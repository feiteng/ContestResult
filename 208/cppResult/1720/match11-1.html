class Solution {
public:
    int minOperations(vector<string>& logs) {
        int cnt = 0;
        for (const auto &s : logs) {
            if (s == "./") {
                continue;
            } else if (s == "../") {
                cnt = max(0, cnt-1);
            } else {
                ++cnt;
            }
        }
        return cnt;
    }
};
