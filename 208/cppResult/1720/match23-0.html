class Solution {
public:
    int minOperations(vector<string>& logs) {
        int cnt = 0;
        for (string s : logs) {
            if (s[0] == '.' && s[1] == '.') {
                cnt = max(0, cnt-1);
            } else if (s[0] != '.') {
                cnt++;
            }
        }
        return cnt;
    }
};
