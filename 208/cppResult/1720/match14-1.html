class Solution {
public:
    int minOperations(vector<string>& logs) {
        int  cur_val = 0;
        for(auto s:logs)
        {
            if(s == "../")
                cur_val--;
            else if(s == "./");
            else
                cur_val++;
            cur_val = max(0,cur_val);
        }
        
        return cur_val;
        
    }
};
