class Solution {
public:
    int minOperations(vector<string>& logs) {
        int res = 0;
        for (auto& log : logs) {
            if (log == "../") {
                res = max(res - 1, 0);
            } else if (log == "./") {
                ;
            } else {
                ++res;
            }
        }
        return res;
    }
};
