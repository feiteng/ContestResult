class Solution {
public:
    int minOperations(vector<string>& logs) {
        int ans = 0;
        for (string& s : logs) {
            if (s.size() >= 3 && s[0] == '.' && s[1] == '.' && s[2] == '/') {
                if (ans != 0)
                    ans--;
            } else if (s.size() >= 2 && s[0] == '.' && s[1] == '/') {
                // no-op
            } else {
                ans++;
            }
        }
        return ans;
    }
};
