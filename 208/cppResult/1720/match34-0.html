class Solution {
public:
    int minOperations(vector<string>& logs) {
        int cur = 0;
        for(auto s : logs) {
            if(s == "../") {
                cur = max(0, cur - 1);
            } else if(s == "./") {
                //
            } else {
                cur++;
            }
        }
        
        return cur;
    }
};
