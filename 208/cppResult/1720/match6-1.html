class Solution {
public:
    int minOperations(vector<string>& logs) {
        int d=0;
        for(string str: logs){
            if(str[0]=='.' && str[1]=='.'){
                if(d>0) d-=1;
            }
            else if(str[0]!='.') d++;
        }
        return d;
    }
};
