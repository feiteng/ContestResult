class Solution {
public:
    int minOperations(vector<string>& logs) {
        int cur = 0;
        for (auto it : logs) {
            if (it == "./") continue;
            else if (it == "../") cur = max(0, cur-1);
            else cur++;
        }
        return cur;
    }
};
