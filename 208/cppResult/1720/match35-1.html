class Solution {
public:
    int minOperations(vector<string>& logs) {
        int depth = 0;
        for (auto &str: logs) {
            if (str.substr(0, 3) == "../") {

                if (depth > 0) --depth;
            } else if (str.substr(0, 2) == "./") {
                ;
            } else {
                ++depth;
            }
        }
        return depth;
    }
};
