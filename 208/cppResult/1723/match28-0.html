class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int res = 0;
        int m = requests.size();
        vector<int> dp(1 << m, 0);
        for(int i = 0; i < (1 << m); i++) {
            vector<int> deg(n, 0);
            int count = 0;
            for(int j = 0; j < m; j++) {
                if(i & (1 << j)) {
                    deg[requests[j][0]]--;
                    deg[requests[j][1]]++;
                    count++;
                }
            }
            bool flag = true;
            for(int j = 0; j < deg.size(); j++) {
                if(deg[j]) {
                    flag = false;
                    break;
                }
            }
            if(flag) res = max(res, count);
        }
        return res;
    }
};
