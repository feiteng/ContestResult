class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int p=requests.size();
        int res=0;
        for (int i=0; i<(1<<p); ++i) {
            vector<int> cnt(n, 0);
            int tmp=0;
            for (int j=0; j<p; ++j) {
                int mask=1<<j;

                if ((mask&i)>0) {
                    cnt[requests[j][0]]--;
                    cnt[requests[j][1]]++;
                    tmp++;
                }
            }
            bool ok=true;
            for (int j=0; j<n; ++j) {
                if (cnt[j]!=0) {
                    ok=false;
                    break;
                }
            }
            if (ok) {
                res=max(res, tmp);
            }
        }
        return res;
    }
};
