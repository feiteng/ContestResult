class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int len = (int)requests.size(), ans = 0;
    for(int i = 1; i < (1 << len); i++)
    {
        vector<int> cnt(n, 0);
        int cur = 0;
        for(int j = 0; j < len; j++) if((i >> j) & 1)
        {
            cur++;
            cnt[requests[j][0]]++;
            cnt[requests[j][1]]--;
        }
        bool ok = true;
        for(int j = 0; j < n && ok; j++)
            if(cnt[j]) ok = false;
        if(ok) ans = max(ans, cur);
    }
    return ans;
    }
};
