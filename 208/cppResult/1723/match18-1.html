class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& a) {
        int m = a.size();
        int res = 0;
        for (int i = 0; i < (1<<m); i++) {
            vector<int> cnt(n+1);
            int bit = 0;
            for (int j = 0; j < m; j++) {
                if (i&(1<<j)) {
                    bit++;
                    cnt[a[j][0]]--;
                    cnt[a[j][1]]++;
                }
            }
            bool flag = true;
            for (int i = 0; i < n; i++) {
                if (cnt[i]) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                res = max(res, bit);
            }
        }
        return res;
    }
};
