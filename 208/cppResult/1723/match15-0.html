class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        
        int ans = 0;
        int len = requests.size();
        
        for(int bitmask = 0; bitmask < (1 << len); bitmask++){
            
            vector<int> cnt(n, 0);
            
            int request = 0;
            for(int i = 0; i < len; i++){
                if((bitmask & (1<<i)) != 0){
                    request++;
                    cnt[requests[i][0]]--;
                    cnt[requests[i][1]]++;
                }
            }
            
            bool canHappen = true;
            for(int i = 0; i < n; i++){
                if(cnt[i] != 0){
                    canHappen = false;
                    break;
                }
            }
            
            if(canHappen){
                ans = max(ans, request);
            }
            
        }
        
        
        
        return ans;
        
    }
};
