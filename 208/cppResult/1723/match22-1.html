class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans=0;
        int sz=requests.size();
        for(int i=0;i<pow(2,sz);i++){
            vector<int> h(n);
            for(int j=0;j<sz;j++){
                if((i&(1ll<<j))){
                    h[requests[j][0]]--;
                    h[requests[j][1]]++;
                }
            }
            

            int f=0;
            for(int j=0;j<n;j++){
                if(h[j]!=0){
                    f=1;
                    break;
                }
            }
            if(f==0){
                ans=max(ans,__builtin_popcount(i));
            }
        }
        return ans;
    }
};
