class Solution {
public:
    bool check(int k, const vector<vector<int>> &r) {
        vector<int> deg(22);
        for (int i = 0; i < 16; i++) {
            if ((k >> i) & 1) {
                deg[r[i][0]]--;
                deg[r[i][1]]++;
            }
        }
        bool flag = true;
        for (auto x : deg) {
            if (x) {
                flag = false;
                break;
            }
        }
        return flag;
    }
    
    int lowbit(int x) {
        return x & -x;
    }
    
    int ones(int x) {
        int cnt = 0;
        while (x) {
            cnt++;
            x -= lowbit(x);
        }
        return cnt;
    }
    
    int maximumRequests(int n, vector<vector<int>>& r) {
        int res = 0;
        int x = r.size();
        int mask = (1 << x) - 1;
        for (int k = 1; k <= mask; k++) {
            if (check(k, r)) {
                res = max(res, ones(k));
            }
        }
        return res;
    }
};
