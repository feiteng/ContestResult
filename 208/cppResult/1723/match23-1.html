class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        int ans = 0;
        for (int r = 0; r < (1 << m); r++){
            vector<int> change(n, 0);
            int changeNum = 0;
            for (int i = 0; i < m; i++){

                if (r &(1<<i)){
                    changeNum ++;
                    change[requests[i][0]] --;
                    change[requests[i][1]] ++;
                }
            }
            bool flag = true;
            for (int i = 0; i < n; i++){
                if (change[i] != 0){
                    flag = false;
                    break;
                }
            }
            if (flag){
                ans = max(ans, changeNum);
            }
        }
        return ans;

    }
};
