class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        
        int ans = 0;
        
        int k = requests.size();
        for(int i = 0;i<(1<<k);i++) {
            
            int num = i;
            
            vector<int> nums(n, 0);
            int cc = 0;
            // cout << (1<<k) << " "; 
            for(int j = 0;j<requests.size();j++) {
                if(i&(1<<j)) {
                    nums[requests[j][0]] --;
                    nums[requests[j][1]] ++;
                    cc++;
                }
            }
            
            bool ok = true;
            for(int j = 0;j<n;j++) {
                if(nums[j]) ok = false;
            }
            if(ok) ans = max(ans, cc);
        }
        return ans;
    }
};
