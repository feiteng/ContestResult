class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int siz=requests.size();
        int ans=0;
        vector<int> deg(n);
        for(int i=1;i<(1<<siz);i++)
        {
              int cnt=0;

              for(int i=0;i<n;i++)deg[i]=0;
              for(int j=0;j<siz;j++)
              {
                  if((1<<j)&i)
                  {
                      cnt++;
                      deg[requests[j][0]]--;
                      deg[requests[j][1]]++;
                  }
              }
              bool g=true;
              for(int i=0;i<n;i++)
              {
                  if(deg[i]!=0)
                  {
                      g=false;
                      break;
                  }
              }
              if(g)
                  ans=max(ans,cnt);
        }
        return ans;
    }
};
