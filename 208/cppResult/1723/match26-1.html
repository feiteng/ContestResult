class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        
        int Ans = 0;
        for (int i = 0; i < 1<< m; ++i) {
            int cnt[n];
            memset(cnt, 0, sizeof(cnt));

            int tot = 0;
            for (int j = 0; j < m; ++j)
                if (i & (1 << j)) {
                    tot += 1;
                    cnt[requests[j][0]] += 1;
                    cnt[requests[j][1]] -= 1;
                }

            int flag = 1;
            for (int j = 0; j < n; ++j)
                if (cnt[j] != 0)
                    flag = 0;
            if (flag == 1)
                Ans = max(Ans, tot);
        }
        return Ans;
    }
};
