class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        int ans = INT_MIN;
        
        for (int i = 0; i < (1 << m); i++) {
            vector<int> total(n, 0);
            int req = 0;
            for (int j = 0; j < m; j++) {
                if (i & (1 << j)) {
                    total[requests[j][0]]--;;
                    total[requests[j][1]]++;
                    req++;
                }
            }
            bool flag = 1;
            for (int j = 0; j < n; j++) {
                if (total[j] != 0) {
                    flag = false; break;
                }
            }
            if (flag) {
                ans = max(ans, req);
            }
        }
        
        return ans;
    }
};
