class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int L = requests.size();
        int res = 0;
        for(int i = 1; i < (1 << L); ++i) {
            vector<int> cap(n, 0);
            int req = 0;
            for(int k = 0; k < L; ++k) {
                if((1 << k) & i) {
                    // request k is fullfilled
                    req++;
                    cap[requests[k][0]]--;
                    cap[requests[k][1]]++;
                }
            }
            
            bool isOK = true;
            for(auto v : cap) {

                if(v != 0) {
                    isOK = false;
                    break;
                } 
            }
            if(isOK) {
                res = max(res, req);
            }
        }
        
        return res;
    }
};
