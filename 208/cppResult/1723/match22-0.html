class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& rq) {
        int r = rq.size();
        int ans = 0;
        for(int state = 0; state < (1<<r); state++){
            vector<int> flow(n,0);
            for(int i = 0; i < r; i++){
                if ((state>>i)&1){
                    flow[rq[i][0]]--;
                    flow[rq[i][1]]++;
                }
            }
            bool ok = true;
            for(int i = 0; i < n; i++){
                if (flow[i] != 0){
                    ok = false;
                    break;
                }
            }
            if (ok){
                ans = max(ans,__builtin_popcount(state));
            }
        }
        
        return ans;
    }
};
