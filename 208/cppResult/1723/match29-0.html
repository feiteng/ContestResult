class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        int max_req=0;
        for (int i=0; i<(1<<m); i++) {
            int freq[22]={0}, c=0;
            for (int j=0; j<m; j++) {
                if (i&(1<<j)) {
                    c++;
                    freq[requests[j][0]]--;
                    freq[requests[j][1]]++;
                }
            }
            bool valid=true;
            for (int i=0;i<n;i++) {
                if (freq[i] != 0) {
                    valid=false;
                    break;
                }
            }
            
            if (valid) max_req=max(max_req, c);
        }
        
        return max_req;
    }
};
