class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        int result = 0;
        for (int st = 0; st < (1 << m); st++) {
            vector<int> deg(n, 0);
            int cur_set_size = 0;
            for (int j = 0; j < m; j++) {
                if (st >> j & 1) {
                    deg[requests[j][0]]--;
                    deg[requests[j][1]]++;
                    cur_set_size++;
                }
            }
            bool all_zero = true;
            for (int i = 0; i < n; i++)  {
                if (deg[i] != 0) {
                    all_zero = false;
                    break;
                }
            }
            if (all_zero) {
                result = max(result, cur_set_size);
            }
        }
        return result;
    }
};
