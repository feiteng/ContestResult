class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& a) {
        int m = a.size();
        int answer = 0;
        for (int i = 0; i < (1 << m); i++) {
            vector<int> c(n);

            int num = 0;
            for (int j = 0; j < m; j++) {
                if (i & (1 << j)) {
                    c[a[j][0]]++;
                    c[a[j][1]]--;
                    num++;
                }
            }
            bool ok = true;
            for (int i = 0; i < n; i++) {
                if (c[i] != 0) {
                    ok = false;
                    break;
                }
            }
            if (ok) answer = max(answer, num);
        }
        return answer;
    }
};
