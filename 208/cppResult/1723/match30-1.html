class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
        int l = requests.size();
        for(int i=1; i<(1<<l); i++) {
            vector<int> in(n, 0);

            for(int j=0; j<l; j++) {
                if(i & (1<<j)) {
                    in[requests[j][1]]++;
                    in[requests[j][0]]--;
                }
            }
            bool f = true;
            for(int j=0; j<n; j++) {
                if(in[j]) {
                    f = false;
                    break;
                }
            }
            if(f) ans = max(ans, __builtin_popcount(i));
        }
        return ans;
    }
};
