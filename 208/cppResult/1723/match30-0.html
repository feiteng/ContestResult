class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& a) {
        int K = (1<<a.size());
        int ret = 0;
        for(int k = 1; k < K; k++) {
            vector<int> cnt(n);
            int tmp = __builtin_popcount(k);
            if (tmp < ret) continue;
            for(int i = 0; i < a.size(); i++) {
                if (k & (1<<i)) {
                    cnt[a[i][0]]--;
                    cnt[a[i][1]]++;
                }
            }
            bool flag = true;
            for(int i = 0; i < n; i++) {
                if (cnt[i] != 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) ret = tmp;
        }
        return ret;
    }
};
