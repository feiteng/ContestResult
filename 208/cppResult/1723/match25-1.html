class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
		int m = requests.size();
		
		for(int i = 0; i < (1<<m); i++){
			int con[22] = {};
			

			int c = 0;
			
			for(int j = 0; j < m; j++){
				if((i>>j)&1) {
					c++;
					con[requests[j][0]]--;
					con[requests[j][1]]++;
				}
			}
			
			bool flag = true;
			for(int j = 0; j < n; j++){
				if(con[j] != 0) flag = false;
			}
			
			if(flag) ans = max(ans, c);
		}
		return ans;
    }
};
