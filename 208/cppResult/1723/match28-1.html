class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
        int m = requests.size();
        int full = (1<<m) -1;

        for(int i = 1; i <= full; i++) {
            vector<int> net(n); // in - out
            int cnt = 0;
            for(int j = 0; j < m; j++) {
                if(i & (1<<j)) {
                    cnt++;

                    auto& r = requests[j];
                    net[r[0]]--;
                    net[r[1]]++;
                }
            }
            bool ok = true;
            for(int j = 0; j < n; j++) {
                if(net[j] != 0) {
                    ok = false;
                    break;
                }
            }
            if(ok) ans = max(ans, cnt);
        }
        return ans;
    }
};
