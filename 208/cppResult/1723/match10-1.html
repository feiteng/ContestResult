class Solution {
public:
    
    int maximumRequests(int n, vector<vector<int>>& requests) {
        
        int mx = requests.size();
        int ans = 0;
        for(int i = 0; i<(1 << mx); i++)
        {
            vector<int> build(n, 0);
            int cnt = 0;
            for(int j = 0; j<requests.size(); j++)
            {

                if (i & (1 << j))
                {
                    cnt++;
                    build[requests[j][0]]--;
                    build[requests[j][1]]++;
                    
                }
            }
            bool poss = true;
            for(int i = 0; i<n; i++)
            {
                if (build[i] != 0)
                {
                    poss = false;
                    break;
                }
            }
            
            if (poss)
            {
                ans = max(ans, cnt);
            }
        }
        return ans;
    }
};
