class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
        vector<int> cnt(n, 0);
        int sz = requests.size();
        for (int mask = 1; mask < (1 << sz); ++mask) {
            fill(cnt.begin(), cnt.end(), 0);

            int d = 0;
            for (int i = 0; i < sz; ++i) {
                if (mask & (1 << i)) {
                    cnt[requests[i][0]]--;
                    cnt[requests[i][1]]++;
                    ++d;
                }
            }

            bool flag = true;
            for (int i = 0; i < n; ++i) {
                if (cnt[i] != 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                ans = max(d, ans);
            }
        }
        return ans;
    }
};
