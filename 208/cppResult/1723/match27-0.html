class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int m = requests.size();
        int in[n], out[n];
        int ans = 0;
        for (int mask = 0; mask < (1 << m); ++ mask) {
            
            memset(in, 0, sizeof in);
            memset(out, 0, sizeof out);
            int cnt = 0;
            for (int i = 0; i < m; ++i) {
                if ((mask >> i) & 1) {
                    cnt ++;
                    in[requests[i][1]]++;
                    out[requests[i][0]]++;
                }
            }
            int flag = 1;
            for (int i = 0; i < n; ++i) {
                if (in[i] != out[i]) {
                    flag = 0;
                    break;
                }
            }
            if (flag) {
                ans = max(cnt, ans);
            }
        }
        return ans;
    }
};
