class Solution {
public:
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
        int m = requests.size();
        for(int mask = 0; mask < (1 << m); ++mask) {
            vector<int> deg(n, 0);
            for(int i = 0; i < m; ++i) {
                if(mask >> i & 1) {
                    auto& r = requests[i];
                    int u = r[0], v = r[1];

                    deg[u] += 1;
                    deg[v] -= 1;
                }
            }
            bool ok = true;
            for(int i = 0; i < n; ++i) {
                if(deg[i] != 0) {
                    ok = false;
                    break;
                }
            }
            if(ok) {
                ans = max(ans, __builtin_popcount(mask));
            }
        }
        return ans;
    }
};
