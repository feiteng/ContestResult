class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int mxv = 0;
        int mxi = -1;
        int noww = 0;
        int rs = 0;
        int cnt = 0;
        int sz = customers.size();
        while(noww>0||cnt<sz)
        {
            if(cnt<sz)
                noww += customers[cnt];
            rs += min(4, noww)*boardingCost - runningCost;
            noww -= min(4, noww);
            cnt++;
            //cout << rs << " " << noww << endl;
            if(rs>mxv)
            {
                mxv = rs;
                mxi = cnt;
            }
        }
        return mxi;
    }
};
