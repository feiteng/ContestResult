class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int idx = 0, ans = -1, mx = 0, tot = 0, profit = 0;
        for(int c : customers)
        {
            ++idx;
            tot += c;
            profit -= runningCost;
            int up = min(4, tot);
            tot -= up;
            profit += up * boardingCost;
            if(profit > mx)
            {
                ans = idx;
                mx = profit;
            }
        }
        while(tot)
        {
            ++idx;
            profit -= runningCost;
            int up = min(4, tot);
            tot -= up;
            profit += up * boardingCost;
            if(profit > mx)
            {
                ans = idx;
                mx = profit;
            }
        
        }
        return ans;
    }
};
