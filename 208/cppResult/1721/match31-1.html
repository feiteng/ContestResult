class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int cur = 0;
        int p = 0;
        int max_p = 0;
        int r = 0;
        int min_r = -1;
        for (int i = 0; i < customers.size(); ++i) {
            cur += customers[i];
            int board = min(cur, 4);

            cur -= board;
            p += boardingCost * board;
            p -= runningCost;
            ++r;
            if (p > max_p) {
                max_p = p;
                min_r = r;
            }
        }
        while (cur > 0) {
            int board = min(cur, 4);
            cur -= board;

            p += boardingCost * board;
            p -= runningCost;
            ++r;
            if (p > max_p) {
                max_p = p;
                min_r = r;
            }
        }
        return min_r;
    }
};
