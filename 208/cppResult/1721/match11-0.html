class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int ptr = 0;
        int profit = 0;
        int onhold = 0;
        int maxprofit = 0;
        int rot = -1;
        int idx = 0;
        for ( int c : customers ) {
            ++ idx;
            onhold += c;
            int curr = min(4, onhold);
            onhold -= curr;
            profit += curr * boardingCost;
            profit -= runningCost;
            if ( profit > maxprofit ) {
                maxprofit = profit;
                rot = idx;
            }
        }
        while ( onhold ) {
            int curr = min(4, onhold);
            ++ idx;
            onhold -= curr;
            profit += curr * boardingCost;
            profit -= runningCost;
            if ( profit > maxprofit ) {
                maxprofit = profit;
                rot = idx;
            }
        }
        return rot;
        
    }
};
