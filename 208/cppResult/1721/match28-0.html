class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int pp = 0;
        int maxcost = 0;
        int cost = 0;
        int res = -1;
        int rolls = 0;
        for(int i = 0; i < customers.size(); i++) {
            pp += customers[i];
            int take = min(pp, 4);
            pp -= take;
            rolls++;
            int c = take * boardingCost - runningCost;
            cost += c;
            if(cost > maxcost) {
                maxcost = cost;
                res = rolls;
            }
   
        }
        while(pp > 0) {
            int take = min(pp, 4);
            pp -= take;
            int c = take * boardingCost - runningCost;
            cost += c;
            rolls++;
            if(cost > maxcost) {
                maxcost = cost;
                res = rolls;
            }
        }
        return res;
    }
};
