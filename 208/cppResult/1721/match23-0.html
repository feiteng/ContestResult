class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int maxProfit=0;
        int tmpProfit=0;
        int people=0;
        int round=0;
        int result=-1;
        int cur=0;
        while(cur<customers.size()||people>0){
            if(cur<customers.size()){
                people+=customers[cur];
            }
            int curPeople=min(people,4);
            people-=curPeople;
            tmpProfit+=curPeople*boardingCost-runningCost;
            if(tmpProfit>maxProfit){
                maxProfit=tmpProfit;
                result=cur+1;
            }
            cur++;
        }
        return result;
    }
};
