class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int remain = 0;
        int ans=-1;
        int load = 0;
        int turn=0;
        int maxturn=-1;
        for (int n:customers)
        {
            turn ++;

            int total = remain+n;
            if (total>=4)
            {
                remain = total-4;
                load +=4;
            }else
            {
                remain = 0;
                load += total;
            }
            //cout<<"load:"<<load<<" total:"<<total<<" remain:"<<remain<<endl;
            if (ans<load*boardingCost-turn*runningCost)
            {
                maxturn = turn;
                ans = load*boardingCost-turn*runningCost;
            }
            //ans =max(ans,load*boardingCost-turn*runningCost);   
            //cout<<"ans:"<<ans<<endl;
        }
        while (remain!=0)
        {
            turn++;

            int total = remain;
            if (total>=4)
            {
                remain = total-4;
                load +=4;
            }else
            {
                remain = 0;
                load += total;
            }
            if (ans<load*boardingCost-turn*runningCost)
            {
                maxturn = turn;
                ans = load*boardingCost-turn*runningCost;
            }
        }
        return ans<0?-1:maxturn;
    }
};
