class Solution {
  public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) 
    {
      int max_profit = 0;
      int max_rotate = 0;
      int profit = 0;
      int rotate = 0;
      int left = 0;
      for (auto c: customers) {
        left += c;
        int board = min(left, 4);
        profit += board * boardingCost;

        profit -= runningCost;
        left -= board;
        ++rotate;
        if (max_profit < profit) {
          max_profit = profit;
          max_rotate = rotate;
        }
      }
      while (left > 0) {
        int board = min(left, 4);
        profit += board * boardingCost;

        profit -= runningCost;
        left -= board;
        ++rotate;
        if (max_profit < profit) {
          max_profit = profit;
          max_rotate = rotate;
        }
      }
      return max_profit > 0 ? max_rotate : -1;
    }
};
