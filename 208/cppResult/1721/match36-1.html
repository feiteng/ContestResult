class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int res = 0;
        int res_profit = 0;
        int circle = 0;
        int profit = 0;
        int cus_count = 0;
        for(int i = 0; (i < customers.size()) || (cus_count > 0); ++i) {
            if(i < customers.size()) cus_count += customers[i];

            circle++;
            if(cus_count >= 4) {
                profit += 4 * boardingCost - runningCost;
                cus_count -= 4;
            }
            else {
                profit += cus_count * boardingCost - runningCost;
                cus_count = 0;
            }
            if(profit > res_profit) {
                res = circle;
                res_profit = profit;
            }
        }
        return res_profit > 0 ? res : -1;
    }
};
