class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int users=0;
        int ansi=-1;
        int ans_cost = 0;
        int costs=0;
        int n=customers.size();

        for(int i=0;i<n;i++){
            users+=customers[i];
            int leave=min(users,4);
            users-=leave;
            costs+=boardingCost*leave-runningCost;
            if(costs>ans_cost){
                ans_cost=costs;
                ansi=i+1;
            }
        }
        int i=n+1;

        while(users>0){
            int leave=min(users,4);
            users-=leave;
            costs+=boardingCost*leave-runningCost;
            if(costs>ans_cost){
                ans_cost=costs;
                ansi=i;
            }
            i++;
        }
        return ansi;
    }
};
