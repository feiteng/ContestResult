class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int profit = 0;
        int best = 0;
        int res = -1;
        int q = 0;
        int counter = 0;
        for(int i = 0; i < customers.size(); i++){
            q += customers[i];
            int c =  q >= 4 ? 4 : q;
            q -= c;
            profit += c * boardingCost - runningCost;
            counter++;
            if(profit > best) {
                best = profit;
                res = counter;
            }
        }
        while(q) {
            int c =  q >= 4 ? 4 : q;
            q -= c;
            profit += c * boardingCost - runningCost;
            counter++;
            if(profit > best) {
                best = profit;
                res = counter;
            }
        }
        
        return res;
    }
};
