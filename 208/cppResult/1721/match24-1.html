class Solution {
public:
    // int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
    int minOperationsMaxProfit(vector<int>& C, int bC, int rC) {
        int ans=-1, best_n=-1, n=0, x=0, c=0;
        for(auto cs:C){

            ++n;
            c+=cs;
            int u=min(c, 4);
            c-=u;
            x+=bC*u-rC;
            if(ans<x)
            {
                best_n=n;
                ans=x;
            }
        }
        while(c)
        {
            ++n;

            int u=min(c, 4);
            c-=u;
            x+=bC*u-rC;
            if(ans<x)
            {
                best_n=n;
                ans=x;
            }
        }
        return best_n;
    }
};
