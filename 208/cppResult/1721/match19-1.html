class Solution {
public:
    int minOperationsMaxProfit(vector<int>& arr, int bc, int rc) {
        int ans = -1;
        int maxProfit = 0;
        int n = arr.size();
        int curr = 0;
        int currProfit = 0;
        int i = 0;

        for(; i < n; i++) {
            curr += arr[i];
            int board = min(curr, 4);
            curr -= board;
            currProfit += ((board * bc) - rc);
            if(currProfit > maxProfit) {
                maxProfit = currProfit;
                ans = i + 1;
            }
        }
        
        while(curr > 0) {
            int board = min(curr, 4);
            curr -= board;
            currProfit += ((board * bc) - rc);
            if(currProfit > maxProfit) {
                maxProfit = currProfit;
                ans = i + 1;
            }
            i++;
        }
        
        return ans;
    }
};
