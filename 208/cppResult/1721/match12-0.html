class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int maxp = -1, remain = 0, curp = 0, turn = 0, best = 0;
        for (auto num : customers) {
            remain += num;
            int move = min(remain, 4);
            remain -= move;
            curp += move * boardingCost - runningCost;
            ++turn;
            if (curp > maxp) {
                best = turn;
                maxp = curp;
            }
        }
        while (remain > 0) {
            int move = min(remain, 4);
            remain -= move;
            curp += move * boardingCost - runningCost;
            ++turn;
            if (curp > maxp) {
                best = turn;
                maxp = curp;
            }
            
        }
        return maxp <= 0 ? -1 : best;
    }
};
