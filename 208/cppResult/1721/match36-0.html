class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int wait_num = 0;
        int profit = 0;
        int res = 0;
        int maxval = 0;
        int k = 0;
        for(int i = 0; i < customers.size(); i++){
            k++;
            wait_num += customers[i];
            if(wait_num < 4){
                profit += wait_num * boardingCost - runningCost;
                wait_num = 0;
            }else{
                profit += 4*boardingCost - runningCost;
                wait_num -= 4;
            }
            if(maxval < profit){
                maxval = profit;
                res = k;
            }
        }
        while(wait_num){
            k++;
            if(wait_num < 4){
                profit += wait_num * boardingCost - runningCost;
                wait_num = 0;
            }else{
                profit += 4*boardingCost - runningCost;
                wait_num -= 4;
            }
            
            
            if(maxval < profit){
                maxval = profit;
                res = k;
            }
            
            
        }
        return maxval > 0?res:-1;   
    }
};
