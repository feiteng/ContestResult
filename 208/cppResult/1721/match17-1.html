class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int x, int y) {
        int a = -1,mx = 0,p = 0,w = 0,r = 0;
        for(int v:customers){

            ++r;
            p += v;
            w += min(p,4) * x - y;
            p -= min(p,4);
            if(w > mx) mx = w,a = r;
        }
        while(p > 0){
            ++r;
            w += min(p,4) * x - y;
            p -= 4;
            if(w > mx) mx = w,a = r;
        }
        return a;
    }
};
