class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int left = 0;
        int profit = 0;
        int rotations = 0;
        int maxProfit = 0;
        int bestRotation = 0;
        
        for (int c: customers) {

            left += c;
            int cur = min(4, left);
            left -= cur;
            profit += cur * boardingCost - runningCost;
            ++rotations;
            if (profit > maxProfit) {
                maxProfit = profit;
                bestRotation = rotations;
            }
        }
        
        while (left) {

            int cur = min(4, left);
            left -= cur;
            profit += cur * boardingCost - runningCost;
            ++rotations;
            if (profit > maxProfit) {
                maxProfit = profit;
                bestRotation = rotations;
            }
        }
        
        return profit >= 0 ? bestRotation : -1;
    }
};
