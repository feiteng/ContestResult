class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int wait = 0;
        int earn = INT_MIN;
        int ans = -1;
        int sum = 0;
        int index = 0;
        for(auto p : customers){
            wait += p;
            int cur = (wait > 4) ? 4 : wait;
            wait -= cur;
            index++;
            sum += boardingCost * cur - runningCost;
            if(sum > earn){
                earn = sum;
                ans = index;
            }
        }
        while(wait > 0){
            int cur = (wait > 4) ? 4 : wait;
            wait -= cur;
            index++;
            sum += boardingCost * cur - runningCost;
            if(sum > earn){
                ans = index;
                earn = sum;
            }
        }
        return earn > 0 ? ans : -1;
    }
};
