class Solution {
public:
    int minOperationsMaxProfit(vector<int>& xs, int prof, int cost) {
        int rot = 0;
        int gain = 0;
        int ans = 0;
        int ans_rot = -1;
        int rest = 0;
        for(int x : xs) {
          rest += x;

          gain -= cost;

          int customer = min(4, rest);
          rest -= customer;
          gain += customer * prof;
          ++rot;
          DEBUG(ans,gain);
          if (ans < gain) {
            ans = gain;
            ans_rot = rot;
          }
        }

        while(rest > 0) {
          gain -= cost;


          int customer = min(4, rest);
          rest -= customer;
          gain += customer * prof;
          ++rot;
          
          if (ans < gain) {
            ans = gain;
            ans_rot = rot;
          }
        }

        return ans_rot;
    }
};
