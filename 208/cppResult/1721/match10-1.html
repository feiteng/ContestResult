class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int waitingCustomers = 0;

        int profit = 0;
        int ans = 0;
        
        int maxTimes = -1;
        
        int times = 0;
        
        int customersBoarded = 0;
        
        for(auto x:customers){

            waitingCustomers += x;
            
            times++;
            
            int y = min(waitingCustomers,4);
            
            customersBoarded += y;
            profit = customersBoarded*boardingCost - times*runningCost;
            
            waitingCustomers -= y;
            
            if(ans < profit){
                maxTimes = times;
                ans = profit;
            }
        }
        
        while(waitingCustomers){
            int x = min(waitingCustomers,4);
            
            times++;
            customersBoarded += x;
            
            profit = customersBoarded*boardingCost - times*runningCost;
            
            waitingCustomers -= x;
            
            if(ans < profit){
                maxTimes = times;
                ans = profit;
            }
        }
        
        return maxTimes;
    }
};
