class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int inc = 0, maxc = -1, wait = 0, on = 0, res = -1, ind = 0;
        for(int i: customers) {
            ind++;

            wait += i;
            if(wait >= 4) {
                wait -= 4;
                on = 4;
            }
            else {
                on = wait;
                wait = 0;
            }
            inc += on * boardingCost - runningCost;
            if(inc > maxc){
                res = ind;
                maxc = inc;
            } 
        }
        while(wait > 0){
            ind++;

            if(wait >= 4) {
                wait -= 4;
                on = 4;
            }
            else {
                on = wait;
                wait = 0;
            }
            inc += on * boardingCost - runningCost;
            if(inc > maxc){
                res = ind;
                maxc = inc;
            } 
        }
        return res;
    }
};
