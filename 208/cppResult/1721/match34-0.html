class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int n = customers.size();
        
        int res = INT_MIN;
        int res_pos = -1;
        
        int cur = 0;
        int remain = 0;
        for(int i = 0; i < n; ++i) {
            remain += customers[i];
            
            int count = min(remain, 4);
            remain -= count;
            cur += count * boardingCost;
            cur -= runningCost;
            if(res < cur) {
                res = cur;
                res_pos = i + 1;
            }
        }
        int a = 0;
        while(remain > 0) {
            a += 1;
            int count = min(remain, 4);
            remain -= count;
            cur += count * boardingCost;
            cur -= runningCost;
            if(res < cur) {
                res = cur;
                res_pos = n + a;
            }
        }
        return res <= 0 ? -1 : res_pos;
    }
};
