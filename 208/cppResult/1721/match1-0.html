class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int res = -1, mx = 0, k = 0, n = 0, cur = 0;
        for (int c : customers) {
            n += c;
            int b = min(n, 4);
            cur += b * boardingCost - runningCost;
            n -= b;
            k++;
            if (cur > mx) {
                mx = cur;
                res = k;
            }
        }
        while (n > 0) {
            int b = min(n, 4);
            cur += b * boardingCost - runningCost;
            n -= b;
            k++;
            if (cur > mx) {
                mx = cur;
                res = k;
            }
        }
        return res;
    }
};
