class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int n=customers.size();
        
        int total_remaining=0;
        
        int ans=-2;
        int max_profit=-1;
        int profit=0;
        
        int i;
        for(i=0;i<n;i++)
        {
            total_remaining+=customers[i];
            int current=min(4,total_remaining);
            total_remaining-=current;
            profit+=boardingCost*current-runningCost;
            if(max_profit<profit)   {max_profit=profit;ans=i;}
        }
        
        while(total_remaining)
        {
            int current=min(4,total_remaining);
            total_remaining-=current;
            profit+=boardingCost*current-runningCost;
            if(max_profit<profit)   {max_profit=profit;ans=i;}
            i++;
        }
        
        return ans+1;
    }
};
