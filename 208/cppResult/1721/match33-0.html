class Solution {
public:
    int minOperationsMaxProfit(vector<int>& a, int b, int c) {
        int ans = -1;
        int used = 0;
        int wait = 0;
        int temp = -1;
        int res = -1;
        for(int i=0; i<a.size(); i++){
            used += min(4, a[i]+wait);
            wait += a[i]- min(4, a[i]+wait);
            temp = ans;
            ans = max(ans, used*b-c*(i+1));
            if(ans>temp){
                res = i+1;
                temp = ans;
            }
        }
        int k = a.size();
        while(wait>0){
            used += min(4, wait);
            wait -= min(4, wait);
            temp = ans;
            ans = max(ans, used*b-c*(++k));
            if(ans>temp){
                res = k;
                temp = ans;
            }
        }
        return res;
    }
};
