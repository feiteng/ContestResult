class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int numw=0;
        int mx=-1;
        int curprof=0;
        int num=0;
        int ans=-1;
        for(int i=0; i<customers.size(); i++){
            numw+=customers[i];
            num++;
            curprof+=boardingCost*min(numw, 4)-runningCost;
            numw-=min(numw, 4);
            if(curprof>mx){
                ans=num; mx=curprof;
            }
        }
        while(numw>0){
            num++;
            curprof+=boardingCost*min(numw, 4)-runningCost;
            numw-=min(numw, 4);
            if(curprof>mx){
                ans=num; mx=curprof;
            }
        }
        return ans;
    }
};
