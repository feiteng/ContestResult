class Solution {
public:
    int minOperationsMaxProfit(vector<int>& a, int b, int c) {
        int tot=0,ans=-1,op=0,t=0,opt=-1;
        for(int i=0;i<a.size();i++){
            t++;tot+=a[i];op+=min(4,tot);tot-=min(tot,4);
            if(op*b-t*c>ans) opt=t;
            ans=max(ans,op*b-t*c);
        }
        while(tot){
            t++;op+=min(4,tot);tot-=min(tot,4);
            if(op*b-t*c>ans) opt=t;
            ans=max(ans,op*b-t*c);
        }
        return opt;
    }
};
