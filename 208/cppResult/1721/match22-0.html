class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int ans = -1;
        int tot = 0;
        int cur = 0;
        int cnt = -1;
        int n = customers.size();
        for(int i = 0; i < customers.size(); ++i){
            tot += customers[i];
            if(tot >= 4) {
                tot -= 4;
                cur += 4;
            }
            else{
               
                cur += tot; tot = 0;
            }
            int s = cur*boardingCost-(i+1)*runningCost;
            if(ans < s){
                ans = s;
                cnt = i+1;
            }
        }
        int t = n;
        while(tot>0){
            if(tot >= 4) {
                tot -= 4;
                cur += 4;
            }
            else {
                
                cur += tot;tot = 0;
            }
            int s = cur*boardingCost-(t+1)*runningCost;
            if(ans < s){
                ans = s;
                cnt = t+1;
            }
            t++;
        }
        return cnt;
    }
};
