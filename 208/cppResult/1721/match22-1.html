class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int opt = 0;
        int time = -1;
        int people = 0;
        int base = 0;
        
        int n = customers.size();
        int i;

        for (i = 1 ; i <= n ; ++i) {
            people += customers[i - 1];
            
            if (people < 4) {
                base += people;
                people = 0;
            } else {
                base += 4;
                people -= 4;
            }
            int sum = base * boardingCost - i * runningCost;
            //printf("%d %d\n", people, sum);
            if (sum > opt) {
                opt = sum;
                time = i;
            }
        }
        

        while (people > 0) {
            if (people < 4) {
                base += people;
                people = 0;
            } else {
                base += 4;
                people -= 4;
            }
            int sum = base * boardingCost - i * runningCost;
            //printf("%d %d\n", people, sum);
            if (sum > opt) {
                opt = sum;
                time = i;
            }
            ++i;
        }
        
        return time;
    }
};
