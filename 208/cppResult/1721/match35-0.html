class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int w = 0;
        int p = 0;
        int p_max = 0;
        int r = 0;
        int br = -1;
        
        for(auto &u : customers)
        {
            w += u;
            int d = min(w,4);
            w -= d;
            p += (d*boardingCost - runningCost);
            r++;
            if(p > p_max)
            {
                br = r;
                p_max = p;
            }
        }
        while(w)
        {
            int d = min(w,4);
            w -= d;
            p += (d*boardingCost - runningCost);
            r++;
            if(p > p_max)
            {
                br = r;
                p_max = p;
            }            
        }
        
        return br;
    }
};
