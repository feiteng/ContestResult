class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int ans = 0, ansi = -1, sum = 0, i = 0, m = 0;
        for(int x : customers){

            m += x;
            int h = min(m, 4);
            m -= h;
            i += 1;
            sum += h * boardingCost - runningCost;
            if(sum > ans){
                sum = ans;
                ansi = i;
            }
        }
        while(m > 0){
            int h = min(m, 4);
            m -= h;
            i += 1;
            sum += h * boardingCost - runningCost;
            if(sum > ans){
                sum = ans;
                ansi = i;
            }
        }
        return ansi;
    }
};
