class Solution {
public:
    int minOperationsMaxProfit(vector<int>& v, int toAdd, int toSub) {
        int ans=0;
        int tot=0;
        int wait=0;
        int n=v.size();
        int cur=0;
        int filled=0;
        int ans1=-1,temp=0;
        while(cur<n || wait!=0)
        {
              if(cur<n)
              {
                  wait+=v[cur];
                  cur++;
              }
                  int people=min(wait,4);
                  wait-=people;
                  tot+=toAdd*people-toSub;
                  temp++;
                  if(tot>0 && ans<tot)
                  {
                      ans=tot;
                      ans1=temp;
                  }
        }
        return ans1;
    }
};
