class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int mx=-1;
        int ans=-1;
        int cnt=0;
        int wait=0;
        int temp=0;
        for (int i=0;i<customers.size();i++){

            wait+=customers[i];
            cnt+=min(wait,4);
            wait=max(wait-4,0);
            temp++;
            if(mx<cnt*boardingCost-temp*runningCost){
                mx=cnt*boardingCost-temp*runningCost;
                ans=temp;
            }
        }
        while (wait){
            cnt+=min(wait,4);
            wait=max(wait-4,0);
            temp++;
            if(mx<cnt*boardingCost-temp*runningCost){
                mx=cnt*boardingCost-temp*runningCost;
                ans=temp;
            }
        }
        return ans;
    }
};
