class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost)
    {
        int ans=-1, curr=0, mx_profit=0, idx=0, c=0;
        for (auto cust: customers)
        {
            c += cust;
            if (c > 4)
            {
                curr += (boardingCost*4) - runningCost;
                c-=4;
            }
            else
            {
                curr += (boardingCost*c) - runningCost;
                c = 0;
            }
            idx++;
            if (curr > mx_profit)
            {
                mx_profit = curr;
                ans = idx;
            }
        }
        while (c)
        {
            if (c > 4)
            {
                curr += (boardingCost*4) - runningCost;
                c-=4;
            }
            else
            {
                curr += (boardingCost*c) - runningCost;
                c = 0;
            }
            idx++;
            if (curr > mx_profit)
            {
                mx_profit = curr;
                ans = idx;
            }
        }
        
        return ans;
    }
};
