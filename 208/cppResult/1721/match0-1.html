class Solution {
public:
    int minOperationsMaxProfit(vector<int>& A, int b, int r) {
        int res = -1;
        int wait = 0;
        int n = A.size();
        int i = 0;
        int round = 0;
        int tot = 0;
        while (i < n || wait) {
            int cur = 0;
            if (i < n) {
                wait += A[i];
            }
            tot += min(wait, 4);
            cur = tot * b - (i + 1) * r;
            if (cur > res) {
                res = cur;
                round = i + 1;
            }
            wait = max(0, wait - 4);
            i++;
            
        }
        return res >= 0 ? round : -1;
    }
};
