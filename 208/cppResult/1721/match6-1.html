class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int ans = 0;
        int last = 0;
        int res = -1;
        int t = - 1;
        int cnt = 0;
        for(auto x:customers){

            ++cnt;
            last += x;
            ans += min(4,last) * boardingCost - runningCost;
            last -= min(4,last);
            if(ans > res){
                res = ans;
                t = cnt;
            }
        }
        while(last){
            ++cnt;
            ans += min(4,last) * boardingCost - runningCost;
            last -= min(4,last);
            if(ans > res){
                res = ans;
                t = cnt;
            }
            
        }
        return t;
    }
};
