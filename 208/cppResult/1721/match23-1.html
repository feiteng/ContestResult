class Solution {
public:
    int minOperationsMaxProfit(vector<int>& cs, int fee, int cost) {
        int max_profit = 0;
        int max_profit_idx = -1;
        int cur_profit = 0;
        int people_waiting = 0;
        int n = cs.size();
        int i = 0;

        while (people_waiting > 0 || i < n) {
            if (i < n) people_waiting += cs[i];

            int people_go = min(4, people_waiting);
            people_waiting -= people_go;
            cur_profit += people_go * fee - cost;
            // cout << cur_profit << endl;
            if (cur_profit > max_profit) {
                max_profit = cur_profit;
                max_profit_idx = i + 1;
            }
            i++;
        }

        return max_profit_idx;
    }
};
