class ThroneInheritance {
    unordered_map<string, vector<string>>children;
    unordered_set<string>alive;
    string king;
public:
    ThroneInheritance(string kingName) {
        king = kingName;
        alive.insert(kingName);
    }
    
    void birth(string parentName, string childName) {
        children[parentName].push_back(childName);
        alive.insert(childName);
    }
    
    void death(string name) {
        alive.erase(name);
    }
    
    vector<string> getInheritanceOrder() {
        vector<string>res;
        helper(res, king);
        return res;
    }
    
    void helper(vector<string>& res, string& person) {
        if (alive.find(person)!=alive.end())
            res.push_back(person);
        for (auto chd: children[person])
            helper(res, chd);
    }
};
