class ThroneInheritance {
public:
    unordered_map<string, vector<string>> inherit;
    unordered_set<string> deaths;
    string kName;
    ThroneInheritance(string kingName) {
        kName = kingName;
    }
    
    void birth(string parentName, string childName) {
        inherit[parentName].push_back(childName);
    }
    
    void death(string name) {
        deaths.insert(name);
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> res;
        dfs(kName, res);
        return res;
    }
    
    void dfs(string name, vector<string>& res) {
        if(deaths.find(name) == deaths.end()) {
            res.push_back(name);
        }
        
        for(auto names : inherit[name]) {
            dfs(names, res);
        }
        
        return;
    }
};

 * obj->death(name);
 * vector<string> param_3 = obj->getInheritanceOrder();
