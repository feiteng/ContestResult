class Node{
public:
    Node(string name){
        cur = name;
        s = true;
    }
    string cur;
    vector<Node*> c;
    bool s;
};
class ThroneInheritance {
public:
    ThroneInheritance(string kingName) {
        root = new Node(kingName);
        map[kingName] = root; 
    }
    
    void birth(string parentName, string childName) {
        Node* cur = map[parentName];
        Node* child = new Node(childName);
        cur->c.push_back(child);
        map[childName] = child;
    }
    
    void death(string name) {
        map[name]->s = false;
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> res;
        helper(root, res);
        return res;
    }
    
    void helper(Node* cur, vector<string>& res){
        if(cur == NULL) return;
        if(cur->s == true){
            res.push_back(cur->cur);
        }
        for(auto c:cur->c){
            helper(c, res);
        }
    }
    Node* root;
    unordered_map<string, Node*> map;
};
