class ThroneInheritance {
public:
    unordered_map<string, vector<string>> m;
    string king;
    unordered_set<string> alive;
    
    ThroneInheritance(string kingName) {
        king = kingName;
        alive.insert(kingName);
    }
    
    void birth(string parentName, string childName) {
        m[parentName].push_back(childName);
        alive.insert(childName);
    }
    
    void death(string name) {
        alive.erase(name);
    }
    
    void helper(vector<string>& ans, string& cur){
        if(alive.count(cur))
            ans.push_back(cur);
        
        for(auto& item : m[cur]){
            helper(ans, item);
        }
        
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> ans;
        
        string cur = king;
        helper(ans, cur);
        
        return ans;
    }
};

 * vector<string> param_3 = obj->getInheritanceOrder();
