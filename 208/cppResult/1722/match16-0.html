class Node {
public:
    string name;
    bool live;
    vector<Node *> child;
    
    Node(string _name) {
        name = _name;
        live = true;
    }
};

class ThroneInheritance {
public:
    ThroneInheritance(string kingName) {
        root = new Node(kingName);
        m[kingName] = root;
    }
    
    void birth(string parentName, string childName) {
        Node *p = m[parentName];
        Node *cur = new Node(childName);
        p->child.push_back(cur);
        m[childName] = cur;
    }
    
    void death(string name) {
        m[name]->live = false;
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> res;
        preorder(root, res);
        return res;
    }
    
    void preorder(Node *root, vector<string> &res) {
        if(root == NULL) return;
        if(root->live) {
            res.push_back(root->name);    
        }
        
        for(auto c: root->child) {
            preorder(c, res);
        }
    }
    
private:
    Node *root;
    unordered_map<string, Node*> m;
};
