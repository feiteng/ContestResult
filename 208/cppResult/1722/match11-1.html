class ThroneInheritance {
public:
    unordered_map<string,vector<string>> u;
    unordered_map<string,bool> ch;
    string k;
    vector<string> ans;
    
    void abc(string na)
    {   if(ch[na]==1)
        ans.push_back(na);
        int n=u[na].size();
        for(int i=0;i<n;i++)
        {   
            abc(u[na][i]);
        }
        return ;
    }
    ThroneInheritance(string kingname) {
        u.clear();
        k=kingname;
        u[kingname].clear();
        ch.clear();
        ch[kingname]=1;
    }
    
    void birth(string p, string c) {
        ch[c]=1;
        u[p].push_back(c);
    }
    
    void death(string name) {
        ch[name]=0;
    }
    
    vector<string> getInheritanceOrder() {
        ans.clear();
         abc(k);
        return ans;
    }
};
