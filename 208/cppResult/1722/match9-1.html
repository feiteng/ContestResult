class ThroneInheritance {
public:
    unordered_map<string,vector<string>>f;
    unordered_map<string,int>f1;
    string k;
    
    ThroneInheritance(string kingName) {
        k=kingName;
        f1[k]=0;
    }
    
    void birth(string p, string c) {
        f[p].push_back(c);
        f1[c]=0;
    }
    
    void death(string name) {
        f1[name]=1;
    }
    
    void dfs(string pa,vector<string> &ans){
        if(f1[pa]!=1)ans.push_back(pa);
        for(auto x:f[pa]){
            dfs(x,ans);
        }
        
        
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> ans;
        dfs(k,ans);
        return ans;
    }
};
