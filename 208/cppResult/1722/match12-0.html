class ThroneInheritance {
public:
    
    unordered_map<string, vector<string>> mp; //name -> sons
    unordered_set<string> ded; //deaded man
    string _root;
    
    ThroneInheritance(string kingName) {
        _root = kingName;
    }
    
    void birth(string a, string b) {
        mp[a].push_back(b);
    }
    
    void death(string name) {
        ded.insert(name);
    }
    
    vector<string> path;
    void dfs(string root){
        if(!ded.count(root)) path.push_back(root);
        for(auto& next: mp[root]){
            dfs(next);
        }
    }
    
    vector<string> getInheritanceOrder() {
        path.clear();
        dfs(_root);
        return path;
    }
};

 * obj->death(name);
 * vector<string> param_3 = obj->getInheritanceOrder();
