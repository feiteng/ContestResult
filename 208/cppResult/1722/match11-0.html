#define pb push_back
class ThroneInheritance {
public:
    unordered_map<string,vector<string>> adj;
    unordered_map<string,bool> visited;
    string t;
    vector<string> res;
    
    void dfs(string &s)
    {   
        if(visited[s]==1)
            res.push_back(s);
        int n=adj[s].size();
        for(int i=0;i<n;i++)
        {   
            dfs(adj[s][i]);
        }
        return;
    }
    ThroneInheritance(string kingname) {
        adj.clear();
        visited.clear();
        visited[kingname]=1;
        t=kingname;
        adj[kingname].clear();
        
    }
    
    void birth(string par, string child) {
        adj[par].push_back(child);
        visited[child]=1;
        
    }
    
    void death(string d) {
        visited[d]=0;
    }
    
    vector<string> getInheritanceOrder() {
        res.clear();
        dfs(t);
        return res;
    }
};
