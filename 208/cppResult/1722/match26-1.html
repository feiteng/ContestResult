class ThroneInheritance {
public:
    unordered_map<string,vector<string>> v;
    unordered_map<string,bool> alive;
    string king;
    ThroneInheritance(string kingName) {
        king = kingName;
        alive[king] = true;
    }
    
    void birth(string parentName, string childName) {
        v[parentName].push_back(childName);
        alive[childName] = true;
    }
    
    void death(string name) {
        alive[name] = false;
    }
    
    void dfs(vector<string>& ans, string person) {
        if(alive[person]) ans.push_back(person);
        for(auto it:v[person]) {
            dfs(ans,it);
        }
    }
    
    vector<string> getInheritanceOrder() {
        ios_base::sync_with_stdio(false);
        cin.tie(0);
        cout.tie(0);
        
        vector<string> ans;
        dfs(ans,king);
        return ans;
    }
};
