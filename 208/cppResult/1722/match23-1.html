class ThroneInheritance {
public:
	#define maxn 100010
	#define pb push_back
	map<string,int> id;int idtot;
	vector<string> to;
	vector<int> e[maxn];
    bool died[maxn];
    ThroneInheritance(string s) {
		id.clear();to.clear();idtot=-1;
		memset(died,false,sizeof(died));
		id[s]=++idtot;to.pb(s);e[idtot].clear();
    }
    void birth(string x, string y) {
    	id[y]=++idtot;to.pb(y);
		e[id[x]].pb(id[y]);e[id[y]].clear();
    }
    void death(string s) {

		died[id[s]]=true;
    }
    void dfs(int x,vector<string> &re){
    	if(!died[x])re.pb(to[x]);
    	for(int y:e[x])dfs(y,re);
    }
    vector<string> getInheritanceOrder() {
		vector<string> re;
		dfs(0,re);
		return re;
    }
};
