class ThroneInheritance {
public:
    ThroneInheritance(string kingName) {
        tr[kingName] = vector<string>();
        king = kingName;
    }
    
    void birth(string parentName, string childName) {
        
        tr[parentName].push_back(childName);
    }
    
    void death(string name) {
        
        d.insert(name);
        //cout << "\n " << name;
    }
    
    vector<string> getInheritanceOrder() {
        
        vector<string> res;
        dfs(king,res);
        
        return res;
    }
    
    void dfs(string n,vector<string> &res)
    {
        //cout << "\n " << n << " " << d.count(n);
        if(d.count(n) < 1)
            res.push_back(n);
        
        for(auto vv:tr[n])
        {
            dfs(vv,res);
        }
    }
    
    unordered_map<string,vector<string>> tr;
    unordered_set<string> d;
    //vector<string> res;
    string king;
};
