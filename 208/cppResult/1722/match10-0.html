class ThroneInheritance {
public:
    map <string, vector <string> > children;
    map <string, bool> died;
    string King;
    ThroneInheritance(string kingName) {
        King = kingName;
        died[King] = 0;
    }
    
    void birth(string parentName, string childName) {
        died[childName] = 0;
        children[parentName].push_back(childName);
    }
    
    void death(string name) {
        died[name] = 1;
    }
    vector <string> inheritanceOrder;
    void move(string head) {
        if (!died[head]) inheritanceOrder.push_back(head);
        for (auto el : children[head]) {
            move(el);
        }
    }
    vector<string> getInheritanceOrder() {
        inheritanceOrder.clear();
        move(King);
        return inheritanceOrder;
    }
};
