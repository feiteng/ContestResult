class ThroneInheritance {
public:
    string king;
    unordered_map<string, bool> mp;
    unordered_map<string, vector<string>> son;
    ThroneInheritance(string kingName) {
        king = kingName;
    }
    
    void birth(string p, string c) {
        son[p].push_back(c);
    }
    
    void death(string name) {
        mp[name] = true;
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> res;
        dfs(res, king);
        return res;
    }
    
    void dfs(vector<string> &res, string cur) {
        if (!mp[cur]) {
            res.push_back(cur);
        }
        for (auto s : son[cur]) {
            dfs(res, s);
        }
    }
};


vector<int> preTraversal(TreeNode* root){
    vector<int> res;
    stack<TreeNode*> stk;
    TreeNode* cur = root;
    while(cur != NULL || !stk.empty()){
        while(cur != NULL){
            res.push_back(cur->val);
            stk.push(cur);
            cur = cur->left;
        }
        TreeNode* p = stk.top();
        stk.pop();
      	cur = p->right;
    }
    return res;
}
