class ThroneInheritance {
public:
    string root;
    set<string> ds;
    map<string, vector<string>> mp;
    
    ThroneInheritance(string kingName) {
        root=kingName;
        ds.clear();
        mp.clear();
        mp[root];
    }
    
    void birth(string parentName, string childName) {
        mp[parentName].push_back(childName);
    }
    
    void death(string name) {
        ds.insert(name);
    }
    
    void dfs(string x, vector<string> &s) {
        if (ds.find(x)==ds.end())s.push_back(x);
        for(string cc: mp[x]) {
            dfs(cc, s);
        }
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> ans; ans.clear();
        dfs(root, ans);
        return ans;
    }
};
