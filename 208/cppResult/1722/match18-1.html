class ThroneInheritance {
public:
    unordered_map<string, vector<string>> pa;
    unordered_map<string, bool> is_dead;
    string kingName;

    ThroneInheritance(string _kingName) {
        // pa[kingName]=new vector<string>();
        is_dead[kingName]=false;
        kingName = _kingName;
    }
    
    void birth(string parentName, string childName) {
        pa[parentName].push_back(childName);
        is_dead[childName]=false;
    }
    
    void death(string name) {
        is_dead[name]=true;
    }
    
    void dfs(string kingName, vector<string> &ret){
        if(!is_dead[kingName]){
            ret.push_back(kingName);
        }
        for(auto& child:pa[kingName]){
            dfs(child, ret);
        }
    }
    
    vector<string> getInheritanceOrder() {
        vector<string> ret;
        dfs(kingName, ret);
        return ret;
    }
};

 * obj->death(name);
 * vector<string> param_3 = obj->getInheritanceOrder();
