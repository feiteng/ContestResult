class ThroneInheritance {
public:
    unordered_map<string, vector<string>> e;
    string root;
    unordered_map<string, int> is_dead;

    ThroneInheritance(string kingName) {
        root = kingName;
    }
    
    void birth(string parentName, string childName) {
        e[parentName].push_back(childName);
    }
    
    void death(string name) {
        is_dead[name] = 1;
    }
    void dfs(string cur, vector<string> &res) {
        if( is_dead[cur] == 0) {
            res.push_back(cur);
        }
        for(auto p: e[cur]) {
            dfs(p, res);
        }
    }
    vector<string> getInheritanceOrder() {
        vector<string> res;
        dfs(root, res);
        return res;
    }
};

 * obj->death(name);
 * vector<string> param_3 = obj->getInheritanceOrder();
