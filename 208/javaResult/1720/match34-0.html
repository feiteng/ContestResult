class Solution {
    public int minOperations(String[] logs) {
        int level = 0;
        for (String log : logs) {
            if (log.charAt(0) != '.') {
                level ++;
            } else if (log.charAt(1) == '.' && level > 0) {
                level --;
            }
        }
        return level;
    }
}
