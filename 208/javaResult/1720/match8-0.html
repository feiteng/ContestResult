class Solution {
    public int minOperations(String[] logs) {
        int cur = 0;
        for (String log : logs) {
            if (log.equals("../")) {
                cur = Math.max(0, cur-1);
            } else if (!log.equals("./")) {
                ++cur;
            }
        }
        return cur;
    }
}
