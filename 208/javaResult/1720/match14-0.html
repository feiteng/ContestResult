class Solution {
    public int minOperations(String[] logs) {
        int result = 0;
        for(String x: logs){
            if(x.startsWith("..")){
                result = Math.max(0,result-1);
            }
            else if(!x.startsWith(".")){
                result++;
            }
        }
        return result;
    }
}
