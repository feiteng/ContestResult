class Solution {
    public int minOperations(String[] logs) {
        int cur = 0;
        for (String s : logs) {
            if (s.equals("../")) {
                if (cur > 0) cur--;
            } else if (s.equals("./")) {
                cur += 0;
            } else {
                cur++;
            }
        }
        return cur;
    }
}
