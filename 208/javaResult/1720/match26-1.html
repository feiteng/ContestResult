class Solution {
    public int minOperations(String[] logs) {
        int depth = 0;
        for (String s : logs) {
            if ("../".equals(s)) {
                depth--;
                depth = Math.max(0, depth);
            } else if ("./".equals(s)) {
            } else {
                depth++;
            }
        }
        return depth;
    }
}
