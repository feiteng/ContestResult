class Solution {
    public int minOperations(String[] logs) {
        int depth = 0;
        for (String log : logs) {
            if (log.equals("../")) {
                depth -= 1;
                if (depth < 0) {
                    depth = 0;
                }
            } else if (log.equals("./")) {
                continue;
            
} else {
                depth += 1;
            }
        }
        return depth;
    }
}
