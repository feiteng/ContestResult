class Solution {
    public int minOperations(String[] logs) {
        int count = 0;
        for (String log : logs) {
            if ("../".equals(log)) {
                count = (count == 0 ? 0 : count - 1);
            } else if ("./".equals(log)) {
            } else {
                count++;
            }
        }
        return count;
    }
}
