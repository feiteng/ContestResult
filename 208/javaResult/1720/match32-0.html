import java.util.*;

class Solution {
    public int minOperations(String[] logs) {

        int level = 0;
        for(String log : logs) {
            if (log.equals("../")) {
                level = level > 0 ? (level-1) : level;
            }
            else if (log.equals("./")) {
                continue;
            }
            else {
                level++;
            }
        }
        return level;
    }
}
