class Solution {
    public int minOperations(String[] logs) {
        int lvl = 0;
        for(String log : logs)
        {
            if(log.equals("../"))
                lvl = Math.max(0, lvl - 1);
            else if(log.equals("./")){}
            else 
lvl++;
        }
        return lvl;
    }
}
