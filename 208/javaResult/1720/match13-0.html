class Solution {
    public int minOperations(String[] logs) {
        int cnt = 0;
        for (String l : logs) {
            if ("../".equals(l)) {
                cnt = Math.max(0, cnt - 1);
            } else if (!"./".equals(l)) {
                cnt++;
            }
        }
        return cnt;
    }
}
