class ThroneInheritance {
    private class Member {
        private String name;
        private boolean isDead;
        private boolean isOrder;
        private List<Member> children;
        
        public Member(String name) {
            this.name = name;
            this.children = new LinkedList<Member>();
        }
    }
    
    private Member king;
    private Map<String, Member> map;

    public ThroneInheritance(String kingName) {
        king = new Member(kingName);
        map = new HashMap<String, Member>();
        map.put(kingName, king);
    }
    
    public void birth(String parentName, String childName) {
        Member parent = map.get(parentName);
        Member child = new Member(childName);
        parent.children.add(child);
        map.put(childName, child);
    }
    
    public void death(String name) {
        Member member = map.get(name);
        member.isDead = true;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> list = new LinkedList<String>();
        add(list, king);
        return list;
    }
    
    private void add(List<String> list, Member member) {
        if (!member.isDead) {
            list.add(member.name);
        }
        for (Member child : member.children) {
            add(list, child);
        }
    }
}
