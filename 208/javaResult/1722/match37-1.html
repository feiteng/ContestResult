class ThroneInheritance {
    
    private final String kingName;
    private Map<String,List<String>> royalTree;
    private Set<String> deadSet;
    
    public ThroneInheritance(String kingName) {
        this.kingName = kingName;
        this.royalTree = new HashMap<>();
        this.deadSet = new HashSet<>();
    }
    
    public void birth(String parentName, String childName) {
        if (!royalTree.containsKey(parentName)) {
            royalTree.put(parentName, new ArrayList<>());
        }
        
royalTree.get(parentName).add(childName);
    }
    
    public void death(String name) {
        deadSet.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> order = new ArrayList<>();
        traverseDepthFirst(kingName, order);
        return order;
    }
    
    private void traverseDepthFirst(String cur, List<String> order) {
        if (cur == null) {
            return;
        }
        if (!
deadSet.contains(cur)) {
            order.add(cur);
        }
        for (String child : royalTree.getOrDefault(cur, new ArrayList<>())) {
            traverseDepthFirst(child, order);
        }
    }
}
