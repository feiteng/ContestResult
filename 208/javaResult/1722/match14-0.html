class ThroneInheritance {
    
    Map<String, List<String>> tree;
    String king;
    Map<String, Boolean> death;

    public ThroneInheritance(String kingName) {
        king = kingName;
        tree = new HashMap<>();
        tree.put(kingName, new ArrayList<>());
        death = new HashMap<>();
    }
    
    public void birth(String parentName, String childName) {
        tree.get(parentName).add(childName);
        tree.put(childName, new ArrayList<>());
    }
    
    public void death(String name) {
        death.put(name, true);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> list = new ArrayList<>();
        dfs(king, list);
        return list;
    }
    private void dfs(String name, List<String> list) {
        if (!death.containsKey(name)) list.add(name);
        for (String child : tree.get(name)) {
            dfs(child, list);
        }
    }
}
