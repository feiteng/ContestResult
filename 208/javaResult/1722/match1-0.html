class ThroneInheritance {
    
    HashMap<String, ArrayList<String>> kingdomHierarchy;
    HashSet<String> deathSet;
    String king;

    public ThroneInheritance(String kingName) {
        kingdomHierarchy = new HashMap<>();
        kingdomHierarchy.put(kingName, new ArrayList<>());
        deathSet = new HashSet<>();
        king = kingName;
    }
    
    public void birth(String parentName, String childName) {
        kingdomHierarchy.get(parentName).add(childName);
        kingdomHierarchy.put(childName, new ArrayList<>());
    }
    
    public void death(String name) {
        deathSet.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> inheritanceList = new ArrayList<>();
        traverse(kingdomHierarchy, king, inheritanceList);
        return inheritanceList;
    }
    
    private void traverse(HashMap<String, ArrayList<String>> kingdomHierarchy, String personName, List<String> inheritanceList) {
        if(!deathSet.contains(personName)) {
            inheritanceList.add(personName);
        }
        
        for(String childName : kingdomHierarchy.get(personName)) {
            traverse(kingdomHierarchy, childName, inheritanceList);
        }
    }
}
