class ThroneInheritance {

    class Node{
        Node p;
        String name;
        boolean dead = false;
        List<Node> c = new LinkedList<>();
        Node(String n) { 
this.name = n; }
    }
    
    Node root;
    
    HashMap<String, Node> map = new HashMap<>();
    
    public ThroneInheritance(String kingName) {
        root = new Node(kingName);
        map.put(kingName, root);
    }
    
    public void birth(String parentName, String childName) {
        Node parent = map.get(parentName);
        Node me = new Node(childName);
        me.p = parent;
        
parent.c.add(me);
        map.put(childName, me);
    }
    
    public void death(String name) {
        Node d = map.remove(name);
        d.dead = true;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> ans = new ArrayList<>();
        dfs(root, ans);
        return ans;
    }
    
    private void dfs(Node root, List<String> ans) {
        if(root == null) return;
        if(!root.dead) ans.add(root.name);
        for(Node ch : root.c) {
            dfs(ch, ans);
        }
    }
}
