class ThroneInheritance {
    Map<String, List<String>> pToC;
    Set<String> dead;
    String king;
    
    public ThroneInheritance(String kingName) {
        pToC = new HashMap<>();
        pToC.put(kingName, new ArrayList<>());
        
        dead = new HashSet<>();
        king = kingName;
    }
    
    public void birth(String parentName, String childName) {
        if (!pToC.containsKey(parentName) || dead.contains(parentName))  {
            return;
        }
        
        pToC.get(parentName).add(childName);
        pToC.put(childName, new ArrayList<>());
    }
    
    public void death(String name) {
        dead.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> res = new ArrayList<>();
        
        dfs(king, res);
        
        return res;
    }
    
    private void dfs(String parent, List<String> list) {
        if (!dead.contains(parent)) {
            list.add(parent);
        }
        
        if (!pToC.containsKey(parent)) {
            return;
        }
        
        for (String child : pToC.get(parent)) {
            dfs(child, list);
        }
    }
}
