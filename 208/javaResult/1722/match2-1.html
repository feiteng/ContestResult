class ThroneInheritance {

    Map<String, List<String>> graph = new HashMap<String, List<String>>();
    Set<String> passed = new HashSet<String>();
    String root;
    public ThroneInheritance(String kingName) {
        root = kingName;
        graph.put(kingName, new ArrayList<String>());
    }
    
    public void birth(String parentName, String childName) {
        graph.get(parentName).add(childName);
        graph.put(childName, new ArrayList<String>());
    }
    
    public void death(String name) {
        passed.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> res = new ArrayList<String>();
        helper(res, root, graph);
        return res;
    }
    
    private void helper(List<String> res, String root, Map<String, List<String>> graph)
    
{
        if(!passed.contains(root))
        {
            res.add(root);
        }

        for(String child : graph.get(root))
        {
            helper(res, child, graph);
        }
    }
}
