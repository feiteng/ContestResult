class ThroneInheritance {
    
    class Node {
        String name;
        List<Node> children;
        
        public Node(String name) {
            this.name = name;
            this.children = new ArrayList<>();
        }
    }
    
    private Node root;
    private Set<String> dead;
    private Map<String, Node> map;

    public ThroneInheritance(String kingName) {
        this.root = new Node(kingName);
        this.dead = new HashSet<>();
        this.map = new HashMap<>();
        this.map.put(kingName, root);
    }
    
    public void birth(String parentName, String childName) {
        Node baby = new Node(childName);
        this.map.get(parentName).children.add(baby);
        this.map.put(childName, baby);
    }
    
    public void death(String name) {
        this.dead.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> order = new ArrayList<>();
        getOrder(order, root);
        return order;
    }
    
    private void getOrder(List<String> order, Node node) {
        if (!dead.contains(node.name)) order.add(node.name);
        for (Node child : node.children) {
            getOrder(order, child);
        }
    }
}
