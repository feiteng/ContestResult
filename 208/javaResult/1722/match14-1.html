class ThroneInheritance {

        HashMap<String,LinkedList<String>> lists;
        HashSet<String> death;
        String king;
        public ThroneInheritance(String kingName)
        {
            king=kingName;
            lists=new LinkedHashMap<>();
            lists.put(kingName,new LinkedList<>());
            death=new HashSet<>();
        }

        public void birth(String parentName, String childName)
        {
            lists.get(parentName).addLast(childName);
            lists.put(childName,new LinkedList<>());
        }

        public void death(String name)
        {
            death.add(name);
        }

        public List<String> getInheritanceOrder()
        {
            List<String> ansr=new ArrayList<>();
            dfs(ansr,king);
            return ansr;
        }
        public void dfs(List<String> ansr,String cur)
        {
            if (!death.contains(cur))
            {
                ansr.add(cur);
            
}

            for (String s : lists.get(cur))
            {
                dfs(ansr, s);
            }
        }
}
