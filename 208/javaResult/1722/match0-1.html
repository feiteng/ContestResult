class ThroneInheritance {
    
    Map<String, List<String>> parent = new HashMap<>();
    Set<String> dead = new HashSet<>();
    String kingName;

    public ThroneInheritance(String kingName) {
        this.kingName = kingName;
        parent.put(kingName, new ArrayList<>());
    }
    
    public void birth(String parentName, String childName) {
        parent.putIfAbsent(parentName, new ArrayList<>());
        parent.get(parentName).add(childName);
        
    }
    
    public void death(String name) {
        dead.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> res = new ArrayList<>();
        dfs(kingName, res);
        return res;
    }
    
    private void dfs(String start, List<String> res) {
        if (!dead.contains(start)) {
            res.add(start);
        }
        if (parent.containsKey(start)) {
            for (String child : parent.get(start)) {
                dfs(child, res);
            }
        }
    }
}
