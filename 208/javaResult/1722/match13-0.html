class ThroneInheritance {
    Map<String, List<String>> graph; 
    String king;
    Set<String> death;

    public ThroneInheritance(String kingName) {
        graph = new HashMap<>();
        king = kingName;
        death = new HashSet<>();
    }
    
    public void birth(String parentName, String childName) {
        graph.putIfAbsent(parentName, new ArrayList<>());
        graph.get(parentName).add(childName);
    }
    
    public void death(String name) {
        death.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> result = new ArrayList<>();
        dfs(king, result);
        return result;
    }
    
    void dfs(String parent, List<String> list){
        if(!death.contains(parent)) list.add(parent);
        if(!graph.containsKey(parent)) return;
        for(String child : graph.get(parent)){
            dfs(child, list);
        }
    }
}
