class ThroneInheritance {

    Map<String, Node> ref;
    Node root;
    
    public ThroneInheritance(String kingName) {
        root = new Node(kingName);
        ref = new HashMap<>();
        ref.put(kingName, root);
    }
    
    public void birth(String parentName, String childName) {
        Node parent = ref.get(parentName);
        Node child = parent.addChild(childName);
        ref.put(childName, child);
    }
    
    public void death(String name) {
        ref.get(name).alive = false;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> order = new ArrayList<> ();
        traverse(root, order);
        return order;
    }
    
    public void traverse(Node root, List<String> container) {
        if (root.alive) {
            container.add(root.name);
        }
        for (Node child : root.children) {
            traverse(child, container);
        }
    }
}

class Node {
    String name;
    boolean alive;
    List<Node> children;
    
    public Node(String name) {
        this.name = name;
        this.alive = true;
        this.children = new ArrayList<>();
    }
    
    public Node addChild(String childName) {
        Node child = new Node(childName);
        children.add(child);
        return child;
    }
}
