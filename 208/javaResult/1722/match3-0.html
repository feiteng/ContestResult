class ThroneInheritance {
    
    Map<String, List<String>> familyMap = new HashMap<>();
    Set<String> deaths = new HashSet<>();
    String king;

    public ThroneInheritance(String kingName) {
        this.familyMap.put(kingName, new ArrayList<>());
        this.king = kingName;
    }
    
    public void birth(String parentName, String childName) {
        this.familyMap.get(parentName).add(childName);
        this.familyMap.put(childName, new ArrayList<>());
    }
    
    public void death(String name) {
        deaths.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> order = new ArrayList<>();
        successor(this.king, order);
        return order;
    }
    
    public void successor(String name, List<String> orderList) {
        if (!this.deaths.contains(name)) orderList.add(name);
        for (String child : this.familyMap.get(name)) {
            successor(child, orderList);
        }
    }
}
