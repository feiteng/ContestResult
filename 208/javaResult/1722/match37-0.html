class ThroneInheritance {
    String king;
    Set<String> dead;
    Map<String, List<String>> kids;

    public ThroneInheritance(String kingName) {
        this.king = kingName;
        this.dead = new HashSet<>();
        this.kids = new HashMap<>();
    }
    
    public void birth(String parentName, String childName) {
        kids.computeIfAbsent(parentName, a -> new ArrayList<>());
        kids.get(parentName).add(childName);
    }
    
    public void death(String name) {
        dead.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> out = new ArrayList<>();
        helper(king, out);
        return out;
    }
    
    private void helper(String name, List<String> out) {
        if(!dead.contains(name)) {
            out.add(name);
        }
        
        for(String child : kids.getOrDefault(name, new ArrayList<>())) {
            helper(child, out);
        }
    }
}
