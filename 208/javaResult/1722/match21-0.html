class ThroneInheritance {
    TreeNode root;
    Map<String, TreeNode> map;
    Set<String> death;
    public ThroneInheritance(String kingName) {
        death = new HashSet<>();
        map = new HashMap<>();
        root = new TreeNode(kingName);
        map.put(kingName, root);
        
    }
    
    public void birth(String parentName, String childName) {
        TreeNode node = map.get(parentName);
        TreeNode newNode = new TreeNode(childName);
        node.child.add(newNode);
        map.put(childName, newNode);
    }
    
    public void death(String name) {
        death.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> list = new ArrayList<>();
        check(root, list);
        return list;
        
    }
    
    public void check(TreeNode node, List<String> list) {
        String curr = node.name;
        if (!death.contains(curr)) {
            list.add(curr);
        }
        List<TreeNode> child = node.child;
        for (TreeNode c : child) {
            check(c, list);
        }
    }
}

class TreeNode {
    String name;
    List<TreeNode> child;
    
    public TreeNode(String name) {
        this.name = name;
        this.child = new ArrayList<>();
    }
}
