class ThroneInheritance {
    
    private static class Person {
        String name;
        List<Person> kids;
        boolean dead;
        
        private Person(String name) {
            this.name = name;
            this.kids = new ArrayList<>();
            this.dead = false;
        }
    }
    
    private Person king;
    Map<String, Person> nameToPerson;
    public ThroneInheritance(String kingName) {
        this.king = new Person(kingName);
        this.nameToPerson = new HashMap<>();
        this.nameToPerson.put(kingName, this.king);
    }
    
    public void birth(String parentName, String childName) {
        Person parent = nameToPerson.get(parentName);
        Person child = new Person(childName);
        parent.kids.add(child);
        nameToPerson.put(childName, child);
    }
    
    public void death(String name) {
        Person p = nameToPerson.get(name);
        p.dead = true;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> result = new ArrayList<>();
        preorder(king, result);
        return result;
    }
    
    private void preorder(Person p, List<String> result) {
        if (!p.dead) {
            result.add(p.name);
        }
        for (Person kid : p.kids) {
            preorder(kid, result);
        }
    }
}
