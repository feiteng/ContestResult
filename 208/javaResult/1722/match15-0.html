class ThroneInheritance {
    
    class TreeNode {
        String name;
        TreeNode parent;
        boolean dead;
        List<TreeNode> children = new ArrayList<>();
        
        TreeNode(TreeNode parent, String name) {
            this.parent = parent;
            this.name = name;
        }
    }
    
    TreeNode root;
    Map<String, TreeNode> map = new HashMap<>();

    public ThroneInheritance(String kingName) {
        root = new TreeNode(null, kingName);
        map.put(kingName, root);
    }
    
    public void birth(String parentName, String childName) {
        TreeNode parent = map.get(parentName);
        TreeNode child = new TreeNode(parent, childName);
        parent.children.add(child);
        map.put(childName, child);
    }
    
    public void death(String name) {
        TreeNode child = map.get(name);
        child.dead = true;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> ans = new ArrayList<>();
        preorder(root, ans);
        return ans;
    }
    
    void preorder(TreeNode root, List<String> ans) {
        if (root == null) return;
        
        if (!root.dead) ans.add(root.name);
        
        for (TreeNode child : root.children) {
            preorder(child, ans);
        }
    }
}
