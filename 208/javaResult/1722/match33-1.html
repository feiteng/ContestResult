class ThroneInheritance {
    
private class Royal {
        public String name;
        public boolean alive;
        public List<Royal> children;
        public Royal(String n_) {
            this.name = n_;
            this.alive = true;
            this.children = new ArrayList<>();
        }
    }
    
    
private Royal king;
    private HashMap<String, Royal> hash;
    public ThroneInheritance(String kingName) {
        this.king = new Royal(kingName);
        this.hash = new HashMap<>();
        this.hash.put(kingName, this.king);
        
    }
    
    public void birth(String parentName, String childName) {
        Royal temp = new Royal(childName);
        this.hash.put(childName, temp);
        
this.hash.get(parentName).children.add(temp);
    }
    
    public void death(String name) {
        this.hash.get(name).alive = false;
    }
    
    public List<String> getInheritanceOrder() {
        List<String> out = new ArrayList<>();
        getInheritanceOrder(this.king, out);
        return out;
    }
    
    private void getInheritanceOrder(Royal r, List<String> out) {
        if (r.alive) { out.add(r.name); }
        for (Royal c : r.children) { getInheritanceOrder(c, out); }
    }
}
