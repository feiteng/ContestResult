class ThroneInheritance {
    
    class TreeNode{
        String name;
        List<TreeNode> children;
        public TreeNode(String name){
            this.name = name;
            children = new LinkedList<>();
        }
        public TreeNode add(String child){
            TreeNode c = new TreeNode(child);
            children.add(c);
            return c;
        
}
    }
    
    TreeNode king;
    HashSet<String> dead;
    HashMap<String,TreeNode> people;
    public ThroneInheritance(String kingName) {
        king = new TreeNode(kingName);
        dead = new HashSet<>();
        people = new HashMap<>();
        people.put(kingName, king);
    }
    
    public void birth(String parentName, String childName) {
        TreeNode p = people.get(parentName);
        TreeNode c = 
p.add(childName);
        people.putIfAbsent(childName, c);
    }
    
    public void death(String name) {
        dead.add(name);
    }
    
    public List<String> getInheritanceOrder() {
        List<String> result = new LinkedList<>();
        dfs(king, result);
        return result;
    }
    
    public void dfs(TreeNode root, List<String> result){
        if(root == null) return;
        if(!
dead.contains(root.name)) result.add(root.name);
        for(TreeNode child : root.children){
            dfs(child, result);
        }
    }
}
