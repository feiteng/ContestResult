class Solution {
    int max = 0;
    public int maximumRequests(int n, int[][] requests) {
        dfs(0, new int[n], requests, 0);
        return max;
    }
    
    public void dfs(int idx, int[] capacity, int[][] requests, int count) {
        if (idx == requests.length) {
            for (int i : capacity)
                if (i != 0)
                    return;
            max = Math.max(max, count);
            return;
        }
        dfs(idx + 1, capacity, requests, count);
        --capacity[requests[idx][0]];
        ++capacity[requests[idx][1]];
        dfs(idx + 1, capacity, requests, count + 1);
        ++capacity[requests[idx][0]];
        --capacity[requests[idx][1]];
        return;
    }
}
