class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int maxApproved = 0;
        for (int i = 0; i < (1 << requests.length); i++) {
            int[] net = new int[n];
            int curApproved = 0;
            for (int j = 0; j < requests.length; j++) {
                if ((i & (1 << j)) > 0) {
                    int from = requests[j][0];
                    int to = requests[j][1];
                    net[from]--; net[to]++;
                    curApproved++;
                
}
            }
            boolean isValidApproval = true;
            for (int j = 0; j < net.length; j++) {
                if (net[j] != 0) {
                    isValidApproval = false;
                    break;
                }
            }
            if (isValidApproval) {
                maxApproved = Math.max(maxApproved, curApproved);
            }
        }
        return maxApproved;
    }
}
