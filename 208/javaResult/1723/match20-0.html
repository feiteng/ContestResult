class Solution {
    public int maximumRequests(int n, int[][] requests) {

        int m = requests.length;
        int mask = 1 << (m);
        // System.out.println("mask " + Integer.toBinaryString(mask - 1));

        int res = 0;
        for (int i = 0; i < mask; i++) {
            int[] buildings = new int[n];

            for (int j = 0; j < m; j++) {
                if ((i & (1 << j)) != 0) {
                    buildings[requests[j][0]]--;
                    buildings[requests[j][1]]++;
                }
            }
            
            boolean valid = true;
            for (int k = 0; k < n && valid; k++) {
                valid = buildings[k] == 0;
            }
            
            if (valid) {
                // System.out.println(Arrays.toString(buildings));
                // System.out.println(Integer.toBinaryString(i));
                res = Math.max(res, Integer.bitCount(i));
            }
        }
        
        // System.out.println(" ");
        
        return res;
    }
}
