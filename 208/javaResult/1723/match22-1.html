class Solution {
    int max = 0;
    public int maximumRequests(int n, int[][] requests) {
        dfs(requests, 0, new int[n], 0);
        return max;
    }
    void dfs(int[][] req, int idx, int[] st, int cost) {
        if (idx == req.length) {
            for (int i : st) if (i != 0) return;
            max = Math.max(cost, max);
            return;
        }
        int f = req[idx][0], t = req[idx][1];
        st[f]--; st[t]++;
        dfs(req, idx + 1, st, cost + 1);
        st[f]++; st[t]--;
        dfs(req, idx + 1, st, cost);
    }
}
