class Solution {
    public int maximumRequests(int n, int[][] requests) {
        final int k = requests.length;
        int best = 0;
        for (int mask = 0; mask < (1 << k); ++mask) {
            if (doable(mask, n, requests)) {
                best = Math.max(best, Integer.bitCount(mask));
            }
        }
        return best;
    }
    
    private static boolean doable(int mask, int n, int[][] requests) {
        int[] deltas = new int[n];
        for (int i = 0; i < requests.length; ++i) {
            if (((mask >> i) & 1) == 1) {
                --deltas[requests[i][0]];
                ++deltas[requests[i][1]];
            }
        }
        for (int v : deltas) {
            if (v != 0) {
                return false;
            }
        }
        return true;
    }
}
