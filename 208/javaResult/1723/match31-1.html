class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int ans = 0;
        int len = requests.length;
        for(int i = 0; i < (1 << len); i++)
        {
            int status = 0;
            for(int j = 0; j < len; j++)
            {
                if((i & (1 << j)) != 0)
                {
                    status |= 1 << j;
                }
            }
            if(canTransfer(status, requests, n))
            {
                // System.out.println(Integer.toBinaryString(status));
                ans = Math.max(ans, Integer.bitCount(status));
            }
        
}
        // System.out.println("...");
        return ans;
    }
    
    boolean canTransfer(int status, int[][] req, int n)
    {
        int[] f = new int[n];
        int len = req.length;
        for(int i = 0; i < len; i++)
        {
            if((status & (1 << i)) != 0)
            {
                int from = req[i][0], to = req[i][1];
                f[from]--;
                f[to]++;
            }
        
}
        // System.out.println(Integer.toBinaryString(status));
        // System.out.println(Arrays.toString(f));
        for(int v : f) if(v != 0) return false;
        return true;
    }
}
