class Solution {
    public int maximumRequests(int n, int[][] req) {
        int m = req.length;
        int sol = 0;
        for (int i = 0; i < (1<<m); i++) {
            int[] ct = new int[n];
            int len = 0;
            for (int j = 0; j < m; j++) {
                if (((1<<j) & i) != 0) {
                    ct[req[j][0]]--;
                    ct[req[j][1]]++;
                    len++;
                }
            }
            
boolean work = true;
            for (int j = 0; j < n; j++) {
                if (ct[j] != 0) {
                    work = false;
                    break;
                }
            }
            if (work) sol = Math.max(sol, len);
        }
        return sol;
    }

}
