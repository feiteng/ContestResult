class Solution {
    public int maximumRequests(int n, int[][] r) {
        int l = r.length;

        int t = (1 << (l + 1));
        int res = 0;
        for(int i = 0; i < t; i++){
            int [] ru = new int[n];
            int [] cu = new int[n];
            int cnt = 0;
            
for(int j = 0; j < l; j++){
                if((i & (1 << j)) != 0){
                    cnt++;
                    ru[r[j][1]] ++;
                    cu[r[j][0]] ++;
                }
            }
            boolean flag = true;
            for(int j = 0; j < n; j++){
                if(ru[j] != cu[j]){
                    flag = false;
                    break;
                }
            }
            if(flag){
                res = 
Math.max(res, cnt);
            }
        }
        return res;
        
    }
}
