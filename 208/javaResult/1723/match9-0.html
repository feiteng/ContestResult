class Solution {
    private int max = 0;

    public int maximumRequests(int n, int[][] requests) {

        int[] cnt = new int[n];
        dfs(cnt, 0, 0, n, requests);

        return max;
    }

    private void dfs(int[] cnt, int k, int curr, int n, int[][] arr) {

        if (curr + arr.length - 1 - k < max) {
            return;
        }

        if (k == arr.length) {
            if (allZero(cnt)) {
                max = Math.max(max, curr);
            }
            return;
        }

        int[] p = arr[k];
        int v = p[0], w = p[1];
        cnt[v]++;
        cnt[w]--;

        dfs(cnt, k + 1, curr + 1, n, arr);
        cnt[v]--;
        cnt[w]++;
        dfs(cnt, k + 1, curr, n, arr);


    }

    private boolean allZero(int[] arr) {
        for (int n : arr) {
            if (n != 0) {
                return false;
            }
        }
        return true;
    }
}
