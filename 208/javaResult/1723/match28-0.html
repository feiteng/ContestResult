class Solution {
    
    int res = 0;
    
    public int maximumRequests(int n, int[][] requests) {
        int[] move = new int[n];
        dfs(requests, 0, 0, move);
        return res;
    }
    
    private void dfs(int[][] req, int cur, int pre, int[] move) {
        if (cur == req.length) {
            for (int m : move) {
                if (m != 0) {
                    return;
                }
            }
            res = Math.max(res, pre);
            return;
        }
        dfs(req, cur + 1, pre, move);
        move[req[cur][0]] -= 1;
        move[req[cur][1]] += 1;
        dfs(req, cur + 1, pre + 1, move);
        move[req[cur][0]] += 1;
        move[req[cur][1]] -= 1;
    }
}
