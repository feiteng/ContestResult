class Solution {
    public int maximumRequests(int n, int[][] reqs) {
        int m = reqs.length;
        int ans = 0;
        for (int i = 1; i < (1<<m); i++)
            ans = Math.max(ans, cnt(i, n, reqs));
        return ans;
    }
    
    int cnt(int cur, int n, int[][] reqs) {
        int[] dx = new int[n];
        int ans = 0;
        for (int i = 0; i < reqs.length; i++)
            if (((1<<i) & cur) > 0) {
                ans++;
                dx[reqs[i][0]]--;
                dx[reqs[i][1]]++;
            }
        for (int x : dx)
            if (x != 0)
                return -1;
        return ans;
    }
}
