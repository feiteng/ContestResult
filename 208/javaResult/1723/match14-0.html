class Solution {
    int[] log ;
    int res = 0;
    public int maximumRequests(int n, int[][] requests) {
        log = new int[n];
        dfs(requests,0,0);
        return res;
    }
    public void dfs(int[][] r, int idx, int cnt){
        if(idx>=r.length){
            for(Integer i: log){
                if(i!=0){
                    return;
                }
            }
            res = Math.max(cnt,res);
            return;
        }
        log[r[idx][0]]--;
        log[r[idx][1]]++;
        dfs(r, idx+1, cnt+1);
        log[r[idx][0]]++;
        log[r[idx][1]]--;
         dfs(r, idx+1, cnt);
    }
}
