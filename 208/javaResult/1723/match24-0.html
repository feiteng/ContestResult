class Solution {
  public int maximumRequests(int n, int[][] requests) {
    final int R = requests.length;
    int res = 0;
    for (int i = 1; i != (1 << R); i++) {
      if (net0(n, i, requests)) {
        res = Math.max(res, Integer.bitCount(i));
      }
    }
    return res;
  }

  private boolean net0(final int N, int bits, int[][] requests) {
    int[] builds = new int[N];
    for (int i = 0; i != requests.length; i++) {
      if (0 != (bits & (1 << i))) {
        builds[requests[i][0]]--;
        builds[requests[i][1]]++;
      }
    }
    for (int i : builds)
      if (i != 0)
        return false;
    return true;
  }
}
