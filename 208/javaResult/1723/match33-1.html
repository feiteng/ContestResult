class Solution {
    public int maximumRequests(int n, int[][] req) {
        int m = req.length;
        int N = (1 << m), res = 0;
        //System.out.println(n);
        for (int i = 0; i < N; i++) {
            
int[] cnt = new int[n];
            int cur = 0;
            for (int j = 0; j < m; j++) {
                if (((1 << j) & i) != 0) {
                    cur++;
                    cnt[req[j][0]]++;
                    cnt[req[j][1]]--;
                }
            }
            boolean can = true;
            for (int j = 0; j < n; j++) {
                //if (i == 53) System.out.print(cnt[j] + ", ");
                if (cnt[j] != 0) {
                    can = false;
                    break;
                }
            }
            if (can) res = Math.max(res, cur);
        }
        return res;
    }
}
