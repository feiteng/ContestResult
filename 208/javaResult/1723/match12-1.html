class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int reqLen = requests.length;
        int max = -1;
        for (int i = 0; i < (1 << reqLen); i++) {
            int res = helper(n, requests, i);
            max = Math.max(max, res);
        
}
        return max;
    }
    
    private int helper(int n, int[][] requests, int choice) {
        int[] net = new int[n];
        int num = 0;
        
for (int i = 0; i < requests.length; i++) {
            if ((choice & (1 << i)) > 0) {
                net[requests[i][0]]--;
                net[requests[i][1]]++;
                num++;
            }
        }
        for (int thisNet : net) {
            if (thisNet != 0) {
                return -1;
            }
        }
        return num;
    }
}
