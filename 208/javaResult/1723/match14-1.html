class Solution {
    int[] cnt;
    int res = 0;
    public int maximumRequests(int n, int[][] requests) {
        cnt = new int[n];
        dfs(0, requests, 0);
        return res;
    }
    private void dfs(int curIdx, int[][] requests, int handled) {
        if(curIdx == requests.length) {
            for(int i = 0; i < cnt.length; i++) if(cnt[i] != 0) return 
;
            
            res = Math.max(res, handled);
            return ;
        }
        
        // handle
        cnt[requests[curIdx][0]]--;
        cnt[requests[curIdx][1]]++;
        dfs(curIdx + 1, requests, handled + 1);
        cnt[requests[curIdx][0]]++;
        cnt[requests[curIdx][1]]--;

        dfs(curIdx + 1, requests, handled);
    }
}
