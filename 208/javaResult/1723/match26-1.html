class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int maxR = 0;
        int rn = requests.length;
        for (int i = 0; i < (1<<rn); i++) {
            int[] stats = new int[n];
            int rc = 0;
            for (int j = 0; j < rn; j++) {
                if ((i & (1<<j)) > 0) {
                    rc++;
                    stats[requests[j][0]]--;
                    stats[requests[j][1]]++;
                }
            
}
            boolean qualified = true;
            for (int k = 0; k < n; k++) 
                if (stats[k] != 0) {
                    qualified = false;
                    break;
                }
            if (qualified) maxR = Math.max(maxR, rc);
        }
        return maxR;
    }
}
