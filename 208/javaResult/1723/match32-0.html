class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int m=requests.length, ans=0;
        for(int state=1; state<(1<<m); state++){
            if(isValid(n,requests,state)){
                ans=Math.max(ans,Integer.bitCount(state));
            }
        }
        return ans;
    }
    
    private boolean isValid(int n,int[][] requests, int state){
        int[] b=new int[n];
        int m=requests.length;
        for(int i=0;i<m; i++){
            if((state & (1<<i)) > 0){
                int from=requests[i][0], to=requests[i][1];
                b[from]--;
                b[to]++;
            }
        }
        
        for(int bi:b){
            if(bi!=0) return false;
        }
        return true;
    }
}
