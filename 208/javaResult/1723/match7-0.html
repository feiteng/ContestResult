class Solution {
    public int maximumRequests(int n, int[][] r) {
        int size = r.length;
        int res = 0;
        for(int i = 1; i < (1<<size); i++){
            int[] arr = new int[n];
            int j = 0;
            int count = 0;
            int copy = i;
            while(copy > 0){
                if(copy%2 == 1) {
                    int[] tt = r[j];
                    arr[tt[0]]++;
                    arr[tt[1]]--;
                    count++;
                }
                j++;
                copy/=2;
            }
            boolean good = true;
            for(int x: arr){
                if(x != 0) {
                    good = false;
                    break;
                }
            }
            if(good) res = Math.max(res, count);
        }
        return res;
    }
}
