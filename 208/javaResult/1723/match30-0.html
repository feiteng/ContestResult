class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int res = 0;
        for (int i = 0; i < 1 << requests.length; i++) {
            int[][] freq = new int[n][2];
            int count = 0;
            for (int j = 0; j < requests.length; j++) {
                if ((i >> j) % 2 == 1) {
                    freq[requests[j][0]][0]++;
                    freq[requests[j][1]][1]++;
                    count++;
                }
            }
            boolean valid = true;
            for (int j = 0; j < n; j++) {
                if (freq[j][0] != freq[j][1]) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                res = Math.max(res, count);
            }
        }
        return res;
    }
}
