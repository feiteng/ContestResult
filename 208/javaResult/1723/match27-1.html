class Solution {
    int res = 0;
    public int maximumRequests(int n, int[][] requests) {
        int[] buffer = new int[n];
        int m = requests.length;
        bt(buffer, requests, 0, 1);
        
        return res;
    }
    private void bt(int[] buffer, int[][] requests, int start, int num){
        for(int i = start; i < requests.length; ++i){
            int[] req = requests[i];
            buffer[req[0]] -= 1;
            buffer[req[1]] += 1;
            if(allZero(buffer)){
                res = Math.max(res, num);
            }
            
bt(buffer, requests, i + 1, num + 1);
            
            // reset
            buffer[req[0]] += 1;
            buffer[req[1]] -= 1;
        }
    }
    private boolean allZero(int[] buffer){
        for(int v: buffer){
            if(v != 0) return false;
        }
        return true;
    }
}
