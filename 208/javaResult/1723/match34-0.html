class Solution {
    int D = 0;
    int res = -1;
    int N = 0;
    public int maximumRequests(int n, int[][] requests) {
        int[] status = new int[n];
        N = n;
        for(int i = 0; i < n; i++){
            status[i] = 0;
        }
        D = requests.length;
        dfs(requests, status, 0, 0);
        return res;
    }
    public void dfs(int[][] requests, int[] status, int depth, int num) {
        if(depth == D) {
            for(int i = 0; i < N; i++) {
                if(status[i] != 0) {
                    return;
                }
            }
            res = Math.max(res, num);
            return;
        }
        status[requests[depth][0]]--;
        status[requests[depth][1]]++;
        dfs(requests, status, depth+1, num+1);
        status[requests[depth][0]]++;
        status[requests[depth][1]]--;
        dfs(requests, status, depth+1, num);
    }
}
