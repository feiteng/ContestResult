class Solution {
    public int maximumRequests(int n, int[][] requests) {
        
        return maxTransfers(new int[n],requests,0,0);
        
    }
    
    public int maxTransfers(int[] dp,int[][] requests,int pos,int count){
        if(pos==requests.length){
            for(int i=0;i<dp.length;i++){
                if(dp[i]!=0){
                    return 0;
                }
            }
            
            return count;
        }
        dp[requests[pos][0]]--;
        dp[requests[pos][1]]++;
        int val1=maxTransfers(dp,requests,pos+1,count+1);
        dp[requests[pos][0]]++;
        dp[requests[pos][1]]--;
        int val2=maxTransfers(dp,requests,pos+1,count);
        
        return Math.max(val1,val2);
    }
}
