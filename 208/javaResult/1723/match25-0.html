class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int m = requests.length, ans = 0;
        for (int s = 1; s < (1 << m); s++) {
            int cnt = 0;
            int[] move = new int[n];
            for (int i = 0; i < m; i++) {
                if ((s & (1 << i)) != 0) {
                    move[requests[m - i - 1][0]]--;
                    move[requests[m - i - 1][1]]++;
                    cnt++;
                }
            }
            boolean valid = true;
            for(int i = 0; i < n; i++) {
                if(move[i] != 0) {
                    valid = false;
                    break;
                }
            }
            if(valid) ans = Math.max(cnt, ans);
        }
        return ans;
    }
}
