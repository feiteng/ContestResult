class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int out = 0;
        int count = 1 << requests.length;
        for (int i = 0; i < count; i++) {
            int[] a = new int[n];
            for (int j = 0; j < requests.length; j++) {
                
if (((1 << j) & i) != 0) {
                    a[requests[j][0]]--;
                    a[requests[j][1]]++;
                }
            }
            boolean valid = true;
            for (int j : a) {
                if (j != 0) {
                    valid = false;
                    break;
                
}
            }
            if (valid) {
                out = Math.max(out, Integer.bitCount(i));
            }
        }
        return out;
    }
}
