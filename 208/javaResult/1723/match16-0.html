class Solution {
    public int maximumRequests(int n, int[][] req) {
        int max = req.length,ans = 0;
        for(int i=(1<<max)-1;i>=0;i--){
            int[] state = new int[n];
            int c = 0;
            for(int j=0;j<max;j++){
                if(((i>>j)&1) == 1){
                    c++;
                    state[req[j][0]]++;
                    state[req[j][1]]--;
                }
            }
            boolean flag = true;
            for(int j=0;j<n;j++){
                if(state[j]!=0){
                    flag = false;
                    break;
                }
            }
            if(flag){
                ans = Math.max(ans,c);
            }
        }
        return ans;
    }
}
