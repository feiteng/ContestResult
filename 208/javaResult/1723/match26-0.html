class Solution {
    public int maximumRequests(int n, int[][] r) {
        int m = r.length;
        int ans = 0;
        for (int i=0; i<(1<<m); i++) {
            int t = 0;
            int[] a = new int[n];
            for (int j=0; j<m; j++)
                if (((i >> j) & 1) == 1) {
                    t++;
                    a[r[j][0]]--;
                    a[r[j][1]]++;
                }
            boolean b = true;
            for (int j=0; j<n; j++)
                if (a[j] != 0) {
                    b = false;
                    break;
                }
            if (b)
                ans = Math.max(ans, t);
        }
        return ans;
    }
}
