class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int res = 0;
        int total = (1<<(requests.length))-1;
        for(int i=1; i<=total; i++){
            int curr=i;
            int net[] = new int[n];
            for(int j=0; j<=requests.length; j++){
                int b=curr&1; curr>>=1;
                if(b==1){
                    net[requests[j][0]]--;
                    net[requests[j][1]]++;
                }
            }
            boolean flag=true;
            for(int x:net){
                if(x!=0) flag=false;
            }
            if(flag){
                res=Math.max(res, Integer.bitCount(i));
            }
        }
        return res;
    }
}
