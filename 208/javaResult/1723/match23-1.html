class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int ans = 0;
        int m = requests.length;
        search: for (int i = 0; i < (1 << m); i++) {
            int[] d = new int[n + 1];
            int t = 0;
            for (int j = 0; j < m; j++) {
                if ((i & (1 << j)) > 0) {
                    t++;
                    d[requests[j][0]]++;
                    d[requests[j][1]]--;
                }
            }
            for (int j = 0; j <= n; j++) {
                if (d[j] != 0) continue search
;
            }
            
            ans = Math.max(ans, t);
        }
        return ans;
    }
}
