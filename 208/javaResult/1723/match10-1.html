class Solution {
    public int maximumRequests(int n, int[][] requests) {
        int res = 0;
        int requestLen = requests.length;
        for(int i = 0; i < (1<<requestLen); ++i){
            int s = 0;
            for(int j = 0; j < requestLen; ++j){
                if((i & (1 << j)) != 0){
                    s = (s | (1<<j));
                }
            }
            
if(check(s, n, requests)){
                res = Math.max(res, Integer.bitCount(s));
            }
        }
        return res;
    }
    
    private boolean check(int s, int size, int[][] requests){
        int[] visited = new int[size];
        int requestLen = requests.length;
        for(int i = 0; i <  requestLen; ++i){
            if((s & (1 << i)) != 0){
                int src = requests[i][0];
                int dst = requests[i][1];
                visited[src]--;
                visited[dst]++;
            }
        }
        for(int i = 0; i < size; ++i){
            if(visited[i] != 0) return false;
        }
        return true;
    }
}
