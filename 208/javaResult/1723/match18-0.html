class Solution {
    
    int max = 0;
    int[][] requests;
    List<List<Integer>> list = new ArrayList<>();
    
    public int maximumRequests(int n, int[][] requests) {
        this.requests = requests;
        dfs(new ArrayList<>(), 0);
        
        int max = 0;
        for (var l : list) {
            int[] change = new int[n];
            for (var i : l) {
                --change[requests[i][0]];
                ++change[requests[i][1]];
            }
            boolean even = true;
            for (int num : change) {
                if (num != 0) {
                    even = false;
                    break;
                }
            }
            if (even) {
                max = Math.max(max, l.size());
            }
        }
        
        return max;
    }
    
    private void dfs(List<Integer> cur, int index) {
        //
        list.add(new ArrayList<>(cur));

        
        //backtracking
        for (int i = index; i < requests.length; ++i) {
            
            cur.add(i);
            dfs(cur, i + 1);
            cur.remove(cur.size() - 1);
            
            // int[] request = requests[index];
            // --change[request[0]];
            // ++change[request[1]];
            // dfs(change, used + 1, i + 1);
            // ++change[request[0]];
            // --change[request[1]];
        }
    }
}
