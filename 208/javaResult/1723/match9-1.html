class Solution {
    int res = 0;
    
    public int maximumRequests(int n, int[][] requests) {
        int[] count = new int[n];
        
        dfs(count, requests, 0, 0);
        
        return res;
    }
    
    private void dfs(int[] count, int[][] requests, int cur, int achieve){
        
if(cur == requests.length){
            if(check(count)){
                res = Math.max(res, achieve);
            }
            
            return;
        }
        
        
int start = requests[cur][0];
        int end = requests[cur][1];
        
        count[start]--;
        count[end]++;
        
        dfs(count, requests, cur + 1, achieve + 1);
        
        count[start]++;
        count[end]--;
        
        dfs(count, requests, cur + 1, achieve);
    }
    
    private boolean check(int[] count){
        for(int i : count){
            if(i != 0){
                return false;
            }
        }
        
        return true;
    }
}
