class Solution {
    List<List<Integer>> g;
    public int maximumRequests(int n, int[][] requests) {
        g = new ArrayList<>();
        
        
generate(requests.length, new ArrayList<>(), 0);
        int max = 0;
        //postprocessing check
        for (var l : g) {
            
            // runover all the request;
            int[] cur = new int[n];
            for (int index : l) {
                ++cur[requests[index][0]] ;
                --cur[requests[index][1]];
            }
            boolean possible = true;
            for (int i = 0; i < n; ++i) {
                if(cur[i] != 0) {
                    possible = false;
                    break;
                }
            }
            if (possible) {
                max = Math.max(max, l.size());
            }
        }
        return max;
        
    }
    private void generate(int total, List<Integer> cur, int index) 
{
        g.add(new ArrayList(cur));
        
        for (int i = index; i < total; ++i) {
            cur.add(i);
            generate(total, cur, i + 1);
            cur.remove(cur.size() - 1);
        }
    }
    
}
