class Solution {
    int[] counts;
    int ans = 0;
    public int maximumRequests(int n, int[][] requests) {
        ans = 0;
        
counts = new int[n];
        dfsMaximumRequests(requests, 0, 0);
        return ans;
    }

    private void dfsMaximumRequests(int[][] requests, int i, int avalibleCount) {
        if (i >= requests.length) {
            for (int item : counts) {
                if (item != 0) {
                    return;
                }
            }
            ans = Math.max(ans, avalibleCount);
            return;
        }
        
        //不接受第i个请求
        dfsMaximumRequests(requests, i + 1, avalibleCount);
        
        //accept i
        counts[requests[i][0]]--;
        counts[requests[i][1]]++;
        dfsMaximumRequests(requests, i + 1, avalibleCount + 1);
        //change back
        counts[requests[i][0]]++;
        counts[requests[i][1]]--;
    }
}
