class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int left = 0;
        int benefit = 0;
        int max = -1;
        int res = -1;
        int round = 1;
        for (int i = 0; i < customers.length; i++, round++) {
            left += customers[i];
            if (left >= 4) {
                left -= 4;
                benefit += boardingCost * 4 - runningCost;
            } else {
                benefit += boardingCost * left - runningCost;
                left = 0;
            }
            if (benefit > max) {
                res = round;
                max = benefit;
            }
        }
        while (left > 0) {
            if (left >= 4) {
                left -= 4;
                benefit += boardingCost * 4 - runningCost;
            } else {
                benefit += boardingCost * left - runningCost;
                left = 0;
            }
            if (benefit > max) {
                res = round;
                max = benefit;
            }
            round++;
        }
        return res;
    }

}
