class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int maxRotationsProfit = 0;
        int maxProfit = 0;
        int profit = 0;
        
        int N = customers.length;
        
        int customersRemaining = 0;
        for (int i = 0; i < N; i++) {
            customersRemaining += customers[i];
            
            
            int serve = Math.min(4, customersRemaining);
            customersRemaining = Math.max(0, customersRemaining - 4);
            
            profit += (boardingCost * serve) - runningCost;
            
if (profit > maxProfit) {
                maxProfit = profit;
                maxRotationsProfit = i+1;
            }
        }
        
        int i = N;
        while (customersRemaining > 0) {
            int serve = Math.min(4, customersRemaining);
            customersRemaining = 
Math.max(0, customersRemaining - 4);
            
            profit += (boardingCost * serve) - runningCost;
            if (profit > maxProfit) {
                maxProfit = profit;
                maxRotationsProfit = i+1;
            }
            i++;
        }
        
        if (maxRotationsProfit <= 0) {
            return -1;
        }
        return maxRotationsProfit;
    }
}
