class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int max =0;
        int cnt =0;
        int wait =0;
        int p =0;
        int ans =-1;
        for(int x : customers){
            cnt++;
            int take = Math.min(wait+x, 4);
            p +=(take*boardingCost- runningCost);
            if(p>max){
                max= p;
                ans=cnt;
            }
            wait=Math.max(wait+x-4,0);
           
        }
        while(wait>0){
            cnt++;
            int take = Math.min(wait, 4);
            p +=(take*boardingCost- runningCost);
            if(p>max){
                max= p;
                ans=cnt;
            }
            wait-=4;
        }
        return ans;
    }
}
