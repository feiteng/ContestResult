class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
    	int runningSum = 0;
    	int waitingCustomers = 0;
    	int rotationCount = 0;
    	int maxProfit = 0;
    	int rotationForMax = -1;
    	if (boardingCost * 4 < runningCost) {
    		return -1;
    	}
    	for (int c: customers) {
    		waitingCustomers += c;
    		int newBoarding = Math.min(4, waitingCustomers);
    		waitingCustomers -= newBoarding;
    		runningSum += newBoarding * boardingCost - runningCost;
    		rotationCount ++;
    		if (runningSum > maxProfit) {
    			maxProfit = runningSum;
    			rotationForMax = rotationCount;
    		}
    	}
    	
    	while (waitingCustomers > 0) {
    		int newBoarding = Math.min(4, waitingCustomers);
    		waitingCustomers -= newBoarding;
    		runningSum += newBoarding * boardingCost - runningCost;
    		rotationCount ++;
    		if (runningSum > maxProfit) {
    			maxProfit = runningSum;
    			rotationForMax = rotationCount;
    		}
    	}

        return rotationForMax;
    }
}
