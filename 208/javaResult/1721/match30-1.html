class Solution {
    public int minOperationsMaxProfit(int[] cus, int bC, int rC) {
        int n = cus.length;
        int max = 0;
        int maxi = -1;
        int res = 0;
        int waiting = 0;
        for(int i = 1; i <= n; i++) {
            waiting += cus[i - 1];
            int on = Math.min(waiting, 4);
            waiting -= on;
            res += on * bC - rC;
            if(res > max) {
                max = res;
                maxi = i;
            }
        }
        int time = n + 1;
        
while(waiting != 0)
        {
            int on = Math.min(waiting, 4);
            waiting -= on;
            res += on * bC - rC;
            if(res > max) {
                max = res;
                maxi = time;
            }
            time++;
        }    
        return maxi;
    }
}
