class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int count = 0;
        int max = -1;
        int min = -1;
        int profit = 0;
        int opt = 0;
        for (int customer : customers) {
            opt++;
            count += customer;
            int boarding = Math.min(count, 4);
            count -= boarding;
            profit += boarding*boardingCost;
            profit -= runningCost;
            if (profit > max) {
                max = profit;
                min = opt;
            }
        }
        while (count > 0) {
            opt++;
            int boarding = Math.min(count, 4);
            count -= boarding;
            profit += boarding*boardingCost;
            profit -= runningCost;
            if (profit > max) {
                max = profit;
                min = opt;
            }
        }
        return min;
    }
}
