class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int bestProfit = 0;
        int currProfit = 0;
        int bestRot = -1;
        int waiting = 0;
        int rot = 0;
        while (rot < customers.length || waiting > 0) {
            if (rot < customers.length) {
                waiting += customers[rot];
            }
            ++rot;
            int board = Math.min(4, waiting);
            currProfit += board * boardingCost - runningCost;
            if (currProfit > bestProfit) {
                bestProfit = currProfit;
                bestRot = rot;
            }
            waiting -= board;
        }
        return bestRot;
    }
}
