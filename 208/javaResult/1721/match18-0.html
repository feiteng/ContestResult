class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int cur = 0;
        int max = 0;
        int res = -1;
        int waiting = 0;
        int index = 0;
        while(waiting >0 || index < customers.length){
            if(index < customers.length){
                waiting += customers[index];
            }
            int onbard = Math.min(waiting, 4);
            cur += boardingCost*onbard - runningCost;
            waiting -= onbard;
            if(max < cur){
                res = index+1;
                max = cur;
            }
            index++;
        }
        // for(int i=0; i<customers.length; i++){
        //     waiting += customers[i];
        //     int onbard = Math.min(waiting, 4);
        //     cur += boardingCost*onbard - runningCost;
        //     waiting -= onbard;
        //     if(max < cur){
        //         res = i;
        //         max = cur;
        //     }
        // }
        return res;
    }
}
