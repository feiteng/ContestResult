class Solution {
    public int minOperationsMaxProfit(int[] a, int b, int r) {
        int ans = -1;
        int ma = 0;
        int t = 0;
        int w = 0;
        int n = a.length;
        for (int i=0; i<n; i++) {
            w += a[i];
            int z = Math.min(4, w);
            w -= z;
            t += z * b - r;
            if (t > ma) {
                ma = t;
                ans = i+1;
            }
        }
        int i = n+1;
        while (w > 0) {
            int z = Math.min(4, w);
            w -= z;
            t += z * b - r;
            if (t > ma) {
                ma = t;
                ans = i;
            }
            i++;
        }
        return ans > 0 ? ans : -1;
    }
}
