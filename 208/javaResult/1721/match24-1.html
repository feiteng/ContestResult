class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = 0, ans = 0, count = 1;
        int left = 0, profit = 0;
        for(int i = 0; i < customers.length; i++){
            left+=customers[i];
            if(left > 4){
                profit+=(boardingCost*4)-runningCost;
                left-=4;
            }else{
                profit+=(boardingCost*left)-runningCost;
                left = 0;
            }
            if(profit > res) {
                ans = count;
                res = profit;
            }
            count++;   
        }
        while(left > 0){
            if(left > 4){
                profit+=(boardingCost*4)-runningCost;
                left-=4;
            }else{
                profit+=(boardingCost*left)-runningCost;
                left = 0;
            }
            if(profit > res) {
                ans = count;
                res = profit;
            }
            count++;
        }
        return res > 0? ans:-1;
    }
}
