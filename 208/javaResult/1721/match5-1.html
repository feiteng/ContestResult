class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {

        int carry = 0;
        int profit = 0, maxProfit = 0;
        int ans = -1;


        int p = 0;
        for (int c : customers) {
            p++;
            carry += c;
            int board = Math.min(carry, 4);
            carry -= board;
            profit += board * boardingCost - runningCost;
            if (maxProfit < profit) {
                maxProfit = profit;
                ans = p;
            }
        }

        while (carry > 0) {
            p++;
            int board = Math.min(carry, 4);
            carry -= board;
            profit += board * boardingCost - runningCost;
            if (maxProfit < profit) {
                maxProfit = profit;
                ans = p;
            }
        }

        return ans;


    }
}
