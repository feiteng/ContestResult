class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = -2;
        int revenue = 0;
        int max = 0;
        int totalCnt = 0;
        for (int i = 0; i < customers.length || (totalCnt * boardingCost > runningCost); i++) {
            if (i < customers.length) {
                totalCnt += customers[i];
            }
            // System.out.println(totalCnt);
            int onboard = Math.min(totalCnt, 4);
            totalCnt -= onboard;
            revenue = revenue - runningCost + onboard * boardingCost;
            if (revenue > max) {
                max = revenue;
                res = i;
            }
        }
        
        return res + 1;
    }
}
