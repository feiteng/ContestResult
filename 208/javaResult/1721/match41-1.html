class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
       int n = customers.length;
    int profit = 0;
    int wait = 0;
    int i = 0;
    int maxr = 0;
    int maxp = 0;
    int rotate = 0;
    while (i < n || wait > 0) {
      if (i < n) {
        wait += customers[i++];
      }
      int m = Math.min(wait, 4);
      profit += m * boardingCost - runningCost;
      wait -= m;
      rotate++;
      
if (profit > maxp) {
        maxp = profit;
        maxr = rotate;
      }
    }
    return maxp > 0 ? maxr : -1; 
    }
}
