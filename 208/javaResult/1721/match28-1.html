class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int total = 0;
        int profit = 0;
        int remain = 0;
        int max = 0;
        int index = -1;
        int count = 0;
        for (int i : customers) 
{
            count++;
            remain += i;
            
            if (remain >= 4) {
                profit += (boardingCost * 4) - runningCost;
                remain -= 4;
            } else {
                profit += (boardingCost * remain) - runningCost;
                remain = 0;
            }
            if (max < profit) {
                max = profit;
                index = count;
            }
        }
        
        while (remain > 0) {
            count++;
            if (remain >= 4) {
                profit += (boardingCost * 4) - runningCost;
                remain -= 4;
            } else {
                profit += (boardingCost * remain) - runningCost;
                remain = 0;
            }
            if (max < profit) {
                max = profit;
                index = count;
            }
        }
        return index;
    }
}
