class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int cur = 0;
        int max = 0;
        int res = 0;
        int times = 0;
        int waiting = 0;
        for(int i=0; i<customers.length; i++){
            waiting += customers[i];
            cur -= runningCost;
            times++;
            if(waiting>=4){
                cur += 4*boardingCost;
                waiting -=4;
            }
            else{
                cur += waiting*boardingCost;
                waiting =0;                
            }
            if(cur > max){
                res = times;
                max = cur;
            }
        }
        while(waiting>0){
            cur -= runningCost;
            times++;
            if(waiting>=4){
                cur += 4*boardingCost;
                waiting -=4;
            }
            else{
                cur += waiting*boardingCost;
                waiting =0;                
            }
            if(cur > max){
                res = times;
                max = cur;
            }           
        }
        return res==0 ? -1 : res;
    }
}
