class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int money = 0;
        int maxMoney = 0;
        
        int res = -1;
        int wait = 0;
        int count = 0;
        
        
        for(int i : customers){
            count++;
            wait += i;
            if(wait >= 4){
                money += 4 * boardingCost;
                wait -= 4;
            }
            else{
                money += wait * boardingCost;
                wait = 0;
            }
            
            money -= runningCost;
            
            if(money > maxMoney){
                maxMoney = money;
                res = count;
            }
        }
        
        while(wait > 0){
            count++;
            if(wait >= 4){
                money += 4 * boardingCost;
                wait -= 4;
            }
            else{
                money += wait * boardingCost;
                wait = 0;
            }
            
            money -= runningCost;
            
            if(money > maxMoney){
                maxMoney = money;
                res = count;
            }
        }
        
        return res;
    }
}
