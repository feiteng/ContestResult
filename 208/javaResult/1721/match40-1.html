class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int cur = 0;
        int max = -1;
        int ans = -1;
        int cnt = 0;
        int w = 0;
        for (int c: customers) {
            w += c;
            int n = Math.min(4, w);
            w -= n;
            cur += n * boardingCost - runningCost;
            cnt++;
            if (cur > max) {
                max = cur;
                ans = cnt;
            }
        }
        
while (w > 0) {
            int n = Math.min(4, w);
            w -= n;
            cur += n * boardingCost - runningCost;
            cnt++;
            if (cur > max) {
                max = cur;
                ans = cnt;
            }
        }
        return ans;
    }
}
