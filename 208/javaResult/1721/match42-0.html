class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int t = 0, remain = 0;
        int max = 0, now = 0, tmax = -1;
        for (int c : customers) {
            remain += c;
            now = now - runningCost + Math.min(remain, 4) * boardingCost ;
            remain -= Math.min(remain, 4);
            t++;
            if (now > max) {
                max = now;
                tmax = t;
            }
        }
        
        while (remain > 0) {
            now = now - runningCost + Math.min(remain, 4) * boardingCost ;
            remain -= Math.min(remain, 4);
            t++;
            if (now > max) {
                max = now;
                tmax = t;
            }
        }
        return tmax;
    }
}
