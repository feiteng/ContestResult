class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int ans = -1;
        int max = 0;
        
        int cur = 0;
        int profit = 0;
        
        for(int i = 0; i < customers.length || cur > 0; i++){
            if(i < customers.length) cur += customers[i];
            int next = Math.min(cur, 4);
            
            profit += next * boardingCost;
            profit -= runningCost;
            cur -= next;
            
            if(profit > max){
                max = profit;
                ans = i + 1;
            }
            
        }
        return ans;
    }
}
