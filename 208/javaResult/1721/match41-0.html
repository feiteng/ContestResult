class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int sum = 0;
        int wait = 0;
        int cur = 0;
        int day = 1;
        int count = 0;
        int max = Integer.MIN_VALUE;
        int res = 0;
//        while (customers[cur] == 0) cur++;
        while (cur != customers.length || wait != 0) {
            if (cur != customers.length) {
                wait += customers[cur++];
            }
            int numP = Math.min(wait, 4);
            wait -= numP;
            sum += numP;
            count = sum * boardingCost;
            count -= day * runningCost;

            // System.out.println("day = " + day + " count = " + count);

            if (count > max) {
                max = count;
                res = day;
            }

            day++;
        }

        return count > 0 ? res : -1;
    }


}
