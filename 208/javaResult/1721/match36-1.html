class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int maxProfit = Integer.MIN_VALUE;
        int maxCnt = 0;
        int cnt = 1;
        int customer = 0;
        int profit = 0;
        for(int i = 0; i < customers.length; i++){
            customer += customers[i];
            int bc = Math.min(4, customer);
            customer -= bc;
            profit += bc * boardingCost - runningCost;
            if(profit > maxProfit){
                maxProfit = profit;
                maxCnt = cnt;
            }
            cnt++;
        }
        
while(customer > 0){
           int bc = Math.min(4, customer);
            customer -= bc;
            profit += bc * boardingCost - runningCost; 
            if(profit > maxProfit){
                maxProfit = profit;
                maxCnt = cnt;
            }
            cnt++;
        }
        return maxProfit < 0 ? -1 : maxCnt;
    }
}
