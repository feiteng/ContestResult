class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int profit = Integer.MIN_VALUE;
        int curProfit = 0;
        int waiting = 0;
        int rot = 0;
        int maxRot = -1;
        
        for (int i = 0; i < customers.length; i++) {
            waiting += customers[i];
            
            int boarding = Math.min(4, waiting);
            waiting -= boarding;
            
            curProfit += boarding * boardingCost - runningCost;
            rot++;
            
            //System.out.printf("CurProfit = %d\n", curProfit);
            if (curProfit > profit) {
                profit = curProfit;
                maxRot = rot;
            }
        }
        
        
        while (waiting > 0) {
            int boarding = Math.min(4, waiting);
            waiting -= boarding;
            
            curProfit += boarding * boardingCost - runningCost;
            rot++;
            //System.out.printf("CurProfit = %d\n", curProfit);
            if (curProfit > profit) {
                profit = curProfit;
                maxRot = rot;
            }
        }
        
        return profit >= 0 ? maxRot : -1;
    }
}
