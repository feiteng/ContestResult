class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int max = 0;
        int currCost = 0;
        int board = 0;
        int len = customers.length;
        int jwb = -1;
        for (int i = 0; i < len; i++) {
            board += customers[i];
            int val = Math.min(board, 4);
            currCost += val * boardingCost - runningCost;
            if (currCost > max) {
                max = currCost;
                jwb = i;
            }
            board -= val;
        }
        int idx = len;
        while (board > 0) {
            int val = Math.min(board, 4);
            currCost += val * boardingCost - runningCost;
            board -= val;
            if (currCost > max) {
                max = currCost;
                jwb = idx;
            }
            idx++;
        }

        if (max == 0) {
            return -1;
        }
        return jwb + 1;
    }
}
