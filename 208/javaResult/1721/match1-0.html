class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = -1;
        int max_profit = -1;
        int time = 0;
        int profit = 0;
        int waiting = 0;
        for (int c: customers) {
            waiting += c;
            int cur = Math.min(waiting, 4);
            waiting -= cur;
            profit += cur * boardingCost - runningCost;
            time += 1;
            if (profit > max_profit) {
                max_profit = profit;
                res = time;
            }
        }
        while (waiting > 0) {
            int cur = Math.min(waiting, 4);
            waiting -= cur;
            profit += cur * boardingCost - runningCost;
            time += 1;
            if (profit > max_profit) {
                max_profit = profit;
                res = time;
            }
        }
        return max_profit>0?res:-1;
    }
}
