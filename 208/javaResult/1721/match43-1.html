class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = -1, Mx = 0, sum = 0;
        int waiters = 0;
        for(int i = 0;i < customers.length || waiters > 0;i++) {
            if(i < customers.length) {
                waiters += customers[i];
            }
            int cur = Math.min(4,waiters) * boardingCost - runningCost;
            sum += cur;
            waiters = Math.max(0,waiters-4);
            
if(sum > Mx) {
                Mx = sum;   res = i+1;
            }
        }

        return res;
    }
}
