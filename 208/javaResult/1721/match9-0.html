class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int waitingCustomer = 0, profit = 0, maxProfit = -1, ans = -1, rotation = 0;
        //get all customer
        for(int customer: customers) {
            rotation++;
            waitingCustomer += customer;
            int boardingCustomer = Math.min(waitingCustomer, 4);
            waitingCustomer -= boardingCustomer;
            profit += (boardingCustomer*boardingCost);
            profit -= runningCost;
            if(profit > maxProfit){
                maxProfit = profit; 
                ans = rotation;
            }        
        }
            
        //boarding rest customer
        while(waitingCustomer > 0){
            rotation++;
            int boardingCustomer = Math.min(waitingCustomer, 4);
            waitingCustomer -= boardingCustomer;
            profit += (boardingCustomer*boardingCost);
            profit -= runningCost;
             if(profit > maxProfit){
                maxProfit = profit; 
                ans = rotation;
            } 
        }
        
        return ans;
    }
}
