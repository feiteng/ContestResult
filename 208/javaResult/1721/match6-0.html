class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int maxProfit = 0;
        int out = -1;
        
        int profit = 0;
        int waiting = 0;
        
        int turn = 0;
        
        for(int cust : customers) {
            waiting += cust;
            
            int board = Math.min(4, waiting);
            profit += board * boardingCost - runningCost;
            waiting -= board;
            turn++;
            
            if(profit > maxProfit) {
                maxProfit = profit;
                out = turn;
            }
        }
        
        while(waiting > 0) {
            int board = Math.min(4, waiting);
            profit += board * boardingCost - runningCost;
            waiting -= board;
            turn++;
            
            if(profit > maxProfit) {
                maxProfit = profit;
                out = turn;
            }
        }
        
        return out;
    }
}
