class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = -1, profit = 0, people = 0, max = -1, t = 0;
        for(int i: customers) {
            t++;
            people += i;
            profit += boardingCost * 
Math.min(4, people);
            people -= Math.min(4, people);
            profit -= runningCost;
            if(profit > max) {
                max = profit;
                res = t;
            }
        }
        while(people > 0) {
            t++;
            profit += boardingCost * 
Math.min(4, people);
            people -= Math.min(4, people);
            profit -= runningCost;
            if(profit > max) {
                max = profit;
                res = t;
            }
        }
        return res;
    }
}
