class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int res = -1;
        int maxProfit = 0;
        int waiting = 0;
        int processed = 0;
        int i = 0;
        for (; i < customers.length; i++) 
{
            waiting += customers[i];
            if (waiting < 4) {
                processed += waiting;
                waiting = 0;
            } else {
                processed += 4;
                waiting -= 4;
            }
            if (maxProfit < processed * boardingCost - runningCost * (i + 1)) {
                maxProfit = processed * boardingCost - runningCost * (i + 1);
                res = i + 1;
            }
        }
        while (waiting > 0) {
            if (waiting < 4) {
                processed += waiting;
                waiting = 0;
            } else {
                processed += 4;
                waiting -= 4;
            }
            if (maxProfit < processed * boardingCost - runningCost * (i + 1)) {
                maxProfit = processed * boardingCost - runningCost * (i + 1);
                res = i + 1;
            }
            i++;
        }
        
        return res;
    }

}
