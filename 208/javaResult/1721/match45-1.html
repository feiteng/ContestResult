class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int profit = 0;
        int wait = 0;
        int boarded = 0;
        int run = 0;
        int best = 0;
        int bestRun = -1;
        
        
        for(int i:customers)
{
            wait+=i;
            if(wait>4){
                wait-=4;
                boarded+=4;
                
            } else {
                boarded+=wait;
                wait=0;
            }
            run++;
            profit = (boarded*boardingCost-runningCost*run);
            
if(profit>best){
                best = profit;
                bestRun = run;
            }   
        }
        while(wait>0){
            if(wait>4){
                wait-=4;
                boarded+=4;
                
            } else {
                boarded+=wait;
                wait=0;
            }
            run++;
            profit = (boarded*boardingCost-runningCost*run);
            
if(profit>best){
                best = profit;
                bestRun = run;
            }
        }
     
        return bestRun;
        
    }
}
