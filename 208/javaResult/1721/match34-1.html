class Solution {
    public int minOperationsMaxProfit(int[] customers, int boardingCost, int runningCost) {
        int max = 0;
        int bestRotation = -1;
        int waiting = 0;
        int current = 0;
        
        int i = 0;
        for (; i < customers.length; i++) 
{
            waiting += customers[i];
            
            if (waiting >= 4) {
                current += boardingCost * 4;
                waiting -= 4;
            } else {
                current += boardingCost * waiting;
                waiting = 0;
            }
            
            current -= runningCost;
            if (current > max) {
                max = current;
                bestRotation = i + 1;
            }
        }
        
        while (waiting > 0) {
            if (waiting >= 4) {
                current += boardingCost * 4;
                waiting -= 4;
            } else {
                current += boardingCost * waiting;
                waiting = 0;
            }
            
            current -= runningCost;
            if (current > max) {
                max = current;
                bestRotation = i + 1;
            }
            
            i++;
        }
        
        return bestRotation;
    }
}
