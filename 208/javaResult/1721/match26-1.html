class Solution {
    public int minOperationsMaxProfit(int[] cus, int b, int r) {
        int p = 0, n = cus.length, cur = 0, res = -1, t = 0;
        for (int i = 0; i < n; i++) {
            p += cus[i];
            int on = Math.min(p, 4);
            p -= on;
            cur += b * on - r;
            if (cur > res) {
                res = cur;
                t = i + 1;
                //System.out.println(res + "," + "t");
            }
        }
        int i = n;
        while (p > 0) {
            int on = Math.min(p, 4);
            p -= on;
            cur += b * on - r;
            if (cur > res) {
                res = cur;
                t = i + 1;
                //System.out.println(res + "," + "t");
            }
            i++;
        }
        return res == -1 ? -1 : t;
    }
}
