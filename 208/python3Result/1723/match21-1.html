class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        
def valid(tri):
            counts = [0 for _ in range(n)]
            for req in tri:
                counts[req[0]] -= 1
                counts[req[1]] += 1
            return not any(counts)
        
                
        
        best = 0
        for i in range(2**len(requests)):
            
            bits = "{0:b}".format(i)
            tri = []
            diff = len(requests) - len(bits)
            
for k in range(len(bits)):
                if int(bits[k]):
                    tri.append(requests[k+diff])
            if len(tri) > best:
                if valid(tri):
                    best = len(tri)
        return best
