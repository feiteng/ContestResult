class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        # n buildings
        net = [0]*n
        l = len(requests)
        
        def dfs(i):
            if i==l:
                
return 0 if all(z==0 for z in net) else -float('inf')
            
            res = dfs(i+1)
            s,e = requests[i]
            net[s]+=1
            net[e]-=1
            res = max(res,dfs(i+1)+1)
            net[s]-=1
            net[e]+=1
            return res
        
        return dfs(0)
