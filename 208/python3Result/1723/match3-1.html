class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        # dfs
        diff = [0] * n
        self.sol = 0
        self.dfs(diff, requests, 0,  0)
        return self.sol
    
    def dfs(self, diff, requests, start, count):
        if start == len(requests):
            if diff == [0] *  len(diff):
                self
.sol = max(self.sol, count)
            return 
        # two options
        f, t = requests[start]
        self.dfs(diff, requests,start +1, count)
        diff[f] -= 1
        diff[t] += 1
        self.dfs(diff, requests, start+1, count + 1)
        diff[f] += 1
        diff[t] -= 1
