class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        
def find(cur: int, req: int):
            if cur == len(requests):
                if all(not a for a in net):
                    self.ans = max(self.ans, req)
                return
            
            net
[requests[cur][0]] -= 1
            net[requests[cur][1]] += 1
            find(cur + 1, req + 1)
            net[requests[cur][0]] += 1
            net[requests[cur][1]] -= 1
            if requests[cur][0] != requests[cur][1]:
                find(cur + 1, req)
        
        self.ans = 0
        net = [0] * n
        find(0, 0)
        return self.ans
