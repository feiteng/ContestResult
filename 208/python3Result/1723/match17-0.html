class Solution:
    def maximumRequests(self, n: int, req: List[List[int]]) -> int:
        cur = [0] * n
        self.res = 0
        def dfs(i, has):
            if i == len(req):
                if all(c == 0 for c in cur):
                    self.res = max(self.res, has)
                return
            dfs(i + 1, has)
            cur[req[i][0]] -= 1
            cur[req[i][1]] += 1
            dfs(i + 1, has + 1)
            cur[req[i][0]] += 1
            cur[req[i][1]] -= 1
        
        dfs(0, 0)
        
        return self.res
