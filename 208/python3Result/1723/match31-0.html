class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        for req_num in range(len(requests), -1, -1):
            for req_id in combinations(requests, req_num):
                cur_req = defaultdict(int)
                for x, y in req_id:
                    cur_req[x] -= 1
                    cur_req[y] += 1
            
                if not any(cur_req.values()):
                    return req_num
                
        return 0
