class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        
        def dfs(k, arr):
            if k == len(requests):
                if all((x == 0 for x in arr)):
                    return 0
                return -(1 << 30)

            x, y = requests[k]
            t0 = dfs(k + 1, arr)
            arr[x] -= 1
            arr[y] += 1
            t1 = dfs(k + 1, arr) + 1
            arr[x] += 1
            arr[y] -= 1
            return max(t0, t1)

        arr = [0] * n
        res = dfs(0, arr)
        return res
