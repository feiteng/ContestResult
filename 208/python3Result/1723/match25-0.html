class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        for chooseCnt in range(len(requests),0,-1):
            comb=combinations(requests,chooseCnt)
            for choosen in comb:
                cnt=[0]*n
                for a,b in choosen:
                    cnt[a]+=1
                    cnt[b]-=1
                if all(v==0 for v in cnt):
                    return chooseCnt
        return 0
