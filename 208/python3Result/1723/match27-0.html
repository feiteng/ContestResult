class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        # n<=20 应该是用状态压缩DP
        # 每个请求是否能完成，标记为2进制中的一位
        # 判断每个状态下 每个请求完成时，入度和出度一致即可
        m = len(requests)
        res = 0
        for i in range(2**m):
            temp = []
            for x in range(m):
                if 1<<x &i:
                    temp.append(requests[x])
            degree = [0]*n
            for u,v in temp:
                degree[u]+=1
                degree[v]-=1
            if all([x==0 for x in degree]):
                res=max(res,len(temp))
        return res
