from itertools import combinations
class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        for k in range(len(requests),-1,-1):
            for a in combinations(requests,k):
                b=[0]*n
                for i,j in a:
                    b[i]+=1
                    b[j]-=1
                if all(not x for x in b): return k
