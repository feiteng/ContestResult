class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        d = [0]*n
        self.ans = 0
        nr = len(requests)
        def dfs(cur,s):
            if cur==nr:
                if all(x==0 for x in d):
                    self.ans = max(self.ans,s)
                return
            dfs(cur+1,s)
            x,y = requests[cur]
            d[x]-=1
            d[y]+=1
            dfs(cur+1,s+1)
            d[x]+=1
            d[y]-=1
        dfs(0,0)
        return self.ans
