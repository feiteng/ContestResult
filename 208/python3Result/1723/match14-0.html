class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        ans = 0
        r = len(requests)
        for bm in range(1<<r):
            cnts = [0]*n
            cand = 0
            for i in range(r):
                if (bm >> i) & 1:
                    cand += 1
                    f, t = requests[i]
                    cnts[f] -= 1
                    cnts[t] += 1
            if all(v == 0 for v in cnts):
                ans = max(ans, cand)
        return ans
