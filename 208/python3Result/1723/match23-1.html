class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        ans = 0
        count = len(requests)
        
for c in reversed(range(1, count + 1)):
            for req in itertools.combinations(requests, c):
                tally = [0] * n
                for f, t in req:
                    tally[f] -= 1
                    tally[t] += 1
                if not any(tally):
                    return c
        return ans
