class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        best = 0
        def valid(reqs):
            buildings = [0 for _ in range(n)]
            for req in reqs:
                buildings[req[0]] -= 1
                buildings[req[1]] += 1
            return not any(buildings)
        
        for i in range(2**len(requests)):
            bin_num = "{0:b}".format(i)
            diff = len(requests) - len(bin_num)
            reqs = []
            for k in range(len(bin_num)):
                if int(bin_num[k]):
                    reqs.append(requests[k+diff])
            if len(reqs) > best:
                if valid(reqs):
                    best = len(reqs)
        
        return best
