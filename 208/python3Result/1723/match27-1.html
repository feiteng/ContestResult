class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        best = 0
        for mask in range(1 << len(requests)):
            req = []
            for i, r in enumerate(requests):
                if mask & (1 << i):
                    req.append(
r)
            deltas = [0] * n
            for x, y in req:
                deltas[x] -= 1
                deltas[y] += 1
            if all(x == 0 for x in deltas):
                best = max(best, len(req))
        return best
