class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        m = len(requests)
        res = 0
        for mask in range((1 << m) - 1, -1, -1):
            states = [0] * n
            cur = 0
            for i in range(m):
                if (1 << i) & mask != 0:
                    cur += 1
                    x, y = requests[i]
                    states[x] += 1
                    states[y] -= 1
            if all([v == 0 for v in states]):
                res = max(res, cur)
        return res
