class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        m = len(requests)
        ans = 0
        for i in range(1 << m):
            tmp = [0]*n
            cnt = 0
            for k in range(m):
                if (i & (1 << k )) != 0:
                    cnt += 1
                    a, b = requests[k]
                    tmp[a] += 1
                    tmp[b] -= 1
            if tmp == [0] * n:        
                ans = max(ans, cnt)
                
        return ans
