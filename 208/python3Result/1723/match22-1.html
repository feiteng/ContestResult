class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        result = 0
        for l in itertools.product([True, False], repeat=len(requests)):
            counter = collections.defaultdict
(int)
            for i, (f, t) in zip(l, requests):
                if i:
                    counter[f] -= 1
                    counter[t] += 1
            if all(i == 0 for i in counter.values()):
                result = max(result, l.count(True))
        return result
