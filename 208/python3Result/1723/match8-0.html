class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        maxres = 0
        for h in range(2 ** len(requests) - 1, 0, -1):
            r = [0] * (n + 1)
            cnt = 0
            for i, (s, e) in enumerate(requests):
                if (1 << i) & h:
                    r[s] -= 1
                    r[e] += 1
                    cnt += 1
            # print(cnt, r)
            if not any(r):
                maxres = max(maxres, cnt)
        return maxres
