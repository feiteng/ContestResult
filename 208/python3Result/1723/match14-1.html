class Solution:
  def maximumRequests(self, n: int, req: List[List[int]]) -> int:
    sz = len(req)
    ret = 0
    
for plan in range(1, 2 ** sz): # from 1 to 2 ^ sz - 1
      arr = [0] * n
      num = 0
      for i in range(sz):
        if (1 << i) & plan > 0:
          num += 1
          x, y = req[i]
          arr[x] -= 1
          arr[y] += 1
      #print("{0:b}".format(plan), arr, num)
      if min(arr) == max(arr) == 0:
        ret = max(num, ret)
        
    return ret
