class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        def dfs(i, net = [0] * n):
            if i == len(requests):
                return 0 if all(n == 0 for n in net) else -float('inf')
            c = dfs(i + 1, net)
            fr, to = requests[i]
            net[fr] -= 1
            net[to] += 1
            c = max(c, 1 + dfs(i + 1, net))
            net[fr] += 1
            net[to] -= 1            
            return c
        
        return dfs(0)
