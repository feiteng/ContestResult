class Solution:
    def maximumRequests(self, n: int, reqs: List[List[int]]) -> int:
        ans = 0
        for mask in range(1 << len(reqs)):
            cnt = 0
            deg = [0] * n
            for i, (u, v) in enumerate(reqs):
                if (1 << i) & mask:
                    cnt += 1
                    deg[u] += 1
                    deg[v] 
-= 1
            if all(x == 0 for x in deg):
                ans = max(ans, cnt)
        return ans
