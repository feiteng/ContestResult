class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        ans 
= 0
        m = len(requests)
        for k in range(1 << m):
            calc = [0] * n
            for i in range(m):
                if (1 << i) & k > 0:
                    calc[requests[i][0]] -= 1
                    calc[requests[i][1]] += 1
            if all(x == 0 for x in calc):
                v = k
                step = 0
                while v > 0:
                    step += 1
                    v = v & (v - 1)
                ans = max(ans, step)
        return ans
