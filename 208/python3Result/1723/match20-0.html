class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        def count1(i: int) -> int:
            ret = 0
            while i:
                i &= (i-1)
                ret += 1
            return ret
            
        res = 0
        m = len(requests)
        for i in range((1 << m) - 1, 0, -1):
            count = [0] * n
            for j in range(m):
                if (1 << j) & i:
                    count[requests[j][0]] += 1
                    count[requests[j][1]] -= 1
            if max(count) == 0:
                res = max(res, count1(i))
        return res
