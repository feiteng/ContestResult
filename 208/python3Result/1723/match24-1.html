class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        for i in range(len(requests), -1, -1):
            for t in itertools.combinations
(requests, i):
                vector = [0]*n
                for x, y in t:
                    vector[x] += 1
                    vector[y] -= 1
                if not any(vector): return i
