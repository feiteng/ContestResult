class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        
        b = collections.defaultdict(int)
        
        for x,y in requests:
            if x != y:
                b[x] -= 1
                b[y] += 1
        
        ans = len(requests)
        
        #print(b.items())
        
        if all([i == 0 for i in b.values()]):
            return ans
        
        for i in range(1, len(requests)):
            comb = list(combinations(requests,i))
            for k in comb:
                copy = b.copy
()
                for m,n in k:
                    copy[m] += 1
                    copy[n] -= 1
                if all([i == 0 for i in copy.values()]):
                    return ans - i
        
        return 0
