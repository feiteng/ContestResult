class Solution:
    def maximumRequests(self, n, requests) -> int:
        requests_num 
= len(requests)
        swap = [0 for _ in range(n)]
        vis = [0 for _ in range(requests_num)]
        total = 0

        def update_total():
            flag = True
            for s in swap:
                if s != 0:
                    flag = False
            if flag:
                count = 0
                for v in vis:
                    if v == 1:
                        count += 1
                nonlocal total
                if count > total:
                    total = count

        def go(index):
            if index == requests_num:
                return
            else:
                request = requests[index]

                p_from = request[0]
                p_to = request[1]
                swap[p_from] += 1
                swap[p_to] -= 1
                vis[index] = 1

                update_total()
                go(index + 1)

                swap[p_from] -= 1
                swap[p_to] += 1
                vis[index] = 0
                go(index + 1)
                update_total()

        go(0)

        return total
