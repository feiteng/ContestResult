class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        build = [0 for _ in range(n)]
        
        ans = 0
        def dfs(cur, l):
            nonlocal ans
            if cur == len(requests):
                satis = True
                for i in build:
                    if i!=0:
                        satis = False
                        break
                if satis: ans = max(ans, l)
                return
            
            i, o = requests[cur]
            build[i] += 1
            build[o] -= 1
            
            dfs(cur+1, l+1)
            build[i] -= 1
            build[o] += 1
            dfs(cur+1, l)
        
        dfs(0, 0)
        return ans
