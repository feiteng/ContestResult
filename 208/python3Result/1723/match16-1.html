class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        num = [0] * n
        m = len(requests)
        r =  requests
        self.ans = 0
        
        def dfs(i,num,cnt):
            if i==m:
                if max(num)==0 and min
(num)==0:
                    self.ans = max(self.ans,cnt)
                return
            dfs(i+1,num,cnt)
            num[r[i][0]]-=1
            num[r[i][1]]+=1
            dfs(i+1,num,cnt+1)
            num[r[i][0]]+=1
            num[r[i][1]]-=1
        dfs(0,num,0)
        return self.ans
