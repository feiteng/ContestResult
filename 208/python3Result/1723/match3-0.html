class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        count = [0] * n
        self.ans = 0
        self.dfs(0, requests, count, 0)
        return self.ans
    
    def dfs(self, idx, requests, count, change):
        if idx == len(requests):
            found = True
            for c in count:
                if c != 0:
                    found = False
            if found:
                self.ans = max(self.ans, change)
            return
        
        u, v = requests[idx]
        self.dfs(idx + 1, requests, count, change)
        count[u] -= 1
        count[v] += 1
        self.dfs(idx + 1, requests, count, change + 1)
        count[u] += 1
        count[v] -= 1
