class Solution:
    def maximumRequests(self, N: int, requests: List[List[int]]) -> int:
        
        M = len(requests)
        ret = 0
        for i in range(1 << M):
            changes = [0] * N
            request_fulfilled = 0
            for j in range(M):
                if i & (1 << j):
                    u, v = requests[j]
                    changes[u] -= 1
                    changes[v] += 1
                    request_fulfilled += 1
            
            if all([x == 0 for x in changes]):
                ret = max(ret, request_fulfilled)
                
        return ret
