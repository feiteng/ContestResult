class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        G = [0] * n
        
        ans = 0
        def all_zero(arr):
            for x in arr:
                if x != 0:
                    return False
            return True
            
        def dfs(d,cnt):
            nonlocal ans
            if d == len(requests):
                if all_zero(G):
                    ans 
= max(ans,cnt)
                return
            f,t = requests[d]
            G[f] -= 1
            G[t] += 1
            dfs(d+1,cnt+1)
            G[f] += 1
            G[t] -= 1
            dfs(d+1,cnt)
        
        dfs(0,0)
        return ans
