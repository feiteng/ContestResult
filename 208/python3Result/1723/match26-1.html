class Solution:
    def maximumRequests(self, n: int, r: List[List[int]]) -> int:
        l=len(r)
        r=[[u,v] for u,v in r if u!=v]
        ans=l-len(r)
        l=len(r)
        self.ans=0
        #print(l)
        
def dfs(i,cur,res):
            if i==l:
                if all(not i for i in cur):
                    self.ans=max(self.ans,res)
                return 
            dfs(i+1,cur,res)
            s,e=r[i]
            cur[s]+=1
            cur[e]-=1
            dfs(i+1,cur,res+1)
            cur[s]-=1
            cur[e]+=1
        cur=[0]*n
        dfs(0,cur,0)
        return self.ans+ans
