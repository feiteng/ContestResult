class Solution:
    def maximumRequests(self, n: int, R: List[List[int]]) -> int:
        mx = 0
        r = len(R)
        for i in range(1<<r):
            cnt = [0] * 20
            for j in range(r):
                if (i>>j) & 1 == 1:
                    cnt[R[j][0]]+=1
                    cnt[R[j][1]]-=1
            if max(cnt) == 0:
                mx = max(mx, bin(i).count('1'))
        return mx
