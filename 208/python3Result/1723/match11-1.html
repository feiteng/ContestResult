class Solution:
    def maximumRequests(self, N: int, requests: List[List[int]]) -> int:
        R = len(requests)
        
        best = 0
        for mask in range(1 << R):
            buildings = [0] * N
            
            count = 0
            for x, (f, t) in enumerate(requests):
                if ((1 << x) & mask) > 0:
                    buildings[f] -= 1
                    buildings[t] += 1
                    count 
+= 1
                    
            if all(x == 0 for x in buildings):
                best = max(best, count)
        return best
