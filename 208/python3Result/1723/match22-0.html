class Solution:
    def maximumRequests(self, n: int, requests: List[List[int]]) -> int:
        maxres = 0
        for comb in itertools.product([0,1], repeat=len(requests)):
            state = [0 for _ in range(20)]
            for c,(a,b) in zip(comb, requests):
                if c == 1:
                    state[a] -= 1
                    state[b] += 1
            # print(comb, state)
            if sum(abs(x) for x in state) == 0:
                maxres = max(maxres, sum(comb))
        return maxres
