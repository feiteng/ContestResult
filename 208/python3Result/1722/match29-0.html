class Node:
    def __init__(self, name):
        self.name = name
        self.children = []

class ThroneInheritance:

    def __init__(self, kingName: str):
        self.king = Node(kingName)
        self.tree = dict()
        self.tree[kingName] = self.king
        self.dead = set()

    def birth(self, parentName: str, childName: str) -> None:
        node = Node(childName)
        self.tree[parentName].children.append(node)
        self.tree[childName] = node

    def death(self, name: str) -> None:
        self.dead.add(name)

    def getInheritanceOrder(self) -> List[str]:
        ans = []
        def preorder(node):
            if not node:
                return
            if node.name not in self.dead:
                ans.append(node.name)
            for each in node.children:
                preorder(each)
        preorder(self.king)
        return ans
