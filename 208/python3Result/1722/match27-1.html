class Node:
    def __init__(self, name):
        self
.children = []
        self.name = name
        self.alive = True

class ThroneInheritance:

    def __init__(self, kingName: str):
        self.node_king = Node(kingName)
        # self.nodes = [self.node_king]
        self.map = {kingName: self
.node_king}

    def birth(self, parentName: str, childName: str) -> None:
        new_node = Node(childName)
        self.map[childName] = new_node
        # self.nodes.append
        self.map[parentName].children.append(new_node)

    def death(self, name: str) -> None:
        self.map[name].alive = False
    
    def pre_order(self,node, result):
        if node.alive:
            result.append(node.name)
        for child in node.children:
            self.pre_order(child, result)

    def getInheritanceOrder(self) -> List[str]:
        result = []
        self.pre_order(self.node_king, result)
        return result
