class Person:
    def __init__(self, name, father):
        self.name=name
        self.father=father
        self.children=[]
        self.isdead=False
        
class ThroneInheritance:

    def __init__(self, kingName: str):
        self.king=Person(kingName, None)
        self.m={}
        self.m[kingName]=self.king
    
    def birth(self, parentName: str, childName: str) -> None:
        p=Person(childName, parentName)
        self.m[childName]=p
        self.m[parentName].children.append(p)
        

    def death(self, name: str) -> None:
        self.m[name].isdead=True

    def find(self, f, res):
        if not f.isdead:
            res.append(f.name)
        for c in f.children:
            self.find(c, res)
        
    def getInheritanceOrder(self) -> List[str]:
        res=[]
        self.find(self.king, res)
        return res
