class Tree:
    def __init__(self,name):
        self.val=name
        self.child=[]
        self.death=False
    
    
class ThroneInheritance:

    def __init__(self, kingName: str):
        self.root=Tree(kingName)
        self.d={kingName:self.root}

    def birth(self, parentName: str, childName: str) -> None:
        cur=self.d[parentName]
        tmp=Tree(childName)
        cur.child.append(tmp)
        self.d[childName]=tmp

    def death(self, name: str) -> None:
        cur=self.d[name]
        cur.death=True

    def getInheritanceOrder(self) -> List[str]:
        ans=[]
        def dfs(root):
            nonlocal ans
            if not root.death:
                ans.append(root.val)
            for child in root.child:
                dfs(child)
        dfs(self.root)
        return ans
