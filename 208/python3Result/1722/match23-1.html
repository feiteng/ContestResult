class TreeNode:
    def __init__(self, name):
        self.name = name
        self.alive = True
        self.children = []

class ThroneInheritance:
    def __init__(self, kingName: str):
        self.root = TreeNode(kingName)
        self.map = {kingName:self.root}

    def birth(self, parentName: str, childName: str) -> None:
        c_node = TreeNode(childName)
        p_node = self.map[parentName]
        p_node
.children.append(c_node)
        self.map[childName] = c_node

    def death(self, name: str) -> None:
        self.map[name].alive = False

    def getInheritanceOrder(self) -> List[str]:
        ans = []
        
        def dfs(root):
            if not root:
                return 
            if root
.alive:
                ans.append(root.name)
            for child in root.children:
                dfs(child)
        dfs(self.root)
        return ans
