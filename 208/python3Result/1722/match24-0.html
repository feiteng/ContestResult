class Node:
    
    def __init__(self, val):
        self.val = val
        self.child = []
        self.is_dead = False
        
class ThroneInheritance:
    
    def __init__(self, kingName: str):
        self.root = Node(kingName)
        self.d = {kingName: self.root}
        
    def birth(self, parentName: str, childName: str) -> None:
        p = self.d[parentName]
        c = Node(childName)
        p.child.append(c)
        self.d[childName] = c

    def death(self, name: str) -> None:
        self.d[name].is_dead = True

    def getInheritanceOrder(self) -> List[str]:
        res = []
        def dfs(node):
            if not node.is_dead:
                res.append(node.val)
            for nei in node.child:
                dfs(nei)
        dfs(self.root)
        return res
