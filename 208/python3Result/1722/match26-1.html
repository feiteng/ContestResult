class ThroneInheritance:

    def __init__(self, kingName: str):
        self.kingName = kingName
        self.famely_dict = {kingName: []}
        self.death_set = set()
        

    def birth(self, parentName: str, childName: str) -> None:
        self
.famely_dict[parentName].append(childName)
        self.famely_dict[childName] = []

    def death(self, name: str) -> None:
        self.death_set.add(name)

    def getInheritanceOrder(self) -> List[str]:
        res_list = []
        def next_step(cur_name):
            if cur_name not in self.death_set:
                res_list.append(cur_name)
                
            for next_name in self.famely_dict[cur_name]:
                next_step(next_name)
        
        next_step(self.kingName)
        return res_list
