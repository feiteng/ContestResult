class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        ans = -1
        profile = 0
        waiting = 0
        used = 0
        rotate = 0
        for num in customers:
            waiting += num
            rotate += 1
            if waiting >= 4:
                waiting -= 4
                used += 4  
            else:
                used += waiting
                waiting = 0
            if profile<boardingCost*used-rotate*runningCost:
                ans = rotate
                profile = boardingCost*used-rotate*runningCost
        while waiting > 0:
            rotate += 1
            if waiting >= 4:
                waiting -= 4
                used += 4  
            else:
                used += waiting
                waiting = 0
            if profile<boardingCost*used-rotate*runningCost:
                ans = rotate
                profile = boardingCost*used-rotate*runningCost
        return ans
