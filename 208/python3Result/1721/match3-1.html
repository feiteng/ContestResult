class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        profit = 0
        waiting = 0
        best = 0
        ans = -1
        i = 0
        while i < len(customers) or waiting > 0:
            if i < len(customers):
                waiting += customers[
i]
            
            boarding = min(waiting, 4)
            waiting -= boarding

            profit += boarding * boardingCost - runningCost
            if profit > best:
                ans = i + 1
                best = profit
                
            i += 1

        return ans
