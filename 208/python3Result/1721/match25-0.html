class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        best = 0
        
        q = [0] * 4
        current = 0
        profit = 0
        best_index = -1
        for index, x in enumerate(customers):
            current += x
            
            getting_on = min(current, 4)
            q[index%4] += getting_on
            profit += boardingCost * getting_on - runningCost
            current -= getting_on
            
            if profit > best:
                best = profit
                best_index = index + 1
                
        index = len(customers)
        while current > 0:
            getting_on = min(current, 4)
            profit += boardingCost * getting_on - runningCost
            current -= getting_on
            
            if profit > best:
                best = profit
                best_index = index + 1
            index += 1
                
            
        return best_index
