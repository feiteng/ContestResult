class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        wait = 0
        profit = 0
        rota = 0
        mon = 0
        i = 0
        
        while i < len(customers) or wait > 0:
            if i < len(customers):
                wait += customers[
i]
            board = min(4, wait)
            wait -= board
            mon += board * boardingCost - runningCost
            if mon > profit:
                profit = mon
                rota = i + 1
            i += 1
        return rota if profit > 0 else -1
