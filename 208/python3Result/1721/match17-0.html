class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        current_shift = 0
        shift_ans = 0
        max_ans = 0
        tmp_profit = 0
        waiting_customers = 0
        for customer_num in customers:
            waiting_customers += customer_num
            customers_to_board = min(waiting_customers, 4)
            waiting_customers -= customers_to_board
            tmp_profit = tmp_profit + customers_to_board * boardingCost - runningCost
            current_shift += 1
            if max_ans < tmp_profit:
                max_ans = tmp_profit
                shift_ans = current_shift
        # 如果还有剩余的顾客，继续轮转
        while waiting_customers > 0:
            customers_to_board = min(waiting_customers, 4)
            waiting_customers -= customers_to_board
            tmp_profit = tmp_profit + customers_to_board * boardingCost - runningCost
            current_shift += 1
            if max_ans < tmp_profit:
                max_ans = tmp_profit
                shift_ans = current_shift
        if max_ans > 0:
            return shift_ans
        else:
            return -1
