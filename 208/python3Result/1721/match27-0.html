class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        ans = -1
        ts = 0
        bs = 0
        cc = 0
        for c in customers:
            cc += c
            # if cc == 0:
            #     continue
            ts += 1
            cbs = bs
            if cc > 4:
                cc -= 4
                cbs = cbs + 4 * boardingCost - runningCost
            else:
                cbs = cbs + cc * boardingCost - runningCost
                cc = 0
            if cbs > bs:
                bs = cbs
                ans = ts
        while cc != 0:
            ts += 1
            cbs = bs
            if cc > 4:
                cc -= 4
                cbs = cbs + 4 * boardingCost - runningCost
            else:
                cbs = cbs + cc * boardingCost - runningCost
                cc = 0
            if cbs > bs:
                bs = cbs
                ans = ts
        return ans
