class Solution:
    def minOperationsMaxProfit(self, c: List[int], b: int, r: int) -> int:
        ans_cnt = 0
        cnt = 0
        ans = -1
        cur = 0
        cur_n = 0
        for n in c:
            cur_n += n
            diff = min(4, cur_n)
            cur_n -= diff
            cur += b * diff - r

            cnt += 1
            # print(cur)
            if cur > ans:
                ans = cur
                ans_cnt = cnt
        while cur_n > 0:
            diff = min(4, cur_n)
            cur_n -= diff
            cur += b * diff - r

            cnt += 1
            # print(cur)
            if cur > ans:
                ans = cur
                ans_cnt = cnt
        #print(ans)
        if ans <= 0: return -1
        else: return ans_cnt
