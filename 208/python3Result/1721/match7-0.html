class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        running_person=0
        waitting_person=0
        index=0
        max_profile=0
        max_index=0
        while(waitting_person>0 or index<len(customers)):
            if index<len(customers):
                waitting_person+=customers[index]
            
            increament_running_person=min(4,waitting_person)
            running_person+=increament_running_person
            waitting_person-=increament_running_person
        
            profile=running_person*boardingCost-(index+1)*runningCost
            if profile>max_profile:
                max_profile=profile
                max_index=index
            
            index+=1
        if max_profile==0:
            return -1
        else:
            return max_index+1
