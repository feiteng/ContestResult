class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        ans = -1
        profit = mx = 0 
        i = ppl = 0
        while i < len(customers) or ppl: 
            if i < len(customers): ppl += customers[i]
            xx = min(4, ppl)
            ppl -= xx 
            profit += xx * boardingCost - runningCost 
            if mx < profit: 
                ans = i+1
                profit = mx 
            i += 1
        return ans
