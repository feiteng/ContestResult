class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        profit = 0
        w = 0
        ans = -1
        count = 0
        max_profit = 0

        for c in customers:
            w += c
            x = min(w, 4)
            profit += x * boardingCost - runningCost
            w -= x
            count += 1
            if profit > max_profit:
                max_profit = profit
                ans = count

        while w > 0:
            x = min(w, 4)
            # print(x)
            profit += x * boardingCost - runningCost
            w -= x
            count += 1
            if profit > max_profit:
                max_profit = profit
                ans = count

        return ans
