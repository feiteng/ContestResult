class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        if boardingCost * 4 < runningCost:
            return -1
        max_profit = -float('inf')
        res = -1
        waiting = 0
        profit = 0
        operations = 0
        for customer in customers:
            waiting += customer
            onboarding = min(4, waiting)
            profit += onboarding * boardingCost - runningCost
            waiting -= onboarding
            operations += 1
            if profit > max_profit:
                max_profit = profit
                res = operations
        while waiting != 0:
            onboarding = min(4, waiting)
            profit += onboarding * boardingCost - runningCost
            waiting -= onboarding
            operations += 1
            if profit > max_profit:
                max_profit = profit
                res = operations
        return res
