class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        maxNum = -1
        maxProfit = 0
        curNum = 0
        curPeople = 0
        curProfit = 0

        for customer in customers:
            curNum += 1
            curPeople += customer

            ridePeople = min(4, curPeople)
            curPeople -= ridePeople
            curProfit += boardingCost * ridePeople - runningCost

            if curProfit > maxProfit:
                maxProfit = curProfit
                maxNum = curNum

        while curPeople > 0:
            curNum += 1
            ridePeople = min(4, curPeople)
            curPeople -= ridePeople
            curProfit += boardingCost * ridePeople - runningCost

            if curProfit > maxProfit:
                maxProfit = curProfit
                maxNum = curNum

        return maxNum
