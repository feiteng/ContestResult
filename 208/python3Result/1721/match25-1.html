class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        best_rotations = 0
        cost = 0
        cur_cost = 0
        cur_count = 0
        cur_rotations = 0
        for i, count in enumerate(customers):
            cur_count += count
            board = min(4, cur_count)
            cur_count 
-= board
            cur_cost += board * boardingCost - runningCost
            if cur_cost > cost:
                cost = cur_cost
                best_rotations = i + 1
        
        cur_rotations = len(customers)
        while cur_count > 0:
            board = min(4, cur_count)
            cur_cost += board * boardingCost - runningCost
            cur_count -= board
            if cur_cost > cost:
                cost = cur_cost
                best_rotations = cur_rotations + 1
            cur_rotations += 1
            
            
        return best_rotations if cost > 0 else -1
