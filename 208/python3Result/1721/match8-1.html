class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        pool = 0
        cur = 0
        ans = 0
        max_profit = 0
        i = 0
        for c in customers:
            pool += c
            serve = min(4, pool)
            pool -= serve
            cur += serve * boardingCost - runningCost
            if cur > max_profit:
                max_profit = cur
                ans = i + 1
            i += 1
        
        while pool > 0:
            serve = min(4, pool)
            pool -= serve
            cur += serve * boardingCost - runningCost
            if cur > max_profit:
                max_profit = cur
                ans = i + 1
            i += 1
        return -1 if ans == 0 else ans
