class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        mp = 0
        mt = -1
        
        cp = 0
        ct = 0
        waiting = 0
        for customer in customers:
            # print("customer: ", customer)
            # print("waiting: ", waiting, "cp: ", cp)
            waiting += customer
            boarding = min(waiting, 4)
            waiting -= boarding
            cp += boarding * boardingCost - runningCost
            ct += 1
            if cp > mp:
                mp = cp
                mt = ct
                
        while waiting > 0:
            boarding = min(waiting, 4)
            waiting -= boarding
            cp += boarding * boardingCost - runningCost
            ct += 1
            if cp > mp:
                mp = cp
                mt = ct
        
        print(cp)
        print(mp, mt)
        return mt
