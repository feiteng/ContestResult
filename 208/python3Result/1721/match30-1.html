class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        profit = 0
        ret = 0
        rot = 0
        b, w = 0, 0
        i = 0
        for c in customers:
            w += c
            b = min(w, 4)
            w -= b            
            profit += b * boardingCost - runningCost
            i += 1
            if profit > ret:
                ret = profit
                rot = i
        while w:
            b = min(w, 4)
            w -= b            
            profit += b * boardingCost - runningCost
            i += 1
            if profit > ret:
                ret = profit
                rot = i            
        return rot if ret > 0 else -1
