class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        best_index = -1
        best_profit = 0
        profit = 0
        queue = 0
        i = 0
        while queue or i < len(customers):
            if i < len(customers):
                c = customers[i]
                queue += c
            on = min(queue, 4)
            queue -= on
            profit += on * boardingCost - runningCost
            # print(profit)
            if profit > best_profit:
                # print(profit)
                best_profit = profit
                best_index = i + 1
            i += 1
        return best_index
