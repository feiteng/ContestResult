class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        res = -1
        waiting = 0
        profit = 0
        maxProfit = -1
        i = 0
        while i < len(customers) or waiting > 0:
            if i < len(customers):
                waiting += customers[i]
            serve_c = min(waiting, 4)
            waiting -= serve_c 
            profit += serve_c * boardingCost
            profit -= runningCost
            if profit > maxProfit:
                maxProfit = profit
                res = i + 1
            i += 1
        if maxProfit > 0:
            return res
        else:
            return -1
