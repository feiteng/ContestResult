class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        ans = 0
        turn = 0
        profit = 0
        wait = 0
        for i, c in enumerate(customers):
            wait += c
            cur = min(wait, 4)
            wait -= cur
            profit += cur*boardingCost
            profit -= runningCost
            if profit > ans:
                ans = profit
                turn = i + 1
        while wait > 0:
            i += 1
            cur = min(wait, 4)
            wait -= cur
            profit += cur*boardingCost
            profit -= runningCost
            if profit > ans:
                ans = profit
                turn = i + 1
        return turn if ans > 0 else -1
