class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        ans = spin = count = current = 0
        i = 0
        for i, c in enumerate(customers):
            count += c
            diff = min(count, 4)
            count -= diff
            current += diff * boardingCost - runningCost
            if current > ans:
                ans = current
                spin = i+1
        while count:        
            i += 1    
            diff = min(count, 4)
            count -= diff
            current += diff * boardingCost - runningCost
            if current > ans:
                ans = current
                spin = i+1
        return spin if spin else -1
