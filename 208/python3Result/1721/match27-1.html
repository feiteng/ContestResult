class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        mx_profit, turn 
= -1, -1
        wait, profit, cost = 0, 0, 0
        count = 0
        for customer in customers:
            count += 1
            cost += runningCost
            wait += customer
            if wait >= 4:
                wait -= 4
                profit += 4 * boardingCost
            else:
                profit += wait * boardingCost
                wait = 0
            if profit - cost > mx_profit:
                mx_profit = profit - cost
                turn = count

        while wait != 0:
            count += 1
            cost += runningCost
            if wait >= 4:
                wait -= 4
                profit += 4 * boardingCost
            else:
                profit += wait * boardingCost
                wait = 0
            if profit - cost > mx_profit:
                mx_profit = profit - cost
                turn = count
        
        return turn
