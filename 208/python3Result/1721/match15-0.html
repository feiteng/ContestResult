class Solution:
    def minOperationsMaxProfit(self, customers: List[int],
                               boardingCost: int, runningCost: int) -> int:
        ans = 0
        res = 0
        p = 0
        w = 0
        i = 0
        for c in customers:
            i += 1
            w += c
            b = min(w, 4)
            w -= b
            p += b * boardingCost - runningCost
            if p > res:
                res = p
                ans = i
        while w:
            i += 1
            b = min(w, 4)
            w -= b
            p += b * boardingCost - runningCost
            if p > res:
                res = p
                ans = i
        return ans if ans else -1
