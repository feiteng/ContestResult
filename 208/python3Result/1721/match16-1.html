class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        pre = 0
        preCnt = 0
        ans = 0
        ansCnt = -1
        nowCnt = 0
        for val in customers:
            nowCnt += 1
            pre += val
            now = min(4,pre)
            pre -= now
            preCnt += now*boardingCost - runningCost
            if preCnt>ans:
                ans = preCnt
                ansCnt = nowCnt
        while pre>0:
            nowCnt += 1
            now = min(4,pre)
            pre -= now
            preCnt += now*boardingCost - runningCost
            if preCnt>ans:
                ans = preCnt
                ansCnt = nowCnt
        return ansCnt
