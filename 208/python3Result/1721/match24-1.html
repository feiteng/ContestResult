class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        result = 0
        cost = 0
        now = 0
        best_cost = 0
        best_result = -1
        for ind, i in enumerate(customers):
            now += i
            tmp = min(4, now)
            now -= tmp
            cost += tmp * boardingCost - runningCost
            if cost > best_cost:
                best_cost = cost
                best_result = ind + 1
        remain = now // 4
        cost += remain * (4 * boardingCost - runningCost)
        now 
-= remain * 4
        if cost > best_cost:
            best_cost = cost
            best_result = len(customers) + remain
        cost += now * boardingCost - runningCost
        if cost > best_cost:
            best_cost = cost
            best_result = len(customers) + remain + 1
        return best_result
