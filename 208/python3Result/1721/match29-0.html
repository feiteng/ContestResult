class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        res=0
        rescur=0
        cus=0
        now=0
        cur=0
        for c in customers:
            cur+=1
            cus+=c
            if cus<4:
                now+=cus*boardingCost-runningCost
                cus=0
            else:
                now+=4*boardingCost-runningCost
                cus-=4
            if now>res:
                res=now
                rescur=cur
        while cus:
            cur+=1
            if cus<4:
                now+=cus*boardingCost-runningCost
                cus=0
            else:
                now+=4*boardingCost-runningCost
                cus-=4
            if now>res:
                res=now
                rescur=cur
        if res==0:
            return -1
        else:
            return rescur
