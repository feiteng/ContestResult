class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        wait = 0
        
        to_ret_i = -1
        to_ret_v = 0
        v_now = 0
        it = 0
        while wait > 0 or it < len(customers):
            if it < len(customers) :
                wait += customers[
it] 
            p_now = min(wait, 4)
            wait -= p_now
            v_now += p_now * boardingCost - runningCost
            if v_now > to_ret_v :
                to_ret_v = v_now
                to_ret_i = it + 1
            it += 1
        # print(to_ret_v)
        return to_ret_i
