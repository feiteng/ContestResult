class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        cap = 4
        ans = -1
        profit = 0
        reserve = 0
        curr = 0
        for i, c in enumerate(customers):
            reserve += c
            onboard = min(reserve, cap)
            reserve -= onboard
            curr += boardingCost * onboard - runningCost
            if curr > profit:
                profit = curr
                ans = i + 1
        div, mod = divmod(reserve, 4)
        curr += 4 * boardingCost - runningCost
        if curr > profit:
            profit = curr
            ans = len(customers) + div
        curr += mod * boardingCost - runningCost
        if curr > profit:
            profit = curr
            ans = len(customers) + div + 1
        return ans
