class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        max_rot = -1
        max_profit = 0
        profit = 0
        available_customers = 0
        idx = 0
        for new_customers in customers:
            available_customers += new_customers
            chosen_customers = min(available_customers, 4)
            available_customers -= chosen_customers
            profit += chosen_customers*boardingCost - runningCost
            if profit > max_profit:
                max_profit = profit
                max_rot = idx + 1
            # print(f"rot {idx+1}, chosen: {chosen_customers}, waiting: {available_customers}, profit={profit}, max={max_profit} ({max_rot})")
            idx += 1
        while available_customers > 0:
            chosen_customers = min(available_customers, 4)
            available_customers -= chosen_customers
            profit += chosen_customers*boardingCost - runningCost
            if profit > max_profit:
                max_profit = profit
                max_rot = idx + 1
            # print(f"rot {idx+1}, chosen: {chosen_customers}, waiting: {available_customers}, profit={profit}, max={max_profit} ({max_rot})")
            idx += 1
            
        return max_rot
