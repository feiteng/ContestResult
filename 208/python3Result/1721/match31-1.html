class Solution:
    def minOperationsMaxProfit(self, customers: List[int], boardingCost: int, runningCost: int) -> int:
        total_board = 0
        remaining = 0
        max_profit = -1
        rotations = 0
        for i, c in enumerate(customers, 1):
            remaining += c
            board = min(4, remaining)
            total_board += board
            remaining = remaining - board
            profit = (total_board * boardingCost) - (i * runningCost)
            #print(i, " profit: ", profit)
            if profit > max_profit:
                rotations = i
                max_profit = profit
        while remaining > 0:
            i += 1
            board = min(4, remaining)
            total_board += board
            remaining = remaining - board
            profit = (total_board * boardingCost) - (i * runningCost)
            if profit > max_profit:
                rotations = i
                max_profit = profit
        if max_profit > 0:
            return rotations
        return -1
