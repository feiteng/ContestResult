class Solution {
public:
    vector<char>letters;
    vector<char>nums;
    string reformat(string s) {
        int len=s.size();
        for(int i=0;i<len;i++)
        {
            if(s[i]<='z'&&s[i]>='a')letters.push_back(s[i]);
            else nums.push_back(s[i]);
        }
        int len1=letters.size(),len2=nums.size();
        string res="";
        if(len1-len2==1)
        {
            for(int i=0;i<len2;i++)
            {
                res+=letters[i];
                res+=nums[i];
            }
            res+=letters[len2];
        }
        else if(len2-len1==1)
        {
            for(int i=0;i<len1;i++)
            {
                res+=nums[i];
                res+=letters[i];
            }
            res+=nums[len1];
        }
        else if(len1==len2)
        {
            for(int i=0;i<len1;i++)
                {
                res+=nums[i];
                res+=letters[i];
            }
        }
        else return "";
        return res;
    }
};
