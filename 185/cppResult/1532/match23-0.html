class Solution {
public:
    string reformat(string s) {
        vector<char>n;
        vector<char>c;
        for(auto ch:s){
            if(isdigit(ch)){
                n.push_back(ch);
            }
            else{
                c.push_back(ch);
            }
        }
        int a=n.size(),b=c.size();
        if(a==b+1){
            string res;
            int idxa=0,idxb=0;
            for(int i=0;i<s.size();i++){
                if(i%2==0){
                    res+=n[idxa];
                    idxa++;
                }
                else{
                    res+=c[idxb];
                    idxb++;
                }
            }
            return res;
        }
        else if(a+1==b||a==b){
            string res;
            int idxa=0,idxb=0;
            for(int i=0;i<s.size();i++){
                if(i%2==0){
                    res+=c[idxa];
                    idxa++;
                }
                else{
                    res+=n[idxb];
                    idxb++;
                }
            }
            return res;
        }
        return "";
    }
};
