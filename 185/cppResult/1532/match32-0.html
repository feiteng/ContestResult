class Solution {
public:
    string reformat(string s) {
     string a, b;
    for(char c:s){
        if(isdigit(c))a.push_back(c);
        else b.push_back(c);
    }
        if(a.size()>b.size())swap(a,b);
        if(a.size()+1<b.size())return "";
        string ans;
        for(int i=0;a.size()||b.size();i++){
            if(i&1){
                ans.push_back(a.back());
                a.pop_back();
            }   else {
                ans.push_back(b.back());
                b.pop_back();
                
            }
        }
        return ans;
    }
};
