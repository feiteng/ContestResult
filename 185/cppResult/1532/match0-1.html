class Solution {
public:
    string reformat(string s) {
    	vector<char> a, b;
    	for (auto ch : s) {
			if (isdigit(ch))
				a.emplace_back(ch);
			else
				b.emplace_back(ch);
		}
		if (a.size() < b.size())
			swap(a, b);
		if (a.size() > b.size() + 1)
			return "";
		stringstream ss;
		while (!a.empty() || !b.empty()) {
			if (!a.empty()) {
				ss << a.back();
				a.pop_back();
			}
			if (!b.empty()) {
				ss << b.back();
				b.pop_back();
			}
		}
		return ss.str();
    }
};
