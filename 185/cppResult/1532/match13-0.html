class Solution {
public:
    string reformat(string s) {
        vector<char> a;
        vector<char>b;
        int nc = 0;
        int zc = 0;
        for(auto x: s) {
            if( x >='0' && x <= '9'){
                nc++; 
                a.push_back(x);
            }else {
                zc++;
                b.push_back(x);
            }
        }
        if(abs(nc-zc) > 1) return ""; 
        string ans = "";
        int ai = 0, bi = 0;
        if( nc > zc) {
            ans+= a[ai++];
            while(ai < nc) {
                ans+= b[bi++];
                ans+= a[ai++];
            }
        } else if( zc > nc){
            ans += b[bi++];
            while(ai < nc) {
                ans+= a[ai++];
                ans+= b[bi++];
            }
        } else {
             while(ai < nc) {
                ans+= a[ai++];
                ans+= b[bi++];
            }
        }
        return ans;
        
    }
};
