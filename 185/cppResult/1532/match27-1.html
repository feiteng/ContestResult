class Solution {
public:
    string reformat(string s) {
        vector<char> num, alp;
        for (auto c: s){
            if ('0' <= c && c <= '9') num.push_back(c); else alp.push_back(c);
        }
        if (num.size() < alp.size()) swap(num, alp);
        if (num.size() > alp.size() + 1) return "";
        
        string ans = "";
        
        for (int i = 0; i < num.size(); i++){
            ans += num[i];
            if (i < alp.size()) ans += alp[i];
        }
        return ans;
    }
};
