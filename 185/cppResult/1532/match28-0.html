class Solution {
public:
    string reformat(string s) {
        string res;
        vector<char> digits, alphas;
        for(auto c : s){
            if(c>='a' && c<= 'z') alphas.push_back(c);
            else digits.push_back(c);
        }
        if(alphas.size() > digits.size()) swap(alphas, digits);
        if(digits.size() >= alphas.size()+2) return "";

        for(int i = 0; i < alphas.size(); i++){
            res.push_back(digits[i]);
            res.push_back(alphas[i]);
        }
        if(digits.size()>alphas.size()) res.push_back(digits.back());
        return res;
    }
};
