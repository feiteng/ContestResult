class Solution {
public:
    string reformat(string s) {
        string a, b;
        for (char x: s) {
            if (x <= '9' && x >= '0') a += x;
            else b += x;
        }
        int n = a.size(), m = b.size();
        if (n < m) {
            swap(a, b);
            swap(n, m);
        }
        if (m + 1 < n) return "";
        string ans;
        for (int i = 0; i < n; i++) {
            ans += a[i];
            if (i < m) ans += b[i];
        }
        return ans;
    }
};
