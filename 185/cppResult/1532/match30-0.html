class Solution {
public:
    string reformat(string s) {
        int n = s.size();
        string res(n, 0);
        int c = 0, d = 0;
        for (int i = 0; i < n; ++i) {
            if (isdigit(s[i])) {
                ++c;
            } else {
                ++d;
            }
        }
        if (abs(c - d) > 1) return "";
        if (c > d) {
            c = 0;
            d = 1;
        } else {
            c = 1;
            d = 0;
        }
        for (int i = 0; i < n; ++i) {
            if (isdigit(s[i])) {
                res[c] = s[i];
                c += 2;
            } else {
                res[d] = s[i];
                d += 2;
            }
        }
        return res;
    }
};
