class Solution {
public:
    string reformat(string s) {
        string d, l;
        for (char c : s) {
            if (c <= '9' && c >= '0') d.push_back(c);
            else l.push_back(c);
        }
        
        if (abs((int)d.size() - (int)l.size()) >= 2) return "";
        
        if (d.size() < l.size()) swap(d, l);
        
        string ans = "";
        for (int i = 0; i < l.length(); i++) {
            ans.push_back(d[i]);
            ans.push_back(l[i]);
        }
        if (d.size() != l.size()) ans.push_back(d[d.length() - 1]);
        return ans;
    }
};
