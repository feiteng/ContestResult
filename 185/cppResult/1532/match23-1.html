class Solution {
public:
    string reformat(string s) {
        string num="",al="";
        for(int i=0;i<s.size();i++)
        {
            if(s[i]>='0' && s[i]<='9')
            {
                num+=s[i];
            }
            else
            {
                al+=s[i];
            }
        }
        int pp=num.size();
        int qq=al.size();
        string temp="";
        if(abs(pp-qq)>1)
            return temp;

        string ans="";
        if(pp>qq)
        {
            int ii=0,jj=0;
            for(int i=0;i<s.size();i++)
            {
                if(i%2==0)
                {
                    ans+=num[ii];
                    ii++;
                    
                }
                else
                {
                    ans+=al[jj];
                    jj++;
                }
            }
        }
        else

        {
            int ii=0,jj=0;
            for(int i=0;i<s.size();i++)
            {
                if(i%2==0)
                {
                    ans+=al[ii];
                    ii++;
                    
                }
                else
                {
                    ans+=num[jj];
                    jj++;
                }
            }
        }
        return ans;
            
    }
};
