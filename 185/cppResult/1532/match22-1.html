class Solution {
public:
    string reformat(string s) {
        string a, b;
        for(char c: s) {
            if('0' <= c && c <= '9') a.push_back(c);
            else b.push_back(c);
        }
        if(abs(int(a.size()) - int(b.size())) > 1) return "";
        if(a.size() < b.size()) a.swap(b);
        string z;
        for(int i = 0; i < a.size(); ++i) {
            z.push_back(a[i]);
            if(i < b.size()) z.push_back(b[i]);
        }
        return z;
    }
};
