class Solution {
public:
    string reformat(string s) {
        string digit,alph;

        for(int i=0;i<s.size();i++)
            if(s[i]>='0' && s[i]<='9')
                digit+=s[i];
            else
                alph+=s[i];
        int a=digit.size(),b=alph.size();
        if(abs(a-b)>1)
            return "";
        string str;
        if(a<b)
            swap(digit,alph);
        
        for(int i=0;i<digit.size();i++)
        {
            str+=digit[i];
            if(i<alph.size())
                str+=alph[i];
        }
        return str;
    }
};
