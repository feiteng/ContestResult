const ll MOD=1e9+7;

class Solution {
public:
  string reformat(string s) {
    string l, d;
    for (char c:s) if (isalpha(c)) l+=c; else d+=c;
    if (abs((int)l.size()-(int)d.size()) > 1) return "";
    if (l.size()<d.size()) swap(l,d);
    string res;
    for (int i=0;i<l.size();i++) {
      res += l[i];
      if (i<d.size()) res+=d[i];
    }
    return res;
  }
};
