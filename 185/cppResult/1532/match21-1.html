class Solution {
public:
    string reformat(string s) {
        vector<char> A, B;
        for(int i = 0; s[i]; i++) {
            if(s[i] >= '0' && s[i] <= '9') A.push_back(s[i]);
            else B.push_back(s[i]);
        }
        string S = "";

        if(A.size() == B.size() || A.size() == B.size() + 1) {
            for(int i = 0; i < B.size(); i++) {
                S += A[i];
                S += B[i];
            }
            if(A.size() == B.size() + 1) {
                S += A[A.size() - 1];
            }
        } else if(A.size() + 1 == B.size()) {
            for(int i = 0; i < A.size(); i++) {
                S += B[i];
                S += A[i];
            }
            S += B[B.size() - 1];
        }
        return S;
    }
};
