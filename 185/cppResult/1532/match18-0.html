class Solution {
public:
    string reformat(string s) {
        string a,b;
        for (char ch : s)
            if(isdigit(ch)) a+=ch;
            else b+=ch;
        if(abs(int(a.size())-int(b.size()))>1)
            return "";
        if(a.size()<b.size())
            swap(a,b);
        string ret;
        for (int i=0; i<b.size(); i++)
            ret+=a[i], ret+=b[i];
        if(a.size()>b.size())
            ret+=a.back();
        return ret;
    }
};
