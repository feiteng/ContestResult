class Solution {
public:
    string reformat(string s) {
        vector<char> num, ch;
        for(int i = 0; i < s.size(); ++i) {
            if('0' <= s[i] && s[i] <= '9') num.push_back(s[i]);
            else ch.push_back(s[i]);

        }
        if(abs((int)num.size() - (int)ch.size()) > 1) return string();
        string res;
        if(num.size() < ch.size()) swap(num, ch);
        while(num.size()) {
            res.push_back(num.back()); num.pop_back();
            if(ch.size()) {
                res.push_back(ch.back());
                ch.pop_back();
            }
        }
        return res;
    }
};
