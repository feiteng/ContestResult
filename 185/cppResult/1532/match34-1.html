public:
    string reformat(string s) {
       /* string a = "" , b = "";

            else if(s[i] >= '0' && s[i] <= '9')b += s[i];
        }

        string d = "" , c = "";
        for(auto& e : s){
            if(e >= '0' && e <= '9')d += e;
            else c += e;
        }
        string ret = "";
        int i = 0 , j = 0;
        while(i < d.size() && j < c.size()){
            ret = ret + d[i] + c[j];
            ++i , ++j;
        }
        if(i == d.size() && j != c.size()){
            ret = c[j] + ret;
            ++j;
        }
        else if(i != d.size() && j == c.size()){
            ret += d[i];
            ++i;
        }
        return (i == d.size() && j == c.size()) ? ret : "";
    }
};
