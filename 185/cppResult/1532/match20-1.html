class Solution {
public:
    string reformat(string s) {
        vector<int> abc(26,0);
        vector<int> digit(10,0);
        int digits=0;
        int abcs=0;
        for(char c:s){
            if(isdigit(c)) {digit[c-'0']++; digits++;}
            else {abc[c-'a']++; abcs++;}
        }
        //for(int i:abc) cout<<i<<" ";
        string res="";
        int n=s.size();
        if(abcs>digits){

            for(int i=0;i<n;i++){
            if(i%2==0){
                for(int j=0;j<26;j++){
                    if(abc[j]>0){

                        res+='a'+j;
                        abc[j]--;
                        break;
                    }
                }
            }else {
                for(int j=0;j<10;j++){
                    if(digit[j]>0){
                    res+='0'+j;
                    digit[j]--;
                    break;}
                }
            }
            if(res.size()!=i+1) return "";
        }
        }else{

            for(int i=0;i<n;i++){
            if(i%2==0){
                for(int j=0;j<10;j++){
                    if(digit[j]>0){

                        res+='0'+j;
                        digit[j]--;
                        break;
                    }
                }
            }else {
                for(int j=0;j<26;j++){
                    if(abc[j]>0){
                    cout<<j<<"\n";
                    res+='a'+j;
                    abc[j]--;
                    break;}
                }
            }
            if(res.size()!=i+1) return "";
        }
        }
        
        return res;
    }
};
