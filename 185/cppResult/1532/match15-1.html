class Solution {
public:
    string reformat(string s) {
        string letters = "";
        string digits = "";
        for (auto x:s){
            if (x>='0' && x<='9') digits += x;
            else letters+=x;
        }
        if (digits.size()>letters.size())swap(digits,letters);
        if (letters.size() > digits.size()+1) return "";
        string ret="";
        for (int i=0;i<digits.size();i++) ret+=letters[i], ret+=digits[i];
        if (letters.size()>digits.size())ret+=letters.back();
        return ret;
    }
};
