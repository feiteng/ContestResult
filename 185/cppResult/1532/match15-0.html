class Solution {
public:
    string reformat(string s) {
        string a = "";
        string b = "";
        for(auto c : s){
            if(c>='0'&&c<='9')a+=c;
            else b+=c;
        }
        if(a.size()<b.size())swap(a,b);
        if(a.size()>b.size()+1){
            return "";
        }
        string ret;
        for(int i=0;i<a.size();i++){
            ret+=a[i];
            if(i<b.size())ret+=b[i];
        }
        return ret;
    }
};
