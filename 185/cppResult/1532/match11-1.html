class Solution {
public:
    string reformat(string s) {
        string digit = "";
        string letter = "";
        for( int i = 0 ; i < s.size(); ++i ){
            if( isdigit(s[i]) ) digit += s[i];
            else letter += s[i];
        }

        if( abs( (int)digit.size() - (int)letter.size()) > 1 ) return "";
        
        string res = "";
        int i = 0, j = 0;
        //cout<<digit.size()<<" "<<letter.size()<<endl;
        if( digit.size() > letter.size() ){
            res += digit[0];
            i++;
            while( i < digit.size() ){
                res += letter[j++];
                res += digit[i++];
            }            
        }else if( letter.size() > digit.size() ){
            res += letter[0];
            j++;
            while( i < digit.size() ){
                res += digit[i++];
                res += letter[j++];
                
            }
        }else{
            while( i < digit.size() ){
                res += letter[j++];
                res += digit[i++];
            }
        }
        return res;
        
        
        
    }
};
