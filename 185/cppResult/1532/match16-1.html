class Solution {
public:
    string reformat(string s) {
        vector<char> num;
        vector<char> alpha;
        int numCnt = 0;
        int alphaCnt = 0;
        for(auto c : s)
        {
            if(c >= 'a' && c <= 'z')
            {
                alphaCnt++;
                alpha.push_back(c);
            }
            else
            {
                numCnt++;
                num.push_back(c);
            }
        }
        if(abs(numCnt - alphaCnt) > 1) return "";
        string res;

        if(numCnt > alphaCnt)
        {
            for(int i = 0; i < alphaCnt; ++i)
            {
                res.push_back(num[i]);
                res.push_back(alpha[i]);
            }
            res.push_back(num.back());
            return res;
        }
        else if(numCnt == alphaCnt)
        {
            for(int i = 0; i < alphaCnt; ++i)
            {
                res.push_back(num[i]);
                res.push_back(alpha[i]);
            }
            return res;
        }
        else
        {
            for(int i = 0; i < numCnt; ++i)
            {
                res.push_back(alpha[i]);
                res.push_back(num[i]);
                
            }
            res.push_back(alpha.back());
            return res;
        }
    }
};
