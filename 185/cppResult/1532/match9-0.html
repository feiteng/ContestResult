class Solution {
public:
    string reformat(string s) {
        string num="";
        string word="";
        for (int i=0;i<s.size();i++){
            if (s[i]>='0' and s[i]<='9') num+=s[i];
            else word+=s[i];
        }
        int numsize=num.size();
        int wordsize=word.size();
        if (abs(numsize-wordsize)>1) return "";
        string res="";
        if (numsize>wordsize){
            for (int i=0;i<wordsize;i++){
                res+=num[i];res+=word[i];
            }
            res+=num[numsize-1];
        }
        else if (wordsize>numsize){
            for (int i=0;i<numsize;i++){
                res+=word[i];res+=num[i];
            }
            res+=word[wordsize-1];
        }
        else{
            for (int i=0;i<numsize;i++){
                res+=word[i];res+=num[i];
            }
        }
        return res;
    }
};
