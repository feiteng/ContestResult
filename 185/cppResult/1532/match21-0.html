class Solution {
public:
    string reformat(string s) {
        string a, b;
        for (auto &c : s) {
            if (c >= '0' && c <= '9') {
                a += c;
            } else {
                b += c;
            }
        }
        int m = a.size();
        int n = b.size();
        if (abs(m-n) <= 1) {
            string res;
            
            if (m >= n) {
                for (int i = 0; i < min(m,n); i++) {
                    res += a[i];
                    res += b[i];
                }
                if (m > n) {
                    res += a[m-1];
                }
            }
            
            else if (m <n) {
                for (int i = 0; i < min(m,n); i++) {
                    res += b[i];
                    res += a[i];
                }
                res += b[n-1];
            }
            
            return res;
        }
        return "";
    }
};
