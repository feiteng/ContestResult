class Solution {
public:
    string reformat(string s) {
      string num, str;       
      for (auto c : s) {
        if (isdigit(c)) {
          num.push_back(c);
        } else {
          str.push_back(c);
        }
      }
      if (abs((int)num.size() - (int)str.size()) > 1) return "";
      if (num.size() < str.size()) num.swap(str);
      string ans;
      while (!num.empty()) {
        ans.push_back(num.back());
        num.pop_back();
        if (!str.empty()) {
          ans.push_back(str.back());
          str.pop_back();
        }
      }
      return ans;
    }
};
