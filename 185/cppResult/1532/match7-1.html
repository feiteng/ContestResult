class Solution {
public:
    string reformat(string s) {
        string alphas, digits;
        for (auto c : s) {
            if ('a' <= c && c <= 'z') {
                alphas += c;
            } else {
                digits += c;
            }
        }
        if (abs(static_cast<int>(alphas.size()) - static_cast<int>(digits.size())) > 1) {
            return "";
        }
        if (alphas.size() < digits.size()) {
            swap(alphas, digits);
        }
        for (size_t i = 0; i < s.length(); ++i) {
            s[i] = i % 2 == 0 ? alphas[i / 2] : digits[i / 2];
        }
        return s;
    }
};
