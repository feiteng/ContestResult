class Solution {
public:
    vector< char > s1, s2;
    bool used[505];
    string reformat(string s) {
        bool type = 0;
        string ans = "";
        for(int i = 0; i < s.length(); i++) {
            if( type == 0 ) {
                for(int j = 0; j < s.length(); j++) {
                    if( !used[j] && s[j] >= 'a' && s[j] <= 'z') {
                        used[j] = 1;
                        type = 1;
                        ans += s[j];
                        break;
                    } 
                }   
            }
            else {
                for(int j = 0; j < s.length(); j++) {
                    if( !used[j] && s[j] >= '0' && s[j] <= '9') {
                        used[j] = 1;
                        type = 0;
                        ans += s[j];
                        break;
                    } 
                }  
            }
        }
        if( ans.length() == s.length() )
            return ans;
        type = 1;
        memset(used,0,sizeof(used));
        ans = "";
        for(int i = 0; i < s.length(); i++) {
            if( type == 0 ) {
                for(int j = 0; j < s.length(); j++) {
                    if( !used[j] && s[j] >= 'a' && s[j] <= 'z') {
                        used[j] = 1;
                        type = 1;
                        ans += s[j];
                        break;
                    } 
                }   
            }
            else {
                for(int j = 0; j < s.length(); j++) {
                    if( !used[j] && s[j] >= '0' && s[j] <= '9') {
                        used[j] = 1;
                        type = 0;
                        ans += s[j];
                        break;
                    } 
                }  
            }
        }
        if( ans.length() == s.length() )
            return ans;
        ans = "";
        return ans;
    }
};
