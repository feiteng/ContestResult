class Solution {
 public:
  string reformat(string s) {
    string a, b;

    for (char c : s) {
      if (isalpha(c)) {
        a += c;
      } else {
        b += c;
      }
    }
    if (abs((int)a.size() - (int)b.size()) > 1) {
      return "";
    }
    if (a.size() < b.size()) {
      swap(a, b);
    }
    string ans;
    for (int i = 0; i < (int)a.size(); ++i) {
      ans += a[i];
      if (i < (int)b.size()) {
        ans += b[i];
      }
    }
    return ans;
  }
};
