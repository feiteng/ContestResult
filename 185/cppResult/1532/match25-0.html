class Solution {
public:
    string reformat(string s) {

        vector<char> a, b;
        for(char c: s)
            if(c >= '0' && c <= '9') a.push_back(c);
            else b.push_back(c);

        if(abs((int)a.size() - (int)b.size()) >= 2) return "";

        if(a.size() < b.size()) swap(a, b);
        string res = "";
        while(b.size()){
            res += a.back();
            res += b.back();
            b.pop_back();
            a.pop_back();
        }

        if(a.size()) res += a.back();
        return res;
    }
};
