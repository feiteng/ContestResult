class Solution {
public:
    string reformat(string s) {
        string digits = "";
        string alpha = "";
        int n = s.size();
        for(int i=0; i<n; i++) {
            if('0'<=s[i] && s[i]<='9') digits+=s[i];
            else alpha+=s[i];
        }
        int na = alpha.size();
        int nd = digits.size();

        if(abs(na-nd)>1) return "";
        string result = "";
        int i = 0, j = 0;
        if(na>nd) {
            result+=alpha[i];
            i++;
            while(i<na) {
                result+=digits[j];
                j++;
                result+=alpha[i];
                i++;
            }
        }else if(nd>na) {
            result+=digits[j];
            j++;
            while(j<nd) {
                result+=alpha[i];
                i++;
                result+=digits[j];
                j++;
            }
        } else {
            while(i<na) {
                result+=digits[j];
                j++;
                result+=alpha[i];
                i++;
            }
        }
        return result;
    }
};
