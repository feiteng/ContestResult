class Solution {
public:
    string reformat(string s) {
        vector<char> a, n;
        for (auto i: s) {
            if (isalpha(i)) {
                a.push_back(i);
            } else {
                n.push_back(i);
            }
        }
        
        if (a.size() < n.size()) swap(a, n);
        if (a.size() - n.size() > 1) return "";
        string ret;
        for (int i = 0; i < n.size(); ++i) {
            ret += a[i];
            ret += n[i];
        }
        if (a.size() > n.size()) ret += a.back();
        return ret;
    }
};
