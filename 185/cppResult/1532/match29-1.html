class Solution {
public:
    string reformat(string s) {
        vector<char> digits, alphas;
        for (const char& c : s) {
            if (isdigit(c)) {
                digits.push_back(c);
            } else {
                alphas.push_back(c);
            }
        }

        string ans = "";
        
        if (alphas.size() > digits.size()) {
            swap(alphas, digits);
        }
        if (digits.size() - alphas.size() > 1) return ans;
        for (int i = 0; i < digits.size(); ++i) {
            ans += digits[i];
            if (i < alphas.size())
                ans += alphas[i];
        }
        return ans;
    }
};
