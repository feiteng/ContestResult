class Solution {
public:
    string reformat(string s) 
    {
        string a="",b="";
        for(auto &e: s)
        {
            if(e>='0'&&e<='9')
                a+=e;
            else
                b+=e;
        }
        int t=(int)a.size()-(int)b.size();
        if(abs(t)>1)
            return "";
        if(a.size()>b.size())
        {
            string res="";
            res+=a[0];
            for(int i=0;i<b.size();i++)
            {
                res+=b[i];
                res+=a[i+1];
            }
            return res;
        }
        else
        {
            string res="";
            res+=b[0];

            for(int i=0;i<a.size();i++)
            {
                res+=a[i];
                if(i+1<b.size())
                    res+=b[i+1];
            }
            return res;
        }
    }
};
