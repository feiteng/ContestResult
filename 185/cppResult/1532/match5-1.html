class Solution {
public:
    string reformat(string s) {

        vector< char > number, other;
        for( char c : s ) {
            if( c >= '0' && c <= '9' ) number.push_back( c );
            else other.push_back( c );
        }
        if( number.size() < other.size() ) swap( number, other );
        if( number.size() > other.size() + 1 ) return "";
        string ret;
        for( int i = 0; i < other.size(); i++ ) {
            ret += number[ i ];
            ret += other[ i ];
        }
        if( number.size() > other.size() ) ret += number.back();
        return ret;
    }
};
