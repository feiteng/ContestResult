class Solution {
public:
    string reformat(string s) {
        string num;
        string let;
        for (auto& iters : s) {
            if (iters <= '9' && iters >= '0') {
                num.push_back(iters);
            }
            else {
                let.push_back(iters);
            }
        }

        int nSize = num.size();
        int lSize = let.size();
        if (abs(nSize - lSize) > 1) {
            return "";
        }
        else {

            if (nSize > lSize) {
                string ret;
                for (int index = 0; index < lSize; ++index) {
                    ret.push_back(num[index]);
                    ret.push_back(let[index]);
                }
                ret.push_back(num[nSize - 1]);
                return ret;
            }
            else if (lSize > nSize) {
                string ret;
                for (int index = 0; index < nSize; ++index) {
                    ret.push_back(let[index]);
                    ret.push_back(num[index]);
                }
                ret.push_back(let[lSize - 1]);
                return ret;
            }
            else {
                string ret;
                for (int index = 0; index < nSize; ++index) {
                    ret.push_back(let[index]);
                    ret.push_back(num[index]);
                }
                return ret;
            }
        }
    }
};
