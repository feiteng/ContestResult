class Solution {
public:
    string reformat(string s) {
        std::vector<char> s1,s2;
        for (auto it:s){
            if (it>='0'&&it<='9') s1.push_back(it);
            else s2.push_back(it);
        }

        if (abs((int)s1.size()-(int)s2.size())>1) return "";
        if (s2.size()>s1.size()) swap(s2,s1);
            string res;
            for (int i=0;i<s2.size();i++) res+=s1[i],res+=s2[i];
            if (s1.size()>s2.size())
            res+=s1[s1.size()-1];
        return res;
    }
};
