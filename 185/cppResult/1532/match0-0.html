class Solution {
public:
    string reformat(string s) {
        vector<char> a,b;
        for(auto &i:s){
            if(i>='0'&&i<='9')b.push_back(i);
            else a.push_back(i);
        }
        if(a.size()<b.size())a.swap(b);
        if(a.size()-b.size()>1)return "";
        string res;
        while(!a.empty()||!b.empty()){
            if(!a.empty()){
                res.push_back(a.back());
                a.pop_back();
            }
            if(!b.empty()){
                res.push_back(b.back());
                b.pop_back();
            }
        }
        return res;
    }
};
