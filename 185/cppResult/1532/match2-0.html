class Solution {
public:
    string reformat(string s) {
        string a, b;
        int n = s.length();
        for (int i = 0; i < n; ++i)
            if (s[i] >= '0' && s[i] <= '9')
                a += s[i];
            else
                b += s[i];
        
        int x = a.length(), y = b.length();
        if (abs(x-y) > 1)
            return "";
        
        if (x < y)
            swap(a, b);
        
        string res;
        for (int i = 0; i < a.length(); ++i) {
            res += a[i];
            if (i < b.length())
                res += b[i];
        }
        return res;
    }
};
