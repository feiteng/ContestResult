class Solution {
public:
    string reformat(string s) {
        string A;
        string C;
        
        for (auto c : s) {
            if (isdigit(c)) {
                C += c;
            } else {
                A += c;
            }
        }
        
        if (abs((int)A.length() - (int)C.length()) > 1) {
            return "";
        }
        
        if (A.length() < C.length()) {
            swap(A, C);
        }
        
        string res;
        for (int i = 0; i < A.length(); ++i) {
            res += A[i];
            if(i < C.length()) {
                res += C[i];
            }
        }
        
        return res;
    }
};
