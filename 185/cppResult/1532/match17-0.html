class Solution {
public:
    string reformat(string s) {
        string d = "", l = "";
        for(char c : s){
            if(c>='0'&&c<='9') d.push_back(c);
            else l.push_back(c);
        }
        string res = "";
        if(d.size()>l.size()+1 || l.size()>d.size()+1){
            return "";
        }
        else if (d.size()==l.size()+1){
            res.push_back(d[0]);
            for(int i = 0; i < l.size(); ++i){
                res.push_back(l[i]);
                res.push_back(d[i+1]);
            }
            return res;
        }
        else if (d.size()==l.size()){
             for(int i = 0; i < l.size(); ++i){
                res.push_back(l[i]);
                res.push_back(d[i]);
            }
            return res;
        }
        else{
                        res.push_back(l[0]);
            for(int i = 0; i < d.size(); ++i){
                res.push_back(d[i]);
                res.push_back(l[i+1]);
            }
            return res;
        }
    }
};
