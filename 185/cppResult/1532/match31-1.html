class Solution {
public:
    string reformat(string s) {
        string num,str,ans;
        for(char x:s){
            if(islower(x))str.push_back(x);
            else num.push_back(x);
        }
        if(abs((int)num.size()-(int)str.size())>1)return "";
        if(num.size()>=str.size()){
            for(int i=0;i<num.size();i++){
                ans.push_back(num[i]);
                if(i<str.size())ans.push_back(str[i]);
            }
        }else{
            for(int i=0;i<str.size();i++){
                ans.push_back(str[i]);
                if(i<num.size())ans.push_back(num[i]);
            }
        }
        return ans;
    }
};
