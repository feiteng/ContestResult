class Solution {
public:
    string reformat(string s) {
        string a, b;
        for (char c : s) {
            if (c >= '0' && c <= '9') a.push_back(c);
            else b.push_back(c);
        }
        if (a.size() < b.size()) swap(a, b);
        if (a.size() > b.size() + 1) return "";
        string ret;
        if (a.size() > b.size()) {
            ret.push_back(a.back());
            a.pop_back();
        }
        while (!b.empty()) {
            ret.push_back(b.back()); b.pop_back();
            ret.push_back(a.back()); a.pop_back();
        }
        return ret;
        
        
    }
};
