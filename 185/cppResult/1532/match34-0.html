class Solution {
public:
    string reformat(string s) {
        string al = "", nu = "";
        for (auto ch: s) {
            if (isalpha(ch)) al += ch;
            else nu += ch;
        }
        string ans = "";
        int i=0, j=0;
        // cout<<al<<" "<<nu<<endl;
        if (al.length() < nu.length()) swap(al, nu);
        for (;i<al.length()&&j<nu.length();i++,j++) {
            ans.push_back(al[i]);
            ans.push_back(nu[j]);
        }
        if (i + 1 < al.length() || j + 1 < nu.length()) return "";
        if (i < al.length()) ans.push_back(al[i]);
        if (j < nu.length()) ans.push_back(nu[j]);
        return ans;
    }
};
