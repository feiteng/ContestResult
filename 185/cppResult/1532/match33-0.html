class Solution {
public:
    string reformat(string s) {
        string s1, s2;
        for(char c: s) {
            if(isalpha(c)) s1.push_back(c);
            else s2.push_back(c);
        }
        if(s2.size() > s1.size()) swap(s1, s2);
        if(s1.size() - s2.size() >= 2) return "";
        string res = "";
        int n = s2.size();
        int i = 0, j = 0;
        for(; j < n; i++, j++) {
            res.push_back(s1[i]);
            res.push_back(s2[j]);
        }
        if(i != s1.size()) res.push_back(s1[i]);
        return res;
    }
};
