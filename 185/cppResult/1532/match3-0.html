class Solution {
public:
    string reformat(string str) {
        string A, B;
        for (auto ch: str) {
            if (ch >= '0' && ch <= '9') {
                A.push_back(ch);
            } else {
                B.push_back(ch);
            }
        }
        if (abs((int)A.size() - (int)B.size()) > 1)
            return "";
        string ans;
        if (A.size() < B.size())
            swap(A, B);
        while (!A.empty() || !B.empty()) {
            if (!A.empty()) {
                ans.push_back(A.back());
                A.pop_back();
            }
            if (!B.empty()) {
                ans.push_back(B.back());
                B.pop_back();
            }
        }
        return ans;
    }
};
