class Solution {
public:
    string reformat(string s) {
        if(s == "") return "";
        string res = s;
        vector<char> digit, letter;
        for(char c : s){
            if('0' <= c && c <= '9') digit.push_back(c);
            else letter.push_back(c);
        }
        
        if(digit.size() > letter.size()) swap(digit, letter);
        if(letter.size() - digit.size() >= 2) return "";
        for(int i = 0; i < digit.size(); i++){
            res[2 * i] = letter[i];
            res[2 * i + 1] = digit[i];
        }
        if(letter.size() > digit.size()) res[s.length() - 1] = letter[letter.size() - 1];
        return res;
    }
};
