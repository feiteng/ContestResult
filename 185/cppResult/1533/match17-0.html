class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> F;
        map<int, map<string, int>> T;
        for (auto& o : orders) {
            F.insert(o[2]);
            T[stoi(o[1])][o[2]] ++;
        }
        vector<vector<string>> r;
        vector<string> header(1, "Table");
        for (auto f : F) { header.push_back(f); }
        r.push_back(header);
        for (auto it: T) {
            vector<string> row;
            row.push_back(to_string(it.first));
            for (string k : F) {
                row.push_back(to_string(it.second[k]));
            }
            r.push_back(row);
        }
        return r;
    }
};
