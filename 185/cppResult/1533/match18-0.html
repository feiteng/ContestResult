class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> strSet; // food name
        set<int> tableNumber;
        unordered_map<string, unordered_map<string, int>> map2; // table, food, count
        for(auto &order : orders)
        {
            strSet.insert(order[2]);
            tableNumber.insert(stoi(order[1]));
            map2[order[1]][order[2]]++;
        }
        
        vector<vector<string>> res;
        vector<string> firstRow;
        firstRow.push_back("Table");
        for(auto &a : strSet)
        {
            firstRow.push_back(a);
        }
        res.push_back(firstRow);
        for(auto &a : tableNumber)
        {
            vector<string> strVec;
            strVec.push_back(to_string(a));
            for(int i = 1; i < firstRow.size(); ++i)
            {
                strVec.push_back(to_string(map2[to_string(a)][firstRow[i]]));
            }
            res.push_back(strVec);
        }
        return res;
    }
};
