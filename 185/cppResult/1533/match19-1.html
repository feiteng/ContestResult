class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> mp;
        set<string> foods;
        for (auto order : orders) {
            mp[stoi(order[1])][order[2]]++;
            foods.insert(order[2]);
        }
        vector<vector<string>> table;
        vector<string> item;
        item.push_back("Table");
        for (auto food : foods) {
            item.push_back(food);
        }
        table.push_back(item);
        for (auto p : mp) {
            vector<string> item;
            item.push_back(to_string(p.first));
            for (auto food : foods) {
                item.push_back(to_string(p.second[food]));
            }
            table.push_back(item);
        }
        return table;
    }
};
