class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<string, int> vis;
        unordered_map<string, int> ma[501];
        for(int i = 0; i < orders.size(); i++) {
            ma[stoi(orders[i][1])][orders[i][2]]++;
            vis[orders[i][2]] = 1;
        }
        vector<vector<string>> ans;
        vector<string> v;
        v.push_back("Table");
        for(auto it : vis) {
            v.push_back(it.first);
        }
        ans.push_back(v);
        v.clear();
        for(int i = 1; i <= 500; i++) {
            if(ma[i].size() == 0) continue;
            v.push_back(to_string(i));
            for(auto it : vis) {
                v.push_back(to_string(ma[i][it.first]));
            }
            ans.push_back(v);
            v.clear();
        }
        return ans;
    }
};
