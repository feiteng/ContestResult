class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> foods;
        map<int, unordered_map<string, int>> map;
        for(auto x: orders) {
            map[stoi(x[1])][x[2]]++;
            foods.insert(x[2]);
        }
        vector<vector<string>> res;
        res.push_back({"Table"});
        for(auto x: foods) {
            res.back().push_back(x);
        }
        for(auto x: map) {
            vector<string> tmp;
            tmp.push_back(to_string(x.first));
            for(int i = 1; i < res[0].size(); i++) {
                tmp.push_back(to_string(x.second[res[0][i]]));
            }
            res.push_back(tmp);
        }
        return res;
    }
};
