class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<string,int> count;
        set<string> food;
        set<int> table;
        for(auto it: orders)
        {
            count[it[1]+it[2]]++;
            table.insert(stoi(it[1]));
            food.insert(it[2]);
        }
        vector<vector<string>> ans;
        vector<string> row;
        row.push_back("Table");
        for(auto it: food)
            row.push_back(it);
        ans.push_back(row);
        for(auto it: table)
        {
            row.clear();
            row.push_back(to_string(it));
            for(auto jt: food)
            {
                row.push_back(to_string(count[to_string(it)+jt]));
            }
            ans.push_back(row);
        }
        return ans;
    }
};
