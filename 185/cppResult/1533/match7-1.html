class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> items;

        map<int, map<string, int> > m;
        set<int> tab;
        for(auto it : orders){
            items.insert(it[2]);
            tab.insert(stoi(it[1]));
            m[stoi(it[1])][it[2]]++;
        }
        vector<vector<string> > ret;
        vector<string> curr;
        curr.push_back("Table");
        for(auto it : items) curr.push_back(it);
        ret.push_back(curr);
        for(auto i : tab){
            vector<string> ncurr;
            ncurr.push_back(to_string(i));
            for(auto j : items){
                ncurr.push_back(to_string(m[i][j]));
            }
            ret.push_back(ncurr);
        }
        return ret;
    }
};
