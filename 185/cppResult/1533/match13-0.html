class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> f;
        map<int, map<string,int>> m;
        for (auto ord: orders) {
            int id = stoi(ord[1]);
            f.insert(ord[2]);
            m[id][ord[2]]++;
        }
        vector<vector<string>> ans;
        vector<string> cols = {"Table"};
        for (auto &foo: f) cols.push_back(foo);
        ans.push_back(cols);
        for (auto &p: m) {
            vector<string> row;
            row.push_back(to_string(p.first));
            for (auto &foo: f) {
                row.push_back(to_string(p.second[foo]));
            }
            ans.push_back(row);
        }
        return ans;
    }
};
