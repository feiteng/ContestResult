class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> mp;
        set<string> foods;
        for (auto &i : orders) {
            mp[stoi(i[1])][i[2]]++;
            foods.insert(i[2]);
        }
        vector<vector<string>> res;
        vector<string> head {"Table"};
        for (auto &i : foods) {
            head.push_back(i);
        }
        res.push_back(head);
        for (auto &i : mp) {
            vector<string> t;
            t.push_back(to_string(i.first));
            for (auto j : foods) {
                auto it = i.second.find(j);
                if (it == i.second.end()) {
                    t.push_back("0");
                } else {
                    t.push_back(to_string(it->second));
                }
            }
            res.push_back(t);
        }
        return res;
    }
};
