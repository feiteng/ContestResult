class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        vector<string> food;
        map<int, map<string, int>> ans;
        for(auto& x: orders) {
            food.push_back(x[2]);
            ++ans[stoi(x[1])][x[2]];
        }
        sort(food.begin(), food.end());
        food.resize(unique(food.begin(), food.end()) - food.begin());
        vector<vector<string>> res(ans.size() + 1);
        res[0].push_back("Table");
        for(auto& s: food) res[0].push_back(s);
        int i = 0;
        for(auto& p: ans) {
            res[++i].push_back(to_string(p.first));
            for(auto& s: food) res[i].push_back(to_string(p.second[s]));
        }
        return res;
    }
};
