class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<string, map<string, int>> f;
        set<string> mn; set<int> cz;
        for (auto x : orders)
        {
            f[x[1]][x[2]]++;
            mn.insert(x[2]);
            cz.insert(stoi(x[1]));
        }
        vector<vector<string>> res;
        res.push_back(vector<string>());
        res[0].push_back("Table");
        for (auto x : mn)  res[0].push_back(x);
        int i = 0;
        for (auto x : cz)
        {
            res.push_back(vector<string>());++i;
            res[i].push_back(to_string(x));
        }
        for (int j = 1; j < res.size(); ++j)
        {
            for (int k = 1; k < res[0].size(); ++k)
            {
                res[j].push_back(to_string(f[res[j][0]][res[0][k]]));
            }
        }
        return res;
    }
};
