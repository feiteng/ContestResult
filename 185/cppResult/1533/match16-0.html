class Solution {
public:
    string go(int k){
        stringstream ss;
        string res;
        ss<<k;
        ss>>res;
        return res;
    }
    int gogo(string str){
        stringstream ss(str);
        int num;
        ss>>num;
        return num;
    }
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> s;
        map<int,map<string,int> > m;
        for(auto &it:orders){
            s.insert(it[2]);
            m[gogo(it[1])][it[2]]++;
        }
        vector<vector<string> > res;
        vector<string> v;
        v.push_back("Table");
        for(auto it:s){
            v.push_back(it);
        }
        res.push_back(v);
        for(auto& it:m){
            vector<string> tmp;
            tmp.push_back(go(it.first));
            for(auto it2:s){
                tmp.push_back(go(it.second[it2]));
            }
            res.push_back(tmp);
        }
        return res;
    }
};
