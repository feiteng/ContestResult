class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> m;
        set<string> dish;
        for(auto o : orders) {
            int table = atoi(o[1].c_str());
            m[table][o[2]]++;
            dish.insert(o[2]);
        }
        
        vector<vector<string>> res;
        vector<string> first = {"Table"};
        for(auto d : dish) first.push_back(d);
        
        res.push_back(first);
        
        for(auto d : m) {
            auto d2 = d.second;
            vector<string> cur = {to_string(d.first)};
            for(auto x : dish) {
                cur.push_back(d2.count(x) == 0 ? "0" : to_string(d2[x]));
            }
            res.push_back(cur);
        }
        
        return res;
    }
};
