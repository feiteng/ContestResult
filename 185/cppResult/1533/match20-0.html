class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> m;
        set<string> s;
        for (auto &o: orders) {
            m[stoi(o[1])][o[2]]++;
            s.insert(o[2]);
        }
        vector<vector<string>> re(1, {"Table"});
        for (auto &it : s) re[0].push_back(it);
        for (auto &t: m) {
            re.push_back({to_string(t.first)});
            for (int i = 1; i < re[0].size(); ++i) {
                re.back().push_back(to_string(t.second[re[0][i]]));
            }
        }
        return re;
    }
};
