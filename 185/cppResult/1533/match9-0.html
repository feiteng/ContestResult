class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<pair<int, string> , int> h;
        vector<string> menu;
        set<int> t;
        
        for (auto &o : orders) {
            menu.emplace_back(o[2]);
            int id;
            sscanf(o[1].c_str() ,"%d", &id);
            h[make_pair(id, o[2])] ++;
            t.insert(id);
        }
        
        vector<vector<string>> res;
        vector<string> e;
        e.emplace_back("Table");
        sort(menu.begin() , menu.end());
        menu.erase(unique(menu.begin(), menu.end()) , menu.end());
        for (auto &m : menu) e.emplace_back(m);
        res.emplace_back(e);
        
        for (auto id : t) {
            e.clear();
            e.push_back(to_string(id));
            for (auto &m : menu) {
                e.push_back(to_string(h[make_pair(id, m)]));
            }
            res.emplace_back(e);
        }
        return res;
        
        
    }
};
