class Solution {
public:
    map<string,int>id[510];
    map<string,int>menu;
    int vis[510]={};
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        int len=orders.size();
        for(int i=0;i<len;i++)
        {
            id[stoi(orders[i][1])][orders[i][2]]++;

            menu[orders[i][2]]=1;
            vis[stoi(orders[i][1])]=1;
        }
        vector<string>temp;
        vector<vector<string>>res;
        temp.push_back("Table");
        for(auto m:menu)
        {
            temp.push_back(m.first);
        }
        res.push_back(temp);
        for(int i=1;i<510;i++)
        {
            if(!vis[i])continue;
            temp.clear();
            temp.push_back(to_string(i));
            for(auto m:menu)
            {
                temp.push_back(to_string(id[i][m.first]));
            }
            res.push_back(temp);
        }
        return res;
    }
};
