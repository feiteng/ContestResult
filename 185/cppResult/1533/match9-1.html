class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> dish;
        map<int, map<string, int>> m;
        for(auto x: orders) {
            dish.insert(x[2]);

            m[atoi(x[1].c_str())][x[2]]++;
        }
        vector<vector<string>> v;
        vector<string>h;
        h.push_back("Table");
        for(auto d: dish) h.push_back(d);
        v.push_back(h);
        for(auto e: m) {
            vector<string> t;
            t.push_back(to_string(e.first));
            for(auto d: dish) {
                t.push_back(to_string(e.second[d]));
            }
            v.push_back(t);
        }
        return v;
    }
};
