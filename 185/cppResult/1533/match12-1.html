class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> tab;
        
        set<string> food;
        for(auto& v : orders) {
            int num = stoi(v[1]);
            tab[num][v[2]]++;
            food.insert(v[2]);

        }
        
        vector<vector<string>> ret;
        
        vector<string> aux;
        aux.push_back("Table");
        for(auto& x : food) aux.push_back(x);
        ret.push_back(aux);
        
        for(auto& m : tab) {
            aux.clear();
            aux.push_back(to_string(m.first));
            for(auto& x : food) {
                aux.push_back(to_string(m.second[x]));
            }
            ret.push_back(aux);
        }
        return ret;
    }
};
