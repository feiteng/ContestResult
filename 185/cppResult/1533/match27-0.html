class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        unordered_set<int> st;
        unordered_set<string> sf;
        for (auto& order : orders) {
            st.insert(stoi(order[1]));
            sf.insert(order[2]);
        }
        
        vector<int> vt(st.begin(), st.end());
        sort(vt.begin(), vt.end());
        unordered_map<string, int> mt;
        for (int i = 0; i < vt.size(); ++i) {
            mt[to_string(vt[i])] = i;
        }
        
        vector<string> vf(sf.begin(), sf.end());
        sort(vf.begin(), vf.end());
        unordered_map<string, int> mf;
        for (int i = 0; i < vf.size(); ++i) {
            mf[vf[i]] = i;
        }
        
        vector<vector<int>> num(vt.size(), vector<int>(vf.size(), 0));
        for (auto& order : orders) {
            ++num[mt[order[1]]][mf[order[2]]];
        }
        
        vector<vector<string>> ans(vt.size() + 1, vector<string>(vf.size() + 1));
        ans[0][0] = "Table";
        for (int i = 0; i < vt.size(); ++i) {
            ans[i + 1][0] = to_string(vt[i]);
        }
        
        for (int j = 0; j < vf.size(); ++j) {
            ans[0][j + 1] = vf[j];
        }
        
        for (int i = 0; i < vt.size(); ++i) {
            for (int j = 0; j < vf.size(); ++j) {
                ans[i + 1][j + 1] = to_string(num[i][j]);
            }
        }
        
        return ans;
    }
};
