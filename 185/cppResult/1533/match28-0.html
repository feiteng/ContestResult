class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> foods;
        map<int, map<string, int>> table;
        for(auto& order: orders){
            table[stoi(order[1])][order[2]]++;
            foods.insert(order[2]);
        }
        vector<string> foodsItem(foods.begin(), foods.end());
        vector<vector<string>> res;
        vector<string> first = {"Table"};
        for(auto& str: foodsItem){
            first.push_back(str);
        }
        res.push_back(first);
        for(auto& pair: table){
            vector<string> temp;
            temp.push_back(to_string(pair.first));
            map<string, int>& foodsCount = pair.second;
            for(auto& str: foodsItem){
                if(foodsCount.count(str)==0){
                    temp.emplace_back("0");
                } else {
                    temp.push_back(to_string(foodsCount[str]));
                }
            }
            res.push_back(temp);
        }
        return res;
    }
};
