class Solution {
public:
    string toString(int x){
        if(!x)return "0";
        stringstream ss;
        ss<<x;
        string ret;
        ss>>ret;
        return ret;
    }
    int toInt(string x){
        stringstream ss;
        ss<<x;
        int ret;
        ss>>ret;
        return ret;
    }
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<string,map<string,int>>mp;
        set<string>foods;
        for(auto s:orders){
            mp[s[1]][s[2]]++;
            foods.insert(s[2]);
        }
        vector<vector<string>> ret;
        ret.push_back(vector<string>());
        ret.back().push_back("Table");
        for(auto it:foods)ret.back().push_back(it);
        for(auto it:mp){
            ret.push_back(vector<string>());
            ret.back().push_back(it.first);
            for(auto it2:foods){
                ret.back().push_back(toString(it.second[it2]));
            }
        }
        sort(ret.begin()+1,ret.end(),[&](vector<string> v1,vector<string> v2){
           return toInt(v1[0])<toInt(v2[0]); 
        });
        return ret;
    }
};
