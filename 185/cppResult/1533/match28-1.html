class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> food;
        map<int, map<string, int>> count;
        vector<vector<string>> ret;
        for (auto o: orders) {
            food.insert(o[2]);
            count[stoi(o[1])][o[2]]++;
        }
        
        vector<string> first = {"Table"};
        for (auto s: food) {
            first.push_back(s);
        }
        ret.push_back(first);
        
        for (auto it: count) {
            vector<string> tbl = {to_string(it.first)};
            for (auto s: food) {
                tbl.push_back(to_string(it.second[s]));
            }
            ret.push_back(tbl);
        }
        return ret;
        
    }
};
