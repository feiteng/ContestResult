class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> a;
        map<int,unordered_map<string,int>> m;
        vector<vector<string>> res(1);
        for(auto& q:orders){
            int table=atoi(q[1].c_str());
            ++m[table][q[2]];
            a.insert(q[2]);

        }
        res[0].push_back("Table");
        for(auto& q:a)
            res[0].push_back(q);
        for(auto& q:m){
            vector<string> t;
            t.push_back(to_string(q.first));
            for(auto& z:a){
                t.push_back(to_string(q.second[z]));
            }
            res.push_back(t);
        }
        return res;
    }
};
