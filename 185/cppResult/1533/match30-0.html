class Solution {
public:
    set<string> name;
    set<int> table;
    map<string, int> order[550];
    string i2s(int x){
        if (x == 0) return "0";
        string ret = "";
        while(x){
            ret = (char)('0' + x % 10) + ret;
            x /= 10;
        }
        return ret;
    }
    int s2i(string x){
        int ret = 0;
        for(auto v: x){
            ret = ret * 10 + v - '0';
        }
        return ret;
    }
    vector<vector<string>> displayTable(vector<vector<string>>& dat) {
        name.clear(); table.clear();
        for (int i = 0; i <= 500; i++) order[i].clear();
        
        for (auto l: dat){
            name.insert(l[2]), table.insert(s2i(l[1]));
            order[s2i(l[1])][l[2]] += 1;
            // printf("%d\n", s2i(l[1]));
        }
        
        vector<vector<string>> ans;
        vector<string> title; title.push_back("Table");
        for (auto n: name) title.push_back(n);
        ans.push_back(title);
        
        for (auto t: table){
            vector<string> row;
            row.push_back(i2s(t));
            for (auto n: name){
                row.push_back(i2s(order[t][n]));
            }
            ans.push_back(row);
        }
        
        return ans;
    }
};
