class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> t;
        set<int> n;
        set<string> d;
        for(int i = 0; i < orders.size(); ++i) {

            t[ atoi(orders[i][1].c_str()) ][ orders[i][2] ] ++;
            n.insert(atoi(orders[i][1].c_str()));
            d.insert(orders[i][2]);
        }
        vector<vector<string>> re;
        vector<string> items;
        items.push_back("Table");
        for(auto it = d.begin(); it != d.end(); ++it) {
            items.push_back(*it);
        }
        re.push_back(items);
        for(auto it = n.begin(); it != n.end(); ++it) {
            vector<string> num;
            num.push_back(to_string(*it));
            for(int j = 1; j < items.size(); ++j) {
                num.push_back(to_string(t[*it][items[j]]));
            }
            re.push_back(num);
        }
        return re;
    }
};
