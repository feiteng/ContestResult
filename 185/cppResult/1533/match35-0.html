class Solution {
 public:
  vector<vector<string>> displayTable(vector<vector<string>>& orders) {
    vector<vector<string>> ans;
    set<string> cai;
    map<int, unordered_map<string, int>> cnt;
    for (auto& v : orders) {
      cnt[stoi(v[1])][v[2]]++;
      cai.insert(v[2]);
    }

    ans.resize(1 + cnt.size());
    ans[0].emplace_back("Table");
    for (auto& s : cai) {
      ans[0].emplace_back(s);
    }
    int i = 1;
    for (auto& p : cnt) {
      ans[i].push_back(to_string(p.first));
      for (auto& s : cai) {
        ans[i].push_back(to_string(p.second[s]));
      }
      i++;
    }
    return ans;
  }
};
