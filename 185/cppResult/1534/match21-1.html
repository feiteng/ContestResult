class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int n = croakOfFrogs.length();
        if (n % 5 != 0) {
            return -1;
        }
        unordered_map<char, int> freq;
        for (const auto& c : croakOfFrogs) {
            freq[c]++;
        }
        for (const auto& p : freq) {
            if (p.second != freq['c']) {
                return -1;
            }
        }
        vector<int> am(6, 0);
        int result = 0;
        for (const auto& c : croakOfFrogs) {
            if (c == 'c') {

                am[1]++;
            } else if (c == 'r') {
                if (am[1] == 0) {
                    return -1;
                } else {
                    am[1]--;
                    am[2]++;
                }
            } else if (c == 'o') {
                if (am[2] == 0) {
                    return -1;
                } else {
                    am[2]--;
                    am[3]++;
                }
            } else if (c == 'a') {
                if (am[3] == 0) {
                    return -1;
                } else {
                    am[3]--;
                    am[4]++;
                }
            } else if (c == 'k') {
                if (am[4] == 0) {
                    return -1;
                } else {
                    am[4]--;
                }
            }
            int cur_frog = accumulate(am.begin(), am.end(), 0);
            result = max(result, cur_frog);
        }
        return result;
    }
};
