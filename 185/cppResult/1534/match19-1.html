class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int c_ = 0, r_ = 0, o_ = 0, a_ = 0, k_ = 0;
        int res = 0;
        int count = 0;
        for(auto &c : croakOfFrogs)
        {
            if(c == 'c')
            {

                c_++;
                count++;
            }
            else if(c == 'r')
            {
                if(c_ == 0) return -1;
                c_--;
                r_++;
            }
            else if(c == 'o')
            {
                if(r_ == 0) return -1;
                r_--;
                o_++;
            }
            else if(c == 'a')
            {
                if(o_ == 0) return -1;
                o_--;
                a_++;
            }
            else
            {
                if(a_ == 0) return -1;
                a_--;
                count--;
            }
            res = max(res, count);
        }
        if(!c_ && !r_ && !o_ && !a_ && !k_)
        {
            return res;
        }
        return -1;
    }
};
