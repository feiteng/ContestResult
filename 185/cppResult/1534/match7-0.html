class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int croaks = 0;
        int counting[10];
        memset(counting, 0, sizeof(counting));
        
        int maxi = 0;
        for (char c : croakOfFrogs) {
            if (c == 'c') {
                counting[0]++;
                
                
            } else if (c == 'r') {
                if (counting[0] == 0) return -1;
                counting[0]--;
                counting[1]++;
            } else if (c == 'o') {
                if (counting[1] == 0) return -1;
                counting[1]--;
                counting[2]++;
                
            } else if (c == 'a') {
                if (counting[2] == 0) return -1;
                counting[2]--;
                counting[3]++;
                
            } else if (c == 'k') {
                if (counting[3] == 0) return -1;
                counting[3]--;
                croaks++;
            }
            int t = 0;
                for (int i = 0; i < 5;i++) t+=counting[i];
                maxi = max(maxi, t);
        }
        
        for (int i = 0; i < 5; i++) if (counting[i] != 0) return -1;
        
        return maxi;
    }
};
