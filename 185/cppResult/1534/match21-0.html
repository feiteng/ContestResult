class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int c = 0, r = 0, o = 0, a = 0, ans = 0, maxAns = 0;
        for (int i = 0; i < croakOfFrogs.length(); ++i) {
            if (croakOfFrogs[i] == 'c') {
                ++ans;
                maxAns = max(ans, maxAns);
                ++c;
            } else if (croakOfFrogs[i] == 'r') {
                if (c <= 0) {
                    return -1;
                } else {
                    --c;
                    ++r;
                }
            } else if (croakOfFrogs[i] == 'o') {
                if (r <= 0) {
                    return -1;
                } else {
                    --r;
                    ++o;
                }
            } else if (croakOfFrogs[i] == 'a') {
                if (o <= 0) {
                    return -1;
                } else {
                    --o;
                    ++a;
                }
            } else if (croakOfFrogs[i] == 'k') {
                if (a <= 0) {
                    return -1;
                } else {
                    --a;
                    --ans;
                }
            }
        }
        if (c > 0 || r > 0 || o > 0 || a > 0) {
            return -1;
        }
        return maxAns;
    }
};
