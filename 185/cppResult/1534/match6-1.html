class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int ans = 0;
        
        map<char, int> m;
        
        
        for(char c : croakOfFrogs)
        {
            int temp = 0;
            for(auto p : m)
            {
                temp += p.second;
            }
            
            ans = max(ans, temp);
            

            if(c == 'c')
            {
                m[c]++;
            }
            else if(c == 'r')
            {
                if(m['c'] > 0)
                {
                    m['c']--;
                    m['r']++;
                }
                else
                {
                    return -1;
                }
            }
            else if(c == 'o')
            {
                if(m['r'] > 0)
                {
                    m['r']--;
                    m['o']++;
                }
                else
                {
                    return -1;
                }
            }
            else if(c == 'a')
            {
                if(m['o'] > 0)
                {
                    m['o']--;
                    m['a']++;
                }
                else
                {
                    return -1;
                }
            }
            else
            {
                if(m['a'] > 0)
                {
                    m['a']--;
                }
                else
                {
                    return -1;
                }
            }
        }
        
        for(auto p : m)
        {
            if(p.second != 0)
            {
                return -1;
            }
        }
        
        return ans;
    }
};
