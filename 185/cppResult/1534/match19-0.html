class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int res=0;
        vector<int> a(4);
        int s=0;
        for(char c:croakOfFrogs){
            if(c=='c'){
                ++a[0];
                res=max(res,++s);
            }
            else if(c=='r'){
                if(!a[0])
                    return -1;
                --a[0];
                ++a[1];
            }
            else if(c=='o'){
                if(!a[1])
                    return -1;
                --a[1];
                ++a[2];
            }
            else if(c=='a'){
                if(!a[2])
                    return -1;
                --a[2];
                ++a[3];
            }
            else{
                if(!a[3])
                    return -1;
                --a[3];
                --s;
            }
        }
        if(a[0]||a[1]||a[2]||a[3])
            return -1;
        return res;
    }
};
