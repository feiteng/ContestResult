int cnt[1001];
class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int tot=0;
        int c,r,o,a;
        c=r=o=a=0;
        int ans = 0;
        for(char d:croakOfFrogs)
        {
            if(d=='c')
            {
                c++;
            }
            else if(d=='r')
            {
                if(c==0)
                    return -1;
                c--;
                r++;
            }
            else if(d=='o')
            {
                if(r==0)
                    return -1;
                r--;
                o++;
            }
            else if(d=='a')
            {
                if(o==0)
                    return -1;
                o--;
                a++;
            }
            else if(d=='k')
            {
                if(a==0)
                    return -1;
                a--;
            }
            
            ans = max(ans,c+r+o+a);
            
        }
        
        if(c+r+o+a>0)
            return -1;
        return ans;
        
    }
};
