class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        map<char, int> cs;
        cs['c'] = cs['r'] = cs['o'] = cs['a'] = cs['k'] = 0;
        int res = 0;
        for(char c: croakOfFrogs){
            if(c == 'c'){
                cs[c]++;
                res = max(res, cs['c'] + cs['r'] + cs['o'] + cs['a'] + cs['k']);
            }else{
                if(c == 'r'){
                    if(cs['c'] > 0){
                        cs['c']--;
                        cs['r']++;
                    }else{
                        return -1;
                    }
                }else if(c == 'o'){
                    if(cs['r'] > 0){
                        cs['r']--;
                        cs['o']++;
                    }else{
                        return -1;
                    }
                }else if(c == 'a'){
                    if(cs['o'] > 0){
                        cs['o']--;
                        cs['a']++;
                    }else{
                        return -1;
                    }
                }else if(c == 'k'){
                    if(cs['a'] > 0){
                        cs['a']--;
                        //res += 1;
                    }else{
                        return -1;
                    }
                }else{
                    return -1;
                }
            }
        }
        if(cs['c'] == 0 && cs['r'] == 0 && cs['o'] == 0 && cs['a'] == 0 && cs['k'] == 0){
            return res;
        }else{
            return -1;
        }
    }
};
