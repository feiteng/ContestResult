#define mst(x, a) memset(x, a, sizeof(x))

using ll = long long;
using P = pair<int, int>;
using T = tuple<int, int, int>;
const int INF = 0x3f3f3f3f;
const ll LLINF = 0x3f3f3f3f3f3f3f3f;
const int MOD = 1e9 + 7;

int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

class Solution {
 public:
  int minNumberOfFrogs(string croakOfFrogs) {
    int cnt[5];
    mst(cnt, 0);
    int open = 0;
    int ans = 0; 
    for (char ch : croakOfFrogs) {
      if (ch == 'c') {
        open++;
        ans = max(ans, open);
        cnt[0]++;
      } else if (ch == 'r') {
        if (cnt[0]) {
          cnt[1]++;
          cnt[0]--;
        } else {
          return -1;
        }
      } else if (ch == 'o') {
        if (cnt[1]) {
          cnt[2]++;
          cnt[1]--;
        } else {
          return -1;
        }
      } else if (ch == 'a') {
        if (cnt[2]) {
          cnt[3]++;
          cnt[2]--;
        } else {
          return -1;
        }
      } else if (ch == 'k') {
        if (cnt[3]) {
          cnt[4]++;
          cnt[3]--;
          open--;
        } else {
          return -1;
        }
      } else {
        return -1;
      }
    }
    return open == 0 ? ans : -1;
  }
};
