class Solution {
public:
    int minNumberOfFrogs(string c)
    {
        if( c[c.length() - 1]!='k' )
            return -1;
        int cnt[5];
        memset( cnt, 0, sizeof(cnt) );
        vector<int> id;
        id.push_back(0);
        int idx = 0;
        for(auto i:c)
        {
            if( i=='c' )
                cnt[0]++;
            else if( i=='r' )
            {
                cnt[1]++;
                if( cnt[1]>cnt[0] )
                    return -1;
            }
            else if( i=='o' )
            {
                cnt[2]++;
                if( cnt[2]>cnt[1] || cnt[2]>cnt[0] )
                    return -1;
            }
            else if( i=='a' )
            {
                cnt[3]++;
                if( cnt[3]>cnt[2] || cnt[3]>cnt[1] || cnt[3]>cnt[0] )
                    return -1;
            }
            else if( i=='k' )
            {
                cnt[4]++, id.push_back(idx);
                if( cnt[4]>cnt[3] || cnt[4]>cnt[2] || cnt[4]>cnt[1] || cnt[4]>cnt[0] )
                    return -1;
            }
            else return -1;
            idx++;
        }
        int v = cnt[0];
        for(int i=0; i<5; i++)
            if( cnt[i]!=v )
                return -1;
        int ans = 0, tot = 0;
        for(int i=0; i<=(id.size()-2); i++)
        {
            int cntc = 0;
            for(int j=id[i]; j<=id[i+1]; j++)
                if( c[j]=='c' )
                    cntc++;
            tot += cntc;
            ans = max( ans, tot-i );
            
        }
        return ans;
    }
};
