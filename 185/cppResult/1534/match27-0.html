class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int cnt[5] = {0},ans = 0,t = 0;
        for(auto c : croakOfFrogs){
            if(c == 'c') cnt[0]++;
            else if(c == 'r'){cnt[1]++;if(cnt[1] > cnt[0]) return -1;}
            else if(c == 'o'){cnt[2]++;if(cnt[2] > cnt[1]) return -1;}
            else if(c == 'a'){cnt[3]++;if(cnt[3] > cnt[2]) return -1;}
            else{cnt[4]++;if(cnt[4] > cnt[3]) return -1;}
        }
        if(cnt[0] != cnt[1] || cnt[0] != cnt[2] || cnt[0] != cnt[3] || cnt[0] != cnt[4])
            return -1;
        for(auto c : croakOfFrogs){
            if(c == 'c') t++;
            if(c == 'k') ans = max(t,ans),t--;
        }
        return ans;
    }
};
