class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int c = 0, r = 0, o = 0, a = 0, k = 0;
        int ans = 0;
        for (char ch: croakOfFrogs) {
            if (ch == 'c') {

                ++c;
            }
            else if (ch == 'r') {
                if (!c) {
                    return -1;
                }
                --c;
                ++r;
            }
            else if (ch == 'o') {
                if (!r) {
                    return -1;
                }
                --r;
                ++o;
            }
            else if (ch == 'a') {
                if (!o) {
                    return -1;
                }
                --o;
                ++a;
            }
            else if (ch == 'k') {
                if (!a) {
                    return -1;
                }
                --a;
                ++k;
            }
            ans = max(ans, c + r + o + a);
        }
        if (c || r || o || a) {
            return -1;
        }
        return ans;
    }
};
