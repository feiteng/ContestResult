class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        if (croakOfFrogs.size() % 5 != 0) return -1;
        int cnt = 0;
        int cur[5] = {0};
        for (auto t: croakOfFrogs) {
            if (t == 'c') {
                if (cur[0] == 0) {
                    ++cnt;
                    ++cur[0];
                }
                --cur[0];
                ++cur[1];
            } else if (t == 'r') {
                if (cur[1] == 0) {
                    return -1;
                }
                --cur[1];
                ++cur[2];
            } else if (t == 'o') {
                if (cur[2] == 0) {
                    return -1;
                }
                --cur[2];
                ++cur[3];
            } else if (t == 'a') {
                if (cur[3] == 0) {
                    return -1;
                }
                --cur[3];
                ++cur[4];
            } else if (t == 'k') {
                if (cur[4] == 0) {
                    return -1;
                }
                --cur[4];
                ++cur[0];
            }
        }
        for (int i = 1; i < 5; ++i) {
            if (cur[i] > 0) return -1;
        }
        return cnt;
    }
};
