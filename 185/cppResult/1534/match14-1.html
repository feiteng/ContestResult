class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs)
    {
        int len = croakOfFrogs.size();
        vector<int> dp(5);
        int ans = 0;
        for (char c : croakOfFrogs) {
            if (c == 'c') {

                dp[0]++;
                ans = max(ans, dp[0]);
            }
            else if (c == 'r') {
                if (dp[0] > 0) {
                    dp[1]++;
                    ans = max(ans, dp[1]);
                }
                else {
                    return -1;
                }
            }
            else if (c == 'o') {
                if (dp[0] > 0 && dp[1] > 0) {
                    dp[2]++;
                    ans = max(ans, dp[2]);
                }
                else {
                    return -1;
                }
            }
            else if (c == 'a') {
                if (dp[0] > 0 && dp[1] > 0 && dp[2] > 0) {
                    dp[3]++;
                    ans = max(ans, dp[3]);
                }
                else {
                    return -1;
                }
            }
            else if (c == 'k') {
                if (dp[0] > 0 && dp[1] > 0 && dp[2] > 0 && dp[3] > 0) {
                    dp[0]--;
                    dp[1]--;
                    dp[2]--;
                    dp[3]--;
                }
                else {
                    return -1;
                }
            }
        }
        for (int i = 0; i < 4; ++i) {
            if (dp[i] > 0) {
                return -1;
            }
        }
        return ans;
    }
};
