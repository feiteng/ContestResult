class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {

        std::map<std::string,int> ma;
        int Max = -1;
        for (auto &it:croakOfFrogs){
            if (it=='c'){
                ma["c"]++;
            }
            else if (it=='r'){
                if (ma["c"]==0){
                    return -1;
                }
                ma["c"]--;
                ma["cr"]++;
            }
            else if (it=='o'){
                if (ma["cr"]==0){
                    return -1;
                }

                ma["cr"]--;
                ma["cro"]++;
            }
            else if (it=='a'){
                if (ma["cro"]==0) return -1;
                ma["cro"]--;
                ma["croa"]++;
            }
            else if (it=='k'){
                if (ma["croa"]==0) return -1;
                ma["croa"]--;
                ma["croak"]++;
            }
            int sum=0;
            for (auto &it1:ma){
                sum+=it1.second;
            }
            Max=max(Max,sum);
            ma["croak"]=0;
        }
        for (auto &it:ma){
            if (it.first!="croak"&&it.second>0) return -1;
        }
        return Max;
    }
};
