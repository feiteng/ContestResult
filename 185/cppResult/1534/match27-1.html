class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int cs=0,rs=0,os=0,as=0,ks=0;

        for(char c:croakOfFrogs){
            if(c=='c') cs++;
            else if(c=='r'){
                rs++;
                cout<<rs<<cs<<"\n";
                if(rs>cs) return -1;
            }else if(c=='o'){
                os++;
                cout<<os<<rs<<"\n";
                if(os>rs) return -1;
            }else if(c=='a'){
                as++;
                cout<<as<<os<<"\n";
                if(as>os) return -1;
            }else{
                ks++;
                cout<<ks<<as<<"\n";
                if(ks>as) return -1;
            }
        }
        if(!(cs==rs && rs==os && os==as && as==ks)) return -1;
        int res=croakOfFrogs.size()/5;
        if(res==1) return 1;
        int finished=0;
        for(char c:croakOfFrogs){
            if(c=='k') finished++;
            if(c=='c'){
                if(finished>0){
                    res--;
                    finished--;
                }
            }
        }
        return res;
    }
};
