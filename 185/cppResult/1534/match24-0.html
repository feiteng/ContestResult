class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        map<char, int> expect = {{'r', 0}, {'o', 0}, {'a', 0}, {'k', 0}};
        int cur = 0;
        int ans = 0;
        map<char, int> cnt;
        for (auto c: croakOfFrogs) {
            if (c == 'c') {
                cur ++;
                ans = max(ans, cur);
                expect['r'] ++;
            } else if (c == 'r') {
                if (expect[c] == 0) return -1; 
                expect['r'] --;
                expect['o'] ++;    
            } else if (c == 'o') {
                if (expect[c] == 0) return -1; 
                expect['o'] --;
                expect['a'] ++;
            } else if (c == 'a') {
                if (expect[c] == 0) return -1;
                expect['a'] --;
                expect['k'] ++;
            } else if (c == 'k') {
                if (expect[c] == 0) return -1; 
                expect['k'] --;
                cur --;
            }
        }
        if (expect['r'] + expect['o'] + expect['a'] + expect['k'] == 0) {
            return ans;
        }
        return -1;
    }
};
