class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        vector<int> v(5, 0);
        int res = 0;

        for (auto &i : croakOfFrogs) {
            if (i == 'c') {
                v[0]++;
            } else if (i == 'r') {
                if (v[0] == 0) return -1;
                v[0]--;
                v[1]++;
            } else if (i == 'o') {
                if (v[1] == 0) return -1;
                v[1]--;
                v[2]++;
            } else if (i == 'a') {
                if (v[2] == 0) return -1;
                v[2]--;
                v[3]++;
            } else {
                if (v[3] == 0) return -1;
                v[3]--;
                v[4]++;
            }
            res = max(res, v[0]+v[1]+v[2]+v[3]+v[4]);
            v[4] = 0;
        }
        if (v[0]+v[1]+v[2]+v[3]+v[4] == 0) return res;
        return -1;
    }
};
