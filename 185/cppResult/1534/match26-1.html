#define per(i,l,r) for(int i=(r)-1;i>=(l);--i)

class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
    	int ret = 0;
    	vector<int> dp(5, 0);

    	for (auto ch : croakOfFrogs) {
			switch(ch) {
			case 'c': ++dp[0]; break;
			case 'r': 
				if (dp[0] == 0) return -1;
				--dp[0], ++dp[1];
				break;
			case 'o': 
				if (dp[1] == 0) return -1;
				--dp[1], ++dp[2];
				break;
			case 'a': 
				if (dp[2] == 0) return -1;
				--dp[2], ++dp[3];
				break;
			case 'k': 
				if (dp[3] == 0) return -1;
				--dp[3];
				break;
			}
			int tmp = 0;
			rep(i, 0, 5) tmp += dp[i];
			ret = max(ret, tmp);
		}
		rep(i, 0, 5) if (dp[i] > 0) return -1;	
		return ret;
    }
};
