class Solution {
public:
    int minNumberOfFrogs(string a) {
      vector<int> d(128);       
      int ans = 0;
      for (auto c : a) {
        if (c == 'c') {

          ++d['c'];
        } else if (c == 'r') {
          --d['c'];
          if (d['c'] < 0) return -1;
          ++d['r'];
        } else if (c == 'o') {
          --d['r'];
          if (d['r'] < 0) return -1;
          ++d['o'];
        } else if (c == 'a') {
          --d['o'];
          if (d['o'] < 0) return -1;
          ++d['a'];
        } else if (c == 'k') {
          --d['a'];
          if (d['a'] < 0) return -1;
          int s = d['c'] + d['r'] + d['o'] + d['a'] + 1;
          ans = max(ans, s);
        } else {
          return -1;
        }
      }
      if (d['c'] || d['r'] || d['o'] || d['a']) return -1;
      return ans;
    }
};
