class Solution {
public:
    int minNumberOfFrogs(string s) {
        vector<int> ct(5,0);
        int ret=0;
        for (char ch : s) {
            switch (ch) {
                case 'c':
                    ct[0]++;
                    break;
                case 'r':
                    if(!ct[0])
                        return -1;
                    ct[0]--; ct[1]++;
                    break;
                case 'o':
                    if(!ct[1])
                        return -1;
                    ct[1]--; ct[2]++;
                    break;
                case 'a':
                    if(!ct[2])
                        return -1;
                    ct[2]--; ct[3]++;
                    break;
                case 'k':
                    if(!ct[3])
                        return -1;
                    ct[3]--; ct[4]++;
                    break;
                default:
                    return -1;
            }
            ret=max(ret,ct[0]+ct[1]+ct[2]+ct[3]+ct[4]);
            ct[4]=0;
        }
        for (int i=0; i<4; i++)
            if(ct[i])
                return -1;
        return ret;
    }
};
