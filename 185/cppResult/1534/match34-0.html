class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int dp[6]={0};
        int i,j,k;
        int num[6]={0};
        int ans=0;
        for(i=0;i<croakOfFrogs.size();i=j)
        {
            if(croakOfFrogs[i]!='c')
                return -1;
            j=i+1;
            num[0]++;
            ans=max(ans,1);
            while(j<croakOfFrogs.size())
            {
                if(croakOfFrogs[j]=='c')
                {
                	num[0]++;
                	ans=max(ans,num[0]);
                }
                else if(croakOfFrogs[j]=='k')
                    num[0]--;
                j++;
                if(num[0]==0)
                    break;
            }
            if(num[0])
                return -1;
            for(k=i;k<j;k++)
            {
                char x = croakOfFrogs[k];
                if(x=='c')
                {
                    dp[0]++;
                }
                else if(x=='r')
                {
                    if(dp[0]==0)
                        return -1;
                    dp[0]--;
                    dp[1]++;
                }
                else if(x=='o')
                {
                    if(dp[1]==0)
                        return -1;
                    dp[1]--;
                    dp[2]++;
                }
                else if(x=='a')
                {
                    if(dp[2]==0)
                        return -1;
                    dp[2]--;
                    dp[3]++;
                }
                else if(x=='k')
                {
                    if(dp[3]==0)
                        return -1;
                    dp[3]--;
                    dp[4]++;
                }
            }
            if(dp[0]||dp[1]||dp[2]||dp[3])
                return -1;
            dp[4]=0;
        }
        
        return ans;
    }
};
