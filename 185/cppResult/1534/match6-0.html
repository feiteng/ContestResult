int cnt[10];
class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int ret = 0;
        memset(cnt, 0, sizeof cnt);
        for(auto c:croakOfFrogs){
            if(c=='c'){
                cnt[0]++;
            }else if(c=='r'){
                if(cnt[0]>0){
                    cnt[0]--;
                    cnt[1]++;
                }else{
                    return -1;
                }
            }else if(c=='o'){
                if(cnt[1]>0){
                    cnt[1]--;
                    cnt[2]++;
                }else{
                    return -1;
                }
            }else if(c=='a'){
                if(cnt[2]>0){
                    cnt[2]--;
                    cnt[3]++;
                }else{
                    return -1;
                }
            }else{
                if(cnt[3]>0){
                    cnt[3]--;
                }else{
                    return -1;
                }
            }
            int temp = 0;
            for(int i=0;i<10;i++){
                temp+=cnt[i];
            }
            ret = max(ret, temp);
        }
        int total = 0;
        for(int i=0;i<10;i++)
            total+=cnt[i];
        if(total>0)
            return -1;
        return ret;
    }
};
