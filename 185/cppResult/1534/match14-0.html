class Solution {
    int c = 0; 
    int r = 0; 
    int o = 0; 
    int a = 0; 
    int k = 0; 
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        for(char i: croakOfFrogs){
            if(i == 'c'){
                if(k) k--;
                c++;
            }
            else if(i == 'r'){
                if(c){
                    c--;
                    r++;
                }
                else{
                    return -1;
                }
            }
            else if(i == 'o'){
                if(r){
                    r--;
                    o++;
                }
                else{
                    return -1;
                }
            }
            else if(i == 'a'){
                if(o){
                    o--;
                    a++;
                }
                else{
                    return -1;
                }
            }
            else if(i == 'k'){
                if(a){
                    a--;
                    k++;
                }
                else{
                    return -1;
                }
            }
        }
        if(!c && !r && !o && !a) return k;
        return -1;
    }
};
