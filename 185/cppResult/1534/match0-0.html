class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        map<string,int> mp;
        int res = 0;
        for(auto ch:croakOfFrogs){
            if(ch=='c'){
                ++mp["c"];
                if(mp["croak"]>0){
                    --mp["croak"];
                }
                else ++res;
            }
            else if(ch=='r'){
                --mp["c"];
                if(mp["c"]<0) return -1;
                ++mp["cr"];
            }
            else if(ch=='o'){
                --mp["cr"];
                if(mp["cr"]<0) return -1;
                ++mp["cro"];
            }
            else if(ch=='a'){
                --mp["cro"];
                if(mp["cro"]<0) return -1;
                ++mp["croa"];
            }
            else if(ch=='k'){
                --mp["croa"];
                if(mp["croa"]<0) return -1;
                ++mp["croak"];
            }
        }
        if(mp["c"]>0||mp["cr"]>0||mp["cro"]>0||mp["croa"]>0) return -1;
        return res;
    }
};
