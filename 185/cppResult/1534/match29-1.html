class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int dp[5] = {0};
        string c = croakOfFrogs;
        cout << c << "\n";
        int ans = 0;
        for (int x = 0; x < c.length(); x++) {
            if (c[x] == 'c') {

                dp[0]++;
            } else if (c[x] == 'r') {
                if (dp[0] == 0) {
                    return -1;
                }
                dp[0]--;
                dp[1]++;
            } else if (c[x] == 'o') {
                if (dp[1] == 0) {
                    return -1;
                }
                dp[1]--;
                dp[2]++;
            } else if (c[x] == 'a') {
                if (dp[2] == 0) {
                    return -1;
                }
                dp[2]--;
                dp[3]++;
            } else if (c[x] == 'k') {
                if (dp[3] == 0) {
                    return -1;
                }
                dp[3]--;
            } else {
                return -1;
            }
            ans = max (ans, dp[0] + dp[1] + dp[2] + dp[3] + dp[4]);
        }
        for (int x = 0; x < 5; x++) {
            if (dp[x] != 0) {
                return -1;
            }
        }
        return ans;
    }
};
