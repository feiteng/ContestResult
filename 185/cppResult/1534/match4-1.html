class Solution {
public:
    
    int minNumberOfFrogs(string croakOfFrogs) {
        int c=0,r=0,o=0,a=0,k=0;
        int ans = 0;
        for(auto cc:croakOfFrogs){
            if(cc=='c')c++;
            else if(cc=='r'){
                r++;
                if(r>c)return -1;

            }
            else if(cc=='o'){
                o++;
                if(o>c || o>r)return -1;
            }
            else if(cc=='a'){
                a++;
                if(a>o || a>c || a>r)return -1;
            }
            else if(cc=='k'){
                k++;
                if(k>c || k>r || k>o || k>a)return -1;
            }
            
            ans = max(ans,c - k);
            // cout<<c<<" "<<"ans = "<<ans<<endl;
        }
        if(c>k)return -1;
        return ans;
    }
};
