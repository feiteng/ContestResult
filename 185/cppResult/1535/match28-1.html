#define MOL 1000000007
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        if (k < 1)
            return 0;
        vector<vector<vector<long long>>> dp(n + 1, vector<vector<long long>>(m + 1, vector<long long>(k + 1, 0)));
        dp[0][0][0] = 1;

        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                for (int l = 1; l <= k; ++l) {
                    for (int jj = 0; jj < j; ++jj) {
                        dp[i][j][l] += dp[i - 1][jj][l - 1];
                        dp[i][j][l] %= MOL;
                    }
                    dp[i][j][l] += (dp[i - 1][j][l] * j) % MOL;

                    dp[i][j][l] %= MOL;
                }
            }
        }
        int ret = 0;
        for (int i = 1; i <= m; ++i) {
            ret += dp[n][i][k];
            ret %= MOL;
        }
        return ret;
    }
};
