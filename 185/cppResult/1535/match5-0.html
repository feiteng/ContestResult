class Solution {
    const int mod = 1e9+7;
public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<long long>>> dp(n, vector<vector<long long>>(m, vector<long long>(k)));
        for (int j = 0; j < m; j++) {
            dp[0][j][0] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int l = 0; l < k; l++) {
                    dp[i][j][l] = dp[i-1][j][l]*(j+1);
                    
                    if (l > 0) {
                        for (int jj = 0; jj < j; jj++) {
                            dp[i][j][l] += dp[i-1][jj][l-1];    
                        }    
                    }
                    dp[i][j][l] %= mod;
                    //out << dp[i][j][l] << " ";
                }
                //cout << "\n";
            }
        }
        long long ans= 0;
        for (int j = 0; j < m; j++){
            ans += dp[n-1][j][k-1];
        }
        return ans%mod;
    }
};
