using ll = long long;
const ll MOD = 1e9+7;
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<ll>>> dp(n+1, vector<vector<ll>>(m+1, vector<ll>(k+1)));

        for(int i=1;i<=m;i++) dp[1][i][1] = 1;
        for(int i=2;i<=n;i++){
            for(int j=1;j<=m;j++){
                for(int l=1;l<=k;l++){
                    if(j>1) dp[i-1][j-1][l-1] = (dp[i-1][j-1][l-1]  + dp[i-1][j-2][l-1]) % MOD;
                    dp[i][j][l] = (dp[i-1][j-1][l-1] + (dp[i-1][j][l]*j)%MOD) % MOD;

                }
            }
        }
        ll ans = 0;
        for(int i=1;i<=m;i++){
            ans = (ans + dp[n][i][k]) % MOD;
        }
        return ans;
    }
};
