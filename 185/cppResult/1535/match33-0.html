class Solution {
public:
    int Add(int a, int b) {
        const int MOD = 1000000007;
        return int((long(a) + long(b)) % MOD);
    }
    int numOfArrays(int n, int m, int k) {
        int X[51][101][51] = { 0 };
        // idx=0
        for (int i=1; i<=m; i++) {
            X[0][i][1] = 1;
        }
        for (int i=1; i<n; i++) {
            for (int elt = m; elt >= 1; elt--) {
                for (int elt1 = m; elt1 >= elt; elt1 --) {
                    for (int prev_cost = k; prev_cost >= 1; prev_cost --) {
                        X[i][elt1][prev_cost] = Add(X[i][elt1][prev_cost], X[i-1][elt1][prev_cost]);
                    }
                }
                for (int elt1 = elt-1; elt1 >= 0; elt1 --) {
                    for (int prev_cost = k; prev_cost >= 1; prev_cost --) {
                        X[i][elt][prev_cost+1] = Add(X[i][elt][prev_cost+1], X[i-1][elt1][prev_cost]);
                    }
                }
            }
        }
        int ret = 0;
        for (int elt = m; elt >= 1; elt --) {
            ret = Add(ret, X[n-1][elt][k]);
        }

		if (0) {
			for (int i=0; i<n; i++) {
				printf("i=%d\n", i);
				for (int e=1; e<=m; e++) {
					printf("  e=%d  ", e);
					for (int c=1; c<=k; c++) {
						printf(" %d", X[i][e][c]);
					}
					printf("\n");
				}
			}
		}

        return ret;
    }
};
