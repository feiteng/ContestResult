#define mod 1000000007
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        int A[n+5][k+5][m+5];
        for(int i=0;i<=n;++i)
        {
            for(int j=0;j<=k;++j)
            {
                for(int p=0;p<=m;++p)
                {
                    A[i][j][p]=0;
                }
            }
        }
        A[0][0][0]=1;
        for(int i=0;i<n;++i)
        {
            
                for(int j=0;j<=k;++j)
                {
                    for(int p=0;p<=m;++p)
                    {
                        for(int q=1;q<=m;++q)
                        {
                            if(q>p)
                            {
                                A[i+1][j+1][q]=(A[i+1][j+1][q]+A[i][j][p])%mod;
                            }
                            else{
                                A[i+1][j][p]=(A[i+1][j][p]+A[i][j][p])%mod;
                            }
                        }
                    }
                }
            
        }
        int ans=0;
        for(int i=1;i<=m;++i)
        {
            ans=(ans+A[n][k][i])%mod;
        }
        return ans;
    }
};
