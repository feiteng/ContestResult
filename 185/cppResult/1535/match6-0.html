class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        long long mod = 1000000007ll;
        long long dp[55][105][55];
        memset(dp, 0, sizeof(dp));
        for (int i = 1; i <= m; i++) {
            dp[0][i][1] = 1;
        }
        
        for (int i = 1; i < n; i++) {
            for (int mx = 1; mx <= m; mx++) {
                for (int c = 1; c <= k; c++) {
                    for (int cur = 1; cur < mx; cur ++) {
                        dp[i][mx][c] = (dp[i][mx][c] + dp[i-1][cur][c-1]) % mod;
                    }
                    
                    dp[i][mx][c] = (dp[i][mx][c] + (dp[i-1][mx][c] * mx) % mod) % mod;
                }
            }       
        }
        
        long long sum = 0;
        for (int mx = 1; mx <= m; mx++) {
            sum = (sum + dp[n-1][mx][k]) % mod;
        }
        return sum;
    }
    
};
