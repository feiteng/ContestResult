class Solution {
public:
  int numOfArrays(int n, int m, int k) {
    long long ways[51][101][51] = {1, 0};

    for (int i = 1; i <= n; i++) {
      for (int j = 1; j <= m; j++) {
        for (int l = 1; l <= k; l++) {
          for (int s = 0; s < j; s++) {
            ways[i][j][l] += ways[i - 1][s][l - 1];
          }
          ways[i][j][l] += ways[i - 1][j][l] * j;
          ways[i][j][l] %= 1000000007;
          // for (int s = j; s <= m; s++)
          //   ways[i][j][l] += ways[i - 1][s][l];
          // cout << i << j << l << ways[i][j][l] << endl;
        }
      }
    }
    long long ans = 0;
    for (int j = m; j; j--) {
      ans += ways[n][j][k];
    }
    ans %= 1000000007;
    return ans;
  }
};
