typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  ll dp[111][111][111];
  int numOfArrays(int N, int M, int K) {
    memset(dp,0,sizeof dp);
    for(int j=1;j<=M;j++) {
      dp[1][j][1] = 1;
    }
    for (int i=2;i<=N;i++){
      for(int j=1;j<=M;j++){
        for(int k=1;k<=K;k++) {
          dp[i][j][k] = (dp[i-1][j][k]*j)%MOD;
          for (int l=1;l<j;l++) {
            dp[i][j][k] += dp[i-1][l][k-1];
            if (dp[i][j][k]>=MOD) dp[i][j][k]-=MOD;
          }
        }
      }
    }
    ll res=0;
    for(int j=1;j<=M;j++) res = (res+dp[N][j][K])%MOD;
    return res;
  }
};
