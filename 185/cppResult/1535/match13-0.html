#define VI vector<int>
#define VII vector<VI>
#define VIII vector<VII>
#define VS vector<string>
#define LL uint64_t
const int MOD=1e9+7;
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        VIII dp(n,VII(m+1,VI(n+1,0)));
        for(int val=1;val<=m;val++){
            dp[0][val][1]=1;
        }
        for(int i=1;i<n;i++){
            for(int kk=1;kk<=k&&kk<=(i+1);kk++){
                for(int val=1;val<=m;val++){
                    for(int last=1;last<=m;last++){
                        if(last<val){
                            dp[i][val][kk]+=dp[i-1][last][kk-1];
                            dp[i][val][kk]%=MOD;
                        }else{
                            dp[i][last][kk]+=dp[i-1][last][kk];
                            dp[i][last][kk]%=MOD;
                        }
                    }
                }
            }
        }
        int ans=0;
        for(int val=1;val<=m;val++){
            ans+=dp[n-1][val][k];
            ans%=MOD;
        }
        return ans;
    }
};
