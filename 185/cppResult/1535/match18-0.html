typedef long long ll;

const ll p = 1e9 + 7;

const int maxn = 59;
const int maxm = 109;
const int maxk = maxn;

ll ways[maxn][maxm][maxn];

class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        if (k == 0) return 0;
        memset(ways, 0, sizeof(ways));
        for (int prev = 1; prev <= m; prev++) {
            ways[n - 1][prev][0] = 1;
        }
        for (int i = n - 2; i >= 0; i--) {
            for (int prev = 1; prev <= m; prev++) {
                for (int rem = 0; rem <= k - 1; rem++) {
                    for (int cur = 1; cur <= m; cur++) {
                        int cost = (cur > prev);
                        if (rem - cost >= 0) {
                            ways[i][prev][rem] = (ways[i][prev][rem] + ways[i + 1][max(cur, prev)][rem - cost]) % p;
                        }
                    }
                    //cout << i << " " << prev << " " << rem << " " << ways[i][prev][rem] << endl;
                }
            }
        }        
        ll ans = 0;
        for (int prev = 1; prev <= m; prev++) {
            //cout << ways[0][prev][k - 1] << endl;
            ans = (ans + ways[0][prev][k - 1]) % p;
        }
        return ans;
    }
};
