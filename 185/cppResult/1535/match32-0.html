constexpr uint64_t kMod = 1e9+7;
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<int>> dp(m+1, vector<int>(k+1, 0)), dp2 = dp;
        for (int i = 1; i <= m; ++i) {
            dp[i][1] = 1;
        }
        for (int i = 1; i < n; ++i) {
            dp2 = dp;
            for (int tm = 1; tm <= m; ++tm) {
                for (int tk = 1; tk <= k; ++tk) {
                    int lt = 0;
                    for (int d = 1; tm-d >= 1; ++d) lt = (lt + dp[tm-d][tk-1]) % kMod;
                    dp2[tm][tk] = (lt + ((uint64_t)dp[tm][tk]*tm%kMod)) % kMod; 
                }
            }
            
            swap(dp2, dp);
        }
        
        int ans = 0;
        for (int i = 1; i <=m; ++i) {
            ans = (ans + dp[i][k]) % kMod;
        }
        return ans;
    }
};
