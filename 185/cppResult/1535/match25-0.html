class Solution {
public:
    int dp[55][105][55];
    const int mod = 1e9 + 7;
    int numOfArrays(int n, int m, int k) {
        dp[0][0][0] = 1;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j <= m; j++) {
                for(int h = 0; h <= k; h++) {
                    for(int l = j + 1; l <= m; l++) {
                        dp[i + 1][l][h + 1] = (dp[i + 1][l][h + 1] + dp[i][j][h]) % mod;
                    }
                    for(int l = 1; l <= j; l++) {
                        dp[i + 1][j][h] = (dp[i + 1][j][h] + dp[i][j][h]) % mod;
                    }
                }
            }
        }
        int ans = 0;
        for(int i = 1; i <= m; i++)
            ans = ( ans + dp[n][i][k]) % mod;
        return ans;
    }
};
