class Solution {
public:

    long long MOD = 1e9 + 7; 
    int numOfArrays(int n, int m, int k) {
        long long dp[55][105][55];
        
        memset(dp, 0, sizeof(dp));
        
        dp[0][0][0] = 1;
        
        for(int i = 0; i < n; ++i)
        {
            for(int end = 1; end <= m; ++end)
            {
                for(int cost = 0; cost <= k; ++cost)
                {
                    for(int maxval = 0; maxval <= m; ++maxval)
                    {
                        if(end > maxval)
                        {
                            // if this end > maxval, then cost will increase 1, and maxval will be changed
   
                            dp[i+1][end][cost + 1] += dp[i][maxval][cost]; 
                            
                            if(dp[i+1][end][cost + 1] >= MOD)
                            {
                                dp[i+1][end][cost + 1] -= MOD;
                            }
                               
                        }
                        else
                        {
                            // if this end <= maxval, then cost will not increase, and maxval stay unchanged
                            dp[i+1][maxval][cost] += dp[i][maxval][cost];
                            if(dp[i+1][maxval][cost] >= MOD)
                            {
                                dp[i+1][maxval][cost] -= MOD;
                            }

                        }             
                    }
                }
            }
        }
        
        long long ans = 0;
        for(int end = 1; end <= m; ++end)
        {
            ans += dp[n][end][k];
            if(ans >= MOD)
            {
                ans -= MOD;
            }
        }
        
        return ans;
    }
};
