class Solution {
public:
    const int mod = 1e9 + 7;
    int dp[51][101][101];
    int numOfArrays(int n, int m, int k) {
        if(k == 0) return 0;

        dp[0][0][0] = 1;
        for(int t = 1; t <= n; t++){
            for(int maxVal = 1; maxVal <= m; maxVal++){
                for(int cost = 1; cost <= k; cost++){

                    dp[t][maxVal][cost] = (1ll * dp[t - 1][maxVal][cost] * maxVal) % mod;
                    for(int i = maxVal - 1; i >= 0; i--){
                        dp[t][maxVal][cost] = (dp[t][maxVal][cost] + dp[t - 1][i][cost - 1]) % mod;
                    }
                }
            }
        }
        // cout << dp[2][2][1] << endl;
        int res = 0;
        for(int i = 1; i <= m; i++) res = (res + dp[n][i][k]) % mod;
        return res;
    }
};
