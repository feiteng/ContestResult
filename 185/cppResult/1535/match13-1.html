class Solution {
public:
    int Q = 1e9 + 7;
    int f[55][105][55];
    int numOfArrays(int n, int m, int K) {
        f[0][0][0] = 1;

        for (int i = 0 ; i < n ; ++ i) {
            for (int j = 0 ; j <= m ; ++ j) {
                for (int k = 0 ; k <= i ; ++ k) {
                    if (!f[i][j][k]) continue;

                    for (int l = 1 ; l <= m ; ++ l) {
                        if (l > j) {
                            f[i + 1][l][k + 1] += f[i][j][k];
                            f[i + 1][l][k + 1] %= Q;
                        } else {
                            f[i + 1][j][k] += f[i][j][k];
                            f[i + 1][j][k] %= Q;
                        }
                        
                    
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0 ; i <= m ; ++ i) {
            res += f[n][i][K];
            res %= Q;
        }
        return res;
    }
};
