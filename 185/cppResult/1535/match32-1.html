class Solution {
    int add(int x, int y) const {
        return (x + y) % 1000000007;
    }
public:
    int numOfArrays(int n, int m, int k) {
        if (k == 0 || k > n) return 0;
        
        vector<vector<vector<int>>> dp(n, vector<vector<int>>(m + 1, vector<int>(n + 1, 0)));

        for (int x = 1; x <= m; x++) {
            dp[0][x][1] = 1;
        }
        
        for (int i = 1; i < n; i++) {

            for (int x = 1; x <= m; x++) {
                for (int c = 1; c <= i + 1; c++) {
                    int f = 0;
                    for (int y = 1; y < x; y++) {
                        f = add(f, dp[i - 1][y][c - 1]);
                    }
                    for (int _ = 0; _ < x; _++) {
                        f = add(f, dp[i - 1][x][c]);
                    }

                    dp[i][x][c] = f;
                }
            }
        }
        
        int ans = 0;
        for (int x = 1; x <= m; x++) {
            ans = add(ans, dp[n - 1][x][k]);
        }
        return ans;
    }
};
