class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        const long long mod = 1e9+7;
        using ll = long long;
        ll dp[60][110][60];
        memset(dp,0,sizeof(dp));
        for (int j=1;j<=m;j++){
            dp[1][j][1]=1;
        }
        for (int i=2;i<=n;i++){
            for (int j=1;j<=m;j++){
                for (int k_=1;k_<=k;k_++){
                    for (int now=1;now<=m;now++){
                        if (now>j){
                            dp[i][now][k_+1]=(dp[i][now][k_+1]+dp[i-1][j][k_])%mod;
                        }
                        else{
                            dp[i][j][k_]=(dp[i][j][k_]+dp[i-1][j][k_])%mod;
                        }
                    }
                }
            }
        }
        ll res = 0;
        for (int j=1;j<=m;j++){
            res=(res+dp[n][j][k])%mod;
        }
        return res;
    }
};
