class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        long long dp[n+1][m+1][k+1];
        memset(dp,0LL,sizeof dp);
        int mod = (int) 1e9 + 7;
        
        for (int j = 1; j <= m;j++){   
            dp[1][j][1] = 1LL;
        }
        
        for(int i = 2; i<= n; i++){
            for(int j = 1; j <= m; j++){
                for(int c = 1; c<= k; c++){
                    
                    // n - 1 length, ends with less number than the current element
                    for(int jj = 1; jj < j; jj++){
                        dp[i][j][c] += dp[i-1][jj][c-1];
                        dp[i][j][c] %= mod;
                    }
                    // n - 1 length, ends with equal or larger element thant the current element
                    for(int jj = j ; jj <= m; jj++){
                        dp[i][jj][c] += dp[i-1][jj][c];
                        dp[i][jj][c] %= mod;
                    }
                }
            }
        }
        
        long long ans = 0LL;
        for(int j = 1; j <= m; j++){
            ans += dp[n][j][k];
            ans %= mod;
        }
        return ans;
        
    }
};
