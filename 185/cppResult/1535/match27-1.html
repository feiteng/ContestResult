long long dp[55][105][105];

class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        if (k>m or k > n) return 0;
        long long P=1000000007;
        memset(dp,0,sizeof(dp));

        for (int i=1;i<=n;++i) {
            for (int j=1;j<=m;++j) {
                for (int k1=1;k1<=k;++k1) {
                    if (i==1 &&k1==1) {
                        dp[i][j][k1]=1;continue;
                    }

                    for (int j2=1;j2<j;j2++) {
                        dp[i][j][k1] += dp[i-1][j2][k1-1];
                        dp[i][j][k1]%=P;
                    }
                    dp[i][j][k1]+=dp[i-1][j][k1]*j;
                    dp[i][j][k1]%=P;
                    //printf("dp[%d][%d][%d]:%d\n",i,j,k1,dp[i][j][k1]);
                }
            }
        }
        long long ans=0;
        for (int i=1;i<=m;++i)ans+=dp[n][i][k],ans%=P;
        return ans;
        
    }
};
