class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        const int MOD = 1000000007;
        vector<vector<vector<int>>> f(n+3);
        for(int i=0; i<n+3; i++) {
            f[i].resize(m+3);
            for(int j=0; j<m+3; j++) {
                f[i][j].resize(k+3, 0);
            }
        }
        for(int num=1; num<=m; num++) {
            f[0][num][1] = 1;
        }

        for(int i=0; i<n-1; i++) {
            for(int j=1; j<=m; j++) {
                for(int cost=1; cost<=k; cost++) {
                    for(int num=1; num<=m; num++) {
                        if(num<=j) {
                            f[i+1][j][cost]+=f[i][j][cost];
                            f[i+1][j][cost]%=MOD;
                        } else {
                            f[i+1][num][cost+1]+=f[i][j][cost];
                            f[i+1][num][cost+1]%=MOD;
                        }
                    }
                }
            }
        }

        int result = 0;
        for(int j=1; j<=m; j++) {
            result+=f[n-1][j][k];
            result%=MOD;
        }
        return result;
    }
};
