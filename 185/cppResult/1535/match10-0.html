typedef long long ll;
const ll mod = 1e9 + 7;

class Solution {
public:
    
    ll DP[55][105][105];
    
    int numOfArrays(int N, int M, int K) {
        memset(DP , 0 , sizeof(DP));
        DP[0][0][0] = 1;
        for(int i = 0; i < N; ++i){
            for(int j = 0; j <= M; ++j){
                for(int k = 0; k <= K; ++k){
                    for(int x = 1; x <= M; ++x){
                        if(!DP[i][j][k])continue;
                        if(x > j){
                            DP[i + 1][x][k + 1] = (DP[i + 1][x][k + 1] + DP[i][j][k]) % mod;
                        }
                        else{
                            DP[i + 1][j][k] = (DP[i + 1][j][k] + DP[i][j][k]) % mod;
                        }
                    }
                }
            }
        }
        ll ret = 0;
        for(int j = 1; j <= M; ++j){
            ret = (ret + DP[N][j][K]) % mod;
        }
        return ret;
    }
};
