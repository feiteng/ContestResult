class Solution {
public:
    using ll = long long;
    ll mod =  1000000007;
    ll dp[55][105][55];
//     dp(n,m,n) = m-n+1;
//     dp[n][j][k] = dp[n-1][j~m][k] + dp[n-1][1~j-1][k-1] 
//     dp(i,j,t) = dp(n-1,1~j-1,t-1) + dp(n-1,j,t)*(j-1)
    int numOfArrays(int n, int m, int k) {
        if(k>m || k>n)return 0;

        memset(dp,0,sizeof(dp));
        for(int j=1;j<=m;j++){
            dp[0][j][1]=1;
        }
        for(int i=1;i<n;i++){
            for(int j=1;j<=m;j++){
                for(int t = 1;t<=i+1 && t<=k;t++){
                    // if(n==2)cout<<i<<" "<<j<<" "<<t<<endl;
                    for(int j2 = 1;j2<j;j2++)
                        dp[i][j][t] = ( dp[i][j][t] + dp[i-1][j2][t-1])%mod;
                    dp[i][j][t] = (dp[i][j][t] + dp[i-1][j][t]*(j))%mod;

                } 
            }
        }
        ll ans = 0;
        for(int j=0;j<=m;j++){
            // if(n==2)cout<<dp[n-1][j][k]<<endl;
            ans = (ans + dp[n-1][j][k])%mod;
        }
        return ans;
    }
};
