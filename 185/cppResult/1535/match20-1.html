class Solution {
public:
    typedef long long ll;
    #define MOD 1000000007
    ll dp[51][101][51];

    int numOfArrays(int n, int m, int l) {
        memset(dp,0,sizeof(dp));
        for(int i=1;i<=m;i++)dp[1][i][1]=1;
        for(int i=2;i<=n;i++){
            for(int j=1;j<=m;j++){
                for(int k=1;k<=l;k++){
                    dp[i][j][k]=(dp[i-1][j][k]*j)%MOD;
                    if(k>0)for(int l=1;l<j;l++)dp[i][j][k]=(dp[i][j][k]+dp[i-1][l][k-1])%MOD;

                }
            }
        }
        ll res=0;
        for(int i=1;i<=m;i++)res=(res+dp[n][i][l])%MOD;
        return res;
    }
};
