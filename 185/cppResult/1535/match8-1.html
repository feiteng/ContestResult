class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        long long dp[51][101][51] = {0};
        int mod = 1e9 + 7;
        dp[0][0][0] = 1;

        for (int i=0; i<n; i++) {
            for (int j=0; j<=m; j++) {
                for (int x=0; x<=k; x++) {
                    dp[i+1][j][x] = (dp[i+1][j][x] + j * dp[i][j][x]) % mod;

                    for (int y=j+1; y<=m; y++) dp[i+1][y][x+1] = (dp[i+1][y][x+1] + dp[i][j][x]) % mod;
                }
            }
        }
        long long ans = 0;
        for (int i=k; i<=m; i++) ans = (ans + dp[n][i][k]) % mod;;
        return ans;
    }
};
