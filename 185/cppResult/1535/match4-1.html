class Solution {
public:
    int dp[51][101][51];
    int numOfArrays(int n, int m, int k) {
        memset(dp, 0, sizeof(dp));

        for(int i = 1; i <= m ;++i) {
            dp[0][i][1] = 1;
        }
        for(int i = 0; i < n; ++i) {
            for(int j = 1; j <= m; ++j) {
                for(int kk = 1; kk <= k; ++kk) {
                    for(int c = 1; c <= m; ++c) {
                        if(c > j) {
                            dp[i+1][c][kk+1] += dp[i][j][kk];
                            dp[i+1][c][kk+1] %= 1000000007;
                        } else {
                            dp[i+1][j][kk] += dp[i][j][kk];
                            dp[i+1][j][kk] %= 1000000007;
                        }
                        
                    }

                }
            }
        }
        int re = 0;
        for(int i = 1; i <= m; ++i) {
            re += dp[n-1][i][k];
            re %= 1000000007;
        }
        return re;
    }
};
