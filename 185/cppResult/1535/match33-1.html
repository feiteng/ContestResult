class Solution {
public:
    int numOfArrays(int N, int M, int K) {
        const int MOD = 1e9 + 7;
        if (K > M) return 0;
        
        long long dp[51][101][51];

        for (int i=0; i<=N; i++) {
            for (int j=0; j<=M; j++) {
                for (int k=0; k<=K; k++) {
                    dp[i][j][k] = 0;
                }
            }
        }
        
        dp[0][0][0] = 1;

        for (int i=1; i<=N; i++) {
            for (int j=1; j<=M; j++) {
                for (int k=1; k<=K; k++) {
                    dp[i][j][k] = (dp[i-1][j][k] * j) % MOD;

                    for (int m=0; m<j; m++) {
                        dp[i][j][k] = (dp[i][j][k] + dp[i-1][m][k-1]) % MOD;
                    }
                    // cout << i << " " << j << " " << k << ": " << dp[i][j][k] << '\n';
                }
            }
        }
        
        int ans = 0;
        for (int i=0; i<=M; i++) {
            ans = (ans + dp[N][i][K]) % MOD;
        }
        return ans;
    }
};
