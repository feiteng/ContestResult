class Solution {

public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<long long>>> dp(n, vector<vector<long long>>(m+1, vector<long long>(k+1, 0)));
        for(int i=1; i<=m; i++){
            dp[0][i][1] = 1;
        }
        for(int i=1; i<n; i++){
            for(int j=1; j<=m; j++){
                for(int d=1; d<=k; d++){
                    long long& ref = dp[i][j][d];
                    ref = dp[i-1][j][d] * j;
                    for(int t=1; t<j; t++){
                        ref += dp[i-1][t][d-1];

                    }
                    ref %= 1000000007;
                }
            }
        }
        long long res = 0;
        for(int i=1; i<=m; i++){
            res += dp[n-1][i][k];
        }
        return res % 1000000007;
    }
};
