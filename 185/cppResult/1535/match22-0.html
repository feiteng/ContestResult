class Solution {
public:
    const int mod=1e9+7;
    long long dp[55][105][55];
    int numOfArrays(int n, int m, int k) {
      memset(dp,0,sizeof(dp));
      for(int i=1;i<=m;i++) dp[1][i][1]=1;
      for(int i=2;i<=n;i++){
        for(int j=1;j<=m;j++){
          for(int jj=1;jj<j;jj++){
            for(int kk=2;kk<=i;kk++){
              dp[i][j][kk]+=dp[i-1][jj][kk-1];
              dp[i][j][kk]%=mod;
            }
          }
          for(int jj=j;jj<=m;jj++){
            for(int kk=1;kk<=i;kk++){
              dp[i][jj][kk]+=dp[i-1][jj][kk];
              dp[i][jj][kk]%=mod;
            }
          }

        }
      }
      long long ans=0;
      for(int i=1;i<=m;i++){
        ans+=dp[n][i][k];
        ans%=mod;
      }
      cout<<ans<<endl;
      return ans;
    }
};
