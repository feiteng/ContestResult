const int mod = 1000000007;
int f[128][128][128]; // have i numbers, max j, k cost
class Solution {
 public:
  int numOfArrays(int n, int m, int cost) {
    memset(f, 0, sizeof(f));
    f[0][0][0] = 1;
    for (int i = 0; i < n; i++) {
      for (int j = 0; j <= m; j++) {
        for (int k = 0; k <= cost; k++) {
          if (f[i][j][k] == 0) continue;
          for (int h = 1; h <= m; h++) {
            if (h <= j) {
              f[i + 1][j][k] = (f[i][j][k] + 0LL + f[i + 1][j][k]) % mod;
            } else {
              f[i + 1][h][k + 1] = (f[i][j][k] + 0LL + f[i + 1][h][k + 1]) % mod;
            }
          }
        }
      }
    }
    long long ret = 0;
    for (int i = 1; i <= m; i++) {
      ret = (ret + 0LL + f[n][i][cost]) % mod;
    }
    return ret;
  }
};
