class Solution {
public:
    static const int mod = 1e9+7;
    int dp[51][101][51];
    int numOfArrays(int n, int m, int k) {
        for(int i=1;i<=m;i++)dp[1][i][1]=1;
        for(int i=1;i<n;i++){
            for(int j=1;j<=m;j++){
                for(int p=1;p<=i;p++)if(dp[i][j][p]){
                    dp[i+1][j][p] = (dp[i+1][j][p]+1ll*j*dp[i][j][p])%mod;
                    for(int l=j+1;l<=m;l++){
                        dp[i+1][l][p+1] = (dp[i+1][l][p+1]+dp[i][j][p])%mod;
                    }
                }
            }
        }
        int ans=0;
        for(int i=1;i<=m;i++)ans=(ans+dp[n][i][k])%mod;
        return ans;
    }
};
