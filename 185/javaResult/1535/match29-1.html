class Solution {
    Long[][][] dp;
    int mod = (int)(1e9 + 7);
    
    public int numOfArrays(int n, int m, int k) {
        dp = new Long[n + 1][101][n + 1];
        
        return recurse(n, m, k, 0, 0, 0);
    }
    
    private int recurse(int n, int m, int k, int ind, int max, int searchCost) {
        if (ind == n) {
            return searchCost == k ? 1 : 0;
        
}
        if (dp[ind][max][searchCost] != null) {
            return (int)((long)dp[ind][max][searchCost]);
        }
        long ans = 0;
        
        for (int i = 1; i <= m; i++) {
            
            if (i > max) {
                ans = (ans + recurse(n, m, k, ind + 1, i, searchCost + 1)) % mod;
            }
            else {
                ans = (ans + 
recurse(n, m, k, ind + 1, max, searchCost)) % mod;
            }
        }
        
        dp[ind][max][searchCost] = ans;
        return (int)ans;
    }
}
