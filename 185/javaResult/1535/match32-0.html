class Solution {
    public int numOfArrays(int n, int m, int k) {
        if (k > m) {
            return 0;
        }
        long MOD = 1000000007L;
        long dp[][][] = new long[n+1][m+1][k+1];
        for (int i = 1; i <=m; i++) {
            dp[1][i][1] = 1;
        }
        for (int i = 2; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int p = 1; p <= k; p++) {
                    dp[i][j][p] = dp[i-1][j][p] * j % MOD;
                    for (int q = 1; q < j; q++) {
                        dp[i][j][p] = (dp[i][j][p] + dp[i-1][q][p-1]) % MOD;
                    }
                }
            }
        }
        long res = 0;
        for(int i = 1; i <= m; i++) {
            res += dp[n][i][k];
        }
        return (int) (res % MOD);
    }
}
