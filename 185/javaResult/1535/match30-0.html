class Solution {
    /**
    2 3 1
    
    dp[1][1][1] = 1
    dp[2][1][1] 
    
    
    
    **/
    private final int MOD = 1000000007;
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n+1][m+1][k+1];
        
        for (int i =1; i <= m; i++) {
            dp[1][i][1] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                // for (int val = 0; val < j; val++) {
                //     for (int cost = 0; cost < k; cost++) {
                //         dp[i][j][cost+1] = (dp[i-1][val][cost+1] + dp[i][j][cost+1]) % MOD;
                //     }
                // }
                // for (int val = j+1; val <= m; val++) {
                //     for (int cost = 0; cost <= k; cost++) {
                //         dp[i][val][cost] = (dp[i-1][val][cost] + dp[i][val][cost]) % MOD;
                //     }
                // }
                for (int cost = 0; cost <= k; cost++) {
                    dp[i+1][j][cost] = (dp[i+1][j][cost] + dp[i][j][cost] * (j)) % MOD;
                    if (cost != k) {
                    for (int max = j+1; max <= m; max++) {
                        dp[i+1][max][cost+1] = (dp[i][j][cost] + dp[i+1][max][cost+1])%MOD;
                    }
                    }
                }
                
            
            }
            
            for (int cost = 0; cost <= k; cost++) {
                dp[i+1][m][cost] = (dp[i][m][cost] * m + dp[i+1][m][cost]) % MOD; 
            }
        }
        
        long res = 0L;
        for (int i =1; i<= m; i++) res = (res + dp[n][i][k])% MOD;
        
        return (int)res;
        // return (int)dp[2][3][1];
    }
}
