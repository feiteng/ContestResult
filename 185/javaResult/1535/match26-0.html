class Solution {
    public int numOfArrays(int n, int m, int target) {
        long[][][] dp = new long[55][105][55];
        for (int j = 1; j <= m; j++) {
            dp[0][j][1] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int k = 1; k <= target; k++) {
                    dp[i][j][k] += (dp[i - 1][j][k] * j) % (1e9 + 7);
                    dp[i][j][k] %= (1e9 + 7);
                    for (int tj = 1; tj < j; tj++) {
                        dp[i][j][k] += dp[i - 1][tj][k - 1] % (1e9 + 7);
                        dp[i][j][k] %= (1e9 + 7);
                    }
                }
            }
        }
        long ans = 0;
        for (int j = 1; j <= m; j++) {
            ans += dp[n - 1][j][target] % (1e9 + 7);
            ans %= 1e9 + 7;
        }
        return (int) ans;
    }
}
