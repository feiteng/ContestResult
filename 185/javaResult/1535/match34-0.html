class Solution {
    public int numOfArrays(int n, int m, int k) {
        long BASE = (long)1e9 + 7;
        
        long[][][] dp = new long[n][k + 1][m + 1];
        
        for (int j = 1; j <= m; j++) {
            dp[0][1][j] = 1;
        }
        for (int i = 0; i < n - 1; i++) {
            for (int k0 = 1; k0 <= k; k0++) {
                for (int j = 1; j <= m; j++) {
                    dp[i + 1][k0][j] += (dp[i][k0][j] * j) % BASE;
                    dp[i + 1][k0][j] %= BASE;
                    
                    if (k0 == k) {
                        continue;
                    }
                    for (int jx = j + 1; jx <= m; jx++) {
                        dp[i + 1][k0 + 1][jx] += dp[i][k0][j];
                        dp[i + 1][k0 + 1][jx] %= BASE;
                    }
                }
            }
        }
        
        long sum = 0;
        for (int j = 1; j <= m; j++) {
            sum += dp[n - 1][k][j];
            sum %= BASE;
        }
        return (int)sum;
    }
}
