class Solution {
    private static final int MOD = 1000000007;
    
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n + 1][m + 1][k + 2];
        dp[0][0][0] = 1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j <= m; ++j) {
                for (int x = 0; x <= k; ++x) {
                    for (int z = 1; z <= m; ++z) {
                        if (z <= j) {
                            dp[i + 1][j][x] += dp[i][j][x];
                            dp[i + 1][j][x] %= MOD;
                            // if (dp[i][j][x] > 0) {
                            //     System.out.format("{%d, %d, %d}: %d -> {%d, %d, %d}: %d\n", i, j, x, dp[i][j][x], i + 1, z, x, dp[i + 1][z][x]);
                            // }
                        } else {
                            dp[i + 1][z][x + 1] += dp[i][j][x];
                            dp[i + 1][z][x + 1] %= MOD;
                            // if (dp[i][j][x] > 0) {
                            //     System.out.format("{%d, %d, %d}: %d -> {%d, %d, %d}: %d\n", i, j, x, dp[i][j][x], i + 1, z, x + 1, dp[i + 1][z][x + 1]);
                            // }
                        }
                    }
                }
            }
        }
        long count = 0;
        for (int j = 1; j <= m; ++j) {
            count += dp[n][j][k];
            count %= MOD;
        }
        return (int)count;
    }
}
