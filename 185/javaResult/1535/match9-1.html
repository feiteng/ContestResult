class Solution {
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n][m+1][k+1];

        for (int i = 1; i <= m; i++) {
            dp[0][i][1] = 1;
        }

        for (int i = 1; i < n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int l = 1; l <= k; l++) {
                    long sum = 0;
                    for (int x = 1; x < j; x++) {
                        sum += dp[i - 1][x][l - 1];
                    }
                    dp[i][j][l] = (int) ((sum + j * dp[i - 1][j][l]) % 1000000007);
                
}
            }
        }
        long res = 0L;
        for (int i = 1; i <= m; i++) {
            res += dp[n-1][i][k];
        }
        return (int) (res % 1000000007);        
    }
}
