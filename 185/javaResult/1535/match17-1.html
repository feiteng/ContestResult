class Solution {
    public int numOfArrays(int n, int m, int k) {
        long mod = (long) 1e9+7;
        
long[][][] dp = new long[n][m][k];
        for (int j = 0; j < m; j++) {
            dp[0][j][0] = 1;
        }
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int kk = 0; kk < k; kk++) {
                    // Does not change max
                    dp[i][j][kk] += (j+1) * dp[i-1][j][kk];
                    dp[i][j][kk] %= mod;
                    // changes max, then choose j this time
                    
if (kk > 0) {
                        for (int b = 0; b < j; b++) {
                            dp[i][j][kk] += dp[i-1][b][kk-1];
                            dp[i][j][kk] %= mod;
                        }
                    }
                }
            }
        }
        
        long total = 0;
        for (int j = 0; j < m; j++) {
            total += dp[n-1][j][k-1];
            total %= mod;
        }
        return (int) total;
    }
}
