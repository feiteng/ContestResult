class Solution {
    
    int mod = 1_000_000_007;
    Long[][][] memo;
    public int numOfArrays(int n, int m, int k) {
        memo = new Long[101][101][101];
        return (int)dfs(0, k, m, n, 0);
    }
    
    private long dfs(int idx, int k, int m, int n, int curMax) {
        if (k < 0) return 0;
        if (idx == n) {
            if (k == 0) {
                return 1;
            }
            return 0;
        }
        
        if (memo[idx][k][curMax] != null) {
            return memo[idx][k][curMax];
        }
        
        long curRes = 0;
        for (int i = 1; i <= m; i++) {
            if (i > curMax) {
                curRes += dfs(idx + 1, k - 1, m, n, i);
                curRes %= mod;
            } else {
                curRes += dfs(idx + 1, k, m, n, curMax);
                curRes %= mod;
            }
        }
        memo[idx][k][curMax] = curRes;
        return curRes;
    }
}
