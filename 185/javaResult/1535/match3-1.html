class Solution {
    
    int MOD = 1000000007;
    
    public int numOfArrays(int n, int m, int k) {
        int[][][] dp = new int[n + 1][m + 1][k + 1];        
        
        // init
        for (int i = 1; i <= m; i++) {
            dp[1][i][1] = 1;
        }
        
        
        for (int i = 2; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int l = 1; l <= k; l++) {
                    dp[i][j][l] =dp[i - 1][j][l];

                    for (int g = 1; g < j; g++) {
                        dp[i][j][l] = (dp[i][j][l] + dp[i - 1][j][l]) % MOD;
                        dp[i][j][l] = (dp[i][j][l] + dp[i - 1][g][l - 1]) % MOD;
                    }
                }
            }
        }
        
        int res = 0;
        for (int i = 1; i <= m; i++)
            res = (res + dp[n][i][k]) % MOD;
        return res;
    }
}
