class Solution {
long[][][] f;

    public int numOfArrays(int n, int m, int k) {
        if(m < k || k < 0) return 0;
        if(n == 0)
        {
            if(k == 0) return 1;
            return 0;
        }

        f = new long[m + 1][n + 1][k + 1];
        
        for(int i = 1; i <= m; i++)
        {
        	for(long[] ff : f[i])
            {
                Arrays.fill(ff, -1);
            }
        }
            
        int mod = 1_000_000_007;
        long re = 0;

        for(int i = 1; i <= m; i++)
        {
            re += dfs(n, i, k - 1, m, 1);
        }
        
        return (int)(re % mod);
    }   

    long dfs(int n, int mini, int k, int m, int pos)
    {

        if(f[mini][pos][k] >= 0) return f[mini][pos][k];
        int mod = 1_000_000_007;

        if(k == 0)
        {
            // from 1 to mini - 1
            int prod = mini;
            long re = 1;
            for(int i = pos; i < n; i++)
            {
                re *= prod;
                re %= mod;
            }
            return re;
        }

        if(pos >= n) return 0;

        long re = 0;
        long prod = 1;

        re += mini * dfs(n, mini, k, m, pos + 1);
        for(int j = mini + 1; j <= m; j++)
        {
            re += dfs(n, j, k - 1, m, pos + 1);
        }

        return f[mini][pos][k] = (re + mod) % mod;
    } 
}
