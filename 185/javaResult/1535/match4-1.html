class Solution {
    long MOD = (long)1e9+7;
    public int numOfArrays(int n, int m, int k) {
        long[][][] DP = new long[1+n][1+m][1+k];
        DP[0][0][0] = 1;
        
for(int i = 1; i<= n; i++){
            for(int j = 1; j<= m; j++){
                for(int l = 1; l <= k; l++){
                    for(int prev = 0; prev <= m; prev++){
                        DP[i][Math.max(j, prev)][l] += DP[i-1][prev][l-(prev<j?1:0)];
                        if(DP[i][Math.max(j, prev)][l]>= MOD)DP[i][Math.max(j, prev)][l] -= MOD;
                    
}
                }
            }
        }
        long ans = 0;
        for(int i = 1; i<= m; i++)ans = (ans+DP[n][i][k])%MOD;
        return (int)ans;
    }
}
