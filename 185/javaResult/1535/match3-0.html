class Solution {
    
    int mod = (int)1e9 + 7;
    
    public int numOfArrays(int n, int m, int k) {
        int[][][] dp = new int[n + 1][m + 1][k + 1];        
        
        // init
        for (int i = 1; i <= m; i++) {
            dp[1][i][1] = 1;
        }
        
        
        for (int i = 2; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int q = 1; q <= k; q++) {
                    dp[i][j][q] = dp[i - 1][j][q];

                    for (int g = 1; g < j; g++) {
                        dp[i][j][q] = (dp[i][j][q] + dp[i - 1][j][q]) % mod;
                        dp[i][j][q] = (dp[i][j][q] + dp[i - 1][g][q - 1]) % mod;
                    }
                }
            }
        }
        
        int ans = 0;
        for (int i = 1; i <= m; i++) ans = (ans + dp[n][i][k]) % mod;
        return ans;
    }
}
