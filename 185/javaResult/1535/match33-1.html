class Solution {
    public int numOfArrays(int N, int M, int K) {


        int mod = 1_000_000_007;

        // n numbers
        // 1<=num<=m
        // cost=k

        int[][][] dp = new int[N + 1][M + 1][K + 1];

        dp[0][0][0] = 1;

        
for (int n = 1; n <= N; n++) {

            for (int m = 1; m <= M; m++) {
                for (int k = 1; k <= K; k++) {

                    long cnt = 0;

                    // System.out.printf("dp[%d][%d][%d]\n", n, m, k);
                    for (int m2 = 0; m2 < m; m2++) {
                        cnt += dp[n - 1][m2][k - 1];

                        cnt %= mod;

                        //System.out.printf("++dp[%d][%d][%d]  (%d)\n", n - 1, m2, k - 1, dp[n - 1][m2][k - 1]);
                    }

                    cnt += (long) dp[n - 1][m][k] * m;
                    cnt %= mod;

                    //System.out.printf("++dp[%d][%d][%d]  (%d)\n", n - 1, m, k, dp[n - 1][m][k]);


                    dp[n][m][k] = (int) cnt;

                    //System.out.printf("dp[%d][%d][%d]=%d\n", n, m, k, dp[n][m][k]);
                
}
            }

        }
        int ans = 0;
        for (int m = 0; m <= M; m++) {
            ans += dp[N][m][K];
            ans %= mod;
        }

        return ans;


    }
}
