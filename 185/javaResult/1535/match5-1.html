class Solution {
    public int numOfArrays(int n, int m, int k) {
        if (k == 0 || m < k) {
            return 0;
        }
        //      length-1  max_m-1  devided -1
        long[][][] dp = new long[n][m][k];

        for (int mm = 0; mm < m; mm++) {
            dp[0][mm][0] = 1;
        }
        for (int nn = 0; nn < n - 1; nn++) {
            for (int kk = 0; kk < k; kk++) {
                for (int mm = 0; mm < m; mm++) {
                    dp[nn][mm][kk] = dp[nn][mm][kk] % 1000000007;
                    dp[nn + 1][mm][kk] += dp[nn][mm][kk] * (mm + 1);
                    
if (kk + 1 < k) {
                        for (int mmm = mm + 1; mmm < m; mmm++) {
                            dp[nn + 1][mmm][kk + 1] += dp[nn][mm][kk];
                        }
                    }
                }
            }
        }
        
        long ret = 0;
        for (int mm = 0; mm < m; mm++) {
            ret += dp[n - 1][mm][k - 1];
        }
        return (int) (ret % 1000000007);
    }
}
