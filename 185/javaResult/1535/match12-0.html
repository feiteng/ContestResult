class Solution {
    int MODE = 1000000007;
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n][k + 1][m + 1];
        for(int i = 1;i <= m;i++) {
            dp[0][1][i] = 1;
        }
        for(int i = 1;i < n;i++) {
            for(int j = 1;j <= k;j++) {
                for(int o = 1;o <= m;o++) {
                    for(int h = 1;h <= m;h++) {
                        if(h >= o) {
                            dp[i][j][h] += dp[i-1][j][h];
                            dp[i][j][o] %= MODE;
                        } else {
                            dp[i][j][o] += dp[i-1][j-1][h];
                            dp[i][j][o] %= MODE;
                        }
                    }
                }
            }
        }
        long sum = 0;
        for(int i = 1;i <= m;i++) {
            sum += dp[n - 1][k][i];
            sum %= MODE;
        }
        return (int)sum;
    }
}
