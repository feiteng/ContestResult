class Solution {
    public int numOfArrays(int n, int m, int k) {
        if (k>m)
            return 0;
        
int[][][] dp = new int[n][m+1][n+1];
        for (int i = 1; i <= m; ++i) {
            dp[0][i][1] = 1;
        }
        for (int a = 1; a < n; ++a) {
            for (int b = 1; b <= a; ++b) {
                for (int x = 1; x <= m; ++x) {
                    for (int y = 1; y <= m; ++y) {
                        if (y>x) {
                            dp[a][y][b+1] += dp[a-1][x][b];
                            dp[a][y][b+1] %= 1000000007;
                        } else {
                            dp[a][x][b] += dp[a-1][x][b];
                            dp[a][x][b] %= 1000000007;
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 1; i <= m; ++i) {
            ans += dp[n-1][i][k];
            ans %= 1000000007;
        }
        return ans;
    }
}
