class Solution {
    public int numOfArrays(int n, int m, int k) {
        int divisor = 1000000000 + 7;
        long[][][] dp = new long[n + 1][m + 1][k + 1];
        for (int i = 1; i <= m; i++) {
            dp[1][i][1] = 1;
        }
        for (int length = 2; length <= n; length++) {
            for (int cost = 1; cost <= k; cost++) {
                for (int i = 1; i <= m; i++) {
                    for (int prevMax = 1; prevMax <= m; prevMax++) {
                        if (i > prevMax) {
                            dp[length][i][cost] += dp[length - 1][prevMax][cost - 1];
                            dp[length][i][cost] %= divisor;
                        } else {
                            dp[length][prevMax][cost] += dp[length - 1][prevMax][cost];
                            dp[length][prevMax][cost] %= divisor;
                        }
                    }
                }
            }
        }
        long sum = 0;
        for (int max = 1; max <= m; max++) {
            sum += dp[n][max][k];
        }
        return (int) (sum % divisor);
    }
}
