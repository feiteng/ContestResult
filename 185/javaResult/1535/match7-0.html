class Solution {
    public int numOfArrays(int n, int m, int K) {
        if (K > m) {
            return 0;
        }
        int[][][] dp = new int[n][K][m];
        for (int i = 0; i < m; i++) {
            dp[0][0][i] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < K; j++) {
                for (int k = 0; k < m; k++) {
                    dp[i][j][k] += (int)((long)dp[i - 1][j][k] * (k + 1) % 1000000007);
                    dp[i][j][k] %= 1000000007;
                    if (j > 0) {
                        for (int l = 0; l < k; l++) {
                            dp[i][j][k] += dp[i - 1][j - 1][l];
                            dp[i][j][k] %= 1000000007;
                        }
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            res += dp[n - 1][K - 1][i];
            res %= 1000000007;
        }
        return res;
    }
}
