class Solution {
    public String reformat(String s) {
        List<Character> chars = new ArrayList<>();
        List<Character> nums = new ArrayList<>();
        for (int i = 0; i < s.length(); ++i) {
            char c = s.charAt(i);
            if (c >= '0' && c <= '9') {
                nums.add(c);
            } else {
                chars.add(c);
            }
        }
        if (Math.abs(chars.size() - nums.size()) > 1) return "";
        StringBuilder sbu = new StringBuilder();
        List<Character> max, min;
        if (chars.size() > nums.size()) {
            max = chars;
            min = nums;
        } else {
            max = nums;
            min = chars;
        }
        for (int i = 0, j = 0; i < max.size() || j < min.size(); ) {
            if (i < max.size()) {
                sbu.append(max.get(i++));
            }
            if (j < min.size()) {
                sbu.append(min.get(j++));
            }
        }
        return sbu.toString();
    }
}
