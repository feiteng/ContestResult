class Solution {
    public String reformat(String s) {
        ArrayList<Character> num = new ArrayList<>(), alpha = new ArrayList<>();
        for(char c:s.toCharArray()){
            if(c>='0'&&c<='9') num.add(c);
            else alpha.add(c);
        }
        if(num.size()> alpha.size()){
            ArrayList<Character> temp = alpha;
            alpha = num;
            num = temp;
        }
        if(alpha.size()>num.size()+1) return "";
        StringBuilder sb = new StringBuilder();
        for(int i=0;i<alpha.size();i++){
            sb.append(alpha.get(i));
            if(i<num.size()) sb.append(num.get(i));
        }
        return sb.toString();
    }
}
