class Solution {
    public String reformat(String s) {
        Queue<Character> q1 = new LinkedList<>();
        Queue<Character> q2 = new LinkedList<>();
        
        for (char c : s.toCharArray()) {
            if (c >= 'a' && c <= 'z') {
                q1.offer(c);
            } else {
                q2.offer(c);
            }
        }
        
        if (Math.abs(q1.size() - q2.size()) > 1) 
            return "";
        
        if (q1.size() >= q2.size())
            return helper(q1, q2);
        else
            return helper(q2, q1);
    }
    
    public String helper(Queue<Character> q1, Queue<Character> q2) {
        StringBuilder res = new StringBuilder();
        while (!q1.isEmpty() && !q2.isEmpty()) {
            res.append(q1.poll());
            res.append(q2.poll());
        }
        while (!q1.isEmpty()) {
            res.append(q1.poll());
        }
        return res.toString();
        
    }
}
