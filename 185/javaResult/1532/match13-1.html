class Solution {
    public String reformat(String s) {
        List<Character> num = new ArrayList<>();
        List<Character> str = new ArrayList<>();
        for (char c: s.toCharArray()) {
            
if (Character.isDigit(c)) {
                num.add(c);
            } else {
                str.add(c);
            }
        }
        if (Math.abs(str.size() - num.size()) > 1) return "";
        StringBuilder sb = new StringBuilder();
        if (str.size() > num.size()) {
            sb.append(str.get(0));
            for (int i = 0; i < num.size(); i++) {
                sb.append(num.get(i));
                sb.append(str.get(i+1));
            }
        } else if (str.size() < num.size()) {
            sb.append(num.get(0));
            for (int i = 0; i < str.size(); i++) {
                sb.append(str.get(i));
                sb.append(num.get(i+1));
            
}
        } else {
            for (int i = 0; i < num.size(); i++) {
                sb.append(num.get(i));
                sb.append(str.get(i));
            }
        }
        return sb.toString();
    }
}
