class Solution {
    public String reformat(String s) {
        List<Character> a = new ArrayList<>();
        List<Character> b = new ArrayList<>();
        int n = s.length();
        for (int i = 0; i < n; i++) {
          char c = s.charAt(i);
          if (c >= '0' && c <= '9') 
{a.add(c);}
          else {b.add(c);}
        }

        if (Math.abs(a.size() - b.size()) > 1) {return "";}

        StringBuilder sb = new StringBuilder();
        if (a.size() > b.size()) {
          sb.append(a.get(a.size() - 1));
          
a.remove(a.size() -1);
          for (int i = 0; i < a.size(); i++) {
            sb.append(b.get(i));
            sb.append(a.get(i));
          }
        } 
        else if (a.size() < b.size()) {
          sb.append(b.get(b.size() - 1));
          
b.remove(b.size() -1);
          for (int i = 0; i < a.size(); i++) {
            sb.append(a.get(i));
            sb.append(b.get(i));
          }
        }

        else {
          for (int i = 0; i < a.size(); i++) {
            sb.append(b.get(i));
            sb.append(a.get(i));
          }
        }

        return sb.toString();



    }
}
