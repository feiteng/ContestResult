class Solution {
        public String reformat(String s) {

        Queue<Character> l1 = new LinkedList<>(), 
                l2 = new LinkedList<>();
        for(char c : s.toCharArray())
        {
            if('0' <= c && c <= '9')
            {
                l1.add(c);
            }        
            else l2.add(c);
        }

        // if(Math.abs(l1.size() - l2.size()) > 1) return "";
        if(l2.size() > l1.size())
        {
            Queue<Character> l3 = l1;
            l1 = l2;
            l2 = l3;
        }
        String str = "";
        while(!l2.isEmpty())
        {
            str += l1.poll();
            str += l2.poll();
        }
        if(!l1.isEmpty()) str += l1.poll();
        if(!l1.isEmpty()) return "";
        return str;
    }    
}
