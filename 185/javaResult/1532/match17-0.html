class Solution {
    public String reformat(String s) {
        List<Character> chars = new ArrayList<>();
        List<Character> digits = new ArrayList<>();
        
        for(int i=0;i<s.length();i++){
            if(s.charAt(i) >= '0' && s.charAt(i) <='9') digits.add(s.charAt(i));
            else if(s.charAt(i) >= 'a' && s.charAt(i) <= 'z') chars.add(s.charAt(i));
        }
        
        if(Math.abs(digits.size() - chars.size()) > 1) return "";
        StringBuilder sb = new StringBuilder();
        
        if(digits.size() > chars.size()){
            for(int i=0;i<chars.size();i++){
                sb.append(digits.get(i));
                sb.append(chars.get(i));
            }
            sb.append(digits.get(digits.size()-1));
        } else {
            for(int i=0;i<digits.size();i++){
                sb.append(chars.get(i));
                sb.append(digits.get(i));
            } 
            if(sb.length() < digits.size() + chars.size()) sb.append(chars.get(chars.size()-1));
        }
        return sb.toString();
    }
}
