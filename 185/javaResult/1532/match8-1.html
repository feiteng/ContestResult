class Solution {
    public String reformat(String s) {
        Queue<Character> cq = new LinkedList<>(), cn = new LinkedList<>();
        for (char c : s.toCharArray()) {
            if (Character.isDigit(c)) 
{
                cn.add(c);
            } else {
                cq.add(c);
            }
        }
        if (Math.max(cn.size(), cq.size()) >= Math.min(cn.size(), cq.size()) + 2) {
            return "";
        }
        Queue<Character> max = cq.size() >= cn.size() ? cq : cn, min = cq.size() >= cn.size() ? cn : cq;
        
StringBuilder res = new StringBuilder();
        while (!max.isEmpty()) {
            res.append(max.poll());
            if (!min.isEmpty()) {
                res.append(min.poll());
            }
        }
        return cq.isEmpty() && cn.isEmpty() ? res.toString() : "";
    }
}
