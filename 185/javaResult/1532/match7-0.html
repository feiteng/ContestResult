class Solution {
    public String reformat(String s) {
        Queue<Character> qc = new LinkedList<>();
        Queue<Character> qn = new LinkedList<>();
        for (char c : s.toCharArray()) {
            if (c >= 'a' && c <= 'z') {
                qc.add(c);
            } else {
                qn.add(c);
            }
        }
        if (qc.size() < qn.size()) {
            Queue<Character> temp = qc;
            qc = qn;
            qn = temp;
        }
        if (qc.size() - qn.size() > 1) {
            return "";
        }
        StringBuilder sb = new StringBuilder();
        while (!qc.isEmpty()) {
            sb.append(qc.poll());
            if (!qn.isEmpty()) {
                sb.append(qn.poll());
            }
        }
        return sb.toString();
    }
}
