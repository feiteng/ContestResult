class Solution {
    public String reformat(String s) {
        List<Character> let = new ArrayList<>();
        List<Character> num = new ArrayList<>();
        
        for (int i = 0; i < s.length();i++){
            char c = s.charAt(i);
            
            if (Character.isDigit(c)){
                num.add(c);
            }
            else {
                let.add(c);
            }
        }
        if (Math.abs(let.size() - num.size()) >= 2) return "";
        
        StringBuilder sb = new StringBuilder();
        
        if (let.size() >= num.size()){
            for (int i = 0; i < let.size();i++){
                sb.append(let.get(i));
                if (i < num.size()) sb.append(num.get(i));
            }
        }
        else {
            for (int i = 0; i < num.size();i++){
                sb.append(num.get(i));
                if (i < let.size()) sb.append(let.get(i));
            }
        }
        return sb.toString();
    }
}
