class Solution {
    public String reformat(String s) {
        char[] cArray = s.toCharArray();
        List<Character> alpha = new ArrayList();
        List<Character> digit = new ArrayList();
        
        for(char c: cArray) {
            if(c >= '0' && c <= '9') {
                digit.add(c);
            } else {
                alpha.add(c);
            }
        }
                
        if(Math.abs(digit.size() - alpha.size()) > 1) return "";
        
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < Math.min(digit.size(), alpha.size()); i++) {
            sb.append(digit.get(i));
            sb.append(alpha.get(i));
        }
        if(digit.size() > alpha.size()) {
            sb.append(digit.get(digit.size()-1));
        } else if(digit.size() < alpha.size()) {
            sb.insert(0, alpha.get(alpha.size()-1));
        }
        return sb.toString();
    }
}
