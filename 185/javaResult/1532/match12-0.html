class Solution {

    public String reformat(String s) {
    	List<Character> digitList = new ArrayList<>() , letterList = new ArrayList<>();
    	for (char ch : s.toCharArray()) {
    		if (Character.isDigit(ch)) {
    			digitList.add(ch);
    		} else {
    			letterList.add(ch);
    		}
    	}
    	if (digitList.size() == letterList.size() + 1) {
    		StringBuilder builder = new StringBuilder();
    		for (int i = 0;i < letterList.size();i ++) {
    			builder.append(digitList.get(i));
    			builder.append(letterList.get(i));
    		}
    		builder.append(digitList.get(digitList.size() - 1));
    		return builder.toString();
    	} else if (digitList.size() == letterList.size() - 1) {
    		StringBuilder builder = new StringBuilder();
    		for (int i = 0;i < digitList.size();i ++) {
    			builder.append(letterList.get(i));
    			builder.append(digitList.get(i));
    		}
    		builder.append(letterList.get(letterList.size() - 1));
    		return builder.toString();
    	} else if (digitList.size() == letterList.size()) {
    		StringBuilder builder = new StringBuilder();
    		for (int i = 0;i < letterList.size();i ++) {
    			builder.append(letterList.get(i));
    			builder.append(digitList.get(i));
    		}
    		return builder.toString();
    	} else {
    		return "";
    	}
    }
    
}
