class Solution {
    public String reformat(String s) {
        Queue<Character> letters = new ArrayDeque<>();
        Queue<Character> numbers = new ArrayDeque<>();
        for (char c : s.toCharArray()) {
            if (Character.isDigit(c)) 
{
                numbers.add(c);
            } else {
                letters.add(c);
            }
        }
        
        if (Math.abs(letters.size() - numbers.size()) > 1) {
            return "";
        }
        
        StringBuilder sb = new StringBuilder();
        if (letters.size() > numbers.size()) {
            sb.append(letters.poll());
        }
        
while (!numbers.isEmpty()) {
            sb.append(numbers.poll());
            if (!letters.isEmpty()) {
                sb.append(letters.poll());
            }
        }
        return sb.toString();
    }
}
