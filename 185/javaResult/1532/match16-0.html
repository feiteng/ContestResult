class Solution {
  public String reformat(String s) {
    StringBuilder sb1 = new StringBuilder();
    StringBuilder sb2 = new StringBuilder();
    for (int i = 0; i != s.length(); i++) {
      char ch = s.charAt(i);
      if (Character.isDigit(ch))
        sb2.append(ch);
      else
        sb1.append(ch);
    }
    if (Math.abs(sb1.length() - sb2.length()) > 1)
      return "";

    return helper(sb1, sb2);
  }

  private String helper(StringBuilder sb1, StringBuilder sb2) {
    if (sb1.length() < sb2.length())
      return helper(sb2, sb1);
    StringBuilder sb = new StringBuilder(sb1.length() + sb2.length());
    for (int len = Math.min(sb1.length(), sb2.length()), i=0; i!=len; i++) {
      sb.append(sb1.charAt(i));
      sb.append(sb2.charAt(i));
    }
    if (sb1.length()>sb2.length())
      sb.append(sb1.charAt(sb1.length() - 1));
    return sb.toString();
  }
}
