class Solution {
    public String reformat(String s) {
        List<Character> d = new ArrayList<>();
        List<Character> l = new ArrayList<>();
        int n = s.length();
        for (int i = 0; i < n; i++) {
            if (Character.isDigit(s.charAt(i))) d.add(s.charAt(i));
            else l.add(s.charAt(i));
        }
        int diff = Math.abs(d.size() - l.size());
        // System.out.println(l);
        // System.out.println(d);
        if (diff != 0 && diff != 1) 
{
            return "";
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0 ; i < Math.min(l.size(),d.size()); i++) {
            if (l.size()>d.size()){
                sb.append(l.get(i));
                sb.append(d.get(i));
            }else{
                sb.append(d.get(i));
                sb.append(l.get(i));
            }
        }
        if (l.size()>d.size()){
            sb.append(l.get(l.size()-1));
        }else if (l.size()<d.size()){
            sb.append(d.get(d.size()-1));
        }
        return sb.toString();
    }
}
