class Solution {
	    public String reformat(String s) {
	        char[] t = s.toCharArray();
	        ArrayList<Character> alf = new ArrayList<>();
	        ArrayList<Character> num = new ArrayList<>();
	        for(char c: t) {
	        	if(c >= '0' && c <= '9') num.add(c);
	        	else alf.add(c);
	        }
	        if(num.size() == alf.size()+1 || num.size() == alf.size()) {
	        	StringBuilder sb = new StringBuilder();
	        	for(int i = 0; i < alf.size(); i++) {
	        		sb.append(num.get(i));
	        		sb.append(alf.get(i));
	        	}
	        	if(num.size() == alf.size()+1) sb.append(num.get(num.size()-1));
	        	return sb.toString();
	        }
	        else if(alf.size() == num.size()+1) {
	        	StringBuilder sb = new StringBuilder();
	        	for(int i = 0; i < num.size(); i++) {
	        		sb.append(alf.get(i));
	        		sb.append(num.get(i));
	        	}
	        	sb.append(alf.get(alf.size()-1));
	        	return sb.toString();
	        }
	        else {
	        	return "";
	        }
	    }
	}
