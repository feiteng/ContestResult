class Solution {
   
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<String, Map<String, Integer>> map = new HashMap<>();
        TreeSet<String> foods = new TreeSet<>();
        TreeSet<Integer> tables = new TreeSet<>();
        for (List<String> o : orders) {
            String table = o.get(1);
            Map<String, Integer> ct = map.getOrDefault(table, new HashMap<>());
            String food = o.get(2);
            ct.put(food, ct.getOrDefault(food, 0) + 1);
            
map.put(table, ct);
            foods.add(food);
            tables.add(Integer.valueOf(table));
        }
        List<List<String>> entries = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for (String f : foods) {
            header.add(f);
        }
        entries.add(header);
        for (Integer table : tables) {
            List<String> entry = new ArrayList<>();
            String stable = table.toString();
            entry.add(stable);
            for (String f : foods) {
                
entry.add(String.valueOf(map.get(stable).getOrDefault(f, 0)));
            }
            entries.add(entry);
        }
        return entries;
    }


}
