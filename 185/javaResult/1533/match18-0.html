class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        // dish : (table : count)
        Map<String, Map<String, Integer>> map = new HashMap<>();
        Set<Integer> tables = new HashSet<>();

        for (List<String> order : orders) {
            String table = order.get(1);
            String dish = order.get(2);
            if (!map.containsKey(dish)) {
                map.put(dish, new HashMap<>());
            }
            Map<String, Integer> tableMap = map.get(dish);
            int count = tableMap.getOrDefault(table, 0) + 1;
            tableMap.put(table, count);
            tables.add(Integer.valueOf(table));
        }
        LinkedList<String> dishList = new LinkedList<>(map.keySet());
        Collections.sort(dishList);
        List<String> header = new ArrayList<>();
        header.add("Table");
        header.addAll(dishList);
        List<Integer> tableList = new LinkedList<>(tables);
        Collections.sort(tableList);

        List<List<String>> result = new LinkedList<>();
        result.add(header);
        for (Integer table : tableList) {
            List<String> row = new ArrayList<>();
            row.add(String.valueOf(table));
            for (String dish : dishList) {
                row.add(String.valueOf(map.get(dish).getOrDefault(String.valueOf(table), 0)));
            }
            result.add(row);
        }
        return result;       
    }
}
