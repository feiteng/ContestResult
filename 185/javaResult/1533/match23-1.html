class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        // [customerNamei,tableNumberi,foodItemi]
        Map<Integer, Map<String, Integer>> dict = new HashMap<>();
        Set<String> foods = new HashSet<>();
        for(List<String> ord : orders) {
            int table = Integer.parseInt(ord.get(1));
            String food = ord.get(2);
            foods.add(food);
            dict.computeIfAbsent(table, v->new HashMap<>());
            
dict.get(table).put(food, dict.get(table).getOrDefault(food, 0) + 1);
        }
        List<List<String>> res = new LinkedList<>();
        List<String> foodsList = new ArrayList<>(foods);
        Collections.sort(foodsList);
        List<String> firstRow = new LinkedList<>();
        firstRow.add("Table");
        for(String food : foodsList) firstRow.add(food);
        
res.add(firstRow);
        
        List<Integer> tables = new ArrayList(dict.keySet());
        Collections.sort(tables);
        for(int table : tables) {
            List<String> row = new LinkedList<>();
            Map<String, Integer> counts = dict.get(table);
            row.add("" + table);
            for(String food : foodsList) {
                
row.add("" + (counts.getOrDefault(food, 0)));
            }
            res.add(row);
        }
        return res;
    }
}
