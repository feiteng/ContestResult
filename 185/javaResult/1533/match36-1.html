class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        
Map<String, Integer> map = new HashMap<>();
        Set<Integer> table = new HashSet<>();
        Set<String> item = new HashSet<>();
        
        for (List<String> order : orders) {
            String key = order.get(1) + order.get(2);
            table.add(
Integer.parseInt(order.get(1)));
            item.add(order.get(2));
            map.put(key, map.getOrDefault(key, 0) + 1);
        }
        
        List<Integer> t = new ArrayList<>(table);
        List<String> i = new ArrayList<>(item);
        Collections.sort(t);
        Collections.sort(i);
        
        List<List<String>> res = new ArrayList<>();
        List<String> head = new ArrayList<>();
        head.add("Table");
        for (String o : i)
            head.add(o);
        
res.add(head);
        for (Integer ta : t) {
            List<String> tmp = new ArrayList<>();
            tmp.add(String.valueOf(ta));
            for (String it : i) {
                String k = String.valueOf(ta) + it;
                if (map.containsKey(k))
                    tmp.add(String.valueOf(map.get(k)));
                else 
                    tmp.add("0");
            
}
            res.add(tmp);
        }
        
        return res;
        
    }
}
