class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, Map<String, Integer>> map = new TreeMap();
        TreeSet<String> set = new TreeSet();
        for (List<String> order : orders) {
            int table = Integer.parseInt(order.get(1));
            String dish = order.get(2);
            
            set.add(dish);
            if (!map.containsKey(table)) {
                map.put(table, new HashMap<String, Integer>());
            }
            map.get(table).put(dish, map.get(table).getOrDefault(dish, 0) + 1);
        }
        
        List<List<String>> res = new ArrayList();
        List<String> list = new ArrayList();
        list.add("Table");
        for (String dish : set) {
            list.add(dish);
        }
        
        res.add(list);
        
        for (int table : map.keySet()) {
            list = new ArrayList();
            list.add(Integer.toString(table));
            for (String dish : set) {
                list.add(Integer.toString(map.get(table).getOrDefault(dish, 0)));
            }
            res.add(list);
        }
        
        return res;
    }
}
