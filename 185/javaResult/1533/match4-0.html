class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<Integer, Map<String, Integer>> table = new TreeMap();
        Map<String, Integer> food = new TreeMap();
        for(List<String> o : orders){
            int tn = Integer.parseInt(o.get(1));
            if(!table.containsKey(tn))
                table.put(tn, new HashMap());
            if(!food.containsKey(o.get(2)))
                food.put(o.get(2), food.size());
            table.get(tn).put(o.get(2), table.get(tn).getOrDefault(o.get(2), 0) + 1);
        }
        List<List<String>> res = new ArrayList();
        List<String> name = new ArrayList();
        name.add("Table");
        for(Map.Entry<String,Integer> f : food.entrySet())
            name.add(f.getKey());
        res.add(name);
        int len = food.size();
        for(Map.Entry<Integer,Map<String, Integer>> t : table.entrySet()){
            List<String> cur = new ArrayList();
            cur.add(Integer.toString(t.getKey()));
            for(int i=0;i<len;++i){
                String cur_food = name.get(i+1);
                cur.add(Integer.toString(t.getValue().getOrDefault(cur_food, 0)));
            }
            res.add(cur);
        }
        return res;
    }
}
