class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        //[customerNamei,tableNumberi,foodItemi]
        TreeMap<Integer, HashMap<String, Integer>> map = new TreeMap<>();
        TreeSet<String> foods = new TreeSet<>();
        for (List<String> order: orders) {
            int table = Integer.valueOf(order.get(1));
            String food = order.get(2);
            foods.add(food);
            
map.putIfAbsent(table, new HashMap<>());
            map.get(table).putIfAbsent(food, 0);
            map.get(table).put(food, map.get(table).get(food) + 1);
        }
        
List<List<String>> results = new ArrayList<>();
        List<String> head = new ArrayList<>();
        head.add("Table");
        for (String s: foods) head.add(s);
        results.add(head);
        for (Map.Entry<Integer, HashMap<String, Integer>> entry: map.entrySet()) {
            List<String> line = new ArrayList<>();
            line.add(String.valueOf(entry.getKey()));
            for (String f: foods) {
                int num = entry.getValue().getOrDefault(f, 0);
                line.add(String.valueOf(num));
            
}
            results.add(line);
        }
        return results;
    }
}
