import java.util.*;


class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, HashMap<String, Integer>> tables = new TreeMap<>();
        TreeSet<String> foods = new TreeSet<>();
        for (List<String> order : orders) {
            Integer tableName = Integer.valueOf(order.get(1));
            tables.putIfAbsent(tableName, new HashMap<>());
            HashMap<String, Integer> table = tables.get(tableName);
            String food = 
order.get(2);
            foods.add(food);
            table.put(food, table.getOrDefault(food, 0) + 1);
        }
        List<List<String>> ret = new ArrayList<>();
        List<String> head = new ArrayList<>();
        head.add("Table");
        for (String food : foods) {
            head.add(food);
        }
        ret.add(head);
        for (int key : tables.keySet()) {
            List<String> line = new ArrayList<>();
            line.add(String.valueOf(key));
            HashMap<String, Integer> table = tables.get(key);
            for (String food : foods) {
                line.add(String.valueOf(table.getOrDefault(food,0)));
            
}
            ret.add(line);
        }
        return ret;
    }
}
