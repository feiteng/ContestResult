class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, Map<String, Integer>> map_order = new TreeMap<>();
        List<String> list_items = new ArrayList<>();

        for (List<String> tl : orders) {
            int table = Integer.parseInt(tl.get(1));
            String food = tl.get(2);
            if (!list_items.contains(food)) list_items.add(food);
            map_order.putIfAbsent(table, new HashMap<>());
            map_order.get(table).putIfAbsent(food, 0);
            map_order.get(table).put(food, map_order.get(table).get(food) + 1);
        }

        Collections.sort(list_items);
        List<List<String>> res = new ArrayList<>();

        List<String> ttl = new ArrayList<>();
        ttl.add("Table");
        for (String fd : list_items) ttl.add(fd);
        res.add(ttl);

        for (Integer key : map_order.keySet()) {
            List<String> tl = new ArrayList<>();
            tl.add(String.valueOf(key));
            for (String fd : list_items) {
                if (!map_order.get(key).containsKey(fd)) {
                    tl.add(String.valueOf(0));
                } else {
                    tl.add(String.valueOf(map_order.get(key).get(fd)));
                }
            }
            res.add(tl);
        }

        return res;
    }
}
