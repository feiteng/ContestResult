class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<Integer, Map<String, Integer>> map = new HashMap<>();
        Set<String> dishSet = new HashSet<>();
        for (List<String> order: orders) {
            Integer table = Integer.parseInt(order.get(1));
            String dish = order.get(2);
            if (!map.containsKey(table)) {
                map.put(table, new HashMap<String, Integer>());
            }
            map.get(table).put(dish, 
map.get(table).getOrDefault(dish, 0) + 1);
            dishSet.add(dish);
        }
        
        List<List<String>> ans = new ArrayList<>();
        List<String> header = new ArrayList<>();
        List<String> dishes = new ArrayList<>(dishSet);
        Collections.sort(dishes);
        header.add("Table");
        for (int i = 0; i < dishes.size(); i++) {
            header.add(dishes.get(i));
        }
        
ans.add(header);
        List<Integer> tables = new ArrayList<>(map.keySet());
        Collections.sort(tables);
        for (Integer table: tables) {
            List<String> list = new ArrayList<>();
            list.add(Integer.toString(table));
            Map<String, Integer> o = map.get(table);
            for (int i = 0; i < dishes.size(); i++) {
                
list.add(Integer.toString(o.getOrDefault(dishes.get(i), 0)));
            }
            ans.add(list);
        }
        return ans;
    }
}
