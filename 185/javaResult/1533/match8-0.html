class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Set<String> foodSet = new HashSet();
        Map<Integer, Map<String, Integer>> table2Orders = new HashMap();
        
        for(List<String> order: orders) {
            int tableNumber = Integer.parseInt(order.get(1));
            String foodItem = order.get(2);
            
            foodSet.add(foodItem);
            Map<String, Integer> tableOrder = table2Orders.getOrDefault(tableNumber, new HashMap());
            tableOrder.put(foodItem, tableOrder.getOrDefault(foodItem, 0) + 1);
            table2Orders.put(tableNumber, tableOrder);
        }
        
        List<List<String>> result = new ArrayList();
        
        List<String> foodList = new ArrayList(foodSet);
        Collections.sort(foodList);
        List<String> firstRow = new ArrayList(foodList);
        firstRow.add(0, "Table");
        result.add(firstRow);
        
        List<Integer> tables = new ArrayList(table2Orders.keySet());
        Collections.sort(tables);
        for(int tableNum : tables) {
            List<String> row = new ArrayList();
            row.add(String.valueOf(tableNum));
            for(String food: foodList) {
                int count = table2Orders.get(tableNum).getOrDefault(food, 0);
                row.add(String.valueOf(count));
            }
            result.add(row);
        }
        return result;
    }
}
