class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
      TreeSet<String> items = new TreeSet<>();
      TreeMap<Integer, TreeMap<String, Integer>> data = new TreeMap<>();
      for(List<String> o : orders) {
        int tid = Integer.parseInt(o.get(1));
        if(!data.containsKey(tid)) data.put(tid, new TreeMap<>());
        items.add(o.get(2));
        int v = data.get(tid).getOrDefault(
o.get(2), 0);
        data.get(tid).put(o.get(2), v+1);
      }
      List<List<String>> res = new ArrayList<>();
      List<String> cres = new ArrayList<>();
      cres.add("Table");
      cres.addAll(items);
      
res.add(cres);
      for(Map.Entry<Integer, TreeMap<String, Integer>> e : data.entrySet()) {
        cres = new ArrayList<>();
        cres.add("" + e.getKey());
        for(String item : items) {
          cres.add("" + e.getValue().getOrDefault(item, 0));
        
}
        res.add(cres);
      }
      return res;
    }
  }
