class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        List<List<String>> ans = new LinkedList<>();
        TreeMap<Integer, Map<String, Integer>> map = new TreeMap<>();
        TreeSet<String> food = new TreeSet<>();
        for (List<String> s : orders) {
            String cus = s.get(0);
            int tb = Integer.parseInt(s.get(1));
            String item = s.get(2);
            food.add(item);
            map.putIfAbsent(tb, new HashMap<>());
            Map<String, Integer> tbmap = map.get(tb);
            tbmap.put(item, tbmap.getOrDefault(item, 0) + 1);
        }
        
        
        List<String> header = new ArrayList<>();
        header.add("Table");
        for (String ff : food) {
            header.add(ff);
        }
        ans.add(header);
        
        
        for (int tb : map.keySet()) {
            List<String> cur = new ArrayList<>();
            cur.add(tb+"");
            Map<String, Integer> tbmap = map.get(tb);
            for (String ff : food) {
                cur.add(tbmap.getOrDefault(ff, 0)+"");
            }
            ans.add(cur);
        }
        
        return ans;
        
        
    }
}
