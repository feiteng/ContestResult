class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        HashSet<String> set = new HashSet<>();
        HashMap<Integer, HashMap<String, Integer>> map = new HashMap<>();
        for(List<String> lx : orders) {
            int tn = Integer.valueOf(lx.get(1));
            String fd = lx.get(2);
            set.add(fd);
            if(!map.containsKey(tn)) map.put(tn, new HashMap<>());
            
HashMap<String, Integer> cx = map.get(tn);
            cx.put(fd, cx.getOrDefault(fd, 0) + 1);
        }
        List<String> la = new ArrayList<>(set);
        Collections.sort(la);
        la.add(0, "Table");
        List<List<String>> res = new ArrayList<>();
        
res.add(la);
        List<Integer> lb = new ArrayList<>(map.keySet());
        Collections.sort(lb);
        for(int x : lb) {
             List<String> lc = new ArrayList<>();
            lc.add(x + "");
            HashMap<String, Integer> cx = map.get(x);
            for(int i = 1; i < la.size(); ++i) {
                String fd = 
la.get(i);
                lc.add("" + (cx.getOrDefault(fd, 0)));
            }
            res.add(lc);
        }
        return res;
    }
}
