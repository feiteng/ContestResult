class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Set<String> foodSet = new TreeSet<>();
        Set<Integer> tableSet = new TreeSet<>();
        Map<String, Integer> count = new HashMap<>();
        
        for (List<String> order : orders) {
            String table = order.get(1);
            String food = order.get(2);
            String tableFood = table + food;
            foodSet.add(food);
            tableSet.add(Integer.parseInt(table));
            count.put(tableFood, count.getOrDefault(tableFood, 0) + 1);
        }
        
        List<List<String>> ans = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for (String food : foodSet) {
            header.add(food);
        }
        ans.add(header);
        // System.out.println(header);
        
        for (int table : tableSet) {
            List<String> row = new ArrayList<>();
            row.add(table + "");
            for (int j = 1; j < header.size(); j++) {
                String food = header.get(j);
                String tableFood = table + "" + food;
                int num = count.getOrDefault(tableFood, 0);
                row.add(num + "");
            }
            ans.add(row);
        }
        return ans;
        
    }
}
