class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        List<List<String>> re = new ArrayList<>();
        TreeMap<Integer, Map<String, Integer>> mapTable = new TreeMap<>();
        TreeSet<String> dishes = new TreeSet<>();
        
        for(List<String> order : orders)
        {
            int table = Integer.valueOf(order.get(1));
            String dish = order.get(2);
            if(!mapTable.containsKey(table)) mapTable.put(table, new HashMap<>());
            Map<String, Integer> map = mapTable.get(table);
            map.put(dish, map.getOrDefault(dish, 0) + 1);
            dishes.add(dish);
        }

        List<String> title = new ArrayList<>();
        title.add("Table");
        
        for(String dish : dishes) title.add(dish);
        re.add(title);
        
        for(int key : mapTable.keySet())
        {
            List<String> row = new ArrayList<>();
            row.add("" + key);
            for(String dish : dishes)
            {
                row.add("" + mapTable.get(key).getOrDefault(dish, 0));
            }
            re.add(row);
        }

        return re;
    }    
}
