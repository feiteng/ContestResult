class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Set<String> food = new HashSet<>();
        Map<Integer, Map<String, Integer>> foodmap = new HashMap<>();
        for(List<String> order: orders){
            int id = Integer.valueOf(order.get(1));
            String item = order.get(2);
            food.add(item);
            foodmap.putIfAbsent(id, new HashMap<>());
            Map<String, Integer> sub = foodmap.get(id);
            sub.put(item, sub.getOrDefault(item, 0) + 1);
        }
        List<String> foods = new ArrayList<>(food);
        Collections.sort(foods);
        List<List<String>> result = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for(String elem: foods){
            header.add(elem);
        }
        result.add(header);
        List<Integer> tables = new ArrayList<Integer>(foodmap.keySet());
        Collections.sort(tables);
        for(int tableId: tables){
            List<String> row = new ArrayList<>();
            row.add(tableId + "");
            for(String elem: foods){
                Map<String, Integer> sub = foodmap.get(tableId);
                row.add(sub.getOrDefault(elem, 0) + "");
            }
            result.add(row);
        }
        return result;
    }
}
