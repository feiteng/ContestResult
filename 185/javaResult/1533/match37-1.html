import java.util.*;


class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {


        TreeSet<Integer> tables = new TreeSet<>();
        Map<String, Map<Integer, Integer>> map = new TreeMap<>();

        for (List<String> order : orders) {

            int table = Integer.parseInt(
order.get(1));
            tables.add(table);

            String food = order.get(2);
            if (!map.containsKey(food)) {
                map.put(food, new HashMap<>());
            }
            map.get(food).put(table, 1 + map.get(food).getOrDefault(table, 0));
        }

        List<List<String>> ans = new ArrayList<>();
        ans.add(new ArrayList<>());
        ans.get(0).add("Table");
        
ans.get(0).addAll(map.keySet());

        for (int table : tables) {

            List<String> row = new ArrayList<>();
            row.add(Integer.toString(table));
            for (String food : map.keySet()) {
                int cnt = 
map.get(food).getOrDefault(table, 0);
                row.add(Integer.toString(cnt));
            }

            ans.add(row);
        }
        return ans;

    }
}
