class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeSet<String> foodItems = new TreeSet<>();
        TreeMap<Integer, Customer> map = new TreeMap<>();
        
        for(List<String> order: orders){
            int table = Integer.parseInt(order.get(1));
            String food = order.get(2);
            if(map.containsKey(table)){
                Customer c = map.get(table);
                c.orders.put(food, c.orders.getOrDefault(food, 0) + 1);
            }
            else{
                Customer c = new Customer(table);
                c.orders.put(food, 1);
                map.put(table, c);
            }
            foodItems.add(food);
        }
        
        List<List<String>> ans = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for(String s: foodItems){
            header.add(s);
        }
        ans.add(header);
        
        for(int t: map.keySet()){
            List<String> row = new ArrayList<>();
            row.add(String.valueOf(t));
            Customer c = map.get(t);
            for(String food: foodItems){
                row.add(String.valueOf(c.orders.getOrDefault(food, 0)));
            }
            ans.add(row);
        }
        
        return ans;
    }
}

class Customer{
    int table;
    Map<String, Integer> orders;
    
    Customer(int t){
        table = t;
        orders = new HashMap<>();
    }
}
