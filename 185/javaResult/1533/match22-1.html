class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<String, Integer> ct = new HashMap<>();
        Set<String> f = new TreeSet<>();
        TreeMap<Integer, TreeMap<String, Integer>> tf = new TreeMap<>();
        for (List<String> order : orders) {            
            int table = Integer.parseInt(order.get(1));
            String food = order.get(2);
            tf.putIfAbsent(table, new TreeMap<>());
            TreeMap<String, Integer> tableFood = 
tf.get(table);            
            tableFood.put(food, tableFood.getOrDefault(food, 0) + 1);
            f.add(food);
        }
        
        List<List<String>> res = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");        
        for (String food : f) {
            header.add(food);
        }
        res.add(header);
        for (int table : tf.keySet()) {
            List<String> temp = new ArrayList<>();
            temp.add(String.valueOf(table));
            for (String food : f) {
                temp.add(tf.get(table).containsKey(food) ? String.valueOf(
tf.get(table).get(food)) : "0");
            }
            res.add(temp);
        }
        return res;
    }
}
