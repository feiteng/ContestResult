class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, HashMap<String, Integer>> tables = new TreeMap<>();
        TreeSet<String> foods = new TreeSet<>();
        for (List<String> order : orders) {
            String customer = order.get(0);
            Integer table = Integer.valueOf(order.get(1));
            String food = order.get(2);
            if (!tables.containsKey(table)) {
                tables.put(table, new HashMap<>());
            }
            HashMap<String, Integer> rc = tables.get(table);
            rc.put(food, rc.getOrDefault(food, 0) + 1);
            foods.add(food);
        }
        List<List<String>> ans = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for (String food : foods) {
            header.add(food);
        }
        ans.add(header);
        for (Map.Entry<Integer, HashMap<String, Integer>> table : tables.entrySet()) {
            List<String> row = new ArrayList<>();
            for (int i = 0; i < header.size(); ++i) {
                if (i == 0) row.add(table.getKey().toString());
                else {
                    row.add(table.getValue().getOrDefault(header.get(i), 0).toString());
                }
            }
            ans.add(row);
        }
        return ans;
    }
}
