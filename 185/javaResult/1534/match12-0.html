class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        if (croakOfFrogs == null || croakOfFrogs.length() < 5){
            return -1;
        }
        int numberOfFrogs = 1;
        int availabeFrogs = 0;
        int[] croaks = new int[5];
        Map<Character, Integer> index = new HashMap<>();
        index.put('c', 0);
        index.put('r', 1);
        index.put('o', 2);
        index.put('a', 3);
        index.put('k', 4);
        for (char chr : croakOfFrogs.toCharArray()) {
            int tempIndex = index.get(chr);
            croaks[tempIndex]++;
            if(tempIndex > 0 && croaks[tempIndex] > croaks[tempIndex - 1]) {
               return -1; 
            }
            if (chr == 'k') {
               for (int i = 0; i < 5; i++) {
                   croaks[i]--;
                } 
            }
            if (chr == 'c') {
                numberOfFrogs = Math.max(numberOfFrogs, croaks[0]);
            }
        }
        for (int number : croaks) {
            if (number > 0) {
                return -1;
            }
        }
        return numberOfFrogs;
    }
}
