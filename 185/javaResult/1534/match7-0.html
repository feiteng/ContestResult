class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] count = new int[4];
        int out = 0;
        for (char c : croakOfFrogs.toCharArray()) {
            if (c == 'c') {
                count[0]++;
            } else if (c == 'r') {
                if (count[0] == 0) {
                    return -1;
                } else {
                    count[0]--;
                    count[1]++;
                }
            } else if (c == 'o') {
                if (count[1] == 0) {
                    return -1;
                } else {
                    count[1]--;
                    count[2]++;
                }
            } else if (c == 'a') {
                if (count[2] == 0) {
                    return -1;
                } else {
                    count[2]--;
                    count[3]++;
                }
            } else if (c == 'k') {
                if (count[3] == 0) {
                    return -1;
                } else {
                    count[3]--;
                }
            }
            out = Math.max(out, Arrays.stream(count).sum());
        }
        return (Arrays.stream(count).sum() == 0) ? out : -1;
    }
}
