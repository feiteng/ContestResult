class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int n = croakOfFrogs.length();
        int c = 0;
        int r = 0;
        int o = 0;
        int a = 0;
        int k = 0;
        int ans = 0;
        for (int i=0; i<n; i++) {
            switch (croakOfFrogs.charAt(i)) {
                case 'c':
                    if (k == 0) ans++;
                    else k--;
                    c++;
                    
break;
                case 'r':
                    if (c <= 0) return -1;
                    r++;
                    c--;
                    break;
                case 'o':
                    if (r <= 0) return -1;
                    o++;
                    r--;
                    break;
                case 'a':
                    if (o <= 0) return -1;
                    a++;
                    o--;
                    break;
                case 'k':
                    if (a <= 0) return -1;
                    k++;
                    a--;
                    break;
                default:
                    return -1;
            }
        
}
        if (c + r + o + a > 0)
            return -1;
        return ans;
    }
}
