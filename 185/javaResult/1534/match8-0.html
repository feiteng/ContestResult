class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] arr = new int[5];
        for(int i = 0; i < croakOfFrogs.length(); i++) {
            char c = croakOfFrogs.charAt(i);
            if(c == 'c') {
                if(arr[4] > 0) {
                    arr[4] -= 1;
                    arr[0] += 1;
                }
                else arr[0] += 1;
            } else if(c == 'r') {
                if(arr[0] == 0) return -1;
                arr[0] -= 1;
                arr[1] += 1;
            } else if(c == 'o') {
                if(arr[1] == 0) return -1;
                arr[1] -= 1;
                arr[2] += 1;
            } else if(c == 'a') {
                if(arr[2] == 0) return -1;
                arr[2] -= 1;
                arr[3] += 1;
            } else if(c == 'k') {
                if(arr[3] == 0) return -1;
                arr[3] -= 1;
                arr[4] += 1;
            } else return -1;
        }
        for(int i = 0; i < 4; i++) {
            if(arr[i] != 0) return -1;
        }
        return arr[4];
        
    }
}
