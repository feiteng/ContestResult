class Solution {
   public int minNumberOfFrogs(String croakOfFrogs) {
        char[] arr = croakOfFrogs.toCharArray();
        int n = arr.length;
        if(n%5!=0) return -1;
        if(n==0) return 0;
        int ans = 1;
        int[] cnt = new int[256];
        for(char c:arr){
            if(c=='c'){
                cnt['c']++;
            } else if(c=='r'){
                cnt['r'] ++;
                if(cnt['r']>cnt['c']) return -1;
            } else if(c=='o'){
                cnt['o']++;
                if(cnt['o']>cnt['r']) return -1;
            } else if(c=='a'){
                cnt['a']++;
                if(cnt['a']>cnt['o']) return -1;
            } else{
                if(cnt['a']<=0) return -1;
                cnt['a'] --;
                cnt['o'] --;
                cnt['r']--;
                ans = Math.max(ans,cnt['c']);
                cnt['c'] --;
            }
        }
        return Math.max(ans,1);
    }
}
