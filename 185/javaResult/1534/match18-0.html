class Solution {
  public int minNumberOfFrogs(String croakOfFrogs) {
    if (croakOfFrogs.length() % 5 != 0)
      return -1;
    int[] counts = new int[128];
    int cur = 0;
    int res = 0;
    for (int i = 0; i != croakOfFrogs.length(); i++) {
      char ch = croakOfFrogs.charAt(i);
      switch (ch) {
        case 'c':
          cur++;
          counts['r']++;
          break;
        case 'r':
          if (--counts['r'] < 0)
            return -1;
          counts['o']++;
          break;
        case 'o':
          if (--counts['o'] < 0)
            return -1;
          counts['a']++;
          break;
        case 'a':
          if (--counts['a'] < 0)
            return -1;
          counts['k']++;
          break;
        case 'k':
          if (--counts['k'] < 0)
            return -1;
          cur--;
          break;
      }
      res = Math.max(res, cur);
    }
    for (int i : counts)
      if (i != 0)
        return -1;
    return res;
  }
}
