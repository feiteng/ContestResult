import java.util.HashMap;
import java.util.Map;


class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        if (croakOfFrogs.length() % 5 != 0) {
            return -1;
        }
        int ans = 0;
        int cc = 0;
        int[] count = new int[5];
        Map<Character, Integer> p = new HashMap<>();
        p.put('c', 0);
        p.put('r', 1);
        p.put('o', 2);
        p.put('a', 3);
        p.put('k', 4);
        count[0] = Integer.MAX_VALUE;
        for (char c : croakOfFrogs.toCharArray()) {
            int po = p.get(c);
            if (count[po] > 0) {
                count[po]--;
                if (po == 4) {
                    cc--;
                } else {
                    count[po + 1]++;
                
}
                if (po == 0) {
                    cc++;
                    ans = Math.max(cc, ans);
                }
            } else 
{
                return -1;
            }
        }
        return ans;
    }
}
