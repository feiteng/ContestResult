class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int result = 0;
        
        int countC = 0;
        int countR = 0;
        int countO = 0;
        int countA = 0;
        int countK = 0;
        
        for(char c: croakOfFrogs.toCharArray()) {
            if(c == 'c') countC++;
            else if(c == 'r') countR++;
            else if(c == 'o') countO++;
            else if(c == 'a') countA++;
            else if(c == 'k') countK++;
            
            if(!(countC >= countR && countR >= countO && countO >= countA && countA >= countK)) return -1;
            
            if(c == 'k') {
                countC--;
                countR--;
                countO--;
                countA--;
                countK--;
            }
            result = Math.max(result, countC);
        }
        if(countC != 0) return -1;
        
        return result;
    }
}
