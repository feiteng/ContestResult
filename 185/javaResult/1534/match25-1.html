class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int max = 0;
        int[] freq = new int[5];
        
for (char c : croakOfFrogs.toCharArray()) {
            if (c == 'c') {
                freq[0]++;
                max = Math.max(max, freq[0]);
            }
            
if (c == 'r') {
                freq[1]++;
                if (freq[0] < freq[1]) {
                    return -1;
                }
            }
            if (c == 'o') {
                freq[2]++;
                if (freq[1] < freq[2]) {
                    return -1;
                }
            }
            if (c == 'a') {
                freq[3]++;
                if (freq[2] < freq[3]) {
                    return -1;
                }
            }
            if (c == 'k') {
                freq[4]++;
                if (freq[3] < freq[4]) {
                    return -1;
                }
                for (int i = 0; i < 5; i++) {
                    freq[i]--;
                }
            
}
        }
        if (freq[0] > 0) {
            return -1;
        }
        return max;
    }
}
