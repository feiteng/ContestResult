class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] fs = new int[5];
        Map<Character, Integer> M = new HashMap();
        M.put('c', 0);
        M.put('r', 1);
        M.put('o', 2);
        M.put('a', 3);
        M.put('k', 4);
        
int ans = 0;
        for (char c : croakOfFrogs.toCharArray()) {
            int pos = M.get(c);
            if (pos > 0 && fs[pos-1] == 0) return -1;
            if (pos > 0) fs[pos-1]--;
            fs[pos]++;
            if (pos == 4) {
                int curAns = 0;
                for (int f : fs) curAns += f;
                ans = Math.max(ans, curAns);
                fs[pos]--;
            
}
        }
        for (int f : fs)
            if (f > 0) return -1;
        return ans;
    }
}
