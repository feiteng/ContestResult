class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        
Map<Character, Integer> order = new HashMap();
        order.put('c', 0);
        order.put('r', 1);
        order.put('o', 2);
        order.put('a', 3);
        order.put('k', 4);
        int[] dict = new int[5];
        int res = 0;
        int opened=0;
        for(char c : croakOfFrogs.toCharArray()) {
            dict[order.get(c)]++;
            if(!ok(dict)) return -1;
            if(c=='c') opened++;
            else if(c=='k') opened--;
            res = Math.max(res, opened);
        }
        if(opened>0) return -1;
        return res;
    }
    
    boolean ok(int[] dict) {
        
for(int i=1; i<dict.length; i++) {
            if(dict[i] > dict[i-1]) return false;
        }
        return true;
    }
}
