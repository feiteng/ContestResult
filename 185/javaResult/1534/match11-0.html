class Solution {
    // croak
    public int minNumberOfFrogs(String croakOfFrogs) {
        int len = croakOfFrogs.length();
        if(len % 5 != 0) return -1;
        int frogs = len/5;
        int [] state = new int[6];
        state[0] = frogs;
        int result = 0;
        int current = 0;
        for(int i=0;i<len;i+=1){
            char ch = croakOfFrogs.charAt(i);
            if(ch == 'c'){
                if(state[0] ==0) return -1;
                state[0]-=1;
                state[1]+=1;
                current+=1;
            }else if(ch == 'r'){
                if(state[1] ==0) return -1;
                 state[1]-=1;
                state[2]+=1;
                
            }else if(ch == 'o'){
                if(state[2] ==0) return -1;
                 state[2]-=1;
                state[3]+=1;
                
            }else if(ch == 'a'){
                if(state[3] ==0) return -1;
                 state[3]-=1;
                state[4]+=1;
                
            }else{
                if(state[4] ==0) return -1;
                 state[4]-=1;
                state[5]+=1;
                current-=1;
            }
            result = Math.max(result, current);
            // for(int s: state){
            //      System.out.print(s+",");
            // }
            // System.out.println();
        }
        
        
        return state[5] == frogs?result:-1;
        
        
        
    }
}
