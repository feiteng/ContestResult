class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int maxTotal = 0;
        int c = 0, r = 0, o = 0, a = 0;
        int total = 0; // c + r + o + a
        boolean bad = false;
        for(char curr: croakOfFrogs.toCharArray()) {
            boolean goodNow = true;
            
if(curr == 'c') {
                c++;
                total++;
                maxTotal = Math.max(maxTotal, total);
            } else if(curr == 'r') {
                if(c == 0) {
                    goodNow = false;
                } else {
                    c--;
                    r++;
                }
            } else if(curr == 'o') {
                if(r == 0) {
                    goodNow = false;
                } else {
                    r--;
                    o++;
                }
            } else if(curr == 'a') {
                if(o == 0) {
                    goodNow = false;
                } else {
                    o--;
                    a++;
                }
            } else {
                // now curr == 'k'
                if(a == 0) {
                    goodNow = false;
                } else {
                    a--;
                    total--;
                }
            }
            if(!goodNow) {
                bad = true;
                break;
            
}
        }
        if(bad || total > 0) {
            return -1;
        }
        return maxTotal;
    }
}
