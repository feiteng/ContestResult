class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {

        if (croakOfFrogs.length() % 5 != 0) {
            return -1;
        }


        
int ans = 0;
        int[] cnt = new int[5];

        for (char c : croakOfFrogs.toCharArray()) {

            int i = getNum(c);

            if (i == 0) {
                cnt[0]++;
                ans = Math.max(ans, sum(cnt));
            } else if (cnt[i - 1] == 0) {
                return -1;
            } else {

                cnt[i - 1]--;
                if (i != 4) {
                    cnt[i]++;
                }
            }
        }
        return ans;

    }

    private int sum(int[] cnt) {
        int sum = 0;
        for (int n : cnt) {
            sum += n;
        
}
        return sum;
    }

    private int getNum(char c) {
        if (c == 'c') {
            return 0;
        } else if (c == 'r') {
            return 1;
        } else if (c == 'o') {
            return 2;
        } else if (c == 'a') {
            return 3;
        } else if (c == 'k') {
            return 4;
        }
        return -1;
    }
}
