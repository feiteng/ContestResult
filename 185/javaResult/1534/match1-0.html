class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int n = croakOfFrogs.length();
        if(n % 5 != 0)
            return -1;
        int min = -1;
        int[] count = new int[4];
        for(char c : croakOfFrogs.toCharArray()){
            if(c =='c'){
                count[0] ++;
            }
            else if(c == 'r'){
                if(count[1] >= count[0])
                    return -1;
                count[1]++;
            }
            else if(c == 'o'){
                if(count[2] >= count[1])
                    return -1;
                count[2]++;
            }
            else if(c == 'a'){
                if(count[3] >= count[2])
                    return -1;
                count[3]++;
            }
            else{
                if(count[3] == 0)
                    return -1;
                min = Math.max(min, count[0]);
                for(int i=0;i<4;++i)
                    count[i]--;
            }
        }
        return min;
    }
}
