class Solution {
	
    public int minNumberOfFrogs(String croakOfFrogs) {
		String s = "croak";
		int ans = 0;
		int cnt[] = new int[5];
		for (char c:croakOfFrogs.toCharArray()) {
			int i = s.indexOf(c);
			if (i==0)
				cnt[i]++;
			else {
				if (cnt[i-1]>0) {
					cnt[i-1]--;
					cnt[i]++;
				}
				else
					return -1;
			}
			ans = Math.max(ans, cnt[0]+cnt[1]+cnt[2]+cnt[3]);
		}
		for (int i=0; i<4; i++)
			if (cnt[i]>0)
				return -1;
		return ans;
        
    }
}
