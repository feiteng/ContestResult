class Solution {
    public int minNumberOfFrogs(String s) {
        
    int count=0;
int ans=0;
    int q=0,u=0,a=0,c=0,k=0;
    for(int i=0;i<s.length();i++){
      if(s.charAt(i)=='c'){
        count++;
      }else if(s.charAt(i)=='k'){
        count--;
      }
      switch(s.charAt(i)){
        case 'c':q++;break;
        case 'r':u++;break;
        case 'o':a++;break;
        case 'a':c++;break;
        case 'k':k++;break;
      }
      if(u>q||a>u||c>a||k>c)return -1;
      if(count<0)return -1
;
      ans=Math.max(ans,count);
    }
    if(count!=0)return -1;
    return ans;
    }
}
