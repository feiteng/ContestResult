class Solution {
    public int minNumberOfFrogs(String s) {
        int max = 0;
        int total = 0;
        
        Map<Character, Integer> count = new HashMap<>();
        
        for (int i =0 ; i < s.length();i++){
            char c = s.charAt(i);
            //System.out.println(count.toString());
            if (c != 'c' && !count.containsKey(c)) return -1;
            if (c == 'c'){
                count.put('r', count.getOrDefault('r', 0) + 1);
                total++;
            }
            else if (c == 'r'){
                
count.put('o', count.getOrDefault('o', 0) + 1);
                count.put('r', count.get('r') - 1);
                if (count.get('r') == 0) count.remove('r');
            }
            else if (c == 'o'){
                
count.put('a', count.getOrDefault('a', 0) + 1);
                count.put('o', count.get('o') - 1);
                if (count.get('o') == 0) count.remove('o');
            }
            else if (c == 'a'){
                
count.put('k', count.getOrDefault('k', 0) + 1);
                count.put('a', count.get('a') - 1);
                if (count.get('a') == 0) count.remove('a');
            }
            else if (c == 'k'){
                
count.put('k', count.getOrDefault('k', 0) - 1);
                if (count.get('k') == 0) count.remove('k');
                total--;
            }
            max = Math.max(max, total);
        }
        
        if (count.size() > 0) return -1;
        return max;
    }
}
