class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {

        int[] count = new int[128];
        
        String s = croakOfFrogs;
        char[] ch = s.toCharArray();
        int len = s.length();
        if(len%5!=0) return -1;
        
        for(char c : ch){
            count[c] += 1;
        }
        int num = len/5;
        if(count['c']!=num || count['r']!=num || count['o']!=num || count['a']!=num || count['k']!=num){
            return -1;
        }
        
        Arrays.fill(count,0);
        
        int cur = 0;
        int res = 0;
        
        for(char c : ch){
            if(c=='c'){
                if(count['k']>0){
                    count['k'] -= 1;
                }
                count[c] += 1;
            } else 
if(c=='r'){
                if(count['c']==0) return -1;
                count['c'] -= 1;
                count['r'] += 1;
            }
            else if(c=='o'){
                if(count['r']==0) return -1;
                count['r'] -= 1;
                count['o'] += 1;
            }
            else if(c=='a'){
                if(count['o']==0) return -1;
                count['o'] -= 1;
                count['a'] += 1;
            }
            else if(c=='k'){
                if(count['a']==0) return -1;
                count['a'] -= 1;
                count['k'] += 1;
            
}
            res = getMax(count);
        }
        return res;
    }
    
    public int getMax(int[] count){
        int max = 0;
        for(int c : count){
            max = Math.max(c,max);
        }
        return max;
    }
}
