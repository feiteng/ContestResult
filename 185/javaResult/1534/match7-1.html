class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] sum = new int[256];
        
int max = 0, frog = 0;
        for (char c : croakOfFrogs.toCharArray()) {
            if (c == 'c') {
                frog += 1;
            } else if (c == 'r') {
                if (sum['c'] == 0) {
                    return -1;
                } else {
                    sum['c'] -= 1;
                }
            } else if (c == 'o') {
                if (sum['r'] == 0) {
                    return -1;
                } else {
                    sum['r'] -= 1;
                }
            } else if (c == 'a') {
                if (sum['o'] == 0) {
                    return -1;
                } else {
                    sum['o'] -= 1;

                }
            } else if (c == 'k') {
                if (sum['a'] == 0) {
                    return -1;
                } else {
                    sum['a'] -= 1;
                    sum[c] -= 1;
                    --frog;
                }
            }
            sum[c] += 1;
            max = Math.max(frog, max);
        }
        return sumNum(sum) == 0 ? max : -1;
    }

    private int sumNum(int[] sum) {
        int res = 0;
        for (int i : sum) {
            res += i;
        }
        return res;
    }
}
