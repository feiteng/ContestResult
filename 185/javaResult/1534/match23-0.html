class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int res = 0, count = 0;
        int[] cnt = new int[5];
        for(char c: croakOfFrogs.toCharArray()){
            int id = getIndex(c);
            if(id == 0){
                count++;
                cnt[id]++;
            } else {
                if(cnt[id - 1] == 0){
                    return -1;
                } 
                cnt[id - 1]--;
                cnt[id]++;
                if(id == 4){
                    count--;
                }
            }
            res = Math.max(res, count);
        }
        for(int i = 0; i < 4; i++){
            if(cnt[i] != 0){
                return -1;
            }
        }
        return res;
    }
    
    private int getIndex(char c){
        if(c == 'c'){
            return 0;
        } else if(c == 'r'){
            return 1;
        } else if(c == 'o'){
            return 2;
        } else if(c == 'a'){
            return 3;
        } else if(c == 'k'){
            return 4;
        } else {
            return -1;
        }
    }
}
