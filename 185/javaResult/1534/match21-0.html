class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        Map<Character, Character> map = new HashMap<>();
        map.put('c', 'r');
        map.put('r', 'o');
        map.put('o', 'a');
        map.put('a', 'k');
        map.put('k', '_');

        int cnt = 0;
        int res = 0;
        Map<Character, Integer> map_cnt = new HashMap<>();
        for (char c : croakOfFrogs.toCharArray()) {
            if (c == 'c') {
                map_cnt.putIfAbsent(c, 0);
                map_cnt.put(c, map_cnt.get(c) + 1);

                cnt ++;
            } else if (c == 'r') {
                if (!map_cnt.containsKey('c')) return -1;

                map_cnt.putIfAbsent(c, 0);
                map_cnt.put(c, map_cnt.get(c) + 1);

                map_cnt.put('c', map_cnt.get('c') - 1);
                if (map_cnt.get('c') == 0) map_cnt.remove('c');
            } else if (c == 'o') {
                if (!map_cnt.containsKey('r')) return -1;

                map_cnt.putIfAbsent(c, 0);
                map_cnt.put(c, map_cnt.get(c) + 1);

                map_cnt.put('r', map_cnt.get('r') - 1);
                if (map_cnt.get('r') == 0) map_cnt.remove('r');
            } else if (c == 'a') {
                if (!map_cnt.containsKey('o')) return -1;

                map_cnt.putIfAbsent(c, 0);
                map_cnt.put(c, map_cnt.get(c) + 1);

                map_cnt.put('o', map_cnt.get('o') - 1);
                if (map_cnt.get('o') == 0) map_cnt.remove('o');
            } else if (c == 'k') {
                if (!map_cnt.containsKey('a')) return -1;

                // map_cnt.putIfAbsent(c, 0);
                // map_cnt.put(c, map_cnt.get(c) + 1);

                map_cnt.put('a', map_cnt.get('a') - 1);
                if (map_cnt.get('a') == 0) map_cnt.remove('a');

                cnt --;
            }

            res = Math.max(res, cnt);
            // System.out.println(map_cnt.toString());
        }

        if (map_cnt.get('c') != null || map_cnt.get('r') != null ||
        map_cnt.get('o') != null || map_cnt.get('a') != null) return -1;

        return res;
    }
}
