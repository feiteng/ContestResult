class Solution {
    int res;
    int[] cnt;
    
public int minNumberOfFrogs(String croakOfFrogs) {
        int n = croakOfFrogs.length();
        if(n % 5 != 0) return -1;
        int maxR = 0;
        res = 0;
        cnt = new int[5];
        for(char x : croakOfFrogs.toCharArray()) {
            if(x == 'c') {
                ++cnt[0];
                ++res;
            
} else if(x == 'r') {
                if(cnt[0] == 0) return -1;
                --cnt[0];
                ++cnt[1];
            } else if(x == 'o') {
                if(cnt[1] == 0) return -1;
                --cnt[1];
                ++cnt[2];
            } else if(x == 'a') {
                if(cnt[2] == 0) return -1;
                --cnt[2];
                ++cnt[3];
            } else {
                if(cnt[3] == 0) return -1;
                --cnt[3];
                if(res > maxR) maxR = res;
                --res;
            }
        }
        return maxR;
    }
}
