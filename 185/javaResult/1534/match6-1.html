class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] count = new int[4];
        int max = 0;
        for (char c: croakOfFrogs.toCharArray()) {
            switch (c) {
                case 'c':
                    count[0]++;
                    
break;
                case 'r':
                    if (count[0] == 0) return -1;
                    count[0]--;
                    count[1]++;
                    break;
                case 'o':
                    if (count[1] == 0) return -1;
                    count[1]--;
                    count[2]++;
                    break;
                case 'a':
                    if (count[2] == 0) return -1;
                    count[2]--;
                    count[3]++;
                    break;
                case 'k':
                    if (count[3] == 0) return -1;
                    count[3]--;
                    break;
            }
            max = Math.max(max, count[0]+count[1]+count[2]+count[3]);
        }
        if (count[0]+count[1]+count[2]+count[3] != 0) return -1;
        return max;
    }
}
