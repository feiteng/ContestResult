class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int res = 0;
        int[] count = new int[5];
        int num = 0;
        for(char c : croakOfFrogs.toCharArray()){
            int i = "croak".indexOf(c);
            if(i < 0)return -1;
            if(i == 0){
                ++count[0];
                if(++num > res)res = num;
            }else{
                if(count[i-1] == 0)return -1;
                count[i-1]--;
                count[i]++;
                if(i == 4)num--;
            }
        }
        for(int i = 0; i < 4; i++)if(count[i] != 0)return -1;
        return res;
    }
    
    
}
