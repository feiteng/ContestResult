class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int n = croakOfFrogs.length();
        Queue<Integer>[] q = new Queue[5];
        for(int i = 0; i < 5; i++)
            q[i] = new LinkedList<>();
        for(int i = 0; i < n; i++)
        {
            char c = croakOfFrogs.charAt(i);
            if(c == 'c') q[0].add(i);
            if(c == 'r') q[1].add(i);
            if(c == 'o') q[2].add(i);
            if(c == 'a') q[3].add(i);
            if(c == 'k') q[4].add(i);
        }

        int size = q[0].size();
        
        int[] preSum = new int[100010];
        while(!q[0].isEmpty())
        {
            try{
            int from = q[0].peek(),
                to = q[4].peek();
            int run = from;
                for(int i = 0; i < 5; i++)
                {
                	if(q[i].isEmpty()) return -1;
                    int val = q[i].poll();
                    if(run > val) return -1;
                    run = val;
                }
                preSum[from]++;
                preSum[to + 1]--;
            }
            catch(Exception e)
            {
                return -1;
            }
            
        }
        for(int i = 0; i < 5; i++)
        {
            if(!q[i].isEmpty()) return -1;
        }
        int sum = 0;
        int maxi = 0;
        for(int i = 0; i <= 100000; i++)
        {
            sum += preSum[i];
            maxi = Math.max(maxi, sum);
        }
        return maxi;
    }    
}
