class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] count = new int[128];
        int res = 0;
        int cur = 0;
        for (char mid: croakOfFrogs.toCharArray()) {
            if (mid == 'c') {
                count[mid] ++;
                cur ++;
            }
            if (mid == 'r') {
                if (count['c'] > 0) {
                    count['r'] ++;
                    
                    count['c'] --;
                } else return -1;
            } if (mid == 'o') {
                if (count['r'] > 0) {
                    count['o'] ++;
                    count['r'] --;
                } else return -1;
            } if (mid == 'a') {
                if (count['o'] > 0) {
                    count['a'] ++;
                    count['o'] --;
                } else return -1;
            } if (mid == 'k') 
{
                if (count['a'] > 0) {
                    count['a'] --;
                } else return -1;
                cur --;
            }
            res = Math.max(res, cur);
        }
        for (int c: count) if (c != 0) return -1;
        return res;
    }
}
