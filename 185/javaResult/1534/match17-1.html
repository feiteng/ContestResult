class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        
int c = 0, r = 0, o = 0, a = 0, ans = 0;
        for (char ch : croakOfFrogs.toCharArray()) {
            switch (ch) {
                case 'c': c++; break;
                case 'r': if (r >= c) return -1; r++; break;
                case 'o': if (o >= r) return -1; o++; break;
                case 'a': if (a >= o) return -1; a++; break;
                case 'k': if (c <= 0 || r <= 0 || o <= 0 || a <= 0) return -1; c--; r--; o--; a--; break;
            }
            int min = Math.min(Math.min(c, r), Math.min(o, a));
            if (min < 0) {
                return -1;
            }
            ans = Math.max(ans, Math.max(Math.max(c, r), Math.max(o, a)));
        }
        return c == 0 && r == 0 && o == 0 && a == 0 ? ans : -1;
    }
}
