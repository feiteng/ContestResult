class Solution {
    public int minNumberOfFrogs(String croaks) {
        // Stack<Integer> s = new Stack<>();
        char[] c = croaks.toCharArray();
        Map<Character, Integer> map= new HashMap<>();
        map.put('c',0); map.put('r',1);
        map.put('o',2); map.put('a',3); map.put('k',4);
        int[] counts = new int[5];
        int ans = 0, open = 0;
        for (int i = 0; i < c.length; i++) {
            if (!map.containsKey(c[i]))    return -1;
            if (c[i] == 'c') {  counts[map.get(c[i])]++; open++; }
            else if (c[i] == 'k') {
                counts[0]--;counts[1]--;counts[2]--;counts[3]--;
                open--;
            }
            else {
                if (counts[map.get(c[i])] >= counts[map.get(c[i])-1])   {
                    // System.out.println("here " + c[i] + " " + i + " " + counts[0]);
                    return -1;   
                }
                counts[map.get(c[i])]++;
            }
            // if (c[i] == 'k') {
            //     if (s.isEmpty())    return -1;
            //     else {  s.pop(); open--;}
            // }
            ans = Math.max(ans, open);
        }
        
        // counts[0];
        for (int i = 0; i < 5; i++) {
            if (counts[i] != counts[0])   return -1;
        }
        
        return ans;
    }
}
