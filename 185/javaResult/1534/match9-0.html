import java.util.Arrays;
import java.util.HashMap;

class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] frogs = new int[4];
        int f = 0;
        int ret = 0;
        for (char c : croakOfFrogs.toCharArray()) {
            if (c == 'c') {
                frogs[0]++;
                f++;
                ret = Math.max(ret, f);
            } else if (c == 'k') {
                if (frogs[3]-- == 0) return -1;                
                f--;
            } else {
                int idx = c == 'r' ? 1 : (c == 'o' ? 2 : (c == 'a' ? 3 : -1));
                if (frogs[idx-1]-- == 0) return -1;                
                frogs[idx] ++;
            }
            // System.out.println(Arrays.toString(frogs));
        }
        for (int ff : frogs){
            if (ff != 0) return -1;
        }
        return ret;
    }
}
