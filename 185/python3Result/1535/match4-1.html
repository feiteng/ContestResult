class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if m < k:
            return 0
        dp = [[[0] * m for _ in range
(k)] for _ in range(n)]
        for k1 in range(m):
            dp[0][0][k1] = 1
        for i in range(1, n):
            for j in range(k):
                for k1 in range(m):
                    dp[i][j][k1] += dp[i-1][j][k1] * (k1 + 1)
            for j in range(k-1):
                for k1 in range(m-1):
                    
for k2 in range(k1+1, m):
                        dp[i][j+1][k2] += dp[i-1][j][k1]
        return sum(dp[-1][-1]) % (10**9 + 7)
