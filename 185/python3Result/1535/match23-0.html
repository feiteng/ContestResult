class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        @lru_cache(None)
        def dfs(cur_max, comp, l):
            if l == n:
                return 1 if comp == k else 0
            elif (m - cur_max) < (k - comp):
                return 0
            else:
                res = 0
                for x in range(1, m + 1):
                    if x > cur_max:
                        res += dfs(x, comp + 1, l + 1)
                    else:
                        res += dfs(cur_max, comp, l + 1)
                    res %= M
                return res % M
        M = 10 ** 9 + 7   
        return dfs(-1, 0, 0)
