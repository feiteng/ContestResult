class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        t = [[[0 for i in range(k+1)] for j in range
(m+1)] for t in range(n)] #(n, max, k)
        for i in range(1, m+1):
            t[0][i][1] = 1
        for i in range(1, n):
            for j in range(1, m+1):
                for s in range(1, k+1):
                    if t[i-1][j][s] > 0:
                        t[i][j][s] = (t[i][j][s] + t[i-1][j][s]*j) % 1000000007
                        for u in range(j+1, m+1):
                            if s < k:
                                t[i][u][s+1] = (t[i][u][s+1] + t[i-1][j][s
]) % 1000000007
        summ = 0
        for i in range(1, m+1):
            summ = (summ + t[n-1][i][k]) % 1000000007
        return summ
