class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if k>n or k>m:
            return 0
        
        dp = [[[0]*(m+1) for _ in range(k+1)] for _ in range(n)]
        for i in range(m):
            dp[0][1][i+1]=1
        
        for i in range(1,n):
            for j in range(1,k+1):
                for l in range(1,m+1):
                    dp[i][j][l]+=dp[i-1][j][l]*l
                    dp[i][j][l]%=1000000007
                    
                    if j+1>k:
                        continue
                    for a in range(1,l):
                        dp[i][j+1][l]+=dp[i-1][j][a]
                        dp[i][j+1][l]%=1000000007
        
        cnt=0
        for i in range(1,m+1):
            cnt+=dp[n-1][k][i]
            cnt%=1000000007
        return cnt
