from functools import lru_cache
class Solution:
    def numOfArrays(self, n: int, m: int, K: int) -> int:
        MOD = 1000000007
        
        @lru_cache(None)
        def helper(idx, lower, k):
            #print(idx, lower, k)
            if k < 0: return 0
            if idx == n: return k == 0
            
            res = 0
            for i in range(1, m + 1):
                res = (res + helper(idx + 1, max(i, lower), k - (i > lower))) % MOD
            #print(idx, lower, k, res)                
            return res                
            
        return helper(0, 0, K)
