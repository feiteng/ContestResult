from functools import lru_cache
class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = 10**9 + 7
        
        @lru_cache(None)
        def dfs(i, currk, currmax):
            if i == n:
                return 1 if currk == k else 0
            cnt = 0
            for d in range(1, m+1):
                if d > currmax:
                    delta = d - currmax
                    cnt 
+= dfs(i+1, currk + 1, d)
                else:
                    cnt += dfs(i+1, currk, currmax)
            return cnt
        return dfs(0, 0, 0) % MOD
