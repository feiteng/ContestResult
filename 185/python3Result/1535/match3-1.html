class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = 1000000007
        
        if k == 0 or k > m:
            return 0
        f 
= [[[0 for _ in range(k+1)] for _ in range(m+1)] for _ in range(n+1)]
        
        for mm in range(1, m+1):
            f[1][mm][1] = 1
        
        for ii in range(2, n+1):
            for kk in range(1, k+1):
                for mm in range(1, m+1):
                    f[ii][mm][kk] += f[ii-1][mm][kk] * mm
                    for j in range(1, mm):
                        f[ii][mm][kk] += f[ii-1][j][kk-1]
        
        ans = 0
        for mm in range(1, m+1):
            ans += f[n][mm][k] % MOD
        return ans % MOD
