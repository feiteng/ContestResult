class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        dp = [[[0] * k for i in range(m)] for j in range(n)]
        for i in range(m):
            dp[0][i][0] = 1
        for i in range(1,n):
            for j in range(m):
                for t in range(k):
                    for tmp in range(m):
                        if tmp > j and t+1 < k:
                            dp[i][tmp][t+1] += dp[i-1][j][t]
                        elif tmp <= j:
                            dp[i][j][t] += dp[i-1][j][t]
        # print(dp)
        return sum(x[k-1] for x in dp[n-1]) % (10 ** 9 + 7)
