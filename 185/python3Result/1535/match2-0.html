class Solution:
    def numOfArrays(self, n: int, m: int, K: int) -> int:
        if m < K:return 0
        mod = 10 ** 9 + 7
        f = [[[0 for _ in range(K+1)] for _ in range(m+1)] for _ in range(n)]
        for i in range(n):
            for j in range(1, m+1):
                for t in range(1, K+1):
                    if i == 0:
                        if t <= 1:f[i][j][t] = 1
                        continue
                    for k in range(1, j+1):
                        if k < j:
                            if t > 1:f[i][j][t] += f[i-1][k][t-1]
                        else:f[i][j][t] += j * f[i-1][k][t]
                        
        return sum([f[n-1][i][K] for i in range(1, m+1)]) % mod
