class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = 1000000007
        @functools.lru_cache(None)
        def dfs(n, k, start):
            # print(n, k, start)
            if n == 0:
                return k == 0
            if k == 0:
                return (start ** n) % MOD
            res = 0
            for i in range(start+1, m+1):
                res += dfs(n-1, k-1, i)
            if start > 0:
                res += dfs(n-1, k, start) * start
            return res % MOD
        return dfs(n, k, 0) % MOD
