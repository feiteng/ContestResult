class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        
        @lru_cache(None)
        def f(i, mx, cmp):
            if i == n:
                if cmp == 0:
                    return 1
                else:
                    return 0
            elif cmp < 0:
                
return 0
            else:
                ret = 0
                for mm in range(1, m + 1):
                    if mm > mx:
                        ret += f(i + 1, mm, cmp - 1)
                    else:
                        ret += f(i + 1, mx, cmp)
                    ret %= 1000000007
                return ret
            
        return f(0, -1, k)
