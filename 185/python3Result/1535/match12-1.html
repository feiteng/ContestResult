from functools import lru_cache
class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = int(1e9 + 7)
        
@lru_cache(None)
        def helper(i, v, k):
            if i == n: return k == 0
            if v == m and k > 0: return 0
            ans = 0
            for x in range(1, m + 1):
                ans += helper(i + 1, max(x, v), k - (x > v))
            return ans
        return helper(0, -1, k) % MOD
