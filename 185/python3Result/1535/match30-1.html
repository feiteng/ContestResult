class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        maxn, maxm = 55, 110
        mod = int(1e9 + 7)
        
        dp 
= [[[0]*(m+1) for _ in range(k+1)] for i in range(n)]
        for i in range(1, m + 1): dp[0][1][i] = 1
            
        for i in range(1, n):
            for p in range(1, k + 1):
                for j in range(1, m + 1):
                    dp[i][p][j] = (dp[i - 1][p - 1][j - 1] + dp[i - 1][p][j] * j) % mod
                    dp[i - 1][p][j] += dp[i - 1][p][j - 1]
        return sum(dp[n - 1][k]) % mod
