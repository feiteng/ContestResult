class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = 100_000_000_7
        dict_ = {}
        def helper(n, k, pre_max):
            if m < k or k < 0:
                return 0
            if not n:
                return 1 if not k else 0
            if (n, k, pre_max) not in dict_:
                ret = 0
                for i in range(pre_max+1, m+1):
                    ret += helper(n-1, k-1, i)
                ret += helper(n-1, k, pre_max)*pre_max
                ret %= MOD
                dict_[(n, k, pre_max)] = ret
            return dict_[(n, k, pre_max)]
        
        return helper(n, k, 0)
