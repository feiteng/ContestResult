class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        modulo = 1000000007
        
if k is 0:
            return 0
        dp = [[[0] * (k+2) for _ in range(m+1)] for __ in range(n+1)]  # dp[pos][max][current]
        for i in range(1, m+1): dp[0][i][1] = 1
        for pos in range(n):
            for i in range(1, m+1):
                for c in range(1, k+1):
                    if dp[pos][i][c]:
                        dp[pos+1][i][c] += dp[pos][i][c] * i
                        dp[pos+1][i][c] %= modulo
                        
for j in range(i+1, m+1):
                            dp[pos+1][j][c+1] += dp[pos][i][c]
                            dp[pos+1][j][c+1] %= modulo
        return sum(dp[n-1][i][k] for i in range(1, m+1)) % modulo
