class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        import functools

        @functools.lru_cache(None)
        def solve(pre,x,k):
            if k>x:
                return 0
            if x==0:
                return 1
            res=0
            for i in range(m):
                if i<=pre:
                    res+=solve(pre,x-1,k)
                elif k>0:
                    res+=solve(i,x-1,k-1)
            return res%(10**9+7)

        return solve(-1,n,k)
