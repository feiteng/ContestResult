class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if k > m:
            return 0
        
        mod = 1000000007
        
        @lru_cache(None)
        def gen(n, cur_m, k):
            if k == 0:
                return cur_m ** n
                # return 1
            if k > n:
                return 0
            
            res = 0
            for i in range(1, m+1):
                res += gen(n-1, max(i, cur_m), k if i <= cur_m else k-1)
                
            return res
        
        res = 0
        for i in range(1, m+1):
            # res = (res + gen(n-1, i, k-1)) % mod
            res += gen(n-1, i, k-1)
            # print(i, res)
            
        return res % mod
