class Solution:
    def numOfArrays(self, n: int, m: int, K: int) -> int:
        """
        state: dp[i][j][p] = len i, max num j, cost k
        transition: 
        init: dp[0][m][0] = 1
        final: dp[n][m][k]
        """
        MOD = 10 ** 9 + 7
        
        ans = 0
        
        if K == 0:
            return 0

        dp 
= [[[0] * (K + 1) for _ in range(m + 1)] for _ in range(n + 1)]
        for j in range(1, m + 1):
            dp[1][j][1] = 1
        
        for i in range(2, n + 1):
            for j in range(1, m + 1):
                for k in range(1, min(j+1, K + 1)):
                    s = 0
                    for l in range(1, j):
                        s += dp[i-1][l][k-1]
                    dp[i][j][k] = (s + dp[i-1][j][
k] * j) % MOD
        
        ans = 0
        for j in range(1, m + 1):
            ans = (ans + dp[n][j][K]) % MOD
        
        return ans % MOD
