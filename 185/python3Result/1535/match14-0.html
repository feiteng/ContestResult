from functools import lru_cache
class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        @lru_cache(None)
        def dfs(i, k,premax):
            if k<0:
                return 0
            if i >n and k == 0:
                return 1
            elif i>n and k>0:
                return 0
            res = 0
            for j in range(1,m+1):
                # print(j)
                if j <= premax:
                    res+=dfs(i+1,k,premax)
                else:
                    res+=dfs(i+1,k-1,j)
                    
            return res%1000000007
        
        return dfs(1,k,0)
