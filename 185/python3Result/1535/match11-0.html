class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = int(1e9 + 7)
        
        sys.setrecursionlimit(1000000)
        from functools import lru_cache
        
        @lru_cache(None)
        def dfs(idx, mx, k):
            if idx == n: return k == 0
            if k < 0: return 0
            res = 0
            for num in range(1, m + 1):
                res += dfs(idx + 1, max(num, mx), k - (num > mx))
            return res % MOD
        

        return dfs(0, -1, k)
