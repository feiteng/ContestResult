class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        memo = {}
        def helper(n, m, k, curr_max):
            if (n, m, k, curr_max) in memo:
                return memo[(n, m, k, curr_max)]
            if m == k == 1:
                return 1
            if m < k:
                return 0
            if m == curr_max and k > 0:
                return 0
            if k == 0:
                if n == 0:
                    return 1
                else:
                    return curr_max ** n
            if n == 0:
                return 0
            res = 0
            for i in range(1, m+1):
                if i > curr_max:
                    res += helper(n-1, m, k-1, i)
                else:
                    res += helper(n-1, m, k, curr_max)
            N = 10 ** 9 + 7
            memo[(n, m, k, curr_max)] = res % N
            return memo[(n, m, k, curr_max)]
        
        return helper(n, m, k, -1)
