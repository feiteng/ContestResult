class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if k == 0: return 0
        # p[a][b][c] = a 1..n, b current max >= a, c current cost
        p = [[[0] * (k + 1) for _ in range(m + 1)] for _ in range(n)]
        modu = 10 ** 9 + 7
        
for b in range(1, m + 1):
            p[0][b][1] = 1

        for a in range(0, n - 1):
            for b in range(1, m + 1):
                for c in range(1, k + 1):
                    if p[a][b][c] > 0:
                        p[a + 1][b][c] = (p[a + 1][b][c] + p[a][b][c] * b) % modu
                        if c == k: continue
                        
for d in range(b + 1, m + 1):
                            p[a + 1][d][c + 1] = (p[a + 1][d][c + 1] + p[a][b][
c]) % modu

        return sum([p[n - 1][b][k] for b in range(m + 1)]) % modu
