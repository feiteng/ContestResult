class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if m < k or n < k:
            return 0
        if m == 1:
            return 1
        if n == 1:
            return m
        mod = 10 ** 9 + 7
        visited = collections.defaultdict(int)
        def dfs(n, k, pre):
            if n == 0:
                if k == 0:
                    return 1
                else:
                    return 0
            
if k < 0:
                return 0
            if m < 0:
                return 0
            if (n, k, pre) not in visited:
                cur = 0
                for i in range(pre + 1, m + 1):
                    cur += dfs(n - 1, k - 1, i)
                cur += dfs(n - 1, k, pre) * pre
                cur = cur % mod
                visited[(n, k, pre)] = cur
            return visited[(n, k, pre)]
        return dfs(n, k, 0)
