class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        curCnt = 0
        maxCnt = 0
        cntList = [0]*4
        for char in croakOfFrogs:
            if char == "c":
                cntList[0] += 1
                curCnt += 1
            elif char == "r":
                if cntList[0] <= 0:
                    return -1
                else:
                    cntList[0] -= 1
                    cntList[1] += 1
            elif char == "o":
                if cntList[1] <= 0:
                    return -1
                else:
                    cntList[1] -= 1
                    cntList[2] += 1        
            elif char == "a":
                if cntList[2] <= 0:
                    return -1
                else:
                    cntList[2] -= 1
                    cntList[3] += 1
            elif char == "k":
                if cntList[3] <= 0:
                    return -1
                else:
                    cntList[3] -= 1
                    curCnt -= 1
            else:
                return -1
            maxCnt = max(curCnt, maxCnt)
        for num in cntList:
            if num != 0:
                return -1
        return maxCnt
