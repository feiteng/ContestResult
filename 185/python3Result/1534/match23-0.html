class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        stats = [0, 0, 0, 0, 0]
        ans = 0
        for i in croakOfFrogs:
            if i == 'c':
                stats[0] += 1
            elif i == 'r':
                if stats[0] == 0:
                    return -1
                stats[0] -= 1
                stats[1] += 1
            elif i == 'o':
                if stats[1] == 0:
                    return -1
                stats[1] -= 1
                stats[2] += 1
            elif i == 'a':
                if stats[2] == 0:
                    return -1
                stats[2] -= 1
                stats[3] += 1
            elif i == 'k':
                if stats[3] == 0:
                    return -1
                stats[3] -= 1
                stats[4] += 1
            ans = max(ans, stats[0] + stats[1] + stats[2] + stats[3])
        if stats[0] > 0 or stats[1] > 0 or stats[2] > 0 or stats[3] > 0:
            return -1
        return ans
