class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        cnt=[0,0,0,0,0]
        r = 0
        for ch in croakOfFrogs:
            if ch == 'c':
                cnt[0] += 1
            elif ch == 'r':
                if cnt[0] > 0:
                    cnt[0] -= 1
                    cnt[1] += 1
                else:
                    return -1
            elif ch == 'o':
                if cnt[1] > 0:
                    cnt[1] -= 1
                    cnt[2] += 1
                else:
                    return -1
            elif ch == 'a':
                if cnt[2] > 0:
                    cnt[2] -= 1
                    cnt[3] += 1
                else:
                    return -1
            elif ch == 'k':
                if cnt[3] > 0:
                    r = max(r, sum(cnt[0:-1]))
                    cnt[3] -= 1
                    cnt[4] += 1
                else:
                    return -1
            else:
                return -1
            print(cnt)
        if sum(cnt[0:-1]) != 0:
            return -1
        return r
