class Solution:
    def minNumberOfFrogs(self, A: str) -> int:
        cnt = Counter(A)
        if not cnt['c'] == cnt['r'] == cnt['o'] == cnt['a'] == cnt['k']:
            return -1
        d = {}
        for i in 'croak':
            d[i] = 0
        mx = 0
        for i in A:
            if i == 'c':
                d['c'] += 1
            elif i == 'r':
                d['r'] += 1
                d['c'] -= 1
                if d['c'] < 0:
                    return -1
            elif i == 'o':
                d['o'] += 1
                d['r'] -= 1
                if d['r'] < 0:
                    return -1
            elif i == 'a':
                d['a'] += 1
                d['o'] -= 1
                if d['o'] < 0:
                    return -1
            elif i == 'k':
                d['a'] -= 1
                if d['a'] < 0:
                    return -1
            mx = max(mx, d['c']+d['r']+d['o']+d['a']+d['k'])
        return mx
