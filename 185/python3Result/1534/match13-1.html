class Solution:
    def minNumberOfFrogs(self, s: str) -> int:
        states = [0,0,0,0,0]
        avail_frogs = 0
        for c in s:
            if c == 'c':
                if avail_frogs:
                    avail_frogs -= 1
                else:
                    states[0] -= 1
                states
[1] += 1
            elif c == 'r':
                if states[1] == 0:
                    return -1
                states[1] -= 1
                states[2] += 1
            elif c == 'o':
                if states[2] == 0:
                    return -1
                states[2] -= 1
                states[3] += 1
            elif c == 'a':
                if states[3] == 0:
                    return -1
                states[3] -= 1
                states[4] += 1
            elif c == 'k':
                if states[4] == 0:
                    return -1
                states[4] -= 1
                avail_frogs += 1
        
        if sum(states[1:]) > 0:
            return -1
        return -states[0]
