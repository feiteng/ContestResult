class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        res = 0
        d = collections.defaultdict(int)
        cnt = 0
        
        for c in croakOfFrogs:
            if c == 'c':
                d['r'] += 1
                cnt 
+= 1
            elif c == 'r':
                if d['r'] > 0:
                    d['r'] -= 1
                    d['o'] += 1
                else:
                    return -1
            elif c == 'o':
                if d['o'] > 0:
                    d['o'] -= 1
                    d['a'] += 1
                else:
                    return -1
            elif c == 'a':
                if d['a'] > 0:
                    d['a'] -= 1
                    d['k'] += 1
                else:
                    return -1
            elif c == 'k':
                if d['k'] > 0:
                    d['k'] -= 1
                    cnt -= 1
                else:
                    return -1
            else:
                return -1
            res = max(res, cnt)
        if cnt != 0:
            return -1
        return res
