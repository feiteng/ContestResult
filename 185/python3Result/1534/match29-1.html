class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        st = {
            c : i
            for i, c in enumerate('croak')
        }
        curr = [0 for _ in range(6)]
        ans = 0
        for c in croakOfFrogs:
            if st[c] == 0:
                curr[st[c]] += 1
            elif curr[st[c] - 1] == 0:
                return -1
            else:
                curr[st[c]] += 1
                curr
[st[c] - 1] -= 1
            ans = max(ans, sum(curr[:4]))
        
        if sum(curr[:4]) != 0:
            return -1
        else:
            return ans
