class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        if len(croakOfFrogs) % 5 != 0:
            return -1
        res = 0
        dp = collections.defaultdict(int)
        dp['c'],dp['r'],dp['o'],dp['a'],dp['k'] = 0, 0, 0, 0, 0
        cur = 0
        for s in croakOfFrogs:
            if s == 'c':
                cur += 1
                res = max(res, cur)
                dp['c'] += 1
                
            elif s == 'r':
                dp['c'] -= 1
                if dp['c'] < 0:
                    return -1
                dp['r'] += 1
            elif s == 'o':
                dp['r'] -= 1
                if dp['r'] < 0:
                    return -1
                dp['o'] += 1
            elif s == 'a':
                dp['o'] -= 1
                if dp['o'] < 0:
                    return -1
                dp['a'] += 1
                
            elif s == 'k':
                dp['a'] -= 1
                if dp['a'] < 0:
                    return -1
                dp['k'] += 1
                cur -= 1
        return res
