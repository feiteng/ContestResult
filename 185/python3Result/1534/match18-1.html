class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        dic = {"c":0,"r":0,"o":0,"a":0,"k":0}
        for i in croakOfFrogs:
            if i not in dic:
                return -1
            elif i == "c":
                if dic["k"] == 0:
                    dic["c"] += 1
                else:
                    dic["k"] -= 1
                    dic["c"] += 1
            elif i == "r":
                
if dic["c"] == 0:
                    return -1
                else:
                    dic["c"] -= 1
                    dic["r"] += 1
            elif i == "o":
                if dic["r"] == 0:
                    return -1
                else:
                    dic["r"] -= 1
                    dic["o"] += 1
            elif i == "a":
                if dic["o"] == 0:
                    return -1
                else:
                    dic["o"] -= 1
                    dic["a"] += 1
            else:
                if dic["a"] == 0:
                    return -1
                else:
                    dic["a"] -= 1
                    dic["k"] += 1
        for i in ["c","r","o","a"]:
            if dic[i] > 0:
                return -1
        return dic["k"]
