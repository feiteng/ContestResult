class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ans=0
        para=dict.fromkeys('croa',0)
        for ch in croakOfFrogs:
            if ch=='c':
                para['c']+=1
                ans=max(ans,sum(para.values()))
            elif ch=='r':
                if para['c']<1:
                    return -1
                para['c']-=1
                para['r']+=1
            elif ch=='o':
                if para['r']<1:
                    return -1
                para['r']-=1
                para['o']+=1
            elif ch=='a':
                if para['o']<1:
                    return -1
                para['o']-=1
                para['a']+=1
            elif ch=='k':
                if para['a']<1:
                    return -1
                para['a']-=1
        if not any(para.values()):
            return ans
        else:
            return -1
