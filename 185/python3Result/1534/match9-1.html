class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        croak = 'croak'
        idx = {}
        for i, c in enumerate(croak):
            idx[c] = i
        #idx = {'c':0, 'r':1, 'o':2, 'a':3, 'k':4}
        cnt = [0] * 5
        
        frogs = 1
        for c in croakOfFrogs:
            if c not in idx:
                return -1
            i = idx[c]
            
if i == 0:
                cnt[i] += 1
            else:
                if cnt[i-1] <= 0:
                    return -1
                cnt[i-1] -= 1
                cnt[i] += 1
                if i == 4:
                    cnt[i] -=
 1
            frogs = max(frogs, sum(cnt))
        if sum(cnt) != 0:
            return -1
        return frogs
