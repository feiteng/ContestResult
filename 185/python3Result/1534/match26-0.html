class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        
        croak = ['c', 'r', 'o', 'a', 'k']
        n = [0] * 5
        frogs = 0
        
        for c in croakOfFrogs:
            
            if c == 'k':
                if n[3] == 0:
                    return -1
                else:
                    n[3] -= 1
            
            elif c == 'a':
                if n[2] == 0:
                    return -1
                else:
                    n[2] -= 1
                    n[3] += 1
            
            elif c == 'o':
                if n[1] == 0:
                    return -1
                else:
                    n[1] -= 1
                    n[2] += 1
            
            elif c =='r':
                if n[0] == 0:
                    return -1
                else:
                    n[0] -= 1
                    n[1] += 1
            
            elif c == 'c':
                n[0] += 1
                frogs = max(frogs, sum(n))
                
            else:
                return -1
        
        if sum(n)>0:
            return -1
        
        return frogs
