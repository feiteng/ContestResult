class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        frog_map = {}
        frog_map['c'] = 0
        frog_map['r'] = 0
        frog_map['o'] = 0
        frog_map['a'] = 0
        frog_map['k'] = 0
       
        res = 0
        for c in croakOfFrogs:
            if c == "c":
                frog_map['c'] += 1
                res = max(res, frog_map['c'])
            elif c == "k":
                frog_map['c'] -= 1
                frog_map['r'] -= 1
                frog_map['o'] -= 1
                frog_map['a'] -= 1
            else:
                frog_map[c] += 1
       
            if frog_map['c'] < 0 or frog_map['r'] < 0 or frog_map['o'] < 0 or frog_map['a'] < 0 or frog_map['k'] < 0:
                return -1
           
            if frog_map['r'] > frog_map['c'] or frog_map['o'] > frog_map['c'] or frog_map['a'] > frog_map['c'] or frog_map['k'] > frog_map['c']:
                return -1

       
        if frog_map['c'] > 0 or frog_map['r'] > 0 or frog_map['o'] > 0 or frog_map['a'] > 0 or frog_map['k'] > 0:
            return -1
        else:
            return res
