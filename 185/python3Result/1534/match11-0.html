class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        mapping = {"c": 0, "r": 1, "o": 2, "a": 3, "k": 4}
        frogstates = [0, 0, 0, 0, 0]  # c, r, o, a, done
        
        for letter in croakOfFrogs:
            val = mapping[letter]
            
            if frogstates[val-1] > 0:
                frogstates[val-1] -= 1
                frogstates[val] += 1
            elif val == 0:
                frogstates[0] += 1
            else:
                return -1
        
        for i in range(4):
            if frogstates[i] != 0:
                return -1
        return frogstates[4]
