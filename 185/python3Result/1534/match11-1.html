class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ret = [0,0,0,0,0]
        l = 6
        d = {}
        std = "croak"
        for i in range(5):
            d[std[i]]=i
        for s in croakOfFrogs:
            num = d[s]
            lst = (num-1)%5
            
if ret[lst]>0:
                ret[lst]-=1
                ret[num]+=1
            elif num==0:
                ret[num]+=1
            else:
                return -1
        for i in range(4):
            if ret[i]>0:
                return -1
        return ret[-1]
