class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        if not croakOfFrogs:
            return 0
        if len(croakOfFrogs) < 5:
            return -1
        res = 0
        idle = 0
        count_map = {'c': 0, 'r':0, 'o':0, 'a':0, 'k':0}
        prev_map = {'r': 'c', 'o':'r', 'a':'o', 'k':'a'}
        for c in croakOfFrogs:
            if c not in count_map:
                return -1
            
if c == 'c':
                if idle == 0:
                    res += 1
                else:
                    idle -= 1
                count_map[c] += 1
            else:
                prev_c = prev_map[c]
                if count_map[prev_c] == 0:
                    return -1
                count_map[prev_c] -= 1
                count_map[c] += 1
                if c == 'k':
                    idle +=
 1
        for k, v in count_map.items():
            if k != 'k' and v > 0:
                return -1
        return res
