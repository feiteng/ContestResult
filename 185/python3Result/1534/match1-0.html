class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ans = 0
        cnt = 0
        memo = collections.defaultdict(int)
        for ch in croakOfFrogs:
            if ch == 'c':
                cnt += 1
                memo['c'] += 1
            elif ch == 'r':
                if memo['c'] == 0:
                    return -1
                else:
                    memo['c'] -= 1
                    memo['r'] += 1
            elif ch == 'o':
                if memo['r'] == 0:
                    return -1
                else:
                    memo['r'] -= 1
                    memo['o'] += 1
            elif ch == 'a':
                if memo['o'] == 0:
                    return -1
                else:
                    memo['o'] -= 1
                    memo['a'] += 1
            elif ch == 'k':
                if memo['a'] == 0:
                    return -1
                else:
                    memo['a'] -= 1
                    cnt -= 1
            else:
                return -1
            ans = max(ans, cnt)
        
        for ch in 'croa':
            if memo[ch] > 0:
                return -1
        return ans
