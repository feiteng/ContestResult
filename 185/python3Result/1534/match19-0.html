class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        cnts = collections.Counter()
        ans = 0
        curr = 0
        for c in croakOfFrogs:
            # print(c)
            # print(cnts)
            if c == 'c':
                cnts['c'] += 1
                curr += 1
            elif c == 'r':
                if cnts['c'] == 0: return -1
                cnts['c'] -= 1
                cnts['r'] += 1
            elif c == 'o':
                if cnts['r'] == 0: return -1
                cnts['r'] -= 1
                cnts['o'] += 1
            elif c == 'a':
                if cnts['o'] == 0: return -1
                cnts['o'] -= 1
                cnts['a'] += 1
            elif c == 'k':
                if cnts['a'] == 0: return -1
                cnts['a'] -= 1
                curr -= 1
            ans = max(curr, ans)
                
        print(cnts)
        if cnts['c'] == cnts['r'] == cnts['o'] == cnts['a'] == 0:
            return ans
        else:
            return -1
