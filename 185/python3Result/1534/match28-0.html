class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        frogs = 0
        stage = {}
        for i in range(5):
            stage[i] = 0
            
        for c in croakOfFrogs:
            if c == 'c':
                if stage[0] > 0:
                    stage[0] -= 1
                    stage[1] += 1
                else:
                    frogs += 1
                    stage[1] += 1
            elif c == 'r':
                if stage[1] > 0:
                    stage[1] -= 1
                    stage[2] += 1
                else:
                    return -1
            elif c == 'o':
                if stage[2] > 0:
                    stage[2] -= 1
                    stage[3] += 1
                else:
                    return -1
            elif c == 'a':
                if stage[3] > 0:
                    stage[3] -= 1
                    stage[4] += 1
                else:
                    return -1
            else:
                if stage[4] > 0:
                    stage[4] -= 1
                    stage[0] += 1
                else:
                    return -1
        
        for i in range(1, 5):
            if stage[i] != 0:
                return -1
            
        return frogs
