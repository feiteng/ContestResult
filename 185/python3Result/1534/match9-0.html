class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        counts = [0] * 4
        ans = 0
        for c in croakOfFrogs:
            if c == 'c':
                counts[0] += 1
            elif c == 'r':
                if counts[0] == 0:
                    return -1
                counts[0] -= 1
                counts[1] += 1
            elif c == 'o':
                if counts[1] == 0:
                    return -1
                counts[1] -= 1
                counts[2] += 1
            elif c == 'a':
                if counts[2] == 0:
                    return -1
                counts[2] -= 1
                counts[3] += 1
            elif c == 'k':
                if counts[3] == 0:
                    return -1
                counts[3] -= 1
            else:
                return -1
            ans = max(ans, sum(counts))
        if sum(counts) != 0:
            return -1
        return ans
