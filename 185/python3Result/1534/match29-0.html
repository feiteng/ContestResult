class Solution:
    def minNumberOfFrogs(self, s: str) -> int:
        # depth, max_depth = 0, 0
        # stack = []
        # seq = {x:y for x, y in zip("croa", "roak")}
        # for c in s:
        #     if stack and seq[stack[-1]] == c:
        #         stack.append(c)
        #         if c == 'k':
        #             for _ in range(5):
        #                 stack.pop()
        #             depth -= 1
        #     else:
        #         if c != 'c':
        #             return -1
        #         stack.append(c)
        #         depth += 1
        #     max_depth = max(max_depth, depth)
        # return max_depth if not stack else -1
        seq = {x:y for x, y in zip("roak", "croa")}
        D = {c:0 for c in "croak"}
        res = 0
        for c in s:
            if c not in "croak":
                return -1
            D[c] += 1
            if c in seq:
                if D[seq[c]] == 0:
                    return -1
                D[seq[c]] -= 1
            res = max(res, sum(D[c] for c in "croa"))
        if sum(D[c] for c in "croa") == 0:
            return res
        else:
            return -1
