class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        progress = {}
        for c in 'croak':
            progress[c] = 0
            
        PREV = {
            'r': 'c',
            'o': 'r',
            'a': 'o',
            'k': 'a',
        }
        
        count = 0
        inactive = 0
        for c in croakOfFrogs:
            if c == 'c':
                if inactive > 0:
                    inactive -= 1
                else:
                    count += 1
                progress[c] += 1
            else:
                prev = PREV[c]
                if progress[prev] == 0:
                    return -1
                progress[prev] -= 1
                progress[c] += 1
            if c == 'k':
                inactive += 1
        for c, v in progress.items():
            if c != 'k' and v > 0:
                return -1
        return count
