class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        croak = "croak"
        froglist = [0,0,0,0] # how many frogs are on each letter e.g. froglist[0] is how many have just said "c"
        maxfrogs = 0
        for char in croakOfFrogs:
            #print(char,froglist)
            if char == "c":
                froglist[0]+=1
            elif char == "r":
                if froglist[0]==0:
                    return -1
                else:
                    froglist[0]-=1
                    froglist[1]+=1
            elif char == "o":
                if froglist[1]==0:
                    return -1
                else:
                    froglist[1]-=1
                    froglist[2]+=1
            elif char == "a":
                if froglist[2]==0:
                    return -1
                else:
                    froglist[2]-=1
                    froglist[3]+=1
            elif char == "k":
                if froglist[3]==0:
                    return -1
                else:
                    froglist[3]-=1
            
            sumfrogs = sum(froglist)
            if maxfrogs<sumfrogs:
                maxfrogs = sumfrogs
        
        if sum(froglist) == 0:
            return maxfrogs
        else:
            return -1
