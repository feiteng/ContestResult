class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        data = collections.defaultdict(lambda :collections.defaultdict(int))
        foodall = set()
        for _, table, food in orders:
            table = int(table)
            data[table][food] += 1
            foodall.add(food)
        foodall = sorted(list(foodall))
        res = [['Table'] + foodall[:]]
        for key in sorted(data.keys()):
            res.append([str(key)] + [str(data[key][food]) for food in foodall])
        return res
