class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foods = set()
        tables = {}
        for name, table, food in orders:
            table = int(table)
            if table not in tables:
                tables[table] = collections.Counter()
            tables[table][food] += 1
            foods.add(food)
        foods = list(sorted(foods))
        ans = [["Table"] + foods]
        for table in sorted(tables):
            row = [table] + [0] * len(foods)
            for food, cnt in tables[table].items():
                row[foods.index(food) + 1] += cnt
            ans.append(list(map(str, row)))
        return ans
