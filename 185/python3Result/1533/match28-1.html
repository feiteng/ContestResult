class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        cols = set()
        tables = set()
        count = collections.defaultdict(lambda:collections.defaultdict(int))
        for name, table, food in orders:
            cols.add(food)
            tables.add(table)
            count[table][food] += 1
        
        foods = sorted(list(cols))
        stables = sorted(int(t) for t in tables)
        ans = []
        ans.append(["Table"] + foods)
        
        for table in stables:
            ans.append([str(table)] + [str(count[str(table)][food]) for food in foods])
        
        return ans
