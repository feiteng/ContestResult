class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        deals = {}
        table = {}
        for _, tid, deal in orders:
            if not deal in deals:
                deals[deal] = len(deals)
            tt = int(tid)
            table.setdefault(tt, {})[deals[deal]] = table.get(tt, {}).get(deals[deal], 0) + 1
        header = ['Table']
        pdeals = sorted(deals.items())
        for k, v in pdeals:
            header.append(k)
        ans = [header]
        for i in sorted(table):
            row = [str(i)]
            for k, v in pdeals:
                row.append(str(table[i].get(v, 0)))
            ans.append(row)
        return ans
