class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        head = ["Table"]
        dishes = set()
        d = {}
        for item in orders:
            dishes.add(item[2])
            d.setdefault(item[1],{})
            d[item[1]].setdefault(item[2],0)
            d[item[1]][item[2]]+=1
        dishes = sorted(list(dishes))
        # print(dishes,d)
        head = head+dishes
        res = [head]
        keys = sorted(d.keys(),key=lambda x: int(x))
        # print(keys)
        for key in keys:
            obj = []
            obj.append
(key)
           
            for dish in dishes:
                # print(dish in d[key])
                if dish in d[key]:
                    obj.append(str(d[key][dish]))
                else:
                    obj.append('0')
            res.append(obj)
        
        return res
