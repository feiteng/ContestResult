from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        _orders = orders
        orders = defaultdict(int)
        for _, table, item in _orders:
            table = int(table)
            orders
[(table, item)] += 1
        tables = sorted(set(table for table, _ in orders))
        items = sorted(set(item for _, item in orders))
        ans = [["Table"] + items]
        for table in tables:
            row = [str(table)]
            for item in items:
                row.append(str(orders[(table, item)]))
            ans.append(row)
        return ans
