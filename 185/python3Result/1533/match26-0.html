class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        DishSet=set()
        tableMap=defaultdict(list)
        tableSet=set()
        
        for name,table,dish in orders :
            DishSet.add(dish)
            tableMap[table].append(dish)
            tableSet.add(int(table))
            
        
        sortedTableList=sorted(list(tableSet))
        #print(sortedTableList)
        #print(list(DishSet))
        DishList=sorted(list(DishSet))
        ans=[["Table"]+DishList]
        
        #print(ans)
        #print(tableMap)
        #cnt=Counter(tableMap['3'])
        #print(cnt)
        ans1=[]
        for table in sortedTableList:
            newL=[str(table)]
            cnt=Counter(tableMap[str(table)])
            #print(cnt)
            for dish in DishList:
                newL.append(str(cnt[dish]))
            ans1.append(newL)
            
        #print(ans)
        #print(ans+sorted(ans1))
            
        return ans+ans1
