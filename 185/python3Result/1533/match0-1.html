class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        display = collections.defaultdict(lambda: collections.defaultdict(int))
        rows 
= set()
        for cus, tab, food in orders:
            tab = int(tab)
            display[tab][food] += 1
            rows.add(food)
        foods = sorted(list(rows))
        ans 
= [['Table'] + foods]
        for tab in sorted(display.keys()):
            row = [str(tab)]
            for food in foods:
                row.append(str(display[tab][food]))
            ans.append(row)
        return ans
