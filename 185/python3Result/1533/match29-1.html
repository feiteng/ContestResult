class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        food_set = set()
        tables = collections.defaultdict(lambda: collections.Counter())
        for costumer, table, food in orders:
            tables[int(table)][food] += 1
            food_set.add(food)
        columns = sorted(food_set)
        ans = [['Table'] + columns]
        for table in sorted(tables.keys()):
            ans += [[str(table)] + [str(tables[table][col]) for col in columns]]
        return ans
