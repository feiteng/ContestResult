class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        food = set()
        tables = set()
        dic = collections.defaultdict(lambda : collections.defaultdict(int))
        for o in orders:
            dic[o[1]][o[2]] += 1
            food.add(o[2])
            tables.add(o[1])
        ret = []
        food = sorted(food)
        tables = sorted(tables, key = lambda x: int(x))
        ret.append(["Table"] + food)
        for t in tables:
            ret.append([t] + [str(dic[t][f]) for f in food])
        return ret
