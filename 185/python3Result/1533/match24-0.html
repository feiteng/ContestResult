class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        items = set()
        table = collections.defaultdict(lambda: collections.defaultdict(int))
        
        for name, number, item in orders:
            items.add(item)
            table[int(number)][item] += 1
        
        it = sorted(items)
        ans = []
        ans.append(['Table'] + it)
        
        for no in sorted(table.keys()):
            sub = [str(no)]
            for thing in it:
                if thing in table[no]:
                    sub.append(str(table[no][thing]))
                else:
                    sub.append("0")
            ans.append(sub)
        return ans
