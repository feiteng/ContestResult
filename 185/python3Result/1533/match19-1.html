class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        food = set()
        table = collections.defaultdict(collections.Counter)
        for _, i, dish in orders:
            food.add(dish)
            table[int(i)][dish] += 1
        food = sorted(food)
        food_d = {f: i for i, f in enumerate(food)}
        num = sorted(table.keys())
        first = ['Table']
        first.extend(food)
        ans = [first]
        for n in num:
            row = [str(n)]
            l = [0] * len(food)
            for k, v in table[n].items():
                k = food_d[k]
                l[k] = v
            row.extend(str(m) for m in l)
            ans.append(row)
        return ans
