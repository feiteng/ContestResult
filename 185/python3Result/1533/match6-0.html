class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        from collections import defaultdict
        items = set()
        order_dict = defaultdict(lambda: defaultdict(int))
        for order in orders:
            items.add(order[2])
            order_dict[int(order[1])][order[2]] += 1
        
        items = sorted(list(items))
        
        ret = []
        header = ["Table"] + items
        ret.append(header)
        for table in sorted(order_dict.keys()):
            row = [str(table)]
            for item in items:
                row.append(str(order_dict[table][item]))
            ret.append(row)
        return ret
