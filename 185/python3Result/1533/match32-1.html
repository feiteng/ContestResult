from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foods = sorted(set(f for _, _, f in orders))
        d = defaultdict(dict)
        for c, t, f in orders:
            t = int(t)
            d[t][f] = d[t].get(f, 0) + 1
        ans = [['Table'] + foods]
        # print(d)
        for t in sorted(d.keys
()):
            tmp = [str(t)]
            for f in foods:
                tmp.append(str(d[t].get(f, 0)))
            ans.append(tmp)
        return ans
