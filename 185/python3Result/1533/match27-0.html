from collections import defaultdict as dd

class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foods = set()
        groups = dd(int)
        tabs = set()
        for cust, tab, food in orders:
            foods.add(food)
            groups[(tab, food)] += 1
            tabs.add(int(tab))
        
        sortedfoods = sorted(list(foods))
        output = [["Table"] + sortedfoods]
        for table in sorted(tabs):
            new_row = [str(table)]
            for food in sortedfoods:
                new_row.append(str(groups[(str(table), food)]))
            output.append(new_row)
        return output
