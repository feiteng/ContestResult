class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        tab = collections.defaultdict(collections.Counter)
        fds = set()
        for _, t, f in orders:
            t = int(t)
            tab[t][f] += 1
            fds.add(f)
        
        fds = list(sorted(list(fds)))
        tit = ['Table'] + fds
        rws = [tit]
        for i in sorted(tab.keys()):
            r = [str(i)]
            for f in fds:
                r.append(str(tab[i][f]))
            rws.append(r)
        return rws
