class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        d = collections.defaultdict(dict)
        foods = set()
        for order in orders:
            table = int(order[1])
            food = order[2]
            foods.add(food)
            
            if food not in d[table]:
                d[table][food] = 1
            else:
                d[table][food] += 1
        sortfood = sorted(list(foods))
        res = [['Table']+sortfood]
        for k in sorted(d.keys()):
            t = []
            t.append(str(k))
            for food in sortfood:
                if food not in d[k]:
                    t.append('0')
                else:
                    t.append(str(d[k][food]))
            res.append(t)
        return res
