from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        
        table = defaultdict(lambda: defaultdict(int))
        
        all_food = set()
        for _, table_num, item in orders:
            table[int(table_num)][item] += 1
            all_food.add(item)
            
        headers = ["Table"] + sorted(all_food)
        sorted_food = sorted(all_food)
        indices = {}
        for i, f in enumerate(headers):
            indices[f] = i
        
        rows = sorted(table.keys())
        
        
        out = [headers]
        for table_num in rows:
            food = [str(table_num)]
            for f in sorted_food:
                food.append(str(table[table_num][f]))
            out.append(food)
        return out
