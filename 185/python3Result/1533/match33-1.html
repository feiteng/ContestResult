class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foods = set([])
        for order in orders:
            foods.add(order[2])
        
        default_ptable_tracker = {}
        for food in foods:
            default_ptable_tracker[food] = 0
        tracker = {}
        for order in orders:
            if int(order[1]) in tracker:
                tracker[int(order[1])][order[2]] += 1
            else:
                tracker[int(order[1])] = dict(default_ptable_tracker)
                tracker[int(order[1])][order[2]] += 1
        
        table_list = list(tracker)
        table_list.sort()
        
        food_list 
= list(foods)
        food_list.sort()
        
        return_list = [["Table"]+food_list]
        for table in table_list:
            added_table_list = [str(table)]
            for food in food_list:
                added_table_list.append(str(tracker[table][food]))
            return_list.append(added_table_list)
        
        return return_list
