class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        table_orders = {}
        foods = set()
        for _, table_num, food in orders:
            table_num = int(table_num)
            if table_num not in table_orders:
                table_orders[table_num] = {}
            table_orders[table_num][food] = table_orders[table_num].get(food, 0) + 1
            foods.add(food)
        
        foods = sorted(list(foods))
        res = [["Table"] + foods]
        for table_num in sorted(table_orders.keys
()):
            row = [str(table_num)]
            for food in foods:
                row.append(str(table_orders[table_num].get(food, 0)))
        
            res.append(row)
                           
        return res
