class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        food 
= set()
        h = defaultdict(lambda: defaultdict(int))
        for item in orders:
            food.add(item[2])
            h[int(item[1])][item[2]] += 1
        food = list(food)
        food.sort()
        tables = list(h.keys())
        tables.sort()
        
        res = []
        res.append(['Table'])
        for f in food:
            res[0].append(f)
            
        for t in tables:
            cur 
= [str(t)]
            for f in food:
                cur.append(str(h[t][f]))
            res.append(cur)
        return res
