from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dt = dict()
        foods = set()
        tables = set()
        for row in orders:
            table = int(row[1])
            tables.add(table)
            food = row[2]
            foods.add(food)
            if table not in dt:
                dt[table] = defaultdict(int)
            dt[table][food] += 1
        tables = sorted(list(tables))
        foods = sorted(list(foods))
        ans = [["Table"]+foods]
        for table in tables:
            row = [str(table)]
            for food in foods:
                row.append(str(dt[table][food]))
            ans.append(row)
        return ans
