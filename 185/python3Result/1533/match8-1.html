class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        d = dict()
        fset = set()
        ans = []
        for cust,t,f in orders:
            tint = int(t)
            if tint not in d:
                d[tint] = collections.Counter()
            d[tint][f] += 1
            fset.add(f)
        food = list(fset)
        food.sort()
        ans
.append(["Table"]+food)
        for tabnum in sorted(d.keys()):
            el = [str(tabnum)]
            for i in food:
                el.append(str(d[tabnum][i]))
            ans.append(el)
        return ans
