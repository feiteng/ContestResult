class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        header = ["Table"]
        foods = set()
        tables = {}
        
        for order in orders:
            [name, table_no, food] = order
            foods.add(food)
            if table_no not in tables:
                tables[table_no] = {}
            
            if food not in tables[table_no]:
                tables[table_no][food] = 1
            else:
                tables[table_no][food] += 1
            
        food_list = sorted(list(foods))
        header += food_list
        
        res = [header]
        
        table_list = sorted(list(tables), key = lambda x: int(x))

        for table_no in table_list:
            line = [str(table_no)]
            for f in food_list:
                if f in tables[table_no]:
                    line.append(str(tables[table_no][f]))
                else:
                    line.append('0')
            res.append(line)
            
        return res
