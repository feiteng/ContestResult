class Solution:
    def reformat(self, s: str) -> str:
        dd=[]
        cc=[]
        for c in s:
            if c.isdigit():dd.append(c)
            else:cc.append(c)
        if abs(len(dd)-len(cc))>1:return ""
        if len(dd)>len(cc):dd,cc=cc,dd
        res=""
        while cc:
            res+=cc.pop()
            if not dd:break
            res+=dd.pop()
        return res
