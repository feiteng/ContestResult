class Solution:
    def reformat(self, s: str) -> str:
        ret = ""
        l = []
        n = []
        for c in s:
            if '0' <= c <= '9':
                n.append(c)
            else:
                l.append(c)
        if abs(len(l) - len(n)) > 1:
            return ""
        elif len
(l) >= len(n):
            for i in range(len(l)):
                ret += l[i]
                if i < len(n): ret += n[i]
        elif len(l) < len(n):
            
for i in range(len(n)):
                ret += n[i]
                if i < len(l): ret += l[i]
        return ret
