class Solution:
    def reformat(self, s: str) -> str:
        cc = []
        nn = []
        for c in s:
            if c.isdigit():
                nn.append(c)
            else:
                cc.append(c)
        if abs(len(cc) - len(nn)) > 1:
            return ''
        if len(cc) < len(nn):
            cc, nn = nn, cc
        res = []
        for i in range(len(cc)):
            res.append(cc[i])
            if i < len(nn):
                res.append(nn[i])
        return ''.join(res)
