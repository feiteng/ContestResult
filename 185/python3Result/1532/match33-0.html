class Solution:
    def reformat(self, s: str) -> str:
        dig, ch = [], []
        for c in s:
            if c.isdigit():
                dig.append(c)
            else:
                ch.append(c)
        if abs(len(dig) - len(ch)) >= 2:
            return ""
        res = ""
        if len(dig) > len(ch):
            for a,b in zip(dig, ch):
                res += a+b
            return res + dig[-1]
        else:
            for a,b in zip(ch, dig):
                res += a+b
            return res + (ch[-1] if len(ch) > len(dig) else "")
