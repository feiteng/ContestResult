class Solution:
    def reformat(self, s: str) -> str:
        digits, chars = [], []
        for c in s:
            if c in list(map(str, range(0, 10))):
                digits.append(c)
            else:
                chars.append(c)
        if abs(len(digits) - len(chars)) >= 2:
            return ""
        if len(digits) > len(chars):
            x, y = digits, chars
        else:
            x, y = chars, digits
        ans = []
        for i, j in zip(x, y):
            ans.append(i)
            ans.append(j)
        if len(x) == len(y):
            return ''.join(ans)
        else:
            ans.append(x[-1])
            return ''.join(ans)
