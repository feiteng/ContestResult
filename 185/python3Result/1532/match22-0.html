class Solution:
    def reformat(self, s: str) -> str:
        nums = []
        lets = []
        for l in s:
            if l.isalpha():
                lets.append(l)
            else:
                nums.append(l)
        if abs(len(nums) - len(lets)) > 1:
            return ""

        output = []
        if len(nums) > len(lets):
            output.append(nums[-1]) 
            for i in range(len(lets)):
                output.append(lets[i])
                output.append(nums[i])
        else:
            if len(nums) < len(lets):
                output.append(lets[-1])
                lets.pop()
            for i in range(len(lets)):
                output.append(nums[i])
                output.append(lets[i])
        return ''.join(output)
