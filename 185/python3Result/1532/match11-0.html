class Solution:
    def reformat(self, s: str) -> str:
        ans=""
        a=""
        b=""
        for i in s:
            if i>='a' and i<='z':
                a=a+i
            else:
                b=b+i
        if abs(len(a)-len(b))>1:
            return ans;
        if len(a)<len(b):
            c=a
            a=b
            b=c
        for i in range(len(b)):
            ans=ans+a[i]
            ans=ans+b[i]
        if len(a)>len(b):
            ans=ans+a[-1]
        return ans
