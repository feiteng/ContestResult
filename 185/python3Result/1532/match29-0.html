class Solution:
    def reformat(self, s: str) -> str:
        left = []
        right = []
        for c in s:
            if c >= '0' and c <= '9':
                left.append(c)
            else:
                right.append(c)
        print(left)
        if abs(len(left) - len(right)) > 1:
            return ""
        ans = ""
        if len(left) > len(right):
            for i in range(len(left)):
                ans += left[i]
                if i < len(right):
                    ans += right[i]
        else:
            for i in range(len(right)):
                ans += right[i]
                if i < len(left):
                    ans += left[i]
        return ans
