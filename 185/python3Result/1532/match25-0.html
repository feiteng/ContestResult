class Solution:
    def reformat(self, s: str) -> str:
        l1 = []
        l2 = []
        for ch in s :
            if ch >= '0' and ch <= '9' :
                l1.append(ch)
            else :
                l2.append(ch)
        diff = len(l1) - len(l2)
        if abs(diff) > 1 :
            return ""
        l = []
        if diff > 0 :
            for i in range(len(l1)) :
                l.append(l1[i])
                if i < len(l2) :
                    l.append(l2[i])
        else :
            for i in range(len(l2)) :
                l.append(l2[i])
                if i < len(l1) :
                    l.append(l1[i])
        return ''.join(l)
