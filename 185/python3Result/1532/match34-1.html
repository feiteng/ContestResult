class Solution:
    def getRes(self, letters, digits):
        if len(letters) > len(digits)+1:
            return ""
        else:
            res = []
            
for i in range(len(digits)):
                res.append(letters[i])
                res.append(digits[i])
            res.append(letters[-1])
        return res
        
    
def reformat(self, s: str) -> str:
        digits = []
        letters = []
        for c in s:
            if ord('9') >= ord(c) >= ord('0'):
                digits.append(c)
            else:
                letters.append(c)
        if len(letters) > len(digits):
            res = self.getRes(letters, digits)
        elif len(digits) > len(letters):
            res = self.getRes(digits, letters)
        else:
            res = []
            for i in range(len(digits)):
                res.append(digits[i])
                res.append(letters[i])
        return "".join(res)
