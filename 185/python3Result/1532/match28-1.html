class Solution:
    def reformat(self, s: str) -> str:
        
        
        num_count = 0
        a_count = 0
        nums = []
        ass = []
        out = []
        for c in s:
            if c.isdigit():
                nums.append(c)
                num_count += 1
            else:
                ass.append(c)
                a_count += 1
        
        if abs(num_count - a_count) > 1:
            return ""
        
        if num_count > a_count:
            
while ass or nums:
                if nums:
                    out.append(nums.pop())
                if ass:
                    out.append(ass.pop())
        else:
            
while ass or nums:
                if ass:
                    out.append(ass.pop())
                if nums:
                    out.append(nums.pop())
            
            
        return ''.join(out)
