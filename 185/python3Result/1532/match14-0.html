class Solution:
    def reformat(self, s: str) -> str:
        letter = []
        number = []
        for ch in s:
            if ch.isdigit():
                number.append(ch)
            else:
                letter.append(ch)
        
        if abs(len(letter) - len(number)) >= 2:
            return ""
        
        ans = ""
        longer, shorter = (letter, number) if len(letter) > len(number) else (number, letter)
        for i in range(len(s)):
            if i % 2 == 0:
                ans += longer.pop()
            else:
                ans += shorter.pop()
        return ans
