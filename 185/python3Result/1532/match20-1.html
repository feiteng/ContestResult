class Solution:
    def reformat(self, s: str) -> str:
        a, b 
= [], []
        for c in s:
            if c.isnumeric():
                a.append(c)
            else:
                b.append(c)
        if len(a) > len(b):
            a, b = b, a
        
        if len(b) > len(a) + 1:
            return ''
        
        else:
            r = ''
            
for i in range(len(s)):
                if i % 2 == 0:
                    r += b.pop()
                else:
                    r += a.pop()
            return r
