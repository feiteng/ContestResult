class Solution:
    def reformat(self, s: str) -> str:
        if len(s)<2:
            return s
        
        alpha = []
        num = []
        
        for char in s:
            if char.isdigit():
                num
.append(char)
            else:
                alpha.append(char)
                
        if abs(len(alpha) - len(num)) >= 2:
            return ""
        
        res = []
        
        if len(alpha) > len(num):
            for i in range(len(num)):
                res.append(alpha[i])
                res.append(num[i])
            res.append(alpha[-1])
        else:
            
for i in range(len(alpha)):
                res.append(num[i])
                res.append(alpha[i])
            if len(alpha) < len(num):
                res.append(num[-1])
                
        return "".join(res)
