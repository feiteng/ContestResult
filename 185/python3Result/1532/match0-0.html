class Solution:
    def reformat(self, s: str) -> str:
        list1 = []
        list2 = []
        for c in s:
            if c.isdigit():
                list1.append(c)
            else:
                list2.append(c)
       
        if abs(len(list2) - len(list1)) > 1:
            return ""
       
        if len(list1) < len(list2):
            list1, list2 = list2, list1
       
        res = []
        for i in range(len(list1)):
            res.append(list1[i])
            if i < len(list2):
                res.append(list2[i])
       
        return "".join(res)
