class Solution:
    def reformat(self, s: str) -> str:
        letter = []
        number = []
        for ch in s:
            if ord('0') <= ord(ch) <= ord('9'):
                number
.append(ch)
            else:
                letter.append(ch)
        if abs(len(letter) - len(number)) > 1:
            return ""
        ret = []
        if len(letter) < len(number):
            letter, number = number, letter
        for idx in range(len(number)):
            ret.append(letter[idx])
            ret.append(number[idx])
        ret.extend(list(letter[len(number):]))
        return "".join(ret)
