class Solution:
    def reformat(self, s: str) -> str:
        d, a = [], []
        for c in s:
            if c in "0123456789":
                d.append(c)
            else:
                a.append(c)
        if abs(len(d) - len(a)) > 1:
            return ""
        r = []
        if len(d) > len(a):
            d, a = a, d
        for i in range(len(d)):
            r.append(a[i])
            r.append(d[i])
        if len(a) > len(d):
            r.append(a[len(a) - 1])
        return ''.join(r)
