class Solution:
    def reformat(self, s: str) -> str:
        digit = []
        chars = []
        for i in s:
            if i.isdigit():
                digit.append(i)
            else:
                chars.append(i)
        if abs(len(digit)-len(chars)) > 1:
            return ""
        if len(digit) >= len(chars):
            res = []
            while digit or chars:
                if digit:
                    res.append(digit.pop())
                if chars:
                    res.append(chars.pop())
        else:
            res = []
            while digit or chars:
                if chars:
                    res.append(chars.pop())
                if digit:
                    res.append(digit.pop())
        return ''.join(res)
