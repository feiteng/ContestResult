class Solution:
    def reformat(self, s: str) -> str:
        num = []
        c = []
        for k in s:
            if k.isdigit():
                num.append(k)
            else:
                c.append(k)
        if abs(len(num) - len(c)) > 1:
            return ""
        a, b = num, c
        if len(b) > len(a):
            a, b = b, a
        ans = ""
        for k in range(len(b)):
            ans += a[k]
            ans += b[k]
        if len(a)!=len(b):
            ans += a[-1]
        return ans
