class Solution:
    def reformat(self, s: str) -> str:
        num = ""
        letter = ""
        for i in s:
            if i.isdigit():
                num 
+= i
            else:
                letter += i
        if abs(len(num) - len(letter))> 1:
            return ""
        if len(num) < len(letter):
            num,letter = letter,num
        re = ""
        
for i in range(len(letter)):
            re += num[i]
            re += letter[i]
        if len(num) > len(letter):
            re += num[-1]
        return re
