class Solution:
    def reformat(self, s: str) -> str:
        # alpha = [chr(ord('a')+i) for i in range(26)]
        num = [str(i) for i in range(10)]
        a, b = [], []
        for si in s:
            if si in num:
                a
.append(si)
            else:
                b.append(si)
        if abs(len(a) - len(b)) > 1:
            return ""
        res = []
        if len(a) < len(b):
            a, b = b, a
        
for i in range(len(b)):
            res.append(a[i])
            res.append(b[i])
        if len(a) > len(b):
            res.append(a[-1])
        return ''.join(res)
