class Solution:
    def reformat(self, s: str) -> str:
        n,d= [],[]
        for c in s:
            if '0'<=c<='9':
                d.append(c)
            else:
                n.append(c)
        if abs(len(n)-len(d))>1:
            return ''
        res=[]    
        if len(n)>=len(d):
            for i in range(len(d)):
                res.append(n[i])
                res.append(d[i])
            if len(n)>len(d):
                res.append(n[-1]) 
        else:        
            for i in range(len(n)):
                res.append(d[i])
                res.append(n[i])
            res.append(d[-1]) 
        return  ''.join(res)
