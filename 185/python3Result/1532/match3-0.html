class Solution:
    def reformat(self, s: str) -> str:
        letters = []
        digits = []
        for c in s:
            if c.isdigit():
                digits.append(c)
            else:
                letters.append(c)
                
        if abs(len(letters) - len(digits)) > 1:
            return ""
        res = ""
        if len(letters) >= len(digits):
            for i in range(len(digits)):
                res += letters[i]
                res += digits[i]
            if len(letters) > len(digits):
                res += letters[-1]
        else:
            for i in range(len(letters)):
                res += digits[i]
                res += letters[i]
            res += digits[-1]
        return res
