class Solution:
    def reformat(self, s: str) -> str:
        digits= []
        alphas = []
        for c in s:
            if c.isdigit():
                digits.append(c)
            else:
                alphas.append(c)
        if abs(len(digits) - len(alphas)) > 1:
            return ''
        ans = []
        chars1, chars2 = (digits, alphas) if len(digits) > len(alphas) else (alphas, digits)
        while chars1 or chars2:
            if chars1:
                ans.append(chars1.pop())
            
if chars2:
                ans.append(chars2.pop())
        return ''.join(ans)
