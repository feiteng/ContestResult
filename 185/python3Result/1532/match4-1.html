class Solution:
    def reformat(self, s: str) -> str:
        letters = []
        nums = []
        
        for ch in s:
            if ch.isalpha():
                letters.append(ch)
            else:
                nums.append(ch)
        
        if abs(len(letters) - len(nums)) > 1:
            return ''
        
        num = 1
        if len(letters) > len(nums):
            num = 0
        
        ans = ''
        while letters or nums:
            if num:
                ans += nums.pop()
            else:
                ans += letters.pop()
            num = 1 - num
        
        return ans
