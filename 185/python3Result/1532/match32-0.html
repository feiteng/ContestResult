class Solution:
    def reformat(self, s: str) -> str:
        letters = []
        digits = []
        for c in s:
            if c.isdigit():
                digits.append(c)
            else:
                letters.append(c)
        if len(letters) < len(digits):
            letters, digits = digits, letters
        if len(letters) - len(digits) > 1:
            return ""
        ret = []
        while len(digits) > 0:
            ret.append(letters.pop())
            ret.append(digits.pop())
        if len(letters) > 0:
            ret.append(letters.pop())
        return "".join(ret)
