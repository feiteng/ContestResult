class Solution:
    def reformat(self, s: str) -> str:
        nums = []
        alphabeta = []
        for ch in s:
            if '0' <= ch <= '9':
                nums.append(ch)
            else:
                alphabeta.append(ch)
                
        if abs(len(nums) - len(alphabeta)) > 1:
            return ""
        
        ans = []
        if len(nums) >= len(alphabeta):
            for i in range(len(alphabeta)):
                ans.append(nums[i])
                ans.append(alphabeta[i])
            if len(nums) > len(alphabeta):
                ans.append(nums[-1])
            return "".join(ans)
        
        else:
            for i in range(len(nums)):
                ans.append(alphabeta[i])
                ans.append(nums[i])
            if len(alphabeta) > len(nums):
                ans.append(alphabeta[-1])
            return "".join(ans)
