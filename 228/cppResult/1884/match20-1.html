class Solution {
public:
    int minOperations(string s) {
        int cand = 0;
        int n = s.size();
        for(int i = 0; i < n; i++){
            if (i%2 == 1 && s[i] == '0') cand++;
            if (i%2==0 && s[i] == '1') cand++;
        }
        int cand1 = 0;
        for(int i = 0; i < n; i++){
            if (i%2 == 1 && s[i] == '1') cand1++;
            if (i%2==0 && s[i] == '0') cand1++;
        }
        
        return min(cand,cand1);
    }
};
