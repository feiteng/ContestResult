class Solution {
public:
    int minOperations(string s) {
        int n = s.size();
        int now=0;
        int num=0;
        int Min = n+1;
        for (int i=0;i<n;i++){
            if (s[i]-'0'!=now){
                num++;
            }
            now^=1;
        }
        Min=min(Min,num);
        num=0;
        now=1;
        for (int i=0;i<n;i++){
            if (s[i]-'0'!=now){
                num++;
            }
            now^=1;
        }
        Min=min(Min,num);
        return Min;
        
    }
};
