class Solution {
public:
    int minOperations(string s) {
        int ones = 0;
        int zeros = 0;
        for (int i = 0; i < s.size(); i++){
            if (i&1){
                if (s[i]=='0') ones++;
                else zeros++;
            }
            else{
                if (s[i]=='1') ones++;
                else zeros++;
            }
        }
        return min(ones, zeros);
    }
};
