class Solution {
public:
    int minOperations(string s) {
        
        int cnt = 0;
        int n = s.length();
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0 && s[i] != '0') cnt++;
            else if (i % 2 == 1 && s[i] != '1') cnt++;
        }
        
        int cnt2 = 0;
        for (int i = 0; i < n; i++) {
            if (i % 2 == 1 && s[i] != '0') cnt2++;
            else if (i % 2 == 0 && s[i] != '1') cnt2++;
        }
        
        return min(cnt, cnt2);
        
    }
};
