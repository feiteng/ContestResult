class Solution {
public:
    int minOperations(string s) {
        int ans=0,temp=0;
        for(int i=0;i<s.size();i++){
            if(i%2==0){
                if(s[i]=='0')ans++;
            }
            else{
                if(s[i]=='1')ans++;
            }
        }
        for(int i=0;i<s.size();i++){
            if(i%2!=0){
                if(s[i]=='0')temp++;
            }
            else{
                if(s[i]=='1')temp++;
            }
        }
        return min(ans,temp);
    }
};
