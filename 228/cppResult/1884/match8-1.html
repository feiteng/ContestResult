class Solution {
public:
    int minOperations(string s) {
        int slen=s.length();
        int ret=slen;
        int pre=0;
        int cnt=0;
        for(int i=0;i<slen;i++){
            if(s[i]-'0'==pre){
                cnt++;
            }
            pre^=1;
        }
        ret=min(ret,cnt);
        pre=1;
        cnt=0;
        for(int i=0;i<slen;i++){
            if(s[i]-'0'==pre){
                cnt++;
            }
            pre^=1;
        }
        ret=min(ret,cnt);
        return ret;
    }
};
