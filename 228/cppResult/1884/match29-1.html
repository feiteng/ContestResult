class Solution {
public:
    int minOperations(string s) {
        int val1=0;
        int n = s.length();
            
        // 1010
        for(int i=0;i<n;i++){
            if(i%2==0 && s[i]=='0')
                val1++;
            else if(i%2==1 && s[i]=='1')
                val1++;
        }
        
        int val2=0;
        // 0101
        for(int i=0;i<n;i++){
            if(i%2==0 && s[i]=='1')
                val2++;
            else if(i%2==1 && s[i]=='0')
                val2++;
        }
        
        return min(val1, val2);
    }
};
