class Solution {
public:
    int minOperations(string s) {
        int res1 = 0, res2 = 0;
        for (int i = 0; i < s.length(); ++i) {
            if (i & 1 && s[i] == '0') ++res1;
            if (!(i & 1) && s[i] == '1') ++res1;
        }
        for (int i = 0; i < s.length(); ++i) {
            if (!(i & 1) && s[i] == '0') ++res2;
            if (i & 1 && s[i] == '1') ++res2;
        }
        return min(res1, res2);
    }
};
