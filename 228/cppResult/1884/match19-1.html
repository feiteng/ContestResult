class Solution {
public:
    int minOperations(string s) {
        int n = s.size();
        int f = 0, se = 0;
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0 && s[i] != '0')
                f++;
            if (i % 2 == 1 && s[i] != '1')
                f++;
        }
        for (int i = 0; i < n; i++) {
            if (i % 2 == 1 && s[i] != '0')
                se++;
            if (i % 2 == 0 && s[i] != '1')
                se++;
        }
        
        return min(f, se);
    }
};
