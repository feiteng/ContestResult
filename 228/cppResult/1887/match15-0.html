class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> graph(n + 1, vector<int>(n + 1, 0));
        vector<int> deg(n + 1, 0);
        for (auto&& edge : edges) {
            graph[edge[0]][edge[1]] = 1;
            graph[edge[1]][edge[0]] = 1;
            deg[edge[0]]++;
            deg[edge[1]]++;
        }
        int minDeg = INT32_MAX;
        for (int i = 1; i <= n; ++i) {
            for (int j = 0; j <= n; ++j) {
                if (graph[i][j]) {
                    for (int k = j + 1; k <= n; ++k) {
                        if (graph[i][k] && graph[j][k]) {
                            minDeg = min(minDeg, deg[i] + deg[j] + deg[k] - 6);
                        }
                    }
                }
            }
        }
        return minDeg == INT32_MAX ? -1 : minDeg;
    }
};
