class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        int ans = INT_MAX;
        vector<vector<bool> > ed(n+1,vector<bool>(n+1,false));
        vector<int> deg(n+1,0);
        for(auto i:edges){
            ed[i[0]][i[1]] = ed[i[1]][i[0]] = true;
            deg[i[0]]++; deg[i[1]]++;
        }
        for(int i=1;i<=n;i++)
            for(int j=i+1;j<=n;j++)
                for(int k=j+1;k<=n;k++)
                    if(ed[i][j] and ed[j][k] and ed[k][i])
                        ans = min(ans,deg[i]+deg[j]+deg[k]-6);
        return (ans == INT_MAX ? -1 : ans);
    }
};
