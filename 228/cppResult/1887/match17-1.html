class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<bool>> v(n, vector<bool>(n, false));
        vector<int> c(n, 0);
        for (auto& a: edges) { v[a[0]-1][a[1]-1] = true; v[a[1]-1][a[0]-1] = true; c[a[0]-1]++; c[a[1]-1]++; }
        int result = INT_MAX;
        for (int i1 = 0; i1 < n; i1++) {

            for (int i2 = i1+1; i2 < n; i2++) {
                if (!v[i1][i2]) continue;
                for (int i3 = i2+1; i3 < n; i3++) {
                    if (!v[i1][i3] || !v[i2][i3]) continue;
                    result = min(result, c[i1]+c[i2]+c[i3]-6);
                }
            }
        }
        return result == INT_MAX ? -1 : result;
    }
};
