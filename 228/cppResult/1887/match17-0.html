class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        // degree
        // find trio
        // a,b; b,c; c,a
        // n^3
        // each edge (b,c): check if (a,b), (a,c) exist
        // then trio
        // b,c nbrs of a -> check pairwise if b,c are nbrs too!
        // adj matrix nxn
        
        vector<vector<bool>> A(n, vector<bool>(n,false));
        vector<int> deg(n,0);
        for(auto& e: edges) {
            A[e[0]-1][e[1]-1] = true;
            A[e[1]-1][e[0]-1] = true;
            deg[e[0]-1]++;
            deg[e[1]-1]++;
        }
        int ans = INT_MAX;
        for(int i=0;i<n;i++) {
            if(deg[i]<2) continue;
            for(int j=i+1;j<n;j++) {
                if(deg[j]<2 || !A[i][j]) continue;
                
                for(int k=j+1;k<n;k++) {
                    if(!A[i][k] || !A[j][k]) continue;
                    
                    // i,j,k trio
                    ans = min(ans, deg[i] + deg[j] + deg[k] - 6);
                }
            }
        }
        return (ans == INT_MAX) ? -1: ans;
        
        
    }
};
