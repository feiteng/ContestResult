class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> e(n, vector<int>(n));
        vector<int> d(n);
        for(auto &v : edges) {
            e[v[0] - 1][v[1] - 1] = e[v[1] - 1][v[0] - 1] = 1;
            d[v[0] - 1]++;
            d[v[1] - 1]++;
        }
        int ans = 1e9;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                if(!e[i][j]) continue;
                for(int k = j + 1; k < n; k++) {
                    if(!e[i][k] || !e[j][k]) continue;
                    ans = min(ans, d[i] + d[j] + d[k] - 6);
                }
            }
        }
        return ans == 1e9 ? -1 : ans;
    }
};
