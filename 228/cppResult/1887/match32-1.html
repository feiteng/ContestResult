class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<int> degrees(n + 1, 0);
        vector<vector<bool>> medges(n + 1,vector<bool>(n + 1, false));
        for (int i = 0; i < edges.size(); ++i)
        {
            degrees[edges[i][0]] += 1;

            degrees[edges[i][1]] += 1;
            medges[edges[i][0]][edges[i][1]] = true;
            medges[edges[i][1]][edges[i][0]] = true;
        }
        int ans = 1e9 + 7;
        for (int i = 1; i <= n; ++i)
        {
            for (int j = i + 1; j <= n; ++j)
            {
                if (medges[i][j])

                for (int k = j + 1; k <= n; ++k)
                {
                    if (medges[i][k] && medges[j][k])
                    {
                        ans = min(ans, degrees[i] + degrees[j] + degrees[k] - 6);
                    }
                }
            }
        }
        if (ans == 1e9 + 7) return -1;
        return ans;
        
    }
};
