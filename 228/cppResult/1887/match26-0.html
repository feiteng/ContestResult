class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>>adj(n,vector<int>(n));
        vector<int>cnt(n);
        
        for(auto&e:edges){
            e[0]--;
            e[1]--;
            adj[e[0]][e[1]]=true;
            adj[e[1]][e[0]]=true;
            cnt[e[0]]++;
            cnt[e[1]]++;
        }
        int ans=INT_MAX;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                // if(i==j)continue;
                for(int k=j+1;k<n;k++){
                    // if(i==k || j==k)continue;
                    if(adj[i][j] && adj[j][k] && adj[i][k]){
                        ans=min(ans,cnt[i]+cnt[j]+cnt[k]-6);
                    }
                }
            }
        }
        if(ans==INT_MAX)return -1;
        return ans;
    }
};
