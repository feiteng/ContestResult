class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> adj(n + 1, vector<int>(n + 1));
        vector<int> deg(n + 1);
        int res = INT_MAX;
        for (auto &e: edges) {
            adj[e[0]][e[1]] = 1, deg[e[0]]++;
            adj[e[1]][e[0]] = 1, deg[e[1]]++;
        }
        for (int u = 1; u <= n; u++) {
            for (int v = u + 1; v <= n; v++) {
                if (!adj[u][v]) continue;
                for (int w = v + 1; w <= n; w++) {
                    if (!adj[v][w] || !adj[u][w]) continue;
                    res = min(res, deg[u] + deg[v] + deg[w] - 6);
                }
            }
        }
        return res == INT_MAX ? -1 : res;
    }
};
