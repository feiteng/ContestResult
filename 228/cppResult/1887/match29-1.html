class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        int ans = INT_MAX;
        vector<vector<int>> graph(n+1, vector<int>(n+1, 0));
        vector<int> degrees(n+1, 0);
        for (auto& edge : edges) {
            graph[edge[0]][edge[1]] = graph[edge[1]][edge[0]] = 1;
            degrees[edge[0]]++, degrees[edge[1]]++;
        }
        
        for (int k=1; k<=n; k++) {
            for (int i=1; i<k; i++) {
                for (int j=1; j<i; j++) {
                    if (graph[i][j] && graph[i][k] && graph[j][k]) {
                        int degree = degrees[i] + degrees[j] + degrees[k] - 6;
                        ans = min(ans, degree);
                    }
                }
            }
        }
        
        return ans == INT_MAX ? -1 : ans;
    }
};
