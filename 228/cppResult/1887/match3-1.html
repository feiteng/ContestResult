class Solution {
public:
    
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        bitset<405> adjList[n+5],B;
        // vector<vector<int> > ADJ(400+1,vector<int>(400+1));
        for(auto p : edges){
            int u = p[0], v = p[1];
            adjList[u][v] = true;
            adjList[v][u] = true;
            // ADJ[u].push_back(v);
            // ADJ[v].push_back(u);
        }
        
        int ans = INT_MAX;
        
        for(int i = 1 ; i <= n ; i++){
            for(int j = i+1 ; j <= n; j++){
                if(i==j)
                    continue;
                if(adjList[i][j] && adjList[j][i] ){
                    B = (adjList[i] & adjList[j]);
                    int cnt1 = B.count();
                    if(cnt1)
                    {
                        for(int k = j+1 ; k <=n ; k++){
                            if(B[k]==true)
                            {
                                int cnt =adjList[i].count() + adjList[j].count() + adjList[k].count() - 6;                               
                                ans = min(ans,cnt);
                            }
                        }
                    }
                }
            }
        }
            
        return ans==INT_MAX ?-1 : ans;
        
    }
};
