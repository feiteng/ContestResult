class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> conn(n, vector<int>(n, 0));
        vector<int> nei(n, 0);
        for (int i = 0; i < edges.size(); ++i) {
            int u = edges[i][0];
            int v = edges[i][1];
            conn[u -1][v -1] = conn[v -1][u -1] = 1;
            ++nei[u -1];
            ++nei[v -1];
        }
        
        int res = edges.size() +1;
        
        for (int i = 0; i < n; ++i) {
            for (int j = i +1; j < n; ++j) {
                for (int k = j +1; k < n; ++k) {
                    if (conn[i][j] && conn[j][k] && conn[k][i]) {
                        res = min(res, nei[i] + nei[j] + nei[k] - 6);
                    }
                }
            }
        }
        
        return (res == edges.size() +1) ? -1: res;
    }
};
