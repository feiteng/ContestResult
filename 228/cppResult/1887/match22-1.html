class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<bool>> ok(n+1,vector<bool>(n+1,false));
        vector<int> cnt(n+1,0);
        for(int i =0;i<edges.size();i++){
            int j = edges[i][0];
            int k = edges[i][1];

            ok[j][k] = ok[k][j] = true;
            cnt[j]++;
            cnt[k]++;
        }
        
        int ans = INT_MAX;
        for(int i =1;i<=n;i++){
            for(int j = i+1;j<=n;j++){

                for(int k = j+1;k<=n;k++){
                    if(ok[i][j]&&ok[j][k]&&ok[i][k]){
                        ans = min(ans,cnt[i]+cnt[j]+cnt[k]-6);
                    }
                }
            }
        }
        if(ans >1000000)
            return -1;
        return ans;
    }
};
