class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<int>deg(n+1);
        vector<vector<int>>arr(n+1,vector<int>(n+1));
        for(int i=0;i<edges.size();i++)
        {
            arr[edges[i][0]][edges[i][1]]=1;
            arr[edges[i][1]][edges[i][0]]=1;
            deg[edges[i][0]]++;
            deg[edges[i][1]]++;
        }
        int ans=INT_MAX;
        for(int i=1;i<=n;i++)
        {
            for(int j=i+1;j<=n;j++)
            {
                for(int k=j+1;k<=n;k++)
                {
                    if(arr[i][j]==1&&arr[i][k]==1&&arr[j][k]==1)
                    {
                        ans=min(ans,deg[i]+deg[j]+deg[k]-6);
                    }
                }
            }
        }
        if(ans==INT_MAX)
            return -1;
        return ans;
    }
};
