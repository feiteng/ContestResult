class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> adj(n,vector<int>(n));
        vector<int> indeg(n);
        for(auto x:edges){
            int a=x[0],b=x[1];
            a--;
            b--;
            adj[a][b]=adj[b][a]=1;
            indeg[a]++;
            indeg[b]++;
        }
        int ans=INT_MAX;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                for(int k=j+1;k<n;k++){
                    if(adj[i][j] && adj[i][k] && adj[j][k]){
                        ans=min(ans,indeg[i]+indeg[j]+indeg[k]-6);
                    }
                }
            }
        }
        return ans==INT_MAX?-1:ans;
    }
};
