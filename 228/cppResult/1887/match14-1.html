class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> mat(n+1, vector<int>(n+1, 0));
        vector<int> dg(n+1, 0);
        for (auto edge: edges) {

            mat[edge[0]][edge[1]] = mat[edge[1]][edge[0]] = 1;
            ++dg[edge[0]]; ++dg[edge[1]];
        }
        int minv = 1e9;
        for (int i=1; i<=n; ++i) {
            for (int j=1; j<i; ++j) {

                for (int k=1; k<j; ++k) {
                    if (mat[i][j] && mat[i][k] && mat[j][k]) {
                        minv = min(
                            dg[i]+dg[j]+dg[k]-6, minv
                        );
                    }
                }
            }
        }
        
        return minv == 1e9 ? -1 : minv;
    }
};
