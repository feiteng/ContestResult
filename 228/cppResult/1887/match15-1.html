class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> g(n, vector<int>(n, 0));
        vector<int> cnt(n, 0);
        for(auto &e : edges) {
            e[0] -= 1; e[1] -= 1;

            g[e[0]][e[1]] = g[e[1]][e[0]] = 1;
            cnt[e[0]]++;
            cnt[e[1]]++;
        }
        int res = INT_MAX;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                if(g[i][j]) {
                    for(int k = j + 1; k < n; k++) {
                        if(g[j][k] && g[i][k]) {
                            res = min(res, cnt[i] + cnt[j] + cnt[k] - 6);
                        }
                    }
                }
            }
        }
        return res == INT_MAX ? -1 : res;
    }
};
