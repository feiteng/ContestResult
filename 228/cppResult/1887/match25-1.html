class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> dis(n , vector<int>(n , INT_MAX >> 2));
        vector<int> deg(n);
        for(auto e : edges){

            int u = e[0] - 1 , v = e[1] - 1;
            dis[u][v] = 1;
            dis[v][u] = 1;
            deg[u]++;
            deg[v]++;
        }
        int ans = INT_MAX;
        for(int i = 0 ; i < n ; i++){
            for(int j = i + 1 ; j < n ; j++){
                for(int k = j + 1 ; k < n ; k++){
                    if(dis[i][j] == 1 && dis[k][j] == 1 && dis[i][k] == 1){
                        ans = min(ans , deg[i] + deg[j] + deg[k] - 6);
                    }
                }
            }
        }
        if(ans == INT_MAX) ans = -1;
        return ans;
    }
};
