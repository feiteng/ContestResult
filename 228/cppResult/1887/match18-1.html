class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
    	vector<vector<int>> g(n+10, vector<int> (n+10, 0));
    	vector<int> sz(n+10, 0);
    	int ans = 0x3f3f3f3f;
    	for(auto &pi : edges) {
    		int x = pi[0], y = pi[1];

    		sz[x]++; sz[y]++;
    		g[x][y] = g[y][x] = 1;
    		// cout << x << ' ' << y << endl;
    	}
    	for(int i = 1; i <= n; i++) {
    		for(int j = 1; j <= n; j++) {
                if(i == j || !g[i][j]) continue;
    			for(int k = 1; k <= n; k++) {
                    if(j == k || i == k || !g[i][k] || !g[j][k]) continue;
    				// cout << i << ' ' << j << ' ' << k << endl;
    				ans = min(ans, sz[i] + sz[j] + sz[k] - 6);
    			}
    		}
    	}
    	return ans == 0x3f3f3f3f ? -1 : ans;
    }
};
