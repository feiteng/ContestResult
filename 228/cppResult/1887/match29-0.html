class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        int res = INT_MAX;
        vector<vector<int>> G(n+1, vector<int>(n+1, 0));
        vector<int> dgr(n+1, 0);
        for(auto e: edges){
            G[e[0]][e[1]] = 1;
            G[e[1]][e[0]] = 1;
            dgr[e[0]]++;
            dgr[e[1]]++;
        }
        for(int i = 1; i <= n; i++){
            for(int j = 1; j < i; j++){
                for(int k = 1; k < j; k++){
                    if(G[i][j] && G[j][k] && G[k][i]){
                        int tmp = dgr[i] - 2 + dgr[j] - 2 + dgr[k] - 2;
                        res = min(res, tmp);
                    }
                }
            }
        }
        if(res == INT_MAX) return -1;
        return res;
    }
};
