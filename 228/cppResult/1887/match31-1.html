class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> g(n,vector<int>(n));
        vector<int> deg(n);
        for ( auto e:edges ) {
            int u=e[0];

            int v=e[1];
            u--;
            v--;
            g[u][v]=1;
            g[v][u]=1;
            deg[u]++;
            deg[v]++;
        }
        int ans=3*n;
        for ( int i=0; i<n; i++ ) {
            for ( int j=i+1; j<n; j++ ) {
                if ( !g[i][j] ) continue;

                for ( int k=j+1; k<n; k++ ) {
                    if ( g[i][j] && g[i][k] && g[j][k] ) {
                        ans=min(ans,deg[i]+deg[j]+deg[k]-6);        
                    }
                }
            }
        }
        if ( ans>=3*n ) return -1;
        return ans;
    }
};
