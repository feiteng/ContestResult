class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>>g(n+1,vector<int>(n+1));
        vector<int>d(n+1);
        for(auto&e:edges){
            ++d[e[0]],++d[e[1]];
            g[e[0]][e[1]]=1;
            g[e[1]][e[0]]=1;
        }
        int m=edges.size(),ans=INT_MAX;
        for(int i=1;i<=n;++i){
            for(int j=i+1;j<=n;++j){
                for(int k=j+1;k<=n;++k){
                    if(g[i][j]&&g[j][k]&&g[i][k]){
                        ans=min(ans,d[i]+d[j]+d[k]-6);
                    }
                }
            }
        }
        return ans==INT_MAX?-1:ans;
    }
};
