class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> adj(n, vector<int>(n, 0));
        vector<int> deg(n, 0);
        for (auto& ed : edges) {
            --ed[0]; --ed[1];
            adj[ed[0]][ed[1]] = 1;
            adj[ed[1]][ed[0]] = 1;
            ++deg[ed[0]];
            ++deg[ed[1]];
        }
        
        constexpr int INF = 1e9;

        int ans = INF;
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                for (int k = j + 1; k < n; ++k) {
                    if (adj[i][j] && adj[i][k] && adj[j][k]) {
                        ans = min(ans, deg[i] + deg[j] + deg[k] - 6);
                    }
                }
            }
        }
        
        if (ans == INF) ans = -1;
        
        return ans;
    }
};
