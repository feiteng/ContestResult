class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
		vector<vector<int>> adj(n,vector<int>(n));
		vector<int> deg(n);
		for(int i=0;i<edges.size();i++){
			int u=edges[i][0]-1;
			int v=edges[i][1]-1;
			adj[u][v]=1;
			adj[v][u]=1;
			deg[u]++;
			deg[v]++;
		}
/*		for(int i=0;i<n;i++){
			cout<<deg[i]<<endl;
		}
		cout<<endl;
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				cout<<adj[i][j]<<" ";
			}
			cout<<endl;
		}
		cout<<endl;*/
		int ret=INT_MAX;
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				for(int k=j+1;k<n;k++){
					if(adj[i][j]&&adj[j][k]&&adj[k][i]){
						ret=min(ret,deg[i]+deg[j]+deg[k]-6);
					}
				}
			}
		}
		if(ret==INT_MAX)return -1;
		return ret;
        
    }
};
