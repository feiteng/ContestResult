class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<unordered_map<int, int>> g;
        vector<int> deg;
        deg.resize(n);
        g.resize(n);
        for(auto edge : edges) {
            int x = edge[0] - 1;
            int y = edge[1] - 1;
            g[x][y] = 1;
            g[y][x] = 1;
            deg[x]++;
            deg[y]++;
            
        }
        int ans = INT_MAX;
        
        
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                if(!g[i].count(j)) continue;
                for(int k = j + 1; k < n; k++) {
                    if(g[i].count(k) && g[j].count(k)) {
                        // cout << i << j << k << endl;
                        ans = min(ans, deg[i] + deg[j] + deg[k] - 6);
                    }
                }
            }
        }
        
        return ans == INT_MAX ? -1 : ans;
    }
};
