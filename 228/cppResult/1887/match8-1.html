class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        int ans = INT_MAX;
        vector<int> degree(n + 1, 0);
        
        vector<vector<bool>> conn(n + 1, vector<bool>(n + 1, false));
        for (auto edge : edges){
            degree[edge[0]]++;
            degree[edge[1]]++;
            conn[edge[0]][edge[1]] = true;
            conn[edge[1]][edge[0]] = true;
        }
        for (int i = 1; i <= n; i++)
            for (int j = i+1; j <= n ; j++)
                for (int k = j+1; k <= n; k++ )
                    if (conn[i][j] && conn[i][k] && conn[j][k] && (degree[i] + degree[j] + degree[k] - 6 < ans))
                        ans = degree[i] + degree[j] + degree[k] - 6;
        
     return ans == INT_MAX ? -1 : ans;   
    }
};
