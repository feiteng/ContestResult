class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int> > v(n+1,vector<int>(n+1));
        vector<int> c(n+1);
        for(auto& e: edges) {
            v[e[0]][e[1]] = 1;
            v[e[1]][e[0]] = 1;
            c[e[0]]++;
            c[e[1]]++;
        }
        int res = INT_MAX;
        for(int i = 1; i <= n; ++i) {
            for(int j = i+1; j <= n; ++j) {
                if(v[i][j] == 0 || c[i]+c[j]-4 > res) continue;
                for(int k = 1; k <= n; ++k) {
                    if(k == i || k == j || v[i][k] == 0 || v[j][k] == 0) continue;
                    res = min(res, c[i]+c[j]+c[k]-6);
                }
            }
        }
        return res == INT_MAX ? -1 : res;
    }
};
