class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> mat(n, vector<int>(n, 0));
        vector<int> degree(n, 0);
        
        for(auto &vec : edges) {
            int u = vec[0], v = vec[1];

            u--; v--;
            degree[u]++; degree[v]++;
            mat[u][v] = 1;
            mat[v][u] = 1;
        }
        
        int ans = 1e9;
        
        for(int a = 0; a < n; a++) {
            for(int b = a + 1; b < n; b++) {

                for(int c = b + 1; c < n; c++) {
                    if(mat[a][b] and mat[b][c] and mat[c][a]) {
                        ans = min(ans, degree[a] + degree[b] + degree[c] - 6);
                    }
                }
            }
        }
        
        return (ans < 1e9 ? ans : -1);
        
    }
};
