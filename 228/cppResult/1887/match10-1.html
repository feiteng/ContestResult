class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> v(n, vector<int>(n, 0));
        vector<int> degree(n, 0);
        for (auto& ed: edges) {
            ed[0]--;ed[1]--;

            v[ed[0]][ed[1]] = 1;
            v[ed[1]][ed[0]] = 1;
            degree[ed[0]]++;
            degree[ed[1]]++;
        }
        int mini = 100000000;
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 1; j < n - 1; j++) {
                if (!v[i][j]) continue;
                for (int k = j + 1; k < n; k++) {
                    if (!v[i][k] || !v[j][k]) continue;
                    mini = min(mini, degree[i] + degree[j] + degree[k] - 6);
                }
            }
        }
        return mini == 100000000 ? -1 : mini;
    }
};
