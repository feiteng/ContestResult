class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& e) {
        vector<int> v(n+1);
        vector<vector<bool>> f(n+1,vector<bool>(n+1));
        for(auto& x:e){
            v[x[0]]++;
            v[x[1]]++;
            f[x[0]][x[1]]=f[x[1]][x[0]]=true;
        }
        int ans=1e9;
        for(int i=1;i<n;i++){
            for(int j=i+1;j<n;j++){
                for(int k=j+1;k<=n;k++){
                    if(f[i][j]&&f[j][k]&&f[i][k]){
                        ans=min(ans,v[i]+v[j]+v[k]-6);
                    }
                }
            }
        }
        if(ans==1e9) return -1;
        else return ans;
    }
};
