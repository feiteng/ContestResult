class Solution {
public:    
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        bitset<405> LIST[n+5],SBIT;
        for(auto vec : edges){
            int u = vec[0], v = vec[1];
            LIST[u][v] = 1;	LIST[v][u] = 1;
        }
        
        int ans = INT_MAX;
        
        for(int i = 1 ; i <= n ; i++){
            for(int j = i+1 ; j <= n; j++){
                if(i==j)
                    continue;
                if(LIST[i][j] && LIST[j][i] ){
                    SBIT = (LIST[i] & LIST[j]);
                    int cnt1 = SBIT.count();
                    if(cnt1)
                    {
                        for(int k = j+1 ; k <=n ; k++){
                            if(SBIT[k]==1)
                            {
                                int cnt =LIST[i].count() + LIST[j].count() + LIST[k].count() - 6;                               
                                ans = min(ans,cnt);
                            }
                        }
                    }
                }
            }
        }
            
        return ans==INT_MAX ?-1 : ans;
        
    }
};
