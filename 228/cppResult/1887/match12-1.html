class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>> grid(n + 1, vector<int>(n + 1, 0));
        vector<int> degree(n + 1, 0);
        for (auto& e : edges) {
            grid[e[0]][e[1]] = 1;
            grid[e[1]][e[0]] = 1;
            degree[e[0]]++;
            degree[e[1]]++;
        }
        
        int res = INT_MAX;
        vector<vector<int>> trios = {};

        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                for (int k = j + 1; k <= n; k++) {
                    if (grid[i][j] && grid[j][k] && grid[i][k]) {
                        res = min(res, degree[i] + degree[j] + degree[k] - 6);
                    }
                }
            }
        }
        
        return res == INT_MAX ? -1 : res;
    }
};
