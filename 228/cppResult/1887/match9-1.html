class Solution {
public:
    int minTrioDegree(int n, vector<vector<int>>& edges) {
        vector<vector<int>>graph(n, vector<int>(n));
        vector<int> cnt(n);
        for(auto & e : edges) {
            graph[e[0] - 1][e[1] - 1] = 1;
            graph[e[1] - 1][e[0] - 1] = 1;
            cnt[e[0] - 1]++;
            cnt[e[1] - 1]++;
        }
        int res = 1e9;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++) {
                if(graph[i][j]== 0) continue;
                for(int k = j + 1; k < n; k++) {
                    if(graph[i][k] == 0 || graph[j][k] == 0) continue;
                    res = min(res, cnt[i] + cnt[j] + cnt[k] - 6);
                }
            }
        }
        return res == 1e9 ? -1 : res;
        
    }
};
