class Solution {
public:
    int countHomogenous(string s) {
        int ans=0;
        int mod=1e9+7;
        for(int i=0;i<s.size();i++)
        {
        int j=i;
            while(j<s.size() && s[j]==s[i])
            {
            j++;
            }
            long long d=j-i;
            long long c=d*(d+1)/2;
            c%=mod;
            ans=(ans+c)%mod;
            i=j-1;
        }
        return ans;
    }
};
