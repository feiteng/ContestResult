class Solution {
public:
    int countHomogenous(string s) {
        const unsigned int M = 1000000007;
        long long res = 0;
        long long count=1;
        for(int i=0;i<s.size()-1;i++)
        {
            if(s[i]==s[i+1])
            {
                count++;
            }
            else
            {
                res=res+(count*(count+1)/2);
                count=1;
            }
        }
        res=res+(count*(count+1)/2);
        return res%M;
    }
};
