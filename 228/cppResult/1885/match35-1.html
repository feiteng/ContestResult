class Solution {
public:
    int countHomogenous(string s) {
        int n = s.size();
        char prev = '?';
        long long now = 0;
        long long ans = 0;
        
        for (int i = 0; i < n; i++) {
            if (s[i] != prev) {
                ans += (now + 1) * now / 2; 
                now = 0;
            }
            prev = s[i];
            now++;
        }
        ans += (now + 1) * now / 2; 
        return ans % 1000000007;
    }
};
