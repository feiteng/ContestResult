#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int countHomogenous(string s) {
        long long n = s.size(), cnt = 1, ans = 0, mod = 1e9 + 7;
        for (int i = 1; i < n; i++) {
            if (s[i] == s[i - 1]) cnt++;
            else {
                ans = (ans + cnt * (cnt + 1) / 2) % mod;
                cnt = 1;
            }
        }
        ans = (ans + cnt * (cnt + 1) / 2) % mod;
        return ans;
    }
};
