class Solution {
public:
    int countHomogenous(string s) {
        int mod = 1e9 + 7;
        int res = 0;
        int n = s.size();
        for(int i = 0; i < n; i ++) {
            int j = i;
            while(j < n && s[j] == s[i]) j ++;
            res = (res + (long long)(j - i + 1) * (j - i) / 2) % mod;
            i = j - 1;
        }
        return res;
    }
};
