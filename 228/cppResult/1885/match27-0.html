const int mod = 1e9+7;
class Solution {
public:
    int countHomogenous(string s) {
        int ans = 0;
        
        int n = s.size();
        
        for(int i = 0; i<n; i++){
            int j = i+1;
            while(j<n && s[j] == s[i]){
                j++;
            }
            long long cnt = j-i;
            ans = ans + cnt*(cnt+1)/2 %mod;
            i = j-1;
        }
        
        return ans;
    }
};
