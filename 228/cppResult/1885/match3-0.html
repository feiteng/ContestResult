class Solution {
public:
    int countHomogenous(string s) {
        long long MOD = 1e9 + 7, ans = 0;
        for(int i = 0; i < s.length(); i ++) {
            int j = i;
            while(j + 1 < s.length() && s[j + 1] == s[i]) j ++;
            long long cnt = j - i + 1;
            ans = (ans + cnt * (cnt + 1) / 2 % MOD) % MOD;
            i = j;
        }
        return ans;
    }
};
