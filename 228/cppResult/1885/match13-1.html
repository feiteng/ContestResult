class Solution {
public:
    #define ll long long
    int countHomogenous(string s) {
        int M = 1e9 + 7;
        ll ret = 0;

        int n = s.length();
        vector<int> dp(n, 0);
        dp[0] = 1;
        ret += 1;
        for (int i = 1; i < n; ++i) {
            if (s[i] == s[i - 1]) {
                dp[i] = dp[i - 1] + 1;
            }
            else {
                dp[i] = 1;
            }
            ret = (ret + dp[i]) % M;
        }
        
        return ret;
    }
};
