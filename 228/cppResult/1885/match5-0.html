constexpr int kMod = 1000000007;

class Solution {
public:
    int countHomogenous(string s) {
      long long count = 0;
      long long run_length = 0;
      for (int i = 0; i < s.length(); ++i) {
        if (i == 0 || s[i - 1] == s[i]) {
          ++run_length;
        } else {
          count += ((run_length + 1) * (run_length)) / 2;
          count %= kMod;
          run_length = 1;
        }
      }
      count += ((run_length + 1) * (run_length)) / 2;
      count %= kMod;
      return count;
    }
};
