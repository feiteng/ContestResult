class Solution {
    static const int MOD = 1e9+7;
    int firstn(int n) {
        return ((1LL * n * (n + 1))/2)%MOD;
    }
public:
    int countHomogenous(string s) {
        int cur = 0, ans = 0;
        for(int i = 0; i < s.length(); i++) {
            if(i && (s[i] != s[i-1])) {
                ans = (ans + firstn(cur))%MOD;
                cur = 1;
            }
            else {
                cur++;
            }
        }
        ans = (ans + firstn(cur))%MOD;
        return ans;
    }
};
