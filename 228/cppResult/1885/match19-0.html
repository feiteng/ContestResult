class Solution {
public:
    int countHomogenous(string s) {
        long long ret = 0, mod = 1e9+7, cur = 0;
        char last = -1;
        for(char c: s) {
            if(c == last) {
                ++cur;
            } else {
                cur = 1;
                last = c;
            }
            ret += cur;
            ret %= mod;
        }
        return ret;
    }
};
