class Solution {
public:
    long long MOD = 1e9 + 7;
    int countHomogenous(string a) {
        int n = (int)a.size();

        long long cur = 1;
        long long answer = 0;
        for (int i = 1; i < n; i++) {
            if (a[i] != a[i - 1]) {
                answer = (answer + (cur + 1) * cur / 2) % MOD;
                cur = 1;
            }
            else cur++;
        }
        answer = (answer + (cur + 1) * cur / 2) % MOD;
        return answer;
    }
};
