class Solution {
public:
    int countHomogenous(string s) {
        long long ans=0;
        s.push_back('0');
        //cout<<s<<endl;
        long long sum=1;
        int n=s.length();
        for(int i=1;i<n;i++){
            if(s[i]!=s[i-1]){
                ans+=(sum+1)*sum/2;
                sum=1;
                ans%=1000000007;
            }else{
                sum++;
            }
        }
        return ans;
    }
};
