class Solution {
public:
    using ll = long long;
    const ll mod= 1e9 + 7;
    int countHomogenous(string s) {
        int n = s.size();
        ll ans = 0;
        ll cnt = 1;
        for(int i = 0; i < n-1; i++){
            if(s[i] == s[i+1])cnt++;
            else{
                ans = (ans + (cnt*cnt+cnt)/2)%mod;
                cnt = 1;
            }
        }
        ans = (ans + (cnt*cnt+cnt)/2)%mod;
        return ans;
    }
};
