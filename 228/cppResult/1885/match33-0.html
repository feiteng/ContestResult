class Solution {
public:
    int countHomogenous(string s) {
        int index = 0;
        int64_t result = 0;
        while (index < s.size())
        {
            int current = index;
            while (current < s.size() && s[current] == s[index]) current++;
            current -= index;
            result = (result + (int64_t)current * (current + 1) / 2) % 1000000007;
            index += current;
        }
        
        return result;
    }
};
