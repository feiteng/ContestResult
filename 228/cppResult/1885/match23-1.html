const int mod = 1000000007;
class Solution {
public:
    int countHomogenous(string s) {
        int n = s.size();

        char c = ' ';
        long long res = 0;
        long long cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (c != s[i]) {
                res += (cnt + 1) * cnt / 2;
                cnt = 1;
                c = s[i];
            } else {
                cnt++;
            }
        }
        res += (cnt + 1) * cnt / 2;
        return res % mod;
    }
};
