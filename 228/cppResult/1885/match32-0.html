typedef long long ll;
class Solution {
public:
    int countHomogenous(string s) {
        ll mod = 1e9+7;
        ll cnt = 1, ans = 0;
        for(int i = 1;i < s.size();i++){
            if(s[i]!=s[i-1]){
                ans += cnt*(cnt+1)/2;
                ans %= mod;
                cnt = 1;
            }else{
                cnt++;
            }
        }
        ans += cnt*(cnt+1)/2;
        return ans%mod;
    }
};
