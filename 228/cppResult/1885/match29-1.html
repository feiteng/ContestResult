#include "bits/stdc++.h"
using namespace std;

const int MOD = 1e9 + 7;

class Solution {
public:

	int countHomogenous(string s) {
		int l = 1, ans = 1;
		for(int i = 1; i < s.size(); i++) {
			if(s[i] == s[i - 1]) {
				l++;
			}
			else {
				l = 1;
			}
			ans += l;
			ans %= MOD;
		}
		return ans;
	}
};
