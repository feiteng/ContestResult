class Solution {
public:
    int countHomogenous(string s) {
        int ans = 1;
        int repeat = 1, M = 1000000007;
        for (int i = 1; i < s.size(); ++i) {
            if (s[i] == s[i - 1]) {
                ++repeat;
            } else {
                repeat = 1;
            }
            ans = (ans + repeat) % M;
        }
        return ans;
    }
};
