class Solution {
public:
    int countHomogenous(string s) {
        if(s.size()==1){return 1;}

        char c=s[0];
        long long cur=1;
        long long res=0;
        for(int i=1;i<s.size();i++){
            if(s[i]==c){cur++;}
            else{
                res+=cur*(cur+1)/2;
                //cout<<res<<endl;
                cur=1;
                c=s[i];
            }
        }
         res+=cur*(cur+1)/2;
        return res%1000000007;
    }
};
