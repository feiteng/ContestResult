#include<bits/stdc++.h>
#define ll long long int
#define pb push_back
#define ff first	
#define ss second
#define MOD 1000000007
using namespace std;


class Solution {
public:
    int countHomogenous(string s) {
        

        ll n,i,j,ans=0,cnt=1;
        
        n=s.length();
        
        for(i=1;i<n;i++){
            
            if(s[i] == s[i-1])
                cnt++;
            
            else{
                ans=(ans+(cnt*(cnt+1))/2)%MOD;
                cnt=1;
            }
            
        }
        
        ans=(ans+(cnt*(cnt+1))/2)%MOD;
        
        return ans;
        
    }
};
