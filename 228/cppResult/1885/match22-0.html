class Solution {
public:
    int countHomogenous(string s) {
        int n = s.size(), i;
        long long t = 1, ans = 0, mod = 1e9 + 7;
        for (i = 1; i < n; ++i) {
            if (s[i] == s[i - 1])
                ++t;
            else {
                ans = (ans + t * (t + 1) / 2) % mod;
                t = 1;
            }
        }
        ans = (ans + t * (t + 1) / 2) % mod;
        return ans;
    }
};
