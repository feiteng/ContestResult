class Solution {
public:

    int countHomogenous(string s) {
        int n = s.length();
        int cnt = 1, ans=0, mod=1e9+7;
        for(int i=0; i<n-1; i++){
            if(s[i]==s[i+1])cnt++;
            else{
                ans =((1ll*cnt*(cnt+1)/2)%mod+ans)%mod;
                cnt=1;
            }
            //printf("%d\n", cnt);
        }
        ans =((1ll*cnt*(cnt+1)/2)%mod+ans)%mod;
        return ans;
    }
};
