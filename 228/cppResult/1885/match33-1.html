class Solution {
public:
    int countHomogenous(string s) {
        const int mod = 1e9 + 7;
        typedef long long ll;
        int index = 0;

        int n = s.size();
        ll ans = 0;
        while (index < n) {
            int next = index;
            while (next < n && s[next] == s[index])
                next++;
            ll cnt = next - index;
            ans += cnt * (cnt + 1) / 2;
            index = next;
        }
        return ans % mod;
    }
};
