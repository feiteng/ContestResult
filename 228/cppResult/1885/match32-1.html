#define MAXN 100010
#define MODN 1000000007
typedef long long ll;
class Solution {
public:
    ll power[MAXN];
    int countHomogenous(string s) {
        int n = s.length();
        power[0] = 0;
        for(int i = 1; i<=n;++i)
            power[i] = (power[i-1]+i)%MODN;

        char b = 0;
        ll ans = 0, cnt = 0;
        for(int i = 0; i < n; ++i){
            if(b != s[i]){
                b = s[i];
                ans = (ans + power[cnt])%MODN;
                cnt = 1;
            }else{
                cnt = cnt + 1;
            }
        }
        ans = (ans + power[cnt])%MODN;
        return ans;
    }
};
