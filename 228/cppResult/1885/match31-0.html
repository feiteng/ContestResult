const int mod = 1e9 + 7;

class Solution {
public:
    int countHomogenous(string str) {
        int n = str.length();
        vector<int> dp(n, 0);
        /* dp[i] : answer starts at i */
        dp[n - 1] = 1;
        for(int i = n - 2; i >= 0; i--) {
            if(str[i] == str[i + 1]) {
                dp[i] = 1 + dp[i + 1];
                dp[i] %= mod;
            }
            else {
                dp[i] = 1;
            }
        }
        
        long long sum = 0;
        for(auto &ele : dp) {
            sum += ele;
            sum %= mod;
        }
        
        return sum;
    }
};
