class Solution {
public:
    int countHomogenous(string s) {
        long long int prime=1e9+7;
        

        int n=s.size();
        long long int cnt=1;
        
        long long int ans=0;
        
        for(int i=1;i<n;i++)
        {
            if(s[i]==s[i-1])    cnt++;
            else
            {
                ans+=cnt*(cnt+1);
                cnt=1;
            }
        }
        
        ans+=cnt*(cnt+1);
        
        ans/=2;
        
        return ans%prime;
    }
};
