class Solution {
    int MOD = pow(10, 9) + 7;
public:
    int countHomogenous(string s) {
        vector<int> dp(s.size(), 0);

        int ans = 1;
        dp[0] = 1;
        for (int i = 1; i < s.size(); i++) {
            if (s[i] != s[i - 1]) {
                dp[i] = 1;
            }
            else {
                dp[i] = dp[i - 1] + 1;
            }
            ans = (ans + dp[i]) % MOD;
        }
        return ans;
    }
};
