class Solution {
public:
    int countHomogenous(string s) {
        long long ans = 0;
        for (int i = 0; i < s.size(); ++i) {
            int j = i;
            while (j + 1 < s.size() && s[j + 1] == s[i]) ++j;
            long long L = j - i + 1;
            ans += L * (L - 1) / 2 + L;
            i = j;
        }
        return ans % (long long)(1e9 + 7);
    }
};
