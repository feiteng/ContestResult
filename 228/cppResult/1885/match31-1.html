class Solution {
public:
    int countHomogenous(string s) {
        int ans = 0;
        int cnt = 1;
        for (int i = 0; i < s.size(); i++) {
            if (i + 1 == s.size() || s[i] != s[i + 1]) {
                ans = (ans + (int64_t)cnt * (cnt + 1) / 2) % kMod;
                cnt = 1;
            } else {
                cnt++;
            }
        }
        return ans;
    }
private:
    static constexpr int kMod = 1e9 + 7;
};
