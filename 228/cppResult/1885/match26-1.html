typedef long long ll;

class Solution {
public:
    int countHomogenous(string s) {
        int M = 1000000007;
        ll res = 0, cur = 1;
        for(int i = 1; i < s.size(); ++i) {
            if(s[i] != s[i-1]) {
                res = (res + cur*(cur+1)/2) % M;
                cur = 1;
            }
            else ++cur;
        }
        return (res + cur*(cur+1)/2) % M;
    }
};
