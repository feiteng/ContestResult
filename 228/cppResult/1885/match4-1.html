class Solution {
public:
    int countHomogenous(string s) {
        long long ret = 0;
        long long count = 1;
        long long mod = 1000000007;

        for(int i=1;i<s.size();i++) {
            if(s[i] == s[i-1]) {
                count++;
            } else {
                ret += ((count * (count+1))/2);   
                ret = ret % mod;
                count = 1;
            }            
        }
        ret += ((count * (count+1))/2);   
        ret = ret % mod;
        return (int) ret;
    }
};
