class Solution {
public:
    int countHomogenous(string s) {
        
        int ret = 1;
        int mod = 1000000007;
        
        int n = s.length();
        int consecutive = 1;
        
        
        for (int i=1; i<n; ++i) {
            if (s.at(i) == s.at(i-1)) {
                ++consecutive;
            }
            else {
                consecutive = 1;
            }
            ret = (ret + consecutive) % mod;
        }
        
        
        
        
        
        return ret;
    }
};
