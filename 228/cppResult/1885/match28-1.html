class Solution {
    const int mod=1e9+7;
public:
    int countHomogenous(string s) {

        long res=0, cnt=1;
        for (int i=1; i<s.size(); ++i) {
            if (s[i]==s[i-1]) {
                cnt++;
            } else {
                res=res+(1+cnt)*cnt/2;
                res%=mod;
                cnt=1;
            }
        }
        res=res+(1+cnt)*cnt/2;
        res%=mod;
        return res;
    }
};
