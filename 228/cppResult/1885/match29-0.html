class Solution {
public:
    int mod=1e9+7;
    int countHomogenous(string s) {
        int ans=1,cnt=1;
        for(int i=1;i<s.size();++i){
            if(s[i]==s[i-1]){
                ++cnt;
            }
            else{
                cnt=1;
            }
            ans+=cnt;
            ans%=mod;
        }
        return ans;
    }
};
