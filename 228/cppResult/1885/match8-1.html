#define ll long long
#define MOD 1000000007

class Solution {
public:
    int countHomogenous(string s) {

        char c = '-';
        ll chain = 0;
        
        ll ans = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s[i] != c) {
                ans += chain * (chain + 1) / 2;
                ans %= MOD;
                c = s[i];
                chain = 1;
            } else chain++;
        }
        
        ans += chain * (chain + 1) / 2;
        ans %= MOD;
        return ans;
    }
};
