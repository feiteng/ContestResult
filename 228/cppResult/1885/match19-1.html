class Solution {
public:
    int countHomogenous(string s) {
        int mod=1000000007;

        int ans=0,now=1,tmp=2;
        char pre=' ';
        for(char c:s) {
            if(c==pre) {
                now++;
            }
            else {
                now=1;
                tmp=2;
            }
            pre=c;
            ans=(ans+now)%mod;
        }
        return ans;
    }
};
