const int mod = 1e9 + 7;
class Solution {
public:
    int countHomogenous(string s) {
        long long res = 0;
        for (int i = 0; i < s.size(); i++) {
            int j = i;
            while(j + 1 <= s.size() && s[j + 1] == s[i]) j++;
            long long len = j - i + 1;
            res = (res + (len + 1) * (len) / 2 % mod) % mod;
            //cout << i << ' ' << j << '\n';
            i = j;
        }
        //cout << "----" << '\n';
        return res;
    }
};
