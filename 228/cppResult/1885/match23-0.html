class Solution {
public:
    long long mod=1e9+7;
    int countHomogenous(string s) {
        int now=s[0];
        long long sum=0;
        long long num=1;
        for (int i=1;i<s.size();i++){
            if (s[i]!=now){
                sum=(sum+(num+1)*num/2)%mod;
                now=s[i];
                num=1;
            }
            else{
                num++;
            }
        }
        sum=(sum+(num+1)*num/2)%mod;
        return sum;
    }
};
