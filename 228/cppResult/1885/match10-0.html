class Solution {
public:
    
    using ll = long long;
    
    ll calc(int x) {
        return (1LL + x) * x / 2;
    }
    int countHomogenous(string s) {
        int n = s.length();
        int cnt = 0;
        
        ll mod = 1e9+7;
        ll ret = 0;
        for (int i = 0; i < n; i++) {
            if (i == 0 || s[i] == s[i-1]) cnt++;
            else {
                ret = (calc(cnt) + ret) % mod;
                cnt = 1;
            }
        }
        ret = (calc(cnt) + ret) % mod;
        return ret;
    }
};
