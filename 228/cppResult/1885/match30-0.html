typedef long long LL;
const int MOD = 1e9 + 7;
class Solution {
public:
    int countHomogenous(string s) {
        LL cnt = 1, ret = 0;
        s += '.';
        for (int i = 1; i < s.size(); i ++ )
        {
            if (s[i] == s[i - 1])
                cnt ++;
            else
            {
                ret += (cnt + 1) * cnt / 2;
                ret %= MOD;
                // printf("i=%d, tmp=%d\n", i, (cnt + 1) * cnt / 2);
                cnt = 1;
            }
        }
        return ret % MOD;
    }
};
