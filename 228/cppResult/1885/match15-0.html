class Solution {
public:
    int MOD = 1000000007;
    int countHomogenous(string s) {
        
        char cur = 'a'-1;
        long long num = 0;
        long long res = 0;
        for(int i = 0; i < s.size(); i++){
            if(s[i] == cur){
                num++;
            }
            else{
                res = (res + (num+1)*num/2)%MOD;
                num = 1;
                cur = s[i];
            }
        }
        res = (res + (num+1)*num/2)%MOD;
        return res;
    }
};
