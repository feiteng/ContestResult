class Solution {
public:
    int countHomogenous(string s) {

        long long ret=0;
        for(int i=0;i<s.size();i++){
            int j = i;
            while(j+1<s.size()&&s[j+1]==s[i])j++;
            long long tt = j-i+1;
            ret += tt*(tt+1)/2;
            i=j;
        }
        return ret %1000000007;
    }
};
