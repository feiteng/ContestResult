class Solution {
public:
    #define ll long long
    int countHomogenous(string s) {
        ll cur = s[0];
        ll tim = 1;
        ll ans = 0;
        for(int i = 1;i<s.size();i++)
        {
            if(cur != s[i])
            {
                ans+=(1+tim)*tim/2;
                ans %= 1000000007;
                tim = 1;
                cur = s[i];
            }
            else tim++;
        } 
        ans+=(1+tim)*tim/2;
        ans %= 1000000007;
        return ans;
    }
};
