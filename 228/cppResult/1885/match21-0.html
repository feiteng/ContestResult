class Solution {
public:
    int countHomogenous(string s) {
        int n=s.length();
        long long int k=1;
        long long int ans=0;
        for(int i=1;i<n;i++)
        {
            if(s[i]==s[i-1])
                k++;
            else
            {
                ans+=(k*(k+1))/2;k=1;
            }
        }
         ans+=(k*(k+1))/2;
        ans=(ans)%1000000007;
        return (int)ans;
    }
};
