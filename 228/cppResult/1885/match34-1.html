#define ll long long 
class Solution {
public:
    int countHomogenous(string s) {
        int bp = 1e9+7;

        ll c=1,ans=0;
        for(int i=1;i<s.size();i++){
            if(s[i]!=s[i-1]){
                ans += ((c*(c+1))%bp)/2;
                c=1;
            }
            else{
                c++;
            }
        }
        ans += ((c*(c+1))%bp)/2;
        return ans;
    }
};
