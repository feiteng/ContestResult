class Solution {
public:
    int countHomogenous(string s) {
        static const int MOD = 1e9+7;
        int n = s.size();
        long long res = 0;
        for (int i = 0, j; i < n; i = j) {
            j = i+1;
            while (j < n && s[j] == s[i]) ++j;
            res += (j-i)*(j-i+1ll)/2;
        }
        return res%MOD;
    }
};
