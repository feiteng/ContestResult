class Solution {
public:
    int countHomogenous(string s) {
        long long res = 0;
        long long mod = 1e9 + 7;
        int n = s.size();
        s.push_back('.');

        long long cnt = 0;
        for(int i = 0; i <= n; ++i) {
            if(i && (s[i] != s[i-1])) {
                res += cnt * (cnt + 1) / 2;
                res %= mod;
                cnt = 1;
            } else {
                cnt += 1;
            }
        }
        
        return res % mod;
    }
};
