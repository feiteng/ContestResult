class Solution {
public:
    bool check(vector<int>&nums,int cur,int k){
        int op=0;
        for(int &x:nums){
            op+=((x-1)/cur);
        }
        return op<=k;
    }
    int minimumSize(vector<int>& nums, int maxOperations) {
        int lo=1,hi=0,n=nums.size();
        for(int &x:nums){
            hi=max(hi,x);
        }
        while(lo<hi){
            int mid=lo + (hi-lo)/2;
            if(check(nums,mid,maxOperations)){
                hi=mid;
            }
            else{
                lo=mid+1;
            }
        }
        return hi;
    }
};
