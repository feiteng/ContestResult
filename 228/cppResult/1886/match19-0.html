class Solution {
    bool ok(int t, vector<int>& nums, int maxOp) {
        int cnt = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] > t) {
                cnt += (nums[i] % t == 0) ? (nums[i]/t-1) : nums[i]/t;
            }
        }
        return cnt <= maxOp;
    }
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int begin = 1, end = *max_element(nums.begin(), nums.end());
        while (begin < end) {
            int mid = begin + (end - begin)/ 2;
            if (ok(mid, nums, maxOperations)) {
                end = mid;
            } else {
                begin = mid + 1;
            }
        }
        return begin;
    }
};
