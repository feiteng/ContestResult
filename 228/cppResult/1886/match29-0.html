class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int L=1;
        int R=1e9;
        
        while(L<R){
            int m=(L+R)/2;
            int op = 0;
            
            for(auto x:nums){
                op+=x/m;
                if((x%m)==0){
                    op--;
                }
            }
            
            if(op>maxOperations){
                L=m+1;
            }else{
                R=m;
            }
        }
        
        
        return L;
    }
};
