class Solution {
public:
    bool works(vector<int>& nums, int op, int balls) {
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] <= balls) {
                continue;
            }
            op -= (nums[i]+balls-1)/balls-1;

        }
        return op >= 0;
    }
    
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1, r = 1e9;
        while(l < r) {
            int m = (l+r)/2;
            if (works(nums, maxOperations, m)) {
                r = m;
            }
            else {
                l = m+1;
            }
        }
        return l;
    }
};
