class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l=1;
        int r=*max_element(nums.begin(),nums.end());
        while(l<r){
            int m=(l+r)/2;
            int t=0;
            for(int q:nums){
                t+=q/m;
                if(q%m==0)
                    --t;
            }
            if(t<=maxOperations)
                r=m;
            else
                l=m+1;
        }
        return l;
    }
};
