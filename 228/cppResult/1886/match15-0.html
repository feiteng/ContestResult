class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int lo = 1, hi = 1e9;
        while (lo < hi) {
            int m = (lo + hi) >> 1;
            long long need = 0;
            for (auto it : nums) {
                need += (it + m - 1) / m - 1;
            }
            if (need <= maxOperations) {
                hi = m;
            } else {
                lo = m + 1;
            }
        }
        return lo;
    }
};
