class Solution {
public:
    int getRequiredOperations (vector<int>& nums, int min_num) {
        int res = 0;
        for (auto num: nums) {
            if (num > min_num) {
                res += ((num-1)/min_num);

            }
        }
        return res;
    }
    int minimumSize(vector<int>& nums, int maxOperations) {
        int lo = 1;
        int hi = 1e9;
        
        while (lo < hi) {
            int mid = (lo+hi)/2;
            if (getRequiredOperations(nums, mid) > maxOperations) {
                lo = mid+1;
            } else {
                hi = mid;
            }
        }
        
        return (lo);
    }
};
