class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int left = 0, right = 1000000001;
        while (right - left > 1) {
            int mid = (left + right) / 2;
            long long t = 0;
            for (int i = 0;i < nums.size();i++) {
                if (nums[i] % mid == 0) {
                    t += nums[i] / mid - 1;
                } else {
                    t += nums[i] / mid;
                }

            }
            if (t <= maxOperations) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
};
