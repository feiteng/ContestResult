class Solution {
public:
    int minimumSizeTgt(vector<int>& nums, int tgt) {
        int res = 0;
        for(int x : nums) {
            res += (x+tgt-1) / tgt - 1;
        }
        return res;
    }
    int minimumSize(vector<int>& nums, int ops) {
        int lb,ub,mb;
        lb = 0;
        ub = 1e9;
        for(;lb+1<ub;) {

            mb = (lb+ub)/2;
            if(minimumSizeTgt(nums,mb) <= ops) {
                ub = mb;
            } else {
                lb = mb;
            }
        }
        return ub;
    }
};
