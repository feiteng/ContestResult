class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1; int h = 1e9;
        while (l < h) {
            int m = (l+h)/2;
            int k = maxOperations;
            for (auto x : nums) {

                if (x > m) k -= (x-1)/m;
            }
            // cout << l << " " << h << " " << k << endl;
            if (k < 0) {
                l = m+1;
            } else {
                h = m;
            }
        }
        return l;
    }
};
