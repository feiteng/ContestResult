class Solution {
public:
    int minimumSize(vector<int>& v, int maxOperations) {
        int lo = 1; int hi = 1000000001;
        
        while (lo != hi) {
            int me = (lo +hi)/2;
            int cnt = 0;
            
            for (int i = 0; i < v.size(); i++) {
                if (v[i] <= me)continue;

                cnt +=  (v[i]-1)/me;
            }
            
            //cout<<me<<" "<<cnt<<endl;
            if (cnt <= maxOperations) {
                hi = me;
            } else {
                lo = me + 1;
            }
        }
        
        return lo;
    }
};
