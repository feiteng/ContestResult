class Solution {
public:
    bool check(vector<int>& a,int x,int s){
        int ans=0;
        for(int& i:a){

            ans+=(i-1)/x;
        }
        return ans<=s;
    }
    int minimumSize(vector<int>& a, int x) {
        int l=0,r=1e9;
        while(l+1<r){
            int mid=(l+r)/2;
            if(check(a,mid,x)) r=mid;
            else l=mid;
        }
        return r;
    }
};
