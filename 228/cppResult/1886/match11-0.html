class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1;
        int r = 1000000000;
        
        int ret = INT_MAX;
        while (l < r) {
            int ops = 0;
            
            int m = l + (r - l) / 2;
            for (int v : nums) {
                ops += (v - 1) / m;
            }
            
            //cout << l << " " << r << " " << m << " " << ops << endl;
            
            if (ops <= maxOperations) {
                ret = min(ret, m);
                r = m;
            } else {
                l = m + 1;
            }
        }
        return ret;
    }
};
