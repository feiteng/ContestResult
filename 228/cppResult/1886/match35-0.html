class Solution {
public:
    int minimumSize(vector<int>& nums, int mx) {
        int l=1,r=*max_element(nums.begin(),nums.end());
        while (l<r) {
            int mid=l+(r-l)/2;
            int cnt=0;
            for (int i:nums) {
                if (i>mid) {
                    int tmp=i/mid;
                    if (i%mid==0) {
                        tmp--;
                    }
                    cnt+=tmp;
                    if (cnt>mx) {
                        break;
                    }
                }
            }
            if (cnt<=mx) {
                r=mid;
            } else {
                l=mid+1;
            }
        }
        return l;
    }
};
