class Solution {
public:
    

    bool test(int x, vector<int> &nums, int &maxOperations) {
        int req = 0;
        for (int i = 0; i < nums.size() && req <= maxOperations; i++) {
            if (nums[i] > x) {
                req += nums[i] / x - (nums[i] % x == 0 ? 1 : 0);
            }
        }
        return req <= maxOperations;
    }
    
    

    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1, r = *max_element(nums.begin(), nums.end()) + 1;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (test(m, nums, maxOperations)) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return l;
    }
};
