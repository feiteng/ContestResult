class Solution {
public:
bool check(vector<int>& a, int lim, int m){
    
    int use = 0;
    for(int& x : a) {
        use += (x - 1) / m;
    }
    // dbg("check ", m, (lim >= use));
    return lim >= use;
}
int minimumSize(vector<int>& a, int lim) {
    int l = 1, r = 1e9;
    while(l < r){
        int m = l + (r - l) / 2;
        if(check(a, lim, m)) r = m;
        else l = m + 1;
    }

    return l;
}

};
