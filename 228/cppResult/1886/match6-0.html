class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1, r = 1000000000;
        while (r > l){
            int mid = (l + r) >> 1;
            long long res = 0;
            for (int x: nums){
                res += (x - 1) / mid;
            }
            if (res <= 1ll * maxOperations)
                r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
