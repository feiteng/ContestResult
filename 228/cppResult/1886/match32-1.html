class Solution {
    bool op(vector<int>& a, int n, int o){
        int count = 0;
        for (int i = 0;i < a.size();++i){
            if (a[i] > n){
                count += (a[i]-1) / n;
                if (count > o) return false;

            }
        }
        return true;
    }
public:
    int minimumSize(vector<int>& nums, int o) {
        int l = 1, h = INT_MIN;
        for (int i = 0;i < nums.size();++i){
            h = std::max(h, nums[i]);
        }

        while (l <= h){
            int m = l + (h-l)/2;
            if (op(nums, m, o)){
                h = m-1;
            }
            else {
                l = m+1;
            }
        }
        return l;
    }
};
