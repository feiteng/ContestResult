class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1;
        int r = 1000*1000*1000+10;
        int res = r;
        while(l<=r) {
            int m = (l+r)/2;
            int v = 0;
            for (int i = 0; v<=maxOperations && i < nums.size(); ++i) {

                v += (nums[i]+m-1)/m - 1;
            }
            if (v<=maxOperations) {
                res = m;
                r=m-1;
            } else {
                l=m+1;
            } 
        }
        return res;
    }
};
