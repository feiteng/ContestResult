class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int n = nums.size();
        int l = 1, r = (int) 1e9, ans = -1;
        while (l <= r) {
            int mid = (r + l) / 2;
            
            int ops = 0;
            for (int i = 0; i < n; i++) {
                ops += nums[i] / mid;
                if (nums[i] % mid == 0) ops--;
            }
            
            if (ops > maxOperations) {
                l = mid+1;
            }
            else {
                ans = mid;
                r = mid-1;
            }
        }
        return ans;
    }
};
