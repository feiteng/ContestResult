class Solution {
public:
    
    bool check(vector<int>& nums, int mo, int mid){
        int cnt = 0;
        for(int x : nums){
            cnt += (x + mid - 1) / mid - 1;
            if(cnt > mo) return false;

        }
        return true;
    }
    
    int minimumSize(vector<int>& nums, int mo) {
        int l = 1, r = 1e9;
        while(l < r){
            int mid = l + r >> 1;
            if(check(nums, mo, mid)) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
