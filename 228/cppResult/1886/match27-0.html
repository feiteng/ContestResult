class Solution {
public:
    
    int aux(vector<int>& a, int g) {
        int res = 0;
        for(auto& x : a) {
            res += max((x - 1)/g, 0);
        }
        return res;
    }
    
    int minimumSize(vector<int>& nums, int maxo) {
        int min = 1;
        int max = *max_element(nums.begin(), nums.end());
        while(min < max) {
            int m = (min + max)/2;
            int c = aux(nums, m);
            if(c <= maxo) {
                max = m;
            }else {
                min = m + 1;
            }
        }
        return min;
    }
};
