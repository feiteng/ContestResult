class Solution {
public:
    bool ok(vector<int>&nums, int target, int maxO){
        int cnt = 0;
        for(auto v:nums){
            if(v<=target){
                //do nothing
            }else{
                int x = v/target;
                int y = v%target;
                if(y==0){
                    cnt+=x-1;
                }else{
                    cnt+=x;
                }
            }
        }
        return cnt<=maxO;
    }
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1;
        int r = 999999999;
        while(l<r){
            int mid = (l+r)/2;
            if(ok(nums, mid, maxOperations)){
                r = mid;
            }else{
                l = mid+1;
            }
        }
        return l;
    }
};
