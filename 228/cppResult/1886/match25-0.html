class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int p = 1;
        int r = *max_element(nums.begin(), nums.end());
        while (p < r) {
            int q = p + (r - p) / 2;
            int cnts = 0;
            for (int num : nums) {
                cnts += (num - 1) / q;
            }
            if (cnts > maxOperations) {
                p = q + 1;
            } else {
                r = q;
            }
        }
        return p;
    }
};
