class Solution {
public:
    bool canDivide(vector<int>& nums, int minimum, int maxOperations) {
        for (int i=0; i<nums.size(); i++) {
            if (nums[i] > minimum) {
                int neededOp = (nums[i] - 1) / minimum;
                maxOperations -= neededOp;
            }
        }
        
        return maxOperations >= 0;
    }
    
    int minimumSize(vector<int>& nums, int maxOperations) {
        int lo = 1, hi = 1e9;
        
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            
            if (canDivide(nums, mid, maxOperations)) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        
        return lo;
    }
};
