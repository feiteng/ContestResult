class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int left = 1, right = nums[n-1];
        while(left < right){
            int mid = (left+right)/2;

            int count = 0;
            for(int i = n-1;i>=0 && nums[i]>mid;i--){
                count += ceil(1.0*nums[i]/mid)-1;
            }
            if(count > maxOperations){
                left = mid+1;
            }else{
                right = mid;
            }
        }
        return left;
    }
};
