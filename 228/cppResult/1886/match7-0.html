class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l = 1, r = 1e9 + 5;
        while (l < r) {
            int m = (l + r) / 2;
            long long ops = 0;
            for (int x : nums)
                ops += (x + m - 1) / m - 1;
            if (ops <= maxOperations)
                r = m;
            else
                l = m + 1;
        }
        return l;
    }
};
