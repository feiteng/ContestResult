typedef long long LL;
class Solution {
public:
    vector<int> v;
    int cnt;
    bool Check(int t)
    {
        LL ret = 0;
        for (auto x : v)
        {
            ret += x / t + (x % t != 0) - 1;
        }
        return ret <= cnt;
    }
    int minimumSize(vector<int>& nums, int maxOperations) {
        v = nums, cnt = maxOperations;
        int l = 1, r = v[0], mid;
        for (auto x : v)    
            r = max(r, x);
        
        while (l < r)
        {
            mid = l + r >> 1;
            if (Check(mid))
            {
                r = mid;
            }
            else
            {
                l = mid + 1;
            }
        }
        return l;       
    }
};
