class Solution {
public:
    int minimumSize(vector<int>& nums, int mO) {
        int n=nums.size();
        int cnt=n+mO;
        int l=1,r=*max_element(nums.begin(),nums.end());

        while(l<=r){
            int mid=l+r>>1;
            int op=0;
            for(int i=0;i<n;i++){
                op+=nums[i]/mid;
                if(nums[i]%mid==0)op--;

            }
            if(op<=mO){
                r=mid-1;
            }
            else{
                l=mid+1;
            }
        }
        return l;
    }
};
