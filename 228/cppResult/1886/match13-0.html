class Solution {
public:
    int check(vector<int>& nums, int mid){
        int cnt = 0;
        for(int num: nums){
            int t = num / mid;
            if(num % mid == 0) cnt += t - 1;
            else cnt += t;
        }
        return cnt;
    }
    int minimumSize(vector<int>& nums, int ops) {
        int l = 1, r = 1e9 + 10;
        // cout << check(nums, 3) << endl;
        while(l < r){
            int mid = l + r >> 1;
            // cout << mid << " " << check(nums, mid) << endl;
            if(check(nums, mid) <= ops) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
