class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int l=1,r=1000000000;
        while(l<r)
        {
            int m = (l+r)/2;
            long long op = 0;
            for(int i=0;i<nums.size();i++)
            {
                op+=((nums[i]-1)/m);
            }
            if(op<=maxOperations)
            {
                r=m;
            }
            else
            {
                l=m+1;
            }
        }
        return r;
    }
};
