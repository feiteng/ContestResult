class Solution {
public:
    
    bool check(int mid,vector<int>& nums,int O){
        int cnt = 0;
        for(auto x : nums){
            if(x % mid == 0) cnt += (x / mid) - 1;
            else cnt += (x / mid);
        }
        return cnt <= O;
    }
    
    int minimumSize(vector<int>& nums, int maxOperations) {
        int r = *max_element(nums.begin(),nums.end());
        int l = 1;
        while(l < r){
            int mid = r + l >> 1;
            if(check(mid,nums,maxOperations)){
                r = mid;
            }else{
                l = mid + 1;
            }
        }
        return l;
    }
};
