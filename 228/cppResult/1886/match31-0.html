class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int n = nums.size();
        int l = 1, r = INT_MIN;
        for (auto num : nums) {
            l = min(num, l);
            r = max(r, num);
        }
        
        while (l < r) {
            int mid = l + r >> 1;
            int tol_op = 0;
            for (int i = 0; i < n; i ++) {
                if (nums[i] > mid) {
                    int op = nums[i] / mid;
                    if (nums[i] % mid == 0) op --;
                    tol_op += op;
                }
            }
            if (tol_op > maxOperations) {
                l = mid + 1;
            }  else {
                r = mid;
            }
        }
        return l;
    }
};
