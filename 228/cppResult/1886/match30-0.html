class Solution {
public:
    int minimumSize(vector<int>& v, int mo) {
        int l = 1, r = *max_element(v.begin(), v.end());
        while(l < r) {
            int m = (l+r)/2;
            int cur = 0;
            for(int i: v) {
                cur += (i-1)/m;
            }
            if(cur > mo) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }
};
