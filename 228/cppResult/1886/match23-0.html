class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        int n=nums.size();
        int l=0;
        int r=0;
        for (int i=0;i<n;i++) r=max(r,nums[i]);
        while (l+1<r) {
            int m=(l+r)/2;
            int cost=0;
            for (int j=0;j<n;j++){
                cost+=(nums[j]+m-1)/m-1;
            }
            if (cost>maxOperations){
                l=m;
            }else{
                r=m;
            }
        }
        return r;
        
    }
};
