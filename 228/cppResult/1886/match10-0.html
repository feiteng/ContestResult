class Solution {
public:
    int minimumSize(vector<int>& nums, int maxOperations) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int lo = 1, hi = nums[n - 1];
        while(lo < hi) {
            int mid = lo + (hi - lo) / 2;
            int lcnt = 0;
            for(int i = 0; i < n; i++) {
                if(nums[i] > mid) {
                    lcnt = lcnt + (nums[i] - 1) / mid;
                }
            }
            if(lcnt > maxOperations) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        return lo;
    }
};
