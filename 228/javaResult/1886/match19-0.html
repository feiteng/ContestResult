class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = 0;
        for (int n : nums) r = Math.max(r, n);
        while (l < r) {
            int mid = (l + r) / 2;
            if (find(nums, maxOperations, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
    private boolean find(int[] nums, int maxOperations, int max) {
        int count = 0;
        for (int n : nums) {
            if (n > max) {
                count += Math.max(n - 1, 0) / max;
                if (count > maxOperations) return false;
            } 
        } 
        return true;
    }
}
