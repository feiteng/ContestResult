class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int len = nums.length;
        int max = 0;
        for(int i = 0; i < len; i++){
            max = Math.max(max, nums[i]);
        }
        int left = 1;
        
int right = max;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (visible(nums, maxOperations, mid)) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
    private static boolean visible(int[] arr, int totalIntervals, int targetGap) {
        int n = arr.length, i = 0, curGap = 0, usedIntervals = 0;
        for (i = 0; i < n; ++i) {
            curGap = arr[i];
            usedIntervals += curGap / targetGap;
            
if (curGap % targetGap == 0) {
                --usedIntervals;
            }
        }
        return usedIntervals <= totalIntervals;
    }

}
