class Solution {
    public int minimumSize(int[] nums, int maxOperations)
	{
		int low = 1;
		int high = 1000000000;
		while(low < high)
		{
			int mid = (low + high) / 2;
			if(check(nums, maxOperations, mid))
			{
				high = mid;
			}
			else
			{
				low = mid + 1;
			}
		}
		
		return low;
	}
	
	boolean check(int[] nums, int ops, int mid)
	{
		long result = 0;
		for(int x = 0; x < nums.length; x++)
		{
			result += (nums[x] - 1) / mid;
		}
		
		return result <= ops;
	}
}
