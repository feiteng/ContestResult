class Solution {
    boolean check(int[] nums, int maxOperations, int max) {
        int opts = 0;
        for(int i = 0; i<nums.length; i++) {
            if(nums[i] > max) {
                opts += Math.ceil((double) (nums[i]-max)/max);
            
}
        }
        return opts <= maxOperations;
    }

    public int minimumSize(int[] nums, int maxOperations) {
        
int l = 1 ,r = (int)1e9;
        while (l < r) {
            int mid = (l+r)/2;
            if(check(nums,maxOperations,mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
}
