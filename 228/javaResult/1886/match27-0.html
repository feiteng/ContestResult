class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        long l = 1;
        long r = 1000000000;
        while(l <= r) {
            long m = (l + r) / 2;
            if(cS(nums, maxOperations, m)) {
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return (int) l;
    }
    boolean cS(int[] A, int t, long y) {
        long cnt = 0;
        for(int x : A) {
            if(x <= y) continue;
            cnt += x / y;
            if(x % y == 0) --cnt;
            if(cnt > t) return false;
        }
        return true;
    }
}
