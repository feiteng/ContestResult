class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, h = (int)1e9;
        while (l < h) {
            int m = (l + h) / 2;
            if (isPossible(nums, m, maxOperations)) {
                h = m;
            } else {
                l = m + 1;
            }
        }
        return l;
    }
    
    public boolean isPossible(int[] nums, int m, int mO) {
        int used = 0;
        for (int i : nums) {
            if (i > m) {
                if (i % m == 0) used += i / m - 1;
                else used += i / m;
            }
        }
        return used <= mO;
    }
}
