class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1;
        int r = Integer.MAX_VALUE;
        while (l < r) {
            int m = l + (r - l) / 2;
            int cnt = count(nums, m);
            if (cnt > maxOperations) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }
    
    int count(int[] nums, int k) {
        int count = 0;
        for (int x : nums) {
            count += (x - 1) / k;
            // x <= k -> 0
            // x = k + 1 -> 1
            // x == 2k -> 1
            // x == 2k + 1. -> 2
        }
        return count;
    }
}
