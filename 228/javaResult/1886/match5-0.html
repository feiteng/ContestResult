class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int left = 1;
        int right = 1000000000;
        while (left < right) {
            int mid = left + (right - left) / 2;
            boolean okay = check(nums, mid, maxOperations);
            if (okay) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
        
    }
    private boolean check(int[] nums, int target, int maxOperation) {
        int count = 0;
        for (int num : nums) {
            if (num > target) {
                int remainder = num % target == 0 ? 0 : 1;
                count += (num / target + remainder) - 1;
            }            
        }
        return count <= maxOperation;
    }
}
