class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
             int l = 1; 
             int r = 0;
             for (int num : nums) {
                 r = Math.max(r, num);
             }
             while (l < r) {
                 int mid = (l + r) / 2;
                 if (requiredOperations(nums, mid) <= maxOperations) {
                     r = mid;
                 } else {
                     l = mid + 1;
                 }
             }
             return l;
        }
        
        private int requiredOperations(int[] nums, int target) 
{
            int result = 0;
            for (int num : nums) {
                if (num > target) {
                    result += num / target;
                    if (num % target == 0) {
                        result--;
                    }
                }
            }
            return result;
        }
}
