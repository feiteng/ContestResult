class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int max = 0;
        for (int i : nums) {
            max = Math.max(max, i);
        }
        
int start = 1, end = max;
        while (start + 1 < end) {
            int mid = start + (end - start) / 2;
            if (canDo(nums, maxOperations, mid)) {
                end = mid;
            } else {
                start = mid;
            }
        }
        if (canDo(nums, maxOperations, start)) {
            return start;
        } else {
            return end;
        }
    
}
    
    private boolean canDo(int[] nums, int ops, int target) {
        for (int i : nums) {
            if (i <= target) {
                continue;
            }
            if (i % target == 0) {
                ops -= i / target - 1;
            } else {
                ops -= i / target;
            }
            
if (ops < 0) {
                return false;
            }
        }
        return true;
    }
}
