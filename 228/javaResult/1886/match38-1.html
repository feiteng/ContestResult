import java.util.Arrays;


class Solution {
    public boolean check(int mid, int[] nums, int maxOperations) {
        int cur = maxOperations;
        for (int i : nums) {
            if (i > mid) {
                if (i % mid == 0) {
                    cur -= (i / mid) - 1;
                } else {
                    cur -= i / mid;
                }

                if (cur < 0) {
                    return false;
                }
            
}
        }
        return true;
    }

    public int minimumSize(int[] nums, int maxOperations) {
        Arrays.sort(nums);
        
int l = 1;
        int r = (int) 1e9;
        while (l < r) {
            int mid = (l + r) / 2;
            if (check(mid, nums, maxOperations)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
}
