class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int left = 1;
        int right = 1;
        for(int num: nums) {
            right = Math.max(num, right);
        }
        while(left < right) {
            int mid = (left + right) / 2;
            if(canSplit(mid, nums, maxOperations)) {
                right = mid;
            }
            else {
                left = mid + 1;
            }
        } 
        return left;
        
    }
    
    public boolean canSplit(int mid, int[] nums, int maxOperations) {
        int sum = 0;
        for(int i = 0; i < nums.length; i++) {
            if(nums[i] % mid == 0) {
                sum = sum + nums[i] / mid - 1;
            }
            else {
                sum = sum + nums[i] / mid;
            }
            if(sum > maxOperations)
                return false;
        }
        return true;
    }
}
