class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        Arrays.sort(nums);
        int lb = 1, hb = nums[nums.length-1];
        while(lb < hb){
            int res = 0, mb = (lb+hb)/2;
            for(int i = 0; i < nums.length; ++i){
                res += (nums[i]+mb-1)/mb - 1;
            
}
            
            if(res <= maxOperations){
                hb = mb;
            }
            else{
                lb = mb+1;
            }
        }
        return lb;
    }
}
