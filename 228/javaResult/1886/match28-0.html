class Solution {
    
    private boolean check(int[] nums, int m, int p) {
        int cnt = 0;
        for (int i = nums.length - 1; i >= 0; --i) {
            if (nums[i] <= m) break;
            cnt += (nums[i] - 1) / m;
            if (cnt > p) return false;
        }
        return cnt <= p;
    }
    public int minimumSize(int[] nums, int p) {
        Arrays.sort(nums);
        int left = 1, right = nums[nums.length - 1];
        while (left < right) {
            int mid = (left + right) / 2;
            if (check(nums, mid, p)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
}
