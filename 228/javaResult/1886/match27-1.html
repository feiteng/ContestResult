class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l=1;
        int h=Integer.MAX_VALUE;
        while(l<h){
            int mid=(l)+(h-l)/2;
            if(check(mid,nums,maxOperations)){
                h=mid;
            }else{
                l=mid+1;
            }
        }
        return l;
    }
