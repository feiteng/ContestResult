class Solution {
    public int minimumSize(int[] nums, int k) {
        int p = 1, q = 1000000000, ans = 0;
        while (p < q) {
            int mid = (p + q) / 2, cnt = 0;
            for (int x : nums)
                cnt += (x % mid == 0 ? x / mid : x / mid + 1) - 1;
            if (cnt > k)
                p = mid + 1;
            else
                q = mid;
        }
        return p;
    }
}
