class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int left = 1;
        int right = Integer.MIN_VALUE;
        for (int num: nums) {
            right = Math.max(right, num);
        }
        while (left  < right) {
            int mid = left + (right - left) / 2;
            int v = helper(nums, mid);
           
            if (v <= maxOperations) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
        // if (helper(nums, right) <= maxOperations) {
        //     return right;
        // } else {
        //     return left;
        // }
    }
    int helper(int[] nums, int target) {
        int sum = 0;
        for (int num: nums) {
            if (num > target) {
                sum += num / target;
                if (num % target == 0) {
                    sum --;
                }
            }
        }
        // System.out.println(target + "," + sum);
        return sum;
    }
}
