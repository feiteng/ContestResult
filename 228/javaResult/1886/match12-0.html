class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        // 二分猜测
        int n = nums.length;
        int left = 1, right = (int)(1e9);
        while (left < right) {
            int guess = (left + right) >> 1;
            if (check(nums, n, guess, maxOperations)) {
                right = guess;
            } else {
                left = guess + 1;
            }
        }
        return left;
    }
    
    private boolean check(int[] nums, int n, int guess, int maxOperations) {
        int op = 0;
        for (int i = 0; i < n; i++) {
            op += (nums[i] - 1) / guess;
        }
        return op <= maxOperations;
    }
}
