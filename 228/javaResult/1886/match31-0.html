class Solution {
    public long getTimes(int[] nums,int pre){
        long res=0;
        for(int num:nums){
           int add=num/pre; 
           if(num%pre!=0){
               add++;
           }
            res+=add;
        }
        return res;
    }
    public int minimumSize(int[] nums, int maxOperations) {
        int n=nums.length;
        long sum=0;
        long max=0;
        for(int num:nums){
            sum+=num;
            max=Math.max(max,num);
        }
        long per=sum/(n+maxOperations);
        if(sum%(n+maxOperations)!=0){
            per++;
        }
        //System.out.println(per);
        int left=(int) per,right=(int)max; 
        int ans=0;
        while(left<=right){
            int mid=(left+right)/2;
            if(getTimes(nums,mid)<=maxOperations+n){
                 ans=mid;
                 right=mid-1;
            }else{
                 left=mid+1;
            }
        }
        return ans;
    }
}
