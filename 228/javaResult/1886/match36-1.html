class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        long lo = 1;
        long hi = 1000000000000000000L;
        while(lo<hi){
            long mi =  (lo+hi)/2L;
            long r = 0;
            for(int nu:nums){
                long m = (nu+mi-1)/mi-1;
                r += m;
            
}
            if(r>maxOperations){
                lo = mi+1;
            }else{
                hi = mi;
            }
            
            
            
        }
        return (int)lo;
        
    }
}
