class Solution {
    public int minimumSize(int[] A, int k) {
        Arrays.sort(A);
        int l=1;
        int r=A[A.length-1];
        int res=-1;
        while(l<=r){
            int mid=l+(r-l)/2;
            //System.out.println(l+"  "+r+"  "+mid);
            if(check(A,mid,k)){
                res=mid;
                
r=mid-1;
            }
            else{
                l=mid+1;
            }
        }
        return res;
    }
    
    public boolean check(int A[],int mid,int k){
        long sum=0;
        for(int i=0;i<A.length;i++){
            int cnt=A[i]/mid;
            if(A[i]%mid==0){
                sum+=(cnt-1);
            }
            else
{
                sum+=(cnt);
            }
            
        }
        //System.out.println(sum);
        return sum<=k;
    }
}
