class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        Arrays.sort(nums);
        long low = 1;
        long high = nums[nums.length - 1];
        while (low < high) {
            long mid = (low + high) / 2;
            long ops = 0;
            for (int i : nums) {
                ops += (long) Math.ceil(((double) i) / mid) - 1;
            
}
            if (ops > maxOperations) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return (int) low;
    }
}
