class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        Arrays.sort(nums);
        int left = 1;
        int right = nums[nums.length - 1];
        while (left < right) {
            int mid = (left + right) / 2;
            int k = maxOperations;
            for (int i = nums.length - 1; i >= 0; i--) {
                if (nums[i] > mid) {
                    k -= (nums[i] - 1) / mid;
                    if (k < 0) {
                        break;
                    }
                } else {
                    break;
                }
            }
            if (k >= 0) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
}
