class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int n = nums.length;
        int max= 0;
        for (int i = 0;i<nums.length;i++){
            max = Math.max(max, nums[i]);
        }
        if (max <=1){
            return max;
        }
        int l = 1; 
int r = 1000000000;
        while (l<r){
            int mid = (l+r)>>1;
            if (canI(nums,maxOperations,mid)){
                r = mid;
            }else{
                l = mid+1;
            }
        }
        
        return l;
    }
    public boolean canI(int[] nums, int o, int top){
        int d = 0;
        for (int i = 0;i<nums.length;i++){
            if (nums[i]>top){
                d+= (nums[i]-1)/top ;
            }
        }
        return d<=o;
    }
}
