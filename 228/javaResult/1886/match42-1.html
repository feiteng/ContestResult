class Solution {
    public int minimumSize(int[] nums, int mops) {
        int n = nums.length;
        int min = 1;
        int max = 1000000000;
        while (min<max) {
            int mid = (min+max)/2;
            long ops = 0;
            for (int i = 0; i < n; ++i) {
                int newops = nums[i]/mid;
                if (nums[i]%mid==0)
                    newops--;
                ops += newops;
            
}
            if (ops<=mops)
                max = mid;
            else
                min = mid+1; 
        }
        return min;
    }
}
