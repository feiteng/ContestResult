class Solution {
    public int minimumSize(int[] nums, int op) {
        int l = 1;
        int r = (int)1e9;
        int res = 0;
        while(l<=r){
            int mid = (l+r)/2;
            boolean ans = check(nums, op, mid);
            if(ans==true){
                res = mid;
                r = mid-1;
            }else{
                l = mid+1;
            }
        }
        return res;
    }
    
    static int max = 0 ;
    public static boolean check(int nums[], int op, int mid){
        int cnt = 0;
        int n = nums.length;
        for(int i=0;i<n;i++){
            int req = (nums[i]%mid==0 ? nums[i]/mid : (nums[i]/mid+1));
            if(nums[i]<mid){
                max = Math.max(max, nums[i]);
            }else{
                max = mid;
            }
            req-=1;
            cnt+=req;
        }
        if(cnt<=op){
            return true;
        }
        return false;
    }
}
