class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = 1000000000, mid, res=0;
        while(l<r) {
            mid = (l+r)/2;
            if(check(nums, maxOperations, mid)) {
                r = mid;
                res = mid;
            }else{
                l = mid+1;
            }
        }
        return res;
    }
    private boolean check(int[] nums, int maxOperations, int cost) {
        int op = 0;
        for(int num: nums) {
            op += (int)(Math.ceil((double)num/cost))-1;
        }
        return op <= maxOperations;
    }
}
