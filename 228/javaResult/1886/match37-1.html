class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int s = 1, e = (int) 1e9;
        
        while (s + 1 < e) {
            int m = (e - s) / 2 + s;
            
            if (canCut(nums, m, maxOperations)) {
                e = m;
            } else {
                s = m;
            }
        }
        
        return canCut(nums, s, maxOperations) ? s : e;
    
}
    
    private boolean canCut(int[] nums, int val, int max) {
        int cnt = 0;
        
        for (int n : nums) {
            if (n > val) {
                if (n % val == 0) {
                    cnt += (n / val - 1);
                } else {
                    cnt += n / val;
                }
                if (cnt > max) {
                    return false;
                
}
            }
        }
        
        return true;
    }
}
