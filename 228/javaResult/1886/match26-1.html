class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int lo = 1, hi = 1_000_000_000;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            long ops = minOperations(nums, mid);
            if (ops > maxOperations) 
{
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        return lo;
    }
    private long minOperations(int[] nums, int maxSize) 
{
        long ops = 0;
        for (int num : nums) {
            ops += (num - 1) / maxSize;
        }
        return ops;
    }
}
