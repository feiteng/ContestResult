class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1;
        int r = nums[0];
        for (int i = 0; i < nums.length; i++) {
            r = Math.max(nums[i], r);
        }
        
        while (l < r) {
            int mid = l + (r - l) / 2;
            int value = getValue(nums, mid);
            if (value <= maxOperations) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return r;
    }
    
    public int getValue(int[] nums, int mid) {
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            int count = nums[i] / mid;
            if (nums[i] % mid != 0) 
{
                count++;
            }
            count--;
            res += count;
        }
        return res;
    }
}
