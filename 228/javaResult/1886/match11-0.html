class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int low = 0;
        int high = (int)1e9 + 1;

        while (low + 1 < high) {
            int mid = low + high >> 1;
            int ops = 0;
            for (int num : nums) {
                ops += (num + mid - 1) / mid - 1;
            }
            if (ops <= maxOperations) high = mid;
            else low = mid;
        }

        return high;
    }
}
