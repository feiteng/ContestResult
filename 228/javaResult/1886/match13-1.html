class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
         int mx = 0, mn = Integer.MAX_VALUE;
        for(int x : nums){
            mx = Math.max(mx, x);
        }
        int lo = 1, hi = mx;
        
while(lo < hi){
            int mid = lo+(hi-lo)/2;
            int k = helper(nums, mid);
            if(k <= maxOperations ){
                hi = mid;
            }else{
                lo = mid+1;
            }

        }
        return lo;
    }
    int helper(int[] nums, int mid){
        int cnt = 0;
        for(int x : nums){
            
if(x > mid){
                cnt += (x + mid-1)/mid-1;
            }

        }
        return cnt;
    }
}
