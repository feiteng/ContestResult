class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        Arrays.sort(nums);
        int n = nums.length;
        int r = nums[n-1];
        int l = 1;
        while(l<r) {
            int m = l+(r-l)/2;  
            if(check(nums,maxOperations,m)) {
                r = m;
            } else {
                l = m+1;
            }
        }
        return l;
    }
    boolean check(int[] nums,int op,int n) {
        int count = 0;
        for(int i=0;i<nums.length;i++) {
            if(nums[i]>n) {
                count+= ((nums[i]-1)/n);
            }
            // if(n==6) System.out.println(String.format("%d,%d",nums[i],count));
        }
        // System.out.println(String.format("%d,%d,%d",op,n,count));
        return count<=op;
    }
    
}
