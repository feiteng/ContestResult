class Solution {
    public int minimumSize(int[] a, int m) {
        int n = a.length;
        
        int low = 1;
        int high = (int)1e9;
        int mid = 0;
        
        while(low<high) {
            mid = low+(high-low)/2;
            
            boolean x = get(mid, a, m);
            
            if (x) {
                
high = mid;
            } else {
                low = mid+1;
            }
        }
        
        return low;
    }
    
    
    boolean get(int max, int[] a, int m) {
        int req = 0;
        int n = a.length;
        
for(int i=0; i<n; i++) {
            req+= (int)Math.ceil((double)a[i]/max)-1;
        }
        
        return req<=m;
    }
}
