class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int left = 1, right = 0;
        for (int i : nums) {
            right = Math.max(right, i);
        }
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (itCan(nums, maxOperations, mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
    
    public boolean itCan(int[] nums, int maxOperations, int mid) {
        int sum = 0;
        for (int i : nums) {
            if (i % mid == 0) {
                sum--;
            
}
            sum += i / mid;
            if (sum > maxOperations) return false;
        }
        return true;
    }
}
