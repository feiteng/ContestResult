class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        return sln1(nums,maxOperations );
    }
    
    private int sln1(int[] nums, int m){
        int l = 0; int r=1_000_000_000;
        int res = 0;
        while(l<=r){
            int mid = l+r>>1;
            if(helper(nums, mid) > m){
                l = mid+1;
            }else{
                res = mid;
                r = mid-1;
            }
        }
        return res;
    }
    
    private int helper(int[] nums, int k){
        if(k==0) return Integer.MAX_VALUE;
        int count = 0;
        for(int i: nums){
            count+= (i-1)/k;
        }
        return count;
    }
}
