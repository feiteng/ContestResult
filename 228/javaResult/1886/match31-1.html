class Solution {
    public long f(int[] nums, int r) {
        long ans = 0;
        for (int a : nums) {
            ans += (a + r - 1) / r - 1;
        
}
        return ans;
    }
    public int minimumSize(int[] nums, int maxOperations) {
        int lo = 1;
        
int hi = 1_000_000_000;
        int ans = hi;
        long target = maxOperations;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            // System.out.printf("mid : %d f: %d\n", mid, f(nums, mid));
            if (f(nums, mid) <= target) {
                ans = mid;
                hi = mid - 1;
            } else {
                lo = mid + 1;
            }
        }
        return ans;
    }
}
