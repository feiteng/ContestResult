class Solution {

	public int minimumSize(int[] nums, int maxOperations) {
		int left = 1, right = 1000000000;
		while (left < right) {
			int mid = (left + right) / 2, count = 0;
			for (int num : nums) {
				count += (num + mid - 1) / mid - 1;
			}

			if (count <= maxOperations) {
				right = mid;
			} else {
				left = mid + 1;
			}
		}
		return left;
	}
}
