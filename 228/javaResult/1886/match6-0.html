class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = 1_000_000_000;
        while (l < r) {
            int m = (l + r) / 2, operations = maxOperations;
            for (int num : nums) {
                if (num > m) {
                    operations -= (num - 1) / m;    // 5 6 7 8 -> 1,   9 10 11 12 -> 2
                    if (operations < 0) {
                        break;
                    }
                }
            }
            if (operations >= 0) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return r;
    }
}
