class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = 1_000_000_010;
        while (l < r) {
            int m = (l + r) / 2, ops = 0;
            for (int i = 0; i < nums.length; i++) {
                ops += Math.ceil(nums[i] * 1.0 / m) - 1;
            
}
            if (ops <= maxOperations) r = m;
            else l = m + 1;
        }
        return r;
    }
}
