class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int end = 0;
        for (int num : nums) {
            end = Math.max(end, num);
        }
        int start = 1;
        while (start < end) {
            int mid = start + (end - start) / 2;
            if (isPossible(nums, maxOperations, mid)) {
                end = mid;
            } else {
                start = mid + 1;
            }
        }
        return start;
    }
    private boolean isPossible(int[] nums, int maxOperations, int thresh) {
        for (int num : nums) {
            if (num <= thresh) {
                continue;
            }
            int used = (num + thresh - 1) / thresh - 1;
            maxOperations -= used;
            if (maxOperations < 0) {
                return false;
            }
        }
        return true;
    }
}
