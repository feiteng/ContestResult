class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = Integer.MAX_VALUE;
        while(l < r) {
            int mid = l + (r - l) / 2;
            if(cutValid(nums, maxOperations, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
    
    public boolean cutValid(int[] nums, int maxOp, int max) {
        int count = 0;
        for(int i: nums) {
            if(i > max) {
                if(i % max == 0) {
                    count += i / max - 1;
                } else {
                    count += i / max;
                
}
            }
        }
        return count <= maxOp;
    }
}
