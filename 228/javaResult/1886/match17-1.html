class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int left = 1, right = 0;
        for (int i : nums) right = Math.max(right, i)
;
        while (left < right) {
            int mid = left + (right - left) / 2;
            // if count == max  move left right = mid
            // if count > max   mid is small we spend too much split   so move left  left = mid + 1
            // if count < max    mid is too large we spend a little split so move right  we can spend more times  right = mid;
            if (helper(nums, maxOperations, mid)) {  // count <= max
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
    
    public boolean helper(int[] nums, int max, int mid) {
        int count = 0;
        for (int i : nums) {
            int temp = 0;
            
if (i % mid == 0) {
                temp = i / mid - 1;
            } else {
                temp = i / mid;
            }
            count += temp;
            if (count > max) return false;
        }
        return true;
    }
}
