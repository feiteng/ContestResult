class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int lo = 1;
        int hi = (int)1e9;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            int op = 0;
            for (int i = 0; i < nums.length; i++) {
                op += (nums[i] - 1) / mid;
                if (op > maxOperations) break;
            
}
            if (op <= maxOperations) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }
}
