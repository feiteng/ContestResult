class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int l = 1, r = 0;
        for (int num : nums) {
            r = Math.max(r, num);
        }
        while (l < r) {
            int mid = l + (r - l) / 2;
            int res = getResult(nums, mid);
            if (res <= maxOperations) 
{
                r = mid;
            }
            else {
                l = mid + 1;
            }
        }
        return l;
    }
    
    private int getResult(int[] nums, int target) {
        int res = 0;
        for (int num : nums) {
            if (num % target == 0) {
                res += num / target - 1;
            
}
            else {
                res += num / target;
            }
        }
        return res;
    }
}
