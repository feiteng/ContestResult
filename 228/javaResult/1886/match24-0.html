class Solution {
    public int minimumSize(int[] nums, int maxOperations) {
        int sum = 0;
        int imax = 0;
        for(int a: nums) {
            sum += a;
            imax = Math.max(imax, a);
        }
        
        int left = Math.max(1, sum / (nums.length + maxOperations));
        int right = imax;
        while(left < right){
            int mid = left + (right-left)/2;
            if(isValid(nums, maxOperations, mid)){
                right = mid;
            }
            else{
                left = mid + 1;
            }
        }
        return left;
    }
    
    public boolean isValid(int[] nums, int maxOperations, int target){
        int ops = 0;
        for(int a: nums){
            if(a>target){
                ops += (a-1)/target;
            }
        }
        return ops <= maxOperations;
    }
}
