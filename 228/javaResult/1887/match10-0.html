class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] graph = new boolean[n + 1][n + 1];
        int[] edgeCount = new int[n + 1];
        for (int[] e : edges) {
            graph[e[0]][e[1]] = graph[e[1]][e[0]] = true;
            edgeCount[e[0]]++;
            edgeCount[e[1]]++;
        }
        int ans = Integer.MAX_VALUE;
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                for (int k = j + 1; k <= n; k++) {
                    if (graph[i][j] && graph[j][k] && graph[i][k]) {
                        ans = Math.min(ans, edgeCount[i] - 2 + edgeCount[j] - 2 + edgeCount[k] - 2);
                    }
                }
            }
        }
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
}
