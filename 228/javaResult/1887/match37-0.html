class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        int[][] ed = new int[n + 1][n + 1];
        int[] count = new int[n + 1];
        for (int[] e : edges) {
            ed[e[0]][e[1]] = ed[e[1]][e[0]] = 1;
            count[e[0]]++;
            count[e[1]]++;
        }
        int res = -1;
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                for (int k = j + 1; k <=n ; k++) {
                    if (ed[i][j] == 1 && ed[i][k] == 1 && ed[j][k] == 1) {
                        if (res == -1 || res > count[i] + count[j] + count[k] - 6) {
                            res = count[i] + count[j] + count[k] - 6;
                        }
                    }
                }
            }
        }
        return res;
    }
}
