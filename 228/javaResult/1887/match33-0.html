class Solution {
    public int minTrioDegree(int N, int[][] edges) {
        boolean[][] m = new boolean[N][N];
        int[] d = new int[N];
        
        for (int i = 0; i < edges.length; i++) {
            int u = edges[i][0] - 1, v = edges[i][1] - 1;
            m[u][v] = m[v][u] = true;
            d[u]++;
            d[v]++;
        }
        
        int total = 1000000000;
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                if (m[i][j]) {
                    for (int k = j + 1; k < N; k++) {
                        if (m[i][k] && m[j][k]) {
                            total = Math.min(total, d[i] + d[j] + d[k] - 6);
                        }
                    }
                }
            }
        }
        if (total == 1000000000) {
            return -1;
        }
        
        return total;
    }
}
