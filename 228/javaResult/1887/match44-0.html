class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] top = new boolean[n+5][n+5];
        int[] val = new int[n+5];
        for(int i=0;i<edges.length;i++){
            int x = edges[i][0],y = edges[i][1];
            top[x][y] = true;
            top[y][x] = true;
            val[x]++;
            val[y]++;
        }
        int ans = (int) 1e9;
        for(int i=1;i<=n;i++){
            for(int j=i+1;j<=n;j++){
                for (int k=j+1;k<=n;k++){
                    if(top[i][j]&&top[i][k]&&top[j][k]){
                        ans = Math.min(ans,val[i]+val[j]+val[k]-6);
                    }
                }
            }
        }
        return (ans == 1e9)?-1:ans;
    }
}
