class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] flag = new boolean[n][n];
        int[] counter = new int[n];
        for (int[] edge: edges) {
            flag[edge[0] - 1][edge[1] - 1] = true;
            flag[edge[1] - 1][edge[0] - 1] = true;
            counter[edge[0] - 1]++;
            counter[edge[1] - 1]++;
        }
        int ret = 1000000000;
        for (int i = 0;i < n;i++) {
            for (int j = i + 1;j < n;j++) {
                for (int k = j + 1;k < n;k++) {
                    if (flag[i][j] && flag[i][k] && flag[j][k]) {
                        ret = Math.min(ret, counter[i] + counter[j] + counter[k] - 6);
                    }
                }
            }
        }
        if (ret == 1000000000) {
            return -1;
        }
        return ret;
    }
}
