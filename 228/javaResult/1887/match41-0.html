class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        int[] degree  = new int[n+1];
        boolean[][] cc = new boolean[n+1][n+1];
        
        for(int[] e:edges){
            cc[e[0]][e[1]] = true;
            cc[e[1]][e[0]] = true;
            degree[e[0]]++;
            degree[e[1]]++;
        }
        int min = Integer.MAX_VALUE;
        for(int i = 1; i <= n; ++i){
            for(int j = i+1; j <= n; ++j){
                for(int k = j+1; k <= n; ++k){
                    if(cc[i][j] && cc[j][k] && cc[i][k]){
                        min = Math.min(min, degree[i] + degree[j] + degree[k] - 6);
                    }
                }
            }
        }
        
        if(min == Integer.MAX_VALUE){
            return -1;
        }
        
        return min;
        
    }
}
