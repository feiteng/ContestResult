class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        List<Set<Integer>> g = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            g.add(new HashSet<Integer>());
        }
        for (int[] e : edges) {
            g.get(e[0]-1).add(e[1]-1);
            g.get(e[1]-1).add(e[0]-1);
        }
        
        int res = Integer.MAX_VALUE / 3;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                for (int k = j + 1; k < n; ++k) {
                    if (g.get(i).contains(j) && g.get(i).contains(k) && g.get(j).contains(k)) {
                        res = Math.min(res, g.get(i).size() + g.get(j).size() + g.get(k).size() - 6);
                    }
                }
            }
        }
        
        return res == Integer.MAX_VALUE / 3? -1 : res;
    }
}
