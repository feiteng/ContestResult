class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        int graph[][] = new int[n+1][n+1];
        int degree[] = new int[n+1];
        for(int i=0;i<edges.length;i++){
            int u = edges[i][0];
            int v = edges[i][1];
            degree[u]++;
            degree[v]++;
            graph[u][v]=1;
            graph[v][u]=1;
        }
        int ans = Integer.MAX_VALUE;
        for(int i=1;i<=n;i++){
            for(int j=i+1;j<=n;j++){
                for(int k=j+1;k<=n;k++){
                    if(graph[i][j]==1 && graph[i][k]==1 && graph[j][k]==1){
                        int op = degree[i]+degree[j]+degree[k];
                        ans = Math.min(ans, op-6);
                    }
                }
            }
        }
        if(ans==Integer.MAX_VALUE){
            return -1;
        }
        return ans;
    }
}
