class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] g = new boolean[n][n];
        int[] d = new int[n];
        for (int[] e : edges) {
            e[0]--;
            e[1]--;
            g[e[0]][e[1]] = g[e[1]][e[0]] = true;
            d[e[0]]++;
            d[e[1]]++;
        }
        
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    if (g[i][j] && g[j][k] && g[i][k]) {
                        int s = d[i] + d[j] + d[k];
                        
ans = Math.min(ans, s - 6);
                    }
                }
            }
        }
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
}
