class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        int[] degree = new int[n + 1];
        int[][] map = new int[n + 1][n + 1];
        for (int[] edge : edges) {
            degree[edge[0]]++;
            degree[edge[1]]++;
            map[edge[0]][edge[1]] = 1;
            map[edge[1]][edge[0]] = 1;
        }
        int res = -1;
        for (int i = 1; i < n - 1; i ++) {
            for (int j = i + 1; j < n ; j++) {
                for (int k = j + 1; k < n + 1; k++) {
                    if (map[i][j] == 1 && map[j][k] == 1 && map[i][k] == 1) {
                        if (res < 0) {
                            res = degree[i] + degree[j] + degree[k] - 6;
                        } else {
                            res = Math.min(res, degree[i] + degree[j] + degree[k] - 6);
                        
}
                    }
                }
            }
        }
        return res;
    }
}
