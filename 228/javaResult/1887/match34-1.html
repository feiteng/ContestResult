class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] map = new boolean[n + 1][n + 1];
        int[] degree = new int[n + 1];
        for(int[] edge : edges){
            map[edge[0]][edge[1]] = true;
            map[edge[1]][edge[0]] = true;
            degree[edge[0]]++;
            degree[edge[1]]++;
        }
        
        int min = n * n;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= n; j++){
                if(map[i][j]){
                    
for(int k = 1; k <= n; k++){
                        if(map[i][k] && map[j][k]){
                            min = Math.min(min, degree[i] + degree[j] + degree[k] - 6);
                        }
                    }
                
}
            }
        }
        
        return min == n * n ? -1 : min;
    }
}
