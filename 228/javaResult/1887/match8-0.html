class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        int[][] dp = new int[n][n];
        int[] d = new int[n];
        for (int[] e : edges) {
            dp[e[0] - 1][e[1] - 1] = dp[e[1] - 1][e[0] - 1] = 1;
            d[e[0] - 1]++;
            d[e[1] - 1]++;
        }
        int ans = n * n;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                for (int k = j + 1; k < n; ++k) {
                    if (dp[i][j] == 1 && dp[j][k] == 1 && dp[i][k] == 1) {
                        ans = Math.min(d[i] + d[j] + d[k] - 6, ans);
                    }
                }
            }
        }
        
        return ans == n * n ? -1 : ans;
    }
}
