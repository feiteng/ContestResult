class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        boolean[][] adjacencyMatrix = new boolean[n][n];
        int[] degree = new int[n];
        for(int[] i: edges) {
            adjacencyMatrix[i[0]-1][i[1]-1] = true;
            adjacencyMatrix[i[1]-1][i[0]-1] = true;
            degree[i[0]-1]++;
            degree[i[1]-1]++;
        }
        int min = edges.length+1;
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                for(int k = j+1; k < n; k++) {
                    if(adjacencyMatrix[i][j] && adjacencyMatrix[i][k] && adjacencyMatrix[j][k]) {
                        min = Math.min(min, degree[i]+degree[j]+degree[k]-6);
                    }
                }
            }
        }
        return min > edges.length ? -1 : min;
    }
}
