class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        Set<Integer>[] G = new HashSet[n + 1];
        for(int i = 1; i <= n; i++) {
            G[i] = new HashSet<>();
        }
        for(int[] edge: edges) {
            G[edge[0]].add(edge[1]);
            G[edge[1]].add(edge[0]);
        }
        int ans = Integer.MAX_VALUE;
        
        for(int u = 1; u <= n; u++) {
            for(int v = u + 1; v <= n; v++) {
                for(int k = v + 1; k <= n; k++) {
                    if(G[u].contains(v) && G[v].contains(k) && G[k].contains(u)) {
                        int cur = G[u].size() + G[v].size() + G[k].size() - 6;
                        ans = Math.min(ans, cur);
                    }
                }
            }
        }

        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
}
