class Solution {
    public int minTrioDegree(int n, int[][] edges) {
        HashSet<Integer>[] set = new HashSet[n + 1];
        for (int i = 1; i <= n; i ++) {
            set[i] = new HashSet<>();
        }
        for (int[] edge: edges) {
            set[edge[0]].add(edge[1]);
            set[edge[1]].add(edge[0]);
        }
        int min = Integer.MAX_VALUE;
        for (int i = 1; i <= n; i ++) {
            for (int j = i + 1; j <= n; j ++) {
                for (int k = j + 1; k <= n; k ++) {
                    if (set[i].contains(j) && set[j].contains(k) && set[i].contains(k)) {
                       int sum = set[i].size() + set[j].size() + set[k].size() - 6;
                        
                        min = Math.min(min, sum);
                    }
                }
            }
        }
        return min == Integer.MAX_VALUE? -1: min;
    }
}
