class Solution {
    public int minOperations(String s) {
        int option1 = 0, option2 = 0;
        for (int i = 0;i < s.length();i++) {
            char c = s.charAt(i);
            if (i % 2 == 0) {
                if (c == '1') {
                    option1++;
                }
                else {
                    option2++;
                }
            }
            else {
                if (c == '1') {
                    option2++;
                }
                else {
                    option1++;
                }
            }
        }
        return Math.min(option1, option2);
    }
}
