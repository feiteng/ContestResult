class Solution {
    public int minOperations(String s) {
        int op = 0;
        boolean one = true;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '0' && one) op++;
            if (s.charAt(i) == '1' && !one) op++
;
            one = !one;
        }
        int op2 = 0;
        one = false;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1' && !one) op2++;
            if (s.charAt(i) == '0' && one) op2++
;
            one = !one;
        }
        return Math.min(op, op2);
    }
}
