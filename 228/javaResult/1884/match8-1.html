class Solution {
    public int minOperations(String s) {
                char[] arr = s.toCharArray();
        int cnt = 0, cnt1 = 0;
        for(int i = 0; i < arr.length; i++){
            if(i%2 == 0){
                if(arr[i] == '0'){
                    cnt++;
                }else
                    cnt1++;
            }else{
                if(arr[i] == '1'){
                    cnt++;
                }else
                    cnt1++;

            }
        }
        return Math.min(cnt, cnt1);
    }
}
