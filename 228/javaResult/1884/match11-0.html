class Solution {
    public int minOperations(String s) {
    	int n = s.length();
    	
    	int ans1=0, ans2=0;
    	for(int i=0; i<n; i++) {
    		if((i&1) == 1) {
    			if(s.charAt(i) == '1') ans1++;
    			else ans2++;
    		}
    		else {
    			if(s.charAt(i) == '0') ans1++;
    			else ans2++;
    		}
    	}
    	
    	return Math.min(ans1, ans2);
    }
}
