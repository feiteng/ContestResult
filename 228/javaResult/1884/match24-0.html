class Solution {
    public int minOperations(String s) {
        int d1 = 0;
        int d2=  0;
        for(int i=0;i<s.length();++i){
            int m =  s.charAt(i)-'0';
            if(i%2!=m){
                d1++;
            }
            if(i%2==m){
                d2++;
            }
        }
        return Math.min(d1,d2);
        
        
    }
}
