class Solution {
    public int minOperations(String s) {
        int n = s.length();
        int res0 = 0, res1 = 0;
        for(int i = 0; i < n; i++){
            char ch = s.charAt(i);
            if(i % 2 == 0 && ch != '0') res0++;
            if(i % 2 == 1 && ch != '1') res0++;
        }
        for(int i = 0; i < n; i++){
            char ch = s.charAt(i);
            if(i % 2 == 0 && ch != '1') res1++;
            if(i % 2 == 1 && ch != '0') res1++;
        }
        return(Math.min(res0, res1));
    }
}
