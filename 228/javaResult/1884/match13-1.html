class Solution {
    public int minOperations(String s) {
        int r1 = 0, r2 = 0;
        // 1, 0
        // 0, 1
        for(int i=0; i<s.length(); i++) {
            char c = s.charAt(i);
            if (i % 2 == 0) {
                if (c=='1') {
                    r2++;
                }
                else {
                    r1++;
                }
            }
            else {
                if (c=='0') {
                    r2++;
                }
                else {
                    r1++;
                }
            }
        }
        return Math.min(r1, r2);
    }
}
