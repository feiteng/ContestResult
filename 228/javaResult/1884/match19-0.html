class Solution {
    public int minOperations(String s) {
        //assume 0101
        int ret0 = 0;
        int ret1 = 0;
        char[] chars = s.toCharArray();
        for(int i = 0 ; i < chars.length;i++){
            if(chars[i]-'0' != i%2){
                ret0++;
            }else{
                ret1++;
            }
        }
        return Math.min(ret0,ret1);
    }
}
