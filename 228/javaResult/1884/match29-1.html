class Solution {
    public int minOperations(String s) {
        int r1 = 0, r2 = 0;
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                if (s.charAt(i) == '1') {
                    r1++;
                
}
            } else {
                if (s.charAt(i) == '0')
                    r1++;
            }
        }
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                if (s.charAt(i) == '0') {
                    r2++;
                
}
            } else {
                if (s.charAt(i) == '1')
                    r2++;
            }
        }
        return Math.min(r1, r2);
    }
}
