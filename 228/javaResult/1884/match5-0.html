class Solution {
    public int minOperations(String s) {
        int n1 = 0, n2 = 0;
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                if (s.charAt(i) == '0') {
                    n1++;
                } else {
                    n2++;
                }
            } else {
                if (s.charAt(i) == '1') {
                    n1++;
                } else {
                    n2++;
                }
            }
        }
        return Math.min(n1, n2);
    }
}
