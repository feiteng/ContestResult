class Solution {
    public int minOperations(String s) {
        int n = s.length();
        int x = 0;
        int y = 0;
        for (int i = 0; i < n; i++) {
            int cur = s.charAt(i) - '0';
            if (i%2 == 0) {
                x += cur;
                y += 1 - cur;
            } else {
                x += 1 - cur;
                y += cur;
            }
        }
        
        return Math.min(x, y);
    }
}
