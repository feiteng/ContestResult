class Solution {
    public int minOperations(String s) {
        char[] cs = s.toCharArray();
        int n = cs.length, a = 0, b = 0;
        for (int i = 0; i < n; ++i) {
            if (0 == (i & 1)) {
                if ('0' == cs[i]) b++;
                else a++;
            } else {
                if ('0' == cs[i]) a++;
                else b++;
            }
        }
        return Math.min(a, b);
    }
}
