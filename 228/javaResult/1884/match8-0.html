class Solution {
    public int minOperations(String s) {
        char[] c = s.toCharArray();
        int a = 0;
        int b = 0;
        for(int i = 0; i < c.length; i++){
            if(i % 2 == 0){
                if(c[i] == '0'){
                    a++;
                }else
                    b++;
            }else{
                if(c[i] == '1'){
                    a++;
                }else
                    b++;
            }
        }
        return Math.min(a,b);
        
    }
}
