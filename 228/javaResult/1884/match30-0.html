class Solution {
    public int minOperations(String s) {
        int op = 0;
        int start = 0;
        for(char c: s.toCharArray()){
            if(c-'0'!=start)op++;
            start = 1 - start;
        }
        
        int op2 = 0;
        start = 1;
        for(char c: s.toCharArray()){
            if(c-'0'!=start)op2++;
            start = 1 - start;
        }
        
        return Math.min(op, op2);
    }
}
