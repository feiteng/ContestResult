class Solution {
    public int minOperations(String s) {
        int len = s.length();
        int c1 = 0;
        int c2 = 0;
        for(int i = 0 ; i < len ; i++) {
            if(i % 2 == 0) {
                if(s.charAt(i) == '0')
                    c2 += 1;
                else
                    c1 += 1;
            }
            else {
                if(s.charAt(i) == '1')
                    c2 += 1;
                else
                    c1 += 1;
            }
        }
        //System.out.println(c1 + "\t" + c2);
        return Math.min(c1, c2);
    }
}
