class Solution {
    public int minOperations(String s) {
        int c1 = 0;
        for (int j=0;j<s.length();j++) {
            if (j%2==0 && s.charAt(j) == '1' ) {
                c1++;
            }
            if (j%2==1 && s.charAt(j) == '0') {
                c1++;
            }
        }
        int c2=0;
        for (int j=0;j<s.length();j++) {
            if (j%2==0 && s.charAt(j) == '0' ) {
                c2++;
            }
            if (j%2==1 && s.charAt(j) == '1') {
                c2++;
            }
        }
        return Math.min(c1, c2);
    }
}
