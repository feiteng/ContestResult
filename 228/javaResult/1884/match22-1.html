class Solution {
    public int minOperations(String s) {
        char start='1';
        int res=0;
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)!=start) res++;
            start=start=='1'?'0':'1';
        }
        int temp=0;
        start='0';
        
for(int i=0;i<s.length();i++){
            if(s.charAt(i)!=start) temp++;
            start=start=='1'?'0':'1';
        }
        
        
        return Math.min(res,temp);
    }
}
