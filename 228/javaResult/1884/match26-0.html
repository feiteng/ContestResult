class Solution {
    public int minOperations(String s) {
        int a0 = 0, a1 = 0;
        char[] cs = s.toCharArray();
        for (int i = 0; i < cs.length; i++) {
            if (i % 2 == 0) {
                a0 += (cs[i] - '0') % 2;
                a1 += ('1' - cs[i]) % 2;
            } else {
                a1 += (cs[i] - '0') % 2;
                a0 += ('1' - cs[i]) % 2;
            }
        }
        return Math.min(a0, a1);
    }
}
