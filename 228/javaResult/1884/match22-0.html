class Solution {
    public int minOperations(String s) {
        int count = 0;
        char num = '0';
        for (int i = 0; i < s.length(); ++i) {
            if (s.charAt(i) != num) ++count;
            num = num == '0' ? '1' : '0';
        }
        int min = count;
        count = 0;
        num = '1';
        for (int i = 0; i < s.length(); ++i) {
            if (s.charAt(i) != num) ++count;
            num = num == '0' ? '1' : '0';
        }
        return Math.min(min, count);
    }
}
