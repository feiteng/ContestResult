class Solution {
    private static final int MOD = 1000000007;
    
    public int countHomogenous(String s) {
        char prev = '#';
        int consec = 0;
        long total = 0;
        for (char c : s.toCharArray()) {
            if (c == prev) {
                ++consec;
            } else {
                prev = c;
                consec = 1;
            }
            total += consec;
            total %= MOD;
        }
        return (int)total;
    }
}
