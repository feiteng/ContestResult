class Solution {
    public int countHomogenous(String s) {
        char[] arr = s.toCharArray();
        long len = 0, mod = (long)1e9 + 7, res = 0;
        for (int i = 0;i < arr.length;i++) {
            if (i == 0 || arr[i] != arr[i - 1]) {
                res = ((long)res + (1 + len) * len / 2) % mod;
                len = 1;
            }
            else {
                len++;
            }
        }
        res = (res + (1 + len) * len / 2) % mod;
        return (int)res;
    }
}
