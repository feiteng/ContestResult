class Solution {
    public int countHomogenous(String s) {
        int last = -1;
        int ans = 0;
        int mod = 1000000007;
        for(int i=0; i < s.length(); i++) {
            if(last == -1 || s.charAt(i) != s.charAt(last)) {
                
last = i;
            }
            ans += (i - last + 1);
            ans %= mod;
        }
        return ans;
    }
}
