class Solution {
    public int countHomogenous(String s) {
        int len = 0, res = 0;
        for (int i = 0; i < s.length(); i++) {
            if (i == 0 || s.charAt(i) != s.charAt(i - 1)) {
                len = 0;
            }
            len++;
            res += len;
            res %= 1000000007;
        }
        return res;
    }
}
