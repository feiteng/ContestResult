class Solution {
    public int countHomogenous(String s) {
        char prev = '\0';
        int MOD = 1_000_000_007;
        long ans = 0, cnt = 0;
        for (char ch : s.toCharArray()) {
            if (ch != prev) {
                ans += (cnt + 1) * cnt / 2;
                prev = ch;
                ans %= MOD;
                cnt = 1;
            } else {
                cnt++;
            }
        }
        ans += (cnt + 1) * cnt / 2;
        ans %= MOD;
        return (int)ans;
    }
}
