class Solution {
    public int countHomogenous(String s) {
        
        int n = s.length();
        
        int i = 0;
        
        long ans = 0;
        int MOD = (int)1e9+7;
        
        while (i<n) {
            int j = i;
            while(j<n && s.charAt(j)==s.charAt(i)) {
                j++;
            }
            ans += (j-i)*1L*(j-i+1)/2;
            ans %=MOD;
            i = j;
        }
        
        return (int)(ans%MOD);
    }
}
