class Solution {

	public int countHomogenous(String s) {
		long count = 1, sum = 0;
		for (int i = 1; i < s.length(); i++) {
			if (s.charAt(i) == s.charAt(i - 1)) {
				count++;
			} else {
				sum += count * (count + 1) / 2;
				count = 1;
			}
		}
		sum += count * (count + 1) / 2;
		return (int) (sum % 1000000007);
	}
}
