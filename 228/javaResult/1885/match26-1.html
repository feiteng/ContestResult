class Solution {
    
public int countHomogenous(String s) {
    	int n = s.length();
    	
    	int cnt=1;
    	
    	long ans=1;
    	for(int i=1; i<n; i++) {
    		if(s.charAt(i) == s.charAt(i-1)) {
    			++cnt;
    			ans += cnt;
    			
    			if(ans > 1000000007) ans %= 1000000007;
    		}

    		else {
    			ans++;
    			cnt=1;
    		}
    	}
    	
    	return (int)ans;
    }
}
