class Solution {
    public int countHomogenous(String s) {
        int mod= (int) (1e9+7);
        long t=1;
        long sum=0;
        for(int i=1;i<s.length();i++){
            if(s.charAt(i)==s.charAt(i-1))t++;
            else{
                sum+=t*(t+1)/2;
                sum%=mod;
                t=1;
            }
        }
        return (int)((sum+t*(t+1)/2)%mod);
    }
}
