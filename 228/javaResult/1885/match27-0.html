class Solution {
    public int countHomogenous(String s) {
        long res = 0, cur = 1;
        for(int i=1;i<s.length();++i){
            if(s.charAt(i)==s.charAt(i-1))++cur;
            else{
                res += (cur+1)*cur/2;
                res = res % 1000000007;
                cur = 1;
            }
        }
        res += (cur+1)*cur/2;
        res = res % 1000000007;
        return (int)res;
    }
}
