class Solution {
    public int countHomogenous(String s) {
        long r = 0;
        long mod = 1000000007L;
        for(int i=0;i<s.length();){
            int j = i;
            while(j<s.length()&&s.charAt(j)==s.charAt(i)) { ++j;}
            long len  = j-i;
            
            r += len*(len+1)/2L;
            r %= mod;
            
            i = j;
        }
        return (int)r;
        
        
    }
}
