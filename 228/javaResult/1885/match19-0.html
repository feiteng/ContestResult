class Solution {
    public int countHomogenous(String s) {
        int count = 1;
        int res = 1;
        for (int i = 1; i < s.length(); ++i) {
            if (s.charAt(i) == s.charAt(i - 1)) {
                res += ++count;
            } else {
                count = 1;
                res += count;
            }
            res = res % ((int)1e9 + 7);
        }
        return res;
    }
}
