class Solution {
  public int countHomogenous(String s) {
    long res = 0;
    long longest = 0;
    for (int i = 0; i <= s.length(); i++) {
      if (i - 1 >= 0 && i < s.length() && s.charAt(i - 1) == s.charAt(i)) {
        longest += 1;
      } else {
        res = (res + longest * (longest + 1) / 2) % 1_000_000_007;
        longest = 1;
      }
    }
    return (int) res;
  }
}
