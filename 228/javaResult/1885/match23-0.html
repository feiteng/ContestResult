class Solution {
    public int countHomogenous(String s) {
        long sol = 0;
        int mod = (int)1e9 + 7;
        
        long ct = 1;
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i-1)) {
                ct++;
            } else {
                sol += ct * (ct + 1) / 2;
                sol %= mod;
                ct = 1;
            }
        }
        sol += (ct + 1) * ct / 2;
        sol %= mod;
        return (int)sol;
    }
}
