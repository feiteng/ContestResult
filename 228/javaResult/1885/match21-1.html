class Solution {
    public int countHomogenous(String s) {
        int mod=1000000007;
        long res=0;
        char[] ss=s.toCharArray();
        int n=ss.length;
        long cur=1;
        for(int i=1;i<n;i++){
            if(ss[i]==ss[i-1]) cur++;
            else
{
                res+=cur*(cur+1)/2;
                cur=1;
            }
        }
        res+=cur*(cur+1)/2;
        return (int)(res%mod);
    }
}
