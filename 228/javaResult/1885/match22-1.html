class Solution {
    public int countHomogenous(String s) {
        long res = 0;
        final int M = (int)(1e9) + 7;
        
        int l = 0, r = 0;
        
        //             l
        // a b b c c c a a
        //                 r
        
        // res: 1 1 2 1 2 3 1 2
        
        while(r < s.length()) {
            if(s.charAt(l) != s.charAt(r)) {
                
l = r;
            }
            
            res += (r - l) + 1;
            r++;
        }
        
        return (int)(res % M);
    }
}
