class Solution {
    private int MOD = 1_000_000_000 + 7;
    
public int countHomogenous(String s) {
        long res = 1, sum = 1;
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i-1)) {
                sum++;
            } else {
                sum = 1;
            }
            res = (res + sum) % MOD;
        }
        return (int)res;
    }
}
