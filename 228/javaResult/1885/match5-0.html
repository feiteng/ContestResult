class Solution {
    public int countHomogenous(String s) {
        int mod = 1000000007;
        long ans = 0;
        long curr = 1;
        for (int i=1; i<s.length(); i++) {
            if (s.charAt(i)==s.charAt(i-1)) {
                curr++;
            } else {
                ans+=(curr+1)*curr/2;
                ans %= mod;
                curr = 1;
            }
        }
        ans += (curr+1)*curr/2;
        ans %=mod;
        return (int) ans;
    }
}
