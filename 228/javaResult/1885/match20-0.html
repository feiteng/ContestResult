class Solution {
    public int countHomogenous(String s) {
        char[] arr = s.toCharArray();
        long ans = 0;
        int start = 0;
        char tar = arr[0];
        for(int i = 1; i < arr.length; i++) {
            if(arr[i] != tar) {
                ans = (ans+((long) (i-start+1)*(i-start)/2)%1000000007)%1000000007;
                start = i;
                tar = arr[i];
            }
        }
        ans = (ans+((long) (arr.length-start+1)*(arr.length-start)/2)%1000000007)%1000000007;
        return (int) ans;
    }
}
