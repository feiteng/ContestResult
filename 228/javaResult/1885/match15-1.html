class Solution {
    public int countHomogenous(String s) {
        if (s.length()==0) return 0;
        long ct = 1, cur = 1;
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i)==s.charAt(i-1)) {
                cur++;
                ct += cur;
            } else 
{
                cur = 1;
                ct += cur;
            }
            ct %= (int)(1e9+7);
        }
        return (int)ct;
    }
}
