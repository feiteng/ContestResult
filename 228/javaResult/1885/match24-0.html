class Solution {
    public int countHomogenous(String s) {
        int pre = 0;
        char preChar = ' ';
        long res = 0L;
        for (char c: s.toCharArray()) {
            if (preChar != c) {
                pre = 0;
                preChar = c;
            }
            pre++;
            res += pre;
            res %= 1_000_000_007;
        }
        
        return (int)res;
    }
}
