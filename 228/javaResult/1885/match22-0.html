class Solution {
    public int countHomogenous(String s) {
        long res = 0;
        long len = 1;
        int mod = (int)1e9 + 7;
        int idx = 0;
        while (idx + 1 < s.length()) {
            if (s.charAt(idx) == s.charAt(idx + 1)) {
                len++;
            } else {
                //System.out.println(idx + ":" + len);
                res += (1 + len) * len / 2;
                len = 1;
            }
            idx++;
        }
        
        res += (1 + len) * len / 2;
        
        return (int)(res % mod);
    }
}
