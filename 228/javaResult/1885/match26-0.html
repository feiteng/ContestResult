class Solution {
    static int mod= (int) (1e9+7);
    public int countHomogenous(String s) {
        int n=s.length();
        int cnt=1;
        int res=1;
        for(int i=1;i<n;i++){
            if ( s.charAt(i)==s.charAt(i-1)){
               cnt++;  
            }
            else {
                
                // res%=mod;
                cnt=1;
            }
            res+=cnt;
            res%=mod;
            
        }
    
        return (int) res%mod;

    }
}
