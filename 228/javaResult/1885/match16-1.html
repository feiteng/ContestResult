class Solution {
    private static final int mod = 1_000_000_007;

    public int countHomogenous(String s) {
        long res = 0;
        int st = 0;
        for (int i = 0; i <= s.length(); ++i) {
            if (i == s.length() || s.charAt(i) != s.charAt(st)) {
                int l = i - st;
                res += (1L + l) * l / 2;
                res %= mod;
                
st = i;
            }
        }
        return (int)(res);
    }
}
