class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        b, e = 1, 10 ** 9
        while b <= e:
            m = (b + e) // 2
            t = 0
            for i in nums:
                t += (i - 1) // m
            if t <= maxOperations:
                e = m - 1
            else:
                b = m + 1
        return b
