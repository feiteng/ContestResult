class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def foo(b):
            cnt = 0
            for i in nums:
                cnt += (i+b-1)//b - 1
            return cnt <= maxOperations
            
        
        lb,ub = 1,max(nums)
        while lb<ub:
            temp = (lb+ub)//2
            if foo(temp):
                ub = temp
            else:
                lb = temp+1
                
        return lb
