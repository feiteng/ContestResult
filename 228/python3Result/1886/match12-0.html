class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        l = 1
        r = max(nums)
        while l < r:
            mid = (l + r) // 2

            cnt = 0
            for c in nums:
                cnt += (c - 1) // mid

            if cnt > maxOperations:
                l = mid + 1
            else:
                r = mid
        return l
