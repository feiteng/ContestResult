class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def helper(nums,k):
            res=0
            for i in nums:
                res+=(i-1)//k
            return res
        start=1
        end=10**9
        ans=0
        while start<=end:
            mid=(end+start)//2
            if helper(nums,mid)<=maxOperations:
                end=mid-1
                ans=mid
            else:
                start=mid+1
        return ans
