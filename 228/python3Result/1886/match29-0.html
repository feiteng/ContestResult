class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        
        def is_possible(m):
            cnt = 0
            for n in nums:
                cnt += n // m - 1
                if n % m > 0:
                    cnt += 1
                    
            return cnt <= maxOperations
        
        mlo, mup = 1, max(nums)
        
        if is_possible(1):
            return 1
        
        while mlo < mup - 1:
            mmd = (mlo + mup) // 2
            if is_possible(mmd):
                mup = mmd
            else:
                mlo = mmd
        
        return mup
