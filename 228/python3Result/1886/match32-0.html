class Solution:
    def minimumSize(self, nums: List[int], o: int) -> int:
        def bs(mid):
            cur = 0
            for x in nums:
                cur += (x-1) // mid
                if cur > o:
                    return False
            return True
        
        l = 1
        r = max(nums)
        
        while l < r:
            mid = (l + r) // 2
            # print(l, r, mid, bs(mid))
            if bs(mid):
                r = mid
            else:
                l = mid + 1
        
        return l
