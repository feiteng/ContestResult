class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        l, r = 1, max(nums)
        while l <= r:
            o = (l + r) // 2
            c = 0
            for x in nums:
                c += x // o - 1
                if x % o > 0:
                    c += 1
            if c <= maxOperations:
                r = o - 1
            else:
                l = o + 1
        return r + 1
