class Solution:
    def minimumSize(self, nums: List[int], v: int) -> int:
        def judge(k):
            res = 0
            for i in nums:
                res += (i - 1) // k
            return res <= v
        left, right = 1, max(nums)
        while left < right:
            mid = left + (right - left) // 2
            if(judge(mid)):
                right = mid
            else:
                left = mid + 1
        return left
