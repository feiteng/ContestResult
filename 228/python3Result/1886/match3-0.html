class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def f(a):
            ans = 0
            for i in nums:
                ans += i // a + (0 if i % a != 0 else -1) 
            return ans <= maxOperations
        lo, hi = 1, max(nums)
        while lo < hi:
            mid = (lo + hi) // 2
            if f(mid):
                hi = mid
            else:
                lo = mid + 1
        return hi
