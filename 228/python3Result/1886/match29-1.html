class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def func(m: int):
            result = 0
            for num in nums:
                result += num//m - 1
                if num%m > 0:
                    result += 1
            return result
        
        L,R = 1,max(nums)
        if func(1) <= maxOperations:
            return 1
        
#        print(nums, maxOperations)
        while L < R-1:
            M = (L+R+1)//2
#            print("  ", L, M, R, func(M))
            if func(M) > maxOperations:
                L = M
            else:
                R = M
        return R
