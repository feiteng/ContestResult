class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        minn,maxn = 1,10**9
        while minn<maxn:
            mid = (minn+maxn)//2
            size = 0
            for i in nums:
                size += (i+mid-1)//mid - 1
            if size <= maxOperations:
                maxn = mid
            else:
                minn = mid + 1
        return maxn
