class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        
        
        lo = 1
        hi = max(nums)
        
        # print()
        while lo < hi:
            
            mid = (lo + hi) // 2
            # print(mid)
            res = 0
            for val in nums:
                res += (val - 1) // mid
                
            if res > maxOperations:
                lo = mid + 1
            else:
                hi = mid
                
        return hi
