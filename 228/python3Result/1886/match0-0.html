class Solution:
    def minimumSize(self, nums: List[int], ma: int) -> int:
        l=1
        r=max(nums)
        def f(x):
            cur=0
            for ch in nums:
                cur+=(ch+x-1)//x-1
            return cur<=ma
        #f(1)
        while l<r:
            mid=(l+r)//2
            if f(mid):
                r=mid
            else:
                l=mid+1
        return l
