class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def isok(ans):
            cnt = 0
            for num in nums:
                if num > ans:
                    cnt += (num-1) // ans
            return cnt <= maxOperations
        
        l, r = 1, max(nums)+1
        while l < r:
            mid = (l+r) // 2
            if isok(mid):
                r = mid
            else:
                l = mid + 1
        return l
