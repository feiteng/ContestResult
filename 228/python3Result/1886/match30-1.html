class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        
        def test(k) :
            total = 0
            for val in nums :
                total += (val + k - 1) // k - 1
            return total <= maxOperations
        
        x = 1
        y = max(nums)
        res 
= y
        while x <= y :
            z = (x + y) >> 1
            if test(z) :
                res = z
                y = z - 1
            else :
                x = z + 1
        return res
