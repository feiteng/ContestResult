class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def valid(mi):
            os = 0
            for v in nums:
                if v >= mi:
                    os += (v-1) // mi
            return os <= maxOperations
        
        lo, hi = 1, max(nums)
        while lo <= hi:
            mi = lo + (hi-lo)//2
            if valid(mi):
                hi = mi - 1
            else:
                lo = mi + 1
        return lo
