class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def possible(target):
            count = 0
            for num in nums:
                count += (num - 1) // target
                if count > maxOperations:
                    return False
            return True
        if maxOperations >= sum(nums):
            return 1
        low = 1
        high 
= max(nums)
        while low <= high:
            mid = (high - low) // 2 + low
            # print(mid, high, low, possible(low))
            # if low == high:
            #     if possible(low):
            #         return low
            #     return low + 1
            if possible(mid):
                high = mid - 1
            else:
                low = mid + 1
        return low
