class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        
        
        def operation(balls: int)-> int:
            ans = 0
            for x in nums:
                if x > balls:
                    ans += (x - 1) // balls
            # print(balls, ans)
            return ans 
        
        left, right = 1, max(nums)
        
        # print(left, right)
        while left < right:
            mid = (left + right) // 2
            if operation(mid) <= maxOperations:
                right = mid
            else:
                left = mid + 1
            # print("lr", left, right)
        
        return left
