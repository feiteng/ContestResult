class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        lo = 1
        hi = max(nums)
        
        def check(val):
            cnt = 0
            
            for num in nums:
                cnt += (num - 1) // val
                
            return cnt <= maxOperations
        
        while lo < hi:
            mid = (lo + hi) // 2
            # print(mid, check(mid))
            if check(mid):
                hi = mid
            else:
                lo = mid + 1
                
        return lo
