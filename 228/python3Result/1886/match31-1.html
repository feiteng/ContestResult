class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def is_valid(n_max):
            n_op = 0
            for num in nums:
                n_op += (num - 1) // n_max
                if n_op > maxOperations:
                    return False
            return True

        l, r = 1, max
(nums)
        while l < r:
            m = l + (r - l) // 2
            if is_valid(m):
                l, r = l, m
            else:
                l, r = m + 1, r
        return l
