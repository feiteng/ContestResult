class Solution:
    # binary search, time O(n*log(max(nums))), space O(n)
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        def works(x):
            cnt = 0
            for num in nums:
                cnt += (num - 1) // x
            return cnt <= maxOperations
        
        l, r = 1, max(nums)
        while l < r:
            mid = l + r >> 1
            if works(mid):
                r = mid
            else:
                l = mid + 1
        return l
