class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        start = 1
        end = max(nums)
        while start != end:
            size = (start + end) // 2
            operations = 0
            for n in nums:
                operations += (n-1)//size
            #operations = sum([(num-1)//size for num in nums])
            if operations <= maxOperations:
                end = size
            else:
                start = size+1
        return start
