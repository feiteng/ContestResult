class Solution:
    def minOperations(self, s: str) -> int:
        end_one = 0
        end_zero = 0
        for c in s:
            if c == '1':
                end_one, end_zero = end_zero, 1 + end_one
            else:
                end_one, end_zero = 1 + end_zero, end_one
        return min(end_one, end_zero)
