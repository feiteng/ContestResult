class Solution:
    def minOperations(self, s: str) -> int:
        def dist(s, start_c):
            ans = 0
            for c in s:
                if c != start_c:
                    ans += 1
                start_c = '1' if start_c == '0' else '0'
            return ans
            
        return min(dist(s, '0'), dist(s, '1'))
