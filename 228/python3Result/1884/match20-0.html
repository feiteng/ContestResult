class Solution:
    def minOperations(self, s: str) -> int:
        odd = 0
        even = 0
        for i, c in enumerate(s):
            if i % 2 == 0:
                if c == "0":
                    even += 1
                else:
                    odd += 1
            else:
                if c == "1":
                    even += 1
                else:
                    odd += 1
        return min(odd, even)
