class Solution:
    def minOperations(self, s: str) -> int:
        
        odd, even = 0, 0
        for i, c in enumerate(s):
            if i % 2:
                odd += int(c) == 1
                even += int(c) == 0
            else:
                odd += int(c) == 0
                even += int(c) == 1
        return min(odd, even)
