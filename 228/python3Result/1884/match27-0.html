class Solution:
    def minOperations(self, s: str) -> int:
        op_0 = 0
        op_1 = 0
        for letter in s:
            if letter == "1":
                op_0, op_1 = op_1 + 1, op_0
            else:
                op_0, op_1 = op_1, op_0+1
        return min(op_0, op_1)
