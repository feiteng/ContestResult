class Solution:
    def minOperations(self, s: str) -> int:
        r1 = 0
        t = 0
        for i in s:
            if int(i) != t: r1 += 1
            t = 1-t
        r2 = 0
        t = 1
        for i in s:
            if int(i) != t: r2 += 1
            t = 1-t
        return min(r1,r2)
