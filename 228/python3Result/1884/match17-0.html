class Solution:
    def minOperations(self, s: str) -> int:
        firstZeroCount = 0
        firstOneCount = 0
        for i, ch in enumerate(s):
            if (i % 2) == 0:
                if ch == '1':
                    firstZeroCount += 1
                else:
                    firstOneCount += 1
            else:
                if ch == '1':
                    firstOneCount += 1
                else:
                    firstZeroCount += 1
        return min(firstZeroCount, firstOneCount)
