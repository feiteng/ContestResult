class Solution:
    def minOperations(self, s: str) -> int:
        l = len(s)
        s1 = ''
        s2 = ''
        for i in range(l):
            if i%2==0:
                s1 += '0'
                s2 += '1'
            else:
                s2 += '0'
                s1 += '1'
        res1 = res2 = 0
        for i in range(l):
            if s1[i]!=s[i]:
                res1 += 1
            if s2[i]!=s[i]:
                res2 += 1
        return min(res1, res2)
