class Solution:
    def minOperations(self, s: str) -> int:
        first_0 = sum(char != '0' for char in s[::2]) + sum(char != '1' for char in s[1::2])
        first_1 = sum(char != '1' for char in s[::2]) + sum(char != '0' for char in s[1::2])
        return min(first_0, first_1)
