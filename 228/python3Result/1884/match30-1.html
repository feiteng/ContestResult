class Solution:
    def minOperations(self, s: str) -> int:
        n = len(s)
        s1, s2 = "", ""
        ans1, ans2 = 0, 0
        for i in range(n):
            if i & 1:
                s1 += "0"; s2 += "1"
            else:
                s1 += "1"; s2 += "0"
        
for i in range(n):
            if s[i] != s1[i]:
                ans1 += 1
            if s[i] != s2[i]:
                ans2 += 1
        return min(ans1, ans2)
