class Solution:
    def minOperations(self, s: str) -> int:
        a = b = 0
        for i in range(len(s)):
            if i % 2:
                a += s[i] != '0'
                b += s[i] != '1'
            else:
                a += s[i] != '1'
                b += s[i] != '0'
        return min(a, b)
