class Solution:
    def minOperations(self, s: str) -> int:
        ret = 0
        ret2 = 0
        for i in range(len(s)):
            if i % 2 == 0:
                if s[i] != '0':
                    ret += 1
                if s[i] != '1':
                    ret2 += 1        
            else:
                if s[i] != '1':
                    ret += 1    
                if s[i] != '0':
                    ret2 += 1    
        return min(ret, ret2)
