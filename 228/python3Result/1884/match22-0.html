class Solution:
    def minOperations(self, s: str) -> int:
        one = 0
        two = 0
        for i, c in enumerate(s):
            if i % 2:
                if c == '0':
                    one += 1
                else:
                    two += 1
            else:
                if c == '0':
                    two += 1
                else:
                    one += 1
        return min(one, two)
