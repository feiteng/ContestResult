class Solution:
    def minOperations(self, s: str) -> int:
        n = len(s)
        cnt1 = cnt2 = 0
        for i in range(n):
            if i % 2:
                if s[i] == '1':
                    cnt1 += 1
                else:
                    cnt2 += 1
            else:
                if s[i] == '0':
                    cnt1 += 1
                else:
                    cnt2 += 1
        return min(cnt1, cnt2)
