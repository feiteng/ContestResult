class Solution:
    def minOperations(self, s: str) -> int:
        mask = 0
        a1 = 0
        for c in s:
            if int(c) != mask:
                a1 += 1
            mask = 1-mask
        mask = 1
        a2 = 0
        for c in s:
            if int(c) != mask:
                a2 += 1
            mask = 1-mask
        return min(a1,a2)
