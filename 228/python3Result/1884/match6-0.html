class Solution:
    def minOperations(self, s: str) -> int:
        ct1, ct2 = 0, 0
        
        for i in range(len(s)):
            if i%2 ==0:
                if s[i] =='0':
                    ct2 += 1
                else:
                    ct1 += 1
            else:
                if s[i] =='1':
                    ct2 += 1
                else:
                    ct1 += 1
        
        return min(ct1, ct2)
