class Solution:
    def minOperations(self, s: str) -> int:
        n = len(s)
        
        cur1 = ''
        cur2 = ''
        
        for i in range(n):
            if i % 2 == 0:
                cur1 += '0'
                cur2 += '1'
            else:
                cur1 += '1'
                cur2 += '0'
        
        res1 = 0
        res2 = 0
        
        for i in range(n):
            if cur1[i] != s[i]:
                res1 += 1
            if cur2[i] != s[i]:
                res2 += 1
        
        return min(res1, res2)
