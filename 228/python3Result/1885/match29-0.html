class Solution:
    def countHomogenous(self, s: str) -> int:
        last_char=-1
        count=0
        ans=0
        for i in s:
            if last_char==i:
                count+=1
            else:
                ans=(ans+(count*(count+1))//2)% (10**9+7)
                last_char=i
                count=1
            # print(ans,count)
        ans=(ans+(count*(count+1))//2)%(10**9+7)
        return ans
