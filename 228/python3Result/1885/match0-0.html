class Solution:
    def countHomogenous(self, s: str) -> int:
        v = "abc"
        c = 0
        t = 0
        for i in s:
            if i == v:
                c += 1
            else:
                t += c*(c+1)//2
                c =1
            v = i
        t += c*(c+1)//2
                
                
                
        return t%((10**9)+7)
