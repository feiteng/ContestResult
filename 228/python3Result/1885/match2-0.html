class Solution:
    def countHomogenous(self, s: str) -> int:
        MAX = 10 ** 9 + 7
        prev = None
        count = 0
        ans = 0
        for i in s:
            if i == prev:
                count += 1
            else:
                ans = (ans + count * (count + 1) // 2) % MAX
                prev = i
                count = 1
        
        return (ans + count * (count + 1) // 2) % MAX
