class Solution:
    def countHomogenous(self, s: str) -> int:
        n = 0
        c_prev = ''
        c_count = 0
        for c in s:
            if c != c_prev:
                n += c_count * (c_count + 1) // 2
                c_prev = c
                c_count = 1
            else:
                c_count += 1
        n += c_count * (c_count + 1) // 2
        return n % int(1e9 + 7)
