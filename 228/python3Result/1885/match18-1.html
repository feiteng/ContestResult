class Solution:
    def countHomogenous(self, s: str) -> int:
        mod = 10**9+7
        pre = None
        cnt = 0
        res = 0
        for c in s:
            if c == pre:
                cnt+=1
            else:
                res=(res+cnt*(cnt+1)//2)%mod
                pre = c
                cnt=1
        res=(res+cnt*(cnt+1)//2)%mod
        return res
