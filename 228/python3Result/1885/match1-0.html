class Solution:
    def countHomogenous(self, s: str) -> int:
        pre = ''
        n = 0
        count = 0
        s = s+'.'
        for c in s:
            if c == pre:
                n += 1
            else:
                count += (n+1)*n//2
                pre = c
                n = 1
        return count%(10**9+7)
