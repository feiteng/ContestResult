class Solution:
    def countHomogenous(self, s: str) -> int:
        i = 0
        res = 0
        m = 10**9+7
        while i<len(s):
            j = i+1
            while j<len(s) and s[j]==s[i]:
                j += 1
            n = j-i
            res = (res+n*(n+1)//2)%m
            i = j
        return res
