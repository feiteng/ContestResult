class Solution:
    def countHomogenous(self, s: str) -> int:
        result = 0
        pre = ' '
        cnt = 0
        for ch in s:
            if ch != pre:
                result += cnt * (cnt + 1) // 2
                cnt = 1
            else:
                cnt += 1
            pre = ch
        result += cnt * (cnt + 1) // 2
        return result % (10 ** 9 + 7)
