class Solution:
    def countHomogenous(self, s: str) -> int:
        mod = 10**9 + 7
        ans = 0
        i = 0
        while i < len(s):
            j = i
            while j < len(s) and s[i] == s[j]:
                j += 1
            n = j - i
            ans += (1 + n) * n // 2 
            i = j
        return ans % mod
