class Solution:
    def countHomogenous(self, s: str) -> int:
        p = None
        cnt = ret = 0
        M = 10**9+7
        for v in s:
            if v!=p:
                p = v
                cnt = 1
            else:
                cnt+=1
            ret+=cnt
        
        return ret%M
