class Solution:
    def countHomogenous(self, s: str) -> int:
        count = 0
        pre = None
        res = 0
        M = 10**9 + 7
        for c in s:
            if c != pre:
                res = (res + count * (count + 1) // 2) % M
                pre = c
                count = 1
            else:
                count += 1
        res += count * (count + 1) // 2
        return res % M
