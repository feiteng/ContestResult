class Solution:
    def countHomogenous(self, s: str) -> int:
        last=''
        # d=defaultdict(int)
        cur=0
        ans=0
        for c in s:
            if c!=last:
                # d[last]=max(d[last],cur)
                ans+=cur*(cur+1)//2
                cur=1
                last=c
            else:
                cur+=1
        ans+=cur*(cur+1)//2
        # d[c]=max(d[c],cur)
        return ans%(10**9+7)
