class Solution:
    def countHomogenous(self, s: str) -> int:
        ans = 0
        pre = '0'
        cnt = 0
        for c in s + '0':
            if c == pre:
                cnt += 1
            else:
                ans = (ans + cnt * (cnt + 1) // 2) % 1000000007
                cnt = 1
                pre = c
        return ans
