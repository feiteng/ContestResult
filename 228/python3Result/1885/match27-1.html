class Solution:
    def countHomogenous(self, s: str) -> int:
        cur = None
        ct = 0
        ret = 0
        for c in s:
            if c != cur:
                ret += ct * (ct+1) // 2
                cur = c
                ct = 1
            else:
                ct += 1
        ret += ct * (ct+1) // 2
                
                
        return ret % (int(1e9)+7)
