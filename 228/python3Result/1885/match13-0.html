class Solution:
    def countHomogenous(self, s: str) -> int:
        prev = ''
        cnt = 0
        res = 0
        for ch in s :
            if ch == prev :
                cnt += 1
            else :
                res += cnt * (cnt + 1) // 2
                prev = ch
                cnt = 1
        res += cnt * (cnt + 1) // 2
        return res % (10**9 + 7)
