class Solution:
    def countHomogenous(self, s: str) -> int:
        M = 10 ** 9 + 7
        res = 0
        count = 0
        cur = ""
        for ch in s:
            if cur == "" or cur != ch:
                cur = ch
                count = 1
            else:
                count += 1
            res += count
            
        return res % M
