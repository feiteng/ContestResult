class Solution:
    def countHomogenous(self, s: str) -> int:
        mod = 10**9+7
        ans = 0
        cnt = 0
        pre = 0
        for c in s:
            if c == pre:
                cnt += 1
            else:
                ans = (ans + cnt*(1+cnt)//2)%mod
                cnt = 1
                pre = c
        ans = (ans + cnt*(1+cnt)//2) % mod
        return ans
