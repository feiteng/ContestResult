class Solution:
    def countHomogenous(self, s: str) -> int:
        prev = ""
        res = 0
        count = 0
        for c in s:
            if c != prev:
                res += count * (count+1) // 2
                count = 0
                prev = c
            count += 1
        return (res + count * (count+1) // 2) % (10**9+7)
