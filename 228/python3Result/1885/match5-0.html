class Solution:
    def countHomogenous(self, s: str) -> int:
        MOD = 10 ** 9 + 7
        pre = " "
        res = 0
        count = 0
        for c in s:
            if c == pre:
                count += 1
            else:
                count = 1
                pre = c
            res = (res + count) % MOD
        return res
