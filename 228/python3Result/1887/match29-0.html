class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        d = {}
        for v in range(1 + n):
            d[v] = set()
        for edge in edges:
            a = edge[0]
            b = edge[1]
            
            d[a].add(b)
            d[b].add(a)
        
        ret = len(edges)
        
        #print(d)
        v = None
        any = False
        for i in range(1, n + 1):
            for j in range(i+1, n + 1):
                for k in range(j+1, n + 1):
                    # is this a trio?
                    # print(f"testing  {i in d[j]}, i ={i}, v = {v} dv = {d[v]} { i in d[v]}, {j in d[v]}")

                    if i in d[j] and i in d[k] and j in d[k]:
                        # is a connected trio
                        degrees = len(d[i]) + len(d[j]) + len(d[k]) - 6
                        #print(i,j,k)
                        ret = min(degrees, ret)
                        any = True
        return -1 if any is False else ret
