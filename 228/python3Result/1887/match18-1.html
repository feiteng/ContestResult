class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        ht = collections.defaultdict(set)
        
        for x, y in edges:
            ht[x].add(y)
            ht[y].add(x)
        
        ans = math.inf
        for i in range(1, n+1):
            for j in range(i+1, n+1):
                for k in range
(j+1, n+1):
                    if j in ht[i] and k in ht[i] and k in ht[j]:
                        ans = min(ans, len(ht[i])-2 + len(ht[j])-2 + len(ht[k])-2)
        if ans == math.inf:
            return -1
        else:
            return ans
