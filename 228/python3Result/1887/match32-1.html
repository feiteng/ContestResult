class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        adj = defaultdict(set)
        for a, b in edges:
            adj[a].add(b)
            adj[b].add(a)
        min_deg = float('inf')
        for i, j in edges:
            ni = adj[i]
            nj = adj[j]
            if not ni & nj:
                continue
            for k in ni & nj:
                
if k < j or k < i:
                    continue
                deg = len(adj[i]) + len(adj[j]) + len(adj[k]) - 6
                # print(i, j, k)
                min_deg = min(min_deg, deg)
        return min_deg if min_deg != float('inf') else -1
