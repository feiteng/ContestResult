class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        liantong = []
        ans = float('inf')
        p2plist = collections.defaultdict(set)
        np = collections.defaultdict(int)
        for u, v in edges:
            p2plist[u].add(v)
            p2plist[v].add(u)
            np[u] += 1
            np[v] += 1
        ans = float('inf')
        for a in range(1, n+1):
            for b in range(a+1, n+1):
                if b not in p2plist[a]: continue
                for c in range(b+1, n+1):
                    if c not in p2plist[b]: continue
                    if a not in p2plist[c]:
                        continue
                    nn = np[a]+np[b]+np[c]
                    ans = min(ans, nn-6)     
        if ans==float('inf'): return -1
        return ans
