class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        g = [set() for _ in range(n + 1)] 
        for n1, n2 in edges:
            g[n1].add(n2)
            g[n2].add(n1)
        res = float('inf')
        
for n1 in range(n):
            for n2 in range(n1 + 1, n + 1):
                if n2 not in g[n1]:
                    continue
                for n3 in range(n2 + 1, n + 1):
                    if n3 not in g[n1] or n3 not in g[n2]:
                        continue
                    res = min(res, len(g[n1]) + len(g[n2]) + len(g[n3]) - 6)
        return res if res != float('inf') else -1
