class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        maps = collections.defaultdict(set)
        for i,j in edges:
            maps[i].add(j)
            maps[j].add(i)
            
        res = float('inf')
        trio = False
        for i in range(1, n+1):
            for j in range(i+1, n+1):
                if j not in maps[i]:
                    continue
                for k in range(j+1, n+1):
                    if k in maps[i] and k in maps[j]:
                        trio = True
                        degree = len(maps[i])+len(maps[j])+len(maps[k])-6
                        res = min(res, degree)
        
        if not trio:
            return -1
        return res
