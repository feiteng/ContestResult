class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        graph = defaultdict(set)
        for a, b in edges:
            graph[a].add(b)
            graph[b].add(a)

        if len(edges) == n * (n - 1) // 2 and n * 3 >= 9:
            return n * 3 - 9
            
        ans = inf
        used = set()
        for a in range(1, n + 1):
            used.add(a)
            for b, c in combinations(graph[a], 2):
                #print(f'b, c {(b, c)}')
                if b not in graph[c]:
                    continue
                if b in used or c in used:
                    continue
                cur = len(graph[a]) + len(graph[b]) + len(graph[c]) - 6
                ans = min(ans, cur)

        return ans if ans != inf else -1
