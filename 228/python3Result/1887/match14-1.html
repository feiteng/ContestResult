class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        g = collections.defaultdict(set)
        for u, v in edges:
            g[u].add(v)
            g[v].add(u)
        
        res = len(edges)
        
        for i in range(1, n+1):
            
for j in range(i+1, n+1):
                if j not in g[i]:
                    continue
                for k in range(j+1, n+1):
                    if k not in g[i] or k not in g[j]:
                        continue
                    ans = len(g[i]) + len(g[j]) +len(g[k]) - 6
                    res = min(res, ans)
                    if res == 0:
                        return 0
        
        return res if res != len(edges) else -1
