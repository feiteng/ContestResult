from collections import defaultdict

class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        g = defaultdict
(set)
        
        for a, b in edges:
            g[a].add(b)
            g[b].add(a)
        
        mtd = float("inf")
        for a in range(1, n + 1):
            for b in range(a + 1, n + 1):
                if b not in g[a]:
                    continue
                for c in range(b + 1, n + 1):
                    if c in g[a] and c in g
[b]:
                        # print((a,b,c), len(g[a]) + len(g[b]) + len(g[c]) - 3 * 2)
                        mtd = min(mtd, len(g[a]) + len(g[b]) + len(g[c]) - 3 * 2)
        return mtd if mtd != float("inf") else -1
