class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        res = 10**9
        edge_size = [0 for _ in range(401)]
        has_edge = [[0 for _ in range(401)] for _ in range
(401)]
        for u,v in edges:
            has_edge[u][v] = 1
            has_edge[v][u] = 1
            edge_size[u] += 1
            edge_size[v] += 1
        
for i in range(1,n+1):
            for j in range(i+1,n+1):
                for k in range(j+1,n+1):
                    if has_edge[i][j] and has_edge[j][k] and has_edge[i][k]:
                        res = min(res,edge_size[i] + edge_size[j] + edge_size[k] - 6)
        return res if res != 10**9 else -1
