class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        g = defaultdict(set)
        for u, v in edges:
            g[u].add(v)
            g[v].add(u)
        
        res = math.inf
        for i in g:
            for j in g[i]:
                if j <= i:
                    continue
                for k in g[j] & g[i]:
                    if k <= j:
                        continue
                    res = min(res, len(g[i]) + len(g[j]) + len(g[k]) - 6)
        return res if res < math.inf else -1
