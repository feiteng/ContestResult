class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        graph=[set() for i in range(n+1)]
        for x,y in edges:
            graph[x].add(y)
            graph[y].add(x)
        edge=[0 for i in range(n+1)]
        for x in range(1,n+1):
            for y in graph[x]:
                edge[x]+=1
        
        mins=10**9
        for x in range(1,n+1):
            for y in range
(x+1,n+1):
                if y not in graph[x]:
                    continue
                for z in range(y+1,n+1):
                    if (z not in graph[x]) or (z not in graph[y]):
                        continue
                    temps=edge[x]+edge[y]+edge[z]-6
                    mins=min(mins,temps)
        if mins==10**9:
            return -1
        return mins
