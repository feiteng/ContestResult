import collections

class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        graph = collections.defaultdict(set)
        for a, b in edges:
            graph[a - 1].add(b - 1)
            graph[b - 1].add(a - 1)
        mindeg = float('inf')
        for node in range(n):
            for node1 in graph[node]:
                if node1 < node:
                    continue
                for node2 in graph[node1]:
                    if node2 < node1:
                        continue
                    if node in graph[node2]:
                        mindeg = min(mindeg, len(graph[node]) + len(graph[node1]) + len(graph[node2]) - 6)
        return mindeg if mindeg < float('inf') else -1
