class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        adj = {e: set([]) for e in range
(1,n+1)}
        for n1, n2 in edges:
            adj[n1].add(n2)
            adj[n2].add(n1)
        
        best = float('inf')
        for e1, lst in adj.items():
            for e2 in lst:
                if e2 <= e1:
                    continue
                for e3 in lst:
                    
if e3 <= e2:
                        continue
                    if e3 in adj[e2]:
                        
                        best = min(best, len(adj[e1]) + len(adj[e2]) + len(adj[e3]) - 6)
        return best if best < float('inf') else -1
