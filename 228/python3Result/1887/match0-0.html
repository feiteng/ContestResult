class Solution:
    def minTrioDegree(self, dsa: int, nisho: List[List[int]]) -> int:
        akao = 100000000
        sando = []
        vimo = {}

        def pop_deu(ano):
            if ano not in vimo:
                vimo[ano] = 0
            vimo[ano] += 1

        for ano in range(dsa):
            sando.append([False] * dsa)
        for ano, ou in nisho:
            sando[ano - 1][ou - 1] = True
            sando[ou - 1][ano - 1] = True
            pop_deu(ano - 1)
            pop_deu(ou - 1)
        for ano1 in range(dsa):
            for ano2 in range(ano1 + 1, dsa):
                for ano3 in range(ano2 + 1, dsa):
                    if sando[ano1][ano2] and sando[ano2][ano3] and sando[ano3][ano1]:
                        current_deg = vimo[ano1] + vimo[ano2] + vimo[ano3] - 6
                        akao = min(akao, current_deg)
        if akao > 10000000:
            akao = -1
        return akao
