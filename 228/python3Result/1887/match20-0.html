class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        m = [0] * (n+1)
        k = [[0 for _ in range(n+1)] for _ in range(n+1)]
        for i in range(len(edges)):
            ui = edges[i][0]
            vi = edges[i][1]
            m[ui] += 1
            m[vi] += 1
            k[ui][vi] = 1
            k[vi][ui] = 1
        ans = float("inf")
        for i in range(1,n+1):
            for j in range(i+1,n+1):
                for l in range(j+1,n+1):
                    if k[i][j] == 1 and k[i][l] == 1 and k[j][l] == 1:
                        ans = min(ans,m[i]+m[j]+m[l]-6)
        if ans != float("inf"): return ans
        else: return -1
