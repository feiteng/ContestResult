class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        con=collections.defaultdict(set)
        for a,b in edges:
            con[a].add(b)
            con[b].add(a)
        re=float('inf')
        trios=0
        # print(con)
        for i in range(n):
            for j in range(i+1,n):
                # print(i,j)
                if j not in con[i]:
                    continue
                coms=con[i].intersection(con[i],con[j])
                for k in coms:
                    if k>j:
                        nv=len(con[i])+len(con[j])+len(con[k])-6
                        # trios.append([i,j,k,nv])
                        trios+=1
                        re
=min(re,nv)
        if not trios:
            return -1
        return re
