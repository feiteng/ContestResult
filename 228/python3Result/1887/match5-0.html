class Solution:
    def minTrioDegree(self, n: int, es: List[List[int]]) -> int:
        mat = [[0 for _ in range(n+1)] for _ in range(n+1)]
        deg = [0 for _ in range(n+1)]
        for x, y in es:
            mat[x][y] = 1
            mat[y][x] = 1
            deg[x] += 1
            deg[y] += 1
        ans = float('inf')
        for a in range(1, n+1):
            for b in range(a+1, n+1):
                for c in range(b+1, n+1):
                    if mat[a][b] and mat[b][c] and mat[a][c]:
                        ans = min(deg[a]+deg[b]+deg[c] - 6, ans)
        return ans if ans != float('inf') else -1
