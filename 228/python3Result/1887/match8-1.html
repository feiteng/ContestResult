from collections import defaultdict

class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        
        d = defaultdict(set)
        for u, v in edges:
            d[u-1].add(v-1)
            d[v-1].add(u-1)
            
        res = float('inf')
        for i in range(n):
            for j in range(i+1, n):
                for k in range(j+1, n):
                    if j in d[i] and k in d[i] and k in d
[j]:
                        res = min(res, len(d[i]) + len(d[j]) + len(d[k]) - 6)
        return res if res != float('inf') else -1
