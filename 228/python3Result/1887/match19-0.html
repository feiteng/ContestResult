class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        con = [[False for i in range(n + 1)] for j in range(n + 1)]
        cnt = [0 for i in range(n + 1)]
        for a, b in edges:
            cnt[a] += 1
            cnt[b] += 1
            con[a][b] = True
            con[b][a] = True
        ans = 10 ** 9
        for i in range(1, n - 1):
            for j in range(i + 1, n):
                for k in range(j + 1, n + 1):
                    if con[i][j] and con[i][k] and con[j][k]:
                        ans = min(ans, cnt[i] - 2 + cnt[j] - 2 + cnt[k] - 2)
        if ans == 10 ** 9:
            return -1
        return ans
