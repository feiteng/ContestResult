class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        graph = defaultdict(set)
        for u, v in edges:
            graph[u].add(v)
            graph[v].add(u)
        res = math.inf
        arr = sorted(graph, key = lambda x: len(graph[x]))
        for i in range(1, n + 1):
            if len(graph[i]) - 6 >= res:
                continue
            for j in range(1, i):
                if len(graph[i]) + len(graph[j]) - 6 >= res or j not in graph[i]:
                    continue
                for k in range(1, j):
                    if k in graph[i] and k in graph[j]:
                        res = min(res, len(graph[i]) + len(graph[j]) + len(graph[k]) - 6)
        return -1 if res == math.inf else res
