class Solution:
    def minTrioDegree(self, n: int, Edges: List[List[int]]) -> int:
        edges 
= collections.defaultdict(set)
        
        deg = collections.Counter() 
        for u, v in Edges:
            edges[u].add(v)
            edges[v].add(u)
            deg[u] += 1
            deg[v] += 1
            
        min_deg = float('inf')
        for i in range(1, n+1):
            for j in range(i+1, n+1):
                if j not in edges[i]: continue
                for k in range(j+1, n+1):
                    if k not in edges[j]: continue
                    if k not in edges[i]: continue
                        
                    min_deg = min(min_deg, deg[i]+deg[j]+deg[k]-6)
                    
        return -1 if min_deg == float('inf') else min_deg
