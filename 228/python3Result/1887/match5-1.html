class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        mp=[[0 for i in range(n)] for i in range(n)]
        du=[0 for i in range(n)]
        for i in edges:
            du[i[0]-1]+=1
            du[i[1]-1]+=1
            mp[i[0]-1][i[1]-1]=mp[i[1]-1][i[0]-1]=1
        flag=0
        ans=100000000
        
for i in range(n):
            for j in range(i+1,n):
                for k in range(j+1,n):
                    if mp[i][j] and mp[j][k] and mp[k][i]:
                        ans=min(ans,du[i]+du[j]+du[k]-6)
                        flag=1
        if not flag:
            return -1
        return ans
