import collections 
class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        d=collections.defaultdict(set)
        for u,v in edges:
            d[u].add(v)
            d[v].add(u)
        res=float("inf")
        for i in range(1,n+1):
            for j in d[i]:                
                if j>i: 
                    ints=d[i].intersection(d[j])
                    for x in ints:
                        if x>j: # find a new trio 
                            cur=0
                            cur+=len(d[i])+len(d[j])+len(d[x])-6
                            res=min(res,cur)
        if res==float("inf"): return -1
        return res
