class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        gr = defaultdict(set)
        for x, y in edges:
            gr[x].add(y)
            gr[y].add(x)
        res = math.inf
        for i in range(1, n + 1):
            ed = list(gr[i])
            for j, x in enumerate(ed):
                if x > i:
                    
for y in ed[j + 1:]:
                        if y > i and y in gr[x]:
                            res = min(res, len(gr[i]) + len(gr[x]) + len(gr[y]) - 6)
        return res if res != math.inf else -1
