class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        graph = defaultdict(set)
        for a,b in edges:
            graph[a].add(b)
            graph[b].add(a)
        
        vis = set()
        ans = float('inf')
        for i in range(1, n+1):
            for adj in graph[i]:
                for adj2 in graph[i]:
                    if i < adj < adj2 and adj2 in graph[adj]:
                        ans = min(ans, len(graph[adj]) + len(graph[i]) + len(graph[adj2]) - 6)
        if ans != float('inf'):
            return ans
        return -1
