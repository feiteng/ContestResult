class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        mydict = collections.defaultdict(set)
        for u, v in edges:
            mydict[v].add(u)
            mydict[u].add(v)
        ans = float('inf')
        for a in range(1, n - 1):
            if a not in mydict:
                continue
            for b in range(a+1, n):
                if b not in mydict[a]:
                    continue
                for c in range(b+1, n+1):
                    if c not in mydict[b] or c not in mydict[a]:
                        continue
                    count = len(mydict[a]) + len(mydict[b]) + len(mydict[c]) - 6
                    ans = min(ans, count)
        return ans if ans != float('inf') else -1
