class Solution:
    def minTrioDegree(self, n: int, edges: List[List[int]]) -> int:
        adj_set = {}
        for u in range(1, n + 1):
            adj_set[u] = set()
        for u, v in edges:
            adj_set[u].add(v)
            adj_set[v].add(u)
            
        degree_min 
= 3 * n
        for u in range(1, n + 1):
            for v in range(u + 1, n + 1):
                for w in range(v + 1, n + 1):
                    if v in adj_set[u] and w in adj_set[v] and u in adj_set[w]:
                        degree = len(adj_set[u]) + len(adj_set[v]) + len(adj_set[w]) - 6
                        degree_min = min(degree_min, degree)
        return degree_min if degree_min < 3 * n else -1
