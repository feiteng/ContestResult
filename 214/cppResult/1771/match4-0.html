typedef long long ll;
int mod=1e9+7;
class Solution {
public:
   
    int maxProfit(vector<int>& inv, int orders) {
       
        ll s = 1, e = 1e9;
        ll res=1,c=0;
        while(s<=e){
            ll mid = (s+e)/2;   ll exp=0;
            for(ll i:inv )
                exp +=max((ll)0 , i-mid+1);
            
            if(exp>=orders){
                res = mid;
                s=res+1;
                c= exp;
            }
            else 
                e= mid-1;
        }
  
       
        ll  ans = 0;
        ll mxVal = ((res*(res-1))/2)%mod;
        ll ex =  c - orders;
        for(int i=0;i<inv.size();i++){
            if(inv[i]>= res){
                ll val =inv[i];
                val=((val*(val+1))/2)%mod;
                ans =(ans +(val-mxVal+mod)%mod)%mod;
            }
        }
        
        return (ans-(ex*res)%mod +mod)%mod;
        
    }
};
