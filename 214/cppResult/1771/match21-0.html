typedef long long ll;
const int maxn=1e5+5;
const int mod=1e9+7;
int a[maxn];
class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        int n=inventory.size();
        ll sum=0;
        for(int i=0;i<n;i++){
            a[i]=inventory[i];
            sum+=a[i];
        }
        sort(a,a+n);
        ll ret=sum-orders;
        ll ans=0;
        for(int i=0;i<n;i++){
            ll t=(ret+n-i-1)/(n-i);
            t=min(t,1LL*a[i]);
            ret-=t;
            ll sel=a[i]-t;
            ans+=(a[i]+a[i]-sel+1)*sel/2%mod;
            // cout<<t<<" "<<ret<<" "<<ans<<endl;
            if(ans>=mod) ans-=mod;
        }
        // cout<<endl;
        return ans;
    }
};
