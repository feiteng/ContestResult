typedef long long ll;
class Solution {
public:
    int M = 1e9 + 7;
    int maxProfit(vector<int>& inv, int ord) {
        int n = inv.size();
        inv.push_back(0);
        sort(inv.rbegin(), inv.rend());
        ll cnt = 1;
        ll ans = 0;
        for(int i = 0; i < n; i++){
            if(inv[i] == inv[i + 1]) {
                cnt++;
                continue;
            }
            int cur = inv[i] - inv[i + 1];
            ll all = (ll)cnt*ll(cur);
            if(all < ord){
                ord -= all;
                ans += (ll(inv[i]) + ll(inv[i + 1]) + 1)*cur/2*cnt;
                ans %= M;
            }
            else{
                ll num = ord/cnt;
                ll res = ord - num*cnt;
                ans += (num * inv[i] - num * (num - 1) /2) *cnt;
                ans %=M;
                inv[i] -= num;
                ans += res*inv[i];
                ans%=M;
                break;
            }
            cnt ++;
        }
        return ans;
    }
};
