class Solution {
public:
    int maxProfit(vector<int>& inventory, long long orders) {
        long long mod = 1e9 + 7, res = 0, idx = 0, len = (int)inventory.size();
        sort(inventory.rbegin(), inventory.rend());        
        while(orders){
            if(idx < len - 1){
                long long cur_len = idx + 1;
                long long diff = inventory[idx] - inventory[idx + 1];
                long long count = min(orders,  cur_len * diff);
                long long take = count / cur_len;
                long long cost =  cur_len * 
                    ((1LL * inventory[idx] * (1LL * inventory[idx] + 1) / 2) - 
                     (1LL * (1LL * inventory[idx] - take) * (1LL * inventory[idx] - take + 1) / 2));
                cost += (count %  cur_len) * (inventory[idx] - take);
                orders -= count;                
                res += cost;
                res %= mod;
                ++idx;
            }else{
                long long cur_len = idx + 1;
                long long diff = inventory[idx];
                long long count = min(orders,  cur_len * diff);
                long long take = count / cur_len;
                long long cost =  cur_len * 
                    ((1LL * inventory[idx] * (1LL * inventory[idx] + 1) / 2) - 
                     (1LL * (1LL * inventory[idx] - take) * (1LL * inventory[idx] - take + 1) / 2));
                cost += (count %  cur_len) * (inventory[idx] - take);
                orders -= count;                
                res += cost;
                res %= mod;
            }
        }
        return res;
    }
};
