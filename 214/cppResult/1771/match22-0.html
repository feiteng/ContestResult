class Solution {
public:
    typedef long long LL;
    const int MOD = 1e9 + 7;
    LL fuck(LL x, LL v) {
        return v * (x - v + 1 + x) / 2 % MOD;
    }
    void upd(LL &x, LL y) {
        x += y;
        x %= MOD;
    }
    int maxProfit(vector<int>& a, int orders) {
        a.push_back(0);
        sort(a.begin(), a.end());
        int n = a.size();
        LL ans = 0;
        for (int i=n-1;i>0;i--) {
            LL now = a[i];
            LL cnt = n - i;
            LL target = a[i - 1];
            LL delta = now - target;
            //printf("%lld %lld %lld %lld %d %lld\n",now,cnt,target,delta,orders,orders - 1LL * cnt * delta);
            if (1LL * cnt * delta <= orders) {
                orders -= 1LL * cnt * delta;
                upd(ans, fuck(now, delta) * cnt % MOD);
            } else {
                LL m = orders % cnt;
                LL nd = orders / cnt;
                upd(ans, fuck(now, nd) * (cnt - m) + fuck(now, nd + 1) * m);
                break;
            }
        }
        return ans;
    }
};
