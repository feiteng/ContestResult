bool cmp(int x, int y) {
    return x > y;
}
class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        inventory.push_back(0);
        sort(inventory.begin(), inventory.end(), cmp);
        long long res = 0;

        const int N = 1e9 + 7 + 0.1;
        long long m = inventory[0];
        long long k = 1;
        for(int i = 1; i < inventory.size(); i++) {
            if(m > inventory[i]) {
                long long n = inventory[i];
                long long temp = k * (m - n);
                if(temp <= orders) {
                    orders -= temp;

                    res += k * (m + n + 1) * (m - n) / 2;
                    res %= N;
                    m = n;
                    k++;
                }
                else {
                    long long len = orders / k;
                    long long r = orders % k;
                    res += r * (m - len);
                    res += k * (m + m - len + 1) * len / 2;
                    res %= N;
                    orders = 0;
                    break;
                }
            }
            else {
                k++;
            }
        }
        
        return (int)res;
    }
};
