const int Exp = 1e9+7;
class Solution {
public:
    const int Exp = 1e9 + 7;
int maxProfit(vector<int>& in, int orders) {
	long long int ret = 0;
	in.push_back(0);
	sort(in.begin(), in.end(), greater<int>());
	int same = 0, len = in.size() - 1, t = 0;
	while (orders) {
		for (int i = same; i < len; ++i) {
			if (in[i] == in[i + 1]) { ++same; continue; }
			break;
		}
		long long int diff = in[same] - in[same + 1];
		t = diff*(same + 1);
		if (t <= orders) {
			orders -= t;
			ret = (ret + (in[same] + in[same] - diff + 1) * t / 2) % Exp;
			in[same] -= diff;
		}
		else {
			t = same + 1;
			diff = orders / t;
			orders = orders % t;
			ret = (ret + (in[same] + in[same] - diff + 1) * diff * t / 2) % Exp;
			ret = (ret + (in[same] - diff) * (orders)) % Exp;
			orders = 0;
		}
	}
	return ret;
}
   
};
