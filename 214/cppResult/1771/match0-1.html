int64_t mod=1e9+7;
class Solution {
public:
    int maxProfit(vector<int>& v, int n) {
        sort (v.begin(),v.end());
        reverse(v.begin(),v.end());

        int l=0,r=1e9;
        while (l<r){
            int mid=l+(r-l+1)/2;
            int64_t tmp=0;
            for (int i=0;i<v.size();i++){

                tmp+=max(0,v[i]-mid);
            }
            if (tmp>=(int64_t)n) l=mid;
            else r=mid-1;
        }
        int64_t tmp=l;
        int64_t sol=0;
        for (int i=0;i<v.size();i++){
            if (v[i]>tmp+1) {
                sol=(sol+((v[i]+tmp+2)*(v[i]-tmp-1)/2)%mod)%mod;
                n-=v[i]-tmp-1;
                v[i]=tmp+1;
            }
        }
        for (int i=0;i<v.size()&&n;i++){
            sol=(sol+v[i])%mod;
            n--;
        }
        return sol;
    }
};
