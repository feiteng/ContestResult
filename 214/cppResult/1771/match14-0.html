class Solution {
public:
    long long getk(vector<int>& inventory,long long x,long long n){
        long long sum=0;
        for(int i=0;i<n;i++) if(inventory[i]>x) sum+=inventory[i]-x;
        return sum;
    }
    int maxProfit(vector<int>& inventory, int orders) {
        long long sum=0;
        long long n=inventory.size();
        for(int i=0;i<n;i++)sum+=inventory[i];
        long long l=0;
        long long r=1000000000;
        long long res=0;
        while(l<r){
            long long mid=(l+r+1)/2;
            long long x=getk(inventory,mid,n);
            if(x>=orders){
                l=mid;
            }
            else{
                r=mid-1;
            }
        }
        //cout<<l<<endl;
        long long m=1000000007;
        long long pos=l+1;
        for(int i=0;i<n;i++){
            long long x=inventory[i];
            if(x<pos) continue;
            orders-=(x-pos);
            res=(res+(x+pos+1)*(x-pos)/2)%m;
            //cout<<res<<" "<<orders<<endl;
            
        }
        res=(res+(l+1)*orders)%m;
        return res;
    }
};
