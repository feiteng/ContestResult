#define ll long long

int Mod = 1e9 + 7;

class Solution {
public:
	ll f(ll x, ll y) {
		ll n = x-y;
		ll tot = (x+y+1)*n / 2;
		return tot;
	}
	
    int maxProfit(vector<int>& a, int orders) {
    	sort(a.rbegin(),a.rend());
		a.push_back(0);
		ll res = 0;
		for(ll i = 1;orders > 0;) {

			ll diff = a[i-1] - a[i];
			ll tot = diff * i; 
			
			if(orders >= tot) {
				res += i*f(a[i-1],a[i]);
				res %= Mod;
				orders -= tot;
			} else {
				ll n = orders / i;
				ll left = orders % i;
				ll v = f(a[i-1],a[i-1]-n) * i + left*(a[i-1]-n);
				res += v;
				res %= Mod;
				orders = 0;
			} 
			
			if(i < a.size()-1)	i++;
		} 
		return res;
    }
};
