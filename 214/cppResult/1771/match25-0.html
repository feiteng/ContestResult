class Solution {
public:
    const int MOD = 1e9 + 7;
    int maxProfit(vector<int>& inv, int ord) {
        int n = inv.size();
        int left = 0;
        int right = 1e9;
        long long res = 0;
        while(left <= right) {
            int mid = (left + right) / 2;
            long long sum = 0;
            bool flag = false;
            for (int x : inv) {
                if (x > mid)
                    sum += (x - mid);
                if (sum >= ord) {
                    flag = true;
                    break;
                }
            }
            if (flag) {
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        }
        for (int x : inv) {
            if (x > left) {
                ord -= (x - left);
                res = (res + (x + left + 1) * 1ll * (x - left) / 2) % MOD;
            }
        }
        //cout <<  left << ' ' <<  res << ' ' << ord << endl;
        res = (res + ord * 1ll * left) % MOD;
        return res;        
    }
};
