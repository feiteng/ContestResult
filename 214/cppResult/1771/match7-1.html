const int MOD = 1e9+7;
class Solution {
public:

  long long Gao(long long from, long long to) {
    return ((from+to)*(to-from+1)/2)%MOD;
  }
    int maxProfit(vector<int>& inventory, int orders) {
      auto A = inventory;
      map<long long, long long> M;
      M[0] = 0;
      for (auto a: A) M[a]++;
      long long tot = 0;
      while (orders) {
        auto it1 = prev(M.end());
        auto it2 = prev(it1);
        long long diff = it1->first-it2->first;

        long long n = it1->second;
        if (diff*n<=orders) {
          (tot += n*Gao(it2->first+1, it1->first))%=MOD;
          orders-=diff*n;
          it2->second += n;
          M.erase(it1);
        } else {
          long long x = orders / n;
          long long y = orders % n;
          (tot += n*Gao(it1->first-x+1, it1->first))%=MOD;
          (tot += y*(it1->first-x))%=MOD;
          break;
        }
      }
      return tot;
    }
};
