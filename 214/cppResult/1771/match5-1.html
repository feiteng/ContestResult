class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {

        const int mod = 1e9 + 7;
        int l = 0, r = 0;
        for (int i : inventory) {
            r = max(i , r);
        }
        while (l < r) {
            int mid = (l + r) / 2;
            long long cnt = 0;
            for (int i : inventory) {
                cnt += max(i - mid, 0);

            }
            if (cnt > orders) {
                l = mid + 1;
            }
            else {
                r = mid;
            }
        }
        long long ans = 0, cnt = 0;
        for (int i : inventory) {
            if (i > l) {
                ans += (i + l + 1) * 1ll * (i - l) / 2;
                ans = ans % mod;
            }
            cnt += max(i - l, 0);
        }
        ans += (orders - cnt) * l;
        ans = ans % mod;
        return ans;
    }
};
