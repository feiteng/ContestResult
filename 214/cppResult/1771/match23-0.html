typedef long long LL;
const int MOD = 1e9 + 7;
class Solution {
public:
    int maxProfit(vector<int>& inv, int orders) {
        sort(inv.begin(), inv.end());
        int n = inv.size();
        LL ans = 0;
        for (int k = n - 1; k; --k) {
            int c = n - k;
            int t = orders / c, d = inv[k] - inv[k - 1];
            if (d > t) {
                ans = (ans + 1ll * c * 1ll * (inv[k] + inv[k] - t + 1) * 1ll * t / 2) % MOD;
                ans = (ans + 1ll * (inv[k] - t) * 1ll * (orders % c)) % MOD;
                orders = 0;
                break ;
            } else {
                ans = (ans + 1ll * c * (inv[k] + inv[k - 1] + 1) * d / 2) % MOD;
                orders -= c * d;
            }
        }
        if (orders) {
            int t = orders / n;
            ans = (ans + 1ll * n * (inv[0] + inv[0] - t + 1) * t / 2) % MOD;
            ans = (ans + 1ll * (inv[0] - t) * (orders % n)) % MOD;
        }
        return ans;
    }
};
