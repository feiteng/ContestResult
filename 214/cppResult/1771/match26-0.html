class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        const int mod = 1e9 + 7;
        sort(inventory.begin(), inventory.end(), greater<int>());
        long long ret = 0;
        int i = 0;
        long long tot = 0;
        int n = inventory.size();
        while (i < n && tot < orders) {
            long long nxt = (i < n - 1 ? inventory[i+1]: 0);
            long long need = (inventory[i] - nxt) * (i + 1);
            if (tot + need <= orders) {
                long long single = (nxt + 1 + inventory[i]) * (inventory[i] - nxt) / 2;
                tot += need;
                ret += single * (i + 1);
                ret %= mod;
                //cout << ret << endl;
            } else {
                long long left = orders - tot;
                long long q = left / (i + 1);
                long long r = left % (i + 1);
                long long small = (inventory[i] + inventory[i] - q + 1) * q / 2;
                long long big = (inventory[i] + inventory[i] - q) * (q + 1) / 2;
                //cout << small << ' ' << big << endl;
                long long diff = big * r + small * (i + 1 - r);
                ret += diff;
                ret %= mod;
                //cout << ret << endl;
                break;
            }
            i++;
        }
        return ret;
    }
};
