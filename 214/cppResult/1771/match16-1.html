class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        sort(inventory.begin(), inventory.end(), std::greater());
        
        long long res = 0;
        for (int i = 0; i < inventory.size(); ++i) {
            if (orders == 0) {
                break;
            }
            
            if (i + 1 < inventory.size()) {
                long long diff = inventory[i] - inventory[i+1];
                if (diff * (i + 1) <= orders) {
                    long long cur = diff * (inventory[i] + inventory[i+1] + 1) / 2;
                    cur = cur * (i + 1);
                    res += cur;
                    orders -= diff * (i + 1);
                } else {
                    

                    long long l = orders / (i + 1);
                    long long r = orders - l * (i + 1);
                    
                    long long cur = l * (inventory[i] + inventory[i] - l + 1) / 2;
                    cur = cur * (i + 1);
                    res += cur;
                    
                    res += (inventory[i] - l) * r;
                    orders = 0;
                }
            } else {

                    long long l = orders / (i + 1);
                    long long r = orders - l * (i + 1);
                    
                    long long cur = l * (inventory[i] + inventory[i] - l + 1) / 2;
                    cur = cur * (i + 1);
                    res += cur;
                    
                    res += (inventory[i] - l) * r;
                    orders = 0;
            }
        }
        
        return res % mod;
    }
    
    long long mod = 1e9 + 7;
};
