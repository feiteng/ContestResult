class Solution {
    typedef long long ll;
    ll mod=1e9+7;
    ll cal(ll x, ll y) {
        return (x + y) * (y - x + 1) / 2;
    }
public:
    int maxProfit(vector<int>& inventory, int orders) {
        map<ll,ll> cnt;
        for(auto x:inventory) cnt[x]++;
        ll res=0;
        cnt[0]=0;
        while(orders) {
            auto it=prev(cnt.end());
            auto it1=prev(it);
            ll c=it->second;

            ll totv=cal(it1->first+1,it->first)*c%mod;
            ll totc=(it->first-it1->first)*c;
            if(totc<=orders) {
                orders-=totc;
                res=(res+totv)%mod;
                it1->second+=it->second;
                cnt.erase(it);
            }
            else {
                ll rm=orders/it->second;
                orders-=rm*it->second;
                res=(res+cal(it->first-rm+1,it->first)*it->second%mod)%mod;
                res+=(it->first-rm)*orders%mod;
                res%=mod;
                orders=0;
            }
        }
        return res;
    }
};
