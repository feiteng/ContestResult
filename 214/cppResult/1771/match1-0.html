class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        int mod = 1e9 + 7;
        sort(inventory.begin(), inventory.end());
        int l = 0, r = 1e9;
        while(l <= r){
            int mid = l + (r - l) / 2;
            if(sum(inventory, mid) > orders) l = mid + 1;
            else r = mid - 1;
        }

        long long total = sum(inventory, l);
        if(total == orders) return solve(inventory, l);
        else return (solve(inventory, l) + ((orders - total) * (l - 1)) % mod) % mod;
    }
    
private:
    long long sum(vector<int> &inventory, int val){
        long long total = 0;
        for(int x : inventory){
            if(x >= val) total += x - val + 1;
        }
        return total;
    }
    
    int solve(vector<int> &inventory, int val){
        long long res = 0, mod = 1e9 + 7;
        for(int x : inventory){
            if(x >= val){
                res = res + static_cast<long long>(x + val) * static_cast<long long>(x - val + 1) / 2;
                res %= mod;
            }
        }
        return static_cast<int>(res % mod);
    }
};
