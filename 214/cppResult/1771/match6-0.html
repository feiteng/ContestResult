class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        sort(inventory.begin(),inventory.end());
        int M=1000000007;
        int l=0,r=1000000000,mid;
        while(l+1<=r){
            mid=(l+r)/2;
            int cnt=0;
            for(auto i:inventory){
                cnt+=max(i-mid,0);
                if(cnt>orders)break;
            }
            if(cnt>orders){
                l=mid+1;
            }else{
                r=mid;
            }
        }
        long long ans=0;
        int tot=0;
        for(auto i:inventory){
            long long cnt=(i-l);
            if(cnt<0)continue;
            ans=(ans+(i+l+1)*cnt/2)%M;
            tot+=cnt;
        }
        while(tot<orders){
            ans=(ans+l)%M;
            tot++;
        }
        return ans;
    }
};
