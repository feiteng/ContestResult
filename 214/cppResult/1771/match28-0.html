class Solution {
public:
    int maxProfit(vector<int>& x, int orders) {
        sort(x.begin(), x.end(),greater<int>());
        x.push_back(0);
        int n = x.size();
        long ans = 0;
        int nmod = 1e9+7;
        for(int i=0;i+1<n;i++) { // todo: x[last]
            long m = i+1;
            long yield = (long) (x[i]-x[i+1]) * m;
            if(yield<=orders) {
                orders -= (long) (x[i]-x[i+1]) * m;
                ans = (ans + (long) m * ( (long) (x[i]+x[i+1]+1) * (long) (x[i]-x[i+1])) / 2)  % nmod;
                if(orders==0) break;
                //x[i] = x[i+1];
            } else { // no need to change all m numbers to x[i+1]
                // orders = 0;
                
                long p = orders / m, q = orders % m;
                ans = (ans + ( (long) m * ((long) (2*x[i]-p+1) * p / 2)) ) % nmod;
                ans = (ans + q * ( (long) x[i]-p))%nmod;
                //cout << orders << " " << m << " " << ans << endl;
                break;
            }
        }
        return ans % nmod;
    }
};
