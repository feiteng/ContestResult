typedef long long ll;
#define mod 1000000007
class Solution {
public:
    ll cal(vector<int>& a , ll k){
        ll sum = 0;
        for(auto &i:a ){
            ll p = max((long long)0 , (ll)(i-k+1));
           // if(k==1)
              //    cout<<p<<" hh ";
            sum = (sum + p);
        }
      //  if(k==1)
     //   cout<<sum<<endl;
        return sum;
    }

    int maxProfit(vector<int>& inventory, int orders) {
       
        ll s = 1, e = 1e9;
        ll ans=1;
        ll count = 0;
        while(s<=e){
            ll mid = (s+e)/2;
            ll p = cal(inventory,mid);
           // cout<<mid<<" "<<p<<endl;
            if(p>=orders){
                ans = mid;
                s= mid+1;
                count = p;
            }
            
            else 
                e= mid-1;
        }
      //  cout<<ans<<" "<<count<<endl;
        ll extra =  count - orders;
        ll  fans = 0;

        int n = inventory.size();
        ll lasi = (ans*(ans-1))/2;
        lasi = (lasi)%mod;
        for(int i=0;i<n;i++){
            if(inventory[i]>= ans){
                ll val = ((ll)(inventory[i])*(ll)(inventory[i]+1))/2;
                val = val%mod;
                ll tot = (val-lasi+mod)%mod;
                fans = (fans+tot)%mod;
            }
        }
        ll ext = ((extra*ans)%mod);
        fans  =(fans-ext +mod)%mod;
        return fans;
        
    }
};
