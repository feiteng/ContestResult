#define LL long long
const LL MOD = 1e9 + 7;


class Solution {
public:
    int maxProfit(vector<int>& inv, int orders) {
        int n = inv.size(); inv.push_back(0);
        sort(inv.begin(), inv.end()); reverse(inv.begin(), inv.end());
        
        LL ans = 0;
        for (int i = 0; i < n && orders > 0; i++){
            LL batch = i + 1;
            LL delta = inv[i] - inv[i + 1];
            LL cnt = min(orders / batch, delta);
            
            LL cost = 1LL * (inv[i] + inv[i] - cnt + 1) * cnt / 2LL * batch;
            orders -= batch * cnt;
            
            if (cnt < delta){
                cost += 1LL * (inv[i] - cnt) * orders;
                orders = 0;
            }
            
            ans += cost;
        }
        
        return ans % MOD;
    }
};
