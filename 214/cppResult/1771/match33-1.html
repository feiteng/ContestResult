class Solution {
public:
    #define ll long long
    const int MOD = 1e9 + 7;
    

    ll range(int a, int b) {
        int tot = b - a + 1;
        int add = a + b;
        return (ll)add * tot / 2 % MOD;
    }
    
    int maxProfit(vector<int>& a, int m) {
        int n = a.size();
        sort(a.rbegin(), a.rend());
        a.push_back(0);

        long long ans = 0;
        for (int i = 0; i < n; ++i) {
            int before = i + 1;
            int dif = a[i] - a[i + 1];
            if ((ll)dif * before < m) {
                m -= dif * before;
                ll x = range(a[i + 1] + 1, a[i]) * before;
                ans = (ans + x) % MOD;
            }
            else {
                int can = m / before;
                ll x = range(a[i] - can + 1, a[i]) * before;
                x += (ll)(a[i] - can) * (m % before);
                ans = (ans + x) % MOD;
                break;
            }
        }
        return ans;
    }
};
