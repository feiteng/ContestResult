class Solution {
public:
    const int mod = 1e9 + 7;
    long long cal(long long a,long long cnt){
        return (2 * a - cnt + 1) * cnt / 2 % mod;
    }
    
    int maxProfit(vector<int>& inv, int o) {
        inv.emplace_back(0);
        sort(inv.rbegin(),inv.rend());
        int n = inv.size();
        // if(n == 1){
        //     return cal(inv[0],inv[0]);
        // }
        // cout << accumulate(inv.begin(),inv.end(),0);
        long long ord = o;
        long long ans = 0;

        long long m = 1;            //等差和的倍数
        long long last = inv[0];    //目前最高价位
        for(int i = 1; i < n && ord > 0; ++i){
            if(inv[i] == last){
                ++m;
            }else{
                long long interval = last - inv[i];
                if(interval * m <= ord){
                    ord -= interval * m;
                    ans += cal(last,interval) * m;
                    ans %= mod;
                    ++m;
                }else{
                    // cout << "here" << endl;
                    // cout << ord << "  " << m << "  " << ans << "  " << last <<"  " << inv[i] << endl;
                    long long u = ord / m;
                    long long v = ord % m;
                    ans += cal(last,u) * m;
                    ans %= mod;
                    last -= u;
                    ans += last * v;
                    ans %= mod;
                    break;
                }
            }
            last = inv[i];
        }
        return ans;
    }
};
