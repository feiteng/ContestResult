class Solution {
public:
    const int mod = 1e9 + 7;
    typedef long long ll;
    int calc(int a, int b) {
        ll d = b - a + 1;
        return d * (a + b) / 2 % mod;
    }
    int maxProfit(vector<int>& inventory, int orders) {
        sort(inventory.rbegin(), inventory.rend());
        inventory.push_back(0);
        int n = inventory.size();
        ll ans = 0;
        for(int i = 0; i < n; i++) {
            ll len = i + 1, x = inventory[i] - inventory[i + 1];
            ll a = inventory[i];
            if(orders <= len * x) {
                ll d = orders / len, y = orders % len, z = len - y;
                return (int)((ans + calc(a - d, a) * y % mod + calc(a - (d - 1), a) * z % mod) % mod);
            }
            else {
                orders -= len * x;
                ans = (ans + calc(inventory[i + 1] + 1, a) * len % mod) % mod;
            }
        }
        return int(ans);
    }
};
