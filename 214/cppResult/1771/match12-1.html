class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        vector<long long> a;
        for(int i=0;i<inventory.size();i++)
        {
            a.push_back(inventory[i]);
        }
        sort(a.begin(),a.end());
        reverse(a.begin(),a.end());
        a.push_back(0);

        long long num=a[0];
        long long ans=0;
        long long mod=1e9+7;
        for(int i=1;i<a.size()&&orders;i++)
        {
            if(a[i]==a[i-1])
            {
                
            }
            else
            {
                if((long long)i*(a[i-1]-a[i])>=orders)
                {
                    long long t=orders/i;
                    ans+=(a[i-1]+a[i-1]-t+1)*t/2*i;

                    ans+=(orders%i)*(a[i-1]-t);
                    ans%=mod;
                    orders=0;
                }
                else
                {
                    long long t=a[i-1]-a[i];
                    ans+=(a[i-1]+a[i-1]-t+1)*t/2*i;
                    ans%=mod;
                    orders-=i*(a[i-1]-a[i]);
                }
            }
        }
        return ans;
    }
};
