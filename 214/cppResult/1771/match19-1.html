#define MOD 1000000007
class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        long long res = 0;
        
        sort(inventory.begin(), inventory.end(), greater<int>());
        inventory.push_back(0);
        int type_of_balls = 0;
        
        for (int i=0; i<inventory.size()-1;i++) {
            //cout << "(" << inventory[i] << "), (" << inventory[i+1] << ")\n";
            type_of_balls += 1;
            if (inventory[i] == inventory[i+1])
                continue;

            if ((inventory[i]-inventory[i+1])*type_of_balls > orders) {
                long long n = inventory[i];
                long long x = orders/type_of_balls;
                long long sum = (n*x*2-(x*(x-1)))/2;
                res += (sum*type_of_balls);
                res = res %MOD;

                orders -= (x*type_of_balls);
                res += (orders*(n-x));
                res = res %MOD;
                orders = 0;
            } else {
                long long n = inventory[i];
                long long x = inventory[i]-inventory[i+1];
                long long sum = (n*x*2-(x*(x-1)))/2;
                res += (sum*type_of_balls);
                res = res %MOD;
                orders -= (x*type_of_balls);
            }
            if (orders <= 0)
                break;
        }
        return res;
    }
};
