class Solution {
public:
    int maxProfit(vector<int>& a, int orders) {
        int n = a.size();
        

        long long lo = 1;
        long long hi = 1e9;
        
        while(lo < hi){
            long long mid = (lo + hi + 1)/2;
            
            long long canbe = 0;
            for(int i=0;i<n;i++){
                if(a[i] >= mid){
                    canbe += (a[i] - mid + 1);
                }
            }
            
            if(canbe >= orders){
                lo = mid;
            } else {
                hi = mid - 1;
            }
        }
        
        long long canbe = 0;
        for(int i=0;i<n;i++){

            if(a[i] >= lo){
                canbe += (a[i] - lo + 1);
            }
        }
        long long extra = canbe - orders;
        
        
        long long ans = 0;
        for(int i=0;i<n;i++){
            long long up = a[i];
            long long down = lo - 1;
            if(up >= down){
                ans += (up*(up+1))/2;
                ans -= (down*(down+1))/2;
            }
        }
        
        ans -= extra*lo;
        
        return ans%(1000000007);
    }
};
