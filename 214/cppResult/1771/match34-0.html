class Solution {
private:
    typedef long long ll;
    ll mod = 1000000007;
    
    ll calc(ll n) {
        return n * (n+1) / (ll)2;
    }
public:
    int maxProfit(vector<int>& invs, int orders) {
        ll n = invs.size();

        vector<ll> is(invs.size());
        for (int i=0; i<n; i++) {
            is[i] = invs[i];
        }
        
        sort(is.begin(), is.end(), greater<int>());
        
        ll result = 0;
        ll _n = 0;
        while (orders > 0) {
            while (_n+1<n && is[_n]==is[_n+1]) {
                _n++;
            }
            
            ll diff = INT_MAX;
            if (_n+1<n) {
                diff = is[_n]-is[_n+1];
            }
            
            ll buckets = _n+1;
            ll total = buckets * diff;
            
            if (total > orders) {
                ll diff = orders / buckets;
                ll b = (calc(is[_n]) - calc(is[_n]-diff)) % mod;
                result += buckets * b;
                result %= mod;
                is[_n] -= diff;
                orders -= diff * buckets;
                
                result += orders * is[_n];
                result %= mod;
                orders = 0;
            } else {
                ll b = (calc(is[_n]) - calc(is[_n]-diff)) % mod;
                result += buckets * b;
                result %= mod;
                is[_n] -= diff;
                orders -= diff * buckets;
            }
        }
        
        return result;
    }
};
