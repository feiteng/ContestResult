#define ll long long
#define mod 1000000007
class Solution {
public:
    ll fpow(ll a, int b){
        ll ans = 1;
        while(b){
            if(b&1){
                ans = (ans * a)%mod;
            }
            b>>=1;
            a = a * a %mod;
        }
        return ans;
    }
    int maxProfit(vector<int>& v, int order) {
        v.push_back(0);
        sort(v.begin(), v.end());
        ll cnt = 0;
        ll ans = 0;
        ll inv = fpow(2,mod-2);
        int n = v.size();
        for(int i = n - 1; i >= 1; i--){
            ll p = v[i];
            ll q = v[i]-v[i-1];
            ll now = ((((2*p*q)%mod - (q*q%mod))%mod+q)%mod)*inv%mod;
            ans = (ans + now * (n-i)%mod )%mod;
            cnt += (n - i) * (q);
            if(cnt >= order){
                ll d = cnt - order;
                q = d / (n-i);
                p = q + v[i-1];
                now = ((((2*p*q)%mod - (q*q%mod))%mod+q)%mod)*inv%mod;
                ans = (((ans - now * (n-i)%mod )%mod)+mod)%mod;
                d = d - q * (n-i);
                p++;
                ans = (((ans - d * p%mod)%mod)+mod)%mod;
                break;
            }
        }
        return ans;
    }
};
