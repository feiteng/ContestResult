class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        //vector<int> inventory=_inventory;
        inventory.push_back(0);
        sort(inventory.rbegin(),inventory.rend());
        long long i,j,ans=0,yu=1e9+7,n=inventory.size(),ball=0,k,m,k2;
        for(i=1;i<=n-1&&ball<orders;i++){
            j=i*(inventory[i-1]-inventory[i]);
            if(j<=orders-ball){
                k=inventory[i-1]+inventory[i]+1;
                k=k*(inventory[i-1]-inventory[i]);
                k/=2;
                //k*=((inventory[i-1]+inventory[i]+1)*(inventory[i-1]-inventory[i])/2);
                ball+=j;
                ans=(ans+i*k)%yu;
            }else{
                m=(orders-ball)/i;
                k=(inventory[i-1]+inventory[i-1]-m+1)*m/2;
                k2=(inventory[i-1]-m)*(orders-ball-i*m);
                ans=(ans+i*k+k2)%yu;
                ball=orders;
            }
        }
        return ans;
    }
};
