class Solution {
public:
    typedef long long ll;
    const ll i2 = 500000004, mod = 1e9 + 7;
    ll get(ll l, ll r){
        if(l>r) return 0;
        
        ll ansr = (r*(r+1)%mod)*i2%mod;
        --l;
        if(l<=0) return ansr;
        ll ansl = (l*(l+1)%mod)*i2%mod;
        return ((ansr - ansl)%mod + mod) %mod;
    }
    

    int maxProfit(vector<int>& a, int orders) {
        a.push_back(0);
        sort(a.rbegin(), a.rend());
        int n = a.size();
        ll ans = 0;
        ll len = 1;
        for(int i = 0; i+1 < n && orders>0; ++i){
            if(len*(a[i]-a[i+1]) <= orders){
                ans += len*get(a[i+1]+1, a[i]);

                ans %= mod;
                orders -= len*(a[i]-a[i+1]);
            }else{
                cout<< 2222222<<endl;
                cout<< orders<<endl;
                cout<< "len "<< len<<endl;
                ll dif = orders/len;
                ans += len*get(a[i] - dif + 1, a[i]);
                ans %= mod;
                cout<<"ans " << ans<<endl;
                cout<< "dif "<< dif<<endl;
                orders -= dif*len;
                ll num = a[i]-dif;
                ans += num*orders;
                ans %= mod;
                orders = 0;
            }
            ++len;
            cout<< ans<<endl;
        }
        return ans;
    }
};
