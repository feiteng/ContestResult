class Solution {
public:
    const int mod = 1e9 + 7;
    long long cal(long long l, long long r) {
        return (l + r) * (r - l + 1) / 2 % mod;
    }
    int maxProfit(vector<int>& inventory, int orders) {
        long long left = 0, right = *max_element(inventory.begin(), inventory.end());
        long long need = 0;
        while(left <= right) {
            long long mid = left + right >> 1;
            long long res = 0;
            for(int i = 0; i < inventory.size(); i++) {
                if(inventory[i] >= mid) {
                    res += (inventory[i] - mid + 1);
                }
            }
            if(res <= orders) {
                right = mid - 1;
                need = mid;
            } else {
                left = mid + 1;
            }
        }
        long long ans = 1e18;
        long long res = 0;
        for(int i = 0; i < inventory.size(); i++) {
            if(inventory[i] >= need) {
                res += cal(need, inventory[i]);
                res %= mod;
                orders -= (inventory[i] - need + 1);
                inventory[i] = need - 1;
            }
        }
        //cout << need << ' ' << orders << '\n';
        for(auto &x : inventory) {
            if(orders && x == need - 1) {
                res += need - 1;
                res %= mod;
                orders--;
            }
        }
        return res % mod;
    }
};
