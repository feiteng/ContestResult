class Solution {
public:

    long long eval(vector<int> &a, int t) {
        long long ans = 0;
        for (int x:a) {
            if (x>=t) ans += x-t+1;
        }
        return ans;
    }
    int maxProfit(vector<int>& a, int orders) {
        const long long MOD = 1e9+7;
        map<int, int> cnt;

        for (int x:a) cnt[x]++;
        int l = 0, r = 1e9;
        while (l!=r) {
            int mid = (l+r) / 2;
            if (eval(a, mid)<orders) r=  mid;
            else l = mid+1;
        }
        long long ans = 0;
        int td = r;
        for (int x:a) {
            if (x >= td) {
                // td + (td+1) + ... + x
                ans += (long long)(x+td) * (x-td+1) / 2;
                ans %= MOD;
                orders -= (x-td+1);
            }
        }
        ans += (long long) orders * (td-1);
        return ans % MOD;
    }
};
