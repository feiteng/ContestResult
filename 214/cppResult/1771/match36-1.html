class Solution {
public:
    int maxProfit(vector<int>& I, int o) {
        I.push_back(0);
        sort(I.begin(), I.end());
        long long c = 0;
        long long s = 0;
        for (int i = I.size() - 1 ; i > 0; i--) {
            c++;
            if ((I[i] - I[i-1]) * c <= o) {
                s += sum(I[i-1], I[i]) * c;
                //cout << I[i] <<"-"<< I[i-1] << "-" << sum(I[i-1], I[i]) << endl;
                o-= (I[i] - I[i-1]) * c;
                s %= 1000000007;
                //cout << s << endl;
                continue;
            }
            
            long long k = o / c;
            //cout << k  << endl;
            long long a = I[i] - k;
            a = max(a, 0ll);
            //cout << "a:" << a << "-" << I[i] << endl;
            s += sum(a, I[i]) * c;
            //cout << sum(I[i], a) * c << endl;
            //o-= (I[i] - a) * c;
            o-= k * c;
            //cout << s << " " << o << a << endl;
            s += o * a;
            s %= 1000000007;
            
            break;
        }
        return s;
    }
    long long sum(int a, int b) {
        return (sum1(b) - sum1(a)) % 1000000007;
    }
    long long sum1(long long a) {
        return a * ( a +1) / 2;
    }
};
