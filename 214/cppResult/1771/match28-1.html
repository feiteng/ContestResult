class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        sort(inventory.begin(),inventory.end(),greater<int>());
        inventory.push_back(0);
        int size=inventory.size();
        long res=0;
        int num=0;
        for(int i=0;i<size-1;i++){
            if((inventory[i]-inventory[i+1])*(i+1)+num>=orders){
                int fre=(orders-num)/(i+1);
                int re=(orders-num)%(i+1);
                res+=(((long)inventory[i]+(long)(inventory[i]-fre+1))*fre*(i+1))/2;
                res+=(long)re*(inventory[i]-fre);
                res%=1000000007;
                break;
            }
            else{
                res+=(((long)inventory[i]+(long)(inventory[i+1]+1))*(inventory[i]-inventory[i+1])*(i+1))/2;
                res%=1000000007;
                num+=(inventory[i]-inventory[i+1])*(i+1);
            }
        }
        return res;
    }
};
