class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        sort(inventory.rbegin(), inventory.rend());
        constexpr long long kPrime = 1e9 + 7;
        long long result = 0;
        for (int i = 0; i < inventory.size() && orders > 0; ++i) {
            int diff = inventory[i] - (i + 1 == inventory.size() ? 0 : inventory[i + 1]);
            if (1LL * diff * (i + 1) <= orders) {
                orders -= diff * (i + 1);
                result = (result + diff * (inventory[i] * 2LL - diff + 1LL) / 2 % kPrime * (i + 1) % kPrime) % kPrime;
            } else {
                diff = orders / (i + 1);
                result = (result + diff * (inventory[i] * 2LL - diff + 1LL) / 2 % kPrime * (i + 1) % kPrime) % kPrime;
                orders -= diff * (i + 1);
                result = (result + 1LL * orders * (inventory[i] - diff) % kPrime) % kPrime;
                break;
            }
        }
        return static_cast<int>(result);
    }
};
