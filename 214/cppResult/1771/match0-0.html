class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        const int mod=1e9+7;
        sort(inventory.begin(), inventory.end());
        int n=inventory.size();
        #define ll long long
        int left=1, right=1e9;
        while(left<right){
            int mid=(left+right)>>1;
            ll cnt=0;
            for(int i=0;i<n;i++){
                if(inventory[i]>=mid) cnt+=inventory[i]-mid+1;
            }
            if(cnt>orders) left=mid+1;
            else right=mid;
        }
        
        ll ans=0, cnt=0;
        for(int i=0;i<n;i++){
            if(inventory[i]>=left) cnt+=inventory[i]-left+1, ans=(ans+1ll*(1ll*inventory[i]+left)*(1ll*inventory[i]-left+1)/2)%mod;
        }
        //cout<<ans<<" "<<left<<endl;
        ans=(ans+(orders-cnt)*(left-1))%mod;
        //cout<<ans<<" "<<left<<endl;
        return ans;
    }
};
