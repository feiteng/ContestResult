class Solution {
public:
    const int mod = 1e9 + 7;
    int maxProfit(vector<int>& inventory, int orders) {
        int l = 0, r = 1e9;
        while (l < r) {
            int mid = (l + r + 1) >> 1;
            if (check(inventory, mid) < orders) {
                r = mid - 1;
            } else {
                l = mid;
            }
        }
        int ans = 0;
        int cnt = 0;

        for (int v: inventory) {
            if (v > l) {
                cnt += v - l;
                ans = (ans + (v + l + 1) * 1ll * (v - l) / 2) % mod;
            }
        }
        ans = (ans + (orders - cnt) * 1ll * l) % mod; 
        return ans;
    }
    long long check(vector<int> &inventory, int t) {
        long long cnt = 0;
        for (int v: inventory) {
            if (v >= t) {
                cnt += v - t + 1;
            }
        }
        return cnt;
    }
};
