class Solution {
public:
	int maxProfit(vector<int>& a, int target) {
		a.push_back(0);
		sort(a.begin(), a.end());
		
		int n = a.size();
		vector<long long> b(n);
		b[0] = a[0];

		for (int i = 1; i < n; ++i)
			b[i] = b[i - 1] + a[i];

		long long sum = 0;
		long long MOD = 1e9 + 7;
		int cur = n - 1;


		while (target > 0) {
			long long dif = a[cur] - a[cur - 1];
			long long s = a[cur - 1] + 1;
			long long e = a[cur];
			long long cnt = e - s + 1;
			long long curS = (e + s) * cnt / 2;

			if (target >= dif * (n - cur)) {
				curS *= n - cur;
				sum += curS % MOD;
				target -= dif * (n - cur);
			}
			else {
				int cnt = n - cur;
				
				long long val = a[cur];

				int take = target / cnt;

				long long s = val - take + 1;
				long long e = val;
				long long cnt1 = e - s + 1;
				long long curS = (e + s) * cnt1 / 2;
				
				sum += curS * cnt;

				target -= cnt * take;
				sum += target * (s - 1);

				target = 0;
			}
			
			sum %= MOD;
			cur--;
		}
		return sum % MOD;
	}
};
