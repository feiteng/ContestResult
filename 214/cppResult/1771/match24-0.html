class Solution {
public:
    long long cal(vector<int>& inventory, int start) {
        long long ret = 0;
        for (int& x: inventory) {
            if (x < start) {
                continue;
            }
            ret += x - start + 1;
        }
        return ret;
    }
    int maxProfit(vector<int>& inventory, int orders) {
        long long ret = 0;
        int start = -1;
        int end = 1000000000;
        while (end - start > 1) {
            int mid = (start + end) / 2;
            if (cal(inventory, mid) > orders) {
                start = mid;
            } else {
                end = mid;
            }
        }
        for (int& x: inventory) {
            if (x < end) {
                continue;
            }
            ret += (long long)(x + end) * (x - end + 1) / 2;
            orders -= (x - end + 1);
        }
        for (int i = 0;i < orders;i++) {
            ret += end - 1;
        }
        return ret % 1000000007;
    }
};
