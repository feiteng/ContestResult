#define ll long long
class Solution {
public:
    int maxProfit(vector<int>& inventory, int orders) {
        ll res=0;
        const int mod=1e9+7;
        inventory.push_back(0);
        sort(inventory.begin(),inventory.end());
        int n=inventory.size();
        for(ll i=n-1;i>0 && orders>0;){
            ll j=inventory.end()-lower_bound(inventory.begin(),inventory.end(),inventory[i]);
            ll k=inventory[i]-inventory[n-j-1];
            if(j*k<=orders){
                ll z= inventory[i]-k;
                ll x=(j*k)%mod;
                res=(res+j*(((ll)inventory[i]*(inventory[i]+1)/2)%mod - (z*(z+1)/2)%mod)+mod)%mod;
                orders-=j*k;
            }
            else{
                ll z=inventory[i]-orders/j,rem=orders%j;
                // ll x=(j*(orders/j))%mod;
                res=(res+j*(((ll)inventory[i]*(inventory[i]+1)/2)%mod - (z*(z+1)/2)%mod)+mod)%mod;
                res=(res+rem*((z)%mod))%mod;
                break;
            }
            i=n-j-1;
            // cout<<res<<" "<<i<<" "<<j<<" "<<k<<"\n";
        }
        return (res+mod)%mod;
    }
};
