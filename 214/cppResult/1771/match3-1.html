typedef long long int lli;

class Solution {
public:
    int maxProfit(vector<int> v, int ord) {
        lli n=v.size();
        lli l=0;
        lli r=1000000001;
        lli mi=0;
        while(l<=r){
            lli m=(l+r)/2ll;
            lli sum=0;
            for(lli i=0;i<n;i++){
                sum+=max(0ll,v[i]-m);
            }
            if(sum>=ord){
                l=m+1;
                mi=max(mi,m);
            }
            else{
                r=m-1;
            }

        }
        lli to=0;
        lli ans=0;
        lli g=(mi*(mi+1))/2ll;
        for(int i=0;i<n;i++){
            if(v[i]>mi){
                ans+=((lli(v[i]))*(lli(v[i]+1)))/2ll;
                ans-=g;    
                to+=v[i]-mi;
                while(ans<0){
                    ans+=1000000007;
                }
                ans%=1000000007;
            }
        }
        if(to>ord){
            ans-=(mi+1)*(to-ord);
            while(ans<0){
                ans+=1000000007;
            }
        }
        while(ans<0){
            ans+=1000000007;
        }
        return ans;
    }
};
