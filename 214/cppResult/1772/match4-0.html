class Solution {
public:
    int BIT[100009];
    
    void add(int pos, int val) {
        while(pos < 100009) {
            BIT[pos] += val;
            pos += (pos & -pos);
        }
    }
    
    int get(int pos) {
        if(pos <= 0) return 0;
        int ans = 0;
        while(pos > 0) {
            ans += BIT[pos];
            pos -= (pos & -pos);
        }
        return ans;
    }
    
    int createSortedArray(vector<int>& a) {
        memset(BIT, 0, sizeof(BIT));
        long long ans = 0;
        int MOD = 1e9 + 7;
        for(int i = 0; i < a.size(); i++) {
            int pos = a[i];
            int low = get(pos - 1);
            int high = get(100007) - get(pos);
            ans = ans + min(low, high);
            ans = ans % MOD;
            add(pos, 1);
        }
        return ans;
    }
};
