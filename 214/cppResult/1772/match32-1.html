class BIT {
private:

    vector<int> tree;
    int n;

public:
    BIT(int _n): n(_n), tree(_n + 1) {}

    static constexpr int lowbit(int x) {
        return x & (-x);
    }

    void update(int x, int d) {
        while (x <= n) {
            tree[x] += d;
            x += lowbit(x);
        }
    }

    int query(int x) const {
        int ans = 0;
        while (x) {
            ans += tree[x];
            x -= lowbit(x);
        }
        return ans;
    }
};




class Solution {
public:
    int createSortedArray(vector<int>& instructions) {
        int mod  = 1000000007;
        long long sum = 0;
        
        int ret = 0;
        BIT bit(100001);
        for (int i = 0; i < instructions.size(); i++) {
            int x = instructions[i];
            ret += min(bit.query(x) - bit.query(0), bit.query(100001)-bit.query(x+1));
            ret = ret%mod;
            bit.update( x+1 , 1);
        }
        return ret;

    }
};
