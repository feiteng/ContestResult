class Solution {
public:
    int n=100005;
    int a[100005];
    int lowbit(int x){
        return x&(-x);
    }
    void insert(int k,int x){
        while(k<=n){
            a[k]+=x;
            k+=lowbit(k);   
        }
    }
    int getsum(int k){
        int sum=0;
        while(k>0){
            sum=(sum+a[k])%1000000007;
            k-=lowbit(k);
        }
        return sum;
    }
    int createSortedArray(vector<int>& instructions) {
        int res=0;
        int m=instructions.size();
        for(int i=0;i<instructions.size();i++){
            int x=instructions[i];
            int y=getsum(x-1);
            int z=getsum(x)-y;
            //cout<<x<<" "<<y<<" "<<min(y,i-y-z)<<endl;
            res=(res+min(y,i-y-z))%1000000007;
            insert(x,1);
        }
        return res;
    }
};
