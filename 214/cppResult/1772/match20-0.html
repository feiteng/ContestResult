class Solution
{
    int n;
	vector<int> c;
	
	void modify(int x, int delta)
	{
	    while (x <= n)
		{
			c[x] += delta;
			x += x & -x;
		}
	}

	int sum(int x)
	{
		int ss = 0;
		while (x > 0)
		{
			ss += c[x];
			x -= x & -x;
		}
		return ss;
	}

public:
	int createSortedArray(vector<int>& instructions)
    {
		n = 100000;
        c = vector<int>(n + 1);
        int ret = 0;
        for (int w: instructions)
        {
            int l = sum(w - 1);
            int r = sum(n) - sum(w);
            ret = (ret + min(l, r)) % 1000000007;
            modify(w, 1);
        }
        return ret;
	}
};
