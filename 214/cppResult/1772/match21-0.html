class Solution {
int a[100010];
int lowbit(int x){
    return x&(-x);
}

void updata(int i,int k){    //在i位置加上k
    while(i < 100010){
        a[i] += k;
        i += lowbit(i);
    }
}

int getsum(int i){        //求A[1 - i]的和
    int res = 0;
    while(i > 0){
        res += a[i];
        i -= lowbit(i);
    }
    return res;
}
public:
    int createSortedArray(vector<int>& instructions) {
        int mod=1e9+7;
        int ans=0;
        int l=instructions.size();
        for(int i=0;i<l;i++)
        {
            int mn=getsum(instructions[i]-1),mx=getsum(100010-1)-getsum(instructions[i]);
            //cout<<mn<<" "<<mx<<endl;
            ans+=min(mx,mn);
            ans%=mod;
            updata(instructions[i],1);
        }
        return ans;
    }
};
