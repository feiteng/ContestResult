class Solution {
public:
    int tree[100007];
    int lowbit(int x) {
        return x & (-x);
    }
    void add(int p,int v) {
        while(p <= 100000) {
            tree[p] += v;
            p += lowbit(p);
        }
    }
    int sum(int p) {
        int ans = 0;
        while(p) {
            ans += tree[p];
            p -= lowbit(p);
        }
        return ans;
    }
    int createSortedArray(vector<int>& instructions) {
        const int mod = 1000000007;
        int ans = 0;
        memset(tree,0, sizeof(tree));

        for(int i = 0;i < instructions.size();++i) {
            int x = instructions[i];
            int l = sum(x - 1);
            int r = i - sum(x);
            ans += 1ll * min(l, r);
            ans %= mod;
            add(x, 1);
        }
        return ans;
    }
};
