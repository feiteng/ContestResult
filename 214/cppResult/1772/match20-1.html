class Solution {
public:

    vector<int> bit;
    
    void add(int pos, int val) {
        while (pos < bit.size()){
            bit[pos] += val;
            pos += pos&(-pos);
        }
    }
    
    int sum(int pos) {
        int ans = 0;
        while (pos>0) {
            ans += bit[pos];
            pos -= pos&(-pos);
        }
        return ans;
    }
    
    int createSortedArray(vector<int>& instructions) {
        const long long MOD = 1e9+7;
        bit = vector<int>(1e5+1);

        long long ans = 0;
        for (int x:instructions) {
            int c1 = sum(x-1);
            int c2 = sum(1e5) - sum(x);
            ans += min(c1, c2);
            add(x, 1);
        }
        return ans % MOD;
    }
};
