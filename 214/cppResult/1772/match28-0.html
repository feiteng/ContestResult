const int maxn = 1e5 + 5;
int A[maxn];

int lb(int x) {return x & -x;}
void add(int x)
{
    while(x < maxn)
    {
        A[x] += 1;
        x += lb(x);
    }
}
int sum(int x)
{
    int ans = 0;
    while(x)
    {
        ans += A[x];
        x -= lb(x);
    }
    return ans;
}

class Solution {
public:
    

int createSortedArray(vector<int>& instructions)
{
    memset(A, 0, sizeof(A));
    const int mod = 1e9 + 7;
    int ans = 0, len = (int)instructions.size();
    for(int i = 0; i < len; i++)
    {
        int c = instructions[i];
        ans = (ans + min(sum(c - 1), sum(maxn - 1) - sum(c))) % mod;
        add(c);
    }
    return ans;
}
};
