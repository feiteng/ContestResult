class Solution {
public:
    using ll=long long;
    constexpr static int mod=1e9+7;

    int bit[100005],n=100001;


    void add(int i,int x){
        while(i<=n){
            bit[i]+=x;
            i+=i&-i;
        }
    }

    int sum(int i){
        int ans=0;
        while(i){
            ans+=bit[i];
            i-=i&-i;
        }
        return ans;
    }

    int createSortedArray(vector<int>& instructions) {
        int ans=0;

        for(int x:instructions){
            int a=sum(x-1);
            int b=sum(n)-sum(x);
            ans+=min(a,b);
            ans%=mod;
            add(x,1);
        }
        return ans;
    }
};
