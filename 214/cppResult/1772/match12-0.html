const int N = 1e5 + 10;
    
int C[N];

class Solution {
public:
    const int mod = 1e9 + 7;
    
    int lowbit(int x) {
        return x & (-x);
    }
    
    void update(int x, int k) {
        while (x < N) {
            C[x] += k;
            x += lowbit(x);
        }
    }
    
    int query(int x) {
        int sum = 0;
        while (x) {
            sum += C[x];
            x -= lowbit(x);
        }
        return sum;
    }
    
    int createSortedArray(vector<int>& ins) {
        int res = 0;
        memset(C, 0, sizeof(C));
        for (int x : ins) {
            int l = query(x-1);
            int r = query(N-1) - query(x);
            res += min(l, r);
            res %= mod;
            update(x, 1);
        }
        
        return res;
    }
};
