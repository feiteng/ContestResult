class Solution {
private:
    vector<int>c;
    void add(int x, int val){
        while(x<c.size()){
            c[x]+=val;
            x+=(x&-x);
        }
    }
    int query(int x){
        int ret = 0;
        while(x>0){
            ret+=c[x];
            x-=(x&-x);
        }
        return ret;
    }
public:
    int createSortedArray(vector<int>& a) {
        int n = a.size();
        c = vector<int>(100005,0);
        long long ret =0;
        for(int i=0;i<a.size();i++){
            int tt = a[i];
            int l = query(tt-1);
            int r = query(100004)-query(tt);
            ret +=min(l,r);
            add(tt,1);
        }
        return ret%1000000007;
    }
};
