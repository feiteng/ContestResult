class Solution {
public:
    int C[200010];
    int N;
    int lowbit(int x){
        return x & (-x);
    }
    

    void insert(int pos,int val){
        while(pos <= N){
            C[pos] += val;
            pos += lowbit(pos);
        }   
    }
    
    int query(int x){
        int ans = 0;
        while(x){
            ans += C[x];
            x -= lowbit(x);
        }
        return ans;
    }
    int createSortedArray(vector<int>& instructions) {
        N = 100005;
        

        long long ans = 0,mod = 1e9 + 7;
        
        for(int v : instructions){
            int l = query(v-1);
            
            int r = query(N) - query(v);
            
            // cout << "l : " << l << endl;
            // cout << "r : " << r << endl;
            
            ans = (ans + min(l,r)) % mod;
            insert(v,1);
        }
        return ans;
    }
};
