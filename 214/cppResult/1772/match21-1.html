int c[100001],cnt[100001];
class Solution {
public:

    int lowbit(int x)
    {
        return x&-x;
    }
    void add(int x,int y)
    {
        while(x<=100000)
        {
            c[x]=c[x]+y;
            x=x+lowbit(x);
        }
    }
    int sum(int x)
    {
        int ans=0;
        while(x)
        {
            ans=ans+c[x];
            x=x-lowbit(x);
        }
        return ans;
    }
    int createSortedArray(vector<int>& instructions) {
        int mod=1000000007;
        memset(c,0,sizeof(c));
        memset(cnt,0,sizeof(cnt));

        int n=instructions.size();
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int ls=sum(instructions[i]-1);
            ans=(ans+min(ls,i-ls-cnt[instructions[i]]))%mod;
            add(instructions[i],1);
            cnt[instructions[i]]++;
        }
        return ans;
    }
};
