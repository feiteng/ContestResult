class Solution {
public:
    vector<int> a;
    void init() {
        a = vector<int>(100001, 0);
    }

    int lowbit(int x) {
        return x & (-x);
    }
    void add(int x) {
        while (x < a.size()) {
            a[x]++;
            x += lowbit(x);
        }
    }
    int sum(int x) {
        int ret = 0;
        while (x != 0){
            ret += a[x];
            x -= lowbit(x);
        }
        return ret;
    }
    int createSortedArray(vector<int>& instructions) {

        long long ret = 0;
        init();
        for (int i = 0;i < instructions.size();i++) {
            int x = instructions[i];
            ret += min(sum(x - 1), i - sum(x));
            add(x);
        }
        return ret % 1000000007;
    }
};
