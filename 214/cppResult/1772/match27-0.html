class Solution {
public:
    vector<int> cnt;
    int N;
    inline int lowbit(int x)
    {
        return (x^(x-1))&x;
    }
    
    void update(int k,int x)
    {
        for(int i=k;i<=N;i+=lowbit(i)) cnt[i]+=x;
    }
    
    int getsum(int x)
    {
        int ans=0;
        for(int i=x;i>0;i-=lowbit(i))
            ans+=cnt[i];
        return ans;
    }
    
    int createSortedArray(vector<int>& instructions) {
        N=100001;
        cnt = vector<int>(100010,0);
        long long ret=0;
        int BP=1e9+7;
        
        for (int i=0;i<instructions.size();i++)
        {
            int a=getsum(instructions[i]-1);
            int b=i-getsum(instructions[i]);
            //cout<<a<<' '<<b<<' '<<getsum(instructions[i])<<' '<<min(a,b)<<endl;
            ret=(ret+min(a,b))%BP;
            update(instructions[i],1);
        }
        
        return (int)ret;
    }
};
