class Solution {
public:
    vector<int> a;
    
    int lowbit(int x) {
        return x & (-x);
    }
    
    void update(int x, int v) {
        while (x <= 100000) {
            a[x] += v;
            x += lowbit(x);
        }
    }
    
    int getsum(int x) {
        int res = 0;
        while (x > 0) {
            res += a[x];
            x -= lowbit(x);
        }
        return res;
    }
    
    int createSortedArray(vector<int>& instructions) {
        int res = 0;
        int Mod = 1000000000 + 7;
        a = vector<int>(100000 + 1, 0);
        for (int i = 0; i < instructions.size(); ++ i) {
            int tmp = instructions[i];
            int l = getsum(tmp - 1);
            int r = getsum(100000) - getsum(tmp);
            res = (res + min(l, r)) % Mod;
            update(tmp, 1);
        }
        return res;
    }
};
