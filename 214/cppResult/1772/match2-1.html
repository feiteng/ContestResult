const int MOD = 1e9 + 7;
const int N = 1e5 + 10;

class Solution {
public:

    int m, t[N];
    int lowbit(int x) {
        return x&(-x);
    }
    void upd(int x, int k) {
        while (x <= m) {
            t[x] += k;
            x += lowbit(x);
        }
    }
    int gsum(int x) {
        int res = 0;
        while (x > 0) {
            res += t[x];
            x -= lowbit(x);
        }
        return res;
    }
    
    int createSortedArray(vector<int>& instructions) {
        int ans = 0, n = instructions.size();
        m = 100001;
        memset(t, 0, sizeof t);

        for (int i = 0; i < n; i ++) {
            int cur = instructions[i];
            int t1 = gsum(cur-1);
            int t2 = gsum(m) - gsum(cur);
            (ans += min(t1, t2)) %= MOD;
            upd(cur, 1);
        }
        return ans;
    }
};
