class Solution {
public:
    const int N = 100002, mod = 1e9+7;
    vector<int>c;
    int lowbit(int x){
        return x & (-x);
    }
    int getsum(int x){
        int sum = 0;
        for(int i = x; i; i-= lowbit(i)) sum += c[i];
        return sum;
    }
    void insert(int x, int n){
        for(int i = x; i <= n; i += lowbit(i)) c[i] += 1;
    }
    int createSortedArray(vector<int>& ins) {
        c.resize(N);
        int ans = 0;
        for(int i = 0; i < ins.size(); ++i){
            int x = ins[i];
            // cout<< x << " " << getsum(x-1) <<" " << getsum(x) << endl;
            ans = (ans + min(getsum(x - 1), i - getsum(x))) % mod;
            insert(x, N);
        }
        return ans;
    }
};
