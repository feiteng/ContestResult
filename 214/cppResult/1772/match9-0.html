constexpr int MAXN = 600000+7;
constexpr int MOD = 1e9+7;
class Solution {
public:
    typedef long long ll;
    int n;
     
// 树状数组 代表 a[1]...a[x]
    int c[MAXN]; // 1...n
    int lowbit(int x) {
        // 算出x二进制的从右往左出现第一个1以及这个1之后的那些0组成数的二进制对应的十进制的数
        return x & -x;
    }

    void add(int x, int k) {
        while (x <= n) {  // 不能越界
            c[x] = c[x] + k;
            x = x + lowbit(x);
        }
    }

    int getsum(int x) {  // a[1]……a[x]的和D
        int ans = 0;
        while (x >= 1) {
            ans = ans + c[x];
            x = x - lowbit(x);
        }
        return ans;
    }

    int createSortedArray(vector<int>& instructions) {
        n = 1e5+7;
        ll ans = 0;
        add(instructions[0], 1);
        for (int i = 1; i < instructions.size(); ++i) {
            int v = instructions[i];
            int lowv = getsum(v -1);
            int loweqv = getsum(v);
            ans = (ans + min(lowv, i - loweqv))%MOD;
            add(v, 1);
        }
        return ans;
    }
};
