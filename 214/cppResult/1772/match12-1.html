const int maxn = 1e5 + 10;
const int mod = 1e9 + 7;
class Solution {
public:
    int a[maxn << 2];
    int N = 1e5;
    

    void add(int pos , int val){
        while(pos <= N){
            a[pos] += val;
            pos += pos & (-pos);
        }
    }
    
    
    int get(int pos){
        int ret = 0;
        while(pos){
            ret += a[pos];
            pos -= pos & (-pos);
        }
        return ret;
    }
    
    int createSortedArray(vector<int>& t) {
        int ret = 0;

        for(int i : t){
            int v1 = get(i - 1);
            int v2 = get(N) - get(i);
            ret += min(v1 , v2);
            ret %= mod;
            add(i , 1);
        }
        return ret;
    }
};
