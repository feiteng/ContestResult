const int MAXN = 1e5;
const int MOD = 1e9 + 7;

int cnt[MAXN + 1];

inline int lowbit(int x) {
    return x & (-x);
}

void update(int i) {
    while (i <= MAXN) {
        ++cnt[i];
        i += lowbit(i);
    }
}

int sum(int i) {
    int s = 0;
    while (i > 0) {
        s += cnt[i];
        i -= lowbit(i);
    }
    return s;
}

class Solution {
public:
    int createSortedArray(vector<int>& instructions) {
        int n = static_cast<int>(instructions.size());
        memset(cnt, 0, sizeof(cnt));
        long long cost = 0;
        for (int i = 0; i < n; ++i) {
            int x = instructions[i];
            cost += min(sum(x - 1), sum(MAXN) - sum(x));
            update(x);
        }
        return cost % MOD;
    }
};
