const int N = 1e5+ 10,mod = 1e9 + 7;
int tr[N];
int lowbit(int x){
    return x & -x;
}
int find(int k){
    int ans = 0;
    for(int i = k;i;i -= lowbit(i))
        ans += tr[i];
    return ans;
}
void add(int k,int x){
    for(int i = k;i < N;i += lowbit(i))
        tr[i] += x;
}
class Solution {
public:
    int createSortedArray(vector<int>& instructions) {
        memset(tr,0,sizeof tr);
        int ans = 0;

        int n = instructions.size();
        for(int i = 1;i <= n;i++){
            int x = instructions[i - 1];
            ans = (0ll + ans + min(find(x - 1) ,i - 1 - find(x)))%mod;
            add(x,1);
        }
        return ans;
    }
};
