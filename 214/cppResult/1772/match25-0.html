class BIT
{
    vector<int> arr;
    int N;

public:
    BIT(int n) : N(n), arr(n + 1) {}
    static constexpr int lowbit(int i)
    {
        return i & -i;
    }
    void update(int index, int val)
    {
        while (index <= N)
        {
            arr[index] += val;
            index += lowbit(index);
        }
    }
    int getsum(int index) const
    {
        int result = 0;
        while (index > 0)
        {
            result += arr[index];
            index -= lowbit(index);
        }
        return result;
    }
};

class Solution {
public:
    int createSortedArray(vector<int>& instructions) {
        BIT bit(100000);
        long long ans = 0;
        for(auto& x : instructions)
        {
            ans = (ans + (long long)min(bit.getsum(x-1), bit.getsum(100000) - bit.getsum(x))) % (1000000000 + 7);
            bit.update(x,1);
        }
        return ans;
    }
};
