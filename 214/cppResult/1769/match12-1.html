class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n == 0) return 0;
        vector<int> vec(n+1);
        vec[0] = 0; vec[1] = 1;
        for(int i = 1;i <= n; ++i){
            if(2*i<=n) vec[2*i] = vec[i];
            if(2*i+1<=n) vec[2*i+1] = vec[i]+vec[i+1];
        }
        return *max_element(vec.begin(),vec.end());
    }
};
