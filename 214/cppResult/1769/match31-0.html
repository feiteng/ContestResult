class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n == 0) return 0;
        vector<int> num(n+1, 0);
        num[0] = 0, num[1] = 1;
        int m1 = 0, m2 = 1;
        for(int i = 2; i < n + 1; ++i){
            if(i&1) num[i] = num[i/2]+num[i/2+1];
            else num[i] = num[i/2];
            m2 = max(m2, num[i]);
        }
        return m2;
    }
};
