class Solution {
public:
    int getMaximumGenerated(int n) {
        
        if(n<1)
            return 0;
        vector <int> a(n+5,0);
        a[0] = 0;
        a[1] = 1;
        int ma = 1;
        for(int i=2;i<=n;i++)
        {
            if(i%2)
                a[i] = a[i/2] + a[i/2 + 1];
            else
                a[i] = a[i/2];
            ma = max(ma,a[i]);
        }
        return ma;
    }
};
