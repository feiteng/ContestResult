class Solution {
public:
    int getMaximumGenerated(int n)
    {
        if (n <= 1)
            return n;
        vector<int> arr(n+1, 0);
        arr[0] = 0;
        arr[1] = 1;
        for (int i=1; i<=n; i++)
        {
            if (2*i <= n)
                arr[2*i] = arr[i];
            if (2*i + 1 <= n)
                arr[2*i+1] = arr[i] + arr[i+1];
        }
        return *max_element(arr.begin(), arr.end());
    }
};
