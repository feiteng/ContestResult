class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n <= 0) return 0;
    vector<int> tab(n + 1);
    tab[0] = 0; tab[1] = 1;
    int ans = 1;
    for(int i = 2; i <= n; i++)
    {
        if(i & 1) tab[i] = tab[i / 2] + tab[i / 2 + 1];
        else tab[i] = tab[i / 2];
        ans = max(ans, tab[i]);
    }
    return ans;
    }
};
