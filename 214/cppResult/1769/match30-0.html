class Solution {
public:
    int getMaximumGenerated(int n) {
        if(!n)
            return 0;
        vector<int> a(n+1);
        a[0]=0;
        a[1]=1;
        int i,ans=1;
        for(i=2;i<=n;i++)
        {
            if(i%2)
                a[i] = a[i/2]+a[(i+1)/2];
            else
                a[i] = a[i/2];
            ans = max(ans, a[i]);
        }
        return ans;
    }
};
