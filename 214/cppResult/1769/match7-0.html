class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n <= 1) {
            return n;
        }
        vector<int> a(n+1);
        a[0] = 0;
        a[1] = 1;
        int ans = 1;
        for(int i = 2; i < n+1; i++) {
            if(i%2 == 0) {
                a[i] = a[i/2];
            }
            else {
                a[i] = a[i/2] + a[(i+1)/2];
            }
            ans = max(ans, a[i]);
        }
        return ans;
    }
};
