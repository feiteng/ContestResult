class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n==1){
            return 1;
        }
        if(n==0){
            return 0;
        }
        vector<int>nums(n+1,0);
        nums[0]=0;
        nums[1]=1;
        for(int i=2;i<=n;++i){
            if(i%2==1){
               nums[i]=nums[(i-1)/2]+nums[(i-1)/2+1];
            }
            else{
                nums[i]=nums[i/2];
            }
        }
        sort(nums.begin(),nums.end());
        return nums[n];
    }
};
