class Solution {
public:
    int getMaximumGenerated(int n) {
        if(n == 0) {
            return 0;
        }
        
        vector<int> vec(n + 1, 0);
        vec[0] = 0;
        vec[1] = 1;
        int ans = 1;
        for(int i = 2;i <= n;++i) {
            if(i % 2 == 0) {
                vec[i] = vec[i / 2];
            }
            else {
                vec[i] = vec[i/2] + vec[i / 2 + 1];
            }
            ans = max(ans, vec[i]);
        }
        return ans;
    }
};
