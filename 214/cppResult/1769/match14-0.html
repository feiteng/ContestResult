class Solution {
public:
    int getMaximumGenerated(int n) {
        vector<int>N(n + 3);
        N[0] = 0;
        N[1] = 1;
        for (int i = 2; i <= n + 1; i++) {
            if ((i % 2) == 0) {
                N[i] = N[i/2];
                //cout << "Y" << N[i / 2];
            } else {
                N[i] = N[(i - 1) / 2] + N[(i + 1) / 2];
            }
            //cout << N[i] <<endl;
        }
        int s = 0;
        for (int i = 0; i <= n; i++) s = max(s, N[i]);
        return s;
    }
};
