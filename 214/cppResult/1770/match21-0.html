class Solution {
public:
    int minDeletions(string s) {
        unordered_map<char, int> m;
        for (char c : s) {
            m[c]++;
        }
        
        map<int, int> f;
        for (auto mi : m) {
            f[mi.second]++;
        }
        
        int res = 0;
        for (auto fi = f.rbegin(); fi != f.rend(); fi++) {
            if (fi->first == 0) break;
            if (fi->second > 1) {
                int ff = fi->first;
                res += f[ff] - 1;
                f[ff-1] += f[ff] - 1;
                f[ff] = 0;
            }
        }
        return res;
    }
};
