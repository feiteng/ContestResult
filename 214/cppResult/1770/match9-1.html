class Solution {
public:
    int minDeletions(string s) {
        vector<int> frequency(26 , 0);
        for(char c : s){
            frequency[c - 'a']++;
        }
        vector<int> dp(1e5 + 100 , 0);
        for(int i = 0 ; i < 26 ; i++){
            if(frequency[i])
                dp[frequency[i]]++;
        }
        int ans = 0;
        for(int i = 1e5 + 1 ; i > 0 ; i--){
            if(dp[i] > 1){
                ans += dp[i] - 1;
                dp[i-1] += dp[i] - 1;
                dp[i] = 1;
            }
        }
        return ans;
    }
};
