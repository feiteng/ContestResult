class Solution {
public:
    int minDeletions(string s) {
        vector<int> count(26);
        for(char c : s) {
            count[c-'a']++;
        }
        
        // unordered_map<int, int> countmap;
        // for(int i = 0; i < 26; i++) {
        //     countmap[count[i]]++;
        // }
        
        vector<int> done(26);
        int rem = 26;
        for(int i = 0; i < 26; i++) {
            if(count[i] == 0) {
                done[i] = 1;
                rem--;
            }
        }
        int ans = 0;
        while(rem > 0) {
            // find max
            //cout << "rem = " << rem << endl;
            int curmax = 0;
            int index = -1;
            for(int i = 0; i < 26; i++) {
                if(done[i] == 0 && count[i] > curmax) {
                    curmax = count[i];
                    index = i;
                }
            }
            if(index == -1) {
                return ans;
            }
            //cout << "curmax, index = " << curmax << ", " << index << endl;
            done[index] = 1;
            rem--;
            for(int i = 0; i < 26; i++) {
                if(done[i] == 0 && count[i] == curmax && i != index) {
                    count[i]--;
                    ans++;
                }
            }
        }
        return ans;
    }
};
