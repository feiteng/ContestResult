class Solution {
public:
    int minDeletions(string s) {
        int arr[26];
        memset(arr,0,sizeof(arr));
        for(int i = 0 ; i<s.length(); ++i)
            arr[s[i] - 'a']++;
        
        sort(arr, arr+26);
      
        int res = 0;
        for(int i = 26 - 1 ; i >= 0; --i){   
            if(arr[i] ==0) break;
            for(int j = i - 1; j >= 0; --j)
                if(arr[i] == arr[j]){
                    arr[j] --;
                    res++;
                }
        }
        
        return res;
    }
};
