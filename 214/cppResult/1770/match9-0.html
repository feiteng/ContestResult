class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(30);
        for (char c : s) {
            cnt[c-'a']++;
        }
        vector<int> a(s.length()+5);
        for (int i = 0; i < 30; i++) {
            if (cnt[i] > 0) a[cnt[i]]++;
        }
        
        int res = 0;
        for (int i = s.length()+1; i >= 1; i--) {
            if (a[i] >= 2) {
                res += a[i]-1;
                a[i-1] += a[i]-1;
            }
        }
        
        return res;
    }
};
