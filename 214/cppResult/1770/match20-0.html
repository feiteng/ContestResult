class Solution {
public:
    int minDeletions(string s) {
        vector<int> count(26, 0);
        for (const char& c : s) {
            count[c-'a']++;
        }
        int res = 0;
        sort(count.begin(), count.end());
        for (int i = 24; i >= 0; --i) {
            if (count[i] == 0) break;
            if (count[i+1] == 0) {
                res += count[i];
                count[i] = 0;
                continue;
            }
            if (count[i] >= count[i+1]) {
                res += (count[i] - count[i+1] + 1);
                count[i] = count[i+1] - 1;
            }
            
        }
        return res;
    }
};
