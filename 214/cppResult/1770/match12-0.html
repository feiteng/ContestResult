class Solution {
public:
    int minDeletions(string s) {
        unordered_map<int, int> c;
        for(auto &i : s) {
            c[i - 'a']++;
        }
        vector<int> v;
        for(int i = 0; i < 26; i++) {
            if(c[i] > 0) {
                v.push_back(c[i]);
            }
        }
        sort(v.begin(), v.end());
        int res = 0;
        for(int i = v.size() - 1; i > 0; i--) {
            while(v[i - 1] >= v[i] && v[i - 1] > 0) {
                v[i - 1]--;
                res++;
            }
        }
        return res;
    }
};
