class Solution {
public:
    int minDeletions(string s) {
        vector<int> count(26, 0);
        for (char c: s) {
            count[c - 'a']++;
        }
        
        sort(count.begin(), count.end(), greater<int>());
        int removed = 0;
        for (int i = 1; i < 26; ++i) {
            if (count[i] > 0 && count[i] == count[i-1]) {
                ++removed;
                count[i]--;
                sort(count.begin(), count.end(), greater<int>());
                --i;
            }
        }
        
        return removed;
    }
};
