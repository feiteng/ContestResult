class Solution {
public:
    int minDeletions(string s) {
        vector<int> dp(26, 0);
        for(auto it:s)

        {
            dp[it-'a']++;
        }
        sort(dp.begin(), dp.end());
        int last = 1e5+5;
        int res = 0;
        for(int i=25;i>=0;i--)
        {
            if(dp[i] == 0) break;

            if(dp[i]>=last){
                res += dp[i]-(last-1);
                last = max(1, last -1);
            }
            else
            {
                last = dp[i];
            }
            
        }
        return res;
    }
};
