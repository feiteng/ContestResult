class Solution
{
public:
    int minDeletions(string s)
    {
        vector<int> c(26);
        for (int i = 0; i < s.size(); i++)
            c[s[i] - 'a']++;
        sort(c.begin(), c.end());
        int ret = 0;
        for (int i = 24; i >= 0 && c[i]; i--)
        {
            if (c[i] >= c[i + 1])
            {
                if (c[i + 1])
                {
                    ret += c[i] - c[i + 1] + 1;
                    c[i] = c[i + 1] - 1;                    
                }
                else
                {
                    ret += c[i];
                    c[i] = 0;
                }
            }
        }
        return ret;
    }
};
