bool cmp(int a, int b) {
    return a > b;
}

class Solution {
public:
    int minDeletions(string s) {
        int n = s.length();
        int cnt[26];
        memset(cnt, 0, sizeof(cnt));

        for (int i = 0; i < n; i++) {
            cnt[s[i] - 'a'] += 1;
        }
        sort(cnt, cnt + 26, cmp);
        int ans = 0;
        // printf("%d ", cnt[0]);
        for (int i = 1; i < 26; i++) {
            // printf("%d ", cnt[i]);
            if (cnt[i] > 0 && cnt[i] >= cnt[i - 1]) {
                if (cnt[i-1] > 0) {
                    ans += cnt[i] - cnt[i - 1] + 1;
                    cnt[i] = cnt[i-1] - 1;
                } else {
                    ans += cnt[i];
                    cnt[i] = 0;
                }
            }
        }
        // printf("\n");
        return ans;
    }
};
