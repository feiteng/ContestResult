class Solution {
public:
    int minDeletions(string s) {
        
        unordered_map<char, int> ump;
        for (char c: s) {
            ++ump[c];
        }
        
        map<int, int> mp;
        for (auto& pr: ump) {
            ++mp[pr.second];
        }
        
        // for (auto& pr: mp) {
        //     cout << pr.first << " " << pr.second << endl;
        // }
        
        int ans = 0;
        for (auto it = mp.rbegin(); it != mp.rend(); ++it) {
            if (it->first == 0) break;
            if (it->second == 1) continue;
            int more = it->second - 1;
            
            // cout << more << endl;
            ans += more;
            mp[it->first - 1] += more;
        }
        
        return ans;
    }
};
