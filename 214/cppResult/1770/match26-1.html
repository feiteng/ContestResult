class Solution {
public:
    int minDeletions(string s) {
        vector<int> count(26, 0);
        vector<int> data(s.size() + 1, 0);
        for (char c: s) {
            count[c - 'a']++;
        }
        int ret = 0;
        for (int i = 0;i < 26;i++) {
            if (count[i] != 0) {
                data[count[i]]++;
            }
        }

        for (int i = s.size();i > 0;i--) {
            if (data[i] > 1) {
                ret += data[i] - 1;
                data[i - 1] += data[i] - 1;
            }
        }
        return ret;
    }
};
