class Solution {
public:
    int minDeletions(string s) {
        vector<int> mem(26, 0);
        for(auto c : s)
            mem[c - 'a']++;
        sort(mem.begin(), mem.end());
        int ret = 0;
        for(int i = 24; i >= 0; --i)
        {
            if(mem[i] == 0)continue;
            while(mem[i] && mem[i] >= mem[i + 1])
            {
                ret++;
                mem[i]--;
            }
        }
        return ret;
    }
};
