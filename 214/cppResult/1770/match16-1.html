class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26, 0);
        for (auto c : s) {
            ++cnt[c - 'a'];
        }
        int total = 0;
        for (int k = 0; k < 26; ++k) {
            int maxF = 0;
            for (int i = 0; i < 26; ++i) {
                maxF = max(maxF, cnt[i]);
            }
            if (maxF == 0) {
                break;
            }

            bool first = true;
            for (int i = 0; i < 26; ++i) {
                if (maxF == cnt[i]) {
                    if (first) {
                        cnt[i] = 0;
                        first = false;
                    } else {
                        --cnt[i];
                        ++total;
                    }
                }
            }
        }
        return total;
    }
};
