class Solution {
public:
    int minDeletions(string s) {
        int a[26] = {0};
        
        for(int i = 0; i < s.length(); i++) {
            a[s[i]-'a']++;
        }
        
        sort(a, a+26);
        reverse(a, a+26);
        
        int res = 0;
        for(int i = 0; i < 26; i++) {
            if(a[i] > 0) {
                for(int j = i+1; j < 26; j++) {
                    if(a[j] == a[i]) {
                        a[j]--;
                        res++;
                    }
                }
            }
        }
        
        return res;
    }
};
