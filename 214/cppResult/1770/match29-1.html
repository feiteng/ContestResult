int cnt[30];
class Solution {
public:
    int minDeletions(string s) {
        memset(cnt, 0, sizeof(cnt));

        for (char v: s) cnt[v - 'a'] += 1;
        
        int ans = 0;
        sort(cnt, cnt + 26);
        for (int i = 24; i >= 0; i--){
            if (cnt[i] == 0) continue;
            while(cnt[i] > 0 && cnt[i] >= cnt[i + 1]){ ans += 1; cnt[i] -= 1; }
        }
        
        return ans;
    }
};
