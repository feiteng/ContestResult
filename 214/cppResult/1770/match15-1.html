class Solution {
public:
    int minDeletions(string s) {
        int a[26]={0};

        for(int i=0;i<s.length();++i){
            a[s[i]-'a']++;
        }
        int ans=0;
        for(int i=0;i<26;++i){
            int maxm=-1;
            for(int j=0;j<26;++j){
                maxm=max(a[j],maxm);
            }
            if(maxm<=0)break;

            int flag=0;
            for(int j=0;j<26;++j){
                if(a[j]==maxm){
                    if(flag==0){
                        flag=1;
                        a[j]=-1;
                    }
                    else{
                        a[j]-=1;
                        ans++;
                    }
                }
            }
            
        }
        return ans;
        
    }
};
