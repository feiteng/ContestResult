class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26);
        for (auto c: s) {
            cnt[c-'a'] ++;
        }
        vector<int> tmp;
        for (auto n: cnt) {
            if (n > 0)
                tmp.push_back(n);
        }
        sort(tmp.rbegin(), tmp.rend());

        int ans = 0;
        for (int i = 1; i < tmp.size(); i++) {
            if (tmp[i] >= tmp[i-1]) {
                if (tmp[i-1] == 0) {
                    ans += tmp[i];
                    tmp[i] = 0;
                } else {
                    ans += (tmp[i] - tmp[i-1]) + 1;
                    tmp[i] = tmp[i-1] - 1;
                }
            }
        }
        return ans;
    }
};
