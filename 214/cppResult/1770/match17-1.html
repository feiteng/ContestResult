class Solution {
public:
    int minDeletions(string s) {
        int c[26]; memset(c,0,sizeof(c));
        for (int i = 0; i < s.length(); i ++)
            ++ c[s[i]-'a'];
        sort(c,c+26);
        int ans=0;
        for (int i=25; i>0; i--)
        {
            if (c[i]==0) break;
            for (int j=i;j>0&&c[j-1]==c[i];--j,++ans,--c[j]);
        }
        return ans;
    }
};
