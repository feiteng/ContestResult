class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26, 0);
        for (char c : s) cnt[c-'a']++;
        sort(cnt.rbegin(), cnt.rend());

        int ans = 0;
        for (int i = 1; i < 26 && cnt[i]; ++i) {
            if (cnt[i] >= cnt[i-1]) {
                if (cnt[i-1]) {
                    ans += cnt[i]-cnt[i-1]+1;
                    cnt[i] = cnt[i-1]-1;
                }
                else {
                    ans += cnt[i];
                    cnt[i] = 0;
                }
            }
        }
        return ans;
    }
};
