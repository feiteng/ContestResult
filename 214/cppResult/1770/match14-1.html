class Solution {
public:
    int minDeletions(string s) {
      vector<int> a(26, 0);

      for (char ch : s) {
        a[ch - 'a'] += 1;
      }
      int ans = 0;
      sort(a.begin(), a.end());
      reverse(a.begin(), a.end());
      for (int i = 0; i < 26; i++) {
        if (a[i] > 0) {
          for (int j = i + 1; j < 26; j++) {
            if (a[i] == a[j]) {
              a[j] -= 1;
              ans += 1;
            }
          }
        }
      }
      return ans;
    }
};
