class Solution {
public:
    int minDeletions(string s) {
        vector<int> freq(26, 0);
        for (char c : s) {
            ++freq[c - 'a'];
        }
        
        sort(freq.begin(), freq.end());
        int res = 0;
        
        unordered_set<int> mm;
        for (int num : freq) {
            int tmp = num;
            while (tmp != 0 && mm.count(tmp)) {
                --tmp;
                ++res;
            }
            mm.insert(tmp);
        }
        
        return res;
    }
};
