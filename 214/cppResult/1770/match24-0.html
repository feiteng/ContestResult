class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26,0);
        for(auto x : s){
            cnt[x-'a']++;
        }
        sort(cnt.begin(), cnt.end());
        int result=0;
        for(int i=24; i>=0; i--) {
            if(cnt[i]>=cnt[i+1]) {
                if(cnt[i+1]==0){
                    result+=cnt[i];
                    cnt[i]=0;
                } else {
                    result+=cnt[i]-cnt[i+1]+1;
                    cnt[i]=cnt[i+1]-1;
                }
            }
        }
        
        return result;
    }
};
