class Solution {
public:
    int minDeletions(string s) {
        vector<int> vec(26,0);
        for(auto i:s){
            vec[i-'a']++;
        }
        sort(vec.begin(),vec.end(),greater<int>());
        int ans=0;
        for(int i=1;i<26;i++){
            if(vec[i]>=vec[i-1]){
                ans+=vec[i]-max((vec[i-1]-1),0);
                vec[i]=max((vec[i-1]-1),0);
            }
        }
        return ans;
    }
};
