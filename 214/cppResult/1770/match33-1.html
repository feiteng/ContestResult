class Solution {
public:
    int minDeletions(string s) {
        vector<int> num(26, 0);
        for (auto& it : s) {
            num[it - 'a']++;
        }
        sort(num.begin(), num.end());
        int ret = 0;
        for (int i = 24; i >= 0; i--) {
            int cur = max(0, min(num[i], num[i + 1] - 1));
            ret += num[i] - cur;
            num[i] = cur;
        }
        return ret;
    }
};
