class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26, 0);
        for(auto a : s) cnt[a - 'a']++;
        sort(cnt.begin(), cnt.end());
        int ans = 0;
        for(int i = 25; i >= 0; --i) {
            if(cnt[i] == 0) return ans;    

            for(int j = 1; i - j >= 0; ++j) {
                if(cnt[i - j] == cnt[i]) {
                    cnt[i - j]--;
                    ans++;
                }
            }
        }
        return ans;
    }
};
