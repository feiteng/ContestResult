int cnt[26];

bool cmp(int a, int b) {
    return a > b;
}

class Solution {
public:    
    int minDeletions(string s) {
        for (int i = 0; i < 26; i++) cnt[i] = 0;
        int n = s.size();

        for (int i = 0; i < n; i++)
            cnt[s[i] - 'a']++;
        sort(cnt, cnt + 26, cmp);
        int ans = 0;
        for (int i = 1; i < 26; i++) {
            if (cnt[i] >= cnt[i - 1]) {
                if (cnt[i - 1] == 0) {
                    ans += cnt[i];
                    cnt[i] = 0;
                } else {
                    ans += cnt[i] - (cnt[i - 1] - 1);
                    cnt[i] = cnt[i - 1] - 1;
                }
            }
        }
        return ans;
    }
};
