class Solution {
public:
    int minDeletions(string s) {
        vector<int> f(26, 0);
        for(char c: s) 
            f[c - 'a']++;
        priority_queue<int> pq;
        for(int i = 0; i < 26; i++) {
            if(f[i] > 0)
                pq.push(f[i]);
        }
        int ans = 0;
        while(pq.size() > 1) {
            int top = pq.top();
            pq.pop();
            if(pq.top() == top) {
                ans++;
                if(top > 1)
                    pq.push(top - 1);
            }
        }
        return ans;
    }
};
