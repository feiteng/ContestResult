class Solution {
public:
    int minDeletions(string s) {
        vector<int> a(26);
        for ( char c:s ) a[c-'a']++;
        priority_queue<int> q;
        for ( int i=0; i<26; i++ ) {
            if ( a[i]>0 ) q.push(a[i]);
        }
        int ans = 0;
        while ( q.size()>1 ) {
            int x=q.top();
            q.pop();
            // cout << x << '\n';
            if ( x==q.top() ) {
                ans++;
            if ( x>1 ) q.push(x-1);
            }
        }
        return ans;
    }
};
