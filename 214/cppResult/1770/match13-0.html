class Solution {
public:
    int minDeletions(string s) {
        int sum[26]={0};
        for(int i=0;i<s.length();i++)
            sum[s[i]-'a']++;
        vector<int> record;
        int res=0,size=0;
        for(int i=0;i<26;i++){
            if(sum[i]>0){
                record.push_back(sum[i]);
                size++;
            }
        }
        sort(record.begin(),record.end());
        for(int i=1;i<size;i++){
            if(record[i]==record[i-1]){
                for(int j=i-1;j>=0;j--){
                    if(record[j]==record[j+1]&&record[j]!=0){
                        record[j]--;
                        res++;
                    }
                }
            }
        }
        return res;
    }
};
