class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26, 0);
        for (int i = 0; i < s.size(); ++ i)
            cnt[s[i] - 'a'] ++;
        sort(cnt.begin(), cnt.end());
        int res = 0;
        for (int i = cnt.size() - 2; i >= 0; --i) {
            if (cnt[i] >= cnt[i + 1]) {
                if (cnt[i + 1] == 0) {
                    res += cnt[i];
                    cnt[i] = 0;
                }
                else {
                    res += cnt[i] - cnt[i + 1] + 1;
                    cnt[i] = cnt[i + 1] - 1;
                }
            }
        }
        return res;
    }
};
