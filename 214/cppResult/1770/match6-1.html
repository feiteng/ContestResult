class Solution {
public:
    int minDeletions(string s) {
        int z[30] = {0};

        for (char c: s) z[c-'a'] ++;
        sort(z,z+26);
        
        int res = 0;
        for (int i=1; i<=26; ++i) {
            for (int j=i-1; j>=0; --j) {
                if (z[j]>0 && z[j] == z[j+1]) {
                    --z[j];
                    ++ res;
                }                
            }            
        }
        return res;
    }
};
