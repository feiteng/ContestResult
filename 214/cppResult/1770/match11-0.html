class Solution {
public:
    int minDeletions(string s) {
        vector<int> freq(26, 0);
        for (auto c : s) {
            freq[c-'a']++;
        }
        vector<int> counts;
        for (int idx = 0; idx < 26; idx++) {
            if (freq[idx] > 0) counts.push_back(freq[idx]);
        }
        sort(counts.begin(), counts.end(), greater<int>());
        int ret = 0;
        for (int idx = 1; idx < counts.size(); idx++) {
            while (counts[idx] > 0 && counts[idx] >= counts[idx-1]) {
                ret++;
                counts[idx]--;
            }
        }
        
        return ret;
    }
};
