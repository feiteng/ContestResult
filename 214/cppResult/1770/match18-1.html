class Solution {
public:
    int minDeletions(string s) {
        int a[30] = {0};
        for(int i = 0; i < s.length(); i++) {
            a[s[i]-'a']++;
        }

        map<int, bool> mp;


        int ans = 0;
        for(int i = 0; i < 26; i++) {
            if(a[i] == 0) continue;
            while(a[i] > 0 && mp[a[i]]) {
                a[i]--;
                ans++;
            }

            if(a[i] > 0) mp[a[i]] = true;
        }

        return ans;
    }
};
