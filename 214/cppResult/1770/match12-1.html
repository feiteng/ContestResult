class Solution {
public:
    int minDeletions(string s) {
        vector<int> a(26, 0);

        for (int i = 0; i < s.length(); ++i) {
            a[s[i] - 'a'] += 1;
        }
        vector<int> b;
        b.clear();
        for (int i = 0; i < 26; ++i) {
            if (a[i] > 0) {
                b.push_back(a[i]);
            }
        }
        sort(b.begin(), b.end());
        int n = b.size();
        if (n < 2) {
            return 0;

        }
        int res = 0;
        // for (int i = 0; i < b.size(); ++i) {
        //     cout << b[i] << ' ';
        // }
        // cout << endl;
        for (int i = n - 2; i >= 0; --i) {
            while (b[i] > 0 && b[i] >= b[i + 1]) {
                --b[i];
                ++res;
            }
        }
        return res;

    }
};
