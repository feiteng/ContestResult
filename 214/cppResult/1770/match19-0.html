class Solution {
public:
    int minDeletions(string s) {
        int cnt[26]{};
        for (char c : s)
            cnt[c - 'a']++;
        int res = 0;
        unordered_set<int> hs;
        for (int i = 0; i < 26; i++) {
            if (!cnt[i])
                continue;
            while (hs.count(cnt[i])) {
                res++;
                cnt[i]--;
            }
            if (cnt[i])
                hs.insert(cnt[i]);
        }
        return res;
    }
};
