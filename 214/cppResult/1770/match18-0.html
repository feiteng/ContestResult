class Solution {
public:
    int minDeletions(string s) {
        int ch[30]={0};
        for(int i=0;i<s.size();i++) {
            int cha=s[i]-'a';
            ch[cha]++;
        }
        int ans=0;
        sort(ch,ch+26);
        for(int i=24;i>=0;i--) {
            if (ch[i]==0) continue;
            while (ch[i+1]<=ch[i]&&ch[i]>0) {
                ch[i]--;
                ans++;
            }
        }
        return ans;
    }
};
