class Solution {
public:
    int minDeletions(string s) {
        vector<int> freq(26,0);
        for(char c:s) {
            ++freq[c-'a'];
        }
        // ignore 0's
        vector<int> v;
        for(int x: freq) {
            if(x!=0) v.push_back(x);
        }
        sort(v.begin(), v.end());
        
        //for(int i=0;i<v.size();i++) cout << v[i] << " "; cout << endl;
        
        
        int ans = 0;
        for(int i=v.size()-2;i>=0;i--) {
            if(v[i]>=v[i+1]) {
                ans += min(v[i], v[i]-v[i+1] + 1);
                v[i] = v[i+1]-1;
            }
                
        }
        return ans;
    }
};
