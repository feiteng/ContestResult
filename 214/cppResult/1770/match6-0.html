class Solution {
  public:
    int minDeletions(string s)
    {
      vector<int> counts(26, 0);
      for (auto c: s) ++counts[c - 'a'];
      sort(counts.begin(), counts.end());
      // cout << counts << endl;
      int count = 0;
      for (int i = counts.size() - 2; i >= 0 and counts[i] > 0; --i) {
        for (int j = counts.size() - 1; j > i; --j) {
          if (counts[j] > 0 and counts[i] == counts[j]) {
            --counts[i];
            ++count;
          }
        }
      }
      return count;
    }
};
