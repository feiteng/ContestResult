class Solution {
public:
    int minDeletions(string s) {
        vector<int> v(26);
        for(auto c:s)v[c-'a']++;
        int ans=0;
        for(int i=0;i<26;i++){
            map<int,int> mp;
            for(int j=0;j<26;j++){
                if(v[j])mp[v[j]]++;
            }
            int mx=0,req=0;
            for(auto P:mp){
                if(P.second>mx){
                    mx=max(P.second,mx);
                    req=P.first;
                }
            }
            cout<<mx<<" "<<req<<"\n";
            if(mx<=1)break;
            ans+=mx-1;

            for(int j=0;j<26;j++){
                if(v[j]==req && mx>1){
                    v[j]--;
                    --mx;
                }
            }
        }
        return ans;
    }
};
