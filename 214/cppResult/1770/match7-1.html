class Solution {
public:
    int minDeletions(string s) {
        vector<int> frequencies(26, 0);
        for (char c : s) {
            ++frequencies[c - 'a'];
        }
        int deletions = 0;
        unordered_set<int> taken;
        for (int i = 0; i < 26; ++i) {

            int frequency = frequencies[i];
            while (frequency > 0 && taken.find(frequency) != taken.end()) {
                --frequency;
                ++deletions;
            }
            taken.emplace(frequency);
        }
        return deletions;
    }
};
