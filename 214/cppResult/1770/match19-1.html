class Solution {
public:
    int minDeletions(string s) {
        vector<int>v(26,0);

        for(char c:s)
            v[c-'a']++;
        int ans = 0;
        unordered_set<int>ss;
        for(int i=0;i<26;++i){
            if(v[i]==0)
                continue;
            while(ss.count(v[i])){
                v[i]--;
                ans++;
            }
            if(v[i]!=0)
            ss.insert(v[i]);
        }
        return ans;
    }
};
