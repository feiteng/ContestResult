class Solution {
public:
    int minDeletions(string s) {
        vector<int> nums(26,0);
        vector<int> step(10005,0);
        for(char c:s){
            nums[c-'a']++;
        }
        for(int i:nums){
            if(i != 0){
                step[i]++;
            }
        }
        int ret = 0;
        for(int i = 10000;i>0;i--){
            if(step[i] != 0){
                ret += (step[i]-1);
                step[i-1] += (step[i]-1);
            }
        }
        return ret;
    }
};
