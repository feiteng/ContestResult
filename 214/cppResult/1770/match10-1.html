class Solution {
public:
    int minDeletions(string s) {
        vector<int> cnt(26, 0);
        for (auto c : s) {
            cnt[c - 'a']++;
        }
        int n = s.size();
        vector<int> p(n + 1, 0);
        for (int i = 0; i < 26; ++i) {
            if (cnt[i]) p[cnt[i]]++;
        }
        

        int ans = 0;
        for (int i = n; i >= 1; --i) {
            if (p[i] > 1) {
                ans += (p[i] - 1);
                p[i - 1] += p[i] - 1; 
            }
        }
        return ans;
    }
};
