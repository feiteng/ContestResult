class Solution:
    def createSortedArray(self, ins: List[int]) -> int:
        from sortedcontainers import SortedList
        nums = SortedList()
        res = 0
        for i in ins:
            left = nums.bisect_left(i)
            right = len(nums) - nums.bisect_right(i)
            # print(nums, i)
            # print(left, right)
            res += min(left, right)
            # print('res',res)
            nums.add(i)
        return res % (10**9 + 7)
