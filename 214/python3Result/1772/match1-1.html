from sortedcontainers import SortedList
class Solution:
    def createSortedArray(self, A: List[int]) -> int:
        mod = 10 ** 9 + 7
        
        sl = SortedList()
        
        res = 0
        # print(len(A))
        for x in A:
            left = sl.bisect_left(x)
            right = sl.bisect_right(x)
            # print(left, right, len(sl))
            res += min(left, len(sl)-right)
            sl.add(x)
        
        return res % mod
