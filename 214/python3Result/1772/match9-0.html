import bisect
class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        nums = list()
        cost = 0
        
        for inst in instructions:
            left = bisect.bisect_left(nums, inst)
            right = len(nums) - bisect.bisect_right(nums, inst)
            cost += min(left,right)
            cost %= 1000000007
            bisect.insort(nums,inst)
        return cost
