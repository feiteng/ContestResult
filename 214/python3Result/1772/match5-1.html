class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        import bisect
        ins = instructions
        n = len(ins)
        mod = 1e9 + 7
        res = 0
        ans = []
        for i in range(n):
            m = len(ans)
            l = bisect.bisect_left(ans, ins[i])
            r = m - bisect.bisect_right(ans, ins[i])
            # print("l: ", l)
            # print("r: ", r)
            # print("============")
            res += min(l,r)
            bisect.insort(ans, ins[i])
#         print(ans)
        
#         a = [1,2,2,2,4]
#         print(bisect.bisect_left(a, 2))
#         print(bisect.bisect_right(a, 2))
        
        return int(res % mod)
