class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        from sortedcontainers import SortedList
        sl 
= SortedList()
        res = 0
        for n in instructions:
            l = sl.bisect_left(n)
            r = sl.bisect_right(n)
            res += min(l,len(sl)-r)
            sl.add(n)
        
        return res%(7+10**9)
