MOD = 10 ** 9 + 7

class FenwickTree:
    def __init__(self, N):
        self.N = N
        self.ft = [0] * (self.N+1)
    
    def update(self, i):
        i = i + 1
        while i <= self.N:
            self.ft[i] += 1
            i += i & -i

    def query(self, i):
        ret = 0
        i = i + 1
        while i > 0:
            ret += self.ft[i]
            i -= i & -i
        return ret
    
class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        N = max(instructions)
        ft = FenwickTree(N+1)
        cost = 0
        
        for x in instructions:
            cost_smaller = ft.query(x-1)
            cost_larger = ft.query(N) - ft.query(x)
            cost += min(cost_smaller, cost_larger) % MOD
            cost %= MOD
            ft.update(x)
        
        return cost
