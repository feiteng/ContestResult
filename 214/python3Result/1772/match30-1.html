class Solution:
    
def createSortedArray(self, instructions: List[int]) -> int:
        nums = []
        res = 0
        for n in instructions:
            res += min(bisect.bisect_left(nums, n), len(nums) - bisect.bisect_right(nums, n))

            bisect.insort(nums, n)
        return res % (10**9 + 7)
