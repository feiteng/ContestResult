import bisect
class Solution:
    
def createSortedArray(self, instructions: List[int]) -> int:
        tmp = []
        ans = 0
        for i in instructions:
            ans+=min(bisect.bisect_left(tmp, i), len(tmp)-bisect.bisect_right(tmp, i))
            bisect.insort_right(tmp, i)
        return ans%int(1e9+7)
