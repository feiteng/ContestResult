class BIT:
    #n:要素数
    def __init__(self, n):
        self.n = n
        self.bit =
 [0]*(n+1)
    
    #i番目(0-indexed)の値にxを足す
    def add(self, i, x):
        i += 1
        
        while i<=self.n:
            self.bit[i] += x
            i += i&(-i)

    #sum[0:i)
    def acc(self, i):
        s 
= 0
        
        while i>0:
            s += self.bit[i]
            i -= i&(-i)
        
        return s

class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        ans = 0
        MOD = 10**9+7
        bit = BIT(10**5+1)
        n = len(instructions)
        
        for i in range(n):
            num = instructions[i]
            l 
= bit.acc(num)
            r = bit.acc(10**5+1)-bit.acc(num+1)
            ans += min(l, r)
            #print(min(l, r))
            ans %= MOD
            bit.add(num, 1)
            
        return ans
