from sortedcontainers import SortedList

class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        mod = 10 ** 9 + 7
        s = SortedList()
        
        total = 0
        for x in instructions:
            index_left = s.bisect_left(x)
            index_right = s.bisect_right(x)
            total += min(index_left, len(s) - index_right)
            s.add(x)
            
        return total % mod
