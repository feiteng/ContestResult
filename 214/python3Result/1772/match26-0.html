class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        nums = []
        ans = 0
        for ni in instructions:
            l = bisect.bisect_left(nums, ni)
            r = len(nums) - bisect.bisect_right(nums, ni)
            ans += min(l, r)
            bisect.insort(nums, ni)
        
        return ans%(10**9+7)
