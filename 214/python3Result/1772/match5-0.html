class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        import bisect
        inss = instructions
        n = len(inss)
        mod = 1e9 + 7
        ress = 0
        ans = []
        for i in range(n):
            m = len(ans)
            l = bisect.bisect_left(ans, inss[i])
            r = m - bisect.bisect_right(ans, inss[i])
            ress += min(l,r)
            bisect.insort(ans, inss[i])
        return int(ress % mod)
