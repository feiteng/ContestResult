import bisect

MOD = 10**9 + 7


class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        ret = 0
        nums 
= []
        for i in instructions:
            j = bisect.bisect_left(nums, i)
            k = bisect.bisect_right(nums, i)
            ret += min(j, len(nums) - k)
            nums.insert(k, i)
        return ret % MOD
