class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        ans, mod =0, 10**9 + 7
        nums = []
        for d in instructions:
            left = bisect.bisect_left(nums, d)
            right = bisect.bisect(nums, d)
            # print(left, right, d)
            ans += min(left, len(nums)-right)
            bisect.insort(nums, d)
        return ans % mod
