from sortedcontainers import SortedList

class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        mod = 10**9+7
        arr = SortedList()
        ans = 0
        
        for idx,i in enumerate(instructions):
            b = arr.bisect_left(i)
            br = arr.bisect_right(i)
            ans = (ans+ min(b, idx - br))%mod
            arr.add(i)
            #print(ans)
        
        return ans
