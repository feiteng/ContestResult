class Solution:
    def createSortedArray(self, A: List[int]) -> int:
        
        ans = 0
        nums 
= []
        
        for a in A:
            r_idx = bisect.bisect_right(nums, a)
            l_idx = bisect.bisect_left(nums, a)
            ans += min(r_idx, len(nums) - r_idx, l_idx, len(nums) - l_idx)
            # print(nums, a, idx, min(idx, len(nums) - idx))
            nums.insert(r_idx, a)
        
        return ans % (10 ** 9 + 7)
