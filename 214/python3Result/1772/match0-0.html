class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        res = 0
        MAX = max(instructions)
        size = int(math.sqrt(MAX))
        buckets = [0] * (MAX//size + 1)
        nums = [[] for i in range(MAX//size + 1)]
        for num in instructions:
            key = num//size
            left = bisect.bisect_left(nums[key], num)
            right = bisect.bisect_right(nums[key], num)            
            res += min(sum(buckets[:key])+left, len(nums[key])-right+ sum(buckets[key+1:]))
            nums[key].insert(left, num)
            buckets[key] += 1
        return res%(10**9+7)
