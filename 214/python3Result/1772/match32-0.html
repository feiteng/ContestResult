class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        import bisect
        s = []
        ans = 0
        for i in instructions:
            ans += min(bisect.bisect_left(s, i), len(s) - bisect.bisect_right(s, i))
            bisect.insort(s, i)
        return ans % (10**9+7)
