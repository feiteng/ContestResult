from sortedcontainers import SortedList

class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        
        nums = SortedList()
        cost = 0
        
        for i in instructions:
            l = nums.bisect_left(i)
            r = nums.bisect_right(i)
            cost += min(l, max(len(nums) - r, 0))
            nums.add(i)
            
        return cost % (10 ** 9 + 7)
