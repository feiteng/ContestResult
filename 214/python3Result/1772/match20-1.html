from sortedcontainers import SortedList
class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        nums 
= SortedList()
        cost = 0
        for x in instructions:
            small_count = nums.bisect_left(x)
            large_count = len(nums) - nums.bisect_right(x)
            cost += min(small_count, large_count)
            nums.add(x)
        return cost % (10**9 + 7)
