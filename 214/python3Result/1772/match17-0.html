import bisect
class Solution:
    def createSortedArray(self, instructions: List[int]) -> int:
        nums = []
        res = 0
        for i in instructions:
            res += min(bisect.bisect_left(nums, i), len(nums) - bisect.bisect_right(nums, i))
            bisect.insort(nums, i)
        return res % (10 ** 9 + 7)
