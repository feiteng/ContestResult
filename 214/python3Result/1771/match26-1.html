class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inv = sorted(inventory, reverse = True)
        inv.append(0)
        left 
= 0
        res = 0
        while orders:
            #print("orders, res, inv", orders, res, inv)
            step = (inv[left] - inv[left + 1]) * (left + 1)
            #print("step", step, )
            if orders > step:
                res += ((inv[left] + inv[left+1]+1) * (inv[left] - inv[left + 1]) // 2) * (left + 1)
                orders -= step
                left += 1
            else:
                # column number: left + 1
                nc = left + 1
                flat = orders // nc
                #print("nc, flat", nc, flat, (orders%nc))
                #print(inv[left], inv[left]-flat+1, flat)
                res += ((inv[left] + (inv[left]-flat+1)) * flat // 2) * nc
                res += (orders % nc) * (inv[left]-flat)
                return res % (10**9 + 7)
                
        return res % (10**9 + 7)
