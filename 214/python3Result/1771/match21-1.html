class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        MOD = 1_000_000_007
        # sell each colored balls until each has #keep balls left
        def sell(keep: int) -> int:
            values, count = 0, 0
            for c in inventory:
                num = max(0, c - keep)
                if num:
                    count += num
                    values 
+= ((c + keep+1) * num) // 2
            return count, values
        
        low, hi = 0, max(inventory)
        while low < hi:
            mid = (low+hi)//2
            count, money = sell(mid)
            if count <= orders:
                hi = mid 
            else:
                low = mid+1
        sell_out, money = sell(hi)
        return (money + (orders - sell_out)*hi) % MOD
