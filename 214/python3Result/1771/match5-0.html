class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inventory.sort(reverse=True)
        inventory.append(0)
        s=0
        for i in range(1,len(inventory)):
            if orders>i*(inventory[i-1]-inventory[i]):
                s+=(inventory[i]+inventory[i-1]+1)*(inventory[i-1]-inventory[i])*i//2
                orders-=(inventory[i-1]-inventory[i])*i
            else:
                k=orders//i
                return (s+(2*inventory[i-1]-k+1)*k*i//2+(inventory[i-1]-k)*(orders%i))%1000000007
