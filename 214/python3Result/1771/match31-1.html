class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        
        mod = 10**9+7
        cnt = collections.Counter
(inventory)
        new = [(-k, v) for k, v in cnt.items()]
        heapq.heapify(new)
        ans = 0
        while new and orders > 0:
            n, cn = heapq.heappop(new)
            n = -n
            nxt = -new[0][0] if new else 0
            total = min((n - nxt)*cn, orders)
            p = total//cn
            q = total % cn
            ans += (((2*n+1-p)*p//2)*(cn-q)+((2*n-p)*(p+1)//2)*q) %mod
            orders -= total
            if new and orders > 0:
                newn,ncn = heapq.heappop(new)
                heapq.heappush(new, (newn, ncn+cn))
                
        
        
        return ans%mod
