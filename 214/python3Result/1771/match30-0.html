class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        mod = 10 ** 9 + 7
        left, right = 0, max(inventory)
        counts = lambda i: sum(count - i + 1 for count in inventory if count >= i)
        while left < right:
            mid = left + (right - left) // 2
            total = counts(mid)
            if total > orders:
                left = mid + 1
            else:
                right = mid
        total = counts(left)
        result = sum((count + left) * (count - left + 1) // 2 for count in inventory if count >= left) % mod
        if (diff := orders - total) > 0:
            result += (left - 1) * diff % mod
        return result % mod
