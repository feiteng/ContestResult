class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inventory = sorted(inventory)
        N = len(inventory)
        idx = N - 1
        value = 0
        const = 1000000007
        while idx >= 0:
            cur = inventory[idx]
            prev = inventory[idx-1] if idx > 0 else 0
            length = N - idx
            if (cur - prev) * length <= orders:
                value += (((prev + 1 + cur) * (cur - prev) // 2) * length) 
                orders -= (cur - prev) * length
            else:
                n = orders // length
                remain = orders % length
                value += (length * (cur - n + 1 + cur) * n // 2) 
                orders -= n * length
                value += (remain * (cur - n)) 
                break
            idx -= 1
        return value % const
