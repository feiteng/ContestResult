class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        l, h = 0, max(inventory)
        while l < h:
            mid = (l + h) // 2
            if self.valid(mid, inventory, orders):
                l = mid + 1
            else:
                h = mid
        
        cur, res = 0, 0
        for num in inventory:
            if num > l:
                res += (num + l + 1) * (num - l) // 2
                cur += num - l
            
        return ((orders - cur) * l + res) % (10 ** 9 + 7)
    
    def valid(self, a, nums, cap):
            res = 0
            for num in nums:
                if num > a:
                    res += num - a
            return res >= cap
