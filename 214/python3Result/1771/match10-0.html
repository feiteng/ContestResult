class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        mod = 10 ** 9 + 7
        
        l = 0
        r = 10 ** 9
        
        def check(mid):
            return sum(max(0, i - mid + 1) for i in inventory) >= orders
        
        while l < r:
            mid = (l + r + 1) // 2
            if check(mid):
                l = mid
            else:
                r = mid - 1
        
        res = 0
        for i in inventory:
            if i > l:
                res += (i + l) * (i - l + 1) // 2
                res %= mod
                orders -= i - l + 1
        res += orders * l
        
        return res % mod
