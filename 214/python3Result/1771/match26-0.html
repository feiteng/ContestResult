class Solution:
    def maxProfit(self, arr: List[int], orders: int) -> int:
        arr.sort()
        arr = arr[::-1]
        arr = arr + [0]
        ans = 0
        mod = 10**9+7
        i = 0
        # print(arr)
        while orders > 0:
            # print(ans)
            tmp = (arr[i] - arr[i+1]) * (i+1)
            if orders >= tmp:
                ans += (i+1) * (arr[i+1] + arr[i] + 1) * (arr[i] - arr[i+1]) // 2
                ans %= mod
                orders -= tmp
            else:
                h = i + 1
                w = int(orders / h) 
                leaset = orders - h*w
                ans += h * (arr[i] + arr[i]-w+1) * w // 2
                ans += leaset * (arr[i] - w)
                ans %= mod
                orders = 0
            i += 1
        return ans
