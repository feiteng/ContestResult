class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inventory.sort()
        cur_item, next_item = len(inventory) - 1, len(inventory) - 2
        multiple 
= 1
        total = 0
        while True:
            cur_price = inventory[cur_item]
            next_price = 0 if next_item < 0 else inventory[next_item]
            max_orders = (cur_price - next_price) * multiple
            #print(orders, max_orders, cur_price, next_price)
            if orders - max_orders > 0:
                orders -= max_orders
                total += (cur_price + next_price + 1) * max_orders // 2
                cur_item, next_item = cur_item - 1, next_item - 1
                multiple += 1
            else:
                rnd = orders // multiple
                total += (cur_price + cur_price - rnd + 1) * rnd * multiple // 2
                left = orders % multiple
                total += (cur_price - rnd) * left
                return total % (10**9 + 7)
