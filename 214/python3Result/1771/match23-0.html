class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inventory.sort()
        count=0
        i=len(inventory)-1
        weight=1
        while orders:
            if i!=0 and (inventory[i]-inventory[i-1])*weight<orders:
                count=(count+(inventory[i]+inventory[i-1]+1)*(inventory[i]-inventory[i-1])*weight//2)%1000000007
                orders-=(inventory[i]-inventory[i-1])*weight
                i-=1
                weight+=1
            else:
                shut=orders//weight
                count=(count+shut*weight*(inventory[i]*2-shut+1)//2+(inventory[i]-shut)*(orders%weight))%1000000007
                orders=0
        return count
