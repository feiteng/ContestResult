class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        inventory = sorted(inventory, reverse=True)
        # make sure i+1 alwasys exists
        inventory.append(0)
        multiply = 1
        out = 0
        while orders > 0:
            #
            if (inventory[0] - inventory[1]) * multiply <= orders:
                # just drain it
                out = (out + (inventory[0] + inventory[1]+1) * (inventory[0] - inventory[1])// 2 * multiply ) % 1000000007
                orders -= (inventory[0] - inventory[1]) * multiply
                multiply += 1
                inventory.pop(0)
            else:
                num = orders // multiply
                out = (out + (inventory[0] + inventory[0]-num+1) * num //2 * multiply) % 1000000007
                out = (out + (orders % multiply) * (inventory[0]-num)) % 1000000007
                orders = 0
        return out
