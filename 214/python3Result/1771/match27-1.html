class Solution:
    def maxProfit(self, inv: List[int], orders: int) -> int:
        mod=10**9+7
        def fun(x):
            res=0
            for a in inv:
                if a>=x:
                    res+=a-x+1
            return res
        l=0
        r=10**9
        while l<r:
            mid=(l+r+1)//2
            tmp=fun(mid)
            if tmp>=orders:
                l
=mid
            else:
                r=mid-1
        ans=0
        #print(l)
        for a in inv:
            if a>=l+1:
                ans+=(a+l+1)*(a-l)//2
                ans%=mod
                orders-=a-l
                #print(ans,a,orders)
        ans+=orders*l
        return ans%mod
