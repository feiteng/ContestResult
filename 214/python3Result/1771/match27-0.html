class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        def is_enough(bar):
            result = 0
            for item in inventory:
                result += max(0, item-bar)
            if result >= orders:
                return True
            return False
        left, right = 0, 10**9
        while left < right:
            mid = (right+left)//2
            if not is_enough(mid):
                right = mid
            else:
                left = mid+1
        value = 0
        for item in inventory:
            if item > left:
                count = item - left
                orders -= count
                value += (left + 1 +item)*count//2
        value += orders * left
        return value % 1000000007
