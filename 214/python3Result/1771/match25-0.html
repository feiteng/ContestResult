class Solution:
    def maxProfit(self, nums: List[int], order: int) -> int:
        n = len(nums)
        nums.sort(reverse=True)
        nums.append(0)
        idx, res = 0, 0
        while order >= 0 and idx < n:
            number = (nums[idx] - nums[idx+1])*(idx+1)
            if order >= number:
                temp = (nums[idx] + nums[idx+1]+1)*(nums[idx] - nums[idx+1]) // 2 * (idx+1)
                idx += 1
                order -= number
                res += temp
            else:
                dev = order // (idx+1)
                mod = order % (idx+1)
                # print(dev, mod)
                res += (2*nums[idx] - dev+1) * dev // 2 * (idx+1)
                res += (nums[idx] - dev) * mod
                break
        return res % (10**9+7)
