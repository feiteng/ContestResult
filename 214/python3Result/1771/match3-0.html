class Solution:
    def maxProfit(self, a: List[int], orders: int) -> int:
        a.sort(reverse=True)
        a.append(0)
        s=0
        for i in range(1,len(a)):
            if orders>i*(a[i-1]-a[i]):
                s+=i*(a[i]+a[i-1]+1)*(a[i-1]-a[i])//2
                orders-=i*(a[i-1]-a[i])
            else:
                k=orders//i
                return (s+i*(2*a[i-1]-k+1)*k//2+(a[i-1]-k)*(orders%i))%1000000007
