class Solution:
    def maxProfit(self, a: List[int], b: int) -> int:
        mod = 10 ** 9 + 7
        ans = 0
        a
.sort()
        a = [0] + a
        n = len(a)
        i = n - 1
        while i > 0 and b > 0:
            if (a[i] - a[i - 1]) * (n - i) <= b:
                ans = (ans + (n - i) * (a[i] + (a[i - 1] + 1)) * (a[i] - a[i - 1]) // 2) % mod
                b -= (a[i] - a[i - 1]) * (n - i)
            else:
                x = min(b // (n - i), a[i] - a[i - 1])
                ans = ans + ((n - i) * (a[i] + a[i] - x + 1) * x // 2) % mod
                b %= (n - i)
                if x < a[i] - a[i - 1]:
                    ans = (ans + b * (a[i] - x)) % mod
                    break
            i -= 1
        return ans
