class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        L, R = 0, 10**9
        while L < R:
            mid = (L + R) // 2
            x = sum(max(a - mid, 0) for a in inventory)
            if x > orders:
                L = mid + 1
            else:
                R = mid
        res = sum(max((a - L) * (a + L + 1) // 2, 0) for a in inventory)
        res += (orders - sum(max(a - L, 0) for a in inventory)) * L
        return res % (10**9 + 7)
