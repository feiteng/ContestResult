import heapq

class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        '''
        2,8,4,10,6  20       -> 10,9      -> 19
        2,8,4, 8,6  18       -> 8*2+7*2   -> 30
        2,6,4, 6,6  14       -> 6*3+5*3   -> 33
        2,4,4, 4,4  14-2*3=8 -> 4*4+3*4   -> 24
        2,2,2, 2,2  0
        '''
        
        '''
        
        10(1),8(1),6(1),4(1),2(1)
        
        10-8 (1)
        
        '''
        
        def sum_of(a, b) -> int:
            n = b - a + 1
            
            s = n * (a + b) // 2
            
            return s
        
        freqs = Counter(inventory)
        queue = [(-price, amount) for price,amount in freqs.items()]
        heapq.heapify(queue)
        
        profit = 0
        
        
        while True:
            max_price, amount = heapq.heappop(queue)
            max_price = -max_price
            next_price = queue[0][0] if queue else 0
            next_price = -next_price
            
            possible_to_take = orders // amount
            
            total_items = max_price - next_price
            
            to_take = min(total_items, possible_to_take)
            
            if to_take == 0:
                heapq.heappush(queue, (-max_price, amount))
                break
            
            
            orders -= to_take*amount
            
            last_price = max_price - to_take
            
            
            total = sum_of(last_price+1, max_price)

            
            profit += total*amount
            
            if queue and last_price == -queue[0][0]:
                queue[0] = (queue[0][0], queue[0][1] + amount)
            else:
                heapq.heappush(queue, (-last_price, amount))
            

        if orders > 0:
            max_price, amount = heapq.heappop(queue)
            max_price = -max_price
            
            profit += max_price*orders
            
        MOD = 10**9 + 7
            
            
        return profit % MOD
