class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        ans = 0
        arr = sorted(inventory, reverse = True)
        arr.append(0)
        for i in range(1, len(arr)):
            if i*(arr[i-1]-arr[i])<=orders:
                ans += (arr[i]+1+arr[i-1])*(arr[i-1]-arr[i])//2*i
                orders -= i*(arr[i-1]-arr[i])
            else:
                x = orders//i
                ans += (2*arr[i-1]-x+1)*x//2*i
                orders -= i*x
                ans += orders*(arr[i-1]-x)
                orders = 0
                break
        return ans % 1000000007
