class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        
        mod = 10 ** 9 + 7
        
        def enough(x):
            ans = 0
            for inv in inventory:
                if inv > x:
                    ans += inv - x
            return ans >= orders
        
        l, r = 0, max(inventory)
        while l < r:
            mid = (l + r) // 2
            if enough
(mid):
                l = mid + 1
            else:
                r = mid
                
        # print(l)
                
        cnt, res = 0, 0
        for inv in inventory:
            if inv > l:
                cnt += inv - l
                res += ((inv + l + 1) * (inv - l) // 2) % mod
            
        # print(cnt, res)
            
        return ((orders - cnt) * l + res) % mod
