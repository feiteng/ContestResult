class Solution:
    def maxProfit(self, a: List[int], c: int) -> int:
        s
=0
        b=1000000001
        while s<b:
            v=0
            m=(s+b+1)//2
            for x in a:
                if m<x:
                    v+=x-m
            if v>=c:
                s=m
            else:
                b=m-1
        print(b)
        r=0
        o=0
        for x in a:
            if x>(b+1):
                r+=(b+2+x)*(x-b-1)//2
                o+=x-b-1
        r=r+(c-o)*(b+1)
        return r%1000000007
