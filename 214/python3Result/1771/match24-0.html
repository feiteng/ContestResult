class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        MAXN = 1000000007
        a = sorted(inventory)
        l = 0
        r = 1000000001
        while l <= r:
            m = (l + r) // 2
            s = 0
            for ele in a:
                if ele > m:
                    s += ele - m
            if s >= orders:
                l = m + 1
            else:
                r = m - 1
        k = l
        ans = 0
        for ele in a:
            if ele > k:
                ans += (ele - k) * (ele + k + 1) // 2
                ans = ans % MAXN
                orders -= ele - k
        ans += orders * k
        ans = ans % MAXN
        return ans
