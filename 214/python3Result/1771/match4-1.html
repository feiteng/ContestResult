class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        n = len(inventory)
        inventory.sort(key = lambda x: -x)
        res = 0
        mod = 10**9+7
        for i in range(n-1):
            count = i+1
            diff = inventory[i] - inventory[i+1] 
            num = count * diff
            if num > orders:
                base = inventory[i]
                diff = orders// count
                res += (count*((2*base-diff+1)*diff)//2) % mod
                orders -= diff*count
                base -= diff
                res += orders*base
                return res%mod
            else:
                orders -= num
                res += (count*((inventory[i]+inventory[i+1]+1)*diff)//2) % mod
        if orders > 0:
            base = inventory[-1]
            count = n
            diff = orders// count
            res += (count*((2*base-diff+1)*diff)//2) % mod
            orders -= diff*count
            base -= diff
            res += orders*base
        return res%mod
