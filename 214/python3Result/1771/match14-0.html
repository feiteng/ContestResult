from typing import List


class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        l,r = 0,max(inventory) + 1
        def cal(v) -> bool:
            res = 0
            for i in inventory:
                if i > v:
                    res += i-v
                if res > orders:
                    return False
            return True
        while l < r:
            mid = (l+r) >> 1
            # print(l,mid,r)
            if cal(mid):
                r = mid
            else:
                l = mid + 1
        # print(r)
        res = 0
        for i in inventory:
            if i > r:
                res += (i + r + 1) * (i-r) // 2
                # print(res)
                orders -= (i-r)
        # print(orders)
        res += orders * r
        return res % 1000000007
