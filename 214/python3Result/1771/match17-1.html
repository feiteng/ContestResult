class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        l,r=0, max(inventory)
        mod 
= 10**9+7
        while l<r:
            mid = (l+r+1)>>1
            s = sum(max(x-mid,0) for x in inventory)
            if s>=orders:
                l = mid
            else:
                r = mid-1
        # print(l)
        s = sum(max(x-l,0) for x in inventory)
        ans = 0
        for c in inventory:
            if c>l:
                ans+= (l+1+c)*(c-l)//2
        ans -= (s-orders)*(l+1)
        return ans%mod
