class Solution:
    def maxProfit(self, inventory: List[int], orders: int) -> int:
        def canfillorder(i):
            val = sum(max(0,e-i+1) for e in inventory)
            #print(val - orders)
            return val >= orders
        i, j = 0, max(inventory)+1
        while i != j:
            mid = (i+j)//2
            if canfillorder(mid): 
                i = mid+1
            else:
                j = mid
        th = j
        #print(th,orders,canfillorder(th))
        ans = 0
        for e in inventory:
            cnt = max(0,e-th+1)
            orders -= cnt
            ans = (ans + (e+th)*cnt//2)%(10**9+7)
        #print('---',orders)
        return (ans + (th-1)*orders)%(10**9+7)
