class Solution:
    def maxProfit(self, iv: List[int], od: int) -> int:
        
        def ok(x) :
            r = 0
            s = 0
            t = 0
            for v in iv :
                u = max(0,v - x + 1)
                r += u
                s += (v+x)*u//2
                if u > 0 :
                    t += 1
            return r,s,t
        
        lb,ub = 1,int
(1e9+1)
        while lb+1<ub :
            mb = (lb+ub)//2
            u,v,w = ok(mb)
            if u >= od :
                lb = mb
            else :
                ub = mb
        u,v,w = ok(lb)
        
        r = v - (u - od) * lb
        M = int(1e9+7)
        return r % M
