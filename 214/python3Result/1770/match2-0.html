class Solution:
    def minDeletions(self, s: str) -> int:
        counts = collections.Counter(s)
        values = list(counts.values())
        values.sort(key= lambda x: -x)        
        vset = set()
        change = []
        for v in values:
            if v in vset:
                change.append(v)
            else:
                vset.add(v)
        rst = 0
        for v in change:
            c = v-1
            while c > 0 and c in vset:
                c -= 1
            if c > 0:
                vset.add(c)
            rst += v-c
        return rst
