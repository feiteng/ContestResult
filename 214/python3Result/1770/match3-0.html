class Solution:
    def minDeletions(self, s: str) -> int:
        d,b={},[]
        for c in s: d[c]=d.get(c,0)+1
        a=sorted(d.values())
        a=a[::-1]
        for x in a:
            if not b or x<b[-1]: b.append(x)
            else: b.append(b[-1]-1)
        return sum(a)-sum([max(x,0) for x in b])
