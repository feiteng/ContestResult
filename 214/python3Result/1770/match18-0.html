class Solution:
    def minDeletions(self, s: str) -> int:
        counter = collections.Counter(s)
        fres = sorted(counter.values(), reverse=True)
        
        used_set = set()
        to_ret = 0
        for kt in fres :
            while kt in used_set and kt > 0 :
                to_ret += 1
                kt -= 1
            used_set.add(kt)
        return to_ret
