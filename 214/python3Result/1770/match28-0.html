class Solution:
    def minDeletions(self, s: str) -> int:
        d = collections.Counter(s)
        res = 0
        cs = set()
        for key in d:
            while d[key] in cs and d[key] != 0:
                d[key] -= 1
                res += 1
            cs.add(d[key])            
        
        return res
