class Solution:
    def minDeletions(self, s: str) -> int:
        cnt = collections.Counter(s)
        res = 0
        ls = [cnt[c] for c in cnt]
        ls.sort
(reverse = True)
        
        #print(ls)
        
        for i in range(1, len(ls)):
            if ls[i] >= ls[i-1]:
                res += ls[i] - max(0, (ls[i-1]-1))
                ls[i] = max(0, (ls[i-1]-1))
            #print(ls, res)
        return res
