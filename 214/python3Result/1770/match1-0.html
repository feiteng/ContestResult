class Solution:
    def minDeletions(self, s: str) -> int:
        from collections import Counter
        counter = Counter(s)
        dic = {}
        res = 0
        for letter, count in counter.items():
            if count in dic:
                c = count
                while c > 0:
                    if c not in dic:
                        dic[c] = letter
                        res += count - c
                        break
                    c -= 1
                if c == 0:
                    res += count
            else:
                dic[count] = letter
        return res
