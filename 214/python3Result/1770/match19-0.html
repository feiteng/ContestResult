class Solution:
    def minDeletions(self, s: str) -> int:
        d = {}
        for c in s:
            if c not in d:
                d[c] = 1
            else:
                d[c] += 1
        f = list(d.values())
        f.sort(reverse = True)
        #print(f)
        cost = 0
        for i in range(1,len(f)):
            if f[i] >= f[i-1]:
                if f[i-1] == 0:
                    cost += f[i]
                    f[i] = 0
                else:
                    cost = cost + f[i] - (f[i-1]-1)
                    f[i] = f[i-1] -1
            #print(f)
        #print(1111111)
        return cost
