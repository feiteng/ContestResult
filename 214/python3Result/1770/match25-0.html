class Solution:
    def minDeletions(self, s: str) -> int:
        cnt = [0] * 26
        for x in s:
            cnt[ord(x)-ord("a")] += 1
        cnt.sort()
        result = 0
        for i in range(len(cnt)-2, -1, -1):
            if cnt[i+1] == 0:
                result += cnt[i]
                cnt[i] = 0
            elif cnt[i] >= cnt[i+1]:
                result += cnt[i] - cnt[i+1] + 1
                cnt[i] = cnt[i+1] - 1
        return result
