class Solution:
    def minDeletions(self, s: str) -> int:
        count = {}
        for x in s:
            if x not in count:
                count[x] = 0
            count[x] += 1
        vs = sorted(count.values(), reverse=True)
        if len(vs) == 1:
            return 0
        r 
= 0
        for i in range(1, len(vs)):
            if vs[i] >= vs[i-1]:
                r += vs[i] - max(0, vs[i-1] - 1)
                vs[i] = max(0, vs[i-1] - 1)
        return r
