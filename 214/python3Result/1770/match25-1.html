class Solution:
    def minDeletions(self, s: str) -> int:
        c = Counter(s)
        vals = list(c.values())
        vals.sort(reverse = True)
        # print(vals)
        res 
= 0
        for i in range(1, len(vals)):
            if vals[i-1] == 0:
                res += vals[i]
                vals[i] = 0
            elif vals[i] >= vals[i-1]:
                res += vals[i] - (vals[i-1] - 1)
                vals[i] = vals[i-1] - 1
        return res
