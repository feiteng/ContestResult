class Solution:
    def minDeletions(self, s: str) -> int:
        dic = {}
        for i in s:
            if i not in dic:
                dic[i] = 0
            dic[i] += 1
        tem = [dic[i] for i in dic]
        tem
.sort(reverse = True)
        re = 0
        for i in range(1,len(tem)):
            if tem[i] >= tem[i-1]:
                if tem[i - 1] == 0:
                    re += tem[i]
                    tem[i] = 0
                else:
                    re += (tem[i] - tem[i-1] + 1)
                    tem[i] = tem[i - 1] - 1

        return re
