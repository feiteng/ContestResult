from collections import Counter
class Solution:
    def minDeletions(self, s: str) -> int:
        ca=Counter(s)
        fin=set()
        ans=0
        for a in sorted(ca.values()):
            cur=a
            while cur and cur in fin:cur-=1
            ans+=a-cur
            if cur:fin.add(cur)
        return ans
