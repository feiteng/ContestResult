class Solution:
    def minDeletions(self, s: str) -> int:
        c = Counter(s)
        s = set()
        ans = 0
        for k, v in c.items():
            while v in s and v > 0:
                v -= 1
                ans += 1
            if v > 0:
                s.add(v)
        return ans
