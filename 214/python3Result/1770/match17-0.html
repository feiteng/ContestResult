class Solution:
    def minDeletions(self, s: str) -> int:
        c=collections.Counter(s)
        xs=list(c.values())
        v=set()
        a=0
        for x in xs:
            while x>0 and x in v:
                x-=1
                a+=1
            v.add(x)
        return a
