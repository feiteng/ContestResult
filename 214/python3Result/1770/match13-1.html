from collections import Counter

class Solution:
    def minDeletions(self, s: str) -> int:
        cnt = Counter(s)
        seen = set()
        ans = 0
        for key, val in cnt.items():
            while val in seen and val != 0:
                ans += 1
                val -= 1
                
            seen.add(val)
        # print(cnt, seen)
        return ans
