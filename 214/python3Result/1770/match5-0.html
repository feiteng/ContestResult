class Solution:
    def minDeletions(self, s: str) -> int:
        char2freq = collections.Counter(s)
        freq2char = set()
        ans = 0
        for c, f in char2freq.items():
            while f > 0 and f in freq2char:
                ans += 1
                f -= 1
            freq2char.add(f)
        return ans
