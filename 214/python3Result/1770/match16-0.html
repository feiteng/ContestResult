class Solution:
    def minDeletions(self, s: str) -> int:
        d = {}
        for c in s:
            d[c] = d.get(c, 0) + 1
            
        cd = set()
        res = 0
        for c, v in d.items():
            while v > 0 and v in cd:
                v -= 1
                res += 1
            cd.add(v)
        return res
