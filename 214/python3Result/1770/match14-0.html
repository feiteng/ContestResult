class Solution:
    def minDeletions(self, s: str) -> int:
        dic = Counter(s)
        dic_2 = set()
        res = 0
        
        for key, val in dic.items():
            if val not in dic_2:
                dic_2.add(val)
            else:
                while val in dic_2 and val > 0:
                    res += 1
                    val -= 1
                if val != 0:
                    dic_2.add(val)
        return res
