class Solution:
    def minDeletions(self, s: str) -> int:
        d = {}
        for ch in s:
            if ch not in d:
                d[ch] = 1
            else:
                d[ch] += 1
        hash_set 
= set()
        res = 0
        for key, val in d.items():
            while val > 0 and val in hash_set:
                res += 1
                val -= 1
            if val > 0:
                hash_set.add(val)
        return res
