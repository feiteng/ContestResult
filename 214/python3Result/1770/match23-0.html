class Solution:
    def minDeletions(self, s: str) -> int:
        char_counter = Counter(s)
        deletions = 0
        seen = set()
        for cnt in sorted(char_counter.values(), reverse=True):
            while cnt in seen:
                deletions += 1
                cnt -= 1
            if cnt > 0:
                seen.add(cnt)
        return deletions
