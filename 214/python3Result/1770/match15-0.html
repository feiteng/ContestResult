from collections import Counter
class Solution:
    def minDeletions(self, s: str) -> int:
        if not s:
            return 0
        c = Counter(s)
        se = set()
        res = 0
        for count in c.values():
            while count != 0 and count in se:
                count -= 1
                res += 1
            se.add(count)
        return res
