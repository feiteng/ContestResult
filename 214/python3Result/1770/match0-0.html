class Solution:
    def minDeletions(self, s: str) -> int:
        from collections import Counter
        c = Counter(s)
        e = sorted(list(c.values()))[::-1]
        j = 0
        ans = 0
        m = max(e)
        for j in range(m,0,-1):
            idx = [i for i,x in enumerate(e) if x==j]
            if idx != []:
                ans += (len(idx) - 1)
                for i in idx[:-1]:
                    e[i] -= 1
        return ans
