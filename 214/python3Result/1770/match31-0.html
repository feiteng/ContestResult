from collections import Counter
class Solution:
    def minDeletions(self, s: str) -> int:
        char_count = Counter(s)
        values = char_count.values()
        ans = 0
        value_set = set()
        for v in values:
            while v in value_set and v != 0:
                v -= 1
                ans += 1
            if v != 0:
                value_set.add(v)

        return ans
