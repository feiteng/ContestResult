class Solution:
    def minDeletions(self, s: str) -> int:
        freq = list(sorted(collections.Counter(s).values(), reverse=True))
        print(freq)
        ans = 0
        for i in range(1, len(freq)):
            if freq[i] >= freq[i-1]:
                ans += freq[i] - max(freq[i-1] - 1, 0)
                freq[i] = max(freq[i-1] - 1, 0)
        return ans
