class Solution:
    def getMaximumGenerated(self, n: int) -> int:
        if n==0:
            return 0
        re = [0]*(n+1)
        re[1] = 1
        for i in range(2,n+1):
            if i%2==0:
                re[i] = re[i//2]
            else:
                re[i] = re[(i-1)//2]+re[(i+1)//2]
         
        return max(re)
