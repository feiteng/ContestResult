class Solution:
    def getMaximumGenerated(self, n: int) -> int:
        if n == 0: return 0
        if n == 1: return 1
        nums = [0, 1]
        ans = 1
        for i in range(2, n + 1):
            j = i // 2
            
if i & 1:
                nums.append(nums[j] + nums[j + 1])
            else:
                nums.append(nums[j])
            ans = max(ans, nums[-1])
        return ans
