class Solution {
    public int maxProfit(int[] inventory, int orders) {
        int MOD = 1_000_000_007;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for(int x : inventory){
            map.put(x, map.getOrDefault(x,0)+1);
        }
        long ans =0;
        while(orders>0){
            Integer last = map.lastKey();
            Integer next = map.lowerKey(last);
            if(next==null) next =0;
            
            int f =map.get(last);
            int cover =(last-next)*f;
            if(orders>=cover){
                ans=(ans+(long)f*((long)next+1+last)*((long)last-next)/2)%MOD;
                
                orders-=cover;
                map.remove(last);
               if(next>0) map.put(next, map.get(next)+f);
            }else{
                int k= orders/f;
                int remain =orders%f;
    
                ans=(ans+(long)f*((long)last+last-k+1)*k/2 +((long)remain*((long)last-k)))%MOD;
                orders=0;
            }
        }

        return (int)ans;
    }
}
