class Solution {
    public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        
int l = 0, r = (int) 1e9, n = inventory.length;
        long MOD = (long) 1e9 + 7;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (getCnt(inventory, mid) > orders) l = mid + 1;
            else r = mid;
        }
        long res = 0;
        for (int i = n - 1; i >= 0 && orders > 0 && inventory[i] > l; --i) {
            res = (res + 1L * (l + 1 + inventory[i]) * (inventory[i] - l) / 2) % MOD;
            orders -= (inventory[i] - l);
        
}
        res = (res + 1L * orders * l) % MOD;
        return (int) res;
    }
    long getCnt(int[] arr, int guess) {
        int n = arr.length;
        long res = 0;
        for (int i = n - 1; i >= 0 && arr[i] > guess; --i) {
            res += (arr[i] - guess);
        }
        return res;
    }
}
