class Solution {
    public int maxProfit(int[] inventory, int orders) {
        PriorityQueue<Integer> maxQ = new PriorityQueue<Integer>(Comparator.reverseOrder());
        for(int  iv : inventory) {
            maxQ.offer(iv);
        }
        
        long ans = 0;
        int freq = 1;
        while(orders>0) {
            int max = maxQ.poll();
            int next = maxQ.isEmpty() ? 0 : maxQ.peek();
            
            if(orders >= freq * (max - next)) {
                ans += (long)freq * (max+next+1) * (max-next) / 2;
                orders -= freq * (max-next);
                freq++;
            } else {
                int k = orders/freq;
                next = max - k;
                ans += (long)freq * (max+next+1) * (max-next) / 2;
                orders -= freq * (max-next);
                ans += (long)next * (orders);
                orders = 0;
            }
        }
        
        
        return (int)(ans%1_000_000_007);
    }
}
