class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long mod = 1000000007;
        TreeMap<Long,Long> map = new TreeMap<>();
        for(int inv : inventory){
            map.put((long)inv,map.getOrDefault((long)inv,0l) + 1l);
        }
        map.put(0l,0l);
        Map.Entry<Long,Long> cur = map.lastEntry();
        long ret = 0;
        while (orders > 0){
            Map.Entry<Long,Long> next = map.lowerEntry(cur.getKey());

            long gap = (next.getKey() + 1 + cur.getKey()) * (cur.getKey() - next.getKey()) / 2 * cur.getValue();
            long moves = (cur.getKey() - next.getKey()) * cur.getValue();
            if(moves <= orders){
                ret += gap % mod;
                map.remove(cur.getKey());
                map.put(next.getKey(),next.getValue() + cur.getValue());
                cur = map.lastEntry();
                orders -= moves;
            }else{
                long terms = orders / cur.getValue();
                long got = (cur.getKey() - terms + 1l + cur.getKey()) * terms / 2 * cur.getValue();
                got %= mod;
                got += orders % cur.getValue() * (cur.getKey() - terms);
                got %= mod;
                ret += got;
                
orders = 0;
            }
            ret %= mod;
        }
        return (int)ret;
    }
}
