class Solution {
    int modulo = 1_000_000_007;
	
	public int maxProfit(int[] inventory, int orders) {
        long result = 0;
        TreeMap<Long, Integer> cntMap = new TreeMap<>();
        for (long i : inventory) {
        	cntMap.put(i, cntMap.getOrDefault(i, 0) + 1);
        }
        while (orders > 0) {
        	Map.Entry<Long, Integer> highest = cntMap.lastEntry();
        	long key = highest.getKey();
        	int val = highest.getValue();
        	long nextKey = cntMap.size() == 1 ? 0 : cntMap.floorKey(key - 1);
        	if (orders > val * (key - nextKey)) {
        		orders -= val * (key - nextKey);
        		result += val * ((key + nextKey + 1)) * (key - nextKey) / 2;
        		result = result % modulo;
        		cntMap.remove(key);
        		cntMap.put(nextKey, cntMap.getOrDefault(nextKey, 0) + val);
        	} else {
        		int item = orders / val;
        		result += ((key + key - item + 1)) * item * val / 2;
        		result = result % modulo;
        		result += (key - item) * (orders % val);
        		result = result % modulo;
        		break;
        	}
        }
        return (int) result % modulo;
    }
}
