class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long mod = (long)1e9+7;
        int n = inventory.length;
        Arrays.sort(inventory);
        
int lo=0, hi = inventory[n-1];
        while(lo+1<hi){
            long temp = 0;
            int mid = (hi+lo)/2;
            for(int cur:inventory){
                temp += Math.max(0, cur-mid);
            }
            if(temp >= orders) lo = mid;
            else hi = mid;
        }
        long ans = 0, now = 0;
        for(int i=0; i<n; i++){
            long all = inventory[i];
            if(all<=lo) continue;
            ans += (all+1)*all/2;
            ans -= ((long)lo+1)*lo/2;
            ans = (ans+mod)%mod;
            now += all-lo;
        }
        for(int i=0; i<n&&now>orders; i++){
            if(inventory[i] <= lo) continue;
            now --;
            ans -= lo+1;
            ans = (ans+mod)%mod;
        }
        
        
        
        ans %= mod;
        return (int)ans;
    }
}
