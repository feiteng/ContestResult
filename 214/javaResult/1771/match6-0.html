class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long lo = 0;
        long hi = 1000000000;
        while (lo < hi) {
            long mid = (lo + hi) / 2;
            long sum = 0;
            for (int i : inventory) {
                sum += Math.max(0, i - mid);
            }
            if (sum > orders) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        long count = 0;
        long out = 0;
        for (int i : inventory) {
            long l = i;
            if (l > lo) {
                out += (l * (l + 1) - lo * (lo + 1)) / 2;
                out %= 1000000000 + 7;
                count += l - lo;
            }
        }
        out += (orders - count) * lo;
        out %= 1000000000 + 7;
        return (int) out;
    }
}
