class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long allvalue=0;
        int allcheck=1000000007;
        Arrays.sort(inventory);
        int e=1;
        for(int i=inventory.length-1;i>0;i--){
            if(inventory[i]==inventory[i-1]){
                e++;
            }else{
                break;
            }
        }
        while(orders>0){
            if(e==inventory.length){
                int size=orders/e;
                int other=orders%e;
                allvalue+=((long)inventory[inventory.length-1]+(long)inventory[inventory.length-1]-size+1)*size/2*e;
                allvalue+=((long)inventory[inventory.length-1]-size)*other;
                break;
            }
            int minus=inventory[inventory.length-1]-inventory[inventory.length-e-1];
            int size=Math.min(orders/e,minus);
            if(size==0){
                int other=orders%e;
                allvalue+=((long)inventory[inventory.length-1]-size)*other;
                break;
            }
            allvalue+=((long)inventory[inventory.length-1]+(long)inventory[inventory.length-1]-size+1)*size/2*e;
            for(int i=inventory.length-1;i>=inventory.length-e;i--){
                inventory[i]=inventory[inventory.length-1]-size;
            }
            orders-=e*size;
            for(int i=inventory.length-e-1;i>=0;i--){
                if(inventory[i]==inventory[inventory.length-1]){
                    e++;
                }else{
                    break;
                }
            }
            
        }
        return allvalue>allcheck?(int)(allvalue%allcheck):(int)allvalue;
    }
}
