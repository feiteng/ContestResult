class Solution {
    static final int MOD = 1_000_000_007;
    
    public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        // System.out.println(Arrays.toString(inventory));
        int len = inventory.length;
        long res = 0;
        for (int i = len - 1; i > 0; i--) {
            long diff = inventory[i] - inventory[i - 1];
            if (orders < diff * (len - i)) {
                long avg = orders / (len - i);
                res += (len - i) * helper(inventory[i] - avg, inventory[i]);
                res += orders % (len - i) * (inventory[i] - avg);
                orders = 0;
                break;
            } else {
                orders -= diff * (len - i);
                res += (len - i) * helper(inventory[i - 1], inventory[i]);
            }
            res %= MOD;
        }
        if (orders > 0) {
            int k = inventory[0];
            long avg = orders / len;
            res += len * helper(k - avg, k);
            res += orders % len * (k - avg);
        }
        return (int) (res % MOD);
    }
    
    private long helper(long a, long b) {
        return (long) (a + 1 + b) * (b - a) / 2 % MOD;
    }
}
