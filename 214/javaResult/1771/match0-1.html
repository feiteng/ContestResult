class Solution {
    long MOD = (long)1e9 + 7;
    public int maxProfit(int[] inv, int orders) {
        int len = inv.length;
        Arrays.sort(inv);

        long l = 0;
        long r = inv[len-1];
        while (l < r) {
            long mid = (l+r)/2;
            long num = 0;
            for (int i=len-1;i>=0 && inv[i] >= mid;i--) {
                num+=inv[i] - mid;
            
}
            if (num > orders) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        long ans = 0;
        int num = 0;
        for (int i = len-1; i >= 0 && inv[i] > l; i--) {
            ans += (inv[i] + l+1)*(inv[i]-l)/2L;
            ans = ans%MOD;
            num += (inv[i]-l);
            inv[i] = (int)l;
        }

        ans += (orders - num) * l;
        return (int)(ans%MOD);

    }
}
