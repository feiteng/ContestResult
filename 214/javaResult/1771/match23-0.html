class Solution {
    public int maxProfit(int[] inv, int ord) {
        int n = inv.length;
        int min = 0;
        int max = 1000000000;
        while (min<max) {
            int mid = (min+max)/2;
            long num = 0;
            for (int x : inv) {
                num += Math.max(0,x-mid);
            }
            if (num<=ord)
                max=mid;
            else
                min=mid+1;
        }
        long ans = 0;
        for (int i = 0; i < n; ++i) {
            long ct = Math.max(0,inv[i]-min);
            long ta = ct*min;
            ta += ct*(ct+1)/2;
            ans += ta;
            ord -= (int)ct;
        }
        //System.out.println(min+" "+ord);
        while (ord>0) {
            ans += min;
            --ord;
        }
        return (int)(ans%1000000007);
    }
}
