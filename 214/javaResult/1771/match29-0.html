class Solution {
    public int maxProfit(int[] inventory, int orders) {
        int n = inventory.length;
        Queue<Integer> pq = new PriorityQueue<>((a,b)->{
            return b-a;
        });
        for(int e:inventory){
            pq.offer(e);
        }
        long ord = orders;
        long res = 0;
        long t =pq.poll();
        long time = 1;
        while(ord > 0){
            while(!pq.isEmpty() && pq.peek()==t){
                pq.poll();
                time++;
            }
            int next = pq.isEmpty() ? 0:pq.peek();
            long count = Math.min(t-next,ord/time);
            if(count == 0)count++;
            res+= time * ((t-count+1)* count + (count * (count-1))/2);
            t-= count;
            ord -=time * count;
        }
        res -= (t+1)*(-ord);
        //864122087
        return (int)(res%1000000007);
    }
}
