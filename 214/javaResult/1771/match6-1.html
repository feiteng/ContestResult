class Solution {
    int mod = (int)1e9+7;
    public boolean check(int cur,int[] cnt,int k){
        long sum = 0;
        for(int t:cnt){
            
if(t>cur){
                sum += t-cur;
            }
        }
        // System.out.println("->"+cur+" "+sum+" "+ k);
        return sum>=k;
    }
    public long get(int l,int r){
        return ((long)l+r)*(r-l+1)/2;
    }
    public int maxProfit(int[] cnt, int k) {
        Arrays.sort(cnt);
        // System.out.println(Arrays.toString(cnt));
        int r = cnt[cnt.length-1]-1,l=0;
        while(l<r){
            // System.out.println(l+" "+r);
            int mid = (l+r+1)/2;
            if(check(mid,cnt,k))
{
                l = mid;
            }
            else{
                r = mid-1;
            }
        }
        //找到那个边界值 sum >= k 
        l++;//此时
        // System.out.print(l);
        long res = 0 ;
        int tot = 0;
        for(int i=cnt.length-1;i>=0;i--){
            if(cnt[i]<=l)continue;
            res = (res+get(l+1,cnt[i]))%mod;
            tot += cnt[i]-l;
        }
        // System.out.println("cur_res->"+res);
        // System.out.println("left->"+(k-tot));
        res = (res+((long)k-tot)*(l))%mod;
        return (int)res;
    }
}
