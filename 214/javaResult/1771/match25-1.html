class Solution {
 public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        int left = 0, right = inventory[inventory.length - 1];
        long ret = 0;
        int baseline = 0;
        while (left <= right) {
            int mid = (left + right) / 2;
            long to = 0;
            for (int i = 0; i < inventory.length; i++) {
                if (inventory[i] > mid) {
                    to += inventory[i] - mid;
                }
            }
            
if (to >= orders) {
                baseline = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        long to = 0;
        for (int i = 0; i < inventory.length; i++) {
            if (inventory[i] > baseline) {
                ret += (long) ((inventory[i] - baseline)) * (inventory[i] + baseline + 1) / 2;
                to += inventory[i] - baseline;
            }
        }
        if (to > orders) {
            ret -= (to - orders) * (baseline + 1);
        }
        return (int) (ret % 1000000007);
    }
}
