class Solution {
    private int MOD = (int) 1e9 + 7;

    public int maxProfit(int[] inventory, int orders) {
        int len = inventory.length;
        Arrays.sort(inventory);
        int low = 0, high = inventory[len - 1];
        while (low < high) {
            int mid = low + (high - low) / 2;
            long cnt = 0;
            for (int i = len - 1; i >= 0; i--) {
                if (inventory[i] <= mid) {
                    break;
                }
                cnt += inventory[i] - mid;
            }
            if (cnt < orders) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        int num = low;
        int idx = len - 1;
        long ans = 0;
        while (orders > 0) {
            if (idx < 0 || inventory[idx] <= num) {
                break;
            }
            int cnt = Math.min(orders, inventory[idx] - num);
            orders -= cnt;
            ans = (ans + ((long) inventory[idx] + inventory[idx] - cnt + 1) * cnt / 2) % MOD;
            inventory[idx] -= cnt;
            idx--;
        }
        if (orders > 0) {
            ans = (ans + (long) orders * num) % MOD;
        }
        return (int) ans;
    }
}
