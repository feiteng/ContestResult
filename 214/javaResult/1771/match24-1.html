class Solution {
    public int maxProfit(int[] a, int m) {
        
        TreeMap<Integer,Integer> map = new TreeMap<>();
        map.put(0,1);
        for(int x : a) map.put(x,map.getOrDefault(x,0)+1);
        long ans = 0;
        int mod = (int)(1e9+7);
        while(map.size() > 0 && m > 0){
            long v = map.lastKey();
            long f = map.get((int)v);
            map.remove((int)v);
            long u = map.lastKey();
            long total = (v-u)*f;
          //  System.out.println(total+" "+m);
            if(m >= total){
                m-=total;
                long sum = get(v) - get(u);
                sum%=mod;
                ans+=(f*sum)%mod;
                map.put((int)u,map.getOrDefault((int)u,0)+(int)f);
             //   System.out.println(ans+" "+v+" "+u);
            }else{
                long q = m/f;
             
                long sum = get(v) - get(v-q);
                sum%=mod;
                ans+=(f*sum)%mod;
                long rem = m%f;
                ans+=(v-q)*rem;
                ans%=mod;
              //  System.out.println(ans);
                break;
            
}
            
        }
        ans%=mod;
        return (int)ans;
    }
    
    long get(long n ){
        return (n*(n+1))/2;
    }
}
