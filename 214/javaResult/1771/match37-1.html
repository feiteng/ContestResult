class Solution {
    public int maxProfit(int[] inv, int orders) {
        int mod = (int)1e9 + 7;
        long sol = 0;
        Arrays.sort(inv);
        
        int n = inv.length;
        
        for (int i = n - 1; i >= 0;) {
            if (i != 0 && inv[i] == inv[i-1]) {
                i--;
            } else {
                int dif = i == 0 ? inv[0] : inv[i] - inv[i-1];
                if ((n - i) * dif >= orders) {
                    int len = orders / (n - i);
                    long tot = (n - i) * ((long)inv[i] * 2 - len + 1) * len / 2;
                    orders -= len * (n - i);
                    sol += tot + (long)orders * (inv[i] - len);
                    sol %= mod;
                    break;
                } else {
                    long tot = (n - i) * ((long)inv[i] * 2 - dif + 1) * dif / 2;
                    sol += tot;
                    orders -= dif * (n - i);
                    sol %= mod;
                }
                sol %= mod;
                i--;
            }
        }
        return (int)sol;
    }
}
