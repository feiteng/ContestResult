class Solution {
    
    long mod = 1000000007;
    public int maxProfit(int[] is, int orders) {
        int lo  = 0;
        int hi = 0;
        
        for(int v:is){
            hi = Math.max(hi,v);
        }
        
        
        
while(lo<hi){
            
            int mid =(lo+hi+1)>>1;
            
            long s = 0;
            
            for(int v:is){
                if(v>mid){
                    s += v - mid;
                
}
            }
            
            if(s>=orders){
                lo = mid;
                
            }else{
                hi = mid-1;
                
            }
            
            
            
        }
        long res = 0;
        long tot = 0;
        for(int v:is){
            if(v>lo){
                res += v - lo;
                tot += (long)(v+lo+1)*(long)(v-(lo+1)+1)/2L;
                tot %= mod;
            }
        }
        
        
        
        
        
        
        
        while(res>orders){
            tot -= lo+1; 
            tot %= mod;
            tot += mod;
            tot%= mod;
            
            
            res--;
        }
        
        
        return (int)tot;
        
        
        
        
        
    }
}
