class Solution {
    public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        
        int min = 0;
        int max = inventory[inventory.length-1]+1;
        while (max > min+1) {
            int mid = (min + max) / 2;
            long sum = 0;
            for (int i = 0; i < inventory.length; i++)
                sum += Math.max(0, inventory[i] - mid + 1);
            // System.out.println(max + " " + min + " " + sum);
            if (sum <= orders)
                max = mid;
            else
                min = mid;
            
        }

        long sum = 0;
        long ans = 0;
        for (int i = 0; i < inventory.length; i++) {
            sum += Math.max(0, inventory[i] - max + 1);
            ans += f(max, inventory[i]);
        }
        ans += (orders-sum)*(max-1);
        
        return (int)(ans % 1000000007);
    }
    
    public long f(int i, int j) {
        if (i > j) return 0;
        return (long)j*(j+1)/2-(long)i*(i-1)/2;
    }
}
