class Solution {
    public int maxProfit(int[] inventory, int orders) {
        PriorityQueue<long[]> pq=new PriorityQueue<>((x,y)->(int)(y[0]-x[0]));
        for(int value:inventory){
            pq.offer(new long[]{value,1});
        }
        
        int mod=(int)1e9+7;
        long sum=0;
        while(orders>0){
            long[] cur=pq.poll();
            long[] next=new long[]{0,1};
            if(!pq.isEmpty()){
                next=pq.peek();
            }
            long diff=(cur[0]-next[0])*(cur[1]);
            if(diff<=orders){
                sum+=(cur[0]+next[0]+1)*diff/2;
                next[1]+=cur[1];
                orders-=diff;
            }else{
                int num=orders/((int)cur[1]);
                if(num>0){
                    sum+=(cur[0]+cur[0]-num+1)*num/2*cur[1];
                    orders-=num*(int)cur[1];
                    cur[0]-=num;
                    pq.offer(cur);
                }else{
                    sum+=cur[0];
                    cur[1]--;
                    pq.offer(cur);
                    pq.offer(new long[]{cur[0]-1,1});
                    orders--;
                }
            }
            sum%=mod;
        }
        return (int)sum;
    }
}
