class Solution {
  public int maxProfit(int[] inventory, int orders) {
    Integer[] inv = new Integer[inventory.length];
    for (int i = 0; i < inv.length; i++) {
      inv[i] = inventory[i];
    }
    Arrays.sort(inv, Collections.reverseOrder());
    int left = 0;
    int right = inv[0];
    while (left < right - 1) {
      int mid = left + (right - left) / 2;
      if (count(inv, mid) < orders) {
        right = mid;
      } else {
        left = mid;
      }
    }
    long total = 0;
    for (int i = 0; i < inv.length && inv[i] > right; i++) {
      total += ((long) inv[i] + right + 1) * ((long) inv[i] - right) / 2;
    }
    total += ((long) orders - count(inv, right)) * right;
    return (int) (total % 1_000_000_007);
  }
  
  private long count(Integer[] inv, int remain) {
    long orders = 0;
    for (int i = 0; i < inv.length && inv[i] > remain; i++) {
      orders += inv[i] - remain;
    }
    return orders;
  }
}
