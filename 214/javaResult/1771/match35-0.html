import java.util.Arrays;

class Solution {
	private static final long MOD = 1000000007;

	public int maxProfit(int[] inventory, int orders) {
		Arrays.sort(inventory);
		tr(inventory);
		long l = 0;
		long r = 10000000000L;
		while (l < r - 1) {
			long m = (l + r) / 2;
			if (check(inventory, m) >= orders)
				l = m;
			else
				r = m - 1;
		}
		if (check(inventory, r) >= orders)
			return get(inventory, r, orders);
		return get(inventory, l, orders);
	}

	private int get(int[] inventory, long r, long total) {
		tr(">", r, total);
		long ret = 0;
		for (int i = inventory.length - 1; i >= 0 && inventory[i] >= r; i--) {
			long last = inventory[i];
			long first = r;
			long cnt = last - first + 1;
			ret += ((last + first) * cnt / 2) % MOD;
//			tr(i, inventory[i], first, cnt, ret);
			total -= cnt;
		}

		ret += total * r;
		ret = ret % MOD + MOD;
		return (int) (ret % MOD);
	}

	private long check(int[] inventory, long m) {
		long ret = 0;
		for (int i : inventory)
			if (i >= m)
				ret += (i - m + 1);
		return ret;
	}

	public static void tr(Object... o) {
		System.out.println(Arrays.deepToString(o));
	}
}
