class Solution {
       int mod=(int)1e9+7;
    public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        int len=inventory.length;
        long res=0;

        for (int i = inventory.length - 1; i > 0; i--) {
            long big = inventory[i],small=inventory[i-1];
         if(orders>=(len-i)*(big-small)){


            res=(res+helper(small+1,big)*(len-i))%mod;
            orders-=(len-i)*(big-small);

           inventory[i]=(int)small;
         }else {

             int cc=orders/(len-i);
           res=(res+helper(big-cc+1,big)*(len-i))%mod;
            orders-=cc*(len-i);
            big=big-cc;
            res=(res+orders*big)%mod;
            return (int) res;
         }
        }
        System.out.println(orders+" "+res);
       if(orders>0){
           int big=inventory[0];
           int cc=orders/(len);
          if(cc!=0){
              res=(res+(helper(big-cc+1,big)%mod)*(len))%mod;
          }
           orders-=cc*(len);
           big=big-cc;
           res=(res+orders*big)%mod;
       
}

        return  (int)res;
    }


     long helper (long a, long b) {
        long res= ((long)a + b) * (b - a + 1) / 2;
        return res%mod;
    }




}
