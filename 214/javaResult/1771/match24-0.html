class Solution {
    public int maxProfit(int[] inventory, int orders) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(0, 0);
        for (int b:inventory) map.put(b, map.getOrDefault(b, 0)+1);
        int mod = 1000000000+7;
        long res = 0;
        while (orders>0) {
            int cur = map.lastKey();
            int fq = map.get(cur);
            map.remove(cur);
            int next = map.lastKey();
            if (orders >= (cur - next)*fq) {
                res = (res + calc(cur, next+1, fq))%mod;
                orders -= (cur - next)*fq;
                map.put(next, map.get(next) + fq);
                
            } else {
                int q = orders/fq;
                res = (res + calc(cur, cur-q+1, fq))%mod;
                res = (res + (orders%fq) * (long)(cur-q))%mod;
                orders = 0;
            }

        }
        return (int)res;
    }
    
    public long calc(int st, int ed, int freq) {
        int mod = 1000000000+7;
        long ret = ((long)(st + ed))*(st-ed+1)/2;
        ret %=mod;
        ret = (ret*freq)%mod;
        return ret;
    }
}
