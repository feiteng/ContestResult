class Solution {
    public int maxProfit(int[] inv, int orders) {
        int n = inv.length;
        long MOD = 1000000007L, res = 0L;
        TreeMap <Long, Long> map = new TreeMap();
        map.put(0L, 0L);
        
for(int i : inv)
            map.put((long)i, map.getOrDefault((long)i, 0L) + 1);
        while(orders > 0){
            //System.out.println(res + " " + map);
            long cur = map.lastKey();
            long num = map.get(cur);
            map.remove(cur);
            
long next = map.lastKey();
            long avail = num * (cur - next);
            if(avail <= orders){
                orders -= avail;
                res += 1L * num * (cur + next + 1) * (cur - next) / 2;
                map.put(next, map.get(next) + num);
            }
            else{
                long a1 = orders / num, a2 = orders % num;
                res += 1L * (num - a2) * (cur + cur - a1 + 1) * (a1) / 2;
                res += 1L * (a2) * (cur + cur - a1) * (a1 + 1) / 2;;
                orders = 0;
            }
        }
        return (int)(res % MOD);
    }
}
