class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long res = 0, sum = 0, right = 0;
        int len = inventory.length;
        for (int v : inventory) {
            sum += v;
            right = Math.max(right, v);
        }
        long left = 0;
        int mod = 1000000007;
        while (left < right) {
            long mid = (left + right) / 2;
            long cur = 0;
            for (int o : inventory) {
                cur += Math.max(0, (long)o - mid);
            }
            if (cur <= orders) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        Arrays.sort(inventory);
        for (int i = len - 1;i >= 0 && orders > 0;i--) {
            int v = inventory[i];
            if (v <= left) {
                break;
            }
            // System.out.println(v + " " + left);
            if (left == v - 1) {
                res = (res + v);
            } else {
                res = (res + (v + left + 1) * (v - left) / 2) % mod;
                // System.out.println(res);
            }
            orders -= (v - left);
        }
        return (int)((res + orders * left) % mod);
    }
}
