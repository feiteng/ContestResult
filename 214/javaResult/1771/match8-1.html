class Solution {
    public int maxProfit(int[] inventory, int orders) {
        PriorityQueue<Long> pq = new PriorityQueue<Long>(Collections.reverseOrder());
        for (int i : inventory) {
            pq.offer((long) i);
        }
        pq.offer(0L);
        long res = 0;
        long MOD = 1_000_000_007;
        long value = pq.poll();
        long fac = 1;
        while (orders > 0) {
            long next = pq.poll();
            long fullfill = (value - next) * fac;
            // System.out.println("value " + value + " next " + next + " orders "  + orders);
            if (fullfill <= orders) {
                res += (value + next + 1) * fullfill / 2;
                res %= MOD;
                value = next;
                fac++;
                orders -= fullfill;
            } else {
                long n = orders / fac;
                res += (value + value - n + 1) * n * fac / 2;
                res %= MOD;
                value -= n;
                orders -= fac * n;
                // System.out.println(res + " fullfill " + (fac*n) + " left "  + orders);
                res += orders * value;
                res %= MOD;
                
orders = 0;
            }
            // System.out.println(res + " fullfill " + fullfill + " left "  + orders);
        }
        return (int) res;
    }
}
