class Solution {
            int mod = (int) (1e9 + 7);

    public int maxProfit(int[] inven, int orders) {
        Arrays.sort(inven);
        long result = 0;
        long count = 1;
        for (int i = inven.length - 1; i >= 1; i--) {
            long cur = inven[i];
            long before = inven[i - 1];
            if (cur == before) {
                count++;
                continue;
            } else {
                if (orders >= (cur - before) * count) {
                    result = result + (helper(cur, before, count, orders) % mod);
                    result = result % mod;
                    orders = (int) (orders - (cur - before) * count);
                    count++;
                } else {
                    result = result + (helper(cur, before, count, orders) % mod);
                    return (int) result % mod;
                }
            }
            if (orders == 0) {
                return (int) result % mod;
            }
        }
        if (orders > 0) {
            result += helper(inven[0], 0, count, orders) % mod;            
        }
        return (int) result % mod;
    }
    public long helper(long cur, long before, long count, long orders) {
        if (orders > (cur - before) * count && orders >= count) {
            return count * ((cur + before + 1)  * ((cur - before) ) / 2);
        } else {
            long ac = orders / count;
            long nn = (orders % count) * (cur - ac);
            long last = cur - ac;
            long fir = ( (cur + last + 1) ) * ((cur - last) ) / 2;
            fir = fir % mod;
            return fir * count + nn;
        }
    }
}
