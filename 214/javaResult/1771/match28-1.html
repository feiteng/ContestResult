class Solution {
    public int maxProfit(int[] inventory, int orders) {
        long max = 0;
        int mod = 1000000007;
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b-a);
        
for(int i : inventory){
            pq.add(i);
        }
        int n = orders;
        int cnt = 1;
        int v = pq.poll();
        pq.add(0);
        
while(n > 0){
            while(!pq.isEmpty() && v == pq.peek()){
                cnt++;
                pq.poll();
            }
            long dif = (long)(v- pq.peek()); 
            if(cnt*dif > n){
                dif = n/cnt;
                n = n%cnt;
                max += cnt*(v+ v-dif+1)*dif/2; 
                max += n*(v-dif );
                n = 0;
            }else{
                max += cnt*(v+ v-dif+1)*dif/2; 
                v-= dif;
                n-= cnt*dif;
            }
        }
        return (int)(max%mod);
    }
}
