class Solution {
    // deal with mod only with the result
    int mod = 1000000007;
    public int maxProfit(int[] inventory, int orders) {
        Arrays.sort(inventory);
        
        int n = inventory.length;
        long max = 0;
        for(int i=n-1; i>0; i--) {
            long[] res = cal(inventory[i], inventory[i-1], n-i, orders);
            // System.out.println(Arrays.toString(res));
            max += res[0];
            orders -= (int) res[1];
            if (orders == 0) break;
        }
        
        if(orders > 0) {
            long[] res = cal(inventory[0], 0, n, orders);
            // System.out.println(Arrays.toString(res));
            max += res[0];
        }
        
        return (int) (max % mod);
    }
    
    private long[] cal(long a, long b, int len, int orders) {
        if((a-b)*len <= orders) {
            // System.out.println ((a+b+1));
            // System.out.println ((a-b));
            // System.out.println (((a+b+1)*(a-b)/2)*len);
            long sum = ((a+b+1)*(a-b)/2)*len % mod;
            long consume = (a-b)*len;
            return new long[] {sum, consume};
        } else {
            long l = orders/len; // from a to (a-l+1);
            long r = orders - l*len; // r times (a-l)
            long sum = (((2*a-l+1)*l/2)*len + (a-l)*r);
            return new long[] {sum, orders};
        }
    }
}
