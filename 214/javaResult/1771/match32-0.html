class Solution {
    public int maxProfit(int[] arr, int orders) {
        long ans = 0L;
        long p =(long)(1e9+7);
        long sum=0L;
        for(int i =0;i<arr.length;i++){
            sum+=arr[i];
        }
        int val = (int)binarySearch(0, sum, arr, orders);
        // 3 4 5 6 7
        for(int i =0;i<arr.length;i++){
            if(arr[i]>val){
                ans+= ((long) (arr[i]+val+1)*(long)(arr[i]-val))/2;
                ans%=p;
                orders -=(arr[i]-val); 
                           System.out.println(ans+" "+i+" "+arr[i]+" "+val);
            }
        }
            ans+=((long)orders*(long)val);
            ans%=p;
        return (int)ans;
    }
    
    public static long binarySearch(long low, long high, int arr[], int orders){
        long ans = 0;
        while(low<=high){
            long mid = low+(high-low)/2;
            if(isPossible(orders, mid, arr)){
               // System.out.println(mid);
                ans = mid;
                low = mid+1;
            } else {
                high = mid-1;
            }
        }
        return ans;
    }
    public static boolean isPossible(int orders, long mid, int arr[]){
        long val =0L;
        for(int i=0;i<arr.length;i++){
            if(arr[i]>=mid){
                val+=(arr[i]-mid+1);
            }
        }
       // System.out.println(mid + " "+val+" "+orders);
        return val>=(long)orders;
    }
}
