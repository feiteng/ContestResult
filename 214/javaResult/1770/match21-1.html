class Solution {
    public int minDeletions(String s) {
        int[] counts = new int[26];
        for (int i = 0; i < s.length(); i++) {
            counts[s.charAt(i) - 'a']++;
        }
        Arrays.sort(counts);
        int res = 0;
        for (int i = 24; i >= 0; i--) {
            if (counts[i] > 0 && counts[i] >= counts[i+1]) {
                int diff = 
Math.min(counts[i], counts[i] - counts[i+1] + 1);
                counts[i] -= diff;
                res += diff;
            }
        }
        return res;
    }
}
