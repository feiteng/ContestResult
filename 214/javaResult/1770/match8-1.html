class Solution {
    public int minDeletions(String s) {
        int[] cntMap = new int[128];
        for (int i = 0; i < s.length(); ++i) {
            cntMap[s.charAt(i)]++;
        }

        PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder());
        
for (char c = 'a'; c <= 'z'; ++c) {
            if (cntMap[c] > 0) {
                maxHeap.offer(cntMap[c]);
            }
        }

        int count = 0;
        while (!maxHeap.isEmpty()) {
            int maxFreq = maxHeap.poll();
            if (maxHeap.isEmpty()) {
                break;
            }

            if (maxFreq == 
maxHeap.peek()) {
                if (maxFreq > 1) {
                    maxHeap.offer(maxFreq - 1);
                }
                count++;
            }
        }

        return count;
    }
}
