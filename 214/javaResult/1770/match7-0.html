class Solution {
    public int minDeletions(String s) {
        int[] cnt = new int[256];
        for(char c : s.toCharArray()){
            cnt[c]++;
        }
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b - a);
        for(int i : cnt){
            if(i!= 0){
                pq.add(i);
            }
        }
        int min =0;
        while(!pq.isEmpty()){
            int i = pq.poll();
            if(!pq.isEmpty() && pq.peek() == i){
                min++;
                if(--i != 0){
                    pq.add(i);
                }
            }
        }
        return min;
    }
}
