class Solution {
    public int minDeletions(String s) {
        int[] hash = new int[26];
        for (int i = 0; i < s.length(); i++) {
            hash[s.charAt(i) - 'a']++;
        }
        Arrays.sort(hash);
        Set<Integer>set=new HashSet<>();
        int ans = 0;
        for (int i = 0; i <26 ; i++) {
            if (set.contains(hash[i])){
                int num=get(set,hash[i]);
                ans+=hash[i]-num;
                set.add(num);
            }else{
                set.add(hash[i]);
            }
        }

        return ans;
    }
    private int get(Set<Integer> set,int num){
        for (int i = num; i >0 ; i--) {
            if (!set.contains(i))return i;
        }
        return 0;
    }
}
