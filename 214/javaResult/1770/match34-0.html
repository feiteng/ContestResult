class Solution {
    public int minDeletions(String s) {
        int[] log = new int[26];
        for(int i = 0;i<s.length();i++){
            log[s.charAt(i)-'a']++;
        }
        HashMap<Integer,Integer> fre = new HashMap<>();
        int res = 0;
        for(int i = 0;i<26;i++){
            if(log[i] == 0)
                continue;
            if(fre.containsKey(log[i])){
                int j = log[i]-1;
                while(fre.containsKey(j)){
                    j--;
                }
                res += log[i]-j;
                if(j!=0){
                    fre.put(j,0);
                }
            } else {
                fre.put(log[i],0);
            }
            
        }
        return res;
        
    }
}
