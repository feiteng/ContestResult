class Solution {
    public int minDeletions(String s) {
        int[] counts = new int[26];
        for (char c: s.toCharArray())
            counts[c-'a']++;
        
        int[] numbers = new int[s.length()+1];
        
        for(int count: counts)
            numbers[count]++;
        
        int ret = 0;
        for (int i = s.length(); i > 0; i--) {
            if (numbers[i] > 1) {
                ret += numbers[i]-1;
                numbers[i-1] += numbers[i]-1;
            }
        }
        
        return ret;
    }
}
