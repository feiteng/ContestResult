class Solution {
    public int minDeletions(String s) {
        int[] arr = new int[26];
        for (char c : s.toCharArray()) ++arr[c - 'a'];
        Arrays.sort(arr);
        int res = 0;
        for (int i = 24; i >= 0; --i) {
            if (arr[i] == 0) break;
            if (arr[i + 1] == 0) {
                res += arr[i];
                arr[i] = 0;
            }
            else if (arr[i] >= arr[i + 1]) {
                int cnt = arr[i] - arr[i + 1] + 1;
                res += cnt;
                arr[i] -= cnt;
            
}
        }
        return res;
    }
}
