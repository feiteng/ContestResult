class Solution {
    public int minDeletions(String s) {
        int[] counts = new int[26];
        for (char c : s.toCharArray()) {
            counts[c - 'a']++;
        }
        Arrays.sort(counts);
        int res = 0;
        //System.out.println(Arrays.toString(counts));
        for (int i = 24; i >= 0 ; i--) {
            if (counts[i] == 0)
                break;
            boolean flag = true;
            for (int j = i + 1; j < 26; j++) {
                if (counts[j] == counts[i]) {
                    flag = false;
                    break;
                }
            }
            if (!flag) {
                    if (counts[i + 1] == 0) {
                        res += counts[i];
                        counts[i] = 0;
                        
                    }
                    else {
                        res += counts[i] - counts[i + 1] +1;
                        counts[i] = counts[i + 1] - 1;
                    }
                    
                    
                }
        }
        //System.out.println(Arrays.toString(counts));
        return res;
    }
}
