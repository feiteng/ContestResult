class Solution {
    public int minDeletions(String s) {
        int[] count = new int[26];
        for (int i = 0; i < s.length(); i++) {
            int ch = s.charAt(i) - 'a';
            count[ch]++;
        }
        
        int[] c = new int[s.length() + 1];
        for (int i = 0; i < 26; i++) {
            if (count[i] > 0) {
                c[count[i]]++;
            }
        
}
        
        int ans = 0;
        for (int i = s.length(); i >= 1; i--) {
            if (c[i] >= 2) {
                ans += c[i] - 1;
                c[i - 1] += c[i] - 1;
            }
        }
        return ans;
    }
}
