class Solution {
    //fequency不同，只能delete，从高到低，有冲突就-1
    public int minDeletions(String s) {
        int[] cnt = new int[26];
        for(int i=0;i<s.length();i++){
            cnt[s.charAt(i) - 'a']++;
        }
        //字母不重要
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b - a);
        for(int i=0;i<26;i++){
            if(cnt[i] != 0){
                pq.offer(cnt[i]);
            }
        }
        
        int deleted = 0;
        Integer pre = null;
        while(!pq.isEmpty()){
            int frequency = pq.poll();
            if(pre == null){
                pre = frequency;
            } else if(pre == frequency){
                if(frequency - 1 != 0){
                    pq.offer(frequency - 1);
                }
                deleted++;
            } else {
                pre = frequency;
            }
        }
        return deleted;
    }
}
