class Solution {
    public int minDeletions(String s) {
        int[] cnt = new int[26];
        for(char c: s.toCharArray()){
            cnt[c-'a']++;
        }
        int ans= 0;
        HashSet<Integer> set = new HashSet<>();
        for(char c= 'a'; c<='z';++c){
            int f =cnt[c-'a'];
            while(f>0 && set.contains(f)){
                ans++;
                --f;
            }
            if(f>0)set.add(f);
            
        }
        return ans;
    }
}
