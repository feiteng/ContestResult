class Solution {
    public int minDeletions(String s) {
        int[] map = new int[26];
        for(char c : s.toCharArray()) {
            map[c-'a']++;
        }
        
        int count = 0;
        Set<Integer> set = new HashSet<>();
        
PriorityQueue<Integer> queue = new PriorityQueue<>((a,b)->(b-a));
        for(int i = 0; i < 26; i++) {
            if(map[i] != 0) {
                queue.offer(map[i]);
            }
        }
        
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            if(!set.contains(cur)) {
                set.add(cur);
            }
            else 
{
                cur--;
                count++;
                if(cur != 0) 
                    queue.offer(cur);
            }
        }
        
        return count;
    }
}
