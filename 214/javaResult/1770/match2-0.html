class Solution {
    public int minDeletions(String s) {
        int[] arr = new int[26];
        for(char c:s.toCharArray()){
            arr[c-'a']++;
        }
        Set<Integer> set = new HashSet();
        int ans=0;
        for(int i = 0;i<26;i++){
            if(arr[i]!=0){
                while(set.contains(arr[i])&&arr[i]>0){
                    ans++;
                    arr[i]--;
                }
                set.add(arr[i]);
            }
        }
        return ans;
    }
}
