class Solution {
    public int minDeletions(String s) {
        int []a = new int[26];
        for(char c : s.toCharArray()){
            a[c-'a'] ++;
        }
        Arrays.sort(a);
        int ans = 0;
        for(int i=24;i>=0;i--){
            if(a[i] > 0 && a[i] >= a[i+1])
{
                ans += a[i]-Math.max(0, a[i+1]-1);
                a[i] = Math.max(0, a[i+1]-1);
            }
        }
        return ans;
    }
}
