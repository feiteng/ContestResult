class Solution {
    public int minDeletions(String s) {
        int[] alphabet = new int[26];
        for (int i = 0; i < s.length(); i++)
            alphabet[s.charAt(i) - 'a']++;
        int n = s.length();
        
int[] freq = new int[n + 1];
        for (int i = 0; i < 26; i++)
            freq[alphabet[i]]++;
        int res = 0;
        for (int i = n; i > 0; i--) {
            if (freq[i] > 1) {
                freq[i - 1] += freq[i] - 1;
                res += freq[i] - 1;
            }
        }
        return res;
    }
}
