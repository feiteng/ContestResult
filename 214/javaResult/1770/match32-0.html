class Solution {
    public int minDeletions(String s) {
       int[] a = new int[26];
    for (char c : s.toCharArray()) {
      a[c-'a']++;
    }
    List<Integer> arr = new ArrayList<>();
    for (int i = 0; i < 26; i++) {
      if (a[i] > 0) {
        arr.add(a[i]);
      }
    }
    Collections.sort(arr, Collections.reverseOrder());
    int ans = 0;
    int prev = arr.get(0);
    for (int i = 1; i < arr.size(); i++) {
      if (arr.get(i) < prev) {
        prev = arr.get(i);
      } else {
        ans += arr.get(i) - prev + 1;
        prev = Math.max(1, prev-1);
      }
    }
    return ans;
    }
}
