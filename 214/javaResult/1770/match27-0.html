class Solution {
    public int minDeletions(String s) {
        int[] cs = new int[26];
        for(char c : s.toCharArray()) {
            cs[c-'a']++;
        }
        
        int ans = 0;
        HashSet<Integer> set = new HashSet<>();
        for(int count : cs) {
            while(count>0 && set.contains(count)) {
                count--;
                ans++;
            }
            set.add(count);
        }
        return ans;
    }
}
