class Solution {
    public int minDeletions(String s) {
        int[] count = new int[26];
        for (char c : s.toCharArray()) {
            count[c - 'a']++;
        }
        Arrays.sort(count);
        int ans = 0;
        int max = s.length();
        for (int i = 25; i >= 0; i--) {
            if (count[i] > 0) {
                if (count[i] > max) {
                    ans += count[i] - max;
                    count[i] = max;
                }
                max = Math.max(0, count[i] - 1);
            }
        }
        return ans;
    }
}
