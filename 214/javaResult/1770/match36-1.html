class Solution {
    public static int minDeletions(String s) {
        int[] map = new int[26];
        for (char c : s.toCharArray()) {
            map[c - 'a']++;
        
}
        Arrays.sort(map);
        Set<Integer> set = new HashSet<>();
        int res = 0;
        for (int i = 0; i < map.length; i++) {
            if (map[i] == 0) continue;
            if (set.contains(map[i])) {
                int j;
                for (j = map[i] - 1; j > 0; j--) {
                    if (!set.contains(j)) {
                        set.add(j);
                        break;
                    }
                }
                res += map[i] - j;
            } else 
{
                set.add(map[i]);
            }
        }
        return res;
    }
}
