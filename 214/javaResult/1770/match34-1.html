class Solution {
    public int minDeletions(String s) {
        int[] cnt = new int['z' - 'a' + 1];
        for(char c : s.toCharArray()){
            cnt[c - 'a']++;
        
}
        Set<Integer> set = new HashSet();
        int ans = 0;
        for(int x : cnt){
            if(x > 0){
                while(x > 0 && set.contains(x)){
                    x--;
                    ans++;
                }
                
set.add(x);
            }
        }
        return ans;
    }
}
