class Solution {
    public int minDeletions(String s) {
        int[] cnt = new int[26];
        for(int i = 0;i < s.length();++i) {
            cnt[s.charAt(i)-'a']++;
        }
        
Set<Integer> has = new HashSet<>();
        int rm = 0;
        for(int i = 0;i < 26;++i) {
            while(has.contains(cnt[i])&&cnt[i]>0) {
                rm++;
                cnt[i]--;
            }
            if(cnt[i]!=0) has.add(cnt[i]);
        }
        return rm;
    }
}
