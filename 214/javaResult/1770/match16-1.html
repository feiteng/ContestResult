class Solution {
    public int minDeletions(String s) {
        int[] ct = new int[26];
        for (char c : s.toCharArray()) {
            ct[c-'a']++;
        }
        Arrays.sort(ct);
        int sol = 0;
        for (int i = 24; i >= 0; i--) {
            if (ct[i] == 0) break;
            
if (ct[i] >= ct[i+1]) {
                if (ct[i+1] == 0) {
                    sol += ct[i];
                    ct[i] = 0;
                } else {
                    sol += ct[i] - (ct[i+1] - 1);
                    ct[i] = ct[i+1] - 1;
                }
                
            }
        }
        return sol;
    }
}
