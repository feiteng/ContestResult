class Solution {
    public int minDeletions(String s) {
        int[] f = new int[26];
        for (int i = 0; i < s.length(); i++) {
            f[s.charAt(i)-'a']++;
        }
        int max = 0;
        for (int i = 0; i < 26; i++)
            max = Math.max(max, f[i]);
        int[] g = new int[max+1];
        for (int i = 0; i < 26; i++)
            g[f[i]]++;
        // System.out.println(Arrays.toString(f));
        // System.out.println(Arrays.toString(g));
        int ans = 0;
        for (int i = max; i > 0; i--) {
            if (g[i] > 1) {
                ans += g[i]-1;
                g[i-1] += g[i]-1;
            }
        }
        return ans;
    }
}
