class Solution {
    public int minDeletions(String s) {
        int[] map = new int[26];
        for (char c: s.toCharArray()) map[c - 97]++;
        Arrays.sort(map);
        
int sum = 0;
        for (int i = 24; i >= 0; i--) {
            if (map[i] == 0) break;
            if (map[i + 1] == 0) {
                sum += map[i];
                map[i] = 0;
            } else if (map[i] >= map[i + 1]) 
{
                sum += map[i] - map[i + 1] + 1;
                map[i] = map[i + 1] - 1;
            }
        }
        return sum;
    }
}
