class Solution {
    public int minDeletions(String s) {
        int[] arr = new int[26];
        for (char c : s.toCharArray()) {
            arr[c - 'a']++;
        }
        PriorityQueue<Integer> queue = new PriorityQueue<>(Comparator.reverseOrder());
        for (int i : arr) {
            if (i > 0) {
                queue.add(i);
            }
        }
        int cur = Integer.MAX_VALUE;
        int res = 0;
        while (!queue.isEmpty()) {
            int e = queue.poll();
            if (cur != e) {
                cur = e;
            } else {
                if (e > 1) {
                    queue.add(e - 1);
                }
                res++;
            }
        }
        return res;
    }
}
