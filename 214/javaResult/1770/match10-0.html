class Solution {
    public int minDeletions(String s) {
        int[] counts = new int[26];
        for (int j = 0;j < s.length(); j++) {
            counts[s.charAt(j)-'a']++;
        }
        PriorityQueue<Integer> queue = new PriorityQueue<>((a, b) -> counts[b] - counts[a]);
        for (int j = 0; j < 26; j++) {
            if (counts[j] > 0) {
                queue.add(j);
            }
        }
        int ret = 0;
        while (queue.size() > 0) {
            int ind = queue.poll();
            boolean good = true;
            for (int k = 0; k < 26; k++) {
                if (k != ind && counts[k] == counts[ind]) {
                    good = false;
                    break;
                }
            }
            if (!good) {
                counts[ind]--;
                ret++;
                if (counts[ind] > 0)
                    queue.add(ind);
            }
        }
        return ret;
    }
}
