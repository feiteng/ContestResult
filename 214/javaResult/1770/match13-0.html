import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

class Solution {
	public int minDeletions(String s) {
		int cnt[] = new int[26];
		for (char c : s.toCharArray()) {
			cnt[c - 'a']++;
		}
		Arrays.sort(cnt);
		Set<Integer> set = new HashSet<Integer>();
		int ans = 0;
		for (int i : cnt) {
			while (set.contains(i)) {
				i--;
				ans++;
			}
			if (i>0)
				set.add(i);
		}
		return ans;
	}
}
