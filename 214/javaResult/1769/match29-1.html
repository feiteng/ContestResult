class Solution {
    public int getMaximumGenerated(int n) {
        if (n == 0) return 0;
        if (n == 1) return 1;
        
        int[] A = new int[n + 1];
        A[0] = 0;
        A[1] = 1;
        
        int res = 1;
        for (int i = 2; i < A.length; ++i) {
            A[i] = (i % 2 == 0) ? A[i / 2] : (A[i / 2] + A[i / 2 + 1]);
            res = Math.max(res, A[i]);
        }
        
        return res;
    }
}
