class Solution {
    public int getMaximumGenerated(int n) {
        if(n == 0) {
            return 0;
        }
        
        int[] out = new int[n+1];
        out[0] = 0;
        
out[1] = 1;
        
        for(int i = 2; i <= n; i++) {
            if(i % 2 == 0) {
                out[i] = out[i/2];
            } else {
                out[i] = out[i/2] + out[i/2 + 1];
            }
        }
        
        return  Arrays.stream(out).max().getAsInt(); 
    }
}
