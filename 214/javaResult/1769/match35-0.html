class Solution {
    public int getMaximumGenerated(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        int[] res = new int[n+1];
        res[1] = 1;
        int ans=0;
        for(int i=2;i<=n;i++){
            if(i%2==0){
                res[i] = res[i/2];
            }else{
                res[i] = res[i/2]+res[i/2+1];
            }
            ans = Math.max(ans,res[i]);
        }
        return ans;
    }
}
