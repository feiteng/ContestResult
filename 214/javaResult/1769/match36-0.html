class Solution {
    public int getMaximumGenerated(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
    int []dp=new int[n+2];
    dp[0]=0;
    dp[1]=1;
    int max=0;
    for(int i=2;i<n+1;i++)
    {
        int mid=i/2;
        if(i%2==0)
        {
            dp[i]=dp[mid];

        }else {
            dp[i]=dp[mid]+dp[mid+1];
        }
        max=Math.max(max,dp[i]);
    }
return max;
    }
}
