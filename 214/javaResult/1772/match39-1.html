class Solution {
    public int createSortedArray(int[] instructions) {
        int mod = 1000000007;
        FenwickTree tree = new FenwickTree(100001);
        long cost = 0;
        for (int ele: instructions) {
            cost = (cost + Math.min(
tree.query(ele - 1), tree.size() - tree.query(ele))) % mod;
            tree.insert(ele, 1);
        }
        return (int)cost;
    }
    
    
class FenwickTree {
        private int[] arr;
        private int size;
        
        public FenwickTree(int n) {
            arr = new int[n + 1];
            size = 0;
        
}
        
        public void insert(int idx, int val) {
            while (idx < arr.length) {
                arr[idx] += val;
                idx += idx & (-idx);
            }
            size++;
        }
        
        public int query(int idx) {
            int sum = 0;
            while (idx > 0) {
                sum += arr[idx];
                idx -= idx & (-idx);
            }
            
            return sum;
        }
        
        public int size(){
            return size;
        }
    }
}
