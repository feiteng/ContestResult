class Solution {
    int[] BIT;
    public int createSortedArray(int[] instructions) {
        long res = 0;
        BIT = new int[100001];
        
for(int n : instructions){
            int smaller = query(n-1), larger = query(100000) - query(n);
            res += Math.min(smaller, larger);
            add(n);
        }
        int mod = (int)Math.pow(10, 9) + 7;
        return (int)(res%mod);
    
}
    
    void add(int n){
        while(n < BIT.length){
            BIT[n]++;
            n += (n & -n);
        }
    
}
    int query(int n){
        int res = 0;
        while(n > 0){
            res += BIT[n];
            n -= (n&-n);
        }
        return res;
    }
}
