class Solution {
    static final int MOD = 1_000_000_007;

    int[] treeArr = null;

    public int createSortedArray(int[] instructions) {
        treeArr = new int[1_000_01];
        int ans = 0;
        int total = 0;
        for (int i : instructions) {
            int cost = Math.min(sum(i - 1), total - sum(i));
            ans += cost;
            ans %= MOD;
            update(i, 1);
            total++;
        }
        return ans;
    }

    private int sum(int pos) {
        int ans = 0;
        while (pos > 0) {
            ans += treeArr[pos];
            pos -= lowbit(pos);
        }
        return ans;
    }

    private void update(int pos, int val) {
        while (pos < treeArr.length) {
            treeArr[pos] += val;
            pos += lowbit(pos);
        }
    }

    private int lowbit(int pos) {
        return pos & -pos;
    }
}
