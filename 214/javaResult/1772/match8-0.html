class Solution {
    
       int[] mins;

    public void update(int i, int delta,int[] nums) {
        while (i < nums.length) {
            nums[i] += delta;
            i += lowbits(i);
        }
    }

    public int query(int x,int[] nums) {
        int sum = 0;
        while (x > 0) {
            sum += nums[x];
            x -= lowbits(x);
        }
        return sum;
    }

    public int lowbits(int x) {
        return (x & (-x));
    }
    
    

   public int createSortedArray(int[] instructions) {
            int n=instructions.length;
            int mod =(int)1e9+7;
            mins=new int[(int)1e5+10];
            int cnt=0;
            long cost=0;
            for (int i :instructions){
                update(i,1,mins);
                cnt++;
                int left = query(i-1,mins);
                int right = cnt-query(i,mins);
                cost=  (cost%mod+ Math.min(left,right))%mod;
            }
            return  (int)cost%mod;
    }


                        
    
}
