class Solution {
    public int createSortedArray(int[] instructions) {
        
        int MOD = (int) (1e9 + 7);
        
int[] BIT = new int[100002];
        int res = 0;
        for (int i : instructions) {
            res = (res + Math.min(getPrefixSum(BIT, i - 1), getSuffixSum(BIT, i + 1)) % MOD) % MOD;
            update(BIT, i);
        }
        
        return res;
    }
    
    private void update(int[] BIT, int index) {
        int val = 1;
        
while (index < BIT.length) {
            BIT[index] += val;
            index += index & (-index);
        }
    }
    
    private int getPrefixSum(int[] BIT, int index) {
        int sum = 0;
        while (index > 0) {
            sum += BIT[index];
            index -= index & (-index);
        }
        return sum;
    }

    private int getSuffixSum(int[] BIT, int index) {
        return getPrefixSum(BIT, (int) 1e5) - getPrefixSum(BIT, index - 1);
    }
}
