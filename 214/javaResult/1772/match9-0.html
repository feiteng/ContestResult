class Solution {

    //数组1..n 大小为4*n
    int[] C;
    //数组的长度+1
    int n;


    public int lowbit(int x){
        return x&(-x);
    }

    //更新单点值，x位其索引
    public void updata(int x,int d){
        while(x<=n){
            C[x]+=d;
            x+=lowbit(x);
        }
    }
    
    //获取区间和sum[1..x](左闭右闭)
    public int getSum(int x){
        int res=0;
        while(x>0){
            res+=C[x];
            x-=lowbit(x);
        }
        return res;
    }

    // public void main() {
    //     //插入值
    //     updata(1, val)
    //     //查询值
    //     getSum(1)
    // }

    public int createSortedArray(int[] instructions) {
        long ans = 0;
        int MOD = 1_000_000_007;
        n = 100001;
        C = new int[4 * n];
        for(int i = 0; i < instructions.length; i++) {
            int num = instructions[i];
            updata(num, 1);
            int less = getSum(num - 1);
            int more = i + 1 - getSum(num);
            ans = (Math.min(less, more) + ans) % MOD;
        }
        return (int)ans;
    }
}
