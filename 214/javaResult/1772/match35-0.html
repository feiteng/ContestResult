import java.util.Map;

class Solution {
    public int createSortedArray(int[] instructions) {
        int MOD = 1000000000 + 7;

        int[] countTable = new int[100050];
        TreeArray treeArray = new TreeArray();

        long cost = 0;

        int count = 0;
        for (int instruction : instructions) {
            count++;
            countTable[instruction]++;
            treeArray.add(instruction, 1);

            int lessCount = treeArray.sum(instruction - 1);
            int moreCount = count - countTable[instruction] - lessCount;

            cost = Math.floorMod(cost + Math.min(lessCount, moreCount), MOD);
        }

        return (int) Math.floorMod(cost, MOD);
    }
}


class TreeArray {
    public static final int N = 100050;
    public int[] c = new int[N];

    private int lowbit(int x) {
        return x & (-x);
    }

    public void add(int x, int d) {
        while (x <= N) {
            c[x] += d;
            x += lowbit(x);
        }
    }

    public int sum(int x) {
        int res = 0;
        while (x > 0) {
            res += c[x];
            x -= lowbit(x);
        }
        return res;
    }
}
