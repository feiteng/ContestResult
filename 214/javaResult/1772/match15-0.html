class Solution {
    
    static class Bit{
        int[] arr;
        public Bit(int n){
            this.arr = new int[n + 1];
        }
        
        public void add(int index){
            index++;
            while(index < this.arr.length){
                arr[index]++;
                index += (index & (-index) );
            }
            return;
        }
        
        public int query(int index){
            index++;
            int ans = 0;
            while(index > 0){
                ans += arr[index];
                index -= (index & (-index) );
            }
            return ans;
        }
    }
    
    public int createSortedArray(int[] inst) {
        Bit bit = new Bit(100001);
        long ans = 0;
        int MOD = (int)(1e9 + 7);
        for(int i = 0;i < inst.length;i++){
            int left = bit.query(inst[i] - 1), right = (i) - bit.query(inst[i]);
            bit.add(inst[i]);
            // System.out.println(left + " " + right);
            ans += Math.min(left, right);
            ans %= MOD;
        }
        return (int)ans;
    }
}
