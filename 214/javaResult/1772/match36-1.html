class Solution {
    public int createSortedArray(int[] instructions) {
        SegTree st=new SegTree(100001);
        int a[]=instructions;
        long mod=(long)1e9+7;
        long gg=0;
        for(int i=0;i<a.length;i++){
            int cur=a[i];
            
int min=Math.min(st.sum(0,cur-1),st.sum(cur+1,100000));
            gg+=min;
            st.add(cur);
        }
        return (int)(gg%mod);
    }
    
    
static class SegTree {
        int n;
        int seg[];
        SegTree(int n){
            this.n=n;
            seg=new int[4*n];
        }
        void add(int v){
            
add(1,0,n-1,v);
        }
        void add(int x, int lx, int rx, int v){
            if(lx==rx){
                seg[x]++;
                
return;
            }
            int mid=lx+rx>>1;
            if(v<=mid) add(2*x,lx,mid,v);
            else add(2*x+1,mid+1,rx,v);
            seg[x]=seg[2*x]+seg[2*x+1];
        
}
        int sum(int l, int r){
           return sum(1,0,n-1,l,r); 
        }
        int sum(int x, int lx, int rx, int l, int r){
            if(l<=lx && r>=rx) return seg[x];
            if(l>rx || r<lx) return 0;
            int mid=lx+rx>>1;
            return sum(2*x,lx,mid,l,r)+sum(2*x+1,mid+1,rx,l,r);
        }
    }
}
