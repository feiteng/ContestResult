class Solution {
    int[] bit;
    int maxn = 100000+5;
    public int createSortedArray(int[] instructions) {
        bit = new int[maxn+1];
        int[] cnt = new int[maxn+1];
        long ans = 0, mod= (long)1e9+7;
        
for(int i=0; i<instructions.length; i++){
            int cur = instructions[i];
            long smaller = sum(cur-1);
            long big = i-cnt[cur]-smaller;
            ans += Math.min(smaller, big);
            ans %= mod;
            addTo(cur, 1);
            cnt[cur]++;
        }
        ans %= mod;
        return (int)ans;
    
}
    void addTo(int pos, int val){
        while(pos<=maxn){
            bit[pos] += val;
            pos += pos&(-pos);
        }
    }
    int sum(int pos){
        int ans = 0;
        while(pos>0){
            ans += bit[pos];
            pos -= pos&(-pos);
        }
        return ans;
    }
}
