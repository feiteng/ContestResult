class Solution {
    int[] bit = new int[100010];
    public int createSortedArray(int[] instructions) {
        
        long cost = 0;
        int mod = (int) 1e9 + 7;
        for(int v : instructions)
        {
            int c1 = count(v - 1);
            int c2 = count(100000) - count(v);
            cost += Math.min(c1, c2);
            cost = (cost + mod) % mod;
            insert(v);
        }
        return (int) cost % mod;
    }
    
    int count(int k)
    {
        int re = 0;
        while(k > 0)
        {
            re += bit[k];
            k -= k & -k;
        }
        return re;
    }
    
    void insert(int k)
    {
        while(k <= 100000)
        {
            bit[k]++;
            k += k & -k;
        }
    }
}
