class Solution {
    
public int createSortedArray(int[] instructions) {
        long res = 0;
        int mod = (int) (1e9 + 7);
        int cnt = 0;
        for (int instruction : instructions) {
            int l = sum(instruction - 1);
            
int r = sum(instruction);
//            int l = upperBoundList(list, instruction - 1);
//            int r = lowerBoundList(list, instruction + 1);
//            list.add(l, instruction);
            res += Math.max(Math.min(l, cnt - r), 0);
            cnt++;
            add(instruction, 1);

        }
        return (int) (res % mod);
    }

    int N = (int) (1e5 + 5);
    int[] t = new int[N];
    int n = N;

    // n&(~n+1) ~n=-1-n
    // Integer.lowestOneBit(x)
    int lowbit(int x) {
        return x & -x;
    }

    // 位置从1开始
    void add(
int x, int c) {
        for (int i = x; i <= n; i += lowbit(i)) {
            t[i] += c;
        }
    }

    int sum(int x) {
        
int res = 0;
        for (int i = x; i > 0; i -= lowbit(i)) {
            res += t[i];
        }
        return res;
    }
}
