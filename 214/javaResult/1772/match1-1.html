class Solution {
    public int createSortedArray(int[] instructions) {
        BIT bit = new BIT(100002);
        int out = 0;
        for (int i : instructions) {
            int lt = bit.query(i);
            int gt = bit.query(100001) - bit.query(i + 1);
            out += Math.min(lt, gt);
            out %= 1000000000 + 7;
            bit.update(i + 1, 1);
        }
        return out;
    }

    class BIT {

        int n;
        int[] a;

        BIT(int n) {
            this.n = n;
            
a = new int[n + 1];
        }

        public void update(int i, int x) {
            for(; i <= n; i += i&-i) 
{
                a[i] += x;
            }
        }

        public int query(int i) {
            int sum = 0;
            for(; i > 0; i -= i&-i) {
                sum += a[i];
            }
            return sum;
        }
    }
}
