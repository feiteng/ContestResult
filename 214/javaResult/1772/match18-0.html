class Solution {
    public int createSortedArray(int[] instructions) {
        int mod = 1000000000+7;
        long res = 0;
        BinaryIndexedTree bit = new BinaryIndexedTree(100000);
        for (int i=0;i<instructions.length;i++) {
            int num = instructions[i];
            int m1 = bit.query(num-1);
            int m2 = i - bit.query(num);
            res = (res + Math.min(m1, m2))%mod;
            bit.add(num);
        }
        return (int)res;
    }

    class BinaryIndexedTree {
        int[] bit;
        int n;
        BinaryIndexedTree(int n) {
            bit = new int[n+1];
            this.n = n;
        }

        public void add(int i){
            for (;i<=n;i+= (i&-i))
            {
                bit[i]++;
            }
        }

        public int query(int i) {
            int res = 0;
            for (;i>0;i-=(i&-i)) {
                res += bit[i];
            }
            return res;
        }

    }
}
