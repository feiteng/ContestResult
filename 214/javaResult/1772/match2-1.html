class Solution {
    class Fenwick {
        int[] arr;
        Fenwick(int n) {
            arr = new int[n + 1];
        }
        void insert(int idx, int val) {
            while (idx < arr.length) {
                arr[idx] += val;
                idx += (idx & -idx);
            }
        }
        int get(int idx) {
            int res = 0;
            while (idx > 0) {
                res += arr[idx];
                idx -= (idx & -idx);
            }
            return res;
        }
    }
    public int createSortedArray(int[] instructions) {
        int res = 0, MOD = (int) 1e9 + 7;
        Fenwick fenwick = new Fenwick((int) 1e5);
        for (int num : instructions) {
            int l = fenwick.get(num - 1);
            int r = fenwick.get((int) 1e5) - 
fenwick.get(num);
            res = (res + Math.min(l, r)) % MOD;
            fenwick.insert(num, 1);
        }
        return res;
    }
}
