class Solution {
    public int createSortedArray(int[] instructions) {
        int max=(int)((1e5)+2);
        tree=new int[4*max];
        long ans=0;
        
for(int x:instructions) {
            ans+=Math.min(query(0,0,max-1,0,x-1),query(0,0,max-1,x+1,max-1));
            update(0,0,max-1,x);
        }
        return (int)(ans%1000000007);
    }
    int tree[];
    void update(
int cur,int l,int r,int ind){
        if(l==r){
            tree[cur]+=1;
            return;
        }
        int mid=(l+r)>>1;
        if(ind<=mid)update(2*cur+1,l,mid,ind);
        else update(2*cur+2,mid+1,r,ind);
        tree[cur]=merge(tree[2*cur+1],tree[2*cur+2]);
    }
    int merge(int a,int b){
        return a+b;
    
}
    int query(int cur,int l,int r,int start,int end){
        if(l>end || r<start)return 0;
        if(l>=start && r<=end)return tree[cur];
        int mid=(l+r)>>1;
        int p1=query(2*cur+1,l,mid,start,end);
        int p2=query(2*cur+2,mid+1,r,start,end);
        return merge(p1,p2);
    }
}
