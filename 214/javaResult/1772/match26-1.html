class Solution {
    int inf = 100000;
    int mod = 1000000007;
    public int createSortedArray(int[] instructions) {
        NumArray f = new NumArray(new int[inf+1]);
        int ans = 0;
        for (int i : instructions) {
            f.update(i, f.sumRange(i, i) + 1);
            int l = f.sumRange(0, i - 1);
            int r = f.sumRange(i + 1, inf);
            // System.out.println(l + " " + r + f.get(i));
            ans += Math.min(l, r);
            ans %= mod;
        }
        
        return ans;
    }
}


class NumArray {
    int[] f;
    int n;
    public NumArray(int[] nums) {
        n = nums.length;
        f = new int[n * 2];
        for (int i = n, j = 0; i < 2 * n; i++, j++) {
            f[i] = nums[j];
        
}
        for (int i = n - 1; i >= 0; i--) {
            f[i] = f[2 * i] + f[2 * i + 1];
        }
    }
    
    public void update(int i, int val) {
        i += n;
        f[i] = val;
        while(i > 0) {
            int left = i, right = i;
            if (i % 2 == 0) {
                right = i +1;
            } else {
                left = i -1;
            }
            f[i / 2] = f[left] + f[right];
            i /= 2;
        }
    }
    
    public int get(int i) {
        return f[i];
    
}
    
    public int sumRange(int i, int j) {
        i += n;
        j += n;
        int sum = 0;
        while (i <= j) {
            if ((i  % 2 ) == 1) {
                sum += f[i];
                i++;
            }
            if ((j  % 2 ) == 0) {
                sum += f[j];
                j--;
            }
            i /= 2;
            j /= 2;
        }
        return sum;
    }
}
