class Solution {
    int fenl=1000000;
    int fena[]=new int[fenl];
    long mod=1000000000+7;
    public int createSortedArray(int[] ins) {
        long ans=0;
        for(int i:ins)
        {
            ans+=
Math.min(getGreater(i),getSmaller(i));
            ans%=mod;
            insert(i);
        }
        return (int)ans;
    }
    public void insert(int pos)
    {
        pos+=1;
        while(pos<fenl)
        {
            fena[pos]+=1;
            pos+=pos&(-pos);
        
}
    }
    public int get(int pos)
    {
        pos+=1;
        int sum=0;
        while(pos>0)
        {
            sum+=fena[pos];
            pos-=(pos&(-pos));
        
}
        return sum;
    }
    public int getGreater(int pos)
    {
        return get(fenl-10)-get(pos);
    }
    public int getSmaller(int pos)
    {
        return get(pos-1);
    }
}
