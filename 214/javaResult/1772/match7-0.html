class Solution {
    private static final int MOD = 1_000_000_007;

    public int createSortedArray(int[] nums) {
        final int N = nums.length;
        int minVal = Integer.MAX_VALUE;
        int maxVal = Integer.MIN_VALUE;
        for (int num : nums) {
            minVal = Math.min(minVal, num);
            maxVal = Math.max(maxVal, num);
        }

        int cost = 0;
        int[] tree = new int[maxVal - minVal + 2];
        for (int i = 0; i < N; ++i) {
            int curr = nums[i] - minVal + 1;
            cost += Math.min(getSum(tree, curr - 1), i - getSum(tree, curr));
            cost %= MOD;
            addNum(tree, curr, 1);
        }

        return cost;
    }

    private int getSum(int[] tree, int idx) {
        int sum = 0;
        for (int i = idx; i > 0; i -= lowbit(i)) {
            sum += tree[i];
            sum %= MOD;
        }
        return sum;
    }

    private void addNum(int[] tree, int idx, int val) {
        for (int i = idx; i < tree.length; i += lowbit(i)) {
            tree[i] += val;
        }
    }

    private int lowbit(int x) {
        return x & -x;
    }
}
