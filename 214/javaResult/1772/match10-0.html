class Solution {
    public final int MOD = 1000000007;
    
    public int createSortedArray(int[] instructions) {
        Bit tree = new Bit();
        int count = 0;
        long res = 0;
        for(int v : instructions) {
            int left = tree.getSum(v - 1);
            int right = count - tree.getSum(v);
            res += Math.min(left, right);
            res %= MOD;
            tree.update(v, 1);
            count++;
        }
        return (int)res;
    }
    
    class Bit {
        private int max = 100000;
        private int[] tree = new int[max + 1];
        
        public void update(int index, int v) {
            for(int i = index; i <= max; i += (i & -i))
                tree[i] += v;
        }
        
        public int getSum(int n) {
            int res = 0;
            for(int i = n; i > 0; i -= (i & -i))
                res += tree[i];
            return res;
        }
    }
}
