class Solution {
    public int createSortedArray(int[] instructions) {
        int max = 100001, mod = 1000000007;
        int[] bits = new int[max];
        Map<Integer, Integer> map = new HashMap<>();
        long res = 0;
        for (int i = 0;i < instructions.length;i++) {
            int in = instructions[i];
            map.put(in, map.getOrDefault(in, 0) + 1);
            insert(bits, in);
            int lessOrEqual = search(bits, in);
            // System.out.println(lessOrEqual);
            res = (res + Math.min(i + 1 - lessOrEqual, lessOrEqual - map.get(in))) % mod;
        }
        return (int)res;
    }
     private void insert(int[] bit, int i) {
        while (i < bit.length) {
            bit[i] += 1;
            i += i & -i;
        }
    }
    private int search(int[] bit, int i) {
        int res = 0;
        while (i > 0) {
            res += bit[i];
            i -= i & -i;
        }
        return res;
    }
}
