class Solution {
    public int createSortedArray(int[] instr) {
        int max = 0;
        for (int v : instr) max = Math.max(max, v);
        
        
        Tree t = new Tree(max);
        long count = 0, res = 0, mod = 1_000_000_007;
        int[] map = new int[max + 1];
        
        for (int v : instr) {
            int under = 
t.sum(v - 1);
            res = (res + Math.min(under, count - under - map[v]))%mod;

            map[v]++;
            t.add(v);
            count++;
        }
        
        return (int) res;
    }
    
    private static class Tree {
        int[] arr;
        int n;

        Tree(int n) {
            this.n = n;
            arr = arr = new int[n + 1];
        }
        
        void add(
int pos) {
            while (pos <= n) {
                arr[pos]++;
                pos += lsb(pos);
            }
        }
        
        int sum(int end) {
            int res = 0;
            while (end > 0) {
                res += arr[end];
                end -= lsb(end);
            }
            return res;
        }
        
        int lsb(int v) {
            return v & (-v);
        }
    }
}
