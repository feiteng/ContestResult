class Solution {
    public int createSortedArray(int[] instructions) {
        TreeSet<Integer> set = new TreeSet<>();
        for(int a : instructions){
            set.add(a);
        }
        int n = set.size();
        int[] cnt = new int[n + 1];
        TreeMap<Integer, Integer> treemap = new TreeMap<>();
        int idx = 1;
        for(Integer i : set){
            treemap.put(i, idx);
            idx++;
        }
        long res = 0;
        BIT bit = new BIT(n + 1);
        for(int a : instructions){
            idx = treemap.get(a);
            // System.out.println(idx);
            bit.update(idx, 1);
            int t1 = idx == 1 ? 0 : bit.query(idx - 1), t2 = bit.query(idx), t3 = bit.query(n);
            int t = Math.min(t1, t3 - t2);
            res += t;
            res %= 1000000007;
        }
        return (int)res;
    }
    class BIT{
        int[] c;
        int n;
        public BIT(int n){
            c = new int[n + 1];
            this.n = n;
        }
        int lowerbit(int u){
            return u & (-u);
        }
        public void update(int u, int cnt){
            while(u <= n){
                c[u] += cnt;
                u += lowerbit(u);
            }
        }
        public int query(int u){
            int res = 0;
            while(u > 0){
                res += c[u];
                u -= lowerbit(u);
            }
            return res;
        }
    }
}
