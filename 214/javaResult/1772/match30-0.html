class Solution {
    int[] C = new int[100001];
    public int createSortedArray(int[] instructions) {
        int MOD = 1000000007;
        long ans = 0;
        for(int num : instructions) {
            add(num, 1);
            ans += Math.min(sum(num - 1), sum(100000) - sum(num));
            ans %= MOD;
        }
        return (int)ans;
    }

    void add(int i, int d) {
        while(i < C.length) {
            C[i] += d;
            i += lowbit(i);
        }
    }

    int sum(int i) {
        int ans = 0;
        while(i > 0) {
            ans += C[i];
            i -= lowbit(i);
        }
        return ans;
    }

    int lowbit(int i) {
        return i &(-i);
    }
}
