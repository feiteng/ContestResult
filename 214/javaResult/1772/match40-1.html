class Solution {
    public int createSortedArray(int[] instructions) {
        FenwickTree ft = new FenwickTree(1_000_01);
        int[] count = new int[100001];
        
        int len = 0;
        long res = 0;
        for(int i : instructions){
            int small = ft.sum(i - 1);
            res += Math.min(small, len - count[i] - small);
            
            count[i]++;
            len++;
            ft.add(i, 1);
        }
        
        return (int) (res % 1_000_000_007);
    
}
}

class FenwickTree {
  int[] array;
  int size;


  public FenwickTree(int size) {
    this.size = size;
    array = new int[size];
  }

  public void add(int loc, int val){
    while(loc < size){
      array[loc] += val;
      loc += loc & (-loc);
    }
  }

  public int sum(int loc){
    int res = 0;
    while(loc > 0){
      res += array[loc];
      loc -= loc & (-loc);
    }

    return res;
  }
}
