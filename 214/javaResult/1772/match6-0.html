class Solution {
    static int mod = (int) 1e9 + 7;
    public int createSortedArray(int[] ins) {
        int max = 0;
        for (int i : ins) if (i > max) max = i;
        long[] bit = new long[max + 1];
        long res = 0;
        for (int i = 0; i < ins.length; ++i) {
            res = (res + Math.min(get(bit, ins[i] - 1), i - get(bit, ins[i]))) % mod;
            update(bit, ins[i]);
        }
        return (int) res;
    }
    private void update(long[] bit, int i) {
        while (i < bit.length) {
            bit[i]++;
            i += i & -i;            
        }
    }
    private long get(long[] bit, int i) {
        long res = 0;
        while (0 < i) {
            res += bit[i];
            i -= i & -i;
        }
        return res;
    }
}
