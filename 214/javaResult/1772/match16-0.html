class Solution {
    
    long[] Bit;
    int mod = (int)(1e9+7);
    public int createSortedArray(int[] a) {
        Bit = new long[(int)(1e5+1)];
        long ans = 0;
        for(int i=0;i<a.length;++i){
            long left = query(a[i]-1);
            long right = i - query(a[i]);
            ans+=Math.min(left,right);
            if(ans>=mod) ans-=mod;
            update(a[i]);
        }
        
        return (int)ans;
    }
    
     long query(int i) {
			long sum = 0;
			while(i > 0) {
				sum+=Bit[i];
				i -= -i&i;
			}
			return sum;
		}
		void update(int i) {
			int n = Bit.length;
			while(i < n) {
				Bit[i]++;
				i += -i&i;
			}
			
		}
}
