class Solution {
	long BIT[];

	void up
date(int ind, int vl) {
		for (int i = ind; i < BIT.length; i += (i & -i))
			BIT[i] += vl;
	}

	long query(int ind) {
		long ans = 0;
		for (int i = ind; i >= 1; i -= (i & -i)) {
			ans += BIT[i];
		}
		return ans;
	}

	public int createSortedArray(int[] instructions) {
		BIT = new long[100010];
		long sum = 0;

		for (int i = 0; i < instructions.length; i++) {
			long lt = query(instructions[i] - 1);
			long gt = i - query(instructions[i]);
			update(instructions[i],1);
			sum += Math.min(lt, gt);
		}
		return (int) (sum%1000000007);
	}
}
