class Solution {
    class BinaryIndexedTree {

        int n = 1;
        long C[];

        BinaryIndexedTree(int sz) {
            while (n <= sz) {
                n <<= 1;
            }
            n = sz + 1;
            C = new long[n];
        }

        
int lowbit(int x) {
            return x & -x;
        }

        void add(int x, long val) {
            while (x < n) {
                C[x] += val;
                x += lowbit(x);
            }
        }

        long getSum(int x) {
            long res = 0;
            while (x > 0) {
                res += C[x];
                x -= lowbit(x);
            }
            return res;
        }

        int binSearch(long sum) {
            if (sum == 0) {
                return 0;
            }
            int n = C.length;
            int mx = 1;
            while (mx < n) {
                mx <<= 1;
            }
            int res = 0;
            for (int i = mx / 2; i >= 1; i >>= 1) {
                if (C[res + i] < sum) {
                    sum -= C[res + i];
                    res += i;
                }
            }
            return res + 1;
        }
    }
    public int createSortedArray(int[] instructions) {
        int n = instructions.length;
        
BinaryIndexedTree bit = new BinaryIndexedTree((int) 1e5);
        long rs = 0;
        long mod = (long) (1e9 + 7);
        for (int i = 0; i < n; i++) {
            bit.add(instructions[i], 1);
            long mn = bit.getSum(instructions[i] - 1);
            long equal = bit.getSum(instructions[i]);
            rs = (rs + Math.min(mn, i + 1 - equal)) % mod;
        }
        return (int) rs;
    }
}
