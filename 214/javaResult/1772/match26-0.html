class Solution {
    public int createSortedArray(int[] arr) {
        int len = arr.length;
        long[] nums = new long[100001], prev = new long[100001];
        SegmentTree st = new SegmentTree(nums);
        long count = 0;
        for(int i = 0; i < len; i++) {
            count = (count + Math.min(st.get(0, arr[i]), st.get(arr[i] + 1, 100001))) % (1000000007);
            st.update(arr[i], prev[arr[i]] + 1);
            prev[arr[i]]++;
        }
        return (int)count;
    }
    
    private class SegmentTree{
        long[] tree;
        int n;
        public SegmentTree(long[] arr) {
            n = arr.length;
            tree = new long[2 * n];
            for(int i = 0; i < n; i++) tree[i + n] = arr[i];
            for(int i = n - 1; i >= 0; i--) {
                tree[i] = tree[2 * i] + tree[2 * i + 1];
            }
        }
        
        public void update(int i, long value) {
            i += n;
            tree[i] = value;
            while(i > 1) {
                i /= 2;
                tree[i] = tree[2 * i] + tree[2 * i + 1];
            }
        }
        
        public long get(int left, int right) {
            left += n;
            right += n;
            long sum = 0;
            while(left < right) {
                if((left&1) == 1) {sum += tree[left]; left++;}
                if((right&1) == 1) {right--; sum += tree[right];}
                left >>= 1;
                right >>= 1;
            }
            return sum;
        }
    }
}
