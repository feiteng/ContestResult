class Solution {
    long ans;
    long[] sum;
    int mod = 1000000007;

    public int createSortedArray(int[] instructions) {
        ans = 0;
        sum = new long[100005];
        
for (int i = 0; i < instructions.length; i++) {
            long x = findSum(100000);
            long y = findSum(instructions[i]);
            long z = findSum(instructions[i] - 1);
            ans = (ans + Math.min(x - y, z)) % mod;
            update(instructions[i]);
        }
        return (int) ans;
    }

    
private long findSum(int x) {
        long res = 0;
        while (x > 0) {
            res += sum[x];
            x -= x & -x;
        }
        return res;
    }

    private void update(int x) {
        while (x <= 100000) {
            sum[x] += 1;
            x += x & -x;
        }
    }
}
