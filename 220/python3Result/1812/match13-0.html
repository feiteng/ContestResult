class Solution:
    def reformatNumber(self, number: str) -> str:
        number = number.replace(" ", "").replace("-", "")
        arr = []
        idx = 0
        while idx < len(number) - 4:
            arr.append(number[idx:idx+3])
            idx += 3
        if idx+4 == len(number):
            arr.append(number[idx:idx+2])
            arr.append(number[idx+2:idx+4])
        else:
            arr.append(number[idx:])
        
        return "-".join(arr)
