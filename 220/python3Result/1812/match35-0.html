class Solution:
    def reformatNumber(self, number: str) -> str:
        
        L = []
        for ch in number:
            if ch == '-' or ch == ' ':
                continue
            L.append(ch)
        
        groups = []
        i = 0
        while len(L) - i > 4:
            groups.append(''.join(L[i:i+3]))
            i += 3
        
        if len(L) - i == 4:
            groups.append(''.join(L[i:i+2]))
            groups.append(''.join(L[i+2:]))
        else:
            groups.append(''.join(L[i:]))
        
        return '-'.join(groups)
