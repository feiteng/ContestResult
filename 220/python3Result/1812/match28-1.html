class Solution:
    def reformatNumber(self, number: str) -> str:
        number = ''.join(''.join(number
.split()).split('-'))
        res = []
        while len(number) > 4:
            res.append(number[:3])
            number = number[3:]
        if len(number) < 4:
            res
.append(number)
        else:
            res.append(number[:2])
            res.append(number[2:])
        return '-'.join(res)
