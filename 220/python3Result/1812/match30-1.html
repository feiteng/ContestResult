class Solution:
    def reformatNumber(self, number: str) -> str:
        s = number.replace(' ', '')
.replace('-', '')
        b = []
        n = len(s)
        i = 0
        while n - i > 4:
            b.append(s[i: i + 3])
            i += 3
        if n - i < 4:
            b.append(s[i:])
        elif n - i == 4:
            b.append(s[i: i + 2])
            b.append(s[i + 2:])
        return '-'.join(b)
