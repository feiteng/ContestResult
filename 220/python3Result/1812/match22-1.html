class Solution:
    def reformatNumber(self, number: str) -> str:
        numbers = "".join([c for c in number if c.isdigit()])
        index = 0
        n = len(numbers)
        result = []
        while index + 4 < n:
            result.append(numbers[index: index + 3])
            index += 3
        if index + 2 == n:
            result.append(numbers[index: index + 2])
        elif index + 3 == n:
            result
.append(numbers[index: index + 3])
        elif index + 4 == n:
            result.append(numbers[index: index + 2])
            result.append(numbers[index + 2:])
        return "-".join(result)
