class Solution:
    def reformatNumber(self, numbers: str) -> str:
        numbers = numbers.replace("-", "").replace(" ", "")
        res = []
        n = len(numbers)
        m = 0
        while m + 4 < n:
            res.append(numbers[m:(m+3)])
            m += 3
        if n - m == 2 or n -m == 3:
            res.append(numbers[m:n])
        elif n - m == 4:
            res.append(numbers[m:(m+2)])
            res.append(numbers[(m+2):(m+4)])
        return "-".join(res)
