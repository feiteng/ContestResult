class Solution:
    def reformatNumber(self, n: str) -> str:
        n = n.replace(' ', '').replace('-', '')
        s = ''
        while len(n) > 4:
            s += n[:3] + '-'
            n = n[3:]
        if len(n) == 4:
            s += n[:2] + '-' + n[2:]
        else:
            s += n
        return s
