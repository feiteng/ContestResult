class Solution:
    def reformatNumber(self, ss: str) -> str:
        ss = ss.replace('-','')
        ss = ss.replace(' ','')
        zs = []
        while len(ss)>4:
            zs.append(ss[:3])
            ss= ss[3:]
        if len(ss) ==4:
            zs.append(ss[:2])
            zs.append(ss[2:])
        else:
            zs.append(ss)
        return '-'.join(zs)
