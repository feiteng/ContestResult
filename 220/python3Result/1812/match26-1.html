class Solution:
    def reformatNumber(self, num: str) -> str:
        num = num.replace(" ", "")
        num = num
.replace("-", "")
        res = ""
        if len(num) % 3 == 1:
            for i in range(0, len(num) - 4, 3):
                res += num[i:i+3] + "-"
            res += num[-4:-2] + "-" + num[-2:]
        if len(num) % 3 == 2:
            for i in range(0, len(num) - 2, 3):
                res += num[i:i+3] + "-"
            res += num[-2:]
        if len(num) % 3 == 0:
            for i in range(0, len(num), 3):
                res += num[i:i+3] + "-"
            res = res[:-1]
        return res
