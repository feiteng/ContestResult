class Solution:
    def reformatNumber(self, number: str) -> str:
        s_list = []
        for char in number:
            if char>="0" and char <="9":
                s_list.append(char)
        s = "".join(s_list)
        # print(s)
        s_len = len(s)
        left = s_len % 3
        tail = 0
        if left == 1:
            tail = 4
        elif left == 2:
            tail = 2
        res = []
        i 
= 0
        while i + tail <s_len:
            res.append(s[i:i+3])
            i += 3
        if tail == 4:
            res.append(s[i:i+2])
            res.append(s[i+2:])
        elif tail == 2:
            res.append(s[i:])
        return "-".join(res)
