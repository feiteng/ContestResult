class Solution:
    def reformatNumber(self, number: str) -> str:
        arr = [x for x in number if x.isnumeric()]
        res 
= []
        i = 0
        while i < len(arr) - 4:
            res.append(''.join(arr[i: i+3]))
            i += 3
        if i == len(arr) - 4:
            res.append(''.join(arr[i: i+2]))
            res.append(''.join(arr[i+2:]))
        else:
            res.append(''.join(arr[i:]))
        return '-'.join(res)
