class Solution:
    def reformatNumber(self, number: str) -> str:
        number = number.replace(" ","").replace("-","")
        ans = []
        n = len(number)
        i = 0
        while n - i > 4:
            ans.append(number[i:i+3])
            i += 3
            
        val = n - i
        if val <= 3:
            ans.append(number[i:i+3])
        else:
            ans.append(number[i:i+2])
            ans.append(number[i+2:i+4])
        return "-".join(ans)
