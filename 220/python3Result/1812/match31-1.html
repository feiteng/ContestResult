class Solution:
    def reformatNumber(self, number: str) -> str:
        n = number.replace(" ", "")
        n 
= n.replace("-", "")
        a = []
        i = 0
        while i + 4 < len(n):
            a.append(n[i:i+3])
            i += 3
        if i + 4 == len(n):
            a.append(n[-4:-2])
            a
.append(n[-2:])
        else:
            a.append(n[i:])
        return "-".join(a)
