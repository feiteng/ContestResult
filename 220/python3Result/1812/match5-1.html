class Solution:
    def reformatNumber(self, number: str) -> str:
        num = "".join(number
.replace(" ", "-").split("-"))
        ret = ""
        while (len(num) > 4):
            ret = ret + num[:3] + "-"
            num = num[3:]
        
        if (len(num) == 4):
            ret = ret + num[:2] + "-" + num[2:]
        else:
            ret = ret + num
        
        return ret
