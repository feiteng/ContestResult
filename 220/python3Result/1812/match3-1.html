class Solution:
    def reformatNumber(self, number: str) -> str:
        number = number.replace('-', '').replace(' ', '')
        out = ''
        while len(number) > 4:
            out = out + number[:3] + '-'
            number = number[3:]
        
        if len(number) == 4:
            out = out + number[:2]+'-'+number[2:]
        else:
            out = out + number
        return out
