class Solution:
    def reformatNumber(self, number: str) -> str:
        numbers = number.replace(' ', '').replace('-', '')
        res = []
        i = 0
        while i < len(numbers) - 4:
            res.append(numbers[i:i+3])
            i+=3
        if i + 4 == len(numbers):
            res.append(numbers[i:i+2])
            res.append(numbers[i+2:])
        else:
            res.append(numbers[i:])
        return '-'.join(res)
