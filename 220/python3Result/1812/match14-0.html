class Solution:
    def reformatNumber(self, number: str) -> str:
        number = number.replace(" ", "")
        number = number.replace("-", "")
        n = len(number)
        i = 0
        res = []
        while n - i > 4:
            res.append(number[i:i+3])
            i += 3
        if n - i != 4:
            res.append(number[i:])
        else:
            res.append(number[i:i+2])
            res.append(number[i+2:])
        return "-".join(res)
