class Solution:
    def reformatNumber(self, number: str) -> str:
        char_list = []
        for c in number:
            if c!='-' and c!=' ':
                char_list.append(c)
        tmp_str = ''.join(char_list)
        res_list = []
        n = len(tmp_str)
        i = 0
        while n-i>4:
            res_list.append(tmp_str[i:i+3])
            i+=3
        if n-i==4:
            res_list.append(tmp_str[i:i+2])
            res_list.append(tmp_str[i+2:i+4])
        else:
            res_list.append(tmp_str[i:n])
        return '-'.join(res_list)
