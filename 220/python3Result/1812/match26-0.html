class Solution:
    def reformatNumber(self, a: str) -> str:
        a = a.replace(" " , "").replace("-" , "")
        ret = ""
        if len(a) % 3 == 1:
            for i in range(0, len(a) - 4, 3):
                ret += a[i : i + 3] + "-"
            ret += a[-4:-2] + "-" + a[-2 :]
        if len(a) % 3 == 2:
            for i in range(0, len(a) - 2, 3):
                ret += a[i:i + 3] + "-"
            ret += a[-2:]
        if len(a) % 3 == 0:
            for i in range(0, len(a), 3):
                ret += a[i : i + 3] + "-"
            ret = ret[: -1]
        return ret
