class Solution:
    def reformatNumber(self, number: str) -> str:
        s=number
        s=s.replace("-","")
        s=s.replace(" ","")
        cur=[]
        i=0
        while len(s)-i>4:
            cur.append(s[i:i+3])
            i+=3
        if len(s)-i==4:
            cur.append(s[i:i+2])
            cur.append(s[i+2:i+4])
        else:
            cur.append(s[i:])
        return "-".join(cur)
