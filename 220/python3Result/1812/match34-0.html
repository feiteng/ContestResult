from typing import *


class Solution:
    def reformatNumber(self, v: str) -> str:
        v = [i for i in v if i != '-' and i != ' ']
        v = "".join(v)
        c = 0
        l = []
        while len(v) - c > 4:
            l.append(v[c:c + 3])
            c += 3
        if len(v)-c == 2:
            l.append(v[c:c + 2])
        elif len(v)-c == 3:
            l.append(v[c:c + 3])
        elif len(v)-c == 4:
            l.append(v[c:c + 2])
            l.append(v[c+2:c + 4])
        return "-".join(l)
