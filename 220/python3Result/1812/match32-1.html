class Solution:
    def reformatNumber(self, number: str) -> str:
        number = number.replace(' ', '')
        number 
= number.replace('-', '')
        
        results = []
        n = len(number)
        index = 0
        while index + 4 < n:
            results.append(number[index:(index + 3)])
            index += 3
        
        if index + 3 >= n:
            results.append(number[index:])
        else:
            results.append(number[index:(index + 2)])
            results.append(number[(index + 2):(index + 4)])
            
        return '-'.join(results)
