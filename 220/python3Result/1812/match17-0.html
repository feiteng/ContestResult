class Solution:
    def reformatNumber(self, number: str) -> str:
        s = number.replace('-', '').replace(' ', '')
        ans = []
        idx = 0
        while idx + 4 < len(s):
            ans.append(s[idx:idx + 3])
            idx += 3
        if idx + 4 == len(s):
            ans.append(s[idx:idx + 2])
            ans.append(s[idx + 2:])
        else:
            ans.append(s[idx:])
        return '-'.join(ans)
