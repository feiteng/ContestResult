def find(arr, i):
    if arr[i] != i:
        arr[i] = find(arr, arr[i])
    return arr[i]

def union(arr, i, j):
    pi, pj = find(arr, i), find(arr, j)
    if pi != pj:
        arr[pi] = pj
        
def same(arr, i, j):
    return find(arr, i) == find(arr, j)

class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edges = sorted(edgeList, key=(lambda _: _[2]))
        queries = sorted([(*q, i) for i,q in enumerate(queries)], key=(lambda _: _[2]))
        
        i = 0 # edges
        j = 0 # queries
        arr = list(range(n))
        ans = [False] * len(queries)
        while j < len(queries):
            while (i < len(edges)) and (edges[i][2] < queries[j][2]):
                a,b,_ = edges[i]
                union(arr, a, b)
                i += 1
                #print(f"adding ({a}, {b}): {arr}")
            a,b,_,k = queries[j]
            ans[k] = same(arr, a, b)
            j += 1
        return ans
