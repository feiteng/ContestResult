    def merge(self, u, v):
        ru = self.root(u)
        rv = self.root(v)
        if ru == rv: return
        du = self.state[ru]
        dv = self.state[rv]
        if du > dv: ru, rv = rv, ru
        if du == dv: self.state[ru] -= 1

class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edgeList.sort(key=lambda x:-x[2])
        uf=UnionFind(n)
        qn=len(queries)
        ans=[None]*qn
        pqd=[]
        for i,(u,v,d) in enumerate(queries):
            pqd.append((d,u,v,i))
        pqd
.sort()

        for lim,p,q,i in pqd:
            while edgeList and edgeList[-1][-1]<lim:
                u,v,d=edgeList.pop()
                uf.merge(u,v)
            ans[i]=(uf.root(p)==uf.root(q))

        return ans
