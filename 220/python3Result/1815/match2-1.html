class DJSet:
    def __init__ (self, n):
        self.a = [i for i in range(n)]
        self.sz = [1 for _ in range(
n)]
        
    def find(self, x):
        if self.a[x] != x:
            self.a[x] = self.find(self.a[x])
        return self.a[x]
        
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        if self.sz[px] < self.sz[py]:
            px, py = py, px
        self.a[py] = px
        self.sz[px] += self.sz[py]
        
return True

class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        qn = len(queries)
        ans = [False for _ in range(qn)]
        
        edgeList.sort(key=lambda x: x[2])
        ei = 0
        en = len(edgeList)
        
        a = [(i, queries[i]) for i in range(qn)]
        a.sort(key=lambda x: x[1][2])
        
        djSet = DJSet(n)
        
        for i, query in a:
            x, y, d = query
            while ei < en and edgeList[ei][2] < d:
                djSet.union(edgeList[ei][0], edgeList[ei][1])
                ei += 1
            if djSet.find(x) == djSet.find(y):
                ans[i] = True
            else:
                ans[i] = False
        
        return ans
