class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edgeList.sort(key=lambda e: e[2])
        queries = [
(p, q, limit, i) for i, (p, q, limit) in enumerate(queries)]
        queries.sort(key=lambda e: e[2])
        ans = []
        ids = list(range(n))
        
def find(i):
            if ids[i] != i:
                ids[i] = find(ids[i])
            return ids[i]
        
        def union(i, j):
            i, j = find(i), find(j)
            if i != j:
                ids[i] = j
        
        index = 0
        for p, q, limit, _ in queries:
            while index < len(edgeList) and edgeList[index][2] < limit:
                i, j, _ = edgeList[index]
                union(i, j)
                index += 1
            ans.append(find(p) == find(q))
        
        ret = [None] * len(queries)
        for res, (_, _, _, i) in zip(ans, queries):
            ret[i] = res
        return ret
