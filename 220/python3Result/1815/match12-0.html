class DSU:
    def __init__(self, n):
        self.d = [i for i in range(n)]
    def find(self, n0):
        if self.d[n0] != n0:
            self.d[n0] = self.find(self.d[n0])
        return self.d[n0]
    def union(self, n0, n1):
        s0 = self.find(n0)
        s1 = self.find(n1)
        self.d[s0] = s1
class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        dsu = DSU(n)
        edgeList.sort(key = lambda x: x[2])
        res = [0] * len(queries)
        idx = sorted(range(len(queries)), key = lambda x: queries[x][2])
        i = 0
        for qi in idx:
            n0, n1, limit = queries[qi]
            while i < len(edgeList) and edgeList[i][2] < limit:
                dsu.union(edgeList[i][0], edgeList[i][1])
                i += 1
            res[qi] = dsu.find(n0) == dsu.find(n1)
        return res
