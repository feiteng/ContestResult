class UF:
    def __init__(self, n):
        self.roots = [i for i in range(n)]
        
    def union(self, node1, node2):
        root1 = self.find(node1)
        root2 = self.find(node2)
        if root1 < root2:
            self.roots[root2] = root1
        else:
            self.roots[root1] = root2
            
    def find(self, node):
        if self.roots[node] != node:
            self.roots[node] = self.find(self.roots[node])
        return self.roots[node]
    
        
class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edgeList.sort(key=lambda x: x[2])
        
        uf = UF(n)
        i = 0
        
        q = [[i, u, v, d, False] for i, (u, v, d) in enumerate(queries)]
        q.sort(key=lambda x: x[3])
        i = 0
        for (u, v, dist) in edgeList:
            while i < len(q) and dist >= q[i][3]:
                root1 = uf.find(q[i][1])
                root2 = uf.find(q[i][2])
                q[i][4] = (root1 == root2)
                # print("query", q[i])
                i += 1
 
            uf.union(u, v)
            # print(u, v, dist, uf.roots)
        
        while i < len(q):
            root1 = uf.find(q[i][1])
            root2 = uf.find(q[i][2])
            q[i][4] = (root1 == root2)
            # print("query", q[i])
            i += 1

        
        q.sort(key=lambda x: x[0])
        # print(q)
        return [r for _, _, _, _, r in q]
