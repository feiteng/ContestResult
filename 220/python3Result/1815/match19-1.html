class UF:
    def __init__(self,n):
        self.p=list(range(n))
        self.s=
[1]*n
    
    def find(self,x):
        if self.p[x]!=x:
            self.p[x]=self.find(self.p[x])
        return self.p[x]
    
    def union(self,x,y):
        xr,yr=self.find(x),self.find(y)
        if self
.s[xr]<self.s[yr]:
            xr,yr=yr,xr
        self.p[yr]=xr
        self.s[xr]+=self.s[
yr]

class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edgeList.sort(key=itemgetter(2))
        qs=[]
        ans=[False]*len(queries)
        for i,v in enumerate(queries):
            pi,qi,lim=v
            qs.append((i,pi,qi,lim))
        qs.sort(key=itemgetter(3))
        
        # print(qs)
        # print(edgeList)
        ind=0
        DJS=UF(n)
        for q in qs:
            i,pi,qi,lim=q
            while ind<len(edgeList):
                u,v,dis=edgeList[ind]
                # print(dis)
                if dis>=lim:
                    break
                DJS.union(u,v)
                ind+=1
            ans[i]=(DJS.find(pi)==DJS.find(qi))
        return ans
