class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        m = len(queries)
        idxs = sorted([i for i in range(m)], key=lambda i: queries[i][2])
        
        f = [i for i in range(10 ** 5 + 5)]
        
def find(i):
            if f[i] != i:
                f[i] = find(f[i])
            return f[i]
        
        def merge(a, b):
            fa = find(a)
            fb = find(b)
            if fa != fb:
                f[fa] = fb
        
        edgeList.sort(key=lambda e:e[2])
        ee = len(edgeList)
        res = [True] * m
        
        j = 0
        for idx in idxs:
            while j < ee and edgeList
[j][2] < queries[idx][2]:
                merge(edgeList[j][0], edgeList[j][1])
                j += 1
            res[idx] = find(queries[idx][0]) == find(queries[idx][1])
        
        return res
