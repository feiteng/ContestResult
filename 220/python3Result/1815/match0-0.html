class DSU:
    def __init__(self, N):
        self.size = N
        self.par = [i for i in range(self.size)]
        self.sz = [1 for _ in range(self.size)]
    
    def find(self, u):
        if self.par[u] != u:
            self.par[u] = self.find(self.par[u])
        return self.par[u]
    
    
    def union(self, u, v):
        pu, pv = self.find(u), self.find(v)
        if pu == pv:
            return False
        
        if self.sz[pu] < self.sz[pv]:
            pu, pv = pv, pu
        
        self.par[pv] = pu
        self.sz[pu] += self.sz[pv]
        return True
    
class Solution:
    def distanceLimitedPathsExist(self, n: int, edges: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edges.sort(key=lambda e: e[2])
        queries = [(i, *q) for i, q in enumerate(queries)]
        queries.sort(key=lambda q: q[3])
        
        ans = [False for _ in range(len(queries))]
        
        dsu = DSU(n)
        i = 0
        for a, p, q, limit in queries:
            while i < len(edges) and edges[i][2] < limit:
                dsu.union(edges[i][0], edges[i][1])
                i += 1
            
            if dsu.find(p) == dsu.find(q):
                ans[a] = True
        
        
        return ans
