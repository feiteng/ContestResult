class DisjointSets:
    
    def __init__(self, N):
        self.parents = [i for i in range(N)]
        
    def getRoot(self, i):
        if self.parents[i] == i:
            return i
        self.parents[i] = self.getRoot(self.parents[i])
        return self.parents[i]
        
    def union(self, i, j):
        self.parents[self.getRoot(i)] = self.getRoot(
j)

class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        disjointsets = DisjointSets(n)
        edgeList.sort(key=lambda l:l[2])
        queries = [queries[i] + [i] for i in range(len(queries))]
        queries.sort(key=lambda l:l[2])
        idx = 0
        res = [None for _ in range(len(queries))]
        for u, v, limit, i in queries:
            
while idx < len(edgeList) and edgeList[idx][2] < limit:
                disjointsets.union(edgeList[idx][0], edgeList[idx][1])
                idx += 1
            res[i] = disjointsets.getRoot(u) == disjointsets.getRoot(v)
        return res
