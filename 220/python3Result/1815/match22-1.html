class UF:
    def __init__(self, n: int):
        self.n = n
        self.root = list(range
(n))
    def find(self, u: int) -> int:     
        if u == self.root[u]:
            return u
        self.root[u] = self.find(self.root[u]) 
        return self.root[u]  
    def unite(self, u: int, v: int) -> bool:  
        ru, rv = self.find(u), self.find(v)
        if ru == rv:
            return False
        self.root[ru] = rv
        return True
    
class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        uf = UF(n)
        m = len(queries)
        ans = [False] * m
        edgeList.sort(key=lambda x: x[2])
        for i in range(m):
            queries[i].append(i)
        queries.sort(key=lambda x: x[2])
        i = 0
        for (u, v, limit, index) in queries:
            
while i < len(edgeList) and edgeList[i][2] < limit:
                uf.unite(edgeList[i][0], edgeList[i][1])
                i += 1
            ans[index] = (uf.find(u) == uf.find(v))       
        return ans
