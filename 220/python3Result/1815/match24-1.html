class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        el = sorted(edgeList, key=lambda x: x[2])
        ans = [False] * len(queries)
        eqv = [i for i in range(n)]
        
def ref(i):
            if eqv[i] != i:
                eqv[i] = ref(eqv[i])
            return eqv[i]
        def unify(i, j):
            ii, jj = ref(i), ref(j)
            if ii != jj:
                eqv[ii] = jj
        iq = list(enumerate(queries))
        k = 0
        for q in sorted(iq, key=lambda x: x[1][2]):
            v = q[1][2]
            while k < len(el) and el[k][2] < v:
                unify(el[k][0], el[k][1])
                k += 1
            ans[q[0]] = (ref(q[1][0]) == ref(q[1][1]))
        return ans
