class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        edgeList.sort(key = lambda x: x[2]) # sort by distance
        allQ = [(queries[i], i) for i in range(len(queries))] # zipped with idx
        allQ.sort(key = lambda x: x[0][2]) # sort by limit
        q = len(queries)
        
        uf = UnionFind(n)
        qIdx = 0
        res = [False for i in range(q)]
        
        eIdx = 0
        for query, idx in allQ:
            x, y, limit = query
            while eIdx < len(edgeList) and limit > edgeList[eIdx][2]:
                u, v, dis = edgeList[eIdx]
                uf.union(u, v)
                eIdx += 1
            
            result = uf.isConnected(x, y)
            res[idx] = result
            
        return res
