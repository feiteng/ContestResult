class UnionFind:
    def __init__(self, n):
        self.head = list(range(n))
        self.rank = [0] * n
    def find_head(self,n):
        hn = self.head[n]
        if(hn!=self.head[hn]):
            self.head[n] = self.find_head(hn)
        return self.head[n]
    def union(self,a,b):
        a = self.find_head(a)
        b = self
.find_head(b)
        if a == b:
            return
        if self.rank[a]>self.rank[b]:
            self.head[b]=a
        elif self.rank[a]<self.rank[b]:
            self.head[a]=b
        else:
            self.rank[a]+=1
            self.head[b]=a
            
            
class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        unionFind = UnionFind(n)
        edgeList.sort(key=lambda x:x[2])
        ans = []
        p_e = 0
        l_e = len(edgeList)
        que 
= []
        for no,(a,b,dis) in enumerate(queries):
            que.append((dis,a,b,no))
        que.sort(key=lambda x:x[0])
        
        for dis,a,b,no in que:
            while(p_e < l_e and edgeList[p_e][2]<dis):
                unionFind.union(edgeList[p_e][0], edgeList[p_e][1])
                p_e += 1
            if unionFind.find_head(a) == unionFind.find_head(b):
                ans.append((no,True))
            else:
                ans.append((no,False))
        ans.sort()
        an = [i[1] for i in ans]
        return an
