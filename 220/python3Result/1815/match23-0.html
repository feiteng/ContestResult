class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:       
        
        def find(x):
            if up[x] != x:
                up[x] = find(up[x])
            return up[x]
        
        def union(x, y):
            a, b = find(x), find(y)
            if a != b:
                up[a] = b
                            
        up = [i for i in range(n)]
        l = len(queries)
        ret = [False] * l
        inq = sorted([[j[2], i] for i, j in enumerate(queries)])
        edgeList.sort(key=lambda x: x[2])
        i = 0
        m = len(edgeList)        
        for limit, idx in inq:
            while i < m and edgeList[i][2] < limit:
                union(edgeList[i][0], edgeList[i][1])                
                i += 1
            a, b, c = queries[idx]
            ret[idx] = find(a) == find(b)
        return ret
