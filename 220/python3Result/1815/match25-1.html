class Solution:
    def distanceLimitedPathsExist(self, n: int, edgeList: List[List[int]], queries: List[List[int]]) -> List[bool]:
        ds = DisjSet(n)
        edgeList.sort(key=lambda x: x[2])
        queries = [tuple(queries[i] + [i]) for i in range(len(queries))]
        queries.sort(key=lambda x: x[2])
        currEdge = 0
        m = {}
        for (p, q, lim, i) in queries:
            while (currEdge < len(edgeList) and edgeList[currEdge][2] < lim):
                u = edgeList[currEdge][0]
                v = edgeList[currEdge][1]
                currEdge += 1
                ds.Union(u, v)
            m[i] = ds.find(p) == ds.find(q)
        return [m[i] for i in range(len(queries))]
                
        
        

class DisjSet: 
    def __init__(self, n): 
        self.rank = [1] * n 
        self.parent = [i for i in range(
n)] 
  
  
    def find(self, x): 
        if (self.parent[x] != x): 
            self.parent[x] = self.find(self.parent[x]) 
        return self.parent[x] 
  
  
    def Union(self, x, y): 
        xset = self.find(x) 
        yset = self
.find(y) 
  
        if xset == yset: 
            return
        if self.rank[xset] < self.rank[yset]: 
            self.parent[xset] = yset 
  
        elif self.rank[xset] > self.rank[yset]: 
            self.parent[yset] = xset 
  
        else: 
            self.parent[yset] = xset 
            self.rank[xset] = self.rank[xset] + 1
