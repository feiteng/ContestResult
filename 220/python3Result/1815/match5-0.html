def getfa(x, fa):
    if not x==fa[x]:
        fa[x] = getfa(fa[x], fa)
    return fa[x]

def merge(x, y, fa):
    x = getfa(x, fa)
    y = getfa(y, fa)
    if not x==y:
        fa[x] = y

class Solution:
    def distanceLimitedPathsExist(self, n: int, e: List[List[int]], q: List[List[int]]) -> List[bool]:
        nq = len(q)
        idx = [i for i in range(nq)]
        idx.sort(key = lambda x:q[x][2])
        fa = [i for i in range(n)]
        ans = [0 for i in range(nq)]
        e.sort(key = lambda x:x[2])
        c = 0
        for i in range(nq):
            j = idx[i]
            while c<len(e) and e[c][2]<q[j][2]:
                merge(e[c][1], e[c][0], fa)
                c += 1
            ans[j] = getfa(q[j][0], fa)==getfa(q[j][1], fa)
        return ans
