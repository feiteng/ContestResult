class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        count = collections.defaultdict(int)
        cursum = 0
        res = 0
        j = 0
        for i in range(len(nums)):
            count[nums[i]]+=1
            cursum+=nums[i]
            while count[nums[i]] > 1 and j < i:
                count[nums[j]]-=1
                cursum-=nums[j]
                j+=1
            res = max(res,cursum)
        return res
