class Solution:
    def maximumUniqueSubarray(self, arr: List[int]) -> int:
        
        i = 0
        j = 1

        set = {}
        set[arr[0]] = 1
        sum = arr[0]
        maxsum = sum

        while (i < len(arr) - 1 and
               j < len(arr)):

            if arr[j] not in set:
                sum = sum + arr[j]
                maxsum = max(sum, maxsum)

                set[arr[j]] = 1
                j += 1

            else:
                sum -= arr[i]
                del set[arr[i]]
                i += 1

        return maxsum
