class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        cur = set()
        i = 0
        j = 0
        res = 0
        s = 0
        while j < len(nums):
            if nums[j] not in cur:
                cur.add(nums[j])
                s += nums[j]
                res = max(res, s)
            else:
                
while i < j and nums[i] != nums[j]:
                    cur.remove(nums[i])
                    s -= nums[i]
                    i += 1
                i += 1
            j += 1
        
        return res
