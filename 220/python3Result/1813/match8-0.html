class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        seen = set()
        i, j = 0, 0
        res = 0
        cnt = 0
        for j in range(len(nums)):
            while nums[j] in seen:
                seen.remove(nums[i])
                cnt -= nums[i]
                i += 1
            seen.add(nums[j])
            cnt += nums[j]
            res = max(res, cnt)
        return res
