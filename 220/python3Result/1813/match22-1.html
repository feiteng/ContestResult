class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        s = set()
        ans = 0
        cnt 
= 0
        i = 0
        for j, v in enumerate(nums):
            while v in s:
                s.remove(nums[i])
                cnt -= nums[i]
                i += 1
            s.add(v)
            cnt += v
            ans = max(ans, cnt)
        return ans
