class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        i = 0
        s = set()
        sm = 0
        res = 0
        for j in range(len(nums)):
            while nums[j] in s:
                s.remove(nums[i])
                sm -= nums[i]
                i += 1
            s.add(nums[j])
            sm += nums[j]
            res = max(res, sm)
        return res
