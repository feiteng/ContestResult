class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        check = set()
        ans = -1
        n = len(nums)
        i = j = cur = 0
        while j < n:
            while nums[j] in check:
                check.remove(nums[i])
                
                cur -= nums[i]
                i += 1
            check.add(nums[j])
            cur += nums[j]
            ans = max(ans, cur)
            j += 1
        return ans
