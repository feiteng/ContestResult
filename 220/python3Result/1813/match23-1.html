class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        ans = 0
        curr 
= 0
        s = set()
        i = 0
        for j in range(len(nums)):
            while nums[j] in s:
                curr -= nums[i]
                s.remove(nums[i])
                i += 1
            s.add(nums[j])
            curr += nums[j]
            ans = max(ans, curr)
        return ans
