class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        L = 0
        res = set()
        sums = 0
        maxs = 0
        for R,n in enumerate(nums):
            while(n in res):
                res.remove(nums[L])
                sums -= nums[L]
                L+=1
            res.add(n)
            sums+=n
            maxs = max(maxs,sums)
        return maxs
