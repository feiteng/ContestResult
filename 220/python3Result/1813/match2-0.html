class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        cnt = collections.Counter()
        j = 0
        res = val = 0
        for i in range(len(nums)):
            cnt[nums[i]] += 1
            val += nums[i]
            while cnt[nums[i]] > 1:
                cnt[nums[j]] -= 1
                val -= nums[j]
                j += 1
            res = max(res, val)
        return res
