class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        presum = nums[:]
        for i in range(1, len(presum)):
            presum[i] += presum[i-1]
        i = -1
        idx = {}
        res = 0
        for j, x in enumerate(nums):
            if x in idx:
                i 
= max(i, idx[x])
            idx[x] = j
            res = max(presum[j] - (presum[i] if i >=0 else 0), res)
        return res
