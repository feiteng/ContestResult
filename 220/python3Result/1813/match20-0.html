class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        counters=[0]*10001
        cur=0
        res=0
        l,r=0,0
        while r<len(nums):
            cur+=nums[r]
            counters[nums[r]]+=1
            while l<r and counters[nums[r]]>1:
                cur-=nums[l]
                counters[nums[l]]-=1
                l+=1
            res=max(res,cur)
            r+=1
        return res
