class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        l = 0
        exist = set()
        now 
= 0
        result = 0
        for i in range(len(nums)):
            # exist.add(nums[i])
            while (nums[i] in exist):
                exist.remove(nums[l])
                now -= nums[l]
                l += 1
            exist.add(nums[i])
            now += nums[i]
            result = max(result, now)
        return result
