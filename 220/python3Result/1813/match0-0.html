class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        d = set()
        cursum = 0
        ans = 0
        i = 0
        for j in range(n):
            while nums[j] in d:
                d.remove(nums[i])
                cursum -= nums[i]
                i += 1
            d.add(nums[j])
            cursum += nums[j]
            ans = max(ans,cursum)
        return ans
