class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        cs = collections.defaultdict(int)

        i
=j=0;
        s = 0
        mx = 0

        while j<len(nums):
            s+= nums[j]
            cs[nums[j]]+=1
            while cs[nums[j]]>1:
                s-=nums[i]
                cs[nums[i]]-=1
                i+=1
            mx = max(mx,s)
            j+=1
        return mx
