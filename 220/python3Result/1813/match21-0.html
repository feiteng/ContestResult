class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        i = 0; has = set(); cur = res = 0
        for j, v in enumerate(nums):
            while v in has:
                has.remove(nums[i])  
                cur -= nums[i]
                i += 1
            has.add(v)
            cur += v
            res = max(res, cur)
        return res
