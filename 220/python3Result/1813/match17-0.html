class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        l = 0
        s = set()
        res = 0
        cur = 0
        for r in range(len(nums)):
            while nums[r] in s:
                s.discard(nums[l])
                cur -= nums[l]
                l += 1
            cur += nums[r]
            s.add(nums[r])
            res = max(res, cur)
        return res
