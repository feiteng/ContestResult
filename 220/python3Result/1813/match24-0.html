class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        mem = set([])
        i = 0
        j = 0
        res = -1
        cur = 0
        while j < len(nums):
            if not nums[j] in mem:
                mem.add(nums[j])
                cur += nums[j]
            else:
                while (i < j and nums[i] != nums[j]):
                    mem.remove(nums[i])
                    cur -= nums[i]
                    i+=1
                i += 1
            j += 1
            res = max(res, cur)
        return res
