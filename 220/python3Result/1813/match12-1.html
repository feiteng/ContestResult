class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        w = set()
        left = 0
        ans = 0
        score = 0
        for right in range(len(nums)):
            while nums[right] in w:
                w.remove(nums[left])
                score -= nums[left]
                left += 1
            w.add(nums[right])
            score += nums[right]
            ans = max(ans,score)
        return ans
