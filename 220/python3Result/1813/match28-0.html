class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        seen = set()
        cur_sum = 0
        max_sum = 0
        i = 0
        for n in nums:
            while (n in seen):
                seen.remove(nums[i])
                cur_sum -= nums[i]
                i += 1
            
            seen.add(n)
            cur_sum += n
            max_sum = max(max_sum, cur_sum)
        return max_sum
