class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        
        window = set()
        l = 0
        res = 0
        cur = 0
        for r in range(len(nums)):
            
            while nums[r] in window:
                window.remove(nums[l])
                cur -= nums[l]
                l += 1
            window.add(nums[r])
            cur += nums[r]
            res = max(res, cur)
        
        return res
