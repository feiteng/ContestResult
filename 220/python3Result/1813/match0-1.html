class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        seen = set()
        N = len(nums)
        
        total = 0
        left = 0
        best = 0
        for right in range(N):
            while nums[right] in seen:
                seen.remove(nums[left])
                total -= nums[left]
                left += 1
            
            seen.add(nums[right])
            total += nums[right]
            best = max(best, total)
        return best
