class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        # total = sum(nums)
        s = set([])
        best = cur = 0
        l = 0
        for r in range(len(nums)):
            while nums[r] in s:
                s.remove(nums[l])
                cur -= nums[l]
                l += 1
            s.add(nums[r])
            cur += nums[r]
            best = max(best, cur)
        return best
