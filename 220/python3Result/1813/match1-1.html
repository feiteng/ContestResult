class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        cnt = collections.Counter()
        i = 0
        res = 0
        summ = 0
        for j in range(len(nums)):
            cnt[nums[j]] += 1
            summ += nums[j]
            while cnt[nums[j]] > 1:
                cnt[nums[i]] -= 1
                summ -= nums[i]
                i += 1
                
            #print(i, j, cnt, summ)
            res = max(res, summ)
        return res
