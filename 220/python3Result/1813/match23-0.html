class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        seen = set()
        start = 0
        largest = float('-inf')
        currSum = 0
        for end in range(len(nums)):
            while nums[end] in seen:
                currSum -= nums[start]
                seen.remove(nums[start])
                start += 1
            seen.add(nums[end])
            currSum += nums[end]
            largest = max(largest, currSum)
        return largest
