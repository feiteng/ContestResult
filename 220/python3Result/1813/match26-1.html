class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        i = 0
        j = 1

        s = dict()

        s
[nums[0]] = 1

        sums = nums[0]
        maxsum = sums

        while (i < len(nums) - 1 and j < len(nums)):

            if nums[j] not in s:
                sums += nums[j]
                maxsum = max(sums, maxsum)
                s[nums[j]] = 1
                j += 1
            else:
                sums -= nums[i]
                del s[nums[i]]
                i += 1

        return maxsum
