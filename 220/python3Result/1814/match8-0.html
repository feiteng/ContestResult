class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        
        dp = [0]*n
        dp[-1] = nums[-1]
        st = deque([(nums[-1], n - 1)])
        
        for i in range(n - 2, -1, -1):
            while st and st[0][1] - i > k:
                st.popleft()
            dp[i] = nums[i] + st[0][0]
            while st and st[-1][0] <= dp[i]:
                st.pop()
            st.append((dp[i], i))
        return dp[0]
