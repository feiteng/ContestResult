import heapq


class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        maxScores = [nums[0]]
        prevVals = [[-nums[0], 0]]
        for i in range(1, len(nums)):
            while i - prevVals[0][1] > k:
                heapq.heappop(prevVals)
            maxScores.append(nums[i] - prevVals[0][0])
            heapq.heappush(prevVals, [-nums[i] + prevVals[0][0], i])
        return maxScores[-1]
