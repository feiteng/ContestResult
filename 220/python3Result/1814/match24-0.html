from collections import deque

class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)

        d = deque()
        d.append((0, nums[0]))
        
        for i in range(1, n):
            while i - d[0][0] > k:
                d.popleft()
            x = d[0][1] + nums[i]
            while d and d[-1][1] <= x:
                d.pop()
            d.append((i, x))
            
        return d[-1][1]
