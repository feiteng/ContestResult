class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        dp = [nums[0]]
        heap = [(-dp[-1], 0)]
        for i in range(1, len(nums)):
            while heap[0][1] < i-k:
                heapq.heappop(heap)
            dp.append(-heap[0][0] + nums[i])
            heapq.heappush(heap, (-dp[-1], i))
        # print(dp)
        return dp[-1]
