class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        dp = [-math.inf] * len(nums)
        dp[0] = nums[0]
        prev_max = 0
        deque = collections.deque([0])
        for i in range(1, len(nums)):
            if deque[0] < i - k:
                deque.popleft()
            dp[i] = dp[deque[0]] + nums[i]
            while deque and dp[deque[-1]] < dp[i]:
                deque.pop()
            deque.append(i)
        return dp[-1]
