class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        import collections
        n=len(nums)
        dp=[float('-inf')]*n
        dp[-1]=nums[-1]
        stack=collections.deque([n-1])
        for i in range(n-2,-1,-1):
            while stack and stack[0]-i>k:
                stack.popleft()
            dp[i]=nums[i]+dp[stack[0]]
            while stack and dp[stack[-1]]<=dp[i]:
                stack.pop()
            stack.append(i)

        return dp[0]
