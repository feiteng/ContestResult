from collections import deque
class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        a=deque([0])
        for i in range(1,len(nums)):
            nums[i]+=nums[a[0]]
            while a and nums[a[-1]]<nums[i]: a.pop()
            a.append(i)
            if a[0]+k==i: a.popleft()
        return nums[-1]
