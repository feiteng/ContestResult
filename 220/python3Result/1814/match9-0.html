class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] * n
        dp[n - 1] = nums[n - 1]
        q = deque([(dp[n - 1], n - 1)])
        
        for i in range(n - 2, -1, -1):
            while q and q[0][1] > i + k:
                q.popleft()
            
            maxi = q[0][0]
            dp[i] = nums[i] + maxi
            
            while q and q[-1][0] < dp[i]:
                q.pop()
            q.append((dp[i], i))
            
            
            
            
        return dp[0]
