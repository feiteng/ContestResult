class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        
        qu = deque()
        qu.append([
0,nums[0]])
        for i in range(1, len(nums)):
            while i-qu[0][0]>k:
                qu.popleft()
            v = nums[i] + qu[0][-1]
            while qu and qu[-1][-1] < v:
                qu.pop()
            qu.append([i,v])
        return qu[-1][-1]
