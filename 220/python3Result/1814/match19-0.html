class Solution:                
    def maxResult(self, nums: List[int], k: int) -> int:  # use dp
        n=len(nums)
        dp=[0]*n
        dp[-1]=nums[-1]
        cur_max=[(nums[-1], n-1)]
        for i in range(n-2,-1,-1):
            while cur_max[0][1]-i>k:
                cur_max.pop(0)
            dp[i]=cur_max[0][0]+nums[i]
            while cur_max and dp[i]>cur_max[-1][0]:
                cur_max.pop(-1)
            cur_max.append((dp[i],i))
        return dp[0]
