class Solution:
    def maxResult(self, A: List[int], k: int) -> int:
        Q = collections.deque([(0, A[0])])
        
        for i in range(1, len(A)):
            
            while Q and Q[0][0] < i - k:
                Q.popleft()
                
            x = A[i] + Q[0][1]
            while Q and x >= Q[-1][1]:
                Q.pop()
            
            Q.append((i, x))
            #print(Q)
        return Q[-1][1]
