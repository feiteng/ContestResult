from collections import deque

class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n=len(nums)
        dp=[-1]*n
        dp[0]=nums[0]
        mx=deque()
        mx.append((
0,nums[0]))

        for i in range(1,n):
            while mx[0][0]<i-k:mx.popleft()
            dp[i]=mx[0][1]+nums[i]
            while mx and mx[-1][-1]<=dp[i]:mx.pop()
            mx.append((i,dp[i]))

        return dp[-1]
