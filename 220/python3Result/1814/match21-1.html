class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        N = len(nums)
        dp = [-1e5] * (N)
        dp
[0] = nums[0]
        pos = [(-dp[0], 0)] # a heap of (-sum, pos)
        for i in range(1, N):
            while(i-pos[0][1]>k):
                heapq.heappop(pos)
            dp[i] = -pos[0][0] + nums[i]
            heapq.heappush(pos, (-dp[i],i))
            
        return dp[N-1]
