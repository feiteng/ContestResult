class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        q = collections.deque([(
nums[0], 0)])

        for i in range(1, n):
            while q and i - q[0][1] > k:
                q.popleft()
            v = q[0][0] + nums[i]
            while q and q[-1][0] <= v:
                q.pop()
            q.append((v, i))

        return q[-1][0]
