from collections import deque

class Solution:
    def worker(nums, k):
        que 
= deque()
        que.append([0, nums[0]])
        for i in range(1, len(nums)):
            while i - que[0][0] > k:
                que.popleft()
            newval = que[0][1] + nums[i]
            while len(que) and (que
[-1][1] < newval):
                que.pop()
            que.append([i, newval])
        return que[-1][1]
    
    def maxResult(self, nums: List[int], k: int) -> int:
        return Solution.worker(nums, k)
        new_nums = [0]
        ans = 0
        for x in nums:
            if x >= 0:
                ans += x
                if len(new_nums) >= k:
                    new_nums.append(0)
                    #print(new_nums)
                    ans += Solution.worker(new_nums, k)
                    new_nums = [0]
            else:
                new_nums.append(x)
        if len(new_nums) > 1:
            ans += nums[-1]
        return ans
