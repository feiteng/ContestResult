class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        # print("-------------")
        n = len(nums)
        ans = nums[0]
        stack = [(0,ans)]
        for i in range(1,n):
            while stack and i-stack[0][0]>k:
                stack.pop(0)
            tmp = stack[0][1] + nums[i]
            # print(stack)
            while stack and stack[-1][1]<tmp:
                stack.pop()
            stack.append((i,tmp))
            # print(stack)
        return stack[-1][1]
