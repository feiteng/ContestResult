class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] * n
        dp
[0] = nums[0]
        que = collections.deque([0])
        for i in range(1, n):
            while que[0] < i - k: que.popleft()
            dp[i] = dp[que[0]] + nums[i]
            while que and dp[que[-1]] <= dp[i]: que.pop()
            que.append(i)
        return dp[-1]
