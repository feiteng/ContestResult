class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        heap = []
        for i in range(len(nums)-1, -1, -1):
            cur = nums[i]
            while heap and heap[0][1] > i+k:
                heapq.heappop(heap)
            if heap:
                cur -= heap[0][0]
            heapq.heappush(heap, (-cur, i))
        return cur
