class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        
        dp = [k for k in nums]
        q = [(nums[-1], n-1)]
        for i in range(n-2, -1, -1):
            # dp[i] += nums[i]
            # print(q)
            while q != [] and q[-1][1] > i + k:
                q.pop()
            dp[i] += q[-1][0]
            while q != [] and q[0][0] < dp[i]:
                q.pop(0)
            q.insert(0, (dp[i], i))
        # print(dp)
        # print("=" * 20)
        return dp[0]
