class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        dp = [nums[0]]
        stack 
= collections.deque([0])
        for i in range(1, len(nums)):
            if i - stack[0] > k:
                stack.popleft()
            dp.append(dp
[stack[0]] + nums[i])
            while stack and dp[-1] > dp[stack[-1]]:
                stack.pop()
            stack.append(i)
        return dp[-1]
