class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        dq = deque()
        dq.append([nums[0], 0])
        for i in range(1, len(nums)):
            while dq and dq[0][1] < i - k:
                dq.popleft()
            v = dq[0][0] + nums[i]
            while dq and dq[-1][0] <= v:
                dq.pop()
            dq.append([v, i])
        return dq[-1][0]
