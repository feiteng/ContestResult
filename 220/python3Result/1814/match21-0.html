class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        print(n)
        dp = [-float('inf')]*n
        dp[0] = nums[0]
        # dp, dp[i-k] to dp[i-1]
        heap = [(-dp[0],0)]
        for i in range(1,n):
            while heap[0][1]<i-k:
                heapq.heappop(heap)
            dp[i] = nums[i]-heap[0][0]
            heapq.heappush(heap,(-dp[i],i))
            
        return dp[-1]
