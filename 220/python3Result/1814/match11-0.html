class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        
        # 滑动窗口最大值
        q = collections.deque()
        q.append((0, nums[0]))
        for i in range(1, len(nums)):
            while q and i - q[0][0] > k:
                q.popleft()
            cur = nums[i] + q[0][1]
            while q and q[-1][1] <= cur:
                q.pop()
            q.append((i, cur))
        return q[-1][1]
