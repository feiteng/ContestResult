class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        L = len(nums)
        dp = [0] * L
        dp
[0] = nums[0]
        pq = [[-dp[0], 0]]
        for i in range(1, L):
            while pq[0][1] < i - k:
                heappop(pq)
            dp[i] = nums[i] - pq[0][0]
            heappush(pq, [-dp[i], i])
        return dp[-1]
