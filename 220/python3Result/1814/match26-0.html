from heapq import heappush, heappop
class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        h = []
        n = len(nums)
        dp = [float('-inf')]*n
        dp[0] = nums[0]
        h = [[-dp[0], 0]]
        for i in range(1, n):
            while h and h[0][1] < i-k:
                heappop(h)
            dp[i] = -h[0][0] + nums[i]
            heappush(h, [-dp[i], i])
        return dp[-1]
