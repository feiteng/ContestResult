class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        le = len(nums)
        dp = [float('-inf')]*le
        dp
[0] = nums[0]
        
        q = [(0, nums[0])]
        
        for i in range(1, le):
            while q and q[0][0]<max(i-k, 0): q
.pop(0)        
            dp[i] = q[0][1]+nums[i]
            while q and q[-1][1]<=dp[i]: q.pop()
            q.append((i, dp[i]))
        
            
        return dp[-1]
