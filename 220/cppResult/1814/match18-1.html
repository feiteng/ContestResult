struct node {
    int pos = 0;
    int val = 0;

    node(int pos, int val) : pos(pos), val(val) {}
};

struct compq {
    bool operator()(const node &n1, const node &n2) {
        return n1.val < n2.val;
    }
};


int dp[100005];
class Solution {
public:

    int maxResult(vector<int>& nums, int k) {
        for(int i=0;i<nums.size();i++) dp[i]=-1E9;
        priority_queue<node,vector<node>,compq> q;
        dp[0]=nums[0];
        q.push(node{0, dp[0]});
        for(int i=1;i<nums.size();i++){
            node temp{0,0};
            while(1){
                temp=q.top();
                if(temp.pos>=i-k)break;
                else q.pop();

            }
            dp[i]=temp.val+nums[i];
            q.push(node{i,dp[i]});
        }
        return dp[nums.size()-1];

    }
};
