class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n=nums.size();
        vector<int> dp(n,INT_MIN);
        priority_queue<pair<int,int>> qu;
        dp[0]=nums[0];
        qu.push({nums[0],0});
        for(int i=1; i<n; i++){
            while(qu.size()&&qu.top().second<i-k) qu.pop();
            dp[i]=qu.top().first+nums[i];
            qu.push({dp[i],i});
        }
        return dp.back();
    }
};
