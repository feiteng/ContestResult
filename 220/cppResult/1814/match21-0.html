class Solution {
public:
	int maxResult(vector<int>& nums, int k) {
		multiset<int> q;
		q.insert(nums[0]);

		int n = nums.size();
		vector<int> dp(n);
		dp[0] = nums[0];

		for (int i = 1; i < n; ++i) {
			int v = nums[i] + *q.rbegin();
			dp[i] = v;
			if (i >= k)
				q.erase(q.find(dp[i-k]));
			q.insert(v);
		}

		return dp[n - 1];;
	}
};
