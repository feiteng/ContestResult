class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        multiset<int> tb;
        int n = nums.size();
        vector<int> dp (n + 1, 0);
        tb.insert(nums[0]);
        dp[1] = nums[0];
        for(int i = 1; i < n; ++i) {
            int last = *tb.rbegin();
            dp[i + 1] = last + nums[i];
            if(i >= k)
                tb.erase(tb.find(dp[i - k + 1]));
            tb.insert(dp[i + 1]);
        }
        //cout<<tb.size()<<endl;
        
        return dp[n];
    }
};
