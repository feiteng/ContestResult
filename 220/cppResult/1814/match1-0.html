class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> f(n);
        multiset<int> S;
        f[0] = nums[0];
        S.insert(f[0]);
        for (int i = 1; i < n; i ++ ) {
            if (i - k - 1 >= 0)
                S.erase(S.find(f[i - k - 1]));
            f[i] = nums[i] + *S.rbegin();
            S.insert(f[i]);
        }
        return f[n - 1];
    }
};
