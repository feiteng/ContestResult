const int MAXN = 1e5;
int ans[MAXN];

class Solution {
public:
    int maxResult(vector<int>& nums, int k){
        // DP using max heap
        int n = int(nums.size());
        multiset<int, greater<int>> choices;
        ans[n-1] = nums[n-1]; choices.insert(ans[n-1]);
        for(int i = n-2; i >= 0; i--){
            ans[i] = (*choices.begin())+nums[i];
            choices.insert(ans[i]);
            // At every turn i choices only store nums[i+1],...,nums[n-1]
            // Next turn: i-1 cannot run to nums[i+k]
            // So this entry should be removed
            if(i < n-k)
                choices.erase(ans[i+k]);
        }
        return ans[0];
    }
};
