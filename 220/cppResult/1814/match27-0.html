class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n, 0);
        dp[0] = nums[0];
        multiset<int> lastk = {nums[0]};
        for (int i = 1; i < n; i++) {
            dp[i] = nums[i] + *lastk.rbegin();
            lastk.insert(dp[i]);
            if (lastk.size() > k) {
                lastk.erase(lastk.find(dp[i - k]));
            }
        }
        return dp[n - 1];
    }
};
