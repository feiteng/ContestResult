class Solution {
public:
    int maxResult(vector<int>& a, int m) {
        int n = a.size();
        vector<int> dp(n);
        dp[n - 1] = a[n - 1];
        deque<int> A;
        A.push_back(n - 1);
        for (int i = n - 2; i >= 0; --i) {
            while (A.size() && A.front() - i > m) A.pop_front();
            dp[i] = a[i] + dp[A.front()];
            while (A.size() && dp[A.back()] <= dp[i]) A.pop_back();
            A.push_back(i);
        }
        return dp[0];
    }
};
