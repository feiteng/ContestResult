class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> ans(n+10,-1e9);
        
        list<int> que;
        ans[0]=nums[0];
        que.push_back(0);
        for(int i=1; i<n; i++){
            while(i-que.front()>k) que.pop_front();
            ans[i] = ans[que.front()]+nums[i];
            while(que.size()&&ans[i]>ans[que.back()]) que.pop_back();
            que.push_back(i);
        }
        return ans[n-1];
    }
};
