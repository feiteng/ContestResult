class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        multiset<int, greater<int>> pq;
        vector<int> dp(n);
        dp[0] = nums[0];
        pq.insert(dp[0]);
        for(int i=1;i<n;i++){
            // for(auto i : pq)
            //     cout<<i<<" ";
            // cout<<"\n";
            dp[i] = *(pq.begin())+nums[i];
            pq.insert(dp[i]);
            if(i>=k)
                pq.erase(dp[i-k]);
        }
        // for(int i=0;i<n;i++)
        //     cout<<dp[i]<<" ";
        // cout<<"\n";
        return dp[n-1];
    }
};
