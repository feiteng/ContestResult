class Solution {
public:
    //维护前k个数的最大值(到达某个位置的最大得分)
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        if (n == 1) return nums[0];
        
        deque<pair<int, int>> dq;
        dq.emplace_back(nums[0], 0);
        
        int cur;
        for (int i = 1; i < n; ++i) {
            while ((i - dq.front().second) > k) 
                dq.pop_front(); 
            cur = dq.front().first + nums[i];
            while (dq.size() && dq.back().first < cur) 
                dq.pop_back();
            dq.emplace_back(cur, i);
            
        }
        return cur;
    }
};
