class Solution {
public:
    // dp[i]
    int maxResult(vector<int>& nums, int k) {
        int n=nums.size();
        vector<int> dp(n,0);
        dp[0]=nums[0];
        deque<int> deq;
        deq.push_back(0);
        for(int i=1;i<n;i++)
        {
            while(i-deq.front()>k) deq.pop_front();
            dp[i]=nums[i]+dp[deq.front()];
            while(!deq.empty()&&dp[deq.back()]<=dp[i])
                deq.pop_back();
            deq.push_back(i);
        }
        return dp[n-1];
    }
};
