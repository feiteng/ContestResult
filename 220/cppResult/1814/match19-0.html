class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        vector<int> dp(nums.size(), 0);
        dp[0] = nums[0];
        multiset<int> count;
        count.insert(dp[0]);
        for(int i = 1; i < nums.size(); i++){
            dp[i] = nums[i] + *count.rbegin();
            if(i >= k) count.erase(count.lower_bound(dp[i-k]));
            count.insert(dp[i]);
        }
        return dp.back();
    }
};
