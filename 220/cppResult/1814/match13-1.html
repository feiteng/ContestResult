class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> f(n);
        deque<int> q;
        f[0] = nums[0];
        q.push_back(0);
        for (int i = 1; i < n; i ++) {
            while (q.front() + k < i) q.pop_front();
            f[i] = f[q.front()] + nums[i];
            while (q.size() && f[q.back()] <= f[i]) q.pop_back();
            q.push_back(i);
        }
        return f[n - 1];
    }
};
