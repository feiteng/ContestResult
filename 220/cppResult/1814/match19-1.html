class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
		int n=nums.size();
		multiset<int> st;
		st.insert(nums[0]);
		for(int i=1;i<n;i++){
			nums[i]+=*prev(st.end());
			st.insert(nums[i]);
			if(i>=k)st.erase(st.find(nums[i-k]));
		}
		return nums[n-1];
    }
};
