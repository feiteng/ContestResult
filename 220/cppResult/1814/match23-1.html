class Solution {
public:
  int maxResult(vector<int>& a, int k) {  
    int n = a.size();
    
    vector<int> f(n + 1, INT_MIN);
    multiset<int> s;
    
    f[0] = a[0];
    s.insert(f[0]);
    
    for (int i = 1; i < n; ++i) {
      // Remove f[i-k-1]
      if (i-k-1 >= 0) s.erase(s.find(f[i-k-1]));
      
      f[i] = *prev(s.end()) + a[i];
      
      // Add f[i]
      s.insert(f[i]);
    }
    
    return f[n-1];
  }
};
