class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        int n=nums.size();
        if(n==1) return nums[0];
        vector<int> dp(n);
        dp[0]=nums[0];

        deque<int> dq;
        dq.push_back(0);
        for(int i=1;i<n;i++){
            while(!dq.empty() && dq.front()<i-k) dq.pop_front();
            dp[i]=dp[dq.front()]+nums[i];
            while(!dq.empty() && dp[i]>=dp[dq.back()]) dq.pop_back();
            dq.push_back(i);
        }
        return dp[n-1];
    }
};
