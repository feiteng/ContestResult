class Solution {
public:
    string reformatNumber(string number) {
        vector<char> digits;
        for (const char c : number) {
            if (c == ' ' || c == '-') {
                continue;
            }
            digits.push_back(c);
        }
        string result;
        int i = 0;
        int n = digits.size();
        while (n - i > 4) {
            result.push_back(digits[i]);
            result.push_back(digits[i + 1]);
            result.push_back(digits[i + 2]);
            result.push_back('-');
            i += 3;
        }
        if (i == n - 4) {
            result.push_back(digits[i]);
            result.push_back(digits[i + 1]);
            result.push_back('-');
            result.push_back(digits[i + 2]);
            result.push_back(digits[i + 3]);
        } else if (i == n - 3) {
            result.push_back(digits[i]);
            result.push_back(digits[i + 1]);
            result.push_back(digits[i + 2]);
        } else {
            result.push_back(digits[i]);
            result.push_back(digits[i + 1]);
        }
        return result;
    }
};
