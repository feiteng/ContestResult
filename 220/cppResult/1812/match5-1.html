class Solution {
public:
    string reformatNumber(string number) {
        string s, res;
        for (auto c : number) if (isdigit(c)) s += c;
        int n = s.length();
        if (n <= 3) return s;


        if (n % 3 == 1) {
            for (int i = 0; i < n-4; i += 3) {
                res += s.substr(i, 3) + "-";
            }
            res += s.substr(n-4, 2) + "-" + s.substr(n-2, 2);
        }
        else if (n % 3 == 0) {
            for (int i = 0; i < n; i += 3) {
                res += s.substr(i, 3) + "-";
            }
            res.pop_back();

        }
        else {
            for (int i = 0; i < n-2; i += 3) {
                res += s.substr(i, 3) + "-";
            }
            res += s.substr(n-2, 2);
        }
        return res;
    }
};
