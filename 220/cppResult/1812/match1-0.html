class Solution {
public:
    string reformatNumber(string number) {
        string ans;
        int count = 0;
        for (auto& i : number) {
            if (isdigit(i)) {
                ans += i;
                if (++count % 3 == 0) {
                    ans += '-';
                }
            }
        }
        if (count % 3 == 0 && ans.length()) {
            ans.pop_back();
        } else if (count % 3 == 1 && ans.length() > 1) {
            swap(ans[ans.length() - 2], ans[ans.length() - 3]);
        }
        return ans;
    }
};
