class Solution {
public:
    string reformatNumber(string number) {
        string digits = "";
        for (auto ch : number) {
            if (isdigit(ch)) {
                digits += ch;
            }
        }
        cout << digits << "\n";
        int n = digits.size();
        string res = "";
        int last = 0;
        string suffix = "";
        if (n % 3 == 1) {
            last = 4;
            suffix += digits.substr(n - 4, 2) + "-" + digits.substr(n - 2);
        } else if (n % 3 == 2) {
            last = 2;
            suffix = digits.substr(n - 2);
        }
        if (n - last == 0) return suffix;
        string group = "";
        int numGroups = (n - last) / 3;
        for (int i = 0; i < numGroups; i++) {
            if (i > 0) res += "-";
            res += digits.substr(3 * i, 3);
        }
        if (!suffix.empty()) {
            res += "-" + suffix;
        }
        return res;
    }
};
