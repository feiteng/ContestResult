class Solution {
public:
    string reformatNumber(string number) {
        string ans = "";
        for(auto p : number){
            if(p >= '0' && p <= '9'){
                ans += p;
            }
        }

        string ret = "";
        for(int i = 0; i < ans.length();){
            if(i + 4 < ans.length()){
                ret += ans.substr(i, 3) + "-";
                i += 3;
            }
            else if(i + 3 < ans.length()){
                ret += ans.substr(i, 2) + "-" + ans.substr(i + 2, 2);
                i += 4;
            }
            else{
                ret += ans.substr(i, ans.length() - i);
                i += ans.length() - i;
            }
        }
        return ret;
    }
};
