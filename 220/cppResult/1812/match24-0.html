class Solution {
public:
    string reformatNumber(string number) {
        string digits;
        for (auto num : number) {
            if (num >= '0' && num <= '9') digits += num;
        }
        
        int n = digits.size(), i = 0;
        vector<string> groups;
        for (; i<n-4; i+=3) {
            groups.push_back(digits.substr(i, 3));
        }
        cout << i << endl;
        if (i == n-2 || i == n-3) groups.push_back(digits.substr(i));
        else if (i == n-4) groups.push_back(digits.substr(i, 2)), groups.push_back(digits.substr(i+2));
        
        string ans;
        for (int i=0; i<groups.size(); i++) {
            ans += groups[i];
            if (i != groups.size()-1) ans += "-";
        }
        
        return ans;
    }
};
