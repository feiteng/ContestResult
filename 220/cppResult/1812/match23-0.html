class Solution {
public:
    string reformatNumber(string number) {
        int n = number.length();
        string other = "";
        for (int i=0; i<n; i++){
            if (number[i] >= '0' && number[i] <= '9') other += number[i];
        }
        int idx = 0;
        n = other.length();
        string ans = "";
        while (n - idx > 4){
            ans += other.substr(idx, 3);
            ans += "-";
            idx += 3;
        }
        if (n - idx == 4){
            ans += other.substr(idx, 2);
            ans += "-";
            idx += 2;
            ans += other.substr(idx, 2);
        } else {
            ans += other.substr(idx);
        }
        return ans;
    }
};
