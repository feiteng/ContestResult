class Solution {
public:
    string reformatNumber(string s) {
        if(s.size() == 0)
        {
            return s;
        }
        
        int n = s.size();
        int l = 0;
        for(int i=0; i<n; i++)
        {
            if(s[i]>='0' && s[i]<='9')
            {
                s[l++] = s[i];
            }

        }
        
        string result;
        int i = 0;
        while(l-i>4)
        {
            result += s.substr(i, 3) + "-";
            i+=3;
        }
        
        if(l-i <=3)
        {
            result += s.substr(i, l-i);
        }
        else
        {
            result += s.substr(i, 2) + "-" + s.substr(i+2, 2);
        }
        
        return result;
    }
};
