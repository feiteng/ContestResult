class Solution {
public:
    string reformatNumber(string number) {
        string tmp = "";
        for(auto item:number)
            if(item>='0'&&item<='9')
            tmp+=char(item);
        int n = tmp.size();
        string ans = "";
        
        for(int i=0;i<n;)
        {
            if(n-i==4)
            {
                ans =ans+ tmp[i]+tmp[i+1]+"-"+tmp[i+2]+tmp[i+3];
                i+=4;
            }
            else if(n-i==3)
            {
                ans =ans+ tmp[i]+tmp[i+1]+tmp[i+2];
                i+=3;
            }
            else if (n-i==2)
            {
                ans =ans+ tmp[i]+tmp[i+1];
                i+=2;
            }
            else
            {
                ans =ans+ tmp[i]+tmp[i+1]+tmp[i+2]+"-";
                i+=3;
            }
        }
        return ans;
    }
};
