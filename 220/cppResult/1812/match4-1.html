class Solution {
public:
    string reformatNumber(string number) {
        string s;
        for (auto c: number) {
            if (isdigit(c))
                s.push_back(c);
        }
        
        string res;
        int m = s.size();
        int i = 0;
        while (i + 4 < s.size()) {
            for (int j = 0; j < 3; ++j) {
                res.push_back(s[i+j]);
            }
            i += 3;
            
            res.push_back('-');
        }
        
        res.push_back(s[i++]);
        res.push_back(s[i++]);
        

        int left = s.size() - i;
        if (left == 2) {
            res.push_back('-');
            res.push_back(s[i++]);
            res.push_back(s[i++]);
        } else if (left == 1) {
            res.push_back(s[i++]);
        }
        
        return res;
    }
};
