class Solution {
public:
    string reformatNumber(string number) {
        queue<char> nums;
        
        for (auto const& num : number) {
            if (num == ' ' || num == '-')
                continue;
            nums.push(num);
        }
        
        string ret;
        while (nums.size() > 4) {
            for (int i = 0; i < 3; i++) {
                ret += nums.front();
                nums.pop();
            }

            ret += '-';
        }
    
        
        if (nums.size() == 4) {
            for (int i = 0; i < 2; i++) {
                ret += nums.front();
                nums.pop();
            }
            ret += '-';
            for (int i = 0; i < 2; i++) {
                ret += nums.front();
                nums.pop();
            }
        } else {
            while (!nums.empty()) {
                ret += nums.front();
                nums.pop();
            }
        }
        return ret;
    }
};
