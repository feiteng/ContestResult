class Solution {
  public:
    string reformatNumber(string number) 
    {
      string ns;
      for (auto c: number) if (c >= '0' and c <= '9') ns += c;
      string r;
      int i = 0;
      while (ns.size() - i > 4) {
        r += ns.substr(i, 3);
        r += '-';
        i += 3;
      }
      if (ns.size() - i == 4) {
        r += ns.substr(i, 2) + '-' + ns.substr(i + 2);
      } else {
        r += ns.substr(i);
      } 
      return r;
    }
};
