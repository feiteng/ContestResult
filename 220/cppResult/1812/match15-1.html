class Solution {
public:
    string reformatNumber(string number) {
    	vector<char> v;
    	for(int i=0;number[i];i++) {
    		if(number[i]>='0' && number[i]<='9') {
    			v.push_back(number[i]);
    		}
    	}

    	string ans="";
    	int now=0,res=v.size();
    	while(res>4) {
    		res-=3;
    		ans+=v[now];

    		ans+=v[now+1];
    		ans+=v[now+2];
    		ans+='-';
    		now+=3;
    	}
    	if(res==2) {
    		ans+=v[now]; ans+=v[now+1];
    	}else if(res==3){
    		ans+=v[now];
    		ans+=v[now+1];
    		ans+=v[now+2];    		
    	}else{
       		ans+=v[now];
    		ans+=v[now+1]; ans += '-';
    		ans+=v[now+2]; ans += v[now+3];
    	}
        return ans;
    }
};
