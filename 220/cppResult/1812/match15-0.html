class Solution {
public:
    string reformatNumber(string number) {
        vector<char> digs;
        for (char dig: number) {
            if ('0' <= dig && dig <= '9') {
                digs.emplace_back(dig);
            }
        }
        reverse(digs.begin(), digs.end());
        
        string phoneNumber = "";
        while (digs.size() > 4) {
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += '-';
        }
        if (digs.size() == 2) {
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
        } else if (digs.size() == 3) {
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
        } else {
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += '-';
            phoneNumber += digs.back(); digs.pop_back();
            phoneNumber += digs.back(); digs.pop_back();
        }
        
        return phoneNumber;
    }
};
