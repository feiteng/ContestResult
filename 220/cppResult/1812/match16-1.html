class Solution {
 public:
  string reformatNumber(string number) {
    string num = "", res = "";

    for (char c : number) {
      if (isdigit(c)) {
        num += c;
      }
    }
    int n = num.size(), i = 0;
    while (n - i > 4) {
      for (int j = 0; j < 3; j++) {
        res += num[i++];
      }
      if (i < n - 1) {
        res += "-";
      }
    }

    if (n - i == 4) {
      res += num[i++];
      res += num[i++];
      res += "-";
      res += num[i++];
      res += num[i++];
    } else {
      while (i < n) {
        res += num[i++];
      }
    }
    return res;
  }
};
