class Solution {
public:
    string reformatNumber(string number) {
        
        string trim;
        
        for(auto ch:number){
            
            if(ch>='0'&&ch<='9')trim.push_back(ch);
        }
        
        int b= trim.size()/3;
        int r= trim.size()%3;
        
        
        string ans;
        
        if(r==1)b--;
        for(int i=0;i<b;i++){
            for(int j=0;j<3;j++)ans.push_back(trim[i*3+j]);
            ans.push_back('-');
        }
        
        if(r==0)ans.pop_back();
        
        if(r==2){
            ans.push_back(trim[trim.size()-2]);
            ans.push_back(trim[trim.size()-1]);
        }
        
        if(r==1){
            ans.push_back(trim[trim.size()-4]);
            ans.push_back(trim[trim.size()-3]);
            ans.push_back('-');
            ans.push_back(trim[trim.size()-2]);
            ans.push_back(trim[trim.size()-1]);
        }
        
        return ans;
        
        
    }
};
