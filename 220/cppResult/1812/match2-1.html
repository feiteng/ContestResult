class Solution {
public:
    string reformatNumber(string number) {
        string s;
        int n = number.size();
        for (int i = 0; i < n; i++) {
            if (number[i] >= '0' && number[i] <= '9') {
                s.push_back(number[i]);
            }
        }
        string ans = "";
        int i;

        n = s.size();
        for (i = 0; i + 4 < n; i += 3) {
            ans += s.substr(i, 3);
            ans.push_back('-');
        }
        if (n - i == 2) {
            ans += s.substr(i);
        } else if (n - i == 3) {
            ans += s.substr(i);
        } else {
            ans += s.substr(i, 2);
            ans.push_back('-');
            i += 2;
            ans += s.substr(i);
        }
        return ans;
    }
};
