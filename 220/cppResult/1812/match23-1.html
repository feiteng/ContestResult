class Solution {
public:
    string reformatNumber(string number) {
        string remove; 
        for (int i = 0; i < number.length(); i++)
        {
            if (number[i] != ' ' and number[i] != '-') remove += number[i]; 
        }
        string ans; 

        int i = 0; 
        while ((remove.length() - i) > 4)
        {
            ans += remove.substr(i, 3); 
            ans += '-';
            i += 3; 
        }
        if (remove.length() - i == 4)
        {
            ans += remove.substr(i, 2), ans += '-', i += 2; 
            ans += remove.substr(i, 2); 
        }
        else
        {
            ans += remove.substr(i);
        }
        return ans; 
    }
};
