class Solution {
public:
    string reformatNumber(string number) {
        int n = number.size();
        string res;
        vector<int> a;
        for (int i = 0; i < n; ++i) {
            if (isdigit(number[i])) {
                a.push_back(number[i] - '0');
            }
        }
        n = a.size();
        int i;

        for (i = 0; n - i > 4; i += 3) {
            res += '0' + a[i];
            res += '0' + a[i + 1];
            res += '0' + a[i + 2];
            res += '-';
        }
        if (i + 3 < n) {
            res += '0' + a[i];
            res += '0' + a[i + 1];
            res += '-';
            res += '0' + a[i + 2];
            res += '0' + a[i + 3];
        } else if (i + 1 < n) {

            for (int j = i; j < n; ++j) {
                res += '0' + a[j];
            }
        }
        return res;
    }
};
