class Solution {
public:
    string reformatNumber(string number) {
        string nums;
        for (auto c: number) {
            if (c >= '0' && c <= '9')
                nums.push_back(c);
        }
        int n = nums.size();
        int right = n % 3;
        if (right == 1 || right == 0)
            right += 3;
        string ans;
        for (int i = 0; i < n - right; i += 3) {
            for (int j = 0; j < 3; ++j)
                ans.push_back(nums[i+j]);
            ans.push_back('-');
        }
        if (right < 4) {
            for (int i = 0; i < right; ++i)
                ans.push_back(nums[n-right+i]);
        } else {
            ans.push_back(nums[n-4]);
            ans.push_back(nums[n-3]);
            ans.push_back('-');
            ans.push_back(nums[n-2]);
            ans.push_back(nums[n-1]);
        }
        return ans;
    }
};
