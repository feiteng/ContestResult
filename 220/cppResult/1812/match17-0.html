class Solution {
public:
    string reformatNumber(string number) {
        string t;
        for(int i=0;i<number.size();i++){
            if(number[i]>='0'&&number[i]<='9'){
                t+=number[i];
            }
        }
        int len=t.size();
        string res;
        if(len%3==1){
            int cnt=0;
            for(int i=0;i<len-4;i++){
                res+=t[i];
                cnt++;
                if(cnt==3){
                    res+='-';
                    cnt=0;
                }
            }
            cnt=0;
            for(int i=max(0,len-4);i<len;i++){
                res+=t[i];
                cnt++;
                if(cnt==2){
                    res+='-';
                    cnt=0;
                }
            }
        }else{
            int cnt=0;
            for(int i=0;i<len;i++){
                res+=t[i];
                cnt++;
                if(cnt==3){
                    res+='-';
                    cnt=0;
                }
            }
        }
        if(res[res.size()-1]=='-'){
            return res.substr(0,res.size()-1);
        }else{
            return res;    
        }
        
    }
};
