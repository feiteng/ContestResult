class Solution {
public:
    string reformatNumber(string number) {
        string num = "";
        for (auto c : number) {
            if (c == '-' || c == ' ') {
                continue;
            }
            num += c;
        }
        int l = num.size();
        string res = "";
        while (num.size() > 0) {
            if (num.size() > 4) {
                res += num.substr(0, 3);
                res += "-";
                num = num.substr(3);
            } else if (num.size() == 4) {
                res += num.substr(0, 2);
                res += "-";
                res += num.substr(2, 2);
                num = "";
            } else {
                res += num;
                num = "";
            }
        }
        return res;
    }
};
