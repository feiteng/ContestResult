class Solution {
public:
    string reformatNumber(string number) {
        string s="";
        string ans="";
        for(char c:number)
        {
            if(c<='9'&&c>='0')
                s+=c;
        }
        int n=s.size();
        if(n<=3)
            return s;
        int k;
        if(n%3==1)
        {
            k=n-4;
        }
        else
        {
            k=n-n%3;
            if(n%3==0)
                k=n-3;
        }
        for(int i=0;i<k;i+=3)
        {
            ans+=s[i];
            ans+=s[i+1];
            ans+=s[i+2];
            ans+='-';
        }
        if(n%3==1)
        {
            ans+=s[k];
            ans+=s[k+1];
            ans+='-';
            ans+=s[k+2];
            ans+=s[k+3];
        }
        else
        {
            for(int i=k;i<n;++i)
            {
                ans+=s[i];
            }
        }
        return ans;
    }
};
