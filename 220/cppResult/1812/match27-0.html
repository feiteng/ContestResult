class Solution {
public:
    string reformatNumber(string number) {
        string numbers = "";
        for(const auto& s : number) {
            if(s >= '0' && s <= '9')
                numbers += s;
        }
        string res = "";
        int i = 0, n = numbers.length();
        for(; i < n - 4; i = i + 3) {
            res += numbers.substr(i, 3) + "-";
        }
        if(n - i == 2) {
            res += numbers.substr(i, 2);
        }
        else if(n - i == 3) {
            res += numbers.substr(i, 3);
        }
        else {
            res += numbers.substr(i, 2) + "-" + numbers.substr(i + 2, 2);
        }

        return res;
    }
};
