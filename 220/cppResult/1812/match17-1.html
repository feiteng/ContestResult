class Solution {
public:
    string reformatNumber(string number) {
        string result;
        int n = number.length();

        for(int i = 0; i < n; i++){
            if(number[i] <= '9' and number[i] >= '0'){
                result+=number[i];
            }
        }
        int sz = result.length();
        string res;
        for(int i = 0; i < sz; i++){
            
        }

        if(sz%3==1){
            int count = 0;
            for(int i = 0; i <= sz-3; i++){ // sz-4 sz-3 sz-2 sz-1
                res += result[i];
                count++;
                if(count % 3 == 0){
                    res += '-';
                }
            }
            res += '-';
            res += result[sz-2];
            res += result[sz-1];

        }
        else {
            int count = 0;
            for(int i = 0; i <= sz-1; i++){ // sz-4 sz-3 sz-2 sz-1
                res += result[i];
                count++;
                if(count % 3 == 0 and i != sz-1){
                    res += '-';
                }
            }
        }
        return res;
    }
};
