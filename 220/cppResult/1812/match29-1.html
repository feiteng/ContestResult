class Solution {
public:
    string reformatNumber(string number) {
        int n=number.size();
        vector<char>res;
        for(int i=0;i<n;i++){
            if(number[i]>='0'&&number[i]<='9'){
                res.push_back(number[i]);
            }
        }
        n=res.size();
        string s="";
        for(int i=0;i<n;i+=3){
            int rest=n-i;

            if(rest<=4){
                if(rest==4){
                    s+=res[i];
                    s+=res[i+1];
                    s+="-";
                    s+=res[i+2];
                    s+=res[i+3];
                }else if(rest==2){
                    s+=res[i];
                    s+=res[i+1];
                }else if(rest==3){
                    s+=res[i];

                    s+=res[i+1];
                    s+=res[i+2];
                }
                break;
            }else{
                s+=res[i];
                s+=res[i+1];
                s+=res[i+2];
                s+="-";
            }
        }
        return s;
    }
};
