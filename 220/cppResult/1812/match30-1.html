class Solution {
public:
    string reformatNumber(string number) {
        string cur;
        for(char c:number) if(isdigit(c)) cur.push_back(c);
        string ans;
        int n = cur.size(), i = 0;
        while(n){

            if(n >4 || n==3){
                for(int j = 0;j<3;j++) ans.push_back(cur[i++]);
                n-=3;
            }else{
                for(int j = 0;j<2;j++) ans.push_back(cur[i++]);
                n-=2;
            }
            ans.push_back('-');
        }
        ans.pop_back();
        return ans;
    }
};
