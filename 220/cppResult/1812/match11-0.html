class Solution {
public:
    string reformatNumber(string number) {
    	string s = "";
    	int n = number.size();
    	for(int i = 0; i < n; i++){
    		if(number[i] >= '0' && number[i] <= '9'){
    			s += number[i];
    		}
    	}
    	//cout << s << '\n';
    	n = s.size();
    	string ans = "";
    	int i = 0;
    	for(i = 0; n - i > 4 ; i += 3){
    		ans += s[i];
    		ans += s[i + 1];
			ans += s[i + 2];
			ans += '-';
    	}
        //cout << i << '\n';
    	if(n - i == 2){
    		ans += s[n - 2];
    		ans += s[n - 1];
    	} else if(n - i  == 3){
    		ans += s[n - 3];
    		ans += s[n - 2];
    		ans += s[n - 1];
    	} else {
    		ans += s[n - 4];
    		ans += s[n - 3];
    		ans += '-';
    		ans += s[n - 2];
    		ans += s[n - 1];
    	}
    	return ans;
    }
};
