class Solution {
public:
    string reformatNumber(string number) {
        string origin;
        for (auto ch: number) {
            if (ch != ' ' && ch != '-') {
                origin += ch;
            }
        }
        
        string res;
        int i = 0;
        while (i < origin.size() && origin.size() - i > 4) {
            res += origin.substr(i, 3);
            res += '-';
            i += 3;
        }
        
        if (origin.size() - i == 2 || origin.size() - i == 3) {
            res += origin.substr(i);
        } else if (origin.size() - i == 4) {
            res += origin.substr(i, 2);
            res += '-';
            res += origin.substr(i + 2);
        }
        return res;
    }
};
