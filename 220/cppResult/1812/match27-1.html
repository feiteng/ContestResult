class Solution {
public:
    string reformatNumber(string num) {
        string temp = "";
        for( int i = num.length() - 1; i >= 0; i--) {
            if(isdigit(num[i])) temp = num[i] + temp ;
        }
        if( temp.length() <= 3) return temp;
        string ret = "" ;
        int j = 0;

        int n = temp.length();
        for( ; j < temp.length() - 4; j+= 3) {
            ret = ret + temp.substr(j,3) + "-";
        }
         if( j == temp.length() - 4) {
             ret += temp.substr(n-4,2) + "-" + temp.substr(n-2);
         }
        else if( j == temp.length() - 2){
            ret += temp.substr(n-2);
        }else {
            ret += temp.substr(n-3);
        }
        return ret;
    }
};
