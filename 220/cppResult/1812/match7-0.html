class Solution {
public:
    string reformatNumber(string number) {
        vector<char> digits;
        for (char c: number) {
            if (c >= '0' && c <= '9') digits.push_back(c);
        }
        int ptr = 0;
        string res;
        while (ptr + 4 < digits.size()) {
            res.push_back(digits[ptr]);
            res.push_back(digits[ptr+1]);
            res.push_back(digits[ptr+2]);
            res.push_back('-');
            ptr += 3;
        }
        
        if (ptr + 4 == digits.size()) {
            res.push_back(digits[ptr]);
            res.push_back(digits[ptr+1]);
            res.push_back('-');
            res.push_back(digits[ptr+2]);
            res.push_back(digits[ptr+3]);
        } else if (ptr + 3 == digits.size()) {
            res.push_back(digits[ptr]);
            res.push_back(digits[ptr+1]);
            res.push_back(digits[ptr+2]);
        } else {
            res.push_back(digits[ptr]);
            res.push_back(digits[ptr+1]);
        }
        return res;
    }
};
