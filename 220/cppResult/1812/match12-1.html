class Solution {
public:
    string reformatNumber(string number) {
        string tmp="",ans="";
        for(char c:number){
            if(c!=' '&&c!='-')
                tmp+=c;
        }
        int L = tmp.size();

        int i=0;
        while(i+4<L){
            ans += tmp.substr(i,3);
            ans+="-";
            i+=3;
        }
        if(L-i==2||L-i==3){
            ans += tmp.substr(i,L-i);
        }else if(L-i==4){
            ans += tmp.substr(i,2);
            ans += "-";
            ans += tmp.substr(i+2,2);
        }
        return ans;
    }
};
