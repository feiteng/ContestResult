class Solution {
public:
    string reformatNumber(string number) {
        string tp1="";
        for(int i=0;i<number.length();i++)
        {
            if(number[i]>='0' && number[i]<='9')
            {
                tp1+=number[i];
            }
        }
        
        int id=0;
        string ans="";
        while(id+4<tp1.length()){
            for(int i=0;i<3;i++)
            {
                ans+=tp1[id];
                id++;
            }
            ans+='-';
        }
        int cnt = tp1.length()-id;
        if(cnt==4){
            ans+=tp1[id];id++;
            ans+=tp1[id];id++;
            ans+='-';
            ans+=tp1[id];id++;
            ans+=tp1[id];id++;
        }else
        {
            for(int i=id;i<tp1.length();i++)
                ans+=tp1[id],id++;
        }
        
        return ans;
    }
};
