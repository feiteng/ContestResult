class Solution {
public:
    string reformatNumber(string number) {
        string tmp = "";
        for (auto & c : number) {
            if (c >= '0' && c <='9')
                tmp+=c;
        }
        int len = tmp.size();
        string res = "";
        for (int i = 0; i < len;) {
            if (len - i > 4) {
                res+=tmp.substr(i, 3);
                i+=3;
            }
            else if (len - i ==4) {
                res+=tmp.substr(i,2);
                i+=2;
            }
            else {res+=tmp.substr(i,len-i);i=len;}
            res+="-";
        }
        res.pop_back();
        return res;
    }
};
