class Solution {
public:
    string reformatNumber(string number) {
        string res;
        for(char c:number){
            if (c!=' ' && c!='-') res+=c;
        }
        string final_res;
        for(int i=0;i<res.size();i++){
            if(res.size()>=4 && res.size()%3==1) {
                if(i<res.size()-4){
                    final_res+=res[i];
                    if(i%3==2){
                        final_res+='-';
                    }
                }else{
                    final_res+=res[i];
                    if(i+2==res.size()-1) final_res+='-';
                }
            }else{
                final_res+=res[i];
                if(i!=res.size()-1 && i%3==2){
                     final_res+='-';
                }
            }
        }
        return final_res;
    }
};
