class Solution {
public:
    string reformatNumber(string number) {
        string pure = "";
        for (int i = 0; i < number.size(); ++i) {
            if (number[i] - '0' >= 0 && number[i] - '0' <= 9) pure += number[i];
        }
        if (pure.size() <= 3) return pure;
        string answer = "";
        if (pure.size() % 3 == 1){
            for (int i = 0; i < pure.size() - 4; i += 3) {
                answer += (pure.substr(i, 3) + "-");
            }
            answer += (pure.substr(pure.size() - 4, 2) + "-" + pure.substr(pure.size() - 2, 2));
        }
        else if (pure.size() % 3 == 2){
            for (int i = 0; i < pure.size() - 5; i += 3) {
                answer += (pure.substr(i, 3) + "-");
            }
            answer += (pure.substr(pure.size() - 5, 3) + "-" + pure.substr(pure.size() - 2, 2));
        }
        else {
            for (int i = 0; i < pure.size(); i += 3) {
                answer += (pure.substr(i, 3) + "-");
            }
            answer = answer.substr(0, answer.size()-1);
        }
        return answer;
    }
};
