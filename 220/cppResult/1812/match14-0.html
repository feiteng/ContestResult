class Solution {
public:
    string reformatNumber(string number) {
        string s="";
        for(char i : number){
            if(i>='0'&&i<='9')s+=i;
        }
        string ans="";
        int n=s.size();
        if(n%3==0){
            for(int i=0;i<n;i++){
                ans+=s[i];
                if(i!=n-1 && i%3==2 && i!=0)ans+='-';
            }
            return ans;
        }
        if(n%3==1){
            for(int i=0;i<n-4;i++){
                ans+=s[i];
                if(i!=n-1 && i%3==2&&i!=0)ans+='-';
            }
            ans+=s[n-4];
            ans+=s[n-3];
            ans+='-';
            ans+=s[n-2];
            ans+=s[n-1];
            return ans;
        }
        if(n%3==2){
            for(int i=0;i<n-2;i++){
                ans+=s[i];
                if(i!=n-1 && i%3==2 && i!=0)ans+='-';
            }
            ans+=s[n-2];
            ans+=s[n-1];
            return ans;
        }
        return ans;
    }
};
