class Solution {
public:
    string reformatNumber(string number) {
        int n=number.size();
        if(n==2 || n==3) return number;
        
        queue<char> q;
        for(char c:number) {
            if(c>='0' && c<='9')
                q.push(c);
        }
        string result="";
        
        
        while(q.size()>4) {
            for(int i=0; i<3; i++) {
                result+=q.front();
                q.pop();
            }
            if(!q.empty()) result+="-";
        }
        
        if(q.size() == 4) {
            for(int i=0; i<2; i++) {
                result+=q.front();
                q.pop();
            }
            result+="-";
            for(int i=0; i<2; i++) {
                result+=q.front();
                q.pop();
            }
        } else {
            while(!q.empty()) {
                result+=q.front();
                q.pop();
            } 
        }
        
        
        return result;
    }
};
