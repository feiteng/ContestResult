class Solution {
public:
    string reformatNumber(string number) {
        string ans;
        int cnt = 0;
        for (auto ch : number) {
            if (ch >= '0' && ch <= '9') {
                ans.push_back(ch);
                cnt++;
            }
            
            if (cnt == 3) {
                ans.push_back('-');
                cnt = 0;

            }
        }
        
        if (ans.back() == '-') {
            ans.pop_back();
        } else if (ans[ans.size() - 2] == '-') {
            swap(ans[ans.size() - 3], ans[ans.size() - 2]);
        }
        
        return ans;
    }
};
