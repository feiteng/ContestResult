class Solution {
public:
    string reformatNumber(string number) {
        string ans = "";
        int N = number.length();
        for (int i=0;i<N;i++) {
            if (number[i] >= '0' && number[i] <= '9') {
                ans += number[i];
            }
        }
        string ans2 = "";
        N = ans.length();
        int i;
        for (i=0;i<N - 4;i+=3) {
            ans2 += ans[i];
            ans2 += ans[i+1];
            ans2 += ans[i+2];
            ans2 += '-';
        }
        if (i == N - 4) {
            ans2 += ans[i];
            ans2 += ans[i+1];
            ans2 += '-';

            ans2 += ans[i+2];
            ans2 += ans[i+3];
            
        } else if (i == N - 3) {
            ans2 += ans[i];
            ans2 += ans[i+1];
            ans2 += ans[i+2];
        } else {
            assert(i == N - 2);
            ans2 += ans[i];
            ans2 += ans[i+1];
        }
        return ans2;
    }
};
