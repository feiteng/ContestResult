class Solution {
public:
    string reformatNumber(string number) {
        string s;
        for(char c : number){
            if(c <= '9' && c >= '0')
                s = s + c;
        }
        for(int i = 0 ; i < s.size() ; i+=4){
            if(s.size() - i <= 4){
                if(s.size() - i == 4){
                    s = s.substr(0,i+2) + "-" + s.substr(i+2);
                }
                break;
            }
            else{
                s = s.substr(0,i+3) + "-" + s.substr(i+3);
            }
        }
        return s;
    }
};
