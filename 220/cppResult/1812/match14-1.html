class Solution {
public:
    string reformatNumber(string number) {
        string s = "";
        for(char c: number) {
            if('0' <= c && c <= '9') s += c;
        }
        int n = s.size();
        // cout << s << " " << n << endl;
        if(n < 4) return s;

        string ans = "";
        if(n % 3 == 0) {
            // cout << "0" << endl;
            for(int i = 0; i < n; ++i) {
                ans += s[i];
                if(i < n - 1 && i % 3 == 2) ans += "-";
            }

        } else if(n % 3 == 1) {
            for(int i = 0; i < n - 2; ++i) {
                ans += s[i];
                if(i < n - 3 && i % 3 == 2) ans += "-";
            }
            ans += "-";
            ans += s[n - 2];
            ans += s[n - 1];

        } else {
            for(int i = 0; i < n - 2; ++i) {
                ans += s[i];
                if(i < n - 3 && i % 3 == 2) ans += "-";
            }
            ans += "-";
            ans += s[n - 2];
            ans += s[n - 1];
        }
        return ans;
    }
};
