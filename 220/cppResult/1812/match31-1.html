class Solution {
public:
    string reformatNumber(string number) {
        string res;
        for(char c:number){
            if(c!=' ' && c!='-')
                res += c;
        }
        string temp;
        int i=0, n=res.size();
        while(n-i>4){
            temp += res.substr(i,3) + "-";
            i += 3;
        }
        if(n-i==4)
            temp += res.substr(i,2) + "-" + res.substr(i+2,2);
        else
            temp += res.substr(i);
        return temp;
    }
};
