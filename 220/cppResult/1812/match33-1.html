class Solution {
public:
    string reformatNumber(string number) {
        int n = number.size();
        string newS;
        for (int i = 0; i < n; i++) {
            if (number[i] >= '0' && number[i] <= '9') {
                newS += number[i];
            }
        }

        int i = 0;
        string res;
        int nn = newS.size();
        while (i < nn - 4) {
            res += newS.substr(i, 3);
            res += '-';
            i += 3;
        }
        if (i > nn - 4) {
            res += newS.substr(i, nn - i + 1);
        } else {
            res += newS.substr(i, 2);
            res += '-';
            res += newS.substr(i+2, 2);
        }
        return res;
    }
};
