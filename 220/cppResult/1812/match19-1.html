class Solution {
public:
    string reformatNumber(string number) {
        string str;
        for (char c : number) {
            if (c == ' '|| c == '-')
                continue;
            
            str+= c;

        }
        
        int i = 0;
        string result;
        while (str.size() - i - 1 >= 4) {
           result += str.substr(i, 3) + "-";
           i+=3;
        }
        
       
        if (str.size() - i == 4) { 
            result += str.substr(i, 2) + "-" + str.substr(i+2);
        } else {
            result += str.substr(i);
        }
        
        return result;
    }
};
