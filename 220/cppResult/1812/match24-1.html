class Solution {
public:
    string reformatNumber(string number) {
        string tmp;
        
        for(auto ch : number) {
            if(ch >= '0' && ch <= '9')
                tmp += ch;
        }
        
        int len = tmp.size();
        
        if(len % 3 == 1)
            return pt(tmp);
        
        return print(tmp);
    }
    
    string print(string tmp) {
        string res;
        for(int i = 0; i < tmp.size(); ++i) {
            res += tmp[i];
            if(i > 0 && i % 3 == 2 && i < tmp.size() - 1)
                res += '-';
        }
        
        return res;
    }
    
    string pt(string tmp) {
        if(tmp.size() == 1)
            return tmp;
        
        string res;
        for(int i = 0; i < tmp.size() - 4; ++i) {
            res += tmp[i];
            if(i > 0 && i % 3 == 2 && i < tmp.size() - 1)
                res += '-';
        }
        

        for(int i = tmp.size() - 4; i < tmp.size(); ++i) {
            res += tmp[i];
            if(i > 0 &&i == tmp.size() - 3)
                res += '-';
        }
        
        return res;
    }
    
};
