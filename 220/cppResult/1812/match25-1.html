class Solution {
public:
    string reformatNumber(string number) {
        string phone = "";
        for (char c : number) {
          if (c != '-' && c != ' ') phone += c;
        }

        string ans = "";
        for (int i = 0; i < phone.length();) {
          if (i + 4 == phone.length()) {
            ans += phone.substr(i, 2) + '-' + phone.substr(i + 2, 2);
            i += 4;
          } else if (i + 3 <= phone.length()) {
            ans += phone.substr(i, 3) + '-';
            i += 3;
          } else if (i + 2 <= phone.length()) {
            ans += phone.substr(i, 2) + '-';
            i += 2;
          } else {
            assert(false);
          }
        }
        if (ans.back() == '-') ans.pop_back();
        return ans;
    }
};
