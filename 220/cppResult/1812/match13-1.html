class Solution {
public:
    string reformatNumber(string number) {
    	string tmp =  "";
    	for (auto c : number) {
    		if (c != ' ' && c != '-') {
    			tmp += c;
    		}
    	}


    	string res = "";
    	for (int i = 0; i < tmp.size(); i += 3) {
    		if (res.size() != 0) res += '-';
    		res += tmp.substr(i, 3);
    	}

    	if (tmp.size() % 3 == 1) {
    		string s;
    		s += tmp[tmp.size() - 2];
    		s += tmp[tmp.size() - 1];
    		res = res.substr(0, res.size() - 3);
    		res += '-';
    		res += s;
    	}
    	return res;

    }
};
