class Solution {
public:
    string reformatNumber(string number) {
        string temp;
        for(char c:number){
            if(c>='0'&&c<='9'){
                temp+=c;
            }
        }
        string ans;
        // cout<<temp;
        if(temp.size()%3==0){
            for(int i=0;i<temp.size();i++){
                
                ans+=temp[i];
                if(i%3==2){
                    ans+='-';
                }
                
            }
            ans.pop_back();
        }else if(temp.size()%3==1){
            for(int i=0;i<temp.size()-4;i++){
                
                ans+=temp[i];
                if(i%3==2){
                    ans+='-';
                }
                
            }
            ans+=temp[temp.size()-4];
            ans+=temp[temp.size()-3];
            ans+='-';
            ans+=temp[temp.size()-2];
            ans+=temp[temp.size()-1];
        }else{
            for(int i=0;i<temp.size();i++){
                
                ans+=temp[i];
                if(i%3==2){
                    ans+='-';
                }
                
            }
            
        }
        return ans;

    }
};
