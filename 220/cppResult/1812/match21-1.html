class Solution {
public:
    string reformatNumber(string number) {
        string s;
        for (char& c : number) if (c >= '0' && c <= '9') s.push_back(c);
        string ret;
        int n = s.length();
        int pos = 0;
        while (n)
        {
            if (!ret.empty()) ret.push_back('-');

            if (n > 4)
            {
                ret += s.substr(pos, 3);
                pos += 3;
                n -= 3;
            }
            else if (n == 4)
            {
                ret += s.substr(pos, 2) + "-";
                pos += 2;
                ret += s.substr(pos, 2);
                n = 0;
            }
            else
            {
                ret += s.substr(pos, n);
                n = 0;
            }
        }
        return ret;
    }
};
