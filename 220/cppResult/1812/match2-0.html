class Solution {
public:
    string reformatNumber(string number) {
        string s;
        stringstream ss;
        int t = 0;
        for (int i = 0; i < number.size(); ++i) {
            if (number[i] != ' ' && number[i] != '-') {
                ss << number[i];
                t++;
            }
        }
        ss >> s;
        int j = 0;
        string s2 = "";
        for (j = 0; j <= t - 5; j = j + 3) {
            s2 = s2 + s.substr(j, 3) + '-';
        }
        if (t-j-4 == 0) {
            s2 = s2 + s.substr(j, 2) + "-" + s.substr(j+2, 2); 
        } else if (t-j-3 == 0) {
            s2 = s2 + s.substr(j, 3);
        } else {
            s2 = s2 + s.substr(j, 2);
        }
        return s2;
    }
};
