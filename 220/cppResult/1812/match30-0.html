class Solution {
public:
    string reformatNumber(string number) {
        string s;
        for (char c: number) {
            if (c >= '0' && c <= '9') s += c;
        }
        
        string res;
        int i = 0, n = s.size();
        while (i < n) {
            if (n - i > 4) {
                for (int j = 0; j < 3; ++j) res += s[i++];
                res += '-';
            }
            
            else if (n - i == 4) {
                res += s[i++];
                res += s[i++];
                res += '-';
                res += s[i++];
                res += s[i++];
            }
            
            else if (n - i == 3) {
                for (int j = 0; j < 3; ++j) res += s[i++];
            }
            
            else {
                for (int j = 0; j < 2; ++j) res += s[i++];
            }
        }
        
        return res;
    }
};
