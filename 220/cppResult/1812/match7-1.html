class Solution {
public:
    string reformatNumber(string number) {
        string str;
        for (char c : number) {
            if (c == '-' || c == ' ') {
                continue;
            }
            str.push_back(c);

        }
        
        string result;
        int idx = 0;
        while (idx + 4 < str.size()) {
            result.push_back(str[idx]);
            result.push_back(str[idx + 1]);
            result.push_back(str[idx + 2]);
            result.push_back('-');
            idx += 3;
        }
        
        if (idx + 4 == str.size()) {
            result.push_back(str[idx]);
            result.push_back(str[idx + 1]);
            result.push_back('-');
            result.push_back(str[idx + 2]);
            result.push_back(str[idx + 3]);
        } else if (idx + 3 == str.size()) {
            result.push_back(str[idx]);
            result.push_back(str[idx + 1]);
            result.push_back(str[idx + 2]);
        } else {
            result.push_back(str[idx]);
            result.push_back(str[idx + 1]);
        }
        return result;
    }
};
