class Solution {
public:
    string reformatNumber(string number) {
		string tmp;
		string res;
		for (int i = 0; i < number.size(); ++i) {
			if (number[i] != '-'&&number[i] != ' ') {
				tmp += number[i];
			}
		}
		if (tmp.size() < 4) {
			return tmp;
		}
		int pos = 0;
		while (pos + 4 < tmp.size()) {
			res += tmp.substr(pos, 3) + "-";
			pos += 3;
		}
		if (pos + 4 == tmp.size()) {
			res += tmp.substr(pos, 2) + "-" + tmp.substr(pos + 2, 2);
		}
		else  {
			res += tmp.substr(pos);
		}
		return res;
	}
};
