class Solution {
public:
  int maximumUniqueSubarray(vector<int>& a) {
    vector<bool> used(10010, false);
    
    int n = a.size(), res = 0, sum = 0;
    
    for (int i = 0, j = -1; i < n; ++i) {
      while (j+1 < n && !used[a[j+1]]) {
        used[a[j+1]] = true;
        sum += a[j+1];
        ++j;
      }
      
      //cout << i << " " << j << " " << sum << endl;
      res = max( res, sum );
      
      used[a[i]] = false;
      sum -= a[i];
    }
    
    return res;
  }
};
