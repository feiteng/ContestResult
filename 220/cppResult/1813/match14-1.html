class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int res=0;
        unordered_set<int> st;
        int left=0;
        int right=0;
        int n=nums.size();
        int sum=0;
        while(right<n){
            while(st.find(nums[right])!=st.end()){
                st.erase(nums[left]);
                sum-=nums[left];
                left++;
            }
            sum+=nums[right];
            st.insert(nums[right]);
            res=max(res,sum);
            right++;
        }
        return res;
    }
};
