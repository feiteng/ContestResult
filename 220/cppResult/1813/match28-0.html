class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int ans=0;
        int l=nums.size();
        int cur=0;
        unordered_set<int> used;
        int j=0;
        for (int i=0;i<l;i++){
            while (j<l and used.count(nums[j])==0){
                used.insert(nums[j]);
                cur+=nums[j];
                j++;
            }
            ans=max(ans,cur);

            used.erase(nums[i]);
            cur-=nums[i];
        }
        return ans;
    }
};
