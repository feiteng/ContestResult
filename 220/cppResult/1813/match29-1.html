class Solution {
public:
    int a[100005], sum[100005], pos[10005], pre[100005];
    int maximumUniqueSubarray(vector<int>& nums) {
        int n = nums.size();
        for(int i = 1; i <= n; i++) a[i] = nums[i-1];
        for(int i = 1; i <= n; i++) sum[i] = sum[i-1]+a[i];
        pre[0] = 1;

        int ans = 0;
        for(int i = 1; i <= n; i++){
            pre[i] = max(pre[i-1],pos[a[i]]+1);
            pos[a[i]] = i;
            ans = max(ans,sum[i]-sum[pre[i]-1]);
        }
        return ans;
    }
};
