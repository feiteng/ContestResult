class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        unordered_set<int> s;
        int sum = 0;
        int tail = 0;
        int ret = 0;
        for(int i = 0;i < nums.size();i++){
            while(tail < i && s.find(nums[i]) != s.end()){
                sum -= nums[tail];
                s.erase(nums[tail++]);
            }
            s.insert(nums[i]);
            sum += nums[i];
            ret = max(ret, sum);
        }
        return ret;
    }
};
