class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int n = 0;
        int start = 0;
        unordered_map<int, int> dict;
        int res = 0;

        int sum = 0;
        for (int i = 0; i < nums.size(); ++i)
        {
            sum += nums[i];
            dict[nums[i]]++;
            while (dict[nums[i]] > 1)
            {
                dict[nums[start]]--;
                sum -= nums[start];
                if (dict[nums[start]] == 0) dict.erase(nums[start]);
                ++start;
            }            
            res = max(res, sum);
        }
        
        return res;
    }
};
