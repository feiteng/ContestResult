class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int ans = 0;
        int sum = 0;
        int left = 0, right = 0;
        int n = nums.size();
        
        unordered_map<int, int> mp;
        
        while(right < n) {
            int curr = nums[right];
            mp[curr]++;
            sum += curr;
            while(mp[curr] > 1) {
                int prev = nums[left];
                mp[prev]--;
                sum -= prev;
                left++;
            }
            ans = max(ans, sum);
            right++;
        }
        
        return ans;
    }
};
