class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int ans = 0;
        int p1 = 0, p2 = 0;
        int curr = 0;

        unordered_set<int> visited;
        while(p2 < nums.size()){
            while(p1 < nums.size() && visited.count(nums[p2])){
                visited.erase(nums[p1]);
                curr -= nums[p1];
                p1++;
            }
            curr += nums[p2];
            visited.insert(nums[p2]);
            ans = max(ans, curr);
            p2++;
        }
        return ans;
    }
};
