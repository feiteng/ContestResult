class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        unordered_map<int, int> count;
        int ans = 0;
        int sum = 0;
        int idx = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (count.find(nums[i]) != count.end()) {
                while (count[nums[i]] > 0) {
                    sum -= nums[idx];
                    count[nums[idx]]--;
                    idx++;
                }
            }
            count[nums[i]] = 1;
            sum += nums[i];
            ans = max(ans, sum);
        }
        return ans;
    }
};
