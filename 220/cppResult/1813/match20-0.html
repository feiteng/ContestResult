class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int output = 0;
        unordered_map<int, int> mp;
        for(int i = 0, j = 0, sm = 0; j < nums.size(); j++){
            mp[nums[j]]++;
            sm += nums[j];
            while(mp[nums[j]] > 1){
                mp[nums[i]]--;
                sm -= nums[i];
                i++;
            }
            output = max(output, sm);
        }
        return output;
    }
};
