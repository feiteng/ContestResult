class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int n = nums.size(), cur = 0, ret = 0;
        vector<bool> has(10005);
        for (int i=0, j=0; i<n; i++) {
            while (j < n && !has[nums[j]]) {
                cur += nums[j];
                has[nums[j]] = true;
                j++;
            }
            ret = max(ret, cur);
            cur -= nums[i];
            has[nums[i]] = false;
        }
        return ret;
    }
};
