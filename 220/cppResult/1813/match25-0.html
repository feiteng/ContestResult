class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int n = nums.size();
        int i = 0, j = 1;
        int out = nums[0], tmp = out;
        unordered_set<int> s;
        s.insert(nums[0]);
        while(i<n && j<n){
            while(s.find(nums[j]) != s.end()){
                s.erase(nums[i]);
                tmp -= nums[i++];
            }
            s.insert(nums[j]);
            tmp += nums[j++];
            out = max(tmp, out);
        }
        return out;
    }
};
