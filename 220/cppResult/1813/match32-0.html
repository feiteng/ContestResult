#include <bits/stdc++.h>

using namespace std;
struct node{
    int val,x;
};
class Solution {
public:
    int cnt[10007];

    int maximumUniqueSubarray(vector<int> &nums) {
        memset(cnt,0,sizeof cnt);
        int j = 0;
        int temp = 0;
        int ans = 0;
        for (int i = 0; i < nums.size(); i++) {
            temp += nums[i];
            cnt[nums[i]]++;
            while (cnt[nums[i]] == 2) {
                temp -= nums[j];
                cnt[nums[j]]--;
                j++;
            }
            ans = max(ans, temp);
        }
        return ans;

    }
};
