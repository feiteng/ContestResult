class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int left = 0;
        int right = 0;
        int len = nums.size();

        int cur = 0;
        int res = 0;
        unordered_map<int, int> rec;
        while(right<len) {
            int i = nums[right];
            cur += i;
            rec[i]++;
            while(rec[i]>1) {
                int j = nums[left];
                cur -= j;
                rec[j]--;
                left++;
            }
            res = max(res, cur);
            right ++;
        }
        return res;
    }
};
