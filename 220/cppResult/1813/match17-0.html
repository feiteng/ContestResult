class Solution {
public:
  int maximumUniqueSubarray(vector<int>& nums) {
    unordered_set<int> s;
    int j = 0;
    int sum = 0;
    int rv = 0;
    for (int i = 0; i < nums.size(); i++) {
      while (s.count(nums[i])) {
        sum -= nums[j];
        s.erase(nums[j++]);
      }
      s.insert(nums[i]);
      sum += nums[i];
      rv = max(rv, sum);
    }
    return rv;;
  }
};
