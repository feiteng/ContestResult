class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        unordered_set<int> s;
        int res = 0, cur = 0;
        for (int j = 0, i = 0; i < nums.size(); i++) {
            while (s.count(nums[i])) {
                cur -= nums[j];
                s.erase(nums[j++]);
            }
            
            s.insert(nums[i]);
            cur += nums[i];
            res = max(res, cur);
        }
        return res;
    }
};
