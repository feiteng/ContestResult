class Solution {
public:
    int maximumUniqueSubarray(vector<int>& N) {
        int s = 0;
        unordered_map<int, int> M;
        int j = 0;
        int bs = 0;
        for (int i = 0 ; i < N.size(); i++) {
            s += N[i];
            M[N[i]]++;
            while (M[N[i]] > 1) {
                M[N[j]]--;
                s -= N[j];
                j++;
            }
            bs = max(bs, s);
            
        }
        return bs;
    }
};
