class Solution {
public:
  int maximumUniqueSubarray(vector<int>& nums) {
    vector<int> cnt(12341);
    int sum = 0;
    int best = 0;
    for (int l = 0, r = 0; l < nums.size(); l++) {
      while (r < nums.size() && cnt[nums[r]] == 0) {
        sum += nums[r];
        cnt[nums[r++]]++;
      }
      best = max(best, sum);
      sum -= nums[l];
      cnt[nums[l]]--;
    }
    return best;
  }
};
