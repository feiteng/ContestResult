class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        // Indicates whether the number is in the subarray
        vector<bool> bitmask(10001);
        int start_i = 0;
        int max_score = 0;
        int curr_score = 0;
        for (int i = 0; i < nums.size(); ++i) {
            while (bitmask[nums[i]]) {
                bitmask[nums[start_i]] = false;
                curr_score -= nums[start_i];
                ++start_i;
            }
            curr_score += nums[i];
            bitmask[nums[i]] = true;
            max_score = max(curr_score, max_score);
        }
        return max_score;
    }
};
