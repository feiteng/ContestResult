class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        map<int, int> ma;
        int cursum = 0;
        int ret = 0;
        int j = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (ma[nums[i]] != 0) {
                while (j != i && ma[nums[i]] == 1) {
                    cursum -= nums[j];
                    ma[nums[j]] = 0;
                    j++;
                }
            }
            cursum += nums[i];
            ma[nums[i]] = 1;
            ret = max(ret, cursum);
            
        }
        return ret;
    }
};
