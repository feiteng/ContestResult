class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int vis[10005];
        fill(vis,vis+10005,0);
        int l=0;
        int sum=0;
        int ans=0;
        for(int i = 0;i<nums.size();i++){
            vis[nums[i]]++;
            sum+=nums[i];
            while(vis[nums[i]]==2){
                sum-=nums[l];
                vis[nums[l]]--;
                l++;
            }
            ans=max(ans,sum);
        }
        return ans;
    }
};
