class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int n = nums.size();
        int sum = 0;
        int ret = 0;
        int r = 0;
        int fr[10001] = {0};

        for(int l = 0;l < n;l ++){
            while(r < n and fr[nums[r]] == 0){
                fr[nums[r]] = 1;
                sum += nums[r];
                r ++;
            }
            ret = max(ret, sum);
            sum -= nums[l];
            fr[nums[l]] = 0;
        }
        return ret;
    }
};
