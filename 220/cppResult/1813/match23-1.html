class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        vector<int> vis(10005, 0);
        int n = nums.size();
        int pre = 0;
        int tt = 0, res = 0;

        for (int i = 0; i < n; ++i) {
            vis[nums[i]]++;
            tt += nums[i];
            while (pre < i && vis[nums[i]] > 1) {
                vis[nums[pre]]--;
                tt -= nums[pre];
                pre ++;
            }
            res = max(res, tt);
        }
        return res;
    }
};
