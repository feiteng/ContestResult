class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int sum = 0;
        unordered_set<int> window;
        int p1 = 0;
        int p2 = 0;
        int n = nums.size();
        int ans = 0;
        while (p1 < n && p2 < n) {
            while (window.count(nums[p2])) {
                sum -= nums[p1];
                window.erase(nums[p1]);
                ++p1;
            }
            
            window.emplace(nums[p2]);
            sum += nums[p2];
            ans = max(ans, sum);
            ++p2;
        }
        return ans;
    }
};
