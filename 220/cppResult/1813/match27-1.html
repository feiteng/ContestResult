class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int s = 0, res = 0, i = 0, j = 0, n = nums.size(), m = *max_element(nums.begin(), nums.end());

        vector<int> freq(m+1,0);
        while(i<n) {
            s+=nums[i];
            freq[nums[i]]++;
            while(j<n && freq[nums[i]]>1) {
                freq[nums[j]]--;
                s-=nums[j];
                j++;
            }
            res = max(res, s);
            i++;
        }
        return res;
    }
};
