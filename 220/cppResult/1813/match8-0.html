class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        unordered_set<int> tb;
        int left = 0;
        int sum = 0;
        int res = 0;
        
        for(int i = 0; i < nums.size(); ++i) {
            while(tb.count(nums[i]) > 0) {
                tb.erase(nums[left]);
                
                sum -= nums[left];
                left++;
            }
            
            tb.insert(nums[i]);
            sum += nums[i];
            res = max(res, sum);
        }
        
        return res;
    }
};
