class Solution {
public:
    int maximumUniqueSubarray(vector<int>& a) {
        bool vis[10005];
        memset(vis, 0, sizeof(vis));
        int n = a.size();
        int ans = 0;
        int cur = 0;
        int l = 0;

        for (int i = 0; i < n; ++i) {
            while (vis[a[i]]) {
                cur -= a[l];
                vis[a[l]] = 0;
                ++l;
            }
            cur += a[i];
            vis[a[i]] = 1;
            ans = max(ans, cur);
        }
        return ans;
    }
};
