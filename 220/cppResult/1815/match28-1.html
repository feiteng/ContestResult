struct Tuple{
    int s, e, w;
    int idx;
    Tuple(){
        s = e = w = idx = 0;
    }
    
    bool operator < (const Tuple &o) const{
        return w < o.w;
    }
};


class Solution {
public:
    int fa[100005];
    
    int find(int x){
        return x == fa[x] ? x : fa[x] = find(fa[x]);    
    }
    

    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        int e_size = edgeList.size();
        int q_size = queries.size();
        vector<bool> ans(q_size, false);
        for(int i=0; i<n; i++) fa[i] = i;
        vector<Tuple> new_e, new_q;
        for(auto &x: edgeList){
            Tuple o;
            o.s = x[0];

            o.e = x[1];
            o.w = x[2];
            o.idx = 0;
            new_e.push_back(o);
        }
        sort(new_e.begin(), new_e.end());
        for(int i=0; i<q_size; i++){
            Tuple o;
            o.s = queries[i][0];
            o.e = queries[i][1];
            o.w = queries[i][2];

            o.idx = i;
            new_q.push_back(o);
        }        
        sort(new_q.begin(), new_q.end());
        int i=0;
        for(int j=0; j<q_size;j++){
            while(i < e_size && new_e[i].w < new_q[j].w){
                int a = find(new_e[i].s);
                int b = find(new_e[i].e);
                if(a != b) fa[a] = b;
                i ++;
            }
            int a = find(new_q[j].s);
            int b = find(new_q[j].e);
            ans[new_q[j].idx] = (a == b);
        }
        return ans;
    }
};
