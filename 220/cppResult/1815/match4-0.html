class Solution {
public:
    vector<int> f;
    int find(int x){
        if(f[x]==-1)
            return x;
        return f[x]=find(f[x]);
    }
    void com(int x, int y){
        int fx=find(x);
        int fy=find(y);
        if(fx!=fy)
            f[fx]=fy;
    }
    static bool cmp(vector<int>& a,vector<int>& b){
        return a[2]<b[2];
    }
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& e, vector<vector<int>>& queries) {
        int m=queries.size();
        int k=e.size();
        f.insert(f.end(),n,-1);
        vector<bool> res(m);
        sort(e.begin(),e.end(),cmp);
        vector<vector<int>> qq;
        for(int i=0;i<m;++i)
            qq.push_back({queries[i][2],queries[i][0],queries[i][1],i});
        sort(qq.begin(),qq.end());
        
        int i=0;
        for(int j=0;j<m;++j){
            while(i<k&&e[i][2]<qq[j][0]){
                com(e[i][0],e[i][1]);
                ++i;
            }
            res[qq[j][3]]=find(qq[j][1])==find(qq[j][2]);
        }
        
        return res;
    }
};
