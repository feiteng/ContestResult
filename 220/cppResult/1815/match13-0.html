class Solution {
public:
    struct node
    {
        int x,y,z;
        int id;
        bool ans;
    }a[100005],b[100005];
    int fa[100005];
    int get(int x)
    {
        return fa[x]==x?x:fa[x]=get(fa[x]);
    }
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        int nn=edgeList.size();
        int m=queries.size();
        for(int i=1;i<=nn;i++)
        {
            a[i].x=edgeList[i-1][0]+1;
            a[i].y=edgeList[i-1][1]+1;
            a[i].z=edgeList[i-1][2];
        }
        for(int i=1;i<=m;i++)
        {
            b[i].x=queries[i-1][0]+1;
            b[i].y=queries[i-1][1]+1;
            b[i].z=queries[i-1][2];
            b[i].id=i;
        }
        sort(a+1,a+nn+1,cmp);
        sort(b+1,b+m+1,cmp);
        for(int i=1;i<=n;i++)
            fa[i]=i;
        int N=1;
        for(int i=1;i<=m;i++)
        {
            while(a[N].z<b[i].z&&N<=nn)
            {
                int fx=get(a[N].x);
                int fy=get(a[N].y);
                if(fx!=fy)
                {
                    fa[fx]=fy;
                }
                N++;
            }
            int fx=get(b[i].x);
            int fy=get(b[i].y);
            if(fx==fy)
            {
                b[i].ans=true;
            }
            else
                b[i].ans=false;
        }
        sort(b+1,b+m+1,cmp1);
        vector<bool>ret;
        for(int i=1;i<=m;i++)
            ret.push_back(b[i].ans);
        return ret;
    }
private:
    static bool cmp(node &aa,node &bb)
    {
        return aa.z<bb.z;
    }
    static bool cmp1(node &aa,node &bb)
    {
        return aa.id<bb.id;
    }
};
