struct note{
    int x,y,z,id;
};
class Solution {
    int fa[100005];
    note a[100005],b[100005];
    int m,q;
public:
    int getfa(int x){
        return fa[x] == x ? x : fa[x] = getfa(fa[x]);
    }
    static bool cmp(note i,note j){
        return i.z<j.z;
    }
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edge, vector<vector<int>>& que) {
        vector<bool> ret;
        m = edge.size();
        for(int i=0;i<m;i++) {
            a[i].x=edge[i][0];
            a[i].y=edge[i][1];
            a[i].z=edge[i][2];
        }
        q = que.size();

        for(int i=0;i<q;i++) {
            ret.push_back(false);
            b[i].x=que[i][0];
            b[i].y=que[i][1];
            b[i].z=que[i][2];
            b[i].id = i;
        }
        sort(a,a+m,cmp);
        sort(b,b+q,cmp);
        for(int i=1;i<=n;i++) fa[i]=i;
        int l=0;
        for(int i=0;i<q;i++) {
            while (l<m && a[l].z < b[i].z) {
                int fx=getfa(a[l].x);
                int fy=getfa(a[l].y);
                if (fx!=fy) {
                    fa[fx]=fy;
                }
                l++;
            }
            int fx=getfa(b[i].x);
            int fy=getfa(b[i].y);
            ret[b[i].id] = (fx==fy);
        }
        return ret;
    }
};
