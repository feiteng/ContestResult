    const static int mxn = 100005;
    
    int f[mxn];
    
    int find(int u) {
        return f[u] == u ? u : f[u] = find(f[u]);
    }
    
    void merge(int a, int b) {
        // cout << "m " << a << " " << b << endl;
        a = find(a);
        b = find(b);
        f[a] = b;
    }
    
    bool query(int a, int b) {
        a = find(a);
        b = find(b);
        return a == b;
    }
    
    bool cmp1(const vector<int> & a, const vector<int> & b) {
        return a[2] < b[2];
    }

class Solution {
private:
    
public:
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        int q = queries.size();
        for (int i = 0; i < q; ++i) {
            queries[i].push_back(i);

        }
        sort(begin(queries), end(queries), cmp1);
        sort(begin(edgeList), end(edgeList), cmp1);
        for (int i = 0; i <= n; ++i)
            f[i] = i;
        int ptr = 0;
        vector<bool> ans(q);
        for (int i = 0; i < q; ++i) {
            // cout << edgeList[ptr][2] << " " <<  queries[i][2] << endl;
            while (ptr < edgeList.size() && edgeList[ptr][2] < queries[i][2]) {
                merge(edgeList[ptr][0], edgeList[ptr][1]);
                ptr++;
            }
            ans[queries[i][3]] = query(queries[i][0], queries[i][1]);
        }
        return ans;
    }
};
