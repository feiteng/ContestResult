class Solution {
public:
    
    vector<int> par;
    int find(int x) {
        return par[x] == x? x: par[x] = find(par[x]);
    }
    
    void unite(int x, int y) {
        x = find(x), y = find(y);
        par[y] = x;
    }
    
    vector<bool> distanceLimitedPathsExist(int nn, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        vector<array<int, 4>> q;
        vector<array<int, 3>> e;
        for (vector<int> &v : edgeList)
            e.push_back({v[2], v[0], v[1]});
        int n = queries.size(), m = e.size();
        for (int i = 0; i < n; i++) {
            vector<int> &v = queries[i];
            q.push_back({v[2], v[0], v[1], i});
        }
        par.resize(nn);
        iota(par.begin(), par.end(), 0);
        vector<bool> ans(n);
        sort(q.begin(), q.end());
        sort(e.begin(), e.end());
        int current = 0;
        for (int i = 0; i < n; i++) {
            while (current < m && e[current][0] < q[i][0]) {
                unite(e[current][1], e[current][2]);
                current++;
            }
            ans[q[i][3]] = (find(q[i][1]) == find(q[i][2]));
        }
        return ans;
    }
};
