typedef pair<int, int> pii;

class Solution {
public:
    
    int find(int a) {
	//m[a] = -size
	// if(m[a] < 0) {
		// return a;
	// }
	
	//m[a] = a
        if(m[a] == a) {
            return a;
        }
        return m[a] = find(m[a]);
    }

    void un(int a, int b) {
        a = find(a);
        b = find(b);
        if(a != b) {

            //m[a] = -size:
            // m[a] += m[b];
            // m[b] = a;

            //m[a] = a:
            m[a] = b;
        }
    }

    //initialized with max value that will be used in before use
    vector<int> m;
    vector<vector<int>> g;
    
    static bool cmp(vector<int> &a, vector<int> &b) {
        return a[2] < b[2];
    }
    
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& e, vector<vector<int>>& q) {
        m.resize(n + 5);
        g.resize(n + 5);
        for(int i = 0; i < q.size(); i++) {
            q[i].push_back(i);
        }
        sort(q.begin(), q.end(), cmp);
        sort(e.begin(), e.end(), cmp);
        for(int i = 0; i < n; i++) {
            m[i] = i;
        }
        int p = 0;
        vector<bool> res(q.size());
        for(auto &i : q) {
            int l = i[2];
            while(p < e.size() && e[p][2] < l) {
                un(e[p][0], e[p][1]);
                p++;
                
            }
            res[i[3]] = (find(i[0]) == find(i[1]));
        }
        return res;
    }
};
