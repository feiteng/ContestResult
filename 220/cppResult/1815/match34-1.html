int per[200000];
int Find(int x){
	if(per[x]==x)return x;

	else return per[x]=Find(per[x]);
}
void Join(int x,int y){
	int p1=Find(x);
	int p2=Find(y);
	if(p1==p2)return ;
	else {
		per[p2]=p1;
	}
}
void init(int n){
	for(int i=0;i<=n;i++){
		per[i]=i;
	}
}

class Solution {
public:

    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        for (int i = 0; i < queries.size(); i++) {
            queries[i].push_back(i);
        }
        sort(queries.begin(), queries.end(), [](vector<int>& a, vector<int>& b) {
            return a[2] < b[2];
        });
        sort(edgeList.begin(), edgeList.end(), [](vector<int>& a, vector<int>& b) {
            return a[2] < b[2];
        });
        init(n + 10);
        int now = 0;
        map<int, bool> mp;
        for (int i = 0; i < queries.size(); i++) {
            int mxx = queries[i][2];
            while (now < edgeList.size() && edgeList[now][2] < mxx) {
                int a = edgeList[now][0];
                int b = edgeList[now][1];
                Join(a, b);
                now++;
            }
            int fa = Find(queries[i][0]);
            int fb = Find(queries[i][1]);
            if (fa == fb) {
                mp[queries[i][3]] = true;
            } else {
                mp[queries[i][3]] = false;
            }
        }
        vector<bool> ret;
        for (int i = 0; i < queries.size(); i++) {
            ret.push_back(mp[i]);
        }
        return ret;
    }
};
