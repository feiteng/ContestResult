class Solution {
public:
    int find(int a[], int x) {
        return (a[x] < 0)? x : (a[x] = find(a, a[x]));
    }
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& e, vector<vector<int>>& Q) {
        int m = e.size(), q = Q.size();
        int qidxs[q];
        for(int i = 0; i < q; ++i) {
            qidxs[i] = i;
        }
        sort(qidxs, qidxs + q, [&](int q1, int q2){
            return Q[q1][2] < Q[q2][2];
        });
        int eidxs[m];
        for(int i = 0; i < m; ++i) {
            eidxs[i] = i;
        }
        sort(eidxs, eidxs + m, [&](int e1, int e2) {
            return e[e1][2] < e[e2][2];
        });
        vector<bool> res(q);
        int uf[n];
        memset(uf, -1, sizeof(uf));
        for(int i = 0, p = 0; i < q; ++i) {
            while(p < m && e[eidxs[p]][2] < Q[qidxs[i]][2]) {
                int f1 = find(uf, e[eidxs[p]][0]), f2 = find(uf, e[eidxs[p]][1]);
                if(f1 != f2) {
                    uf[f2] = f1;
                }
                ++p;
            }
            if(find(uf, Q[qidxs[i]][0]) == find(uf, Q[qidxs[i]][1])) {
                res[qidxs[i]] = true;
            } else {
                res[qidxs[i]] = false;
            }
        }
        return res;
    }
};
