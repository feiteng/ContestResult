class Solution {
public:
    int f[100003]; 

    int find(int x) {
        if(f[x] == x) return x;
        f[x] = find(f[x]); 
        return f[x];
    }

    void uni(int x, int y) {
        f[find(x)] = find(y);
    }
   
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        int ne = edgeList.size(), nq = queries.size();
        
        sort(edgeList.begin(), edgeList.end(), [](const vector<int>& lhs, const vector<int>& rhs) {
            return lhs[2] < rhs[2];
        });
        vector<vector<int>> que(0); 
        for(int i = 0; i < nq; i ++) { // [3] = orig index
            vector<int> ele = queries[i];
            ele.push_back(i);
            que.push_back(ele);
        }
        sort(que.begin(), que.end(), [](const vector<int>& lhs, const vector<int>& rhs) {
            return lhs[2] < rhs[2];
        });
        
        for(int i = 0; i < n; i ++) {
            f[i] = i;
        }
        
        vector<bool> ans(nq);
        int ie = 0; 
        for(int i = 0; i < nq; i ++) {
            while(ie < ne && edgeList[ie][2] < que[i][2]) {
                uni(edgeList[ie][0], edgeList[ie][1]); 
                ie ++;
            }
            ans[que[i][3]] = (find(que[i][0]) == find(que[i][1])); 
        }
        
        return ans;
    }
};
