class Solution {
public:
    int fa[100010];
    int tfind(int x){
        if (fa[x]==x) return x;
        return fa[x]=tfind(fa[x]);
    }
    void tunion(int x,int y){
        int u=tfind(x);
        int v=tfind(y);
        if (u==v) return ;
        fa[u]=v;
    }
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        for (int i=0;i<n;i++) fa[i]=i;
        for (int i=0;i<queries.size();i++){
            queries[i].push_back(i);
        }
        sort(edgeList.begin(),edgeList.end(),[](auto &a,auto &b){
            return a[2]<b[2];
        });        

        sort(queries.begin(),queries.end(),[](auto &a,auto &b){
            return a[2]<b[2];
        });
        int now=0;
        vector<bool> res(queries.size());
        for (int i=0;i<queries.size();i++){
            while (now<edgeList.size()&&edgeList[now][2]<queries[i][2]){
                tunion(edgeList[now][0],edgeList[now][1]);
                now++;
            }
            if (tfind(queries[i][0])==tfind(queries[i][1])){
                res[queries[i][3]]=true;
            }
            else res[queries[i][3]]=false;
        }
        return res;
    }
};
