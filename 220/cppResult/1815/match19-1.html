class Solution {
public:
    vector<int> fa;
    int find(int x){
        return fa[x]==-1? x: fa[x]=find(fa[x]);
    }
    
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& e, vector<vector<int>>& q) {
        fa = vector<int>(n+10, -1);

        sort(e.begin(), e.end(), [](vector<int>&a, vector<int>&b){
            return a[2]>b[2];
        });
        vector<int> qq(q.size());
        for(int i=0; i<q.size(); i++){
            qq[i] = i;
        }
        sort(qq.begin(), qq.end(), [&](int x, int y){
            return q[x][2]<q[y][2];
        });
        vector<bool> ans(q.size());
        for(int i=0; i<qq.size(); i++){
            while(e.size()&&e.back()[2]<q[qq[i]][2]){
                int fa1 = find(e.back()[0]);
                int fa2 = find(e.back()[1]);
                e.pop_back();
                if(fa1==fa2) continue;
                fa[fa2] = fa1;
               
            }
            ans[qq[i]] = (find(q[qq[i]][0])==find(q[qq[i]][1]));
            // cout<<find(q[qq[i]][0])<<" "<<find(q[qq[i]][1])<<endl;
        }
        return ans;
    }
};
