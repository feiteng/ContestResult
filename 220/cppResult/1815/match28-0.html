class Solution {
struct yqh{
    int l,r;
    int nums;
}a[100020];
struct query{
    int l,r;
    int nums;
    int id;
}b[100020];
int fa[100020];
static bool cmp1(yqh x, yqh y){
    return x.nums < y.nums;
}
static bool cmp2(query x, query y){
    return x.nums < y.nums;
}
int getfa(int x){
    if(fa[x] == x)return x;
    fa[x] = getfa(fa[x]);
    return fa[x];
}
public:
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        vector<bool> ans;
        int m1 = edgeList.size();
        int m2 = queries.size();
        for(int i=0;i<m1;++i){
            a[i].l = edgeList[i][0];
            a[i].r = edgeList[i][1];
            a[i].nums = edgeList[i][2];
        }
        for(int i=0;i<m2;++i){
            b[i].l = queries[i][0];
            b[i].r = queries[i][1];
            b[i].nums = queries[i][2];
            b[i].id = i;
            ans.push_back(true);
        }
        sort(a,a+m1,cmp1);
        sort(b,b+m2,cmp2);
        for(int i=0;i<n;++i){
            fa[i] = i;
        }
        int now = 0;
        for(int i=0;i<m2;++i){
            while(now < m1 && a[now].nums < b[i].nums){
                int x = a[now].l;
                int y = a[now].r;
                //printf("%d %d %d %d\n", x, y, a[now].nums, b[i].nums);
                x = getfa(x);
                y = getfa(y);
                fa[x] = y;
                ++now;
            }
            
            int x = b[i].l;
            int y = b[i].r;
            x = getfa(x);
            y = getfa(y);
            if(x != y)ans[b[i].id] = false;
        }
        return ans;
    }
};
