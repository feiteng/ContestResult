const int N=100005;


bool cmp(vector<int> &a,vector<int> &b) {
    return a[0]<b[0] || (a[0]==b[0] && a[1]<b[1]);
}


class Solution {
public:
    

    int p[N];
    
    int fnd(int u) {
        if(u==p[u]) return u;
        return p[u]=fnd(p[u]);
    }
    
    
    void un(int a,int b) {
        int u=fnd(a);
        int v=fnd(b);
        if(rand()&1) swap(u,v);
        if(u!=v) p[u]=v;
    }
    
    

    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& ee, vector<vector<int>>& qq) {
        for(int i=0;i<=n;i++) {
            p[i]=i;
        }
        
        vector<bool> ans(qq.size(),false);
        vector<vector<int> > e;
        for(int i=0;i<ee.size();i++) {
            e.push_back({ee[i][2],1,ee[i][0],ee[i][1]});
        }
        for(int i=0;i<qq.size();i++) {
            e.push_back({qq[i][2],0,qq[i][0],qq[i][1],i});
        }
        
        sort(e.begin(),e.end(),&cmp);
        
        for(int i=0;i<e.size();i++) {
            int a=e[i][2];
            int b=e[i][3];
            if(e[i][1]==1) {
                un(a,b);
            } else {
                if(fnd(a) == fnd(b)) {
                    ans[e[i][4]]=true;
                }
            }
        }
        
        return ans;
    }
};
