class Solution {
    public String reformatNumber(String number) {
        LinkedList<Character> list = new LinkedList<>();
        for (char c : number.toCharArray()) {
            if (c != ' ' && c != '-') {
                list.add(c);
            }
        }
        
        StringBuilder res = new StringBuilder();
        while (list.size() > 0) {
            if (list.size() > 4) {
                for (int i = 0; i < 3; i ++) {
                    res.append(list.pollFirst() + "");
                }
                res.append("-");
                continue;
            }
            
            if (list.size() == 4) {
                for (int i = 0; i < 2; i ++) {
                    res.append(list.pollFirst() + "");
                }
                res.append("-");
                for (int i = 0; i < 2; i ++) {
                    res.append(list.pollFirst() + "");
                }
                continue;
            }
            
            while (list.size() > 0) {
                res.append(list.pollFirst() + "");
            }
        }
        
        return res.toString();
    }
}
