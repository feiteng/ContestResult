class Solution {
    public String reformatNumber(String number) {
        Queue<Character> q = new LinkedList<>();
        for (char c : number.toCharArray()) {
            if (c-'0' >= 0 && c - '0' <= 9) {
                q.offer(c);
            }
        }
        StringBuilder sb = new StringBuilder();
        
while (q.size() > 4) {
            sb.append(q.poll());
            sb.append(q.poll());
            sb.append(q.poll());
            sb.append("-");
        }
        
        if (q.size() < 4) {
            while (!q.isEmpty()) {
                sb.append(q.poll());
            }
        } else {
            sb.append(q.poll());
            sb.append(q.poll());
            
            sb.append("-");
            sb.append(q.poll());
            sb.append(q.poll());
        }
        
        return sb.toString();
    }
}
