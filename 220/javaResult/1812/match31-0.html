class Solution {
    public String reformatNumber(String number) {
         number = number.replaceAll(" ","").replaceAll("-","");
            StringBuilder sb = new StringBuilder();
           int le = number.length();
            if(le<=3){
                return number;
            }
            if(le==4){
               sb.append(number.charAt(le-4));
                        sb.append(number.charAt(le-3));
                        sb.append("-");
                        sb.append(number.charAt(le-2));
                        sb.append(number.charAt(le-1));
                        return sb.toString(); 
            }
            
            for(int i=0;i<le;i++){
                sb.append(number.charAt(i));
                if((i+1)%3==0){
                    sb.append("-");
                    if(i==le-5){
                        sb.append(number.charAt(le-4));
                        sb.append(number.charAt(le-3));
                        sb.append("-");
                        sb.append(number.charAt(le-2));
                        sb.append(number.charAt(le-1));
                        return sb.toString();
                    }
                    else if(i==le-4){
                        sb.append(number.charAt(le-3));
                        sb.append(number.charAt(le-2));
                        sb.append(number.charAt(le-1));
                        return sb.toString();
                    }
                    else if(i==le-3){
                        sb.append(number.charAt(le-2));
                        sb.append(number.charAt(le-1));
                        return sb.toString();
                    }
                }
            }
            return sb.toString();
    }
}
