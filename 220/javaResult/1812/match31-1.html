class Solution {
    public String reformatNumber(String number) {
        number = number.replace("-", "").replace(" ", "");
        int n = number.length();
        StringBuilder sb = new StringBuilder();
        
if (n == 4) {
            sb.append(number.charAt(0));
            sb.append(number.charAt(1));
            sb.append('-');
            sb.append(number.charAt(2));
            sb.append(number.charAt(3));
            return sb.toString();
        }
        for (int i = 0; i < n; i++) {
            sb.append(number.charAt(i));
            if (i % 3 == 2 && i != n - 1) {
                sb.append('-');
                if (n - 1 - i == 4) {
                    sb.append(number.charAt(i + 1));
                    sb.append(number.charAt(i + 2));
                    sb.append('-');
                    sb.append(number.charAt(i + 3));
                    sb.append(number.charAt(i + 4));
                    return sb.toString();
                
} 
            } 
        }
        return sb.toString();
    }
}
