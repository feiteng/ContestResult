class Solution {
    public String reformatNumber(String number) {
        number = number.replace(" ", "");
        number = number.replace("-", "");
        String result = "";
        while(number.length() > 4) {
            result = result + number.substring(0, 3) + "-";
            number = number.substring(3);
        }
        
        if(number.length() < 4) {
            result += number;
        }
        else {
            result += number.substring(0, 2) + "-" + number.substring(2);
        }
        
        return result;
    }
}
