class Solution {
    public String reformatNumber(String number) {
        StringBuilder sb = new StringBuilder();
        List<Integer> list = new ArrayList<>();
        
        for (char c : number.toCharArray()) {
            if (Character.isDigit(c)) {
                list.add(c-'0');
            }
        }
        
        int i = 0;
        while (i < list.size() - 4) {
            sb.append(list.get(i++));
            sb.append(list.get(i++));
            sb.append(list.get(i++));
            sb.append('-');
        }
        
        if (list.size() - i <= 3) {
            while (i < list.size()) {
                sb.append(list.get(i++));
            }
        } else {
            sb.append(list.get(i++));
            sb.append(list.get(i++));
            sb.append('-');
            sb.append(list.get(i++));
            sb.append(list.get(i++));
        }
        
        return sb.toString();
    }
}
