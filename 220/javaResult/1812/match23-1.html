class Solution {
    public String reformatNumber(String number) {
        char[] chs = new char[number.length()];
        int len = 0;
        for (int i = 0; i < number.length(); ++i) {
            char ch = number.charAt(i);
            
if (ch <= '9' && ch >= '0') {
                chs[len++] = ch;
            }
        }
        StringBuilder sbu = new StringBuilder();
        for (int i = 0; i < len; ++i) {
            if (len - i > 4 || len - i == 3) {
                sbu.append(chs[i]);
                sbu.append(chs[i + 1]);
                sbu.append(chs[i + 2]);
                if (len - i > 4) {
                    sbu.append('-');
                }
                i += 2;
            } else if (len - i  == 4) {
                sbu.append(chs[i]);
                sbu.append(chs[i + 1]);
                sbu.append('-');
                sbu.append(chs[i + 2]);
                sbu.append(chs[i + 3]);
                i = len;
            } else {
                sbu.append(chs[i]);
                
sbu.append(chs[i + 1]);
                i++;
            }
        }
        return sbu.toString();
    }
}
