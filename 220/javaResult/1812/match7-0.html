class Solution {
    public String reformatNumber(String number) {
        StringBuilder digits = new StringBuilder();
        for(char d : number.toCharArray()){
            if(Character.isDigit(d)){
                digits.append(d);
            }
        }
        StringBuilder ans = new StringBuilder();
        int i = 0;
        while(i+4 < digits.length()){
            ans.append(digits.charAt(i));
            ans.append(digits.charAt(i+1));
            ans.append(digits.charAt(i+2));
            ans.append('-');
            i+=3;
        }
        if(i+2 == digits.length()){
            ans.append(digits.charAt(i));
            ans.append(digits.charAt(i+1));
        }
        else if(i+3 == digits.length()){
            ans.append(digits.charAt(i));
            ans.append(digits.charAt(i+1));
            ans.append(digits.charAt(i+2));
        }
        else{
            ans.append(digits.charAt(i));
            ans.append(digits.charAt(i+1));
            ans.append('-');
            ans.append(digits.charAt(i+2));
            ans.append(digits.charAt(i+3));
        }
        return ans.toString();
    }
}
