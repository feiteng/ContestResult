class Solution {
    public String reformatNumber(String number) {
        List<Character> digs = new ArrayList<>();
        for(char c: number.toCharArray()){
            if(c>='0'&&c<='9'){
                digs.add(c);
            }
        }
        int n = digs.size();
        StringBuilder sb = new StringBuilder();
        int i=0;
        while(i<n-4){
            for(int x =0; x<3;++x){
                sb.append(digs.get(i++));
            }
            sb.append('-');
        }
        if(i==n-4){
            sb.append(digs.get(i++));
            sb.append(digs.get(i++));
            sb.append('-');
            sb.append(digs.get(i++));
            sb.append(digs.get(i++));
        }else{
            while(i<n){
                 sb.append(digs.get(i++));
            }
        }
        return sb.toString();
    }
}
