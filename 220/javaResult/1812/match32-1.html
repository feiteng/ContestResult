class Solution {
    public String reformatNumber(String number) {
        List<Character> l = new ArrayList<Character>();
        
        for (char c : number.toCharArray()) {
            if (c >= '0' && c <= '9') {
                l.add(c);
            }
        }
        
        StringBuilder sb = new StringBuilder();
        
        while (l.size() > 4) {
            for (int i = 0; i < 3; i++) {
                sb.append(l.remove(0));
            }
            
sb.append("-");
        }
        
        if (l.size() == 2) {
            sb.append(l.remove(0));
            sb.append(l.remove(0));
        }
        else if (l.size() == 3) {
            sb.append(l.remove(0));
            sb.append(l.remove(0));
            sb.append(l.remove(0));
        }
        else {
            sb.append(l.remove(0));
            sb.append(l.remove(0));
            sb.append("-");
            sb.append(l.remove(0));
            sb.append(l.remove(0));
        }
        
        return sb.toString();
    }
}
