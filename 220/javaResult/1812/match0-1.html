		public String reformatNumber(String number) {
			number = number.replace("-", "");
			number = number.replace(" ", "");
			String ret = "";
			while(number.length() > 4){
				ret += number.substring(0, 3);

				ret += "-";
				number = number.substring(3);
			}
			if(number.length() <= 3){
				ret += number;
			}else{
				ret += number.substring(0, 2) + "-" + number.substring(2);
			}
			return ret;
		}
	}
