class Solution {
    public String reformatNumber(String number) {
        StringBuilder sb = new StringBuilder();
        StringBuilder ans = new StringBuilder();
        for(char ch: number.toCharArray())
        {
            if(ch >= '0' && ch <= '9')
                sb.append(ch);
        }
        int len = sb.length();
        int i = 0;
        while(len > 4)
        {
            len -= 3;
            ans.append(sb.charAt(i)+"");
            i++;
            ans.append(sb.charAt(i)+"");
            i++;
            ans.append(sb.charAt(i)+"");
            i++;
            ans.append("-");
        }
        if(len == 4)
        {
                ans.append(sb.charAt(i)+"");
                i++;
                ans.append(sb.charAt(i)+"");
                i++;
                ans.append("-");
                ans.append(sb.charAt(i)+"");
                i++;
                ans.append(sb.charAt(i)+"");
                i++;
        }
        else
        {
            while(len > 0)
            {
                len--;
                ans.append(sb.charAt(i)+"");
                i++;
            }
        }
        return ans.toString();
    }
}
