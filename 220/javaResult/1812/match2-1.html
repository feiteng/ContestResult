class Solution {
    public String reformatNumber(String number) {
        List<Character> list = new ArrayList<Character>();
        for(char c: number.toCharArray()){
            if(c != '-' && c != ' '){
                list.add(c);
            }
        }
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while(i < list.size()){
            if(i + 4 < 
list.size()){
                sb.append(list.get(i++));
                sb.append(list.get(i++));
                sb.append(list.get(i++));
                sb.append('-');
                // i += 3;
            }else {
                if(list.size() - i == 4) 
{
                    sb.append(list.get(i++));
                    sb.append(list.get(i++));
                    sb.append('-');
                    sb.append(list.get(i++));
                    sb.append(list.get(i++));
                }else 
{
                    while(i < list.size()){
                        sb.append(list.get(i++));
                    }
                }
            }
        }
        return sb.toString();
        
        
    }
}
