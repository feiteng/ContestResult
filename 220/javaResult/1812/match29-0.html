class Solution {
    public String reformatNumber(String s) {
        StringBuilder res = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (c >= '0' && c <= '9') {
                res.append(c);
            }
        }
        s = res.toString();
        res = new StringBuilder();
        // System.out.println(s);
        for (int i = 0; i < s.length(); i++) {
            if (i < s.length() - 4) {
                for (int j = 0; j < 3; j++) {
                    res.append(s.charAt(i));
                    i++;
                }
                i--;
                res.append("-");
            } else {
                if (i == s.length() - 4) {
                    for (int j = 0; j < 2; j++) {
                        res.append(s.charAt(i));
                        i++;
                    }
                    res.append("-");
                    for (int j = 0; j < 2; j++) {
                        res.append(s.charAt(i));
                        i++;
                    }
                } else {
                   res.append(s.charAt(i));
                }
            }
        }
        return res.toString();
    }
}
