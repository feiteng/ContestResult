class Solution {
    public String reformatNumber(String number) {
        Stack<Character> stack= new Stack<>();
        for(int i=0;i<number.length();i++){
            char c=number.charAt(i);
            
if(c>='0'&&c<='9'){
                stack.push(c);
            }
        }
        
        StringBuilder sb=new StringBuilder();
        if(stack.size()%3==1)
{
            sb.append(stack.pop());
            sb.append(stack.pop());
            sb.append('-');
            sb.append(stack.pop());
            sb.append(stack.pop());
        }else if(stack.size()%3==2){
            sb.append(stack.pop());
            sb.append(stack.pop());
        }
        
        
while(!stack.isEmpty()){
            sb.append('-');
            sb.append(stack.pop());
            sb.append(stack.pop());
            sb.append(stack.pop());
        }
        
        String s= sb.reverse().toString();
        return s.charAt(s.length()-1)=='-'?s.substring(0,s.length()-1):s;
    }
}
