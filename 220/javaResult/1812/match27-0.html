class Solution {
    public String reformatNumber(String number) {
        String s = number.replaceAll(" ", "");
        s = s.replaceAll("-", "");
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); ) {
            if(s.length() - i <= 4){
                if (s.length() - i == 4) {
                    for (int j = 0; j < 2; j++) {
                        sb.append(s.charAt(i));
                        i++;
                    }
                    sb.append("-");
                    for (int j = 0; j < 2; j++) {
                        sb.append(s.charAt(i));
                        i++;
                    }
                } else if (s.length() - i == 2) {
                    for (int j = 0; j < 2; j++) {
                        sb.append(s.charAt(i));
                        i++;
                    }
                } else if (s.length() - i == 3) {
                    for (int j = 0; j < 3; j++) {
                        sb.append(s.charAt(i));
                        i++;
                    }
                }
                break;
            }
            for (int j = 0; j < 3; j++) {
                sb.append(s.charAt(i));
                i++;
            }
            sb.append("-");
        }
        return sb.toString();
    }
}
