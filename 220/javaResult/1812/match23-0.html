class Solution {
    public String reformatNumber(String number) {
        char[] chars = new char[number.length()];
        int len = 0;
        for (char c : number.toCharArray()) {
            if (c != ' ' && c != '-') {
                chars[len++] = c;
            }
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; ) {
            if (i + 4 == len) {
                sb.append(chars[i]);
                sb.append(chars[i + 1]);
                sb.append("-");
                sb.append(chars[i + 2]);
                sb.append(chars[i + 3]);
                i += 4;
            } else if (i + 3 <= len) {
                sb.append(chars[i]);
                sb.append(chars[i + 1]);
                sb.append(chars[i + 2]);
                if (i + 3 < len) {
                    sb.append("-");
                }
                i += 3;
            } else {
                sb.append(chars[i++]);
            }
        }
        return sb.toString();
    }
}
