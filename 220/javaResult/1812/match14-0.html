class Solution {
    public String reformatNumber(String number) {
        List<Character> s = new ArrayList<>();
        for (char c: number.toCharArray()) {
            if (c == ' ' || c == '-') continue;
            s.add(c);
        }
        StringBuilder sb = new StringBuilder();
        int index = 0;
        while (s.size() - index > 4) {
            for (int i = 0; i < 3; i++) sb.append(s.get(index++));
            if (index < s.size() - 1) sb.append('-');
        }
        if (s.size() - index == 2) {
            sb.append(s.get(index++));
            sb.append(s.get(index++)); 
        } else if (s.size() - index == 3) {
            sb.append(s.get(index++)); 
            sb.append(s.get(index++)); 
            sb.append(s.get(index++)); 
        } else if (s.size() - index == 4) {
            sb.append(s.get(index++)); 
            sb.append(s.get(index++)); 
            sb.append('-');
            sb.append(s.get(index++)); 
            sb.append(s.get(index++)); 
        }
        return sb.toString();        
    }
}
