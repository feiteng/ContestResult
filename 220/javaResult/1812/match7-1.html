class Solution {
    public String reformatNumber(String number) {
        StringBuffer sb = new StringBuffer();
        for (char ch: number.toCharArray()) {
            if (ch == ' ' || ch == '-') continue;
            sb.append(ch);
        }
        
        StringBuffer res = new StringBuffer();
        int idx = 0;
        while (idx < sb.length()) {
            if (idx + 4 < 
sb.length()) {
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
                res.append('-');
            } else 
if (idx + 2 == sb.length()) {
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
            } else if (idx + 3 == sb.length()) {
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
            } else {
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
                res.append('-');
                res.append(sb.charAt(idx++));
                res.append(sb.charAt(idx++));
            }
        }
        
        return res.toString();
    }
}
