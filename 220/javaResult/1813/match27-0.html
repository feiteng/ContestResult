class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int len = nums.length;
        int l = 0;
        int sum = 0;
        int windowSum = 0;
        Set<Integer> window = new HashSet<>();
        
        for(int r = 0; r < len; r++) {
            int rightNum = nums[r];
            while(window.contains(rightNum)) {
                int leftNum = nums[l];
                windowSum -= leftNum;
                window.remove(leftNum);
                l++;
            }
            
            windowSum += rightNum;
            window.add(rightNum);
            
            sum = Math.max(sum, windowSum);
        }
        
        return sum;
    }
}
