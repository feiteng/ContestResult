class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int ans = 0;
        int f = 0;
        int[] a = new int[100005];
        int l = 0;
        for (int i =0; i < nums.length; i++) {
            f += nums[i];
            a[nums[i]]++;
            while(a[nums[i]] > 1) {
                f -= nums[l];
                a[nums[l]]--;
                l++;
            }
            ans = Math.max(ans, f);
        }
        
        return ans;
    }
}
