class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int left = 0;
        int n = nums.length;
        int res = 0;
        int sum = 0;
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < n; i++) {
            while (set.contains(nums[i])) {
                set.remove(nums[left]);
                sum -= nums[left];
                left++;
            }
            set.add(nums[i]);
            sum += nums[i];
            res = Math.max(res, sum);
        }
        return res;
    }
}
