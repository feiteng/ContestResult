class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int max = 0;
        Set<Integer> set = new HashSet<>();
        int start = 0;
        int cur = 0;
        for (int i = 0; i < nums.length; i++) {
            while (set.contains(nums[i])) {
                set.remove(nums[start]);
                cur -= nums[start++];
            }
            set.add(nums[i]);
            cur += nums[i];
            max = Math.max(max, cur);
        }
        return max;
    }
}
