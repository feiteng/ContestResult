class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int begin = 0; 
        int end = 0; 
        int n = nums.length; 
        long sum = (long)0; 
        long result = (long)0; 
        Set<Integer> set=  new HashSet<>(); 
        
        while(begin < n) {
            int curr = nums[begin];
            while (set.contains(curr)){
                set.remove(nums[end]);
                sum -= (long)nums[end];
                end++;
            }
            set.add(curr);
            begin++;
            sum += (long)curr;
            result = Math.max(sum, result);
        }
        return (int) result; 
    }
}
