class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int out = 0;
        
        int sum = 0;
        int start = 0;
        Set<Integer> curr = new HashSet<>();
        
        for(int i = 0; i < nums.length; i++) {
            while(curr.contains(nums[i])) {
                curr.remove(nums[start]);
                sum -= nums[start];
                start++;
            }
            
            curr.add(nums[i]);
            sum += nums[i];
            out = Math.max(out, sum);
        }
        
        return out;
    }
}
