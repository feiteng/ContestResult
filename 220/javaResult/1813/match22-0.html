class Solution {
    public int maximumUniqueSubarray(int[] a) {
        int max = 0, n = a.length, sum = 0, left = 0;
        Set<Integer> set = new HashSet();
        for(int i = 0; i < n; i++){
            while(set.contains(a[i])){
                sum -= a[left];
                set.remove(a[left++]);
            }
            sum += a[i];
            set.add(a[i]);
            max = Math.max(sum, max);
            // System.out.println(sum);
        }
        return max;
    }
}
