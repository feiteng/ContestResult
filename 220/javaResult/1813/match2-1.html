class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int sum = 0, l = 0, maxV = 0;
        Set<Integer> numSet = new HashSet<>();
        
        for (int r = 0; r < nums.length; ++r) {
            while (numSet.contains(nums[r])) {
                sum -= nums[l];
                numSet.remove(nums[l++]);  
            }
            numSet.add(nums[r]);
            sum += nums[r];
            maxV = Math.max(sum, maxV);
        }
        
        return maxV;
    }
}
