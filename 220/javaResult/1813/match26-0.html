class Solution {
  public int maximumUniqueSubarray(int[] nums) {
    int n = nums.length;

    int i = 0;
    int j = 1;

    Set<Integer> set = new HashSet<>(Set.of(nums[0]));

    int current = nums[0];
    int max = current;

    while (i < n - 1 && j < n) {
      if (!set.contains(nums[j])) {
        current += nums[j];
        max = Math.max(max, current);
        set.add(nums[j]);
        j++;
      } else {
        current -= nums[i];
        set.remove(nums[i]);
        i++;
      }
    }
    return max;
  }
}
