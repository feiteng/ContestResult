class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int l = 0, r = 0, n = nums.length;
        Set<Integer> set = new HashSet<>();
        int max = 0;
        int sum = 0;
        while (r < n) {
            while (set.contains(nums[r])) {
                set.remove(nums[l]);
                sum -= nums[l];
                l++;
            }
            sum += nums[r];
            set.add(nums[r]);
            // System.out.println(sum);
            r++;
            max = Math.max(sum, max);
        }
        return max;
    }
}
