class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int l = 0;
        int r = 0;
        Set<Integer> set = new HashSet<>();
        int count = 0;
        int max = 0;
        while(r < nums.length){
            while(set.contains(nums[r])){
                set.remove(nums[l]);
                count -= nums[l++];
            }
            set.add(nums[r]);
            count += nums[r++];
            max = Math.max(max, count);
        }
        return max;
    }
}
