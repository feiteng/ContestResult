class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int[] counter = new int[10001];
        int i = 0;
        int sum = 0;
        int output = 0;
        for(
int j = 0; j < nums.length; j++) {
            counter[nums[j]]++;
            sum += nums[j];
            while(counter[nums[j]] >= 2) {
                counter[nums[i]]--;
                sum -= nums[i];
                i++;
            }
            output = Math.max(output, sum);
        }
        return output;
    }
}
