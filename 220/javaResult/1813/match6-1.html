class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int result = 0;
        int tmp = 0;
        int left = 0;
        Set<Integer> set = new HashSet<>();
        for (int right = 0; right < nums.length; right++) {
            while (set.contains(nums[right])) {
                set.remove(nums[left]);
                tmp -= nums[left++];
            }
            set.add(nums[right]);
            tmp += nums[right];
            result = Math.max(result, tmp);
        }
        
        return result;
    }
}
