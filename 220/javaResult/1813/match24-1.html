class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        
        int start = 0;
        int sum = 0;
        int maxScore = 0;
        
        for(int end = 0; end < nums.length; end++){
            sum += nums[end];
            
            while(set.contains(nums[end])){
                sum -= nums[start];
                set.remove(nums[start]);
                start++;
            }
            
            set.add(nums[end]);
            maxScore = Math.max(maxScore, sum);
        }
        
        return maxScore;
    }
}
