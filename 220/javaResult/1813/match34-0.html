class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int[] visited = new int[10010];
        int l = 0, sum = 0, ans = 0;
        for (int i = 0; i < nums.length; i++) {
            int v = nums[i];
            visited[v]++;
            sum += v;
            while (visited[v] > 1) {
                visited[nums[l]]--;
                sum -= nums[l];
                l++;
            }
            ans = Math.max(ans, sum);
        }
        return ans;
    }
}
