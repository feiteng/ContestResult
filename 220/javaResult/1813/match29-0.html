class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        
        int n = nums.length;
        int l = 0;
        int r = 0;
        int sum = 0;
        int max = 0;
        Set<Integer> set = new HashSet();
        
        while(r<n){
            
            while(l<r && set.contains(nums[r])){
                sum -= nums[l];
                set.remove(nums[l]);
                l++;
            }
            sum += nums[r];
            set.add(nums[r]);
            max = Math.max(max,sum);
            r++;
        }
        
        return max;
    }
}
