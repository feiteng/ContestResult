class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int sum = 0, res = 0, left = 0;
        HashSet<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            while (set.contains(nums[i])) {
                sum -= nums[left];
                set.remove(nums[left++]);
            }
            set.add(nums[i]);
            sum += nums[i];
            res = Math.max(res, sum);
        }
        return res;
    }
}
