class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int[] count = new int[10001];
        int sum = 0;
        int res = 0;
        int n = nums.length;
        for (int i = 0, j = 0; j < n; j++) {
            count[nums[j]]++;
            sum += nums[j];
            while (count[nums[j]] > 1) {
                count[nums[i]]--;
                sum -= nums[i];
                i++;
            }
            res = Math.max(sum, res);
        }
        return res;
    }
}
