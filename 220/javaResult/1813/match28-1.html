class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int sum = 0;
        int i = 0;
        int j = 0;
        Set<Integer> set = new HashSet<>();
        int ans = 0;
        int n = nums.length;
        while (j < n) {
            int t = nums[j];
            
while (set.contains(t)) {
                sum -= nums[i];
                set.remove(nums[i]);
                i++;
            }
            
            sum += t;
            set.add(t);
            j++;
            ans = Math.max(ans, sum);
        }
        return ans;
    }
}
