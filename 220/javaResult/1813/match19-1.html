import java.util.*;



class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int l = 0, r = 0;
        int max =0;
        int curSum = 0;
        int n = nums.length;
        Set<Integer> seen = new HashSet<>();
        while (r < n) {
            while (seen.contains(nums[r])) {
                curSum -= nums[l];
                seen.remove(nums[l]);
                l++;
            }
            seen.add(nums[r]);
            curSum += nums[r];
            r++;
            max = Math.max(max, curSum);
        }
        return max;
    }
}
