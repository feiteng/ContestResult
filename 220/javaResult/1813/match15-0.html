class Solution {
    public int maximumUniqueSubarray(int[] nums) {
		int i = 0, s = 0, m = 0;
		Set<Integer> set = new HashSet<>();
		for (int j = 0; j < nums.length; j++) {
			while (set.contains(nums[j])) {
				s -= nums[i];
				set.remove(nums[i]);
				i++;
			}
			s += nums[j];
			set.add(nums[j]);
			m = Math.max(m, s);
		}
		return m;
    }
}
