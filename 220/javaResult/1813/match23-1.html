class Solution {
     public int maximumUniqueSubarray(int[] nums) {
        int l = 0;
        int r = 0;
        Set<Integer> set = new HashSet<>();
        int sum = 0;
        int res = 0;
        while(r < nums.length){
            while(set.contains(nums[r]) && l < r){
                set.remove(nums[l]);
                sum-=nums[l];
                l++;
            }
            sum += nums[r];
            set.add(nums[r]);
            r++;
            res = Math.max(sum ,res);
        }
        return res;
    }
}
