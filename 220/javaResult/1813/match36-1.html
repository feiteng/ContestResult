class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int []count = new int[10002];
        int l = 0;
        int sum = 0;
        int ans = 0;
        for (
int num : nums) {
            count[num]++;
            sum += num;
            while (count[num] > 1) {
                count[nums[l]]--;
                sum -= nums[l];
                l++;
            }
            ans = Math.max(ans, sum);
        }
        return ans;
    }
}
