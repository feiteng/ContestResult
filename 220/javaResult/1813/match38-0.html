class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        // sliding window
        int n = nums.length, sum = 0, res = 0;
        HashSet<Integer> visit = new HashSet<>();
        
        for (int i = 0, j = 0; i < n; i++) {
            while (visit.contains(nums[i])) {
                sum -= nums[j];
                visit.remove(nums[j]);
                j++;
            }
            visit.add(nums[i]);
            sum += nums[i];
            res = Math.max(sum, res);
        }
        
        return res;
    }
}
