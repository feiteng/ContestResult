public class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        int[] occur = new int[(int) 1e4 + 1];
        int n = nums.length;
        int ans = 0;
        int sum = 0;
        for (int i = 0, r = -1; i < n; i++) {
            while (r + 1 < n && occur[nums[r + 1]] == 0) {
                occur[nums[r + 1]]++;
                sum += nums[r + 1];
                r++;
            }
            ans = Math.max(ans, sum);
            occur[nums[i]]--;
            sum -= nums[i];
        }
        return ans;
    }
}
