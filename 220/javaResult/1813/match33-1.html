import java.util.BitSet;


class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        BitSet bit = new BitSet();
        int sum = 0;
        int l = 0;
        int ans = 0;
        for (int x : nums) {
            while (bit.get(x)) {
                int y = nums[l];
                sum -= y;
                
bit.set(y, false);
                l++;
            }
            sum += x;
            bit.set(x);
            ans = Math.max(ans, sum);
        }
        return ans;
    }
}
