class Solution {
    public int maximumUniqueSubarray(int[] nums) {
        //just keep track of maximum size array
        int n = nums.length;
        int l = 0;
        int sum = 0;
        int ans = 0;
        Set<Integer> s = new HashSet<Integer>();
        
        for (int r = 0; r < n; r++) {
            int num = nums[r];
            
            while (s.contains(num)) {
                
s.remove(nums[l]);
                sum -= nums[l++];
            }
            
            sum += nums[r];
            s.add(nums[r]);
            // System.out.printf("r=%d, l=%d, sum=%d\n", r, l, sum);
            ans = Math.max(ans, sum);
        }
        
        return ans;
    }
}
