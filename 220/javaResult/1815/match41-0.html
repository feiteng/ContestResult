class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] A, int[][] query) {
        int arr[]=new int[n];
        for(int i=0;i<arr.length;i++){
            arr[i]=i;
        }
        
        int cost=0;
        int j=0;
        boolean res[]=new boolean[query.length];
        int q[][]=new int[query.length][4];
        for(int i=0;i<query.length;i++){
            q[i][0]=query[i][0];
            q[i][1]=query[i][1];
            q[i][2]=query[i][2];
            q[i][3]=i;
        }
        
        Arrays.sort(q,(a,b)->{
            return a[2]-b[2];
        });
        
        
        
        
        Arrays.sort(A,(a,b)->{
            return a[2]-b[2];
        });
        
        
        //for(int pair[]:A){
           // System.out.println(pair[0]+" "+pair[1]+" "+pair[2]+" ");
       // }
        
        for(int i=0;i<A.length;i++){
            int rr1=find(arr,A[i][0]);
            int rr2=find(arr,A[i][1]);
            
           
            
            int oldj=j;
            cost=Math.max(cost,A[i][2]);
            while(j<q.length&&q[j][2]<=cost){
                j++;
            }

            //System.out.println(oldj+"  "+j+"  "+cost);
            
            
            for(int k=oldj;k<j;k++){
               
                int  r1=find(arr,q[k][0]);
                int r2=find(arr,q[k][1]);
                //if(k==2)System.out.println("yes"+"  "+q[k][0]+"  "+q[k][1]+"  "+r1+"  "+r2);
                int index=q[k][3];
                if(r1==r2){
                    res[index]=true;
                }
            }
            
            if(rr1==rr2)continue;
            arr[rr1]=rr2;
           
        }
        
        
        for(int i=j;i<q.length;i++){
            if(q[i][2]>cost){
                int r1=find(arr,q[i][0]);
                int r2=find(arr,q[i][1]);
                int index=q[i][3];
                if(r1==r2){
                    res[index]=true;
                }
            }
        }
        
        
        return res;
    }
    
    
    public int find(int nums[],int x){//union find => find method
		if(nums[x]==x)return x;
		int root=find(nums,nums[x]);
		nums[x]=root;
		return root;
	}
}
