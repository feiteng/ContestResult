class Solution {

    int[] father;

    int find(int u) {
        if (father[u] == u) return u;
        return father[u] = find(father[u]);
    }

    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        father = new int[n];
        for (int i = 0; i < n; i++) father[i] = i;
        Arrays.sort(edgeList, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[2] - o2[2];
            }
        });

        int[][] nq = new int[queries.length][4];
        for (int i = 0; i < queries.length; i++) {
            for (int j = 0; j < 3; j++) {
                nq[i][j] = queries[i][j];
            }
            nq[i][3] = i;
        }
        Arrays.sort(nq, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[2] - o2[2];
            }
        });

        int idx = 0;
        boolean[] res = new boolean[queries.length];
        for (int i = 0; i < nq.length; i++) {
            while (idx < edgeList.length && edgeList[idx][2] < nq[i][2]) {
                int u = edgeList[idx][0];
                int v = edgeList[idx][1];
                father[find(u)] = find(v);
                idx++;
            }
            res[nq[i][3]] = find(nq[i][0]) == find(nq[i][1]);
        }
        return res;
    }
}
