class Solution {
    private class DSU{
        int[] parent;

        public DSU(int N) {
            parent = new int[N];
            for (int i = 0; i < N; ++i) {
                parent[i] = i;
            }
        }

        public int find(int x) {
            if (parent[x] != x) parent[x] = find(parent[x]);
            return parent[x];
        }

        public void union(int x, int y) {
            parent[find(x)] = find(y);
        }
    }

    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        boolean[] res = new boolean[queries.length];
        Arrays.sort(edgeList, (a, b) -> (a[2] - b[2]));
        Integer[] sort = new Integer[queries.length];
        for (int i = 0; i < queries.length; ++i) {
            sort[i] = i;
        }
        Arrays.sort(sort, (a, b) -> (queries[a][2] - queries[b][2]));
        DSU uf = new DSU(n);
        int pos = 0;
        for (int i = 0; i < sort.length; ++i) {
            int[] q = queries[sort[i]];
            while (pos < edgeList.length && edgeList[pos][2] < q[2]) {
                uf.union(edgeList[pos][0], edgeList[pos][1]);
                ++pos;
            }
            res[sort[i]] = (uf.find(q[0]) == uf.find(q[1]));
        }
        return res;
    }
}
