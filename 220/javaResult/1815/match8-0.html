class Solution {
    int[] fathers;
    private int find(int a) {
        if (fathers[a] == a) {
            return a;
        }
        fathers[a] = find(fathers[a]);
        return fathers[a];
    }
    private void union(int a, int b) {
        int fa = find(a);
        int fb = find(b);
        if (fa != fb) {
            fathers[fa] = fb;
        }
    }
    class Query {
        int[] query;
        int idx;
        Query(int[] _query, int _idx) {
            query = _query;
            idx = _idx;
        }
    }
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        fathers = new int[n];
        for (int i = 0; i < n; ++i) {
            fathers[i] = i;
        }
        Arrays.sort(edgeList, (a, b) -> 
            a[2] - b[2]
        );
        int len = queries.length;
        boolean[] ans = new boolean[len];
        Query[] q = new Query[len];
        for (int i = 0; i < len; ++i) {
            q[i] = new Query(queries[i], i);
        }
        Arrays.sort(q, (a, b) -> 
           a.query[2] - b.query[2]
        );
        
        int edgeIdx = 0;
        for (int i = 0; i < len; ++i) {
            int limit = q[i].query[2];
            int src = q[i].query[0];
            int dest = q[i].query[1];
            while (edgeIdx < edgeList.length && edgeList[edgeIdx][2] < limit) {
                union(edgeList[edgeIdx][0], edgeList[edgeIdx][1]);
                ++edgeIdx;
            }
            ans[q[i].idx] = (find(src) == find(dest));
        }
        
        return ans;
    }
}
