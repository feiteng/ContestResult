class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] q) {
        
boolean[] res = new boolean[q.length];
        int[][] queries = new int[q.length][4];
        for (int i = 0; i < q.length; ++i) {
            queries[i][0] = q[i][0];
            queries[i][1] = q[i][1];
            queries[i][2] = q[i][2];
            queries[i][3] = i;
        }
        Arrays.sort(queries, (a, b) -> { return a[2] - b[2]; });
        Arrays.sort(edgeList, (a, b) -> { return a[2] - b[2]; });
        
        int[] parent = new int[100001];
        for (int i = 0; i < parent.length; ++i) parent[i] = i;
        
        int j = 0;
        for (int i = 0; i < queries.length; ++i) {
            int max = queries[i][2];
            while (j < edgeList.length && edgeList[j][2] < max) {
                int p1 = find(parent, edgeList[j][0]), p2 = find(parent, edgeList[j][1]);
                parent[p1] = parent[p2];
                ++j;
            }
            // System.out.println(queries[i][3]);
            res[queries[i][3]] = find(parent, queries[i][0]) == find(parent, queries[i][1]);
        
}
        return res;
        
    }
    
    public int find(int[] parent, int i) {
        if (i == parent[i]) return i;
        return parent[i] = find(parent, parent[i]);
    }
}
