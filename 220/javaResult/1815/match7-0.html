import java.util.Arrays;
import java.util.Comparator;

class Solution {
    int[] fa;

    private int find(int x) {
        if (x == fa[x]) {
            return fa[x];
        }
        return fa[x] = find(fa[x]);
    }

    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        int m = edgeList.length;
        int Q = queries.length;
        boolean[] ans = new boolean[Q];
        int[][] query = new int[Q][4];
        for (int i = 0; i < Q; i++) {
            query[i] = new int[]{queries[i][0], queries[i][1], queries[i][2], i};
        }
        Arrays.sort(edgeList, Comparator.comparingInt(a -> a[2]));
        Arrays.sort(query, Comparator.comparingInt(a -> a[2]));
        fa = new int[n];
        for (int i = 0; i < n; i++) {
            fa[i] = i;
        }
        int edg = 0;
        for (int i = 0; i < Q; i++) {
            while (edg < m && edgeList[edg][2] < query[i][2]) {
                int fx = find(edgeList[edg][0]);
                int fy = find(edgeList[edg][1]);
                if (fx != fy) {
                    fa[fx] = fy;
                }
                edg++;
            }
            ans[query[i][3]] = find(query[i][0]) == find(query[i][1]);
        }
        return ans;
    }
}
