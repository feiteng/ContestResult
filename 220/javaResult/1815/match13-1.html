class Solution {
 
public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        Arrays.sort(edgeList, (a, b)->{
            return a[2] - b[2];
        });
        int qn = queries.length;
        Integer[] queryIndex = new Integer[queries.length];

        for(int i = 0; i < queryIndex.length; ++i){
            queryIndex[i] = i;
        
}


        Arrays.sort(queryIndex, (a, b)->{
            return queries[a][2] - queries[b][2];
        });
        
        boolean[] res = new boolean[qn];
        Uset ss = new Uset(n);
        int ei = 0;
        for(int i = 0; i < qn; ++i){
            int w = queries[queryIndex[i]][2];
            
            while(ei < edgeList.length  && edgeList[ei][2] < w){
                ss.union(edgeList[ei][0], edgeList[ei][1]);
                ei++;
            }
            
            res[queryIndex[i]] = ss.isInOnTree(queries[queryIndex[i]][0],queries[queryIndex[i]][1]);
        }
        
        return res;
    }
    
    static class Uset{
        int[] parent;
        
        public Uset(int n){
            parent = new int[n];
            for(int i = 0; i < n; ++i){
                parent[i] = i;
            }
        }
        
        int getRoot(int a){
            
            if(parent[a] != a){
                parent[a] = getRoot(parent[a]);
            }
            
            return parent[a];
        }
        
        public boolean isInOnTree(int a, int b){
            return getRoot(a) == 
getRoot(b);
        }
        
        public void union(int a, int b){
            int pa = getRoot(a);
            int pb = getRoot(b);
            if(pa != pb){
                parent[pa] = pb;
            }
        }
    }
}
