class Solution {
    private int[] fn;

    private int findx(int n) {
        if (fn[n] == n) {
            return n;
        } else {
            fn[n] = findx(fn[n]);
            return fn[n];
        }
    }

    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        Arrays.sort(edgeList, (a, 
b) -> (a[2] - b[2]));
        fn = new int[n];
        for (int idx=0;idx<n;idx++) {
            fn[idx] = idx;
        }
        
int[][] tQueries = new int[queries.length][];
        boolean[] ans = new boolean[queries.length];
        for (int idx=0;idx<queries.length;idx++) {
            tQueries[idx] = new int[]{queries[idx][0], queries[idx][1], queries[idx][2], idx};
        }
        Arrays.sort(tQueries, (a,b) -> (a[2] - b[2]));
        
int start = 0;
        for (int idx=0;idx< queries.length;idx++) {
            while ((start < edgeList.length) && (edgeList[start][2] < tQueries[idx][2])) {
                int x = findx(edgeList[start][0]);
                int y = findx(edgeList[start][1]);
                if (x < y) {
                    fn[y] = x;
                } else {
                    fn[x] = y;
                }
                start += 1;
            }
            int x = findx(tQueries[idx][0]);
            int y = findx(tQueries[idx][1]);
            ans[tQueries[idx][3]] = (x==y);
        }
        return ans;
    }
}
