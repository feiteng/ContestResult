class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        Arrays.sort(edgeList, (a, b) -> a[2] - b[2]);
        
        int[][] newQueries = new int[queries.length][4];
        
        for (int i = 0; i < queries.length; i++) {
            newQueries[i][0] = queries[i][0];
            newQueries[i][1] = queries[i][1];
            newQueries[i][2] = queries[i][2];
            newQueries[i][3] = i;
        }
        
        Arrays.sort(newQueries, (a, b) -> a[2] - b[2]);

        int[] par = new int[n];
        for (int i = 0; i < n; i++) par[i] = i;
        
        int edgeInd = 0;
        
        boolean[] ans = new boolean[queries.length];
        
        for (int i = 0; i < newQueries.length; i++) {
            int p = newQueries[i][0];
            int q = newQueries[i][1];
            int cost = newQueries[i][2];
            int ind = newQueries[i][3];
            
            // System.out.printf("query=(%d,%d,%d,%d)\n", p, q, cost, ind);
            
            while (edgeInd < edgeList.length && edgeList[edgeInd][2] < cost) {
                
                int u = edgeList[edgeInd][0];
                int v = edgeList[edgeInd][1];
                
                // System.out.printf("Adding (%d, %d, %d)\n", u, v, edgeList[edgeInd][2]);
                int parU = find(par, u);
                int parV = find(par, v);
                
                if (parU != parV) {
                    union(par, parU, parV);
                }
                
                edgeInd++;
            }
            
            int parP = find(par, p);
            int parQ = find(par, q);
            
            if (parP == parQ) {
                ans[ind] = true;
            }
            else {
                ans[ind] = false;
            }
        }
        
        return ans;
    }
    
    private int find(int[] par, int x) {
        if (par[x] == x) {
            return x;
        }
        
        return par[x] = find(par, par[x]);
    }
    
    private void union(int[] par, int x, int y) {
        par[y] = par[x];
    }
}
