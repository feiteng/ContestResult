class Solution {
    
    int[] fa = new int[100005];
    
    private int find(int x) {
        if (x == fa[x]) return x;
        return fa[x] = find(fa[x]);
    }
    
    private void union(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            fa[fx] = fy;
        }
    }
    
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        Arrays.sort(edgeList, (x, y) -> {
            return x[2] - y[2];
        });
        int m = queries.length;
        Integer[] q = new Integer[m];
        for (int i = 0; i < m; ++i) q[i] = i;
        for (int i = 0; i < n; ++i) fa[i] = i;
        Arrays.sort(q, (x, y) -> {
            return queries[x][2] - queries[y][2];
        });
        boolean[] ans = new boolean[m];
        for (int i = 0, j = 0; i < m; ++i) {
            int idx = q[i];
            while (j < edgeList.length && edgeList[j][2] < queries[idx][2]) {
                union(edgeList[j][0], edgeList[j][1]);
                ++j;
            } 
            
            int f1 = find(queries[idx][0]);
            int f2 = find(queries[idx][1]);
            ans[idx] = f1 == f2;
        }
        return ans;
    }
}
