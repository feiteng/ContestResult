class Solution {
    class Query {
        int x;
        int y;
        int dist;
        int ind;
        public Query(int x, int y, int dist, int ind) {
            this.x = x;
            this.y = y;
            this.dist = dist;
            this.ind = ind;
        }
    }
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        PriorityQueue<Query> edges = new PriorityQueue<>((a,b) ->a.dist -b.dist);
        PriorityQueue<Query> queryQ = new PriorityQueue<>((a,b) ->a.dist -b.dist);
        for (int j = 0; j < queries.length; j++) {
            int[] query = queries[j];
            queryQ.add(new Query(query[0], query[1], query[2], j));
        }
        for (int[] edge : edgeList) {
            edges.add(new Query(edge[0], edge[1], edge[2], 0));
        }
        int[] uf = new int[n];
        boolean[] ret = new boolean[queries.length];
        Arrays.fill(uf, -1);
        while (queryQ.size() > 0) {
            Query query = queryQ.poll();
            while (edges.size() > 0 && edges.peek().dist < query.dist) {
                Query edge = edges.poll();
                int repA = find(uf, edge.x);
                int repB = find(uf, edge.y);
                if (repA != repB) {
                    uf[repA] = repB;
                }
            }
            int repA = find(uf, query.x);
            int repB = find(uf, query.y);
            ret[query.ind] = repA == repB;
        }
        return ret;
    }
    
    public int find(int[] uf, int ind) {
        if (uf[ind] == -1) {
            return ind;
        }
        uf[ind] = find(uf, uf[ind]);
        return uf[ind];
    }
}
