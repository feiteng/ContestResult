class Solution {
    
    class UnionFind {
        int[] roots;
        int[] ranks;
        
        UnionFind(int N) {
            roots = new int[N];
            
ranks = new int[N];
            for (int i = 0; i < N; i++) {
                roots[i] = i;
            }
        }
        
        int find(int i) {
            while (roots[i] != i) {
                roots[i] = roots[roots[i]];
                i = roots[i];
            }
            
            
return i;
        }
        
        void union(int i, int j) {
            i = find(i);
            j = find(j);
                
            if (i == j) return;
            
            if (ranks[i] < ranks[j]) {
                roots[i] = j;
            } else if (ranks[i] > ranks[j]) {
                roots[j] = i;
            } else {
                roots[j] = i;
                ranks[i]++;
            }
        
}
    }
    
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        int Q = queries.length;
        for (int i = 0; i < Q; i++) {
            queries[i] = new int[] { queries[i][0], queries[i][1], queries[i][2], i };
        }
        
        PriorityQueue<int[]> eq = new PriorityQueue<>((a, b) -> a[2] - b[2]);
        for (int i = 0; i < edgeList.length; i++) eq.offer(edgeList[i]);
        
        Arrays.sort(queries, (a, b) -> a[2] - b[2]);
        
        UnionFind uf = new UnionFind(n);
        boolean[] ans = new boolean[Q];
        for (int i = 0; i < Q; i++) {
            int l = queries[i][2];
            int u = queries[i][0];
            int v = queries[i][1];
            
            while (!eq.isEmpty() && eq.peek()[2] < l) {
                int[] e = eq.poll();
                
uf.union(e[0], e[1]);
            }
            
            ans[queries[i][3]] = uf.find(u) == uf.find(v);
        }
        
        return ans;
    }
}
