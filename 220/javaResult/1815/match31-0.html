class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        UnionFind uf = new UnionFind(n);

        List<query> qs = new ArrayList<>();
        for (int i=0;i<queries.length;i++) {
            qs.add(new query(queries[i][0], queries[i][1], queries[i][2], i));
        }
        Collections.sort(qs, (a,b)->a.limit-b.limit);
        boolean[] res = new boolean[queries.length];
        Arrays.sort(edgeList, (a,b)->a[2]-b[2]);

        int id = 0;
        for (query qy:qs) {
            while (id<edgeList.length && edgeList[id][2]<qy.limit) {
                uf.union(edgeList[id][0], edgeList[id][1]);
                id++;
            }
            res[qy.index] = uf.find(qy.p) == uf.find(qy.q);
        }

        return res;
    }

    class query{
        int p;
        int q;
        int limit;
        int index;
        query(int p, int q, int limit, int id) {
            this.p = p;
            this.q = q;
            this.limit = limit;
            index = id;
        }
    }
    class UnionFind {
        int[] fa;
        UnionFind(int size) {
            fa = new int[size];
            Arrays.fill(fa, -1);
        }

        public int find(int i) {
            if (fa[i]==-1) return i;
            int t = find(fa[i]);
            return fa[i] = t;
        }

        public void union(int u, int v) {
            int f1 = find(u);
            int f2 = find(v);
            if(f1!=f2) {
                fa[f2] = f1;
            }
        }
    }
}
