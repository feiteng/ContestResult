class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] q) {
        int[][] queries = new int[q.length][4];
        for(int i = 0; i < q.length; i++) {
            queries[i][0] = q[i][0];
            queries[i][1] = q[i][1];
            queries[i][2] = q[i][2];
            queries[i][3] = i;
        }
        Arrays.sort(queries, (a, b) -> {
            return a[2] - b[2];
        });
        Arrays.sort(edgeList, (a, b) -> {
            return a[2] - b[2];
        });
        boolean[] output = new boolean[queries.length];
        DSU dsu = new DSU(n);
        int j = 0;
        for(int i = 0; i < queries.length; i++) {
            while(j < edgeList.length && edgeList[j][2] < queries[i][2]) {
                dsu.union(edgeList[j][0], edgeList[j][1]);
                j++;
            }
            if(dsu.findParent(queries[i][0]) == dsu.findParent(queries[i][1])) {
                output[queries[i][3]] = true;
            }
        }
        return output;
    }
    
    static class DSU {
        int[] parent;
        
        DSU(int n) {
            parent = new int[n];
            for(int i = 0; i < n; i++) parent[i] = i;
        }
        
        void union(int a, int b) {
            parent[findParent(a)] = parent[findParent(b)];
        }
        
        int findParent(int a) {
            if(parent[a] != a) {
                parent[a] = findParent(parent[a]);
            }
            return parent[a];
        }
        
    }
    
}
