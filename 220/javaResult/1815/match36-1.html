class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        if (edgeList == null || queries == null) {
            return new boolean[]{};
        }
        
        Arrays.sort(edgeList, (a, b) -> a[2] - b[2]);
        
int[][] queriesSorted = new int[queries.length][4];
        for (int i = 0; i < queries.length; i++) {
            queriesSorted[i][0] = queries[i][0];
            queriesSorted[i][1] = queries[i][1];
            queriesSorted[i][2] = queries[i][2];
            queriesSorted[i][3] = i;
        }
        Arrays.sort(queriesSorted, (a, b) -> a[2] - b[2]);
        
        boolean[] res = new boolean[queries.length];
        int[] map = new int[n];
        Arrays.fill(map, -1);
        int edgeIndex = 0;
        for (int i = 0; i < queriesSorted.length; i++) {
            while (edgeIndex < edgeList.length && edgeList[edgeIndex][2] < queriesSorted[i][2]) {
                union(map, edgeList[edgeIndex]);
                edgeIndex++;
            }
            
            res[queriesSorted[i][3]] = find(map, queriesSorted[i]);
        
}
        
        return res;
    }
    
    private void union(int[] map, int[] edge) {
        int x = parent(map, edge[0]);
        int y = parentSet(map, edge[1], x);
        if (x == y) {
            return;
        }
        
        map[y] = x;
        map[edge[0]] = x;
        map[edge[1]] = x;
    
}
    
    private int parent(int[] map, int x) {
        if (map[x] == -1 || map[x] == x) {
            return x;
        }
        return parent(map, map[x]);
    }

    private int parentSet(
int[] map, int x, int set) {
        if (map[x] == -1 || map[x] == x) {
            return x;
        }
        int res = parent(map, map[x]);
        map[x] = set;
        
return res;
    }
    
    private boolean find(int[] map, int[] query) {
        return parent(map, query[0]) == parent(map, query[1]);
    }
}
