class Solution {
    public boolean[] distanceLimitedPathsExist(int n, int[][] edgeList, int[][] queries) {
        Arrays.sort(edgeList, (a, b) -> (a[2] - b[2]));
        int[][] q = new int[queries.length][4];
        for (int i = 0;i < queries.length;i++) {
            q[i] = new int[]{queries[i][0], queries[i][1], queries[i][2], i};
        
}
        Arrays.sort(q, (a, b) -> (a[2] - b[2]));
        int[] roots = new int[n];
        for (int i = 0;i < n;i++) {
            roots[i] = i;
        }
        int j = 0;
        boolean[] res = new boolean[q.length];
        for (int i = 0;i < q.length;i++) {
            int max = q[i][2];
            
            while (j < edgeList.length && max > edgeList[j][2]) {
                union(roots, edgeList[j][0], edgeList[j][1]);
                j++;
            }
            int root1 = q[i][0], root2 = q[i][1];
            root1 = findRoot(roots, root1);
            root2 = findRoot(roots, root2);
            res[q[i][3]] = root1 == root2;
            
        
} 
        return res;
    }
    private int findRoot(int[] roots, int root) {
        return roots[root] == root ? root : (
roots[root] = findRoot(roots, roots[root]));
    }
    private boolean union(int[] roots, int root1, int root2) {
        root1 = findRoot(roots, root1);
        root2 = findRoot(roots, root2);
        if (root1 != root2) {
            roots[root2] = root1;
            return true;
        }
        return false;
    }
}
