class Solution {
    
    int arr[];
    int rank[];
    
    public boolean[] distanceLimitedPathsExist(int n, int[][] edges, int[][] queries) {
        arr=new int[n];
        rank=new int[n];
        for(int j=0;j<n;j++)
        {
            arr[j]=j;
        }
        Arrays.fill(rank,1);
        
        
ArrayList<edge> edgeList=new ArrayList<>();
        for(int i[]:edges)
        {
            edgeList.add(new edge(i[0],i[1],i[2]));
        }
        
        Collections.sort(edgeList,new cust());
                ArrayList<query> querylist=new ArrayList<>();

        
        for(int j=0;j<queries.length;j++)
        {
            int i[]=queries[j];
            querylist.add(new query(i[0],i[1],i[2]-1,j));
        }
        
        
        
        Collections.sort(querylist,new comp());
        
        
        int ptr=0;
        int max=0;
        
        boolean ans[]=new boolean[queries.length];
        for(int i=0;i<edgeList.size();i++)
        {
            edge e=edgeList.get(i);
            max=Math.max(e.w,max);
            //System.out.println("max is now "+max);
            union(e.a,e.b);
            if(i<edgeList.size()-1)
            {
                int nextMax=Math.max(max,edgeList.get(i+1).w);
                //System.out.println("next max is "+nextMax);
                while(ptr<querylist.size()&&querylist.get(ptr).w<nextMax)
                {
                    
                    query ele=querylist.get(ptr);
                    //System.out.println(ele);
                    
                    //System.out.println("parent of "+ele.a+" is "+getParent(ele.a));
                    //System.out.println("parent of "+ele.b+" is "+getParent(ele.b));
                    
                    if(getParent(ele.a)==getParent(ele.b)&&ele.w>=max)
                    {
                        ans[ele.ind]=true;
                    }
                    ptr++;
                }
            }
            else
            {
                while(ptr<querylist.size())
                {
                    
                    query ele=querylist.get(ptr);
                    
                    if(getParent(ele.a)==getParent(ele.b)&&ele.w>=max)
                    {
                        ans[ele.ind]=true;
                    }
                    ptr++;
                }
            }
            
        
}
        
        return ans;
        
        
        
        
    }
    
    class query
    {
        int a;int b;int w;int ind;
        query(int a,int b,int w,int ind)
        {
            this.a=a;this.b=b;this.w=w;this.ind=ind;
        }
        public String toString()
        {
            return w+" "+ind;
        }
    }
    
    class comp implements Comparator<query>
    {
        public int compare(query q1,query q2)
        {
            if(q1.w!=q2.w)
            {
                return q1.w-q2.w;
            }
            return q1.ind-q2.ind;
        }
    }
    
    class cust implements Comparator<edge>
    {
        public int compare(edge e1,edge e2)
        {
            return e1.w-e2.w;
        }
    }
    
    class edge
    {
        int a;int b;int w;
        edge(int a,int b,int w)
        {
            this.a=a;this.b=b;this.w=w;
        }
    }
    
    
    
    
public void union(int x,int y)
    {
        int px=getParent(x);
        int py=getParent(y);
        if(rank[px]>rank[py])
        {
            arr[py]=px;
        }
        else if(rank[px]<rank[py])
        {
            arr[px]=py;
        }
        else
        {
            arr[px]=py;
            rank[py]+=1;
        }
    
}
    public int getParent(int x)
    {
        if(arr[x]==x)
            return x;
        int parent=getParent(arr[x]);
        arr[x]=parent;
        return parent;
    }
    
}
