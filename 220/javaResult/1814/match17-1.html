class Solution {
  public int maxResult(int[] nums, int k) {
    final int N = nums.length;
    int[] dp = new int[N];
    dp[0] = nums[0];
    Deque<Integer> deq = new ArrayDeque<>();
    deq.offerLast(0);
    for (int i = 1 - k, j = 1; j != N; i++, j++) {
      
dp[j] = dp[deq.peekFirst()] + nums[j];
      while (!deq.isEmpty() && deq.peekFirst() <= i) // if
        deq.pollFirst();
      while (!deq.isEmpty() && dp[j] >= dp[deq.peekLast()])
        deq.pollLast();
      deq.offerLast(j);
    }
    return dp[N - 1];
  }

}
