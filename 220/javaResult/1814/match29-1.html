class Solution {
    public int maxResult(int[] nums, int k) {
        int n=nums.length;
        //dp[i] = max score landing on i
        
int[] dp=new int[n];
        Arrays.fill(dp,Integer.MIN_VALUE);
        dp[0]=nums[0];
        
        //mono-decreasing 
        Deque<Integer> deque=new ArrayDeque<>(); 
        deque.offerLast(0);
        
        for(int i=1; i<n; i++){
            // TLE 
            // for(int j=Math.max(0,i-k); j<i; j++){
            //     dp[i]=Math.max(dp[i], dp[j]+nums[i]);
            // }
            
            // use mono queue to track largest value within k range before me.
            if(deque.peekFirst()+k<i){
                deque.pollFirst();
            }
            dp[i]=dp[deque.peekFirst()]+nums[i];
            while(deque.isEmpty()==false && dp[i]>=dp[deque.peekLast()]){
                deque.pollLast();
            }
            deque.offerLast(i);
        }
        
        return dp[n-1];
    }
}
