class Solution {
    
private class Node{
        int data, loc;
        public Node(int data, int loc) {
            this.data = data;
            this.loc = loc;
        }
    }
    
public int maxResult(int[] nums, int k) {
        int[] dp=new int[nums.length];
        dp[nums.length-1]=nums[nums.length-1];
        PriorityQueue<Node> maxHeap=new PriorityQueue<>((x,y)->y.data-x.data);
        maxHeap.add(new Node(dp[nums.length-1],nums.length-1));
        if (nums.length>=2){
            dp[nums.length-2]=nums[nums.length-1]+nums[nums.length-2];
            maxHeap.add(new Node(dp[nums.length-2],nums.length-2));
        }
        
        
        
for (int i = nums.length-3; i >=0 ; i--) {
            while (maxHeap.peek().loc-i>k){
                maxHeap.poll();
            }
            dp[i]=nums[i]+maxHeap.peek().data;
            maxHeap.add(new Node(dp[i],i));
        }
        return dp[0];
    }
}
