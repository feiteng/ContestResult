class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Arrays.fill(dp, Integer.MIN_VALUE);
        dp[0] = nums[0];
        for(int i = 0; i < n - 1; ++i) {
            for(int j = 1; j <= k && i + j < n; ++j) {
                dp[i + j] = Math.max(dp[i + j], dp[i] + nums[i + j]);
                if(nums[i + j] > 0) break;
            }
        }
        return dp[n - 1];
    }
}
