class Solution {
    public int maxResult(int[] ns, int k) {
        int n = ns.length;
        int[] a = new int[n];
        a[0] = ns[0];
        TreeMap<Integer, Integer> ks = new TreeMap<>();
        ks.put(a[0], 1);
        for (int i = 1; i < n; i++) {
            if (i - k - 1 >= 0) {
                ks.put(a[i-k-1], ks.get(a[i-k-1]) - 1);
                if (ks.get(a[i-k-1]) == 0) ks.remove(a[i-k-1]);
            }
            a[i] = ns[i];
            a[i] += ks.lastKey();
            ks.put(a[i], ks.getOrDefault(a[i], 0) + 1);
        }
        return a[n-1];        
    }
}
