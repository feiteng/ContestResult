class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Deque<Integer> dq = new LinkedList();
        for (int i = 0; i < n; ++i) {
            while (!dq.isEmpty() && k < i - dq.peek()) dq.poll();
            dp[i] = nums[i] + (dq.isEmpty() ? 0 : dp[dq.peek()]);
            while (!dq.isEmpty() && dp[dq.peekLast()] <= dp[i]) dq.pollLast();
            dq.add(i);
        }
        return dp[n - 1];
    }
}
