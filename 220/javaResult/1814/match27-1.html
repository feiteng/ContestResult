class Solution {    
    public int maxResult(int[] nums, int k) {
        int length = nums.length;
        int[] fn = new int[length];
        fn[0] = nums[0];
        Deque<Integer> mDeque = new ArrayDeque<Integer>();
        mDeque.addFirst(0);
        for (int idx=1;idx<length;idx++) {
            while( 
mDeque.getFirst() + k < idx) {
                mDeque.removeFirst();
            }
            fn[idx] = fn[mDeque.getFirst()] + nums[idx];
            while ((!mDeque.isEmpty()) && (fn[mDeque.getLast()] <= fn[idx])) {
                mDeque.removeLast();
            }
            mDeque.addLast(idx);
        }
        return fn[length-1];
    }
}
