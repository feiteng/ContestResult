class Solution {
	public int maxResult(int[] nums, int k) {
		int[] a = new int[nums.length];
		a[0] = nums[0];
		PriorityQueue<Node> q = new PriorityQueue<>((x, y) -> y.val - x.val);
		q.offer(new Node(nums[0], 0));
		for (int i = 1; i < a.length; i++) {
			while (q.peek().index < i - k) {
				q.poll();
			}
			a[i] = q.peek().val + nums[i];
			q.offer(new Node(a[i], i));
		}
		return a[a.length - 1];
	}

	private class Node {
		int val;
		int index;

		public Node(int val, int index) {
			this.val = val;
			this.index = index;
		}
	}
}
