class Solution {
    public int maxResult(int[] nums, int k) {
        int len = nums.length;
        int[] dp = new int[len];
        Arrays.fill(dp, Integer.MIN_VALUE);
        dp[0] = nums[0];
        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j <= i + k && j < len; j++) {
                dp[j] = Math.max(dp[j], dp[i] + nums[j]);
                if (nums[j] >= 0) break;
            }
        }
        return dp[len - 1];
    }
}
