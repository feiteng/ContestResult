import java.util.*;


class Solution {
    public int maxResult(int[] A, int k) {
        int n = A.length;

        // let dp[i] be the maximum score that can be obtained to reach the end of the array starting from A[i]
        int[] dp = new int[n];
        dp[n-1] = A[n-1];

        // We want to maintain the maximum value of dp[i+1]...dp[i+k]
        // This is a fixed size sliding window maximum
        // This can be done using a deque
        Deque<Integer> maximums = new LinkedList<>();
        maximums.add(dp[n-1]);

        for (int i = n - 2; i >= 0; i--) {
            if (i + k + 1 < n) {
                if (maximums.peekFirst() == dp[i+k+1]) {
                    maximums.pollFirst();
                }
            
}
            dp[i] = A[i] + maximums.peekFirst();
            while (!maximums.isEmpty() && maximums.peekLast() < dp[i]) {
                maximums.pollLast();
            }
            maximums.addLast(dp[i]);
        }

        return dp[0];

    }
}
