class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Deque<Integer> dq = new LinkedList<>();
        dq.addLast(0);
        dp[0] = nums[0];
        
for (int i = 1; i < n; i++) {
        //     System.out.println(Arrays.toString(dp));
        //     System.out.println(dq);
            while (!dq.isEmpty() && i - dq.getFirst() > k) dq.removeFirst();
            // System.out.println(dq);
            dp[i] = nums[i] + dp[
dq.getFirst()];
            while (!dq.isEmpty() && dp[dq.getLast()] <= dp[i]) dq.removeLast();
            dq.add(i);
        }
        return dp[n - 1];
    }
}
