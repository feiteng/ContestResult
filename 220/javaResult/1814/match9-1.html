class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        LinkedList<Integer> window = new LinkedList<Integer>();
        window.addFirst(0);
        for (int i = 1; i < n; i++) {
            while ((window.size() > 0) && (window.peekFirst() < i - k)) window.pollFirst();
            
dp[i] = dp[window.peekFirst()] + nums[i];
            while ((window.size() > 0) && (dp[i] >= dp[window.peekLast()])) window.pollLast();
            window.addLast(i);
        }
        /* for (int i = 0; i < n; i++) System.out.print(dp[i] + " ");
        System.out.println(); */
        return dp[n-1];
    }
}
