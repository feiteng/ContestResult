import java.util.TreeMap;


class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] scores = new int[n];
        scores[0] = nums[0];
        
        TreeMap<Integer, Integer> lastScores = new TreeMap<>();
        lastScores.put(scores[0], 1);
        for (int j = 1; j < n; j++) {
            if (j - k - 1 >= 0) {
                lastScores.put(scores[j - k - 1], lastScores.get(scores[j - k - 1]) - 1);
                if (lastScores.get(scores[j - k - 1]) == 0) {
                    lastScores.remove(scores[j - k - 1]);
                
}
            }
            scores[j] = nums[j] + lastScores.lastKey();
            lastScores.put(scores[j], lastScores.getOrDefault(scores[j], 0) + 1);
        }
        return scores[n-1];

    }
}
