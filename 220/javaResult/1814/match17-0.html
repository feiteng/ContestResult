class Solution {
    public int maxResult(int[] nums, int k) {
        int len = nums.length;
        
        int[] dp = new int[len + 1];
        dp[0] = nums[0];
        Deque<Integer> d = new LinkedList<>();
        d.addLast(0);
        for (int i = 1; i < len; ++i) {
            int cur = nums[i];
            dp[i] = cur + dp[d.peekFirst()];
            while (!d.isEmpty() && d.peekFirst() <= i - k) d.removeFirst();
            while (!d.isEmpty() && dp[d.peekLast()] < dp[i]) d.removeLast();
            d.addLast(i);
        }
        return dp[len - 1];

    }
}
