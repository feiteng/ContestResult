class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        
        //dp[i] = dp[i-m] get max
        int[] dp = new int[n];
        
        Arrays.fill(dp, Integer.MIN_VALUE);
        
        dp[0] = nums[0];
        for (int i = 1; i < n; i++) {
            
            for (int j = i-1; j >= Math.max(0, i-k); j--) {
                if (dp[j] == Integer.MIN_VALUE) {
                    continue;
                }
                if (nums[j] > 0) {
                    dp[i] = Math.max(dp[i], dp[j] + nums[i]);
                    
break;
                } else {
                    dp[i] = Math.max(dp[i], dp[j] + nums[i]);
                }
            }
            
            // for (int j = Math.max(0, i - k); j <= i-1; j++) {
            //     if (dp[j] == Integer.MIN_VALUE) {
            //         continue;
            //     }
            //     dp[i] = Math.max(dp[i], dp[j] + nums[i]);
            // }
        }
        
        return dp[n-1];
    }
}
