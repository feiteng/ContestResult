class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Deque<Integer> q= new LinkedList<>();
        for(int i=0; i<n;++i){
            if(!q.isEmpty()&& q.peekFirst()<i-k) q.pollFirst();
            dp[i]=nums[i]+(q.isEmpty()?0 : dp[q.peekFirst()]);
            while(!q.isEmpty()&& dp[q.peekLast()]<=dp[i]) q.pollLast();
            
            q.offerLast(i);
            
        }
        return dp[n-1];
    }
}
