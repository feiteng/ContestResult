class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        Deque<Integer> que = new LinkedList<>();
        que.add(0);
        for(int i=1; i<n; i++){
            while(que.size()>0&&que.peek()+k<i) que.poll();
            
dp[i] = dp[que.peek()]+nums[i];
            while(que.size()>0&&dp[que.peekLast()]<=dp[i]) que.pollLast();
            que.add(i);
        }
        return dp[n-1];
    }
}
