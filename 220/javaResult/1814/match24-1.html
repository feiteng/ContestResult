class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        LinkedList<Integer> ll = new LinkedList();
        dp[n-1]=nums[n-1];
        ll.add(n-1);
        for(int i=n-2;i>=0;--i){
            if(ll.peek()-i>k)ll.poll();
            
dp[i] = nums[i] + dp[ll.peek()];
            while(!ll.isEmpty() && dp[i]>=dp[ll.peekLast()])ll.pollLast();
            ll.add(i);
        }
        return dp[0];
    }
}
