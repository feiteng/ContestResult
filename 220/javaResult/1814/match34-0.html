class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Arrays.fill(dp, Integer.MIN_VALUE);
        dp[n - 1] = nums[n - 1];
        ArrayDeque<int[]> q = new ArrayDeque<>();
        q.offer(new int[]{nums.length - 1, nums[n - 1]});
        for (int i = n - 2; i >= 0; i --) {
            while (!q.isEmpty() && q.peek()[0] > i + k) {
                q.poll();
            }
            dp[i] = nums[i] + q.peek()[1];
            while (!q.isEmpty() && dp[i] >= q.peekLast()[1]) {
                q.pollLast();
            }
            q.offer(new int[]{i, dp[i]});
        }
        return dp[0];
    }
}
