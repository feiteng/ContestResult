class Solution {
    public int maxResult(int[] nums, int k) {
        int sum = nums[0];
        int[] dp = new int[nums.length];
        Arrays.fill(dp, -1000000000);
        dp[0] = nums[0];
        for (int i = 1; i < dp.length; i++) {
            for (int j = 1; j <= k && i - j >= 0; j++) {
                if (nums[i - j] >= 0) {
                    dp[i] = dp[i - j] + nums[i];
                    break;
                } else {
                    dp[i] = Math.max(dp[i], dp[i - j] + nums[i]);
                }
            }
        }
        return dp[nums.length - 1];
    }
}
