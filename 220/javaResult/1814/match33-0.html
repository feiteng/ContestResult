class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[n-1] = nums[n-1];
        Deque<Integer> deque = new LinkedList<>();
        deque.add(n-1);
        for (int i = n-2; i >= 0; i--) {
            while (deque.peekLast() > i + k) deque.pollLast();
            dp[i] = nums[i] + dp[deque.peekLast()];
            if (i >= n - k - 1) dp[i] = Math.max(dp[i], nums[i] + nums[n-1]);
            while (!deque.isEmpty() && dp[deque.peekFirst()] <= dp[i]) deque.pollFirst();
            deque.addFirst(i);
        }
        return dp[0];
    }
}
