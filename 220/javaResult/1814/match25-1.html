class Solution {
    public int maxResult(int[] nums, int k) {
        int l = nums.length;
        int[] dp = new int[l];
        Arrays.fill(dp,Integer.MIN_VALUE);
        ArrayDeque<Integer> deque = new ArrayDeque<Integer>();
        dp[0] = nums[0];
        deque.add(0);
        
for(int i = 1;i < l; i ++){
            while(!deque.isEmpty() && i - deque.peekFirst() > k){
                deque.pollFirst();
            }
            int idx = deque.peekFirst();
            int v = dp[idx] + nums[i];
            dp[i] = v;
            
while(!deque.isEmpty() && dp[deque.peekLast()] < v){
                deque.pollLast();
            }
            deque.addLast(i);
        }

        return dp[l-1];
    }
}
