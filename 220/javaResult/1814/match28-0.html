class Solution {
    public int maxResult(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        
        LinkedList<Integer> l = new LinkedList<>();
        l.add(0);
        
        for (int i = 1; i < n; i++) {
            if (i - k > l.peekFirst()) l.removeFirst();
            dp[i] = nums[i] + dp[l.peekFirst()];
            while (!l.isEmpty() && dp[l.peekLast()] <= dp[i]) l.removeLast();
            l.addLast(i);
        }
        
        /*for (int i = 1; i < n; i++) {
            dp[i] = nums[i] + dp[i - 1];
            for (int j = Math.max(0, i - k); j < i; j++) {
                dp[i] = Math.max(dp[i], dp[j] + nums[i]);
            }
        }*/
        
        return dp[n - 1];
    }
}
