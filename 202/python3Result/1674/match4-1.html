class Solution:
    def minOperations(self, n: int) -> int:
        s = 0
        for i in range(n):
            s += (2*i + 1)
        each = s//n
        res = 0
        for i in range(n):
            if (2*i + 1) < each:
                res += (each - 2*i - 1)
        return res
