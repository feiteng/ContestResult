class Solution:
    def minOperations(self, n: int) -> int:
        arr_sum = 0
        for i in range(n):
            arr_sum += 2*i+1
        arr_sum /= n
        ops = 0
        for i in range(n):
            if 2*i+1 < arr_sum:
                ops += arr_sum - 2*i - 1
        return int(ops)
