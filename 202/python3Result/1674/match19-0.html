class Solution:
    def minOperations(self, n: int) -> int:
        res = 0
        if n % 2:
            mid = (n + 1) // 2
            for i in range(1, mid):
                res += 2 * mid - 2 * i
            return res    
        mid1, mid2 = n // 2, n // 2 + 1 
        mid = mid1 + mid2 + 1
        for i in range(1, mid1 + 1):
            res += mid - 2 * i - 1
        return res
