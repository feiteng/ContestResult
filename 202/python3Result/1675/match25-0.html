class Solution:

    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()

        def check(dist):
            num = 1
            cur = 0
            for i in range(1, len(position)):
                cur += position[i] - position[i-1]
                if cur >= dist:
                    num += 1
                    cur = 0
                if num >= m:
                    return True
            return False


        lo, hi = 0, position[-1] - position[0]
        while lo < hi:
            mid = (lo + hi) // 2 + 1
            if check(mid):
                lo = mid
            else:
                hi = mid - 1
        return lo
