class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def valid(ball, f):
            pre = -f
            for p in position:
                if p - pre >= f:
                    ball -= 1
                    pre = p
            return ball <= 0
        l = 1
        r = position[-1] - position[0]
        while l <= r:
            mid = (l + r) // 2
            if valid(m, mid):
                l = mid + 1
            else:
                r = mid - 1
        return r
