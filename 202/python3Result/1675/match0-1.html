class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def tryDist(d, m):
            prevPos = None
            for pos in position:
                if prevPos is None or pos-prevPos>=d:
                    m-=1
                    prevPos = pos
            return m<=0
        
        left, right = 1, position
[-1] #left is ??, right is impossible
        while left<right:
            # print(left, right)
            mid = (left+right)//2
            if tryDist(mid, m):
                left = mid+1
            else:
                right = mid
        return right-1
