class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        
        left, right = 1, position[-1] - position[0]
        
        
        while left < right - 1:
            mid = left + (right - left) // 2
            if self.valid(position, m, mid):
                left = mid
            else:
                right = mid - 1
        if self.valid(position, m, right):
            return right
        return left
        
    def valid(self, position, m, force):
        # 把第一个球放在position[0]
        count = 1
        last = 0
        curr = 1
        while curr < len(position):
            if position[curr] - position[last] >= force:
                count += 1
                last = curr
                if count == m:
                    return True
            curr += 1
        return False
