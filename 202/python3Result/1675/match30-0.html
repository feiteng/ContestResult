class Solution:
    def maxDistance(self, p: List[int], m: int) -> int:
        p = sorted(p)
        i, j = 1, p[-1]
        def isBad(mid):
            cnt = 0
            tmp = -inf
            for e in p:
                if e - tmp >= mid:
                    cnt += 1
                    tmp = e
            return cnt < m
        while i < j:
            mid = (i+j)//2
            #print(mid, isBad(mid))
            if isBad(mid):
                j = mid
            else:
                i = mid +1
        return i-1
