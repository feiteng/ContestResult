class Solution:
    def maxDistance(self, p: List[int], m: int) -> int:
        p.sort()
        d=[]
        for i in range(1,len(p)):
            d.append(p[i]-p[0])
        # print(d)
        left,right=0,d[-1]
        def check(n):
            last=0
            num=0
            for i in d:
                if i-last>=n:
                    num+=1
                    last=
i
            return num>=m-1
        while left<right:
            mid=(left+right+1)//2
            if check(mid):
                left=mid
            else:
                right=mid-1
            # print(left,right,mid)
        return left
