class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def is_good(x):
            prev = -float('inf')
            remain = m
            for p in position:
                if p - prev >= x:
                    # place here.
                    remain -= 1
                    if remain == 0:
                        return True
                    prev = p
            return False
        
        lo, hi = 1, position[-1] - position[0]
        
        while lo <= hi:
            mid = (lo + hi) // 2
            if is_good(mid):
                # try to better
                lo = mid + 1
            else:
                hi = mid - 1
                
        return hi
