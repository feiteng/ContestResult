class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position = sorted(position)
        
        l, r = 0, position[-1] - position[0] + 1
        while r - l > 1:
            mid = (l+r)//2
            pre_pos = 0
            cnt = 1
            for i in range(1, len(position)):
                if position[i] - position[pre_pos] >= mid:
                    cnt += 1
                    pre_pos = i
            if cnt >= m:
                l = mid
            else:
                r = mid
        return l
