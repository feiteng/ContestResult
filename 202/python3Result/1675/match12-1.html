class Solution:
    def maxDistance(self, p: List[int], m: int) -> int:
        p.sort()
        l, r = 1, p
[-1]+1
        while l < r - 1:
            middle = (l + r)//2
            
            count = 1
            last = p[0]
            for b in p[1:]:
                if b >= last + middle:
                    count += 1
                    last = b
            
            if count >= m:
                l = middle
            else:
                r = middle
        return l
