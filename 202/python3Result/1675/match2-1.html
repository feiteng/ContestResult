class Solution:
    def maxDistance(self, pos: List[int], m: int) -> int:
        pos.sort()
        l = 0
        r = pos[-1]
        
        def check(mid):
            cur = 1
            pre = pos[0]
            for p in pos[1:]:
                if p - pre >= mid:
                    cur += 1
                    pre =
 p
            return cur >= m
        
        while l < r:
            mid = (l + r + 1) // 2 
            if check(mid):
                l = mid
            else:
                r = mid - 1
        
        return l
