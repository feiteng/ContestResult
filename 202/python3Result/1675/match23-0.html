class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        l = 1
        r = (position[-1] - position[0]) // (m - 1)
        
        while l < r:
            mid = (l + r + 1) // 2
            if self.isPossible(position, m, mid):
                l = mid
            else:
                r = mid - 1
        return l
        
    def isPossible(self, position, m, min_dist):
        last_pos = position[0]
        n_placed = 1
        for pos in position[1:]:
            if pos - last_pos >= min_dist:
                last_pos = pos
                n_placed += 1
                if n_placed == m:
                    return True
        return False
