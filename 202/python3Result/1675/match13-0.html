class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        # 用二分法做，先定一个最小距离，然后遍历这些球，每当距离>=最小距离，取出，如果cnt>m 则可增大距离
        position.sort()
        l = 0
        r = 2*10**9
        while l<r:
            mid = (l+r+1)//2
            cnt = 1
            last = position[0]
            for i in position[1:]:
                if i-last>=mid:
                    cnt+=1
                    last = i
            # 如果小于m，则区间太大，要缩小
            if cnt<m:
                r = mid-1
            else:
                l = mid
        return l
