class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        
        def yes(dist):
            nonlocal position, m
            
            last = -float('INF')
            cnt = 0
            for pos in position:
                if pos - last >= dist:
                    cnt += 1
                    if cnt >= m:
                        return True
                    last = pos
            return False         
            
        l, r = 1, max(position)-min(position)
        while l < r:
            mid = (l+r+1)//2
            if yes(mid):
                l = mid
            else:    
                r = mid-1
        return l
