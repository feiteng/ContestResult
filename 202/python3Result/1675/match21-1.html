inf = float('inf')

class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        p = position
        p.sort()
        
        def ok(x):
            last = -inf
            a = 0
            for y in p:
                
if y - x >= last:
                    a += 1
                    last = y
                if a == m:
                    return True
            return False
        
        a, b = 1, p[-1] - p[0] + 1
        while a + 1 < b:
            c = (a + b) >> 1
            if ok(c):
                a = c
            else:
                b = c
        return a
