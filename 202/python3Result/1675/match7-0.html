class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        def chk(mid):
            cnt = 1
            prev = 0
            for i in range(1, len(position)):
                if position[i]-position[prev] >= mid:
                    cnt += 1
                    prev = i
            return cnt >= m
                
        position.sort()
        L,H = 1,position[-1]-position[0]+1
        while L<H:
            mid = (L+H+1)//2
            if chk(mid): L = mid
            else: H = mid-1
        return L
