class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        l, r = 0, max(position)-min(position)
        while l<r:
            mid = (l+r+1) // 2
            cur = position[0]
            cnt = 1
            for pos in position:
                if pos-cur >= mid:
                    cnt += 1
                    cur = pos
            if cnt >= m:
                l = mid
            else:
                r = mid-1
        return l
