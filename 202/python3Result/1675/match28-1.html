class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def check(d):
            t,n=-1e18,0
            for x in position:
                if x-t>=d:
                    t,n
=x,n+1
            return n
        l,h=1,max(position)-min(position)
        while l<h:
            mid=(l+h)//2+1
            if check(mid)>=m:
                l=mid
            else:
                h=mid-1
        return l
