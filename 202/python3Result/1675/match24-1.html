class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        left, right = 1, 10**9 + 1
        position = sorted(position)
        
        
while left < right:
            mid = (left + right) // 2
            if self.check(position, m, mid):
                left = mid + 1
            else:
                right = mid
                
        return left - 1
                
    def check(self, position, m, force):
        lastIndex = position[0]
        m -= 1
        
        for pos in position[1:]:
            if pos - lastIndex >= force:
                m -= 1
                lastIndex = pos
        
        return m <= 0
