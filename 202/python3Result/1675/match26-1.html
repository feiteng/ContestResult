class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        n = len(position)
        start = 0
        end = position[n - 1] - position[0]
        
        def valid(m, mid):
            last = position[0]
            count 
= 1
            for p in position:
                if p - last >= mid:
                    last = p
                    count += 1
            return count >= m
        
        while start < end:
            mid = (start + end + 1) // 2
            if valid(m, mid):
                start = mid
            else:
                end = mid - 1
        return start
