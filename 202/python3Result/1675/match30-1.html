class Solution:
    def maxDistance(self, pos: List[int], m: int) -> int:
        pos.sort()
        n=len(pos)
        lo=min([pos[i+1]-pos[i] for i in range(n-1)])
        hi=pos[-1]-pos[0]
        if m==2:
            return hi
        if m==n:
            return lo
        
def can(d):
            cnt=0
            nxt=-1
            for p in pos:
                if p>=nxt:
                    cnt+=1
                    nxt=p+d
            return cnt>=m
        while lo<hi:
            mid=(lo+hi+1)//2
            if can(mid):
                lo=mid
            else:
                hi=mid-1
        return lo
