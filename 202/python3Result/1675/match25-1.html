class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def possible(d):
            placed, last = 0, -math.inf
            for p in position:
                if p - last >= d:
                    placed, last 
= placed + 1, p
                if placed >= m:
                    return True
            return False
        low, high = 1, position[-1] - position[0] + 1
        while low < high:
            mid = (low + high) // 2
            if possible(mid):
                low = mid + 1
            else:
                high = mid
        return low - 1
