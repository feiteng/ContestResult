class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position = sorted(position)
        l, r = 1, position[-1] - position[0]
        while l <= r:
            mid = (l + r) // 2
            last, cnt = 0, 1
            ok = False
            for i in range(1, len(position)):
                if position[i] - position[last] >= mid:
                    last = i
                    cnt += 1
                if cnt >= m:
                    ok = True
                    break
            if ok:
                l = mid + 1
            else:
                r = mid - 1

        return r
