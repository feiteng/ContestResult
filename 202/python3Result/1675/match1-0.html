class Solution:
    def maxDistance(self, positions: List[int], m: int) -> int:
        positions.sort()
        def count(mid):
            res = 1
            last = positions[0]
            for pos in positions[1:]:
                if pos >= last + mid:
                    res += 1
                    last = pos
            return res
            
        left, right = 0, positions[-1]
        while left <= right:
            mid = (left + right) // 2
            if count(mid) >= m:
                left = mid + 1
            else:
                right = mid - 1
        return left - 1
