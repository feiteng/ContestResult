class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()

        def isok(target: int)->bool:
            count = m-1
            pre = 0
            for i in range(1,len(position)):
                if position[i]-position[pre] >= target:
                    pre = i
                    count -= 1
                if count == 0:
                    break
            return count == 0
        # 二分
        l, r = 0, position[-1] - position[0]
        res = 0
        while l <= r:
            mid = (l+r)//2
            if isok(mid):
               res = mid
               l = mid+1
            else:
                r = mid-1
        return res
