class Solution:
    def maxDistance(self, a: List[int], m: int) -> int:
        a.sort()
        l, r = 1, a[-1] - a[0]
        
        def test(k):
            remain = m - 1
            last = a[0]
            for x in a[1:]:
                if x - last >= k:
                    remain -= 1
                    if remain == 0:
                        return True
                    last =
 x
            return False
        
        while l < r:
            k = (l + r) // 2 + 1
            if test(k):
                l = k
            else:
                r = k - 1
        return l
