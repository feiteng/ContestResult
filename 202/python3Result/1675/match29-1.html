class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position = sorted(position)
        lower = 0
        upper 
= position[-1] - position[0] + 1
        
        while lower + 1 < upper:
            mid = (lower + upper) // 2
            if self.works(position, m, mid):
                lower = mid
            else:
                upper = mid
        
return lower
    
    def works(self, position, m, force):
        prev = 0
        m -= 1
        i = 0
        while i < len(position) and m > 0:
            if position[i] - position[prev] >= force:
                m -= 1
                prev = i
            else:
                i += 1
        return m == 0
