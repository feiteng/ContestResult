class Solution:
    def maxDistance(self, nums: List[int], m: int) -> int:
        nums.sort()
        lb = 0
        ub = nums[-1] // (m - 1)
        
        def can_fit(dist):
            count = 1
            curr_pos = nums[0]
            for num in nums[1:]:
                if num - curr_pos < dist:
                    continue
                count += 1
                curr_pos = num
                if count == m:
                    return True
            return False
        
        while lb < ub:
            mid = (lb + ub + 1) // 2
            if can_fit(mid):
                lb = mid
            else:
                ub = mid - 1
                
        return lb
