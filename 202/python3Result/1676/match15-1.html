class Solution:
    def minDays(self, n: int) -> int:
        from functools import lru_cache
        @lru_cache(maxsize=None)
        
def _minDays(n):
            if n == 1:
                return 1
            cur_min = n
            if (n%2) == 0:            
                cur_min = min(cur_min, _minDays(n//2) + 1)
            if (n%3) == 0:
                cur_min = min(cur_min, _minDays(n//3) + 1)
            if (n%2) or (n%3):
                cur_min = min(cur_min, _minDays(n-1) + 1)

            return cur_min
        return _minDays(n)
