class Solution:
    def minDays(self, n: int) -> int:
        
        step = 0
        q, seen = [n], set([n])
        while q:
            nq = []
            for x in q:
                if x == 0:
                    return step
                if x % 3 == 0 and x // 3 not in seen:
                    nq.append(x // 3)
                    seen.add(x // 3)
                if x % 2 == 0 and x // 2 not in seen:
                    nq.append(x // 2)
                    seen.add(x // 2)
                if x - 1 not in seen:
                    nq.append(x - 1)
                    seen.add(x - 1)
            step += 1
            q = nq
            # print(q)
