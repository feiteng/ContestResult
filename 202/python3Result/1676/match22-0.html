class Solution:
    def minDays(self, n: int) -> int:
        cur = {n}
        visited = set()
        count = 0
        while len(cur)!=0:
            ncur = set()
            for c in cur:
                if c==0:
                    return count
                if c%3==0 and c//3 not in visited:
                    visited.add(c//3)
                    ncur.add(c//3)
                if c%2==0 and c//2 not in visited:
                    visited.add(c//2)
                    ncur.add(c//2)
                if c-1 not in visited:
                    visited.add(c-1)
                    ncur.add(c-1)
            cur = ncur
            count += 1
