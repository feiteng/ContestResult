"""
吃掉一个橘子。
如果剩余橘子数 n 能被 2 整除，那么你可以吃掉 n/2 个橘子。
如果剩余橘子数 n 能被 3 整除，那么你可以吃掉 2*(n/3) 个橘子。
"""
from functools import lru_cache


class Solution:
    def minDays(self, n: int) -> int:
        @lru_cache(None)
        def eat(k):
            if k == 1:
                return 1
            ans = float('inf')
            if k >= 3:
                ans = min(ans, k % 3 + 1 + eat(k//3))
            if k >= 2:
                ans = min(ans, k%2 + 1 + eat(k//2))
            return ans
        return eat(n)
