from collections import deque
class Solution:
    def minDays(self, n: int) -> int:
        
        visited = set()
        visited.add(n)
        bfs = deque([(0, n)])
        
        while bfs:
            step, num = bfs.popleft()
            if num == 0: return step
            if num - 1 not in visited:
                visited.add(num - 1)
                bfs.append((step + 1, num - 1))
            if not num % 2 and num // 2 not in visited:
                visited.add(num // 2)
                bfs.append((step + 1, num // 2))
            if not num % 3 and num // 3 not in visited:
                visited.add(num // 3)
                bfs.append((step + 1, num // 3))
