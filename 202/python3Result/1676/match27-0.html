class Solution:
    def minDays(self, n: int) -> int:
        from collections import deque
        ans = n 
        dq = deque()
        dq.append((n, 0))
        vis = set()
        while dq:
            node, dis = dq.popleft()
            if node in vis:
                continue
            vis.add(node)
            if node <= 0:
                # ans = min(ans, dis)
                # continue 
                return dis
            #print('{}, {}'.format(node, dis))
            if node % 3 == 0:
                dq.append(((node // 3), dis + 1))
            if node % 2 == 0:
                dq.append((node // 2, dis + 1))

            dq.append((node - 1, dis + 1))
        return ans
