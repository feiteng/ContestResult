class Solution:
    def minDays(self, n: int) -> int:
        sys.setrecursionlimit(10**7)
        @lru_cache(None)
        
def dfs(n,t):
            if n==0:
                return 0
            ans=inf
            if n%2==0:
                ans=min(ans,1+dfs(n//2,4))
            if n%3==0:
                ans=min(ans,1+dfs(n-2*(n//3),4))
            if t>0:
                ans=min(ans,1+dfs(n-1,t-1))
            return ans
        return dfs(n,4)
