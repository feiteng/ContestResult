class Solution:
    def minDays(self, n: int) -> int:
        a,d,s=[n],set([n]),1
        while 1 not in d:
            b,s=[],s+1
            for x in a:
                
if x-1 not in d:
                    d.add(x-1)
                    b.append(x-1)
                if not x%2 and x//2 not in d:
                    d.add(x//2)
                    b.append(x//2)
                if not x%3 and x//3 not in d:
                    d.add(x//3)
                    b.append(x//3)
            a=b
        return s
