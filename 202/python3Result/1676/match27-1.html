class Solution:
    def minDays(self, n):
        if n == 0:
            return 0

        seen = set()
        q = collections.deque([[n, 0]])
        
while True:
            p, step = q.popleft()
            if p in seen:
                continue
            seen.add(p)
            if p == 0:
                return step
            if p % 3 == 0:
                q.append([p//3, step+1])
            if p % 2 == 0:
                q.append([p//2, step+1])
            q.append([p-1, step+1])

        return 0
