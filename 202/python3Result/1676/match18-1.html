class Solution:
    def minDays(self, n: int) -> int:
        
        if n <= 0:
            return 0
        q = [n]
        vis = {n}
        step 
= 0
        while q:
            temp = []
            for x in q:
                if x == 0:
                    return step
                if x%2==0 and x//2 not in vis:
                    temp.append(x//2)
                    vis.add(x//2)
                if x%3==0 and x//3 not in vis:
                    temp.append(x//3)
                    vis.add(x//3)
                if x-1 not in vis:
                    temp.append(x-1)
                    vis.add(x-1)
            q=temp
            step+=1
        return 0
