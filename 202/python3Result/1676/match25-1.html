class Solution:
    def minDays(self, n: int) -> int:
        heap = [(0, n)]
        seen = set([n])
        while len(heap) > 0:
            cost, x 
= heapq.heappop(heap)
            if x == 0:
                return cost
            if x - 1 not in seen:
                seen.add(x - 1)
                heapq.heappush(heap, (cost + 1, x - 1))
            if x % 2 == 0 and x // 2 not in seen:
                seen.add(x // 2)
                heapq.heappush(heap, (cost + 1, x // 2))
            if x % 3 == 0 and x // 3 not in seen:
                seen.add(x // 3)
                heapq.heappush(heap, (cost + 1, x // 3))
