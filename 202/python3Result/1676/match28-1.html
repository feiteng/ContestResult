class Solution:
    @lru_cache(None)
    def minDays(self, n: int) -> int:
        if n == 1:
            return 1
        res = float("inf")
        
if n%2!=0 or n%3!=0:
            res = self.minDays(n-1)+1
        if n%2==0:
            res = min(res, self.minDays(n//2)+1)
        if n%3==0:
            res = min(res, self.minDays(n//3)+1)
        return res
