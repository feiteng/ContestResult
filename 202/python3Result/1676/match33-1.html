from functools import lru_cache
class Solution:
    def minDays(self, n: int) -> int:
        @lru_cache(None)
        def helper(n):
            if n <= 1: return n
            ans = helper(n - 1)
            if n % 2 == 0:
                ans = min(ans, helper(n // 2))
            if n % 3 == 0:
                ans = min(ans, helper(n - 2 * (n // 3)))
            return ans + 1
        @lru_cache(None)
        def helper1(n):
            if n <= 1: return n
            if n % 3 == 0:
                ans = 1 + helper1(n // 3)
                if n % 2 == 1:
                    ans = min(ans, 1 + helper1(n - 1))
                else:
                    ans = min(ans, 1 + helper1(n // 2))
                return ans
            if n % 2 == 0:
                return min(1 + helper1(n // 2), (n % 3) + helper1(n - (n % 3)))
            return 1 + helper1(n - 1)
        # for k in range(1000):
        #     if helper(k) != helper1(k):
        #         print(k, helper(k), helper1(k))
        return helper1(n)
