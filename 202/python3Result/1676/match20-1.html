class Solution:
    def minDays(self, n: int) -> int:
        # d = {1:1,2:2,3:2}
        # for i in range(4,10**5):
        #     d[i] = 1+d[i-1]
        #     if i%2 == 0:
        #         d[i] = min(d[i],1+d[i//2])
        #     if i%3 == 0:
        #         d[i] = min(d[i],1+d[i//3])
        
        d = {1:1,2:2,3:2}
        def solve(n):
            
if n in d:
                return d[n]
            
            ret = n
            if n%2 == 0:
                ret = min(ret,1+solve(n//2))
            else:
                ret = min(ret,2+solve((n-1)//2))
            if n%3 == 0:
                ret = min(ret,1+solve(n//3))
            elif n%3 == 1:
                ret = min(ret,2+solve((n-1)//3))
            else:
                ret = min(ret,3+solve((n-2)//3))
            d[n] = ret
            return ret
        return solve(n)
