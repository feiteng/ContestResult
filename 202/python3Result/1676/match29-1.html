from functools import lru_cache
class Solution:
    def minDays(self, n: int) -> int:
        
        @lru_cache(maxsize=None)
        def f(n):
            
if n == 0: return 0
            if n == 1: return 1

            return min(f(n // 3) + n % 3 + 1, f(n // 2) + n % 2 + 1)
        
        return f(n)
