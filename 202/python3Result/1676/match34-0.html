class Solution:
    def minDays(self, n: int) -> int:
        seen = set()
        seen.add(n)
        layer = [n]
        count = 0
        
        while True:
            count += 1
            new_layer = []
            for x in layer:
                if x == 1:
                    return count
                if x - 1 not in seen:
                    seen.add(x - 1)
                    new_layer.append(x - 1)
                if x % 2 == 0 and (x // 2) not in seen:
                    seen.add(x // 2)
                    new_layer.append(x // 2)
                if x % 3 == 0 and (x // 3) not in seen:
                    seen.add(x // 3 )
                    new_layer.append(x // 3 )
            layer = new_layer

        return count
