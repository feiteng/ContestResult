import functools
class Solution:
    def minDays(self, n: int) -> int:  
        if n == 1:
            return 1

        visited = {n}
        layer = [n]
        ops = 1

        while True:
            next_layer = []

            for num in layer:
                if num == 1:
                    return ops

                if num - 1 not in visited:
                    next_layer.append(num - 1)
                    visited.add(num - 1)
                if not num % 2 and num // 2 not in visited:
                    next_layer.append(num // 2)
                    visited.add(num // 2)
                if not num % 3 and num // 3 not in visited:
                    next_layer.append(num // 3)
                    visited.add(num // 3)

            layer = next_layer
            ops += 1

        return 0
