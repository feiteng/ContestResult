class Solution:
    def minDays(self, n: int) -> int:
        @functools.lru_cache(None)
        def solve(n, cnt):
            
if n == 1:
                return 1
            res = float('inf')
            if n % 3 == 0:
                res = min(res, solve((n // 3), cnt) + 1)
            if n % 2 == 0:
                res = min(res, solve((n // 2), cnt) + 1)
            if cnt < 15:
                res = min(res, solve((n - 1), cnt + 1) + 1)
            return res
        return solve(n, 0)
