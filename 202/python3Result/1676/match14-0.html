from functools import lru_cache
class Solution:
    def minDays(self, n: int) -> int:
        @lru_cache(None)
        def helper(m, tries=2):
            if m <= 1:
                return m
            ret = float('inf')
            if tries >= 0:
                ret = min(ret, helper(m - 1, tries - 1))
            if m % 3 == 0:
                ret = min(ret, helper(m // 3))
            if m % 2 == 0:
                ret = min(ret, helper(m //2))
                
            return 1 + ret
        
        return helper(n, 2)
