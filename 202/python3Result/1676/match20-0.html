class Solution:
    def minDays(self, n: int) -> int:
        inf = 10 ** 16
        memo = {}

        def dfs(a):
            if a == 1: return 1
            if a == 2: return 2
            if a == 3: return 2
            if a in memo: return memo[a]
            res = inf
            if a % 2 == 0:
                res = min(res, dfs(a // 2) + 1)
            else:
                res = min(res, dfs((a - 1) // 2) + 2)
            if a % 3 == 0:
                res = min(res, dfs(a // 3) + 1)
            elif a % 3 == 1:
                res = min(res, dfs((a - 1) // 3) + 2)
            else:
                res = min(res, dfs((a - 2) // 3) + 3)
            memo[a] = res
            return res

        return dfs(n)
