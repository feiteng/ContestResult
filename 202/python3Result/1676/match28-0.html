class Solution:
    @functools.lru_cache(None)
    def minDays(self, n: int, m1 = 0) -> int:
        if n == 1:
            return 1
        arr = math.inf
        if m1 <= 3:
            arr = self.minDays(n-1, m1 + 1)
        if n % 2 == 0:
            arr = min(arr, self.minDays(n // 2))
        if n % 3 == 0:
            arr = min(arr, self.minDays(n // 3))
        return arr + 1
