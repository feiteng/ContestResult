class Solution:
    def minDays(self, n: int) -> int:
        @lru_cache(None)
        def check(m:int):
            
if m == 0:
                return 0
            if m == 1:
                return 1
            return min(check(m//2) + m%2 + 1,check(m//3) + m%3 + 1)
        return check(n)
