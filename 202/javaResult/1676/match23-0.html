class Solution {

  private Map<Integer, Integer> map = new HashMap<>();

  public int minDays(int n) {
    map = new HashMap<>();
    return dfs(n);
  }

  private int dfs(int n) {
    if (n == 1) {
      return 1;
    }
    if (map.containsKey(n)) {
      return map.get(n);
    }
    int res = Integer.MAX_VALUE;
    if (n % 2 == 0) {
      res = Math.min(res, dfs(n >> 1) + 1);
    }
    if (n % 3 == 0) {
      res = Math.min(res, dfs(n / 3) + 1);
    }
    if (n % 2 != 0 || n % 3 != 0) {
      res = Math.min(res, dfs(n - 1) + 1);
    }
    map.put(n, res);
    return res;
  }
}
