class Solution {
    public int minDays(int n) {
        Queue<Integer> q = new LinkedList<>();
        HashSet<Integer> nums = new HashSet<>();
        q.offer(n);
        nums.add(n);
        int res = 0;
        while (true) 
{
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int m = q.poll();
                if (m == 0) {
                    return res;
                }
                if (!nums.contains(m - 1)) {
                    q.offer(m - 1);
                    nums.add(m - 1);
                }
                if (m % 2 == 0 && !nums.contains(m / 2)) {
                    q.offer(m / 2);
                    nums.add(m / 2);
                }
                if (m % 3 == 0 && !nums.contains(m / 3)) {
                    q.offer(m / 3);
                    nums.add(m / 3);
                }
                
            }
            res++;
        }
    }
}
