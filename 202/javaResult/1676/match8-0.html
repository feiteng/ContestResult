class Solution {
    public int minDays(int n) {
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(n);
        visited.add(n);
        int ret = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i += 1) {
                int cur = queue.poll();
                if (cur == 0) {
                    return ret;
                }
                if (!visited.contains(cur - 1)) {
                    queue.offer(cur - 1);
                    visited.add(cur - 1);
                }
                if (cur % 2 == 0 && !visited.contains(cur / 2)) {
                    queue.offer(cur / 2);
                    visited.add(cur / 2);
                }
                if (cur % 3 == 0 && !visited.contains(cur - cur / 3 * 2)) {
                    queue.offer(cur - cur / 3 * 2);
                    visited.add(cur - cur / 3 * 2);
                }
            }
            ret += 1;
        }
        return -1;
