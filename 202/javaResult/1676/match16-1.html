class Solution {
    public int minDays(int n) {
        Set<Integer> vis = new HashSet<>();
        vis.add(n);
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(n);
        int dist = 0;
        while (!queue.isEmpty()) {
            dist++;
            int size = queue.size();
            while (size-- > 0) 
{
                int front = queue.poll();
                if (front == 1) {
                    return dist;
                }
                if (!vis.contains(front-1)) {
                    vis.add(front-1);
                    queue.offer(front-1);
                }
                if (front%2 == 0 && !vis.contains(front/2)) {
                    vis.add(front/2);
                    queue.offer(front/2);
                }
                if (front%3 == 0 && !vis.contains(front/3)) {
                    vis.add(front/3);
                    queue.offer(front/3);
                }
                
            }
        }
        return dist;
    }
}
