class Solution {
    Map<Integer, Integer> hm;
    public int minDays(int n) {
        hm = new HashMap<>();
        return rec(n);
    }

    int rec(int n) {
        if (n <= 1) {
            return n;
        }
        if (hm.containsKey(n)) {
            return hm.get(n);
        }

        int res = Integer.MAX_VALUE;
        if (n % 3 == 0) {
            res = Math.min(res, rec(n / 3) + 1);
        } else 
{
            res = Math.min(res, rec(n - 1) + 1);
        }
        if (n % 2 == 0) {
            res = Math.min(res, rec(n / 2) + 1);
        } else {
            res = Math.min(res, rec(n - 1) + 1);
        }

        hm.put(n, res);
        return res;
    }
}
