class Solution {
    public int minDays(int n) {
        Queue<Integer> q = new LinkedList<Integer>();
        HashSet<Integer> visited = new HashSet<Integer>();
        int result=0;
        
        q.offer(n);
        
        while(!q.isEmpty()) {
            int m=q.size();
            
            for(int i=0; i<m; i++) {
                int rem=q.poll();
                
                if(rem==0) {
                    return result;
                }
                
                if(!visited.contains(rem-1)) {
                    q.offer(rem-1);
                    visited.add(rem-1);
                }
                
                if((rem%2 == 0) && !visited.contains(rem/2)) {
                    q.offer(rem/2);
                    visited.add(rem/2);
                }
                
                if((rem%3 == 0) && !visited.contains(rem - (2*(rem/3)))) {
                    q.offer(rem - (2*(rem/3)));
                    visited.add(rem - (2*(rem/3)));
                }
            }
            
            result++;
        }
        
        return result;
    }
}
