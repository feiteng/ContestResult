class Solution {
    public int minDays(int n) {
        Set<Integer> set = new HashSet<>();
        Queue<Integer> q = new LinkedList<>();
        set.add(n);
        q.add(n);
        int ans = 0;
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int t = q.remove();
                if (t == 0) return ans;
                if (!set.contains(t - 1)) {
                    q.add(t - 1);
                    set.add(t - 1);
                }
                if (t % 2 == 0 && !set.contains(t / 2)) {
                    q.add(t / 2);
                    set.add(t / 2);
                }
                if (t % 3 == 0 && !set.contains(t / 3)) {
                    q.add(t / 3);
                    set.add(t / 3);
                }
            }
            ans++;
        }
        return -1;
    }
}
