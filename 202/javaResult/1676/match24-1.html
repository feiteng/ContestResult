class Solution {
    public int minDays(int n) {
        
        ArrayDeque<Integer> ad = new ArrayDeque<>();
        HashSet<Integer> hm = new HashSet<Integer>();
        ad.add(n);
        int ret = 0;
        
        while(ad.size()!=0){
            int size = ad.size();
            
            ret++;
            for(int i=0;i<size;i++){
                int r = ad.pop();
                
                if(r%3==0&&!
hm.contains(r%3)){
                    ad.add(r/3);
                    hm.add(r/3);
                } 
                if(r%2==0&&!hm.contains(r%2)){
                    ad.add(r/2);
                    hm.add(r/2);
                }
                if(!hm.contains(r-1)){
                    ad.add(r-1);
                    hm.add(r-1);
                    if(r-1==0) return ret;
                
}
                

            }
        }
        
        return -1;
      
    }
        
}
