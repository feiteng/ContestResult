class Solution {
    public int minDays(int n) {
        Deque<Integer> deque=new ArrayDeque<>();
        Set<Integer> set=new HashSet<>();
        int ans=0;
        deque.addLast(n);
        while (!deque.isEmpty()){
            int len=deque.size();
            for(int i=0;i<len;i++){
                int cur=deque.removeFirst();
                if(cur==0){
                    return ans;
                }
                if(cur%3==0&&!set.contains(cur/3)){
                    deque.addLast(cur/3);
                    set.add(cur/3);
                }
                if(cur%2==0&&!set.contains(cur/2)){
                    deque.addLast(cur/2);
                    set.add(cur/2);
                }
                if(!set.contains(cur-1)){
                    deque.addLast(cur-1);
                    set.add(cur-1);
                }
            }
            ans++;
        }
        return ans;
    }
}
