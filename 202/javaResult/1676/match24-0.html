class Solution {
    public int minDays(int n) {
        Set<Integer> set = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(n);
        int step = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            step++;
            for (int i=0; i<size; i++) {
                int num = queue.poll();
                if (num == 1) {
                    return step;
                }
                if (num % 3 == 0 && !set.contains(num / 3)) {
                    queue.offer(num / 3);
                    set.add(num / 3);
                }
                if (num % 2 == 0 && !set.contains(num / 2)) {
                    queue.offer(num / 2);
                    set.add(num / 2);
                }
                if (!set.contains(num - 1)) {
                    queue.offer(num - 1);
                    set.add(num - 1);
                }
            }
        }
        return n;
    }
}
