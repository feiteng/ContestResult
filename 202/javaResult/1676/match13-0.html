class Solution {
    public int minDays(int n) {
        Map<Integer, Integer> memo = new HashMap<>();
        return minDaysHelper(n, memo);
    }

    private int minDaysHelper(int n, Map<Integer, Integer> memo) {
        if (n == 0) {
            return 0;
        }
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        int ans = n;
        if (n % 3 == 0) {
            ans = Math.min(ans, minDaysHelper(n / 3, memo) + 1);
        } else {
            ans = Math.min(ans, minDaysHelper(n - 1, memo) + 1);
        }
        if (n % 2 == 0) {
            ans = Math.min(ans, minDaysHelper(n / 2, memo) + 1);
        } else {
            ans = Math.min(ans, minDaysHelper(n - 1, memo) + 1);
        }
        memo.put(n, ans);
        return ans;
    }
}
