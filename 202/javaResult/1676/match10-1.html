class Solution {
    public int minDays(int n) {
        Queue<Integer> queue = new LinkedList();
        queue.offer(n);
        Set<Integer> used = new HashSet();
        
int step = 1;
        while (!queue.isEmpty()) {
            // System.out.println(queue);
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int node = queue.poll();
                if (node == 1) {
                    return step;
                }
                if (node % 3 == 0 && !used.contains(node / 3)) {
                    used.add(node / 3);
                    queue.offer(node / 3);
                }
                if (node % 2 == 0 && !used.contains(node / 2)) {
                    used.add(node / 2);
                    queue.offer(node / 2);
                }
                if (!used.contains(node - 1)) {
                    used.add(node - 1);
                    queue.offer(node - 1);
                }
            }
            step++;
        }
        return -1;
    }
}
