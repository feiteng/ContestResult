class Solution {
    public int minDays(int n) {
        Deque<Integer> q = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        q.offer(n);
        set.add(n);
        int level = 0;
        while (!q.isEmpty()) 
{
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int j = q.poll();
                if (j == 0) return level
;
                if (!set.contains(j - 1)) {
                    q.offer(j - 1);
                    set.add(j - 1);
                }
                if (j % 2 == 0 && !set.contains(j / 2)) {
                    q.offer(j / 2);
                    set.add(j / 2);
                }
                if (j % 3 == 0 && !set.contains(j / 3)) {
                    q.offer(j / 3);
                    set.add(j / 3);
                }
            }
            level++;
        } 
        return -1;
    }
}
