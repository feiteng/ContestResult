        return dfs(n, new HashMap<>());

    private static int dfs(int n, Map<Integer,Integer> dp) {
        if (n == 0) {
            return -1;
        }
        if (dp.containsKey(n)) {
            return dp.get(n);
        }
        int res = 1 + Math.min(n % 2 + dfs(n / 2, dp), n % 3 + dfs(n / 3, dp));
        dp.put(n, res);
        return res;
    }
