class Solution {
    public int minDays(int n) {
        HashMap<Integer, Integer> memo = new HashMap<>();
        return help(n, memo);
    }
    
    public int help(int rem, HashMap<Integer, Integer> memo) {
        if (rem <= 1) {
            return rem;
        }
        if (memo.containsKey(rem)) {
            return memo.get(rem);
        }
        int ret = Integer.MAX_VALUE;
        if (rem % 2 == 0) {
            ret = Math.min(ret, 1 + help(rem/2, memo));
        } else {
            ret = Math.min(ret, 2 + help((rem-1)/2, memo));
        }
        if (rem % 3 == 0) {
            ret = Math.min(ret, 1 + help(rem/3, memo));
        } else {
            ret = Math.min(ret, 1 + (rem % 3) + help(rem/3, memo));
        }
        memo.put(rem, ret);
        return ret;
    }
}
