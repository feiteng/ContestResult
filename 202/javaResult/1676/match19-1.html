class Solution {
    public int minDays(int n) {
        Set<Integer> visited = new HashSet();
        Queue<Integer> queue = new LinkedList();
        visited.add(n);
        queue.add(n);
        int steps = 0;
        while (!queue.isEmpty()) 
{
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int curr = queue.poll();
                if (curr == 0) {
                    return steps;
                }
                if (curr % 2 == 0 && !visited.contains(curr / 2)) {
                    visited.add(curr / 2);
                    queue.add(curr / 2);
                }
                if (curr % 3 == 0 && !visited.contains(curr / 3)) {
                    visited.add(curr / 3);
                    queue.add(curr / 3);
                }
                if (!visited.contains(curr - 1)) {
                    visited.add(curr - 1);
                    queue.add(curr - 1);
                }
            }
            steps++;
        }
        return -1;
    }
}
