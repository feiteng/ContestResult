class Solution {
    public int minDays(int n) {
        HashSet<Integer> set = new HashSet<>();
        Queue<Integer> q = new LinkedList<>();
        set.add(n);
        q.offer(n);
        int res = 0;
        while(!q.isEmpty()) 
{
            int size = q.size();
            for(int i = 0; i < size; i++) {
                int cur = q.remove();
                if(cur == 0) {
                    return res;
                }else{
                    
if(cur % 2 == 0 && !set.contains(cur/2)) {
                        q.offer(cur/2);
                        set.add(cur/2);
                    }
                    if(cur % 3 == 0 && !set.contains(cur - 2*(cur/3))) {
                        q.offer(cur - 2*(cur/3));
                        set.add(cur - 2*(cur/3));
                    }
                    if(!set.contains(cur -1)) {
                        q.offer(cur - 1);
                        set.add(cur - 1);
                    }
                }
            }
            res++;
        }
        return res;
    }
}
