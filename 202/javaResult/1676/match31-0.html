	class Solution {
	    public int minDays(int n) {
			Map<Integer, Integer> ds = new HashMap<>();
			ds.put(n, 0);
			Queue<Integer> q = new ArrayDeque<>();
			q.add(n);
			while(true){
				int cur = q.poll();
				int d = ds.get(cur);
				if(cur == 0){
					return d;
				}
				if(!ds.containsKey(cur-1)){
					ds.put(cur-1, d+1);
					q.add(cur-1);
				}
				if(cur % 2 == 0 && !ds.containsKey(cur/2)){
					ds.put(cur/2, d+1);
					q.add(cur/2);
				}
				if(cur % 3 == 0 && !ds.containsKey(cur/3)){
					ds.put(cur/3, d+1);
					q.add(cur/3);
				}
			}
	    }
	}
