	class S
olution {
	    public int maxDistance(int[] a, int m) {
	    	Arrays.sort(a);
	        int low = 0, high = 1000000007;
	        while(high - low > 1){
	        	int h = high+low>>1;
	        	if(ok(a, h, m)){
	        		low = h;
	        	}else{
	        		high = h;
	        	}
	        }
	        return low;
	    }
	    
	    boolean ok(int[] a, int h, int m)
	    {

	    	int r = -1;
	    	int u = 0;
	    	for(int v : a){
	    		if(r <= v){
	    			r = v + h;
	    			u++;
	    		}
	    	}
	    	return u >= m;
	    }
	}
