class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int n = position.length;
        int low = 1;
        int high = position[n - 1] - position[0];
        while (low < high) {
            int mid = (low + high + 1) / 2;
            if (canFit(position, mid, m - 1)) {
                low = mid;
            } else {
                high = mid - 1;
            }
        }
        return low;
    }
    
    private boolean canFit(int[] p, int d, int m) {
        int prev = p[0];
        
for (int i=1; i<p.length; i++) {
            if (p[i] - prev >= d) {
                prev = p[i];
                m--;
                if (m == 0) {
                    return true;
                }
            }
        }
        return false;
    }
}
