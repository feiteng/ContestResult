class Solution {
    public int maxDistance(int[] position, int m) {
        int n = position.length;
        Arrays.sort(position);
        int lo = 1; 
        int hi = position[n - 1];
        while (lo < hi) {
            int mid = lo + (hi - lo + 1) / 2;
            if (canSeparate(position, mid, m)) {
                lo = mid;
            }
            else {
                hi = mid - 1;
            }
        }
        return lo;
    }
    
    private boolean canSeparate(int[] position, int force, int m) {
        m--;
        int last = position[0];
        int i = 0;
        for (; i < position.length; i++) {
            if (position[i] - last >= force) {
                m--;
                last = position[i];
                if (m == 0) {
                    return true;
                }
            }
        }
        return false;
    }
}
