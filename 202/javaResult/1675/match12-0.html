class Solution {
    public int maxDistance(int[] pos, int m) {
        Arrays.sort(pos);
        
        int out = 0;
        
        int low = 1;
        int high = Integer.MAX_VALUE;
        
        while(low < high) {
            int mid = low + (high - low) / 2;
            
            if(isPossible(pos, m, mid)) {
                out = Math.max(mid, out);
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        
        return out;
    }
    
    private boolean isPossible(int[] pos, int m, int dist) {
        int p = pos[0];
        
        int elems = 1;
        
        for(int i = 1; i < pos.length; i++) {
            if(pos[i] - p >= dist) {
                elems++;
                p = pos[i];
            }
        }
        
        return elems >= m;
    }
}
