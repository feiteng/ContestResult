class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int a = 1;
        int b = position[position.length -1 ] - position[0] + 1;
        while (a < b - 1) {
            int c = (int)((long) a + (long)b) / 2;
            if (isPos(position, m, c)) {
                a = c;
            } else {
                b = c;
            }
        }
        return a;
    }

    boolean isPos(int [] position, int m, int k) {
        int lastP = -1;
        int r = 0;
        for (int i = 0; i < position.length; i++) {
            if (position[i] - lastP >= k || i == 0) {
                r++;
                if (r >= m) return true;
                lastP = position[i];
            }
        }
        return r >= m;
    }
}
