class Solution {
    private boolean isValid(int[] position, int m, int distance) {
        int curr = position[0];
        int total = 1;
        for(int i=1;i<position.length; i++) {
            if(position[i]-curr >= distance) {
                total++;
                curr = position[i];
            }
        }
        return total >= m;
    }
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        
        int start = 0;
        int end = position[position.length-1];
        int result = 0;
        while(start<=end) {
            int mid = start + (end-start)/2;
            if(isValid(position, m, mid)) {
                result = mid;
                start = mid +1;
            } else {
                end = mid-1;
            }
        }
        return result;
    }
}
