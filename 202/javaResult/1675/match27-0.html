import java.util.Arrays;

class Solution {

    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int left = 1, right = 1000000000;
        while (left < right) {
            int mid = (left + right + 1) / 2;
            if (maxDistance(mid, position, m)) {
                left = mid;
            } else {
                right = mid - 1;
            }
        }
        return left;
    }

    private boolean maxDistance(int value, int[] position, int m) {
        int prev = position[0];
        for (int i : position) {
            if (i - prev >= value) {
                m--;
                prev = i;
            }
        }
        return m <= 1;
    }
}
