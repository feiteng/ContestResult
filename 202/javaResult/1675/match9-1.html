class Solution {
    boolean valid(int[] position, int dis, int m){
        int cnt = 1,last = 0;
        for (int i = 1; i < position.length; i++) {
            if(position[i] - position[last] >= dis){
                cnt++;
                last = i;
            }
        }
        return cnt >= m;
    }

    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int min = 1,max = position[position.length - 1] - position[0],last = min;
        while (min <= max){
            int mid = (min + max) / 2;
            if(valid(position,mid,m)){
                last = mid;
                min = mid + 1;
            }else{
                max = mid - 1;
            }
        }
        return last;
    }
}
