class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int len = position.length;
        int left = 0, right = position[len - 1];
        while (left < right) {
            int mid = left + (-left + right) / 2;
            if (isValid(position, m, mid)) {
                left = mid + 1;
            } else {
                right = mid;
            
}
        }
        return left - 1;
    }
    
    private boolean isValid(int[] positions, int m, int gap) {
        int cur = 1, pre = positions[0];
        for (int i = 1;i < positions.length;i++) {
            if (positions[i] - pre >= gap) {
                cur++;
                pre = positions[i];
            }
            if (cur >= m) {
                return true;
            }
        }
        return false; 
    }
}
