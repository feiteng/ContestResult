class Solution {
    public int maxDistance(int[] arr, int k) {
        Arrays.sort(arr);
        
        int result = 0;
        int n = arr.length;
        
        int l = 0, r = arr[n-1];
        
        while (l <= r) {
            int mid = (l + r) / 2;
            
            if (check(arr, mid, k)) {
                result = Math.max(result, mid);
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        
        return result;
    }
    
    private boolean check(int[] arr, int mid, int k) {
        int n = arr.length;
        
int curr = arr[0], cnt = 1;
        for (int i = 1; i < n; i++) {
            if (arr[i] - curr >= mid) {
                curr = arr[i];
                cnt++;
                
                if (cnt == k) {
                    return true;
                }
            }
        }
        return false;
    }
}
