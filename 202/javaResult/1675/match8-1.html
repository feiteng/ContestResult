class Solution {
    
    /* bsearch or minmax;
        bsearch is ok. the distance is 1 to Max_val;
        given distance, put balls see if if can put m; if > m distance is to small; if < m; distance is too large.
        if == m, we want it to move right.
    */
    
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int left = 1, right = Integer.MAX_VALUE;
        while (left < right) {
            int mid = right - (right - left) / 2;
            if (requiredBall(position, mid) >= m) left = mid;
            else right = mid - 1;
        }
        
        return left;
    }
    
    private int requiredBall(int[] pos, int minDist) 
{
        int prev = pos[0];
        int count = 1;
        for (int i = 1; i < pos.length; i++) {
            if (pos[i] - prev >= minDist) {
                prev = pos[i];
                count++;
            }
        }
        
        return count;
    }
}
