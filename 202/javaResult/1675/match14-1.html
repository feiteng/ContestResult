class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int res = 0, lo = 0, hi = Integer.MAX_VALUE;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (canPlace(position, mid, m)) {
                res = Math.max(res, mid);
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        return res;      
    }
    
    private boolean canPlace(int[] position, int force, int m) {
        int current = position[0]; m--;
        
for (int i = 1; i < position.length; i++) {
            if (position[i] - current >= force) {
                current = position[i];
                m--;
                if (m == 0) return true;
            }
        }
        return false;
    }
}
