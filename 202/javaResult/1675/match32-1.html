class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int l = 1;
        int r = 1000000000 - 1;
        while (r - l > 1) {
            int mid = l + (r - l) / 2;
            if (canUseAll(position, mid, m)) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        if (canUseAll(position, r, m)) return r;
        int mid = l + (r - l) / 2;
        if (canUseAll(position, mid, m)) return mid;
        
return l;

    }
     private boolean canUseAll(int position[], int dis, int m) {
        int pre = position[0];
        m--;
        for (int i = 1; i < position.length && m > 0; i++) {
            if (position[i] - pre >= dis) {
                pre = position[i];
                m--;
            }
        }
        return m == 0;
    }
}
