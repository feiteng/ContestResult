class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int low = 1;
        int high = position[position.length - 1];
        int result = -1;
        while(low < high){
            int mid = low + (high - low)/2;
            if(!helper(position, m, mid)){
                high = mid;
            }else{
                low = mid + 1;
                result = Math.max(result, mid);
            }
        }
        return result;
    }
    
    public boolean helper(int[] position, int m, int k){
        int pos = position[0];
        int elements = 1;
        for(int i=0; i<position.length; i++){
            if(position[i] - pos >= k){
                pos = position[i];
                elements++;
            }
            if(elements == m){
                return true;
            }
        }
        return false;
    }
}
