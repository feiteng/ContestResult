class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int left = 1;
        int right = position[position.length - 1] - position[0];
        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (canForm(position, mid, m)) {
                left = mid;
            } else {
                right = mid - 1;
            }
        }
        return canForm(position, right, m) ? right : left;
    
}
    private boolean canForm(int[] pos, int force, int cnt) {
        int prev = pos[0];
        cnt--;
        for (int i = 1; i < pos.length; i++) {
            if (pos[i] - prev >= force) {
                if (--cnt <= 0) {
                    return true;
                }
                
prev = pos[i];
            }
        }
        return false;
    }
}
