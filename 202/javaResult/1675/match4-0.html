class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int l = 0, r = Integer.MAX_VALUE;
        int res = 0;
        while (l < r) {
            int mid = l + (r - l) / 2;
            int t = Integer.MAX_VALUE;
            int ball = 0;
            int begin = 0;
            for (int index : position) {
                if (ball == 0) {
                    begin = index;
                    ++ball;
                } else {
                    if (index - begin >= mid) {
                        t = Math.min(t, index - begin);
                        ++ball;
                        begin = index;
                        if (ball == m) res = Math.max(res, t);
                    }
                }
            }
            if (ball >= m) l = mid + 1;
            else r = mid;
        }
        return res;
    }
}
