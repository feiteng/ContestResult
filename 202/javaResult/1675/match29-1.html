class Solution {
    public int maxDistance(int[] position, int m) {
        int n=position.length;
        Arrays.sort(position);
        int l=0,r=1000000000;
        while(l<r){
            int c=l+(r-l+1)/2;
            if(isOk(position,m,c)){
                l=c;
            }else{
                r=c-1;
            }
        }
        return l;
    }
    boolean isOk(int[] p,int m,int len){
        int pre=p[0];
        m--;
        for(int i=1;i<p.length;i++)
{
            if(p[i]-pre>=len){
                m--;
                pre=p[i];
                if(m<=0){
                    return true;
                }
            }
        }
        return false;
    }
}
