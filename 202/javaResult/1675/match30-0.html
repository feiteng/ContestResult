class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int min = 0;
        int high = position[position.length -1];
        while ( min <= high)  {
            int dist = (min + high) / 2;
            if (canFit (position, dist, m)) {
                min = dist + 1;
            }else {
                high = dist - 1;
            }
        }
            
        return min -1;
    }
    private boolean canFit (int[] position, int dist, int m) {
        int prev = position[0];
        m--;
        for (int i = 1; i < position.length; i++) {
            if (position[i] - prev >= dist) {
                prev = position[i];
                m--;
                if (m <= 0) {
                    return true;
                }
            }
        }
        return false;
        
    }
}
