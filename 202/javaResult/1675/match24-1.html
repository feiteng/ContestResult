class Solution {
    public int maxDistance(int[] position, int balls) {
        Arrays.sort(position);
        int l = 1, r = position[position.length-1]; //binary search for the max minimum magnetic force between any two balls 
        while(l<=r){
            int m = l + (r-l)/2; 
            if(can(position, m, balls))
                l = m + 1;
            else 
                r = m-1;
        }
        return r; 
    }
    
    private boolean can(int[] position, int d, int balls)
{
        int ballsPlaced = 1; 
        int prevPos = position[0]; 
        for(int pos : position){
            if(pos - prevPos >= d){
                ballsPlaced++; 
                prevPos = pos;
            }
        }
        return ballsPlaced >= balls;
    }
}
