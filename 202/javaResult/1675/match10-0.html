class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int l = 1, r = position[position.length - 1] - position[0] + 1;
        while (l < r) {
            int mid = (l + r) / 2;
            int cnt = 0, last = Integer.MIN_VALUE / 2;
            for (int i = 0; i < position.length && cnt < m; i++)
            {
                if (position[i] - last >= mid) {
                    cnt++;
                    last = position[i];
                }
            }
            if (cnt == m) l = mid + 1;
            else r = mid;
        }
        return l - 1;
    }
}
