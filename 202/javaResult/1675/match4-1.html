class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int left = 0, right = Integer.MAX_VALUE;
        int res = 0;
        while (left < right) {
            int mid = left + (right - left) / 2;
            int temp = Integer.MAX_VALUE;
            int ball = 0;
            int start = 0;
            for (int pos : position) {
                if (ball == 0) {
                    start = pos;
                    ++ball;
                } else {
                    if (pos - start >= mid) {
                        temp = Math.min(temp, pos - start);
                        ++ball;
                        start = pos;
                        if (ball == m) res = Math.max(res, temp);
                    }
                }
            }
            if (ball >= m) left = mid + 1;
            else right = mid;
        }
        return res;
    }
}
