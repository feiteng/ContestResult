class Solution {
    public int maxDistance(int[] position, int b) {
        Arrays.sort(position);
        
        int s = 1;
        int e = position[position.length -1] - position[0];
        int a = 1;
        while (s <= e) {
            int m = (s + e)/2;
            if (possible(position, m, b)) {
                a = m;
                
s = m + 1;
            } else {
                e = m - 1;
            }
        }
        return a;
    }
    
    private boolean possible(int[] position, int m, int b) {
        int lb = b -1;
        int lp = 0;
        for (int i = 1; i<position.length && lb > 0; i++) {
            if (position[i] - position[lp] >= m) {
                lb--;
                lp = i;
            }
        }
        
        return lb == 0;
        
    }
}
