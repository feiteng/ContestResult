class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int n = position.length;
        int res = -1;
        int left = 0, right = 2*((int)1e9);
        while (left < right) {
            int mid = (left + right) / 2;
            if (isFeasible(mid, position, n, m)) {
                res = Math.max(res, mid);
                left = mid + 1;
            } else { 
                right = mid;
            }
        }

        return res;
    }

    static boolean isFeasible(int mid, int[] positions, int n, int m) {
        int pos = positions[0];
        int elements = 1;
        for (int i = 1; i < n; i++) {
            if (positions[i] - pos >= mid) {
                pos = positions[i];
                elements++;
                if (elements == m) { 
                    return true;
                }
            }
        }
        return false;
    }
}
