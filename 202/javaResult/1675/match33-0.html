class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int low = 0;
        int high = position[position.length - 1] - position[0];
        while(low <= high) {
            int mid = low + (high - low) / 2;
            int count = m - 1;
            int pre = position[0];
            for(int i = 1; i < position.length; i++) {
                if(position[i] - pre >= mid) {
                    pre = position[i];
                    count--;
                } else {
                    continue;
                }
                if(count == 0) {
                    break;
                }
            }
            if(count == 0) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return high;
    }
}
