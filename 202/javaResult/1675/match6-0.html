class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);

        int n = position.length;
        int r = position[n - 1] - position[0];

        int l = 0;
        while (l < r) {
            int mid = l + (r - l + 1 >> 1);
            if (checkValid(position, m , mid)) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        return l;
    }

    private boolean checkValid(int[] position, int m, int mid) {
        int count = m;
        count--;
        int pre = position[0];
        for (int pos : position) {
            if (pos - pre >= mid) {
                pre = pos;
                count--;
            }
            if (count == 0) {
                return true;
            }
        }
        return false;
    }
}
