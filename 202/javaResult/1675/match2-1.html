class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        
        int l = 1;
        int r = 999999999;
        int ans = -1;
        
        while(l <= r){
            int mid = l + (r-l)/2;
            if(check(position,mid,m)){
                ans = mid;
                
l = mid+1;
            } else {
                r = mid-1;
            }
        }
        return ans;
    }
    
    public boolean check(int[] position, int x, int m){
        
        int lastused = 0;
        int used = 1;
        for(int i = 1; i < position.length; i++){
            if(position[i] - position[lastused] >= x){
                lastused = i;
                used++;
            }
            
        }
        return used >= m;
    }
}
