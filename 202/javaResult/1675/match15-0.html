class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int n = position.length;
        int left = 0;
        int right = position[n - 1] - position[0] + 1;
    //     while(l < r) {
    //     int mid = l + (r - l) / 2;
    //     if(A[mid] < target) {
    //         l = mid + 1;
    //     } else {
    //         r = mid;
    //     } 
    // }
    // return l;
        while(left < right) {
            int mid = left + (right - left) / 2;
            if (isValid(position, m, mid)) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left - 1;
    }
    
    boolean isValid(int[] position, int m, int mid) {
        int sum = 1;
        int last = 0;
        for (int i = 1; i < position.length; ++i) {
            if (position[i] - position[last] >= mid) {
                sum++;
                last = i;
                if (sum >= m) {
                    return true;
                }
            }
        }
        return false;
    }
}
