class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        int start = 1;
        int end = position[position.length - 1];
        while (start + 1 < end) {
          int mid = (end - start) / 2  + start;
          if (canMeet(position, m, mid)) {
            start = mid;
          } else {
            end = mid;
          }
        }
        if (canMeet(position, m, end)) {
          return end;
        }
      
return start;
    }
    
    private boolean canMeet(int []position, int m, int min) {
      
      int lastPos = position[0] ;
      m--;
      for (int i = 1; i < position.length; i++) {
        if (position[i] - lastPos >= min) {
          m--;
          lastPos = position[i];
          if (m == 0) {
            return true;
          }
        }
      }
      return false;
    }

}
