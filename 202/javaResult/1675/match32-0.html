class Solution {
    public int maxDistance(int[] position, int m) {
        Arrays.sort(position);
        long left = 0;
        long right = position[position.length-1];
        while(left <= right){
            long mid = left + (right - left)/2;
            if(put(position, m, mid)){
                left = mid+1;
            }else{
                right = mid-1;
            }
        }
        return (int)(left-1);
    }
    
    private boolean put(int[] arr, int m, long dist){
        int prev = arr[0];
        m--;
        for(int i =1;i<arr.length && m > 0;i++){
            if(arr[i] - prev >= dist){
                m--;
                prev = arr[i];
            }
        }
        return m == 0;
    }
}
