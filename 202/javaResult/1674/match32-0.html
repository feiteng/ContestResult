class Solution {
    public int minOperations(int n) {
        int sum = 0;
        for(int i = 1;i <= n; i++) {
            sum += (2 * i) + 1;
        }
        int reqAvg = sum/n;
        int diff = 0;
        for(int i = 1; i <= n; i++) {
            diff += Math.abs(((2 * i) + 1) - reqAvg);
        }
        return diff/2;
    }
}
