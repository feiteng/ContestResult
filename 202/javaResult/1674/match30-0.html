class Solution {
    public int minOperations(int n) {
        long sum = 0;
        for (int i = 0; i < n; i++) {
            sum += 2 * i + 1;
        }
        long avg = sum / n;
        int r = 0;
        for (int i = 0; i < n; i++) {
            r += Math.abs(avg - (2 * i + 1));
        }
        return (r + 1) / 2;
    }
}
