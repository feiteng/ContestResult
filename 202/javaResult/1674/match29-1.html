class Solution {
    public int minOperations(int n) {
        int ans = 0;
        for (int i = 0; i < n / 2; i++) {
            int v1 = 2 * i + 1;
            int v2 = 2 * (n - 1 - i) + 1;
            
            ans += (v1 + v2) / 2 - v1;
        }
        
        return ans;
    }
}
