class Solution {
    public int minOperations(int n) {
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += 2 * i + 1;
        }
        int m = sum / n, res = 0;
        for (int i = 0; i < n; i++) {
            res += Math.abs(2 * i + 1 - m);
        }
        return res / 2;
    }
}
