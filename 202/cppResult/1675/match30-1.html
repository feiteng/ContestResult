class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int p = 1, r = INT_MAX;
        while (p < r) {
            int q = r - (r - p) / 2;
            int index = position[0];
            int cnt = 1;
            for (int i = 1; i < position.size(); ++i) {
                if (position[i] - index >= q) {
                    ++cnt;
                    index = position[i];
                }

            }
            if (cnt >= m) {
                p = q;
            } else {
                r = q - 1;
            }
        }
        return p;
    }
};
