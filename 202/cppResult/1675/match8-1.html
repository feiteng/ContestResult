class Solution {
public:
    bool isFeasible(int mid, vector<int> &arr, int n, int m) 
    { 
        int pos = arr[0]; 
        int elements = 1; 
        for (int i=1; i<n; i++) 
        { 
            if (arr[i] - pos >= mid) 
            { 
                pos = arr[i]; 
                elements++; 
                if (elements == m) 
                  return true; 
            } 
        } 
        return 0; 
    } 
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(),position.end());
        if (m==2)
            return position.back()-position[0];

        int res = -1; 
        int left = 0, right = position.back(); 
        while (left < right) 
        { 
            int mid = (left + right)/2; 
            if (isFeasible(mid, position, position.size(), m)) 
            { 
                res = max(res, mid); 
                left = mid + 1; 
            } 
            else
                right = mid; 
        } 

        return res; 
    }
};
