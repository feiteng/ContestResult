class Solution {
public:
    
    int get(vector<int>& position, int mid) {
        int pre = -1e9-7;
        int cnt = 0;
        for (auto x : position) {
            if (x - pre >= mid) {
                cnt++;
                pre = x;
            }
        }
        return cnt;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int l = 0, r = 1e9+7;
        while (r - l > 1) {
            int mid = (r + l)/2;
            if (get(position, mid) >= m) l = mid;
            else r = mid;
        }
        return l;
    }
};
