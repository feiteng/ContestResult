class Solution {
public:
    
    bool check(vector<int>& position, int mid, int m) {
        int last_brancket = -1e9;
        int ans = 0;
        for (int i = 0; i < position.size(); ++i) {
            if (last_brancket + mid <= position[i]) {
                ++ans;
                last_brancket = position[i];
            }
        }
        return ans >= m;
    }
    
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int l = 0, r = 1e9 + 1, mid;
        while (l + 1 < r) {

            mid = (l + r) / 2;
            if (check(position, mid, m)) {
                l = mid;
            }
            else {
                r = mid;
            }
        }
        return l;
    }
};
