class Solution {
public:
    int maxDistance(vector<int>& a, int m) {
        sort(a.begin(), a.end());
        int n = a.size();
        int l = 1, r = 1000000000, ans = 0;
        while (l <= r) {
            int mid = (l + r) >> 1;
            int cnt = 1, prev = a[0];

            for (int i = 1; i < n; ++i) {
                if (a[i] - prev >= mid) {
                    ++cnt;
                    prev = a[i];
                }
            }
            if (cnt >= m) {
                ans = mid;
                l = mid + 1;
            }
            else {
                r = mid - 1;
            }
        }
        return ans;
    }
};
