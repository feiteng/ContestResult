class Solution {
public:
    int maxDistance(vector<int>& v, int m) {
        sort(v.begin(), v.end());
        
        int low = 0;
        int high = 1e9 + 10;
        
        while (low < high) {
            int mid = (low + high) / 2;
            
            int cnt = 1;
            int last = v[0];
            for (int j = 1; j < v.size() && cnt < m; j++) {
                if (v[j] - last >= mid) {
                    cnt++;
                    last = v[j];
                }
            }
            
            if (cnt < m) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low - 1;
    }
};
