class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int lo = 0, hi = position.back();
        
        while (lo < hi) {
            int mid = lo + (hi - lo + 1) / 2;
            
            int rem = m - 1;
            int tmp = position.front();
            for (int i = 1; i < position.size(); ++i) {
                if (position[i] - tmp >= mid) {
                    --rem;
                    tmp = position[i];
                }
            }
            
            if (rem <= 0) lo = mid;
            else hi = mid - 1;
        }
        
        return lo;
    }
};
