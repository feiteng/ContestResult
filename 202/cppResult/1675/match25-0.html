class Solution {
public:
    int getCount(vector<int>& p, int dis) {
        int s = p[0], cnt = 1;
        for(int i=0; i<p.size(); i++) {
            if (p[i] - s >= dis) {
                cnt++;
                s = p[i];
            }
        }
        return cnt;
    }
    int maxDistance(vector<int>& p, int m) {
        sort(p.begin(), p.end());
        int size = p.size();
        int end = p[size - 1] - p[0], start = 0;
        int ret = 0;
        while (start <= end) {
            int mid = start + (end - start)/2;
            int cnt = getCount(p, mid);
            if (cnt >= m) {
                ret = max(ret, mid);
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return ret;
    }
};
