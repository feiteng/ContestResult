class Solution {
public:
    int maxDistance(vector<int>& a, int b) {
        sort(a.begin(), a.end());
        int l = 0, r = a.back() - a.front();
        while (l < r) {

            const int m = (l + r + 1) >> 1;
            int x = 0;
            int last = -m;
            for (int i = 0; i < a.size(); ++i) {
                if (a[i] - last >= m) {
                    last = a[i];
                    x++;
                }
            }
            if (x >= b) {
                l = m;
            } else {
                r = m - 1;
            }
        }
        return r;
    }
};
