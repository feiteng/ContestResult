class Solution {
public:
    int maxDistance(vector<int>& p, int m) {
        sort(p.begin(),p.end());
        int l=1; int h=p.back();
        int ans=-1;
        while(l<=h){
            int g=(l+h)/2;
            int pr=p[0]; int pl=1;
            for(int i=1;i<p.size();i++){
                if(p[i]-pr>=g){
                    pr=p[i];
                    pl++;
                }
            }
            
            if(pl>=m){
                l=g+1;
                ans=g;
            }else{
                h=g-1;
            }
        }
        return ans;
    }
};
