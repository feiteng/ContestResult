class Solution {
public:
    int maxDistance(vector<int>& a, int m) {
        int l = 1;
        int r = 1e9;
        int ans = 1;
        sort(a.begin(), a.end());
        while(l <= r) {
            int mid = (l + r) / 2;
            int rem = m - 1;
            int last = a[0];
            for(int i = 1; i < a.size(); i++) {
                if(a[i] - last >= mid) {
                    last = a[i];
                    rem--;
                }
            }
            if(rem <= 0) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
};
