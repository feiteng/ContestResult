class Solution {
public:
    int maxDistance(vector<int>& p, int m) {
        sort(p.begin(), p.end());
        
        int l = 1, r = *max_element(p.begin(), p.end());
        
        while(l < r)
        {
            int mid = l+1 + (r-l)/2;
            //cout<<l<<" "<<mid<<" "<<r<<endl;
            if(can(p, m, mid))
            {
                l = mid;
            }
            else r = mid-1;
        }
        
        return l;
    }
    
    bool can(vector<int>& p, int m, int mid)
    {
        int res = 1;
        int pre = p[0];
        
        for(int i = 1; i < p.size(); i++)
        {
            if(p[i] - pre >= mid)
            {
                res++;
                pre = p[i];
            }
        }
        //cout<<mid<<" "<<res<<endl;
        return res >= m;
    }
};
