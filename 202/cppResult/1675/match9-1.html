class Solution {
public:
    int find(vector<int>& position, int p) {
        int last = position[0];
        int ret = 1;
        for (int i = 1;i < position.size();i++) {
            if (position[i] - last >= p) {
                ret++;
                last = position[i];
            }
        }
        return ret;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        
        int left = 1, right = position[position.size() - 1] - position[0] + 1;
        while (right - left > 1) {
            int t = (left + right) / 2;
            // cout << t << " " << find(position, t) << endl;
            if (find(position, t) >= m) {
                left = t;
            } else {
                right = t;
            }
        }
        return left;
    }
};
