class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int a = 1;
        int b = (position.back() - position[0]) / (m - 1) + 1;
        int res = a;
        while (a <= b) {
            int mid = (a + b) / 2;
            int cnt = 0;
            int lasti = 0;
            for (int i = 1; i < position.size(); ++i) {
                if (position[i] - position[lasti] >= mid) {
                    ++cnt;
                    if (cnt >= m - 1)
                        break;

                    lasti = i;
                }
            }
            if (cnt >= m - 1) {
                a = mid + 1;
                res = mid;
            } else {
                b = mid - 1;
            }
        }
        return res;
    }
};
