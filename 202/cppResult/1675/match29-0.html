class Solution {
public:
    bool check(int mid,vector<int>& pos, int n, int m){ 
        int cur = pos[0]; 
        int elements = 1; 
        for (int i=1; i<n; i++){ 
            if (pos[i] - cur >= mid){ 
                cur = pos[i]; 
                elements++; 
                if (elements == m) 
                  return true; 
            } 
        } 
        return false; 
    } 
    int maxDistance(vector<int>& pos, int m) {
        int n = pos.size();
        sort(pos.begin(),pos.end());
        int res = -1; 
        int left = 1, right = pos[n-1]; 
        while (left < right) { 
            int mid = (left + right)/2; 
            if (check(mid, pos, n, m)){ 
                res = max(res, mid); 
                left = mid + 1; 
            }
            else
                right = mid; 
        } 
        return res; 
    }
};
