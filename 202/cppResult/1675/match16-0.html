class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int n = position.size();
        int l = 1;
        int r = position[n - 1] - position[0] + 1;
        while (r - l >= 2) {
            int mid = (l + r) / 2;
            int last = position[0];
            int q = 1;
            for (int j = 0; j < n; j++) {
                if (position[j] - last >= mid) {
                    q++;
                    last = position[j];
                }
            }
            if (q >= m) {
                l = mid;
            } else {
                r = mid;
            }
        }
        return l;
    }
};
