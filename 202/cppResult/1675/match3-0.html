class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(begin(position), end(position));
        int n = position.size();
        int lb = 1, ub = position.back();
        while (ub - lb > 1) {
            int mid = lb + ub >> 1;
            
            int j = 0, prev = -mid;
            for (int i = 0; i < n; ++ i) {
                if (position[i] - prev >= mid) {
                    ++ j;
                    prev = position[i];
                }
            }
            
            if (j >= m) lb = mid;
            else ub = mid;
            
        }
        return lb;
    }
};
