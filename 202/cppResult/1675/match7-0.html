class Solution {
public:
    
    bool isFeasible(int mid, vector<int> arr, int n, int k) 
    { 
       int pos = arr[0];
       int elements = 1;
       for (int i=1; i<n; i++){
          if (arr[i] - pos >= mid){
             pos = arr[i];
             elements++;
             if (elements == k)
                return true;
          }
       }
       return 0;
    }
    
    int maxDistance(vector<int>& arr, int k) {

        sort(arr.begin(),arr.end()); 

        int res = -1; 
        int n = arr.size();

        int left = 1, right = arr[n-1]; 
        // for(int i=0;i<n;i++){
        //     cout<<arr[i]<<" ";
        // }
        // cout<<endl;
        while (left < right) 
        { 
            int mid = (left + right)/2; 
            // cout<<left<<" "<<mid<<" "<<right<<endl;
            if (isFeasible(mid, arr, n, k)) 
            { 
                // cout<<"isfeasible"<<endl;
                res = max(res, mid); 
                left = mid + 1; 
            } 
            else
                right = mid; 
        } 

        return res;         
    }
};
