class Solution {
public:
    int maxDistance(vector<int>& p, int m) {
        sort(p.begin(),p.end());
        int left = 1;
        int right = p.back() - p[0];
        while(left <= right)
        {
            int mid = left + (right-left)/2;
            int last = -1e9;
            int sum = 0;
            for(int i=0;i<p.size();i++)
            {
                if(last + mid <= p[i])
                {
                    sum++;
                    last = p[i];
                }
                if(sum>m)
                {
                    break;
                }

            }
            if(sum>=m)
            {
                left = mid + 1;
            }
            else
            {
                right = mid - 1;
            }
            
        }
        return right;
        
    }
};
