class Solution {
public:
    
    bool can(vector<int>& position, int m, int d) {
        int count = 0; int j = 0;
        for (int i = 0; i < position.size(); i++) {
            if (position[i]-position[j] >= d) {
                j = i;
                count++;
            }
        }
        return count+1 >= m;
    }
    
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int n = position.size();
        int len = position[n-1]-position[0];
        int d = len/(m-1)+1;

        int from = 1; int to = d;
        while (from < to) {
            int mid = (from+to+1)/2;
            if (can(position, m, mid)) {
                from = mid;
            } else {
                to = mid-1;
            }
        }
        return from;
    }
};
