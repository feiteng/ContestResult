class Solution {
public:
    bool trial(int x, vector<int> &vec, int m) {
        int cur = -x;
        int can = 0;
        
        for (auto v: vec) {
            if (cur + x <= v) {
                cur = v;
                can++;
            }
        }
        
        return can >= m;
    }
    int maxDistance(vector<int>& position, int m) {
        int n = (int)position.size();
        
        sort(position.begin(), position.end());
        

        int lb = 0, ub = position.back() - position[0] + 1;
        
        while (ub - lb > 1) {
            int mid = (ub + lb) >> 1;
            if (trial(mid, position, m)) lb = mid;
            else ub = mid;
        }
        
        return lb;
    }
};
