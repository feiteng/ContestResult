class Solution {
public:
    bool isCheck(vector<int>& arr, int d, int count) {
        
        int left=arr[0];
        count-=1;
        for(int i=1; i<arr.size(); i++) {
            if(arr[i]-left>=d) {
                count--;
                left = arr[i];
            }
        }
        
        return count<=0;
    }
    
    int maxDistance(vector<int>& arr, int m) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        
        if(n==m) {
            int ans = arr[n-1]-arr[0];
            for(int i=0; i<arr.size()-1; i++)
                ans = min(ans, arr[i+1]-arr[i]);
            return ans;
        }
        

        int left = 0, right = 1e9;
        int ans = 0;
        while(left<=right) {
            int mid = (left+right)/2;
            if(isCheck(arr, mid, m)) {
                ans = mid;
                left = mid+1;
            }
            else {
                right = mid-1;
            }
        }
        return ans;
    }
};
