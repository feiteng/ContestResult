class Solution {
public:
    bool possible(vector<int> &p,int mx,int m){
        int prev=p[0],cnt=1;
        for(int i=1;i<p.size();i++){
            if(p[i]-prev>=mx){
                cnt++;
                prev=p[i];
            }
        }
        return cnt>=m;
    }
    int maxDistance(vector<int>& p, int m) {
        sort(p.begin(),p.end());
        int l=0,r=p.back()-p[0];
        int ans=0;
        while(l<=r){
            
            int mid=(l+r)/2;
            if(possible(p,mid,m))
            { //  cout<<ans<<" "<<mid<<"\n";
                ans=mid;
                l=mid+1;
            }else{
                r=mid-1;
            }
        }
        return ans;
    }
};
