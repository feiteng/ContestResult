class Solution {
public:
    bool isValid(vector<int>& position, int m, int dist){
        int used = 1, pos = position[0];
        for(int i = 1; i < position.size() && used < m; i++){
            if(position[i] - pos >= dist){
                pos = position[i];
                used += 1;
            }
        }
        return used >= m;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int res = 0, low = 0, high = position.back() + 1;

        while(low < high){
            int mid = (low + high) / 2;
            
            if(isValid(position, m, mid)){
                res = max(res, mid);
               low = mid + 1;
            }else{
                high = mid;
            }
        }
        return res;
    }
};
