class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(begin(position), end(position));
        int l = 1, r = 1e9;
        while (l + 1 < r) {
            int mid = (l + r) / 2;
            int last = -1;
            int tot = 0;
            for (int i = 0; i < position.size(); i++) {
                if (last == -1) {
                    last = i;
                    tot++;
                } else {
                    if (position[last] + mid <= position[i]) {
                        tot++;
                        last = i;
                    }
                }
            }
            if (tot >= m) {
                l = mid;
            } else {
                r = mid;
            }
        }
        return l;
    }
};
