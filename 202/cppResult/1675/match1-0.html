class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        
        int l=0,r=1000000001;
        sort(position.begin(),position.end());
        while(r-l>1)
        {
            int mid=(l+r)/2; // gap
            int last=-1e9;
            int take=0;
            for(auto &z:position)
            {
                if(z-last >= mid)
                {
                    last=z;take++;
                }
            }
            if(take>=m)l=mid;
            else r=mid;
        }
        return l;
        
    }
};
