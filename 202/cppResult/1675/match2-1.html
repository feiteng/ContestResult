class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int lp = 1, rp = 1e9;
        while(lp < rp){
            int mid = lp + (rp - lp + 1) / 2;
            int last = - mid, cnt = 0;

            for(auto& v: position){
                if(v - last >= mid){
                    cnt ++;
                    last = v;
                }
            }
            if(cnt >= m) lp = mid;
            else rp = mid - 1;
        }
        return lp;
    }
};
