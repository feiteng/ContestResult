class Solution {
public:
    bool check(int key, vector<int>& arr, int m) {
        int pre = arr[0];
        m--;
        for (int i=1; i<arr.size(); i++) {
            if (arr[i] - pre >= key) {
                m--;
                pre = arr[i];
            }
            if (m==0) return true;

        }
        return false;
    }
    
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int l = 1, r = 1e9;
        int ans = 1;
        while (l <= r) {
            int mid = (l + (r-l)/2);
            if (check(mid, position, m)) {
                ans = mid;
                l = mid+1;
            } else {
                r = mid-1;
            }
        }
        return ans;
    }
};
