class Solution {
public:
    int maxDistance(vector<int>& a, int m) {
         sort(a.begin(), a.end());
        int l = 0, r = a.back() - a[0];
        while (l < r) {
            int mid = (l + r + 1) / 2;
            int now = a[0], s = 1;
            for (int i = 1; i < a.size(); i++) {
                if (a[i] - now >= mid) {
                    now = a[i];
                    s++;
                }
            }
            if (s >= m) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        return l;
    }
};
