class Solution {
public:
    int maxDistance(vector<int>& pos, int m) {
        sort(pos.begin(), pos.end());
        int l = 1, r = pos.back() - pos[0];
        while ( l + 1 < r ) {
            int mid = ( r - l ) / 2 + l;
            if ( check(pos, mid, m) ) {
                l = mid;
            } else r = mid;
        }
        if ( check(pos, r, m) ) return r;

        else return l;
    }
    
    bool check(vector<int>& pos, int x, int m) {
        int n = pos.size();
        int v = -1e9;
        for ( int i = 0; i < n; ++i ) {
            if ( pos[i] - v >= x ) {
                m--;
                v = pos[i];
            }
        }
        return m <= 0;
    }
};
