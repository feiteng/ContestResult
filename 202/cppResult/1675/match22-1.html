class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int L = 1, R = 1000000000;
        while(L + 1 < R){
            int M = (L + R) / 2;
            int k = 1, pre = position[0];
            for(int i = 1; i < (int)position.size(); i += 1)

                if(position[i] >= pre + M){
                    k += 1;
                    pre = position[i];
                }
            if(k >= m) L = M;
            else R = M;
        }
        return L;
    }
};
