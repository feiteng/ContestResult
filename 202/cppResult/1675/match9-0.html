class Solution {
public:
    int place(vector<int> &position, int target) {
        int c=0;
        int start=-target-1;
        for (int i=0; i<position.size(); i++) {
            if (position[i]>start+target) {
                start=position[i];
                c++;
            }
        }
        return c;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int l=1, r=1e9;
        while (l<r) {
            int mid=l+(r-l)/2;
            if (place(position, mid)>=m) {
                l=mid+1;
            } else {
                r=mid;
            }
        }
        return l;
    }
};
