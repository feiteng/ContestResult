class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(),position.end());
        int Max=1e9,Min=1;
        while(Max>Min+1){
            int mid=(Max+Min)/2;
            int tot=0;
            int last=-1e9;
            for(auto it:position){
                if(it-last>=mid){
                    tot++;
                    last =it;
                }
            }
            if(tot>=m)Min=mid;
            else Max=mid;
        }
        return Min;
    }
};
