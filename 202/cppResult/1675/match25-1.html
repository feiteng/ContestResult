class Solution {
public:
    bool ischeck(vector<int> &a, int d, int len){
        int left = a[0];
        len-=1;
        for(int i=1;i<a.size();i++){
            if(a[i]-left>=d){
                len--;
                left = a[i];
            }
        }
        return len<=0;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(),position.end());
        int n = position.size();
        int left = 0, right = 1e9;
        int ans=0;
        while(left<=right){
            int mid = left+(right-left)/2;

            if(ischeck(position,mid,m)){
                ans = mid;
                left = mid+1;
            }
            else{
                right = mid-1;
            }
        }
        return ans;
    }
};
