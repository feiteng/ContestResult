class Solution {
public:
    int maxDistance(vector<int>& p, int m) {
        int n=p.size();
        sort(p.begin(),p.end());
        int l=1, h=1e9, ans=0;
        while(l<h){
            int mid=(h-l)/2 + l;
            //cout<<l<<' '<<mid<<' '<<h<<endl;
            int ok=0, last=p[0], pick=1;
            for(int i=1;i<n;i++){
                if(p[i]-last>=mid) { pick++; last=p[i];}
            }
            if(pick>=m){
                ans=max(ans,mid);
                l=mid+1;
            }
            else{
                h=mid;
            }
        }
        return ans;
    }
};
