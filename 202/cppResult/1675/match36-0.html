class Solution {
public:
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        
        int l = 1, h = position.back() - position[0];
        while(l<h){
            int f = (h-l)/2+l;
            // cout<<l<<h<<f<<endl;
            if(possible(position, m, f)){
                l = f;
            }
            else{
                h = f-1;
            }
        // cout<<possible(position, m, f)<<endl;
            if(l +1 == h){
                if(possible(position, m, h))
                    return h;
                else
                    return l;
            }
        }
        // cout<<possible(position, m, 3)<<endl;
        return l;
    }
    bool possible(vector<int>& position, int m, int force){
        int i = 0;
        m--;
        for(int j = 1;m>0 && j<position.size();j++){
            if(position[j] - position[i] >= force){
                i = j;
                m--;
            }
        }
        return m == 0;
    }
};
