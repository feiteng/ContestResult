class Solution {
public:
    unordered_map<int,int>dp;
    int rec(int n){
        if(n==1)
            return 1;
        if(n==2||n==3)
            return 2;
        if(dp.count(n))
            return dp[n];
        int ans=0;
        ans = 1+n%2+rec(n/2);
        ans = min(ans,1+n%3+rec(n/3));
        dp[n]=ans;
        return dp[n];
    }
    int minDays(int n) {
        // dp=vector<long long int>(n+5,INT_MAX);
        return rec(n);
    }
};
