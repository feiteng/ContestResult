class Solution {
public:
    map<int, int> dp;
    int minDays(int n) {
        if (n == 1) return 1;
        if (n <= 3) return 2;
        if (dp.count(n)) return dp[n];
        
        int ans = n;
        ans = min(ans, minDays(n/2)+(n%2)+1);
        ans = min(ans, minDays(n/3)+(n%3)+1);        
        return dp[n] = ans;
    }
};
