class Solution {
public:
    int minDays(int n) {
        //using bfs
        if(n<2) return n;
        queue<int> q;
        unordered_set<int> v;
        q.push(n);
        v.insert(n);
        int step=0;
        while(q.size()){
            int sz=q.size();
            
            while(sz--){
                int cur=q.front();
                q.pop();
                if(cur==0) return step;
                if(cur%3==0 && !v.count(cur/3)) q.push(cur/3),v.insert(cur/3);
                if(cur%2==0 && !v.count(cur/2)) q.push(cur/2),v.insert(cur/2);
                if(!v.count(cur-1)) q.push(cur-1),v.insert(cur-1);
            }
            step++;
        }
        return -1;
    }
};
