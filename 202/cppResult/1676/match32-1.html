class Solution {
public:
    int dfs(int n,unordered_map<int,int> &dp){
        if(n==0) return 0;
        if(dp.find(n) != dp.end()) return dp[n];
        dp[n] = n;

        dp[n] = min(dp[n],(n%2)+1+dfs((n-(n%2))/2,dp));
        dp[n] = min(dp[n],(n%3)+1+dfs((n-(n%3))/3,dp));
        return dp[n];
    }
    int minDays(int n) {
        unordered_map<int,int> dp;
        // int res = 0;
        // for(int i=1;i<=31;i++){
        //     dp[1<<(i-1)] = i;
        // }
        // int res = 0;
        // while(n){
        //     if(dp.find(n)) return dp[n]+res;
        //     if()
        // }
        // for(int i=1;i<10;i++) cout<<dfs(i,dp)<<endl;
        // cout<<dfs(2,dp)<<endl;
        return dfs(n,dp);
    }
};
