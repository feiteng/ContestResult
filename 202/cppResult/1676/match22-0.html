class Solution {
public:
    map<int,int>mp;
    int DP(int n) {
        if(mp[n]) return mp[n];
        if(n <= 2) return n;
        if(n == 3) return 2;
        int x1 = n % 2,x2 = n % 3;
        mp[n] = min(DP((n - x1) / 2) + x1,DP((n - x2) / 3) + x2) + 1;
        return mp[n];
    }
    int minDays(int n) {
        return DP(n);
    }
};
