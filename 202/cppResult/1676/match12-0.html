class Solution {
public:
    int minDays(int n) {
        /*
        vector<int> f(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            f[i] = 1 + f[i - 1];
            if (i % 2 == 0) {
                f[i] = min(f[i], f[i / 2] + 1);
            }
            if (i % 3 == 0) {
                f[i] = min(f[i], f[i / 3] + 1);
            }
        }
        return f[n];
        */
        queue<int> q;
        unordered_set<int> visited;
        q.push(n);
        visited.insert(n);
        int level = 0;
        while (!q.empty()) {
            int n = q.size();
            for (int i = 0; i < n; i++) {
                int curr = q.front();
                q.pop();
                if (curr == 0) return level;
                if (!visited.count(curr - 1)) {
                    q.push(curr - 1);
                }
                if (curr % 2 == 0 && !visited.count(curr / 2)) {
                    visited.insert(curr / 2);
                    q.push(curr / 2);
                }
                if (curr % 3 == 0 && !visited.count(curr / 3)) {
                    visited.insert(curr / 3);
                    q.push(curr / 3);
                }
            }
            level++;
        }
        return level;
    }
};
