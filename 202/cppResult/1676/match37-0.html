class Solution {
public:
    map<int, int> dp;
    int minDays(int n) {
        if(n == 0) return 0;
        if(n == 1) return 1;
        if(dp.count(n)) return dp[n];
        int res = min(minDays(n / 2) + n % 2 + 1, minDays(n / 3) + n % 3 + 1);
        dp[n] = res;
        return res;
    }
};
