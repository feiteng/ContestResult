class Solution {
public:
    int minDays(int n) {
            queue<int> q;
        unordered_set<int> vis;
        q.push(n);
        vis.insert(n);
        int step = 0;
        while (!q.empty()) {
            int size = q.size();
            while (size--) {
                int u = q.front();
                if (u == 0) return step;
                q.pop();
                if (u % 2 == 0 && !vis.count(u / 2)) {
                    q.push(u / 2);
                    vis.insert(u / 2);
                }
                if (u % 3 == 0 && !vis.count(u - u / 3 * 2)) {
                    q.push(u - u / 3 * 2);
                    vis.insert(u - u / 3 * 2);
                }
                if (!vis.count(u - 1)) {
                    q.push(u - 1);
                    vis.insert(u - 1);
                }
            }
            step++;
        }
        return step;
    }
};
