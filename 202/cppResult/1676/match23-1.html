class Solution {
public:
    unordered_map<int,int>mp;
  
    
     int go(int n)
    {
        if(n==0)return 0;
        if(mp.find(n)!=mp.end())return mp[n];
        
        int ans=INT_MAX;
            ans=min(ans,n%2+1+go(n/2));
            ans=min(ans,n%3+1+go(n/3));
        mp[n]=ans;
        return mp[n];
    }
    
    int minDays(int n) {
        int ans=go(n);
        return ans-1;
    }
};
