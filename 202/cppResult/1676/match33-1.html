class Solution {
public:
    int minDays(int n) {
        // T(n) = min(T(n - 1), T(n / 2 [if n % 2 == 0]), T(n / 3 if [N % 3 == 0])) + 1
        unordered_map<int, int> visited;
        return DFSEatOrange(n, visited);
    }
    

    int DFSEatOrange(int n , unordered_map<int, int>& visited) {
        if (visited.find(n) != visited.end()) return visited[n];
        if (n <= 1) return n;
        
        int min_move = n;
        min_move = min(min_move, DFSEatOrange(n / 2, visited) + 1 + n % 2);
        
        min_move = min(min_move, DFSEatOrange(n / 3, visited) + 1 + n % 3);

       
        visited[n] = min_move;
        return min_move;
    }
};
