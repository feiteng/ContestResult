class Solution {
public:
    unordered_map<int, int> m;
    
    int solve(int n){
        if(n==0 || n==1 || n==2) return n;

        if(m[n]!=0) return m[n];
        
        int ans = INT_MAX;
        if(n%3==0) ans = min(ans, 1 + solve(n/3));
        if(n%2==0) ans = min(ans, 1 + solve(n/2));
        if((n-1)%2==0) ans = min(ans, 1 + solve(n-1));
        if((n-1)%3==0) ans = min(ans, 1 + solve(n-1));
        if((n-2)%3==0) ans = min(ans, 2 + solve(n-2));

        return m[n] = ans;
    }
    
    int minDays(int n) {
        return solve(n);
    }
};
