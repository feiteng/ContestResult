class Solution {
public:
    int minDays(int n) {
        if ( n <= 2 ) return n;

        queue<int> q;
        q.push(n);
        unordered_set<int> st;
        st.insert(n);
        int res = 0;
        while ( !q.empty() ) {
            int k = q.size();
            while ( k-- ) {
                int cur = q.front();
                q.pop();
                if ( cur == 0 ) return res;
                if ( !st.count(cur - 1) ) { 
                    q.push(cur - 1);
                    st.insert(cur - 1);
                }
                
                if ( cur % 2 == 0 && !st.count(cur / 2) ) { 
                    q.push(cur / 2);
                    st.insert(cur / 2);
                }
                
                if ( cur % 3 == 0 && !st.count(cur / 3) ) { 
                    q.push(cur / 3);
                    st.insert(cur / 3);
                }
            }
            ++res;
        }
        return -1;
    }
};
