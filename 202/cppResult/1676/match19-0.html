class Solution {
public:
    unordered_map<int, int> dp;
    int utility(int n) {
        if(dp.find(n)!=dp.end())
            return dp[n];
        dp[n] = min(n%2+1+utility(n/2), n%3+1+utility(n/3));
        return dp[n];
    }
    
    int minDays(int n) {
        dp.clear();
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 2;
        return utility(n);
    }
};
