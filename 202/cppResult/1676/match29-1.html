class Solution {
public:
    int minDays(int n) {

        queue<int> q;
        q.push(n);
        unordered_set<int> vis;
        int steps = 0;
        vis.insert(n);
        while (!q.empty()) {
            int cnt = q.size();
            while (cnt--) {
                int u = q.front(); q.pop();
                if (u == 0) return steps;
                if (!vis.count(u - 1)) vis.insert(u - 1), q.push(u - 1);
                if (u % 2 == 0 && !vis.count(u / 2)) q.push(u/ 2), vis.insert(u / 2);
                if (u % 3 == 0 && !vis.count(u / 3)) q.push(u/ 3), vis.insert(u / 3);
                
            }
            steps++;
        }
        return steps;
    }
};
