class Solution {
public:
    int minDays(int n) {
        queue<pair<int, int>> q;
        unordered_set<int> mm;
        q.push({n, 0});
        while (true) {
            auto& f = q.front();
            if (f.first == 0)
                return f.second;
            if (mm.find(f.first) != mm.end()) {
                q.pop();
                continue;
            }
            mm.insert(f.first);

            q.push({f.first - 1, f.second + 1});
            if ((f.first % 2) == 0) {
                q.push({f.first / 2, f.second + 1});
            }
            if ((f.first % 3) == 0) {
                q.push({f.first / 3, f.second + 1});
            }
            q.pop();
        }
        return -1;
    }
};
