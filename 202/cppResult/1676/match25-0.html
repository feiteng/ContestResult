class Solution {
public:
    int minDays(int n) {
        unordered_set<int> s;
        queue<pair<int, int>> q;
        q.push({n, 1});
        while (true) {
            auto p = q.front();
            q.pop();
            int x = p.first, step = p.second;
            if (x == 1) return step;
            if (x % 3 == 0 && s.find(x / 3) == s.end()) {
                q.push({x / 3, step + 1});
                s.insert(x / 3);
            }
            if (x % 2 == 0 && s.find(x / 2) == s.end()) {
                q.push({x / 2, step + 1});
                s.insert(x / 2);
            }
            if (s.find(x - 1) == s.end()) {
                q.push({x - 1, step + 1});
                s.insert(x - 1);
            }
        }
        return -1;
    }
};
