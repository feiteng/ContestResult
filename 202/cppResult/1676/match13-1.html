class Solution {
public:
    unordered_map<int,int> ans;
    int minDays(int n) {
        if(n==0)    return 0;
        if(n==1)    return 1;

        if(ans.count(n)) return ans[n];
        // if(isp(n)){
        //     ans[n]==ans
        // }
        int res = INT_MAX;
        if(n%2==0)   res = min(res,minDays(n/2)+1);
        else res = min(res,minDays(n/2)+2);
        if(n%3==0)   res = min(res,minDays(n/3)+1);
        else if(n%3==1) res = min(res,minDays(n/3)+2);
        else res = min(res,minDays(n/3)+3);
        ans[n]=res;
        return res;
    }
};
