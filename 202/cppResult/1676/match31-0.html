class Solution {
public:
    unordered_map<int, int> mm;
    int minDays(int n) {
        if(mm.count(n)) return mm[n];
        if (n == 0) return 0;
        if (n == 1) return 1;
        int res =  min(n % 2 + 1 + minDays(n/2), n % 3 + 1 + minDays(n/3));
        mm[n] = res;
        return res;
    }
};
