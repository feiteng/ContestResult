class Solution {
public:

    int minDays(int n) {
        queue<int> q;
        q.push(n);
        unordered_set<int> visited; visited.insert(n);
        int ret = 0;
        while (!q.empty()) {

            ret++;
            for (int i = q.size(); i > 0; --i) {
                int cur = q.front(); q.pop();
                if (cur == 1) return ret;
                if (cur % 2 == 0 && !visited.count(cur/2)) {
                    visited.insert(cur/2);
                    q.push(cur/2);
                }
                if (cur % 3 == 0 && !visited.count(cur/3)) {
                    visited.insert(cur/3);
                    q.push(cur/3);
                }
                if (!visited.count(cur-1)) {
                    visited.insert(cur-1);
                    q.push(cur-1);
                }
            }
        }
        return ret;
    }
};
