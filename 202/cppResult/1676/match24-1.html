class Solution {
public:
    unordered_map<int,int> dp;
    int getAns(int n){
        if(n==1)
            return 0;
        if(n==0)
            return -1;
        if(dp.find(n)!=dp.end())
            return dp[n];
        return dp[n]=1+min(n%2+getAns(n/2),n%3+getAns(n/3));
    }
    
    int minDays(int n) {
        int count=getAns(n);
        return count+1;
    }
};
