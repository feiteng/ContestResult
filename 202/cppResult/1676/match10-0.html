class Solution {
public:
    int minDays(int n) {
        int res = 0;
        queue<int> q;
        q.push(n);
        unordered_set<int> visited;
        visited.insert(n);
        while (!q.empty()) {
            int size = q.size();
            while (size--) {
                int curr = q.front();
                q.pop();
                if (curr == 1)
                    return res + 1;
                if (visited.find(curr - 1) == visited.end()) {
                    q.push(curr - 1);
                    visited.insert(curr - 1);
                }
                if (curr % 2 == 0 && visited.find(curr / 2) == visited.end()) {
                    q.push(curr / 2);
                    visited.insert(curr / 2);
                }
                if (curr % 3 == 0 && visited.find(curr / 3) == visited.end()) {
                    q.push(curr / 3);
                    visited.insert(curr / 3);
                }
            }
            res++;
        }
        
        
        return res;
    }
};
