class Solution {
public:
    int helper(int n, unordered_map<int, int>& dp) {
        if (n == 1) return 1;
        if (n == 2) return 2;
        if (n == 3) return 2;
        unordered_map<int, int>::iterator it = dp.find(n);
        if (it != dp.end())
            return (*it).second;
        int d2 = 0;
        if ( (n&1) == 1)
            d2 = 2 + helper((n-1)/2, dp);
        else
            d2 = 1 + helper(n/2, dp);
        int d3 = 0;
        int rem = n%3;
        if ( rem == 0)
            d3 = 1 + helper(n/3, dp);
        else if ( rem == 1) 
            d3 = 2 + helper((n-1)/3, dp);
        else
            d3 = 3 + helper((n-2)/3, dp);
        int ans = min(d2, d3);
        dp[n] = ans;
        return ans;
    }
    int minDays(int n) {
        unordered_map<int, int> dp;
        return helper(n, dp);
    }
};
