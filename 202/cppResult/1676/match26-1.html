class Solution {
public:
    int minDays(int n) {
        unordered_set<int> isVisited;
        queue<int> orangeNumberQueue;
        orangeNumberQueue.push(n);
        int res = 0;
        while (!orangeNumberQueue.empty()) {
            res++;
            int size = orangeNumberQueue.size();

            for(int i = 0; i < size; i++) {
                int cur = orangeNumberQueue.front();
                orangeNumberQueue.pop();
                if (cur - 1 == 0)
                    return res;
                if (!isVisited.count(cur - 1)) {
                    orangeNumberQueue.push(cur-1);
                    isVisited.insert(cur - 1);       
                }
                    
                if (cur % 2 == 0) {
                    if (!isVisited.count(cur/2)) {
                        orangeNumberQueue.push(cur/2);
                        isVisited.insert(cur/2);
                    }
                        
                }
                if (cur % 3 == 0) {
                    if (!isVisited.count(cur/3)) {
                        orangeNumberQueue.push(cur/3);
                        isVisited.insert(cur/3);
                    }
                }   
            }
        }
        return 0;
    }
};
