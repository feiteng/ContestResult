class Solution {
public:
    int minDays(int nn) {
        queue<int> bfs, bfs_next;
        bfs.push(nn);
        unordered_set<int> vis;
        vis.insert(nn);
        int step = 0;
        while (true) {
            while (!bfs.empty()) {
                int n = bfs.front(); bfs.pop();
                if (n == 0) return step;
                if (vis.find(n - 1) == vis.end()) {
                    vis.insert(n - 1); bfs_next.push(n - 1);
                }
                if (n % 2 == 0 && vis.find(n / 2) == vis.end()) {
                    vis.insert(n / 2); bfs_next.push(n / 2);
                }
                if (n % 3 == 0 && vis.find(n / 3) == vis.end()) {
                    vis.insert(n / 3); bfs_next.push(n / 3);
                }
            }
            step++, bfs.swap(bfs_next);
        }
        return INT_MAX;
    }
};
