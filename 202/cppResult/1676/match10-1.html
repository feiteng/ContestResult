class Solution {
public:
    int minDays(int n) {
        int dist=0;
        queue<int> q;
        q.push(n);
        unordered_set<int> seen;
        while (!q.empty()) {

            for (int l=q.size()-1; l>=0; l--) {
                int curr=q.front(); q.pop();
                if (curr==0) return dist;
                if (seen.find(curr-1)==seen.end()) {
                    seen.insert(curr-1);
                    q.push(curr-1);
                }
                if (curr%2==0&&seen.find(curr/2)==seen.end()) {
                    seen.insert(curr/2);
                    q.push(curr/2);
                }
                if (curr%3==0&&seen.find(curr/3)==seen.end()) {
                    seen.insert(curr/3);
                    q.push(curr/3);
                }
            }
            dist++;
        }
        return dist;
    }
};
