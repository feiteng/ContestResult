class Solution {
public:
    int minDays(int n) {
        queue<pair<int,int>> q;
        q.push({n,0});
        unordered_set<int> s;
        while(true){
            int num=q.front().first,day=q.front().second;
            q.pop();
            if(num==1)return day+1;
            if(num%2==0&&!s.count(num/2)){
                s.insert(num/2);
                q.push({num/2,day+1});
            }
            if(num%3==0&&!s.count(num/3)){
                s.insert(num/3);
                q.push({num/3,day+1});
            }
            if(!s.count(num-1)){
                s.insert(num-1);
                q.push({num-1,day+1});
            }
        }
        return -1;
    }
};
