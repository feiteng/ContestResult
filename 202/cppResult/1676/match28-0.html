class Solution {
public:
    int minDays(int n) {
        
        set<int> st;
        queue<int> qu;
        qu.push(n);
        st.insert(n);
        int day=0;
        while(!qu.empty())
        {
            
            int size = qu.size();
            day++;
        
            for(int i=0;i<size;i++)
            {
                long long count=qu.front();
                qu.pop();
                
                if(count==0) return day-1;
                
                if(st.count(count-1)==0)
                {
                    st.insert(count-1);
                    qu.push(count-1);
                }
                
                if(count%2==0&&st.count(count-count/2)==0)
                {
                    st.insert(count-count/2);
                    qu.push(count-count/2);
                }
                
                if(count%3==0&&st.count(count-2*count/3)==0) 
                {
                    st.insert(count-2*count/3);
                    qu.push(count-2*count/3);
                }
                
            }
            
            
        }
        
        return 0;

    }
};
