class Solution {
public:
    map<int, int> dp;
    
    int minDays(int n) {
        queue<int> que;
        
        que.push(n);
        dp[n] = 0;
        
        while (!que.empty()) {
            int v = que.front(); que.pop();
            
            if (v == 0) break;
            

            if (!dp.count(v - 1)) {
                dp[v - 1] = dp[v] + 1;
                que.push(v - 1);
            }
            
            if (v % 2 == 0 && !dp.count(v - v / 2)) {
                dp[v - v / 2] = dp[v] + 1;
                que.push(v - v / 2);
            }
            
            if (v % 3 == 0 && !dp.count(v - v / 3 * 2)) {
                dp[v - v / 3 * 2] = dp[v] + 1;
                que.push(v - v / 3 * 2);
            }
        }
        
        return dp[0];
    }
};
