int dp[10000007];
bool flag=false;

class Solution {
public:
    unordered_map<int,int>m;
    int dfs(int i){
        if(i<=1e7)return dp[i];
        if(m.count(i))return m[i];
        return m[i]=min(1+i%2+dfs(i/2),1+i%3+dfs(i/3));
    }
    int minDays(int n) {
        if(!flag){
            flag=true;
            dp[0]=0;
            for(int i=1;i!=1e7+1;i++){
                dp[i]=dp[i-1]+1;
                if(i%2==0)dp[i]=min(dp[i],dp[i/2]+1);
                if(i%3==0)dp[i]=min(dp[i],dp[i/3]+1);
            }
        }
        if(n<1e7)return dp[n];
        return dfs(n);
    }
};
