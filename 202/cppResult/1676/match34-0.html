class Solution {
public:
    unordered_map<int, int> dp;
    int fun(int n) {
        if(n == 0) return 0;
        if(n == 1) return 1;
        if(dp[n]) return dp[n];
        return dp[n] = min(n % 2 + 1 + fun(n / 2), n % 3 + 1 + fun(n / 3));
    }
    
    int get(int n) {
        // queue<pair<int, int>> q;
        // q.push({n, 0});
        int ans = 0;
        while(n > 0) {
            while(n % 2 == 0) {
                n /= 2; 
                ans++;
            }
            while(n % 3 == 0) {
                n /= 3;
                ans++;
            }
            n -= 1;
            ans++;
        }
        return ans;
    }
    
    int minDays(int n) {
        // queue<pair<int, int>> q;
        // q.push({n, 0});
        return fun(n);
        int ans = 0;
        while(n > 0) {
            while(n % 2 == 0) {
                n /= 2; 
                ans++;
            }
            while(n % 3 == 0) {
                n /= 3;
                ans++;
            }
            n -= 1;
            ans++;
        }
        return ans;
    }
};
