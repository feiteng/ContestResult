class Solution {
public:
    unordered_map<int,int>mp;
    int dfs(int n){
        if(mp[n]>0)return mp[n];
        if(n==1)return 1;
        else if(n<=3)return 2;
        int ans=n;
        if(n%2==0)ans=min(ans,dfs(n/2)+1);
        else ans=min(ans,dfs(n/2)+2);
        if(n%3==0)ans=min(ans,dfs(n/3)+1);
        else if(n%3==1)ans=min(ans,dfs(n/3)+2);
        else ans=min(ans,dfs(n/3)+3);
        mp[n]=ans;
        return ans;
    }
    int minDays(int n) {
        return dfs(n);
    }
};
