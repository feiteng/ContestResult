class Solution {
public:
    map<int,int> dp;
    int minDays(int n) {
        if(n <= 2) return n;
        if(dp.find(n)!=dp.end()) 
            return dp[n];
        int c = INT_MAX;
        c = min(c,minDays(n/3)+n%3+1);
        c = min(c,minDays(n/2)+n%2+1);
        
        // if(n%3==0)
        // {
        //     c = min(c,minDays(n/3)+1);
        //     if(n%2==0)  c = min(c,minDays(n/2)+1);
        //     c = min(c,minDays(n/3)+1);
        // }
        // else 
        // {   
        //     if(n%2==0)
        //     { 
        //         c = min(c,minDays(n/2)+1);
        //         c = min(c,minDays(n-1)+1);
        //     }
        //     else
        //         c = min(c,minDays(n-1)+1);
        // }
            
        dp[n] = c;
        return dp[n];
    }
};
