class Solution {
public:
    int minDays(int n) {
        queue<int> q;
        q.push(n);
        unordered_set<int> v;
        int res = 0;
        while(true) {

            res++;
            for(int k = q.size(); k > 0; k--) {
                int t = q.front(); q.pop();
                if(t==1) return res;
                if(!v.count(t-1)) {
                    v.insert(t-1);
                    q.push(t-1);
                }
                if(t%2==0) {
                    if(!v.count(t/2)) {
                        v.insert(t/2);
                        q.push(t/2);
                    }
                }
                if(t%3==0) {
                    if(!v.count(t/3)) {
                        v.insert(t/3);
                        q.push(t/3);
                    }
                }
            }
        }
        return -1;
    }
};
