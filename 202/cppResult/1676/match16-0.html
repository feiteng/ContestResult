using namespace std;
using ll = long long;
using ld = double;
typedef pair<int, int> II;
typedef pair<II, int> III;
typedef complex<ld> cd;
typedef vector<cd> vcd;

class Solution {
public:
    map<int, int> ma;
    
    int solve(int n) {
        if (ma.count(n)) return ma[n];
        if (n <= 2) return n;
        int res = 1000000000;
        if (n % 2 == 0) res = min(res, 1 + solve(n/2));
        if (n % 2 == 1) res = min(res, 2 + solve((n-1) / 2));
        if (n % 3 == 0) res = min(res, 1 + solve(n/3));
        if (n % 3 == 1) res = min(res, 2 + solve((n-1) / 3));
        if (n % 3 == 2) res = min(res, 3 + solve((n-2) / 3));
        return ma[n] = res;
    }
    
    int minDays(int n) {
        ma.clear();
        return solve(n);
    }
};
