class Solution {
public:
    
    
    int dfs(int n, unordered_map<int, int>& memo){
        // cout << n << endl;
        if (memo.count(n)) return memo[n];
        if (n == 0) return 0;
        if (n == 1) return 1;
        if (n == 2) return 2;
        int a = dfs(n / 3, memo) + 1 + dfs(n % 3, memo);
        int b = dfs(n / 2, memo) + 1 + dfs(n % 2, memo);
        memo[n] = min(a, b);
        return memo[n];
    }
    
    
    int minDays(int n) {
        unordered_map<int, int> memo;
        return dfs(n, memo);
    }
};
