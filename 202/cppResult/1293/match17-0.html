class Solution {
public:
    bool threeConsecutiveOdds(vector<int>& arr) {
        int odds = 0;
        for (auto n : arr) {
            if (n % 2)
                ++odds;
            else
                odds = 0;
            if (odds == 3)
                return true;
        }
        return false;
    }
};
