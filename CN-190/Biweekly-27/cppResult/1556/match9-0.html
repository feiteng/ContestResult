class Solution {
public:
    bool canBeEqual(vector<int>& v, vector<int>& vv) {
        sort(v.begin(), v.end());
        sort(vv.begin(), vv.end());
        if (v.size() != vv.size()) return false;
        for (int i = 0; i < v.size(); i++) {
            if (v[i] != vv[i]) return false;
        }
        return true;
    }
};
