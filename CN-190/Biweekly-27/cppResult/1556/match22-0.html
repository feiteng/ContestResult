class Solution {
public:
    bool canBeEqual(vector<int>& target, vector<int>& arr) {
        unordered_map<int, int> mp;
        for (auto& t: arr)
            mp[t]++;
        for (auto& t: target) {
            if (mp.find(t) == mp.end())
                return false;
            mp[t]--;
            if (mp[t] < 0)
                return false;
        }
        return true;
    }
};
