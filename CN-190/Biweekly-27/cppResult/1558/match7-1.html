class Solution {
public:
    
    void dfs(  vector<vector<int>> &g, set<int> &s,int st,int *vis){
        vis[st]=1;
        for(int n:g[st]){
            if(!vis[n])
                dfs(g,s,n,vis);
        }
        s.insert(st);
    }
    
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
       
        vector<vector<int>> g(n);
        for(auto &l:prerequisites)
            g[l[1]].push_back(l[0]);
        
        vector<set<int>> v;

        for(int i=0;i<n;i++){
            set <int> s;
            int *visited=new int[n]();
            dfs(g,s,i,visited);
            v.push_back(s);
        }
        vector<bool> ans;
        for(auto &l:queries){
            if(v[l[1]].find(l[0])!=v[l[1]].end())
                ans.push_back(true);
            else
                ans.push_back(false);
        }
        return ans;
    }
};
