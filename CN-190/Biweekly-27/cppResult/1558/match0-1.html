class Solution {
public:
    vector<vector<bool>>ret;
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& p, vector<vector<int>>& q) {
        ret=vector<vector<bool>>(n,vector<bool>(n));
        for(auto& it:p)
            ret[it[0]][it[1]]=true;
        for (int k = 0; k <n; k++)
      for (int i = 0; i <n; i++)
       for (int j = 0; j <n; j++)
        if (ret[i][k]&&ret[k][j])
         ret[i][j]=true;
        vector<bool>ans(q.size());
        for(int i=0;i<q.size();i++)
        {
            auto& it=q[i];
            if(ret[it[0]][it[1]])
                ans[i]=true;
        }
        return ans;
    }
};
