char G[111][111];
class Solution {

public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& e, vector<vector<int>>& queries) {
        memset(G, false, sizeof(G));
        for(auto edge: e)
            G[edge[0]][edge[1]] = true;
        for(int k = 0; k < n; k ++)
            for(int i = 0; i < n; i ++)
                for(int j = 0; j < n; j ++) 
                    G[i][j] |= (G[i][k] & G[k][j]);
        vector<bool> ans;
        for(auto q: queries) {
            ans.push_back(G[q[0]][q[1]]);;
        }
        return ans;
    }
};
