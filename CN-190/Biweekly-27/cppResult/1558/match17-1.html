class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<int>> mat(n+1, vector<int>(n+1, 1000));
        
        for(auto& p : prerequisites)
        {
            int x = p[0]; int y = p[1];

            mat[x][y] = 1;
        }
        
        for(int mid = 0; mid < n; ++mid)
        {
            for(int start = 0; start < n; ++start)
            {
                for(int dest = 0; dest < n; ++dest)
                {
                    int dist = mat[start][mid] + mat[mid][dest];

                    if(dist < mat[start][dest])
                    {
                        mat[start][dest] = dist;
                    }
                }
            }
        }
        vector<bool> ans;
        for(auto& q: queries)
        {
            if(mat[q[0]][q[1]] < n)
            {
                ans.push_back(true);
            }
            else
                ans.push_back(false);
        }
        
        return ans;
    }
};
