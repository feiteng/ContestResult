class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& edges, vector<vector<int>>& queries) {
        vector<vector<int>> distances(n, vector<int>(n, INT_MAX / 2));
        for (const auto& e : edges)
            distances[e[0]][e[1]] = 1;
        for (int k = 0; k < n; ++k)
            for (int a = 0; a < n; ++a)
                for (int b = 0; b < n; ++b)
                    if (distances[a][k] + distances[k][b] < distances[a][b])
                        distances[a][b] = distances[a][k] + distances[k][b];
        
        vector<bool> rs;
        for (const auto& q : queries) {
            if (distances[q[0]][q[1]] != INT_MAX / 2)
                rs.push_back(true);
            else rs.push_back(false);
        }
        return rs;
    }
};
