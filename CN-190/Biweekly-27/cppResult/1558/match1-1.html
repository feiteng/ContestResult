class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& p, vector<vector<int>>& q) {
        vector<vector<int>> dist(n, vector<int>(n, 1000000));
        for (auto &item : p)
        {
            dist[item[0]][item[1]] = 1;
        }
        
        for (int k = 0; k < n; k++)
        {
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (dist[i][k] + dist[k][j] < dist[i][j])
                    {
                        dist[i][j] = dist[i][k] + dist[k][j];
                    }
                }
            }
        }
        
        vector<bool> sol;
        for (auto &item : q)
        {
            if (dist[item[0]][item[1]] == 1000000)
            {
                sol.push_back(false);
            }
            else
            {
                sol.push_back(true);
            }
        }
        
        return sol;
    }
};
