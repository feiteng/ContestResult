class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& pre, vector<vector<int>>& q) {
        vector<vector<bool>> graph(n, vector<bool>(n, false));
        
        for (vector<int> &e : pre) {
            graph[e[0]][e[1]] = true;
        }
        
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (!graph[i][j]) {
                        graph[i][j] = graph[i][k] & graph[k][j];
                    }
                }
            }
        }
        
        vector<bool> ans;
        for (vector<int> &e : q) {
            if (graph[e[0]][e[1]]) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
