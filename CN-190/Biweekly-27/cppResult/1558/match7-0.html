class Solution {
public:
  void dfs(vector<vector<int>>&ed , set<int>& dep,int s,int* vis){
        vis[s]=1;
        for(int i:ed[s]){
            if(!vis[i])
                dfs(ed,dep,i,vis);
        }
        dep.insert(s);
    }
    
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& p, vector<vector<int>>& q) {
       vector<vector<int>> ed(n);
       vector<set<int>> qu;
       vector<bool> ans;
        for(auto i: p){
           ed[i[1]].push_back(i[0]);
        }

        for(int i=0;i<n;i++){
            set<int> dep;
            int* vis =new int[n]();
            dfs(ed,dep,i,vis);
            qu.push_back(dep);
        }
        for(auto i:q){
            if(qu[i[1]].find(i[0])!=qu[i[1]].end())
                ans.push_back(true);
            else
                ans.push_back(false);
        }
        return ans;
    
    }
};
