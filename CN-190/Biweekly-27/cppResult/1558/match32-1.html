class Solution {
    bool a[111][111];
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) a[i][j]=0;
        for(const auto &P:prerequisites) a[P[0]][P[1]]=1;

        for(int k=0;k<n;k++) for(int i=0;i<n;i++) for(int j=0;j<n;j++) a[i][j]|=a[i][k]&a[k][j];
        vector<bool> ans;
        for(const auto &P:queries) ans.push_back(a[P[0]][P[1]]);
        return ans;
    }
};
