class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        
        vector< vector< int > > Edge( n, vector< int >( n, 1000000 ) );
        for( auto x: prerequisites ) Edge[ x[ 0 ] ][ x[ 1 ] ] = 1;
        
        for( int i = 0; i < n; i ++ )
        for( int j = 0; j < n; j ++ )
        for( int k = 0; k < n; k ++ )
        {
            Edge[ j ][ k ] = min( Edge[ j ][ k ], Edge[ j ][ i ] + Edge[ i ][ k ] ); 
        }
        
        vector< bool > goal;
        for( auto x: queries )
        {
            if( Edge[ x[ 0 ] ][ x[ 1 ] ] >= 1000000 ) goal.push_back( false );
            else goal.push_back( true );
        }
        return goal;
    }
};
