class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        int dp[111][111];
        memset(dp, 0, sizeof(dp));
        for(auto vec :prerequisites) {
            int u = vec[0] + 1;
            int v = vec[1] + 1;

            dp[u][v] = 1;
        }
        
        for(int k=1; k<=n; k++)
        {
            for(int i=1; i<=n; i++)
            {
                for(int j=1; j<=n; j++)
                {
                    if(dp[i][k] && dp[k][j])
                        dp[i][j] = 1;
                }
            }
        }
        vector<bool> ans;
        for(auto vec : queries) 
        {
            int u = vec[0] + 1;
            int v = vec[1] + 1;
            if(dp[u][v]) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;

        
        
    }
};
