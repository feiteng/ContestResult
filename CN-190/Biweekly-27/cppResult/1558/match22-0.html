class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        int q = queries.size();
        vector<bool> solution;
        
        bool pre[n][n];
        memset(pre, false, sizeof pre);
        
        for(auto i : prerequisites) {
            pre[i[0]][i[1]] = true;
        }
        
        
        for(int k = 0;k < n;k++) {
            for(int i = 0;i < n;i++) {
                for(int j = 0;j < n;j++) {
                    if (pre[i][k] == true and pre[k][j] == true) {
                        pre[i][j] = true;
                    }
                }
            }
        }
        for(auto i : queries) {
            solution.push_back(pre[i[0]][i[1]]);
        }
        return solution;
    }
};
