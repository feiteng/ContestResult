class Solution {
    bool g[111][111];
    const int n = 111;
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& edges, vector<vector<int>>& queries) {
        memset(g, 0, sizeof g);

        for (auto &e : edges) {
            g[e[0]][e[1]] = true;
        }
        for (int k = 0; k < n; ++k) {
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    g[i][j] |= g[i][k] & g[k][j];
                }
            }
        }
        vector<bool> ans;
        ans.reserve(queries.size());
        for (auto &q : queries) {
            ans.push_back(g[q[0]][q[1]]);
        }
        return ans;
    }
};
