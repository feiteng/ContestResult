class Solution {
private:
    bool dp[105][105];
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        memset(dp, false, sizeof(dp));
        for (auto it : prerequisites)
            dp[it[0]][it[1]] = true;
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < n; ++ j)
                for (int k = 0; k < n; ++ k)
                    dp[j][k] = dp[j][k] || (dp[j][i] && dp[i][k]);
        vector<bool> res;
        for (auto it : queries)
            res.push_back(dp[it[0]][it[1]]);
        return res;
    }
};
