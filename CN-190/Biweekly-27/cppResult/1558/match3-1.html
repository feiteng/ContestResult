class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& pre, vector<vector<int>>& q) {
        int req[111][111];
        memset(req,0,sizeof(req));

        for(int i=0;i<pre.size();i++){
            req[pre[i][0]][pre[i][1]]=1;
        }
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                for(int k=0;k<n;k++){
                    if(req[j][i]&&req[i][k]){
                        req[j][k]=1;
                    }
                }
            }
        }
        vector<bool> ans;
        for(auto a:q){
            if(req[a[0]][a[1]]){
                ans.push_back(true);
            } else{
                ans.push_back(false);
            }
        }
        return ans;
    }
};
