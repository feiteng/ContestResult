class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<bool>ans;
        
        vector<vector<bool>>dist(n,vector<bool>(n));
        for(auto x:prerequisites){
            dist[x[0]][x[1]]=true;
        }
        for(int k=0;k<n;k++){
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    if(dist[i][k] && dist[k][j])dist[i][j]=true;
                }
            }
        }
        for(auto qu:queries){    
            ans.push_back(dist[qu[0]][qu[1]]);
        }
        return ans;
    }
};
