class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for(auto v : prerequisites) dp[v[0]][v[1]] = 1;
        for(int _ = 0; _ < 3; _ += 1)
        for(int i = 0; i < n; i += 1)
            for(int j = 0; j < n; j += 1)
                for(int k = 0; k < n; k += 1)
                    dp[i][j] |= dp[i][k] and dp[k][j];
        vector<bool> res;
        for(auto v : queries) res.push_back(dp[v[0]][v[1]]);
        return res;
    }
};
