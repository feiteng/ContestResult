class Solution {
public:

    void dfs(vector<vector<int>>& v,vector<bool>& vis, int i) {
        if(vis[i]) return;
        vis[i] = true;
        for(auto it:v[i]) {
            dfs(v,vis,it);
        }
    }
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<int>> v(n);
        for(auto it:prerequisites) {
            v[it[1]].push_back(it[0]);
            
        }
        vector<vector<bool>> poss(n,vector<bool>(n,false));
        vector<bool> ans;
        vector<bool> vis;

        for(int i=0;i<n;i++) {
            dfs(v,poss[i],i);
        }
        for(auto it:queries) {
            //vis = vector<bool> (n,false);
            //dfs(v,vis,it[1]);
            //ans.push_back(vis[it[0]]);
            ans.push_back(poss[it[1]][it[0]]);
        }
        return ans;
    }
};
