int mat[101][101];

class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int> >& edges, vector<vector<int> >& queries) {
        
        memset(mat, 9999, sizeof(mat));
        
        for(int i=0;i<n;i++)
            mat[i][i]=0;
        
        for(int i=0;i<edges.size();i++)
        {
            int src = edges[i][0];
            int dest = edges[i][1];
            
            mat[src][dest] = 1;
            //mat[dest][src] = 1e9;
        }
        
        for (int k = 0; k < n; k++)  
        {  
            for (int i = 0; i < n; i++)  
            {  
                for (int j = 0; j < n; j++)  
                {   
                    if (mat[i][k] + mat[k][j] < mat[i][j])  
                        mat[i][j] = mat[i][k] + mat[k][j];  
                }  
            }  
        }
        
        vector<bool> ans;
        
        for(int i=0;i<queries.size();i++)
        {
            int src = queries[i][0];
            int dest = queries[i][1];
            
            
            if(mat[src][dest]<9999 && mat[src][dest]>0)
                ans.push_back(true);
            else
                ans.push_back(false);
            
        }
        return ans;
        
    }
};
