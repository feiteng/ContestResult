const ll MOD=1e9+7;
class Solution {
public:
  int dp[111][111][111];
  int cherryPickup(vector<vector<int>>& g) {
    int N=g.size(),M=g[0].size();
    for(int j=0;j<M;j++)

      for(int k=0;k<M;k++)
        dp[0][j][k]=-12341234;
    dp[0][0][M-1]=g[0][0]+g[0][M-1];
    for(int i=1;i<N;i++) {
      for(int j=0;j<M;j++) {
        for(int k=0;k<M;k++) {
            dp[i][j][k]=-12341234;

          for(int jj=j-1;jj<=j+1;jj++){
            for(int kk=k-1;kk<=k+1;kk++){
              if (jj>=0&&kk>=0&&jj<M&&kk<M) {
                dp[i][j][k] = max(dp[i-1][jj][kk],dp[i][j][k]);
              }
            }
          }
          if (j==k) dp[i][j][k]+=g[i][j];

          else dp[i][j][k]+=g[i][j]+g[i][k];
        }
      }
    }
    int res=0;
    for(int j=0;j<M;j++){
      for(int k=0;k<M;k++){
        res=max(res,dp[N-1][j][k]);
      }
    }
    return res;
  }
};
