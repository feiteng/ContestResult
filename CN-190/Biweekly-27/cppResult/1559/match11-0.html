class Solution {
public:
    int g[80][80];
    int dp[80][80][80];
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        for (int i=0;i<80;i++){
            for (int j=0;j<80;j++){
                for (int k=0;k<80;k++){
                    dp[i][j][k]=-1*100;
                }
            }
        }
        dp[0][0][m-1] = grid[0][0]+grid[0][m-1];
        for(int i=1;i<n;i++){
            for (int j=0;j<m;j++){
                for (int k=0;k<m;k++){
                    if (dp[i-1][j][k]==-1*100) continue;
                    for (int diffj=-1;diffj<=1;diffj++){
                        if (j+diffj<0||j+diffj>=m) continue;
                        for (int diffk=-1;diffk<=1;diffk++){
                            if (k+diffk<0||diffk+k>=m) continue;
                            //cout<<"k+diffk="<<k+diffk<<" "<<j+diffj<<endl;
                            //cout<<"i="<<dp[i-1][j][k]<<" "<<grid[i][j+diffj]<<" "<<grid[i][k+diffk]<<endl;
                            int sum =  dp[i-1][j][k]+grid[i][j+diffj]+grid[i][k+diffk];
                            if (j+diffj==k+diffk) sum-=grid[i][j+diffj];
                            dp[i][j+diffj][k+diffk] = max(dp[i][j+diffj][k+diffk],sum);
                        }
                    }
                }
            }
        }
        int Max = -1*100;
        for(int i=0;i<m;i++){
            for (int j=0;j<m;j++){
                Max = max(Max,dp[n-1][i][j]);
            }
        }
        return Max;
    }
};
