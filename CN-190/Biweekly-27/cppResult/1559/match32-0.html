class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int width = grid[0].size();
        vector<vector<int>> dp(width, vector<int>(width, -1));
        dp[0][width - 1] = grid[0][0] + grid[0][width - 1];
        
        for (int i = 1; i < grid.size(); ++i) {
            vector<vector<int>> next(width, vector<int>(width, -1));
            for (int robot1 = 0; robot1 < width; ++robot1) {
                for (int robot2 = 0; robot2 < width; ++robot2) {
                    if (dp[robot1][robot2] == -1) {
                        continue;
                    }
                    for (int move_robot1 = -1; move_robot1 <= 1; ++move_robot1) {
                        for (int move_robot2 = -1; move_robot2 <= 1; ++move_robot2) {
                            int next_pos_robot1 = robot1 + move_robot1;
                            int next_pos_robot2 = robot2 + move_robot2;
                            if (next_pos_robot1 < 0 || next_pos_robot1 >= width
                                || next_pos_robot2 < 0 || next_pos_robot2 >= width) {
                                continue;
                            }
                            if (next_pos_robot1 == next_pos_robot2) {
                                next[next_pos_robot1][next_pos_robot2] = max(dp[robot1][robot2] + grid[i][next_pos_robot1], next[next_pos_robot1][next_pos_robot2]);
                            } else {
                                next[next_pos_robot1][next_pos_robot2] = max(dp[robot1][robot2] + grid[i][next_pos_robot1] + grid[i][next_pos_robot2], next[next_pos_robot1][next_pos_robot2]);
                            }
                        }
                    }
                }
            }
            swap(next, dp);
        }
        int max_cnt = 0;
        for (int i = 0; i < width; ++i) {
            for (int j = 0; j < width; ++j) {
                max_cnt = max(dp[i][j], max_cnt);
            }
        }
        return max_cnt;
    }
};
