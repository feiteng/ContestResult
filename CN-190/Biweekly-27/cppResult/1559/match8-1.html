class Solution {
public:
    static const int maxn = 80;
    int dp[maxn][maxn][maxn];
    int cherryPickup(vector<vector<int>>& grid) {
        memset(dp, -1, sizeof(dp));
        int n = grid.size(), m = grid[0].size();

        dp[0][0][m-1] = grid[0][0]+grid[0][m-1];
        for(int i = 0; i < n-1; i++) {
            for(int u = 0; u < m; u++) {
                for(int v = u+1; v < m; v++) {
                    for(int s = -1; s <= 1; s++) {
                        for(int t = -1; t <= 1; t++) {
                            if(u+s >= 0 && u+s < m && v+t >= 0 && v+t < m && u+s < v+t && dp[i][u][v] != -1) {
                                dp[i+1][u+s][v+t] = max(dp[i+1][u+s][v+t], dp[i][u][v]+grid[i+1][u+s]+grid[i+1][v+t]);
                                // printf("dp[%d][%d][%d]=%d\n", i+1, u+s, v+t, dp[i+1][u+s][v+t]);
                            }

                        }
                    }
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < m; i++) {
            for(int j = i+1; j < m; j++) {
                ans = max(ans, dp[n-1][i][j]);
            }
        }
        return ans;
    }
};
