int dp[71][71][71];
class Solution {
public:

    int solve(int x, int y1, int y2, vector<vector<int>>& grid){
        if(x == grid.size()) return 0;
        if(y1 < 0 || y1 == grid[0].size() || y2 < 0 || y2 == grid[0].size()) return 0;
        if(dp[x][y1][y2] != -1) return dp[x][y1][y2];
        int ans = 0;
        if(y1 == y2){
            ans = max(ans, grid[x][y1] + solve(x + 1, y1 + 1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1 + 1, y2 - 1, grid));
            ans = max(ans,grid[x][y1] + solve(x + 1, y1 + 1, y2, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1 - 1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1 - 1, y2 - 1, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1 - 1, y2, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + solve(x + 1, y1, y2 - 1, grid));
            ans = max(ans,grid[x][y1] + solve(x + 1, y1, y2, grid));
        }
        else{
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 + 1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 + 1, y2 - 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 + 1, y2, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 - 1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 - 1, y2 - 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1 - 1, y2, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1, y2 + 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1, y2 - 1, grid));
            ans = max(ans, grid[x][y1] + grid[x][y2] + solve(x + 1, y1, y2, grid));
        }
        return dp[x][y1][y2] = ans;
    }
    int cherryPickup(vector<vector<int>>& grid){
        int n = grid.size();
        int m = grid[0].size();
        memset(dp, -1, sizeof(dp));
        return solve(0, 0, m - 1, grid);
    }
};
