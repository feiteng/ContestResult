class Solution {
    int dp[71][71][71];
public:
    int cherryPickup(vector<vector<int>>& grid) {
        memset(dp,0,sizeof(dp));
        int rn = grid.size();
        int cn = grid[0].size();
        dp[1][1][cn] = grid[0][0] + grid[0][cn-1];
        for(int i1=2;i1<=rn;i1++)
        {
            for(int j1=1;j1<=min(i1,cn);j1++)
            {
                for(int j2=cn;j2>=max(1,cn-i1+1);j2--)
                {
                    int val;
                    if(j1==j2)
                        val = grid[i1-1][j1-1];
                    else
                        val = grid[i1-1][j1-1] + grid[i1-1][j2-1];
                    for(int os1=-1;os1<=1;os1++)
                    {
                        if(j1+os1==0||j1+os1>cn)
                            continue;
                        for(int os2=-1;os2<=1;os2++)
                        {
                            if(j2+os2==0||j2+os2>cn)
                                continue;
                            dp[i1][j1][j2] = max(dp[i1][j1][j2],dp[i1-1][j1+os1][j2+os2]+val);
                        }
                    }
                }
            }
        }
        int maxx = 0;
        for(int j1=1;j1<=cn;j1++)
        {
            for(int j2=1;j2<=cn;j2++)
            {
                maxx = max(dp[rn][j1][j2],maxx);
            }
        }
        return maxx;
    }
};
