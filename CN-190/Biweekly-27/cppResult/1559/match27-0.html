class Solution {
    int dp[71][71][71];
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int row = grid.size();
        int col = grid[0].size();
        int res = 0;
        memset(dp,-1,sizeof(dp));
        
        dp[0][0][col-1] = grid[0][0] + grid[0][col-1];
        for(int i = 1; i < row; ++i){
            for(int j = 0; j < col; ++j){
                for(int k = 0; k < col; ++k){
                    for(int l1 = -1; l1 <= 1; ++l1){
                        for(int l2 = -1; l2 <= 1; ++l2){
                            int x = j + l1;
                            int y = k + l2;
                            if(x >= 0 && x < col && y >= 0 && y < col && dp[i-1][x][y] >= 0){
                                dp[i][j][k] = max(dp[i][j][k],dp[i-1][x][y]);
                            }
                        }
                    }
                    if(dp[i][j][k] >= 0){
                        if(j == k){
                            dp[i][j][k] += grid[i][j];
                        }else{
                            dp[i][j][k] += grid[i][j] + grid[i][k];
                        }
                    }
                }
            }
        }
        
        for(int i = 0; i < col; ++i){
            for(int j = 0; j < col; ++j){
                res = max(res,dp[row-1][i][j]);
            }
        }
        
        return res;
    }
};
