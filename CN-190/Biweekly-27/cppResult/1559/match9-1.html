class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {

        vector<vector<int>> pre(grid[0].size(), vector<int>(grid[0].size(), -100000));
        pre[0][grid[0].size() - 1] = grid[0][0] + grid[0][grid[0].size() - 1];
        for (int r = 1; r < grid.size(); r++) {
            vector<vector<int>> dp(grid[0].size(), vector<int>(grid[0].size(), 0));
            for (int i = 0; i < grid[0].size(); i++) {
                for (int j = 0; j < grid[0].size(); j++) {

                    if (i != j) {
                        dp[i][j] = grid[r][i] + grid[r][j];
                    } else {
                        dp[i][j] = grid[r][i];
                    }
                    int max_pre = -1;
                    for (int ii = max(i - 1, 0); ii <= i + 1 and ii < grid[0].size(); ii++) {
                        for (int jj = max(j - 1, 0); jj <= j + 1 and jj < grid[0].size(); jj++) {
                            if (pre[ii][jj] >= 0)

                                max_pre = max(max_pre, pre[ii][jj]);
                        }
                    }
                    if (max_pre == -1) {
                        dp[i][j] = -1;
                    } else {
                        dp[i][j] += max_pre;
                    }
                    
                }
            }
            pre = dp;
        }
        
        int res = INT_MIN;
        for (int i = 0; i < grid[0].size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                res = max(res, pre[i][j]);
            }
        }
        return res;
    }
};
