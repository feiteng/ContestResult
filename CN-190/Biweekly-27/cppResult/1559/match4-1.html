class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int r = grid.size();
        int c = grid[0].size();
        vector<vector<vector<int>>> dp(r, vector<vector<int>>(c, vector<int>(c, INT_MIN)));

        dp[0][0][c - 1] = grid[0][0] + grid[0][c - 1];
        for (int i = 1; i < r; i++) {
            for (int j = 0; j < c; j++) {
                for (int k = 0; k < c; k++) {
                    int num = dp[i - 1][j][k];

                    for (int p = max(0, j - 1); p <= min(c - 1, j + 1); p++) {
                        for (int q = max(0, k - 1); q <= min(c - 1, k + 1); q++) {
                            num = max(num, dp[i - 1][p][q]);
                        }
                    }
                    dp[i][j][k] = num + (j == k ? grid[i][j] : grid[i][j] + grid[i][k]);

                }
            }
        }
        int ans = 0;
        for (int j = 0 ; j < c; j++) {
            for (int k = 0; k < c; k++) {
                ans = max(ans, dp[r - 1][j][k]);
            }
        }
        return ans;
    }
};
