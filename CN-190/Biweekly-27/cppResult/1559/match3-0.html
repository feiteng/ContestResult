class Solution {
public:
    int cherryPickup(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<vector<int>>> dp(n, vector<vector<int>>(m, vector<int>(m, -1000000000)));
        vector<vector<vector<bool>>> ok(n, vector<vector<bool>>(m, vector<bool>(m, false)));
        dp[0][0][m - 1] = a[0][0] + a[0][m - 1];
        ok[0][0][m - 1] = true;
        
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = 0; k < m; ++k) {
                    int x = 0;
                    for (int dx = -1; dx <= 1; ++dx) {
                        for (int dy = -1; dy <= 1; ++dy) {
                            int j1 = j + dx;
                            int k1 = k + dy;
                            if (j1 < 0 || j1 >= m || k1 < 0 || k1 >= m) continue;
                            if (!ok[i - 1][j1][k1]) continue;
                            ok[i][j][k] = true;
                            x = max(x, dp[i - 1][j1][k1]);
                        }
                    }
                    int y = a[i][j] + a[i][k];
                    if (j == k) y -= a[i][k];
                    dp[i][j][k] = x + y;
                }
            }
        }
        int ans = 0;
         for (int j = 0; j < m; ++j) {
            for (int k = 0; k < m; ++k) {
                ans = max(ans, dp[n - 1][j][k]);
            }
         }
        
        return ans;
        
    }
};
