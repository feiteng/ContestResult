class Solution {
    const int NINF=-1000000000;
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n=grid.size(),m=grid[0].size();
        vector<vector<int>> dp(m,vector<int>(m,NINF));
        dp[0][m-1]=grid[0][0]+grid[0][m-1];
        for(int t=1;t<n;t++){
            vector<vector<int>> dp2(m,vector<int>(m,NINF));
            for(int i=0;i<m;i++)
                for(int ii=i-1;ii<=i+1;ii++){
                    if(ii==-1||ii==m)continue;
                    for(int j=0;j<m;j++)
                        for(int jj=j-1;jj<=j+1;jj++){
                            if(jj==-1||jj==m)continue;
                            int val=dp[ii][jj]+grid[t][i];
                            if(i!=j)val+=grid[t][j];
                            dp2[i][j]=max(dp2[i][j],val);
                        }
                }
            dp=dp2;
        }
        int ans=0;
        for(int i=0;i<m;i++)
            for(int j=0;j<m;j++)
                ans=max(ans,dp[i][j]);
        return ans;
    }
};
