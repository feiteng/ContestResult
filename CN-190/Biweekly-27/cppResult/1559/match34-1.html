class Solution {
public:
    int dp[75][75][75];
    const int inf = 1e9;
    int cherryPickup(vector<vector<int>>& grid) {
        for(int i = 0; i < 75; i++) {
            for(int j = 0; j < 75; j++) {
                for(int k = 0; k < 75; k++) {
                    dp[i][j][k] = -inf;
                }
            }
        }

        int r = grid.size(), c = grid[0].size();
        dp[0][0][c-1] = grid[0][0] + grid[0][c - 1];
        for(int i = 0; i < r - 1; i++) {
            for(int j = 0; j < c; j++) {
                for(int k = 0; k < c; k++) {
                    if( dp[i][j][k] != -inf ) {

                        for(int l1 = -1; l1 <= 1; l1++) {
                            for(int l2 = -1; l2 <= 1; l2++) {
                                if( j + l1 >= 0 && j + l1 < c && k + l2 >= 0 && k + l2 < c ) {
                                    dp[ i + 1 ][j + l1 ][k + l2] = max( dp[ i + 1 ][j + l1 ][k + l2], dp[i][j][k] + grid[i + 1][j + l1] + grid[i + 1][k + l2] - ( j + l1 == k + l2 ) * grid[i + 1][j + l1] );
                                }
                            }
                        }
                    }

                }
            
            }
        }
        int mx = 0;
        for(int j = 0; j < c; j++) {
                for(int k = 0; k < c; k++) {
                        mx = max( mx, dp[r - 1][j][k]);
                    }
                }
        return mx;
    }
};
