class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid)
    {
        int H = grid.size();
        int W = grid[0].size();
        const int inf = 1 << 9;
        vector<vector<vector<int>>> dp(
            H + 1, vector<vector<int>>(W, vector<int>(W, -inf)));

        dp[0][0][W - 1] = grid[0][0] + grid[0][W - 1];
        for (int h = 1; h < H; ++h) {
            for (int w1 = 0; w1 < W; ++w1) {
                for (int w2 = w1; w2 < W; ++w2) {
                    for (int d1 = max(0, w1 - 1); d1 <= min(W - 1, w1 + 1);
                         ++d1) {
                        for (int d2 = max(0, w2 - 1); d2 <= min(W - 1, w2 + 1);
                             ++d2) {
                            if (w1 != w2) {
                                dp[h][w1][w2] =
                                    max(dp[h][w1][w2], dp[h - 1][d1][d2] +
                                                           grid[h][w1] +
                                                           grid[h][w2]);
                            }
                            else {
                                dp[h][w1][w2] =
                                    max(dp[h][w1][w2], dp[h - 1][d1][d2]);
                            }
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int w1 = 0; w1 < W; ++w1) {
            for (int w2 = w1; w2 < W; ++w2) {
                ans = max(ans, dp[H - 1][w1][w2]);
            }
        }
        return ans;
    }
};
