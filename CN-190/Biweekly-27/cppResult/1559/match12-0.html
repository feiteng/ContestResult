class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        int dp[80][80][80];
        memset(dp,-1,sizeof dp);
        dp[0][0][m-1]=grid[0][0]+grid[0][m-1];
        vector<int>dir={-1,0,1};
        for(int i=1;i<n;i++)
            for(int j=0;j<m;j++)
                for(int k=0;k<m;k++)
                {
                    for(int a=0;a<3;a++)
                        for(int b=0;b<3;b++)
                        {
                            int nj=j+dir[a];
                            int nk=k+dir[b];
                            if(nj>=0&&nj<m&&nk>=0&&nk<m&&dp[i-1][nj][nk]!=-1)
                                dp[i][j][k]=max(dp[i][j][k],dp[i-1][nj][nk]);            
                        }
                    if(dp[i][j][k]==-1)continue;
                    dp[i][j][k]+=(grid[i][j]+grid[i][k]);
                    if(j==k)
                        dp[i][j][k]-=grid[i][j];
                }
        int max_=0;
        for(int i=0;i<m;i++)
            for(int j=0;j<m;j++)
                max_=max(max_,dp[n-1][i][j]);
        return max_;
    }
};
