class Solution {
public:
    int cherryPickup(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector<vector<int>> d(m, vector<int>(m, -1));

        vector<vector<int>> e(m, vector<int>(m));
        d[0][m - 1] = a[0][0] + a[0][m - 1];
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = j + 1; k < m; ++k) {
                    e[j][k] = -1;
                }
            }

            for (int j = 0; j < m; ++j) {
                for (int k = j + 1; k < m; ++k) {
                    if (d[j][k] < 0) continue;
                    for (int u = -1; u <= 1; ++u) {
                        if (j + u < 0 || j + u >= m) continue;
                        for (int v = -1; v <= 1; ++v) {
                            if (k + v < 0 || k + v >= m) continue;
                            int cur = a[i][j + u] + a[i][k + v];
                            if (j + u == k + v) cur -= a[i][j + u];
                            e[j + u][k + v] = max(e[j + u][k + v], d[j][k] + cur);
                        }
                    }
                }
            }
            swap(d, e);
        }
        int ret = 0;
        for (int j = 0; j < m; ++j) {
            for (int k = j + 1; k < m; ++k) {
                ret = max(ret, d[j][k]);
            }
        }
        return ret;
    }
};
