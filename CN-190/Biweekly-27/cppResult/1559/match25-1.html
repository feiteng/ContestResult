class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {

        int m = grid.size(), n = grid[0].size();
        vector<vector<vector<int>>> dp(m, vector<vector<int>>(n, vector<int>(n, INT_MIN)));
        dp[0][0][n-1] = grid[0][0] + grid[0][n-1];
        for(int i = 1; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                for(int k = 0; k < n; ++k) {

                    for(int s1 = -1; s1 <=1; ++s1) {
                        if(j + s1 <0 || j + s1 >= n) continue;
                        for(int s2 = -1; s2 <=1; ++s2) {
                            if(k + s2 < 0 || k + s2 >=n) continue;
                            if(dp[i-1][j+s1][k+s2] == INT_MIN) continue;
                            dp[i][j][k] = max(dp[i][j][k], dp[i-1][j+s1][k+s2]);
                        }
                    }
                    
                    if(dp[i][j][k] > INT_MIN) {
                        dp[i][j][k] += (k == j? grid[i][j] : (grid[i][j] + grid[i][k]));
                    }

                }
            }
        }
        
        int res = 0;
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                res = max(res, dp[m-1][i][j]);
            }
        }
        return res;
    }
};
