class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        int dp[n][m][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < m; k++) {
                    dp[i][j][k] = -1;
                }
            }
        }
        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < m; k++) {
                    dp[i][j][k] = -1;
                    for (int x = -1; x <= 1; x++) {
                        for (int y = -1; y <= 1; y++) {
                            int xj = j + x, yk = k + y;
                            if (xj >= 0 && xj < m && yk >=0 && yk < m && dp[i-1][xj][yk] >= 0) {
                                int tmp = dp[i-1][xj][yk] + grid[i][j] + grid[i][k];
                                if (j == k) tmp -= grid[i][j];
                                dp[i][j][k] = max(dp[i][j][k], tmp);
                            }
                        }
                    }
                }
            }
        }
        int res = 0;
        for (int j = 0; j < m; j++) {
            for (int k = 0; k < m; k++) {
                res = max(res, dp[n-1][j][k]);
            }
        }
        return res;
    }
};
