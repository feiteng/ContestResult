class Solution {
    int getVal(vector <vector <vector <int>>> &dp, int row,  int col1, int col2){
        int result = -1;
        int cols = dp[0].size();
        for(int i = -1; i <=1; ++i){
            for(int j = -1; j <=1; ++j){
                if(col1 + i < 0 || col1 + i >=cols || col2 + j < 0 || col2 + j >=cols )
                    continue;
                result = max(result, dp[row][col1+i][col2+j]);
            }
        }
        return result;
    }
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int rows = grid.size(), cols = grid[0].size();
        vector <vector <vector <int>>> dp(rows, vector <vector <int>>(cols, vector <int>(cols, -1)));
        dp[0][0][cols - 1] = grid[0][0] + grid[0][cols - 1];
        for(int i = 1; i < rows; ++i){
            for(int c1 = 0; c1 < cols; ++c1){
                for(int c2 = 0; c2 < cols; ++c2){
                    int val = getVal(dp, i-1, c1, c2);
                    if(val == -1)
                        continue;
                    dp[i][c1][c2] = val + grid[i][c1] + grid[i][c2];
                    if(c1 == c2)
                        dp[i][c1][c2] -= grid[i][c1];
                }
            }
        }
        int result = 0;
        for(int i = 0; i < cols; ++i){
            for(int j = 0; j < cols; ++j){
                result = max(result, dp[rows - 1][i][j]);
            }
        }
        return result;
    }
};
