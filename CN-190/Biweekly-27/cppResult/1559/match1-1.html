class Solution {
	int f[71][71][71];

public:

    int cherryPickup(vector<vector<int>>& a) {
    	int n = a.size(), m = a[0].size();
        memset(f, 0x8f, sizeof f);
        f[0][0][m - 1] = a[0][0] + a[0][m - 1];
        for (int i = 1; i < n; ++i) {
        	for (int l = 0; l < m - 1; ++l)
        		for (int r = l + 1; r < m; ++r) {

        			for (int dx = -1; dx <= 1; ++dx)
        				for (int dy = -1; dy <= 1; ++dy) {
        					int x = l + dx, y = r + dy;
        					if (!(x >= 0 && x < m && y >= 0 && y < m)) continue;
        					f[i][l][r] = max(f[i][l][r], f[i - 1][x][y]);
        				}
        			f[i][l][r] += a[i][l] + a[i][r];

        		}
        }
        int ans = 0;
        for (int l = 0; l < m - 1; ++l)
    		for (int r = l + 1; r < m; ++r)
    			ans = max(ans, f[n - 1][l][r]);
    	return ans;
    }
};
