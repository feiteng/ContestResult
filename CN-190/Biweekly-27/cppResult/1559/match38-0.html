class Solution {
private:
    int dp[77][77][77];
public:
    int cherryPickup(vector<vector<int>>& grid) {
        memset(dp, 0xc1, sizeof(dp));
        dp[1][1][grid[0].size()] = grid[0][0] + grid[0][grid[0].size() - 1];
        for (int i = 1; i <= grid.size(); ++ i) {
            for (int j = 1; j <= grid[0].size(); ++ j) {
                for (int k = j + 1; k <= grid[0].size(); ++ k) {
                    for (int p = -1; p <= 1; ++ p) {
                        for (int q = -1; q <= 1; ++ q) {
                            dp[i][j][k] = max(dp[i][j][k], dp[i - 1][j + p][k + q] + grid[i - 1][j - 1] + grid[i - 1][k - 1]);
                        }
                    }
                }
            }
        }
        int res = 0;
        for (int i = 1; i <= grid[0].size(); ++ i) {
            for (int j = i + 1; j <= grid[0].size(); ++j) {
                res = max(res, dp[grid.size()][i][j]);
            }
        }
        return res;
    }
};
