class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int N=grid.size();
        int M=grid[0].size();
        vector<vector<vector<int>>> dp(N+1,vector<vector<int>>(M+2,vector<int>(M+2,-1)));


        dp[1][1][M]=grid[0][0]+grid[0][M-1];
        for(int r=1;r<N;r++){
            for(int i=0;i<M;i++){
                for(int j=0;j<M;j++){
                    
                    for(int a=-1;a<=1;a++){
                        for(int b=-1;b<=1;b++){
                            dp[r+1][i+1][j+1]=max(dp[r+1][i+1][j+1],dp[r][i+1+a][j+1+b]);
                        }
                    }
                    if(dp[r+1][i+1][j+1]<0){
                        continue;
                    }
                    dp[r+1][i+1][j+1]+=grid[r][i]+(i==j?0:grid[r][j]);
                }
            }
        }
        int ans=0;

        for(int i=0;i<M;i++){
            for(int j=0;j<M;j++){  
                ans=max(ans,dp[N][i+1][j+1]);
            }
        }
        
        return ans;
    }
};
