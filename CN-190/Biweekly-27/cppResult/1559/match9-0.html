class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<int>> dp(n,vector<int>(n,-1));
        //base case;
        dp[0][n-1] = grid[0][0] + grid[0][n-1];
        
        
        
        for (int i = 1; i<m; ++i) {
            
            vector<vector<int>> newdp(n,vector<int>(n,-1));
            for (int j1=0; j1<n; ++j1) {
                for (int j2=j1; j2<n; ++j2) {
                    
                    // j1,j2 could have come from... up to 9 possibilities;
                    
                    int cand;
                    if (j1==j2) {
                        cand = grid[i][j1];
                    } else {
                        cand = grid[i][j1]+grid[i][j2];
                    }
                    
                    int bestprev=-1;
                    for (int k1 = j1-1; k1 <= j1+1; ++k1) {
                        for (int k2 = j2-1; k2<= j2+1; ++k2) {
                            if (k1>k2 || k1<0 || k1>=n || k2<0 || k2>=n) continue;
                            
                            if (dp[k1][k2]==-1) continue;
                            
                            bestprev = max(bestprev,dp[k1][k2]);
                        }
                    }
                    
                    if (bestprev==-1) {
                        newdp[j1][j2] = -1;
                    } else {
                        newdp[j1][j2] = cand + bestprev;
                    }
                }
            }
            
            dp=newdp;
            
        }
        
        
        
        
        
        int ans = 0;
        for (int j1 = 0; j1<n; ++j1) {
            for (int j2=j1; j2<n; ++j2) {
                ans = max(ans,dp[j1][j2]);
            }
        }
        
        
        return ans;
        
        
        
    }
};
