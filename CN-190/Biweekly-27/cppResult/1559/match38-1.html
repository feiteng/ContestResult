class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int dp[72][72][72];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < m; k++) {
                    dp[i][j][k] = -1e9;
                }
            }
        }

        dp[0][0][m-1] = grid[0][0] + grid[0][m-1]; 
        
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = j; k < m; k++) {
                    for (int nj = j - 1; nj <= j + 1; nj++) {
                        for (int nk = k - 1; nk <= k + 1; nk++) {
                            if (nj < 0 || nj >= m || nk < 0 || nk >= m || nk < nj) {
                                continue;
                            }
                            if (nj == nk) {
                                dp[i+1][nj][nk] = max(dp[i+1][nj][nk], dp[i][j][k] + grid[i+1][nj]);
                            } else {
                                dp[i+1][nj][nk] = max(dp[i+1][nj][nk], dp[i][j][k] + grid[i+1][nj] + grid[i+1][nk]);
                            }

                        }
                    }
                }
            }
        }
        int mx = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < m; j++) {
                mx = max(mx, dp[n-1][i][j]);
            }
        }
        return mx;
    }
};
