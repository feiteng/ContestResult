class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        int dp[n][m][m];
        fill(dp[0][0], dp[n][0], -1);

        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];

        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < m; k++) {
                    for (int d1 = j - 1; d1 <= j + 1; d1++) {
                        for (int d2 = k - 1; d2 <= k + 1; d2++) {
                            if (dp[i][j][k] != -1 && d1 >= 0 && d1 < m && d2 >= 0 && d2 < m) {
                                dp[i+1][d1][d2] = max(dp[i+1][d1][d2], dp[i][j][k] + (d1 == d2 ? grid[i+1][d1] : grid[i+1][d1] + grid[i+1][d2]));
                            }

                        }
                    }
                }
            }
        }

        int ret = 0;
        for (int j = 0; j < m; j++) {
            for (int k = 0; k < m; k++) {
                ret = max(ret, dp[n-1][j][k]);
            }
        }

        return ret;
    }
};
