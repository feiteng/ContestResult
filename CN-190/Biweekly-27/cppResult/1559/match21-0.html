class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        vector<int> d{-1, 0, 1};
        int r=grid.size(), c=grid[0].size();
        vector<vector<int>> dp(c, vector<int>(c, -1000000));
        dp[0][c-1]=grid[0][0]+grid[0][c-1];
        for (int i=1; i<r; ++i) {
            vector<vector<int>> nxt(c, vector<int>(c, -1000000));
            for (int j=0; j<c; ++j) {
                for (int k=0; k<c; ++k) {
                    for (int dj=0; dj<3; ++dj) {
                        for (int dk=0; dk<3; ++dk) {
                            int jj=j+d[dj], kk=k+d[dk];
                            if (jj>=0&&jj<c&&kk>=0&&kk<c) {
                                // cout << jj << "  " << kk << " " << j << " "<<k<< endl;
                                if (jj==kk) {
                                    nxt[jj][kk] = max(nxt[jj][kk], dp[j][k]+grid[i][kk]);
                                } else {
                                    nxt[jj][kk] = max(nxt[jj][kk], dp[j][k]+grid[i][jj]+grid[i][kk]);
                                }
                                // cout << jj << " " << kk << " " << dp[j][k] << " " << j << " " << k << " "<< nxt[jj][kk]<< endl;
                            }
                        }
                    }
                }
            }
            dp=nxt;
        }
        int res=0;
        for (int i=0; i<c; ++i) {
            for (int j=0; j<c; ++j) {
                // cout << i <<" "<<j<<" "<<dp[i][j]<<endl;
                res = max(res, dp[i][j]);
            }
        }
        return res;
    }
};
