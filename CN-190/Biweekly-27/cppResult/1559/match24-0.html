class Solution {
public:
    int dp[71][71][71];
    int dp2[71][71];
    int n, m;
    vector<vector<int>> v;
    int func(int x, int y, int q) {
        if (x >= n || y < 0 || y >= m || q < 0 || q >= m) return 0;
        if (q < y) return 0;
        if (dp[x][y][q]) return dp[x][y][q];
        
        int ans = 0;
        if (y == q) {
            ans = max(ans, v[x][y] + func(x+1, y-1, q-1));
            ans = max(ans, v[x][y] + func(x+1, y-1, q));
            ans = max(ans, v[x][y] + func(x+1, y-1, q+1));

            ans = max(ans, v[x][y] + func(x+1, y, q-1));
            ans = max(ans, v[x][y] + func(x+1, y, q));
            ans = max(ans, v[x][y] + func(x+1, y, q+1));

            ans = max(ans, v[x][y] + func(x+1, y+1, q-1));
            ans = max(ans, v[x][y] + func(x+1, y+1, q));
            ans = max(ans, v[x][y] + func(x+1, y+1, q+1));
        } else {
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y-1, q-1));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y-1, q));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y-1, q+1));

            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y, q-1));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y, q));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y, q+1));

            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y+1, q-1));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y+1, q));
            ans = max(ans, v[x][y] + v[x][q] + func(x+1, y+1, q+1));
        }
        return dp[x][y][q] = ans;
    }
    int cherryPickup(vector<vector<int>>& vv) {
        v = vv;
        n = v.size();
        m = v[0].size();
        return func(0, 0, m - 1);
    }
};
