class Solution {
public:
    int dp[105][105][105];
    int d[3]={-1,0,1};
    int cherryPickup(vector<vector<int>>& grid) {

        int n=grid.size(),m=grid[0].size();
        memset(dp,-1,sizeof dp);
        dp[0][0][m-1]=grid[0][0]+grid[0][m-1];
        for(int step=1;step<n;++step){
            for(int i=0;i<m;++i){
                for(int j=i;j<m;++j){
                    if(dp[step-1][i][j]==-1)continue;

                    for(int ii=0;ii<3;++ii){
                        for(int jj=0;jj<3;++jj){
                            int ti=i+d[ii];
                            int tj=j+d[jj];
                            if(ti<0||ti>=m||tj<0||tj>=m)continue;

                            if(ti<=tj){
                                if(ti==tj){
                                    dp[step][ti][tj]=max(dp[step][ti][tj],dp[step-1][i][j]+grid[step][ti]);
                                }else{
                                    dp[step][ti][tj]=max(dp[step][ti][tj],dp[step-1][i][j]+grid[step][ti]+grid[step][tj]);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        int res=0;

        for(int i=0;i<m;++i){
            for(int j=0;j<m;++j){
                res=max(res,dp[n-1][i][j]);
            }
        }
        
        return res;
    }
};
