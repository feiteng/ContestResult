int dp[75][75][75];
class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int r = grid.size(), c = grid[0].size();
        for(int i = 0; i < r; i++){
            for(int j = 0; j < c ; j++){
                for(int k = 0; k < c; k++){
                    dp[i][j][k] = -1;
                }
            }
        }
        dp[0][0][c - 1] = grid[0][0] + grid[0][c-1];
        for(int i = 1; i < r; i++){
            for(int j = 0; j < c ; j++){
                for(int k = j; k < c ;k++){
                    for(int p1 = -1; p1 <= 1; p1++){
                        for(int p2 = -1; p2<=1;p2++){
                            if(j+p1 < 0) continue;
                            if(k+p2 == c) continue;
                            if(k+p2 < p1+j) continue;
                            if(dp[i-1][j+p1][k+p2]==-1) continue;
                            dp[i][j][k] = max(dp[i][j][k],dp[i-1][j+p1][k+p2] + grid[i][j] + grid[i][k] - (j==k?grid[i][k]:0));
                        }
                    }
                    // cout<<i<<","<<j<<","<<k<<":"<<dp[i][j][k]<<endl;
                }
            }
        }
        int ans = -1;
        for(int j =0; j < c;j++){
            for(int k = j ; k< c;k++){
                ans = max(dp[r-1][j][k],ans);
            }
        }
        return ans;
        
    }
};
