class Solution {
public:
    int dp[100][100][100]; //走到第i行，s1->x ,s2->y 的最大方案数
    int vis[100][100][100];
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        memset(dp,0,sizeof(dp));
        memset(vis,0,sizeof(vis));
        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];
        vis[0][0][m-1] = 1;
        for(int i = 1; i < n; i++){
            for(int j = 0; j < m; j++){
                for(int k = 0; k < m; k++){
                    for(int c = -1; c <= 1; c++){
                        for(int d = -1; d <= 1; d++){
                            if(j+c >= 0 && j+c < m && k+d >= 0 && k+d < m && vis[i-1][j+c][k+d]){
                                
                                if(j == k) dp[i][j][k] = max(dp[i-1][j+c][k+d]+grid[i][j], dp[i][j][k]);
                                else dp[i][j][k] = max(dp[i-1][j+c][k+d]+grid[i][j]+grid[i][k], dp[i][j][k]);
                                vis[i][j][k] = 1;
                            }
                            
                        }
                    }
                    
                    
                }
            }
        }
        int ans = 0;
            for(int i = 0; i < m; i++){
                for(int j = 0; j < m; j++){
                    ans = max(ans,dp[n-1][i][j]);
                }
            }
        return ans;
    }
};
