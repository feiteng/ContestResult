class Solution {

public:
    int cherryPickup(vector<vector<int>>& grid) {
        int m=grid.size(), n=grid[0].size();
        vector<vector<int>> dp(n,vector<int>(n,-1e9)), nx(n,vector<int>(n,-1e9));
        dp[0][n-1]=grid[0][0]+grid[0][n-1];
        for(int i=1;i<m;++i){
            for(int j=0;j<n;++j){
                for(int k=j+1;k<n;++k){
                    int& now=nx[j][k];

                    now=-1e9;
                    for(int d1=j-1;d1<=j+1;++d1){
                        for(int d2=k-1;d2<=k+1;++d2){
                            if(d1<d2&&d1>=0&&d2<n) now=max(now,dp[d1][d2]);
                        }
                    }

                    now+=grid[i][j]+grid[i][k];
                }
            }
            swap(dp,nx);
        }
        int ans=0;
        for(int j=0;j<n;++j){
            for(int k=j+1;k<n;++k){
                //cout<<dp[j][k]<<" ";
                ans=max(ans,dp[j][k]); 
            }
            //cout<<endl;
        }
        return ans;
    }
};
