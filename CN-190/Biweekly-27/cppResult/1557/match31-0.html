class Solution {
public:
    bool hasAllCodes(string s, int k) {
        
        int n=pow(2,k-1);
        if(s.size()<2*n) return false;
        vector<bool> arr(2*n,true);
        int tmp=0;
        for(int i=0;i<k;i++){
            tmp=tmp*2+s[i]-'0';
        }
        arr[tmp]=false;
        for(int i=0;i<s.size()-k;i++){
            //cout<<tmp<<endl;
            tmp-=(s[i]-'0')*n;
            tmp*=2;
            tmp+=s[i+k]-'0';
            arr[tmp]=false;
        }
        
        
        for(int i=0;i<arr.size();i++) {if(arr[i]) return false;}
        return true;
    }
};
