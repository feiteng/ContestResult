class Solution {
public:
    bool hasAllCodes(string s, int k) {
        vector<int> v(1<<k);
        for (int i = 0; i + k - 1 < s.size(); i++) {
            int x = 0;
            for (int j = 0; j < k; j++) {
                x *= 2;
                x += s[i+j] - '0';
            }
            v[x]++;
        }
        for (int i = 0; i < (1<<k); i++) {
            if (v[i] == 0) return false;
        }
        return true;
    }
};
