class Solution {
public:
    bool hasAllCodes(string s, int k) {
        vector<bool> vis((1 << k), 0);
        int l = s.length();
        if (l < k) return 0;
        int rec = 0;
        for (int i = 0; i < k; ++i) {
            rec <<= 1;

            rec += s[i] - '0';
        }
        vis[rec] = 1;
        for (int i = k; i < l; ++i) {
            rec = rec & ((1 << (k - 1)) - 1);
            rec = rec << 1;
            rec += s[i] - '0';
            vis[rec] = 1;
        }
        for (int i = 0; i < (1 << k); ++i) {
            if (!vis[i]) return 0;
        }
        return 1;
    }
};
