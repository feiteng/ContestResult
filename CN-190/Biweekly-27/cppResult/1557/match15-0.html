class Solution {
public:
    bool hasAllCodes(string s, int k) {
      set<int> st;
      int n = s.size();
      for (int i = 0; i + k <= n; i++) {
        int w = 0;
        for (int j = 0; j < k; j++) {
          w = w * 2 + (s[i + j] - '0');
        }
        st.insert(w);
      }
      return st.size() == (1 << k);
    }
};
