class Solution(object):
    def checkIfPrerequisite(self, n, prerequisites, queries):
        adj = collections.defaultdict(list)
        for u,v in prerequisites:
            adj[u].append(v)
    
        return [self.bfs(u,v,adj) for u,v in queries]
    def bfs(self,u,v,adj):
        q = collections.deque([u])
        vis = set([])
        while(q):
            uu = q.popleft()
            if uu == v:
                return True
            vis.add(uu)
            
for vv in adj[uu]:
                if vv not in vis:
                    vis.add(vv)
                    q.append(vv)
        return False
