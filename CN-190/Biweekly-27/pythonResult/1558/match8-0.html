class Solution(object):
    def checkIfPrerequisite(self, n, prerequisites, queries):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        g = collections.defaultdict(list)
        for u, v in prerequisites:
            g[u] += v,
        
        def bfs(start, end):
            q = [start]
            visited = set()
            visited.add(start)
            while q:
                node = q.pop(0)
                if node == end:
                    return True
                for nei in g[node]:
                    if nei not in visited:
                        q.append(nei)
                        visited.add(nei)
            return False
        res = [bfs(start, end) for start, end in queries]
        return res
