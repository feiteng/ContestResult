from collections import defaultdict

class Solution(object):
    def checkIfPrerequisite(self, n, prerequisites, queries):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        graph = defaultdict(list)
        in_degree = defaultdict(int)
        for [u, v] in prerequisites:
            graph[u].append(v)
            in_degree[v] += 1
            
        hashtable = [None for i in range(n)]
        def dfs_children(node):
            if hashtable[node] is not None:
                return hashtable[node]
            ans = set([node])
            for nb in graph[node]:
                temp = dfs_children(nb)
                ans = ans.union(temp)
            hashtable[node] = ans
            return ans
        for i in range(n):
            if in_degree[i] == 0:
                dfs_children(i)
        ans = []
        for [u, v] in queries:
            
if hashtable[u] is not None and v in hashtable[u]:
                ans.append(True)
            else:
                ans.append(False)
        return ans
