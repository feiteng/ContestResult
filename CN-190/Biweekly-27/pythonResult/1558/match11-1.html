class Solution(object):
    def checkIfPrerequisite(self, n, prerequisites, queries):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        g = [[False] * n for _ in range(n)]
        for pre in prerequisites:
            g[pre[0]][pre[1]] = True
        for i in range(n):
            g[i][i] = True
        for k in range(n):
            for i in range
(n):
                for j in range(n):
                    if g[i][k] and g[k][j]:
                        g[i][j] = True
        ans = []
        for q in queries:
            ans.append(g[q[0]][q[1]])
        return ans
