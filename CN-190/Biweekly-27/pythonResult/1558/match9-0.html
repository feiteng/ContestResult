class Solution(object):
    def checkIfPrerequisite(self, n, p, q):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        ans = [set() for _ in xrange(n)]
        path = [[] for _ in xrange(n)]
        for pp in p:
            path[pp[1]].append(pp[0])
        def dfs(x, root):
            if v[x]:
                return
            ans[root].add(x)
            v[x] = True
            for pp in path[x]:
                dfs(pp, root)
        for i in xrange(n):
            v = [False] * n
            dfs(i, i)
        res = []
        for qq in q:
            res.append(qq[0] in ans[qq[1]])
        return res
