class Solution(object):
    def checkIfPrerequisite(self, n, prerequisites, queries):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        
        dist = [[False] * n for _ in xrange(n)]
        
        for preq in prerequisites:
            a, b = preq
            dist[a][b] = True
            #dist[b][a] = True
        
        for k in xrange(n):
            for i in xrange(n):
                for j in xrange(n):
                    if dist[i][k] and dist[k][j]:
                        dist[i][j] = True

        
        ans = []
        for query in queries:
            a, b = query
            if dist[a][b]:
                ans.append(True)
            else:
                ans.append(False)
        return ans
