class Solution(object):
    def checkIfPrerequisite(self, n, ps, qs):
        """
        :type n: int
        :type prerequisites: List[List[int]]
        :type queries: List[List[int]]
        :rtype: List[bool]
        """
        adj = [[] for _ in range(n)]
        deg = [0] * n
        for f, t in ps:
            adj[f].append(t)
            deg[t] += 1
        pars = [set() for _ in range(n)]
        que = collections.deque()
        for i in range(n):
            if deg[i] == 0:
                que.append(i)
        while que:
            cur = que.popleft()
            for nei in adj[cur]:
                deg[nei] -= 1
                pars[nei].add(cur)
                for a in pars[cur]:
                    pars[nei].add(a)
                if deg[nei] == 0:
                    que.append(nei)
        #print(pars)
        num = len(qs)
        ans = [False] * num
        
for i in range(num):
            if qs[i][0] in pars[qs[i][1]]:
                ans[i] = True
        return ans
