class Solution(object):
    def checkIfPrerequisite(self, n, p, q):
        z=[[1<<32]*n for _ in xrange(n)]
        for x,y in p:
            z
[x][y]=1
        for k in xrange(n):
            for i in xrange(n):
                for j in xrange(n):
                    z[i][j]=min(z[i][j],z[i][k]+z[k][j])
        return [z[x][y]<(1<<32) for x,y in q]
