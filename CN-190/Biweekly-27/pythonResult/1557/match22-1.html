class Solution(object):
    def hasAllCodes(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: bool
        """
        if len(s)+1<k+(1<<k):
            return False
        #print 1<<k
        A=[0]*(1<<k)
        s1=map(int,list(s))
        n=len(s1)   
        tmp=0
        p=1
        for i in range(k):
            tmp+=s1[i]*p
            p<<=1
        #print tmp
        A[tmp]+=1
        p>>=1
        for i in range(k,n):
            tmp>>=1
            tmp+=s1[i]*p
            #print tmp
            A[tmp]
+=1
        for i in range(1<<k):
            if A[i]==0:
                return False
        return True
