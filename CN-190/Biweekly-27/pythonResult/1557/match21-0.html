class Solution(object):
    def hasAllCodes(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: bool
        """
        a = set()
        n = len(s)
        if n < k:
            return False
        t = 2 ** k
        p = 0
        for i in range(k - 1):
            p <<= 1
            p += ord(s[i]) - ord('0')
        for i in range(k - 1, n):
            p <<= 1
            p &= (t - 1)
            p += ord(s[i]) - ord('0')
            a.add(p)
        return len(a) == t
