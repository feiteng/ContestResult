class Solution(object):
    def hasAllCodes(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: bool
        """
        l = len(s)
        if l < k:
            return False
        has = [False] * (1 << k)
        cur = 0
        for i in range(k - 1):
            if s[i] == '1':
                cur |= 1
            cur <<= 1
        for i in range(k - 1, l):
            if s[i] == '1':
                cur |= 1
            #print(cur)
            has[cur] = True
            if s[i - (k - 1)] == '1':
                cur ^= 1 << (k - 1)
            cur <<= 1
        for i in range(1 << k):
            if not has[i]:
                return False
        return True
