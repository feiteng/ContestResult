class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        d = {}
        
        
        def dfs(x1,y,x2):
            if x1<0 or x1>=len(grid[0]) or x2<0 or x2>=len(grid[0]):
                return float('-inf')
            if y == len(grid)-1:
                if x1!=x2:
                    return grid[y][x1]+grid[y][x2]
                return grid[y][x1]
            if (x1,y,x2) in d:
                return d[(x1,y,x2)]
            cur = grid[y][x1]+(grid[y][x2] if x1!=x2 else 0)
            nm = 0
            for i in range(-1, 2):
                for j in range(-1, 2):
                    nm = max(nm, dfs(x1+i, y+1, x2+j))
            d[x1,y,x2] = cur + nm
            return d[x1,y,x2]
        
        return dfs(0, 0, len(grid[0])-1)
