class Solution(object):

    def cherryPickup(self, A):
        m, n = len(A), len
(A[0])
        dp = {(0, n - 1): A[0][0] + A[0][n - 1]}
        print dp
        for i in xrange(1, m):
            dp2 = {}
            for j2 in xrange(n):
                for j1 in xrange(j2):
                    for d1 in [-1, 0, 1]:
                        for d2 in [-1, 0, 1]:
                            if 0 <= j1 + d1 < n and 0 <= j2 + d2 < n:
                                if (j1 + d1, j2 + d2) in dp:
                                    dp2[j1, j2] = max(dp2.get((j1, j2), 0), dp.get(
                                        (j1 + d1, j2 + d2), -float('inf')) + A[i][j1] + A[i][j2])
            dp = dp2
            # print dp
        return max(dp.values())
