class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        M, N = len(grid), len(grid[0])
        dp = {(0, N - 1): grid[0][0] + grid[0][-1]}
        for i in xrange(1, M):
            newDp = collections.defaultdict(int)
            for (a, b), v in dp.items():
                for (x, y) in [(a - 1, b - 1), (a, b - 1), (a + 1, b - 1), (a - 1, b), (a, b), (a + 1, b), (a - 1, b + 1), (a, b + 1), (a + 1, b + 1)]:
                    if x  < 0 or y < 0 or x >= N or y >= N:
                        continue
                    newDp[(x, y)] = max(newDp[(x, y)], v + grid[i][x] + (grid[i][y] if x != y else 0))
            dp = newDp
        return max(dp.values())
