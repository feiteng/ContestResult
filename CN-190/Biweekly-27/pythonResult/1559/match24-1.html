class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m, n = len(grid), len(grid[0])
        dp = [[[-1 for k in range
(n)] for j in range(n)] for i in range(m)]
        dp[0][0][n - 1] = grid[0][0] + grid[0][n - 1]
        for i in range(1, m):
            for j in range(n):
                for k in range(n):
                    for delta1 in [-1, 0, 1]:
                        for delta2 in [-1, 0, 1]:
                            y1 = j + delta1
                            y2 = k + delta2
                            if y1 >= 0 and y2 >= 0 and y1 < n and y2 < n and dp
[i - 1][y1][y2] != -1:
                                if k != j:
                                    dp[i][j][k] = max(dp[i][j][k], dp[i - 1][y1][y2] + grid[i][j] + grid[i][k])
                                else:
                                    dp[i][j][k] = max(dp[i][j][k], dp[i - 1][y1][y2] + grid[i][j])
        #print(dp)
        ans = 0
        for j in range(n):
            ans = max(ans, max(dp[m - 1][j]))
        return ans
