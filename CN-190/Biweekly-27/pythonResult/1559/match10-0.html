class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        M, N = len(grid), len(grid[0])
        dp1, dp2 = [[0]*N for _ in range(N)], [[0]*N for _ in range(N)]
        dp1[0][N-1] = grid[0][0]+grid[0][-1]+1
        for row in grid[1:]:
            for i in range(N):
                for j in range(N):
                    for i_ in range(max(i-1, 0), min(i+2, N)):
                        for j_ in range(max(j-1, 0), min(j+2, N)):
                            if dp1[i_][j_] > 0:
                                dp2[i][j] = max(dp2[i][j], dp1[i_][j_]+row[i]+row[j] if i!=j else dp1[i_][j_]+row[i])
            dp1 = dp2
            dp2 = [[0]*N for _ in range(N)]
        ans = 0
        for i in range(N):
            for j in range(N):
                ans = max(ans, dp1[i][j])
        return ans-1
