class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m, n = len(grid), len(grid[0])

        dp = [[[0 for _ in range(n)] for _ in range(n)] for _ in range(m)]
        d = [-1, 0, 1]
        if n == 1:
            val = dp[0][0][0] = grid[0][0]
        else:
            val = dp[0][0][-1] = grid[0][0] + grid[0][-1]
        r = 0
        res = val
        q = [[0, n - 1]]
        while r < m:
            if r == m - 1:
                return res
            tmp = []
            for c1, c2 in q:
                val = dp[r][c1][c2]
                res = max(res, val)
                for o1 in d:
                    for o2 in d:
                        oc1 = o1 + c1
                        oc2 = o2 + c2
                        if 0 <= oc1 < n and 0 <= oc2 < n:
                            v = dp[r][c1][c2] + grid[r + 1][oc1] if oc1 == oc2 else dp[r][c1][c2] + grid[r + 1][oc1] + grid[r + 1][oc2]
                            if v >= dp[r + 1][oc1][oc2]:
                                dp[r + 1][oc1][oc2] = v
                                tmp.append((oc1, oc2))
                                res = max(res, v)
            r += 1
            q = list(set(tmp))
            # for row in grid:
            #     print row
            # print r, q, [dp[r][i][j] for i, j in q]
        return res
