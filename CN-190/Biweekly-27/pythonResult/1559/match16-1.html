class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        r, c = len(grid), len(grid[0])
        dp = [[-1] * c for _ in range(c)]
        dp[0][c - 1] = grid[0][0] + grid[0][-1]
        #print(dp)
        for i in range(1, r):
            tmp = [[-1] * c for _ in range(c)]
            for a in range(c):
                for b in range(c):
                    if dp[a][b] == -1:
                        continue
                    for da in range(-1, 2):
                        x = a + da
                        if x < 0 or x >= c:
                            continue
                        for db in range(-1, 2):
                            y = b + db
                            if y < 0 or y >= c:
                                continue
                            if x == y:
                                tmp[x][y] = max(tmp
[x][y], dp[a][b] + grid[i][x])
                            else:
                                tmp[x][y] = max(tmp[x][y], dp[a][b] + grid[i][x] + grid[i][y])
                                #print(i, a, b, x, y, tmp[x][y], dp[a][b])
            dp, tmp = tmp, dp
            #print(dp)
        ans = 0
        for v in dp:
            ans = max(ans, max(v))
        return ans
