class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        m = len
(grid[0])
        d = {(0, m - 1): grid[0][0] + grid[0][m - 1]}
        for i in range(1, n):
            nxt = {}
            for x, y in d.keys():
                l = [(x - 1, y - 1), (x - 1, y), (x - 1, y + 1), (x, y - 1), (x, y), (x, y + 1), (x + 1, y - 1), (x + 1, y), (x + 1, y + 1)]
                for p, q in l:
                    if p >= 0 and q >= 0 and p < m and q < m:
                        v = d[(x, y)] + grid[i][p] + grid[i][q]
                        if p == q:
                            v -= grid[i][p]
                        if not (p, q) in nxt or v > nxt[(p, q)]:
                            nxt[(p, q)] = v
            d = nxt
        return max(d.values())
