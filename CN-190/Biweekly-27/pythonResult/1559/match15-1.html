class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m = len(grid)
        n = len(grid[0])
        
        dp = [[float('-inf') for _ in range(n)] for _ in range
(n)]
        dp[0][n - 1] = grid[0][0] + grid[0][n - 1]
        
        for i in range(1, m):
            next_dp = [[float('-inf') for _ in range(n)] for _ in range(n)]
            
            for j in range(n):
                for k in range(n):
                    
                    if dp[j][k] == float('-inf'):
                        continue
                        
                    for next_j in (j - 1, j, j + 1):
                        for next_k in (k - 1, k, k + 1):
                            if 0 <= next_j < n and 0 <= next_k < n:
                                next_dp[next_j][next_k] = max(next_dp[next_j][next_k], dp[j][k] + grid[i][next_j]) \
                                if next_j == next_k else \
                                max
(next_dp[next_j][next_k], dp[j][k] + grid[i][next_j] + grid[i][next_k])
            dp = next_dp
            
        ans = 0
        for i in range(n):
            ans = max(ans, max(dp[i]))
        return ans
