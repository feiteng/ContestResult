class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        rows = len(grid)
        cols = len(grid[0])
        dp=[[[float('-inf')] * (cols+2) for _ in xrange(cols+2)] for _ in xrange(rows)]
        dp[0][1][cols] = grid[0][0] + grid[0][cols-1]
        for row in xrange(1, rows):
            #print('-')
            #print(dp[row-1])
            for c1 in xrange(1,cols+1):
                for c2 in xrange(1, cols+1):
                    most = grid[row][c1-1] + grid[row][c2-1]
                    if c1 == c2: most = grid[row][c1-1]
                    #print(c1,c2, most)
                    dp
[row][c1][c2] = most + max(
                        dp[row-1][c1-1][c2-1],
                        dp[row-1][c1-1][c2],
                        dp[row-1][c1-1][c2+1],
                        dp[row-1][c1][c2-1],
                        dp[row-1][c1][c2],
                        dp[row-1][c1][c2+1],
                        dp[row-1][c1+1][c2-1],
                        dp[row-1][c1+1][c2],
                        dp[row-1][c1+1][c2+1])
            #print(dp[row])

        res = float('-inf')
        for i in xrange(cols+1):
            for j in xrange(cols+1):
                if dp[rows-1][i][j] > res: res = dp[rows-1][i][j]
        return res
