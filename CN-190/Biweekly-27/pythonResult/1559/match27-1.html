class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        cache = {}
        def doFind(i, j1, j2):
            if (i, j1, j2) in cache:
                return cache[(i, j1, j2)]
            if i == len(grid):
                res = 0
            else:
                res = None
                curr = grid[i][j1] + grid[i][j2] if j1 != j2 else grid[i][j1]
                for dj1 in [-1, 0, 1]:
                    nj1 = j1 + dj1
                    if (0 <= nj1 < len(grid[0])):
                        for dj2 in [-1, 0, 1]:
                            nj2 = j2 + dj2
                            if (0 <= nj2 < len(grid[0])):
                                cand = curr + doFind(i + 1, nj1, nj2)
                                res = cand if res is None else max(res, cand)
            cache[(i, j1, j2)] = res
            return res
        
        return doFind(0, 0, len(grid[0]) - 1)
