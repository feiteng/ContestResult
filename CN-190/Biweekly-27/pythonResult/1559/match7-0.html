class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m, n = len(grid), len(grid[0])
        dp = {}
        def rec(row_idx, r1, r2):
            if r1<0 or r2>=n or r1>=r2 or row_idx>=m: return 0
            if (row_idx, r1, r2) in dp: return dp[(row_idx, r1, r2)]
            val = grid[row_idx][r1] + grid[row_idx][r2]
            maxval = -float("inf")
            for i in [r1-1,r1,r1+1]:
                for j in [r2-1,r2,r2+1]:
                    maxval = max(maxval, rec(row_idx+1,i,j))
            dp[(row_idx, r1, r2)] = val + maxval
            return val + maxval
        return rec(0,0,n-1)
