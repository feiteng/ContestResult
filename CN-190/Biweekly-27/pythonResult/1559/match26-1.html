class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        memo = {}
        
        
def pick(x, y1, y2):
            if (x, y1, y2) in memo:
                return memo[(x, y1, y2)]
            if x >= len(grid):
                return 0
            cur = 0
            for yy1 in [y1 + 1, y1, y1 - 1]:
                for yy2 in [y2+1, y2, y2 - 1]:
                    if 0 <= yy1 < len(grid[0]) and 0 <= yy2 < len(grid[0]):
                        cur = max(cur, pick(x+1, yy1, yy2))
            memo[(x, y1, y2)] = cur
            if y1 != y2:
                memo[(x, y1, y2)] +=  grid[x][y1]
            memo[(x, y1, y2)] += grid[x][y2]           
            return memo[(x, y1, y2)]
        
        pick(0, 0, len(grid[0])-1)
        return memo[(0, 0, len(grid[0])-1)]
