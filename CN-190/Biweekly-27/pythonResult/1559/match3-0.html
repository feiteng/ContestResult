class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m = len(grid)
        n = len(grid[0])
        dp = [[[-1 for _ in range(n)] for _ in range(n)] for _ in range(m)]
        dp[0][0][n-1] = grid[0][0] + grid[0][n-1]
        for i in range(1,m):
            for j in range(n):
                for k in range(n):
                    for s in range(j-1, j+2):
                        for t in range(k-1, k+2):
                            if s>=0 and s<n and t>=0 and t<n and dp[i-1][s][t] >= 0:
                                if j == k:
                                    dp[i][j][k] = max(dp[i][j][k], dp[i-1][s][t] + grid[i][k])
                                else:
                                    dp[i][j][k] = max(dp[i][j][k], dp[i-1][s][t] + grid[i][j] + grid[i][k])
        res = -1
        for j in range(n):
            for k in range(n):
                res = max(res, dp[m-1][j][k])
        return res
