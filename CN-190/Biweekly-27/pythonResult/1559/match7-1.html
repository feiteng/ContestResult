class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        r = len(grid)
        c = len(grid[0])
        d = [(1, -1), (1, 0), (1, 1)]
        hash_map = {}
        def solve(i1, j1, i2, j2):
            if i1 == i2 and j1 == j2:
                return float('-inf')
            
            if i1 == r-1 or i2 == r-1:
                return grid[i1][j1] + grid[i2][
j2]
            
            if (i1, j1, i2, j2) in hash_map:
                return hash_map[(i1, j1, i2, j2)]
            
            cur_sum = grid[i1][j1] + grid[i2][j2]
            
            _max = float('-inf')
            for r1, c1 in d:
                for r2, c2 in d:
                    if j1 + c1 >= c or j2 + c2 >= c or j1+c1 < 0 or j2 + c2 < 0:
                        continue
                    _max 
= max(_max, solve(i1+r1, j1+c1, i2+r2, j2+c2))
            hash_map[(i1, j1, i2, j2)] = cur_sum + _max
            return cur_sum + _max
        return solve(0,0,0,c-1)
