class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n,m=len(grid),len(grid[0])
        dp=[[[0]*m for i in range(m)] for j in range(2)]
        for i in range(n-1,-1,-1):
            for j in range(m):
                for k in range(m):
                    tmp
=0
                    if j==k:
                        tmp=grid[i][j]
                    else:
                        tmp=grid[i][j]+grid[i][k]
                    tmp2=0
                    for d1 in range(-1,2):
                        for d2 in range(-1,2):
                            if 0<=j+d1<=m-1 and 0<=k+d2<=m-1:
                                tmp2=max(tmp2, dp[(i+1)%2][j+d1][k+d2])
                    dp[i%2][j][k]=tmp+tmp2
            #print dp[i%2]
        #print dp[1]
        return dp[0][0][m-1]
