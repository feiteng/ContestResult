class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        M,N= len(grid), len(grid[0])
        memo = {}
        def dp(i,j, row):
            key = (i,j, row)
            # print (i,j, row)
            if key in memo:
                return memo[key]
            board = grid[row][i]
            if i!=j:
                board += grid[row][j]
            if row == M-1:
                return board
            ans = board
            for pos in [
(i-1,j),(i,j),(i+1,j),(i-1,j-1),(i,j-1),(i+1,j-1),(i-1,j+1),(i,j+1),(i+1,j+1)]:
                if 0<=pos[0]<N and 0<=pos[1]<N and pos[0]!=pos[1]:
                    ans = max(ans, board + dp(pos[0], pos[1], row+1))
            memo[key] = ans
            return ans
        ans = dp(0,N-1,0)
        return ans
