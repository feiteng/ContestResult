class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m=len(grid)
        n=len(grid[0])
        dp={}
        def possible(i,j):
            if i<0 or j<0 or i>=m or j>=n:
                return False
            return True
        cx=[1,1,1]
        cy=[1,-1,0]
        
        def sol(i1,j1,i2,j2):
            if i1>=m and i2>=m:
                return 0
            key=str(i1)+"_"+str(j1)+"_"+str(j2)
            if key in dp:
                return dp[key]
            a=0
            
            if j1==j2:
                
                s=grid[i1][j1]
            else:
                s=grid[i1][j1]+grid[i2][j2]
            a=s
            for x in range(3):
                for y in range(3):
                    first=possible(i1+cx[x],j1+cy[x])
                    sec=possible(i2+cx[y],j2+cy[y])
                    fx=i1+cx[x]
                    fy=j1+cy[x]
                    sx=i2+cx[y]
                    sy=j2+cy[y]

                    if first and sec:
                        a=max(a,sol(fx,fy,sx,sy)+s)
            dp[key]=a
            return dp[key]
        return sol(0,0,0,n-1)
