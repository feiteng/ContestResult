class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        cache = {} # (row, col1, col2): num_cherries
        return self._search(grid, 0, 0, len(grid[0])-1, cache)
    
    def _search(self, grid, r, c1, c2, cache):
        if (r, c1, c2) in cache:
            return cache[(r, c1, c2)]
        if r >= len(grid):
            return 0
        col = len(grid[0])
        cnt, max_cnt = 0, 0
        curr = grid[r][c1] if c1 == c2 else grid[r][c1] + grid[r][c2]
        for d1 in [-1, 0, 1]:
            if not 0 <= c1 + d1 < col:
                continue
            for d2 in [-1, 0, 1]:
                if not 0 <= c2 + d2 < col:
                    continue
                cnt = curr + self._search(grid, r + 1, c1 + d1, c2 + d2, cache)
                max_cnt = max(max_cnt, cnt)
        cache[(r, c1, c2)] = max_cnt
        return max_cnt
