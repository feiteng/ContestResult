class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        c = len(grid[0])  
        dp = [[-float('inf') for v in range(c)] for u in range(c)]
        dp[0][c-1] = grid[0][0] + grid[0][c-1]
        
        
        for i in range(1, len(grid)):
            
            prev = dp[::]
            
            dp = [[-float('inf') for v in range(c)] for u in range(c)]
            for u in range(c):
                for v in range(c):
                    for pu in [u-1, u, u+1]:
                        for pv in [v-1,v,v+1]:
                            if 0 <= pu < c and 0<= pv < c:
                                dp[u][v] = max(dp[u][v], grid[i][u] + grid[i][v] -(grid[i][v] if u == v else 0)+ prev[pu][pv])
        return max([max(l) for l in dp])
