class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        l = len(grid)
        w = len(grid[0])
        dp = [[[0 for _ in range(w)] for _ in range(w)] for _ in range(l)]
        for i in range(w):
            dp[-1][i][i] = grid[-1][i]
            for j in range(i+1,w):
                dp[-1][i][j] = grid[-1][i]+grid[-1][j]
                #print('dp -1 ',i,' ',j,'  = ',dp[-1][i][j])
        for i in range(l-2,-1,-1):
            #print('i is ',i)
            #print('Row  ',i)
            for j in range(w):                
                for k in range(j,w):
                    #print(' pos ',j,' ',k)
                    tans = grid[i][j]+grid[i][k]
                    if j==k:
                        tans = tans//2
                    mx = 0
                    #print('temp ',tans)
                    for p in [-1,0,1]:
                        for q in [-1,0,1]:
                            if j+p >=0 and j+p<w and k+q>=0 and k+q<w:
                                mx = max(mx,dp[i+1][j+p][k+q])
                    dp[i][j][k] = tans+mx
                    #print('dp ',i,' ',j,' ',k,'  = ',dp[i][j][k])
            #print()
        return dp[0][0][w-1]
