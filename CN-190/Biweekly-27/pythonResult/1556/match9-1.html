class Solution(object):
    def canBeEqual(self, target, arr):
        """
        :type target: List[int]
        :type arr: List[int]
        :rtype: bool
        """
        cs1 ,cs2 
= {}, {}
        for n in target:
            if n not in cs1: cs1[n]=0
            cs1[n]+=1
        for n in arr:
            if n not in cs2: cs2[n]=0
            cs2[n]+=1
        for n in cs1:
            if n not in cs2:
                return False
            if cs1[n]!=cs2[n]: return False
        for n in cs2:
            if n not in cs1:
                return False
        return True
