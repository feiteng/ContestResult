class Solution(object):
    def canBeEqual(self, target, arr):
        """
        :type target: List[int]
        :type arr: List[int]
        :rtype: bool
        """
        m = {}
        m2 = {}
        for t in target:
            if t not in m:
                m[t] = 0
            m[t] += 1
        for t in arr:
            if t not in m2:
                m2[t] = 0
            m2[t] += 1
        for key in m2:
            if key not in m:
                return False
            if m2[key] != m[key]:
                return False
        return len(arr) == len(target)
