class Solution(object):
    def canBeEqual(self, target, arr):
        """
        :type target: List[int]
        :type arr: List[int]
        :rtype: bool
        """
        d = {}
        
        for key in target:
            if key not in d:
                d[key] = 1
            else:
                d[key] += 1
                
        for key in arr:
            if key not in d or d[key] == 0:
                return False
            
            d[key] -= 1
            
        for key in d:
            if d[key] != 0:
                return False
        
        return True
