class Solution {
    int[][][] dp;
    
    int m;
    int n;
    
    public int cherryPickup(int[][] grid) {
        m = grid.length;
        n = grid[0].length;
        dp = new int[m][n][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                Arrays.fill(dp[i][j], -1);
            }
        }
        
        return dfs(grid, 0, 0, n - 1);
    }
    
    private int dfs(int[][] grid, int x, int y1, int y2) {
        if (y1 < 0 || y1 >= n || y2 < 0 || y2 >= n) return Integer.MIN_VALUE;
        if (x >= m) return 0;
        if (dp[x][y1][y2] != -1) return dp[x][y1][y2];
        if (y1 == y2) {
            int temp = Integer.MIN_VALUE;
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    temp = Math.max(temp, dfs(grid, x + 1, y1 + i, y2 + j));
                }
            }
            dp[x][y1][y2] = grid[x][y1] + temp;
        }
        else {
            int temp = Integer.MIN_VALUE;
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    temp = Math.max(temp, dfs(grid, x + 1, y1 + i, y2 + j));
                }
            }
            dp[x][y1][y2] = grid[x][y1] + grid[x][y2] + temp;
        }
        return dp[x][y1][y2];
    }
}
