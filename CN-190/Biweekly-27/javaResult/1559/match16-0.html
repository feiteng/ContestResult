class Solution {
    public int cherryPickup(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;
        int dp[][][][] = new int[row][col][row][col];
        for (int i = row - 1; i >= 0; i--) {
            for (int j = 0; j < col; j++) {
                for (int k = 0; k < col; k++) {
                    dp[i][j][i][k] = grid[i][j] + (k != j ? grid[i][k] : 0);
                    if (i < row - 1) {
                        int max = 0;
                        if (j - 1 >= 0 && k - 1 >= 0) {
                            max = Math.max(dp[i + 1][j - 1][i + 1][k - 1], max);
                        }
                        if (j - 1 >= 0) {
                            max = Math.max(dp[i + 1][j - 1][i + 1][k], max);
                        }
                        if (k - 1 >= 0) {
                            max = Math.max(dp[i + 1][j][i + 1][k - 1], max);
                        }
                        max = Math.max(dp[i + 1][j][i + 1][k], max);
                        if (j - 1 >= 0 && k + 1 < col) {
                            max = Math.max(dp[i + 1][j - 1][i + 1][k + 1], max);
                        }
                        if (j + 1 < col && k - 1 >= 0) {
                            max = Math.max(dp[i + 1][j + 1][i + 1][k - 1], max);
                        }
                        if (j + 1 < col && k + 1 < col) {
                            max = Math.max(dp[i + 1][j + 1][i + 1][k + 1], max);
                        }
                        if (k + 1 < col) {
                            max = Math.max(dp[i + 1][j][i + 1][k + 1], max);
                        }
                        if (j + 1 < col) {
                            max = Math.max(dp[i + 1][j + 1][i + 1][k], max);
                        }
                        dp[i][j][i][k] += max;
                    }
                }
            }
        }
        return dp[0][0][0][col - 1];

    }
}
