class Solution {
    static int dp[][][];
    public int cherryPickup(int[][] grid) {
        dp=new int[grid.length][grid[0].length][grid[0].length];
        // PriorityQueue<Pair>q=new PriorityQueue<>((a,b)->(b.c-a.c));
        return dfs(0,0,grid[0].length-1,grid);
       
    }
    int dfs(int i, int y1, int y2, int grid[][])
    {
        if(i==grid.length){return 0;}
        if(dp[i][y1][y2]!=0){return dp[i][y1][y2];}
        int a=(y1==y2?grid[i][y1]:grid[i][y1]+grid[i][y2]);
        
int max=0;
        for(int j=y1-1;j<=y1+1;j++)
        {
            for(int k=y2-1;k<=y2+1;k++){
                if(j>=0 && k>=0 && j<grid[0].length &&k<grid[0].length )
                {
                
                        max=Math.max(dfs(i+1,j,k,grid),max);
                   
                }
            }
        }
        dp[i][y1][y2]=a+max;
        return a+max;
    }
    
}
