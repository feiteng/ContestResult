class Solution {
    int n;
    int m;
    int[][][] memo;
    public int cherryPickup(int[][] grid) {
        n = grid.length;
        m = grid[0].length;
        
memo = new int[n][m][m];
        return calc(grid, 0, 0, m-1);
    }
    
    public int calc(int[][] grid, int i, int j1, int j2){
        if(i==n || j1<0 || j1>=m || j2<0 || j2>=m){
            return 0;
        }
        if(memo[i][j1][j2]!=0){
            return memo[i][j1][j2];
        }
        int res = j1==j2 ? grid[i][j1] : grid[i][j1]+grid[i][j2];
        int max = 0;
        for(int m1=-1; m1<=1; m1++){
            for(int m2=-1; m2<=1; m2++){
                max = Math.max(max, calc(grid, i+1, j1+m1, j2+m2));
            
}
        }
        memo[i][j1][j2] = res+max;
        return memo[i][j1][j2];
    }
}
