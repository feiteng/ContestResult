class Solution {
    public int cherryPickup(int[][] grid) {
        int R = grid.length;
        int C = grid[0].length;
        
        int[][][] dp = new int[R + 1][C + 1][C + 1]; // dp[i][j][k] row i, robot 1 on col j, robot 2 on col k
        for (int[][] d : dp) {
            for (int[] e : d) {
                Arrays.fill(e, IMPOSSIBLE);
            }
        }
        
        dp[0][0][C - 1] = grid[0][0] + grid[0][C - 1];
        
        for (int i = 1; i < R; i++) {
            for (int j = 0; j < C; j++) {
                for (int k = 0; k < C; k++) {
                    int max = IMPOSSIBLE;
                    for (int m1 = -1; m1 <= 1; m1++) {
                        int jNew = j + m1;
                        if (jNew < 0 || jNew >= C) continue;
                        
                        for (int m2 = -1; m2 <= 1; m2++) {
                            int kNew = k + m2;
                            if (kNew < 0 || kNew >= C) continue;
                            if (dp[i - 1][jNew][kNew] == IMPOSSIBLE) continue;
                            
                            max = Math.max(max, dp[i - 1][jNew][kNew]);
                        }
                    }
                    
                    if (max != IMPOSSIBLE) {
                        dp[i][j][k] = max + grid[i][j] + (j == k ? 0 : grid[i][k]);
                    }
                }
            }
        }
        
        int cherry = 0;
        for (int[] e : dp[R - 1]) {
            for (int f : e) {
                cherry = Math.max(cherry, f);
            }
        }
        return cherry;
    }
    
    private static final int IMPOSSIBLE = -1;
    private static final int[][] MOVES = new int[][]{
        {+1,-1},
        {+1,0},
        {+1,+1}
    };
}
