class Solution {
    int[] horizontalMoves = new int[] {-1, 0, 1};
    
public int cherryPickup(int[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        Integer[][] preRow = new Integer[cols][cols]; 
        preRow[0][cols-1] = grid[0][0] + grid[0][cols-1];
        for(int row = 1; row < rows; ++row) {
            Integer[][] curRow = new Integer[cols][cols];
            for(int left = 0; left < cols; ++left) {
                for(int right = left; right < cols; ++right) {
                    if(preRow[left][right] != null) {
                        
for(int leftDelta : horizontalMoves) {
                            for(int rightDelta : horizontalMoves) {
                                int newLeft = left + leftDelta;
                                int newRight = right + rightDelta;
                                if(newLeft > -1 && newLeft <= newRight && newRight < cols) {
                                    int newCollectionAmount = preRow[left][right] + grid[row][newLeft] + (newRight == newLeft ? 0 : grid[row][newRight]);
                                    
if(curRow[newLeft][newRight] == null || curRow[newLeft][newRight] < newCollectionAmount) {
                                        curRow[newLeft][newRight] = newCollectionAmount;
                                    }
                                }
                            }
                        }
                    }
                }
            }
            
            
preRow = curRow;
        }
        
        int max = 0;
        for(int left = 0; left < cols; ++left) {
            for(int right = left; right < cols; ++right) {
                if(preRow[left][right] != null && preRow[left][right] > max) {
                    max = preRow[left][right];
                }
            }
        }
        
        return max;
    }
}
