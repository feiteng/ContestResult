class Solution {
    public int cherryPickup(int[][] grid) {
        int N = grid.length, M = grid[0].length;
        int[][][] dp = new int[N][M][M];
        int res = 0;
        for (int k = 0; k < N; k++) {
            for (int i = 0; i < M; i++) {
                for (int j = 0; j < M; j++) {
                    dp[k][i][j] = -1;
                    if (k == 0) {
                        dp[k][i][j] = i == 0 && j == M - 1 ? grid[0][0] + grid[0][M - 1] : -1;
                    } else {
                        
for (int p = -1; p <= 1; p++) {
                            for (int q = -1; q <= 1; q++) {
                                if (i + p >= 0 && i + p < M && j + q >= 0 && j + q < M) {
                                    dp[k][i][j] = Math.max(dp[k][i][j], dp[k - 1][i + p][j + q]);
                                }
                            }
                        }
                        if (dp[k][i][j] >= 0) {
                            dp[k][i][j] += i == j ? grid[k][i] : grid[k][i] + grid[k][j];
                        }
                    }
                    res = Math.max(res, dp[k][i][j]);
                
}
            }
        }
        return res;
    }
}
