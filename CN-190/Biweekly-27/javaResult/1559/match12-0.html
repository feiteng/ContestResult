class Solution {

      Integer[][][] dp;
  public int cherryPickup(int[][] grid) {

    dp = new Integer[grid.length][grid[0].length][grid[0].length];
    return dfs(grid, 0, 0, grid[0].length - 1);
  }

  int dfs(int[][] grid, int row, int col1, int col2) {
    if (dp[row][col1][col2] != null){
      return dp[row][col1][col2];
    }
    if (row == grid.length - 1) {
      if (col1 == col2) {
        return grid[row][col1];
      }
      return grid[row][col1] + grid[row][col2];
    }

    int max = 0;
    for (int i = -1; i <= 1; i++) {
      for (int j = -1; j <= 1; j++) {
        if (col1 + i >= 0 && col1 + i < grid[0].length
            && col2 + j >= 0 && col2 + j < grid[0].length) {
          max = Math.max(dfs(grid, row+1, col1 + i, col2 + j), max);
        }
      }
    }
    int ans = max + grid[row][col1];
    if (col1 != col2) {
      ans += grid[row][col2];
    }
    dp[row][col1][col2] = ans;
    return ans;
  }

}
