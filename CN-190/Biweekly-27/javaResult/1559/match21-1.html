class Solution {
    public int cherryPickup(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        int[][][] dp = new int[m][n][n];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                Arrays.fill(dp[i][j], -1);
            
}
        }
        
        dp[0][0][n - 1] = grid[0][0] + grid[0][n - 1];
        
        
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    int max = -1;
                    if (j != 0 && k != 0) {
                        max = Math.max(max, dp[i - 1][j - 1][k - 1]);
                    }
                    
                    if (j != 0) {
                        max = Math.max(max, dp[i - 1][j - 1][k]);
                    }
                    
                    if (j != 0 && k != n - 1) {
                        max = Math.max(max, dp[i - 1][j - 1][k + 1]);
                    }
                    
                    if (j != n - 1 && k != 0) {
                        max = Math.max(max, dp[i - 1][j + 1][k - 1]);
                    }
                    
                    if (j != n - 1) {
                        max = Math.max(max, dp[i - 1][j + 1][k]);
                    }
                    
                    if (j != n - 1 && k != n - 1) {
                        max = Math.max(max, dp[i - 1][j + 1][k + 1]);
                    }
                    
                    if (k != 0) {
                        max = Math.max(max, dp[i - 1][j][k - 1]);
                    }
                    
                    
                    max = Math.max(max, dp[i - 1][j][k]);
                    
                    if (k != n - 1) {
                        max = Math.max(max, dp[i - 1][j][k + 1]);
                    }
                    
                    
                    
if (max != -1) {
                        if (j == k) {
                            dp[i][j][k] = max + grid[i][j];
                        } else {
                            dp[i][j][k] = max + grid[i][j] + grid[i][k];
                        }
                    }
                    
                    
                }
            }
        }
        
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                res = Math.max(res, dp[m - 1][i][j]);
            }
        }
        
        return res;
    }
}
