class Solution {
    public int cherryPickup(int[][] grid) {
        
        final int rows = grid.length;
        final int cols = grid[0].length;
        
        // prev and current row
        // positions of r1 and r2
        // total cherries
        int[][][] dp = new int[2][cols][cols];
        
        dp[0][0][cols - 1] = grid[0][0] + grid[0][cols - 1];
        
        for (int r = 1; r < rows; r++) {
            

            
            for (int i = 0; i < Math.min(r + 1, cols); i++) {
                for (int j = Math.max(i, cols - 1 - r); j < cols; j++) {
                    
                    int max = 0;
                    
                    for (int k = -1; k <= 1; k++) {
                        for (int l = -1; l <= 1; l++) {
                            int oi = i + k;
                            int oj = j + l;
                            if (oi >= 0 && oi < cols && oj >= 0 && oj < cols) {
                                if (max < dp[0][oi][oj]) {
                                    max = dp[0][oi][oj];
                                }
                            }
                        }
                    }
                                        
                    dp[1][i][j] = grid[r][i] + (i != j ? grid[r][j] : 0) + max;
                        
                }
            }
            
            int[][] tmp = dp[0];
            dp[0] = dp[1];
            dp[1] = tmp;
        }
        
        int result = 0;
        
        for (int i = 0; i < cols; i++) {
            for (int j = i; j < cols; j++) {
                if (dp[0][i][j] > result) {
                    result = dp[0][i][j];
                }
            }
        }
        
        return result;
    }
}
