class Solution {
    int c;
    Integer[][][] dp;
    public int cherryPickup(int[][] grid) {
        
        c = grid[0].length;
        dp = new Integer[c][c][grid.length];
        int i = 0;
        int j = c-1;
        return dfs(i, j, 0, grid);
    }
    
    int[] d = {-1, 0, 1};
    
    int dfs(int i, int j, int l, int[][] grid) {
        if (l >= grid.length) {
            return 0;
        }
        if (dp[i][j][l] != null) {
            return dp[i][j][l];
        }
        int cur = 0;
        cur += grid[l][i];
        if (i != j) {
            cur += grid[l][j];
        }
        int max = 0;
        for (int di = 0; di < 3; ++di) {
            for (int dj = 0; dj < 3; ++dj) {
                int ti = i + d[di];
                int tj = j + d[dj];
                if (isValid(ti) && isValid(tj)) {
                    max = Math.max(max, dfs(ti, tj, l+1, grid));
                }
            }
        }
        dp[i][j][l] = max + cur;
        return max + cur;
    }
    
    boolean isValid(int i) {
        return i >= 0 && i < c;
    }
    
}
