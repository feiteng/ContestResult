class Solution {
    public int cherryPickup(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][][] dp = new int[m][n][n];
        dp[0][0][n - 1] = n - 1 == 0 ? grid[0][0] : grid[0][0] + grid[0][n - 1];
        for(int i = 1; i < m; i++) {
            for(int j = 0; j <= Math.min(i, n - 1); j++) {
                for(int k = n - 1; k >= Math.max(0, n - 1 - i); k--) {
                    dp[i][j][k] = (j == k) ? grid[i][j] : grid[i][j] + grid[i][k];
                    int max = Integer.MIN_VALUE;
                    for(int jj = -1; jj <= 1; jj++) {
                        for(int kk = -1; kk <= 1; kk++) {
                            int nj = j + jj, nk = k + kk;
                            if(nj < 0 || nj >= n || nk < 0 || nk >= n) continue;
                            max = Math.max(max, dp[i - 1][nj][nk]);
                        }
                    }
                    dp[i][j][k] += max;
                }
            }
        }
        int res = 0;
        for(int j = 0; j < n; j++) {
            for(int k = 0; k < n; k++) {
                res = Math.max(res, dp[m - 1][j][k]);
            }
        }
        return res;
    }
}
