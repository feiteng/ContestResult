class Solution {
    int[] dc = new int[]{-1, 0, 1};
    public int cherryPickup(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        return walk(grid, new int[m][n][n], new boolean[m][n][n], 0, 0, n - 1);
    }

    private int walk(int[][] grid, int[][][] dp, boolean[][][] visit, int r, int x, int y) {
        if (x < 0 || y < 0 || x >= grid[0].length || y >= grid[0].length) {
            return 0;
        }
        if (visit[r][x][y]) {
            return dp[r][x][y];
        }
        
        int ans = 0;
        if (r < grid.length - 1) {
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    ans = Math.max(walk(grid, dp, visit, r + 1, x + dc[i], y + dc[j]), ans);    
                }
                
            }
        }
        if (x == y) {
            ans += grid[r][x];
        } else {
            ans += grid[r][x] + grid[r][y];
        }
        visit[r][x][y] = true;
        return dp[r][x][y] = ans;
    }

}
