class Solution {
    int[][] grid;
    int[][][] dp;
    int n;
    int m;
    
    int inf = (int)1e9;
    public int dp(int i, int j, int k){
        if(i >= n){
            return 0;
        }
        if(j < 0 || j >= k || k >= m){
            return -inf;
        }
        if(dp[i][j][k] == -1){
            dp[i][j][k] = 0;
            for(int dx = -1; dx <= 1; dx++){
                for(int dy = -1; dy <= 1; dy++){
                    dp[i][j][k] = Math.max(dp[i][j][k], dp(i + 1, j + dx, k + dy));
                }
            }
            dp[i][j][k] += grid[i][j] + grid[i][k];
        }
        return dp[i][j][k];
    }
    
    public int cherryPickup(int[][] grid) {
        this.grid = grid;
        n = grid.length;
        m = grid[0].length;
        dp = new int[n][m][m];
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                Arrays.fill(dp[i][j], -1);
            }
        }
        int ans = dp(0, 0, m - 1);
        return ans;
    }
}
