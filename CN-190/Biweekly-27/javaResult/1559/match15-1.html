class Solution {
    public int cherryPickup(int[][] grid) {
        int r = grid.length;
        int c = grid[0].length;
        int dp[][][] = new int[r][c][c];
        dp[0][0][c - 1] = grid[0][0] + grid[0][c - 1];
        boolean ok[][][] = new boolean[r][c][c];
        ok[0][0][c - 1] = true;
        for (int i = 1; i < r; i++) {
            for (int j = 0; j < c; j++) {
                for (int k = j + 1; k < c; k++) {
                    
for (int l = -1; l <= 1; l++) {
                        for (int m = -1; m <= 1; m++) {
                            int x = j + l;
                            int y = k + m;
                            if (x < 0 || x >= c || y < 0 || y >= c || x >= y || !ok[i - 1][x][y]) {
                                continue;
                            }
                            ok[i][j][k] = true;
                            
dp[i][j][k] = Math.max(dp[i][j][k], dp[i - 1][x][y] + grid[i][j] + grid[i][k]);
                        }
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < c; i++) {
            for (int j = i + 1; j < c; j++) {
                res = Math.max(res, dp[r - 1][i][j]);
            }
        }
        return res;
    }
}
