class Solution {
    
    public int cherryPickup(int[][] grid) {
        int[][][] dp = new int[grid.length + 1][grid[0].length][grid[0].length];
        for (int i = grid.length - 1; i >= 0; i--) {
            for (int j = 0; j < grid[0].length; j++) {
                for (int k = 0; k < grid[0].length; k++) {
                    int sum = 0;
                    if (j == k) {
                        sum += grid[i][j];
                    } else {
                        sum += grid[i][j] + grid[i][k];
                    }
                    
                    int[][] next = dp[i+1];
                    int max = next[j][k];
                    for (int m = -1; m <= 1; m++) {
                        for (int n = -1; n <= 1; n++) {
                            if (j + m < 0 || j + m >= grid[0].length || k + n < 0 || k + n >= grid[0].length) continue;
                            max = Math.max(max, next[j+m][k+n]);
                        }
                    }
                    sum += max;
                    dp[i][j][k] = sum;
                }
            }
        }
        return dp[0][0][grid[0].length -1];
        
    }
    
    class Pair {
        int[] r1 = new int[2];
        int[] r2 = new int[2];
        
        Pair(int ra, int ca, int rb, int cb) {
            r1[0] = ra;
            r1[1] = ca;
            r2[0] = rb;
            r2[1] = cb;
        }
    }
}
