class Solution {
    public int cherryPickup(int[][] grid) {
        Integer[][][] dp = new Integer[grid.length][grid[0].length][grid[0].length];
        return dfs(grid, 0, 0, grid[0].length - 1, dp);
    }
    
    private int dfs(int[][] grid, 
int curRow, int left, int right, Integer[][][] dp) {
        if (curRow >= grid.length) {
            return 0;
        }
        
        if (dp[curRow][left][right] != null) {
            return dp[curRow][left][right];
        }
        int max = 0;
        int curSum = 0;
        if (left == right) {
            curSum += grid[curRow][left];
        } else {
            curSum += grid[curRow][left];
            curSum += grid[curRow][right];
        }
        
for (int i = left - 1; i <= left + 1; i++) {
            for (int j = right - 1; j <= right + 1; j++) {
                if (i >= 0 && i < grid[0].length && j >= 0 && j < grid[0].length) {
                    max = Math.max(max, dfs(grid, curRow + 1, i, j, dp) + curSum);
                } 
            }
        }
        dp[curRow][left][right] = max;
        return max;
    }
}
