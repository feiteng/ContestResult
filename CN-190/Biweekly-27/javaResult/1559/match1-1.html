class Solution {
    
public int cherryPickup(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        return walk(grid, new int[m][n][n], new boolean[m][n][n], 0, 0, n - 1);
    }

    private int walk(int[][] grid, int[][][] dp, boolean[][][] visit, int layer, int x, int y) {
        if (x < 0 || y < 0 || x >= grid[0].length || y >= grid[0].length) {
            return 0;
        }
        if (visit[layer][x][y]) {
            return dp[layer][x][y];
        }
        int ans = 0;
        if (layer < grid.length - 1) {
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    ans = Math.max(walk(grid, dp, visit, layer + 1, x + i, y + j), ans);
                }
            }
        }
        if (x == y) {
            ans += grid[layer][x];
        } else {
            ans += grid[layer][x] + grid[layer][y];
        }
        visit[layer][x][y] = true;
        dp[layer][x][y] = ans;
        return ans;
    }
}
