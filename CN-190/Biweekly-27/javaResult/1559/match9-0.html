class Solution {
    public int cherryPickup(int[][] grid) {
         int N = grid.length, M = grid[0].length;
        int [][][]mem = new int[N][M][M];
        for(int i = 0; i < N; i++)
        {
            for(int j = 0; j < M; j++)
            {
                for(int l = 0; l < M; l++)
                    mem[i][j][l]=-1;
            }
        }
        
        return cherryPickupUtil(grid, mem, N, M, 0, 0, M-1);
      // print3D(mem); 
      //  return mem[0][0][M-1];
    }

    private boolean isValid(int N, int M, int x, int y1, int y2)
    {
        return (x >= 0 && x < N && y1 >=0 && y1 < M && y2 >=0 && y2 < M);
    }

    
    private int cherryPickupUtil(int arr[][], int mem[][][], int N, int M,
                                int x, int y1, int y2)
    {
        //System.out.println("x: " + x +" y1: "+ y1 +" y2: " + y2);

        if (!isValid(N, M, x, y1, y2)) return 0;
        
        if (x == N){
            return 0;
        }
        
        if (mem[x][y1][y2] != -1) return mem[x][y1][y2];
        
        int ans = 0;
        
        int temp = (y1 == y2)? arr[x][y1]:
                arr[x][y1] + arr[x][y2];
        
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1, y2-1));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1, y2+1));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1, y2));

        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1-1, y2));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1-1, y2-1));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1-1, y2+1));

        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1+1, y2));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1+1, y2-1));
        ans = Math.max(ans, temp +
                cherryPickupUtil(arr, mem, N, M,x+1, y1+1, y2+1));

       mem[x][y1][y2] = ans;
        return ans;
    }
}
