class Solution {
    public int cherryPickup(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        Integer[][][] dp = new Integer[n][m][m];
        
return dfs(grid, dp, 0, 0, m - 1);
    }
    
    private int[] dirs = {1, 0, -1};
    
    private int dfs(int[][] grid, Integer[][][] dp, int r, int c1, int c2){
        int n = grid.length, m = grid[0].length;
        
if(r == n){
            return 0;
        }
        if(dp[r][c1][c2] != null){
            return dp[r][c1][c2];
        }
        int num = c1 == c2 ? grid[r][c1]: grid[r][c1] + grid[r][c2];
        
int max = 0;
        for(int i: dirs){
            for(int j: dirs){
                int nc1 = c1 + i, nc2 = c2 + j;
                if(nc1 <= nc2 && nc1 >= 0 && nc2 < m)
{
                    max = Math.max(max, dfs(grid, dp, r + 1, nc1, nc2));
                }
            }
        }
        dp[r][c1][c2] = max + num;
        return dp[r][c1][c2];
    }
}
