class Solution {
    public int cherryPickup(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][][] cache = new int[m][n][n];
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cache[0][i][j] = -1;
            }
        }
        cache[0][0][n-1] = grid[0][0] + grid[0][n-1];
        
        int ans = 0;
        for (int r = 1; r < m; r++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) 
{
                    int max = cache[r-1][i][j];
                    if (i > 0) {
                        max = Math.max(max, cache[r-1][i-1][j]);
                    }
                    if (i < n-1) {
                        max = Math.max(max, cache[r-1][i+1][j]);
                    }
                    if (j > 0) {
                        max = Math.max(max, cache[r-1][i][j-1]);
                        
if (i > 0) {
                            max = Math.max(max, cache[r-1][i-1][j-1]);
                        }
                        if (i < n-1) {
                            max = Math.max(max, cache[r-1][i+1][j-1]);
                        }
                    }
                    if (j < n-1) {
                        
max = Math.max(max, cache[r-1][i][j+1]);
                        if (i > 0) {
                            max = Math.max(max, cache[r-1][i-1][j+1]);
                        }
                        if (i < n-1) {
                            max = Math.max(max, cache[r-1][i+1][j+1]);
                        }
                    }
                    if (max == -1) {
                        cache[r][i][j] = -1;
                        continue;
                    }
                    
                    cache[r][i][j] = max;
                    if (i != j) {
                        cache[r][i][j] += (grid[r][i] + grid[r][j]);
                    } else {
                        cache[r][i][j] += grid[r][i];
                    }
                    if (r == m-1) ans = Math.max(ans, cache[r][i][j]);
                
}
                // System.out.println(r + " " + i + " " + Arrays.toString(cache[r][i]));
            }
            // System.out.println("  ");
        }
        return ans;
    }
}
