class Solution {
    public static int cherryPickup(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][][] dp = new int[m][n][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    dp[i][j][k] = -100000000;
                }
            }
        }
        dp[0][0][n-1] = grid[0][0] + grid[0][n-1];
        
for (int i = 0; i < m - 1; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    if(dp[i][j][k] == -100000000) continue;
                    for(int x = -1; x <= 1; ++x) {
                        if(j + x < 0 || j + x >= n) continue;
                        for (int y = -1; y <= 1; y++) {
                            if(k + y < 0 || k + y >= n) continue;
                            int p = dp[i][j][k] + grid[i+1][j+x] + grid[i+1][k+y];
                            if(j + x == k + y) p -= grid[i+1][j+x];
                            dp[i+1][j+x][k+y] = Math.max(dp[i+1][j+x][k+y],p);
                        
}
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                ans = Math.max(dp[m-1][i][j],ans);
            }
        }
        return ans;
    }
}
