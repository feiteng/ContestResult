class Solution {
    public int cherryPickup(int[][] grid) {
        int r = grid.length;
        int c = grid[0].length;
        int[][][] dp = new int[r][c][c];
        for (int c1 = 0; c1 < c; c1++) {
            for (int c2 = 0; c2 < c; c2++) {
                dp[0][c1][c2] = -1;
            }
        }
        dp[0][0][c - 1] = grid[0][0] + grid[0][c - 1];
        int[] dir = {-1, 0, 1};
        
for (int row = 1; row < r; row++) {
            for (int c1 = 0; c1 < c; c1++) {
                for (int c2 = 0; c2 < c; c2++) {
                    int sum = (c1 == c2) ? grid[row][c1] : (grid[row][c1] + grid[row][c2]);
                    
int max = -1;
                    for (int d1 = 0; d1 < 3; d1++) {
                        int cc1 = c1 + dir[d1];
                        if (cc1 < 0 || cc1 >= c) continue;
                        for (int d2 = 0; d2 < 3; d2++) {
                            int cc2 = c2 + dir[d2];
                            if (cc2 < 0 || cc2 >= c) continue;
                            if (dp[row - 1][cc1][cc2] == -1) continue;
                            max = Math.max(max, dp[row - 1][cc1][cc2]);
                        }
                    }
                    if (max == -1) dp[row][c1][c2] = -1;
                    else dp[row][c1][c2] = max + sum
;
                }
            }
        }
        
        int res = 0;
        for (int c1 = 0; c1 < c; c1++) {
            for (int c2 = 0; c2 < c; c2++) {
                res = Math.max(res, dp[r - 1][c1][c2]);
            }
        }
        
        return res;
    }
}
