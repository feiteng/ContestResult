import java.util.Arrays;

class Solution {

    public int cherryPickup(int[][] grid) {
        dp = new int[grid.length][grid[0].length][grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                Arrays.fill(dp[i][j], -1);
            }
        }
        return cherryPickup(grid, 0, 0, grid[0].length - 1);
    }
    
    private final int[] next = {-1, 0, 1};
    private int[][][] dp;
    
    private int cherryPickup(int[][] grid, int row, int col1, int col2) {
        if (grid.length == row) {
            return 0;
        }
        if (dp[row][col1][col2] != -1) {
            return dp[row][col1][col2];
        }
        int max = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                int next1 = col1 + next[i];
                int next2 = col2 + next[j];
                if (next1 != next2 && next1 >= 0 && next1 < grid[0].length && next2 >= 0 && next2 < grid[0].length) {
                    max = Math.max(max, cherryPickup(grid, row + 1, next1, next2));
                }
            }
        }
        
        return dp[row][col1][col2] = (grid[row][col1] + grid[row][col2] + max);
    }

}
