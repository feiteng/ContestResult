class Solution {
    int[][] g;
    int n;
    int m;
    int[][][] dp;
    public int cherryPickup(int[][] grid) {
        g = grid;
        
n = g.length;
        m = g[0].length;
        
        dp = new int[n+1][m+1][m+1];
        for(int i=0; i<dp.length; i++) {
            for(int j=0; j<m+1; j++) {
                Arrays.fill(dp[i][j], -1);
            }
        }
        return Math.max(0, solve(0, 0, m-1));
    }
    
    private int solve(
int i, int j1, int j2) {
        if (j1<0 || j1>=m || j2<0 || j2>=m) return -(int)1e9;
        if (i==n) return 0;
        
        if (dp[i][j1][j2]!=-1) return dp[i][j1][j2];
        
        int ans = -(int)1e9;
        
        if (j1==j2) {
            
for(int y1=-1; y1<=1; y1++) {
                for(int y2=-1; y2<=1; y2++) {
                    ans = Math.max(ans, g[i][j1]+solve(i+1, j1+y1, j2+y2));
                }
            }
            
        } else {
            
for(int y1=-1; y1<=1; y1++) {
                for(int y2=-1; y2<=1; y2++) {
                    ans = Math.max(ans, g[i][j1]+g[i][j2]+solve(i+1, j1+y1, j2+y2));
                }
            }
            
        }
        
        return dp[i][j1][j2]=ans;
        
        
    }
}
