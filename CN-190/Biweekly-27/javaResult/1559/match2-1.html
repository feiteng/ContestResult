class Solution {
    public int cherryPickup(int[][] grid) {
        
        
int[][] dp = new int[grid[0].length][grid[0].length];
        
        for(int i = 0; i < dp.length; i++){
            for(int j = 0; j < dp.length; j++){
                dp[i][j] = -99999999;
            }
        }
        
        dp[0][dp.length-1] = grid[0][0] + grid[0][dp.length-1];
        
        
        for(int k = 1; k < grid.length; k++){
            
            int[][] newdp = new int[grid[0].length][grid[0].length];
            
            
for(int i = 0; i < dp.length; i++){
                for(int j = 0; j < dp.length; j++){
                    int max = dp[i][j];
                    if(i > 0 && j > 0) max = Math.max(max, dp[i-1][j-1]);
                    if(i > 0) max = Math.max(max, dp[i-1][j]);
                    if(i > 0 && j < dp.length-1) max = Math.max(max, dp[i-1][j+1]);
                    if(j > 0) max = Math.max(max, dp[i][j-1]);
                    if(j < dp.length-1) max = Math.max(max, dp[i][j+1]);
                    if(i < dp.length-1 && j > 0) max = Math.max(max, dp[i+1][j-1]);
                    if(i < dp.length-1) max = Math.max(max, dp[i+1][j]);
                    if(i < dp.length-1 && j < dp.length-1) max = Math.max(max, dp[i+1][j+1]);
                    
                    newdp[i][j] = max;
                    if(i != j) newdp[i][j] += (grid[k][i] + grid[k][j]);
                    else newdp[i][j] += grid[k][i];
                }
            }
            dp = newdp;
        
}
        
        int ans = -1;
        for(int i = 0; i < dp.length; i++){
            for(int j = 0; j < dp.length; j++){
                ans = Math.max(ans, dp[i][j]);
            }
        }
        return ans;
    }
}
