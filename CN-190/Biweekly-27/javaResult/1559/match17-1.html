class Solution {
    public int cherryPickup(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;
        
        // i,j,k: 在第 i 行的时候，robot1 位于 j 列，robot2 位于 k 列
        int[][][] dp = new int[row][col][col];
        
        int i = 0;
        dp[i][0][col - 1] = grid[i][0] + grid[i][col - 1];
        
        int res = 0;
        for (i = 1; i < row; i++) {
            // 左边最多达到第 i 列，右边做多达到第 (col - 1 -i) 列
            for (int j = 0; j < Math.min(i + 1, col); j++) {
                for (int k = Math.max(0, col - 1 - i); k < col; k++) 
{
                    int maxStatus = dp[i - 1][j][k];
                    
                    if (k != 0) {
                        maxStatus = Math.max(maxStatus, dp[i-1][j][k-1]);
                    }
                    if (k != col - 1) {
                        maxStatus = Math.max(maxStatus, dp[i-1][j][k+1]);
                    }
                    
                    if (j != 0) {
                        maxStatus = Math.max(maxStatus, dp[i-1][j-1][k]);
                        
if (k != 0) {
                            maxStatus = Math.max(maxStatus, dp[i-1][j-1][k-1]);
                        }
                        if (k != col - 1) {
                            maxStatus = Math.max(maxStatus, dp[i-1][j-1][k+1]);
                        }
                    }
                    if (j != col - 1) {
                        
maxStatus = Math.max(maxStatus, dp[i-1][j+1][k]);
                        if (k != 0) {
                            maxStatus = Math.max(maxStatus, dp[i-1][j+1][k-1]);
                        }
                        if (k != col - 1) {
                            maxStatus = Math.max(maxStatus, dp[i-1][j+1][k+1]);
                        }
                    }
                    dp[i][j][k] = ( j == k ?  0 : grid[i][j]) + grid[i][k] + maxStatus;
                    if (i == row - 1) {
                        res  = Math.max(res, dp[i][j][k]);
                    }
                
} 
            }
            
        }
        return res;
    } 
}
