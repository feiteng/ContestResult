class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] pre,int[][] q) {
        boolean aa[][]=new boolean[n][n];
        for(int i=0;i<pre.length;i++)
        {
           aa[pre[i][0]][pre[i][1]]=true;
        }
        
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(!aa[i][j] && i!=j)
                {
for(int k=0;k<n;k++)
                {
                    // if(i==3 && j==4){System.out.println((aa[i][k]&aa[k][j])+" "+k);}
                    if(!aa[i][j]){aa[i][j]=aa[i][k]&aa[k][j];}
                    
                }}
            }
        }
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(!aa[i][j] && i!=j)
                {for(int k=0;k<n;k++)
                {
                    // if(i==3 && j==4){System.out.println((aa[i][k]&aa[k][j])+" "+k);}
                    if(!aa[i][j]){aa[i][j]=aa[i][k]&aa[k][j];}
                    
                }}
            
}
        }
        // System.out.println(aa[3][4]);
        List<Boolean>arr=new ArrayList<>();
        for(int i=0;i<q.length;i++)
        {
            arr.add(aa[q[i][0]][q[i][1]]);
        }
        return arr;
    }
}
