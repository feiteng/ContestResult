class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Map<Integer, Set<Integer>> neighborMap = new HashMap<>();
        for (int i = 0; i < n; i++) {
            neighborMap.put(i, new HashSet<>());
        }
        for (int[] prerequisite : prerequisites) {
            neighborMap.get(prerequisite[0]).add(prerequisite[1]);
        }
        Map<Integer, Set<Integer>> dp = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (!dp.containsKey(i)) {
                dfs(i, neighborMap, dp);
            }
        }
        List<Boolean> result = new ArrayList<>();
        for (int[] query : queries) {
            result.add(dp.get(query[0]).contains(query[1]));
        }
        return result;
    }
    
    private Set<Integer> dfs(int node, Map<Integer, Set<Integer>> neighborMap, Map<Integer, Set<Integer>> dp) {
        if (dp.containsKey(node)) return dp.get(node);
        Set<Integer> result = new HashSet<>();
        for (int next : neighborMap.get(node)) {
            result.addAll(dfs(next, neighborMap, dp));
            result.add(next);
        }
        dp.put(node, result);
        return result;
    }
}
