class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        List<Set<Integer>> cr = new ArrayList();
        for (int i = 0; i < n; i++) cr.add(new HashSet())
;
        
        for (int[] p : prerequisites) {
            cr.get(p[1]).add(p[0]);
        }
        
        boolean[] visited = new boolean[n];
        
        for (int i = 0; i < n; i++) {
            aux(i, cr, visited);
        
}
        
        List<Boolean> ans = new ArrayList();
        for (int[] q : queries) {
            ans.add(cr.get(q[1]).contains(q[0]));
        }
        return ans;
    }
    
    private Set<Integer> aux(int c, List<Set<Integer>> cr, boolean[] visited) {
        if (visited[c]) return cr.get(c);
        visited[c] = true;
        
Set<Integer> pre = new HashSet();
        for (int p : cr.get(c)) {
            pre.add(p);
            pre.addAll(aux(p, cr, visited));
        }
        cr.set(c, pre);
        return pre;
    }
}
