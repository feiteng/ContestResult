class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Map<Integer, List<Integer>> coursesByPrerequisite = new HashMap<>();
        for (int i = 0; i < n; i++) {
            coursesByPrerequisite.put(i, new ArrayList<>());
        }
        for (int[] p : prerequisites) {
            coursesByPrerequisite.get(p[0]).add(p[1]);
        }

        Map<Integer, Set<Integer>> coursesToPrerequisiteRecursive = new HashMap<>();
        for (int i = 0; i < n; i++) {
            coursesToPrerequisiteRecursive.put(i, new HashSet<>());
        }
        for (int i = 0; i < n; i++) {
            Queue<Integer> q = new LinkedList<>();
            q.add(i);
            boolean[] visited = new boolean[n];
            while (!q.isEmpty()) {
                int c = q.remove();
                if (visited[c]) {
                    continue;
                }
                visited[c] = true;
                coursesToPrerequisiteRecursive.get(c).add(i);
                q.addAll(coursesByPrerequisite.get(c));
            }
            
coursesToPrerequisiteRecursive.get(i).remove(i);
        }
        
        List<Boolean> out = new ArrayList<>();
        for (int[] q : queries) {
            out.add(coursesToPrerequisiteRecursive.get(q[1]).contains(q[0]));
        }
        return out;
    }
}
