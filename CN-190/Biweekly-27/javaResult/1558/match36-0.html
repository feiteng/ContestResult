class Solution {
      public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        List<Set<Integer>> list = new ArrayList<>();
        for (int i = 0;i<n;i++){
            list.add(new HashSet<>());
        }
        for (int[] each : prerequisites){
            list.get(each[1]).add(each[0]);
        }
        boolean[] visited = new boolean[n];
        for (int i = 0;i<n;i++){
            if(visited[i]){
                continue;
            }
            dfs(i,list,list.get(i),visited);
        }

        List<Boolean> res = new ArrayList<>(queries.length);
        for (int[] eachQuery : queries){
            res.add(list.get(eachQuery[1]).contains(eachQuery[0]));
        }
        return res;
    }

    private Set<Integer> dfs(int n,List<Set<Integer>> list,Set<Integer> eachSet,boolean[] visited){
        if(visited[n]){
            return list.get(n);
        }
        
        Set<Integer> cur = new HashSet<>();
        for (int eachP : eachSet){
            cur.add(eachP);
            Set<Integer> p = list.get(eachP);
            Set<Integer> pList =  dfs(eachP,list,p,visited);
            cur.addAll(pList);
        }

        for (Integer eachP : cur){
            eachSet.add(eachP);
        }

        visited[n] = true;
        return cur;
    }
}
