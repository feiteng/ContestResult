class Solution {
    public boolean hasAllCodes(String s, int k) {
        Set<String> subs = new HashSet<>();
        int possibilities = (int)Math.pow(2, k);
        for (int i = 0; i + k <= s.length(); i++) 
{
            String sub = s.substring(i, i + k);
            subs.add(sub);
        }
        
        return subs.size() == possibilities;
    }
}
