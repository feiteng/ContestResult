class Solution {
    public boolean hasAllCodes(String s, int k) {
        Set<String> kLenStrs = new HashSet<>();
        for (int i = 0, j = k; j <= s.length(); i++, j++) {
            kLenStrs.add(s.substring(i, j));
        }
        // System.out.println(kLenStrs);
        // System.out.println(2 << k);
        return kLenStrs.size() >= (1 << k);
    }
}
