class Solution {
public:
int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
	int m = nums1.size();
	int n = nums2.size();
	vector<vector<int>> dp(m, vector<int>(n, INT_MIN));
	int value = nums1[0] * nums2[0];
	dp[0][0] = value;
	for (int i = 1; i < m; i++) {
		value = max(value, nums2[0] * nums1[i]);
		dp[i][0] = value;
	}
	value = nums1[0] * nums2[0];
	for (int j = 1; j < n; j++) {
		value = max(value, nums2[j] * nums1[0]);
		dp[0][j] = value;
	}
	for (int i = 1; i < m; i++) {
		for (int j = 1; j < n; j++) {
			int value = nums1[i] * nums2[j];
			dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
			if (value >= 0) {
				dp[i][j] = max(dp[i][j], max(0,dp[i - 1][j - 1]) + value);
			}
		}
	}
	return dp[m-1][n-1];
}
};
