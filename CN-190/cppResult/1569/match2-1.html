class Solution {
public:
    
    
    
    
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        
        
        
        int n = nums1.size(), m = nums2.size();
        
        int dp[n + 1][m + 1];
        
        int max1, min1, max2, min2;
        max1 = min1 = nums1[0];

        max2 = min2 = nums2[0];
        
        for(int i = 0; i <= n; i++){
            dp[i][0] = 0;
        }
        for(int j = 0; j <= m; j++){
            dp[0][j] = 0;
        }
        
        int maxTerm = nums1[0] * nums2[0];
        
        for(int i = 0; i < n; ++i)
            for(int j = 0; j < m; ++j){
                maxTerm = max(maxTerm, nums1[i] * nums2[j]);
            }
        
        if(maxTerm < 0)return maxTerm;
        
        
        
        nums1.insert(nums1.begin(), 0);
        nums2.insert(nums2.begin(), 0);
        
        
        

        for(int i = 1; i <= n; ++i){
            for(int j = 1; j <= m; ++j){
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
                
                dp[i][j] = max(dp[i][j], dp[i-1][j-1] + nums1[i] * nums2[j]);
                
                /*int maxv, minv;
                maxv = minv = nums1[i];
                for(int k = i - 1; k >= 0; k--){
                    dp[i][j] = max(dp[i][j], dp[k][j - 1] + max(maxv * nums2[j], minv * nums2[j]));
                    maxv = max(maxv, nums1[k]); minv = min(minv, nums1[k]);
                }
                                   
                maxv = minv = nums2[j];
                for(int k = j - 1; k >= 0; k--){
                    dp[i][j] = max(dp[i][j], dp[i - 1][k] + max(maxv * nums1[i], minv * nums1[i]));
                    maxv = max(maxv, nums2[k]); minv = min(minv, nums2[k]);
                }*/
            }
        }
        
        return dp[n][m];
    }
};
