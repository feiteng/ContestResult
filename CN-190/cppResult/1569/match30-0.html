int mul[505][505],dp[505][505];
class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n=nums1.size(),m=nums2.size();
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                mul[i+1][j+1]=nums1[i]*nums2[j];
        int ans=INT_MIN;
        for(int i=1;i<=n;i++)
            for(int j=1;j<=m;j++)
            {
                dp[i][j]=max(max(dp[i-1][j],dp[i][j-1]),dp[i-1][j-1]+mul[i][j]);
                ans=max(ans,dp[i][j]);
            }
        if(!ans){
            ans=INT_MIN;
            for(int i=1;i<=n;i++)
                for(int j=1;j<=m;j++)
                    ans=max(ans,mul[i][j]);
        }
        return ans;
    }
};
