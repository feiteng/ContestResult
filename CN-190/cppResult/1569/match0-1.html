class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size();
        int n = nums2.size();
        int dp[m+1][n+1];

        memset(dp,0,sizeof(dp));
        for(int i = 0 ;i < m ;i++){
            for(int j = 0 ;j < n ;j++){
                //if(i> 0 && j > 0)
                    dp[i+1][j+1] = max(dp[i][j]+nums1[i]*nums2[j],max(dp[i][j],max(dp[i][j+1],dp[i+1][j])));
                /*else if(i > 0){
                    dp[i+1][j+1] = max(nums1[i]*nums2[j],dp[i][j+1]);
                }
                else if(j > 0){
                    dp[i+1][j+1] = max(nums1[i]*nums2[j],dp[i+1][j]);
                }
                else{
                    dp[i+1][j+1] = nums1[i]*nums2[j];
                }*/
            }
        }
        if(dp[m][n] == 0){

            int mmax = -2147483648;
            for(int i = 0 ;i < m ;i++){
                for(int j = 0 ;j < n ;j++){
                    mmax = max(mmax,nums1[i]*nums2[j]);
                }
            }
            return mmax;
        }
        return dp[m][n];
    }
};
