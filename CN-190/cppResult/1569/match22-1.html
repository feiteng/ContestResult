class Solution {
public:
    int maxDotProduct(vector<int>& num1, vector<int>& num2) {
        int rows = num1.size();
        int cols = num2.size();
        vector<vector<int>> dp(rows, vector<int>(cols, 0));
        int tmp1 = dp[0][0] = num1[0]*num2[0];
        for(int i=1; i<cols; i++){
             dp[0][i] = tmp1 = max(tmp1, num1[0]*num2[i]);
            //cout << dp[0][i] << " ";
        }
       // cout << endl;
        tmp1 = dp[0][0];
        for(int j=1; j<rows; j++) {
            dp[j][0] = tmp1 = max(tmp1, num1[j]*num2[0]);
            //cout << dp[j][0] << " ";
        }
       // cout << endl;
        for(int i=1; i<rows; i++) 
        {
            for(int j=1; j<cols; j++)
            {
                dp[i][j] = max( max(dp[i-1][j], dp[i][j-1]), dp[i-1][j-1] + num1[i]*num2[j]);
                dp[i][j] = max(dp[i][j], num1[i]*num2[j]);
                // cout << i << " " << j << " " << dp[i][j] << endl;
            }
           
        }
        return dp[rows-1][cols-1];
    }
};
