class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size();
        int n = nums2.size();
        int matrix[m][n]; // matrix[i][j] = nums1[i] * nums2[j]
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                matrix[i][j] = nums1[i] * nums2[j];
            }
        }

        int res = matrix[0][0];
        int dp[m][n];
        dp[0][0] = matrix[0][0];
        for(int i = 1; i < m; i++){
            dp[i][0] = max(dp[i - 1][0], nums1[i] * nums2[0]);
            res = max(res, dp[i][0]);
        } 
        for(int j = 1; j < n; j++){
            dp[0][j] = max(dp[0][j - 1], nums1[0] * nums2[j]);
            res = max(res, dp[0][j]);
        } 
        
        for(int i = 1; i < m; i++){
            for(int j = 1; j < n; j++){
                dp[i][j] = max(nums1[i] * nums2[j], max(max(dp[i - 1][j - 1], dp[i - 1][j - 1] + nums1[i] * nums2[j]), max(dp[i][j - 1], dp[i - 1][j])));
                res = max(res, dp[i][j]);
                //printf("%d ",dp[i][j]);
            }
            //printf("\n");
        }
        return res;
    }
};
