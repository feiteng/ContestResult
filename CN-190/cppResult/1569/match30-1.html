int dp[505][505];
class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
		int n=nums1.size(),m=nums2.size();
		for (int i=0;i<=n;++i)
			for (int j=0;j<=m;++j) dp[i][j]=-1e9;

		for (int i=1;i<=n;++i)
			for (int j=1;j<=m;++j)
			{
				dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
				dp[i][j]=max(dp[i][j],max(0,dp[i-1][j-1])+nums1[i-1]*nums2[j-1]);
			}
		return dp[n][m];
    }
};
