class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();
        vector<vector<int> > dp(n+1,vector<int> (m+1,-0x3f3f3f3f));
        for(int i=0;i<n;i++){
            dp[i][0] = 0;
        }
        for(int i=0;i<m;i++){
            dp[0][i] = 0;
        }
        int fl=0;
        int mx = -0x3f3f3f3f;
        int mm = mx;
        int res= -0x3f3f3f3f;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                int t=nums1[i-1]*nums2[j-1]; 
                if(t<=0){
                    dp[i][j] = max(dp[i-1][j],dp[i][j-1]);
                    mx = max(mx,t);
                }else{
                    dp[i][j] = dp[i-1][j-1] + t;
                    int x = max(dp[i-1][j],dp[i][j-1]);
                    dp[i][j] = max(dp[i][j],x);
                    fl = 1;
                }
                
                //cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
                res = max(res,dp[i][j]);
            }
        }
       // cout<<fl<<" "<<mx<<endl;
        if(fl)
        return res;
        else
            return mx;

    }
};
