class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size();
        int n2 = nums2.size();
        
        vector<vector<int>> dp(n1, vector<int>(n2, 0));
        
        dp[0][0] = nums1[0]*nums2[0];
        
        for(int i = 1; i < n1; i++){
            dp[i][0] = max(dp[i-1][0], nums1[i]*nums2[0]);
        }

        for(int j = 1; j < n2; j++){
            dp[0][j] = max(dp[0][j-1], nums1[0]*nums2[j]);

        }
        
        for(int i = 1; i < n1; i++){
            for(int j = 1; j < n2; j++){
                
                if(nums1[i]*nums2[j] >= 0){
                    dp[i][j] = max(dp[i-1][j-1] + nums1[i]*nums2[j],  nums1[i]*nums2[j]);
                }
                else{
                    dp[i][j] = nums1[i]*nums2[j] > dp[i-1][j-1]? nums1[i]*nums2[j] : dp[i-1][j-1];
                }
                
                //dp[i][j] = nums1[i]*nums2[j] < 0 ? dp[i-1][j-1] : dp[i-1][j-1] + nums1[i]*nums2[j];
                
                dp[i][j] = max(dp[i][j], max(dp[i-1][j], dp[i][j-1]));
            }
        }
        
        //  for(int i = 0; i < n1; i++){
        //     for(int j = 0; j < n2; j++){
        //         cout << dp[i][j] << " ";
        //     }
        //      cout << endl;
        // }
        
        return dp[n1-1][n2-1];
    }
};
