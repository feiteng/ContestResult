class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int len1 = nums1.size(), len2 = nums2.size();
        vector<vector<int>> V(len1+1, vector<int>(len2+1, 0));
        
        //for (int i=0; i<=len1; i++) V[i][0] = 0;
        //for (int j=0; j<=len2; j++) V[0][j] = 0;
        
        for (int i=1; i<=len1; i++) {
            for (int j=1; j<=len2; j++) {
                V[i][j] = nums1[i-1] * nums2[j-1];
                if (i>1) V[i][j] = max(V[i][j], V[i-1][j]);
                if (j>1) V[i][j] = max(V[i][j], V[i][j-1]);
                /*
                for (int k=j; k>=1; k--) {
                    V[i][j] = max(V[i][j], nums1[i-1]*nums2[k-1] + V[i-1][k-1]);
                }
                for (int k=i; k>=1; k--) {
                    V[i][j] = max(V[i][j], nums1[k-1]*nums2[j-1] + V[k-1][j-1]);
                }
                */
                V[i][j] = max(V[i][j], nums1[i-1] * nums2[j-1] + V[i-1][j-1]);
                //cout<<i<<j<<" ->"<<V[i][j]<<endl;
            }
        }
        return V[len1][len2];
    }
};
