class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {

        int dp[600][600];
        dp[0][0] = nums1[0] * nums2[0];
        for(int i = 1; i < nums2.size(); ++i) {
            dp[0][i] = max(dp[0][i - 1], nums1[0] * nums2[i]);
        }
        for(int i = 1; i < nums1.size(); ++i) {
            dp[i][0] = max(dp[i - 1][0], nums1[i] * nums2[0]);
        }
        for(int i = 1; i < nums1.size(); ++i) {
            for(int j = 1; j < nums2.size(); ++j) {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                dp[i][j] = max(dp[i][j], dp[i - 1][j - 1]);
                dp[i][j] = max(dp[i - 1][j - 1] + nums1[i] * nums2[j], dp[i][j]);
                dp[i][j] = max(dp[i][j], nums1[i] * nums2[j]);
                /*
                dp[i][j] = max(dp[i][j], dp[0][j]);
                dp[i][j] = max(dp[i][j], dp[i][0]);
                for(int k = 1; k <= i; ++k) {
                    dp[i][j] = max(dp[i][j], dp[k - 1][j - 1]);
                    dp[i][j] = max(dp[i][j], dp[k - 1][j - 1] + nums2[j] * nums1[k]);
                }
                for(int k = 1; k <= j; ++k) {
                    dp[i][j] = max(dp[i][j], dp[i - 1][k - 1]);
                    dp[i][j] = max(dp[i][j], dp[i - 1][k - 1] + nums2[k] * nums1[i]);
                }*/
            }
        }
        
        return dp[nums1.size() - 1][nums2.size() - 1];
    }
};
