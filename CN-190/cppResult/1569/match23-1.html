class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int INF = 0x3f3f3f3f;
        int n1 = nums1.size();
        int n2 = nums2.size();
        int dp[n1 + 1][n2 + 1];

        for (int i = 0; i <= n1; i++) {
            dp[i][0] = -INF;
        }
        for (int j = 0; j <= n2; j++) {
            dp[0][j] = -INF;
        }
        for (int i = 1; i <= n1; i++) {
            for (int j = 1; j <= n2; j++) {
                if (nums1[i - 1] * nums2[j - 1] + dp[i - 1][j - 1] < nums1[i - 1] * nums2[j - 1]) {
                    dp[i][j] = nums1[i - 1] * nums2[j - 1];
                } else {
                    dp[i][j] = nums1[i - 1] * nums2[j - 1] + dp[i - 1][j - 1];
                }
                dp[i][j] = max(dp[i][j], max(dp[i - 1][j], dp[i][j - 1]));
            }
        }
        return dp[n1][n2];
    }
};
