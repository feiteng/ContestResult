class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int l1 = nums1.size();
        int l2 = nums2.size();
        vector<vector<int>> dp(l1,vector<int> (l2,0));
        dp[0][0] = nums1[0]*nums2[0];
        for(int i = 1;i<l1;++i){
            dp[i][0] = max(dp[i-1][0],nums1[i]*nums2[0]); 
        }
        for(int i = 1;i<l2;++i){
            dp[0][i] = max(dp[0][i-1],nums1[0]*nums2[i]); 
        }
        for(int i = 1;i<l1;++i){
            for(int j = 1;j<l2;++j){
                dp[i][j] = max(dp[i][j-1],max(dp[i-1][j],max(dp[i-1][j-1],dp[i-1][j-1]+nums1[i]*nums2[j])));
                dp[i][j] = max(dp[i][j],nums1[i]*nums2[j]);
            }
        }
        return dp[l1-1][l2-1];
    }
};
