class Solution {
public:    
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        const int m=nums1.size();
        const int n=nums2.size();

        vector<vector<int>> dp(m,vector<int>(n,INT_MIN));
        dp[m-1][n-1]=nums1.back()*nums2.back();
        for(int i=m-2;i>=0;--i){
            dp[i][n-1]=max(nums1[i]*nums2[n-1],dp[i+1][n-1]);
        }
        for(int i=n-2;i>=0;--i){
            dp[m-1][i]=max(nums1[m-1]*nums2[i],dp[m-1][i+1]);
        }
        
        for(int i=m-2;i>=0;--i){
            for(int j=n-2;j>=0;--j){

                dp[i][j]=max(max(dp[i+1][j],dp[i][j+1]),max(nums1[i]*nums2[j],nums1[i]*nums2[j]+dp[i+1][j+1]));
            }
        }
        for(int i=0;i<m;++i){
            for(int j=0;j<n;++j){
                cout<<dp[i][j]<<" ";
            }
            cout<<endl;
        }
        return dp[0][0];
    }
};
