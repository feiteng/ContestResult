class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int ans[501][501];
        ans[0][0]=nums1[0]*nums2[0];
        for(int i=1;i<nums1.size();i++)
            ans[i][0]=max(ans[i-1][0],nums1[i]*nums2[0]);       
        for(int j=1;j<nums2.size();j++)
            ans[0][j]=max(ans[0][j-1],nums1[0]*nums2[j]);
        for(int i=1;i<nums1.size();i++){
            for(int j=1;j<nums2.size();j++){
                ans[i][j]=max(nums1[i]*nums2[j],ans[i-1][j-1]+nums1[i]*nums2[j]);
                ans[i][j]=max(ans[i-1][j],ans[i][j]);
                ans[i][j]=max(ans[i][j-1],ans[i][j]);                
            }
        }
        
        
         
     
        return ans[nums1.size()-1][nums2.size()-1];
    }
};
