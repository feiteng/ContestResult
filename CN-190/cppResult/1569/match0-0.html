class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int len1 = nums1.size();
        int len2 = nums2.size();
        int dp[505][505];
        int maxx = -0x3f3f3f3f;
        for(int i=0;i<len1;i++){
            for(int j=0;j<len2;j++){
                maxx = max(maxx, nums1[i]*nums2[j]);
            }
        }
        memset(dp,0,sizeof(dp));
        for(int i=1;i<=len1;i++){
            for(int j=1;j<=len2;j++){
                dp[i][j] = max(dp[i-1][j], max(dp[i][j-1], dp[i-1][j-1]+nums1[i-1]*nums2[j-1]));
            }
        }
        if(dp[len1][len2]==0){
            return maxx;
        }
        return dp[len1][len2];
    }
};
