class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int N1=nums1.size();
        int N2=nums2.size();
        vector<vector<int>> dp(N1+1,vector<int>(N2+1,-1000*1000));
        
        
        for(int i=0;i<N1;i++){
            for(int j=0;j<N2;j++){
                dp[i+1][j+1]=max(dp[i][j],nums1[i]*nums2[j]);
       
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i][j]+nums1[i]*nums2[j]);
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i+1][j]);
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i][j+1]);
            }
        }
        return dp[N1][N2];
    }
};
