/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int valCnt[10];
	int pseudoPalindromicPaths(TreeNode* root) {
		for (int i = 0; i < 10; ++i) {
			valCnt[i] = 0;
		}
		return dfs(root);
	}

	int dfs(TreeNode* node) {
		if (node == nullptr)
			return 0;
		int result = 0;
		++valCnt[node->val];
		if (node->left == nullptr && node->right == nullptr) {
			if (Check()) {
				result = 1;
			}

		}
		else {
			result = dfs(node->left) + dfs(node->right);
		}
		--valCnt[node->val];
		return result;
	}

	bool Check() {
		bool onlyOne = false;
		for (int i = 0; i < 10; ++i) {
			if ((valCnt[i] & 1) == 1) {
				if (!onlyOne) {
					onlyOne = true;
				}
				else {
					return false;
				}
			}
		}
		return true;
	}
};
