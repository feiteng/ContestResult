/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution 
{
public:
	int pseudoPalindromicPaths(TreeNode* root) 
	{
		vector<int>cnt(10);
		return work(root, cnt);
	}

	int work(TreeNode* root, vector<int>&cnt)
	{
		int ret = 0;
		cnt[root->val]++;
		if (root->left == NULL && root->right == NULL)
		{
			int k = 0;
			for (int i = 1; i <= 9; i++)
				if (cnt[i] % 2 == 1)
					k++;
			if (k <= 1)
				ret = 1;
		}
		if (root->left)
			ret += work(root->left, cnt);
		if (root->right)
			ret += work(root->right, cnt);
		cnt[root->val]--;
		return ret;
	}
};
