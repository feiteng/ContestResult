/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    int sum=0;
    int a[10];
public:
    bool judge(int t)
    {
        int sum=0;
        while(t>0)
        {
            sum+=t & 0x01;
            t>>=1;
        }
        return sum<2;
    }
    
    void search(TreeNode* t,int m)
    {

        m^=a[t->val];
        if(t->left==nullptr && t->right==nullptr)
        {
            //cout<<m<<endl;
            if(judge(m))
                sum++;
            return;
        }
        if(t->left!=nullptr)
        search(t->left,m);
        if(t->right!=nullptr)
        search(t->right,m);
    }
    
    int pseudoPalindromicPaths (TreeNode* root) {
        if(root == nullptr)
            return sum;
        int t=1;
        a[1]=1;
        for(int i=2;i<10;i++)
        {
            t<<=1;
            a[i]=t;
        }
        search(root,0);
        return sum;
    }
};
