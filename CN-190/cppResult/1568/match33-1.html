/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class Solution {
public:
    int ans = 0;
    void helper(TreeNode* p, vector<int> nums){
        if(p == NULL) return;
        nums[p->val]++;
        if(p->left == NULL && p->right == NULL){
            int count = 0;
            for(int i = 1; i <= 9; i++)
                if(nums[i] % 2 == 1) count++;
            if(count <= 1) ans++;
            return;
        }
        if(p->left != NULL) helper(p->left, nums);
        if(p->right != NULL) helper(p->right, nums);
        return;
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> nums(10, 0);
        helper(root, nums);
        return ans;
    }
};
