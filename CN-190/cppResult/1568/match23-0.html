class Solution {
public:
    int c[14],ans = 0;
    bool ok(){
        int s = 0;
        for(int i = 0;i < 10;i++)
            s += c[i] & 1;
        return s <= 1;
    }
    void dfs(TreeNode * r){
        if(r == NULL) return;
            ++c[r->val];
        if(r->left == NULL && r->right == NULL)
            ans += ok();
        dfs(r->left);dfs(r->right);
            --c[r->val];
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        dfs(root);
        return ans;
    }
};
