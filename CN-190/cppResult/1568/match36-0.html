/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int res=0;
    bool get(vector<int>v){
        int ans=0;
        for(int i=1;i<10;i++){
            if(v[i]%2)
                ans++;
        }
        if(ans>1)
            return false;
        return true;
    }
    void solver(TreeNode *root,vector<int>v){
        v[root->val]++;
        if(root->left==NULL&&root->right==NULL){
            if(get(v))
                res++;
            return;
        }
        if(root->left)
            solver(root->left,v);
        if(root->right)
            solver(root->right,v);
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        if(root==NULL)
            return 0;
        vector<int>v(10,0);
        solver(root,v);
        return res;
    }
};
