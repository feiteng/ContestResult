/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int res;
    

    inline bool ispsePalPath(vector<int> cur){
        unordered_map<int,int> mem;
        for(auto i : cur){
            ++mem[i];
        }
        int odd=0;
        for(auto i:mem){
            if(i.second&1){
                ++odd;
                if(odd>1){
                    return false;
                }
            }
        }
        return true;
    }
    
    void helper(TreeNode* root,vector<int> cur){
        cur.push_back(root->val);
        if(!root->left && !root->right && ispsePalPath(cur)){

            ++res;
            return;
        }
        if(root->left){
            helper(root->left,cur);
        }
        if(root->right){
            helper(root->right,cur);
        }
    }
    
    int pseudoPalindromicPaths (TreeNode* root) {
        res=0;
        vector<int> cur;
        helper(root,cur);
        return res;
    }
};
