class Solution {
public:
    int maxVowels(string s, int k) {
        int n=s.size();
        vector<int> dp(n+10, 0);
        for(int i=0;i<n;i++){
            dp[i+1]=dp[i];
            if(s[i]=='a'||s[i]=='e'|| s[i]=='i'|| s[i]=='o'|| s[i]=='u'){
                dp[i+1]++;
            }
        }
        int ret=0;
        for(int i=k;i<=n;i++){
            ret=max(ret,dp[i]-dp[i-k]);
        }
        return ret;

    }
};
