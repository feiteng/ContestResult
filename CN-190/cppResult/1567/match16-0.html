class Solution {
public:
    
    bool ck(char ch)
    {
        if(ch=='a' || ch=='e'|| ch=='i'|| ch=='o'|| ch=='u')
            return true;
        else
            return false;
    }
    int maxVowels(string s, int k) {
        int cnt=0;
        int maxn=0;
        
        for(int i=0;i<k;i++)
        {
            if(ck(s[i]))
                cnt++;
        }
        maxn = cnt;
        
        for(int i=0;i+k<s.length();i++)
        {
            if(ck(s[i]))
                cnt--;
            if(ck(s[i+k]))
                cnt++;
            
            maxn = max(maxn, cnt);
        }
        
        return maxn;
    }
};
