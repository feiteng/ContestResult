class Solution {
public:
    int ans=0;

    int maxVowels(string s, int k) {
        int n =s.length();
        for(int i=0;i<k;i++) if(flag(s[i])) ans++;
        int t=ans;
        for(int i=k;i<n;i++){
            if(flag(s[i])) t++;
            if(flag(s[i-k])) t--;
            ans=max(ans,t);
        }
        return ans;
    }
    bool flag(char c){
        if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u') return true;
        return false;
    }
};
