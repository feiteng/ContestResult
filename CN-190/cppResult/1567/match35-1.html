class Solution {
public:
    int maxVowels(string s, int k) {
        int max = 0;
        for (int i = 0; i < k; i++) {
            if (is(s[i])) max++;
        }

        int tmp = max;
        for (int i = k ; i < s.length(); i++) {
            if (is(s[i - k])) tmp--;
            if (is(s[i])) tmp++;
            max = max > tmp ? max : tmp;
        }
        
        return max;
    }
    
    bool is(char a) {
        if (a == 'a' || a == 'o' || a == 'i' || a == 'e' || a == 'u') return true;
        return false;
    }
};
