class Solution {
public:
    int maxVowels(string s, int k) {
        vector<int> count(s.size()+1, 0);
        for (int i = 0; i < s.size(); i++)
        {
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
            {
                count[i+1] = count[i] + 1;
            }
            else 
            {
                count[i+1] = count[i];
            }
        }
        
        int max_value = -1;
        for (int j = 1; j < s.size() - k + 2; j++)
        {
            if (count[j+k-1] - count[j-1] > max_value) max_value = count[j+k-1] - count[j-1];
        }
        return max_value;
    }
};
