class Solution {
public:
    int maxVowels(string s, int k) {
        vector<int> vec(s.size() + 1, 0);
        for (int i = 0; i < s.size(); ++ i) {
            vec[i + 1] = vec[i];
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
                vec[i + 1] ++;
        }
        
        int maxNum = 0;
        for (int i = k; i < vec.size(); ++ i)
            maxNum = max(maxNum, vec[i] - vec[i-k]);
        
        return maxNum;
    }
};
