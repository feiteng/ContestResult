class Solution {
public:
    int maxVowels(string s, int k) {
        int ans = 0, num = 0;
        for(int i = 0; i < k; i++) {
            if(check(s[i])) num++;
        }
        ans = num;
        for(int i = k; i < s.size(); i++) {
            if(check(s[i - k])) num--;
            if(check(s[i])) num++;
            ans = max(ans, num);
        }
        return ans;
    }
    bool check(char c) {
        if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') return true;
        return false;
    }
};
