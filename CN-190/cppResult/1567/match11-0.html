class Solution {
public:
    int help(char c){
        if(c=='a')return 1;
        if(c=='e')return 1;
        if(c=='i')return 1;
        if(c=='o')return 1;
        if(c=='u')return 1;
        return 0;
    }
    int maxVowels(string s, int k) {
        int m=0;
        for(int i=0;i<k;i++)
            m+=help(s[i]);
        int M=m;
        for(int i=k;i<s.size();i++){
            m+=help(s[i]);
            if(help(s[i-k]))m--;
            M=max(m,M);
        }
        return M;
    }
};
