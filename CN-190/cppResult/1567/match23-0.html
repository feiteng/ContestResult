class Solution {
public:
    int maxVowels(string s, int k) {
        unordered_set<char> se;
        se.insert('a');
        se.insert('e');
        se.insert('o');
        se.insert('i');
        se.insert('u');
        int x=0;
        int m=0;
        for(int i=0;i<k;i++)
            if(se.count(s[i]))
                x++;
        m=x;
        for(int i=k;i<s.size();i++)
        {
            if(se.count(s[i-k]))
                x--;
            if(se.count(s[i]))
                x++;
            m=max(x,m);
        }
        return m;
    }
};
