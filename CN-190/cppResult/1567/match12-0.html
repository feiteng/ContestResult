class Solution {
public:
	int maxVowels(string s, int k) 
	{
		//aeiou
		vector<int>sum(s.size() + 1, 0);
		for (int i = 0; i < s.size(); i++)
		{
			sum[i + 1] = sum[i];
			if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
				sum[i + 1]++;
		}
		int ret = 0;
		for (int i = k; i < s.size() + 1; i++)
			if (sum[i] - sum[i - k] > ret)
				ret = sum[i] - sum[i - k];
		return ret;
	}
};
