class Solution {
public:
    int maxVowels(string s, int k) {
        vector<char> word_vec = {'a', 'e', 'i', 'o', 'u'};

        int max_length = -1;
        int word_sum = 0;
        for(int i = 0; i < s.size();i++){
            if(find(word_vec.begin(), word_vec.end(), s[i]) != word_vec.end()){
                word_sum += 1;
            }
            if(i - k + 1 > 0){
                if(find(word_vec.begin(), word_vec.end(), s[i-k]) != word_vec.end()){
                    word_sum -= 1;
                }
            }
            max_length = max(max_length, word_sum);
        }
        return max_length;
    }
};
