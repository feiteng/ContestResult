class Solution {
public:
    int maxVowels(string s, int k) {
        unordered_set<char> sets;
        sets.insert('a');
        sets.insert('e');
        sets.insert('i');
        sets.insert('o');
        sets.insert('u');
        
        int res = 0;
        int cur = 0;
        for (int i = 0; i < k; ++i) {
            if (sets.count(s[i])) {
                ++cur;
            }
        }
        res = cur;
        for (int i = k; i < s.size(); ++i) {
            if (sets.count(s[i])) {
                ++cur;
            }
            if (sets.count(s[i-k])) {
                --cur;
            }
            res = max(res, cur);
        }
        return res;
    }
};
