class Solution {
public:
    int maxVowels(string s, int k) {
        int n = s.length();
        vector<int> a(n+1);
        for (int i = 0; i < n; i++) {
            a[i+1] = a[i];
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') a[i+1]++;

        }
        int ans = 0;
        for (int i = k; i <= n; i++) {
            ans = max(a[i]-a[i-k],ans);
        }
        return ans;
        
    }
};
