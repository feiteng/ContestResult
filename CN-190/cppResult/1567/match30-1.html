class Solution {
public:
    int maxVowels(string s, int k) {
        int n=s.size();
        vector<int> dp(n+1);
        dp[0]=0;
        for(int i=1;i<=n;++i){
            dp[i]=dp[i-1];

            if(s[i-1]=='a'||s[i-1]=='e'||s[i-1]=='i'||s[i-1]=='o'||s[i-1]=='u'){
                ++dp[i];
            }
        }
        int ans=0;
        for(int i=k;i<=n;++i){
            ans=max(ans,dp[i]-dp[i-k]);
        }
        return ans;
    }
};
