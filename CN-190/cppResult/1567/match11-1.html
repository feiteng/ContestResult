class Solution {
public:
    int c2i(char x){
        if (x == 'a') return 1;
        if (x == 'e') return 1;
        if (x == 'i') return 1;
        if (x == 'o') return 1;
        if (x == 'u') return 1;
        return 0;
    }
    int maxVowels(string s, int k) {
        int n = s.length();
        int sum = 0;

        for (int i = 0; i < k; i++) sum += c2i(s[i]);
        
        int ans = sum;
        for (int i = k; i < n; i++){
            sum += c2i(s[i]) - c2i(s[i - k]);
            ans = max(ans, sum);
        }
        return ans;
    }
};
