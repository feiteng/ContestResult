class Solution {
public:
    int maxVowels(string s, int k) {
        int res=0;
        for(int i=0;i<k;++i)
            if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
                ++res;
        int t=res;
        for(int i=k;i<s.size();++i){
            if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
                ++t;
            if(s[i-k]=='a'||s[i-k]=='e'||s[i-k]=='i'||s[i-k]=='o'||s[i-k]=='u')
                --t;
            res=max(res,t);
        }
        return res;
    }
};
