class Solution {
public:
    int maxVowels(string s, int k) {
        int res=0, f=0;
        set<char> yuanyin;
        yuanyin.insert('a');
        yuanyin.insert('e');
        yuanyin.insert('i');
        yuanyin.insert('o');
        yuanyin.insert('u');
        
        for(int i=0;i<k;i++){
            if(yuanyin.find(s[i])!=yuanyin.end()){
                f+=1;
            }    
        }
        res=max(res, f);
        // sliding window
        for(int i=k;i<s.size();i++){
            if(yuanyin.find(s[i-k])!=yuanyin.end()){
                f-=1;
            }
            if(yuanyin.find(s[i])!=yuanyin.end()){
                f+=1;
            }
            res=max(res, f);
            
        }
        
        return res;
        
    }
};
