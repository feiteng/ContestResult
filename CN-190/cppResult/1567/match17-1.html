class Solution {
public:
    int maxVowels(string s, int k) {
        vector<int> aeiou(s.length() + 1, 0);
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') {
                aeiou[i + 1] = aeiou[i] + 1;
            } else {
                aeiou[i + 1] = aeiou[i];
            }
        }
        
        int maxk = 0;
        for (int i = k - 1; i < s.length(); i++) {
            maxk = max(maxk, aeiou[i + 1] - aeiou[i + 1 - k]);
            if (maxk == k) return maxk;
        }
        
        return maxk;
    }
};
