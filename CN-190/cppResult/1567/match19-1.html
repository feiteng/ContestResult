class Solution {
public:
    int maxVowels(string s, int k) {

        int start=0;
        unordered_set<char> st;
        st.insert('a');
        st.insert('e');
        st.insert('i');
        st.insert('o');
        st.insert('u');
        for(int i=0;i<k;i++){
            if(st.find(s[i])!=st.end()){
                start++;
            }
        }

        int tmpRes=start;
        for(int i=k;i<s.size();i++){
            if(st.find(s[i])!=st.end()){
                start++;
            }
            if(st.find(s[i-k])!=st.end()){
                start--;
            }
            tmpRes=max(tmpRes,start);
        }
        return tmpRes;
    }
};
