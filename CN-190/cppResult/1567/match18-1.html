class Solution {

public:
    int maxVowels(string s, int k) {
        int mx=0;
        int c=0;
        for(int i=0;i<k;i++)
            if(s[i]=='a'||s[i]=='o'||s[i]=='e'||s[i]=='i'||s[i]=='u'){
                c++;
            }

        mx=c;
        for(int i=k;i<s.size();i++){
            c+=(s[i]=='a'||s[i]=='o'||s[i]=='e'||s[i]=='i'||s[i]=='u');
            c-=(s[i-k]=='a'||s[i-k]=='o'||s[i-k]=='e'||s[i-k]=='i'||s[i-k]=='u');
            mx=max(mx,c);
        }
        return mx;
    }
};
