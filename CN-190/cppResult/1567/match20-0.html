class Solution {
public:
    bool check(char c) {
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
            return true;
        return false;
    }
    int maxVowels(string s, int k) {
        int sum = 0;
        int ans = 0;
        int l = s.size();
        for (int i = 0; i < l ; i++) {
            if (check(s[i]))
                sum++;
            if (i >= k && check(s[i-k]))
                sum--;
            ans = max(ans, sum);
        }
        return ans;
    }
};
