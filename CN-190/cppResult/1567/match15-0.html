class Solution {
public:
    bool judge(char c){
        return c=='a'||c=='o'||c=='e'||c=='i'||c=='u';
    }
    int maxVowels(string s, int k) {
        queue<char>c;
        int res=0;
        for(int i=0;i<k;i++){
            c.push(s[i]);
            if(judge(s[i])){
                res++;
            }
        }
        int now=res;
        for(int i=k;i<s.size();i++){
            if(judge(c.front())){
                now--;
            }
            c.pop();
            c.push(s[i]);
            if(judge(s[i])){now++;}
            res=max(res,now);
        }
        return res;
    }
};
