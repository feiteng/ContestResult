class Solution {
public:
    int maxVowels(string s, int k) {
        set<char> vowel{'a', 'e', 'i', 'o', 'u'};
        int left=0;
        int right=left+k;
        int result=0;
        int count=0;
        for(int i=0;i<k;i++){
            if(vowel.find(s[i])!=vowel.end())
                result++;
        }
        count=result;
        for(int i=k;i<s.size();i++){
            if(vowel.find(s[i])!=vowel.end())
                count++;
            if(vowel.find(s[i-k])!=vowel.end())
                count--;
            result=max(result,count);
        }
        return result;
    }
};
