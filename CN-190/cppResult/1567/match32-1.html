class Solution {
public:
    bool yuanyin(char ch){
        if(ch == 'a' || ch == 'e' || ch=='i'||ch == 'o' || ch == 'u')return true;
        return false;
    }
    int maxVowels(string s, int k) {
        int count = 0;
        int res = 0;
        for(int i = 0 ;i < k ;i++){
            if(yuanyin(s[i]))count++;
        }
        res = max(res,count);

        int beg = 0;
        for(int i = k ; i < s.size() ;i++){
            if(yuanyin(s[beg])) count--;
            if(yuanyin(s[i]))count++;
            res = max(res,count);
            if(res >= k)return res;
            beg++;
        }
        return res;
    }
};
