class Solution {
private:
    unordered_set<char> vowel{'a','e','i','o','u'};
public:
    int maxVowels(string s, int k) {
        int n=s.size();
        int count=0;
        for (int i=0; i<min(k,n); i++) 
            if (vowel.find(s[i])!=vowel.end())
                count++;
        int res=count;
        for (int i=k; i<n; i++) {
            count+=(vowel.find(s[i])!=vowel.end());
            count-=(vowel.find(s[i-k])!=vowel.end());
            res=max(res,count);
        }
        return res;
    }
};
