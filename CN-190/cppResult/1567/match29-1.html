class Solution {
public:
    int maxVowels(string s, int k) {
        set<char> vowels;
        vowels.insert('a');
        vowels.insert('e');
        vowels.insert('i');
        vowels.insert('o');
        vowels.insert('u');

        int left = 0; 
        int right = k - 1;
        int answer = 0;
        int cur = 0;
        for (int i = left; i <= right; i++) {
            if (vowels.count(s[i])) {
                cur++;
            }

        }
        answer = cur;
        for (int i = right + 1; i < s.length(); i++) {
            if (vowels.count(s[i - k])) cur--;
            if (vowels.count(s[i])) cur++;
            answer = max(answer, cur);
        }
        return answer;
    }
};
