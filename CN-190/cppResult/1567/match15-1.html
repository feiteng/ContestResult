class Solution {
public:
    int maxVowels(string s, int k) {
        unordered_set<char> vowels = {'a', 'e', 'i', 'o', 'u'};

        int num = 0;
        for(int i=0;i<k;i++){
            if(vowels.find(s[i]) != vowels.end()){
                num++;
            }
        }
        int max_num = num;
        for(int i = k; i<s.size(); i++){
            if(vowels.find(s[i - k]) != vowels.end()){
                num--;
            }
            if(vowels.find(s[i]) != vowels.end()){
                num++;
            }
            max_num = max(num, max_num);
        }
        return max_num;
    }
};
