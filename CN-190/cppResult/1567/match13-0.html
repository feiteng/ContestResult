class Solution {
public:
    int maxVowels(string s, int k) {
        int res = 0;
        set<char> st = {'a', 'e', 'i', 'o', 'u'};
        int tmp = 0;
        for(int i = 0; i < k; ++i) {
            if(st.find(s[i]) != st.end()) {
                tmp++;
            }
        }
        res = max(res, tmp);
        for(int i = 1; i <= s.length() - k; ++i) {
            if(st.find(s[i - 1]) != st.end()) {
                tmp--;
            }
            if(st.find(s[i + k - 1]) != st.end()) {
                tmp++;
            }
            res = max(res, tmp);
        }
        return res;
    }
};
