class Solution {
public:
    int maxVowels(string s, int k) {
        vector<int> vOK(256);
        vOK['a']=1;
        vOK['e']=1;
        vOK['i']=1;
        vOK['o']=1;
        vOK['u']=1;
        int N=s.size();

        int ans=0;
        int sum=0;
        for(int i=0;i<k;i++){
            if(vOK[s[i]]>0){
                sum++;
            }
        }
        ans=sum;
        for(int i=k;i<N;i++){
            if(vOK[s[i]]>0){
                sum++;
            }
            if(vOK[s[i-k]]>0){
                sum--;
            }
            ans=max(ans,sum);
        }
        return ans;
    }
};
