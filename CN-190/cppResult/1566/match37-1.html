class Solution {
public:
	int isPrefixOfWord(string sentence, string searchWord) {
		int cnt = 0;
		int len = sentence.size();
		int slen = searchWord.size();
		for (int i = 0; i < len; ++i)
		{
			++cnt;
			bool find = true;
			int j = 0;
			for (; i < len; ++i)
			{
				if (sentence[i] == ' ')
				{
					break;
				}
				else if( j < slen)
				{
					if ( find && sentence[i] == searchWord[j++])
					{
						if( j == slen)
							return cnt;
					}
					else
					{
						find = false;
					}
				}
			}
		}
		return -1;
	}
};
