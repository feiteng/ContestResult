class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        string temp = "";
        vector<string> res;
         for(int i = 0;i<sentence.size();++i){
            if(sentence[i] != ' '){
                temp.push_back(sentence[i]);
            }
             else {
                 res.push_back(temp);
                 temp = "";
             }
        }
        res.push_back(temp);
        int len = searchWord.size();
        for(int i = 0;i<res.size();++i){
            if(res[i].size()<len){
                continue;
            }
            ///cout<<res[i].substr(0,len)<<endl;
            if(res[i].substr(0,len) == searchWord){
                
                return i+1;
            }
        }
        return -1;
    }
};
