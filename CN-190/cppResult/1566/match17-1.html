class Solution {
public:
int isPrefixOfWord(string sentence, string searchWord) {
    int ssz = sentence.size(), wsz = searchWord.size(), cnt = 0;
    for (int i = 0; i < ssz; i++)
        if ((i == 0 || sentence[i] == ' ') 
            && ++cnt &&
            sentence.compare(i==0?0:(i+1), wsz,
                searchWord) == 0)
            return cnt;
    return -1;
}
};
