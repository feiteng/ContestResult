class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int cnt = 0;
        istringstream in(sentence);
        string str;
        while(in>>str)
        {
            ++cnt;
            int i = 0, j = 0;
            //cout<<str<<endl;
            for(;i < searchWord.size() && j < str.size(); i++, j++)
            {
                if(searchWord[i] != str[j])break;
            }
            if(i == searchWord.size())return cnt;
        }
        return -1;
    }
};
