bool judge(string a, string b) {
    if (a.size() < b.size()) {
        return false;
    }
    for (int i = 0; i < b.size(); i++) {
        if (a[i] != b[i])
            return false;
    }
    return true;
}

class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int last = 0;
        int cnt = 0;
        for (int i = 0; i < sentence.size(); i++) {
            if (sentence[i] == ' ') {
                cnt++;
                if (judge(sentence.substr(last, i-last), searchWord)) {
                    return cnt;
                    break;
                }
                last = i+1;
            }
        }
        cnt++;
        if (judge(sentence.substr(last), searchWord)) {
            return cnt;
        }
        return -1;
    }
};
