class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream ss(sentence);
        string t;
        for(int i = 1 ; ss >> t ; i++) {
            if(t.size() >= searchWord.size() && t.substr(0,searchWord.size()) == searchWord) return i;
        }
        return -1;
    }
};
