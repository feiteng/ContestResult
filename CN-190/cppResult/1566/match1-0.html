class Solution {
public:
    stringstream ss;
    string s;
    bool ok(string s,string t){
        return s.size() >= t.size() && s.substr(0,t.size()) == t;
    }
    int isPrefixOfWord(string sentence, string searchWord) {
        ss << sentence;
        int g = 0;
        while(ss >> s){
            ++g;
            if(ok(s,searchWord)) return g;
        }
        return -1;
    }
};
