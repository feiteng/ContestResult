class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
		int wordIdx = 1;
		int subIdx = 0;
		for (int i = 0; i < sentence.size(); ++i) {
			if (sentence[i] == ' ') {
				++wordIdx;
				subIdx = 0;
			}
			else {
				if (sentence[i] == searchWord[subIdx]) {
					++subIdx;
					if (searchWord.size() == subIdx) {
						return wordIdx;
					}
				}
				else {
					while (i < sentence.size() && sentence[i] != ' ') {
						++i;
					}
					++wordIdx;
					subIdx = 0;
				}
			}
		}
		return -1;
	}
};
