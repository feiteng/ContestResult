class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        
        vector<string> v;
        string temp;
        stringstream ss(sentence);
        while(ss>>temp)
        {
            v.push_back(temp);
        }
        int size=v.size();
        for(int i=1;i<=size;i++)
        {
            if(searchWord==v[i-1].substr(0,searchWord.size()))
                return i;
        }
        return -1;
    }
};
