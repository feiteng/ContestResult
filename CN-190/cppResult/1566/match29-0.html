class Solution {
public:
    bool isMatch(string a, string b){
        string tmp = a.substr(0, b.size());
        if(tmp == b) return true;
        return false;
    }
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream ss(sentence);
        string tmp = "";
        int cnt = 1;
        bool flag = false;
        while( ss >> tmp){
            if(isMatch(tmp, searchWord)){
                return cnt;
            }
            cnt++;
        }
        
        return -1;
    }
};
