class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int q = 0;
        int result = 1;
        while(q < sentence.size())
        {
            if(q + searchWord.size()-1 >= sentence.size())
                break;
            if(sentence.substr(q, searchWord.size()) == searchWord)
                return result;
            while(q < sentence.size() && sentence[q] != ' ')
                q++;
            if(q < sentence.size() && sentence[q] == ' ')
                q++;
            result ++;
        }
        return -1;
    }
};
