class Solution {
public:
    string word[100];
    int isPrefixOfWord(string sentence, string searchWord) {
     stringstream out(sentence);
        int cnt=1;
     while(out>>word[cnt])

         cnt++;
      for(int i=1;i<=cnt;i++)
      {
          if(word[i].find(searchWord)==0)
              return i;
       }
        return -1;
    }
};
