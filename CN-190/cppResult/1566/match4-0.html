class Solution {
public:
    int isPrefixOfWord(string s, string w) {
        int index = 1;
        for(int i = 0; i < s.size(); ++i) {
            if(i == 0 || s[i-1] == ' ') {
                int j = 0;
                while(i+j < s.size() && j < w.size() && s[i+j] == w[j])
                    ++j;
                if(j == w.size())
                    return index;
                ++index;
            }
        }
        return -1;
    }
};
