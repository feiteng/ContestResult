class Solution {
public:
    string str1,str2;
    
    bool check(int id){
        int len = str2.length();
        for(int i=0;i<len;i++)
        {
            if(str1[id+i]!=str2[i])
                return false;
        }
        return true;
    }
    int isPrefixOfWord(string sentence, string searchWord) {
        int len = sentence.length();
        int cnt=0;
        str1 = sentence;
        str2 = searchWord;
        for(int i=0;i<len;i++){
            if(i==0 || sentence[i-1]==' '){
                cnt++;
                if(check(i)){
                    return cnt;
                }
            }
        }
        return -1;
    }
};
