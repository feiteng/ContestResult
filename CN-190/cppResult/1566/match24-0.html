class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int i = 0;
        int num = 1;
        while(i<sentence.size()){
            if(sentence[i]==searchWord[0]&&(i == 0||sentence[i-1]==' ')){
                i+=1;
                int j = 1;
                while(j < searchWord.size()){
                    if(sentence[i] == searchWord [j]){
                        i+=1;
                        j+=1;
                    }
                    else break;
                }
                if(j ==searchWord.size()) return num;
                
            }
            else i+=1;
            if(sentence[i] == ' ') num+=1;
        }
            return -1; 

    }
};
