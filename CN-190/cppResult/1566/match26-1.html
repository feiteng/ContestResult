class Solution {
public:
    int isPrefixOfWord(string sen, string sw) {

        int ind=0,cnt=0;
        while(ind<sen.size()){
            ++cnt;
            int j=ind,i=0;
            for(;i<sw.size()&&j<sen.size();i++,j++){
                if(sen[j]!=sw[i])break;
            }
            if(i==sw.size()){
                return cnt;
            }
            while(ind<sen.size()&&sen[ind]!=' ')ind++;
            ++ind;
        }
        return -1;
    }
};
