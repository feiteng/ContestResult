class Solution {
public:
    bool isMatch(string a, string b)
    {
        int x = a.size();
        int y = b.size();
        if(y > x) return false;
        int i=0;
        while(i<y)
        {
            if(a[i] != b[i])
                return false;
            ++i;
        }
        return true;
    }
    int isPrefixOfWord(string sen, string se) {
        vector<string> s;
        int len = sen.size();
        string temp;
        for(int i=0; i<len; ++i)
        {
            if(sen[i] != ' ')
            {
                temp += sen[i];
            }
            else
            {
                if(temp != "")
                    s.emplace_back(temp);
                temp = "";
            }
        }
        if(temp != "")
            s.emplace_back(temp);

        int n = s.size();
        for(int i=0; i<n; ++i)
        {
            if(isMatch(s[i], se))
                return i+1;
        }
        return -1;
    }
};
