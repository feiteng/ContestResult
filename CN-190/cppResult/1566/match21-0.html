class Solution {
public:
    bool fun(string s1, string s2)
    {
        if(s1.length() < s2.length()) return false;
        int p = 0;
        while(p < s2.length() && s1[p] == s2[p]) ++p;
        if(p == s2.length()) return true;
        return false;
    }
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream ss;
        ss << sentence;
        string str;
        int cnt = 1;
        while(ss >> str)
        {
            if(fun(str, searchWord)) return cnt;
            ++cnt;
        }
        return -1;
    }
};
