class Solution {
public:

    int isPrefixOfWord(const string& sentence, const string& searchWord) {
        stringstream ss(sentence);
        string tmp;int tot = 0;
        while(ss >> tmp) {
            ++tot;
            if(tmp.find(searchWord) == 0)
                return tot;
        }
        return -1;
    }
};
