class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int wordCnt = 1;
        int start = 1;
        for (int i = 0; i < sentence.length(); i++) {
            if (sentence[i] == ' ') {
                wordCnt++;
                start = 1;
                continue;
            }
            
            if (start == 0) {
                continue;
            }

            int match = 1;
            for (int j = 0; j < searchWord.length(); j++, i++) {
                if (sentence[i] == searchWord[j]) continue;
                
                match = 0;
                if (sentence[i] == ' ') {
                    wordCnt++;
                    start = 1;
                } else {
                    start = 0;
                }
                break;
            }
            
            if (match) return wordCnt;
        }
        
        return -1;
    }
};
