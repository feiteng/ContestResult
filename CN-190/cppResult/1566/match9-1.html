class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        sentence += ' ';
        vector<string> str;
        string tmp;
        for (int i = 0; i < sentence.size(); i++) {
            if (sentence[i] == ' ') {
                if (tmp != "")
                str.push_back(tmp);
                tmp = "";

            } else {
                tmp += sentence[i];
            }
        }
        int ans = -1;
        for (int i = 0; i < str.size(); i++) {
            if (str[i].size() >= searchWord.size()) {
                string t = "";
                for (int j = 0; j < searchWord.size(); j++) {
                    t += str[i][j];
                }
                cout << "t = " << t << endl;
                if (t == searchWord) {
                    ans = i + 1;
                    break;
                }
            }
        }
        return ans;
    }
};
