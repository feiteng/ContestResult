class Solution {
    bool check(string &s, string &t) {
        if (s.size() < t.size()) return false;
        return s.substr(0, t.size()) == t;
    }

   public:
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream s;
        s.str(sentence);
        string word;

        int now = 1;
        while (s >> word) {
            if (check(word, searchWord)) {
                return now;
            }
            now++;
        }
        return -1;
    }
};
