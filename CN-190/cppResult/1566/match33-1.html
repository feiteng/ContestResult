class Solution {
public:
    int isPrefixOfWord(string s, string se) {
        stringstream ss;
        ss<<s;
        int i=1;
        string t;
        while(ss>>t){
            int size=se.size();
            if(t.size()>=se.size()&&t.substr(0,size)==se) return i;
            i++;
        }
        return -1;
    }
};
