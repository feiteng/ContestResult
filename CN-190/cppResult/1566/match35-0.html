class Solution {
public:
    int isPrefixOfWord(string s, string t) {
        int ans = -1;
        int len = t.size();
        stringstream ss(s);
        string sss;
        int p = 0;
        while(ss >> sss){
            ++ p;
            int u = sss.size();
            if(u < len) continue;
            if(sss.substr(0 , len) == t){
                return p;
            }
            
        }
        return ans;
    }
};
