class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        sentence+=" ";
        vector<string> container;
        string temp="";
        for (int i=0;i<sentence.size();i++){
            if (sentence[i]==' ' and temp!=""){
                container.push_back(temp);
                temp="";
            }
            else if (sentence[i]!=' '){
                temp+=sentence[i];
            }
        }
        for (int i=0;i<container.size();i++){
            if (container[i].size()>=searchWord.size()){
                if (tryit(container[i],searchWord)) return i+1;
            }
        }
        return -1;
    }

    bool tryit(string a,string b){
        for (int i=0;i<b.size();i++){
            if (a[i]!=b[i]) return false;
        }
        return true;
    }
};
