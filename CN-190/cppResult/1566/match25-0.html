class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int ans = -1;
        int i = 0;
        stringstream ss(sentence);
        string word;
        while (ss >> word) {
            if (word.size() >= searchWord.size()) {
                bool flag = true;
                for (int j = 0; j < searchWord.size(); j++) {
                    if (word[j] != searchWord[j]) {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    ans = i;
                    break;
                }
            }
            i++;
        }
        return ans == -1 ? ans : ans + 1;
    }
};
