class Solution {
public:

    int isPrefixOfWord(string se, string sw) {
        stringstream ss(se);
        
        int idx = 0;
        while (ss >> se) {
            idx++;
            if (se.size() >= sw.size() && memcmp(se.c_str(), sw.c_str(), sw.size()) == 0) return idx;
            
        }
        return -1;
        
        
    }
};
