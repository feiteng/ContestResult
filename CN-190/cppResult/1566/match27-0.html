class Solution {
public:
    bool isIn(string orig, string searchWord) {
        if (orig.length() < searchWord.length()) {
            return false;
        }    
        for (int i = 0; i < searchWord.length(); i++) {
            if (orig[i] != searchWord[i]) {
                return false;
            }
        }
        return true;
    }
    
    int isPrefixOfWord(string sentence, string searchWord) {
        int n = sentence.length();
        int cur = 0;
        int ret = 1;
        for (int i = 0; i <= n; i++) {
            if (i == n || sentence[i] == ' ') {
                if (isIn(sentence.substr(cur, i - cur), searchWord)) {
                    return ret;
                }
                cur = i + 1;
                ret++;
            }
        }
        return -1;
    }
};
