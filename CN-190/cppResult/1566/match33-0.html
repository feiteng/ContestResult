class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int pos = 0;
        while(1){
            int newpos = sentence.find(searchWord,pos);
            if(newpos==string::npos){
                return -1;
            }else if(newpos==0 || sentence[newpos-1]==' '){
                int numspace = 1;
                for(int i = newpos; i>=0;i--){
                    if(sentence[i]==' '){
                        numspace++;
                    }
                }
                return numspace;
            }
            pos = newpos+1;
        }
    }
};
