class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int cnt=0;
        for(int i=0;i<sentence.size();i++){
            if(i==0||sentence[i-1]==' '){
                cnt++;
                if(sentence.substr(i,searchWord.size())==searchWord){
                    return cnt;
                }
            }
        }
        return -1;
    }
};
