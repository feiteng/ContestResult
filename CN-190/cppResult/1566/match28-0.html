class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        int z=0;
        vector<string> v;
        for(int i=0;i<sentence.size();i++)
        {
            if(sentence[i]==' ')
            {
                v.push_back(sentence.substr(z,i-z));
                z=i+1;
            }
        }
         v.push_back(sentence.substr(z,sentence.size()-z));
        int i=0;
        for(;i<v.size();i++)
        {
            if(v[i].size()<searchWord.size())
                    continue;
            if(v[i].substr(0,searchWord.size())==searchWord)
                break;
        }
        return i==v.size()?-1:i+1;
    }
};
