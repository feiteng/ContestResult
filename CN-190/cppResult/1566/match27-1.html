class Solution {
public:
    bool CheckPrefix(string sentence, string word) {
        if(word.size() > sentence.size()) {
            return false;
        }
        
        for(int i = 0; i < word.size(); i++) {
            if(sentence[i] != word[i]) {
                return false;
            }
        }
        
        return true;
    }
    
    int isPrefixOfWord(string sentence, string searchWord) {
        vector<string> vecs;
        int start = 0;
        for(int i = 0; i < sentence.size(); i++) {
            if(sentence[i] == ' ') {
                vecs.push_back(sentence.substr(start, i - start));
                start = i + 1;
            }
        }
        vecs.push_back(sentence.substr(start, sentence.size() - start));
        
        
        for(int i = 0; i < vecs.size(); i++) {
            cout<<vecs[i]<<" ";
            if(CheckPrefix(vecs[i], searchWord)) {
                return i + 1;
            }
        }
        
        return -1;
    }
};
