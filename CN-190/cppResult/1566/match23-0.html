class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream ss(sentence);
        string s;
        int ans=1;
        int n=searchWord.size();
        while(ss>>s){
            if(s.size()>=n){
                int i=0;
                for(;i<n;++i){
                    if(searchWord[i]!=s[i]){
                        break;
                    }
                }
                if(i==n) return ans;
            }
            ++ans;
        }
        return -1;
    }
};
