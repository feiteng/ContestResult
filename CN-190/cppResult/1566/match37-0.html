class Solution {
public:
    int isPrefixOfWord(string s, string w) {
        int id = 1;
        int n =s.size();
        int l = w.size();
        for (int i=0; i<n; ++i){
            int start = i;
            while(i<n && s[i]!=' ') i++;
            if (s.substr(start, l) == w) return id;
            id ++;
        }
        return -1;

    }
};
