class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        vector<string> words;
        int n = sentence.length();
        int len = 0;
        for(int i = 0; i < n; ++i) {
            if(sentence[i] != ' ') {
                len++;
            }
            else {
                words.push_back(sentence.substr(i - len, len));
                len = 0;
            }
        }
        words.push_back(sentence.substr(n - len, len));
        for(int i = 0; i < words.size(); ++i) {
            if(words[i].length() < searchWord.length()) {
                continue;
            }
            else {
                if(words[i].substr(0, searchWord.length()) == searchWord) {
                    return i + 1;
                }
            }
        }

        return -1;
    }
};
