/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    private int pseudo = 0;
    
    private boolean checkCnts(int[] cnts) {
        int cntOdd = 0;
        for (int cnt: cnts) {
            if (cnt % 2 == 1) {
                ++cntOdd;
            }
        }
        return cntOdd <= 1;
    }
    
    private void dfs(TreeNode root, int[] cnts) {
        ++cnts[root.val-1];
        if (root.left == null && root.right == null) {
            if (checkCnts(cnts)) {
                ++pseudo;
            }
        }
        if (root.right != null) {
            dfs(root.right, cnts);
        }
        if (root.left != null) {
            dfs(root.left, cnts);
        }
        --cnts[root.val-1];
    }
    
    public int pseudoPalindromicPaths (TreeNode root) {
        pseudo = 0;
        int[] cnts = new int[9];
        dfs(root, cnts);
        return pseudo;
    }
}
