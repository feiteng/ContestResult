/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int ans = 0;
    
    
public int pseudoPalindromicPaths (TreeNode root) {
        if(root==null){
            return 0;
        }
        int[] map = new int[10];
        find(root,map);
        return ans;
    }
    
    public void find(
TreeNode root,int[] map){
        map[root.val]++;
        // System.out.println("val="+root.val+" map["+root.val+"]"+map[root.val]);
        
        if(root.left==null && root.right==null){
            // 叶子节点
            int inValid = 0;
            for(int i=1;i<=9;i++){
                if(map[i]%2!=0){
                    inValid++;
                
}
            }
            if(inValid<=1){
                ans++;
            }
        }else{
            if(root.left!=null){
                find(root.left,map);
            }
            if(root.right!=null){
                find(root.right,map);
            }
        }
        
        map[root.val]--;
    }
}
