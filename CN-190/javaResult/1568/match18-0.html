class Solution {
    private int[] cnts;
    private int ans;

    public int pseudoPalindromicPaths(TreeNode root) {
        cnts = new int[10];
        ans = 0;
        dfs(root);
        return ans;
    }

    private void dfs(TreeNode node) {
        cnts[node.val]++;
        if (node.left == null && node.right == null) {
            if (isValid()) {
                ans++;
            }
        } else {
            if (node.left != null) {
                dfs(node.left);
            }
            if (node.right != null) {
                dfs(node.right);
            }
        }
        cnts[node.val]--;
    }

    private boolean isValid() {
        int oddCnt = 0;
        for (int i = 1; i <= 9; i++) {
            if (cnts[i] % 2 == 1) {
                oddCnt++;
            }
        }
        return oddCnt <= 1;
    }
}
