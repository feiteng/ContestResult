class Solution {
        int rs = 0;
        void dfs(TreeNode treeNode, int map[]) {
            map[treeNode.val] ++;
            if(treeNode.left != null) {
                dfs(treeNode.left, map);
            }
            if(treeNode.right != null) {
                dfs(treeNode.right, map);
            }
            if(treeNode.left == null && treeNode.right == null) {
                boolean flag = false;
                boolean valid = true;
                for(int i=1; i<=9; i++) {
                    if(map[i]%2!=0) {
                        if(flag == false) {
                            flag = true;
                        } else {
                            valid = false;
                            break;
                        }
                    }
                }
                if(valid) {
                    rs ++;
                }
            }
            map[treeNode.val] --;
        }
        public int pseudoPalindromicPaths (TreeNode root) {
            int map[] = new int[10];
            dfs(root, map);
            return rs;
        }
    }
