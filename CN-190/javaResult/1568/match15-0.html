/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] path = new int[10];
        int[] count = new int[1];
        pseudoPalindromicPaths(root, path, count);
        return count[0];
    }
    void pseudoPalindromicPaths(TreeNode root, int[] path, int[] count) {
        path[root.val]++;
        // 到了叶子节点了
        if (root.left == null && root.right == null) {
            int odd = 0;
            for (int i = 1; i <= 9; ++i) {
                if ((path[i] & 1) == 1) {
                    odd++;
                    if (odd > 1) {
                        break;
                    }
                }
            }
            if (odd <= 1) {
                count[0]++;
            }
        }
        else {
            if (root.left != null) {
                pseudoPalindromicPaths(root.left, path, count);
            }
            if (root.right != null) {
                pseudoPalindromicPaths(root.right, path, count);
            }
        }
        path[root.val]--;
    }
}
