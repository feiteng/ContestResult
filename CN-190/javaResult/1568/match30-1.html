/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int[] count = new int[10];
    
int res = 0;
    
    public int pseudoPalindromicPaths (TreeNode root) {
        helper(root);
        return res;
    }
    
    private void helper(TreeNode root) {
        count[root.val] += 1;
        if (root.left == null && root.right == null) {
            if (isPalin(count))
                res++;
            count[root.val] += 1;
            return;
        
}
        
        if (root.left != null)
            helper(root.left);
        if (root.right != null)
            helper(root.right);
        
        count[root.val] -= 1;
    }
    
    private boolean isPalin(int[] count) {
        //for (int i = 0; i < 10; i++)
        //    System.out.print(count[i]);
        //System.out.println();
        boolean flag = false;
        for (int i = 0; i < 10; i++) {
            if (count[i] % 2 == 0) continue;
            if (flag)
                return false;
            flag = true;
        }
        return true;
    }
}
