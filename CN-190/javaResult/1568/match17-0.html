/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
  int[] map;
    int ans;
    //当它满足：路径经过的所有节点值的排列中，存在一个回文序列。
    public int pseudoPalindromicPaths (TreeNode root) {
        //root = [2,3,1,3,1,null,1]
        //输出：2
        map = new int[10];
        ans = 0; 
        dfs(root);
        return ans;
    }

    private void dfs(TreeNode root) {
        map[root.val]++;
        if (root.left == null && root.right == null) {
            ans = verify(map) ? ans+1 : ans;
        }
        if (root.left != null) {
            dfs(root.left);
        }
        if (root.right != null) {
            dfs(root.right);
        }
        map[root.val]--;
    }

    private boolean verify(int[] map) {
        int k = 0;
        for (int i = 1; i < map.length; i++) {
            if (map[i] % 2 != 0) {
                k++;
            }
        }
        return k <= 1;
    }
}
