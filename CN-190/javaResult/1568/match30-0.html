/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count;
    public int pseudoPalindromicPaths (TreeNode root) {
        dfs(root, 0);
        return count;
    }
    private void dfs(TreeNode root, int flag) {
        int mask = 1 << root.val;
        flag ^= mask; // root.val位: flag取异或
        if (root.left == null && root.right == null && (flag == 0 || (flag & (flag - 1)) == 0)) // flag只有一位1
            count++;
        if (root.left != null)
            dfs(root.left, flag);
        if (root.right != null)
            dfs(root.right, flag);
    }
}
