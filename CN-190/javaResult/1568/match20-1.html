/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int ans = 0;
    
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root == null) {
            return 0;
        }
        
        
int[] cnt = new int[10];
        dfs(root, cnt);
        return ans;
    }
    
    public void dfs(TreeNode root, int[] cnt) {
        
if (root.left == null && root.right == null) {
            cnt[root.val]++;
            if (check(cnt)) {
                ans++; 
            }
            cnt[root.val]--;
            return;
        }
        
        cnt[root.val]++;
        if (root.left != null) {
            dfs(root.left, cnt);
        }
        if (root.right != null) {
            dfs(root.right, cnt);
        }
        cnt[root.val]--;
    }
    
    public boolean check(int[] cnt) {
        int odd = 0;
        for (int c : cnt) {
            if ((c & 1) == 1) {
                odd++;
            }
            if (odd > 1) {
                return false;
            }
        }
        return true;
    }
}
