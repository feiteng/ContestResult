/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int result;

    public int pseudoPalindromicPaths (TreeNode root) {
        Deque<TreeNode> queue = new ArrayDeque<>();
        if (root == null) {
            return 0;
        }
        queue.add(root);
        
result = 0;
        int[] r = new int[10];
        r[root.val] ++;
        search(root, r);
        return result;
    }

    private void search(TreeNode root, int[] nums) {
        if (root.left == null && root.right == null) {
            int single = 0;
            System.out.println(Arrays.toString(nums));
            
for (int n: nums) {
                if (n % 2 == 1) {
                    single ++;
                }
            }
            if (single <= 1) {
                result ++;
            }
        
}
        
        if (root.left != null) {
            nums[root.left.val] ++;
            search(root.left, nums);
            nums[root.left.val] --;
        }
        if (root.right != null) {
            nums[root.right.val] ++;
            search(root.right, nums);
            nums[root.right.val] --;
        }
    }
}
