/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    private int ret;
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root.left == null && root.right == null) {
            return 1;
        }
        ret = 0;
        
dfs(root, "");
        return ret;
    }

    private void dfs(TreeNode root, String path) {
        path += root.val;
        if (root.left == null && root.right == null) {
            if (isPalindromeSeq(path)) 
{
                ret += 1;
            }
            return;
        }
        if (root.left != null) {
            dfs(root.left, path);
        }
        if (root.right != null) {
            dfs(root.right, path);
        }
    }

    private boolean isPalindromeSeq(String xs) {
        if (xs.length() == 1) {
            return false;
        }
        Set<Character> remains = new HashSet<>();
        for (int i = 0; i < xs.length(); i++) { 
            char x = xs.charAt(i);
            if (!remains.contains(x)) {
                remains.add(x);
            } else {
                remains.remove(x);
            }
        }
        return remains.size() <= 1;
    }
}
