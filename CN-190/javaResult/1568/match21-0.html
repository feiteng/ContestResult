/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int ans=0;
    public int pseudoPalindromicPaths (TreeNode root) {
        dfs(root,1<<root.val-1);
        return ans;
    }
    public void dfs(TreeNode root,int s){
        System.out.println(root.val+" "+s);
        if(root.left==null && root.right==null){
            int res=0;
            while(s!= 0) 
            {
                res += s & 1;
                s >>= 1;
            }
            System.out.println("res"+res);
            if(res<2)ans++;
            return;
        }
        if(root.left!=null)dfs(root.left,s^1<<root.left.val-1);
        if(root.right!=null)dfs(root.right,s^1<<root.right.val-1); 
    }
}
