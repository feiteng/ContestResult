/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int res = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] t = new int[10];
        dfs(root,t);
        return res;
    }public void dfs(TreeNode root,int[] t){
        if(root == null) return;
        if(root.left == null && root.right == null){
            
int count = 0;
            t[root.val]++;
            for(int i = 1; i< 10; i++){
                if(t[i] %2 == 1){
                    count++;
                }
            }
            t[root.val]--;
            if(count < 2){
                res++;
            }return ;
        }
        t[root.val]++;
        dfs(root.left,t);
        dfs(root.right,t);
        t[root.val]--;
    }
}
