class Solution {
    private int[] count;
    private int res;

    public int pseudoPalindromicPaths (TreeNode root) {
        count = new int[10];
        res = 0;
        DFS(root);
        return res;
    }

    private void DFS(TreeNode root) {
        if (root == null) {
            return;
        }
        count[root.val] += 1;
        if (root.left == null && root.right == null) {
            res += isPalin() ? 1 : 0;
        } else {
            DFS(root.left);
            DFS(root.right);
        
}
        count[root.val] -= 1;
    }

    private boolean isPalin() {
        int odd = 0;
        for (int i : count) {
            if (i % 2 == 1) {
                ++odd;
            }
        }
        return odd <= 1;
    }
}
