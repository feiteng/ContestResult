/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] map = new int[10];
        
int[] count = new int[1];
        dfs(root,map,count);
        return count[0];
    }

    private void dfs(TreeNode root,int[] map,int[] count){
        if(root == null){
            return;
        }
        
        map[root.val]++;
        
        if(root.left == null && root.right == null){
            if(isP(map))
{
                count[0]++;
            }
            map[root.val]--;
            return;
        }
        
        if(root.left != null){
            dfs(root.left,map,count);
        }
        if(root.right != null){
            dfs(root.right,map,count);
        }
        map[root.val]--;
    }
    
    private boolean isP(int[] map)
{
        int jiCount = 0;
        for (int i = 0;i<map.length;i++){
            if((map[i] & 1) == 1){
                jiCount++;
            }
        }
        return jiCount <= 1;
    }
}
