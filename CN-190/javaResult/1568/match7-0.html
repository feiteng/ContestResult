/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count;

    public int pseudoPalindromicPaths(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 1;
        }
        count = 0;
        int[] freq = new int[10];
        F(root, freq);
        return count;
    }

    private void F(TreeNode root, int[] freq) {
        freq[root.val]++;
        if (root.left == null && root.right == null) {
            int odd = 0;
            for (int i = 1; i <= 9; i++) {
                if (freq[i] % 2 != 0) {
                    odd++;
                }
            }
            if (odd == 0 || odd == 1) {
                count++;
            }
            freq[root.val]--;
            return;
        }
        if (root.left != null) {
            F(root.left, freq);
        }
        if (root.right != null) {
            F(root.right, freq);
        }
        freq[root.val]--;
    }

}
