/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int totalNumber = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] nums = new int[10];
        backTrack(root, nums);
        return totalNumber;
    }

    public void backTrack(TreeNode node, int[] nums){
        nums[node.val]++;
        if(node.left != null){
            backTrack(node.left, nums);
        }
        if(node.right != null){
            backTrack(node.right, nums);
        }
        if(node.left == null && node.right == null){
            if(isPseudoPalindromic(nums)){
                totalNumber++;
            }
        }
        nums[node.val]--;
    }

    public boolean isPseudoPalindromic(int [] nums){
        int oneNumber = 0;
        for(int i=0; i<nums.length; i++){
            if(nums[i]%2==1){
                oneNumber++;
            }
        }
        return oneNumber<=1;
    }
}
