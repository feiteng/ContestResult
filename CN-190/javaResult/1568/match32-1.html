class TreeNode {
      int val;
      TreeNode left;
      TreeNode right;
    TreeNode() {}
     TreeNode(int val) { this.val = val; }
      TreeNode(int val, TreeNode left, TreeNode right) {
          this.val = val;
          this.left = left;
          this.right = right;
      }
}

class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] arr = new int[10];
        return dfs(root, arr);
    }

    public int dfs(TreeNode node, int[] arr) {
        if (node == null) return 0;
        arr[node.val]++;
        if (node.left == null && node.right == null) {
            int ans = check(arr) ? 1 : 0;
            arr[node.val]--;
            return ans;
        }
        int l = dfs(node.left, arr);
        int r = dfs(node.right, arr);
        arr[node.val]--;
        
return l + r;
    }

    public boolean check(int[] arr) {
        int so = 0;
        for (int i : arr) {
            if ((i & 0x1) == 1) so++;
        }
        return so < 2;
    }
}
