/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int r =0;
    
void go(TreeNode rt, int d[]){
        if(rt==null) return;
        d[rt.val]++;
        if(rt.left==null&&rt.right==null){
            int ct = 0;
            for(int i=1;i<=9;++i){
                if(d[i]%2==1){
                    ct++;
                }
            }
            if(ct==1||ct==0){
                r++;
            }
            d[rt.val]--;
            return;
        }
        go(rt.left,d);
        go(rt.right,d);
        d[rt.val]--;
        
    }
    
    
    public int pseudoPalindromicPaths (TreeNode root) {
        go(root,new int[10]);
        return r;
    }
}
