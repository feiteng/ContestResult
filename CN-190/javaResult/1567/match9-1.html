class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> ss = new HashSet<>();
        ss.add('a');
        ss.add('e');
        ss.add('i');
        ss.add('o');
        ss.add('u');
        
int left = 0, right = k-1, nums = 0;
        
        for(int i = left;i <= right;i++) {
            if(ss.contains(s.charAt(i))) nums++;
        }
        int res = nums;
        right++;
        for(;right < s.length();right++) 
{
            if(ss.contains(s.charAt(right))) nums++;
            if(ss.contains(s.charAt(left))) nums--;
            left++;
            res = Math.max(res, nums);
        }
        return res;
    }
}
