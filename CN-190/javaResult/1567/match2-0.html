class Solution {
    public int maxVowels(String s, int k) {
        
        int ans = 0;
        for (int i=0;i<k;i++) {
            if (check(s.charAt(i))) {
                ans++;
            }
        }
        int max = ans;
        for (int i=k;i<s.length();i++) {
            if (check(s.charAt(i))) {
                ans++;
            }
            if (check(s.charAt(i-k))) {
                ans--;
            }
            max = Math.max(max,ans);
        }
        return max;

    }
    boolean check(char ch) {
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
            return true;
        }
        return false;
    }
}
