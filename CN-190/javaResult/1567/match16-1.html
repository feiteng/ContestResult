class Solution {
    public int maxVowels(String s, int k) {
        int start = 0;
        int end = 0;
        
int cnt = 0;
        int res = 0;
        for (int i = 0; i < k; i++) {
            if (isYuanyin(s.charAt(i))) {
                cnt++;
            }
        }
        res = Math.max(res, cnt);
        
for (int i = 1; i + k - 1 < s.length(); i++) {
            if (isYuanyin(s.charAt(i - 1))) {
                cnt--;
            }
            if (isYuanyin(s.charAt(i + k - 1))) {
                cnt++;
            }
            res = Math.max(res, cnt);
        
}
        return res;
    }
    public boolean isYuanyin(char a) {
        return a == 'a' 
               || a == 'e' 
               || a == 'i'
               || a == 'o'
               || a == 'u';
    }
}
