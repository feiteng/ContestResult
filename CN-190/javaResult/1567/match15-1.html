class Solution {
    public int maxVowels(String s, int k) {
        int n = s.length();
        int[] dp = new int[n];
        
for(int i = 0; i < k; i++) {
            if(isValid(s.charAt(i))) {
                dp[k-1]++;
            }
        }
        int ans = dp[k-1];
        for(int i = k; i < n; i++) {
            int temp = dp[i-1];
            
if(isValid(s.charAt(i-k))) {
                temp--;
            }
            if(isValid(s.charAt(i))) {
                temp++;
            }
            dp[i] = temp;
            
ans = Math.max(ans, dp[i]);
        }
        return ans;
    }
    public boolean isValid(char c) {
        if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            return true;
        } else {
            return false;
        }
    }
}
