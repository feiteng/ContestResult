class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');set.add('i');
        set.add('o');set.add('u');
        int left = 0;
        int max = 0;
        int sum = 0;
        for (int right = 0; right < s.length(); right++) {
            char l = s.charAt(left);
            char r = s.charAt(right);
            if (right - left >= k) {
                if (set.contains(l)) {
                    sum--;
                }
                left++;
            }
            if (set.contains(r)) {
                sum++;
            }
            max = Math.max(max, sum);
        }
        return max;
    }    
}
