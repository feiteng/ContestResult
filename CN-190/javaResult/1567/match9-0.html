class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> set = new HashSet<>();
        set.add('a');set.add('e');set.add('i');set.add('o');set.add('u');
        int res = 0;
        int len = s.length();
        int temp = 0;
        for(int i = 0; i < k; i++){
            if(set.contains(s.charAt(i))) temp++;
        }
        res = Math.max(res, temp);
        for(int i = k; i < len; i++){
            if(set.contains(s.charAt(i))) temp++;
            if(set.contains(s.charAt(i - k))) temp--;
            res = Math.max(res, temp);
        } 
        return res;
    }
}
