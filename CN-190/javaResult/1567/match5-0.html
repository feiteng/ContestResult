class Solution {
    public int maxVowels(String s, int k) {
        int max = 0;
        for (int i = 0; i < k; i++) {
            char c = s.charAt(i);
            if (vowel(c)) {
                max++;
            }
        }
        int num = max;
        for (int i = k; i < s.length(); i++) {
            if (vowel(s.charAt(i))) {
                num++; 
            }
            if (vowel(s.charAt(i - k))) {
                num--;
            }
            max = Math.max(max,num);
        }
        return max;
    }
    private boolean vowel(char c) {
        switch (c) { 
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u': return true;
        }
        return false;
    }
}
