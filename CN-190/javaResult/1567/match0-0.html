class Solution {
    public int maxVowels(String s, int k) {
        int max = 0;
        int count = 0;
        // initialize
        for (int i = 0; i < k; i++) {
            if (isVowel(s.charAt(i)))
                count++;
        }
        max = count;
        for (int i = k; i < s.length(); i++) {
            if (isVowel(s.charAt(i-k)))
                count--;
            if (isVowel(s.charAt(i)))
                count++;
            max = Math.max(max, count);
        }
        return max;
    }
    private boolean isVowel(char c) {
        if ('a' == c || 'e' == c || 'i' == c || 'o' == c || 'u' == c)
            return true;
        return false;
    }
}
