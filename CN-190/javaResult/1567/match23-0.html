class Solution {
    public int maxVowels(String s, int k) {
        char[] ch = s.toCharArray();
        int ans = 0;
        int p = 0;
        for (int i = 0; i < ch.length; i++) {
            if (ch[i] == 'a' || ch[i] == 'e' || ch[i] == 'i' || ch[i] == 'o' || ch[i] == 'u') {
                p++;
            }
            
            if(i>=k && (ch[i-k] == 'a' || ch[i-k] == 'e' || ch[i-k] == 'i' || ch[i-k] == 'o' || ch[i-k] == 'u')){
                p--;
            }
            ans = Math.max(ans,p);
        }
        return ans;
    }
}
