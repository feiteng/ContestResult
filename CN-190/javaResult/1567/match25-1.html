class Solution {
    public int maxVowels(String s, int k) {
        //    init
        char[] c = s.toCharArray();
        int cnt=0;
        
for (int i = 0; i < k; i++) {
            if(is(c[i])){
                cnt++;
            }
        }
        int max=cnt;
        for (int i = k; i < c.length; i++) 
{
            if(is(c[i])){
                cnt++;
            }
            if(is(c[i-k])){
                cnt--;
            }
            max=Math.max(max,cnt);
        
}

        return max;
    }

    private boolean is(char c) {
        return c=='a'|| c=='e' ||c=='i'|| c=='o'||c=='u';
                
    }
}
