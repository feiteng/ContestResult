class Solution {
    public int maxVowels(String s, int k) {
        int start = 0;
        int count = 0;
        for (int i = 0; i < k; i++) {
            if (isVowel(s.charAt(i))) {
                count++;
            }
        }
        int result = count;
        while (start + k < 
s.length()) {
            if (isVowel(s.charAt(start))) {
                count--;
            }
            if (isVowel(s.charAt(start + k))) {
                count++;
                result = Math.max(result, count);
            }
            start++;
        }
        
return result;
    }

    private boolean isVowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
}
