class Solution {
    public boolean isvalid(char c)
    {
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c== 'u'){
            return true;
        }else{
            return false;
        }
    }
    
public int maxVowels(String s, int k) {
        int len = s.length();
        int res = 0;
        int num = 0;
        for (int i = 0 ; i < k; i++)
        {
            if (isvalid(s.charAt(i))){
                num++;
            }
        }
        res = num;
        for (int i = k; i < len ; i++)
        {
            if (isvalid(s.charAt(i))){
                num++;
            } 
            if (isvalid(s.charAt(i - k)))
{
                num--;
            }
            res = Math.max(res,num);
        }
        return res;
    }
}
