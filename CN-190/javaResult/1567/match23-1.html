class Solution {
    
    public int maxVowels(String s, int k) {
        char[] chs = s.toCharArray();
        int max = 0, count = 0;
        for (int i = 0; i < chs.length; i++) {
            if (i < k) {
                if (chs[i] == 'a' || chs[i] == 'e' || chs[i] == 'i' || chs[i] == 'o' || chs[i] == 'u') count++;
            } else {
                if (chs[i] == 'a' || chs[i] == 'e' || chs[i] == 'i' || chs[i] == 'o' || chs[i] == 'u') count++;
                if (chs[i - k] == 'a' || chs[i - k] == 'e' || chs[i - k] == 'i' || chs[i - k] == 'o' || chs[i - k] == 'u')
                    count--;
            
}
            max = Math.max(max, count);
        }
        return max;
    }
}
