class Solution {
    public int maxVowels(String s, int k) {
        int max = 0;
        
int tmpMax = 0;
        for(int i=0; i<k; i++){
            if(isVowels(s.charAt(i))){
                tmpMax++;
            }
        }
        max = tmpMax;
        for(int i=k; i<s.length(); i++){
            
if(isVowels(s.charAt(i))){
                tmpMax++;
            }
            if(isVowels(s.charAt(i-k))){
                tmpMax--;
            }
            if(tmpMax > max){
                max = tmpMax;
            }
        
}
        return max;
    }

    public boolean isVowels(char tmp){
        return tmp == 'a'||
        tmp == 'e' ||
        tmp == 'i' ||
        tmp == 'o' ||
        tmp == 'u';
    }
}
