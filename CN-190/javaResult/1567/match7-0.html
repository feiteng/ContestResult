class Solution {
    public int maxVowels(String s, int k) {
        int res = 0;
        for (int i = 0; i < k; i++) {
            if (isV(s.charAt(i))){
                res++;
            }
        }
        int max = res;
        for (int i = 0;  (i+k)<s.length(); i++) {
            int add = isV(s.charAt(i+k))?1:0;
            int des = isV(s.charAt(i))?1:0;
            res += (add-des);
            if (max < res)max = res;
        }
        return max;
    }

    private boolean isV(char charAt) {
        return charAt=='a'||charAt=='e'||charAt=='i'||charAt=='o'||charAt=='u';
    }
}
