class Solution {
    public int maxVowels(String s, int k) {

        int curcount = 0;
        for (int i = 0;i<k;i++){
            if(isYuan(s.charAt(i))){
                curcount++;
            }
        }

        int max = curcount;
        for (int j = k;j<s.length();j++){
            int needRemoveIndex = j - k;
            if(isYuan(s.charAt(needRemoveIndex))){
                curcount--;
            }
            if(isYuan(s.charAt(j))){
                curcount++;
            }
            max = Math.max(max,curcount);
        }
        return max;

    }

    private boolean isYuan(char c){
        return c == 'a'|| c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
}
