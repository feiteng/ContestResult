class Solution {
    private boolean is(char x)
    {
        boolean flag = false;
        switch (x)
        {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                flag = true;
                break;
        }

        return flag;
    }


    public int maxVowels(String s, int k)
    {
        if(s == null) return 0;

        int ans = 0;
        int tmp = 0;

        int left = 0;
        for (int i = 0; i < k; i++)
        {
            if(is(s.charAt(i)))
            {
                tmp ++;
            }
        }
        ans = tmp;

        for (int i = k; i < s.length(); i++)
        {
            if(is(s.charAt(i)))
            {
                tmp++;
            }

            if(is(s.charAt(left++)))
            {
                tmp--;
            }
            ans = Math.max(ans,tmp);
            if(ans == k) break;
        }


        return ans;
    }
}
