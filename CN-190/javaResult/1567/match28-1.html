class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> vols=new HashSet<>();
        vols.add('a');vols.add('e');vols.add('i');vols.add('o');vols.add('u');
        
        int max=0;
        int cur=0;
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            if(vols.contains(ch)){
                cur++;
            
}
            if(i>=k&&vols.contains(s.charAt(i-k))){
                cur--;
            }
            max=Math.max(max,cur);
        }
     return max;   
    }
}
