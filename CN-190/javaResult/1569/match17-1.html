class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int[][] dp = new int[len1][len2];
        
        int max = Integer.MIN_VALUE;
        for(int i = 0; i < len1; ++i){
            for(int j = 0; j < len2; ++j){
                if(i > 0 && j > 0){
                    dp[i][j] = Math.max(Math.max(dp[i-1][j], dp[i][j-1]), Math.max(nums1[i] * nums2[j], dp[i-1][j-1] + nums1[i] * nums2[j]));
                
}else if(i > 0){
                    dp[i][j] = Math.max(dp[i-1][j], nums1[i] * nums2[j]);
                }else if(j > 0){
                    dp[i][j] = Math.max(dp[i][j-1], nums1[i] * nums2[j]);
                }else{
                    dp[i][j] = nums1[i] * nums2[j];
                
}
                max = Math.max(max, dp[i][j]);
                //System.out.printf("%d %d %d\n", i, j, dp[i][j]);
            }
        }
        
        return max;
    }
}
