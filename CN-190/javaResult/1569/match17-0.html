class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int[][] dp = new int[m][n];
        int res = Integer.MIN_VALUE;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                //include nums1[]i * nums2[j]
                //not include
                if (i == 0 && j == 0) {
                    dp[i][j] = nums1[i] * nums2[j];
                }
                else if (i == 0) {
                    dp[i][j] = Math.max(nums1[i] * nums2[j], dp[i][j-1]);
                }
                else if (j == 0) {
                    dp[i][j] = Math.max(nums1[i] * nums2[j], dp[i-1][j]);
                }
                else {
                    dp[i][j] = Math.max(Math.max(nums1[i] * nums2[j], dp[i-1][j-1] + nums1[i] * nums2[j]), Math.max(dp[i][j-1], dp[i-1][j]));  
                }    
                res = Math.max(res, dp[i][j]);
            }
        }
        return res;
    }
}
