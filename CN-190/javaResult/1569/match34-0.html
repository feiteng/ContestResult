class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        int[][] dp = new int[m][n];
        int max = Integer.MIN_VALUE;
        int ans = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            max = Math.max(max, nums1[0] * nums2[i]);
            dp[0][i] = max;
            ans = Math.max(dp[0][i], ans);
            // System.out.print(dp[0][i]+ ", ");
        }
        // System.out.println();
        max = Integer.MIN_VALUE;
        for (int i = 0; i < m; i++) {
            max = Math.max(max, nums1[i] * nums2[0]);
            dp[i][0] = max;
            ans = Math.max(dp[i][0], ans);
            // System.out.println(dp[i][0] + ",");
        }
        // System.out.println();
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = Math.max(dp[i - 1][j - 1] + nums1[i] * nums2[j], Math.max(dp[i - 1][j - 1], dp[i][j - 1]));
                dp[i][j] = Math.max(dp[i][j], Math.max(nums1[i] * nums2[j], dp[i - 1][j]));
                // System.out.print(dp[i][j] + ",");
                ans = Math.max(ans, dp[i][j]);
            }
            // System.out.println();
        }
        return ans;
    }
}
