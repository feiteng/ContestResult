class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int n = nums1.length, m = nums2.length;
        int[][] dp = new int[n][m];
        dp[n - 1][m - 1] = nums1[n - 1] * nums2[m - 1];
        for (int i = n - 2; i >= 0; i--) {
            dp[i][m - 1] = Math.max(dp[i + 1][m - 1], nums1[i] * nums2[m - 1]);
        }
        for (int j = m - 2; j >= 0; j--) {
            dp[n - 1][j] = Math.max(dp[n - 1][j + 1], nums1[n - 1] * nums2[j]);
        }
        for (int i = n - 2; i >= 0; i--) {
            for (int j = m - 2; j >= 0; j--) {
                dp[i][j] = Math.max(Math.max(dp[i + 1][j + 1], dp[i][j + 1]), Math.max(dp[i + 1][j + 1] + nums1[i] * nums2[j], dp[i + 1][j]));
                dp[i][j] = Math.max(dp[i][j], nums1[i] * nums2[j]);
            }
        }
        
        return dp[0][0];
    }
}
