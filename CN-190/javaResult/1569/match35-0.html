class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        int[][] dp = new int[m + 1][n + 1];
        int[] minandmax1 = findminandmax(nums1);
        int[] minandmax2 = findminandmax(nums2);
        if (minandmax1[0] < 0 && minandmax1[1] < 0) {
            if (minandmax2[0] > 0 && minandmax2[1] > 0) {
                return minandmax1[1] * minandmax2[0];
            }
        } else if (minandmax1[0] > 0 && minandmax1[1] > 0) {
            if (minandmax2[0] < 0 && minandmax2[1] < 0) {
                return minandmax1[0] * minandmax2[1];
            }
        }
        for (int i = 1; i < m + 1; i++) {
            for (int j = 1; j < n + 1; j++) {
                dp[i][j] = Math.max(Math.max(dp[i - 1][j - 1], dp[i - 1][j - 1] + nums1[i - 1] * nums2[j - 1]), Math.max(dp[i - 1][j], dp[i][j - 1]));
            }
        }
        return dp[m][n];
    }
    
    private int[] findminandmax(int[] a) {
        int[] minandmax = new int[2];
        minandmax[0] = Integer.MAX_VALUE;
        minandmax[1] = Integer.MIN_VALUE;
        for (int i = 0; i < a.length; i++) {
            minandmax[0] = Math.min(minandmax[0], a[i]);
            minandmax[1] = Math.max(minandmax[1], a[i]);
        }
        return minandmax;
    }
    
}
