class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;

        int[][] dp = new int[len1][len2];

        dp[len1 - 1][len2 - 1] = nums1[len1 - 1] * nums2[len2 - 1];
        for (int i = len1 - 2; i >= 0; i--) {
            dp[i][len2 - 1] = Math.max(dp[i + 1][len2 - 1], nums1[i] * nums2[len2 - 1]);
        }
        for (int i = len2 - 2; i >= 0; i--) {
            dp[len1 - 1][i] = Math.max(dp[len1 - 1][i + 1], nums1[len1 - 1] * nums2[i]);
        }

        for (int i = len1 - 2; i >= 0; i--) {
            for (int j = len2 - 2; j >= 0; j--) {
                int dpij = Math.max(0, dp[i + 1][j + 1]) + nums1[i] * nums2[j];
                dpij = Math.max(dpij, dp[i + 1][j]);
                dpij = Math.max(dpij, dp[i][j + 1]);
                dp[i][j] = dpij;
            }
        }

        return dp[0][0];
    }
}
