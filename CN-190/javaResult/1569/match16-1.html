class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int[][] t = new int[m][n];
        t[0][0] = nums1[0] * nums2[0];
        for(int i=1;i<m;i++){
            t[i][0] = Math.max(nums1[i] * nums2[0], t[i-1][0]);
        }
        for(int j=1;j<n;j++){
            t[0][j] = Math.max(nums1[0] * nums2[j], t[0][j-1]);
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                t[i][j] = Math.max(Math.max(t[i-1][j], t[i][j-1]), (t[i-1][j-1] > 0 ? t[i-1][j-1]: 0) +nums1[i]*nums2[j]);
            }
        }
        return t[m-1][n-1];
    }
}
