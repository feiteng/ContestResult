class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        int[][] dp = new int[m + 1][n + 1];
        // for (int i = 1; i <= m; i++) {
        //     dp[i][0] = nums1[i - 1] * nums2[0];
        // }
        // for (int j = 1; j <= n; j++) {
        //     dp[0][j] = nums1[0] * nums2[j - 1];
        // }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = Math.max(dp[i - 1][j - 1] + nums1[i - 1] * nums2[j - 1], Math.max(dp[i - 1][j], dp[i][j - 1]));
            }
        }
        if (dp[m][n] == 0) {
            
int ans = Integer.MIN_VALUE;
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    ans = Math.max(ans, nums1[i] * nums2[j]);
                }
            }
            return ans;
        } else {
            return dp[m][n];
        }
    }
}
