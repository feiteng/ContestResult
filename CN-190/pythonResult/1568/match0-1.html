# Definition for a binary tree node.

class TreeNode(object):
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        listpoint=[0 for i in range(9)]
        listpoint[root.val-1]=1
        rootqueue=[root]
        listqueue=[listpoint]
        res=0
        while rootqueue:
            point=rootqueue.pop(0)
            listpoint=listqueue.pop(0)
            if ((not point.left) and (not point.right)):
                if self.checkp(listpoint):
                    res+=1
            else:
                if point.left:
                    listleft=self.addup(listpoint,point.left.val)
                    listqueue.append(listleft)
                    rootqueue.append(point.left)
                if point.right:
                    listright=self.addup(listpoint,point.right.val)
                    listqueue.append(listright)
                    rootqueue.append(point.right)
        return res
                
    def addup(self,l1,val):
        l3=l1[:]
        l3[val-1]+=1
        return l3
    def checkp(self,l):
        num=0
        for i in range(len(l)):
            if l[i]%2==1:
                num+=1
        if num<=1:
            return True
        return False
