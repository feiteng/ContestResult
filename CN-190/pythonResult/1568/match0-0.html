class TreeNode(object):
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


class Solution(object):
    def pseudoPalindromicPaths(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """

        self.res = 0
        self.dfs(root, curr_set=set([root.val]))
        return self.res

    def dfs(self, node, curr_set):
        if node.left is None and node.right is None:
            if len(curr_set) <= 1:
                self.res += 1
                return

        if node.left is not None:
            left_set = set(curr_set)
            if node.left.val in left_set:
                left_set.remove(node.left.val)
            else:
                left_set.add(node.left.val)
            self.dfs(node.left, left_set)
        if node.right is not None:
            right_set = set(curr_set)
            if node.right.val in right_set:
                right_set.remove(node.right.val)
            else:
                right_set.add(node.right.val)
            self.dfs(node.right, right_set)
        return
