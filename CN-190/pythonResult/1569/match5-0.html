class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        import sys
        dp = [[-sys.maxsize] * len(nums2) for _ in range(len(nums1))]

        for i in range(len(nums1)):
            for j in range(len(nums2)):
                if i > 0 and j > 0:
                    dp[i][j] = max(dp[i-1][j-1]+nums1[i]*nums2[j], dp[i-1][j-1], dp[i-1][j],dp[i][j-1])
                elif i > 0:
                    dp[i][j] = max(dp[i-1][0],nums1[i]*nums2[j])
                elif j > 0:
                    dp[i][j] = max(dp[0][j-1],nums1[i] * nums2[j])
                else:
                    dp[i][j] = max(0,nums1[i]*nums2[j])
        if dp[-1][-1] == 0:
            if sum(nums1) > 0:
                return min(nums1) * max(nums2)
            else:
                return max(nums1) * min(nums2)
        return dp[-1][-1]
