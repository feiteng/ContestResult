class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        sta = []
        for x1 in nums1:
            tmp = [x1*x2 for x2 in nums2]
            sta.append(tmp)
        #print (sta)
        for i in range(1, len(sta)):
            sta[i][0] = max(sta[i][0],sta[i-1][0])
        for j in range(1,len(sta[0])):
            for i in range(len(sta)):
                if i == 0:
                    sta[i][j] = max(sta[i][j],sta[i][j-1])
                    continue
                # use
                sta[i][j] = max(sta[i][j], sta[i-1][j-1] + sta[i][j])
                # not use
                sta
[i][j] = max(sta[i][j],sta[i][j-1])
                sta[i][j] = max(sta[i][j], sta[i-1][j])
        #print(sta)
        return sta[len(sta)-1][len(sta[0])-1]
