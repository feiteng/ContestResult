class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        len1 = len(nums1) + 1
        len2 = len(nums2) + 1
        dp = [[[float("-inf")] * 2 for _ in range(len2)]  for _ in range(len1)]
        for i in range(len1):
            dp[i][0][0] = 0
            #dp[i][0][1] = float("-inf")
        for j in range(len2):
            dp[0][j][0] = 0
            #dp[0][j][1] = float("-inf")
        #print dp[1][0][1]
        for i in range(1, len1):
            for j in range(1, len2):
                #dp[i][j] = max(nums1[i-1]*nums2[j-1] + dp[i-1][j-1], dp[i-1][j], dp[i][j-1])
                #print dp[1]
                dp[i][j][1] = max(nums1[i-1]*nums2[j-1] + max(dp[i-1][j-1][0], dp[i-1][j-1][1]), dp[i-1][j][1], dp[i][j-1][1])
                dp[i][j][0] = max(dp[i-1][j][0], dp[i][j-1][0])
                #print i,j,dp[i][j][0],dp[i][j][1]
                #print nums1[i-1]*nums2[j-1] + max(dp[i-1][j-1][0], dp[i-1][j-1][1]), dp[i-1][j][1], dp[i][j-1][1]
                #print dp[1]
        return dp[-1][-1][1]
