class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        dp = [[float('-inf')]*len(nums2) for i in range(len(nums1))]

        for i in range(len(dp)):
            for j in range(len(dp[0])):

                if i == 0 and j == 0:
                    dp[i][j] = nums1[i] * nums2[j]

                elif i >=1 and j>=1:

                    dp[i][j] = max(nums2[j]*nums1[i],dp[i-1][j-1] + nums2[j]*nums1[i], dp[i-1][j],dp[i][j-1])
                elif i == 0:
                    dp[i][j] = max([nums1[0]*nums2[k] for k in range(0,j+1)])
                elif j == 0:
                    dp[i][j] = max([nums1[k]*nums2[j] for k in range(0,i+1)])

        return dp[-1][-1]
