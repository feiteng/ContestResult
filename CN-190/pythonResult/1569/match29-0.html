class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        n1, n2 = len(nums1), len(nums2)

        dp = [[0 for _ in range(n2)] for _ in range(n1)]
        f = [[-1000005 for _ in range(n2)] for _ in range(n1)]
        ans = -1000005

        for i in range(n1):
            for j in range(n2):
                if i > 0 and j > 0:
                    t_max = max(0, f[i - 1][j - 1])
                else:
                    t_max = 0

                t_max = t_max + nums1[i] * nums2[j]
                ans = max(ans, t_max)

                if i > 0:
                    f[i][j] = max(f[i][j], f[i - 1][j])
                if j > 0:
                    f[i][j] = max(f[i][j], f[i][j - 1])
                f[i][j] = max(f[i][j], t_max)

                dp[i][j] = t_max

        return ans
