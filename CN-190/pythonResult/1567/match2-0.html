class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        y = "aeiou"
        if k == 1:
            if any([i for i in s if i in y]):
                return 1
            return 0
        i,j = 0,k-1

        res = len([e for e in s[:k] if e in y])
        maxy = res
        while(j<len(s)-1):        
            j+=1
            if s[i] in y:
                res -=1
            if s[j] in y:
                res +=1
            i+=1
            maxy = max(maxy,res)
        return maxy
