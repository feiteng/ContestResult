class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """

        set_ = set(list('aeiou'))

        curr_count 
= 0
        for i in range(k):
            if s[i] in set_:
                curr_count += 1
        max_count = curr_count
        for i in range(k, len(s)):
            if s[i] in set_:
                curr_count += 1
            if s[i - k] in set_:
                curr_count -= 1
            max_count = max(curr_count, max_count)
        return max_count
