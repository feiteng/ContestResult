class Solution(object):
    def maxVowels(self, s, k):
        alphas = ['a', 'e', 'i', 'o', 'u']
        dp = []
        n = len(s)
        for i in range(n):
            curr = 1 if s[i] in alphas else 0
            prev = 0 if i == 0 else dp[i - 1]
            dp.append(curr + prev)

        ans = 0
        
for i in range(k - 1, n):
            curr = dp[i]
            if i - k >= 0:
                curr -= dp[i - k]
            ans = max(ans, curr)

        return ans
