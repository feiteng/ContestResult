class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        i = 0
        j = k-1
        count = {'a':0, 'e':0,'i':0,'o':0,'u':0}
        vowel = ['a', 'e','i','o','u']
        res = 0
        cnt = 0
        for m in range(k):
            if s[m] in vowel:
                cnt += 1
        res 
= cnt
        while j < len(s) - 1:
            j += 1
            if s[j] in vowel:
                cnt += 1
            if s[i] in vowel:
                cnt -= 1
            i +=1
            res = max(res, cnt)
        return res
