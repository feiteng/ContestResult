class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        eset = ('a','i','u','e','o')
        ecount = 0
        for i in range(k):
            if s[i] in eset:
                ecount+=1
        emax = ecount
        for i in range(k, len(s)):
            if s[i] in eset:
                ecount+=1
            if s[i-k] in eset:
                ecount-=1
            emax = max(emax, ecount)
            if emax==k:
                break
        return emax
