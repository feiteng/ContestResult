class Solution(object):
    def isPrefixOfWord(self, sentence, searchWord):
        """
        :type sentence: str
        :type searchWord: str
        :rtype: int
        """
        tmp=split(sentence)
        for i,it in enumerate(tmp):
            if it.find(searchWord)==0:
                return i+1
            
        return -1
