class Solution(object):
    def isPrefixOfWord(self, sentence, searchWord):
        """
        :type sentence: str
        :type searchWord: str
        :rtype: int
        """
        lst = sentence.split()
        for i in range(len(lst)):
            if (searchWord == lst[i][:len(searchWord)]):
                return i + 1
        return -1
