class Solution {
public:
    string entityParser(string t) {
        string res;
        for(int i = 0; i < t.size(); i++) {
            if(t[i] == '&') {
                if(t.substr(i, 6) == "&quot;") {
                     res += "\"";
                    i += 5;
                } else if(t.substr(i, 6) == "&apos;") {
                    res += "\'";
                    i += 5;
                } else if(t.substr(i, 5) == "&amp;") {
                    res += "&";
                    i += 4;
                } else if(t.substr(i, 4) == "&gt;") {
                    res += ">";
                    i += 3;
                } else if(t.substr(i, 4) == "&lt;") {
                    res += "<";
                    i += 3;
                } else if(t.substr(i, 7) == "&frasl;") {
                    res += "/";
                    i += 6;
                } else {
                    res += "&";
                    continue;
                }
            } else {
                res += t[i];
            }
        }
        return res;
    }
};
