class Solution {
public:
    string entityParser(string r) {
        string ret = "";
        int n = r.size();
        cout << r.substr(0, 5) << endl;
        for (int i = 0; i < n; i++) {
            if (i + 5 < n && r.substr(i, 6) == "&quot;") {
                ret += '\"';
                i += 5;
            } else if (i + 5 < n && r.substr(i, 6) == "&apos;") {
                ret += '\'';
                i += 5;
            } else if (i + 6 < n && r.substr(i, 7) == "&frasl;") {
                ret += '/';
                i += 6;
            } else if (i + 4 < n && r.substr(i, 5) == "&amp;") {
                ret += '&';
                i += 4;
            } else if (i +  3< n && r.substr(i, 4) == "&lt;") {
                ret += '<';
                i += 3;
            } else if (i + 3 < n && r.substr(i, 4) == "&gt;") {
                ret += '>';
                i += 3;
            } else {
                ret += r[i];
            }
        }
        return ret;
    }
};
