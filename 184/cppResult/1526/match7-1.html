class Solution {
public:
    string entityParser(string text) {
        string ans;
        for(int i = 0, len = text.size(); i < len;){
            if(text[i] == '&'){
                if(i <= len - 6 && memcmp(&text[i], "&quot;", 6) == 0){
                    ans += '"';
                    i += 6;
                }else if(i <= len - 6 && memcmp(&text[i], "&apos;", 6) == 0){
                    ans += '\'';
                    i += 6;
                }else if(i <= len - 5 && memcmp(&text[i], "&amp;", 5) == 0){
                    ans += '&';
                    i += 5;
                }else if(i <= len - 4 && memcmp(&text[i], "&gt;", 4) == 0){
                    ans += '>';
                    i += 4;
                }else if(i <= len - 4 && memcmp(&text[i], "&lt;", 4) == 0){
                    ans += '<';
                    i += 4;
                }else if(i <= len - 7 && memcmp(&text[i], "&frasl;", 7) == 0){
                    ans += '/';
                    i += 7;
                }else{
                    ans += text[i++];
                }
            }else{
                ans += text[i++];
            }
        }
        return ans;
    }
};
