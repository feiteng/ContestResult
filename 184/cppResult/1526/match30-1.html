class Solution {
public:
    string entityParser(string text) {
        string rs;
        int len = text.length();
        for(int i=0; i<len; i++)
            if(text[i]!='&')
                rs += text[i];
            else

            {
                if(len-i>5&&text[i+1]=='q'&&text[i+2]=='u'&&text[i+3]=='o'&&text[i+4]=='t'&&text[i+5]==';')
                {
                    rs += '"';
                    i += 5;
                }
                else if(len-i>5&&text[i+1]=='a'&&text[i+2]=='p'&&text[i+3]=='o'&&text[i+4]=='s'&&text[i+5]==';')
                {
                    rs += '\'';
                    i += 5;
                }
                
                else if(len-i>4&&text[i+1]=='a'&&text[i+2]=='m'&&text[i+3]=='p'&&text[i+4]==';')
                {
                    rs += '&';
                    i += 4;
                }
                else if(len-i>3&&text[i+1]=='g'&&text[i+2]=='t'&&text[i+3]==';')
                {
                    rs += '>';
                    i += 3;
                }
                else if(len-i>3&&text[i+1]=='l'&&text[i+2]=='t'&&text[i+3]==';')
                {
                    rs += '<';
                    i += 3;
                }
                else if(len-i>6&&text[i+1]=='f'&&text[i+2]=='r'&&text[i+3]=='a'&&text[i+4]=='s'&&text[i+5]=='l'&&text[i+6]==';')
                {
                    rs += '/';
                    i += 6;
                }
                else
                    rs += text[i];
            }
        return rs;
    }
};
/*
Quotation Mark: the entity is &quot; and symbol character is ".
Single Quote Mark: the entity is &apos; and symbol character is '.
Ampersand: the entity is &amp; and symbol character is &.
Greater Than Sign: the entity is &gt; and symbol character is >.
Less Than Sign: the entity is &lt; and symbol character is <.
Slash: the entity is &frasl; and symbol character is /.
*/
