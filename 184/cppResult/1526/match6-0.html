class Solution {
public:
    string entityParser(string text) {
        string result;
        for (int i = 0; i < text.size(); i++) {
            if (text[i] == '&') {
                if (text.substr(i+1,5) == "quot;") {
                    result += '\"';
                    i += 5;
                } else if (text.substr(i+1, 5) == "apos;") {
                    result += '\'';
                    i += 5;
                } else if (text.substr(i+1, 4) == "amp;") {
                    result += "&";
                    i += 4;
                } else if (text.substr(i+1, 3) == "gt;") {
                    result += ">";
                    i += 3;
                } else if (text.substr(i+1, 3) == "lt;") {
                    result += "<";
                    i += 3;
                } else if (text.substr(i+1, 6) == "frasl;") {
                    result += "/";
                    i += 6;
                } else {
                    result += text[i];
                }
            } else {
                result += text[i];
            }
        }
        return result;
    }
};
