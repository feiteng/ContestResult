class Solution {
public:
    string entityParser(string text) {
        string ans = "";
        int len = text.length();
        for(int i = 0; i < len;) {
            if(text[i] == '&') {
                if(i + 5 < len && text[i + 1] == 'q' && text[i + 2] == 'u' && text[i + 3] == 'o' && text[i + 4] == 't' && text[i + 5] == ';') {
                    ans += '\"';
                    i += 6;
                } else if(i + 5 < len && text[i + 1] == 'a' && text[i + 2] == 'p' && text[i + 3] == 'o' && text[i + 4] == 's' && text[i + 5] == ';') {
                    ans += '\'';
                    i += 6;
                } else if(i + 4 < len && text[i + 1] == 'a' && text[i + 2] == 'm' && text[i + 3] == 'p' && text[i + 4] == ';') {
                    ans += '&';
                    i += 5;
                } else if(i + 3 < len && text[i + 1] == 'g' && text[i + 2] == 't' && text[i + 3] == ';') {
                    ans += '>';
                    i += 4;
                } else if(i + 3 < len && text[i + 1] == 'l' && text[i + 2] == 't' && text[i + 3] == ';') {
                    ans += '<';
                    i += 4;
                } else if(i + 6 < len && text[i + 1] == 'f' && text[i + 2] == 'r' && text[i + 3] == 'a' && text[i + 4] == 's' && text[i + 5] == 'l' && text[i + 6] == ';') {
                    ans += '/';
                    i += 7;
                } else {
                    ans += text[i++];
                }
            } else {
                ans += text[i++];
            }
        }
        return ans;
    }
};
