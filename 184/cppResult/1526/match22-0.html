class Solution {
public:
    string entityParser(string text) {
        unordered_map<string, string> mp;
        mp["quot;"] = "\"";
        mp["apos;"] = "'";
        mp["amp;"] = "&";
        mp["gt;"] = ">";
        mp["lt;"] = "<";
        mp["frasl;"] = "/";
        
        const int n = text.size();
        string res;
        for (int i = 0; i < n; ) {
            if (text[i] == '&') {
                if (i+6 < n && text.substr(i+1, 6) == "frasl;") {
                    res += "/";
                    i += 7;
                } else if (i+5 < n && text.substr(i+1, 5) == "quot;") {
                    res += "\"";
                    i += 6;
                } else if (i+5 < n && text.substr(i+1, 5) == "apos;") {
                    res += "'";
                    i += 6;
                } else if (i+4 < n && text.substr(i+1, 4) == "amp;") {
                    res += "&";
                    i += 5;
                } else if (i+3 < n && text.substr(i+1, 3) == "gt;") {
                    res += ">";
                    i += 4;
                } else if (i+3 < n && text.substr(i+1, 3) == "lt;") {
                    res += "<";
                    i += 4;
                } else {
                    res += '&';
                    i++;
                } 
            } else {
                res += text[i];
                i++;
            }
        }
        
        return res;
    }
};
