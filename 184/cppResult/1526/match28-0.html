class Solution {
public:
    string entityParser(string text) {
        string res;
        int n = text.length();
        int i = 0;
        while (i < n) {
            if (text[i] == '&') {
                if (match(text, "&quot;", i)) {
                    res.push_back('\"');
                    i += 6;
                } else if (match(text, "&apos;", i)) {
                    res.push_back('\'');
                    i += 6;
                } else if (match(text, "&amp;", i)) {
                    res.push_back('&');
                    i += 5;
                } else if (match(text, "&gt;", i)) {
                    res.push_back('>');
                    i += 4;
                } else if (match(text, "&lt;", i)) {
                    res.push_back('<');
                    i += 4;
                } else if (match(text, "&frasl;", i)) {
                    res.push_back('/');
                    i += 7;
                } else {
                    res.push_back(text[i]);
                    i++;
                }
            } else {
                res.push_back(text[i]);
                i++;
            }
        }
        return res;
    }
    
    bool match(string& s, string pat, int i) {
        int n = s.length(), m = pat.length();
        if (i + m > n) return false;
        for (int j = 0; j < m; j++) {
            if (pat[j] != s[i+j]) {
                return false;
            }
        }
        return true;
    }
};
