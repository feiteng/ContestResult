class Solution {
  vector<string> split(string& s,char c){
    int n=s.size();
    vector<string> res;
    for(int i=0;i<n;i++){
      if(s[i]==c) continue;
      string t;
      while(i<n&&s[i]!=c) t.push_back(s[i++]);
      res.push_back(t);
    }
    return res;
  }

//INSERT ABOVE HERE
public:
  string entityParser(string text) {
    text="&uku;"+text;
    auto vs=split(text,'&');
    string ans;

    for(auto s:vs){
      if(s.substr(0,5)=="quot;"){
        ans+="\"";
        ans+=s.substr(5);
      }else if(s.substr(0,5)=="apos;"){
        ans+="\'";
        ans+=s.substr(5);
      }else if(s.substr(0,4)=="amp;"){
        ans+="&";
        ans+=s.substr(4);
      }else if(s.substr(0,3)=="gt;"){
        ans+=">";
        ans+=s.substr(3);
      }else if(s.substr(0,3)=="lt;"){
        ans+="<";
        ans+=s.substr(3);
      }else if(s.substr(0,6)=="frasl;"){
        ans+="/";
        ans+=s.substr(6);
      }else{
        ans+='&'+s;
      }
    }
    return ans.substr(5);
  }
};
