class Solution {
public:
    string entityParser(string s) {
        string t;
        for(int i = 0; i < s.size(); i++) {
            if(i + 7 <= s.size() && s.substr(i, 7) == "&frasl;") {
                t += "/";
                i += 6;
            }
            else if(i + 6 <= s.size() && s.substr(i, 6) == "&quot;") {
                t += "\"";
                i += 5;
            }
            else if(i + 6 <= s.size() && s.substr(i, 6) == "&apos;") {
                t += "'";
                i += 5;
            }
            else if(i + 5 <= s.size() && s.substr(i, 5) == "&amp;") {
                cout << "yoo";
                t += "&";
                i += 4;
            }
            else if(i + 4 <= s.size() && s.substr(i, 4) == "&gt;") {
                t += ">";
                i += 3;
            }
            else if(i + 4 <= s.size() && s.substr(i, 4) == "&lt;") {
                t += "<";
                i += 3;
            }
            else
                t += s[i];
        }
        return t;
    }
};
