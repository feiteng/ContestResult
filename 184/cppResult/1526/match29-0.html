class Solution {
public:
    string entityParser(string s) {
        string ans;
        int n = s.size();
        for(int i = 0; i < n;) {
            if(s[i] == '&' && (i + 3 < n)) {
                if(i + 6 < n && s.substr(i+1, 6) == "frasl;") {
                    ans.push_back('/');
                    i += 7;
                } else if(i + 5 < n && s.substr(i+1, 5) == "quot;") {
                    ans.push_back('"');
                    i += 6;
                } else if(i + 5 < n && s.substr(i+1, 5) == "apos;") {
                    ans.push_back('\'');
                    i += 6;
                } else if(i + 4 < n && s.substr(i+1, 4) == "amp;") {
                    ans.push_back('&');
                    i += 5;
                } else if(i + 3 < n && s.substr(i+1, 3) == "gt;") {
                    ans.push_back('>');
                    i += 4;
                } else if(i + 3 < n && s.substr(i+1, 3) == "lt;") {
                    ans.push_back('<');
                    i += 4;
                } else {
                    ans.push_back(s[i]);
                    i++;
                }
            } else {
                ans.push_back(s[i]);
                i++;
            }
        }
        return ans;
    }
};
