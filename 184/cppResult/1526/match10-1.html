class Solution {
public:
    string entityParser(string text) {

        string result = "";
        int n = text.length();
        int i = 0;
        while (i < n) {
            if (i + 5 < n && text.substr(i, 6) == "&quot;") {
                result.push_back('\"');
                i += 6;
            } else if (i + 5 < n && text.substr(i, 6) == "&apos;") {
                result.push_back('\'');
                i += 6;
            } else if (i + 4 < n && text.substr(i, 5) == "&amp;") {
                result.push_back('&');
                i += 5;
            } else if (i + 3 < n && text.substr(i, 4) == "&gt;") {
                result.push_back('>');
                i += 4;
            } else if (i + 3 < n && text.substr(i, 4) == "&lt;") {
                result.push_back('<');
                i += 4;
            } else if (i + 6 < n && text.substr(i, 7) == "&frasl;") {
                result.push_back('/');
                i += 7;
            } else {
                result.push_back(text[i]);
                i++;
            }
        }
        return result;
    }
};
