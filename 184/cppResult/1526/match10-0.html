class Solution {
public:
    string entityParser(string text) {
        string res;
        int i=0;
        int n=(int)text.size();
        string s=text;
        while(i<n){
            /*
            
    Greater Than Sign: the entity is &gt; and symbol character is >.
    Less Than Sign: the entity is &lt; and symbol character is <.
    Slash: the entity is &frasl; and symbol character is /.

            */
            if(i+5<n&&s[i]=='&'&&s[i+1]=='q'&&s[i+2]=='u'&&s[i+3]=='o'&&s[i+4]=='t'&&s[i+5]==';'){
                res.push_back('"');
                i+=6;
            }
            else if(i+5<n&&s[i]=='&'&&s[i+1]=='a'&&s[i+2]=='p'&&s[i+3]=='o'&&s[i+4]=='s'&&s[i+5]==';'){
                res.push_back('\'');
                i+=6;
            }
            else if(i+4<n&&s[i]=='&'&&s[i+1]=='a'&&s[i+2]=='m'&&s[i+3]=='p'&&s[i+4]==';'){
                res.push_back('&');
                i+=5;
            }
            else if(i+3<n&&s[i]=='&'&&s[i+1]=='g'&&s[i+2]=='t'&&s[i+3]==';'){
                res.push_back('>');
                i+=4;
            }
            else if(i+3<n&&s[i]=='&'&&s[i+1]=='l'&&s[i+2]=='t'&&s[i+3]==';'){
                res.push_back('<');
                i+=4;
            }
            else if(i+6<n&&s[i]=='&'&&s[i+1]=='f'&&s[i+2]=='r'&&s[i+3]=='a'&&s[i+4]=='s'&&s[i+5]=='l'&&s[i+6]==';'){
                res.push_back('/');
                i+=7;
            }
            else{
                res.push_back(s[i]);
                i++;
            }
        }
        return res;
    }
};
