class Solution {
public:
    string entityParser(string text) {
        const char* str = text.c_str();
        string out;
        while (*str != '\0')
        {
            if (strncmp(str, "&quot;", 6) == 0)
            {
                out += '"';
                str += 6;
            }
            else if (strncmp(str, "&apos;", 6) == 0)
            {
                out += '\'';
                str += 6;
            }
            else if (strncmp(str, "&amp;", 5) == 0)
            {
                out += '&';
                str += 5;
            }
            else if (strncmp(str, "&gt;", 4) == 0)
            {
                out += '>';
                str += 4;
            }
            else if (strncmp(str, "&lt;", 4) == 0)
            {
                out += '<';
                str += 4;                
            }
            else if (strncmp(str, "&frasl;", 7) == 0)
            {
                out += '/';
                str += 7;
            }
            else
            {
                out += *str;
                ++str;
            }
        }
        return out;
    }
};
