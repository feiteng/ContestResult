class Solution {
public:
    vector<string> stringMatching(vector<string>& words) {
        int n = words.size();
        vector<string> res;
        for (int i = 0; i < n; ++i)
        {
            for (int t = 0; t < n; ++t)
            {
                if (i == t) continue;
                if (words[t].find(words[i]) != string::npos)
                {
                    res.push_back( words[i] );
                    break;
                }
            }
        }
        return res;
    }
};
