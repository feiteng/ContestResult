class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int>num(m);
        for(int i=0;i<m;i++)
            num[i]=i+1;
        vector<int>ret;
        for(int i=0;i<queries.size();i++)
        {
            for(int j=0;j<m;j++)
            {
                if(num[j]==queries[i])
                {
                    ret.push_back(j);
                    int tmp=num[j];
                    for(int k=j;k>0;k--)
                    {
                        num[k]=num[k-1];
                    }
                    num[0]=tmp;
                    break;
                }
            }
        }
        return ret;
    }
};
