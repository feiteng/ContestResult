class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> arr;
        vector<int> ans;
        map<int, int> pos;
        for (int i = 1; i <= m; ++i) {
            arr.push_back(i);
            pos[i] = arr.size() - 1;
        }
        for (auto q : queries) {
            int p = pos[q];
            ans.push_back(p);
            for (int i = p - 1; i >= 0; --i) {
                arr[i + 1] = arr[i];
                pos[arr[i + 1]] = i + 1;
            }
            arr[0] = q;
            pos[q] = 0;
        }
        return ans;
    }
};
