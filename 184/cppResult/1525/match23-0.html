class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        int dp[m];
        for(int i=0;i<m;i++) dp[i]=i;
        vector<int> ret(queries.size());
        for(int i=0;i<queries.size();i++){
            ret[i]=dp[queries[i]-1];
            for(int j=0;j<m;j++){
                if(dp[j]<dp[queries[i]-1]) dp[j]++;
            }
            dp[queries[i]-1]=0;
        }
        return ret;
    }
};
