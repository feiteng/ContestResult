class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p;
        for(int i = 1; i<=m; i++){
            p.push_back(i);
        }
        vector<int> ans;
        for(int q : queries){
            int pos;
            for(int i = 0; i < m; i++){
                if(p[i] == q){
                    pos = i;
                    break;
                }
            }
            ans.push_back(pos);
            for(int i = pos; i > 0; i--){
                p[i] = p[i-1];
            }
            p[0] = q;
        }
        return ans;
    }
};
