class Solution {
public:
    vector<int> processQueries(vector<int>& q, int m) {
        vector <int> v;
        for (int i = 0; i < m; i++) {
            v.push_back(i + 1);
        }
        vector <int> ret;
        for (int i = 0; i < q.size(); i++) {
            int now = q[i];
            int r = 0;
            for (int j = 0; j < v.size(); j++) {
                if (v[j] == now) {
                    r = j;
                    break;
                }
            }
            ret.push_back(r);
            for (int j = r; j >= 1; j--) {
                v[j] = v[j - 1];
            }
            v[0] = now;
        }
        return ret;
    }
};
