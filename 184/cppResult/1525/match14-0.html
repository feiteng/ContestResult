class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> data;
        for (int i = 1; i <= m; ++i) {
            data.push_back(i);
        }
        
        vector<int> res;
        for (int key : queries) {
            auto pos = find(data.begin(), data.end(), key);
            int idx = pos - data.begin();
            
            res.push_back(idx);
            for (int i = idx - 1; i >= 0; --i) {
                data[i + 1] = data[i];
            }
            data.front() = key;
        }
        
        return res;
    }
};
