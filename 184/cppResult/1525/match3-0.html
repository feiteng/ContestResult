class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> data(m);
        for(int i = 0; i < m; ++i)
            data[i] = i + 1;
        
        vector<int> res;
        for(int q: queries) {
            for(int i = 0; i < m; ++i) {
                if(data[i] == q) {
                    res.push_back(i);
                    data.erase(data.begin() + i);
                    data.insert(data.begin(), q);
                    break;
                }
            }
        }
        return res;
    }
};
