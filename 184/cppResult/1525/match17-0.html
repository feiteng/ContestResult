class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> res;
        vector<int> perm(m, 0);
        for(int i=0; i<m; ++i){
            perm[i] = i+1;
        }
        for(auto q: queries){
            // auto pos = perm.find(q);
            auto pos = find(perm.begin(), perm.end(), q);
            int npos = pos - perm.begin();
            res.emplace_back(npos);
            perm.erase(pos, pos+1);
            perm.insert(perm.begin(), q);
        }
        
        
        return res;
        
    }
};
