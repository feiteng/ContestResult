class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> v(m);
        unordered_map<int, int> f;
        for (int i = 0; i < m; ++i) {
            v[i] = i + 1;
            f[i + 1] = i;
        }
        
        vector<int> res;
        for (int i = 0; i < queries.size(); ++i) {
            int t = f[queries[i]];
            res.push_back(t);
            for (int j = t; j > 0; --j) {
                v[j] = v[j - 1];
                f[v[j]] = j;
            }
            v[0] = queries[i];
            f[queries[i]] = 0;
        }
        return res;
    }
};
