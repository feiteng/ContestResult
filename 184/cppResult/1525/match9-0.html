class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        unordered_map<int, int> p;
        for(int i = 1; i <= m; ++i) {
            p[i] = i - 1;
        }
        vector<int> ans;
        for(int i = 0; i < queries.size(); ++i) {
            int t = p[queries[i]];
            ans.push_back(t);
            for(int i = 1; i <= m; ++i) {
                if(p[i] < t)
                    ++p[i];
            }
            p[queries[i]] = 0;
        }
        return ans;
    }
};
