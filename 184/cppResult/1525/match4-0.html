class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> a(m);
        iota(a.begin(), a.end(), 1);
        vector<int> ans;
        for (auto& x: queries) {
            auto itr = find(a.begin(), a.end(), x);
            ans.push_back(itr - a.begin());
            a.erase(itr);
            a.insert(a.begin(), x);
        }
        return ans;
    }
};
