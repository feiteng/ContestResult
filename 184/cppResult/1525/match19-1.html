class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p(m);
        for (int i = 0; i < m; ++i) {
            p[i] = i + 1;
        }
        int n = queries.size();

        vector<int> result(n);
        for (int q = 0; q < n; ++q) {
            int pos = -1;
            for (int i = 0; i < m; ++i) {
                if (p[i] == queries[q]) {
                    pos = i;
                    break;
                }
            }
            result[q] = pos;

            for (int i = pos; i > 0; --i) {
                p[i] = p[i - 1];
            }
            p[0] = queries[q];
        }
        return result;
    }
};
