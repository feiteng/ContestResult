class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p;
        for (int i = 0; i < m; i++) {
            p.push_back(i + 1);
        }

        vector<int> ret;
        for (int a : queries) {
            int pos;
            for (int i = 0; i < m; i++) {
                if (p[i] == a) {
                    pos = i;

                }
            }

            ret.push_back(pos);

            for (int i = pos; i > 0; i--) {
                p[i] = p[i-1];
            }
            p[0] = a;
        }

        return ret;
    }
};
