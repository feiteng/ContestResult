class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m)
    {
        vector<int> data(m, 0);
        for (int i = 0; i < m; ++i) {
            data[i] = i + 1;
        }
        vector<int> result;
        for (int idx : queries) {
            int pos = 0;
            for (int i = 0; i < m; ++i) {
                if (data[i] == idx) {
                    pos = i;
                    break;
                }
            }
            result.push_back(pos);
            
            for (int i = pos - 1; i >= 0; --i) {
                data[i + 1] = data[i];
            }
            data[0] = idx;
        }
        
        return result;
    }
};
