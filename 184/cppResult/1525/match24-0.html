class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> pos(m+1);
        for(int i = 0; i < m; i++) {
            pos[i+1] = i;
        }
        int n = queries.size();
        vector<int> ans(n);
        for(int i = 0; i < n; i++) {
            int x = pos[queries[i]];
            ans[i] = x;
            for(int j = 1; j <= m; j++) {
                if(pos[j] < x) pos[j]++;
            }
            pos[queries[i]] = 0;
        }
        return ans;
    }
};
