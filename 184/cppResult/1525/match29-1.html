class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> v(m, 0), res;
        for (int i = 0; i < m; i++) v[i] = i+1;

        for (int i = 0; i < queries.size(); i++) {
            int q = queries[i];
            int idx = 0;
            for (int j = 0; j < m; j++) {
                if (v[j] == q) {
                    idx = j;
                    break;
                }
            }
            res.push_back(idx);
            int tmp = v[idx];

            for (int j = idx; j > 0; j--) {
                v[j] = v[j-1];
            }
            v[0] = tmp;
        }
        return res;
    }
};
