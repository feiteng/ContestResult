class Solution {
public:
  vector<int> processQueries(vector<int>& queries, int m) {
    vector<int> p(m);
    vector<int> res;
    iota(p.begin(), p.end(), 1);
    for (auto v : queries) {
      auto it = find(p.begin(), p.end(), v);
      res.emplace_back(it - p.begin());
      p.erase(it);
      p.emplace(p.begin(), v);
    }
    return res;
  }
};
