class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> pos(m+1);
        for(int i = 0; i<m+1; i++){
            pos[i] = i-1;
        }

        vector<int> res(queries.size(),0);
        for(int j = 0; j < queries.size(); j++){
            int q = queries[j];
            res[j] =  pos[q];
            for(int i = 1; i < m+1; i++){
                if(pos[i] < pos[q])
                    pos[i]++;
            }
            pos[q] = 0;
        }
        return res;
    }
};
