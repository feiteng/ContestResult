class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> nums;
        for (int i = 0; i < m; i++) {
            nums.push_back(i + 1);
        }
        vector<int> result;
        for (const auto& a : queries) {
            int pos = -1;
            for (int i = 0; i < m; i++) {
                if (nums[i] == a) {
                    result.push_back(i);
                    pos = i;
                    break;
                }
            }
            int target = nums[pos];
            for (int i = pos - 1; i >= 0; i--) {
                nums[i + 1] = nums[i];
            }
            nums[0] = target;
        }
        return result;
    }
};
