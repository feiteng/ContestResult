class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        int P[m];
        unordered_map<int, int> valueToIdx;
        for (int i = 0; i < m; i++){
            P[i] = i+1;
            valueToIdx[i+1] = i;
        }
        vector<int> answer;
        for (int i = 0; i < queries.size(); ++i){
            int idx = valueToIdx[queries[i]];
            answer.push_back(idx);
            
            // Change the array
            for (int j = idx-1; j >= 0; --j){
                P[j+1] = P[j]; valueToIdx[P[j+1]] = j+1;
            }
            P[0] = queries[i]; valueToIdx[P[0]] = 0;
        }
        return answer;
    }
};
