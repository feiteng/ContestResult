class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p(m), ans;
        
        for (int i = 0; i < m; i++) p[i] = i+1;
        
        for (int q : queries) {
            for (int i = 0; i < m; i++) {
                if (q == p[i]) {
                    ans.push_back(i); p.erase(p.begin()+i);
                    p.insert(p.begin(), q); break;
                }
            }
        }
        return ans;
    }
};
