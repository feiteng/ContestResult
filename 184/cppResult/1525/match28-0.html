class Solution {
public:
    
    vector<int> processQueries(vector<int> &a, int m) {
        vector<int> b(m);
        for(int i=0;i<m;++i) b[i]=i+1;
        vector<int> ans;
        for(int q:a) {
            int x;
            for(int i=0;i<m;++i) if(b[i]==q) {x=i;break;}
            b.erase(b.begin()+x);
            ans.push_back(x);
            b.insert(b.begin(),q);
        }
        return ans;
    }
};
