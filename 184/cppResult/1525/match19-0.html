class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> arr(m);
        for (int i=0; i<m; ++i) {
            arr[i] = i + 1;
        }
        vector<int> res(queries.size());
        for (int t=0; t<res.size(); ++t) {
            int v = queries[t];
            for (int i=0; i<arr.size(); ++i) {
                if (arr[i] == v) {
                    res[t] = i;
                    break;
                }
            }
            for (int i=res[t]; i>=1; --i) {
                arr[i] = arr[i-1];
            }
            arr[0] = v;
        }
        return res;
    }
};
