class Solution {
    int f[5005][20];
public:
    int numOfWays(int n) {
        int ways[20][3] = {};
        bool match[20][20] = {};
        int m = 0;
        for (int i = 0 ; i < 3 ; ++ i) {
            for (int j = 0 ; j < 3 ; ++ j) {
                for (int k = 0 ; k < 3 ; ++ k) {
                    if (i != j && j != k) {
                        ways[m][0] = i;
                        ways[m][1] = j;
                        ways[m][2] = k;
                        ++ m;
                        
                    }
                }
            }
        }
        for (int i = 0 ; i < m ; ++ i) {
            for (int j = 0 ; j < m ; ++ j) {
                match[i][j] = ways[i][0] != ways[j][0] && ways[i][1] != ways[j][1] && ways[i][2] != ways[j][2];
            }
        }
        
        int Q = 1e9 + 7;
        memset(f, 0 , sizeof(f));
        for (int i = 0 ; i < m ; ++ i) {
            f[1][i] = 1;
        }
        for (int i = 2 ; i <= n ; ++ i) {
            for (int j = 0 ; j < m ; ++ j) {
                for (int k = 0 ; k < m ; ++ k) {
                    if (match[k][j]) {
                        f[i][j] += f[i - 1][k];
                        f[i][j] %= Q;
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0 ; i < m ; ++ i) {
            res += f[n][i];
            res %= Q;
        }
        return res;
        
    }
};
