const int M = 1e9 + 7;

class Solution {
public:
    int numOfWays(int n) {
        int a = 6, b = 6;
        for (int i = 1; i < n; ++i) {
            int c = a * 2 % M + 2 * b % M;
            c %= M;
            int d = a * 2 % M + 3LL * b % M;
            d %= M;
            a = c;
            b = d;
        }
        return (a + b) % M;
    }
};
