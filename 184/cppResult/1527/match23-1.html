class Solution {
public:
    int numOfWays(int n) {
        if (n == 1) return 12;

        long long abc = 6, aba = 6;
        for(int i = 2; i <= n; i++) {
            long long  n_aba = (3 * aba) % (1000000000 + 7) + (2 * abc) % (1000000000 + 7);
            long long n_abc = (2 * aba) % (1000000000 + 7) + (2 * abc) % (1000000000 + 7);
            aba = n_aba % (1000000000 + 7);
            abc = n_abc % (1000000000 + 7);
        }
        return (aba + abc) % (1000000000 + 7);
    }
};
