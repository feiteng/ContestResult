class Solution {
public:
    const int MOD = 1e9 + 7;
    int numOfWays(int n) {
        long long t1 = 6, t2 = 6;
        for (int i = 2; i <= n; i++) {
            long long nt1 = 0, nt2 = 0;
            nt1 += t1 * 3; nt2 += t1 * 2;
            nt1 += t2 * 2; nt2 += t2 * 2;
            t1 = nt1 % MOD; t2 = nt2 % MOD;
        }
        return (t1 + t2) % MOD;
    }
};
