class Solution {
public:
    
    int mod = 1e9+7;
    
    bool valid(int i, int j, int k)
    {
        return (i != j && j != k);
    }
    
    int dp[5100][3][3][3];
    
    int numOfWays(int n) {
        
        memset(dp, 0, sizeof(dp));
        

        for (int j1 = 0; j1 <= 2; j1++)
            for (int k1 = 0; k1 <= 2; k1++)
                for (int l1 = 0; l1 <= 2; l1++)
                    if (valid(j1, k1, l1))
                        dp[1][j1][k1][l1] = 1;
        for (int i = 2; i <= n; i++)
        {
            for (int j1 = 0; j1 <= 2; j1++)
                for (int k1 = 0; k1 <= 2; k1++)
                    for (int l1 = 0; l1 <= 2; l1++)
                    {
                        if (valid(j1, k1, l1))
                        {

                            for (int j2 = 0; j2 <= 2; j2++)
                                for (int k2 = 0; k2 <= 2; k2++)
                                    for (int l2 = 0; l2 <= 2; l2++)
                                    {
                                        if (valid(j2, k2, l2) && j2 != j1 && k2 != k1 && l2 != l1)
                                        {
                                            dp[i][j2][k2][l2] = (dp[i - 1][j1][k1][l1] + dp[i][j2][k2][l2]) % mod;
                                        }
                                    }
                        }
                    }
        }
        

        int ans = 0;
            for (int j1 = 0; j1 <= 2; j1++)
                for (int k1 = 0; k1 <= 2; k1++)
                    for (int l1 = 0; l1 <= 2; l1++)
                        ans = (ans + dp[n][j1][k1][l1]) % mod;
        return ans;
        
    }
};
