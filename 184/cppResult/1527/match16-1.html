class Solution {
public:
    int f[5005][3][3][3];
    
    int numOfWays(int n) {

        for (int i = 0; i < n; i++) {
            for (int a = 0; a < 3; a++) {
                for (int b = 0; b < 3; b++) {
                    for (int c = 0; c < 3; c++) {
                        f[i][a][b][c] = 0;
                    }
                }
            }
        }
        
        int MOD = 1000000007;

        for (int i = 0; i < n; i++) {
            for (int a = 0; a < 3; a++) {
                for (int b = 0; b < 3; b++) {
                    for (int c = 0; c < 3; c++) {
                        if (a == b || b == c) {
                            continue;
                        }
                        if (i == 0) {
                            f[i][a][b][c] = 1;
                        } else {

                            for (int aa = 0; aa < 3; aa++) {
                                for (int bb = 0; bb < 3; bb++) {
                                    for (int cc = 0; cc < 3; cc++) {
                                        if (aa != a && bb != b && cc != c) {
                                            (f[i][a][b][c] += f[i - 1][aa][bb][cc]) %= MOD;
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        int ans = 0;
        for (int aa = 0; aa < 3; aa++) {
                                for (int bb = 0; bb < 3; bb++) {
                                    for (int cc = 0; cc < 3; cc++) {
                                        (ans += f[n - 1][aa][bb][cc]) %= MOD;
                                    }
                                }
        }
        return ans;
    }
};
