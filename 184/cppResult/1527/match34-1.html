class Solution {
public:
    int numOfWays(int n) {
        const int mod = 1e9+7;
        int dp[n][12];
        memset(dp,0,sizeof(dp));
        vector<string> st = { "RYR", "RYG", "RGR", "RGY", "YRY", "YRG", "YGR", "YGY", "GRY", "GRG", "GYR", "GYG"};
        

        for(int i=0; i<12; ++i){
            dp[0][i] = 1;
        }
        
        for(int i=1; i<n; ++i){
            for(int j=0; j<12; ++j){
                string curr = st[j];

                for(int k=0; k<12; ++k){
                    if( curr[0] != st[k][0] and curr[1] != st[k][1] and curr[2] != st[k][2] ){
                        dp[i][j] += dp[i-1][k];
                        dp[i][j] %= mod;
                    }
                }
            }
        }
        int ans = 0;
        for(int i=0; i<12; ++i){
            ans += dp[n-1][i];
            ans %= mod;
        }
        return ans;
        
    }
};
