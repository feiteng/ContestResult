class Solution {
public:
    int numOfWays(int n) {
        if(n == 1) return 12;
        const long long MOD = 1000000007;
        long long x12 = 6,x111 = 6;
        for(int i = 2;i <= n;i++){
            long long a = x12 * 3 + x111 * 2;
            long long b = x12 * 2 + x111 * 2;
            x12 = a % MOD;
            x111 = b % MOD;
        }
        return (x12 + x111) % MOD;
    }
};
