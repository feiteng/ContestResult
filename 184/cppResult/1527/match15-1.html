class Solution {
public:

    int numOfWays(int n) {
        long long a = 6, b = 6;
        for (int i = 1;i < n;++i){
            long long p = 0, q = 0;
            p += 2 * a;
            q += 2 * a;
            p += 2 * b;
            q += 3 * b;
            a = p % 1000000007;
            b = q % 1000000007;
        }
        return (a + b) % 1000000007;
    }
};
