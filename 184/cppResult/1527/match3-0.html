#define MAX 1000000007
class Solution {
public:
    int numOfWays(int n) {
        int64_t aba = 6; 
        int64_t abc = 6;
        for (int i = 2; i <= n; ++i)
        {
            int64_t newAba = aba * 3 + abc * 2;
            int64_t newAbc = aba * 2 + abc * 2;
            aba = newAba % MAX;
            abc = newAbc % MAX;
            //cout << aba << ' ' << abc << endl;
        }
        
        return (aba + abc) % MAX;
    }
};
