class Solution {
public:
    int numOfWays(int n) {
        constexpr long long kPrime = 1e9 + 7;
        long long f = 1, g = 1;
        for (int i = 1; i < n; ++i) {
            long long last_f = f;
            long long last_g = g;
            f = (2 * last_f % kPrime + 2 * last_g % kPrime) % kPrime;
            g = (2 * last_f % kPrime + 3 * last_g % kPrime) % kPrime;
        }
        return static_cast<int>((f + g) % kPrime * 6 % kPrime);
    }
};
