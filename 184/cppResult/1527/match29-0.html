class Solution {
public:
    
    typedef long long ll;
    ll dp[5005][3][3][3];
    static const int mod  = 1e9+7;
    int numOfWays(int n) {
        
        for(int i=0;i<3;++i)
            for(int j=0;j<3;++j)
                for(int k=0;k<3;++k)
                    if(i!=j && j!=k) dp[1][i][j][k] = 1;
        for(int p=2;p<=n;++p) {
            for(int i=0;i<3;++i)
                for(int j=0;j<3;++j)
                    for(int k=0;k<3;++k)
                        for(int ii=0;ii<3;++ii)
                            for(int jj=0;jj<3;++jj)
                                for(int kk=0;kk<3;++kk)
                                    if(i!=j && j!=k && ii!=i && jj!=j && kk!=k)
                                        (dp[p][i][j][k] += dp[p-1][ii][jj][kk]) %= mod;
        }
        ll ans = 0;
        for(int i=0;i<3;++i)
            for(int j=0;j<3;++j)
                for(int k=0;k<3;++k)
                    ans += dp[n][i][j][k];
        ans %= mod;
        return ans;
    }
};
