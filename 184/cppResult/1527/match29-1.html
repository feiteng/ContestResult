#include <bits/stdc++.h>

using namespace std;

int f[5005][3][3][3];
const int mod = (int )1e9 + 7;
class Solution {
public:
    void upd(int& a, int b) {
        a += b;
        if (a >= mod) a -= mod;
    }
    int numOfWays(int n) {
        memset(f, 0, sizeof(f));

         for (int a = 0; a < 3; ++ a)
                for (int b = 0; b < 3; ++ b)
                    for (int c = 0; c < 3; ++ c) 
                        if (a != b && b != c) f[1][a][b][c] = 1;

        int ans = 0;
        for (int i = 2; i <= n; i ++) {
            for (int a = 0; a < 3; ++ a)
                for (int b = 0; b < 3; ++ b)
                    for (int c = 0; c < 3; ++ c) {
                        if (a == b || b == c) continue;

                        for (int x = 0; x < 3; ++ x)
                            for (int y = 0; y < 3; ++ y)
                                for (int z = 0; z < 3; ++ z) {
                                    if (x != a && y != b && z != c && x != y && y != z)
                                        upd(f[i][x][y][z], f[i - 1][a][b][c]);
                                }
                    }
        }

            for (int a = 0; a < 3; ++ a)
                for (int b = 0; b < 3; ++ b)
                    for (int c = 0; c < 3; ++ c) upd(ans, f[n][a][b][c]);
        return ans;
    }
};
