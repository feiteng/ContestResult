class Solution {
public:
    ll dp[5123][2];
    
    int numOfWays(int n) {
        dp[1][0] = 6;
        dp[1][1] = 6;
        
        for(int i = 2; i <= n; i++){
            dp[i][0] = 2 * dp[i-1][0] + 2 * dp[i-1][1];
            dp[i][1] = 2 * dp[i-1][0] + 3 * dp[i-1][1];
            
            dp[i][0] %= MOD;
            dp[i][1] %= MOD;
        }
        
        return (dp[n][0] + dp[n][1]) % MOD;
    }
};
