typedef long long ll;
const ll mod = 1e9 + 7;

class Solution {
public:
    
    ll DP[5005][3][3][3];
    
    int numOfWays(int N) {
        for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j){
                for(int k = 0; k < 3; ++k){
                    if(i == j || j == k)continue;
                    DP[1][i][j][k] = 1;
                }
            }
        }
        for(int row = 2; row <= N; ++row){
            for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j){
                for(int k = 0; k < 3; ++k){
                    if(i == j || j == k)continue;
                    for(int x = 0; x < 3; ++x){
                        for(int y = 0; y < 3; ++y){
                            for(int z = 0; z < 3; ++z){
                                if(x == y || y == z)continue;
                                if(x == i || y == j || z == k)continue;
                                DP[row][i][j][k] = (DP[row][i][j][k] + DP[row - 1][x][y][z]) % mod;
                            }
                        }
                    }
                }
            }
        }
        }
        
        ll ret = 0;
        for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j){
                for(int k = 0; k < 3; ++k){
                    if(i == j || j == k)continue;
                    ret = (ret + DP[N][i][j][k]) % mod;
                }
            }
        }
        return ret;
    }
};
