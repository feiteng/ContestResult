class Solution {
public:
    int numOfWays(int n) {
      long long kMod = 1e9 + 7;
      long long same = 6, diff = 6;
      
      for (int i = 2; i <= n; ++i) {
        long long ns = (3 * same + 2 * diff) % kMod;
        long long nd = (2 * same + 2 * diff) % kMod;
        same = ns;
        diff = nd;
      }
      return (same + diff) % kMod;
    }
};
