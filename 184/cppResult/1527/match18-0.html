typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  int numOfWays(int n) {
    vector<ll> s(n),d(n);
    s[0]=d[0]=6;
    for(int i=1;i<n;i++) {
      s[i] = (3*s[i-1]+2*d[i-1])%MOD;
      d[i] = (2*s[i-1]+2*d[i-1])%MOD;
    }
    return (s[n-1]+d[n-1])%MOD;
  }
};
