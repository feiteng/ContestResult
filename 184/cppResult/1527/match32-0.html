class Solution {
public:
    long long f[5010][2];
    const int mods = 1e9+7;
    int numOfWays(int n) {
        f[1][1] = f[1][0] = 6;
        for(int i=2;i<=n;i++){
            f[i][1] = f[i-1][1]*3%mods+f[i-1][0]*2%mods;
            f[i][1] %= mods;
            f[i][0] = f[i-1][1]*2%mods+f[i-1][0]*2%mods;
            f[i][0]%=mods;
        }
        long long ans = 0;
        ans = (f[n][0]+f[n][1])%mods;
        return ans;
    }
};
