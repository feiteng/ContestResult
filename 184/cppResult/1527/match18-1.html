long long int dp[10000][2];



class Solution {
public:
    int numOfWays(int n) {
        dp[1][0]=6;
        dp[1][1]=6;
        for(int i=2;i<=n;i++)
        {
            dp[i][0]=(2*(dp[i-1][0]+dp[i-1][1]))%1000000007;
            dp[i][1]=((2*dp[i-1][0]+3*dp[i-1][1]))%1000000007;
        }
        return (dp[n][0]+dp[n][1])%1000000007;
        
    }
};
