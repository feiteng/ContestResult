const int MOD = 1e9+7;
class Solution {
public:
    int numOfWays(int n) {
        int same = 6, diff = 6;
        for (int i=2; i<=n; ++i) {
            int nextsame = (((same + same) % MOD + same) % MOD + (2 * diff) % MOD) % MOD;
            int nextdiff = (2 * same % MOD + 2 * diff % MOD) % MOD;
            same = nextsame;
            diff = nextdiff;
        }
         return ((same + diff) % MOD + MOD) % MOD;
    }
};
