class Solution {
public:
    int dp[5005][3][3][3];
    const int MOD=1e9+7;
    int numOfWays(int n) {
        memset(dp,0,sizeof(dp));

        for (int i=0; i<=2; ++i) {
        for (int j=0; j<=2; ++j) {
        for (int k=0; k<=2; ++k) {
            if (i==j || j==k) continue;

            dp[1][i][j][k]=1;
        }}}
        for (int x=2; x<=n; ++x) {
        for (int i=0; i<=2; ++i) {
        for (int j=0; j<=2; ++j) {
        for (int k=0; k<=2; ++k) {
            if (i==j || j==k) continue;

            for (int ii=0; ii<=2; ++ii) {
            for (int jj=0; jj<=2; ++jj) {
            for (int kk=0; kk<=2; ++kk) {
                if (ii==i || jj==j || kk==k) continue;
                dp[x][i][j][k]=(dp[x][i][j][k]%MOD+dp[x-1][ii][jj][kk]%MOD)%MOD;
            }}}
        }}}}

        int ans=0;
        for (int i=0; i<=2; ++i) {
        for (int j=0; j<=2; ++j) {
        for (int k=0; k<=2; ++k) {
            ans=(ans%MOD+dp[n][i][j][k]%MOD)%MOD;
        }}}
        return ans;
    }
};
