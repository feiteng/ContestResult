using ll = long long;
ll MOD = 1e9+7;
class Solution {
public:
    int numOfWays(int n) {
        ll a = 6, b=6;
        for(int i=2;i<=n;i++){
            ll x = (3*a + 2*b)%MOD;
            ll y = 2*(a+b)%MOD;
            a = x;
            b = y;
        }
        return (a+b) % MOD;
    }
};
