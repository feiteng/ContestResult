class Solution {
public:
    int numOfWays(int n) {
        long long dp[5005][3][3][3] = {0};
        long long MOD = 1e9 + 7;
        for (int x = 0; x < 3; x++) {
            for (int i = 0;i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    if (x == i || i == j) {
                            continue;
                        }
                    dp[1][x][i][j] = 1;
                }
            }
        }
        for (int x = 2; x <= n; x++) {
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    for (int y = 0; y < 3; y++) {
                        
                        if (i == j || j == y) {
                            continue;
                        }
                        
                        for (int a = 0; a < 3; a++) {
                            for (int b= 0; b < 3; b++) {
                                for (int c = 0; c < 3; c++) {
                                    if (a != i && b != j && c != y) {
                                        dp[x][i][j][y] += dp[x-1][a][b][c];
                                        dp[x][i][j][y] %= MOD;
                                    }    
                                    
                                }
                            
                            }
                        }
                        
                    }
                }
            }
        }
        long long ans = 0;
        for (int x = 0; x < 3; x++) {
            for (int i = 0;i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    ans += dp[n][x][i][j];
                    ans %= MOD;
                }
            }
        }
        return ans;
    }
};
