	class Solution {
	    public List<String> stringMatching(String[] words) {
	        List<String> ret = new ArrayList<>();
	        for(String w : words){
	        	for(String q : words){
	        		if(q.equals(w))continue;
	        		if(q.contains(w)){
	        			ret.add(w);
	        			break;
	        		}
	        	}
	        }
	        return ret;
	    }
	}
