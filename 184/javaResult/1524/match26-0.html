class Solution {
    public List<String> stringMatching(String[] words) {
        List<String> ret = new ArrayList<>();
        for (String w : words) {
            for (String a : words) {
                if (!w.equals(a) && a.indexOf(w)>=0) {
                    ret.add(w);
                    break;
                }
            }
        }
        return ret;
        
    }
}
