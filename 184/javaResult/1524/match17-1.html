class Solution {
    public List<String> stringMatching(String[] words) {
        List<String> res = new ArrayList<>();
        for (String s : words) {
            for (String t : words) {
                if (t != s && t.contains(s)) {
                    res.add(s);
                    break;
                }
            }
        }
        return res;
    }
}
