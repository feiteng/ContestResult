import java.util.ArrayList;
import java.util.List;

class Solution {
  public List<String> stringMatching(String[] words) {
    ArrayList<String> ret = new ArrayList<>();
    
    for (int i = 0; i < words.length; i++) {
      for (int j = 0; j < words.length; j++) {
        if (i == j) {
          continue;
        }
        String s1 = words[i];
        String s2 = words[j];
        if (s2.contains(s1)) {
          ret.add(s1);
          break;
        }
      }
    }
    
    return ret;
  }
}
