class Solution {
    public List<String> stringMatching(String[] w) {
        List<String> result = new ArrayList<>();
        int n = w.length;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i == j) {
                    continue;
                }
                
                
if (w[j].contains(w[i])) {
                    result.add(w[i]);
                    break;
                }
            }
        }
        return result;
    }
}
