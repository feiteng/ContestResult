class Solution {
    public List<String> stringMatching(String[] w) {
        List<String> ans = new ArrayList<>();
        for(int i = 0; i < w.length; i++){
            for(int j = 0; j < w.length; j++){
                if(!w[j].equals(w[i]) && w[j].contains(w[i])){
                    ans.add(w[i]);
                    break;
                }
            }
        }
        return ans;
    }
}
