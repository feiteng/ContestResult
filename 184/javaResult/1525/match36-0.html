class Solution {
    public int[] processQueries(int[] queries, int m) {
        LinkedList<Integer> linkedList = new LinkedList<>();
        for (int i = 1; i <= m; i++) {
            linkedList.add(i);
        }
        int n = queries.length;
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            int t = linkedList.indexOf(queries[i]);
            ans[i] = t;
            linkedList.addFirst(linkedList.remove(t));
        }
        return ans;
    }
}
