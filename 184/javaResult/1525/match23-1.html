class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] res = new int[queries.length];
        List<Integer> l = new ArrayList<>();
        for (int i = 1; i <= m; i++) {
            l.add(i);
        
}
        
        for (int i = 0; i < queries.length; i++) {
            int idx = l.indexOf(queries[i]);
            res[i] = idx;
            l.remove(idx);
            l.add(0, queries[i]);
        }
        
        return res;
    }
}
