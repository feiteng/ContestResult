class Solution {
    public int[] processQueries(int[] queries, int m) {
        LinkedList<Integer> vals = new LinkedList<>();
        for(int i = 1; i <= m; i++){
            vals.add(i);
        }
        int pos = 0;
        int [] result = new int[queries.length];
        for(int val: queries){
            int front = 0;
            int back = vals.size() - 1;
            int index = -1;
            while(front <= back){
                if(vals.get(front) == val){
                    index = front;
                }else if(vals.get(back) == val){
                    index = back;
                }
                front++;
                back--;
            }
            
vals.remove(index);
            vals.addFirst(val);
            result[pos++] = index;
        }
        return result;
    }
}
