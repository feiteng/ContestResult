class Solution {
    public int[] processQueries(int[] queries, int m) {
        LinkedList<Integer> res = new LinkedList<>();
        for(int i = 1; i<=m; i++) res.add(i);
        int[] ans = new int[queries.length];
        for(int i = 0; i<queries.length; i++) {
            int q = queries[i];
            
for(int j = 0; j<res.size(); j++) {
                if(res.get(j)==q) {
                    ans[i]=j;
                    res.remove(j);
                    res.addFirst(q);
                    break;
                }
            }
        }
        return ans;
    }
}
