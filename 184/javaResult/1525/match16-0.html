class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] data = new int[m];
        for (int i = 0; i < m; i++) {
            data[i] = i + 1;
        }
        int[] res = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            for (int j = 0; j < m; j++) {
                if (data[j] == queries[i]) {
                    res[i] = j;
                    for (int k = j - 1; k >= 0; k--) {
                        data[k + 1] = data[k];
                    }
                    data[0] = queries[i];
                }
            }
        }
        return res;
    }
}
