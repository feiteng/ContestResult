class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> list = new LinkedList<>();
        for(int i = 1; i <= m; ++i) list.add(i);
        int n = queries.length;
        int[] ans = new int[n];
        int cnt = 0;
        for(int x : queries){
            for(int i = 0; i < list.size(); ++i){
                if(x == list.get(i)){
                    ans[cnt++] = i;
                    list.remove(i);
                    list.add(0, x);
                    break;
                }
            }
        }
        return ans;
    }
}
