	class S
olution {
	    public int[] processQueries(int[] queries, int m) {
	        int[] a = new int[m];
	        for(int i = 0;i < m;i++){
	        	a[i] = i+1;
	        }
	        int[] ret = new int[queries.length];
	        for(int i = 0;i < queries.length;i++){
	        	int q = queries[i];
	        	for(int j = 0;
j < m;j++){
	        		if(a[j] == q){
	        			ret[i] = j;
	        			for(int k = j;k >= 1;k--){
	        				a[k] = a[k-1];
	        			}
	        			a[0] = q;
	        			break;
	        		}
	        	}
	        }
	        return ret;
	    }
	}
