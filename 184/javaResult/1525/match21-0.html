class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] arr = new int[m];
        for (int i = 0; i < m; i++) {
            arr[i] = i+1;
        }
        int j = 0;
        int[] res = new int[queries.length];
        for (int q : queries) {
            // System.out.println(Arrays.toString(arr));
            for (int i = 0; i < m; i++) {
                if (arr[i] == q) {
                    res[j] = i;
                    j++;
                    for (int e = i; e > 0; e--) {
                        arr[e] = arr[e-1];
                    }
                    arr[0] = q;
                    break;
                }
            }
        }
        return res;
    }
}
