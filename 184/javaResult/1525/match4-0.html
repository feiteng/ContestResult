class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> nums = new ArrayList<>();
    for (int i = 1; i <= m; i++) {
      nums.add(i);
    }
    int[] output = new int[queries.length];
    for (int i = 0; i < queries.length; i++) {
      int k = 0;
      while (nums.get(k) != queries[i]) {
        k++;
      }
      output[i] = k;
      nums.remove(k);
      nums.add(0, queries[i]);
    }
    return output;
    }
}
