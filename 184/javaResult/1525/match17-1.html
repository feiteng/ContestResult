class Solution {
    public int[] processQueries(int[] q, int m) {
        int[] arr = new int[m];
        for (int i = 0; i < m; i++) arr[i] = i + 1;
        int n = q.length;
        
int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (arr[j] == q[i]) {
                    res[i] = j;
                    //int temp = arr[j];
                    for (int k = j; k >= 1; k--) {
                        arr[k] = arr[k - 1];
                    }
                    arr[0] = q[i];
                    break;
                
}
            }
        }
        return res;
    }
}
