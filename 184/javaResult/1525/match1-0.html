class Solution {
    public int[] processQueries(int[] queries, int m) {
        ArrayList<Integer> temp  =new ArrayList<>();
        
        for(int i=1;i<=m;i++)
        {
            temp.add(i);
        }
        
        int[] ans =new int[queries.length];
        
        for(int i=0;i<queries.length;i++)
        {
            int pos=  temp.indexOf(queries[i]);
            ans[i]=pos;
            temp.remove(pos);
            temp.add(0,queries[i]);
        }
        return ans;
    }
}
