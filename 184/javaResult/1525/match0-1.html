class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] pos = new int[m + 1];
        int[] arr = new int[m];
        for (int i = 0; i < m; i++) {
            pos[i] = i - 1;
            arr[i] = i + 1;
        }
        pos[m] = m - 1;
        // System.out.println(Arrays.toString(pos) + Arrays.toString(arr));
        int[] res = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int num = queries[i];
            int position = pos[num];
            res[i] = position;
            for (int j = position; j > 0; j--) {
                int prev = arr[j - 1];
                arr[j] = prev;
                pos[prev] = j;
            }
            pos[num] = 0;
            arr[0] = num;
            // System.out.println(Arrays.toString(pos) + Arrays.toString(arr));
        }
        
        return res;
    }
}
