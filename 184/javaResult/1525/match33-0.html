class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> p = new ArrayList<>(m);
        for (int i = 1; i <= m; i++) {
            p.add(i);
        }
        int[] ans = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int idx = p.indexOf(queries[i]);
            ans[i] = idx;
            p.remove(idx);
            p.add(0, queries[i]);
        }
        return ans;
    }
}
