import java.util.LinkedList;

class Solution {
    public int[] processQueries(int[] queries, int m) {

        int[] ans = new int[queries.length];

        LinkedList<Integer> list = new LinkedList<>();
        for (int i = 0; i < m; i++) {
            list.addLast(i + 1);
        }

        for (int i = 0; i < queries.length; i++) {
            int q = queries[i];
            int p = 0;
            while (!list.get(p).equals(q)) {
                p++;
            }

            ans[i] = p;

            list.remove(p);
            list.addFirst(q);
        }
        return ans;

    }
}
