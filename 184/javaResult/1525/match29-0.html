class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] ans = new int[queries.length];
        int[] p = new int[m];
        for (int i = 0; i < m; i++) {
            p[i] = i + 1;
        }
        for (int i = 0; i < queries.length; i++) {
            ans[i] = peek(p, queries[i]);
        }
        return ans;
    }
    private int peek(int[] p, int k) {
        int count = -1;
        for (int i = 0; i < p.length; i++) {
            if (p[i] == k) {
                count = i;
                break;
            }
        }
        for (int i = count; i >= 1; i--) {
            p[i] = p[i - 1];
        }
        p[0] = k;
        return count;
    }
}
