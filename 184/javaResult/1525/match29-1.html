class Solution {
    public int[] processQueries(int[] queries, int m) {
        int len = queries.length;
        
int[] res = new int[len];
        int[] arr = new int[m];
        for (int i = 0; i < m; i++) {
            arr[i] = i+1;
        }
        for (int i = 0; i < len; i++) {
            int q = queries[i];
            
res[i] = findAndMove(arr, q);
        }
        return res;
    }
    private int findAndMove(int[] arr, int query) {
        int len = arr.length;
        
int pos = -1;
        for (int i = 0; i < len; i++) {
            if (arr[i] == query) {
                pos = i;
                break;
            }
        }
        for (int i = pos; i >= 1; i--) {
            arr[i] = arr[i-1];
        }
        arr[0] = query;
        return pos;
    }
}
