class Solution {
    public int[] processQueries(int[] queries, int m) {
        ArrayList<Integer> P = new ArrayList<>(m);
        for (int i = 0; i < m; i++) {
            P.add(i);
        }
        int[] ans = new int[queries.length];
        int k = 0;
        
for (int q : queries) {
            q--;
            int index = findValue(P, q);
            ans[k++] = index;
            P.remove(index);
            P.add(0, q);
        }
        return ans;
    }

    private int findValue(ArrayList<Integer> arr, int value) {
        for (int i = 0; i < arr.size(); i++)
            if (arr.get(i) == value)
                return i;
        return -1;
    }
}
