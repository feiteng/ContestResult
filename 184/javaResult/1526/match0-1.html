class Solution {
    public String entityParser(String text) {
        String res0 = text.replaceAll("&quot;", "\"");
        String res1 = res0.replaceAll("&apos;", "\'");
        String res2 = res1.replaceAll("&gt;", ">");
        String res3 = res2.replaceAll("&lt;", "<");
        String res4 = res3.replaceAll("&frasl;", "/");
        String res5 = res4.replaceAll("&amp;", "&");
        return res5;
    }
}
