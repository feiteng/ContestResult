class Solution {
    public String entityParser(String t) {
        t = t.replaceAll("&quot;", "\"");
        t = t.replaceAll("&apos;", "'");
        t = t.replaceAll("&amp;", "&");
        t = t.replaceAll("&gt;", ">");
        t = t.replaceAll("&lt;", "<");
        t = t.replaceAll("&frasl;", "/");
        return t;
    }
}
