class Solution {
    public String entityParser(String text) {
        String text1 = text.replace("&quot;", "\"");
        String text2 = text1.replace("&apos;", "\'");
        String text3 = text2.replace("&amp;", "&");
        String text4 = text3.replace("&gt;", ">");
        String text5 = text4.replace("&lt;", "<");
        String text6 = text5.replace("&frasl;", "/");
        return text6;
    }
}
