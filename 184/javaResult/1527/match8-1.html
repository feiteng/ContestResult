class Solution {
    public int numOfWays(int n) {
        int mod = 1000000007;
        long two = 6, three = 6;
        for (int i = 2; i <= n; i++) {
            long newTwo = (two * 3 + three * 2) % mod;
            long newThree = (two * 2 + three * 2) % mod;
            two = newTwo;
            three = newThree;
        }
        return (int)((two + three) % mod);
    }
}
