class Solution {
    public int numOfWays(int n) {
        if (n == 1) return 12;
        long same = 6;
        long notSame = 6;
        long res = 0;
        for (int i=0; i<n-1; ++i) {
            long nextSame = same * 3 + notSame * 2;
            long nextNotSame = same * 2 + notSame * 2;
            same = nextSame % 1000000007;
            notSame = nextNotSame % 1000000007;
            res = 0;
            res += same;
            res += notSame;
            res = res % 1000000007;
        }
        return (int) res;
    }
}
