class Solution {
    public int numOfWays(int n) {
        long a = 6;
        long b = 6;
        int mod = 1000000007;
        while(--n > 0){
            long c = 3*a+2*b;
            long d = 2*a+2*b;
            a = c%mod;
            b = d%mod;
        }
        return (int) (a+b)%mod;
    }
}
