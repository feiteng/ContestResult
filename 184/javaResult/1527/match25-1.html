class Solution {
    public int gm(int c1, int c2, int c3) {
        return (c1 << 4) | (c2 << 2) | c3;
    }
    public boolean check(int c1, int c2, int c3, int d1, int d2, int d3) {
        return (c1 != c2 && c2 != c3 && d1 != d2 && d2 != d3 && c1 != d1 && c2 != d2 && c3 != d3);
    }
    
public int numOfWays(int n) {
        int MV = 1000000007;
        int[][] f = new int[2][43];
        for (int i = 0; i <= 2; ++i) {
            for (int j = 0; j <= 2; ++j) {
                for (int k = 0; k <= 2; ++k) {
                    if (i != j && j != k)
                        f[0][gm(i, j, k)] = 1;
                
}
            }
        }
        for (int i = 1; i < n; ++i) {
            for (int c1 = 0; c1 <= 2; ++c1) {
                for (int c2 = 0; c2 <= 2; ++c2) {
                    for (int c3 = 0; c3 <= 2; ++c3) {
                        int mc = gm(c1, c2, c3);
                        f[i % 2][mc] = 0;
                        
for (int d1 = 0; d1 <= 2; ++d1) {
                            for (int d2 = 0; d2 <= 2; ++d2) {
                                for (int d3 = 0; d3 <= 2; ++d3) {
                                    if (check(c1, c2, c3, d1, d2, d3)) {
                                        f[i % 2][mc] += f[(i + 1) % 2][gm(d1, d2, d3)];
                                        if (f[i % 2][mc] >= MV) f[i % 2][mc] -= MV;
                                    
}
                                }
                            }
                        }
                    }
                }
            }
        }
        int s = 0;
        for (int c1 = 0; c1 <= 2; ++c1) {
            for (int c2 = 0; c2 <= 2; ++c2) {
                for (int c3 = 0; c3 <= 2; ++c3) {
                    int mc = gm(c1, c2, c3);
                    s += f[(n - 1) % 2][mc];
                    if (s > MV) s -= MV;
                
}
            }
        }
        return s;
    }
}
