class Solution {
    public int numOfWays(int n) {
            final long MOD = 1000000007L;
    // dp2[m] is the way to paint where the top rows is a fixed A B A style
    // (One of the 6 such possibilities ABA ACA BAB BCB CAC CBC)
    long[] dp2 = new long[n+1];
    // dp3[m] is the way to paint where the top rows is a fixed A B C style
    // One of the 6 ways ABC ACB BAC BCA CAB CBA
    long[] dp3 = new long[n+1];

    dp2[1] = 1;
    dp3[1] = 1;

    for (int m = 2; m <= n; m++) {
      dp2[m] = (3L * dp2[m-1] + 2 * dp3[m-1]) % MOD;
      dp3[m] = (2L * dp2[m-1] + 2 * dp3[m-1]) % MOD;
    }
    int output = (int) ((6L * (dp2[n] + dp3[n])) % MOD);
    return output;
    }
}
