class Solution {
    public int numOfWays(int n) {
        long color2 = 6;
        long color3 = 6;
        
        int max = (int) (1e9 + 7);
        
        for(int i = 2; i <= n; i++)
        {
            long temp = color3; 
            color3 = (2 * color3 + 2 * color2 ) % max; 
            color2 = (2 * temp + 3 * color2 ) % max;
        }
        
        return (int) ((color3 + color2) % max);
    }
}
