class Solution {
    public int numOfWays(int n) {
        
        long color2 = 6;
        long color3 = 6;
        long MOD = (int) (1e9 + 7);
        
        for (int i = 2; i <= n; ++i) {
            long temp = color3;
            color3 = (2 * color3 + 2 * color2) % MOD;
            color2 = (2 * temp + 3 * color2) % MOD;
        }
        
        long res = (color2 + color3) % MOD;
        return (int) res;
    }
}
