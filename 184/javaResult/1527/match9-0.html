class Solution {
    public int numOfWays(int n) {
        // aba   abc
        long aba = 6;
        long abc = 6;
        int mod = 1000000007;
        for (int i = 1; i < n; i++) {
            long x = aba * 3 + abc * 2;
            long y = aba * 2 + abc * 2;
            aba = x % mod;
            abc = y % mod;
        }
        return (int)((aba + abc) % mod);
    }
}
