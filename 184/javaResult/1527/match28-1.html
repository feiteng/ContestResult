class Solution {
    private int mod = 1000000007;
    
public int numOfWays(int n) {
        if(n == 1)
            return 12;
        int f = 6;
        int g = 6;
        int f_new, g_new;
        for(int i = 1; i < n; i++) {
            // iterate
            // f(n) = 3*f(n-1) + 2*g(n-1)
            // g(n) = 2*f(n-1) + 2*g(n-1)
            // f: ABA type
            // g: ABC type
            g_new = (f + g) % mod;
            
g_new = (g_new + g_new) % mod;
            f_new = (f + g_new) % mod;
            f = f_new;
            g = g_new;
        }
        return (f + g) % mod;
    }
}
