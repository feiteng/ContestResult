class Solution {
    private static final int MOD = 1000000007;
    public int numOfWays(int n) {
        long color3 = 6;  
        long color2 = 6; 
        long temp = 0; 
        for (int i = 2; i <= n; i++)         
        { 
            long tmp = color3; 
            color3 = (2*color3 + 2*color2) % MOD; 
            color2 = (2*tmp + 3*color2) % MOD; 
        } 
        long num = (color3 + color2) % MOD; 
        return (int)num;
    }
    
    private int pow(int x, int n)
    {
        long res = 1;
        while(n > 0)
        {
            if (n%2==1)
            {
                res *= x;
                res %= MOD;
            }
            x = (x*x)%MOD;
            n /= 2;
        }
        return (int) res%MOD;
    }
}
