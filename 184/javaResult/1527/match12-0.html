class Solution {
    public int numOfWays(int n) {
        long a=6,b=6;
        long mod=1000000007;
        for(int i=2;i<=n;i++){
            long na=(3*a+2*b)%mod;
            b=(2*b+2*a)%mod;
            a=na;
        }
        int ans=((int)(a+b))%(int)mod;
        return ans;
    }
}
