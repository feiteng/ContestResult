class Solution {
    public int numOfWays(int n) {
        final int MOD = 1000000007;
        int[][][][] dp = new int[n][3][3][3];
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                for (int k = 0; k < 3; k++) {
                    dp[0][i][j][k] = (i != j) && (j != k) ? 1 : 0;
                }
            }
        }
        for (int I = 1; I < n; I++) {
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    for (int k = 0; k < 3; k++) {
                        for (int i2 = 0; i2 < 3; i2++) {
                            for (int j2 = 0; j2 < 3; j2++) {
                                for (int k2 = 0; k2 < 3; k2++) {
                                    if (i == j || j == k) continue;
                                    if (i2 == j2 || j2 == k2) continue;
                                    if (i2 == i || j2 == j || k2 == k) continue;
                                    dp[I][i][j][k] += dp[I - 1][i2][j2][k2];
                                    dp[I][i][j][k] %= MOD;
                                }
                            }
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                for (int k = 0; k < 3; k++) {
                    ans = (ans + dp[n - 1][i][j][k]) % MOD;
                }
            }
        }
        return ans;
    }
}
