class Solution {
    public int numOfWays(int n) {
        long[][] f = new long[n+1][2];
        f[0][0]=6;
        f[0][1]=6;
        for(int i=1;i<=n;i++){
            f[i][0]=(2*f[i-1][0]+2*f[i-1][1])%(1000_000_000+7);
            f[i][1]=(2*f[i-1][0]+3*f[i-1][1])%(1000_000_000+7);
        }
        return (int)(f[n-1][0]+f[n-1][1])%(1000_000_000+7);
    }
}
