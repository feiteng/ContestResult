class Solution {
    public int numOfWays(int n) {
        long aba = 6;
        long abc = 6;
        for(int i = 2; i <= n; i++){
            long newABA = (aba*3+abc*2)%1000000007;
            long newABC = (aba*2+abc*2)%1000000007;
            aba = newABA;
            abc = newABC;
        }
        long res = (aba+abc)%1000000007;
        return (int)res;
    }
}
