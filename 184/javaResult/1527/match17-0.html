class Solution {
    public int numOfWays(int n) {
        int mod = (int)1e9 + 7;
        Long[] dp1 = new Long[n];
        Long[] dp2 = new Long[n];
        dp1[0] = (long)6;
        dp2[0] = (long)6;
        for (int i = 1; i < n; i++) {
            dp1[i] = (3 * dp1[i - 1] + 2 * dp2[i - 1]) % mod;
            dp2[i] = (2 * dp1[i - 1] + 2 * dp2[i - 1]) % mod;
        }
        return (int) (((dp1[n - 1] % mod) + (dp2[n - 1] % mod)) % mod);
    }
}
