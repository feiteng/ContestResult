class Solution {
    private static final int MOD = 1000000007;
    
    public int numOfWays(int n) {
        
int[][][][] dp = new int[n][3][3][3];
        for (int c1 = 0; c1 < 3; ++c1) {
            for (int c2 = 0; c2 < 3; ++c2) {
                for (int c3 = 0; c3 < 3; ++c3) {
                    if (c1 != c2 && c2 != c3) {
                        dp[0][c1][c2][c3] = 1;
                    }
                
}
            }
        }
        
        for (int r = 1; r < n; ++r) {
            for (int c1 = 0; c1 < 3; ++c1) {
                for (int c2 = 0; c2 < 3; ++c2) {
                    for (int c3 = 0; c3 < 3; ++c3) {
                        if (c1 == c2 || c2 == c3) {
                            continue;
                        }
                        
                        
for (int lc1 = 0; lc1 < 3; ++lc1) {
                            for (int lc2 = 0; lc2 < 3; ++lc2) {
                                for (int lc3 = 0; lc3 < 3; ++lc3) {
                                    if (lc1 == lc2 || lc2 == lc3) {
                                        continue;
                                    }
                                    
                                    if (lc1 == c1 || lc2 == c2 || lc3 == c3) {
                                        continue;
                                    
}
                                    
                                    dp[r][lc1][lc2][lc3] += dp[r - 1][c1][c2][c3];
                                    dp[r][lc1][lc2][lc3] %= MOD;
                                }
                            }
                        }
                    }
                }
            }
        }
        
        int total = 0;
        for (int c1 = 0; c1 < 3; ++c1) {
            for (int c2 = 0; c2 < 3; ++c2) {
                for (int c3 = 0; c3 < 3; ++c3) {
                    total += dp[n - 1][c1][c2][c3];
                    total %= MOD;
                
}
            }
        }
        return total;
    }
}
