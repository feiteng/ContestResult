class Solution {
    public int numOfWays(int n) {
        long k = 1;
        long m = 1;
        long mod = (long)1e9 + 7;
        for (int i = 0; i < n-1; i++) {
            long k1 = 3 * k + 2 * m;
            long m1 = 2 * (k + m);
            k = k1 % mod;
            m = m1 % mod;
        }
        return (int)(((k + m) * 6) % mod);
    }
}
