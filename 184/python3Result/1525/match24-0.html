class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        res = []
        P = list(range(1, m+1))
        for q in queries:
            q_index = P.index(q)
            res.append(q_index)
            P = [q]+P[:q_index]+P[q_index+1:]
        return res
