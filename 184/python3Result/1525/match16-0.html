class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        a = list(range(1, m + 1))
        ret = []
        for q in queries:
            i = a.index(q)
            ret.append(i)
            del a[i]
            a.insert(0, q)
        return ret
