class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        l = list(range(1, m+1))
        result = []
        for q in queries:
            idx = l.index(q)
            result.append(idx)
            l.pop(idx)
            l.insert(0, q)
        return result
