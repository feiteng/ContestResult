class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        l = list(range(1,m+1))
        ans = []
        for x in queries:
            i = l.index(x)
            l.pop(i)
            l.insert(0, x)
            ans.append(i)
        return ans
