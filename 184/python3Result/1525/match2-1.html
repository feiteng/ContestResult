class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        q = list(range(1, m + 1))
        res = []
        for x in queries:
            idx = q.index(x)
            res.append(idx)
            q.pop(idx)
            q = [x] + q
        return res
