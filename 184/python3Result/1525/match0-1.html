class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        P = [i + 1 for i in range(m)]
        res = []
        for q in queries:
            idx = P.index(q)
            res.append(idx)
            P.pop(idx)
            P = [q] + P
            
        return res
