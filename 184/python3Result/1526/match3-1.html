class Solution:
    def entityParser(self, text: str) -> str:
        e = {
            '&quot;': '"',
            '&apos;': "'",
            '&gt;': '>',
            '&lt;': '<',
            '&frasl;': '/',
            '&amp;': "&",
        }
        for k, v in e.items():
            text = text.replace(k, v)
        return text
