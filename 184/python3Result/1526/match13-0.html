class Solution:
    def entityParser(self, text: str) -> str:
        encoding = {"&quot;": "\"", 
                   "&apos;": "'",
                   "&amp;": "&",
                   "&gt;": ">",
                   "&lt;": "<",
                   "&frasl;": "/"}
        
        for en, de in encoding.items():
            text = text.replace(en, de)
        return text
