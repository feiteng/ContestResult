class Solution:
    def entityParser(self, text: str) -> str:
        check = {
            "&quot;": '"',
            "&apos;": "'",
            "&amp;": "&",
            "&gt;": ">",
            "&lt;": "<",
            "&frasl;": "/"
        }
        
        for key, val in check.items():
            text = text.replace(key, val)
        
        return text
