class Solution:
    def entityParser(self, text: str) -> str:
        entities = {
            "&quot;": "\"",
            "&apos;": "'",
            "&amp;": "&",
            "&gt;": ">",
            "&lt;": "<",
            "&frasl;": "/"
        }

        for e, x in entities.items():
            text = text.replace(e, x)

        return text
