class Solution:
    def entityParser(self, text: str) -> str:
        mapped = {
            '&quot;':'"',
            "&apos;": "'",
            "&amp;":'&',
            '&gt;':'>',
            '&lt;': '<',
            '&frasl;':'/'
        }
        for f, t in mapped.items():
            text = text.replace(f, t)
        return text
