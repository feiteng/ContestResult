class Solution:
    def entityParser(self, text: str) -> str:
        DIC = {
            "&quot;"    :'"',
            "&apos;"    :"'",
            "&amp;"     :'&',
            "&gt;"      :'>',
            "&lt;"      :'<',
            "&frasl;"   : r"/",
        }
        for key, val in DIC.items():
            text = text.replace(key,val)
        return text
