class Solution:
    def numOfWays(self, n: int) -> int:
        a = 6
        b = 6
        mod = 10**9+7
        for i in range(n-1):
            new_a = (a + a + a + b + b) % mod
            new_b = (a + a + b + b) % mod
            a = new_a
            b = new_b
        return (a+b) % mod
