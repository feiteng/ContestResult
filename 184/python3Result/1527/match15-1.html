class Solution:
    def numOfWays(self, n: int) -> int:
        color3 = 6
        color2 = 6
        temp = 0

        for i in range(2, n + 1, 1): 
            temp = color3 
            color3 = (2 * color3 + 2 * color2 ) % 1000000007

            color2 = ( 2 * temp + 3 * color2 ) % 1000000007

        num = (color3 + color2) % 1000000007

        return num
