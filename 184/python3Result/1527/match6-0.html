class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9 + 7
        
        
        n2, n3 = 6, 6
        nn2, nn3 = 0, 0
        for _ in range(n-1):
            nn2 = 3*n2 + 2*n3
            nn3 = 2*n2 + 2*n3
            n2 = nn2
            n3 = nn3
        return (n2 + n3) % MOD
