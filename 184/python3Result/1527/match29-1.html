class Solution:
    def numOfWays(self, n: int) -> int:
        mod = 1000000007
        ans1 = [0] * 5001
        ans2 = [0] * 5001
        ans1[1] = ans2[1] = 6
        for i in range(2, 5001):
            ans1[i] = (2 * ans1[i - 1] + 2 * ans2[i - 1]) % mod
            ans2[i] = (2 * ans1[i - 1] + 3 * ans2[i - 1]) % mod
        return (ans1[n] + ans2[n]) % mod
