class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9 + 7
        dp1 = [0] * (n+1)
        dp2 = [0] * (n+1)
        dp1[1] = 6
        dp2[1] = 6
        for i in range(2, n+1):
            dp1[i] = (dp1[i-1] * 2 + dp2[i-1] * 2) % MOD
            dp2[i] = (dp1[i-1] * 2 + dp2[i-1] * 3) % MOD
        return (dp1[n] + dp2[n]) % MOD
