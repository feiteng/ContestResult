class Solution:
    def numOfWays(self, n: int) -> int:
        n1 = 6
        n2 = 6
        M = 1000000007
        for i in range(n-1):
            t1 = ((3*n1) % M + (2 * n2) % M) % M
            t2 = ((2 * n1) % M + (2 * n2) % M) % M
            n1, n2 = t1, t2
        return (n1+n2)%M
