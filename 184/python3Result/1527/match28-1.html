class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = ((10**9) + 7)
        three = 6
        two = 6
        for i in range(1, n):
            n_three = 2 * three + 2 * two
            n_two = 2 * three + 3 * two
            three, two = n_three, n_two
        return (three + two) % MOD
