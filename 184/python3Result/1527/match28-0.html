class Solution:

    def numOfWays(self, n: int) -> int:
        mod = 10**9 + 7
        uniq = same = 6
        for _ in range(n-1):
            u, s = uniq, same
            uniq = (u + s) % mod * 2 % mod
            same = (2 * u % mod + 3 * s % mod) % mod
        return (uniq + same) % mod
