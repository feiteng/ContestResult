class Solution:
    def numOfWays(self, n: int) -> int:
        mod = 10 ** 9 + 7
        num_sym = 6
        num_diff = 6
        for _ in range(n - 1):
            next_sym = (3 * num_sym + 2 * num_diff) % mod
            next_diff = (2 * num_sym + 2 * num_diff) % mod
            num_sym = next_sym
            num_diff = next_diff
        
        return (num_sym + num_diff) % mod
