class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9+7
        aba = 6
        abc = 6
        for i in range(2, n+1):
            naba = aba * 3 + abc * 2
            nabc = aba * 2 + abc * 2
            aba = naba % MOD
            abc = nabc % MOD
        return (aba + abc) % MOD
