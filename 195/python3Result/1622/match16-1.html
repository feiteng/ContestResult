class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        que = collections.deque()
        res = float('-inf')
        n = len(points)
        for i in range
(n):
            x, y = points[i]
            while que and x - que[0][0] > k:
                que.popleft()
            if que:
                res = max(res, que[0][1] + y + x - que[0][0])
            while que and x - que[-1][0] + que[-1][1] < y:
                que.pop()
            que.append((x, y))
        return res
