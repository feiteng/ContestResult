class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        # store decreasing queue
        q = deque() # store (val, x)
        ret = -float('inf')
        for x, y in points:
            val = y-x
            while q and q[0][1] < x-k:
                q.popleft()
            if q:
                ret = max(ret, q[0][0]+x+y)
            while q and q[-1][0] <= val:
                q.pop()
            q.append([val,x])
        return ret
