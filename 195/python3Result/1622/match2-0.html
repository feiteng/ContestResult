class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        
        dq1 = collections.deque()
        dq2 = collections.deque()
        dq1.append(0)
        dq2.append(0)
        res = -sys.maxsize
        
        for j in range(1, len(points)):
            xj, yj = points[j]
            while dq1 and xj - points[dq1[0]][0] > k:
                dq1.popleft()
            if dq1: 
                res = max(res, yj + points[dq1[0]][1] + xj - points[dq1[0]][0])
            while dq1 and yj - points[dq1[-1]][1] - (xj - points[dq1[-1]][0]) > 0:
                dq1.pop()
            dq1.append(j)
            
            while dq2 and xj - points[dq2[0]][0] > k:
                dq2.popleft()
            if dq2: 
                res = max(res, yj + points[dq2[0]][1] - xj + points[dq2[0]][0]) 
            while dq2 and yj - points[dq2[-1]][1] - (points[dq2[-1]][0] - xj) > 0:
                dq2.pop()
            dq2.append(j)
            
        return res
