class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        deque = []
        max_val = -float('inf')
        for x, y in points:
            while deque and x-deque[0][0] > k:
                deque.pop(0)
            deque.append([x, y])
            if len(deque) == 1: continue    
            
            a, b = deque[0]
            max_val = max(max_val, b+y+x-a)
            
            if deque[-1][0] == x and y > deque[-1][1]:
                deque[-1][1] = y

            if deque:
                newx, newy = deque[-1]
            newx, newy = deque.pop()
                
            
while deque and newy - deque[-1][1] > newx - deque[-1][0]:
                deque.pop()
            deque.append([newx, newy])
            # print([x, y],'-----', deque, max_val)
        return max_val
