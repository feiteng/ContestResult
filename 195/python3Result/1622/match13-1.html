class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        res = -sys.maxsize

        deque1 = collections.deque()
        deque2 = collections.deque()
        deque1.append(0)
        deque2.append(0)

        
for j in range(1, len(points)):
            xj, yj = points[j]
            while deque1 and xj - points[deque1[0]][0] > k:
                deque1.popleft()
            
while deque2 and xj - points[deque2[0]][0] > k:
                deque2.popleft()
            if deque1:
                res = max(res, yj + xj + points[deque1[0]][1] - points[deque1[0]][0])
            
if deque2:
                res = max(res, yj - xj + points[deque2[0]][1] + points[deque2[0]][0])
            while deque1 and yj - xj > points[deque1[-1]][1] - points[deque1[-1]][0]:
                deque1.pop()
            
while deque2 and yj + xj > points[deque2[-1]][1] + points[deque2[-1]][0]:
                deque2.pop()
            deque1.append(j)
            deque2.append(j)
        return res
