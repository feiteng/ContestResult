class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        stack = []
        ans = -10**9
        for a,b in points:
            
            # print(stack)
            # i = len(stack) - 1
            # while i >= 0 and a-stack[i][1] <= k:
            #     ans = max(ans, a+b+stack[i][0])
            #     i -= 1
            # 二分优化
            if stack and a-stack[-1][1] <= k: # 存在
                left = 0
                right = len(stack)-1
                while left <= right:
                    # print(left, right)
                    mid = left + right >> 1
                    if a-stack[mid][1] <= k:
                        right = mid - 1
                    else:
                        left = mid + 1
                ans 
= max(ans, a+b+stack[left][0])

            
            while stack and stack[-1][0] < b-a:
                stack.pop()
            stack.append([b-a, a])
        
        return ans
