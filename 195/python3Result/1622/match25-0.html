class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        deque = collections.deque()
        res = float('-inf')
        n = len(points)
        for i in range(n):
            cx = points[i][0]
            cy = points[i][1]
            while deque and cx - points[deque[0]][0] > k:
                deque.popleft()
            if deque:
                res = max(res, cx - points[deque[0]][0] + cy + points[deque[0]][1])
            while deque and (points[deque[-1]][1] - points[deque[-1]][0]) <= cy - cx:
                deque.pop()
            deque.append(i)
        return res
