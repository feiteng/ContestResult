class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        q = []
        sol = -float('inf')
        for i, (x, y) in enumerate(points):
            while q and abs(q[0][1] - x) > k:
                heapq.heappop(q)
            if q:
                yx2, x2, i2 = q[0]
                sol = max(sol, (-yx2)+y+x)
            heapq.heappush(q, (x-y, x, i))

        return sol
