class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        stack = deque()
        ans 
= float('-inf')
        for i,j in points:
            while stack and (i - stack[0][0] > k):
                stack.popleft()
            if stack:
                ans = max(i+j+stack[0][1], ans)
            while stack and (stack
[-1][1] < j-i):
                stack.pop()
            stack.append([i, j-i])
            # print(i,j,ans, stack)
        return ans
