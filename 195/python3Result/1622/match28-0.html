class Solution:
    def findMaxValueOfEquation(self, p: List[List[int]], k: int) -> int:
        stack=[]
        ans=-inf
        for x,y in p:
            ind=bisect.bisect(stack,(x-k,-inf))
            if ind!=len(stack):
                ans=max(ans,stack[ind][1]+x+y)
            while stack and stack[-1][1]<=y-x:
                stack.pop()
            stack.append((x,y-x))
        return ans
