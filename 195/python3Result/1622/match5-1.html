class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        q = collections.deque()
        res = -float('inf')
        
        for i in range(len(points)):
            x, y = points[i]
            
while q and x - q[0][0] > k:
                q.popleft()
            if q:
                res = max(res, y + q[0][1] + x - q[0][0])

            while q and y - x > (q[-1][1] - q[-1][0]):
                q.pop()
            
            q.append((x, y))
        
        return res
