from collections import deque
class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        monoq=deque() # [(i, x, y)]
        ans=float('-inf')
        for i,(x,y) in enumerate(points):
            
while monoq and monoq[0][1]<x-k: 
                monoq.popleft()
            if monoq:
                ans=max(ans, y + monoq[0][2] + abs(monoq[0][1] - x))
            while monoq and monoq[-1][2] + x - monoq[-1][1]  <= y:
                monoq.pop()
            monoq.append((i, x, y))
        return ans
