class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        res = float('-inf')
        stack = deque()

        
for i in range(len(points)):
            x, y = points[i]
            while stack and points[stack[0]][0] + k < x:
                stack.popleft()

            if stack:
                xx, yy = points[stack[0]]
                res = max(
res, x + y + yy - xx)

            while stack and points[stack[-1]][1] - points[stack[-1]][0] <= y - x:
                stack.pop()
            stack.append(i)

        return res
