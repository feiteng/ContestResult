from collections import deque

class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        dec = deque([points[0]])
        ans = -math.inf
        for i in range(1, len(points)):
            x, y = points[i]
            while dec and x - dec[0][0] > k:
                dec.popleft()
            if dec:
                ans = max(ans, x - dec[0][0] + y + dec[0][1])
            while dec and dec[-1][1] - dec[-1][0] < y - x:
                dec.pop()
            dec.append((x, y))
        return ans
