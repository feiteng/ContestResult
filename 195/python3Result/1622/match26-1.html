import heapq
class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        res = -math.inf
        q = []
        for x, y in points:
            while q and x - q
[0][1] > k:
                heapq.heappop(q)
            if q:
                _, x_, y_ = q[0]
                res = max(res, y + y_ + x - x_)
            heapq.heappush(q, (x - y, x, y))
        return res
