class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        que = collections.deque()
        res = -99999999999999
        for x, y in reversed
(points):
            while que and que[0][0] - x > k:
                que.popleft()
            if que:
                res = max(res, que[0][1] + y - x)
            while que and x + y > que
[-1][1]:
                que.pop()
            que.append((x, x + y))
        return res
