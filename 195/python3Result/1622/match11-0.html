class Solution:
    def findMaxValueOfEquation(self, points: List[List[int]], k: int) -> int:
        stack = []
        res = -float('inf')
        for x,y in points:
            while stack and x-stack[0][0]>k:
                stack.pop(0)
            if stack:
                res = max(res, y+stack[0][1]+abs(x-stack[0][0]))
            while stack and y-x>=stack[-1][1]-stack[-1][0]:
                stack.pop()
            stack.append([x,y])
        return res
