class Solution:
    def isPathCrossing(self, path: str) -> bool:
        visited = set()
        visited.add((0,0))
        
        cur_x = 0
        cur_y = 0
        
        for c in path:
            if c == 'N':
                cur_y += 1
            elif c == 'S':
                cur_y -= 1
            elif c == "E":
                cur_x += 1
            else:
                cur_x -= 1
            
            if (cur_x, cur_y) in visited:
                return True
            visited.add((cur_x, cur_y))
            
        return False
