class Solution:
    def isPathCrossing(self, path: str) -> bool:
        direction = {"N":(0, 1), "S": (0, -1), "E": (1, 0), "W": (-1, 0)}
        
        visited = set()
        visited.add((0, 0))
        x, y = 0, 0
        for c in path:
            di, dj = direction[c]
            x += di
            y += dj
            if (x, y) in visited:
                return True
            visited.add((x, y))
        return False
