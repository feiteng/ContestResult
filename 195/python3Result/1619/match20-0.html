class Solution:
    def isPathCrossing(self, path: str) -> bool:
        x,y=0,0
        s=set()
        s.add((x,y))
        for i in path:
            if i=='N':
                y+=1
            elif i=='S':
                y-=1
            elif i=='E':
                x+=1
            else:
                x-=1
            #print(x,y,s)
            if (x,y) in s:
                return True
            s.add((x,y))
            
        return False
