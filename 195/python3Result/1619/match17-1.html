class Solution:
    def isPathCrossing(self, path: str) -> bool:
        dd = {
            'N': (1, 0),
            'S': (-1,0),
            'E': (0,1),
            'W': (0,-1),
        }
        v = set()
        v.add((0,0))
        x, y = 0,0
        for d in path:
            dx, dy = dd[d]
            x += dx
            y += dy
            if (x,y) in v: return True
            v.add((x,y))
            
        return False
