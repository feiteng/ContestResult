class Solution:
    def isPathCrossing(self, path: str) -> bool:
        s = set()
        s.add((0,0))
        curr_x = 0
        curr_y = 0
        for c in path:
            if c == 'N':
                curr_y += 1
            elif c == 'S':
                curr_y -= 1
            elif c == 'E':
                curr_x += 1
            elif c == 'W':
                curr_x -= 1
            if (curr_x, curr_y) in s:
                return True
            s.add((curr_x, curr_y))
        return False
