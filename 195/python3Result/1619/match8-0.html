class Solution:
    def isPathCrossing(self, path: str) -> bool:
        visited = set()
        currentX = currentY = 0##this is how you declare two variables to be the same initially with equal signs, not comma
        visited.add((currentX,currentY))
        for direction in path:
            if direction == 'N':
                currentY += 1
            elif direction == 'S':
                currentY -= 1
            elif direction == 'E':
                currentX += 1
            else:
                currentX -= 1
            if (currentX,currentY) in visited:
                return True
            else:
                visited.add((currentX,currentY))
        return False
