class Solution:
    def isPathCrossing(self, path: str) -> bool:
        save = set()
        save.add((0,0))
        x ,y = 0, 0
        for di in path:
            if di == 'N':
                y += 1
            elif di == 'S':
                y -= 1
            elif di == 'E':
                x += 1
            else:
                x -= 1
            if (x,y) in save:
                return True
            save.add((x,y))
        return False
