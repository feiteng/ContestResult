class Solution:
    def isPathCrossing(self, path: str) -> bool:
        visited = set()
        visited.add((0, 0))
        x, y = 0, 0
        for direction in path:
            if direction == 'N':
                x -= 1
            elif direction == 'S':
                x += 1
            elif direction == 'E':
                y += 1
            else:
                y -= 1
            if (x, y) in visited:
                return True
            visited.add((x, y))
        return False
