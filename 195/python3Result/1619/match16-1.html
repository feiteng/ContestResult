class Solution:
    def isPathCrossing(self, path: str) -> bool:
        deja = {(0, 0)}
        r, l = 0, 0
        for i in path:
            if i == "N":
                r += 1
            elif i == "S":
                r -= 1
            elif i == "E":
                l += 1
            else:
                l -= 1
            if (r, l) in deja:
                return True
            deja.add((r,l))
        return False
