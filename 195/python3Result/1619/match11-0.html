class Solution:
    def isPathCrossing(self, path: str) -> bool:
        
        visited = set()
        visited.add((0,0))
        
        currX, currY = 0,0
        
        for char in path:
            if char == "N":
                currY += 1
            elif char == "S":
                currY -= 1
            elif char == "E":
                currX += 1
            else:
                currX -= 1
            
            if (currX, currY) in visited:
                return True
            visited.add((currX, currY))
        return False
