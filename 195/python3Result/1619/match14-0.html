class Solution:
    def isPathCrossing(self, path: str) -> bool:
        pos = set()
        pos.add((0, 0))
        
        i, j = 0, 0
        for p in path:
            if p == 'N':
                i += 1
            elif p == 'S':
                i -= 1
            elif p == 'E':
                j += 1
            else:
                j -= 1
            if (i, j) in pos:
                return True
            pos.add((i, j))
        
        return False
