class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        rem_cnts = [0] * k
        for val in arr:
            rem_cnts[val % k] += 1
        
        if rem_cnts[0] % 2 != 0:
            return False
        
        for p in range(1, k):
            if rem_cnts[p] != rem_cnts[k - p]:
                return False
        
        return True
