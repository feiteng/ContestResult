from collections import defaultdict

class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        
        cnter = defaultdict(int)
        for num in arr:
            cnter[num % k] += 1
        if cnter[0] % 2 == 1: return False
        for i in range(1, k):
            if cnter[i] != cnter[k-i]: return False
        return True
