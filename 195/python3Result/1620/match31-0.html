class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        dic = defaultdict(int)
        for a in arr:
            dic[a % k] += 1
        # print(dic)
        
        if dic[0] % 2 != 0:
            return False
        for i in range(1, k):
            if dic[i] != dic[k-i]:
                return False     
        return True
