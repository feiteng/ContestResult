from collections import Counter

class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        counter = Counter()
        for n in arr:
            counter[(n % k)] += 1
        
        if counter[0] % 2 == 1:
            return False
        for i in range(1, k):
            if counter[i] != counter[(k-i)]:
                return False
        return True
