class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        k_list = [0 for i in range(k)]
        for i in arr:
            k_list[i % k] += 1

        # print(k_list)
        if k_list[0] % 2 != 0:
            return False

        for i in range(1, k // 2):
            if k_list[i] != k_list[k - i]:
                return False

        return True
