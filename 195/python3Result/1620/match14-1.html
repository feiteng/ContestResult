class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        memo = [0] * k
        for i in arr:
            memo[i % k] += 1
        if memo[0] % 2: return False
        for i in range(1, k):
            if memo[i] != memo[k - i]:
                return False
        return True
