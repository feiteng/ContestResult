class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        rem = [0] * k
        for num in arr:
            rem[num%k] += 1
        if rem[0] % 2 != 0:
            return False
        for i in range(1, k // 2):
            if rem[i] != rem[k-i]:
                return False
        return True
