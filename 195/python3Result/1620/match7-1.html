class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        modValues = defaultdict(int)
        for number in arr:
            modValues[number%k] += 1
        if modValues[0]%2 != 0:
            return False

        for i in range(1,(k+1)//2):##k even go from 1 to (k-2)//2. k odd go from 1 to (k-1)//2
            if modValues[i] != modValues[k-i]:
                return False
        if k%2 == 0:
            if modValues[k//2]%2 != 0:
                return False
        return True
