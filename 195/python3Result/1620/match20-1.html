class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        cand = [0 for _ in range(k)]
        for num in arr:
            cand[num%k] += 1
        if cand[0]%2 == 1:
            return False
        for i in range(1,k//2):
            if cand[i] != cand[-i]:
                return False
        if k%2 == 0:
            if cand[k//2]%2 == 1:
                return False
        return True
