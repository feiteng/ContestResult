class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        table = defaultdict(int)
        for n in arr:
            table[n % k] += 1
        
        for key in table:
            if key == 0:
                continue

            if key == k / 2 and table[key] % 2 != 0:
                return False

            if table[k - key] != table[key]:
                return False
        
        return True
