class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        save = defaultdict(int)
        for num in arr:
            save[num % k] += 1
        if save[0] % 2 != 0:
            return False
        for i in range(1,k//2):
            if save[i] != save[k-i]:
                return False
        if k % 2 == 0:
            if save[k//2]%2 != 0:
                return False
        return True
