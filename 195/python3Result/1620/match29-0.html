class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        num = [0] * k
        for x in arr:
            num[x % k] += 1
        if num[0] % 2 != 0:
            return False
        for i in range(1, k):
            if num[i] != num[k - i]:
                return False
        return True
