class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        d = [0]*k
        for x in arr:
            d[x%k] += 1
        if d[0]%2 != 0:
            return False
        for i in range(1, k):
            if (d[i] + d[k-i]) % 2 != 0:
                return False
        return True
