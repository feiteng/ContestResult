class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        mds = [0]*k
        for a in arr:
            mds[a%k] += 1
        # print(mds)
        if mds[0]%2 > 0: return False
        for i in range(1, (k//2)+1):
            # print(i, k-i)
            if mds[i] != mds[k-i]:
                return False
        return True
