class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        count = [0 for _ in range(k)]
        for a in arr:
            count[a % k] += 1
        # print(count)
        if count[0] % 2:
            return False
        for i in range(1, k//2):
            # print(i, k-i-1)
            if count[i] != count[k - i]:
                return False
        return True
