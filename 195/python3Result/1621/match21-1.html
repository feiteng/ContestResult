class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()

        i, j = 0, len(nums) - 1
        res = 0
        while i <= j:
            if nums[i] + nums[j] > target: j -= 1
            else:
                # for all subsets include j not i, since i is a must include
                res += (1 << (j - i))
                i += 1

        return res % (10**9 + 7)
