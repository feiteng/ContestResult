class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        left, right, res = 0, n-1, 0
        while left <= right:
            if nums[left] + nums[right]>target:
                right -= 1
            else:
                k = right - left
                res += pow(2, k)
                left += 1
        return res % (pow(10, 9) + 7)
