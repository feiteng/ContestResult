import bisect

class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        res = 0
        for start in range(len(nums)):
            if nums[start] > target:
                break
            end = bisect.bisect_right(nums, target-nums[start], lo=start+1)
            if end-start == 1:
                res += 1 if nums[start] * 2 <= target else 0
            else:
                res += 2 ** (end-start-1)
        return res % (10**9+7)
