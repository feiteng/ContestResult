class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        MOD = 10**9 + 7
        nums.sort()
        i = 0
        j 
= len(nums) - 1
        ret = 0
        
        while i <= j:
            if nums[i] + nums[j] <= target:
                # add the number of subsequences that contain nums[i] up to nums[j]
                # can view it as a binary number where the lsb must be 1 -
                # so, we have 2^(j - i) possibilities
                ret = (ret + 2 ** (j - i)) % MOD
                
                # keep going, this time with a higher i
                i += 1
            else:
                # can't be certain that we can make valid subsequences from i to j,
                # so decrease j and try again.
                j -= 1
            
        return ret
