class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        i,j = 0,len(nums) - 1
        res = 0
        while i <= j:
            if nums[i] + nums[j] <= target:
                res += 2**(j-i)
                i += 1
            else:
                j -= 1
        return res%(10**9+7)
