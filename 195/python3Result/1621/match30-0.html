class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        res = 0
        for i, num in enumerate(nums):
            if num > target - num:
                break
            j = bisect.bisect(nums, target - num)  # nums[i:j] now satisfies the requirement; must include nums[i]
            res += 2 ** (j - i - 1)
        return res % (10 ** 9 + 7)
