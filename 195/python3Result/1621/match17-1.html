class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        head, tail 
= 0, len(nums)-1
        res = 0
        while head <= tail:
            if nums[head] + nums[tail] <= target:
                res += 1 << (tail - head)
                head += 1
            else:
                tail -= 1
        return res % (10**9 + 7)
