class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        ret=0
        for i,v in enumerate(nums):
            if v*2 > target: break
            r = bisect.bisect_right(nums, target-v, i)
            if r-i == 0: continue
            ret = (ret+pow(2, r-i-1,1000000007))%1000000007
        return ret
