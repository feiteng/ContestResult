class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        MOD = 10**9 + 7
        
        N = len(nums)
        total = (1 << N) - 1
        
        for n in nums:
            if n + n > target:
                total -= 1
        
        nums.sort()
        L = 0
        R = len(nums) - 1
        while L < R:
            if nums[R] + nums[L] > target:
                total -= (1 << (R-L)) - 1
                R -= 1
            else:
                L += 1
        
        return total % MOD
