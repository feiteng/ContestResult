class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        nums.sort()
        if nums[0] * 2 > target : return 0 
        l, r = 0, len(nums) - 1
        ans = 0
        while l <= r :
            if nums[l] + nums[r] <= target :
                ans += 2 ** (r - l + 1 - 1) 
                l += 1
            else :
                r -= 1
                
        return (ans) % (10**9 + 7)
