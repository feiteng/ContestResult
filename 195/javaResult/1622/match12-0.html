class Solution {

    public int findMaxValueOfEquation(int[][] points, int k) {
    	TreeMap<Integer , Integer> map = new TreeMap<>();
    	int i , j = 0 , n = points.length , ans = Integer.MIN_VALUE;
    	for (i = 0;i < n;i ++) {
    		int result = points[i][0] + points[i][1];
    		while (j < i && points[i][0] - points[j][0] > k) {
    			int value = points[j][1] - points[j][0];
    			map.put(value , map.get(value) - 1);
    			if (map.get(value) == 0) {
    				map.remove(value);
    			}
    			j ++;
    		}
    		if (!map.isEmpty()) {
    			result += (map.lastKey());
    			ans = Math.max(ans , result);
    		}
    		int value = points[i][1] - points[i][0];
    		map.put(value , map.getOrDefault(value , 0) + 1);
    	}    	
    	return ans;
    }
}
