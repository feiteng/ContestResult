class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
int out = Integer.MIN_VALUE;
        LinkedList<Integer> stack = new LinkedList<>();
        
for (int i = 0; i < points.length; ++i) {
            while (!stack.isEmpty() && points[i][0] - points[stack.peek()][0] > k)
                stack.pollFirst();
            if (stack.isEmpty()) { stack.addLast(i); continue; }
            int f = stack.peek();
            out = Math.max(out, points[i][0] - points[f][0] + points[i][1] + points[f][1]);
            
while (!stack.isEmpty() && points[i][1] > points[i][0] - points[stack.peekLast()][0] + points[stack.peekLast()][1]) 
                stack.pollLast();
            stack.addLast(i);
        }
        return out;
    }
}
