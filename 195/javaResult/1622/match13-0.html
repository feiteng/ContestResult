class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        LinkedList<Integer> d = new LinkedList<>();
        d.add(0);
        int max = Integer.MIN_VALUE;
        for (int i = 1; i < points.length; i++){
            int x = points[i][0], y = points[i][1];
            while((!d.isEmpty()) && ((x - points[d.peekLast()][0]) > k)) d.pollLast();
            if (!d.isEmpty()){
                int j = d.peekLast();
                max = Math.max(max, x + y + points[j][1] - points[j][0]);
            } 
            int diff = y - x;
            while (!d.isEmpty() && diff >= points[d.peekFirst()][1] - points[d.peekFirst()][0]) d.pollFirst();
            d.addFirst(i);
        }
        return max;
    }
}
