class Solution {
  public int findMaxValueOfEquation(int[][] points, int k) {
        int n = points.length;

        TreeSet<int[]> set = new TreeSet<>((a, b) -> {
            if (a[0] != b[0]) return Integer.compare(a[0], b[0]);
            else return Integer.compare(a[1], b[1]);
        });

        int ans = Integer.MIN_VALUE;
        for (int i = 0, j = 0; i < n; i++) {
            int x = points[i][0];
            int y = points[i][1];
            while (j < n && points[j][0] - x <= k) {
                set.add(new int[] {points[j][1] + points[j][0], j});
                j++;
            }

            while (!set.isEmpty() && set.last()[1] <= i) {
                set.pollLast();
            }

            if (!set.isEmpty()) 
{
                int index = set.last()[1];
                ans = Math.max(ans, points[index][1] + y + points[index][0] - x);
            }
        }
        return ans;
    }
}
