class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int ans = Integer.MIN_VALUE;
        int i = 0;
        int x, y;
        int cur;
        for (int j = 0; j < points.length; j++) {
            while (i < j && points[j][0] - points[i][0] > k) {
                x = points[i][0];
                y = points[i][1];
                
if (map.get(y - x) <= 1) {
                    map.remove(y - x);
                } else {
                    map.put(y - x, map.get(y - x) - 1);
                }
                i += 1;
            }
            x = points[j][0];
            y = points[j][1];
            if (map.size() > 0) {
                cur = map.lastKey() + x + y;
                ans = Math.max(cur, ans);
            }
            
if (map.containsKey(y - x)) {
                map.put(y - x, map.get(y - x) + 1);
            } else {
                map.put(y - x, 1);
            }
        }
        return ans;

    }
}
