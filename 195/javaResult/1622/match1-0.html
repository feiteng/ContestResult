class Solution {
    public int findMaxValueOfEquation(int[][] p, int k) {
        int ans = -1999999999;
       int[] res = new int[p.length];
       for (int i = 0; i < p.length; i++) {
          res[i] = p[i][1] - p[i][0];
       }
       TreeMap<Integer, Integer> map = new TreeMap<>((x, y) -> (y - x));
 
      int pre = 0;
        //System.out.println(p[0][1] - p[0][0]);
      map.put(p[0][1] - p[0][0], 1);
      for (int i = 1; i < p.length; i++) {
          int curX = p[i][0], curY = p[i][1];
          while (pre < i && curX - p[pre][0] > k) {
              int preVal = p[pre][1] - p[pre][0];
              if (map.get(preVal) == 1) map.remove(preVal);
              else map.put(preVal, map.get(preVal) - 1);
              
              pre++;
          }

          if (map.size() != 0) {
              //System.out.println(map.firstKey());
              ans = Math.max(ans, curX + curY + map.firstKey());
          }
          int curKey = curY - curX;
          int preValue = map.containsKey(curKey) ? map.get(curKey) : 0;
          map.put(curKey, preValue + 1);
       } 
        return ans;
    }
}
