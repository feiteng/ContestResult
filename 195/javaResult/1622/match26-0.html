import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {

    public int findMaxValueOfEquation(int[][] points, int k) {
        int fu = Integer.MIN_VALUE;
        var deng = new ArrayDeque<Integer>();
        for (int i = 0, j = i + 1; i < points.length; i++) {
            while (j < points.length && points[j][0] - points[i][0] <= k) {
                while (!deng.isEmpty() && points[j][0] + points[j][1] >= points[deng.peekLast()][0] + points[deng.peekLast()][1])
                    deng.pollLast();
                deng.addLast(j);
                j++;
            }
            while (!deng.isEmpty() && deng.peekFirst() <= i)
                deng.pollFirst();
            if (!deng.isEmpty()) {
                int ii = deng.peekFirst();
                fu = Math.max(fu, points[i][1] - points[i][0] + points[ii][0] + points[ii][1]);
            }
        }
        return fu;
    }
}
