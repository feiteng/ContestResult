class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int[] p : points) {
            map.put(p[0], p[1]);
        }
        
        int n = points.length, res = Integer.MIN_VALUE;
        Deque<Integer> deque = new ArrayDeque<>();
        for (int[] p : points) {
            
int x = p[0], y = p[1];
            while (!deque.isEmpty() && x-deque.getFirst() > k) {
                deque.removeFirst();
            }
            if (!deque.isEmpty()) {
                res = Math.max(res, x-deque.getFirst()+map.get(deque.getFirst())+y);
            }
            while (!deque.isEmpty() && y-map.get(deque.getLast()) > x-deque.getLast()) {
                deque.removeLast();
            
}
            deque.add(x);
        }
        return res;
    }
}
