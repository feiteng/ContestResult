class Solution {
    public int findMaxValueOfEquation(int[][] A, int k) {
        int n = A.length;
        
        // x - x0 + y
        TreeMap<Integer, Integer> sum = new TreeMap<>();
        
        int res = Integer.MIN_VALUE;
        for (int i = 0, j = 1; i < n; i++) {
            if (i > 0 && j > i) {
                int s = A[i][0] - A[0][0] + A[i][1];
                int f = sum.get(s) - 1;
                if (f > 0) {
                    sum.put(s, f);
                } else {
                    sum.remove(s);
                }
            }
            j = Math.max(j, i + 1);
            while (j < n && A[j][0] - A[i][0] <= k) {
                int sj = A[j][0] - A[0][0] + A[j][1];
                sum.put(sj, sum.getOrDefault(sj, 0) + 1);
                j++;
            }
            
            if (j - i > 1) {
                int ms = A[i][1] + sum.lastKey() + A[0][0] - A[i][0];
                res = Math.max(res, ms);
            }
            // System.out.println(i + ", sum=" + sum + ", res=" + res);
        }
        return res;
        
    }
}
