class Solution {
    
public int findMaxValueOfEquation(int[][] points, int k) {
        Deque<Integer> low = new ArrayDeque<>();
        Deque<Integer> high = new ArrayDeque<>();
        low.offer(0);
        high.offer(0);
        int result = Integer.MIN_VALUE;
        for (int i = 1;i<points.length;i++) {
            while (!low.isEmpty() && points[i-1][1] - points[i-1][0] >= points[low.peekLast()][1] - points[low.peekLast()][0]) {
                low.pollLast();
            }
            low.offer(i-1);
            
while (!high.isEmpty() && points[i][1] + points[i][0] >= points[high.peekLast()][1] + points[high.peekLast()][0]) {
                high.pollLast();
            }
            high.offer(i);
            while (!low.isEmpty() && points[i][0] - points[low.peek()][0] > k) {
                low.poll();
            }
            while (!high.isEmpty() && points[i][0] - points[high.peek()][0] > k) {
                high.poll();
            }
            if (!high.isEmpty() && high.peek().equals(low.peek())) {
                high.poll();
            }
            // System.out.println(low);
            // System.out.println(high);
            // System.out.println();
            if (!low.isEmpty() && !high.isEmpty()) {
                result = Math.max(result, points[low.peek()][1] + points[high.peek()][1] + Math.abs(points[high.peek()][0] - points[low.peek()][0]));
            }
        }
        return result;
    }
}
