class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        int max = Integer.MIN_VALUE, n = points.length, r = 1;
        
        LinkedList<Integer> deque = new LinkedList();
        
for (int i = 0; i < n; i++) {
            while (!deque.isEmpty() && deque.peekFirst() <= i) {
                deque.pollFirst();
            }
            
            r = Math.max(r, i+1);
            while (r < n && points[r][0] - points[i][0] <= k) 
{
                while (!deque.isEmpty() &&
                       points[deque.peekLast()][0] + points[deque.peekLast()][1] <= points[r][0] + points[r][1] ) {
                    deque.pollLast();
                }
                deque.add(r++);
            }
            // if (i == 8)
            //     System.out.println(i + " " + r + " " + deque);
            
            if (!deque.isEmpty()) {
                int cur =  points[deque.peekFirst()][0] + points[deque.peekFirst()][1] + points[i][1] - points[i][0];
                max = Math.max(max, cur);
            }
        }
        
        return max;
    }
}
