class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        Deque<Integer> dq = new LinkedList<>();
        int res = Integer.MIN_VALUE;
        for (int j = 0; j < points.length; j++) {
            
while (!dq.isEmpty() && points[dq.peek()][0] < points[j][0] - k) {
                dq.pollFirst();
            }
            if (!dq.isEmpty()) {
                int i = dq.peekFirst();
                int vi = points[i][1] - points[i][0];
                int vj = points[j][1] + points[j][0];
                
res = Integer.max(res, vi + vj);
            }
            int vj = points[j][1] - points[j][0];
            while (!dq.isEmpty()) {
                int vi = points[dq.peekLast()][1] - points[dq.peekLast()][0];
                if (vi > vj) {
                    break;
                }
                dq.pollLast();
            
}
            dq.offerLast(j);
        }
        return res;
    }
}
