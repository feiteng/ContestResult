class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        
        
        Deque<Integer> indices = new LinkedList<>();
        int max = -1000000000;
        int n = points.length;
        for (int i = 0; i < n; ++i) {
            int xcur = points[i][0];
            int ycur = points[i][1];

            //remove expired
            while(indices.size() > 0 && ( xcur - points[indices.getFirst()][0] > k))indices.removeFirst();

            //calculate
            if(indices.size() > 0 ){
            int il = indices.getFirst();
            int v = (xcur - points[il][0]) + points[il][1] + ycur;
            max = max < v? v: max;
        }
            //remove unuseful
            
while (indices.size() > 0 && (ycur - xcur >= points[indices.getLast()][1] - points[indices.getLast()][0]))indices.removeLast();
            indices.addLast(i);

        }
        return max;
    }
}
