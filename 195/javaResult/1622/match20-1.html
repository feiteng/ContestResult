class Solution {
    public int findMaxValueOfEquation(int[][] p, int k) {
        int res = Integer.MIN_VALUE, l = p.length;

        Deque<Integer> dq = new ArrayDeque<>();
        for (int i = 0; i < l; i++) {	
            
while (!dq.isEmpty() && (p[i][0] - p[dq.peekFirst()][0]) > k) {
                dq.pollFirst();
            }
            // System.out.println(i + " " + p[i][0] + " " + p[i][1] + " " + dq.size());
            if (!dq.isEmpty()) {
                res = Math.max(res, p[i][1] + p[dq.peekFirst()][1] + p[i][0] - p[dq.peekFirst()][0]);
            }    
            while (!
dq.isEmpty() && p[i][1] - (p[i][0] - p[dq.peekLast()][0]) > p[dq.peekLast()][1]) {
                dq.pollLast();
            }
            dq.offerLast(i);
        }
        
        return res;
    }
}
