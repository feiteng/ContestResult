class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        return sln1(points, k);
    }
    
    private int sln1(int[][] ps, int k){
        Deque<Integer> dq1 = new ArrayDeque<>(); //for yi+xi+yj-xj
        Deque<Integer> dq2 = new ArrayDeque<>(); //for yi-xi+yj+xj
        
        dq1.addLast(0);
        dq2.addLast(0);
        int max = Integer.MIN_VALUE;
        
        for(int j =1;j<ps.length;j++){
            
            while(dq2.size()>0 && ps[dq2.getFirst()][0]+k<ps[j][0]) dq2.removeFirst();
            
            if(dq2.size() >0){
                int r2 = ps[j][1]+ps[j][0];
                
int i2 = dq2.getFirst();
                int l2 = ps[i2][1]-ps[i2][0];
                max = Math.max(max, l2+r2);
                //System.out.printf("r2:%d, i2:%d, l2:%d, max:%d\n", r2,i2,l2,max);
                
            }
            
            int l2 = ps[j][1]-ps[j][0];
            while(dq2.size()>0 && ps[dq2.getLast()][1]-ps[dq2.getLast()][0] < l2){
                dq2.removeLast();
            
}
            dq2.addLast(j);
            
        }
        return max;
    }
}
