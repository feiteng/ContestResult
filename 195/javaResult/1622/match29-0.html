class Solution {
    public int findMaxValueOfEquation(int[][] pts, int k) {
        int n = pts.length;
        TreeMap<Integer, Integer> xpy = new TreeMap<>();
        int a = 0;
        int b = 0;
        int ans = -1000000000;
        while (b < n) {
            if (pts[b][0]-pts[a][0]<=k) {
                if (xpy.containsKey(pts[b][0]+pts[b][1])) {
                    xpy.put(pts[b][0]+pts[b][1], xpy.get(pts[b][0]+pts[b][1])+1);
                } else {
                    xpy.put(pts[b][0]+pts[b][1], 1);
                }
                ++b;
                continue;
            }
            if (xpy.get(pts[a][0]+pts[a][1])==1) {
                xpy.remove(pts[a][0]+pts[a][1]);
            } else {
                xpy.put(pts[a][0]+pts[a][1], xpy.get(pts[a][0]+pts[a][1])-1);
            }
            if (!xpy.isEmpty())
                ans = Math.max(ans, pts[a][1]+xpy.lastKey()-pts[a][0]);
            //System.out.println(a+" "+b+" "+ans);
            ++a;
        }
        while (a < n) {
            if (xpy.get(pts[a][0]+pts[a][1])==1) {
                xpy.remove(pts[a][0]+pts[a][1]);
            } else {
                xpy.put(pts[a][0]+pts[a][1], xpy.get(pts[a][0]+pts[a][1])-1);
            }
            if (!xpy.isEmpty())
                ans = Math.max(ans, pts[a][1]+xpy.lastKey()-pts[a][0]);
            //System.out.println(a+" "+b+" "+ans);
            ++a;
        }
        return ans;
    }
}
