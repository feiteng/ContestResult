import java.util.*;
import java.lang.*;


class Solution {
    public int findMaxValueOfEquation(int[][] points, int k) {
        TreeMap<Integer, Integer> YMinusXToCount = new TreeMap<>();
        YMinusXToCount.put(points[0][1] - points[0][0], 1);
        int left = 0;
        int result = Integer.MIN_VALUE;
        for (int j = 1; j < points.length; j++) {
            // System.out.println("j: " + j);
            // System.out.println("Start: YMinusXToCount: " + YMinusXToCount.toString());
            while (points[j][0] - points[left][0] > k) {
                int key = points[left][1] - points[left][0];
                YMinusXToCount.put(key, YMinusXToCount.get(key) - 1);
                if (YMinusXToCount.get(key) == 0) {
                    YMinusXToCount.remove(key);
                }
                left += 1;
            
}
            if (!YMinusXToCount.isEmpty()) {
                Integer maxKey = YMinusXToCount.lastKey();
                result = Math.max(result, maxKey + points[j][1] + points[j][0]);
            }
            int key = points[j][1] - points[j][0];
            YMinusXToCount.put(key, YMinusXToCount.getOrDefault(key, 0) + 1);
            // System.out.println("End: YMinusXToCount: " + YMinusXToCount.toString());
        }
        return result;

    }
}
