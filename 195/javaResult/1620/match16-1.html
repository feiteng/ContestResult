class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] mo=new int[k];
        for(int i=0;i<arr.length;i++) {
            while(arr[i]<0)
                arr[i]+=k;
            mo[arr[i]%k]++;
        
}

        if(mo[0]%2!=0)
            return false;
        for(int i=1;i<=k/2;i++) {
            if(mo[i]!=mo[k-i])
                return false;
        }
        return true;

    }
}
