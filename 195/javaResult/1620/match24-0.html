class Solution {
    public boolean canArrange(int[] arr, int k) {
        int rem[]=new int[k];
        for(int i:arr)
        {
            int re=Math.abs(i)%k;
            if(i<0)
                re=-re;
            if(re<0)
                rem[k+re]++;
            else
                rem[re]++;
        }
        //System.out.println(Arrays.toString(rem));
        for(int i=1;i<k;i++)
        {
            if(rem[i]!=rem[k-i])
                return false;
        }
        
        if(rem[0]%2!=0)
            return false;
        return true;
    }
}
