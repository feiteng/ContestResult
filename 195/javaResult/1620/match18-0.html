class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] counts = new int[k];
        for (int num : arr) {
            counts[(num%k+k)%k]++;
        }
        if (counts[0]%2 != 0) {
            return false;
        }
        for (int i = 1; i < k; i++) {
            if (counts[i] != counts[k-i]) {
                return false;
            }
        }
        if (k%2 == 0 && counts[k/2]%2 != 0) {
            return false;
        }
        return true;
    }
}
