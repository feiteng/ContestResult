class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] dp = new int[k];
        for (int a : arr) {
            if (a >= 0) dp[a % k]++;
            else dp[(a % k + k) % k]++;
        }
        if (dp[0] % 2 != 0) return false;
        for (int i = 1; i <= k / 2; i++) {
            if (dp[i] != dp[k - i]) return false;
        }
        return true;
    }
}
