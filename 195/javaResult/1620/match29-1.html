class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] L = new int[k];
        Arrays.fill(L, 0);
        for (int i : arr) {
            if(i>0)L[i % k]++;
            else L[(k-(-i)%k)%k]++;
        }
        if (L[0] % 2 == 1) return false;
        
if (k % 2 == 0 && L[k / 2] % 2 == 1) return false;
        for (int i = 1; i < k / 2; ++i) {
            if (L[i] != L[k - i]) return false;
        }
        return true;
    }
}
