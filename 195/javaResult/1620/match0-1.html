class Solution {
    public boolean canArrange(int[] arr, int k) {
      for (int i = 0; i < arr.length; i++) {
         arr[i] = arr[i] % k;
          while(arr[i] < 0) arr[i] += k;
      }
      Map<Integer, Integer> map = new HashMap<>();
      for (int ind : arr) {
       int pair = ind== 0 ? 0 :k - ind;
       if (!map.containsKey(pair)) map.put(ind, map.getOrDefault(ind, 0) + 1);
       else {
           if (map.get(pair) == 1) map.remove(pair);
           else map.put(pair, map.get(pair) - 1);
         }      
      }
      return map.size() == 0;
    }
}
