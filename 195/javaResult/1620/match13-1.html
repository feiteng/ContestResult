	    public boolean canArrange(int[] a, int k) {
	        int[] f = new int[k];
	        for(int v : a){
	        	v %= k;
	        	if(v < 0)v += k;
	        	f[v]++;
	        }
	        for(int
 i = 1;i < k;i++){
	        	if(f[i] != f[k-i])return false;
	        }
	        return f[0] % 2 == 0;
	    }
