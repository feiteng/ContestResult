class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] cnt = new int[k];
        for (int num : arr) {
            int re = num % k;
            if (re < 0) re += k
;
            cnt[re]++;
        }
        if (cnt[0] % 2 > 0) return false;
        for (int i = 1; i < k / 2; ++i) {
            int j = k - i;
            
if (cnt[i] != cnt[j]) return false;
        }
        return true;
    }
}
