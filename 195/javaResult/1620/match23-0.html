class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] mod = new int[k];
        for (int x : arr)
            mod[x%k >= 0 ? x%k : x%k+k]++;
        for (int i = 1; i <= k/2; i++)
            if (mod[i] != mod[k-i])
                return false;
        if (k%2 == 0 && mod[k/2]%2 != 0)
            return false;
        return true;
    }
}
