class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] bucket = new int[k];
        
        for(int a: arr){
            int rem = (a % k + k)% k;
            bucket[rem]++;
        
}
        if(bucket[0] % 2 == 1)return false;
        if(k % 2 == 0 && bucket[k/2] % 2 == 1)return false;
        for(
int i = 1, j = k - 1; i < j; i ++, j--){
            if(bucket[i] != bucket[j])return false;
        }
        return true;
    }
}
