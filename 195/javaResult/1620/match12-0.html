class Solution {
    public boolean canArrange(int[] nums, int k) {
        int[] q = new int[nums.length];
		for (int i = 0; i < nums.length; i++) {
			int cur = nums[i] % k;
			q[i] = cur < 0 ? cur + k : cur;
		}
		int[] map = new int[k];
		for (int i : q) map[i]++;
		if (map[0] % 2 == 1) return false;
		for (int i = 1; i <= k / 2; i++) {
			if (map[i] != map[k - i]) return false;
		}
		return true;

    }
}
