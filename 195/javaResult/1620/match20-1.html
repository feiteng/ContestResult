class Solution {
    public boolean canArrange(int[] arr, int k) {
        
        int rems[]=new int[k];
        for(int i=0;i<arr.length;i++)rems[((arr[i]%k)+k)%k]++
;
        if(rems[0]%2!=0)return false;
        for(int i=1;i<k;i++) {
            if (rems[i]!=rems[k-i])return false;
        }
        return true;
    }
}
