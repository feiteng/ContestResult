class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] count = new int[k];
        for(int a : arr){
            int d = a % k;
             if(d < 0){
                d += k;
            }
            count[d]++;
        }
        
        if(count[0] % 2 == 1) return false;
        int i = 1, j = k - 1;
        while(i <= j){
            if(count[i] != count[j]) return false;
            i++;
            j--;
        }
        return true;
    }
}
