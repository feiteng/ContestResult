class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] remainder = new int[k];
        for (int i : arr) remainder[i % k < 0 ? i % k + k : i % k]++;
        for (int r = 1; r < k; r++) if (remainder[r] != remainder[k - r]) return false;
        return remainder[0] % 2 == 0;
    }
}
