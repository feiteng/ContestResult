class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] help = new int[k]; // help[i] means the # of elem in arr that mod k == i
        for(int x: arr) {
            int tmp = x % k;
            if(tmp < 0) tmp += k
;
            help[tmp]++;
        }
        
        if(help[0] % 2 != 0) return false;
        
        int i = 1;
        int j = k-1;
        while(i < j) {
            if(help[i] != help[j]) return false;
            i++;
            j--;
        }
        return true;
    }
}
