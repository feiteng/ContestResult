class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] map = new int[k+1];
        for (int i = 0; i < arr.length; ++i) {
            arr[i] = arr[i] % k;
            if (arr[i] < 0) arr[i] += k;
            map[arr[i]]++;
        }
        if (map[0] % 2 != 0) return false;
        for (int i = 1; i < k; ++i) {
            if (map[i] != map[k-i]) return false;
        }
        return true;
    }
}
