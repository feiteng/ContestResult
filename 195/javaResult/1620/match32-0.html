class Solution {
    public boolean canArrange(int[] arr, int k) {
        int[] nums = new int[k];
        for (int i : arr) {
             int j = i % k;
            if (j < 0) j += k;
            nums[j]++;
        }
        if (nums[0] % 2 != 0) return false;
        if (k % 2 == 0) {
            if (nums[k / 2] % 2 != 0) return false;
        }

        for (int i = 1; i <= (k-1) / 2; i++) {
            if(nums[i]!=nums[k-i]) return false;
        }
        return true;
    }
}
