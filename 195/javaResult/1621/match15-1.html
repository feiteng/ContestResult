class Solution {
    
private static final int MOD = 1000000007;

    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int left = 0;
        int right = nums.length - 1;
        int result = 0;
        while (left <= right) {
            while (left <= right && nums[left] + nums[right] > target) {
                right--;
            }
            if (left > right) {
                return result;
            
}
            result = (result + (int)modPow(right - left)) % MOD;
            left++;
        }
        return result;
    }
    
    public long modPow(int pow) {
        if (pow == 0) return 1;
        long temp = modPow(pow / 2);
        if (pow % 2 == 0) {
            return (temp * temp) % MOD;
        }
        return (temp * temp * 2) % MOD;
    }
}
