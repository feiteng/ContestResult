class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int N = nums.length;
        int l = 0;
        int r = N - 1;
        int mod = 1_000_000_007;
        int[] table = new int[100010];
        table[0] = 1;
        for (int i = 1; i < table.length; i++) {
            table[i] = table[i - 1] * 2;
            table[i] %= mod;
        }
        int res = 0;
        while (r >= l) {
            if (nums[l] + nums[r] <= target) {
                res += table[r - l];
                res %= mod;
                l++;
            } else {
                r--;
            }
        }
        return res;

    }
}
