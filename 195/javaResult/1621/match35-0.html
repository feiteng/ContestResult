class Solution {
    int mod = (int) 1e9 + 7;
    public int numSubseq(int[] nums, int target) {
        long res = 0;
        Arrays.sort(nums);
        int l = nums.length;
        int left = 0, right = l - 1;
        
        while (left <= right) {
            if (nums[left] + nums[right] > target) {
                right--;
            }
            else {
                res = (res + pow(2L, right - left));
                left++;
            }
        }
        
        return (int) (res % mod);
    }
    
    private long pow(long base, int p) {
        if (p == 0) return 1;
        return p % 2 == 0 ? pow(base * base % mod, p / 2) : base * pow(base, p - 1) % mod;
    }
}
