class Solution {
    private static final int MOD = 1000000007;
    
    private static final long[] POW2 = new long[100001];
    static {
        POW2[0] = 1;
        for (int i = 1; i < POW2.length; ++i) {
            POW2[i] = (2 * POW2[i - 1]) % MOD;
        }
    }
    
    public int numSubseq(int[] nums, int target) {
        final int N = nums.length;
        
        Arrays.sort(nums);
        
        int L = 0;
        int R = N - 1;
        long total = 0;
        while (L <= R) {
            if (nums[L] + nums[R] > target) {
                --R;
            } else {
                int size = R - L + 1;
                total += POW2[size - 1];
                total %= MOD;
                ++L;
            }
        }
        return (int)total;
    }
}
