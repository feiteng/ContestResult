class Solution {
    long MOD = 1000000007;
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        long res = 0;
        for(int i=0;i<nums.length;i++) {
            int s=i;
            int e=nums.length-1;
            int ans = -1;
            while(s<=e) {
                int mid = (s+e)>>1;
                if(nums[i]+nums[mid]<=target) {
                    ans=mid;
                    s=mid+1;
                } else {
                    e=mid-1;
                }
            
}
            if(ans !=-1) {
                res+=pow(ans-i);
                res%=MOD;
            }
        }
        return (int)res;
    }
    
    public long pow(int x) 
{
        if(x==0)return 1;
        if(x==1)return 2;
        long m = pow(x/2);
        m%=MOD;
        long ans = m*m;
        ans %=MOD;
        if(x%2==1) {
            ans *= 2;
        }
        ans %=MOD;
        return ans;
    }
}
