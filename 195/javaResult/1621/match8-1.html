class Solution {
 
  private int mod = (int) 1e9 + 7;

  public int numSubseq(int[] nums, int target) {
    int n = nums.length;
    Arrays.sort(nums);
    long res = 0;
    for (int i = 0; i < n; i++) {
      if ((long)nums[i] * 2 <= target) {
        res++;
      }
      
int l = i + 1, r = n - 1;
      while (l <= r) {
        int m = l + ((r - l) >> 1);
        if (nums[m] <= target - nums[i]) {
          l = m + 1;
        } else {
          r = m - 1;
        }
      }
      if (r > i) {
        res += cal(r - i) - 1;
        res = (res + mod) % mod;
      }
    }
    return (int) (res % mod);
  }

  private long cal(int n){
    long base = 2 ,res = 1;
    while(n > 0){
      if((n & 1) == 1){
        res *= base;
        res %= mod;
      }
      base *= base;
      base %= mod;
      n >>= 1;
    }
    return res;
  }
}
