import java.util.Arrays;

class Solution {
    private static final int MAX_N = 1000000007;
    private int[] k2;

    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        k2 = 
new int[nums.length];
        k2[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            k2[i] = (k2[i - 1] << 1) % MAX_N;
        }
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            int oidx = binary_search(nums, target - nums[i]);
            if (oidx < i) {
                break;
            }
            sum = (sum + k2[oidx - i]) % MAX_N;
        
}
        return sum;
    }

    private int binary_search(int[] nums, int target) {
        int s = 0, e = nums.length - 1;
        while (s <= e) {
            int mid = s + ((e - s) >> 1);
            if (nums[mid] > target) {
                e = mid - 1;
            } else {
                if (mid + 1 > e || nums[mid + 1] > target) {
                    return mid;
                }
                s = mid + 1;
            }
        }
        return -1;
    }
}
