class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int mod = (int)1e9 + 7;
        long[] base = new long[nums.length];
        base[0] = 1L;
        for(int i = 1; i < base.length; i++) {
            base[i] = base[i - 1] * 2 % mod;
        }
        
        long cnt = 0;
        for(int i = 0; i < nums.length; i++) {
            if(nums[i] * 2 > target) {
                break;
            }
            int r = bs(nums, target - nums[i]);
            if(r > i) {
                cnt = (cnt + base[r - i]) % mod;
            }
            else {
                cnt = (cnt + 1) % mod;
            }
        }
        
        return (int)cnt;
    }
    private int bs(int[] nums, int t) {
        int l = 0, r = nums.length - 1;
        while(l < r - 1) {
            int mid = l + (r - l) / 2;
            if(nums[mid] <= t) {
                l = mid;
            }
            else {
                r = mid - 1;
            }
        }
        if(nums[r] <= t) {
            return r;
        }
        return l;
    }
}
