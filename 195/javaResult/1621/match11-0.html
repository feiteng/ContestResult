class Solution {
    public int numSubseq(int[] nums, int tg) {
        int n = nums.length;
        Arrays.sort(nums);
        int a = 0;
        int b = n-1;
        int[] pows = new int[n+1];
        pows[0] = 1;
        for (int i = 1; i <= n; ++i)
            pows[i] = (pows[i-1]*2)%1000000007;
        int ans = 0;
        while (a<=b) {
            if (nums[a]+nums[b]<=tg) {
                ans += pows[b-a];
                ans %= 1000000007;
                a++;
            } else {
                b--;
            }
        }
        return ans;
    }
}
