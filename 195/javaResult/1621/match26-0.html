class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        long[] mem = new long[nums.length + 1];
        mem[0] = 1;
        for(int i = 1; i <= nums.length; i++){
            mem[i] = (mem[i - 1] + mem[i - 1]) % 1000000007;
        }
        long re = 0;
        for(int i = nums.length - 1; i >= 0; i--){
            int low = i, high = nums.length - 1;
            while(low < high){
                int mid = (low + high + 1) / 2;
                if(nums[mid] + nums[i] <= target){
                    low = mid;
                }else{
                    high = mid - 1;
                }
            }
            if(nums[low] + nums[i] > target)
                continue;
            re = (re + mem[low - i]) % 1000000007;
        }
        return (int) re;
    }
}
