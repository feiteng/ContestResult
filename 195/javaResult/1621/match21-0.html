class Solution {
    //32 + 16 + 8 + 4 = 
    private int MOD = (int)1e9 + 7;
    Map<Integer, Long> memo = new HashMap<>();
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        long result = 0;
        int left = 0, right = nums.length - 1;
        long preMult = 1;
        for (int i = 1; i < nums.length; i ++) {
            preMult = (preMult * 2) % MOD;
            memo.put(i, preMult);
        }
        
        while (left <= right) {
            int sum = nums[left] + nums[right];
            if (sum <= target) {
                result = (result + memo.getOrDefault(right - left - 1, 1l)) % MOD;
                if (left != right) {
                    result = (result + memo.getOrDefault(right - left - 1, 1l)) % MOD;
                }
                left ++;
            } else {
                right --;
            }
        }
        
        return (int)result;
    }
