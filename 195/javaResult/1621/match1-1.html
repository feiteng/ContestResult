class Solution
{
    private int mod = 1_000_000_007;
    private long calc2pow(int n)
    {
        if (n == 0) return 1;
        if (n == 1) return 2;
        long half = calc2pow(n / 2);
        long ans =  (half * half) % mod;
        if (n % 2 == 1) ans = (ans * 2) % mod;
        return ans;
    }
    
public int numSubseq(int[] nums, int target)
    {
        Arrays.sort(nums);
        int l = nums.length - 1;
        int ans = 0;
        for (int i = 0; i < nums.length; i++)
        {
            while (i <= l && nums[i] + nums[l] > target) l--;
            if (i > l) break;
            ans = (ans + (int)calc2pow(l - i)) % mod;
        }
        return ans;
    }
}
