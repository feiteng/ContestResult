class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int mod = 1000000007;
        long[] mi = new long[nums.length];
        mi[0] = 1;
        for (int i = 1; i < mi.length; i++) {
            mi[i] = 2 * mi[i - 1];
            mi[i] %= mod;
        }
        long res = 0;
        int k = nums.length - 1;
        for (int i = 0; i < nums.length; i++) {
            int j = k;
            if (j < i)
                break;
            if (j == i && nums[j] + nums[i] > target)
                break;
            for (; j >= i; j--) {
                if (nums[i] + nums[j] <= target)
                    break;
            }
            if (j < i)
                continue;
            res += mi[j - i];
            res %= mod;
            k = j;
        }
        return (int) res;
    }
}
