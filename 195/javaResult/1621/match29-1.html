class Solution {

    private int mod = 1000000007;

    public int numSubseq(int[] nums, int target) {

        Arrays.sort(nums);

        int n = nums.length;
        long ans = 0;

        for (int i = 0; i < n; i++) {
            if (nums[i] > target)
                break;
            
int l = i, r = n - 1, p = -1;
            while (l <= r) {
                int mid = (l + r) / 2;
                if (nums[i] + nums[mid] <= target) {
                    p = mid;
                    l = mid + 1;
                } else
                    r = mid - 1;
            }

            if (p != -1)
                ans = (ans + Sum(p - i + 1)) % mod;
        }

        return (int)ans;

    }

    private long Sum(int x) {

        return q(2, x - 1) % mod;

    }

    private long q(long x, long y) {
        
long res = 1;
        while (y > 0) {
            if (y % 2 == 1)
                res = res * x % mod;
            x = x * x % mod;
            y /= 2;
        }
        return res;
    }
}
