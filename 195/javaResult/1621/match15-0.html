class Solution {
    int mod = 1000000007;
    int[] buf;
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int l = 0, r = nums.length-1;
        int ans = 0;
        while (l <= r) {
            if (nums[l] + nums[r] > target) {
                r--;
                continue;
            }
            ans = (ans + cnt(r-l))%mod;
            l++;
        }
        return ans;
    }
    
    int cnt(int n) {
        if (buf == null)
            buf = new int[n+1];
        else
            return buf[n];
        buf[0] = 1;
        for (int i = 1; i <= n; i++) {
            buf[i] = buf[i-1]*2%mod;
        }
        return buf[n];
    }
}
