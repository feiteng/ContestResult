class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int mod = 1000000007;
        int[] map = new int[nums.length+1];
        map[0] = 1;
        for (int i = 1;i<=nums.length;i++){
            map[i] = (map[i-1] * 2) % mod;
        }

        int[] preSum = new int[map.length];
        preSum[0] = map[0];
        for (int i = 1;i<preSum.length;i++){
            preSum[i] = preSum[i-1] + map[i];
            if (preSum[i] >= mod) {
                preSum[i] -= mod;
            }
        }

        int i = 0;
        int j = nums.length-1;
        int res = 0;
        while (i <= j){
            if(nums[i] + nums[j] <= target){
                if(i == j){
                    res++;
                }else {
                    res += preSum[j - i - 1];
                    res++;
                    if (res >= mod) {
                        res -= mod;
                    }
                }
                i++;
            }else{
                j--;
            }
        }
        return res;
    }
}
