class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        long sum=0;
        long mod=(long)1e9+7;
        int start=0,end=nums.length-1;
        
long[] powa=new long[nums.length];
        powa[0]=1;
        for(int i=1;i<powa.length;i++){
            powa[i]=2*powa[i-1]%mod;
        }
        
while(start<=end){
            if(nums[start]+nums[end]>target){
                
                end--;
                continue;
            }
            sum+=powa[end-start];
            sum%=mod;
             start++;
        }
        return (int)sum;
    }
