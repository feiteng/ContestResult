class Solution {
    static long mod=1000000007;
    public int numSubseq(int[] nums, int target) {
        return get(nums,target,nums.length);
    }
    static int get(int arr[], int K, 
                                       int N) 
    { 
        Arrays.sort(arr); 

        int left, right; 
        left = 0; 
        right = N - 1; 

        long ans = 0; 

        while (left <= right) 
        { 
            if (arr[left] + arr[right] <= K) 
            { 
                ans = (ans%mod+power(2,right-left))%mod; 
                left++; 
            } 
            else 
            { 
                right--; 
            } 
        } 
        return (int)(ans%mod); 
    }
    static int power(long a,long b){
        long res=1;
        while(b>0){
            if(b%2!=0){
                res=(res%mod*a%mod)%mod;b--;
            }
            a=(a%mod*a%mod)%mod;
            b/=2;
        }
        return (int)(res%mod);
    }
}
