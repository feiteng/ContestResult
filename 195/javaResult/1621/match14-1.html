class Solution {
    int MOD = (int)1e9 + 7;
    public int numSubseq(int[] A, int T) {
        Arrays.sort(A);
        int n = A.length;
        if (A[n-1] * 2 <= T) {
            return (int)pow(2, n) - 1;
        }
        long ans = 0;
        for (int i = 0; i <n; i++) {
            int lo = i, hi = n-1;
            while (lo < hi) {
                int mid = (lo + hi+1) >>> 1;
                int cntT = T - A[i];
                if (A[mid] <= cntT) {
                    lo = mid;
                } else {
                    hi = mid-1;
                }
            }
            // System.out.println("" + A[i] + ":" + lo);
            if (lo - i == 0 && A[i] * 2 > T) continue;
            ans = (ans + pow(2,lo - i)) % MOD;
        }
        
return (int)(ans % MOD);
    }
    
    long pow(long x, long n) {
        long ans = 1;
       
        while (n != 0) {
            if ((n & 1) == 1) {
                ans = (ans * x) % MOD;
            }
            x = (x * x) % MOD;
            n >>= 1;
        }
        return ans;
    }
}
