import java.util.Arrays;
import java.util.Collections;

class Solution {
    public int numSubseq(int[] nums, int target) {
        int mo=(int)1e9+7;
        long ans=0;
        Arrays.sort(nums);
        int left=0;
        int right=nums.length-1;
        long[] mi=new long[nums.length+1];
        mi[0]=1;
        for(int i=1;i<mi.length;i++) {
            mi[i]=(mi[i-1]*2)%mo;
        }
        while(left<right) {
            if(nums[left]+nums[right]<=target) {
                int len=right-left-1;
                //ans=(ans+(long)Math.pow(2,len+1)-1)%mo;
                ans=(ans+mi[len+1]-1)%mo;
                left++;
            } else {
                right--;
            }
        }
        for(int i=0;i<nums.length;i++) {
            if(nums[i]*2<=target)
                ans=(ans+1)%mo;
        }
        return (int)ans%mo;
    }

}
