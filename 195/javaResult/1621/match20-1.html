class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int sum = 0;
        
        int pow[] = new int[100005];
        int mod = 1000000007;
        pow[0] =1;
        for(int i=1;i<=100002;++i){
            pow[i]  =pow[i-1]*2;
            pow[i] %= mod;
        }
        
        
        
        
        for(int i=0;i<nums.length;++i){
            if(nums[i]*2>target){
                return sum;
            }
            
            
            
int lo = i;
            int hi =  nums.length-1;
            
            while(lo<hi){
                int mi = lo+hi+1; mi >>=1;
                
                if(nums[i]+nums[mi]<=target){
                    lo = mi;
                }else{
                    hi  = mi-1;
                }
                
            }
            
            sum += pow[lo-i];
            sum %= mod;
            
        }
        return sum;
        
        
        
    }
}
