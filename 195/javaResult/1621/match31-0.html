class Solution {
    public int numSubseq(int[] nums, int target) {
        int n = nums.length;
        Arrays.sort(nums);
        int j = n - 1;
        long ans = 0, mod = 1_000_000_007;
        long[] pow = new long[1_000_10];
        pow[0] = 1;
        for(int i = 1; i <= 100_000; i++)
        {
            pow[i] = pow[i - 1] * 2 % mod;
        }
        for(int i = 0; i < n; i++)
        {
            while(j >= i && nums[i] + nums[j] > target) j--;
            if(j < i) break;
            int len = j - i;
            ans += pow[len];
        }
        return (int)(ans % mod);
    }
    
    long pow(int k)
    {
        long ans = 0, prod = 1;
        long mod = 1_000_000_007;
        if(k == 0) return 1;
        for(int i = 1; i <= k; i++)
        {
            prod *= 2;
            prod %= mod;
        }
        return prod;
    }
}
