class Solution {
    int MOD = (int)1e9+7;
    
    public int numSubseq(int[] a, int target) {
        Arrays.sort(a);
        int n = a.length;
        
        long ans = 0;
        
        for(int i=0; i<n; i++) {
            if (2*a[i]>target) continue;
            // if (2*a[i]==target) {
            //     ans++;
            //     ans%=MOD;
            //     continue;
            // }
            int low = i;
            int high = n-1;
            int mid = 0;
            while(low<high) {
                mid = low+(high-low+1)/2;
                if (a[i]+a[mid]>target) {
                    high=mid-1;
                } else {
                    low=mid;
                }
            }
            ans = (ans+(modExp(2, low-i+1-2+1)))%MOD;
            // System.out.println(i+" "+ans);
        }
        return (int)ans%MOD;
    }
    
    private long modExp(long a, long b) {
		if (b == 0)
			return 1;

		a %= MOD;

		long exp = modExp(a, b / 2);

		if (b % 2 == 0) {
			return (exp * exp) % MOD;
		} else {
			return (a * ((exp * exp) % MOD)) % MOD;
		}
	}
}
