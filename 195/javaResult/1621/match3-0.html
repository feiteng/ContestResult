import java.util.Arrays;

class Solution {

    private int MOD = 1_000_000_007;

    public int numSubseq(int[] nums, int target) {

        int[] pow = new int[nums.length];
        pow[0] = 1;
        for (int i = 1; i < pow.length; i++) {
            pow[i] = (pow[i - 1] * 2) % MOD;
        }
        int ans = 0;
        Arrays.sort(nums);

        int lo = 0, hi = nums.length - 1;
        while (lo <= hi) {
            if (nums[lo] + nums[hi] > target) {
                hi--;
                continue;
            }
            ans += pow[hi - lo];
            ans %= MOD;
            lo++;
        }
        return ans;

    }
}
