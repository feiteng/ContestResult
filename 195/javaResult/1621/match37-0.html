class Solution {
    public int numSubseq(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        long mod = 1000000007;
        long res = 0;
        long[] count = new long[n+1];
        count[0] = 1;
        for(int i=1; i<=n; i++){
            count[i] = count[i-1]*2%mod;
        }
        int lo = 0;
        int hi = n-1;
        while(lo<=hi){
            while(lo<hi && nums[lo]+nums[hi]>target){
                hi--;
            }
            if(nums[lo]+nums[hi]<=target){
                res = (res+count[hi-lo])%mod;
            }
            lo++;
        }
        return (int)(res%mod);
    }
    
}
