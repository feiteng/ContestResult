class Solution {
    Map<Integer, Integer> map = new HashMap<>();
    
    public int numSubseq(int[] nums, int target) {
        
        Arrays.sort(nums);
        int sum = 0;
        int i = 0;
        int j = nums.length - 1;
        while (i <= j) {
            while (i <= j && nums[i] + nums[j] > target) {
                j--;
            }
            if (i <= j) {
                sum += pow(j - i);
                sum = sum % 1000000007;
            } else {
                break;
            }
            i++;
        }
        return sum;
    }
    
    int pow(int i) {
        if (map.containsKey(i)) {
            return map.get(i);
        }
        if (i == 0) {
            return 1;
        }
        if (i == 1) {
            return 2;
        } else {
            int val = (2 * pow(i - 1)) % 1000000007;
            map.put(i, val);
            return val;
        }
    }
}
