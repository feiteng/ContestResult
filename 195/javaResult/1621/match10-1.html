class Solution {
    public int numSubseq(int[] nums, int target) {
        int ans=0;
        Arrays.sort(nums);
        int p=0,q=nums.length-1;
        int help=1;
        for(int i=0;i<nums.length-1;i++){
            help*=2;
            help%=1000000007;
        }
        
while(p<=q){
            if(nums[p]+nums[q]<=target){
                ans+=help;
                ans%=1000000007;
                p++;
            }
            else {
                q--;
            }
            if(help%2!=0){
                help+=1000000007;
            }
            help/=2;
        }
        return ans;
    }
}
