class Solution {
    public boolean isPathCrossing(String path) {
        int x = 0;
        int y = 0;
        
        Map<Integer, Set<Integer>> map = new HashMap<>();
        
        map.computeIfAbsent(x, k->new HashSet<>()).add(y);
        
        for(char c : path.toCharArray()){
            if(c == 'N'){
                y++;
            }else if(c == 'S'){
                y--;
            }else if(c == 'E'){
                x--;
            }else if(c == 'W'){
                x++;
            }
            
            Set<Integer> set =  map.computeIfAbsent(x, k->new HashSet<>());
            if(set.contains(y)){
                return true;
            }
            set.add(y);
        }
        return false;
    }
}
