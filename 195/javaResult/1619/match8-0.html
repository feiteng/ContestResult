class Solution {
    public boolean isPathCrossing(String s) {
        HashSet<String> set=new HashSet<>();
        int x=0;
        int y=0;
        set.add(0+" "+0);
        for(int i=0;i<s.length();i++)
        {
            char ch=s.charAt(i);
            if(ch=='N')
            {
                y+=1;
            }
            else if(ch=='W')
            {
                x-=1;
            }
            else if(ch=='E')
            {
                x+=1;
            }
            else if(ch=='S')
            {
                y-=1;
            }
            if(set.contains(x+" "+y))
            {
                return true;
            }
            set.add(x+" "+y);
        }
        return false;
    }
}
