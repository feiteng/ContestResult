class Solution {
    public boolean isPathCrossing(String path) {
        Set<Integer> set = new HashSet<>();
        set.add(0);
        int x = 0, y = 0;
        for (char c : path.toCharArray()) {
            switch(c) {
                case 'N': y++; break;
                case 'S': y--; break;
                case 'E': x++; break;
                case 'W': x--; break;
            }
            
if (set.contains(x * 10001 + y)) {
                return true;
            }
            set.add(x * 10001 + y);
        }
        return false;
    }
}
