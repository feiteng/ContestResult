class Solution {
 public boolean isPathCrossing(String path) {
        Set<String> loc = new HashSet<>();


        int x = 0;
        int y = 0;
        loc.add(x + "," + y);
        for (int i = 0; i < path.length(); i++) {
            char ch = path.charAt(i);
            if (ch == 'N') {
                y++;
            } else if (ch == 'E') {
                x++;
            } else if (ch == 'S') {
                y--;
            } else {
                x--;
            
}

            String key = x + "," + y;
            if (loc.contains(key)) return true;
            loc.add(key);
        }
        return false;

    }

}
