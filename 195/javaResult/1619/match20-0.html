class Solution {
    public boolean isPathCrossing(String path) {
        HashSet<String> set = new HashSet();
        int[] cur = new int[]{0, 0};
        set.add(cur[0] + "_" + cur[1]);
        for(char c : path.toCharArray()) {
            if(c == 'N') {
                cur[1] += 1;
            } else if(c == 'S') {
                cur[1] -= 1;
            } else if(c == 'E') {
                cur[0] += 1;
            } else if(c == 'W') {
                cur[0] -= 1;
            }
            String s = cur[0] + "_" + cur[1];
            if(set.contains(s)) {
                return true;
            }
            set.add(s);
        }
        return false;
    }
}
