class Solution {
    public boolean isPathCrossing(String path) {
        Set<Integer> st = new HashSet<>();
        int x = 0, y = 0;
        st.add(0);
        for (int i = 0; i < path.length(); ++i) {
            char ch = path.charAt(i);
            if (ch == 'N') --x;
            else if (ch == 'S') ++x;
            else if (ch == 'E') ++y;
            else --y;
            int mark = x * 10000 + y;
            if (st.contains(mark)) return true;
            st.add(mark);
        }
        return false;
    }
}
