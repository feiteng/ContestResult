class Solution {
    public boolean isPathCrossing(String path) {
        int x = 0,y = 0;
        Set<Integer> p = new HashSet<>();
        p.add(x* 10001 + y);
        
for(char c: path.toCharArray()){
            if(c == 'N'){
                y++;
            } else if(c == 'S'){
                y --;
            }else if(c == 'E'){
                x++;
            }else if(c == 'W'){
                x--;
            }
            int cur =x* 10001 + y;
            //System.out.println(x+" "+y);
            
if(p.contains(cur)){
                
                return true;
            }
            p.add(cur);
        }
        return false;
    }
}
