class Solution {
    public boolean isPathCrossing(String path) {
        int max = (int) Math.pow(10, 4);
        
int x = max, y = max;
        Set<Integer> set = new HashSet<>();
        set.add(x*(2*max+1)+y);
        for (char dir : path.toCharArray()) {
            
switch (dir) {
            case 'N':
                y++;
                break;
            case 'S':
                y--;
                break;
            case 'E':
                x++;
                break;
            case 'W':
                x--;
                break;
            }
            int hashcode = x*(2*max+1)+y;
            if (set.contains(hashcode)) {
                return true;
            }
            set.add(hashcode);
        }
        return false;
    }
}
