class Solution {
    
    static class Point {
        int x;
        int y;
        Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
        Point(Point p) {
            this.x = p.x;
            this.y = p.y;
        }
        
        public boolean equals(Object o) {
            Point p = (Point) o;
            return p.x == this.x && p.y == this.y;
        }
    }
    
    
public boolean isPathCrossing(String path) {
        Set<Integer> set = new HashSet<>();
        // Point cnt = new Point(0, 0);
        int[] p = new int[]{0,0};
        set.add(0);
        for (char c : path.toCharArray()) {
            if (c == 'N') {
                p[0] -= 1;
            } else if (c == 'S') {
                p[0] += 1;
            } else if (c == 'W') {
                p[1] -= 1;
            } else if (c == 'E') {
                p[1] += 1;
            }
            int z = p[0] * 10007 + p[1];
            if (set.contains(z)) {
                return true;
            }
            set.add(z);
        }    
        return false;
    }
}
