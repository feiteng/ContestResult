class Solution {
    public boolean isPathCrossing(String path) {
        Set<String> set = new HashSet();
        int i = 0, j = 0;
        set.add(i + "#" + j);
        for(char c : path.toCharArray()){
            if(c == 'N'){
                i--;
            }else if(c == 'S'){
                i++;
            }else if(c == 'E'){
                j++;
            }else{
                j--;
            }
            if(set.contains(i + "#" + j)){
                return true;
            }else{
                set.add(i + "#" + j);
            }
        }
        return false;
    }
}
