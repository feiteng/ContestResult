class Solution {
    public boolean isPathCrossing(String path) {
        HashSet<String> visited = new HashSet<>();
        visited.add("0,0");
        int r = 0, c = 0;
        for (char c2 : path.toCharArray()) {
            switch (c2) {
                case 'N': r++; break;
                case 'S': r--; break;
                case 'W': c--; break;
                case 'E': c++; break;
            }
            String key = r + "," + c;
            if (visited.contains(key)) {
                return true;
            }
            visited.add(key);
        }
        return false;
    }
}
