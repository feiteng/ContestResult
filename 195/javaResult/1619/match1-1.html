class Solution {
    public boolean isPathCrossing(String path) {
        Set<String> set = new HashSet<>();
        int i = 0;
        int j = 0;
        set.add(0 + "," + 0);
        for(char c : path.toCharArray()){
            if(c == 'N'){
                i++;
            }else if(c == 'S'){
                i--;
            }else if(c == 'E'){
                j++;
            }else if(c == 'W'){
                j--;
            }
            
if(set.contains(i + "," + j)){
                return true;
            }
            set.add(i + "," + j);
        }
        return false;
    }
}
