class Solution {
    public boolean isPathCrossing(String path) {
        HashSet<String> visited = new HashSet();
        int x = 0;
        int y = 0;
        visited.add("0,0");
        for(char c : path.toCharArray()){
            if(c=='S'){
                y--;
            }
            else if(c=='N'){
                y++;
            }
            else if(c=='E'){
                x++;
            }
            else{
                x--;
            }
            String cur = x+","+y;
            if(visited.contains(cur)){
                return true;
            }
            visited.add(cur);
        }
        return false;
    }
}
