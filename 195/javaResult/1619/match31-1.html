class Solution {
    public boolean isPathCrossing(String path) {
        int x = 0, y = 0;
        HashSet<String> coords = new HashSet<>();
        coords.add("0#0");
        for (char c : path.toCharArray()){
            if (c == 'N') x++;
            else if (c == 'S') x--;
            else if (c == 'E') y++;
            else y--
;
            String coord = x + "#" + y;
            if (coords.contains(coord)) return true;
            coords.add(coord);
        }
        return false;
    }
}
