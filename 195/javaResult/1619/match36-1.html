class Solution {
    public boolean isPathCrossing(String path) {
        return sln1(path);
    }
    
    
private boolean sln1(String path){
        Set<String> set =new HashSet<>();
        int x = 0; int y = 0;
        set.add(x+","+y);
        for(int i =0;i<path.length();i++){
            char c= path.charAt(i);
            if(c == 'N'){
                y++;
            }else if(c=='S'){
                y--;
            }else if(c == 'E'){
                x++;
            }else if(c=='W'){
                x--;
            
}
            String key = x+","+y;
            if(set.contains(key)) return true;
            set.add(key);
        }
        return false;
    }
}
