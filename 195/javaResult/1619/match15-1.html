class Solution {
    public boolean isPathCrossing(String path) {
        Set<String> visited = new HashSet<>();
        int x = 0, y = 0;
        visited.add(x + ":" + y);
        for (char c : path.toCharArray()) {
            if (c == 'N') {
                x --;
            } else if (c == 'S') {
                x ++;
            } else if (c == 'W') {
                y --;
            } else {
                y ++;
            }
            
            if (!visited.add(x + ":" + y)) {
                return true;
            }
        }
        
        return false;
    }
}
