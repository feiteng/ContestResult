class Solution {
    public int RANGE = 10002;
    
    public boolean isPathCrossing(String path) {
        int x = 0, y = 0;
        Set<Integer> set = new HashSet<>();
        set.add(0);
        
        for (int i = 0; i < path.length(); i++) {
            char c = path.charAt(i);

            switch (c) {
                case 'N':
                    y++;
                    break;
                case 'S':
                    y--;
                    break;
                case 'E':
                    x++;
                    break;
                case 'W':
                    x--;
                    break;
            }
            
            int tmp = x * RANGE + y;
            
            if (set.contains(tmp)) {
                return true;
            }
            
            set.add(tmp);
        }
        
        return false;
    }
}
