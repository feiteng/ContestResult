class Solution {
    public boolean isPathCrossing(String path) {
        int x = 0, y = 0;
        HashSet <String> set = new HashSet();
        set.add("0#0");
        for(char c : path.toCharArray())
        {
            if(c == 'N') y++;
            else if(c == 'S') y--;
            else if(c == 'E') x++;
            else x--;
            String tmp = x + "#" + y;
            if(set.contains(tmp)) return true;
            set.add(tmp);
        }
        return false;
    }
}
