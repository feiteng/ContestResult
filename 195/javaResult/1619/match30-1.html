class Solution {
    public boolean isPathCrossing(String path) {
        int hor = 0;
        int ver = 0;
        
        Set<String> set = new HashSet();
        set.add(hor + "_" + ver);
        
        for(char c: path.toCharArray()) {
            if(c == 'N') {
                ver++;
            } else if(c == 'S') {
                ver--;
            } else if(c == 'E') {
                hor++;
            } else { // c == 'W'
                hor--;
            
}
            
            String s = hor + "_" + ver;
            
            if(set.contains(s)) return true;
            set.add(s);
        }
        return false;
    }
}
