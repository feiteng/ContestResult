class Solution {

    public boolean isPathCrossing(String path) {
        HashSet<List<Integer>> set = new HashSet<>();
        int x = 0, y = 0;
        set.add(List.of(0, 0));
        for (char c : path.toCharArray()) {
            if (c == 'N') {
                x++;
            } else if (c == 'S') {
                x--;
            } else if (c == 'E') {
                y++;
            } else {
                y--;
            }
            if (set.contains(List.of(x, y))) {
                return true;
            }
            set.add(List.of(x, y));
        }
        return false;
    }
}
