class Solution {
    public boolean isPathCrossing(String path) {
        Set<Integer> seen = new HashSet<>();
        int i = 0, j = 0;
        seen.add(0);
        for(char c: path.toCharArray()){
            if(c == 'N'){
                j++;
            } else if(c == 'S'){
                j--;
            } else if(c == 'W'){
                i--;
            } else {
                i++;
            }
            int mask = i * 100000 + j;
            // System.out.println(i + " " + j + " " + mask);
            
if(seen.contains(mask)){
                return true;
            }
            seen.add(mask);
        }
        return false;
    }
}
