class Solution {
    public boolean isPathCrossing(String path) {
        Set<Integer> set = new HashSet<>();
        set.add(0);
        
        int i = 0, j = 0;
        for (char c : path.toCharArray()) {
            if (c == 'N') {
                i++;
            }
            else if (c == 'E') {
                j++;
            }
            else if (c == 'S') {
                i--;
            }
            else if (c == 'W') {
                j--;
            }
            if (set.contains(i * 100000 + j)) {
                return true;
            }
            set.add(i * 100000 + j);
        }
        
        return false;
    }
}
