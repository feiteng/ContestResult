import java.util.HashSet;
import java.util.Set;

class Solution {
    public boolean isPathCrossing(String path) {
        int x = 0;
        int y = 0;
        Set<String> paths = new HashSet<>();
        paths.add("0_0");
        for (char c : path.toCharArray()) {
            if (c == 'E') {
                x++;
            } else if (c == 'W') {
                x--;
            } else if (c == 'N') {
                y++;
            } else if (c == 'S') {
                y--;
            }
            String v = x + "_" + y;
            if (paths.contains(v)) {
                return true;
            }
            paths.add(v);
        }
        return false;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().isPathCrossing("NNSWWEWSSESSWENNW"));
    }
}
