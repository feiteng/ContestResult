class Solution {
public:

    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        int q[n + 10];
        int hh = 0, tt = -1;
        int ans = INT_MIN;
        for (int i = 0; i < n; i ++) {
            int x = points[i][0], y = points[i][1];
            // 队头元素出队

            while (hh <= tt && x - points[q[hh]][0] > k) hh ++;
            if (hh <= tt) ans = max(ans, x - points[q[hh]][0] + y + points[q[hh]][1]);
            while (hh <= tt && y - x >= points[q[tt]][1] - points[q[tt]][0]) tt --;
            q[++ tt] = i;
        }
        return ans;
    }
};
