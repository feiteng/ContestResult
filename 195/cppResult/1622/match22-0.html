class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        priority_queue<array<int,3>> q;
        int n=points.size();
        int ans=-1e9;
        for ( int i=0; i<n; i++ ) {
            while ( q.size()&&q.top()[1]+k<points[i][0] ) q.pop();
            if ( q.size() ) {
                ans=max(ans,q.top()[0]+points[i][0]+points[i][1]);
            }
            q.push({points[i][1]-points[i][0],points[i][0],points[i][1]});
        }
        return ans;
    }
};
