class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        deque<int> Q;
        int res = -1e9;
        for (int i = 0; i < n; i++) {
            if (Q.size() == 0) {
                Q.push_back(i);
                continue;
            }
            int x = points[i][0];
            while (!Q.empty()) {
                int idx = Q.front();
                if (x-points[idx][0] > k) {
                    Q.pop_front();
                } else {
                    break;
                }
            }
            
            int y = points[i][1];
            if (!Q.empty()) {
                int idx = Q.front();
                res = max(res, x+y+points[idx][1]-points[idx][0]);
            }
            
            while (!Q.empty()) {
                int idx = Q.back();
                if (y-x >= points[idx][1]-points[idx][0]) {
                    Q.pop_back();
                } else {
                    break;
                }
            }
            
            Q.push_back(i);
        }
        return res;
    }
};
