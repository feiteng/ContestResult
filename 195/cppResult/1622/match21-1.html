class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& a, int k) {
        int l = 0;
        multiset <int> st;

        int ans = -1e9;
        for (int i = 0; i < a.size(); i++) {
            while (a[i][0] - a[l][0] > k) {
                st.erase(st.find(a[l][1] - a[l][0]));
                l++;
            }
            if (st.size()) ans = max(ans, a[i][0] + a[i][1] + *st.rbegin());
            st.insert(a[i][1] - a[i][0]);
        }
        return ans;
    }
};
