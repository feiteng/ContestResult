class Solution {
public:
	int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
		multiset<int> s;
		int n = points.size();
		int pos2 = 0;
		int res = -1e9;
		for (int pos = 0; pos < n; ++pos) {
			int x = points[pos][0];
			int val = -x + points[pos][1];


			while (pos2 < n && points[pos2][0] <= x + k) {
				s.insert(points[pos2][0] + points[pos2][1]);
				++pos2;
			}

			s.erase(s.find(x + points[pos][1]));
			if (!s.empty()) {
				res = max(res, val + *s.rbegin());
			}

		}

		return res;
	}
};
