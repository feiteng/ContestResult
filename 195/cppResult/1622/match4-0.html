class Solution {
public:
  int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
    multiset<int> a;
    int n = points.size();
    int ret = -1000010000;
    for (int i = 0, j = -1; i < n; ++i) {
      while (points[i][0] - points[j + 1][0] > k) {
        ++j;
        a.erase(a.lower_bound(points[j][1] - points[j][0]));
      }
      if (!a.empty()) {
        ret = max(ret, points[i][0] + points[i][1] + *a.rbegin());
      }
      a.insert(points[i][1] - points[i][0]);
    }
    return ret;
  }
};
