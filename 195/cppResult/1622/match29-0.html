class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        deque<pair<int,int> > q;
        q.push_back(make_pair(points[0][1]-points[0][0],points[0][0]));
        int ans=-2e8-5;;
        for(int i=1;i<points.size();i++){
            int pos=points[i][0]-k;
            while(!q.empty()&&q.front().second<pos) q.pop_front();
            if(!q.empty()) ans=max(ans,points[i][0]+points[i][1]+q.front().first);
            int nvalue=points[i][1]-points[i][0];
            while(!q.empty()&&nvalue>=q.back().first) q.pop_back();
            q.push_back(make_pair(nvalue,points[i][0]));
        }
        return ans;
    }
};
