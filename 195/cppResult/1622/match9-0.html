struct tmp1 //运算符重载<
{
    int x;
    int value;
    tmp1(int a, int b) {x = a; value = b;}
    bool operator<(const tmp1& a) const
    {
        return value < a.value; //大顶堆
    }
};

class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        priority_queue<tmp1> pq;
        int first_x = points[0][0];
        int ans = -300000000;
        for (int i = 0; i < points.size(); ++i)
        {
            while (!pq.empty() && points[i][0] - pq.top().x > k)
            {
                pq.pop();
            }
            if (!pq.empty())
            {
                ans = max(ans, points[i][1] + points[i][0] - first_x + pq.top().value);
            }
            pq.push(tmp1(points[i][0], points[i][1] + first_x - points[i][0]));
        }
        return ans;
    }
};
