class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& p, int k) {

      multiset<int> s;
      int n = p.size();
      int start = 0;
      int ma = INT_MIN;
      for (int i=0; i<n; ++i){
        // pop
        int curx = p[i][0];

        while(p[start][0] < curx - k){
          s.erase(s.find(p[start][1] - p[start][0]));
          start++;
        }

        //find

        if (!s.empty()){
          ma = max(ma, p[i][0] + p[i][1] + *s.rbegin());
        }

        //add
        s.insert(-p[i][0] + p[i][1]);
      }
      return ma;

    }
};
