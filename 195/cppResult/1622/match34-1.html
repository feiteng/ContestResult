class Solution {
public:
    
    struct node
    {
        long long x,y;
        long long sum;
        
        bool operator<(const node& a) const
        {
            return sum<a.sum;
        }
    }itm[100005];
    
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {

        long long ans=-1e9+7;
        int cnt=0;
        for(auto tp: points)
        {
            itm[cnt].x = tp[0];
            itm[cnt].y = tp[1];
            itm[cnt].sum = tp[0]+tp[1];
            cnt++;
        }
        
        priority_queue<node> q;

        int cur=0;
        for(int i=0;i<cnt;i++)
        {
            while(cur<cnt && (itm[cur].x-itm[i].x)<=k)
            {
                q.push(itm[cur]);
                cur++;
            }
            
            while(!q.empty() && q.top().x<=itm[i].x)
            {
                q.pop();
            }
            
            if(!q.empty())
            {
                ans=max(ans, q.top().sum+itm[i].y-itm[i].x);
            }
        }
        
        return ans;
    }
};
