class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& p, int k) {
        int n = p.size();
        deque<int> q;
        q.push_back(0);
        int ans = 0xcfcfcfcf;
        for(int i=1; i<n; i++){
            while(q.size() && (p[i][0] - p[q.front()][0] > k)) q.pop_front();
            if(q.size()) ans = max(ans, p[i][1]+p[i][0]+p[q.front()][1]-p[q.front()][0]);
            while(q.size() && (p[i][1]-p[i][0])>(p[q.back()][1]- p[q.back()][0])) q.pop_back();
            q.push_back(i);
        }
        return ans;
    }
};
