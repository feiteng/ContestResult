class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& a, int k) {
        priority_queue<pair<int, int>> f;
        int ans = -999999999;
        for (auto e : a) {
            int x = e[0], y = e[1];
            while (!f.empty() && x - f.top().second > k) f.pop();
            if (!f.empty()) {
                ans = max(ans, y + f.top().first + x);
            }
            f.push(make_pair(y - x, x));
        }
        return ans;
    }
};
