class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        deque<pair<int,int>>q;
        int res = -1e9;
        for(int i = 0;i<points.size();i++){
            while(q.size()&&points[i][0]-q.front().first>k)q.pop_front();
            if(i)res = max(res,points[i][0]+points[i][1]+q.front().second);
            while(q.size()&&points[i][1]-points[i][0]>=q.back().second)q.pop_back();
            q.push_back({points[i][0],-points[i][0]+points[i][1]});
        }
        return res;
    }
};
