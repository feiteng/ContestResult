class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        deque<int> q;
        int top = 0;
        q.clear();
        int res = -2000000001;
        int n = points.size();
        auto p = points;
        for (int i = 0; i < n; ++i) {
            int x = p[i][0], y = p[i][1];
            while (!q.empty()) {
                int id = q.front();
                if (x - p[id][0] > k) {
                    q.pop_front();
                } else {
                    break;
                }
            }
            if (!q.empty()) {
                int id = q.front();
                res = max(res, y + x + p[id][1] - p[id][0]);
            }
            while (!q.empty()) {
                int id = q.back();
                if (p[id][1] - p[id][0] <= y - x) {
                    q.pop_back();
                } else {
                    break;
                }
            }
            q.push_back(i);
        }
        return res;
    }
};
