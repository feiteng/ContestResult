class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& A, int k) {
        typedef pair<int, int> PII;
        deque<PII> dq;
        dq.push_back({A[0][0], A[0][1] - A[0][0]});
        int res = -2e9;

        int n = A.size();
        for (int i = 1; i < n; i ++ )
        {
            while (dq.size() && A[i][0] - dq.front().first > k) dq.pop_front();
            if (dq.size()) res = max(res, dq.front().second + A[i][0] + A[i][1]);
            
            while (dq.size() && dq.back().second <= A[i][1] - A[i][0]) dq.pop_back();
            dq.push_back({A[i][0], A[i][1] - A[i][0]});
        }
        
        return res;
    }
};
