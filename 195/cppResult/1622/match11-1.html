class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& p, int k) {
        int ans = INT_MIN, n = p.size();
        deque<int> stk;
        

        for (int i = 0; i < n; ++i ) {
            while (stk.size() && p[i][0]-p[stk.front()][0] > k) stk.pop_front();
            
            if (stk.size()) ans = max(ans, p[stk.front()][1]+p[i][1]+abs(p[stk.front()][0]-p[i][0]));
            
            while (stk.size() && p[i][1]-p[i][0] >= p[stk.back()][1]-p[stk.back()][0]) {
                stk.pop_back();
            }
            
            stk.push_back(i);
        }
        
        return ans;
    }
};
