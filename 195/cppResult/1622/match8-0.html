bool cmp(const vector<int> &p,const vector<int> &q){
    return p[0]<q[0];
}
class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        sort(points.begin(),points.end(),cmp);
        multiset<int> s;
        int now=0;
        int ans=-1e9;
        for(int i = 0;i<points.size();i++){
             while(points[now][0]+k<points[i][0]){
                 s.erase(s.find(points[now][1]-points[now][0]));
                 now++;
             }
            if(s.size()){
                ans=max(ans,points[i][1]+points[i][0]+*s.rbegin());
            }
            s.insert(points[i][1]-points[i][0]);
        }
        return ans;
    }
};
