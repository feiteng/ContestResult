class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        deque<int> eq;
        int n = points.size();
        eq.push_back(0);
        int ans = -2000000000;
        for (int i = 1; i < n; i++)
        {
            while (!eq.empty() && points[i][0] - points[eq.front()][0] > k) eq.pop_front();
            if (!eq.empty()) {
            int cnt = eq.front();
            int cur = points[i][0] + points[i][1] + points[cnt][1] - points[cnt][0];
            ans = max(cur, ans);}
            while (!eq.empty() && points[i][1] - points[i][0] >= points[eq.back()][1] - points[eq.back()][0]) eq.pop_back();
            eq.push_back(i);
        }
        return ans;
    }
};
