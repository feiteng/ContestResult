class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        map<int, set<int>> m;
        int res = -0x3f3f3f3f;
        int index = 0;
        for (int i = 0; i < points.size(); i++) {
            while (index < points.size() && points[index][0] <= k + points[i][0]) {
                m[points[index][0] + points[index][1]].insert(points[index][0]);
                index++;
            }
            while (m.size() > 0 && *(m.rbegin()->second.rbegin()) <= points[i][0]) {
                m.erase(m.rbegin()->first);
            }
            if (m.size() > 0) {
                res = max(res, points[i][1] - points[i][0] + m.rbegin()->first);
            }
        }
        return res;
    }
};
