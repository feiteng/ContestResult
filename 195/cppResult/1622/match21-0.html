class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        vector<vector<int>>x(points); 
        int n = points.size();
        
        multiset<int> k2;
        int i = 0;
        int ans = INT_MIN;
        for(int j = 0 ; j < n ; j++) {
            while(points[j][0]-points[i][0] > k) {
                k2.erase(k2.find(x[i][1]-x[i][0]));
                i++;
            }
            if(!k2.empty())
                ans = max(ans,*k2.rbegin()+x[j][0]+x[j][1]);
            k2.insert(x[j][1]-x[j][0]);
        }
        return ans;
    }
};
