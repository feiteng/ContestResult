class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        deque<int> monoq;
        int ans = INT_MIN;
        for (int i = 0; i < points.size(); i++) {
            while (!monoq.empty() && points[monoq.front()][0] + k < points[i][0]) {
                monoq.pop_front();
            }
            if (!monoq.empty()) {
                ans = max(ans, points[i][0] + points[i][1] + points[monoq.front()][1] - points[monoq.front()][0]);
            }
            while (!monoq.empty() && 
                   points[monoq.back()][1] - points[monoq.back()][0] < points[i][1] - points[i][0]) {
                monoq.pop_back();
            }
            monoq.push_back(i);
        }
        return ans;
    }
};
