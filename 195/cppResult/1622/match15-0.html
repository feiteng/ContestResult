class Solution {
public:
    
    int get_f(vector<vector<int>>& po , int a,int b){
    return po[a][1] + po[b][1] + (abs(po[a][0]-po[b][0]));
    }
    
    int findMaxValueOfEquation(vector<vector<int>>& po, int k) {
    int n = po.size();
    vector<int> d(n+10);
    int st=0,ed=-1;
    int ans = -1e9;
    for (int i=0;i<n;i++)
        {
            while ( (st<=ed) && ( po[ d[st] ][0] < po[i][0] -k ) )
                st++;
            if ( st<=ed )
                ans = max( ans , get_f( po , d[st] , i) );
            while ( ( st<=ed) && ( po[i][1]-po[i][0] >= po[ d[ed] ][1] - po[ d[ed] ][0] ) )
                ed--;
            d[++ed]= i;
        }
    // reverse(po.begin(),po.end());
    // for (int i=0;i<n;i++)
    //     po[i][0]= - po[i][0];
    // for (int i=0;i<n;i++)
    //     {
    //         while ( (st<=ed) && ( po[ d[st] ][0] < po[i][0] -k ) )
    //             st++;
    //         if ( st<=ed )
    //             ans = max( ans , get_f( po , d[st] , i)  );
    //         while ( ( st<=ed) && ( po[i][1]-po[i][0] >= po[ d[ed] ][1] - po[ d[ed] ][0] ) )
    //             ed--;
    //     }
    return ans;
    }
};
