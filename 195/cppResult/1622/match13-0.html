class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        int ans = INT_MIN;
        deque<pair<int,int>> dq;
        int i = 0;
        dq.push_back({points[i][0]-points[i][1],points[i][0]});
        for(int i=1,j=0;i<n;i++) {
            while(!dq.empty() && dq.front().second<points[i][0]-k) {
                dq.pop_front();
            }
            if(!dq.empty()) {
                ans = max(ans,points[i][0]+points[i][1]-dq.front().first);
            }
            while(!dq.empty() && dq.back().first>=(points[i][0]-points[i][1])) {
                dq.pop_back();
            }
            dq.push_back({points[i][0]-points[i][1],points[i][0]});
        }
        return ans;
    }
};
