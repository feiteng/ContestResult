struct T{
    int x, y;

    T(int X, int Y): x(X), y(Y) { }
    bool operator < (const T& that) const {
        return y - x < that.y - that.x;
    }
};
class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        int ans = INT32_MIN;
        priority_queue<T> q;
        for (int i = 0; i < n; i++) {
            while (!q.empty() && q.top().x + k < points[i][0]) {
                q.pop();
            }
            if (!q.empty()) {
                ans = max(ans, points[i][0] - q.top().x + points[i][1] + q.top().y);
            }
            q.push(T(points[i][0], points[i][1]));
        }
        return ans;
    }
};
