class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        deque<int> dq;
        dq.push_front(n - 1);
        int result = INT_MIN;
        for (int i = n - 2; i >= 0; --i) {
            while (dq.size() > 0 && points[dq.back()][0] - points[i][0] > k) {
                dq.pop_back();
            }
            if (dq.size() > 0) {
                result = max(result, points[i][1] + points[dq.back()][0] - points[i][0] + points[dq.back()][1]);
            }
            while (dq.size() > 0 && points[i][1] + points[i][0] >= points[dq.front()][1] + points[dq.front()][0]) {
                dq.pop_front();
            }
            dq.push_front(i);
        }
        return result;
    }
};
