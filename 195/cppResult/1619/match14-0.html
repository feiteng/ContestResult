class Solution {
public:
    bool isPathCrossing(string path) {
        int x = 0, y = 0;
        unordered_set<int> mp;
        mp.insert(0);
        for(char c : path) {
            if(c == 'N')
                x--;
            if(c == 'S')
                x++;
            if(c == 'E')
                y--;
            if(c == 'W')
                y++;
            int key = (1e4 + 1) * x + y;
            if(mp.count(key))
                return true;
            mp.insert(key);
        }
        return false;
            
    }
};
