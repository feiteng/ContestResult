class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int,int>> s;
        int cur_x = 0, cur_y = 0;
        s.insert(make_pair(cur_x, cur_y));
        for(char c : path){
            if(c == 'N') ++cur_y;
            else if(c == 'S') --cur_y;
            else if(c == 'E') ++cur_x;
            else --cur_x;
            if(s.find(make_pair(cur_x, cur_y)) != s.end())
                return true;
            s.insert(make_pair(cur_x, cur_y));
        }
        return false;
    }
};
