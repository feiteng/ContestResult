class Solution {
public:
    bool isPathCrossing(string st) {
   set<pair<int,int>> s;
   int x = 0,y=0;
   s.insert({0,0});
   for(auto i: st){
   	  
   	  if(i=='S')
   	     y = y-1;
   	     
   	     
   	   if(i=='N')
   	     y = y+1;
   	     
   	     
   	     if(i=='E')
   	       x = x+1;
   	       
   	       
   	     if(i=='W')
   	       x = x-1;
   	       
   	       
   	     if(s.find({x,y})!=s.end())
   	       return true;
  
   	  s.insert({x,y});
   }
   
   return false;
    }
};
