class Solution {
public:
    bool isPathCrossing(string path) {
        int x=0,y=0;
        map<pair<int,int>,bool> mp;
        mp[{x,y}] = true;
        for(auto it:path) {
            if(it=='N') x++;
            else if(it=='S') x--;
            else if(it=='E') y++;
            else y--;
            if(mp[{x,y}]) return true;
            mp[{x,y}] = true;
        }
        return false;
    }
};
