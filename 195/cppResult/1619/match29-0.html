class Solution {
public:
    bool isPathCrossing(string str) {
        set<pair<int, int>> m;
		m.insert({ 0, 0 });
		int x = 0, y = 0;
		for (char c : str) {
			if (c == 'N') {
				y += 1;
			}
			else if (c == 'W') {
				x -= 1;
			}
			else if (c == 'E') {
				x += 1;
			}
			else {
				y -= 1;
			}
			if (m.count({ x, y })) {
				return true;
			}
			m.insert({ x, y });
		}
        return false;
    }
};
