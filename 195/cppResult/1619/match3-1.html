class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> f;
        int x = 0; int y = 0;
        for (char c : path) {
            if (f.count(make_pair(x, y))) return true;
            f.insert(make_pair(x, y));
            if (c == 'N') y++;
            if (c == 'S') y--;
            if (c == 'W') x--;
            if (c == 'E') x++;
        }
        return f.count(make_pair(x, y));
    }
};
