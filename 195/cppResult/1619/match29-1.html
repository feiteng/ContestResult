class Solution {
public:
    bool isPathCrossing(string s) {
        // return false;
        set<pair<int,int> > s1;
        s1.insert({0,0});
        int rl=0,ud=0;
        for(int i=0;i<s.size();i++)

        {
            if(s[i]=='W'){rl--;}
            else if(s[i]=='E'){rl++;}
            else if(s[i]=='N'){ud++;}
            else{ud--;}
            if(s1.find({rl,ud})!=s1.end())
            {
                return true;
            }
            s1.insert({rl,ud});
        }
        return false;
    }
};
