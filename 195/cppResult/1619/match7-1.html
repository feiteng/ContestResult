class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> s;
        int x = 0, y = 0;
        s.insert(make_pair(0, 0));
        for (auto& i : path) {
            if (i == 'N') ++y;
            else if (i == 'S') --y;
            else if (i == 'E') ++x;
            else --x;
            if (s.count(make_pair(x, y))) return 1;
            s.insert(make_pair(x, y));
        }
        return 0;
    }
};
