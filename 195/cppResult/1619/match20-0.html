class Solution {
public:
    bool isPathCrossing(string path) {
        map<pair<int,int>,bool> mp;
        int i=0,j=0;
        mp[{0,0}]=true;
        for(char c:path){
            if(c=='N')
                j++;
            else if(c=='S')
                j--;
            else if(c=='E')
                i++;
            else
                i--;
            if(mp.find({i,j})!=mp.end())
                return true;
            mp[{i,j}]=true;
        }
        return false;
    }
};
