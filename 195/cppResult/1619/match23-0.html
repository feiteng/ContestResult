class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> seen;
        int x = 0, y = 0;
        seen.insert({x, y});
        for(char c: path) {
            if(c == 'N') {
                y += 1;
            } else if(c == 'S') {
                y -= 1;
            } else if(c == 'W') {
                x -= 1;
            } else {
                x += 1;
            }
            if(seen.count({x, y})) return true;
            seen.insert({x, y});
        }
        return false;
    }
};
