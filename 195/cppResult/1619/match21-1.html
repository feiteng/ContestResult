class Solution {
public:
    bool isPathCrossing(string path) {
        int x = 0, y = 0;
        map<pair<int, int>, bool> vis;
        vis[{0, 0}] = true;
        for (auto c : path) {
            if (c == 'N') y++;
            else if (c == 'S') y--;
            else if (c == 'E') x++;
            else if (c == 'W') x--;

            
            if (vis[{x, y}]) return true;
            vis[{x, y}] = true;
        }
        
        return false;
    }
};
