class Solution {
public:
    bool isPathCrossing(string path) {
        unordered_map<int, unordered_set<int>> map;
        map[0].insert(0);
        int i = 0, j = 0;
        for (char c : path) {
            if (c == 'N') j++;
            else if (c == 'S') j--;
            else if (c == 'E') i++;
            else i--;
            if (!map[i].insert(j).second) return true;
        }
        return false;
    }
};
