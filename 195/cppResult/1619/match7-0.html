class Solution {
public:
    bool isPathCrossing(string path) {
        unordered_map<int, unordered_set<int>> M;
        
        int x = 0, y = 0;
        M[0].insert(0);
        for (auto c : path) {
            if (c == 'N') ++y;
            else if (c == 'S') --y;
            else if (c == 'E') ++x;
            else --x;
            if (M[x].count(y)) return true;
            M[x].insert(y);
        }
        return false;
    }
};
