class Solution {
public:
    bool isPathCrossing(string path) {
        
        set<pair<int,int>> st;
        
        st.insert({0,0});
        
        int up = 0,right = 0;
        for(char ch : path)
        {
            if(ch=='N') up++;
            else if(ch=='S')up--;
            else if(ch=='W') right--;
            else right++;
            
            // cout<<up<<" "<<right<<endl;
            
            if(st.count({up,right})) return 1;
            st.insert({up,right});
        }
        
        return 0;
        
    }
};
