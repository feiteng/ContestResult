class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int,int>> g;
        int x = 0, y = 0;
        g.insert({0, 0});
        for(int i = 0; i < path.size(); i ++){

            if (path[i] == 'N'){
                x += 1;
            } else if (path[i] == 'S'){
                x -= 1;
            } else if (path[i] == 'W'){
                y --;
            } else {
                y ++;
            }
            if (g.count({x, y})) return true;
            g.insert({x, y});
        }
        return false;
    }
};
