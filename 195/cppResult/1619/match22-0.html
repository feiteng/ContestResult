class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> se;
        int x = 0, y = 0;
        se.insert({x, y});
        for (auto ch : path) {
            if (ch == 'N') {
                y ++;
            }else if (ch == 'S'){
                y --;
            }else if (ch == 'E'){
                x ++;
            }else if (ch == 'W'){
                x --;
            }
            if (se.count({x, y})) return true;
            se.insert({x, y});
        }
        return false;
    }
};
