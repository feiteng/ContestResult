class Solution {
public:
    bool isPathCrossing(string path) {
        map<pair<int,int>, bool> p;
        int x = 0, y = 0;
        p[{x, y}] = true;
        for(int i = 0; i < path.length(); i++) {
            if(path[i] == 'N')  x++;
            else if(path[i] == 'S') x--;
            else if(path[i] == 'W') y--;
            else if(path[i] == 'E') y++;
            if(p[{x, y}])   return true;
            // cout << x << ' ' << y << endl;
            p[{x, y}] = true;
        }
        return false;
    }
};
