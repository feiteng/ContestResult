class Solution {
public:
    bool isPathCrossing(string path) {
     set<pair<int,int> >st;
        int x=0,y=0;
        st.insert({0,0});
        for(auto ch: path)

        {
            if(ch=='N')
            {
                y++;
            }
            if(ch=='E')
            {
                x++;
            }
            if(ch=='S')
            {
                y--;
            }
            if(ch=='W')
            {
                x--;
            }
            if(st.count({x,y}))
                return true;
            st.insert({x,y});
        }
        return false;
    }
};
