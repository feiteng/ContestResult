class Solution {
public:
    bool isPathCrossing(string path) {
        int x = 0, y = 0;
        set<pair<int,int>> s;
        s.insert({0,0});
        for (char c: path) {
            if (c == 'N') ++x;
            else if (c == 'S') --x;
            else if (c == 'E') ++y;
            else --y;
            if (s.count({x, y})) return true;
            s.insert({x, y});
        }
        return false;
    }
};
