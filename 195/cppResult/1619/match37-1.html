class Solution {
public:
    bool isPathCrossing(string path) {
        map <pair<int,int>,int> mp;
        mp[{0,0}] = 1;
        int x=0,y=0;
        for(auto ch : path){

            if(ch=='N')y++;
            if(ch=='W')x--;
            if(ch=='E')x++;
            if(ch=='S')y--;
            if(mp[{x,y}])return true;
            mp[{x,y}] = 1;
        }
        return false;
    }
};
