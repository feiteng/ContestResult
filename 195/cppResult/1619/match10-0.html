class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> H;
        int x = 0, y = 0;
        H.insert({x, y});
        for (auto c : path)
        {
            if (c == 'N') ++y;
            if (c == 'S') --y;
            if (c == 'E') ++x;
            if (c == 'W') --x;
            if (H.count({x, y})) return true;
            H.insert({x, y});
        }
        return false;
    }
};
