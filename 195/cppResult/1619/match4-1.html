class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> h;
        pair<int, int> pos(0, 0);
        h.insert(pos);
        for (char ch : path) {
            if (ch == 'N')
                pos.first++;
            else if (ch == 'S')
                pos.first--;
            else if (ch == 'W')
                pos.second--;
            else
                pos.second++;
            if (h.find(pos) != h.end())
                return true;
            h.insert(pos);
        }
        return false;
    }
};
