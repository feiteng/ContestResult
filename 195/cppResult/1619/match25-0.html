class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int> > SET;
        int x = 0, y = 0;
        SET.insert({0, 0});
        for (auto it : path) {
            if (it == 'N')
                x++;
            else if (it == 'S')
                x--;
            else if (it == 'E')
                y++;
            else
                y--;
            if (SET.count({x, y}))
                return true;
            SET.insert({x, y});
        }
        return false;
    }
};
