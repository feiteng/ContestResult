class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> S;
        int x = 0, y = 0;
        S.insert({x, y});
        for(int i = 0; i < path.length(); i++){
            if(path[i] == 'N')
                x++;
            else if(path[i] == 'S')
                x--;
            else if(path[i] == 'E')
                y++;
            else
                y--;
            if(S.count({x, y}))
                return true;
            S.insert({x, y});
        }
        return false;
    }
};
