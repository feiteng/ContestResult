class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int, int>> sp;
        int x = 0, y = 0;
        sp.insert({x, y});
        for(char c : path){
            if(c == 'N') y += 1;
            if(c == 'S') y -= 1;
            if(c == 'E') x += 1;
            if(c == 'W') x -= 1;
            if(sp.count({x, y})) return true;
            sp.insert({x, y});
        }
        return false;
    }
};
