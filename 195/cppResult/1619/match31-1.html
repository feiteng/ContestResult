class Solution {
public:
    bool isPathCrossing(string path) {
        map<pair<int,int>, bool>mp;
        int curx = 0, cury = 0;
        mp[{curx, cury}] = 1;
        for(auto c : path){

            if(c == 'N') curx++;
            else if(c == 'S') curx--;
            else if(c == 'E') cury++;
            else if(c == 'W') cury--;
            if(mp[{curx, cury}])
                return 1;
            mp[{curx, cury}] = 1;
        }
        return 0;
    }
};
