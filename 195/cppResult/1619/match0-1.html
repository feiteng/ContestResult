class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int,int>> vis;
        int x=0, y=0;
        vis.insert({x, y});
        for (char ch : path) {
            if (ch == 'N') x++;
            else if (ch == 'E') y++;
            else if (ch == 'S') x--;
            else y--;
            if (vis.count({x, y})) return true;
            vis.insert({x, y});
        }
        return false;
    }
};
