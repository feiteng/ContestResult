class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int,int>> se;
        int x=0, y=0;
        se.insert({x,y});
        for(char p:path)

        {
            if(p=='N')
                x--;
            else if(p=='S')
                x++;
            else if(p=='W')
                y--;
            else
                y++;
            if(se.count({x,y})) return true;
            se.insert({x,y});
        }
        return false;
    }
};
