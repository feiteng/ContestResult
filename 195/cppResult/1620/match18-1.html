class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        unordered_map<int,int> mp;
        for(int i = 0; i < arr.size(); ++i){
            mp[(arr[i] + k * (abs(arr[i])/k + 1))%k]++;
            //cout<<arr[i]%k<<endl;
        }
        for(auto p : mp){
            //cout<<p.first<< " "<<p.second<<endl;
            int b = k - p.first;

            if(p.first == 0 || b == p.first){
                if(p.second % 2 != 0) return false;
            }else{
                if(mp.count(b) == 0 || mp[b] != p.second) return false;;
            }
        }
        return true;
    }
};
