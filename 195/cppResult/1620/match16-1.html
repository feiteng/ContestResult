using namespace std;

class Solution 
{
public:
	bool canArrange(vector<int>& arr, int k) 
	{
		vector<vector<int>>mod(k);
		for (int i = 0; i < arr.size(); i++)
		{
			mod[(arr[i] % k+k)%k].push_back(arr[i]);
		}

		if (mod[0].size() % 2 != 0)
			return false;
		for (int i = 1; i <= k / 2; i++)
		{
			if (mod[i].size() != mod[k - i].size())
				return false;
		}
		return true;
	}
};
