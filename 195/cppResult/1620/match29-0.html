class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> m(k, 0);
        for (int i : arr) m[(k + i % k) % k]++;
        if (m[0] & 1) return false;
        for (int i = 1; i < k; i++) if (m[i] != m[k - i]) return false;
        return (k & 1) ? true : ! (m[k / 2] & 1);
    }
};
