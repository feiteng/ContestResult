class Solution {
public:
    bool canArrange(vector<int>& nums, int k) {
        map<int, int> m;
        for (int i: nums) {
            m[((i % k) + k)%k]++;
        }
        // cout<<m[0]<< ' '<<m[1]<<' '<<m[2];
        if(m[0]%2) return 0;
        for (int i = 1; i < k; i++) {
            if(m[i] != m[k-i]) return 0;
        }
        return 1;
    }
};
