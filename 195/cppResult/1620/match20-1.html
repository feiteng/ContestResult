class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {

        unordered_map<int, int> M;
        for (auto x : arr) {
            int y = x % k;
            if (y < 0) y += k;
            ++M[y];
        }
        
        for (auto &p : M) {

            if (p.first == 0) {
                if (p.second % 2 != 0) return false;
            }
            else {
                if (M[k - p.first] != p.second) return false;
            }
        }
        
        return true;
    }
};
