class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int n = arr.size();
        if (n & 1) return false; 
        unordered_map<int, int> freq; 
        for (int i = 0; i < n; i++) freq[(arr[i] % k+k)%k]++; 
        for (int i = 0; i < n; i++) {     
            int rm =  (arr[i] % k+k)%k;  
            if  (2*rm == k) {             
                if (freq[rm] &1 ) return false; 
            } 

            else if (rm == 0) { 
                if (freq[rm] & 1) return false; 
            }         

            else if (freq[rm] != freq[k - rm])  return false; 
        }
        return true;
    }
};
