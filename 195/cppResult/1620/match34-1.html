class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int>mod(k+1);
        
        for(int i=0;i<n;i++){
            if(arr[i]<0){
                mod[(arr[i]%k + k)%k]+=1;
            }else {
                mod[arr[i]%k]+=1;
            }
            
        }
        for(int i=0;i<k;i++){
            cout<< mod[i] << " ";
        }cout<<endl;
        if(mod[0]&1){
            return false;
        }

        for(int i=1;i<k;i++){
            if(mod[i]!=mod[k-i]){
                return false;
            }
        }
        return true;
    }
};
