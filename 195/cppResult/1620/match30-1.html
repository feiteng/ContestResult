class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        unordered_map<int, int> um;
    for (auto a : arr) {
        um[(a % k + k) % k]++;
    }
    for (auto p : um) {
        if (p.first == 0) {
            if (p.second % 2 == 1) return false;
        } else {
            if (p.second != um[k - p.first]) return false;
        }
    }
    return true;
    }
};
