class Solution {
public:
  bool canArrange(vector<int>& arr, int k) {
    vector<int> cnt(k, 0);
    for (int x : arr) {
      int d = x % k;
      if (d < 0) d += k;
      ++cnt[d];
    }
    
    if (cnt[0] % 2 == 1) return false;
    for (int i = 1; i < k; ++i) {
      if (cnt[i] != cnt[k-i]) return false;
    }
    return true;
  }
};
