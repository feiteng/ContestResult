class Solution {
public:
    bool canArrange(vector<int>& a, int k) {
        int n = a.size();
        vector<int> d(k);
        for (int i = 0; i < n; i++) {
            int m = a[i] % k;
            if (m < 0) m += k;
            d[m]++;
        }
        if (d[0] % 2) return false;
        for (int i = 1; i < k; i++) {
            if ((d[i] + d[k - i]) % 2 != 0) return false;
        }
        return true;
    }
};
