class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int> a(k);
        for(int i = 0; i < k; i++) a[i] = 0;
        for(int i = 0; i < n; i++){
            int b = arr[i] % k;

            if(b < 0) b = b+k;
            a[b]++;
        }
        if(a[0] % 2 != 0) return false;
        for(int i = 1; i*2 < k; i++){
            if(a[i] != a[k-i]) return false;
        }
        return true;
    }
};
