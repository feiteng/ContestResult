class Solution {
public:
  bool canArrange(vector<int>& arr, int k) {
    vector<int> kmod(k);
    for (int a : arr) {
      ++kmod[(a % k + 2 * k) % k];
    }
    
    for (int i = 1; i + i < k; ++i) {
      if (kmod[i] != kmod[k - i]) {
        return false;
      }
    }
    
    if (kmod[0] % 2 > 0) {
      return false;
    }
    
    if (k % 2 == 0 && kmod[k/2] % 2 > 0) {
      return false;
    }
    
    return true;
  }
};
