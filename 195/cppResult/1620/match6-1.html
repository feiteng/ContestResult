class Solution {
public:
    bool canArrange(vector<int>& a, int k) {
      vector<int> b(k);
      for (int x : a) {
        int y = x % k;
        if (y < 0) y += k;
        b[y]++;
      }
      if (b[0] & 1) {
        return false;
      }
      for (int i = 1; i < k; ++i) {
        if (b[i] != b[k - i]) {
          return false;
        }
      }
      return true;
    }
};
