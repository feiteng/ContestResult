class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> c(k);
        for(int a : arr)
            c[(a%k+k)%k]++;
        for(int i=1; i<k/2; ++i)
            if(c[i]^c[k-i])
                return 0;
        if(c[0]&1)
            return 0;
        return 1;
    }
};
