class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int>res(k,0);
        for(int i = 0;i<arr.size();i++){
            res[(arr[i]%k+10*k)%k]++;
        }
        if(res[0]%2!=0)return false;
        for(int i = 1;i<k;i++){
            if(res[k-i] != res[i])return false;
        }
        return true;
        
    }
};
