class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        unordered_map<int,int> m;
        for(int i=0;i<arr.size();i++)
        {
            arr[i]=arr[i]%k;
            if(arr[i]<0)
                arr[i]=arr[i]+k;

            m[arr[i]]++;
        }
        
        // for(int i=0;i<arr.size();i++)
        //     cout<<arr[i]<<" ";
        // int i=0,j=arr.size()-1;
        // while(i<j)
        // {
        //     if((arr[i]+arr[j])%k != 0)
        //         return false;
        //     i++;
        //     j--;
        // }
        
        for(auto i:m)
        {
            if(i.first==0)
            {
                if(i.second%2!=0)
                    return false;
            }
            else {
                if(m[k-i.first]!=i.second)
                    return false;
            }
        }
        
        
        return true;
    }
};
