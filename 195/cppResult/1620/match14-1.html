class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int> cnt(k, 0);
        for (int i = 0; i < n; ++i) {
            arr[i] %= k;
            if (arr[i] < 0)
                arr[i] += k;
            cnt[arr[i]]++;
        }
        if (cnt[0] & 1)
            return false;

        if (k % 2 == 0 && (cnt[k / 2] & 1))
            return false;
        for (int i = 1; i <= k / 2; ++i) {
            if (cnt[i] != cnt[k - i])
                return false;
        }
        return true;
    }
};
