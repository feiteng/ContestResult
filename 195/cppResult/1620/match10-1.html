class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> cnt(k);
        for (auto n: arr) {
            cnt[((n % k) + k) % k] ++;
        }
        if (cnt[0] % 2)
            return false;
        for (int i = 1; i < k; i++) {
            if (cnt[i] != cnt[(k-i) % k])
                return false;
        }
        return true;
    }
};
