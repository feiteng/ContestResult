class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> v(k, 0);
        for(int i : arr) {
            int mod = i % k;
            if(mod < 0) mod += k;
            v[mod]++;
        }
        for(int i = 1, j = k - 1; i < j; i++, j--)
            if(v[i] != v[j]) return false;
        if(k % 2 == 0) return v[k/2] % 2 == 0;
        return true;
    }
};
