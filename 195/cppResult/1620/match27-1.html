class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> vis(k+1, 0);
        for (int i=0; i< arr.size(); ++i)
        {
            int tem = arr[i] % k;

            if (tem<0) tem = tem + k;
            //printf("arr[i]: %d, mod: %d\n", arr[i], tem);
            vis[tem] ++;
        }
        for (int i=1; i<k/2; ++i)
        {
            if (vis[i] != vis[k-i])
                return false;
        }
        //printf("stage1\n");
        if (vis[0] % 2) return false;
        if (k % 2 ==0 && vis[k/2] % 2) return false;
        return true;
    }
};
