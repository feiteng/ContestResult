class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> mods(k, 0);
        for (int x : arr) {
            int tmp = x%k;
            if (tmp < 0) tmp += k;
            mods[tmp]++;
        }
        if (mods[0] % 2 == 1) {
            return false;
        }
        for (int i = 1; i < k; ++i) {
            if (mods[i] != mods[k-i]) {
                return false;
            }
        }
        return true;
    }
};
