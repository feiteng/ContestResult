class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> a(k);
        for(int q:arr)
            ++a[((q%k)+k)%k];
        if(a[0]%2)
            return false;
        for(int i=1;i<k/2;++i)
            if(a[i]!=a[k-i])
                return false;
        if(k%2==0&&a[k/2]%2)
            return false;
        return true;
    }
};
