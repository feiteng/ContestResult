class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> cnt(k);
        for(int x: arr)
            ++cnt[(x%k+k)%k];
        for(int i=1;i<k;++i)
            if(cnt[i] != cnt[k-i])
                return false;
        if(!(k&1) && (cnt[k>>1]&1))
            return false;
        return true;
    }
};
