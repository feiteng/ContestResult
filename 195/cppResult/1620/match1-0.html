class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int cnt[100005];
        fill(cnt,cnt+k,0);
        for(auto it:arr){
            cnt[(it%k+k)%k]++;
        }
        for(int i = 0;i<k;i++){
            if(i==(k-i)%k){
                if(cnt[i]%2)return false;
            }
            else {
                if(cnt[i]!=cnt[(k-i)%k])return false;
            }
        }
        return true;
    }
};
