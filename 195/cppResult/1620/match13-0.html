class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        
        int n=arr.size();
        
        vector<int> rem_arr(k,0);
        
        for(int i=0;i<n;i++)
        {
            int rem=arr[i]%k;
            rem=(rem+k)%k;
            
            rem_arr[rem]++;
        }
        
        if(rem_arr[0]%2==1)     return false;
        
        for(int i=1;i<k;i++)
        {
            if(rem_arr[i]!=rem_arr[k-i])    return false;
        }
        
        return true;
    }
};
