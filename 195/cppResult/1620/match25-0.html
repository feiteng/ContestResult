class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        int n=arr.size();
        vector<int>v(k,0);
        for(int i=0;i<n;++i)
        {
            arr[i]%=k;
            if(arr[i]<0) arr[i]+=k;
            ++v[arr[i]];
        }
        if(v[0]&1) return false;
        if(k%2==0&&v[k/2]%2) return false;
        for(int i=1;i<k;++i)
        {
            if(i*2==k) continue;
            if(v[i]!=v[k-i])
                return false;
        }
        return true;
    }
};
