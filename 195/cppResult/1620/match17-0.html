class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> cnt;
        cnt.resize(k, 0);
        for(int i=0;i<arr.size();i++){
            int x;
            if(arr[i]>=0) x = arr[i]%k;
            else{
                arr[i]=-arr[i];
                x = (k-(arr[i]%k))%k;
            }
            cnt[x]++;
        }
        if(cnt[0]%2!=0) return false;
        for(int i=1;i<k;i++){
            if(cnt[i]!=cnt[k-i]) return false;
        }
        return true;
    }
};
