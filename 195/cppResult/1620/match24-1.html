class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int> f(k, 0);
        for (auto x : arr) {
            ++f[((x % k) + k) % k];
        }
        for (int i = 1; i <= (k / 2); ++i) {
            if (f[i] != f[k - i]) return false;
        }
        if (k % 2 == 0) return f[k / 2] % 2 == 0;
        return true;
    }
};
