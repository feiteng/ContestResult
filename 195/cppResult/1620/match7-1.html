class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        unordered_map<int,int> cnt;
        for(int i:arr) {
            //cout<<i%k<<endl;
            cnt[(i%k+k)%k]++;
        }
        if(cnt[0]%2!=0) return false;
        for(int i=1;i<=k/2;i++){
            if(cnt[i]!=cnt[k-i]) return false;
        }
        return true;
    }
};
