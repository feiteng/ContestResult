class Solution {
public:
int mod = 1e9 + 7;
long long powmod(long long a , long long b) {
    long long ans = 1;

    a = a % mod;
    while (b) {
        if(b & 1) ans = ans * a % mod;
        a = a * a % mod;
        b /= 2;
    }
    return ans;
}
int numSubseq(vector<int>& nums, int target) {
    nums.push_back(1e7);
    sort(nums.begin(), nums.end());

    long long ans = 0;
    for (int i = 0; i < nums.size() - 1; i++) {
        int t1 = upper_bound(nums.begin(), nums.end(), target - nums[i]) - nums.begin();
        if(nums[t1] != target - nums[i]) t1--;
        if(t1 < i) continue;
        t1 -= i;
        ans = (ans + powmod(2, t1)) % mod;
    }
    return ans;
}
};
