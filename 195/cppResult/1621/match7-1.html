class Solution {
public:
    const int mods = 1e9+7;

    long long qpow(long long n,long long b){
        long long res=1;
        while(b){
            if(b&1) res = res*n%mods;
            n= n*n%mods;
            b>>=1;
        }
        return res;
    }
    int n;
    int solve(vector<int>& a,int x){
        int l = 0,r=n-1;
        int mid,res=0;
        while(l<=r){
            mid = (l+r)>>1;
            if(a[mid]>=x){
                r = mid-1;
            }
            else{
                res = mid;
                l = mid+1;
            }
        }
        return res;
    }
    int numSubseq(vector<int>& a, int tar) {
        n = a.size();
        sort(a.begin(),a.end());
        if(n==1){
            if(2*a[0]<=tar) return 1;
            else return 0;
        }

        long long ans = 0;
        for(int i =0;i<n;i++){
            int k = solve(a,tar-a[i]+1);  //last < tar-a[i]
            //cout<<k<<endl;
            if(k-i>=0) ans=(ans+qpow(2,k-i))%mods;
        }
        return ans;
    }
};
