class Solution {
public:
    int numSubseq(vector<int>& a, int tar) {
        int MOD = 1000000007;
        sort(a.begin(), a.end());
        int n = a.size();
        vector<int> f(n+2);
        f[0] = 1; f[1] = 2; int t = 1;
        for (int i = 2; i <= n; i++) {
            t = t * 2 % MOD;
            f[i] = (f[i-1] + t) % MOD;
        }
        int ans = 0;
        int p = 0;
        for (int i = 0; i < n; i++) {
            if (a[i] + a[0] > tar) break;
            while (p < i && a[p+1] + a[i] <= tar) p++;
            while (p >=0 && a[p] + a[i] > tar) p--;
            ans += f[i]; ans %= MOD;
            if (p < i) {
                ans -= f[i-p-1]; ans %= MOD;
                ans = (ans + MOD) % MOD;
            }
             
        }
        return ans;
    }
};
