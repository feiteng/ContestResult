using namespace std;

#define MOD 1000000007

class Solution {
private:

public:
  Solution() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
  }

  int numSubseq(vector<int>& nums, int target) {
    const int n = nums.size();
    sort(nums.begin(), nums.end());

    vector<unsigned int> pow(n);
    pow[0] = 1;
    for (int i = 1; i < n; ++i)
      pow[i] = (pow[i - 1] * 2) % MOD;

    int result = 0;
    int last = n - 1;
    for (int first = 0; first <= last; ++first) {
      while (first <= last && nums[first] + nums[last] > target) --last;
      if (first > last) break;

      result = result + pow[last - first];
      result %= MOD;
    }

    return result;
  }
};

#ifdef LOCALRUN
void run_test(vector<int> n, int t) {
  printf("%d\n", Solution().numSubseq(n, t));
}

int main() {
  run_test({3,5,6,7}, 9);
  return 0;
}
#endif
