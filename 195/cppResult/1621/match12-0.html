#define     mod      1000000007
#define     inf      1e18
#define ll long long

class Solution {
public:
    ll powermod(ll _a,ll _b,ll _m)    {ll _r=1;while(_b){if(_b%2==1)_r=(_r*_a)%_m;_b/=2;_a=(_a*_a)%_m;}return _r;}
    int numSubseq(vector<int>& arr, int k) {
        sort(arr.begin(),arr.end());
        ll ans=0;
        for(int i=0;i<arr.size();i++){
            if(arr[i]*2<=k)
                ans++;
            int idx = upper_bound(arr.begin(),arr.end(),k-arr[i])-arr.begin()-1;
            // cout<<idx<<endl;
            if(idx<=i)
                continue;
            ans+=powermod(2,idx-i,mod);
            ans--;
            ans=(ans+mod)%mod;
        }
        return ans;
    }
};
