class Solution {
    
    const int MOD = 1e9+7;
    
    vector<int> fact;
    
    int add(int a, int b) {
        a += b;
        if (a >= MOD) a -= MOD;
        return a;
    }
    
    int mul(int a, int b) {
        return a * 1ll * b % MOD;
    }
    
    int fastPow(int a, int n) {
        int res = 1;
        while (n) {
            if (n & 1) res = mul(res, a);
            a = mul(a, a);
            n >>= 1;
        }
        return res;
    }
    
public:
    int numSubseq(vector<int>& nums, int target) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        int ans = 0;
        int right = n-1, left = 0;
        while (left <= right) {
            //cout << nums[left] << ' ' << nums[right] << '\n';
            if (nums[left] + nums[right] > target) {
                --right;
                continue;
            }
            int cnt = right - left;
            ans = add(ans, fastPow(2, cnt));
            //cout << fastPow(2, cnt) << '\n';
            ++left;
        }
        return ans;
    }
};
