class Solution {
public:
    #define MOD 1000000007
    #define lli long long int
    lli pov(lli x, lli y, lli p=MOD){
        lli res = 1;	 
        x = x % p; 
        while (y > 0) { 
            if (y & 1) 
                res = (res*x) % p; 
            y = y>>1; 
            x = (x*x) % p; 
        } 
        return res; 
    }
    int numSubseq(vector<int>& a, int tgt) {
        sort(a.begin(), a.end());
        int n = a.size();
        int l = 0, r = n-1, ans = 0;
        while(r>=l) {
            if(a[l]+a[r] > tgt) {
                --r;
                continue;
            }
            ans = (ans + pov(2, r-l)) % MOD;
            ++l;
        }
        return ans;
    }
};
