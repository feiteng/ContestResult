using ll = long long;
const ll mod = 1e9 + 7;

class Solution {
public:
    int numSubseq(vector<int>& a, int target) {
        ll res = 0;
        sort(a.begin(), a.end());
        int n = a.size();
        
        vector<ll> pw2(n+2, 0);
        pw2[0] = 1;
        for (int i=1; i<n+2; i++) {
            pw2[i] = 2LL*pw2[i-1]%mod;
        }
        
        for (int i=0,j=n-1; i<n; i++) {
            while (j>=i && a[i]+a[j]>target) j--;
            if (j<i) break;

            ll cur = pw2[j-i];
            // cout<<i<<": "<<cur<<endl;
            res += cur;
            res %= mod;
            
        }
        
        res %= mod;
        res += mod;
        res %= mod;
        return res;
    }
};
