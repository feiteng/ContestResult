class Solution {
public:
    

    long long pow_mod(long long x, long long n, long long mod) {
        long long res = 1;
        while(n) {
            if(n & 1) res = (res * x) % mod;
            x = (x * x) % mod;
            n >>= 1;
        }
        return res;
    }
    
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        long long mod = 1e9 + 7;
        long long ans = 0;
        for(int i = 0; i < nums.size(); i++) {
            int k = target - nums[i];
            int d = upper_bound(nums.begin() + i, nums.end(), k) - nums.begin() - i - 1;
            // cout<<">>> "<<d<<endl;
            if(d < 0) break;
            ans = (ans + pow_mod(2, d, mod)) % mod;
        }
        
        return int(ans);
        
    }
};
