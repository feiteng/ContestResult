class Solution {
public:
    const int MOD = 1e9 + 7;

    long long fast(long long x, int y){
        long long ret = 1;
        while(y){
            if(y&1) ret=ret*x%MOD;
            y>>=1;
            x=x*x%MOD;
        }
        return ret;
    }

    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        long long ans = 0;
        for(int i=0; i<n; i++){
            int idx = upper_bound(nums.begin(), nums.end(), target-nums[i])-nums.begin()-1;
            if(idx-i >= 0)
                ans += fast(2, idx-i);
        }
        return ans % MOD;
    }
};
