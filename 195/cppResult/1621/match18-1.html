class Solution {
public:

    static const int MOD = 1e9 + 7;
    
    int qmi(int a, int k) {
        int res = 1;
        while (k) {
            if (k & 1) res = (long long) res * a % MOD;
            a = (long long) a * a % MOD;
            k >>= 1;
        }
        return res;
    }
    
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();

        int res = 0;
        for (int i = 0; i < n; i ++) {
            int l = i, r = n - 1;
            while (l < r) {
                int m = l + r + 1 >> 1;
                if (nums[i] + nums[m] <= target) l = m;
                else r = m - 1;
            }
            if (nums[i] + nums[r] > target) break;
            int k = r - i;
            res = (res + qmi(2, k)) % MOD;
            
        }
        return res;
    }
};
