#define ll long long 
ll m=1e9+7;
ll power(ll a,ll b)
{

    ll res=1;
    while(b>0)
    {
        if(b&1) res=(res*a)%m;
        b=b/2;
        a=(a*a)%m;
    }
    return res;
}
class Solution {
public:
    int numSubseq(vector<int>& a, int k) 
    {
        ll n=a.size(),ans=0;
        sort(a.begin(),a.end());
        for(ll i=0;i<a.size();i++) 
        {
            int x=k-a[i];
            ll j=upper_bound(a.begin(),a.end(),x)-a.begin();
            if(j-1>=i) ans=(ans+power(2ll,j-1-i))%m;
        }
        return (int)ans;
    }
};
