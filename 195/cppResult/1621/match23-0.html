class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        int n = nums.size();
        int mod = 1e9 + 7;
        sort(nums.begin(), nums.end());
        vector<int> pre(n + 1, 1);
        for (int i = 1; i <= n; ++i) {
            pre[i] = pre[i - 1] * 2;
            pre[i] %= mod;
        }
        int res = 0;
        for (int i = 0; i < n; ++i) {
            int p = upper_bound(nums.begin(), nums.end(), target - nums[i]) - nums.begin();
            int len = p - i - 1;
            if (len < 0) continue;
            if (len == 0 && nums[i] * 2 > target) continue;
            res += pre[len];
            res %= mod;
        }
        return res;
    }
};
