class Solution {
    const int mod = 1e9 + 7;
    int my_pow(int x) {
        int ret = 1, base = 2;
        while (x) {
            if (x & 1) ret = ret * 1ll * base % mod;
            base = base * 1ll * base % mod;
            x = x >> 1;
        }
        return ret;
    }
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int ans = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] * 2 <= target) ans++;
            int idx = upper_bound(nums.begin(), nums.begin() + i, target - nums[i]) - nums.begin();
            ans += (my_pow(idx) - 1) * 1ll * my_pow(i - idx) % mod;
            ans %= mod;
        }
        return ans;
    }
};
