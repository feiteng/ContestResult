lint mod = 1e9 + 7;
lint mypow(lint a, lint n){
  lint res = 1;
  lint base = a;
  while(n > 0){
    if (n%2) res = res*base%mod;
    base = base*base%mod;
    n = n/2;
  }
  return res;
}

class Solution {
public:
    int numSubseq(vector<int>& a, int t) {
      sort(a.begin(), a.end());
      int n = a.size();
      lint ans = 0;
      for (int i = 0; i<n; ++i){
        int big = t - a[i];
        if (big < a[i]) break;
        
        auto it = upper_bound(a.begin(), a.end(), big);
        int len = (it - a.begin()) - i;
        int can = len - 1;
        ans = (ans + mypow(2, can))%mod;
      }
      return ans;

    }
};
