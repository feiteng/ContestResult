typedef long long ll;
const ll MOD=1e9+7;

long long int binexp(long long int a,long long  int b){
    long long int  x=1, y=a; 
    ll n=MOD;
    while (b > 0) {
        if (b%2 == 1) {
            x = (x*y) % n; // multiplying with base
        }
        y = (y*y) % n; // squaring the base
        b /= 2;
    }
    return x % n;
}

class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        int n=nums.size();
        int ans=0;
        sort(nums.begin(),nums.end());
        for(int i=0; i<n; i++)
        {
            if(nums[i]>target)
                break;
         
            int lo=i,hi=n-1,mid;
            int req=-1;
            while(lo<=hi)
            {
                int mid=lo+(hi-lo)/2;
                if(nums[i]+nums[mid]<=target)
                {
                    req=mid;
                    lo=mid+1;
                }
                else
                    hi=mid-1;
            }
            // db2(i,binexp(2,req-i));
            // db1(req);
            if(req==-1)
                    break;
                ans=(ans+binexp(2,req-i))%MOD;
                // db1(ans);
        }
        // cout<<endl;
        return ans;
    }
};
