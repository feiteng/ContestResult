class Solution {
public:
    static constexpr int mod = 1e9+7;

    int qpow(long long a, int n) {
        long long ret = 1;
        while(n) {
            if(n&1) ret = ret*a%mod;
            a = a*a%mod;
            n >>= 1;
        }
        return ret;
    }
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        long long ans = 0;
        for(int i = 0; i < n; i++) {
            int l = i, r = upper_bound(nums.begin(), nums.end(), target-nums[i])-nums.begin();
            if(r <= l) break;
            int len = r-l;
            long long cur = qpow(2, len-1);
            ans = (ans+cur)%mod;
        }
        return ans;
    }
};
