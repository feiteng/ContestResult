class Solution {
public:


    int numSubseq(vector<int>& nums, int target) {
        
        int mod=1e9+7;
        int n=nums.size();
        unordered_map<int,int> mp;

        mp[0]=1;
        for(int i=1;i<=n;i++){
            
            mp[i]=mp[i-1]<<1;
            mp[i]%=mod;
        }
        sort(nums.begin(),nums.end());

        int mx=target/2;
        int end=upper_bound(nums.begin(),nums.end(),mx)-nums.begin();
        
        if(end==n)return (1<<n)-1;
        int res=0;
        
        for(int i=0;i<end;i++){
            
            int r=upper_bound(nums.begin(),nums.end(),target-nums[i])-nums.begin();

            res+=mp[r-i-1];
            res%=mod;
        }
        return res;
    }
};
