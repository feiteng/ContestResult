const int MOD = 1e9 + 7;
class Solution {
public:
    int numSubseq(vector<int>& a, int target) {
        int n = a.size();
        sort(a.begin(), a.end());
        vector<int> p2(n + 1, 1);

        for (int i = 1; i <= n; ++i) p2[i] = p2[i - 1] * 2 % MOD;
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            int j = upper_bound(a.begin(), a.end(), target - a[i]) - a.begin();
            if (j <= i) continue;
            ret = (ret + p2[j - i - 1]) % MOD;
        }
        return ret;
    }
};
