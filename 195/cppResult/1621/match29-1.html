class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int ans = 0;
        vector<int> twopows;
        twopows.push_back(1);

        int po = 1;
        for (int i = 1; i < nums.size(); i ++)
        {
            po = (po+po) % (1000000007);
            twopows.push_back(po);
        }
        for (int head = 0, tail = nums.size()-1; head <= tail; head ++)
        {
            while (head <= tail && nums[head] + nums[tail] > target) -- tail;
            if (head > tail) break;
            ans = (ans + twopows[tail-head]) % 1000000007;
        }
        return ans;
    }
};
