class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
    int n =nums.size();
    vector<int> f(n);
    int P = 1e9+7;
    f[0]= 1;
    for (int i=1;i<n;i++)
        f[i] = f[i-1]*2ll%P;
    sort(nums.begin(),nums.end());
    int r=n-1;
    int ans=0;
    for (int i=0;i<n;i++)
    {
        while ( ( r>=i ) && ( nums[r] + nums[i] >target ) )
            r--;
        if ( r>=i )
            ans= (ans + f[ r-i ] ) %P;
        else
            break;
    }
    return ans;
    }
};
