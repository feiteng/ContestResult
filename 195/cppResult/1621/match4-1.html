class Solution {
public:
    //递归快速幂（对大素数取模）
    #define MOD 1000000007

    typedef long long ll;
    ll qpow(ll a, ll n)
    {
        if (n == 0)
            return 1;
        else if (n % 2 == 1)
            return qpow(a, n - 1) * a % MOD;
        else
        {
            ll temp = qpow(a, n / 2) % MOD;
            return temp * temp % MOD;
        }
    }
    int numSubseq(vector<int>& nums, int target) {
        long long ans=0;
        sort(nums.begin(),nums.end());

        int len=nums.size();
        for(int i=0;i<len;i++){
            int tmp=target-nums[i];
            if(tmp<nums[i]) break;
            int idx=upper_bound(nums.begin()+i,nums.end(),tmp)-nums.begin()-1;
            ans=(ans+qpow(2,idx-i))%MOD;
        }
        return ans;
    }
};
