class Solution {
public:
	ll power(ll x,ll n){
		ll ans=1;
		while(n){
			if(n&1)
				ans=ans*x%MOD;
			x=x*x%MOD;
			n/=2;
		}
		return ans;
	}
    int numSubseq(vector<int>& nums, int target) {
        sort(all(nums));
        ll ans=0;
        for(int i=0;i<nums.size();++i){
        	int mx=target-nums[i];
        	if(mx<nums[i])
        		break;

        	int indx=upper_bound(all(nums),mx)-nums.begin();
        	ans=(ans+power(2,indx-i-1))%MOD;
        }
        return ans;
    }
};
