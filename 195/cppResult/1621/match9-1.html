class Solution {
public:
    const int M=1e9+7;
    int numSubseq(vector<int>& nums, int target) {
        int n=nums.size();
        sort(nums.begin(), nums.end());
        vector<int> p2(n);
        p2[0]=1;
        for(int i=1; i<n; ++i)
            p2[i]=p2[i-1]*2%M;

        int ans=0;
        for(int i=0, j=n-1; i<n; ++i) {
            while(j>i&&nums[i]+nums[j]>target)
                --j;
            if(nums[i]+nums[j]>target)
                break;
            ans=(ans+p2[j-i])%M;
        }
        return ans;
    }
};
