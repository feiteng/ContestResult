const int MOD = 1e9 + 7;

class Solution {
private:
    int add(int x, int y) {return (x+y) % MOD;}

    int sub(int x, int y) {return add(x+MOD, -y);}
    int mul(int x, int y) {return (x*1ll*y) % MOD;}
    int bp(int x, int y) {
        int ans = 1;
        while (y) {
            if (y&1) ans = mul(ans, x);
            x = mul(x, x);
            y >>= 1;
        }
        return ans;
    }
    
public:
    int numSubseq(vector<int>& a, int k) {
        sort(a.begin(), a.end());
        int n = a.size();
        int ans = 0;
        for (int i=0; i<n; i++) {
            int j = upper_bound(a.begin(), a.end(), k-a[i]) - a.begin() - 1;
            if (j < i) break;
            ans = add(ans, bp(2, j-i));
        }
        return ans;
    }
};
