class Solution {
public:
    const int mod = 1000000007;
    int cnt[100005];
    long long ans;
    

    long long nPow(long long x, int n) {
        long long res = 1;
        while(n) {
            if(n & 1) res = (res * x) % mod;
            x = (x * x) % mod;
            n >>= 1;
        }
        return res;
    }
    
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        memset(cnt, 0, sizeof(cnt));

        int len = nums.size();
        int cc = 0;
        for(int i = 0; i < len; i++) {
            int p = upper_bound(nums.begin(), nums.end(), target - nums[i]) - nums.begin();
            if(p < i) cnt[i] = i - p;
            if(nums[i] > (target / 2)) cc++;
        }
        cc = len - cc;
        ans = nPow(2, len) - len - 1;
        for(int i = 0; i < len; i++) {
            int tmp = nPow(2, cnt[i]) - 1;
            ans = (ans + mod - tmp) % mod;
        }
        ans += cc;
        return ans;
    }
};
