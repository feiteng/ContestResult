class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int md = 1e9+7;
        int n = nums.size();
        int p[n+1];
        p[0] = 1;
        for (int i = 1 ; i <= n ; i++) {
            p[i] = (p[i-1]*2)%md;
        }
        long long ans = 0;
        for (int i = 0 ; i < n ; i++) {
            int idx = upper_bound(nums.begin(), nums.end(), target-nums[i])-nums.begin()-1;
            // cout << idx << endl;
            idx -= i;
            if (idx < 0)
                continue;
            ans = (ans+p[idx])%md;
        }
        return (int)(ans%md);
    }
};
