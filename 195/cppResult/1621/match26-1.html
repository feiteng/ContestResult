#define rep(i,init,n) for(int i=init;i<n;i++)
int pow2[200005];
const int MOD=1e9+7;
void init()
{
    pow2[0]=1;
    rep(i,1,200001)
    {
        pow2[i]=pow2[i-1]*2;
        while(pow2[i]>=MOD)pow2[i]-=MOD;
    }

}
class Solution {
public:
    int numSubseq(vector<int>& nums, int x) {
        sort(nums.begin(),nums.end());
        int n=nums.size();
        int out1=0;
        init();
        rep(i,0,nums.size())
        {
             int cur=nums[i];

             int l=i,r=n-1;
             int ans=-1;
            while(l<=r)
            {
                int mid=(l+r)>>1;
                if(cur+nums[mid]<=x)
                {
                    ans=max(ans,mid);
                    l=mid+1;
                }
                else
                r=mid-1;    
            }
            if(ans==-1)break;
            out1+=pow2[ans-i];
            while(out1>=MOD)out1-=MOD;
        }
        return out1;
    }
};
