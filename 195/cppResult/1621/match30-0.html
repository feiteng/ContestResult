const int mod = 1e9 + 7;
class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        vector<int> arr(n);
        arr[0] = 1;
        for ( int i = 1; i < n; ++i ) arr[i] = arr[i - 1] * 2 % mod;
        int res = 0;
        for ( int i = 0; i < nums.size(); ++i ) {
            if ( nums[i] > target ) break;
            int index = binSearch(nums, target - nums[i], 0, i - 1);
            // cout << i << " " << index << endl;
            
            int xx = nums[i] * 2 <= target;
            if ( index != -1 ) {
                long x = arr[index + 1] - 1;
                // cout << "x: " << x << endl;
                long y = arr[i - index - 1];
                // cout << "y: " << y << endl;
                long z = x * y % (long)mod;
                xx += (int)z;
            }
            
            // cout << xx << endl;
            
            res = (res + xx) % mod;
        }
        return res%mod;
    }
    
    int binSearch(vector<int>& nums, int t, int l, int r) {
        if ( l > r ) return -1;
        while ( l + 1 < r ) {
            int mid = ( r - l ) / 2 + l;
            if ( nums[mid] <= t ) l = mid;
            else r = mid;
        }
        
        if ( nums[r] <= t ) return r;
        else if ( nums[l] <= t ) return l;
        else return -1;
    }
    
};
