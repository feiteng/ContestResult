const int mod = 1e9+7;

class Solution {
public:
    long long qpow(long long a,long long b){
        long long ans=1;

        while(b){
            if(b&1) ans=ans*a%mod;
            a=a*a%mod;
            b>>=1;
        }
        return ans%mod;
    }
    int numSubseq(vector<int>& nums, int target) {
        int n=(int)nums.size();
        sort(nums.begin(),nums.end());
        long long ans=0;
        for(int i=0;i<n;i++){
            int nxt=upper_bound(nums.begin(),nums.end(),target-nums[i])-nums.begin();
            nxt--;
            // cout<<i<<' '<<nxt<<endl;
            if(nxt>=i){
                long long now=qpow(2,nxt-i);
                // now--;//减去空
                if(now<0) now+=mod;
                ans=ans+now;
                ans%=mod;
            }
        }
        return ans;
    }
};
