class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(),nums.end());
        int ans = 0;
        int MOD = 1e9 + 7;
        int powersMod[100002];
        powersMod[0] = 1;
        for (int i = 1; i < 100002; ++i) {
            powersMod[i] = powersMod[i-1] * 2;
            powersMod[i] %= MOD;
        }
        for (int i = 0; i < nums.size(); ++i) {
            int val = nums[i];
            if (val + val > target) break;
            int lb = i, rb = nums.size() - 1;
            while (lb < rb) {
                int mb = (lb+rb+1)/2;
                if (val + nums[mb] > target) {
                    rb = mb - 1;
                } else {
                    lb = mb;
                }
            }
            if (i == lb) {
                ++ans;
            } else {
                int x = lb - i;
                ans += powersMod[x];
            }
            ans %= MOD;
        }
        return ans;
    }
};
