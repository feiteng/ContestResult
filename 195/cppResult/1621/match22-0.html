class Solution {
public:
    int mod = 1e9 + 7;
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        
        long long int res = 0;
        int l = 0, r = n - 1;
        while(l <= r)
        {
            if(nums[l] + nums[r] > target) r -- ;
            else res = (res + qmi(2, r - l)) % mod, l ++ ;
        } 
        return res;
    }
    
    int qmi(long long int a, int k)
    {
        long long int res = 1 % mod;
        while(k)
        {
            if(k & 1) res = res * a % mod;
            a = a *a % mod;
            k >>= 1;
        }
        return res;
    }
};
