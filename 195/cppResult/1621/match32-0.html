class Solution {
public:
    long long int power(long long int x, long long int y, long long int p)  
{  
    long long int res = 1;     
  
    x = x % p;  
   
    if (x == 0) return 0; 
  
    while (y > 0)  
    {  
        // If y is odd, multiply x with result  
        if (y & 1)  
            res = (res*x) % p;  
  
        // y must be even now  
        y = y>>1; // y = y/2  
        x = (x*x) % p;  
    }  
    return res;  
}  
    
    int numSubseq(vector<int>& nums, int target) {
        sort(begin(nums),end(nums));
        int p=target/2;
        long long int ans=0;
        int i=upper_bound(begin(nums),end(nums),p)-begin(nums);
        long long int m=1000000007;
        ans+=(power(2,i,m)-1)%m;
        
        while(i<nums.size())
        {
            p=target-nums[i];
            if(p<=0)
                break;
            int j=upper_bound(begin(nums),end(nums),p)-begin(nums);
            int mid=i-j;
            
            
            if(j==0)
                break;
            
            
            ans+=(((power(2,j,m)-1)%m)*(power(2,mid,m)))%m;
            i++;
            
        }
        return ans%m;
        
        
        
    }
};
