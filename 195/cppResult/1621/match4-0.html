class Solution {
public:
    
    //递归快速幂（对大素数取模）
    const int MOD = 1e9+7;
    typedef long long ll;
    ll qpow(ll a, ll n)
    {
        if (n == 0)
            return 1;
        else if (n % 2 == 1)
            return qpow(a, n - 1) * a % MOD;
        else
        {
            ll temp = qpow(a, n / 2) % MOD;
            return temp * temp % MOD;
        }
    }
    
    
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(),nums.end());
        ll res = 0;
        for(int i = 0;i<nums.size();i++){
            int x = nums[i];
            if(x>target)break;
            int idx = upper_bound(nums.begin(),nums.end(),target-x)-nums.begin();
            if(idx==nums.size())idx--;
            else if(nums[idx]>target-x)idx--;
            if(idx-i<0)break;
            // cout<<i<<"   "<<idx<<endl;
            res = (res+qpow(2,idx-i))%MOD;
        }
        return res;
    }
};
