#define mo (1000000007)
int two[100003];


class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        int n= nums.size();

        two[0] = 1;
        for (int i = 1; i <= n; i++) two[i] = (two[i-1]*2)%mo;
        
        
        sort(nums.begin(),nums.end());
        
        int r = n-1;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            while (r >= i && nums[r]+nums[i] > target) --r;
            if (r < i) break;
            ans += two[r-i];
            ans %= mo;
        }
        return ans;
        
    }
};
