typedef long long LL;

const int mod = 1e9 + 7;

int qmi(int a, int k)
{
    int res = 1;
    while(k)
    {
        if(k & 1)  res = (LL) res * a % mod;
        k >>= 1;
        a = (LL)a * a % mod;
    }
    return res;
}

class Solution {
public:
    int numSubseq(vector<int>& a, int sum) {
        
        a.push_back(-1);
        sort(a.begin(), a.end());
        
        int res = 0;
        for(int i = 1; i < a.size(); i ++)
        {
            int l = 0, r = a.size() - 1;
            while(l < r)
            {
                int mid = l + r + 1 >> 1;
                if(a[i] + a[mid] <= sum)  l = mid;
                else  r = mid - 1;
            }
            if(l >= i)
            {
                int j = l - i;
                res = (res + qmi(2, j)) % mod;
            }
        }
        return res;
    }
};
