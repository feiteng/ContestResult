const int mod = 1e9+7;
class Solution {
public:
    long long qpow(long long a,long long b)
    {
        long long res = 1;
        for(;b;b>>=1){
            if(b&1)res=res*a%mod;
            a=a*a%mod;
        }
        return res;
    }
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(),nums.end());
        long long ans = 0;
        for(int i = 0;i < nums.size();++i){
            int tmp = upper_bound(nums.begin(),nums.end(),target-nums[i])-nums.begin();
            tmp--;
            if(tmp<i)break;
            else{
                ans += qpow(2,tmp-i);
                ans%=mod;
            }
        }    
        return ans;
    }
};
