class Solution {
public:
    int numSubseq(vector<int>& a, int t) {
        int M = 1e9 + 7;
        vector<int> pos(1000000 + 10);
        sort(a.begin(), a.end());
        int n = a.size();
        for (int i = 0; i < n; i++) {
            pos[a[i]] = i;
        }
        for (int i = 1; i < 1000010; i++) {
            if (pos[i] == 0) {
                pos[i] = pos[i - 1];
            }
        }
        long long answer = 0;
        for (int i = 0; i < n && t - a[i] >= a[i]; i++) {
            int p = pos[t - a[i]] - i;
            long long res = 1;
            long long cur = 2;
            for (int j = 0; j <= 21; j++) {
                if (p & (1 << j)) {
                    res = (res * cur) % M;
                }
                cur = (cur * cur) % M;
            }
            answer = (answer + res) % M;
        }
        return answer;
    }
};
