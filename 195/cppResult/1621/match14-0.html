class Solution {
public:
    int numSubseq(vector<int>& nums, int target)
    {
        sort(nums.begin(), nums.end());
        int ret = 0;
        for(int i = 0; i < nums.size(); i++)
        {
            if(nums[i] * 2 <= target)
                ret++;
        }
        
        int lo = 0;
        while(lo < nums.size() && target - nums[lo] >= nums[lo])
        {
            int hiMax = search(nums, target - nums[lo]);
            int hi = hiMax;
            int cnt = hi - lo - 1;
            ret = (ret + pow(cnt + 1) - 1) % 1000000007;
            lo++;
        }
        return ret;
    }
    
int pow(int x){
    long long res;
    if (x == 0) res=1;
    else if (x == 1)    res=2;
    else {
        res = pow(x/2);
        if (x % 2 == 0) 
            res = (res * res) % 1000000007;
        else
            res = (((res*res) % 1000000007) * 2) % 1000000007;
    }
    return res;
}
    
    int search(vector<int>& nums, int target)
    {
        int lo = 0, hi = nums.size() - 1;
        while(lo < hi)
        {
            int mid = lo + (hi - lo + 1) / 2;
            if(nums[mid] > target)
                hi = mid - 1;
            else
                lo = mid;
        }
        return lo;
    }
};
