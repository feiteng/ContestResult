typedef long long LL;

class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int res=0;
        for(int i=0;i<nums.size();++i) {
            if(nums[i]>target || target-nums[i]<nums[i]) break;
            int pos=search(nums, target-nums[i]);
            res=((LL)res+qmi(pos-i))%MOD;
        }
        return res;
    }
private:
    int MOD = 1e9+7;
    int qmi(int x) {
        int res=1, k=2;
        while(x) {
            if(x&1) res=((LL)res*k)%MOD;
            x=x>>1;

            k=((LL)k*k)%MOD;
        }
        return res;
    }
    int search(vector<int>& nums, int x) {
        int l=0, r=nums.size()-1;
        while(l<r) {
            int mid=(l+r+1)>>1;
            if(nums[mid]<=x) l=mid;
            else r=mid-1;
        }
        return r;
    }
};
