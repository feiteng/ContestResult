const int MOD = 1000000007;
using int64 = long long;

int64 power(int64 a, int64 b = MOD - 2) {
    int64 c = 1;
    while (b) {
        if (b & 1)
            c = c * a % MOD;
        b /= 2;
        a = a * a % MOD;
    }
    return c;
}


class Solution {
public:
    
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int64 ans = 0;
        for(int i=0;i<n;i++) {
            int cnt = ( upper_bound(nums.begin()+i, nums.end(), target-nums[i]) - nums.begin() )-i;
            if(cnt>0)
                ans += power(2,cnt-1);
        }
        return ans % MOD;
    }
};
