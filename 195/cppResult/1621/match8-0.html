class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        const int mod = 1000000007;
        int n=nums.size();
        sort(nums.begin(), nums.end());
        vector<int> pow2(n);
        pow2[0]=1;
        for(int i=1;i<n;++i)
            pow2[i] = (pow2[i-1]<<1) % mod;
        int ans = 0;
        for(int j=n-1,i=0;i<n;++i)
        {
            while(j>=i && nums[i] + nums[j] > target)
                --j;
            if(j<i) break;
            ans += pow2[j-i];
            ans %= mod;
        }
        return ans%mod;
    }
};
