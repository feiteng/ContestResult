class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(),nums.end());
        int mod=1000000000+7;
        int p[100010];
        p[0]=1;
        int ret=0;
        for(int i=1;i<=nums.size();i++) 
        {
            p[i]=2*p[i-1];
            p[i]%=mod;
        }
        for(int i=0;i<nums.size();i++){
            //以nums[i]为最小数
            int aim=target-nums[i];
            if(aim<nums[i]) break;
            int l=0,r=nums.size()-1;
            while(l<=r)
            {
                int mid=(l+r)/2;
                if(nums[mid]<=aim) l=mid+1;
                else r=mid-1;
            }
            // printf("num=%d l=%d r=%d\n",nums[i],l,r);
            ret+=p[r-i];
            ret%=mod;
        }
        return ret;
    }
};
