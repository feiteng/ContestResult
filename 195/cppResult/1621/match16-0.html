class Solution {
public:
    long long pow(long long b,long long c){
        long long a=2,ans=1;
        while(b){
            if(b&1)
                ans=ans*a%c;
            a=a*a%c;
            b>>=1;
        }
        return ans;
    }
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(),nums.end());
        int l=0,n=nums.size();
        long long ans=0,mod=1e9+7;
        int r=n-1;
        while(l<=r){
            while(l<=r&&nums[l]+nums[r]>target){
                r--;
            }
            if(l>r)
                break;
            
            ans=(ans+pow(r-l,mod))%mod;
            l++;
        }
        return ans;
    }
};
