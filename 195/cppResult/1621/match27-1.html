const int base = 1000000007;
class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());

        int n = nums.size();
        int z2[100000+10];
        z2[0] = 1;
        for (int i=1; i<=100000; ++i) {
            z2[i] = (z2[i-1]*2) % base;
        }
        long long res = z2[n] -1;
        for (int i=0; i<n; ++i) {
            int x = target - nums[i];
            if (x>= nums[i]) continue;
            auto p = upper_bound(nums.begin(), nums.begin() +i, x);
            int g = p-nums.begin();
            res -= z2[i-g];
        }
        res %= base;
        if (res<0) res+=base;
        return res;
    }
};
