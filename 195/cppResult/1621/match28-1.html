class Solution {
    const int mod = 1e9 + 7;
    long long qmi(int t)
    {

        long long v = 2, ans = 1;
        while(t > 0)
        {
            if(t & 1)ans = ans*v%mod;
            t = t/2;
            v = v*v%mod;
        }
        
        return ans;
    }
    
public:
    int numSubseq(vector<int>& nums, int target) {
        
        sort(nums.begin(), nums.end());
        
        int ans = 0;
        for(int i = 0; i < nums.size(); ++i)
        {
            int a = nums[i];
            int k = upper_bound(nums.begin(), nums.end(), target - a) - nums.begin();
            if(k < i)break;
            if(k == i)
            {
                if(nums[i] + nums[i] <= target)++ans;
                break;
            }
            // printf("%d\t%d\n", nums[i], nums[k]);
            
            ans = (ans + qmi(k - i - 1))%mod;
        }
        
        return ans;
    }
};
