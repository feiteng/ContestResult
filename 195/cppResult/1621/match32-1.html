    long long int mypow(long long int x, long long int k){
        long long int ans = 1;

        while(k){
            if(k&1)ans = (ans*x)%1000000007;
            x= (x*x)%1000000007;
            k>>=1;
        }
        return ans;
    }
    
    int numSubseq(vector<int>& nums, int target) {
        int n = (int)nums.size();
        sort(nums.begin(), nums.end());
        long long int ans = 0;
        int l = 0, r = n-1;
        while(l<=r){
            if(nums[l] + nums[r] > target)r--;
            else{
                ans = (ans + mypow(2, r-l))%1000000007;
                l++;
            }
        }
        return ans;
    }
};
