class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        int mod = 1000000007;
        int n=nums.size();
        vector<int> p(n+1 , 0);
        p[0]=1;
        for (int i=1;i<p.size();i++)(p[i] = p[i-1]*2)%=mod;

        vector<int> c(1000009 , 0);
        for(int x: nums){
            c[x]++;
        }
        vector<int> f = c;
        for (int i=1;i<c.size();i++)(c[i] += c[i-1])%=mod;

        int ans=0;
        for(int i=0;i<f.size(); i++){
            int mi = i;
            int ma = target - mi;
            if(ma < mi)continue;
            ans += (p[f[i]] - 1) * (long long)(p[c[ma]- c[mi]]) % mod;
            ans %=mod;
            
        }
        
        return ans;
    }
};
