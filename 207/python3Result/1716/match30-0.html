class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        n=len(grid)
        m=len(grid[0])
        minne=[[0 for j in range(len(grid[0]))] for i in range(len(grid))]
        maxpo=[[0 for j in range(len(grid[0]))] for i in range(len(grid))]
        minne[0][0]=grid[0][0]
        maxpo[0][0]=grid[0][0]
        for i in range(1,len(grid)):
            minne[i][0]=minne[i-1][0]*grid[i][0]
            maxpo[i][0]=maxpo[i-1][0]*grid[i][0]
        for j in range(1,len(grid[0])):
            minne[0][j]=minne[0][j-1]*grid[0][j]
            maxpo[0][j]=maxpo[0][j-1]*grid[0][j]
        for i in range(1,len(grid)):
            for j in range(1,len(grid[0])):
                minne[i][j]=min(minne[i-1][j]*grid[i][j],minne[i][j-1]*grid[i][j])
                minne[i][j]=min(minne[i][j],maxpo[i][j-1]*grid[i][j])
                minne[i][j]=min(minne[i][j],maxpo[i-1][j]*grid[i][j])
                
                maxpo[i][j]=max(minne[i-1][j]*grid[i][j],minne[i][j-1]*grid[i][j])
                maxpo[i][j]=max(maxpo[i][j],maxpo[i][j-1]*grid[i][j])
                maxpo[i][j]=max(maxpo[i][j],maxpo[i-1][j]*grid[i][j])
        #print(minne)
        #print(maxpo)
        if maxpo[n-1][m-1]<0:
            return -1
        else:
            return maxpo[n-1][m-1]%(10**9+7)
