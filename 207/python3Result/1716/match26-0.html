class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        if not grid:
            return 0
        
        m, n = len(grid), len(grid[0])
        dp1 = [[1]*n for _ in range(m)]
        dp2 = [[1]*n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if not i and not j:
                    dp1[i][j] = grid[i][j]
                    dp2[i][j] = grid[i][j]
                if i == 0 and j != 0:
                    dp1[i][j] = dp1[i][j-1]*grid[i][j]
                    dp2[i][j] = dp2[i][j-1]*grid[i][j]
                if i != 0 and j == 0:
                    dp1[i][j] = dp1[i-1][j] * grid[i][j]
                    dp2[i][j] = dp1[i-1][j] * grid[i][j]
                elif i != 0 and j != 0:
                    dp1[i][j] = max(dp1[i-1][j]*grid[i][j], dp1[i][j-1]*grid[i][j], dp2[i-1][j]*grid[i][j], dp2[i][j-1]*grid[i][j])
                    dp2[i][j] = min(dp1[i-1][j]*grid[i][j], dp1[i][j-1]*grid[i][j], dp2[i-1][j]*grid[i][j], dp2[i][j-1]*grid[i][j])      
        return -1 if dp1[-1][-1] < 0 else dp1[-1][-1]%1000000007
