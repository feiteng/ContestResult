class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m,n = len(grid),len(grid[0])
        minimum = [[1 for _ in range(n)] for _ in range(m)]
        maximum = [[1 for _ in range(n)] for _ in range(m)]
        minimum[0][0]=maximum[0][0]=grid[0][0]

        i=0
        for j in range(1,n):
            minimum[i][j] = minimum[i][j-1]*grid[i][j]  
            maximum[i][j] = maximum[i][j-1]*grid[i][j] 

        j=0
        for i in range(1,m):
            minimum[i][j] = minimum[i-1][j]*grid[i][j]
            maximum[i][j] = maximum[i-1][j]*grid[i][j]                

        for i in range(1,m):                
            for j in range(1,n):
                minimum[i][j] = min(minimum[i-1][j]*grid[i][j],maximum[i-1][j]*grid[i][j],minimum[i][j-1]*grid[i][j],maximum[i][j-1]*grid[i][j])
                maximum[i][j] = max(minimum[i-1][j]*grid[i][j],maximum[i-1][j]*grid[i][j],minimum[i][j-1]*grid[i][j],maximum[i][j-1]*grid[i][j])

        return maximum[-1][-1]%(10**9+7) if maximum[-1][-1]>=0 else -1
