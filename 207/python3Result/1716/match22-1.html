class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        mod = 10 ** 9 + 7
        m, n = len(grid), len(grid[0])
        dp = [[None for _ in range(n)] for _ in range(m)]

        dp[0][0] = [grid[0][0], grid[0]
[0]]

        for i in range(1, m):
            cur = grid[i][0]
            if cur >= 0:
                dp[i][0] = [dp[i-1][0][0]*cur, dp[i-1][0][1]*cur]
            else:
                dp[i][0] = [dp[i-1][0][1]*cur, dp[i-1][0][0]*cur]
        
        for j in range(1, n):
            cur = grid[0][j]
            if cur >= 0:
                dp[0][j] = [dp[0][j-1][0]*cur, dp[0][j-1][1]*cur]
            else:
                dp[0][j] = [dp[0][j-1][1]*cur, dp[0][j-1][0]*cur]           


        for i in range(1,m):
            for j in range(1,n):
                cur = grid[i][j]
                if cur > 0:
                    dp[i][j] = [max
(dp[i][j-1][0]* cur, dp[i-1][j][0]* cur), min(dp[i][j-1][1]*cur, dp[i-1][j][1]*cur)]
                else:
                    dp[i][j] = [max(dp[i][j-1][1]* cur, dp[i-1][j][1]* cur), min(dp[i][j-1][0]*cur, dp[i-1][j][0]*cur)]

        return -1 if dp[m-1][n-1][0] < 0 else (dp[m-1][n-1][0]%mod)
