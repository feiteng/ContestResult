class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        r = len(grid)
        c = len(grid[0])
        
        dp = [[[-sys.maxsize, sys.maxsize] for _ in range(c)] for _ in range(r)]
        
        for i in range(r-1, -1, -1):
            for j in range(c-1, -1, -1):
                if i == r-1 and j == c-1:
                    dp
[i][j][0] = grid[i][j]
                    dp[i][j][1] = grid[i][j]
                elif i == r-1 and j != c-1:
                    dp
[i][j][0] = max(dp[i][j+1][0] * grid[i][j], dp[i][j+1][1] * grid[i][j])
                    dp[i][j][1] = min(dp[i][j+1][0] * grid[i][j], dp[i][j+1][1] * grid[i][j])
                elif i != r-1 and j == c-1:
                    dp
[i][j][0] = max(dp[i+1][j][0] * grid[i][j], dp[i+1][j][1] * grid[i][j])
                    dp[i][j][1] = min(dp[i+1][j][0] * grid[i][j], dp[i+1][j][1] * grid[i][j])
                else:
                    dp
[i][j][0] = max(
                        dp[i+1][j][0] * grid[i][j], 
                        dp[i][j+1][0] * grid[i][j],
                        dp[i+1][j][1] * grid[i][j], 
                        dp[i][j+1][1] * grid[i][j]
                    )
                    dp[i][j][1] = min(
                        dp[i+1][j][0] * grid[i][j], 
                        dp[i][j+1][0] * grid[i][j],
                        dp[i+1][j][1] * grid[i][j], 
                        dp[i][j+1][1] * grid[i][j]
                    )
                    
                    
        # print(dp)
        return -1 if max(dp[0][0]) < 0 else (max(dp[0][0]) % (10**9+7))
