class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        if (grid[m-1][n-1] == 0):
            return 0
        
        pos 
= [[0 for x in range(n)] for y in range(m)]
        neg = [[0 for x in range(n)] for y in range(m)]
        
        pos[0][0] = max(0, grid[0][0])
        neg[0][0] = min(0, grid[0][0])
        
        for i in range(1, m):
            temp = grid[i][0]
            if (temp > 0):
                pos[i][0] = pos[i-1][0] * grid
[i][0]
                neg[i][0] = neg[i-1][0] * temp
            else:
                pos[i][0] = neg[i-1][0] * temp
                neg[i][0] = pos[i-1][0] * temp
                
        for j in range(1, n):
            temp = grid[0][j]
            if (temp > 0):
                pos[0][j] = pos[0][j-1] * temp
                neg[0][j] = neg[0][j-1] * temp
            else:
                pos[0][j] = neg[0][j-1] * temp
                neg[0][j] = pos[0][j-1] * temp
        
        for i in range(1,m):
            for j in range(1,n):
                temp = grid[i][j]
                if (temp > 0):
                    pos[i][j] = max(pos[i-1][j] * temp, pos[i][j-1]*temp)
                    neg[i][j] = min(neg[i-1][j] * temp, neg[i][j-1]*temp)
                else:
                    pos[i][j] = max(neg[i-1][j] * temp, neg[i][j-1]*temp)
                    neg[i][j] = min(pos[i-1][j] * temp, pos[i][j-1]*temp)
        
        if (pos[m-1][n-1] == 0):
            #check 0
            for i in range(m):
                for j in range(n):
                    if (grid[i][j] == 0):
                        return 0
            return -1
        
        return (pos[m-1][n-1] % 1000000007)
