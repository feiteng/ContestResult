class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:

        r,c=len(grid),len(grid[0])
        Max=[[0 for _ in range(c)] for _ in range(r)]
        Min=[[0 for _ in range(c)] for _ in range(r)]
        
        Max[0][0]=Min[0][0]=grid[0][0]
        for i in range(1,r):
            Max[i][0]=grid[i][0]*Max[i-1][0]
            Min[i][0]=grid[i][0]*Min[i-1][0]
        for i in range(1,c):
            Max[0][i]=grid[0][i]*Max[0][i-1]
            Min[0][i]=grid[0][i]*Min[0][i-1]
        
        for i in range(1,r):
            for j in range(1,c):

                temp_max=max(grid[i][j]*Max[i-1][j],grid[i][j]*Max[i][j-1],grid[i][j]*Min[i-1][j],grid[i][j]*Min[i][j-1])
                temp_min=min(grid[i][j]*Max[i-1][j],grid[i][j]*Max[i][j-1],grid[i][j]*Min[i-1][j],grid[i][j]*Min[i][j-1])
                Max[i][j]=temp_max
                Min[i][j]=temp_min

        if Max[-1][-1]<0:
            return -1
        else:
            return Max[-1][-1]%(10**9+7)
