class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        rows,cols = len(grid),len(grid[0])
        dpmax = [ [0 for _ in range(cols)] for _ in range(rows)]
        dpmin = [ [0 for _ in range(cols)] for _ in range(rows)]
        dpmax[0][0],dpmin[0][0] = grid[0][0],grid[0][0]
        
for i in range(1,cols):
            dpmax[0][i] = dpmax[0][i-1]*grid[0][i]
            dpmin[0][i] = dpmin[0][i-1]*grid[0][i]
        
for i in range(1,rows):
            dpmax[i][0] = dpmax[i-1][0]*grid[i][0]
            dpmin[i][0] = dpmin[i-1][0]*grid[i][0]
        for i in range(1,rows):
            for j in range(1,cols):
                dpmax[i][j] = max(dpmax[i-1][j]*grid[i][j],dpmin[i-1][j]*grid[i][j],dpmax[i][j-1]*grid[i][j],dpmin[i][j-1]*grid[i][j])
                dpmin[i][j] = min(dpmax[i-1][j]*grid[i][j],dpmin[i-1][j]*grid[i][j],dpmax[i][j-1]*grid[i][j],dpmin[i][j-1]*grid[i][j])               
        if dpmax[rows-1][cols-1] < 0:
            return -1
        else:
            return dpmax[rows-1][cols-1]%(pow(10,9)+7)
