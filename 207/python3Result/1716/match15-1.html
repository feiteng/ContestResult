class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        
        # max prod in array 的2d 扩展
        
        m, n = len(grid), len(grid[0])
        max_prod 
= [[0] * n for i in range(m)]
        min_prod = [[0] * n for i in range(m)]
        
        max_prod[0][0] = grid[0][0] 
        min_prod[0][0] = grid[0][0] 
        for i in range(1, m):
            # j = 0
            if grid[i][0] >= 0:
                max_prod[i][0] = max_prod[i-1][0] * grid[i][0]
                min_prod[i][0] = min_prod[i-1][0] * grid[i][0]
            else:
                max_prod[i][0] = min_prod[i-1][0] * grid[i][0]
                min_prod[i][0] = max_prod[i-1][0] * grid[i][0]
                
        for j in range(1, n):
            # i = 0
            if grid[0][j] >= 0:
                max_prod[0][j] = max_prod[0][j-1] * grid[0][j]
                min_prod[0][j] = min_prod[0][j-1] * grid[0][j]
            else:
                max_prod[0][j] = min_prod[0][j-1] * grid[0][j]
                min_prod[0][j] = max_prod[0][j-1] * grid[0][j]
                
        for i in range(1, m):
            for j in range(1, n):
                if grid[i][j] >= 0:
                    max_prod[i][j] = max(max_prod[i][j-1], max_prod[i-1][j]) * grid[i][j]
                    min_prod[i][j] = min(min_prod[i][j-1], min_prod[i-1][j]) * grid[i][j]
                else:
                    max_prod[i][j] = min(min_prod[i][j-1], min_prod[i-1][j]) * grid[i][j]
                    min_prod[i][j] = max(max_prod[i][j-1], max_prod[i-1][j]) * grid[i][j]

                
        return max_prod[m-1][n-1] % (10**9 + 7) if max_prod[m-1][n-1] >= 0 else -1
