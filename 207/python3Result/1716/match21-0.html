class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        
        m, n = len(grid), len(grid[0])
        dp = [[(0, 0) for _ in range(n)] for _ in range(m)]
        
        dp[0][0] = (grid[0][0], grid[0][0])
        for i in range(1, m):
            dp[i][0] = (grid[i][0] * dp[i - 1][0][0], grid[i][0] * dp[i - 1][0][0])
        
        for i in range(1, n):
            dp[0][i] = (grid[0][i] * dp[0][i - 1][0], grid[0][i] * dp[0][i - 1][0])
            
        #print(dp)
        for i in range(1, m):
            for j in range(1, n):
                bigP = max(dp[i - 1][j][0] * grid[i][j], dp[i - 1][j][1] * grid[i][j], dp[i][j - 1][0] * grid[i][j], dp[i][j - 1][1] * grid[i][j])
                smallN = min(dp[i - 1][j][0] * grid[i][j], dp[i - 1][j][1] * grid[i][j], dp[i][j - 1][0] * grid[i][j], dp[i][j - 1][1] * grid[i][j])
                dp[i][j] = (bigP, smallN)
        
        return dp[-1][-1][0] % ( 10 ** 9 + 7 ) if dp[-1][-1][0] >= 0 else -1
