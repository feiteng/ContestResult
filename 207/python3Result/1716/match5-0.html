class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])
        dp_min = [[0] * m for _ in range(n)]
        dp_max = [[0] * m for _ in range(n)]
        dp_min[0][0] = dp_max[0][0] = grid[0][0]
        for i in range(1, n):
            dp_min[i][0] = dp_max[i][0] = dp_min[i-1][0] * grid[i][0]
        for j in range(1, m):
            dp_min[0][j] = dp_max[0][j] = dp_min[0][j-1] * grid[0][j]
        
        for i in range(1, n):
            for j in range(1, m):
                dp_min[i][j] = min(dp_max[i-1][j] * grid[i][j], dp_max[i][j-1] * grid[i][j], dp_min[i-1][j] * grid[i][j], dp_min[i][j-1] * grid[i][j])
                dp_max[i][j] = max(dp_max[i-1][j] * grid[i][j], dp_max[i][j-1] * grid[i][j], dp_min[i-1][j] * grid[i][j], dp_min[i][j-1] * grid[i][j])
        if dp_max[-1][-1] < 0:
            return -1
        else:
            return dp_max[-1][-1] % (pow(10, 9) + 7)
