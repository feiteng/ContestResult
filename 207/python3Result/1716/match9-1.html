class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        rowCount 
= len(grid)
        columnCount = len(grid[0])

        minimums = [[1] * columnCount for _ in range(rowCount)]
        maximums = [[1] * columnCount for _ in range(rowCount)]
        minimums[0][0] = grid[0][0]
        maximums
[0][0] = grid[0][0]

        for j in range(1, columnCount):
            maximums[0][j] = max(maximums[0][j - 1] * grid[0][j], minimums[0][j - 1] * grid[0][j])
            minimums[0][j] = min(maximums[0][j - 1] * grid[0][j], minimums[0][j - 1] * grid[0][j])

        for i in range(1, rowCount):
            maximums[i][0] = max(maximums[i - 1][0] * grid[i][0], minimums[i - 1][0] * grid[i][0])
            minimums[i][0] = min(maximums[i - 1][0] * grid[i][0], minimums[i - 1][0] * grid[i][0])

        for i in range(1, rowCount):
            for j in range(1, columnCount):
                maximums[i][j] = max(
                    maximums[i - 1][j] * grid[i][j],
                    minimums[i - 1][j] * grid[i][j],
                    maximums[i][j - 1] * grid[i][j],
                    minimums[i][j - 1] * grid[i][j],
                )
                minimums[i][j] = min(
                    maximums[i - 1][j] * grid[i][j],
                    minimums[i - 1][j] * grid[i][j],
                    maximums[i][j - 1] * grid[i][j],
                    minimums[i][j - 1] * grid[i][j],
                )

        if maximums[-1][-1] < 0:
            return -1
        else:
            return maximums[-1][-1] % (10 ** 9 + 7)
