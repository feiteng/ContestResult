class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        M = 10**9+7
        
        m 
= len(grid)
        n = len(grid[0])
        max_ = [[None for j in range(n)] for i in range(m)]
        min_ = [[None for j in range(n)] for i in range(m)]
        
        ## initialization
        max_[m-1][n-1] = grid[m-1][n-1]
        min_[m-1][n-1] = grid[m-1][n-1]
        
        for i in range(m-2, -1, -1):
            max_[i][n-1] = grid[i][n-1] * max_[i+1][n-1]
            min_[i][n-1] = grid[i][n-1] * min_[i+1][n-1]
            
        for j in range(n-2, -1, -1):
            max_[m-1][j] = grid[m-1][j] * max_[m-1][j+1]
            min_[m-1][j] = grid[m-1][j] * min_[m-1][j+1]
            
#         for i in range(m):
#             print(max_[i])
            
#         print('')
        
#         for i in range(m):
#             print(min_[i])

        for i in range(m-2, -1, -1):
            for j in range(n-2, -1, -1):
                if grid[i][j] == 0:
                    max_[i][j] = 0
                    min_[i][j] = 0
                elif grid[i][j] > 0:
                    max_[i][j] = grid[i][j] * max(max_[i+1][j], max_[i][j+1])
                    min_[i][j] = grid[i][j] * min(min_[i+1][j], min_[i][j+1])
                else:
                    max_[i][j] = grid[i][j] * min(min_[i+1][j], min_[i][j+1])
                    min_[i][j] = grid[i][j] * max(max_[i+1][j], max_[i][j+1])
                    
        
#         for i in range(m):
#             print(max_[i])
            
#         print('')
        
#         for i in range(m):
#             print(min_[i]) 
        if max_[0][0]>=0:
            return max_[0][0] % M
        else:
            return -1
