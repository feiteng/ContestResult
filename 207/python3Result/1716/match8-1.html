class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        if grid[0][0] == 0 or grid[-1][-1] == 0: return 0
        n, m = len(grid), len
(grid[0])
        max_value = [[0] * m for _ in range(n)]
        min_value = [[0] * m for _ in range(n)]
        max_value[0][0] = min_value[0][0] = grid[0][0]
        for i in range(1, n):
            max_value[i][0] = max(max_value[i-1][0]*grid[i][0], min_value[i-1][0]*grid[i][0])
            min_value[i][0] = min(max_value[i-1][0]*grid[i][0], min_value[i-1][0]*grid[i][0])
        for i in range(1, m):
            max_value[0][i] = max(max_value[0][i-1]*grid[0][i], min_value[0][i-1]*grid[0][i])
            min_value[0][i] = min(max_value[0][i-1]*grid[0][i], min_value[0][i-1]*grid[0][i])
        
        for i in range(1, n):
            for j in range(1, m):
                max_value[i][j] = max(max_value[i-1][j]*grid[i][j], min_value[i-1][j]*grid[i][j], max_value[i][j-1]*grid[i][j], min_value[i][j-1]*grid[i][j])
                min_value[i][j] = min(max_value[i-1][j]*grid[i][j], min_value[i-1][j]*grid[i][j], max_value[i][j-1]*grid[i][j], min_value[i][j-1]*grid[i][j])

        return max_value[-1][-1] % 1000000007 if max_value[-1][-1] >= 0 else -1
