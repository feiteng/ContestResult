class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        dp1 = [[-10**18]*m for _ in range(n)]
        dp2 = [[10**18]*m for _ in range(n)]
        dp1[0][0] = grid[0][0]
        dp2[0][0] = grid[0][0]
        
        for i in range(n):
            for j in range(m):
                if i>0:
                    dp1[i][j] = max(dp1[i][j], grid[i][j]*dp1[i-1][j], grid[i][j]*dp2[i-1][j])
                
                if j>0:
                    dp1[i][j] = max(dp1[i][j], grid[i][j]*dp1[i][j-1], grid[i][j]*dp2[i][j-1])
                
                if i>0:
                    dp2[i][j] = min(dp2[i][j], grid[i][j]*dp1[i-1][j], grid[i][j]*dp2[i-1][j])
                
                if j>0:
                    dp2[i][j] = min(dp2[i][j], grid[i][j]*dp1[i][j-1], grid[i][j]*dp2[i][j-1])
        
        if dp1[i][j]<0:
            return -1
        else:
            return dp1[i][j]%(10**9+7)
