class Solution:
    def maxProductPath(self, G: List[List[int]]) -> int:
        n,m = len(G),len(G[0])
        d = [[0] * m for _ in range(n)]
        e = [[0] * m for _ in range(n)]
        d[0][0] = G[0][0]
        e[0][0] = G[0][0]
        for i in range(1,n):
            d
[i][0] = d[i-1][0] * G[i][0]
            e[i][0] = d[i-1][0] * G[i][0]
        
for i in range(1,m):
            d[0][i] = d[0][i-1] * G[0][i]
            e[0][i] = d[0][i-1] * G[0][i]
        for i in range(1,n):
            for j in range(1,m):
                d
[i][j] = max(d[i][j-1]*G[i][j],d[i-1][j]*G[i][j],e[i][j-1]*G[i][j],e[i-1][j]*G[i][j])
                e[i][j] = min(d[i][j-1]*G[i][j],d[i-1][j]*G[i][j],e[i][j-1]*G[i][j],e[i-1][j]*G[i][j])
        # print(d)
        if d[-1][-1] < 0:
            return -1
        else:
            return d[-1][-1] % (10 ** 9 + 7)
