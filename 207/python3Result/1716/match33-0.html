class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        dp = [[[None, None] for __ in range(m)] for __ in range(n)]
        dp[0][0] = [grid[0][0], grid[0][0]]
        for i in range(1, n):
            dp[i][0] = [dp[i-1][0][0]*grid[i][0], dp[i-1][0][0]*grid[i][0]]
        # print(dp)
        for i in range(1, m):
            dp[0][i] = [dp[0][i-1][0]*grid[0][i], dp[0][i-1][0]*grid[0][i]]
        # print(dp)
        for i in range(1, n):
            for j in range(1, m):
                dp[i][j][0] = min(dp[i-1][j][0]*grid[i][j], dp[i][j-1][0]*grid[i][j], dp[i-1][j][1]*grid[i][j], dp[i][j-1][1]*grid[i][j])
                dp[i][j][1] = max(dp[i-1][j][0]*grid[i][j], dp[i][j-1][0]*grid[i][j], dp[i-1][j][1]*grid[i][j], dp[i][j-1][1]*grid[i][j])
        p = dp[-1][-1][1]
        return -1 if p<0 else dp[-1][-1][1]%(10**9 + 7)
