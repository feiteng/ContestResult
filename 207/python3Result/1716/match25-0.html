class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        
        dp1 = [[0]*m for _ in range(n)]
        dp2 = [[0]*m for _ in range(n)]
        
        for i in range(n):
            for j in range(m):                
                if i==0 and j==0:
                    dp1[i][j] = grid[i][j]
                    dp2[i][j] = grid[i][j]
                elif i==0:
                    dp1[i][j] = grid[i][j]*dp1[i][j-1]
                    dp2[i][j] = grid[i][j]*dp2[i][j-1]
                elif j==0:
                    dp1[i][j] = grid[i][j]*dp1[i-1][j]
                    dp2[i][j] = grid[i][j]*dp2[i-1][j]
                else:
                    dp1[i][j] = min(grid[i][j]*dp1[i][j-1], grid[i][j]*dp1[i-1][j], grid[i][j]*dp2[i][j-1], grid[i][j]*dp2[i-1][j])
                    dp2[i][j] = max(grid[i][j]*dp1[i][j-1], grid[i][j]*dp1[i-1][j], grid[i][j]*dp2[i][j-1], grid[i][j]*dp2[i-1][j])
        print(dp1,dp2)
        if max(dp1[-1][-1], dp2[-1][-1]) < 0:
            return -1
        else:
            return max(dp1[-1][-1], dp2[-1][-1])%(10**9+7)
