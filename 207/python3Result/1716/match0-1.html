INF = float('inf')
MOD = 10 ** 9 + 7
class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        R 
= len(grid)
        C = len(grid[0])
        mx = [[-INF] * C for _ in range(R)]
        mi = [[INF] * C for _ in range(R)]
        mx[0][0] = grid[0][0]
        mi
[0][0] = grid[0][0]
        for i in range(1, R):
            mx[i][0] = max(mx[i - 1][0] * grid[i][0], mi[i - 1][0] * grid[i][0])
            mi[i][0] = min(mx[i - 1][0] * grid[i][0], mi[i - 1][0] * grid[i][0])
        for j in range(1, C):
            mx[0][j] = max(mx[0][j - 1] * grid[0][j], mi[0][j - 1] * grid[0][j])
            mi[0][j] = min(mx[0][j - 1] * grid[0][j], mi[0][j - 1] * grid[0][j])
        for i in range(1, R):
            for j in range(1, C):
                mx[i][j] = max(mx[i][j - 1] * grid[i][j], mi[i][j - 1] * grid[i][j], mx[i - 1][j] * grid[i][j], mi[i - 1][j] * grid[i][j])
                mi[i][j] = min(mx[i][j - 1] * grid[i][j], mi[i][j - 1] * grid[i][j], mx[i - 1][j] * grid[i][j], mi[i - 1][j] * grid[i][j])
        return -1 if mx[R - 1][C - 1] < 0 else mx[R - 1][C - 1] % MOD
