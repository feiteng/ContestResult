class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        max_dp = [[1]*n for _ in range(m)]
        min_dp = [[1]*n for _ in range(m)]
        max_dp[0][0] = grid[0][0]
        min_dp[0][0] = grid[0][0]
        token = (10**9+7)
        for j in range(1, n):
            max_dp[0][j] = grid[0][j] * max_dp[0][j-1]
            min_dp[0][j] = grid[0][j] * min_dp[0][j-1]
        for i in range(1, m):
            for j in range(n):
                if j == 0:
                    max_dp[i][j] = (grid[i][j] * max_dp[i-1][j]) #% token
                    min_dp[i][j] = (grid[i][j] * min_dp[i-1][j]) #% token
                else:
                    max_dp[i][j] = max(grid[i][j] * max_dp[i-1][j], grid[i][j] * max_dp[i][j-1], grid[i][j] * min_dp[i-1][j], grid[i][j] * min_dp[i][j-1]) #% token
                    min_dp[i][j] = min(grid[i][j] * max_dp[i-1][j], grid[i][j] * max_dp[i][j-1], grid[i][j] * min_dp[i-1][j], grid[i][j] * min_dp[i][j-1]) #% token
        
        maxval = max_dp[-1][-1]#, max(min_dp[-1]))
#         for row in max_dp:
#             print(row)
            
#         for row in min_dp:
#             print(row)
        if maxval < 0:
            return -1
        return maxval % token
