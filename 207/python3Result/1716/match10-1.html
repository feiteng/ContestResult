class Solution:
    def maxProductPath(self, arr: List[List[int]]) -> int:
        N=len(arr)
        M=len(arr[0])
        
        mp = [[0 for i in range(M)] for j in range(N)]; 
        mpp = [[0 for i in range(M)] for j in range(N)]; 

        for i in range(N): 
            for j in range(M): 
                miv = sys.maxsize; 
                mav = -sys.maxsize; 
                if (i == 0 and j == 0): 
                    mav = arr[i][j]; 
                    miv = arr[i][j]; 
                    
                if (i > 0): 
                    tempMax = max(mp[i - 1][j] * arr[i][j], mpp[i - 1][j] * arr[i][j]); 
                    mav = max(mav, tempMax); 

                    tempMin = min(mp[i - 1][j] * arr[i][j],mpp[i - 1][j] * arr[i][j]); 
                    miv = min(miv, tempMin); 
                    
                if (j > 0): 
                    tempMax = max(mp[i][j - 1] * arr[i][j],mpp[i][j - 1] * arr[i][j]); 
                    mav = max(mav, tempMax); 

                    tempMin = min(mp[i][j - 1] * arr[i][j],mpp[i][j - 1] * arr[i][j]); 
                    miv = min(miv, tempMin); 
                    
                mp[i][j] = mav; 
                mpp[i][j] = miv; 

        if(mp[N - 1][M - 1] < 0):
            return -1
        return mp[N - 1][M - 1]%(1000000007);
