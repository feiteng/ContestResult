class Solution:
  def maxProductPath(self, grid: List[List[int]]) -> int:
    n, m = len(grid), len(grid[0])
    dp = [[[0, 0] for _ in range(m)] for _ in range(n)]
    
    dp[0][0] = [grid[0][0], grid[0][0]]
    
    for i in range(1, n):
      dp[i][0] = [grid[i][0] * dp[i-1][0][0], grid[i][0] * dp[i-1][0][1]]
    
    for j in range(1, m):
      dp[0][j] = [grid[0][j] * dp[0][j-1][0], grid[0][j] * dp[0][j-1][1]]
    
    for i in range(1, n):
      for j in range(1, m):
        dp[i][j] = [
          min(grid[i][j] * dp[i-1][j][0], grid[i][j] * dp[i][j-1][0], grid[i][j] * dp[i-1][j][1], grid[i][j] * dp[i][j-1][1]),
          max(grid[i][j] * dp[i-1][j][0], grid[i][j] * dp[i][j-1][0], grid[i][j] * dp[i-1][j][1], grid[i][j] * dp[i][j-1][1])
        ]
    
    return dp[-1][-1][1] % (10**9 + 7) if dp[-1][-1][1] >= 0 else -1
