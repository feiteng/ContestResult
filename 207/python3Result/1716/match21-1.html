class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m,n=len(grid),len(grid[0])
        dp={}
        dp
[(0,0)]=(grid[0][0],grid[0][0])
        for i in range(1,m):
            dp[(i,0)]=(dp[(i-1,0)][0]*grid[i][0],dp[(i-1,0)][0]*grid[i][0])
        for j in range(1,n):
            dp[(0,j)]=(dp[(0,j-1)][0]*grid[0][j],dp[(0,j-1)][0]*grid[0][j])
        for i in range(1,m):
            for j in range(1,n):
                max_=max(dp[(i-1,j)][0]*grid[i][j],dp[(i-1,j)][1]*grid[i][j],dp[(i,j-1)][0]*grid[i][j],dp[(i,j-1)][1]*grid[i][j])
                min_=min(dp[(i-1,j)][0]*grid[i][j],dp[(i-1,j)][1]*grid[i][j],dp[(i,j-1)][0]*grid[i][j],dp[(i,j-1)][1]*grid[i][j])
                dp[(i,j)]=(max_,min_)
        ans=dp[(m-1,n-1)][0]
        return -1 if ans<0 else ans%(10**9+7)
