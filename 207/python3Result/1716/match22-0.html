class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        grid[0][0]=[grid[0][0],grid[0][0]]
        rows, cols=len(grid), len(grid[0])
        for i in range(1, rows):
            cur=grid[i][0]
            if cur>0:
                grid[i][0]=[cur*grid[i-1][0][0], cur*grid[i-1][0][1]]
            else:
                grid[i][0]=[cur*grid[i-1][0][1], cur*grid[i-1][0][0]]
        for i in range(1, cols):
            cur=grid[0][i]
            if cur>0:
                grid[0][i]=[cur*grid[0][i-1][0], cur*grid[0][i-1][1]]
            else:
                grid[0][i]=[cur*grid[0][i-1][1], cur*grid[0][i-1][0]]
        for i in range(1, rows):
            for j in range(1, cols):
                cur=grid[i][j]
                if cur==0:
                    grid[i][j]=[0,0]
                elif cur>0:
                    grid[i][j]=[cur*min(grid[i-1][j][0], grid[i][j-1][0]), cur*max(grid[i-1][j][1], grid[i][j-1][1])]
                else:
                    grid[i][j]=[cur*max(grid[i-1][j][1], grid[i][j-1][1]), cur*min(grid[i-1][j][0], grid[i][j-1][0])]
        return grid[-1][-1][1]%(10**9+7) if grid[-1][-1][1]>=0 else -1
