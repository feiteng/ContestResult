class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        dp = [[[float('inf'), -float('inf')] for j in range(n)] for i in range(m)]
        
        dp[0][0][0] = grid[0][0]
        dp[0][0][1] = grid[0][0]
        for j in range(1, n):
            dp[0][j][0] = min(dp[0][j-1][0]*grid[0][j], dp[0][j-1][1]*grid[0][j])
            dp[0][j][1] = max(dp[0][j-1][0]*grid[0][j], dp[0][j-1][1]*grid[0][j])
            
        for i in range(1, m):
            dp[i][0][0] = min(dp[i-1][0][0]*grid[i][0], dp[i-1][0][1]*grid[i][0])
            dp[i][0][1] = max(dp[i-1][0][0]*grid[i][0], dp[i-1][0][1]*grid[i][0])
        
        zero = False
        for i in range(1, m):
            for j in range(1, n):
                dp[i][j][0] = min(dp[i][j-1][0]*grid[i][j], dp[i][j-1][1]*grid[i][j], dp[i-1][j][0]*grid[i][j], dp[i-1][j][1]*grid[i][j])
                dp[i][j][1] = max(dp[i][j-1][0]*grid[i][j], dp[i][j-1][1]*grid[i][j], dp[i-1][j][0]*grid[i][j], dp[i-1][j][1]*grid[i][j])
                
                if grid[i][j] == 0:
                    zero = True
        #print(dp)
        
        if dp[-1][-1][1] >= 0:
            return dp[-1][-1][1]%(10**9 + 7)
        if zero:
            return 0
        return -1
