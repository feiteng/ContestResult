class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m = 1e9+7
        
        rows 
= len(grid)
        cols = len(grid[0])
        dp1 = [[None] * cols for _ in range(rows)]
        dp2 = [[None] * cols for _ in range(rows)]
        
        for i in range(rows):
            for j in range(cols):
                if i==0 and j==0:
                    dp1[i][j] = grid[i][j]
                    dp2[i][j] = grid[i][j]
                elif i==0:
                    dp1[i][j] = dp1[i][j-1]*grid[i][j]
                    dp2[i][j] = dp2[i][j-1]*grid[i][j]
                elif j==0:
                    dp1[i][j] = dp1[i-1][j]*grid[i][j]
                    dp2[i][j] = dp2[i-1][j]*grid[i][j]
                else:
                    dp1[i][j] = min(dp1[i-1][j]*grid[i][j], dp1[i][j-1]*grid[i][j], dp2[i-1][j]*grid[i][j], dp2[i][j-1]*grid[i][j])
                    dp2[i][j] = max(dp1[i-1][j]*grid[i][j], dp1[i][j-1]*grid[i][j], dp2[i-1][j]*grid[i][j], dp2[i][j-1]*grid[i][j])
        # print(dp1)
        # print(dp2)
        
        ans = dp2[-1][-1]
        return int(dp2[-1][-1]%m) if ans >= 0 else -1
