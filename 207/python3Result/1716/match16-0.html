class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        if not grid:
            return 0
        m = len(grid)
        n = len(grid[0])
        max_positive = [[float('-inf')] * n for _ in range(m)]
        min_negative = [[float('inf')] * n for _ in range(m)]
        max_positive[-1][-1] = grid[-1][-1]
        min_negative[-1][-1] = grid[-1][-1]
        for i in range(m - 2, -1, -1):
            max_positive[i][n - 1] = grid[i][n - 1] * max_positive[i + 1][n - 1]
            min_negative[i][n - 1] = grid[i][n - 1] * min_negative[i + 1][n - 1]
        for j in range(n - 2, -1, -1):
            max_positive[m - 1][j] = grid[m - 1][j] * max_positive[m - 1][j + 1]
            min_negative[m - 1][j] = grid[m - 1][j] * min_negative[m - 1][j + 1]
        
        for i in range(m - 2, -1, -1):
            for j in range(n - 2, -1, -1):
                if grid[i][j] == 0:
                    max_positive[i][j] = 0
                    min_negative[i][j] = 0
                elif grid[i][j] > 0:
                    max_positive[i][j] = grid[i][j] * max(max_positive[i + 1][j], max_positive[i][j + 1])
                    min_negative[i][j] = grid[i][j] * min(min_negative[i + 1][j], min_negative[i][j + 1])
                else:
                    max_positive[i][j] = grid[i][j] * min(min_negative[i + 1][j], min_negative[i][j + 1])
                    min_negative[i][j] = grid[i][j] * max(max_positive[i + 1][j], max_positive[i][j + 1])
        return int(max_positive[0][0] % (1e9 + 7)) if max_positive[0][0] >= 0 else -1
