class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        import math
        const = 1000000007
        rows, cols = len(grid), len(grid[0])
        f = [[[-10, -10] for jj in range(cols)] for ii in range(rows)]
        if grid[0][0] >= 0:
            f[0][0][0] = grid[0][0]
        else:
            f[0][0][1] = -grid[0][0]
        for i in range(rows):
            for j in range(cols):
                if i == 0 and j == 0:
                    continue
                if i == 0:
                    if grid[i][j] < 0:
                        f[i][j][0] = f[i][j-1][1] * (-grid[i][j])
                        f[i][j][1] = f[i][j-1][0] * (-grid[i][j])
                    else:
                        f[i][j][0] = f[i][j-1][0] * (grid[i][j])
                        f[i][j][1] = f[i][j-1][1] * (grid[i][j])
                    continue
                if j == 0:
                    if grid[i][j] < 0:
                        f[i][j][0] = f[i-1][j][1] * (-grid[i][j])
                        f[i][j][1] = f[i-1][j][0] * (-grid[i][j])
                    else:
                        f[i][j][0] = f[i-1][j][0] * (grid[i][j])
                        f[i][j][1] = f[i-1][j][1] * (grid[i][j])
                    continue
                if grid[i][j] < 0:
                    f[i][j][0] = max(f[i-1][j][1], f[i][j-1][1]) * (-grid[i][j])
                    f[i][j][1] = max(f[i-1][j][0], f[i][j-1][0]) * (-grid[i][j])
                else:
                    f[i][j][0] = max(f[i-1][j][0], f[i][j-1][0]) * (grid[i][j])
                    f[i][j][1] = max(f[i-1][j][1], f[i][j-1][1]) * (grid[i][j])
        if f[rows-1][cols-1][0] >= 0:
            return f[rows-1][cols-1][0] % const
        else:
            return -1
