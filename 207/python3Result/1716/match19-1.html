class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n 
= len(grid), len(grid[0])
        
        f = [[[0, 0] for _ in range(n)] for _ in range(m)]  #f[i][j][0]: 最大正积, f[i][j][1]: 最大负积
        
        if grid[0][0] > 0:
            f[0][0][0] = grid[0][0]
        else:
            f[0][0][1] = grid[0][0]
        
        for i in range(1, n):
            
if grid[0][i] > 0: 
                f[0][i][0] = f[0][i - 1][0] * grid[0][i]
                f[0][i][1] = f[0][i - 1][1] * grid[0][i]
            else:
                f[0][i][0] = f[0][i - 1][1] * grid[0][i]
                f[0][i][1] = f[0][i - 1][0] * grid[0][i]
        
        for i in range(1, m):
            
if grid[i][0] > 0: 
                f[i][0][0] = f[i - 1][0][0] * grid[i][0]
                f[i][0][1] = f[i - 1][0][1] * grid[i][0]
            else:
                f[i][0][0] = f[i - 1][0][1] * grid[i][0]
                f[i][0][1] = f[i - 1][0][0] * grid[i][0]
                
        for i in range(1, m):
            for j in range(1, n):
                
if grid[i][j] > 0: 
                    f[i][j][0] = max(f[i - 1][j][0], f[i][j - 1][0]) * grid[i][j]
                    f[i][j][1] = min(f[i - 1][j][1], f[i][j - 1][1]) * grid[i][j]
                else:
                    f[i][j][0] = min(f[i - 1][j][1], f[i][j - 1][1]) * grid[i][j]
                    f[i][j][1] = max(f[i - 1][j][0], f[i][j - 1][0]) * grid[i][j]
                    
        # for i in f:
        #     print(i)
        
        if f[m - 1][n - 1][0] == 0:
            if all(grid[i][j] != 0 for i in range(m) for j in range(n)):
                return -1
            
        return f[m - 1][n - 1][0] % (10 ** 9 + 7)
