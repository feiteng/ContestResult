class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        M = 10**9 + 7
        m = len(grid)
        n = len(grid[0])
        minv = [[0]*n for _ in range(m)]
        maxv = [[0]*n for _ in range(m)]
        minv[0][0] = maxv[0][0] = grid[0][0]
        for k in range(1, n):
            minv[0][k] = min(minv[0][k-1]*grid[0][k], maxv[0][k-1]*grid[0][k])
            maxv[0][k] = max(minv[0][k-1]*grid[0][k], maxv[0][k-1]*grid[0][k])
        for k in range(1, m):
            minv[k][0] = min(minv[k-1][0]*grid[k][0], maxv[k-1][0]*grid[k][0])
            maxv[k][0] = max(minv[k-1][0]*grid[k][0], maxv[k-1][0]*grid[k][0])
        for k in range(1, m):
            for v in range(1, n):
                minv[k][v] = min(maxv[k-1][v] * grid[k][v], maxv[k][v-1]*grid[k][v], minv[k-1][v] * grid[k][v], minv[k][v-1]*grid[k][v])
                maxv[k][v] = max(maxv[k-1][v] * grid[k][v], maxv[k][v-1]*grid[k][v], minv[k-1][v] * grid[k][v], minv[k][v-1]*grid[k][v])
        return maxv[-1][-1] % M if maxv[-1][-1] >= 0 else -1
