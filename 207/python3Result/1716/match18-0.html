class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        lo = [[0] * n for _ in range(m)]
        hi = [[0] * n for _ in range(m)]
        lo[0][0] = grid[0][0]
        hi[0][0] = grid[0][0]
        for i in range(1, m):
            lo[i][0] = min(lo[i-1][0] * grid[i][0], hi[i-1][0] * grid[i][0])
            hi[i][0] = max(lo[i-1][0] * grid[i][0], hi[i-1][0] * grid[i][0])
            
        for j in range(1, n):
            lo[0][j] = min(lo[0][j-1] * grid[0][j], hi[0][j-1] * grid[0][j])
            hi[0][j] = max(lo[0][j-1] * grid[0][j], hi[0][j-1] * grid[0][j])
        
        for i in range(1, m):
            for j in range(1, n):
                lo[i][j] = min(lo[i-1][j] * grid[i][j], 
                               hi[i-1][j] * grid[i][j], 
                               lo[i][j-1] * grid[i][j], 
                               hi[i][j-1] * grid[i][j])
                hi[i][j] = max(lo[i-1][j] * grid[i][j], 
                               hi[i-1][j] * grid[i][j], 
                               lo[i][j-1] * grid[i][j], 
                               hi[i][j-1] * grid[i][j])
        res = max(lo[m-1][n-1], hi[m-1][n-1])
        if res < 0:
            return -1
        return res % (10 ** 9 + 7)
