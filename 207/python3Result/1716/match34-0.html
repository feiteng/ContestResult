class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        pos = [[0 for x in range(n)] for y in range(m)]
        neg = [[0 for x in range(n)] for y in range(m)]
        pos[0][0] = max(-1, grid[0][0])
        neg[0][0] = min(1, grid[0][0])
        for i in range(1, m):
            tmp = grid[i][0]
            if (tmp > 0):
                pos[i][0] = pos[i-1][0] * tmp
                neg[i][0] = neg[i-1][0] * tmp
            else:
                pos[i][0] = neg[i-1][0] * tmp
                neg[i][0] = pos[i-1][0] * tmp
        for j in range(1, n):
            tmp = grid[0][j]
            if (tmp > 0):
                pos[0][j] = pos[0][j-1] * tmp
                neg[0][j] = neg[0][j-1] * tmp
            else:
                pos[0][j] = neg[0][j-1] * tmp
                neg[0][j] = pos[0][j-1] * tmp
        
        for i in range(1, m):
            for j in range(1, n):
                tmp = grid[i][j]
                if (tmp > 0):
                    pos[i][j] = max(pos[i-1][j] * tmp, pos[i][j-1] * tmp)
                    neg[i][j] = min(neg[i-1][j] * tmp, neg[i][j-1] * tmp)
                else:
                    pos[i][j] = max(neg[i-1][j] * tmp, neg[i][j-1] * tmp)
                    neg[i][j] = min(pos[i-1][j] * tmp, pos[i][j-1] * tmp)
        if (pos[m-1][n-1] < 0):
            return -1
        else:
            return (pos[m-1][n-1]%1000000007)
