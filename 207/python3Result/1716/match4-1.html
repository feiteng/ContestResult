class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        big = [[0] * n for i in range(m)]
        small = [[0] * n for i in range(m)]
        big[0][0] = small[0][0] = grid[0][0]
        for i in range(1, m):
            big[i][0] = small[i][0] = big[i - 1][0] * grid[i][0]
        for j in range(1, n):
            big[0][j] = small[0][j] = big[0][j - 1] * grid[0][j]
        for i in range(1, m):
            for j in range(1, n):
                big[i][j] = max(big[i - 1][j] * grid[i][j], big[i][j - 1] * grid[i][j], small[i - 1][j] * grid[i][j], small[i][j - 1] * grid[i][j])
                small[i][j] = min(big[i - 1][j] * grid[i][j], big[i][j - 1] * grid[i][j], small[i - 1][j] * grid[i][j], small[i][j - 1] * grid[i][j])
        if big[-1][-1] < 0:
            return -1
        return big[-1][-1] % (10 ** 9 + 7)
