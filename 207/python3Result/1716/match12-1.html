def maxProductPath(arr): 
    n 
= len(arr)
    m = len(arr[0])
    maxp = [[0 for j in range(m)] for i in range(n)]
    minp = [[0 for j in range(m)] for i in range(n)] 
    for i in range(n): 
        for j in range(m): 
            mini = 10000000000 
            maxi = -10000000000 
            
if (i==0 and j==0): 
                maxi = arr[i][j]
                mini = arr[i][j]
            if (i>0): 
                tn = min(maxp[i-1][j]*arr[i][j],minp[i-1][j]*arr[i][j])
                mini = min(mini,tn)
                tm = max(maxp[i-1][j]*arr[i][j],minp[i-1][j]*arr[i][j]) 
                maxi = max(maxi,tm) 
            if (j>0):  
                tn = min(maxp[i][j-1]*arr[i][j],minp[i][j-1]*arr[i][j])
                mini = min(mini,tn)
                tm = max(maxp[i][j-1]*arr[i][j],minp[i][j-1]*arr[i][j])
                maxi = max(maxi,tm)
            maxp[i][j] = maxi; 
            minp[i][j] = mini; 
    return maxp[n-1][m-1]; 
class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        x = maxProductPath(grid)
        return x%(10**9+7) if x>=0 else -1
