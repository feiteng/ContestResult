class Solution:
    # dp, time O(HW), space O(HW)
    def maxProductPath(self, A: List[List[int]]) -> int:
        INF = float('inf')
        H, W = len(A), len(A[0])
        small = [[INF] * W for _ in range(H)]
        large = [[-INF] * W for _ in range(H)]
        has_zero = False
        for i in range(H):
            if has_zero:
                break
            for j in range(W):
                if A[i][j] == 0:
                    has_zero = True
                    break
        small[0][0] = large[0][0] = A[0][0]
        for i in range(1, H):
            small[i][0] = min(small[i - 1][0] * A[i][0], large[i - 1][0] * A[i][0])
            large[i][0] = max(small[i - 1][0] * A[i][0], large[i - 1][0] * A[i][0])
        for j in range(1, W):
            small[0][j] = min(small[0][j - 1] * A[0][j], large[0][j - 1] * A[0][j])
            large[0][j] = max(small[0][j - 1] * A[0][j], large[0][j - 1] * A[0][j])
        for i in range(1, H):
            for j in range(1, W):
                small[i][j] = min(small[i - 1][j] * A[i][j], small[i][j - 1] * A[i][j], large[i - 1][j] * A[i][j], large[i][j - 1] * A[i][j])
                large[i][j] = max(small[i - 1][j] * A[i][j], small[i][j - 1] * A[i][j], large[i - 1][j] * A[i][j], large[i][j - 1] * A[i][j])
        res = max(small[-1][-1], large[-1][-1])
        # print(small, large, res)
        if res < 0:
            return 0 if has_zero else -1
        return res % (10**9 + 7)
