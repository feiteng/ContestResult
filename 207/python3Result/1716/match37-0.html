class Solution:
    def maxProductPath(self, A: List[List[int]]) -> int:
        if not A: return -1
        N = len(A)
        M = len(A[0])
        dp = [[[float('inf'), float('-inf')] for _ in range(M)] for _ in range(N)]
        dp[0][0] = [A[0][0], A[0][0]]
        for i in range(N):
            for j in range(M):
                if i == 0 and j == 0: continue
                if i > 0:
                    dp[i][j][0] = min(dp[i][j][0], A[i][j] * dp[i-1][j][1], A[i][j] * dp[i-1][j][0])
                    dp[i][j][1] = max(dp[i][j][1], A[i][j] * dp[i-1][j][1], A[i][j] * dp[i-1][j][0])
                if j > 0:
                    dp[i][j][0] = min(dp[i][j][0], A[i][j] * dp[i][j-1][1], A[i][j] * dp[i][j-1][0])
                    dp[i][j][1] = max(dp[i][j][1], A[i][j] * dp[i][j-1][1], A[i][j] * dp[i][j-1][0])
        if dp[-1][-1][0] < 0 and dp[-1][-1][1] < 0:
            return -1
        return max(dp[-1][-1][0], dp[-1][-1][1]) % (10**9 + 7)
