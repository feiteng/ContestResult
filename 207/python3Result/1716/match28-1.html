class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len
(grid[0])
        dp = [[[float('inf'), float('-inf')] for _ in range(n)] for _ in range(m)]
        dp[0][0] = [grid[0][0], grid[0][0]]
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    continue
                
if j > 0:
                    dp[i][j][0] = min(dp[i][j - 1][0] * grid[i][j], dp[i][j - 1][1] * grid[i][j], dp[i][j][0])
                    dp[i][j][1] = max(dp[i][j - 1][0] * grid[i][j], dp[i][j - 1][1] * grid[i][j], dp[i][j][1])
                if i > 0:
                    dp[i][j][0] = min(dp[i - 1][j][0] * grid[i][j], dp[i - 1][j][1] * grid[i][j], dp[i][j][0])
                    dp[i][j][1] = max(dp[i - 1][j][0] * grid[i][j], dp[i - 1][j][1] * grid[i][j], dp[i][j][1])
        if max(dp[-1][-1]) < 0:
            return -1
        return max(dp[-1][-1]) % 1_000_000_007
