class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.ans = 0
        def dfs(mylist, index):
            if index == len(s):
                self.ans = max(self.ans, len(mylist))
            tmpstr = ''
            for i in range(index, len(s)):
                tmpstr += s[i]
                if tmpstr not in mylist:
                    dfs(mylist+[tmpstr], i+1)
        dfs([], 0)
        return self.ans
