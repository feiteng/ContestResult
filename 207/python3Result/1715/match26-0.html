class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        if not s:
            return 0
        self.res = 0
        self.dfs(s, 0, set())
        return self.res
    
    def dfs(self, s, i, visited):
        if i == len(s):
            self.res = max(self.res, len(visited))
            return
        
        for j in range(i, len(s)):
            substr = s[i:j + 1]
            if substr not in visited:
                visited.add(substr)
                self.dfs(s, j + 1, visited)
                visited.discard(substr)
