class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        res 
= [0]
        self.helper(s, 0, set(), res)
        return res[0]
    
    def helper(self, s, start, words, res):
        if start == len(s):
            res[0] = max(res[0], len(words))
            return
        
        for i in range(start, len(s)):
            temp = s[start:i + 1]
            if temp not in words:
                words.add(temp)
                self.helper(s, i + 1, words, res)
                words.remove(temp)
