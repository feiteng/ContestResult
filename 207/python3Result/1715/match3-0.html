class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.res=0
        def rec(ind, hash, word):
            if ind==len(s):
                # print(hash)
                # if len(hash)>self.res:
                #     print(hash)
                self.res=max(self.res, len(hash))
                
            new=''
            for i in range(ind, len(s)):
                new+=s[i]
                if new not in hash:
                    rec(i+1, hash+[new], new)
                
                    
        
        rec(0, [], '')
        return self.res
