class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        
        def dfs(i, soFar):
            if i == len(s):
                return len(soFar)
            res = 0
            for j in range(i+1, len(s)+1):
                word = s[i:j]
                if word not in soFar:
                    soFar.append(word)
                    res= max(res, dfs(j, soFar))
                    soFar.pop()
            return res
        return dfs(0,[])
