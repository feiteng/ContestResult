class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.ans = 0
        self.helper(s, 0, set())
        return self.ans

    def helper(self, s, start, tmp):
        if start == len(s):
            #print(tmp)
            self.ans = max(self.ans, len(tmp))
            return

        for i in range(start+1, len(s)+1):
            if s[start:i] not in tmp:
                tmp.add(s[start:i])
                self.helper(s, i, tmp)
                tmp.remove(s[start:i])
