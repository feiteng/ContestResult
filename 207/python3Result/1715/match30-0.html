class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        return self.max_unique_substrings(s)
    def max_unique_substrings(self, s, seen=()):
        return max((1 + self.max_unique_substrings(s[i:], {candidate, *seen}) for i in range(1, len(s) + 1) if (candidate := s[:i]) not in seen), default=0)
