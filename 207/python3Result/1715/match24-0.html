class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        n = len(s)
        global res
        res = 0
        
        def dfs(idx, prev):
            if idx == n:
                global res
                res = max(res, len(prev))
                return
            
            for j in range(idx, n):
                new = s[idx:(j + 1)]
                if new not in prev:
                    prev.add(new)
                    dfs(j + 1, prev)
                    prev.remove(new)

        dfs(0, set())
        return res
