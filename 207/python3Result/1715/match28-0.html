class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.ans = 0
        self.dfs(0, s, set())
        return self.ans
    
    
    def dfs(self, l, s, visited):
        if l >= len(s):
            # print(visited)
            self.ans = max(self.ans, len(visited))
            return 
        
        for i in range(l, len(s)):
            t = s[l:i+1]
            if t in visited:
                continue
            visited.add(t)
            self.dfs(i+1, s, visited)
            visited.remove(t)
        
        return
