class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        n = len(s)
        chars 
= list(s)
        max_split = 1
        def search(start, curr_split_set):
            nonlocal max_split
            if start == n:
                max_split = max(max_split, len(curr_split_set))
                return
            for end in range(start+1, n+1):
                split = s[start:end]
                if split not in curr_split_set:
                    curr_split_set.add(split)
                    search(end, curr_split_set)
                    curr_split_set.discard(split)
        search(0, set())
        return max_split
