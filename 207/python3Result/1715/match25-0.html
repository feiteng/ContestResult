class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        if not s:
            return 0
        
        N = len(s)
        res = 1
        def dfs(index, visited):
            nonlocal res
            if index == N:
                res = max(res, len(visited))
                return
            
            for next_index in range(index + 1, N + 1):
                ss = s[index:next_index]
                if ss not in visited:
                    visited.add(ss)
                    dfs(next_index, visited)
                    visited.remove(ss)
                    
        dfs(0, set())
        return res
