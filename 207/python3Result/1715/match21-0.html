class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        def helper(start, prev):
            if start >= len(s):
                return len(prev)
            res = 0
            #print(f"Prev: {prev}")
            for i in range(1, len(s) - start+1):
                subs = s[start:start+i]
                
                if not subs in prev:
                    prev.add(subs)
                    res = max(res, helper(start+i, prev))
                    prev.remove(subs)
            return res
        return helper(0, set())
