class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        def process(s, subs):
            if s=='':
                return 0
            count = 0
            for i in range(1, len(s)+1):
                if s[:i] not in subs:
                    subs.add(s[:i])
                    count = max(count, 1+process(s[i:], subs))
                    subs.remove(s[:i])
            return count
        return process(s, set([]))
