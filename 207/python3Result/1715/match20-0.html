class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.ans = 0
        def helper(lis,s):
            if s == '':
                self.ans = max(self.ans,len(lis))
                return
            
            for i in range(1,len(s)+1):
                if s[:i] not in lis:
                    lis.append(s[:i])
                    helper(lis,s[i:])
                    lis.pop()
                    
         
            
        
        helper([],s)
        
        return self.ans
