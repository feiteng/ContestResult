class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.max = 1
        self.dfs(s, set())
        
        return self.max
    
    def dfs(self, s, seen):
        if not s:
            self
.max = max(self.max, len(seen))
            return
        
        for i in range(len(s)):
            w = s[:i+1]
            if w in seen:
                continue
            seen.add(w)
            self.dfs(s[i+1:], seen)
            seen.remove(w)
