class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.max_len = 0
        self.dfs(s, 0, set())
        return self.max_len
        
    def dfs(self, s: str, start: int, cur_set):
        # print(f"dfs: start: {start}, cur_set: {cur_set}")
        if start >= len(s):
            self.max_len = max(self.max_len, len(cur_set))
            # print(f"return: max_len: {self.max_len}, cur_set: {cur_set}")
            return
        
        
for i in range(start + 1, len(s) + 1):
            word = s[start: i]
            if word not in cur_set:
                cur_set.add(word)
                # print(f"not in cur set: word: {word}, i: {i}, cur_set: {cur_set}")
                self.dfs(s, i, cur_set)
                cur_set.remove(word)
                # print(f"removed: word: {word}, i: {i}, cur_set: {cur_set}")
