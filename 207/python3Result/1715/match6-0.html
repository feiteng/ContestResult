class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        def dfs(s, dct, l):
            if not s:
                return l
            ans = 0
            for i in range(len(s)):
                if s[:i+1] not in dct:
                    dct.add(s[:i+1])
                    ans = max(ans, dfs(s[i+1:], dct, l + 1))
                    dct.remove(s[:i+1])
            return ans
        return dfs(s, set(), 0)
