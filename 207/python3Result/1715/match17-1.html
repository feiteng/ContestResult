class Solution:
    best = 0
    
def maxUniqueSplit(self, s: str) -> int:
        self.best = 0
        self.helper(s, 0, set())
        return self.best
    
    def helper(self, s, start, cur):
        if start >= len(s):
            #print(cur)
            self.best = max(self.best, len(cur))
            return 
        
for i in range(start, len(s)+1):
            temp = s[start:i]
            if temp != '' and temp not in cur:
                cur.add(temp)
                self.helper(s, i, cur)
                cur.remove(temp)
