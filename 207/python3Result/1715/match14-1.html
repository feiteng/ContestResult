class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        n = len(s)
        self.ans = 0
        def dfs(cur,st):
            if cur==n:
                self.ans = max(self.ans,len(st))
                
return
            for j in range(cur+1,n+1):
                if s[cur:j] not in st:
                    st.add(s[cur:j])
                    dfs(j,st)
                    st.remove(s[cur:j])
        dfs(0,set())
        return self.ans
