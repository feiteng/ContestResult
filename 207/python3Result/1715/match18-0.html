class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.res = 0
        
        self.dfs(0, set(), s)
        
        return self.res
        
    def dfs(self, idx, used, s):
        if idx == len(s):
            self.res = max(self.res, len(used))
            return
        
        cur = ''
        for i in range(idx, len(s)):
            cur += s[i]
            if cur not in used:
                used.add(cur)
                self.dfs(i+1, used, s)
                used.remove(cur)
