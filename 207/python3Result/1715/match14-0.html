class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        n = len(s)
        self.res = 0
        
        def dfs(i,seen,cur):
            if i == n:
                self.res = max(self.res, cur)
                return 
            for j in range(i,n):
                if s[i:j+1] not in seen:
                    seen.add(s[i:j+1])
                    dfs(j+1,seen,cur+1)
                    seen.remove(s[i:j+1])
        
        dfs(0, set(), 0)
        return self.res
