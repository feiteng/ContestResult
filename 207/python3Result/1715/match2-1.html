class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        def helper(string, visited):
            ret = 0
            for i in range(len(string)):
                x = string[:i+1]
                if x not in visited:
                    visited.add(x)
                    ret = max(ret, 1 + helper(string[i+1:],visited))
                    visited.remove(x)
            return ret
        return helper(s, set())
