class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        st = set()
        N = len(s)
        
        def rec(cur):
            # print(cur)
            if cur == N:
                # print(st)
                return len(st)
            
            ans = 0
            for i in range(cur, N):
                sub = s[cur:i+1]
                if sub not in st:
                    st.add(sub)
                    ans = max(ans, rec(i+1))
                    st.discard(sub)
            
            return ans
        
        return rec(0)
