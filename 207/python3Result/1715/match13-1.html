class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        self.res = 0
        
        self.backtrack(s, [], 0, set())
        
        return self.res
    
    def backtrack(self, s, cur_path, i, words_used):
        if i == len(s):
            self.res = max(self.res, len(cur_path))
            return
        
        for j in range(i + 1, len(s) + 1):
            substr = s[i: j]
            if substr not in words_used:
                cur_path.append(substr)
                words_used.add(substr)
                
                self.backtrack(s, cur_path, j, words_used)
                
                words_used.remove(substr)
                cur_path.pop()
