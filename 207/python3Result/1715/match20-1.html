class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        cur = []
        self
.ans = 0
        def dfs(s):
            if not s:
                self.ans = max(self.ans, len(cur))
                return
            for i in range(1, len(s) + 1):
                if s[:i] not in cur:
                    cur.append(s[:i])
                    dfs(s[i:])
                    cur.pop()
        dfs(s)
        return self.ans
