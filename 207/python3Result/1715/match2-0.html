class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        
        def helper(s, seen):
            res = 0
            for i in range(1, len(s) + 1):
                cand = s[:i]
                if cand not in seen:
                    seen.add(cand)
                    res = max(res, 1 + helper(s[i:], seen))
                    seen.remove(cand)
            return res
        
        return helper(s, set())
