class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        self.size1 = len(cost)
        self.size2 = len(cost[0])
        self.fullVisited = (1 << self.size2) - 1
        self.cost = cost
        self.costT = [[self.cost[i][j] for i in range(self.size1)] for j in range(self.size2)]
        self.cache = dict()
        return self.connectTwoGroupsRecursive(0, 0)
        
    def connectTwoGroupsRecursive(self, i, status):
        key = (i, status)
        if key in self.cache:
            return self.cache[key]
        if i == self.size1:
            result = 0
            if status == self.fullVisited:
                result = 0
            else:
                for j in range(self.size2):
                    if status & (1 << j) == 0:
                        result += min(self.costT[j])
            self.cache[key] = result
            return result
        result = float("inf")
        for j in range(self.size2):
            result = min(result, self.cost[i][j] + self.connectTwoGroupsRecursive(i+1, status | (1 << j)))
        self.cache[key] = result
        return result
