class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        
        s1 = len(cost)
        s2 = len(cost[0])
        a = (1 << s1) - 1
        b = (1 << s2) - 1
        dp = {}
        lowest = [min(cost[i][j] for i in range(s1)) for j in range(s2)]
        for i in range(b+1):
            dp[(0, i)] = 0
            for k in range(s2):
                if (1 << k) & i:
                    dp[0,i] += lowest[k]
        # print(lowest, dp)
        def dfs(m1, left, m2):
            if (m1, m2) in dp:
                return dp[m1, m2]
            # left = m1 & (~(m1-1))
            tmp = m1 ^ (1 << left)
            dp[m1, m2] = min(cost[left][i] + dfs(tmp, left+1, m2 - (m2 & (1 << i))) for i in range(s2))
            return dp[m1, m2]
        return dfs(a, 0, b)
