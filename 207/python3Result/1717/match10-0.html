class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        N1 = len(cost)
        N2 = len(cost[0])
        #print(N1, N2)
        
        @lru_cache(None)
        def go(index, used, index2, mask):
            if index == N1:
                if mask == (1 << N2) - 1:
                    return 0
                return 10 ** 20
            if index2 == N2:
                if used:
                    return go(index + 1, False, 0, mask)
                return 10 ** 20
            
            
            best = 10 ** 20

            total = cost[index][index2]
            if used:
                best = min(best, go(index + 1, False, 0, mask))
            
            best = min(best, go(index, used, index2 + 1, mask))
            best = min(best, go(index, True, index2 + 1, mask | (1 << index2)) + total)
            return best
        
        return go(0, False, 0, 0)
