class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        n1, n2 = len(cost), len(cost[0])
        sys.setrecursionlimit(1000000000)
        
@lru_cache(None)
        def dfs(ind1: int, mask2: int) -> int:
            if ind1 == n1:
                c = 0
                for j in range(n2):
                    if not mask2 >> j & 1:
                        _min = float('inf')
                        for i in range(n1):
                            _min = min(cost[i][j], _min)
                        c += _min          
                return c         
            ans = float('inf') 
            _min = float('inf')
            k = -1
            for i in range(n2):
                if mask2 >> i & 1:
                    if _min > cost[ind1][i]:
                        _min = cost[ind1][i]
                        k = i
                else:         
                    ans = min(ans, cost[ind1][i] + dfs(ind1 + 1, mask2 | (1 << i)))
            if not k == -1:           
                ans = min(ans, _min + dfs(ind1 + 1, mask2))
            return ans                
        
        return dfs(0, 0)
