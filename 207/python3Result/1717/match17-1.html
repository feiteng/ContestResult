class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        m = len(cost)
        n = len(cost[0])
        dic = {}
        a = {}
        b = {}
        for i in range(m):
            tem = cost[i][0]
            ind = 0
            for j in range(1,n):
                if cost[i][j] < tem:
                    tem = cost[i][j]
                    ind = j
            a[i] = [ind,tem]
        for j in range(n):
            tem = cost[0][j]
            ind = 0
            for i in range(1,m):
                if cost[i][j] < tem:
                    tem = cost[i][j]
                    ind = i
            b[j] = [ind,tem]
                
        
        
        
        def solve(c,d):
            sign = (c,tuple(d))
            if sign in dic:
                return dic[sign]
            if c == m and not d:
                return 0
            elif c == m:
                re = 0  
                for i in d:
                    re += b[i][1]
                dic[sign] = re
                return re
            elif not d:
                re = 0
                for i in range(c,m):
                    re += a[i][1]
                dic[sign] = re
                return re
            else:
                re = 100000000
                        
                for i in list(d):
                    d.remove(i)
                    re = min(re,cost[c][i] + solve(c + 1,d))
                    d.add(i)
                if a[c][0] not in d:
                    re = min(re,a[c][1] + solve(c + 1,d))
      
                        
                        
                dic[sign] = re
                return re

        return solve(0,set([i for i in range(n)]))
