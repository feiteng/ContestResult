class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        n1, n2 = len(cost), len(cost[0])
        self.cost = cost
        self.D = {}
        return self.dp((1<<n1)-1, (1<<n2)-1, n1, n2)
    
    @lru_cache(None)
    def dp(self, s1, s2, n1, n2):
        if s1 == s2 == 0:
            return 0
        #if (s1, s2) in self.D:
        #    return self.D[(s1, s2)]
        res = float('inf')
        for j in range(n2):
            if (1<<j) & s2:
                for k in range(n1):
                    if (1<<k) & s1:
                        res = min(res, self.cost[k][j] + self.dp(s1-(1<<k), s2-(1<<j), n1, n2))
                    else:
                        res = min(res, self.cost[k][j] + self.dp(s1, s2-(1<<j), n1, n2))
                break
            else:
                for k in range(n1):
                    if (1<<k) & s1:
                        res = min(res, self.cost[k][j] + self.dp(s1-(1<<k), s2, n1, n2))
                        break
        # for k in range(n1):
        #     if (1<<k) & s1:
        #         for j in range(n2):
        #             if not (1<<j) & s2:
        #                 res = min(res, self.cost[k][j] + self.dp(s1-(1<<k), s2, n1, n2))
        #self.D[(s1, s2)] = res
        return res
