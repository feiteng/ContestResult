import sys
sys.setrecursionlimit(999999999)

from typing import List
class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:

        n1, n2 = len(cost), len(cost[0])

        from functools import lru_cache
        @lru_cache(typed=False, maxsize=128000000)
        def dp(stat1, stat2):
            if stat1 == 0 and stat2 == 0:
                return 0

            ans = 0x7fffffff
            if stat1 != 0:
                for i in range(n1):
                    if stat1 & (1 << i):
                        for j in range(n2):
                            s1, s2 =  stat1 & (~(1 << i)), stat2 & (~(1 << j))
                            ans = min(ans, cost[i][j] + dp(s1, s2))
                        break
            else:
                for j in range(n2):
                    if stat2 & (1 << j):
                        for i in range(n1):
                            s1, s2 = stat1 & (~(1 << i)), stat2 & (~(1 << j))
                            ans = min(ans, cost[i][j] + dp(s1, s2))
                        break

            return ans

        return dp((1 << n1) - 1, (1 << n2) - 1)
