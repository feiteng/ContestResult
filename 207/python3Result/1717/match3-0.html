class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        M, N = len(cost), len(cost[0])
        g2 = (1 << N) -1
        def transpose():
            tmp = [[0] * M for _ in range(N)]
            for i in range(M):
                for j in range(N):
                    tmp[j][i] = cost[i][j]
            return tmp
        tcost = transpose()
        @lru_cache(None)
        def dfs(i, gr2):
            if i == M and gr2 == 0: return 0
            if i == M: 
                res = 0
                for j in range(N):
                    if gr2 & (1 << j):
                        res += min(tcost[j])
                return res
            res = math.inf
            for j in range(N):
                res = min(res, cost[i][j] + dfs(i+1, gr2 & (~(1 << j))))
            return res
        return dfs(0, g2)
