class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        s1 = len(cost)
        s2 = len(cost[0])
        if s1 < s2:
            new_cost 
= [[0] * s1 for i in range(s2)]
            for i in range(s1):
                for j in range(s2):
                    new_cost[j][i] = cost[i][j]
            cost = new_cost
            s1 = len(cost)
            s2 = len(cost[0])
        dp = [[1000000] * (1 << s2) for i in range(s1+1)]
        dp[0][0] = 0
        for i in range(s1):
            for j in range(s2):
                ct = cost[i][j]
                for s in range(1 << s2):
                    dp[i+1][s | (1 << j)] = min(dp[i+1][s | (1 << j)], dp[i][s] + ct)
        res = 100000
        mins = []
        for j in range(s2):
            mins.append(min([cost[i][j] for i in range(s1)]))
        for s in range((1 << s2)):
            tot = dp[s1][s]
            for i in range(s2):
                if not s & (1 << i):
                    tot += mins[i]
            res = min(res, tot)
        return res
