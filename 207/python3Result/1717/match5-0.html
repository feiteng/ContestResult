class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        
        rows = len(cost)
        cols = len(cost[0])
    
        
        @lru_cache(None)
        def helper(i, rem):
            if i==rows:
                ans = 0
                for j in range(cols):
                    if not rem & 1<<j:
                        m = float('inf')
                        for r in range(0, rows):
                            m = min(m, cost[r][int(j)])
                        ans += m
                return ans
            
                    
                
                
            ans = float('inf')
            for j in range(0, cols):
                nrem = rem | 1<<j
                    
                ans = min(ans, cost[i][j]+helper(i+1, nrem))
            
            return ans
        
        
        return helper(0, 0)
