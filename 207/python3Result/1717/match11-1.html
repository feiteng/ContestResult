class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        m = len(cost)
        n = len(cost[0])
        @functools.lru_cache(None)
        def f(cur,p):
            ans = 0
            for i in range(n):
                if p&(1<<i):
                    ans+= cost[cur][i]
            return ans
        @functools.lru_cache(None)
        def fm(cur,p):
            ans = float("inf")
            for i in range(n):
                if p&(1<<i):
                    ans = min(ans,cost[cur][i])
            return ans
        @functools.lru_cache(None)
        def dfs(cur,p):
            if cur==m:
                if p== ((1<<n)-1):
                    return 0
                return float("inf")
            ans = float("inf")
            mc = 0
            if p:
                mc = fm(cur,p)
            q = (1<<n)-1-p
            k = q
            while k:
                ans = min(ans,f(cur,k)+dfs(cur+1,p|k))
                k = (k-1)&q
            if p:
                ans = min(ans,mc+dfs(cur+1,p))
            return ans

        return dfs(0,0)
