class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        m,n=len(cost),len(cost[0])
        @lru_cache(None)
        def dp(i,mask,pt):
            # mask is which in group2 alreadyu hit
            if pt==0:
                if i==m:
                    return dp(0,mask,1)
                ret=inf
                for j in range(n):
                    ret=min(ret,dp(i+1,mask|(1<<j),0)+cost[i][j])
                return ret
            else:
                if i==n:
                    return 0
                if mask&(1<<i):
                    return dp(i+1,mask,1)
                # have to pick one to connect to
                mn=inf
                for j in range(m):
                    mn=min(mn,cost[j][i])
                return mn+dp(i+1,mask,1)
        return dp(0,0,0)
