class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        
        cnta = len(cost)
        cntb = len(cost[0])
        
        @functools.lru_cache(None)
        def bitcnt(num):
            cnt = 0
            for i in range(12):
                if num&(1<<i): cnt+=1
            return cnt
        
        @functools.lru_cache(None)
        def f(g1, g2):
            # print("{0} {1}".format(g1, g2))
            nonlocal cnta, cntb
            
            if bitcnt(g1) == cnta and bitcnt(g2) == cntb: 
                return 0
            
            ans = float('inf')
            if bitcnt(g1) < cnta:
                for i in range(cnta):
                    if not g1&(1<<i):
                        for j in range(cntb):
                            ans = min(ans, 
                                     f(g1|(1<<i), g2|(1<<j))+cost[i][j])
                        break
            if bitcnt(g1) == cnta:
                for j in range(cntb):
                    if not g2&(1<<j):
                        for i in range(cnta):
                            ans = min(ans, 
                                     f(g1|(1<<i), g2|(1<<j))+cost[i][
j])
                        break
            return ans
        
        return f(0,0)
