class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        m,n=len(cost),len(cost[0])
        mi=[min(cost[i][j] for i in range(m)) for j in range(n)]
        @lru_cache(None)
        def dp(i,used):
            if i==m:
                ans=0
                for j in range(n):
                    if not 1<<j&used:
                        ans+=mi[j]
                return ans
            res=inf
            
for j in range(n):
                res=min(res,cost[i][j]+dp(i+1,used|1<<j))
            return res
        return dp(0,0)
