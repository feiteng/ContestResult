from heapq import *

class Solution:
    def connectTwoGroups(self, cost: List[List[int]]) -> int:
        N, M = len(cost), len(cost[0])
        
        @lru_cache(None)
        def f(i, j, st, flag):
            if j == M:
                if flag == 0:
                    return 1e99
                i += 1
                j = 0
                flag = 0
            if i == N:
                if st == (1 << M) - 1:
                    return 0
                else:
                    return 1e99
            
            tk = cost[i][j] + f(i, j + 1, st | (1 << j), 1)
            ntk = f(i, j + 1, st, flag)
            return min(tk, ntk)
                
            
        return f(0, 0, 0, 0)
