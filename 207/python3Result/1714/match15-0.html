class Solution:
    def reorderSpaces(self, text: str) -> str:
        num_spaces = text.count(' ')
        texts = text.split()
        if len(texts) == 1:
            return texts[0]+' '*num_spaces
        space_between = num_spaces//(len(texts)-1)
        space_end = num_spaces%(len(texts)-1)
        ret = []
        return (' '*space_between).join(texts) + ' '*space_end
