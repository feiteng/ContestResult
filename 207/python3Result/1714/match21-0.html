class Solution:
    def reorderSpaces(self, text: str) -> str:
        words = [w for w in text.split(' ') if w]
        spaces = text.count(' ')
        if len(words) == 1:
            return words[0] + ' ' * spaces
        swidth, r = divmod(spaces, len(words) - 1)
        sep = ' ' * swidth
        res = sep.join(words)
        res += ' ' * r
        return res
