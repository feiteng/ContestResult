class Solution:
    def reorderSpaces(self, text: str) -> str:
        sp = text.split(" ")
        nonempty = [i for i in sp if i != '']
        num = text.count(" ")
        if len(nonempty) == 1:
            return nonempty[0] + ' ' * num
        k = num // (len(nonempty) - 1)
        j = num % (len(nonempty) - 1)
        return (' ' * k).join(nonempty) + ' ' * j
