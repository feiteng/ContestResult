class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt = text.count(' ')
        text = text.strip().split()
        if len(text) == 1:
            return text[0] + ' '*cnt
        
        div, mod = divmod(cnt, len(text)-1)
        t = ' '*div
        return t.join(text)+' '*mod
