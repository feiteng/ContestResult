class Solution:
    def reorderSpaces(self, text: str) -> str:
        words = text.strip().split()
        if len(words) == 0:
            return text
        letters = sum
(map(len, words))
        spaces = len(text) - letters
        if len(words) == 1:
            return words[0] + ' '*spaces
        per = spaces // (len(words)-1)
        rest = spaces % (len(words)-1)
        return (' '*per).join(words) + ' '*rest
