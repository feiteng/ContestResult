class Solution:
    def reorderSpaces(self, text: str) -> str:
        spaces = text.count(' ')
        words 
= text.strip().split()
        if len(words) == 1:
            return words[0] + ' ' * spaces
        s = ' ' * (spaces // (len(words)-1))
        return s.join(words) + ' ' * (spaces % (len(words)-1))
