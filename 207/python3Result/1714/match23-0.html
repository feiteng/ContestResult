class Solution:
    def reorderSpaces(self, text: str) -> str:
        words=text.split()
        spacecnt=text.count(' ')
        if len(words)<2:
            return words[0]+' '*spacecnt
        div,mod=divmod(spacecnt,(len(words)-1))
        return (' '*div).join(words)+' '*mod
