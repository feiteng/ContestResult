class Solution:
    def reorderSpaces(self, text: str) -> str:
        numSpaces = text.count(' ')
        words = text.strip().split()
        if len(words) == 1:
            return words[0] + (' ' * numSpaces)
        q, r = divmod(numSpaces, len(words) - 1)
        return (' ' * q).join(words) + (' ' * r)
