class Solution:
    def reorderSpaces(self, text: str) -> str:
        word = []
        cur = ''
        cnt = 0
        for c in text:
            if c == ' ':
                cnt += 1
                if cur:
                    word.append(cur)
                    cur = ''
            else:
                cur += c
        if cur:
            word.append(cur)
        # print(cnt)
        if len(word) == 1:
            return word[0] + ' '*cnt
        return (' '*(cnt//(len(word)-1))).join(word) + ' '*(cnt%(len(word)-1))
