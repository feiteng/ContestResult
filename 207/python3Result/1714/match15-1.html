class Solution:
    def reorderSpaces(self, text: str) -> str:
        s = text.split()
        n = text.count(' ')
        if len(s) == 1:
            return s[0] + ' '*n
        x = n // (len(s)-1)
        a = n % (len(s) - 1)
        pre = ' ' * x
        return pre.join(s) + ' ' * a
