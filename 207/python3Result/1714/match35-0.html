class Solution:
    def reorderSpaces(self, text: str) -> str:
        space = 0
        for c in text:
            if c == ' ':
                space += 1
        words = [ word for word in text.split(' ') if word != '']
        if len(words) == 1:
            return words[0]+' '*space
        d = ' '*(space//(len(words)-1))
        extra = ' '*(space%(len(words)-1))
        return d.join(words)+extra
