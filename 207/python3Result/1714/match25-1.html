class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt = 0
        for c in text:
            if c == ' ':
                cnt += 1
        text = text.strip().split()
        word_num = len(text)
        if word_num == 1:
            return text[0] + ' ' * cnt
        avg, res = cnt // (word_num - 1), cnt % (word_num - 1)
        ct = avg * ' ' if avg else ''
        rt = res * ' ' if res else ''
        return ct.join(text) + rt
