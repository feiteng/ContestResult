class Solution:
    def reorderSpaces(self, text: str) -> str:
        blanks = 0
        for e in text:
            if e == " ":
                blanks += 1
        words = text.split()
        if len(words) == 1:
            return words[0] + " " * blanks
        
        averageBlanks = blanks // (len(words) - 1)
        remainingBlanks = blanks % (len(words) - 1)
        return (" " * averageBlanks).join(words) + " " * remainingBlanks
