class Solution:
    def reorderSpaces(self, text: str) -> str:
        space = 0
        for c in text:
            if c == ' ':
                space += 1

        words = text.split()
        if len(words) == 1:
            return words[0] + ' ' * space

        between, end = divmod(space, len(words)-1)
        between_space = ' ' * between
        return between_space.join(words) + ' ' * end
