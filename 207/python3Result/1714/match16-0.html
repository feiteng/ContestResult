class Solution:
    def reorderSpaces(self, text: str) -> str:
        text_split = text.split()
        words = len(text_split)
        spaces = len(text) - len("".join(text_split))
        if words == 1:
            return text_split[0] + " " * spaces
        intervals = " " * (spaces // (words - 1))
        rest = " "*(spaces%(words - 1))
        # print("space: ", spaces // (words - 1), "rest: ", spaces%(words - 1))
        return intervals.join(text_split) + rest
