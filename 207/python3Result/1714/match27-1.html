class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt = 0
        for c in text:
            if c==" ": cnt+=1
        
        text = text.split()
        n = len(text)
        if n==1:
            return text[0]+cnt*" "
        s = " "*(cnt//(n-1))
        return s.join(text)+" "*(cnt-cnt//(n-1)*(n-1))
