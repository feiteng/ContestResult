class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt=0
        for c in text:
            if c==' ':
                cnt+=1
        list1=text.strip().split()
        num_word=len(list1)
        if num_word==1:
            return list1[0]+" "*cnt
        num_space=cnt//(num_word-1)
        res=(num_space*' ').join(list1)
        res+=(' '*(cnt%(num_word-1)))
        return res
