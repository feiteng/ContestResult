class Solution:
    def reorderSpaces(self, text: str) -> str:
        words = [w for w in text.split(" ") if w != '']
        n = len(words)
        spaces = len(text) - len(''.join(words))
        if n == 1:
            return words[0] + ' ' * spaces
        remainder = spaces % (n-1)
        s = spaces // (n-1)
        
        return (' ' * s).join(words) + ' ' * remainder
