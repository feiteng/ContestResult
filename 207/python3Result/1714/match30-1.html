class Solution:
    def reorderSpaces(self, text: str) -> str:
        counter = 0
        for c in text:
            if c == " ":
                counter += 1
        words = text.split()
        if len(words) == 1:
            return words[0] + " "*counter
        btw = counter // (len(words)-1)
        extra = counter % (len(words)-1)
        res = []
        for w in words:
            res.append(w)
            res.append(" "*btw)
        res.pop()
        res.append(" "*extra)
        return "".join(res)
