class Solution:
    def reorderSpaces(self, text: str) -> str:
        words = [s for s in text.split() if len(s) > 0]
        spaces = len([c for c in text if c == ' '])
        if len(words) == 1:
            return words[0] + ' ' * spaces
        # print(words)
        # print(spaces)
        avg = (spaces // (len(words) - 1)) * ' '
        rem = (spaces % (len(words) - 1)) * ' '
        return avg.join(words) + rem
