class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt = 0
        for c in text:
            if c == ' ':
                cnt += 1
        s = text.split()
        if len(s) == 1:
            return s[0] + ' ' * cnt
        m = len(s)
        n = cnt // (len(s) - 1)
        s = (' ' * n).join(s)
        return s + (cnt - n * (m - 1)) * ' '
