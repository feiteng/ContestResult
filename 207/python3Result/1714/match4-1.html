class Solution:
    def reorderSpaces(self, text: str) -> str:
        spaces = text.count(' ')
        sp 
= text.split()
        if len(sp) == 1:
            return sp[0] + ' ' * spaces
        num_space = spaces // (len(sp) - 1)
        extra_space = spaces % (len(sp) - 1)
        return (' ' * num_space).join(sp) + ' ' * extra_space
