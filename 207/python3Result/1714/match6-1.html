class Solution:
    def reorderSpaces(self, text: str) -> str:
        spaces = text.count(" ")
        texts = text.split()
        
        if len(texts) == 1:
            return texts[0] + (" "*spaces)
        
        factor = spaces // (len(texts) - 1)
        remainder = spaces % (len(texts) - 1)
        
        return (" "*factor).join(texts) + (" "*remainder)
