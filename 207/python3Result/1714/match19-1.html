class Solution:
    def reorderSpaces(self, text: str) -> str:
        space = sum
(c == ' ' for c in text)
        words = text.split()
        if len(words) == 1:
            return words[0] + (' ' * space)
        d, m = divmod(space, len(words) - 1)
        return (' ' * d).join(words) + (' ' * m)
