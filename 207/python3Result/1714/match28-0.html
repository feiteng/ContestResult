class Solution:
    def reorderSpaces(self, text: str) -> str:
        nSpaces = 0
        for i in text:
            if(i == ' '):
                nSpaces += 1
        words = text.split()
        nWords = len(words)
        if(nWords == 1):
            return words[0] + " " * nSpaces
        d = nSpaces // (nWords-1)
        r = nSpaces % (nWords-1)
        return (" " * d).join(words) + " " * r
