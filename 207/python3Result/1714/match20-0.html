class Solution:
    def reorderSpaces(self, text: str) -> str:
        cnt = 0
        for c in text:
            if c == " ":
                cnt += 1
        words = text.split()
        if len(words) == 1:
            return words[0] + cnt * " "
        else:
            empty = cnt // (len(words) - 1)
            ret = words[0]
            for i in range(1, len(words)):
                ret += " " * empty + words[i]
            ret += " " * (cnt - (len(words) - 1) * empty)
            return ret
