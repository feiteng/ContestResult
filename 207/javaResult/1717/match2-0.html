class Solution {
  private static final int INF = 1_000_000_001;
    private static final long MOD = 1_000_000_007;
    private static final int UNSET = -123;
    private static final int[][] DIR = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    public int connectTwoGroups(List<List<Integer>> cost) {
        int m = cost.size();
        int n = cost.get(0).size();
        int[][] dp = new int[m][1 << n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < dp[i].length; j++) {
                dp[i][j] = INF;
            }
        }

        int[][] c = new int[m][1 << n];

        for (int i = 0; i < m; i++) {
            for (int s = 1; s < (1 << n); s++) {
                int sum = 0;
                for (int j = 0; j < n; j++) {
                    if ((s & (1 << j)) != 0) {
                        sum += cost.get(i).get(j);
                    }
                }
                c[i][s] = sum;
                if (i == 0) {
                    dp[0][s] = sum;
                }
            }
        }
        

        for (int i = 1; i < m; i++) {
            for (int s = 0; s < (1 << n); s++) {
                if (dp[i - 1][s] >= INF) continue;
                for (int j = 1; j < (1 << n); j++) {
                    dp[i][s | j] = Math.min(dp[i][s | j], dp[i - 1][s] + c[i][j]);
                }
            }
        }
        return dp[m - 1][(1 << n) - 1];
    }

}
