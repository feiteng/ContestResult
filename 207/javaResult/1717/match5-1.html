class Solution {
    public int connectTwoGroups(List<List<Integer>> cost) {
        int l1 = cost.size();
        int l2 = cost.get(0).size();
        
        long dp[][]  =new long[l1+1][1<<l2];
        
        long cos[][] = new long[l1][1<<l2];
        
        for(int i =0;i<l1;++i){
            for(int st=1;st<1<<l2;++st){
                long tot = 0;
                for(int j=0;j<l2;++j){
                    if(((1<<j)&st)>0){
                        tot  += cost.get(i).get(j);
                    }
                }
                cos[i][st] = tot;
            }
        }
        
for(long m[]:dp){
            Arrays.fill(m,1000000000L);
        }
        
        dp[0][0] = 0;
        for(int i=1;i<=l1;++i){
            for(int c=0;c<1<<l2;++c){
                for(int st=1;st<1<<l2;++st){
                    dp[i][c|st] = Math.min(dp[i][c|st], dp[i-1][c] + cos[i-1][st]);
                }
                
            }
        }
        return (int)dp[l1][(1<<l2)-1];
        
        
        
        
        
    }
}
