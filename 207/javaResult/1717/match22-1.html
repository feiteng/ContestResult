class Solution {
    static int n;
    static int m;
    static int[][]mat;
    static int[][] cos;
    static int[][]dp;
    static int solve(int ind,int mask)
    {
        if(ind==n)
        {
            if(mask==((1<<m)-1))
                return 0;
            return 1000000000;
        }
        if(dp[ind][mask]!=-1)
            return dp[ind][mask];
        int res=1000000000;
        int mx=(1<<m)-1;
        for(int i=1;i<=mx;i++)
        {
            res=Math.min(res,cos[ind][i]+solve(ind+1,mask|i));
        }
        dp[ind][mask]=res;
        return res;
    }
    public int connectTwoGroups(List<List<Integer>> cost) {
        n=cost.size();
        m=cost.get(0).size();
        mat=new int[n][m];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                mat[i][j]=cost.get(i).get(j);
            }
        }
        int mx=(1<<m)-1;
        
cos=new int[n][mx+1];
        for(int k=0;k<n;k++){
        for(int i=1;i<=mx;i++)
        {
            int c=0;
            for(int j=0;j<m;j++)
            {
                if((i&(1<<j))!=0)
                    c=c+mat[k][j]
;
            }
            cos[k][i]=c;
        }}
        dp=new int[n+1][(1<<m)];
        for(int i=0;i<=n;i++)
        {
            Arrays.fill(dp[i],-1);
        }
        return solve(0,0);
    }
}
