class Solution {
    public int connectTwoGroups(List<List<Integer>> cost) {
        int s1 = cost.size();
        int s2 = cost.get(0).size();
        int[][] dp = new int[s1 + 1][1 << s2];
        dp[0][0] = 1;
        for (int i = 0; i < s1; i++) {
            for (int j = 0; j < (1 << s2); j++) {
                if (dp[i][j] == 0) continue;
                for (int r = 0; r < s2; r++) {
                    int x = j | (1 << r);
                    if (dp[i + 1][x] == 0 || dp[i + 1][x] > dp[i][j] + cost.get(i).get(r)) {
                        dp[i + 1][x] = dp[i][j] + cost.get(i).get(r);
                    }
                }
            }
        }
        int[] min = new int[s2];
        Arrays.fill(min, 101);
        for (int i = 0; i < s1; i++)
            for (int j = 0; j < s2; j++)
                min[j] = Math.min(min[j], cost.get(i).get(j));
        int minn = 100*12;
        for (int i = 0; i < (1 << s2); i++) {
            if (dp[s1][i] == 0) continue;
            int sum = dp[s1][i] - 1;
            for (int j = 0; j < s2; j++) {
                if ((i & (1 << j)) == 0) {
                    sum += min[j];
                }
            }
            minn = Math.min(minn, sum);
        }
        return minn;
    }
}
