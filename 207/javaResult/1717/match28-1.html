class Solution {
    
    int INF = Integer.MAX_VALUE >> 2;
    
List<List<Integer>> cost;
    int N;
    int M;
    Integer[][][][] R;
    
    public int connectTwoGroups(List<List<Integer>> cost) {
        this.cost = cost;
        this.N = cost.size();
        this.M = cost.get(0).size();
        this.R = new Integer[N][M][2][1 << M];
        
        return dp(0, 0, false, 0);
    }
    
    int dp(int i, int j, 
boolean taken, int state) {
        if (i >= N && state == (1 << M) - 1) return 0;
        if (i >= N) return INF;
        if (R[i][j][taken ? 1 : 0][state] != null) return R[i][j][taken ? 1 : 0][state];
        
        int ans = INF;
        int mask = 1 << j;
        int c = cost.get(i).get(j);
        
        if (j + 1 < M) {
            ans = Math.min(c + dp(i, j + 1, true, state | mask), dp(i, j + 1, taken, state));
        } else {
            ans = Math.min(c + dp(i + 1, 0, false, state | mask), taken ? dp(i + 1, 0, false, state) : INF);
        }
        
        R[i][j][taken ? 1 : 0][state] = ans;
        return ans;
    }
}
