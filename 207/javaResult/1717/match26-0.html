class Solution {
    public int connectTwoGroups(List<List<Integer>> cost) {
        int row = cost.size();
        int col = cost.get(0).size();
        int[][] dp = new int[row][1<<col];
        for(int i=1;i<1<<col;i++) {
            for(int j=0;j<col;j++) {
                if ((i & 1<<j) >0) dp[0][i]+=cost.get(0).get(j);
            }
        }
        for(int i=1;i<row;i++) {
            for(int j=1;j<1<<col;j++) {
                int min = Integer.MAX_VALUE;
                for(int k=0;k<col;k++) {
                    if ((j & 1<<k)>0) min = Math.min(min,cost.get(i).get(k));
                }
                min += dp[i-1][j];
                for(int k=1;k<j;k++) {
                    if ((k | j)==j) {
                        int d = j-k;
                        int sum = dp[i-1][k];
                        for(int l = 0;l<col;l++) {
                            if ((d & 1<<l)>0) sum += cost.get(i).get(l);
                        }
                        min = Math.min(sum,min);
                        //System.out.println(String.format("%d,%d,%d",j,k,min));
                    }
                }
                dp[i][j] = min;
            }
            //System.out.println(Arrays.toString(dp[i]));
        }
        return dp[row-1][(1<<col)-1];
    }
}
