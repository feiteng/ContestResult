class Solution {
    private static final int INF = 987654321;
    
    
public int connectTwoGroups(List<List<Integer>> cost) {
        final int N = cost.size();
        final int M = cost.get(0).size();
        
        int[][] C = new int[N][M];
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) 
{
                C[i][j] = cost.get(i).get(j);
            }
        }
        
        int[] best = new int[1 << M];
        Arrays.fill(best, INF);
        best[0] = 0;
        for (int i = 0; i < N; ++i) {
            int[] next = new int[1 << M];
            Arrays.fill(next, INF);
            for (int j = 0; j < M; ++j) {
                for (int mask = 0; mask < best.length; ++mask) {
                    int nextMask = mask | (1 << j);
                    next[nextMask] = Math.min(next[nextMask], next[mask] + C[i][j]);
                    next[nextMask] = Math.min(next[nextMask], best[mask] + C[i][j]);
                }
            }
            best = next;
            // for (int mask = 0; mask < best.length; ++mask) {
            //     System.out.format("i = %d, mask = %3s: best = %d\n", i, Integer.toBinaryString(mask), best[mask]);
            // }
        }
        return best[best.length - 1];
    }
}
