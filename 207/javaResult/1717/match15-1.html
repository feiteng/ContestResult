class Solution {
    int n;
    int m;
    int mar[];
    long dp[][];
    List<List<Integer>> c;
    public int connectTwoGroups(List<List<Integer>> cost) {
        n=cost.size();
        m=cost.get(0).size();
        c=cost;
        mar=new int[n];
        dp=
new long[1<<n][m+1];
        for(int i=0;i<n;i++)
        {
            int min=200;
            for(int j=0;j<m;j++)
            {
                min=Math.min(cost.get(i).get(j),min);
            }
            mar[i]=min;
        }
        return (int)recurse((1<<n)-1,0);
    }
    public long recurse(int bs1,int pos)
    {
        if(pos==m)
        {
            
long ans=0;
            for(int i=0;i<n;i++)
            {
                if((bs1&(1<<i))!=0)
                {
                    ans+=mar[i];
                }
            }
            return ans;
        }
        if(dp[bs1][pos]!=0)
            return dp[bs1][pos]-1
;
        long min=Long.MAX_VALUE;
        for(int i=0;i<n;i++)
        {
            int nbs1=bs1&(~(1<<i));
            min=Math.min(min,c.get(i).get(pos)+recurse(nbs1,pos+1));
        }
        dp[bs1][pos]=min+1;
        return min;
    }
}
