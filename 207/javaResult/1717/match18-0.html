class Solution {
    public int connectTwoGroups(List<List<Integer>> cost) {
        int s1 = cost.size();
        int s2 = cost.get(0).size();
        int[][] dp = new int[s1+1][1<<s2];
        for (int i = 0; i <= s1; ++i) {
            for (int j = 0; j < (1<<s2); ++j)
                dp[i][j] = 1000000;
        }
        dp[0][0] = 0;
        for (int i = 0; i < s1; ++i) {
            ArrayDeque<Integer> q = new ArrayDeque<>();
            for (int j = 0; j < (1<<s2); ++j) {
                for (int k = 0; k < s2; ++k) {
                    if (dp[i+1][j|(1<<k)]>dp[i][j]+cost.get(i).get(k)) {
                        q.add(j|(1<<k));
                        dp[i+1][j|(1<<k)] = dp[i][j]+cost.get(i).get(k);
                    }
                }
            }
            while (!q.isEmpty()) {
                int u = q.removeFirst();
                for (int k = 0; k < s2; ++k) {
                    int v = u|(1<<k);
                    if (dp[i+1][v]>dp[i+1][u]+cost.get(i).get(k)) {
                        q.add(v);
                        dp[i+1][v] = dp[i+1][u]+cost.get(i).get(k);
                    }
                }
            }
        }
        return dp[s1][(1<<s2)-1];
    }
}
