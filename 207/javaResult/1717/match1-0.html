class Solution {
  int result = Integer.MAX_VALUE, m, n;
  int[][] matrix;

  public int connectTwoGroups(List<List<Integer>> cost) {
    m = cost.size();
    n = cost.get(0).size();
    matrix = new int[m][1 << n];
    for (int k = 0; k < m; k++) {
      for (int i = 0; i < (1 << n); i++) {
        int sum = 0;
        for (int j = 0; j < n; j++) {
          if ((i & (1 << j)) > 0)
            sum += cost.get(k).get(j);
        }
        matrix[k][i] = sum;
      }
    }
    for (int i = 0; i < m; i++)
      System.out.println(Arrays.toString(matrix[i]));
    int[][] dp = new int[m][1 << n];
    for (int i = 1; i < m; i++)
      Arrays.fill(dp[i], Integer.MAX_VALUE);
    dp[0] = matrix[0];
    for (int i = 1; i < m; i++) {
      for (int j = 1; j < (1 << n); j++) {
        for (int k = 1; k < (1 << n); k++)
          dp[i][j | k] = Math.min(dp[i][j | k], dp[i - 1][k] + matrix[i][j]);
      }
    }
    return dp[m - 1][(1 << n) - 1];
  }
}
