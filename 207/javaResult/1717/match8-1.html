	class S
olution {
		public int connectTwoGroups(List<List<Integer>> cost) {
			int n = cost.size();
			int m = cost.get(0).size();
			int[] dp = new int[1<<m];
			Arrays.fill(dp, Integer.MAX_VALUE/2);
			dp[0] = 0;
			for(int i = 0;i < n;i++){
				int[] co = new int[1<<m];
				for(int k = 1;k < 1<<m;k++){
					int b = Integer.numberOfTrailingZeros(k);
					co[k] = co[k&k-1] + cost.get(i).get(b);
				}


				int[] ndp = new int[1<<m];
				Arrays.fill(ndp, Integer.MAX_VALUE/2);
				for(int j = 0;j < 1<<m;j++){
					for(int k = 1;k < 1<<m;k++){
						ndp[j|k] = Math.min(ndp[j|k], dp[j] + co[k]);
					}
				}
				dp = ndp;
			}
			return dp[(1<<m)-1];
		}
	}
