class Solution {
 int getRet(List<Integer> cc, int j, int k, int m) {
        int ret = 0;
        for (int x = 0; x < m; x++) {
            if ((j & (1 << x)) != 0) continue;
            if ((k & (1 << x)) != 0) {
                ret += cc.get(x);
            }
        }
        return ret;
    }

    public int connectTwoGroups(List<List<Integer>> cost) {
        int n = cost.size();
        int m = cost.get(0).size();
        int dp[][] = new int[n + 1][1 << m];
        for (int i = 0; i <= n; i++) {
            Arrays.fill(dp[i], -1);
        }
        dp[0][0] = 0;
        int len = 1 << m;
        for (int i = 1; i <= n; i++) {
            List<Integer> cc = cost.get(i - 1);
            for (int j = 0; j < len; j++) {
                if (dp[i - 1][j] == -1) continue;
                for (int k = 0; k < m; k++) {
                    int s = j | (1 << k);
                    int v = cc.get(k) + dp[i - 1][j];
                    if (dp[i][s] == -1 || v < dp[i][s]) dp[i][s] = v;
                }
                for (int k = 0; k < len; k++) {
                    if (j == k || (j & k) != j) continue;
                    int v = getRet(cc, j, k, m) + dp[i-1][j];
                    if(dp[i][k] == -1 || v < dp[i][k]) dp[i][k] = v;
                }
            }
        }
        return dp[n][len - 1];
    }
}
