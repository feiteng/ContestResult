class Solution {
    static int [][] dp;
    static List<List<Integer>> a;
    static int n, m;
    public int connectTwoGroups(List<List<Integer>> cost) {
        a = cost;
        n = cost.size(); m = cost.get(0).size();
        dp = new int[(1 << n)][1 << m];
        for (int [] arr: dp) Arrays.fill(arr, -1);
        return go(0, 0);
    }
    
    static int go(int left, int right) {
        if (dp[left][right] != -1) return dp[left][right];
        if (left == (1 << n) - 1 && right == (1 << m) - 1) return 0;
        int match = -1;
        for (int i = 0; i < n; i++) {
            if (((left >> i) & 1) == 0) {
                match = i; break;
            }
        }
        if (match == -1) {
            for (int i = 0; i < m; i++) {
                if (((right >> i) & 1) == 0) {
                    match = i;
                    break;
                }
            }
            int res = Integer.MAX_VALUE;
            for (int i = 0; i < n; i++) {
                res = Math.min(res, a.get(i).get(match) + go(left | (1 << i), right | (1 << match)));
            }
            return dp[left][right] = res;
        } else {
            int res = Integer.MAX_VALUE;
            for (int i = 0; i < m; i++) {
                res = Math.min(res, a.get(match).get(i) + go(left | (1 << match), right | (1 << i)));
            }
            return dp[left][right] = res;
        }
    }
}
