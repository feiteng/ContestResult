class Solution {
    public int connectTwoGroups(List<List<Integer>> costt) {
        int s1 = costt.size(), s2 = costt.get(0).size(), m = 1 << s2;
        int[][] cost = new int[s1][s2];
        for(int i = 0; i < s1; i++)
            for(int j = 0; j < s2; j++)
                cost[i][j] = costt.get(i).get(j);
        int [][] dp = new int[s1 + 1][m];
        for(int i = 0; i <= s1; i++)
            Arrays.fill(dp[i], 1000000);
        dp[0][0] = 0;
        
int[] min = new int[s2];
        Arrays.fill(min, 10000000);
        for(int j = 0; j < s2; j++)        
            for(int i = 0; i < s1; i++)
                min[j] = Math.min(min[j], cost[i][j]);        
        for(int i = 0; i < s1; i++)
        {
            for(int j = 0; j < m; j++)
            {
                for(int k = 0; k < s2; k++)
                    dp[i + 1][j | (1 << k)] = Math.min(dp[i + 1][j | (1 << k)], dp[i][j] + cost[i][k]);
            }
        }
        int ans = 1000000;
        for(int j = 1; j < m; j++)
        {
            int res = dp[s1][j];
            for(int k = 0; k < s2; k++)
            {
                if((j & (1 << k)) == 0)
                    res += min[k]
;
            }
            ans = Math.min(ans, res);
        }
        return ans;
    }
}
