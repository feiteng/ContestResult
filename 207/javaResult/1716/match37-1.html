class Solution {
    public int maxProductPath(int[][] grid) {
        
long[][][] dp = new long[grid.length][grid[0].length][2];
        dp[0][0][0] = grid[0][0];
        dp[0][0][1] = grid[0][0];
        for (int j = 1; j < grid[0].length; j++) {
            dp[0][j][0] = dp[0][j-1][0] * grid[0][j];
            dp[0][j][1] = dp[0][j-1][1] * grid[0][j];
        }
        for (int i = 1; i < grid.length; i++) {
            dp[i][0][0] = dp[i-1][0][0] * grid[i][0];
            dp[i][0][1] = dp[i-1][0][1] * grid[i][0];
        }
        for (int i = 1; i < grid.length; i++) {
            for (int j = 1; j < grid[0].length; j++) {
                if (grid[i][j] >= 0) {
                    dp[i][j][0] = 
Math.min(dp[i-1][j][0], dp[i][j-1][0]) * grid[i][j];
                    dp[i][j][1] = Math.max(dp[i-1][j][1], dp[i][j-1][1]) * grid[i][j];
                } else {
                    dp[i][j][1] = Math.min(dp[i-1][j][0], dp[i][j-1][0]) * grid[i][j];
                    dp[i][j][0] = Math.max(dp[i-1][j][1], dp[i][j-1][1]) * grid[i][j];                    
                }
            
}
        }
        return dp[grid.length-1][grid[0].length-1][1] >= 0L ? (int)(dp[grid.length-1][grid[0].length-1][1] % 1000000007) : -1;
    }
}
