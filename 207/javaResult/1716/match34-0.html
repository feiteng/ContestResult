class Solution {
    public int maxProductPath(int[][] grid) {
        int row=grid.length;
        int col=grid[0].length;
        long dp[][][]=new long[row][col][2];
        int e=1000000007;
        dp[0][0][0]=grid[0][0];
        for(int i=1;i<row;i++){
            dp[i][0][0]=dp[i-1][0][0]*grid[i][0];
            dp[i][0][1]=dp[i-1][0][0]*grid[i][0];
        }
        for(int i=1;i<col;i++){
            dp[0][i][0]=dp[0][i-1][0]*grid[0][i];
            dp[0][i][1]=dp[0][i-1][0]*grid[0][i];
        }
        for(int i=1;i<row;i++){
            for(int j=1;j<col;j++){
                if(grid[i][j]>0){
                    dp[i][j][0]=grid[i][j]*Math.max(dp[i-1][j][0],dp[i][j-1][0]);
                    dp[i][j][1]=grid[i][j]*Math.min(dp[i-1][j][1],dp[i][j-1][1]);
                }
                else if(grid[i][j]<0){
                    dp[i][j][0]=grid[i][j]*Math.min(dp[i-1][j][1],dp[i][j-1][1]);
                    dp[i][j][1]=grid[i][j]*Math.max(dp[i-1][j][0],dp[i][j-1][0]);
                }
            }
        }
        return (dp[row-1][col-1][0])<0?-1: (int) (dp[row - 1][col - 1][0] % e);
    }

}
