class Solution {
    
public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        long[][] max = new long[m][n];
        long[][] min = new long[m][n];
        max[0][0] = (long)grid[0][0];
        min[0][0] = (long)grid[0][0];
        for (int i = 1; i < m; i++) {
            max[i][0] = max[i - 1][0] * (long)grid[i][0];
            min[i][0] = min[i - 1][0] * (long)grid[i][0];
        }
        for (int i = 1; i < n; i++) {
            max[0][i] = max[0][i - 1] * (long)grid[0][i];
            min[0][i] = min[0][i - 1] * (long)grid[0][i];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                max[i][j] = Math.max(Math.max((long)grid[i][j] * max[i - 1][j], (long)grid[i][j] * max[i][j - 1]),
                                     Math.max((long)grid[i][j] * min[i - 1][j], (long)grid[i][j] * min[i][j - 1]));
                min[i][j] = Math.min(
Math.min((long)grid[i][j] * min[i - 1][j], (long)grid[i][j] * min[i][j - 1]),
                                     Math.min((long)grid[i][j] * max[i - 1][j], (long)grid[i][j] * max[i][j - 1]));
            }
        }
        // show(max);
        // System.out.println("----");
        // show(min);
        if (max[m - 1][n - 1] < 0) {
            return -1;
        }
        return (int)(max[m - 1][n - 1] % 1000000007);
    }
