class Solution {
    public int maxProductPath(int[][] grid) {
        
long[][][] dp = new long[grid.length][grid[0].length][2];
        dp[0][0][0] = 1L*grid[0][0];
        dp[0][0][0] = 1L*grid[0][0];
        for (int i = 1; i < dp.length; i++) {
            dp[i][0][0] = dp[i - 1][0][0]*grid[i][0];
            dp[i][0][1] = dp[i][0][0];
        }
        for (int i = 1; i < dp[0].length; i++) {
            dp[0][i][0] = dp[0][i - 1][0]*grid[0][i];
            dp[0][i][1] = dp[0][i][0];
        } 
        for (int i = 1; i < grid.length; i++) {
            for (int j = 1; j < grid[0].length; j++) {
                dp[i][j][0] = Math.max((1L*grid[i][j])*dp[i - 1][j][0], (1L*grid[i][j])*dp[i - 1][j][1]);
                dp[i][j][0] = Math.max(dp[i][j][0], Math.max((1L*grid[i][j])*dp[i][j - 1][0], (1L*grid[i][j])*dp[i][j - 1][1]));
                dp[i][j][1] = Math.min((1L*grid[i][j])*dp[i - 1][j][0], (1L*grid[i][j])*dp[i - 1][j][1]);
                dp[i][j][1] = Math.min(dp[i][j][1], Math.min((1L*grid[i][j])*dp[i][j - 1][0], (1L*grid[i][j])*dp[i][j - 1][1]));
            }
        }
        if (dp[grid.length - 1][grid[0].length - 1][0] < 0) {
            return -1;
        }
        return (int)(dp[grid.length - 1][grid[0].length - 1][0]%1000000007);
    }
}
