class Solution {
    public int maxProductPath(int[][] grid) {
        //we need to use long as dp 
        //also when a number is positive, dp[i][j][0] = Math.max(dp[i-1][j][0], dp[i][j-1][0]) * grid[i][j]
        //dp[i][j][1] = Math.min(dp[i-1][j][1], dp[i][j-1][1]) * grid[i][j]
        //if a number is 0, the max and min are all 0
        int m = grid.length, n = grid[0].length;
        long[][][] dp = new long[m][n][2];
        int mod = (int)(1e9+7);
        dp[0][0][0] = dp[0][0][1] = grid[0][0];
        for (int j = 1; j < n; j++) {
            dp[0][j][0] = dp[0][j][1] = grid[0][j] * dp[0][j-1][0];
        }
        for (int i = 1; i < m; i++) {
            dp[i][0][0] = dp[i][0][1] = grid[i][0] * dp[i-1][0][0];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (grid[i][j] > 0) {
                    dp[i][j][0] = Math.max(dp[i-1][j][0], dp[i][j-1][0]) * grid[i][j];
                    dp[i][j][1] = Math.min(dp[i-1][j][1], dp[i][j-1][1]) * grid[i][j];
                } else if (grid[i][j] < 0) {
                    dp[i][j][0] = Math.min(dp[i-1][j][1], dp[i][j-1][1]) * grid[i][j];
                    dp[i][j][1] = Math.max(dp[i-1][j][0], dp[i][j-1][0]) * grid[i][j];
                } else {
                    dp[i][j][0] = dp[i][j][1] = 0;
                }
            }
        }
        if (dp[m-1][n-1][0] >= 0) {
            return (int)(dp[m-1][n-1][0] % mod);
        } else {
            return -1;
        }
    }
}
