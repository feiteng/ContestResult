class Solution {
    public int maxProductPath(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;
        long[][] min = new long[row][col];
        long[][] max = new long[row][col];
        min[0][0] = grid[0][0];
        max[0][0] = grid[0][0];
        for (int i = 1; i < row; i++) {
            min[i][0] = Math.min(max[i-1][0] * grid[i][0], min[i-1][0] * grid[i][0]);
            max[i][0] = Math.max(max[i-1][0] * grid[i][0], min[i-1][0] * grid[i][0]);
        }
        for (int i = 1; i < col; i++) {
            min[0][i] = Math.min(max[0][i-1] * grid[0][i], min[0][i-1] * grid[0][i]);
            max[0][i] = Math.max(max[0][i-1] * grid[0][i], min[0][i-1] * grid[0][i]);
        }
        for (int i = 1; i < row; i++) {
            for (int j = 1; j < col; j++) {
                min[i][j] = Math.min(max[i-1][j] * grid[i][j], min[i-1][j] * grid[i][j]);
                min[i][j] = Math.min(min[i][j], max[i][j-1] * grid[i][j]);
                min[i][j] = Math.min(min[i][j], min[i][j-1] * grid[i][j]);
                max[i][j] = Math.max(max[i-1][j] * grid[i][j], min[i-1][j] * grid[i][j]);
                max[i][j] = Math.max(max[i][j], max[i][j-1] * grid[i][j]);
                max[i][j] = Math.max(max[i][j], min[i][j-1] * grid[i][j]);
            }
        }
        int res = (int)(max[row-1][col-1] % 1000000007);
        
        return res < 0 ? -1 : res;
    }
}
