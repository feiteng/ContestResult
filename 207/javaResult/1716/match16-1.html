class Solution {
    final int mod = (int)1e9 + 7;
    
public int maxProductPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        long[][] arr1 = new long[n][m];
        long[][] arr2 = new long[n][m];
        arr1[n - 1][m - 1] = grid[n - 1][m - 1];
        arr2[n - 1][m - 1] = grid[n - 1][m - 1];
        
        for(int i = m - 2; i >= 0; i--){
            arr1[n - 1][i] = arr1[n - 1][i + 1] * grid[n - 1][i];
            arr2[n - 1][i] = arr2[n - 1][i + 1] * grid[n - 1][i];
        }
        
        for(int i = n - 2; i >= 0; i--){
            arr1[i][m - 1] = arr1[i + 1][m - 1] * grid[i][m - 1];
            arr2[i][m - 1] = arr2[i + 1][m - 1] * grid[i][m - 1];
        }
        
        for(int i = n - 2; i >= 0; i--){
            for(int j = m - 2; j >= 0; j--){
                
if(grid[i][j] > 0){
                    arr1[i][j] = Math.max(arr1[i + 1][j] * grid[i][j], arr1[i][j + 1] * grid[i][j]);
                    arr2[i][j] = Math.min(arr2[i + 1][j] * grid[i][j], arr2[i][j + 1] * grid[i][j]);
                }else{
                    arr1[i][j] = Math.max(arr2[i + 1][j] * grid[i][j], arr2[i][j + 1] * grid[i][j]);
                    arr2[i][j] = Math.min(arr1[i + 1][j] * grid[i][j], arr1[i][j + 1] * grid[i][j]);
                }
                
            
}
        }
        
        if(arr1[0][0] < 0){
            return -1;
        }
        
        return (int)(arr1[0][0] % mod);
    }
}
