class Solution {
    private static final long MOD = 1_000_000_007;

    public int maxProductPath(int[][] grid) {
        final int N = grid.length;
        final int M = grid[0].length;

        long[][] min = new long[N][M];
        long[][] max = new long[N][M];

        min[0][0] = grid[0][0];
        max[0][0] = grid[0][0];
        for (int i = 1; i < N; ++i) {
            min[i][0] = grid[i][0] * Math.min(min[i - 1][0], max[i - 1][0]);
            max[i][0] = grid[i][0] * Math.max(min[i - 1][0], max[i - 1][0]);
        }
        for (int j = 1; j < M; ++j) {
            min[0][j] = grid[0][j] * Math.min(min[0][j - 1], max[0][j - 1]);
            max[0][j] = grid[0][j] * Math.max(min[0][j - 1], max[0][j - 1]);
        }
        for (int i = 1; i < N; ++i) {
            for (int j = 1; j < M; ++j) {
                min[i][j] = grid[i][j] * Math.min(
                    Math.min(min[i - 1][j], max[i - 1][j]),
                    Math.min(min[i][j - 1], max[i][j - 1])
                );
                max[i][j] = grid[i][j] * Math.max(
                    Math.max(min[i - 1][j], max[i - 1][j]),
                    Math.max(min[i][j - 1], max[i][j - 1])
                );
            }
        }

        long res = Math.max(max[N - 1][M - 1], min[N - 1][M - 1]);
        return res < 0 ? -1 : (int) (res % MOD);
    }
}
