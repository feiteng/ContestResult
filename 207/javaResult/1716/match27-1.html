class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        long[][] maxP = new long[m][n];
        long[][] minP = new long[m][n];
        maxP[0][0] = grid[0][0];
        minP[0][0] = grid[0][0];
        
        for(int i = 1;i < n;++i) {
            maxP[0][i] = Math.max(grid[0][i] * maxP[0][i - 1], grid[0][i] * minP[0][i - 1]);
            minP[0][i] = Math.min(grid[0][i] * maxP[0][i - 1], grid[0][i] * minP[0][i - 1]);
        }
        
        for(int i = 1;i < m;++i) {
            maxP[i][0] = Math.max(grid[i][0] * maxP[i - 1][0], grid[i][0] * minP[i - 1][0]);
            minP[i][0] = Math.min(grid[i][0] * maxP[i - 1][0], grid[i][0] * minP[i - 1][0]);
        }
        
        for(int i = 1;i < m;++i) {
            for(int j = 1;j < n;++j) {
                maxP[i][j] = Math.max(grid[i][j] * maxP[i - 1][j], grid[i][j] * minP[i - 1][j]);
                maxP[i][j] = Math.max(maxP[i][j], grid[i][j] * maxP[i][j - 1]);
                maxP[i][j] = Math.max(maxP[i][j], grid[i][j] * minP[i][j - 1]);
                
                minP[i][j] = Math.min(grid[i][j] * maxP[i - 1][j], grid[i][j] * minP[i - 1][j]);
                minP[i][j] = Math.min(minP[i][j], grid[i][j] * maxP[i][j - 1]);
                minP[i][j] = Math.min(minP[i][j], grid[i][j] * minP[i][j - 1]);
            }
        }
        if(maxP[m - 1][n - 1] < 0) return -1;
        return Math.toIntExact(maxP[m - 1][n - 1] % 1000000007L);
    }
}
