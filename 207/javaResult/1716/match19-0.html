class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length, n = grid[0].length, mod = 1000000007;
        long[][] min = new long[m][n], max = new long[m][n];
        
        max[0][0] = min[0][0] = (int)grid[0][0];
        
        for (int i = 1; i < m; ++i) {
            max[i][0] = min[i][0] = min[i-1][0] * (long)grid[i][0];
        }
        
        for (int i = 1; i < n; ++i) {
            max[0][i] = min[0][i] = min[0][i-1] * (long)grid[0][i];
        }
        
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                if (grid[i][j] == 0) {
                    min[i][j] = max[i][j] = 0;
                }
                else if (grid[i][j] < 0) {
                    max[i][j] = Math.max(min[i-1][j] * (long)grid[i][j], min[i][j-1] * (long)grid[i][j]);
                    min[i][j] = Math.min(max[i-1][j] * (long)grid[i][j], max[i][j-1] * (long)grid[i][j]);
                }
                else {
                    min[i][j] = Math.min(min[i-1][j] * (long)grid[i][j], min[i][j-1] * (long)grid[i][j]);
                    max[i][j] = Math.max(max[i-1][j] * (long)grid[i][j], max[i][j-1] * (long)grid[i][j]);
                }
            }
        }
        
        return max[m-1][n-1] < 0 ? -1 : (int)(max[m-1][n-1] % mod);
    }
}
