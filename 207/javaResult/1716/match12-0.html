class Solution {
    public int maxProductPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        long[][] maxp = new long[n][m];
        long[][] minp = new long[n][m];
        maxp[0][0] = grid[0][0];
        minp[0][0] = grid[0][0];
        for (int i = 1; i < m; ++i) {
            maxp[0][i] = maxp[0][i-1]*grid[0][i];
            minp[0][i] = minp[0][i-1]*grid[0][i];
        }
        for (int i = 1; i < n; ++i) {
            maxp[i][0] = maxp[i-1][0]*grid[i][0];
            minp[i][0] = minp[i-1][0]*grid[i][0];
            for (int j = 1; j < m; ++j) {
                maxp[i][j] = Math.max(Math.max(maxp[i-1][j]*grid[i][j], minp[i-1][j]*grid[i][j]),Math.max(maxp[i][j-1]*grid[i][j], minp[i][j-1]*grid[i][j]));
                minp[i][j] = Math.min(Math.min(maxp[i-1][j]*grid[i][j], minp[i-1][j]*grid[i][j]),Math.min(maxp[i][j-1]*grid[i][j], minp[i][j-1]*grid[i][j]));
            }
        }
        return maxp[n-1][m-1]<0?-1:(int)(maxp[n-1][m-1]%1000000007);
    }
}
