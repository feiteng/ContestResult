class Solution {
    public int maxProductPath(int[][] a) {
        
int r = a.length, c = a[0].length, mod = (int) 1e9 + 7;
        long[][] dp1 = new long[r][c], dp2 = new long[r][c];
        dp1[0][0] = dp2[0][0] = a[0][0];
        for(int i = 1; i < r; i++) {
            dp1[i][0] = dp1[i - 1][0] * a[i][0];
            dp2[i][0] = dp1[i][0];
        }
        for(int i = 1; i < c; i++) {
            dp1[0][i] = dp1[0][i - 1] * a[0][i];
            dp2[0][i] = dp1[0][i];
        }
        
        // for(int i = 0; i < r; i++){
        //     for(int j = 0; j < c; j++){
        //         System.out.print(dp1[i][j] + " ");
        //     }
        //     System.out.println();
        // }
        for(int i = 1; i < r; i++){
            for(int j = 1; j < c; j++){
                dp1[i][j] = Math.max(Math.max(dp1[i][j - 1] * a[i][j], dp1[i - 1][j] * a[i][j]), 
                                     Math.max(dp2[i][j - 1] * a[i][j], dp2[i - 1][j] * a[i][j]));
                dp2[i][j] = Math.min(Math.min(dp1[i][j - 1] * a[i][j], dp1[i - 1][j] * a[i][j]), 
                                     Math.min(dp2[i][j - 1] * a[i][j], dp2[i - 1][j] * a[i][j]));
            }
        }
        
        return dp1[r - 1][c - 1] < 0 ? -1: (int)(dp1[r - 1][c - 1] % mod);
    }
}
