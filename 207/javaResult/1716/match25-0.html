class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int mod = (int)(1e9 + 7);
        long[][][] dp = new long[m + 1][n + 1][2];
        dp[1][1][0] = grid[0][0];
        dp[1][1][1] = grid[0][0];
        for (int i = 2; i <= m; i++) {
            dp[i][1][0] = dp[i - 1][1][0] * grid[i - 1][0];
            dp[i][1][1] = dp[i - 1][1][1] * grid[i - 1][0];
        }
        for (int i = 2; i <= n; i++) {
            dp[1][i][0] = dp[1][i - 1][0] * grid[0][i - 1];
            dp[1][i][1] = dp[1][i - 1][1] * grid[0][i - 1];
        }
        for (int i = 2; i <= m; i++) {
            for (int j = 2; j <= n; j++) {
                if (grid[i - 1][j - 1] == 0) {
                    dp[i][j][0] = 0;
                    dp[i][j][1] = 0;
                } else if (grid[i - 1][j - 1] > 0) {
                    dp[i][j][0] = Math.max(dp[i - 1][j][0], dp[i][j - 1][0]) * grid[i - 1][j - 1];
                    dp[i][j][1] = Math.min(dp[i - 1][j][1], dp[i][j - 1][1]) * grid[i - 1][j - 1];
                } else {
                    dp[i][j][0] = Math.min(dp[i - 1][j][1], dp[i][j - 1][1]) * grid[i - 1][j - 1];
                    dp[i][j][1] = Math.max(dp[i - 1][j][0], dp[i][j - 1][0]) * grid[i - 1][j - 1];
                }
            }
        }
        if (dp[m][n][0] < 0) return -1;
        return (int)(dp[m][n][0] % mod);
    }
}
