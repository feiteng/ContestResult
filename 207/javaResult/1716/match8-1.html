class Solution {
  public int maxProductPath(int[][] grid) {
    
final int n = grid.length;
    final int m = grid[0].length;
    final long[][] maxSoFar = new long[n][m];
    final long[][] minSoFar = new long[n][m];
    maxSoFar[0][0] = minSoFar[0][0] = grid[0][0];
    for (int i = 1; i < n; i++) {
      maxSoFar[i][0] = minSoFar[i][0] = maxSoFar[i-1][0]*grid[i][0];
    }
    for (int j = 1; j < m; j++) {
      maxSoFar[0][j] = minSoFar[0][j] = maxSoFar[0][j-1]*grid[0][j];
    }
    for (int i = 1; i < n; i++) {
      for (int j = 1; j < m; j++) {
        maxSoFar[i][j] = Math.max(Math.max(maxSoFar[i-1][j]*grid[i][j], minSoFar[i-1][j]*grid[i][j]), Math.max(maxSoFar[i][j-1]*grid[i][j], minSoFar[i][j-1]*grid[i][j]));
        minSoFar[i][j] = Math.min(Math.min(maxSoFar[i-1][j]*grid[i][j], minSoFar[i-1][j]*grid[i][j]), Math.min(maxSoFar[i][j-1]*grid[i][j], minSoFar[i][j-1]*grid[i][j]));
      }
    }
    return maxSoFar[n-1][m-1] < 0L ? -1 : (int) (maxSoFar[n-1][m-1]%(1000000007));
  }
}
