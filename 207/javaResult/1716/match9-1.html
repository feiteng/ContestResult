class Solution {
    public int maxProductPath(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        long[][] max = new long[n][m];
        long[][] min = new long[n][m];
        
max[0][0] = grid[0][0];
        min[0][0] = grid[0][0];
        for (int i = 1; i < n; i++) {
            max[i][0] = max[i - 1][0] * grid[i][0];
            min[i][0] = min[i - 1][0] * grid[i][0];
        }
        for (int i = 1; i < m; i++) {
            max[0][i] = max[0][i - 1] * grid[0][i];
            min[0][i] = min[0][i - 1] * grid[0][i];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {
                int num = grid[i][j];
                
max[i][j] = Math.max(Math.max(min[i - 1][j] * num, max[i - 1][j] * num), 
                                     Math.max(min[i][j - 1] * num, max[i][j - 1] * num));
                // System.out.println("max " + i + " " + j + ":" + max[i][j]);
                min[i][j] = Math.min(Math.min(min[i - 1][j] * num, max[i - 1][j] * num), 
                                     Math.min(min[i][j - 1] * num, max[i][j - 1] * num));
                // System.out.println("min " + i + " " + j + ":" + min[i][j]);
            }
        }
        // System.out.println(max[n - 1][m - 1]);
        if (max[n - 1][m - 1] < 0) return -1;
        return (int) (max[n - 1][m - 1] % (1e9 + 7));
    }
}
