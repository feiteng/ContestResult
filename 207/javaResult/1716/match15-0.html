class Solution {
    public int maxProductPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        long[][] dp = new long[n][m];
        long[][] dp2 = new long[n][m];
        int result = Integer.MIN_VALUE;
        double mod = 1e9 + 7;
        dp[0][0] = grid[0][0];
        dp2[0][0] = grid[0][0];
        for(int i = 1; i<n; i++){
            dp[i][0] = dp[i-1][0] * grid[i][0];
            dp2[i][0] = dp[i-1][0] * grid[i][0];
        }
        for(int i=1; i<m; i++){
            dp[0][i] = dp[0][i-1] * grid[0][i];
            dp2[0][i] = dp[0][i-1] * grid[0][i];
        }
        for(int i = 1; i<n; i++){
            for(int j=1; j<m; j++){
                dp[i][j] = Math.max(Math.max(dp[i-1][j] * grid[i][j] , dp[i][j-1] * grid[i][j]), Math.max(dp2[i-1][j] * grid[i][j] , dp2[i][j-1] * grid[i][j]));
                dp2[i][j] = Math.min(Math.min(dp[i-1][j] * grid[i][j] , dp[i][j-1] * grid[i][j]), Math.min(dp2[i-1][j] * grid[i][j] , dp2[i][j-1] * grid[i][j]));
                
            }
        }
        if(dp[n-1][m-1] <0) return -1;
        return (int)(dp[n-1][m-1]%mod);
    }
    
    
}
