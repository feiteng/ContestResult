class Solution {
    int mod = 1_000_000_007;
    public int maxProductPath(int[][] grid) {
        System.out.println((-5)%4);

        int m = grid.length;
        int n = grid[0].length;
        
        long[][] maxs = new long[m][n];
        long[][] mins = new long[m][n];
        
        maxs[0][0] = grid[0][0];
        mins[0][0] = grid[0][0];
        
        for(int i=1; i<n; i++){
            maxs[0][i] = maxs[0][i-1]*grid[0][i];
            mins[0][i] = maxs[0][i];
        }
        
        for(int i=1; i<m; i++){
            maxs[i][0] = maxs[i-1][0]*grid[i][0];
            mins[i][0] = maxs[i][0];
        }
        
        for(int i=1; i<m; i++){
            for(int j=1; j<n; j++){
                int v = grid[i][j];
                long r1 = v * maxs[i-1][j];
                long r2 = v * maxs[i][j-1];
                long r3 = v * mins[i-1][j];
                long r4 = v * mins[i][j-1];
                maxs[i][j] = Math.max(Math.max(r1,r2),Math.max(r3,r4));
                mins[i][j] = Math.min(Math.min(r1,r2),Math.min(r3,r4));
            }
        }
        // System.out.println(Arrays.deepToString(maxs));
        // System.out.println(Arrays.deepToString(mins));
        return maxs[m-1][n-1] >= 0 ? (int)(maxs[m-1][n-1] % mod): -1;
    }
}
