class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        long[][] max = new long[m][n];
        long[][] min = new long[m][n];
        max[0][0] = min[0][0] = grid[0][0];
        for(int i = 1; i < n; i++) {
            max[0][i] = min[0][i] = (long)grid[0][i] * max[0][i - 1];
        }
        for(int i = 1; i < m; i++) {
            max[i][0] = min[i][0] = (long)grid[i][0] * max[i - 1][0];
        }
        for(int i = 1; i < m; i++) {
            for(int j = 1; j < n; j++) {
                max[i][j] = Math.max(max[i][j - 1] * (long)grid[i][j], min[i][j - 1] * (long)grid[i][j]);
                min[i][j] = Math.min(max[i][j - 1] * (long)grid[i][j], min[i][j - 1] * (long)grid[i][j]);
                
                max[i][j] = Math.max(max[i][j], Math.max(max[i - 1][j] * (long)grid[i][j], min[i - 1][j] * (long)grid[i][j]));
                min[i][j] = Math.min(min[i][j], Math.min(max[i - 1][j] * (long)grid[i][j], min[i - 1][j] * (long)grid[i][j]));
            }
        }
        int mod = (int)(1e9 + 7);
        return max[m - 1][n - 1] >= 0 ? (int)(max[m - 1][n - 1] % mod) : -1;
    }
}
