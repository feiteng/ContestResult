class Solution {
    
    private static final int MODULO = (int) Math.pow(10, 9) + 7;
    
    
public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        long[][] dp = new long[m][n];
        long[][] min = new long[m][n];
        dp[0][0] = grid[0][0];
        min[0][0] = grid[0][0];
        for (int i = 1; i < m; i++) {
            dp[i][0] = grid[i][0] * dp[i-1][0];
            min[i][0] = grid[i][0] * min[i-1][0];
        }
        for (int j = 1; j < n; j++) {
            dp[0][j] = grid[0][j] * dp[0][j-1];
            min[0][j] = grid[0][j] * min[0][j-1];
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = Math.max(grid[i][j] * dp[i-1][j], grid[i][j] * dp[i][j-1]);
                dp[i][j] = Math.max(dp[i][j], Math.max(grid[i][j] * min[i-1][j], grid[i][j] * min[i][j-1]));
                
                min[i][j] = 
Math.min(grid[i][j] * dp[i-1][j], grid[i][j] * dp[i][j-1]);
                min[i][j] = Math.min(min[i][j], Math.min(grid[i][j] * min[i-1][j], grid[i][j] * min[i][j-1]));
            }
        }
        
        int result = (int)(dp[m-1][n-1] % MODULO);
        return result >= 0 ? result : -1;
    }
}
