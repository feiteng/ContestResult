class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        long mod = 1000000007;
        
        long[][] negative = new long[m][n];
        long[][] possitive = new long[m][n];
        
        negative[0][0] = (long)grid[0][0];
        possitive[0][0] = (long)grid[0][0];
        
        for (int i = 1; i < m; ++i) {
            negative[i][0] = negative[i - 1][0] * (long)grid[i][0];
            possitive[i][0] = possitive[i - 1][0] * (long)grid[i][0];
        }
        for (int j = 1; j < n; ++j) {
            negative[0][j] = negative[0][j - 1] * (long)grid[0][j];
            possitive[0][j] = possitive[0][j - 1] * (long)grid[0][j];
        }
        
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                long cur = (long)grid[i][j];
                
negative[i][j] = Math.min(negative[i - 1][j] * cur, Math.min(negative[i][j - 1] * cur, Math.min(possitive[i][j - 1] * cur, possitive[i - 1][j] * cur)));
                
                possitive[i][j] = Math.max(negative[i - 1][j] * cur, Math.max(negative[i][j - 1] * cur, Math.max(possitive[i][j - 1] * cur, possitive[i - 1][j] * cur)));
            }
        }
        int max = (int)(possitive[m - 1][n - 1] % mod);
        return max < 0 ? -1 : max;
    }
}
