class Solution {
    public int maxProductPath(int[][] grid) {
        int n = grid.length;
        int h = grid[0].length;
        
        long[][] max = new long[n][h];
        long[][] min = new long[n][h];
        
        max[n-1][h-1] = grid[n-1][h-1];
        min[n-1][h-1] = grid[n-1][h-1];
        
        for(int j = h - 2; j >= 0; j--) {
            max[n-1][j] = min[n-1][j] = max[n-1][j+1] * grid[n-1][j];
        }
        for (int i = n - 2; i >= 0; i--) {
            max[i][h-1] = min[i][h-1] = max[i+1][h-1] * grid[i][h-1];
        }
        
        for (int i = n - 2; i >= 0; i--) {
            for (int j = h - 2; j >= 0; j--) {
                int value = grid[i][j];
                if (value == 0) {
                    max[i][j] = min[i][j] = 0;
                } else if (value > 0) {
                    max[i][j] = value * Math.max(max[i][j+1], max[i+1][j]);
                    min[i][j] = value * Math.min(min[i][j+1], min[i+1][j]);
                } else {
                    min[i][j] = value * Math.max(max[i][j+1], max[i+1][j]);
                    max[i][j] = value * Math.min(min[i][j+1], min[i+1][j]);
                }
            }
        }
        
        if (max[0][0] >= 0) {
            return (int)(max[0][0] % 1000000007L);
        }
        return -1;
    }
}
