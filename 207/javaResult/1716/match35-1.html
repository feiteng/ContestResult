class Solution {
    public int maxProductPath(int[][] grid) {
        
        int base = 1000000007;
        if (grid == null || grid.length == 0 || grid[0] == null || grid[0].length == 0) {
            return -1;  // no positive product
        }
        
        
int m = grid.length;
        int n = grid[0].length;
        
        long[][] max = new long[m][n];
        long[][] min = new long[m][n];
        
        max[0][0] = min[0][0] = grid[0][0];
        
        for (int j = 1; j < n; j++) {
            max[0][j] = max[0][j - 1] * grid[0][j];
            min[0][j] = min[0][j - 1] * grid[0][j];
        }
        
        for (int i = 1; i < m; i++) {
            max[i][0] = max[i - 1][0] * grid[i][0];
            min[i][0] = min[i - 1][0] * grid[i][0];
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                long maxTop = max[i][j - 1] * grid[i][j];
                long maxLeft = max[i - 1][j] * grid[i][j];
                long minTop = min[i][j - 1] * grid[i][j];
                long minLeft = min[i - 1][j] * grid[i][j];
                
max[i][j] = Math.max(maxTop, Math.max(maxLeft, Math.max(minTop, minLeft)));
                min[i][j] = Math.min(maxTop, Math.min(maxLeft, Math.min(minTop, minLeft)));
            }
        }
        
        if (max[m - 1][n - 1] < 0) return -1;
        return (int)(max[m - 1][n - 1] % base);
        
    }
}
