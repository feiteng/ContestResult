class Solution {
    public int maxProductPath(int[][] grid) {
        
int row = grid.length;
        int col = grid[0].length;
        long[][] maxdp = new long[row][col];
        long[][] mindp = new long[row][col];
        maxdp[0][0] = grid[0][0];
        mindp[0][0] = grid[0][0];
        for (int i = 1; i < col; i++){
            maxdp[0][i] = maxdp[0][i-1] * grid[0][i];
            mindp[0][i] = mindp[0][i-1] * grid[0][i];
        }
        for (int i = 1; i < row; i++){
            maxdp[i][0] = maxdp[i-1][0] * grid[i][0];
            mindp[i][0] = mindp[i-1][0] * grid[i][0];
        }
        for (int i = 1; i < row; i++){
            for (int j = 1; j < col; j++){
                long num1 = maxdp[i-1][j] * grid[i][j];
                long num2 = mindp[i-1][j] * grid[i][j];
                long num3 = maxdp[i][j-1] * grid[i][j];
                long num4 = mindp[i][j-1] * grid[i][j];
                
                
if (grid[i][j] == 0){
                    maxdp[i][j] = 0;
                    mindp[i][j] = 0;
                }
                else if (grid[i][j] > 0){
                    maxdp[i][j] = Math.max(num1, num3);
                    mindp[i][j] = Math.min(num2, num4);
                }
                else {
                    maxdp[i][j] = Math.max(num2, num4);
                    mindp[i][j] = Math.min(num1, num3);
                }
                
            }
        }
        return maxdp[row-1][col-1] >= 0 ? (int)(maxdp[row-1][col-1] % (1e9+7)): -1;
    }
}
