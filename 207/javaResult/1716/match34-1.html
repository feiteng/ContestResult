class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        long[][] dpMax = new long[m + 1][n + 1];
        long[][] dpMin = new long[m + 1][n + 1];
        
        
for (int i = 0; i <= n ; i++) {
            dpMax[0][i] = 1;
            dpMin[0][i] = 1;
        }
        
        for (int i = 0; i <= m; i++) {
            dpMax[i][0] = 1;
            dpMin[i][0] = 1;
        }
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (i == 1 && j == 1) {
                    dpMax[i][j] = grid[i - 1][j - 1];
                    dpMin[i][j] = grid[i - 1][j - 1];
                }
                else if (i == 1) {
                    dpMax[i][j] = 
Math.max(grid[i - 1][j - 1] * dpMax[i][j - 1], grid[i - 1][j - 1] * dpMin[i][j - 1]);
                    dpMin[i][j] = Math.min(grid[i - 1][j - 1] * dpMax[i][j - 1], grid[i - 1][j - 1] * dpMin[i][j - 1]);
                }
                else if (j == 1) {
                     dpMax[i][j] = Math.max(grid[i - 1][j - 1] * dpMax[i - 1][j], grid[i - 1][j - 1] * dpMin[i - 1][j]);
                    dpMin[i][j] = Math.min(grid[i - 1][j - 1] * dpMax[i - 1][j], grid[i - 1][j - 1] * dpMin[i - 1][j]);
                }
                else {
                    
               
                
                dpMax[i][j] = Math.max(Math.max(grid[i - 1][j - 1] * dpMax[i][j - 1], grid[i - 1][j - 1] * dpMin[i][j - 1]), Math.max(grid[i - 1][j - 1] * dpMax[i - 1][j], grid[i - 1][j - 1] * dpMin[i - 1][j]));
                
                dpMin[i][j] = Math.min(Math.min(grid[i - 1][j - 1] * dpMax[i][j - 1], grid[i - 1][j - 1] * dpMin[i][j - 1]), Math.min(grid[i - 1][j - 1] * dpMax[i - 1][j], grid[i - 1][j - 1] * dpMin[i - 1][j]));
              
}  
                
            }
        }
        
        return dpMax[m][n] < 0 ? -1 : (int)(dpMax[m][n] % 1000000007);
        
    }
}
