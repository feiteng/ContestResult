class Solution {
    public String reorderSpaces(String text) {
        
        int space = 0;
        
        for (char c : text.toCharArray()) {
            if (c == ' ') space++;
        }
        
        String[] s = text.trim().split(" +", 0);
        
        StringBuilder sb = new StringBuilder();
        
        int count = s.length;
        
if (count == 1) {
            sb.append(s[0]);
            for (int i = 0; i < space; i++) sb.append(" ");
            return sb.toString();
        }
        int sn = space / (count - 1);
        int extra = space % (count - 1);
        
        for (int j = 0; j < count; j++) {
            sb.append(s[j]);
            if (j != count - 1) {
                for (int i = 0; i < sn; i++) sb.append(" ")
;
            }
            
        }
        

        
        for (int i = 0; i < extra; i++) sb.append(" ");
        
        return sb.toString();
        
        
        
        
    }
}
