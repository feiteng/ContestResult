class Solution {
    public String reorderSpaces(String text) {
        int spaceCounts = 0;
        for (char c : text.toCharArray()) {
            if (c == ' ') {
                spaceCounts++;
            
}
        }
        
        String[] words = text.trim().split("\\s+");
        
        
        int gap = words.length <= 1 ? 0 : spaceCounts / (words.length - 1);
        int trailing = spaceCounts - gap * (words.length - 1);
        
        String spaces = "";
        for (int i = 0; i < gap; i++) {
            spaces += " ";
        }
        
        StringBuilder sb = new StringBuilder();
        for (String word : words) {
            sb.append(sb.length() == 0 ? "" : spaces).append(word);
        
}
        
        for (int i = 0; i < trailing; i++) {
            sb.append(" ");
        }
        
        //System.out.println(Arrays.toString(words));
        return sb.toString();
    }
}
