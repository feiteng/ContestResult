class Solution {
    public String reorderSpaces(String s) {
        String[] words = s.split("\\s+");
        int cnt = 0;
        for (char c : s.toCharArray()) {
            if (c == ' ') cnt++;
        }
        int p = 0;
        for (String word : words) {
            if (word.trim().length() != 0)
                words[p++] = word
;
        }
        StringBuilder sb = new StringBuilder(); 
        if (p == 1) {
            sb.append(words[0]);
            for (int j = 0; j < cnt; j++) sb.append(" ");
            return sb.toString();
        }
        // System.out.println(Arrays.toString(words));
        int avg = cnt / (p-1);
        int left = cnt % (p-1);
        
        for (int i = 0; i < p - 1; i++) {
            sb.append(words[i]);
            for (int j = 0; j < avg; j++) sb.append(" ");
        }
        sb.append(words[p-1]);
        for (int j = 0; j < left; j++) sb.append(" ");
        return sb.toString();
    }
}
