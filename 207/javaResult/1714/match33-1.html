class Solution {
    public String reorderSpaces(String s) {
        int len = s.length();
        int cnt = 0;
        int slots = 0;
        String text = s.trim();
        
for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                cnt++;
            }
            if (text.charAt(i) != ' ' && (i == 0 || text.charAt(i - 1) == ' ')) {
                slots++;
            }
        }
        slots--;
        cnt += s.length() - text.length();
        if (slots == 0) {
            StringBuilder sb = new StringBuilder();
            sb.append(text);
            for (int i = 0; i < cnt; i++) {
                sb.append(" ");
            }
            return sb.toString();
        }
        
        StringBuilder sb =new StringBuilder();
        String[] words = s.split("\\s+");
        int a = cnt / slots;
        int b = cnt % slots;
        
for (String word : words) {
            if (sb.length() != 0) {
                for (int i = 0; i < a; i++) {
                    sb.append(" ");
                }
            }
            sb.append(word);
        }
        for (int i = 0; i < b; i++) {
            sb.append(" ");
        }
        
        return sb.toString();
    }
}
