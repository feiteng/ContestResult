class Solution {
    public String reorderSpaces(String text) {
        
        int spaceCount = 0;
        
        for(char ch : text.toCharArray()) {
            spaceCount += (ch == ' ') ? 1 : 0;
        }
        
        String[] strList = text.trim().split("\\s+");
        
        StringBuffer sb = new StringBuffer();
        
        if(strList.length == 1) {
            sb.append(strList[0]);
            for(int i = 0; i < spaceCount; i++) {
                sb.append(' ');
            }
            return sb.toString();
        }
        
        int spaceBetween = spaceCount/(strList.length-1);
        
        sb.append(strList[0]);
        for(int j = 1; j < strList.length; j++) {
            for(int i = 0; i < spaceBetween; i++) {
                sb.append(' ');
            }
            sb.append(strList[j]);
        }
        int remaining = spaceCount - (spaceBetween * (strList.length-1));
        for(int i = 0; i < remaining; i++) {
            sb.append(' ');
        }
        
        return sb.toString();
    }
}
