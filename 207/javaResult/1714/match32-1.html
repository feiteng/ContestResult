import java.util.List;

class Solution {
    public String reorderSpaces(String text) {
        String[] words = 
text.split(" ");
        int num = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                num++;
            }
        }
        int n = 0;
        for (int i = 0; i < words.length; i++) {
            if (words[i] == null || words[i].length() == 0) {
                continue;
            }
            n++;
        }

        int x = n == 1 ? 0 : num / (n - 1);
        int y = n == 1 ? num : num % (n - 1);
        StringBuilder sb = new StringBuilder();
        boolean fg = false;
        for (int i = 0; i < words.length; i++) {
            if (words[i] == null || words[i].length() == 0) {
                continue;
            }
            if (fg) 
{
                for (int j = 0; j < x; j++) {
                    sb.append(' ');
                }
            }
            sb.append(words[i]);
            fg = true;

        
}
        for (int j = 0; j < y; j++) {
            sb.append(' ');
        }
        return sb.toString();
    }

}
