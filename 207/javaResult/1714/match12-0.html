class Solution {
    public String reorderSpaces(String text) {
        int spaceCnt = 0;
        for(int i = 0; i < text.length(); i++) {
            if(text.charAt(i) == ' ') {
                spaceCnt += 1;
            }
        }
        String[] words = text.trim().split("\\s+");
        if(words.length == 1) {
            return words[0] + space(spaceCnt);
        }
        int gap = spaceCnt / (words.length - 1);
        int last = spaceCnt % (words.length - 1);
        
        StringBuilder ans = new StringBuilder();
        ans.append(words[0]);
        for(int i = 1; i < words.length; i++) {
            ans.append(space(gap)).append(words[i]);
        }
        ans.append(space(last));
        return ans.toString();
    }
    
    private String space(int n) {
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < n; i++) {
            sb.append(" ");
        }
        return sb.toString();
    }
}
