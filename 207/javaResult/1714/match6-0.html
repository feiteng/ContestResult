class Solution {
    public String reorderSpaces(String text) {
        int leadingSpace = 0;
        for (int i = 0; i < text.length() && text.charAt(i) == ' '; i++) leadingSpace++;
        String[] words = text.substring(leadingSpace,text.length()).split("\\s+");
        
        int sCount = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') sCount++;
        }
        StringBuilder newWords = new StringBuilder();
        if (words.length == 1) {
            newWords.append(words[0]);
            for (int i = 0; i < sCount; i++) newWords.append(' ');
            return newWords.toString();
        }
        int perSpace = sCount / (words.length - 1);
        for (int i = 0; i < words.length - 1; i++) {
            newWords.append(words[i]);
            for (int j = 0; j < perSpace; j++) newWords.append(' ');
            sCount -= perSpace;
        }
        newWords.append(words[words.length - 1]);
        for (int i = 0; i < sCount; i++) newWords.append(' ');
        return newWords.toString();
    }
}
