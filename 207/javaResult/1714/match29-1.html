class Solution {
    public String reorderSpaces(String text) {
        int num = 0;
        for (char t : text.toCharArray()) {
            if (t == ' ') 
{
                num++;
            }
        }
        if (num == 0) {
            return text;
        }
        //System.out.println(num);
        String[] s = text.trim().split("\\s+");
        //System.out.println(Arrays.toString(s));
        StringBuilder sb = new StringBuilder();
        if (s.length == 1) {
            
sb.append(s[0]);
            //System.out.println("1");
            for (int i = 0; i < num; i++) {
                sb.append(" ");
            }
            return sb.toString();
        }
        if (num % (s.length - 1) == 0) {
            int n = num / (s.length - 1);
            //System.out.println(n);
            for (int j = 0; j < s.length - 1; j++) {
                String w = s[j];
                sb.append(w);
                for (int i = 0; i < n; i++) {
                    sb.append(" ");
                }
            }
            sb.append(s[s.length - 1]);
            return sb.toString();
        } else {
            //System.out.println(n);
            int n = num / (s.length - 1);
            int remain = num - n * (s.length - 1);
            for (int j = 0; j < s.length - 1; j++) {
                String w = s[j];
                
sb.append(w);
                for (int i = 0; i < n; i++) {
                    sb.append(" ");
                }
            }
            sb.append(s[s.length - 1]);
            for (int i = 0; i < remain; i++) {
                sb.append(" ");
            }
            return sb.toString();
        }
    } 
}
