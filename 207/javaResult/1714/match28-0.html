class Solution {
    public String reorderSpaces(String text) {
    int spaces = 0;
    for (int i = 0; i < text.length(); i++) {
      if (text.charAt(i) == ' ') {
        spaces++;
      }
    }
    if (spaces == 0) {
      return text;
    }
    List<String> ls = new ArrayList<>();
    int start = 0;
    while (start < text.length()) {
      if (text.charAt(start) == ' ') {
        start++;
        continue;
      }

      int end = start;
      while (end < text.length() && text.charAt(end) != ' ') {
        end++;
      }
      ls.add(text.substring(start, end));
      start = end;
    }

    int sz = ls.size();
    StringBuilder sb = new StringBuilder();
    if (sz == 1) {
      sb.append(ls.get(0));
      for (int i = 0; i < spaces; i++) {
        sb.append(" ");
      }
      return sb.toString();
    }

    int each = spaces / (sz - 1);
    for (int i = 0; i < ls.size() - 1; i++) {
      sb.append(ls.get(i));
      for (int j = 0; j < each; j++) {
        sb.append(" ");
      }
    }
    sb.append(ls.get(ls.size() - 1));
    int rem = spaces % (sz - 1);
    for (int i = 0; i < rem; i++) {
      sb.append(" ");
    }
    return sb.toString();
  }
}
