class Solution {
    public String reorderSpaces(String text) {
        int spaces = 0;
        for(int i = 0; i < text.length(); i++){
            if(text.charAt(i) == ' ') spaces++;
        }
        text = text.trim();
        String[] ss = text.split("\\s+", -1);
        System.out.println(Arrays.toString(ss));
        int cnt = 0;
        int left = spaces;
        if(ss.length != 1){
            cnt = spaces / (ss.length - 1);
            left = spaces % (ss.length - 1);
        
}
        StringBuilder ans = new StringBuilder("");
        for(int i = 0; i < ss.length - 1; i++){
            ans.append(ss[i]);
            for(int j = 0; j < cnt; j++) ans.append(' ');
        }
        ans.append(ss[ss.length-1]);
        for(int j = 0; j < left; j++) ans.append(' ');
        return ans.toString();
    }
}
