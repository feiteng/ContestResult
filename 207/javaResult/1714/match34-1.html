class Solution {
    public String reorderSpaces(String text) {
        int spaceCount = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                spaceCount++;
            }
        }
        
        // System.out.println(spaceCount);
        String[] texts = text.split(" ");
        int wordCount = 0;
        for (String t : texts) {
            // System.out.println(t);
            if (t.length() > 0) {
                wordCount++;
            }
        }
        
        // System.out.println(wordCount);
        if (wordCount == 1) {
            return helper(texts, spaceCount, 0, wordCount);
        }
        int numSpace = spaceCount / (wordCount - 1);
        int remainSpace = spaceCount % (wordCount - 1);
        
        return helper(texts, remainSpace, numSpace, wordCount);
    }
    
    String helper(String[] texts, int remainSpace, int numSpace, int wordCount) {
        StringBuilder sb = new StringBuilder();
        for (String t : texts) {
            if (t.length() > 0) 
{
                wordCount--;
                sb.append(t);
                if (wordCount > 0) {
                    for (int i = 0; i < numSpace; i++) {
                        sb.append(" ");
                    }
                }
            
}
        }
        
        for (int i = 0; i < remainSpace; i++) {
            sb.append(" ");
        }
        
        return sb.toString();
    }
}
