class Solution {
    public String reorderSpaces(String text) {
        int spaces = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') spaces++;
        }
        String[] strs = text.trim().split("\\s+");
        StringBuilder sb = new StringBuilder();
        if (strs.length == 1) {
            sb.append(strs[0]);
            for (int j = 0; j < spaces; j++) sb.append(' ');
            return sb.toString();
        }
        
        int div = spaces / (strs.length - 1);
        int r = spaces - (strs.length - 1) * div;
        
        for (int i = 0; i < strs.length; i++) {
            sb.append(strs[i]);
            if (i != strs.length - 1) {
                for (int j = 0; j < div; j++) {
                    sb.append(' ');
                }
                spaces -= div;
            }
        }
        
        for (int i = 0; i < r; i++) sb.append(' ');
        return sb.toString();
    }
}
