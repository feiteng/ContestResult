class Solution {
    public String reorderSpaces(String text) {
        String[] words = text.trim().split("\\s+");
        int spaces = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                spaces++;
            }
        }
        
        int len = words.length > 1 ? spaces / (words.length - 1) : 0;
        int leftover = words.length > 1 ? spaces % (words.length - 1) : spaces;
        StringBuilder sb = new StringBuilder();
        for (String word : words) {
            if (sb.length() > 0) {
                for (int i = 0; i < len; i++) {
                    sb.append(" ");
                }
            }
            sb.append(word);
        }
        for (int i = 0; i < leftover; i++) {
            sb.append(" ");
        }
        return sb.toString();
    }
}
