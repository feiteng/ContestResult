class Solution {
    public String reorderSpaces(String text) {
        List<String> words = new ArrayList<>();
        String[] splited = 
text.split(" ");
        
        int blanks = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                blanks++;
            }
        }
        
        for (String w : splited) {
            if (w.length() > 0) {
                words.add(w);
            }
        }
        
        StringBuilder b = new StringBuilder();
        int N = words.size();
        int remain;
        if (N > 1) {
            remain = blanks % (N - 1);
            blanks /= (N - 1);
            
            
for (String w : words) {
                if (b.length() > 0) {
                    for (int i = 0; i < blanks; i++) {
                        b.append(' ');
                    }
                }
                
                b.append(w);
            }
            
            
        } else {
            remain = blanks;
            for (String w : words) {
                b.append(w);
            }
        }
        
        
for (int i = 0; i < remain; i++) {
            b.append(' ');
        }
        
        return b.toString();
    }
}
