class Solution {
    public String reorderSpaces(String text) {
        
String[] words = text.trim().split("\\s+");
        
        int space = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ')
                ++space;
        }
        int n = words.length;
        int k = (n == 1) ? 0 : space / (n - 1);
        int r = 
(n == 1) ? space : space % (n - 1);
        StringBuilder sb = new StringBuilder();
        for (int j = 0; j < n; j++) {
            sb.append(words[j]);
            if (j != n -1) {
                for (int i = 0 ; i < k; i++) {
                    sb.append(' ');
                }
            }
        }
        
        
for (int i = 0; i < r; i++)
            sb.append(' ');
        
        return sb.toString();
        
    }
}
