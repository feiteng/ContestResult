class Solution {
    public String reorderSpaces(String text) {
        int count = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') count ++;
        }
        String[] words = text.trim().split("\\s+");
        StringBuffer sb = new StringBuffer();
        sb.append(words[0]);
        
        if (words.length == 1) {
            for (int j = 0; j < count; j++) sb.append(' ');
        } else {
            int gap = count / (words.length - 1);
            int rest = count - gap * (words.length - 1);
            
            for (int i = 1; i < words.length; i++) {
                for (int j = 0; j < gap; j++) sb.append(' ');
                sb.append(words[i]);
            }
            
            for (int j = 0; j < rest; j++) sb.append(' ');
        }
        return sb.toString();
    }
}
