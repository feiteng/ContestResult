class Solution {
    public String reorderSpaces(String text) {
        int blankCount = 0;
        for (char c : text.toCharArray()) {
            if (c == ' ') blankCount++;
        }
        String[] word = text.trim().split(" +");
        StringBuilder res = new StringBuilder();
        if (word.length == 1) {
            res.append(word[0]);
            
for (int i = 0; i < blankCount; i++) res.append(' ');
        } else {
            int ceil = blankCount / (word.length - 1);
            int mod = blankCount - ceil * (word.length - 1);
            for (int j = 0; j < word.length; j++) {
                res.append(word[j]);
                if (j != word.length - 1)
                    for (int i = 0; i < ceil; i++) res.append(' ');
            
}
            for (int i = 0; i < mod; i++) res.append(' ');
        }

        return res.toString();
    }
}
