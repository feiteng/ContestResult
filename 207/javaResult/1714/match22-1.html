class Solution {
    public String reorderSpaces(String text) {
        int co = 0;
        
        
for (char c : text.toCharArray()) {
            if (c == ' ') {
                co++;
            }
        }
        
        
String[] words = text.split(" ");
        
        List<String> w = new ArrayList<>();
        
        for (String str : words) {
            if (str.length() != 0) {
                w.add(str);
            }
        }
                StringBuilder b = new StringBuilder();

        
        if (w.size() == 1) {
            b.append(w.get(0));
            
for(int j = 0 ;j < co; j++) {
                b.append(' ');
            }
            return b.toString();
        }
        
        int n = co / (w.size() - 1);
        boolean e = (
w.size() - 1) * n == co ? false : true;
        
        
        for (int i = 0; i < w.size() - 1; i++) {
            b.append(w.get(i));
            for(int j = 0 ;j < n; j++) {
                b.append(' ');
            }
        }
        
        b.append(w.get(w.size() - 1));
        if(e) {
            for(int i = 0 ; i< co - ((w.size() - 1) * n); i++) {
                b.append(' ');    
            }
            
        }
        return b.toString();
        
    }
}
