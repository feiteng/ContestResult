class Solution {
    public String reorderSpaces(String text) {
        int spaceCount = 0;
        for(char c : text.toCharArray()){
            if(c == ' '){
                spaceCount++;
            }
        }
        String[] words = text.trim().split("\\s+");
        int spacesCount = (words.length - 1);
        int space = 0;
        if(spacesCount != 0){
            space = spaceCount / spacesCount;
        }
        var sb = new StringBuilder();
        sb.append(words[0]);
        for(int i = 1; i < words.length; i++){
            for(int j = 0; j < space; j++){
                sb.append(' ');
            }
            sb.append(words[i]);
        }
        for(int i = 0; i < spaceCount - (space * spacesCount); i++){
            sb.append(' ');
        }
        return sb.toString();
    }
}
