class Solution {
    public String reorderSpaces(String text) {
        int count=0;
        for(int i=0;i<text.length();++i)
            if(text.charAt(i)==' ')
                count++;
        String[] dicts=text.split(" ");
        int len=0;
        for(int i=0;i<dicts.length;++i)
            if(dicts[i].length()!=0)
                len++;
        len--;
        String ans="";
        //System.out.println("count="+count+" len="+len);
        int num=len==0?count:count/len;
        for(int i=0;i<dicts.length;++i){
            if(dicts[i].length()!=0){
                ans+=dicts[i];
                for(int j=0;count>0&&j<num;++j){
                    ans+=" ";
                    count--;
                }
                
            }
        
}
        for(int j=0;j<count;++j)
            ans+=" ";
        return ans;
    }
}
