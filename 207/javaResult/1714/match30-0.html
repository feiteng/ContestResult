class Solution {
    public String[] split(String s){
        //boolean b = false;
        List<String> l = new ArrayList<>();
        String tmp = "";
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)!=' '){
                tmp+=s.charAt(i);
            }
            else{
                if(!tmp.equals(""))
                {
                    l.add(tmp);
                    String n = "";
                    tmp = n;
                }
            }
        }
        if(!tmp.equals("")){
            l.add(tmp);
        }
        String[] res = new String[l.size()];
        for(int i=0;i<l.size();i++){
            res[i]=l.get(i);
        }
        return res;
    }
    public String reorderSpaces(String text) {
        StringBuilder sb = new StringBuilder();
        int ct = 0;
        //int wc = 0;
        String[] toks = split(text);
        
        for(int i=0;i<text.length();i++){
            if(text.charAt(i)==' '){
                ct++;
            }
        }
        if(toks.length==1){
            sb.append(toks[0]);
            for(int i=0;i<ct;i++){
                sb.append(" ");
            }
            return sb.toString();
        }
        int sp = ct/(toks.length-1);
        int rem = ct%(toks.length-1);
        for(int i=0;i<toks.length-1;i++){
            sb.append(toks[i]);
            for(int j=0;j<sp;j++){
                sb.append(" ");
            }
        }
        sb.append(toks[toks.length-1]);
        for(int i=0;i<rem;i++){
            sb.append(" ");
        }
        return sb.toString();
    }
}
