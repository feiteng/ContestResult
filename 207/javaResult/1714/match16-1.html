class Solution {
    public String reorderSpaces(String text) {
        
int count = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                   count++;
            }
        }
        String[] w = text.trim().split(" +");
        
if (w.length == 1) {
            StringBuilder sb = new StringBuilder();
            sb.append(w[0]);
            while (count-- > 0) {
                sb.append(' ');
            }
            return sb.toString();
        }
        
        StringBuilder sb = 
new StringBuilder();
        for (int i = 0; i < w.length; i++) {
            sb.append(w[i]);
            int a = count / (w.length - 1);
            while (i + 1 < w.length && a-- > 0) {
                sb.append(' ');
            }
        }
        int a = count % (w.length - 1);
        while (a-- > 0) {
            sb.append(' ');
        }
        return sb.toString();
    }
}
