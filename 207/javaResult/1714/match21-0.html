class Solution {
    public String reorderSpaces(String text) {
        int sps = 0;
        for(char x : text.toCharArray()) {
            if(x == ' ') ++sps;
        }
        String[] sx = text.trim().split("\\s+");
        int n = sx.length;
        int itx = n - 1;
        int sp = 0;
        if(itx != 0) sp = sps / itx;
        int rm = sps;
        if(itx != 0) rm = sps % itx;
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < n - 1; ++i) {
            sb.append(sx[i]);
            for(int k = 0; k < sp; ++k) sb.append(' ');
        }
        sb.append(sx[n - 1]);
        for(int k = 0; k < rm; ++k) sb.append(' ');
        return sb.toString();
    }
}
