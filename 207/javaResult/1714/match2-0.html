class Solution {
    public String reorderSpaces(String text) {
        int count = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') count++;
        }
        
        String[] st = text.trim().split("\\s+");
        int even = st.length == 1 ? 0 : count / (st.length - 1);
        int end = st.length == 1 ? count : count % (st.length - 1);
        //System.out.println(Arrays.toString(st));
        StringBuffer sb = new StringBuffer();
        for (int i = 0; i < st.length; i++) {
            sb.append(st[i]);
            if (i != st.length - 1) {
                for (int k = 0; k < even; k++) sb.append(" ");
            }
        }
        
        for (int k = 0; k < end; k++) {
            sb.append(" ");
        }
        return sb.toString();
    }
}
