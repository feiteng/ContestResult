class Solution {
    public String reorderSpaces(String text) {
        String[] ss = text.trim().split("\\s+");
        int num = 0;
        for(int i = 0;i < text.length();i++) {
            char c = text.charAt(i);
            if(c == ' ') num++
;
        }
        StringBuilder sb = new StringBuilder();
        if(ss.length == 1) {
            sb.append(ss[0]);
            for(int j = 0;j < num;j++) sb.append(' ');
            return sb.toString();
        }
        int val1 = num / (ss.length - 1);
        
int val2 = num % (ss.length - 1);
        
        for(int i = 0;i < ss.length - 1;i++) {
            sb.append(ss[i]);
            for(int j = 0;j < val1;j++) sb.append(' ');
        }
        sb.append(ss[ss.length - 1]);
        for(int j = 0;j < val2;j++) sb.append(' ');
        //for(String s : ss) System.out.println(ss.length);
        return sb.toString();
    }
}
