class Solution {
    public String reorderSpaces(String text) {
        int numSpace = 0;
        for (char c : text.toCharArray()) {
            if (c == ' ')
                numSpace ++;
        }
        text = text.trim();
        String[] strs = text.split("[ ]+");
        int maxLen = 0;
        if (strs.length > 1)
            maxLen = numSpace / (strs.length-1);
        numSpace -= maxLen * (strs.length-1);
        StringBuilder sb = new StringBuilder();
        for (int j = 0; j < strs.length; ++j) {
            sb.append(strs[j]);
            if (j != strs.length - 1) {
                for (int i = 0; i < maxLen; ++i) {
                    sb.append(' ');
                }
            }
        }
        
        for (int j = 0; j < numSpace; ++j) {
            sb.append(' ');
        }
        return sb.toString();
    }
}
