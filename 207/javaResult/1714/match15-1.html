class Solution {
    public String reorderSpaces(String text) {
        String[] args = text.trim().split("\\s+");
        int count = 0;
        for(String word : args)
            count += word.length();
        
        int space = args.length > 1 ? (text.length() - count)/(args.length-1) : 0;
        int left = args.length > 1 ? (text.length() - count) % (args.length-1) : text.length() - count;
        
        
StringBuilder sb = new StringBuilder();
        for(int i = 0; i<args.length; i++) {
            sb.append(args[i]);
            for(int j = 0; i < args.length-1 && j<space; j++)
                sb.append(" ");
        }
        
        
for(int j = 0; j<left; j++)
            sb.append(" ");
        
        return sb.toString();
    }
}
