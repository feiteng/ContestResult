class Solution {
    
    int res;
    
    public int maxUniqueSplit(String s) {
        res = 0;
        backtrack(s, new HashSet<String>(), 0, 0);
        return res;
    }
    
    private void backtrack(String s, Set<String> seen, int pos, int count) {
        if (pos == s.length()) {
            res = Math.max(res, count);
            return;
        }
        
        for (int i = pos + 1; i <= s.length(); i++) {
            String str = s.substring(pos, i);
            if (!seen.contains(str)) {
                seen.add(str);
                backtrack(s, seen, i, count+1);
                seen.remove(str);
            }
        }
    }
}
