class Solution {
    int res = 1;
    public int maxUniqueSplit(String s) {
        dfs(s, 0, new HashSet<>());
        return res;
    }
    
    private void dfs(String s, int i, Set<String> set) {
        int n = s.length();
        if (i == n) {
            res = Math.max(res, set.size());
            return;
        
}
        for (int j = i; j < n; j++) {
            String cur = s.substring(i, j + 1);
            if (set.contains(cur)) continue;
            set.add(cur);
            dfs(s, j + 1, set);
            set.remove(cur);
        }
    }
}
