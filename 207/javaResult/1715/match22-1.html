class Solution {
    public int maxUniqueSplit(String s) {
        int[] ans = {0};
        maxUniqueSplit(s, 0, new HashSet<>(), ans);
        return ans[0];
    }

    private void maxUniqueSplit(String s, int i, Set<String> visited, int[] ans) {
        int len = s.length();
        if (i == len) {
            ans[0] = Math.max(ans[0], visited.size());
            return;
        }
        for (int j = i + 1; j <= len && visited.size() + len - j >= ans[0]; j++) 
{
            String sub = s.substring(i, j);
            if (visited.add(sub)) {
                maxUniqueSplit(s, j, visited, ans);
                visited.remove(sub);
            }
        }
    }
}
