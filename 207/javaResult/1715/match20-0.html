class Solution {
    int ans = 0;
    public int maxUniqueSplit(String s) {
        HashSet<String>hashset = new HashSet(); 
        dfs(0,s,hashset);
        return ans;
    }
    public void dfs(int index,String s,HashSet<String>path){
        if(index==s.length()){
            ans=Math.max(ans,path.size());
            return ;
        }
        StringBuilder t = new StringBuilder();
        for(int i=index;i<s.length();i++){
            t.append(s.charAt(i));
            if(path.contains(t.toString()))continue;
            path.add(t.toString());
            dfs(i+1,s,path);
            path.remove(t.toString());
        }
    }
}
