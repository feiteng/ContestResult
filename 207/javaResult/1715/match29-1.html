class Solution {
    Set<String> set = new HashSet();
    int max = 0;
    public int maxUniqueSplit(String s) {
        visit(s, 0);
        return max;
    }
    private void visit(String s, int index) {
        if (index == s.length()) {
            max = Math.max(max, set.size());
            return ;
        }
        for (int i = index; i < s.length(); i++) {
            String cur = s.substring(index, i + 1);
            if (set.contains(cur)) {
                continue;
            
} 
            set.add(cur);
            visit(s, i + 1);
            set.remove(cur);
        }
    }
}
