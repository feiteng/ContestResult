class Solution {
    int max =1;
    public int maxUniqueSplit(String s) {
        Set<String> visited =new HashSet<>();
        core(s, 0, visited);
        return max;
    }
    private void core(String s, int index, Set<String> visited){
        if(index == s.length()){
            max = Math.max(visited.size(), max);
            return;
        }
        
        for(int i = 1; i+index<=s.length(); i++){
            String sub = s.substring(index, i+index);
            if(visited.contains(sub)){
                continue;
            }
            visited.add(sub);
            core(s, i+index, visited);
            visited.remove(sub);
            
        }
    }
}
