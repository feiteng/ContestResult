class Solution {
    int res=0;
    public int maxUniqueSplit(String s) {
        dfs(s,0,new HashSet<>());
        return res;
    }
    
    public void dfs(String s,int index,Set<String>cur){
        if(index>=s.length()){
            res=Math.max(res,cur.size());
            return;
        
}
        for(int i=index;i<s.length();i++){
            String sub=s.substring(index,i+1);
            if(!cur.contains(sub)){
                cur.add(sub);
                dfs(s,i+1,cur);
                cur.remove(sub);
            }
        }
    }
}
