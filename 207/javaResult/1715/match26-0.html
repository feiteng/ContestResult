class Solution {
    int res = 1;
    public int maxUniqueSplit(String s) {
        dfs(new HashSet<>(), s, 0);
        return res;
    }public void dfs(Set<String> set, String s, int index){
        int n = s.length();
        if(index == n){
            res = Math.max(res, set.size());
        }
        for(int i = index + 1; i <= n; i++){
            String str = s.substring(index, i);
            if(set.contains(str)) continue;
            set.add(str);
            dfs(set,s,i);
            set.remove(str);
        }
    }
}
