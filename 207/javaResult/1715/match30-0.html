class Solution {
    public int maxUniqueSplit(String s) {
        return rec(s, 0, new HashSet<>());
    }
    
    int rec(String s, int start, Set<String> set) {
        int output = 0;
        for(int i = start; i < s.length(); i++) {
            String sub = s.substring(start, i + 1);
            if(!set.contains(sub)) {
                set.add(sub);
                output = Math.max(output, 1 + rec(s, i + 1, set));
                set.remove(sub);
            }
        }
        return output;
    } 
    
}
