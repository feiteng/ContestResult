class Solution {
    public int maxUniqueSplit(String s) {
        return count(s, 0, new HashSet());
    }
    
    private int count(String s, int start, Set<String> set){
        if(start >= s.length()){
            return set.size();
        }
        int result = 0;
        for(int i = start + 1; i <= s.length(); i++){
            String sub = s.substring(start, i);
            if(!set.contains(sub)){
                set.add(sub);
                result = Math.max(result, count(s, i, set));
                set.remove(sub);
            }
        }
        return result;
    }
}
