class Solution {
    int res = 0;
    public int maxUniqueSplit(String s) {
        dfs(s, 0, new HashSet<>(), 0);
        return res;
    }
    private void dfs(String s, int idx, Set<String> set, int cur) {
        if (idx == s.length()) {
            res = Math.max(cur, res);
            return;
        }
        for (int i = idx;i < s.length();i++) {
            String sub = s.substring(idx, i + 1);
            if (set.contains(sub)) continue;
            set.add(sub);
            dfs(s, i + 1, set, cur + 1);
            set.remove(sub);
        }
    }
}
