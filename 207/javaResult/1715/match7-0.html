class Solution {
    int ans = 0;
    public int maxUniqueSplit(String s) {

        dfs(s, new HashSet<>(), 0);
        return ans;
    }
    
    private void dfs(String s, Set<String> used, int index) {
        //
        if (index == s.length()) {
            ans = Math.max(ans, used.size());
        }
        
        //backtracking
        for (int i = index; i < s.length(); ++i) {
            String substring = s.substring(index, i + 1);
            if (!used.contains(substring)) {
                used.add(substring);
                dfs(s, used, i + 1);
                used.remove(substring);
            }
        }
        
    }
}
