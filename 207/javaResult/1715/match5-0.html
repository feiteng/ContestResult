class Solution
{
    private int max = 1;
    
    public int maxUniqueSplit(String s)
    {
        maxUniqueSplitHelper(s, new HashSet<>(), 0);
        return max;
    }
    
    private void maxUniqueSplitHelper(String s, Set<String> strSet, int startIdx)
    {
        if (startIdx == s.length())
        {
            max = Math.max(max, strSet.size());
            return;
        }
        for (int i = startIdx + 1; i <= s.length(); i++)
        {
            String subStr = s.substring(startIdx, i);
            if (!strSet.add(subStr))
            {
                continue;
            }
            maxUniqueSplitHelper(s, strSet, i);
            strSet.remove(subStr);
        }
    }
}
