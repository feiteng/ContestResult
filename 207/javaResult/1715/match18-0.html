class Solution {
  int res = 0;
  public int maxUniqueSplit(String s) {
    Set<String> has = new HashSet<>();
    rec(s, 0, has);
    return res;
  }

  void rec(String s, int id, Set<String> has) {
    if (id == s.length()) {
      res = Math.max(res, has.size());
      return;
    }
    for (int nxt = id; nxt < s.length(); nxt++) {
      String cur = s.substring(id, nxt + 1);
      if (!has.contains(cur)) {
        has.add(cur);
        rec(s, nxt + 1, has);
        has.remove(cur);
      }
    }
  }
}
