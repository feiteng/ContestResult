class Solution {
    int res = 1;
    public int maxUniqueSplit(String s) {
        calc(s, 0, new HashSet());
        return res;
    }
    
    public void calc(String s, int start, Set<String> seen){
        if(start==s.length()){
            res = Math.max(res, seen.size());
        }
        for(int i=start; i<s.length(); i++){
            String cur = s.substring(start, i+1);
            if(!seen.contains(cur)){
                seen.add(cur);
                calc(s, i+1, seen);
                seen.remove(cur);
            }
        }
    }
}
