class Solution {
    int max = 0;
    public int maxUniqueSplit(String s) {
        Set<String> set = new HashSet<>();
        dfs(s, 0, set);
        return max;
    }
    
    private void dfs(String s, int start, Set<String> set) {
        if(start == s.length()) {
            max = Math.max(set.size(), max);
            return ;
        }
        for(int i = start; i < s.length(); ++i) {
            String sub = s.substring(start, i + 1);
            if(set.contains(sub))
                continue
;
            set.add(sub);
            dfs(s, i + 1, set);
                
            set.remove(sub);
        }
       
    }
}
