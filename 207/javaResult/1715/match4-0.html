class Solution {
    int max = 0;
    public int maxUniqueSplit(String s) {
        dfs(s,0,new HashSet<String>(), 0);
        return max;
    }
    private void dfs(String s, int cur, HashSet<String>set, int num){
        if(cur == s.length()){
            max = Math.max(num,max);
            return;
        }

        for(int i = cur; i < s.length(); i ++){
            String temp = s.substring(cur,i+1);
            if(set.contains(temp)) continue;
            set.add(temp);
            dfs(s,i+1,set,num+1);
            set.remove(temp);
        }
    }
}
