class Solution {
    int ansr;
    public int maxUniqueSplit(String s)
    {
        ansr=0;
        backTracking(new HashSet<>(),s,0,0);
        return ansr;
    }
    public void backTracking(HashSet<String> exists,String s,int index,int num)
    {
        if (index>=s.length())
        {
            ansr = Math.max(ansr, num);
            return;
        }
        if (num+s.length()-index<=ansr)
        {
            return;
        }
        HashSet<String> next=new HashSet<>(exists);
        
for (int i = index+1; i <= s.length(); i++)
        {
            String sub = s.substring(index, i);
            if (!exists.contains(sub))
            {
                next.add(sub);
                backTracking(next, s, i, num + 1);
                next.remove(sub);
            }
        }
    }
}
