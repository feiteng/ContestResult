class Solution {
    public int maxUniqueSplit(String s) {
        int[] res = new int[]{1};
        
        find(s, 0, new HashSet<>(), res);
        return res[0];
    }
    
    public void find(String s, int start, Set<String> set, int[] res) {
        if (start == s.length()) {
            res[0] = Math.max(res[0], set.size());
            return;
        }
        
        for (int i = start + 1; i <= s.length(); i++) {
            String subs = s.substring(start, i);
            if (!set.contains(subs)) {
                set.add(subs);
                find(s, i, set, res);
                set.remove(subs);
            }
        }
    }
}
