class Solution {
    private int max = 0;
    public int maxUniqueSplit(String s) {
        Set<String> set = new HashSet<>();
        dfs(s, 0, set);
        return max;
    }
    
    private void dfs(String s, int start, Set<String> set) {
        if (start >= s.length()) {
            max = Math.max(max, set.size());
            return;
        }
        
for (int i = 1; i <= s.length() - start; i++) {
            String sb = s.substring(start, start + i);
            if (!set.contains(sb)) {
                set.add(sb);
                dfs(s, start + i, set);
                set.remove(sb);
            } 
        }
    }
}
