class Solution {
    int ans = 0;
    public int maxUniqueSplit(String s) {
        dfs(s, 0, new HashSet<>());
        return ans;
    }
    
    private void dfs(String s, int start, Set<String> split) {
        if (start >= s.length()) {
            ans = Math.max(ans, split.size());
            return;
        }
        for (int i = start; i < s.length(); i++) {
            String prefix = s.substring(start, i + 1);
            if (!split.add(prefix)) {
                continue;
            }
            dfs(s, i + 1, split);
            split.remove(prefix);
        }
    }
}
