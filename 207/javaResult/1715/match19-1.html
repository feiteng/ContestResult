class Solution {
    public int maxUniqueSplit(String s) {
        return backtrack(s, 0, new HashSet<>());
    }
    
    private int backtrack(String s, int pos, Set<String> set){
        if(pos == s.length()){
            return set.size();
        }
        int max = 0;
        for(int i = pos; i<s.length(); i++){
            String curr = s.substring(pos, i+1);
            if(!set.contains(curr)){
                set.add(curr);
                max = Math.max(max, backtrack(s, i+1, set));
                set.remove(curr);
            }
        }
        return max;
    }
}
