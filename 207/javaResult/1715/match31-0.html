import java.util.HashSet;

class Solution {
  
  public static void main(String args[]) {
    String input = "ababccc";
    System.out.println(maxUniqueSplit(input));
  }

  public static int maxUniqueSplit(String s) {
    return getMax(s, 0, new HashSet<>());
  }
  
  public static int getMax(String s, int index, HashSet<String> set) {
    if (index >= s.length()) {
      return set.size();
    }
    int best = 0;
    
    for (int i = index + 1; i <= s.length(); i++) {
      String substring = s.substring(index, i);
      if (!set.contains(substring)) {
        set.add(substring);
        best = Math.max(best, getMax(s, i, set));
        set.remove(substring);
      }
    }
    return best;
  }
}
