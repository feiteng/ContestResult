class Solution {
    public int maxUniqueSplit(String s) {
        int[] res = new int[] {1};
        dfs(s, 0, new HashSet<>(), res);
        return res[0];
    }
    
    private void dfs(String s, int index, Set<String> set, int[] res) {
        int n = s.length();
        if (n == index) {
            res[0] = Math.max(set.size(), res[0]);
            return;
        }
        for (int i = index; i < n; i++) {
            String sub = s.substring(index, i + 1);
            if (set.add(sub)) {
                dfs(s, i + 1, set, res);
                set.remove(sub);
            }
        }
    }
}
