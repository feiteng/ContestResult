class Solution {
    int max = 0;
    public int maxUniqueSplit(String s) {
        Set<String> set = new HashSet<>();
        helper(set, s, 0);
        return max;
    }
    
    private void helper(Set<String> set, String s, int start) {
        if(start == s.length()) {
            max = Math.max(max, set.size());
            return;
        }
        
        for(int i=start+1; i<=s.length(); i++) {
            String curr = s.substring(start, i);
            if(set.contains(curr)) {
               continue; 
            }
            set.add(curr);
            helper(set, s, i);
            set.remove(curr);
        }
    }
}
