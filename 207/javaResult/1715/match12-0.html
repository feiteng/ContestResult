        Set<String> set = new HashSet<>();
        backtrack(s, 0, set);
        return ans;
    }
    
    private void backtrack(String s, int idx, Set<String> set) {
        if (idx == s.length()) {
            ans = Math.max(ans, set.size());
            return;
        }
        for (int i = idx+1; i <= s.length(); i++) {
            String temp = s.substring(idx, i);
            if (set.contains(temp)) {
                continue;
            }
            set.add(temp);
            backtrack(s, i, set);
            set.remove(temp);
        }
    }
}
