class Solution {
    private int res = 0;

    public int maxUniqueSplit(String s) {
        Set<String> set = new HashSet<>();
        helper(s, 0, set);
        return res;
    }
    
    private void helper(String s, int i, Set<String> set) {
        if (i == s.length()) {
            res = Math.max(res, set.size());
            return;
        }
        int res = 0;
        
for (int j = i+1; j <= s.length(); j++) {
            String str = s.substring(i, j);
            if (!set.contains(str)) {
                set.add(str);
                helper(s, j, set);
                set.remove(str);
            }
        }
    }
}
