class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size(), m = cost[0].size();
        std::vector<int> f(1 << m, std::numeric_limits<int>::max());
        for (int i = 0; i < m; i++) {
            f[1 << i] = cost[0][i];
        }
        for (int i = 1; i < n; i++) {
            std::vector<int> g(1 << m, std::numeric_limits<int>::max());
            for (int j = 0; j < (1 << m); j++) {
                if (f[j] == std::numeric_limits<int>::max()) { continue; }
                for (int k = 0; k < m; k++) {
                    int val = f[j] + cost[i][k];
                    g[j | (1 << k)] = std::min(val, g[j | (1 << k)]);
                }
            }
            f = g;
        }
        int ans = f[(1 << m) - 1];
        for (int i = 1; i < (1 << m); i++) {
            int val = f[i];
            for (int j = 0; j < m; j++) {
                if (i & (1 << j)) { continue; }
                int num = cost[0][j];
                for (int k = 0; k < n; k++) {
                    if (cost[k][j] < num) {
                        num = cost[k][j];
                    }
                }
                val += num;
            }
            ans = std::min(ans, val);
        }
        return ans;
    }
};
