class Solution {
public:
    static const int maxn=13;
    int dp[maxn][1<<maxn];
    int connectTwoGroups(vector<vector<int>>& cst) {
        int n=cst.size();
        int m=cst[0].size();

        memset(dp, 0x3f, sizeof(dp));
        dp[0][0]=0;
        for (int i=1;i<=n;++i){
            for (int j=0;j<(1<<m);++j){
                for (int cas=0;cas<m;++cas){
                    for (int k=0;k<m;++k){
                        dp[i][j|(1<<k)]=min(dp[i][j|(1<<k)], dp[i-1][j]+cst[i-1][k]);
                        dp[i][j|(1<<k)]=min(dp[i][j|(1<<k)], dp[i][j]+cst[i-1][k]);
                    }
                }
            }
        }
        
        return dp[n][(1<<m)-1];
    }
};
