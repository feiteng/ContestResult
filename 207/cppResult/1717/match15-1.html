class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int s1 = cost.size();
        int s2 = cost[0].size();
        vector<vector<int>> dp(s1 + 1, vector<int>(1 << s2, 1e9));
        dp[0][0] = 0;

        for (int i = 0; i < s1; i++) {
            for (int j = 0; j < (1 << s2); j++) {
                if (dp[i][j] != 1e9) {
                    for (int k = 0; k < s2; k++) {
                        int msk = 1 << k;
                        dp[i + 1][j | msk] = min(dp[i + 1][j | msk], dp[i][j] + cost[i][k]);
                    }
                }
            }
        }

        vector<int> mini(s2, 1e9);
        for (int i = 0; i < s2; i++) {
            for (int j = 0; j < s1; j++) {
                mini[i] = min(mini[i], cost[j][i]);
            }
        }
        
        int ans = 1e9;

        for (int j = 1; j < (1 << s2); j++) {
            int val = dp[s1][j];
            for (int k = 0; k < s2; k++) {
                if (!(j & (1 << k))) {
                    val += mini[k];
                }
            }
            ans = min(ans, val);
        }
        return ans;
    }
};
