class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size(), m = cost[0].size(), inf = numeric_limits<int>::max();
        vector<int> low(n, inf);
        for (int i=0; i<n; ++i) {
            for (int j=0; j<m; ++j) {
                low[i] = min(low[i], cost[i][j]);
            }
        }
        int t = 1<<n;
        vector<int> f(t, 0);
        for (int i=0; i<t; ++i) {
            for (int j=0; j<n; ++j) {
                if (i & (1<<j)) f[i] += low[j];
            }
        }
        for (int j=0; j<m; ++j) {
            vector<int> ff(t, inf);
            for (int x=0; x<t; ++x) {
                for (int i=0; i<n; ++i) {
                    ff[x] = min(ff[x], cost[i][j] + min(f[x], f[x^(1<<i)]));
                }
            }
            swap(f, ff);
        }
        return f[t-1];
    }
};
