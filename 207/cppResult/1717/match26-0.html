class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        const int m = cost.size();
        const int n = cost[0].size();
        vector<int> dp(1 << n, INT_MAX);
        vector<int> last_dp(1 << n, INT_MAX);
        dp[0] = 0;
        for (int i = 0; i < m; ++i) {
            swap(dp, last_dp);
            fill(dp.begin(), dp.end(), INT_MAX);
            for (int state = 1; state < (1 << n); ++state) {
                for (int j = 0; j < n; ++j) {
                    if (state & (1 << j)) {
                        int prev = (state & ~(1 << j));
                        int prev_val = min(last_dp[prev], min(last_dp[state], dp[prev]));
                        if (prev_val < INT_MAX) {
                            dp[state] = min(dp[state], prev_val + cost[i][j]);
                        }
                    }
                }
            }
        }
        return dp.back();
    }
};
