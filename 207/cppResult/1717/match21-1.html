class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int INF = 0x3fffffff;
        int m = cost[0].size();
        vector<int> dp(1 << m, INF);
        dp[(1 << m) - 1] = 0;
        for (int i = 0;i < cost.size();i++) {
            vector<int> test(1 << m, 0);
            vector<int> temp(1 << m, INF);
            

            for (int j = 0;j < (1 << m);j++) {
                for (int k = 0;k < m;k++) {
                    if (j & (1 << k)) {
                        test[j] += cost[i][k];
                    }
                }
            }
            for (int j = 0;j < (1 << m);j++) {
                if (dp[j] == INF) {
                    continue;
                }
                for (int k = j;k != 0;k = (k - 1) & j) {
                    int key = j - k;
                    int cost = dp[j] + test[k];
                    temp[key] = min(temp[key], cost);
                }
                int p = INF;
                for (int k = 0;k < m;k++) {
                    if (!(j & (1 << k))) {
                        p = min(p, cost[i][k]);
                    }
                }
                if (p != INF) {
                    temp[j] = min(temp[j], dp[j] + p);
                }
            }
            dp = temp;
        }
        return dp[0];
    }
};
