class Solution {
    int f[15][5005];
    int a[15][5005];
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size();
        int m=cost[0].size();
        for(int i=0;i<n;i++) {
            for(int j=0;j<(1 << m);j++) {
                for(int k=0;k<m;k++) {
                    if (j & (1 << k)) a[i][j]+=cost[i][k];
                }
            }
        }
        memset(f,63,sizeof(f));
        for(int j=1;j<(1 << m);j++) f[0][j]=a[0][j];
        for(int i=0;i<n-1;i++){
            for(int j=1;j<(1 << m);j++) {
                int tmp = ((1 << m)-1) ^ j;
                for(int k=tmp;k>0;k=(k-1) & tmp) {
                    f[i+1][j | k] = min(f[i+1][j | k],f[i][j]+a[i+1][k]);
                }
                for(int l=0;l<m;l++) {
                    int k = (1 << l);
                    f[i+1][j | k] = min(f[i+1][j | k],f[i][j]+a[i+1][k]);
                }
            }
        }
        return f[n-1][(1 << m)-1];
    }
};
