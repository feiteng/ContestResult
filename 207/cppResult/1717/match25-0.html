class Solution {
public:
    int n, m;
    int connectTwoGroups(vector<vector<int>>& c) {
        n = c.size();
        m = c[0].size();
        long long dp[15][9000] = {0};
        for (int i = 0; i <= n; i++) {
            for (int k = 0; k < (1 << m); k++){
                dp[i][k] = INT_MAX;
            }
        }
        dp[0][0] = 0;
        
        for (int i = 0; i < n; i++) {
            for (int k = 0; k < (1 << m); k++){
                for (int r = 0; r < m; r++) {
                    for (int j = 0; j < m; j++) {
                        if (((k >> j) & 1) != 0) {
                            dp[i+1][k] = min(dp[i+1][k], dp[i][k] + c[i][j]);
                            dp[i+1][k] = min(dp[i+1][k], dp[i+1][k] + c[i][j]); 
                        }
                        else {
                            int s = k | (1 << j);
                            dp[i+1][s] = min(dp[i+1][s], dp[i][k] + c[i][j]);
                            dp[i+1][s] = min(dp[i+1][s], dp[i+1][k] + c[i][j]);
                        }
                    }
                    
                }
                 
            }
        }
        return dp[n][(1 << m) - 1];
    }
};
