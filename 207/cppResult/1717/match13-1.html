using namespace std;
#define rep(i, a, b) for(int i = a; i < (b); ++i)
#define trav(a, x) for(auto& a : x)
#define all(x) begin(x), end(x)
#define sz(x) (int)(x).size()
typedef long long ll;
typedef pair<int, int> pii;
typedef vector<int> vi;


class Solution {
public:
    int F[13][(1 << 12)];
    int connectTwoGroups(vector<vector<int>>& cost) {
        // 0, ..., n-1 matches 0, ..., m-1
        // init F[n][i]
        int n = cost.size();

        int m = cost[0].size();
        for(int S = 0; S < (1 << m); S++){
            int t = 0;
            for(int i = 0; i < m; i++){
                if((S & (1 << i)) == 0){
                    int m = 105;
                    for(int j = 0; j < n; j++){
                        m = min(m, cost[j][i]);
                    }
                    t += m;
                }
            }
            F[n][S] = t;
        }
        

        for(int k = n-1; k >= 0; k--){
            for(int S = 0; S < (1 << m); S++){
                // Compute F[k][S]
                int t = 24000;
                for(int i = 0; i < m; i++){
                    t = min(t, F[k+1][S | (1 << i)] + cost[k][i]);
                }
                F[k][S] = t;
            }
        }
        
        return F[0][0];
    }
};
