class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size(), m = cost[0].size();
        vector<vector<int>> dp(2, vector<int>(1<<m, INT_MAX));

        int cur = 0, nxt = 1;
        dp[0][0] = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < (1<<m); j++) {
                if(dp[cur][j] == INT_MAX) continue;
                for(int k = 0; k < m; k++) {
                    dp[nxt][j|(1<<k)] = min(dp[nxt][j|(1<<k)], dp[cur][j]+cost[i][k]);
                }
            }
            for(int j = 0; j < (1<<m); j++) {
                if(dp[nxt][j] == INT_MAX) continue;

                for(int k = 0; k < m; k++) {
                    dp[nxt][j|(1<<k)] = min(dp[nxt][j|(1<<k)], dp[nxt][j]+cost[i][k]);
                }
            }
            swap(cur, nxt);
            fill(begin(dp[nxt]), end(dp[nxt]), INT_MAX);
        }
        return dp[cur][(1<<m)-1];
    }
};
