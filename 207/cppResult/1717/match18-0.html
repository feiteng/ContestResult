class Solution {
public:
	int connectTwoGroups(vector<vector<int>> cost) {
		int n = cost.size(), m = cost[0].size();
		vector<vector<int>>dp(n+1, vector<int>(1 << m, 1e8));
		dp[0][0] = 0;
        
        vector<vector<int>>a(1<<m);
        
        for(int i=0;i<(1<<m);i++){
            for(int j=1;j<(1<<m);j++){
                if((i&j)==0)a[i].push_back(j);
            }
            for(int j=0;j<m;j++){
                a[i].push_back(1<<j);
            }
        }
		for (int i = 1; i <= n; i++) {
			vector<int>p(1 << m, 0);
			for (int j = 0; j < (1 << m); j++) {
				for (int k = 0; k < m; k++) {
					if (j & (1 << k))p[j] += cost[i - 1][k];
				}
			}

			for (int j = 0; j < (1 << m); j++) {
				for (auto k:a[j]) {
					dp[i][j | k] = min(dp[i][j | k], dp[i - 1][j] + p[k]);
				}
			}
		}
		return dp[n][(1 << m) - 1];
	}
};
