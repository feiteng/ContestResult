class Solution {
 public:
  int connectTwoGroups(vector<vector<int>>& c) {
    int n = c.size(), m = c[0].size();
    vector<int> F(1 << n), G(1 << n);
    auto cal = [&](int v, vector<int>& A) {
      fill(A.begin(), A.end(), 0);
      for (int i = 1; i < (1 << n); ++i) {
        for (int j = 0; j < n; ++j)
          if (i >> j & 1) {
            A[i] += c[j][v];
          }
      }
    };
    cal(0, F);
    for (int i = 1; i < m; ++i) {
      cal(i, G);

      vector<int> tp(1 << n, 1e9);
      for (int a = 1; a < (1 << n); ++a) {
        for (int b = a; b; b = (b - 1) & a) {
          if (a ^ b) tp[a] = min(tp[a], F[b] + G[a ^ b]);
        }
      }
      for (int a = 1; a < (1 << n); ++a) {

        for (int b = a; b; b = (b - 1) & a) {
          tp[b] = min(tp[a], tp[b]);
        }
      }
      swap(tp, F);
    }
    return F[(1 << n) - 1];
  }
};
