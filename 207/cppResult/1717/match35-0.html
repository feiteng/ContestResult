class Solution {
    int N, M;
    vector<vector<int>> C;
    
    vector<vector<int>> memo;
    
    int f(int pos, int mask)
    {
        if(pos>=N)
        {
            /*

            else
            {

            */
            int result=0;
            for(int j=0; j<M; j++)
            {
                if(bit(mask, j)) continue;
                
                int curr=INF;
                for(int i=0; i<N; i++) curr=min(curr, C[i][j]);
                
                result+=curr;
            }
            return result;
        }
        
        if(memo[pos][mask]!=-1) return memo[pos][mask];
        int result=INF;
        
        for(int j=0; j<M; j++)
        {
            result=min(result, C[pos][j]+f(pos+1, mask|(1<<j)));
        }
        
        memo[pos][mask]=result;
        return result;
    }
    
public:
