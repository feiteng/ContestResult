int dp[20][100000];
class Solution {
public:
    
    int a[20];
    
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n=cost.size(),m=cost[0].size();
        memset(a,0x3f,sizeof a);

        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                a[i]=min(a[i],cost[j][i]);
            }
        }
        int ed=1<<m;
        memset(dp,0x3f,sizeof dp);

        dp[0][0]=0;
        for(int i=1;i<=n;i++){
            for(int j=0;j<ed;j++){
                for(int k=0;k<m;k++){
                    int st=j|(1<<k);
                    dp[i][st]=min(dp[i-1][j]+cost[i-1][k],dp[i][st]);
                }
            }
        }
        int ans=0x3f3f3f3f;
        for(int i=0;i<ed;i++){
            int res=dp[n][i];
            for(int j=0;j<m;j++){
                if((i>>j)&1) continue;
                res+=a[j];
            }
            ans=min(ans,res);
        }
        return ans;
    }
};
