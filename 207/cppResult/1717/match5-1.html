class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int left=cost.size();
        int right=cost[0].size();
        int **dp=new int*[left+1];
        for (int i=0;i<=left;i++){
            dp[i]=new int[1<<right];
            for (int j=0;j<(1<<right);j++){
                dp[i][j]=INT_MAX-100;
            }
        }

        dp[0][0]=0;
        for (int i=1;i<=left;i++){
            for (int j=0;j<right;j++){
                for (int k=0;k<(1<<right);k++){
                    dp[i][k|(1<<j)]=min(dp[i][k|(1<<j)],dp[i-1][k]+cost[i-1][j]);
                    dp[i][k|(1<<j)]=min(dp[i][k|(1<<j)],dp[i][k]+cost[i-1][j]);
                }
            }
        }
        return dp[left][(1<<right)-1];
    }
};
