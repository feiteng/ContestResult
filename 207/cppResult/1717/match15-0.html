class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int N = cost.size();
        int M = cost[0].size();
        int F[N][1<<M];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < (1<<M); j++) {
                F[i][j] = 1000000000;
            }
        }
        for (int j = 1; j < (1<<M); j++) {
            int c = 0;
            for (int b = 0; b < M; b++) {
                if (((j>>b)&1)==1) {
                    c += cost[0][b];
                }
            }
            F[0][j] = c;
        }
        for (int j = 1; j < (1<<M); j++) {
            for (int k = 0; k < (1<<M); k++) {
                if ((j&k)==k) {
                    F[0][k] = min(F[0][k], F[0][j]);
                }
            }
        }
        for (int i = 1; i < N; i++) {
            for (int j = 1; j < (1<<M); j++) {
                for (int k = 1; k <= j; k++) {
                    if ((j&k)==k) {
                        int c = 0;
                        for (int b = 0; b < M; b++) {
                            if (((k>>b)&1)==1) {
                                c += cost[i][b];
                            }
                        }
                        F[i][j] = min(F[i][j], F[i-1][j-(j&k)] + c);
                    }
                }
            }
            for (int j = 1; j < (1<<M); j++) {
                for (int k = 0; k < (1<<M); k++) {
                    if ((j&k)==k) {
                        F[i][k] = min(F[i][k], F[i][j]);
                    }
                }
            }
        }
        return F[N-1][(1<<M)-1];
    }
};
