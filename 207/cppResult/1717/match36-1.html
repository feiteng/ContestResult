class Solution {
public:
    int dp[14][4096];
    void check(int &a, int b) {
        if(a == -1 || a > b) a = b;
    }
    int connectTwoGroups(vector<vector<int>>& cost) {
        memset(dp, -1, sizeof(dp));
        int n = cost.size();
        int m = cost[0].size();

        dp[0][0] = 0;
        for(int i = 0; i < n; ++i){
            for(int j = 0; j < (1<<m); ++j) {
                if(dp[i][j] != -1) {
                    for(int k = 0; k < m; ++k) {
                        check(dp[i+1][j|(1<<k)], dp[i][j] + cost[i][k]);
                    }
                }
                if(dp[i+1][j] != -1) {
                    for(int k = 0; k < m; ++k) {
                        check(dp[i+1][j|(1<<k)], dp[i+1][j] + cost[i][k]);
                    }
                }
            }
        }
        return dp[n][(1<<m)-1];
    }
};
