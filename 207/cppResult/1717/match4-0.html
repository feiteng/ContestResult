class Solution {
public:
    int n,m;
    int f[13][1<<12],now[1<<12];
    int connectTwoGroups(vector<vector<int>>& cost) {
        n=cost.size(),m=cost[0].size();
        for (int i=0;i<(1<<m);i++) 
            f[0][i]=(1e9)*(i>0);
        for (int i=0;i<n;i++){
            for (int j=0;j<(1<<m);j++)
                f[i+1][j]=1e9;
            for (int T=0;T<(1<<m);T++){
                now[T]=0;
                for (int j=0;j<m;j++)
                    if ((T>>j)&1) now[T]+=cost[i][j];
            }
            for (int S=0;S<(1<<m);S++)
                if (f[i][S]<1e9){
                    int mi=cost[i][0];
                    for (int j=0;j<m;j++)
                        mi=min(mi,cost[i][j]);
                    f[i+1][S]=min(f[i+1][S],f[i][S]+mi);
                    int inv=((1<<m)-1)^S;
                    for (int T=inv;T;T=(T-1)&inv){
                        f[i+1][S|T]=min(f[i+1][S|T],f[i][S]+now[T]);
                    }
                }
        }
        return f[n][(1<<m)-1];
    }
};
