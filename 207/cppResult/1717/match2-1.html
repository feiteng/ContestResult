class Solution {
public:
    int d[15][(1 << 12)+5];
    const int oo = 10000000;
    int size1, size2;

    int connectTwoGroups(vector<vector<int>>& cost) {
        size1 = cost.size();
        size2 = cost[0].size();
        
        for (int j = 0; j <= size1; ++j)
            for (int i = 0; i < (1 << size2); ++i) d[j][i] = oo;

        d[0][0] = 0;
    
        for (int i = 0; i < size1; ++i)
            for (int j = 0; j < size2; ++j) {
                int state = (1 << (j));
                int c = cost[i][j];
                for (int s = 0; s < (1 << size2); ++s) {
                    d[i+1][s|(1<<j)] = min(d[i+1][s|(1<<j)], d[i][s] + c);
                    d[i+1][s] = min(d[i+1][s], d[i][s] + c);
                    d[i+1][s|(1<<j)] = min(d[i+1][s|(1<<j)], d[i+1][s] + c);
                }
            }
        
        return d[size1][(1<<size2)-1];
    }
};
