class Solution {
public:
    int dp[14][1<<13];
    
    int connectTwoGroups(vector<vector<int>>& cost) {
        memset(dp, 0x3f, sizeof dp);
        dp[0][0] = 0;
        int len = cost[0].size();
        int lenr = cost.size();
        for(int i = 1; i <= lenr; ++i)
        {
            for(int s = 0; s < 1 << len; ++ s)
            {
                for(int j = 0; j < len; ++j)
                {
                    dp[i][s | (1 << j)] = min(dp[i][s | (1 << j)], dp[i - 1][s] + cost[i - 1][j]);
                    dp[i][s | (1 << j)] = min(dp[i][s | (1 << j)], dp[i][s] + cost[i - 1][j]);
                }
            }
        }
        
        return dp[lenr][(1 << len) - 1];
    }
};
