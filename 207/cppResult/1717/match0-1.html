class Solution {
 public:
  int connectTwoGroups(vector<vector<int>>& cost) {
    int n = cost.size(), m = cost[0].size();
    constexpr int INF = 1e9;
    vector<int> dp(1 << m, INF);
    dp[0] = 0;
    for (int i = 0; i < n; ++i) {
      vector<int> next_dp(1 << m, INF);
      for (int k = 0; k < 1 << m; ++k) {
        for (int j = 0; j < m; ++j) {
          next_dp[k | (1 << j)] =
              min(next_dp[k | (1 << j)], dp[k] + cost[i][j]);
          next_dp[k | (1 << j)] =
              min(next_dp[k | (1 << j)], next_dp[k] + cost[i][j]);
        }
      }
      dp = move(next_dp);
    }
    return dp.back();
  }
};
