class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int N = cost.size(), M = cost[0].size();
        int f[13][1<<12];
        memset(f, 60, sizeof(f));
        f[0][0] = 0;
        for (int i = 0; i < N; i ++)
        {
            for (int j = 0; j < (1<<M); j ++)
            {
                if (f[i][j] < 1e4 || f[i+1][j] < 1e4)
                {
                    for (int k = 0; k < M; k ++)
                        f[i+1][j|(1<<k)] = min(f[i+1][j|(1<<k)], min(f[i][j],f[i+1][j])+cost[i][k]);
                }
            }
        }
        //printf("%d %d %d %d\n", f[0][0], f[1][1], f[2][3]);
        return f[N][(1<<M)-1];
    }
};
