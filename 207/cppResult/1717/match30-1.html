class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int m = cost.size(), n = cost[0].size();
        int N = (1 << n);
        vector<int> min_cost(N, 1E6);
        min_cost[0] = 0;
        for (int i = 0; i < m; ++i) {
            vector<int> new_min_cost(N, 1E6);

            for (int j = 0; j < N; ++j) {
                for (int k = 0; k < n; ++k) {
                    int mask = (j | (1 << k));
                    new_min_cost[mask] = min(min(min_cost[j], new_min_cost[j]) + cost[i][k], new_min_cost[mask]);
                }
            }
            min_cost.swap(new_min_cost);
        }
        return min_cost[N - 1];
    }
};
