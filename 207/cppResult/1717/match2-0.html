const int maxn=10000+10;
int f[20][maxn],n,m,c[20][20];
class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        n=cost.size();
        m=cost[0].size();
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                c[i+1][j+1]=cost[i][j];
        memset(f,127/2,sizeof(f));
        f[0][0]=0;
        
        for(int i=0;i<n;i++)
            for(int j=0;j<(1<<m);j++)
                for(int k=1;k<=m;k++)
                {
                    int nj=j|(1<<(k-1));
                    f[i+1][nj]=min(f[i+1][nj],f[i][j]+c[i+1][k]);
                    f[i][nj]=min(f[i][nj],f[i][j]+c[i+1][k]);
                }
        return f[n][(1<<m)-1];
    }
};
