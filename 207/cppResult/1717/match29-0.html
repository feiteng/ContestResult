class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size(), m = cost[0].size(); //n must be greater
        vector<int> pre(1<<m, 1e9), cur(1<<m, 1e9); pre[0]=0;
        for (int i = 0; i < n; i++){
            for (int j = 0; j < (1<<m); j++) for (int k = 0; k < m; k++) {
                cur[j|(1<<k)]=min(cur[j|(1<<k)], pre[j]+cost[i][k]);
            }
            for (int j = (1<<m)-1; j >= 0; j--) for (int k = 0; k < m; k++){
                cur[j|(1<<k)]=min(cur[j|(1<<k)], cur[j]+cost[i][k]);
            }
            for (int j = 0; j < (1<<m); j++) pre[j]=cur[j], cur[j]=1e9;
        }
        return pre[(1<<m)-1];
    }
};
