class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& w) {
        int n = w.size(), m = w[0].size();
        int tot = 1 << m, lim = INT_MAX / 2;

        vector<vector<int>> f(n, vector<int>(tot+1, lim));
        
        f[0][0] = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < tot; ++j) {
                if (f[i][j] != lim) {
                    int cur = f[i][j];
                    

                    for (int k = 0; k < m; ++k) {
                        // if ((j & (1 << k)) == 0) {
                            int t = j | (1 << k);
                            f[i][t] = min(f[i][t], cur + w[i][k]);
                            if (i+1 < n && j != 0)
                                f[i+1][t] = min(f[i+1][t], cur + w[i+1][k]);
                        // }
                    }
                    // printf("%d %d:%d\n", i, j, f[i][j]);
                }
            }
        }
        return f[n-1][tot-1];
    }
};
