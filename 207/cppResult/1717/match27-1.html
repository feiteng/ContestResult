class Solution {
public:
    const int MAX_INF = 1e9 + 7;
    int connectTwoGroups(vector<vector<int>>& cost) {
        int m = cost.size(), n = cost[0].size();
        vector<vector<int>> dp(m + 1, vector<int>(1 << n, MAX_INF));

        dp[0][0] = 0;
        for (int i = 0; i < m; ++i)
        {
            for (int j = 0; j < n; ++j)
            {
                for (int k = 0; k < (1 << n); ++k)
                {
                    int x = k | (1 << j);
                    dp[i + 1][x] = min(dp[i + 1][x], dp[i][k] + cost[i][j]);
                    dp[i + 1][x] = min(dp[i + 1][x], dp[i + 1][k] + cost[i][j]);
                }
            }
        }
        return dp[m][(1 << n) - 1];
    }
};
