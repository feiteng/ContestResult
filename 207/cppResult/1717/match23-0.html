const int N = 13;
const int M = 1 << 13;

int f[N][M];

class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        
        memset(f,0x3f,sizeof f);
        
        int n = cost.size();
        int m = cost[0].size();
        
        f[0][0] = 0;
        
        for(int i=0;i<n;i++)
            for(int j=0;j<1<<m;j++)
            {
                for(int v=0;v<m;v++)
                {
                    int state = j | (1 << v);
                    f[i + 1][state] = min(f[i + 1][state],f[i][j]+cost[i][v]);
                }
                    
            }
        
        int res = 0x3f3f3f3f;
        
        for(int i=0;i<1<<m;i++)
        {
            int value = 0;
            for(int j=0;j<m;j++)
            {
                if(i >> j & 1) continue;
                int t = 0x3f3f3f3f;
                for(int k=0;k<n;k++)
                    t = min(t,cost[k][j]);
                value += t;
                
            }

            res = min(res,f[n][i] + value);
        }
        
        
        return res;
    }
};
