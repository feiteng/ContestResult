class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int n = cost.size();
        int m = cost[0].size();
        vector<int> mycost(1 << m, 1000000000);
        mycost[0] = 0;
        for(int i=0;i<n;i++){
            vector<int> tc(1<<m, 0);
            for(int j=1;j<1<<m;j++){
                for(int k=0;k<m;k++) {
                    if(j & (1<<k)) {tc[j] = tc[j ^ (1<<k)] + cost[i][k]; break;}
                }
            }
            int mi = 100;
            for(int j=0;j<m;j++){
                mi = min(mi, cost[i][j]);
            }
            vector<int> precost = mycost;
            for(int &j: mycost) j+= mi;
            for(int j=1;j<1<<m;j++){
                int k = j;
                while(k){
                    mycost[j] = min(mycost[j], precost[j ^ (k)] + tc[k]);
                    k = (k - 1) & j;
                }
            }
        }
        return mycost[(1<<m)-1];
    }
};
