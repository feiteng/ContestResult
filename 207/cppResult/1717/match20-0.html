class Solution {
public:
    int connectTwoGroups(vector<vector<int>>& cost) {
        int N=cost.size();
        int M=cost[0].size();
        int total=1<<M;
        vector<vector<int>> dp(N+1,vector<int>(total,-1));
        dp[0][0]=0;
        for(int i=0;i<N;i++){
            for(int ls=0;ls<total;ls++){
                if(dp[i][ls]<0){
                    continue;
                }
                for(int j=0;j<M;j++){
                    int ns=ls|(1<<j);
                    int nco=cost[i][j]+dp[i][ls];
                    if(dp[i+1][ns]<0){
                        dp[i+1][ns]=nco;
                    }else{
                        dp[i+1][ns]=min(dp[i+1][ns],nco);
                    }

                }
            }
            for(int ls=0;ls<total;ls++){
                if(dp[i+1][ls]<0){
                    continue;
                }   
                for(int j=0;j<M;j++){
                    int ns=ls|(1<<j);
                    int nco=cost[i][j]+dp[i+1][ls];
                    if(dp[i+1][ns]<0){
                        dp[i+1][ns]=nco;
                    }else{
                        dp[i+1][ns]=min(dp[i+1][ns],nco);
                    }

                }
            }
        }
        
        return dp[N][total-1];
    }
};
