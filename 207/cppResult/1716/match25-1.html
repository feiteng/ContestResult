class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<long long int>>dpmax(n+1,vector<long long int>(m+1,0));
        vector<vector<long long int>>dpmin(n+1,vector<long long int>(m+1,0));

        dpmax[0][0]=grid[0][0];
        dpmin[0][0]=grid[0][0];
        for(int i=1;i<m;i++)
        {
            dpmax[0][i]=grid[0][i]*dpmax[0][i-1];
            dpmin[0][i]=dpmax[0][i];
        }
        for(int i=1;i<n;i++)
        {
            dpmax[i][0]=grid[i][0]*dpmax[i-1][0];
            dpmin[i][0]=dpmax[i][0];
        }
        for(int i=1;i<n;i++)
        {
            for(int j=1;j<m;j++)
            {
                if(grid[i][j]>0)
                {
                    dpmax[i][j]=max(dpmax[i-1][j],dpmax[i][j-1])*grid[i][j];
                    dpmin[i][j]=min(dpmin[i-1][j],dpmin[i][j-1])*grid[i][j];
                }
                else if(grid[i][j]<0)
                {
                    dpmin[i][j]=max(dpmax[i-1][j],dpmax[i][j-1])*grid[i][j];
                    dpmax[i][j]=min(dpmin[i-1][j],dpmin[i][j-1])*grid[i][j];
                }
            }
        }
        if(dpmax[n-1][m-1]<0)
            return -1;
        int ans=dpmax[n-1][m-1]%1000000007;
        return ans;
    }
};
