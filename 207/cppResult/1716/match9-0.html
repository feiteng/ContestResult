class Solution {
public:
    long mod = 1e9+7;
    int maxProductPath(vector<vector<int>>& grid) {
        int m, n;
        m = grid.size();
        n = grid[0].size();

        vector<vector<long>> pos(m, vector<long>(n, -1));
        vector<vector<long>> neg(m, vector<long>(n, -1));
        vector<vector<long>> posDir(m, vector<long>(n));
        vector<vector<long>> negDir(m, vector<long>(n));
        
        if (grid[0][0] > 0)
            pos[0][0] = grid[0][0];
        if (grid[0][0] < 0)
            neg[0][0] = -grid[0][0];
        if (grid[0][0] == 0){
            pos[0][0] = 0;
            neg[0][0] = 0;
        }
            
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++){
                if (grid[i][j] > 0){
                    //printf("point %d %d\n", i, j);
                    if (i > 0 ){
                       
                        pos[i][j] = max(pos[i][j], pos[i-1][j] * grid[i][j]);
                        neg[i][j] = max(neg[i][j], neg[i-1][j] * grid[i][j]);      
                        //printf("%d %d\n", )
                    }
                    if (j > 0){
                        pos[i][j] = max(pos[i][j], pos[i][j-1] * grid[i][j]);
                        neg[i][j] = max(neg[i][j], neg[i][j-1] * grid[i][j]);
                        
                    }
                }
                if (grid[i][j] < 0){
                    //printf("%d %d\n", i, j);
                    if (i > 0 ){
                        pos[i][j] = max(pos[i][j], -neg[i-1][j] * grid[i][j]);
                        neg[i][j] = max(neg[i][j], -pos[i-1][j] * grid[i][j]);
                    }
                    if (j > 0){
                        pos[i][j] = max(pos[i][j], -neg[i][j-1] * grid[i][j]);
                        neg[i][j] = max(neg[i][j], -pos[i][j-1] * grid[i][j]);
                    }
                }
                if (grid[i][j] == 0){
                    pos[i][j] = 0;
                    neg[i][j] = 0;
                }
                    
            }
        // for (int i = 0; i < m; i++){
        //     for (int j = 0; j < n; j++){
        //         printf("(%d, %d) ", pos[i][j], neg[i][j]);
        //     }
        //     cout <<endl;
        // }
        if (pos[m-1][n-1] >= 0)
            return pos[m-1][n-1] % mod;
        return -1;
    }
};
