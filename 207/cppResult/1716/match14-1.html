class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {

        int n = grid.size(), m = grid[0].size();
        vector<vector<long long>> f(n, vector<long long>(m, LLONG_MIN)), h(n, vector<long long>(m, LLONG_MAX));
        
        f[0][0] = h[0][0] = grid[0][0];
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if (!(i + j))
                    continue;

                if (grid[i][j] >= 0) {
                    if (i) {
                        f[i][j] = max(f[i][j], f[i - 1][j] * grid[i][j]);
                        h[i][j] = min(h[i][j], h[i - 1][j] * grid[i][j]);
                    }
                    if (j) {
                        f[i][j] = max(f[i][j], f[i][j - 1] * grid[i][j]);
                        h[i][j] = min(h[i][j], h[i][j - 1] * grid[i][j]);
                    }
                } else {
                    if (i) {
                        f[i][j] = max(f[i][j], h[i - 1][j] * grid[i][j]);
                        h[i][j] = min(h[i][j], f[i - 1][j] * grid[i][j]);
                    }
                    if (j) {
                        f[i][j] = max(f[i][j], h[i][j - 1] * grid[i][j]);
                        h[i][j] = min(h[i][j], f[i][j - 1] * grid[i][j]);
                    }
                }
            }
        }
        
        int res = f[n - 1][m - 1] % 1000000007;
        
        return res < 0 ? -1 : res;
    }
};
