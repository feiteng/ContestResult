class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<long long>> dp0(n+1,vector<long long>(m+1,0));
        vector<vector<long long>> dp1(n+1,vector<long long>(m+1,0));
        
        for (int i=0;i<n;i++)
        {
            for (int j=0;j<m;j++)
            {
                if (i==0 && j==0)
                {
                    dp0[i][j]=grid[i][j];
                    dp1[i][j]=grid[i][j];
                    continue;
                }
                long long Min=2147483647,Max=-2147483648;

                if (i!=0)
                {
                    Min=min(dp0[i-1][j]*grid[i][j],dp1[i-1][j]*grid[i][j]);
                    Max=max(dp0[i-1][j]*grid[i][j],dp1[i-1][j]*grid[i][j]);
                }
                if (j!=0)
                {
                    Min=min(Min,min(dp0[i][j-1]*grid[i][j],dp1[i][j-1]*grid[i][j]));
                    Max=max(Max,max(dp0[i][j-1]*grid[i][j],dp1[i][j-1]*grid[i][j]));
                }
                //cout<<i<<' '<<j<<' '<<Min<<' '<<Max<<endl;
                dp0[i][j]=Min;
                dp1[i][j]=Max;
            }
        }
        int BP=1e9+7;
        if (dp1[n-1][m-1]>=0) return (dp1[n-1][m-1]%BP);
        else return -1;
    }
};
