class Solution {
 public:
  int maxProductPath(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<int64_t>> maxs(
        m, vector<int64_t>(n, numeric_limits<int64_t>::min()));
    vector<vector<int64_t>> mins(
        m, vector<int64_t>(n, numeric_limits<int64_t>::max()));
    maxs[0][0] = grid[0][0];
    mins[0][0] = grid[0][0];
    for (int i = 0; i < m; ++i) {
      for (int j = 0; j < n; ++j) {
        if (i == 0 && j == 0) continue;
        int64_t p = grid[i][j];
        if (i - 1 >= 0) {
          maxs[i][j] =
              max(maxs[i][j], max(maxs[i - 1][j] * p, mins[i - 1][j] * p));
          mins[i][j] =
              min(mins[i][j], min(maxs[i - 1][j] * p, mins[i - 1][j] * p));
        }
        if (j - 1 >= 0) {
          maxs[i][j] =
              max(maxs[i][j], max(maxs[i][j - 1] * p, mins[i][j - 1] * p));
          mins[i][j] =
              min(mins[i][j], min(maxs[i][j - 1] * p, mins[i][j - 1] * p));
        }
      }
    }
    if (maxs[m - 1][n - 1] < 0) return -1;
    return maxs[m - 1][n - 1] % 1000000007;
  }
};
