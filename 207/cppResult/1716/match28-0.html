class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<long long>> maxRes(m, vector<long long>(n, 0));
        vector<vector<long long>> minRes(m, vector<long long>(n, 0));
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                long long preMax = INT64_MIN;
                long long preMin = INT64_MAX;
                if(i == 0 && j == 0) {
                    maxRes[i][j] = grid[i][j];
                    minRes[i][j] = grid[i][j];
                    continue;
                }
                if(i > 0) {
                    preMax = max(preMax, maxRes[i-1][j]);
                    preMin = min(preMin, minRes[i-1][j]);
                }
                if(j > 0) {
                    preMax = max(preMax, maxRes[i][j-1]);
                    preMin = min(preMin, minRes[i][j-1]);
                }
                
                maxRes[i][j] = max(preMax * grid[i][j], preMin * grid[i][j]);
                minRes[i][j] = min(preMax * grid[i][j], preMin * grid[i][j]);
            }
        }
        
        return maxRes[m-1][n-1] < 0? -1 : maxRes[m-1][n-1] % (long long)(1e9 + 7);
    }
};
