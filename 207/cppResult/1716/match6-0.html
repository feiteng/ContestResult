class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int i,j,m=grid.size(),n=grid[0].size();
        
        vector<vector<long>> min_grid(m,vector<long>(n,0));
        vector<vector<long>> max_grid(m,vector<long>(n,0));
        
        
        
        max_grid[0][0] = min_grid[0][0] = grid[0][0];
        for(i=1;i<m;i++) {
            max_grid[i][0] = min_grid[i][0] = min_grid[i-1][0]*grid[i][0];
        }
        
        for(j=1;j<n;j++) {
            max_grid[0][j] = min_grid[0][j] = min_grid[0][j-1]*grid[0][j];
        }
        //puts("hello");
        for(i=1;i<m;i++) {
            for(j=1;j<n;j++) {
                if (grid[i][j] ==0) {
                    min_grid[i][j] = max_grid[i][j] = 0;
                }
                else if (grid[i][j]>0) {
                    min_grid[i][j] = grid[i][j] * min(min_grid[i-1][j], min_grid[i][j-1]);
                    max_grid[i][j] = grid[i][j] * max(max_grid[i-1][j], max_grid[i][j-1]);
                }
                else {
                    max_grid[i][j] = grid[i][j] * min(min_grid[i-1][j], min_grid[i][j-1]);
                    min_grid[i][j] = grid[i][j] * max(max_grid[i-1][j], max_grid[i][j-1]);
                }
                
            }
        }
        
        //printf("%d\n",max_grid[m-1][n-1]);
        
        if(max_grid[m-1][n-1]<0) {return -1;}
        return max_grid[m-1][n-1]%1000000007;
    }
};
