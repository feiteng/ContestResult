typedef long long ll;
class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        ll M=1000000007;
        int n=grid.size();
        int m=grid[0].size();
        
        vector<vector<ll>> dpmax(n,vector<ll>(m));
        vector<vector<ll>> dpmin(n,vector<ll>(m));
        
        dpmax[n-1][m-1]=grid[n-1][m-1];
        dpmin[n-1][m-1]=grid[n-1][m-1];
        
        for(int j=m-2;j>=0;j--){
            dpmax[n-1][j]=((grid[n-1][j])*dpmax[n-1][j+1]);
            dpmin[n-1][j]=((grid[n-1][j])*dpmin[n-1][j+1]);
        }
        
        for(int i=n-2;i>=0;i--){
            dpmax[i][m-1]=((grid[i][m-1])*dpmax[i+1][m-1]);
            dpmin[i][m-1]=((grid[i][m-1])*dpmin[i+1][m-1]);
        }
        
        for(int i=n-2;i>=0;i--){
            for(int j=m-2;j>=0;j--){
                if(grid[i][j]==0){
                    dpmax[i][j]=0;
                    dpmin[i][j]=0;
                }
                else if(grid[i][j]<0){
                    dpmax[i][j]=(grid[i][j]*min(dpmin[i][j+1],dpmin[i+1][j]));
                    dpmin[i][j]=(grid[i][j]*max(dpmax[i][j+1],dpmax[i+1][j]));
                }
                else{
                    dpmin[i][j]=(grid[i][j]*min(dpmin[i][j+1],dpmin[i+1][j]));
                    dpmax[i][j]=(grid[i][j]*max(dpmax[i][j+1],dpmax[i+1][j]));
                }
            }
        }
        
        if(dpmax[0][0]<0){
            return -1;
        }
        
        return dpmax[0][0]%M;
    }
};
