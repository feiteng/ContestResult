class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<long long>> pres(m, vector<long long>(n, 1));
        vector<vector<long long>> nres(m, vector<long long>(n, 1));
        
        pres[0][0] = grid[0][0];
        nres[0][0] = grid[0][0];
        
        for(int j = 1; j < n; j++){
            pres[0][j] = grid[0][j] * pres[0][j-1];
            nres[0][j]  = grid[0][j] * nres[0][j-1];
        }
        
        for(int i = 1; i < m; i++){
            pres[i][0] = grid[i][0] * pres[i-1][0];
            nres[i][0] = grid[i][0] * nres[i-1][0];
        }
        
        for(int i = 1; i < m; i++){
            for(int j = 1; j < n; j++){
                
                if(grid[i][j] >= 0){
                    pres[i][j] = grid[i][j] * max(pres[i][j-1], pres[i-1][j]);
                    nres[i][j] = grid[i][j] * min(nres[i][j-1], nres[i-1][j]);
                } else {
                    pres[i][j] = grid[i][j] * min(nres[i][j-1], nres[i-1][j]);
                    nres[i][j] = grid[i][j] * max(pres[i][j-1], pres[i-1][j]);
                }
                
                // cout<<i<<" "<<j<<" "<<pres[i][j]<<" "<<nres[i][j]<<endl;
                
            }
        }
        
        long long tmp = 1000000007;
        if(pres[m-1][n-1] >= 0){
            return pres[m-1][n-1] % tmp;
        }
        else return -1;
        
    }
};
