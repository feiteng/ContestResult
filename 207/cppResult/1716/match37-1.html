class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        long res[15][15][2];
        res[0][0][0]=res[0][0][1]=grid[0][0];

        int row=grid.size(),col=grid[0].size();
        for(int j=1;j<col;j++){
            if(grid[0][j]<0){
                res[0][j][0]=grid[0][j]*res[0][j-1][1];
                res[0][j][1]=grid[0][j]*res[0][j-1][0];
            }
            else{
                res[0][j][0]=grid[0][j]*res[0][j-1][0];
                res[0][j][1]=grid[0][j]*res[0][j-1][1];
            }
        }
        for(int i=1;i<row;i++){
            if(grid[i][0]<0){
                res[i][0][0]=grid[i][0]*res[i-1][0][1];
                res[i][0][1]=grid[i][0]*res[i-1][0][0];
            }
            else{
                res[i][0][0]=grid[i][0]*res[i-1][0][0];
                res[i][0][1]=grid[i][0]*res[i-1][0][1];
            }
        }
        for(int i=1;i<row;i++){
            for(int j=1;j<col;j++){
                if(grid[i][j]<0){
                    res[i][j][0]=grid[i][j]*min(res[i-1][j][1],res[i][j-1][1]);
                    res[i][j][1]=grid[i][j]*max(res[i-1][j][0],res[i][j-1][0]);
                }
                else{
                    res[i][j][0]=grid[i][j]*max(res[i-1][j][0],res[i][j-1][0]);
                    res[i][j][1]=grid[i][j]*min(res[i-1][j][1],res[i][j-1][1]);
                }
            }
        }
        if(res[row-1][col-1][0]<0)
            return -1;
        return res[row-1][col-1][0]%1000000007;
    }
};
