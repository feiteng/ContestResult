class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {

        int row=grid.size(),col=grid[0].size();
        vector<vector<long long>> mmax(row,vector<long long>(col));
        vector<vector<long long>> mmin(row,vector<long long>(col));
        mmax[0][0]=grid[0][0];
        mmin[0][0]=grid[0][0];
        for(int i=1;i<row;++i)
        {
            mmax[i][0]=mmax[i-1][0]*grid[i][0];
            mmin[i][0]=mmax[i][0];
        }
        for(int j=1;j<col;++j)
        {
            mmax[0][j]=mmax[0][j-1]*grid[0][j];
            mmin[0][j]=mmax[0][j];
        }
        for(int i=1;i<row;++i)
        {
            for(int j=1;j<col;++j)
            {
                int t=grid[i][j];
                if(t>0)
                {
                    mmax[i][j]=max(mmax[i-1][j],mmax[i][j-1])*t;
                    mmin[i][j]=min(mmin[i-1][j],mmin[i][j-1])*t;
                }

                else if(t<0)
                {
                    mmin[i][j]=max(mmax[i-1][j],mmax[i][j-1])*t;
                    mmax[i][j]=min(mmin[i-1][j],mmin[i][j-1])*t;
                }
                else
                {
                    mmin[i][j]=0;
                    mmax[i][j]=0;
                }
            }
        }
        long long a=mmax[row-1][col-1];
        if(a<0) return -1;
        return a%(long long)(1e9+7);
    }
};
