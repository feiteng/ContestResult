class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        
        int m = grid.size();
        int n = grid[0].size();
        
        long long int dp[m][n];
        long long int dp2[m][n];
        

        dp[0][0] = grid[0][0];
        dp2[0][0] = grid[0][0];
        for(int i=1;i<m;i++)
        {
            dp[i][0] = dp[i-1][0]*grid[i][0];
            dp2[i][0] = dp2[i-1][0]*grid[i][0];
        }
        
        for(int j=1;j<n;j++)
        {
            dp[0][j] = dp[0][j-1]*grid[0][j];
            dp2[0][j] = dp2[0][j-1]*grid[0][j];
        }
        
        for(int i=1;i<m;i++)
        {
            for(int j=1;j<n;j++)
            {
                dp[i][j] = max(max(dp[i][j-1]*grid[i][j],dp[i-1][j]*grid[i][j]),max(dp2[i-1][j]*grid[i][j],dp2[i][j-1]*grid[i][j]));
                dp2[i][j]= min(min(dp[i][j-1]*grid[i][j],dp[i-1][j]*grid[i][j]),min(dp2[i-1][j]*grid[i][j],dp2[i][j-1]*grid[i][j]));
            }
        }
        
        if(dp[m-1][n-1]>=0) return dp[m-1][n-1]%1000000007;
        return -1;
        
    }
};
