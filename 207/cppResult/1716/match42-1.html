using LL=long long;

class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {

      int m = grid.size(), n = grid[0].size();
      vector<vector<LL>> minl(m, vector<LL>(n, LLONG_MAX)), maxl(m, vector<LL>(n, LLONG_MIN));
      for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
          if (i == 0 && j == 0) {
            minl[0][0] = grid[0][0]; maxl[0][0] = grid[0][0];
            continue;
          }
          if (i > 0) {
            if (grid[i][j] >= 0) {
              minl[i][j] = min(minl[i][j], minl[i-1][j] * grid[i][j]);
              maxl[i][j] = max(maxl[i][j], maxl[i-1][j] * grid[i][j]);
            }
            else {
              minl[i][j] = min(minl[i][j], maxl[i-1][j] * grid[i][j]);
              maxl[i][j] = max(maxl[i][j], minl[i-1][j] * grid[i][j]);
            }
          }
          if (j > 0) {
            if (grid[i][j] >= 0) {
              minl[i][j] = min(minl[i][j], minl[i][j-1] * grid[i][j]);
              maxl[i][j] = max(maxl[i][j], maxl[i][j-1] * grid[i][j]);
            }
            else {
              minl[i][j] = min(minl[i][j], maxl[i][j-1] * grid[i][j]);
              maxl[i][j] = max(maxl[i][j], minl[i][j-1] * grid[i][j]);
            }
          }
        }
      }
      LL mod = 1e9+7;
      if (maxl[m-1][n-1] < 0) return -1;
      return maxl[m-1][n-1] % mod;
      
    }
};
