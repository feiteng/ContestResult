class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        long long dp[n + 1][m + 1][2], mod = 1e9 + 7;
        for(int i = 0; i <= n; ++i) {
            dp[i][0][0] = 1;
            dp[i][0][1] = -1;
        }
        for(int i = 0; i <= m; ++i) {
            dp[0][i][0] = 1;
            dp[0][i][1] = -1;
        }
        int tmp;
        for(int i = 1; i <= n; ++i) {
            for(int j = 1; j <= m; ++j) {
                dp[i][j][0] = 1;
                dp[i][j][1] = -1;
                if( i == 1 and j == 1)dp[1][1][0] = dp[1][1][1] = grid[0][0];
                for(int k = 0; k < 2; ++k) {
                    if( k == 0)
                        tmp = 1;
                    else
                        tmp = -1;
                    if( dp[i-1][j][k] != tmp)
                        dp[i][j][0] = min(dp[i][j][0], dp[i-1][j][k] * grid[i-1][j-1]);
                    if( dp[i][j - 1][k] != tmp)
                        dp[i][j][0] = min(dp[i][j][0], dp[i][j - 1][k] * grid[i-1][j-1]);
                    
                    if( dp[i-1][j][k] != tmp)
                        dp[i][j][1] = max(dp[i][j][1], dp[i-1][j][k] * grid[i-1][j-1]);
                    if( dp[i][j - 1][k] != tmp)
                        dp[i][j][1] = max(dp[i][j][1], dp[i][j - 1][k] * grid[i-1][j-1]);
                }
                // cout << dp[i][j][1] << " ";
            }
            // cout << endl;
        }
        if( dp[n][m][1] < 0)
            return -1;
        return dp[n][m][1] % mod;
    }
};
