class Solution {
public:
    const int mod=1000000007;

    int maxProductPath(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<long long>>dp(n,vector<long long>(m,1));
        vector<vector<long long>>dp1(n,vector<long long>(m,1));
        dp[n-1][m-1]=grid[n-1][m-1];
        dp1[n-1][m-1]=grid[n-1][m-1];
        for(int i=n-2;i>=0;i--){
            dp[i][m-1]=grid[i][m-1]*dp[i+1][m-1];
            dp1[i][m-1]=grid[i][m-1]*dp[i+1][m-1];
        }
        for(int j=m-2;j>=0;j--){
            dp[n-1][j]=grid[n-1][j]*dp[n-1][j+1];
            dp1[n-1][j]=grid[n-1][j]*dp[n-1][j+1];
        }
        for(int i=n-2;i>=0;i--){

            for(int j=m-2;j>=0;j--){
                if(grid[i][j]<0){
                    dp[i][j]=max(grid[i][j]*dp1[i+1][j],grid[i][j]*dp1[i][j+1]);
                    dp1[i][j]=min(grid[i][j]*dp[i+1][j],grid[i][j]*dp[i][j+1]);
                }
                else{
                    dp[i][j]=max(grid[i][j]*dp[i+1][j],grid[i][j]*dp[i][j+1]);
                    dp1[i][j]=min(grid[i][j]*dp1[i+1][j],grid[i][j]*dp1[i][j+1]);
                }
            }
        }
        if(dp[0][0]<0){
            return -1;
        }
        dp[0][0]%=mod;
        return dp[0][0];
    }
};
