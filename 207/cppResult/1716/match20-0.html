using LL = long long;

class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<LL>> u(m, vector<LL>(n));
        vector<vector<LL>> v(m, vector<LL>(n));
        u[0][0] = v[0][0] = grid[0][0];
        for (int i = 1; i < n; ++i) {
            u[0][i] = v[0][i] = u[0][i - 1] * grid[0][i];
        }
        for (int i = 1; i < m; ++i) {
            u[i][0] = v[i][0] = u[i - 1][0] * grid[i][0];
        }
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                tie(u[i][j], v[i][j]) = minmax({u[i - 1][j] * grid[i][j], v[i - 1][j] * grid[i][j], u[i][j - 1] * grid[i][j], v[i][j - 1] * grid[i][j]});
            }
        }
        return v[m - 1][n - 1] < 0 ? -1 : v[m - 1][n - 1] % 1000000007;
    }
};
