class Solution {
	using ll = long long;
	const ll INF = 2 * 0x3f3f3f3f3f3f3f3fLL;
	const int mod = 1e9 + 7;
public:
	int maxProductPath(vector<vector<int>> &v) {

		int n = v.size(), m = v[0].size();
		vector<vector<ll>> dpMax(n, vector<ll>(m, -INF));
		vector<vector<ll>> dpMin(n, vector<ll>(m, INF));
		for (int i = n - 1; i >= 0; i--) {
			for (int j = m - 1; j >= 0; j--) {
				if (i == n - 1 && j == m - 1)

					dpMax[i][j] = dpMin[i][j] = v[i][j];
				if (j + 1 < m) {
					dpMax[i][j] = max(v[i][j] * dpMax[i][j + 1],
							v[i][j] * dpMin[i][j + 1]);
					dpMin[i][j] = min(v[i][j] * dpMax[i][j + 1],
							v[i][j] * dpMin[i][j + 1]);
				}
				if (i + 1 < n) {
					dpMax[i][j] = max(dpMax[i][j],
							max(v[i][j] * dpMax[i + 1][j],
									v[i][j] * dpMin[i + 1][j]));
					dpMin[i][j] = min(dpMin[i][j],
							min(v[i][j] * dpMax[i + 1][j],
									v[i][j] * dpMin[i + 1][j]));
				}
			}
		}
		if (dpMax[0][0] < 0)
			return -1;
		return dpMax[0][0] % mod;
	}
};
