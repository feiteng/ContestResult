using LL=long long;

class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int N=grid.size();
        int M=grid[0].size();
        vector<vector<vector<LL>>> dp(N+1,vector<vector<LL>>(M+1,vector<LL>(2,1)));
        
        for(int i=0;i<M;i++){
            dp[1][i+1][0]=dp[1][i][0]*grid[0][i];
            dp[1][i+1][1]=dp[1][i+1][0];
        }
        for(int i=0;i<N;i++){
            dp[i+1][1][0]=dp[i][1][0]*grid[i][0];
            dp[i+1][1][1]=dp[i+1][1][0];
        }
        for(int i=1;i<N;i++){
            for(int j=1;j<M;j++){
                int x=grid[i][j];
                if(x==0){
                    dp[i+1][j+1][0]=0;
                    dp[i+1][j+1][1]=0;
                }else if(x>0){
                    
                    dp[i+1][j+1][0]=max(x*dp[i+1][j][0],x*dp[i][j+1][0]);
                    dp[i+1][j+1][1]=min(x*dp[i+1][j][1],x*dp[i][j+1][1]);
   
                }else{
                    dp[i+1][j+1][0]=max(x*dp[i+1][j][1],x*dp[i][j+1][1]);
                    dp[i+1][j+1][1]=min(x*dp[i+1][j][0],x*dp[i][j+1][0]);                    
                }
                //cout<<i<<" "<<j<<" "<<dp[i+1][j+1][0]<<" "<<dp[i+1][j+1][1]<<endl;
            }
        }
        
        if(dp[N][M][0]<0){
            return -1;
        }
       
        LL BASE=pow(10,9)+7;
        return dp[N][M][0]%(BASE);
    }
};
