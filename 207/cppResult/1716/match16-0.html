class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        // dmax[i][j] := max 
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<long long>> dpMax(m, vector<long long>(n,INT_MIN));
        vector<vector<long long>> dpMin(m, vector<long long>(n,INT_MAX));
        
        dpMax[0][0] = grid[0][0];
        dpMin[0][0] = grid[0][0];
        
        for (int i = 1; i < m; i++) {
            dpMax[i][0] = dpMax[i - 1][0] * grid[i][0];
            dpMin[i][0] = dpMax[i][0];
        }
        
        for (int i = 1; i < n; i++) {
            dpMax[0][i] = dpMax[0][i - 1] * grid[0][i];
            dpMin[0][i] = dpMax[0][i];
        }
        
        for (int i = 1; i < m; i++)
            for (int j = 1; j < n; j++) {
                if (grid[i][j] == 0){
                    dpMax[i][j] = 0;
                    dpMin[i][j] = 0;
                } else {
                    dpMax[i][j] = max({dpMax[i-1][j] * grid[i][j], dpMin[i-1][j] * grid[i][j], dpMax[i][j-1] * grid[i][j], dpMin[i][j-1] * grid[i][j]});
                    dpMin[i][j] = min({dpMax[i-1][j] * grid[i][j], dpMin[i-1][j] * grid[i][j], dpMax[i][j-1] * grid[i][j], dpMin[i][j-1] * grid[i][j]});
                }
            }
        
        int kMod = 1e9 + 7;
        return dpMax[m-1][n-1] < 0 ? -1 : dpMax[m-1][n-1] % kMod;
    }
};
