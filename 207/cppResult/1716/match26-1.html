class Solution {
public:
int maxProductPath(vector<vector<int>>& grid) {
    int m=grid.size();
    int n=grid[0].size();
    long long cont_val=1000000007;
    vector<vector<long long>>h_max(m,vector<long long>(n,INT_MIN));
    vector<vector<long long>>h_min(m,vector<long long>(n,INT_MAX));

    h_max[0][0]=grid[0][0];
    h_min[0][0]=grid[0][0];
    for(int i=1;i<m;++i){
        h_max[i][0]=h_max[i-1][0]*grid[i][0];
        h_min[i][0]=h_max[i][0];
    }
    for(int i=1;i<n;++i){
        h_max[0][i]=h_max[0][i-1]*grid[0][i];
        h_min[0][i]=h_max[0][i];
    }
    for(int i=1;i<m;++i){
        for(int j=1;j<n;++j){
            h_max[i][j]=max(h_max[i][j], h_max[i-1][j]*grid[i][j]);
            h_max[i][j]=max(h_max[i][j], h_max[i][j-1]*grid[i][j]);
            h_max[i][j]=max(h_max[i][j], h_min[i-1][j]*grid[i][j]);
            h_max[i][j]=max(h_max[i][j], h_min[i][j-1]*grid[i][j]);

            h_min[i][j]=min(h_min[i][j], h_min[i-1][j]*grid[i][j]);
            h_min[i][j]=min(h_min[i][j], h_min[i][j-1]*grid[i][j]);
            h_min[i][j]=min(h_min[i][j], h_max[i-1][j]*grid[i][j]);
            h_min[i][j]=min(h_min[i][j], h_max[i][j-1]*grid[i][j]);
            
        }
    }
    return h_max[m-1][n-1]>=0?h_max[m-1][n-1]%cont_val:-1;

}
};
