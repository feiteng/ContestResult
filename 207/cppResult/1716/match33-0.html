class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        long long dp[2][n][m];
        dp[0][0][0] = dp[1][0][0] = grid[0][0];
        for (int i = 1; i < n; ++i) {
            dp[0][i][0] = dp[1][i][0] = dp[0][i - 1][0] * grid[i][0];
        }
        for (int i = 1; i < m; ++i) {
            dp[0][0][i] = dp[1][0][i] = dp[0][0][i - 1] * grid[0][i];
        }
        for (int i =  1; i < n; ++i) {
            for (int j = 1; j < m; ++j) {
                dp[0][i][j] = LONG_LONG_MIN;
                dp[1][i][j] = LONG_LONG_MAX;
                for (int k = 0; k < 2; ++k) {
                    long long t1 = dp[k][i - 1][j] * grid[i][j];
                    long long t2 = dp[k][i][j - 1] * grid[i][j];
                    dp[0][i][j] = max(dp[0][i][j], t1);
                    dp[0][i][j] = max(dp[0][i][j], t2);
                    dp[1][i][j] = min(dp[1][i][j], t1);
                    dp[1][i][j] = min(dp[1][i][j], t2);
                }
            }
        }
        // for (int i = 0;i < 2; ++i) {
        //     for (int j = 0; j < n; ++j) {
        //         for (int k = 0; k<m; ++k) {
        //             printf("%lld%c", dp[i][j][k], k == m - 1? '\n' : ' ');
        //         }
        //     }
        // }
        if (dp[0][n - 1][m - 1] >= 0) {
            return (int)(dp[0][n - 1][m - 1] % 1000000007);
        } else {
            return -1;
        }
    }
};
