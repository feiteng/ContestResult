class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<vector<long>>> dp(n, vector<vector<long>>(m, vector<long>(2, 1)));
        int mod = 1e9 + 7;

        dp[0][0] = {grid[0][0], grid[0][0]};
        for (int i = 1; i < n; ++i) {
            long tmp = dp[i - 1][0][0] * grid[i][0];
            dp[i][0] = {tmp, tmp};
        }
        for (int i = 1; i < m; ++i) {
            long tmp = dp[0][i - 1][0] * grid[0][i];
            dp[0][i] = {tmp, tmp};
        }
        for (int i = 1; i < n; ++i) {
            for (int j = 1; j < m; ++j) {
                long tmp1 = dp[i - 1][j][0] * grid[i][j];

                long tmp2 = dp[i - 1][j][1] * grid[i][j];
                long tmp3 = dp[i][j - 1][0] * grid[i][j];
                long tmp4 = dp[i][j - 1][1] * grid[i][j];
                long max_val = max(max(tmp1, tmp2), max(tmp3, tmp4));
                long min_val = min(min(tmp1, tmp2), min(tmp3, tmp4));
                dp[i][j] = {min_val, max_val};
            }
        }
        if (dp[n - 1][m - 1][1] >= 0) {
            return dp[n - 1][m - 1][1] % mod;
        }
        return -1;
    }
};
