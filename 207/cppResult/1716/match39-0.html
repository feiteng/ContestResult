class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<long long>> dp1(n,vector<long long>(m,INT_MIN)),dp2(n,vector<long long>(m,INT_MAX));
        dp1[0][0]=grid[0][0];
        dp2[0][0]=grid[0][0];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(i==0&&j==0)continue;
                
                long long a=grid[i][j];
               
                if(grid[i][j]<0){
                    if(i){
                        dp1[i][j]=dp2[i-1][j]*a;
                        dp2[i][j]=dp1[i-1][j]*a;
                    }
                    if(j){
                        dp1[i][j]=max(dp1[i][j],dp2[i][j-1]*a);
                        dp2[i][j]=min(dp2[i][j],dp1[i][j-1]*a);
                    }
                    
                }else{
                    if(i){
                        dp1[i][j]=dp1[i-1][j]*a;
                        dp2[i][j]=dp2[i-1][j]*a;
                    }
                    if(j){
                        dp1[i][j]=max(dp1[i][j],dp1[i][j-1]*a);
                        dp2[i][j]=min(dp2[i][j],dp2[i][j-1]*a);
                    }
                }
                
            }
        }
        long long mod=1e9+7;
        return (dp1[n-1][m-1]<0)?-1:dp1[n-1][m-1]%mod;
        
    }
};
