class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        #define ll long long
        ll d[20][20][2];
        int m=grid.size(), n=grid[0].size();

        d[0][0][0]=d[0][0][1]=grid[0][0];
        for(int i=1;i<n;i++){
            d[0][i][1]=d[0][i][0]=d[0][i-1][0]*grid[0][i];
        }
        for(int i=1;i<m;i++){
            d[i][0][1]=d[i][0][0]=d[i-1][0][0]*grid[i][0];
        }
        
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                if(grid[i][j]<=0){
                    d[i][j][1]=min(d[i][j-1][0], d[i-1][j][0])*grid[i][j];
                    d[i][j][0]=max(d[i][j-1][1], d[i-1][j][1])*grid[i][j];
                }
                else{
                    d[i][j][1]=max(d[i][j-1][1], d[i-1][j][1])*grid[i][j];
                    d[i][j][0]=min(d[i][j-1][0], d[i-1][j][0])*grid[i][j];
                }
            }
        }
        if(d[m-1][n-1][1]<0) return -1;
        return d[m-1][n-1][1]%1000000007;
    }
    
};
