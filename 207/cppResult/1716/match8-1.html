class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m=grid.size(),n=grid[0].size();
        vector<vector<long long>> mx(m,vector<long long>(n,INT_MIN));
        vector<vector<long long>> mn(m,vector<long long>(n,INT_MAX));
        mx[0][0]=grid[0][0];
        mn[0][0]=grid[0][0];
        int mod=pow(10,9)+7;
        for (int i=0;i<m;i++)
        {

            for (int j=0;j<n;j++)
            {
                if (grid[i][j]>0)
                {
                    if (i>0)
                    {
                        mx[i][j]=mx[i-1][j]*grid[i][j];
                        mn[i][j]=mn[i-1][j]*grid[i][j];
                    }
                    if (j>0)
                    {
                        mx[i][j]=max(mx[i][j],mx[i][j-1]*grid[i][j]);
                        mn[i][j]=min(mn[i][j],mn[i][j-1]*grid[i][j]);
                    }
                }
                else if (grid[i][j]==0)
                {
                    mx[i][j]=0;
                    mn[i][j]=0;
                }
                else

                {
                    if (i>0)
                    {
                        mx[i][j]=mn[i-1][j]*grid[i][j];
                        mn[i][j]=mx[i-1][j]*grid[i][j];
                        //cout << mx[i][j] << " " << mn[i][j] << endl;
                    }
                    if (j>0)
                    {
                        mx[i][j]=max(mx[i][j],mn[i][j-1]*grid[i][j]);
                        mn[i][j]=min(mn[i][j],mx[i][j-1]*grid[i][j]);
                        //cout << mx[i][j] << " " << mn[i][j] << endl;
                    }
                }
                //cout << mx[i][j] << " " << mn[i][j] << endl;
            }
        }
        if (mn[m-1][n-1]<0 && mx[m-1][n-1]<0)
            return -1;
        return max(mn[m-1][n-1],mx[m-1][n-1])%mod;
    }
};
