class Solution {
public:
    int maxProductPath(vector<vector<int>>& arr) {
        #define ll long long
        int n = arr.size();
        int m = arr[0].size();
        vector<vector<ll>> dpmin(n, vector<ll>(m, -1));
        vector<vector<ll>> dpmax(n, vector<ll>(m, -1));
        dpmin[0][0] = dpmax[0][0] = arr[0][0];
        for(int i = 1; i < n; i++){
            dpmin[i][0] = dpmax[i][0] = dpmax[i-1][0]*arr[i][0];
        }
        
        for(int j = 1; j < m; j++){
            dpmin[0][j] = dpmax[0][j] = dpmax[0][j-1]*arr[0][j];
        }
        
        for(int i = 1; i < n; i++){
            for(int j = 1; j < m; j++){
                ll cur1 = dpmin[i][j-1]*arr[i][j];
                ll cur2 = dpmax[i][j-1]*arr[i][j];
                ll mi = min(cur1, cur2);
                ll ma = max(cur1, cur2);
                cur1 = dpmin[i-1][j]*arr[i][j];
                cur2 = dpmax[i-1][j]*arr[i][j];
                ll mi1 = min(cur1, cur2);
                ll ma1 = max(cur1, cur2);
                
                dpmin[i][j] = min(mi, mi1);
                dpmax[i][j] = max(ma, ma1);
            }
        }
        if(dpmax[n-1][m-1] < 0) return -1;
        else return dpmax[n-1][m-1]%1000000007;
    }
};
