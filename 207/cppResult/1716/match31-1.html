long long dp[20][20][2];
class Solution {
public:
    int maxProductPath(vector<vector<int>>& g) {
        int m = g.size(), n = g[0].size();

        dp[0][0][0] = g[0][0];
        dp[0][0][1] = g[0][0];
        for (int i = 1; i < m; ++i) {
            dp[i][0][0] = min(dp[i - 1][0][0] * g[i][0], dp[i - 1][0][1] * g[i][0]); 
            dp[i][0][1] = max(dp[i - 1][0][0] * g[i][0], dp[i - 1][0][1] * g[i][0]); 
        }
        for (int i = 1; i < n; ++i) {
            dp[0][i][0] = min(dp[0][i - 1][0] * g[0][i], dp[0][i - 1][1] * g[0][i]);
            dp[0][i][1] = max(dp[0][i - 1][0] * g[0][i], dp[0][i - 1][1] * g[0][i]);
        }
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {

                int v = g[i][j];
                dp[i][j][0] = min({dp[i][j - 1][0] * v, dp[i][j - 1][1] * v, dp[i - 1][j][0] * v, dp[i - 1][j][1] * v});
                dp[i][j][1] = max({dp[i][j - 1][0] * v, dp[i][j - 1][1] * v, dp[i - 1][j][0] * v, dp[i - 1][j][1] * v});
            }
        }
        return dp[m - 1][n - 1][1] < 0 ? -1 : dp[m - 1][n - 1][1] % 1000000007;
    }
};
