#define ll long long
class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {

        ll n=grid.size(),m=grid[0].size();
        vector<vector<ll> >vpos(n,vector<ll>(m,0)),vneg(n,vector<ll>(m,0));
        vpos[0][0]=grid[0][0];
        vneg[0][0]=grid[0][0];
        for(int j=1;j<m;j++){
            vpos[0][j]=vpos[0][j-1]*grid[0][j];
            vneg[0][j]=vpos[0][j];
        }for(int i=1;i<n;i++){
            vpos[i][0]=vpos[i-1][0]*grid[i][0];
            vneg[i][0]=vpos[i][0];
        }
        
        for(int i=1;i<n;i++){
            for(int j=1;j<m;j++){
                if(grid[i][j]>=0){
                    vpos[i][j]=max(vpos[i-1][j]*grid[i][j],vpos[i][j-1]*grid[i][j]);
                    vneg[i][j]=min(vneg[i-1][j]*grid[i][j],vneg[i][j-1]*grid[i][j]);
                }else{
                    vneg[i][j]=min(vpos[i-1][j]*grid[i][j],vpos[i][j-1]*grid[i][j]);
                    vpos[i][j]=max(vneg[i-1][j]*grid[i][j],vneg[i][j-1]*grid[i][j]);
                }
            }
        }
        
        ll k=1e9+7;
        if(vpos[n-1][m-1]>=0){
            return vpos[n-1][m-1]%k;
        }
        return -1;
    }
};
