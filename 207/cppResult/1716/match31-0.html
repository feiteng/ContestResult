class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid)
    {
        int n = grid.size();
        int m = grid[0].size();

        vector<vector<long long>> mn(n, vector<long long>(m, 1));
        vector<vector<long long>> mx(n, vector<long long>(m, 1));

        mn[0][0] = grid[0][0];
        mx[0][0] = grid[0][0];

        for (int i = 1; i < n; ++i)
        {
            mn[i][0] = grid[i][0] * mn[i - 1][0];
            mx[i][0] = grid[i][0] * mx[i - 1][0];
        }

        for (int i = 1; i < m; ++i)
        {
            mn[0][i] = grid[0][i] * mn[0][i - 1];
            mx[0][i] = grid[0][i] * mx[0][i - 1];
        }

        for (int i = 1; i < n; ++i)
        {
            for (int j = 1; j < m; ++j)
            {
                long long v1 = mn[i - 1][j] * grid[i][j];
                long long v2 = mn[i][j - 1] * grid[i][j];
                long long v3 = mx[i - 1][j] * grid[i][j];
                long long v4 = mx[i][j - 1] * grid[i][j];

                mn[i][j] = min({ v1, v2, v3, v4 });
                mx[i][j] = max({ v1, v2, v3, v4 });
            }
        }

        return mx.back().back() >= 0 ? mx.back().back() % (1000 * 1000 * 1000 + 7) : -1;
    }
};
