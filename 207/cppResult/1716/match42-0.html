const int dx[] = {0,1};
const int dy[] = {1,0};
class Solution {
public:
    typedef long long ll;
    ll ans = LLONG_MIN;
    int mod = 1e9+7;
    int m , n;
    int maxProductPath(vector<vector<int>>& g) {
        m = g.size();
        n = g[0].size();
        ll dp[20][20][2];
        for(int i=0;i<20;i++){
            for(int j=0;j<20;j++){
                dp[i][j][0] = -1;
                dp[i][j][1] = 1;
            }
        }
        dp[0][0][0] =dp[0][0][1] = g[0][0];
        
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(g[i][j]==0){
                    dp[i][j][0] = dp[i][j][1] = 0;
                    continue;
                }
                
                if(i){
                   if(g[i][j]>0){
                       dp[i][j][0] = max(dp[i][j][0],dp[i-1][j][0]*g[i][j]);
                       dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][1]*g[i][j]);
                   }else{
                       dp[i][j][0] = max(dp[i][j][0],dp[i-1][j][1]*g[i][j]);
                       dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][0]*g[i][j]);                       
                   }
                }
                if(j){
                   if(g[i][j]>0){
                       dp[i][j][0] = max(dp[i][j][0],dp[i][j-1][0]*g[i][j]);
                       dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][1]*g[i][j]);
                   }else{
                       dp[i][j][0] = max(dp[i][j][0],dp[i][j-1][1]*g[i][j]);
                       dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][0]*g[i][j]);                       
                   }                    
                }
            }
        }
        if(dp[m-1][n-1][0]<0) return -1;
        return dp[m-1][n-1][0]%mod;
    }

};
