#define ll long long

class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        ll mod=1e9+7;
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<ll>> mx(n, vector<ll>(m, 0));
        vector<vector<ll>> mn(n, vector<ll>(m, 0));
        mx[0][0]=grid[0][0];

        mn[0][0]=grid[0][0];
        for (int i=0; i<n; ++i) {
            for (int j=0; j<m; ++j) {
                if (i==0&&j==0) continue;

                if (i==0) {
                    mx[i][j]=mx[i][j-1]*grid[i][j];
                    mn[i][j]=mn[i][j-1]*grid[i][j];
                } else if (j==0) {
                    mx[i][j]=mx[i-1][j]*grid[i][j];
                    mn[i][j]=mn[i-1][j]*grid[i][j];
                } else {
                    mx[i][j]=max(mx[i][j-1]*grid[i][j], mx[i-1][j]*grid[i][j]);
                    mx[i][j]=max(mx[i][j], max(mn[i][j-1]*grid[i][j], mn[i-1][j]*grid[i][j]));
                    mn[i][j]=min(mx[i][j-1]*grid[i][j], mx[i-1][j]*grid[i][j]);
                    mn[i][j]=min(mn[i][j], min(mn[i][j-1]*grid[i][j], mn[i-1][j]*grid[i][j]));
                }
            }
        }
        // for (int i=0; i<n; ++i) {
        //     for (int j=0; j<m; ++j) {
        //         cout << mx[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        // cout << endl;
        // for (int i=0; i<n; ++i) {
        //     for (int j=0; j<m; ++j) {
        //         cout << mn[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        if (mx[n-1][m-1]<0) return -1;
        return mx[n-1][m-1]%mod;
    }
};
