class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<long long>> f(n, vector<long long>(m, -1)), g(n, vector<long long>(m, 1));
        f[0][0] = g[0][0] = grid[0][0];
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {

                if (j + 1 < m) {
                    if (f[i][j] >= 0) {
                        f[i][j + 1] = max(f[i][j + 1], f[i][j] * grid[i][j + 1]);
                        g[i][j + 1] = min(g[i][j + 1], f[i][j] * grid[i][j + 1]);
                    }
                    if (g[i][j] <= 0) {
                        f[i][j + 1] = max(f[i][j + 1], g[i][j] * grid[i][j + 1]);
                        g[i][j + 1] = min(g[i][j + 1], g[i][j] * grid[i][j + 1]);
                    }
                }

                if (i + 1 < n) {
                    if (f[i][j] >= 0) {
                        f[i + 1][j] = max(f[i + 1][j], f[i][j] * grid[i + 1][j]);
                        g[i + 1][j] = min(g[i + 1][j], f[i][j] * grid[i + 1][j]);
                    }
                    if (g[i][j] <= 0) {
                        f[i + 1][j] = max(f[i + 1][j], g[i][j] * grid[i + 1][j]);
                        g[i + 1][j] = min(g[i + 1][j], g[i][j] * grid[i + 1][j]);
                    }
                }
            }
        return f[n - 1][m - 1] % 1000000007;
    }
};
