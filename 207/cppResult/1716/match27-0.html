class Solution {
    typedef long long ll;
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int num_row = grid.size(), num_col = grid[0].size();
        int MOD = 1e9 + 7;
        vector<vector<ll>> max_grid(num_row + 1, vector<ll>(num_col + 1, 1));
        vector<vector<ll>> min_grid(num_row + 1, vector<ll>(num_col + 1, 1));
        for (int i = 0; i < num_row; i++) {
            for (int j = 0; j < num_col; j++) {
                
                if (i == 0 && j == 0) {
                    max_grid[i + 1][j + 1] = grid[i][j];
                    min_grid[i + 1][j + 1] = grid[i][j];

                }
                else if (i == 0) {
                    max_grid[i + 1][j + 1] = max((ll) grid[i][j] * max_grid[i + 1][j], (ll) grid[i][j] * min_grid[i + 1][j]);
                    min_grid[i + 1][j + 1] = min(grid[i][j] * max_grid[i + 1][j], grid[i][j] * min_grid[i + 1][j]);

                }
                else if (j == 0) {
                    max_grid[i + 1][j + 1] = max(grid[i][j] * max_grid[i][j + 1], grid[i][j] * min_grid[i][j + 1]);
                    min_grid[i + 1][j + 1] = min(grid[i][j] * max_grid[i][j + 1], grid[i][j] * min_grid[i][j + 1]);
                } else {
                    
                    max_grid[i + 1][j + 1] = max(grid[i][j] * max_grid[i][j + 1], grid[i][j] *  max_grid[i + 1][j]);
                    max_grid[i + 1][j + 1] = max(max_grid[i + 1][j + 1], grid[i][j] * min(min_grid[i][j + 1], min_grid[i + 1][j]));

                    min_grid[i + 1][j + 1] = grid[i][j] * max(max_grid[i][j + 1], max_grid[i + 1][j]);
                    min_grid[i + 1][j + 1] = min(min_grid[i + 1][j + 1], grid[i][j] * min(min_grid[i][j + 1], min_grid[i + 1][j]));
                }
                
                
            }
        }
        
        if (max_grid[num_row][num_col] < 0) return -1;
        return max_grid[num_row][num_col] % MOD;
    }
};
