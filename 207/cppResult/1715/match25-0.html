class Solution {
public:
	int maxUniqueSplit(string s) {
		vector<string> ans;
		int sum = 0;
		maxSplit(s, 0, ans, sum);
		return sum;
	}

	void maxSplit(string s, int start, vector<string>& ans, int& sum) {
		if (start >= s.size()) {
			sum = max(sum, (int)ans.size());
			return;
		}
		for (int i = start; i < s.size(); i++) {
			string tem = s.substr(start, i - start + 1);
			if (find(ans.begin(), ans.end(), tem) == ans.end()) {
				ans.push_back(tem);
				maxSplit(s, i+1, ans, sum);
				ans.pop_back();
			}
		}

	}
};
