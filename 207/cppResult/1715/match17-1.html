class Solution {
public:
    int cnt = 0;
    int maxUniqueSplit(string s) {
        unordered_set<string> st;
        dfs(s, st, 0);
        return cnt;
    }
    
    void dfs(string& s, unordered_set<string>& st, int idx) {
        if (idx == s.size()) {
            cnt = max((int)st.size(), cnt);

        }
        for (int i = 1; i <= s.size() - idx; i++) {
            string sub = s.substr(idx, i);
            if (st.count(sub)) continue;
            st.insert(sub);
            dfs(s, st, idx + i);
            st.erase(sub);
        }
    }
};
