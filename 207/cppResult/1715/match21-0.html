class Solution {
public:
    set<string> vis;
    int dfs(string &s,int l)
    {
        int ret=0;
        for (int i=l;i>=0;i--)
        {
            string now=s.substr(i,l-i+1);
            if (!vis.count(now))
            {
                vis.insert(now);
                ret=max(ret,dfs(s,i-1)+1);
                vis.erase(now);
            }
        }
        return ret;
    }
    int maxUniqueSplit(string s) {
        vis.clear();
        return dfs(s,s.length()-1);
    }
};
