class Solution {
public:
    int res = 1;
    int maxUniqueSplit(string s) {
        unordered_set<string> subset;
        helper(s, subset, 0);
        return res;
    }
    
    void helper(string& s, unordered_set<string>& subset, int idx) {
        if (idx == s.size()) {
            res = max(res, (int)subset.size());
        }
        for (int i = idx; i < s.size(); i++) {
            string sub = s.substr(idx, i-idx+1);
            if (!subset.count(sub)) {
                subset.insert(sub);
                helper(s, subset, i+1);
                subset.erase(sub);
            }
        }
    }
};
