class Solution {
public:
    int ans = 1;
    void dfs(int i, string &s, set<string> &st){
        if(i >= s.size()){
            if(st.size() > ans)
                ans = st.size();
            return;
        }
        int len = s.size() - i;
        if(len + st.size() <= ans)
            return;
        for(int l = 1; l <= len; l++){
            string sub = s.substr(i,l);
            if(st.find(sub) == st.end()){
                st.insert(sub);
                dfs(i+l,s,st);
                st.erase(sub);
            }
        }
    }
    int maxUniqueSplit(string s) {
        set<string> st;
        dfs(0,s,st);
        return ans;
    }
};
