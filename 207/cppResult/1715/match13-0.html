class Solution {
public:
    int maxUniqueSplit(string s) {
        unordered_set<string> unique;
        int maxunique = 1;
        recurse(s,0,unique,maxunique);
        return maxunique;
    }
    void recurse(string & s, int index, unordered_set<string> & unique,int &maxunique){
        if (index >= s.size()){
            maxunique = max(maxunique,int(unique.size()));
            return;
        }
        string temp;
        for (int i =index;i<s.size();i++){
            temp += s[i];
            if (!unique.count(temp)){
                //cout<<temp<<endl;
                unique.insert(temp);
                recurse(s,i+1,unique,maxunique);
                unique.erase(temp);
            }
        }
        return;
    }
};
