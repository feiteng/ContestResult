class Solution {
public:
    int maxUniqueSplit(string s)
    {
        /*unordered_set<string> st;
        string cur;
        for (auto ch : s)
        {
            cur += ch;
            if (st.count(cur))
                continue;
            st.insert(cur);
            cur.clear();
        }
        
        return st.size();*/
        
        unordered_set<string> vals;
        return impl(s, 0, vals);
    }
    
private:
    int impl(string s, int p, unordered_set<string>& vals)
    {
        if (p == s.size())
            return vals.size();
        
        int res = 0;
        string cur;
        for (int i = p; i < s.size(); ++i)
        {
            cur += s[i];
            if (!vals.count(cur))
            {
                vals.insert(cur);
                res = max(res, impl(s, i + 1, vals));
                vals.erase(cur);
            }
        }
        
        return res;
    }
};
