class Solution {
public:
    int maxUniqueSplit(string s) {
        return dfs(s,0);
    }
    
private:
    unordered_set<string> seen;
    int dfs(string& s,int idx){
        if(idx==s.size())return 0;
        string cur;
        int res=0;

        int n=s.size();
        for(int i=idx;i<n;i++){
            cur+=s[i];
            if(!seen.count(cur)){
                seen.insert(cur);
                res=max(res,1+dfs(s,i+1));
                seen.erase(cur);
            }
        }
        return res;
    }
};
