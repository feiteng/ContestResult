class Solution {
private:
    unordered_set<string>st;
    int ret;
    void dfs(string&s, int pos, int cnt){
        if(pos==s.size()){
            ret =max(ret, cnt);
            return;
        }
        string tt;
        for(int i=pos;i<s.size();i++){
            tt+=s[i];
            if(st.find(tt)==st.end()){
                st.insert(tt);
                dfs(s,i+1,cnt+1);
                st.erase(tt);
            }
        }
    }
public:
    int maxUniqueSplit(string s) {
        st.clear();
        ret = 0;
        dfs(s, 0, 0);
        return ret;
    }
};
