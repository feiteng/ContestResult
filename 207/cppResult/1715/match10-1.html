class Solution {
public:
    int maxUniqueSplit(string s) {
        unordered_set<string> counter;
        int res = 0;
        helper(s, 0, counter, res);
        return res;
    }
    
    void helper(string s, int start, unordered_set<string>& counter, int& res) {
        if (start == s.size()) {
            res = max(res, int(counter.size()));
            return;
        }
        for (int i = start + 1; i <= s.size(); i++) {
            string cur = s.substr(start, i - start);
            if (counter.count(cur) == 0) {
                counter.insert(cur);
                helper(s, i, counter, res);
                counter.erase(cur);
            }
        }
    }
};
