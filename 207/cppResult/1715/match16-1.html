class Solution {
public:
    int ans;
    void dfs(unordered_set<string>& mem, const string& s, int idx) {
        if (idx == s.size()) {
            ans = max(ans, (int)mem.size());
            return;
        }
        
        for (int i = idx; i < s.size(); ++i) {
            string tmp = s.substr(idx, i - idx + 1);
            if (mem.find(tmp) == mem.end()) {
                mem.insert(tmp);
                dfs(mem, s, i+1);
                mem.erase(tmp);
            }   
        }
    }
    
    int maxUniqueSplit(string s) {
        ans = 0;
        unordered_set<string> mem;
        dfs(mem, s, 0);
        return ans;
    }
};
