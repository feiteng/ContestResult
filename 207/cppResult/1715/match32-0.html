class Solution {
    int ans = 0;
    void dfs(string& s, int cur, int cnt, set<string>& ss) {
        if(cur == s.size()) {
            ans = max(ans, cnt);
            return;
        }
        for(int j = cur; j < s.size(); ++j) {
            string t = s.substr(cur, j - cur + 1);
            if(ss.count(t) == 0) {
                ss.insert(t);
                dfs(s, j + 1, cnt + 1, ss);
                ss.erase(t);
            }
        }
    }
public:
    int maxUniqueSplit(string s) {
        ans = 0;
        set<string> ss;
        dfs(s, 0, 0, ss);
        return ans;
    }
};
