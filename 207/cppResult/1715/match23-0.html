class Solution {
public:
    int maxCtr = 0;
    void checkAllOptions(string& s, int pos, unordered_set<string>& visited) {
        if (pos == s.size()) {
            if (visited.size() > maxCtr)
                maxCtr = visited.size();
            return;
        }
        
        string curStr = "";
        for (int i = pos; i < s.size(); i++) {
            curStr += s[i];
            if (visited.find(curStr) == visited.end()) {
                visited.insert(curStr);
                checkAllOptions(s, i + 1, visited);
                visited.erase(curStr);
            }
        }
    }
    
    int maxUniqueSplit(string s) {
        unordered_set<string> visited;
        checkAllOptions(s, 0, visited);
        return maxCtr;
    }
};
