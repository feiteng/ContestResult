class Solution {
public:
    int maxUniqueSplit(string s) {
        int res = 1;
        unordered_set<string> set;
        dfs(s, 0, set, res);
        return res;
    }
    
    void dfs(string & s, int start, unordered_set<string> & set, int & res) {
        if (start >= s.size()) {
            res = max(res, (int)set.size());
            return;
        }
        
        for (int i=start; i<s.size(); ++i) {
            string t = s.substr(start, i - start + 1);
            if (set.count(t)) continue;
            
            set.insert(t);
            dfs(s, i + 1, set, res);
            set.erase(t);
        }
    }
};
