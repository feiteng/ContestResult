class Solution {
public:
    int ans;

    int N;
    void dfs(string& s, unordered_set<string>& vis, int cur) {
        // cout << vis.size() << " " << cur << endl;
        if (cur == s.size()) {
            ans = max(ans, (int)vis.size());
            return;
        }
        for (int i=cur+1; i<=N; i++) {
            string t = s.substr(cur, i-cur);
            if (vis.count(t)) continue;
            vis.insert(t);
            dfs(s, vis, i);
            vis.erase(t);
        }
    }
    int maxUniqueSplit(string s) {
        N = s.size();
        ans = 1;
        unordered_set<string> vis;
        dfs(s, vis, 0);
        return ans;
    }
};
