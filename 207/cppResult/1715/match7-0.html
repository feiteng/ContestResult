class Solution {
public:
    int solve(int id, string s,set < string > &st){
        if(id == s.size())
            return 0;
        string cur = "";
        int ans = 0;
        for(int i = id; i < s.size(); i++) {
            cur += s[i];
            if(st.count(cur) == 0) {
                st.insert(cur);
                ans = max(ans, solve(i + 1, s, st) + 1);
                st.erase(cur);
            }
        }
        return ans;
    }
    int maxUniqueSplit(string s) {
        set < string > st;
        return solve(0, s,st);
    }
};
