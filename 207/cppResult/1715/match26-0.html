class Solution {
    int res = 1;
public:
    int maxUniqueSplit(string s) {
        set<string> m;
        dfs(s,m,0);
        return res;
    }
    void dfs(string& s,set<string>& m, int start){
        if(start == s.size()){
            int q = m.size();
            res = max(res,q);
            return;
        }
        for(int i = start; i < s.size(); ++i){
            string t = s.substr(start,i-start+1);
            if(m.find(t) == m.end()){
                m.insert(t);
                dfs(s,m,i+1);
                m.erase(t);
            }
        }
    }
};
