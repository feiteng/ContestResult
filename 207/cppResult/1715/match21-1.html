class Solution {
public:
    
    int go(int left, string &s, unordered_set<string>& ss)
    {
        int ans = 0;
        for (int len = 1; len <= s.size(); len ++)
        {
            int r = left + len-1;
            
            if (r >= s.length())
                return ans;
            

            string the = s.substr(left, len);
            if (ss.count(the) == 0)
            {
                ss.insert(the);
                ans = max(ans, 1 + go(left + len, s, ss));
                ss.erase(the);
            }
        }
        return ans;
    }
    
    int maxUniqueSplit(string s) {
        unordered_set<string> ss;
        return go(0, s, ss);
    }
};
