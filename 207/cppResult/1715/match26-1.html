class Solution {
    int res = 0;
public:
    int maxUniqueSplit(string s) {
        set<string> m;
        
        helper(s, 0, m);
        
        return res;
    }
    
    void helper(string& s, int st, set<string>& m)
    {
        if(st == s.size())
        {
            res = max(res, int(m.size()));

            return;
        }
        
        for(int end = st; end < s.size(); end++)
        {
            string sub = s.substr(st, end-st+1);
            
            if(m.find(sub) == m.end())
            {
                m.insert(sub);
                
                helper(s, end+1, m);
                
                m.erase(sub);
            }
        }
    }
};
