class Solution {
public:
    int maxUniqueSplit(string s) {
        unordered_set<string> res;
        int cnt = 1;
        dfs(res, s, 0, cnt);
        return cnt;
    }
    void dfs(unordered_set<string> & res, string &s, int start, int & cnt) {
        if (start == s.size()) {
            cnt = max(cnt, (int)res.size());
            return;
        }
        for (int i = start; i < s.size(); ++i) {
            string tmp = s.substr(start, i - start + 1);
            if (res.count(tmp) == 0) {
                res.insert(tmp);
                dfs(res, s, i+1, cnt);
                res.erase(tmp);
            }
        }
    }
};
