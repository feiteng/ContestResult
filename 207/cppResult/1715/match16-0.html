class Solution {
public:
    unordered_set<string> uniqStr;
    int ans;
    void dfs(const string &s, int idx) {
        if (idx == s.size()) {
            ans = max(ans, (int)uniqStr.size());
            return;
        }
        for (int nextIdx = idx + 1; nextIdx <= s.size(); ++ nextIdx) {
            string w = s.substr(idx, nextIdx - idx);
            if (uniqStr.find(w) == uniqStr.end()) {
                uniqStr.insert(w);
                dfs(s, nextIdx);
                uniqStr.erase(w);
            }
        }
    }
    int maxUniqueSplit(string s) {
        ans = 0;
        dfs(s, 0);
        return ans;
    }
};
