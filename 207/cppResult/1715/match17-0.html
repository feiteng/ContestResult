class Solution {
public:
    int ans = 0;
    unordered_set<string> m;
    int maxUniqueSplit(string s) {
        //回溯法：每个位置可拆可不拆
        backtrack(s,0);
        return ans;
    }
    void backtrack(string s,int start){
        if(start == s.size()){
            ans = max(ans,(int)m.size());
            return;
        }
        //选择拆分
        for(int i = start;i < s.size();i++){
            string cur = s.substr(start,i - start + 1);
            if(m.count(cur))continue;
            m.insert(cur);
            backtrack(s,i+1);
            m.erase(cur);
        }
    }
};
