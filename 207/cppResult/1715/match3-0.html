class Solution {
public:
    int maxUniqueSplit(string s) {
        unordered_set<string> st;
        int res = 1;
        dfs(st,s,0,res);
        return res;
    }
    
    void dfs(unordered_set<string>& st, string& s, int start, int& res)
    {
        if(start == s.size())
        {
            res = max(res,(int)st.size());
            return;
        }
        for(int i = 1; i <= s.size() - start; i++)
        {
            string next = s.substr(start,i);
            if(st.count(next)) continue;
            st.insert(next);
            dfs(st,s,start+i,res);
            st.erase(next);
        }
    }
};
