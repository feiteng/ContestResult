#define all(x) (x).begin(), (x).end()
typedef long long ll;
typedef pair<int, int> ii;


class Solution {
    int res = 0;
    
    void backtracking(string s, unordered_set<string>& visited, int start, int cnt) {
        if (start >= s.length()) {
            res = max(res, cnt);
            return;
        }
        string cur = "";

        for (int i = start; i < s.length(); i++) {
            cur += s[i];
            if (visited.count(cur) == 0) {
                visited.insert(cur);
                backtracking(s, visited, i + 1, cnt + 1);
                visited.erase(cur);
            }
        }
    }
public:
    int maxUniqueSplit(string s) {
        int n = s.length();
        unordered_set<string> v;
        backtracking(s, v, 0, 0);
        return res;
    }
};
