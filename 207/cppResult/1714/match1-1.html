class Solution {
public:
    string reorderSpaces(string text) {
        int cnt = 0;
        string pre;
        vector<string> words;
        for(int i=0;i<text.size();++i){
            if(text[i]==' '){
                ++cnt;
                if(pre!=""){
                    words.push_back(pre);
                    pre = "";
                }
            }
            else
                pre += text[i];
        }
        if(pre!="") words.push_back(pre);
        string ret;

        if(words.size()>1){
            int each = cnt/(words.size()-1);
            for(int i=0;i<words.size();++i){
                ret += words[i];
                if(i!=words.size()-1){
                    for(int j=1;j<=each;++j){
                        ret += " ";
                    }
                }
            }
            for(int i=0;i<cnt%(words.size()-1);++i){
                ret += " ";
            }
        }
        else{
            ret = words[0];
            for(int i=0;i<cnt;++i){
                ret += " ";
            }
        }
        return ret;
    }
};
