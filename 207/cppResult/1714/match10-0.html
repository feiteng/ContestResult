class Solution {
public:
    string reorderSpaces(string text) {
        int spaces = 0;
        vector<string> s;
        string curr = "";
        for(int i = 0; i < text.length(); i++) {
            if(text[i] == ' ') {
                if(curr.length() > 0) {
                    s.push_back(curr);
                    curr = "";
                }
                spaces++;
            } else {
                curr += text[i];
            }
        }
        if(curr.length() > 0) {
            s.push_back(curr);
            curr = "";
        }
        string ans = "";
        int totalWords = s.size() - 1;
        if(totalWords == -1) {
            return text;
        }
        if(totalWords == 0) {
            ans += s[0];
            for(int j = 0; j < spaces; j++) {
                ans += " ";
            }
            return ans;
        }
        int diffInEach = spaces / totalWords;
        for(int i = 0; i < s.size(); i++) {
            ans += s[i];
            if(i != s.size() - 1) {
                for(int j = 0; j < diffInEach; j++) {
                    ans += " ";
                }
            }
        }
        int extra = spaces - (diffInEach * totalWords);
        for(int j = 0; j < extra; j++) {
            ans += " ";
        }
        return ans;
    }
};
