class Solution {
public:
    string reorderSpaces(string text) 
    {
        vector<string>words;
        int count = 0;
        for (int i=0; i<text.size(); i++)
        {
            if (text[i]==' ')
            {
                count++;
                continue;                
            }
            int j=i;
            while (j<text.size() && text[j]!=' ')
                j++;
            words.push_back(text.substr(i,j-i));
            i = j-1;
        }
        
        cout<<count<<endl;
        
        int k = words.size();
        string ret;
        if (k==1)
        {
            ret+=words[0];
            for (int i=0; i<count; i++)
                ret.push_back(' ');
            return ret;
        }
        
        int q = count/(k-1);
        int r = count%(k-1);
        ret+=words[0];
        for (int i=1; i<words.size(); i++)
        {
            for (int i=0; i<q; i++)
                ret.push_back(' ');
            ret+=words[i];
        }
        for (int i=0; i<r; i++)
            ret.push_back(' ');
        return ret;
    }
};
