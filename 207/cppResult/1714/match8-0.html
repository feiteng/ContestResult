class Solution {
public:
    string reorderSpaces(string text) {
        int kong = 0,cnt = 0,n = text.size();
        int i = 0;
        vector<string> rec;
        while(i < n){
            while(i < n && text[i] == ' '){
                kong++;
                i++;
            }
            int p = i;
            while(i < n && text[i] >= 'a' && text[i] <= 'z')i++;
            if(text[i - 1] >= 'a' && text[i - 1] <= 'z'){
                cnt++;
                rec.push_back(text.substr(p,i - p));
            }
        }
        string ans;
        if(cnt == 1){
            ans = rec[0];
            for(int i = 0;i < kong;i++)ans = ans + ' ';
            return ans;
        }
        int x = kong / (cnt - 1);
        int y = kong % (cnt - 1);
        for(int i = 0;i < rec.size();i++){
            ans = ans + rec[i];
            if(i != rec.size() - 1)for(int k = 0;k < x;k++)ans = ans + ' ';
        }
        for(int k = 0;k < y;k++)ans = ans + ' ';
        return ans;
    }
};
