class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> strs;
        int len = 0;
        for(int i =0;i<text.length();){
            if(text[i]==' '){
                i++;
                continue;
            }
            int j =i;
            string tmp="";
            while(j<text.length()&&text[j]!=' '){
                tmp.push_back(text[j]);
                j++;
            }
            strs.emplace_back(tmp);
            len+=tmp.length();
            i = j;
        }
        int empty  = text.length()-len;
        if(strs.size()==1){
            string ans = strs[0];
            for(int i =0;i<empty;i++){
                ans+=" ";
            }
            return ans;
        }
        int avg = empty/(strs.size()-1);
        int rem = empty%(strs.size()-1);
        string empty_s = "";
        for(int i =0;i<avg;i++){
            empty_s+=" ";
        }
        string ans = strs[0];
        for(int i =1;i<strs.size();i++){
            ans+=empty_s;
            ans+=strs[i];
        }
        for(int i =0;i<rem;i++){
            ans+=" ";
        }
        return ans;
    }
};
