class Solution {
public:
    string reorderSpaces(string text) {
        string word = "";
        int space = 0;
        vector<string> words; 
        for(int i = 0; i < text.size(); i++) {
            if(text[i] != ' ') {
                word+=text[i];
            } else {
                space++;
                if(word != "") {
                    words.push_back(word);
                    word = "";
                }
            }
        }
        if(word != "") {
            words.push_back(word);
        }
        int per = 0;
        if(words.size()>1) {
            per = space/(words.size()-1);
        }
        string spa = "";

        for(int i = 0; i < per; i++) {
            spa += " ";
        }
        string ans = "";
        for(int i = 0; i+1 < words.size(); i++) {
            ans += words[i]+spa;
        }
        ans += words[words.size()-1];
        while(ans.size()<text.size()) {
            ans += " ";
        }
        return ans;
    }
};
