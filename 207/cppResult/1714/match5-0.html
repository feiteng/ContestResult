class Solution {
public:
    string reorderSpaces(string text) {
        int len=text.size();
        int cnt=0;
        vector<string> vec;
        for(int i=0; i<len; i++) {
            char ch=text[i];
            if (ch==' ') cnt++;
            else {
                int j=i;
                string  cur;
                cur+=text[j];
                while(j+1<len&&text[j+1]!=' ') {
                    j++;
                    cur+=text[j];
                }
                vec.push_back(cur);
                i=j;
            }
        }
        if (cnt==0) return text;
        if (vec.size()==1) {
            string ret;
            ret+=vec[0];
            for(int j=0; j<cnt; j++) ret+=' ';
            return ret;
        }
        int div=cnt/(vec.size()-1);
        int rem=cnt-div*(vec.size()-1);
        string ret;
        for(int i=0; i<vec.size(); i++) {
            ret+=vec[i];
            if (i+1==vec.size()) {
                for(int j=0; j<rem; j++) ret+=' ';
            }
            else {
                for(int j=0; j<div; j++) ret+=' ';
            }
        }
        return ret;
    }
};
