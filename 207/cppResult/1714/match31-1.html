class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        int spaces = 0;
        string word = "";
        for (char c : text) {
            if (c == ' ') {
                spaces++;
                if (!word.empty()) {
                    words.push_back(word);
                    word = "";
                }
            } else {
                word += c;
            }
        }
        if (!word.empty()) {
            words.push_back(word);
        }
        int spaces_between_words = words.size() == 1 ? 0 : spaces / (words.size() - 1);
        int remaining_spaces = spaces - spaces_between_words * (words.size() - 1);
        string res = words[0];
        for (int i = 1; i < words.size(); ++i) {
            for (int j = 0; j < spaces_between_words; ++j) {
                res += ' ';
            }
            res += words[i];
        }
        for (int j = 0; j < remaining_spaces; ++j) {
            res += ' ';
        }
        return res;
    }
};
