class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> v;
        int cnt = 0, last = -1;
        for(int i = 0;i < text.size(); ++i) {
            if(text[i] == ' ') {
                if(last != -1) {
                    v.push_back(text.substr(last, i - last));
                    last = -1;
                }
                cnt++;
            }
            else {
                if(last == -1) {
                    last = i;
                }
            }
        }
        
        if(cnt == 0) {
            return text;
        }
        if(last != -1) {
            v.push_back(text.substr(last));
        }
        
        string res;
        
        if(v.size() == 1) {
            res += v[0];
            for(int i = 0; i < cnt; ++i) {
                res += ' ';
            }
            return res;
        } else {
            int d = cnt / (v.size() - 1);
            int r = cnt % (v.size() - 1);
            res += v[0];
            for(int i = 1; i < v.size(); ++i) {
                for(int j = 0; j < d; ++j) {
                    res += ' ';
                }
                res += v[i];
            }
            for(int i = 0; i < r; ++i) {
                res += ' ';
            }
            return res;
        }
        
        return "";
    }
};
