class Solution {
public:
    string reorderSpaces(string text) {
        int len = text.length(), space = 0, wstart = -1;
        bool word = false;
        vector<string> words;
        for(int i = 0; i < len; ++i) {
            if(text[i] == ' ') {
                ++space;
                if(word) {
                    word = false;
                    words.push_back(text.substr(wstart, i - wstart));
                }
            }
            else {
                if(!word) {
                    word = true;
                    wstart = i;
                }
            }
        }
        if(word) words.push_back(text.substr(wstart, len - wstart));
        string ans = "";
        if(words.size() == 1) {
            ans += words[0];
            for(int i = 0; i < space; ++i) ans += " ";
        } else {
            int between = space / (words.size() - 1), tail = space % (words.size() - 1);
            ans += words[0];
            for(int i = 1; i < words.size(); ++i) {
                for(int j = 0; j < between; ++j) ans += " ";
                ans += words[i];
            }
            for(int i = 0; i < tail; ++i) ans += " ";
        }
        return ans;
        
    }
};
