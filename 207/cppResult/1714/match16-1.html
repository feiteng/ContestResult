class Solution {
public:
    string reorderSpaces(string s) {
        int n=s.size();
        vector<string> A;
        int cnt=0;
        for (int i=0;i<n;++i) if (s[i]==' ') ++cnt;
        for (int i=0,j;i<n;++i)
        {
            if (s[i]==' ') continue;
            string a;
            j=i;
            while (j+1<n && s[j+1]!=' ') ++j;
            for (int k=i;k<=j;++k) a.push_back(s[k]);
            A.push_back(a);
            i=j;
        }
        //for (int i=0;i<A.size();++i) cout<<A[i]<<endl;
        if (A.size()==1)
        {
            string ans=A[0];

            for (int i=1;i<=cnt;++i) ans.push_back(' ');
            return ans;
        }
        int v=cnt/(A.size()-1);
        int r=cnt%(A.size()-1);
        string ans=A[0];
        for (int i=1;i<A.size();++i)
        {
            for (int j=1;j<=v;++j) ans.push_back(' ');
            ans+=A[i];
        }
        for (int j=1;j<=r;++j) ans.push_back(' ');
        return ans;
    }
};
