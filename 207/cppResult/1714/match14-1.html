class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;

        int cnt = 0;
        for(int i = 0; i < text.length(); i ++){
            if(text[i] == ' ') cnt ++;
            else{
                int t = i;
                while(t < text.length() && text[t] != ' ')
                    t ++;
                words.push_back(text.substr(i, t - i));
                i = t - 1;
            }
        }
        string rnt = words[0];
        if(words.size() > 1){

            int each = cnt / (words.size() - 1);
            for(int i = 1; i < words.size(); i ++){
                for(int j = 0; j < each; j ++)
                    rnt += ' ';
                cnt -= each;
                rnt += words[i];
            }
        }
        while(cnt--) rnt += ' ';
        return rnt;
    }
};
