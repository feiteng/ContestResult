class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> data;
        int sum = 0;
        int i = 0;
        int n = text.size();
        string now = "";
        while (i < n){
            while (i < n && text[i]!=' ') now+=text[i++];
            if (now.size() > 0) data.push_back(now);
            now = "";
            while (i < n && text[i]==' ') ++sum, ++i;
        }
        int m = data.size();
        string ans;

        if (m == 1){
            ans = data[0];
            for (int i = 0; i < sum; ++i) ans += " ";
        }
        else{
            int cnt = sum / (m-1);
            ans = data[0];
            for (int i = 1; i < m; ++i){
                for (int j = 0; j < cnt; ++j) ans += " ";
                ans += data[i];
            }
            for (int j = 0; j < sum%(m-1); ++j)
                ans += " ";
        }
        return ans;
    }
};
