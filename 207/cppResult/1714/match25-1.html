class Solution {
public:
    string reorderSpaces(string s) {
        vector<string> words;
        int spaces = 0;
        bool f = false;

        for (auto c : s) {
            if (c == ' ') {
                f = false;
                spaces++;
            } else {
                if (f == false) {
                    words.push_back("");
                } 
                words.back().push_back(c);
                f = true;
            }
        }
        
        //printf("spaces = %d, %d\n", spaces, words.size()); 
        
        string ans;
        int n = words.size();
        int t = n == 1 ? 0 : spaces / (n-1);
        int re = spaces - t * (n-1);
        

        for (int i = 0; i < words.size(); ++i) {
            ans += words[i];
            
            if (i != n - 1)
                ans += string(t, ' ');
        }
        ans += string(re, ' ');
        return ans;
    }
};
