class Solution {
public:
    string reorderSpaces(string text) {
        string res;
        int count = 0;
        string tmp;
        vector<string> words;
        for(int i = 0; i < text.size(); i++)
        {
            if(text[i] == ' ') {
                if(tmp != "") words.push_back(tmp);
                tmp = "";
                count++;
            }
            else{
                tmp.push_back(text[i]);
            }
        }
        if(tmp != "") words.push_back(tmp);
        if(words.size() == 1)
        {

            return words[0] + string(count,' ');
        }
        int len = count / (words.size() - 1);
        int left = count % (words.size() - 1);
        for(int i = 0; i < words.size(); i++)
        {
            res += words[i];
            if(i+1 != words.size()) res += string(len,' ');
        }
        res += string(left,' ');
        return res;
    }
};
