class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        
        int cnt = 0; string cur = "";
        for (char c: text){
            if (c == ' '){
                cnt += 1;
                if (cur.length() > 0) words.push_back(cur);
                cur = "";
            }else cur += c;
        }
        if (cur.length() > 0) words.push_back(cur);
        
        int n = words.size(), p, q;

        if (n == 1){
            p = 0; q = cnt;
        }else{
            p = cnt / (n - 1); q = cnt % (n - 1);
        }
        
        string ans = "";
        for (int i = 0; i < n; i++){
            if (i > 0)

                for (int k = 0; k < p; k++) ans += ' ';
            ans += words[i];
        }
        for (int k = 0; k < q; k++) ans += ' ';
        
        return ans;
    }
};
