class Solution {
public:
    string reorderSpaces(string text) {
        int cnt=0;
        vector<string> vec;
        string res;
        for(int i=0;i<text.size();){
            string str;
            while(i<text.size()&&text[i]!=' '){
                str+=text[i];
                i++;
            }
            if(str.length()) vec.push_back(str);
            while(i<text.size()&&text[i]==' ') cnt++,i++;
        }
        cout<<vec.size()<<endl;
        if(!vec.size()) return text;
        if(vec.size()==1){
            res+=vec[0];
            for(int i=vec[0].size();i<text.size();i++) res+=' ';
            return res;
        }
        int t=cnt/(vec.size()-1);
        int mod=cnt%(vec.size()-1);
        for(int i=0;i<vec.size()-1;i++){
            res+=vec[i];
            for(int j=0;j<t;j++) res+=' ';
        }
        res+=vec[vec.size()-1];
        for(int i=0;i<mod;i++) res+=' ';
        return res;
    }
};
