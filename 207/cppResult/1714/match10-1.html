class Solution {
public:
    string reorderSpaces(string text) {
        int blank = 0,word = 0;

        int n = text.size();
        vector<string> words;
        string s = "";
        for (int i = 0; i < n; i++) {
            if (text[i] == ' ') {
                if (s.size()) {
                    words.push_back(s);
                    s = "";
                    word++;

                }
                blank++;
            } else {
                s += text[i];
            }
        }
        if (s.size()) {
            words.push_back(s);
            word++;
        }
        int avg = word == 1 ? 0:blank / (word-1);
        int remainder = word == 1? blank:blank % (word-1);
        // cout << avg << " " << remainder << endl;

        string ans = "";
        for (int i = 0; i < words.size(); i++) {
            ans += words[i];
            if (i != words.size() - 1) {
                for (int j = 0; j < avg; j++) {
                    ans += ' ';
                }
            }
        }
        for (int i = 0; i < remainder; i++) {
            ans += ' ';
        }
        return ans;
    }
};
