class Solution {
public:
    string reorderSpaces(string s) {
        int len = s.size();
        vector<string> vs;
        string ss = s;
        char* p = strtok((char*)ss.c_str()," ");
        while(p){
            vs.push_back(p);
            p = strtok(nullptr," ");
        }
        int sum = 0;
        for(char c:s){
            if(c==' ') sum++;
        }

        string ans= "";

        if(vs.size()==1){
            ans = vs[0];
            for(int i=0;i<sum;i++) ans += ' ';
            return ans;
        }
        int gap = sum/(vs.size()-1);
        int r = sum%(vs.size()-1);
        
        for(int i=0;i<vs.size()-1;i++){
            ans += vs[i];
            for(int i=0;i<gap;i++) ans += ' ';
        }
        ans += vs.back();
        for(int i=0;i<r;i++) ans += ' ';
        return ans;
    }
};
