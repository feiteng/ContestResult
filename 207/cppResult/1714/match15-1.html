class Solution {
public:
    string reorderSpaces(string text) {
        string res;
        int n_space = 0;
        int n_word = 0;
        for(int i=0; i<text.size(); i++){
            if(text[i] == ' ')
                n_space++;
        }
        //return res;
        
        vector<string> words;
        int stt = 0;
        int end = 0;
        bool flag = false;
        while(end < text.size()){
            if(text[end] == ' '){
                if(flag)
                    words.push_back(text.substr(stt, end-stt));
                flag = false;
                stt = end + 1;
                end = end + 1;
            }else{
                flag = true;
                end = end + 1;
            }
        }
        if(flag)
            words.push_back(text.substr(stt, end-stt+1));
        
        n_word = words.size();
        int average_space;
        int last_space;
        if(n_word > 1){
            average_space = n_space / (n_word-1);
            last_space = n_space - average_space * (n_word-1);
        }else{
            average_space = 0;
            last_space = n_space;
        }

        res = res + words[0];
        for(int i=1;i<words.size();i++){
            for(int k=0; k<average_space;k++)
                res = res + ' ';
            res = res + words[i];
        }
        for(int k=0; k<last_space; k++)
            res = res + ' ';
        return res;
    }
};
