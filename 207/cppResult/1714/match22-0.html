class Solution {
public:
    string reorderSpaces(string text) {
        int n = text.length();
        int cnt = 0;
        vector<string> v;
        string s="";
        for(int i=0;i<n;i++){
            if(text[i]==' '){
                cnt++;
                if(!s.empty())
                    v.push_back(s);
                s="";
            }
            else{
                s+=text[i];
            }
        }
        if(!s.empty())
            v.push_back(s);
        int rl = v.size();
        if(rl<=1){
            if(rl==1){
                string ans="";
                ans+=v[0];
                for(int i=0;i<cnt;i++)
                    ans+=' ';
                return ans;
            }
            return text;
        }
        int d = cnt/(rl-1);
        int rs = cnt%(rl-1);
        string ans="";
        for(int i=0;i<v.size();i++){
            ans+=v[i];
            if(i<v.size()-1){
                for(int j=0;j<d;j++)
                    ans+=' ';
            }
        }
        for(int i=0;i<rs;i++){
            ans+=' ';
        }
        return ans;
    }
};
