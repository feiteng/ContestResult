class Solution {
public:
    string reorderSpaces(string text) {
        string ans = "";
        int sp = 0;
        
        for(int i=0;i<text.size();i++){
            if(text[i]==' '){
                sp++;
            }
        }
        stringstream ss(text);
        string word;
        vector<string> words;
        while(ss >> word){
            words.push_back(word);
        }
        if(words.size()==1){
            ans = words[words.size()-1];
            for(int i=0;i<sp;i++){
                ans += " ";
            }
            return ans;
        }
        int z = sp/(words.size()-1);
        int last = sp%(words.size()-1);
        string spaces = "";
        for(int i=0;i<z;i++){
            spaces += " ";
        }
        for(int i=0;i<words.size()-1;i++){
            ans += words[i]+spaces;
        }
        ans += words[words.size()-1];
        for(int i=0;i<last;i++){
            ans += " ";
        }
        return ans;
    }
};
