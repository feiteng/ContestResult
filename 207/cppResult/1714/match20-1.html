class Solution {
public:
    string reorderSpaces(string text) {
        int cnt_space = 0;
        for (auto ch : text) {
            cnt_space += ch == ' ';
        }
        vector<string> words;
        for (int i = 0; i < text.size(); ++i) {
            if (text[i] == ' ') {
                continue;
            }
            int start = i;
            while (i < text.size() && text[i] != ' ') {

                i++;
            }
            words.push_back(text.substr(start, i - start));
        }
        string res;
        if (words.size() == 1) {
            res += words[0];
            for (int i = 0; i < cnt_space; ++i) {
                res += ' ';
            }
            return res;
        }

        int num_space = cnt_space / (words.size() - 1);
        for (int i = 0; i < words.size() - 1; ++i) {
            res += words[i];
            for (int i = 0; i < num_space; ++i) {
                res += ' ';
            }
        }
        res += words.back();
        for (int i = 0; i < cnt_space - (words.size() - 1) * num_space; ++i) {
            res += ' ';
        }
        return res;
    }
};
