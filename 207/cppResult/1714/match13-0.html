class Solution {
public:
    string reorderSpaces(string text) {
        int i=0;
        int n=text.length();
        int sum=0;
        vector<string>vec;
        while(i<n){
            while(i<text.length() && text[i]==' '){
                sum++;
                i++;
            }
            string str="";
            while(i<text.length() && text[i]!=' '){
                str+=text[i];
                i++;
            }
            if(str.length()!=0){
                vec.push_back(str);
            }
        }
        int si=vec.size();
        if(si==1){
            string str=vec[0];
            for(int i=0;i<sum;i++){
                str+=" ";
            }
            return str;
        }
        int num_space=sum/(si-1);
        int rem_space=sum%(si-1);
        string space="";
        for(int i=0;i<num_space;i++){
            space+=" ";
        }
        string str=vec[0];
        for(int i=1;i<vec.size();i++){
            str+=(space+vec[i]);
        }
        for(int i=0;i<rem_space;i++){
            str+=" ";
        }
        return str;
    }
};
