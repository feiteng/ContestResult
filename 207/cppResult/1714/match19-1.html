class Solution {
public:
    string reorderSpaces(string text) {
        int c=0;
        for (char ch: text) {
            if (ch==' ') c++;
        }
        stringstream s(text);
        string word;
        vector<string> words;
        while (s>>word) {
            words.push_back(word);
        }
        string res;
        if (words.size()==1) {
            res+=words[0];
            res+=string(c, ' ');

            return res;
        }
        int rem=c%(words.size()-1);
        int a=c/(words.size()-1);
        for (int i=0; i<words.size(); i++) {
            res+=words[i];
            if (i+1<words.size()) res+=string(a, ' ');
        }
        res+=string(rem, ' ');
        return res;
    }
};
