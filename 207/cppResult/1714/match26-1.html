class Solution {
public:
    string reorderSpaces(string text) {
        int n = text.size(), pos = 0, cnt = 0;
        vector<string> str;
        while (pos < n) 
        {
            if (text[pos] == ' ') {
                ++cnt, ++pos;
                continue;
            }
            string cur;
            while (pos < n && text[pos] != ' ') {
                cur += text[pos];
                ++pos;
            }
            str.push_back(cur);
        }
        int sz = str.size();

        if (sz == 1) {
            string ans = str[0];
            for (int i = 1; i <= cnt; ++i) {
                ans += ' ';
            }
            return ans;
        }

        int interval = cnt / (sz - 1);
        string ans = str[0];
        for (int i = 1; i < str.size(); ++i) {
            for (int j = 0; j < interval; ++j) {
                ans += ' ';
            }
            ans += str[i];
        }
        for (int i = 0; i < cnt - interval * (sz - 1); ++i) {
            ans += ' ';
        }
        return ans;
    }
};
