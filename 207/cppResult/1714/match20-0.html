class Solution {
public:
    string reorderSpaces(string text) {
        int cw=0,cb=0;
        vector<string> words;
        for(int i=0;i<text.length();){
            if(text[i]!=' '){
                cw++;
                string w = "";
                while(i<text.length()&&text[i]!=' '){
                    w +=text[i];
                    i++;
                }
                words.emplace_back(w);
            }else{
                cb++;
                i++;
            }
        }
        
        // cout<<"cb:"<<cb<<endl;
        // cout<<"cw:"<<cw<<endl;
        
        string ans;
        if(cw==1){
            ans = words[0];
            for(int i=0;i<cb;i++){
                ans += " ";
            }
            return ans;
        }else{
            int bn = cb/(cw-1);
            int left = cb-bn*(cw-1);
            ans = "";
            
            for(int i=0;i<cw-1;i++){
                ans += words[i];
                
                for(int j=0;j<bn;j++){
                    ans += " ";
                }
            }
            
            ans += words[cw-1];
            
            for(int i=0;i<left;i++){
                ans += " ";
            }
        }
        
        return ans;
    }
};
