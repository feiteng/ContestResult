class Solution {
public:
    string reorderSpaces(string text) {
        int spaceNum = -1;
        vector<string> words;
        string word;
        text += ' ';
        for (char c:text) {
            if (c == ' ') {
                spaceNum++;
                if (word != "") {
                    words.push_back(word);
                    word = "";
                } 
            } else {
                word += c;
            }
        }
        string ans = words[0];
        if (words.size() == 1) return ans+string(spaceNum, ' ');
        string spaceWord(spaceNum/(words.size()-1), ' ');
        for (int i=1; i<words.size(); i++) {
            ans += spaceWord + words[i];
        }
        return ans + string(spaceNum%(words.size()-1), ' ');
    }
};
