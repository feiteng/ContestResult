class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;

        int cnt=0;
        string s="";
        for (int i=0;i<text.size();i++)
        {
            if (text[i]==' ')
            {
                cnt++;
                if (!s.empty())
                    words.push_back(s);
                s="";
            }
            else
                s+=text[i];
        }
        if (!s.empty())
            words.push_back(s);
        string res;
        if (words.size()==1)
        {
            res+=words[0];
            for (int i=0;i<cnt;i++)
                res+=" ";
            return res;
        }

        int len=words.size()-1,avg=cnt/len,remain=cnt-avg*len;
        for (int i=0;i<words.size();i++)
        {
            res+=words[i];
            if (i!=words.size()-1)
            {
                for (int j=0;j<avg;j++)
                res+=" ";
            }
        }
        for (int i=0;i<remain;i++)
            res+=" ";
        return res;
    }
};
