class Solution {
public:
    string reorderSpaces(string text) {
        istringstream ss(text);
        string w = "";
        vector<string> t;
        int cnt = 0;
        for(char c : text) {
            if(c == ' ') ++cnt;
        }
        while(getline(ss, w, ' '))  {
            if(!w.empty()) t.push_back(w);
        }
        string res = "";

        if(t.size() == 1) {
            res = t[0];
            for(int i=0; i<cnt; ++i) res+=" ";
            return res;
        }
        int even = cnt/(t.size()-1);
        int residue = cnt%(t.size()-1);
        //cout << cnt << " " << t.size()-1 << " " << even << " " << residue << endl;
        for(int i= 0; i<t.size(); ++i) {
            string tt = t[i];
            res += tt;

            if(i != t.size()-1)
                for(int i=0; i<even; ++i) res+=" ";
        }
        for(int i=0; i<residue; ++i) res+=" ";
        return res;
    }
};
