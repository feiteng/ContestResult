class Solution {
public:
    string reorderSpaces(string t) {
        int n = t.size();
        stringstream ss(t);
        vector<string> v;
        string s;
        int sz = 0;
        while(ss >> s) {
            sz += s.size();
            v.push_back(s);
        }

        if(v.size() == 1) {
            return v[0] + string(n - v[0].size(), ' ');
        }
        int space = (n-sz) / (v.size()-1);
        int extra = (n-sz) % (v.size()-1);
        string res;
        for(int i = 0; i < v.size(); i++) {
            res += v[i];
            if(i < v.size()-1)  
                res += string(space, ' ');
        }
        res += string(extra, ' ');
        return res;
    }
};
