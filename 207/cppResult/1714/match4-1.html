class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        string word;
        int space = 0;
        for(char c : text) if(c == ' '){
            space += 1;
            if(word.size()){
                words.push_back(word);
                word = "";
            }
        }
        else word.push_back(c);
        cout << space << " " << words.size() << "\n";
        if(word.size())
            words.push_back(word);
        string res;

        if(words.size() == 1){
            res = words[0];
            for(int i = 0; i < space; i += 1) res += ' ';
        }
        else{
            int m = space / (words.size() - 1);

            res = words[0];
            for(int i = 1; i < words.size(); i += 1){
                for(int j = 0; j < m; j += 1) res += ' ';
                res += words[i];
            }
            for(int j = 0; j < space % (words.size() - 1); j += 1) res += ' ';
        }
        return res;
    }
};
