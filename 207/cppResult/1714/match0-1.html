class Solution {
public:
	string reorderSpaces(string text) 
	{
		int blank = 0;
		for (int i = 0; i < text.size(); i++)
			if (text[i] == ' ')
				blank++;
		stringstream ss(text);
		vector<string>str;
		string s;
		while (ss >> s)
			str.push_back(s);
		

		if (str.size() == 1)
			return str[0] + string(blank, ' ');
		string ret = str[0];
		int n = blank / (str.size() - 1);
		for (int i = 1; i < str.size(); i++)
			ret += string(n, ' ') + str[i];
		ret += string(blank % (str.size() - 1), ' ');
		return ret;
	}
};
