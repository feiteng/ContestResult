class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        int cnt(0);
        bool nw = true;
        for (const char& c: text) {
            if (c == ' ') {
                cnt++;
                nw = true;
            } else {
                if (nw) {
                    words.push_back("");
                }
                words.back() += c;
                nw = false;
            }
        }
        //cout << words.size() << endl;
        int space, extra;
        if (words.size() == 1) {
            space = 0;
            extra = cnt;
        } else {
            space = cnt / ((int) words.size() - 1);
            extra = cnt % ((int) words.size() - 1);
        }
        
        string ans;
        for (int i = 0; i < words.size(); ++i) {
            ans += words[i];
            if (i + 1 != words.size()) ans += string(space, ' ');
        }
        ans += string(extra, ' ');
        return ans;
    }
};
