class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> v; int spaces = 0; bool end=true;
        for (auto c : text){
            if (c==' ') spaces++, end=true;
            else {
                if (end) v.push_back(string(1, c)), end=false;
                else v.back()+=c, end=false;
            }
        }

        if (v.size()==1){
            string ans=v[0];
            for (int j = 0; j < spaces; j++) ans += " ";
            return ans;
        }
        int nm = (int)v.size()-1, dist = spaces/nm;

        string ans="";
        for (int i = 0; i < nm; i++){
            ans+=v[i];
            for (int j=0; j < dist; j++) ans+=" ";
        }
        ans += v[nm];
        for (int i = 0; i < spaces-dist*nm; i++) ans += " ";
        return ans;
    }
};
