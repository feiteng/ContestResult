class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        int blankCount = 0;
        string nowWord;
        for (auto &c : text) {
            if (c == ' ') {
                if (nowWord.empty()) {
                    blankCount++;
                } else {
                    blankCount++;
                    words.push_back(nowWord);
                    nowWord.clear();
                }
            } else {
                nowWord += c;
            }
        }
        if (!nowWord.empty()) {
            words.push_back(nowWord);
            nowWord.empty();
        }
        // printf("%s: %d\n", text.c_str(), blankCount);
        int sep = 0;
        int tail = 0;
        if (words.size() == 1) {
            sep = 0;
            tail = blankCount;
        } else {
            sep = blankCount / (words.size() - 1);
            tail = blankCount % (words.size() - 1);
        }
        // result
        string result = words[0];
        for (int i = 1; i < words.size(); i++) {
            result += string(sep, ' ');
            result += words[i];
        }
        result += string(tail, ' ');
        return result;
    }
};
