class Solution {
public:
    string reorderSpaces(string text) {
        int koge = 0;
        vector<string> words;
        string word;
        for (int i = 0; i < text.size(); ++i)
        {
            if (text[i] == ' ') 
            {
                koge++;
                if (word.size() > 0)
                {
                    words.push_back(word);
                    word = "";
                }
            }
            else
            {
                word += text[i];
            }
        }
        if (word.size() > 0)
        {
            words.push_back(word);
            word = "";
        }
        
        string ans;
        if (words.size() >= 2)
        {
            int x = koge / (words.size() - 1);
            
            for (int i = 0; i < words.size(); ++i)
            {
                ans += words[i];
                if (i != words.size() - 1)
                {
                    for (int j = 0; j < x; ++j)
                    {
                        ans += ' ';
                    }    
                }
            }
            for (int j = 0; j < koge % (words.size() - 1); ++j)
            {
                ans += ' ';
            }
        }
        else
        {
            ans = words[0];
            for (int j = 0; j < koge; ++j)
            {
                ans += ' ';
            }
        }
        
        return ans;
    }
};
