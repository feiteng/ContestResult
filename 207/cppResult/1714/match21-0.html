class Solution {
public:
    vector<string> ssplit(string s){
        vector<string> vs;
        string ss;
        for(int i = 0; i < s.size(); i++){
            ss = "";
            if(s[i] != ' '){
                while(s[i] != ' ' && i < s.size()){
                    ss = ss + s[i];
                    i++;
                }
                vs.push_back(ss);
            }
        }
        return vs;
    }
    
    
    string reorderSpaces(string text) {
        string res;
        int cnt = 0;
        for(int i = 0; i < text.size(); i++){
            if(text[i] == ' ')cnt++;
        }
        vector<string> vs;
        vs = ssplit(text);
        if(vs.size() == 1){
            res = vs[0];
            for(int i = 0; i < cnt; i++){
                res = res + ' ';
            }
            return res;
        }
        
        
        for(int i = 0; i < vs.size() - 1; i++){
            res = res + vs[i];
            for(int j = 0; j < cnt / (vs.size() - 1); j++){
                res = res + ' ';
            }
        }
        res = res + vs[vs.size() - 1];
        for(int j = 0; j < cnt % (vs.size() - 1); j++){
            res = res + ' ';
        }
        return res;
    }
};
