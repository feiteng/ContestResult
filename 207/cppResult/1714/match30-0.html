class Solution {
public:
    vector<string>s;
    string reorderSpaces(string t) {
        int num=0;
        string temp="";
        for(int i=0;i<t.length();i++) {
            if(t[i]==' ') {
                num++;
                if(temp.length()) {
                    s.push_back(temp);
                    temp="";
                }
            }
            else temp+=t[i];
        }
        if(temp.length()) s.push_back(temp);
        string ans="";
        int now=0;
        if(s.size()>1)now=num/((int)s.size()-1);
        for(int i=0;i<s.size();i++) {
            ans+=s[i];
            if(i!=s.size()-1){
            for(int j=0;j<now;j++) ans+=" ";
            num-=now;
            }
        }
        while(num--) ans+=" ";
        return ans;
    }
};
