class Solution {
public:
    vector<string> a;
    string reorderSpaces(string text) {
        a.clear();
        string s="";
        int cnt=0;
        for(int i=0;i<text.size();i++){
            if(text[i]==' '){
                if(s!="") a.push_back(s);
                s="";
                cnt++; 
            } 
            else{
                s+=text[i];
            }
        }
        if(s!="") a.push_back(s);
        
        

        int n=a.size();
        if(n==1){
            string ans=a[0];
            for(int i=1;i<=cnt;i++){
                ans+=" ";
            }
            return ans;
        }
        
        int m=cnt/(n-1);
        int k=cnt%(n-1);
        string ans;
        ans=a[0];
        for(int i=1;i<n;i++){
            for(int j=0;j<m;j++)
                ans+=" ";

            ans+=a[i];
        }
        for(int i=1;i<=k;i++){
            ans+=" ";
        }
        return ans;
    }
};
