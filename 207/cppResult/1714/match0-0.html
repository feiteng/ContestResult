class Solution {
public:
    string reorderSpaces(string text) {
        int num = 0;
        for (int i = 0; i < text.length(); ++i)
            if (text[i] == ' ')
                ++num;
        vector<string> tmp;
        stringstream ss(text);
        string x;
        while (ss >> x)
            tmp.push_back(x);
        int total = tmp.size() - 1;
        if (total == 0)
            return tmp[0] + string(num, ' ');
        string res = tmp[0];
        int spaces = num / total;
        for (int i = 1; i < tmp.size(); ++i)
            res += (string(spaces, ' ') + tmp[i]);
        res += string(num % total, ' ');
        
        return res;
    }
};
