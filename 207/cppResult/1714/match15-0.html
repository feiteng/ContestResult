class Solution {
public:
    string reorderSpaces(string text) {
        int N = text.size();
        int space = 0;
        vector<string> words;
        for (int i=0; i<N; i++) {
            if (text[i] == ' ') space++;
            else {
                int o = i;
                while(i<N and text[i] != ' ') i++;
                words.push_back(text.substr(o, i-o));
                i--;
            }
        }
        if (words.size() == 1) {
            string ans = words[0];
            for (int i=0; i<space; i++) ans += ' ';
            return ans;
        }
        int between = space/(words.size()-1);
        int end = space % (words.size()-1);
        string ans = words[0];
        for (int i=1; i<words.size(); i++) {
            for (int i=0; i<between; i++) ans += ' ';
            ans += words[i];
        }
        for (int i=0; i<end; i++) ans += ' ';
        return ans;
    }
};
