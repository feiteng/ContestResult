class Solution {
public:
    string reorderSpaces(string s) {
        int N = s.size(), cnt = 0;
        for (char c : s) {
            cnt += c == ' ';
        }
        string word;
        istringstream ss(s);
        vector<string> words;
        while (ss >> word)  {
            words.push_back(word);
        }
        string ans;

        if (words.size() == 1) {
            ans += words[0];
            for (int i = 0; i < cnt; ++i) ans += ' ';
            return ans;
        }
        int d = cnt / (words.size() - 1), r = cnt - d * (words.size() - 1);

        for (int i = 0; i < words.size(); ++i) {
            ans += words[i];
            if (i < words.size() - 1) {
                for (int k = 0; k < d; ++k) ans += ' ';
            } else {
                for (int k = 0; k < r; ++k) ans += ' ';
            }
        }
        return ans;
    }
};
