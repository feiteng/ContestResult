class Solution {
public:
    string reorderSpaces(string text) {
        vector<string> words;
        string w;
        int spaceCount = 0;
        for (int i = 0; i < text.size(); ++ i) {
            if (text[i] != ' ') {
                w += text[i];
            } else {
                ++ spaceCount;
                if (!w.empty()) {
                    words.push_back(w);
                    w = "";
                }
            }
        }

        if (!w.empty()) words.push_back(w);
        string ans;
        if (words.size() == 1) {
            ans = words[0];
            for (int i = 0; i < spaceCount; ++ i) ans += ' ';
        } else {
            ans = words[0];

            int betweenSize = spaceCount / (words.size() - 1);
            for (int i = 1; i < words.size(); ++ i) {
                for (int j = 0; j < betweenSize; ++ j) ans += ' ';
                ans += words[i];
            }
            for (int i = 0; i < spaceCount - betweenSize * (words.size() - 1); ++ i) ans += ' ';
        }
        return ans;
    }
};
