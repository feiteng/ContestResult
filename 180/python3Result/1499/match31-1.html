from heapq import heappop, heappush

class Solution:
    def maxPerformance(self, n, speed, efficiency, k):
        dat 
= sorted([(efficiency[i], speed[i]) for i in range(n)], reverse=True)
        egnr, opt, s = [], 0, 0
        for eff, sp in dat:
            if len(egnr) < k: 
                heappush(egnr, sp)
                s += sp
                opt = max(opt, eff * s)
            elif sp > egnr[0]:
                s += -heappop(egnr) + sp
                heappush(egnr, sp)
                opt = max(opt, eff * s)
        return opt % (10 ** 9 + 7)
