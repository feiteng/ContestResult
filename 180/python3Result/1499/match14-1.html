class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        en = [(efficiency[i], speed[i]) for i in range(n)]
        en.sort(key = lambda x: -x[0])
        print(en)
        h = []
        me = 10 ** 9
        ms = 0
        s = 0
        for ef, sp in en:
            me = min(me, ef)
            heapq
.heappush(h, sp)
            s += sp
            if len(h) > k:
                s -= heapq.heappop(h)
            
            ms = max(ms, s * me)
            
        return ms % 1000000007
