import heapq
import math

class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        mod = 10 ** 9 + 7
        eff_speed = sorted(zip(efficiency, speed), reverse=True)
        ans = total_speed = 0
        min_eff = math.inf
        # print(eff_speed)
        heap = []
        for eff, speed in eff_speed:
            min_eff = min(min_eff, eff)
            total_speed += speed
            heapq.heappush(heap, speed)
            if len(heap) > k:
                total_speed -= heapq.heappop(heap)
            ans = max(ans, total_speed * min_eff)
        return ans % mod
