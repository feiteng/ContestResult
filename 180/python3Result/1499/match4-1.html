class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        points = [(efficiency[i], speed[i]) for i in range(n)]
        points.sort(reverse=True)
        ans = 0
        vq = []
        speedsum = 0
        
        for eff, v in points:
            heapq
.heappush(vq, v)
            speedsum += v
            if len(vq) > k:
                speedsum -= heapq.heappop(vq)
                
            
            ans = max(ans, eff * speedsum)
        
        return ans % (10**9 + 7)
