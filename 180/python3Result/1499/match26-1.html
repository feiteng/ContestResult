class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        mod = 10 ** 9 + 7
        p = []
        for i in range(n):
            p.append((speed[i], efficiency[i]))

        p.sort(key=lambda x: -x[1])
        bsp = []
        aspeed = 0
        ma = 0
        for i, (s, e) in enumerate(p):
            bisect.insort(bsp, s)
            aspeed += s
            while len(bsp) > k:
                aspeed -= bsp.pop(0)
            ma = max(ma, aspeed*e)

        return ma % mod
