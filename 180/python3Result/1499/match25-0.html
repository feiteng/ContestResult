class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        tmp=sorted([[e,s] for s,e in zip(speed,efficiency)],reverse=True)
        que=[]
        res,cursum=0,0
        mod=10**9+7
        for e,s in tmp:
            cursum+=s
            heapq.heappush(que,s)
            if len(que)>k:
                s2=heapq.heappop(que)
                cursum-=s2
            res=max(res,cursum*e)
        return res%mod
