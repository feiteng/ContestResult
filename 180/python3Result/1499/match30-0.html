import heapq
class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        newlist = [(speed[i],  efficiency[i]) for i in range(n)]
        newlist.sort(reverse=True)
        total = 0
        people = []
        ans = 0
        for (spe, eff) in newlist:
            total += spe
            heapq.heappush(people, (eff, spe))
            if len(people) > k:
                (tmpeff, tmpspe) = heapq.heappop(people)
                total -= tmpspe
            sofar = total * people[0][0]
            ans = max(ans, sofar)
        while len(people) > 1:
            (tmpeff, tmpspe) = heapq.heappop(people)
            total -= tmpspe
            sofar = total * people[0][0]
            ans = max(ans,  sofar)
        return ans % 1000000007
