import heapq

class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        data = list(zip(efficiency, speed))
        data.sort(reverse=True)
        h = []
        tot, ans = 0, 0
        for e, s in data:
            heapq.heappush(h, s)
            tot += s
            if len(h) > k:
                tot -= heapq.heappop(h)
            ans = max(ans, tot * e)
        return ans % (10**9 + 7)
