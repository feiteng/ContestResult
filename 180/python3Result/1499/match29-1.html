import heapq
class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        t = list((efficiency[i],speed[i]) for i in range(n))
        t.sort(reverse = True)
        #print(t)
        h 
= []
        cur_speed,ans = 0,0
        for i in range(k):
            heapq.heappush(h,t[i][1])
            cur_speed += t[i][1]
            cur_efficiency = t[i][0]
            ans = max(ans,cur_speed*cur_efficiency)
        for i in range(k,n):
            popped = heapq.heapreplace(h,t[i][1])
            cur_speed = cur_speed+t[i][1] - popped
            cur_efficiency = t[i][0]
            ans = max(ans,cur_speed*cur_efficiency)
        return ans % (10**9 + 7)
