class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = 10**9 + 7
        people = [(efficiency[i], speed[i]) for i in range(n)]
        people.sort(key=lambda x:(-x[0]))
        ans = 0
        speed_sum = 0
        heap = []
        for i in range(n):
            heapq.heappush(heap, people[i][1])
            speed_sum += people[i][1]
            if i >= k:                
                speed_sum -= heap[0]
                heapq.heappop(heap)
            ans = max(ans, speed_sum*people[i][0])
        return ans % MOD
