import heapq


class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        engineers = list(zip(efficiency, speed))
        engineers = sorted(engineers, reverse=True)
        res = 0
        totalSpeed = 0
        heap = []
        for e, s in engineers:
            totalSpeed += s
            heapq.heappush(heap, s)
            if len(heap) > k:
                totalSpeed -= heapq.heappop(heap)
            t = e * totalSpeed
            if t > res:
                res = t
        return res % (10 ** 9 + 7)
