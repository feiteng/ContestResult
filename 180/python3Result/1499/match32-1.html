class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = ((10 ** 9) + 7)
        se = [(speed[i], efficiency[i]) for i in range(n)]
        se.sort(key=lambda x: -x[1])
        ss = 0
        sol = 0
        q = []
        for s, e in se:
            sol = max(sol, (ss+s) * e)
            if len(q) < k-1:
                heapq.heappush(q, s)
                ss += s
            elif q and q[0] < s:
                ss -= heapq.heappop(q)
                ss += s
                heapq.heappush(q, s)

        return sol % MOD
