class Solution:
    def maxPerformance(self, N: int, speed: List[int], efficiency: List[int], K: int) -> int:
        
        worker = sorted([ (-efficiency[i], speed[i]) for i in range(N) ])
        result 
= 0
        total_speed = 0
        heap = []
        for e, s in worker:
            total_speed += s
            heapq.heappush(heap, s)
            if len(heap) > K: total_speed -= heapq.heappop(heap)
            result = max(result, total_speed * (-e))
        # print(618114759971930%(10**9 + 7))
        return result % (10**9 + 7)
