class Solution:
    def maxPerformance(self, n: int, S: List[int], E: List[int], k: int) -> int:
        res, m, acc = float('-inf'), float('inf'), 0
        tmp = sorted((e, s) for e, s in zip(E, S))[::-1]
        MOD = 10**9 + 7
        q = []
        for e, s in tmp:
            m = min(m, e)
            if k <= len(q):
                acc -= heapq.heappop(q)
            heapq.heappush(q, s)
            acc += s
            res = max(res, acc * m)
        return res % MOD
