class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = 10 ** 9 + 7
        
        people = [[speed[i], efficiency[i]] for i in range(n)]
        
        people.sort(key = lambda x: -x[1])
        
        summ = 0
        pq = []
        res = 0
        for i in range(k):
            summ += people[i][0]
            res = max(res, summ * people[i][1])
            pq.append(people[i][0])
        
        heapq.heapify(pq)
            
        #res = summ * people[k-1][1]
        for i in range(k, n):
            summ += people[i][0]
            heapq.heappush(pq, people[i][0])
            summ -= heapq.heappop(pq)
            res = max(res, summ * people[i][1])
        
        return res % MOD
