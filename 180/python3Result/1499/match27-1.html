class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        a 
= []
        for i in range(n):
            a.append((speed[i], efficiency[i]))
        a.sort(key=lambda x: -x[1])
        result 
= 0
        ss = 0
        h = []
        for i in range(k):
            heapq.heappush(h, a[i][0])
            ss += a[i][0]
            if ss * a[i][1] > result:
                result = ss * a[i][1]
        for i in range(k, n):
            ss += a[i][0]
            s = heapq.heappushpop(h, a[i][0])
            ss -= s
            if ss * a[i][1] > result:
                result = ss * a[i][1]
        return result % (10 ** 9 + 7)
