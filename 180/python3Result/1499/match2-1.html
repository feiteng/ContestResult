class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        perf = sorted(zip(efficiency, speed), reverse=True)
        heap = []
        sumSpeed = 0
        result = float("-inf")
        
for e, s in perf:
            heapq.heappush(heap, s)
            sumSpeed += s
            if len(heap) > k:
                sumSpeed -= heapq.heappop(heap)
            result = max(result, sumSpeed * e)
        return result % (10 ** 9 + 7)
