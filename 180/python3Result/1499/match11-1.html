class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = 10 ** 9 + 7
        people = [(e, s) for s, e in zip(speed, efficiency)]
        people.sort(reverse=True)
        
        cur_speed = []
        max_perform = float("-inf")
        cur_sum 
= 0
        
        for e, s in people:
            heapq.heappush(cur_speed, s)
            cur_sum += s
            if len(cur_speed) > k:
                cur_sum -= heapq.heappop(cur_speed)
            max_perform = max(max_perform, cur_sum * e)
        
        return max_perform % MOD
