class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        import heapq
        if k == 0:
            return 0
        li = [[s,e] for s, e in zip(speed, efficiency)]
        res = 0
        s, cur = [], 0
        #print(sorted(list(li), key = lambda x: -x[1]))
        for i, j in  sorted(li, key = lambda x: -x[1]):
            heapq.heappush(s, i)
            cur += i
            if len(s) > k:
                cur -= heapq.heappop(s)
            #print(j)
            res = max(res, cur*j)
            #print(s, j)
            #print(i , j)
        return res%(10**9 + 7)
