class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = 1000000007
        A = []
        for i in range(n):
            A.append([speed[i], efficiency[i]])
        
        A.sort(key = lambda x : -x[1])

        B = []
        s = 0
        ret = 0
        for i in range(n):
            hpush(B, A[i][0])
            s += A[i][0]

            if len(B) > k :
                c = hpop(B)
                s -= c
            ret = max(ret, s * A[i][1])
        return ret % MOD
