class Solution:
    def maxPerformance(self, n: int, s: List[int], e: List[int], k: int) -> int:
        l = [[a,b] for a, b in zip(s, e)]
        l = sorted(l, key=lambda x :(-x[1], -x[0]))
        res, th, ans = float('-inf'), float('inf'), 0
        import heapq
        heap 
= []
        for a, b in l:
            th = min(b, th)
            if len(heap) >= k:
                ans -= heapq.heappop(heap)
            heapq.heappush(heap, a)
            ans += a
            res = max(res, ans * th)
        return res % (10**9 + 7)
