class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        mod = 10**9 + 7
        
        es = sorted(zip(efficiency, speed), reverse=True)
        # print(es)
        
        pq = []
        heapq.heapify(pq)
        
        res = 0
        heaps = 0
        for i in range(k):
            heapq.heappush(pq, es[i][1])
            heaps += es[i][1]
            cur_e = es[i][0]
            res = max(res, cur_e * heaps)
        
        for i in range(k, n):
            pop_v = heapq.heappop(pq)
            push_v = es[i][1]
            heapq.heappush(pq, push_v)
            heaps -= pop_v
            heaps += push_v
            cur_e = es[i][0]
            
            res = max(res, cur_e * heaps)
        
        return res % mod
