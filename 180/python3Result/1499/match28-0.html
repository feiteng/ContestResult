import heapq

class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        engs = []
        for i in range(n):
            engs.append([speed[i], efficiency[i]])
        
        engs.sort(key= lambda x: -x[1])
        
        
        speeds = []
        total = 0
        res = 0
        for i in range(n):
            s = engs[i][0]
            heappush(speeds, s)
            total += s
            min_eff = engs[i][1]
            if len(speeds) > k:
                min_s = heappop(speeds)
                total -= min_s
            res = max(res, total * min_eff)
            
        return res%(pow(10,9)+7)
