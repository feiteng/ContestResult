import heapq



class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        MOD = int(1e9) + 7
        ls = len(speed)
        z = []
        for i in range(ls):
            z.append((efficiency[i], speed[i]))
        z.sort()
        y = [0] * ls
        h = []
        sm = 0
        best = 0
        # print(z)
        for i in range(ls-1, -1, -1):
            heapq.heappush(h, z[i][1])
            sm += z[i][1]
            if len(h) > k:
                x = heapq.heappop(h)
                # print(x)
                sm -= x
            y[i] = sm
            r = y[i] * z[i][0]
            
            best = max(best, r)
            # print(sm, r, best)
        return best % MOD
