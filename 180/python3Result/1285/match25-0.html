# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        nodes = []
        
        def dfs(node):
            if not node:
                return
            dfs(node.left)
            nodes.append(node.val)
            dfs(node.right)
    
        dfs(root)
        
        def make_tree(start, end):
            if start == end:
                return None
            mid = (start + end) // 2
            node = TreeNode(nodes[mid])
            node.left = make_tree(start, mid)
            node.right = make_tree(mid + 1, end)
            return node
        
        return make_tree(0, len(nodes))
