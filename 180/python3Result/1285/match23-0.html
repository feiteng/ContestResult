# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        data = []
        def dfs1(u):
            if u is None:
                return
            dfs1(u.left)
            data.append(u.val)
            dfs1(u.right)
        
        def dfs2(data):
            if not data:
                return None
            mid = len(data) // 2
            node = TreeNode(data[mid])
            node.left = dfs2(data[:mid])
            node.right = dfs2(data[mid + 1:])
            return node

        dfs1(root)
        return dfs2(data)
