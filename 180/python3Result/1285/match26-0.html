# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        nodes = []
        def dfs(root):
            if root:
                dfs(root.left)
                nodes.append(root.val)
                dfs(root.right)
        dfs(root)
        def buildTree(le, ri):
            if le>ri:
                return None
            else:
                mid = (le + ri) // 2
                root = TreeNode(nodes[mid])
                root.left = buildTree(le, mid-1)
                root.right = buildTree(mid+1, ri)
                return root
        return buildTree(0, len(nodes)-1)
