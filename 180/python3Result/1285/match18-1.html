# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        inorder = []
        def dfs(node):
            if node is None:
                return
            dfs(node.left)
            inorder.append(node.val)
            dfs(node.right)
        dfs(root)
        def balTree(i, j):
            if i == j:
                return None
            k = (i+j)//2
            val = inorder[k]
            node = TreeNode(val)
            node
.left = balTree(i,k)
            node.right = balTree(k+1, j)
            return node
        return balTree(0, len(inorder))
