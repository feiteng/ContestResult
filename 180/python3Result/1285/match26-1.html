# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        
        arr = []
        
        def preorder(cur):
            nonlocal arr
            if cur:
                preorder(cur.left)
                arr.append(cur.val)
                preorder(cur.right)
        
        preorder(root)
        
        def buildtree(startIdx, endIdx):
            nonlocal arr
            if (startIdx > endIdx):
                return None
            elif startIdx == endIdx:
                return TreeNode(arr[startIdx])
            else:
                midIdx 
= (startIdx + endIdx) // 2
                t = TreeNode(arr[midIdx])
                t.left = buildtree(startIdx, midIdx - 1)
                t.right = buildtree(midIdx + 1, endIdx)
                return t
        
        return buildtree(0, len(arr) - 1)
