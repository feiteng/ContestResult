# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        l=[]
        def dfs(node):
            if not node:return
            dfs(node.left)
            l.append(node.val)
            dfs(node.right)
        dfs(root)
        #("l:%s"%l)
        def dfs2(nodes, begin, size):
            #print("dfs2 %d,%d"%(begin,size))
            if not size: return None
            if size==1:
                #print("node %d"%(nodes[begin]))
                return TreeNode(nodes[begin])
            m=begin+size//2
            node=TreeNode(nodes[m])
#print("node:%d"%(node.val))
            node.left=dfs2(nodes,begin,m-begin)
            node.right=dfs2(nodes,m+1,begin+size-m-1)
            #print("return %d,left:%s,right:%s"%(node.val, node.left.val if node.left else "None", node.right.val if node.right else "None"))
            return node
        return dfs2(l,0,len(l))
