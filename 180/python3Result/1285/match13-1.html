# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        res = []
        def dfs(root):
            if not root:
                return
            dfs(root.left)
            res.append(root)
            dfs(root.right)
        dfs(root)
        # print([a.val for a in res])
        def build(cur):
            if not cur:
                return
            k = len(cur) // 2
            root = cur[k]
            root.left = build(cur[:k])
            root.right = build(cur[k+1:])
            return root
        return build(res)
