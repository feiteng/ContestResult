# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        #with same node value = traverse and get the array and construct the balanced tree
        array = []
        def inorder(node):
            if not node:
                return
            inorder(node.left)
            array.append(node.val)
            inorder(node.right)
            
        def construct(left, right):
            if left > right:
                return None
            if left == right:
                return TreeNode(array[left])
            mid = (left+right) // 2
            curr = TreeNode(array[mid])
            curr.left = construct(left, mid-1)
            curr.right = construct(mid+1, right)
            return curr
        
        inorder(root)
        return construct(0, len(array)-1)
