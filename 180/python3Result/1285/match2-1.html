# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        arr = []
        def add(node):
            if not node: return 
            add(node.left)
            arr.append(node.val)
            add(node.right)
        add(root)
        
        def build(l, r):
            if l == r:
                return TreeNode(arr[l])
            
if l > r:
                return None
            m = (l+r) >> 1
            node = TreeNode(arr[m])
            node.left = build(l, m-1)
            node.right = build(m+1, r)
            return node
        return build(0, len(arr)-1)
