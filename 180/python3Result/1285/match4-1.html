# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        sarr = []
        def inorder(node):
            if not node:
                return
            inorder(node.left)
            sarr.append(node.val)
            inorder(node.right)
        inorder(root)
        # print(sarr)
        def build(i, j):
            if i > j:
                return None
            m 
= (i + j) // 2
            node = TreeNode(sarr[m])
            node.left = build(i, m - 1)
            node.right = build(m + 1, j)
            return node
        return build(0, len(sarr) - 1)
