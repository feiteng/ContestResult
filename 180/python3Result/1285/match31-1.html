# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        
        def get_seq(root):
            if not root:
                return []
            return get_seq(root.left) + [root.val] + get_seq(root.right)
        
        def build(seq, i, j):
            if i == j:
                return TreeNode(seq[i])
            if i + 1 == j:
                root = TreeNode(seq[i])
                root.right = TreeNode(seq[j])
                
return root
            mid = (i + j) // 2
            root = TreeNode(seq[mid])
            root.left = build(seq, i, mid - 1)
            root.right = build(seq, mid + 1, j)
            return root
            
        
        seq = get_seq(root)
        return build(seq, 0, len(seq) - 1)
