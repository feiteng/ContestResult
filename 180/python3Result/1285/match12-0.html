# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        arr = []
        def flat(root):
            if not root: return
            flat(root.left)
            arr.append(root.val)
            flat(root.right)
        flat(root)
        
        def build(l, r):
            if l > r: return None
            if l == r: return TreeNode(arr[l])
            m = (l + r) // 2
            ret = TreeNode(arr[m])
            ret.left = build(l, m - 1)
            ret.right = build(m + 1, r)
            return ret
        
        return build(0, len(arr) - 1)
