# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        stack = []
        def tostack(root):
            if not root: return
            tostack(root.left)
            stack.append(root)
            tostack(root.right)
            
        tostack(root)
        
        def totree(stack):
            if not stack:
                return None
            index = len(stack)//2
            mid = stack[index]
            mid.left = totree(stack[:index])
            mid.right = totree(stack[index+1:])
            return mid
        
        return totree(stack)
