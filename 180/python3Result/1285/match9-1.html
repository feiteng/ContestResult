# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        # fake
        vals = []
        def collect(root):
            if not root: return
            collect(root.left)
            vals.append(root.val)
            collect(root.right)
            
        collect(root)
        if not vals: return None
        
        
def build(i, j):
            if i > j: return None
            if i == j: return TreeNode(vals[i])
            mid = (i + j) // 2
            cur = TreeNode(vals[mid])
            cur.left = build(i, mid - 1)
            cur.right = build(mid + 1, j)
            return cur
        
        return build(0, len(vals) - 1)
