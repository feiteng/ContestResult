class Solution:
    def balanceBST(self, root):
        def combine(node):
            if node is None:
                return []
            return combine(node.left) + [node.val] + combine(node.right)

        def resolve(arr, lo, hi):
            if lo > hi:
                return None

            m = (hi + lo) >> 1
            r = TreeNode(arr[m])
            r.left = resolve(arr, lo, m - 1)
            r.right = resolve(arr, m + 1, hi)
            return r

        a = combine(root)
        return resolve(a, 0, len(a) - 1)
