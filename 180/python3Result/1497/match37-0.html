class CustomStack:

    def __init__(self, maxSize: int):
        
        self.L=[]
        self.maxSize=maxSize
    def push(self, x: int) -> None:
        
        if len(self.L)<self.maxSize:
            self.L.append(x)
        

    def pop(self) -> int:
        
        if len(self.L)==0:
            return -1
        
        return self.L.pop()

    def increment(self, k: int, val: int) -> None:
        
        for i in range(min(len(self.L),k)):
            self.L[i]+=val
