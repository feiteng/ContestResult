class CustomStack:

    def __init__(self, maxSize: int):
        self.maxSize = maxSize
        self.li = []

    def push(self, x: int) -> None:
        if len(self.li) < self.maxSize:
            self.li.append(x)
        

    def pop(self) -> int:
        if len(self.li) == 0:
            return -1
        return self.li.pop()

    def increment(self, k: int, val: int) -> None:
        for i in range(min(len(self.li), k)):
            self.li[i] += val
