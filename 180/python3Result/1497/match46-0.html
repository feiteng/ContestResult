class CustomStack:

    def __init__(self, maxSize: int):
        self.max=maxSize
        self.data=[]

    def push(self, x: int) -> None:
        if len(self.data)<self.max:
            self.data.append(x)

    def pop(self) -> int:
        if len(self.data)==0:
            return -1
        else:
            r=self.data.pop()
            return  r

    def increment(self, k: int, val: int) -> None:
        for i in range(min(k,len(self.data))):
            self.data[i]+=val
