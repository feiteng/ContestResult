class CustomStack:

    def __init__(self, maxSize: int):
        self.data = []
        self.max_size = maxSize

    def push(self, x: int) -> None:
        if len(self.data) < self.max_size:
            self.data.append(x)

    def pop(self) -> int:
        if not self.data:
            return -1
        return self.data.pop()

    def increment(self, k: int, val: int) -> None:
        for i in range(min(k, len(self.data))):
            self.data[i] += val
