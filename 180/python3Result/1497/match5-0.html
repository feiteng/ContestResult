class CustomStack:

    def __init__(self, maxSize: int):
        self.st = []
        self.maxsize = maxSize 

    def push(self, x: int) -> None:
        if len(self.st) < self.maxsize:
            self.st.append(x)

    def pop(self) -> int:
        if len(self.st) == 0:
            return -1
        else:
            return self.st.pop()

    def increment(self, k: int, val: int) -> None:
        for i in range(min(len(self.st), k)):
            self.st[i] += val
