class CustomStack:
    def __init__(self, maxSize: int):
        self.a = []
        self.maxSize = maxSize

    def push(self, x: int) -> None:
        if len(self.a) < self.maxSize:
            self.a.append(x)

    def pop(self) -> int:
        if self.a:
            return self.a.pop()
        else:
            return -1

    def increment(self, k: int, val: int) -> None:
        for i in range(min(k, len(self.a))):
            self.a[i] += val
