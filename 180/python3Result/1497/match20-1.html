class CustomStack:

    def __init__(self, maxSize: int):
        self.A = []
        self.ms = maxSize
        

    def push(self, x: int) -> None:
        if len(self.A) < self.ms:
            self.A.append(x)

    def pop(self) -> int:
        if self.A:
            return self.A.pop()
        return -1

    def increment(self, k: int, val: int) -> None:
        for i in range(min(k, len(self.A))):
            self.A[i] += val
