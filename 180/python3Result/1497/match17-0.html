class CustomStack:

    def __init__(self, maxSize: int):
        self.maxsize = maxSize
        self.stk = []

    def push(self, x: int) -> None:
        if len(self.stk) < self.maxsize:
            self.stk.append(x)

    def pop(self) -> int:
        if len(self.stk) == 0:
            return -1
        return self.stk.pop()

    def increment(self, k: int, val: int) -> None:
        for i in range(min(k, len(self.stk))):
            self.stk[i] += val
