class CustomStack:

    def __init__(self, maxSize: int):
        self.m = maxSize
        self.s = []

    def push(self, x: int) -> None:
        if len(self.s) != self.m:
            self.s.append(x)


    def pop(self) -> int:
        if not self.s:
            return -1
        return self.s.pop()


    def increment(self, k: int, val: int) -> None:
        for i in range(0, min(len(self.s), k)):
            self.s[i] += val
