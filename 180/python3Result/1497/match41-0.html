class CustomStack:

    def __init__(self, maxSize: int):
        self.max_size = maxSize
        self.st = []
        

    def push(self, x: int) -> None:
        
        if len(self.st) < self.max_size:
            self.st.append(x)

    def pop(self) -> int:
        if self.st:
            return self.st.pop()            
        else:
            return -1
        

    def increment(self, k: int, val: int) -> None:
        
        for i in range(min(k,len(self.st))):
            self.st[i] += val
