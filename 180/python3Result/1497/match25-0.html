class CustomStack:

    def __init__(self, maxSize: int):
        self.maxSize = maxSize
        self.ss = []
    def push(self, x: int) -> None:
        if len(self.ss) < self.maxSize:
            self.ss.append(x)
    def pop(self) -> int:
        if self.ss:
            return self.ss.pop()
        return -1
    def increment(self, k: int, val: int) -> None:
        for i in range(min(k, len(self.ss))):
            self.ss[i] += val
