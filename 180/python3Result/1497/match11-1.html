class CustomStack:

    def __init__(self, maxSize: int):
        self.a=[]
        self.n=maxSize

    def push(self, x: int) -> None:
        if len(self.a)<self.n:
            self.a.append(x)

    def pop(self) -> int:
        if len(self.a)>0:
            return self.a.pop()
        else:
            return -1

    def increment(self, k: int, val: int) -> None:
        for i in range(min(len(self.a),k)):
            self.a[i]+=val
