class Solution:
    def luckyNumbers (self, mat: List[List[int]]) -> List[int]:
        n, m, ans = len(mat), len(mat[0]), []
        rmn, cmx = [10**5+1 for i in range(n)], [-10**5+1 for i in range(m)]
        for i in range(n):
            for j in range(m):
                rmn[i] = min(rmn[i], mat[i][j])
                cmx[j] = max(cmx[j], mat[i][j])
        for i in range(n):
            for j in range(m):
                if mat[i][j] == rmn[i] and mat[i][j] == cmx[j]:
                    ans.append(mat[i][j])
        return ans
