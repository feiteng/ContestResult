class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        import numpy as np
        matrix = np.asarray(matrix)
        m, n = matrix.shape
       
        row_min = [0 for i in range(m)]
        col_max = [0 for j in range(n)]
       
        for i in range(m):
            row_min[i] = np.argmin(matrix[i, :])
       
        for j in range(n):
            col_max[j] = np.argmax(matrix[:, j])
           
        res = []
       
        if m <= n:
            for i in range(m):
                if col_max[row_min[i]] == i:
                    res.append(matrix[i, row_min[i]])
       
        else:
            for j in range(n):
                if row_min[col_max[j]] == j:
                    res.append(matrix[col_max[j], j])
       
        return res
