class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        n=len(matrix)
        m=len
(matrix[0])
        r=[]
        for i in range(n):
            for j in range(m):
                if matrix[i][j]==min(matrix[i]) and matrix[i][j]==max(matrix[k][j] for k in range(n)):
                    r.append(matrix[i][j])
        return r
