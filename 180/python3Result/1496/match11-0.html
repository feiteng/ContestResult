class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        R, C = len(matrix), len(matrix[0])
        row_min = [min(matrix[r][c] for c in range(C)) for r in range(R)]
        col_max = [max(matrix[r][c] for r in range(R)) for c in range(C)]
        res = []
        for r in range(R):
            for c in range(C):
                if matrix[r][c] == row_min[r] and matrix[r][c] == col_max[c]:
                    res.append(matrix[r][c])
        return res
