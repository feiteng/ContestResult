class Solution:
    def luckyNumbers(self, mat):
        if not mat: return []
        res = []
        for i in range(len(mat)):
            j = mat[i].index(min(mat[i]))
            if mat[i][j] == max(mat[k][j] for k in range(len(mat))):
                res.append(mat[i][j])
        return res
