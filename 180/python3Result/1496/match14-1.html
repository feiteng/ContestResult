class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        ret 
= []
        for i in range(len(matrix)):
            j = matrix[i].index(min(matrix[i]))
            if matrix[i][j] == max([matrix[ii][j] for ii in range(len(matrix))]):
                ret.append(matrix[i][j])
        return ret;
