class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        cols = zip(*matrix)
        col_max = [max(col) for col in cols]
        row_min = [min
(row) for row in matrix]
        ans = []
        for r in range(len(matrix)):
            for l in range(len(matrix[r])):
                if matrix[r][l] == row_min[r] == col_max[l]:
                    ans.append(matrix[r][l])
        return ans
