class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m = len(matrix)
        n = len(matrix[0])
        rmin = [matrix[r][0]  for r in range(m)]
        cmin = [matrix[0][c]  for c in range(n)]
        ret = []
        
for r in range(m):
            for c in range(n):
                rmin[r] = min(rmin[r], matrix[r][c])
                cmin[c] = max(cmin[c], matrix[r][c])
                
        for r in range(m):
            for c in range(n):
                if matrix[r][c] == rmin[r] and matrix[r][c] == cmin[c]:
                    ret.append(matrix[r][c])
        return ret
