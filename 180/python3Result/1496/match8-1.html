class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m = len(matrix)
        n = len(matrix[0])
        row = [pow(10,6) for i in range(m)]
        col = [0 for i in range
(n)]
        for i in range(m):
            for j in range(n):
                row[i] = min(row[i], matrix[i][j])
                col[j] = max(col[j], matrix[i][j])
        res = []
        for i in range(m):
            for j in range(n):
                if row[i] == col[j]:
                    res.append(row[i])
        return res
