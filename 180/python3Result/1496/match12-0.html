class Solution(object):
    def luckyNumbers (self, matrix):
        h = len(matrix)
        w = len(matrix[0])
        min_row = [min(matrix[row]) for row in range(h)]
        max_col = [max([matrix[row][col] for row in range(h)]) for col in range(w)]
        res = []
        for row in range(h):
            for col in range(w):
                if (matrix[row][col] == min_row[row] and matrix[row][col] == max_col[col]):
                    res.append(matrix[row][col])
        return res
