class Solution:
    def luckyNumbers (self, mat: List[List[int]]) -> List[int]:
        m, n = len(mat), len(mat[0])
        mini, maxi = [1000000] * m, [0] * n
        for i in range(m):
            for j in range(n):
                mini[i] = min(mini[i], mat[i][j])
                maxi[j] = max(maxi[j], mat[i][j])
        ret = []
        for i in range(m):
            for j in range(n):
                if mat[i][j] == mini[i] and mat[i][j] == maxi[j]:
                    ret.append(mat[i][j])
        return ret
