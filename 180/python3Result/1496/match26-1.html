class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        min_ = [min
(x) for x in matrix]
        max_ = [max([matrix[i][j] for i in range(m)]) for j in range(n)]
        res = []
        for i in range(m):
            for j in range(n):
                if matrix[i][j] == min_[i] and matrix[i][j] == max_[j]:
                    res.append(matrix[i][j])
        return res
