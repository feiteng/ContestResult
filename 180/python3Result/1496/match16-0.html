class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m = len(matrix)
        n = len(matrix[0])
        a = matrix
        
        minrow = [10**10 for _ in range(m)]
        maxcol = [-10**10 for _ in range(n)]
        for i in range(m):
            for j in range(n):
                minrow[i] = min(a[i][j], minrow[i])
                maxcol[j] = max(a[i][j], maxcol[j])
             
        res = []
        for i in range(m):
            for j in range(n):
                if a[i][j] == minrow[i] and a[i][j] == maxcol[j]:
                    res.append(a[i][j])
                
        return res
