class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m = len(matrix)
        n = len(matrix[0])
        min_r = [1e6] * m
        max_c = [0] * n
        
for i in range(m):
            for j in range(n):
                min_r[i] = min(min_r[i], matrix[i][j])
                max_c[j] = max(max_c[j], matrix[i][j])
        r = []
        for i in range(m):
            for j in range(n):
                if matrix[i][j] == min_r[i] and min_r[i] == max_c[j]:
                    r.append(matrix[i][j])
        return r
