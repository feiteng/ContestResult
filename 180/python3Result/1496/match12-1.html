class Solution:
    def luckyNumbers (self, a: List[List[int]]) -> List[int]:
        n, m = len(a), len
(a[0])
        r = [min(a[i]) for i in range(n)]
        c = [max(a[i][j] for i in range(n)) for j in range(m)]
        ans = []
        for i in range(n):
            for j in range(m):
                if a[i][j] == r[i] and a[i][j] == c[j]:
                    ans.append(a[i][j])
        return ans
