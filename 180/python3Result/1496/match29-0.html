class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        R, C = len(matrix), len(matrix[0])
        rowmin, colmax = [sys.maxsize] * R, [-sys.maxsize] * C
        for r in range(R):
            for c in range(C):
                rowmin[r] = min(rowmin[r], matrix[r][c])
                colmax[c] = max(colmax[c], matrix[r][c])
        
        res = []
        for r in range(R):
            for c in range(C):
                if matrix[r][c] == rowmin[r] and matrix[r][c] == colmax[c]:
                    res.append(matrix[r][c])
        
        return res
