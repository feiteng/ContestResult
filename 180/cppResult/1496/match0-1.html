class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        vector<int> mx; mx.resize(m); fill(mx.begin(), mx.end(), 0);
        vector<int> mn; mn.resize(n); fill(mn.begin(), mn.end(), 1000000);
        for(int i=0; i<n; ++i) {
            for(int j=0; j<m; ++j) {
                mx[j] = max(mx[j], matrix[i][j]);
                mn[i] = min(mn[i], matrix[i][j]);
            }
        }
        vector<int> ans;
        for(int i=0; i<n; ++i) {
            for(int j=0; j<m; ++j) {
                if(matrix[i][j]==mn[i]&&matrix[i][j]==mx[j]) ans.push_back(matrix[i][j]);
                // printf("%d %d %d\n", matrix[i][j], mn[i], mx[j]);
            }
        }
        // for(int i=0; i<n; ++i) printf("%d ", mn[i]); puts("");
        // for(int i=0; i<m; ++i) printf("%d ", mx[i]); puts("");
        return ans;
    }
};
