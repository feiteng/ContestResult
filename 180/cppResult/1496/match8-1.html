class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<int> mi(m, INT_MAX);
        vector<int> ma(n, INT_MIN);
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                mi[i] = min(mi[i], matrix[i][j]);
                ma[j] = max(ma[j], matrix[i][j]);
            }
        }
        
        vector<int> res;

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (matrix[i][j] == mi[i] && matrix[i][j] == ma[j]) res.push_back(matrix[i][j]);
            }
        }
        
        return res;
    }
};
