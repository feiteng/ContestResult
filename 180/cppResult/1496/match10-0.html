class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& A) {
        int n = A.size();
        int m = A[0].size();
        vector<int> row(n, 1e9), col(m, -1e9);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                row[i] = min(row[i], A[i][j]);
                col[j] = max(col[j], A[i][j]);
            }
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (A[i][j] == row[i] && A[i][j] == col[j]) {
                    ans.push_back(A[i][j]);
                }
            }
        }
        return ans;
    }
};
