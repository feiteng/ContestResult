class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        int R = matrix.size(), C = matrix[0].size();
        vector<int> rowMins(R), colMaxs(C);
        for(int i=0; i<R; i++){
            int v = matrix[i][0];
            for(int j=0; j<C; j++){
                v = min(v, matrix[i][j]);
            }
            
            rowMins[i] = v;
        }
        
        for(int j=0; j<C; j++){
            int v = matrix[0][j];
            for(int i=0; i<R; i++){
                v = max(v, matrix[i][j]);
            }
            
            colMaxs[j] = v;
        }
        
        vector<int> res;
        for(int i=0; i<R; i++){
            for(int j=0; j<C; j++){
                int v = matrix[i][j];
                if(v == rowMins[i] && v == colMaxs[j]){
                    res.push_back(v);
                }
            }
        }
        
        return res;
    }
};
