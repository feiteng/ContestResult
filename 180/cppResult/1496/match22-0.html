class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& v) {
        int n = v.size(),m = v[0].size();
        vector<int>a;
        for(int i = 0;i < n;i++)
            for(int j = 0;j < m;j++){
                bool f = 1;
                for(int x = 0;x < n;x++)
                    if(v[x][j] > v[i][j]) f = 0;
                for(int y = 0;y < m;y++)
                    if(v[i][y] < v[i][j]) f = 0;
                if(f) a.push_back(v[i][j]);
            }
        return a;
    }
};
