class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& a) {
        vector <int> ret;
        int n = a.size();
        int m = a[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                bool ok = true;
                for (int k = 0; k < n; k++) {
                    if (a[k][j] > a[i][j]) {
                        ok = false;
                        break;
                    }
                }
                for (int k = 0; k < m; k++) {
                    if (a[i][k] < a[i][j]) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    ret.push_back(a[i][j]);
                }
            }
        }
        return ret;
    }
};
