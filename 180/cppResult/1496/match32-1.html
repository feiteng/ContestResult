class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        vector<int> minrow(m, 1e9), maxcol(n, 0);
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                minrow[i] = min(minrow[i], mat[i][j]);

                maxcol[j] = max(maxcol[j], mat[i][j]);
            }
        }
        vector<int> ans;
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(mat[i][j] == minrow[i] && minrow[i] == maxcol[j]) {
                    ans.push_back(mat[i][j]);
                }
            }
        }
        return ans;
    }
};
