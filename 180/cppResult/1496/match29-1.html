class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        vector<int> res;
        int n = matrix.size();
        int m = matrix[0].size();
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                
                int x = i;
                int y = j;
                bool ok = true;

                for(int k=0;k<n;++k){
                    if(matrix[x][y]<matrix[k][y])ok = false;
                }
                for(int k=0;k<m;++k){
                    if(matrix[x][y]>matrix[x][k])ok = false;
                }
                if(ok) res.push_back(matrix[x][y]);
            }
        }
        return res;
    }
};
