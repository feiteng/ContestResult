class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& mtx) {
        int m = mtx.size();
        int n = mtx[0].size();
        vector<int> cols(n, INT_MIN);
        vector<int> rows(m, INT_MAX);
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                cols[j] = max(mtx[i][j], cols[j]);
                rows[i] = min(mtx[i][j], rows[i]);
            }
        }
        vector<int> anse;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(mtx[i][j] == cols[j] && mtx[i][j] == rows[i]){
                    anse.emplace_back(mtx[i][j]);
                }
            }
        }
        return anse;
    }
};
