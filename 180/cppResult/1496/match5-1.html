class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& ma) {
        vector<int> res;
        int n=ma.size();
        int m=ma[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                bool f=true;
                for(int ii=0;ii<n;ii++){
                    if(ma[i][j]<ma[ii][j]){
                        f=false;
                        break;
                    }
                }
                for(int jj=0;jj<m;jj++){
                    if(ma[i][j]>ma[i][jj]){
                        f=false;
                        break;
                    }
                }
                if(f==true){
                    res.push_back(ma[i][j]);
                }
            }
        }
        return res;
    }
};
