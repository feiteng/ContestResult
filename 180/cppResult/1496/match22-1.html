class Solution {
public:
    vector<int> luckyNumbers(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector<int> ans;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {
                bool ok = true;
                for (int k = 0; k < m; ++k)
                    if (!(a[i][k] >= a[i][j])) ok = false;
                for (int k = 0; k < n; ++k)
                    if (!(a[k][j] <= a[i][j])) ok = false;
                if (ok) ans.push_back(a[i][j]);
            }
        return ans;
    }
};
