class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& mt) {
        int m = mt.size();
        int n = mt[0].size();
        vector<int> cols(n, INT_MIN);
        vector<int> rows(m, INT_MAX);
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                cols[j] = max(mt[i][j], cols[j]);
                rows[i] = min(mt[i][j], rows[i]);
            }
        }
        vector<int> ans;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(mt[i][j] == cols[j] && mt[i][j] == rows[i]){
                    ans.emplace_back(mt[i][j]);
                }
            }
        }
        return ans;
    }
};
