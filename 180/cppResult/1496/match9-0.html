class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        vector<int> ans;
        for(int i=0;i<matrix.size();++i){
            for(int j=0;j<matrix[i].size();++j){
                int Max=0,Min=INT_MAX;
                for(int k=0;k<matrix.size();++k)Max=max(Max,matrix[k][j]);
                for(int k=0;k<matrix[i].size();++k)Min=min(Min,matrix[i][k]);
                if(matrix[i][j]==Max&&matrix[i][j]==Min)ans.push_back(matrix[i][j]);
            }
        }
        return ans;
    }
};
