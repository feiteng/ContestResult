class Solution {
public:
	vector<int> luckyNumbers(vector<vector<int>>& matrix) {
		int n = matrix.size(), m = matrix[0].size();
		vector<int>a(n, 1e6),b(m,0);
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				a[i] = min(a[i], matrix[i][j]);
				b[j] = max(b[j], matrix[i][j]);
			}
		}
		vector<int>ret;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (matrix[i][j] == a[i] && matrix[i][j] == b[j])ret.push_back(matrix[i][j]);
			}
		}
		return ret;
	}
};
