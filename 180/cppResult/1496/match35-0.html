class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        int r = matrix.size();
        int c = matrix.back().size();
        
        vector<int> row(r, INT_MAX), col(c, INT_MIN);
        for (int i = 0; i < r; ++i) {
            for (int j = 0; j < c; ++j) {
                row[i] = min(row[i], matrix[i][j]);
            }
        }
        
        for (int i = 0; i < c; ++i) {
            for (int j = 0; j < r; ++j) {
                col[i] = max(col[i], matrix[j][i]);
            }
        }
        
        vector<int> res;
        for (int i = 0; i < r; ++i) {
            for (int j = 0; j < c; ++j) {
                if (matrix[i][j] == row[i] && matrix[i][j] == col[j]) {
                    res.push_back(matrix[i][j]);
                }
            }
        }
        
        return res;
    }
};
