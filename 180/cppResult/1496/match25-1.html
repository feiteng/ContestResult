class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& m) {
        int N=m.size();
        int M=m[0].size();
        vector<int> res;
        for(int a=0;a<N;a++){
            for(int b=0;b<M;b++){

                int good=1;
                for(int i=0;i<N;i++){
                    if(m[i][b]>m[a][b]){
                        good=0;
                        break;
                    }
                }
                for(int j=0;j<M;j++){
                    if(m[a][j]<m[a][b]){
                        good=0;
                        break;
                    }
                }
                if(good) res.push_back(m[a][b]);
            }
        }
        return res;
    }
};
