class CustomStack {
public:
    int size;
    vector<int> data;
    CustomStack(int maxSize) {
        size = maxSize;
        data.clear();
    }
    
    void push(int x) {
        if (data.size() < size) {
            data.push_back(x);
        }
    }
    
    int pop() {
        if (data.size() == 0) {
            return -1;
        }

        int x = data[data.size() - 1];
        data.pop_back();
        return x;
    }
    
    void increment(int k, int val) {
        for (int i = 0;i < k && i < data.size();i++) {
            data[i] += val;
        }
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
