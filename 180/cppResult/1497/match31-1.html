class CustomStack {
public:
    vector<int> vs;
    int size;
    CustomStack(int maxSize) {
        size = maxSize;
        vs.clear();
    }
    
    void push(int x) {
        if(vs.size() == size) return;
        vs.push_back(x);
    }
    
    int pop() {
        if(vs.size() == 0) return -1;
        int a = *vs.rbegin();
        vs.pop_back();
        return a;
    }
    
    void increment(int k, int val) {
        for(int i=0; i<k && i<vs.size(); i++){
            vs[i] += val;
        }
    }
};
