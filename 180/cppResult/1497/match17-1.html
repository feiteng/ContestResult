class CustomStack {
public:
    int cap = 0;
    vector<int> stk;
    int count = 0;
    CustomStack(int maxSize) {
        cap = maxSize;
        return;
    }
    
    void push(int x) {
        //cout << "push" << endl;
        if (count < cap) { 
            stk.push_back(x);
            count++;
        }
        return;
    }
    
    int pop() {
        if (stk.size() == 0) { 
            return -1;
        }
        //cout << "pop" << endl;
        int ans = stk[stk.size() - 1];
        stk.pop_back();
        count--;
        return ans;
    }
    
    void increment(int k, int val) {
        //cout << "increment" << endl;
        for (int i = 0; i < min(k, count); i++) { 
            //cout << "i is " << i << endl;
            stk[i] += val;
        }
        return;
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
