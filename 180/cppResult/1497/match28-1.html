class CustomStack {
public:
    CustomStack(int maxSize) {
        num.resize(maxSize);
        maxNum = maxSize;
        cur = 0;
    }
    
    void push(int x) {
        // cout << "push: " << x << ", cur: " << cur << endl;
        if (cur < maxNum) {
            num[cur] = x;
            ++cur;
        }
    }
    
    int pop() {
        // cout << "pop: " << ", cur: " << cur << endl;
        if (cur > 0) {
            --cur;
            return num[cur];
        }
        return -1;

    }
    
    void increment(int k, int val) {
        for (int i = min(k - 1, cur - 1); i >= 0; --i) {
            num[i] += val;
        }
    }
    
    vector<int> num;
    int maxNum;
    int cur;
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
