class CustomStack {
    vector<int> st;
    int ms;
public:
    CustomStack(int maxSize) {
        ms = maxSize;
    }
    
    void push(int x) {
        if (st.size() < ms)
            st.push_back(x);

    }
    
    int pop() {
        if (st.empty())
            return -1;
        int v = st.back();
        st.pop_back();
        return v;
    }
    
    void increment(int k, int val) {
        for (int i = 0; i < min((int)st.size(), k); i++)
            st[i] += val;
    }
};

 * CustomStack* obj = new CustomStack(maxSize);

 * obj->increment(k,val);
