class CustomStack {
public:
    int maxSize;
    int count = 0;
    int tq = 0;
    stack<vector<int>> q;
    vector<vector<int>> increase;
    CustomStack(int maxSize) {
        this->maxSize = maxSize;
        this->count = 0;
    }
    
    void push(int x) {
        if(maxSize == 0 || count >= maxSize) return;
        q.push({x, q.size()+1 , ++tq});
        count++;
    }
    
    int pop() {
        if(maxSize == 0 || count == 0) return -1;
        vector<int> v = q.top(); q.pop();
        int res = v[0];
        for(int i = 0; i < increase.size(); ++i){
            if(v[2] > increase[i][2] || v[1] > increase[i][0]){
                continue;
            }else{
                res += increase[i][1];
            }
        }
        count--;
        return res;
    }
    
    void increment(int k, int val) {
        increase.push_back({k, val, ++tq});
    }
};

 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
