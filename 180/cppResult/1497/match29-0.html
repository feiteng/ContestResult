class CustomStack {
    vector<int> stc;
    int size;
public:
    CustomStack(int maxSize) {
        size = maxSize;
    }
    
    void push(int x) {
        // cout << "push " << x << endl;
        if (stc.size() >= size) return;
        stc.push_back(x);
    }
    
    int pop() {
        // cout << "pop " << endl;
        if (stc.empty()) return -1;
        int res = stc.back();
        stc.pop_back();
        return res;
    }
    
    void increment(int k, int val) {
        // cout << "inc" << endl;
        for (int i = 0; i < min(k, (int)stc.size()); ++i) {
            stc[i] += val;
        }
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
