class CustomStack {
    int ms;
    vector<int> s;
public:
    CustomStack(int maxSize) {
        ms=maxSize;
    }
    
    void push(int x) {
        if(s.size()==ms)return;
        s.push_back(x);
    }
    
    int pop() {
        if(s.empty())return -1;
        int a=s[s.size()-1];
        s.pop_back();
        return a;
    }
    
    void increment(int k, int val) {
        for(int i=0;i<min((int)s.size(),k);++i)s[i]+=val;
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
