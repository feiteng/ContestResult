class CustomStack {
public:
    int MAX_SIZE;
    vector<int> stack;
    CustomStack(int maxSize) {
        MAX_SIZE = maxSize;
    }
    
    void push(int x) {
        if (stack.size() < MAX_SIZE) {
            stack.push_back(x);
        }
    }
    
    int pop() {
        int ret = -1;
        if (stack.size()) {
            ret = stack.back();
            stack.pop_back();
        }
        return ret;
    }
    
    void increment(int k, int val) {
        for (int i = 0; i < k && i < stack.size(); ++ i) {
            stack[i] += val;
        }
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
