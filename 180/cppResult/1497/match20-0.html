class CustomStack {
public:
    int max_len;
    vector<int> data;
    CustomStack(int maxSize) {
        max_len = maxSize;
        data.clear();
    }
    
    void push(int x) {
        if(data.size() < max_len) data.push_back(x);
    }
    
    int pop() {
        if(data.size() > 0){
            int ans = data[data.size()-1];
            data.pop_back();
            return ans;
        }
        return -1;
    }
    
    void increment(int k, int val) {
        for(int i = 0; i < min(data.size(),(size_t)k); i++) 
            data[i] += val;
    }
};

 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
