class CustomStack {
private:

    vector<int> nums;
    int s;
public:
    CustomStack(int maxSize) {
        s=maxSize;
    }
    
    void push(int x) {
        if(s==nums.size()) return;
        nums.push_back(x);
    }
    
    int pop() {
        if(nums.empty()) return -1;
        int v=nums.back();
        nums.pop_back();
        return v;
    }
    
    void increment(int k, int val) {
        for(int i=0;i<min((int)nums.size(), k); ++i){
            nums[i]+=val;
        }
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
