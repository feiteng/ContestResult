class CustomStack {
public:
    vector<int> cur;
    int maxSize;
    CustomStack(int mx) {
        this->maxSize = mx;
        cur.clear();
    }
    
    void push(int x) {
        if(cur.size() < maxSize) cur.push_back(x);
    }
    
    int pop() {
        if(cur.size() == 0) return -1;
        int x = cur.back();
        cur.pop_back();
        return x;
    }
    
    void increment(int k, int val) {
        for(int i = 0; i < min(k, (int)cur.size()); i ++) cur[i] += val;
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
