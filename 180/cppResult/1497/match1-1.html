class CustomStack {
public:
    
    vector<int> s;
    int size;
    CustomStack(int maxSize) {
        size=maxSize;
    }
    
    void push(int x) {
        if (s.size()<size) {
            s.push_back(x);
        }
    }
    
    int pop() {
        int ans=-1;
        if (s.size()>0) {
            ans = s[s.size()-1];
            s.pop_back();
        }
        return ans;
    }
    
    void increment(int k, int val) {
        for(int i=0;i<s.size()&&i<k;i++){
            s[i]+=val;
        }
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
