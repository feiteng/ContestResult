class CustomStack {
public:
    vector <int> v;
    int s;
    CustomStack(int maxSize) {
        s = maxSize;
        v.clear();
    }
    
    void push(int x) {
        if (v.size() < s) {
            v.push_back(x);
        }
    }
    
    int pop() {
        int ret = -1;
        if (v.size() > 0) {
            ret = v.back();
            v.pop_back();
        }
        return ret;
    }
    
    void increment(int k, int val) {
        for (int i = 0; i < k && i < v.size(); i++) {
            v[i] += val;
        }
    }
};

 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
