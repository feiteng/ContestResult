class CustomStack {
public:
    vector<int> vec;
    int maxsize;
    CustomStack(int maxSize) {
        maxsize = maxSize;
    }
    
    void push(int x) {
        if(vec.size() < maxsize)
            vec.push_back(x);
    }
    
    int pop() {
        if(!vec.empty()){
            int ans = vec[vec.size()-1];
            vec.pop_back();
            return ans;
        }
        else

            return -1;
    }
    
    void increment(int k, int val) {
        for(int i=0;i<k && i<vec.size();i++)
            vec[i] = vec[i] + val;
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
