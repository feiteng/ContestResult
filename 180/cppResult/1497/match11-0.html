class CustomStack {
public:
    vector <int> vec;
    int maxSize;
    CustomStack(int m) {
        maxSize = m;
        vec.clear();
    }
    
    void push(int x) {
        if(vec.size() < maxSize)
            vec.push_back(x);
    }
    
    int pop() {
        if(vec.empty())
            return -1;
        int val = vec.back();
        vec.pop_back();
        return val;
    }
    
    void increment(int k, int val) {
        for(int i = 0; i < min(k, int(vec.size())); i++)
            vec[i] += val;
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
