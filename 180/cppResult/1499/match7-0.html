class Solution {
public:
    int maxPerformance(int n, vector<int>& S, vector<int>& E, int k) {
        vector<pair<int, int>> p(n);
        for (int i = 0; i < n; i++)
            p[i] = {E[i], S[i]};
        sort(p.rbegin(), p.rend());
        priority_queue<int, vector<int>, greater<int>> pq;
        int64_t sum = 0, ans = 0, mod = 1e9+7;
        for (int i = 0; i < n; i++) {
            pq.push(p[i].second);
            sum += p[i].second;
            if (pq.size() > k) {
                int u = pq.top();
                pq.pop();
                sum -= u;
            }
            ans = max(ans, sum * p[i].first);
        }
        return ans % mod;
    }
};
