class Solution {
public:
    int maxPerformance(int n, vector<int>& sp, vector<int>& ef, int k) {
        vector<pair<int,int>> x;
        for(int i = 0; i < n; i++)
            x.push_back({sp[i],ef[i]});
        sort(x.begin(),x.end(),[](pair<int,int>x, pair<int,int>y){
            return x.second<y.second;
        });
        multiset<int> st;
        long long sum=0,res=0,mod=1e9+7;
        for(int i = n-1;i>=0;i--) {
            sum+=x[i].first;
            st.insert(x[i].first);
            if(st.size()>k) {
                sum-=*st.begin();
                st.erase(st.begin());
            }
            res=max(res,x[i].second*sum);
        }
        return res%mod;
    }
};
