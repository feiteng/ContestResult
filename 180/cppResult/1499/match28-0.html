class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int,int>> arr;
        for(int i=0; i<n; i++){
            arr.push_back({efficiency[i],i});
        }
        sort(arr.begin(), arr.end());
        multiset<int> ms;
        long long int ans=0, s=0;
        for(int i=n-1; i>=0; i--) {
            s+=speed[arr[i].second];
            ans=max(ans,s*arr[i].first);
            ms.insert(speed[arr[i].second]);
            if(ms.size()>=k) {
                s-=*(ms.begin());
                ms.erase(ms.begin());
            }
        }
        return ans%(1000000007);
    }
};
