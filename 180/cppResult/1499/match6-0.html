class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int, int> > p;
        for(int i = 0;i < n;i++){
            p.emplace_back(efficiency[i], speed[i]);
        }
        sort(p.begin(), p.end());
        priority_queue<int, vector<int>, greater<int> > pq;
        long long ans = 0;
        long long sum = 0;
        for(int i = n - 1;i >= 0;i--){
            //cout << sum << " " << p[i].first << " " << p[i].second << endl;
            sum += p[i].second;
            pq.push(p[i].second);
            if(pq.size() > k){
                sum -= pq.top();
                pq.pop();
            }
            ans = max(ans, sum * p[i].first);
        }
        int mod = 1e9 + 7;
        return ans % mod;
    }
};
