struct Eng {
    int s, e;
    bool operator < (const Eng rhs) const {
        if(e != rhs.e) return e > rhs.e;
        else return s > rhs.s;
    }
};

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<Eng> vec;
        for(int i = 0; i < n; i++) {
            vec.push_back({speed[i], efficiency[i]});
        }
        sort(vec.begin(), vec.end());
        int mod = 1e9+7;
        priority_queue<int, vector<int>, greater<int>> pq;

        long long sum = 0, ans = 0;
        for(int i = 0; i < n; i++) {
            pq.push(vec[i].s);
            sum += vec[i].s;
            if(pq.size() > k) {
                sum -= pq.top(); pq.pop();
            }
            ans = max(ans, sum*vec[i].e);
        }
        return ans%mod;
    }
};
