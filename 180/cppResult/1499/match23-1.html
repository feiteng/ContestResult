#define MOL 1000000007
#define M 1000000007

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<int> id(n);
        for (int i = 0; i < n; ++i)
            id[i] = i;
        sort(id.begin(), id.end(), [&](int a, int b) {return efficiency[a] < efficiency[b];});

        long long sumn = 0, ret = -1;
        priority_queue<int, vector<int>, greater<int>> Q;
        for (int i = 0; i < k; ++i) {
            Q.push(speed[id[n - 1 - i]]);
            sumn += speed[id[n - 1 - i]];
            ret = max(ret, efficiency[id[n - 1 - i]] * sumn);
        }
        for (int i = n - 1 - k; i >= 0; --i) {
            Q.push(speed[id[i]]);
            sumn += speed[id[i]];
            sumn -= Q.top();
            ret = max(ret, efficiency[id[i]] * sumn);
            Q.pop();
        }
        return ret % MOL;
    }
};
