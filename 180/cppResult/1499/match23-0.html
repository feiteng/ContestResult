class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<vector<int>> eng;
        for (int i = 0; i < n; ++i) {
            eng.push_back({efficiency[i], speed[i]});
        }
        sort(eng.begin(), eng.end());
        // for (int i = 0; i < n; ++i) {
        //     printf("%d %d\n", eng[i][0], eng[i][1]);
        // }
        // puts("----");
        long long int sum = 0;
        long long int res = 0;
        priority_queue<int,vector<int>,greater<int> >pq;
        for (int i = n -1; i >= n - k; --i) {
            sum += eng[i][1];
            pq.push(eng[i][1]);
            res = max(res, sum * eng[i][0]);
        }
        // long long int res = sum * eng[n - k][0];
        // printf("%lld %lld %d\n", sum, res, eng[n - k][0]);
        for (int i = n - k -1; i >= 0; --i) {
            sum -= pq.top();
            // printf("pop = %d, ", pq.top());
            pq.pop();
            sum += eng[i][1];
            pq.push(eng[i][1]);
            // printf("push = %d\n", eng[i][1]);
            res = max(res, sum * eng[i][0]);
            // printf("%lld %lld, eff = %d\n", sum, res, eng[i][0]);
        }
        return res % 1000000007;
    }
};
