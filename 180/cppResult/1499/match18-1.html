class Solution {
public:
    int maxPerformance(int n, vector<int>& s, vector<int>& e, int k) {
        vector<pair<int, int>> all;
        for(int i = 0; i < n; i++) all.push_back(make_pair(e[i], s[i]));
        sort(all.begin(), all.end());

        multiset<int> S;
        long long sum = 0, ans = 0;
        for(int i = n - 1; i >= 0; i--) {
            ans = max(ans, (sum + all[i].second) * all[i].first);
            S.insert(all[i].second);
            sum += all[i].second;
            if(S.size() > k - 1) {
                sum -= (*S.begin());
                S.erase(S.begin());
            }
        }
        int mod = 1e9 + 7;
        return ans % mod;
    }
};
