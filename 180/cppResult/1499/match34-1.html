class Solution {
public:
    vector<pair<long long ,long long > >a;

    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        for(int i=0;i<n;i++){
            a.push_back(make_pair(efficiency[i],speed[i]));
        }
        
        sort(a.begin(),a.end());
       long long ans=0;
        long long sum=0;
        priority_queue<int,vector<int>,greater<int> >q;

        for(int i=n-1;i>=0;i--){
            q.push(a[i].second);
            sum+=a[i].second;
            if(q.size()>k){
                sum-=q.top();
                q.pop();
            }
            ans=max(ans,sum*a[i].first);
        }
       // cout<<ans<<endl;
        return ans%(1000000000+7);
    }
};
