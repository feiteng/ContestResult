class Solution {
public:
    long long MOD = 1000000007;
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<vector<long long>> vec(n);
        for(int i = 0; i < n; ++i){
            vec[i] = {efficiency[i], speed[i]};
        }
        sort(vec.begin(), vec.end(), greater<vector<long long>>());
        priority_queue<vector<long long>, vector<vector<long long>>, greater<vector<long long>> > pq;
        long long res = INT_MIN;
        long long total = 0;
        for(int i = 0; i < n; ++i){
            pq.push({vec[i][1], vec[i][0]});
            total += vec[i][1];
            if(pq.size() > k){
                total -= pq.top()[0];
                pq.pop();
            }
            if(pq.size() <= k){
                res = max(res, total*vec[i][0]);
            }
        }
        return res%MOD;
    }
};
