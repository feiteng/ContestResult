/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    vector<int> numbers;
    void walk(TreeNode* root) {
        if (!root) return;
        walk(root->left);
        numbers.push_back(root->val);
        walk(root->right);
    }
    TreeNode* balance(int l, int r) {
        if (l > r) return NULL;
        const int m = (l + r) >> 1;
        TreeNode* root = new TreeNode(numbers[m]);
        root->left = balance(l, m - 1);
        root->right = balance(m + 1, r);
        return root;
    }
public:
    TreeNode* balanceBST(TreeNode* root) {
        numbers.clear();
        walk(root);
        return balance(0, numbers.size() - 1);
    }
};
