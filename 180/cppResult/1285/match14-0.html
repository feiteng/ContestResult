/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        if(root == nullptr) return nullptr;
        vector<TreeNode*> nodes;
        saveNode(root, nodes);
        return dfsBuildTree(0, nodes.size() - 1, nodes);
    }
    
    void saveNode(TreeNode* root, vector<TreeNode*>& nodes){
        if(root == nullptr) return;
        saveNode(root->left, nodes);
        nodes.emplace_back(root);
        saveNode(root->right, nodes);
    }
    
    TreeNode* dfsBuildTree(int l, int e, vector<TreeNode*>& nodes){
        if(l > e) return nullptr;
        int m = (l + e)/2;
        TreeNode* root = nodes[m];
        root->left = dfsBuildTree(l , m - 1, nodes);
        root->right = dfsBuildTree(m + 1 , e, nodes);
        return root;
    }
    
};
