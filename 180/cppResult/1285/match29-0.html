/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void storeBSTNodes(TreeNode* root, vector<TreeNode*> &nodes) { 
        if (root==NULL) return; 
        storeBSTNodes(root->left, nodes); 
        nodes.push_back(root); 
        storeBSTNodes(root->right, nodes); 
    } 
    TreeNode* buildTree(vector<TreeNode*> &nodes, int start, int end) { 
        if (start > end) return NULL; 
        int mid = (start + end)/2; 
        TreeNode *root = nodes[mid]; 

        root->left  = buildTree(nodes, start, mid-1); 
        root->right = buildTree(nodes, mid+1, end); 

        return root; 
    } 
    TreeNode* balanceBST(TreeNode* root) {
        vector<TreeNode*> nodes; 
        storeBSTNodes(root, nodes); 

        int n = nodes.size(); 
        return buildTree(nodes, 0, n-1); 
    }
};
