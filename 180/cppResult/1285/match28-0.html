/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    std::vector<int> v;
    void dfs(TreeNode* root) {
        if(root == NULL) {
            return;
        }
        v.push_back(root->val);
        dfs(root->left);
        dfs(root->right);
    }
    TreeNode* buildTree(int le, int ri) {
        if(le > ri) {
            return NULL;
        }
        if(le == ri) {
            TreeNode* root = new TreeNode(v[le]);
            return root;
        }

        int mid = (le + ri) / 2;
        TreeNode* root = new TreeNode(v[mid]);
        root->left = buildTree(le, mid - 1);
        root->right = buildTree(mid + 1, ri);
        return root;
    }
    TreeNode* balanceBST(TreeNode* root) {
        v.clear();
        dfs(root);
                sort(v.begin(), v.end());
        return buildTree(0,v.size()-1);
    }
};
