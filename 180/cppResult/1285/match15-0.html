/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> element;

    void pre_dfs(TreeNode *root) {
        if (root != NULL) {
            pre_dfs(root->left);
            element.push_back(root->val);
            pre_dfs(root->right);
        }
    }
    TreeNode *create(const vector<int> &x) {
        int len = x.size();
        auto root = new TreeNode(x[len / 2]);
        vector<int> left;
        vector<int> right;
        for (int i = 0; i < len / 2; i++) {
            left.push_back(x[i]);
        }
        for (int i = len / 2 + 1; i < len; i++) {
            right.push_back(x[i]);
        }
        if (!left.empty()) {
            root->left = create(left);
        }
        if (!right.empty()) {
            root->right = create(right);
        }
        return root;
    }
    TreeNode *balanceBST(TreeNode *root) {
        element.clear();
        pre_dfs(root);
        return create(element);
    }
};
