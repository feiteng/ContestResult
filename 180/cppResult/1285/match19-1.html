/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
  vector<int> data;
  void dfs(TreeNode* root) {
    if (!root) return;
    dfs(root->left);
    data.emplace_back(root->val);
    dfs(root->right);
  }
  TreeNode* build(int l, int r) {
    if (l > r) return nullptr;

    int mid = (l + r) / 2;
    TreeNode* res = new TreeNode(data[mid]);
    res->left = build(l, mid - 1);
    res->right = build(mid + 1, r);
    return res;
  }
  TreeNode* balanceBST(TreeNode* root) {
    dfs(root);
    TreeNode* res = build(0, data.size() - 1);
    return res;
  }
};
