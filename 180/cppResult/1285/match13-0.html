/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    void fillData(TreeNode* root, vector<int>& data)
    {
        if (root == NULL) return;
        fillData(root->left, data);
        data.push_back(root->val);
        fillData(root->right, data);
    }
    TreeNode* makeTree(vector<int>& data, int start, int end)
    {
        if (start == end) return NULL;
        int mid = (start + end) / 2;
        TreeNode* node = new TreeNode(data[mid]);
        node->left = makeTree(data, start, mid);
        node->right = makeTree(data, mid + 1, end);
        return node;
    }
public:
    TreeNode* balanceBST(TreeNode* root)
    {
        vector<int> data;
        fillData(root, data);
        return makeTree(data, 0, data.size());
    }
};
