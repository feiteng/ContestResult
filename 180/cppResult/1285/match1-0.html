/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> data;
    void dfs(TreeNode * rt){
        if(rt == NULL) return ;
        dfs(rt->left);
        data.push_back(rt->val);
        dfs(rt->right);
    }
    TreeNode * build(int left, int right){
        if(left > right)  return NULL;
        int mid = (left + right) / 2;
        TreeNode * rt = new TreeNode(data[mid]);
        rt->left = build(left, mid-1);
        rt->right = build(mid+1, right);
        return rt;
    }
    TreeNode* balanceBST(TreeNode* root) {
        data.clear();
        dfs(root);
        return build(0,data.size()-1);
    }
};
