/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> a;

    void dfs(TreeNode* cur) {
        if (!cur) return;
        dfs(cur->left);
        a.push_back(cur->val);
        dfs(cur->right);
    }
    
    TreeNode* get_tree(int l, int r) {
        if (l > r) return NULL;
        else if (l == r) return new TreeNode(a[l]);
        int mid = l + r >> 1;
        TreeNode* root = new TreeNode(a[mid]);
        root->left = get_tree(l, mid - 1);
        root->right = get_tree(mid + 1, r);
        return root;
    }
    
    TreeNode* balanceBST(TreeNode* root) {
        dfs(root);
        return get_tree(0, a.size() - 1);
    }
};
