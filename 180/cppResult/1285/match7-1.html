/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        vector<TreeNode*> nodes;
        storeTree(root, nodes);
        return buildTree(nodes, 0, nodes.size() - 1); 
    }
    
    void storeTree(TreeNode* root, vector<TreeNode*>& nodes) {
        if (root == nullptr) return;
        storeTree(root->left, nodes);
        nodes.push_back(root);
        storeTree(root->right, nodes);
    }
    
    TreeNode* buildTree(vector<TreeNode*> &nodes, int start, int end) {
        if (start > end)
            return nullptr;

        int mid = (start + end)/2;
        TreeNode *root = nodes[mid];

        root->left  = buildTree(nodes, start, mid - 1);
        root->right = buildTree(nodes, mid + 1, end);

        return root;
    }
};
