/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> val;
    void dfs(TreeNode* root) {
        if (root == NULL) return;
        val.push_back(root->val);
        dfs(root->left);
        dfs(root->right);
    }
    
    TreeNode* build(int l, int r) {
        if (l > r) return NULL;
        if (l == r) return new TreeNode(val[l]);
        int mid = (l + r) / 2;
        TreeNode* ans = new TreeNode(val[mid]);
        ans->left = build(l, mid-1);
        ans->right = build(mid+1, r);
        return ans;
    }
    
    TreeNode* balanceBST(TreeNode* root) {
        dfs(root);
        sort(val.begin(), val.end());
        // for (auto num : val) cout << num << endl;
        return build(0, val.size()-1);
    }
};
