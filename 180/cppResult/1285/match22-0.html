/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    void dfs(TreeNode* r) {
        if (!r) return;
        dfs(r->left);
        v.push_back(r->val);
        dfs(r->right);
    }
    
    TreeNode* build(int b, int e) {
        if (b > e) return NULL;
        
        int m = (b + (e - b) / 2);
        TreeNode* node = new TreeNode(v[m]);
        node->left = build(b, m - 1);
        node->right = build(m + 1, e);
        return node;
    }
    
    TreeNode* balanceBST(TreeNode* root) {
        v.clear();
        dfs(root);
        // create tree
        return build(0, v.size() - 1);
    }
};
