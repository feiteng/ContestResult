/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */


class Solution {
public:
    std::vector<int> elems;
    
    void traverse(TreeNode* root){
        if(root == NULL){
            return;
        }
        
        traverse(root->left);
        elems.push_back(root->val);
        traverse(root->right);
    }
    
    TreeNode* createTree(int l, int r){
        if(l >= r){
            return NULL;
        }
        

        int mid = (l + r) / 2;
        
        TreeNode* c = new TreeNode(elems[mid]);
        c->left = createTree(l, mid);
        c->right = createTree(mid + 1, r);
        
        return c;
    }

    TreeNode* balanceBST(TreeNode* root) {
        elems.clear();
        traverse(root);
        return createTree(0, elems.size());
    }
};
