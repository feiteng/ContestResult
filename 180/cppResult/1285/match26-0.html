/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    void getv(TreeNode* root)
    {
        if(!root) return;
        getv(root->left);
        v.push_back(root->val);
        getv(root->right);
    }
    TreeNode* construct(int l, int r)
    {
        if(l == r) return new TreeNode(v[l]);
        if(l > r) return NULL;
        int m = (r + l) / 2;
        TreeNode* newNode = new TreeNode(v[m]);
        newNode->left = construct(l,m-1);
        newNode->right = construct(m+1, r);
        return newNode;
    }
    TreeNode* balanceBST(TreeNode* root) {
        getv(root);
        return construct(0, v.size() - 1);
    }
};
