/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<int> v;
    
    void to_list(TreeNode* root) {
        if (root == NULL) {
            return;
        }
        
        to_list(root->left);
        v.push_back(root->val);
        to_list(root->right);
    }
    
    TreeNode* to_BST(int l, int r) {
        if (l == r) {
            return NULL;
        }
        
        int mid = (l + r) >> 1;
        TreeNode* root = new TreeNode(v[mid]);
        root->left = to_BST(l, mid);
        root->right = to_BST(mid + 1, r);
        return root;
    }
    
public:
    TreeNode* balanceBST(TreeNode* root) {
        to_list(root);
        return to_BST(0, v.size());
    }
};
