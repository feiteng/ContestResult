/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> v;
    void dfs(TreeNode *u)
    {
        if(u==NULL) return;
        dfs(u->left);
        v.push_back(u->val);
        dfs(u->right);
    }
    TreeNode* dfs2(int l,int r)
    {
        if(l>r) return NULL;

        int mid=(l+r)>>1;
        TreeNode *u=new TreeNode(v[mid]);
        u->left=dfs2(l,mid-1);
        u->right=dfs2(mid+1,r);
        return u;
    }
    TreeNode* balanceBST(TreeNode* root) {
        dfs(root);
        return dfs2(0,v.size()-1);
    }
};
