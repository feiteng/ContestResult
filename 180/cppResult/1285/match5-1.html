/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution 
{
public:
	TreeNode* balanceBST(TreeNode* root) 
	{
		vector<int>a;
		go(root, a);
		return create(a, 0, a.size() - 1);
	}
	void go(TreeNode* root, vector<int>&a)
	{
		if (root == NULL) return;
		go(root->left, a);
		a.push_back(root->val);
		go(root->right, a);
	}
	TreeNode* create(vector<int>&a, int l, int r)
	{
		if (l > r) return NULL;
		int m = (l + r) / 2;
		TreeNode* root = new TreeNode(a[m]);
		root->left = create(a, l, m - 1);
		root->right = create(a, m + 1, r);
		return root;
	}
};
