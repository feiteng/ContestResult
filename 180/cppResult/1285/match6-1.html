/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
    vector<int> vals;
    void traverse(TreeNode *node) {
        if (node == NULL) {
            return;
        }
        traverse(node->left);
        vals.push_back(node->val);
        traverse(node->right);
    }
    TreeNode *gen(int first, int last) {
        if (first > last) {
            return NULL;
        }
        int mid = first + (last - first) / 2;
        TreeNode *node = new TreeNode(vals[mid]);
        node->left = gen(first, mid - 1);
        node->right = gen(mid + 1, last);
        return node;
    }
public:
    TreeNode* balanceBST(TreeNode* root) {
        traverse(root);
        return gen(0, vals.size() - 1);
    }
};
