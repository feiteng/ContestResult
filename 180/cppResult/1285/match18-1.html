/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
private:
    vector<int>a;
    void dfs(TreeNode* root){
        if(!root)return;
        dfs(root->left);
        a.push_back(root->val);
        dfs(root->right);
    }
    TreeNode* dfs2(int l, int r){
        if(l>r)return NULL;

        int mid = (l+r)>>1;
        TreeNode* root = new TreeNode(a[mid]);
        root->left = dfs2(l, mid-1);
        root->right = dfs2(mid+1,r);
        return root;
    }
public:
    TreeNode* balanceBST(TreeNode* root) {
        a.clear();
        dfs(root);
        return dfs2(0, a.size()-1);
    }
};
