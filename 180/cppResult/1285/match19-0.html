/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    void dfs(TreeNode* node)
    {
        if(!node) return;
        v.push_back(node->val);
        dfs(node->left);
        dfs(node->right);
    }
    TreeNode* construct(int l,int r)
    {
        if(l>=r) return NULL;
        int mid=(l+r)/2;
        TreeNode * now=new TreeNode(v[mid]);
        now->left=construct(l,mid);
        now->right=construct(mid+1,r);
        return now;
    }
    TreeNode* balanceBST(TreeNode* root) {
        dfs(root);
        sort(v.begin(),v.end());
        TreeNode *ans=construct(0,v.size());
        return ans;
    }
};
