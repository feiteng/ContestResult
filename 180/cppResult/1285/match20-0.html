/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int>a;
    void dfs(TreeNode* root){
        if(!root)return;
        dfs(root->left);
        dfs(root->right);
        a.push_back(root->val);
    }
    TreeNode* build(int l,int r){
        if(l > r)return NULL;
        int mid = (l+r)/2;
        auto no = new TreeNode(a[mid]);
        no->left = build(l,mid-1);
        no->right = build(mid+1,r);
        return no;
    }
    TreeNode* balanceBST(TreeNode* root) {
        a.clear();
        dfs(root);
        sort(begin(a),end(a));
        return build(0,a.size()-1);
    }
};
