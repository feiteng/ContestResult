/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        vector<int> nums;
        collect_num(root, nums);
        return constructBST(nums, 0, nums.size()-1);
    }
    
    void collect_num(TreeNode* root, vector<int>& nums){
        if(root == NULL) return ;
        collect_num(root->left, nums);
        nums.push_back(root->val);
        collect_num(root->right, nums);
    }
    
    TreeNode* constructBST(vector<int>& nums, int l, int r) {
        // nums[l ... r]
        if(l > r) return NULL;
        int mid = (l + r) / 2;
        // partition to nums[l... mid-1], nums[mid], nums[mid+1,...r]
        TreeNode* res = new TreeNode(nums[mid]);
        res->left = constructBST(nums, l, mid-1);
        res->right = constructBST(nums, mid+1, r);
        
        return res;
    }
};
