/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        if(root == nullptr) return nullptr;
        vector<TreeNode*> nodes;
        saveNode(root, nodes);
        return help(0, nodes.size() - 1, nodes);
    }
    
    void saveNode(TreeNode* root, vector<TreeNode*>& nodes){
        if(root == nullptr) return;
        saveNode(root->left, nodes);
        nodes.emplace_back(root);
        saveNode(root->right, nodes);
    }
    
    TreeNode* help(int l, int e, vector<TreeNode*>& vec){
        if(l > e) return nullptr;
        int m = (l + e)/2;
        TreeNode* root = vec[m];
        root->left = help(l , m - 1, vec);
        root->right = help(m + 1 , e, vec);
        return root;
    }
    
};
