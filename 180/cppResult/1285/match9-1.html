/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> a;
    
    void dfs(TreeNode* r) {
        if (r==NULL) return ;
        dfs(r->left);
        a.push_back(r->val);
        dfs(r->right);
    }
    
    TreeNode* build(int x, int y) {
        if (x>y)return NULL;
        int m = (x+y)/2;
        TreeNode *r = new TreeNode(a[m]);
        
        r->left = build(x, m-1);
        r->right = build(m+1,y);
        return r;
    }
    
    TreeNode* balanceBST(TreeNode* root) {
        dfs(root);
        return build(0, a.size()-1);    
    }
};
