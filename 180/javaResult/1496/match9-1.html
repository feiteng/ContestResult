class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int row = matrix.length, col = matrix[0].length;
        int[] r = new int[row], l = new int[col];
        Arrays.fill(r, 555555);
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                r[i] = Math.min(r[i], matrix[i][j]);
                l[j] = Math.max(l[j], matrix[i][j]);
            }
        }
        List<Integer> list = new ArrayList<>();
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(matrix[i][j] == r[i] && r[i] == l[j])
                    list.add(r[i]);
            }
        }
        return list;
    }
}
