import java.util.ArrayList;
import java.util.List;

class Solution {
    public List<Integer> luckyNumbers (int[][] g) {
        int h = g.length;
        int w = g[0].length;
        int[] rowMin = new int[h];
        int[] colMax = new int[w];

        for(int i=0;i<h;i++){
            int min = Integer.MAX_VALUE;
            for(int j=0;j<w;j++) {
                min=Math.min(min, g[i][j]);
                colMax[j]=Math.max(colMax[j],g[i][j]);
            }
            rowMin[i]=min;
        }
        
        List<Integer> rl = new ArrayList<>();
        for(int i=0;i<h;i++){
            for(int j=0;j<w;j++) {
                if(g[i][j]==rowMin[i]&&g[i][j]==colMax[j]){
                    rl.add(g[i][j]);
                }
            }
        }
        
        
        return rl;
    }
}
