class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        List<Integer> ans = new ArrayList<>();
        for(int i=0; i < n; i++) {
            for(int j=0;j < m; j++) {
                int min = -1;
                for(int k=0; k < m; k++) {
                    if(min == -1 || matrix[i][k] < min) {
                        min = matrix[i][k];
                    }
                }
                int max = -1;
                for(int k=0; k < n; k++) {
                    if(max == -1 || matrix[k][j] > max) {
                        max = matrix[k][j];
                    }
                }
                if(min == max && min == matrix[i][j]) {
                    ans.add(min);
                }
            }
        }
        return ans;
    }

}
