class Solution {
    public List<Integer> luckyNumbers (int[][] a) {
        int m = a.length;
        int[] rm = new int[m];
        int n = a[0].length;
        int[] cm = new int[n];
        for(int i=0; i<m;i++){
            rm[i]  = Integer.MAX_VALUE;
            for(int j=0; j<n;j++){
                rm[i] = Math.min(rm[i], a[i][j]);
            }
        }
        for(int j=0; j<n;j++){
            cm[j] = Integer.MIN_VALUE;
            for(int i=0; i<m;i++){
                cm[j] = Math.max(cm[j], a[i][j]);
            }
        }
        List<Integer> r = new ArrayList<>();
        for(int i=0; i<m;i++){
            for(int j=0; j<n;j++){
                if(a[i][j] == rm[i] && a[i][j] == cm[j]){
                    r.add(a[i][j]);
                }
            }
        }
        return r;
    }
}
