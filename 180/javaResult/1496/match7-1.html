class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[] rows = new int[n], columns = new int[m];
        for(int i = 0 ; i < n ; i++){
            rows[i] = matrix[i][0];
            for(int j = 1 ; j < m ; j++){
                rows[i] = Math.min(matrix[i][j], rows[i]);
            }
        }
        for(int j = 0 ; j < m ; j++){
            columns[j] = matrix[0][j];
            for(int i = 1 ; i < n ; i++){
                columns[j] = Math.max(matrix[i][j], columns[j]);
            }
        }
        List<Integer> res = new ArrayList<>();
        for(int i = 0 ; i< n ; i++){
            for(int j = 0 ; j < m ; j++){
                if(rows[i] == columns[j]){
                    res.add(rows[i]);
                }
            }
        }
        return res;
    }
}
