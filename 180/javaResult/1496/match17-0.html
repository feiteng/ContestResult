class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        List<Integer> res = new ArrayList<>();
        
        OUTER:
        for(int i = 0; i < n; i++){
            int min = 1000_000;
            int loc = -1;
            for(int j = 0; j < m; j++){
                if(matrix[i][j] < min){
                    min = matrix[i][j];
                    loc = j;
                }
            }
            
            for(int j = 0; j < n; j++){
                if(matrix[j][loc] > min){
                    continue OUTER;
                }
            }
            
            res.add(min);
        }
        
        return res;
    }
}
