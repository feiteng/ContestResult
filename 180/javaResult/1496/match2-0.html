class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int R = matrix.length, C = matrix[0].length;
        int[] min = new int[R];
        for (int r = 0; r < R; r++) {
            int m = 0;
            for (int c = 1; c < C; c++) {
                if (matrix[r][c] < matrix[r][m]) {
                    m = c;
                }
            }
            min[r] = m;
        }
        int[] max = new int[C];
        for (int c = 0; c < C; c++) {
            int m = 0;
            for (int r = 1; r < R; r++) {
                if (matrix[r][c] > matrix[m][c]) {
                    m = r;
                }
            }
            max[c] = m;
        }
        List<Integer> res = new ArrayList();
        for (int r = 0; r < R; r++) {
            if (max[min[r]] == r) {
                res.add(matrix[r][min[r]]);
            }
        }
        return res;
    }
}
