class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        int[] line = new int[m], col = new int[n];
        Arrays.fill(line, Integer.MAX_VALUE);
        Arrays.fill(col, Integer.MIN_VALUE);
        for (int i = 0 ; i < m ; i++) {
            
for (int j = 0 ; j < n; j++) {
                line[i] = Math.min(line[i], matrix[i][j]);
                col[j] = Math.max(col[j], matrix[i][j]);
            }
        
}
        List<Integer> ans = new ArrayList<>();
        for (int i = 0 ; i < m ; i++) {
            for (int j = 0 ; j < n; j++) {
                if (line[i] == matrix[i][j] && col[j] == matrix[i][j]) {
                    ans.add(matrix[i][j]);
                }
            }
        }
        return ans;

    }
}
