import java.util.*;

class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {

        final int nrow = matrix.length;
        final int ncol = matrix[0].length;
        int[] rowmin = new int[nrow];
        int[] colmax = new int[ncol];

        for (int row = 0; row < nrow; row++) {
            for (int col = 0; col < ncol; col++) {

                if (rowmin[row] == 0) {
                    rowmin[row] = matrix[row][col];
                }
                else {
                    rowmin[row] = Math.min(matrix[row][col], rowmin[row]);
                }

                colmax[col] = Math.max(matrix[row][col], colmax[col]);

            }
        }

        List<Integer> magic = new ArrayList<>();
        for (int row = 0; row < nrow; row++) {
            for (int col = 0; col < ncol; col++) {
                if (matrix[row][col] == rowmin[row] && matrix[row][col] == colmax[col]) {
                    magic.add(matrix[row][col]);
                }
            }
        }
        return magic;
    }
}
