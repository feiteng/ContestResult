class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        
        int[] rows = new int[m];
        int[] cols = new int[n];
        
        for(int i =0;i<m;i++){
            rows[i] = matrix[i][0];
            for(int j =1;j<n;j++){
                rows[i] = Math.min(rows[i], matrix[i][j]);
            }
        }
        
        for(int j =0;j<n;j++){
            cols[j] = matrix[0][j];
            for(int i =1;i<m;i++){
                cols[j] = Math.max(cols[j], matrix[i][j]);
            }
        }
        
        List<Integer> list = new ArrayList<>();
        for(int i =0;i<m;i++){
            for(int j =0;j<n;j++){
                if(matrix[i][j] == rows[i] && matrix[i][j] == cols[j]){
                    list.add(matrix[i][j]);
                }
            }
        }
        return list;
    }
}
