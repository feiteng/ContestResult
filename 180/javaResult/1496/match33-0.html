class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
    	List<Integer> ans= new ArrayList<>();
    	
    	for(int i = 0; i < matrix.length; i++) {
    		int min = 1000000;
    		int minIdx = -1;
    		for(int j = 0; j < matrix[0].length; j++) {
    			if(matrix[i][j] < min) {
    				min = matrix[i][j];
    				minIdx = j;
    			}
    		}
    		boolean ok = true;
    		for(int ii = 0; ii < matrix.length; ii++) {
    			if(matrix[ii][minIdx] > min) {
    				ok = false;
    				break;
    			}
    		}
    		if(ok) ans.add(min);
    		
    	}
        return ans;
    }
}
