	class S
olution {
	    public List<Integer> luckyNumbers (int[][] a) {
	    	int n = a.length;
	    	int m = a[0].length;
	    	List<Integer> ret = new ArrayList<>();
	    	outer:
	        for(int i = 0;i < n;i++){
	        	int min = Integer.MAX_VALUE;
	        	int arg = -1;
	        	for(int j = 0;j < m;j++){
	        		if(a[i][j] < min){
	        			min = a[i][j];
	        			arg = j;
	        		}
	        	}
	        	for(int k = 0;k < n;k++){
	        		if(min < a[k][arg]){
	        			continue outer;
	        		}
	        	}
	        	ret.add(min);
	        }
	    	return ret;
	    }
	}
