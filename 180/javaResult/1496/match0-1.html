class Solution {
    public List<Integer> luckyNumbers (int[][] arr) {
        List<Integer> list = new ArrayList<>();
        int h = arr.length, w = arr[0].length;
        for(int i = 0; i < h; i++){
            int cur = arr[i][0], index = 0;
            for(int j = 0; j < w; j++){
                if(cur > arr[i][j]){
                    cur = arr[i][j];
                    index = j;
                }
            }
            boolean flag = false;
            for(int j = 0; j < h; j++){
                if(cur < arr[j][index])
                    flag = true;
            }
            if(!flag)
                list.add(cur);
        }
        return list;
    }
}
