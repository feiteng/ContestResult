class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        List<Integer> ans = new ArrayList<>();
        int n = matrix.length;
        int m = matrix[0].length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int num = matrix[i][j];
                boolean luck = true;
                for (int k = 0; k < n; k++) {
                    if (matrix[k][j] > num) {
                        luck = false;
                        break;
                    }
                }
                for (int k = 0; k < m; k++) {
                    if (matrix[i][k] < num) {
                        luck = false;
                        break;
                    }
                }
                if (luck) {
                    ans.add(num);
                }
            }
        }
        return ans;
    }
}
