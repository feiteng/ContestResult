class Solution {
    public List<Integer> luckyNumbers (int[][] m) {
        List<Integer> res = new ArrayList<>();
        int h = m.length, l = m[0].length;
        for (int i = 0; i < h; i++) {
            int minj = 0;
            for (int j = 0; j < l; j++) {
                if (m[i][j] < m[i][minj]) {
                    minj = j;
                }
            }
            int maxi = i;
            for (int j = 0; j < h; j++) {
                if (m[maxi][minj] < m[j][minj]) {
                    maxi = j;
                }
            }
            if (maxi == i) res.add(m[maxi][minj]);
        }
        return res;
    }
}
