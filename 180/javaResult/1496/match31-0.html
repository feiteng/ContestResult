class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        int[] row = new int[n];
        int[] col = new int[m];
        for(int i = 0; i < n; i++) {
            int min = matrix[i][0];
            for(int j = 1; j < m; j++) {
                min = Math.min(min, matrix[i][j]);
            }
            row[i] = min;
        }
        for(int j = 0; j < m; j++) {
            int max = matrix[0][j];
            for(int i = 1; i < n; i++) {
                max = Math.max(max, matrix[i][j]);
            }
            col[j] = max;
        }
        List<Integer> res = new ArrayList<>();
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(matrix[i][j] == row[i] && matrix[i][j] == col[j]) res.add(matrix[i][j]);
            }
        }
        return res;
        
    }
}
