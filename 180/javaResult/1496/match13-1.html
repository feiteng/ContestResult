class Solution {
        public List<Integer> luckyNumbers (int[][] matrix) {
            List<Integer> list  = new ArrayList<>();
            int R = matrix.length, C = matrix[0].length;
            int[] r = new int[R], c = new int[C];

            for(int i = 0; i < R; i++) {
                r[i] = matrix[i][0];
                for(int j = 0; j < C; j++) {
                    r[i] = Math.min(r[i], matrix[i][j]);
                }
            }
            
            for(int j = 0; j < C; j++) {
                c[j] = matrix[0][j];
                for(int i = 0; i < R; i++) {
                    c[j] = Math.max(c[j], matrix[i][j]);
                }
            }

            for(int i = 0; i < R; i++) {
                for(int j = 0; j < C; j++) {
                    int val = matrix[i][j];
                    if(val == r[i] && val == c[j]) {
                        list.add(val);
                    }
                }
            }
            return list;

        }
    }
