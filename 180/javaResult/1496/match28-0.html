class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int r = matrix.length;
        int c = matrix[0].length;
        int[] m1 = new int[r];
        int[] m2 = new int[c];
        for (int i = 0; i < r; i++) {
            m1[i] = Arrays.stream(matrix[i]).min().getAsInt();
        }
        for (int i = 0; i < c; i++) {
            int max = 0;
            for (int j = 0; j < r; j++) {
                max = Math.max(max,matrix[j][i]);
            }
            m2[i] = max;
        }
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                int a = matrix[i][j];
                if (a == m1[i] && a == m2[j])res.add(a);
            }
        }
        return res;
    }
}
