/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> all = new ArrayList<>();
    public TreeNode balanceBST(TreeNode root) {
        in(root);
        return help(0, all.size() - 1);
    }
    void in(TreeNode cur) {
        if (cur == null) return;
        in(cur.left);
        all.add(cur.val);
        in(cur.right);
        
    }
    TreeNode help(int s, int e) {
        if (s > e) return null;
        if (s == e) return new TreeNode(all.get(s));
        
        int mid = (s + e) / 2;
        TreeNode r = new TreeNode(all.get(mid));
        r.left = help(s, mid - 1);
        r.right = help(mid + 1, e);
        return r;
    }
}
