/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> val = new ArrayList<>();
        midOrder(root, val);
        return build(val, 0, val.size() - 1);
    }
    
    private void midOrder(TreeNode root, List<Integer> res){
        if(root == null){
            return;
        }
        
        
        midOrder(root.left, res);
        res.add(root.val);
        midOrder(root.right, res);
    }
    
    private TreeNode build(List<Integer> val, int s, int e){
        if(s > e){
            return null;
        }
        
        int mid = s + ((e - s) >> 1);
        TreeNode res = new TreeNode();
        res.val = 
val.get(mid);
        res.left = build(val, s, mid - 1);
        res.right = build(val, mid + 1, e);
        
        return res;
    }
}
