/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> values = new ArrayList<>();
        dfsInOrder(root, values);
        return buildTree(0, values.size()-1, values);
    }
    
    private TreeNode buildTree(int lo, int hi, List<Integer> list) {
        if (lo > hi) {
            return null;
        }
        if (lo == hi) {
            return new TreeNode(list.get(lo));
        }
        int mid = lo + (hi - lo) / 2;
        TreeNode node = new TreeNode(list.get(mid));
        node.left = buildTree(lo, mid-1, list);
        node.right = buildTree(mid+1, hi, list);
        return node;
    }
    
    private void dfsInOrder(TreeNode node, List<Integer> res) {
        if (node == null) {
            return;
        }
        dfsInOrder(node.left, res);
        res.add(node.val);
        dfsInOrder(node.right, res);
    }
}
