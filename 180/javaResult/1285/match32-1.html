/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    List<Integer> inorder = new ArrayList<>();
    public TreeNode balanceBST(TreeNode root) {
        traverse(root);
        return toBST(0, inorder.size()-1);
    }
    
    void traverse(TreeNode n) {
        if(n == null )return;
        traverse(n.left);
        inorder.add(n.val);
        traverse(n.right);
    }
    
    TreeNode toBST(int low, int high) { 
        if (low > high)
            return null
; 
  
        int mid = (low + high) / 2; 
        TreeNode n = new TreeNode(inorder.get(mid));

        n.left = toBST(low, mid - 1); 
        n.right = toBST(mid + 1, high); 
        return n; 
    }
}
