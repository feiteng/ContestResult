/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<TreeNode> nodes = new ArrayList<>();
        traverseTree(root, nodes);
        
        return buildTree(nodes, 0, nodes.size() - 1);
    }
    
    public void traverseTree(TreeNode root, List<TreeNode> nodes) {
        if (root == null) {
            return;
        }
        traverseTree(root.left, nodes);
        nodes.add(root);
        traverseTree(root.right, nodes);
    }
    
    public TreeNode buildTree(List<TreeNode> nodes, int start, int end) {
        if (start > end) {
            return null;
        }
        
        int mid = (start + end) / 2;
        TreeNode root = nodes.get(mid);
        root.left = buildTree(nodes, start, mid - 1);
        root.right = buildTree(nodes, mid + 1, end);
        
        return root;
    }
}
