/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<TreeNode> nodes = new ArrayList<>();
        midOrder(root, nodes);
        return balanceBST(nodes, 0, nodes.size() - 1);
    }

    private void midOrder(TreeNode root, List<TreeNode> res) {
        if (root == null) return;
        midOrder(root.left, res);
        res.add(root);
        midOrder(root.right, res);
    }

    private TreeNode balanceBST(List<TreeNode> nodes, int i, int j) {
        if (i > j) return null;
        int mid = (i + j) / 2;
        TreeNode root = nodes.get(mid);
        root.left = balanceBST(nodes, i, mid - 1);
        root.right = balanceBST(nodes, mid + 1, j);
        return root;
    }
}
