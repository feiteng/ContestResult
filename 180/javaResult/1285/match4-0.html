/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<TreeNode> L = new ArrayList<>();
        collect(root, L);
        return connect(0, L.size() - 1, L);
    }
    
    TreeNode connect(int lo, int hi, List<TreeNode> L) {
        if (hi < lo) return null;
        
        int mid = (lo + hi) / 2;
        TreeNode root = L.get(mid);
        root.left = connect(lo, mid - 1, L);
        root.right = connect(mid + 1, hi, L);
        return root;
    }
    
    void collect(TreeNode root, List<TreeNode> L) {
        if (root == null) return;
        
        collect(root.left, L);
        L.add(root);
        collect(root.right, L);
    }
}
