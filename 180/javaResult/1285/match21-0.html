/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> vals = new ArrayList<>();
        inorder(root, vals);
        return build(vals, 0, vals.size()-1);
    }
    
    private void inorder(TreeNode root, List<Integer> vals) {
        if(root == null) {
            return;
        }
        inorder(root.left, vals);
        vals.add(root.val);
        inorder(root.right, vals);
    }
    
    private TreeNode build(List<Integer> vals, int s, int e) {
        if(s > e) {
            return null;
        }
        int mid = (s + e) / 2;
        TreeNode root = new TreeNode(vals.get(mid));
        root.left = build(vals, s, mid - 1);
        root.right = build(vals, mid + 1, e);
        
        return root;
    }
}
