import java.util.ArrayList;
import java.util.List;

class TreeNode {
     int val;
     TreeNode left;
     TreeNode right;
     TreeNode(int x) { val = x; }
}

class Solution {
    List<Integer> list = new ArrayList<>();
    public TreeNode balanceBST(TreeNode root) {
        midF(root);
        return bin(0, list.size() - 1);
    }

    public TreeNode bin(int l, int r) {
        if (l > r) return null;
        if (l == r) return new TreeNode(list.get(l));

        int mid = (l + r) / 2;
        TreeNode node = new TreeNode(list.get(mid));
        node.left = bin(l, mid - 1);
        node.right = bin(mid + 1, r);
        return node;
    }

    public void midF(TreeNode root) {
        if (root == null) return;
        midF(root.left);
        list.add(root.val);
        midF(root.right);
    }
}
