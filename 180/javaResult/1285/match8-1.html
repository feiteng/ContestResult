/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode balanceBST(TreeNode root) {
        
        List<TreeNode> nodes = new LinkedList<>();
        traverse(root, nodes);
        return rebalance(nodes, 0, nodes.size() - 1);
    }
    
    private void traverse(TreeNode root, List<TreeNode> nodes) {
        
        if (root == null) return;
        
        traverse(root.left, nodes);
        nodes.add(root);
        traverse(root.right, nodes);
    }
    
    private TreeNode rebalance(List<TreeNode> nodes, int l, int r) {
        
        if (l > r) return null;
        
        int m = (l + r) / 2;
        TreeNode root = 
nodes.get(m);
        
        root.left = rebalance(nodes, l, m - 1);
        root.right = rebalance(nodes, m + 1, r);
        
        return root;
    }
    
}
