/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> values = new ArrayList<>();
        preOrder(values, root);
        return buildBST(values, 0, values.size() - 1);
    }

    private TreeNode buildBST(List<Integer> values, int l, int r) {
        if (l > r) {
            return null;
        }
        if (l == r) {
            return new TreeNode(values.get(l));
        }
        int mid = (l + r) / 2;
        TreeNode node = new TreeNode(values.get(mid));
        node.left = buildBST(values, l, mid-1);
        node.right = buildBST(values, mid+1, r);
        return node;
    }

    private void preOrder(List<Integer> values, TreeNode root) {
        if (root == null) {
            return;
        }
        preOrder(values, root.left);
        values.add(root.val);
        preOrder(values, root.right);
    }
}
