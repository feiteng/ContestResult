/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> list = new ArrayList<>();
    public TreeNode balanceBST(TreeNode root) {
        getList(root);
        return construct(0, list.size() - 1);
    }
    
    private void getList(TreeNode root) {
        if(root == null) return;
        getList(root.left);
        list.add(root.val);
        getList(root.right);
    }
    
    private TreeNode construct(int start, int end) {
        if(start == end) return new TreeNode(list.get(start));
        if(start > end) return null;
        int i = start + (end - start) / 2;
        TreeNode node = new TreeNode(list.get(i));
        node.left = construct(start, i - 1);
        node.right = construct(i + 1, end);
        return node;
    }
}
