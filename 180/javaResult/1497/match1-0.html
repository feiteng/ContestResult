class CustomStack {

   int[] data = null;
        int top = 0;
        int capacity = 0;

        public CustomStack(int maxSize) {
            data = new int[maxSize];
            top = 0;
            capacity = maxSize;
        }

        public void push(int x) {
            if (top < data.length) {
                data[top] = x;
                top++;
            }
        }

        public int pop() {
            int retVal = -1;
            if (top > 0) {
                top--;
                retVal = data[top];
            }
            return retVal;
        }

        public void increment(int k, int val) {
            for (int idx = 0; idx < Math.min(k, top); idx++) {
                data[idx] += val;
            }
        }
}

 * Your CustomStack object will be instantiated and called as such:

 * obj.increment(k,val);
