class CustomStack {

    List<Integer> deque;
    int maxSize;
    
    public CustomStack(int maxSize) {
        deque = new ArrayList<>();
        this.maxSize = maxSize;
    }
    
    public void push(int x) {
        if (deque.size() < maxSize) {
            deque.add(x);
        }
    }
    
    public int pop() {
        if (deque.size() == 0) {
            return -1;
        }
        int val = deque.get(deque.size() - 1);
        deque.remove(deque.size() - 1);
        return val;
    }
    
    public void increment(int k, int val) {
        k = 
Math.min(k, deque.size());
        for (int i = 0; i < k; i++) {
            deque.set(i, deque.get(i) + val);
        }
    }
}
