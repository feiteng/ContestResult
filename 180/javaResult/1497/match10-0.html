	class CustomStack {

		int[] arr;
		int top = 0;
		public CustomStack(int maxSize) {
			arr = new int[maxSize];
			top = 0;
		}

		public void push(int x) {
			if (top == arr.length) return;
			arr[top++] = x;
		}

		public int pop() {
			if (top > 0) {
				return arr[--top];
			}
			return -1;
		}

		public void increment(int k, int val) {
			for (int i = 0; i < k && i < top; i++) {
				arr[i] += val;
			}
		}
	}

 * obj.increment(k,val);
