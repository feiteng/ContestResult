class CustomStack {
    private int[] vals;
    private int maxSize;
    private int curSize;
    
    public CustomStack(int maxSize) {
        this.vals = new int[maxSize];
        this.maxSize = maxSize;
        this.curSize = 0;
    }
    
    public void push(int x) {
        if (this.curSize < this.maxSize) {
            this.vals[this.curSize++] = x;
        }
    }
    
    public int pop() {
        if (this.curSize > 0) {
            return this.vals[--this.curSize];
        
}
        return -1;
    }
    
    public void increment(int k, int val) {
        for (int i = 0; i < Math.min(k, this.curSize); i++) {
            this.vals[i] += val;
        }
    }
}
