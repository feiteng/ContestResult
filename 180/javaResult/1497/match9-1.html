class CustomStack {

    private int[] stack;
    private int p;


    public CustomStack(int maxSize) {
        stack = new int[maxSize];
        p = 0;

    }

    public void push(int x) {
        if (p < stack.length) {
            stack[p++] = x;
        }
    }

    public int pop() {
        if (p == 0) {
            return -1;
        } else {
            return stack[--p];
        }

    
}

    public void increment(int k, int val) {
        for (int i = 0; i < k && i < p; i++) {
            stack[i] += val;
        }
    }
}
