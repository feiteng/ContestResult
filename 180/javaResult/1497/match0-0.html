class CustomStack {

    private int[] res;
    private int id = 0, n;
    
    public CustomStack(int maxSize) {
        res = new int[maxSize];
        id = 0;
        n = maxSize;
    }
    
    public void push(int x) {
        if(id < n){
            res[id++] = x;
        }
    }
    
    public int pop() {
        if(id == 0){
            return -1;
        }
        return res[--id];
    }
    
    public void increment(int k, int val) {
        for(int i = 0; i < Math.min(k, n); i++){
            res[i] += val;
        }
    }
}

 * CustomStack obj = new CustomStack(maxSize);

 * obj.increment(k,val);
