class CustomStack {

    int[] data;
    int size = 0;

    public CustomStack(int maxSize) {
        data = new int[maxSize];
    }

    public void push(int x) {
        if (size < data.length) {
            data[size++] = x;
        }
    }

    public int pop() {
        if (size > 0) {
            return data[--size];
        } else 
{
            return -1;
        }
    }

    public void increment(int k, int val) {
        for (int i = 0; i < data.length && i < k; i++) {
            data[i] += val;
        }
    }
}

 * obj.increment(k,val);
