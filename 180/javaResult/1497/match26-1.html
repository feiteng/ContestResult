class CustomStack {
    int[] stack;
    int index;
    public CustomStack(int maxSize) {
        stack = new int[maxSize];
        index = -1;
    }
    
    public void push(int x) {
        if (index < stack.length - 1) {
            index++;
            stack[index] = x;
        }
    }
    
    public int pop() {
        if (index == -1) {
            return -1;
        }
        int ret = stack[index];
        index--;
        return ret;
    
}
    
    public void increment(int k, int val) {
        for (int i = 0; i < k && i <= index; i++) {
            stack[i] += val;
        }
    }
}

 * obj.increment(k,val);
