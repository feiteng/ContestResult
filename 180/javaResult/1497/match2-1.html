class CustomStack {

    Deque<Integer> que;
    int max = 0;
    Deque<Integer> tmp;
    
    public CustomStack(int maxSize) {
        que = new LinkedList<>();
        max = maxSize;
        tmp = new LinkedList<>();
    }
    
    public void push(int x) {
        if(max <= que.size())
            return;
        que.addLast(x);
    }
    
    public int pop() {
        if(que.isEmpty())
            return -1;
        return que.pollLast();
    }
    
    public void increment(int k, int val) {
        int cnt = 0;
        while(!que.isEmpty() && cnt++ < k){
            tmp.addFirst(que.pollFirst());
        }
        while(!tmp.isEmpty()){
            que.addFirst(tmp.pollFirst() + val);
        }
    }
}

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack obj = new CustomStack(maxSize);
 * obj.push(x);
 * int param_2 = obj.pop();
 * obj.increment(k,val);
