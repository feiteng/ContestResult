class Solution {
    
public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        long mod = 1000000007;
        List<Person> eng = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            eng.add(new Person(speed[i], efficiency[i]));
        }
        Collections.sort(eng, (p1, p2) -> {
            if (p1.eff < p2.eff || (p1.eff == p2.eff && p1.speed < p2.speed)) {
                return 1;
            }
            if (p1.eff == p2.eff && p1.speed == p2.speed) {
                return 0;
            }
            return -1;
        });
        // System.out.println(eng);
        long ans = 0;
        long sumSpeed = 0;
        // PriorityQueue<Person> maxSpeed = new PriorityQueue<>(k, (p1, p2) -> p2.speed - p1.speed);
        PriorityQueue<Person> minSpeed = new PriorityQueue<>(n, (p1, p2) -> p1.speed - p2.speed);

        for (int i = 0; i < n; i++) {
            Person e = eng.get(i);
            
minSpeed.add(e);
            sumSpeed += e.speed;
            if (minSpeed.size() > k) {
                sumSpeed -= minSpeed.poll().speed;
            }
            ans = Math.max(ans, sumSpeed * e.eff);
        }
        return (int)(ans % mod);
    }
    
    
private class Person {
        int speed;
        int eff;
        public Person(int speed, int eff) {
            this.speed = speed;
            this.eff = eff;
        }
        
        public String toString() {
            return this.eff + " " + this.speed;
        }
    }
}
