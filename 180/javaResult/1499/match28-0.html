class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
            long[][] es = new long[n][2];
    for (int i = 0; i < n; i++) {
      es[i][0] = efficiency[i];
      es[i][1] = speed[i];
    }
    Arrays.sort(es, new Comparator<long[]>() {
      @Override
      public int compare(long[] o1, long[] o2) {
        if (o1[0] != o2[0]) {
          return Long.compare(o2[0], o1[0]);
        } else {
          return Long.compare(o2[1], o1[1]);
        }
      }
    });
    // for (long[] v : es) {
    //   System.out.println(Arrays.toString(v));
    // }
    PriorityQueue<Long> pq = new PriorityQueue<>();
    long maxv = 0;
    long sum = 0;
    for (long[] v : es) {
      if (pq.size() >= k && pq.peek() < v[1]) {
        long min = pq.poll();
        sum -= min;
        // System.out.format("rem %2d sum %2d\n", min, sum);
      }
      if (pq.size() < k) {
        pq.add(v[1]);
        sum += v[1];
        // System.out.format("add %2d sum %2d\n", v[1], sum);
      }
      long value = v[0] * sum;
      if (value > maxv) {
        maxv = value;
      }
    }
    return (int) (maxv % 1000000007L);
    }
}
