class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][] fn = new int[n][2];
        for (int idx=0;idx<n;idx++) {
            fn[idx][0] = speed[idx];
            fn[idx][1] = efficiency[idx];
        }
        Arrays.sort(fn, new Comparator<int[]>() {
            @Override
            public int compare(int[] x, int[] y) {
                return (y[1] - x[1]);
            }
        });
        System.out.println(fn);
        Queue<Integer> pQue = new PriorityQueue<Integer>(
                new Comparator<Integer>(){
                    public int compare(Integer x, Integer y) {
                        return (x-y);
                    }
                });
        int cnt = 0;
        long ans = 0;
        long max_sum = 0;
        for (int idx=0;idx<n;idx++) {
            System.out.println(fn[idx][0] +":" + fn[idx][1]);
            if (pQue.size() == k) {
                if (pQue.peek() < fn[idx][0]) {
                    max_sum -=pQue.poll();
                }
            }
            if (pQue.size()<k) {
                pQue.add(fn[idx][0]);
                max_sum += fn[idx][0];
            }

            ans = Math.max(ans, max_sum * fn[idx][1]);
        }
        return (int) (ans % 1000000007);
    }
}
