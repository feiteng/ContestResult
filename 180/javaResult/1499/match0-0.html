class pair {
    int s, e;
    pair (int x, int y) {
        s = x;
        e = y;
    }
}
//speed = [2,10,3,1,5,8]
// efficie[5,4,3,9,7,2]
class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        List<pair> list = new ArrayList<>();
        for (int i = 0; i < speed.length; i++) {
            list.add(new pair(speed[i], efficiency[i]));
        }
        Collections.sort(list, (a, b) -> (a.e - b.e));
        Queue<Integer> q = new PriorityQueue<>();
        long sum = 0, total = 0;
        for (int i = n-1; i>= 0; i--) {
            sum+= list.get(i).s;
            q.offer(list.get(i).s);
            while (q.size() > k) sum-= q.poll();
            total = Math.max(total, sum * list.get(i).e);
        }
        return (int)(total % 1000000007);
    }
}
