class Solution {
    final static int MOD = 1000000007;
    
    
class Person{
        int speed;
        int eff;
        
        public Person(int speed, int eff) {
            this.speed = speed;
            this.eff = eff;
        }
    }
    
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        Person[] persons = new Person[n];
        for(int i = 0; i < n; i++) {
            persons[i] = new Person(speed[i], efficiency[i]);
        }
        Arrays.sort(persons, (a, b) -> (a.eff == b.eff ? b.speed - a.speed : b.eff - a.eff));
        
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> (a - b));
        long sum = 0;
        int eff = 0;
        long res = 0;
        for(int i = 0; i < n; i++) {
            Person person = persons[i];
            eff = person.eff;
            
if(pq.size() < k) {
                sum += person.speed;
                pq.offer(person.speed);
            }
            else {
                if(person.speed > pq.peek()) {
                    sum -= pq.poll();
                    sum += person.speed;
                    pq.offer(person.speed);
                }
            }
            res = Math.max(eff * sum, res);
            // System.out.print(person.speed + " " + person.eff + " " + sum + " " + res);
        }
        return (int)(res % MOD);
    }
}
