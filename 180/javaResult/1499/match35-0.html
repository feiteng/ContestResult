class Solution {
    class Node{
        long eff, speed;
        public Node(int e, int s){
            eff = e;
            speed = s;
        }
    }
    
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        List<Node> list = new ArrayList<>();
        for(int i = 0; i < n; i++){
            list.add(new Node(efficiency[i], speed[i]));
        }
        Collections.sort(list, new Comparator<Node>(){
            @Override
            public int compare(Node a, Node b){
                if(b.eff == a.eff)
                    return 0;
                return b.eff - a.eff > 0 ? 1 : -1;
            }
        });
        PriorityQueue<Node> qu = new PriorityQueue<>(new Comparator<Node>(){
            @Override
            public int compare(Node a, Node b){
                if(a.speed == b.speed)
                    return 0;
                return a.speed - b.speed > 0 ? 1 : -1;
            }
        });
        long res = 0;
        long sum = 0, max = list.get(0).eff;
        int base = 1000000007;
        for(int i = 0; i < n; i++){
            Node node = list.get(i);
            max = Math.min(max, node.eff);
            sum += node.speed;
            qu.offer(node);
            if(qu.size() > k){
                sum -= qu.poll().speed;
            }
            //sum = (sum % base + base) % base;
            //max = max % base;
            res = Math.max(res, sum * max);
        }
        return (int)(res % base);
    }
}
