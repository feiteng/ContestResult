class Solution {
    int MOD = 1_000_000_007;
    
    
public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
//         if(k == 1){
//             long res = 0;
//             for(int i = 0; i < n; i++){
//                 res = Math.max((long) speed[i] * efficiency[i]);
//             }
            
//             return (int) (res % MOD);
//         }
        
        // sort by efficiency
        Integer[] order = new Integer[n];
        for(int i = 0; i < n; i++){
            order[i] = i;
        }
        
        Arrays.sort(order, new Comparator<Integer>(){
            public int compare(Integer i1, Integer i2){
                return Integer.compare(efficiency[i1], efficiency[i2]);
            }
        });
        
        int[] temp = Arrays.copyOf(speed, n);
        for(int i = 0; i < n; i++){
            speed[i] = temp[order[i]];
        }
        Arrays.sort(efficiency);
        // end of sort
        
        
PriorityQueue<Integer> speeds = new PriorityQueue<>();
        long sum = 0;
        long res = 0;
        for(int i = n - 1; i >= n - k; i--){
            speeds.add(speed[i]);
            sum += speed[i];
            res = Math.max(res, sum * efficiency[i]);
        }
        
        for(int i = n - k - 1; i >= 0; i--){
            int cur = speed[i];
            int smallest = speeds.poll();
            sum -= smallest;
            sum += Math.max(cur, smallest);
            
speeds.add(Math.max(cur, smallest));
            res = Math.max(res, sum * efficiency[i]);
        }
        
        return (int) (res % MOD);
    }
}
