class Solution {
    class Guy {
        public int speed;
        public int efficiency;
        
        public Guy(int speed, int efficiency) {
            this.speed = speed;
            this.efficiency = efficiency;
        }
    }
    
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        List<Guy> guys = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            guys.add(new Guy(speed[i], efficiency[i]));
        }
        
        Collections.sort(guys, (a, b) -> b.efficiency - a.efficiency);
        
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        int count = 0;
        long answer = 0;
        long sum = 0;
        for (int i = 0; i < n; i++) {
            Guy guy= guys.get(i);
            if (count < k) {
                queue.add(guy.speed);
                sum += guy.speed;
                count++;
            } else {
                queue.add(guy.speed);
                int out = queue.poll();
                sum = sum + guy.speed - out;
            }
            answer = Math.max(answer, sum * guy.efficiency);
        }
        
        return (int) (answer % 1000000007L);
    }
}
