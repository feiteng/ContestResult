class Solution {
    private static long MODULE = 1000000007L;

    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][] speedAndEfficiency = new int[speed.length][3];
        for (int idx = 0; idx < speed.length; idx++) {
            speedAndEfficiency[idx][0] = speed[idx];
            speedAndEfficiency[idx][1] = efficiency[idx];
            speedAndEfficiency[idx][2] = idx;
        }
        Arrays.sort(speedAndEfficiency, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[1] - o2[1];
            }
        });


        // Queue<Integer> minQueue = new PriorityQueue<>();
        Queue<Integer> minQueue = new PriorityQueue<>();
        long speedSum = 0;
        long performance = -1;
        //initial
        for (int idx = n - 1; idx >= n - k; idx--) {
            speedSum += speedAndEfficiency[idx][0];
            minQueue.offer(speedAndEfficiency[idx][0]);
            if (speedSum * speedAndEfficiency[idx][1] > performance) {
                performance = speedSum * speedAndEfficiency[idx][1];
            }
        }
        for (int idx = n - k - 1; idx >= 0; idx--) {
            minQueue.offer(speedAndEfficiency[idx][0]);
            speedSum = speedSum + speedAndEfficiency[idx][0] - minQueue.poll();
            if (speedSum * speedAndEfficiency[idx][1] > performance) {
                performance = speedSum * speedAndEfficiency[idx][1];
            }
        }
        return (int) (performance % MODULE);
    }
}
