class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][] arr = new int[n][2];
        long mod = 1000000007;
        
for(int i = 0; i < n; i++){
            arr[i][0] = speed[i];
            arr[i][1] = efficiency[i];
        }
        
        Arrays.sort(arr, (i1, i2) -> (i2[1] - i1[1]));
        // System.out.println(Arrays.deepToString(arr));
        PriorityQueue<int[]> pq = new PriorityQueue<>((i1, i2) -> (i1[0] - i2[0]));
        long cur = 0, res = 0;
        
for(int i = 0; i < k; i++){
            cur += arr[i][0];
            pq.add(arr[i]);
            res = Math.max(res, arr[i][1] * cur);
        }
        
        // System.out.println(res);
        for(int i = k; i < n; i++){
            if(pq.peek()[0] < arr[i][0]){
                cur -= pq.poll()[0];
                pq.add(arr[i]);
                cur += arr[i][0];
                res = Math.max(res, cur * arr[i][1]);
            }
        }
        return (int)(res % mod);
    }
}
