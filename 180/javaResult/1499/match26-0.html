class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][] entries = new int[n][2];
        for (int i = 0; i < n; i += 1) {
            entries[i][0] = speed[i];
            entries[i][1] = efficiency[i];
        }
        Arrays.sort(entries, (e1, e2) -> -Integer.compare(e1[1], e2[1]));
        PriorityQueue<Integer> pq = new PriorityQueue();
        long res = 0, mod = 1000000007, curSum = 0;
        for (int i = 0; i < n; i += 1) {
            pq.offer(entries[i][0]);
            curSum += entries[i][0];
            if (pq.size() > k) {
                int s = pq.poll();
                curSum -= s;
            }
            res = Math.max(res, curSum * (long)entries[i][1]);
        }
        return (int)(res % mod);
    }
}
