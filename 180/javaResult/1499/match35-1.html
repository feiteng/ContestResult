class Solution {
    private static final long MOD = (long)(1e9+7);
    
public class Person {
        long s;
        long e;
        Person(long s, long e) {
            this.s = s;
            this.e = e;
        }
    }
    
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        List<Person> ps = new ArrayList<>();
        for(int i = 0; i < n; i++) {
            ps.add(new Person(speed[i], efficiency[i]));
        }
        Collections.sort(ps, (a, b) -> (int)(b.e - a.e));
        
        Queue<Person> que = new PriorityQueue<Person>((a, b) -> (int)(a.s - b.s));
        long cure = 0;
        long sum = 0;
        long res = 0;
        for(Person p : ps) {
            cure = p.e; 
            sum += p.s;
            
que.offer(p);
            if(que.size() > k) {
                sum -= que.poll().s;
            }
            res = Math.max(res, sum * cure);
        }
        return (int) (res%MOD);
    }
    
}
