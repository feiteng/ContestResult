class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
		Integer[] idx = new Integer[n];
		for (int i = 0; i < n; i++) idx[i] = i;
		Arrays.sort(idx, new Comparator<Integer>() {
			@Override
			public int compare(Integer o1, Integer o2) {
				return efficiency[o2] - efficiency[o1];
			}
		});
		TreeMap<Integer, Integer> selected = new TreeMap<>();
		long res = 0;
		long tot = 0;
		for (int i = 0; i < n; i++) {
			long e = efficiency[idx[i]];
			int s = speed[idx[i]];
			if (i < k) {
				tot += s;
				selected.put(s, selected.getOrDefault(s, 0) + 1);
			} else if (selected.firstKey() < speed[idx[i]]) {
				tot -= selected.firstKey();
				Integer fisrt = selected.firstKey();
				selected.put(fisrt, selected.get(fisrt) - 1);
				if (selected.get(fisrt) == 0) selected.remove(fisrt);
				tot += s;
				selected.put(s, selected.getOrDefault(s, 0) + 1);
			}
			res = Math.max(res, tot * e);
		}
		return (int) (res % 1000000007);
	}
}
