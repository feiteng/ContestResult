class Solution {
    private static final int MOD = 1_000_000_007;

    public int maxPerformance(final int N, int[] speed, int[] efficiency, final int K) {
        Node[] nodes = new Node[N];
        for (int i = 0; i < N; ++i) {
            nodes[i] = new Node(speed[i], efficiency[i]);
        }

        Arrays.sort(nodes, (a, b) -> a.efficiency - b.efficiency);
        PriorityQueue<Node> minHeap = new PriorityQueue<>((a, b) -> a.speed - b.speed);

        long maxPerf = 0L;
        long speedSum = 0L;
        for (int i = 1; i <= K; ++i) {
            speedSum += nodes[N - i].speed;
            minHeap.offer(nodes[N - i]);
            maxPerf = Math.max(maxPerf, speedSum * nodes[N - i].efficiency);
        }

        // maxPerf = Math.max(maxPerf, speedSum * nodes[N - K].efficiency);
        for (int i = N - K - 1; i >= 0; --i) {
            speedSum -= minHeap.poll().speed;
            speedSum += nodes[i].speed;
            minHeap.offer(nodes[i]);
            maxPerf = Math.max(maxPerf, speedSum * nodes[i].efficiency);
        }

        return (int) (maxPerf % MOD);
    }

    static class Node {
        int speed;
        int efficiency;

        Node(int speed, int efficiency) {
            this.speed = speed;
            this.efficiency = efficiency;
        }
    }
}
