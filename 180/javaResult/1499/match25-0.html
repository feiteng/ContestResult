import java.util.*;

class Solution {

    class Candidate {
        long speed;
        int efficiency;
        public Candidate(int speed, int efficiency) {
            this.speed=speed;
            this.efficiency=efficiency;
        }
    }

    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {

        Candidate[] candidates = new Candidate[n];
        for(int i = 0; i < n; i++) {
            candidates[i] = new Candidate(speed[i], efficiency[i]);
        }
        Arrays.sort(candidates, (a,b)-> -1*(a.efficiency - b.efficiency));

        PriorityQueue<Long> topSpeeds = new PriorityQueue<>();

        long sumSpeeds = candidates[0].speed;
        topSpeeds.add(candidates[0].speed);
        long currentEfficiency = candidates[0].efficiency;
        long currentProd = sumSpeeds * currentEfficiency;

        for(int idx = 1; idx < n; idx++) {

            long incomingSpeed = candidates[idx].speed;
            if (topSpeeds.size() < k) {
                topSpeeds.add(incomingSpeed);
                sumSpeeds += incomingSpeed;
            }
            else {
                if (topSpeeds.peek() < incomingSpeed) {
                    sumSpeeds -= topSpeeds.poll();
                    topSpeeds.add(incomingSpeed);
                    sumSpeeds += incomingSpeed;
                }
            }

            currentEfficiency = candidates[idx].efficiency;
            currentProd = Math.max(currentProd, sumSpeeds * currentEfficiency);
        }

        return (int)(currentProd % 1000000007L);
    }
}
