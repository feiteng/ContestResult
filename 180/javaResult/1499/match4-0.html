class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        long MOD = 1_000_000_007;
        List<Integer> idx = new ArrayList();
        for (int i = 0; i < n; i++) {
            idx.add(i);
        }
        Collections.sort(idx, (a, b) -> efficiency[b] - efficiency[a]);
        List<Integer> speedList = new ArrayList();
        List<Integer> effList = new ArrayList();
        for (int i = 0; i < n; i++) {
            speedList.add(speed[idx.get(i)]);
            effList.add(efficiency[idx.get(i)]);
        }
        for (int i = 0; i < n; i++) {
            speed[i] = speedList.get(i);
            efficiency[i] = effList.get(i);
        }
        long max = 0, maxSum = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
        for (int i = 0; i < k; i++) {
            maxSum += speed[i];
            pq.offer(speed[i]);
            max = Math.max(max, maxSum * efficiency[i]);
        }
        System.out.println(Arrays.toString(speed));
        System.out.println(Arrays.toString(efficiency));
        System.out.println(pq.toString());
        for (int i = k; i < n; i++) {
            if (pq.peek() >= speed[i]) {
                continue;
            }
            long curSum = maxSum + speed[i] - pq.peek();
            if (curSum > maxSum) {
                pq.poll();
                pq.offer(speed[i]);
                maxSum = curSum;
            }
            System.out.println(i + ": " + curSum);
            max = Math.max(max, maxSum * efficiency[i]);
        }
        return (int) (max % MOD);
    }
}
