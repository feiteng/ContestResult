class Solution {

    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][] pairs = new int[n][2];
        for (int i = 0; i < n; i++) {
            pairs[i] = new int[]{speed[i], efficiency[i]};
        
}
        Arrays.sort(pairs, new Comparator<int[]>() {
            @Override
            public int compare(int[] l, int[] r) {
                if (r[1] == l[1]) {
                    return r[0] - l[0];
                } else {
                    return r[1] - l[1];
                
}
            }
        });

        PriorityQueue<Integer> pq = new PriorityQueue<>(k);
        long sum = 0;
        long ans = 0;
        for (int i = 0; i < k; i++) {
            pq.offer(pairs[i][0]);
            sum += pairs[i][0];
            ans = Math.max(ans, sum * pairs[i][1]);
        }

        for (int i = k; i < n; i++) {
            int eff = pairs[i][1];
            int min = pq.peek();
            if (min < pairs[i][0]) {
                pq.offer(pairs[i][0]);
                pq.poll();
                sum -= min;
                sum += pairs[i][0];
            }
            
if (sum * eff > ans) {
                ans = sum * eff;
            }
        }
        return (int) (ans % 1000000007);
    }
}
