class Solution {
    int m = 1000000007;
    public int maxPerformance(int n, int[] sp, int[] eff, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b)->Integer.compare(sp[a], sp[b]));
        Integer[] idx = new Integer[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Arrays.sort(idx, (a,b)->Integer.compare(eff[b], eff[a]));
        long sd = 0;
        long max = 0;
        for (int i = 0; i < k; i++) {
            sd += sp[idx[i]];
            max = Math.max(sd * eff[idx[i]], max);
            pq.offer(idx[i]);
        }
        for (int i = k; i < n; i++) {
            int id = pq.poll();
            sd -= sp[id];
            int next = idx[i];
            sd += sp[next];
            pq.offer(next);
            max = Math.max(max, sd * eff[next]);
        }
        return (int) (max % m);
    }
}
