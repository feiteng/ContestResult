class Solution {
    
class State {
        int speed;
        int efficiency;
        State(int speed, int efficiency) {
            this.speed = speed;
            this.efficiency = efficiency;
        }
    }
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int MOD = (int)1e9 + 7;
        List<State> list = 
new ArrayList<>();
        for (int i = 0; i < n; i++) {
            list.add(new State(speed[i], efficiency[i]));
        }
        Collections.sort(list, (a, b) -> (a.efficiency == b.efficiency ? b.speed - a.speed : b.efficiency - a.efficiency));
        PriorityQueue<State> pq = new PriorityQueue<>((a, b) -> a.speed - b.speed);
        long sum = 0;
        long res = 0;
        for (State state : list) {
            
pq.offer(state);
            sum += state.speed;
            if (pq.size() > k) {
                sum -= pq.poll().speed;
            }
            res = Math.max(res, sum * state.efficiency);
        }
        return (int)(res % MOD);
        
    }
}
