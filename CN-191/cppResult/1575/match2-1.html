class Solution {
public:
    
    typedef long long ll ;
    ll module = 1e9 + 7 ;

    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        sort(horizontalCuts.begin() , horizontalCuts.end()) ;
        sort(verticalCuts.begin() , verticalCuts.end()) ;
        int maxh = max(horizontalCuts[0] , h - horizontalCuts[horizontalCuts.size() - 1]) ;
        int maxw = max(verticalCuts[0] , w - verticalCuts[verticalCuts.size() - 1]) ;
        
        //cout << h - horizontalCuts[horizontalCuts.size() - 1] << endl ;
        
        //cout << maxw << " " << maxh << endl ;
        
        for (int i = 1 ; i < verticalCuts.size() ; ++i) {
            maxw = max(maxw , verticalCuts[i] - verticalCuts[i - 1]) ;
        }
        for (int i = 1 ; i < horizontalCuts.size() ; ++i) {
            maxh = max(maxh , horizontalCuts[i] - horizontalCuts[i - 1]) ;
        }
        //cout << maxw << " " << maxh << endl ;
        ll res = (ll) maxw * (ll) maxh % module ;
        return res ;
    }
};
