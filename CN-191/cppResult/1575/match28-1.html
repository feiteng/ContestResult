class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) 
    {
        sort(horizontalCuts.begin(),horizontalCuts.end());
        sort(verticalCuts.begin(),verticalCuts.end());
        int r=horizontalCuts[0]-0,c=verticalCuts[0]-0;
        for(int i=1;i<horizontalCuts.size();i++)
            r=max(r,horizontalCuts[i]-horizontalCuts[i-1]);
        r=max(r,h-horizontalCuts.back());
        for(int i=1;i<verticalCuts.size();i++)
            c=max(c,verticalCuts[i]-verticalCuts[i-1]);
        c=max(c,w-verticalCuts.back());
        int mo=1e9+7;
        long long res=((long long)r)*c%mo;
        return res;
    }
};
