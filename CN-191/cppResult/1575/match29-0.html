class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        sort(hc.begin(), hc.end());
        sort(vc.begin(), vc.end());
        set<int> hs, vs;
        int mx = 0;
        for (int i = 0; i < hc.size(); ++i) {
            hs.insert(hc[i] - mx);
            mx = hc[i];
        }
        hs.insert(h - mx);
        mx = 0;
        for (int i = 0; i < vc.size(); ++i) {
            vs.insert(vc[i] - mx);
            mx = vc[i];
        }
        vs.insert(w - mx);
        return ((long long)*hs.rbegin()) * (*vs.rbegin()) % int(1e9+7);
    }
};
