const int mod = 1000000007;

class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        horizontalCuts.push_back(0);
        verticalCuts.push_back(0);
        sort(horizontalCuts.begin(), horizontalCuts.end());
        sort(verticalCuts.begin(), verticalCuts.end());
        int x = h -  horizontalCuts.back(), y = w - verticalCuts.back();
        for (int i = 1, n = horizontalCuts.size(); i < n; ++i) x = max(x, horizontalCuts[i] - horizontalCuts[i-1]);
        for (int i = 1, n = verticalCuts.size(); i < n; ++i) y = max(y, verticalCuts[i] - verticalCuts[i-1]);
        return ((long long)x * y) % mod;
    }
};
