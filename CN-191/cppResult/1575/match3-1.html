class Solution {
public:
    int maxArea(int h, int w, vector<int>& hv, vector<int>& wv) {

        sort(hv.begin(),hv.end());
        sort(wv.begin(),wv.end());
        int x = max(hv[0],h-hv.back());
        int y = max(wv[0],w-wv.back());
        for(int i = 0;i + 1 < hv.size();i++)
            x = max(x,hv[i+1] - hv[i]);
        for(int j = 0;j + 1 < wv.size();j++)
            y = max(y,wv[j+1] - wv[j]);
        return 1ll * x * y % 1000000007;
    }
};
