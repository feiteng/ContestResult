class Solution {
public:
    int maxArea(int h, int w, vector<int>& z1, vector<int>& z2) {
        sort(z1.begin(), z1.end());
        sort(z2.begin(), z2.end());
        z1.insert(z1.begin(),0);
        z1.push_back(h);
        z2.insert(z2.begin(),0);
        z2.push_back(w);
        int r1 = 0, r2 = 0;
        for (int i=1; i<z1.size(); ++i) {
            r1 = max(r1, z1[i] - z1[i-1]);
        }
        
        for (int i=1; i<z2.size(); ++i) {
            r2 = max(r2, z2[i] - z2[i-1]);
        }
        long long res = r1;
        res *= r2;
        return res % 1000000007;
    }   
};
