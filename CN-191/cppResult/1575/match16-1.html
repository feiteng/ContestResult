class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        int x = 0, y = 0;
        int n = hc.size();
        int m = vc.size();
        sort(hc.begin(),hc.end());
        sort(vc.begin(),vc.end());

        x = hc[0], y = vc[0];
        for(int i=1;i<n;++i){
            x = max(x,hc[i]-hc[i-1]);
        }
        for(int i=1;i<m;++i){
            y = max(y,vc[i]-vc[i-1]);
        }
        x = max(x, h-hc[n-1]);
        y = max(y, w-vc[m-1]);
        int mod = 1e9+7;
        return x*1LL*y % mod;
    }
};
