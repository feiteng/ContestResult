class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        hc.push_back(0);
        hc.push_back(h);
        vc.push_back(0);
        vc.push_back(w);
        sort(hc.begin(), hc.end());
        sort(vc.begin(), vc.end());
        int hmax = 0, vmax = 0;
        for(int i = 1; i < hc.size(); ++i)
            hmax = max(hmax, hc[i] - hc[i-1]);
        for(int j = 1; j < vc.size(); ++j)
            vmax = max(vmax, vc[j] - vc[j-1]);
        long hm = hmax, vm = vmax, mod = 1e9 + 7;
        return (hm * vm) % mod;
    }
};
