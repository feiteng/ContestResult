class Solution {
public:
    int maxArea(int h, int w, vector<int>& hor, vector<int>& ver) {
        hor.insert(hor.begin(), 0);
        hor.push_back(h);
        ver.insert(ver.begin(), 0);
        ver.push_back(w);
        sort(hor.begin(), hor.end());
        sort(ver.begin(), ver.end());
        int n = hor.size(), m = ver.size();
        int mh = 0, mw = 0;
        for (int i = 1; i < n; i++) {
            mh = max(mh, hor[i] - hor[i - 1]);
        }
        for (int i = 1; i < m; i++) {
            mw = max(mw, ver[i] - ver[i - 1]);
        }
        return (int)((1LL * mh * mw) % ((int)1e9 + 7));
    }
};
