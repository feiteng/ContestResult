class Solution {
public:
    int maxArea(int h, int w, vector<int>& A, vector<int>& B) {
        sort(A.begin(),A.end());
        sort(B.begin(),B.end());
        int n=A.size();
        int maxa=max(A[0],h-A[n-1]);
        for (int i=1;i<n;++i) maxa=max(maxa,A[i]-A[i-1]);
        int m=B.size();
        int maxb=max(B[0],w-B[m-1]);
        for (int i=1;i<m;++i) maxb=max(maxb,B[i]-B[i-1]);
        //printf("%d %d\n",maxa,maxb);
        int MOD=1e9+7;
        int ans=1LL*maxa*maxb%MOD;
        return ans;
    }
};
