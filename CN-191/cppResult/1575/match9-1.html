class Solution {
public:
    int maxArea(int h, int w, vector<int>& hcs, vector<int>& vcs) {

        int Mod=1e9 + 7;
        sort(hcs.begin(),hcs.end());
        sort(vcs.begin(),vcs.end());
        hcs.push_back(h);
        vcs.push_back(w);
        int mh=hcs[0],mw=vcs[0];
        for (int i=1;i<hcs.size();i++) {
            mh=max(mh,hcs[i]-hcs[i-1]);
        }
        for (int i=1;i<vcs.size();i++) {
            mw=max(mw,vcs[i]-vcs[i-1]);
        }
        return 1LL*mh*mw%Mod;
    }
};
