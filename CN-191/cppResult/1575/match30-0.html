class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        using i64 = long long;
        constexpr i64 MODULO = 1e9+7;
        horizontalCuts.push_back(0);horizontalCuts.push_back(h);
        verticalCuts.push_back(0);verticalCuts.push_back(w);
        sort(horizontalCuts.begin(),horizontalCuts.end());
        sort(verticalCuts.begin(),verticalCuts.end());
        int hori = 0, vert = 0;
        for(int i=1;i<horizontalCuts.size();++i){
            hori = max(horizontalCuts[i]-horizontalCuts[i-1],hori);
        }
        for(int i=1;i<verticalCuts.size();++i){
            vert = max(verticalCuts[i]-verticalCuts[i-1],vert);
        }
        i64 result = 1; result *= hori; result *= vert; result %= MODULO;
        return result;
    }
};
