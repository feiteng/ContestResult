class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        sort(hc.begin(), hc.end());
        sort(vc.begin(), vc.end());
        int mh = hc[0], mw = vc[0];
        for(int i = 1; i < hc.size(); i++) {
            mh = max(mh, hc[i] - hc[i - 1]);
        }
        mh = max(mh, h - hc.back());
        for(int i = 1; i < vc.size(); i++) {
            mw = max(mw, vc[i] - vc[i - 1]);
        }
        mw = max(mw, w - vc.back());
        long long mmh = mh, mmw = mw, M = 1e9 + 7;
        long long res = mmh * mmw % M;
        return res;
    }
};
