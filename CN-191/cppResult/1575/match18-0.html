class Solution {
public:
    int maxArea(int h, int w, vector<int>& hArr, vector<int>& vArr) 
    {
        hArr.push_back(0);
        hArr.push_back(h);
        sort(hArr.begin(), hArr.end());
        vArr.push_back(0);
        vArr.push_back(w);
        sort(vArr.begin(), vArr.end());
        
        int dH = 0;
        for (int i = 0; i+1 < hArr.size(); i++)
        {
            dH = max(dH, hArr[i+1]-hArr[i]);
        }
        int dW = 0;
        for (int i = 0; i+1 < vArr.size(); i++)
        {
            dW = max(dW, vArr[i+1]-vArr[i]);
        }
        
        unsigned long long ans = dH;
        ans *= dW;
        ans %= 1000000007;
        return ans;
    }
};
