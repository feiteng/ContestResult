#include <bits/stdc++.h>

int a[200001], b[200001];
class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        int n = (int )horizontalCuts.size();
        int m = (int )verticalCuts.size();
        for (int i = 0; i < n; i ++) a[i + 1] = horizontalCuts[i];
        for (int i = 0; i < m; i ++) b[i + 1] = verticalCuts[i];
        sort(a + 1, a + 1 + n);
        sort(b + 1, b + 1 + m);
        a[0] = 0; a[n + 1] = h;
        int ans1 = 0, ans2 = 0;
        b[0] = 0; b[m + 1] = w;
        for (int i = 0; i <= n; i ++) ans1 = max(ans1, a[i + 1] - a[i]);
        for (int i = 0; i <= m; i ++) ans2 = max(ans2, b[i + 1] - b[i]);
        return 1LL * ans1 * ans2 %  ((int )(1e9) + 7);
    }
};
