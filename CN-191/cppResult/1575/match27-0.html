        long long rest = 1000000007;

            pre_horizontal.push_back(horizontalCuts[i]);
        }
        pre_horizontal.push_back(h);
        
        pre_vertical.push_back(0);
        for (int i = 0; i < verticalCuts.size(); i++)
        {
            pre_vertical.push_back(verticalCuts[i]);
        }
        pre_vertical.push_back(w);
        
        long long h_diff = -1;
        for (int i = 0; i < pre_horizontal.size() - 1; i++)
        {
            long long diff = pre_horizontal[i+1] - pre_horizontal[i];
            h_diff = max(diff, h_diff);
        }
        
        long long w_diff = -1;
        for (int i = 0; i < pre_vertical.size() - 1; i++)
        {
            long long diff = pre_vertical[i+1] - pre_vertical[i];
            w_diff = max(diff, w_diff);
        }
        long long max_value = w_diff * h_diff % rest;
        return max_value;
