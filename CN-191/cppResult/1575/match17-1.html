class Solution {
public:
    int maxArea(int hi, int wi, vector<int>& h, vector<int>& w) {
        int lh=0,lw=0;
        int hh=0,ww=0;
        sort(h.begin(),h.end());
        sort(w.begin(),w.end());
        h.push_back(hi);
        w.push_back(wi);
        for (int i=0; i<h.size(); i++)
        {
            hh=std::max(hh,h[i]-lh);
            lh=h[i];
        }
        for (int i=0; i<w.size(); i++)
        {
            ww=std::max(ww,w[i]-lw);
            lw=w[i];
        }
        return (long long)hh*(long long)ww%1000000007LL;
    }
};
