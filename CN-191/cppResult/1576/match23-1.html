class Solution {
public:
    typedef pair<int, int> PII;
    int n;
    vector<PII> E[50050];
    int dfs(int u, int fa) {
        int cnt = 0;
        for (auto &v : E[u]) {
            if (v.first != fa) {

                cnt += dfs(v.first, u);
                cnt += v.second;
            }
        }
        return cnt;
    }
    int minReorder(int n, vector<vector<int>>& cc) {
        for (int i = 0; i < n; i++) E[i].clear();
        for (auto &x : cc) {
            int p = x[0], q = x[1];
            E[p].emplace_back(q, 1);
            E[q].emplace_back(p, 0);
        }
        return dfs(0, -1);
    }
};
