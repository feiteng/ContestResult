class Solution {
public:
	vector<set<int>> adj;
	vector<set<int>> radj;
	vector<int> visit;
	int minReorder(int n, vector<vector<int>>& connections) {
		adj = vector<set<int>>(n);
		visit = vector<int>(n, 0);
		radj = adj;
		for (auto i : connections) {
			adj[i[0]].insert(i[1]);
			radj[i[1]].insert(i[0]);
		}
		int ans = 0;
		queue<int> Q;
		Q.push(0);
		visit[0] = 1;
		while (!Q.empty()) {
			int cur = Q.front();
			Q.pop();
			for (auto i : radj[cur]) {
				if (!visit[i]) {
					visit[i] = 1;
					Q.push(i);
				}
			}
			for (auto i : adj[cur]) {
				if (!visit[i]) {
					++ans;
					visit[i] = 1;
					Q.push(i);
				}
			}
		}
		return ans;
	}
};
