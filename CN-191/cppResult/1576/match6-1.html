class Solution {
public:
    int dfs(int root, int fa, vector<vector<pair<int, bool>>>& next) {

        int ans = 0;
        for (auto& [n, flag] : next[root]) {
            if (n == fa) continue;
            if (flag) ++ans;
            ans += dfs(n, root, next);
        }
        return ans;
    }
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<pair<int, bool>>> next(n);
        for (auto& c : connections) {
            next[c[0]].emplace_back(c[1], true);
            next[c[1]].emplace_back(c[0], false);
        }
        return dfs(0, -1, next);
    }
};
