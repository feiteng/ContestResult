class Solution {
public:
    
    struct Edge {
        int v , pr ;
        Edge *next ;
    } E[1000010] , *header[50010] ;
    int tot , res = 0 ;
    bool vis[50010] ;
    
    void addEdge(int u , int v) {
        E[++tot].v = v ; E[tot].pr = 1 ; E[tot].next = header[u] ; header[u] = &E[tot] ;
        E[++tot].v = u ; E[tot].pr = 0 ; E[tot].next = header[v] ; header[v] = &E[tot] ;
    }
    
    void dfs(int u) {
       // cout << u << endl ;
        vis[u] = true ;
        for (Edge *e = header[u] ; e ; e = e->next) {
            if (!vis[e->v]) {
                if (e->pr == 0) {
                    ++res ;
                }
                dfs(e->v) ;
            }
        }
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        tot = res = 0 ;
        memset(vis , false , sizeof(vis)) ;
        for (int i = 0 ; i < connections.size() ; ++i) {
            addEdge(connections[i][0] , connections[i][1]) ;
        }
        dfs(0) ;
        return (n - 1) - res ;
    }
};
