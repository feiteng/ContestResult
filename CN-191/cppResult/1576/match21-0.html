class Solution {
public:
    void helper(vector<vector<int>>& graph, vector<unordered_set<int>>& out_nb, vector<bool>& visited, int start, int& ans) {
        visited[start] = true;
        for (auto a:graph[start]) {
            if (visited[a]) continue;
            if (out_nb[start].count(a)) ++ans;
            helper(graph, out_nb, visited, a, ans);
        }
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<int>> graph(n);
        vector<unordered_set<int>> out_nb(n);
        vector<bool> visited(n, false);
        for (auto connection:connections) {
            graph[connection[0]].push_back(connection[1]);
            graph[connection[1]].push_back(connection[0]);
            out_nb[connection[0]].insert(connection[1]);
        }
        int ans = 0;
        helper(graph, out_nb, visited, 0, ans);
        return ans;
    }
};
