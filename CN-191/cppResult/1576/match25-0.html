vector<pair<int,int> > v[50005];
class Solution {
public:
	int dfs(int x,int fa)
	{
		int ret=0;
		for (int i=0;i<v[x].size();++i)
		{
			int y=v[x][i].first;
			if (y!=fa){
				if (v[x][i].second==0) ret++;
				ret+=dfs(y,x);
			}
		}
		return ret;
	}
    int minReorder(int n, vector<vector<int>>& ee) {
		for (int i=0;i<n;++i) v[i].clear();
		for (int i=0;i<n-1;++i)
		{
			v[ee[i][0]].push_back(make_pair(ee[i][1],0));
			v[ee[i][1]].push_back(make_pair(ee[i][0],1));
		}
		return dfs(0,-1);
    }
};
