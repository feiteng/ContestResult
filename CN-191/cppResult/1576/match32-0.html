class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        auto es = vector<vector<int>>(n, vector<int>{});
        auto res = vector<vector<int>>(n, vector<int>{});
        for (auto &p: connections) {
            es[p[0]].push_back(p[1]);
            res[p[1]].push_back(p[0]);
        }
        auto q = deque<int>{0};
        auto visited = vector<int>(n, 0);
        visited[0] = 1;
        int ans = 0;
        while (!q.empty()) {
            int s = q.front();
            q.pop_front();
            for (auto t: es[s]) {
                if (!visited[t]) {
                    q.push_back(t);
                    visited[t] = 1;
                    ans ++;
                }
            }
            for (auto t: res[s]) {
                if (!visited[t]) {
                    q.push_back(t);
                    visited[t] = 1;
                }
            }
        }
        return ans;
    }
};
