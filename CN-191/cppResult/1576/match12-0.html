class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        unordered_map<int, vector<pair<int,bool>>> node_to_node_direction(n); // true 表示正向
        int sum = 0;
        vector<bool> visited(n, false);
        for (int i = 0; i < connections.size(); i++)
        {
            int begin = connections[i][0];
            int end = connections[i][1];
            node_to_node_direction[begin].push_back(make_pair(end, true));
            node_to_node_direction[end].push_back(make_pair(begin, false));
        }
        
        vector<int> queue;
        queue.push_back(0);
        visited[0] = true;
        int begin = 0; 
        int end = 1;
        
        while (begin < end)
        {
            for (int index = begin; index < end; index++)
            {
                int cur = queue[index];
                for (auto& pair_value: node_to_node_direction[cur])
                {
                    if (visited[pair_value.first] == false)
                    {
                        if (pair_value.second == true) sum++;
                        queue.push_back(pair_value.first);
                        visited[pair_value.first] = true;
                    }
                }
            }
            begin = end;
            end = queue.size();
        }
        return sum;
    }
};
