class Solution {
public:
    int minReorder(int n, vector<vector<int>>& cArr) 
    {
        vector<vector<int>> indexArr(n);
        for (int i = 0; i < cArr.size(); i++)
        {
            indexArr[cArr[i][0]].push_back(i);
            indexArr[cArr[i][1]].push_back(i);
        }
        
        vector<bool> dealArr(cArr.size(), false);
        
        vector<int> dealList;
        dealList.push_back(0);
        int ans = 0;
        for (int i = 0; i < dealList.size(); i++)
        {
            int cur = dealList[i];
            for (int j = 0; j < indexArr[cur].size(); j++)
            {
                if (dealArr[indexArr[cur][j]])
                {
                    continue;
                }
                dealArr[indexArr[cur][j]] = true;
                if (cArr[indexArr[cur][j]][0] == cur)
                {
                    ans += 1;
                    dealList.push_back(cArr[indexArr[cur][j]][1]);
                    //cout << cArr[indexArr[cur][j]][0] << "   " << cArr[indexArr[cur][j]][1] << "\n"; 
                }
                else
                {
                    dealList.push_back(cArr[indexArr[cur][j]][0]);
                }
            }
        }
        return ans;
    }
};
