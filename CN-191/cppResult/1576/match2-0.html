class Solution {
public:
    struct E{
        int next,to,val;
    }e[100004];
    int head[50004],tot = 0,ans = 0;
    void add(int x,int y,int z){
        e[tot] = E{y,head[x],z};
        head[x] = tot++;
    }
    void dfs(int u,int fa){
        for(int i = head[u];~i;i = e[i].to){
            int v = e[i].next;
            if(v == fa) continue;
            ans += e[i].val;
            dfs(v,u);
        }
    }
    int minReorder(int n, vector<vector<int>>& connections) {
        memset(head,-1,sizeof(head));
        for(vector<int>&f:connections){
            int u = f[0],v = f[1];
            add(v,u,0);add(u,v,1);
        }
        dfs(0,-1);
        return ans;
    }
};
