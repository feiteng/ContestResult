class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<pair<int, int>> G[n];
        for(auto& p : connections) {
            G[p[0]].push_back({p[1], 1});
            G[p[1]].push_back({p[0], 0});
        }
        int res = 0;
        int vis[n];
        memset(vis, 0, sizeof(vis));
        vis[0] = 1;
        queue<int> q({0});
        while(q.size()) {
            int cur = q.front(); q.pop();
            for(auto& ne : G[cur]) {
                if(!vis[ne.first]) {
                    vis[ne.first] = 1;
                    res += ne.second;
                    q.push(ne.first);
                }
            }
        }
        return res;
    }
};
