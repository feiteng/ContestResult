class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<bool> visited(n, false);
        queue<int> Q;
        vector<vector<int>> edges(n, vector<int>());
        for (vector<int>& connection: connections) {
            edges[connection[0]].push_back(connection[1]);
            edges[connection[0]].push_back(1);
            edges[connection[1]].push_back(connection[0]);
            edges[connection[1]].push_back(0);
        }
        Q.push(0);
        visited[0] = true;

        int ret = 0;
        while (!Q.empty()) {
            int cur = Q.front();
            Q.pop();
            for (int i = 0;i < edges[cur].size();i += 2) {
                int y = edges[cur][i];
                int v = edges[cur][i + 1];
                if (visited[y]) {
                    continue;
                }
                visited[y] = true;
                ret += v;
                Q.push(y);
            }
        }
        return ret;
    }
};
