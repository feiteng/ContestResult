#include <bits/stdc++.h>

vector<pair<int, int> > E[200001];
int ans = 0;
class Solution {
public:
    void dfs(int x, int fa) {
        for (int i = 0; i < (int )E[x].size(); i ++) {
            if (E[x][i].first == fa) continue;
            ans += E[x][i].second;
            dfs(E[x][i].first, x);
        }

    }
    int minReorder(int n, vector<vector<int>>& connections) {
        for (int i = 1; i<= n; i ++) E[i].clear();
        for (int i = 0; i < n - 1; i ++) {
            int x = connections[i][0], y = connections[i][1];
            ++ x; ++ y;
            E[x].push_back(make_pair(y, 1));
            E[y].push_back(make_pair(x, 0));
        }
        ans = 0;
        dfs(1, 0);
        return ans;
    }
};
