class Solution {
public:
    int minReorder(int n, vector<vector<int>>& eg) {
        using PII = pair<int,int>;
        vector<PII> G[n];
        for (const auto &p : eg) {
            G[p[1]].push_back({p[0],0});
            G[p[0]].push_back({p[1],1});
        }
        queue<int> q;
        q.push(0);
        vector<bool> vis(n,0);
        vis[0]=true;
        int ans=0;
        while (!q.empty()) {
            auto u=q.front();
            q.pop();

            for (const auto &p : G[u]) {
                if (!vis[p.first]) {
                    vis[p.first]=true;
                    q.push(p.first);
                    ans += p.second;
                }
            }
        }
        return ans;
    }
};
