class Solution {
public:
  int minReorder(int n, vector<vector<int>>& connections) {
    vector<vector<pair<int, int>>> G(n);
    for (auto c: connections)  {
      G[c[0]].push_back({c[1], 1});
      G[c[1]].push_back({c[0], 0});
    }
    vector<bool> visited(n);
    queue<int> Q;
    visited[0] = true;
    Q.push(0);
    int ans = 0;
    while (Q.size()) {
      int s = Q.front(); Q.pop();
      for (auto [t, d]: G[s]) if (!visited[t]) {
        visited[t] = true;
        Q.push(t);
        ans += d;
      }
    }
    return ans;
  }
};
