class Solution {
public:
    int ret=0;
    struct node
    {
        int t;
        bool h;
    };
    void dfs(int c,int f,vector<vector<node>>& graph)
    {
        for(auto& it:graph[c])
        {
            if(it.t==f)continue;
            if(it.h==false)
                ret++;
            dfs(it.t,c,graph);
        }
    }
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<node>>graph(n);
        for(auto& it:connections)
        {
            graph[it[0]].push_back({it[1],true});
            graph[it[1]].push_back({it[0],false});
        }
        ret=0;
        dfs(0,-1,graph);
        return n-1-ret;
    }
};
