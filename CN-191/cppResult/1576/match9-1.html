class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) 
    {
        vector<pair<int,int> > vt[n];
        vector<int> vis(n,0);

        for(auto &e: connections)
        {
            vt[e[0]].push_back({e[1],1});
            vt[e[1]].push_back({e[0],0});
        }
        queue<int> q;
        q.push(0);
        vis[0]=1;
        int res=0;
        while(!q.empty())
        {
            int u=q.front();
            q.pop();
            for(auto &e: vt[u])
            {
                if(!vis[e.first])
                {
                    if(e.second)
                        res++;
                    vis[e.first]=1;
                    q.push(e.first);
                }
            }
        }
        return res;
    }
};
