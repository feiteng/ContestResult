class Solution {
public:
    vector<int> e[50005];
    set<long long> s;
    int cnt=0;
    void dfs(int p,int f){
        for(int i=0;i<e[p].size();i++){
            if(e[p][i]==f) continue;

            if(!s.count((long long)p*50001+e[p][i])) cnt++;
            dfs(e[p][i],p);
        }
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        
        for(int i=0;i<connections.size();i++){
            e[connections[i][0]].push_back(connections[i][1]);
            e[connections[i][1]].push_back(connections[i][0]);
            s.insert((long long)connections[i][0]*50001+connections[i][1]);
        }
        dfs(0,-1);
        return connections.size()-cnt;
    }
};
