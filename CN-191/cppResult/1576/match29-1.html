class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<int> G[n];
        vector<int> f(n, -1);
        for(int i = 0; i < connections.size(); i++) {
            int u = connections[i][0], v = connections[i][1];
            G[u].push_back(v);
            G[v].push_back(u);
        }
        dfs(0, -1, f, G);
        int ans = 0;
        for(int i = 0; i < connections.size(); i++) {
            int u = connections[i][0], v = connections[i][1];
            if(f[v] == u) ans++;
        }
        return ans;
    }

    void dfs(int u, int fa, vector<int>& f, vector<int> G[]) {
        f[u] = fa;
        for(auto v : G[u]) {
            if(v == fa) continue;
            dfs(v, u, f, G);
        }
    }
};
