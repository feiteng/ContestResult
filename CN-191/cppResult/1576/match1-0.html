class Solution {
public:
    int ans;
    vector<int> G[50005];
    set<pair<int,int>> S;
    void dfs(int u, int fa) {
        
        for (int i=0; i<G[u].size(); i++) {
            int v=G[u][i];
            if (v==fa) continue;
            dfs(v,u);
            if (S.count(make_pair(u,v))) ans++;
        }
    }
    int minReorder(int n, vector<vector<int>>& g) {
        for (int i=0; i<g.size(); i++) {
            S.insert(make_pair(g[i][0],g[i][1]));
            G[g[i][0]].push_back(g[i][1]);
            G[g[i][1]].push_back(g[i][0]);
        }
        dfs(0,0);
        return ans;
    }
};
