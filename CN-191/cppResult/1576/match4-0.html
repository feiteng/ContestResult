class Solution {
public:
    vector<pair<int, bool> > edg[500010];
    int poi(int v, int f) {
        int ret = 0;
        for (auto it : edg[v]) {
            if (it.first != f) {
                ret += it.second;
                ret += poi(it.first, v);
            }
        }
        return ret;
    }
    int minReorder(int n, vector<vector<int>>& connections) {
        memset(edg, 0, sizeof(edg));
        for (auto it : connections) {
            edg[it[0]].push_back({it[1], true});
            edg[it[1]].push_back({it[0], false});
        }
        
        return poi(0,  -1);
    }
};
