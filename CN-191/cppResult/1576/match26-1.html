class Solution 
{
public:
	int minReorder(int n, vector<vector<int>>& connections) 
	{
		vector<vector<int>>es(n);
		vector<bool>vis(n, false);
		for (int i = 0; i < n - 1; i++)
		{
			es[connections[i][0]].push_back(connections[i][1]);
			es[connections[i][1]].push_back(-connections[i][0]);
		}
		return dfs(es, vis, 0);
	}
	int dfs(vector<vector<int>>&es, vector<bool>&vis, int root)
	{
		int ret = 0;
		vis[root] = true;
		for (int i = 0; i < es[root].size(); i++)
		{
			int t = es[root][i];
			if (vis[abs(t)] == true)
				continue;
			if (t > 0)
				ret++;
			ret += dfs(es, vis, abs(t));
		}
		return ret;
	}
};
