class Solution {
    int ans=0;
    void dfs(vector<vector<pair<int,int>>>& G,int n,int p){
        for(auto ne:G[n]){
            int nt;
            if(ne.second==n){
                nt=ne.first;
            }else nt=ne.second;
            if(nt==p) continue;
            if(ne.first==n) ans++;
            dfs(G,nt,n);
        }
    }
public:
    int minReorder(int n, vector<vector<int>>& connections) {

        vector<vector<pair<int,int>>> G(n);
        for(auto p:connections){
            G[p[0]].push_back({p[0],p[1]});
            G[p[1]].push_back({p[0],p[1]});
        }
        dfs(G,0,-1);
        return ans;
    }
};
