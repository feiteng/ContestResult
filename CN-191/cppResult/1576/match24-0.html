class Solution {
    vector<int> In[50010];
    vector<int> Out[50010];
    int Ans;
    void dfs(int x,int fa)
    {
        for (int i=0;i<Out[x].size();++i)
            if (Out[x][i]!=fa)
                ++Ans,dfs(Out[x][i],x);
        for (int i=0;i<In[x].size();++i)
            if (In[x][i]!=fa)
                dfs(In[x][i],x);
    }
public:
    int minReorder(int n, vector<vector<int>>& E) {
        int m=E.size();
        for (int i=0;i<m;++i)
        {
            int a=E[i][0];
            int b=E[i][1];
            In[b].push_back(a);
            Out[a].push_back(b);
        }
        dfs(0,-1);
        return Ans;
    }
};
