class Solution {
private:
    vector<vector<pair<int,int>>> graph;
    vector<bool> visited;
    int findCost(int pos){
        visited[pos] = true;
        int total = 0;
        for(auto [next,cost]:graph[pos]){
            if(!visited[next]){
                total += cost + findCost(next);
            }
        }
        return total;
    }
public:

    int minReorder(int n, vector<vector<int>>& connections) {
        graph.resize(n);
        visited.resize(n,false);
        for(auto edge:connections){
            graph[edge[0]].push_back({edge[1],1});
            graph[edge[1]].push_back({edge[0],0});
        }
        return findCost(0);
    }
};
