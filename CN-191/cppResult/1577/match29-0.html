using ll = long long;
using ld = double;

ld f[11][55][1<<8][1<<8];
ld C[55][55];
ld fact[55];

class Solution {
public:
    double getProbability(vector<int>& b) {
        
        
        int n = b.size();
        int sum = 0;
        for (auto x : b) sum += x;
        
        
        for (int i = 0; i <= sum; ++ i)
            for (int j = 0; j <= sum; ++ j)
                C[i][j] = 0;
        for (int i = 0; i <= sum; ++ i)
        {
            C[i][0] = 1;
            for (int j = 1; j <= i; ++ j)
                C[i][j] = C[i-1][j-1]+C[i-1][j];
        }
        fact[0] = 1;
        for (int i = 1; i <= sum; ++ i) fact[i] = fact[i-1]*i;
        
        for (int i = 0; i <= n; ++ i)
           for (int j = 0; j <= sum; ++ j)
               for (int s = 0; s < 1<<n; ++ s)
                   for (int t = 0; t < 1<<n; ++ t)
                       f[i][j][s][t] = 0;
        f[0][0][0][0] = 1;
        for (int i = 0; i < n; ++ i)
           for (int j = 0; j <= sum; ++ j)
               for (int s = 0; s < 1<<n; ++ s)
                   for (int t = 0; t < 1<<n; ++ t)
                   {
                       if (f[i][j][s][t] == 0) continue;
                       for (int d = 0; d <= b[i]; ++ d)
                       {
                           if (j+d <= sum) f[i+1][j+d][s|(d > 0)<<i][t|(d < b[i]) << i] += C[b[i]][d]*f[i][j][s][t];
                       }
                   }
        double ret = 0;
        for (int s = 0; s < 1<<n; ++ s)
            for (int t = 0; t < 1<<n; ++ t)
                if (__builtin_popcount(s) == __builtin_popcount(t))
                    ret += f[n][sum/2][s][t];
        ret *= fact[sum/2];
        ret *= fact[sum/2];
        ret /= fact[sum];
        return ret;
    }
};
