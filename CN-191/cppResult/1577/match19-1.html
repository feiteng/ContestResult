#include <bits/stdc++.h>

using namespace std;
double ans = 0;
double fac[101];
int a[101], b[101];
double f[9][101], g[101][1001];
int stk[1001];
double C(int x, int y) {
    return fac[x] / fac[y] / fac[x - y];
}
int n, N;
class Solution {
public:
    void dfs(int x) {
        if (x == n + 1) {
            int left = 0, right = 0, c0 = 0, c1 = 0;
            for (int i = 1; i <= n; i ++)
                if (b[i] == 2) {
                    ++ left;
                    ++ right;
                    ++ c0; ++ c1;
                }
                else if (b[i] == 1) {
                    left += a[i]; ++ c0;
                }
                else {
                    right += a[i];
                    ++ c1;
                }
            if (left > N / 2 || right > N / 2) return ;
            if (c0 != c1) return;

            int top = 0;
            for (int i = 1; i <= n; i ++)
                for (int j = 0; j <= N / 2; j ++) f[i][j] = 0;
            f[0][0] = 1;
            int cur = 0;
            for (int i = 1; i <= n; i ++) {
                cur += a[i];
                if (b[i] != 2) {
                    int res1 = b[i] == 1 ? a[i] : 0;
                    int res2 = b[i] == 1 ? 0 : a[i];
                    for (int j = N / 2; j >= res1; j --) {
                        int k = cur - j;
                        if (k < 0 || k > N / 2) continue;
                        if (k < res2) continue;
                        f[i][j] += f[i - 1][j - res1] * g[j - res1 + 1][j] / fac[res1] * g[k - res2 + 1][k] / fac[res2];
                    }
                }
                else {

                    for (int r1 = 1; r1 < a[i]; ++ r1) {
                        int r2 = a[i] - r1;
                        for (int j = N / 2; j >= r1; j --) {
                            int k = cur - j;
                            if (k < 0 || k > N / 2) continue;
                            if (k < r2) continue;
                            f[i][j] += f[i - 1][j - r1] * g[j - r1 + 1][j] / fac[r1] * g[k - r2 + 1][k] / fac[r2];
                        }
                    }
                }
            }
            ans += f[n][N / 2];
            return ;
        }
        for (int i = 0; i < 3; i ++) {
            if (i == 2 && a[x] < 2) continue;
            b[x] = i;
            dfs(x + 1);
        }
    }
    double getProbability(vector<int>& balls) {
        n = (int )balls.size();
        N = 0;
        for (int i = 0; i < n; i ++) N += balls[i];
        fac[1] = fac[0] = 1;
        for (int i = 2; i <= N; i ++) fac[i] = fac[i - 1] * (long long )i;
        double tans = fac[N];
        for (int i = 1; i <= N; i ++) {
            g[i][i - 1] = 1;
            for (int j = i; j <= N; j ++)
                g[i][j] = g[i][j - 1] * j;
        }
        for (int i = 0; i < n; i ++) tans /= fac[balls[i]];
        for (int i = 1; i <= n; i ++) a[i] = balls[i - 1];
        ans = 0;
        dfs(1);
        return ans * 1. / tans;
    }
};
