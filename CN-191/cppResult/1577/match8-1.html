const int N = 48 + 2;
const int M = 8 + 2;
double dp[N][N][M][M]; // l r lcs rcs
double pro[M][M];

class Solution {
public:
    double getProbability(vector<int>& balls) {
        pro[0][0] = 1;
        for (int i = 1; i < M; ++i) {
            pro[i][0] = pro[i-1][0] / 2;
            for (int j = 1; j <= i; ++j) {
                pro[i][j] = pro[i-1][j] / 2 + pro[i-1][j-1] / 2;
            }
        }
            
        for (int i = 0; i < N; ++i) for (int j = 0; j < N; ++j)
            for (int k = 0; k < M; ++k) for (int l = 0; l < M; ++l)
                dp[i][j][k][l] = 0;
        dp[0][0][0][0] = 1;
        
        int n = balls.size(), cnt = 0;
        for (int i = 0; i < n; ++i) {
            int x = balls[i];
            for (int p = 0; p <= x; ++p) {
                int dl = p == 0 ? 0 : 1;
                int dr = p == x ? 0 : 1;

                for (int l = 0; l <= cnt; ++l) {
                    int r = cnt - l;
                    for (int lcs = 0; lcs <= i; ++lcs) {
                        for (int rcs = 0; rcs <= i; ++rcs) {
                            dp[l+p][r+x-p][lcs+dl][rcs+dr] += dp[l][r][lcs][rcs] * pro[x][p];
                        }
                    }
                }
            }
            cnt += x;
        }
        
        int h = cnt/2;
        double ans1 = 0, ans2 = 0;
        for (int i = 1; i <= n; ++i) {
            ans1 += dp[h][h][i][i];
        }
        for (int i = 0; i <= n; ++i) {
            for (int j = 0; j <= n; ++j)
                ans2 += dp[h][h][i][j];
        }
        return ans1 / ans2;
    }
};
