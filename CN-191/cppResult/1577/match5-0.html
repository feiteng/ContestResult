double dp[10][30][10][10];

class Solution {
public:
    
    double C(int n, int m){
        if (m > n) return 0;
        if (m == n) return 1;
        double ret = 1;
        for (int i = n; i > m; i--) ret *= i;
        for (int i = 1; i <= n - m; i++) ret /= i;
        return ret;
    }
    double getProbability(vector<int>& balls) {
        int m = balls.size(), n = 0;
        for (auto b: balls) n += b; n /= 2;
        
        for (int i = 0; i <= m; i++){
            for (int k = 0; k <= n; k++){
                for (int p = 0; p <= m; p++){
                    for (int q = 0; q <= m; q++){
                        dp[i][k][p][q] = 0;
                    }
                }
            }
        }
        dp[0][0][0][0] = 1.0;
        
        int sum = 0;
        for (int c = 1; c <= m; c++){
            int cnt = balls[c - 1];
            for (int i = 0; i <= n; i++){
                for (int p = 0; p <= m; p++){
                    for (int q = 0; q <= m; q++){
                        // if (dp[c - 1][i][p][q] > 0) printf("%d %d %d %d = %.f\n", c - 1, i, p, q, dp[c - 1][i][p][q]);
                        double tot = 0;
                        for (int l = 0; l <= cnt; l++)
                            if (i + l <= n && sum - i + (cnt - l) <= n) tot += C(n - i, l) * C(n - (sum - i), cnt - l);
                        for (int l = 0; l <= cnt; l++){
                            if (i + l <= n && sum - i + (cnt - l) <= n) {
                                int np = p + (l > 0 ? 1 : 0);
                                int nq = q + ((cnt - l) > 0 ? 1 : 0);
                                double v = C(n - i, l) * C(n - (sum - i), cnt - l) / tot;
                                dp[c][i + l][np][nq] += dp[c - 1][i][p][q] * v;
                            }
                        }
                    }
                }
            }
            sum += cnt;
        }
        
        double ans = 0;
        for (int p = 0; p <= m; p++)
            ans += dp[m][n][p][p];
        
        return ans;
    }
};
