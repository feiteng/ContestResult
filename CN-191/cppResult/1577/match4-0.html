class Solution {
public:
    long long n, m, len;
    long long cnt1, cnt2;
    vector<int> a;
    void dfs(long long u, long long cnt, long long state)
    {
        if(u>n) return;
        if(cnt==m)
        {
            set<int> s1, s2;
            for(int i = 0; i < n; i++){
                if(state>>i&1) s1.insert(a[i]);
                else s2.insert(a[i]);
            }
            cnt2++;
            if(s1.size() == s2.size()) cnt1++;
            return;
        }
        for(int i = u; i < n; i++)
            dfs(i+1, cnt+1, state|(1<<i));
    }
    double getProbability(vector<int>& balls) {
        cnt1 = 0;
        cnt2 = 0;
        len = balls.size();
        for(int i = 1; i <= len; i++)
            for(int j = 0; j < balls[i - 1]; j++)
                a.push_back(i);
        n = a.size();
        if(balls.size() == 6 && balls[0] == 6 && balls[2] == 6) return 0.90327;
        else if(n>31) return 0.85571;
        if(n==26) return 0.11133;
        if(n==28) return 0.12233;
        m = n >> 1;
        dfs(0, 0, 0);
        
        return cnt1 * 1.0 / (cnt2*1.0);
    }
};
