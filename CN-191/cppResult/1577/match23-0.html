class Solution {
public:
    double res=0;
    double mu;
    int k;
    long long a[10];
    long long h[10];
    long long jiecheng[10];
    void init(){
        jiecheng[0] = 1;
        for(long long i=1;i<10;i++){
            jiecheng[i] = jiecheng[i-1]*i;
            cout<<jiecheng[i]<<endl;
        }
    }
    
    double solve(long long p[]){
        double gg = 1;
        for(int i=0;i<k;i++){
            gg = gg*(1.0/(jiecheng[p[i]]));
        }
        for(int i=0;i<k;i++){
            gg = gg*(1.0/jiecheng[a[i]-p[i]]);
        }
        return gg;
    }
    
    void dfs(int x,int lef,int rig,int numlef,int numrig){
        if(x>=k){
            if(numlef==numrig){
                double tmp = solve(h);
                //cout<<tmp<<endl;
                if(lef==rig) res+=tmp;
                mu+=tmp;
            }
            return;
        }
        for(int i=0;i<=a[x];i++){
            h[x] = i;
            if(i==0){
                dfs(x+1,lef,rig+1,numlef+i,numrig+a[x]-i);
            }
            else if(i==a[x]){
                dfs(x+1,lef+1,rig,numlef+i,numrig+a[x]-i);
            }
            else dfs(x+1,lef+1,rig+1,numlef+i,numrig+a[x]-i);
        }
    }
    double getProbability(vector<int>& b) {
        init();
        k = b.size();
        for(int i=0;i<k;i++){
            a[i] = b[i];
        }
        dfs(0,0,0,0,0);
        cout<<res<<mu<<endl;
        return res/mu;
    }
};
