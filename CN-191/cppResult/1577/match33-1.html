class Solution {
public:
    double ans=0,mi=0;int t=0;
    vector<int> v,r;
    double ln[55];
    void cal(vector<int>& balls){
        double x=2*ln[t];
        for(int i=0;i<balls.size();i++){
            x-=ln[v[i]]+ln[r[i]];
        }
        ans+=pow(2,x-mi);
    }
    void dfs(int pos,int l,vector<int>& balls){
        if(!l){
            int ct1=0,ct2=0;
            for(int i=0;i<balls.size();i++){
                r[i]=balls[i]-v[i];
                ct1+=!!v[i];
                ct2+=!!r[i];
            }
            if(ct1==ct2){
                cal(balls);
            }
            return;
        }
        if(pos==balls.size()) return;
        for(int i=0;i<=balls[pos]&&i<=l;i++){
            v[pos]=i;
            dfs(pos+1,l-i,balls);

            v[pos]=0;
        }
    }
    
    double getProbability(vector<int>& balls) {
        ln[0]=0;
        for(int i=1;i<=50;i++) ln[i]=log(i)/log(2);
        for(int i=2;i<=50;i++) ln[i]+=ln[i-1];
        int n=0;
        for(int i=0;i<balls.size();i++){
            n+=balls[i];
            mi-=ln[balls[i]];
            v.push_back(0);
            r.push_back(0);
        } 
        mi+=ln[n];
        cout<<mi<<endl;
        n/=2;t=n;
        dfs(0,n,balls);
        return ans;
    }
};
