class Solution {
public:
    double ans=0;
    double lo[50];
    double getlog(vector<int>& balls){
        int n=balls.size();
        int sum=0;
        for(int i=0;i<balls.size();i++)
            sum+=balls[i];
        double res=0;; 
        for(int i=0;i<balls.size();i++)
        {
            int x=balls[i];
            /*if(x!=0){
                for(int j=1;j<=x;j++){
                    res-=log(j);
                }
            }*/
            res-=lo[x];
        }
        //for(int j=1;j<=sum;j++)
        //    res+=log(j);
        res+=lo[sum];
        return res;
    }
    bool pd(vector<int>&l,vector<int>&r){
        int sum1=0;
        int sum2=0;
        int tot1=0;
        int tot2=0;
        for(int i=0;i<l.size();i++){
            sum1+=l[i];
            if(l[i]>0) tot1+=1;
        }
        for(int i=0;i<r.size();i++){
            sum2+=r[i];
            if(r[i]>0) tot2+=1;
        }
        if(sum1==sum2&&tot1==tot2) return true;
        else return false;
    }
    void printff(vector<int>& l){
        for(int i=0;i<l.size();i++)
            std::cout<<l[i]<<" ";
        std::cout<<endl;
    }
    void dfs(int pos,vector<int>&l,vector<int>&r,vector<int>& balls,double x){
        if(pos==balls.size()){
            //printff(l);
            //std::cout<<"------"<<endl;
            //    printff(r);
            if(pd(l,r)){
                
                ans+=exp(getlog(l)+getlog(r)-x);}
            return;
        }
        for(int i=0;i<=balls[pos];i++){
            l.push_back(i);
            r.push_back(balls[pos]-i);
            dfs(pos+1,l,r,balls,x);
            l.pop_back();
            r.pop_back();
        }
    }
    double getProbability(vector<int>& balls) {
        int n=balls.size();
        lo[1]=0;
        for(int i=2;i<=48;i++){
            lo[i]=lo[i-1]+log(i);
        }
        vector<int> l;
        vector<int> r;
        double x=getlog(balls);
        //std::cout<<x<<endl;
        dfs(0,l,r,balls,x);
        return ans;
    }
};
