class Solution {
public:
    int n, m;
    int a[100];
    int c[100], d[100];
    double fac[100];
    double sum, ret;
    double calc(int *f) {
        double o = fac[n];

        for (int i = 0; i < m; i++) {
            o /= fac[f[i]];
        }
        return o;
    }
    void dfs(int idx, int cnt, int flag, const vector<int>& b) {
        if (idx == m) {
            if (cnt == n) {
                double shb = calc(c) * calc(d);
                flag = 0;
                int _f1 = 0, _f2 = 0;
                for (int i = 0; i < m; i++) if (c[i] > 0) _f1++;
                for (int i = 0; i < m; i++) if (d[i] > 0) _f2++;
                flag = _f1 == _f2;
                // for (int i = 0; i < m; i++) {
                //     if (c[i] && a[c[i]] > 0) _f1 = 1;
                //     a[c[i]]++;
                // }
                // for (int i = 0; i < m; i++) {
                //     a[c[i]]--;
                // }
                // for (int i = 0; i < m; i++) {
                //     if (d[i] && a[d[i]] > 0) _f2 = 1;
                //     a[d[i]]++;
                // }
                // for (int i = 0; i < m; i++) {
                //     a[d[i]]--;
                // }
                // flag = _f1 || _f2;
                // for (int i = 0; i < m; i++) a[c[i]]++;
                // for (int i = 0; i < m; i++) {
                //     if (d[i] && a[d[i]] > (d[i] == c[i])) flag = 1;
                // }
                // cout << "??? " << shb << endl;
                // for (int i = 0; i < m; i++) cout << c[i] << ' '; cout << flag << endl;
                // for (int i = 0; i < m; i++) cout << d[i] << ' '; cout << flag << endl;
                // for (int i = 0; i < m; i++) a[c[i]]--;
                sum += shb;
                ret += flag * shb;
            }
            return ;
        }
        for (int i = 0; i <= b[idx]; i++) {
            if (i + cnt <= n) {
                c[idx] = i;
                d[idx] = b[idx] - i;
                dfs(idx + 1, cnt + i, 1, b);
            }
        }
    }
    double getProbability(vector<int>& b) {
        m = b.size();
        fac[0] = 1;
        for (int i = 1; i < 100; i++) fac[i] = fac[i-1] * i;
        n = 0;
        for (auto &x : b) n += x;
        n /= 2;
        for (int i = 1; i <= 50; i++) a[i] = 0;
        dfs(0, 0, 0, b);
        // cout << ret << ' ' << sum << endl;
        return 1.0 * ret / sum;
    }
};
