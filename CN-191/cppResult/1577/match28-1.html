class Solution {
public:
    double a[10000];
    int b[100];
    double son = 0;
    int cnt = 0;
    int n;
    int getcha = 0;
    void dfs(int sum, int x, vector<int>& balls) {
        if (sum > cnt / 2) return;
        if (x == balls.size()) {
            if (sum != cnt / 2) return;
            getcha++;
            int ca = 0;
            int cb = 0;
            for (int i = 0; i < n; i++) {
                if (b[i] > 0) ca ++;
                if (balls[i] - b[i] > 0) cb++;
            }
            if (ca != cb) return;
            double vin = a[cnt / 2];
            for (int i = 0; i < n; i++) {
                if (b[i] == 0) continue;
                vin /= a[b[i]];
            }
            double kay = a[cnt / 2];
            for (int i = 0; i < n; i++) {
                if (balls[i] - b[i] == 0) continue;
                kay /= a[balls[i] - b[i]];
            }
            son += kay * vin;
            return;
        }

        for (int i = 0; i <= balls[x]; i++) {
            b[x] = i;
            dfs(sum + i, x + 1, balls);
        }
    }
    double getProbability(vector<int>& balls) {
        n = balls.size();

        a[0] = 1;
        for (int i = 1; i <= 1000; i++) {
            a[i] = a[i - 1] * i;
        }
        for (int i = 0; i < balls.size(); i++) {
            cnt += balls[i];
        }
        double all = a[cnt];
        for (int i = 0; i < n; i++) {
            all /= a[balls[i]];
        }
        dfs(0, 0, balls);
        //return getcha;
        return son / all;
    }
};
