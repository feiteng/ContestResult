class Solution {
    public int maxProduct(int[] nums) {
        int maxProduct = Integer.MIN_VALUE;
        int length = nums.length;
        for (int i = 0; i < length; i++) {
            int num1 = nums[i] - 1;
            for (int j = i + 1; j < length; j++) {
                int num2 = nums[j] - 1;
                int product = num1 * num2;
                maxProduct = Math.max(maxProduct, product);
            }
        }
        return maxProduct;
    }
}
