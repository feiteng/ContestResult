class Solution {
    
   
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        List<Integer> list = new ArrayList<>();
        List<Integer> list1 = new ArrayList<>();
        for (int i = 0; i <= horizontalCuts.length; i++) {
            if (i == 0) list.add(horizontalCuts[i]);
            else if (i == horizontalCuts.length) list.add(h - horizontalCuts[i - 1]);
            else list.add(horizontalCuts[i] - horizontalCuts[i - 1]);
        }
        for (int i = 0; i <= verticalCuts.length; i++) {
            if (i == 0) list1.add(verticalCuts[i]);
            else if (i == verticalCuts.length) list1.add(w - verticalCuts[i - 1]);
            else list1.add(verticalCuts[i] - verticalCuts[i - 1]);
        }
        Collections.sort(list);
        Collections.sort(list1);
        return (int) ((((long) list.get(list.size() - 1) % (1000000007l)) * ((long) list1.get(list1.size() - 1) % (1000000007l))) % (1000000007l)); }
}
