class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        int maxRow = 0, maxCol = 0;
        int pre = 0;
        Arrays.sort(horizontalCuts);
        for (int cut : horizontalCuts) {
            maxRow = Math.max(maxRow, cut - pre);
            pre = cut;
        }
        maxRow = Math.max(maxRow, h - pre);

        pre = 0;
        Arrays.sort(verticalCuts);
        
for (int cut : verticalCuts) {
            maxCol = Math.max(maxCol, cut - pre);
            pre = cut;
        }
        maxCol = Math.max(maxCol, w - pre);
        
        return (int) ((long) maxRow * maxCol % 1_000_000_007);
    }
}
