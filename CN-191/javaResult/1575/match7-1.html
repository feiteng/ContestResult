class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        int maxH = 0, cur = 0;
        for(int i: horizontalCuts){
            maxH = Math.max(maxH, i - cur);
            cur = i;
        }
        maxH = Math.max(maxH, h - cur);
        int maxV = 0;
        cur = 0;
        for(int v: verticalCuts){
            maxV = Math.max(maxV, v - cur);
            cur = v;
        }
        maxV = Math.max(maxV, w - cur);
        return (int)((long)maxV * maxH % ((int)1e9 + 7));
    }
}
