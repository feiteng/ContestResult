class Solution {

    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        long max1 = 0, max2 = 0;
        for (int i = 0; i <= horizontalCuts.length; i++) {
            max1 = Math.max(max1,
                    (i < horizontalCuts.length ? horizontalCuts[i] : h) - (i > 0 ? horizontalCuts[i - 1] : 0));
        }
        for (int j = 0; j <= verticalCuts.length; j++) {
            max2 = Math.max(max2, (j < verticalCuts.length ? verticalCuts[j] : w) - (j > 0 ? verticalCuts[j - 1] : j));
        }
        return (int) (max1 * max2 % 1000000007);
    }
}
