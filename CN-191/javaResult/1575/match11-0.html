class Solution {
    public int maxArea(int h, int w, int[] hs, int[] vs) {
        int maxH = 0;
        int maxV = 0;
        shuffle(hs, 0, hs.length);
        shuffle(vs, 0, vs.length);
        Arrays.sort(hs);
        Arrays.sort(vs);
        
        for(int i = 1; i < hs.length; i++){
            maxH = Math.max(hs[i] - hs[i - 1], maxH);
        }
        for(int i = 1; i < vs.length; i++){
            maxV = Math.max(vs[i] - vs[i - 1], maxV);
        }
        maxH = Math.max(maxH, hs[0]);
        maxH = Math.max(maxH, h - hs[hs.length - 1]);
        maxV = Math.max(maxV, vs[0]);
        maxV = Math.max(maxV, w - vs[vs.length - 1]);
        return (int)((long)maxH * maxV % (int)(1e9 + 7));
    }
    
    Random random = new Random();
    
    public int nextInt(int l, int r){
        return random.nextInt(r - l + 1) + l;
    }
    
    public void shuffle(int[] data, int from, int to) {
        to--;
        for (int i = from; i <= to; i++) {
            int s = nextInt(i, to);
            int tmp = data[i];
            data[i] = data[s];
            data[s] = tmp;
        }
    }
}
