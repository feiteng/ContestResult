class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        long maxH=getMax(horizontalCuts,h);
        long maxV=getMax(verticalCuts,w);
        return (int)(maxH*maxV%1000000007);
    }
    
    private int getMax(int[] h,int n){
        Arrays.sort(h);
        int max=0;
        int last=0;
        for(int val:h){
            max=Math.max(val-last,max);
            last=val;
        }
        max=Math.max(n-last,max);
        return max;
    }
}
