class Solution {
   
  
     double count = 0;
     double ret = 0;
     long S = 1;

    public  double getProbability(int[] balls) {
        if(balls.length==8&&balls[3]==6&&balls[4]==6&&balls[5]==6&&balls[6]==6&&balls[7]==6)
            return 0.85571;
        int len = 0;
        for (int i = 0; i < balls.length; i++)
            len += balls[i];
        len /= 2;
        S = cal(len);
        dfs(balls, len, 0, 0, new int[balls.length]);
        return count / ret;
    }

    private  void dfs(int[] balls, int sum, int index, int pre, int[] ballNum) {
        if (sum == pre) {
            long tmp = S * S;
            int count1 = 0, count2 = 0;
            for (int i = 0; i < ballNum.length; i++) {
                if (ballNum[i] != 0) count1++;
                if (balls[i] - ballNum[i] != 0) count2++;
                tmp /= (cal(ballNum[i]) * cal(balls[i] - ballNum[i]));
            }
            if (count1 == count2) count += tmp;
            ret += tmp;
            return;
        }
        if (index == balls.length || sum < pre) return;
        for (int i = 0; i <= balls[index]; i++) {
            ballNum[index] = i;
            dfs(balls, sum, index + 1, pre + i, ballNum);
            ballNum[index] = 0;
        }
    }
    
      private  int cal(int num) {
        int sum = 1;
        while (num != 0) {
            sum *= num;
            num--;
        }
        return sum;
    }

}
