class Solution {
    public int minReorder(int n, int[][] connections) {
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for (int[] conn : connections) {
            map.computeIfAbsent(conn[0], k -> new HashMap<>()).put(conn[1], 1);
            map.computeIfAbsent(conn[1], k -> new HashMap<>()).put(conn[0], 0);
        }

        return minReorder(map, 0, -1);
    }

    private int minReorder(Map<Integer, Map<Integer, Integer>> map, int cur, int pre) {
        int res = 0;
        Map<Integer, Integer> nexts = map.get(cur);
        if (nexts == null) return res;
        for (Map.Entry<Integer, Integer> entry : nexts.entrySet()) {
            if (entry.getKey() == pre) continue;
            res += entry.getValue();
            res += minReorder(map, entry.getKey(), cur);
        }
        return res;
    }
}
