class Solution {
    class Node{
        List<Node> zEdges=new ArrayList<>();
        List<Node> nEdges=new ArrayList<>();
        
    }
    public int minReorder(int n, int[][] connections) {
        Node[] nodes=new Node[n];
        for(int i=0;i<n;i++){
            nodes[i]=new Node();
        }
        for(int[] c:connections){
            int bf=c[0];
            int af=c[1];
            nodes[bf].zEdges.add(nodes[af]);
            nodes[af].nEdges.add(nodes[bf]);
        }
        
        //从0开始遍历
        int res=BFS(nodes);
        return res;
    }
    
    private int BFS(Node[] nodes){
        Set<Node> visited=new HashSet<>(nodes.length);
        Queue<Node> queue=new LinkedList<>();
        queue.add(nodes[0]);
        visited.add(nodes[0]);
        int res=0;
        while(!queue.isEmpty()){
            Node cur=queue.poll();
            for(Node nxt:cur.zEdges){
                if(visited.add(nxt)){
                                res++;
        queue.add(nxt);
                }
            }
            for(Node nxt:cur.nEdges){
                if(visited.add(nxt)){
                    queue.add(nxt);
                }
            }        
        }
        return res;
    }
}
