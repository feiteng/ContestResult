class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horizontalCuts.sort()
        verticalCuts.sort()
        horizontalCuts += [h]
        horizontalCuts[1:] = [horizontalCuts[i] - horizontalCuts[i-1] for i in range(1,len(horizontalCuts))] 
        verticalCuts += [w]
        verticalCuts[1:] = [verticalCuts[i] - verticalCuts[i-1] for i in range(1,len(verticalCuts))] 
        return max(verticalCuts) * max(horizontalCuts) % (10 ** 9 +7)
