class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        L = 0
        maxh = 0
        for i in sorted(horizontalCuts):
            maxh = max(maxh,i-L)
            L = i
        maxh = max(maxh,h-L)
        L = 0
        maxw = 0
        for i in sorted(verticalCuts):
            maxw = max(maxw,i-L)
            L = i
        maxw = max(maxw,w-L)
        return (maxh*maxw)%(10**9+7)
