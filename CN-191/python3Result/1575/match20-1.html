class Solution:
    def maxArea(self, h: int, w: int, hc: List[int], vc: List[int]) -> int:
        hc.sort()
        vc.sort()
        aw, ah = max(hc[0], h - hc[-1]), max(vc[0], w - vc[-1])
        n, m = len(hc), len(vc)
        for i in range(1, n):
            aw = max(aw, hc[i] - hc[i-1])
        for i in range(1, m):
            ah = max(ah, vc[i] - vc[i-1])
        return aw * ah % (10 ** 9 + 7)
