class Solution:
    def maxArea(self, h: int, w: int, hori: List[int], verti: List[int]) -> int:
        hori.sort()
        verti.sort()
        hori.append(h)
        verti.append(w)
        max1 = 0
        max2 = 0
        prev = 0
        for h in hori:
            max1 = max(max1, h - prev)
            prev = h
        prev = 0
        for v in verti:
            max2 = max(max2, v - prev)
            prev = v
        return (max1 * max2 ) % (10 ** 9 + 7)
