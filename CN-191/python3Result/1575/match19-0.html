class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        mh, mv = 0, 0
        h = [0] + sorted(horizontalCuts) + [h]
        for i in range(len(h) - 1):
            mh = max(mh, h[i + 1] - h[i])
        h = [0] + sorted(verticalCuts) + [w]
        for i in range(len(h) - 1):
            mv = max(mv, h[i + 1] - h[i])
        return mh * mv % int(1e9 + 7)
