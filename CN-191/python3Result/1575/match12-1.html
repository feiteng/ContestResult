class Solution:
    def maxArea(self, h: int, w: int, hh: List[int], vv: List[int]) -> int:
        hh.append(h)
        hh.append(0)
        vv.append(0)
        vv.append(w)
        
        hh.sort()
        vv.sort()
        mh = max(b - a for a, b in zip(hh, hh[1:]))
        mv = max(b - a for a, b in zip(vv, vv[1:]))
        
        return mh * mv % (10 ** 9 + 7)
