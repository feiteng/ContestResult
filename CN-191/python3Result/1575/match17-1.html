class Solution:
    def maxArea(self, h: int, w: int, hcs: List[int], vcs: List[int]) -> int:
        hcs.sort()
        vcs.sort()
        hcs = [0] + hcs + [h]
        vcs = [0] + vcs + [w]
        maxh, maxv = 0, 0
        for i, h1 in enumerate(hcs[:-1]):
            h2 = hcs[i + 1]
            maxh = max(maxh, h2 - h1)
        for i, v1 in enumerate(vcs[:-1]):
            v2 = vcs[i + 1]
            maxv = max(maxv, v2 - v1)
        return maxh * maxv % (10 ** 9 + 7)
