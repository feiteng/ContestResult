class Solution:
	def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
		horizontalCuts.sort()
		verticalCuts.sort()
		MOD = 10 ** 9 + 7

		mw = 0
		mh = 0
		for i in range(len(verticalCuts)):
			mw = max(mw, verticalCuts[i] - (0 if i == 0 else verticalCuts[i-1]))
		mw = max(mw, w - verticalCuts[-1])
		for i in range(len(horizontalCuts)):
			mh = max(mh, horizontalCuts[i] - (0 if i == 0 else horizontalCuts[i-1]))
		mh = max(mh, h - horizontalCuts[-1])
		return mw * mh % MOD
