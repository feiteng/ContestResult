class Solution:
    def maxArea(self, h: int, w: int, ha: List[int], wa: List[int]) -> int:
        mod = 10 ** 9 + 7
        
        ha
.append(0)
        ha.append(h)
        
        wa.append(0)
        wa.append(w)
        
        ha.sort()
        wa.sort()
        
        hm = 0
        for i in range(1, len(ha)):
            hm = max(hm, ha[i]-ha[i-1])
        
        wm = 0
        
        for i in range(1, len(wa)):
            wm = max(wm, wa[i]-wa[i-1])
        
        # print(hm, wm)
        
        return (hm * wm) % mod
