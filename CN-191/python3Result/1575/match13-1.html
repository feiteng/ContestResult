class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: 'List[int]', verticalCuts: 'List[int]') -> int:
        a1=[0,h]+horizontalCuts
        a2=[0,w]+verticalCuts
        a1.sort()
        a2.sort()
        x1
=[a1[i]-a1[i-1] for i in range(1,len(a1))]
        x2=[a2[i]-a2[i-1] for i in range(1,len(a2))]
        mh,mw=max(x1),max(x2)
        return mh*mw%(10**9+7)
