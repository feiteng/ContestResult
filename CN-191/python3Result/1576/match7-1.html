from collections import deque
class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        g 
= [[] for _ in range(n)]
        for u, v in connections:
            g[u].append((v, 1))
            g[v].append((u, 0))
        vis = set([0])
        q = deque([0])
        ans = 0
        while q:
            u = q.popleft()
            for v, d in g[u]:
                if v not in vis:
                    ans += d
                    vis.add(v)
                    q.append(v)
        return ans
