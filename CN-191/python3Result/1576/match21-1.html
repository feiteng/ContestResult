class Solution:
    def minReorder(self, n: int, connections: 'List[List[int]]') -> int:
        correct=set([tuple(c) for c in connections])
        d={}
        for c in connections:
            a1,a2=c
            if a1 not in d:
                d[a1]=[]
            d[a1].append(a2)
            if a2 not in d:
                d[a2]=[]
            d[a2].append(a1)

        self.ans=0
        visited=set([0])
        def search(x):
            for nx in d[x]:
                if nx in visited:
                    continue
                visited.add(nx)
                
if (x,nx) in correct:
                    self.ans+=1
                search(nx)
        search(0)
        return self.ans
