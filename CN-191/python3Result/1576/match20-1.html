class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        dic1 = collections.defaultdict(set)
        dic2 = collections.defaultdict(set)
        for c in connections:
            dic1
[c[1]].add(c[0])
            dic2[c[0]].add(c[1])
        vis = [False for _ in range(n)]
        def dfs(x, dic1, dic2, vis, flag):
            if vis[x]: return 0
            vis[x] = True
            ret = 0 if not flag else 1
            for can in dic1[x]:
                ret += dfs(can, dic1, dic2, vis, False)
            for cannot in dic2[x]:
                ret += dfs(cannot, dic1, dic2, vis, True)
            return ret
        return dfs(0, dic1, dic2, vis, False)
