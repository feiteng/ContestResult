class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        pre, nex = [[] for i in range(n)], [[] for i in range(n)]
        for x, y in connections:
            nex[x].append(y)
            pre[y].append(x)
        stack = [0]
        visited = [False] * n
        visited[0] = True
        ans = 0
        while len(stack) > 0:
            x = stack.pop()
            for y in nex[x]:
                if not visited[y]:
                    ans += 1
                    visited[y] = True
                    stack.append(y)
            for y in pre[x]:
                if not visited[y]:
                    visited[y] = True
                    stack.append(y)
        return ans
