class Solution:
    
def minReorder(self, n: int, con: List[List[int]]) -> int:
        i = defaultdict(set) # 从u出去
        o = defaultdict(set) # 到v
        
        for u, v in con:
            i[u].add(v)
            o[v].add(u)
           
        self.res = 0
        seen = set()
        def right(node):
            if node in seen:
                return 
            seen.add(node)
            for x in i[node]:
                left(x)
            for x in o[node]:
                right(x)
        def left(node):
            
            if node in seen:
                return 
            print(node)
            seen.add(node)
            self.res += 1
            for x in i[node]:
                left(x)
            for x in o[node]:
                right(x)
        right(0)
        return self.res
