class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        # 从根节点出发，进行bfs
        searched = set()
        bfs = [0]
        searched.add(0)
        wayTo = [[] for _ in range(n)]
        wayFrom = [ [] for _ in range(n)]
        for a,b in connections:
            wayTo[a].append(b)
            wayFrom[b].append(a)
        res = 0
        while bfs:
            x = bfs.pop(0)
            for y in wayTo[x]:
                if y not in searched:
                    searched.add(y)
                    bfs.append(y)
                    res += 1
            for y in wayFrom[x]:
                if y not in searched:
                    searched.add(y)
                    bfs.append(y)
        return res
