class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        def solve(flag, g, i):
            flag[i] = True
            ans = 0
            for x in g[i]:
                if not flag[x[0]]: 
                    ans += solve(flag, g, x[0])
                    ans += x[1]
            return ans
        g = [[] for _ in range(n)]
        flag = [False for _ in range(n)]
        for x in connections:
            g[x[0]].append([x[1],1])
            g[x[1]].append([x[0],0])
        return solve(flag, g, 0)
