class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        ans = 0
        to = [[] for i in range(n)]
        fro = [[] for i in range(n)]
        for a, b in connections:
            to[a].append(b)
            fro[b].append(a)
        vis = [False for i in range(n)]
        from queue import Queue
        q = Queue()
        q.put(0)
        vis[0] = True
        while q.qsize() > 0:
            f = q.get()
            for v in to[f]:
                if not vis[v]:
                    vis[v] = True
                    ans += 1
                    q.put(v)
            for v in fro[f]:
                if not vis[v]:
                    vis[v] = True
                    q.put(v)
        return ans
