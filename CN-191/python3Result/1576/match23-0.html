class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        nei = [[] for _ in range(n)]
        ok = set()
        rev = [[] for _ in range(n)]
        for y, x in connections:
            nei[x].append(y)
            rev[y].append(x)
            
        ok.add(0)
        q = collections.deque()
        q.append(0)
        res = 0
        while q:
            cur = q.popleft()
            for v in nei[cur]:
                q.append(v)
                ok.add(v)
            for v in rev[cur]:
                if v in ok: continue
                res += 1
                ok.add(v)
                q.append(v)
                
        return res
