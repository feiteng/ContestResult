class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        start = [[] for _ in range(n)]
        end = [[] for _ in range
(n)]

        for connection in connections:
            start[connection[0]].append(connection[1])
            end[connection[1]].append(connection[0])
        
        visited = [False] * n
        visited[0] = True
        queue = [0]

        ans = 0
        while queue:
            first = queue.pop(0)
            for e in start[first]:
                if not visited[e]:
                    visited[e] = True
                    ans += 1
                    queue.append(e)
            for s in end[first]:
                if not visited[s]:
                    visited[s] = True
                    queue.append(s)

        return ans
