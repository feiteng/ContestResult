class Solution:
    def minReorder(self, n: int, conn: List[List[int]]) -> int:
        conn_set = set(tuple(x) for x in conn)
        tot = 0
        adj = [[] for _ in range(n)]
        for f, t in conn:
            adj[f].append((t, True))
            adj[t].append((f, False))
        visited = [False] * n 
        visited[0] = True
        q = [0]
        while q:
            temp_q = []
            for i in q:
                for nei, flag in adj[i]:
                    if not visited[nei]:
                        temp_q.append(nei)
                        visited[nei] = True
                    else:
                        if (nei, i) in conn_set:
                            tot += 1
            q = temp_q 
        return tot
