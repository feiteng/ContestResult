from collections import defaultdict


class Solution:
	def minReorder(self, n: int, connections) -> int:
		a = set()
		b = defaultdict(list)
		for conn in connections:
			a.add(tuple(conn))
			b[conn[0]].append(conn[1])
			b[conn[1]].append(conn[0])

		ans = [0]
		vis = set()

		def dfs(root):
			if root in vis:
				return
			vis.add(root)
			for ch in b[root]:
				if ch in vis:
					continue
				if (ch, root) not in a:
					ans[0] += 1
				dfs(ch)

		dfs(0)

		return ans[0]
