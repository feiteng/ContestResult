class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        ans = None
        for i in range(len(nums)):
            for j in range(i + 1, len
(nums)):
                if ans is None or ans < (nums[i] - 1) * (nums[j] - 1):
                    ans = (nums[i] - 1) * (nums[j] - 1)
        return ans
