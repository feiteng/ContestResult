class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        m = 10**9 + 7
        horizontalCuts.sort()
        verticalCuts.sort()
        
        
        hor = []
        ver = []
        for i in range(len(horizontalCuts)):
            hor.append(horizontalCuts[0]) if i == 0 else hor.append(horizontalCuts[i]-horizontalCuts[i-1])
        hor.append(h-horizontalCuts[-1])
        for j in range(len(verticalCuts)):
            ver.append(verticalCuts[0]) if j == 0 else ver.append(verticalCuts[j]-verticalCuts[j-1])
        ver.append(w-verticalCuts[-1])
        hor.sort()
        ver.sort()
        return (hor[-1]*ver[-1]) % m
