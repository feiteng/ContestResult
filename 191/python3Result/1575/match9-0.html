class Solution:
    def maxArea(self, h: int, w: int, hor: List[int], ver: List[int]) -> int:
        hor = [0] + sorted(hor) + [h]
        ver = [0] + sorted(ver) + [w]
        x = max(y-x for x, y in zip(hor[:-1], hor[1:]))
        y = max(y-x for x, y in zip(ver[:-1], ver[1:]))
        return (x*y) % (10**9+7)
