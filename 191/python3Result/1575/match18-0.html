class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horiz_cuts = [0] + sorted(horizontalCuts) + [h]
        vert_cuts = [0] + sorted(verticalCuts) + [w]
        height = max(horiz_cuts[i+1] - horiz_cuts[i] for i in range(len(horiz_cuts) - 1))
        width = max(vert_cuts[i+1] - vert_cuts[i] for i in range(len(vert_cuts) - 1))
        return (height * width) % (10**9 + 7)
