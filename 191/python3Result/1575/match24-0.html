class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horizontalCuts.sort()
        verticalCuts.sort()
        h_org = [0] + horizontalCuts + [h]
        v_org = [0] + verticalCuts + [w]
        h_max = 0
        v_max = 0
        for i in range(1, len(h_org)):
            h_max = max(h_max, h_org[i] - h_org[i - 1])
        for i in range(1, len(v_org)):
            v_max = max(v_max, v_org[i] - v_org[i - 1])
        return (h_max * v_max) % (10 ** 9 + 7)
