class Solution:
    def maxArea(self, h: int, w: int, H: List[int], V: List[int]) -> int:
        H.sort()
        V.sort()
        H.insert(0, 0)
        V.insert(0, 0)
        H.append(h)
        V.append(w)
        hmx = max(H[i + 1] - H[i] for i in range(len(H) - 1))
        rmx = max(V[i + 1] - V[i] for i in range(len(V) - 1))
        return (hmx * rmx) % (10 ** 9 + 7)
