class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horizontalCuts.sort()
        verticalCuts.sort()
        hc = [0] + horizontalCuts + [h]
        vc = [0] + verticalCuts + [w]
        hd = [hc[i] - hc[i-1] for i in range(1,len(hc))]
        vd = [vc[i] - vc[i-1] for i in range(1,len(vc))]
        return (max(hd) * max(vd)) % 1000000007
