class Solution:
    def maxArea(self, h: int, w: int, hc: List[int], vc: List[int]) -> int:
        hc.sort()
        vc.sort()
        hc = [0] + hc + [h]
        vc = [0] + vc + [w]
        mh = -1e9
        mw = -1e9
        for i in range(len(hc)-1):
            mh = max(mh, hc[i+1]-hc[i])
        for i in range(len(vc)-1):
            mw = max(mw, vc[i+1]-vc[i])
        return (mh * mw) % (int(1e9) + 7)
