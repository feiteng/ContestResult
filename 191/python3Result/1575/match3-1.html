class Solution:
    def maxArea(self, h: int, w: int, hor: List[int], ver: List[int]) -> int:
        hor.append(0)
        hor.append(h)
        ver.append(0)
        ver.append(w)
        hor = sorted(hor)
        ver = sorted
(ver)
        maxH, maxV = 0, 0
        for i in range(len(hor) - 1):
            maxH = max(maxH, hor[i + 1] - hor[i])
        for i in range(len(ver) - 1):
            maxV = max(maxV, ver[i + 1] - ver[i])
        return maxH * maxV % (10**9+7)
