class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horizontalCuts.sort()
        verticalCuts.sort()
        
        hmax=horizontalCuts[0]
        for i in range(1,len(horizontalCuts)):
            hmax=max(hmax,horizontalCuts[i]-horizontalCuts[i-1])
        hmax=max(hmax,h-horizontalCuts[-1])
        wmax=verticalCuts[0]
        for i in range(1,len(verticalCuts)):
            wmax=max(wmax,verticalCuts[i]-verticalCuts[i-1])
        wmax=max(wmax,w-verticalCuts[-1])
        
        return (wmax*hmax)%(10**9+7)
