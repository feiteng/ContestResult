class Solution:
    def maxArea(self, h: int, w: int, ho: List[int], ve: List[int]) -> int:
        ho.sort()
        ve.sort()
        ho.append(h)
        ve.append(w)
        ho.insert(0,0)
        ve.insert(0,0)
        m1, m2 = 0, 0
        for i in range(1, len(ho)):
            m1 = max(m1, ho[i]-ho[i-1])
        for i in range(1, len(ve)):
            m2 = max(m2, ve[i]-ve[i-1])
        return m1*m2 % 1000000007
