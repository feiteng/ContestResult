class Solution:
    def maxArea(self, h: int, w: int, a: List[int], b: List[int]) -> int:
        a.sort()
        b.sort()
        a = [0] + a + [h]
        b = [0] + b + [w]
        ma = 0
        mb = 0
        for i in range(len(a) - 1):
            ma = max(ma, a[i+1]-a[i])
        for i in range(len(b) - 1):
            mb = max(mb, b[i+1]-b[i])
        return int(ma*mb % (1e9+7))
