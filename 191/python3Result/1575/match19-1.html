class Solution:
    def maxArea(self, h: int, w: int, hc: List[int], vc: List[int]) -> int:
        hc.append(0)
        hc.append(h)
        vc.append(0)
        vc.append(w)
        hc.sort()
        vc.sort()
        x, y 
= 0, 0
        for i in range(1, len(hc)):
            x = max(hc[i] - hc[i - 1], x)
        for i in range(1, len(vc)):
            y = max(vc[i] - vc[i - 1], y)
        return (x * y) % (10**9+7)
