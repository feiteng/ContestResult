class Solution:
    def maxArea(self, h: int, w: int, hs: List[int], vs: List[int]) -> int:
        hs.sort()
        vs.sort()
        hs = [0] + hs + [h]
        vs = [0] + vs + [w]
        x = [hs[i] - hs[i - 1] for i in range(1, len(hs))]
        y = [vs[i] - vs[i - 1] for i in range(1, len(vs))]
        # print(x, y)
        return max(x) * max(y) % (int(1e9 + 7))
