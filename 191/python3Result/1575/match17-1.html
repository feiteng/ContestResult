class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        horizontalCuts.sort()
        verticalCuts.sort()
        hh = [0] + horizontalCuts + [h]
        ww = [0] + verticalCuts + [w]

        
        mh = 0
        for i in range(len(hh)-1):
            mh = max(mh, hh[i+1] - hh[i])
        
        mw = 0
        for i in range(len(ww)-1):
            mw = max(mw, ww[i+1] - ww[i])
        
        return (mh*mw) % (10**9+7)
