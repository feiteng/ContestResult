class Solution:
    def maxArea(self, h: int, w: int, hor: List[int], ver: List[int]) -> int:
        hor.sort()
        ver.sort()
        hor.append(h)
        ver.append(w)
        mh,mv=hor[0],ver[0]
        for i in range(1,len(hor)):
            mh=max(mh,hor[i]-hor[i-1])
        for i in range(1,len(ver)):
            mv=max(mv,ver[i]-ver[i-1])
        return mh*mv%1000000007
