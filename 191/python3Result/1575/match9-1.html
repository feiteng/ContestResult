class Solution:
    def maxArea(self, h: int, w: int, hc: List[int], vc: List[int]) -> int:
        hc = [0] + sorted(hc) + [h]
        vc = [0] + sorted(vc) + [w]
        a = max(b - a for a, b in zip(hc[:-1], hc[1:]))
        b = max(b - a for a, b in zip(vc[:-1], vc[1:]))
        return (a * b) % 1000000007
