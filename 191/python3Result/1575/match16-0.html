class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        hc = 0
        prev = 0
        for num in sorted(horizontalCuts):
            hc = max(hc, num - prev)
            prev = num
        hc = max(hc, h - prev)
        
        wc = 0
        prev = 0
        for num in sorted(verticalCuts):
            wc = max(wc, num - prev)
            prev = num
        wc = max(wc, w - prev)
        
        return (wc * hc)%(10**9 + 7)
