class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        maxi1 = 0
        last = 0
        for n in sorted(horizontalCuts):
            maxi1 = max(maxi1, n-last)
            last = n
        maxi1 = max(h-last, maxi1)

        maxi2 = 0
        last = 0
        for n in sorted(verticalCuts):
            maxi2 = max(maxi2, n-last)
            last = n
        maxi2 = max(w-last, maxi2)
        return (maxi1*maxi2)%(10**9+7)
