class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
        last = 0
        mH = 0
        for n in sorted(horizontalCuts):
            # print(n-last, 's')
            mH = max(mH, n - last)
            last = n
        mH = max(mH, h - last)
        last = 0
        mV = 0
        for n in sorted(verticalCuts):
            # print(n-last)
            mV = max(mV, n - last)
            last = n
        mV = max(mV, w - last)
        # print(mV, mH)
        return (mV * mH) % 1000000007
