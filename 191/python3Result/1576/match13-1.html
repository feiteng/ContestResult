class Solution:
    def minReorder(self, n: int, con: List[List[int]]) -> int:
        D = collections.defaultdict(list)
        for x, y in con:
            D[x].append((y, 1))
            D[y].append((x, -1))
        q = [0]
        visited = {0}
        count = 0
        while q:
            newq = []
            for x in q:
                
for next_x, d in D[x]:
                    if next_x not in visited:
                        visited.add(next_x)
                        newq.append(next_x)
                        if d == 1:
                            count += 1
            q = newq
        return count
