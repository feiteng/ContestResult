class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        f = [[] for _ in range(n)]
        r = [[] for _ in range(n)]
        
        for s, t in connections:
            f[s].append(t)
            r[t].append(s)
        
            
        seen = set([0])
        q = [0]
        ans = 0
        while q:
            n = q.pop()
            
            for m in r[n]:
                if m not in seen:
                    seen.add(m)
                    q.append(m)
                
            for m in f[n]:
                if m not in seen:
                    seen.add(m)
                    q.append(m)
                    ans += 1
        return ans
