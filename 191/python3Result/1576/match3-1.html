class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        edges = collections.defaultdict(set)
        redges = collections.defaultdict(set)
        for s, e in connections :
            edges[e].add(s)
            redges[s].add(e)
        
        visit_list = [0]
        visited = set([0])
        ip = 0
        to_ret = 0
        while ip < len(visit_list) :
            nt = visit_list[ip]
            ip += 1
            
for nn in edges[nt] :
                if nn in visited :
                    continue
                visit_list.append(nn)
                visited.add(nn)
            for nn in redges[nt] :
                if nn in visited :
                    continue
                visit_list.append(nn)
                visited.add(nn)
                to_ret += 1
        return to_ret
