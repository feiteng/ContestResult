class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        to = collections.defaultdict(list)
        frm = collections.defaultdict(list)
        for u, v in connections:
            to[u].append(v)
            frm[v].append(u)
        queue = [0]
        visited = set()
        visited.add(0)
        res = 0
        while queue:
            node = queue.pop(0)
            for nei in to[node]:
                if nei not in visited:
                    res += 1
                    visited.add(nei)
                    queue.append(nei)
            for nei in frm[node]:
                if nei not in visited:
                    visited.add(nei)
                    queue.append(nei)
        return res
