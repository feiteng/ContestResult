class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        g = collections.defaultdict(set)
        g_rev = collections.defaultdict(set)
        for a, b in connections:
            g[a].add(b)
            g_rev[b].add(a)
        
        res = 0
        visited = set()
        
        queue = collections.deque([0])
        
        while queue:
            node = queue.popleft()
            if node in visited:
                continue
            visited.add(node)
            for nei in g_rev[node]:
                queue.append(nei)
            for nei in g[node]:
                if nei not in visited:
                    res += 1
                queue.append(nei)
                
        return res
