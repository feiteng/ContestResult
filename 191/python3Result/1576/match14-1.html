from collections import defaultdict as ddict

class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        graph = ddict(list)
        d = ddict
(list)
        
        for a,b in connections:
            graph[a].append(b)
            graph[b].append(a)
            d[a].append(b)
        
        visited = [False]*n
        
        
def reversedfs(v):
            nonlocal graph, d, visited
 
            visited[v] = True
            ans = 0
            for k in graph[v]:
                if not visited[k]:
                    if k in d[v]:
                        ans += 1
                    ans += reversedfs(k)
            return ans
            
        return reversedfs(0)
