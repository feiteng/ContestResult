import collections
class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        ans = 0
        adj = collections.defaultdict(list)
        for a,b in connections:
            adj[a].append((b,True))
            adj[b].append((a,False))
        
        q = collections.deque([0])
        visited = set()
        while q:
            city = q.popleft()
            if city in visited:
                continue
            visited.add(city)
            for adj_city,direction in adj[city]:
                if adj_city in visited:
                    continue
                q.append(adj_city)
                if direction:
                    ans += 1
        return int(ans)
