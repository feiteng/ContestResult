from collections import defaultdict

class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        adj_map = defaultdict
(list)
        for i, j in connections:
            adj_map[i].append((j, True))
            adj_map[j].append((i, False))
        stack = [0]
        visited = {0}
        count = 0
        while stack:
            city = stack
.pop()
            for neighbor, direct in adj_map[city]:
                if neighbor in visited:
                    continue
                visited.add(neighbor)
                if direct:
                    count += 1
                stack.append(neighbor)
        return count
