class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        roads = set(tuple
(c) for c in connections)
        dic = collections.defaultdict(list)
        for a, b in connections:
            dic[a].append(b)
            dic[b].append(a)
        q = collections.deque([0])
        seen = set([0])
        ret = 0
        while q:
            # print(q)
            curr = q.popleft()
            for nxt in dic[curr]:
                if nxt in seen:
                    continue
                seen.add(nxt)
                if (curr, nxt) in roads:
                    ret += 1
                q.append(nxt)
        return ret
