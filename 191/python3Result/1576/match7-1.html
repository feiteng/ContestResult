class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        edges = [[] for _ in range(n)]
        for x, y in connections:
            edges[x].append([y, 1])
            edges[y].append([x, -1])
        
        q = [0]
        v = [False] * n
        v[0] = True
        ans = 0
        
        while q:
            x = q.pop()
            for y, f in edges[x]:
                if v[y]: continue
                v[y] = True
                if f == 1: ans += 1
                q.append(y)
        return ans
