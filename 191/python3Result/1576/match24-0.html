class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        # make tree, count the number of switch
        
        graph = collections.defaultdict(list)
        graph_rev = collections.defaultdict(list)
        
        for v1, v2 in connections:
            graph[v1].append(v2)
            graph_rev[v2].append(v1)
        
        q = collections.deque([0])
        processed = set([0])
        ans = 0
        while q:
            node = q.popleft()
            for node2 in graph[node]:
                if node2 not in processed:
                    q.append(node2)
                    processed.add(node2)
                    ans += 1
            for node2 in graph_rev[node]:
                if node2 not in processed:
                    q.append(node2)
                    processed.add(node2)
                    
        return ans
