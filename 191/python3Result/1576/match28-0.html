class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        graph = [[] for i in range(n)]
        
        for a, b in connections:
            graph[a].append((b, False))
            graph[b].append((a, True))
        
        q = [0]
        visited = [False] * n
        visited[0] = True
        
        ans = 0
        while len(q) > 0:
            new_q = []
            for i in q:
                for j, way in graph[i]:
                    if visited[j]:
                        continue
                    if not way:
                        ans += 1
                    new_q.append(j)
                    visited[j] = True
            
            q = new_q
        
        return ans
