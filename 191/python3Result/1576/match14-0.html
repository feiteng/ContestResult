class Solution:
    def minReorder(self, n: int, conn: List[List[int]]) -> int:
        g1 = [set() for i in range(n)]
        g2 = [set() for i in range(n)]
        
        for i,j in conn:
            g1[i].add(j)
            g1[j].add(i)
            g2[i].add(j)
            
        
        def dfs(i,visit):
            visit[i] = True
            res = 0
            for vertex in g1[i]:
                if visit[vertex] == False:
                    if vertex in g2[i]:
                        res += 1
                    res += dfs(vertex,visit)
            return res
        
        
        
        visit = [False for i in range(n)]
        return dfs(0,visit)
