from collections import defaultdict, deque
class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        
        pre, pro = defaultdict(set), defaultdict(set)
        for u, v in connections:
            pre[v].add(u)
            pro[u].add(v)
            
        q, seen, res = deque([0]), {0}, 0
        while q:
            u = q.popleft()
            for v in pre[u]:
                if v not in seen:
                    q.append(v)
                    seen.add(v)
            for v in pro[u]:
                if v not in seen:
                    q.append(v)
                    seen.add(v)
                    res += 1
        return res
