class Solution:
    def minReorder(self, n: int, ee: List[List[int]]) -> int:
        e=[[] for _ in range(n)]
        for v in ee:
            e[v[0]].append([v[1],1])
            e[v[1]].append([v[0],0])
        ans=0
        vis=[False]*n
        stk=[0]
        while stk:
            u=stk[-1]
            if vis[u]:
                stk.pop()
                continue
            vis[u]=True
            for v,c in e[u]:
                if vis[v]:
                    continue
                ans+=c
                # print("par",u,v)
                stk.append(v)
        return ans
