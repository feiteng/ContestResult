class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:

        t = collections.defaultdict
(list)
        for u, v in connections:
            t[u].append((v, 1))
            t[v].append((u, -1))

        ans = 0

        q = collections.deque([0])
        vis = [False for _ in range(n)]
        vis[0] 
= True
        while q:
            u = q.popleft()
            for v, d in t[u]:
                if vis[v]:
                    continue
                vis[v] = True
                if d == 1:
                    ans += 1
                q.append(v)

        return ans
