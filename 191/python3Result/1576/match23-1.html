class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        nei = collections.defaultdict(set)
        mei = collections.defaultdict(set)
        for a,b in connections:
            mei[a].add(b)
            nei[b].add(a)
            
        pool = collections.deque([0])
        seen = set()
        res = 0
        while pool:
            x = pool.popleft()
            seen.add(x)
            for y in nei[x]:
                if y not in seen:
                    pool.append(y)
            for z in mei[x]:
                if z not in seen:
                    pool.append(z)
                    res += 1               
        return res
