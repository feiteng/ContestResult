class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        unordered_connection = collections.defaultdict(list)
        for a, b in connections:
            unordered_connection[a].append((b, 1))
            unordered_connection[b].append((a, 0))
        
        q = collections.deque([0])
        visited = set([0])
        ans = 0
        while q:
            node = q.popleft()
            for child, original_dir in unordered_connection[node]:
                if child not in visited:
                    visited.add(child)
                    q.append(child)
                    if original_dir == 1: ans += 1
        return ans
