class Solution:
    def minReorder(self, n: int, c: List[List[int]]) -> int:
        g = [[] for _ in range(n)]
        for a, b in c:
            g[a].append((b, 1))
            g[b].append((a, 0))

        
        def dfs(u, p):
            ret = 0
            for v, z in g[u]:
                if v == p: continue
                ret += dfs(v, u) + z
            return ret
            
        return dfs(0, 0)
