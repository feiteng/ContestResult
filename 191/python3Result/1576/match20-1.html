class Solution:
    def minReorder(self, n: int, cs: List[List[int]]) -> int:
        in_list = [set() for i in range(n)]
        out_list 
= [set() for i in range(n)]
        for x, y in cs:
            in_list[y].add(x)
            out_list[x].add(y)
        res = 0
        vs = set()
        vs.add(0)
        q = deque([0])
        
while q:
            n = q.popleft()
            for x in in_list[n]:
                if x not in vs:
                    q.append(x)
                    vs.add(x)
            for x in out_list[n]:
                if x not in vs:
                    q.append(x)
                    res += 1
                    vs.add(x)
        return res
