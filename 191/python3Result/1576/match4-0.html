class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        tree = collections.defaultdict(set)
        rev = collections.defaultdict(set)
        for a, b in connections:
            tree[b].add(a)
            rev[a].add(b)
        res = 0
        visited = set([0])
        que = collections.deque([0])
        while que:
            x = que.popleft()
            for y in tree[x]:
                if y in visited: continue
                visited.add(y)
                que.append(y)
            for y in rev[x]:
                if y in visited: continue
                visited.add(y)
                que.append(y)
                res += 1
        return res
