class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        # wait, connections.length == n-1, this is easy
        
        forwardsdict = {x:[] for x in range(n)}
        backwarddict = {x:[] for x in range(n)}
        for c in connections:
            forwardsdict[c[0]].append(c[1])
            backwarddict[c[1]].append(c[0])
        
        #print(forwardsdict)
        #print(backwarddict)
        
        final = 0
        seen = {0}
        tocheck = [0]
        while len(tocheck)!=0:
            newtocheck = []
            for city in tocheck:
                for connect in backwarddict[city]:
                    if connect not in seen:
                        seen.add(connect)
                        newtocheck.append(connect)
                for connect in forwardsdict[city]:
                    if connect not in seen:
                        seen.add(connect)
                        newtocheck.append(connect)
                        final+=1
            tocheck = newtocheck
        #print(seen)
        return final
