class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        
        outs = collections.defaultdict(list)
        ins = collections.defaultdict(list)
        
        for i,j in connections:
            outs[i].append(j)
            ins[j].append(i)
            
        # start from citi 0 find all its outwards citis and reorder the path
        queue = collections.deque()
        queue.append(0)
        res = 0
        seen = {0}
        # bfs
        while queue:
            i = queue.popleft()
            for j in ins[i]:
                if j not in seen:
                    # fina its indegree node
                    queue.append(j)
                    seen.add(j)
                
            for j in outs[i]:
                if j not in seen:
                    queue.append(j)
                    seen.add(j)
                    res += 1
            # first extend
        return res
