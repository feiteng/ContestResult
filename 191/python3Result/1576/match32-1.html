import collections

class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        graph = collections.defaultdict(list)
        dest = collections.defaultdict(list)
        for s, d in connections:
            graph[d].append(s)
            dest[s].append(d)
        
        edges = set()
        def bfs(s, graph, dest, edges):
            visited = set([s])
            
            count = 0
            queue = collections.deque([s])
            
            while len(queue) > 0:
                cur 
= queue.popleft()
                for n in graph[cur]:
                    if n not in visited:
                        visited.add(n)
                        queue.append(n)
                for u in dest[cur]:
                    if u not in visited:
                        visited.add(u)
                        queue.append(u)
                        count += 1
            return count
        return bfs(0, graph, dest, edges)
