class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        edges = [[] for i in range(n)]
        for a, b in connections:
            edges[a].append((b, 1))
            edges[b].append((a, 0))
        
        def dfs(u, parent):
            ans = 0
            for v, e in edges[u]:
                if v != parent:
                    ans += e + dfs(v, u)
            return ans
        
        return dfs(0, -1)
