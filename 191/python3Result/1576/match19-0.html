class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        edges = [[] for _ in range(n)]
        for a, b in connections:
            edges[a].append([b, 1])
            edges[b].append([a, 0])
        
        def dfs(root, fa):
            cost = 0
            for x, c in edges[root]:
                if x == fa: continue
                #print (root, x, c)
                cost += c
                cost += dfs(x, root)
            return cost
        
        #print ("xxxx")
        return dfs(0, -1)
