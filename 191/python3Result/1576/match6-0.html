class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        adj = [[] for _ in range(n)]
        for u, v in connections:
            adj[u].append((v, 1))
            adj[v].append((u, 0))
        
        result = 0
        seen = {0}
        q = [0]
        for u in q:
            for v, d in adj[u]:
                if v not in seen:
                    seen.add(v)
                    q.append(v)
                    result += d
        return result
