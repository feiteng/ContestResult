import math
class Solution:
    def getProbability(self, balls: List[int]) -> float:
        N = sum(balls)
        k = len(balls)
        
        res = 0
        
        def helper(i, left, right, left_count, prev):
            nonlocal res
            if i == k:
                if left == right and left_count == N // 2:
                    res += prev
                return
            # balls[i]: total num of this color
            for j in range(balls[i] + 1):
                if j == 0:
                    helper(i+1, left, right + 1, left_count + j, prev)
                elif j == balls[i]:
                    helper(i+1, left+1, right, left_count + j, prev)
                else:
                    helper(i+1, left+1, right+1, left_count + j, prev * self.comb(balls[i], j))
                
            
        helper(0, 0, 0, 0, 1)

        return res * 1.0 / self.comb(N, N//2)
    
    def comb(self, n, p):
        return math.factorial(n) / (math.factorial(p) * math.factorial(n-p))
