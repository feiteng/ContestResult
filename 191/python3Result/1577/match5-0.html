class Solution:
    def getProbability(self, balls: List[int]) -> float:
        
        def count(balls):
            n = sum(balls)
            ans = math.factorial(n)
            for num in balls:
                ans //= math.factorial(num)
            return ans

        def bt(n, leNum, riNum, le, ri, k):
            if k == len(balls):
                le1 = [i for i in le if i>0]
                ri1 = [i for i in ri if i>0]
                if leNum == riNum and len(le1) == len(ri1):
                    self.ans += count(le1) * count(ri1)
            else:
                for i in range(balls[k] + 1):
                    leNum2 = i + leNum
                    riNum2 = riNum + balls[k] - i
                    if leNum2 <= n and riNum2 <= n:
                        bt(n, leNum2, riNum2, le + [i], ri + [balls[k] - i], k+1)
        self.ans = 0
        n = sum(balls)
        tot = count(balls)
        bt(n//2, 0, 0, [], [], 0)
        return self.ans / tot
