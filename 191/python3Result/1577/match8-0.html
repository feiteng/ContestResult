class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n = sum(balls) // 2
        
        def count_comb(arr):
            total = math.factorial(sum(arr))
            for a in arr:
                total /= math.factorial(a)
            
            return total
        
        
        total = ans = 0
        for A in itertools.product(*[range(ball + 1) for ball in balls]):
            if sum(A) == n:
                B = [balls[i] - a for i, a in enumerate(A)]
                comb = count_comb(A) * count_comb(B)
                
                total += comb
                if sum(a > 0 for a in A) == sum(b > 0 for b in B):
                    ans += comb
        
        return ans / total
