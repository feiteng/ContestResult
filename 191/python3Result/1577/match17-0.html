from functools import lru_cache
import math

class Solution:
    def getProbability(self, balls: List[int]) -> float:
        numchoices = math.factorial(sum(b for b in balls))
        for b in balls:
            numchoices //= math.factorial(b)
        # print(f"numchoices = {numchoices}")
        numgood = 0
        

# After that we add the first two balls to the first box and the second two balls to the second box.
#We can see that 8 of these 12 possible random distributions have the same number of distinct colors of balls in  # each box.
        
        # [1,1 / 2,3], [1,1 / 3,2], [1,2 / 1,3], [1,2 / 3,1], [1,3 / 1,2], [1,3 / 2,1], [2,1 / 1,3],
        # [2,1 / 3,1], [2,3 / 1,1], [3,1 / 1,2], [3,1 / 2,1], [3,2 / 1,1]
        
        def getnum(B):
            num = math.factorial(sum(b for b in B))
            for b in B:
                num //= math.factorial(b)
            return num
        
        box1, box2 = [0 for _ in range(len(balls))], [0 for _ in range(len(balls))]
        def dfs(i):
            nonlocal numchoices, numgood
            if i == len(balls):
                if sum(box1) == sum(box2):
                    # numchoices += 1
                    a = sum(1 for i in range(len(box1)) if box1[i] > 0)
                    b = sum(1 for i in range(len(box2)) if box2[i] > 0)
                    if a == b:
                        # print(f"box1 = {box1}")
                        # print(f"box2 = {box2}")
                        # print("---")
                        # numgood += (math.factorial(a)*math.factorial(b))
                        numgood += (getnum(box1)*getnum(box2))
                return
            for x in range(balls[i]+1):
                box1[i], box2[i] = x, balls[i] - x
                dfs(i + 1)
                box1[i], box2[i] = 0, 0
        
        dfs(0)
        # print(numgood, numchoices)
        # print(f"~~~~")
        return numgood/numchoices
