class Solution:
    def getProbability(self, AA: List[int]) -> float:
        n = sum(AA) // 2
        k = len(AA)
        
        def f(A, idx) :
            sum_a = sum(A)
            if sum_a > n :
                return 0
            if idx == k :
                if sum_a != n :
                    return 0
                cnt0 = cnt1 = 0
                for i in range(k) :
                    if A[i] : cnt0 += 1
                    if AA[i] - A[i] : cnt1 += 1
                if cnt0 != cnt1 :
                    return 0
                ret = math.factorial(n) ** 2
                for i in range(k):
                    ret //= math.factorial(A[i])
                    ret //= math.factorial(AA[i] - A[i])
                #print(A, ret)
                return ret
            
            if sum_a == n :
                return f(A, idx + 1)
                
            ret = 0
            for i in range(AA[idx] + 1):
                A[idx] = i
                ret += f(A, idx + 1)
                A[idx] = 0
            return ret
                
        
        total = math.factorial(sum(AA))
        for a in AA :
            total //= math.factorial(a)
        ret = f([0] * k, 0) / total
        return ret
