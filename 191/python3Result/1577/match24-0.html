from math import comb
class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n = sum(balls) // 2
        k = len(balls)
        a = [0] * k
        self.ans = 0.0
        
        def check():
            n1, n2 = 0, 0
            for i in range(k):
                if a[i] > 0:
                    n1 += 1
                if balls[i] - a[i] > 0:
                    n2 += 1
            return n1 == n2
        
        def check_same():
            for i in range(k):
                if a[i] * 2 != balls[i]:
                    return False
            return True
        
        def search(i, left):
            if i == k:
                if left == 0 and check():
                    temp = 1.0
                    n1, n2 = n, n
                    for i in range(k):
                        temp *= comb(n1, a[i]) * comb(n2, balls[i] - a[i])
                        n1 -= a[i]
                        n2 -= balls[i] - a[i]
                    self.ans += temp
                return
            
            for j in range(0, min(left, balls[i]) + 1):
                a[i] = j
                search(i + 1, left - j)
                
        search(0, n)
        total = 1.0
        n1 = 2 * n
        for i in range(k):
            total *= comb(n1, balls[i])
            n1 -= balls[i]
        return self.ans / total
