class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n 
= sum(balls)//2
        nfac = math.factorial(n)
        denom = math.factorial(n*2)
        for b in balls:
            denom //= math.factorial(b)
        
        
        # num ways for first box to get count tup
        def ways(tup):
            # mult combinations * n!
            cnt = nfac*nfac
            for i, k in enumerate(tup):
                cnt //= math.factorial(k)
                cnt //= math.factorial(balls[i]-k)
            return cnt
        possible_tups = [range(b+1) for b in balls]
        tot = 0
        for tup in itertools.product(*possible_tups):
            if sum(tup) == n:
                l = tup.count(0)
                r = [balls[i] - tup[i] for i in range(len(balls))].count(0)
                if l == r:
                    # print(tup)
                    w = ways(tup)
                    # print(w)
                    tot += w
        print(tot, denom)
        return tot/denom
