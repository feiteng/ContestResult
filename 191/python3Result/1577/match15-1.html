class Solution:

    def getProbability(self, balls: List[int]) -> float:
        balls.sort(reverse = True)
        m = len(balls)
        half = sum(balls) // 2
        cur1 = []
        cur2 = []
        self.a = 0
        self.b = 0
        fp = [1]
        for i in range(1, 25):
            fp.append(fp[-1] * i)
        # print(f'{fp=}')
    
        # @lru_cache(None)
        def dp(start):
            if start == len(balls):
                if sum(cur1) != half:
                    return
                add_b = fp[half]
                for cc in cur1:
                    add_b //= fp[cc]

                add_b *= fp[half]
                for cc in cur2:
                    add_b //= fp[cc]
                self.b += add_b
                cx = Counter(cur1)
                cy = Counter(cur2)
                if cx[0] == cy[0]:
                    self.a +
= add_b
            else:
                for i in range(0, balls[start]+1):
                    cur1.append(i)
                    cur2.append(balls[start] - i)
                    if sum(cur1) <= half and sum(cur2) <= half:
                        dp(start + 1)
                    cur1.pop()
                    cur2.pop()
         
        dp(0)
        # print(f'{self.a=},{self.b=}')
        return self.a / self.b
