class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n = len(balls)
        total = sum(balls) // 2
        first = [0] * n
        all_pos = 0
        valid = 0
        def c(n, m):
            return math.factorial(n) / math.factorial(n-m) / math.factorial(m)
        def cal(arr, s):
            res = 1
            for x in arr:
                if x > 0:
                    res *= c(s, x)
                s -= x
            return res
        def rec(i, s):
            if i == n:
                if s != total: return
                nonlocal all_pos
                nonlocal valid
                t = cal(first, s) * cal([balls[j] - first[j] for j in range(n)], s)
                all_pos += t
                a = 0
                b = 0
                for j in range(n):
                    if first[j] != 0:
                        a += 1
                    if balls[j] - first[j] != 0:
                        b += 1
                if a == b: valid += t
                return
            for x in range(balls[i] + 1):
                if s + x > total: break
                first[i] = x
                rec(i+1, s + x)
            first[i] = 0
        rec(0, 0)
        return valid / all_pos
