class Solution:
    def getProbability(self, balls: List[int]) -> float:
        from math import factorial as F
        
        
def C(y):
            x = sum(y)
            x = F(x)
            for yi in y:
                x /= F(yi)
            return x
        
        s = sum(balls)
        
        def f(idx, x1, x2, y1, y2):
            if sum(y1) > s / 2 or sum(y2) > s / 2: 
                # print(sum(y1), sum(y2))
                return 0
            
            if idx == len(balls):
                if x1 != x2 or sum(y1) != s // 2: return 0
                return C(y1) * C(y2)
            
            ret = 0
            for i in range(balls[idx] + 1):
                
                z1 = y1[:]
                z2 = y2[:]
                z1[idx] = i
                z2[idx] = balls[idx] - i
                ret += f(idx + 1, x1 + (i != 0), x2 + (balls[idx] - i != 0), z1, z2)
                
            return ret
        
        return f(0, 0, 0, [0 for _ in range(len(balls))], [0 for _ in range(len(balls))]) / C(balls)
