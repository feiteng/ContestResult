class Solution:
    def getProbability(self, balls: List[int]) -> float:
        def fact(n):
            ans=1
            for i in range(1,n+1):
                ans*=i
            return ans
        def possible(arr):
            sm=sum(arr)
            ans=fact(sm)
            for n in arr:
                ans/=fact(n)
            return ans
        sm=sum(balls)
        t=sm//2
        ans=0
        def dfs(remain, got, beg, t, acc):
            nonlocal ans
            if t==0 or beg == len(remain):
                if t==0:
                    c1=sum(x>0 for x in got)
                    c2=sum(x>0 for x in remain)
                    if c1==c2:
                        ans+=possible(got)*possible(remain)
                return
            acc-=remain[beg]
            for i in range(0, remain[beg]+1):
                if i>t:break
                if acc<t-i:continue
                remain[beg]-=i
                got[beg]+=i
                dfs(remain,got,beg+1,t-i, acc)
                got[beg]-=i
                remain[beg]+=i
        dfs(balls, [0]*len(balls), 0, t, sm)
        return ans/possible(balls)
