from itertools import product
class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n = len(balls)
        prod = product(*[range(x + 1) for x in balls])
        case = 0
        for num in prod:
            if sum(num) != sum(balls) // 2:
                continue
            num1, num2 
= 0, 0
            for i in range(n):
                if num[i] != 0:
                    num1 += 1
                if num[i] != balls[i]:
                    num2 += 1
            if num1 != num2:
                continue
            ans = 1
            for i in range(n):
                ans *= combine(balls[i], num[i])
            case += ans
        n = sum(balls) // 2
        ans = 1.0
        for i in range(1, n + 1):
            ans *= i / (n + i)
        return ans * case
            
def combine(n, m):
    ans = 1
    for i in range(1, m + 1):
        ans *= (n - m + i) / i
    return ans
