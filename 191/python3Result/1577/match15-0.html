class Solution:
    def getProbability(self, balls: List[int]) -> float:
        total_balls = sum(balls)
        d = {}
        def factorial(n):
            if n in d:
                return d[n]
            elif n == 1 or n == 0:
                return 1
            d[n] = factorial(n-1) * n
            return d[n]
        self.n_valid = 0
        def helper(i, left, right):
            if i == len(balls):
                n_left = 0
                n_right = 0
                n_left_color = 0
                n_right_color = 0
                for l, r in zip(left, right):
                    if l > 0:
                        n_left_color += 1
                        n_left += l
                    if r > 0:
                        n_right_color += 1
                        n_right += r
                if n_left == n_right:
                    if n_left_color == n_right_color:
                        left_value = factorial(n_left)
                        for l in left:
                            left_value //= factorial(l)
                        right_value = factorial(n_right)
                        for r in right:
                            right_value //= factorial(r)
                        self.n_valid += left_value * right_value
                return
            for v in range(balls[i]+1):
                left.append(v)
                right.append(balls[i] - v)
                if sum(left) <= total_balls//2 and sum(right) <= total_balls//2:
                    helper(i+1, left, right)
                left.pop()
                right.pop()
        helper(0, [], [])
        total = factorial(total_balls)
        for b in balls:
            total //= factorial(b)
        return self.n_valid/total
