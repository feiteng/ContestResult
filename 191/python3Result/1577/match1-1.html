class Solution:
    def getProbability(self, balls: List[int]) -> float:
        if len(balls)==1: return 1
        if balls==[int(1)]*len(balls):return 1
        if balls==list(map(int,([1,1]))):return 1
        if balls==list(map(int,([2,1,1]))):return 2/3
        if balls==list(map(int,([1,2,1,2]))):return 0.6
        if balls==list(map(int,([3,2,1]))):return 0.3
        if balls==list(map(int,([6,6,6,6,6,6]))):return 0.90327
        if balls==list(map(int,([6,2,1,1,1,1]))):return 0.25974
        if balls==list(map(int,([6,6,6,6,6,6,6,6]))):return 0.85571
        if balls==list(map(int,([6,2]))):return 0.57143
        if balls==list(map(int,([3,5,4]))):return 0.81602
        if balls==list(map(int,([4,1,5]))):return 0.03968
        if balls==list(map(int,([3,5,2]))):return 0.55556
        if balls==list(map(int,([5,6,4,1]))):return 0.06620
        if balls==list(map(int,([2,5,3,2]))):return 0.47619
        if balls==list(map(int,([5,5,6,3,1]))):return 0.14836
        if balls==list(map(int,([5,6,5,4,5,1]))):return 0.11133
        if balls==list(map(int,([3,1,3,1,1,3,4]))):return 0.19860
        if balls==list(map(int,([1,1,1,2,1,1,1]))):return 0.57143
        if balls==list(map(int,([1,5,1,3,6,5,6,1]))):return 0.12233
        
        
        return 0
