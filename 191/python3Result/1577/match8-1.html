class Solution:
    def getProbability(self, balls: List[int]) -> float:
        target = sum(balls) // 2
        print(target)
        
        ans = 0
        
        def count(target, counter):
            total = math.factorial(target)
            for v in counter.values():
                total //= math.factorial(v)
            return total
            
        total = 0
        for me in itertools.product(*[range(b+1) for b in balls]):
            # print(me)
            if sum(me) == target:
                
                other = [b - m for b, m in zip(balls, me)]
                # print(me,other)
                cme = collections.Counter()
                cother = collections.Counter()
                for i, c in enumerate(me):
                    if c > 0:
                        cme[i] += c
                for i, c in enumerate(other):
                    if c > 0:
                        cother[i] += c
                
                # print(cme, cother)
                comb 
= count(target, cme) * count(target, cother)
                total += comb
                if len(cme) == len(cother):
                    ans += comb
        
        return ans / total
