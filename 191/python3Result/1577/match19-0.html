class Solution:
    def getProbability(self, balls: List[int]) -> float:
        t_num = sum(balls)
        t_color = len(balls)
        t_comb = math.factorial(t_num//2)
        
        t_comb2 = math.factorial(t_num)
        for num in balls:
            t_comb2 /= math.factorial(num)
        
        balls_copy = balls.copy()
        
        ans = 0
        def dfs(idx, aval, color, num, rem_color):
            nonlocal ans
            if num == t_num//2 and color == rem_color: 
                t = t_comb
                t2 = t_comb
                for idx, cnt in enumerate(aval):
                    t /= math.factorial(cnt)
                    t2 /= math.factorial(balls_copy[idx] - aval[idx])
                ans += t * t2
                return
            if idx >= len(balls): return
            for take in range(aval[idx] + 1):
                temp = aval[idx]
                aval[idx] -= take
                if take == 0:
                    dfs(idx + 1, aval, color, num, rem_color)
                elif temp == take: 
                    dfs(idx + 1, aval, color + 1, num + take, rem_color - 1)
                else: 
                    dfs(idx + 1, aval, color + 1, num + take, rem_color)
                aval[idx] += take
        dfs(0, balls, 0, 0, t_color)
        
        return ans/t_comb2
