class Solution:
    def getProbability(self, balls: List[int]) -> float:
        self.n = sum(balls)//2
        
        #calculate how many possible ways to distribute
        def base(n):
            return math.factorial(2*n)/math.factorial(n)/math
.factorial(n)
        
        def choose(m,k):
            return math.factorial(m)/math.factorial(m-k)/math.factorial(k)
        
        denominator = base(self.n)
        #print(real_balls)
        self.res = 0
        
        def dfs(balls,index,count1,count2,diff1,diff2,times):
            if index==len(balls):
                if diff1==diff2:
                    self.res+=1*times
                return
            
            #print(index,count1,count2,diff1,diff2)
            if count1>self.n or count2>self.n:
                return
            val = balls[index]  
            for give1 in range(0,val+1):
                give2 = val-give1
                
                if give1>self.n-count1:
                    break
                if give2>self.n-count2:
                    continue
                nt = times*choose(val,give1)
                add1 = 1 if give1!=0 else 0
                add2 = 1 if give2!=0 else 0
                dfs(balls,index+1,count1+give1,count2+give2,diff1+add1,diff2+add2,nt)
        
        dfs(balls,0,0,0,0,0,1)
                
                
        
        return self.res/denominator
