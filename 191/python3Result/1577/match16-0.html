class Solution:
    def getProbability(self, balls: List[int]) -> float:
        num = sum(balls)
        total = math.factorial(num)  
        sss = [] 
        for i, n in enumerate(balls): 
            total /= math.factorial(n)
            temp = [] 
            if not sss:
                temp.extend([[j] for j in range(n+1) if j <= num // 2])
            else:
                for s in sss: 
                    temp.extend([s + [j] for j in range(n+1) if sum(s)+j <= num // 2])
                    
            sss = temp
        cnt = 0
        for s in sss:
            if sum(s) == num // 2:
                buji = [balls[i]-j for i, j in enumerate(s)]
                if sum(1 for i in s if i > 0) == sum(1 for i in buji if i > 0): 
                    t = math.factorial(num//2)**2 
                    for i in s:
                        t /= math.factorial(i)  
                    for i, j in enumerate(s):
                        t /= math.factorial(balls[i]-j)
                    cnt += t
        return cnt / total
