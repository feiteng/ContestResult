class Solution:
    def getProbability(self, balls: List[int]) -> float:
        self.balls = balls
        combs = self.search(0, 0, sum(balls) // 2, [])
        return sum([self.calccomb(c) for c in combs]) / calccomb_singleside(balls)
        
    def search(self, idx, colordiff, vac, hist):
        if vac < 0:
            return []
        if idx == len(self.balls):
            return [tuple(hist)] if colordiff == 0 and vac == 0 else []
        res = []
        for cnt in range(self.balls[idx] + 1):
            hist.append(cnt)
            res.extend(self.search(idx + 1, colordiff + (1 if cnt == self.balls[idx] else -1 if cnt == 0 else 0), vac - cnt, hist))
            hist.pop()
        return res
    
    def calccomb(self, side1):
        side2 = [x - y for x, y in zip(self.balls, side1)]
        return calccomb_singleside(side1) * calccomb_singleside(side2)
    

def calccomb_singleside(arr):
    res = math.factorial(sum(arr))
    for n in arr:
        res /= math.factorial(n)
    return res
