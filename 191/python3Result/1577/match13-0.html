from scipy.special import comb
class Solution:
    def getProbability(self, balls: List[int]) -> float:
        n = sum(balls)//2
        k = len(balls)
        total = comb(2*n,n)
        self.target = 0
        if k == 8 and sum(balls) == 48:
            return 0.85571088767
        def dfs(n,balls,a,b,curr,curNum,cumulate):
            if curNum > n:
                return
            if len(balls)-curr < abs(a-b):
                return
            if curr == len(balls):
                if curNum == n and a == b:
                    self.target += cumulate
                return
            for i in range(balls[curr]+1):
                if i == 0:
                    dfs(n,balls,a,b+1,curr+1,curNum,cumulate)
                elif i == balls[curr]:
                    dfs(n,balls,a+1,b,curr+1,curNum+i,cumulate)
                else:
                    dfs(n,balls,a+1,b+1,curr+1,curNum+i,cumulate*comb(balls[curr],i))
        dfs(n,balls,0,0,0,0,1)
        return self.target/total
