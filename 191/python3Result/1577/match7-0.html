import math

class Solution:
    def getProbability(self, balls: List[int]) -> float:
        def perms(arr):
            res = math.factorial(sum(arr))
            
            for n in arr:
                res //= math.factorial(n)
                
            return res
        
        
        denom = perms(balls)
        numer = [0]
        
        def split(a, b, i):
            if i == len(balls):
                if sum(a) != sum(b):
                    return
                
                if len(list(filter(lambda x: x > 0, a))) != len(list(filter(lambda x: x > 0, b))):
                    return
                
                numer[0] += perms(a) * perms(b)
                return
            
            for c in range(balls[i] + 1):
                split(a + [c], b + [balls[i] - c], i + 1)
                
        split([], [], 0)
        return numer[0] / denom
