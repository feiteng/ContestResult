class Solution:
    def getProbability(self, balls: List[int]) -> float:
        
        def nAk(n, k):
            ans = 1
            for i in range(k):
                ans *= n - i
            return ans
        def nCk(n, k):
            ans = 1
            for i in range(k):
                ans *= n - i
                ans /
/= i + 1
            return ans
        
        nn = sum(balls)
        n = nn // 2
        total = 0
        def dfs(i, b1, c1, n1, b2, c2, n2):
            nonlocal total
            if i == len(balls):
                if c1 == c2 and n1 == n2:
                    r = 1
                    for ki, ni in zip(b1, balls):
                        r *= nCk(ni, ki)
                    r *= nAk(n, n)
                    total += r
                return
            if n1 > n or n2 > n:
                return
            for x in range(balls[i]+1):
                dc1 = 1 if x != 0 else 0
                dc2 = 1 if balls[i] - x != 0 else 0
                b1.append(x)
                b2.append(balls[i] - x)
                dfs(i+1, b1, c1+dc1, n1+x, b2, c2+dc2, n2+balls[i] - x)
                b1.pop()
                b2.pop()  
        
        dfs(0, [], 0, 0, [], 0, 0)
        return float(total / nAk(nn, n))
