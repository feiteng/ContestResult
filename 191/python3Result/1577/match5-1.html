import math
class Solution:
    def getProbability(self, balls: List[int]) -> float:
        m = len(balls)
        b1 = [None] * m
        b2 = [None] * m
        self.cnt = 0
        
        
def calc(b):
            s = sum(b)
            ans = math.factorial(s)
            for x in b:
                ans //= math.factorial(x)
            return ans
        
        def dfs(i):
            if i == m:
                # print(b1, b2)
                if sum(b1) == sum(b2) and sum(b > 0 for b in b1) == sum(b > 0 for b in b2):
                    self.cnt += calc(b1) * calc(b2)
                return
            for j in range(balls[i] + 1):
                b1[i] = j
                b2[i] = balls[i] -  j
                dfs(i + 1)
        dfs(0)
        tot = calc(balls)
        # print(self.cnt, tot)

        return self.cnt / tot
