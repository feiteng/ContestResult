class Solution:
    def getProbability(self, balls: List[int]) -> float:
        N = len(balls)
        num = sum(balls) // 2
        self.tot = 0
        @functools.lru_cache(None)
        def fac(n):
            if n == 1:
                return 1
            return n * fac(n - 1)
        
        def perm(balls):
            n = sum(balls)
            res = fac(n)
            for m in balls:
                if m > 0:
                    res //= fac(m)
            return res
        
        def dfs(l_balls, s, cur_t):
            if cur_t > num:
                return
            if len(l_balls) == N:
                r_balls = [balls[i] - l_balls[i] for i in range(N)]
                if sum
(l_balls) == num and sum([x > 0 for x in l_balls]) == sum([x > 0 for x in r_balls]):
                    self.tot += perm(l_balls) * perm(r_balls)
                return
            for n in range(balls[s] + 1):
                dfs(l_balls + [n], s + 1, cur_t + n)
        dfs([], 0, 0)
        return self.tot / float(perm(balls))
