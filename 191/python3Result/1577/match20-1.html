import math

def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)

class Solution:

    def getProbability(self, balls: List[int]) -> float:
        solutions = [0]
        good_solutions = [0]
        
        def isGood(balls, leftBalls):
            rightBalls = [balls[i] - leftBalls[i] for i in range(len(balls))]
            return len([x for x in leftBalls if x != 0]) == len([x for x in rightBalls if x != 0])
        
        def dfs(i, balls, leftBalls, solutions, good_solutions):
            if i >= len(balls):
                if sum(leftBalls) == sum(balls) // 2:
                    this_division = 1
                    for i in range(len(leftBalls)):
                        this_division *= nCr(balls[i], leftBalls[i])
                    solutions[0] += this_division
                    good_solutions[0] += this_division if isGood
(balls, leftBalls) else 0
                return
            for k in range(0, balls[i] + 1):
                dfs(i + 1, balls, leftBalls + [k], solutions, good_solutions)
        
        dfs(0, balls, [], solutions, good_solutions)
        return good_solutions[0] / solutions[0]
