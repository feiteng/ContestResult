class Solution:
    def __init__(self,):
        self.flag = None
        self.num = 0
        self.fun = [0,1]
        for x in range(2,50):
            self.fun.append(x*self.fun[x-1])
        # print(self.fun)
    
    def dfs(self,balls,idx,n):
        # print(idx,self.flag,n)
        if sum(self.flag) == n:
            # print(self.flag)
            t1 = 0
            t2 = 0
            for x,y in zip(self.flag,balls):
                if x > 0:
                    t1 +=1 
                if x != y:
                    t2 += 1
            if t1 != t2:
                return
            tmp = self.fun[n] * self.fun[n]
            for x,y in zip(self.flag,balls):
                if x != 0:
                    tmp /= self.fun[x]
                if y-x != 0:
                    tmp /= self.fun[y-x]
            self.num += tmp
            # print(self.flag,tmp,self.num)
            return 
        
        if idx >= len(balls):
            
return
        
        for i in range(balls[idx]+1):
            self.flag[idx] = i
            # print(idx,i)
            self.dfs(balls,idx+1,n)
            self.flag[idx] = 0

        
    def getProbability(self, balls: List[int]) -> float:
        total  = self.fun[sum(balls)]
        for x in balls:
            total /= self.fun[x]
        self.flag = [0]*len(balls)
        self.dfs(balls, 0, sum(balls)//2)
        # print(self.num, total)
        return self.num / total
