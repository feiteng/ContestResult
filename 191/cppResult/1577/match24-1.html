vector<int> a;
int n;
double ans;
int m;
double ms;
int fa[100], fb[100];
double fac[100];

void dfs(int p, int x, int y, int da, int db, double ma, double mb) {
    if (p >= n) {
        if (x != y || da != db) return;
        ans += ma * mb;
        return;
    }
    
    for (int i = 0; i <= a[p]; i++) {
        int nx = x + i;
        int ny = y + a[p] - i;
        int nda = da, ndb = db;
        if (i != 0) nda++;
        if (i != a[p]) ndb++;
        double nma = ma / fac[i], nmb = mb / fac[a[p] - i];
        dfs(p + 1, nx, ny, nda, ndb, nma, nmb);
    }
    
}

class Solution {

public:
    double getProbability(vector<int>& balls) {
        fac[0] = 1;
        for (int i = 1; i <= 10; i++)
            fac[i] = fac[i-1] * i;
        
        a = balls;
        n = a.size();
        
        m = 0; ans = 0;
        for (int t : a) m += t;
        m /= 2;
        ms = 1;

        for (int i = 1; i <= m; i++)
            ms *= i;
        dfs(0, 0, 0, 0, 0, ms, ms);

        for (int i = 1; i <= 2 * m; i++)
            ans /= i;
        for (int i = 0; i < n; i++)
            ans *= fac[a[i]];
        return ans;
    }
};
