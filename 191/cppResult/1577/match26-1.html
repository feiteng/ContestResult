double C(int n, int m) {
    double ans = 1.0;
    for (int i = 1; i <= m; ++i)
        ans *= (double)(n - i + 1) / i;
    return ans;
}

class Solution {
    vector<int> a, l;
    int sum;
    double ans;


    void dfs(int x) {
        if (x == a.size()) {
            int lc = 0, rc = 0, cnt = 0;
            for (int i = 0; i < a.size(); ++i) {
                cnt += l[i];
                if (l[i] > 0) ++lc;
                if (a[i] - l[i] > 0) ++rc;
            }
            if (cnt != sum - cnt) return;
            if (lc == rc) {
                double prob = 1.0 / C(sum, cnt);
                for (int i = 0; i < a.size(); ++i) {
                    prob *= C(a[i], l[i]);

                }
                ans += prob;
            }
            // if (lc == rc) ++ans;
            // ++tot;
            return;
        }
        for (int i = 0; i <= a[x]; ++i) {
            l[x] = i;
            dfs(x + 1);
        }
    }

public:
    double getProbability(vector<int>& balls) {
        a = balls;
        l = vector<int>(a.size());
        // tot = ans = 0;
        ans = 0;
        sum = accumulate(a.begin(), a.end(), 0);
        dfs(0);
        return ans;
        // return (double)ans / tot;
    }
};
