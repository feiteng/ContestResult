class Solution {
public:
	double ans = 0.0;
	int tot = 0, right = 0, s = 0;
	vector <int> x;
	void dfs(int i, int s1, int s2, int c1, int c2, double pro) {
		if (s1 > s / 2 || s2 > s / 2) {
			return;
		}
		if (i == x.size()) {
			if (c1 == c2) {
				ans += pro;
			}
			return;
		}
		for (int now = 0; now <= x[i]; now++) {
			dfs(i + 1, s1 + now, s2 + (x[i] - now), c1 + (now > 0), c2 + (now < x[i]),
					pro * f[s / 2 - s1][s / 2 - s2][x[i]][now]);
		}
	}

	double f[33][33][8][8];
	double getProbability(vector<int>& a) {
		x = a;
		int n = a.size();
		s = 0;
		for (int i= 0; i < n; i++) {
			s += a[i];
		}

		int mx = s / 2;
		for (int i = 0; i <= mx; i++) {
			for (int j = 0; j <= mx; j++) {
				for (int l = 0; l <= 6; l++) {
					for (int r = 0; r <= 6; r++) {
						f[i][j][l][r] = 0.0;
					}
				}
				f[i][j][0][0] = 1.0;
			}
		}

		for (int i = 0; i <= mx; i++) {

			for (int j = 0; j <= mx; j++) {
				for (int l = 0; l < 6; l++) {
					for (int r = 0; r <= l; r++) {
						double ll = i - r;
						double rr = j - (l - r);
						if (ll > -1e-9 && rr > -1e-9) {
							f[i][j][l + 1][r] += f[i][j][l][r] * rr / (ll + rr);
							f[i][j][l + 1][r + 1] += f[i][j][l][r] * ll / (ll + rr);
						}
					}
				}
			}
		}

		ans = 0;
		dfs(0, 0, 0, 0, 0, 1.0f);
		return ans;
	}
};
