class Solution {
    double fz;
    vector<int> tmp;
    int kd;
    void dfs(vector<int> &x,int n,int pos)
    {
        if(pos==n)
        {
            int sum=0,cnt1=0,cnt2=0;
            for(int i=0;i<n;i++)
            {
                sum+=tmp[i];
                cnt1+=(tmp[i]>0);
                cnt2+=(x[i]-tmp[i]>0);
            }
            if(sum==kd&&cnt1==cnt2)
            {
                double tot=1;
                for(int i=1;i<=sum;i++)
                    tot*=i;
                for(int i=0;i<n;i++)
                {
                    if(tmp[i])
                    {
                        for(int j=1;j<=tmp[i];j++)
                            tot/=j;
                    }
                }
                
                for(int i=1;i<=sum;i++)
                    tot*=i;
                for(int i=0;i<n;i++)
                {
                    if(x[i]-tmp[i])
                    {
                        for(int j=1;j<=x[i]-tmp[i];j++)
                            tot/=j;
                    }
                }
                fz+=tot;
            }
            return ;
        }
        
        for(int i=0;i<=x[pos];i++)
        {
            tmp[pos]=i;
            dfs(x,n,pos+1);
        }
    }
public:
    double getProbability(vector<int>& balls) 
    {
        int n=balls.size();
        tmp.clear();
        tmp.resize(n);
        int sum=0;
        for(auto &e: balls)
            sum+=e;
        kd=sum/2;
        double tot=1;
        for(int i=1;i<=sum;i++)
            tot*=i;
        for(auto &e: balls)
        {
            for(int i=1;i<=e;i++)
                tot/=i;
        }
        
        fz=0;
        dfs(balls,n,0);
        return fz/tot;
        
        
    }
};
