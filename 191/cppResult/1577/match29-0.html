class Solution {
public:
    double getProbability(vector<int>& balls) {
        int n = 0, k = balls.size();
        for (int b : balls) n += b;
        n /= 2;
        vector<vector<double>> dp(4 * n + 1, vector<double>(2 * k + 1));
        dp[2 * n][k] = 1;
        for (int b : balls) {
            // cout << "start"<< b << endl;
            vector<vector<double>> dp2(4 * n + 1, vector<double>(2 * k + 1));
            int comb = 1;
            for (int t = 0; t <= b; t++) {
                // cout << b << " "<< t << " " << comb << endl;
                int dn = t - (b - t);
                for (int i = 0; i < dp.size(); i++) {
                    for (int j = 0; j < dp[0].size(); j++) {
                        if (dp[i][j] > 0) {
                            // cout << i + dn << endl;
                            // cout << j - 1 << " " << j + 1 << endl;
                            if (t == 0) {
                                dp2[i + dn][j - 1] += comb * dp[i][j];
                            }
                            else if (t == b) {
                                dp2[i + dn][j + 1] += comb * dp[i][j];
                            }
                            else {
                                dp2[i + dn][j] += comb * dp[i][j];
                            } 
                        }
                    }
                }
                comb *= (b - t);
                comb /= (t + 1);
                
            }
            swap(dp, dp2);
        }
        double total = 0;
        for (int j = 0; j <= 2 * k; j++) {
            // cout << j - k << " "<< dp[2 * n][j] << endl;
            total += dp[2 * n][j];
        }
        return dp[2 * n][k] / total;
    }
    
// private:

//         if (dp[i][d] > -0.5) return dp[i][d];
        
//     }
};
