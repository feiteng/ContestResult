class Solution {
    int n,m;
    double ans;
    int A[10],B[10],Pow[10];
    void dfs(int x)
    {
        if (x==m)
        {
            int sum=0;

            int a=0,b=0;
            for (int i=0;i<m;++i)
            {
                sum+=B[i];
                if (B[i]) ++a;
                if (A[i]-B[i]) ++b;
            }
            if (a!=b) return ;
            if (sum!=n) return ;
            //for (int i=0;i<m;++i) printf("%d %d\n",B[i],A[i]-B[i]);
            double v=1;
            for (int i=1;i<=n;++i) v=v*i;
            for (int i=0;i<m;++i) v/=Pow[B[i]],v/=Pow[A[i]-B[i]];
            ans+=v;
            //printf("v=%.5lf\n",v);
            return ;
        }
        for (int i=0;i<=A[x];++i) B[x]=i,dfs(x+1);
    }
public:
    double getProbability(vector<int>& nums) {
        Pow[0]=1;
        for (int i=1;i<10;++i) Pow[i]=Pow[i-1]*i;
        m=nums.size();
        n=0;
        for (int i=0;i<m;++i) A[i]=nums[i],n+=A[i];

        n/=2;
        dfs(0);
        //printf("%.8lf\n",ans);
        for (int i=n*2;i>n;--i) ans/=i;
        for (int i=0;i<m;++i) ans*=Pow[A[i]];
        return ans;
    }
};
