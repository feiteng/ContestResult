class Solution {
    int n;
    double factorial(double n) {
        // cout << n << endl;
        if (n <= 1) {
            return 1.0;
        }
        return n * factorial(n - 1);
    }
    double fact(double n, double r) {
        if (n < r) {
            return 0;
        }
        if (n == r or r == 0) {
            return 1.0;
        }
        return fact(n - 1, r - 1) + fact(n - 1, r);
    }
    double func(int i, int cd1, int cd2, int n1, int n2, vector<int>& balls) {
        // base Case 
        
        // cout << i << " " << cd1 << " " << cd2 << " " << n1 << " " << n2 << endl;
        if (i == balls.size()) {
            
            if (n1 == 0 and n2 == 0 and cd1 == cd2) {
                return 1.0;
            }
            return 0.0;
        }
        double sol = 0.0;
        for(int curr = 0;curr <= balls[i];curr++) {
            if (n1 - curr >= 0 and n2 - (balls[i] - curr) >= 0) {
                // den = den + 1.0;
                double temp;

                if (curr != 0 and curr != balls[i]) {
                    temp = func(i + 1, cd1 + 1, cd2 + 1, n1 - curr, n2 - (balls[i] - curr), balls);
                } else if (curr == 0) {
                    temp = func(i + 1, cd1, cd2 + 1, n1 - curr, n2 - (balls[i] - curr), balls);
                } else {
                    temp = func(i + 1, cd1 + 1, cd2, n1 - curr, n2 - (balls[i] - curr), balls);
                }
                sol += fact(n1, curr) * fact(n2, n2 - (balls[i] - curr)) * temp;
            }
        }
        return sol;
    }
public:
    double getProbability(vector<int>& balls) {
        n = accumulate(balls.begin(), balls.end(), 0);
        double num = func(0, 0, 0, n / 2, n / 2, balls);

        double den = factorial(n);
        for(int i = 0;i < balls.size();i++ ){
            den /= factorial(balls[i]);
        }
        return num / den;
        
    }
};
