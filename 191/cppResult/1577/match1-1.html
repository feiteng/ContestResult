class Solution {
    int m, n;
    double ans, tot, basic, fac[50];
public:
    double getProbability(vector<int>& a) {
        m = a.size();
        vector<int> b(m);
        
        n = 0;

        ans = 0;
        fac[1] = 1;
        
        for (int i = 0; i < m; ++i) {
            n += a[i];
        }
        for (int i = 2; i <= n; ++i) {
            fac[i] = fac[i-1] * i;
        }
        

        tot = fac[n];
        for (int i = 0; i < m; ++i) {
            tot /= fac[a[i]];
        }
        // cout << tot << endl;
        
        dfs(0, n/2, 0, a, b);
        return ans / tot;
        // return ans;
    }
    
    void dfs(int i, int rem, int x, vector<int> &a, vector<int> &b) {
        // printf("i:%d rem:%d %d %d\n", i, rem, x, y);
        // for (int j = 0; j < m; ++j) {
        //         printf("%d ", b[j]);
        // }
        // printf("\n");
        if (rem == 0) {
            int y = 0;
            double cur = fac[n/2] * fac[n/2];
            for (int j = 0; j < m; ++j) {
                if (b[j] > 0) {
                    cur /= fac[b[j]];
                }
                
                if (a[j] > b[j]) {
                    ++y;
                    cur /= fac[a[j]-b[j]];
                }
                    
                // printf("%d ", b[j]);
            }
            // printf("\n%d %lf\n", x, cur);
            if (x == y)
                ans += cur;
            

            return;
        }
        if (i == m) {
            return;
        }
        
        for (int t = 0; t <= a[i] && t <= rem; ++t) {
            b[i] = t;
            dfs(i+1, rem - t, x + (t > 0), a, b);
            b[i] = 0;
        }
    }
};
