class Solution {
	double prop(vector<int>& p) {
		double d = 1;
		for (int i = 2; i <= sum; i++)
			d *= i * i;
		for (int i = 0; i < p.size(); i++) {
			for (int j = 2; j <= p[i]; j++)
				d /= j;
			for (int j = 2; j <= B[i] - p[i]; j++)
				d /= j;
		}
		return d;
	}
	double prop2(vector<int>& p) {
		double d = 1;
		for (int i = 2; i <= sum * 2; i++)
			d *= i;
		for (int i = 0; i < p.size(); i++) {
			for (int j = 2; j <= p[i]; j++)
				d /= j;
		}
		return d;
	}

	void dfs(int cur, int t, vector<int>& p) {
		if (cur == p.size() && t)
			return;
		if (!t) {
			int x = 0, y = 0;
			for (int i = 0; i < p.size(); i++) {
				if (p[i]) x++;
				if (B[i] - p[i]) y++;
			}
			if (x == y) {
				ans += prop(p);
			}
			return;
		}
		if (p[cur] < B[cur]) {
			p[cur]++;
			dfs(cur, t - 1, p);
			p[cur]--;
		}
		dfs(cur + 1, t, p);
	}
	int sum;
	vector<int> B;
	double ans;
public:
	double getProbability(vector<int>& b) {
		B = b;
		sum = accumulate(B.begin(), B.end(), 0) / 2;
		vector<int> p(B.size());
		ans = 0;
		dfs(0, sum, p);
		return ans / prop2(b);
	}
};
