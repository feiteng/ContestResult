double dp[10][10][10][50];
// after i-th ball j type in first k type in second l balls in the first half

//double C[50][50];

int C(int n,int m)
{
    int ans=1;
    for(int i=n,j=1;j<=m;i--,j++)
        ans=ans*i/j;
    return ans;
}

class Solution {
public:
    double EPS=1e-10;
    double getProbability(vector<int>& balls) {
        int n=balls.size();
        int sum=0;
        for(int i:balls) sum+=i;
        

        int m=sum/2;
        
        for(int i=0;i<=n;i++)
            for(int j=0;j<=n;j++)
                for(int k=0;k<=n;k++)
                    for(int l=0;l<=m;l++)
                        dp[i][j][k][l]=0.0;
        dp[0][0][0][0]=1.0;
        
        // for(int i=0;i<50;i++)
        // {
        //     C[i][0]=C[i][i]=1.0;
        //     for(int j=1;j<i;j++)
        //         C[i][j]=C[i-1][j-1]+C[i-1][j];
        // }
        
        sum=0;
        for(int i=0;i<n;i++)
        {

            sum+=balls[i];
            for(int j=0;j<=i;j++)
                for(int k=0;k<=i;k++)
                    for(int l=0;l<=m;l++)
                    {
                        if(dp[i][j][k][l]<EPS) continue;
                        //cout<<i<<" "<<j<<" "<<k<<" "<<l<<" "<<dp[i][j][k][l]
                        int cnt=0;
                        for(int t=0;t<=balls[i];t++) // put t into first half
                        {
                            if(t+l>m) break;
                            if(sum-l-t>m) continue;
                            cnt+=C(m-l,t)*C(m-(sum-balls[i]-l),balls[i]-t);
                        }
                        //cout<<i<<" "<<j<<" "<<k<<" "<<l<<" "<<cnt<<endl;
                        for(int t=0;t<=balls[i];t++) // put t into first half
                        {
                            if(t+l>m) break;
                            if(sum-l-t>m) continue;
                            dp[i+1][j+(t!=0)][k+(t!=balls[i])][l+t]+=dp[i][j][k][l]*C(m-l,t)*C(m-(sum-balls[i]-l),balls[i]-t)/cnt;
                        }
                        
                    }
        }
            
        
        double ans=0.0;
        for(int j=0;j<=n;j++)
            ans+=dp[n][j][j][m];
        // cout<<ans<<endl;
        // double down=1.0;
        // for(int i=1;i<=sum;i++)
        //     down*=i;
        // cout<<down<<endl;
        // for(int i=0;i<n;i++)
        //     for(int j=1;j<=balls[i];j++)
        //         down/=j;
        // cout<<down<<endl;
        return ans;
    }
};
