class Solution {
public:
    int maxArea(int h, int w, vector<int>& a, vector<int>& b) {
        a.push_back(0);
        a.push_back(h);
        sort(a.begin(), a.end());
        b.push_back(0);
        b.push_back(w);
        sort(b.begin(), b.end());
        long long x = 0, y = 0;
        for (int i = 0; i+1 < a.size(); i++) {
            x = max(x, (long long)a[i+1]-a[i]);
        }
        for (int i = 0; i+1 < b.size(); i++) {
            y = max(y, (long long)b[i+1]-b[i]);
        }
        int mod = 1e9 + 7;
        long long ret = x*y % mod;
        return ret;
        
        
    }
};
