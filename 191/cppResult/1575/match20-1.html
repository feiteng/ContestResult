#define LL long long
const LL mod =  1e9 + 7;



class Solution {
public:
    int maxArea(int h, int w, vector<int>& x, vector<int>& y) {
      int n = x.size(), m = y.size();
      sort(x.begin(), x.end());
      sort(y.begin(), y.end());
      
      int maxx = max(x[0], h - x[n - 1]), maxy = max(y[0], w - y[m - 1]);
      for (int i = 1; i < n; i++) {
          maxx = max(maxx, x[i] - x[i - 1]);
      }
      for (int i = 1; i < m; i++) {
          maxy = max(maxy, y[i] - y[i - 1]);
      }
      return (LL)maxx * maxy % mod;
    }
};
