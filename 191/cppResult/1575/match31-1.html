#define ALL(x) (x).begin(), (x).end()
const int MOD = 1e9+7;


class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        int dh = 0;
        int dv = 0;
        hc.push_back(0);
        hc.push_back(h);
        vc.push_back(0);
        vc.push_back(w);
        sort(ALL(hc));
        sort(ALL(vc));
        int M = hc.size();

        int N = vc.size();
        for (int i = 1; i < M; ++i) {
            dh = max(hc[i] - hc[i-1], dh);
        }
        for (int i = 1; i < N; ++i) {
            dv = max(vc[i] - vc[i-1], dv);
        }
        int ret = 1LL * dh * dv % MOD;
        return ret;
    }
};
