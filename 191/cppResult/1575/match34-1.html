class Solution {
 public:
  int maxArea(int h, int w, vector<int>& hs, vector<int>& vs) {
    int mod = 1e9 + 7;
    int max_v = 0, max_h = 0;
    sort(hs.begin(), hs.end());
    sort(vs.begin(), vs.end());
    for (int i = 0; i < hs.size() - 1; i++) {
      max_h = max(max_h, hs[i + 1] - hs[i]);
    }
    for (int i = 0; i < vs.size() - 1; i++) {
      max_v = max(max_v, vs[i + 1] - vs[i]);
    }
    max_v = max(max_v, vs[0] - 0);
    max_v = max(max_v, w - vs.back());
    max_h = max(max_h, hs[0] - 0);
    max_h = max(max_h, h - hs.back());
    cout << max_v << ", "<<max_h <<endl;
    return (long)max_v * max_h % mod;
  }
};
