class Solution {
public:
    int maxArea(int h, int w, vector<int>& hh, vector<int>& vv) {
        sort(hh.begin(), hh.end());
        sort(vv.begin(), vv.end());
        int p = max(hh[0], h - hh.back());
        for (int i = 0; i < hh.size() - 1; ++i) {
            p = max(p, hh[i + 1] - hh[i]);
        }
        int q = max(vv[0], w - vv.back());
        for (int i = 0; i < vv.size() - 1; ++i) {
            q = max(q, vv[i + 1] - vv[i]);
        }
        return (long long)p * q % 1000000007;
    }
};
