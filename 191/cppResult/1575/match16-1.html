class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
        hc.push_back(0);
        hc.push_back(h);
        vc.push_back(0);
        vc.push_back(w);
        
        sort(hc.begin(), hc.end());
        sort(vc.begin(), vc.end());
        int maxh = 0;
        for (int i = 0; i < hc.size() -1; i++) {
            maxh = max(maxh, hc[i+1]-hc[i]);
        }
        int maxv = 0;
        for (int i = 0; i < vc.size() -1; i++) {
            maxv = max(maxv, vc[i+1]-vc[i]);
        }
        int mod = 1e9+7;
        return ((long long)maxh*maxv) %mod;
    }
};
