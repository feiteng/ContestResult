class Solution {
public:
    int maxArea(int h, int w, vector<int>& hh, vector<int>& ww) {
        sort(hh.begin(),hh.end());
        sort(ww.begin(),ww.end());
        hh.push_back(h);
        ww.push_back(w);
        int n1=hh.size(), n2=ww.size();
        for(int i=n1-1;i>0;i--) hh[i]-=hh[i-1];
        for(int i=n2-1;i>0;i--) ww[i]-=ww[i-1];
        return *max_element(hh.begin(),hh.end())* (long long)*max_element(ww.begin(),ww.end()) % (long long)(1e9+7);
    }
};
