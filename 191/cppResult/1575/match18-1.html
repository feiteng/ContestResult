class Solution {
public:
    int maxArea(int h, int w, vector<int>& hh, vector<int>& ww) {
        hh.push_back(0);
        hh.push_back(h);
        ww.push_back(0);
        ww.push_back(w);
        sort(hh.begin(), hh.end());
        sort(ww.begin(), ww.end());
        int ma = 0, mb = 0;
        for (int i = 1; i < hh.size(); i++)
            ma = max(ma, hh[i] - hh[i - 1]);
        for (int i = 1; i < ww.size(); i++)
            mb = max(mb, ww[i] - ww[i - 1]);
        return (long long)ma * mb % 1000000007;
    }
};
