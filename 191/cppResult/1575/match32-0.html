typedef long long ll;
ll mod = 1e9 + 7;
class Solution {
public:
    int maxArea(int h, int w, vector<int>& H, vector<int>& V) {
        sort(H.begin(), H.end());
        sort(V.begin(), V.end());
        H.insert(H.begin(), 0);
        H.push_back(h);
        V.insert(V.begin(), 0);
        V.push_back(w);
        
        int n = H.size(); int m = V.size();
        ll res = 0;
        ll maxh = 0; ll maxw = 0;
        for (int i=0; i+1<n; i++) {
            maxh = max(maxh, 1LL*(H[i+1]-H[i]));
        }
        for (int i=0; i+1<m; i++) {
            maxw = max(maxw, 1LL*(V[i+1]-V[i]));
        }
        res = maxh*maxw;
        cout<<res<<endl;
        return res%mod;
    }
};
