class Solution {
public:
    int maxArea(int h, int w, vector<int>& hc, vector<int>& vc) {
      sort(hc.begin(), hc.end());
      sort(vc.begin(), vc.end());
      
      int r = max(hc[0], h-hc[hc.size()-1]);
      int c = max(vc[0], w-vc[vc.size()-1]);
      
      for (int i = 0; i < hc.size()-1; i++) {
        r = max(r, hc[i+1] -hc[i]);
      }
           for (int i = 0; i < vc.size()-1; i++) {
        c = max(c, vc[i+1] -vc[i]);
      }
      
      return (int) (((long long)r*c) % (1000000007));
      
      
        
    }
};
