class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        vector<int>& hv = horizontalCuts;
        vector<int>& vv = verticalCuts;
        hv.push_back(0);
        hv.push_back(h);
        vv.push_back(0);
        vv.push_back(w);
        sort(hv.begin(), hv.end());
        sort(vv.begin(), vv.end());
        int hval = 0, vval = 0;
        for (int i = 1; i < hv.size(); ++i) {
            hval = max(hval, hv[i] - hv[i - 1]);
        }
        for (int i = 1; i < vv.size(); ++i) {
            vval = max(vval, vv[i] - vv[i - 1]);
        }
        return ((long long)hval * vval) % 1000000007;
    }
};
