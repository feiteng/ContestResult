class Solution {
public:
    int maxArea(int h, int w, vector<int>& H, vector<int>& V) {
        H.push_back(0);
        H.push_back(h);
        V.push_back(0);
        V.push_back(w);
        sort(H.begin(), H.end());
        sort(V.begin(), V.end());
        
        int mxh = 0;
        int mxv = 0;
        int n = H.size();
        int m = V.size();
        for (int i = 1; i < n; ++i) {
            mxh = max(mxh, H[i] - H[i - 1]);
        }
        for (int i = 1; i < m; ++i) {
            mxv = max(mxv, V[i] - V[i - 1]);
        }
        
        long long MOD = 1e9 + 7;
        
        long long res = mxh * 1LL * mxv;
        res %= MOD;
        return res;
    }
};
