class Solution {
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        sort(horizontalCuts.begin(), horizontalCuts.end());
        sort(verticalCuts.begin(), verticalCuts.end());
        long long max_h = max(verticalCuts[0], w - verticalCuts.back());

        long long max_w = max(horizontalCuts[0], h - horizontalCuts.back());
        for (int i = 1; i < verticalCuts.size(); ++i) {
            max_h = max(max_h, (long long)(verticalCuts[i] - verticalCuts[i - 1]));
        }
        for (int i = 1; i < horizontalCuts.size(); ++i) {
            max_w = max(max_w, (long long)(horizontalCuts[i] - horizontalCuts[i - 1]));
        }
        return int(max_h * max_w % (long long)(1E9 + 7));
    }
};
