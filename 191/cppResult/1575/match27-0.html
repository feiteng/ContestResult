class Solution {
public:
    int maxArea(int h, int w, vector<int>& ho, vector<int>& ve) {
        sort(ho.begin(), ho.end());
        sort(ve.begin(), ve.end());
        int hh = max(ho[0], h - ho[ho.size() - 1]), v = max(ve[0], w - ve[ve.size() - 1]);
        
        for (int i = 1; i < ho.size(); ++i) {
            hh = max(hh, ho[i] - ho[i-1]);
        }
        
        for (int i = 1; i < ve.size(); ++i) {
            v = max(v, ve[i] - ve[i-1]);
        }
        
        int mod = 1e9 + 7;
        return 1l * hh * v % mod;
    }
};
