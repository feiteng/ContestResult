class Solution {
	#define		MOD		1000000007
	
public:
    int maxArea(int h, int w, vector<int>& horizontalCuts, vector<int>& verticalCuts) {
        int i;
		
		horizontalCuts.push_back(h);
		verticalCuts.push_back(w);
		sort(horizontalCuts.begin(), horizontalCuts.end());
		sort(verticalCuts.begin(), verticalCuts.end());
		
		int M = horizontalCuts.size();
		int N = verticalCuts.size();
		int MaxX = horizontalCuts[0];
		int MaxY = verticalCuts[0];
		
		for(i = 1; i < M; i++)
		{
			MaxX = max(MaxX, (horizontalCuts[i] - horizontalCuts[(i - 1)]));
			//cout << horizontalCuts[i] << ":" << horizontalCuts[(i - 1)] << endl;
		}

		for(i = 1; i < N; i++)
		{
			MaxY = max(MaxY, (verticalCuts[i] - verticalCuts[(i - 1)]));
			//cout << verticalCuts[i] << ";" << verticalCuts[(i - 1)] << endl;
		}
		
		//cout << MaxX << " " << MaxY << endl;
		
		return (((long long)MaxX * MaxY) % MOD);
    }
};
