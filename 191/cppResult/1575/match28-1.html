class Solution {
public:
    int maxArea(int _h, int _w, vector<int>& h, vector<int>& v) {
        h.push_back(0); h.push_back(_h);
        v.push_back(0); v.push_back(_w);
        sort(h.begin(), h.end());
        sort(v.begin(), v.end());
        int maxH = 0, maxV = 0;
        for (int i = 1; i < h.size(); i++)
            maxH = max(maxH, h[i] - h[i-1]);
        for (int i = 1; i < v.size(); i++)
            maxV = max(maxV, v[i] - v[i-1]);
        
        long long res = maxH;
        res *= maxV;
        res %= 1000000000 + 7;
        return res;
    }
};
