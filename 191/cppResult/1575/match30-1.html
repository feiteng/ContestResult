class Solution {
public:
    int maxArea(int h, int w, vector<int>& H, vector<int>& W) {
        H.push_back(0); H.push_back(h); sort(H.begin(), H.end());
        W.push_back(0); W.push_back(w); sort(W.begin(), W.end());
        int hh = 0, ww = 0;
        for(int i=1; i<H.size(); i++)
            hh = max(hh, H[i] - H[i-1]);
        for(int i=1; i<W.size(); i++)
            ww = max(ww, W[i] - W[i-1]);
        int res = (long long)hh*ww%1000000007;
        return res;
    }
};
