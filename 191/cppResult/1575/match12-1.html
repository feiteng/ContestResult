typedef long long int64;
const int MOD = 1e9 + 7;

class Solution {
public:
    int maxArea(int h, int w, vector<int>& a, vector<int>& b) {
        a.push_back(0);
        a.push_back(h);
        b.push_back(0);
        b.push_back(w);
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        int n = a.size(), m = b.size();

        int64 H = 0, W = 0;
        for (int i = 1; i < n; ++i) H = max(H, (int64)(a[i] - a[i - 1]));
        for (int i = 1; i < m; ++i) W = max(W, (int64)(b[i] - b[i - 1]));
        return H * W % MOD;
    }
};
