class Solution {
public:
	vector<int>p;
	vector<bool>vis;
	void dfs(int x,vector<vector<int>>&adj) {
		vis[x] = true;

		for (auto it : adj[x]) {
			if (!vis[it]) {
				p[it] = x;
				dfs(it,adj);
			}
		}
	}
	int minReorder(int n, vector<vector<int>>& connections) {
		vector<vector<int>>adj(n);

		for (int i = 0; i < connections.size(); i++) {
			int x = connections[i][0], y = connections[i][1];
			adj[x].push_back(y);
			adj[y].push_back(x);
		}
		p.assign(n, -1);
		vis.assign(n, false);
		dfs(0, adj);
		int ct = 0;
		for (auto it : connections) {
			int x = it[0], y = it[1];
			if (p[x] == y)continue;
			ct++;
		}
		return ct;
	}
};
