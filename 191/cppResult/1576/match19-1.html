class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<int>> vi(n);
        vector<vector<int>> vo(n);
        for(auto p: connections){
            vo[p[0]].push_back(p[1]);
            vi[p[1]].push_back(p[0]);
        }
        vector<bool> flag(n,true);
        flag[0]=false;
        queue<int> q;
        q.push(0);
        int ans=0;
        while(!q.empty()){
            int len=q.size();

            for(int pp=0;pp<len;pp++){
                int head=q.front();
                q.pop();
                for(auto i: vo[head]){
                    if(flag[i]){
                        ans++;
                        flag[i]=false;
                        q.push(i);
                    }
                }
                for(auto i: vi[head]){
                    if(flag[i]){
                        flag[i]=false;
                        q.push(i);
                    }
                }
            }
        }
        return ans;
    }
};
