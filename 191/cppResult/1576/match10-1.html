class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<int>> G(n);
        for(auto v : connections){
            G[v[0]].push_back(v[1]);
            G[v[1]].push_back(v[0]);
        }
        vector<int> d(n, -1);
        d[0] = 0;
        queue<int> q;
        q.push(0);

        while(not q.empty()){
            int u = q.front();
            q.pop();
            for(int v : G[u]) if(d[v] == -1){
                d[v] = d[u] + 1;
                q.push(v);
            }
        }
        int ans = 0;
        for(auto v : connections) ans += d[v[0]] < d[v[1]];
        return ans;
    }
};
