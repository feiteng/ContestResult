class Solution {
public:
    
    int n;
    vector<vector<int>> inedges, outedges;
    vector<int> visited;
    int ans;
    
    void dfs(int v) {
        visited[v] = 1;

        for(int next: outedges[v]) {
            if(!visited[next]) {
                ans++;
                dfs(next);
            }
        }
        for(int next: inedges[v]) {
            if(!visited[next]) {
                dfs(next);
            }
        }
    }
    
    int minReorder(int _n, vector<vector<int>>& connections) {
        n = _n;
        inedges.resize(n);
        outedges.resize(n);
        visited.resize(n);
        for(auto p: connections) {
            inedges[p[1]].push_back(p[0]);
            outedges[p[0]].push_back(p[1]);
        }
        ans = 0;
        dfs(0);
        return ans;
    }
};
