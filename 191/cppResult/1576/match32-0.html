ll inverse (ll a, ll m) {
    a = MOD (a, m);
    if (a <= 1)return a;
    return MOD ((1 - inverse (m, a) * m) / a, m);
}

ll gcd (ll a, ll b) {
    a = abs (a), b = abs (b);
    while (b != 0) {
        a %= b;
        swap (a, b);
    }
    return a;
}

ll fast (ll a, ll b, ll mod) {
    if (b < 0)a = inverse (a, mod), b = -b;
    ll ans = 1;
    while (b) {
        if (b & 1)ans = ans * a % mod;
        a = a * a % mod;
        b /= 2;
    }
    return ans % mod;
}


VI adj[100010];
int dep[100100];

class Solution {
public:
    void dfs (int cur, int fa) {
        for (auto i:adj[cur]) {
            if (i != fa) {
                dep[i] = dep[cur] + 1;
                dfs (i, cur);
            }
        }
    }
    
    int minReorder (int n, vector<vector<int>> &connections) {
        REP(i, 0, n)adj[i].clear ();
        dep[0] = 0;
        for (auto v:connections) {
            int a = v[0], b = v[1];
            adj[a].PB (b);
            adj[b].PB (a);
        }
        dfs (0, -1);
        int ans = 0;
        for (auto v:connections) {
            int a = v[0], b = v[1];
            ans += dep[a] < dep[b];
        }
        return ans;
    }
};
