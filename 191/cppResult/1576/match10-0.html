class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<int>> adj(n);
        for(auto &e:connections){
            adj[e[0]].push_back(e[1]);
            adj[e[1]].push_back(e[0]);
        }
        queue<int> q;
        vector<int> d(n,-1);
        q.push(0);
        d[0]=0;
        while(!q.empty()){
            int u=q.front();
            q.pop();
            for(int v:adj[u])
                if(d[v]==-1){
                    d[v]=d[u]+1;
                    q.push(v);
                }
        }
        int ans=0;
        for(auto &e:connections)
            if(d[e[0]]<d[e[1]])ans++;
        return ans;
    }
};
