class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        unordered_map<int, unordered_set<int>> from;
        unordered_map<int, unordered_set<int>> to;
        for(auto c:connections){
            from[c[0]].insert(c[1]);
            to[c[1]].insert(c[0]);
        }
        int res = 0;
        unordered_set<int> visited;
        queue<int> q;
        q.push(0);
        while(q.size() > 0){
            auto cur = q.front();
            q.pop();
            visited.insert(cur);
            for(auto next:to[cur]){
                if(visited.count(next) < 1)
                    q.push(next);
            }
            for(auto next:from[cur]){
                if(visited.count(next) < 1){
                    res++;
                    q.push(next);
                }
            }
        }
        return res;
        
    }
};
