class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<int>> s(n,vector<int>(0));
        vector<vector<int>> d(n,vector<int>(0));
        for(int i=0;i<connections.size();i++)
        {
            int x=connections[i][0];
            int y=connections[i][1];
            s[x].push_back(y);
            s[y].push_back(x);
            d[x].push_back(y);
        }
        int vis[n];
        memset(vis,0,sizeof(vis));
        queue<int> q;
        q.push(0);
        int res=0;
        while(!q.empty()){
            int now=q.front();
            vis[now]=1;
            q.pop();
            for(int i=0;i<s[now].size();i++){
                int nex=s[now][i];
                if(vis[nex]==0){
                    for(int j=0;j<d[now].size();j++)
                        if(d[now][j]==nex) res+=1;
                    q.push(nex);
                    vis[nex]=1;
                }
            }
        }
      return res;
    }
};
