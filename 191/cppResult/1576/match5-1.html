class Solution {
public:
    int minReorder(int n, vector<vector<int>>& c) {
        vector<vector<pair<int,int>>> g(n);
        for(auto e : c){
            g[e[0]].push_back({e[1], 1});
            g[e[1]].push_back({e[0], 0});

        }
        vector<int> vis(n);
        queue<int> q;
        q.push(0);
        vis[0] = 1;
        int ans = 0;
        while(!q.empty()){
            int u = q.front();
            q.pop();
            for(auto e : g[u]){
                if(!vis[e.first]){
                    vis[e.first] = 1;
                    ans += e.second;
                    q.push(e.first);
                }
            }
        }
        return ans;
    }
};
