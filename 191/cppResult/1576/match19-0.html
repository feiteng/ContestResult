class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<pair<int, int>> adj[n];
        for(auto i : connections) {
            adj[i[0]].emplace_back(i[1], 1);
            adj[i[1]].emplace_back(i[0], 0);
        }
        vector<int> visited(n, false);
        int cost = 0;
        stack<int> st;
        st.push(0);
        visited[0] = true;
        while (!st.empty()) {
            int curr = st.top();
            st.pop();
            for(auto i : adj[curr]) {
                if (!visited[i.first]) {
                    visited[i.first] = true;
                    st.push(i.first);
                    cost += i.second;
                }
            }
        }
        return cost;
    }
};
