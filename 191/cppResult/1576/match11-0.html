class Solution {
    vector<pair<int,int>> G[50005];
    int ans;
    
    void dfs(int v, int p) {
        for(auto& x : G[v]) {
            if(x.first == p) continue;
            dfs(x.first, v);
            ans += !x.second;
        }
    }
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        for(auto& e : connections) {
            G[e[0]].emplace_back(e[1], 0);
            G[e[1]].emplace_back(e[0], 1);
        }
        
        ans = 0;
        dfs(0, 0);
        return ans;
    }
};
