class Solution {
public:
    int res;
    vector<vector<int>> zg1, zg2;
    void dfs(int r, int par) {
        for (int nxt: zg1[r]) {
            if (nxt==par) continue;
            dfs(nxt,r);            
            ++res;
        }
        for (int nxt: zg2[r]) {
            if (nxt==par) continue;
            dfs(nxt,r);
        }
    }
    int minReorder(int n, vector<vector<int>>& connections) {
        zg1.resize(n);
        zg2 = zg1;
        for (auto & z: connections) {
            zg1[z[0]].push_back(z[1]);
            zg2[z[1]].push_back(z[0]);
        }
        res = 0;
        dfs(0, -1);
        return res;
    }
};
