class Solution {
public:
    
    int dfs(int u, vector<bool> &vis, vector<pair<int, int> > adj[]) {
        int ret = 0;
        vis[u] = true;
        for(pair<int, int> p : adj[u]) {
            int nxt = p.first;
            int directed = p.second;
            if(!vis[nxt]) {
                ret += directed;

                ret += dfs(nxt, vis, adj);
            }
        }
        return ret;
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<pair<int, int> > adj[n];
        for(vector<int> v : connections) {
            adj[v[0]].push_back({v[1], 1});
            adj[v[1]].push_back({v[0], 0});
        }
        vector<bool> vis(n, false);
        return dfs(0, vis, adj);
    }
};
