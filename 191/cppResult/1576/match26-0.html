class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
		int i;
		int N = connections.size();
		vector<vector<int> > Adj(n, vector<int>());
		vector<vector<int> > Adj1(n, vector<int>());
		vector<int> V(n);
		
		for(i = 0; i < N; i++)
		{
			Adj[connections[i][0]].push_back(connections[i][1]);
			Adj1[connections[i][1]].push_back(connections[i][0]);
		}
		
        queue<int> Q;
		Q.push(0);
		V[0] = 1;
		
		int Result = 0;
		
		while(Q.size())
		{
			int T = Q.front();
			Q.pop();
			
			for(auto it : Adj[T])
			{
				if(V[it] == 0)
				{
					V[it] = 1;
					Result++;
					Q.push(it);
				}
			}

			for(auto it : Adj1[T])
			{
				if(V[it] == 0)
				{
					V[it] = 1;
					Q.push(it);
				}
			}
		}
		
		return Result;
    }
};
