class Solution {
public:
    int dfs(int start, vector<vector<int>>& adj, vector<vector<int>>& adjo, vector<bool>& vis) {
        int ans = 0;
        vis[start] = true;
        for (int ne : adjo[start]) {
            if (!vis[ne]) {
                ans += dfs(ne, adj, adjo, vis) + 1;
            }
        }
        for (int ne : adj[start]) {
            if (!vis[ne]) {
                ans += dfs(ne, adj, adjo, vis);
            }
        }
        return ans;
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<bool> vis(n);
        vector<vector<int>> adj(n);
        vector<vector<int>> adjo(n);
        for (vector<int> con : connections) {
            adjo[con[0]].push_back(con[1]);
            adj[con[1]].push_back(con[0]);
        }
        return dfs(0, adj, adjo, vis);
    }
};
