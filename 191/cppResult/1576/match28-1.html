class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector < vector <pair <int, int> > > conn(n, vector<pair <int, int> > ());
        for (auto c : connections) {
            conn[c[0]].push_back(make_pair(c[0], c[1]));
            conn[c[1]].push_back(make_pair(c[0], c[1]));
        }
        
        stack <int> st; 

        vector <int> vis(n, 0);
        st.push(0);
        int out = 0;
        while (st.size() > 0) { 
            int idx = st.top();
            st.pop();
            vis[idx] = 1;
            for (int i = 0; i < conn[idx].size(); i++) {
                if (vis[conn[idx][i].first] && vis[conn[idx][i].second]) continue;
                int next = conn[idx][i].first;
                if (next == idx) {
                    next = conn[idx][i].second;
                    out++;   
                }
                st.push(next);
            }
        }
        return out;
    }
};
