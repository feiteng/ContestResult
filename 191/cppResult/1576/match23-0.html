class Solution {
public:
    int DFS(vector<vector<pair<bool, int>>> &g, int root, vector<bool> &visited) {
        int ret = 0;
        if (visited[root]) return 0;
        visited[root] = true;
        for (auto &c: g[root]) {
           // printf("%d %d %d\n", root, c.first, c.second);
            if (visited[c.second]) continue;
            if (c.first)
                ++ret;
            ret += DFS(g, c.second, visited);
        }
        return ret;
    }
    
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<pair<bool, int>>> g(n);
        for (auto &c: connections) {        
            g[c[0]].push_back({true, c[1]});
            g[c[1]].push_back({false, c[0]});
        }

        vector<bool> visited(n, false);
        return DFS(g, 0, visited);

    }
};
