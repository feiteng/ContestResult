class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<pair<int, int>>> g(n);
        for(auto e : connections) {
            g[e[0]].emplace_back(e[1], 1);
            g[e[1]].emplace_back(e[0], 0);
        }
        queue<int> q;
        q.push(0);
        int ans = 0;
        vector<int> vis(n);
        vis[0] = 1;
        while(!q.empty()) {
            int u = q.front(); q.pop();
            for(auto p : g[u]) {
                if(vis[p.first]) continue;
                ans += p.second;
                q.push(p.first);
                vis[p.first] = 1;
            }
        }
        return ans;
    }
};
