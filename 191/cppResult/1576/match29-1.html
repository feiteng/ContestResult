class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        unordered_map<int,vector<pair<int,int>>> m;
        for (auto& c:connections) {
            m[c[0]].push_back({c[0],c[1]});
            m[c[1]].push_back({c[0],c[1]});
        }
        int res=0;
        queue<int> q;
        q.push(0);
        unordered_set<int> d;//done
        while (!q.empty()) {
            int t=q.front();
            q.pop();
            d.insert(t);
            for (auto p:m[t]) {
                if (p.first==t){
                    if (d.count(p.second)) continue;
                    ++res;
                    q.push(p.second);
                }else {
                    if (d.count(p.first)) continue;
                    q.push(p.first);
                }
            }
        }
        return res;
    }
};
