class Solution {
public:
    vector<vector<int>> to;

    vector<vector<int>> rto;
    int ans;
    void dfs(int cur, int pre)
    {
        for (int t : to[cur]) {
            if (t == pre) continue;
            ans++;
            dfs(t, cur);
        }
        for (int t : rto[cur]) {
            if (t == pre) continue;
            dfs(t, cur);
        }
    }
    int minReorder(int n, vector<vector<int>>& connections)
    {
        to = vector<vector<int>>(n);
        rto = vector<vector<int>>(n);
        for (auto& c : connections) {
            int a = c[0];
            int b = c[1];
            to[a].push_back(b);
            rto[b].push_back(a);
        }
        ans = 0;
        dfs(0, -1);
        return ans;
    }
};
