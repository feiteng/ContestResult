class Solution {
public:
    
    vector<vector<pair<int,int>>> g; 
    
    int minReorder(int n, vector<vector<int>>& cons) {
        g = vector<vector<pair<int,int>>>(n);
        for (auto e : cons)
        {
            int a = e[0], b = e[1];
            g[a].push_back({b, 0});
            g[b].push_back({a, 1});
        }
        
        return dfs(0, -1);
    }
    
    int dfs(int u, int father)
    {
        int res = 0;
        for (auto e : g[u])
        {
            if (e.first == father) continue;
            res += 1 - e.second + dfs(e.first, u);
        }
        return res;
    }
};
