class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<pair<int,int>>> graph(n);
        
        for (auto& e : connections) {
            int u = e[0], v = e[1];
            graph[u].emplace_back(v, 1);
            graph[v].emplace_back(u, 0);
        }
        
        vector<int> que;
        que.reserve(n);
        
        vector<bool> visited(n, false);
        visited[0] = true;
        que.push_back(0);
        
        int res = 0;
        for (int i = 0; i < que.size(); ++i) {
            int u = que[i];
            
            for (auto& e : graph[u]) {
                int v = e.first;
                int w = e.second;
                if (!visited[v]) {
                    visited[v] = true;
                    res += w;
                    que.push_back(v);
                }
            }
        }
        
        return res;
    }
};
