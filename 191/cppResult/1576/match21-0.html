class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<int> v[50005],v2[50005];
        for(auto it:connections){
            v[it[1]].push_back(it[0]);
            v2[it[0]].push_back(it[1]);
        }
        queue<int> q;
        int vis[50005];
        fill(vis,vis+n+1,0);
        int ans=0;
        vis[0]=1;
        q.push(0);
        while(!q.empty()){
            int x=q.front();
            q.pop();
            for(auto it:v[x]){
                if(!vis[it]){
                    q.push(it);
                    vis[it]=1;
                }
            }
            for(auto it:v2[x]){
                if(!vis[it]){
                    q.push(it);
                    vis[it]=1;
                    ans++;
                }
            }
        }
        return ans;
    }
};
