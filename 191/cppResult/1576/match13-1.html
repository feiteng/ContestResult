class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        vector<vector<vector<int>>> edges(n);

        for (auto& c : connections) {
            edges[c[0]].push_back({c[1], 1});
            edges[c[1]].push_back({c[0], 0});
        }
        
        queue<int> q;
        q.push(0);
        vector<bool> vis(n, false);
        vis[0] = true;
        
        int ans = 0;
        while (q.empty() == false) {

            for (auto& e : edges[q.front()]) {
                if (vis[e[0]] == false) {
                    ans += e[1];
                    
                    q.push(e[0]);
                    vis[e[0]] = true;
                }
            }
            
            q.pop();
        }
        
        return ans;
    }
};
