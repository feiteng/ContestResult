class Solution {
public:
    int minReorder(int n, vector<vector<int>>& connections) {
        unordered_map<int, vector<int>> mp;
        unordered_map<int, unordered_set<int>> st;
        for (auto & con : connections) {
            mp[con[0]].push_back(con[1]);
            mp[con[1]].push_back(con[0]);
            st[con[0]].insert(con[1]);
        }
        int res = 0;
        unordered_set<int> visited;
        visited.insert(0);
        queue<int> que;
        que.push(0);
        while (!que.empty()) {
            auto cur = que.front();
            que.pop();
            for (auto next : mp[cur]) {
                if (!visited.count(next)) {
                    visited.insert(next);
                    que.push(next);
                    if (st[cur].count(next)) {
                        ++res;
                    }
                }
            }
        }
        return res;
    }
};
