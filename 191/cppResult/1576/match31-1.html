typedef pair<int, int> ii;
const int N = 5e4 + 10;
vector<ii> a[N];

class Solution {
public:
    int ret;

    void DFS(int u, int parent) {
        for (auto& [v, k] : a[u]) {
            if (v == parent) continue;
            ret += k;
            DFS(v, u);
        }
    }
    int minReorder(int n, vector<vector<int>>& e) {

        for (int i = 0; i < n; ++i) a[i].clear();
        for (auto& v : e) {
            int x = v[0], y = v[1];
            a[x].push_back({y, 1});
            a[y].push_back({x, 0});
        }
        ret = 0;
        DFS(0, -1);
        return ret;
    }
};
