const int maxn = 5e4 + 100;

class Solution {
public:
    vector<int> g[maxn];
    set<pair<int,int>> edges;
    int res;
    void dfs(int at, int p) {
        if (at != 0) {
            //check edge
            if (!edges.count(make_pair(at, p))) {
                res++;
            }
        }
        for (int to: g[at]) {
            if (to != p) {
                dfs(to ,at);
            }
        }
    }
    int minReorder(int n, vector<vector<int>>& v) {
        res=0;
        for (auto ed: v) {
            g[ed[0]].push_back(ed[1]);
            g[ed[1]].push_back(ed[0]);
            edges.insert({ed[0], ed[1]});
        }
        
        dfs(0, -1);
        
        return res;
    }
};
