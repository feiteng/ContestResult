class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int max_product = -1E6;
        for (int i = 0; i < nums.size(); ++i) {
            for (int j = i + 1; j < nums.size(); ++j) {
                max_product = max(max_product, (nums[i] - 1) * (nums[j] - 1));
            }
        }
        return max_product;
    }
};
