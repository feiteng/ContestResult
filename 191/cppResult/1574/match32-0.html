class Solution {
public:
    int maxProduct(vector<int>& nums) {
        	int ret = -1e9;
					for(int i = 0; i < sz(nums); i++) for(int j = 0; j < i; j++) ret = max(ret, (nums[i]-1) * (nums[j]-1));
					return ret;
    }
};
