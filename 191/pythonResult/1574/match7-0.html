class Solution(object):
    def maxProduct(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        ans = float('-inf')
        
        for i in xrange(n):
            for j in xrange(i + 1, n):
                ans = max(ans, (nums[i]-1)*(nums[j]-1))
        
        return ans
