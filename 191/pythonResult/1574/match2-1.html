class Solution(object):
    def maxProduct(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        max_val = (nums[0]-1)*(nums[1]-1)
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                cur = (nums[i]-1)*(nums[j]-1)
                if cur > max_val:
                    max_val = cur
        return max_val
