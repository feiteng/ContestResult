class Solution(object):
    def getProbability(self, balls):
        """
        :type balls: List[int]
        :rtype: float
        """
        def arrange(n, m):
            ans = 1
            if n == 0:
                return 1
            for i in range(m, n, -1):
                ans *= i
            for i in range(2, m - n + 1, 1):
                ans //= i
            #print(n, m, ans)
            return ans
            
        def decode(num, m, k, n, balls):
            arr = []
            for i in range(k - 1):
                arr.append(num % m)
                if arr[-1] > balls[k - 1 - i]:
                    return None
                num = (num // m)
            head = (n // 2) - sum(arr)
            if head > balls[0]:
                return None
            arr.append(head)
            arr.reverse()
           
            return arr
        
        def is_equal(arr, balls):
            box2 = [balls[i] - arr[i] for i in range(len(balls))]
            num1, num2 = 0, 0
            for i in range(len(arr)):
                if arr[i] == 0:
                    num1 += 1
                if box2[i] == 0:
                    num2 += 1
            return num1 == num2
        
        def count(arr, balls):
            ans = 1
            for i in range(len(balls)):
                ans *= arrange(arr[i], balls[i])
            return ans
            
            
        if len(balls) == 1:
            return 1.0
        m = max(balls) + 1
        n = sum(balls)
        k = len(balls)
        N = m ** (len(balls) - 1)
        acc = 0
        for i in range(N):
            arr = decode(i, m, k, n, balls)
            if arr is None:
                continue
            if is_equal(arr, balls):
                acc += count(arr, balls)
        return float(acc) / arrange(n // 2, n)
