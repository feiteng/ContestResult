class Solution(object):
    def getProbability(self, balls):
        """
        :type balls: List[int]
        :rtype: float
        """
        self.count = 0
        checklist = [1, 1]
        tot = sum(balls)
        self.half = tot / 2
        for i in range(2, tot + 1):
            checklist.append(checklist[-1] * i)
        
        
        def dfs(aes, box1):
            if not aes:
                if sum(box1.values()) == self.half:
                    c1 = {key:box1[key] for key in box1 if box1[key]}
                    c2 = {key:stat[key] - box1[key] for key in box1 if stat[key] - box1[key]}
                    self.count += cc(c1.values()) * cc(c2.values()) if len(c1)==len(c2) else 0
            else:
                for possible in range(aes[0][1] + 1):
                    cur = dict(box1)
                    cur[aes[0][0]] = possible
                    dfs(aes[1:], cur)
                    
        def cc(input):
            cur = tot = sum(input)
            res 
= 1
            for num in input:
                res *= (checklist[cur] / checklist[cur - num]) / checklist[num]
                cur -= num
            return res
        stat = {yuu:num for yuu,num in enumerate(balls)}
        dfs([[i,num] for i, num in enumerate(balls)], {})
        return 1.0 * self.count / cc(balls)
