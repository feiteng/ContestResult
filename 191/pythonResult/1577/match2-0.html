class Solution(object):
    def getProbability(self, balls):
        """
        :type balls: List[int]
        :rtype: float
        """
        K = len(balls)
        n = sum(balls)/2
        self.balls = balls
        self.box1 = [0 for _ in range(K)]
        self.box2 = [0 for _ in range(K)]
        self.cnt = 0
        self.total = self.combin(2*n, n)
        self.dfs(0, K, n)
        return self.cnt*1.0/self.total

    def combin(self, n, k):
        if k == 0 or k == n:
            return 1
        res = 1
        for i in range(1, n+1):
            res *= i
        for i in range(1, k+1):
            res /= i 
        for i in range(1, n-k+1):
            res /= i
        return res
    
    def dfs(self, i, K, n):
        x = 0
        y = 0
        for c in range(i):
            x += self.box1[c]
            y += self.box2[c]
        if i == K:
            if x == y:
                t1 = 0
                t2 = 0
                for c in range(K):
                    if self.box1[c] > 0:
                        t1+=1
                    if self.box2[c] > 0:
                        t2+=1
                if t1 == t2:
                    count = 1
                    for c in range(K):
                        count *= self.combin(self.balls[c], self.box1[c])
                    self.cnt+=count
            return
        for c in range(max(0, self.balls[i]-n+y), min(n-x,self.balls[i]) + 1):
            self.box1[i] = c
            self.box2[i] = self.balls[i]-c
            self.dfs(i+1, K, n)
