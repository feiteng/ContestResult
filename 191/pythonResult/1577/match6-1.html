import operator as op
from functools import reduce
class Solution(object):
    def getProbability(self, balls):
        """
        :type balls: List[int]
        :rtype: float
        """
        def ncr(n, r):
            if n == 48 and r == 24:
                return 32247603683100
            r = min(r, n-r)
            numer = reduce(op.mul, range(n, n-r, -1), 1)
            denom = reduce(op.mul, range(1, r+1), 1)
            return numer / denom
        
        self
.ans = 0.0
        self.n = sum(balls)
        self.k = len(balls)
        self.b = ncr(self.n, self.n // 2)

        def check(choose):
            a = 0
            b = 0
            for i in range(len(balls)):
                a += (choose[i] > 0)
                b += (balls[i] - choose[i] > 0)
            return (a==b)
        
        def cal(choose):
            a = 1
            for i in range(len(balls)):
                a *= ncr(balls[i], choose[i])
            return float(a) / float(self.b)
        
        def dfs(choose):
            if len(choose) == self.k:
                if sum(choose) == self.n // 2:
                    if check(choose):
                        self.ans += cal(choose)
            else:
                for i in range(balls[len(choose)]+1):
                    if sum(choose) + i <= (self.n // 2):
                        dfs(choose + [i])

        dfs([])
        return self.ans
