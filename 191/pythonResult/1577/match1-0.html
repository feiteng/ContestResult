class Solution(object):
    def getProbability(self, balls):
        """
        :type balls: List[int]
        :rtype: float
        """
        N = sum(balls)
        k = len(balls)
        def ncr(n, r):
            r = min(r, n-r)
            numer = reduce(operator.mul, range(n, n-r, -1), 1)
            denom = reduce(operator.mul, range(1, r+1), 1)
            return numer / denom
        def helper(cur, num, cof):
            if num * 2 == N:
                return (len(filter(None, cur)) == sum(len(cur) <= i or balls[i] != cur[i] for i in xrange(k))) * cof
            if len(cur) == k:
                return 0
            ans = 0
            for i in xrange(min(N / 2 - num, balls[len(cur)]) + 1):
                ans += helper(cur + [i], num + i, cof * ncr(balls[len(cur)], i))
            return ans
        return helper([], 0, 1) / float(ncr(N, N / 2))
