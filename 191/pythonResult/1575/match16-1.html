class Solution(object):
    def maxArea(self, h, w, hc, vc):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        hc = [0] + sorted(hc) + [h]
        vc = [0] + sorted(vc) + [w]
        m, n = 1, 1
        
for i in range(1,len(hc)):
            m = max(m, hc[i]-hc[i-1])
        for i in range(1,len(vc)):
            n = max(n, vc[i]-vc[i-1])
        return (m*n)%(10**9+7)
