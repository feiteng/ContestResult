class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.sort()
        verticalCuts.sort()
        wi = horizontalCuts[0]
        for i in range(1,len(horizontalCuts)):
            wi = max(wi, horizontalCuts[i]-horizontalCuts[i-1])
        wi = max(wi,h-horizontalCuts[-1])
        hi = verticalCuts[0]
        for i in range(1,len(verticalCuts)):
            hi = max(hi, verticalCuts[i]-verticalCuts[i-1])
        hi = max(hi,w-verticalCuts[-1])
        return (hi*wi)%1000000007
