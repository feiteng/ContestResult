class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.append(0)
        horizontalCuts.append(h)
        horizontalCuts.sort()
        verticalCuts.append(0)
        verticalCuts.append(w)
        verticalCuts.sort()
        H = len(horizontalCuts)
        V = len(verticalCuts)
        
        largestH = 0
        for i in range(1, H):
            if horizontalCuts[i] - horizontalCuts[i-1] > largestH:
                largestH = max(largestH, horizontalCuts[i] - horizontalCuts[i-1])
        
        largestV = 0
        for i in range(1, V):
            if verticalCuts[i] - verticalCuts[i-1] > largestV:
                largestV = max(largestV, verticalCuts[i] - verticalCuts[i-1])
        
        return (largestH * largestV) % (10**9 + 7)
