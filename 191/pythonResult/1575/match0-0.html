class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.sort()
        verticalCuts.sort()
        maxHori = horizontalCuts[0]
        for i in range(len(horizontalCuts)-1):
            maxHori = max(horizontalCuts[i+1]-horizontalCuts[i],maxHori)
        maxHori = max(maxHori,h-horizontalCuts[-1])
        maxVerti = verticalCuts[0]
        for i in range(len(verticalCuts)-1):
            maxVerti = max(verticalCuts[i+1]-verticalCuts[i],maxVerti)
        maxVerti = max(maxVerti,w-verticalCuts[-1])
        return int((maxHori * maxVerti)%(1e9+7))
