class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        horizontalCuts = [0] + sorted(horizontalCuts) + [h]
        verticalCuts = [0] + sorted(verticalCuts) + [w]
        
        res = 0
        height = 0
        width = 0
        
for i in range(1, len(horizontalCuts)):
            height = max(height, horizontalCuts[i] - horizontalCuts[i - 1])    
        for j in range(1, len(verticalCuts)):
            width = max(width, verticalCuts[j] - verticalCuts[j - 1])
        
        return (height * width) % (10 ** 9 + 7)
