class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.extend([h, 0])
        verticalCuts.extend([w, 0])
        horizontalCuts.sort()
        verticalCuts.sort()
        return max(horizontalCuts[i+1]-horizontalCuts[i] for i in range(len(horizontalCuts)-1)) * max(verticalCuts[i+1]-verticalCuts[i] for i in range(len(verticalCuts)-1)) % (10**9+7)
