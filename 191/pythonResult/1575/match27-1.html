class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.append(0)
        horizontalCuts.append(h)
        horizontalCuts.sort()
        verticalCuts.append(0)
        verticalCuts.append(w)
        verticalCuts.sort()
        disH = max([horizontalCuts[i+1] - horizontalCuts[i] for i in range(len(horizontalCuts)-1)])
        disW = max([verticalCuts[i+1] - verticalCuts[i] for i in range(len(verticalCuts)-1)])
        return (disH * disW) % (10 ** 9 + 7)
