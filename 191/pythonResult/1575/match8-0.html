class Solution(object):
    def maxArea(self, h, w, horizontalCuts, verticalCuts):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        horizontalCuts.sort()
        verticalCuts.sort()
        rows = [0] + horizontalCuts + [h]
        cols = [0] + verticalCuts + [w]
        ans = 0
        row_max = 0
        col_max = 0
        for i in range(1, len(rows)):
            row_max = max(row_max, rows[i] - rows[i - 1])
        for i in range(1, len(cols)):
            col_max = max(col_max, cols[i] - cols[i - 1])
        return (row_max * col_max) % (10 ** 9 + 7)
