class Solution(object):
    def maxArea(self, h, w, hs, vs):
        """
        :type h: int
        :type w: int
        :type horizontalCuts: List[int]
        :type verticalCuts: List[int]
        :rtype: int
        """
        mh = mv = 0
        hs.append(0)
        hs.append(h)
        vs.append(0)
        vs.append(w)
        hs.sort()
        vs.sort()
        for i in range(len(hs)-1):
            mh = max(mh, hs[i+1] - hs[i])
        for i in range(len(vs)-1):
            mv = max(mv, vs[i+1] - vs[i])
        return mh * mv % (10**9+7)
