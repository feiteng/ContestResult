from collections import defaultdict
class Solution(object):
    def minReorder(self, n, A):
        edgeMap = self.build(A) 
        que = [0]
        # print que
        # usedEdges = set(que)
        visited = set([0])
        turns = 0
        while que:
            size = len(que)
            for _ in range(size):
                element = que.pop(0)
                for a, b in edgeMap[element]:
                    for ele in [a, b]:
                        if ele not in visited:
                            visited.add(ele)
                            if b == ele:
                                turns += 1
                            que.append(ele)
        return turns
    
    def build(self, A):
        edgeMap = defaultdict(set)
        for a, b in A:
            edgeMap[a].add((a, b))
            edgeMap[b].add((a, b))
        return edgeMap
