class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        ans = 0
        nxt = collections.defaultdict(list)
        pre = collections.defaultdict(list)
        for conn in connections:
            nxt[conn[1]].append(conn[0])
            pre[conn[0]].append(conn[1])
        visited = set()
        visited.add(0)
        q = [0]
        while q:
            t = []
            for city in q:
                for i in nxt[city]:
                    if i not in visited:
                        t.append(i)
                        visited.add(i)
                for i in pre[city]:
                    if i not in visited:
                        t.append(i)
                        visited.add(i)
                        ans += 1
            # print(t)
            q = t
        return ans
