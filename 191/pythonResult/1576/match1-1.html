class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        adjMatrix = [[] for _ in range(n)]
        ht = [set() for _ in range(n)]
        for c in connections:
            ht[c[0]].add(c[1])
            adjMatrix
[c[0]].append(c[1])
            adjMatrix[c[1]].append(c[0])
        visited = [False for _ in range(n)]
        visited[0] = True
        queue = [0]
        w = len(queue)
        res = 0
        while w > 0:
            for _ in range(w):
                cur = queue[0]
                queue.pop(0)
                for node in adjMatrix[cur]:
                    if visited[node] == False:
                        visited[node] = True
                        queue.append(node)
                        if node in ht[cur]:
                            res += 1
            w = len(queue)
        return res
