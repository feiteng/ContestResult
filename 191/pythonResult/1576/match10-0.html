class Solution(object):
    def minReorder(self, n, cs):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        g = collections.defaultdict(set)
        jump = collections.defaultdict(set)
        for x,y in cs:
            g[y].add(x)
            g[x].add(y)
            jump[y].add(x)
        #print jump
            
        
        stack = [0]
        seen = set()
        seen.add(0)
        self.res = 0
        
        while stack:
            cur = stack.pop()
            #print cur
            for nei in g[cur]:
                if nei not in seen:
                    seen.add(nei)
                    if nei not in jump[cur]:
                        self.res += 1
                    stack.append(nei)
            
        return self.res
