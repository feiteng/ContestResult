class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        connects = collections.defaultdict(list)
        for f, t in connections:
            connects[f].append(t)
            connects[t].append(f)
        connections = set(map(tuple, connections))
        visited = set([0])
        ans = 0
        q = [0]
        for f in q:
            for t in connects[f]:
                if t in visited:
                    continue
                q.append(t)
                visited.add(t)
                if (f, t) in connections:
                    ans += 1
        return ans
