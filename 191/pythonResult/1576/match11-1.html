class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        graph = collections.defaultdict(set)
        for x, y in connections:
            graph[x].add(y)
            graph[y].add(x)
        frontier = deque()
        frontier.append(0)
        dist = collections
.defaultdict(int)
        dist[0] = 0
        while frontier:
            curr = frontier.popleft()
            for nbr in graph[curr]:
                if nbr not in dist:
                    dist[nbr]=dist[curr]+1
                    frontier.append(nbr)
        ans = 0
        for x, y in connections:
            if dist[x] < dist[y]:
                ans += 1
        return ans
