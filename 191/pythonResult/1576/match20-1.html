class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        adj = [[] for _ in range(n)]
        direc = [set() for _ in range(n)]
        for c in connections:
            adj[c[0]].append(c[1])
            adj
[c[1]].append(c[0])
            direc[c[0]].add(c[1])
        visited = [False for _ in range(n)]
        
        q = [0]
        visited[0] = True
        cnt = 0
        width = len(q)
        while width > 0:
            for _ in range(width):
                cur = q[0]
                q.pop(0)
                for node in adj[cur]:
                    if visited[node] == False:
                        visited[node] = True
                        q.append(node)
                        if node not in direc[cur]:
                            cnt+=1
            width = len(q)
            #print q
        return n-1-cnt
