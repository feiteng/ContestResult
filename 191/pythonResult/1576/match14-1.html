class Solution(object):
    def minReorder(self, n, connections):
        graph = [set() for _ in xrange(n)]
        rgraph = [set() for _ in xrange(n)]
        for v,u in connections:
            graph[u].add(v)  # actually, backwards
            rgraph[v].add(u)
        stack = [0]
        seen = {0}
        ans = 0
        while stack:
            node = stack.pop()
            
for nei in graph[node]:
                if nei not in seen:
                    stack.append(nei)
                    seen.add(nei)
            for neir in rgraph[node]:
                if neir not in seen:
                    stack.append(neir)
                    ans += 1
                    seen.add(neir)
        return ans
