class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        adj = [[] for _ in xrange(n)]
        for i, j in connections:
            adj[i].append((j,1))
            adj[j].append((i,0))
        r = [0]
        def dfs(u, p):
            for v, d in adj[u]:
                if v==p: continue
                r[0] += d
                dfs(v, u)
        dfs(0, -1)
        return r[0]
