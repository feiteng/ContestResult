class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        
        E = collections.defaultdict(lambda: 0)
        D = collections.defaultdict(set)
        for c in connections:
            E[tuple(c)] = 0
            D[c[0]].add(c[1])
            D[c[1]].add(c[0])
        
        queue = collections.deque()
        queue.append(0)
        
        count = 0
        visited = set([0])
        while len(queue) > 0:
            node = queue.popleft()
            for new in D[node]:
                if new not in visited:
                    if (new, node) not in E: 
                        count += 1
                    visited.add(new)
                    queue.append(new)
        return count
