class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        a,b,s=[set() for _ in range(n)],[set() for _ in range(n)],0
        for i,j in connections:
            a[j].add(i)
            b[i].add(j)
        st=[0]
        while st:
            i=st.pop()
            for j in a[i]:
                st.append(j)
                b[j].remove(i)
            for j in b[i]:
                st.append(j)
                s+=1
                a[j].remove(i)
        return s
