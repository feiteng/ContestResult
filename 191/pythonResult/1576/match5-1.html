class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        undirected_graph 
= collections.defaultdict(list)
        directed_graph = collections.defaultdict(set)
        
        for u, v in connections:
            undirected_graph[u].append(v)
            undirected_graph[v].append(u)
            directed_graph[u].add(v)
        
        num_changes = 0
        queue = collections.deque([0])
        visited = {0}
        while queue:
            u = queue.popleft()
            for child in undirected_graph[u]:
                if child not in visited:
                    if u not in directed_graph[child]:
                        num_changes += 1
                    queue.append(child)
                    visited.add(child)
        return num_changes
