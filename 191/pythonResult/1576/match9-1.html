from collections import defaultdict, deque
class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        G = defaultdict(list)
        G2 = defaultdict(list)
        visited = set([0])
        stack = deque
([0])
        for src, dst in connections:
            G[src].append(dst)
            G2[dst].append(src)
        
        ct = 0
        while stack:
            city = stack.popleft()
            for x in G[city]:
                if x in visited:
                    continue
                else:
                    visited.add(x)
                    stack.append(x)
                    ct += 1
            for y in G2[city]:
                if y not in visited:
                    stack.append(y)
                    visited.add(y)
        return ct
