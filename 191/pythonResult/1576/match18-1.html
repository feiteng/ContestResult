class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        diction=dict()
        for i in range(n):
            diction[i]=[[],[]]
        for i in range(len(connections)):
            diction[connections[i][0]][1].append(connections[i][1])#出度
            diction
[connections[i][1]][0].append(connections[i][0])
        visited=set([0])
        up=[0]
        down=[]
        ans=0
        while len(visited)<n:
            while up:
                test=up.pop(0)
                for i in range(len(diction[test][0])):
                    if diction[test][0][i] in visited:
                        continue
                    else:
                        down.append(diction[test][0][i])
                    visited.add(diction[test][0][i])
                for j in range(len(diction[test][1])):
                    if diction[test][1][j] in visited:
                        continue
                    else:
                        down.append(diction[test][1][j])
                        ans+=1
                    visited.add(diction[test][1][j])
            up=down[:]
            down=[]
        return ans
