class Solution(object):
    def minReorder(self, n, connections):
        """
        :type n: int
        :type connections: List[List[int]]
        :rtype: int
        """
        from collections import defaultdict
        g = defaultdict(list)
        
        for c in connections:
            a, b = c
            # a -> b
            g
[b].append((a, 1))
            g[a].append((b, 0))
        
        def recurse(node, parent):
            ans = 0
            
            for child in g[node]:
                c, fromChild = child
                if c != parent:
                    if not fromChild:
                        ans += 1
                    ans += recurse(c, node)
            
            return ans
        
        return recurse(0, -1)
