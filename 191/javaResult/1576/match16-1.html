class Solution {
    public int minReorder(int n, int[][] connections) {
        List<List<Integer>> start = new ArrayList(), end = new ArrayList();
        for (int i = 0; i < n; i++) {
            start.add(new ArrayList());
            end.add(new ArrayList());
        }
        for (int[] c : connections) {
            start.get(c[0]).add(c[1]);
            end.get(c[1]).add(c[0]);
        }
        
        Queue<Integer> q = new LinkedList();
        q.offer(0);
        int count = 0;
        boolean[] visited = new boolean[n];
        visited[0] = true;
        while (!q.isEmpty()) {
            int cur = q.poll();
            for (int nb : end.get(cur)) {
                if (!visited[nb]) {
                    q.offer(nb);
                    visited[nb] = true;
                }
            }
            for (int nb : start.get(cur)) {
                if (!visited[nb]) {
                    count++;
                    // System.out.println(cur + " " + nb);
                    q.offer(nb);
                    visited[nb] = true;
                
}
            }
        }
        return count;
    }
}
