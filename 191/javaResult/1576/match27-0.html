class Solution {
    public int minReorder(int n, int[][] cs) {
        ArrayList<HashSet<Integer>> g = new ArrayList<>();
        ArrayList<HashSet<Integer>> g2 = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            g.add(new HashSet<>());
            g2.add(new HashSet<>());
        }
        for (int[] c : cs) {
            g.get(c[0]).add(c[1]);
            g2.get(c[0]).add(c[1]);
            g2.get(c[1]).add(c[0]);
        }
        int[] p = new int[n];
        for (int i = 0; i < n; ++i)
            p[i] = -1;
        ArrayDeque<Integer> q = new ArrayDeque<>();
        q.add(0);
        int ans = 0;
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            for (int v : g2.get(u)) {
                if (p[v]==-1&&v!=0) {
                    p[v] = u;
                    q.add(v);
                    //System.out.println(u+" "+v);
                    if (g.get(u).contains(v))
                        ans++;
                }
            }
        }
        return ans;
    }
}
