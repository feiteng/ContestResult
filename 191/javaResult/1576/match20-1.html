class Solution {
    List<Integer>[] a;
    List<Integer>[] b;
    int[] v;
    
int ans;
    public int minReorder(int n, int[][] e) {
        a = new List[n];
        b = new List[n];
        v = new int[n];
        ans = 0;
        
for (int i=0; i<n; i++) {
            a[i] = new ArrayList<>();
            b[i] = new ArrayList<>();
        }
        for (int i=0; i<e.length; i++) {
            a[e[i][0]].add(e[i][1]);
            
b[e[i][0]].add(1);
            a[e[i][1]].add(e[i][0]);
            b[e[i][1]].add(0);
        }
        search(0);
        return ans;
    }
    private void search(int r) {
        v[r] = 1;
        for (int i=0; i<a[r].size(); i++) {
            if (v[a[r].get(i)] == 0) {
                ans += b[r].get(i);
                search(a[r].get(i));
            }
        }
    }
}
