class Solution {
    ArrayList<Integer>[] in;
    ArrayList<Integer>[] out;
    boolean visited[];
    int ans;
    public int minReorder(int n, int[][] connections) {
        in = new ArrayList[n];
        out = new ArrayList[n];
        ans = 0;
        
visited = new boolean[n];
        for(int i=0;i<n;i++) {
            in[i]=new ArrayList();
            out[i]=new ArrayList();
        }
        for(int i=0;i<connections.length;i++) {
            int from = connections[i][0];
            int to = connections[i][1];
            in[to].add(from);
            out[from].add(to);
        
}
        dfs(0);
        return ans;
    }
    
    public void dfs(int pos) {
        visited[pos]=true;
        for(int i=0;i<in[pos].size();i++) {
            int pos1=in[pos].get(i);
            if (visited[pos1])continue;
            dfs(pos1);
        }
        for(int i=0;i<out[pos].size();i++) {
            int pos1=out[pos].get(i);
            if (visited[pos1])continue;
            ans++;
            dfs(pos1);
        }
        
    }
    
}
