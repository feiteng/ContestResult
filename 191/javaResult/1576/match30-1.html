class Solution {
    public int minReorder(int n, int[][] connections) {
        HashSet<Integer>[] graph = new HashSet[n];
        List<Integer>[] uG = new List[n];
        for(int i=0; i<n; i++){
            graph[i] = new HashSet();
            uG[i] = new ArrayList();
        }
        for(int[] e : connections){
            graph[e[0]].add(e[1]);
            uG[e[0]].add(e[1]);
            uG[e[1]].add(e[0]);
        }
        int res = 0;
        boolean[] visited = new boolean[n];
        Queue<Integer> q = new LinkedList();
        q.offer(0);
        visited[0] = true;
        while(!
q.isEmpty()){
            int cur = q.poll();
            for(int next : uG[cur]){
                if(visited[next]){
                    continue;
                }
                visited[next] = true;
                if(!graph[next].contains(cur)){
                    res++;
                }
                q.offer(next);
            }
        }
        return res;
    }
}
