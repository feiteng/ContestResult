class Solution {
    public int minReorder(int n, int[][] c) {
        List<List<Integer>> g = new ArrayList<>();
        List<List<Integer>> gr = new ArrayList<>();
        for (int i=0; i < n; ++i) {
            g.add(new ArrayList<>());
            gr.add(new ArrayList<>());
        }
        for (int i=0; i < c.length; ++i) {
            g.get(c[i][0]).add(c[i][1]);
            gr.get(c[i][1]).add(c[i][0]);
        }
        
        int rev = 0;
        boolean[] vis = new boolean[n];
        Queue<Integer> q = new LinkedList<>();
        q.add(0);
        while (q.size() > 0) {
            int node = q.remove();
            
            vis[node] = true;
            for (int i = 0; i < gr.get(node).size(); ++i) {
                if (!vis[gr.get(node).get(i)]) {
                    
                    rev++;
                    q.add(gr.get(node).get(i));
                }
                
            }
            for (int i = 0; i < g.get(node).size(); ++i) {
                if (!vis[g.get(node).get(i)]) {
                    
                    q.add(g.get(node).get(i));
                }
                
            }
            
        }
        return n - 1 - rev;
        
        
    }
}
