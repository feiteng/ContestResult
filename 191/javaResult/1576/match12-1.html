class Solution {
    public int minReorder(int n, int[][] connections) {
        List<Integer>[] lists = new List[n];
        
List<Integer>[] revLists = new List[n];
        for (int i = 0; i < n; ++i) {
            lists[i] = new ArrayList<>();
            revLists[i] = new ArrayList<>();
        }
        for (int[] conn : connections) {
            lists[conn[0]].add(conn[1]);
            revLists[conn[1]].add(conn[0]);
        }
        Queue<Integer> que = new LinkedList<>();
        boolean[] visited = new boolean[n];
        que.offer(0);
        visited[0] = true;
        int res = 0;
        
while (!que.isEmpty()) {
            int cur = que.poll();
            for (int next : revLists[cur]) {
                if (visited[next]) continue;
                visited[next] = true;
                que.offer(next);
            }
            for (int next : lists[cur]) {
                if (visited[next]) continue;
                visited[next] = true;
                que.offer(next);
                ++res;
            }
        }
        return res;
    }
}
