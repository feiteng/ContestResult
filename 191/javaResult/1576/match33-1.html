class Solution {
    public int minReorder(int n, int[][] connections) {
        // bfs
        Map<Integer, Set<Integer>> nextMap = new HashMap<>();
        Map<Integer, Set<Integer>> fromMap = new HashMap<>();
        for (int[] conn : connections) {
            nextMap.putIfAbsent(conn[0], new HashSet<>());
            fromMap.putIfAbsent(conn[1], new HashSet<>());
            nextMap.get(conn[0]).add(conn[1]);
            fromMap.get(conn[1]).add(conn[0]);
        }
        Queue<Integer> q = new ArrayDeque<>();
        Set<Integer> visited = new HashSet<>();
        visited.add(0);
        q.add(0);
        int res = 0;
        while (!q.isEmpty()) 
{
            int cur = q.poll();
            if (fromMap.containsKey(cur)) {
                for (int from : fromMap.get(cur)) {
                    // put all into
                    if (!visited.contains(from)) {
                        visited.add(from);
                        q.add(from);
                    }
                }
            }
            if (nextMap.containsKey(cur)) {
                // change the direction
                
for (int next : nextMap.get(cur)) {
                    if (!visited.contains(next)) {
                        // System.out.println("cur=" + cur);
                        // System.out.println("next=" + next);
                        visited.add(next);
                        res++;
                        q.add(next);
                    }
                }
            }
        }
        return res;
    }
}
