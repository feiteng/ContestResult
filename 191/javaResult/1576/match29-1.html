class Solution {
    public int minReorder(int n, int[][] connections) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        Set<String> path = new HashSet<>();
        for (int[] conn : connections) {
            path.add(conn[0] + "|" + conn[1]);
            List<Integer> list = map.getOrDefault(conn[0], new ArrayList<>());
            list.add(conn[1]);
            map.put(conn[0], list);
            
            list = map.getOrDefault(conn[1], new ArrayList<>());
            
list.add(conn[0]);
            map.put(conn[1], list);
        }
        
        Set<Integer> visited = new HashSet<>();
        int revCount = 0;
        Deque<Integer> q = new ArrayDeque<>();
        q.offerLast(0);
        visited.add(0);
        while (!q.isEmpty()) {
            int curr = q.pollFirst();
            List<Integer> nei = map.getOrDefault(curr, new ArrayList<>());
            for (int ne : nei) {
                if (visited.add(ne)) {
                    if (!path.contains(ne + "|" + curr)) {
                        revCount++;
                    }
                    
q.offerLast(ne);                    
                }
            }            
        }
        return revCount;
    }
}
