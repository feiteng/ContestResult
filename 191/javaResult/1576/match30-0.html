class Solution {
    public int minReorder(int n, int[][] connections) {
        List<Integer>[] g1 = new List[n];
        List<Integer>[] g2 = new List[n];
        for (int i = 0; i < n; i++) {
            g1[i] = new ArrayList<>();
            g2[i] = new ArrayList<>();
        }
        for (int[] e : connections) {
            g1[e[0]].add(e[1]);
            g2[e[1]].add(e[0]);
        }
        boolean[] visited = new boolean[n];
        int reach = 1, cnt = 0;
        Deque<Integer> q = new ArrayDeque<>();
        q.offerLast(0);
        visited[0] = true;
        while (reach < n && !q.isEmpty()) {
            for (int i = q.size() - 1; i >= 0; i--) {
                int u = q.pollFirst();
                for (int v : g1[u]) {
                    if (!visited[v]) {
                        cnt++;
                        reach++;
                        q.offerLast(v);
                        visited[v] = true;
                    }
                }
                for (int v : g2[u]) {
                    if (!visited[v]) {
                        reach++;
                        q.offerLast(v);
                        visited[v] = true;
                    }
                }
            }
        }
        return cnt;
    }
}
