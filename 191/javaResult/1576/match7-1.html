class Solution {
    public int minReorder(int n, int[][] connections) {
        ArrayList<ArrayList<Integer>> in = new ArrayList<>();
        ArrayList<ArrayList<Integer>> out = new ArrayList<>();
        
HashSet<Integer> visited = new HashSet<>();
        Queue<Integer> q = new LinkedList<>();
        
        for (int i = 0; i < n; ++i) {
            in.add(new ArrayList<Integer>());
            out.add(new ArrayList<Integer>());
        }
        
        for (int[] connection : connections) {
            int v1 = connection[0];
            int v2 = connection[1];
            out.get(v1).add(v2);
            in.get(v2).add(v1);
        }
        
        int count = 0;
        q.add(0);
        visited.add(0);
        
while (!q.isEmpty()) {
            int cur = q.remove();
            for (int outNode : out.get(cur)) {
                if (!visited.contains(outNode)) {
                    q.add(outNode);
                    visited.add(outNode);
                }
            }
            for (int inNode : in.get(cur)) {
                if (!visited.contains(inNode)) {
                    ++count;
                    q.add(inNode);
                    visited.add(inNode);
                
}
            }
        }
        return n - 1 - count;
    }
}
