class Solution {
    
    int c = 0;
    void go(int rt,int fa){
        for(Integer g1:g[rt]){
            if(g1==fa) continue;
            if(!st[g1].contains(rt)){
               // System.out.println(rt+","+g1);
                c++;
            }
            go(g1,rt);
        }
    }
    
    Set<Integer> st[] ;    
    List<Integer> g[] ;
    public int minReorder(int n, int[][] con) {
        
         st = new Set[n+1];
        
        g = new List[n+1];
        
        for(int i=0;i<=n;++i){
            st[i] = new HashSet<Integer>();
            g[i] =new ArrayList<Integer>();
        }
        
        for(int[] c:con){
            g[c[0]].add(c[1]);
            g[c[1]].add(c[0]);
            st[c[0]].add(c[1]);
        }
        
        go(0,-1);
        return c;
        
        
    }
}
