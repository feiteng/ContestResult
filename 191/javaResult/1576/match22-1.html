class Solution {
    public int minReorder(int n, int[][] connections) {
        Map<Integer, Set<Integer>> from = new HashMap<>();
        Map<Integer, Set<Integer>> to = new HashMap<>();
        
        for(int[] co: connections){
            Set<Integer> r1 = from.getOrDefault(co[0], new HashSet<Integer>());
            r1.add(co[1]);
            from.put(co[0], r1);
            Set<Integer> r2 = to.getOrDefault(co[1], new HashSet<Integer>());
            r2.add(co[0]);
            to.put(co[1], r2);
        }
        
        Queue<Integer> q= new LinkedList<>();
        q.offer(0);
        int res = 0;
        boolean[] v = new boolean[n];
        v[0] = true;
        while(q.size() > 0){
            int cur = q.poll();
            if(from.containsKey(cur)){
                Set<Integer> r1 = from.get(cur);
                for(int t: r1)
{
                    if(!v[t]){
                        q.offer(t);
                        v[t] = true;
                        res++;
                        //System.out.println(cur + " " + t);
                    }
                }
            }
            if(to.containsKey(cur)){
                Set<Integer> r1 = to.get(cur);
                for(int t: r1)
{
                    if(!v[t]){
                        q.offer(t);
                        v[t] = true;
                        
                    }
                }
            }
        }
        return res;
    }
}
