class Solution {
    int ans = 0;
    public int minReorder(int n, int[][] connections) {
        ans = 0;
        Map<Integer, List<Integer>> original = new HashMap<>();
        Map<Integer, List<Integer>> reverse = new HashMap<>();

        for(int[] connection: connections) {
            
if(!original.containsKey(connection[0])) {
                original.put(connection[0], new ArrayList<>());
            }
            original.get(connection[0]).add(connection[1]);

            if(!reverse.containsKey(connection[1])) {
                reverse.put(connection[1], new ArrayList<>());
            }
            reverse.get(connection[1]).add(connection[0]);
        }


        Set<Integer> visited = new HashSet<>();
        visited.add(0);
        LinkedList<Integer> list = new LinkedList<>();
        list.add(0);

        while(!list.isEmpty()) {
            int city = list.removeFirst();
            List<Integer> neis = getNeis(visited, original, reverse, city);
            list.addAll(neis);
        }

        return ans;
    }

    private List<Integer> getNeis(Set<Integer> visited, Map<Integer, List<Integer>> original, Map<Integer, List<Integer>> reverse, int city) {
        List<Integer> list = new ArrayList<>();
        
if(original.containsKey(city)) {
            for(int each: original.get(city)) {
                if(!visited.contains(each)) {
                    ans++;
                    list.add(each);
                    visited.add(each);
                }
            }
        }
        if(reverse.containsKey(city)) {
            
for(int each: reverse.get(city)) {
                if(!visited.contains(each)) {
                    list.add(each);
                    visited.add(each);
                }
            }
        }
        return list;
    }
}
