class Solution {
    public int minReorder(int n, int[][] connections) {
        Map<Integer, Set<Integer>> edges = new HashMap<>();
        Map<Integer, Set<Integer>> relation = new HashMap<>();
        
        for (int[] con : connections) {
            relation.computeIfAbsent(con[0], v -> new HashSet<>()).add(con[1]);
            edges.computeIfAbsent(con[0], v -> new HashSet<>()).add(con[1]);
            edges.computeIfAbsent(con[1], v -> new HashSet<>()).add(con[0]);
        }
        
        Deque<Integer> queue = new ArrayDeque<>();
        Set<Integer> visited = new HashSet<>();
        int count = 0;
        
        queue.offer(0);
        visited.add(0);
        
        
while (!queue.isEmpty()) {
            int cur = queue.poll();
            
            for (Integer next : edges.get(cur)) {
                if (!visited.contains(next)) {
                    visited.add(next);
                    if (!relation.containsKey(next) || !relation.get(next).contains(cur)) {
                        count++;
                    }
                    queue.offer(next);
                
}
            }
        }
        
        return count;
        
    }
}
