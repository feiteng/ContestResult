class Solution {
    public int minReorder(int n, int[][] connections) {
      Map<Integer, List<Integer>> min = new HashMap<>();
      Map<Integer, List<Integer>> mout = new HashMap<>();
      for(int[] c : connections) {
        if(!min.containsKey(c[1])) {
          min.put(c[1], new ArrayList<>());
        }
        min.get(c[1]).add(c[0]);

        if(!mout.containsKey(c[0])) {
          mout.put(c[0], new ArrayList<>());
        }
        mout.get(c[0]).add(c[1]);
      }

      boolean[] vis = new boolean[n];
      vis[0] = true;
      LinkedList<Integer> q = new LinkedList<>();
      int res = 0;
      q.add(0);
      while(!q.isEmpty()) {
        int v = q.poll();
        if(mout.containsKey(v)) {
          for (int v1 : mout.get(v)) {
            if(!vis[v1]) {
              res++;
              q.add(v1);
              vis[v1] = true;
            }
          }
        }
        if(min.containsKey(v)) {
          for (int v1 : min.get(v)) {
            if(!vis[v1]) {
              q.add(v1);
              vis[v1] = true;
            }
          }
        }
      }
      return res;
    }
  }
