class Solution {
    public int minReorder(int n, int[][] edges) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        Map<Integer, List<Integer>> rmap = new HashMap<>();
        
        for(int[] edge: edges){
            map.computeIfAbsent(edge[0], k->new ArrayList<>(2)).add(edge[1]);
            rmap.computeIfAbsent(edge[1], k->new ArrayList<>(2)).add(edge[0]);
        }
        
        boolean[] visited = new boolean[n];
        int count = 0;
        Queue<Integer> queue = new ArrayDeque<>();
        queue.add(0);
        visited[0] = true;
        
        int changes = 0;
        
        while(!queue.isEmpty()){
            int node = queue.poll();
            for(int next : rmap.getOrDefault(node, Collections.emptyList())){
                if(!visited[next]){
                    visited[next] = true;
                    queue.add(next);
                }
            }
            for(int next : map.getOrDefault(node, Collections.emptyList())){
                if(!visited[next]){
                    changes++;
                    visited[next] = true;
                    queue.add(next);
                }
            }
        }
        return changes;
        
    }
}
