class Solution {
    public int minReorder(int n, int[][] connections) {
        List<List<Integer>> fromNode = new ArrayList(); // an edge from this node to other nodes
        List<List<Integer>> toNode = new ArrayList(); // an edge from other nodes to this node
        
        for(int i = 0; i < n; i++) {
            fromNode.add(new ArrayList());
            toNode.add(new ArrayList());
        }
        
        for(int[] connection: connections) {
            int from = connection[0];
            int to = connection[1];
            
            fromNode.get(from).add(to);
            toNode.get(to).add(from);
        }
        
        Set<Integer> visited = new HashSet();
        int result = 0;
        Queue<Integer> queue = new LinkedList();
        queue.offer(0);
        
        while(!queue.isEmpty()) {
            int curr = queue.poll(); // want to caculate node from/to this node
            visited.add(curr);
            List<Integer> nodesToCurr = toNode.get(curr);
            for(int tmp: nodesToCurr) {
                if(!visited.contains(tmp)) {
                    // visited.add(tmp);
                    queue.offer(tmp);
                }
            }
            
            List<Integer> currToNodes = fromNode.get(curr);
            for(int tmp: currToNodes) {
                if(!visited.contains(tmp)) {
                    result++;
                    // visited.add(tmp);
                    queue.offer(tmp);
                }
            }
        }
        
        return result;
    }
}
