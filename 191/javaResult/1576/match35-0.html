class Solution {
  Set<String> conn = new HashSet<>();
    List<Integer>[] adj;
    boolean[] vst;
    public int minReorder(int n, int[][] connections) {
        adj = new List[n];
        vst = new boolean[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }

        for (int i = 0; i < connections.length; i++) {
            int x = connections[i][0];
            int y = connections[i][1];
            adj[x].add(y);
            adj[y].add(x);
            conn.add(x + "," + y);
        }
        
        return dfs(0);
    }
    
    int dfs(int x) {
        if (vst[x]) return 0;
        vst[x] = true;
        
        int ret = 0;
        for (int y : adj[x]) {
            if (vst[y]) continue;
            String key = y + "," + x;
            if (!conn.contains(key)) {
                ret++;
            }
            ret += dfs(y);
        }
        return ret;
    }
}
