class Solution {
    public int minReorder(int n, int[][] connections) {
        Set<Integer>[] nextAdj = new Set[n];
        Set<Integer>[] adj = new Set[n];
        for(int i = 0 ; i < n ; i++){
            adj[i] = new HashSet<>();
            nextAdj[i] = new HashSet<>();
        }
        int[] degree = new int[n];
        for(int[] connection: connections){
            adj[connection[0]].add(connection[1]);
            adj[connection[1]].add(connection[0]);
            nextAdj[connection[1]].add(connection[0]);
            degree[connection[1]]++;
        }
        boolean[] seen = new boolean[n];
        Deque<Integer> queue = new ArrayDeque<>();
        queue.add(0);
        seen[0] = true;
        // System.out.println(Arrays.toString(degree));
        int res = 0;
        while(!queue.isEmpty()){
            int cur = queue.poll();
            res += degree[cur];
            for(int next: nextAdj[cur]){
                degree[next]++;
            }
            if(cur != 0 && degree[cur] > 0){
                res--;
            }
        // System.out.println(cur + " " + res);
            for(int next: adj[cur]){
                if(seen[next]){
                    continue;
                }
                seen[next] = true;
                queue.add(next);
            }
        }
        return n - 1 - res;
    }
}
