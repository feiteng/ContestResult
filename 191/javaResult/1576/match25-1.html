class Solution {
    ArrayList<Integer>[] adj;
    HashSet<Integer>[] edge;
    int[] par;
    int count = 0;
    public int minReorder(int n, int[][] connections) {
        adj = new ArrayList[n];
        edge = new HashSet[n];
        par = new int[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<Integer>();
            edge[i] = new HashSet<Integer>();
        }
        
for (int i = 0; i < connections.length; i++) {
            int a = connections[i][0];
            int b = connections[i][1];
            adj[a].add(b);
            adj[b].add(a);
            edge[b].add(a);
        }
        dfs(0);
        return count;
    }
    public void dfs(
int x) {
        for (int y: adj[x]) {
            if (y != par[x]) {
                if (edge[y].contains(x)) {
                    count++;
                    par[y] = x;
                    dfs(y);
                }
                else {
                    par[y] = x;
                    dfs(y);
                }
            }
        }
    }
}
