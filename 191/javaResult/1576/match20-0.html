class Solution {
    Set<Integer>undirect[];
    Set<Integer>direct[];
    Set<Integer>set=new HashSet<>();
    int res=0;
    public int minReorder(int n, int[][] connections) {
        undirect=new HashSet[n];
        direct=new HashSet[n];
        for(int i=0;i<n;i++){
            undirect[i]=new HashSet<>();
            direct[i]=new HashSet<>();
        }
        for(int pair[]:connections){
            int v1=pair[0];
            int v2=pair[1];
            direct[v1].add(v2);
            undirect[v1].add(v2);
            undirect[v2].add(v1);
        }
        dfs(0);
        return res;
    }
    public void dfs(int root){
        set.add(root);
        Set<Integer>dchilds=direct[root];
        Set<Integer>udchilds=undirect[root];
        for(int uchild:udchilds){
            if(set.contains(uchild))continue;
            if(dchilds.contains(uchild))res++;
            dfs(uchild);
        }
    }
}
