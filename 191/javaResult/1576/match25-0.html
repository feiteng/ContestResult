import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {
    List<Integer>[] g;
    Set<Long> edges;
    int fu;
    int n;

    public int minReorder(int n, int[][] connections) {
        g = new List[n];
        this.n = n;
        for (int i = 0; i < n; i++)
            g[i] = new ArrayList<>();
        edges = new HashSet<>();

        for (int[] pan : connections) {
            int u = pan[0];
            int v = pan[1];
            g[u].add(v);
            g[v].add(u);
            edges.add((long) u * n + v);
        }
        fu = 0;
        dfs(0, -1);
        return fu;
    }

    void dfs(int u, int p) {
        for (int v : g[u]) {
            if (v != p) {
                if (!edges.contains((long) v * n + u)) {
                    fu++;
                }
                dfs(v, u);
            }
        }
    }
}
