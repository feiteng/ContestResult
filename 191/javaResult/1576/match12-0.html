class Solution {
    public int minReorder(int n, int[][] connections) {
          LinkedList<Integer>[][] relas = new LinkedList[n][2];
        for(int i=0;i<n;i++) {
            relas[i][0] = new LinkedList<>();
            relas[i][1] = new LinkedList<>();
        }
        for(int[] con:connections) {
           relas[con[0]][0].add(con[1]);
           relas[con[1]][1].add(con[0]);
        }
        LinkedList<Integer> queue = new LinkedList<>();
        boolean[] visited = new boolean[n];
        queue.offer(0);
        int cnt=  0;
        visited[0] = true;
       
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            //逆序
            for(int next:relas[cur][1]) {
                if(!visited[next]) {
                    queue.offer(next);
                    visited[next] = true;
                   
                }
            }
            //正序可以达到的点
            for(int next:relas[cur][0]) {
                if(!visited[next]) {
                    queue.offer(next);
                    visited[next] = true;
                     cnt++;
                }
            }
            
        }
        return cnt;
    }
}
