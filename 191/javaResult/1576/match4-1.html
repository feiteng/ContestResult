class Solution {
    private HashMap<Integer, List<Integer>> inEdges = new HashMap<>();
    private HashMap<Integer, List<Integer>> outEdges = new HashMap<>();
    
    public int minReorder(int n, int[][] connections) {
        Queue<Integer> queue = new LinkedList<>();
        int num = 0;
        
        
for (int i = 0; i <= connections.length; i += 1) {
            inEdges.put(i, new LinkedList<Integer>());
            outEdges.put(i, new LinkedList<Integer>());
        }
        
        for (int i = 0; i < connections.length; i += 1) {
            int a = connections[i][0];
            int b = connections[i][1];
            List<Integer> l_in = inEdges.get(b);
            List<Integer> l_out = outEdges.get(a);
            l_in.add(a);
            l_out.add(b);
            inEdges.put(b, l_in);
            outEdges.put(a, l_out);
        }
        
        queue.add(0);
        
        Set<Integer> visited = new HashSet<>();
        
visited.add(0);
        
        while (!queue.isEmpty()) {
            int a = queue.poll();
            for (int b : inEdges.get(a)) {
                if (!visited.contains(b)) {
                    queue.add(b);
                    visited.add(b);
                }
            }
            
            for (int b : outEdges.get(a)) {
                if (!visited.contains(b)) {
                    queue.add(b);
                    visited.add(b);
                    num += 1;
                }
            }
        }
        
        return num;
    }
}
