class Solution {
    static long MOD = 1000000007L;
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        long diffw = 1;
        long diffh = 1;
        int prev = 0;
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        for (int i : horizontalCuts) {
            diffh = Math.max(diffh, i-prev);
            prev = i;
        }
        diffh = Math.max(diffh, h - prev);
        prev = 0;
        for (int i : verticalCuts) {
            diffw = Math.max(diffw, i-prev);
            prev = i;
        }
        diffw = Math.max(diffw, w - prev);
        return (int)((diffh*diffw) % MOD);
    }
}
