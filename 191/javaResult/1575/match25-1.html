class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        long height = horizontalCuts[0];
        long width = verticalCuts[0];
        int a = horizontalCuts.length;
        int b = verticalCuts.length;
        
for (int i = 0; i < a-1; i++) {
            height = Math.max(height, horizontalCuts[i+1]-horizontalCuts[i]);
        }
        for (int i = 0; i < b-1; i++) {
            width = Math.max(width, verticalCuts[i+1]-verticalCuts[i]);
        }
        height = Math.max(height, h-horizontalCuts[a-1]);
        width = Math.max(width, w-verticalCuts[b-1]);
        long ans = (long) height* (long) width;
        return (int) (ans % 1000000007);
    }
}
