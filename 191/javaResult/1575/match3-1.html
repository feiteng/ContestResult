class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        int l = horizontalCuts.length;
        int m = verticalCuts.length;
        
long maxH = 0;
        long maxW = 0;
        int last = 0;
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        for (int i = 0; i < l; ++i) {
            maxH = Math.max(maxH, horizontalCuts[i] - last);
            last = horizontalCuts[i];
        }
        maxH = Math.max(maxH, h - last);
        last = 0;
        for (int i = 0; i < m; ++i) {
            maxW = Math.max(maxW, verticalCuts[i] - last);
            last = verticalCuts[i];
        }
        maxW = Math.max(maxW, w - last);
        return (int)((maxH * maxW) % 1000000007);
    }
}
