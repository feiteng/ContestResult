class Solution {
    public int maxArea(int h, int w, int[] hc, int[] vc) {
        
Arrays.sort(hc);
        Arrays.sort(vc);
        int x = hc[0];
        for (int i = 1; i < hc.length; i++) {
            x = Math.max(x, hc[i] - hc[i - 1]);
        }
        x = Math.max(x, h - hc[hc.length - 1]);
        
        int y = vc[0];
        for (int i = 1; i < vc.length; i++) {
            y = Math.max(y, vc[i] - vc[i - 1]);
        }
        y = Math.max(y, w - vc[vc.length - 1]);
        
        long area = (long)x * (long)y;
        return (int)(area % 1000000007);
        
    }
}
