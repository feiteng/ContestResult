class Solution {
    int mod = 1000000007;
    public int maxArea(int h, int w, int[] hs, int[] vs) {
        Arrays.sort(hs);
        Arrays.sort(vs);
        int mh = 0, vh = 0;
        int prev = 0;
        
for (int n : hs) {
            mh = Math.max(n - prev, mh);
            prev = n;
        }
        mh = Math.max(h - hs[hs.length - 1], mh);
        prev = 0;
        
for (int n : vs) {
            vh = Math.max(vh, n - prev);
            prev = n;
        }
        vh = Math.max(vh, w - vs[vs.length - 1]);
        long res = (long)mh*(long)vh;
        res %= mod;
        return (int)res;
    }
}
