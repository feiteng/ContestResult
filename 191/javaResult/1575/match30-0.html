class Solution {
    public int maxArea(int h, int w, int[] hc, int[] vc) {
        Arrays.sort(hc);
        Arrays.sort(vc);
        long mh = Math.max(hc[0],h-hc[hc.length-1]);
        long mv = Math.max(vc[0],w-vc[vc.length-1]);
        for (int i = 1; i < hc.length; ++i) {
            mh = Math.max(mh, hc[i]-hc[i-1]);
        }
        for (int i = 1; i < vc.length; ++i) {
            mv = Math.max(mv, vc[i]-vc[i-1]);
        }
        return (int)((mh*mv)%1000000007);
    }
}
