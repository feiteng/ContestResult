class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        
Arrays.sort(horizontalCuts);
        int r = Math.max(horizontalCuts[0], h - horizontalCuts[horizontalCuts.length - 1]);
        for (int i = 1; i < horizontalCuts.length; i++) {
            r = Math.max(r, horizontalCuts[i] - horizontalCuts[i - 1]);
        }

        Arrays.sort(verticalCuts);
        
int c = Math.max(verticalCuts[0], w - verticalCuts[verticalCuts.length - 1]);
        for (int i = 1; i < verticalCuts.length; i++) {
            c = Math.max(c, verticalCuts[i] - verticalCuts[i - 1]);
        }

        return (int) (((long) r * c) % (1000000000 + 7));
    }
}
