class Solution {
    public int maxArea(int h, int w, int[] hcuts, int[] vcuts) {
        
        long max = 0;
        Arrays.sort(hcuts);
        Arrays.sort(vcuts);
       
        int diff1 = hcuts[0];
        for(int i =1;i<hcuts.length;i++){
            diff1 = Math.max(diff1, hcuts[i]-hcuts[i-1]);
        }
        diff1 = Math.max(diff1, h - hcuts[hcuts.length-1]);
        
        int diff2 = vcuts[0];
        for(int i =1;i<vcuts.length;i++){
            diff2 = Math.max(diff2, vcuts[i]-vcuts[i-1]);
        }
        diff2 = Math.max(diff2, w - vcuts[vcuts.length-1]);
        
        long area = (long)diff1 * diff2;
        return (int)(area % (int)(1e9+7));
        
    }
}
