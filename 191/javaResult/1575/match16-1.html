class Solution {
    public int maxArea(int h, int w, int[] hC, int[] vC) {
        Arrays.sort(hC);
        Arrays.sort(vC);
        long maxR = hC[0];
        long maxC = vC[0];
        for(int i=1; i<hC.length; i++){
            maxR = Math.max(maxR, hC[i]-hC[i-1]);
        }
        
for(int i=1; i<vC.length; i++){
            maxC = Math.max(maxC, vC[i]-vC[i-1]);
        }
        maxR = Math.max(maxR, h-hC[hC.length-1]);
        maxC = Math.max(maxC, w-vC[vC.length-1]);
        return (int)(maxR*maxC%1000000007);
    }
}
