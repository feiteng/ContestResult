class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        final int N = horizontalCuts.length;
        
final int M = verticalCuts.length;
        
        Arrays.sort(horizontalCuts);
        long bestHoriz = Math.max(horizontalCuts[0], h - horizontalCuts[N - 1]);
        for (int i = 0; i + 1 < N; ++i) {
            bestHoriz = Math.max(bestHoriz, horizontalCuts[i + 1] - horizontalCuts[i]);
        }
        
        Arrays.sort(verticalCuts);
        long bestVert = Math.max(verticalCuts[0], w - verticalCuts[M - 1]);
        
for (int i = 0; i + 1 < M; ++i) {
            bestVert = Math.max(bestVert, verticalCuts[i + 1] - verticalCuts[i]);
        }
        
        return (int)((bestHoriz * bestVert) % 1000000007);
    }
}
