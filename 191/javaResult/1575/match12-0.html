class Solution {
    public static final long MOD = 1000000007L;
    
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(verticalCuts);
        Arrays.sort(horizontalCuts);
        
        int wMax = 0, vLast = 0, hMax = 0, hLast = 0;
        
        for (int v: verticalCuts) {
            wMax = Math.max(wMax, v - vLast);
            vLast = v;
        }
        
        wMax = Math.max(wMax, w - vLast);
                
        for (int ho: horizontalCuts) {
            hMax = Math.max(hMax, ho - hLast);
            hLast = ho;
        }
        
        hMax = Math.max(hMax, h - hLast);
        
        return (int)((((long)wMax) * ((long)hMax)) % MOD);
    }
}
