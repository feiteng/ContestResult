class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        
int maxH = 0, curH = 0;
        for (int hc : horizontalCuts) {
            maxH = Math.max(maxH, hc-curH);
            curH = hc;
        }
        maxH = Math.max(maxH, h-curH);
        
        Arrays.sort(verticalCuts);
        int maxW = 0, curW = 0;
        
for (int wc : verticalCuts) {
            maxW = Math.max(maxW, wc-curW);
            curW = wc;
        }
        maxW = Math.max(maxW, w-curW);
        
        long ans = (long)maxH*(long)maxW;
        return (int)(ans % 1_000_000_007);
    }
}
