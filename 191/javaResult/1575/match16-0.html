class Solution {
    public int maxArea(int h, int w, int[] a, int[] b) {
        
        Arrays.sort(a);
        Arrays.sort(b);
        int maxH = a[0];
        int maxW = b[0];
        
        for(int i=1; i<a.length; i++) {
            maxH = Math.max(maxH, a[i]-a[i-1]);
        }
        
        maxH = Math.max(maxH, h-a[a.length-1]);
        
        for(int i=1; i<b.length; i++) {
            maxW = Math.max(maxW, (b[i]-b[i-1]));
        }
        maxW = Math.max(maxW, w-b[b.length-1]);
        
        
        return (int)((maxH*1L*maxW)%((int)1e9+7));
    }
}
