class Solution {
    public int maxProduct(int[] nums) {
        int out = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                out = Math.max(out, (nums[i] - 1) * (nums[j] - 1));
            }
        }
        return out;
    }
}
