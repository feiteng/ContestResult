class Solution {
    public int maxProduct(int[] nums) {
        int max = Integer.MIN_VALUE;
        for(int idx1=0;idx1<nums.length;idx1++) {
            for(int idx2=idx1+1;idx2<nums.length;idx2++) {
                max = Math.max(max, (nums[idx1]-1)*(nums[idx2]-1));
            }
        }
        return max;
    }
}
