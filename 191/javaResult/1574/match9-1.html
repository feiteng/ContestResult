class Solution {
    public int maxProduct(int[] nums) {
        int max = 0;
        int secMax = 0;
        for (int i = 0; i < nums.length; i += 1) {
            if (max < nums[i]) {
                secMax = max;
                max = nums[i];
            }
            else if (secMax < nums[i]) {
                secMax = nums[i];
            }
        }
        
        return (max - 1) * (secMax - 1);
    }
}
