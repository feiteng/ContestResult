class Solution {
    public int maxProduct(int[] nums) {
        int mx = 0;
        for(int i=0;i<nums.length;++i){
            for(int j=i+1;j<nums.length;++j){
                int v =(nums[i]-1)*(nums[j]-1);
                mx = Math.max(mx,v);
            }
        }
        return mx;
    }
}
