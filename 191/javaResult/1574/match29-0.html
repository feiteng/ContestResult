class Solution {
    public int maxProduct(int[] nums) {
        Arrays.sort(nums);
        // for (int n : nums) System.out.println(n);
        int l = nums.length;
        int a = (nums[0] - 1)*(nums[1] - 1);
        int b = (nums[l - 1] - 1)*(nums[l - 2] - 1);
        return Math.max(a, b);
    }
}
