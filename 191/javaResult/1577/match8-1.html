class Solution {

    private double total = 0;
    private double valid = 0;
    private int[] cnt1, cnt2;
    
private Set<String> seen;

    public double getProbability(int[] balls) {
        int sum = 0;
        for(int ball: balls){
            sum += ball;
        }
        seen = new HashSet<>();
        int n = balls.length;
        cnt1 = new int[n];
        cnt2 = new int[n];
        
dfs(balls, 0, 0, sum / 2);
        return valid / total;
    }

    private void dfs(int[] ball, int cur, int sum, int target){
        if(sum == target){
            handle(ball, target);
            return;
        }
        if(cur== ball.length){
            return;
        }
        for(int i = 0 ; i <= Math.min(ball[cur], target - sum); i++){
            cnt1[cur] = i;
            dfs(ball, cur + 1, sum + i, target);
            cnt1[cur] = 0;
        
}
    }

    private void handle(int[] ball, int target){
        int diff1 = 0, diff2 = 0;
        for(int i = 0 ; i < ball.length; i++){
            cnt2[i] = ball[i] - cnt1[i];
            if(cnt1[i] > 0){
                diff1++;
            }
            if(cnt2[i] > 0){
                diff2++;
            }
        }
        double num = 1;
        int tmp = target;
        for(int i: cnt1){
            num *= count(i, tmp);
            tmp -= i;
        }
        tmp = target;
        for(int i: cnt2){
            num *= count(i, tmp);
            tmp -= i;
        }
        total += num;
        if(diff1 == diff2){
            valid += num;
        }
    }

    private double count(int a, int b){
        if(a == 0){
            return 1;
        }
        
double mul1 = 1, mul2 = 1;
        for(int i = 0; i < a ; i++){
            mul1 *= a - i;
            mul2 *= b - i;
        }
        return mul2 / mul1;
    }
}
