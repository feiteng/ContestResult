class Solution {
    private double res = 0;
    private double total = 0;
    
    public double getProbability(int[] balls) {
        int n = balls.length;
        dfs(balls, new int[n], 0);
        // System.out.println("res=" + res);
        // System.out.println("total=" + total);
        return res * 1.0 / total;
    }
    
    private void dfs(int[] balls, int[] nb, int idx) {
        if (idx == balls.length) {
            // System.out.println("idx=" + idx);
            // print(nb);
            // check if the split is equal and the color is equal
            check(balls, nb);
            
return;
        }
        int n = balls.length;
        for (int i = 0; i <= balls[idx]; i++) {
            // System.out.println("idx=" + idx);
            // System.out.println("balls[idx]=" + balls[idx]);
            // System.out.println("i=" + i);
            nb[idx] = i;
            dfs(balls, nb, idx + 1);
        }
        // restore 0
        nb[idx] = 0;
    
}
    
    private void check(int[] balls, int[] nb) {
        int n = balls.length;
        // get reverse
        int[] revb = new int[n];
        for (int i = 0; i < n; i++) {
            revb[i] = balls[i] - nb[i];
        }
        // check the split is equal
        double pos = p(sum(revb)) * p(sum(nb));
        for (int i = 0; i < n; i++) {
            pos /= p(revb[i]);
            pos /= p(nb[i]);
        }
        
        if (sum(revb) == sum(nb)) {
            total += pos;
            // print(revb);
            // System.out.println("pos=" + pos);
            // System.out.println("sum(revb)=" + sum(revb));
            // System.out.println("p(sum(revb))=" + p(sum(revb)));
            // System.out.println("p(sum(nb))=" + p(sum(nb)));
            if (color(revb) == color(nb)) res += pos;
        }
        // if (revb[0] == 2 && revb[1] == 1 && revb[2] == 0) {
        //     System.out.println("pos=" + pos);
        // }
    }
    
    private double p(double i) {
        if (i == 0 || i == 1) return 1;
        return i * p(i - 1);
    
}
    
    private int sum(int[] balls) {
        int sum = 0;
        for (int b : balls) sum += b;
        return sum;
    
}
    
    private int color(int[] balls) {
        int col = 0;
        for (int b : balls) {
            if (b != 0) col++;
        }
        return col;
    }
    
    private void print(int[] nb) {
        System.out.println("print nb!");
        for (int i = 0; i < nb.length; i++) {
            System.out.println("i=" + i);
            System.out.println("nb[i]=" + nb[i]);
        }
    }
}
