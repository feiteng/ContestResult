class Solution {
    double total = 0, valid = 0;
    public double getProbability(int[] balls) {
        total = 0;
        valid = 0;
        total = getTotal(balls);
        int cnt = 0;
        for(int each: balls)
            cnt += each;

        helper(balls, new int[balls.length], 0, cnt / 2, 0);
        
return valid / total;
    }

    private void helper(int[] balls, int[] curr, int cnt, int target, int idx) {
        if(cnt == target) {
            int[] another = new int[curr.length];
            for(int i = 0; i < balls.length; i++) {
                another[i] = balls[i] - curr[i];
            }

            if(isValid(another, curr)) {
                /*System.out.println("valid...");

                for(int i = 0; i < curr.length; i++) {
                    System.out.print(curr[i] + " ");
                }
                System.out.println();

                for(int i = 0; i < another.length; i++) {
                    System.out.print(another[i] + " ");
                }
                System.out.println();

                System.out.println("total case: " + (getTotal(curr) * getTotal(another)));*/
                valid = valid + getTotal(curr) * 
getTotal(another);
            }
            return;
        }

        if(cnt > target || idx == balls.length)
            return;

        for(int i = 0; i <= balls[idx]; i++) {
            curr[idx] = i;
            helper(balls, curr, cnt + i, target, idx+1);
            curr[idx] = 0;
        
}
    }

    private boolean isValid(int[] another, int[] curr) {
        int a = 0, b = 0;
        for(int i = 0; i < curr.length; i++) {
            if(curr[i] != 0)
                a++;
            
if(another[i] != 0) {
                b++;
            }
        }
        return a == b;
    }

    public double getTotal(int[] balls) {
        double ans = 1;
        int total = 0;
        for(int each: balls)
            total += each;

        for(int ball: balls) {
            if(ball == 0)
                continue;
            ans = ans * combine(total, ball);
            total = total - ball;
        }

        return ans;
    
}

    private double combine(int m, int n) {
        double ans = 1;
        for(int i = 0; i < n; i++) {
            ans = ans * (m - i);
        }

        for(int i = 1; i <= n; i++) {
            ans = ans / i;
        }
        return ans;
    }
}
