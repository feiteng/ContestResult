class Solution {
    double p1 = 0.0, p0 = 0.0;
    public double getProbability(int[] balls) {
        int k = balls.length;
        int n = 0, sum = 0;
        for(int b: balls){
            sum += b;
        }
        n = sum / 2;
        
        int[] plan = new int[k];
        gen(balls, plan, 0, 0, n, k);
        
return p1/(p1 + p0);
    }
    
    public void gen(int[] balls, int[] plan, int ind, int sum, int n, int k){
        if(ind == k){// time to calculate
            if(sum != n)return;
            int[] rp = new int[k];
            for(int i = 0; i < k; i++)rp[i] = balls[i] - plan[i];
            double v = weight(plan, rp);
            if(check(plan, rp)){
                p1 += v;
            } else {
                p0 += v;
            }
        } else{//gen next 
            int rem = n - sum;
            if(rem > balls[ind])rem = balls[ind]
;
            for(int next = rem; next >= 0; next--){
                plan[ind] = next;
                gen(balls, plan, ind + 1, sum + next, n, k);
                plan[ind] = 0;
            }
        }
    }
    
    public int C(int x){
        int base = 1;
        for(int i = 1; i <= x; i++)base *= i;
         //System.out.println(base +" " + x);
        return base;
    
}
    
    public boolean check(int[] p, int[] rp){
        int c1 = 0, c2 = 0;
        for(int i = 0; i < p.length; i++){
            if(p[i] > 0)c1++;
            if(rp[i] > 0)c2++;
        
}
        return c1 == c2;
    }
    
    public double weight(int[] p, int[] rp){
        double base = 1.0;
        for(int i = 0; i < p.length; i++){
            if(p[i] > 0) base /= (double)C(p[i]);
            if(rp[i] > 0) base /= (double)C(rp[i]);
        }
        //System.out.println(base +" " + Arrays.toString(p) +" & " + Arrays.toString(rp));
        return base;
    }
}
