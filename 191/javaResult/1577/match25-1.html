class Solution {
    double[] factors;
    double possible;
    public double getProbability(int[] balls) {
        factors = new double[49];
        possible = 0.0;
        
for (int i = 0; i < factors.length; ++i) {
            if (i == 0) {
                factors[0] = 1.0;
            } else {
                factors[i] = factors[i - 1] * i;
            }
        }
        
        double total = calculate(balls);
        int[] left = new int[balls.length];
        int[] right = new int[balls.length];
        solve(left, right, balls, 0);
        return possible / total;
    }
    
    private void solve(int[] left, int[] right, int[] balls, int idx) {
        if (idx == balls.length) {
            if (isSame(left, right)) {
                double res = calculate(left) * calculate(right);
                possible += res;
            }
            return;
        }
        for (int i = 0; i <= balls[idx]; ++i) {
            left[idx] = i;
            right[idx] = balls[idx] - i;
            solve(left, right, balls, idx + 1);
        
}
        return;
    }
    
    private boolean isSame(int[] l, int[] r) {
        int countl = 0;
        int countr = 0;
        int totall = 0;
        int totalr = 0;
        for (int i = 0; i < l.length; ++i) {
            totall += l[i];
            totalr += r[i];
            if (l[i] != 0) {
                ++countl;
            }
            if (r[i] != 0) {
                ++countr;
            }
        }
        return (countl == countr) && (totall == totalr);
    }
    
    private double calculate(int[] array) {
        double divisor = 1.0;
        int total = 0;
        
for (int i = 0; i < array.length; ++i) {
            total += array[i];
            divisor *= factors[array[i]];
        }
        return factors[total] / divisor;
    }
}
