class Solution {
    public double getProbability(int[] balls) {
        long all = solve(balls, 0, 0, 0, 0, 0);
        // System.out.println("fuck");
        long ans = some(balls, 0, 0, 0, 0, 0);
        // System.out.println(all);
        // System.out.println(ans);
        return ((1.0)*ans/all);
    }
    
    long nCr(long n, long r) { 
        return fact(n) / (fact(r) * fact(n - r)); 
    } 
  
    
    long fact(long n) { 
        long res = 1; 
        for (int i = 2; i <= n; i++) 
            res = res * i; 
        return res; 
    
} 
    
    public long solve(int[] b, int pos, int f, int s, int disF, int disS) {
        if (pos == b.length) {
            
            // System.out.println(f + " " + s + " " + disF + " " + disS);
            if (f == s) return 1;
            return 0;
        }
        long answer = solve(b, pos+1, f, s + b[pos], disF, disS+1);
        answer += solve(b, pos+1, f+ b[pos], s , disF+1, disS);
        for (int i = 1; i < b[pos]; ++i) {
            answer += (solve(b, pos+1, f+i, s + b[pos]-i, disF+1, disS+1)*
nCr(b[pos], i));
        }
        return answer;
        
    }
    public long some(int[] b, int pos, int f, int s, int disF, int disS) {
        if (pos == b.length) {
            // System.out.println(f + " " + s + " " + disF + " " + disS);
            if (f == s && disF == disS) return 1;
            return 0;
        }
        
long answer = some(b, pos+1, f, s + b[pos], disF, disS+1);
        answer += some(b, pos+1, f+ b[pos], s , disF+1, disS);
        for (int i = 1; i < b[pos]; ++i) {
            answer += (some(b, pos+1, f+i, s + b[pos]-i, disF+1, disS+1)*nCr(b[pos], i));
        }
        return answer;
        
    }
}
