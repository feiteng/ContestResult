class Solution {
    double res1 = 0, res2 = 0;
    int n = 0;
    public double getProbability(int[] balls) {
        int len = balls.length;
        for (int ball : balls) {
            n += ball;
        }
        n /= 2;
        helper(balls, new int[len], new int[len], 0, 1.0);
        return res1 / res2;
    }
    private void helper(int[] balls, int[] box1, int[] box2, int idx, double pro) {
        if (idx == balls.length) {
            if (check1(box1, box2)) {
                if (check(box1, box2)) res1 += pro;
                res2 += pro;
            }
            return;
        }
        for (int i = 0;i <= balls[idx];i++) {
            box1[idx] += i;
            box2[idx] += (balls[idx] - i);
            helper(balls, box1, box2, idx + 1, pro * Math.pow(0.5, i) * Math.pow(0.5, balls[idx] - i) * get(balls[idx], i));;
            box1[idx] -= i;
            box2[idx] -= (balls[idx] - i);
        }
    }
    private int get(int n, int i) {
        if (i == 0 || n == i) return 1;
        int res1 = 1, res2 = 1;
        for(int j = 0;j < i;j++) {
            res1 *= (n - j);
            res2 *= (j + 1);
        }
        return res1 / res2;
    }
    
    private boolean check(int[] box1, int[] box2) {
        int cnt1 = 0, cnt2 = 0;
        for (int box : box1) {
            cnt1 = box == 0 ? cnt1 : (cnt1 + 1);
        }
        for (int box : box2) {
            cnt2 = box == 0 ? cnt2 : (cnt2 + 1);
        }
        return cnt1 == cnt2;
    }
    private boolean check1(int[] box1, int[] box2) {
        int cnt1 = 0;
        for (int box : box1) {
            cnt1 += box;
        }
        return cnt1 == n;
    }
}
