class Solution {
    double total = 0;
    
double same = 0;
    public double getProbability(int[] balls) {
        int n = 0, m = balls.length;
        for (int b : balls) n += b;
        n /= 2;

        aux(0, balls, n, new int[m]);
        
        // System.out.println(same + " " + total);
        return (double)same/(double)total;
    }
    
    private void aux(int cur, int[] balls, int remain, int[] dis) {
        if (remain < 0) return;
        if (remain == 0) {
            // System.out.println(Arrays.toString(dis));
            calc(dis, balls);
            return;
        }
        if (cur >= balls.length) return;
        for (int i = 0; i <= balls[cur] && i <= remain; i++) {
            dis[cur] = i;
            aux(cur+1, balls, remain-i, dis);
            dis[cur] = 0;
        
}
    }
    
    private void calc(int[] dis, int[] balls) {
        int c1 = 0, c2 = 0;
        int[] dis2 = new int[dis.length];
        for (int i = 0; i < balls.length; i++) {
            if (dis[i] > 0) c1++;
            dis2[i] = balls[i] - dis[i];
            if (dis2[i] > 0) c2++;
        }
        
        double way1 = findWay(dis), way2 = findWay(dis2);
        double way = way1*way2;
        
        total += way;
        if (c1 == c2) same += way;
    }
    
    private double findWay(int[] dis) {
        int cur = 0;
        double way = 1;
        for (int d : dis) {
            cur += d;
            // System.out.println(way + " " + comb(cur, d));
            way *= comb(cur, d);
            // if (way < 0) System.out.println(Arrays.toString(dis));
        
}
        return way;
    }
    
    private double comb(int n, int r) { 
        return fact(n) / (
fact(r) * fact(n - r)); 
    } 

    private double fact(int n)  { 
        double res = 1; 
        for (int i = 2; i <= n; i++) 
            res = res * i; 
        return res; 
    } 
}
