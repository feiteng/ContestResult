class Solution {
    double numerator = 0.0;
    
double denominator = 0.0;
    public double getProbability(int[] balls) {
        int total = 0;
        for (int b : balls) {
            total += b;
        }
        
        dfs(balls, 0, 0, new int[balls.length], total / 2);    
        
        
return numerator / denominator;
    }
    
    private void dfs(int[] balls, int start, int num, int[] set, int total) {
        if (start == balls.length) {
            if (num != total) {
                return;
            }
        }
        if (num == total) {
            double count = 1.0;
            for (int i = 0; i < set.length; i++) {
                count *= comb(balls[i], set[i]);
            }
            if (valid(set, balls)) {
                numerator += count;
            }
            denominator += count;
            return;
        }
        
        for (int i = 0; i <= balls[start]; i++) {
            if (num + i <= total) {
                int tmp = set[start];
                set[start] += i;
                dfs(balls, start + 1, num + i, set, total);
                set[start] = tmp;
            }
        
}
    }
    
    private boolean valid(int[] set, int[] balls) {
        int[] rest = new int[set.length];
        int counts = 0;
        int countr = 0;
        for (int i = 0; i < balls.length; i++) {
            rest[i] = balls[i] - set[i];
            
            if (rest[i] != 0) {
                countr++;
            }
            
            
if (set[i] != 0) {
                counts++;
            }
        }
        return counts == countr;
    }
    
    private long comb(int n, int k) {
        return perm(n) / (perm(k) * perm(n - k));
    }
    
    private long perm(int n) {
        if (n < 2) {
            return 1l;
        }
        return n * perm(n - 1);
    }
}
