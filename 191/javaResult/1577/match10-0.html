class Solution {
    long count = 0;
    long[][] f = new long[10][10];
    long p = 0;
    public double getProbability(int[] balls) {
        int sum = 0;
        for (int b : balls) {
            sum += b;
        }
        
        for (int i = 0; i < 10; i++) {
            for (int j = 0; j <= i; j++) {
                if (j == 0) f[i][j] = 1;
                else f[i][j] = f[i - 1][j - 1] + f[i - 1][j];
            }
        }
        p = per(sum / 2, sum / 2);
        long all = per(sum, sum / 2);
        // System.out.println(all);
        List<Integer> list = new ArrayList<>();
        helper(balls, list, 0, sum / 2);
        
        return (double) count / all;
    }
    
    private void helper(int[] b, List<Integer> list, int index, int k) {
        if (k < 0) return;
        if (index == b.length) {
            if (k == 0) {
                if (same(b, list)) {
                    // System.out.println(list.toString());
                    count += combo(b, list);
                }
            }
            return;
        }
        
        for (int i = 0; i <= b[index]; i++) {
            list.add(i);
            helper(b, list, index + 1, k - i);
            list.remove(list.size() - 1);
        }
    }
    
    private boolean same(int[] b, List<Integer> list) {
        
        int a = 0, c = 0;
        for (int i = 0; i < b.length; i++) {
            int num = i >= list.size() ? 0 : list.get(i);
            if (num > 0) a++;
            if (b[i] - num > 0) c++;
        }
        return a == c;
    }
    
    private long combo(int[] b, List<Integer> list) {
        long res = 1;
        for (int i = 0; i < b.length; i++) {
            int num = i >= list.size() ? 0 : list.get(i);
            res *= f[b[i]][num];
        }
        return res;
    }
    
       
    private long per(int n, int m) {
        long res = 1;
        for (int i = 1; i <= m; i++) {
            res *= n - i + 1;
            res /= i;
        }
        return res;
    }
}
