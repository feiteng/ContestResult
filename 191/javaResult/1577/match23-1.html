class Solution {
    public double getProbability(int[] balls) {
        int k = balls.length;
        int n = Arrays.stream(balls).sum();
        return (double) dfs(balls, k, n, 0, 0, 0, 0, 0, 1) / 
choose(n, n / 2);
    }

    private long dfs(int[] balls, int k, int n, int next, int aDistinct, int bDistinct, int aSum, int bSum, long product) {
        if (aSum > n / 2 || bSum > n / 2) {
            return 0;
        }
        if (k == next) {
            return (aDistinct == bDistinct) ? product : 0;
        }
        long out = 0;
        for (int i = 0; i <= balls[next]; i++) {
            out += dfs(balls, k, n, next + 1, aDistinct + (i > 0 ? 1 : 0), bDistinct + (i < balls[next] ? 1 : 0), aSum + i, bSum + balls[next] - i, product * 
choose(balls[next], i));
        }
        return out;
    }

    private static long choose(int a, int b) {
        b = Math.min(b, a - b);
        
long out = 1;
        for (int i = 0; i < b; i++) {
            out *= a - i;
            out /= 1 + i;
        }
        return out;
    }
}
