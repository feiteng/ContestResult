class Solution {
    public double getProbability(int[] balls) {
        int len = balls.length;

        double total = 1;
        
        int num = 0;
        for (int i = 0; i < balls.length; i += 1) {
            num += balls[i];
        }
        
        total = factorial(num);
        
        for (int i = 0; i < balls.length; i += 1) {
            double crt = factorial(balls[i]);
            total /= crt;
        }
        
        double sum = helper(balls, 0, new int[len], num, 0, factorial(num / 2));
        return sum / total;
    }
    
    public double factorial(int n) {
        
double sum = 1;
        for (int i = 1; i <= n; i += 1){
            sum *= i;
        }
        return sum;
    }
    public double helper(int[] balls, int k, int[] chosen, int total, int left_total, double fac) {
        if (k == balls.length) {
            if (left_total * 2 != total) {
                return 0;
            }
            
int left = 0;
            int right = 0;
            for (int i = 0; i < chosen.length; i += 1) {
                if (chosen[i] < balls[i]) {
                    right += 1;
                }
                if (chosen[i] > 0) {
                    left += 1;
                }
            }
            
            if (left != right) {
                return 0;
            }
            
            double sum = 1;
            
            for (int i = 0; i < chosen.length; i += 1) {
                double crt = factorial(chosen[i]);
                double crt_2 = factorial(balls[i] -  chosen[i]);
                sum *= crt * crt_2;
            }
            
            return fac * fac / sum;
        }
        
        double sum = 0;
        for (int i = 0; i <= balls[k]; i += 1) {
            chosen[k] = i;
            sum += helper(balls, k + 1, chosen, total, left_total + i, fac);
        }
        
        return sum;
    }
}
