class Solution {
    double[] A = new double[49];
    int n = 0;
    double res = 0;
    public double getProbability(int[] balls) {
        A[0] = 1.0;
        for(int i=1; i<A.length; i++){
            A[i] = A[i-1]*i;
        }
        for(int count : balls){
            n += count;
        }
        calc(balls, 0, new int[balls.length], n/2);
        double denom = A[n];
        for(int num : balls){
            denom /= A[num];
        }
        // System.out.println(res +"," + denom);
        return res / denom;
    }
    
    public void calc(int[] balls, int index, int[] a, int m){
        if(m<0){
            return;
        }
        
        if(m==0){
            int colors = 0;
            int otherColors = 0;
            for(int i=0; i<a.length; i++){
                if(a[i]>0){
                    colors++;
                }
                if(balls[i]-a[i]>0){
                    otherColors++;
                }
            }
            if(otherColors==colors){
                double curRes = 0.0;
                double aCount = A[n/2];
                for(int i=0; i<a.length; i++){
                    aCount /= A[a[i]];
                }
                double bCount = A[n/2];
                for(int i=0; i<a.length; i++){
                    bCount /= A[balls[i]-a[i]];
                }
                curRes = aCount*bCount;
                // System.out.println(aCount+","+bCount);
                res+=curRes;
            }
            return;
        }
        
        if(index==balls.length){
            return;
        }
        
        for(int take=0; take<=balls[index]; take++){
            a[index] += take;
            calc(balls, index+1, a, m-take);
            a[index] -= take;
        }  
    }
}
