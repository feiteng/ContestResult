class Solution {
    double p = 0.0;
    public double getProbability(int[] balls) {
        int n = 0;
        for (int num : balls) {
            n += num;
        }
        n /= 2;
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer, Integer> map2 = new HashMap<>();
        bt(map1, map2, 0, 0, balls, 0, n);
        return p;
    }
    
    private void bt(Map<Integer, Integer> map1, Map<Integer, Integer> map2, int sum1, int sum2, int[] balls, int idx, int n) {
        if (idx == balls.length) {
            if (isValid(map1, map2)) {
                p += getP(map1, map2, balls, n);
            }
            return;
        }
        if (sum1 > n) return;
        for (int i = 0; i <= balls[idx]; i++) {
            map1.put(idx, i);
            map2.put(idx, balls[idx] - i);
            bt(map1, map2, sum1 + i, sum2 + balls[idx] - i, balls, idx + 1, n);
        }
    }
    
    private boolean isValid(Map<Integer, Integer> map1, Map<Integer, Integer> map2) {
        int cnt1 = 0, cnt2 = 0, sum1 = 0, sum2 = 0;
        for (int c : map1.values()) {
            if (c > 0) {
                cnt1++;
                sum1 += c;
            }
        }
        for (int c : map2.values()) {
            if (c > 0) {
                cnt2++;
                sum2 += c;
            }
        }
        return cnt1 == cnt2 && sum1 == sum2;
    }
    
    private double getP(Map<Integer, Integer> map1, Map<Integer, Integer> map2, int[] balls, int n) {
        double curr = 1;
        for (int i = 0; i < balls.length; i++) {
            if (map1.get(i) > 0) {
                curr *= C(balls[i], map1.get(i));
            }
        }
        return curr * A(n, n) / A(n * 2, n);
    }
    
    private double C(int n, int k) {
        return A(n, k) / A(k, k);
    }
    
    private double A(int n, int k) {
        double res = 1.0;
        for (int i = n; i > n - k; i--) {
            res *= i;
        }
        return res;
    }
}
