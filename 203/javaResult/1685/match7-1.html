class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n=stoneValue.length;
        int[] sum=new int[n+1];
        int[][] dp=new int[n][n];
        
for(int i=0;i<n;i++){
            sum[i+1]=sum[i]+stoneValue[i];
        }
        
        for(int i=1;i<n;i++){
            for(int j=0;j+i<n;j++){
                for(int k=j;k<j+i;k++){
                    int lSum=sum[k+1]-sum[j], rSum=sum[j+i+1]-sum[k+1];
                    
if(lSum>rSum){
                        dp[j][j+i]=Math.max(dp[j][j+i],rSum+dp[k+1][j+i]);
                    }else if(lSum<rSum){
                        dp[j][j+i]=Math.max(dp[j][j+i],lSum+dp[j][k]);
                    }else{
                        dp[j][j+i]=Math.max(dp[j][j+i],rSum+dp[k+1][j+i]);
                        dp[j][j+i]=
Math.max(dp[j][j+i],lSum+dp[j][k]);
                    }
                }
            }
        }
        
        return dp[0][n-1];
    }
}
