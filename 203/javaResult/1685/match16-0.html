class Solution {
    int INF = Integer.MAX_VALUE;
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] a = new int[n + 1];
        for(int i = 1; i <= n; i++){
            a[i] = a[i - 1] + stoneValue[i - 1];
        }
        int[][] dp = new int[n + 1][n + 1];
        for(int len = 2; len <= n; len++){
            for(int i = 1; i + len - 1 <= n; i++){
                int l = i; int r = i + len - 1;
                //dp[l][r] = INF;
                for(int k = l; k < r; k ++){
                    //dp[l][r] = Math.max(dp[l][r], Math.min(dp[l][k], dp[k + 1][r]) + Math.min(a[k] - a[l - 1], a[r] - a[k]));
                    int left = a[k] - a[l - 1];
                    int right = a[r] - a[k];
                    if(left > right){
                        dp[l][r] = Math.max(dp[l][r], dp[k + 1][r] + right);
                    }else if(left < right){
                        dp[l][r] = Math.max(dp[l][r], dp[l][k] + left);
                    }else{
                        int left_sum = Math.max(dp[l][r], dp[k + 1][r] + right);
                        int right_sum = Math.max(dp[l][r], dp[l][k] + left);
                        /*if(r - k - 1 > k - l){
                            dp[l][r] = Math.max(dp[l][r], dp[k + 1][r] + right);
                        }else if(r - k - 1 < k - l){
                            dp[l][r] = Math.max(dp[l][r], dp[l][k] + left);
                        }else{ 
                            dp[l][r] = Math.max(dp[l][r], Math.max(left_sum, right_sum));
                        }*/
                        dp[l][r] = Math.max(dp[l][r], Math.max(left_sum, right_sum));
                    }
                }
            }
        }
        return dp[1][n];
    }
}
