class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] sum = new int[n];
        sum[0] = stoneValue[0];
        for (int i = 1; i < stoneValue.length; i++) {
            sum[i] = sum[i - 1] + stoneValue[i];
        }        
        int[][] dp = new int[n][n];
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i + 1; j < n; j++) {
                for (int k = i; k < j; k++) {
                    int leftSum = sum[k] - sum[i] + stoneValue[i];
                    int rightSum = sum[j] - sum[k];
                    if (leftSum < rightSum) {
                        dp[i][j] = Math.max(leftSum + dp[i][k], dp[i][j]);
                    } else if (leftSum == rightSum) {
                        dp[i][j] = Math.max(leftSum + Math.max(dp[i][k], dp[k + 1][j]), dp[i][j]);
                    } else {
                        dp[i][j] = Math.max(rightSum + dp[k + 1][j], dp[i][j]);
                    }
                }
            }
        }
        
        return dp[0][n - 1];
    }
}
