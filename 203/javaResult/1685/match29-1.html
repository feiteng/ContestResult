class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] sums = new int[n];
        sums[0] = stoneValue[0];
        for (int i = 1; i < n; i++) {
            sums[i] = sums[i - 1] + stoneValue[i];
        }

        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dp[i], -1);
            dp[i][i] = 0;
        }

        
for (int d = 1; d < n; d++) {
            for (int i = 0; i < n - d; i++) {
                for (int k = 0; k < d; k++) {
                    int left = sums[i + k] - sums[i] + stoneValue[i];
                    int right = sums[i + d] - sums[i + k];
                    if (left < right) {
                        dp[i][i + d] = Math.max(dp[i][i + d], left + dp[i][i + k]);
                    } else if (left > right) {
                        dp[i][i + d] = Math.max(dp[i][i + d], right + dp[i + k + 1][i + d]);
                    } else {
                        dp[i][i + d] = Math.max(dp[i][i + d], left + dp[i][i + k]);
                        
dp[i][i + d] = Math.max(dp[i][i + d], right + dp[i + k + 1][i + d]);
                    }
                }
            }
        }
        return dp[0][n - 1];
    }
}
