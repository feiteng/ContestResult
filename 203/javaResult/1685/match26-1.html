class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] sums = new int[n][n];
        for (int j = 0; j < n; j++) {
            int sum = 0;
            for (int k = j; k < n; k++) {
                sum += stoneValue[k];
                sums[j][k] = sum;
            }
        }
        int[][] memo = new int[n][n];
        return help(0, n - 1, sums, memo);
    
}
    
    public int help(int lo, int hi, int[][] sums, int[][] memo) {
        if (lo == hi) {
            return 0;
        }
        if (memo[lo][hi] > 0) {
            return memo[lo][hi] - 1;
        }
        int ret = 0;
        for (int j = lo; j < hi; j++) {
            int left = sums[lo][j];
            int right = sums[j+1][hi];
            if (left < right) {
                ret = Math.max(ret, left + help(lo, j, sums, memo));
            } else if (right < left) {
                ret = Math.max(ret, right + help(j + 1, hi, sums, memo));
            } else {
                ret = Math.max(ret, Math.max(left + help(lo, j, sums, memo), right + help(j + 1, hi, sums, memo)));
            }
        }
        memo[lo][hi] = ret + 1;
        return ret;
    }
}
