class Solution {
    
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] dp = new int[n][n];
        int[] sum = new int[n + 1];
        for(int i = 1; i <= n; i++){
            sum[i] = sum[i - 1] + stoneValue[i - 1];
        }
        for(int i = 0; i < n; i++){
            dp[i][i] = 0;
        }
        for(int len = 1; len < n; len++){
            for(int i = 0; i + len < n; i++){
                int j = i + len;
                int max = 0;
                for(int k = i; k < j; k++){
                    int left = sum[k + 1] - sum[i];
                    int right = sum[j + 1] - sum[k + 1];
                    if(left > right){
                        max = Math.max(max, right + dp[k + 1][j]);
                    }else if(left < right){
                        max = Math.max(max, left + dp[i][k]);
                    }else{
                        max = Math.max(max, Math.max(left + dp[i][k], right + dp[k + 1][j]));
                    }
                }
                dp[i][j] = max;
            }
        }
        
        
        
        
        return dp[0][n - 1];
        
        
    }
}
