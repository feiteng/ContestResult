class Solution {
    public int stoneGameV(int[] s) {
        int n = s.length;
        int[] sum = new int[n];
        int[][] dp = new int[n][n];
        sum[0] = s[0];
        for (int i = 1; i < n; i++) {
            sum[i] = sum[i - 1] + s[i];
        }
        for (int len = 1; len < n; len++) {
            for (int l = 0; l + len < n; l++) {
                int sumL = s[l], r = l + len, sumR = sum[r] - sum[l];
                for (int m = l + 1; m <= r; m++) {
                    if (sumL > sumR) {
                        dp[l][r] = Math.max(dp[l][r], dp[m][r] + sumR);
                    } else if (sumL < sumR) {
                        dp[l][r] = Math.max(dp[l][r], dp[l][m - 1] + sumL);
                    } else {
                        dp[l][r] = Math.max(dp[l][r], Math.max(dp[l][m - 1], dp[m][r]) + sumL);
                    }
                    sumL += s[m];
                    sumR -= s[m];
                }
            }
        }
        return dp[0][n - 1];
    }
}
