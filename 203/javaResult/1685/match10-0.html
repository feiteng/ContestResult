class Solution {
    public int stoneGameV(int[] stoneValue) {
        int l = stoneValue.length;
        int[] pre = new int[l+1];
        for (int i = 0; i < l; i++) {
            pre[i+1] = pre[i] + stoneValue[i];
        }
        
        int[][] dp = new int[l][l];
        for (int len = 2; len <= l; len++) {
            for (int i = 0; i + len - 1 < l; i++) {
                int j = i + len - 1, sum = 0;
                for (int k = i; k < j; k++) {
                    int left = pre[k+1] - pre[i], right = pre[j+1] - pre[k+1], val = 0;
                    if (left < right) {
                        val = left + dp[i][k];
                    } else if (left > right) {
                        val = right + dp[k+1][j];
                    } else {
                        val = left + Math.max(dp[i][k], dp[k+1][j]);
                    }
                    dp[i][j] = Math.max(dp[i][j], val);
                }
            }
        }
        return dp[0][l-1];
    }
}
