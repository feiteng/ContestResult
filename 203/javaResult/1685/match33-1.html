class Solution {
    public int stoneGameV(int[] stoneValue) {
        
int[] prefixSum = new int[stoneValue.length + 1];
        for (int i = 0; i < stoneValue.length; i++) {
            prefixSum[i + 1] = prefixSum[i] + stoneValue[i]; 
        }
        Integer[][] dp = new Integer[stoneValue.length][stoneValue.length];
        return helper(prefixSum, 0, stoneValue.length - 1, dp);
    }
    
    private int helper(int[] prefixSum, int left, int right, Integer[][] dp) 
{
        if (left == right) {
            return 0;
        }
        if (dp[left][right] != null) {
            return dp[left][right];
        }
        int result = 0;
        for (int i = left; i < right; i++) {
            
if (prefixSum[i + 1] - prefixSum[left] > prefixSum[right + 1] - prefixSum[i + 1]) {
                result = Math.max(result, prefixSum[right + 1] - prefixSum[i + 1] + helper(prefixSum, i + 1, right, dp));
            } else if (prefixSum[i + 1] - prefixSum[left] < prefixSum[right + 1] - prefixSum[i + 1]) {
                result = Math.max(result, prefixSum[i + 1] - prefixSum[left] + helper(prefixSum, left, i, dp));
            } else {
                result = Math.max(result, Math.max(prefixSum[right + 1] - prefixSum[i + 1] + 
helper(prefixSum, i + 1, right, dp), prefixSum[i + 1] - prefixSum[left] + helper(prefixSum, left, i, dp)));
            }
        }
        dp[left][right] = result;
        return result;
    }
}
