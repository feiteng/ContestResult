class Solution {
    public int stoneGameV(int[] stoneValue) {
        int[][] dp = new int[stoneValue.length][stoneValue.length];
        for(int i = 0; i < dp.length; i++) {
            Arrays.fill(dp[i], -1);
        }
        return helper(dp, stoneValue, 0, stoneValue.length - 1);
    }
    
    int helper(int[][] dp, int[] stoneVal, int l, int r) {
        if(r == l) {
            return 0;
        }
        if(dp[l][r] != -1) {
            return dp[l][r];
        }
        int res = 0, sum = 0, leftSum = 0;
        for(int i = l; i <= r; i++) {
            sum += stoneVal[i];
        }
        for(int i = l; i < r; i++) {
            leftSum += stoneVal[i];
            if(leftSum > sum - leftSum) {
                res = Math.max(sum - leftSum + helper(dp, stoneVal, i + 1, r), res);
            } else if(leftSum < sum - leftSum) {
                res = Math.max(leftSum + helper(dp, stoneVal, l, i), res);
            } else {
                res = Math.max(res, Math.max(sum - leftSum + helper(dp, stoneVal, i + 1, r), leftSum + helper(dp, stoneVal, l, i)));
            }
        }
        dp[l][r] = res;
        return res;
    }
}
