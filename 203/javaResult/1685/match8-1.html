class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] preSum = new int[n + 1];
        
        for (int i = 0; i < n; i++) {
            preSum[i + 1] = preSum[i] + stoneValue[i];
        }
        
        int[][] dp = new int[n][n];
        
        for (int len = 1; len < n; len++) {
            for (int i = 0; i + len < n; i++) {
                
for (int j = i; j < i + len; j++) {
                    int l = preSum[j + 1] - preSum[i];
                    int r = preSum[i + len + 1] - preSum[j + 1];
                    
                    int sum = 0;
                    if (l < r) {
                        sum = l + dp[i][j];
                    } else if (l > r) {
                        sum = r + dp[j + 1][i + len];
                    } else {
                        sum = Math.max(l + dp[i][j], r + dp[j + 1][i + len]);
                    }
                    
                    dp[i][i + len] = Math.max(dp[i][i + len], sum);
                }
            }
        }
        
        return dp[0][n - 1];
    }
}
