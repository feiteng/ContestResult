class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] sum = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + stoneValue[i - 1];
        }
        int[][] dp = new int[n][n];
        for(int g = 1; g < n; g++) {
            for(int l = 0;l + g < n; l++) {
                int r = l + g;
                int tot = sum[r + 1] - sum[l];
                int left = 0;
                for(int i = l; i < r; i++) {
                    left += stoneValue[i];
                    
int right = tot - left;
                    if(left > right) {
                        dp[l][r] = Math.max(dp[l][r], right + dp[i + 1][r]);
                    } else if(right > left) {
                        dp[l][r] = Math.max(dp[l][r], left + dp[l][i]);
                    } else {
                        dp[l][r] = Math.max(dp[l][r], right + dp[i + 1][r]);
                        dp[l][r] = 
Math.max(dp[l][r], left + dp[l][i]);
                }
            }
        }
    }
        return dp[0][n - 1];
}
}
