class Solution {
        
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] preSum = new int[n + 1];
        for (int i = 0; i < n; i++)
            preSum[i + 1] = preSum[i] + stoneValue[i];
        int[][] memo = new int[n][n];
        return dfs(stoneValue, preSum, 0, n - 1, memo);
    }
    
    private int dfs(int[] stone, int[] preSum, int left, int right, int[][] memo) {
        if (left >= right) return 0;
        
        if (left + 1 == right) return Math.min(stone[left], stone[right]);
        
        if (memo[left][right] != 0) return memo[left][right];
        
        int res = 0;
        
        for (int i = left; i <= right; i++) {
            int leftSum = preSum[i + 1] - preSum[left];
            int rightSum = preSum[right + 1] - preSum[i + 1];
            
            if (leftSum > rightSum) {
                res = Math.max(res, rightSum + dfs(stone, preSum, i + 1, right, memo));
            } else if (leftSum < rightSum) {
                res = Math.max(res, leftSum + dfs(stone, preSum, left, i, memo));
            } else {
                res = Math.max(res, leftSum + Math.max(dfs(stone, preSum, i + 1, right, memo), dfs(stone, preSum, left, i, memo)));
            }
            //System.out.println(leftSum + ":" + rightSum);
        }
        
        memo[left][right] = res;
        return res;
    }
}
