class Solution {
    public int stoneGameV(int[] A) {
        int n = A.length;
        int[] prefix = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            prefix[i] = A[i - 1] + prefix[i - 1];
        }
        
        int[][] dp = new int[n][n];
        // dp[i][i] = 0;
        
        for (int d = 1; d < n; d++) {
            for (int i = 0; i + d < n; i++) {
                int j = i + d;
                for (int k = i; k < j; k++) {
                    int l = prefix[k + 1] - prefix[i];
                    int r = prefix[j + 1] - prefix[k + 1];
                    int c = 0;
                    if (l < r) {
                        c = l + dp[i][k];
                    } else if (l > r) {
                        c = r + dp[k + 1][j];
                    } else {
                        c = l + Math.max(dp[i][k], dp[k + 1][j]);
                    }
                    dp[i][j] = Math.max(dp[i][j], c);
                }
            }
        }
        return dp[0][n - 1];
    }
}
