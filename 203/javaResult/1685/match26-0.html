class Solution {
    public int stoneGameV(int[] nums) {
        final int N = nums.length;
        int[] sum = new int[N + 1];
        for (int i = 0; i < N; ++i) {
            sum[i + 1] = sum[i] + nums[i];
        }
        return dfs(new Integer[N][N], sum, 0, N - 1);
    }

    private int dfs(Integer[][] memo, int[] sum, final int start, final int end) {
        if (start >= end) {
            return 0;
        }
        if (memo[start][end] != null) {
            return memo[start][end];
        }

        int res = Integer.MIN_VALUE;
        for (int k = start; k < end; ++k) {
            int leftSum = sum[k + 1] - sum[start];
            int rightSum = sum[end + 1] - sum[k + 1];
            if (leftSum < rightSum) {
                res = Math.max(res, leftSum + dfs(memo, sum, start, k));
            } else if (leftSum > rightSum) {
                res = Math.max(res, rightSum + dfs(memo, sum, k + 1, end));
            } else {
                res = Math.max(res, leftSum + Math.max(dfs(memo, sum, start, k), dfs(memo, sum, k + 1, end)));
            }
        }
        memo[start][end] = res;

        return res;
    }
}
