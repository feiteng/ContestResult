class Solution {
    
public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] + stoneValue[i];
        }
        
        
int[][] dp = new int[n][n];
        
        for (int len = 2; len <= n; len++) {
            for (int start = 0; start + len <= n; start++) {
                int max = 0;
                for (int end = start; end < start + len - 1; end++) {
                    
if ((sum[end + 1] - sum[start]) > (sum[start + len] - sum[end + 1])) {
                        max = Math.max(max, sum[start + len] - sum[end + 1] + dp[end + 1][start + len - 1]);
                    } else if ((sum[end + 1] - sum[start]) < (sum[start + len] - sum[end + 1])) {
                        max = Math.max(max, sum[end + 1] - sum[start] + dp[start][end]);
                    } else {
                        max = Math.max(max, sum[start + len] - sum[end + 1] + dp[end + 1][start + len - 1]);
                        max = 
Math.max(max, sum[end + 1] - sum[start] + dp[start][end]);
                    }
                }
                dp[start][start + len - 1] = max;
            }
        }
        
        return dp[0][n - 1];
        
    }
}
