class Solution {
    public int stoneGameV(int[] sv) {
        int n = sv.length;
        int[][] dp = new int[n][n];
        int[][] val = new int[n][n];
        for (int i = 0; i < n; ++i) {
            val[i][i] = sv[i];
            for (int j = i+1; j < n; ++j) {
                val[i][j] = val[i][j-1]+sv[j];
            }
        
}
        for (int z = 1; z < n; ++z) {
            for (int i = 0; i+z<n; ++i) {
                int j = i+z;
                for (int k = i; k < j; ++k) {
                    
if (val[i][k]<=val[k+1][j]) {
                        dp[i][j] = Math.max(dp[i][j], val[i][k]+dp[i][k]);
                    }
                    if (val[i][k]>=val[k+1][j]) {
                        dp[i][j] = Math.max(dp[i][j], val[k+1][j]+dp[k+1][j]);
                    }
                }
            }
        }
        return dp[0][n-1];
    }
}
