class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] pre = new int[n + 1];
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++) {
            pre[i + 1] = pre[i] + stoneValue[i];
        }
        
        for (int len = 1; len <= n - 1; len++) {
            for (int i = 0; i < n - len; i++) {
                dp[i][i + len] = 0;
                for (int j = i; j <= i + len - 1; j++) {
                    int leftSum = getSum(pre, i, j), rightSum = getSum(pre, j + 1, i + len);
                    if (leftSum < rightSum) {
                        dp[i][i + len] = Math.max(dp[i][i + len], dp[i][j] + leftSum);
                    } else if (leftSum > rightSum) {
                        dp[i][i + len] = Math.max(dp[i][i + len], dp[j + 1][i + len] + rightSum);
                    } else {
                        dp[i][i + len] = Math.max(dp[i][i + len], leftSum + Math.max(dp[i][j], dp[j + 1][i + len]));
                    }
                }     
            }
        }
        
        return dp[0][n - 1];
    }
    
    private int getSum(int[] pre, int left, int right) {
        return pre[right + 1] - pre[left];
    }
}
