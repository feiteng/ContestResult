class Solution {
    int[][] dp;
    int[] sum;
    public int stoneGameV(int[] sv) {
        int len = sv.length;
        dp = new int[len+2][len+2];
        sum = new int[len+2];
        for (int i=0;i<len;i++) {
            sum[i+1] = sum[i] + sv[i];
        }

        return work(0,len-1);
    }
    int work(int l,int r) {
        if (dp[l][r] > 0) {
            return dp[l][r];
        }
        if (l == r) {
            return 0;
        }
        int ans = 0;
        for (int i=l;i<r;i++) {
            int lsum = sum[i+1] - sum[l];
            int rsum = sum[r+1] - sum[i+1];
            if (lsum > rsum) {
                ans = Math.max(ans,work(i+1,r) + rsum);
            } else if (lsum < rsum) {
                ans = Math.max(ans,work(l,i) + lsum);
            } else {
                ans = Math.max(ans,work(i+1,r) + rsum);
                ans = Math.max(ans,work(l,i) + lsum);
            }
        }
        return dp[l][r] = ans;
    }
    
}
