class Solution {
    public int stoneGameV(int[] stoneValue) {
        int len = stoneValue.length;
        int[] sum = new int[len+1];
        for (int i = 0 ; i < len ; ++i) {
            sum[i+1] = sum[i] + stoneValue[i];
        }
        
        int[][] dp = new int[len][len];
        for (int l = 0; l < len ; ++l) {
            for (int i = 0 ; i + l < len ; ++i) {
                int j = i+l;
                if (i == j) {
                    dp[i][j] = 0;
                    continue;
                }
                
                int tmp = 0;
                for (int k = i ; k < j ; ++k) {
                    int left = rangeSum(sum, i, k);
                    int right = rangeSum(sum, k+1, j);
                    
if (left < right) {
                        tmp = Math.max(tmp, left+dp[i][k]);
                    } else if (left > right) {
                        tmp = Math.max(tmp, right+dp[k+1][j]);
                    } else {
                        tmp = Math.max(tmp, left + 
Math.max(dp[i][k], dp[k+1][j]));
                    }
                }
                dp[i][j] = tmp;
            }
        }
        return dp[0][len-1];
    }
    
    private int rangeSum(int[] sum, int s, int e) {
        return sum[e+1] - sum[s];
    }
}
