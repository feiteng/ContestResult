class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] dp = new int[n+1][];
        int[] sum = new int[n+1];
        for(int i=1;i<=n;i++){
            dp[i] = new int[n+1];
            sum[i] = sum[i-1] + stoneValue[i-1];
        }
        for(int i=2;i<=n;i++) {
            for(int j=1;j+i-1<=n;j++){
                int r = j+i-1;
                for(int k=j;k<r;k++){
                    int s1 = sum[k]-sum[j-1];
                    int s2 = sum[r]-sum[k];
                    if (s1 >= s2) {
                        dp[j][r] = Math.max(dp[j][r], dp[k+1][r]+s2);
                    }
                    if (s1 <= s2) {
                        dp[j][r] = Math.max(dp[j][r], dp[j][k]+s1);
                    }
                }
            }
        }
        return dp[1][n];
    }
}
