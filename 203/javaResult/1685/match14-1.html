class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] + stoneValue[i];
        }

        int[][] dp = new int[n][n];
        for (int len = 2; len <= n; len++) {
            for (int start = 0; start < n; start++) {
                int end = start + len - 1;
                if (end >= n) {
                    continue;
                }
                
for (int firstend = start; firstend < end; firstend++) {
                    int firstsum = sum[firstend + 1] - sum[start];
                    int secondsum = sum[end + 1] - sum[firstend + 1];
                    int v;
                    if (firstsum == secondsum) {
                        v = Math.max(firstsum + dp[start][firstend], secondsum + dp[firstend + 1][end]);
                    
} else if (firstsum > secondsum) {
                        v = secondsum + dp[firstend + 1][end];
                    } else {
                        v = firstsum + dp[start][firstend];
                    
}
                    dp[start][end] = Math.max(dp[start][end], v);
                }
            }
        }

        return dp[0][n - 1];
    }
}
