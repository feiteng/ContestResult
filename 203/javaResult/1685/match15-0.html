class Solution {
    int[] sum;

    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        sum = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + stoneValue[i - 1];
        }
        int[][] dp = new int[n][n];
        for (int len = 1; len < n; len++) {
            for (int i = 0; i + len < n; i++) {
                int end = i+len;
                for (int j = i; j < end; j++) {
                    int left = sum[j+1] - sum[i];
                    int right = sum[end+1] - sum[j+1];
                    int max = 0;
                    if (left == right) {
                        max = Math.max(left + dp[i][j], right + dp[j+1][end]);
                    } else if (left < right) {
                        max = left + dp[i][j];
                    } else {
                        max = right + dp[j+1][end];
                    }
                    dp[i][end] = Math.max(dp[i][end], max);
                }
            }
        }
        return dp[0][n-1];
    }
}
