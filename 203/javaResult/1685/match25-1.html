class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i += 1) {
            dp[i][i] = 0;
        }
        int[] prefixSum = new int[n + 1];
        for (int i = 1; i <= n; i += 1) {
            prefixSum[i] = prefixSum[i - 1] + stoneValue[i - 1];
        }
        for (int i = 2; i <= n; i += 1) {
            for (int j = 0; j <= n - i; j += 1) {
                // [j, j + i - 1]
                
for (int k = j; k <= j + i - 2; k += 1) {
                    // [j, k], [k + 1][j + i - 1]
                    int left = prefixSum[k + 1] - prefixSum[j];
                    int right = prefixSum[j + i] - prefixSum[k + 1];
                    if (left <= right) {
                        dp[j][j + i - 1] = Math.max(dp[j][j + i - 1], left + dp[j][k]);
                    } 
                    if (left >= right) {
                        dp[j][j + i - 1] = Math.max(dp[j][j + i - 1], right + dp[k + 1][j + i - 1]);
                    } 
                    // else {
                    //     dp[j][j + i - 1] = Math.max(dp[j][j + i - 1], left + dp[k + 1][j + i - 1]);
                    //     dp[j][j + i - 1] = Math.max(dp[j][j + i - 1], right + dp[j][k]);
                    // }
                }
            }
        }
        return dp[0][n - 1];
    }
}
