class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[] preSum = new int[n + 1];
        for (int i = 1; i < preSum.length; ++i) {
            preSum[i] = preSum[i - 1] + stoneValue[i - 1];
        
}
        int[][] dp = new int[n][n + 1];
        return dfs(stoneValue, 0, n, dp, preSum);
    }
    private static int dfs(int[] sv, int start, int end, int[][] memo, int[] preSum) {
        if (end - start < 2) return 0
;
        if (memo[start][end] != 0) return memo[start][end];
        int m = 0;
        for (int i = start; i < end; ++i) {
            int l = preSum[i + 1] - preSum[start];
            int r = preSum[end] - preSum[i + 1];
            if (l < r) {
                m = Math.max(m, l + dfs(sv, start, i + 1, memo, preSum));
            }
            else if (l > r) {
                m = Math.max(m, r + dfs(sv, i + 1, end, memo, preSum));
            }
            else {
                m = Math.max(m, l + Math.max(dfs(sv, start, i + 1, memo, preSum), dfs(sv, i + 1, end, memo, preSum)));
            }
        }
        return memo[start][end] = m;
    }
}
