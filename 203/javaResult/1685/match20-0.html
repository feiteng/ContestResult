class Solution {
    public int stoneGameV(int[] stoneValue) {
        int len = stoneValue.length;
        int[][] cache = new int[len][len];
        
        int[] sum = new int[len+1];
        for (int i=0; i<len; i++) {
            sum[i+1] = sum[i] + stoneValue[i];
        }
        
        for (int L=2; L<=len; L++) {
            for (int s=0; s+L<=len; s++) {
                int e = s+L-1;
                for (int m=s; m<e; m++) {
                    int v1 = sum[m+1] - sum[s];
                    int v2 = sum[e+1] - sum[m+1];
                    if (v1 > v2) {
                        cache[s][e] = Math.max(cache[s][e], v2 + cache[m+1][e]);
                    } else if (v1 == v2) {
                        cache[s][e] = Math.max(cache[s][e], Math.max(v2 + cache[m+1][e], v1 + cache[s][m]));
                    } else {
                        cache[s][e] = Math.max(cache[s][e], v1 + cache[s][m]);
                    }
                    
                }    
            }
        }
        return cache[0][len-1];
    }
    
}
