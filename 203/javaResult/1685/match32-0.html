class Solution {
    public int stoneGameV(int[] stoneValue) {
        int len = stoneValue.length;
        int[][] dp = new int[len][len];
        int[] preSum = new int[len];
        for (int i = 1; i < len; i++) {
            preSum[i] = preSum[i - 1] + stoneValue[i];
        }
        for (int j = 1; j < len; j++) {
            for (int i = j - 1; i >= 0; i--) {
                dp[i][j] = 0;
                for (int k = i; k < j; k++) {
                    int sum1 = preSum[k] - preSum[i] + stoneValue[i];
                    int sum2 = preSum[j] - preSum[k + 1] + stoneValue[k + 1];
                    if (sum1 > sum2) {
                        dp[i][j] = Math.max(dp[i][j], sum2 + dp[k + 1][j]);
                    } else if (sum1 < sum2) {
                        dp[i][j] = Math.max(dp[i][j], sum1 + dp[i][k]);
                    } else {
                        dp[i][j] = Math.max(dp[i][j], sum1 + Math.max(dp[i][k], dp[k + 1][j]));
                    }
                }
            }
        }
        return dp[0][len - 1];
    }
}
