class Solution {
    int[][] dp;
    int[] sum;
    
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        dp = new int[n][n];
        for(int i = 0; i < n; i++){
            Arrays.fill(dp[i], -1);
        }
        
        
sum = new int[n + 1];
        for(int i = 1; i <= n; i++){
            sum[i] = sum[i - 1] + stoneValue[i - 1];
        }
        
        return findMaxValue(stoneValue, 0, n - 1);
    }
    
    private int findMaxValue(int[] val, int i, int j)
{
        if(i == j){
            return 0;
        }
        if(dp[i][j] != -1){
            return dp[i][j];
        }
        
        int ans = 0;
        int total = sum[j + 1] - sum[i];
        int curr = 0;
        for(int k = i; k < j; k++){
            curr += val[k];
            total -= val[k];
            
            
if(curr < total){
                ans = Math.max(ans, curr + findMaxValue(val, i, k));
            }
            else if(curr > total){
                ans = Math.max(ans, total + findMaxValue(val, k + 1, j));
            }
            else{
                ans = Math.max(ans, curr + Math.max(
findMaxValue(val, i, k), findMaxValue(val, k + 1, j)));
            }
        }
        
        return dp[i][j] = ans;
    }
}
