class Solution {
    public int stoneGameV(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] dp = new int[n][n];

        int[][] sum = new int[n][n];
        for (int i = 0;i<n;i++) {
            for(int j = i;j<n;j++) {
                for (int k = i;k<=j;k++) {
                    sum[i][j] += stoneValue[k];
                }
            }
        
}

        for (int len = 2;len <=n;len++) {
            for (int left = 0;left + len - 1 < n;left++) {
                int right = left + len -1;
                int maxValue = 0;
                for (int mid = left;mid<right;mid++) {
                    int sumLeftValue = sum[left][mid];
                    int sumRightValue = sum[mid+1][right];
                    if (sumRightValue > sumLeftValue) {
                        maxValue= Math.max(maxValue, sumLeftValue + dp[left][mid]);
                    } else if (sumRightValue < sumLeftValue) {
                        maxValue = Math.max(maxValue, sumRightValue + dp[mid+1][right]);
                    } else {
                        maxValue = Math.max(maxValue, sumLeftValue + Math.max(dp[left][mid],
                                dp[mid+1][right]));
                    }
                }
                dp[left][right] = maxValue;
            }
        }
        return dp[0][n-1];
    }
}
