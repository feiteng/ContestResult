class Solution {
    
public int stoneGameV(int[] stoneValue) {
        int len = stoneValue.length;
        int[] preSum = new int[len+1];
        for(int i = 0; i< len; i++){
            preSum[i+1] = preSum[i] + stoneValue[i];
        }
        int[][] dp = new int[len][len];
        for(int width = 2; width <= len; width ++){
            for(int i= 0; i+width-1<len; i++){
                int j = i + width - 1;
                
for(int k = i; k<j; k++){
                    //[i,k],[k+1,j]
                    int sum1 = preSum[k+1] - preSum[i];
                    int sum2 = preSum[j+1] - preSum[k+1];
                    if(sum1>sum2){
                        dp[i][j] = Math.max(dp[i][j], dp[k+1][j] + sum2);
                    }else if(sum1<sum2){
                        dp[i][j] =Math.max(dp[i][j], dp[i][k] + sum1);
                    }else{
                        int max = Math.max(dp[k+1][j], dp[i][k]);
                        dp[i][j] = 
Math.max(dp[i][j], max + sum1);
                    }
                }
            }
        }
        return dp[0][len-1];
    }
}
