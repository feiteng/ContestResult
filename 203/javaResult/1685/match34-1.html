class Solution {
    public int stoneGameV(int[] stones) {
        int n = stones.length, dp[][] = new int[n][n], sum[] = new int[n];
        
        sum[0] = stones[0];
        for(int i = 1; i < n; i++) {
            // dp[i][i] = stones[i];
            sum[i] = sum[i-1] + stones[i];
        }
        
        for(int len = 2; len <= n; len++) {  
            for(int i = 0; i < n - len + 1; i++) {
                for(int j = i + 1; j <= i + len - 1; j++) { // idx of split point
                    int lSum, rSum;
                    if(i == 0) {
                        lSum = sum[j-1];
                        rSum = sum[i + len - 1] - sum[j - 1];
                    } else {
                        lSum = sum[j-1] - sum[i-1];
                        rSum = sum[i + len - 1] - sum[j-1];
                    }
                    
if(lSum > rSum) { // right sum left
                        dp[i][i + len - 1] = Math.max(dp[i][i + len - 1], dp[j][i + len - 1] + rSum);
                    } else if(lSum < rSum) { // left part left
                        dp[i][i + len - 1] = Math.max(dp[i][i + len - 1], dp[i][j-1] + lSum);
                    } else {
                        dp[i][i + len - 1] = Math.max(dp[i][i + len - 1], dp[j][i + len - 1] + rSum);
                        dp[i][i + len - 1] = 
Math.max(dp[i][i + len - 1], dp[i][j-1] + lSum);
                    }
                }
            }
        }
        return dp[0][n-1];
    }
}
