class Solution {
    public int stoneGameV(int[] stoneValue) {
        if (stoneValue == null || stoneValue.length < 2) {
            return 0;
        }
        int n = stoneValue.length;
        int[] prefixSum = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            prefixSum[i] = prefixSum[i - 1] + stoneValue[i - 1];
        }
        int[][] memo = new int[n][n];
        return dfs(stoneValue, prefixSum, memo, 0, n - 1);
    }
    private int dfs(int[] stoneValue, int[] prefixSum, int[][] memo, int i, int j) {
        if (i == j) {
            return 0;
        }
        if (memo[i][j] != 0) {
            return memo[i][j];
        }

        int val = 0;
        for (int k = i; k < j; k++) {
            int left = prefixSum[k + 1] - prefixSum[i];
            int right = prefixSum[j + 1] - prefixSum[k + 1];
            if (left > right) {
                val = Math.max(val, right + dfs(stoneValue, prefixSum, memo, k + 1, j));
            }
            else if (left < right) {
                val = Math.max(val, left + dfs(stoneValue, prefixSum, memo, i, k));                                
            } 
            else {
                int rightVal = right + dfs(stoneValue, prefixSum, memo, k + 1, j);
                int leftVal = left + dfs(stoneValue, prefixSum, memo, i, k);                                
                val = Math.max(val, Math.max(leftVal, rightVal));
            }
        }
        memo[i][j] = val;
        return val;
    }
}
