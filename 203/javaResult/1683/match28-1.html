class Solution {
    public int maxCoins(int[] piles) {
        int n = piles.length / 3;
        Arrays.sort(piles);
        int res = 0;
        int cnt = 0;
        for (int i = 3 * n; cnt < n; i -= 2) {
            res += piles[i - 2];
            cnt++;
        }
        
        return res;
    }
}
