class Solution {
    public int maxCoins(int[] piles) {
        
        int n = piles.length;
        Arrays.sort(piles);
        
        int cnt = n / 3;
        int res = 0;
        for(int i = n - 2; cnt > 0; cnt--, i -= 2) {
            res += piles[i];
        }
        return res;
    }
}
