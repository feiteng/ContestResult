class Solution {
    public int maxCoins(int[] piles) {
        int n = piles.length;
        Arrays.sort(piles);
        int chos = 0;
        int ans = 0;
        for (int i = piles.length-2; chos < n/3; i-=2) {
            chos++;
            ans += piles[i];
        }
        return ans;
    }
}
