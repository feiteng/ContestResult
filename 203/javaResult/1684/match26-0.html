class Solution {
    
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        Map<Integer, int[]> indexToRange = new HashMap<>();
        Map<Integer, Integer> endToSize = new HashMap<>();
        Set<Integer> added = new HashSet<>();
        Map<Integer, Integer> sizeCount = new HashMap<>();
        sizeCount.put(m, 0);
        int ret = -1;
        for (int i = 0; i < n; i += 1) {
            int cur = arr[i] - 1;
            int len = 1;
            int left = cur;
            int right = cur;
            if (cur - 1 >= 0 && added.contains(cur - 1)) {
                left = indexToRange.get(cur - 1)[0];
                int leftLen = endToSize.get(cur - 1);
                len += leftLen;
                sizeCount.put(leftLen, sizeCount.get(leftLen) - 1);
            } 
            if (cur + 1 < n && added.contains(cur + 1)) {
                right = indexToRange.get(cur + 1)[1];
                int rightLen = endToSize.get(right);
                len += rightLen;
                sizeCount.put(rightLen, sizeCount.get(rightLen) - 1);
            }
            indexToRange.put(left, new int[] {left, right});
            indexToRange.put(right, new int[] {left, right});
            endToSize.put(right, len);
            added.add(cur);
            sizeCount.put(len, sizeCount.getOrDefault(len, 0) + 1);
            if (sizeCount.get(m) > 0) {
                ret = i + 1;
            }
        }
        
        // List<Integer> lst = new LinkedList<>();
        // lst.add(0);
        // lst.add(n + 1);
        // if (m == n) return n;
        // for (int i = n - 1; i >= 0; i -= 1) {
        //     int cur = arr[i];
        //     int index = Bsearch(lst, cur);
        //     lst.add(index, cur);
        //     if (lst.get(index + 1) - lst.get(index) - 1 == m || lst.get(index) - lst.get(index - 1) - 1 == m) {
        //         return i;
        //     }
        // }
        return ret;
    }
}
