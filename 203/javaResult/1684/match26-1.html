class Solution {
    public int findLatestStep(int[] arr, int m) {
        int ans = -1;
        Map<Integer, Integer> count = new HashMap<>();
        
        Map<Integer, Integer> stoE = new HashMap<>();
        Map<Integer, Integer> etoS = new HashMap<>();
        
        for (int i = 0; i < arr.length; i++) {
            int curr = arr[i];
            
int prev = curr-1;
            int after = curr+1;
            
            int news = curr;
            int newe = curr;
            
            if (etoS.containsKey(prev)) {
                news = etoS.get(prev);
                //update count
                int len = prev - news + 1;
                count.put(len, count.get(len)-1);
                etoS.remove(prev);
            }
            if (stoE.containsKey(after)) {
                newe = stoE.get(after);
                
                int len = newe - after + 1;
                count.put(len, count.get(len)-1);
                stoE.remove(after);
            }
            
            stoE.put(news, newe);
            etoS.put(newe, news);
            int len = newe - news + 1;
            count.put(len, count.getOrDefault(len, 0) + 1);
            
            if (count.containsKey(m) && count.get(m) >= 1) {
                ans = i;
            }
        }
        if (ans == -1) {
            return ans;
        }
        return ans+1;
        
    }
}
