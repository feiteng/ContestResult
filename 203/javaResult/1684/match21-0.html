class Solution {
    public int findLatestStep(int[] arr, int m) {
        final int n = arr.length;
        int[] lens = new int[n+2];
        int mCnt = 0;
        int maxStep = -1;
        for (int i = 0; i < n; i++) {
            int pos = arr[i];
            int len = 1 + lens[pos-1] + lens[pos+1];
            if (len == m) {
                mCnt++;
            }
            if (lens[pos-1] == m) {
                mCnt--;
            }
            if (lens[pos+1] == m) {
                mCnt--;
            }
            lens[pos-lens[pos-1]] = len;
            lens[pos+lens[pos+1]] = len;
            if (mCnt > 0)
                maxStep = Math.max(maxStep, i+1);
        }
        return maxStep;
    }
}
