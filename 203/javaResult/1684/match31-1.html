class Solution {
    int[] s;
    int find(int x) {
        if (s[x] == x) {
            return x;
        }
        s[x] = find(s[x]);
        return s[x];
    }
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) {
            return n;
        
}
        
        TreeSet<Integer> tree = new TreeSet<>();
        tree.add(0);
        tree.add(n+1);
        
        for (int i = n-1; i >= 0; i--) {
            Integer integer = tree.lower(arr[i]);
            if (arr[i] - integer.intValue() - 1 == m) {
                return i;
            }
            integer = tree.higher(arr[i]);
            if (integer.intValue() - arr[i] - 1 == m) 
{
                return i;
            }
            tree.add(arr[i]);
        }
        
        return -1;
    }
}
