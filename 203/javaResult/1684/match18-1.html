class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if(m == n) return n;
        TreeSet<Integer> zeros = new TreeSet<>();
        zeros.add(0);
        zeros.add(n+1);
        for(int i = n - 1; i >= 0; i--) {
            Integer left = zeros.floor(arr[i]);
            Integer right = zeros.ceiling(arr[i]);
            zeros.add(arr[i]);
            if(arr[i] - left - 1 == m) return i;
            if(right - arr[i] - 1 == m) return i;
        }
        return -1;
    }
}
