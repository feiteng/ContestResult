class Solution {

    public int findLatestStep(int[] arr, int m) {
        TreeSet<Integer> S = new TreeSet<>();
        int n = arr.length;
        if (m == n) return n;
        S.add(0);
        S.add(n + 1);
        for (int i = n - 1; i >= 0; --i) {
            int l = S.lower(arr[i]), r = S.higher(arr[i]);
            if (r - arr[i] - 1 == m || arr[i] - l - 1 == m) return i;
            S.add(arr[i]);
        }
        return -1;
    }
}
