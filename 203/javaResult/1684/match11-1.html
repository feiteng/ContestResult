class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if(m == n)
            return n;
        TreeMap<Integer,Integer> map = new TreeMap<>();
        
map.put(1,n);
        for(int i = n - 1; i >= 0; i--){
            int p = arr[i];
            int l = map.floorKey(p);
            int r = map.get(l);
            if(p - l == m || r - p == m)
                return i;
            map.remove(l);
            map.put(l,p - 1);
            map.put(p + 1, r);
        }
        return -1;
    }
}
