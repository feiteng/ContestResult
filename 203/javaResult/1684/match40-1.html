class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) {
            return n;
        }
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(1, n);
        for (int i = n-1; i >= 0; i--) {
            int idx = arr[i];
            int start = map.floorKey(idx), end = map.get(start);
            map.remove(start);
            if (start == end) {
                continue;
            }
            if (start == idx) {
                map.put(idx+1, end);
            } else if (end == idx) {
                map.put(start, idx-1);
            } else {
                map.put(start, idx-1);
                map.put(idx+1, end);
            }
            
if (idx-start == m || end-idx == m) {
                return i;
            }
        }
        return -1;
    }
}
