import java.util.ArrayList;
import java.util.List;

class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n=arr.length;
        List<Integer> list=new ArrayList<>(n);
        list.add(0);
        list.add(n+1);
        if(n==m){
            return n;
        }
        for(int i=n-1;i>=0;i--){
            int index=f(list,arr[i]);
            if(list.get(index)-arr[i]-1==m){
                return i;
            }
            if(arr[i]-list.get(index-1)-1==m){
                return i;
            }
            list.add(index,arr[i]);
        }
        return -1;
    }
    int f(List<Integer> list,int v){
        int l=0;
        int r=list.size()-1;
        while(l<r){
            int m=(l+r)/2;
            if (v<=list.get(m)){
                r=m;
            }else{
                l=m+1;
            }
        }
        return r;
    }
}
