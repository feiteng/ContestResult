class Solution {
    private static class UF {
        private int[] parents;
        private int[] sizes;
        private int[] counts;
        
        UF(int n) {
            parents = new int[n];
            sizes = new int[n];
            counts = new int[n + 1];
            for (int i = 0; i < n; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        private int findRoot(int a) {
            if (a == parents[a]) {
                return a;
            }
            return parents[a] = findRoot(parents[a]);
        }
        
        private void union(int a, int b) {
            int rA = findRoot(a);
            int rB = findRoot(b);
            if (rA != rB) {
                int sA = sizes[rA];
                int sB = sizes[rB];
                int size = sA + sB;
                parents[rA] = rB;
                counts[sA]--;
                counts[sB]--;
                counts[size]++;
                sizes[rB] = size;
            }
        }
        
        private void add(int a) {
            counts[1]++;
        }
    }
    
    
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        int[] nums = new int[n + 1];
        UF uf = new UF(n + 1);
        int res = -1;
        for (int i = 0; i < n; i++) {
            int a = arr[i];
            nums[a] = 1;
            uf.add(a);
            if (a > 1 && nums[a - 1] == 1) {
                uf.union(a, a - 1);
            }
            if (a < n && nums[a + 1] == 1) {
                uf.union(a, a + 1);
            }
            if (uf.counts[m] >= 1) {
                res = i + 1;
            }
        }
        return res;
    }
}
