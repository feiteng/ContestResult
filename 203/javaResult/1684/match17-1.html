class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if(n == m) return n;
        TreeSet<Integer> p = new TreeSet<>();
        p.add(0);
        p.add(n + 1);
        for(int i = n; i >= 1; i--){
            int cur = arr[i - 1];
            if(p.lower(cur) != null){
                int lo = p.lower(cur);
                if(cur - lo == m + 1) return i - 1; 
            }
            if(p.higher(cur) != null){
                int hi = p.higher(cur);
                if(hi - cur == m + 1) return i - 1;
            
}
            p.add(cur);
        }
        return -1;
    }
}
