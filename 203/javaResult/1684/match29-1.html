class Solution {
    public int findLatestStep(int[] arr, int m) {
        Map<Integer, Node> start = new HashMap<>();
        Map<Integer, Node> end = new HashMap<>();
        int res = -1;
        
int count = 0;
        for(int i = 0; i < arr.length; i++){
            int s = arr[i];
            int e = arr[i];
            int val = 1;
            if(end.containsKey(arr[i] - 1)){
                val += end.get(arr[i] - 1).val;
                s = end.get(arr[i] - 1).start;
                if (end.get(arr[i] - 1).val == m){
                    count--;
                }
                end.remove(arr[i] - 1);
            }
            if(start.containsKey(arr[i] + 1)){
                val += start.get(arr[i] + 1).val;
                e = start.get(arr[i] + 1).end;
                if (start.get(arr[i] + 1).val == m){
                    count--;
                }
                start.remove(arr[i] + 1);
            }
            Node node = new Node(s, e, val);
            start.put(s, node);
            end.put(e, node);
            if (val == m){
                count++;
            }
            if(count > 0)
{
                res = i + 1;
            }
        }
        return res;
    }
}

class Node{
    int start;
    int end;
    int val;
    public Node(int start, int end, int val){
        this.start = start;
        this.end = end;
        this.val = val;
    }
}
