class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) {
            return n;
        }
        
        TreeMap<Integer, Integer> treeMap = new TreeMap<>();
        treeMap.put(1, n);
        for (int i = n - 1; i >= 0; i--) {
            int start = treeMap.floorKey(arr[i]);
            int end = treeMap.get(start);
            
if (arr[i] - start == m || end - arr[i] == m) {
                return i;
            }
            
            if (start == end) {
                treeMap.remove(start);
            } else if (start == arr[i]) {
                treeMap.remove(start);
                treeMap.put(arr[i] + 1, end);
            } else if (end == arr[i]) {
                treeMap.put(start, arr[i] - 1);
            } else {
                treeMap.remove(start);
                treeMap.put(start, arr[i] - 1);
                treeMap.put(arr[i] + 1, end);
            }
        }
        
        return -1;
    }
}
