class Solution {
    public int findLatestStep(int[] arr, int m) {
        TreeMap<Integer, Integer> tree = new TreeMap<>();
        int n = arr.length;
        if (n == m) return n;
        tree.put(0, n);
        for (int i = n - 1; i >= 0; i--) {
            int t = arr[i] - 1;
            int k = tree.floorKey(t);
            int v = tree.remove(k);
            
            int k1 = k;
            int v1 = t - k;
            int k2 = t + 1;
            int v2 = v - (v1 + 1);
            // System.out.printf("%d %d %d %d %d\n", k, k1, v1, k2, v2);
            if (v1 == m || v2 == m) return i
;
            if (v1 > 0) tree.put(k1, v1);
            if (v2 > 0) tree.put(k2, v2);
        }
        return -1;
    }
}
