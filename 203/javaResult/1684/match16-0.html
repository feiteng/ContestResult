/*
111 0 111

l..i-1   
i+1..r 

[]
 r 
 l
[] 

treemap 
l  -> r 



*/
class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;

        int curr= n;
        
        if(m==n) return n;
        
        TreeMap<Integer, Integer> map = new TreeMap<>(); //l -> r
        map.put(0, n-1);

        for(int i=n-1; i>=0; i--){
            int v = arr[i]-1;
            int l = map.floorKey(v);
            int r = map.remove(l);

            if(l<=v-1){
                if((v-1)-l+1 == m) return i;
                map.put(l, v-1);
            }
            
            if(v+1<=r){
                if(r-(v+1)+1 == m) return i;
                map.put(v+1, r);
            }
        }
        
        return -1;
        
    }
}
