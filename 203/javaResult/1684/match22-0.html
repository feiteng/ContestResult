class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        int[] uf = new int[n+1];
        int[] l = new int[n+1];
        int[] r = new int[n+1];
        for (int i=0; i<n; i++) {
            l[i] = i;
            r[i] = i;
        }
        int ans = -1;
        int[] map = new int[n+1];
        for (int i=0; i<n; i++) {
            int t = arr[i] - 1;
            int ll = t;
            int rr = t;
            int x = 1;
            if (t > 0 && uf[t-1] > 0) {
                ll = l[t-1];
                x += uf[ll];
                map[uf[ll]]--;
            }
            if (t < n-1 && uf[t+1] > 0) {
                rr = r[t+1];
                x += uf[rr];
                map[uf[rr]]--;
            }
            map[x]++;
            if (map[m] > 0)
                ans = i+1;
            uf[ll] = x;
            uf[rr] = x;
            l[rr] = ll;
            r[ll] = rr;
        }
        return ans;
    }
}
