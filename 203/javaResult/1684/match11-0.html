class Solution {
    public int findLatestStep(int[] arr, int m) {
        int len = arr.length;
        if(m==len) return len;
        TreeSet<Integer> set = new TreeSet<>();
        set.add(0);
        set.add(len+1);
        for(int i=len-1;i>=0;i--){
            int cur = arr[i];
            int big = set.higher(cur);
            int small = set.lower(cur);
            if(big-cur-1==m||cur-small-1==m)return i;
            set.add(cur);
        }
        return -1;
    }
}
