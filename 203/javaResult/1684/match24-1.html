class Solution {
    public int findLatestStep(int[] arr, int m) {
        if(arr.length == m) return m;
        int res = 0;
        // TreeMap<Integer, Integer> count = new TreeMap<>();
        TreeSet<Integer> ts = new TreeSet<>();
        
ts.add(0);
        ts.add(arr.length + 1);
        for(int i=arr.length-1; i>=0; i--){
            Integer low = ts.floor(arr[i]);
            Integer high = ts.ceiling(arr[i]);
            int a = arr[i] - low - 1;
            int b = high - arr[i] - 1;
            
            
if(a==m || b==m) return i;
            ts.add(arr[i]);
        }
        return -1;

    }
}
