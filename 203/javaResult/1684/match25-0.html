class Solution {
    public int findLatestStep(int[] arr, int m) {
        int res = -1, count = 0;
        int[] len = new int[arr.length];
        for (int k = 0, j, i; k < arr.length; k++) {
            i = arr[k] - 1;
            if (i < arr.length - 1 && len[i + 1] > 0 && 
                i > 0 && len[i - 1] > 0) {
                if (len[i - 1] == m) count--;
                if (len[i + 1] == m) count--;
                
                j = len[i - len[i - 1]] + len[i + 1] + 1;
                len[i - len[i - 1]] = j;
                len[i + len[i + 1]] = j;
                if (j == m) count++;
            } else if (i > 0 && len[i - 1] > 0) {
                if (len[i - 1] == m) count--;
                j = len[i - len[i - 1]] + 1;
                len[i - len[i - 1]] = j;
                len[i] = j;
                if (j == m) count++;
            } else if (i < arr.length - 1 && len[i + 1] > 0) {
                if (len[i + 1] == m) count--;
                j = len[i + 1] + 1;
                len[i] = j;
                len[i + len[i + 1]] = j;
                if (j == m) count++;
            } else {
                j = 1;
                len[i] = j;
                if (j == m) count++;
            }
            
            if (count > 0) res = k + 1;
        }
        
        return res;
    }
}
