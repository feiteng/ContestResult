class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) return n;
        TreeSet<Integer> cnt = new TreeSet<>();
        cnt.add(0); cnt.add(n+1);
        
        for (int i=n-1; i>=0; i--) {
            Integer l = cnt.floor(arr[i]) + 1;
            Integer h = cnt.higher(arr[i]) - 1;
            if (arr[i]-l == m || h-arr[i] == m) {
                return i;
            }
            cnt.add(arr[i]);
        }
        
        return -1;
    }
}
