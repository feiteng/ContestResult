class Solution {
    public int findLatestStep(int[] a, int m) {
        int n=a.length;
        int c[]=new int[n+1];
        int dp[]=new int[n+2];
        int ans=-1;
        for(int i=0;i<n;i++){
            int p=a[i],prev=dp[p-1],suc=dp[p+1];
            int k=prev+1+suc;
            c[prev]--;c[suc]--;
            dp[p-prev]=dp[p+suc]=k;
            c[k]++;
            
if(c[m]!=0)
                ans=i+1;
        }
        return ans;
    }
}
