class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length; 
        if (m == n) return n;
        TreeSet<Integer> set = new TreeSet<>();
        for (int i = n - 1; i >= 1; i--) {
            int l = arr[i];
            set.add(l);
            Integer higher = set.higher(l);
            if (higher != null && higher - l == m + 1) return i;
            if (higher == null && n + 1 - l == m + 1) return i;
            Integer lower = set.lower(l);
            if (lower != null && l - lower == m + 1) return i;
            if (lower == null && l == m + 1) return i;
        }
        return -1;
        
    }
}
