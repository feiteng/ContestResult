class Solution {
    public int findLatestStep(int[] arr, int m) {
        int[] bits = new int[arr.length + 2];
        int mCount = 0;
        int last = -1;
        for (int i = 0; i < arr.length; ++i) {
            int idx = arr[i];
            if (bits[idx] != 0) continue;
            
if (bits[idx + 1] != 0 && bits[idx - 1] != 0) {
                if (bits[idx + 1] == m) --mCount;
                if (bits[idx - 1] == m) --mCount;
                bits[idx] = bits[idx - 1] + bits[idx + 1] + 1;
                bits[idx + 1 + (bits[idx + 1] - 1)] = bits[idx];
                bits[idx - 1 - (bits[idx - 1] - 1)] = bits[idx];
                if (bits[idx] == m) ++mCount;
            }
            else if (bits[idx + 1] != 0) {
                if (bits[idx + 1] == m) --mCount;
                bits[idx] = bits[idx + 1] + 1;
                ++bits[idx + 1 + (bits[idx + 1] - 1)];
                
if (bits[idx] == m) ++mCount;
            }
            else if (bits[idx - 1] != 0) {
                if (bits[idx - 1] == m) --mCount;
                bits[idx] = bits[idx - 1] + 1;
                ++bits[idx - 1 - (bits[idx - 1] - 1)];
                if (bits[idx] == m) ++mCount;
            }
            else {
                
bits[idx] = 1;
                if (m == 1) ++mCount;
            }
            if (mCount >= 1) last = i;
            // System.out.println(bits[idx]);
            // System.out.println(mCount);
        }
        return last == -1 ? -1 : last + 1;
    }
}
