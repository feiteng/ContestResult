class Solution {
    
public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        boolean[] mark = new boolean[n + 2];
        UnionFind uf = new UnionFind(n);
        int ans = -1;
        int steps = 1;
        for (int i : arr) {
            
mark[i] = true;
            uf.selfUnion(i);
            if (mark[i - 1]) {
                uf.union(i, i - 1);
            } 
            if (mark[i + 1]) {
                uf.union(i, i + 1);
            }
    
            if (uf.map.containsKey(m) && uf.map.get(m) > 0) 
{
                ans = steps;
            }
            steps++;
            //System.out.println(uf.map.get(m));
        }
        return ans;
    }
    
    
private class UnionFind {
        int[] parents;
        Map<Integer, Integer> map;
        int[] cnt;
        
        public UnionFind (int n) {
            parents = new int[n + 1];
            map = new HashMap();
            
cnt = new int[n + 1];
            for (int i = 0; i <= n; i++) {
                parents[i] = i;
                cnt[i] = 1;
            }
            map.put(1, 0);
        }
        
        private void selfUnion(int i) {
            map.put(1, map.getOrDefault(1, 0) + 1);
        
}
        
        public int find (int i) {
            if (parents[i] == i) {
                return i;
            }
            return 
find(parents[i]);
        }
        
        public void union (int i, int j) {
            int pi = find(i);
            int pj = find(j);
            if (pi == pj) return;
            parents[pi] = pj;
            map.put(cnt[pj], map.get(cnt[pj]) - 1);
            map.put(cnt[pi], map.get(cnt[pi]) - 1);
            cnt[pj] += cnt[pi];
            map.put(cnt[pj], map.getOrDefault(cnt[pj], 0) + 1);
        }
    }
}
