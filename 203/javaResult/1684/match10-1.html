class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) return n;
        TreeSet<Integer> ts = new TreeSet<>();
        ts.add(0);
        ts.add(n+1);
        int step = n;
        for (; step >= 1; step--) {
            int cur = arr[step-1];
            int hi = ts.higher(cur);
            if (hi - cur == m + 1) return step - 1;
            
int lo = ts.lower(cur);
            if (cur - lo == m + 1) return step - 1;
            ts.add(cur);
        }
        return -1;
    }
}
