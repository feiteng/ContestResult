class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        
int step = n;
        if(m == n){
            return n;
        }
        TreeSet<Integer> set = new TreeSet<>();
        set.add(0);
        set.add(n+1);
        for(int i = n-1; i >= 0; i--){
            int cur = arr[i];
            int l = set.lower(cur);
            int h = set.higher(cur);
            if(cur - l-1 == m || h - cur-1 == m)
{
                return i;
            }
            set.add(cur);
        }
        return -1;
    }
}
