class Solution {
    public int findLatestStep(int[] arr, int m) {
        if(m == arr.length)
        {
            return arr.length;
        }
        Map<Integer,Integer> map = new HashMap();
        
map.put(1,arr.length);
        for(int i=arr.length-1;i>=0;i--)
        {
            int index = arr[i];
            if(check(map,index,m))
            {
                return i;
            }
            if(map.size() == 0)
            {
                return -1;
            }
        }
        return -1;
    }
    
    private boolean check(Map<Integer,Integer> map,int index,int m)
    {
        for(Map.Entry<Integer,Integer> en : map.entrySet())
        {
            if(en.getKey() > index || en.getValue() < index)
            {
                continue;
            }
            int start = en.getKey(),end = en.getValue();
            int left = index-start,right = end-index;
            if(left == m || right == m)
            {
                return true;
            }
            map.remove(en.getKey());
            if(left > m)
            {
                map.put(start,index-1);
            }
            if(right > m)
            {
                map.put(index+1,end);
            }
            break;
        }
        return false;
    }
}
