class Solution {
    public int findLatestStep(int[] arr, int m) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int cnt = 0;
        int res = -1;
        int index = 1;
        for (int a : arr) {
            Integer floorKey = map.floorKey(a);
            Integer ceilingKey = map.ceilingKey(a);
            if (floorKey == null && ceilingKey == null) {
                map.put(a, a);
                if (m == 1) {
                    cnt++;
                }
            } else if (floorKey == null) {
                if (ceilingKey == a + 1) {
                    int value = map.get(ceilingKey);
                    if (m - 1 == value - ceilingKey) {
                        cnt--;
                    }
                    map.remove(ceilingKey);
                    map.put(a, value);
                    if (m - 1 == value - a) {
                        cnt++;
                    }
                } else {
                    map.put(a, a);
                    if (m == 1) {
                        cnt++;
                    }
                }
            } else if (ceilingKey == null) {
                int value = map.get(floorKey);
                if (value == a - 1) {
                    if (m - 1 == value - floorKey) {
                        cnt--;
                    }
                    map.put(floorKey, a);
                    if (m - 1 == a - floorKey) {
                        cnt++;
                    }
                } else {
                    map.put(a, a);
                    if (m == 1) {
                        cnt++;
                    }
                }
            } else {
                int value1 = map.get(floorKey);
                int value2 = map.get(ceilingKey);
                if (value1 == a-1 && ceilingKey == a+1) {
                    
if (m - 1 == value1 - floorKey) {
                        cnt--;
                    }
                    if (m - 1 == value2 - ceilingKey) {
                        cnt--;
                    }
                    if (m - 1 == value2 - floorKey) {
                        cnt++;
                    }
                    map.remove(ceilingKey);
                    map.put(floorKey, value2);
                } else if (value1 == a-1) {
                    if (m - 1 == value1 - floorKey) {
                        cnt--;
                    }
                    map.put(floorKey, a);
                    if (m - 1 == a - floorKey) {
                        cnt++;
                    }
                } else if (ceilingKey == a+1) {
                    if (m - 1 == value2 - ceilingKey) {
                        cnt--;
                    }
                    map.remove(ceilingKey);
                    map.put(a, value2);
                    if (m - 1 == value2 - a) {
                        cnt++;
                    }
                } else {
                    map.put(a, a);
                    if (m == 1) {
                        cnt++;
                    }
                }
            }

            
if (cnt > 0) {
                res = index;
            }
            index++;
        }
        return res;
    }
}
