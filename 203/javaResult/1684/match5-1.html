class Solution {
    public int findLatestStep(int[] arr, int m) {
        if (arr.length == m) {
            return m;
        }
        TreeSet<Integer> set = new TreeSet<>();
        set.add(0);
        set.add(arr.length + 1);
        m++;
        for (int i = arr.length - 1; i >= 0; i--) {
            int floor = set.floor(arr[i]);
            if (arr[i] - floor == m) {
                return i;
            }
            int cel = set.ceiling(arr[i]);
            if (cel - arr[i] == m) {
                return i;
            
}
            set.add(arr[i]);
        }
        return -1;
    }
}
