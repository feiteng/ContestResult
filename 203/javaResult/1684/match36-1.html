class Solution {
    public int findLatestStep(int[] arr, int m) {
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer, Integer> map2 = new HashMap<>();
        int n = arr.length;
        
int res = 0;
        for (int i = 0; i < n; i++) {
            int index = arr[i];
            int left = map1.getOrDefault(index - 1, 0);
            int right = map1.getOrDefault(index + 1, 0);
            int len = left + right + 1;
            
            
map1.put(index, len);
            map1.put(index - left, len);
            map1.put(index + right, len);
            
            map2.put(len, map2.getOrDefault(len, 0) + len);
            map2.put(left, map2.getOrDefault(left, 0) - left);
            map2.put(right, map2.getOrDefault(right, 0) - right);
            
            if (map2.getOrDefault(m, 0) > 0) {
                res = i + 1;
            }
        }
        return res == 0 ? -1 : res;
    }
}
