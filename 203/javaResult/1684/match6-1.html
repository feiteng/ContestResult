class Solution {
    public int findLatestStep(int[] arr, int m) {
        if (m == arr.length) {
            return m;
        }
        TreeSet<Integer> removed = new TreeSet<>();
        removed.add(0);
        removed.add(arr.length + 1);
        m += 1;
        
for (int i = arr.length - 1; i >= 0; i--) {
            int toRemoved = arr[i];
            int ceiling = removed.ceiling(toRemoved);
            int floor = removed.floor(toRemoved);
            if (ceiling - toRemoved == m || toRemoved - floor == m) {
                return i;
            
}
            removed.add(toRemoved);
        }
        
        return -1;
    }
}
