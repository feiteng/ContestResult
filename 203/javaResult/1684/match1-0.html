import java.util.TreeSet;

class Solution {
    public int findLatestStep(int[] arr, int m) {

        if (m == arr.length) {
            return arr.length;
        }

        TreeSet<Integer> set = new TreeSet<>();
        set.add(0);
        set.add(arr.length + 1);
        for (int i = arr.length - 1; i >= 0; i--) {
            int removed = arr[i];

            if (set.lower(removed) == removed - m - 1) {
                return i;
            }
            if (set.higher(removed) == removed + m + 1) {
                return i;
            }
            set.add(removed);
        }
        return -1;

    }
}
