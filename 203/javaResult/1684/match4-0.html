class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (m == n) {
            return n;
        }
        
        TreeSet<Integer> treeSet = new TreeSet<>();
        treeSet.add(0);
        treeSet.add(n + 1);
        
        for (int i = n - 1; i >= 1; i--) {
            treeSet.add(arr[i]);
            int higher = treeSet.higher(arr[i]);
            // System.out.println(higher + ", " + arr[i] + ", " + m);
            if (higher - arr[i] - 1 == m) {
                return i;
            }
            int lower = treeSet.lower(arr[i]);
            if (arr[i] - lower - 1 == m) {
                return i;
            }
        }
        return -1;
    }
}
