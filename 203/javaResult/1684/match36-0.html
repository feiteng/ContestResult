class Solution {
    public int findLatestStep(int[] arr, int m) {
        Map<Integer, Integer> s2e = new HashMap<>();
        Map<Integer, Integer> e2s = new HashMap<>();
        
        Map<Integer, Integer> len2Cnt = new HashMap<>();
        
        int latest = -1;
        for (int i=1; i<=arr.length; i++) {
            int a = arr[i-1];
            
            Integer prev = e2s.get(a-1);
            Integer next = s2e.get(a+1);
            
            if (prev == null && next == null) {
                s2e.put(a, a);
                e2s.put(a, a);
                
                len2Cnt.put(1, len2Cnt.getOrDefault(1, 0) + 1);
            } else if (prev == null) {
                s2e.remove(a+1);
                s2e.put(a, next);
                e2s.put(next, a);
                
                len2Cnt.put(next - a, len2Cnt.getOrDefault(next - a, 0) - 1);
                len2Cnt.put(next - a + 1, len2Cnt.getOrDefault(next - a + 1, 0) + 1);
            } else if (next == null) {
                e2s.remove(a-1);
                e2s.put(a, prev);
                s2e.put(prev, a);
                
                if (a - prev +1 == m) latest = i;
                len2Cnt.put(a - prev, len2Cnt.getOrDefault(a - prev, 0) - 1);
                len2Cnt.put(a - prev +1, len2Cnt.getOrDefault(a - prev +1, 0) + 1);
            } else {
                s2e.remove(a+1);
                e2s.remove(a-1);
                s2e.put(prev, next);
                e2s.put(next, prev);
                
                len2Cnt.put(next - a, len2Cnt.getOrDefault(next - a, 0) - 1);
                len2Cnt.put(a - prev, len2Cnt.getOrDefault(a - prev, 0) - 1);
                len2Cnt.put(next - prev + 1, len2Cnt.getOrDefault(next - prev + 1, 0) + 1);
            }
            
            if (len2Cnt.getOrDefault(m, 0) > 0) latest = i;
        }
        
        return latest;
    }
}
