class Solution {
    public int findLatestStep(int[] arr, int m) {
        //two maps one set; need section object:len, beginning, end
        Map<Integer, Section> map1 = new HashMap<>();
        Map<Integer, Section> map2 = new HashMap<>();
        Map<Integer, Integer> map3 = new HashMap<>();
        Set<Section> set = new HashSet<>();
        int ans = -1;
        for(int i = 0; i < arr.length; i++) {
            int tempLen = 1;
            int tempbegin = arr[i];
            int tempend = arr[i];
            if(map1.containsKey(arr[i] + 1)) {
                Section pre1 = map1.get(arr[i] + 1);
                tempLen += pre1.len;
                tempend = pre1.end;
                map1.remove(pre1.begin);
                map2.remove(pre1.end);
                set.remove(pre1);
                map3.put(pre1.len, map3.get(pre1.len) - 1);
                //System.out.println("removed begin is " + pre1.begin + " end is " + pre1.end);
            }
            if(map2.containsKey(arr[i] - 1)) {
                Section pre2 = map2.get(arr[i] - 1);
                tempLen += pre2.len;
                tempbegin = pre2.begin;
                map2.remove(pre2.end);
                map1.remove(pre2.begin);
                set.remove(pre2);
                map3.put(pre2.len, map3.get(pre2.len) - 1);
                //System.out.println("removed begin is " + pre2.begin + " end is " + pre2.end);
            }

            Section s = new Section(tempLen, tempbegin, tempend);
            set.add(s);
            map1.put(tempbegin, s);
            map2.put(tempend, s);
            map3.put(tempLen, map3.getOrDefault(tempLen, 0) + 1);

            //System.out.println("final begin is " + tempbegin + " end is " + tempend);
            if(map3.containsKey(m) && map3.get(m) != 0) {
                ans = i + 1;
                //System.out.println("still found length m here");
            }
            
        }
        return ans;
        
    }
}
class Section {
    int begin;
    int end;
    int len;
    public Section(int l, int b, int e) {
        this.begin = b;
        this.end = e;
        this.len = l;
    }
}
