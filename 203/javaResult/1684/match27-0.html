class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        int[] a = new int[n + 2];
        int[] l = new int[n + 2];
        int[] r = new int[n + 2];
        int ans = -1;
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            int x = arr[i];
            a[x] = 1;
            int left = l[x - 1] == 0 ? x : l[x - 1];
            int right = r[x + 1] == 0 ? x : r[x + 1];
            r[left] = right;
            l[right] = left;
            if (x - left == m) cnt--;
            if (right - x == m) cnt--;          
            if (right - left + 1 == m) cnt++;
            if (cnt > 0) ans = i + 1;
        }
        return ans;
    }
}
