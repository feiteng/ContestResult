class Solution {
    public int findLatestStep(int[] arr, int m) {
        int n = arr.length;
        if (n == m) {
            return n;
        }
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(1, n);
        for (int i = n - 1; i >= 0; i--) {
            int start = map.floorKey(arr[i]); // start <= arr[i]
            int end = map.get(start);
            int len1 = arr[i] - start;
            if (len1 == m) {
                return i;
            }
            if (len1 > 0) {
                map.put(start, arr[i] - 1);
            } else {
                map.remove(start);
            }

            int len2 = end - arr[i];
            if (len2 == m) {
                return i;
            }
            if (len2 > 0) {
                map.put(arr[i] + 1, end);
            }
        }
        return -1;
    }
}
