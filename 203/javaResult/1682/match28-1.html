class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] fq = new int[n+1];
        fq[rounds[0]]++;
        for (int i = 1; i < rounds.length; i++) {
            for (int j = rounds[i-1] % n + 1; ; j = j % n + 1) {
                fq[j]++;
                if (j == rounds[i]) {
                    break;
                }
            }
        }
        
        List<Integer> result = new ArrayList<>();
        int max = 0;
        for (int i = 1; i <= n; i++) {
            int tmp = fq[i];
            
if (tmp > max) {
                max = tmp;
                result.clear();
                result.add(i);
            } else if (tmp == max) {
                result.add(i);
            }
        }
        
        return result;
    }
}
