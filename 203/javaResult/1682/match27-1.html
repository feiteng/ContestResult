class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] count = new int[n];
        count[rounds[0]-1]++;
        int max = 1;
        for (int j = 0; j+1 < rounds.length; j++) {
            int start = rounds[j]-1;
            int end = rounds[j+1]-1;
            if (end < start) {
                end += n;
            }
            for (int i = start+1; i <= end; i++) {
                int idx = i%n;
                count[idx]++;
            }
        }
        
        
for (int i = 0; i < n; i++) {
            max = Math.max(max, count[i]);
        }
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (count[i] == max) {
                ans.add(i+1);
            }
        }
        return ans;
        
        
    }
}
