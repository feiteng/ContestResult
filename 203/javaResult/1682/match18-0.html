class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        List<Integer> ans = new ArrayList<>();
        int[] res = new int[n];
        res[rounds[0] - 1]++;
        for(int i = 1; i < rounds.length; i++) {
            int start = rounds[i - 1], end = rounds[i] - 1;
            while(start != end) {
                res[start % n]++;
                start++;
                start = start % n;
            }
            res[end]++;
        }
        int max = 0;
        for(int i : res) {
            max = Math.max(max, i);
        }
        for(int i = 0; i < n; i++) {
            if(res[i] == max) {
                ans.add(i + 1);
            }
        }
        return ans;
    }
}
