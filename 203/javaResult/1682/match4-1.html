class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] fre = new int[n + 1];
        
        for(int i = 1; i < rounds.length; i++) {
            int p = rounds[i-1];
            // offset duplicate compute
            while(p != rounds[i]) {
                fre[p++]++;
                if(p == n + 1) p = 1;
            
}
        }
        fre[rounds[rounds.length - 1]]++; // end point is not counted
        // System.out.println(Arrays.toString(fre));
        int max = 0;
        for(int nn : fre) max = Math.max(max, nn);
        List<Integer> res = new ArrayList<>();
        for(int i = 1; i <= n; i++) if(fre[i] == max) res.add(i);
        
        return res;
    }
}
