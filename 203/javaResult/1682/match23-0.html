class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] count = new int[n+1];
        for(int i = 0; i < rounds.length - 1; i++) {
            int cur = rounds[i];
            while(cur % (n+1) != rounds[i+1]) {
                count[cur % (n+1)]++;
                cur++;
            }
        }
        count[rounds[rounds.length-1]]++;
        int max = -1;
        for(int i = 1; i < count.length; i++)  max  = Math.max(max, count[i]);
        List<Integer> res = new ArrayList<>();
        for(int i = 1; i < count.length; i++)   if(count[i] == max) res.add(i);
        // for(int i = 1; i < count.length; i++) System.out.print(count[i]+" ");
        // System.out.println();
        return res;

    }
}
