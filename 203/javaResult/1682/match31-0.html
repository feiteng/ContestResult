class Solution {
    public List<Integer> mostVisited(int n, int[] rs) {

        int rd[] = new int[n+1];
        int len = rs.length;
        rd[rs[0]]++;
        for(int i=1;i<len;++i){
            int p = rs[i-1]+1;
            if(p==n+1){
                p =1;
            }
            
            int q = rs[i];
            
            while(p!=q){
                rd[p]++;
                p++;
                if(p==n+1){
                    p = 1;
                }
            }
            rd[q]++;
            
        }
        int mx = 0;
        for(int i=1;i<=n;++i){
            mx = Math.max(mx,rd[i]);
        }
        List<Integer> l = new ArrayList<>();
        
         for(int i=1;i<=n;++i){
          if(rd[i]==mx){
              l.add(i);
          }   
             
         }
        return l;
        
    }
}
