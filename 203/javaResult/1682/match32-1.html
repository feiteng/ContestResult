class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] count = new int[n];
        count[rounds[0]-1]++;
        for(int i = 1;i<rounds.length;i++){
            int start  = rounds[i-1]-1;
            
for(int j = 1;j<=(n+rounds[i]-rounds[i-1])%n;j++){
                count[(start + j)%n]++;
            }
        }
        List<Integer> res = new ArrayList<>();
        int max = -1;
        for(int i = 0;i<count.length;i++){
            if(count[i] > max){
                res.clear();
                
res.add(i+1);
                max = count[i];
            }else if(count[i] == max){
                res.add(i+1);
            }
        }
        return res;
    }
}
