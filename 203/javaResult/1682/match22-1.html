import java.util.ArrayList;
import java.util.List;

class Solution {
	public List<Integer> mostVisited(int n, int[] rounds) {
    	int[] arr = new int[n+1];
    	for (int i=1; i<rounds.length; i++) {
    		int l = rounds[i-1];
    		int r = rounds[i];
    		if (l <= r) {
    			for (int j=l; j<r; j++) {
    				arr[j] ++;
    			}
    		} else {
    			for (int j=l; j<=n; j++) {
    				arr[j] ++;
    			}
    			for (int j=1; j<r; j++) {
    				arr[j] ++;
    			}
    		}
    		if (i == rounds.length-1) {
    			arr[r] ++;
    		}
    	}

    	int max = 0;
    	for (int x : arr) {
    		max = Math.max(max, x);
    	}
    	List<Integer> list = new ArrayList<Integer>();
    	for (int i=1; i<=n; i++) {
    		if (arr[i] == max) {
    			list.add(i);
    		}
    	}
    	return list;
    }
}
