class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] visited = new int[n + 1];
        for (int i = 0; i < rounds.length - 1; i++) {
            int start = rounds[i];
            int end = rounds[i + 1];
            int addition = 0;
            if (end < start) {
                addition = end;
                end = n;
            }     
            for (int j = start + 1; j <= end; j++) {
                visited[j]++;
            }
            for (int j = 1; j <= addition; j++) {
                visited[j]++;
            }
        }
        visited[rounds[0]]++;
        List<Integer> res = new ArrayList<>();
        int max = 0;
        for (int i = 1; i <= n; i++) {
            if (visited[i] > max) {
                res = new ArrayList<>();
                res.add(i);
                max = visited[i];
            } else if (visited[i] == max) {
                res.add(i);
            }
            
        }
        return res;
        
    }
}
