class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        
        int cnt[] = new int[n + 1];
        int len = rounds.length;
        for(int i =0; i < len - 1; i++){
            int s = rounds[i];
            int t = rounds[i + 1];
            while(s != t){
                cnt[s]++;
                s++;
                if(s > n){
                    s = 1;
                }
            }
        }
        cnt[rounds[len - 1]]++;
        
        int max = 0;
        for(int i = 1; i <= n; i++){
            max = Math.max(max, cnt[i]);
            // System.out.println(i + " " + cnt[i]);
        }
        List<Integer> result = new ArrayList();
        for(int i = 1; i <= n; i++){
            if(cnt[i] == max){
                result.add(i);
            }
        }
        
        return result;
 
    }
}
