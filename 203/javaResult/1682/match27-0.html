class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] counts = new int[n];
        
        int m = rounds.length;
        counts[rounds[0] - 1]++;
        
        for (int i = 1; i < m; i++) {
            int start = rounds[i - 1] - 1;
            int end = rounds[i] - 1;
            if (end <= start) {
                end += n;
            }
            
            for (int j = start + 1; j <= end; j++) {
                counts[j % n]++;
            }
        }
        
        int max = 0;
        for (int i = 0; i < n; i++) {
            max = Math.max(max, counts[i]);
        }
        
        List<Integer> rst = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (counts[i] == max) {
                rst.add(i + 1);
            }
        }
        
        return rst;
    }
}
