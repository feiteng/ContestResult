class Solution {
public List<Integer> mostVisited(int n, int[] rounds) {

        int[] count = new int[n];
        count[rounds[0]-1]++;
        for(int i = 0; i < rounds.length-1; ++i){
            for(int j = rounds[i]%n;;j =(j+1)%n){
                count[j]++;
                if(j == rounds[i+1]-1){
                    break;
                }
            }

        }

        int max = 0;
        List<Integer> res = new LinkedList<>();

        for(int i = 0; i < n; ++i){
            if(count[i] > max){
                max = count[i];
                res.clear();
                res.add(i+1);
            }else if(count[i] == max){
                res.add((i+1));
            }
        }

        return res;

    }
}
