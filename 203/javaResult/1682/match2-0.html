class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int last = rounds[rounds.length - 1], start = rounds[0];
        List<Integer> res = new ArrayList<>();
        if(start <= last) {
            for(int i = start; i <= last; i++) res.add(i);
        } else {
            for(int i = 1; i <= last; i++) res.add(i);
            for(int i = start; i <= n; i++) res.add(i);
        }
        
        return res;
    }
}
