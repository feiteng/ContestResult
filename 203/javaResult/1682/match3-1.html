class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] freqs = new int[n+1];
        for (int i = 0; i < rounds.length-1; i++) {
            int start = rounds[i];
            while (start != rounds[i+1]) {
                freqs[start++]++;
                if (start > n) start = 1;
            }
        }
        freqs[rounds[rounds.length-1]]++;
        int maxFreq = 0;
        for (int x : freqs) maxFreq = Math.max(maxFreq, x);
        List<Integer> ret = new ArrayList();
        for (int i = 1; i <= n; i++)
            if (freqs[i] == maxFreq) ret.add(i);
        return ret;
    }
}
