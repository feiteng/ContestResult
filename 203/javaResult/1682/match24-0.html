class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] v = new int[n + 1];
        int l = rounds[0];
        ++v[l];
        for (int i = 1; i < rounds.length; ++i) {
            while (l != rounds[i]) {
                ++l;
                if (l > n) {
                    l %= n;
                }
                ++v[l];
            }
        }
        int max = 0;
        for (int i = 1; i <= n; ++i) {
            max = Math.max(max, v[i]);
        }
        List<Integer> res = new LinkedList<>();
        for (int i = 1; i <= n; ++i) {
            if (v[i] == max) {
                res.add(i);
            }
        }
        return res;
    }
}
