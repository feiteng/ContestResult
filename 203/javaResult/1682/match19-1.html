class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] freq = new int[n + 1];
        for(int i = 0; i < rounds.length - 1; i++){
            int src = rounds[i];
            while(src != rounds[i + 1]){
                freq[src]++;
                src = src % n + 1; 
            
}
        }
        freq[rounds[rounds.length - 1]]++;

        int maxFreq = 0;
        for(int i = 1; i <= n; i++){
            maxFreq = Math.max(maxFreq, freq[i]);
        }
        
        List<Integer> res = new ArrayList<>();
        for(int i = 1; i <= n; i++){
            if(freq[i] == maxFreq){
                res.add(i);
            }
        }
        
        return res;
    }
}
