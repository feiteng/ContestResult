import java.util.ArrayList;
import java.util.List;

class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] arr = new int[n+1];
        arr[rounds[0]]++;
        for(int i=0;i<rounds.length-1;i++){
            int t1 = rounds[i];
            int t2 = rounds[i+1];
            if(t1 < t2){
                for(int j=t1+1;j<=t2;j++){
                    arr[j]++;
                }
            }else {
                for(int j=t1+1;j<=n;j++){
                    arr[j]++;
                }
                for(int j=1;j<=t2;j++){
                    arr[j]++;
                }
            }
        }
        int max = 0;
        for(int t:arr){
            max = Math.max(max, t);
        }
        List<Integer> list = new ArrayList<>();
        for(int i=1;i<=n;i++){
            if(arr[i] == max){
                list.add(i);
            }
        }
        return list;
    }
}
