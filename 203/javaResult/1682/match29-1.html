class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int start = rounds[0];
        int len = rounds.length;
        int end = rounds[len - 1];
        List<Integer> res = new ArrayList<>();
        if(start <= end){
            for(int i = start; i <= end; i++){
                res.add(i);
            }
        }else{
            for(int i = start; i <= n; i++){
                res.add(i);
            }
            for(int i = 1; i <= end; i++){
                res.add(i);
            }
        }
        Collections.sort(res);
        return res;
        
    }
}
