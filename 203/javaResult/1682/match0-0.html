class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int l = rounds[0];
        int r = rounds[rounds.length-1];
        List<Integer> ans = new ArrayList<>();
        if (l <= r) {
            for (int i = l; i <= r; i++)
                ans.add(i);
        } else {
            for (int i = 1; i <= r; i++)
                ans.add(i);
            for (int i = l; i <= n; i++)
                ans.add(i);
        }
        return ans;
    }
}
