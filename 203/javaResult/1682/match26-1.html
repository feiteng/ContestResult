class Solution {
    public List<Integer> mostVisited(final int N, int[] rounds) {
        int[] map = new int[N + 1];
        for (int i = 1; i < rounds.length; ++i) {
            for (int j = rounds[i - 1]; j != rounds[i]; ) {
                map[j++]++;
                if (j > N) {
                    j = 1;
                }
            
}
        }
        map[rounds[rounds.length - 1]]++;

        int maxCnt = 0;
        for (int i = 1; i <= N; ++i) {
            maxCnt = Math.max(maxCnt, map[i]);
        }

        List<Integer> res = new ArrayList<>();
        for (int i = 1; i <= N; ++i) {
            if (map[i] == maxCnt) {
                res.add(i);
            }
        }

        return res;
    }
}
