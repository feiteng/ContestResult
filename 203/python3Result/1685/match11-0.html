class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        presum = [stoneValue[0]]
        for sv in stoneValue[1:]:
            presum.append(presum[-1] + sv)
        presum.append(0)
        
        @lru_cache(None)
        def find(i, j):
            if i == j:
                return 0
            res = 0
            for sp in range(i, j):
                left = presum[sp] - presum[i-1]
                right = presum[j] - presum[sp]
                if left > right:
                    candidate = right + find(sp+1, j)
                elif left < right:
                    candidate = left + find(i, sp)
                else:
                    candidate = max(find(sp+1, j), find(i, sp)) + left
                res = max(candidate, res)
            return res
        return find(0, len(stoneValue)-1)
