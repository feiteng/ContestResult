class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        n = len(stoneValue)
        prefix_sum = [0] * (n + 1)
        
        for i in range
(1, n + 1):
            prefix_sum[i] = prefix_sum[i - 1] + stoneValue[i - 1]
        
        @lru_cache(None)
        def dp(l, r):
            if l == r:
                return 0
            
            score = 0
            for l_end in range(l, r):
                
                left_score = prefix_sum[l_end + 1] - prefix_sum[l]
                right_score = prefix_sum[r + 1] - prefix_sum[l_end + 1]
                
                if left_score == right_score:
                    score = max(score, dp(l, l_end) + left_score, dp
(l_end + 1, r) + right_score)
                elif left_score > right_score:
                    score = max(score, dp(l_end + 1, r) + right_score)
                else:
                    score = max(score, dp(l, l_end) + left_score)            
                
            return score
        
        return dp(0, n - 1)
