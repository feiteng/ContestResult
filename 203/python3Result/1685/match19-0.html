class Solution:
    def stoneGameV(self, s: List[int]) -> int:
        n = len(s)
        presum = [0] * (n + 1)
        presum[0] = 0
        presum[1] = s[0]
        for i in range(2, n + 1):
            presum[i] = presum[i-1] + s[i - 1]

        @functools.lru_cache(None)
        def dfs(l, r): # max score alice can get afterwards when remaining is l to r 
            if l >= r:
                return 0 
            maxx = 0 
            for right in range(l, r):
                lft = presum[right + 1] - presum[l]
                rt = presum[r + 1] - presum[right + 1]
                if lft > rt:
                    maxx = max(maxx, rt + dfs(right + 1, r))
                elif lft < rt:
                    maxx = max(maxx, lft + dfs(l, right))
                else:
                    maxx = max(maxx, lft + max(dfs(l, right), dfs(right + 1, r)))
            return maxx 
        return dfs(0, n - 1)
