from sys import setrecursionlimit


sys.setrecursionlimit(1000000)


class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        stones = stoneValue
        p = [0]
        for x in stones:
            p.append(p[-1] + x)

        @lru_cache(None)
        def dp(i, j):
            if j - i == 1:
                return 0
            best = -math.inf
            for k in range(i + 1, j): # split at k, with k becoming part of right
                left = p[k] - p[i]
                right = p[j] - p[k]
                if left > right: # Bob throws away left, alice gets right
                    best = max(best, right + dp(k, j))
                elif left < right: # Bob throws away right, alice gets left
                    best = max(best, left + dp(i, k))
                else:
                    best = max(best, left + dp(i, k), right + dp(k, j))
            return best

        return dp(0, len(stones))
