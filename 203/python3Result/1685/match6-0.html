class Solution:
    def stoneGameV(self, stoneValue):
        s = [0]
        for val in stoneValue:
            s.append(s[-1] + val)

        # form i to j inclusive = s[j + 1] - s[i]

        @functools.lru_cache(None)
        def dp(i, j):
            if i == j:
                return 0
            ans = 0
            for cut in range(i, j):
                s1 = s[cut + 1] - s[i]
                s2 = s[j + 1] - s[cut + 1]
                if s1 < s2:
                    ans = max(ans, s1 + dp(i, cut))
                elif s1 > s2:
                    ans = max(ans, s2 + dp(cut + 1, j))
                else:
                    ans = max(ans, s1 + max(dp(i, cut), dp(cut + 1, j)))

            return ans

        return dp(0, len(stoneValue) - 1)
