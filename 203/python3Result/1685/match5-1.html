from functools import lru_cache

class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        tmppresum = [0]
        for i in stoneValue:
            tmppresum.append(tmppresum[-1]+i)
        @lru_cache(None)
        def dfs(i, j):
            if i == j:
                return 0
            tmpans = []
            for p in range(i, j):
                tmpleftans = tmppresum[p+1] - tmppresum[i] 
                tmprightans = tmppresum
[j+1] - tmppresum[p+1] 
                if tmpleftans > tmprightans:
                    tmpans.append(dfs(p+1, j)+tmprightans)
                elif tmpleftans < tmprightans:
                    tmpans.append(dfs(i, p)+tmpleftans)
                else:
                    tmpans.append(max(dfs(i, p), dfs(p+1, j))+tmpleftans)
            return max(tmpans)
        return dfs(0, len(stoneValue)-1)
