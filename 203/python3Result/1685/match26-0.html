class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        @lru_cache(None)
        def dfs(left, right):
            if left >= right:
                return 0
            total_sum = sum(stoneValue[left:right+1])
            cum_sum = 0
            res = 0
            for i in range(left, right):
                cum_sum += stoneValue[i]
                if cum_sum * 2 < total_sum:
                    tmp = dfs(left, i) + cum_sum
                elif cum_sum * 2 > total_sum:
                    tmp = dfs(i+1, right) + total_sum - cum_sum
                else:
                    tmp = max(dfs(i+1, right), dfs(left, i)) + cum_sum
                res = max(res, tmp)
            return res
        return dfs(0, len(stoneValue) - 1)
