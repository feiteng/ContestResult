class Solution:
    def stoneGameV(self, s: List[int]) -> int:
        n=len(s)
        pre=[0]
        for i in range(n):
            pre.append(pre[-1]+s
[i])
        #print(pre)
        @lru_cache(None)
        def dp(l,r):
            if l==r:
                return 0
            res=0
            for i in range(l,r):
                lc=pre[i+1]-pre[l]
                rc=pre[r+1]-pre[i+1]
                if lc>rc:
                    res=max(res,rc+dp(i+1,r))
                elif lc<rc:
                    res=max(lc+dp(l,i),res)
                else:
                    res=max(res,lc+max(dp(l,i),dp(i+1,r)))
            return res
        return dp(0,n-1)
