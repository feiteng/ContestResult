class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        presum 
= [0]
        for v in stoneValue:
            presum.append(presum[-1] + v)
        @lru_cache(None)
        def dfs(lo, hi):
            if lo == hi - 1:
                return 0
            cur = 0
            for m in range(lo+1, hi): #[lo, m), [m, hi)
                l = presum[m] - presum[lo]
                r = presum[hi] - presum[m]
                if l < r:
                    cur = max(cur, l+dfs(lo, m))
                elif l > r:
                    cur = max(cur, r+dfs(m, hi))
                else:
                    cur = max(cur, l+max(dfs(lo, m), dfs(m, hi)))
            return cur
        return dfs(0, len(stoneValue))
