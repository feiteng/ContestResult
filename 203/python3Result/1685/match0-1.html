class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        ps = [0]
        
        for n in stoneValue:
            ps.append(ps[-1] + n)
        
        mem = {}
        def dp(i, j):
            # one stone left
            if i >= j:
                return 0
            
            if (i,j) in mem:
                return mem[i,j]
            
            res = 0
            ## take at least one stone at i, until j-1
            for mid in range(i, j):
                leftsum = ps[mid+1]-ps[i]
                rightsum = ps[j+1]-ps[mid+1]
                if leftsum > rightsum:
                    res = max(res, rightsum + dp(mid+1, j))
                
                elif leftsum < rightsum:
                    res = max(res, leftsum + dp(i, mid))
                
                else:
                    res = max(res, leftsum + max(dp(i, mid), dp(mid+1, j)))
            mem[i,j] = res
            return res    
                
        return dp(0, len(stoneValue)-1)
