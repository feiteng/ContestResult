class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        
        # presum[i] means res for arr[:i]
        presum = [0]
        for val in stoneValue:
            presum+=[presum
[-1]+val]
        n = len(stoneValue)
        # return the max score A can get,[i,j)
        @functools.lru_cache(None)
        def dfs(i,j):
            if i==j:
                return 0
            
            res = 0
            for k in range(i,j+1):
                # []
                s1,s2 = presum[k]-presum[i],presum
[j]-presum[k]
                if s1<s2:
                    res = max(res,s1+dfs(i,k))
                elif s1>s2:
                    res = max(res,s2+dfs(k,j))
                else:
                    res = max(res,s2+dfs(k,j),s1+dfs(i,k))
            return res
        return dfs(0,n)
