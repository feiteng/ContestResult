class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        prefix 
= [0]
        for v in stoneValue: prefix.append(v + prefix[-1])
        @lru_cache(None)
        def helper(i, j):
            if i == j:
                return 0
            ret = 0
            for k in range(i, j):
                left = prefix[k+1] - prefix[i]
                right = prefix[j+1] - prefix[k + 1]
                #print(i, j, k, left, right)
                if right > left:
                    ret = max(ret, helper(i, k) + left)
                elif right < left:
                    ret = max(ret, helper(k + 1, j) + right)
                else:
                    ret = max(ret, max(helper(i, k), helper(k + 1, j)) + right)
            return ret
        
        return helper(0, len(stoneValue) - 1)
