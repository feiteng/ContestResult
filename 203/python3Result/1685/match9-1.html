class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        prefix=[0]
        for v in stoneValue:
            prefix.append(prefix[-1]+v)
        @functools.lru_cache(None)
        def dp(start,end):
            if start==end:
                return 0
            res=0
            for i in range(start,end):
                s1=prefix[i+1]-prefix[start]
                s2=prefix[end+1]-prefix[i+1]
                if s1>s2:
                    cur=s2+dp(i+1,end)
                elif s2>s1:
                    cur=s1+dp(start,i)
                else:
                    cur=max(dp(i+1,end),dp
(start,i))+s1
                res=max(res,cur)
            return res
        return dp(0,len(stoneValue)-1)
