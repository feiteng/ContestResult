class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        @lru_cache(None)
        def solve(b, e):
            if b >= e - 1:
                return 0
            x = sum(stoneValue[b:e])
            left = 0
            right = sum(stoneValue[b:e])
            ans = 0
            for p in range(b + 1, e):
                left += stoneValue[p - 1]
                right -= stoneValue[p - 1]
                
                temp = min(left, right)
                
if left > right:
                    temp += solve(p, e)
                elif left < right:
                    temp += solve(b, p)
                else:
                    temp += max(solve(b, p), solve(p, e))
                ans = max(ans, temp)
            return ans
        return solve(0, len(stoneValue))
