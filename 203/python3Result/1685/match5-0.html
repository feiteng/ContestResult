class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        
        prefix = [0]
        for s in stoneValue:
            prefix.append(prefix[-1] + s)
        
        @functools.lru_cache(None)
        def dfs(i, j):
            # return the Alice can get
            if i == j:
                return 0
            cand = []
            for k in range(i, j):
                left, right = prefix[k+1] - prefix[i], prefix[j+1] - prefix[k+1]
                if left > right:
                    cand.append(dfs(k+1, j) + right)
                elif right > left:
                    cand.append(dfs(i, k) + left)
                else:
                    cand.append(max(dfs(i, k) + left, dfs(k+1, j) + right))
            return max(cand)
        return dfs(0, len(stoneValue) - 1)
