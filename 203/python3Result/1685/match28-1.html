class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        acc = list(itertools.accumulate(stoneValue, initial=0))
        @functools.lru_cache(None)
        
def dp(i, j):
            if i == j:
                return 0
            best = 0
            for k in range(i + 1, j + 1):
                left = acc[k] - acc[i]
                right = acc[j + 1] - acc[k]
                if left > right:
                    best = max(best, right + dp(k, j))
                elif right > left:
                    best = max(best, left + dp(i, k - 1))
                else:
                    best = max(best, right + dp(k, j), left + dp(i, k - 1))
            # print(i, j, best)
            return best
        return dp(0, len(stoneValue) - 1)
