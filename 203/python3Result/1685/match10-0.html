class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        sums = [0]
        
        for v in stoneValue:
            sums.append(sums[-1] + v)
        
        @lru_cache(None)
        def solve(i, j):
            if i == j:
                return 0
            if i + 1 == j:
                return min(stoneValue[i:i+2])
            
            max_score = 0
            for border in range(i+1, j+1):
                left_value = sums[border] - sums[i]
                right_value = sums[j+1] - sums[border]
                
                if left_value == right_value:
                    score = left_value + max(solve(i, border-1), solve(border, j))
                if left_value < right_value:
                    score = left_value + solve(i, border-1)
                if left_value > right_value:
                    score = right_value + solve(border, j)
                
                max_score = max(max_score, score)
            
            return max_score
        
        return solve(0, len(stoneValue)-1)
