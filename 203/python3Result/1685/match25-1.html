class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        presums = [0] + list(accumulate(stoneValue))
        n = len(stoneValue)
        
@lru_cache(None)
        def dp(l, r):
            if l == r:
                return 0
            ans = 0
            for i in range(l, r):
                ls = presums[i+1] - presums[l]
                rs = presums[r+1] - presums[i+1]
                if ls < rs:
                    ans = max(ans, ls + dp(l, i))
                elif ls > rs:
                    ans = max(ans, rs + dp(i+1, r))
                else:
                    ans = max(ans, ls + dp(l, i), rs + dp(i+1, r))
            return ans
        return dp(0, n-1)
