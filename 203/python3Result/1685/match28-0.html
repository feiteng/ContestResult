from functools import lru_cache

class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        
        pre = [0]
        for s in stoneValue:
            pre.append(pre[-1] + s)
        
        @lru_cache(maxsize=None)
        def search(i, j):
            if i >= j: return 0
            
            ret = 0
            for k in range(i, j):
                ls = pre[k+1] - pre[i]
                rs = pre[j+1] - pre[k+1]
                if ls > rs:
                    ret = max(ret, search(k+1, j) + rs)
                elif ls < rs:
                    ret = max(ret, search(i, k) + ls)
                else:
                    # same
                    ret = max(ret, search(i, k) + ls, search(k+1, j) + ls)
            return ret
            
            
        return search(0, len(stoneValue)-1)
