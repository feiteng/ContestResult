class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        # dp[i][j] 表示i,j闭区间内 alice能获得的最大值
        sumstone= [0]
        for i in stoneValue:
            sumstone.append(sumstone[-1]+i)
        # print(sumstone)
        n = len(stoneValue)
        
        @lru_cache(None)
        def dfs(i,j):
            if i==j:
                return 0
            
            res = 0
            for k in range(i,j):
                l = sumstone[k+1]-sumstone[i]
                r = sumstone[j+1]-sumstone[k+1]
                if l>r:
                    res = max(res,dfs(k+1,j) + r )
                elif r>l:
                    res= max(res,dfs(i,k) + l)
                else:
                    res = max(res,dfs(k+1,j) + r, dfs(i,k) + l)
            return res
        return dfs(0,n-1)
