class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        
        n = len(stoneValue)
        S = [0]*(n+1)
        for i in range(1, len(S)):
            S[i] = S[i-1] + stoneValue[i-1]
        
        @lru_cache(None)
        def dp(i, j):
            if i >= j:
                return 0
            
            res = 0
            for k in range(i, j):
                left = S[k+1]-S[i]
                right = S[j+1]-S[k+1]
                #print(left, right)
                if left > right:
                    res = max(res, right + dp(k+1, j))
                elif left < right:
                    res = max(res, left + dp(i, k))
                else:
                    res = max(res, left + dp(i, k))
                    res = max(res, left + dp(k+1, j))
            return res
        return dp(0, n-1)
