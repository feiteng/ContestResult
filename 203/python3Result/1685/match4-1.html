class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        presum = [0]
        for num in stoneValue:
            presum.append(presum[-1] + num)
        
        def ssum(i, j):
            return presum[j+1] - presum[i]
        
        @lru_cache(None)
        
def dp(l, r):
            if r <= l:
                return 0
            ans = 0
            for k in range(l, r):
                left = ssum(l, k)
                right = ssum(k + 1, r)
                if left < right:
                    ans = max(ans, left + dp(l, k))
                elif left > right:
                    ans = max(ans, right + dp(k + 1, r))
                else:
                    ans = max(ans, left + dp(l, k), right + dp(k + 1, r))
            return ans
        
        
        return dp(0, len(stoneValue) - 1)
