class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        ps = [0]
        for n in stoneValue:
            ps.append(ps[-1] + n)
            
        mem = {}
        
        def dp(i, j):
            if i >= j:
                return 0
            if (i, j) in mem:
                return mem[i, j]
            res = 0
            for k in range(i, j):
                sum_left = ps[k+1] - ps[i]
                sum_right = ps[j+1] - ps[k+1]
                if sum_left > sum_right:
                    res = max(res, dp(k+1, j) + sum_right)
                elif sum_left < sum_right:
                    res = max(res, dp(i, k) + sum_left)
                else:
                    res = max(res, max(dp(i, k), dp(k+1, j)) + sum_left)
            mem[i, j] = res
            return res
        
        return dp(0, len(stoneValue) - 1)
