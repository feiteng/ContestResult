class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        pre 
= [0]
        for v in stoneValue:
            pre.append(pre[-1] + v)
        @functools.lru_cache(None)
        def dp(i, j):
            if j - i == 1:
                return 0
            m = 0
            for k in range(i + 1, j):
                l, r = pre[k] - pre[i], pre
[j] - pre[k]
                if l < r:
                    m = max(m, l + dp(i, k))
                elif l > r:
                    m = max(m, r + dp(k, j))
                else:
                    m = max(m, l + dp(i, k), r + dp(k, j))
            return m
        return dp(0, len(pre) - 1)
