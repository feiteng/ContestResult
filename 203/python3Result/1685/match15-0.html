class Solution:
    def stoneGameV(self, v: List[int]) -> int:
        a=list(accumulate(v,initial=0))
        sys.setrecursionlimit(10**7)
        # print(a)
        @lru_cache(None)
        def dfs(left,right):
            if right-left+1<=1:
                return 0
            ans=0
            for i in range(left,right+1):
                t1,t2=a[i+1]-a[left],a[right+1]-a[i+1]
                # print(left,i,right,t1,t2)
                if t1<t2:
                    ans=max(ans,t1+dfs(left,i))
                elif t1>t2:
                    ans=max(ans,t2+dfs(i+1,right))
                else:
                    ans=max([ans,t1+dfs(left,i),t2+dfs(i+1,right)])
            # print(left,right,ans)
            return ans
        return dfs(0,len(v)-1)
