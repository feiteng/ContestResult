class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        n = len(stoneValue)
        prefix = [0]
        for i, value in enumerate(stoneValue):
            prefix.append(prefix[-1] + value)
            
        @lru_cache(None)
        def dfs(be, en):
            if be == en:
                return 0
            res = 0
            for split in range(be + 1, en + 1):
                sum_left = prefix[split] - prefix[be]
                sum_right = prefix[en + 1] - prefix[split]
                if sum_left > sum_right:
                    res = max(res, sum_right + dfs(split, en))
                elif sum_left < sum_right:
                    res = max(res, sum_left + dfs(be, split - 1))
                else:
                    res = max(res, sum_left + dfs(be, split - 1), sum_right + dfs(split, en))
                                
            return res
                
                
                
        return dfs(0, n - 1)
