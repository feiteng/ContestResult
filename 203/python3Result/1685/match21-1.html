class Solution:
    def stoneGameV(self, stones: List[int]) -> int:
        n = len(stones)
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + stones[i]
        @lru_cache(None)
        def dfs(s: int, t: int) -> int:
            if s == t: return 0
            ret = 0
            for i in range(s, t):
                u, v = prefix[i + 1] - prefix[s], prefix
[t + 1] - prefix[i + 1]
                if u > v:
                    ret = max(v + dfs(i + 1, t), ret)
                elif u < v:    
                    ret = max(u + dfs(s, i), ret)
                else:    
                    ret = max(u + dfs(s, i), u + dfs(i + 1, t), ret)
            return ret            
        return dfs(0, n - 1)
