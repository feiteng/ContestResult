class Solution:
    def stoneGameV(self, arr: List[int]) -> int:
        n = len(arr)
        dp = [[0]*n for _ in range(n)]
        pre = [0]
        for i in range(n):
            pre.append(pre[-1]+arr[i])
        
        print(pre)
        # for i in range(n):
        #     for j in range(i+1,n):
        #         dp[i][j] = pre[j]-pre[i-1]
                
        ans = 0
        @lru_cache(None)
        def solve(i,j):
            if i > j:
                return 0
            # print(i,j)
            ret = 0
            for k in range(i+1,j):
                # print(k)
                x = pre[k]-pre[i]
                y = pre[j]-pre[k]
                # print('x',x,'y',y)
                if x < y:
                    ret = max(ret,x+solve(i,k))
                elif y < x:
                    ret = max(ret,y+solve(k,j))
                else:
                    ret = max(ret,x+solve(i,k),y+solve(k,j))
            # print(i,j, ret)
            return ret
        return solve(0,n)
