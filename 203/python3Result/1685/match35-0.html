class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        self.pre=[0]
        c=0
        for i in stoneValue:
            c+=i
            self.pre.append(c)
        @lru_cache(None)
        def dfs(a,b):
            if b<=a+1:
                return 0
            maxn=0
            for m in range(a+1,b):
                l=self.pre[m]-self.pre[a]
                r=self.pre[b]-self.pre[m]
                if l>r:
                    maxn=max(maxn,r+dfs(m,b))
                elif l<r:
                    maxn=max(maxn,l+dfs(a,m))
                else:
                    maxn=max(maxn,r+max(dfs(a,m),dfs(m,b)))
            return maxn
        return dfs(0,len(stoneValue))
