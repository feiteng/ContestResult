from functools import lru_cache
class Solution:
    def stoneGameV(self, stoneValue: List[int]) -> int:
        prefix = [0]
        for value in stoneValue:
            prefix.append(prefix[-1] + value)
        
        def sum_(i, j):
            return prefix[j + 1] - prefix[i]
        
        
        
        @lru_cache(maxsize=None)
        def get(i, j):

            if i == j:
                return 0
            best = 0
            for mid in range(i + 1, j + 1):
                left = sum_(i, mid - 1)
                right = sum_(mid, j)
                if left < right: 
                    best = max(best, left + get(i, mid - 1))
                elif left > right: 
                    best = max(best, right + get(mid, j))
                else:
                    best = max(best, left + get(i, mid - 1), right + get(mid, j))
            return best
            
        
        return get(0, len(stoneValue) - 1)
