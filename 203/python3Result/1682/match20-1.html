class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        d = collections.defaultdict(int)
        
        cur = rounds[0]
        d[cur] += 1
        
        for r in rounds[1:]:
            while cur != r:
                # print(cur)
                cur += 1
                if cur == n + 1:
                    cur = 1
                d[cur] += 1
                
            # d[cur] += 1
            
        
        # print(d)
        
        m = max(d.values())
        
        res = []
        
        for k in d:
            if d[k] == m:
                res.append(k)
                
        res.sort()
        
        return res
