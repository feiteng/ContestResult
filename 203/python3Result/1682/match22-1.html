class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        pool=[0 for i in range(n)]
        for m,i in enumerate(rounds[:-1]):
            
if rounds[m+1]>rounds[m]:
                for o in range(rounds[m]-1,rounds[m+1]-1):
                    pool[o]+=1
            else:

[m+1]-1):
                    pool[o%n]+=1
        pool[rounds[-1]-1]+=1
        maxn=max(pool)
        r=[]
        for m,i in enumerate(pool):
            if i==maxn:
