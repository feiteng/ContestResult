class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        d = [0]*(n+1)
        for i in range(1,len(rounds)):
            start = rounds[i-1]
            while start!=rounds[i]:
                d[start]+=1
                start+=1
                if start==n+1:
                    start=1
        d[rounds[-1]]+=1
        m = max(d)
        res = []
        for i in range(len(d)):
            if d[i]==m:
                res.append(i)
        return res
