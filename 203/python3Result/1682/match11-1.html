class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        visit = [0] * n
        visit
[rounds[0] - 1] += 1
        for i in range(len(rounds) - 1):
            if rounds[i + 1] > rounds[i]:
                for j in range(rounds[i] + 1, rounds[i + 1] + 1):
                    visit[j - 1] += 1
            else:
                for j in range(rounds[i] + 1, n + 1):
                    visit[j - 1] += 1
                for j in range(1, rounds[i + 1] + 1):
                    visit[j - 1] += 1
        maxvisit 
= max(visit)
        ans = []
        for i in range(n):
            if visit[i] == maxvisit:
                ans.append(i + 1)
        return ans
