class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        
        a = rounds[0]
        b = rounds[-1]
        
        ans = []
        if a <= b:
            for i in range(a, b+1):
                ans.append(i)
        else:
            for i in range(a, n+1):
                ans.append(i)
                
            for i in range(1, b+1):
                ans.append(i)
            
        return sorted(ans)
