class Solution:
    def mostVisited(self, n: int, r: List[int]) -> List[int]:
        cnt = [0] * (n + 1)
        m = len(r)
        for i in range
(1, m):
            if r[i-1] < r[i]:
                for j in range(r[i-1], r[i]):
                    cnt[j]+=1
            else:
                for j in range(r[i-1], r
[i] + n):
                    cnt[(j - 1) % n + 1] += 1
        cnt[r[-1]]+=1
        a = max(cnt)
        ret = []
        for i in range(1, n + 1):
            if cnt[i] == a: ret.append(i)
        return ret
