class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        cnt = collections.defaultdict(int)
        cnt
[rounds[0]]+=1
        for i in range(1, len(rounds)):
            s, e = rounds[i-1], rounds[i]
            if s < e:
                for j in range(s+1,e+1):
                    cnt[j]+=1
            else:
                for j in range(s+1, n+1):
                    cnt[j]+=1
                for j in range(1, e+1):
                    cnt[j]+=1
        mx = max(cnt.values())
        res = []
        for k in cnt:
            if cnt[k] == mx:
                res.append(k)
        return sorted(res)
