class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        cnt=[0]*n
        for i in range(1,len(rounds)):
            if rounds[i]>rounds[i-1]:
                for j in range(rounds[i-1]-1,rounds[i]-1):
                    cnt[j]+=1
            else:
                for j in range(rounds[i-1]-1,n):
                    cnt[j]+=1
                for j in range(rounds[i]-1):
                    cnt[j]+=1
        cnt[rounds[-1]-1]+=1
        max_=max(cnt)
        res=[]
        for i,c in enumerate(cnt):
            if c==max_:
                res.append(i+1)
        return sorted(res)
