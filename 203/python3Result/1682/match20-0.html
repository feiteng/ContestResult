class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        counts = {pos: 0 for pos in range(1, n+1)}
        pos = rounds[0]
        counts[pos] += 1
        for target in rounds[1:]:
            while pos != target:
                pos += 1
                if pos > n:
                    pos -= n
                counts[pos] += 1
        max_count = max(counts.values())
        return [pos for pos in range(1, n+1) if counts[pos] == max_count]
