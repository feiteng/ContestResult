class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        curr = 1
        cnt = collections.Counter()
        cnt[rounds[0]] += 1
        for i in range(1, len(rounds)):
            if rounds[i] > rounds[i-1]:
                for sec in range(rounds[i-1]+1, rounds[i]+1):
                    cnt[sec] += 1
            else:
                for sec in range(rounds[i-1]+1, n+1):
                    cnt[sec] += 1
                for sec in range(1, rounds[i]+1):
                    cnt[sec] += 1
        
        maxval = max(cnt.values())
        res = []
        for key in cnt:
            if cnt[key] == maxval:
                res.append(key)
        return sorted(res)
