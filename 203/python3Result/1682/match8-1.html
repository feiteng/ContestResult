class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        
        ans = [0] * n
        ans[rounds[0]-1] += 1
        for idx in range(1, len(rounds)):
            s, e = rounds[idx-1], rounds[idx]
            if s <= e:
                for i in range(s, e):
                    ans[i] += 1
            else:
                for i in range(s, n):
                    ans[i] += 1
                for i in range(e):
                    ans[i] += 1
        max_use = max(ans)
        return [idx + 1 for idx in range(n) if ans[idx] == max_use]
