class Solution:
    def mostVisited(self, n: int, A: List[int]) -> List[int]:
        
        d = defaultdict(int)
        
        for i in range(1, len(A)):
            a, b = A[i - 1], A[i]
            if a < b:
                for j in range(a + 1, b + 1):
                    d[j] += 1
            else:
                for j in range(a + 1, n + 1):
                    d[j] += 1
                for j in range(1, b + 1):
                    d[j] += 1
        d[A[0]] += 1
        t = max(d.items(), key=lambda x:x[1])[1]
        #print(d)
        res = []
        for ele in d:
            if d[ele] == t:
                res.append(ele)
        
        return sorted(res)
