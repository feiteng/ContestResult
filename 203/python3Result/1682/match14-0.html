class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        sectors = [0] * (n + 1)
        for i in range(1, len(rounds)):
            if rounds[i-1] < rounds[i]:
                for k in range(rounds[i-1],rounds[i]):
                    sectors[k] += 1
            else:
                for k in range(rounds[i-1],n + 1):
                    sectors[k] += 1
                for k in range(rounds[i]):
                    sectors[k] += 1
        sectors[rounds[-1]] += 1
        
        res = []
        maxx = max(sectors)
        for i in range(1, len(sectors)):
            if sectors[i] == maxx:
                res.append(i)
        return res
