class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        c = Counter()
        c[rounds[0]] += 1
        for i in range(1, len(rounds)):
            if rounds[i] > rounds[i-1]:
                for v in range(rounds[i-1]+1, rounds[i]+1):
                    c[v] += 1
            else:
                for v in range(rounds[i-1]+1, n+1):
                    c[v] += 1
                for v in range(1, rounds[i]+1):
                    c[v] += 1
        most = c.most_common()[0][1]
        # print(c.most_common())
        res = [k for k in c.keys() if c[k]==most]
        return sorted(res)
