class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        counter = collections.Counter()
        for i in range(len(rounds) - 1):
            if rounds[i + 1] > rounds[i]:
                for r in range(rounds[i], rounds[i + 1]):
                    counter[r] += 1
            else:
                for r in range(rounds[i], n + 1):
                    counter[r] += 1
                for r in range(1, rounds[i + 1]):
                    counter[r] += 1
        counter[rounds[-1]] += 1
        m = counter.most_common()[0][1]
        ans = []
        for x, t in counter.items():
            if t == m:
                ans.append(x)
        return sorted(ans)
