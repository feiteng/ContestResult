class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        m = len(rounds) - 1
        v = [0] * (n + 1)
        for i in range(m):
            a, b = rounds[i], rounds[i + 1]
            if b < a:
                for j in range(a, n + 1):
                    v[j] += 1
                for j in range(1, b):
                    v[j] += 1
            else:
                for j in range(a, b):
                    v[j] += 1
        v[rounds[-1]] += 1
        r = max(v)
        return [i for i in range(1, n + 1) if v[i] == r]
