class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        res = []
        start = rounds[0]
        end = rounds[-1]
        
if start <= end:
            for i in range(start, end + 1):
                res.append(i)
        else:
            for i in range(start, n + 1):
                res.append(i)
            for i in range(1, end + 1):
                res.append(i)
        return sorted(res)
