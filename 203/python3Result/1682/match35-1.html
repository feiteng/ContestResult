class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        cnts = [0] * (n+1)
        for i in range(len(rounds) - 1):
            if rounds[i] < rounds[i+1]:
                for x in range(rounds[i], rounds[i+1]):
                    cnts[x] += 1
            else:
                for x in range(rounds[i], n+1):
                    cnts[x] += 1
                for x in range(1, rounds[i+1]):
                    cnts[x] += 1
        print(cnts)
        cnts[rounds[-1]] += 1
        max_cnt = max(cnts)
        return [i for i, x in enumerate(cnts) if x == max_cnt]
