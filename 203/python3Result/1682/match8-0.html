class Solution:
  def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
    data = [0] * (n + 1)
    data[rounds[0]] += 1
    
    for i in range(len(rounds) - 1):
      s, e = rounds[i], rounds[i + 1]
      if s < e:
        for p in range(s + 1, e + 1):
          data[p] += 1
      else:
        for p in range(s + 1, n + 1):
          data[p] += 1
        
        for p in range(1, e + 1):
          data[p] += 1

    m = max(data)
    return sorted([p for p, v in enumerate(data) if v == m])
