class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        lib = [0] * n
        lib[rounds[0] - 1] += 1
        for i in range(1, len(rounds)):
            start = rounds[i - 1]
            end = rounds[i]
            while start != end:
                start += 1
                if start > n:
                    start -= n
                lib[start - 1] += 1
        val = max(lib)
        ans = []
        for i in range(n):
            if lib[i] == val:
                ans.append(i + 1)
        return ans
