class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        count = [0]*(n+1)
        for x,y in zip(rounds, rounds
[1:]):
            if x<y:
                for i in range(x,y): count[i] += 1
            else:
                for i in range(x, n+1): count [i] += 1
                for i in range(1, y): count [i] += 1
        count[rounds[-1]] += 1
        m = max(count)
        return [x for x in range(1,n+1) if count[x]==m]
