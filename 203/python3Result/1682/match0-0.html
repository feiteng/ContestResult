class Solution:
    def mostVisited(self, n: int, rounds: List[int]) -> List[int]:
        count = [0]*(n+1)
        for i in range(1, len(rounds)):
            j = rounds[i-1]
            while j != rounds[i]:
                count[j] += 1
                j += 1
                if j > n:
                    j = 1
        count[rounds[-1]] += 1
        Max = max(count)
        res = []
        for i in range(1, len(count)):
            if count[i]==Max:
                res.append(i)
        return res
