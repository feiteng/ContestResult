class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        lefts = {}  # left -> right
        rights = {}  # right -> left
        sizes = defaultdict(int)
        ans = -1
        for idx, num in enumerate(arr):
            if num - 1 in rights and num + 1 in lefts:
                current_group = rights[num - 1]
                lefts[rights[num - 1]] = lefts[num + 1]
                rights[lefts[num + 1]] = rights[num - 1]
                sizes[(num - 1) - rights[num - 1] + 1] -= 1
                sizes[lefts[num + 1] - (num + 1) + 1] -= 1
                del rights[num - 1]
                del lefts[num + 1]
            elif num - 1 in rights:
                current_group = rights[num - 1]
                lefts[rights[num - 1]] = num
                rights[num] = rights[num - 1]
                sizes[(num - 1) - rights[num - 1] + 1] -= 1
                del rights[num - 1]
            elif num + 1 in lefts:
                current_group = num
                lefts[num] = lefts[num + 1]
                rights[lefts[num + 1]] = num
                sizes[lefts[num + 1] - (num + 1) + 1] -= 1
                del lefts[num + 1]
            else:
                current_group = num
                lefts[num] = num
                rights[num] = num
            sizes[lefts[current_group] - current_group + 1] += 1
            if sizes[m] > 0:
                ans = idx + 1
        return ans
