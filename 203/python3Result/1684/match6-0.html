class Solution:
    def findLatestStep(self, arr, m) -> int:
        d1, d2 = {}, {}  # {end:start}, {start:end}
        cnt = defaultdict(int)
        ans = -1
        for i, pos in enumerate(arr):
            if pos - 1 in d1:
                start = d1.pop(pos - 1)
                if pos + 1 in d2:
                    end = d2.pop(pos + 1)
                    d2[start] = end
                    d1[end] = start
                    cnt[pos - start] -= 1
                    cnt[end - pos] -= 1
                    cnt[end - start + 1] += 1
                else:
                    d2[start] = pos
                    d1[pos] = start
                    cnt[pos - start] -= 1
                    cnt[pos - start + 1] += 1
            elif pos + 1 in d2:
                end = d2[pos + 1]
                d1[end] = pos
                d2[pos] = end
                cnt[end - pos] -= 1
                cnt[end - pos + 1] += 1
            else:
                d1[pos] = pos
                d2[pos] = pos
                cnt[1] += 1

            if cnt[m]:
                ans = i + 1

        return ans
