class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        nums = defaultdict(int)
        mem = {}
        # 3
        # 3 5
        # 1 3 5
        # 
        ans = -1
        for i, num in enumerate(arr):
            if num - 1 in mem and num + 1 in mem:
                mem[num] = 1
                length = mem[num-1] + mem[num+1] +1
                
                nums[length] += 1
                nums[mem[num-1]] -= 1
                nums[mem[num+1]] -= 1
                
                mem[num-mem[num-1]] = length
                mem[num+mem[num+1]] = length

                
            elif num - 1 in mem:
                length = mem[num-1] +1
                mem[num] = length
                nums[length] += 1
                nums[mem[num-1]] -= 1     
                mem[num-mem[num-1]] = length
                
            elif num + 1 in mem:
                length = mem[num+1] +1
                mem[num] = length
                nums[length] += 1
                nums[mem[num+1]] -= 1     
                mem[num+mem[num+1]] = length
                
            else:
                length = 1
                mem[num] = 1
                nums[1] += 1
            
            # print(nums)
            # print(mem)
            if nums[m]:
                ans = i + 1
                    
                
        # print(mem)
        
        return ans
