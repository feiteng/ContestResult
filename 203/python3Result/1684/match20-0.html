class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        mp = {}
        cnt = {}
        fcnt = {}
        def find(x):
            if mp.get(x) is None:
                cnt[x] = 1
                fcnt[1] = fcnt.get(1,0)+1
                mp[x] = x
                return x
            if mp[x] != x:
                mp[x] = find(mp[x])
            return mp[x]
                
        def union(x,y):
            x = find(x)
            y = find(y)
            if x==y:
                return
            mp[x] = y
            # print(x,y,mp,cnt,fcnt)
            fcnt[cnt[x]] -= 1
            fcnt[cnt[y]] -= 1
            cnt[y] += cnt[x]
            fcnt[cnt[y]] = fcnt.get(cnt[y],0)+1
        ans = -1
        for i,num in enumerate(arr):
            find(num)
            # print(mp)
            if num-1 in mp:
                union(num-1,num)
            if num+1 in mp:
                union(num,num+1)
            
            # print(num,fcnt)
            if fcnt.get(m,0)>0:
                ans = i+1
        # print(ans)
        return ans
