class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        l_lens, r_lens = {}, {}
        res = -1
        m_set = set()
        for i, x in enumerate(arr):
            r_len = r_lens.pop(x-1, 0)
            l_len = l_lens.pop(x+1, 0)
            new_len = l_len + r_len + 1
            if r_len == m:
                m_set.remove(x-1)
            if l_len == m:
                m_set.remove(
x+l_len)
            r_lens[x+l_len] = new_len
            l_lens[x-r_len] = new_len
            if new_len == m:
                m_set.add(x+l_len)
            if m_set:
                res = i+1
        return res
