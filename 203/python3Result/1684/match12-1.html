class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        roots = [*range(n)]
        size = [1] * n
        
        
def find(i):            
            if i != roots[i]: roots[i] = find(roots[i])
            return roots[i]                
        
        def unite(i, j):
            cnt[size[find(i)]] -= 1
            cnt[size[find(j)]] -= 1
            tmp = size[find(i)] + size[find(j)]
            cnt[tmp] += 1
            size[find(i)] = size[find(j)] = tmp
            roots[find(i)] = roots[find(j)]
            
        cnt = [0] * (n + 1)
        curr = [0] * n                
        res = -1
        
for k, a in enumerate(arr, 1):
            i = a - 1
            curr[i] = 1
            cnt[1] += 1
            if i and curr[i - 1]:
                unite(i, i - 1)
            if i < n - 1 and curr[i + 1]:
                unite(i, i + 1)
            if cnt[m] > 0:
                res = k
       
        return res
