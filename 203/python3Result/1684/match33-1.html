class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        LR = {}
        cnt = collections.Counter()
        latest = -1
        for i,c in enumerate(arr):
            L = c if c not in LR else LR
[c]
            R = c+1 if c+1 not in LR else LR[c+1]
            cnt[R-L]+=1
            cnt[R-c-1]-=1
            cnt[c-L]-=1
            LR[L]=R
            LR[R]=L
            if cnt[m]>0:
                latest = i+1
        return latest
