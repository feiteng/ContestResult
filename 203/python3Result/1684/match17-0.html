class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        arr.reverse()
        cuts = [0, len(arr)+1]
        if m == len(arr):
            return len(arr)
        for i, e in enumerate(arr):
            x = bisect.bisect_left(cuts, e)
            cuts.insert(x, e)
            # print(cuts)
            if cuts[x+1] - cuts[x] == m+1 or cuts[x] - cuts[x-1] == m+1:
                return len(arr) - i - 1
        return -1
