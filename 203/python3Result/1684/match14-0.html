from sortedcontainers import SortedList

class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        freq = Counter()
        sl = SortedList()
        t = 0
        maxT = -1
        for x in arr:
            x -= 1
            a, b = (x, x + 1)
            index = sl.bisect_left((a, b))
            if index < len(sl) and sl[index][0] == b:
                c, d = sl.pop(index)
                freq[d - c] -= 1
                b = d
            if index - 1 >= 0 and sl[index - 1][1] == a:
                c, d = sl.pop(index - 1)
                freq[d - c] -= 1
                a = c
            sl.add((a, b))
            freq[b - a] += 1
            
            t += 1
            if freq[m]:                
                maxT = max(maxT, t)
        return maxT
