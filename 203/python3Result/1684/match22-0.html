class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        
        res = -1
        
        nums = [0] * (len(arr) + 2)
        
        d = defaultdict(int)
        
        for step, i in enumerate(arr):
            a, b = nums[i - 1], nums[i + 1]
            if a > 0 and b > 0:
                d[a] -= 1
                d[b] -= 1
                d[a + b + 1] += 1
                nums[i - a] = a + b + 1
                nums[i + b] = a + b + 1
            elif a > 0:
                d[a] -= 1
                d[a + 1] += 1
                nums[i - a] = a + 1
                nums[i] = a + 1
            elif b > 0:
                d[b] -= 1
                d[b + 1] += 1
                nums[i + b] = b + 1
                nums[i] = b + 1
            else:
                d[1] += 1
                nums[i] = 1
            if d[m] > 0:
                res = step + 1
        return res
