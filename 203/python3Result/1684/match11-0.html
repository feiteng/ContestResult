import bisect 
class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        lst = [0, n + 1]
        if n == m:
            return n
        if n < m:
            return -1 
        for i in range(n, 0, -1):
            p = arr[i - 1]
            j = bisect.bisect(lst, p)
            lst.insert(j, p)
            if lst[j] - lst[j-1] == m + 1 or lst[j+1] - lst[j] == m + 1:
                return i - 1
        return -1
