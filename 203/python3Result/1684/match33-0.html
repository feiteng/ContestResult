from collections import defaultdict
class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        bits = [0 for i in range(n)]
        left = [i for i in range(n)]
        right = [i for i in range(n)]
        d = defaultdict(int)
        ans = -1
        for t, idx in enumerate(arr):
            i = idx - 1
            bits[i] = 1
            if i + 1 < n and bits[i + 1] == 1:
                u, v = i + 1, right[i + 1]
                right[i] = v
                d[v - u + 1] -= 1
                d[v - u + 2] += 1
            else:
                right[i] = i
                d[1] += 1
            if i - 1 >= 0 and bits[i - 1] == 1:
                u, v = left[i - 1], i - 1
                left[i] = u
                d[v - u + 1] -= 1
                p, q = i, right[i]
                d[q - p + 1] -= 1
                d[q - u + 1] += 1
                left[q] = u
                right[u] = q
            else:
                u, v = i, right[i]
                right[u] = v
                left[v] = u
            # print(bits, left, right, d)
            if m in d and d[m] > 0:
                ans = t + 1
        return ans
