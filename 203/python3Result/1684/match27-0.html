class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        from sortedcontainers import SortedList
        sl=SortedList()
        ans=-1
        t=0
        for ss,i in enumerate(arr):
            k=[i,i]
            r=[]
            j=sl.bisect_left(k)
            if j-1>=0 and sl[j-1][1]==i-1:
                k[0]=sl[j-1][0]
                if sl[j-1][1]-sl[j-1][0]+1==m:
                    t-=1
                r.append(sl[j-1])
            if j<len(sl) and sl[j][0]==i+1:
                k[1]=sl[j][1]
                if sl[j][1]-sl[j][0]+1==m:
                    t-=1
                r.append(sl[j])
            for w in r:
                sl.remove(w)
            sl.add(k)
            if k[1]-k[0]+1==m:
                t+=1
            if t>0:
                ans=ss+1
            # print(ss,i,k,sl,ans)
        return ans
