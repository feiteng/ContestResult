class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        cnts = collections.defaultdict(int)
        iToCnt = {}
        res = -1
        for index, x in enumerate(arr):
            i = x - 1
            left = 0
            right = 0
            start = i
            end = i
            if i - 1 >= 0 and i - 1 in iToCnt:
                left = iToCnt[i - 1]
                start -= left
            if i + 1 < n and i + 1 in iToCnt:
                right = iToCnt[i + 1]
                end += right
            cnts
[left] -= left
            cnts[right] -= right
            cnts[left + right + 1] += left + right + 1
            iToCnt[start] = left + right + 1
            iToCnt[end] = left + right + 1
            if cnts[m] > 0:
                res = max(res, index + 1)
        return res
