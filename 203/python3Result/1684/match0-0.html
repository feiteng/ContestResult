class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        # 维护一个0的有序队列，每次从后往前插入一个0的坐标，然后计算这个坐标距离他两侧的值-1的大小是否包含 m
        index = [0,len(arr)+1]
        if index[-1]-index[0]-1==m:
            return len(arr)
        
        for i in range(len(arr)-1,-1,-1):      
            a = bisect.bisect(index,arr[i])
            index.insert(a,arr[i])
            if index[a]-index[a-1]-1==m or index[a+1]-index[a]-1==m:
                return i
        return -1
