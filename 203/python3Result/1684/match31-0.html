from collections import Counter

class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        arr = [it-1 for it in arr]
        ans = -1
        gsi = {} # si -> (si, ei)
        gei = {} # ei -> (si, ei)
        scount = Counter()
        ans = -1
        for i, it in enumerate(arr, 1):
            if it+1 in gsi and it-1 in gei:
                g1 = gsi[it+1]
                g2 = gei[it-1]
                del gsi[g1[0]]
                del gei[g1[1]]
                del gsi[g2[0]]
                del gei[g2[1]]
                scount[g1[1]-g1[0]+1] -= 1
                scount[g2[1]-g2[0]+1] -= 1
                g = (g2[0], g1[1])
            elif it+1 in gsi:
                g1 = gsi[it+1]
                del gsi[g1[0]]
                del gei[g1[1]]
                scount[g1[1]-g1[0]+1] -= 1
                g = (it, g1[1])
            elif it-1 in gei:
                g2 = gei[it-1]
                del gsi[g2[0]]
                del gei[g2[1]]
                scount[g2[1]-g2[0]+1] -= 1
                g = (g2[0], it)
            else:
                g = (it, it)
            scount[g[1]-g[0]+1] += 1
            gsi[g[0]] = g
            gei[g[1]] = g
            if scount[m] > 0:
                ans = i
        return ans
