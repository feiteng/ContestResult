class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        res = -1
        starts, ends = [0] * n, [0] * n
        s = set()
        
        for step, idx in enumerate(arr):
            i = idx - 1
            if i > 0 and ends[i - 1] > 0:
                start = i - ends[i - 1]
                if start in s:
                    s.remove(start)
                ends[i] = ends[i - 1] + 1
            else:
                ends[i] = 1
            
            if i < n - 1 and starts[i + 1] > 0:
                start = i + 1
                if start in s:
                    s.remove(start)
                starts[i] = starts[i + 1] + 1
            else:
                starts[i] = 1
            
            start = i + 1 - ends[i]
            l = ends[i]
            if i < n - 1:
                l += starts[i + 1]
            starts[start] = l
            ends[start + l - 1] = l
            if l == m:
                s.add(start)
            
            if s:
                res = step + 1
        return res
