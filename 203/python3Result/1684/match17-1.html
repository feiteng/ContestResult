class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        if m == len(arr):
            return m
        bb = [0,len(arr)+1]
        # print(arr)
        for i in range(-1,-len
(arr)-1,-1):
            ins = bisect.bisect_left(bb, arr[i], 0, len(bb))
            # print(bb,ins,arr[i])
            if (arr[i]-bb[ins-1]-1 == m) or (bb[ins]-arr[i]-1 == m):
                return i+len(arr)
            bb.insert(ins, arr[i])

        return -1
