class Solution:
    def findLatestStep(self, a: List[int], m: int) -> int:
        n=len(a)
        grp=[i for i in range(n)]
        sz=[0]*n
        vis=[False]*n
        d=defaultdict(int)
        def gt(n):
            a=[]
            while grp[n]!=n:
                a.append(n)
                n=grp[n]
            for x in a:
                grp[x]=n
            return n

        def unite(l,r):
            l=gt(l)
            r=gt(r)
            d[sz[l]]-=1
            d[sz[r]]-=1
            sz[r]+=sz[l]
            d[sz[r]]+=1
            grp[l]=r
        ans=-1
        for i,x in enumerate(a):
            x-=1
            vis[x]=True
            sz[x]=1
            d[1]+=1
            if x>=1 and vis[x-1]:
                unite(x-1,x)
            if x+1<n and vis[x+1]:
                unite(x+1,x)
            x=gt(x)
            if d[m]:
                ans=i+1
        return ans
