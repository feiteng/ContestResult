class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        tar = [0] * n
        parents = list(range(n))
        sizes = [1] * n
        cnts = defaultdict(int)
        ans = -1
        step = 1
        
def find(p):
            if parents[p] != p:
                parents[p] = find(parents[p])
            return parents[p]
        def union(p, q):
            nonlocal ans, step
            rp, rq = find(p), find(q)
            if rp != rq:
                parents[rq] = rp
                cnts[sizes[rq]] -= 1
                cnts[sizes[rp]] -= 1
                sizes[rp] += sizes[rq]
                cnts[sizes[rp]] += 1
                
        for k in arr:
            k 
-= 1
            tar[k] = 1
            cnts[1] += 1
            if k > 0 and tar[k-1] == 1:
                union(k, k-1)
            if k < n-1 and tar[k+1] == 1:
                union(k, k+1)
            if cnts[m] > 0:
                # print('union', m, cnts[m])
                ans = step
            step += 1
            # print(cnts)
            # print(sizes)
            # print(tar)
        return ans
