class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        end2start = dict()
        start2end = dict()
        c = 0
        
        def blockSize(pos, isStart):
            if isStart:
                return start2end[pos]-pos+1
            else:
                return pos-end2start[pos]+1
        latest = -1
        for j in range(n):
            i = arr[j]
            if i-1 in end2start and i+1 in start2end:
                bsl = blockSize(i-1, False)
                bsr = blockSize(i+1, True)
                s = bsl + bsr + 1
                if bsl == m: c -= 1
                if bsr == m: c -= 1
                if s == m: c += 1
                st = end2start[i-1]
                en = start2end[i+1]
                start2end[st] = en
                end2start[en] = st
            elif i-1 in end2start:
                bsl = blockSize(i-1, False)
                # print('.',bsl)
                s = bsl + 1
                if bsl == m: c -= 1
                if s == m: c += 1                
                st = end2start[i-1]
                start2end[st] = i
                end2start[i] = st
            elif i+1 in start2end:
                bsr = blockSize(i+1, True)
                s = bsr + 1
                if bsr == m: c -= 1
                if s == m: c += 1                
                en = start2end[i+1]
                end2start[en] = i
                start2end[i] = en
            else:
                start2end[i] = i
                end2start[i] = i
                if m == 1: c += 1
            if c >= 1:
                latest = j+1
            # print(c)
        return latest
