class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        C = collections.Counter()
        a = [0]*(len(arr)+2)
        ans = -1
        for i, x in enumerate(arr):
            C[a[x-1]] -= 1
            C[a[x+1]] -= 1
            l 
= a[x-1] + a[x+1] + 1
            C[l] += 1
            a[x-a[x-1]] = l
            a[x+a[x+1]] = l
            if C[m]: ans = i+1
        return ans
