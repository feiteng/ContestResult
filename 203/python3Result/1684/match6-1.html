class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        ret=-2
        s=dict()
        e=dict()
        c=Counter()
        for i,v in enumerate(arr):
            if v-1 in e and v+1 in s:
                ss=e[v-1]
                ee=s[v+1]
                del e[v-1]
                del s[v+1]
                s[ss]=ee
                e[ee]=ss
                c[ee-(v+1)+1]-=1
                c[v-1-ss+1]-=1
                c[ee-ss+1]+=1
            elif v-1 in e:
                ss=e[v-1]
                del e[v-1]
                s[ss]=v
                e[v]=ss
                c[v-1-ss+1]-=1
                c[v-ss+1]+=1
            elif v+1 in s:
                ee=s[v+1]
                del s
[v+1]
                s[v]=ee
                e[ee]=v
                c[ee-(v+1)+1]-=1
                c[ee-v+1]+=1
            else:
                s[v]=v
                e[v]=v
                c[1]+=1
            if c[m]>=1:
                ret=i
            # print(i,v,s,e,c,ret+1)
        return ret+1
