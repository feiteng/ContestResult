class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n=len(arr)
        if m==n or m==0:
            return m
        
        d={}
        c=Counter()
        ans=-1
        for i in range(n):
            if arr[i]-1 in d and arr[i]+1 in d:
                c[arr[i]-d[arr[i]-1]]-=1
                c[d[arr[i]+1]-arr[i]]-=1
                c[d[arr[i]+1]-d[arr[i]-1]+1]+=1
                d[d[arr[i]-1]],d[d[arr[i]+1]]=d[arr[i]+1],d[arr[i]-1]
                #d.remove(arr[i]-1)
                #d.remove(arr[i]+1)
            elif arr[i]-1 in d:
                c[arr[i]-d[arr[i]-1]]-=1
                c[arr[i]-d[arr[i]-1]+1]+=1
                d[arr[i]]=d[arr[i]-1]
                d[d[arr[i]-1]]=arr[i]
                #d.remove(arr[i]-1)
            elif arr[i]+1 in d:
                c[d[arr[i]+1]-arr[i]]-=1
                c[d[arr[i]+1]-arr[i]+1]+=1
                d[arr[i]]=d[arr[i]+1]
                d[d[arr[i]+1]]=arr[i]
                #d.remove(arr[i]+1)
            else:
                d[arr[i]]=arr[i]
                c[1]+=1
            #print(d,c)
            if c[m]>0:
                ans=i+1
        #print('111')
        return ans
