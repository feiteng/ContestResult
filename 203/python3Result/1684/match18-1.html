class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        n = len(arr)
        c = {}
        f = {}
        h = collections.Counter
()

        def bcj(x):
            a = []
            while x != f[x]:
                a.append(x)
                x = f[x]
            for y in a:
                f[y] = x
            return x

        def merge(a, b):
            a, b = bcj(a), bcj(b)
            f[a] = b
            h
[c[b]] -= 1
            h[c[a]] -= 1
            c[b] += c[a]
            h[c[b]] += 1

        ret = -1
        for i in range(len(arr)):
            x = arr[i]
            f[x] = x
            c[x] = 1
            h[1] += 1
            if x - 1 in f:
                merge(x, x - 1)
            if x + 1 in f:
                merge(x, x + 1)
            if h[m] > 0:
                ret = i + 1
        return ret
