import bisect
class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        if m == len(arr):
            return m
        tmp = [0, len(arr)+1]
        tmpans = len(arr)
        index = len(arr)-1
        while(tmpans):
            tmpans-=1
            tmptmp 
= arr[index]
            tmpindex = bisect.bisect(tmp, tmptmp)
            bisect.insort(tmp, tmptmp)
            if tmp[tmpindex] - tmp[tmpindex-1] - 1 == m or  tmp[tmpindex+1] - tmp[tmpindex] - 1 == m:
                return tmpans
            index -= 1
        return -1
