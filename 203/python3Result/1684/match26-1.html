class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        right = {}
        left = {}
        counter = collections.Counter()
        ans = -1
        for i, n in enumerate(arr):
            if n - 1 in left:
                counter[n - 1 - left[n - 1] + 1] -= 1
            if n + 1 in right:
                counter[right[n + 1] - (n + 1) + 1] -= 1
            l = left.pop(n - 1, n)
            r = right.pop(n + 1, n)
            left
[r] = l
            right[l] = r
            counter[r - l + 1] += 1
            if counter[m]:
                ans = i + 1
        return ans
