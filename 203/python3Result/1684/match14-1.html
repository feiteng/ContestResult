class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        import bisect
        import collections
        stack = []
        res = -1
        d = collections.Counter()
        for i in range(len(arr)):
            num = arr[i]
            idx = bisect.bisect_left(stack,(num,num))
            start = end = num
            
if idx < len(stack) and stack[idx][0] == num+1:
                x,y = stack.pop(idx)
                d[y-x+1] -= 1
                end = y
            if idx > 0 and stack[idx-1][1] == num-1:
                x,y = stack.pop(idx-1)
                d[y-x+1] -= 1
                start = x
            bisect.insort(stack,(start,end))
            d[end-start+1] += 1
            if d[m]:
                res = i+1
        return res
