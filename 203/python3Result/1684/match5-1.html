class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        # O(n)/ nlogn
        n = len(arr)
        s = [0]*n
        # (index, length)
        counter = collections.defaultdict(int)
        # s/e: length
        res = -1
        dic = collections.defaultdict(int)
        for step,i in enumerate(arr,1):
            l,r = i-1,i+1
            ll,rl = dic[l],dic[r]
            if ll:
                if rl:
                    total = ll+rl+1
                    counter
[ll]-=1
                    counter[rl]-=1
                    counter[total]+=1
                    dic[l-ll+1] = total
                    dic[r+rl-1] = total
                else:
                    counter[ll]-=1
                    counter[ll+1]+=1
                    dic[l-ll+1] = ll+1
                    dic[i] = ll+1
            else:
                if rl:
                    counter
[rl]-=1
                    counter[rl+1]+=1
                    dic[r+rl-1] = rl+1
                    dic[i] = rl+1
                else:
                    dic[i] = 1
                    counter[1]+=1
            if counter[m]>0:
                res = step
                
        
        return res
