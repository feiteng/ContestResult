class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        runningCount = collections.defaultdict(int)
        result = -1
        startMap,endMap = {},{}
        for event,index in enumerate(arr):
            leftPosition = index - 1
            rightPosition = index + 1
            start,end = index,index
            if leftPosition in endMap:
                leftStart,leftEnd = endMap[leftPosition]
                start = leftStart
                del endMap[leftEnd]
                del startMap[leftStart]
                runningCount[leftEnd-leftStart+1] -= 1
                
            if rightPosition in startMap:
                rightStart,rightEnd = startMap[rightPosition]
                end = rightEnd
                del startMap[rightStart]
                del endMap[rightEnd]
                runningCount[rightEnd-rightStart+1] -= 1
            
            startMap[start] = (start,end)
            endMap[end] = (start,end)
            runningCount[end-start+1] += 1
            if runningCount[m] != 0:
                result = event + 1
        return result
