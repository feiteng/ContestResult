class Solution:
    def findLatestStep(self, arr: List[int], m: int) -> int:
        r = [False] * len(arr)
        last = -1
        groups = defaultdict(int)
        lera = {}
        for j in range(len(arr)):
            ind = arr[j] - 1
            r[ind] = True
            l = [ind, ind]
            if ind > 0 and r[ind - 1]:
                l[0] = lera[ind - 1][0]
            if ind < len(arr) - 1 and r[ind + 1]:
                l[1] = lera[ind + 1][1]
                lera[lera[ind + 1][1]][0] = l[0]
            if ind > 0 and r[ind - 1]:
                lera[lera[ind - 1][0]][1] = l[1]
            lera[ind] = l
            groups[l[1] - l[0] + 1] += 1
            groups[ind - l[0]] -= 1
            groups[l[1] - ind] -= 1
 
            if groups[m]:
                last = j + 1
        return last
