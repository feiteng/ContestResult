class Solution:
    def maxCoins(self, piles: List[int]) -> int:
        piles.sort()
        # you can never get best, so settle for second. give bob the worst. repeat until nothing is left
        rounds = len(piles)//3
        
        ans = 0
        for j in range(rounds):
            #print(piles[(j+1)*-2])
            ans+=piles[(j+1)*-2]
        return ans
