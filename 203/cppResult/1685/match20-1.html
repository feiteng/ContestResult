class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> sums(n,0);
        for(int i = 0; i < n; i++){
            sums[i] = (i-1>=0?sums[i-1]:0)+stoneValue[i];
        }
        dp = vector<vector<int>>(n,vector<int>(n,-1));
        return dfs(stoneValue,sums,0,n-1);
    }
private:

    vector<vector<int>> dp;
    int dfs(vector<int>& stoneValue, vector<int>& sums,int l, int r){
        if(l>=r) return 0;
        if(dp[l][r] != -1) return dp[l][r];
        int ans = 0;
        int sum = 0;
        int cur = sums[r]-(l-1>=0?sums[l-1]:0);
        for(int i = l; i <= r-1; i++){

            sum += stoneValue[i];
            int left = cur-sum;
            int nxt = 0;
            if(sum > left){
                nxt = dfs(stoneValue,sums,i+1,r);
            }else if(sum<left){
                nxt = dfs(stoneValue,sums,l,i);
            }else{
                nxt = max(dfs(stoneValue,sums,i+1,r),dfs(stoneValue,sums,l,i));
            }
            ans = max(ans,min(sum,left)+nxt);
        }
        //cout<<l<<r<<ans<<endl;
        return dp[l][r] = ans;
    }
};
