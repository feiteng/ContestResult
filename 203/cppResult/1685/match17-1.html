class Solution {
public:
    
    int dp[510][510];
    int s[510];
    
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        s[0] = stoneValue[0];

        for (int i = 1; i < n; i ++) s[i] = s[i - 1] + stoneValue[i];
        for (int step = 2; step <= n; step ++) {
            for (int i = 0; i + step - 1 < n; i ++) {
                int j = i + step - 1;
                for (int k = i; k < j; k ++) {
                    int sl, sr;
                    if (i > 0) sl = s[k] - s[i - 1];
                    else sl = s[k];

                    sr = s[j] - s[k];
                    if (sl > sr) dp[i][j] = max(dp[i][j], sr + dp[k + 1][j]);
                    else if (sl < sr) dp[i][j] = max(dp[i][j], sl + dp[i][k]);
                    else  {
                        dp[i][j] = max(dp[i][j], sr + dp[k + 1][j]);
                        dp[i][j] = max(dp[i][j], sl + dp[i][k]);
                    }
                }
            }
        }
        return dp[0][n - 1];
    }
};
