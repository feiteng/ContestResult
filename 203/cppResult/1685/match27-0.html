struct cmp {
    bool operator()(const pair<int, int> &p) const {
        return p.first * 500 + p.second;
    }
};

class Solution {
public:
    
    int dp[505][505];
    
    int solve(vector<int>& A, int start, int end) {
        if (start == end) return 0;
        if (dp[start][end] != -1) return dp[start][end];
        
        int total = 0;
        for (int i = start; i <= end; i += 1) total += A[i];
        
        int left = 0;
        int best = 0;
        for (int i = start; i < end; i += 1) {            
            left += A[i];
            int right = total - left;
            
            // int solve_left = ;
            // int solve_right = ;
            
            if (left > right) best = max(best, right + solve(A, i+1, end));
            else if (right > left) best = max(best, left + solve(A, start, i));
            else best = max(best, max(left + solve(A, start, i), right + solve(A, i+1, end)));
            
            // if (start == 0 && end == A.size() - 1) {
                // printf("[%d %d] [left=%d right=%d best=%d]\n", start, end, left, right, best);
            // }
        }
        return dp[start][end] = best;
    }
    
    int stoneGameV(vector<int>& A) {
        //unordered_map<pair<int, int>, int, cmp> dp;
        memset(dp, -1, sizeof dp);
        return solve(A, 0, A.size() - 1);
    }
};
