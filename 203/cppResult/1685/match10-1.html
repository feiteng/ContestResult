class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector memo(n, vector(n, -1));
        vector<int> psum(n + 1);
        for (int i = 1; i <= n; i++)
            psum[i] = psum[i - 1] + stoneValue[i - 1];
        function<int(int, int)> dp = [&] (int l, int r) {

            if (l == r)
                return 0;
            if (memo[l][r] != -1)
                return memo[l][r];
            int res = INT_MIN;
            for (int i = l + 1; i <= r; i++) {
                int lsum = psum[i] - psum[l];

                int rsum = psum[r + 1] - psum[i];
                if (lsum > rsum) {
                    res = max(res, rsum + dp(i, r));
                } else if (rsum > lsum) {
                    res = max(res, lsum + dp(l, i - 1));
                } else {
                    res = max(res, lsum + dp(l, i - 1));
                    res = max(res, lsum + dp(i, r));
                }
            }
            return memo[l][r] = res;
        };
        return dp(0, n - 1);
    }
};
