int dp[600][600];
int value[600];

class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();

        value[0] = 0;
        for (int i = 0; i < n; i++) {
            value[i + 1] = value[i] + stoneValue[i];
        }
        for (int g = 1; g < n; g++) {
            for (int l = 0; l + g < n; l++) {
                int r = l + g;

                dp[l][r] = 0;
                for (int i = l; i + 1 <= r; i++) {
                    int rv = value[r+1] - value[i+1];
                    int lv = value[i+1] - value[l];
                    if (lv == rv) {
                        dp[l][r] = max(dp[l][r],max(dp[l][i],dp[i+1][r]) + lv);
                    } else if (lv < rv) {
                        dp[l][r] = max(dp[l][r],lv + dp[l][i]);
                    } else {
                        dp[l][r] = max(dp[l][r],rv + dp[i+1][r]);
                    }
                }
            }
        }
        return dp[0][n-1];
    }
};
