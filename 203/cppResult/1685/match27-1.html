class Solution {
public:
  
  int f[505][505];
  int sum[505];
  
  int dfs(int l, int r) {
    if (l >= r) return 0;
    if (f[l][r] != -1) return f[l][r];
    int ans = 0;
    for (int i = l; i < r; i ++) {

      int lsum = sum[i] - sum[l - 1];
      int rsum = sum[r] - sum[i];
      if (lsum < rsum)
        ans = max(ans, dfs(l, i) + lsum);
      else if (lsum > rsum)
        ans = max(ans, dfs(i + 1, r) + rsum);
      else
        ans = max(ans, max(dfs(l, i), dfs(i + 1, r)) + lsum);
    }
    return f[l][r] = ans;
  }
  
  int stoneGameV(vector<int>& stoneValue) {
    memset(f, -1, sizeof(f));
    sum[0] = 0;
    for (int i = 0; i < stoneValue.size(); i ++)
      sum[i + 1] = sum[i] + stoneValue[i];
    return dfs(1, stoneValue.size());
  }
};
