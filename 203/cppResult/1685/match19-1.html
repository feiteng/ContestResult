class Solution {
public:
    int dp[506][506];

    int val[507];
    int dfs(int i,int j){
        if(i>=j)return 0;
        if(dp[i][j]!=-1)return dp[i][j];
        int ans=0;
        for(int k=i;k!=j;k++){
            if(val[k+1]*2>=val[i]+val[j+1]) ans=max(ans,dfs(k+1,j)+val[j+1]-val[k+1]);
            if(val[k+1]*2<=val[i]+val[j+1]) ans=max(ans,dfs(i,k)+val[k+1]-val[i]);

        }
        return dp[i][j]=ans;
    }
    int stoneGameV(vector<int>& stoneValue) {
        val[0]=0;
        for(int i=0;i!=stoneValue.size();i++)val[i+1]=val[i]+stoneValue[i];
        memset(dp,-1,sizeof(dp));
        return dfs(0,int(stoneValue.size())-1);
    }
};
