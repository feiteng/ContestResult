class Solution {
public:
    vector<vector<int>> value;
    int dfs(vector<int>& nums,vector<int>& sum,int l,int r)
    {
        if (value[l][r]!=INT_MIN)
        {
            return value[l][r];
        }
        if (l==r)
        {
            value[l][r]=0;
            return value[l][r];
        }
        int ans=INT_MIN;
        for (int i=l;i<r;i++)
        {
            int suml = sum[i+1]-sum[l];
            int sumr = sum[r+1]-sum[i+1];
            if (sumr>suml)
            {
                ans = max(ans,dfs(nums,sum,l,i)+suml);
            }else if (sumr<suml)
            {
                ans = max(ans,dfs(nums,sum,i+1,r)+sumr);
            }else
            {
                ans = max(ans,max(dfs(nums,sum,l,i),dfs(nums,sum,i+1,r))+suml);
            }
        }
        value[l][r]=ans;
        return ans;
    }
    int stoneGameV(vector<int>& stoneValue) {
        value=vector<vector<int>>(stoneValue.size(),vector<int>(stoneValue.size(),INT_MIN));
        vector<int> sum(stoneValue.size()+1,0);
        for (int i=1;i<sum.size();++i)
        {
            sum[i]=sum[i-1]+stoneValue[i-1];
        }
        return dfs(stoneValue,sum,0,stoneValue.size()-1);
    }
};
