class Solution {
public:
    int get(vector<int>& stoneValue, int l, int r, vector<int>& preSum, vector<vector<int>>& dp){
        if(l == r) return 0;
        if(dp[l][r]) return dp[l][r];
        for(int mid = l; mid < r; mid ++){
            int leftSum = preSum[mid + 1] - preSum[l];
            int rightSum = preSum[r + 1] - preSum[mid + 1];
            if(leftSum > rightSum){
                dp[l][r] = max(dp[l][r], rightSum + get(stoneValue, mid + 1, r, preSum, dp));
            } else if(leftSum < rightSum){
                dp[l][r] = max(dp[l][r], leftSum + get(stoneValue, l, mid, preSum, dp));
            } else{
                dp[l][r] = max({dp[l][r], 
                                rightSum + get(stoneValue, mid + 1, r, preSum, dp),
                                leftSum + get(stoneValue, l, mid, preSum, dp)
                                });
            }
        }
        // cout << l << " " << r << " -> " << dp[l][r] << endl;
        return dp[l][r];
    }
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> preSum(n + 1, 0);
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for(int i = 0; i < stoneValue.size(); i++){
            preSum[i + 1] = preSum[i] + stoneValue[i];
        }
        return get(stoneValue, 0, n - 1, preSum, dp);
    }
};
