const int N = 555;
int dp[N][N];

int s[N];
class Solution {
public:
    int stoneGameV(vector<int>& a) {
        int n = a.size();
        for(int i= 1; i <= n; i++) dp[i][i] = 0;
        for(int i = 1; i <= n; i++) s[i] = s[i - 1] + a[i - 1];

        for(int delta = 1; delta < n; delta++) {
        
            for(int i = 1; i + delta <= n; i++) {
                int j = i + delta;
                dp[i][j] = 0;
                for(int k = i; k + 1 <= j; k++) {

                    if(s[k] - s[i - 1] <= s[j] - s[k]) {
                        dp[i][j] = max(dp[i][j], dp[i][k] + s[k] - s[i - 1]);
                    }
                    if(s[k] - s[i - 1] >= s[j] - s[k]) {
                        dp[i][j] = max(dp[i][j], dp[k + 1][j] + s[j] - s[k]);
                    }
                }
            }
        
        }
        return dp[1][n];
    }
};
