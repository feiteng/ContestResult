int dp[505][505];
int presum[505];
class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        if (n == 1) return 0;

        for (int i = 0; i < n; i++)
            presum[i + 1] = presum[i] + stoneValue[i];
        
        for (int len = 1; len < n; len++)
            for (int i = 0; i < n - len; i++)
            {
                int j = i + len;
                dp[i][j] = 0;

                for (int k = i; k < j; k++)
                {
                    if (presum[j + 1] - presum[k + 1] >= presum[k + 1] - presum[i])
                        dp[i][j] =max(dp[i][j], dp[i][k] + presum[k + 1] - presum[i]);
                    if (presum[j + 1] - presum[k + 1] <= presum[k + 1] - presum[i])
                        dp[i][j] =max(dp[i][j], dp[k + 1][j] + presum[j + 1] - presum[k + 1]);
                    
                }
            }
        return dp[0][n - 1];
    }
};
