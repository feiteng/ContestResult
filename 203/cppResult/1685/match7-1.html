class Solution {
public:
    int pre[501];
    int dp[501][501];
    int stoneGameV(vector<int>& stoneValue) {
        int n =stoneValue.size();
        memset(pre,0,sizeof(pre));
        memset(dp,0,sizeof(dp));

        for(int i=1;i<=n;i++){
            pre[i] = pre[i-1]+stoneValue[i-1];
        }
        
        for(int l=1;l<n;l++){
            for(int i=0;i<n-l;i++){
                int j = i+l;

                for(int k=i+1;k<=j;k++){
                    int prel = pre[k]-pre[i];
                    int prer = pre[j+1]-pre[k];
                    if(prel < prer){
                        dp[i][j] = max(dp[i][j],prel + dp[i][k-1]);
                    }
                    else if(prel > prer){
                        dp[i][j] = max(dp[i][j], prer + dp[k][j]);
                    }else{
                        dp[i][j] = max(dp[i][j],max(prel + dp[i][k-1], prer + dp[k][j]));
                    }
                }
            }
        }
        return dp[0][n-1];
    }
};
