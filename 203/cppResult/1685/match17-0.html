int dp[510][510];
int presum[510];
class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        memset(dp, 0, sizeof dp);
        memset(presum, 0, sizeof presum);
        for(int i=1;i<=n;i++)
            presum[i] = presum[i-1]+stoneValue[i-1];
        for(int len = 2;len<=n;len++){
            for(int i=1;i+len-1<=n;i++){
                int j = i+len-1;
                for(int k=i;k<j;k++){
                    int sumleft = presum[k]-presum[i-1];
                    int sumright = presum[j]-presum[k];
                    if(sumleft>sumright)
                        dp[i][j] = max(dp[i][j], dp[k+1][j]+sumright);
                    else if(sumleft<sumright)
                        dp[i][j] = max(dp[i][j], dp[i][k]+sumleft);
                    else{
                        dp[i][j] = max(dp[i][j], dp[k+1][j]+sumright);
                        dp[i][j] = max(dp[i][j], dp[i][k]+sumleft);
                    }
                }
            }
        }
        return dp[1][n];
    }
};
