class Solution {
public:
    int get(vector<int> &sv, vector<vector<int>> &dp, vector<int> &sum, int start, int end) {
        if (start == end) return 0;
        if (dp[start][end] >= 0) return dp[start][end];
        int s = 0, rs = sum[end] - (start == 0?0:sum[start - 1]);
        for (int i = start; i < end; ++i) {
            s += sv[i];
            

            if (s * 2 > rs) {
                dp[start][end] = max(dp[start][end], rs - s + get(sv, dp, sum, i + 1, end));
            } else if (s * 2 < rs) {
                dp[start][end] = max(dp[start][end], s + get(sv, dp, sum, start, i));
            } else {
                dp[start][end] = max(dp[start][end], max(rs - s + get(sv, dp, sum, i + 1, end), s + get(sv, dp, sum, start, i)));
            }
        }
        return dp[start][end];
    }
    
    int stoneGameV(vector<int>& sv) {
        vector<int> sum(sv.size(), 0);
        for (int i = 0; i < sv.size(); ++i) sum[i] = i == 0?sv[0]:(sum[i - 1] + sv[i]);
        
        vector<vector<int>> dp(sv.size(), vector<int>(sv.size(), -1));
        
        return get(sv, dp, sum, 0, sv.size() - 1);
    }
};
