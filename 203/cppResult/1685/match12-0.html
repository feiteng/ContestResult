class Solution
{
public:
    int dfs(int l, int r, vector<int>& sum, vector<vector<int>>& dp)
    {
        // cout << "l: " << l << " r: " << r << endl;
        // if (l == r) return sum[r] - sum[l - 1];
        if (l == r) return 0;
        if (dp[l][r] != -1) return dp[l][r];
        int maxn = 0;
        for (int i = l; i < r; i++)
        {
            int ls = sum[i] - sum[l - 1];
            int rs = sum[r] - sum[i]; 
            if (ls == rs)
            {
                maxn = max(maxn, max(dfs(l, i, sum, dp), dfs(i + 1, r, sum, dp)) + ls);
            }
            else if (ls < rs)
            {
                maxn = max(maxn, dfs(l, i, sum, dp) + ls);
            }
            else
            {
                maxn = max(maxn, dfs(i + 1, r, sum, dp) + rs);
            }
        }
        return dp[l][r] = maxn;
    }
    int stoneGameV(vector<int>& stoneValue)
    {
        int n = stoneValue.size();
        vector<int> sum(n + 1);
        vector<vector<int>> dp(n + 1, vector<int>(n + 1, -1));
        for (int i = 1; i <= n; i++) sum[i] = sum[i - 1] + stoneValue[i - 1];
        return dfs(1, n, sum, dp);
    }
};
