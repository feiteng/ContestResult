class Solution {
public:
    int dp[501][501];
    vector<int>v;
    vector<int>pre;
    int solve(int l, int r) {
        if (~dp[l][r]) {
            return dp[l][r];
        }
        if (l >= r) {
            return dp[l][r] = 0;

        }
        int mx = 0;
        for (int i = l ; i < r ; i++) {
            int left = pre[i] - pre[l - 1];
            int right = pre[r] - pre[i];
            if (left < right) {
                mx = max(mx, left + solve(l, i));
            }
            else if (left > right) {
                mx = max(mx, right + solve(i + 1, r));
            }
            else {
                mx = max(mx, left + max(solve(l, i), solve(i + 1, r)));
            }
        }
        return dp[l][r] = mx;
    }
    int stoneGameV(vector<int>& stoneValue) {
        memset(dp, -1, sizeof(dp));
        v = stoneValue;
        pre.push_back(0);
        for (auto &i : v) {
            pre.push_back(pre.back() + i);
        }
        return solve(1, v.size());
    }
};
