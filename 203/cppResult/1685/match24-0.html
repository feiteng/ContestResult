class Solution {
    int f[502][502], sum[502];
public:
    int stoneGameV(vector<int>& a) {
        int n = a.size();
        // memset(f, 0, sizeof(f));
        // cout << n << endl;
        // vector<vector<int>> f(n+1, vector<int>(n+1));
        
        // vector<int> sum(n+1);
        for (int i = 0; i < n; ++i)
            sum[i+1] = sum[i] + a[i];
        
        for (int j = 2; j <= n; ++j) {
            for (int i = j-1; i >= 1; --i) {
                int c = 0;
                for (int k = i; k < j; ++k) {
                    int a = sum[k] - sum[i-1], b = sum[j] - sum[k];
                    if (a > b)
                        c = max(c, b + f[k+1][j]);
                    else if (a < b)
                        c = max(c, a + f[i][k]);
                    else
                        c = max(c, a + max(f[i][k], f[k+1][j]));
                }
                f[i][j] = c;
            }
        }
        
        return f[1][n];
    }
};
