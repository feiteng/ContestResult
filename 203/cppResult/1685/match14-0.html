#define maxn 503
int f[maxn][maxn];
int sum[maxn];

class Solution {
public:
    int stoneGameV(vector<int>& s) {
        memset(f,0,sizeof(f));
        int n = s.size();
        sum[0] = 0;
        for (int i = 1; i <= n; i++) sum[i] = sum[i-1]+s[i-1];
        for (int l = 2; l <= n; l++) {
            for (int i = 1; i+l-1 <= n; i++) {
                int j = i+l-1;
                for (int k = i; k < j; k++) {
                    if (sum[k]-sum[i-1] < sum[j]-sum[k]) {
                        f[i][j] = max(f[i][j], f[i][k]+sum[k]-sum[i-1]);
                    }
                    else if (sum[k]-sum[i-1] > sum[j]-sum[k]) {
                        f[i][j] = max(f[i][j], f[k+1][j]+sum[j]-sum[k]);
                    }
                    else {
                        f[i][j] = max(f[i][j], max(f[i][k],f[k+1][j])+sum[k]-sum[i-1]); 
                    }
                    
                }
            }
        }
        return f[1][n];
    }
};
