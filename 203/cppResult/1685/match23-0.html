class Solution {
public:
    int f[505][505],sum[505];
    int stoneGameV(vector<int>& stoneValue) {
        int n=stoneValue.size();
        for (int i=0;i<n;i++)
            sum[i+1]=sum[i]+stoneValue[i];
        for (int i=1;i<=n;i++) f[i][i]=0;
        for (int l=2;l<=n;l++)
            for (int i=1;i+l-1<=n;i++){
                int j=i+l-1;
                for (int k=i;k<j;k++){
                    if (sum[k]-sum[i-1]<=sum[j]-sum[k]) f[i][j]=max(f[i][j], sum[k]-sum[i-1]+f[i][k]);
                    if (sum[j]-sum[k]<=sum[k]-sum[i-1]) f[i][j]=max(f[i][j], sum[j]-sum[k]+f[k+1][j]);
                }
            }
        return f[1][n];
    }
};
