#define mymax(a,b)(a>b?a:b)
class Solution {
public:
    const static int maxn=505;

    int n,arr[maxn];
    int dp[maxn][maxn];
    
    int stoneGameV(vector<int>& stoneValue) {
        n=stoneValue.size();
        arr[0]=0;
        for(int i=1;i<=n;++i){
            arr[i]=stoneValue[i-1]+arr[i-1];
            dp[i][i]=0;
        }

        for(int len=1;len<=n;++len){
            for(int i=1;i+len<=n;++i){
                int j=i+len;
                dp[i][j]=0;
                for(int k=i;k+1<=j;++k){
                    int Lsum=arr[k]-arr[i-1];
                    int Rsum=arr[j]-arr[k];
                    if(Lsum<Rsum){
                        dp[i][j]=mymax(dp[i][j],dp[i][k]+Lsum);
                    }
                    else if(Lsum>Rsum){
                        dp[i][j]=mymax(dp[i][j],dp[k+1][j]+Rsum);
                    }
                    else{
                        dp[i][j]=mymax(dp[i][j],dp[i][k]+Lsum);
                        dp[i][j]=mymax(dp[i][j],dp[k+1][j]+Rsum);
                    }
                }
            }
        }
        return dp[1][n];
    }
};
