class Solution {
public:
    int DP(vector<vector<int>>& dp, vector<int>& s, vector<int>& sum, int a, int b) {
        int n = s.size();

        if (a == b) return 0;
        if (dp[a][b] != -1) return dp[a][b];
        
        // [a .. i] & [i + 1 .. b]
        int tmp = 0;
        for (int i = a; i < b; ++i) {
            int l = sum[i] - sum[a - 1];
            int r = sum[b] - sum[i];
            // cout << a << " "<<i<<" "<<b<<" "<<l<<" "<<r<<endl;
            if (l > r) {
                tmp = max(tmp, DP(dp, s, sum, i + 1, b) + r);
            } else if (l < r) {
                tmp = max(tmp, DP(dp, s, sum, a, i) + l);
            } else {
                tmp = max(tmp, DP(dp, s, sum, i + 1, b) + r);
                tmp = max(tmp, DP(dp, s, sum, a, i) + l);

            }
        }
        
        return dp[a][b] = tmp;
    }
    
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        //   [6,2,3,4,5,5]
        //  0 6 8 11 ...
        vector<int> sum(n + 1, 0);
        vector<vector<int>> dp(n + 2, vector<int>(n + 2, -1));
        for (int i = 0; i < n; ++i) {
            sum[i + 1] = sum[i] + stoneValue[i];
            // cout<<i + 1<<" "<<sum[i+1]<<endl;
        }
        return DP(dp, stoneValue, sum, 1, n);
    }
};
