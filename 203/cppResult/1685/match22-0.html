class Solution {
    vector<int> psum;
public:
    int dp[505][505];
    int maxScore(vector<int>& s, int i, int j) {
        int ans = INT_MIN;
        
        if (i == j)
            return 0;
        
        if (dp[i][j] != -1)
            return dp[i][j];
        
        for (int k = i; k < j; k++) {
            int first = psum[k] - psum[i-1];
            int second = psum[j] - psum[k];
            //cout << i << " " << j << " " << k << " "<< first << " " << second << endl;
            if (first > second) {
                ans = max(ans, second + maxScore(s, k+1, j));
            } else if (first < second) {
                ans = max(ans, first + maxScore(s, i, k));
            } else {
                ans = max(ans, first + maxScore(s, k+1, j));
                ans = max(ans, first + maxScore(s, i, k));
            }
        }
        return dp[i][j] = ans;
    }
    int stoneGameV(vector<int>& s) {
        int sum = 0;
        
        memset(dp, -1, sizeof(dp));
        psum.push_back(0);
        for (int i = 0; i < s.size(); i++) {
            sum += s[i];
            psum.push_back(sum);
        }
        
        return maxScore(s, 1, s.size());
    }
};
