class Solution {
public:
    int dp[505][505],pre[505],n;
    int d(int l,int r){
        if(dp[l][r]!=-1)return dp[l][r];
        if(l==r)return 0;
        int ans = 0;
        for(int i=l;i<r;i++){
            if(pre[r]-pre[i]>pre[i]-pre[l-1])ans = max(ans,d(l,i)+pre[i]-pre[l-1]);
            else if(pre[r]-pre[i]<pre[i]-pre[l-1])ans = max(ans,d(i+1,r)+pre[r]-pre[i]);
            else{
                int L = d(l,i)+pre[i]-pre[l-1];
                int R = d(i+1,r)+pre[i]-pre[l-1];
                ans = max(ans,max(L,R));
            }
        }
        return dp[l][r]=ans;
    }
    int stoneGameV(vector<int>& st) {
        n = st.size();
        for(int i=1;i<=n;i++)pre[i] = pre[i-1]+st[i-1];
        memset(dp,-1,sizeof(dp));
        return d(1,n);
    }
};
