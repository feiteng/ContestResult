typedef long long ll;
class Solution {
public:
    vector<vector<int>> dp;
    vector<int> v;
    int dfs(int l, int r){
        if (r == l) return 0;
        if (r == l+1) return min(v[l], v[r]);
        if (dp[l][r] != -1) return dp[l][r];
        int sum = 0, ps = 0;
        for (int i = l; i <= r; i++) sum += v[i];
        for (int i = l; i < r; i++){
            ps += v[i];
            int sl = ps, sr = sum-ps;
            if (sl > sr){
                dp[l][r] = max(dp[l][r], sr+dfs(i+1, r));
            } else if (sl < sr){
                dp[l][r] = max(dp[l][r], sl+dfs(l, i));
            } else {
                dp[l][r] = max(dp[l][r], max(sr+dfs(i+1, r), sl+dfs(l, i)));
            }
        }
        return dp[l][r];
    }
    int stoneGameV(vector<int>& stoneValue) {
        int n = (int)stoneValue.size();
        v = stoneValue;
        dp.assign(n, vector<int>(n, -1));
        return dfs(0, n-1);
    }
};
