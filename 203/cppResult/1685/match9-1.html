const int maxn=505;
int sum[maxn],dp[maxn][maxn];
class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {

        int n=stoneValue.size();
        for (int i=0; i<=n; i++) for (int j=0; j<=n; j++) dp[i][j]=0;
        for (int i=1; i<=n; i++) sum[i]=sum[i-1]+stoneValue[i-1];
        for (int len=2; len<=n; len++) {
            for (int i=1; i+len-1<=n; i++) {
                int j=i+len-1;

                for (int k=i; k<j; k++) {
                    int L=sum[k]-sum[i-1];
                    int R=sum[j]-sum[k];
                    if (L==R) dp[i][j]=max(dp[i][j],max(dp[i][k],dp[k+1][j])+L);
                    else if (L<R) dp[i][j]=max(dp[i][j],dp[i][k]+L);
                    else dp[i][j]=max(dp[i][j],dp[k+1][j]+R);
                }
            }
        }
        return dp[1][n];
    }
};
