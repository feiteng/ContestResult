class Solution {
public:
    int dp[505][505];
    vector<int> val;
    int DP(int l,int r){
        if(l==r)return 0;
        if(dp[l][r]!=-1)return dp[l][r];
        int sum=0;
        for(int i = l;i<=r;i++)sum+=val[i];

        int now=0;
        dp[l][r]=0;
        for(int i=l;i<r;i++){
            now+=val[i];
            int left=now,right=sum-now;
            if(left>right){
                dp[l][r]=max(dp[l][r],right+DP(i+1,r));
            }
            else if(right>left){
                dp[l][r]=max(dp[l][r],left+DP(l,i));
            }
            else{
                dp[l][r]=max(dp[l][r],right+max(DP(l,i),DP(i+1,r)));
            }
        }
        return dp[l][r];
    }
    int stoneGameV(vector<int>& stoneValue) {
        val=stoneValue;
        memset(dp,-1,sizeof(dp));
        return DP(0,val.size()-1);
    }
};
