class Solution {
    vector<vector<int>> map;
    int n;
    int takeStone(vector<int>& stoneValue, int l, int r){
        if(l >= r) return 0;
        if(map[l][r] >= 0) return map[l][r];
        int res = 0; 
        for(int i = l; i < r; i++){
            int lnum, rnum;
            if(l){
                lnum = stoneValue[i] - stoneValue[l - 1];
            }
            else{
                lnum = stoneValue[i];
            }
            rnum = stoneValue[r] - stoneValue[i];
            int tmp = min(lnum, rnum);
            if(lnum > rnum){
                tmp += takeStone(stoneValue, i + 1, r);
            }
            else if(lnum < rnum){
                tmp += takeStone(stoneValue, l, i);
            }
            else{
                tmp += max(takeStone(stoneValue, i + 1, r), takeStone(stoneValue, l, i));
            }
            res = max(res, tmp);
        }
        map[l][r] = res;
        // cout<<l<<" "<<r<<" "<<res<<endl;
        return res;
    }
public:
    int stoneGameV(vector<int>& stoneValue) {
        n = stoneValue.size();
        map = vector<vector<int>>(n, vector<int>(n, -1));
        for(int i = 1; i < n; i++){
            stoneValue[i] += stoneValue[i - 1];
        }
        return takeStone(stoneValue, 0, n - 1);
    }
};
