class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int l = stoneValue.size();
        vector<int> sums(l + 1);
        for (int i = 0; i < l; i++) sums[i + 1] = sums[i] + stoneValue[i];
        vector<vector<int>> dp(l + 1, vector<int>(l + 1, -1));
        return f(sums, 0, l, dp);
    }
    
private:
    int f(vector<int>& sums, int i, int j, vector<vector<int>>& dp) {
        if (i == j) return 0;
        if (dp[i][j] != -1) return dp[i][j];
        int res = 0;
        for (int k = i + 1; k < j; k++) {
            int sum2 = sums[j] - sums[k], sum1 = sums[k] - sums[i];
            if (sum1 > sum2) {
                res = max(res, sum2 + f(sums, k, j, dp));
                // if (res == 22) cout << k << endl;
            }
            else if (sum1 < sum2) {
                res = max(res, sum1 + f(sums, i, k, dp));
                // if (res == 22) cout << k << endl;
            }
            else {
                res = max(res, sum1 + max(f(sums, i, k, dp), f(sums, k, j, dp)));
                // if (res == 22) cout << k << endl;
            }
        }
        // cout << i << " "<< j << " " << res<< endl;
        dp[i][j] = res;
        return res;
    }
};
