class Solution {
public:
    
    static const int N = 510;
    int f[N][N];
    int sum[N];
    
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        for (int i = 1; i <= n; i++) sum[i] = sum[i - 1] + stoneValue[i - 1];
        

        for (int len = 2; len <= n; len++){
            for (int i = 1; i + len - 1 <= n; i++){
                int j = i + len - 1;
                for (int k = i; k <= j - 1; k++){
                    int cur_l = sum[k] - sum[i - 1];
                    int cur_r = sum[j] - sum[k];
                    if (cur_l < cur_r){
                        f[i][j] = max(f[i][j], cur_l + f[i][k]);
                    }
                    else if(cur_l > cur_r){
                        f[i][j] = max(f[i][j], cur_r + f[k + 1][j]);
                    }
                    else{
                        f[i][j] = max(f[i][j], cur_r + max(f[i][k], f[k + 1][j]));
                    }
                }
            }
        }
        
        return f[1][n];
    }
};
