class Solution {
public:
    int n;
    vector<int> s;
    vector<vector<int>> dp;
    
    int get(int l, int r)
    {
        if(dp[l][r]!=-1) return dp[l][r];
        if(r-l==1)
        {
            dp[l][r]=min(s[r]-s[r-1],s[l]-s[l-1]);
            return dp[l][r];
        }
        else if(r-l==0)
        {
            dp[l][r]=0;
            return dp[l][r];
        }
        else
        {
            int sum=s[r]-s[l-1];
            int tmp=-1;
            for(int k=l+1;k<=r;++k)
            {
                if(s[k-1]-s[l-1]<s[r]-s[k-1])
                {
                    dp[l][r]=max(dp[l][r],s[k-1]-s[l-1]+get(l,k-1));
                }
                else if(s[k-1]-s[l-1]>s[r]-s[k-1])
                {
                    dp[l][r]=max(dp[l][r],s[r]-s[k-1]+get(k,r));
                }
                else
                {
                    dp[l][r]=max(dp[l][r],s[k-1]-s[l-1]+get(l,k-1));
                    dp[l][r]=max(dp[l][r],s[r]-s[k-1]+get(k,r));
                }
            }
            return dp[l][r];
        }
    }
    
    int stoneGameV(vector<int>& v) {
        n=v.size();
        s = vector<int>(n+1,0);
        dp=vector<vector<int>>(n+1,vector<int>(n+1,-1));
        
        for(int i=1;i<=n;++i)
        {
            s[i]=v[i-1];
            s[i]+=s[i-1];
        }
        return get(1,n);
        
    }
};
