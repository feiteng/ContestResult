class Solution {
public:
    int dp[501][501];
    vector<int>stone;
    vector<int>pre;
    
    int maxval(int i,int j){
        if(i>=j)return 0;
        if(dp[i][j]!=-1)return dp[i][j];

        dp[i][j]=INT_MIN;
        for(int k=i+1;k<=j;k++){
            if(pre[j+1]-pre[k] > pre[k]-pre[i]){
                dp[i][j] = max(dp[i][j], maxval(i,k-1)+pre[k]-pre[i]);
            }
            else if(pre[j+1]-pre[k] < pre[k]-pre[i]){
                dp[i][j] = max(dp[i][j], maxval(k,j)+pre[j+1]-pre[k]);
            }
            else{
                dp[i][j] = max(dp[i][j], maxval(i,k-1)+pre[k]-pre[i]);
                dp[i][j] = max(dp[i][j], maxval(k,j)+pre[j+1]-pre[k]);
            }
            
        }
        return dp[i][j];
    }
    
    int stoneGameV(vector<int>& stoneValue) {
        memset(dp,-1,sizeof(dp));
        stone=stoneValue;
        pre={0};
        for(int x:stoneValue)pre.push_back(pre.back()+x);
        
        return maxval(0,(int)stone.size()-1);
    }
};
