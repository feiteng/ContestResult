class Solution {
public:
    int dp(vector<int>& stone, vector<vector<int>>& mem, int i, int j, int sum) {
        if (i >= j)             return 0;
        if (mem[i][j] != -1)    return mem[i][j];
        mem[i][j] = 0;
        int running = 0;
        for (int cut = i; cut < j; cut++) {
            running += stone[cut];

            sum -= stone[cut];
            if (running < sum) {
                mem[i][j] = max(mem[i][j], running + dp(stone, mem, i, cut, running));
            } else if (running > sum) {
                mem[i][j] = max(mem[i][j], sum + dp(stone, mem, cut + 1, j, sum));
            } else {
                mem[i][j] = max(mem[i][j], running + dp(stone, mem, i, cut, running));
                mem[i][j] = max(mem[i][j], sum + dp(stone, mem, cut + 1, j, sum));

            }
        }
        return mem[i][j];
    }
    
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<vector<int>> mem(n, vector<int>(n, -1));
        int sum = 0;
        for (int i = 0; i < n; i++)     sum += stoneValue[i];
        
        return dp(stoneValue, mem, 0, n - 1, sum);
    }
};
