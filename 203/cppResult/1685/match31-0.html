int dp[505][505];
int s[505];
class Solution {
public:
    int stoneGameV(vector<int>& stoneValue) {
        int n = stoneValue.size();
        memset(s, 0, sizeof(s));
        for (int i = 1; i <= n; ++i) {
            s[i] = s[i - 1] + stoneValue[i - 1];
        }
        memset(dp, 0, sizeof(dp));
        for (int m = 2; m <= n; ++m) {
            for (int i = 0; i + m <= n; ++i) {
                int j = i + m;
                for (int k = i + 1; k < j; ++k) {
                    int l = s[j] - s[k], r = s[k] - s[i];
                    if (l < r) {
                        dp[i][j] = max(dp[i][j], l + dp[k][j]);
                    } else if (l > r) {
                        dp[i][j] = max(dp[i][j], r + dp[i][k]);
                    } else {
                        dp[i][j] = max(dp[i][j], l + dp[i][k]);
                        dp[i][j] = max(dp[i][j], r + dp[k][j]);
                    }
                }
                //cout << i << " " << i + m << " " << dp[i][j] << endl;
            }
        }
        //cout << endl;
        return dp[0][n];
    }
};
