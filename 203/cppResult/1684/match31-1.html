class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        unordered_map<int,int> pos;
        unordered_map<int,int> cnt;
        int res = -1;
        for (int i = 0; i < arr.size(); i++) {
            int b = arr[i];
            int len = 1 + pos[b - 1] + pos[b + 1];
            cnt[pos[b - 1]]--; 
            cnt[pos[b + 1]]--;
            cnt[len]++;
            pos[b - pos[b - 1]] = len;
            pos[b + pos[b + 1]] = len;
            if (cnt[m] > 0) res = i + 1;
        }
        return res;
    }
};
