class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int ans=-1;
        int n=arr.size();
        vector<int>l(n+1),r(n+1),ok(n+1,0),len(n+1,0);
        for(int i=0;i<n;++i) {
            int p=arr[i];

            ok[p]=1;l[p]=p,r[p]=p;
            if(p>1&&ok[p-1]) {
                l[p]=l[p-1];
                --len[r[p-1]-l[p-1]+1];
                r[p-1]=r[p];
            }
            if(p<n&&ok[p+1]) {
                r[p]=r[p+1];
                --len[r[p+1]-l[p+1]+1];
                l[p+1]=l[p];
            }
            r[l[p]]=r[p];
            l[r[p]]=l[p];
            ++len[r[p]-l[p]+1];
            if(len[m]) ans=i+1;
        }
        return ans;
    }
};
