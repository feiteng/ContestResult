class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        set<pair<int,int>> S;
        int latest = -1;
        int count = 0;
        for(int i = 0; i < arr.size(); i++)
        {
            int a = arr[i];
            vector<set<pair<int,int>>::iterator> D;
            int l = a, r = a;
            auto u = S.lower_bound({a,a});
            if(u != S.end())
            {

                if(u->first == a+1)
                {
                    r = u->second;
                    D.push_back(u);
                }
            }
            if(!S.empty() && u != S.begin())
            {
                u--;
                if(u->second == a-1)
                {
                    l = u->first;
                    D.push_back(u);
                }
            }
            
            if(r-l+1 == m)
            {
                count++;
            }
            
            //cout << "inserting " << l << ", " << r << endl;
            S.insert({l,r});

            for(auto &z : D)
            {
                if(z->second - z->first + 1 == m)
                {
                    count--;
                }
                S.erase(z);
            }
            
            //cout << "# of arrays of size " << m << ": " << count << endl;
            
            if(count > 0)
            {
                latest = i+1;
            }
            
        }
        
        return latest;
    }
};
