class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        reverse(arr.begin(), arr.end());
        int n = arr.size(), x;

        if(n == m) return n;

        set<int>s;

        for(int i=0; i<n; i++){
        	s.insert(arr[i]);
        	auto it = s.upper_bound(arr[i]);
        	if(it == s.end())
        		x = n-arr[i];
        	else
        		x = *it - arr[i] -1;
        	if(x == m)
        		return n-1-i;

        	it = s.lower_bound(arr[i]);
        	if(it == s.begin())
        		x = arr[i] -1;
        	else{
        		it--;
        		x = arr[i] - *it -1;
        	}
        	if(x == m)
        		return n-1-i;

        }

        return -1;
    }
};
