class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        set<pair<int, int> > section;
        section.insert(make_pair(-100, -100));
        section.insert(make_pair(1000000, 1000000));
        
        int ret = -1;
        int ret_count = 0; 
        for (int i = 0; i < arr.size(); i++) {
            auto it_2 = section.upper_bound(make_pair(arr[i], arr[i]));
            auto it_1 = section.upper_bound(make_pair(arr[i], arr[i])); it_1--;
            
            int left = arr[i], right = arr[i];
            if (it_1->second == arr[i] - 1) {
                left = it_1->first;
                if (it_1->second - it_1->first == m - 1) {
                    ret_count--;
                }
                section.erase(it_1);
            }
            if (it_2->first == arr[i] + 1) {
                right = it_2->second;
                if (it_2->second - it_2->first == m - 1) {
                    ret_count--;
                }
                section.erase(it_2);
            }
            
            if (right - left == m - 1) {
                ret_count++;
            }
            
            if (ret_count > 0) {
                ret = i + 1;
            }
            
            section.insert(make_pair(left, right));
        }
        
        return ret;
    }
};
