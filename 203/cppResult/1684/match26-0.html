#include <bits/stdc++.h>
const int maxn = 101110;

int pre[maxn],sz[maxn],a[maxn],b[maxn];

int find(int x)
{
    return x==pre[x]?x:pre[x]=find(pre[x]);
}

void unite(int x,int y)
{
    x = find(x);
    y = find(y);
    if(x == y) return;
    pre[x] = y;
    sz[y] += sz[x];
}

class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        int ans = -1;
        for(int i = 1; i <= n; i++) 
        {
            pre[i] = i;sz[i] = 1;
        }
        for(int i = 0; i <= n+1; i++) 
        {
            a[i] = b[i] = 0;
        }
        for(int i = 0; i < n; i++) 
        {
            int x = arr[i];
            a[x] = 1;
            if(a[x-1])
            {   
                b[sz[find(x-1)]]--;
                unite(x,x-1);
            }
            if(a[x+1])
            {   
                b[sz[find(x+1)]]--;
                unite(x,x+1);
            }
            b[sz[find(x)]]++;
            if(b[m]) ans = i+1;
        }
        return ans;
    }
};
