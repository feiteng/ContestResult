class Solution {
 public:
  int findLatestStep(vector<int>& arr, int m) {

    int n = arr.size();
    vector<int> left(n + 2, -1), right(n + 2, -1);
    int res = -1;
    int gs = 0;
    for (int i = 0; i < n; i++) {
      int pos = arr[i];
      int l = left[pos - 1] == -1 ? pos : left[pos - 1];

      int r = right[pos + 1] == -1? pos : right[pos + 1];
      left[pos] = l;
      right[pos] = r;
      if (r - l + 1 == m)
        gs++;
      if (pos - l == m) 
        gs--;
      if (r - pos == m) 
        gs--;
      if (gs != 0) res = i + 1;
      right[l] = r;
      left[r] = l;
    }
    return res;
  }
};
