class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n=arr.size();
        vector<int> v(n+1,0);
        vector<int> cnt(n+1,0);
        int ans=-1;
        for(int i=0;i<arr.size();i++){

            int x=arr[i];
            int l=x,r=x;
            if(x-1>0&&v[x-1]){
                cnt[v[x-1]]--;
                l=x-v[x-1];
            }
            if(x+1<=n&&v[x+1]){
                cnt[v[x+1]]--;
                r=x+v[x+1];
            }
            v[l]=1+v[l]+v[r];
            v[r]=v[l];
            cnt[v[l]]++;
            if(cnt[m])ans=i+1;
        }
        return ans;
    }
};
