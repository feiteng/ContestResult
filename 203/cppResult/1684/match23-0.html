class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        if( n == m ) return n;
        
        set<int> hash;
        hash.insert(n + 1);
        hash.insert(0);
        int res = -1;
        
        for( int i = n - 1 ; i >= 0 ; --i ){
            int pos = arr[i];
            hash.insert(pos);
            auto it = hash.find(pos);
            it++;
            int up = *it;
            it--;
            it--;
            int low = *it;
            if( up - pos - 1 == m || pos - low - 1 == m ){
                res = i;
                break;
            }
        }
        
        /*int res = -1;
        
        for( int i = n - 1 ; i >= 0 ; --i ){
            int pos = arr[i];

            int low = *hash.upper_bound(-pos);
            if( up - pos - 1 == m || pos + low - 1 == m ){
                res = i;
                break;
            }
            hash.insert(pos);
            hash.insert(-pos);
        }*/
        return res;
    }
};
