class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        string s(n, '0');
        vector<int> tag(n, -1);
        int cnt = 0;
        int ret = -1;
        for(int i=0; i<n; ++i) {
            int pos = arr[i] - 1;
            s[pos] = '1';
            int left = pos;
            int right = pos;
            if(pos != 0 && tag[pos-1] != -1) {
                left = tag[pos-1];
                if(pos-left == m) --cnt;
            }
            if(pos != n-1 && tag[pos+1] != -1) {
                right = tag[pos+1];
                if(right - pos == m) --cnt;
            }
            tag[left] = right;
            tag[right] = left;
            int length = right - left + 1;
            if(length == m) ++cnt;
            if(cnt > 0) ret = i+1;
        }
        return ret;
    }
};
