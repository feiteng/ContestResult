class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        if (n == m) return n;
        set<int> a;
        a.insert(0);
        a.insert(n + 1);
        for (int i = n - 1; i >= 0; i--)
        {
            auto tmpp = a.lower_bound(arr[i]);
            auto tmp = tmpp; tmp--;
            if (*tmpp - arr[i] == m + 1 || arr[i] - *tmp == m + 1) return i;
            a.insert(arr[i]);
        }
        return -1;
    }
};
