class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        

        int n=arr.size();
        vector<int> bits(arr.size(),0);
        int ans=0;
        
        int res=-1;
        for(int i=0;i<arr.size();i++){
            
            
            
            int left= arr[i]-1>=1?bits[arr[i]-1-1]:0;
            int right= arr[i]-1<n-1?bits[arr[i]-1+1]:0;
            
            int tleng=left+right+1;
            //cout<<tleng<<endl;
            
            bits[arr[i]-1]=tleng;

            bits[arr[i]-1-left]=tleng;
            bits[arr[i]-1+right]=tleng;
            
            if(left==m)ans--;
            if(right==m)ans--;
            if(tleng==m)ans++;
            
            if(ans>0)res=i+1;
        }
        
        
        
        return res;
        
    }
};
