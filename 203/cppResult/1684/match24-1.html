class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        if (m == arr.size())
            return m;
        set<int> memo;
        int n = arr.size();
        memo.insert(0), memo.insert(n+1);
        for (int i = 1; i <= n; i++){
            auto l = --memo.lower_bound(arr[n - i]);
            auto r = memo.upper_bound(arr[n - i]);
            // cout << *l << " " << *r << endl;
            if (arr[n - i] - *l - 1== m || *r - arr[n - i] - 1 == m)  
                return n - i;
            memo.insert(arr[n - i]);
        }
        return -1;
    }
};
