class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        set<pair<int,int>> st;
        st.insert({1,n});
        if(n == m)
            return n;
        for(int i = n - 1;i>=0;i--){
            auto it = st.upper_bound({arr[i],n+1});
            if(it == st.begin())
                continue;
            it--;
            pair<int,int> p =*it;
            st.erase(it);
            if(arr[i] - p.first == m)
                return i;
             if(p.second - arr[i] == m)
                return i;
            if(arr[i] - p.first > 0)
                st.insert({p.first,arr[i] - 1});
            if(p.second - arr[i] > 0)
                st.insert({arr[i]+1,p.second});
        }
        return -1;
    }
};
