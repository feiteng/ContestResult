class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int cur = 0;
        int n = arr.size() + 2;
        vector<int> v(n, 0);
        int cnt = 0;
        int res = -1;
        for (int i = 0; i < arr.size(); ++i) {
            int p = arr[i];
            v[p] = v[p - 1] + v[p + 1] + 1;
            if (v[p + 1] == m) {
                --cnt;
            } 
            if (v[p - 1] == m) {
                --cnt;
            }
            if (v[p] == m) {
                ++cnt;
            }
            v[p - v[p - 1]] = v[p];
            v[p + v[p + 1]] = v[p];
            if (cnt > 0) {
                res = i + 1;
            }
        }
        return res;
        
    }
};
