class Solution {
public:
    int findLatestStep(vector<int>& arr, int xxx) {
        int n = arr.size();
        if(n == xxx) return n;
        
        map<int, int> m;
        m[-1] = -n;
        
        for(int i = arr.size()-1; i >= 0; i --)
        {
            int t = arr[i];
            
            auto itr = m.lower_bound(-t);
            
            int a = -itr->first, b = -itr->second;
            m.erase(itr);
            
            // cout<<m.size()<<": "<<a<<" "<<b<<" "<<t<<endl;
            
            if(t-a == xxx || b-t == xxx) return i;
            
            if(a <= t-1)m[-a] = -t+1;
            if(t+1 <= b)m[-t-1] = -b;
            // for(auto& p: m) cout<<p.first<<" "<<p.second<<endl;
            // cout<<"**********"<<endl;
        }
        return -1;
    }
};
