class Solution {
public:
    int findLatestStep(const vector<int>& arr, const int m) {
            const int n = arr.size();
            vector<int> bounds(n+1, 0);
            int count = 0;
            int answer = -1;
            for (int step = 1; step <= n; ++step) {
                const int num = arr.at(step-1);
                const int left = (num>1 && bounds.at(num-1) != 0) ? bounds.at(num-1) : num;
                const int right = (num<n && bounds.at(num+1) != 0) ? bounds.at(num+1) : num;
                bounds.at(left) = right;
                bounds.at(right) = left;
                if (num - left == m) count--;
                if (right - num == m) count--;
                if (right - left + 1 == m) count++;
                if (count > 0) answer = step;
                //cout << left << ' ' << right << ' ' << step << '\n';
            }
        return answer;
    }
};
