int left1[100010];
int right1[100010];
int flags[100010];
class Solution {
public:

    int findLatestStep(vector<int>& arr, int m) {
        int ret  = -1;
        map<int,int> ma;
        memset(left1, 0, sizeof left1);
        memset(right1, 0, sizeof right1);
        memset(flags, 0, sizeof flags);
        for(int i=0;i<arr.size();i++ ){
            int cur = arr[i];
            if(flags[cur+1]==0&&flags[cur-1]==0){
                flags[cur]=1;
                left1[cur]=1;
                right1[cur]=1;
                ma[1]++;
            }else if(flags[cur+1]==0&&flags[cur-1]!=0){
                flags[cur]=1;
                ma[left1[cur-1]]--;
                left1[cur] = left1[cur-1]+1;
                right1[cur-left1[cur-1]]+=1;
                ma[left1[cur-1]+1]++;
            }else if(flags[cur+1]!=0&&flags[cur-1]==0){
                flags[cur]=1;
                ma[right1[cur+1]]--;
                right1[cur] = right1[cur+1]+1;
                left1[cur+right1[cur+1]]+=1;
                ma[right1[cur+1]+1]++;
            }else{
                flags[cur]=1;
                ma[left1[cur-1]]--;
                ma[right1[cur+1]]--;
                ma[left1[cur-1]+1+right1[cur+1]]++;
                int len = left1[cur-1]+1+right1[cur+1];
                right1[cur-left1[cur-1]]=len;
                left1[cur+right1[cur+1]] = len;
                
            }
            if(ma[m]>0)
                ret = i+1;
        }
        return ret;
    }
};
