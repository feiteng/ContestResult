class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        vector<int> as(n, 0);
        multiset<int> st;
        int ans = -1;
        for(int s=0; s<arr.size(); s++){
            int a = arr[s];
            int i = a - 1;
            as[i] = 1;
            if(i-1>=0 && as[i-1] != 0 && i+1<n && as[i+1] != 0){
                int lnL = as[i-1], lnR = as[i+1];
                int j = i - 1 - lnL + 1, k = i+1+lnR-1;
                st.erase(st.find(lnL));
                st.erase(st.find(lnR));
                int ln = lnL + lnR + 1;
                as[j] = ln; as[k] = ln;
                st.insert(ln);
            }else if(i-1>=0 && as[i-1] != 0){
                int lnL = as[i-1];
                int j = i - 1 - lnL + 1;
                st.erase(st.find(lnL));
                int ln = lnL + 1;
                as[j] = ln; as[i] = ln;
                st.insert(ln);
            }else if(i+1<n && as[i+1] != 0){
                int lnR = as[i+1];
                int k = i+1+lnR - 1;
                st.erase(st.find(lnR));
                int ln = lnR + 1;
                as[k] = ln; as[i] = ln;
                st.insert(ln);
            }else{
                st.insert(1);
                as[i] = 1;
            }

            if(st.find(m) != st.end()){
                ans = s + 1;
            }
        }      

        return ans;

    }
};
