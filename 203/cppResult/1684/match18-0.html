class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        
        int n = arr.size();
        set<pair<int, int>> vs{make_pair(1, n)};
        if (m == n)
            return n;
        for (int i = n - 1; i >= 0; --i)
        {
            auto it = vs.lower_bound(make_pair(arr[i], n + 1));
            --it;
            auto p = *it;
            vs.erase(it);
            if (arr[i] - p.first == m || p.second - arr[i] == m)
                return i;
            if (arr[i] > p.first)
                vs.insert(make_pair(p.first, arr[i] - 1));
            if (p.second > arr[i])
                vs.insert(make_pair(arr[i] + 1, p.second));
        }
        
        return -1;
    }
};
