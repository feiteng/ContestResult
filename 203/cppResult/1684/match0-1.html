class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        vector<int> map(n, 0);
        int res = 0; 
        int record = 0; 
        for(int i = 0; i < n; i++){
            int now = arr[i] - 1;
            // cout<< now<<endl;

            int l = 0; int r = 0;
            if(now) {
                l = map[now - 1];
                if(l == m) record--;
            }
            if(now < n - 1) {
                r = map[now + 1];
                if(r == m) record--;
            }
            int sum = 1 + l + r;
            if(sum == m) record++;
            map[now - l] = sum;
            map[now + r] = sum;
            if(record) res = i + 1;
        }
        return res ? res : -1;
    }
};
