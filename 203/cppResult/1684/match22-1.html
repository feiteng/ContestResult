class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        if (n == m) return n;
        map<int,int> mp;
        mp[0] = n;
        for (int i = n - 1; i >= 0; --i) {
            int x = arr[i] - 1;
            auto it = prev(mp.upper_bound(x));
            int l = it->first, r = it->second;
            if (l < x) {
                if (x - l == m) return i;
                mp[l] = x;
            }
            if (r > x + 1) {
                if (r - x - 1 == m) return i;
                mp[x + 1] = r;
            }
        }
        return -1;
    }
};
