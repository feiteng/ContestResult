class Solution {
public:
    int fa[110000],sz[110000];
    int ct[110000];//记录长度为i的组的个数
    int vis[110000];
    int ffind(int x)
    {
        if(fa[x]==x) return x;
        return fa[x]=ffind(fa[x]);
    }
    void mmerge(int x,int y){
        int fx=ffind(x);
        int fy=ffind(y);
        if(fx!=fy){
            ct[sz[fx]]--;ct[sz[fy]]--;
            fa[fy]=fx;
            sz[fx]+=sz[fy];sz[fy]=0;
            ct[sz[fx]]++;
        }
    }
    
    int findLatestStep(vector<int>& arr, int m) {
        memset(fa,-1,sizeof(fa));
        memset(sz,0,sizeof(sz));
        memset(ct,0,sizeof(ct));
        memset(vis,0,sizeof(vis));
        int ans=-1;
        for(int i=0;i<arr.size();i++){
            int now=arr[i];
            vis[now]=1;fa[now]=now;sz[now]=1;ct[1]++;
            if(now-1>=1&&vis[now-1]) mmerge(now,now-1);
            if(now+1<=arr.size()&&vis[now+1]) mmerge(now,now+1);
            if(ct[m]) ans=i+1;
        }
        return ans;
    }
};
