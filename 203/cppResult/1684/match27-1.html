class Solution {
public:
    
    vector<int> f, cnt;
    unordered_map<int, int> mp;
    int find(int a) {
        if (a != f[a]) f[a] = find(f[a]);
        return f[a];
    }
    int last = -1;
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        f.resize(n + 1);
        cnt.resize(n+1);
        for (int i = 1; i <= n; i ++) f[i] = i;
        for (int i = 0; i < n; i ++) {
            int x = arr[i];

            cnt[x] = 1;
            if (x > 1 && cnt[x - 1]) {
                int fl = find(x - 1);
                f[fl] = x;
                cnt[x] += cnt[fl];
                mp[cnt[fl]] --;
            }
            if (x < n && cnt[x + 1]) {
                int fr = find(x + 1);
                f[fr] = x;
                cnt[x] += cnt[fr];
                mp[cnt[fr]] --;
            }
            mp[cnt[x]] ++;
            if (mp[m]) last = i + 1;
        }
        return last;
    }
};
