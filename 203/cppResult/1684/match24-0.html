class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        if (m == arr.size()) return m;
        vector<int> cnt(arr.size()+2, 0);
        int ret = -1;
        for (int i = 0; i < arr.size(); ++i) {
            int p = arr[i];
            int p1 = p - cnt[p-1];
            int p2 = p + cnt[p+1];
            if (cnt[p-1] == m || cnt[p+1] == m) ret = i;
            int x = 1 + cnt[p-1] + cnt[p+1];
            cnt[p1] = cnt[p2] = cnt[p] = x;
        }
        return ret;
    }
};
