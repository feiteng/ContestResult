class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n=arr.size();
        if (m==n) return n;
        map<int,int> mm;
        mm[1]=n;
        for (int i=n-1;i>=0;--i) {
            // round i, remove arr[i+1];
            int a=arr[i];
            auto it=mm.upper_bound(a);
            if (it==mm.begin()) continue;
            --it;
            int f=it->first, e=it->second;
            if (f<=a&&e>=a) {
                mm.erase(it);
                if (e-a==m||a-f==m) return i;
                if (e-a>m) mm[a+1]=e;
                if (a-f>m) mm[f]=a-1;
            }
        }
        return -1;
    }
};
