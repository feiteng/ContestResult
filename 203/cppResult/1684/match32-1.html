class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int N = arr.size();
        map<int, pair<int, int>> p;
        p[N] = make_pair(1, N);
        if (m == N)
            return N;

        for (int i = N-1; i>=0; --i) {
            int now = arr[i];
            // printf("now %d\n", now);
            auto iter = p.lower_bound(now);
            auto pr = iter->second;
            p.erase(iter);
            // printf("%d %d\n", pr.first, pr.second);
            if (now - pr.first == m || pr.second - now == m)            
                return i;
            if (now-1 >= pr.first) {
                p[now-1] = make_pair(pr.first,now-1);
                // printf("- %d %d\n", pr.first, now-1);
            }
            if (pr.second >= now+1) {
                p[pr.second] = make_pair(now+1,pr.second);
                // printf("- %d %d\n", now+1, pr.second);
            }
        }
        return -1;
    }
};
