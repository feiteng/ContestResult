class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        vector<int> right(n+1, -1), left(n+1, -1), cnt(n+1, 0);
        int step = 0;
        int ans = -1;
        for (auto k : arr) {
            step++;
            right[k] = k;
            left[k] = k;
            
            int start = k, end = k;
            if (k > 1 && left[k-1] != -1) {
                cnt[right[k-1]-left[k-1]+1]--;
                start = left[k-1];
            }
            if (k < n && right[k+1] != -1) {

                cnt[right[k+1]-left[k+1]+1]--;
                end = right[k+1];
            }
            
            right[start] = end;
            right[end] = end;
            left[start] = start;
            left[end] = start;
            
            cnt[end-start+1]++;
            if (cnt[m] > 0)
                ans = step;
        }
        
        return ans;
    }
};
