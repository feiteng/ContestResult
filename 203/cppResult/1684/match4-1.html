class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        if (n == m) return n;
        map<int, int> mp;
        mp[n] = 1;
        auto handle = [&mp](int l, int r) {
            int k = r - l + 1;
            if (k > 0) mp[r] = l;
            return k;
        };

        for (int i = n; i >= 1; --i) {
            int j = arr[i-1];
            auto find = mp.lower_bound(j);
            int r = find->first, l = find->second;
            mp.erase(find);
            int l1 = l, r1 = j-1;
            int l2 = j+1, r2 = r;
            if (handle(l1, r1) == m || handle(l2, r2) == m) return i-1;
        }
        return -1;
    }
};
