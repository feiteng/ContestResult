class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n=arr.size();
        if ( m==n ) return n;
        set<int> s;
        s.insert(0);
        s.insert(n+1);
        for ( int i=n-1; ~i; i-- ) {
            auto it=s.lower_bound(arr[i]);
            int r=*it;
            auto l=*--it;
            if ( r-arr[i]==m+1 ) return i;
            else if ( arr[i]-l==m+1 ) return i;
            s.insert(arr[i]);
        }
        return -1;
    }
};
