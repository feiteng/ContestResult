class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        vector<int> s(n, 0);
        vector<int> l(n, -1);
        vector<int> r(n, -1);
        unordered_map<int, int> numbers;
        
        int res = -1, step = 0;
        for (auto a: arr) {
            step ++;
            int i = a - 1;
            
            int count = 1;
            int tail = i, head = i;
            if (i>= 1 && s[i-1] == 1) {
                head = l[i-1];
                count += (i - head);
            } 
            if (i < arr.size() - 1 && s[i+1] == 1) {
                tail = r[i+1];
                count += (tail - i);
            }
            numbers[count] ++;
            if (head != i) {
                numbers[i - head] --;
                
            }
            if (tail != i) {
                numbers[tail - i] --;
            }
            
            
            
            if (numbers[m] > 0)
                res = step;
            
            s[i] = 1;
            
            
            r[head] = tail;
            l[tail] = head;
            
        }
        
        
        return res;
    }
};
