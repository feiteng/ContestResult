class Solution {
public:
    int findLatestStep(vector<int>& a, int m) {
        set<int> dp;
        int n = a.size();
        dp.insert(0);
        dp.insert(n+1);
        if(m==n) return n;
        for(int i=n-1;i>=0;i--)
        {
            int cur = a[i];
            auto tmp = dp.upper_bound(cur);
            int y = *tmp;
            tmp--;
            int x = *tmp;
            //cout<<x<<y<<cur<<endl;
            if(cur-x==m+1||y-cur==m+1) return i;
            dp.insert(cur);
        }
        return -1;
    }
};
