class Solution {

public:
    int findLatestStep(vector<int>& arr, int m) {
        int n = arr.size();
        if(m == n) return n;
        
        set<int> s;
        s.insert(0);
        s.insert(n + 1);
        
        for(int i = n - 1; i >= 0; --i) {
            auto iter = s.lower_bound(arr[i]);
            int r = *iter;
            int l = *(--iter);
            if(m == r - arr[i] - 1 || m == arr[i] - l - 1) return i;
            s.insert(arr[i]);
        }
        return -1;
    }
    
};
