class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {

        vector<int> str(arr.size(), 0);
        int count = 0;
        int result = -1;
        for (int round = 0; round < arr.size(); ++round) {
            int index = arr[round] - 1;
            str[index] = 1;
            if (index > 0 && str[index - 1] > 0 && (arr[round] == arr.size() || str[index + 1] == 0)) {
                int prev_length = str[index - 1];
                int new_length = prev_length + 1;
                if (prev_length == m) {
                    --count;
                }
                if (new_length == m) {
                    ++count;
                }
                str[index - prev_length] = new_length;
                str[index] = new_length;
            } else if (arr[round] < arr.size() && str[index + 1] > 0 && (index == 0 || str[index - 1] == 0)) {
                int prev_length = str[index + 1];
                int new_length = prev_length + 1;
                if (prev_length == m) {
                    --count;
                }
                if (new_length == m) {
                    ++count;
                }
                str[index + prev_length] = new_length;
                str[index] = new_length;
            } else if (index > 0 && str[index - 1] > 0 && arr[round] < arr.size() && str[index + 1] > 0) {
                int prev_length1 = str[index - 1];
                int prev_length2 = str[index + 1];
                int new_length = prev_length1 + prev_length2 + 1;
                if (prev_length1 == m) {

                    --count;
                }
                if (prev_length2 == m) {
                    --count;
                }
                if (new_length == m) {
                    ++count;
                }
                str[index - prev_length1] = new_length;
                str[index + prev_length2] = new_length;
                str[index] = new_length;
            } else {
                if (m == 1) {
                    ++count;
                }
            }
            if (count > 0) {
                result = round + 1;
            }
        }
        return result;
    }
};
