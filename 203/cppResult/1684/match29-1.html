class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        vector<int> mark(arr.size() + 2);
        int result = -1;
        vector<int> lens(arr.size() + 1);
        for(int i = 0; i < arr.size(); i++) {
            int current = arr[i];
            int lenss;
            if (mark[current - 1] == 0 && mark[current + 1] == 0) {

                mark[current] = current;
                lens[1]++;
            } else if (mark[current - 1] > 0 && mark[current + 1] == 0) {
                mark[current] = mark[current - 1];
                mark[mark[current - 1]] = current;
                lenss = current - mark[current] + 1;
                lens[lenss]++;
                lens[lenss - 1]--;
            } else if (mark[current - 1] == 0 && mark[current + 1] > 0) {
                mark[current] = mark[current + 1];
                mark[mark[current + 1]] = current;
                lenss = mark[current] - current + 1;
                lens[lenss]++;
                lens[lenss - 1]--;
            } else {
                lenss = (current - mark[current - 1]) + (mark[current + 1] - current) + 1;
                lens[lenss]++;
                lens[current - mark[current - 1]]--;
                lens[mark[current + 1] - current]--;
                int temp = mark[current - 1];
                mark[mark[current - 1]] = mark[current + 1];
                mark[mark[current + 1]] = temp;
            }
            // cout <<"m:";
            // for(int j = 1; j <= arr.size(); j++) {
            //     cout << mark[j] << " ";
            // }
            // cout << endl;
            // cout << "l:";
            // for(int j = 1; j <= arr.size(); j++) {
            //     cout << lens[j] << " ";
            // }
            // cout << endl;
            if (lens[m] > 0) {
                result = i + 1;
            }
        }
        return result;
    }
};
