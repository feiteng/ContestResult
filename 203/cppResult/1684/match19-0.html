class Solution {
public:
    int findLatestStep(vector<int>& arr, int m) {
        vector<int> c(arr.size() + 1, 0);
        vector<int> r(arr.size() + 2, 0);
        int lastStep = -1;
        for (int i = 0; i < arr.size(); ++i) {
            int id = arr[i], left = id, right = id;
            if (r[id - 1] > 0) {
                left = abs(r[id - 1]);
                c[id - left]--;
                r[id - 1] = 0;
            }
            if (r[id + 1] < 0 || r[id + 1] == id + 1) {
                right = abs(r[id + 1]);
                r[id + 1] = 0;
                c[right - id]--;
            }
            r[left] = -right;
            r[right] = left;
            c[right - left + 1]++;
            if (c[m] > 0) lastStep = i + 1;
           /* for (int j = 0; j <= arr.size(); ++j) {
                printf("%d %d %d\t", j, c[j], r[j]);
            }
            printf("\n");*/
        }
            
        return lastStep;
    }
};
