class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& ro) {
        int i,j,s,e;
        s=ro[0];
        e=ro[ro.size()-1];
        //swap(s,e);
        vector<int> rs;
        if(s<=e)
        for(;s<=e;s++){
            rs.push_back(s);
        }
        else{
            for(i=1;i<=e;i++){
                rs.push_back(i);
            }
            for(i=s;i<=n;i++){
                rs.push_back(i);
            }
        }
        return rs;
    }
};
