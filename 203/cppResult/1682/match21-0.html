class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cnt(n, 0);
        for (int i = 0; i < rounds.size() - 1; i++) {
            for (int s = rounds[i] - 1; s != rounds[i + 1] - 1; s = (s + 1) % n) {
                cnt[s] += 1;
            }
        }
        int m = rounds.size();
        cnt[rounds[m - 1] - 1] += 1;
        int maxv = 0;
        for (int i = 0; i < n; i++) {
            maxv = max(maxv, cnt[i]);
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            if (cnt[i] == maxv) {
                ans.push_back(i + 1);
            }
        }
        return ans;
    }
};
