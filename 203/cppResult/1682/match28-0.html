class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& d) {
        int m = d.size();
        int s = d[0];
        int e = d.back();
        vector<int> res;
        if(s<=e){
            for(int i=s;i<=e;i++)
            {
                res.push_back(i);
            }
        } 
        else
        {
            for(int i=1;i<=e;i++)
            {
                res.push_back(i);
            }
            for(int i=s;i<=n;i++)
            {
                res.push_back(i);
            }
        }
        return res;
    }
};
