class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> re;
        if(rounds[0]<=rounds.back())
        {
            for(int i=rounds[0];i<=rounds.back();++i) re.push_back(i);
        }
        else
        {
            for(int i=1;i<=rounds.back();++i) re.push_back(i);
            for(int i=rounds[0];i<=n;++i) re.push_back(i);

        }
        
        return re;
    }
};
