class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cnt(n);
        int N = rounds.size();
        for (int i=0; i<N-1; i++) {
            int s = rounds[i]-1;
            while (s != rounds[i+1]-1) {
                cnt[s]++;
                s = (s+1) % n;

            }
            // cnt[s]++;
        }
        cnt[rounds.back()-1]++;
        int m = *max_element(cnt.begin(), cnt.end());
        vector<int> ans;
        for (int i=0; i<n; i++) {
            // cout << cnt[i] << " ";
            if (cnt[i] == m)
                ans.push_back(i+1);
        }
        // cout << endl;
        return ans;
    }
};
