class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cnt(n);
        int m = rounds.size();
        for (int i = 0; i + 1 < m; ++i) {
            for (int j = rounds[i] + (i != 0); j <= rounds[i + 1] + (rounds[i + 1] < rounds[i] ? n : 0); ++j) {
                ++cnt[(j - 1) % n];
            }
        }
        int maxc = *max_element(cnt.begin(), cnt.end());
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            if (cnt[i] == maxc) {
                ans.push_back(i + 1);
            }
        }
        return ans;
    }
};
