class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> ret;
        if (rounds.back() >= rounds.front()) {
            for (int i = rounds.front(); i <= rounds.back(); ++i)
                ret.push_back(i);
        } else {
            for (int i = 1; i <= rounds.back(); ++i)
                ret.push_back(i);
            for (int i = rounds.front(); i <= n; ++i)
                ret.push_back(i);
        }
        return ret;
    }
};
