class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cnt(n, 0);
        int x = rounds[0] - 1;
        for (int i = 1; i < rounds.size(); i++) {
            int y = rounds[i] - 1;
            while (x != y) {
                cnt[x]++;
                x = (x + 1) % n;
            }
        }
        cnt[x]++;
        int mx = *max_element(cnt.begin(), cnt.end());
        vector<int> ans;
        for (int i =0 ; i < cnt.size(); i++) {
            if (cnt[i] == mx) ans.push_back(i + 1);
        }
        return ans;
    }
};
