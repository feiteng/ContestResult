class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        // int vis[110];
        // memset(vis, 0, sizeof(vis));
        // int now = rounds[0];
        int l = rounds.size();
        // vis[now]++;
        // for (int i = 1; i < l; i++) {
        //     while (now != rounds[i]) {
        //         now++;
        //         vis[now]++;
        //     }
        // }
        int a = rounds[0];
        int b = rounds[l - 1];
        set<int> ans;
        for (int i = a; ; i++) {
            if (i == n + 1)
                i = 1;
            ans.insert(i);
            if (i == b) {
                break;
            }
        }
        ans.insert(b);
        vector<int> ret;
        for (auto it : ans) {
            ret.push_back(it);
        }
        return ret;
    }
};
