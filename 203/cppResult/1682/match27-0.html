class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cc(n+1,0);
        int m = rounds.size();
        cc[rounds[0]%n]++;
        for(int i = 0;i<m-1;i++)
        {
            int start = rounds[i]%n;
            int end = rounds[i+1]%n;
            if(end<start)
                end+=n;
            for(int j = start+1;j<=end;j++)
            {
                cc[j%n]++;
            }   
        }
        int maxv=-1;
        for(int i = 0;i<cc.size();i++)
            maxv=max(maxv, cc[i]);
        vector<int> res;
        for(int i = 1;i<=n;i++)
        {
            if(cc[i%n]==maxv)
                res.push_back(i);
        }
        return res;
    }
};
