class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> sector_count(n);
        int curr = rounds[0] - 1;
        
        for (int i = 1; i < rounds.size(); i++) {

            while (curr != rounds[i] - 1) {
                sector_count[curr]++;
                curr = (curr + 1) % n;
            }
        }
        sector_count[curr]++;
        
        int mx = *max_element(sector_count.begin(), sector_count.end());
        
        vector<int> result;
        
        for (int i = 0; i < sector_count.size(); i++) {
            if (sector_count[i] == mx) {
                result.push_back(i + 1);
            }
        }
        
        
        return result;
    }
};
