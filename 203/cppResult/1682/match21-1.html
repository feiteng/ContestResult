class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> t(n, 0);
        for (int i = 1; i < rounds.size(); i++) {
            for (int s = rounds[i - 1] - 1; s != rounds[i] - 1; s = (s + 1) % n) {
                t[s]++;
            }
        }
        t[rounds.back() - 1]++;
        int m = INT_MIN;

        for (int i : t) {
            m = max(m, i);
        }
        vector<int> r;
        for (int i = 0; i < t.size(); i++) {
            if (t[i] == m) {
                r.push_back(i + 1);
            }
        }
        return r;
    }
};
