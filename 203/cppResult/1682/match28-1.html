class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        int left = rounds.front();
        int right = rounds.back();
        
        vector<int> res;
        if (left == right) return vector<int> { left };
        else {

            if(left < right) {
                for (int i = left; i <= right; i++) {
                    res.push_back(i);
                }
            }
            else {
                for (int i = left; i <= n; i++) {
                    res.push_back(i);
                }
                for (int i = 1; i <= right; i++) {
                    res.push_back(i);
                }
                sort(res.begin(), res.end());
            }
        }
        
        return res;
    }
};
