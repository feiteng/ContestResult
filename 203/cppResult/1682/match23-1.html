class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector <int> ans;
        ans.resize(n + 1);
        for (int i = 0; i < rounds.size() - 1; ++ i) {
            int j = rounds[i];
            while (j != rounds[i + 1]) {
                ans[j] ++;
                j ++;

                if (j == n + 1) j = 1;
            }
        }
        ans[rounds[rounds.size() - 1]] ++;
        int mx = 0;
        for (int i = 1; i <= n; ++ i) mx = max(mx, ans[i]);
        vector <int> ret;
        for (int i = 1; i <= n; ++ i) if (ans[i] == mx) ret.push_back(i);
        return ret;
    }
};
