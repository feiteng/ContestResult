class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> count(n + 1, 0);
        for (int i = 1; i < rounds.size(); ++i) {
            int from = rounds[i - 1];
            int to = rounds[i];

            int j = from;
            while (j != to) {
                ++count[j];
                ++j;
                if (j == n + 1) {
                    j = 1;
                }
            }
        }
        ++count[rounds.back()];
        int max = INT_MIN;
        vector<int> result;
        for (int i = 0; i < count.size(); ++i) {
            if (count[i] > max) {
                max = count[i];
                result.clear();
                result.push_back(i);
            } else if (count[i] == max) {
                result.push_back(i);
            }
        }
        return result;
    }
};
