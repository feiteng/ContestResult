class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> v(n + 2);
        for( int i = 0 ; i < rounds.size() - 1; ++i ){
            for( int j = rounds[i] - 1; ; j = (j + 1) % n ){
                v[j + 1]++;
                if( j == rounds[i + 1] - 1) break;
            }
            rounds[i + 1] = (rounds[i + 1] - 1 + 1 ) % n + 1;
        }
        int maxi = 0;
        for( int x: v ){
            maxi = max(maxi, x);
        }
        
        vector<int> res;
        for( int i = 1 ; i <= n ; ++i ){
            if( v[i] == maxi ){
                res.push_back(i);
            }
        }
        return res;
    }
};
