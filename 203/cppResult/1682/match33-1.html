class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& r) {
        int st = r[0];
        int m = r.size() - 1;
        int ed = r[m];
        vector<int> ans;
        for (int i = st; i != ed; i = (i % n) + 1) ans.push_back(i);
        ans.push_back(ed);
        sort(ans.begin(), ans.end());
        return ans;
    }
};
