class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        map<int,int> M;
        for(int i = 1; i < rounds.size(); i ++) {
            int x = rounds[i - 1];

            int y = rounds[i];
            while(x != y) {
                M[x] ++;
                x = (x + 1);
                if(x == n + 1) x = 1;
            }
        }
        M[rounds.back()] ++;
        int mx = -1;
        for(auto p: M) {
            mx = max(mx, p.second);
        }
        vector<int> ans;
        for(auto p: M) if(p.second == mx) {
            ans.push_back(p.first);
        }
        sort(ans.begin(), ans.end());
        return ans;
    }
};
