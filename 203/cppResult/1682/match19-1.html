class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> cnt(n, 0);
        int m = rounds.size();
        for (int i = 1 ; i < m ; i++) {
            int s = rounds[i - 1] - 1, t = rounds[i] - 1;
            while (s != t) {
                cnt[s]++;
                s++, s %= n;
            }
        }
        cnt[rounds.back() - 1]++;

        int maxV = *max_element(cnt.begin(), cnt.end());
        vector<int> ret;
        for (int i = 0 ; i < n ; i++) {
            if (cnt[i] == maxV)
                ret.push_back(i+  1);
        }
        return ret;
    }
};
