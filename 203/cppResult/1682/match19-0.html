class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        vector<int> aux(n + 1);
        int m = rounds.size();
        for (int i = 1; i < m; ++i) {
            int start = rounds[i - 1], end = rounds[i];
            while (start != end) {
                ++aux[start];
                if (++start == n + 1) start = 1;
            }
            if (i == m - 1) ++aux[end];
        }
        int mn = *max_element(aux.begin(), aux.end());
        vector<int> ans;
        for (int i = 1; i <= n; ++i) {
            if (aux[i] == mn) ans.push_back(i);
        }
        return ans;
    }
};
