class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        int i,s,e;
        s=rounds[0],e=rounds[rounds.size()-1];
        vector<int> ans;
        if(s<=e){
            for(i=s;i<=e;++i){
                ans.push_back(i);
            }

        }
        else{
            for(i=1;i<=e;++i){
                ans.push_back(i);
            }
            for(i=s;i<=n;++i){
                ans.push_back(i);
            }
        }
        return ans;
    }
};
