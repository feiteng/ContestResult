class Solution {
public:
    int maxCoins(vector<int>& piles) {
        int result = 0;
        sort(piles.begin(), piles.end());
        int len = piles.size();
        for(int i = 0; i < len / 3; ++i) {
            result += piles[len - 2 - 2 * i];
        }
        return result;
    }
};
