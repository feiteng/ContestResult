class Solution:
    def maxScore(self, s: str) -> int:
        ones = 0
        for c in s:
            if c == "1":
                ones += 1
        sofar = ones
        if s[0] == "1":
            sofar -= 1
        else:
            sofar += 1
        res = sofar
        for i in range(1, len(s) - 1):
            if s[i] == "1":
                sofar -= 1
            else:
                sofar += 1
            res = max(res, sofar)
        return res
