from typing import List

class Solution:
    def maxScore(self, s: str) -> int:
        n = len(s)
        left = [None] * n
        right = [None] * n

        count = 0
        for i in range(n):
            if s[i] == '0':
                count += 1
            left[i] = count
        
        count = 0
        for i in range(n - 1, -1, -1):
            if s[i] == '1':
                count += 1
            right[i] = count
        
        return max(left[i - 1] + right[i] for i in range(1, n))
