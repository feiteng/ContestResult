class Solution:
    def maxScore(self, s: str) -> int:
        one = 0
        for c in s:
            if c == '1': one += 1
        zero = 0
        ret = -1
        for c in s[:-1]:
            if c == '0':
                zero += 1
            else:
                one -= 1
            ret = max(ret, zero + one)
        return ret
