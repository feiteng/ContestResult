class Solution:
    def maxScore(self, s: str) -> int:
        rones = s.count('1')
        lzeros = 0
        res = -math.inf
        for c in s[:-1]:
            if c == '1':
                rones -= 1
            else:                
                lzeros += 1
            res = max(res, rones + lzeros)
        return res
