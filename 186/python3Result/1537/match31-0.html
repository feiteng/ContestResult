class Solution:
    def maxScore(self, s: str) -> int:
        one = s.count('1')
        zero = 0
        res = 0
        for i in range(0, len(s)-1):
            if s[i] == '1':
                one -= 1
            if s[i] == '0':
                zero += 1
            res = max(res, one+zero)
        return res
