class Solution:
    def maxScore(self, s: str) -> int:
        cnt = collections.Counter([c for c in s])
        cs = cnt['1']
        ms = 0
        for c in s[:-1] :
            if c == '0' : 
                cs += 1
            else :    
                cs -= 1
            ms = max(cs,ms)
        return ms
