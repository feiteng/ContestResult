class Solution:
    def maxScore(self, s: str) -> int:
        n = len(s)
        num_zeros = [0] * n
        num_ones = [0] * n
        zeros = 0
        for i in range(n):
            if s[i] == '0':
                zeros += 1
            num_zeros[i] = zeros
        ones = 0
        for i in range(n - 1, -1, -1):
            if s[i] == '1':
                ones += 1
            num_ones[i] = ones
        return max(num_zeros[i] + num_ones[i + 1] for i in range(n - 1))
