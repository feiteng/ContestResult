from collections import Counter

class Solution:
    def maxScore(self, s: str) -> int:
        cnt = Counter
(s)
        
        zcnt = 0
        ocnt = 0
        result = 0
        for c in s[:-1]:
            if c == '0':
                zcnt += 1
            else:
                ocnt += 1
            result = max(result, zcnt + cnt['1'] - ocnt)
        return result
