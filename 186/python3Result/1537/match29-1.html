class Solution:
    def maxScore(self, s: str) -> int:
        c1 = collections.Counter(s)
['1']
        
        to_ret = 0
        for c in s[:-1] :
            if c == '0' :
                c1 += 1
            elif c == '1' :
                c1 -= 1
            to_ret = max(to_ret, c1)
        return to_ret
