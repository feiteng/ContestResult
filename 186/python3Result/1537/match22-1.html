class Solution:
    def maxScore(self, s: str) -> int:
        num0, num1 
= 0, 0
        for c in s:
            if c == '0':
                num0 += 1
            else:
                num1 += 1
        ans = 0
        left0 = 0
        for i in range(len(s) - 1):
            if s[i] == '0':
                left0 += 1
            else:
                num1 -= 1
            ans = max(ans, left0 + num1)
        return ans
