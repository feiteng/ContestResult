class Solution:
    def maxScore(self, s: str) -> int:
        ans = 0;
        for i in range(1, len(s)):
            sub_ans = 0;
            left = s[:i]
            right = s[i:]
            for x in left:
                if x == '0':
                    sub_ans += 1
            for x in right:
                if x == '1':
                    sub_ans += 1
            ans = max(ans, sub_ans)
        return ans
