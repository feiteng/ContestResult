class Solution:
    def maxScore(self, s: str) -> int:
        oneCount = 0
        for n in s:
            if n == '1': oneCount += 1
        
        res = 0
        rScore = oneCount
        lScore = 0
        for i in range(len(s)-1):
            if s[i] == '0':
                lScore += 1
            else:
                rScore -= 1
            res = max(res,lScore + rScore)
        return res
