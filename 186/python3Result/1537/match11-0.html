class Solution:
    def maxScore(self, s: str) -> int:
        res = 0
        for i in range(1, len(s)):
            tmp = 0
            tmp1 = s[:i]
            tmp2 = s[i:]
            for j in tmp1:
                if j == '0':
                    tmp += 1
            for j in tmp2:
                if j == '1':
                    tmp += 1
            res = max(res, tmp)
        return res
