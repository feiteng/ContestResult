class Solution:
    def maxScore(self, s: str) -> int:
        score = 0
        
        for c in s:
            if c == '1':
                score += 1
        
        if s[0] == '1':
            score -= 1
        else:
            score += 1
        
        best = score
        
        for i in range(1, len(s)-1):
            if s[i] == '0':
                score += 1
            else:
                score -= 1
            
            best = max(best, score)
        
        return best
