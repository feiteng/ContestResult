class Solution:
    def maxScore(self, s: str) -> int:
        s = list(s)
        s_1 = 0
        s_0 = 0
        for i in s:
            if i=='0':
                s_0+=1
            else:
                s_1+=1
                
        sm = 0
        m = -1
        
        for i in range(len(s)-1):
            if s[i]=='0':
                sm+=1
            else:
                s_1-=1
                
            m = max(m, sm+s_1)
            
        return m
