class Solution:
    def maxScore(self, s: str) -> int:
        zero = 0
        one = 0
        for cha in s:
            if cha == '0':
                zero += 1
            if cha == '1':
                one += 1
        leftOne = 0
        leftZero 
= 0
        ans = 0
        for idx in range(len(s) - 1):
            if s[idx] == '0':
                leftZero += 1
            if s[idx] == '1':
                leftOne += 1
            ans = max(ans, leftZero + one - leftOne)
        return ans
