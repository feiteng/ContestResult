class Solution:
    def maxScore(self, s: str) -> int:
        left = 1 if s[0] == '0' else 0
        right = s[1:].count('1')
        cur = left + right
        ans 
= cur
        for i in range(1,len(s)-1):
            if s[i] == '0': cur += 1
            else: cur -= 1
            ans = max(ans,cur)
        return ans
