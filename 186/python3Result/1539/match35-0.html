class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        n = len(nums)
        p = []
        for i in range(n):
            for j in range(len(nums[i])):
                p.append([i + j, i, nums[i][j]])
        p.sort(key=lambda x: (x[0], -x[1]))
        ans = []
        for i in range(len(p)):
            ans.append(p[i][2])
        return ans
