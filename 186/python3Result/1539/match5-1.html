class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        # n = len(nums)
        # m = {}
        # for i in range(n):
        #     m[i] = len(nums[i])
        # maxxm = max(m.values())
        # total = n + maxxm - 1
        # res = []
        # for tt in range(total):
        #     for i in range(min(0,), ):
        res = []
        for i,vec in enumerate(nums):
            for j, v in enumerate(vec):
                res.append((v,i+j,i)) 
        res.sort(key = lambda x: (x[1],-x[2]))
        tmp = []
        for val,_,_ in res:
            tmp.append(val)
        return tmp
