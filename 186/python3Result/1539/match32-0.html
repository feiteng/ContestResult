class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        
        q = []
        for i, r in enumerate(nums):
            for j, num in enumerate(r):
                heapq.heappush(q, (i+j, j, num))
        
        res = []
        while q:
            _, _, num = heapq.heappop(q)
            res.append(num)
        return res
