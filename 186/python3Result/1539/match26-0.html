class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        d = collections.defaultdict(list)
        n = len(nums)
        for i in range(n):
            for j in range(len(nums[i])):
                d[i + j].append(nums[i][j])
        
        res = []
        
        for c in sorted(list(d.keys())):
            res += d[c][::-1]
        
        return res
