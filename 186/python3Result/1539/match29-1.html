class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        line = defaultdict(list)
        for start in range(len(nums)):
            for i in range(len(nums[start])):
                line[start + i].append(nums[start][i])
        ans = []
        for i in line:
            while line[i]:
                ans.append(line[i].pop())
        return ans
