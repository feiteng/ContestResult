class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        d = collections.defaultdict(list)
        maxval = 0
        for i, arr in enumerate(nums):
            for j, num in enumerate(arr):
                d[i+j].append(num)
                maxval = max(
i+j, maxval)
        res = []
        for i in range(0, maxval+1):
            res.extend(d[i][::-1])
        return res
