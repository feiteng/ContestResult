import heapq


class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        q = []
        for i, row in enumerate(nums):
            for j, x in enumerate(row):
                heapq.heappush(q, (i + j, j, x))
        res = []
        while q:
            _, _, x = heapq.heappop(q)
            res.append(x)
        return res
