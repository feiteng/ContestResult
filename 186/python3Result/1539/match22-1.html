class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        a = []
        for i in range(len(nums)):
            for j in range(len(nums[i])):
                a.append((nums[i][j], (i + j) * 1000000 - i))
        a.sort(key=lambda x:x[1])
        ret = []
        for x in a:
            ret.append(x[0])
        return ret
