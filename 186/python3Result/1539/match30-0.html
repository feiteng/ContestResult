class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        ans = []
        for row_idx, row in enumerate(nums):
            for col_idx, num in enumerate(row):
                ans.append((row_idx + col_idx, -row_idx, num))
        ans.sort()
        return [num for _, _, num in ans]
