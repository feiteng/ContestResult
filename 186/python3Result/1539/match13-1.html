class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        q = []
        for i, ll in enumerate(nums):
            for j, x in enumerate(ll):
                q.append(((i+j), j, x))
        q.sort()
        sol = [a[2] for a in q]
        return sol
