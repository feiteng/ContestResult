class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        ret = max(nums)
        if ret < 0:
            return ret
        q = []
        t = 0
        for i in range(n):
            while t < len(q) and i - q[t][1] > k:
                t += 1
            now = nums[i]
            if t < len(q):
                now += q[t][0]
            if now > 0:
                ret = max(now, ret)
                while len(q) > t and q
[-1][0] <= now:
                    q.pop()
                q.append((now, i))
        return ret
