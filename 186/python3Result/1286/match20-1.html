from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        # f = []
        q = deque([(0, nums[0])])
        ans = nums[0]
        for i, x in enumerate(nums[1:], 1):
            if i - q[0][0] > k:
                q.popleft()
            # print(q)
            fval = x
            if q[0][1] > 0:
                fval += q[0][1]
            ans 
= max(ans, fval)
            while q and q[-1][1] <= fval:
                q.pop()
            q.append((i, fval))
        return ans
