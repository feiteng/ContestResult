import queue
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        pq = queue.PriorityQueue()
        
        to_ret = nums[0]
        pq.put([-nums[0], 0])
        for i in range(1, len(nums)) :
            while True :
                a = pq.get()
                if not i-a[1] > k :
                    break
            pq.put(a)
            now = nums[i]-a
[0] if a[0] < 0 else nums[i]
            # print([now, i], a)
            to_ret = max(to_ret, now)
            pq.put([-now, i])
        return to_ret
