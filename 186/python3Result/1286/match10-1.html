class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        q = [(-nums[0], 0)]
        sol = nums[0]
        
for i in range(1, n):
            while q and q[0][1] < i - k:
                heapq.heappop(q)
            v = nums[i] + max(0, -q[0][0] if q else 0)
            sol = max(sol, v)
            heapq.heappush(q, (-v, i))
        return sol
