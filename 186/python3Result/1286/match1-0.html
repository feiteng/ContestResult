class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        q = []
        h = 0
        s = -1e10
        for i in range(len(nums)):
            while h < len(q) and i - q[h][0] > k:
                h += 1
            v = nums[i]
            if h < len(q):
                v = max(q[h][1] + nums[i], v)
            s = max(s, v)
            while h < len(q) and v >= q[-1][1]:
                q.pop()
            q.append((i, v))
        return s
