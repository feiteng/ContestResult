class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        
        d = collections.deque([(-1, float('-inf'))])
        ans = max(nums)
        for i, n in enumerate(nums):
            idx, lastKSumMax = d[0]
            newSum = max(lastKSumMax, 0) + n
            ans = max(ans, newSum)
            while d and d[-1][1] < newSum:
                d.pop()
            d.append((i, newSum))
            if d[0][0] == i - k:
                d.popleft()
        return ans
