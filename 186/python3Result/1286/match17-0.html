import collections


class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        q = collections.deque()
        n = len(nums)
        res = -float('inf')
        vals = []
        for i in range(n):
            if i == 0:
                q.append(i)
                vals.append(nums[i])
                res = nums[i]
                continue
            vals.append(max(vals[q[0]], 0) + nums[i])
            res = max(res, vals[-1])
            while q and vals[q[-1]] <= vals[i]:
                q.pop()
            q.append(i)
            if q[0] == i - k:
                q.popleft()
        return res
