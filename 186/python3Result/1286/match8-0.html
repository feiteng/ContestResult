class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = [nums[0]]
        hp = [(-nums[0],0)]
        for i in range(1, len(nums)):
            while hp[0][1] < i-k: heapq.heappop(hp)
            dp.append(nums[i] + max(0, -hp[0][0]))
            heapq.heappush(hp, (-dp[-1], i))
        return max(dp)
