class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        ans=nums[0]
        h=[(-nums[0],0)]
        for i in range(1,len(nums)):
            n=nums[i]
            while h[0][1]<i-k:
                heapq.heappop(h)
            tmp=max(-h[0][0]+n,n)
            ans=max(ans,tmp)
            heapq.heappush(h,(-tmp,i))
        return ans
