class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = [-10**10]*len(nums)
        dp[0] = nums[0]
        m = nums[0]


        for i in range(1, len(nums)):
            for j in range(i-1, i-k-1, -1):
                dp[i] = max(dp[i], dp[j]+nums[i], nums[i])
                m = max(m, dp[i])
                if nums[j]>0:
                    break
                
        return m
