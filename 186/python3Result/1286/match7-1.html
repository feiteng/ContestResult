from heapq import *

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [float('-inf')] * n
        
        h = []
        
        for i, num in enumerate(nums):
            while h and i - h[0][1] > k:
                heappop(h)
                
            preMax = 0
            if h:
                preMax = max(-h[0][0], 0)
            
            dp
[i] = max(dp[i], preMax + nums[i])
            
            heappush(h, (-dp[i], i))
            
        return max(dp)
