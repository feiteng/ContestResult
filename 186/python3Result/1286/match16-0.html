class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] + [k for k in nums]
        dp[0] = 0
        dp[1] = nums[0]
        hq = [(-dp[1],1)]
        for i in range(2, n + 1):
            while hq:
                if i - hq[0][1] <= k:
                    break 
                heapq.heappop(hq)
            dp[i] = max(dp[i], -hq[0][0] + nums[i-1])
            heapq.heappush(hq,(-dp[i],i))
        
        dp[0] = -float(inf)
        return max(dp)
