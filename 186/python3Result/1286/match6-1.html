class Solution:
    def constrainedSubsetSum(self, nums: List
[int], k: int) -> int:
        
        dp = nums[:]
        
        for i in range(len(nums) - 1, -1, -1):            
            for j in range(i + 1, i + k + 1):
                dp[i] = max(dp[i], nums[i] + (dp[j] if j < len(nums) else 0))

                if j >= len(nums) or nums[j] >= 0:
                    break
        
        #print(dp)
        ans = max(dp)
        
        return ans
