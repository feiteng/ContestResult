class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        # dp0[i] select i th element, the max sum
        # dp1[i] until ith element, the max sum
        """
        dp0[i] = max(dp0[i-1], ... dp0[i-k]) + nums[i] for 
        return max(dp0)
        """
        n = len(nums)
        dp = [float('-inf')] * (n+1)
        dp[0] = 0
        # sliding window max 
        q = collections.deque()
        q.append((0, 0)) # index, val 
        for i in range(1, n+1):
            # for j in range(i-k, i):
            #     dp[i] = max(dp[i], dp[j]) 
            dp[i] = max(0,q[0][1])
            dp[i] += nums
[i-1]
            # enqueue:
            while q and q[-1][1] <= dp[i]:
                q.pop()
            q.append((i, dp[i]))
            # dequeue:
            if q[0][0] <= i-k:
                q.popleft()
        # print(dp)
        return max(dp[1:])
