import heapq

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        maxResult = nums[0]
        queue = [(-nums[0], 0)]
        for i in range(1, len(nums), 1):
            negMaxResult, index = queue[0]
            while i - index > k:
                heapq.heappop(queue)
                negMaxResult, index = queue[0]
            currentResult = nums[i] + (- negMaxResult if -negMaxResult > 0 else 0)
            maxResult = max(maxResult, currentResult)
            heapq.heappush(queue, (-currentResult, i))
        return maxResult
