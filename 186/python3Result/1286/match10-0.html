class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        res=nums[0]
        T=[]
        for i,v in enumerate(nums):
            while T and T[0][1]+k<i: T.pop(0)
            t=v+ (0 if not T else max(0,T[0][0]))
            while T and T[-1][0]<=t: T.pop()
            res=max(res, t)
            T.append((t,i))
            # print(res, t, T)
        return res
