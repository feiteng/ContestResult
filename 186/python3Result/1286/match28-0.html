class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        
        if all(x < 0 for x in nums):
            return max(nums)
        
        
        qe = deque()
        ret = 0
        
        for i, x in enumerate(nums):
            
            while qe and i - qe[0][1] > k:
                qe.popleft()
            
            
            bans = 0 if not qe else qe[0][0]
            
            ans = max(0, x+bans)
            ret = max(ret, ans)
            if ans > 0:
                while qe and qe[-1][0] < ans:
                    qe.pop()
                    
                qe.append((ans, i))
        return ret
