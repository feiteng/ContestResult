from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        N = len(nums)
        dp = [0] * N
        q = deque([])
        ans = -float('inf')
        
        for i in range(N):
            
            if q and q[0] < i - k:
                q.popleft()
            dp[i] = nums[i]
            if q and dp[q[0]] > 0:
                dp[i] += dp[q[0]]
            ans = max(ans, dp[i])
            while q and dp[q[-1]] <= dp[i]:
                q.pop()
            q.append(i)
        return ans
