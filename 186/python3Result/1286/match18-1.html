class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        
        dp = [float('-inf')] * len(nums)
        
        dp[0] = nums[0]
        stack = [0]
        for i in range(1,len(nums)):
            while stack and i - stack[0] > k:
                stack.pop(0)
            max_value = dp[stack[0]]
            dp[i] 
= max(0 , max_value) + nums[i]
            while stack and dp[stack[-1]] <= dp[i]:
                stack.pop()
            stack.append(i)
            # print(dp)
            # print(stack)
        return max(dp)
