class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = [0] * len(nums)
        for i, x in enumerate(nums):
            dp[i] = x

        S = [(-nums[0], 0)]
        for i in range(1, len(nums)):
            while S[0][1] < i - k: heapq
.heappop(S)
            dp[i] = max(dp[i], dp[i] - S[0][0])
            heapq.heappush(S, (-dp[i], i))
        return max(dp)
