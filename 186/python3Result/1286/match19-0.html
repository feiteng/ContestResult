class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        res = max(nums)
        if res <= 0: return res
        s = 0
        while nums[s] <= 0: s += 1
        res = nums[s]
        que = collections.deque([(res, s)])
        for i in range(s + 1, len(nums)):
            while que and (i - que[0][1] > k):
                que.popleft()
            s = max((que[0][0] if que else 0) + nums[i], 0)
            res = max(res, s)
            while que and s >= que[-1][0]:
                que.pop()
            que.append((s, i))
        return res
