class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        ans = nums[0]
        prev_sums = collections.deque([(0, nums[0])])
        for end_idx in range(1, n):
            while prev_sums[0][0] < end_idx - k:
                prev_sums.popleft()
            #print(end_idx, prev_sums)
            max_sum = max(prev_sums[0][1], 0) + nums[end_idx]
            #print(end_idx, max_sum, prev_sums, prev_sums[0][1])
            while len(prev_sums) > 0 and (prev_sums[0][1] <= max_sum or prev_sums[0][0] <= end_idx - k):
                prev_sums.popleft()
            prev_sums.append((end_idx, max_sum))
            ans = max(ans, max_sum)
        return ans
