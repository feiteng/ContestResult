class MaxQueue:
    def __init__(self):
        self.q_ = collections.deque()

    def push(self, e):
        while self.q_ and e > self.q_[-1]: self.q_.pop()
        self.q_.append(e)

    def pop(self):
        self.q_.popleft()

    def max(self):
        return self.q_[0]

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        
        n = len(nums)
        dp = [0] * n
        dp[-1] = nums[-1]
        q = MaxQueue()
        q.push(nums[-1])
        for i in range(n -2, -1, -1):
            
            
            dp[i] = max(nums[i],  nums[i] + q.max())
            if i <= n - k - 1:
                if dp[i + k] == q.max(): q.pop()
                print (i, i+k)
            q.push(dp[i])
        
        # print (dp)
        return max(dp)
