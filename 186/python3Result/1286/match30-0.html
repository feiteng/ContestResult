class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        l = len(nums)
        st = 0
        ed = l
        for i in range(l):
            if nums[i] > 0:
                st = i
                break
        i = l - 1
        while i > st:
            if nums[i] > 0:
                ed = i + 1
                break
            i -= 1
        if i == st:
            return max(nums)
        nums = nums[st:ed]
        l = ed - st
        re = [0,nums[0]]
        m = []
        index = []
        for i in range(2,l + 1):
            if index and index[0] < i - k:
                m.pop(0)
                index.pop(0)
            while m and re[-1] > m[-1]:
                m.pop()
                index.pop()
            m += [re[-1]]
            index += [i - 1]
            re  += [max(0,m[0]) + nums[i-1]]

        return re[-1]
