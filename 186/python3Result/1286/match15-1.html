class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        result = -float("inf")
        dp = MaxQueue()
        for x in nums:
            r = x + (max(dp.getMax(), 0) if dp else 0)
            result = max(result, r)
            dp.append(r)
            if len(dp) > k:
                dp.popleft()
        return result
