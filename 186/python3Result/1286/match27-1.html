class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = [-math.inf] * len(nums)
        heap = []
        dp[0] = nums[0]
        heapq.heappush(heap, (-nums[0], 0))
        res = nums[0]
        for i in range(1, len
(nums)):
#             print(heap)
            while i - heap[0][1] > k:
                heapq.heappop(heap)
            largest = max(0, -heap[0][0])
            heapq.heappush(heap, (-largest - nums[i], i))
            res = max(res, -heap[0][0])
        return res
