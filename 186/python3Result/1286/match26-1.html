import heapq


class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        
        res = nums[0]
        
        heap = []
        heapq.heappush(heap, (-nums[0], 0))
        
        for i in range(1, n):
            v, j = heap[0]
            while i - j > k:
                heapq.heappop(heap)
                v, j = heap[0]
            
            v = -v
            m = nums[i]
            
            if v > 0:
                m += v
                
            heapq.heappush(heap, (-m, i))
            res = max(res, m)
        
        return res
