class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        # keep a queue that tracks max
        dp = [-float('inf')] * len(nums)
        dp[-1] = nums[-1]
        q = deque([(dp[-1], len(nums)-1)])
        for i in range(len(nums)-1)[::-1]:
            dp[i] = max(nums[i], q[0][0]+nums[i])
            while q and q[-1][0] < dp[i]:
                q.pop()
            q.append((dp[i],i))
            if q[0][1] - i == k:
                q.popleft()
        return max(dp)
    
    def constrainedSubsetSum2(self, nums: List[int], k: int) -> int:
        dic = {}
        def backtrack(i, currIdx):
            if i == len(nums):
                return 0
            ret = -float('inf')
            # use
            if nums[i]-1 in dic and currIdx - dic[nums[i]-1] > k:
                pass
            else:
                dic[nums[i]] = currIdx
                ret = max(ret, nums[i] + backtrack(i+1, currIdx+1))
                del dic[nums[i]]
            ret = max(ret, backtrack(i+1, currIdx))
            return ret
        return backtrack(0, 0)
