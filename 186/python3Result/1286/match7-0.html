class Solution:
    def constrainedSubsetSum(self, nums, k):
        opt, pq = [0] * len(nums), []
        for i, v in enumerate(nums):
            while pq and i + pq[0][1] > k:
                heappop(pq)
            opt[i] = v - min(0, pq[0][0]) if pq else v
            heappush(pq, (-opt[i], -i))
        return max(opt)
