class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n,f,j=len(nums),[(-1,0)],0
        ans=-1e99
        for i in range(n):
            x=nums[i]
            while f[j][0]<i-k:
                j+=1
            temp=max(f
[j][1],0)+x
            ans=max(ans,temp)
            while f and f[-1][1]<=temp:
                f.pop()
            f.append((i,temp))
            if j>=len(f):
                j=len(f)-1
        return ans
