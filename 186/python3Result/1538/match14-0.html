class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        ans = curr = sum(cardPoints[:k])
        for i in range(k):
            curr -= cardPoints[k - i - 1]
            curr += cardPoints[~i]
            ans = max(ans, curr)
        return ans
