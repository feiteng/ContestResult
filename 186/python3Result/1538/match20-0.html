class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        
        ret = 0
        left_sum = 0
        right_sum = sum(cardPoints[-k:])
        ret = left_sum + right_sum
        for i in range(0, k):
            #ret = max(ret, sum(cardPoints))
            left_sum += cardPoints[i]
            right_sum -= cardPoints[-(k-i)]
            ret = max(ret, left_sum + right_sum)
        
        return ret
