class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        
        n = len(cardPoints)
        total = sum(cardPoints)
        
        l = n - k
        
        if l <= 0:
            return total
        
        s_ = sum(cardPoints[:l])
        ans = s_
        for i in range(l, n):
            s_ += cardPoints[i]
            s_ -= cardPoints[i - l]
            ans = min(s_,  ans)
        return total - ans
