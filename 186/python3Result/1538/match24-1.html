class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        ss = [0] * (n+1)
        for i in range(n):
            ss[i+1] = ss[i] + cardPoints[i]
        ans = 0
        for i in range(k+1):
            ans = max(ans, ss[n] - ss[n-k+i] + ss[i])
        return ans
