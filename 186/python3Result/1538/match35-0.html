class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        c = cardPoints
        ans = 0
        temp = sum(c[n-k:])
        left = 0
        ans = temp
        for i in range(1,k+1):
            left += c[i-1]
            temp -= c[n-k+i-1]
            ans = max(ans,left+temp)
        return ans
