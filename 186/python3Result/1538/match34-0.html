class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        
        m = n - k
        total = sum(cardPoints)
        if k == n:
            return total
        
        i, j = 0, m-1
        s = sum(cardPoints[0:m])
        res = s
        i += 1
        j += 1
        while j < n:
            s += cardPoints[j] - cardPoints[i-1]
            res = min(s, res)
        
            i += 1
            j += 1
        
        return total - res
