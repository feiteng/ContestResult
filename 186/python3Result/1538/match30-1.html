class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        s = sum(cardPoints)
        remain 
= sum(cardPoints[:n-k])
        res = s - remain
        for i in range(1,k+1):
            remain -= cardPoints[i-1]
            remain += cardPoints[n-k+i-1]
            res = max(res, s-remain)
        return res
