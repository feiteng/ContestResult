class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        curSum = 0
        for i in range(k):
            curSum += cardPoints[i]
        
        res = curSum
        # shift to right one by one
        for i in range(k):
            curSum -= cardPoints[k-i-1]
            curSum += cardPoints[-i-1]
            res = max(res,curSum)
        return res
