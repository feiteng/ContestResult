class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        
        ans = cur = sum(cardPoints[:k])
        
        for i in range(1, k+1):
            cur -= cardPoints[k-i]
            cur += cardPoints[-(i)]
            ans = max(ans, cur)
        return ans
