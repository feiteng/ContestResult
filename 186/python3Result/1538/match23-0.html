class Solution:
    def maxScore(self, card: List[int], k: int) -> int:
        pre = [0]
        for c in card:
            pre.append(pre[-1] + c)
        post = [0]
        for c in card[::-1]:
            post.append(post[-1] + c)
        
        res = 0
        for i in range(k + 1):
            res = max(res, pre[i] + post[k - i])
        
        return res
