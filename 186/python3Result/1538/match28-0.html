class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        l = len(cardPoints)
        re = sum(cardPoints[0:k])
        tem = re
        j = k - 1
        i = -1
        while j > -1:
            tem = tem - cardPoints[j] + cardPoints[i]
            re = max(re,tem)
            j -= 1
            i -= 1
        return re
