class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        curscore 
= sum(cardPoints[:k])
        maxscore= curscore
        for i in range(1, k + 1):
            curscore += cardPoints[-i]
            curscore -= cardPoints[k - i]
            maxscore = max(maxscore, curscore)
        return maxscore
