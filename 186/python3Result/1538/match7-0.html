class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        max_val = sum(cardPoints[:k])
        sum_prev = max_val
        for i in range(1,k+1):
            sum_prev = sum_prev-cardPoints[k-i]+cardPoints[-1*i]
            max_val = max(max_val,sum_prev)
        return max_val
