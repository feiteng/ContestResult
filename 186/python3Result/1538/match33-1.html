class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        if cardPoints is None or len(cardPoints) == 0:
            return 0
        
        n = len(cardPoints)
        if k > n:
            return 0
        
        score 
= 0
        for i in range(k):
            score += cardPoints[i]
        
        mScore = score
        
        # try other points, which takes cards from the end
        for i in range(k):
            score += cardPoints[n - i - 1]
            score -= cardPoints[k - i - 1]
            mScore = max(mScore, score)
        
        return mScore
