class Solution:
    def maxScore(self, c: List[int], k: int) -> int:
        n = len(c)
        tot = [0] * (n + 1)
        for i in range(n):
            tot[i + 1] = tot[i] + c[i]
        ret = 0
        for i in range(k + 1):
            ret = max(ret, tot[i] + tot[n] - tot[n - (k - i)])
        return ret
