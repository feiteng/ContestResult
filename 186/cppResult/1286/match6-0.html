class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int size = nums.size();
        multiset<int> container;
        
        vector<int> records(size);

        int maximum = INT_MIN;
        for (int i = 0; i < size; i++) {
            
            int count = nums[i];
            if (!container.empty()) {
                auto pre = container.rbegin();
                if (*pre > 0)
                    count += *pre;
            }

            container.insert(count);
            if (container.size() > k)
                container.erase(records[i-k]);
            
            records[i] = count;

            maximum = max({nums[i], maximum, count});
        }

        return maximum;
    }
};
