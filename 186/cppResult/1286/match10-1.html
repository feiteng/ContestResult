class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        vector<int> res(nums.size(), 0);
        deque<int> dq;
        res[0] = nums[0];
        int res_num = res[0];
        dq.push_back(0);
        for(int i = 1; i < nums.size(); i++){
            // cout << dq.front()<< " "<< i << endl;
            if(dq.size() && dq.front() < i-k)dq.pop_front();
            res[i] = nums[i] + max(0, res[dq.front()]);
            res_num = max(res_num, res[i]);
            cout << res[i] <<" ";
            while(dq.size() && res[dq.back()] < res[i])dq.pop_back();
            cout << dq.size() << endl;
            dq.push_back(i);
        }
        return res_num;
    }
};
