class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        priority_queue<pair<int, int>> q;
        int n = nums.size();
        q.emplace(0, n);
        int ans = nums[0];
        q.emplace(ans, 0);
        
        for (int i = 1; i < n; i++) {
            while (!q.empty() && q.top().second + k < i) {
                q.pop();
            }
            // printf("%d %d %d\n", i, q.top().first + nums[i], q.top().second);
            int cur = q.top().first + nums[i];
            q.emplace(cur, i);
            ans = max(ans, cur);
        }
        
        return ans;
    }
};
