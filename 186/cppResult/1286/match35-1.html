class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int ans = -100000000;
        priority_queue<pair<int, int>> q;
        for (int i = 0; i < nums.size(); i++) {
            bool done = false;
            while(!q.empty()) {
                pair<int, int> a = q.top();
                if (i - a.second > k) {
                    q.pop();
                    continue;
                } else {
                    pair<int, int> now;
                    now.first = a.first + nums[i];
                    now.second = i;
                    ans = max(ans, now.first);
                    q.push(now);
                    break;
                }
            }
            q.push(make_pair(nums[i], i));
            ans = max(ans, nums[i]);
        }
        return ans;
    }
};
