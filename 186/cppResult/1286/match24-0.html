class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> v(n);
        multiset<int> s;
        int ans = INT_MIN;
        for (int i=0; i<n; i++) {
            if (i - k - 1 >= 0) s.erase(s.find(v[i-k-1]));
            v[i] = nums[i];
            if (!s.empty()) v[i] = max(v[i], (*s.rbegin()) + nums[i]);
            ans = max(ans, v[i]);
            s.insert(v[i]);
        }
        return ans;
    }
};
