class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        map<int, int> maxWindow;
        vector<int> dp(n, 0);
        int ans = INT_MIN;
        for(int i = 0; i < n; i++) {
            dp[i] = nums[i];
            if(!maxWindow.empty()) {
                dp[i] = max(dp[i], nums[i] + maxWindow.rbegin()->first);
            }
            maxWindow[dp[i]]++;
            if(i >= k) {
                auto it = maxWindow.find(dp[i-k]);
                assert(it != maxWindow.end());
                if(--it->second == 0) {
                    maxWindow.erase(it);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
