class Solution {
public:
    int dp[100005];
int constrainedSubsetSum(vector<int>& nums, int k) {
    deque<int> q;
    int n = nums.size();

    dp[0] = -1000000007;
    q.push_back(0);
    int res = dp[0];
    for (int i = 0; i < nums.size(); i++) {
        while (!q.empty() && (i+1) - q.front() > k) q.pop_front();
        dp[i+1] = max(0, dp[q.front()]) + nums[i];
        while (!q.empty() && dp[q.back()] < dp[i+1]) q.pop_back();
        q.push_back(i+1);

        res = max(res, dp[i+1]);
    }
    return res;
}
};
