class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        multiset<int> s;
        int n = nums.size();
        vector<int> dp(n);
        dp[0] = nums[0];
        s.insert(nums[0]);
        for(int i=1;i<n;i++){
            int m = *s.rbegin();
            int c = nums[i];
            if(m > 0) c += m;
            dp[i] = c;
            s.insert(c);
            if(s.size() > k){
                auto itr = s.find(dp[i-k]);
                s.erase(itr);
            }
        }
        return *max_element(dp.begin(), dp.end());
    }
};
