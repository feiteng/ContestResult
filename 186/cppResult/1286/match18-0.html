class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int res = INT_MIN;
        deque<pair<int, int>> prev;
        for (int i = 0; i < nums.size(); ++i) {
            //remove the old
            if (!prev.empty() && prev.front().first < i - k) {
                prev.pop_front();
            }
            int last = prev.empty() ? 0 : prev.front().second;
            last = max(last, 0);
            int cur = last + nums[i];
            res = max(res, cur);
            while (!prev.empty() && prev.back().second <= cur) {
                prev.pop_back();
            }
            prev.push_back({i, cur});
        }
        return res;
    }
};
