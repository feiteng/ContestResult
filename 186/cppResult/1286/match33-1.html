class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        deque<pair<int, int>> q;
        int result = nums[0];
        for (int i = 0; i < n; i++) {
            if (i == 0) {
                q.push_back({nums[i], i});
                result = max(result, nums[i]);
            } else {
                int prevMax = q.front().first;
                int cur = max(nums[i] + prevMax, nums[i]);
                result = max(result, cur);
                while (!q.empty() && q.back().first <= cur) {
                    q.pop_back();
                }
                q.push_back({cur, i});
                if (i >= k) {
                    if (q.front().second == i - k) {
                        q.pop_front();
                    }
                }
            }
        }
        return result;
    }
};
