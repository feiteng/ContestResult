class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        vector<int> ln(nums.size(), 0);
        ln[0] = nums[0];
        map<int, int> vc;
        vc[nums[0]]++;
        int r = ln[0];
        for (int i = 1; i < nums.size(); i++) {
            ln[i] = nums[i];
            int m = vc.crbegin()->first;
            if (m > 0) {
                ln[i] += m;
            }
            vc[ln[i]]++;
            if (i >= k) {
                auto iter = vc.find(ln[i - k]);
                iter->second--;
                if (iter->second == 0) {
                    vc.erase(iter);
                }
            }
            r = max(r, ln[i]);
        }
        return r;
    }
};
