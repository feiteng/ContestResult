class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n=nums.size();
        deque<pair<int,int>> rec;
        int ans = nums[0];
        for(int i=0;i<n;i++){
            ans = max(ans,nums[i]);
            
            while(!rec.empty()&&i-rec.front().first>k)
                rec.pop_front();
            int x = 0;
            if(!rec.empty())x=rec.front().second;
            x = max(nums[i],x+nums[i]);
            while(!rec.empty()&&x>rec.back().second){
                rec.pop_back();
            }
            rec.push_back(make_pair(i,x));
            // interval++;
            ans=max(ans,rec.back().second);
            
        }
        return ans;
    }
};
