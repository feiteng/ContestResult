class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int ans = *max_element( nums.begin(), nums.end());
        deque<int> D;
        deque<int> index;
        
        D.push_back(0);
        index.push_back(nums.size());
        
        for(int i=nums.size()-1;i>=0;--i){
            int cur = nums[i];
            while( index.front() - i > k ){
                index.pop_front();
                D.pop_front();
            }
            cur = max( cur , cur + D.front() );
            ans = max( cur, ans);
            while( !D.empty() and cur >= D.back()){
                D.pop_back();
                index.pop_back();
            }
            D.push_back(cur);
            index.push_back(i);
        }
        return ans;
    }
};
