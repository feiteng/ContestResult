class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        multiset<int> s;
        vector<int> f;
        int ans = -1e9 + 1;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            if (i > k) {
                s.erase(f[i - k - 1]);
            }
            f.push_back(nums[i]);
            if (s.size() > 0) {
                int last_max = *s.rbegin();
                //printf("%d %d\n", i, last_max);
                f[i] = max(f[i], f[i] + last_max);
            }
            ans = max(f[i], ans);
            s.insert(f[i]);
        }
        return ans;
    }
};
