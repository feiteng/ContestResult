class Solution {
public:
	int constrainedSubsetSum(vector<int> &nums, int k) {
		int n = nums.size();
		vector<int> dp(n);
		deque<pair<int, int>> q;
		for (int i = 0; i < n; i++) {
			dp[i] = nums[i];
			while (!q.empty() && i - q.front().second > k)
				q.pop_front();
			if (!q.empty())
				dp[i] = max(dp[i],dp[i] + q.front().first);
			while (!q.empty() && q.back().first <= dp[i])
				q.pop_back();
			q.push_back( { dp[i], i });
		}
		return *max_element(dp.begin(), dp.end());
	}
};
