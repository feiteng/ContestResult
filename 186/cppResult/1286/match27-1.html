const int INF = 0x3f3f3f3f;

class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int ans = -INF;
        deque< pair<int, int> > dq;
        for (int i=0; i<nums.size(); i++) {
            // get rid of idx
            while (!dq.empty() && dq.back().first < i-k)
                dq.pop_back();
            int dpVal = nums[i];
            if (!dq.empty() && dq.back().second > 0)
                dpVal += dq.back().second;
            // printf("%d : %d\n", i, dpVal);
            ans = max(ans, dpVal);
            // insert current dpVal
            while (!dq.empty() && dq.front().second < dpVal)
                dq.pop_front();
            dq.push_front(make_pair(i, dpVal));
        }
        return ans;
    }
};
