class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) 
    {
        multiset<int>Set;
        Set.insert(0);
        queue<int>q;
        q.push(0);
        int ret = INT_MIN;
        
        for (int i=0; i<nums.size(); i++)
        {
            if (q.size()>k)
            {
                int t=q.front();
                q.pop();
                Set.erase(Set.lower_bound(t));                
            }
                        
            int cur = max(nums[i],*(Set.rbegin())+nums[i]);
            ret = max(ret, cur);
            q.push(cur);
            Set.insert(cur);
        }
        
        return ret;
    }
};
