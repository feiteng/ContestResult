class Solution {
public:
    int constrainedSubsetSum(vector<int>& v, int k) {
        int n = v.size();
        vector <int> f(n + 5, 0);
        multiset <int> s;
        f[0] = v[0];
        int ans = v[0];
        s.insert(v[0]);
        for (int i = 1; i < n; i++) {
            if (i - k - 1 >= 0) {
                s.erase(s.find(f[i - k - 1]));
            }
            f[i] = max(0, *s.rbegin()) + v[i];
            s.insert(f[i]);
            ans = max(ans, f[i]);
        }
        return ans;
    }
};
