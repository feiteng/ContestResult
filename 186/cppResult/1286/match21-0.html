class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n);
        multiset<int, greater<int>> ms;
        int ans = INT_MIN;
        for(int i = 0; i < n; i += 1){
            ans = max(ans, dp[i] = nums[i] + (ms.empty() ? 0 : max(0, *ms.begin())));
            ms.insert(dp[i]);
            if(i - k >= 0) ms.erase(ms.find(dp[i - k]));
        }
        return ans;
    }
};
