class Solution {
public:
    int constrainedSubsetSum(vector<int> A, int k) {
		deque<int> mono;
		int n = A.size();
		vector<int> dp(n, 0);
		dp[0] = A[0];
		mono.push_back(0);
		for (int i = 1; i < n; ++i) {
			while (mono.size() && mono.front() < i - k)
				mono.pop_front();
			int max_cur = mono.size() ? dp[mono.front()] : 0;
			dp[i] = max(max_cur,0) + A[i];
			while (!mono.empty() && dp[i] > dp[mono.back()]) {
				mono.pop_back();
			}
			mono.push_back(i);
		}
		return *max_element(dp.begin(), dp.end());
	}
};
