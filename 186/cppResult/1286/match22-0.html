class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n + 1, 0);
        int ans = nums[0];
        dp[0] = nums[0];
        deque<int> dq;
        dq.push_back(0);
        for (int i = 1; i < n; i++) {
            while (!dq.empty() && dq.front() < i - k) dq.pop_front();
            int preMax = (dq.empty() ? 0 : dp[dq.front()]);
            dp[i] = nums[i] + (preMax > 0 ? preMax : 0);
            while (!dq.empty() && dp[i] > dp[dq.back()]) dq.pop_back();
            dq.push_back(i);
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
