class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        deque<int> monoQ;
        vector<int> dp(nums.size(), INT_MIN);
        dp[0] = nums[0];
        monoQ.push_back(0);
        int ans = dp[0];
        for (int i = 1; i < nums.size(); i++) {
            dp[i] = nums[i];
            while (!monoQ.empty() && monoQ.front() + k < i) {
                monoQ.pop_front();
            }
            if (!monoQ.empty() && dp[monoQ.front()] > 0) {
                dp[i] += dp[monoQ.front()];
            }
            while (!monoQ.empty() && dp[monoQ.back()] <= dp[i]) {
                monoQ.pop_back();
            }
            //printf("%d, %d\n", i, dp[i]);
            monoQ.push_back(i);
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
