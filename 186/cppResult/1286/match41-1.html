class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        multiset<int> s;
        int n = nums.size();
        vector<int> dp(n);
        for (int i = 0; i < n; ++i) {
            int pre = 0;
            if (s.size())
                pre = max(pre, *s.rbegin());
            int cur = nums[i] + pre;
            dp[i] = cur;
            s.insert(cur);
            if (s.size() > k)
                s.erase(dp[i - k]);
        }
        
        return *max_element(dp.begin(), dp.end());
    }
};
