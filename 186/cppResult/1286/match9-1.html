class Solution {
public:
    int f[100010];
    int constrainedSubsetSum(vector<int>& a, int k) {
        deque<int> Q;
        int n=a.size();
        f[0] = a[0];
        Q.push_back(0);
        int ans = a[0];
        for(int i=1;i<n;i++){
            f[i] = max(a[i],a[i]+f[Q.front()]);
            ans = max(ans,f[i]);
            while(Q.size()&&f[i]>f[Q.back()]){
                Q.pop_back();
            }
            Q.push_back(i);
            while(i-Q.front()>k-1){
                Q.pop_front();
            }
        }
        return ans;
    }
};
