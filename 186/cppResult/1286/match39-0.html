class Solution {
    using pii = pair<int, int>;
    struct cmp{
      bool operator()(pii& a, pii& b){
          return a.first < b.first;
      }
    };
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        int ret = INT_MIN;
        vector<int> dp(n);
        
        priority_queue<pii, vector<pii>, cmp> pq;
        for(int i = 0; i < n; i++) {
            int ma = 0, mi = i-k;
            while(!pq.empty()) {
                pii tmp = pq.top();
                if(tmp.second < mi)
                    pq.pop();
                else {
                    ma = max(ma, tmp.first);
                    break;
                }
            }
            dp[i] = nums[i] + ma;
            pq.push(make_pair(dp[i], i));
            ret = max(ret, dp[i]);
        }
        return ret;
    }
};
