class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int len = nums.size();
        vector<int> dp(len, 0);
        int res = -1e5;
        map<int, int> MM;
        for(int i=0;i<len ;i ++)
        {
            int cur = nums[i];
            if(!MM.empty())
            {
                auto iter = MM.end();
                iter --;
                cur = max(cur , nums[i]+iter->first);
            }

            dp[i] = cur;
            res = max(res, cur);
            //cout<<i << "," <<cur<<endl;
            MM[cur] ++;
            if(i-k>=0)
            {
                MM[dp[i-k]]--;
                if(MM[dp[i-k]] == 0)
                {
                    MM.erase(dp[i-k]);
                }
            }
        }
        return res;
    }
};
