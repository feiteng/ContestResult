class Solution {
public:
    
    int constrainedSubsetSum(vector<int>& nums, int k) {
 /*       int maxN = INT_MIN;
        for (auto n: nums) maxN = max(maxN, n);
        if (maxN <= 0) return maxN;
   */     
        vector<int> dp(nums.size() + 1, INT_MIN);
        dp[nums.size()] = 0;
        multiset<int, greater<int>> q;
        q.insert(0);
        int ret = INT_MIN;
        for (int i = nums.size() - 1; i >= 0; --i) {
            int maxDP = *q.begin();
            dp[i] = max(nums[i], maxDP + nums[i]);
            ret = max(ret, dp[i]);
            q.insert(dp[i]);
            if (i + k <= nums.size()) q.erase(q.equal_range(dp[i + k]).first);
        } 
        return ret;
    }
};
