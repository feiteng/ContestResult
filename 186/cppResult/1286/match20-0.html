#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define FOREACH(i,c) for(__typeof((c).begin())i = (c).begin();i!=(c).end(); ++i)

class Solution {
public:
  int constrainedSubsetSum(vector<int>& nums, int k) {
    int best = nums[0];
    map<int, int> r;
    r[best]++;
    vector<int> sum(nums.size());
    sum[0] = best;
    FOR(i, 1, nums.size() - 1) {
      int v = max(0, r.rbegin()->first);
      sum[i] = v + nums[i];
      best = max(best, sum[i]);
      r[sum[i]]++;
      if (i - k >= 0) {
        r[sum[i - k]]--;
        if (r[sum[i - k]] == 0) {
          r.erase(sum[i - k]);
        }
      }
    }
    return best;
  }
};
