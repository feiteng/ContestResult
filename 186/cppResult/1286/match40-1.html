class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n);
        map<int, int> mp;
        for(int i = 0; i < n; ++i){
            dp[i] = nums[i];
            if(i){
                int mx = mp.rbegin()->first;
                dp[i] = max(dp[i], nums[i] + mx);
            }
            if(i >= k){
                mp[dp[i - k]]--;
                // cout << i << " " << dp[i - k] << endl;
                if(mp[dp[i - k]] == 0){
                    mp.erase(dp[i - k]);
                }
                // cout << "cnt " << mp[dp[i - k]] << endl;
            }
            mp[dp[i]]++;
            cout << dp[i] << " ";
        }
        return *max_element(dp.begin(), dp.end());
    }
};
