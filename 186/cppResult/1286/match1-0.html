class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        priority_queue<pair<int,int>> pq;
        pq.push(make_pair(nums[0],0));
        
        vector<int> dp(n+1,0);
        dp[0]=nums[0];
        int ans = nums[0];
        for(int i=1;i<n;i++)
        {
            // cout<<i-pq.top().secn
            while(i-pq.top().second>k)
                pq.pop();
            int c = 0;
            if(!pq.empty())
                c = pq.top().first;
            if(c<0)c=0;
            pq.push(make_pair(c+nums[i],i));
            dp[i]=c+nums[i];
            ans = max(ans,dp[i]);
        }
        // for(auto i:dp)
        //     cout<<i<<' ';cout<<endl;
        return ans;
        
    }
};
