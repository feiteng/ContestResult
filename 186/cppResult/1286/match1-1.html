class Solution {
public:
    int constrainedSubsetSum(vector<int>& a, int k) {
        int n = a.size();
        deque<pair<int,int>> sums;
        sums.emplace_back(a[0], 0);
        int sumsBegin = 0;
        int result = a[0];
        for(int i=1; i<n; i++) {
            while(!sums.empty() && i-sums.front().second>k) sums.pop_front();
            int newSum = a[i];
            if(!sums.empty()) newSum = max(newSum, sums.front().first + a[i]);
            result = max(result, newSum);
            while(!sums.empty() && sums.back().first<=newSum) sums.pop_back();
            sums.emplace_back(newSum, i);
        }
        return result;
    }
};
