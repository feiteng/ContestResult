class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(nums);
        map<int,int> m;
        m[0]=1;
        for (int i = n-1;i>=0;--i) {
            if (i+1<n) {
                if (m.count(dp[i+1])) {
                    m[dp[i+1]]++;
                } else {
                    m[dp[i+1]]=1;
                }
            }
            if (i+k+1<n) {
                if (m[dp[i+k+1]]==1) m.erase(dp[i+k+1]);
                else --m[dp[i+k+1]];
            }
            auto iter = m.end();
            --iter;
            dp[i]+=(iter->first);
            // int r = 0;
            // for (int j = 1; j<=k&&i+j<n; ++j) r = max(r, dp[i+j]);
            // dp[i]+=r;
        }
        int res = dp[0];
        for (int r:dp) {
            res = max(res,r);
            // cout<<r<<" ";
        }
        // cout<<endl;
        return res;
    }
};
