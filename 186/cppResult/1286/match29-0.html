class Solution {
public: 
    int constrainedSubsetSum(vector<int>& nums, int k) {
        set<pair<int, int>> s;
        int l = nums.size();
        int res = -0x3f3f3f3f;
        for (int i = 0; i < l; i++) {
            while (s.size() > 0 && (i - s.rbegin()->second > k || s.rbegin()->first <= 0)) {
                s.erase(*s.rbegin());
            }
            if (s.size() > 0) {
                res = max(res, s.rbegin()->first + nums[i]);
                s.insert(make_pair(s.rbegin()->first + nums[i], i));
            } else {
                s.insert(make_pair(nums[i], i));
                res = max(res, nums[i]);
            }
        }
        return res;
    }
};
