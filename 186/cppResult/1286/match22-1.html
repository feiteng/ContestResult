class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n+1, -10000000);
        int ans = INT_MIN;
        dp[0] = 0;
        multiset<int> m;
        m.insert(0);
        for (int i = 1; i <= n; ++i) {
            dp[i] = nums[i-1];
            dp[i] = max(dp[i], *(--m.end()) + nums[i-1]);
            m.insert(dp[i]);
            if (i >= k) {
                auto p = m.lower_bound(dp[i-k]);
                m.erase(p);
            }
            ans = max(dp[i], ans);
        }
        return ans;
    }
};
