class Solution {
public:
    int constrainedSubsetSum(vector<int>& a, int k) {
        vector<int> dp = a;
        int n = a.size();
        priority_queue<pair<int, int>> pq;
        pq.emplace(dp[0], 0);
        for (int i = 1; i < n; ++i) {
            while (!pq.empty() && pq.top().second < i - k) {
                pq.pop();
            }
            if (!pq.empty())
                dp[i] = max(dp[i], a[i] + pq.top().first);
            pq.emplace(dp[i], i);
        }
        return *max_element(dp.begin(), dp.end());
    }
};
