class Solution {
public:
    int maxScore(string s) {
        int total_zero = 0, total_one = 0;
        for (int i = 0; i < s.length(); ++i)
        {
            if (s[i] == '0') total_zero += 1;
            else total_one += 1;
        }

        int ans = 0, left_zero = 0, right_one = total_one;
        for (int i = 0; i < s.length() - 1; ++i)
        {
            if (s[i] == '0') left_zero += 1;
            else right_one -= 1;
            ans = max(ans, left_zero + right_one);
        }
        return ans;
    }
};
