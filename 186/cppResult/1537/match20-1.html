class Solution {
public:
    int maxScore(string s) {
        int ans=0;
        int tot=0;
        int i,j,k,l,m,n;
        n=s.size();  int pre=0;
        for(i=0;i<n;i++)  if(s[i]=='1')  tot++;

        for(i=0;i<(n-1);i++)
        {
            if(s[i]=='1')  tot--;
            else pre++;
            ans=max(ans,pre+tot);
        }
        return ans;
    }
};
