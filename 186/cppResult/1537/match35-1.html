class Solution {
public:
    int maxScore(string s) {
        int ans = 0;
        int len = s.length();
        for (int i = 0; i < len - 1; i++) {
            int t1 = 0;
            for (int j = 0; j <= i; j++) {
                if (s[j] == '0')
                    t1++;
            }

            for (int j = i + 1; j < len; j++) {
                if (s[j] == '1')
                    t1++;
            }
            ans = max(ans, t1);
        }
        return ans;
    }
};
