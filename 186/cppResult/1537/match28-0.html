class Solution {
public:
    int maxScore(string s) {
        int n = s.size();
        vector<int> prefix(n+1, 0);
        for (int i = 0; i < n; i++) {
            if (s[i] == '0') prefix[i+1] = prefix[i]+1;
            else prefix[i+1] = prefix[i];
        }
        
        int ans = 0;
        for (int i = 1; i < n; i++) {
            int score = prefix[i] + n - i - prefix[n] + prefix[i];
            // cout<<"here "<<i<<" "<<score<<" "<<prefix[n]<<" "<<prefix[i]<<endl;
            ans = max(ans, score);
        }
        return ans;
    }
};
