class Solution {
public:
    int maxScore(string s) {
        int n=s.size();
        vector<int> left0(n,0);
        left0[0]=(s[0]=='0');
        for(int i=1;i<n;i++){
            left0[i]=left0[i-1]+(s[i]=='0');
        }
        int ans = 0;
        int right1 = s[n-1]=='1';
        for(int i=n-2;i>=0;i--){
            ans = max(ans,left0[i]+right1);
            right1 = right1+(s[i]=='1');
        }
        // ans = max(ans,right1);
        return ans;
        
    }
};
