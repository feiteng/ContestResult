class Solution {
public:
    int maxScore(string s) {
        int maximum = INT_MIN;
        for (int i = 1;i < s.length(); i++)      {
            int count = 0;
            for (int j = 0; j < i; j++)
                if (s[j] == '0')
                    count++;
            for (int j = i; j < s.length(); j++)
                if(s[j] == '1')
                    count++;
            maximum = max(count, maximum);
        }
        return maximum;
    }
};
