class Solution {
public:
    int maxScore(string s) {
        int n = s.length();
        int maxScore = 0;
        for (int i = 0; i < n - 1; i++) {
            int leftZero = 0;
            int rightOne = 0;
            for (int j = 0; j <= i; j++) {
                if (s[j] == '0') {
                    leftZero++;
                }
            }

            for (int j = i + 1; j < n; j++) {
                if (s[j] == '1') {
                    rightOne++;
                }
            }
            maxScore = max(maxScore, leftZero + rightOne);
        }
        return maxScore;
    }
};
