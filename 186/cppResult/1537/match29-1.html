class Solution {
public:
    int maxScore(string s) {
        vector<int> cntone(s.length(), 0);
        vector<int> cntzero(s.length(), 0);

        cntzero[0] = (s[0] == '0');
        for (int i = 1; i < s.length(); i++) {
            if (s[i] == '0') {
                cntzero[i] = cntzero[i - 1] + 1;
            }
            else cntzero[i] = cntzero[i - 1];
        }
        cntone[s.length() - 1] = (s[s.length() - 1] == '1');

        for (int i = s.length() - 2; i>= 0; i--) {
            if (s[i] == '1') {
                cntone[i] = cntone[i + 1] + 1;
            }
            else cntone[i] = cntone[i + 1];
        }
        int res = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            res = max(cntzero[i] + cntone[i + 1], res);
        }
        return res;
    }
};
