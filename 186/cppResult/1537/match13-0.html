class Solution {
public:
    int maxScore(string s) {
        int n = s.size();
        int best = 0;
        for (int i = 0; i + 1 < n; i++) {
            int sum = 0;
            for (int j = 0; j <= i; j++) {
                if (s[j] == '0') sum++;
            }
            for (int j = i + 1; j < n; j++) {
                if (s[j] == '1') sum++;
            }

            best = max(best, sum);
        }
        return best;
    }
};
