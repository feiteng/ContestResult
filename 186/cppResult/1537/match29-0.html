class Solution {
public:
    int maxScore(string s) {
        int n=s.size();
        int a0[n+1];
        int a1[n+1];
        a0[0]=0;
        a1[n]=0;
        for(int i=0;i<n-1;i++){
            if(s[i]=='0'){
                a0[i+1]=a0[i]+1;
            }
            else
                a0[i+1]=a0[i];
        }
        for(int i=n-1;i>0;i--){
            if(s[i]=='1'){
                a1[i]=a1[i+1]+1;
            }
            else
                a1[i]=a1[i+1];
        }
        int r=0;
        for(int i=1;i<n;i++){
            r=max(r,a0[i]+a1[i]);
        }
        return r;
    }
};
