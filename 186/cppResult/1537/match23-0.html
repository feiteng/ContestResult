class Solution {
public:
    int maxScore(string s) {
        int n = s.length();
        
        int ans = 0;
        for (int i = 0; i < n - 1; i++) {
            int score = 0;
            for (int j = 0; j < n; j++) {
                if ((j <= i) ^ (s[j] == '1')) {
                    score++;
                }
            }
            ans = max(ans, score);
        }
        return ans;
    }
};
