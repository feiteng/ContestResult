class Solution {
public:
    int maxScore(string s) {
        vector<int> left0(s.size(), (s[0] == '0')), right1(s.size(), (s.back() == '1'));
        for (int i = 1; i < s.size(); i++) {
            left0[i] = left0[i-1] + (s[i] == '0');
        }
        for (int i = s.size()-2; i >= 0; --i) {
            right1[i] = right1[i+1] + (s[i] == '1');
        }
        int maxV = 0;
        for (int i = 0; i < s.size()-1; i++) {
            maxV = max(maxV, left0[i] + right1[i+1]);
        }
        return maxV;
    }
};
