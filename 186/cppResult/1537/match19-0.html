class Solution {
public:
    int maxScore(string s) {
        int length = s.length();
        vector<int> left(length, 0);
        vector<int> right(length, 0);
        right[length - 1] = s[length - 1] - '0';
        for (int i = 1; i < length; i++) {
            left[i] = left[i - 1] + '1' - s[i - 1];
            right[length - 1 - i] = right[length - i] + s[length - 1 - i] - '0';
        }
        int ret = 0;
        for (int i = 1; i < length; i++) {
            ret = max(ret, left[i] + right[i]);
        }
        return ret;
    }
};
