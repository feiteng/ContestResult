class Solution {
public:
    int maxScore(string s) {
        int n = s.size();

        int ans = 0;
        for(int i=1;i<n;i++)
        {
            int m = 0;
            for(int j=0;j<i;j++)
                if(s[j]=='0')
                    m++;
            for(int j=i;j<n;j++)
                if(s[j]=='1')
                    m++;
            ans = max(m,ans);
        }
        return ans;
    }
};
