class Solution {
public:
    int maxScore(string s) {
        int n = s.length();
        
        int maxscore=0;
        for (int i = 1; i<n; ++i) {
            int leftscore=0, rightscore=0;

            
            for (int j = i; j<n; ++j) {
                if (s[j]=='1') ++rightscore;
            }
            for (int k=0; k<i; ++k) {
                if (s[k]=='0') ++leftscore;
            }
            maxscore = max(maxscore,leftscore+rightscore);
        }
        return maxscore;
    }
};
