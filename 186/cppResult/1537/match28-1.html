class Solution {
public:
    int maxScore(string s) {
        int n = s.length();
        vector<int> ones(n + 1, 0);
        for (int i = 0; i < n; ++i) {

            ones[i + 1] = ones[i] + s[i] - '0';
        }
        
        int res = 0;
        for (int i = 0; i < n - 1; ++i) {
            int ans = i + 1 - ones[i + 1];
            ans += ones[n] - ones[i + 1];
            res = max(res, ans);
        }
        
        return res;
    }
};
