class Solution {
public:
    int maxScore(string s) {
        int len = s.size();
        int best = 0;
        for (int i = 1; i < len; i++) {
            int tmp = 0;

            for (int a = 0; a < i; a++) {
                if (s[a] == '0') tmp++;
            }
            for (int a = i; a < len; a++) {
                if (s[a] == '1') tmp++;
            }
            best = max(best, tmp);
        }
        return best;
    }
};
