class Solution {
public:
    int maxScore(string s) {
        int len = s.size();
        vector<int> left(len + 1, 0), right(len + 1, 0);
        for (int i = 1; i <= s.size(); i++) {
            left[i] = left[i - 1] + (s[i - 1] == '0');
        }
        for (int i = s.size() - 1; i >= 0; i--) {
            right[i] = right[i + 1] + (s[i] == '1');
        }
        int ans = 0;
        for (int i = 1; i < s.size(); i++) {
            ans = max(ans, left[i] + right[i]);
        }
        return ans;
    }
};
