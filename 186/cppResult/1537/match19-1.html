class Solution {
public:
    int maxScore(string s) {
        int l = s.size();
        vector<int> v;
        int c = 0;
        for (int i = 0; i < l; i++) {
            if (s[i] == '0') {
                c++;
            }
            v.push_back(c);

        }
        
        int res = 0;
        for (int i = 0; i < l - 1; i++) {
            res = max(res, v[i] + l - v.back() - (i + 1 - v[i]));
        }
        return res;
    }
};
