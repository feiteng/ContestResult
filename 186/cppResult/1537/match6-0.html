class Solution {
public:
    int maxScore(string s) {
        int n = s.size();
        vector<int> zeros(n + 1, 0), ones(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            zeros[i] = zeros[i - 1] + (s[i - 1] == '0');
        }
        for (int i = n - 1; i >= 0; i--) {
            ones[i] = ones[i + 1] + (s[i] == '1');
        }
        int res = 0;
        for (int i = 1; i <= n - 1; i++) {
            res = max(res, zeros[i] + ones[i]);
        }
        return res;
    }
};
