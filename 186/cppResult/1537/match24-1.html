const ll MOD=1e9+7;

class Solution {
public:
  int maxScore(string s) {
    int N=s.size();
    int res=0;
    for(int i=1;i<N;i++){

      int k=0;
      for(int j=0;j<N;j++){
        if (j<i&&s[j]=='0' || j>=i&&s[j]=='1') k++;
      }
      res=max(res,k);
    }
    return res;
  }
};
