class Solution {
public:
	int maxScore(string s) {

		int l = 0, r = 0;
		if (s[0] == '0')l = 1;
		for (int i = 1; i < s.size(); i++) {
			if (s[i] == '1')r++;
		}
		int maxx = l + r;
		for (int i = 1; i < s.size()-1; i++) {
			if (s[i] == '0')l++;
			else r--;
			maxx = max(maxx, l + r);
		}
		return maxx;
	}
};
