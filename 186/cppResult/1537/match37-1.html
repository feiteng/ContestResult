class Solution {
public:
    int maxScore(string s) {
        int n = s.size();
        vector<int> numZeroes(n), numOnes(n);
        int zeroCnt = 0;
        for(int i = 0; i < n; i++) {
            zeroCnt += (s[i] == '0');
            numZeroes[i] = zeroCnt;
        }
        int oneCnt = 0;
        for(int i = n - 1; i >= 0; i--) {
            oneCnt += (s[i] == '1');
            numOnes[i] = oneCnt;

        }
        int ans = 0;
        for(int split = 0; split + 1 < n; split++) {
            ans = max(ans, numZeroes[split] + numOnes[split + 1]);
        }
        return ans;
    }
};
