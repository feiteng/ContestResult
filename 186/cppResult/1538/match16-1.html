class Solution {
public:
    int maxScore(vector<int>& ar, int k) {
        int n = ar.size();
        int score = 0;
        for (int i = 0; i < k; i++) score += ar[i];
        int res = score;

        int i = k-1, j = n-1;
        while (i >= 0) {
            score -= ar[i];
            score += ar[j];
            res = max(res, score);
            i--;
            j--;
        }
        return res;
    }
};
