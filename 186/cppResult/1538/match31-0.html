class Solution {
public:
    int maxScore(vector<int>& a, int k) {
        int n = a.size();
        int m = n - k;
        int sum = accumulate(a.begin(), a.begin() + m, 0);
        int ans = sum;
        for (int i = 1; i + m <= n; ++i) {
            sum += a[i + m - 1] - a[i - 1];
            ans = min(ans, sum);
        }
        return accumulate(a.begin(), a.end(), 0) - ans;
    }
};
