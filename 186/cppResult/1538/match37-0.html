class Solution {
public:
    int maxScore(vector<int>& a, int k) {
        vector<int> p = {0}, s;
        partial_sum(a.begin(), a.end(), back_inserter(p));
        // for (auto pi : p) {
        //     cout << pi << " ";
        // }
        // cout << "\n";
        partial_sum(a.rbegin(), a.rend(), back_inserter(s));
        reverse(s.begin(), s.end());
        s.push_back(0);
        // for (auto si : s) {
        //     cout << si << " ";
        // }
        // cout << "\n";
        int n = a.size();
        int ans = 0;
        for (int l = 0; l <= k; ++l) {
            // cout << l << " " << n + 1 - k + l << "\n";
            ans = max(ans, p[l] + s[n - k + l]);
        }
        // cout << "\n";
        return ans;
    }
};
