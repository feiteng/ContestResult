class Solution {
public:
    int maxScore(vector<int>& card, int k) {
        int n=card.size();
        vector<int> sum(n+1,0);
        for(int i=1;i<=n;i++){
            sum[i]=sum[i-1]+card[i-1];
        }
        int res=0;
        int len=n-k;
        for(int left=1;left+len-1<=n;left++){
            res=max(res,sum[n]-(sum[left+len-1]-sum[left-1]));
        }
        return res;
    }
};
