class Solution {
public:
    int maxScore(vector<int>& v, int k) {
        int n = v.size();
        vector<int> sumL(n + 1),sumR(n + 1);
        for(int i =1;i<=n;i++)
            sumL[i] = sumL[i - 1] + v[i - 1];
        reverse(v.begin(),v.end());
         for(int i =1;i<=n;i++)
            sumR[i] = sumR[i - 1] + v[i - 1];
        reverse(v.begin(),v.end());
        int ans = 0;
        for(int i =0;i<=k;i++)
            ans = max(ans,sumL[i] +sumR[k - i]);
        return ans;
    }
};
