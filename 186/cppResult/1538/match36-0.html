class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n=cardPoints.size();
        int sumleft[k+1];
        int sumright[k+1];
        sumleft[0]=0;
        sumright[0]=0;
        for(int i=0;i<k;i++){
            sumleft[i+1]=sumleft[i]+cardPoints[i];
            sumright[i+1]=sumright[i]+cardPoints[n-1-i];
        }
        int r=0;
        for(int i=0;i<=k;i++){
            r=max(r,sumleft[i]+sumright[k-i]);
        }
        return r;
    }
};
