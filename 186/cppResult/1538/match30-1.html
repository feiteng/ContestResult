class Solution {
public:
	int maxScore(vector<int>& cardPoints, int k) {
		int n = cardPoints.size();
		vector<int>sum(n + 1, 0);
		for (int i = 1; i <= n; i++) {
			sum[i] = sum[i - 1] + cardPoints[i-1];

		}
		int maxx = 0;
		for (int i = 0; i <= k; i++) {
			maxx = max(maxx, sum[i] + sum[n] - sum[n - k + i]);
		}
		return maxx;
	}
};
