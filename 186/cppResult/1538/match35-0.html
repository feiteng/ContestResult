class Solution {
public:
    int pre[110000];
    int lst[110000];
    int maxScore(vector<int> &cardPoints, int k) {
        int len = cardPoints.size();
        pre[0] = 0;
        lst[0] = 0;
        for (int i = 0; i < len; i++) {
            pre[i + 1] = pre[i] + cardPoints[i];
            lst[i + 1] = lst[i] + cardPoints[len - i - 1];
        }
        int ans = 0;
        for (int i = 0; i <= k; i++) {
            ans = max(ans, pre[i] + lst[k - i]);
        }
        return ans;
    }
};
