class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int ans = 0;
        int sum = 0;
        for(int i = 0; i < k; i++) sum += cardPoints[i];
        int lo = k - 1, hi = cardPoints.size() - 1;
        ans = sum;
        while(lo >= 0) {
            sum += cardPoints[hi];
            sum -= cardPoints[lo];
            ans = max(ans, sum);
            hi--;
            lo--;
        }
        return ans;
    }
};
