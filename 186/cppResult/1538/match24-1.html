class Solution {
public:
    int maxScore(vector<int>& A, int k) {
        //take 0..k from left, k..0 from right
        int n = A.size();
        vector<int> prefix(n+1), suffix(n+1); //suffix is inverse
        for(int i=0; i<n; i++)
            prefix[i+1] = prefix[i]+A[i];

        for(int i=0; i<n; i++)
            suffix[i+1] = suffix[i]+A[n-i-1];
        int ans = 0;
        for(int i=0; i<=k; i++){
            //take i from left, k-i from right
            ans = max(ans, prefix[i] + suffix[k-i]);
        }
        return ans;
    }
};
