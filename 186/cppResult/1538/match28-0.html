class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        vector<int> pf(n + 1);
        for (int i = 0; i < n; ++i) {
            pf[i + 1] = pf[i] + cardPoints[i];
        }
        int answer = INT_MIN;
        for (int ll = 0; ll <= k; ++ll) {
            int rr = k - ll;
            answer = max(answer, pf[ll] + pf[n] - pf[n - rr]);
        }
        return answer;
    }
};
