class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int len = cardPoints.size();
        int sum = accumulate(cardPoints.begin(), cardPoints.end(), 0);
        if (k == len) return sum;

        int cur = accumulate(cardPoints.begin(), cardPoints.begin() + len - k, 0);
        int ans = cur;
        for (int i = len - k; i < len; i++) {
            cur += cardPoints[i] - cardPoints[i - (len - k)];
            ans = min(ans, cur);
        }
        return sum - ans;
    }
};
