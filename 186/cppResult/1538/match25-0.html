class Solution {
public:
    int maxScore(vector<int>& c, int k) {
        int n = c.size();
        vector<int>sum(n+2,0),suf(n+2,0);
        for(int i = 0;i < c.size();++i){
            sum[i+1]=sum[i]+c[i];
        }
        for(int i = n;i >= 1;--i)suf[i]=suf[i+1]+c[i-1];
        int ans = 0;
        for(int i = 0;i <= k;++i){
            ans=max(ans,sum[i]+suf[n-k+i+1]);
        }
        return ans;
    }
};
