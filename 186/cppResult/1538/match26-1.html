class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) 
    {
        int sum = 0;        
        // initial
        for (int i=0; i<k; ++i)
            sum += cardPoints[i];
        

        int N = cardPoints.size();
        int res = sum;
        // greedy
        for (int i=0; i<k; ++i)
        {
            sum -= cardPoints[k-i-1];
            sum += cardPoints[N-i-1];
            
            
            // cout << cardPoints[k-i-1] << ' ' << cardPoints[N-i-1] << ' ' << sum << endl;
            res = max(res, sum);
        }
        return res;
    }
};
