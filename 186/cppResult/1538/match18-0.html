class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
		int n = cardPoints.size();
        vector<int> lsum(n+1, 0), rsum(n+1, 0);
		for ( int i = 0; i < n; ++i ) {
			lsum[i+1] = lsum[i]+cardPoints[i];
			rsum[i+1] = rsum[i]+cardPoints[n-1-i];
		}
		int res = 0;
		for ( int i = 0; i <= k; ++i ) {
            //cout << i << " " << lsum[i] << " " << rsum[k-i] << endl;
			res = max(res, lsum[i]+rsum[k-i]);
		}
		return res;
    }
};
