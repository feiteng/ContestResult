class Solution {
public:
    int maxScore(vector<int>& v, int k) {
        int n=v.size();
        vector<int> pre(n+1,0), suf(n+1,0);
        for (int i=1; i<=n; i++)
            pre[i]=pre[i-1]+v[i-1];
        for (int i=1; i<=n; i++)
            suf[i]=suf[i-1]+v[n-i];
        int ret=0;
        for (int i=0; i<=k; i++)
            ret=max(ret,pre[i]+suf[k-i]);
        return ret;
    }
};
