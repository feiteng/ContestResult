class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        vector<int> left(k), right(k);
        for (int i = 0; i < k; i++) {
            left[i] = i == 0 ? cardPoints[i] : left[i - 1] + cardPoints[i];
            // cout<<left[i]<<" ";
        }
        cout<<endl;
        for (int i = k - 1; i >= 0; i--) {
            int ni = i + (n - k);
            right[i] = i == k - 1 ? cardPoints[ni] : right[i + 1] + cardPoints[ni];
            // cout<<right[i]<<" ";
        }
        cout<<endl;
        
        int MAX = max(right[0], left[k - 1]);
        for (int i = 0; i < k - 1; i++) {
            MAX = max(MAX, left[i] + right[i + 1]);
        }
        
        return MAX;
    }
};
