class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size(), sum = 0;
        vector<int> front(n+1, 0), back(n+1, 0);
        for(int i = 0; i < n+1; i++){
            front[i] = sum;
            if(i < n) sum += cardPoints[i];
        }
        sum = 0;
        for(int i = 0; i < n+1; i++){
            back[i] = sum;
            if(i < n) sum += cardPoints[n-1-i];
        }
        int maxi = 0;
        for(int i = 0; i <= k; i++){
            maxi = max(maxi, front[i]+back[k-i]);
        }
        return maxi;
    }
};
