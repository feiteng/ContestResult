class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int sum = 0;
        for(int i = 0; i < k; i++){
            sum += cardPoints[i];
        }
        int res = sum;
        int n = cardPoints.size();
        for(int i = 1; i <= k; i ++){
            sum += cardPoints[n-i] - cardPoints[k-i];
            res = max(res, sum);
        }
        return res;
    }
};
