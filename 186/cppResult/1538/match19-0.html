class Solution {
public:
    int maxScore(vector<int>& card, int k) {
        int n = card.size();
        vector<int> sum(n + 1);
        for (int i=1; i<=n; ++i)
            sum[i] = sum[i-1] + card[i-1];
        
        int res = 0;
        for (int i=0; i<=k; ++i)
        {
            res = max(res, sum[i] + sum[n] - sum[n-k+i]);
        }
        return res;
    }
};
