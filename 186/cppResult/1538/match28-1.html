class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        vector<int> s(n + 1, 0);  
        for(int i = 1; i <= n; ++i)
            s[i] = s[i - 1] + cardPoints[i - 1];

        int ans = 0;
        for(int i = 0; i <= k; ++i) {
            int tmp = s[i] - s[0] + s[n] - s[n - k + i];
            //cout << tmp << endl;
            ans = max(ans, tmp);
        }
        return ans;
    }
};
