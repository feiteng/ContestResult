class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int left = 0;
        int right = 0;
        int len = cardPoints.size();
        for (int i=len - k; i <len; i++) {
            right += cardPoints[i];
        }

        int ans = left + right;
        // cout << "left = " << left << ", right = " << right << ", ans = " << ans << endl;
        for (int i=0; i<k; i++) {
            left += cardPoints[i];
            right -= cardPoints[len-k+i];
            ans = max(ans, left+right);
            // cout << "left = " << left << ", right = " << right << ", ans = " << ans << endl;
        }
        return ans;
    }
};
