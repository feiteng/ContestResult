class Solution {
public:
    int pre[100010];
    int suf[100010];
    int maxScore(vector<int>& c, int k) {
        int n = c.size();
        pre[0] = 0;
        suf[n+1] = 0;
        for(int i=1;i<=n;i++){
            pre[i] = pre[i-1] + c[i-1];
        }
        for(int i=n;i>=1;i--){

            suf[i] = suf[i+1] + c[i-1];
        }
        int ans= 0;
        for(int i=0;i<=k;i++){
            ans = max(ans,pre[i]+suf[n-k+i+1]);
        }
        return ans;
    }
};
