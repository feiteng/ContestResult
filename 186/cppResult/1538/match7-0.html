class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int sum = 0;
        for (int i=0; i<k; ++i){
            sum += cardPoints[i];
        }
        int res = sum, n = cardPoints.size();
        for (int i=0; i<k; ++i){
            sum -= cardPoints[k-1-i] - cardPoints[n-1-i];
            res = max(res, sum);
        }
        return res;
    }
};
