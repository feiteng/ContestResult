class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n=cardPoints.size();
        int len=n-k;
        int sum=0;
        for(auto x:cardPoints)
            sum+=x;
        if(len==0)
            return sum;
        int tmp=sum;
        sum=0;
        for(int i=0;i<len;i++){
            sum+=cardPoints[i];
        }

        int res=sum;
        for(int i=len;i<n;i++){
            sum+=cardPoints[i]-cardPoints[i-len];
            res=min(res,sum);
        }
        return tmp-res;
    }
};
