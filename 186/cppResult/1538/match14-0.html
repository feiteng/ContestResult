class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int sz = cardPoints.size();
        int subSize = sz - k;
        int subSum = 0;
        for (int i = 0; i < subSize; i++)
            subSum += cardPoints[i];
        int minSub = subSum;
        int sum = subSum;
        for (int i = subSize; i < sz; i++) {
            subSum -= cardPoints[i - subSize];
            subSum += cardPoints[i];
            minSub = min(minSub, subSum);
            sum += cardPoints[i];
        }
        return sum - minSub;
    }
};
