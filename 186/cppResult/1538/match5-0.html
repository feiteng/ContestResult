class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int N = cardPoints.size();
        vector<int> sumL(N + 1), sumR(N + 1);
        sumL[0] = 0;
        sumR[0] = 0;
        for (int i = 1; i <= N; ++i) {
            sumL[i] = sumL[i - 1] + cardPoints[i - 1];
            sumR[i] = sumR[i - 1] + cardPoints[N - i];
        }
        int ans = 0;
        for (int i = 0; i <= k; ++i) {
            ans = max(ans, sumL[i] + sumR[k - i]);
        }
        return ans;
    }
};
