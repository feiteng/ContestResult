class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n=cardPoints.size();
        int tot=0;
        for(int i=0;i<n;i++)
        {
            tot=tot+cardPoints[i];
        }
        int now=0;
        for(int i=0;i<n-k;i++)
        {
            now=now+cardPoints[i];
        }
        int minm=now;
        for(int i=n-k;i<n;i++)
        {
            now=now+cardPoints[i]-cardPoints[i-(n-k)];
            minm=min(minm,now);
        }
        return tot-minm;
    }
};
