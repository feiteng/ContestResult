class Solution {
public:
    int maxScore(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int> A(n);
        vector<int> B(n);
        A[0] = arr[0];
        B[n - 1] = arr[n - 1];
        for (int i = 1; i < n; i++) {
            A[i] = A[i - 1] + arr[i];
        }
        for (int i = n - 2; i >= 0; i--) {

            B[i] = B[i + 1] + arr[i];
        }
        int ans = max(A[k - 1], B[n - k]);
        for (int i = 1; i < k; i++) {
            ans = max(ans, A[i - 1] + B[n - (k - i)]);
        }
        return ans;
    }
};
