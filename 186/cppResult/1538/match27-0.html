class Solution {
public:
    int maxScore(vector<int>& a, int k) {
        int n=a.size(),s=0;
        for(int i=0;i<k;i++)s+=a[i];
        int ans=s;
        for(int i=k-1;i>=0;i--){
            s-=a[i];
            s+=a[n-(k-i)];
            ans=max(s,ans);
        }
        return ans;
    }
};
