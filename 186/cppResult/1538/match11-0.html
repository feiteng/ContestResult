class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n=cardPoints.size();
        vector<int> preSum(n+1,0), sufSum(n+1,0);
        for (int i=0; i<n; i++) preSum[i+1]=preSum[i]+cardPoints[i];
        for (int i=0; i<n; i++) sufSum[i+1]=sufSum[i]+cardPoints[n-1-i];
        // for (int i=0; i<n; i++) cout << sufSum[i] << endl;
        int ans=0;
        
        for (int i=0; i<=k; i++) {
            ans = max(ans, preSum[i] + sufSum[k-i]);
        }
        return ans;
    }
};
