class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        vector<pair<int, pair<int, int> > > v;
        
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 0; j < nums[i].size(); j++) {
                v.push_back({i + j, {j, nums[i][j]}});
            }
        }
        
        vector<int> ans;
        sort(v.begin(), v.end());
        
        for (auto i : v) {
            ans.push_back(i.second.second);
        }
        
        return ans;
    }
};
