class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        map<int,vector<int>> mp;
        for(int i=0;i<nums.size();i++)
        {
            for(int j=0;j<nums[i].size();j++)
            {
                mp[i+j].push_back(nums[i][j]);
            }
        }
        for(auto &x: mp)
        {
            reverse(x.second.begin(),x.second.end());

        }
        vector<int> ans;
        for(auto x:mp)
        {
            for(auto y:x.second)
            {
                ans.push_back(y);
            }
        }
        return ans;
    }
    
};
