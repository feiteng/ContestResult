class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        vector<vector<int>>sm;
        for(int i=0;i<nums.size();++i)
        {
            for(int j=0;j<nums[i].size();++j)
            {
                if(sm.size()<=i+j)
                {
                    vector<int>v;
                    sm.push_back(v);
                }
                sm[i+j].push_back(nums[i][j]);

            }
        }
        vector<int>ans;
        for(int i=0;i<sm.size();++i)
        {
            for(int j=sm[i].size()-1;j>=0;--j)
            {
                ans.push_back(sm[i][j]);
            }
        }
        return ans;
    }
};
