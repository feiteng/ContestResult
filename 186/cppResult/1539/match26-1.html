class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        vector<vector<int>> p(200000, vector<int>());
        for (int i = 0; i < nums.size(); ++i) {
            for (int j = 0; j < nums[i].size(); ++j) {
                p[i + j].push_back(nums[i][j]);
            }
        }
        
        vector<int> ret;
        for (int i = 0; i < p.size(); ++i) {
            if (!p[i].empty()) ret.insert(ret.end(), p[i].rbegin(), p[i].rend());
        }
        
        return ret;
    }
};
