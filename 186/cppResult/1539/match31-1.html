class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        map<int, vector<int>> m;
        for (int i=0; i<nums.size(); ++i)
        {
            for (int j=0; j<nums[i].size(); ++j)
            {
                m[i + j].push_back(nums[i][j]);
            }
        }
        vector<int> res;
        for (auto & it: m)
            res.insert(end(res), rbegin(it.second), rend(it.second));
        return res;
    }
};
