class Solution {
	private static class Element {
		public int value;
		public int index;
		public Element(int value , int index) {
			this.value = value;
			this.index = index;
		}
	}
	
    public int constrainedSubsetSum(int[] nums, int k) {
    	Deque<Element> queue = new LinkedList<>();
    	int sum = 0 , ans = Integer.MIN_VALUE , n = nums.length;    	
    	for (int i = 0;i < k;i ++) {
    		int result = 0;    		
    		if (nums[i] >= 0) {
    			sum += nums[i];
    		}
    		if (nums[i] < 0) {
    			result = nums[i] + sum;    			
    		} else {
    			result = sum;
    		}
    		while (!queue.isEmpty() && queue.peekLast().value <= result) {
    			queue.pollLast();    		
    		}
    		queue.addLast(new Element(result , i));
    		ans = Math.max(ans , result);
    	}
    	for (int i = k;i < n;i ++) {
    		int result = Math.max(queue.peekFirst().value , 0) + nums[i];
    		ans = Math
.max(ans , result);
    		while (!queue.isEmpty() && queue.peekLast().value <= result) {
    			queue.pollLast();    			
    		}
    		queue.addLast(new Element(result , i));
    		if (queue.peek
First().index == i - k) {
    			queue.pollFirst();
    		}
    	}
    	return ans;
    }
}
