class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        if (nums.length < k) {
            return 0;
        }
        
        int n = nums.length;
        
TreeSet<Integer> set = new TreeSet<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer a, Integer b) {
                return b - a;
            }
        });
        set.add(0);
        int[] constrainedSums = new int[n];
        int result = nums[0];
        
        for (int i = 0; i < n; i ++) {
            if (i > k) {
                set.remove(constrainedSums[i - k - 1]);
            }
            
            constrainedSums[i] = nums[i] + set.first().intValue();
            set.add(constrainedSums[i]);
            result = Math.max(result, constrainedSums[i]);
        }
        
        return result;
    }
}
