class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] cache = new int[n];
        int max = nums[0];
        
        LinkedList<Integer> deque = new LinkedList();
        
        for (int i = 0; i < n; i++) {
            while (!deque.isEmpty() && deque.peekFirst() < i-k) {
                deque.pollFirst();
            }
            // System.out.println(nums[i] + " " + deque);
            
            cache[i] = nums[i];
            if (!deque.isEmpty() && cache[deque.peekFirst()] > 0) cache[i] += cache[deque.peekFirst()];
            
            while (!deque.isEmpty() && cache[deque.peekLast()] <= cache[i] ) {
                deque.pollLast();
            }
            
            deque.add(i);
            // System.out.println(nums[i] + " " + deque);
            max = Math.max(max, cache[i]);
        }
        // System.out.println(Arrays.toString(cache));
        return max;
    }
}
