class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] d = new int[n];
        
        Deque<int[]> deque = new LinkedList();
        
        for (int i = 0; i< n; i++) {
            d[i] = nums[i];
            
            if (!deque.isEmpty() && i - deque.peekFirst()[0] > k) {
                deque.pollFirst();
            }
            if (!deque.isEmpty()) {
                d[i] = Math.max(deque.peekFirst()[1] + nums[i], d[i]);
            }
            
            while (!deque.isEmpty() && deque.peekLast()[1] < d[i]) {
                deque.pollLast();
            }
            deque.addLast(new int[]{i, d[i]});
        }
        int res = Integer.MIN_VALUE;
        for (int i = 0; i< n; i++) {
            res = Math.max(res, d[i]);
        }
        return res;
    }
}
