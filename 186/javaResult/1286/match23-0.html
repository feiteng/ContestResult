class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n+1];
        Arrays.fill(dp,(int)-1e9-1);
        int max = Integer.MIN_VALUE;
        PriorityQueue<Integer> queue = new PriorityQueue<>((a1,a2)->dp[a2]-dp[a1]);
        queue.add(0);
        for(int i=1;i<=n;i++){
            dp[i]=nums[i-1];
            int add = dp[queue.peek()];
            if(add>0){
                dp[i]=nums[i-1]+add;
            }
            while(!queue.isEmpty()&&(i-queue.peek()>k-1||dp[queue.peek()]<=dp[i])){
                queue.poll();
            }
            queue.add(i);
            max = Math.max(max,dp[i]);
        }
        return max;
    }
}
