class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] dp = new int[nums.length];
        
        dp[0] = nums[0];
        int max = dp[0];
        for(int i=1 ; i<nums.length ; i++){
            dp[i] = nums[i];
            for(int j =i-1 ; j>=Math.max(0,i-k) ; j--){
                dp[i] = Math.max(dp[i],dp[j]+nums[i]);
                if(nums[j]>0){
                    break;
                }
            }
            max = Math.max(max,dp[i]);
        }
        return max;
    }
}
