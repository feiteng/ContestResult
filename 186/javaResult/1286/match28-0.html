class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] dp = new int[nums.length + 1];
        ArrayDeque<Integer> dq = new ArrayDeque<Integer>();
        dq.offerLast(0);
        int max = -10000;
        for (int i = 0; i < nums.length; i++) {
            int idx = dq.getFirst();
            dp[i + 1] = Math.max(dp[i + 1], dp[idx]);
            dp[i + 1] += nums[i];
            max = Math.max(max, dp[i + 1]);
            
            if (!dq.isEmpty() && (i + 1) - dq.getFirst() >= k) {
                dq.pollFirst();
            }
            while (!dq.isEmpty() && dp[dq.getLast()] <= dp[i + 1]) {
                dq.pollLast();
            }
            dq.offerLast(i + 1);
        }
        return max;
    }
}
