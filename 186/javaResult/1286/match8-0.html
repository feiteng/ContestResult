import java.util.LinkedList;

class Solution {
    public int constrainedSubsetSum(int[] arr, int k) {


        int ans = arr[0];

        LinkedList<int[]> q = new LinkedList<>();
        q.addLast(new int[]{0, arr[0]});


        for (int i = 1; i < arr.length; i++) {

            while (!q.isEmpty() && q.peekFirst()[0] < i - k) {
                q.pollFirst();
            }


            if (q.isEmpty()) {
                int sum = arr[i];
                ans = Math.max(ans, sum);
                q.addLast(new int[]{i, sum});
            } else {

                int sum = Math.max(arr[i], arr[i] + q.peekFirst()[1]);
                ans = Math.max(ans, sum);
                while (!q.isEmpty() && q.peekLast()[1] <= sum) {
                    q.pollLast();
                }
                q.addLast(new int[]{i, sum});
            }
        }
        return ans;

    }

    public static void main(String[] a) {
        Solution sol = new Solution();
        int[] arr = {10, 2, -10, 5, 20};
        System.out.println(sol.constrainedSubsetSum(arr, 2));
    }
}
