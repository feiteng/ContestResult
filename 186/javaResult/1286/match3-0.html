class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] dp = new int[nums.length];

        dp[0] = nums[0];
        int result = nums[0];
        TreeSet<int[]> tree = new TreeSet<>((a, b) -> (a[1] - b[1]));
        tree.add(new int[] { 0, dp[0] });
        for (int j = 1; j < nums.length; j++) {
            if (j - k > 0) {
                tree.remove(new int[] { j - k - 1, dp[j - k - 1] });
            }
            int max = tree.size() > 0 ? Math.max(0, tree.last()[1]) : 0;
            dp[j] = nums[j] + max;
            tree.add(new int[] { j, dp[j] });
            result = Math.max(result, dp[j]);
        }

        return result;
    }
}
