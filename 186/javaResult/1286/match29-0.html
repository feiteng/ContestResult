class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n + 1];
        int ans = Integer.MIN_VALUE;
        LinkedList<Integer> deque = new LinkedList<>();
        deque.add(0);
        for (int i = 1; i <= n; ++i) {
            while (!deque.isEmpty() && deque.getFirst() < i - k) deque.removeFirst();
            dp[i] = nums[i - 1] + Math.max(0, dp[deque.getFirst()]);
            ans = Math.max(ans, dp[i]);
            while (!deque.isEmpty() && dp[deque.getLast()] <= dp[i]) deque.removeLast();
            deque.add(i);
        }
        return ans;
    }
}
