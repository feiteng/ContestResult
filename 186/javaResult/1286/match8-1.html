class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        
        Deque<int[]> deque = new LinkedList<>();
        int sum = Integer.MIN_VALUE;
        
        for(int i = 0; i < n; i++)
        {
            int cur = nums[i];
            
while(!deque.isEmpty() && deque.peekFirst()[0] + k < i) 
            {
            	deque.pollFirst();
            }
            if(!deque.isEmpty())
            {
            	cur += Math.max(0, deque.peekFirst()[1]);
            }
            
sum = Math.max(sum, cur);
            while(!deque.isEmpty() && deque.peekLast()[1] < cur)
            {
            	deque.pollLast();
            }
            deque.offer(new int[]{i, cur});
        }
        
        return sum;
    }   
}
