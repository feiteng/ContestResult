class Pair{
    final int index;
    final int value;
    
    public Pair(int index, int value){
        this.index = index;
        this.value = value;
    }
}

class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        ArrayDeque<Pair> queue = new ArrayDeque<>();
        int result= Integer.MIN_VALUE;
        for(int i =0;i<nums.length;i++){
            result = Math.max(result, nums[i]);
            // consider the case where we take nums[i];
            while(!queue.isEmpty() && i - queue.peekFirst().index >= k+1){
                queue.pollFirst();
            }
            int sum = Math.max((queue.isEmpty()?0:queue.peekFirst().value), 0) + nums[i];
            while(!queue.isEmpty() && queue.peekLast().value < sum){
                queue.pollLast();
            }
            queue.add(new Pair(i, sum));
            result = Math.max(result, sum);
        }
        return result;
    }
}
