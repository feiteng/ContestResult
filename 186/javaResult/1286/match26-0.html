class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        TreeMap<Integer, Integer> tm = new TreeMap<>();
        tm.put(nums[0], 1);
        int count = 1;
        for (int i = 1; i < n; i++) {
            int max = tm.lastKey();
            if (max < 0) dp[i] = nums[i];
            else dp[i] = max + nums[i];
            if (count < k) {
                if (!tm.containsKey(dp[i])) tm.put(dp[i], 0);
                tm.put(dp[i], tm.get(dp[i]) + 1);
                count++;
            } else {
                int removeKey = dp[i - k];
                int times = tm.get(removeKey);
                if (times <= 1) tm.remove(removeKey);
                else tm.put(removeKey, times - 1);
                if (!tm.containsKey(dp[i])) tm.put(dp[i], 0);
                tm.put(dp[i], tm.get(dp[i]) + 1);
            }
        }
        int ans = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            ans = Math.max(dp[i], ans);
        }
        return ans;
    }
}
