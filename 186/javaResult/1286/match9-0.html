class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> {
            if (a[0] == b[0]) {
                return a[1] - b[1];
            }
            return b[0] - a[0];
        });
        pq.add(new int[]{nums[0], 0});
        int best = nums[0];
        for (int i = 1; i < n; ++i) {
            int val = nums[i];
            int minIdx = i - k;
            while (pq.size() > 0 && pq.peek()[1] < minIdx) {
                pq.poll();
            }
            if (!pq.isEmpty() && pq.peek()[0] > 0) {
                val += pq.peek()[0];
                // System.out.println("i: " + i + " pq.peek()[0]: " + pq.peek()[0] + " val: " + val);
            }
            pq.add(new int[]{val, i});
            best = Math.max(best, val);
        }
        return best;
    }
}
