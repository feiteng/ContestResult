class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length, maxValue = Integer.MIN_VALUE;
        for (int num : nums) {
            maxValue = Math.max(maxValue, num);
        }
        if (maxValue < 0) {
            return maxValue;
        }
        int[] maxSum = new int[n + 1];
        int res = 0;
        Deque<Integer> deque = new ArrayDeque<>();
        deque.add(0);
        for (int i = n - 1; i >= 0; i--) {
            maxSum[i] = deque.peekFirst() + nums[i];
            res = Math.max(res, maxSum[i]);
            if (i + k < n && 
deque.peekFirst() == maxSum[i + k]) {
                deque.removeFirst();
            }
            while (!deque.isEmpty() && deque.peekLast() < maxSum[i]) {
                deque.removeLast();
            }
            deque.add(maxSum[i]);
        }
        return res;
    }
}
