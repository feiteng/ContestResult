class Solution {
   public int constrainedSubsetSum(int[] nums, int k) {
       
       int len = nums.length;
       
       int[] dp = new int[len];
       
       int res = Integer.MIN_VALUE;
       int max = Integer.MIN_VALUE;
       
       Deque<int[]> dq = new ArrayDeque<>();
       
       for (int i = 0; i < len; i++){
           dp[i] = nums[i];
           
           while (!dq.isEmpty() && i - dq.peekFirst()[1] > k)
               dq.pollFirst();
           
           if (!dq.isEmpty())
            dp[i] = Math.max(dp[i], dq.peek()[0] + nums[i]);
           
           while (!dq.isEmpty() && dq.peekLast()[0] <= dp[i]) 
               dq.pollLast();
           
           
dq.offer(new int[]{dp[i], i});
           
           res = Math.max(res, dp[i]);
       }
       
       return res;
       
   }
}
