class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n+1];
        dp[0] = Integer.MIN_VALUE/2;
        int max = Integer.MIN_VALUE/2;
        Deque<int[]> queue = new LinkedList<>();  // int[0] pos, int[1] max
        for (int i = 1; i <= n; i++) {
            int num = nums[i-1];
            
            // get max from mono queue
            while (!queue.isEmpty() && queue.peekFirst()[0] < i - k) {
                queue.pollFirst();
            }
            int win_max = queue.isEmpty() ? Integer.MIN_VALUE/2 : queue.peekFirst()[1];
            
            dp[i] = Math.max(num, win_max + num);
            
            //System.out.println(String.format("%s -%s", i, dp[i]));
            max = Math.max(max, dp[i]);
            
            // put current max into mono queue
            while (!queue.isEmpty() && queue.peekLast()[1] <= dp[i]) {
                queue.pollLast();
            }
            queue.offer(new int[]{i, dp[i]});
        }
        return max;
    }
}
