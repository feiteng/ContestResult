class Solution {
    private static final Comparator<int[]> CMP = new Comparator<int[]>() {
        public int compare(int[] a, int[] b) {
            int cmp = Integer.compare(b[0], a[0]);
            if (cmp != 0) return cmp;
            return Integer.compare(a[1], b[1]);
        }  
    };
    
    
public int constrainedSubsetSum(int[] nums, int k) {
        int len = nums.length;
        
        // dp[i] == best sum that contains nums[i] 
        int[] dp = new int[len];
        
        PriorityQueue<int[]> maxQueue = new PriorityQueue<>(CMP);
        
        for (int i = 0; i < len; i += 1) {
            dp[i] = nums[i];
            if (!maxQueue.isEmpty()) {
                int[] p = maxQueue.peek();
                while (p[1] < i - k) {
                    maxQueue.poll();
                    if (maxQueue.isEmpty()) break;
                    p = maxQueue.peek();
                }
                
                if (!maxQueue.isEmpty()) dp[i] = Math.max(dp[i], dp[i] + p[0]);
            }
            
            
maxQueue.offer(new int[] { dp[i], i });
        }
        
        int max = Integer.MIN_VALUE;
        
        for (int n : dp) if (n > max) max = n;
        
        return max;
    }
}
