class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n=nums.length;
        int res=Integer.MIN_VALUE;
        PriorityQueue<int[]> queue=new PriorityQueue<int[]>(k,(a,b) -> b[1] - a[1]);
        for(int i=0;i<n;i++){
            
while(queue.size()>0&&i-queue.peek()[0]>k){
                queue.poll();
            }
            int max=queue.size()>0?queue.peek()[1]:0;
            int val=max>0?max+nums[i]:nums[i];
            res=Math.max(val,res);
            System.out.println(i+" "+val);
            
queue.add(new int[]{i,val});
        }
        return res;
    }
}
