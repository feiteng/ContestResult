class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int N = nums.length;
        
        int ans = Integer.MIN_VALUE;
        LinkedList<Integer> L = new LinkedList<>();
        int[] dp = new int[N];
        
        for (int i = N - 1; i >= 0; i--) {
            while (!L.isEmpty() && L.peekFirst() - i > k) L.pollFirst();
            int submax = (L.isEmpty() ? 0 : dp[L.peekFirst()]);
            int sum = nums[i] + (submax > 0 ? submax : 0);
            
            dp[i] = sum;
            
ans = Math.max(ans, sum);
            
            while (!L.isEmpty() && dp[L.peekLast()] < sum) L.pollLast();
            L.offerLast(i);
        }
        
        return ans;
    }
}
