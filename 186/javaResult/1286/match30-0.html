class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int max = 0;
        int[] dp = new int[nums.length];
        PriorityQueue<Integer> pq = new PriorityQueue<>(k, Collections.reverseOrder());
        for(int i = 0; i < nums.length; ++i) {
            if(i == 0) {
                int t = nums[i];
                max = t;
                dp[i] = t;
                pq.add(t);
            } else {
                int t = nums[i] + pq.peek();
                t = Math.max(nums[i], t);
                dp[i] = t;
                if(t > max) max = t;
                if(i >= k) {
                    int tp = dp[i - k];
                    if(tp == pq.peek()) pq.poll();
                }
                while(!pq.isEmpty() && t > pq.peek()) pq.poll();
                pq.add(t);
            }
        }
        return max;
    }
}
