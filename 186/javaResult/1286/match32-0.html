class Solution {

	public int constrainedSubsetSum(int[] nums, int k) {
		int dp[] = new int[nums.length], max = Integer.MIN_VALUE;
		TreeMap<Integer, Integer> map = new TreeMap<>();
		map.put(0, 1);
		for (int i = 0; i < dp.length; i++) {
			dp[i] = nums[i] + map.lastKey();
			if (i - k >= 0) {
				map.put(dp[i - k], map.get(dp[i - k]) - 1);
				if (map.get(dp[i - k]) == 0) {
					map.remove(dp[i - k]);
				}
			}
			map.put(dp[i], 1 + map.getOrDefault(dp[i], 0));
		}
		for (int i : dp) {
			max = Math.max(max, i);
		}
		return max;
	}
}
