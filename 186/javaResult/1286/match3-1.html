class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        int ans = nums[0];
        dp[0] = nums[0];
        
PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->b[0]-a[0]);    // val, ind
        pq.add(new int[]{ dp[0], 0 });
        for(int i = 1; i < n; i++) {
            dp[i] = nums[i];
            int[] mx = pq.poll();
            while(mx[1] < i-k) mx = pq.poll();
            dp[i] = Math.max(dp[i], mx[0] + nums[i]);
            pq.add(mx);
            
pq.add(new int[]{ dp[i], i });
            ans = Math.max(ans, dp[i]);
        }
        return ans;
    }
}
