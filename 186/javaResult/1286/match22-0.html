class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
         int n = nums.length;
        int[] dp = new int[n];
        int res =nums[0];
        Deque<Integer> dq = new ArrayDeque<>();
        for (int i=n-1;i>=0;i--){
            while (!dq.isEmpty()&&(dq.peekLast()-i>k||dp[dq.peekLast()]<0)){
                dq.pollLast();
            }
            dp[i] = nums[i]+(dq.isEmpty()?0:dp[dq.peekLast()]);
            while (!dq.isEmpty()&&dp[i]>dp[dq.peekFirst()]){
                dq.pollFirst();
            }
            res =Math.max(res,dp[i]);
            dq.addFirst(i);
            //System.out.println(Arrays.toString(dp));
        }
        return res;
    }
}
