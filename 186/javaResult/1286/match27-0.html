class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        if(nums==null || nums.length==0){
            return 0;
        }
        
        PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>(){
            @Override
            public int compare(int[] o1,int[] o2){
                return o2[1] - o1[1];
            }
        });
        
        int n = nums.length;
        // int[] dp = new int[n];
        // int max = nums[0];
        // for(int i=0;i<n;i++){
        //     dp[i] = nums[i];
        //     for(int j=1;j<=k;j++){
        //         if(i-j>=0){
        //             dp[i] = Math.max(dp[i],dp[i-j] + nums[i]);
        //         }else{
        //             break;
        //         }
        //     }
        //     max = Math.max(max,dp[i]);
        // }
        
        int max = nums[0];
        for(int i=0;i<n;i++){
            int cur = nums[i];
            while(!queue.isEmpty()){
                if(queue.peek()[0]+k<i){
                    queue.poll();
                }else{
                    cur = Math.max(cur,cur+queue.peek()[1]);
                    break;
                }
            }
            queue.add(new int[] {i,cur});
            max = Math.max(max,cur);
        }
        
        return max;
    }
}
