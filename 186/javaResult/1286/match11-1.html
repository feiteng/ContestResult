class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        Deque<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < n; i++) {
            if (q.size() > 0 && i - q.peekFirst() > k) q.pollFirst(); 
            int tmp = (q.size() > 0) ? Math.max(dp[q.peekFirst()] + nums[i], nums[i]) : nums[i];
            while (!q.isEmpty() && dp[q.peekLast()] < tmp) q.pollLast();
            q.offerLast(i);
            // for (int j = Math.max(0,i - k); j < i; j++) {
            //     tmp = Math.max(tmp, dp[j]);
            // }
            dp[i] = tmp;
        
}
        int res = Integer.MIN_VALUE;
        for (int i : dp) {
            res = Math.max(res, i);
        }
        return res;
    }
}
