class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        //System.out.println(nums.length + " " + k);
        
        int n = nums.length;
        
        
        
        
int[] dp = new int[n];
        dp[0] = nums[0];
        
        int ans = nums[0];
        for (int i = 1; i < n; i++) {
            dp[i] = nums[i];//choose from i
            
            for (int j = i-1; j >= Math.max(0, i-k); j--) {
                int temp = dp[j] + nums[i];
                
dp[i] = Math.max(temp, dp[i]);
                
                //positive -> must pick
                if (nums[j] > 0) {
                    break;
                }
                
            }
            
            // for (int j = Math.max(0, i-k); j <= i-1; j++) {
            //     int temp = dp[j] + nums[i];
            //     dp[i] = Math.max(temp, dp[i]);
            // }
            ans = Math.max(ans, dp[i]);
        }
        
        return ans;
        
        
    }
}
