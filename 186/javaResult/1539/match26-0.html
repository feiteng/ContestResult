class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        TreeMap<Integer, Stack<Integer>> map = new TreeMap();
        int N = 0;
        for (int i = 0; i < nums.size(); ++i) {
            for (int j = 0; j < nums.get(i).size(); ++j) {
                int idx = i + j;
                map.putIfAbsent(idx, new Stack());
                map.get(idx).push(nums.get(i).get(j));
                N ++;
            }
        }
        
        int[] ans = new int[N]; int k = 0;
        for (Map.Entry<Integer, Stack<Integer>> entry : map.entrySet()) {
            Stack<Integer> stack = entry.getValue();
            while(!stack.isEmpty())
                ans[k++] = stack.pop();
        }
        return ans;
    }
}
