class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        int m = nums.size();
        List<Integer> list = new ArrayList<>();
        
        Queue<int[]> qq = new LinkedList<>();
        qq.offer(new int[]{0, 0});

        Map<Integer, Set<Integer>> map = new HashMap<>();

        
while(!qq.isEmpty())
        {
            int[] cur = qq.poll();
            int x = cur[0], y = cur[1];
            if(!map.containsKey(x))
            {
                map.put(x, new HashSet<>());
            
}
            if(map.get(x).contains(y)) continue;
            if(x < 0 || x >= m) continue;
            if(nums.get(x).size() <= y) continue;

            list.add(nums.get(x).get(y));

            map.get(x).add(y);

            qq.offer(new int[]{x + 1, y});
            qq.offer(new int[]{x, y + 1});
        }

        int size = list.size();
        int[] ans = new int[size];
        for(int i = 0; i < size; i++) 
        {
            ans[i] = list.get(i);
        }
        return ans;

    }
}
