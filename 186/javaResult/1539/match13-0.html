class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> vec) {
        
        int m = vec.size();
        
        Queue<int[]> qu = new LinkedList<>();
        qu.offer(new int[]{0, 0});
        Map<Integer, Set<Integer>> visited = new HashMap<>();
        List<Integer> arrayList = new ArrayList<>();
        int i1 = 0, j1 = 1;
        int i2 = 1, j2 = 0;
        while(!qu.isEmpty()){
            
            int[] curVec = qu.poll();
            int i = curVec[0], j = curVec[1];
            if(!visited.containsKey(i)) visited.put(i, new HashSet<>());
            if(visited.get(i).contains(j)) continue;
            if(i < 0 || i >= m) continue;
            if(vec.get(i).size() <= j) continue;
            arrayList.add(vec.get(i).get(j));
            visited.get(i).add(j);
            qu.offer(new int[]{i + i2, j + j2});
            qu.offer(new int[]{i + i1, j + j1});
        }
        
        int sizeN = arrayList.size();
        int[] res = new int[sizeN];
        
        for(int i = 0; i < sizeN; i++){
            res[i] = arrayList.get(i);
        }
        
        return res;

    }
}
