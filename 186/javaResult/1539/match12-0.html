class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        int m = nums.size();
        List<Integer> list = new ArrayList<>();
        
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{0, 0});
        Map<Integer, Set<Integer>> map = new HashMap<>();
        while(!q.isEmpty())
        {
            int[] cur = q.poll();
            int x = cur[0], y = cur[1];
            if(!map.containsKey(x)) map.put(x, new HashSet<>());
            if(map.get(x).contains(y)) continue;
            if(x < 0 || x >= m) continue;
            if(nums.get(x).size() <= y) continue;
            list.add(nums.get(x).get(y));
            map.get(x).add(y);
            q.offer(new int[]{x + 1, y});
            q.offer(new int[]{x, y + 1});
        }

        int s = list.size();
        int[] re = new int[s];
        for(int i = 0; i < s; i++) re[i] = list.get(i);
        return re;

    }
}
