class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
       
        
        PriorityQueue<Point> sizeIn =new PriorityQueue<Point>(new Comparator<Point>() {

			@Override
			public int compare(Point o1, Point o2) {

				// TODO Auto-generated method stub
				if((o1.i+o1.j)!=(o2.i+o2.j)) {
					return o1.i+o1.j-o2.i-o2.j;
				}
				return o2.i-o1.i;
				
			}

        	
        });
    
        int numElements = 0;
        
for(int i=0;i<nums.size();i++) {
        	for(int j=0;j<nums.get(i).size();j++) {
        		sizeIn.add(new Point(i,j));
                numElements++;
        	}
        }
        int [] ret = new int[numElements];
        
int counter =0;
        while(!sizeIn.isEmpty()) {
        	Point x = sizeIn.poll();
        	ret[counter] = nums.get(x.i).get(x.j);
            counter++;
        }
        return ret;
        
    }
	public class Point{
		int i; int j;
		public Point(int i,int j) {
			this.i = i; this.j = j;
		}
	}
}
