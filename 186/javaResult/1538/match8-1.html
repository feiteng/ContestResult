class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int[] sum = new int[k + 1];
        for (int i = 1; i <= k; i++) {
            sum[i] = sum[i - 1] + cardPoints[n - i];
        }
        int maxScore = sum[k], currentSum = 0;
        
for (int i = 0; i < k; i++) {
            currentSum += cardPoints[i];
            maxScore = Math.max(currentSum + sum[k - i - 1], maxScore);
        }
        return maxScore;
    }
}
