class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int left = 0, right = 0, n = cardPoints.length, ans = 0;
        if (k >= n) {
            for (int i = 0; i < n; i++) {
                ans += cardPoints[i];
            }
            return ans;
        }
        
        for (int i = n - 1; i >= n - k; i--) {
            right += cardPoints[i];
        }
        ans = left + right;
        for (int i = 0; i < k; i++) {
            left += cardPoints[i];
            right -= cardPoints[n - k + i];
            ans = Math.max(ans, left + right);
        }
        return ans;
    }
}
