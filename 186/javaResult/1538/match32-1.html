class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int result = 0;
        int ls = 0, rs = 0;
        int n = cardPoints.length;
        
        
for (int i = 0; i < k; i++) {
            ls += cardPoints[i];
        }
        
        result = ls;
        
        for (int i = k - 1; i >= 0; i--) {
            ls -= cardPoints[i];
            rs += cardPoints[n - 1 - (k - 1 - i)];
            result = Math.max(result, ls + rs);
        }
        
        return result;
    }
}
