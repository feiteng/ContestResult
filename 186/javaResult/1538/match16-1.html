class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int len = cardPoints.length;
        int[] pre = new int[k + 1];
        int[] post = new int[k + 1];
        for (int i = 0; i < k; i++) {
            pre[i + 1] = pre[i] + cardPoints[i];
            post[i + 1] = post[i] + cardPoints[len - 1 - i];
        
}
        int res = 0;
        for (int i = 0; i <= k; i++) {
            res = Math.max(res, pre[i] + post[k - i]);
        }
        return res;
    }
}
