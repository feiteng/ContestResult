class Solution {
    public int maxScore(int[] cardPoints, int k) {
    	int n=cardPoints.length;
    	int[] head=new int[k+1], bottom=new int[k+1];
    	for(int i=1;i<=k;i++){
			head[i]=head[i-1]+cardPoints[i-1];
		}
    	if(n==k) return head[k];
    	for(int i=1;i<=k;i++){
			bottom[i]=bottom[i-1]+cardPoints[n-i];
		}
    	int max=0;
    	for(int i=0;i<=k;i++){
			max=Math.max(max, head[i]+bottom[k-i]);
		}
    	return max;
    }
}
