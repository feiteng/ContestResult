class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] + cardPoints[i];
        }
        int totalsum = sum[n];
        int max = 0;
        for (int i = 0; i + (n - k) <= n; i++) {
            max = Math.max(max, totalsum - (sum[i + n - k] - sum[i]));
        }
        return max;
    }
}
