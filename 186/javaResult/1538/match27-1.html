class Solution {
    public int maxScore(int[] cp, int k) {
        int n = cp.length;
        int[] fb = new int[k+1];
        int[] fe = new int[k+1];
        for (int i = 0; i < k; ++i) {
            fb[i+1] = fb[i]+cp[i];
        }
        
for (int i = 1; i <= k; ++i) {
            fe[i] = fe[i-1]+cp[n-i];
        }
        int ans = 0;
        for (int i = 0; i <= k; ++i) {
            ans = Math.max(ans, fb[i]+fe[k-i]);
        }
        return ans;
    }
}
