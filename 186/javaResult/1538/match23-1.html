class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int[] nums = new int[n*2];
        for (int i = 0; i < n; i++) {
            nums[i] = cardPoints[i];
            nums[n+i] = cardPoints[i];
        }
        
        int ans = 0;
        for (int i = n-k; i < n; i++) {
            ans += nums[i];
        
}
        
        int curr = ans;
        for (int i = 0; i < k; i++) {
            curr += nums[n + i];
            curr -= nums[n-k+i];
            ans = Math.max(ans, curr);
        }
        return ans;
        
    }
