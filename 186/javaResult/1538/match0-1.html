class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int[] presum = new int[n + 1];
        int[] rearsum = new int[n + 1];
        for(int i = 1; i <= n ; i++){
            presum[i] = presum[i-1] + cardPoints[i-1];
            rearsum[i] = rearsum[i-1] + cardPoints[n-i];
        }
        int res = 0;
        for(int i = 0 ; i <= k; i++){
            res = Math.max(res, presum[i] + rearsum[k - i]);
        }
        return res;
    }
}
