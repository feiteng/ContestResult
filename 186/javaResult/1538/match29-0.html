class Solution {
    public int maxScore(int[] points, int k) {
        int n = points.length;
        int[] left = new int[n];
        int[] right = new int[n];
        left[0] = points[0];
        for (int i = 1; i < n; i++) {
            left[i] = left[i - 1] + points[i];
        }
        right[n - 1] = points[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            right[i] = right[i + 1] + points[i];
        }
        int score = Math.max(left[k - 1], right[n - k]);
        for (int i = 1; i < k; i++) {
            score = Math.max(score, left[i - 1] + right[n - (k - i)]);
        }
        return score;
    }
}
