class Solution {
    public int maxScore(String s) {
        char[] chars = s.toCharArray();
        int n = chars.length;
        int max = 0;
        int os = 0, zs = 0;
        for (int i = 0; i < n; i++) {
            if (chars[i] == '1') {
                os++;
            
}
        }
        for (int i = 0; i < n - 1; i++) {
            if (chars[i] == '0') {
                zs++;
            } else {
                os--;
            }
            max = Math.max(max, zs + os);
        }
        return max;
    }
}
