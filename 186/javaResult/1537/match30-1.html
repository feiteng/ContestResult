class Solution {
    public int maxScore(String s) {

        int ones = 0, zeros = 0;
        for (char c : s.toCharArray()) {
            
if (c == '1') {
                ones++;
            } else {
                zeros++;
            }
        }

        int ans = 0;

        int l1 = 0, l0 = 0;
        
for (int i = 0; i < s.length() - 1; i++) {
            char c = s.charAt(i);
            if (c == '1') {
                l1++;
            } else {
                l0++;
            }

            ans = Math.max(ans, l0 + ones - l1);

        }
        return ans;

    }
}
