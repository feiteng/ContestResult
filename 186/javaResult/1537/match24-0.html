class Solution {
    public int maxScore(String s) {
    	int n=s.length();
    	int[] zero_num=new int[n], one_num=new int[n];
    	int zero_count=0,one_count=0;
    	for(int i=0;i<n;i++){
			if(s.charAt(i)=='0') zero_count++;
			if(s.charAt(n-i-1)=='1') one_count++;
			zero_num[i]=zero_count;
			one_num[n-i-1]=one_count;
		}
    	int max=0;
    	for(int i=0;i<n-1;i++){
			max=Math.max(max, zero_num[i]+one_num[i+1]);
		}
    	return max;
    }
}
