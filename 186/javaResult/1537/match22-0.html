class Solution {
    public int maxScore(String s) {
        int result = 0;
        int zs = 0, os = 0;
        
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                os++;
            }
        }
        
        for (int i = 0; i < s.length() - 1; i++) {
            if (s.charAt(i) == '1') {
                os--;
            } else {
                zs++;
            }
            
            result = Math.max(result, zs + os);
        }
        
        return result;
    }
}
