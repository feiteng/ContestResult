class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int maxi = 0;
        int one = 0;
        for(char c : s.toCharArray())
        {
            one += c - '0';
        }

        int zero = 0;
        for(int i = 0; i < n - 1; i++)
        {
            char c = s.charAt(i);
            if(c == '0') zero++;
            else one--;
            maxi = Math.max(maxi, zero + one);
        }

        return maxi;
    }    
}
