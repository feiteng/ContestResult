class Solution {
    public int maxScore(String s) {
        int left_one = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') 
{
                left_one++;
            }
        }
        
        int count_zero = 0;
        int ans = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            if (s.charAt(i) == '1') {
                left_one--;
            } else {
                                count_zero++;

            }
            
            ans = Math.max(ans, count_zero + left_one);
        }
        return ans;
    }
}
