class Solution {

    public int maxScore(String s) {
        int n = s.length();
        int[] zero = new int[n];
        int[] one = new int[n];
        zero[0] = s.charAt(0) == '0' ? 1 : 0;
        one[n-1] = s.charAt(n-1) == '1' ? 1 : 0;
        for (int i = 1; i < n; i++) {
            zero[i] = zero[i-1];
            if (s.charAt(i) == '0') {
                zero[i]++;
            }
        }
        for (int i = n-2; i >= 0; i--) {
            one[i] = one[i+1];
            if (s.charAt(i) == '1') {
                one[i]++;
            }
        }

        int maxScore = 0;
        for (int i = 1; i < n; i++) {
            maxScore = Math.max(maxScore, zero[i-1] + one[i]);
        }

        return maxScore;
    }

}
