class Solution {
    public int maxScore(String s) {
        
int oneCnt = 0;
        for(int i = 0; i < s.length(); i++) {
            if(s.charAt(i) == '1') { oneCnt++; }
        }
        int zeroCnt = 0;
        int score = -1;
        
for(int i = 0; i < s.length() - 1; i++) {
            if(s.charAt(i) == '0') { zeroCnt++; }
            else { oneCnt--; }
            score = Math.max(score, oneCnt + zeroCnt);
        }
        return score;
    }
}
