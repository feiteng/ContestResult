class Solution {
    public int maxScore(String s) {
        int res = 0, l = s.length();
        for (int i = 0; i < l - 1; i++) 
{
            int temp = 0;
            for (int j = 0; j <= i; j++) {
                if (s.charAt(j) == '0') temp++;
            }
            for (int j = i + 1; j < l; j++) 
{
                if (s.charAt(j) == '1') temp++;
            }
            res = Math.max(res, temp);
        }
        return res;
    }
}
