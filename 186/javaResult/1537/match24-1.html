class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int[] left = new int[n+1];
        int[] right = new int[n+1];    
        int count0 = 0, count1 = 0;
        for(int i = 1; i <= n; i++){
            if(s.charAt(i-1) == '0')
                count0++;
            left[i] = count0;
        }
        
        for(int i = n-1; i >= 0; i--){
            if(s.charAt(i) == '1')
                count1++;
            
right[i] = count1;
        }
        
        int max = 0;
        for(int i = 1; i <= n-1; i++){
            max = Math.max(max, left[i] + right[i]);
        }
        
        return max;
    }
}
