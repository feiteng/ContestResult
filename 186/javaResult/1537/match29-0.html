class Solution {
    public int maxScore(String s) {
        int totalZero = 0; 
        
        for (char c : s.toCharArray()) {
            if (c == '0') {
                totalZero++;
            }
        }
        
        int leftZero = 0;
        int result = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            if (s.charAt(i) == '0') {
                leftZero++;
            }
            int currScore = leftZero + (s.length() - i - 1 - totalZero + leftZero);
            result = Math.max(result, currScore);
        }
        
        return result;
    }
}
