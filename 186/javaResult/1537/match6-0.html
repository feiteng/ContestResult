class Solution {

	public int maxScore(String s) {
		int ans = 0;
		for (int i = 1; i < s.length(); i++) {
			int score = 0;
			for (int j = i; j < s.length(); j++) {
				if (s.charAt(j) == '1') {
					score++;
				}
			}
			for (int j = 0; j < i; j++) {color="#77bfc7">
				if (s.charAt(j) == '0') {
					score++;
				}
			}
			ans = Math.max(ans, score);
		}
		return ans;
	}
}
