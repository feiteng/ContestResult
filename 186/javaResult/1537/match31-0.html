class Solution {
    public int maxScore(String s) {
        int[] zCount = new int[s.length()];
        int[] oneCount = new int[s.length()];
        
        int pre = 0;
        for(int i = 0; i < zCount.length; ++i){
            if(s.charAt(i) == '0'){
                zCount[i] = ++pre;
            }else{
                zCount[i] = pre;
            }
        }
        
        pre = 0;
        for(int i = s.length()-1; i >= 0; i--){
            pre = pre + (s.charAt(i) == '1' ? 1:0);
            oneCount[i] = pre;
        }
        
        int max = 0;
        for(int i = 0; i < s.length()-1; ++i){
            max = Math.max(zCount[i]+oneCount[i+1], max);
        }
        return max;
    }
}
