class Solution {
    public int maxScore(String s) {
        char[] ch = s.toCharArray();
        int maxSum=0, split=0, lz=0, ro=0;
        for( char i: ch ){
            if(i=='1') ro++;
        }
        for( int i=0; i<ch.length-1; i++ ){
            if( ch[i] == '1' ){
                ro--;
            }else{
                lz++;
            }
            if( lz+ro>maxSum )
            maxSum = lz+ro;
        }
        return maxSum;
    }
}
