class Solution {
    public int maxScore(String s) {
        int one = 0;
        int zero = 0;
        for(char c : s.toCharArray()){
            if(c == '0'){
                zero++;
            }
            else{
                one++;
            }
        }
        
        int cur_zero = 0;
        int cur_one = 0;
        int res = 0;
        int loc = 0;
        for(char c : s.toCharArray()){
            if(c == '0'){
                cur_zero++;
            }
            else{
                cur_one++;
            }
            
            loc++;
            if(loc == s.length()){
                break;
            
}
            res = Math.max(res, cur_zero + one - cur_one);
        }
        
        return res;
    }
}
