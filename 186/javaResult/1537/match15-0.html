class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int sum0 = 0, sum1 = 0;
        for (int i = 0; i < n; i++) {
            if (s.charAt(i) == '0') {
                sum0++;
            }else {
                sum1++;
            }
        }
        
        int res = 0;
        int curr0 = 0, curr1 = sum1;
        for (int i = 0; i < n - 1; i++) {
            if (s.charAt(i) == '0') {
                curr0++;
            }else {
                curr1--;
            }
            res = Math.max(res, curr0 + curr1);
        }
        return res;
    }
}
