class Solution {
    public int maxScore(String s) {
        int res = 0;
        int num0 = 0;
        int num1 = 0;
        
        for(int i=0 ; i<s.length() ; i++){
            int num = s.charAt(i)-'0';
            if(num==0){
                ++num0;
            } else{
                ++num1;
            }
        }
        int tmp0 = 0;
        int tmp1 = 0;
        if(s.charAt(0)-'0'==0){
            ++tmp0;
        } else{
            ++tmp1;
        }
        res = tmp0 + num1 - tmp1;
        for(int i=1 ; i<s.length()-1 ; i++){
            int num = s.charAt(i)-'0';
            if(num==1){
                ++tmp1;
            } else{
                ++tmp0;
            }
            res = Math.max(res,tmp0 + num1 - tmp1);
        }
        return res;
    }
}
