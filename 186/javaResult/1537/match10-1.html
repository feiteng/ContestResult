class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int[] zero = new int[n];
        int[] one = new int[n];
        if (s.charAt(0) == '0') {
            zero[0] = 1;
        }
        
for (int i = 1; i < n; i++) {
            zero[i] = zero[i-1];
            if (s.charAt(i) == '0') {
                zero[i]++;
            }
        }
        
        if (s.charAt(n-1) == '1') {
            one[n-1] = 1;
        }
        
for (int i = n-2; i>=0; i--) {
            one[i] = one[i+1];
            if (s.charAt(i) == '1') {
                one[i]++;
            }
        }
        
        int ans = 0;
        for (int i = 0; i+1 < n; i++) {
            ans = Math.max(ans, zero[i] + one[i+1]);
        }
        return ans;
    }
}
