class Solution {
    public int maxScore(String s) {
        int t = 0;
        int a = 0;
        for (char ch : s.toCharArray()) if (ch == '1') t++;
        int ans = 0;
        for (int i = 0; i < s.length() - 1; i++) 
{
            char ch = s.charAt(i);
            if (ch == '1') t--;
            else a++;
            ans = Math.max(ans, t + a);
        }
        return ans;
    }
}
