class Solution {
    public int maxScore(String s) {
        int zeros = 0;
        int ones = 0;
        
        for (char c : s.toCharArray()) {
            if (c == '0') zeros += 1;
            else ones += 1;
        }
        
        int rZ = (s.charAt(0) == '0') ? 1 : 0;
        int rO = ones - (1 - rZ);
        
        int max = rZ + rO;
        
        int len = s.length();
        for (int i = 1; i < len - 1; i += 1) {
            char c = s.charAt(i);
            if (c == '0') rZ += 1;
            else rO -= 1;
            
            max = Math.max(max, rZ + rO);
        }
        return max;
    }
}
