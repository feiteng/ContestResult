class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int[] c0 = new int[n], c1 = new int[n];
        c0[0] = s.charAt(0) == '0' ? 1 : 0;
        c1[n - 1] = s.charAt(n - 1) == '1' ? 1 : 0;
        for (int i = 1; i < n; i++) {
            c0[i] = s.charAt(i) == '0' ? c0[i - 1] + 1 : c0[i - 1];
            c1[n - i - 1] = s.charAt(n - i - 1) == '1' ? c1[n - i] + 1 : c1[n - i];
        }
        int max = 0;
        for (int i = 1; i < n; i++) {
            max = Math.max(max, c0[i - 1] + c1[i]);
        }
        return max;
    }
}
