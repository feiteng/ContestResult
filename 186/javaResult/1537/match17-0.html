class Solution {
    public int maxScore(String s) {
        int max = 0;
        for (int i = 0; i < s.length() - 1; i ++) {
            int sum = 0;
            for (int j = 0; j <= i; j ++) {
                if(s.charAt(j) == '0') sum++;
            }
            for (int k = i + 1; k < s.length(); ++k) {
                if(s.charAt(k) == '1') sum++;
            }
            max = Math.max(sum, max);
        }
        return max;
    }
}
