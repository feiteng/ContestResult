class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        rows 
= []
        for row in grid:
            all_zeros = True
            for colind in range(n - 1, -1, -1):
                if row[colind] == 1:
                    rows.append(colind)
                    all_zeros = False
                    break
            if all_zeros:
                rows.append(0)
        res = 0
        for i in range(n):
            if rows[0] <= i:
                rows.pop
(0)
            else:
                found = False
                for j in range(1, len(rows)):
                    if rows[j] <= i:
                        res += j
                        rows.pop(j)
                        found = True
                        break
                if not found:
                    return -1
        return res
