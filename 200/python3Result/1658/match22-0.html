class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        rows = []
        for row in grid:
            num_zeros = 0
            for val in reversed(row):
                if val == 1:
                    break
                num_zeros += 1
            rows.append(num_zeros)
        ans = 0
        for i in range(n):
            lo = n - i - 1
            for j in range(i, n):
                if rows[j] >= lo:
                    for k in range(j, i, -1):
                        rows[k-1], rows[k] = rows[k], rows[k-1]
                        ans += 1
                    break
            else:
                return -1
        return ans
