class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        # use bubble sort
        R, C = len(grid), len(grid[0])
        cnt0 = []
        
        
for r in range(R):
            cnt = 0
            for c in range(C-1, -1, -1):
                if grid[r][c] == 0:
                    cnt += 1
                else:
                    break
            cnt0.append(cnt)
                    
            
        # sort from the last element
        res = 0
        for r in range(R):
            p = r
            while p < R and cnt0[p] < R - 1 - r:
                p += 1
            if p >= R: return -1
            res += p-r
            cnt0 = cnt0[:r] + [cnt0[p]] + cnt0[r:p] + cnt0[p+1:]
            
        return res
