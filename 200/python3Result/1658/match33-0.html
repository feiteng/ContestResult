class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        zero = [0]*n  # 记录从哪个位置开始全为0
        for i in range(n):
            for j in range(n-1,-1,-1):
                if grid[i][j]:
                    zero[i] = j
                    break
        print(zero)
        # 对所有的i 使得zero[i] <= i
        res = 0
        for i in range(n):
            if zero[i] <= i:
                continue
            check = False
            for j in range(i+1,n):
                if zero[j] <= i:
                    res += j - i
                    # print(i,j,zero,zero[i+1:j+1],zero[i:j])
                    zero[i+1:j+1] = zero[i:j]
                    # print(i,j,zero,zero[i+1:j+1],zero[i:j])
                    check = True
                    
                    break
            if not check:
                return -1
        return res
