class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len
(grid)
        h = [0] * n
        for i in range(n):
            height = 0
            for j in range(n-1,-1,-1):
                if grid[i][j] == 0:
                    height += 1
                else:
                    break
            h[i] = height
        print(h)
        
        ans = 0
        for i in range(n):
            f = 0
            if h[i] >= n - i - 1:
                continue 
            for j in range(i+1, n):
                if h[j] >= n - i - 1:
                    h[i:j+1] = [h[j]] + h[i:j]
                    # h[i], h[j] = h[j], h[i]
                    ans 
+= j - i
                    f = 1
                    break
            if f == 0:
                return -1
        return ans
