class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        d = []
        n = len(grid)
        for i, row in enumerate(grid):
            c = 0
            for j in range(n - 1, -1, -1):
                if grid[i][j] == 0:
                    c += 1
                else:
                    break
            d.append(c)
        ans = 0
        for i in range(n - 1, -1, -1):
            k = i
            for j, c in enumerate(d):
                if c >= k:
                    ans += j
                    d.remove(c)
                    break   
            else:
                return -1
        return ans
