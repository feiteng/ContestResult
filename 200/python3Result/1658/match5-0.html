class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        length = len(grid)
        zeros = [0]*length
        for i in range(length):
            t = 0
            for j in range(length-1,-1,-1):
                if grid[i][j] == 0:
                    t += 1
                else:
                    break
            zeros[i] = t
        res = 0
        for i in range(length):
            if zeros[i]>=(length-1-i):
                continue
            else:
                flag = True
                for j in range(i+1,length):
                    if zeros[j]>=(length-i-1):
                        res += j-i
                        flag = False
                        t = zeros.pop(j)
                        zeros.insert(i,t)
                        break
                if flag:
                    return -1
        return res
