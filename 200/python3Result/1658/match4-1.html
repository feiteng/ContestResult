class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        temp = [0]*n
        for i in range(n):
            count = 0
            for j in range(n-1,-1,-1):
                if grid[i][j] == 0:
                    count += 1
                else:
                    break
            temp[i] = count
            
        ans = 0
        for i in range(n):
            if temp[i] < n-i-1:
                for j in range(i+1,n):
                    if temp[j] >= n-i-1:
                        ans += j-i
                        break
                else:
                    return -1
                a = j
                while a > i:
                    temp[a],temp[a-1] = temp[a-1],temp[a]
                    a -= 1
        return ans
