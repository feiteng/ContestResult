class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        for c in range(n):
            columnsum = 0
            for row in grid:
                if row[c]:
                    columnsum += 1
            if columnsum > n - c:
                return -1
        rowsums = []
        swaps = 0
        for row in grid:
            flag = False
            for i in range(n - 1, -1, -1):
                if row[i]:
                    rowsums.append(i + 1)
                    flag = True
                    break
            if not flag:
                rowsums.append(0)
                    
        for i in range(n):
            flag = False
            for j in range(len(rowsums)):
                if rowsums[j] <= i + 1:
                    swaps += j
                    rowsums.pop(j)
                    flag = True
                    break
            if not flag:
                return -1
        return swaps
