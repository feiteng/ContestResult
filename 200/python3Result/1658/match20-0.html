class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        arr=[] #右边0的数量
        n=len(grid)
        for i in range(n):
            curr=n-1
            for j in range(n-1,-2,-1):
                if grid[i][j]==1:
                    curr=n-1-j
                    break
            arr.append(curr)
        ans=0
        #print(arr)
        for i in range(n-1):
            j=0
            while arr[j]<n-1-i:
                j+=1
                if j>=n-i:
                    return -1
            ans+=j
            arr.pop(j)
        return ans
