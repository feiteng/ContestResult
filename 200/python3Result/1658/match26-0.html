class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        r0 = []
        for line in grid:
            zeros = 0
            for i in range(n - 1, 0, -1):
                if line[i] == 0:
                    zeros += 1
                else:
                    break
            r0.append(zeros)
            
        res = 0
        for i in range(0, n):
            lower = n - i - 1
            if r0[i] >= lower:
                continue
            
            target = -1
            for j in range(i + 1, n):
                if r0[j] >= lower:
                    target = j
                    break        
            if target == -1:
                return target
            for j in range(target, i, -1):
                r0[j], r0[j - 1] = r0[j - 1], r0[j]
                res += 1
            
        return res
