class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        N=len(grid)
        #慢慢来，也就是说，对于第j列，需要有j个0位于对角线上方
        #所以可以先统计下，如果某列j的全部0数目小于j的话，那就说明无论如何也完成不了，return-1
        #接着可以统计每一行的末尾0的数目，
        for j in range(N):
            cur=0
            for i in range(N):
                if grid[i][j]==0:
                    cur+=1
            if cur<j:
                return -1
        
        num=[]
        
for i in range(N):
            cur=0
            for j in range(N-1,-1,-1):
                if grid[i][j]==0:
                    cur+=1
                else:
                    break
            num.append(cur)
        result=0
        #现在要将之构成排列
        for i in range(N):
            if num[i] < N-i-1:
                flag=True
                #这个时候就要想办法换行了，找到在它之后的，第一个满足要求的j，这两者换行，然后num[i+1:j+1]=num[i:j]
                for j in range(i+1,len(num)):
                    if num[j] >= N-i-1:
                        result+=(j-i)
                        num[i+1:j+1]=num[i:j]
                        flag
=False
                        break
                if flag:
                    return -1
        return result
