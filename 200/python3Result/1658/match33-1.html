class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        rcnt = [0] * n
        for i in range(n):
            for j in range(n - 1, -1, -1):
                if grid[i][j] != 0:
                    break
                else:
                    rcnt[i] += 1
                
        srcnt = rcnt.copy()
        srcnt.sort()
        ok = True
        for i in range(n):
            if srcnt[i] < i:
                ok = False
                break
                
        if not ok: return -1
        
        # print(rcnt)
        cnt 
= 0
        for i in range(n):
            if rcnt[i] >= n - i - 1: continue
            jj = -1
            for j in range(i + 1, n):
                # print(j, rcnt[j], n - i - 1)
                if rcnt[j] >= n - i - 1:
                    jj = j
                    break
                    
            # print(i, jj)
            cnt += jj - i
            # rcnt[i], rcnt[jj] = rcnt[jj], rcnt[i]
            rcnt[i + 1: jj + 1] = rcnt[i: jj]
            # print(rcnt)
        
        # print("-------")
        return cnt
