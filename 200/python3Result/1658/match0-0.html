class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        dp = [[0] * n for _ in range(n)]
        
        for i in range(n):
            dp[i][n-1] = int(grid[i][n-1] == 0)
            for j in range(n-2, -1, -1):
                dp[i][j] = dp[i][j+1] + int(grid[i][j] == 0)
                
        print(dp)
        cnt = 0
        for i in range(n-1):
            j = i + 1
            if dp[i][j] < n - j:
                flag = False
                for k in range(i+1, n):
                    if dp[k][j] >= n - (i+1):
                        tmp = dp.pop(k)
                        dp.insert(i, tmp)
                        cnt += (k - i)
                        flag = True
                        # print(i, k)
                        # print(dp)
                        break
                if not flag:
                    return -1
                
        
        return cnt
