class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        n_zero = [-1] * n
        for i in range(n):
            zero_num = 0
            for j in range(n-1, 0, -1):
                if grid[i][j] == 0:
                    zero_num += 1
                else:
                    break
            n_zero[i] = zero_num
        res = 0
        # 几个0
        for i in range(n):
            j = i
            while j < n and n_zero[j] < (n-1) - i:
                j += 1
            if j == n:
                return -1
            res += j - i
            n_zero = n_zero[:i] + [n_zero[j]] + n_zero[i:j] + n_zero[j+1:]
        return res
