class Solution:
	def minSwaps(self, grid: [[int]]) -> int:
		arr 
= []
		n = len(grid)
		for i in range(n):
			c = 0
			for j in range(n-1, -1, -1):
				if grid[i][j] == 0:
					c += 1
				else:
					break
			arr.append(c)
		ans = 0
		for i in range(n-1, -1, -1):
			found = False
			for j in range(len(arr)):
				if arr[j] >= i:
					ans += j
					del arr[j]
					found = True
					break
			if not found:
				return -1
			
		return ans
