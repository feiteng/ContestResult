def check(row, x):
    n = len(row)
    for i in range(x + 1, n):
        if row[i] == 1:
            return False
    return True
def move(grid, x, y):
    if x == y:
        return grid
    # print("x", x, y, grid)
    # print(grid[0:y], grid[x:x+1], grid[y:x], grid[x+1:-1])
    new_grid = grid[0:y] + grid[x:x+1] + grid[y:x] + grid[x+1:]
    return new_grid
    
class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        ans = 0
        for i in range(n):
            flag = True
            for j in range(i, n):
                if check(grid[j], i):
                    grid = move(grid, j, i)
                    # print(grid)
                    ans += j - i
                    flag = False
                    break
            if flag:
                return -1
        return ans
