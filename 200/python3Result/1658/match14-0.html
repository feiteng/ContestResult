class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        A = []
        for i in range(n):
            cnt = 0
            for j in range(n-1, -1, -1):
                if grid[i][j] == 0:
                    cnt += 1
                else:
                    break
            A.append(cnt)
        ans = 0
        for i in range(n):
            j = i
            while j < n:
                if A[j] >= n-i-1:
                    break
                j += 1
            if j == n:
                return -1
            
            for k in range(j, i, -1):
                A[k], A[k-1] = A[k-1], A[k]
                ans += 1
        
        return ans
