class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        ans = 0
        for i in range
(1, n):
            flag = False
            for j in range(i - 1, n):
                f = True
                for k in range(i, n):
                    if grid[j][k] == 1:
                        f = False
                        break
                if f:
                    temp = [t for t in grid[j]]
                    for k in reversed(range(i, j + 1)):
                        grid[k] = [t for t in grid[k - 1]]
                    grid[i - 1] = temp
                    ans 
+= j - i + 1
                    flag = True
                    break
            if not flag:
                return -1
            
        return ans
