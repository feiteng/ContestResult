class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        l = [0] * n
        for i in range(n):
            count = 0
            for j in range(n - 1, -1, -1):
                if grid[i][j] == 0:
                    count += 1
                else:
                    break
            l[i] = count
        #print(l)
        res = 0
        for i in range(n):
            target = n - 1 - i
            found = False
            for j in range(i, n):
                if l[j] >= target:
                    res += j - i
                    found = True
                    l = [l[j]] + l[:j] + l[j+1:]
                    break
            if not found:
                return -1
        return res
