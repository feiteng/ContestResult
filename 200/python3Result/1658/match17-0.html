class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        zeros = []
        for i in range(n):
            count = 0
            for j in range(n - 1, -1, -1):
                if grid[i][j] == 0:
                    count += 1
                else:
                    break
            zeros.append(count)
        if sum(zeros) < (n - 1) * n // 2:
            return -1
        
        result = 0
        for i in range(n - 1, -1, -1):
            k = n - i - 1
            found = False
            for j in range(k, n):
                if zeros[j] >= i:
                    zeros[k:] = [zeros[j]] + zeros[k:j] + zeros[j + 1:]
                    result += j - k
                    found = True
                    break
            if not found:
                return -1
        return result
