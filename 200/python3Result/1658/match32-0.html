class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        N = len(grid)
        zeros = collections.deque()
        for i in range(N):
            count = 0
            for j in range(N-1, -1, -1):
                if grid[i][j] == 0:
                    count += 1
                else:
                    break
            zeros.append(count)
        res = 0
        for i in range(N):
            require = N - i - 1
            if zeros[i] < require:
                find = False
                for j in range(i+1, N):
                    if zeros[j] >= require:
                        res += j - i
                        del zeros[j]
                        zeros.appendleft(0)
                        find = True
                        break
                if not find:
                    return -1
        return res
