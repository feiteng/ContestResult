class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        
        suffixZeros = [0] * n
        for i in range(n):
            j = n - 1
            while grid[i][j] == 0 and j >= 0:
                j -= 1
            suffixZeros[i] = (n - 1) - j
            
        total = 0
            
        for row in range(n):
            # We need n - row - 1 zeros at the end.
            found = False
            for i in range(row, n):
                if suffixZeros[i] >= n - row - 1:
                    if i == row:
                        found = True
                        break
                    
                    else:
                        total += i - row
                        print(i, row, total)
                        
                        suffixZeros.insert(row, suffixZeros[i])
                        suffixZeros.pop(i + 1)
                        
                        found = True
                        break
                    
            if not found:
                return -1
                    
        return total
