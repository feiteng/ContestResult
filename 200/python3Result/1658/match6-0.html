class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        nums = []
        R, C = len(grid), len(grid[0])
        for r in range(R):
            j = C - 1
            while j >= 0 and grid[r][j] == 0:
                j -= 1
            nums.append(C - 1 - j)
        
        steps = 0
        for r in range(R):
            i = r
            while i < R and nums[i] < R - 1 - r:
                i += 1
            if i == R:
                return -1
            steps += i - r
            nums = nums[:r] + [nums[i]] + nums[r:i] + nums[i+1:]
            
        return steps
