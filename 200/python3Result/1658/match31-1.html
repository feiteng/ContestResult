class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        A = [0] * n
        for i in range(n):
            j = 0
            while j < n and grid[i][-1-j] == 0: j += 1
            A[i] = j
        # print(A)
        ans = 0
        for i in range(n-1, 0, -1):
            j = 0
            while j < len(A) and A[j] < i: j += 1
            if j >= len(A): return -1
            ans += j
            A.pop(j)
        return ans
