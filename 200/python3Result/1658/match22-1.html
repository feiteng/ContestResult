class Solution:
    def minSwaps(self, a: List[List[int]]) -> int:
        n = len(a)
        
        score = []
        
        for row in a:
            scr = 0
            for i in range(n-1, -1, -1):
                if row[i] == 1:
                    
break
                scr += 1
            score.append(scr)
            
        ret = 0
        
        for i in range(n):
            
            pivot = None
            
            for j in range(i, n):
                if score[j] >= n-i-1:
                    pivot = j
                    break
            
            if pivot == None:
                return -1
            
            ret += pivot - i
            score = score[:i] + [score[pivot]] + score[i: pivot] + score[pivot+1:]
        return ret
