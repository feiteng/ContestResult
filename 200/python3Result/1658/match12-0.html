class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        zero_cnt = []
        for i in range(n):
            cnt = 0
            for j in range(n-1, -1, -1):
                if grid[i][j] != 0:
                    break
                else:
                    cnt += 1
            zero_cnt.append(cnt)
            
        step = 0
        for i in range(n):
            flag = 0
            for j in range(n-i):
                if zero_cnt[j] >= n-i-1:
                    step += j
                    zero_cnt.pop(j)
                    flag = 1
                    break
            if flag == 0:
                return -1
            
        return step
