class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        arr = []
        for i in range(n):
            count = 0
            for j in range(n - 1, -1, -1):
                if grid[i][j] == 0:
                    count += 1
                else:
                    break
            arr.append(count)
        
        ans = 0
        for i in range(n):
            for j in range(len(arr)):
                if arr[j] >= n - i:
                    ans += j
                    arr.pop(j)
                    break
            if len(arr) > n - i:
                return -1
        return ans
