class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        n = len(grid)
        end_z = []
        for i in range(n):
            cnt = 0
            for j in range(n-1, -1, -1):
                if grid[i][j] == 0:
                    cnt += 1
                else:    
                    break
            end_z.append(cnt)
        
        total_move = 0
        for row_idx in range(n):
            print(end_z)
            need = n-row_idx-1
            
            for pick in range(row_idx, n):
                if end_z[pick] >= need:
                    found = pick
                    break
            else:        
                return -1
            
            # print(found)
            total_move += pick-row_idx
            
            end_z.insert(row_idx, end_z.pop(found))
        return total_move
