class Solution:
    def countGoodTriplets(self, arr: List[int], a: int, b: int, c: int) -> int:
        res = 0
        N = len(arr)
        for i in range(N):
            for j in range(i+1, N):
                for k in range(j+1, N):
                    if abs(arr[j]-arr[i]) <= a and abs(arr[k]-arr[j]) <= b and abs(arr[k]-arr[i]) <= c:
                        res += 1
        return res
