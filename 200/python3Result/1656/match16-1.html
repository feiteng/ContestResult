class Solution:
    def countGoodTriplets(self, ar: List[int], a: int, b: int, c: int) -> int:
        ans=0
        n=len(ar)
        for i in range(n):
            for j in range(i+1,n):
                for k in range(j+1,n):
                    if abs(ar[i]-ar[j])<=a and abs(ar[j]-ar[k])<=b and abs(ar[i]-ar[k])<=c:
                        ans+=1
        return ans
