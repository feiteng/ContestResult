class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        l1 = len(nums1)
        l2 
= len(nums2)
        length = max(l1, l2)
        top = [0] * (l1 + 1)
        bot = [0] * (l2 + 1)
        one = l1 - 1
        two = l2 - 1
        while one >= 0 and two >= 0:
            if (nums1[one] > nums2[two]):
                top[one] += (nums1[one] + top[one + 1])
                one -= 1
            elif (nums1[one] < nums2[two]):
                bot[two] += (nums2[two] + bot[two + 1])
                two -= 1
            else:
                top
[one] += (nums1[one] + top[one + 1])
                bot[two] += (nums2[two] + bot[two + 1])
                top[one] = max(top[one], bot[two])
                bot[two] = top[one]
                one 
-= 1
                two -= 1
        while one >= 0:
            top[one] += (nums1[one] + top[one + 1])
            one -= 1
        while two >= 0:
            bot[two] += (nums2[two] + bot[two + 1])
            two -= 1
        return max(top[0], bot[0]) % 1000000007;
