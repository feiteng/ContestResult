class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        curr1, curr2=0, 0
        sub1, sub2=0, 0
        res=0
        while curr1<len(nums1) and curr2<len(nums2):
            if nums1[curr1]<nums2[curr2]:              
                sub1+=nums1[curr1]
                curr1+=1
            elif nums2[curr2]<nums1[curr1]:
                sub2+=nums2[curr2]
                curr2+=1
            else:
                res+=max
(sub1, sub2)+nums1[curr1]
                curr1+=1
                curr2+=1
                sub1, sub2=0, 0
        while curr1<len(nums1):
            sub1+=nums1[curr1]
            curr1+=1
        while curr2<len(nums2):
            sub2+=nums2[curr2]
            curr2+=1
        res+=max(sub1, sub2)
        return res%(10**9+7)
