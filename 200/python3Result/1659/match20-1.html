class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        p1 = len(nums1) - 1
        p2 = len(nums2) - 1
        sum1 = 0
        sum2 = 0
        while p1 >= 0 and p2 >= 0:
            if nums1[p1] > nums2[p2]:
                sum1 += nums1[p1]
                p1 -= 1
            elif nums1[p1] < nums2[p2]:
                sum2 += nums2[p2]
                p2 -= 1
            else:
                max_sum = max(sum1, sum2)
                sum1 = max_sum + nums1[p1]
                sum2 = max_sum + nums2[p2]
                p1 -= 1
