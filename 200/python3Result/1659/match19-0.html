class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        mask = 10**9+7
        cur1, cur2 = 0, 0
        pos1, pos2 = 0, 0
        while pos1 < len(nums1) and pos2 < len(nums2):
            if nums1[pos1] < nums2[pos2]:
                cur1 += nums1[pos1]
                pos1 += 1
            elif nums1[pos1] > nums2[pos2]:
                cur2 += nums2[pos2]
                pos2 += 1
            else:
                cur = max(cur1, cur2) + nums1[pos1]
                cur1, cur2 = cur, cur
                pos1 += 1
                pos2 += 1
        if pos1 == len(nums1):
            cur2 += sum(nums2[pos2:])
        else:
            cur1 += sum(nums1[pos1:])
        return max(cur1, cur2) % mask
