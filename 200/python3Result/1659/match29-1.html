class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        dp1, dp2 = [-float('inf')]*n1+[0], [-float('inf')]*n2+[0]  # nums_i[j]到末尾的最大和

        cur1, cur2 
= n1-1, n2-1 # 双指针
        while cur1>=0 or cur2>=0:
            if cur1>=0 and cur2>=0 and nums1[cur1]==nums2[cur2]:
                dp2[cur2] = max(dp1[cur1+1],dp2[cur2+1])+nums2[cur2]
                dp1[cur1] = max(dp1[cur1+1],dp2[cur2+1])+nums1[cur1]
                cur2 -= 1
                cur1 -= 1
            elif cur2<0 or( cur1>=0 and nums1[cur1]>nums2[cur2]):
                dp1[cur1] = nums1[cur1]+dp1[cur1+1]
                cur1 -= 1
            else:
                dp2[cur2] = nums2[cur2]+dp2[cur2+1]
                cur2 -= 1

        
        return max(dp1[0],dp2[0])%(10**9+7)
