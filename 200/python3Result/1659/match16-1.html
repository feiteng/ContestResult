class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        i 
= j = 0
        nums1_sum = 0
        nums2_sum = 0
        ans = 0
        while i<len(nums1) and j<len(nums2):
            if nums1[i] < nums2[j]:
                nums1_sum += nums1[i]
                i += 1
            elif nums1[i] > nums2[j]:
                nums2_sum += nums2[j]
                j += 1
            else:
                ans += max(nums1_sum, nums2_sum)+nums1[i]
                nums1_sum = 0
                nums2_sum = 0
                i += 1
                j += 1
        # print(ans)
        # print(i, j)
        # print(nums1_sum, nums2_sum)
        ans += max(nums1_sum+sum(nums1[i:]), nums2_sum+sum(nums2[j:]))
        # if i<len(nums1):
        #     ans += sum(nums1[i:])
        # if j<len(nums1):
        #     ans += sum(nums2[j:])
                
        return ans%1000000007
