class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        
        i = 0
        j = 0
        sum1 = 0
        sum2 = 0
        total = 0
        while i < len(nums1) or j < len(nums2):
            if i == len(nums1):
                total += max(sum1, sum2 + sum(nums2[j:]))
                break
            if j == len(nums2):
                total += max(sum1 + sum(nums1[i:]), sum2)
                break
            if nums1[i] < nums2[j]:
                sum1 += nums1[i]
                i += 1
            elif nums1[i] > nums2[j]:
                sum2 += nums2[j]
                j += 1
            else:
                total += max(sum1, sum2)
                sum1 = 0
                sum2 = 0
                total += nums1[i]
                i += 1
                j += 1
        return total % 1000000007
