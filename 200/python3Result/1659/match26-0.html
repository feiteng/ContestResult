class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        dp1,dp2=0,0
        s1,s2=0,0
        L1,L2=len(nums1),len(nums2)
        while s1<L1 and s2<L2:
            if nums1[s1]<nums2[s2]:
                dp1+=nums1[s1]
                s1+=1
            elif nums1[s1]>nums2[s2]:
                dp2+=nums2[s2]
                s2+=1
            else:
                dp1=max(dp1,dp2)+nums1[s1]
                dp2=dp1
                s1+=1
                s2+=1
        while s1<L1:
            dp1+=nums1[s1]
            s1+=1
        while s2<L2:
            dp2+=nums2[s2]
            s2+=1
        return max(dp1,dp2)%(10**9+7)
