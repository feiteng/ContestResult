class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        i = len(nums1) - 1
        j = len(nums2) - 1
        max1 = max2 = 0
        while i>=0 and j>=0:
            if nums1[i] == nums2[j]:
                max1 = max2 = max(max1,max2)+nums1[i]
                i -= 1
                j -= 1
            elif nums1[i] > nums2[j]:
                max1 += nums1[i]
                i -= 1
            elif nums1[i] < nums2[j]:
                max2 += nums2[j]
                j -= 1
            # print(nums1[i],nums2[j], max1,max2)
        if i>= 0: max1 += sum(nums1[:i+1])
        if j>= 0: max2 += sum(nums2[:j+1])
        return max(max1,max2) % (10**9 +7)
