class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        p = 0
        q = 0
        a = 0
        b = 0
        while p != len(nums1) and q != len(nums2):
            if nums1[p] < nums2[q]:
                a += nums1[p]
                p += 1
            elif nums1[p] > nums2[q]:
                b += nums2[q]
                q += 1
            else:
                a = max(a, b) + nums1[p]
                b = a
                p += 1
                q += 1
        if p != len(nums1):
            for i in range(p, len(nums1)):
                a += nums1[i]
        if q != len(nums2):
            for i in range(q, len(nums2)):
                b += nums2[i]
        #print(a % 1000000007, b % 1000000007)
        return max(a, b) % 1000000007
