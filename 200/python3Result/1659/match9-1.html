class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = 0
        sum2 = 0
        p1 
= 0
        p2 = 0
        while (p1 < len(nums1)) and (p2 < len(nums2)):
            if nums1[p1] < nums2[p2]:
                sum1 += nums1[p1]
                p1 += 1
            elif nums1[p1] > nums2[p2]:
                sum2 += nums2[p2]
                p2 += 1
            elif nums1[p1] == nums2[p2]:
                sum1 
= max(sum1,sum2) + nums1[p1]
                sum2 = sum1
                p1 += 1
                p2 += 1
        if p1 < len(nums1):
            sum1 += sum(nums1[p1:])
        if p2 < len(nums2):
            sum2 += sum(nums2[p2:])
        return max(sum1,sum2) % 1000000007
