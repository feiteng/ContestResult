class Solution:
    def maxSum(self, a: List[int], b: List[int]) -> int:
        m, n = len(a), len(b)
        dp1 = [0] * (m + 1)
        dp2 = [0] * (n + 1)
        i, j = 0, 0
        while i < m and j < n:
            if a[i] == b[j]:
                dp1[i + 1] = max(dp1[i], dp2[j]) + a[i]
                dp2[j + 1] = max(dp2[j], dp1[i]) + b[j]
                i += 1
                j += 1
            elif a[i] < b[j]:
                dp1[i + 1] = dp1[i] + a[i]
                i += 1
            else:
                dp2[j + 1] = dp2[j] + b[j]
                j += 1
        while i < m:
            dp1[i + 1] = dp1[i] + a[i]
            i += 1
        while j < n:
            dp2[j + 1] = dp2[j] + b[j]
            j += 1
        return max(dp1[m], dp2[n]) % (10 ** 9 + 7)
