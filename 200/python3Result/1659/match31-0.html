class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        m = len(nums2)
        scores1 = Counter()
        scores2 = Counter()
        i = j = 0
        mod = 10**9 + 7
        while i < n or j < m:
            if i < n and j < m:
                if nums1[i] < nums2[j]:
                    scores1[i] = scores1[i-1] + nums1[i]
                    #scores1[i] %= mod
                    i += 1
                elif nums1[i] > nums2[j]:
                    scores2[j] = scores2[j-1] + nums2[j]
                    #scores2[j] %= mod
                    j += 1
                else:
                    scores1[i] = max(scores1[i-1], scores2[j-1]) + nums1[i]
                    #scores1[i] %= mod
                    scores2[j] = max(scores1[i-1], scores2[j-1]) + nums2[j]
                    #scores2[j] %= mod
                    i += 1
                    j += 1
            elif i < n:
                scores1[i] = scores1[i-1] + nums1[i]
                #scores1[i] %= mod
                i += 1
            else:
                scores2[j] = scores2[j-1] + nums2[j]
                #scores2[j] %= mod
                j += 1
        return max(scores1[n-1], scores2[m-1]) % mod
