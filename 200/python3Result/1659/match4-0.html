class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        p=0
        q=0
        ans=0
        sum1=0
        sum2=0
        while p<len(nums1) and q<len(nums2):
            if nums1[p]==nums2[q]:
                ans+=max(sum1,sum2)
                sum1=sum2=nums1[p]
                p+=1
                q+=1
            elif nums1[p]>nums2[q]:
                sum2+=nums2[q]
                q+=1
            else:
                sum1+=nums1[p]
                p+=1
        while p<len(nums1):
            sum1+=nums1[p]
            p+=1
        while q<len(nums2):
            sum2+=nums2[q]
            q+=1
        ans+=max(sum1,sum2)
        return int(ans%(1e9+7))
