class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        sum1, sum2 = 0, 0
        i , j = 0, 0
        
        while i < len(nums1) and j < len(nums2):  
            if nums1[i] > nums2[j]:
                sum2 += nums2[j]
                j += 1
            elif nums1[i] < nums2[j]:
                sum1 += nums1[i]
                i += 1
            else:
                m_sum = max(sum1, sum2) + nums1[i]
                sum1, sum2 = m_sum, m_sum
                i += 1
                j += 1
        
        if i < len(nums1):
            sum1 += sum(nums1[i:])
        if j < len(nums2):
            sum2 += sum(nums2[j:])
        
        return max(sum1, sum2) % (7 + 10 ** 9)
