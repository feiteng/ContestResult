class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        nums1 = [0] + nums1
        nums2 = [0] + nums2
        dp = {0:0}
        l, r = 1, 1
        while l < len(nums1) and r < len(nums2):
            if nums1[l] < nums2[r]:
                dp[nums1[l]] = dp[nums1[l-1]] + nums1[l]
                l += 1
            elif nums1[l] > nums2[r]:
                dp[nums2[r]] = dp[nums2[r-1]] + nums2[r]
                r += 1
            else:
                dp[nums1[l]] = max(dp[nums1[l-1]], dp[nums2[r-1]]) + nums1[l]
                l += 1
                r += 1
        while l < len(nums1):
            dp[nums1[l]] = dp[nums1[l-1]] + nums1[l]
            l += 1
        while r < len(nums2):
            dp[nums2[r]] = dp[nums2[r-1]] + nums2[r]
            r += 1
        return max(list(dp.values()))%1000000007
