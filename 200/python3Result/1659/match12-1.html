class Solution:
    def maxSum(self, arr1: List[int], arr2: List[int]) -> int:
        mod = int(10 ** 9 + 7)
        dp1, dp2 
= 0, 0
        i, j = 0, 0
        while i < len(arr1) and j < len(arr2):
            if arr1[i] == arr2[j]:
                x = max(dp1, dp2)
                dp1, dp2 = arr1[i] + x, arr1[i] + x
                i, j 
= i + 1, j + 1
            elif arr1[i] < arr2[j]:
                dp1 += arr1[i]
                i += 1
            else:
                dp2 += arr2[j]
                j += 1
        while i < len(arr1):
            dp1 += arr1[i]
            i += 1
        while j < len(arr2):
            dp2 += arr2[j]
            j += 1
        return max(dp1, dp2) % mod
