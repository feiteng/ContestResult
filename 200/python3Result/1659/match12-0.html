class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        mod = 10**9+7
        p1 = p2 = 0
        # from collections import Counter
        # c1 = Counter(nums1)
        # c2 = Counter(nums2)
        # c = c1 & c2
        pre1 = pre2 = 0
        ans = 0
        while p1<len(nums1) and p2<len(nums2):
            if nums1[p1] == nums2[p2]:
                ans += max(pre1,pre2)
                ans += nums1[p1]
                ans %= mod
                p1+=1
                p2+=1
                pre1=pre2=0
            elif nums1[p1]<nums2[p2]:
                pre1+= nums1[p1]
                p1+=1
            else:
                pre2+= nums2[p2]
                p2+=1
        while p1<len(nums1):
            pre1+= nums1[p1]
            p1+=1
        while p2<len(nums2):
            pre2+= nums2[p2]
            p2+=1
        ans += max(pre1,pre2)
        return ans%mod
