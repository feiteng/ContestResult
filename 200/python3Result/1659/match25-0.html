class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        n1 = len(nums1)
        n2 = len(nums2)
        dp1 = [0] * n1
        dp2 = [0] * n2
        ptr1 = 0
        ptr2 = 0
        while ptr1 != n1 and ptr2 != n2:
            if nums1[ptr1] > nums2[ptr2]:
                dp2[ptr2] = (dp2[ptr2 - 1] + nums2[ptr2]) if ptr2 != 0 else nums2[ptr2]
                ptr2 += 1
            elif nums1[ptr1] < nums2[ptr2]:
                dp1[ptr1] = (dp1[ptr1 - 1] + nums1[ptr1]) if ptr1 != 0 else nums1[ptr1]
                ptr1 += 1
            else:
                if ptr1 == 0 and ptr2 == 0:
                    dp1[0], dp2[0] = nums1[0], nums1[0]
                elif ptr1 == 0:
                    dp1[0] = nums1[0] + dp2[ptr2 - 1]
                    dp2[ptr2] = max(nums1[0], dp2[ptr2 - 1] + nums2[ptr2])
                elif ptr2 == 0:
                    dp2[0] = nums2[0] + dp1[ptr1 - 1]
                    dp1[ptr1] = max(nums2[0], dp1[ptr1 - 1] + nums1[ptr1])
                else:
                    dp1[ptr1] = max(nums1[ptr1] + dp2[ptr2 - 1], nums1[ptr1] + dp1[ptr1 - 1])
                    dp2[ptr2] = max(nums2[ptr2] + dp1[ptr1 - 1], nums2[ptr2] + dp2[ptr2 - 1])
                
                ptr1 += 1
                ptr2 += 1
                
        while ptr1 != n1:
            dp1[ptr1] = dp1[ptr1 - 1] + nums1[ptr1]
            ptr1 += 1
        
        while ptr2 != n2:
            dp2[ptr2] = dp2[ptr2 - 1] + nums2[ptr2]
            ptr2 += 1
      
        return max(dp1[n1 - 1], dp2[n2 - 1]) % (10 ** 9 + 7)
