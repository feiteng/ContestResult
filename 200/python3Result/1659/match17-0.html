class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        nums1_idx, nums2_idx = 0, 0
        split_sum1, split_sum2 = [], []
        sum1, sum2 = 0, 0
        while nums1_idx < len(nums1) and nums2_idx < len(nums2):
            if nums1[nums1_idx] < nums2[nums2_idx]:
                sum1 += nums1[nums1_idx]
                nums1_idx += 1
            elif nums2[nums2_idx] < nums1[nums1_idx]:
                sum2 += nums2[nums2_idx]
                nums2_idx += 1
            else:
                split_sum1.append(sum1 + nums1[nums1_idx])
                split_sum2.append(sum2 + nums2[nums2_idx])
                nums1_idx += 1
                nums2_idx += 1
                sum1, sum2 = 0, 0
        
        while nums1_idx < len(nums1):
            sum1 += nums1[nums1_idx]
            nums1_idx += 1
        while nums2_idx < len(nums2):
            sum2 += nums2[nums2_idx]
            nums2_idx += 1
        split_sum1.append(sum1)
        split_sum2.append(sum2)
        #print(split_sum1, split_sum2)

        total = 0
        for num1, num2 in zip(split_sum1, split_sum2):
            total += max(num1, num2)
        return int(total % (1e9 + 7))
