class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        ans 
= s1 = s2 = 0
        i = ii = 0
        while i < len(nums1) and ii < len(nums2): 
            if nums1[i] < nums2[ii]: 
                s1 += nums1[i]
                i += 1
            elif nums1[i] > nums2[ii]:
                s2 += nums2[ii]
                ii += 1
            else: 
                ans += max(s1, s2)
                s1 = s2 = 0
                while i < len(nums1) and ii < len(nums2) and nums1[i] == nums2[ii]: 
                    ans += nums1[i]
                    i += 1
                    ii += 1
        s1 += sum(nums1[i:])
        s2 += sum(nums2[ii:])
        ans += max(s1, s2)
        return ans % 1_000_000_007
