class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        p1, p2 = 0, 0
        cnt1, cnt2 = 0, 0
        res = 0
        bp1, bp2 = -1, -1
        while p1 < len(nums1) and p2 < len(nums2):
            if nums1[p1] < nums2[p2]:
                cnt1 += nums1[p1]
                p1 += 1
            elif nums1[p1] > nums2[p2]:
                cnt2 += nums2[p2]
                p2 += 1
            else:
                res += max(cnt1, cnt2) + nums1[p1]
                cnt1, cnt2 = 0, 0
                bp1, bp2 = p1, p2
                p1 += 1
                p2 += 1
        cnt1, cnt2 = sum(nums1[bp1+1:]), sum(nums2[bp2+1:])
        res += max(cnt1, cnt2)
        return res % (10 ** 9 + 7)
