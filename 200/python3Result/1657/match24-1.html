class Solution:
    def getWinner(self, A: List[int], k: int) -> int:
        n 
= len(A)
        top = A[0]
        count = 0
        for i in range(1, n):
            if top < A[i]:
                top = A[i]
                count = 1
            else:
                count += 1
            if count == k:
                return top
        return max(A)
