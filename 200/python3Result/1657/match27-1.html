class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        i=1
        cnt=0
        head=0
        n=len(arr)
        while i<n:
            
if arr[head]>arr[i]:
                cnt+=1
                i+=1
                if cnt==k:
                    return arr[head]
            else:
                head=i
                i=head+1
                cnt=1
                if cnt==k:
                    return arr[head]
        return arr[head]
