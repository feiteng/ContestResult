class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k >= len(arr):
            return max(arr)
        
        t = 0
        mmax = arr[0]
        for i in range(1, len(arr)):
            if arr[i] > mmax:
                t = 1
                mmax = arr[i]
            else:
                t += 1
            if t == k:
                return mmax
        return max(arr)
