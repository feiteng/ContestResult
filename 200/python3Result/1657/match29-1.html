class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k >= len(arr):
            return max(arr)
        ans = max(arr[0], arr
[1])
        cnt = 1
        for i in range(2, len(arr)):
            if cnt == k:
                return ans
            if arr[i] > ans:
                ans = arr[i]
                cnt = 1
            else:
                cnt += 1
        return ans
