class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        k = min(k, n - 1)
        res = arr[0]
        count = 0
        for i in range(1, n):
            if arr[i] > res:
                res = arr[i]
                count = 1
            else:
                count += 1
            if count == k:
                return res
        return res
