class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        if k >= n - 1:
            return max(arr)
        cur = 0
        while cur < k:
            i = 1
            while cur < k and arr[0] > arr[i]:
                i += 1
                cur += 1
            if cur == k:
                return arr[0]
            arr = arr[i:] + arr[1:i][::-1] + [arr[0]]
            cur = 1
            # print(arr)
        return arr[0]
