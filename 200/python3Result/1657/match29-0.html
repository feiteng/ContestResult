class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        ret=max(arr[:2])
        wins=1
        for i in range(2,len(arr)):
            if wins>=k:
                return ret
            if arr[i]>ret:
                ret=arr[i]
                wins=1
            else:
                wins+=1
        return ret
