class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        wins = 0
        winner = arr[0]
        
for i in arr[1:]:
            if winner > i:
                wins += 1
            else:
                winner = i
                wins = 1
            if wins == k:
                return winner
        return winner
