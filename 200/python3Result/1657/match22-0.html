class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k >= len(arr):
            return max(arr)
        champion = arr[0]
        wins = 0
        for num in arr[1:]:
            if num < champion:
                wins += 1
                if wins >= k:
                    return champion
            else:
                champion = num
                wins = 1
                if k == 1:
                    return champion
        return champion
