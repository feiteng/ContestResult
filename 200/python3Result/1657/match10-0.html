class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        cur = arr[0]
        count = 0
        for num in arr[1:]:
            if num < cur:
                count += 1
                if count == k:
                    return cur
            else:
                cur = num
                count = 1
                if count == k:
                    return cur
        
        return cur
