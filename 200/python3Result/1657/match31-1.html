class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        curr_win 
= 0
        prev_max = arr[0]
        for i in range(1, len(arr)):
            if arr[i] > prev_max:
                curr_win = 1
                prev_max = arr[i]
            else:
                curr_win += 1
                
            if curr_win >= k:
                return prev_max
        return prev_max
