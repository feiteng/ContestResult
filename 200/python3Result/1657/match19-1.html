class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        if k >= n - 1:
            return max(arr)
        
        win = 0
        while True:
            i = 1
            while win < k and arr[i] < arr[0]:
                i += 1
                win += 1
            if win == k:
                return arr[0]
            
            arr = arr[i:n] + arr[1:i] + [arr[0]]
            win = 1
