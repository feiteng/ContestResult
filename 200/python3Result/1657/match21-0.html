class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k > len(arr):
            return max(arr)
        cnt = 0
        while(True):
            if arr[0] > arr[1]:
                cnt += 1
                arr.append(arr[1])
                del arr[1]
            else:
                cnt = 1
                arr.append(arr[0])
                del arr[0]
            if(cnt == k):
                return arr[0]
