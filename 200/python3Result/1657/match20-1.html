class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k>=len(arr)-1: 
return max(arr)
        count=k
        while count:
            if arr[0]<arr[1]:
                count=k-1
                arr.append(arr.pop(0))
            else:
                count-=1
                arr.append(arr.pop(1))
        return arr[0]
