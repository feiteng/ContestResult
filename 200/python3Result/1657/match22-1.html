class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        a 
= arr[0]
        m = 0
        for x in arr[1:]:
            if a > x:
                m += 1
                if m == k:
                    return a
            else:
                a = x
                m = 1
                if m == k:
                    return a
        return a
