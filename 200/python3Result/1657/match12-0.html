class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        if k >= len(arr)-1:
            return max(arr)
        local = arr[0]
        cnt = 0
        for i in range(1,len(arr)):
            #print(cnt,k,arr[i],local)
            if cnt == k:
                break
            if arr[i]>local:
                #print("?")
                local = arr[i]
                cnt = 1
            else:
                cnt += 1
        return local
