class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        count = 0
        ans = arr[0]
        
for v in arr[1:]:
            if v > ans:
                ans = v
                count = 1
            else:
                count += 1
            # print(ans, count)
            if count == k:
                return ans
        return ans
