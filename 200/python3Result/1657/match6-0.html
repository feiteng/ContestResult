class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        cur_max = arr[0]
        cur_cnt = 0
        for x in arr[1:]:
            if x > cur_max:
                cur_max = x
                cur_cnt = 1
            else:
                cur_cnt += 1
            if cur_cnt == k:
                return cur_max
        return cur_max
