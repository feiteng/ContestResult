class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        max_num = arr[0]
        count = 0
        
for i in range(1, len(arr)):
            if arr[i] <= max_num:
                count += 1
            else:
                max_num = arr[i]
                count = 1
            if count == k:
                return max_num
        return max_num
