class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        if k >= n - 1:
            return max(*arr)
        
        win = 0
        while win < k:
            if arr[0] > arr[1]:
                win += 1
                arr.append(arr.pop(1))
            else:
                win = 1
                arr.append(arr.pop(0))
                
        return arr[0]
