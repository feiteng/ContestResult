class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        cur = arr[0]
        win = 0
        
        for n in arr[1:]:
            if n > cur:
                cur = n
                win = 1
            else:
                win += 1
                
            if win >= k:
                return cur
            
        return cur
