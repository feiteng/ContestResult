class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        arr2 = arr + arr
        if k >= len(arr):
            return max(arr)
        win = 0
        curr_best = arr2[0]
        for i in range(1, len(arr2)):
            #print(curr_best, arr2[i])
            if curr_best > arr2[i]:
                win += 1
            else:
                curr_best = arr2[i]
                win = 1
            if win >= k:
                return curr_best
        return arr[i]
