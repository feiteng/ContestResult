class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        winner = arr[0]
        cnt = 0
        for i in range(1, n):
            if arr[i] > winner:
                winner = arr[i]
                cnt = 1
            else:
                cnt += 1
            if cnt == k:
                return winner
        return winner
