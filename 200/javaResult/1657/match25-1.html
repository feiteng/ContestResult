class Solution {
    public int getWinner(int[] arr, int k) {
        int count = 0;
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (max > arr[i]) {
                count++;
                if (count == k) return max
;
            }
            else {
                max = arr[i];
                count = 1;
                if (count == k) {
                    return max;
                }
            }
        }
        return max;
    }
}
