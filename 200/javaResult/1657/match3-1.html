class Solution {
    public int getWinner(int[] arr, int k) {
        int value = arr[0];
        int count = 0;
        for (int idx = 1; idx < arr.length; idx++) {
            if (value >= arr[idx]) {
                count++;
            } else {
                value = arr[idx];
                count=1;
            }
            if (count == k) return value;
        }
        return value;
    }
}
