class Solution {
    public int getWinner(int[] arr, int k) {
        int n = arr.length;
        int currIdx = 0;
        int cnt = 0;
        for (int i = 1; i < n; i++) {
            
if (arr[currIdx] > arr[i]) cnt++;
            else {
                currIdx = i;
                cnt = 1;
            }
            if (cnt == k) return arr[currIdx];
        }
        return arr[currIdx];
    }
}
