class Solution {
    public int getWinner(int[] arr, int k) {
        int num=0,cur=0,i=1;
        while (i<arr.length){
            if(arr[i]<arr[cur]){
                i++;
                num++;
                if(num==k)
                    return arr[cur];
            }
            else{
                cur=i;
                i++;
                num=1;
                if(num==k)
                    return arr[cur];
            }
        }
        return arr[cur];
    }
}
