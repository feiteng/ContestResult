class Solution {
    public int getWinner(int[] arr, int k) {
        int len = arr.length,max = Math.max(arr[0],arr[1]),cnt = 1;
        if(k==cnt){
            return max;
        }
        for(int i = 2;i<len;i++) {
            if(arr[i] > max) {
                max = arr[i];
                cnt = 1;
            } else {
                cnt++;
            }
            if(cnt == k){
                return max;
            }
        }
        return max;
    }
}
