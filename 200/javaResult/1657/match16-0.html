class Solution {
    public int getWinner(int[] arr, int k) {
        int cur = arr[0];
        int winCount = 0;
        for (int i = 1; i < arr.length; i++) {
            if (cur > arr[i]) {
                winCount++;
            } else {
                cur = arr[i];
                winCount = 1;
            }
            if (winCount == k) {
                return cur;
            }
        }
        return cur;
    }
}
