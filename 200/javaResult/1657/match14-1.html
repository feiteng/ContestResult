class Solution {
    public int getWinner(int[] arr, int k) {
        int ans=arr[0],cur=0;
        
        for(int i=1;i<arr.length;i++){
            if(ans>arr[i])cur++;
            else 
{
                cur=1;
                ans=arr[i];
            }
            if(cur==k)return ans;
        }
        return ans;
    }
}
