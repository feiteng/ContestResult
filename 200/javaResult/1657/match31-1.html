class Solution {
    public int getWinner(int[] arr, int k) {
        int maxi = 0;
        int n = arr.length;
        for(int i = 1; i < n; ++i) {
            if (arr[i] > arr[maxi]) {
                maxi = i;
            }
        }
        for(int i = 1, won = 0, v = arr[0]; i < maxi; ++i) {
            if (arr[i] > v) 
{
                v = arr[i];
                won = 0;
            }
            if (++won == k) {
                return v;
            }
        }
        return arr[maxi];
    }
}
