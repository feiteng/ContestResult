class Solution {
    public int getWinner(int[] arr, int k) {
        int res = arr[0];
        int acc = 0;
        for(int i = 1; i < arr.length; ++i) {
            int t = arr[i];
            
if(res > t) {
                ++acc;
            } else {
                acc = 1;
                res = t;
            }
            if(acc == k) return res;
        }
        return res;
    }
}
