class Solution {
    public int countGoodTriplets(int[] arr, int a, int b, int c) {
        int result = 0;
        int n = arr.length;
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 1; j < n - 1; j++) {
                for (int k = j + 1; k < n; k++) {
                    if (arr[i] - arr[j] <= a && arr[j] - arr[i] <= a && arr[j] - arr[k] <= b && arr[k] - arr[j] <= b && arr[i] - arr[k] <= c && arr[k] - arr[i] <= c) {
                        result++;
                    }
                }
            }
        }
        return result;
    }
}
