class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int fa = 0;
        for(int i = 0; i<n-1; i++){
            int temp = -1;
            for(int j = i; j< n; j++){
                int k = i+1;
                for(k = i+1; k<n;k++){
                    if(grid[j][k]!=0){
                        break;
                    }
                }
                if(k==n){
                    temp = j;
                    break;
                }
            }
            if(temp==-1){
                return -1;
            }
            fa += (temp-i);
            for(int j = temp; j>i; j--){
                swap(grid, j, j-1, n);
            }
            //System.out.println(fa);
        }
        return fa;
    }
    public void swap(int[][] grid, int i, int j, int n){
        for(int x = 0; x<n;x++){
            int temp = grid[i][x];
            grid[i][x] = grid[j][x];
            grid[j][x] = temp;
        }
    }
}
