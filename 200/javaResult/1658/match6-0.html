class Solution {
    public int minSwaps(int[][] grid) {
        List<Integer> cnts = new LinkedList<>();
        for (int i = 0; i < grid.length; i++) {
            int cnt = 0;
            for (int j = grid[0].length -1; j >= 0; j--) {
                if(grid[i][j] != 0) break;
                cnt++;
            }
            cnts.add(cnt);
        }
        int move = 0;
        int needed = grid[0].length - 1;
        while (needed > 0){
            boolean found = false;
            for (int i = 0; i < cnts.size(); i++) {
                if(cnts.get(i) >= needed){
                    move+=i;
                    cnts.remove(i);
                    found = true;
                    break;
                }
            }
            if(!found) return -1;
            needed--;
        }
        return move;
    }
}
