class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = n - 1; j > 0; j--) {
                if (grid[i][j]==0) arr[i]++;
                else break;
            }
        }
        //System.out.println(Arrays.toString(arr));
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int target = n - i - 1;
            for (int j = i; j < n; j++) {
                if (arr[j] >= target) {
                    //System.out.println(i + ":" + j);
                    ans += j - i;
                    for (int k = j; k > i; k--) {
                        arr[k] = arr[k-1];
                    }
                    arr[i] = target;
                    break;
                }
            }
            if (arr[i] < target) return -1;
        
        }
        return ans;
    }
}
