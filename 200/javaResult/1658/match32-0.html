class Solution {
    public int minSwaps(int[][] g) {
        int len  =g.length;
        int s[]  =new int[len];
        int ll = g[0].length;
        for(int i=0;i<len;++i){
            for(int j=ll-1;j>=0;--j){
                if(g[i][j]==0){
                    s[i]++;
                }else{
                    break;
                }
                
            }
        }
        int ss[] = s.clone();
        Arrays.sort(ss);
        for(int i=0;i<len;++i){
            if(ss[i]<i){
                return -1;
            }
        }
        int tm  =0;
        for(int i=0;i<len;++i){
            int nd = len-1-i;
            int fd = -1;
            for(int j=i;j<len;++j){
                if(s[j]>=nd){
                    fd = j; 
                    break;
                }
            }
            for(int j=fd;j>i;--j){
                int p = s[j];
                s[j] = s[j-1];
                s[j-1]  = p;
                tm++;
            }
            
            
        }
        return tm;
    }
}
