class Solution {
    public int minSwaps(int[][] grid) {
        //sanity
        int row = grid.length;
        int col = grid[0].length;
        
List<Integer> list = new ArrayList<>();
        for (int r = 0; r < row; r++) {
            int num = 0;
            for (int c = col - 1; c >= 0; c--) {
                if (grid[r][c] == 1) {
                    num = c;
                    
break;
                }
            }
            list.add(num);
        }
        int swaps = 0;
        for (int r = 0; r < row; r++) {
            
for (int i = r; i < row; i++) {
                if (list.get(i) <= r) {
                    swaps += i - r;
                    int num = list.get(i);
                    list.remove(i);
                    list.add(r, num);
                    break;
                }
            }
            if (list.get(r) > r) {
                return -1;
            }
        }
        return swaps;
    }
}
