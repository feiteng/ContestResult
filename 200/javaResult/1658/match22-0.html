class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] rowNums = new int[n];
        int[] copy = new int[n];
        for (int j = 0; j < n; j++) {
            int num = 0;
            for (int k = n-1; k >=0; k--) {
                if (grid[j][k] == 1) {
                    break;
                }
                num++;
            }
            rowNums[j] = num;
            copy[j] = num;
        }
        boolean[] used = new boolean[n];
        int ret = 0;
        for (int j = 0; j < n; j++) {
            int needed = n - j - 1;
            boolean bad = true;
            for (int k = 0; k < n; k++) {
                if (used[k]) {
                    continue;
                }
                if (rowNums[k] >= needed) {
                    used[k] = true;
                    bad = false;
                    break;
                }
                ret++;
            }
            if (bad) {
                return -1;
            }
        }
        return ret;
    }
}
