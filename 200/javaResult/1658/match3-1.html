class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] zs = new int[n];
        
        for (int i = 0; i < n; i++) {
            
for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    zs[i]++;
                } else {
                    break;
                }
            }
        }
        
        int result = 0;
        
        for (int i = 0; i < n; i++) {
            if (zs[i] >= n - 1 - i) {
                continue;
            }
            
int j = i + 1;
            for (; j < n; j++) {
                if (zs[j] >= n - 1 - i) {
                    break;
                }
            }
            
            if (j == n) {
                return -1;
            }
            
            int tmp = zs[j];
            for (int k = j; k > i; k--) {
                zs[k] = zs[k-1];
            }
            result += (j - i);
            zs[i] = tmp;
        }
        
        return result;
    }
}
