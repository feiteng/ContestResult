class Solution {
    public int minSwaps(int[][] grid) {
        final int N = grid.length;
        int[] target = new int[N];
        for (int i = 0; i < N; ++i) {
            int pos = -1;
            
for (int j = N - 1; j >= 0; --j) {
                if (grid[i][j] == 1) {
                    pos = j;
                    break;
                }
            }
            // if (pos < 0) {
            //     return -1;
            // }
            target[i] = pos;
        
}
        // System.out.println(Arrays.toString(target));
        int swaps = 0;
        for (int i = 0; i < N; ++i) {
            int want = -1;
            for (int j = i; j < N; ++j) {
                
if (target[j] <= i) {
                    want = j;
                    break;
                }
            }
            if (want < 0) {
                return -1;
            }
            for (int j = want; j > i; --j) {
                swap(target, j, j - 1);
                ++swaps;
            }
        }
        return swaps;
        // // boolean[] seen = new boolean[N];
        // // for (int i = 0; i < N; ++i) {
        // //     if (seen[target[i]]) {
        // //         return -1;
        // //     }
        // //     seen[target[i]] = true;
        // // }
        // int swaps = 0;
        // while (true) {
        //     boolean swapped = false;
        //     for (int i = 1; i < N; ++i) {
        //         if (target[i] < target[i - 1]) {
        //             swap(target, i, i - 1);
        //             ++swaps;
        //             swapped = true;
        //         }
        //     }
        //     if (!swapped) {
        //         break;
        //     }
        // }
        // for (int i = 0; i < N; ++i) {
        //     if (target[i] > i) {
        //         return -1;
        //     }
        // }
        // return swaps;
    }
    
    private static void swap(int[] A, int i, int j) {
        int tmp = A[i];
        A[i] = A[j];
        A[j] = tmp;
    }
}
