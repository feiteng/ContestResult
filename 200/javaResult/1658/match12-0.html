class Solution {
    public int minSwaps(int[][] grid) {
        List<Integer> list = new ArrayList();
        for(int i=0;i<grid.length;i++){
            int count = 0;
            for(int j=grid[i].length-1;j>=0;j--){
                if(grid[i][j] == 1){
                    break ;
                }
                count ++;
            }
            list.add(count);
        }
        int n = grid.length;
        int ans = 0;
        for(int i=0;i<grid.length;i++){
            n --;
            if(list.get(i) >= n){
                continue ;
            }
            for(int j=i+1;j<grid.length;j++){
                if(list.get(j) >= n){
                    ans += j-i;
                    int val = list.get(j);
                    list.remove(j);
                    list.add(i, val);
                    break ;
                }
            }
            if(list.get(i) < n){
                return -1;
            }
        }
        return ans;
    }
}
