class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] zeroEndCounts = new int[n];
        
        for(int row = 0; row < n; row++){
            for(int col = n - 1; col >= 0; col--){
                if(grid[row][col] == 0){
                    zeroEndCounts[row]++;
                }
                else{
                    break;
                }
            }
        }
        int swaps = 0;
        for(int i = 0; i < n - 1; i++){
            int minCount = n - i - 1;
            
            
int j = i;
            for(; j < n; j++){
                if(zeroEndCounts[j] >= minCount){
                    break;
                }
            }
            if(j >= n){
                return -1;
            }
            for(; j > i; j--){
                int temp = zeroEndCounts[j];
                zeroEndCounts[j] = zeroEndCounts[j - 1];
                zeroEndCounts[j - 1] = temp;
                swaps++;
            }
        }
        return swaps;
    }
}
