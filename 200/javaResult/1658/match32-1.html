class Solution {
    public int minSwaps(int[][] grid) {
        int n=grid.length;
        int[] count=new int[n];
        
for(int i=0;i<n;i++){
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]==0){
                    count[i]++;
                }else{
                    break;
                }
            }
        }
        
        int res=0;
        for(int i=n-1;i>=1;i--){
            
int num=-1;
            for(int j=n-i-1;j<n;j++){
                if(count[j]>=i){
                    num=j;
                    break;
                }
            }
            if(num==-1) return -1;
            res+=num-(n-i-1);
            
for(int j=num;j>n-i-1;j--){
                int temp=count[j];
                count[j]=count[j-1];
                count[j-1]=temp;
            }
        }
        return res;
    }
}
