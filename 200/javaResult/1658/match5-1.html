class Solution {
    public int minSwaps(int[][] grid) {
        
int len = grid.length;
        int[] zeros = new int[len];
        for(int i = 0; i < len; i++) {
            int count = 0;
            for(int j = len - 1; j > -1; j--) {
                if(grid[i][j] == 0) {
                    count++;
                } else {
                    break;
                }
            }
            zeros[i] = count;
        }
        int res = 0;
        for(int i = 0; i < len; i++) {
            int zeroLen = len - i - 1;
            int j = i;
            for(; j < len; j++) {
                if(zeros[j] >= zeroLen) {
                    
break;
                }
            }
            if(j == len) {
                return -1;
            }
            int tmp = zeros[j];
            
for(int k = j; k > i; k--) {
                zeros[k] = zeros[k - 1];
                res++;
            }
            zeros[i] = tmp;
        }
        return res;
    }
}
