class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] c = new int[n];
        for (int i = 0; i < n; i++){
            for (int j = grid[0].length - 1; j >= 0; j--){
                if (grid[i][j] == 1) 
break;
                c[i]++;
            }
        }
        
        //for (int i = 0; i < n; i++)
        //    System.out.print(c[i] + " ");
        //System.out.println();
        
        int ans = 0;
        for (int i = 0; i < n; i++){
            if (c[i] >= n - 1 - i) continue;
            int k = c[i];
            int dist = 0;
            for (int j = i + 1; j < n; j++)
                if (c[j] >= n - 1 - i){
                    dist = j - i;
                    for (int p = j; p > i; p--)
                        swap(c, p, p - 1);
                    break;
                
}
            if (dist == 0) return -1;
            ans += dist;
        }
        
        return ans;
    }
    
    
    private void swap(int[] c, int i, int j){
        int tmp = c[i]; c[i] = c[j]; c[j] = tmp;
    }
}
