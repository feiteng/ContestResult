class Solution {
    public int minSwaps(int[][] grid) {
        int len = grid.length;
        int[] status = new int[len];
        for (int i = 0; i < len; i++) {
            for (int j = len - 1; j >= 0; j--) {
                if (grid[i][j] == 1) {
                    status[i] = j;
                    break;
                }
            }
        }
        int step = 0;
        for (int i = 0; i < len; i++) {
            boolean f = false;
            for (int j = i; j < len; j++) {
                if (j > i) {
                    int p = status[i];
                    status[i] = status[j];
                    status[j] = p;
                }
                if (status[i] < i + 1) {
                    step += j - i;
                    f = true;
                    break;
                }
            }
            if (!f) {
                return -1;
            }
        }
        return step;
    }
}
