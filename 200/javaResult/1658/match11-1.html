class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] c = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = n - 1; j >= -1; j--) {
                if (j == -1 || grid[i][j] == 1) {
                    c[i] = n - 1 - j;
                    break;
                
}
            }
        }
        // System.out.println(Arrays.toString(c));
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (c[i] >= n - 1 - i) continue;
            int prev = c[i];
            // System.out.println(Arrays.toString(c));
            int j = i + 1;
            for (; j < n; j++) {
                int t = c[j];
                c[j] = prev;
                prev = t;
                if (prev >= n - 1 - i) {
                    
c[i] = prev;
                    ans += j - i;
                    break;
                }
            }
            if (j == n) return -1;
        }
        return ans;
    }
}
