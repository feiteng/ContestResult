class Solution {
    public int minSwaps(int[][] grid) {
        int N = grid.length;
        int[] rowCnt = new int[N];
        
        for (int i = 0; i < N; i++) {
            for (int j = N - 1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    rowCnt[i]++;
                } else {
                    break;
                }
            }
        }
        
        int ret = 0;
        for (int i = 0; i < N; i++) {
            int cnt = N - 1 - i;
            if (rowCnt[i] >= cnt) {
                continue;
            }
            
            // find next j
            int j = i + 1;
            for (; j < N; j++) {
                if (rowCnt[j] >= cnt) {
                    break;
                }
            }
            
            if (j == N) {
                return -1;
            }
            
            // swap i and j
            for (int k = j - 1; k >= i; k--) {
                // swap k and j
                ret++;
                swap(rowCnt, k, k + 1);
            }
        }
        
        return ret;
    }
    
    void swap(int[] arr, int i, int j) {
        int t = arr[i];
        arr[i] = arr[j];
        arr[j] = t;
    }
    
}
