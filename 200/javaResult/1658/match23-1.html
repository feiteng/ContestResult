class Solution {
    public int minSwaps(int[][] grid) {
        int N = grid.length;
        int[] zeroCnt = new int[N];
        for (int i = 0; i < N; i++) {
            for (int j = N-1; j >= 0; j--) {
                if (grid[i][j] == 0) zeroCnt[i]++;
                else 
break;
            }
        }
        int res = 0;
        for (int i = 0; i < N; i++) {
            boolean found = false;
            for (int j = i; j < N; j++) {
                if (zeroCnt[j] >= N-i-1) {
                    for (int k = j; k > i; k--) swap(zeroCnt, k, k-1);
                    res += 
Math.abs(j-i);
                    found = true;
                    break;
                }
            }
            if (!found) return -1;
        }
        return res;
    }
    
    private void swap(int[] a, int i, int j) {
        int tmp = a[i];
        a[i] = a[j];
        a[j] = tmp;
    }
}
