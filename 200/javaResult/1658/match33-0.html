class Solution {
    public int minSwaps(int[][] grid) {
        
        int N = grid.length;
        
        List<Integer> list = new LinkedList<>();
        
        for (int i = 0; i < N; ++i) {
            int count = 0;
            for (int j = N - 1; j >= 0; --j) {
                if (grid[i][j] == 0) count++;
                else break;
            }
            list.add(count);
        }
        
        int res = 0;
        int moves = 0;
        
        for (int i = 0; i < N - 1; ++i) {
            int needed = (N - i - 1);
            boolean noFound = true;
            
            for (int j = i; j < list.size(); ++j) {
                
                int value = list.get(j);
                if (value >= needed) {
                    noFound = false;
                    res += (j - i);
                    
                    list.remove(j);
                    list.add(i, value);
                    break;
                }
            }
            
            if (noFound) return -1;
        }
        
        return res;
    }
}
