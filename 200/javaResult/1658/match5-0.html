class Solution {
    public int minSwaps(int[][] grid) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        //PriorityQueue<int[]> tmp = new PriorityQueue<>((a, b) -> b[0] - a[0] == 0 ? a[1] - b[1] : b[0] - a[0]);
        int sum = 0;
        int size = grid.length;
        int[] arr = new int[size];
        for (int i = 0; i < size; i++) {
            int count = 0;
            for (int j = size - 1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    count++;
                } else {
                    break;
                }
            }
            arr[i] = count;
        }
        for (int i = 0; i < size - 1; i++) {
            int limit = size - i - 1;
            int j = i;
            int target = -1;
            for (; j < size; j++) {
                if (arr[j] >= limit) {
                    sum += (j - i);
                    target = arr[j];
                    break;
                }
            }
            if (j == size) {
                return -1;
            }
            for (int k = j - 1; k >= i; k--) {
                arr[k + 1] = arr[k];
            }
            arr[i] = target;
        }
        return sum;
        
        
    }
}
