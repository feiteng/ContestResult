class Solution {
    public int minSwaps(int[][] grid) {
        ArrayList<Integer> array = new ArrayList<Integer>();
        for(int i = 0;i<grid.length;i++){
            int count = 0;
            for(int j = grid.length-1;j>=0;j--){
                if (grid[i][j] == 0)
                    count++;
                else
                    break;
            }
            array.add(count);
        }
        int swaps = 0;
        for(int i=0;i<grid.length-1;i++){
            boolean found = false;
            for(int j = 0;j<array.size();j++){
                if (array.get(j) >= grid.length-i-1){
                    found = true;
                    swaps+=j;
                    array.remove(j);
                    break;
                }
            }
            if (!found)
                return -1;
        }
        return swaps;
    }
}
