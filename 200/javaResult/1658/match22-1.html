class Solution {
    public int minSwaps(int[][] grid) {
        int result = 0;
        
int n = grid.length;
        boolean[] used = new boolean[n];
        int[] counts = new int[n];
        for (int i = 0; i < n; i++) {
            int thisCount = 0;
            for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    thisCount++;
                } else {
                    break;
                }
            }
            counts[i] = thisCount;
        }
        
        
for (int i = n - 1; i >= 1; i--) {
            int steps = 0;
            boolean found = false;
            for (int index = 0; index < n; index++) {
                if (used[index]) {
                    continue;
                }
                if (counts[index] >= i) {
                    used[index] = true;
                    found = true;
                    break;
                } else {
                    steps++;
                
}
            }
            if (!found) {
                return -1;
            }
            result += steps;
        }
        return result;
    }
}
