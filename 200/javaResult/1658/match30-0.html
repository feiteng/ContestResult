class Solution {
    public int minSwaps(int[][] grid) {
        int length=grid.length;
        int[] size=new int[length];
        for(int i=0;i<grid.length;i++){
            int len=grid[i].length;
            for(int j=len-1;j>=0;j--){
                if(grid[i][j]==0){
                    size[i]++;
                }
                else
                    break;
            }
        }
        int sum=0;
        for(int i=0;i<length;i++){
            boolean suc = false;
            for(int j=i;j<length;j++){
                if(size[j]>=length-i-1){
                    sum=sum+j-i;
                    size=ex(i,j,size);
                    suc = true;
                    break;
                }
            }
            if (!suc) return -1;
        }
        return sum;
    }
    public int[] ex(int i,int j,int[] size){
       for(int k=j;k>i;k--){
                        int t=size[k];
                        size[k]=size[k-1];
                        size[k-1]=t;
                    }
        return size;
    }
}
