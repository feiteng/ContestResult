class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        List<Integer> counts = new ArrayList<>();
        for(int i = 0; i < n; ++i) {
            int count = 0;
            for(int j = n - 1; j >= 0; --j) {
                if (grid[i][j] == 0) {
                    ++count;
                } else {
                    break;
                }
            }
            counts.add(count);
        }
        
        int result = 0;
        for(int i = n - 1; i > 0; --i) {
            boolean found = false;
            for(int j = 0, k = 
counts.size(); j < k; ++j) {
                if (counts.get(j) >= i) {
                    result += j;
                    counts.remove(j);
                    found = true;
                    break;
                }
            }
            if (!found) {
                return -1;
            }
        }
        return result;
    }
}
