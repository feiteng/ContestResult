class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] suffix = new int[n];
        for(int i=0;i<n;i++) {
            int num = 0;
            for(int j=n-1;j>=0;j--) {
                if(grid[i][j] == 1) break;
                num++;
            }
            suffix[i] = num;
            //System.out.println("i = " + i + ", suffix[i] = " + suffix[i]);
        }
        int num = n-1;
        int result = 0;
        for(int i=0;i<n;i++) {
            int row = -1;
            for(int j=i;j<n;j++) {
                if(suffix[j] >= num) {row = j; break;}
            }
            if(row == -1) return -1;
            int current = suffix[row];
            for(int j=row-1;j>=i;j--) {
                suffix[j+1] = suffix[j];
            }
            result += row-i;
            suffix[i] = current;
            num--;
        }
        return result;
    }
}
