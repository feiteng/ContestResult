class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        
        int zero[][] = new int[n][];
        // int ind[][] = new int[n][];
        
        for(int i=0;i<n;i++){
            zero[i] = new int[]{i,0};
            // ind[i] = zerro[i];
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]==0)
                    zero[i][1]++;
                else
                    break;
            }
        }
        
        // Arrays.sort(zero,(a,b)->(b[1]==a[1])?a[0]-b[0]:b[1]-a[1]);
        
        int ans=0;
        
        int rec = n-1;
        
        outer:
        for(int i=0;i<n;i++){
            
            for(int j=i;j<n;j++){
                if(zero[j][1]>=rec){
                    var t = zero[j];
                    for(int k=j;k>i;k--){
                        ans++;
                        zero[k] = zero[k-1];
                    }
                    zero[i] = t;
                    rec--;
                    continue outer;
                }
            }
            return -1;
        }
        return ans;
    }
}
