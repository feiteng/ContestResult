class Solution {
    public int minSwaps(int[][] grid) {
        int N = grid.length;
        int cnt[] = new int[N];
        for(int i = 0; i < N; i++){
            int c = 0;
            
for(int j = N - 1; j >= 0; j--){
                if(grid[i][j] == 0){
                    c++;
                }else{
                    break;
                }
            }
            cnt[i] = c;
        }
        
        int result = 0;
        
        for(int i = 0; i < N; i++){
            
            int require = N - 1 - i;
            int swapPos = -1;
            for(int j = i; j < N; j++){
                if(cnt[j] >= require){
                    result += (j - i);
                    swapPos = j;
                    
break;
                }
            }
            if(swapPos == -1){
                return -1;
            }
            int tmp = cnt[swapPos];
            for(int j = swapPos; j > i; j--){
                cnt[j] = cnt[j - 1];
            }
            cnt[i] = tmp;
        }
        
        return result;

    }
}
