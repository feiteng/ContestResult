class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        
List<Integer> rows = new LinkedList<>();
        for (int[] ints : grid) {
            int zero = 0;
            for (int j = n - 1; j >= 0; j--) {
                if (ints[j] == 0) {
                    zero++;
                } else
                    
break;
            }
            rows.add(zero);
        }
        int target = n - 1;
        int steps = 0;
        while (target > 0) {
            boolean out = true;
            for (int i = 0; i < rows.size(); i++) {
                if (rows.get(i) >= target) {
                    steps+=i;
                    target--;
                    rows.remove(i);
                    out = false;
                    break;
                }
            }
            if (out)
                return -1;
        }
        return steps;

        // 0: >= n-1
        // 1: >= n-2


    }
}
