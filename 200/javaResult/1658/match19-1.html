class Solution {
 public int minSwaps(int[][] grid) {
        int n = grid.length;
        if (n == 1) return 0;
        int[] nums = new int[n];

        for (int i = 0; i < n; i++) {
            int count = 0;
            
for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    count++;
                } else {
                    break;
                }
            }
            nums[i] = count;
        }

        int ans = 0;
        for (
int k = n - 1, start = 0; k >= 1; k--, start++) {
            int index = -1;
            for (int i = start; i < n; i++) {
                if (nums[i] >= k) {
                    index = i;
                    break;
                }
            }
            if (index < 0) return -1;
            ans += index - start;
            
            for (int i = index; i > start; i--) {
                int t = nums[i];
                nums[i] = nums[i - 1];
                nums[i - 1] = t;
            }
        }
        return ans;
    }

}
