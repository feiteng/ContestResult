class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        if (n == 1) {
            return 0;
        }

        
int res = 0;
        for (int i = 0; i < n - 1; i++) {
            // [i + 1, n - 1] all zero
            int first = -1;
            for (int j = i; j < n; j++) {
                boolean valid = true;
                for (int k = i + 1; k < n; k++) {
                    if (grid[j][k] != 0) {
                        valid = false;
                    
}
                }
                if (valid) {
                    first = j;
                    break;
                }
            }

            if (first == -1) {
                return -1;
            }
            for (int j = first; j > i; j--) {
                swap(grid, j, j - 1);
            }
            res += first - i;
        }
        return res;
    }

    void swap(int[][] grid, int i, int j) {
        int n = grid.length;
        
for (int k = 0; k < n; k++) {
            int tmp = grid[i][k];
            grid[i][k] = grid[j][k];
            grid[j][k] = tmp;
        }
    }
}
