class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        
        int[] rowRep = new int[n];
        
        for (int i = 0; i < n; i++) {
            for (int j = n - 1; j >= 0 && grid[i][j] == 0; j--) {
                rowRep[i]++;
            }
        }
        
        int res = 0;
        for (int i = 0; i < n - 1; i++) {
            int rowReq = n - i - 1;
            if (rowRep[i] >= rowReq) continue;
            
            
int swapRow = -1;
            
            for (int j = i + 1; j < n; j++) {
                if (rowRep[j] >= rowReq) {
                    swapRow = j;
                    break;
                }
            }
            
            if (swapRow == -1) return -1;
            
            for (int j = swapRow - 1; j >= i; j--) {
                int temp = rowRep[j];
                rowRep[j] = rowRep[j + 1];
                rowRep[j + 1] = temp;
                res++;
            }
        }
        return res;
        
        
    }
}
