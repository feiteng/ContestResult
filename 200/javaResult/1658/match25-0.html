class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] ct = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 1) {
                    ct[i] = n - 1 - j;
                    break;
                }
                if (j == 0) ct[i] = n;
            }
            //System.out.println(ct[i]);
        }
        int sol = 0;
        for (int i = 0; i < n; i++) {
            boolean found = false;
            for (int j = i; j < n; j++) {
                if (ct[j] >= n - i - 1) {
                    int cur = ct[j];
                    for (int k = j; k > i; k--) {
                        ct[k] = ct[k-1];
                    }
                    ct[i] = cur;
                    sol += j - i;
                    found = true;
                    break;
                }
            }
            if (!found) return -1;
        }
        return sol;
    }
}
