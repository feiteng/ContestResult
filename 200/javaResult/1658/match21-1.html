class Solution {
    public int minSwaps(int[][] grid) {
        int len = grid.length;
        int[] support = new int[len];
        for (int i = 0; i < len; ++i) {
            for (int j = len - 1; j >= 0; --j) {
                if (grid[i][j] == 1) {
                    support[i] = j;
                    break;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < len; ++i) {
            int count = 0;
            boolean isSwap = false;
            for (int j = 0; j < len; ++j) {
                if (support[j] != Integer.MAX_VALUE) {
                    if (support[j] <= i) {
                        support[j] = Integer.MAX_VALUE;
                        res += count;
                        isSwap = true;
                        break;
                    } else {
                        ++count;
                    }
                
}
            }
            if (!isSwap) {
                return -1;
            }
        }
        return res;
    }
}
