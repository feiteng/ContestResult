class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        List<Integer> zeroLens = new ArrayList<Integer>();
        for (int i = 0; i < n; i++) {
            int j = n - 1;
            while (j >= 0 && grid[i][j] == 0) j--;
            zeroLens.add(n - 1 - j);
        }
        
        
int swaps = 0;
        for (int i = 0; i < n; i++) {
            int neededLen = n - 1 - i;
            boolean found = false;
            for (int j = 0; j < zeroLens.size(); j++) {
                if (zeroLens.get(j) >= neededLen) {
                    swaps += j;
                    zeroLens.remove(j);
                    found = true;
                    
break;
                }
            }
            if (!found) return -1;
        }
        
        return swaps;
    }
}
