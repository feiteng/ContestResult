class Solution {
   public int minSwaps(int[][] grid) {
            int n = grid.length;
            int[] leftZeroNum = new int[n];
            for (int i = 0; i < n; ++i) {
                leftZeroNum[i] = 0;
                for (int j = n - 1; j >= 0; --j) {
                    if (grid[i][j] == 1) {
                        break;
                    }
                    leftZeroNum[i]++;
                }
            }
            int count = 0;
            for (int i = 0; i < n; ++i) {
                boolean find = false;
                for (int j = i; j < n; ++j) {
                    if (leftZeroNum[j] >= n - i - 1) {
                        count += moveJToI(j, i, leftZeroNum);
                        find = true;
                        break;
                    }
                }
                if (!find) {
                    return -1;
                }
            }
            return count;

        }

        private int moveJToI(int j, int i, int[] a) {
            for (int now = j; now > i; --now) {
                int tmp = a[now];
                a[now] = a[now - 1];
                a[now - 1] = tmp;
            }
            return j - i;
        }
}
