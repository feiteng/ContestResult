class Solution {
    public int minSwaps(int[][] grid) {
        int[] n = new int[grid.length];
        for(int i=0;i<n.length;i++){
            int temp=0;
            
for(int j=grid[i].length-1;j>-1;j--){
                if(grid[i][j]==0) temp++;
                else break;
            }
            n[i] = temp;
        
}
        int m = n.length;
        int res=0;
        // System.out.println(Arrays.toString(n));
        t1:for(int i=0;i<m;i++){
            for(int j=i;j<m;j++){
                if(n[j] >= m-i-1){
                    res += j-i;
                    int temp = n[j];
                    for(int k=j;k>i;k--)
                        n[k] = n[k-1]
;
                    n[i] = temp;
        //             System.out.println(i+"  "+j+"  "+res);
        // System.out.println(Arrays.toString(n));
                    continue t1;
                }
            }
            return -1;
        }
        return res;
    }
}
