class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        List<Integer> l = new LinkedList<>();
        for(int i = 0 ; i < n ; i++) {
            int c = 0;
            for(int j = n-1 ; j >= 0 ; j--){
                if(grid[i][j] == 0){
                    c++;
                }else{
                    break;
                }
            }
            l.add(c);
        }
        int ret = 0;
        for(int i=0; i<n; i++){
            boolean find = false;
            for(int j = 0; j < l.size(); j++){
                if(l.get(j) >= n - 1 - i){
                    ret += j;
                    l.remove(j);
                    find = true;
                    break;
                }
            }
            if(!find){
                return -1;
            }
        }
        return ret;
    }
}
