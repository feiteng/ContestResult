class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;
        int[] lo = new int[n];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j]==1)
                    lo[i]=j;
            }
        }
        boolean[] vis = new boolean[n];
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int vc = 0;
            for (int j = 0; j < n; ++j) {
                if (vis[j])
                    continue;
                if (lo[j]<=i) {
                    ans += vc;
                    vis[j] = true;
                    vc = 0;
                    break;
                }
                ++vc;
            }
            //System.out.println(i+" "+vc);
            if (vc>0)
                return -1;
        }
        return ans;
    }
}
