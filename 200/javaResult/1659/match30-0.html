class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        //是不是可以直接贪心呀，试试
        int idx1 = 0;
        int idx2 = 0;
        long sum1 = 0;
        long sum2 = 0;
        long maxSum = 0;
        while (idx1 < nums1.length || idx2 < nums2.length) {
            if (idx1 == nums1.length) {
                sum2 += nums2[idx2];
                idx2++;
            } else if (idx2 == nums2.length) {
                sum1 += nums1[idx1];
                idx1++;
            } else {
                if (nums1[idx1] == nums2[idx2]) {
                    sum1 += nums1[idx1];
                    sum2 += nums2[idx2];
                    maxSum = (maxSum + Math.max(sum1, sum2)) % 1000000007;
                    idx1++;
                    idx2++;
                    sum1 = 0;
                    sum2 = 0;
                } else if (nums1[idx1] < nums2[idx2]) {
                    sum1 += nums1[idx1];
                    idx1++;
                } else {
                    sum2 += nums2[idx2];
                    idx2++;
                }
            }
        }
        return (int)((maxSum + Math.max(sum1, sum2)) % 1000000007);
    }
}
