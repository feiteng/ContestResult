class Solution {
    private int DIV = 1000000007;
    public int maxSum(int[] nums1, int[] nums2) {
        long[] dp1 = new long[nums1.length + 1];
        long[] dp2 = new long[nums2.length + 1];
        int p1 = 0;
        int p2 = 0;
        dp1[0] = 0;
        dp2[0] = 0;
        while (p1 < nums1.length && p2 < nums2.length) {
            if (nums1[p1] < nums2[p2]) {
                dp1[p1 + 1] = (dp1[p1] + nums1[p1]);
                p1++;
            } else if (nums1[p1] > nums2[p2]){
                dp2[p2 + 1] = (dp2[p2] + nums2[p2]);
                p2++;
            } else {
                dp1[p1 + 1] = (Math.max((dp1[p1] + nums1[p1]), (dp2[p2] + nums2[p2])));
                dp2[p2 + 1] = (Math.max(dp1[p1] + nums1[p1], dp2[p2] + nums2[p2]));
                p1++;
                p2++;
            }
        }
        while (p1 < nums1.length) {
            dp1[p1 + 1] = (dp1[p1] + nums1[p1]);
            p1++;
        }
        while (p2 < nums2.length) {
            dp2[p2 + 1] = (dp2[p2] + nums2[p2]);
            p2++;
        }
        return (int)(Math.max(dp1[nums1.length], dp2[nums2.length]) % DIV);
    }
}
