class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        long[] map1 = new long[nums1.length + 1];
        long[] map2 = new long[nums2.length + 1];
        int cur1 = nums1.length - 1;
        int cur2 = nums2.length - 1;
        while (cur1 > -1 && cur2 > -1) {
            if (nums1[cur1] > nums2[cur2]) {
                map1[cur1] = map1[cur1 + 1] + nums1[cur1];
                cur1--;
            } else if (nums1[cur1] < nums2[cur2]) {
                map2[cur2] = map2[cur2 + 1] + nums2[cur2];
                cur2--;
            } else {
                map1[cur1] = map2[cur2] = Math.max(map1[cur1 + 1], map2[cur2 + 1]) + nums1[cur1];
                cur1--;
                cur2--;
            
}
        }
        while (cur1 != -1) {
            map1[cur1] = map1[cur1 + 1] + nums1[cur1];
            cur1--;
        }
        while (cur2 != -1) {
            map2[cur2] = map2[cur2 + 1] + nums2[cur2];
            cur2--;
        }
        return (int) (Math.max(map1[0],map2[0]) % 1000000007);
    }
}
