class Solution {
    int M = (int) 1e9 + 7;
    
public int maxSum(int[] n1, int[] n2) {
        long res = 0;
        int i = 0, j = 0;
        long pre1 = 0, pre2 = 0;
        while (i < n1.length && j < n2.length) {
            int a = n1[i], b = n2[j];
            
if (a == b) {
                res += Math.max(pre1, pre2) + a;
                res %= M;
                i++;
                j++;
                pre1 = 0; pre2 = 0;
            } else if (a < b) {
                pre1 += a;
                i++;
            } else {
                pre2 += b;
                j++;
            }
        }
        while (i < n1.length) {
            pre1 += n1[i++];
        }
        while (j < n2.length) {
            pre2 += n2[j++];
        }
        return (int) ((res + Math.max(pre1, pre2)) % M);
    }
}
