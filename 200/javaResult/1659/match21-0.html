class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        int MOD = 1000000007;
        long suma = 0; // sums between 2 common points in both arr, inclusive of last common point
        long sumb = 0;
        // int points[] = new int[l];
        int l1 = nums1.length;
        int l2 = nums2.length;
        int i=0;
        int j=0;
        long ans = 0;
        while(i<l1 && j< l2) {
           // System.out.println(i + " " + j + " " + nums1[i] + " " + nums2[j] + " " + ans);
            if(nums1[i] == nums2[j]) {
                suma += nums1[i];
                sumb += nums2[j];
                ans += Math.max(suma, sumb);
                i++;
                j++;
                suma = 0;
                sumb = 0;
                ans = ans % MOD;
            } else if(nums1[i] < nums2[j]){
                suma += nums1[i];
                i++;
            } else {
                sumb += nums2[j];
                j++;
            } 
        }
        while(i<l1) {
            suma += nums1[i];
            i++;
        }
        while(j < l2) {
            sumb += nums2[j];
            j++;
        }
        ans += Math.max(suma, sumb);
        
        return (int)(ans %MOD);
        
    }
}
