class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        
        long count = 0;
        int i=0,j=0;
        long a = 0;
        long b = 0;

        while (i<nums1.length&&j<nums2.length){
            if (nums1[i]==nums2[j]){
                count += Math.max(a,b)+nums1[i];
                i++;
                j++;
                a = b = 0;
            }else
            if (nums1[i]<nums2[j]){
                a += nums1[i++];
            } else {
                b += nums2[j++];
            }
        }
        while (i<nums1.length){
            a += nums1[i++];
        }
        while (j<nums2.length){
            b += nums2[j++];
        }
        count+= Math.max(a,b);
        long max = 1000_000_007;
        while (count>=max){
            count-=max;
        }
        return (int) count;
    }
}
