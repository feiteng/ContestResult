class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        long sum1 = 0;
        long sum2 = 0;
        
        long totalSum = 0;
        
        int i = 0;
        int j = 0;
        
        while( i < nums1.length && j < nums2.length){
            if(nums1[i] == nums2[j]){
                totalSum += Math.max(sum1, sum2);
                sum1 = nums1[i];
                sum2 = nums2[j];
                i++;
                j++;
            }else if(nums1[i] > nums2[j]){
                sum2 += nums2[j];
                j++;
            }else{
                sum1 += nums1[i];
                i++;
            }
        }
        
        while(i < nums1.length){
            sum1 += nums1[i];
            i++;
        }
        
        while(j < nums2.length){
            sum2 += nums2[j];
            j++;
        }
        
        totalSum += Math.max(sum1, sum2);
        
        long modulo = 1000000007;
        
        return (int)(totalSum%modulo);
    }
}
