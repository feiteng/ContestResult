class Solution {
    final static int MOD = (int)1e9 + 7; 
    public int maxSum(int[] nums1, int[] nums2) {
        int N1 = nums1.length, N2 = nums2.length;
        int i1 = 0, i2 = 0;
        
long max1 = 0, max2 = 0;
        
        while(i1 < N1 && i2 < N2) {
            if(nums1[i1] < nums2[i2]) {
                max1 = max1 + nums1[i1];
                ++i1;
            } else if(nums1[i1] > nums2[i2]) {
                max2 = max2 + nums2[i2];
                ++i2;
            } else {
                long thisMax = Math.max(max2, max1);
                max1 = max2 = thisMax + nums2[i2];
                ++i1; ++i2;
            }
        }
        
        while(i1 < N1) {
            max1 = max1 + nums1[i1];
            ++i1;
        }
        while(i2 < N2) {
            max2 = max2 + nums2[i2];
            ++i2;
        }
        
        return (int)(Math.max(max1, max2) % MOD);
        
    }
}
