// 10^5 * 10^7 = 10^12 < 2^63 - 1
class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        
int mod = (int)1e9 + 7;
        long dp1 = 0;
        long dp2 = 0;
        int idx1 = nums1.length - 1;
        int idx2 = nums2.length - 1;
        while (idx1 >= 0 && idx2 >= 0) {
            if (nums1[idx1] > nums2[idx2]) {
                dp1 += nums1[idx1--];
            } else if (nums1[idx1] < nums2[idx2]) {
                dp2 += nums2[idx2--];
            } else {
                dp1 = Math.max(dp1, dp2) + nums1[idx1];
                
dp2 = dp1;
                idx1--;
                idx2--;
            }
        }
        while (idx1 >= 0) {
            dp1 += nums1[idx1--];
        }
        while (idx2 >= 0) {
            dp2 += nums2[idx2--];
        }
        return Math.max((int)(dp1 % mod), (int)(dp2 % mod));
    }
}
