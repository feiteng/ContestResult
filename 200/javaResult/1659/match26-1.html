class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        
long sum = 0;
        int i = 0;
        int j = 0;
        long up = 0;
        long down = 0;
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] < nums2[j]) {
                up += nums1[i++];
            }
            else if (nums1[i] > nums2[j]) {
                down += nums2[j++];
            }
            else {
                up += nums1[i++];
                down += nums2[j++];
                sum += Math.max(up, down);
                up = 0;
                down = 0;
            }
        }
        while (i < nums1.length) up += nums1[i++];
        while (j < nums2.length) down += nums2[j++];
        sum += Math.max(up, down);
        return (int)(sum % 1000000007);
    }
}
