class Solution {
    public int maxSum(int[] nums1, int[] nums2) {


        int MOD = 1_000_000_007;
        long dp1 = nums1[0], dp2 = nums2[0];
        int p1 = 0, p2 = 0;

        while (true) {


            int n1 = nums1[p1], n2 = nums2[p2];


            //System.out.printf("p1=%d dp1=%d\n", p1, dp1);
            //System.out.printf("p2=%d dp2=%d\n", p2, dp2);


            if (n1 == n2) {

                dp1 = Math.max(dp1, dp2);
                dp2 = Math.max(dp1, dp2);
                if (p2 == nums2.length - 1 && p1 == nums1.length - 1) {
                    break;

                } else if (p1 == nums1.length - 1) {
                    p2++;
                    dp2 += nums2[p2];
                } else if (p2 == nums2.length - 1) {
                    p1++;
                    dp1 += nums1[p1];

                
} else if (nums1[p1 + 1] <= nums2[p2 + 1]) {
                    p1++;
                    dp1 += nums1[p1];
                } else {
                    p2++;
                    dp2 += nums2[p2];
                }
            } else if (p2 == nums2.length - 1 && p1 == nums1.length - 1) {
                break;

            } else if (p1 == nums1.length - 1) {
                p2++;
                dp2 += nums2[p2];
            } else 
if (p2 == nums2.length - 1) {
                p1++;
                dp1 += nums1[p1];

            } else if (n1 < n2) {
                p1++;
                dp1 += nums1[p1];
            } else {
                p2++;
                dp2 += nums2[p2];
            
}


        }

        return (int)(Math.max(dp1, dp2)%MOD);


    }
}
