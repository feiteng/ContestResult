class Solution {
    private final static int MODULE = 1000000007;

    
public int maxSum(int[] nums1, int[] nums2) {
        long sum1 = 0;
        long sum2 = 0;
        long sum = 0;
        int idx1 = 0;
        int idx2 = 0;
        while (idx1 < nums1.length && idx2 < nums2.length) {
            if (nums1[idx1] == nums2[idx2]) {
                sum += Math.max(sum1, sum2) + nums1[idx1];
                idx1++;
                idx2++;
                sum1 = 0;
                sum2 = 0;
            } else if (nums1[idx1] < nums2[idx2]) {
                sum1 += nums1[idx1];
                idx1++;
            } else {
                sum2 += nums2[idx2];
                idx2++;
            }
        }
        while (idx1 < nums1.length) {
            sum1 += nums1[idx1];
            idx1++;
        }
        while (idx2 < nums2.length) {
            sum2 += nums2[idx2];
            idx2++;
        }
        return (int)((sum + Math.max(sum1, sum2))%MODULE);
    }
}
