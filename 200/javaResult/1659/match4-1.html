class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        int i = 0;
        
int j = 0;
        long res = 0;
        long sum1 = 0;
        long sum2 = 0;
        int mod = 1000000007;

        while (i<n1 && j<n2) {
            if (nums1[i]==nums2[j]) {
                res += Math.max(sum1,sum2);
                sum1 = 0;
                sum2 = 0;
                res += nums1[i];
                i++;
                j++;
            } else if (nums1[i]>nums2[j]) {
                sum2 += nums2[j];
                j++;
            } else {
                sum1 += nums1[i];
                i++;
            }
        }

        while (i<n1) {
            sum1 += nums1[i];
            i++;
        }

        while (j<n2) {
            sum2 += nums2[j];
            j++;
        }
        res += Math.max(sum1,sum2);

        res %= mod;

        return (int)res;
    }
}
