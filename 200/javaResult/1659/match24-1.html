class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        long p1 = 0l;
        long p2 = 0l;
        int ind1 = 0;
        int ind2 = 0;
        int mod = (int)1e9+7;
        while(ind1 < nums1.length && ind2 < nums2.length) {
            if(nums1[ind1] < nums2[ind2]) {
                p1+=nums1[ind1];
                ind1++;
            } else if(nums1[ind1] > nums2[ind2]) {
                p2+=nums2[ind2];
                ind2++;
            } else {
                long l = 
Math.max(p1,p2) + nums1[ind1];
                p1 = l%mod;
                p2 = p1;
                ind1++;
                ind2++;
            }
        }
        while(ind1 < nums1.length) {
            p1+=nums1[ind1];
            ind1++;
        }
        while(ind2 < nums2.length) {
            p2+=nums2[ind2];
            ind2++;
        }
        long l = Math.max(p1,p2);
        return (int)(l%mod);
    }
}
