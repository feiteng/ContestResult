class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        long[] dp1 = new long[nums1.length+1];
        long[] dp2 = new long[nums2.length+1];
        //best path in arri starting from this idx
        int c1 = nums1.length-1;
        int c2 = nums2.length-1;
        
        while (c1 >= 0 && c2 >= 0) {
            if (nums1[c1] > nums2[c2]) {
                dp1[c1] = dp1[c1+1] + nums1[c1];
                c1--;
            } else if (nums2[c2] > nums1[c1]) {
                dp2[c2] = dp2[c2+1] + nums2[c2];
                c2--;
            } else {
                long bestPath = nums1[c1] + Math.max(dp1[c1+1],dp2[c2+1]);
                dp1[c1] = bestPath;
                dp2[c2] = bestPath;
                c1--;
                c2--;
            }
        }
        while (c1 >= 0) {
            dp1[c1] = dp1[c1+1] + nums1[c1];
            c1--;
        }
        while (c2 >= 0) {
            dp2[c2] = dp2[c2+1] + nums2[c2];
            c2--;
        }
        return (int) (Math.max(dp1[0], dp2[0]) % 1000000007);
    }
}
