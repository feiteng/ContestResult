class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        long[] dp1 = new long[nums1.length + 1];
        long[] dp2 = new long[nums2.length + 1];
        int p1 = nums1.length - 1, p2 = nums2.length - 1;
        while (p1 >= 0 && p2 >= 0) {
            if (nums1[p1] == nums2[p2]) {
                long tmp = Math.max(dp1[p1 + 1], dp2[p2 + 1]);
                dp1[p1] = 
dp2[p2] = tmp + nums1[p1];
                p1--;
                p2--;
            } else if (nums1[p1] < nums2[p2]) {
                dp2[p2] = dp2[p2 + 1] + nums2[p2];
                p2--;
            } else {
                
dp1[p1] = dp1[p1 + 1] + nums1[p1];
                p1--;
            }
        }
        while (p1 >= 0) {
            dp1[p1] = dp1[p1 + 1] + nums1[p1];
            p1--;
        }

        while (p2 >= 0) {
            dp2[p2] = dp2[p2 + 1] + nums2[p2];
            p2--;
        }
        int a = (int)(dp1[0] % 1000000007);
        int b = (int)(dp2[0] % 1000000007);
        return Math.max(a, b);
    }
}
