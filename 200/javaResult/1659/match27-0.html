class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        long maxDp1 = 0;
        long maxDp2 = 0;
        
        int i = 0, j = 0;
        while (i < len1 && j < len2) {
            if (nums1[i] < nums2[j]) {
                maxDp1 += nums1[i];
                i++;
            } else if (nums1[i] > nums2[j]) {
                maxDp2 += nums2[j];
                j++;
            } else {
                long curMax = Math.max(maxDp1, maxDp2) + nums1[i];
                maxDp1 = curMax;
                maxDp2 = curMax;
                i++; j++;
            }
        }
        
        while (i < len1) {
            maxDp1 += nums1[i++];
        }
        
        while (j < len2) {
            maxDp2 += nums2[j++];
        }
        
        return (int) (Math.max(maxDp1, maxDp2) % 1_000_000_007);
    }
}
