class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        
int i1 = 0;
        int i2 = 0;
        long sum1 = 0;
        long sum2 = 0;
        long res = 0;
        // 1 3 5
        // 2 4 6
        while (i1 < nums1.length && i2 < nums2.length) {
            if (nums1[i1] < nums2[i2]) {
                sum1 += (long)nums1[i1];
                i1++;
            } else if (nums1[i1] > nums2[i2]) {
                sum2 += (long)nums2[i2];
                i2++;
            } else {
                res += (Math.max(sum1, sum2) + nums1[i1]) % 1000000007;
                sum1 = 0l;
                sum2 = 0l;
                i1++;
                i2++;
            }
        }
        while (i1 < nums1.length) {
            sum1 += nums1[i1];
            i1++;
        }
        while (i2 < nums2.length) {
            sum2 += nums2[i2];
            i2++;
        }
        res += Math.max(sum1, sum2) % 1000000007;
        res %= 1000000007;
        return (int)res;
    }
}
