class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        int len1=nums1.length;
        int len2=nums2.length;
        long sum=0;
        long sum1=0,sum2=0;
        int i=0,j=0;
        while(i<len1 && j<len2){
            if(nums1[i]<nums2[j]){                                
                sum1+=nums1[i];
                i++;
            }
            else if(nums1[i]>nums2[j]){                
                sum2+=nums2[j];
                j++;
            }
            else{
                sum+=Math.max(sum1,sum2)+nums1[i];
                sum1=0;
                sum2=0;
                i++;
                j++;
            }
        }
        while(i<len1){
            sum1+=nums1[i];
            i++;
        }
        while(j<len2){
            sum2+=nums2[j];
            j++;
        }
        sum+=Math.max(sum1,sum2);
        return (int)(sum%1000000007);
    }
}
