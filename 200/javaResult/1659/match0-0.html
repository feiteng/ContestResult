class Solution {
    public int maxSum(int[] nums1, int[] nums2) {
        int pos1 = 0;
        int pos2 = 0;
        int n1 = nums1.length;
        int n2 = nums2.length;
        long pre1 = 0;
        long pre2 = 0;
        long ans = 0;
        while(pos1 < n1 && pos2 < n2){
            if(nums1[pos1] < nums2[pos2]){
                pre1 += nums1[pos1];
                pos1++;
            }else if(nums1[pos1] > nums2[pos2]){
                pre2 += nums2[pos2];
                pos2++;
            }else{
                ans += Math.max(pre1, pre2);
                pre1 = nums1[pos1];
                pre2 = nums2[pos2];
                pos1++;
                pos2++;
            }
        }
        while(pos1 < n1){
            pre1 += nums1[pos1];
            pos1++;
        }
        while(pos2 < n2){
            pre2 += nums2[pos2];
            pos2++;
        }
        ans += Math.max(pre1, pre2);
        return (int)(ans % 1000000007);
    }
}
