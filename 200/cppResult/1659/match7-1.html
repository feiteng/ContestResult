class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        
        int i=0,j=0;
        long long sum1=0,sum2=0;

        long long answer=0;
        int l1=nums1.size();
        int l2=nums2.size();

        while(i<l1 && j<l2){
            if(nums1[i]<nums2[j]){
                sum1+=nums1[i];
                i++;
            }
            else if(nums1[i]>nums2[j]){
                sum2+=nums2[j];
                j++;
            }
            else{
                answer+=max(sum1,sum2);
                sum1=0;
                sum2=0;
                while(i<l1 && j<l2 && nums1[i]==nums2[j]){
                    answer+=nums1[i];
                    i++;
                    j++;
                }
            }
        }
        while(i<l1){
            sum1+=nums1[i];
            i++;
        }
        while(j<l2){
            sum2+=nums2[j];
            j++;
        }

        answer+=max(sum1,sum2);
        return (int)(answer%1000000007);
    }
};
