class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int n=nums1.size(),mod=1e9+7;
        int m=nums2.size();
        long long int dp[n+1];
        long long int dp1[m+1];
        dp[n]=0;
        dp1[m]=0;
        int i=n-1,j=m-1;
        while(i>=0||j>=0){
            if(i>=0&&(j<0||nums1[i]>nums2[j])){
                dp[i]=dp[i+1]+nums1[i];
                i--;
            }
            else if(j>=0&&(i<0||nums1[i]<nums2[j])){
                dp1[j]=dp1[j+1]+nums2[j];
                j--;
            }
            else{
                dp[i]=nums2[j]+max(dp[i+1],dp1[j+1]);
                dp1[j]=dp[i];
                i--;
                j--;
            }
        }
        long long int res=max(dp[0],dp1[0]);
        res%=mod;
        return res;
    }
};
