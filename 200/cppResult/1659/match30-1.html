class Solution {
public:
    int maxSum(vector<int>& n1, vector<int>& n2) {
        long ans=0;
        long s1=0,s2=0;
        

        int i=0,j=0;
        while(i<n1.size() and j<n2.size()){
            if(n1[i]<n2[j]){s1+=n1[i];i++;}
            else if(n1[i]>n2[j]){s2+=n2[j];j++;}
            else{
                ans+=max(s1,s2);
                ans%=1000000007;
                s1=0;
                s2=0;
                ans+=n1[i];
                //ans+=n2[j];
                i++;
                j++;

            }
        }
        
        while(i<n1.size()){s1+=n1[i];i++;}
        while(j<n2.size()){s2+=n2[j];j++;}
        
        ans+=max(s1,s2);
        return ans%1000000007;
    }
};
