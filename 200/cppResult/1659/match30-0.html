class Solution {
public:
    long long max(long long &n1,long long&n2){
        if(n1>n2) return n1;
        return n2;
    }
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        long long ans=0;
        int i=0,j=0;
        long long tmp1=0,tmp2=0;
        while(i<nums1.size() && j<nums2.size()){
            if(nums1[i]<nums2[j]){
                tmp1+=nums1[i++]%(1000000007);
            }else if(nums1[i]>nums2[j]){
                tmp2+=nums2[j++]%(1000000007);
            }else{
                ans += max(tmp1,tmp2);
                ans += nums1[i];
                i++;j++;
                tmp1=tmp2=0;
            }
        }
        while(i<nums1.size()){
            tmp1+=nums1[i++]%(1000000007);
        }
        while(j<nums2.size()){
            tmp2+=nums2[j++]%(1000000007);
        }
        ans+=max(tmp1,tmp2);
        return ans%(1000000007);
    }
};
