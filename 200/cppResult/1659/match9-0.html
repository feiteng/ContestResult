class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int mod = 1e9 + 7;
        int i = 0, j = 0;
        int len1 = nums1.size(), len2 = nums2.size();
        long sum = 0, sum1 = 0, sum2 = 0;
        while (i < len1 && j < len2) {
            if (nums1[i] == nums2[j]) {
                sum += max(sum1, sum2) + nums1[i];
                sum %= mod;
                sum1 = sum2 = 0;
                i ++;
                j ++;
            }
            else if (nums1[i] < nums2[j]) {
                sum1 += nums1[i++];
            }
            else if (nums1[i] > nums2[j]) {
                sum2 += nums2[j++];
            }
        }
        while (i < len1) sum1 += nums1[i++];
        while (j < len2) sum2 += nums2[j++];
        sum += max(sum1, sum2);
        sum %= mod;
        return sum;
    }
};
