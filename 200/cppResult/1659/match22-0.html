class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        long long  l1 = nums1.size(), l2 = nums2.size(),
                r1 = 0, r2 = 0, mod = 1e9 + 7.5,
                p1 = 0, p2 = 0;
        while(p1 < l1 && p2 < l2){
            if(nums1[p1] == nums2[p2]){
                r1 = r2 = max(r1, r2) + nums1[p1];
                p1++;
                p2++;
            }else if(nums1[p1] < nums2[p2]){
                r1 = r1 + nums1[p1];
                p1++;
            }else{
                r2 = r2 + nums2[p2];
                p2++;
            }
        }
        while(p1 < l1){
            r1 = (r1 + nums1[p1]);
            p1++;
        }
        while(p2 < l2){
            r2 = (r2 + nums2[p2]);
            p2++;
        }
        return max(r1, r2) % mod;
    }
};
