class Solution {
public:
    using ll = long long;
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        ll ans1 = 0, ans2 = 0;
        int i1 = 0, i2 = 0;
        while (i1 < nums1.size() || i2 < nums2.size()) {
            int n1 = i1 < nums1.size() ? nums1[i1] : INT_MAX;
            int n2 = i2 < nums2.size() ? nums2[i2] : INT_MAX;
            if (n1 == n2) {
                ans1 = ans2 = max(ans1, ans2);
                ans1 += n1;
                ans2 += n2;
                i1++;
                i2++;
            } else {
                if (n1 < n2) {
                    ans1 += n1;
                    i1++;
                } else {
                    ans2 += n2;
                    i2++;
                }
            }
        }
        return max(ans1, ans2) % 1000000007;
    }
};
