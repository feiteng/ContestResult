class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {

        long long score1 = 0, score2 = 0;
        int i = nums1.size() - 1, j = nums2.size() - 1;
        while (i >= 0 && j >= 0) {
            if (nums1[i] == nums2[j]) {
                score1 = max(score1, score2) + nums1[i];
                score2 = score1;
                --i;
                --j;
            } else if (nums1[i] > nums2[j]) {
                score1 = score1 + nums1[i];
                --i;
            } else {
                score2 = score2 + nums2[j];
                --j;
            }
        }
        while (i >= 0) {
            score1 += nums1[i];
            --i;
        }
        while (j >= 0) {
            score2 += nums2[j];
            --j;
        }
        long long ans = max(score1, score2), mod = 1E9 + 7;
        return (int)(ans % mod);
    }
};
