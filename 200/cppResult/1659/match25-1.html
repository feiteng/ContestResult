#pragma GCC optimize ("Ofast")
#include<bits/stdc++.h>
using namespace std;
#define MD (1000000007U)
template<class S, class T> inline S chmax(S &a, T b){
  if(a<b){
    a=b;
  }
  return a;
}
#define main dummy_main
int main(){
  return 0;
}
#undef main

class Solution{
  public:
  int maxSum(vector<int>& A, vector<int>& B){
    int i = 0;
    int j = 0;
    int As = A.size();
    int Bs = B.size();
    long long dpA = 0;
    long long dpB = 0;
    long long res = 0;
    A.push_back(1000000000);
    B.push_back(1000000000);
    while(i < As || j < Bs){
      chmax(res, dpA);
      chmax(res, dpB);
      if(A[i]==B[j]){
        chmax(dpA, dpB);
        chmax(dpB, dpA);
      }

      if(A[i] <= B[j]){
        dpA += A[i];
        i++;
      }
      else{
        dpB += B[j];
        j++;
      }
    }
    chmax(res, dpA);
    chmax(res, dpB);
    return res % MD;
  }
}
