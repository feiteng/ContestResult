class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        const int MAX = 1000000007;
        int res = 0;
        int n1=nums1.size(),n2=nums2.size();
        int sum1=0,sum2=0;
        int cnt1=0,cnt2=0;

        int i1=0,i2=0;
        while(i1<n1&&i2<n2){
            if(nums1[i1] <nums2[i2]){
                cnt1 += (sum1+nums1[i1])/ MAX;
                sum1 = (sum1+nums1[i1])% MAX;
                i1++;
                
            }
            else if(nums1[i1] > nums2[i2]){
                cnt2 += (sum2+nums2[i2])/ MAX;
                sum2 =(sum2+nums2[i2])% MAX;
                i2++;
            }
            else{
                if(cnt1==cnt2){
                    res += (sum1>sum2?sum1:sum2);
                    res%=MAX;
                }
                else{
                    res+=(cnt1>cnt2?sum1:sum2);
                    res%=MAX;
                }
                res+=nums1[i1];
                res%=MAX;
                cnt1=0;

                cnt2=0;
                sum1=0;
                sum2=0;
                i1++;i2++;
            }
        }
        while(i1<n1){
            cnt1 += (sum1+nums1[i1])/ MAX;
            sum1 = (sum1+nums1[i1])% MAX;
            i1++;
        }
        while(i2<n2){
            cnt2 += (sum2+nums2[i2])/ MAX;
            sum2 =(sum2+nums2[i2])% MAX;
            i2++;
        }
        if(cnt1==cnt2){
            res += (sum1>sum2?sum1:sum2);
            res%=MAX;
        }
        else{
            res+=(cnt1>cnt2?sum1:sum2);
            res%=MAX;
        }
        return res;
    }
};
