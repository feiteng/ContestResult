const int mod = 1e9 + 7;

class Solution {
public:
    int maxSum(vector<int>& a, vector<int>& b) {
        long long ans = 0;

        int n = a.size(), m = b.size();
        
        long long s1 = 0, s2 = 0;
        int i = 0, j = 0;
        
        while(i < n and j < m){
            if(a[i] < b[j]) s1 += a[i], i++;
            else if(a[i] > b[j]) s2 += b[j], j++;
            else {
                ans += max(s1, s2);
                ans += a[i];
                i++, j++;
                s1 = 0, s2 = 0;

            }
        }
        while(i < n) s1 += a[i], i++;
        while(j < m) s2 += b[j], j++;
        
        ans += max(s1, s2);
        return ans % mod;
    }
};
