#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod=1e9+7;
const int mn=1e5+10,mv=1e7+10;
ll dp1[mn],dp2[mn];
class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        memset(dp1,0,sizeof(dp1));
        memset(dp2,0,sizeof(dp2));
        int n=nums1.size(),m=nums2.size();
        nums1.insert(nums1.begin(),-1),nums2.insert(nums2.begin(),-1);
        for(int i=n,j=m;i||j;){
            if(nums1[i]>nums2[j]){
                dp1[i]=dp1[i+1]+nums1[i];
                i--;
            }
            else if(nums2[j]>nums1[i]){
                dp2[j]=dp2[j+1]+nums2[j];
                j--;
            }
            else{
                dp1[i]=dp2[j]=max(dp1[i+1]+nums1[i],dp2[j+1]+nums2[j]);
                i--,j--;
            }
        }
        return max(dp1[1],dp2[1])%mod;
    }
};
