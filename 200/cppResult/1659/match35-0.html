class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int64_t ans = 0;
        vector<vector<int64_t>> dp(2, vector<int64_t>(max(nums1.size() , nums2.size()) + 1, 0));
        size_t i = 1, j = 1;
        while (i <= nums1.size() && j <= nums2.size()) {
            if (nums1[i - 1] < nums2[j - 1]) {
                dp[0][i] = dp[0][i - 1] + nums1[i - 1];
                ans = max(ans, dp[0][i]);
                i++;
            } else if (nums1[i - 1] > nums2[j - 1]) {
                dp[1][j] = dp[1][j - 1] + nums2[j - 1];
                ans = max(ans, dp[1][j]);
                j++;
            } else {
                dp[0][i] = dp[1][j] = max(dp[0][i - 1], dp[1][j - 1]) + nums1[i - 1];
                ans = max(ans, dp[0][i]);
                i++;
                j++;
            }
        }
        while (i <= nums1.size()) {
            dp[0][i] = dp[0][i - 1] + nums1[i - 1];
            ans = max(ans, dp[0][i]);
            i++;
        }
        while (j <= nums2.size()) {
            dp[1][j] = dp[1][j - 1] + nums2[j - 1];
            ans = max(ans, dp[1][j]);
            j++;
        }
        return ans % 1000000007;
    }
};
