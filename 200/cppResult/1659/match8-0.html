#define inf 1000000007
class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int i = nums1.size()-1,j = nums2.size()-1;
        long long int global = 0, curr1 = 0, curr2 = 0;
        while(i >= 0 && j >= 0){
            if(nums1[i] == nums2[j]){
                global += max(curr1,curr2) + nums1[i];
                curr1 = 0;
                curr2 = 0;
                i--;
                j--;
            }
            else if(nums1[i] < nums2[j]){
                curr2 += nums2[j];
                j--;
            }
            else{
                curr1 += nums1[i];
                i--;
            }
        }
        while(i >= 0){
            curr1 += nums1[i];
            i--;
        }
        while(j >= 0){
            curr2 += nums2[j];
            j--;
        }
        global += max(curr1,curr2);
        return global%inf;
    }
};
