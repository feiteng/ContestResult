class Solution {
    using ll = long long;
    const ll MOD = 1e9 + 7;
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<int, ll> dp;
        nums1.insert(nums1.begin(), 0);
        nums2.insert(nums2.begin(), 0);
        int i1 = 1, i2 = 1;
        ll ans = numeric_limits<int>::min();

        while (i1 < nums1.size() && i2 < nums2.size()) {
            if (nums1[i1] < nums2[i2]) {
                dp[nums1[i1]] = max(dp[nums1[i1]], dp[nums1[i1-1]] + nums1[i1]);
                ans = max(ans, dp[nums1[i1]]);
                ++i1;
            } else if (nums1[i1] > nums2[i2]) {
                dp[nums2[i2]] = max(dp[nums2[i2]], dp[nums2[i2-1]] + nums2[i2]);
                ans = max(ans, dp[nums2[i2]]);
                ++i2;
            } else {
                dp[nums1[i1]] = max(dp[nums1[i1]], dp[nums1[i1-1]] + nums1[i1]);
                dp[nums2[i2]] = max(dp[nums2[i2]], dp[nums2[i2-1]] + nums2[i2]);
                ans = max(ans, dp[nums2[i2]]);
                ++i1;
                ++i2;
            }
        }
        while (i1 < nums1.size()) {
            dp[nums1[i1]] = max(dp[nums1[i1]], dp[nums1[i1-1]] + nums1[i1]);
            ans = max(ans, dp[nums1[i1]]);
            ++i1;
        }
        while (i2 < nums2.size()) {
            dp[nums2[i2]] = max(dp[nums2[i2]], dp[nums2[i2-1]] + nums2[i2]);
            ans = max(ans, dp[nums2[i2]]);
            ++i2;
        }
        return ans % MOD;
    }
};
