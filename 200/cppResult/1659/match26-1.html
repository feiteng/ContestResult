class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        long res1 = 0, res2 = 0;
        

        int n1 = nums1.size(), n2 = nums2.size();
        for (int i1 = 0, i2 = 0; i1 < n1 || i2 < n2; ) {
            if (i2 == n2 || i1 < n1 && nums1[i1] < nums2[i2]) {
                res1 = res1 + nums1[i1++];
            }
            else if (i1 == n1 || i2 < n2 && nums2[i2] < nums1[i1]) {
                res2 = res2 + nums2[i2++];
            }
            else {
                res1 = max(res1, res2) + nums1[i1];
                res2 = res1;
                ++i1;
                ++i2;
            }
        }
        
        return max(res1, res2) % 1000000007;
    }
};
