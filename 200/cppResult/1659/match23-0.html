class Solution {
public:
    typedef long long LL;
    
    int mod = 1e9 + 7;
    
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size(), m = nums2.size();
        vector<LL> f1(n+1, 0);
        vector<LL> f2(m+1, 0);
        
        int i = n - 1, j = m - 1;
        while (i >= 0 && j >= 0) {
            if (nums1[i] == nums2[j]) {
                f1[i] = f2[j] = max(f1[i + 1], f2[j + 1]) + nums1[i];
                --i;
                --j;
            }
            else if (nums1[i] > nums2[j]) {
                f1[i] = nums1[i] + f1[i+1];
                --i;
            }
            else if (nums1[i] < nums2[j]) {
                f2[j] = nums2[j] + f2[j+1];
                --j;
            }
        }
        while (i >= 0) {
            f1[i] = nums1[i] + f1[i + 1];
            --i;
        }
        while (j >= 0) {
            f2[j] = nums2[j] + f2[j + 1];
            --j;
        }
        return max(f1[0], f2[0]) % mod;
    }
};
