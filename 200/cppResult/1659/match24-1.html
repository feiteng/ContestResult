class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size();
        int n2 = nums2.size();
        int i = 0;
        int j = 0;
        long long score1 = 0;
        long long score2 = 0;
        long long result = 0;
        long long mod = 1000000007;
        while(i < n1 && j < n2){
            if(nums1[i] == nums2[j]){

                result = (result + max(score1, score2)) % mod;
                result = (result + nums1[i]) % mod;
                i++;
                j++;
                score1 = 0;
                score2 = 0;
            }
            else if(nums1[i] < nums2[j]){
                score1 += nums1[i];
                i++;
            }
            else{
                score2 += nums2[j];
                j++;
            }
        }
        while(i < n1){
            score1 += nums1[i];
            i++;
        }
        while(j < n2){
            score2 += nums2[j];
            j++;
        }
        result = (result + max(score1, score2)) % mod;
        return result;
    }
};
