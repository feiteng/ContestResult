class Solution {
    using i64 = int64_t;
public:
    int maxSum(vector<int>& x1, vector<int>& x2) {
        i64 result = 0;
        
        int n1 = x1.size(), n2 = x2.size();
        int i1 =  0, i2 = 0;

        i64 cur1 = 0, cur2 = 0;
        while (i1 < n1 and i2 < n2) {
            if (x1[i1] < x2[i2]) {
                cur1 += x1[i1];
                i1++;
            } else if (x1[i1] > x2[i2]) {
                cur2 += x2[i2];
                i2++;
            } else {
                result += max(cur1, cur2);
                cur1 = 0;
                cur2 = 0;
                result += x1[i1];
                i1++;
                i2++;

            }
        }
        while (i1 < n1) {
            cur1 += x1[i1];
            i1++;
        }
        while (i2 < n2) {
            cur2 += x2[i2];
            i2++;
        }
        result += max(cur1, cur2);
        return result % 1000000007;
    }
};
