#define mod 1000000007
class Solution {
public:
    int maxSum(vector<int>& a, vector<int>& b) {
    int i = 0, j = 0;  
    long long   ans= 0;
    long long val1 = 0;
    long long  val2 = 0; 
    int m= a.size();
    int n = b.size();
    while (i < m && j < n){ 
        if (a[i] < b[j]) {
            val1 = val1 + a[i]; 
            i++;
        }
  
        else if (a[i] > b[j]) {
           val2 =val2+ b[j]; 
            j++;
        }
  
        else{ 

            ans =(ans +  max(val1, val2))%mod; 
            val1 = 0, val2 = 0; 
            while (i < m &&  j < n && a[i] == b[j]) 
            { 
                ans = (ans + a[i++])%mod; 
                j++; 
            } 
        } 
    }
    while (i < m) 
        val1  = val1 +  a[i++]; 
        
    while (j < n) 
        val2 =  val2 + b[j++];  
        
        
    ans = (ans +   max(val1, val2))%mod; 
  
    return ans; 
} 
    
};
