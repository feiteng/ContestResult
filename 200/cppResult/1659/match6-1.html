const int MOD = 1e9 + 7;

class Solution {
public:

	int maxSum(vector<int>& v1, vector<int>& v2) {
		LL r1 = 0, r2 = 0;
		int p1 = v1.size() - 1, p2 = v2.size() - 1;
		while (p1 >= 0 || p2 >= 0) {
			int a = p1 >= 0 ? v1[p1] : 0;
			int b = p2 >= 0 ? v2[p2] : 0;

			if (a == b) {
				r1 = r2 = a + max(r1, r2);
				--p1; --p2;

			}
			else {
				if (a > b) {
					r1 = a + r1;
					--p1;
				}
				else {
					r2 = b + r2;
					--p2;
				}
			}
		}

		return max(r1, r2) % MOD;
	}
};
