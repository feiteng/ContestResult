class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        const int mod=1000000007;
        int ret=0;
        int64_t sum1=0,sum2=0;
        auto it1=nums1.cbegin(),it2=nums2.cbegin();
        while(it1!=nums1.cend() && it2!=nums2.cend()){
            if(*it1<*it2){
                sum1+=*it1++;
            }else if(*it2<*it1){
                sum2+=*it2++;
            }else{
                ret+=max(sum1,sum2)%mod;
                ret+=*it1;
                sum1=0;
                sum2=0;
                ret%=mod;
                ++it1;
                ++it2;
            }
        }
        while(it1!=nums1.cend()){
            sum1+=*it1++;
        }
        while(it2!=nums2.cend()){
            sum2+=*it2++;
        }
        ret+=max(sum1,sum2)%mod;
        ret%=mod;
        return ret;
    }
};
