class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
    long long  left = 0, right = 0;

    int n1 = nums1.size(), n2 = nums2.size();
    int l = 0, r = 0;
    while ((l < n1) && (r < n2)) {
        if (nums1[l] < nums2[r]) {
            left += nums1[l];
            l++;
        }
        else if (nums1[l] > nums2[r]) {
            right += nums2[r];
            r++;
        }
        else {
            long long  mid = max(left, right) + nums1[l];

            left = mid;
            right = mid;
            l++;
            r++;
        }
    }
    while (l < n1) {
        left += nums1[l];
        l++;
    }
    while (r < n2) {
        right += nums2[r];
        r++;
    }
    long long ans;
    if (left> right) ans=left;
        else ans=right;
    ans%=1000000007;
    return ans;
}
};
