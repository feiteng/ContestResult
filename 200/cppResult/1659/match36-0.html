class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        vector<long long> sum1, sum2;
        sum1.push_back(0);
        sum2.push_back(0);
        int n1 = nums1.size(), n2 = nums2.size();
        
        int i = 0, j = 0;
        while (i < n1 && j < n2) {
            if (nums1[i] == nums2[j]) {
                sum1.back() += nums1[i];
                sum2.back() += nums2[j];
                sum1.push_back(0);
                sum2.push_back(0);
                ++i;
                ++j;
            }else if (nums1[i] > nums2[j]) {
                sum2.back() += nums2[j];
                ++j;
            }else {
                sum1.back() += nums1[i];
                ++i;
            }
        }
        
        while (i < n1) {
            sum1.back() += nums1[i];
            ++i;
        }
        while (j < n2) {
            sum2.back() += nums2[j];
            ++j;
        }
        
        long long ans = 0;
        const long long MOD = 1e9 + 7;
        int m = sum1.size();
        for (int i = 0; i < m; ++i) {
            if (sum1[i] > sum2[i]) {
                ans = (ans + sum1[i]) % MOD;
            }else {
                ans = (ans + sum2[i]) % MOD;
            }
        }
        
        return ans;
    }
};
