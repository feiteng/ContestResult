class Solution {
public:
    int maxSum(vector<int>& nums1, vector<int>& nums2) {
        nums1.push_back(INT_MAX);
        nums2.push_back(INT_MAX);
        int len1 = nums1.size();
        int len2 = nums2.size();
        int ptr1 = 0;
        int ptr2 = 0;
        long long count1 = 0;
        long long count2 = 0;
        long long res = 0;
        while(ptr1<len1 || ptr2 < len2) {
            int i = nums1[ptr1];
            int j = nums2[ptr2];
            if(i==j) {
                int val = i;
                if(val == INT_MAX) val = 0;
                count1 += val;
                count2 += val;
                res += max(count1, count2);
                res %= 1000000007;
                count1 = 0;
                count2 = 0;
                ptr1++;
                ptr2++;
            } else if(i<j) {
                count1 += i;
                ptr1++;
            } else {
                count2 += j;
                ptr2++;
            }
        }
        return res;
    }
};
