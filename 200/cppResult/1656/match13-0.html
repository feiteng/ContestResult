class Solution {
public:
    int countGoodTriplets(vector<int>& arr, int a, int b, int c) {
        int rs = 0;
        int sz = arr.size();
        for(int i=0; i<sz; i++)
            for(int j=i+1; j<sz; j++)
                for(int k=j+1; k<sz; k++)
                    if(max(arr[i]-arr[j],arr[j]-arr[i])<=a&&max(arr[j]-arr[k],arr[k]-arr[j])<=b&&max(arr[i]-arr[k],arr[k]-arr[i])<=c)
                        rs++;
        return rs;
    }
};
