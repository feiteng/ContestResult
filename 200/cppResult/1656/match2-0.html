class Solution {
public:
    int countGoodTriplets(vector<int>& A, int a, int b, int c) {
        int ans = 0, n = A.size();
        for(int i = 0; i < n; ++i) {
            for(int j = i + 1; j < n; ++j) {
                for(int k = j + 1; k < n; ++k) {
                    if(abs(A[i] - A[j]) <= a && abs(A[j] -A[k]) <= b && abs(A[i] - A[k]) <= c) {
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
