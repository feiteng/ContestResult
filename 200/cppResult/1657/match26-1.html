class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        if (arr.size() <= k) {
            int res = 0;
            for (int n : arr) {
                res = max(res, n);
            }
            return res;
        }

        int cnt = 0;
        int curi = 0;
        for (int i = 1; i < arr.size(); ++i) {
            if (arr[curi] > arr[i]) {
                ++cnt;
                if (cnt >= k) {
                    return arr[curi];
                }
            } else {
                curi = i;
                cnt = 1;
                if (cnt >= k) {
                    return arr[curi];
                }
            }
        }
        return arr[curi];
    }
};
