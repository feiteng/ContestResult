class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int n = arr.size();
        int ma = -1;
        for (int& v : arr) ma = max(ma, v);
        if (k >= n - 1) return ma;
        vector<int> q;
        for (int i = 1; i < n; i++) q.push_back(arr[i]);
        int cur = arr[0];
        int cnt = 0;
        int p = 0;
        while (1) {
            if (cnt >= k) return cur;
            // cout << q[p] << " " << cur << endl;
            if (q[p] < cur) {
                q.push_back(q[p]);
                p++;
                cnt++;
            } else {
                cnt = 1;
                q.push_back(cur);
                cur = q[p];
                p++;
            }
            // cout << cur << " " << cnt << endl;
        }
        return -1;
    }
};
