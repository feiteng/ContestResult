class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        if(k > arr.size()){
            int result = 0;
            for(int i = 0; i < arr.size(); i++) result = max(result, arr[i]);
            return result;
        }else{
            int tk = 0;
            int spoint = arr[0];
            for(int i = 1; i < arr.size(); i++){
                if(arr[i] < spoint){
                    tk ++;
                    if(tk >= k) {
                        return spoint;
                    }
                }else{
                    tk = 1;
                    spoint = arr[i];
                    if(tk >= k) {
                        return spoint;
                    }
                }
            }
            return spoint;
        }
    }
};
