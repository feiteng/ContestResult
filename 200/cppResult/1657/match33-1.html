class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int res_i = 0;
        int res_cnt = 0;
        if (arr[0] > arr[1]) {
            res_i = 0;
            res_cnt = 1;
        } else {
            res_i = 1;
            res_cnt = 1;
        }
        
        if (res_cnt == k) {
            return arr[res_i];
        }
        

        for (int i = 2; i < arr.size(); ++i) {
            if (arr[i] > arr[res_i]) {
                res_i = i;
                res_cnt = 1;
            } else {
                res_cnt++;
            }
            if (res_cnt == k) {
                return arr[res_i];
            }
        }
        
        return arr[res_i];
    }
};
