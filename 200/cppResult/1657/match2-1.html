class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int w=arr[0],c=0;
        for (int a:arr) {
            if (w==a) continue;

            if (w>a) {
                ++c;
            }
            else {
                c=1;
                w=a;
            }
            if (c>=k) return w;
        }
        return w;
    }
};
