class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int n = arr.size();
        int curr = arr[0];
        int cnt = 0;
        for(int i = 1; i < n; ++i){
            if(arr[i] > curr){
                cnt = 1;
                curr = arr[i];
            }else{
                cnt++;
            }
            
            if(cnt == k){
                return curr;
            }
        }
        
        return curr;
    }
};
