class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int winner_num = arr[0], winner_count=0;
        for (int i = 1; i < arr.size(); i++)
        {
            if (winner_num > arr[i])
            {
                winner_count += 1;
            }
            else
            {
                winner_num = arr[i];
                winner_count = 1;
            }
            if (winner_count == k) return winner_num;
        }
        return winner_num;
    }
};
