class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int winner = arr[0], win_cnt = 0;
        for (int i = 1; i < arr.size(); ++i) {
            if (arr[i] < winner) {
                ++win_cnt;
            } else {
                winner = arr[i];
                win_cnt = 1;
            }
            if (win_cnt >= k) return winner;
        }
        return winner;
    }
};
