class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int cur = arr[0];
        int cnt = 0;
        for(int i = 1; i < arr.size(); i++) {
            if(arr[i] > cur) {
                cur = arr[i];
                cnt = 1;
                // cout << "no" << endl;
            } else {
                // arr[i] < cur
                cnt++;
                
                // cout << cnt << endl;
            }
            if(cnt == k) {
                return cur;
            }
        }
        return cur;
    }
};
