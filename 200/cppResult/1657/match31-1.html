class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int current = arr[0];
        int t = k;
        for(int i=1;i<arr.size() and t > 0 ;++i){
            if( current > arr[i] ){
                --t;
            } 
            else{
                current = arr[i];
                t = k-1;
            }
        }
        return current;
    }
};
