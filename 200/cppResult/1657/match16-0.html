class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int p=1;
        int maxx=0;
        int win=0;
        for(int j=1;j<arr.size();j++){
            if(arr[maxx]>arr[j]){
                win++;
            }
            else{
                maxx=j;
                win=1;
            }
            if(win==k) return arr[maxx];
        }
        return arr[maxx];
    }
};
