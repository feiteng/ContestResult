class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int N=arr.size();
        int kn=0; ///连续赢的次数
        for(int i=1;i<N;i++){
            if(arr[i-1]>arr[i]){
                swap(arr[i-1],arr[i]);

                kn++;  //连续赢了
                
            }else{
                kn=1;
            }
            if(kn>=k){
                return arr[i];
            }
        }
        
        return arr[N-1];
    }
};
