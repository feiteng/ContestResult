class Solution {
public:
    int getWinner(vector<int>& A, int k) {
        int n = A.size();
        int max_ = *max_element(A.begin(), A.end());
        if(k >= n) {
            return max_;
        }
        int cur = A[0], cnt = 0;
        for(int i = 1; i < n; ++i) {
            if(A[i] > cur) {
                cur = A[i];
                cnt = 1;
            }else {
                cnt++;
            }
            if(cnt == k) {
                return cur;
            }
        }
        return max_;
    }
};
