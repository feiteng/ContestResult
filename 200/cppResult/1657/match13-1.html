class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int ret = arr[0];
        int now = 0;
        for (int i = 1;i < arr.size();i++) {
            if (ret > arr[i]) {
                now++;
            } else {
                ret = arr[i];
                now = 1;
            }
            if (now >= k) {
                return ret;
            }
        }
        return ret;
    }
};
