class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int winner = max(arr[0], arr[1]);
        int cnt = 1;
        for (int i = 2; cnt != k && i < arr.size(); ++i) {
            int new_winner = max(winner, arr[i]);
            if (new_winner == winner) {
                ++cnt;
            } else {
                winner = new_winner;
                cnt = 1;
            }
        }
        return winner;
    }
};
