class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int winCount = 0;
        int maxVal = arr[0];
        int current = arr[0];
        for (int i = 1; i < arr.size(); ++ i) {
            maxVal = max(maxVal, arr[i]);

            if (current > arr[i]) {
                ++ winCount;
            } else {
                current = arr[i];
                winCount = 1;
            }
            // printf("%d %d\n", i, current);
            if (winCount == k) return current;
        }
        return maxVal;
    }
};
