class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int mx = *max_element(arr.begin(), arr.end());
        int n = arr.size();
        int cnt = 0, cur_mx = arr[0];
        for(int i = 1; i < n; i ++) {
            if(arr[i] > cur_mx) {
                cnt = 1;
                cur_mx = arr[i];
            } else {
                cnt ++;
            }
            if(cnt == k) return cur_mx;
        }
        return mx;
    }
};
