class Solution {
public:
    bool ok(vector<int>& row, int from, int to) {
        vector<int> tmp;
        for (int i = from; i <= to; i++) tmp.push_back(row[i]);
        sort(tmp.begin(), tmp.end(), greater<int>());
        for (int i = 0, need = (int)row.size() - 1 - from; i < tmp.size(); i++, need--) {
            if (tmp[i] < need) return false;
        }
        return true;
    }
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> row(n);
        for (int i = 0; i < n; i++) {
            int cnt = 0;
            for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 0) cnt++;
                else break;
            }
            row[i] = cnt;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (row[i] >= n - 1 - i && ok(row, i + 1, n -1)) continue;
            
            int candi = -1;
            for (int j = i + 1; j < n; j++) {
                swap(row[i], row[j]);
                bool ck = row[i] >= n - 1 - i && ok(row, i + 1, n - 1);
                swap(row[i], row[j]);
                
                if (ck) {
                    candi = j;
                    break;
                }
            }
            
            if (candi == -1) return -1;
            for (int j = candi; j > i; j--) {
                swap(row[j], row[j - 1]);
                res++;
            }
        }
        return res;
    }
};
