class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n=grid.size();
        vector<int> cnt(n,0);
        for(int i=0;i<n;i++){
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]==0) cnt[i]++;
                else break;
            }
        }
        int ans=0;
        for(int i=0;i<n;i++){
            bool f=false;
            for(int j=0;j<cnt.size();j++){
                if(cnt[j]>=n-i-1){
                    ans+=j;
                    cnt.erase(cnt.begin()+j);
                    f=true;
                    break;
                }
            }
            if(!f) return -1;
        }
        return ans;
    }
};
