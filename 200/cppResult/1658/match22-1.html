class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> a;
        for(int i=0; i<n; ++i) {
            int sz = 0;
            for(int j=n-1; j>=0; --j) {
                if(grid[i][j]) break;
                sz++;
            }
            sz = n- sz;
            a.push_back(sz);
        }
        int ans = 0;
        for(int i=0; i<n; ++i) {

            if(a[i]<=i+1) continue;
            int x = -1;
            for(int j=i+1; j<n; ++j) {
                if(a[j]<=i+1) {
                    x = j;
                    break;
                }
            }
            if(x==-1) return -1;
            for(int j=x; j>i; --j) {
                swap(a[j], a[j-1]);
                ++ans;
            }
        }
        return ans;
    }
};
