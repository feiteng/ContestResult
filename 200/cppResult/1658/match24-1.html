class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        vector<int> my_size;
        for(int i = 0;i < grid.size(); i++){
            int can0 = 0;

            for(int j = grid[i].size()-1; j >= 0; j--){
                if(grid[i][j] == 0){
                    can0++;
                }
                else{
                    break;
                }
            }
            //cout<<can0<<endl;
            my_size.push_back(can0);
        }
        int tans = 0;
        for(int i = 0; i < grid.size(); i++){
            int need = grid[i].size() - 1 - i;
            int tmin = -1;
            for(int j = 0; j< my_size.size(); j++){
                if(my_size[j] >= need){
                    tmin = j;
                    break;
                }
            }
            if(tmin == -1){
                tans = -1;
                break;
            }
            my_size.erase(my_size.begin()+tmin);
            tans += tmin;
        }
        return tans;
    }
};
