class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = static_cast<int>(grid.size());

        vector<int> zeroCnt(n);
        for (int i = 0; i < n; ++i) {
            int cnt = 0;
            for (int j = n - 1; j >= 0; --j) {
                if (grid[i][j] == 1) {
                    break;
                }

                ++cnt;
            }
            zeroCnt[i] = cnt;
        }
        int swaps = 0;
        for (int i = 0; i < n; ++i) {
            int req = n - i - 1;
            int pos = -1;
            for (int j = i; j < n; ++j) {
                if (zeroCnt[j] >= req) {
                    pos = j;
                    break;
                }
            }
            if (pos == -1) {
                return -1;
            }

            swaps += pos - i;
            for (int j = pos; j > i; --j) {
                zeroCnt[j] = zeroCnt[j - 1];
            }
        }
        return swaps;
    }
};
