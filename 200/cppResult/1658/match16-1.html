class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n=grid.size();
        vector<int> newix(n);
        for(int i=0;i<n;i++){
            int c=0;
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]){
                    break;
                }
                c++;

            }//cout<<c<<" ";
            newix[i]=c;
        }
        int ans=0;
        for(int i=0;i<n;i++){
            int req=n-i-1;
            bool f=true;
            int ix=i;
            for(int j=i;j<n;j++){//cout<<newix[j]<<" "<<req<<endl;
                if(newix[j]>=req){
                    ix=j;

                    f=false;
                    break;
                }
                
            }
            if(f){
                return -1;
            }
            while(ix>i){
                swap(newix[ix],newix[ix-1]);
                ix--;
                ans++;
            }
            
        }
        return ans;
    }
};
