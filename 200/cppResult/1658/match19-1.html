class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> a(n);
        for(int i=0;i<n;i++) {
            int cnt = 0;
            int k = n-1;
            while(k>=0 && grid[i][k]==0) {
                cnt++;
                k--;
            }
            a[i] = cnt;

        }
        
        int ans = 0;
        for(int i=0;i<n;i++) {
            int loc = -1;
            for(int j=i;j<n;j++) {
                if(a[j] >= n - 1 - i) {
                    loc = j;
                    break;
                }
            }
            if(loc == -1) return -1;
            ans += (loc - i);
            int tmp = a[loc];
            for(int s = loc;s>i;s--)
                a[s] = a[s-1];
            a[i] = tmp;
        }
        return ans;
    }
};
