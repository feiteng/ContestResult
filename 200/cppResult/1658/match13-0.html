class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        
        int n = grid.size();
        if(n == 1)
            return 0;
        vector<int> zm(n, 0);
        for(int i = 0; i < n; i++) {
            int zeros = 0;
            for(int j = n - 1; j >= 0; j--) {
                if(grid[i][j] != 0)
                    break;
                zeros++;
            }
            zm[i] = zeros;
        }
        
        int res = 0;
        for(int i = 0; i < n; i++) {
            int need = n - i - 1;
            int index = n;
            for(int j = i; j < n; j++) {
                if(zm[j] >= need) {
                    index = j;
                    break;
                }
            }
            if(index == n)
                return -1;
            int steps = index - i;
            res += steps;
            for(int j = index; j > i; j--) {
                zm[j] = zm[j-1];
            }
        }
        return res;
    }
};
