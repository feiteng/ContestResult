class Solution {
public:
    
    void swap_(int x,int y,int fre[])
    {
        if(x==y){return;}
        for(int i=y-1;i>=x;i--)
        {
            swap(fre[i],fre[i+1]);
        }
        return;
    }
    
    int minSwaps(vector<vector<int>>& a) 
    {
        int n = a.size();
        int fre[n+1],ans=0;
        memset(fre,0,sizeof(fre));

        for(int i=0;i<n;i++)
        {
            int c=0;
            for(int j=n-1;j>=0;j--)
            {
                if(a[i][j]==0){c++;}
                else{break;}
            }
            fre[i] = c;
        }

        for(int i=0;i<n;i++)
        {
            bool f = false;;
            for(int j=i;j<n;j++)
            {
                if(fre[j]>=n-1-i)
                {
                    f=true;
                    swap_(i,j,fre);
                    ans += (j-i);
                    break;
                }
            }
            if(f==false){
                return -1;
            }
        }
        return ans;
    }
};
