class Solution {
public:

    int minSwaps(vector<vector<int>>& grid) {
        int l = grid.size();
        vector<int> res(l, 0);
        for(int i = 0; i < l; i++)
        {
            int count = 0;
            for(int j = l - 1; j >= 0; j--)
            {
                if(grid[i][j] == 0)
                    count++;
                else
                    break;
            }
            res[i] = count;
        }
        int ans = 0;
        for(int i = l - 1; i >= 0; i--)
        {
            int j = l - 1 - i;
            for(; j < l; j++)
                if(res[j] >= i)
                    break;

            if(j == l)
                return -1;
            for(int k = j; k >= l - i; k--)
            {
                swap(res[k], res[k - 1]);
                ans++;
            }
        }
        return ans;
    }
};
