class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> vec(n);
        for (int i = 0; i < n; i++) {
            int t = 0;
            for (int j = n - 1; j > 0; j--) {
                if (grid[i][j] == 1) break;
                t++;
            }
            vec[i] = t;
        }
        int res = 0;
        for (int i = 0; i < n - 1; i++) {
            int t = 0;

            bool found = false;
            for (int j = 0; j < n; j++) {
                if (vec[j] >= n - i - 1) {
                    vec[j] = -1;
                    found = true;
                    break;
                }
                else if (vec[j] != -1) t++;
            }
            if (!found) return -1;
            res += t;
        }
        return res;
    }
};
