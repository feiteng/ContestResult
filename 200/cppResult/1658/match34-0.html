class Solution {
public:
	
	
    int minSwaps( vector<vector<int>>& grid ) {
		int n=grid.size();
		vector<int> tmp;
		for( int i=0;i<grid.size();i++ )
		{
			int p=0;
			for( int j=grid[i].size()-1;j>=0;j-- )
			{
				if( grid[i][j]==0 ) p++;
				else break;
			}
			tmp.push_back(p);
		}
		int ans=0;
		//for(int v:tmp ) cout<<v<<" ";
		for( int i=0;i<tmp.size();i++ )
		{
			if( tmp[i]>=n-i-1 )
			{
				continue;
			}
			else
			{
				bool f=false;
				for( int j=i+1;j<tmp.size();j++ )
				{
					if( tmp[j]>=n-i-1 )
					{
						ans+=j-i;
						for( int k=j;k>i;k-- )
						{
							swap(tmp[k],tmp[k-1]);
						}
						f=true;
						break;
					}
				}
				if( !f ) return -1;
			}
		}
		return ans;
    }
    
};
