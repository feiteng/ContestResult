class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> v;
        for(int i = 0; i < n; i += 1){
            v.push_back(0);

            for(int j = n - 1; j >= 0; j -= 1)
                if(grid[i][j] == 0)
                    v.back() += 1;
                else break;
        }
        int ans = 0;
        for(int i = 0; i < n; i += 1) if(v[i] >= (n - i - 1)) continue;
        else{
            int k = -1;
            for(int j = i + 1; j < n; j += 1) if(v[j] >= (n - i - 1)){
                k = j;
                break;
            }

            if(k == -1) return -1;
            for(int j = k; j > i; j -= 1){
                swap(v[j], v[j - 1]);
                ans += 1;
            }
        }
        return ans;
    }
};
