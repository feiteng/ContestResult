class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> dp(n,0);
        for(int i=0;i<n;i++){

            int j = n-1;
            while(j>=0 && grid[i][j]==0){
                dp[i]++;
                j--;
            }
        }
        int res = 0;
        for(int i=0;i<n;i++){
            int cnt = n-i-1;
            int idx = -1;

            for(int j=i;j<n;j++){
                if(dp[j]>=cnt){
                    idx = j;
                    break;
                }
            }
            if(idx==-1) return -1;
            res+=idx-i;
            while(idx--){
                swap(dp[idx+1],dp[idx]);
            }
        }
        return res;
    }
};
