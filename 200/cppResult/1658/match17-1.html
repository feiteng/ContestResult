class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            int cnt = 0;
            for (int j = n-1; j >= 0; j--) {
                if (grid[i][j]) break;
                ++cnt;

            }
            a[i] = cnt;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int need = n - 1 - i;
            if (a[i] >= need) continue;

            int row = -1;
            for (int j = i+1; j < n; j++) {
                if (a[j] >= need) {
                    row = j;
                    break;
                }
            }
            if (row == -1) {
                return -1;
            }
            ans += row - i;
            while (row > i) {
                swap(a[row], a[row-1]);
                --row;
            }
        }
        return ans;
    }
};
