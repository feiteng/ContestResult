class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        
        int n=grid.size();
        
        
        vector<int> zero_len(n,0);
        for(int i=0;i<n;i++) {
            for(int j=n-1;j>=0;j--) {
                if (grid[i][j]==1) break;
                zero_len[i]++;
            }
        }
        int ans=0;
        for(int i=0;i<n;i++) {
            // zero_len >= n-1-i.

            int jopt=-1;
            for(int j=i;j<n;j++) {
                if (zero_len[j] >= n-1-i) {
                    jopt=j;
                    break;
                }
            }
            if (jopt==-1) return -1;
            
            for(int k=jopt;k>=i+1;k--) {
                swap(zero_len[k],zero_len[k-1]);
                ans++;
            }
            
        }
        return ans;
    }
};
