class Solution {
public:
    int num[220];

    int minSwaps(vector<vector<int>>& grid) {
        int n=grid[0].size();
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int id=-1;
            for(int j=i;j<n;j++)
            {
                int tmp=0;
                for(int k=n-1;k>=0;k--)
                {
                    if(grid[j][k]==0) tmp++;
                    else break;

                }
                if(tmp>=n-i-1) 
                {
                    id=j; break;
                }
            }
            if(id==-1) return -1;
            ans+=id-i;
            for(int j=id;j>=i+1;j--) swap(grid[j],grid[j-1]);
        }
        return ans;
    }
};
