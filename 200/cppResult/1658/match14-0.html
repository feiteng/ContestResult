class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        int cnt = 0;
        for(int i = 0; i < n; i++) {
            int fil = -1;
            for(int j = i; j < n; j++) {
                bool can = 1;
                for(int k = i + 1; k < n; k++) {
                    if(grid[j][k] == 1) can = 0;
                }
                if(can) {
                    fil = j;
                    break;
                }
            }
            if(fil == -1) return -1;
            cnt += fil - i;
            swap(grid[fil], grid[i]);
            for(int j = fil; j - 1 > i; j--) {
                swap(grid[j], grid[j - 1]);
            }
        }
        return cnt;
    }
};
