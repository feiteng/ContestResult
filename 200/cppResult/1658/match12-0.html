class Solution {
public:
    int minSwaps(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector <int> v;
        v.resize(n);
        for (int i = 0; i < n; i++) {
            int cur = 0;
            for (int j = m - 1; j >= 0; j--) {
                if (a[i][j] != 0) {
                    break;
                } else {
                    cur++;
                }
            }
            v[i] = cur;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int need = n - i - 1;
            int sel = -1;
            for (int j = i; j < n; j++) {
                if (v[j] >= need) {
                    sel = j;
                    break;
                }
            }
            if (sel == -1) {
                return -1;
            }
            
            ans += (sel - i);
            for (int j = sel; j > i; j--) {
                v[j] = v[j - 1];
            }
        }
        return ans;
    }
};
