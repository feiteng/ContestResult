class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        int ans = 0;
        
        for(int i=0; i<n; i++) {
            // [0 - i] 1  > i -- 0
            
            int  choose = -1;
            
            
            for(int row = i; row < n; row++)
            {
                int max1 = -1;
                for(int j=0; j<n; j++) {
                    if(grid[row][j] == 1) {
                        max1 = j;
                    }

                }
                if(max1 <= i) {
                    choose = row;
                    break;
                }
            }
            if(choose == -1) return -1;
            // move row to i
            ans += choose - i;
            for(int j=choose-1; j>=i; j--) {
                swap(grid[j], grid[j+1]);
            }
        }
        return ans;

    }
};
