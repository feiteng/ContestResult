class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> data(n, 0);
        for (int i = 0;i < n;i++) {
            int count = 0;
            for (int j = n - 1;j >= 0;j--) {
                if (grid[i][j] == 1) {
                    break;
                } else {
                    count++;
                }
            }
            data[i] = count;
        }
        
        int ret = 0;
        for (int i = 0;i < n;i++) {
            int t = 0;
            int k = -1;
            for (int j = i;j < n;j++) {
                if (data[j] >= n - i - 1) {
                    k = j;
                    break;
                }
            }
            if (k == -1) {
                return -1;
            }
            while (k != i) {
                swap(data[k], data[k - 1]);
                ret++;
                k--;
            }
        }
        return ret;
    }
};
