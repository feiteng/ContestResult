class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> zeros(n);
        for (int i = 0; i < n; ++i) {
            int count = 0;
            for (int j = n - 1; j >= 0; --j) {
                if (grid[i][j] == 0) {
                    ++count;
                } else {
                    break;
                }
            }
            zeros[i] = count;
        }
        
        int res = 0;
        for (int i = n - 1; i >= 0; --i) {
            bool find = false;
            for (int j = 0; j < zeros.size(); ++j) {
                if (zeros[j] >= i) {
                    res += j;
                    zeros.erase(zeros.begin() + j);
                    find = true;
                    break;
                }
            }
            
            if (!find) {
                return -1;
            }
        }
        
        return res;
    }
};
