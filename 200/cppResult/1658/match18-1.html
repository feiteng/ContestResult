class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n=grid.size();
        
        int pos[n];
        for(int i=0;i<n;i++)
        {
            int count=0;
            for(int j=n-1;j>=0 && grid[i][j]==0;j--,count++);

            pos[i]=count;
            //cout<<pos[i]<<" ";
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int index=-1;
            for(int j=i;j<n;j++)
                if(pos[j]>=n-i-1)
                {
                    index=j;
                    break;

                }
            //cout<<i<<" "<<index<<endl;
            if(index==-1)
                return -1;
            
            ans+=index-i;
            int temp=pos[index];
            for(int j=index;j>i;j--)
                pos[j]=pos[j-1];
            pos[i]=temp;
        }
        
        return ans;
        
        
            
        
    }
};
