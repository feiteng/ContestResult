class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
		vector<int> num(n);
		int res = 0;
		
		for(int i = 0; i < n; i++){
			for(int j = n-1; j >= 0; j--){
				if(grid[i][j] == 1) break;
				num[i]++;
			}
		}
		//cout<<"[][]"<<endl;
		for(int i = n-1; i >= 1; i--){ //cout<<"i = "<<i<<endl;
			int ind = -1;
			
			for(int j = n - i - 1; j < n; j++){
				if(num[j] >= i) {
					ind = j;
					break;
				}
			}
			
			if(ind == -1) return -1;
			
			//cout<<"ind = "<<ind<<" x = "<<n-i-1<<endl;
			for(int j = ind; j > n - i - 1; j--){
				swap(num[j], num[j-1]);
				res++;
			}
		}
		
		return res;
    }
};
