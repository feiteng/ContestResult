#define pb push_back
class Solution {
public:
    int minSwaps(vector<vector<int>>& a) {
        
        int n = a.size();
        
        vector<int> v;
        
        for(int i=0;i<n;i++){
            int cnt=0;
            int j=n-1;
            while(j>=0 && a[i][j]==0){
                cnt++;
                j--;
            }
            v.pb(cnt);
        }
        
        
        int ans=0;
        for(int i=0;i<n;i++){
            
            int row = -1;
            for(int j=i;j<n;j++){
                if(v[j]>=n-i-1){
                    row = j;
                    break;
                }
            }
            
            if(row==-1)
                return -1;
            
            ans+=row-i;
            
            for(int j=row;j>i;j--){
                swap(v[j],v[j-1]);
            }
        }
        
        return ans;
    }
};
