class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        vector<int> arr(grid.size(), 0);
        for (int i = 0; i < grid.size(); i++) {
            for (int j = grid[0].size()-1; j >= 0; j--) {
                if (grid[i][j] == 0) {
                    arr[i]++;
                } else {
                    break;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < grid.size(); i++) {
            int target = grid.size() - i - 1;
            int row = -1;
            for (int j = i; j < grid.size(); j++) {
                if (arr[j] >= target) {
                    row = j;
                    break;
                }
            }
            if (row == -1) { return -1; }
            int temp = arr[row];
            for (int j = row; j > i; j--) {
                arr[j] = arr[j-1];
                ans++;
            }
            arr[i] = temp;
        }
        return ans;
    }
};
