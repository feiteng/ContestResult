class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n=grid.size();
        vector<int> zeros(n,0);
        for(int i=0;i<n;i++){
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]==0) zeros[i]++;
                else break;
            }
        }
        int ans=0;
        for(int i=0;i<n-1;i++){
            bool flag=true;
            for(int j=i;j<n;j++){
                if(zeros[j]>=n-i-1){
                    ans+=j-i;
                    int temp=zeros[j];
                    for(int k=j-1;k>=i;k--) zeros[k+1]=zeros[k];
                    zeros[i]=temp;
                    flag=false;
                    break;
                }
            }
            if(flag) return -1;
        }
        return ans;
    }
};
