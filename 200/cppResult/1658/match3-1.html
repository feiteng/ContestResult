class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int row = grid.size();
        vector<int> vecGrid(row,0);
        
        for(int i = 0; i < row; i++) {

            for(int j = row-1; j >= 0; j--) {
                if(grid[i][j] == 0) {
                    vecGrid[i]++;
                }
                else break;
            }
        }
        
        int ans = 0;

        bool canDo = false;
        
        for(int i = 0; i < row; i++) {
            canDo = false;
            for(int j = 0; j < vecGrid.size(); j++) {
                if( vecGrid[j] >= (row-i-1) ) {
                    ans += j;
                    canDo = true;
                    vecGrid.erase(vecGrid.begin()+j);
                    break;
                }
            }
            if(canDo == false) return -1;
        }
        
        
        
        return ans;
    }
};
