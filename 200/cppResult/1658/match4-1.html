class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> nums;
        for (int i = 0; i < n; ++i) {
            int curnum = 0;
            for (int j = n - 1; j >= 0; --j) {
                if (grid[i][j] == 0) {
                    curnum++;
                }
                else break;
            }
            nums.push_back(curnum);

        }
        // cout << nums[0] << " " << nums[1] << " " << nums[2] << endl;
        int res = 0;
        for (int i = 0; i < n; ++i) {
            bool found = false;
            for (int j = i; j < n; ++j) {
                if (nums[j] >= n - i - 1) {
                    found = true;
                    res += (j - i);
                    nums.erase(nums.begin() + j);
                    nums.insert(nums.begin(), 0);
                    break;
                }
            }
            if (!found) return -1;
        }
        return res;
    }
};
