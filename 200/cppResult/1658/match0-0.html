class Solution {
public:
    int minSwaps(vector<vector<int>>& a) {
        int n = a.size();
        vector<int>v(n, 0);
        for(int i=0;i<n;i++){
            for(int j=n-1;j>=0;j--){
                if(a[i][j])break;
                v[i]++;
            }
        }
        int ret = 0;
        for(int i=0;i<n;i++){
            if(v[i]>=n-i-1)continue;
            int pos = -1;
            for(int j=i+1;j<n;j++){
                if(v[j]>=n-i-1){
                    pos = j;
                    break;
                }
            }
            if(pos==-1)return -1;
            for(int j=pos;j>i;j--){
                swap(v[j],v[j-1]);
                ret++;
            }
        }
        return ret;
    }
};
