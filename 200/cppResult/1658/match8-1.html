class Solution {
public:
    int minSwaps(vector<vector<int>>& g) {
        int n = g.size();
        vector<int> pos(n);
        for(int i = 0; i < n; i++) {
            int c = 0;
            for(int j = n - 1; j >= 0; j--) {
                if(g[i][j] == 0) {
                    c++;
                } else {
                    break;
                }
            }
            pos[i] = c;
        }
        // for(auto &i : pos) {
        //     cout << i << endl;
        // }
        vector<int> v(n);
        int res = 0;
        for(int i = 0; i < n; i++) {

            int f = 0;
            // cout << i << endl;
            for(int j = i; j < n; j++) {
                if(pos[j] >= n - i - 1) {
                    res += j - i;
                    pos.erase(pos.begin() + j, pos.begin() + j + 1);
                    pos.insert(pos.begin(), 0);
                    f = 1;
                    break;
                }
            }
            if(!f) {
                return -1;
            }
        }
        return res;
    }
};
