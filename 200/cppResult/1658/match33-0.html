class Solution {
public:
    void swap(vector<int>& v, int i, int j) {
        if(v[i]==v[j]) return;
        v[i] ^= v[j];
        v[j] ^= v[i];
        v[i] ^= v[j];
    }
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> vec(n,0);
        for(int i=0; i<n; i++) {
            int t = 0;
            for(int j=n-1; j>=0; j--) {
                if(grid[i][j]==1) break;
                t++;
            }
            vec[i] = t;
        }
        int res = 0;
        for(int i=0; i<n; i++) {
            if(vec[i]>=(n-i-1)) continue;
            int t = -1;
            for(int j=i+1; j<n; j++) {
                if(vec[j]>=(n-i-1)) {
                    t = j;
                    break;
                }
            }
            if(t==-1) return -1;
            res += (t-i);
            int cur = vec[t];
            for(int j=t-1; j>=i; j--) {
                vec[j+1] = vec[j];
            }
            vec[i] = cur;
        }
        return res;
    }
};
