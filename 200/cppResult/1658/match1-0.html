class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> t(n);
        for(int i = 0; i < n; i++) {
            for(int j = n - 1; j >= 0; j--) {
                if(grid[i][j]) break;
                t[i]++;
            }
        }
        vector<int> id(n);
        for(int i = 0; i < n; i++) id[i] = i;
        int ans = 0;
        for(int i = 0; i < n - 1; i++) {
            int r = -1;
            for(int j = i; j < n; j++) {
                if(t[id[j]] >= n - i - 1) {
                    r = j;
                    break;
                }
            }
            if(r == -1) return -1;
            for(int j = r - 1; j >= i; j--) {
                swap(id[j], id[j + 1]);
                ans++;
            }
        }
        return ans;
    }
};
