class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int m=grid.size();

        int n=grid[0].size();
        vector<int> rowzero(grid.size());
        
        
        for(int i=0;i<grid.size();i++){
            
            int k=0;
            for(int j=n-1;j>=0;j--){
                if(grid[i][j]!=0)break;
                k++;
            }
            //cout<<k<<endl;
            rowzero[i]=k;
        }
        
        int ans=0;
        for(int i=0;i<grid.size()-1;i++){
            //cout<<n-i-1<<endl;
            if(rowzero[i]>=n-i-1)continue;
            else{

                int target=-1;
                for(int j=i+1;j<grid.size();j++){
                    if(rowzero[j]>=n-i-1){
                        //cout<<j<<endl;
                        target= j;
                        break;
                    }   
                }
                if(target==-1)return -1;
                
                ans+= target-i;
                int tmp= rowzero[target];
                for(int j=target;j>i;j--){
                    rowzero[j]=rowzero[j-1];
                    //cout<<rowzero[j]<<endl;
                }
                rowzero[i]=tmp;
            }
            
        }
        return ans;
    }
};
