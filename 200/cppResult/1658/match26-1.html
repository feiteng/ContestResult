class Solution {
public:
    int minSwaps(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> len(n);
        for (int i = 0; i < n; ++i) {

            for (int j = n - 1;  j >= 0; --j) {
                if (grid[i][j] == 0)
                    len[i]++;
                else 
                    break;
            }
        }

        int counter = 0;
        for (int i = 0; i < n; ++i) {
            bool flag = true;

            for (int j = i; j < n; ++j) {
                if (len[j] + i >= n - 1) {
                    flag = false;
                    for (int k = j; k > i; --k) {
                        swap(len[k], len[k - 1]);
                        counter++;
                    }
                    break;
                }
            }
            if (flag) {
                cout << i << endl;
                return -1;
            }
        }
        return counter;
    }
};
