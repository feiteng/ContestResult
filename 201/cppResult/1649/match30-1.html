class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int, int> Pos;
        Pos[0] = -1;
        
        int sum = 0;
        int lastIdx = -1;

        int result = 0;
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
            if (Pos.find(sum-target) != Pos.end() && Pos[sum-target] >= lastIdx) {
                result++;
                lastIdx = i;
            }
            Pos[sum] = i;
        }
        
        return result;
    }
};
