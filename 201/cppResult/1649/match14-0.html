class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int len = nums.size();
        vector<int> dp(len, 0);
        unordered_map < int , int > pref_sum;
        pref_sum[0] = -1;
        int pref = 0;
        for(int i = 0; i < len; ++i) {
            pref += nums[i];
            if(i > 0) {
                dp[i] = dp[i - 1];
            }
            int reqd = pref - target;
            if(pref_sum.find(reqd) != pref_sum.end()) {
                int indx = pref_sum[reqd];
                dp[i] = max(dp[i], 1 + (indx >= 0 ? dp[indx] : 0));
            } 
            pref_sum[pref] = i;
        }
        return dp[len - 1];
    }
};
