class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        //res[i] is the max none overlapping arrary from [0, i] with sum to target
        int n = nums.size();
        vector<int> res(n + 1, 0);
        
        // start from nums[i], find the the largets previous sum[j] = sum[i] - target
        vector<int> sum(n + 1, 0);
        map<int, int> m;

        m[0] = 0;
        for(int i = 0; i < n; ++i) {
            sum[i + 1] = sum[i] + nums[i];
            // try to find if any previous sum 9s sum[i + 1] - target
            res[i + 1] = res[i];
            if(m.find(sum[i + 1] - target) != m.end()) {
                int pre = m[sum[i + 1] - target];
                res[i + 1] = max(res[i + 1], res[pre] + 1);
            }
            
            m[sum[i + 1]] = i + 1;
        }
        
        return res[n];
    }
};
