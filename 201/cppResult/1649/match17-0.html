class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int m = nums.size();
        unordered_map<int, int> seen;
        seen[0] = -1;
        // cout << "  ";
        // for (auto c: nums) cout << c << " ";
        // cout << endl;
        
        vector<int> dp(m+1);
        int max_so_far = 0;
        int sum = 0;
        for (int i = 0; i < m; ++i) {
            sum += nums[i];
            // cout << nums[i] << ": ";
            auto it = seen.find(sum - target);
            if (it != seen.end()) {
                max_so_far = max(max_so_far, 1 + dp[it->second+1]);
                // cout << it->second << " -> ";
            }
            // cout << sum << " " << max_so_far << endl;
            dp[i+1] = max_so_far;
            seen[sum] = i;
        }
        
        // for (auto c: dp) cout << c << " ";
        // cout << endl;
        
        return dp[m];
    }
};
