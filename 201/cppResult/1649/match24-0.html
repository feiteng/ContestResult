class Solution {
public:
	#define maxn 100010
	int sum[maxn],f[maxn];
	map<int,int>mp;
    int maxNonOverlapping(vector<int>& a, int k) {
		int n=a.size();
		sum[1]=a[0];
		for(int i=2;i<=n;i++)sum[i]=sum[i-1]+a[i-1];
		mp.clear();mp[0]=0;
		memset(f,0,sizeof(f));
		for(int i=1;i<=n;i++){
			f[i]=f[i-1];
//			printf("%d %d : ",i,sum[i]);
			if(mp.count(sum[i]-k)){
				f[i]=max(f[i],mp[sum[i]-k]+1);
			}
			if(mp.count(sum[i]))mp[sum[i]]=max(mp[sum[i]],f[i]);
			else mp[sum[i]]=f[i];
//			printf(" -> %d\n",f[i]);
		}
		return f[n];
    }
}K;
