class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        map<int,int> mp ;
        mp[0] = 1 ;
        int ans = 0 ;
        int f = 0 ;

        for(auto i : nums){
            f += i ;
            if(mp.count(f-target)){
                f = 0 ;
                mp.clear() ;
                ans++ ;
            }
            mp[f]++ ;
        }
        return ans ;
    }
};
