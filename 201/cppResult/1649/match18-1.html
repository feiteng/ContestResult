class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        set<int> S;
        S.insert(0);
        int n = (int)nums.size();
        int sum = 0, ans = 0;

        for(int i = 0; i < n; i ++) {
            sum += nums[i];
            if(S.find(sum - target) != S.end()) {
                S.clear();
                S.insert(0);
                sum = 0;
                ans ++;
                //cout << "index " << i << endl;
            } else {
                S.insert(sum);
            }
        }
        return ans;
    }
};
