class Solution {
public:
    int maxNonOverlapping(vector<int>& v, int tar) 
    {
        unordered_map<int,int> mp;
        mp[0] = 1;
        int ans = 0;
        int cur = 0;
        for(int x: v)
        {
            cur += x;
            if(mp[cur-tar]==1)
            {
                ans++;
                mp.clear();
                mp[0] = 1;
                cur = 0;
            }
            else
                mp[cur] = 1;
            //cout << cur << " " << ans << endl;
        }
        //if(mp[tar]==1)
            //ans++;
        return ans;
    }
};
