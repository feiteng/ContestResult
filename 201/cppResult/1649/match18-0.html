class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int i;
		int N = nums.size();
		unordered_set<int> Set;
		int Sum = 0;
		int Result = 0;
		
		Set.insert(0);
		
		for(i = 0; i < N; i++)
		{
			Sum += nums[i];
			
			if(Set.count(Sum - target))
			{
				Result++;
				Set.clear();
				Set.insert(0);
				Sum = 0;
			}
			else
			{
				Set.insert(Sum);
			}
		}
		
		return Result;
    }
};
