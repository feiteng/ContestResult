class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
      int n = nums.size();
      vector<int> DP(n+1);
      unordered_map<int, int> pos;
      int subsum = 0;
      pos[0] = -1;
      for (int i=0; i<n; ++i) {
        DP[i+1] = DP[i];
        subsum += nums[i];
        if (pos.count(subsum-target)) {
          int j = pos[subsum - target];
          // cout<<j<<"\t"<<i<<"\t"<<subsum<<endl;
          DP[i+1] = max(DP[i+1], DP[j+1]+1);
        }
        pos[subsum] = i;
      }
      
      return DP.back();
    }
};
