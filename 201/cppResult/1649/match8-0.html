class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        map<int, int> M;
        int res = 0;
        int sum = 0;
        M[0]=1;
        for(int i=0; i<nums.size(); i++){
            sum+=nums[i];
            if(M.count(sum-target)){
                res++;
                sum=0;
                M.clear();
                M[0]=1;
                continue;
            }
            M[sum]=true;
        }
        return res;
    }
};
