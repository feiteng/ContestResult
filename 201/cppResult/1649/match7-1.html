class Solution {
public:
    int maxNonOverlapping(vector<int>& a, int t) {
        map<int, int> F;
        int n = a.size();
        F[0] = 0;
        int s = 0;
        int la = 0;
        int S = 0;
        for (int i = 0; i < n; i ++) {
            s += a[i];

            if (F.find(s-t) != F.end()) {
                int z = F[s-t];
                if (z >= la) {
                    S++; la = i+1;
                }
            }
            F[s] = i+1;
        }
        return S;
    }
};
