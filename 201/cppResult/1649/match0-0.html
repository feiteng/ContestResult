class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int ans = 0;
        unordered_map<int, int> mp;
        mp[0] = -1;
        int t = 0;
        int last_i = -1;
        int n = nums.size();
        for (int i = 0; i < n; i ++) {
            t += nums[i];
            if (mp.count(t - target) && mp[t - target] >= last_i) {
                ans ++;
                last_i = i;
            }
            // cout << i << " " << ans << endl;
            mp[t] = i;
        }
        return ans;
        
    }
};
