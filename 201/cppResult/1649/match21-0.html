class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int ans = 0;
        set<int> tab;
        tab.insert(0);
        int cur = 0;
        
        for(int x : nums) {
            cur += x;
            int rest = cur - target;
            if(tab.count(rest)) {
                ans++;
                tab.clear();
                tab.insert(0);
                cur = 0;
            }
            else {
                tab.insert(cur);
            }
        }
        
        return ans;
    }
};
