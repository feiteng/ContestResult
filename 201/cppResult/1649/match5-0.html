class Solution {
public:
    map<int,int>mp;
    int maxNonOverlapping(vector<int>& nums, int target) {
        int len = nums.size();
        int now = 0,ans = 0;
        mp[0] = 1;
        for(int i = 0;i < len;i++) {
            now = (now + nums[i]);
            if(mp[now - target]) {
                ans++;
                mp.clear();
                now = 0;
                mp[0] = 1;
            } else {
                mp[now] = 1;
            }
        }
        return ans;
    }
};
