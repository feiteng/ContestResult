class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int, int> um;
        um[0] = -1;
        int index = 0;
        int res = 0;
        int c = 0;
        for (int i = 0; i < nums.size(); i++) {
            c += nums[i];
            if (um.find(c - target) != um.end()) {
                if (um[c - target] + 1 >= index) {
                    res++;
                    index = i + 1;
                } 
            }
            
            um[c] = i;
        }
        return res;
    }
};
