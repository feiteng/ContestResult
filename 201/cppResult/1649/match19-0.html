class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> st;
        st.insert(0);
        int acc = 0;
        int ans = 0;
        for(int i=0;i<nums.size();i++){
            acc+=nums[i];
            int goal = acc-target;
            if(st.find(goal)!=st.end()){
                //printf("i=%d nums[i]=%d found\n", i, nums[i]);
                ans++;
                st.clear();
                st.insert(0);
                acc=0;
            }
            else{
                st.insert(acc);
            }
        }
        return ans;
    }
};
