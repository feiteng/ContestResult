class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int n = nums.size();
        vector<int> f(n + 1, 0);
        unordered_map<int, int> h; h[0] = -1;
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            f[i + 1] = max(f[i + 1], f[i]);

            sum += nums[i];
            if (h.find(sum - target) != h.end()) {
                int idx = h[sum - target];
                f[i + 1] = max(f[i + 1], (idx >= 0 ? f[idx] : 0) + 1);
            }
            h[sum] = i + 1;
        }
        return f[n];
    }
};
