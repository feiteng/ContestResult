class Solution {
public:
    int maxNonOverlapping(vector<int>& A, int target) {
        int n = A.size();
        set<int> S;
        int sum = 0;
        int ans = 0;
        S.insert(0);

        for (int i = 0; i < n; i++) {
            sum += A[i];
         //   cout << sum << endl;
            if (S.count(  sum- target) != 0) {
                ans ++; 
                S.clear();
                S.insert(0);
                sum = 0;
            } else {
                S.insert(sum);
            }
        }
        return ans;
    }
};
