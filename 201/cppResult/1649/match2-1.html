class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> s;
        s.insert(0);
        int sum = 0;
        int ans = 0;        
        for(int num: nums) {
            sum += num;
            if(s.find(sum - target) != s.end()) {
                ans ++;
                s.clear();
                sum = 0;
                s.insert(0);
            }
            else s.insert(sum);
        }
        return ans;
    }
};
