#include <bits/stdc++.h>
using namespace std;


class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int n = nums.size();
        vector<int> jump(n, -2);
        map<int, int> lasttime;
        lasttime[0] = -1;

        int running = 0;
        for (int i = 0; i < n; ++i) {
            running += nums[i];
            if (lasttime.count(running-target)) {
                jump[i] = lasttime[running-target];
                // cout << i << " " << jump[i] << endl;
            }
            lasttime[running] = i;
        }
        vector<int> dp(n);
        for (int i = 0; i < n; ++i) {
            dp[i] = 0;
            if (i-1 >= 0) dp[i] = max(dp[i], dp[i-1]);
            if (jump[i] != -2) {
                if (jump[i] == -1) {
                    dp[i] = max(dp[i], 1);
                } else {
                    dp[i] = max(dp[i], dp[jump[i]]+1);
                }
            }
            // cout << i << ": " << dp[i] << endl;
        }
        return dp[n-1];
    }
};
