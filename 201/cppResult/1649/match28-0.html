class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> s;
        int offset = 0;
        int ans = 0;
        for (int num: nums) {
                offset += num;
                // insert num
                s.insert(num - offset);
            
            
            // a value in s + offset == target
            if (s.find(target - offset) != s.end()) {
                ans++;
                s.clear();
                offset = 0;
            }
        }
        return ans;
        
    }
};
