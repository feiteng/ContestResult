class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> sums;
        sums.insert(0);
        int cur_sum = 0;
        int ans = 0;
        for (int n: nums){
            cur_sum = cur_sum + n;
            if (sums.find(cur_sum - target) != sums.end()){
                ans++;
                sums.clear();
                cur_sum = 0;
                sums.insert(0);
            }
            else
                sums.insert(cur_sum);
        }
        return ans;
    }
};
