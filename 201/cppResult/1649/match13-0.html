class Solution {
public:
    int maxNonOverlapping(vector<int>&v, int t) {
        set <int> s;
        s.insert(0);
        int sum = 0, ans = 0;
        for (int x: v) {
            sum += x;
            if (s.count(sum-t)) {
                ans++;
                s.clear();
                s.insert(0);
                sum = 0;
            } else s.insert(sum);
        }
        return ans;
    }
};
