class Solution {
public:
  int maxNonOverlapping(vector<int>& nums, int target) {
    set<int> seen;
    int sum=0,ans=0;
    for(int v:nums){
      seen.emplace(sum);
      sum+=v;
      if(seen.count(sum-target)){
        ans++;
        seen.clear();
      }
    }
    return ans;
  }
};
