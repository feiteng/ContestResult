class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int>st;
        st.insert(0);
        int sum = 0, res=0;
        for (auto num: nums) {

            sum+=num;
            if (st.find(sum-target)!=st.end()) {
                res++;
                st.clear();
                sum=0;
                st.insert(0);
            } else
                st.insert(sum);
                
        }
        return res;
    }
};
