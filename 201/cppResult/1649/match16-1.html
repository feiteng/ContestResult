class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> s;
        s.insert(0);
        int cumsum = 0, ans = 0;
        for (auto n : nums) {
            cumsum += n;

            int req = cumsum - target;
            if (s.count(req)) {
                ans++;
                s.clear();
                s.insert(0);
                cumsum = 0;
            } else {
                s.insert(cumsum);
            }
        }
        return ans;
    }
};
