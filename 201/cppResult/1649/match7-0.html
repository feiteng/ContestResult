class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        map<int, int> dp;
        int len = nums.size();
        dp[0]=-1;
        int last = 0;
        int lastIdx = -2;
        int res = 0;
        for(int i=0;i<len;i++)
        {
            last += nums[i];
            int tmp = last - target;
            if(dp.find(tmp) != dp.end())
            {
                int start = dp[tmp];
                if(start>=lastIdx)
                {
                    res++;
                    lastIdx = i;
                }
            }
            dp[last] = i;
        }
        return res;
    }
};
