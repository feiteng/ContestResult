class Solution {
public:
	static const int N=100010;

	int sum[N],f[N];
	map<int,int>s;
    int maxNonOverlapping(vector<int>& a, int k) {
		int n=a.size();
		sum[1]=a[0];
		for(int i=2;i<=n;i++)sum[i]=sum[i-1]+a[i-1];
		s.clear();s[0]=0;
		memset(f,0,sizeof(f));
		for(int i=1;i<=n;i++){
			f[i]=f[i-1];
			if(s.count(sum[i]-k)){
				f[i]=max(f[i],s[sum[i]-k]+1);
			}
			if(s.count(sum[i]))s[sum[i]]=max(s[sum[i]],f[i]);
			else s[sum[i]]=f[i];
		}
		return f[n];
    }
}K;
