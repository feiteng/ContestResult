class Solution {
public:
  int maxNonOverlapping(vector<int>& nums, int target) {
    int N = nums.size();
    vector<int> A;
    A.push_back(0);
    partial_sum(nums.begin(), nums.end(), back_inserter(A));
    vector<int> best(N+1);
    map<int, int> M;
    M[0] = 0;
    for (int i = 1; i <= N; i++) {
      best[i] = best[i-1];
      if (M.count(A[i] - target)) {
        best[i] = max(best[i], best[M[A[i]-target]] + 1);
      }
      M[A[i]] = i;
    } 
    return best.back();
  }
};
