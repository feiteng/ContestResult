class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        map<int, int> maps;
        int cnt = 0, ans = 0;
        maps[0] = 1;

        for (auto t : nums) {
            cnt += t;
            if (maps.find(cnt - target) != maps.end()) {
                maps.clear();
                maps[0] = 1;
                cnt = 0;
                ans++;
            } else {
                maps[cnt] = 1;
            } 
        }
        return ans;
    }
};
