class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int ans = 0;
        unordered_set<int> s;
        s.insert(0);
        int sum = 0;
        for (int a: nums) {
            sum += a;
            if (s.find(sum - target) != s.end()) {
                ans++;
                s.clear();
                s.insert(0);
                sum = 0;
            } else {
                s.insert(sum);
            }
        }
        return ans;
    }
};
