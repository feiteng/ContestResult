class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int, int> pos;
    int len = (int)nums.size(), sum = 0, ans = 0, R = -1;
    pos[0] = -1;
    for(int i = 0; i < len; i++)
    {
        sum += nums[i];
        if(pos.count(sum - target) && pos[sum - target] >= R)
        {
            ans++;
            R = i;
        }
        pos[sum] = i;
    }
    return ans;
    }
};
