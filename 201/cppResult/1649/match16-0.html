class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_set<int> s_s;
        s_s.insert(0);
        int sum = 0, ans = 0;
        for (const int& each: nums)
        {
            sum += each;
            if (s_s.find(sum - target) != s_s.end())
            {
                ans += 1;
                sum = 0;
                s_s.clear();
                s_s.insert(0);
            }
            else
            {
                s_s.insert(sum);
            }
        }
        return ans;
    }
};
