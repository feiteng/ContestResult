class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        map<int,int> m;
        int s = 0;
        int n = nums.size();
        int t = target;
        int last = -2;
        m[0] = -1;

        int ans = 0;
        for (int i = 0; i < n; i++) {
            s += nums[i];
            if (m.count(s - target) && m[s-target] >= last ) {
                ans++;
                last = i;
            } 
            m[s] = i;
        }
        return ans;
    }
};
