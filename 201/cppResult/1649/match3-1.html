class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        int n=nums.size();
        int ans=0;
        set<int> s;
        s.insert(0);
        int now=0;
        for(int i=0;i<n;++i)
        {
            now+=nums[i];

            // cout<<i<<" : "<<now<<endl;
            if(s.find(now-target)!=s.end())
            {
                ++ans;
                now=0;
                s.clear();
                s.insert(0);
            }
                        s.insert(now);
        }
        return ans;
    }
};
