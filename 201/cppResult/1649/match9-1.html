class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int,int> ump;
        ump[0] = 1;
        int sum = 0;
        int ans = 0;
        for(int i=0;i<nums.size();i++)

        {
            sum += nums[i];
            if(ump.find(sum-target) != ump.end())
            {
                ans++;
                ump.clear();
                ump[0] = 1;
                sum = 0;
            }
            else
                ump[sum]++;
        }
        return ans;
    }
};
