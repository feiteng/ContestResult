class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);

        int m=cuts.size();
        sort(cuts.begin(), cuts.end());
        
        vector<vector<int>> dp(m+1, vector<int>(m+1, n*m));
        for(int i=0;i<m;i++){
            dp[i][i+1]=0;
        }

     
        cout<<endl;
        for(int i=m;i>=0;i--){
            for(int j=i+1;j<m;j++){
                for(int k=i+1;k<j;k++){
                    dp[i][j]=min(dp[i][j], cuts[j]-cuts[i] + dp[i][k] + dp[k][j]);
                }
            }
        }
        return dp[0][m-1];
    }
};
