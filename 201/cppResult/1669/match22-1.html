typedef long long int lli;
lli dp[105][105];

class Solution {
public:
    lli dfs(int i, int j, int x, int y, vector <int>& v){
        
        if(x > y) return 0;
        //cout << x << " " << y << endl;
        if(dp[x][y] != -1) return dp[x][y];

        lli ret = INT_MAX;
        for(int mid = x; mid <= y; mid ++){
            ret = min(ret, dfs(i, v[mid], x, mid - 1, v) + dfs(v[mid], j, mid + 1, y, v) + j - i);
        }
        return dp[x][y] = ret;
        
    }
    int minCost(int n, vector<int>& cuts) {
        sort(cuts.begin(), cuts.end());
        memset(dp, -1, sizeof dp);
        return dfs(0, n, 0, (int)cuts.size() - 1, cuts);
    }
};
