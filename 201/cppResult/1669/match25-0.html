class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        int len = cuts.size();
        sort(cuts.begin(), cuts.end());
        vector<vector<int>> dp(len+1, vector<int>(len+1, 0));
        for(int d = 1; d<=len; d++) {
            for(int i=0; i<=len-d; i++) {
                int j = i+d;
                int right = n;
                if(j<len) {
                    right = cuts[j];
                }
                int left = 0;
                if(i>0) {
                    left = cuts[i-1];
                }
                int cur_len = right-left;
                int res = 0;
                int cur = INT_MAX;
                for(int k=i; k<j; k++) {
                    int temp = dp[i][k] + dp[k+1][j] + cur_len;
                    if(temp<cur) cur = temp;
                }
                dp[i][j] = cur;
            }
        }
        return dp[0][len];
    }
};
