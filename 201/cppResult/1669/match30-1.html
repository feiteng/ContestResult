class Solution {
public:
	int minCost(int n, vector<int>& cuts) {
		cuts.push_back(0);
		cuts.push_back(n);
		sort(cuts.begin(), cuts.end());
		int l = cuts.size();
		vector<vector<int>> dp(l, vector<int>(l, 1e9));
        for(int i = 0; i < l; i++) {
            dp[i][i] = 0;

            if(i + 1 < l) dp[i][i+1] = 0;
        }
		for(int j = 2; j < l; j++) {
			for(int i = 0; i + j < l; i++) {

				for(int k = i + 1; k <= i + j; k++) {
					dp[i][i+j] = min(dp[i][i+j], dp[i][k] + dp[k][i+j] + cuts[i+j] - cuts[i]);
				}
                // printf("%d %d %d\n", i, i + j, dp[i][i + j]);
			}
		}
		return dp[0][l-1];
	}
};
