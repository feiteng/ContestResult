class Solution {
public:
    int minCost(int m, vector<int>& cuts) {
        sort(cuts.begin(), cuts.end());
        cuts.insert(cuts.begin(), 0);
        cuts.push_back(m);
        
        int n = cuts.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for(int l=3; l<=n; l++){
            for(int i=0; i<=n-l; i++){
                int j = i+l-1;
                int minCost = INT_MAX, currCut = cuts[j]-cuts[i];

                for(int p=i+1; p<j; p++){
                    int currCost = dp[i][p] + dp[p][j] + currCut;
                    if(currCost < minCost)
                        minCost = currCost;
                }
                dp[i][j] = minCost;
            }
        }
        return dp[0][n-1];
    }
};
