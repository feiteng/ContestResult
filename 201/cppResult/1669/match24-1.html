class Solution {
public:
    int minCost(int m, vector<int>& cuts) {
        
        cuts.push_back(0);
        cuts.push_back(m);
        
        sort(cuts.begin(),cuts.end());
        
        int n=cuts.size();
        
        vector<vector<int>> DP(n,vector<int> (n,1e8));
        
        for(int i=1;i<=n;i++)
        {
            for(int j=0;j+i-1<n;j++)//subarray start point
            {
                int start=j;
                int end=j+i-1;
                
                if(i==1 or i==2)
                {
                    DP[start][end]=0;
                }
                

                for(int cut_pos=start+1;cut_pos<end;cut_pos++)//cut points
                {
                    DP[start][end]=min(DP[start][end],DP[start][cut_pos]+DP[cut_pos][end]+cuts[end]-cuts[start]);
                }
            }
        }
        
        return DP[0][n-1];
    }
};
