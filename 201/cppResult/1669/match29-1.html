class Solution {
public:
    
    
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        sort(cuts.begin(), cuts.end());

        n = cuts.size();
        vector<vector<int>> v(n, vector<int> (n, 0x3f3f3f3f));
        for (int i = 0; i < n - 1; i++) {
            v[i][i + 1] = 0;
        }
        for (int i = 2; i <= n - 1; i++) {
            for (int j = 0; j + i < n; j++) {
                int l = cuts[j + i] - cuts[j];

                for (int k = j + 1; k < j + i; k++) {
                    v[j][j + i] = min(v[j][j + i], v[j][k] + v[k][j + i] + l);
                }
            }
        }
        return v[0][n - 1];
    }    
    
};
