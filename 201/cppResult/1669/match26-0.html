class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        cuts.insert(cuts.begin(),0);
        cuts.push_back(n);
        sort(cuts.begin(),cuts.end());
        vector<vector<int>>dp(cuts.size(),vector<int>(cuts.size(),INT_MAX));
        for(int i=0;i < dp.size();i++){
            dp[i][i] = 0;
            // cout << cuts[i]<<" ";
        }
        cout<<endl;
        for(int i=0;i < dp.size()-1;i++){
            dp[i][i+1] = 0;
        }
        
        for(int l=2;l <= cuts.size();l++){
            for(int start=0;start+l < cuts.size();start++){
                int endindex = start+l;
                for(int cindex = start+1;cindex < endindex;cindex++){
                    dp[start][endindex] = min(dp[start][endindex],dp[start][cindex]+dp[cindex][endindex]+cuts[endindex]-cuts[start]);
                }
                // cout << cuts[start]<<' '<<cuts[endindex]<<" "<<dp[start][endindex]<<endl;
            }
        }
        return dp[0][cuts.size()-1];
    }
    
    

};
