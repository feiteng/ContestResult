class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        if (cuts.size() == 1) return n;
        sort(cuts.begin(), cuts.end());
        cuts.insert(cuts.begin(), 0);
        cuts.push_back(n);
        int num = cuts.size();

        vector<vector<int>> dp(num, vector<int>(num, INT_MAX));
        for (int i = 0; i < num - 1; ++i) {
            dp[i][i + 1] = 0;
        }
        for (int i = 0; i < num - 2; ++i) {
            dp[i][i + 2] = cuts[i + 2] - cuts[i];
        }
        for (int kk = 3; kk < num; ++kk) {
            for (int st = 0; st < num - kk; ++st) {
                int ed = st + kk;
                for (int mid = st + 1; mid < ed; ++mid) {
                    dp[st][ed] = min(dp[st][ed], dp[st][mid] + dp[mid][ed] + cuts[ed] - cuts[st]);
                }
            }
        }
        
        return dp[0][num - 1];
    }
};
