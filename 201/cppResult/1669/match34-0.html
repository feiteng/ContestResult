class Solution {
public:
    int place[150];
    int dp[150][150];
    int m;
    
    int solve(int a, int b){
        if(b-a <= 1) return 0;
        if(dp[a][b])return dp[a][b];
        int mn = 1e9;
        for(int i = a+1; i < b; i++){
            mn = min(solve(a,i) + solve(i,b), mn);
        }
        return dp[a][b] = mn + place[b] - place[a];
    }
    
    int minCost(int n, vector<int>& cuts) {
        sort(cuts.begin(), cuts.end());
        place[0] = 0;
        for(int i = 0; i < cuts.size(); i++){
            place[i+1] = cuts[i];;
        }
        place[cuts.size()+1] = n;
        int m = cuts.size() +2;
        return solve(0,m-1);
    }
};
