class Solution {
public:
  
  vector<int> vec;
  int f[104][104];
  
  int dfs(int l, int r) {
    if (f[l][r] != -1) return f[l][r];
    if (l >= r - 1) return 0;
    int ans = 0x3f3f3f3f;
    for (int i = l + 1; i < r; i ++) {
      ans = min(ans, dfs(l, i) + dfs(i, r) + vec[r] - vec[l]);
    }
    // cout << l << " " << r << " " << ans << endl;
    return f[l][r] = ans;
  }
  
  int minCost(int n, vector<int>& cuts) {
    int ed = 0;
    memset(f, -1, sizeof(f));
    cuts.push_back(0); cuts.push_back(n);
    sort(cuts.begin(), cuts.end());
    vec = cuts;
    return dfs(0, cuts.size() - 1);
  }
};
