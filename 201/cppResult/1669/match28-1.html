class Solution {
public:
    int dp(vector<vector<int>>& mem, vector<int>& cuts, int i, int j, int l, int r) {
        if (i > j)  return 0;
        if (i == j) return r - l;
        if (mem[i][j] != -1)    return mem[i][j];

        mem[i][j] = 1 << 25;
        for (int c = i; c <= j; c++) {
            mem[i][j] = min(mem[i][j], dp(mem, cuts, i, c - 1, l, cuts[c]) + dp(mem, cuts, c + 1, j, cuts[c], r));
        }
        mem[i][j] += r - l;
        return mem[i][j];
    }
    
    int minCost(int n, vector<int>& cuts) {
        int c = cuts.size();
        sort(cuts.begin(), cuts.end());
        vector<vector<int>> mem(c, vector<int>(c, -1));
        int ret = dp(mem, cuts, 0, c - 1, 0, n);
        // for (int i = 0; i < c; i++) {
            // for (int j = 0; j < c; j++) cout << mem[i][j] << " ";
            // cout << endl;
        // }
        return ret;
    }
};
