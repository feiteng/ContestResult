class Solution {
public:
    
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        sort(cuts.begin(), cuts.end());
        int m = cuts.size();
        vector<vector<int>> dp(m, vector<int>(m, n * m));
        for (int i = 0; i < m - 1; ++i)
        {
            dp[i][i + 1] = 0;
        }
        for (int i = 2; i < m; ++i)
        {
            for (int l = 0; l + i < m; ++l)
            {
                int r = l + i;
                for (int k = l + 1; k < r; ++k)
                {
                   dp[l][r] = min(dp[l][r], dp[l][k] + dp[k][r] + cuts[r] - cuts[l]); 
                }
                
            }
        }
        return dp[0][m - 1];
    }
};
