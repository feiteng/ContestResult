class Solution {
public:
    int minCost(int n, vector<int>& c) {
        sort(c.begin(), c.end());
        c.insert(c.begin(), 0);
        c.push_back(n);
        int m = c.size();
        vector<vector<int>> dp(m+5, vector<int>(m+5, 1e8));
        for (int i = 0; i < m; i++) {
            dp[i][i] = 0; dp[i][i+1] = 0;

        }
        
        for (int l = 2; l <= m; l++) {
            for (int i = 0; i < m; i++) {
                int j = i+l;
                if (j >= m) break;

                for (int k = i+1; k < j; k++) {
                    dp[i][j] = min(dp[i][j], dp[i][k]+dp[k][j] + c[j]-c[i]);
                }
            }
        }
        
        return dp[0][m-1];
    }
};
