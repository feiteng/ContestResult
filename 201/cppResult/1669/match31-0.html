class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        int len = cuts.size();
        vector<vector<int>> segCost(len + 2, vector<int>(len + 2, 0));
        
        sort(cuts.begin(), cuts.end());
        for (int l = 2; l < len + 2; l++) {
            for (int i = 0; i + l < len + 2; i++) {
                int cost;
                
                if (i > 0 && i + l < len + 1) {
                    cost = cuts[i + l - 1] - cuts[i - 1];
                } else if (i == 0 && i + l == len + 1) {
                    cost = n;
                } else if (i == 0) {
                    cost = cuts[i + l - 1];
                } else {
                    cost = n - cuts[i - 1];
                }
                segCost[i][i + l] = INT_MAX;
                
                for (int j = i + 1; j < i + l; j++) {
                    segCost[i][i + l] = min(segCost[i][i + l], cost + segCost[i][j] + segCost[j][i + l]);
                }
            }
        }
        
        return segCost[0][len + 1];
    }
};
