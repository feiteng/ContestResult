class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        sort(cuts.begin(), cuts.end());
        int sz = cuts.size();
        vector<vector<int>> dp(sz, vector<int>(sz, 0x3f3f3f3f));
        // dp[i][j]: stick[i...j] inclusive cost;
        for(int i = 0; i < sz-1; i ++) {
            dp[i][i+1] = 0;
        }
        for(int len = 2; len < sz; len ++) {
            for(int i = 0; i + len < sz; i ++) {
                int left = i;
                int right = i + len;

                for(int j = left + 1; j < right; j ++) {
                    dp[left][right] = min(dp[left][right], dp[left][j] + dp[j][right] + cuts[right] - cuts[left]);
                }
            }
        }
        return dp[0][sz-1];
    }
};
