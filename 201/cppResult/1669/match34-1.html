#include <bits/stdc++.h>
using namespace std;


class Solution {
public:
    int N;
    vector<int> C;
    vector<vector<int>> dp;
    int f(int l, int r) {
        if (l+1==r) return 0;
        if (dp[l][r] != -1) return dp[l][r];
        int ans = 1e9;
        for (int k = l+1; k < r; ++k) {
            ans = min(ans, f(l, k)+f(k,r)+(C[r]-C[l]));
        }
        // cout << l << " " << r << " " << ans << endl;
        return dp[l][r] = ans;
    }
    int minCost(int n, vector<int>& cuts) {
        sort(begin(cuts), end(cuts));
        C.push_back(0);
        C.insert(end(C), begin(cuts), end(cuts));
        C.push_back(n);
        N = C.size();
        dp.assign(N, vector<int>(N, -1));
        // for (int c : C) {
        //     cout << c << endl;
        // }
        return f(0, C.size()-1);
    }
};
