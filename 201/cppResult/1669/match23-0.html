class Solution {
public:
    int dp[102][102];
    int solve(int l,int r,vector<int>& a){
        //if(l>=r) return 0;
        if(r-l<=1) return 0;
        if(dp[l][r]!=-1) return dp[l][r];
        int mn=1e9;
        for(int i=l+1;i<r;i++){
            mn=min(mn,solve(l,i,a)+solve(i,r,a));
        }
        return dp[l][r]=mn+a[r]-a[l];
    }
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        memset(dp,-1,sizeof(dp));
        sort(cuts.begin(),cuts.end());
        return solve(0,cuts.size()-1,cuts);
    }
};
