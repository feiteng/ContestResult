class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        cuts.insert(cuts.begin(),0);
        cuts.insert(cuts.end(),n);
        sort(cuts.begin(),cuts.end());
        int len = cuts.size();
        vector<vector<int>> dp(len,vector<int> (len));
        for(int step=1;step<len;++step)
            for(int i=0,j=i+step;i<len-step;++i,++j)
            {
                if(j-i==1) dp[i][j]=0;
                else

                {
                    int tmp=INT_MAX;
                    for(int k=i+1;k<j;++k)
                    {
                        tmp=min(tmp,dp[i][k]+dp[k][j]);
                    }
                    dp[i][j]=tmp+cuts[j]-cuts[i];
                }
            }
        // for(int i=0;i<len;++i)
        // {
        //     for(int j=0;j<len;++j)
        //     {
        //         cout<<dp[i][j]<<' ';
        //     }
        //     cout<<endl;
        // }
        return dp[0][len-1];
        
    }
};
