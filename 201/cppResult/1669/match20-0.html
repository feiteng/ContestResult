class Solution {
public:
    int dp[105][105];
    int solve(int l, int r, vector<int>& cuts) {
        if(l <= 0 || r >= cuts.size() - 1|| l > r) return 0;
        if(dp[l][r] != -1) return dp[l][r];
        int res = cuts[r + 1] - cuts[l - 1], mx = INT_MAX;
        for(int i = l; i <= r; i++) {
             mx = min(mx,solve(l,i-1,cuts) + solve(i + 1,r,cuts));
        }
        return dp[l][r] = res + mx;
    }
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        int sz = cuts.size();
        sort(cuts.begin(),cuts.end());
        memset(dp,-1,sizeof(dp));
        return solve(1,sz-2,cuts);
    }
};
