class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        int m = cuts.size();
        sort(cuts.begin(), cuts.end());
        vector<vector<int>> f(m+2, vector<int>(m+2));
        for (int k = 2; k < m+2; k++) {
            for (int s = 0; s + k < m+2; s++) {
                int e = s + k;
                f[s][e] = INT_MAX;
                int len = (e == m+2-1 ? n : cuts[e-1]) - (s == 0 ? 0: cuts[s-1]);
                for (int i = s + 1; i <= e - 1; i++) {
                    f[s][e] = min(f[s][e], f[s][i] + f[i][e] + len);
                }
            }
        }
        return f[0][m+2-1];
    }

};
