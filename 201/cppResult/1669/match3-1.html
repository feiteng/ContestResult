class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
        cuts.push_back(0);
        cuts.push_back(n);
        sort(cuts.begin(), cuts.end());
        int l = cuts.size(), t;
        vector<vector<int>> m(l, vector<int>(l));
        for (int i = 2; i < l; ++i) {
            for (int j = 0; j + i < l; ++j) {
                t = 2e9;

                for (int k = 1; k < i; ++k) {
                    t = min(t, m[j][j + k] + m[j + k][j + i]);
                }
                m[j][j + i] = t + cuts[j + i] - cuts[j];
            }
        }
        return m[0][l - 1];
    }
};
