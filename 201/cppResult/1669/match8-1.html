class Solution {
public:
    int minCost(int n, vector<int>& cuts) {
      int m = cuts.size();
      vector<vector<int>> costs(m+2, vector<int>(m+2, INT_MAX));
      cuts.push_back(0);
      cuts.push_back(n);
      sort(cuts.begin(), cuts.end());
      
      for (int i=0; i<m+1; ++i) {
        costs[i][i+1] = 0;
        if (i<m) {
          costs[i][i+2] = cuts[i+2]-cuts[i];
          // cout<<i<<"\t"<<costs[i][i+2]<<endl;
        }

      }
      
      for (int l = 3; l <= m+1; ++l) {
        for (int i=0; i+l<m+2; ++i) {
          int j = i+l;
          for (int k = i+1; k<j; ++k) {
            costs[i][j] = min(costs[i][j], costs[i][k] + costs[k][j]);
          }
          costs[i][j] += (cuts[j] - cuts[i]);
          // cout<<i<<"\t"<<j<<"\t"<<costs[i][j]<<"\t"<<cuts[i]<<"\t"<<cuts[j]<<endl;
        }
      }
      
      
      // for (int i=0; i+2<m+2; ++i) {
      //   costs[i][i+1] = 0;
      //   costs[i][i+2] = cuts[i+2] - cuts[i];
      //   for (int j=i+3; j<m+2; ++j) {
      //     for (int k = i+1; k+1<j; ++k) {
      //       costs[i][j] = min(costs[i][j], costs[i][k] + costs[k+1][j]);
      //     }
      //   }
      // }
      
      return costs[0][m+1];
    }
};
