class Solution {
public:
    string makeGood(string s) {
        std::vector<char> vec;
        for (int i = 0; i < s.size(); i++) {
            if (vec.size() == 0) {
                vec.push_back(s[i]);
            } else {
                if (vec[vec.size()-1] <= 'Z' && vec[vec.size()-1] >= 'A' && s[i] <='z' && s[i] >= 'a' && vec[vec.size()-1] - 'A' == s[i] - 'a') {
                    vec.pop_back();
                } else if (vec[vec.size()-1] <= 'z' && vec[vec.size()-1] >= 'a' && s[i] <='Z' && s[i] >= 'A' && vec[vec.size()-1] - 'a' == s[i] - 'A') {
                    vec.pop_back();
                } else {
                    vec.push_back(s[i]);
                }
            }
        }
        
        return std::string(vec.begin(), vec.end());
    }
};
