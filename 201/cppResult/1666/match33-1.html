class Solution {
public:
    string makeGood(string s) {
        // cout << s << ' ' ;
        for(int i =  0 ; i + 1 < s.length() ; ++i){
            if(s[i] != s[i + 1] && tolower(s[i]) == tolower(s[i + 1])){
                s.erase(s.begin() + i  , s.begin() + i + 2);
                return makeGood(s);
            }
        }
        return s ;
    }
};
