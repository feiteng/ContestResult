class Solution {
public:
    string makeGood(string s) {
        deque<char> st;
        for(auto x:s){
            if(st.size()==0){
                st.push_back(x);
            }else{
                
                char a = x;
                char b = st.back();
                if(a>='a' && a<='z' && b>='A' && b<='Z' &&((a-'a')==(b-'A'))){
                    st.pop_back();
                }
                
                else if(a>='A' && a<='Z' && b>='a' && b<='z' &&((a-'A')==(b-'a'))){
                    st.pop_back();
                }else{
                    st.push_back(x);
                }
                
            }
        }
        string ans;
        while(st.size()){
            ans.push_back(st.front());
            st.pop_front();
        }
        return ans;
    }
};
