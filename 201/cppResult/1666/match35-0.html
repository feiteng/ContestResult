class Solution {
private:
    char normal(char c) {
        if (isupper(c)) return (char)(c - 'A' + 'a');
        return c;
    }
public:
    string makeGood(string s) {
        int n = s.size();
        stack<char> st;
        string ans;
        for (int i = 0; i < n; i++) {
            if (!st.empty() && normal(st.top()) == normal(s[i]) && (islower(st.top()) ^ islower(s[i]))) {
                st.pop();
            } else {
                st.push(s[i]);
            }
        }
        while (!st.empty()) {
            ans.insert(ans.begin(), st.top());
            st.pop();
        }
        return ans;
    }
};
