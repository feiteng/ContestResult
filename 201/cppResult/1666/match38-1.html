class Solution {
public:
    
    string help(string s, int n)
    {
        string res = "";

        for(int i=0;i<n;i++)
        {
            if((i+1<n)&&(s[i]-s[i+1])==32 || (s[i+1]-s[i])==32)
            {
                // cout<<"hello";
                i++;
            }
            else
            {
                res += s[i];
            }
        }
        // cout<<res<<endl;
        return res;
    }
    
    string makeGood(string s) 
    {
        
        if(s==""){
            return s;
        }
        while(true)
        {
            string t = s;
            int n = s.size();
            s = help(s,n);
            if(s=="") break;
            if(t==s){
                break;
            }
        }
        return s;
    }
};
