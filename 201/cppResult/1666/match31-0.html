class Solution {
public:
    string makeGood(string s) {
        bool change = true;
        while (change) {
            change = false;
            for (int i = 0; i + 1 < s.size(); ++i) {
                if (s[i] != s[i + 1] && tolower(s[i]) == tolower(s[i + 1])) {
                    s = s.substr(0, i) + s.substr(i + 2);
                    change = true;
                    break;
                }
            }
        }
        return s;
    }
};
