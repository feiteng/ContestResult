class Solution {
public:
    string makeGood(string s) {
        string tmp;

        for(int i =0;i <s.size();i++){
            if(tmp.size() && abs(tmp.back() - s[i]) == 32)
                tmp.pop_back();
            else
                tmp += s[i];
        }
        return tmp;
    }
};
