class Solution {
public:
    string makeGood(string s) {
        string ans;
        for (int i = 0; i < s.length(); ++i)
        {
            if (ans.length() > 0 && (ans[ans.length() - 1] + 32 == s[i] || ans[ans.length() - 1] == s[i] + 32))
            {
                ans = ans.substr(0, ans.length() - 1);
            }
            else{
                ans = ans + s[i];
            }
        }
        return ans;
    }
};
