class Solution {
public:
    string makeGood(string s) {
        stack<char> p;
        for (int i = 0;i < s.size();i++) {
            if (p.empty()) {
                p.push(s[i]);
            } else {
                if (s[i] <= 'Z' && s[i] >= 'A' && p.top() <= 'z' && p.top() >= 'a' && p.top() - s[i] == 'a' - 'A') {
                    p.pop();

                } else if (s[i] <= 'z' && s[i] >= 'a' && p.top() <= 'Z' && p.top() >= 'A' && s[i] - p.top() == 'a' - 'A') {
                    p.pop();
                } else {
                    p.push(s[i]);
                }
            } 
        }
        string ret = "";
        while (!p.empty()) {
            ret = p.top() + ret;
            p.pop();
        }
        return ret;
    }
};
