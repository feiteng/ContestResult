const int inf = 1e9;


class Solution {
public:
    string makeGood(string s) {
      while (true) {
        bool flag = true;
        int n = s.size();
        for (int i = 0; i + 1 < n; i++) {
          if (s[i] - s[i + 1] == 'a' - 'A'
            || s[i + 1] - s[i] == 'a' - 'A') {
            s = s.substr(0, i) + s.substr(i + 2, n);
            flag = false;
            break;
          }
        }
        if (flag) return s;
      }
      return "";
    }
};
