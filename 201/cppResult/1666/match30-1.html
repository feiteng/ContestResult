class Solution {
public:
    string makeGood(string s) {
        string res;

        for(char c: s) {
            if(res.size() and toupper(res.back()) == toupper(c) and res.back() != c) {
                res.pop_back();
            } else {
                res.push_back(c);
            }
        }
        return res;
    }
};
