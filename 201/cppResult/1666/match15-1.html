class Solution {
public:
    string makeGood(string s) {
        while (s.size()) {
            bool isChanged = false;
            for (int i = 0; i < s.size()-1; i++) {
                if ( (s[i] != s[i+1]) && (tolower(s[i]) == tolower(s[i+1]))) {
                    s.erase(i, 2);
                    isChanged = true;
                    break;
                }
            }
            
            if (!isChanged)
                break;
        }
        return s;
    }
};
