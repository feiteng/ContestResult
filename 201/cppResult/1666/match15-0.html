class Solution {
public:
    string makeGood(string s) {
        while(s.length()){
            bool flag=false;
            for(int i=0;i<s.length()-1;++i){
                if(s[i+1]==s[i]-'a'+'A' || s[i]==s[i+1]-'a'+'A'){
                    // cout<<s<<endl;
                    s.erase(i,2);
                    flag=true;
                    break;
                }
            }
            if(!flag)return s;
        }
        return s;
    }
};
