class Solution {
public:
    string makeGood(string s) {
        deque<char> st;
        for (const char& c: s) {
            if (!st.empty()) {
                if (c - 32 == st.back() || st.back() - 32 == c) {
                    st.pop_back();
                } else {
                    st.push_back(c);
                }
            } else {
                st.push_back(c);
            }
        }
        string ans(st.begin(), st.end());
        return ans;
    }
};
