class Solution {
public:
    string makeGood(string s) {
        if(s=="")return s;
        string temp;
        for(int i=0;i<s.length()-1;i++){
            if((isupper(s[i]) && tolower(s[i])==s[i+1]) || (isupper(s[i+1]) && s[i]==tolower(s[i+1]))){
                temp+=s.substr(i+2);
                return makeGood(temp);
                break;
            }
            temp+=s[i];
        }
        
        return s;
    }
};
