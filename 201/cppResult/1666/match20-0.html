class Solution {
public:
    string makeGood(string s) {
        int len=s.size();
        while(1)
        {
            bool flag=true;
            for(int i=0;i<len-1;i++)
            {
                if(s[i]-s[i+1]==32||s[i+1]-s[i]==32)
                {
                    flag=false;
                    s.erase(i,2);
                    len=len-2;
                    break;
                }
            }
            if(flag)break;
        }
        return s;
    }
};
