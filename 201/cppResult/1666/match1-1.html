class Solution {
public:
    string makeGood(string s) {
        string ret;
        for (char ch: s) {
            if (ret.size() && (ret.back()-'a'==ch-'A' || ret.back()-'A'==ch-'a'))
                ret.pop_back();
            else ret.push_back(ch);
        }
        return ret;
    }
};
