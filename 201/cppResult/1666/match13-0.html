class Solution {
public:
    string makeGood(string s) {
        string ans = "";
        int n = s.length();
        for(int i = 0; i < n; i++) {
            if(ans.empty()) {
                ans.push_back(s[i]);
            }
            else {
                if(ans.back() != s[i] && tolower(ans.back()) == tolower(s[i])) {
                    ans.pop_back();
                }
                else {
                    ans.push_back(s[i]);
                }
            }
        }
        return ans;
    }
};
