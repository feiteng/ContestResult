class Solution {

public:
    string makeGood(string s) {
        int n = s.size();
        vector<char> ans;
        for(int i = 0; i < n; i++) {
            if(ans.empty() || islower(s[i]) == islower(ans.back()) || tolower(s[i]) != tolower(ans.back())) {
                ans.push_back(s[i]); 
            } else {
                ans.pop_back();
            }
        }
        return string(ans.begin(), ans.end());
    }
};
