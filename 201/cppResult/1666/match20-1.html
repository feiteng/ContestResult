class Solution {
public:
    string makeGood(string s) {
        
        while (1) {
            bool flag = 0;

            int n = s.size();
            for (int i = 0; i < n-1; i++) {
                if (s[i]-s[i+1] == 'a'-'A' || s[i]-s[i+1] == 'A'-'a') {
                    s = s.substr(0, i) + s.substr(i+2, n);
                    flag = 1;
                    break;
                }
            }
            if (!flag) break;
        }
        return s;
        
    }
};
