class Solution {
public:
    string makeGood(string s) {
        stack<char> st;
        
        for (int i = 0; i < s.size(); i++) {
            if (st.empty()) {
                st.push(s[i]);
            }
            else {
                if (st.top() - 'a' == s[i] - 'A') {
                    st.pop();
                    continue;
                }
                else if (st.top() - 'A' == s[i] - 'a') {
                    st.pop();
                }
                else {
                    st.push(s[i]);
                }
            }
        }
        
        string t = "";
        while (!st.empty()) {
            t += st.top();
            st.pop();
        }
        reverse(t.begin(), t.end());
        
        return t;
    }
};
