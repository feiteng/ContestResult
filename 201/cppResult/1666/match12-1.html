class Solution {
public:
    string makeGood(string s) {
        string ans;
        stack<char> st;
        for (auto c: s) {
            if (st.empty()) {
                st.push(c);
            } else {

                if (islower(st.top()) && isupper(c) && st.top() == tolower(c)) {
                    st.pop();
                } else if (islower(c) && isupper(st.top()) && c == tolower(st.top())) {
                    st.pop();
                } else {
                    st.push(c);
                }
            }
        }
        while (!st.empty()) {
            ans.push_back(st.top());
            st.pop();
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
