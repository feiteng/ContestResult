class Solution {
public:
    bool same(char a,char b)
    {
        if((isupper(a) && islower(b) && tolower(a)==b) ||
           (islower(a) && isupper(b) && toupper(a)==b))
        {
            return true;
        }
        return false;
    }
    string makeGood(string s) {
        int n=s.size(),i=1;
        string ss="#";
        for(char c:s)
        {
            if(same(ss.back(),c))
            {
                ss.pop_back();
            }
            else
            {
                ss.push_back(c);
            }
        }
        return ss.substr(1);
    }
};
