class Solution {
public:
    string makeGood(string s) {
        s="#"+s;
        bool f=1;
        while(f){
            f=0;
            for(int i=0;i<s.size()-1;i++){
                if(s[i]-s[i+1]=='a'-'A' or s[i+1]-s[i]=='a'-'A'){
                    s.erase(s.begin()+i);
                    s.erase(s.begin()+i);
                    f=1;
                    break;
                }
            }
        }
        return s.substr(1);
    }
};
