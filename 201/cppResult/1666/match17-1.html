class Solution {
public:
    bool ok (char a, char b) {
        if (a > b) swap (a,b);
        a += ('a'-'A');
        return a == b;
    }
    string makeGood(string s) {
        bool sb = true;
        while (sb) {
            sb = false;
            for (int i = 1; i < s.size(); i++) {
                if (ok(s[i-1],s[i])) {
                    s.erase(i-1,2);
                    sb = true;
                    break;
                    }
            }
        }
        return s;
    }
};
