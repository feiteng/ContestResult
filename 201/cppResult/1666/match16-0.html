class Solution {
public:
    string makeGood(string s) {
        vector<char> stack;
        stack.reserve(s.size());
        for (auto c : s) {
            if (stack.empty()) {
                stack.emplace_back(c);
            } else if (stack.back() != c && tolower(stack.back()) == tolower(c)) {
                stack.pop_back();
            } else {
                stack.emplace_back(c);
            }
        }
        return string(begin(stack), end(stack));
    }
};
