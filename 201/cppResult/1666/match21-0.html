class Solution {
public:
    stack<char> st;
    string makeGood(string s) {
        if(s == "") return s;
        
        st.push(s[0]);
        
        for(int i = 1; i < s.length(); i ++) {
            if((!st.empty()) && abs(st.top() - s[i]) == abs('a' - 'A')) {
                st.pop();
            } else {
                st.push(s[i]);
            }
        }
        
        string res = "";
        
        while(!st.empty()) {
            res += st.top();
            st.pop();
        }
        
        reverse(res.begin(), res.end());
        
        return res;
    }
};
