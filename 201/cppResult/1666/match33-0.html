class Solution {
public:
    string makeGood(string s) {
        for (int i = 0; i + 1 < s.size(); ++ i) {
            if (s[i] >= 'a' && s[i] <= 'z' && s[i + 1] >= 'A' && s[i + 1] <= 'Z' && s[i + 1] - 'A' == s[i] - 'a') {
                return makeGood(s.substr(0, i) + s.substr(i + 2));
            }
            else if (s[i + 1] >= 'a' && s[i + 1] <= 'z' && s[i] >= 'A' && s[i] <= 'Z' && s[i + 1] - 'a' == s[i] - 'A') {
                return makeGood(s.substr(0, i) + s.substr(i + 2));
            }
        }
        return s;
    }
};
