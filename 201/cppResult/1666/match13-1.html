class Solution {
public:
    string makeGood(string s) {
        string res;

        int n = s.size();
        for (int i = 0; i < n; i++) {
            if (res.size() == 0) {
                res.push_back(s[i]);
            } else {
                if (res.back() + 'A' - 'a' == s[i] || s[i] + 'A' - 'a' == res.back()) {
                    res.pop_back();
                } else {
                    res.push_back(s[i]);
                }
            }
        }
        return res;
    }
};
