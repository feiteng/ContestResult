class Solution {
private:

public:
  Solution() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
  }

  string makeGood(string s) {
    const int delta = 'a' - 'A';


    bool need_check = true;
    while (need_check) {
      need_check = false;
      for (int i = 0; i + 1 < s.size(); ++i) {
        if (('A' <= s[i] && s[i] <= 'Z' && s[i] + delta == s[i + 1]) ||
            ('a' <= s[i] && s[i] <= 'z' && s[i] - delta == s[i + 1])) {
          s = s.substr(0, i) + s.substr(i + 2);
          need_check = true;
          break;
        }
      }
    }
    return s;
  }
};
