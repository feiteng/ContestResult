class Solution {
public:
    string makeGood(string s) {
        
        string ans;
        
        ans.push_back(s[0]);
        
        for(int i=1;i<s.size();i++)
        {
            if(ans.empty() or abs(s[i]-ans.back())!=abs('A'-'a'))
            {
                ans.push_back(s[i]);
            }
            else
            {
                ans.pop_back();
            }
        }
        
        return ans;
    }
};
