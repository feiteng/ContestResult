class Solution {
public:
    string makeGood(string s) {
        if(s.size() <=1)
            return s;
        for(int i=0;i<s.size()-1;i++)
        {
            if((s[i+1] <= 'Z' && s[i] >= 'a' && s[i]-s[i+1] == 32) || (s[i+1] >= 'a' && s[i] <= 'Z' && s[i+1]-s[i] == 32))
            {
                string temp = s.substr(0,i) + s.substr(i+2);
                return makeGood(temp);
            }
        }
        return s;
    }
};
