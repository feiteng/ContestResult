class Solution {
public:
    string invert(string s){
        for(int i=0;i<s.length();i++){
            if(s[i]=='0') s[i]='1';
            else s[i]='0';
        }
        return s;
    }
    string reverse(string s){
        for(int i=0;i<s.length()/2;i++){
            swap(s[i], s[s.length()-1-i]);
        }
        return s;
    }
    char findKthBit(int n, int k) {
        string s = "0";
        for(int i=1;i<n;i++){
            s = s+"1"+reverse(invert(s));
        }
        //cout<<s<<endl;
        return s[k-1];
    }
};
