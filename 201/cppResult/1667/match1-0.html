class Solution {
public:
    char findKthBit(int n, int k) {
        string s="0";
        for(int i=2;i<=n;i++){
            string tmp=s;
            for(auto& it:tmp){
                if(it=='1') it='0';
                else it='1';
            }
            reverse(tmp.begin(),tmp.end());
            s=s+"1"+tmp;
        }
        return s[k-1];
    }
};
