class Solution {
public:
    char findKthBit(int n, int k) {
        // cout << n << " " << k << endl;
        if ( n == 1 ) return '0';
        int len = 0;
        for ( int i = 1; i <= n; ++i ) len = len * 2 + 1;
        int m = len / 2 + 1;
        if ( k == m ) return '1';
        else if ( k < m ) return findKthBit(n - 1, k);
        else return findKthBit(  n - 1, (len / 2 + 1) -  (k - m)) ^ 1;
    }
};
