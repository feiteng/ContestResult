class Solution {
public:
    char findKthBit(int n, int k) {
        string s = "0";
        for(int i=0; i<n; i++){
            string t = s;
            reverse(t.begin(), t.end());
            for(int j=0; j<t.size(); j++){
                if(t[j]=='0') t[j]='1';
                else t[j]='0';
            }
            s += '1' + t;
        }
        return s[k-1];
    }
};
