class Solution {
public:
    string a[50];
    char findKthBit(int n, int k) {
        a[1] = '0';
        for (int i = 2 ; i <= n ; i++){
            a[i] = a[i - 1] + "1";           
            for (auto & g : a[i - 1]){
                if (g == '1') g = '0';
                else g = '1';
            }
            reverse(a[i - 1].begin(),a[i - 1].end());
            
            a[i] += a[i - 1];
        }
      //  cout << a[4] << endl;
        return a[n][k - 1];
    }
};
