class Solution {
public:
    char findKthBit(int n, int k) {
        string s="0";
        for(int i=0;i<n-1;i++)
        {
            string ans=s;
            for(int j=0;j<ans.size();j++)
            {
                if(ans[j]=='1') ans[j]='0';
                else ans[j]='1';
            }
            reverse(ans.begin(), ans.end());
            s=s+"1"+ans; 
        }
        return s[k-1];
    }
};
