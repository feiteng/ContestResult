class Solution {
public:
    char findKthBit(int n, int k) {
        string str="0";
        for(int i=1;i<n;++i){
            string rev=str;

            int len=rev.length();
            for(int j=0;j<len;++j){
                if(rev[j]=='0') rev[j]='1';
                else rev[j]='0';
            }
            reverse(rev.begin(),rev.end());
            str=str+'1'+rev;
        }
        //cout<<str<<endl;
        return str[k-1];
    }
};
