class Solution {
public:
    char findKthBit(int n, int k) {
        vector<string> v(21,"");
        v[0]="0";
        for(int i=1;i<=n;i++){
            v[i]=v[i-1]+"1"+func(v[i-1]);
        }
        return (v[n][k-1]);
    }
private:
    string func(string s){
        reverse(s.begin(),s.end());
        for(int i=0;i<s.length();i++){
            if(s[i]=='0')s[i]='1';
            else s[i]='0';
        }
        return s;
    }
};
