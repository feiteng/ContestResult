class Solution {
public:
    char findKthBit(int n, int k) {
      if (n == 1) return '0';
     // if (n == 2) return ("010")[k-1];
      int len = (1<<(n-1))-1;
      if (k <= len) return findKthBit(n-1, k);
      if (k == len+1) return '1';
      return findKthBit(n-1, 2*len-k+2) == '0' ? '1' : '0';
    }
};
