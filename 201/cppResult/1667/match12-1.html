class Solution {
public:
  char findKthBit(int n, int k) {
    string s1 = "0";
    for (int i = 1; i < n; ++i){
      string s2 = s1;
      reverse(s2.begin(), s2.end());
      for (int j = 0; j < s2.length(); ++j) s2[j] = s2[j] == '0' ? '1' : '0';
      s1 += "1" + s2;
//      cout << s1 << endl;
    }
    return s1[k-1];
  }
};
