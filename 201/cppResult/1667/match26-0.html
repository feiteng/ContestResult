class Solution {
public:
    char findKthBit(int n, int k) {
        cout << n << " " << k << endl;
        if (n == 1)
            return '0';
        int cnt[n + 1];
        cnt[0] = 0;
        for (int i = 1; i <= n; ++i)
            cnt[i] = cnt[i - 1] * 2 + 1;
        int mid = (cnt[n] + 1) / 2;
        if (k == mid)
            return '1';
        else if (k < mid)
            return findKthBit(n - 1, k);
        else
            return 1 - (findKthBit(n - 1, cnt[n] + 1 - k) - '0') + '0';
    }
};
