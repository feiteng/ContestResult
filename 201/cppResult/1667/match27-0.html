class Solution {
public:
    string revflip(string& s) {
      string ans; 
      for (int i = s.size()-1; i >= 0; i--) {
        if (s[i] == '1') {
          ans += "0";
        } else {
          ans += "1";
        }
      }
      return ans;
    }
    char findKthBit(int n, int k) {
      string s = "0";
      s.reserve(1000000);
      for (int i = 2; i <= n; i++) {
        s = s + "1" + revflip(s);
      }
      
      return s[k-1];
    }
};
