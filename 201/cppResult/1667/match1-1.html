class Solution {
public:
    char findKthBit(int n, int k) {
        string s = "0";
        for (int i = 1; i < n; i++) {
            string tt = s;
            for (auto& ch : tt) {
                if (ch == '0') ch = '1';
                else ch = '0';
            }
            reverse(tt.begin(), tt.end());
            s = s + "1" + tt;
        }
        return s[k - 1];
    }
};
