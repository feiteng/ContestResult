inline LL fastpo(LL x, LL n, LL mod) {
	LL res(1);
	while(n) {
		if(n & 1) {
			res = res * (LL)x % mod;
		}
		x = x * (LL) x % mod;
		n /= 2;
	}
	return res;
}

inline string itoa(int x, int width = 0) {
  string res;
  if(x == 0) res.push_back('0');
  while(x) {
    res.push_back('0' + x % 10);
    x /= 10;
  }
  while((int)res.size() < width) res.push_back('0');
  reverse(res.begin(), res.end());
  return res;
}

string s[21];
class Solution {
public:
    char findKthBit(int n, int k) {
        s[1] = "0";
        for(int i = 2; i <= n; i++) {
            s[i] = s[i - 1];
            s[i].pb('1');
            reverse(all(s[i - 1]));
            for(char & c : s[i - 1]) c = '0' + '1' - c;
            s[i] = s[i] + s[i - 1];
        }
        return s[n][k - 1];
    }
};
