class Solution {
public:
    char findKthBit(int n, int k) {
        // aab
        // aab1abb
        string s = "0";
        for (int i = 2; i <= n; i++) {
            string t = s; reverse(t.begin(), t.end());
            for (int j = 0; j < t.size(); j++) t[j] = t[j] == '0' ? '1' : '0';
            s += "1" + t;
        }
        return s[k-1];
    }
};
