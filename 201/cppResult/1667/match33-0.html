class Solution {
public:
    char findKthBit(int n, int k) {
        string s = "0";
        for(int i = 0; i < n - 1; ++i) {
            string t;
            for(int i = s.size() - 1; i >= 0; --i) {
                t += (s[i] != '1') + '0';
            }
            s = s + "1" + t;
        }
        return s[k - 1];
    }
};
