class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        s = "0"
        for i in range(1, n):
            s = s + '1' + ''.join(list(reversed(self.invert(s))))
        return s[k-1]

    def invert(self, s):
        ret = []
        for c in s:
            if c == '0':
                ret.append('1')
            else: ret.append('0')
        return ret
