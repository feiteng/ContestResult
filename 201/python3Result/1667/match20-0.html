class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        if n == 1 and k == 1:
            return '0'
        m = 2**(n-1)
        if k == m:
            return '1'
        elif k < m:
            return self.findKthBit(n-1, k)
        else:
            return str(1-int(self.findKthBit(n-1, m*2-k)))
