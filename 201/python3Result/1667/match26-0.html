class Solution:
    
    def findKthBit(self, n: int, k: int) -> str:
        if n==1:
            return '0'
        N = 2**(n-1)
        if k==N:
            return '1'
        if k<N:
            return self.findKthBit(n-1, k)
        else:
            v = self.findKthBit(n-1, 2*N-k)
            return {'1':'0','0':'1'}[v]
