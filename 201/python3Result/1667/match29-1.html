class Solution:
    
def findKthBit(self, n: int, k: int) -> str:
        if n == 1: return '0'
        mid = (2 ** n - 1) // 2 + 1
        if k == mid: return '1'
        if k > mid: 
            return '0' if '1' == self.findKthBit(n - 1, 2 ** n - k) else '1'
        return self.findKthBit(n - 1, k)
