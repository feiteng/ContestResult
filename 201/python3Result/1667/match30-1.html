class Solution:
    
def findKthBit(self, n: int, k: int) -> str:
        '''1,3,7,15,31   l(n-1)*2+1 '''
        base = '0'
        for i in range(1, n+1):
            base = base + "1" + ''.join("1" if b == "0" else "0" for b in base[::-1])
            #print(base)
        return base[k-1]
