class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        s = '0'
        def invertt(s):
            res = ''
            for c in s:
                if c == '0':
                    res += '1'
                else:
                    res += '0'
            return res
        
        for i in range(19):
            s = s + '1' + invertt(s)[::-1]
            # print(s)
        return s[k-1]
