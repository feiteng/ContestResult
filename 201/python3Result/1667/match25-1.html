class Solution:
    def kth(self, n, k):
        if n == 1:
            assert(k == 1)
            
return 0
        mid = 2**(n-1)
        if k == mid: return 1
        if k < mid:
            return self.kth(n-1, k)
        if k > mid:
            return 1-self.kth(n-1, 2**(n-1) - (k-mid))
        
    def findKthBit(self, n: int, k: int) -> str:
        return str(self.kth(n, k))
