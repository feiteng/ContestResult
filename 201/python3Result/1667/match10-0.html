class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        def invert(s):
            arr = []
            for ch in s:
                if ch == '1':
                    arr.append('0')
                else:
                    arr.append('1')
            return ''.join(arr[::-1])
        s = '0'
        for i in range(1, n):
            s = s + '1' + invert(s)
        return s[k - 1]
