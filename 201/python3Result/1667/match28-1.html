class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        def helper(s):
            dic = {'0':'1','1':'0'}
            ret = [dic[c] for c in s]
            
return ''.join(reversed(ret))
        s = '0'
        for _ in range(n-1):
            s += '1'+helper(s)
            # print(s)
        return s[k-1]
