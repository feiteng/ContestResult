def rev_invert(s):
    sn = []
    for c in reversed(s):
        sn.append('0' if c == '1'else '1')
    return ''.join(sn)

class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        s = '0'
        for i in range(2, n + 1):
            s = s + '1' + rev_invert(s)
        return s[k - 1]
