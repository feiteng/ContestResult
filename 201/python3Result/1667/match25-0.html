class Solution:
    def helper(self,n,k,invert):
        if n==1:
            return invert%2
        mid=2**(n-1)
        if k==mid:
            return 1-invert%2
        if k<mid:
            return self.helper(n-1,k,invert)
        if k>mid:
            return self.helper(n-1,mid*2-k,invert+1)
    def findKthBit(self, n: int, k: int) -> str:
        return str(self.helper(n,k,0))
