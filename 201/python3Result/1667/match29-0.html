class Solution:
    def rev(self, a):
        return '0' if a == '1' else '1'
    
    def findKthBit(self, n: int, k: int) -> str:
        if k == 1:
            return '0'
        m = 2 ** (n - 1)
        if k == m:
            return '1'
        if k < m:
            return self.findKthBit(n - 1, k)
        return self.rev(self.findKthBit(n - 1, 2 ** n - k))
