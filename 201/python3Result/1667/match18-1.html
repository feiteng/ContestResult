class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        
        s = "0"
        
        for i in range(n - 1):
            
            temp = ""
            for ch in s:
                if ch == "0":
                    temp += "1"
                else:
                    temp += "0"
                    
            s = s + "1" + temp[::-1]
            
        return s[k - 1]
