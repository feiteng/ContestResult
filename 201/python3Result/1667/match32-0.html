class Solution:
    def invert(self, s):
        newS = ""
        for ch in s:
            newS = newS + ("1" if ch=="0" else "0")
        return newS
    def findKthBit(self, n: int, k: int) -> str:
        s = "0"
        for i in range(n-1):
            s = s + "1" + (self.invert(s))[::-1]
        return s[k-1]
