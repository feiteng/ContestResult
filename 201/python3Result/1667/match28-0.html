class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        def inv(s):
            return ''.join('1' if x == '0' else '0' for x in reversed(s))
        s = "0"
        for i in range(n - 1):
            s = s + "1" + inv(s)
            #print('"' + s + '",')
        return s[k - 1]
