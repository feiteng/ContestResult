class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        
        def recurse(n, k):
            if n == 1:
                return 0
            if k == 2 ** (n-1):
                return 1
            elif k < 2 ** (n-1):
                return recurse(n-1, k)
            else:
                return 1 - recurse(n-1, 2**n - k)
            
        
        return str(recurse(n, k))
