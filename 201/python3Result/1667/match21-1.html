class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        s = "0"
        def invert(s):
            res = ""
            for char in s:
                if char == "0":
                    res += "1"
                else:
                    res += "0"
            return res
        for i in range(n):
            s = s + "1" + invert(s)[::-1]
        return s[k-1]
