class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        if n == 1 : r
eturn '0'
        length = 2 ** n - 1
        mid = (length + 1) // 2
        
        if k == mid : return '1'
        
        if k < mid : return self.findKthBit(n - 1, k)
        
        #k - mid in none reversed order
        digit = self.findKthBit(n - 1, 2 ** (n - 1) - (k - mid))
        
        return '1' if digit == '0' else '0'
