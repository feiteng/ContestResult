class Solution:
    def findKthBit(self, n: int, k: int) -> str:
        def helper(s):
            res = []
            for c in s:
                if c == '1':
                    res.append('0')
                else:
                    res.append('1')
            return ''.join(res)[::-1]
        s = '0'
        for i in range(n-1):
            s = s + '1' + helper(s)
       # print(s)
        return s[k-1]
