class Solution:

    def findKthBit(self, n: int, k: int) -> str:
        s = "0"
        for i in range(1, n): 
            s = s + '1' + ''.join(list(reversed(self.invert(s))))
        
        return s[k-1]
    
    def invert(self, s): 
        new = []
        for val in s: 
            if val == '1': 
                new.append('0')
            else:
                new.append('1')
        return ''.join(new)
