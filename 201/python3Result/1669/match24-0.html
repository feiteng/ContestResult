class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        cuts.append(n)
        idxs = [cuts[0]]
        for i in range(len(cuts)-1):
            idxs.append(cuts[i+1] - cuts[i])
        dp = [[0]*len(cuts) for _ in range(len(cuts))]
        for i in range(len(cuts)):
            for j in range(0, len(cuts)-i):
                if i == 0:
                    dp[j][j] = 0
                elif i == 1:
                    dp[j][j+1] = idxs[j] + idxs[j+1]
                else:
                    dp[j][i+j] = float('inf')
                    for k in range(j, i+j):
                        dp[j][i+j] = min(dp[j][i+j], dp[j][k] + dp[k+1][i+j] + cuts[i+j] - (cuts[j-1] if j >=1 else 0))
        return dp[0][-1]
