class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.extend([0, n])
        cuts.sort()
        print(cuts)
        m = len(cuts)
        
        @functools.lru_cache(m**2)
        def cost(start, end):
            nonlocal cuts
            
            if start == end:
                return 0
            
if start+1 == end:
                return 0
            
            min_cost = float('INF')
            for cut in range(start+1, end):
                min_cost = min(min_cost, cost(start, cut)+cost(cut, end))
                
            # print('min_child_cost for', start, end, 'is', min_cost)
            return min_cost+cuts[end]-cuts[start]         
        
        return cost(0, m-1)
