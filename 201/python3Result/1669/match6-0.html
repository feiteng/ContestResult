from functools import lru_cache

class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        @lru_cache(None)
        def f(i, j):
            res = float('inf')
            
            for x in cuts:
                if i < x < j:
                    res = min(res, f(i, x) + f(x, j) + j-i)
                    
            return res if res != float('inf') else 0
        
        return f(0, n)
