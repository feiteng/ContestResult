class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        nums = []
        cuts.sort()
        for idx, val in enumerate(cuts):
            if idx == 0:
                nums.append(val)
            else:
                nums.append(val - cuts[idx- 1])
        nums.append(n - cuts[-1])
        k = len(nums)
        prefix = [0] + cuts + [n]
        # for idx, val in enumerate(nums):
        #     prefix.append(val + prefix[-1])
        print(nums)
        print(prefix)
        from functools import lru_cache
        @lru_cache(None)
        def dp(l, r):
            if l == r:
                return 0
            if l + 1 == r:
                return nums[l] + nums[r]
            tmp = float('inf')
            for mid in range(l, r):
                tmp = min(tmp, dp(l, mid) + dp(mid + 1, r))
            return tmp + prefix[r + 1] - prefix[l]
        return dp(0, k - 1)
