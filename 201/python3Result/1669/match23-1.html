class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        m = len(cuts)
        cuts.sort()
        cuts = [0] + cuts + [n]

        from functools import lru_cache
        
        sys.setrecursionlimit
(7654321)
        @lru_cache(None)
        def dfs(l ,r):
            if r - l <= 1: return 0
            res = int(1e9)
            for i in range(l + 1, r):
                res = min(res, dfs(l, i) + dfs(i, r))
            return res + cuts[r] - cuts[l]
            
            
        return dfs(0, m + 1)
