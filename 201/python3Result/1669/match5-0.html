class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        @lru_cache(None)
        def dfs(left, right):
            result = float('inf')
            for cut in cuts:
                if left < cut < right:
                    result = min(result, right - left + dfs(left, cut) + dfs(cut, right))
                    
            return 0 if result == float('inf') else result
        
        return dfs(0, n)
