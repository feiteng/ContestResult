class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        @lru_cache(None)
        def dfs(start, end):
            # print(start, end)
            if end-start == 1:
                return 0
            res = float('inf')
            left, right = bisect.bisect_right(cuts, start), bisect.bisect_left(cuts, end)-1
            if left > right:
                # print('cut', start, end, 0, left, right)
                return 0
            # print('left', cuts[left], 'right', cuts[right],)
            
for i in range(left, right+1):
                # print('curt', i)
                res = min(res, dfs(start, cuts[i])+dfs(cuts[i], end)+end-start)
            # print('cut', start, end, res)
            return res
        return dfs(0, n)
