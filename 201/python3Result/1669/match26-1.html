class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        cuts.append(0)
        cuts.append(n)
        cuts.sort()
        
        arr = []
        for i in range(len(cuts) - 1):
            arr.append(cuts[i+1] - cuts[i])
        
        
@functools.lru_cache(None)
        def recurse(begin, end):
            if end == begin + 1:
                return 0
            return cuts[end] - cuts[begin] + min([recurse(begin, i) + recurse(i, end) for i in range(begin + 1, end)])
                
        
        return recurse(0, len(arr))
