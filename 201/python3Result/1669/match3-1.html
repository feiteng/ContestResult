class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        cuts = sorted
(cuts)
        
        @lru_cache(None)
        def cut(stick_l, stick_r, cuts_l, cuts_r):
            if cuts_l > cuts_r:
                return 0
            
            min_cost = float(inf)
            for i in range(cuts_l, cuts_r + 1):
                min_cost = min(min_cost, cut(stick_l, cuts[i], cuts_l, i-1) + cut(cuts[i], stick_r, i+1, cuts_r))
            
            return min_cost + stick_r - stick_l
        
        return cut(0, n, 0, len(cuts)-1)
