class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        @lru_cache(None)
        def helper(start, end, i, j):
            if i==j: return 0
            ans = math.inf
            for x in range(i,j):
                ans = min(ans, end-start + helper(start,cuts[x], i, x) + helper(cuts[x], end, x+1, j))
            return ans
        return helper(0,n,0,len(cuts))
