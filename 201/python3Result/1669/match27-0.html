class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        dp = dict()
        def calc(lb, rb, ci, cj): # [ci, cj)
            if ci >= cj: return 0
            if (lb, rb) in dp: return dp[(lb, rb)]
            dp[(lb, rb)] = 10**9
            for i in range(ci, cj):
                x = calc(lb, cuts[i], ci, i) + calc(cuts[i], rb, i + 1, cj)
                dp[(lb, rb)] = min(dp[(lb, rb)], x + rb - lb)
            return dp[(lb, rb)]
        return calc(0, n, 0, len(cuts))
