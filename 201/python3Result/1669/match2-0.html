class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        @functools.lru_cache(None)
        def cut(l, r, cut_l, cut_r):
            if cut_l == cut_r:
                return 0
            res = float('inf')
            for i in range(cut_l, cut_r):
                res = min(res, cut(l, cuts[i], cut_l, i) + cut(cuts[i], r, i + 1, cut_r) + r - l)
            return res
        
        return cut(0, n, 0, len(cuts))
