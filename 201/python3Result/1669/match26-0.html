class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        from functools import lru_cache
        
        c = [0] + sorted(cuts) + [n]
        @lru_cache(None)
        def helper(i, j):
            if j == i + 1:
                return 0
            return c[j] - c[i] + min([helper(i, k) + helper(k, j) for k in range(i + 1, j)])
        
        return helper(0, len(c) - 1)
