class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        cuts = [0] + cuts + [n]
        m = len(cuts)
        
        # min cost cutting stick[cuts[i]:cuts[j]]
        from functools import lru_cache
        @lru_cache(None)
        def dp(i: int, j: int) -> int:
            
if j - i == 1:
                return 0
            return cuts[j] - cuts[i] + min(dp(i, k) + dp(k, j) for k in range(i + 1, j))
        
        return dp(0, m - 1)
