class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts = [0] + sorted(cuts) + [n]
        @functools.lru_cache(None)
        def dp(left=0, right=len(cuts)-1):
            if right - left < 2:
                return 0
            return cuts[right] - cuts[left] + min(dp(left, j) + dp(j, right) for j in range(left+1, right))
            
        return dp()
