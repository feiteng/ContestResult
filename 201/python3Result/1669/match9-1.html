class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts += [0, n]
        cuts
.sort()
        l = len(cuts)
        dp = [[float('inf')] * l for _ in range(l)]
        for i in range(l-1):
            dp[i][i+1] = 0
        for ln in range(2, l+1):
            for i in range(l-ln+1):
                j = i + ln - 1
                for k in range(i+1, j):
                    dp[i][j] = min(dp[i][j], dp[i][k] + dp[k][j])
                dp[i][j] += cuts[j] - cuts[i]
        res = int(dp[0][l-1]) - n
        return res
