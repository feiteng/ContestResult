INF=999999999999999999

class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.append(n)
        cuts.append(0)
        cuts.sort()
        pn=len(cuts)-1
        f=[[INF for _ in range(pn)] for _ in range(pn)]
        for i in range(pn):
            f[i][i]=0
        for width in range(2,pn+1):
            for leftIdx in range(pn):
                rightIdx=leftIdx+width-1
                if rightIdx>=pn: break
                
for sepIdx in range(leftIdx,rightIdx):
                    f[leftIdx][rightIdx]=min(f[leftIdx][rightIdx],f[leftIdx][sepIdx]+f[sepIdx+1][rightIdx]+cuts[rightIdx+1]-cuts[leftIdx])
        return f[0][pn-1]
