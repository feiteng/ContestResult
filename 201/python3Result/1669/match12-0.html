import functools

class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        
        @functools.lru_cache(None)
        def dp(s, e, tup):
            if not tup:
                return 0
            
            res = float('inf')
            for i, cut in enumerate(tup):
                res = min(res, dp(s, cut, tup[:i]) + dp(cut, e, tup[i + 1:]) + e - s)
            return res
        
        return dp(0, n, tuple(cuts))
