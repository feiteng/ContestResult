class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        #sort cuts
        #dp[i][j]: minimal cost for position from cuts[i] to cuts[j], i: 0-m, j: 0-m
        #dp[i][j] = min((k-i)*(j-k) + dp[i][k] + dp[k][j] for k in range(i, j)), where k is the first cut. 
        cuts+=[0, n]
        cuts
.sort()
        m = len(cuts)
        dp = [[0]*m for _ in range(m)]
        for i in range(m-3, -1, -1):
            for j in range(i+2, m):
                dp[i][j] = min((cuts[k]-cuts[i])+(cuts[j]-cuts[k]) + dp
[i][k] + dp[k][j] for k in range(i+1, j))
        return dp[0][m-1]
