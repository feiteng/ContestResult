class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        sys.setrecursionlimit(10**7)
        @lru_cache(None)
        def dfs(left,right,l,r):
            if left>=right:
                return 0
            if l>r:
                return 0
            ans=inf
            for i in range(l,r+1):
                ans=min(ans,right-left+dfs(left,cuts[i],l,i-1)+dfs(cuts[i],right,i+1,r))
            return ans
        return dfs(0,n,0,len(cuts)-1)
