from functools import lru_cache
class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        cuts.insert(0,0)
        cuts.append(n)
        @lru_cache(None)
        def dp(i,j):
            if j <= i+1: return 0
            ans=float('inf')
            for k in range(i+1,j):
                ans=min(ans,
                        dp(i,k)+dp(k,j)+(cuts[j]-cuts[i])
                       )
            return ans
                    
            
        return dp(0,len(cuts)-1)
