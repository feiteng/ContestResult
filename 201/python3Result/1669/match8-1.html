class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        if len(cuts) == 1:
            return n
            
        cuts.sort()
        lens = [0] + cuts + [n]
        
        n = len(lens) - 1
        
        dp = [[sys.maxsize] * n for _ in range
(n)]
        
        for i in range(n):
            dp[i][i] = 0
            
        for step in range(1, n):
            for start in range(n - step):
                end = start + step
                for left_end in range(start, end):
                    dp[start][end] = min(dp[start][end], dp[start][left_end] + dp[left_end + 1][end] + lens[end + 1] - lens[start])
                    
        return dp[0][-1]
