class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        cuts.insert(0,0)
        cuts.append(n)
        #print(cuts)
        dp=[[9999999 for _ in range(len
(cuts))] for _ in range(len(cuts))]
        for i in range(1,len(cuts)):
            for j in range(0,len(cuts)-i):
                if i==1:
                    dp[j][j+i]=0
                elif i==2:
                    dp[j][j+i]=cuts[j+i]-cuts[j]
                else:
                    
for k in range(1,i):
                        dp[j][j+i]=min(dp[j][j+i],cuts[j+i]-cuts[j]+dp[j][j+k]+dp[j+k][j+i])
        return dp[0][len(cuts)-1]
