class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        cuts.sort()
        if cuts[0] != 0:
            cuts = [0] + cuts
        if cuts[-1] != n:            
            cuts = cuts + [n]
        m = len(cuts)
        
        @lru_cache(maxsize=None)
        def dp(i, j): 
            #boundary cuts were cuts[i] cuts[j]
            assert i != j
            if i + 1 == j:
                return 0
            
            best = inf
            for k in range(i + 1, j):
                best = min(best, dp(i, k) + dp(k, j))
                
            return cuts[j] - cuts[i] + best
            
        return dp(0, m - 1)
