INF = float('inf')
class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts
.sort()
        @lru_cache(None)
        def dp(lo, hi, left, right):
            if lo > hi:
                return 0
            
            if lo == hi:
                return right - left
            
            ans = INF
            for i in range(lo, hi + 1):
                ans = min(ans, dp(lo, i - 1, left, cuts[i]) + dp(i + 1, hi, cuts[i], right))
            return ans + right - left
        return dp(0, len(cuts) - 1, 0, n)
