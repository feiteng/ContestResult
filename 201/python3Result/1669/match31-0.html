class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts = [0] + sorted(cuts) + [n]
        @lru_cache(None)
        def dp(i,j):
            if j == i + 1:
                return 0
            res = float('inf')
            for k in range(i+1,j):
                res = min(res, dp(i,k) + dp(k,j))
            return res + cuts[j] - cuts[i]
        return dp(0, len(cuts)-1)
