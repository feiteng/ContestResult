from functools import lru_cache

class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        m = len(cuts)
        
        cuts += [0, n]
        
        cuts.sort()
        
        # print (cuts)
        
        @lru_cache(None)
        def cut(l, r):
            
            if l == r:
                # print (l, r, cuts[l + 1] - cuts[l - 1])
                return cuts[l + 1] - cuts[l - 1]
            
            maxv = min( cut(l + 1, r), cut(l, r - 1) )
            
            for j in range(l + 1, r):
                
                # print ('j', j)
                
                maxv = min(maxv, cut(l, j - 1) + cut(j + 1, r) )
                
                
            # print (l, r, maxv, (cuts[r + 1] - cuts[l - 1]))
                
            return maxv + (cuts[r + 1] - cuts[l - 1])
        
        ans = cut(1, m)
        
        return ans
