class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts=[0]+sorted(cuts)+[n]
        m=len(cuts)
        #print(cuts)
        @lru_cache(None)
        def dp(l,r):
            
if l>=r-1:
                return 0
            res=inf
            for j in range(l+1,r):
                res=min(res,dp(l,j)+dp(j,r))
            return res+cuts[r]-cuts[l]
        return dp(0,m-1)
