class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        
        cuts.sort()
        
        INF = float('inf')
        
        @lru_cache(None)
        def dfs(be, en):
            if be == en:
                return 0 
            needed 
= [cut for cut in cuts if be < cut < en]
            
            if not needed:
                return 0
            
            res = INF
            for cut in needed:
                res = min(res, dfs(be, cut) + dfs(cut, en) + en - be)
            return res
        
        return dfs(0, n)
