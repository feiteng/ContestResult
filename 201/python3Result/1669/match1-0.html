class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        
        @lru_cache(None)
        def dp(i, j, start, end):
            if i > j:
                return 0
            p = math.inf
            for k in range(i, j + 1):
                p = min(p, dp(i, k - 1, start, cuts[k]) + dp(k + 1, j, cuts[k], end))
            return p + end - start
        return dp(0, len(cuts) - 1, 0, n)
