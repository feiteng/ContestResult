class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        res = []
        cuts.insert(0, 0)
        cuts.append(n)
        for i in range(1, len(cuts)):
            res.append(cuts[i] - cuts[i-1])
        
        K = 2
        N = len(res)
        dp = [[float('inf')] * N for _ in range(N)]

        for i in range(N):
            dp[i][i] = 0
        for i in range(N):
            for j in range(i-1, -1, -1):
                if i == j+K-1:
                    dp[i][j] = sum(res[j:i+1])
                    continue
                for t in range(j, i, K-1):
                    tmp = (i-t-1)%(K-1)+2
                    if tmp == K:
                        dp[i][j] = min(dp[i][j], dp[t][j] + dp[i][t+1] + sum(res[j:i+1]))
                    elif tmp < K:
                        dp[i][j] = min(dp[i][j], dp[t][j] + dp[i][t+1])
        return dp[-1][0]
