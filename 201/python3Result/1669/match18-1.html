class Solution:
    def minCost(self, n: int, c: List[int]) -> int:
        c.append(0)
        c.append(n)
        c.sort()
        
        @functools.lru_cache(None)
        def dp(s, e):
            if s == e - 1:
                return 0
            else:
                cost 
= float('inf')
                for i in range(s+1, e):
                    cost = min(dp(s, i) + dp(i, e) + c[e] - c[s], cost)
                return cost
        return dp(0, len(c)-1)
