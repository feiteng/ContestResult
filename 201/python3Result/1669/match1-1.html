class Solution:
    def minCost(self, n: int, cuts: List[int]) -> int:
        cuts.sort()
        cuts = tuple(cuts)
        @lru_cache(None)
        def helper(lb, ub, t_cuts):
            if len(t_cuts) == 0: 
return 0
            ans = sys.maxsize
            for idx, num in enumerate(t_cuts):
                ans = min(ans, helper(lb, num, t_cuts[:idx]) + helper(num, ub, t_cuts[idx+1:]))
            return ub-lb + ans
        return helper(0, n, cuts)
