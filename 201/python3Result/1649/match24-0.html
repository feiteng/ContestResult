class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        n = len(nums)
        dp = [0] * (n+1)
        d = {0:-1}
        cur = 0
        for i in range(n):
            cur += nums[i]
            dp[i+1] = dp[i]
            if cur - target in d:
                i0 = d[cur - target]
                dp[i+1] = max(dp[i+1], dp[i0+1] + 1)
            d[cur] = i
        return dp[-1]
