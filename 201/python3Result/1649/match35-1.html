class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        sum_dict = {0: -1}
        
        r = 0
        s = 0
        p 
= -1
        for i, n in enumerate(nums):
            s += n
            if  s - target in sum_dict and sum_dict[s - target] >= p:
                r += 1
                p = i
            
            sum_dict[s] = i
        return r
