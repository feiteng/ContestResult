class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        
        from collections import defaultdict
        
        D = defaultdict(int)
        rightmost = -float('inf')
        D
[0] = -1
        cursum = 0
        res = 0
        for i, num in enumerate(nums):
            cursum += num
            if cursum - target in D and D[cursum - target] >= rightmost:
                res += 1
                rightmost = i
            D[cursum] = i
        return res
