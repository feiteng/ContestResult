class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        st = {0}
        a = 0
        ans = 0
        for i in range(len(nums)):
            a += nums[i]
            if a-target in st:
                ans += 1
                st = {0}
                a = 0
            else:
                st.add(a)
        return ans
