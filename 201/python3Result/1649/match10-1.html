class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        cumsum = {0:-1} # total -> ind mapping
        ans = 0
        total 
= 0
        taken = -2
        for i, n in enumerate(nums):
            total += n
            if total - target in cumsum and cumsum[total-target] >= taken:
                #print(cumsum[total-target], i)
                ans += 1
                taken = i
            cumsum[total] = i
        return ans
