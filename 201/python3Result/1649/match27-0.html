class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        d=collections.defaultdict(int)
        d[0]=-1
        res, lower, s=0, -1, 0
        for i, v in enumerate(nums):
            s+=v
            if s-target in d and d[s-target]>=lower:
                res+=1
                lower=i
            d[s]=i
        return res
