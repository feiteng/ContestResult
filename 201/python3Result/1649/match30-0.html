class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        d = {0: 0}
        s, ans, j = 0, 0, -1
        for i, num in enumerate(nums):
            s += num
            if s - target in d and d[s - target] > j:
                # print(i, j, s)
                ans += 1
                j = i
            d[s] = i + 1
        return ans
