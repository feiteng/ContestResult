class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        accu = set([0])
        total = 0
        res = 0
        for num in nums:
            total += num
            if total - target in accu:
                accu = set([0])
                total = 0
                res += 1
            else:
                accu.add(total)
        return res
