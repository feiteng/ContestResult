class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        rightmost = {0 : 0}
        total = 0
        intervals = []
        for i, a in enumerate(nums, 1):
            total += a
            if total - target in rightmost:
                intervals.append((rightmost[total - target], i))
            rightmost[total] = i
        
        intervals.sort(key = lambda x : x[1])
        
        now = 0
        ans = 0
        for l, r in intervals:
            if now <= l:
                ans += 1
                now = r
        return ans
