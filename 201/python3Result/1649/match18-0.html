class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        
        ans = 0
        
        vs = {0}
        # vs.add(0)
                
        c = 0
        
        for i in nums:
            
            c += i
            
            
            
            if c - target in vs:
                ans += 1
                vs = {0}
                c = 0
            else:
                vs.add(c)
                
            # print (vs, i, c, ans)
                
            
                
        return ans
