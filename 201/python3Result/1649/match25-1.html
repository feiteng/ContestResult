class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        pre=[0]
        for n in nums:
            pre.append(pre[-1]+n)
        seen
={}
        intervals=[]
        for i,n in enumerate(pre):
            comp=n-target
            if comp in seen:
                intervals.append((seen[comp],i))
            seen[n]=i
        intervals.sort(key=lambda x: x[1])
        last=-1
        ans=0
        for s,e in intervals:
            if s>=last:
                ans+=1
                last=e
        return ans
