class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        res = 0
        d = {0:-1}
        cur = 0
        end = -2
        for i,v in enumerate(nums):
            cur += v
            if cur - target in d and d[cur - target] >= end:
                res += 1
                end = i
                # print(i)
            d[cur] = i
        return res
