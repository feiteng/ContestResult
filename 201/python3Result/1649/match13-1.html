class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        used = -2
        pre = {
0:-1}
        cur = 0
        ans = 0
        for i, e in enumerate(nums):
            cur += e
            # print(cur,used)
            if cur - target  in pre and pre[cur - target] >= used:
                ans += 1
                used = i
            pre[cur] = i
        return ans
