class Solution:
    def maxNonOverlapping(self, nums: List[int], t: int) -> int:
        s = 0
        dic = {0: -1}
        pre = -3
        res 
= 0
        for i, v in enumerate(nums):
            s += v
            if s - t in dic and dic[s - t] >= pre:
                pre = i
                res += 1
            dic[s] = i
        #print(dic)
        return res
