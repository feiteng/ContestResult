class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        sums = {0,}
        sum_ = 0
        res = 0
        for num in nums:
            sum_ += num
            # print(target, sum_, sums)
            if sum_ - target in sums:
                res += 1
                sums = {0,}
                sum_ = 0
            else:
                sums.add(sum_)
        return res
