class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        v = {0:-1}
        ans = 0
        left = -1
        cur 
= 0
        for i,num in enumerate(nums):
            cur += num
            if cur-target in v and v[cur-target] >= left:
                left = i
                ans += 1
            v[cur] = i
            # print(v,ans,left)
        return ans
