class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        pre = -1
        cur = 0
        sums 
= {0: 0}
        res = 0
        for i, v in enumerate(nums):
            cur += v
            # cur - pre = target
            if cur - target in sums and sums[cur-target] > pre:
                res += 1
                pre = i
            sums[cur] = i+1
        return res
