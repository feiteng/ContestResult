class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        
        n = len(nums)
        dp = [0] * (n + 1)
        
        d = {0: -1}
        s = 0
        for i, num in enumerate(nums):
            s += num
            dp
[i + 1] = dp[i]
            if s - target in d:
                prev_idx = d[s - target]
                dp[i + 1] = max(dp[prev_idx + 1] + 1, dp[i])
            d[s] = i
        # print(dp)
        return dp[-1]
