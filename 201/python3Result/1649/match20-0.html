class Solution:
    def maxNonOverlapping(self, nums: List[int], target: int) -> int:
        sums = set()
        
        count = 0
        cur_sum = 0
        for val in nums:
            cur_sum += val
            if cur_sum - target in sums or cur_sum == target:
                sums = set()
                cur_sum = 0
                count += 1
            else:
                sums.add(cur_sum)

        return count
