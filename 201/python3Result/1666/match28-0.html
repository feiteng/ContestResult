class Solution:
    def makeGood(self, s: str) -> str:
        ret = []
        for ch in s:
            if ret and ((ch == ret[-1].upper() and ret[-1].islower()) or \
                           (ch == ret[-1].lower() and ret[-1].isupper())):
                ret.pop()
            else:
                ret.append(ch)
                
        return ''.join(ret)
