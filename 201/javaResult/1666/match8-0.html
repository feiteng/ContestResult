class Solution {
    public String makeGood(String s) {
        // char[] arr = s.toCharArray();
        while(true){
            boolean change = false;
            for(int i=0;i<s.length();i++){
                if(i<s.length()-1 && ((s.charAt(i)-'a'==s.charAt(i+1)-'A') || (s.charAt(i)-'A'==s.charAt(i+1)-'a')))
                {
                    s = s.substring(0,i)+s.substring(i+2);
                    change = true;
                    break;
                }
            }
            if(!change)
                break;
        }
        return s;
    }
}
