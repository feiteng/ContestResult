class Solution {
    public String makeGood(String s) {
	        for(int i = 0; i + 1 < s.length(); i++) {
	        	if(s.charAt(i) - s.charAt(i + 1) == 'a' - 'A' || s.charAt(i) - s.charAt(i + 1) == 'A' - 'a') {
	        		return makeGood(s.substring(0, i) + s.substring(i + 2));
	        	}
	        }
	        return s;
    }
}
