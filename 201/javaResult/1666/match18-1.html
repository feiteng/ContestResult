class Solution {
    public String makeGood(String s) {
        return test(s);
    }
    
    
public String test(String s){
        int n = s.length();
        for(int i = 0; i < n - 1; i++){
            char c1 = s.charAt(i);
            char c2 = s.charAt(i+1);
            if(c1 != c2 && (c1 - c2 == 32 || c2 - c1 == 32)){
                return test(s.substring(0, i) + s.substring(i + 2));
            }
        }
        return s;
    }
}
