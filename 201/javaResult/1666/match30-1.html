class Solution {
    public String makeGood(String s) {
        Stack<Character> stk = new Stack<>();
        for (int i = 0; i < s.length(); ++i) {
            if (!stk.isEmpty()) {
                char prev = stk.peek(), cur = s.charAt(i);
                if (prev >= 'a' && prev <= 'z' && cur >= 'A' && cur <= 'Z' && prev - cur == 'a' - 'A')
                    stk.pop();
                else if (prev >= 'A' && prev <= 'Z' && cur >= 'a' && cur <= 'z' && prev - cur == 'A' - 'a')
                    stk.pop();
                else
                    stk.push(cur);
            }
            else
                stk.push(
s.charAt(i));
        }
        StringBuilder res = new StringBuilder();
        while (!stk.isEmpty()) res.append(stk.pop());
        return res.reverse().toString();
    }
}
