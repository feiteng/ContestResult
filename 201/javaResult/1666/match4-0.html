class Solution {
    public String makeGood(String s) {
        Deque<Character> stack = new LinkedList<>();
        for (char c : s.toCharArray()) {
            if (stack.isEmpty()) {
                stack.push(c);
            } else {
                char top = stack.peek();
                if (Character.toLowerCase(top) == Character.toLowerCase(c) && Character.compare(top, c) != 0) {
                    stack.pop();
                } else {
                    stack.push(c);
                }
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for (char c : stack) {
            sb.insert(0, c);
        }
        return sb.toString();
    }
}
