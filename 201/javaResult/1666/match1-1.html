class Solution {
    public String makeGood(String s) {
        Stack<Character> stk = new Stack<>();
        for (char c : s.toCharArray()) {
            if (!stk.isEmpty()) {
                char p = stk.peek();
                if (Math.abs(p - c) == 32) 
{
                    stk.pop();
                    continue;
                }                
            }
            stk.push(c);
        }
        StringBuilder b = new StringBuilder();
        while (!stk.isEmpty()) b.insert(0, stk.pop());
        return b.toString();
    }
}
