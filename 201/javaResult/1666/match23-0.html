class Solution {
    public String makeGood(String s) {
        Deque<Character> dq = new LinkedList<>();
        for(char c : s.toCharArray()){
            if(dq.isEmpty() || Character.toLowerCase(dq.peekLast()) != Character.toLowerCase(c) || dq.peekLast() == c){
                dq.offerLast(c);
            } else
                dq.pollLast();
        }
        StringBuilder sb = new StringBuilder();
        while(!dq.isEmpty())
            sb.append(dq.pollFirst());
        return sb.toString();
    }
}
