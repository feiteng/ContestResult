class Solution {
    public String makeGood(String s) {
        
        while(!s.equals(helper(s))){
            s = helper(s);
        }
        return s;
    }
    
    String helper(String s){
        
        Stack<Character> stk = new Stack<>();
        for(char ch : s.toCharArray()){
            if (stk.isEmpty()){
                stk.push(ch);
            } else {
                if (stk.peek() != ch && (Character.toLowerCase(stk.peek()) == ch || Character.toLowerCase(ch) == stk.peek())){
                    stk.pop();
                } else {
                    stk.push(ch);
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        while(!stk.isEmpty()){
            sb.append(stk.pop());
        }
        return sb.reverse().toString();
    }
}
