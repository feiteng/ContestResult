class Solution {
    public String makeGood(String s) {
        Stack<Character> stk = new Stack<>();
        for (int i = 0; i < s.length(); ++i) {
            char c = s.charAt(i);
            if (!stk.isEmpty() && sameButDiffCase(stk.peek(), c)) {
                stk.pop();
            } else {
                stk.push(c);
            }
        }
        StringBuilder sb = new StringBuilder();
        for (char c : stk) {
            sb.append(c);
        }
        return sb.toString();
    }
    
    private static boolean sameButDiffCase(char a, char b) {
        return a != b && Character.toLowerCase(a) == Character.toLowerCase(b);
    }
}
