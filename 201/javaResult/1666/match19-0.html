class Solution {
    public String makeGood(String s) {
        int n = s.length();
        for (int i=0; i<n-1; i++) {
            char ch0 = s.charAt(i); char ch1 = s.charAt(i+1);
            if (ch0-'a' == ch1-'A' || ch0-'A' == ch1-'a') {
                return makeGood(s.substring(0, i) + s.substring(i+2, n));
            }
        }
        return s;
    }
}
