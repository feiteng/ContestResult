class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Set<Integer> seen = new HashSet<>();
        seen.add(0);
        int runningSum = 0;
        int ret = 0;
        for (int num: nums) {
            runningSum += num;
            if (seen.contains(runningSum - target)) {
                ret++;
                runningSum = 0;
                seen = new HashSet<>();
                seen.add(0);
            }
            else {
                seen.add(runningSum);
            }
        }
        return ret;
    }
}
