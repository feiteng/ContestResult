import java.util.HashMap;
import java.util.Map;

class Solution {
	public int maxNonOverlapping(int[] nums, int target) {
		Map<Integer, Integer> m = new HashMap<Integer, Integer>();
		m.put(0, 1);
		int sum = 0;
		int ans = 0;
		for (int i : nums) {
			sum += i;
			if (m.containsKey(sum - target)) {
				ans++;
				m.clear();
				m.put(0, 1);
				sum = 0;
			} else
				m.put(sum, 1);
		}
		return ans;
	}
}
