class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map= new HashMap<>();
        int sum=0, pos=-1, res=0;
        map.put(0,-1);
        for (int i=0; i<nums.length; i++){
            sum+=nums[i];
            int remain = sum - target;
            // System.out.println(map);
            // System.out.println(sum+" "+remain+" "+pos);
            if (map.containsKey(remain) && map.get(remain)>=pos){
                res++;
                pos=i;
            }
            map.put(sum, i);
        }
        return res;
    }
}
