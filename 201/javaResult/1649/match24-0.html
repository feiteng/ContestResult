class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        int n = nums.length;
        int res = 0;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int lastStart = -1, lastEnd = -1;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (map.containsKey(sum - target)) {
                int start = map.get(sum - target) + 1;
                if (start > lastEnd || i <= lastEnd) {
                    res++;
                    lastStart = start;
                    lastEnd = i;
                }
            }
            map.put(sum, i);
        }
        return res;
    }
}
