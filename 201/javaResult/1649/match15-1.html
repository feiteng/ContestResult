class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        
        Set<Integer> set = new HashSet<>();
        set.add(0);
        int prefix = 0;
        
        int count = 0;
        for (int i=0; i<nums.length; i++) {
            prefix += nums[i];
            
            int temp = prefix - target;
            
if (set.contains(temp)) {
                count++;
                prefix = 0;
                set.clear();
                set.add(0);
            } else {
                set.add(prefix);
            }
        }
        return count;
    }
}
