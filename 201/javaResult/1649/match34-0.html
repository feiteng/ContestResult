class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 0);
        int n = nums.length;
        int[] dp = new int[n + 1];
        int sum = 0;
        for(int i = 1 ; i <= n ; i++){
            dp[i] = dp[i-1];
            sum += nums[i - 1];
            if(map.containsKey(sum - target)){
                int id = map.get(sum - target);
                dp[i] = Math.max(dp[i], dp[id] + 1);
            }
            map.put(sum, i);
        }
        return dp[n];
    }
}
