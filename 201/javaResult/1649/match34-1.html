class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int sum = 0;
        int n = nums.length;
        int[] dp = new int[n + 1];
        dp[0] = 0;
        
int max = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            dp[i + 1] = max;
            if (map.containsKey(sum - target)) {
                int idx = map.get(sum - target) + 1;
                // System.out.println(i + 1);
                // System.out.println(i + 1 - idx);
                dp[i + 1] = Math.max(dp[i + 1], dp[idx] + 1);
                max = Math.max(max, dp[i + 1]);
            
}
            map.put(sum, i);
            // System.out.println((sum - target) + sum + " " + dp[i + 1]);
        }
        
        return max;
    }
}
