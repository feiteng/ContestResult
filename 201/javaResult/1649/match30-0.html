class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        //greedy
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int prefix = 0;
        int last = -1;
        int cnt = 0;
        for (int i = 0; i < nums.length; i++) {
            prefix += nums[i];
            if (map.containsKey(prefix - target)) {
                int idx = map.get(prefix - target);
                if (idx >= last) {
                    last = i;
                    cnt++;
                }
            }
            map.put(prefix, i);
        }
        return cnt;
    }
}
