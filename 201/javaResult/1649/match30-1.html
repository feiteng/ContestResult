class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int last = -1;
        int sum = 0;
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (map.getOrDefault(sum - target, -2) >= last) {
                res++;
                last = i;
            
}
            map.put(sum, i);
        }
        return res;
    }
}
