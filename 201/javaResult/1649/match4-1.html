class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Set<Integer> prev = new HashSet<>();
        prev.add(0);
        int sum = 0;
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (prev.contains(sum - target)) {
                result++;
                prev = new HashSet<>();
                
prev.add(0);
                sum = 0;
            } else {
                prev.add(sum);
            }
        }
        return result;
    }
}
