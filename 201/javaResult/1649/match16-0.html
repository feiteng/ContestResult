class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Set<Integer> sum = new HashSet<>();
        sum.add(0);
        int cur = 0;
        int res = 0;
        for (int num : nums) {
            cur += num;
            if (sum.contains(cur-target)) {
                res++;
                cur = 0;
                sum.clear();
                sum.add(0);
            }
            else {
                sum.add(cur);
            }
        }
        return res;
    }
}
