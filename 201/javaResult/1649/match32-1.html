class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 0); int n = nums.length;
        int [] pref = new int[n + 1];
        for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + nums[i - 1];
        
int last = -1; int ans = 0;
        for (int i = 1; i <= n; i++) {
            if (map.containsKey(pref[i] - target) && map.get(pref[i] - target) >= last) {
                ans++; last = i;
            }
            map.put(pref[i], i);
        }
        return ans;
    }
}
