class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        int sum = 0;
        int result = 0;
        int flag = -1;
        
        Map<Integer, Integer> preSum = new HashMap<>();
        preSum.put(0, -1);
        
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (preSum.containsKey(sum - target) && preSum.get(sum - target) >= flag) {
                result++;
                flag = i;
            }
            preSum.put(sum, i);
        }
        
        return result;
    }
}
