class Solution {
    
    public int maxNonOverlapping(int[] nums, int target) {
        
        HashSet<Integer> set = new HashSet<>();
        set.add(0);
        
        int sum=0, cnt=0;
        
        for(int i=0; i<nums.length; i++) {
            sum+=nums[i];
            // System.out.println(sum+" "+set);
            if(set.contains(sum-target)) {
                cnt+=1;
                set.clear();
                set.add(0);
                sum=0;
            } else
                set.add(sum);
            
        }
        
        return cnt;
    }
}
