import java.util.HashSet;
import java.util.Set;

class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Set<Integer> sum = new HashSet<>();
        sum.add(0);
        int s = 0;
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            s += nums[i];
            if (sum.contains(s - target)) {
                ans++;
                sum.clear();
                sum.add(0);
                s=0;
            } else {
                sum.add(s);
            }
        }
        return ans;
    }
}
