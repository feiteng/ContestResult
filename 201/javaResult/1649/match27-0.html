class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        
        int sol = 0;
        
        Set<Integer> sums = new HashSet<>();
        sums.add(0);
        int upToNum = 0;
        
        for (int num : nums) {
            upToNum += num;
            if (sums.contains(upToNum-target)) {
                sol++;
                sums = new HashSet<>();
                sums.add(upToNum);
            } else {
                sums.add(upToNum);
            }
        }
        return sol;
        
    }
}
