class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> prefix2idx = new HashMap<>();
        int maxIdx = -2, prefix = 0, count = 0;
        prefix2idx.put(0, -1);
        for (int i = 0; i < nums.length; i++){
            prefix += nums[i];
            int diff = prefix - target;
            if (prefix2idx.containsKey(diff) && prefix2idx.get(diff) + 1 > maxIdx){
                count++;
                maxIdx = i;
            }
            prefix2idx.put(prefix, i);
        }
        return count;
    }
}
