class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        HashMap<Long,Integer> map=new HashMap<>();
        map.put(0L,0);
        long sum=0;
        int prev=0;
        for(int i:nums)
        {
            sum+=i;
            if(map.containsKey(sum-target))
            {
                
                int nprev=map.get(sum-target)+1;
                
prev=Math.max(prev,nprev);
                map.put(sum,prev);
                //System.out.println("putting 1"+sum+" "+prev);
            }
            else
            {
                map.put(sum,prev);
                //System.out.println("putting 2"+sum+" "+prev);
            }
            //System.out.println("prev is now "+prev);
            
        }
        return prev;
    }
}
