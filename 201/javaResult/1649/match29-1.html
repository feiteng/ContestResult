class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int sum = 0;
        int lastIdx = -1;
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (map.containsKey(sum - target)) {
                if (map.get(sum - target) >= lastIdx) 
{
                    lastIdx = i;
                    ans++;
                }
            }
            map.put(sum, i);
        }
        return ans;
    }
}
