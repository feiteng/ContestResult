class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
            Set<Integer> prefix = new HashSet<>();
            prefix.add(0);
            int cnt = 0;
            int roll = 0;
            for (int x: nums) {
                roll += x;
                if (prefix.contains(roll - target)) {
                    cnt++;
                    prefix.clear();
                    prefix.add(0);
                    roll = 0;
                } else {
                    prefix.add(roll);
                }
            }
            return cnt;
        }
}
