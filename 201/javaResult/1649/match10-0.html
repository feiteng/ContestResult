class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Set<Integer> set = new HashSet<>();
        set.add(0);
        int result = 0;
        int sum = 0;
        for(int i=0;i<nums.length;i++) {
            sum += nums[i];
            if(set.contains(sum-target)) {
                result++;
                sum = 0;
                set = new HashSet<>();
                set.add(0);
            } else {
                set.add(sum);
            }
        }
        return result;
    }
}
