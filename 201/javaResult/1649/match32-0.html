class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int sum = 0;
        int end = -1;
        int sol = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (map.containsKey(sum-target) && map.get(sum-target) + 1 > end) {
                sol++;
                end = i;
            }
            map.put(sum, i);
        }
        return sol;
    }
}
