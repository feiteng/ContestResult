class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
	        int n = nums.length;

	        int res = 0;
	        int min = -1;
	        Map<Long, Integer> map = new HashMap<>();
	        long cur = 0;
	        map.put
(0L, -1);
	        for(int i = 0; i < n; i++) {
	        	cur += nums[i];
        		if(map.containsKey(cur - target) && map.get(cur - target) >= min) {
        			res++;
        			min = i;
        		}
        		map.put(cur, i);

	        }
	        return res;
    }
}
