class Solution {
    Map<Long, Integer> result = new HashMap<>();
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return helper(cuts, 0, cuts.length - 1, 0, n);
    }
    private int helper(int[] cuts, int cl, int cr, int sl, int sr) {
        if(cl > cr) {
            return 0;
        }
        long state = sr;
        state *= (long)1e6;
        state += sl;
        if(result.containsKey(state)) {
            return result.get(state);
        }
        int diff = sr - sl;
        int totalCost = diff;
        int minCost = Integer.MAX_VALUE;
        
        for(int i = cl; i <= cr; i++) {
            minCost = Math.min(minCost, helper(cuts, cl, i - 1, sl, cuts[i]) + helper(cuts, i + 1, cr, cuts[i], sr));    
        }
        totalCost += minCost;
        result.put(state, totalCost);
        return totalCost;
    }
}
