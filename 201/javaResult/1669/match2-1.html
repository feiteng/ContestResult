class Solution {
      public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return run(cuts, 0, n, 0, cuts.length - 1, new HashMap<>());
    }
    
    private int run(int[] cuts, int lb, int rb, int l, int r, HashMap<String, Integer> cache) {
        if (l > r) return 0;
        
if (l == r) return rb - lb;
        String id = l + ":" + r;
        if (cache.containsKey(id)) return cache.get(id);
        int out = Integer.MAX_VALUE;
        for (int i = l; i <= r; ++i) {
            out = Math.min(run(cuts, lb, cuts[i], l, i - 1, cache) + run(cuts, cuts[i], rb, i + 1, r, cache) + 
                    rb - lb, out);
        }
        cache.put(id, out);
        return out;
    }
}
