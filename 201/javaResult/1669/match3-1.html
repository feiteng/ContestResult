class Solution {

    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return minCost(0, n, 0, cuts.length - 1, cuts, new HashMap<>());
    }

    private int minCost(int l, int r, int left, int right, int[] cuts, HashMap<String, Integer> map) {
        if (left > right) {
            return 0;
        } else {
            if (map.containsKey(l + " " + r + " " + left + " " + right)) {
                return map.get(l + " " + r + " " + left + " " + right);
            
}
            int min = Integer.MAX_VALUE;
            for (int i = left; i <= right; i++) {
                min = Math.min(min,
                        minCost(l, cuts[i], left, i - 1, cuts, map) + minCost(cuts[i], r, i + 1, right, cuts, map));
            }
            map.put(l + " " + r + " " + left + " " + right, min + r - l);
            return min + r - l;
        }
    }
}
