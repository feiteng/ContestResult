class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int m = cuts.length;
        Map<String, Integer> dp = new HashMap<>();
        return minCostSub(0, n, cuts, 0, m - 1, dp);
    }
    
    private int minCostSub(int start, int end, int[] cuts, int cStart, int cEnd, Map<String, Integer> dp) {
        int n = cuts.length;
        if (cEnd < 0 || cStart >= n || cStart > cEnd) {
            return 0;
        }
        
        String key = start + " " + end + " " + cStart + " " + cEnd;
        if (dp.containsKey(key)) {
            return dp.get(key);
        }
            
        int cost = end - start;
        int rst = Integer.MAX_VALUE;
        for (int i = cStart; i <= cEnd; i++) {
            rst = Math.min(
                rst,
                minCostSub(start, cuts[i], cuts, cStart, i - 1, dp) + minCostSub(cuts[i], end, cuts, i + 1, cEnd, dp) + cost
            );
        }
        
        dp.put(key, rst);
        return rst;
    }
}
