class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        
int[] newCuts = new int[cuts.length + 2];
        for (int i = 0; i < cuts.length; i++) {
            newCuts[i + 1] = cuts[i];
        }
        newCuts[newCuts.length - 1] = n;
        int[][] ans = new int[newCuts.length][newCuts.length];
        for (int i = 0; i < newCuts.length - 1; i++) {
            ans[i][i + 1] = 0;
        }
        
for (int i = 2; i < newCuts.length; i++) {
            for (int j = 0; j < newCuts.length - i; j++) {
                ans[j][j + i] = Integer.MAX_VALUE;
                for (int k = j + 1; k < j + i; k++) {
                    ans[j][j + i] = Math.min(ans[j][j + i], newCuts[j + i] - newCuts[j] +
                            ans[j][k] + ans[k][j + i]);
                }
            }
        }
        return ans[0][newCuts.length - 1];
    }
}
