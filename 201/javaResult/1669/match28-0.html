class Solution {
    private static final int INF = 987654321;
    
    public int minCost(int n, int[] cuts) {
        final int N = cuts.length + 2;
        final int[] A = new int[N];
        System.arraycopy(cuts, 0, A, 0, cuts.length);
        A[N - 1] = n;
        Arrays.sort(A);
        
        int[][] best = new int[N][N];
        for (int i = 0; i < N; ++i) {
            Arrays.fill(best[i], INF);
        }
        for (int i = 0; i + 1 < N; ++i) {
            best[i][i + 1] = 0;
        }
        for (int sz = 2; sz < N; ++sz) {
            for (int i = 0; i + sz < N; ++i) {
                int k = i + sz;
                for (int j = i + 1; j < k; ++j) {
                    int c = A[k] - A[i];
                    best[i][k] = Math.min(best[i][k], c + best[i][j] + best[j][k]);
                }
            }
        }
        return best[0][N - 1];
    }
}
