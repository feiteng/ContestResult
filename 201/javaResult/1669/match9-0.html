class Solution {
    public int minCost(int n, int[] cuts) {
        int m = cuts.length;
        Arrays.sort(cuts);
        int[][] dp = new int[m + 2][m + 2];
        for (int i = 0; i < dp.length - 1; i += 1) {
            dp[i][i + 1] = 0;
        }
        for (int i = 2; i <= m + 1; i += 1) {
            //[j, j + i]
            for (int j = 0; j <= m + 1 - i; j += 1) {
                dp[j][j + i] = Integer.MAX_VALUE;
                for (int k = j + 1; k <= j + i - 1; k += 1) {
                    // [j, j + i]
                    int start = j == 0 ? 0 : cuts[j - 1];
                    int end = j + i == m + 1 ? n : cuts[i + j - 1];
                    int cur = dp[j][k] + dp[k][j + i] + end - start;
                    dp[j][j + i] = Math.min(dp[j][j + i], cur);
                }
            }
        }
        return dp[0][m + 1];
    }
}
