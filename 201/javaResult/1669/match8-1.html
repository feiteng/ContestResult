/*
1. def: int[][] dp, dp[i][j] := min cost to cut stick[i : j]
2. base case: dp[x][x + 1] = 0
3. induction rule: dp[i][j] = min(dp[i][k] + dp[k][j]) + stick[j] - stick[i]
*/

class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int[] stick = new int[cuts.length + 2];
        for (int i = 0; i < cuts.length; i++) {
            stick[i + 1] = cuts[i];
        }
        int len = stick.length;
        stick[len - 1] = n;
        int[][] dp = new int[len][len];
        
for (int i = len - 1; i >= 0; i--) {
            for (int j = i + 2; j < len; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                for (int k = i + 1; k < j; k++) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k][j]);
                }
                dp[i][j] += stick[j] - stick[i];
            }
        }
        return dp[0][len - 1];
    }
}
