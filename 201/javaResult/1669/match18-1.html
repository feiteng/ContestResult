class Solution {
    HashMap<List<Integer>, Integer> map = new HashMap<>();
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return cut(cuts, 0, cuts.length - 1, 0, n);
    }

    private int cut(int[] cuts, int startIndex, int endIndex, int start, int end) {
        if (startIndex == endIndex) {
            return end - start;
        } else if (startIndex > endIndex) {
            return 0;
        }
        List<Integer> list = Arrays.asList(startIndex, endIndex);
        if (map.containsKey(list)) return map.get(list);
        
int min = Integer.MAX_VALUE;

        for (int i = startIndex; i <= endIndex; i++) {
            min = Math.min(min, end - start + cut(cuts, startIndex, i - 1, start, cuts[i]) + cut(cuts, i + 1, endIndex, cuts[i], end));
        }
        map.put(list, min);
        return min;
    }
}
