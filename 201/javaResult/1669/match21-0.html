class Solution {
    public int minCost(int n, int[] cuts) {
        if(cuts == null || cuts.length < 0) {
            return 0;
        }
        
        Arrays.sort(cuts);
        int[] segments = new int[cuts.length + 1];
        segments[0] = cuts[0];
        for(int i = 1; i < cuts.length; ++i) {
            segments[i] = cuts[i] - cuts[i-1];
        }
        
        segments[cuts.length] = n - cuts[cuts.length -1];
        int[][][] costs = new int[segments.length][segments.length][2];
        for(int i = 0; i < costs.length; ++i) {
            costs[i][i][1] = segments[i];
        }
        
        for(int len = 2; len <= costs.length; ++len) {
            for(int i = 0; i + len <= costs.length; ++i) {
                costs[i][i+len-1][1] = costs[i][i][1] + costs[i+1][i+len-1][1];
                costs[i][i+len-1][0] = costs[i][i][0] + costs[i+1][i+len-1][0];
                for(int j = i + 1; j < i + len; ++j) {
                    costs[i][i+len-1][0] = Math.min(costs[i][i+len-1][0], costs[i][j-1][0] + costs[j][i+len-1][0]);
                }
                
                costs[i][i+len-1][0] += costs[i][i+len-1][1];
            }
            
            
        }
        
        return costs[0][segments.length-1][0];
    }
}
