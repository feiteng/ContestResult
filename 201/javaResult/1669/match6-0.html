class Solution {
    
  public int minCost(int n, int[] cuts) {
    Arrays.sort(cuts);
    int m = cuts.length + 2;
    int[] sum = new int[m];
    for(int i = 1; i <= cuts.length; i++){
      sum[i] = cuts[i-1];
    }
    sum[m - 1] = n;
    int[][] dp = new int[m][m];
    for(int len = 2; len < m; len++){
      for(int i = 0; i + len < m; i++) {
        int j = i + len;
        dp[i][j] = Integer.MAX_VALUE;
        for (int k = i + 1; k < j; k++) {
          dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k][j] + sum[j] - sum[i]);
        }
      }
    }
    return dp[0][m - 1];
  }
}
