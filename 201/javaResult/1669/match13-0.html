class Solution {
    public int minCost(int n, int[] cuts) {
        int x = cuts.length;
        int[] nums = new int[x+2];
        for(int i = 0; i < x; ++i) {
            nums[i+1] = cuts[i];
        }
        nums[x+1] = n;
        Arrays.sort(nums);
        int m = x + 1;
        int[][] dp = new int[m+1][m+1];
        for(int j = 2; j <= m; ++j) {
            for(int i = j - 2; i >= 0; --i) {
                // dp[i][j] is the minimum cost to cut (i,j).
                dp[i][j] = Integer.MAX_VALUE;
                for(int k = i + 1; k < j; ++k) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k][j]);
                }
                dp[i][j] += nums[j] - nums[i];
            }
        }
        return dp[0][m];
    }
}
