class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int n1=cuts.length;
        int[] d = new int[n1+2];
        
for(int i=0;i<n1;i++){
            d[i+1]=cuts[i];
        }
        d[n1+1]=n;
        int[][] dp = new int[n1+2][n1+2];
        for(int j=2;j<n1+2;j++){
            for(int i=0;i<n1+2;i++){
                if(i+j>=n1+2){
                    break;
                }
                dp[i][i+j]=dp[i][i+j-1]+d[i+j]-d[i];
                
for(int k=i+1;k<i+j;k++){
                    dp[i][i+j]=Math.min(dp[i][k]+dp[k][i+j]+d[i+j]-d[i],dp[i][i+j]);
                }
            }
        }
        // System.out.println(Arrays.deepToString(dp));
        return dp[0][n1+1];
        
    }
}
