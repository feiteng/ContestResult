class Solution {
    public int minCost(int n, int[] cuts) {
        int m = cuts.length + 2;
        int[][] dp = new int[m][m];
        
int[] c = new int[m];
        for (int i = 0; i < m - 2; i++) {
            c[i] = cuts[i];
        }
        c[m - 1] = n;
        Arrays.sort(c);
        System.out.println(Arrays.toString(c));
        
for (int i = m - 1; i >= 0; i--) {
            for (int j = i + 2; j < m; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                for (int k = i + 1; k < j; k++) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k][j] + c[j] - c[i]);
                }
                // System.out.printf("%d, %d, %d\n", i, j, dp[i][j]);
            }
        }
        return dp[0][m - 1];
    }
}
