class Solution {
    
public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        for(int[] drow : dp)
        {
            Arrays.fill(drow, -1);
        }
        return cut(0, n, cuts, 0, cuts.length - 1);
    }
    
    int N = 1_000_010;
    int[][] dp = 
new int[110][110];
    
    int cut(int from, int to, int[] cuts, int cfrom, int cto)
    {
        // System.out.printf("%d %d %d %d\n", from, to, cfrom, cto);
        if(cfrom > cto) return 0;
        if(dp[cfrom][cto] >= 0) {
            // System.out.println("...!");
            return dp[cfrom][cto];
        
}
        int total = 1 << 30;
        for(int i = cfrom; i <= cto; i++)
        {
            if(cuts[i] < from || cuts[i] > to) continue;
            total = Math.min(total, to - from + 
                cut(from, cuts[i], cuts, cfrom, i - 1) + 
                cut(cuts[i], to, cuts, i + 1, cto));
        }
        return dp[cfrom][cto] = total;
    }    
}
