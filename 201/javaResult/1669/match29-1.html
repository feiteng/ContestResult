class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int[][] memo = new int[cuts.length][cuts.length];
        for (int j = 0; j < cuts.length; j++) {
            for (int k = 0; k < cuts.length; k++) {
                memo[j][k] = -1;
            }
        }
        
return help(0, cuts.length-1, cuts, n, memo);
    }
    
    public int help(int lo, int hi, int[] cuts, int n, int[][] memo) {
        //System.out.printf("lo :%d, hi:%d\n", lo, hi);
        if (lo > hi) {
            return 0;
        }
        if (memo[lo][hi] > 0) {
            return memo[lo][hi];
        }
        
        int ret = Integer.MAX_VALUE;
        
        int big = 
hi < cuts.length - 1 ? cuts[hi + 1] :n;
        int small = lo > 0 ? cuts[lo - 1] : 0;
        int len = big - small;
        
        for (int j = lo; j <= hi; j++) {
            ret = Math.min(ret, len + help(lo, j - 1, cuts, n, memo) + help(j + 1, hi, cuts, n, memo));
        }
        memo[lo][hi] = ret;
        return ret;
    }
}
