class Solution {
    Map<Integer, Integer> map = new HashMap<>();
    int mul = 10000000;
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        
return dfs(n, cuts, 0, cuts.length - 1, 0, n);
    }
    private int dfs(int n, int[] cuts, int left, int right, int startRod, int endRod) {
        if (left > right) {
            return 0;
        }
        
if (left == right) {
            return endRod - startRod;
        }
        int key = startRod * mul + endRod;
        if (map.containsKey(key)) {
            return map.get(key);
        }
        int cur = endRod - startRod, res = Integer.MAX_VALUE;
        for (int i = left; i <= right;i++) {
            res = Math.min(res, cur + dfs(n, cuts, left, i - 1, startRod, cuts[i]) + 
                           dfs(n, cuts, i + 1, right, cuts[i], endRod));
        }
        map.put(key, res);
        return res;
    }
}
