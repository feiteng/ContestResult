class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return backtracking(0, n, cuts, 0, cuts.length, new HashMap<>());
    }
    
    public int backtracking(int left, int right, int[] cuts, int start, int end, Map<String, Integer> map) {
        if(start == end) return 0;
        String key = start + "," + end + "," + (right - left);
        if(map.containsKey(key)) return map.get(key);
        int res = Integer.MAX_VALUE;
        for(int i = start; i < end; i++) {
            int val = cuts[i];
            res = Math.min(res, right - left + backtracking(left, val, cuts, start, i, map) + backtracking(val, right, cuts, i + 1, end, map));
        }
        map.put(key, res);
        return res;
    }
}
