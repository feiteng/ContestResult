class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        return cutter(0, n, cuts, 0, cuts.length-1, new HashMap<String, Integer>());
    }
    private int cutter(int lo, int hi, int[] cuts, int i, int j, HashMap<String, Integer> dp) {
        if (i > j) return 0;
        int len = hi - lo;
        if (i == j)
            return len;
        String key = lo + "#" + hi;
        if (dp.containsKey(key)) return dp.get(key);
        int ret = Integer.MAX_VALUE;
        for (int k = i; k <= j; k++)
            ret = Math.min(ret, cutter(lo, cuts[k], cuts, i, k-1, dp) + cutter(cuts[k], hi, cuts, k+1, j, dp));
        ret += len;
        dp.put(key, ret);
        return ret;
    }
}
