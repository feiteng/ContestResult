class Solution {
    public int minCost(int n, int[] cuts) {
        int[] nods = new int[cuts.length + 2];
        for(int i = 0; i < cuts.length; i++) {
            nods[i] = cuts[i];
        }
        nods[cuts.length] = 0;
        nods[cuts.length + 1] = n;
        
Arrays.sort(nods);
        int m = nods.length;
        int[][] dp = new int[m][m];
        for(int i = 0; i < m - 1; i++) {
            dp[i][i + 1] = 0; 
        }
        
for(int g = 2; g <= m - 1; g++) {
            for(int l = 0; l + g < m; l++) {
                int r = l + g;
                dp[l][r] = 1 << 29;
                for(int mid = l + 1; mid < r; mid++) {
                    dp[l][r] = Math.min(dp[l][r], dp[l][mid] + dp[mid][r] + nods[r] - nods[l]);
                }
            }
        }
        //System.out.println(Arrays.toString(nods));
       // System.out.println(Arrays.deepToString(dp));
        return dp[0][m - 1];
    }
}
