class Solution {
    public int minCost(int n, int[] cuts) {
        int[][] memo = new int[cuts.length + 2][cuts.length + 2];
        Arrays.sort(cuts);
        return bt(n, cuts, memo, 0, cuts.length + 1);
    }
    private int bt(int n, int[] cuts, int[][] memo, int left, int right) {
        if (left == right - 1) {
            return 0;
        }
        if (memo[left][right] > 0) {
            return memo[left][right];
        }
        int start = left == 0 ? 0 : cuts[left - 1];
        int end = right == cuts.length + 1 ? n : cuts[right - 1];
        int res = Integer.MAX_VALUE;
        for (int i = left + 1; i < right; i++) {
            res = Math.min(res, bt(n, cuts, memo, left, i) + bt(n, cuts, memo, i, right));
        }
        memo[left][right] = res + end - start;
        return memo[left][right];
    }
}
