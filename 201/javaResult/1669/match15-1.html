class Solution {
    public int minCost(int n, int[] cutsN) {
        
int[] cuts = new int[cutsN.length + 2];
        Arrays.sort(cutsN);
        for (int i = 0; i < cutsN.length; i++) {
            cuts[i + 1] = cutsN[i];
        }
        cuts[cuts.length - 1] = n;
        Integer[][] dp = new Integer[cuts.length][cuts.length];
        
return helper(0, cuts.length - 1, cuts, dp);
    }
    
    private int helper(int left, int right, int[] cuts, Integer[][] dp) {
        if (right - left == 1) {
            return 0;
        }
        if (dp[left][right] != null) {
            return dp[left][right];
        }
        int min = Integer.MAX_VALUE;
        for (int i = left + 1; i < right; i++) {
            min = Math.min(min, cuts[right] - cuts[left] + helper(left, i, cuts, dp) + helper(i, right, cuts, dp));
        }
        dp[left][right] = min;
        return min;
    }
}
