class Solution {
    public int minCost(int n, int[] cuts) {
        int numCut = cuts.length;
        int[][] cache = new int[numCut + 2][numCut + 2];
        int[] rod = new int[numCut + 2];
        Arrays.sort(cuts);
        for (int i = 0; i < numCut; i++) {
            rod[i + 1] = cuts[i];
        }
        rod[0] = 0;
        rod[numCut + 1] = n;
        // for (int i : rod) {
        //     System.out.print(i + " ");
        // }
        return check(cache, 0, numCut + 1, rod);
    }
    public int check(int[][] cache, int s, int e, int[] rod) {
        if (s + 1 == e) {
            return 0;
        }
        if (cache[s][e] > 0) {
            return cache[s][e];
        }
        int res = Integer.MAX_VALUE;
        for (int i = s + 1; i < e; i++) {
            res = Math.min(res, check(cache, s, i, rod) + check(cache, i, e, rod) + rod[e] - rod[s]);
        }
        // if (res == Integer.MAX_VALUE) {
        //     res = 0;
        // }
        cache[s][e] = res;
        return res;
    }
}
