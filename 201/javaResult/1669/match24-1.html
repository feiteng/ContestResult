class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int m = cuts.length;
        
int[][] dp = new int[m][m];
        return dfs(0, n, cuts, 0, m - 1, dp);
    }
    private int dfs(int i, int j, int[] cuts, int lo, int hi, int[][] dp) {
        if (i > j || lo > hi) return 0;
        if (0 != dp[lo][hi]) return dp[lo][hi];
        
int cost = j - i, min = Integer.MAX_VALUE;
        for (int m = lo; m <= hi; ++m) {
            min = Math.min(min, dfs(i, cuts[m], cuts, lo, m - 1, dp) + dfs(cuts[m], j, cuts, m + 1, hi, dp));
        }
        return dp[lo][hi] = cost + min;
    }
}
