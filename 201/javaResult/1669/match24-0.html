class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int[][] dp=new int[cuts.length+5][cuts.length+5];
        return slove(0,cuts.length-1,dp,0,n,cuts);
    }

    private int slove(int left, int right, int[][] dp, int s,int n, int[] cuts) {
        if(left>right)return 0;
        if(left==right) return n-s;
        if(dp[left][right]!=0)return dp[left][right];

        int res=Integer.MAX_VALUE;
        for (int i=left;i<=right;i++)
        {
            res=Math.min(res,n-s+slove(left,i-1,dp,s,cuts[i],cuts)
                    +slove(i+1,right,dp,cuts[i],n,cuts));
        }

        return dp[left][right]=res;
    }
}
