class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int len=cuts.length+1;
        int[] a=new int[len];int i=0;
        int l=0;
        for(int c:cuts){
            a[i++]=c-l;
            l=c;
        }
        a[i]=n-l;

        int[][] dp=new int[len][len];

        int[] pre=new int[len+1];
        int t=0;
        pre[0]=t;
        for(i=0;i<len;i++){
            t+=a[i];
            pre[i+1]=t;
        }
        //sum[l,r]= pre[r+1]-pre[l-1+1];
        
for(int cnt=2;cnt<=len;cnt++){
            for(int ll=0;ll+cnt-1<len;ll++){
                int rr=ll+cnt-1;
                dp[ll][rr]=Integer.MAX_VALUE;
                for(int tt=ll;tt<rr;tt++){
                    dp[ll][rr]=Math.min(dp[ll][rr],
                            dp[ll][tt]+dp[tt+1][rr]+pre[rr+1]-pre[ll]
                    );
                }
            }
        }

        return dp[0][len-1];

    }
}
