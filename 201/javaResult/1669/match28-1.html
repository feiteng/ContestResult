class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int[] cut = new int[cuts.length + 2];
        System.arraycopy(cuts, 0, cut, 1, cuts.length);
        cuts = cut;
        cut[0] = 0;
        cut[cut.length-1] = n;
        
int[][] dp = new int[cuts.length][cuts.length];
        for (int[] nums: dp) {
            Arrays.fill(nums, Integer.MAX_VALUE);
        }
        for (int i = 0; i < cuts.length - 1; i ++) {
            dp[i][i+1] = 0;
        }
        for (int cutNum = 1; cutNum <= cuts.length - 2; cutNum ++) {
            for (int left = 0; left < cuts.length - cutNum - 1; left ++) {
                int right = left + cutNum + 1;
                for (int mid = left + 1; mid < right; mid ++) {
                    
dp[left][right] = Math.min(dp[left][right],
                            dp[left][mid] + dp[mid][right] + cuts[right] - cuts[left]);
                }
            }
        }
//        System.out.println(Arrays.deepToString(dp));
        return dp[0][cuts.length-1];
    }
