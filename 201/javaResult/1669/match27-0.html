class Solution {
    public int minCost(int n, int[] cuts) {
        int p = cuts.length;
        int[] arr = new int[p + 2];
        for(int i = 0; i < p; i++){
            arr[i] = cuts[i];
        }

        arr[p] = 0;
        arr[p + 1] = n;
        Arrays.sort(arr);
        p = arr.length;

        int[][] dp = new int[p][p];
        for(int i = 0; i < p; i++){
            Arrays.fill(dp[i], Integer.MAX_VALUE / 2);
        }

        for(int i = 1; i < p; i++){
            dp[i-1][i] = 0;
        }
        for(int i = 2; i < p; i++){
            for(int j = 0; j + i < p; j++){
                for(int k = j + 1; k < j + i; k++){
                    dp[j][j + i] = Math.min(dp[j][j + i],arr[j + i] - arr[j] + dp[j][k] + dp[k][j + i]);
                }
            }
        }


        return dp[0][p - 1];
    }
}
