class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int m = cuts.length+1;
        int[] st = new int[m];
        st[0] = cuts[0];
        for (int i = 1; i < m-1; ++i)
            st[i] = cuts[i]-cuts[i-1];
        st[m-1] = n-cuts[m-2];
        int[][] ss = new int[m][m];
        for (int i = 0; i < m; ++i) {
            ss[i][i] = st[i];
            for (int j = i+1; j < m; ++j)
                ss[i][j] = ss[i][j-1]+st[j];
        }
        
int[][] dp = new int[m][m];
        for (int z = 1; z < m; ++z) {
            for (int i = 0; i < m; ++i) {
                int j = i+z;
                if (j>=m)
                    break;
                
dp[i][j] = 100000000;
                for (int k = i+1; k <= j; ++k) {
                    dp[i][j] = Math.min(dp[i][j],dp[i][k-1]+dp[k][j]+ss[i][j]);
                }
            }
        }
        return dp[0][m-1];
    }
}
