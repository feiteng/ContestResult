class Solution {
    public int minCost(int n, int[] cuts) {
        int c = cuts.length;
        int[] cs = new int[c+2];
        for (int i = 0; i < c; i++)
            cs[i] = cuts[i];
        cs[c] = 0;
        
cs[c+1] = n;
        Arrays.sort(cs);
        int m = cs.length;
        int[][] dp = new int[m][m];
        for (int l = 2; l < m; l++) {
            for (int i = 0; i < m-l; i++) {
                int j = i + l;
                dp[i][j] = Integer.MAX_VALUE;
                int cost = cs[j] - cs[i];
                
for (int k = i+1; k < j; k++) {
                    dp[i][j] = Math.min(dp[i][j], cost+dp[i][k]+dp[k][j]);
                }
            }
        }
        return dp[0][m-1];
    }
}
