class Solution {
    public char findKthBit(int n, int k) {
        String init = "0";
        for (int i = 0; i < n; i++) {
            StringBuilder sb = new StringBuilder(init);
            sb.reverse();
            StringBuilder nxt = new StringBuilder();
            nxt.append(init);
            nxt.append("1");
            for (int j = 0; j < sb.length(); j++) {
                nxt.append((char) ('0' + ('1' - sb.charAt(j))));
            
}

            init = nxt.toString();
        }

        return init.charAt(k - 1);
    }
}
