class Solution {
    public char findKthBit(int n, int k) {
        String s = "0";
        for (int i = 1; i<= n; i++){
            s = 
getNext(s);
        }
        return s.charAt(k-1);
    }

    private String getNext(String s){
        StringBuilder reverse = new StringBuilder();
        for (int i = s.length() - 1; i >= 0; i--){
            if(s.charAt(i) == '0') reverse.append("1");
            else 
reverse.append("0");
        }
        return s + "1" + reverse.toString();
    }
}
