class Solution {
    public char findKthBit(int n, int k) {
        return find(n, k - 1);
    }
    
    
public char find(int n, int idx) {
        if (n == 1) {
            return '0';
        }
        int len = (int) Math.pow(2, n) - 1;
        if (idx == len / 2) {
            return '1';
        } else if (idx > len / 2) {
            char c = find(n - 1, len - 1 - idx);
            return c == '1' ? '0' : '1';
        
} else {
            return find(n - 1, idx);
        }
    }
}
