class Solution {
    public char findKthBit(int n, int k) {
        byte[] t = new byte[1 << 22];
        
        int end = 0;
        for (int i = 1; i <= n; i++) {
            t[end + 1] = 1;
            int newEnd = (end + 1) * 2;
            for (int j = end + 2; j <= newEnd; j++) {
                byte a = t[end - (j - 2 - end)];
                t[j] = (byte) (a == 1 ? 0 : 1);
            }
            end = newEnd;
        }
        
        return t[k - 1] == 1 ? '1' : '0';
    }
}
