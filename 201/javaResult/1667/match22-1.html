class Solution {
    public char findKthBit(int n, int k) {
        return sln1(n,k);
    }
    
    
private char sln1(int n ,int k){
        List<Integer> list = new ArrayList<>();
        list.add(0);
        for(int i =1;i< n;i++){
            list.add(1);
            for(int l =list.size()-1, j = 0;l>0;l--,j++){
                list.add(
list.get(l-1)^1);
            }
        }
        //for(int i: list) System.out.print(i);
        //System.out.println();
        return (char)(list.get(k-1)+'0');
    }
}
