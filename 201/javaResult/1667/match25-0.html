class Solution {

    public char findKthBit(int n, int k) {
        StringBuilder s = new StringBuilder("0");
        while (s.length() < k) {
            s.append("1");
            for (int i = s.length() - 2; i >= 0; i--) {
                s.append(s.charAt(i) == '0' ? "1" : "0");
            }
        }
        return s.charAt(k - 1);
    }
}
