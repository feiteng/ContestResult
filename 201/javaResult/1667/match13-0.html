class Solution {
    public char findKthBit(int n, int k) {
        String prev = "0";
        n--;
        while(n > 0) {
            prev = getNext(prev);
            n--;
        }
        return prev.charAt(k - 1);
    }
    private String getNext(String prev) {
        StringBuilder next = new StringBuilder(prev);
        next.append('1');
        for(int i = prev.length() - 1; i >= 0; i--) {
            next.append(prev.charAt(i) == '0' ? '1' : '0');
        }
        return next.toString();
    }
}
