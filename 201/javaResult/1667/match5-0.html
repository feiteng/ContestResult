class Solution {
    public char findKthBit(int n, int k) {
        String s = "0";
        for (int i = 1; i < n; i++) {
            s = next(s);
        }
        return s.charAt(k - 1);
    }
    
    private String next(String s) {
        return s + "1" + help(s);
    }
    
    private String help(String s) {
        StringBuilder sb = new StringBuilder();
        for (int i = s.length() - 1; i >= 0; i--) {
            if (s.charAt(i) == '0') sb.append('1');
            else sb.append('0');
        }
        return sb.toString();
    }
}
