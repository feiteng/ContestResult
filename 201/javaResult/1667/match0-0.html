class Solution {
    public char findKthBit(int n, int k) {
        if (n == 1) {
            return '0';
        }
        int num = 1 << (n - 1);
        if (k == num) {
            return '1';
        } else if (k < num) {
            return findKthBit(n - 1, k);
        } else {
            k -= num;
            k = num - k;
            return findKthBit(n - 1, k) == '0' ? '1' : '0';
        }
    }
}
