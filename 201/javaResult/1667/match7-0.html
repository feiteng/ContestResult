class Solution {
    public char findKthBit(int n, int k) {
        int[] len = new int[n];
        len[0] = 1;
        for (int i = 1; i < n; i++) len[i] = len[i-1] * 2 + 1;
        return solve(n, k, len);
    }
    
    public char solve(int n, int k, int[] len) {
        if (n == 1) return '0';
        if (k == len[n-1] / 2 + 1) return '1';
        if (k <= len[n-2]) return solve(n-1, k, len);
        return rev(solve(n-1, len[n-1] + 1 - k, len));
    }
    
    public char rev(char c) {
        return c == '0' ? '1' : '0';
    }
    
}
