class Solution {
    public char findKthBit(int n, int k) {
        if (n == 1) {
            return '0';
        }
        int mid = (2 << (n - 2));
        
if (k == mid) {
            return '1';
        } else if (k < mid) {
            return findKthBit(n - 1, k);
        } else {
            return invert(findKthBit(n - 1, mid - (k - mid)));
        }
    }
    private char invert(char a) {
        return a == '0' ? '1' : '0';
    }
}
