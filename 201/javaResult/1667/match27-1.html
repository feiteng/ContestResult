class Solution {
    
    public char findKthBit(int n, int k) {
        String x = "0";
        for(int i=2; i<=n; i++) {
            StringBuilder sb = new StringBuilder();
            sb.append(x);
            sb.append("1");
            for(int j=x.length()-1; j>=0; j--) {
                if(x.charAt(j)=='0') sb.append('1');
                else sb.append('0');
            
}
            x = sb.toString();
        } 
        // System.out.println(x);
        return x.charAt(k-1);
    }
}
