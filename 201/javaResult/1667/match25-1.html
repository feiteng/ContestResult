class Solution {
    public char findKthBit(int n, int k) {
        StringBuilder s = new StringBuilder("0");
        while (k > s.length()) {
            int len = s.length();
            s.append(1);
            for (int i = len - 1; i >= 0; i--) {
                s.append(1 - (
s.charAt(i) - '0'));
            }
        }
        return s.charAt(k - 1);
    }
}
