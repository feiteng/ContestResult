class Solution {
    public char findKthBit(int n, int k) {
        // if (n == 0) {
        //     return '0';
        // }
        if (k == 1) {
            return '0';
        }
        int len = (int)Math.pow(2, n) - 1; 
        
int mid = (len + 1) / 2;
        if (k == mid) {
            return '1';
        } else if (k < mid) {
            return findKthBit(n - 1, k);
        } else {
            
return findKthBit(n - 1, 2 * mid  - k) == '0' ? '1' : '0';
        }
    }
}
