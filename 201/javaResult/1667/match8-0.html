class Solution {
    public char findKthBit(int n, int k) {
        int[] len = new int[n+1];
        len[1] = 1;
        for (int i = 2; i <= n; i++) {
            len[i] = 2*len[i-1] + 1;
        }
        
        return aux(n, k, len, false);
    }
    
    private char aux(int n, int k, int[] len, boolean rev) {
        // System.out.println(n + " " + k + " " + rev);
        if (n == 1) {
            return rev ? '1' : '0';
        }
        
        int l = len[n];
        if (k == (l+1)/2) return rev ? '0' : '1';
        if (k < (l+1)/2) return aux(n-1, k, len, rev);
        return aux(n-1, l - k + 1, len, !rev);
    }
}
