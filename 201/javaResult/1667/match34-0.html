class Solution {
    public char findKthBit(int n, int k) {
        len = new int[n + 1];
        len[1] = 1;
        for(int i = 2; i <= n; i++){
            len[i] = len[i - 1] * 2 + 1;
        }
        int ans = find(n, k);
        return (char)ans;
    }
    
    public int find(int n, int k){
        if(n == 1){
            return '0';
        }
        if(len[n - 1] >= k){
            return find(n - 1, k);
        }
        if(len[n - 1] + 1 == k){
            return '1';
        }
        k -= len[n - 1] + 1;
        int c = find(n - 1, len[n - 1] + 1 - k);
        return '1' - c + '0';
    }
    
    int[] len;
}
