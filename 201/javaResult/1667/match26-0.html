class Solution {
    public char findKthBit(int n, int k) {
        String s = "0";
        for (int i = 0; i < n - 1; i++){
            StringBuilder revInv = new StringBuilder();
            for (int j = s.length() - 1; j >= 0; j--)
                revInv.append((char) ('0' + '1' - s.charAt(j)));
            s = s + "1" + revInv.toString();
        }
        //System.out.println(s);
        return s.charAt(k - 1);
    }
}
