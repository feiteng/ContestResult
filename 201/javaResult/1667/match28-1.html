class Solution {
    public char findKthBit(int n, int k) {
        boolean[] arr = new boolean[1];
        int len = 1;
        
        for (int i = 1; i < n; i++) {
            len = len * 2 + 1;
            
            boolean[] next = new boolean[len];
            int p = 0;
            for (int j = 0; j < arr.length; j++) {
                next[p++] = arr[j];
            }
            next[p++] = true;
            for (int j = arr.length - 1; j >= 0; j--) {
                next[p++] = !arr[j];
            
}
            
            arr = next;
        }
        return arr[k - 1] ? '1' : '0';
    }
}
