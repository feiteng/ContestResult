class Solution {
    public char findKthBit(int n, int k) {
        String s = "0";
        for (int i = 2;i <= n;i++) {
            s = s + "1" + reverse(invert(s));
        }
        
return s.charAt(k - 1);
    }
    
    String invert(String s) {
        StringBuilder res = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (c == '0') res.append(1);
            else res.append(0);
        
}
        return res.toString();
    }
    
    String reverse(String s) {
        StringBuilder res = new StringBuilder();
        for (int i = s.length() - 1;i >= 0;i--) {
            char c = s.charAt(i);
            res.append(c);
        }
        return res.toString();
    }
}
