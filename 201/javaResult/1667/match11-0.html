class Solution {
    public char findKthBit(int n, int k) {
        return findK(n, k - 1);
    }
    
    private static char findK(int n, int k) {
        if (n == 1) {
            return '0';
        }
        int sz = (1 << n) - 1;
        if (k == sz / 2) {
            return '1';
        }
        if (k < sz / 2) {
            return findK(n - 1, k);
        } else {
            char c = findK(n - 1, sz - k - 1);
            if (c == '0') {
                return '1';
            } else {
                return '0';
            }
        }
    }
}
