/*
S1 = "0"
S2 = "011"
S3 = "0111001"
S4 = "011100110110001"
*/
class Solution {
    public char findKthBit(int n, int k) {
        if (k == 1) return '0';
        
StringBuilder sb = new StringBuilder();
        sb.append('0');
        for (int i = 2; i <= n; i++) {
            sb.append('1');
            for (int j = sb.length() - 2; j >=0; j--) {
                sb.append(sb.charAt(j)  == '0'? '1' : '0');
            }
           // System.out.println(sb.toString());
        }
        

        
        return sb.charAt(k-1);
    }
}
