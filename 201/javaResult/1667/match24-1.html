class Solution
{
    private int solve(int n, int k)
    {
        if (n == 1) return 0;
        int len = (1 << n)  - 1;
        if (k == len / 2) return 1;
        else if (k < len / 2) return solve(n - 1, k);
        else return 1 ^ solve(n - 1, len - k -  1);
    
}

    public char findKthBit(int n, int k)
    {
        char res =  (char)(solve(n, k - 1) + '0');
        return res;
    }
}
