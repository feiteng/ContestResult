class Solution {



	public char findKthBit(int n, int k) {
		int res = getN(n, k-1);
		
		return  res > 0 ? '1':'0';
	}


	int getN(int n, int k){
		if(n == 1){
			return 0;
		}
		int mid = power(2, n-1) -1;
		if(k == mid){
			return 1;
		}
		
		if(k < mid){
			return getN(n-1, k);
		}else{
			return  1-getN(n-1, 2*mid -k);
		}
		
	
	}
	
	int power(int x, int y){
		if(y== 0){
			return 1;
		}
		
		return x*power(x, y-1);
	}
}
