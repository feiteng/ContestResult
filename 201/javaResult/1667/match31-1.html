class Solution {
    public char findKthBit(int n, int k) {
        if (k == 1) {
            return '0';
        }
        int len = (int) Math.pow(2, n) - 1;
        if (k == len / 2 + 1) {
            return '1';
        }
        if (find(n, k) == 0) {
            return '0';
        }
        else {
            return '1';
        }
    }

    private int find(int num, int k) {
        if (num == 1) {
            return 0;
        }
        int len = (int) Math.pow(2, num) - 1;
        if (k == len) {
            return 1;
        }
        int mid = len / 2 + 1;
        if (k == mid) {
            return 1;
        }
        
if (k > mid) {
            return 1 - find(num - 1, len - k + 1);
        }
        else {
            return find(num - 1, k);
        }
    }
}
