class Solution {
  public static void main(String args[]) {
    Solution sol = new Solution();
    int n = 4;
    for (int i = 1; i <= pow2[n] - 1; i++) {
      System.out.print(sol.findKthBit(n, i));
    }
    System.out.println("\n011100110110001");
  }
  public char findKthBit(int n, int k) {
    return findKthBit2(n, k - 1);
  
}
  
  public char findKthBit2(int n, int k) {
    if (n <= 1) {
      return '0';
    }
    
    int mid = pow2[n - 1] - 1;
    
if (k < mid) {
      return findKthBit2(n - 1, k);
    }
    if (k == mid) {
      return '1';
    }
    
    k -= mid;
    char result = findKthBit2(n - 1, mid - k);
    return result == '0' ? '1' : '0';
  }
  
  static int[] pow2 = new int[21];
  
static {
    pow2 = new int[21];
    pow2[0] = 1;
    for (int i = 1; i < pow2.length; i++) {
      pow2[i] = 2 * pow2[i - 1];
    }
  }
}
