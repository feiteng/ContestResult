class Solution {
    public char findKthBit(int n, int k) {
        if (n == 1) {
            return '0';
        }
        int half = ((1 << n) - 1) / 2;
        System.out.println(half + " " + k);
        if (k == half+1) {
            return '1';
        } else if (k > half+1) {
            return (char)('0'+('1'-findKthBit(n-1, half+1-k+half+1)));
        } else {
            return findKthBit(n-1, k);
        }
    }
}
