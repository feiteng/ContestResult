class Solution {
    public char findKthBit(int n, int k) {
        int[] len = new int[n+1];
        len[0] = 0;
        for (int i = 1; i <= n; i++) len[i] = 2*len[i-1] + 1
;
        return (char)(helper(len, n, k) + '0');
    }
    public int helper(int[] len, int n, int k){
        if (n == 1 && k == 1) return 0;
        if (k == len[n]/2 + 1) return 1
;
        if (k > len[n]/2 + 1) return 1^helper(len, n, len[n]+1-k);
        return helper(len, n-1, k);
    }
}
