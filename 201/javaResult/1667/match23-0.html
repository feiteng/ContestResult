class Solution {
    public char findKthBit(int n, int k) {
        
        String ret = "0"; 
        for (int i = 2; i <= n; i++) {
            ret = ret + "1" + calc(ret); 
        }
        // System.out.println(ret);
        return ret.charAt(k-1);
    }
    private String calc(String s) {
        
       StringBuilder sb = new StringBuilder(); 
        char[] cs = s.toCharArray(); 
        for (char c : cs) {
            if (c == '1') {
                sb.append("0");
            } else {
                sb.append("1");
            }
        }
        return sb.reverse().toString();
    }
}
