class Solution {
    /*
    0      1       n = 0
    
    011      3     n = 1
    
    0111001     7    n = 2
    
    011100110110001   2 * 7 + 1 = 15 n = 3
    
    ...
    
    011100110110001
    ->
    0111001 1 011000 1 11100100110001
            |        |
            |        |
            |        |
            
       |
       |
       
       
    
    
    given n
    length is
         2 * len(n-1) + 1
         
         2 * 2 * len(n - 2) + 2 + 1
         
         2 ^ n + 2 ^ n - 1
         
    k === center ------> 1
    
       
    
    */
    public char findKthBit(int n, int k) {
        return (char) (find(n, k) + '0');
    
}
    
    private int find(int n, int k) {
        if (n == 0) return 0;
        
        int length = (int) (2 * Math.pow(2, n) - 1);
        
int center = length / 2;
        if (k == center + 1) return 1;
        if (k > center + 1) return 1 - find(n - 1, length - k + 1);
        return find(n - 1, k);
    }
}
