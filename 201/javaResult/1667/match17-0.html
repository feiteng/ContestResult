class Solution {
    public char findKthBit(int n, int k) {   
        int ans = process(n,k);
        return (char)(ans+'0');
    }
    
    private int process(int n, int k){
        if(n == 1) return 0;
        int half = (1<<(n-1));
        if(k == half) return 1;
        if(k < half) return process(n-1,k);
        return ((process(n-1,2*half - k))+1)%2;
        
    }
}
