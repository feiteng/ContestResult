class Solution {
    public ListNode swapNodes(ListNode head, int k) {
        int len = lengthOfList(head);
        ListNode n1 = findTarget(head, k), n2 = findTarget(head, len - k + 1);
        int tmp = n2.val;
        n2.val = n1.val;
        n1.val = tmp;
        return head;
    }

    private int lengthOfList(ListNode head) {
        int res = 0;
        while (head != null) {
            res += 1;
            head = head.next;
        }
        return res;
    }

    private ListNode findTarget(ListNode head, int target) {
        while (head != null && target != 1) {
            head = head.next;
            --target;
        }
        return head;
    }
}
