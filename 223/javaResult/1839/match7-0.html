class Solution {
  public int[] decode(int[] encoded, int first) {
    int[] ret = new int[encoded.length + 1];
    ret[0] = first;
    for (int i = 0; i < encoded.length; i++) {
      //encoded[i] = ret[i] ^ ret[i + 1]
      // encoded[i] ^ ret[i] = ret[i + 1]
      ret[i + 1] = encoded[i] ^ ret[i];
    }
    
    return ret;
  }
}
