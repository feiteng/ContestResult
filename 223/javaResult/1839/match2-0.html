class Solution {
    public int[] decode(int[] encoded, int first) {
        int[] re = new int[encoded.length + 1];
        re[0] = first;
        for (int i = 0; i < encoded.length; i++) {
            re[i+1] = first ^ encoded[i];
            first = re[i+1];
        }
        return re;
    }
}
