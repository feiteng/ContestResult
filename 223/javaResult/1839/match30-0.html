class Solution {
    public int[] decode(int[] e, int first) {
        int[] out = new int[e.length+1];
        out[0] =first;
        for(int i = 1,j = 0;j<e.length;i++,j++){
            first = first^e[j];
            out[i] = first;     
        }
        return out;
    }
}
