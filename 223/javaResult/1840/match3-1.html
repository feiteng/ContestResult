class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        UF uf = new UF(n);
        for (int[] pair : allowedSwaps) {
            uf.union(pair[0], pair[1]);
        }
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int r = uf.find(i);
            map.putIfAbsent(r, 
new HashMap<>());
            Map<Integer, Integer> cnt = map.get(r);
            cnt.put(target[i], cnt.getOrDefault(target[i], 0) + 1);
        }
        int dist = n;
        for (int i = 0; i < n; i++) {
            int r = uf.find(i);
            Map<Integer, Integer> cnt = map.get(r);
            if (cnt.containsKey(source[i])) {
                cnt.put(source[i], cnt.get(source[i]) - 1);
                dist--;
                if (cnt.get(source[i]) == 0) {
                    cnt.remove(source[i]);
                }
            }
        }
        return dist;
    }
    
    
class UF {
        int[] root;
        UF(int n) {
            root = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
        }
        
        int find(int i) {
            if (root[i] == i) {
                return i;
            }
            int r = find(root[i]);
            root[i] = r;
            
return r;
        }
        
        void union(int i, int j) {
            int ip = find(i);
            int jp = find(j);
            if (jp != ip) {
                root[jp] = ip;
            }
        }
    }
}
