class Solution {
    
    public void union(int[] uf, int x, int y) {
        int a = find(uf, x), b = find(uf, y); 
        if(a == b) return ; 
        uf[a] = uf[b];
    }
    public int find(int[] uf, int x) {
        if(uf[x] == x) return uf[x]; 
        uf[x] = find(uf, uf[x]); 
        return uf[x];
    }
    public int minimumHammingDistance(int[] s, int[] t, int[][] r) {
        int[] uf = new int[s.length]; 
        for(int i = 0; i < uf.length; i++) uf[i] = i;
        for(int[] i: r) {
            union(uf, i[0], i[1]); 
        }
        
        Map<Integer, List<Integer>> map = new HashMap<>(); 
        for(int i = 0; i < uf.length; i++) {
            int root = find(uf, i); 
            map.putIfAbsent(root, new ArrayList<>()); 
            map.get(root).add(i); 
        }
        int ret = 0; 
        for(int key : map.keySet()) {
            int sz = map.get(key).size();
            List<Integer> index = map.get(key); 
            //for(int i : index) System.out.print(i + " ");
            Map<Integer, Integer> tmp = new HashMap<>(); 
            for(int i = 0; i < sz; i++) {
                int val = s[index.get(i)];
                tmp.putIfAbsent(val, 0); 
                tmp.put(val, tmp.get(val) + 1); 
            }
            for(int i = 0; i < sz; i++) {
                int val = t[index.get(i)]; 
                if(!tmp.containsKey(val)) {
                    ret++;
                } else {
                    tmp.put(val, tmp.get(val) - 1); 
                    if(tmp.get(val) == 0) tmp.remove(val);
                }
            }
            
        }
        return ret;
    }
}
