class Solution {
    private class DJS{
        int[] parent;
        Map<Integer, Integer>[] maps;
        
        DJS(int n, int[] src){
            parent = new int[n];
            maps = new Map[n];
            for(int i = 0; i < n; i++){
                parent[i] = i;
                maps[i] = new HashMap<>();
                maps[i].put(src[i], maps[i].getOrDefault(src[i], 0)+1);
            }
        }
        
        int find(int i){
            if(parent[i] != i){
                parent[i] = find(parent[i]);
            }
            return parent[i];
        }
        
        void union(int i, int j){
            int parI = find(i);
            int parJ = find(j);
            if(parI == parJ){
                return;
            }
            parent[parI] = parJ; 
            for(Map.Entry<Integer, Integer> count: maps[parI].entrySet()){
                maps[parJ].put(count.getKey(), maps[parJ].getOrDefault(count.getKey(), 0) + count.getValue());
            }
            maps[parI].clear();
        }
    }
    
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        DJS djs = new DJS(n, source);
        for(int[] curr: allowedSwaps){
            djs.union(curr[0], curr[1]);
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            int curPar = djs.find(i);
            if(djs.maps[curPar].containsKey(target[i])){
                int count = djs.maps[curPar].get(target[i]);
                if(count == 1){
                    djs.maps[curPar].remove(target[i]);
                }else{
                    djs.maps[curPar].put(target[i], count - 1);
                }
            }else{
                res++;
            }
        }
        return res;
    }
}
