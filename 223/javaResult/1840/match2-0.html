class Solution {
    int[] parents;
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = target.length;
        parents = new int[n];
        for(int i=0; i<n; i++){
            parents[i] = i;
        }
        for(int[] arr : allowedSwaps){
            union(arr[0],arr[1]);
        }
        // System.out.println(Arrays.toString(parents));
        Map<Integer,Map<Integer,Integer>> groupMap = new HashMap<>();
        for(int i=0; i<n; i++){
            int groupId = find(i);
            groupMap.putIfAbsent(groupId,new HashMap<Integer,Integer>());
            Map<Integer,Integer> map = groupMap.get(groupId);
            map.put(source[i],map.getOrDefault(source[i],0)+1);
            map.put(target[i],map.getOrDefault(target[i],0)-1);
        }
        int res = 0;
        for(Map<Integer,Integer> map : groupMap.values()){
            // System.out.println(map);
            int c = 0;
            for(int val : map.values()){
                c += Math.abs(val);
            }
            res+= c/2;
        }
        return res;
    }
    
    private void union(int x, int y){
        int px = find(x);
        int py = find(y);
        parents[px] = py; 
    }
    
    private int find(int x){
        if(parents[x] != x){
            parents[x] = find(parents[x]);
        }
        return parents[x];
    }
}
