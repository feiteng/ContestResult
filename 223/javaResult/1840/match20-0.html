class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        Union u = new Union(n);
        
        for (int[] s : allowedSwaps) u.union(s[0], s[1]);
        
        Map<Integer, Map<Integer, LinkedList<Integer>>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int p = u.getParent(i);
            map.computeIfAbsent(p, k -> new HashMap<>())
                .computeIfAbsent(source[i], k -> new LinkedList<>())
                .addLast(i);
        }
        
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            int p = u.getParent(i);
            Map<Integer, LinkedList<Integer>> m = map.computeIfAbsent(p, k -> new HashMap<>());
            
            if (m.containsKey(target[i]) && !m.get(target[i]).isEmpty()) {
                m.get(target[i]).removeFirst();
            } else {
                res++;
            }
        }
        
        return res;
    }
    
    private static class Union {
        int[] parents;
        
        Union(int n) {
            parents = new int[n];
            for (int i = 0; i < n; i++) parents[i] = i;
        }
        
        int getParent(int i) {
            if (parents[i] != i) parents[i] = getParent(parents[i]);
            return parents[i];
        }
        
        void union(int i1, int i2) {
            int p1 = getParent(i1), p2 = getParent(i2);
            if (p1 != p2) {
                parents[p1] = p2;
            }
        }
    }
}
