class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        UF uf = new UF(source.length);
        for (int[] swap : allowedSwaps) {
            uf.union(swap[0],swap[1]);
        }
        Map<Integer, Map<Integer, Integer>> m = new HashMap<>();
        for (int i = 0; i < source.length; i ++) {
            m.putIfAbsent(uf.find(i), new HashMap<Integer, Integer>());
            Map<Integer, Integer> map = m.get(uf.find(i));
            map.put(source[i], map.getOrDefault(source[i],0)+1);
            map.put(target[i], map.getOrDefault(target[i],0)-1);
        }
        
        int ans = 0;
        for(Map<Integer, Integer> map : m.values()) {
            for (Integer v : map.values()) {
                ans += Math.abs(v);
            }
        }
        return ans/2;
    }
    
    class UF {
        int[] p;
        UF(int n) {
            p = new int[n];
            for (int i = 0; i < n; i ++) {
                p[i] = i;
            }
        }
        
        int find(int x) {
            while (p[x] != x) {
                p[x] = p[p[x]];
                x = p[x];
            }
            return x;
        }
        
        void union(int x, int y) {
            int px = find(x);
            int py = find(y);
            if (px == py)
                return;
            p[px] = py;
        }
    }
}
