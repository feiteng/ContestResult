class Solution {
    int[] label;
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        label = new int[n];
        for (int i = 0; i < n; i++) {
            label[i] = i;
        }
        
        for (int[] swap : allowedSwaps) {
            union(swap[0], swap[1]);
        }
        
        Map<Integer, Map<Integer, Integer>> map1 = new HashMap<>();
        Map<Integer, Map<Integer, Integer>> map2 = new HashMap<>();
        
        for (int i = 0; i < n; i++) {
            int key = find(i);
            if (!map1.containsKey(key)) {
                map1.put(key, new HashMap<>());
                map2.put(key, new HashMap<>());
            }
            Map<Integer, Integer> counter1 = map1.get(key);
            Map<Integer, Integer> counter2 = map2.get(key);
            
counter1.put(source[i], counter1.getOrDefault(source[i],0) + 1);
            counter2.put(target[i], counter2.getOrDefault(target[i],0) + 1);
        }
        
        
        int ans = 0;
        for (int key : map1.keySet()) {
            Map<Integer, Integer> counter1 = map1.get(key);
            Map<Integer, Integer> counter2 = map2.get(key);
            for (int key2 : counter1.keySet()) {
                if (!counter2.containsKey(key2)) {
                    ans += counter1.get(key2);
                } else {
                    int count1 = counter1.get(key2);
                    int count2 = counter2.get(key2);
                    ans += Math.max(0, count1 - count2);
                }
            
}
        }
        return ans;
    }
    
    private int find(int i) {
        while(label[i] != i) {
            
i = label[i];
        }
        return i;
    }
    
    private void union(int i, int j) {
        int pi = find(i);
        int pj = find(j);
        label[pi] = pj;
    }
}
