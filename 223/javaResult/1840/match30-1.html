class Solution {
    
public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        nodes = new int[n];
        for (int i = 0; i < n; i++) {
            nodes[i] = i;
        }
        for (int[] allowedSwap : allowedSwaps) {
            union(allowedSwap[0], allowedSwap[1]);
        }
        HashMap<Integer, HashMap<Integer, Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int r = find(i);
            map.putIfAbsent(r, new HashMap<>());
            HashMap<Integer, Integer> maplet = map.get(r);
            
maplet.put(source[i], maplet.getOrDefault(source[i], 0) + 1);
            maplet.put(target[i], maplet.getOrDefault(target[i], 0) - 1);
        }
        int ret = 0;
        for (HashMap<Integer, Integer> maplet : map.values()) {
            for (int i : maplet.values()) {
                ret += Math.abs(i);
            }
        }
        return ret/2;
    }

    
private int[] nodes;

    private int find(int i) {
        if (nodes[i] != i) {
            nodes[i] = find(nodes[i]);
        }
        return nodes[i];
    }

    private void union(int i, int j) {
        nodes[find(i)] = find(j);
    }
}
