class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        DSU dsu = new DSU(source.length);
        
for(int[] swap : allowedSwaps) {
            dsu.union(swap[0], swap[1]);
        }
        Map<Integer, Map<Integer, Integer>> freq = new HashMap<>();
        for(int i = 0; i < source.length; i++) {
            int parent = dsu.findParent(i);
            freq.putIfAbsent(parent, new HashMap<>());
            Map<Integer, Integer> freqMap = freq.get(parent);
            freqMap.put(source[i], freqMap.getOrDefault(source[i], 0) + 1);
            freqMap.put(target[i], freqMap.getOrDefault(target[i], 0) - 1);
        }
        // System.out.println(freq);
        int output = 0;
        for(Map<Integer, Integer> v : freq.values()) {
            if(v == null) continue;
            int temp = 0;
            
for(int vv : v.values()) {
                temp += Math.abs(vv);
            }
            output += temp / 2;
        }
        return output;
    }
    
    class DSU {
        int[] parent;
        DSU(int n) {
            parent = new int[n];
            for(int i = 0; i < n; i++) parent[i] = i;
        }
        
        
int findParent(int n) {
            if(parent[n] != n) {
                parent[n] = findParent(parent[n]);
            }
            return parent[n];
        }
        
        void union(int a, int b) {
            parent[findParent(a)] = parent[findParent(b)];
        }
        
    }
}
