class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        int[] par = new int[n];
        for(int i = 0; i < n; i ++){
            par[i] = i;
        }
        
        for(int[] a : allowedSwaps){
            int x = find(par,a[0]);
            int y = find(par,a[1]);
            if(x != y){
                par[x] = y;
            }
        }
//        fatherIdx, index
        HashMap<Integer,List<Integer>> map = new HashMap<>();
        for(int i = 0; i < n; i ++){
            int pp = find(par,i);
            map.putIfAbsent(pp,new ArrayList<>());
            map.get(pp).add(i);
        }
        int ret = 0;
        for(int key : map.keySet()){
            List<Integer> cur = map.get(key);
            HashMap<Integer,Integer> temp = new HashMap<>();
            for(int i = 0; i < cur.size(); i ++){
                int idx = cur.get(i);
                temp.put(source[idx],temp.getOrDefault(source[idx],0)+1);
                temp.put(target[idx],temp.getOrDefault(target[idx],0)-1);
            }
            for(int k : temp.keySet()){
                ret += Math.abs(temp.get(k));
            }
            
        }
        return ret/2;
    }
    private int find(int[] par, int i){
        if(par[i] == i) return i;
        par[i] = find(par,par[i]);
        return par[i];
    }
}
