class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        int[] parent = new int[n];
        for(int i=0;i<n;i++) {
            parent[i] = i;
        }
        for(int i=0;i<allowedSwaps.length;i++) {
            union(parent,allowedSwaps[i][0],allowedSwaps[i][1]);
        }
        
HashMap<Integer,ArrayList<Integer>> hm = new HashMap<>();
        for(int i=0;i<n;i++) {
            int p = find(parent,i);            
            if(hm.containsKey(p)) {
                hm.get(p).add(i);
            } else {
                hm.put(p,new ArrayList<>());
                hm.get(p).add(i);
            }
        }
        int count = 0;
        for(Integer key:hm.keySet()) {
            ArrayList<Integer> arr = hm.get(key);
            HashMap<Integer,Integer> cnt = new HashMap<>();
            
for(Integer i:arr) {
                cnt.put(target[i],cnt.getOrDefault(target[i],0)+1);
            }
            for(Integer i:arr) {
                cnt.put(source[i],cnt.getOrDefault(source[i],0)-1);
            }
            for(Integer v:cnt.values()) {
                if(v>0) count+= v
;
            }
        }
        return count;
    }
    
    int find(int[] parent,int p) {
        if(parent[p] == p) {
            return p;
        } else {
            parent[p] = find(parent,parent[p]);
            return parent[p];
        }
    }
    
    void union(int[] parent,int a,
int b) {
        int pa = find(parent,a);
        int pb = find(parent,b);
        if(pa != pb) {
            parent[pa] = pb;
        }
    }
}
