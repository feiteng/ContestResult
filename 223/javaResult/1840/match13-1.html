class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int res = 0;
        
int[] dst = new int[source.length];
        for(int i = 0; i < dst.length; i++) dst[i] = i;
        for(int i = 0; i < allowedSwaps.length; i++) {
            union(dst, allowedSwaps[i][0], allowedSwaps[i][1]);
        }
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int i = 0; i < source.length; i++) {
            int root = find(dst, i);
            map.putIfAbsent(root, new ArrayList<>());
            map.get(root).add(i);
        }
        for(Map.Entry<Integer, List<Integer>> e : map.entrySet()) {
            List<Integer> l = e.getValue();
            Map<Integer, Integer> valueCount = new HashMap<>();
            for(int i : l) {
                
valueCount.put(source[i], valueCount.getOrDefault(source[i], 0) + 1);
            }
            for(int i : l) {
                int count = valueCount.getOrDefault(target[i], 0);
                if(count == 0) res++;
                else {
                    valueCount.put(target[i], count - 1);
                
}
            }
        }
        return res;
    }
    
    
private void union(int[] dst, int a, int b) {
        int ra = find(dst, a);
        int rb = find(dst, b);
        if(ra == rb) return;
        dst[ra] = rb;
    }
    
    private int find(int[] dst, int a) {
        if(dst[a] == a) return a;
        dst[a] = find(dst, dst[a]);
        return dst[a];
    }
}
