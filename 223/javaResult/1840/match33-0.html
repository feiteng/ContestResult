class Solution {
    public int minimumHammingDistance(int[] A, int[] B, int[][] map) {
        int[] root = new int[A.length];
        int n = A.length;
        for (int i = 0; i < root.length; i++) {
            root[i] = i;
        }
        for (int[] m : map) {
            union(root, m[0], m[1]);
        }
        Map<Integer, List<Integer>> m1 = new HashMap<>();
        Map<Integer, List<Integer>> m2 = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int ri = find(root, i);
            m1.putIfAbsent(ri, new ArrayList<>());
            m1.get(ri).add(A[i]);
            m2.putIfAbsent(ri, new ArrayList<>());
            m2.get(ri).add(B[i]);
        }
        int res = 0;
        for (int key : m1.keySet()) {
            List<Integer> l1 = m1.get(key), l2 = m2.get(key);
            res += add(l1, l2);
        }
        return res;
    }
    
    private int add(List<Integer> l1, List<Integer> l2) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : l1) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int res = 0;
        for (int i : l2) {
            if (!map.containsKey(i)) {
                res++;
            } else {
                map.put(i, map.get(i) - 1);
                if (map.get(i) == 0) map.remove(i);
            }
        }
        return res;
    }
    
    private int find(int[] root, int i) {
        if (root[i] != i) {
            root[i] = find(root, root[i]);
        }
        return root[i];
    }
    
    private void union(int[] root, int i, int j) {
        int ri = find(root, i), rj = find(root, j);
        if (ri != rj) {
            root[ri] = rj;
        }
    }
}
