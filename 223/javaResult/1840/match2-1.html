class Solution {
    int[] parent;
    
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        parent = new int[n];
        for(int i = 0;i < n;i++){parent[i] = i;}
        for(int[] s : allowedSwaps){
            union(s[0],s[1]);
        }
        Map<Integer,Set<Integer>> map = new HashMap<>();
        for(int i = 0;i < n;i++){
            int p = find(i);
            map.putIfAbsent(p,new HashSet<>());
            
map.get(p).add(i);
        }
        int res = n;
        for(Set<Integer> set : map.values()){
            Map<Integer,Integer> targetMap = new HashMap<>();
            for(int i : set){
                targetMap.put(target[i],targetMap.getOrDefault(target[i],0)+1);
            }
            for(int i : set){
                if(targetMap.containsKey(source[i]) && targetMap.get(source[i]) > 0){
                    res--;
                    targetMap.put(source[i],targetMap.get(source[i])-1);
                }
            
}
        }
        return res;
    }
    
    public void union(int i,int j){
        int p1 = find(i), p2 = find(j);
        if(p1 != p2){
            parent[p1] = p2;
        }
    }
    
    public int find(int i){
        while(parent[i] != i){
            parent[i] = find(parent[i]);
            i = parent[i];
        }
        return parent[i];
    }
}
