class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        
        
int[] parents = new int[n];
        for(int i = 0; i < n; i ++) {
            parents[i] = i;
        }
        
        for(int[] i : allowedSwaps) {
            union(parents, i[0], i[1]);
        }
        
        Map<Integer, Integer>[] map = new Map[n];
        for(int i = 0; i < n; i ++) {
            int p = find(parents, i);
            if (map[p] == null) map[p] = new HashMap<>();
            
map[p].put(source[i], map[p].getOrDefault(source[i], 0) + 1);
        }
        int count = 0;
        
        for(int i = 0; i < n; i ++) {
            int p = find(parents, i);
            if (map[p].containsKey(target[i])) {
                map[p].put(target[i], map[p].get(target[i]) - 1);
                if (map[p].get(target[i]) == 0) map[p].remove(target[i]);
            
} else {
                count ++;
            }
        }
        
        return count;
    }
    
    private int find(int[] parents, int x) {
        if (x != parents[x]) {
            parents[x] = find(parents, parents[x]);
        }
        return parents[x];
    }
    
    private void union(int[] parents, int x, int y) {
        int px = find(parents, x);
        int py = find(parents, y);
        parents[py] = px;
    }
}
