class Solution {
    
class DSU {
        int n;
        int[] parent;
        DSU(int n) {
            this.n = n;
            parent = new int[n];
            for (int i = 0; i < n; i++) parent[i] = i;
        }
        int find(int i) {
            if (parent[i] != i) parent[i] = find(parent[i])
;
            return parent[i];
        }
        void union(int a, int b) {
            int xa = find(a), xb = find(b);
            parent[xb] = xa;
        }
    }
    
public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        DSU dsu = new DSU(n);
        for (int[] s : allowedSwaps) {
            dsu.union(s[0], s[1]);
        }
        Map<Integer, Map<Integer, Integer>> map = new HashMap();
        for (int i = 0; i < n; i++) {
            int p = dsu.find(i);
            if (!map.containsKey(p)) map.put(p, new HashMap());
            map.get(p).put(source[i], map.get(p).getOrDefault(source[i], 0) + 1);
            map.get(p).put(target[i], 
map.get(p).getOrDefault(target[i], 0) - 1);
        }
        int ret = 0;
        for (Map<Integer, Integer> m : map.values()) {
            for (Integer v : m.values()) {
                if (v > 0) ret += v;
            }
        }
        return ret;
    }
}
