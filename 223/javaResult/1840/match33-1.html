class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int[] dsu = new int[source.length];
        int n = source.length;
        for (int i=0;i<n;i++){
            dsu[i]=i;
        }
        for (int[] swap: allowedSwaps){
            union(dsu, swap[0], swap[1]);
        }
        HashMap<Integer,List<Integer>> map = new HashMap<>();
        for (int i=0;i<n;i++){
            int root = 
find(dsu,i);
            map.putIfAbsent(root, new ArrayList<>());
            map.get(root).add(i);
        }
        int answer = 0;
        for (List<Integer> list:map.values()){
            answer+=getDiff(source, target, list);
        }
        return answer;      
    }
    private int getDiff(int[] source, 
int[] target, List<Integer> list){
        HashMap<Integer,Integer> map = new HashMap<Integer,Integer>();
        for (int index:list){
            map.put(source[index], map.getOrDefault(source[index],0)+1);
            map.put(target[index], map.getOrDefault(target[index],0)-1);
        }
        int diff = 0;
        for (int val:map.values()){
            if (val>0){
                diff+=val;
            
}
        }
        return diff;
        
    }
    private int find(int[] dsu, int index){
        return dsu[index]==index?index:find(dsu,dsu[index]);
    
}
    private void union(int[] dsu, int index1, int index2){
        int root1 = find(dsu, index1);
        int root2 = find(dsu, index2);
        dsu[root1]=root2;
    }
}
