class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        int[] par = new int[n];
        for (int i = 0; i < n; i++) {
            par[i] = i;
        }
        for (int[] swap : allowedSwaps) {
            int a = swap[0], b = swap[1];
            
int ra = find(par, a);
            int rb = find(par, b);
            
            par[rb] = ra;
        }
        Map<Integer, Set<Integer>> map = new HashMap<>();
       
        for (int i = 0; i < n; i += 1) {
            int rt = find(par, i);
            map.putIfAbsent(rt, new HashSet<>());
            map.get(rt).add(i);
        }
        int ans = 0;
        for (int k : map.keySet()) {
            
            int d = dist(map.get(k), source, target);
            ans += d;
            //System.out.println(map.get(k).toString() + " " + d);
        }
        return ans;
    }
    private int dist(Set<Integer> set, int[] source, int[] target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : set) {
            map.put(target[i], map.getOrDefault(target[i], 0)+1);
        }
        int d = 0;
        //System.out.println(map.toString());
        for (int e : set) {
            if (map.getOrDefault(source[e], 0) > 0) {
                map.put(source[e], map.get(source[e])-1);
            } else {
                d += 1;
            }
        
}
        return d;
    }
    private int find(int[] par, int x) {
        while (x != par[x]) {
            par[x] = par[par[x]];
            x = par[x];
        }
        return x;
    }
}
