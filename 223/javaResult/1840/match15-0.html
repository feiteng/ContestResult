class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int[] ds = new int[source.length];
        
        for(int i = 0; i < ds.length; i++) {
            ds[i] = i;
        }
        
        for(int[] swap : allowedSwaps) {
            join(ds, swap[0], swap[1]);
        }
        
        Map<Integer, Map<Integer, Integer>> pos = new HashMap<>();
        
        for(int i = 0; i < ds.length; i++) {
            int iSet = getSet(ds, i);
            
            pos.computeIfAbsent(iSet, a -> new HashMap<>());
            
            Map<Integer, Integer> setPos = pos.get(iSet);
            
            setPos.put(source[i], setPos.getOrDefault(source[i], 0) + 1);
        }
        
        int out = 0;
        
        for(int i = 0; i < target.length; i++) {
            int count = pos.get(getSet(ds, i)).getOrDefault(target[i], 0);
            
            if(count > 0) {
                pos.get(getSet(ds, i)).put(target[i], count - 1);
            } else {
                out++;
            }
        }
        
        return out;
    }
    
    private int getSet(int[] ds, int n) {
        if(ds[n] != n) {
            ds[n] = getSet(ds, ds[n]);
        }
        
        return ds[n];
    }
    
    private void join(int[] ds, int x, int y) {
        int xs = getSet(ds, x);
        int ys = getSet(ds, y);
        
        ds[xs] = ys;
    }
}
