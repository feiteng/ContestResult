class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int N = source.length;
        DSU dsu = new DSU(N);
        for (int[] p: allowedSwaps){
            dsu.union(p[0], p[1]);
        }
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for (int i = 0; i < N; i++){
            int root = dsu.find(i);
            if (!map.containsKey(root)) map.put(root, new HashMap<>());
            Map<Integer, Integer> secMap = map.get(root);
            int count = 
secMap.getOrDefault(source[i], 0);
            secMap.put(source[i], count+1);
        }
        int res = 0;
        for (int i = 0; i < N; i++){
            int root = dsu.find(i);
            Map<Integer, Integer> secMap = map.get(root);
            if (secMap.containsKey(target[i])){
                if (secMap.get(target[i]) == 1) secMap.remove(target[i]);
                else secMap.put(target[i], secMap.get(target[i])-1);
            
}else {
                res++;
            }
        }
        return res;
    }
}


class DSU{
    int[] parent;
    
    DSU(int n){
        parent = new int[n];
        for (int i  = 0; i < n; i++){
            parent[i] = i;
        }
    }
    
    public int find(int x){
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    
    public void union(int x, int y){
        if (find(x) > find(y)) parent[find(x)] = parent[find(y)];
        else parent[find(y)] = parent[find(x)];
    }
}
