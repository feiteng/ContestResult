import java.util.*;

class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        int[] f = new int[n];
        for (int i = 0; i < f.length; i++) {
            f[i] = i;
        }
        for (int[] e : allowedSwaps) {
            int fa = find(e[0], f);
            int fb = find(e[1], f);
            if (fa != fb) {
                f[fa] = fb;
            }
        }
        int count = 0;
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
           int fa = find(i, f);
           map.putIfAbsent(fa, new HashSet<>());
           map.get(fa).add(i);
        }
        for (Map.Entry<Integer, Set<Integer>> e : map.entrySet()) {
            Map<Integer, Integer> ok = new HashMap<>();
            for (int index : e.getValue()) {
                ok.put(source[index], ok.getOrDefault(source[index], 0)  + 1);
            }
            for (int index : e.getValue()) {
                if (ok.containsKey(target[index])) {
                    int number = ok.get(target[index]);
                    if (number == 1) ok.remove(target[index]);
                    else ok.put(target[index], number - 1);
                } else {
                    count++;
                }
            }
        }
        return count;
    }

    private int find(int start, int[] f) {
        if (f[start] == start) {
            return start;
        }
        return f[start] = find(f[start], f);
    }
}
