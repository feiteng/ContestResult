class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int N = source.length;
        DSU dsu = new DSU(N);
        for(int[] swap : allowedSwaps) {
            dsu.union(swap[0], swap[1]);
        }
        
        Map<Integer,Map<Integer,Integer>> map = new HashMap<>(); 
        for(int i = 0; i < N; i++) {
            int p = dsu.find(i);
            if(!map.containsKey(p)) map.put(p, new HashMap<>());
            
            Map<Integer, Integer> m = map.get(p);
            m.put(source[i], m.getOrDefault(source[i], 0) + 1);
        }
        
        int dis = 0;
        for(int i = 0; i < N; i++) {
            int v = target[i];
            int p = dsu.find(i);
            Map<Integer, Integer> m = map.get(p);
            if(m.containsKey(v) && m.get(v) > 0) {
                m.put(v, m.get(v) - 1);
            } else {
                dis++;
            }
        }
        return dis;
    }
    
    class DSU {
        int[] parent;
        int[] rank;
        public DSU(int N) {
            parent = new int[N];
            rank = new int[N];
            for (int i = 0; i < N; ++i){
                parent[i] = i;
                rank[i] = 1;
            }
        }
        public int find(int x) {
            if (parent[x] != x) parent[x] = find(parent[x]);
            return parent[x];
        }
        public void union(int x, int y) {
            int px = find(x), py = find(y);
            if(px != py) {
                if(rank[px] > rank[py]) {
                    rank[px] += rank[py];
                    parent[py] = px;
                } else {
                    rank[py] += rank[px];
                    parent[px] = py;
                }
            }
        }
    }
}
