class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length ; 
        int[] uf = new int[n] ; 
        for ( int i = 0 ; i < n ; i ++ ) uf[i] = i ; 
        for ( int[] as: allowedSwaps ) {
            int pa = find(uf, as[0]), pb = find(uf, as[1]) ; 
            if ( pa != pb ) uf[pb] = pa ;  
        }
        // parent -> (value -> count)
        Map<Integer, Map<Integer, Integer>> mapS = new HashMap<>() ; 
        Map<Integer, Map<Integer, Integer>> mapT = new HashMap<>() ; 
        for ( int i = 0 ; i < n ; i ++ ) {
            int p = find(uf, i) ; 
            if ( ! mapS.containsKey(p) ) mapS.put(p, new HashMap<>()) ; 
            if ( ! mapT.containsKey(p) ) mapT.put(p, new HashMap<>()) ; 
            mapS.get(p).put(source[i], mapS.get(p).getOrDefault(source[i], 0) + 1) ; 
            mapT.get(p).put(target[i], mapT.get(p).getOrDefault(target[i], 0) + 1) ; 
        }
        // System.out.println("mapS: " + mapS) ; 
        // System.out.println("mapT: " + mapT) ;         
        int ans = 0, common = 0, total = 0; 
        for ( int key: mapS.keySet() ) {
            Map<Integer, Integer> ms = mapS.get(key) ; 
            Map<Integer, Integer> mt = mapT.get(key) ; 
            for ( int value: ms.keySet() ) {
                total += ms.get(value) ; 
                if ( mt.containsKey(value) ) {
                    common += Math.min(ms.get(value), mt.get(value)) ; 
                }
            }
        }
        return total - common ; 
    }
    
    private int find(int[] uf, int node) {
        while ( node != uf[node] ) {
            uf[node] = uf[uf[node]] ; 
            node = uf[node] ; 
        }
        return node ; 
    }
}
