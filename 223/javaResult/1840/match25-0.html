class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        int[] uf = new int[n];
        Arrays.fill(uf, -1);
        for (int[] swap : allowedSwaps) {
            int repA = find(uf, swap[0]);
            int repB = find(uf, swap[1]);
            if (repA != repB) {
                //if (uf[repA] < uf[repB]) {
                    //uf[repA] += uf[repB];
                    uf[repB] = repA;
                //} else {
                    
                    //uf[repB] += uf[repA];
                    //uf[repA] = repB;
                //}
            }
        }
        int ret = 0;
        HashMap<Integer, List<Integer>> map = new HashMap<>();
        for (int j = 0; j < n; j++) {
            int rep = find(uf, j);
            if (!map.containsKey(rep))
                map.put(rep, new ArrayList<>());
            map.get(rep).add(j);
        }
        for (Map.Entry<Integer, List<Integer>> entry : map.entrySet()) {
            int rep = entry.getKey();
            List<Integer> list = entry.getValue();
            HashMap<Integer, Integer> targetCount = new HashMap<>();
            int hamDist = list.size();
            for (int index : list) {
                targetCount.put(target[index], targetCount.getOrDefault(target[index], 0) + 1);
            }
            for (int index : list) {
                int count = targetCount.getOrDefault(source[index], 0);
                if (count > 0) {
                    hamDist--;
                    targetCount.put(source[index], count-1);
                }
            }
            ret += hamDist;
        }
        return ret;
    }
    
    public int find(int[] uf, int ind) {
        if (uf[ind] < 0) {
            return ind;
        } else {
            uf[ind] = find(uf, uf[ind]);
            return uf[ind];
        }
    }
}
