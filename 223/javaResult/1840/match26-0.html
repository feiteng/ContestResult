class Solution {
    
    int[] parents;
    
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = source.length;
        parents = new int[n];
        for (int i = 0; i < n; i++) {
            parents[i] = i;
        }
        for (int[] as : allowedSwaps) {
            union(as[0], as[1]);
        }
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.computeIfAbsent(find(i), l -> new ArrayList<>()).add(i);
        }
        int ans = 0;
        for (List<Integer> list : map.values()) {
            Map<Integer, Integer> countMap = new HashMap<>();
            for (int idx : list) {
                countMap.put(source[idx], countMap.getOrDefault(source[idx], 0)+1);
                countMap.put(target[idx], countMap.getOrDefault(target[idx], 0)-1);
            }
            int sum = 0;
            for (int value : countMap.values()) {
                sum += Math.abs(value);
            }
            ans += sum/2;
        }
        return ans;
    }
    
    private int find(int i) {
        while (parents[i] != i) {
            parents[i] = parents[parents[i]];
            i = parents[i];
        }
        return i;
    }
    
    private void union(int p, int q) {
        int pRoot = find(p), qRoot = find(q);
        if (pRoot != qRoot) {
            parents[qRoot] = pRoot;
        }
    }
}
