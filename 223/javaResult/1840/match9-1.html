class Solution {
    public int minimumHammingDistance(int[] source, int[] target, int[][] allowedSwaps) {
        int n = target.length;
        int[] map = new int[n];
        for(int i = 0; i < n; i++){
            map[i] = i;
        
}
        
        for(int[] a: allowedSwaps){
            int c = find(map, a[0]);
            int d = find(map, a[1]);
            if(c != d){
                map[c] = d;
            
}
        }
        
        HashMap<Integer, List<Integer>> map1 = new HashMap();
        HashMap<Integer, List<Integer>> map2 = new HashMap();
        
        for(int i = 0; i < n; i++){
            int id = find(map, i);
            map1.putIfAbsent(id, new LinkedList());
            map2.putIfAbsent(id, new LinkedList());
            map1.get(id).add(source[i]);
            map2.get(id).add(target[i]);
        }
        
        
int res = 0;
        
        for(int key:map1.keySet()){
            List<Integer> l1 = map1.get(key);
            List<Integer> l2 = map2.get(key);
            HashMap<Integer, Integer> count = new HashMap();
            for(int i: l1){
                count.put(i, count.getOrDefault(i, 0)+1);
            }
            for(int j: l2){
                if(count.containsKey(j)){
                    if(count.get(j) > 0){
                        count.put(j, count.get(j) - 1);
                    }else{
                        res++;
                    }
                }else{
                    res++;
                
}
            }
        }
        return res;
    }
    
    public int find(int[] map, int id){
        while(map[id] != id){
            map[id] = map[map[id]];
            id = map[id];
        }
        return id;
    }
}
