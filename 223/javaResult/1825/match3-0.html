class Solution {
    int res = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        int[] buckets = new int[k];
        Arrays.sort(jobs);
        bt(buckets,jobs,0);
        return res;
    }
    private void bt(int[] buckets, int[] jobs, int job) {
        if (job == jobs.length) {
            int temp = 0;
            for (int num : buckets) {
                temp = Math.max(temp, num);
            }
            res = Math.min(temp, res);
            return;
        }
        for (int i = 0; i < buckets.length; i++) {
            buckets[i] += jobs[job];
            bt(buckets,jobs,job+1);
            buckets[i] -= jobs[job];
            if (buckets[i] == 0) {
                break;
            }
        }
    }
}
