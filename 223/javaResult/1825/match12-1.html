class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int jobNum = jobs.length;
        int cost[] = new int[1 << jobNum];
        for (int i = 1; i < cost.length; i++) {
            cost[i] = 0;
            
for (int j = 0; j < jobs.length; j++) {
                if (((1 << j) & i) > 0)
                    cost[i] += jobs[j];
            }
        }
        int jobStatus = (1 << jobNum) - 1;
        int res[][] = new int[k + 1][jobStatus + 1];

        
for (int i = 0; i <= jobStatus; i++) {
            res[1][i] = cost[i];
        }
        for (int i = 1; i <= k - 1; i++) {
            for (int j = 0; j < jobStatus+1; j++)
                res[i + 1][j] = cost[j];
            for (int nowStatus = 0; nowStatus <= jobStatus; nowStatus++) {
                for (int add = 1; add <= jobStatus; add++) {
                    if ((nowStatus & add) == 0) {
//                        if (i+1==3 && (nowStatus | add) ==7 )
//                            System.out.println(cost[add] + " " + res[i][nowStatus] + " "+ Math.max(cost[add], res[i][nowStatus]));
                        res[i + 1][nowStatus | add] = Math.min(Math.max(cost[add], res[i][nowStatus]), res[i + 1][nowStatus | add]);
//                        if (i+1==3 && (nowStatus | add) ==7 )
//                        System.out.println(nowStatus + " " + add + " "+ res[i + 1][nowStatus | add] );


                    }
                
}
            }
        }
//        for (int i = 1; i <= k ;i++){
//            for (int i1 : res[i]) {
//                System.out.print(i1+" ");
//            }
//            System.out.println();
//        }


        return res[k][jobStatus];



    }
}
