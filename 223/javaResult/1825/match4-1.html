class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        int[][] f = new int[k + 1][1 << n];
        for(int i = 0; i <= k; ++i) {
            Arrays.fill(f[i], Integer.MAX_VALUE / 2);
        }
        f[0][0] = 0;
        int[] sum = new int[1 << n];
        for(int i = 0; i < (1 << n); ++i) {
            for(int j = 0; j < n; ++j) {
                if(((i >> j) & 1) > 0) {
                    sum[i] += jobs[j];
                
}
            }
        }
        
        for(int i = 1; i <= k; ++i) {
            for(int s = 0; s < (1 << n); ++s) {
                
for(int ns = s; ns > 0; ns = (ns - 1) & s) {
                    
                    f[i][s] = Math.min(f[i][s], Math.max(f[i - 1][s ^ ns], sum[ns]));
                }
            }
        }
        return f[k][(1 << n) - 1];
    }
}
