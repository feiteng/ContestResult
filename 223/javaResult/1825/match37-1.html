class Solution {
    int res = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        int total = 0;
        for (int i : jobs) {
            total += i;
        }
        Arrays.sort(jobs);
        int[] arr = new int[jobs.length];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = jobs[arr.length - i - 1];
        }
        jobs = arr;
        int[] sumArr = new int[k];
        helper(jobs, sumArr, 0);
        
return res;
    }
    public void helper(int[] jobs, int[] sumArr, int level) {
        if (level == jobs.length) {
            int max = 0;
            for (int i : sumArr) {
                max = Math.max(max, i);
            }
            res = Math.min(res, max);
            return;
        }        
        for (int i = 0; i < sumArr.length; i++) {
            if (sumArr[i] + jobs[level] >= res) {
                continue;
            }
            sumArr[i] = sumArr[i] + jobs[level];
            helper(jobs, sumArr, level + 1);
            sumArr[i] = sumArr[i] - jobs[level];
            while (i != sumArr.length - 1 && sumArr[i] == sumArr[i + 1]) {
                i++;
            }
        }
    }
 
}
