class Solution {
    int ans = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        ans = Integer.MAX_VALUE;
        Arrays.sort(jobs);
        int[] ttl = new int[k];
        int idx = jobs.length - 1;
        for(int i = 0; i < k; i++){
            ttl[i] = jobs[idx];
            idx--;
        }
        
        int[] newjobs = new int[jobs.length - k];
        for(int i = 0; i < newjobs.length; i++){
            newjobs[i] = jobs[idx];
            idx--;
        }
        //System.out.println(Arrays.toString(newjobs));
        recursion(newjobs, ttl, 0);
        return ans;
    }
    
    private void recursion(int[] jobs, int[] ttl, int idx){
        if(idx == jobs.length){
            int max = -1;
            for(int t : ttl){
                max = Math.max(max, t);
            }
            
            ans = Math.min(ans, max);
            return;
        }
        
        for(int i = 0; i < ttl.length; i++){
            ttl[i] += jobs[idx];
            recursion(jobs, ttl, idx+1);
            ttl[i] -= jobs[idx];
        }
    }
}
