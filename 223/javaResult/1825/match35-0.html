class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int[][] dp = new int[k][1 << jobs.length];
        int mask = (1 << jobs.length) - 1;
        int[] jobInterval = new int[1 << jobs.length];
        for (int i = 0; i < jobInterval.length; i++) {
            for (int j = 0; j < jobs.length; j++) {
                if (((1 << j) & i) != 0) {
                    jobInterval[i] += jobs[j];
                }
            }
            dp[0][i] = jobInterval[i];
        }
        //dp每一位记录。有k个工人时，搬运i 状态job时，的最大值
        for (int i = 1; i < dp.length; i++) {
            for (int j = 1; j < dp[0].length; j++) {
                dp[i][j] = jobInterval[j];
                for (int l = j; l > 0 ; l = j & (l - 1)) {
                    dp[i][j] = Math.min(dp[i][j], Math.max(jobInterval[l], dp[i - 1][l ^ j]));
                }
            }
        }

        return dp[k - 1][dp[0].length - 1];
    }
}
