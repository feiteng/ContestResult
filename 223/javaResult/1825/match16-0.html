import java.util.Arrays;

class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int l = Arrays.stream(jobs).max().getAsInt();
        int r = Arrays.stream(jobs).sum();
        Arrays.sort(jobs);
        while (l < r) {
            int m = (l + r) / 2;
            if (possible(jobs, k, m)) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return l;
    }

    private boolean possible(int[] jobs, int k, int w) {
        int[] available = new int[k];
        available[0] = jobs[jobs.length - 1];
        return bp(jobs, available, w, jobs.length - 2);
    }

    private boolean bp(int[] jobs, int[] available, int w, int index) {
        if (index == -1) {
            return true;
        }
        for (int i = 0; i < available.length; i++) {
            if (jobs[index] + available[i] <= w) {
                available[i] += jobs[index];
                if (bp(jobs, available, w, index - 1)) {
                    return true;
                }
                available[i] -= jobs[index];
            }
        }
        return false;
    }
}
