class Solution {
     public static int minimumTimeRequired(int[] jobs, int k) {
        int[][] dp = new int[(1<<jobs.length)][k+1];
        int[] sum1 = new int[(1<<jobs.length)];
         for(int i = 0;i<(1<<jobs.length);i++){
             for(int j = 0;j<jobs.length;j++){
               if (((1<<j)&i) != 0)
                    sum1[i]+=jobs[j];
            }
         }
        for(int j = 0;j<(1<<jobs.length);j++){
            Arrays.fill(dp[j], Integer.MAX_VALUE);
        }
        Arrays.fill(dp[0], 0);
        for(int t = 1;t<=k;t++){
            for(int i = 1;i<(1<<jobs.length);i++){
                for(int s = i;s>=0;s = (s-1)&i){
                    int sum = sum1[i^s];
                    if (dp[s][t-1] != Integer.MAX_VALUE)
                    	dp[i][t] = Math.min(dp[i][t], Math.max(dp[s][t-1], sum));
                    if (s == 0)
                    	break;
                }
            }
        }
        return dp[(1<<jobs.length)-1][k];
        
    }
}
