class Solution {
    private int min = Integer.MAX_VALUE;

    public int minimumTimeRequired(int[] jobs, int k) {
        int[] result = new int[k];
        pushNum(result, k, jobs, 0);
        return min;
    }

    private void pushNum(int[] result, int emptyCount, int[] jobs, int jobIndex) {
        int remain = jobs.length - jobIndex;
        if (remain < emptyCount) {
            return;
        }

        
if (jobIndex == jobs.length) {
            int max = result[0];
            for (int i = 0; i < result.length; i++) {
                max = Math.max(max, result[i]);
            }
            min = Math.min(max, min);
            return;
        }
        for (int i = 0; i < result.length; i++) {
            if (result[i] == 0) {
                emptyCount--;
            }
            result[i] += jobs[jobIndex];
            
pushNum(result, emptyCount, jobs, jobIndex + 1);
            result[i] -= jobs[jobIndex];
            if (result[i] == 0) {
                break;
            }
        }
    }
}
