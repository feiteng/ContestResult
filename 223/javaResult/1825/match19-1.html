class Solution {
    private int min;

    public int minimumTimeRequired(int[] jobs, int k) {
        min = Integer.MAX_VALUE;
        dfs(jobs, k, 0, new int[k]);
        return min;
    }

    private void dfs(int[] jobs, int k, int i, int[] assignments) {
        if (i == jobs.length) {
            min = Math.min(min, 
Arrays.stream(assignments).max().getAsInt());
        } else {
            for (int j = 0; j < k; j++) {
                assignments[j] += jobs[i];
                if (assignments[j] < min) {
                    dfs(jobs, k, i + 1, assignments);
                }
                assignments[j] -= jobs[i];
                if (assignments[j] == 0) {
                    break;
                
}
            }
        }
    }
}
