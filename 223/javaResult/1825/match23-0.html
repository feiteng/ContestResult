class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        int[] tots = new int[1<<n];
        for (int i = 0; i < (1<<n); ++i) {
            for (int j = 0; j < n; ++j) {
                if ((i&(1<<j))>0)
                    tots[i]+=jobs[j];
            }
        }
        int[][] dp = new int[k][1<<n];
        for (int i = 0; i < (1<<n); ++i)
            dp[0][i] = tots[i];
        for (int i = 1; i < k; ++i) {
            for (int j = 0; j < (1<<n); ++j) {
                dp[i][j] = 1000000000;
            }
            for (int j = 0; j < (1<<n); ++j) {
                for (int a = j+1; a < (1<<n); ++a) {
                    dp[i][j|a] = Math.min(dp[i][j|a], Math.max(dp[i-1][j], tots[a]));
                }
            }
        }
        return dp[k-1][(1<<n)-1];
    }
}
