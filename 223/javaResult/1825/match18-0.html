/**
only 12 workers, 12 jobs

**/
class Solution {
    
    private int minTotal = Integer.MAX_VALUE;
    
    public int minimumTimeRequired(int[] jobs, int k) {
        Arrays.sort(jobs);
        int[] workers = new int[k];
        int n = jobs.length;
        for (int i = 0; i < k; i++) {
            workers[i] += jobs[n-1-i];
        }        
        assignWork(jobs, workers, 0);
        return minTotal;
    }
    
    private void assignWork(int[] jobs, int[] workers, int cur) {
        if (cur == jobs.length - workers.length) {
            minTotal = Math.min(minTotal, checkMax(workers));     
            return;
        }
        
        for (int i = 0; i < workers.length; i++) {
            workers[i] += jobs[cur];
            assignWork(jobs, workers, cur + 1);
            workers[i] -= jobs[cur];
        }        
    }
    
    private int checkMax(int[] workers) {
        int res = workers[0];        
        for (int i = 1; i < workers.length; i++) {
            res = Math.max(res, workers[i]);
        }
        return res;
    }
}
