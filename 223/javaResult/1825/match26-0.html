class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        int[][] dp = new int[n][1 << n];
        for (int i = 0; i < (1 << n); i++) {
            int sum = 0;
            for (int j = 0; j < n; j++) {
                if ((i >> j & 1) != 0)
                    sum += jobs[j];
            }
            dp[0][i] = sum;
        }
        for (int i = 1; i < k; i++)
            Arrays.fill(dp[i], 1000000000);
        for (int i = 1; i < k; i++) {
            for (int j = 0; j < (1 << n); j++) {
                for (int h = j; h != 0; h = (h - 1) & j) {
                    if ((j & h) != h) continue;
                    dp[i][j] = Math.min(dp[i][j], Math.max(dp[i - 1][h], dp[0][j] - dp[0][h]));
                }
            }
        }
        return dp[k - 1][(1 << n) - 1];
    }
}
