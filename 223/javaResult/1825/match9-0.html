class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int[] timeCost = new int[1 << jobs.length];
        for (int i = 0; i < timeCost.length; i++) {
            for (int j = 0; j < jobs.length; j++) {
                if ((i & (1 << j)) != 0) {
                    timeCost[i] += jobs[j];
                }
            }
        }

        int[] dp = new int[1 << jobs.length];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = 0;
        for (int i = 0; i < k; i++) {
//            int[] dpNew = new int[1 << jobs.length];
//            Arrays.fill(dpNew, Integer.MAX_VALUE);
            for (int j = dp.length - 1; j >= 0; j--) {
                for (int j2 = 0; j2 < j; j2++) {
                    if ((~j & j2) != 0) {
                        continue;
                    }
                    dp[j] = Math.min(dp[j], Math.max(dp[j2], timeCost[j & ~j2]));
                }
            }
            
//            dp = dpNew;
        }
        
        return dp[dp.length - 1];
    }
}
