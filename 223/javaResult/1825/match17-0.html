class Solution {
    private int min_max = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        Arrays.sort(jobs);
        int n = jobs.length;
        int[] workers = new int[k];
        for (int i = 0; i < k; i++) {
            workers[i] += jobs[n - 1 - i];
        }
        getMinMax(jobs, workers, n - 1 - k);
        return min_max;
    }
    
    private void getMinMax(int[] jobs, int[] workers, int index) {
        int cur_max = getMax(workers);
        if (cur_max > min_max) {
            return ;
        }
        if (index < 0) {
            min_max = Math.min(min_max, cur_max);
            return ;
        }
        for (int i = 0; i < workers.length; i++) {
            workers[i] += jobs[index];
            getMinMax(jobs, workers, index - 1);
            workers[i] -= jobs[index];
        }
        return ;
    }
    
    private int getMax(int[] workers) {
        int max = Integer.MIN_VALUE;
        for (int w : workers) {
            max = Math.max(w, max);
        }
        
        return max;
    }
}
