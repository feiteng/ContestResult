class Solution {
    int min;
    public int minimumTimeRequired(int[] jobs, int k) {
        min = Integer.MAX_VALUE;
        int[] record = new int[k];
        Arrays.sort(jobs);
        int len = jobs.length;
        for(int i = 0, j = len - 1; i < j; i++, j--){
            int temp = jobs[i];
            jobs[i] = jobs[j];
            jobs[j] = temp;
        }
        
for(int i = 0; i < k; i++){
            record[i] = jobs[i];
        }
        dfs(record, jobs, k, k);
        return min;
    }
    
    public void dfs(int[] record, int[] jobs, int pos, int k){
        if(pos == jobs.length){
            int max = 0;
            for(int num : record){
                max = Math.max(max, num);
            }
            min = Math.min(max, min);
            return;
        }
        for(int i = 0; i < k; i++){
            if(record[i] + jobs[pos] >= min){
                continue;
            }
            record[i] += jobs[pos];
            dfs(record, jobs, pos + 1, k);
            record[i] -= jobs[pos];
        }
    }
}
