class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int l = jobs.length;
        
        
int dp[][] = new int[k+2][1<<l];
        
        int sum[] = new int[1<<l];
        
        for(int st = 0 ;st<1<<l;++st){
            for(int j=0;j<l;++j){
                if((st&(1<<j))!=0){
                   sum[st] += jobs[j]; 
                
}
            }
        }
        
        for(int d[]:dp){
            Arrays.fill(d,1000000000);
        }
        dp[0][0] = 0;
        
        
for(int i=1;i<=k;++i){
            
            for(int st = 0 ;st<1<<l;++st){
                
                for(int f =st;f>0; f = (f-1)&st){
                    
                    dp[i][st] = Math.min( dp[i][st] ,  Math.max(dp[i-1][st-f], sum[f]));
                    
                
}
                
                
                
            }
            
        }
        
        return dp[k][(1<<l)-1];
        
        
    }
}
