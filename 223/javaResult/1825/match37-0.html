class Solution {
    int min = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        helper(jobs, 0, new int[k]);
        return min;
    }
    
    public void helper(int[] jobs, int index, int[] work) {
        if(index == jobs.length) {
            int max = work[0];
            for(int i = 1; i < work.length; i++) {
                max = Math.max(max, work[i]);
            }
            min = Math.min(min, max);
            return;
        }
        for(int i = 0; i < work.length; i++) {
            if(i > 0 && work[i] == work[i-1]) {
                continue;
            }
            work[i] += jobs[index];
            helper(jobs, index + 1, work);
            work[i] -= jobs[index];
        }
    }
}
