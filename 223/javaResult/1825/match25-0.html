class Solution {
    public int minimumTimeRequired(int[] js, int k) {
        int n = js.length;
        int[] time = new int[1 << n];
        for (int i = 0; i < (1 << n); i++) {
            time[i] = 0;
            for (int j = 0; j < n; j++) {
                if ((i & (1 << j)) > 0) {
                    time[i] += js[j];
                }
            }
        }
        
        int[][] dp = new int[1 << n][k + 1];
        for (int i = 0; i < (1 << n); i++) {
            Arrays.fill(dp[i], Integer.MAX_VALUE);
        }
        dp[0][0] = 0;
        
        for (int i = 1; i <= k; i++) {
            for (int j = 1; j < (1 << n); j++) {
                int l = j;
                while (l > 0) {
                    if (dp[j ^ l][i - 1] != Integer.MAX_VALUE) {
                        dp[j][i] = Math.min(dp[j][i], Math.max(dp[j ^ l][i - 1], time[l]));
                    }
                    l = j & (l - 1);
                }
            }
        }
        
        return dp[(1 << n) - 1][k];
    }
}
