class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        //Solution1 Binary Search
        Arrays.sort(jobs);
        int min = jobs[0], sum = 0;
        for(int j:jobs){
            sum += j;
        }
        if(k == 1){return sum;}
        int low = min, high = sum;
        while(low < high){
            int mid = (high-low)/2+low;
            if(valid(jobs,k,mid)){
                high = mid;
            }else{
                low = mid+1;
            }
        }
        return low;
    }
    
    public boolean valid(int[] jobs,int k,int max){
        int[] times = new int[k];
        return dfs(jobs,jobs.length-1,times,max);
    }
    
    public boolean dfs(int[] jobs,int index,int[] times,int max){
        if(index == -1){return true;}
        for(int i = 0;i < times.length;i++){
            if(jobs[index]+times[i] > max){continue;}
            times[i] += jobs[index];
            if(dfs(jobs,index-1,times,max)){return true;}
            times[i] -= jobs[index];
        }
        return false;
    }
}
