class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        int[] cost = new int[1 << n];
        
        for (int i = 1; i < (1 << n); ++i) {
            for (int j = 0; j < n; ++j) {
                if ((i & (1 << j)) != 0) {
                    cost[i] += jobs[j];
                }
            
}
        }
        
        int[][] dp = new int[1 << n][k];
        for (int i = 0; i < (1<<n); ++i) {
            dp[i][0] = cost[i];
        }
        
        
for (int j = 1; j < k; ++j) {
            for (int i = 0; i < (1<<n); ++i) {
                for (int l = 0; l < (1<<n); ++l) {
                    if ((i & l) == 0) {
                        if (dp[i | l][j] == 0) 
                            dp[i | l][j] = Math.max(dp[i][j-1], cost[l]);
                        else
                            dp[i | l][j] = Math.min(dp[i | l][j], Math.max(dp[i][j-1], cost[l]));
                    }
                
}
            }
        }
        
        return dp[(1 << n) - 1][k-1];
    }
}
