class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int max = 0, sum = 0;
        for (int j : jobs) {
            max = Math.max(max, j);
            sum += j;
        }
        int left = max, right = sum;
        while (left < right) {
            int mid = (left + right) / 2;
            if (helper(jobs, new int[k], mid)) 
{
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
    private boolean helper(int[] jobs, int[] p, int maxLoad) {
        
return dfs(jobs, 0, p, maxLoad);
    }
    private boolean dfs(int[] jobs, int idx, int[] p, int maxLoad) {
        if (idx == jobs.length) {
            return true;
        }
        // Arrays.sort(p);
        for (int i = 0;i < p.length;i++) {
            if (p[i] + jobs[idx] > maxLoad) continue;
            if (i > 0 && p[i] == p[i - 1]) continue
;
            p[i] += jobs[idx];
            if (dfs(jobs, idx + 1, p, maxLoad)) {
                return true;
            }
            p[i] -= jobs[idx];
        }
        return false;
    }
}
