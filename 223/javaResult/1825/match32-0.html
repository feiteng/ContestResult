class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        int[][] dp = new int[k][1<<n];
        for(int i=0;i<(1<<n);i++) {
            for(int j=0;j<n;j++) {
                if(((1<<j)&i)>0) dp[0][i] += jobs[j];
            }
        }
        // System.out.println(Arrays.toString(dp[0]));
        for(int i=1;i<k;i++) {
            for(int j=1;j<(1<<n);j++) {
                dp[i][j] = dp[0][j];
                for(int k1=0;k1<j;k1++) {
                    if((j&k1) ==k1) {
                        dp[i][j] = Math.min(dp[i][j],Math.max(dp[i-1][k1],dp[0][j-k1]));
                    }
                }
            }
            // System.out.println(Arrays.toString(dp[i]));
        }
        // int min =Integer.MAX_VALUE;
        // for(int i=1;i<(1<<n);i++) min = Math.min(min,dp[k-1][i]);
        return dp[k-1][(1<<n)-1];
    }
}
