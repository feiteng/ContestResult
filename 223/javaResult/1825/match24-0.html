class Solution {
    public int minimumTimeRequired(int[] a, int k) {
        int n = a.length;
        int[][] dp = new int[k + 1][1 << n];
        for (int i = 0; i <= k; i++)
        {
            for (int j = 0; j < 1 << n; j++)
            {
                dp[i][j] = 1000000000;
            }
        }
        int[] cost = new int[1 << n];
        for (int mask = 0; mask < 1 << n; mask++)
        {
            int v = 0;
            for (int i = 0; i < n; i++)
            {
                if ((mask & (1 << i)) != 0) v += a[i];
            }
            cost[mask] = v;
        }
        dp[k][(1 << n) - 1] = 0;
        for (int i = k - 1; i >= 0; i--)
        {
            for (int mask = (1 << n) - 2; mask >= 0; mask--)
            {
                for (int cur = 1; cur < 1 << n; cur++)
                {
                    if ((cur & mask) != 0) continue;
                    int v = Math.max(cost[cur], dp[i + 1][cur | mask]);
                    dp[i][mask] = Math.min(dp[i][mask], v);
                }
            }
        }
        return dp[0][0];
    }
}
