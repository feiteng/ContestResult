class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        // System.out.println("split");
        int max = 0;
        int l = jobs.length;
        Arrays.sort(jobs);
        int min = jobs[l-1];
        for(int i = 0; i < l; i++)
            max += jobs[i];
        while(min < max) {
            int mid = min + (max - min) / 2;
            // System.out.println(" mid " + mid);
            if(possible(jobs, k, mid)) {
                // System.out.println(" max " + max);
                max = mid;
            } else {
                min = mid + 1;
            }
        }
        // System.out.println(min + " " + max);
        return min;
    }
    
    private boolean possible(int[] jobs, int k, int top) {
        int[] sum = new int[k];
        return helper(sum, 0, jobs, 0, k, top);
    }
    
    private boolean helper(int[] sum, int ind, int[] jobs, int ii, int k, int top) {
        if(ii == jobs.length) {
            return true;
        }
        for(int i = 0; i < ind+1 && i < sum.length; i++) {
            if(sum[i] + jobs[ii] <= top) {
                sum[i] += jobs[ii];
                if(helper(sum, ind+1, jobs, ii+1, k, top))
                    return true;
                sum[i] -= jobs[ii];
            }
        }
        return false;
    }
}
