class Solution {
    int min = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        int [] dp  = new int[k];
        Arrays.sort(jobs);
        int n = jobs.length;
        
for(int i = 1; i <= k; i++ ){
            dp[i-1] = jobs[n-i];
        }
        dfs(dp, jobs, 0, k);
        return min;
    }
    void dfs(int [] dp, int [] jobs, int i, int k){
        if(i == jobs.length - k){
            int m = 0;
            for(int j = 0; j < k; j++){
                if(m < dp[j]){
                    m = dp[j];
                }
            }
            if(m < min){
                min = m;
            }
            
return ;
        }
        for(int j = 0; j < k; j++){
            dp[j] += jobs[i];
            dfs(dp, jobs, i + 1, k);
            dp[j] -= jobs[i];
        }
    }
}
