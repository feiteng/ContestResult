class Solution {
    int res = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        int[] times = new int[k];
        Arrays.sort(jobs);
        for (int i = 0; i < jobs.length / 2; i++) {
            int v = jobs[i];
            jobs[i] = jobs[jobs.length - 1 - i];
            jobs[jobs.length - 1 - i] = v;
        }
        
backtrack(0, jobs, times);
        return res;
    }
    
    private void backtrack(int i, int[] jobs, int[] times) {
        if (i < times.length) {
            times[i] = jobs[i];
            backtrack(i + 1, jobs, times);
            return;
        }
        
if (i == jobs.length) {
            int value = 0;
            for (int t : times) {
                value = Math.max(value, t);
            }
            res = Math.min(res, value);
            return;
        }
        for (int pos = 0; pos < times.length; pos++) {
            times[pos] += jobs[i];
            backtrack(i + 1, jobs, times);
            times[pos] -= jobs[i];
        }
    }
}
