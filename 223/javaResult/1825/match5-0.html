class Solution {
    public int minimumTimeRequired(int[] jobs, int k) {
        int full = (1 << jobs.length) - 1;
        int[][] dp = new int[k][full + 1];
        for (int i = 0; i <= full; i++) {
            for (int j = 0; j < jobs.length; j++) {
                if (((i >> j) & 1) == 1) dp[0][i] += jobs[j];
            }
        }
        for (int i = 1; i < dp.length; i++) {
            for (int mask = 0; mask <= full; mask++) {
                if (mask != 0) dp[i][mask] = Integer.MAX_VALUE;
                for (int state = mask; state > 0; state = (state - 1) & mask) {
                    dp[i][mask] = Math.min(dp[i][mask], Math.max(dp[0][state], dp[i - 1][mask ^ state]));
                }
            }
        }
        return dp[k - 1][full];
    }
}
