class Solution {
	private int min;

	public int minimumTimeRequired(int[] jobs, int k) {
		min = Integer.MAX_VALUE;
		minimumTimeRequired(0, jobs, new int[k]);
		return min;
	}

	private void minimumTimeRequired(int index, int[] jobs, int[] sum) {
		if (index == jobs.length) {
			int max = 0;
			for (int i : sum) {
				max = Math.max(max, i);
			}
			min = Math.min(min, max);
		} else {
			for (int i = 0; i < sum.length; i++) {
				if (sum[i] == 0) {
					sum[i] += jobs[index];
					minimumTimeRequired(index + 1, jobs, sum);
					sum[i] -= jobs[index];
					return;
				} else {
					sum[i] += jobs[index];
					minimumTimeRequired(index + 1, jobs, sum);
					sum[i] -= jobs[index];
				}
			}
		}
	}
}
