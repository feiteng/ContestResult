class Solution {
    int min = Integer.MAX_VALUE;
    public int minimumTimeRequired(int[] jobs, int k) {
        int[] f = new int[k];
        Arrays.sort(jobs);
        for(int i=jobs.length-1;i>=jobs.length-k;i--){
            f[jobs.length-i-1] = jobs[i];
        }
        dfs(jobs,f,0,jobs.length-k);
        return min;
    }

    private void dfs(int[] jobs, int[] f, int k, int n) {
        if(k == n){
            int max = 0;
            for(int i=0;i<f.length;i++){
                max = Math.max(max,f[i]);
            }
            min = Math.min(min,max);
            return;
        }
        for(int i=0;i<f.length;i++){
            f[i] += jobs[k];
            dfs(jobs,f,k+1,n);
            f[i] -= jobs[k];
        }
    }
}
