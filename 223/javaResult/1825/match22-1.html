class Solution {
public int minimumTimeRequired(int[] jobs, int k) {
        int n = jobs.length;
        
        
        int[] values = new int[1<<n];
        for(int i = 0; i < values.length; ++i){
            
            for(int j = 0; j <n; ++j){
                if((i&(1<<j)) > 0){
                    values[i] += jobs[j];
                }
            }
            
            
        }
        
        int[][] dp = new int[1<<n][k+1];
        for(int i = 0;i  < dp.length; ++i){
            Arrays.fill(dp[i], -1);
        }
        dp[0][0] = 0;
        
        
        
        
        
        for(int i = 1; i < dp.length; ++i){
            for(int j = 1; j <= k; ++j){
                int use = i;
                dp[i][j] = values[i];
                for(;use > 0; use = (use-1)&i){
                    if(dp[i^use][j-1] != -1){
                         dp[i][j] = Math.min(dp[i][j], Math.max(values[use], dp[i^use][j-1]));
                    
}
                    
                }
                
                
                
            }
        }
        return dp[(1<<n) -1][k];
    }
}
