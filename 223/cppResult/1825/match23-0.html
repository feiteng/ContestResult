class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        int m = 1 << n;
        vector<int> t(m);
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (i >> j & 1) {
                    t[i] += jobs[j];
                }
            }
        }
        vector<vector<int>> dp(k, vector<int>(m, INT_MAX));
        dp[0] = t;
        for (int i = 1; i < k; i++) {
            for (int j = 1; j < m; j++) {
                int z = (j - 1) & j;
                for (int x = z; x > 0; x = (x - 1) & j) {
                    dp[i][j] = min(dp[i][j], max(dp[i - 1][x], t[x ^ j]));
                }
            }
        }
        return dp[k - 1][m - 1];
    }
};
