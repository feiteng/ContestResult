class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        vector<int> values(1<<12,0);
        vector<vector<int>> dp(k,(vector<int>(1<<12,2147483647)));

        int n = jobs.size();
        for(int i=0;i<1<<n;i++){
            for(int j=0;j<n;j++){
                if(1<<j & i){
                    values[i]+=jobs[j];
                }
            }
            dp[0][i] = values[i];
        }

        for(int i=1;i<k;i++){
            for(int status=1;status<1<<n;status++){
                for (int sub = status; sub; sub = (sub - 1) & status) {
                    dp[i][status] = min(max(values[sub],dp[i-1][status-sub]),dp[i][status]);
                }
            }   
        }
        return dp[k-1][(1<<n)-1];
    }
};
