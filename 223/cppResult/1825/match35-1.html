class Solution {
public:
    const int BIG_INT = 1e9 + 7;

    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<int> sum(1 << n, 0);
        for (int i = 0; i < (1 << n); ++i)
        {
            int x = i, j = 0;
            while (x > 0)
            {
                if (x % 2) sum[i] += jobs[j];
                x = x >> 1;
                ++j;
            }
        }

        vector<vector<int>> dp(k + 1, vector<int>(1 << n, BIG_INT));
        dp[0][0] = 0;
        for (int i = 1; i <= k; ++i)
        {
            for (int j = 0; j < (1 << n); ++j)
            {
                for (int z = 0; z <= j; ++z)
                {
                    if ((j | z) == j)
                    {
                        dp[i][j] = min(dp[i][j], max(dp[i - 1][j - z], sum[z]));
                    }
                }
            }
        }
        return dp[k][(1 << n) - 1];
    }
};
