class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        rec.resize(k, 0);
        sort(jobs.begin(), jobs.end());
        int n = jobs.size(), p = n - 1;
        for(int i = 0; i < k; ++i){
            rec[i] = jobs[p--];
        }
        res = INT_MAX;
        dfs(jobs, p, k);
        return res;
    }
    
private:
    vector<int> rec;
    int res;
    void dfs(vector<int>& jobs, int idx, int k){
        if(idx < 0){
            int max_val = 0;
            for(int x : rec) max_val = max(max_val, x);
            res = min(res, max_val);
            return;
        }
        for(int i = 0; i < k; ++i){
            rec[i] += jobs[idx];
            dfs(jobs, idx - 1, k);
            rec[i] -= jobs[idx];
        }
    }
};
