class Solution {
public:
    int dp[1<<12][15];
    int val[1<<12];
    int DP(int x,int k){
        if(dp[x][k]!=-1)return dp[x][k];
        if(x==0)return 0;

        if(k==0)return 1e9;
        dp[x][k]=1e9;
        for(int i = x;i;i=(i-1)&x){
            dp[x][k]=min(dp[x][k],max(DP(x-i,k-1),val[i]));
        }
        return dp[x][k];
    }
    int minimumTimeRequired(vector<int>& jobs, int k) {
        memset(dp,-1,sizeof(dp));
        int n=jobs.size();

        for(int i = 0;i<(1<<n);i++){
            int sum=0;
            for(int j = 0;j<n;j++){
                if(i&(1<<j))sum+=jobs[j];
            }
            val[i]=sum;
        }
        return DP((1<<n)-1,k);
    }
};
