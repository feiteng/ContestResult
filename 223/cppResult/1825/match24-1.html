class Solution {
public:
    int lowbit(int x)
    {
        return x&(-x);
    }
    int INF=0x3f3f3f3f;

    int minimumTimeRequired(vector<int>& jobs, int k) {
        int m=jobs.size();
        int mask=1<<m;
        vector<int> sum(mask,0);
        for(int i=0;i<m;i++)
            sum[1<<i]=jobs[i];
        for(int i=1;i<mask;i++)
            sum[i]=sum[i^lowbit(i)]+sum[lowbit(i)];

        vector<int> dp(mask,INF);
        dp[0]=0;
        for(int i=0;i<k;i++)
        {
            for(int j=mask-1;j>=1;j--)
            {

                for(int k=j;k;k=(k-1)&j)
                {
                    dp[j]=min(dp[j],max(dp[j^k],sum[k]));
                }
            }
        }
        return dp[mask-1];
    }
};
