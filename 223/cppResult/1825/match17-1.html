class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n=jobs.size();
        int T=1<<n;
        vector<vector<int>> dp(T,vector<int>(k+1,INT_MAX));

        vector<int> tt(T,0);
        for(int i=0;i<T;i++){
            int t=0;
            for(int j=0;j<n;j++){
                if(i&(1<<j)){
                    t+=jobs[j];
                }
            }
            tt[i]=t;
        }
        for(int i=0;i<=k;i++) dp[0][i]=0;

        for(int i=1;i<=k;i++){
            for(int j=1;j<T;j++){
                for(int a=j;a>0;a=(a-1)&j){
                    int r=j-a;
                    if(dp[r][i-1]!=INT_MAX){
                        dp[j][i]=min(dp[j][i],max(dp[r][i-1],tt[a]));
                    }
                }
            }
        }
        return dp[T-1][k];
    }
};
