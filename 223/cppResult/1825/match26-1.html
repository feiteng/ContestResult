class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n=jobs.size();

        vector<int>w(1<<n);
        w[0]=0x3f3f3f3f;
        for(int st=1;st<(1<<n);++st){
            for(int i=0;i<n;++i){
                if(st&(1<<i))w[st]+=jobs[i];
            }
        }
        // for(int x:w)cout<<x<<" ";
        // cout<<endl;
        vector<vector<int>>dp(k,vector<int>(1<<n,0x3f3f3f3f));
        for(int st=0;st<(1<<n);++st)dp[0][st]=w[st];

        for(int i=1;i<k;++i){
            for(int j=1;j<(1<<n);++j){
                for(int sub=j;sub>0;sub=j&(sub-1)){
                    dp[i][j]=min(dp[i][j],max(dp[i-1][j-sub],w[sub]));
                }
            }
        }
        return dp.back().back();
    }
};
