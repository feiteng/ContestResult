class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        int sum[1<<12] = {0};

        for (int state = 0; state < 1 << n; state++) {
            for (int i = 0; i < n; i++) {
                if ((1 << i) & state) sum[state] += jobs[i];
            }
        }
        
        int dp[1<<12] = {0};
        for (int state = 0; state < 1 << n; state++) {

            dp[state] = sum[state];
        }
        
        for (int i = 1; i < k; i++) {
            for (int state = (1 << n) - 1; state; state--) {

                for (int tmp = state; tmp; tmp = state & (tmp-1)) {
                    dp[state] = min(dp[state], max(sum[tmp], dp[tmp^state]));
                }
            }
        }
        return dp[(1<<n)-1];
    }
};
