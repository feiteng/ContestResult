class Solution {
public:
  int minimumTimeRequired(vector<int>& jobs, int k) {
    vector<vector<int>> dp(k, vector<int>(1 << jobs.size(), INT_MAX));
    
    dp[0][0] = 0;

    vector<int> work(1 << jobs.size(), 0);
    
    for (int i = 1; i < work.size(); i++) {
      for (int j = 0; j < jobs.size(); j++) {
        if (i & (1 << j)) {
          work[i] = work[i - (1 << j)] + jobs[j];
          break;
        }
      }
    }
    
    // vector<int> index(1 << jobs.size());
    // iota(index.begin(), index.end(), 0);
    // sort(index.begin(), index.end(), [&](int a, int b) {
    //   return work[a] < work[b]
    // });
    
    // for (int i = 0; i < index.size(); i++) {
    //   int rv = work[index[i]];
    //   int mask = (1 << jobs.size()) - 1;
    //   mask ^= i;
    //   for (int j = 1; j < k; j++) {
    //     for (int h = 0; h < i; h++) {
    //       if (h ^ mask != h) {
    //         continue;
    //       }
    //     }
    //   }
    // }
    for (int j = 0; j < work.size(); j++) {

      dp[0][j] = work[j];
    }
    
    
    for (int i = 1; i < k; i++) {
      for (int j = 0; j < work.size(); j++) {
        // int mask = ((1 << jobs.size()) - 1) ^ j;

        for (int h = j; h > 0; h = (h - 1) & j) {
          dp[i][j] = min(dp[i][j], max(dp[i - 1][j ^ h], work[h]));
        }
      }    
      // for (int j = 0; j < work.size(); j++)  cout << dp[i + 1][j]; cout << endl;
    }
    return dp.back().back();
  }
  
  // void run(int worker, int mask, vector<vector<int>>& dp, vector<int>& work)
};
