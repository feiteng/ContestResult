class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size(), m = 1<<n;
        vector<vector<int>> dp(1<<n, vector<int> (k+1, 1e9));
        vector<int> cost(m, 0);
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i & (1<<j)) {
                    cost[i] += jobs[j];
                }
            }
        }
        
        for (int i = 0; i < m; ++i) dp[i][1] = cost[i];
        //dp[0][0] = 0;
        
        for (int i = 0; i < m; ++i) {
            for (int j = 1; j < k; ++j) {
                //cout << i << " " << j << " " << dp[i][j] << "\n";
                
                if (dp[i][j] >= 1e9) continue;
                int t = m-1-i;
                for (int a = t; a != 0; a = (a-1)&t) {
                   // if (i == 4 && j == 1) 
                   //     cout << i << " " << a << " " << cost[t] << "\n";
                    
                    dp[a|i][j+1] = min(dp[a|i][j+1], max(dp[i][j], cost[a]));
                }
            }            
        }
        return dp[m-1][k];
    }
};
