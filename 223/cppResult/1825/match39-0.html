class Solution {
public:
    typedef long long llt;
    const int INF = 1e9+7;
    const int mod = 1e9+7;
    static const int N = 100010;

    int dp[15][10010];

    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<int> sum(1<<n);
        for(int i = 0; i < (1<<n); ++i){
            for(int j = 0; j < n; ++j){
                if(i&(1<<j)) sum[i] += jobs[j];
            }
        }

        for(int i = 0; i < k; ++i){
            for(int j = 0; j < (1<<n); ++j){
                dp[i][j] = INF;
                if(i == 0){
                    dp[i][j] = sum[j]; continue;
                }
                for(int state = j; state; state = j&(state-1)){
                    dp[i][j] = min(dp[i][j], max(dp[i-1][state], sum[j^state]));
                }
            }
        }

        return dp[k-1][(1<<n)-1];
    }
};
