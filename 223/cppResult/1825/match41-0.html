class Solution {
public:
    
    int total_time[1<<12];
    int dp[1<<12][13];
    int Solve(int n,int job_mask, int k) {
        
        if (job_mask == 0)
            return 0;
        if (k == 0)
            return (1<<30);
        if (dp[job_mask][k]!=-1)
            return dp[job_mask][k];
        int ans = (1<<30);
        for(int curr_assign=job_mask;curr_assign!=0;curr_assign=((curr_assign-1)&job_mask)) {
            ans=min(ans, max(total_time[curr_assign],Solve(n, job_mask^curr_assign,k-1)));
        }
        return dp[job_mask][k]=ans;
        
    }
    
    int minimumTimeRequired(vector<int>& jobs, int k) {
        
        int n=jobs.size();
        for(int mask=0;mask<(1<<n);mask++) {
            total_time[mask]=0;
            for(int i=0;i<n;i++) {
                if ((mask>>i)&1)
                    total_time[mask]+=jobs[i];
            }
            for(int i=0;i<=k;i++)
              dp[mask][i]=-1;
        }
        return Solve(n, (1<<n)-1,k);
        
        
    }
};
