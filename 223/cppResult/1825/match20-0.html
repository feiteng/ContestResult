class Solution {
public:
    vector<int>cost;
    vector<vector<int>>dp;
    
    int func(int mask,int k){
        // cout<<cost[mask]<<" "<<mask<<" "<<k<<"\n";
        
        if(mask==0) return 0;
        if(k==0)     return INT_MAX;
        
        int &ans=dp[mask][k];
        if(ans!=-1) return ans;
        
        
        ans=cost[mask];
        for(int submask=mask;submask;submask=(mask&(submask-1))){
            ans=min(ans, max(cost[(mask^submask)], func(submask, k-1)));    
        }
        return ans;
    }
    int minimumTimeRequired(vector<int>& a, int k) {
        int n=a.size();
        int MASK=(1ll<<n);
        cost.resize(MASK);
        
        dp=vector<vector<int>>(MASK, vector<int>(k+1, -1));
        
        for(int mask=0;mask<MASK;mask++){
            int ma=0;
            for(int i=0;i<n;i++){
                if((mask&(1ll<<i))){
                    ma+=a[i];
                }
            }
            cost[mask]=ma;
        }
        
        return func(MASK-1, k);   
    }
};
