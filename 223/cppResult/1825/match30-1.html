class Solution {
public:
    bool check(vector<int>&jobs,int k,int mid){

        vector<int>vis((int)jobs.size(),0);
        for(int i=1;i<=k;i++){
            int sum=mid;
            for(int i=0;i<jobs.size();i++){
                if(vis[i]==0 && jobs[i]<=sum){
                    vis[i]=1;
                    sum-=jobs[i];
                }
            }
        }
        bool flag=true;
        for(int i=0;i<vis.size();i++) if(vis[i]==0) flag=false;
        return flag;
    }
    int f[(1<<12)+5][15];
    int sum[(1<<12)+5];
    int minimumTimeRequired(vector<int>& jobs, int k) {
        /*sort(jobs.begin(),jobs.end());
        reverse(jobs.begin(),jobs.end());
        int sum=0;
        for(int i=0;i<jobs.size();i++) sum+=jobs[i];
        int l=1,r=sum,mid,ans;
        while(l<=r){
            mid=(l+r)>>1;
            if(check(jobs,k,mid)){
                ans=mid;
                r=mid-1;
            }else l=mid+1;
        }
        return ans;*/
        int n=jobs.size();
        memset(sum,0,sizeof(sum));
        memset(f,0x3f,sizeof(f));
        for(int i=0;i<(1<<n);i++){
            for(int j=0;j<jobs.size();j++){
                if((i>>j)&1){
                    sum[i]+=jobs[j];
                }
            }
        }

        f[0][0]=0;
        for(int i=0;i<(1<<n);i++){
            for(int j=i;j!=0;j=(j-1)&i){
                for(int l=1;l<=k;l++){
                    f[i][l]=min(f[i][l],max(sum[j],f[i-j][l-1]));
                }
            }
        }
        return f[(1<<n)-1][k];
    }
};
