class Solution {
public:

    int result;
    void dfs(vector<int>& jobs, int j, vector<int>& workers) {
        if (j == jobs.size()) {
            int maxi = 0;
            for (int& w : workers)
                maxi = max(maxi, w);
            result = min(result, maxi);
            return;
        }
        for (int i = 0; i < workers.size(); ++i) {
            if (workers[i] + jobs[j] >= result) continue;
            workers[i] += jobs[j];
            dfs(jobs, j + 1, workers);
            workers[i] -= jobs[j];
        }
    }


    int minimumTimeRequired(vector<int>& jobs, int k) {
        int sum = 0, maxi = 0;
        for (int i = 0; i < jobs.size(); ++i) {
            sum += jobs[i];
            maxi = max(maxi, jobs[i]);
        }
        if (k >= jobs.size()) return maxi;
        sort(jobs.begin(), jobs.end(), greater<int>());
        result = INT_MAX;
        vector<int> worker(k);
        for (int i = 0; i < k; ++i)
            worker[i] = jobs[i];
        dfs(jobs, k, worker);
        return result;
    }
};
