// dp[s][i] 表示i + 1个人完成s工作需要的最大工作时间是多少
class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        int lim = (1 << n);
        vector<vector<int>> dp(lim, vector<int>(k));
        
        vector<int> f(lim);
        for (int s = 0; s < lim; ++s) {
            for (int i = 0; i < n; ++i) {
                if (s >> i & 1) f[s] += jobs[i];
            }
            dp[s][0] = f[s];
        }
        
        for (int s = 1; s < lim; ++s) {
            for (int i = 1; i < k; ++i) {
                dp[s][i] = f[s];
                for (int sub = s; sub; sub = (sub - 1) & s) {
                    dp[s][i] = min(dp[s][i], max(dp[sub][i - 1], f[s - sub]));
                }
            }
        }
        
        return dp[lim - 1][k - 1];
    }
};
