class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        int t = (1 << n);
        int dp[t][k + 1];
        memset(dp, 0x3f, sizeof dp);
        dp[0][0] = 0;
        int psum[t];
        
        for (int i = 0; i < t; ++i) {
            int sum = 0;
            for (int j = 0; j < n; ++j) {
                if (i & (1 << j))
                    sum += jobs[j];
            }
            psum[i] = sum;
        }
        
        for (int num = 0; num < k; ++num) {
            for (int i = 0; i < t; ++i) {
                for (int j = i; j >= 0; j = ((j - 1) & i)) {
                    int mask = i ^ j;
                    int sum = psum[mask];
                    dp[i][num + 1] = min(dp[i][num + 1], max(dp[j][num], sum));
                    if (!j)
                        break;
                }
            }
        }
        return dp[t - 1][k];
    }
};
