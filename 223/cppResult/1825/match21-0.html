class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n=jobs.size();
        int mmask=1<<n;
        vector<vector<int>> dp(mmask, vector<int>(k+1,1e9));
        vector<int> cnt(mmask);
        for(int mask=1;mask<mmask;mask++) {
            for(int i=0;i<n;i++) {
                if((1<<i)&mask) {
                    cnt[mask]+=jobs[i];
                }
            }
        }
        for(int mask=1;mask<mmask;mask++) {
            dp[mask][1]=cnt[mask];
            for(int kk=2;kk<=k;kk++) {
                for(int tmp=mask; tmp; tmp = mask & (tmp - 1)) {
                    dp[mask][kk]=min(dp[mask][kk],max(cnt[tmp],dp[mask-tmp][kk-1]));
                }
            }
        }
        return dp[mmask-1][k];
    }
};
