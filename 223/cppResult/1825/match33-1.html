class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<vector<int>> dp(1<<n, vector<int>(k+1, 1e9));

        for(int i=0;i<(1<<n);i++) {
            dp[i][1] = cal(jobs, i);
        }
        for(int d=2;d<=k;d++) {
            for(int i=0;i<(1<<n);i++) {
                for (int j = i; j; j = (j - 1) & i) {
                    // cout<<i<<" "<<j<<" "<<(i^j)<<endl;
                    dp[i][d] = min(dp[i][d], max(dp[i^j][1], dp[j][d-1]));
                }
            }
        }
        return dp[(1<<n)-1][k];
    }

    int cal(vector<int>& jobs, int mask) {
        int res = 0;
        for(int i=0;i<jobs.size();i++) {
            if((1<<i)&mask) res+=jobs[i];
        }
        return res;
    }
};
