class Solution {
public:
    int minimumTimeRequired(vector<int>& a, int k) {
        int n = a.size();
        vector<int> t(1 << n);
        for (int i = 1; i < (1 << n); ++i) {
            int o = __builtin_ctz(i & (-i));
            t[i] = t[i ^ (i & (-i))] + a[o];
        }
        vector<int> f(1 << n);
        fill(f.begin(), f.end(), INT_MAX / 2);
        f[0] = 0;
        for (int j = 1; j <= k; ++j) {
            for (int mask = (1 << n) - 1; mask > 0; --mask) {
                for (int sub = mask; sub; sub = (sub - 1) & mask) {
                    f[mask] = min(f[mask], max(f[mask ^ sub], t[sub]));
                }
            }
        }
        return f[(1 << n) - 1];
    }
};
