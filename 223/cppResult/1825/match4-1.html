class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<vector<int>> v((1<<n) + 1, vector<int>(k + 1, 1e9));

        vector<int> gg(1<<n);
        for(int i = 0; i < (1<<n); ++i){
            int s = 0;
            for(int j = 0; j < n; ++j){
                if(i&(1<<j)){
                    s += jobs[j];
                }
            }
            gg[i] = s;
        }
        for(int i = 0; i < (1<<n); ++i){
            int sum = 0;
            v[i][1] = gg[i];

            for(int j = i; j; j=(i&(j-1))){
                for(int u = 1; u < k; u++){
                    v[i][u+1] = min(v[i][u+1], max(v[j][u], gg[i^j]));
                }
            }
        }
        return v[(1<<n)-1][k];
    }
};
