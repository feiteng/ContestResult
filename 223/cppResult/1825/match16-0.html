class Solution {
public:
    int minimumTimeRequired(vector<int>& a, int m) {
        int n = a.size();
        vector<int>f=vector<int>(1<<n, 1000000007);
        f[0] = 0;
        vector<int>g=vector<int>(1<<n, 0);
        for(int i=1;i<(1<<n);i++){
            int sum=0;
            for(int j=0;j<n;j++){
                if(i&(1<<j))sum+=a[j];
            }
            g[i] = sum;
        }
        for(int j=1;j<=m;j++)
        for(int i=(1<<n)-1;i>0;i--){
            for(int k=i;k;k=(k-1)&i){
                    int tt = max(f[i^k],g[k]);
                    f[i] = min(f[i], tt);
            }
        }
        int ret = 1000000007;
        for(int j=1;j<=m;j++)ret=min(ret, f[(1<<n)-1]);
        return ret;
    }
};
