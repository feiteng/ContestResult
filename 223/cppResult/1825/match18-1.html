int f[13][1 << 12];
int s[1 << 12];

class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        int m = 1 << n;
        memset(f, 0x3f, sizeof f);
        memset(s, 0, sizeof s);

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i & (1 << j))
                    s[i] += jobs[j];
            }
            f[0][i] = s[i];
        }
        for (int i = 1; i < k; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int x = j; x; x = (x - 1) & j) {
                    f[i][j] = min(f[i][j], max(f[i - 1][x], s[j ^ x]));
                }
            }
        }
        return f[k - 1][m - 1];
    }
};
