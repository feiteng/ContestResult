class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<int> time(1<<n);
        for (int i = 0; i < 1<<n; i++) {
            int cost = 0;
            for (int j = 0; j < n; j++) {
                if (i>>j & 1) {
                    cost += jobs[j];
                }
            }
            time[i] = cost;
        }
        vector<vector<int>> dp(k+1, vector<int>(1<<n, INT_MAX));
        dp[0][0] = 0;
        for (int i = 1; i <= k; i++) {
            for (int j = 1; j < 1<<n; j++) {
                for (int m = j; m; m = (m-1) & j) {
                    dp[i][j] = min(dp[i][j], max(time[m], dp[i-1][j-m]));
                }
            }
        }
        return dp[k][(1<<n)-1];
    }
};
