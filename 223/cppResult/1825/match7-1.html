class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = jobs.size();
        vector<int> cost(1 << n, 0);
        vector<vector<int> > dp(1 << n, vector<int> (k + 1, 1e9));
        
        for (int mask = 0; mask < 1 << n; mask++) {
            for (int j = 0; j < n; j++) {
                if (mask >> j & 1) cost[mask] += jobs[j];
            }
        }
        
        dp[0][0] = 0;
        for (int mask = 0; mask < 1 << n; mask++) {
            for (int i = 1; i <= k; ++i) {
                int &v = dp[mask][i];

                v = 1e9;
                for (int submask = mask; submask > 0; submask = (submask - 1) & mask) {
                    v = min(v, max(dp[mask - submask][i - 1], cost[submask]));
                }
            }
        }
        
        //cout << dp[1][1] << ' ' << dp[3][1] << endl;
        return dp[(1 << n) - 1][k];

    }
};
