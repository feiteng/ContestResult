class Solution {
public:
    int minimumTimeRequired(vector<int>& a, int m) {
        int n = a.size();
        vector<int> sum(1 << n);
        for (int S = 0; S < (1 << n); ++S) {
            for (int i = 0; i < n; ++i) {
                if ((S >> i) & 1) sum[S] += a[i];
            }
        }
        vector<vector<int>> dp(1 << n, vector<int>(m + 1));
        for (int S = 1; S < (1 << n); ++S) {
            dp[S][0] = 1e9;
            for (int j = 1; j <= m; ++j) {
                dp[S][j] = 1e9;
                for (int T = S; T; T = (T - 1) & S) {
                    dp[S][j] = min(dp[S][j], max(sum[T], dp[S ^ T][j - 1]));
                }
            }
        }
        return dp[(1 << n) - 1][m];
    }
};
