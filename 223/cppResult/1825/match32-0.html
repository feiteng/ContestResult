class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int m) {
        int n = jobs.size();
        vector<vector<int>> f(1 << n, vector<int>(m + 1, 0x3f3f3f3f));
        f[0][0] = 0;
        vector<int> cost(1 << n);
        for(int i = 0; i < 1 << n; i++) {
            for(int j = 0; j < n; j++) {
                if(i >> j & 1)
                    cost[i] += jobs[j];
            }
        }
        for(int i = 0; i < 1 << n; i++) {
            for(int j = 1; j <= m; j++) {
                for(int k = i; k; k = (k - 1) & i) {
                    if(__builtin_popcount(i ^ k) >= j - 1) f[i][j] = min(f[i][j], max(f[i ^ k][j - 1], cost[k]));
                }
            }
        }
        return f[(1 << n) - 1][m];
    }
};
