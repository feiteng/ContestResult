class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int n = 1 << jobs.size();
        vector<int> time(n, 0);
        vector<vector<int>> record(k, vector<int>(n, INT_MAX));
        
        for(int j = 0; j < n; ++j) {
            for(int i = 0; i < jobs.size(); ++i) {
                if((1 << i) & j) time[j] += jobs[i];
            }
            record[0][j] = time[j];
        }
        
        for(int i = 1; i < k; ++i) {
            for(int j = 0; j < n; ++j) {
                record[i][j] = record[i - 1][j];
                for(int jj = j; jj; jj = j & (jj - 1)) {
                    record[i][j] = min(record[i][j], max(record[i - 1][(jj ^ j) & j], time[jj]));
                }
            }
        }
        
        return record.back().back();
    }
};
