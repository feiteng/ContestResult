#include <bits/stdc++.h>
using namespace std;


class Solution {
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int size = jobs.size();
        vector<int> times(1<<size,0);
        for(int i = 0;i<(1 << size);i++){
            int sum = 0;
            for(int j = 0;j<size;j++){
                if((i>>j&1))sum += jobs[j];
            }
            times[i] = sum;
        }
        vector<vector<int>> dp(k+1,vector<int>(1<<size,INT_MAX));
        for(int i = 0;i<(1<<size);i++){dp[1][i] = times[i];}

        for(int i = 2;i<=k;i++){
            for(int j = 0;j < (1 << size);j++){
                for(int s = j;s != 0;s = ((s-1)&j)){
                    int prev = j - s;
                    dp[i][j] = min(dp[i][j],max(dp[i-1][prev],times[s]));
                }
            }
        }
        return dp[k][(1<<size)-1];
    }
};
