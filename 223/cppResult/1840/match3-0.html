class dsu{
    vector<int>parent;
    vector<int>rank;
    public:
    dsu(int n) {
        parent.resize(n);
        iota(parent.begin(),parent.end(),0);
        rank.resize(n,1);
    }
    int find_par(int x) {
        if( parent[x] == x) return x;
        return parent[x] = find_par(parent[x]);
    }
    void uni(int a,int b) {
        int x = find_par(a);
        int y = find_par(b);
        if(x == y) return ;
        if(rank[x]>rank[y]) {
            parent[y] = x;
            rank[x] += rank[y];
        } else {
            parent[x] = y;
            rank[y] += rank[x];
        }
    }
};
class Solution {
public:
    int minimumHammingDistance(vector<int>& A, vector<int>& B, vector<vector<int>>& C) {
        int n = A.size();
    dsu d(n);
    for(auto p : C) {
        d.uni(p[0],p[1]);
    }
    map<int,unordered_map<int,int>> m;
    for( int i = 0; i < n; ++i) {
        int x = d.find_par(i);
        m[x][A[i]] += 1;
    }
    int ans = 0;
    for(int i = 0; i < n;++i) {
        int x = d.find_par(i);
        if(m[x][B[i]]-- > 0)  {
            ans += 1;
        }
    }
    return  A.size() - ans;
    }
};
