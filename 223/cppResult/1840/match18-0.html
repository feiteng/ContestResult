class Solution {
    int getf(vector<int>& f, int a)
    {
        if(f[a] != a)
        {
            f[a] = getf(f, f[a]);
        }
        return f[a];
    }
    
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        vector<int> f(n, 0);
        for(int i = 1; i < n; ++i)f[i] = i;
        
        for(auto x:allowedSwaps)
        {
            int a = x[0], b = x[1];
            int ta = getf(f, a);
            int tb = getf(f, b);
            if(ta != tb)f[ta] = tb;
        }
        
        unordered_map<int, unordered_map<int, int>> ss;
        for(int i = 0; i < n; ++i)
        {
            int t = getf(f, i);
            ++ss[t][source[i]];
        }
        
        int ans = 0;
        for(int i = 0; i < n; ++i)
        {
            int t = getf(f, i);
            if(--ss[t][target[i]] < 0)++ans;
        }
        
        return ans;
    }
};
