class Solution {
    int find(vector<int>& parent, int child) {
        if (parent[child] != child) {
            parent[child] = find(parent, parent[child]);
        }
        return parent[child];
    }
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        vector<int> parent(n, 0);
        for (int i = 0; i < n; ++i) parent[i] = i;
        for (vector<int> sw : allowedSwaps) {
            int pp = find(parent, sw[0]);
            int pc = find(parent, sw[1]);
            parent[pc] = pp;
        }
        unordered_map<int, unordered_map<int, int>> map;
        for (int i = 0; i < n; ++i) {

            ++((map[find(parent, i)])[source[i]]);
        }
        int res = 0;
        for (int i = 0; i < n; ++i) {
            int circle = find(parent, i);
            if ((map[circle])[target[i]] == 0) ++res;
            else --((map[circle])[target[i]]);
        }
        return res;
    }
};
