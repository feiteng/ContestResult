int h[100005] = {0};
class Solution {
public:
    int find(int x, int a[]) {
        return (a[x] == -1)? x : (a[x] = find(a[x], a));
    }
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        int dsu[n];
        memset(dsu, -1, sizeof(dsu));
        
        for(auto&p : allowedSwaps) {
            int x = find(p[0], dsu);
            int y = find(p[1], dsu);
            if(x != y) {
                dsu[x] = y;
            }
        }
        
        vector<int> v[n];
        
        for(int i = 0; i < n; ++i) {
            int x = find(i, dsu);
            v[x].push_back(i);
        }
        
        int res = 0;
        
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < v[i].size(); ++j) {
                h[source[v[i][j]]]++;
            }
            for(int j = 0; j < v[i].size(); ++j) {
                if(h[target[v[i][j]]] > 0) {
                    h[target[v[i][j]]]--;
                } else {
                    ++res;
                }
            }
            for(int j = 0; j < v[i].size(); ++j) {
                h[source[v[i][j]]] = 0;
            }
        }
        
        return res;
    }
};
