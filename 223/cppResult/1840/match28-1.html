const int mx = (int)1e5+1;
vector<int> pa(mx);
int f(int x){return (x==pa[x]) ? x : pa[x] = f(pa[x]);}

class Solution {
public:

    int minimumHammingDistance(vector<int>& arr, vector<int>& brr, vector<vector<int>>& swaps) {
          int n = arr.size();
          for(int i=0;i<n;i++)pa[i] = i;
          for(auto it : swaps){
              int x = f(it[1]);
              int y = f(it[0]);
              if(x != y){
                  pa[x] = y;
              }

          }
          unordered_map<int,unordered_map<int,int>> mp;
        for(int i=0;i<n;i++){
            int p = f(i);
            mp[p][arr[i]]+=1;
        }
          int ans = 0;
        for(int i=0;i<n;i++){
             int p = f(i);
             if(mp[p].find(brr[i]) != mp[p].end()){
                 ans++;
                 if(mp[p][brr[i]] == 1)mp[p].erase(brr[i]);
                 else mp[p][brr[i]]-=1;
             }
        }
        return n-ans;
    }
};
