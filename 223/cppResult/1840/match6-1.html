class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        vector<int> root(n);
        iota(begin(root), end(root), 0);
        for (auto & p: allowedSwaps) {
            int a = find(root, p[0]);
            int b = find(root, p[1]);
            if (a != b) root[a] = b;

        }
        unordered_map<int, vector<int>> m;
        for (int i=0; i<n; ++i) {
            int r = find(root, i);
            m[r].push_back(i);
        }
        
        int res = 0;
        for (auto & it: m) {
            auto & v = it.second;
            unordered_map<int, int> m1, m2;
            for (auto & x: v) {
                ++m1[source[x]];
                ++m2[target[x]];
            }
            
            int same = 0;

            for (auto & it2: m2) {
                same += min(it2.second, m1[it2.first]);
            }
            
            res += v.size() - same;
        }
        
        return res;
    }
    
    int find(vector<int> & root, int x) {
        if (root[x] != x) root[x] = find(root, root[x]);
        return root[x];
    }
};
