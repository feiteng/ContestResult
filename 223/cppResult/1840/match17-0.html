class Solution {
public:
    vector<int> fa;
    int get(int x){
        return x==fa[x]?x:fa[x]=get(fa[x]);
    }
    int minimumHammingDistance(vector<int>& a, vector<int>& b, vector<vector<int>>& c) {
        int n=a.size();
        vector<vector<int>> v(n+1);
        fa.resize(n+1);
        for(int i=0;i<=n;i++) fa[i]=i;
        for(int i=0;i<c.size();i++){
            int x=get(c[i][0]);
            int y=get(c[i][1]);
            if(x!=y) fa[y]=x;
        }
        for(int i=0;i<n;i++){
            int x=get(i);
            v[x].push_back(i);
        }
        int ans=0;
        for(int i=0;i<n;i++){
            unordered_map<int,int> cnt;
            for(int j=0;j<v[i].size();j++){
                cnt[a[v[i][j]]]++;
            }
            for(int j=0;j<v[i].size();j++){
                if(cnt[b[v[i][j]]]){
                    ans++;
                    cnt[b[v[i][j]]]--;
                }
            }
        }
        return n-ans;
        
    }
};
