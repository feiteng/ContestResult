const int N = 1e5 + 5;

class Solution {
public:
    int fa[N];
    
    int getfa(int x) {
        if (x == fa[x]) return fa[x];
        return fa[x] = getfa(fa[x]);
    }
    
    void merge(int x, int y) {
        x = getfa(x) , y = getfa(y);
        if (x == y) return;
        fa[x] = y;

    }
    
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        for (int i = 0; i < n; ++i) fa[i] = i;
        
        for (auto it : allowedSwaps) {
            merge(it[0], it[1]);

        }
        
        map< int, map<int,int>> st;
        for (int i = 0; i < n; ++i) {
            int p = getfa(i);
            st[p][source[i]]++;
        }
        
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int p = getfa(i);
            if (!st[p][target[i]]) ans++;
            else st[p][target[i]]--;
        }
        
        return ans;
        
    }
};
