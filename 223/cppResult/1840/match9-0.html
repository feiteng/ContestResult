class Solution {
public:
    vector<int> fa;

    int find(int x) {
        if (x != fa[x]) {
            fa[x] = find(fa[x]);
        }
        return fa[x];
    }

    void us(int x, int y) {
        x = find(x);
        y = find(y);
        fa[x] = y;
    }
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        fa.resize(n);
        for (int i = 0; i < n; ++i) {
            fa[i] = i;
        }
        for (auto &it : allowedSwaps) {
            us(it[0], it[1]);
        }
        unordered_map<int, vector<int>> um;
        for (int i = 0; i < n; ++i) {
            um[find(i)].push_back(i);
        }
        unordered_map<int, int> cnt1;
        int ans = 0;
        for (auto &it : um) {
            cnt1.clear();
            for (auto &itt : it.second) {
                cnt1[source[itt]]++;
                cnt1[target[itt]]--;
            }
            for (auto &itt : cnt1) {
                ans += abs(itt.second);
            }
        }
        return ans / 2;
    }
};
