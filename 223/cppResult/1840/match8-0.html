class Solution {
    static const int MAXN = 1e5 + 10;
    int p[MAXN];
public:
    int find(int x) {
        if (p[x] == x) return x;
        return p[x] = find(p[x]);
    }
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        for (int i = 0; i < n; i++) p[i] = i;
        for (auto s : allowedSwaps) {
            int x = s[0];
            int y = s[1];
            p[find(x)] = find(y);
        }
        map<int, vector<int>> m;
        for (int i = 0; i < n; i++) {
            int x = find(i);
            m[x].push_back(i);
        }
        int ans = 0;
        for (const auto& pa : m) {
            const auto& v = pa.second;
            map<int, int> c1, c2;
            for (auto i : v) {
                //cout << "i:" << i << " ";
                c1[source[i]]++;
                c2[target[i]]++;
            }
            int match = 0;
            for (auto p : c1) {
                match += min(c1[p.first], c2[p.first]);
            }
            //cout << v.size() << " " << match << endl;
            ans += v.size() - match;
        }
        return ans;
    }
};
