int pre[100005];
int Find(int x)
{
    if(x!=pre[x])
    {
        pre[x]=Find(pre[x]);
    }
    return pre[x];
}

class Solution {

public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n=source.size();
        map<pair<int,int>,int> cnt;
        for(int i=0;i<=n;i++)
        {
            pre[i]=i;
        }
        for(int i=0;i<allowedSwaps.size();i++)
        {
            int a=allowedSwaps[i][0],b=allowedSwaps[i][1];
            int x=Find(a);
            int y=Find(b);
            if(x!=y)
            {
                pre[x]=y;
            }

        }
        for(int i=0;i<n;i++)
        {
            int t=Find(i);
            cnt[make_pair(t,target[i])]++;
        }
        int ans=n;
        for(int i=0;i<n;i++)
        {
            int t=Find(i);
            if(cnt[make_pair(t,source[i])])
            {
                cnt[make_pair(t,source[i])]--;
                ans--;
            }
        }
        return ans;
    }
};
