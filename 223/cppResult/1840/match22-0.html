class Solution {
    vector<int> idx;
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        for (int i=0; i<n; i++)
            idx.push_back(i);
        for (auto &temp:allowedSwaps) {
            join(temp[0],temp[1]);
        }
        
        
        unordered_map<int, unordered_map<int,int>> counter;
        for (int i=0; i<n; i++) {
            counter[find(i)][source[i]]++;
            counter[find(i)][target[i]]--;
        }
        
        int dist = 0;
        for (auto &temp1: counter) {
            for (auto &temp2: temp1.second) {
                dist += temp2.second>0? temp2.second:0;
            }
        }
        return dist;
        
    }
    
    int find(int x) {
        if (idx[x]==x)
            return x;
        return idx[x] = find(idx[x]);
    }
    
    void join(int x, int y) {
        int idxx = find(x);
        int idxy = find(y);
        if (idxx!=idxy) {
            idx[idxy] = idxx;
            idx[y] = idxx;
        }
        return;
    }
};
