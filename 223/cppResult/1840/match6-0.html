class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int L = source.size();
        vector<int> parent(L);
        for(int i=0; i<L; i++) {
            parent[i] = i;
        }
        for(auto s: allowedSwaps) {
            int p1 = find(parent, s[0]);
            int p2 = find(parent, s[1]);
            if(p1 != p2) {
                parent[p1] = min(p1, p2);
                parent[p2] = min(p1, p2);
            }
        }
        unordered_map<int, vector<int>> groups;
        for(int i=0; i<L; i++) {
            int p = find(parent, i);
            groups[p].push_back(i);
        }
        // get array at a given group indexes and compare
        int ans = 0;
        for(auto p : groups) {
            unordered_map<int, int> freq;
            for(auto i : p.second) {
                freq[source[i]]++;
                freq[target[i]]--;
            }
            for(auto t : freq) {
                ans += abs(t.second);
            }
            ans = ans;
        }
        return ans/2;
    }
    
    int find(vector<int>& parent, int i) {
        if(parent[i] != i) {
            parent[i] = find(parent, parent[i]);
        }
        return parent[i];
    }
};
