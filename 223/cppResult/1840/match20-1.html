const int maxn = 1e5 + 6;
class Solution {
public:
    int minimumHammingDistance(vector<int>& s, vector<int>& t, vector<vector<int>>& es) {
        n = s.size();
        init(n);
        for(vector<int>& e : es) merge(e[0] , e[1]);

        for(int i = 0; i < n; ++i){
            a[find(i)].push_back(s[i]);
            b[find(i)].push_back(t[i]);
        }
        int ret = 0;
        for(int i = 0; i < n; ++i){
            unordered_map<int , int> cnt;
            for(int j : a[i]) cnt[j]++;
            for(int j : b[i]) if(cnt[j] > 0)--cnt[j];
            for(auto& e : cnt)ret += e.second;
        }
        return ret;
    }
private:
    int ma[maxn], n;
    unordered_map<int, vector<int>> a, b;
    void init(int n) {for(int i = 0; i < n; ++i) ma[i] = i;}

    int find(int x){ return ma[x] == x ? x : ma[x] = find(ma[x]);}
    void merge(int x , int y) { x = find(x), y = find(y); ma[x] = y;}
};
