const int N = 1e5 + 10;
int f[N];
int find(int x) {
    if (f[x] != f[f[x]]) f[x] = find(f[x]);
    return f[x];
}
class Solution {
public:

    int minimumHammingDistance(vector<int>& a, vector<int>& b, vector<vector<int>>& c) {
        int n = a.size();
        for (int i = 0; i < n; ++i) f[i] = i;
        for (auto& v : c) {
            int x = v[0], y = v[1];
            int rx = find(x), ry = find(y);
            if (rx != ry) f[ry] = rx;
        }

        vector<vector<int>> pos(n);
        for (int i = 0; i < n; ++i) {
            int k = find(i);
            pos[k].push_back(i);
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            if (pos[i].size() == 0) continue;

            map<int, int> A, B;
            for (auto& x : pos[i]) {
                A[a[x]]++;
                B[b[x]]++;
            }
            int cur = pos[i].size();
            for (auto& [k, v] : A) {
                cur -= min(A[k], B[k]);
            }
            ret += cur;
        }
        return ret;
    }
};
