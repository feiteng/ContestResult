class Solution {
public:
	vector<int>p;

	int getroot(int x) {
		if (p[x] == x)return x;
		return p[x] = getroot(p[x]);
	}
	void join(int x, int y) {
		int a = getroot(x), b = getroot(y);
		if (a == b)return;
		p[a] = b;
	}
	int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& swaps) {
		int n = source.size();
		p.resize(n);

		for (int i = 0; i < n; i++)p[i] = i;
		for (auto it : swaps) {
			join(it[0], it[1]);
		}
		unordered_map<int, unordered_map<int, int>>ct,ct2;
		for (int i = 0; i < n; i++) {
			ct[getroot(i)][source[i]]++;
			ct2[getroot(i)][target[i]]++;
		}
		int ret = 0;
		for (auto& it : ct) {
			for (auto& jt : it.second) {
				ret += min(jt.second, ct2[it.first][jt.first]);
			}
		}
		return n - ret;
	}
};
