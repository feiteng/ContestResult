const int maxn = 2e5 + 10;
class Solution {
public:
  int fa[maxn], s[maxn], t[maxn];

multiset<int> st[maxn];
int ff(int x)
{
	return fa[x] == x ? x : fa[x] = ff(fa[x]);
}
int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
	int n = source.size();

	for (int i = 1; i <= n; i++)
		fa[i] = i;
	for (int i = 0; i < allowedSwaps.size(); i++)
	{
		int fax = ff(allowedSwaps[i][0] + 1), fay = ff(allowedSwaps[i][1] + 1);
		if (fax != fay)
			fa[fax] = fay;
	}
	for (int i = 0; i < n; i++)
	{
		s[i + 1] = source[i];

		t[i + 1] = target[i];
	}
	for (int i = 1; i <= n; i++)
	{
		int fai = ff(i);
		st[fai].insert(s[i]);
	}
	int ans = 0;
	for (int i = 1; i <= n; i++)
	{
		int fai = ff(i);
		if (st[fai].find(t[i]) == st[fai].end()) ans++;
		else
			st[fai].erase(st[fai].find(t[i]));
	}
	return ans;
}
};
