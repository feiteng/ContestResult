const int maxn = 1e5 + 10;
class Solution {
public:
    
    int fa[maxn];
    int n;
    
    vector<int> a[maxn] , b[maxn];
    
    int find(int x){
        return fa[x] == x ? x : fa[x] = find(fa[x]);
    }
    
    void merge(int x , int y){
        x = find(x);
        y = find(y);
        fa[x] = y;
    }
    
    int minimumHammingDistance(vector<int>& s, vector<int>& t, vector<vector<int>>& es) {
        n = s.size();
        for(int i = 0; i < n; ++i)fa[i] = i;
        for(vector<int>& e : es){
            merge(e[0] , e[1]);
        }
        for(int i = 0; i < n; ++i){
            a[find(i)].push_back(s[i]);
            b[find(i)].push_back(t[i]);
        }
        int ret = 0;
        for(int i = 0; i < n; ++i){
            unordered_map<int , int> cnt;
            for(int j : a[i])cnt[j]++;
            for(int j : b[i])if(cnt[j] > 0)--cnt[j];
            for(auto& e : cnt)ret += e.second;
        }
        return ret;
    }
};
