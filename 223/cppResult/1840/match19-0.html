class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = target.size();
        vector<int>root(n, 0);
        for(int i = 0; i < n; i++)
        {
            root[i] = i;
        }
        for(auto a:allowedSwaps)
        {
            int x = a[0], y = a[1];
            uni(root, x, y);
            
        }
        unordered_map<int, unordered_map<int, int>>m;
        for(int i = 0; i < source.size(); i++)
        {
            int idx = find(root, i);
            m[idx][source[i]]++;
        }
        int res = 0;
        for(int i = 0; i < target.size(); i++)
        {
            int idx = find(root, i);
            if(m[idx][target[i]] > 0)
            {
                ++res;
                --m[idx][target[i]];
            }
        }
        return n - res;
    }

    void uni(vector<int>& root, int x, int y) {
        root[find(root, x)] = find(root, y);
    }
    
    int find(vector<int>& root, int x) {
        return root[x] == x?root[x]:root[x] = find(root, root[x]);
    }
};
