class Solution {
public:
    int par[100005];
    
    int getRoot(int u) {
        if (u != par[u]) {
            par[u] = getRoot(par[u]);
        }

        return par[u];
    }
   
    void merge(int u, int v) {
        u = getRoot(u);
        v = getRoot(v);
        if (u != v) {
            par[v] = u;
        }
    }
    
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        for (int i = 0; i < n; i++) {
            par[i] = i;
        }
        

        for (auto p : allowedSwaps) {
            merge(p[0], p[1]);
        }
        
        map<int, vector<int> > mp;
        for (int i = 0; i < n; i++) {
            mp[getRoot(i)].push_back(i);
        }
        
        
        int ret = 0;
        for (auto it : mp) {
            auto idx = it.second;
            

            ret += idx.size();
            
            map<int,int> sc;
            for (auto i : idx) {
                sc[source[i]]++;
            }
            for (auto i : idx) {
                if (sc[target[i]] > 0) {
                    ret--;
                }
                sc[target[i]]--;
            }
        }
        
        return ret;
    }
};
