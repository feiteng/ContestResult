class Solution {
public:
    int f[100005];
    int Find(int x){
        if(f[x]==x)return x;
        return f[x]=Find(f[x]);
    }
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        for(int i = 0;i<source.size();i++){
            f[i]=i;
        }

        for(auto it:allowedSwaps){
            f[Find(it[0])]=Find(it[1]);
        }
        vector<vector<int> > v(source.size()),v2(source.size());
        for(int i = 0;i<target.size();i++){
            v[Find(i)].push_back(source[i]);
            v2[Find(i)].push_back(target[i]);
        }
        int ans=0;
        for(int i = 0;i<source.size();i++){
            sort(v[i].begin(),v[i].end());
            sort(v2[i].begin(),v2[i].end());
            int l=0,r=0;
            while(l!=v[i].size()&&r!=v2[i].size()){
                if(v[i][l]==v2[i][r]){
                    l++,r++;
                    continue;
                }
                else if(v[i][l]<v2[i][r]){
                    ans++;
                    l++;
                }
                else{
                    r++;
                }
            }
            ans+=v[i].size()-l;
        }
        return ans;
    }
};
