int f[100005];
int c1[100005];
vector <int> v[100005];

class Solution {
public:

    int find(int i) {
        return f[i] == i ? i : f[i] = find(f[i]);
    }
    
    int minimumHammingDistance(vector<int>& a, vector<int>& b, vector<vector<int>>& g) {
        int n = a.size();

        for (int i = 0; i <= n; i++) {
            f[i] = i;
        }
        for (int i = 0; i < g.size(); i++) {
            int fa = find(g[i][0]), fb = find(g[i][1]);

            f[fa] = fb;
        }
        for (int i = 0; i < n; i++) {
            int fi = find(i);
            v[fi].push_back(i);
        }
        int ans = n;
        for (int i = 0; i < n; i++) {
            for (int j : v[i]) {
                c1[a[j]]++;
            }
            for (int j : v[i]) {
                if (c1[b[j]] > 0) {
                    ans--;
                    c1[b[j]]--;
                }
            }
            for (int j : v[i]) {
                c1[a[j]] = 0;
            }
        }
        for (int i = 0; i < n; i++) {
            v[i].clear();
        }
        return ans;
    }
};
