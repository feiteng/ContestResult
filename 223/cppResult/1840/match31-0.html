class Solution {
public:
	int find(int x,vector<int> &par){
		if(par[x]==x) return x;
		return par[x] = find(par[x],par);
	}
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
    	int n = source.size();
    	vector<int> par(n);
    	for(int i=0;i<n;i++) par[i] = i;
    	for(auto i: allowedSwaps){
    		par[find(i[0],par)] = find(i[1],par);
    	}
    	vector<multiset<int> > s(n);
    	for(int i=0;i<n;i++){
    		s[find(i,par)].insert(source[i]);
    	}
    	int ans = 0;
    	for(int i=0;i<n;i++){
    		int x = find(i,par);
    		if(!s[x].count(target[i])) ans++;
    		else{
    			s[x].erase(s[x].find(target[i]));
    		}
    	}
    	return ans;
    }
};
