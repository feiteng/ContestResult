class Solution {
public:
    int f[100005];
    int find(int x){
        return x==f[x]?x:f[x]=find(f[x]);
    }
    void merge(int x,int y){
        int tx=find(x),ty=find(y);
        if(tx!=ty){
            f[tx]=ty;
        }
    }
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& a) {
        int n=source.size();
        for(int i=0;i<n;i++){
            f[i]=i;
        }
        for(int i=0;i<a.size();i++){
            merge(a[i][0],a[i][1]);
        }
        map<int,map<int,int>>mp;
        for(int i=0;i<n;i++){
            int t=find(i);
            mp[t][target[i]]++;
        }
        int res=0;
        for(int i=0;i<n;i++){
            int t=find(i);
            if(mp[t][source[i]]>0){
                res++;
                mp[t][source[i]]--;
            }
        }
        return n-res;
    }
};
