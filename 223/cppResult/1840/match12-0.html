class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        vector<int> parent(n);
        iota(parent.begin(), parent.end(), 0);
        for (auto& v : allowedSwaps)    uni(parent, v[0], v[1]);
        unordered_map<int, vector<int>> ggg;
        for (int i = 0; i < n; i++)    ggg[find(parent, i)].push_back(i);
        
        int res = 0;
        for (auto& [k, gg] : ggg) {
            unordered_map<int, int> cnt;
            for (int g : gg)    cnt[source[g]]++;
            for (int g : gg) {
                if (cnt.count(target[g]) > 0) {
                    if (--cnt[target[g]] == 0)    cnt.erase(target[g]);
                } else    res++;
            }
        }
        return res;
    }
    
private:
    int find(vector<int>& parent, int a) {
        if (parent[a] != a)
            parent[a] = find(parent, parent[a]);
        return parent[a];
    }

    void uni(vector<int>& parent, int a, int b) {
        int ra = find(parent, a), rb = find(parent, b);
        if (ra < rb)    parent[rb] = ra;
        else    parent[ra] = rb;
    }
};
