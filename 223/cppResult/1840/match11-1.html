class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {
        int n = source.size();
        vector<int> root(n);
        for (int i = 0; i < n; ++i) root[i] = i;
        for (auto& p: allowedSwaps) {
            int r1 = getRoot(root, p[0]);
            int r2 = getRoot(root, p[1]);
            // cout << p[0] << ' ' << p[1] << endl;
            // cout << r1 << ' ' << r2 << endl;
            if (r1 != r2) {
                root[r1] = r2;
            } 
        }
        
        // for (int i: root) cout << i << ' ';
        // cout << endl;
        

        unordered_map<int, vector<int>> groups;
        for (int i = 0; i < n; ++i) groups[getRoot(root, i)].push_back(i);
        int res = n;
        for (auto& g: groups) {
            // for (int i: g.second) cout << i << ' ';
            // cout << endl;
            vector<int> t;
            unordered_map<int, int> s;
            for (int i: g.second) s[source[i]] ++;
            for (int i: g.second) {
                if (s.count(target[i]) && s[target[i]] > 0) {
                    res --;

                    s[target[i]] --;
                }
            }
        }
        
        return res;
    }
    
    int getRoot(vector<int>& root, int cur) {
        if (root[cur] != cur) root[cur] = getRoot(root, root[cur]);
        return root[cur];
    }
};
