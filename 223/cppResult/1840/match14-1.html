class Solution {
public:
    vector<int> f;
    int find(int i) {
        if (f[i]==i) return f[i];
        f[i] = find(f[i]);
        return f[i];

    }
    int minimumHammingDistance(vector<int>& s, vector<int>& t, vector<vector<int>>& a) {
        int n = s.size();
        f.clear();
        for (int i = 0; i < n; ++i)
            f.push_back(i);
        for (int i = 0; i < a.size(); ++i)
            f[find(a[i][0])] = find(a[i][1]);
        vector<unordered_map<int,int>> sub;
        for (int i = 0; i < n; ++i) {
            find(i);
            sub.push_back(unordered_map<int,int>());
        }
        

        for (int i = 0; i < n; ++i) {
            int fa = find(i);
            sub[fa][t[i]]++;
        }
        int ans = n;
        for (int i = 0; i < n; ++i) {
            int fa = find(i);
            if (sub[fa][s[i]]>0) {
                sub[fa][s[i]]--;
                --ans;
            }
        }
        return ans;
    }
};
