class Solution {
  public:
    vector<int> decode(vector<int>& encoded, int first) 
    {
      vector<int> origin(encoded.size() + 1);
      origin[0] = first;
      for (int i = 1; i < origin.size(); ++i) {
        origin[i] = origin[i - 1] ^ encoded[i - 1];
      }
      return origin;
    }
};
