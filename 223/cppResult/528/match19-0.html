class Solution {
   public:
    ListNode *swapNodes(ListNode *head, int k) {
        ListNode *L = head;
        for (int i = 0; i < k - 1; ++i) L = L->next;
        ListNode *fast = L->next, *low = head;
        while (fast) {
            fast = fast->next;
            low = low->next;
        }
        swap(L->val, low->val);
        return head;
    }
};
