/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int id=1;
        ListNode* hd = head;
        while(hd->next!=NULL)
        {
            id++;
            hd = hd->next;
        }
        
        ListNode* tp1 = head;
        ListNode* tp2 = head;
        
        for(int i=0;i<k-1;i++)
            tp1 = tp1->next;
        
        for(int i=0;i<id-k+1-1;i++)
            tp2 = tp2->next;
        
        swap(tp1->val, tp2->val);
        return head;
        
    }
};
