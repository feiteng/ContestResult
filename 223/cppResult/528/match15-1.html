/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {

        ListNode* cur = head;
        int num = 1;
        while(num<k){
            cur = cur->next;
            num++;
        }
        ListNode* num1 = cur;
        ListNode* pre = head;
        while(cur->next){
            cur = cur->next;
            pre = pre->next;
        }
        int temp = pre->val;
        pre->val = num1->val;
        num1->val = temp;
        return head;
    }
};
