/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int n = 0;
        ListNode* p = head;
        while (p != nullptr) {
            ++n;
            p = p->next;
        }
        ListNode *p1 = head;
        ListNode *p2 = head;
        int i = 1;
        while (i < k) {
            p1 = p1->next;
            ++i;
        }
        i = 0;

        while (i < n - k) {
            p2 = p2->next;
            ++i;
        }
        int tmp = p2->val;
        p2->val = p1->val;
        p1->val = tmp;
        return head;
    }
};
