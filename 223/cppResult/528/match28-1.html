/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* n1 = head;
        ListNode* n2 = head;
        for(int i = 1; i < k; i++)
            n1 = n1->next;

        auto n3 = n1->next;
        while(n3){
            n3 = n3->next;
            n2 = n2->next;
        }
        
        // cout << n1->val;
        // cout << n2->val;
        // cout << n3->val;
        int tmp = n1->val;
        n1->val = n2->val;
        n2->val = tmp;
        return head;
    }
};
