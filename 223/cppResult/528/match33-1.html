/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* first, *last;
        vector<ListNode*> arr;
        last = head;
        while(last){
            arr.push_back(last);
            last = last->next;
        }
        swap(arr[k-1]->val, arr[arr.size()-k]->val);
        return head;
    }
};
