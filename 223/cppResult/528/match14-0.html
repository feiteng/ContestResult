/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* h, int k) {
        ListNode *a, *b, *p = h, *q = h;
        for(int i = 1; i < k; ++i)
            p = p->next;
        a = p;
        while(p->next)
            q = q->next, p = p->next;
        swap(a->val, q->val);
        return h;
    }
};
