/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        vector<ListNode*> ls;
        ls.reserve(100000);
        ListNode* probe = head;
        while(probe){
            ls.push_back(probe);
            probe = probe->next;
        }
        swap(ls[k - 1]->val, ls[ls.size() - k]->val);
        return head;
    }
};
