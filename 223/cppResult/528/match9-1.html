/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int sz = 0;
        ListNode* oth = head;
        while(oth) sz++, oth= oth->next;
        int othk = sz-k;
        k--;
        ListNode* oth2 = head;
        oth = head;
        while(k--) oth = oth->next;
        while(othk--) oth2 = oth2->next;
        swap(oth->val, oth2->val);
        return head;
    }
};
