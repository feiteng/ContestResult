/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* l, *r;
        l = r = head;
        for(int i=0; i<k-1; i++){
            r = r->next;
        }
        ListNode* node1 = r;
        while(r->next){
            r = r->next;
            l = l->next;
        }
        ListNode* node2 = l;
        swap(node1->val, node2->val);
        return head;
    }
};
