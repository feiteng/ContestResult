/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int n = 0;
        ListNode *cur = head;
        while (cur != nullptr) {
            cur = cur->next;
            n++;
        }
        int i = 1;
        cur = head;
        ListNode *node1 = nullptr;
        ListNode *node2 = nullptr;
        while (cur != nullptr && (node1 == nullptr || node2 == nullptr)) {
            if (i == k) {
                node1 = cur;
            }
            if (i == n + 1 - k) {
                node2 = cur;
            }
            cur = cur->next;
            i++;
        }
        int tmp = node1->val;
        node1->val = node2->val;
        node2->val = tmp;
        return head;
    }
};
