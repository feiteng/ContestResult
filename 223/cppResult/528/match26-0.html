/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int n = 0;
        auto p = head;
        while (head) {
            n++;
            head = head->next;
        }
        
        int f1 = 0;
        int f2 = 0;
        
        int c = 0;
        head = p;
        while (head) {
            if (c == k-1) f1 = head->val;
            if (c == n - k) f2 = head->val;
            c++;
            head = head->next;
        }
        
        head = p;
        c = 0;
        while (head) {
            if (c == k-1)  head->val = f2;
            if (c == n - k) head->val = f1;
            c++;
            head = head->next;
        }
        
        return p;
    }
};
