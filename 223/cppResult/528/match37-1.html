/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        vector<int> a;
        ListNode* iter;
        
        iter = head;
        while (iter != NULL) {
            a.push_back(iter->val);
            iter = iter->next;
        }
        int n = a.size();
        
        int z = a[k-1]; a[k-1] = a[n-k]; a[n-k]=z;
        

        iter = head;
        int i = 0;
        while (iter != NULL) {
            iter->val = a[i++];
            iter = iter->next;
        }
        return head;
    }
};
