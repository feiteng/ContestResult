/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* p1=head;
        for (int i=0; i<k-1; i++)
            p1=p1->next;
        ListNode* p2=head, *p=p1->next;
        while(p) {
            p=p->next;
            p2=p2->next;
        }
        int val=p1->val;
        p1->val=p2->val;
        p2->val=val;
        return head;
    }
};
