/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* pre=head;
        int ks=k-1;
        while(ks){
            pre=pre->next;
            ks--;
        }
        ListNode* ans1=pre;
        ListNode* last=head;
        while(ans1->next){
            ans1=ans1->next;
            last=last->next;
        }
        int kv=pre->val;
        pre->val=last->val;
        last->val=kv;
        return head;
        
    }
};
