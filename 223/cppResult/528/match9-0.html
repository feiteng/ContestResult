/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int sz = 0;
        ListNode* node = head;
        while(node) sz ++, node = node->next;
        int pos = sz - k;
        ListNode* node2 = head;
        node = head;
        k --;
        while(k--) node = node->next;
        while(pos--) node2 = node2->next;
        swap(node2->val, node->val);
        return head;
    }
};
