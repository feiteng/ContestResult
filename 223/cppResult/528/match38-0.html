/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        int step = 1;
        ListNode* firstk = head;
        while (step < k)
        {
            ++step;
            firstk = firstk->next;
        }
        ListNode* now = firstk->next, *lastk = head;
        while (now != nullptr)
        {
            now = now->next;
            lastk = lastk->next;
        }
        int tmp = firstk->val;
        firstk->val = lastk->val;
        lastk->val = tmp;
        return head;
    }
};
