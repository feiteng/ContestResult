/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapNodes(ListNode* head, int k) {
        ListNode* l = head;
        ListNode* r = head;
        int i = 1;
        while (i < k) {
            l = l->next;
            i++;
        }
        ListNode* l2 = l;
        while (l2->next != nullptr) {
            l2 = l2->next;
            r = r->next;
        }
        swap(l->val, r->val);
        return head;
    }
};
