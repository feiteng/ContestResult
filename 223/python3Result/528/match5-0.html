# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        nums = []
        p = head
        while p:
            nums.append(p)    
            p = p.next
        nums[k - 1].val, nums[-k].val = nums[-k].val, nums[k - 1].val
        
        return head
