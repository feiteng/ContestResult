# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        
        l = []
        curr = head
        while curr:
            l.append(curr)
            curr = curr.next
        
        l[k - 1].val, l[-k].val = l[-k].val, l[k - 1].val
        return head
