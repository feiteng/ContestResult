# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        
        a = head
        for _ in range(k-1):
            a = a.next
        
        temp = a
        b = head
        while temp.next:
            temp = temp.next
            b = b.next
            
        a.val, b.val = b.val, a.val
        return head
