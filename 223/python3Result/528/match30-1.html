# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        h = head
        n = 0
        while h:
            h = h.next
            n += 1
        #print(n)
        kth, _kth = head, head
        for _ in range(1, k):
            kth = kth.next
        
for _ in range(n-k):
            _kth = _kth.next
        kth.val, _kth.val = _kth.val, kth.val
        return head
