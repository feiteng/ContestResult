# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        q = p = head 
        for _ in range(1,k):
            q = q.next 
        firstK = q 
        while q.next!=None:
            p = p.next 
            q = q.next 
        lastK = p 
        firstK.val, lastK.val = lastK.val, firstK.val
        return head
