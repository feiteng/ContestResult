# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        
        d = []
        temp = head
        
        while temp:
            d.append(temp)
            temp = temp.next
        
        d[k-1].val,d[~(k-1)].val = d[~(k-1)].val,d[k-1].val
        return head
