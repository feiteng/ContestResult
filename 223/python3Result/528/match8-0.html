# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        th=head
        l=[]
        while th!=None:
            l.append(th)
            th=th.next
        l[k-1].val,l[-k].val=l[-k].val,l[k-1].val
        return head
