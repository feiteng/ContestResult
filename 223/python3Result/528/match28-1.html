class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        n1 = n2 = head
        for _ in range(k - 1):
            n2 = n2.next

        n3 = n2  # 正数第k个节点

        while n2.next:
            n1 = n1.next
            n2 = n2.next

        n4 = n1  # 倒数第k个节点

        n3.val, n4.val = n4.val, n3.val

        return head
