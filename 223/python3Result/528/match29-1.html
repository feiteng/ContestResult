# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        A = []
        p = head
        while p:
            A.append(p)
            p = p.next
        q = len(A) - k
        A[k-1].val, A[q].val = A[q].val, A[k-1].val
        return head
