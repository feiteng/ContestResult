# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        f = head
        for i in range(k-1):
            f = f.next
        l = f 
        s = head
        while f.next is not None:
            f = f.next
            s = s.next
        s.val, l.val = l.val, s.val
        return head
