# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: ListNode, k: int) -> ListNode:
        nodes = []
        cur = head
        while cur:
            nodes.append(cur)
            cur = cur.next
        n = len(nodes)
        nodes[k-1].val, nodes[n-k].val = nodes[n-k].val, nodes[k-1].val
        return head
