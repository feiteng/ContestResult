class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        n = len(jobs)
        jobs = sorted(jobs, reverse=True)
        workers = [jobs[i] for i in range(k)]
        def f(i):
            if i == n:
                return max(workers)
            workers[0] += jobs[i]
            ans = f(i+1)
            workers[0] -= jobs[i]
            
for j in range(1, k):
                workers[j] += jobs[i]
                ans = min(ans, f(i+1))
                workers[j] -= jobs[i]
            return ans
        return f(k)
