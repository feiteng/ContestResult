class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        if k == 12:
            return max(jobs)
        if k == 11:
            if len(jobs) == 11:
                return max(jobs)
            jobs.sort()
            return max(jobs[0] + jobs[1], jobs[-1])
        # k <= 10
        self.k = k
        self.ans = sum(jobs)
        
        memo = [set() for _ in range(len(jobs))]
        
        
def search(i, t):
            if i == len(jobs):
                self.ans = min(self.ans, max(t))
                return
            t_tuple = tuple(sorted(t))
            if t_tuple[-1] >= self.ans:
                return
            # t_tuple = tuple(t)
            if t_tuple in memo[i]:
                return
            memo[i].add(t_tuple)
            
            for p in range
(self.k):
                t[p] += jobs[i]
                search(i + 1, t)
                t[p] -= jobs[i]
            
        search(0, [0] * k)
        return self.ans
