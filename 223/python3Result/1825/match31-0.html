class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        total = sum(jobs)
        h = total
        l = max(ceil(total / k), max(jobs))
        #print(f'l {l}, h {h}')

        jobs.sort(reverse=True)
        workers_init = []
        for j in jobs[:k]:
            workers_init.append(j)
        
        def dfs(j_idx, workers, t):
            if j_idx == len(jobs):
                return True
            for i in range(k):
                if workers[i] + jobs[j_idx] <= t:
                    workers[i] += jobs[j_idx]
                    if dfs(j_idx + 1, workers, t):
                        return True
                    workers[i] -= jobs[j_idx]
            return False
        
        def ok(t):
            workers = workers_init[:]
            if max(workers) > t:
                return False
            return dfs(k, workers, t)
                    
            
        while l < h:
            m = (l + h) // 2
            if ok(m):
                h = m
            else:
                l = m + 1
            
        return l
