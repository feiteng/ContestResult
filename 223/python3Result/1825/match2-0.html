class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        def backtracking(i, jobs, worker_times):
            if i == len(jobs):
                return max(worker_times)
            
            min_schedule = float("inf")
            for j in range(k):
                worker_times[j] += jobs[i]
                min_schedule = min(min_schedule, backtracking(i + 1, jobs, worker_times))
                worker_times[j] -= jobs[i]
                
            return min_schedule
        
        n = len(jobs)
        jobs.sort()
        
        starting_state = [0] * k
        starting_state = jobs[n - k:]
        
        print(starting_state)
        return backtracking(0, jobs[:n - k], starting_state)
