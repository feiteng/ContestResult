class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        if not jobs or k == 0:
            return 0
        
        self.ans = 0x7fffffff
        jobs.sort(reverse=True)
        times = [jobs[i] for i in range(k)]
        
        def solveByDfs(job):
            if job >= len(jobs):
                self.ans = min(self.ans, max(times))
                return
            
            for j in range(k):
                times[j] += jobs[job]
                solveByDfs(job + 1)
                times[j] -= jobs[job]
                
        solveByDfs(k)
        return self.ans
