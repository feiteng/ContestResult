class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs.sort(reverse=True)
        working_time = [jobs[i] for i in range(k)]
        N = len(jobs)
        def dfs(arr, i):
            if i == N:
                return max(arr)
            ans = 1000000000000
            for j in range(len(arr)):
                arr[j] += jobs[i]
                ans = min(ans, dfs(arr, i+1))
                arr[j] -= jobs[i]
            return ans
        
        return dfs(working_time, k)
