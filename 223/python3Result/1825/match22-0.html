class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        def recur(curr):
            nonlocal assign
            if not curr < len(jobs):
                return max(assign)
            
            res = float('inf')
            for i in range(k):
                assign[i] += jobs[curr]
                res = min(res, recur(curr + 1))
                assign[i] -= jobs[curr]
            return res
        
        jobs.sort(reverse=True)
        assign = [0] * k
        tmp = min(len(jobs), k)
        for i in range(tmp):
            assign[i] = jobs[i]
        return recur(tmp)
