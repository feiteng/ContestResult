class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs.sort()
        self.ans = inf
        
def dfs(cur, jobs, i, k):
            if i == len(jobs):
                self.ans = min(self.ans, max(cur))
                return
            for j in range(0, k):
                cur[j] += jobs[i]
                if cur[j] < self.ans:
                    dfs(cur, jobs, i + 1, k)
                cur[j] -= jobs[i]
                if cur[j] == 0:
                    break
        cur = [0] * k
        dfs(cur, jobs, 0, k)
        return self.ans

        def check(jobs, k, m):
            def dfs(cur, jobs, i, k, m):
                if i == len(jobs):
                    return True
                for j in range(k):
                    cur[j] += jobs[i]
                    if cur[j] <= m:
                        if dfs(cur, jobs, i + 1, k, m):
                            return True
                    cur[j] -= jobs[i]
                return False
            cur = [0] * k
            return dfs(cur, jobs, 0, k, m)
        l, r = 1, sum(jobs)
        while l <= r:
            m = (l + r) // 2
            if check(jobs, k, m):
                ans = m
                r = m - 1
            else:
                l = m + 1
        return ans
