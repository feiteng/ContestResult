class Solution:
    def minimumTimeRequired(self, jobs, k):
        jobs = sorted(jobs, reverse=True)
        a = jobs[:k]
        jobs = jobs[k:]

        def dfs(dep):
            ans = 1000000000
            if dep == len(jobs):
                return max(a)
            for i in range
(k):
                a[i] += jobs[dep]
                ans = min(ans, dfs(dep + 1))
                a[i] -= jobs[dep]
            return ans
        return dfs(0)
