class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        def helper(i, arr):
            if i == len(jobs):
                self.ans = min(self.ans, max(arr))
                return
            
            for j in range(k):
                arr[j] += jobs[i]
                helper(i+1, arr)
                arr[j] -= jobs[i]
            
        
        jobs.sort()
        arr = []
        for i in range(k):
            arr.append(jobs.pop())
        
        self.ans = float('inf')
        helper(0, arr)
        return self.ans
