class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        if k == 1:
            return sum(jobs)
        if k == len(jobs):
            return max(jobs)
        
        jobs.sort(reverse=True)
        num_left = [sum(jobs)]
        
        for j in jobs:
            num_left.append(num_left[-1] - j)
        
        @functools.lru_cache(None)
        def recurse2(l, pos):

            if pos == len(jobs) or l[0] - l[-1] >= num_left[pos]:
                return l[0]
            
            l = list(l)
            ans = float('inf')
            
            
for i in range(len(l)):
                l[i] += jobs[pos]
                ans = min(ans, recurse(l, pos+1))
                l[i] -= jobs[pos]
            
            return ans
        
        
    
        def recurse(l, pos):
            return recurse2(tuple(sorted(l, reverse=True)), pos)
            
        return recurse(tuple([0] * k), 0)
