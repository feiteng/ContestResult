class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs.sort()
        n = len(jobs)

        memo = [0] * (1 << n)
        for i in range(1 << n):
            cur = 0
            for j in range(n):
                if (i >> j) & 1 == 1:
                    cur += jobs[j]
            memo[i] = cur
        
        @functools.lru_cache(None)
        def dp(ind, state):
            if ind == k:
                if state == 0:
                    return 0
                else:
                    return float('inf')
            sub = state
            ans = dp(ind+1, state)
            while sub:
                if memo[sub] < ans:
                    ans = min(ans, max(memo[sub], dp(ind+1, state ^ sub)))
                sub = (sub - 1) & state
            # print(ind, state, ans)
            return ans
        
        return dp(0, (1 << n) - 1)
