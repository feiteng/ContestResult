class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
                   n=len(jobs)
                   jobs.sort(reverse=True)
                   arr=[0]*k
                   for i in range(k):
                        arr[i]=jobs[i]
                   self.ans=float('inf')
                   def dfs(i):
                       if i==n:
                          self.ans=min(self.ans,max(arr))
                          return
                       val=jobs[i]
                       for j in range(k):
                            arr[j]+=val
                            dfs(i+1)
                            arr[j]-=val
                   dfs(k)
                   return self.ans
