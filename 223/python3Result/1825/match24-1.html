class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        time = [0] * k
        answer = math.inf
        
        def recurse(j):
            
            nonlocal answer
            
            if j == len(jobs):
                answer 
= min(answer, max(time))
                return
            
            for i in range(k):
                if time[i] + jobs[j] <= answer:
                    time[i] += jobs[j]
                    recurse(j+1)
                    time[i] -= jobs[j]
                if time[i] == 0:
                    break
        
        recurse(0)
        
        return answer
