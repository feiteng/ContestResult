class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs.sort(reverse=True)
        summ = sum(jobs)
        maxx = max(jobs)
        if k == len(jobs):
            return maxx
        
        load = [0] * k
        for j in range(k):
            load[j] = jobs[j]
        
        def doit(load, i):
            if i == len(jobs):
                return max(load)
            ans = summ
            for j in range(k):
                load[j] += jobs[i]
                ans = min(ans, doit(load, i + 1))
                load[j] -= jobs[i]
            
            return ans
        
        return doit(load, k)
