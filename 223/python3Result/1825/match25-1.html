class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs = sorted(jobs, reverse=True)
        res = jobs[:k]
        ans = float('inf')
        def dfs(i):
            nonlocal ans
            if i == len(jobs):
                ans = min(ans, max
(res))
                return 
            j = jobs[i]
            for m in range(k):
                res[m] += j
                dfs(i+1)
                res[m] -= j
        dfs(k)
        return ans
