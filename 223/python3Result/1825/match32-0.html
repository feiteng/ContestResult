        jobs.sort()
        # print(jobs)
        count=[0]
        # @cache
        def search(idx,state,k=k):
            count[0]+=1
            if idx==len(jobs):
                return max(state)
            if k-len(state)>=len(jobs)-idx:
                return max(state+jobs[idx:])
            # state=list(state)
            if len(state)<k:
                state.append(jobs[idx])
                res=search(idx+1,state)
                state.pop()

                res=math.inf
            for i in range(len(state)):
                state[i]+=jobs[idx]
                res=min(res,search(idx+1,state))
                state[i]-=jobs[idx]
            return res
