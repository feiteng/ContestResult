class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        arr = [0] * k
        n = len(jobs)
        self.ans = math.inf
        def backtrack(i):
            if i == n:
                self.ans = min(self.ans, max(arr))
            if max(arr) >= self.ans:
                return
            visited = set()
            for j in range(k):
                if arr[j] not in visited:
                    visited.add(arr[j])
                    arr[j] += jobs[i]
                    backtrack(i + 1)
                    arr[j] -= jobs[i]
        backtrack(0)
        return self.ans
