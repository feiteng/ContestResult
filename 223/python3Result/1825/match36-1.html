class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        avg = sum(jobs+[k-1])//k
        #print(avg)
        jobs = sorted(jobs)[::-1]
        #print(jobs)
        xs = [0] * k
        for j in jobs:
            mn = min(xs)
            i = xs.index(mn)
            xs[i]+=j
        bound = max(xs)
        
        n = len(jobs)
        
        xs = [0] * k
        ys = [[] for _ in range(k)]
        ys[0] = [jobs[0]]
        xs[0] = jobs[0]
        
def dfs(ji):
            nonlocal bound
            nonlocal ones
            #if bound == avg:
            #    return
            if ji == len(jobs):
                bound = min(max(xs), bound)
                return
            j = jobs[ji]
            #order = sorted(list(range(k)), key=lambda x:xs[x])
            for i in range(ones, k):#order:
                if xs[i] + j < bound:                
                    #ys[i].append(j)
                    xs[i]+=j
                    dfs(ji+1)         
                    #ys[i].append(j)
                    xs[i]-=j
        
        # each k at least one job
        # n-k jobs left
        # if k> n-k, k-(n-k)=2k-n will have only 1 job 
        ones = max(0, k*2-n)
        if k*2-n > 0:
            for i in range(ones):
                xs[i] = jobs[i]
            dfs(ones)
        dfs(1)
                
        #print(xs, ys)
        #return max(xs)
        return bound
