class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:

        def helper(nl, idx):
            if idx == len(jobs):
                return 1
            for i in range(k):
                if nl[i] >= jobs[idx] and (i == 0 or nl[i] != nl[i-1]):
                    nl[i] -= jobs[idx]
                    if helper(nl, idx + 1):
                        return 1
                    else:
                        nl[i] += jobs[idx]
            else:
                return 0

        left = max(sum(jobs) // k, max(jobs))
        right = sum(jobs)

        while left < right:
            mid = (left + right) // 2
            if helper([mid] * k, 0):
                right = mid
            else:
                left = mid + 1
        return left
