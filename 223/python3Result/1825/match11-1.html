class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        def dfs(a, p):
            if p == len(jobs):
                return max(a)
            res = 1 << 31
            for i in range
(k):
                a[i] += jobs[p]
                res = min(dfs(a, p + 1), res)
                a[i] -= jobs[p]
            return res
        
        a = [0] * k
        jobs.sort(reverse=True)
        p = 0
        for i in range(k):
            a[i] = jobs[i]
        return dfs(a, k)
