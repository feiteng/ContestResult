class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        
        def canFit(total, i):
            if i == len(jobs):
                #print("canFit ", total)
                return True
            else:
                for j in range( min( i + 1, len
(time))):
                    if time[j] + jobs[i] <= total:
                        time[j] += jobs[i]
                        if canFit(total, i + 1):
                            #print("canFit ", total)
                            return True
                        time[j] -= jobs[i]
            #print("can not fit ", total)
            return False            
        
        worst_case = sum(jobs)
        
        left , right = 0, worst_case
        while left < right:
            mid = left + (right - left) // 2
            time = [0] * k
            if canFit(mid, 0):
                right = mid
            else:
                left = mid + 1
        return left
