class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        n = len(jobs)
        time = [0] * k
        self.ans = sum(jobs)
        
        def dfs(cur, ind):
            visited = set()
            if ind == n:
                #print(cur)
                self.ans = max(cur)
                return
            for i in range(k):
                if i > 0 and cur[i] == cur[i - 1]:
                    continue
                if cur[i] in visited:
                    continue
                if cur[i] + jobs[ind] <= self
.ans:
                    visited.add(cur[i])
                    cur[i] += jobs[ind]
                    dfs(cur, ind + 1)
                    cur[i] -= jobs[ind]
        dfs(time, 0)
        return self.ans
