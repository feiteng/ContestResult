class Solution:
    # 回溯法
    # 超出时间限制
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        group = [0] * k
        res = sum(jobs)
        
        def helper(i = 0):
            nonlocal res 
            if i == len(jobs):
                res = min(res, max(group))
                return 
        
            if max(group) >= res: #剪枝
                return
        
            for j in range(k):
                group[j] += jobs[i]
                helper(i + 1)
                group[j] -= jobs[i]
            
        helper()
        return res 
    
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        
        res = sum(jobs)
        
        def helper(i = 0, group = []):
            nonlocal res 
            if i == len(jobs):
                res = min(res, max(group))
                return 
            if group and max(group) >= res: #剪枝
                return
            
            if not group:
                group.append(jobs[i])
                helper(i + 1, group)
                group.pop()
                return 
            elif len(group) < k:
                for j in range(len(group)):
                    group[j] += jobs[i]
                    helper(i + 1, group)
                    group[j] -= jobs[i]
                group.append(jobs[i])
                helper(i + 1, group)
                group.pop()
                return 
            else:
                for j in range(len(group)):
                    group[j] += jobs[i]
                    helper(i + 1, group)
                    group[j] -= jobs[i]
                return 
        
        helper()
        return res
