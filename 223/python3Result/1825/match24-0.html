class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        ans = sum(jobs)
        L = len(jobs)
        if k == L:
            return max(jobs)
        
        time = [0 for _ in range(L)]
        
        def dfs(time, index):
            nonlocal ans
            if index == L:
                # print("ans", ans, "max(time)", max(time), "time", time)
                ans = min(ans, max(time))
                return

            for i in range(k):
                if time[i] + jobs[index] <= ans:
                    time[i] += jobs[index]
                    dfs(time, index + 1)
                    time[i] -= jobs[index]
                    if time[i] == 0:
                        break
        
        dfs(time, 0)
        return ans
