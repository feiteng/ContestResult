class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        def get(v, i):
            # print(v,i,d)
            if i == len(jobs):
                return 1
            for j in range(k):
                if j and d[j] == d[j-1]:
                    continue
                if d[j] + jobs[i] <= v:
                    d[j] += jobs[i]
                    if get(v,i+1):
                        d[j] -= jobs[i]
                        return 1
                    d[j] -= jobs[i]
            return 0
        
        jobs.sort(reverse = 1)
        d = [0] * k
        l,r 
= max(jobs), 2 * 10 ** 8
        while l < r:
            m = (l + r) // 2
            # print(m,l,r)
            if get(m, 0):
                r = m
            else:
                l = m + 1
        return l
