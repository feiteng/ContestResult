class DSU:
    def __init__(self, n):
        self.cache = [i for i in range(n)]
    
    def find(self, i):
        if self.cache[i] != i:
            self.cache[i] = self.find(self.cache[i])
        return self.cache[i]
    
    def union(self, i, j):
        self.cache[self.find(i)] = self.cache[self.find(j)]

        
class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        
        dsu = DSU(n)
        for i, j in allowedSwaps:
            dsu.union(i, j)
        
        group = collections.defaultdict(lambda : collections.Counter())
        for i in range(n):
            group[dsu.find(i)][source[i]] += 1
        
        same_count = 0
        for ind in range(n):
            if group[dsu.find(ind)][target[ind]] > 0:
                same_count += 1
                group[dsu.find(ind)][target[ind]] -= 1
        return n - same_count
