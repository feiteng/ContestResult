from collections import defaultdict

class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        
        n = len(source)
        parent = list(range(n))
        
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        for i, j in allowedSwaps:
            rooti, rootj = find(i), find(j)
            if rooti != rootj:
                parent[rooti] = 
rootj
            
        d = defaultdict(set)
        for i in range(n):
            d[find(i)].add(i)
            
        res = 0
        for l in d.values():
            sd, td = defaultdict(int), defaultdict(int)
            for i in l:
                sd
[source[i]] += 1
                td[target[i]] += 1
            
            for num, cnt in sd.items():
                if td[num] < sd[num]:
                    res += sd[num] - td[num]
        return res
