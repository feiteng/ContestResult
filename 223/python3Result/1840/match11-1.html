class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        p = list(range(n))
        def pi(i):
            if p[i] != i:
                p[i] = pi(p[i])
            return p[i]
        def join(i,j):
            a,b = pi(i), pi(j)
            p[b]=a
        for (a,b) in allowedSwaps:
            join
(a,b)
        
        sets = {
        }
        for i in range(n):
            x = pi(i)
            if x not in sets:
                sets[x] = []
            sets[x].append(i)
        def dist(s,t):
            from collections import Counter
            cs = Counter(s)
            ct = Counter(t)
            ret = len(s)
            for k,v in cs.items():
                if k in ct:
                    ret -= min(v,ct[k])
            return  ret
        ret = 0
        for se in sets.values():
            s = [source[i] for i in se]
            t = [target[i] for i in se]
            ret += dist(s,t)
        return ret
