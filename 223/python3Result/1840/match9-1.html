class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        par = list(range(n))
        def find(x: int) -> int:
            res = par[x]
            if res != par[res]:
                res = find(res)
                par[x] = res
            return res
        def join(x: int, y: int):
            par[find(x)] = find(
y)
        for a, b in allowedSwaps:
            join(a, b)
        dic = defaultdict(list)
        for i in range(n):
            dic[find(i)].append(i)
        res = 0
        for val in dic.values():
            count = defaultdict(int)
            for i in val:
                count[source[i]] += 1
            for i in val:
                if target[i] in count and count[target[i]] > 0:
                    count[target[i]] -= 1
                else:
                    res += 1
        return res
