class DisjointSet:
	def __init__(self, n):		
		self.parent = [i for i in range(n)]
		self.rank = [1 for i in range(n)]
	def find_parent(self, x):
		if self.parent[x] == x:
			return x
		
if self.parent[x] != x:
			self.parent[x] = self.find_parent(self.parent[x])
		return self.parent[x]
	def union(self, x, y):
		xset = self.find_parent(x)
		yset = self.find_parent(y)
		if xset == yset:
			return
		if self.rank[xset] > self.rank[yset]:
			self.parent[yset] = xset
		elif self.rank[yset] > self.rank[xset]:
			self.parent[xset] = yset
		else:
			self.parent[yset] = xset
			self.rank[xset] += 1


class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        obj = DisjointSet(len(source))
        for [x, y] in allowedSwaps:
            obj.union(x, y)
        src = defaultdict(list)
        dest = defaultdict(Counter)
        for i in range(len(target)):
            obj.parent[i] = obj.find_parent(i)
            src[obj.parent[i]].append(source[i])
            dest[obj.parent[i]][target[i]] += 1
        sameCount = 0
        for key in src:
            for entry in src[key]:
                if dest[key][entry] > 0:
                    sameCount += 1
                    dest[key][entry] -= 1
        return len(target)-sameCount
