class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        p = [i for i in range(n)]
        def find(x:int):
            if x != p[x]:
                p[x] = find(p[x])
            return p[x]
        def union(x,y):
            px, py = find(x), find(y)
            if px != py:
                p[px] = py
        def helper(s:set):
            n = len(s)
            ca, cb = Counter(), Counter()
            for i in s:
                ca[source[i]] += 1
                cb[target[i]] += 1
            same = 0
            for key in ca:
                if key in cb:
                    same += min(ca[key], cb[key])
            return n - same
            
        for a,b in allowedSwaps:
            union(a,b)
        lookup = defaultdict(set)
        for i in range(n):
            lookup[find(i)].add(i)

        ans = 0
        for g in lookup:
            ans += helper(lookup[g])
        return ans
