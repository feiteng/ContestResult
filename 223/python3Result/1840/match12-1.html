from collections import defaultdict, Counter


def find(arr, i):
    if arr[i] != i:
        arr[i] = find(arr, arr[i])
    return arr[i]

def union(arr, i, j):
    ai, aj = find(arr, i), find(arr, j)
    if ai != aj:
        arr[ai] = aj


class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        data = list(range(len(source)))
        
for i,j in allowedSwaps:
            union(data, i, j)
        
        connected = defaultdict(list) # maps representative to all the indices
        for i,rep in enumerate(data):
            connected[find(data, rep)].append(i)
        # print(data, connected)
        ans = 0
        for _,indices in connected.items
():
            source_count, target_count = Counter(), Counter()
            for i in indices:
                source_count[source[i]] += 1
                target_count[target[i]] += 1
            
            keys = set(source_count.keys()) | set(target_count.keys())
            dist = 0
            for k in keys:
                dist += abs(source_count[k] - target_count[k])
            ans += (dist // 2)
        
        return ans
