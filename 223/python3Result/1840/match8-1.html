class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], swaps: List[List[int]]) -> int:
        n = len(source)
        f = [i for i in range(n)]
        
        def find(x):
            if f[x] != x:
                f[x] = find(f[x])
            return f[x]
        
        def merge(x, y):
            fx = find(x)
            fy = find(y)
            f[fx] =
 fy
        
        for a, b in swaps:
            merge(a, b)
        
        g = collections.defaultdict(list)
        for i in range(n):
            g[find(i)].append(i)
        
        res = 0
        for gg in g:
            cnt1 = collections.Counter([source[i] for i in g[gg]])
            cnt2 = collections.Counter([target[i] for i in g[gg]])
            res += len(g[gg]) - sum(min(cnt1[c], cnt2[c]) for c in cnt1)
        
        return res
