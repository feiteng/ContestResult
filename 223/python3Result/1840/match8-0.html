class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n=len(source)
        
        g=[i for i in range(n)]
        def find(i):
            if g[i]!=i:
                g[i]=find(g[i])
            return g[i]
        def union(i,j):
            gi=find(i)
            gj=find(j)
            g[gi]=g[gj]
            
        for i,j in allowedSwaps:
            union(i,j)
        
        d=collections.defaultdict(set)
        for i in range(n):
            d[find(i)].add(i)
        
        possible_matches=0
        for x in d:
            a=[]
            b=[]
            for i in d[x]:
                a.append(source[i])
                b.append(target[i])
            ca=collections.Counter(a)
            cb=collections.Counter(b)
            for y in ca:
                if y in cb:
                    possible_matches+=min(ca[y],cb[y])
        ans=n-possible_matches
        return ans
