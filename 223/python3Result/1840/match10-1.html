class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        parent = list(range(n))
        
        
def get_parent(i):
            if parent[i] == i:
                return i
            parent[i] = get_parent(parent[i])
            return parent[i]
        
        def union(i, j):
            pi, pj = get_parent(i), get_parent(j)
            parent
[pi] = pj
        
        for i, j in allowedSwaps:
            union(i, j)
        
        groups = defaultdict(list)
        
        for i in range(n):
            groups[get_parent(i)].append(i)
        
        res = 0
        
        for _, indices in groups.items():
            source_count = defaultdict(int)
            for i in indices:
                source_count[source[i]] += 1
            for i in indices:
                if source_count[target[i]] > 0:
                    source_count[target[i]] -= 1
                else:
                    res += 1
        
        return res
