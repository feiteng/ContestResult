class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        tree = list(range(n))

        def find(st):
            if tree[st] != st:
                tree[st] = find(tree[st])
            return tree[st]

        for x, y in allowedSwaps:
            tree[find(x)] = find(y)

        t = [[] for _ in range(n)]
        for i in range(n):
            t[find(i)].append(i)
        ans = 0
        for i in range(n):
            if len(t[i]) > 0:
                s1 = collections.defaultdict(int)
                s2 = collections.defaultdict(int)
                for c in t[i]:
                    s1[source[c]] += 1
                    s2[target[c]] += 1
                for k in s1.keys():
                    if s2[k] >= s1[k]:
                        ans += 0
                    else:
                        ans += s1[k] - s2[k]
        return ans
