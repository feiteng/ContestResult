class DSU:
    def __init__(self, N):
        self.p = list(range(N))
        # p: pointer to parent
        self.rank =
 [0]*N
        # rank: the bigger, more closer to root, more eliment in the following

    def find(self, x):
        if self.p[x] != x:
            self.p[x] = self.find(self.p[x])
        return self.p[x]

    def union(self, x, y):
        xr = self.find(x)
        yr = self.find(y)
        if xr == yr: return
        if self.rank[xr] < self.rank[yr]:
            self.p[xr] = yr
        else:
            self.p[yr] = xr
            if self.rank[xr] == self.rank[yr]:
                self.rank[xr] += 1

class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        N = len(source)
        dsu = DSU(N)
        for swap in allowedSwaps:
            dsu.union(swap[0], swap[1])
        
        group = [0] * N
        for i in range(N):
            p = dsu.find(i)
            if group[p] == 0:
                group[p] = defaultdict(int)
                group
[p][source[i]] += 1
            else:
                group[p][source[i]] += 1
        
        res = 0
        #print(group)
        for i in range(N):
            p = dsu.find(i)
            d = group[p]
            if target[i] in d and d[target[i]] > 0:
                d[target[i]] -= 1
            else:
                res += 1
    
        return res
