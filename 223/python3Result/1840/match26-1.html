class DSU:
    def __init__(self, n):
        self.p = list(range
(n))
    def find(self, x):
        if x!=self.p[x]:
            self.p[x] = self.find(self.p[x])
        return self.p[x]
    def union(self, x, y):
        self.p[self.find(x)] = self.find
(y)
        
        
class Solution:
    def minimumHammingDistance(self, s: List[int], t: List[int], a: List[List[int]]) -> int:
        n = len(s)
        dsu = DSU(n)
        for i, j in a:
            dsu.union(i, j)
        d = collections.defaultdict(list)
        for i in range(n):
            d[dsu.find(i)].append(i)
        res = 0
        for _, l in d.items():
            dt = collections.defaultdict(int)
            for i in l:
                dt[s[i]] += 1
            for i in l:
                if dt[t[i]]==0:
                    res += 1
                else:
                    dt[t[i]] -= 1
        return res
