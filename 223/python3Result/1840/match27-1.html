class DSU:
    def __init__(self, n):
        self.uf = {i: i for i in range(n)}
        self.rank = {i: 1 for i in range
(n)}

        if self.uf[x] != x:
            self.uf[x] = self.find(self.uf[x])
        return self.uf[x]

class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        dsu = DSU(n)
        for a, b in allowedSwaps:
            dsu.union(a, b)
        dic = defaultdict
(list)
        for i in range(n):
            dic[dsu.find(i)].append(i)
        ans = 0
        for v in dic.values():
            cnt1, cnt2 = Counter(), Counter()
            for i in v:
                cnt1[source[i]] += 1
                cnt2[target[i]] += 1
            cur = list((cnt1 & cnt2).elements())
            ans += len(cur)
        return n - ans
