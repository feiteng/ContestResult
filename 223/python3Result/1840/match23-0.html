class DSU:
    def __init__(self, n):
        self.roots = [i for i in range(n)]
        self.size = [1] * n
    
    def find_root(self, x):
        if self.roots[x] != x:
            self.roots[x] = self.find_root(self.roots[x])
        return self.roots[x]

    def union(self, x, y):
        x_root, y_root = self.find_root(x), self.find_root(y)
        if x_root != y_root:
            if self.size[x_root] > self.size[y_root]:
                self.roots[y_root] = x_root
            else:
                self.roots[x_root] = y_root
                self.size[y_root] += self.size[x_root] == self.size[y_root]
        

class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        dsu = DSU(n)
        
        for x, y in allowedSwaps:
            dsu.union(x, y)
        
        res = 0
        
        # indicies with same roots
        groups = collections.defaultdict(collections.Counter)
        for i in range(n):
            groups[dsu.find_root(i)][source[i]] += 1
            
        for i in range(n):
            groups[dsu.find_root(i)][target[i]] -= 1
            if groups[dsu.find_root(i)][target[i]] < 0:
                res += 1
                
        return res
