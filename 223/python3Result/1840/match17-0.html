class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        # 取哪几个元素联通
        n = len(source)
        d = list(range(n))
        
        def find(x):
            if d[x]!=x:
                d[x]=find(d[x])
            return d[x]
        
        
        def union(x,y):
            u,v = find(x),find(y)
            if u!=v:
                d[u]=v
        for x,y in allowedSwaps:
            union(x,y)
        s,t = defaultdict(Counter),defaultdict(Counter)
        for i in range(n):
            temp = find(i)
            s[temp][source[i]]+=1
            t[temp][target[i]]+=1
        print(s,t)
        res = 0
        for v in s:
            a,b = s[v],t[v]
            res+=sum((a-b).values())
        return res
