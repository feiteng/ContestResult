import collections
from typing import List



class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        N = len(source)

        djs = list(range(N))

        def find(i: int):
            if djs[i] != i:
                djs[i] = find(djs[i])
            return djs[
i]

        def merge(i: int, j: int):
            pi = find(i)
            pj = find(j)
            djs[pi] = pj

        for i, j in allowedSwaps:
            merge(i, j)

        groups = collections.defaultdict(list)
        for i in range(N):
            gid = find(i)
            groups[gid].append(i)

        hammingDistance = 0
        for group in groups.values():
            counter = collections
.Counter()
            for i in group:
                counter[source[i]] += 1
                counter[target[i]] -= 1
            hammingDistance += sum(map(abs, counter.values())) // 2
        return hammingDistance
