class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        
        n = len(source)
        parent = list(range(n))
        
        def find(i):
            if parent[i]!=i:
                parent[i] = find(parent[i])
            return parent[i]
        
        for i, j in allowedSwaps:
            pi = find(i)
            pj = find(j)
            if pi != pj:
                parent[pi] = pj
        
        groups = collections.defaultdict(list)
        v = [False] * n
        
        for i in range(n):
            groups[find(i)].append(i)
        
        ans = 0
        for g, ind in groups.items():
            arr1 = collections.Counter()
            arr2 = []
            for i in ind:
                arr1[source[i]] += 1
                arr2.append(target[i])
                v[i] = True
            
            for num in arr2:
                if arr1[num] > 0:
                    arr1[num] -= 1
                    continue
                ans += 1 
        for i in range(n):
            if not v[i]:
                ans += (source[i] != target[i])
        return ans
