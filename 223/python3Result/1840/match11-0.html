from collections import Counter
class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]])   -> int:
        n = len(source)
        eqv = list(range(n))
        def ref(i):
            if eqv[i] != i:
                eqv[i] = ref(eqv[i])
            return eqv[i]
        def unify(i, j):
            ii, jj = ref(i), ref(j)
            eqv[ii] = jj
        for i, j in allowedSwaps:
            unify(i, j)
        groups = {}
        for i in range(n):
            ii = ref(i)
            if ii not in groups:
                groups[ii] = []
            groups[ii].append(i)
        #print(groups)
        ans = 0
        for i in groups:
            sc = Counter(source[j] for j in groups[i])
            tc = Counter(target[j] for j in groups[i])
            #print('groups[i]', groups[i], 'sc', sc, 'tc', tc)
            for v in sc:
                ans += sc[v] - min(sc[v], tc[v])
        return ans
