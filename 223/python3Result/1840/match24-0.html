class DSU: 
    def __init__(self, N): 
        self.par = list(range(N)) 
        self.sz = [1] * N 
    def find(self, x): 
        if self.par[x] != x: 
            self.par[x] = self.find(self.par[x]) 
        return self.par[x] 
    def union(self, x, y): 
        xr, yr = self.find(x), self.find(y) 
        if xr == yr: 
            return False 
        if self.sz[xr] < self.sz[yr]: 
            xr, yr = yr, xr 
        #print(self.par[yr])
        self.par[yr] = xr 
        #print(yr, self.par[yr])
        self.sz[xr] += self.sz[yr] 
        self.sz[yr] = self.sz[xr] 
        return True 
    def size(self, x): 
        return self.sz[self.find(x)]
                            
class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        dsu = DSU(n)
        for s in allowedSwaps:
            dsu.union(s[0], s[1])
        ans = 0
        partition1 = {}
        partition2 = {}
        #print(dsu.par)
        for i in range(n):
            idx = dsu.find(i)
            if idx not in partition1:
                partition1[idx] = {}
                partition2[idx] = {}
                partition1[idx] = defaultdict(int)
                partition1[idx][source[i]] += 1
                partition2[idx] = defaultdict(int)
                partition2[idx][target[i]] += 1
            else:
                partition1[idx][source[i]] += 1
                partition2[idx][target[i]] += 1
        #print(partition1, partition2)      
        for idx in partition2:
            for x in partition2[idx]:
                ans += max(0, partition2[idx][x] - partition1[idx][x])
        return ans
