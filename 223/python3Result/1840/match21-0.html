class UF:
    def __init__(self,n):
        self.p=list(range(n))
        self.s=[1]*n
    
    def find(self,x):
        if self.p[x]!=x:
            self.p[x]=self.find(self.p[x])
        return self.p[x]

    def union(self,x,y):
        xr,yr=self.find(x),self.find(y)
        if xr==yr:
            return
        if self.s[xr]<self.s[yr]:
            xr,yr=yr,xr
        self.p[yr]=xr
        self.s[xr]+=self.s[yr]
        
class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n=len(source)
        g=UF(n)
        for a,b in allowedSwaps:
            g.union(a,b)
        
        d={}
        for i in range(n):
            c=target[i]
            root=g.find(i)
            if root not in d:
                d[root]=defaultdict(int)
            d[root][c]+=1
        
        for i in range(n):
            c=source[i]
            root=g.find(i)
            if d[root][c]>0:
                d[root][c]-=1
        
        ans=0
        for k in d:
            ans+=sum(d[k].values())
        return ans
