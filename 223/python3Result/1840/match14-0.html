class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        N = len(source)
        UF = list(range(N))
        def find(i):
            while i != UF[i]:
                i = UF[i]
            return i
        def union(a, b):
            pa = find(a)
            pb = find(b)
            UF[pa] = pb
        for a, b in allowedSwaps:
            union(a, b)
        
        cc = collections.defaultdict(set)
        for i in range(N):
            cc[find(i)].add(i)
        
        res = 0
        for s in cc.values():
            src, tgt = collections.Counter(), collections.Counter()
            for i in s:
                src[source[i]] += 1
                tgt[target[i]] += 1
            res += sum((src - tgt).values())
        return res
