class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], tups: List[List[int]]) -> int:
        n = len(source)
        
        tmp = [tuple(sorted(t)) for t in tups]
        tups = set(tmp)
        root = [i for i in range(n)]
        
        def find(x):
            if root[x]!=x:
                root[x] = find(root[x])
            return root[x]
        
        def union(a,b):
            ra,rb = find(a),find(b)
            if ra!=rb:
                root[ra]=rb
                
                
        for s,e in tups:
            #print([s,e],root[s],root[e],root)
            union(s,e)
            
        dic = collections.defaultdict(set)
        for i in range(n):
            dic[find(i)].add(i)
        #print(dic)
        res = 0
        for i,val in dic.items():
            tmp = 0
            sets = collections.Counter([source[v] for v in val])
            sett = collections.Counter([target[v] for v in val])
            for k,v in sets.items():
                tmp+=min(v,sett[k])
            res += 2*len(val)-2*tmp
        return res//2
