class DSU:
    def __init__(self, n):
        self.parents = list(range(n))
        self.cc = n
    
    def find(self, u):
        orig = u
        while u!=self.parents[u]:
            u = self.parents[u]
        
        parent,u=u,orig
        
        while u!=self.parents[u]:
            u,self.parents[u] = self.parents[u],parent
        
        return parent
    
    def union(self, u, v):
        u,v = sorted([self.find(u),self.find(v)])
        if u!=v:
            self.parents[v] = u
            self.cc-=
1
            
    
class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        
        n = len(source)
        
        dsu = DSU(n)
        
        for u,v in allowedSwaps:
            dsu.union(u,v)
        
        groups = collections.defaultdict
(set)
        
        for idx in range(n):
            groups[dsu.find(idx)].add(idx)
        
        #print(groups)
        res = 0
        for k,group in groups.items():
            cc = collections.Counter
()
            for i in group:
                cc[source[i]]+=1
                cc[target[i]]-=1
            res += sum(v for v in cc.values() if v>0)
        return res
