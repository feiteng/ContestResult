class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        ids = list(range(n))
        def find(i):
            if ids[i] != i:
                ids[i] = find(ids[i])
            return ids[i]
        def union(i, j):
            i, j = find(i), find(j)
            if i != j:
                ids[j] = i
        for i, j in allowedSwaps:
            union(i, j)
        group = collections.defaultdict(list)
        for i in range(n):
            group[find(i)].append(i)
        ans = 0
        for g in group.values():
            c1 = collections.Counter(source[i] for i in g)
            c2 = collections.Counter(target[i] for i in g)
            c = c1 - c2
            ans += sum(c.values())
        return ans
