from collections import defaultdict, Counter

class DSU:
    def __init__(self, n):
        self.p = list(range(n))
    def find(self, x):
        if x != self.p[x]:
            self.p[x] = self.find(self.p[x])
        return self.p[x]
    def union(self, x, y):
        xp = self.find(x)
        yp = self.find(y)
        self.p[xp] = yp
