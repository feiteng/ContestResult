class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n=len(source)
        g=[-1]*n
        def find(x):
            if g[x]==-1:
                return x
            g[x]=find(g[x])
            return g[x]
        def join(x,y):
            x,y=find(x),find(y)
            if x==y:
                return
            g[x]=y
        for a,b in allowedSwaps:
            join(a,b)
        #print('g:%s'%g)                        
        d=defaultdict(list)            
        for x in range(n):
            d[find(x)].append(x)
        match=0
        for k, v in d.items():
            src=[source[x] for x in v]            
            dst=[target[x] for x in v]            
            cntSrc=Counter(src)
            cntDst=Counter(dst)
            for k1,v1 in cntSrc.items():
                match+=min(v1,cntDst[k1])
            #print("v:%s,src:%s,dst:%s"%(v,src,dst))
        return n-match
