class Solution:
  def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
    n = len(source)
    uf = UF(n)
    for p1, p2 in allowedSwaps:
      uf.union(p1, p2)
    ccs = defaultdict(list)
    for i in range(n):
      ccs[uf.find(i)].append(i)
    res = 0
    for cc, l in ccs.items():
      vals = Counter([source[k] for k in l])
      for t in [target[k] for k in l]:
        if vals[t] > 0:
          vals[t] -= 1
        else:
          res += 1
    return res     
        
        
        
        
        
class UF:
  def __init__(self, n):
    self.parent, self.count = [i for i in range(n)], n*[1]
  def find(self, i):
    root = self.parent[i]
    while root != self.parent[root]: root = self.parent[root]
    while self.parent[i] != root: i, self.parent[i] = self.parent[
i], root
    return root
  def union(self, i, j):
    root_i = self.find(i)
    root_j = self.find(j)
    if root_i == root_j: return
    if self.count[root_i] < self.count[root_j]: 
      self.parent[root_i] = root_j
      self.count[root_j] += self.count[root_i]
    else:
      self.parent[root_j] = root_i
      self.count[root_i] += self.count[root_j]
