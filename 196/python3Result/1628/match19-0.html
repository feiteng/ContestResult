class Solution:
    def numSubmat(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        def findPrefixCount(p_mat, mat):
            for i in range(0, m):  
                for j in range(n - 1, -1, -1):  
                    if not mat[i][j]:  
                        continue
                    if j != n - 1:  
                        p_mat[i][j] += p_mat[i][j + 1]  

                    p_mat[i][j] += mat[i][j]  
        p_mat = [[0 for i in range(n)] for j in range(m)]  

        findPrefixCount(p_mat, mat) 
        res = 0
        for j in range(0, n):  
            i = m - 1
            q = []
            to_sum = 0
            while i >= 0:
                c = 0
                while len(q) != 0 and q[-1][0] > p_mat[i][j]:  
                    to_sum -= (q[-1][1] + 1)*(q[-1][0] - p_mat[i][j])  
                    c += q[-1][1] + 1
                    q.pop()  
                to_sum += p_mat[i][j]  
                res += to_sum  
                q.append((p_mat[i][j], c))  
                i -= 1
        return res
