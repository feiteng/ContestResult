class Solution:
    def numSubmat(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len
(mat[0])
        pre = [[0 for j in range(n)] for i in range(m)]
        for i in range(m):
            for j in range(n - 1, -1, -1):
                if mat[i][j] == 0:
                    continue
                if j != n - 1:
                    pre[i][j] += pre[i][j + 1]
                pre[i][j] += mat[i][j]
        ans = 0
        for j in range(n):
            i = m - 1
            q = []
            s = 0
            while i >= 0:
                c = 0
                while len(q) != 0 and q[-1][0] > pre[i][j]:
                    s -= (q[-1][1] + 1) * (q[-1][0] - pre[i][j])
                    c += q[-1][1] + 1
                    q.pop()
                s += pre[i][j]
                ans += s
                q.append((pre[i][j], c))
                i -= 1
        return ans
