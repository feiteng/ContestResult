class Solution:
    def numSubmat(self, arr: List[List[int]]) -> int:
        n = len(arr)
        m = len
(arr[0])
        
        def findPrefixCount(p_arr, arr, n, m):  
            for i in range(0, n):   
                for j in range(m - 1, -1, -1):  
                    if not arr[i][j]:  
                        continue
                    if j != m - 1:  
                        p_arr[i][j] += p_arr[i][j + 1]  
                    p_arr[i][j] += arr[i][j]  
        
        
        p_arr = [[0 for i in range(m)] for j in range(n)]  
        findPrefixCount(p_arr, arr, n, m)    
        ans = 0
        for j in range(0, m):  
            i = n - 1
            q = []   
            to_sum = 0
            while i >= 0:  
                c = 0
                while len(q) != 0 and q[-1][0] > p_arr[i][j]:  
                    to_sum -= (q[-1][1] + 1) * (q[-1][0] - p_arr[i][j])  
                    c += q[-1][1] + 1
                    q.pop()  
                to_sum += p_arr[i][j]  
                ans += to_sum  
                q.append((p_arr[i][j], c))  
                i -= 1

        return ans
