class Solution:
    def numSubmat(self, mat: List[List[int]]) -> int:
        m = len(mat)
        n = len(mat[0])
        ones_count = [[0 for i in range(n + 1)] for _ in range(m)]
        
        for i in range(m):
            for j in range(n - 1, -1, -1):
                if mat[i][j]:
                    ones_count[i][j] = ones_count[i][j
 + 1] + 1
                    
        res = 0
        for j in range(n):
            i = m - 1
            stack = []
            
            subs = 0
            while i >= 0:
                temp = 0
                while len(stack) and stack[-1][0] > ones_count[i][j]:
                    subs -= (stack[-1][1] + 1) * (stack[-1][0] - ones_count[i][j])
                    temp += stack[-1][1] + 1
                    stack.pop()
                subs += ones_count[i][j]
                res += subs
                stack.append((ones_count[i][j], temp))
                i -= 1
        return res
