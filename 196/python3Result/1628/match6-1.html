class Solution:
    def numSubmat(self, mat: List[List
[int]]) -> int:
        def findPrefixCount(p_arr, arr):  
            for i in range(0, m):  
                for j in range(n - 1, -1, -1):  
                    if not arr[i][j]:  
                        continue
                    if j != n - 1:  
                        p_arr[i][j] += p_arr[i][j + 1]  
                    p_arr[i][j] += arr[i][j]  
        m, n = len(mat), len(mat[0])
        p_arr 
= [[0 for i in range(n)] for j in range(m)]  
        findPrefixCount(p_arr, mat)  
        ans  = 0
        for j in range(0, n):  
            i = m - 1
            q = []  
            to_sum = 0
            while i >= 0:  
                c = 0
                while len(q) != 0 and q[-1][0] > p_arr[i][j]:  
                    to_sum -= (q[-1][1] + 1) * (q[-1][0] - p_arr[i][j])   
                    c += q[-1][1] + 1
                    q.pop()   
                to_sum += p_arr[i][j]  
                ans += to_sum  
                q.append((p_arr[i][j], c))  
                i -= 1 
        return ans
