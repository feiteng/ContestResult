class Solution:
    def numSubmat(self, mat: List[List[int]]) -> int:
        rows = len(mat)
        col = len(mat[0])
        tmat = [[0 for i in range(col)] for j in range(rows)]
        for i in range(rows):
            for j in range(col-1, -1, -1):
                if mat[i][j] != 1:
                    continue
                if j != col - 1:
                    tmat[i][j] += tmat[i][j+1]
                tmat[i][j] += mat[i][j]
        sol = 0
        for j in range(col):
            i = rows - 1
            temp = []
            t = 0
            while i > -1:
                count = 0
                while len(temp) != 0 and temp[-1][0] > tmat[i][j]:
                    t -= (temp[-1][1] + 1) * (temp[-1][0] - tmat[i][j])
                    count += temp[-1][1] + 1
                    temp.pop(-1)
                t += tmat[i][j]
                sol += t
                temp.append((tmat[i][j], count))
                i -= 1
        return sol
