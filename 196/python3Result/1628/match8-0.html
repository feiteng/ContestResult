class Solution:
    def numSubmat(self, A: List[List[int]]) -> int:
        
        def find(pre, A):
            for i in range(m):
                for j in range(n)[::-1]:
                    if A[i][j] == 0:
                        continue
                    if j != n-1:
                        pre[i][j] += pre[i][j+1]
                    pre[i][j] += A[i][j]
        m, n = len(A), len(A[0])
        pre = [[0 for i in range(n)] for j in range(m)]
        
        find(pre, A)
        
        res = 0
        for j in range(n):
            i = m-1
            q = []
            sm = 0
            while i >= 0:
                c = 0
                while len(q) and q[-1][0] > pre[i][j]:
                    sm -= (q[-1][1] + 1) * (q[-1][0] - pre[i][j])
                    
                    c += q[-1][1] + 1
                    
                    q.pop()
                    
                sm += pre[i][j]
                res += sm
                
                q.append((pre[i][j], c))
                i -= 1
        return res
