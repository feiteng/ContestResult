class Solution:
    def numSubmat(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        p = [[0 for i in range(n)] for j in range(m)]  
    
        for i in range(m):  
            for j in range(n - 1, -1, -1):  
                if not mat[i][j]:  
                    continue

                if j != n - 1:  
                    p[i][j] += p[i][j + 1]  

                p[i][j] += mat[i][j]
        
        
        # print(p)
        res = 0

        for j in range(0, n):  
            i = m - 1
            s = []  
            cur = 0

            while i >= 0:  
                c = 0
                while len(s) and s
[-1][0] > p[i][j]:  
                    cur -= (s[-1][1] + 1) * (s[-1][0] - p[i][j])  
                    c += s[-1][1] + 1
                    s
.pop()
                
                # print(i, j)
                cur += p[i][j]  
                res += cur  

                s.append((p[i][j], c))  
                i -= 1

        return res
