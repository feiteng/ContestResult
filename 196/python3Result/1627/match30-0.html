import heapq
import collections
import functools
import math
from typing import List


class Solution:
    def getLastMoment(self, n: int, left: List[int], right: List[int]) -> int:
        ans = 0
        for elem in left:
            ans = max(ans, elem)
        for elem in right:
            ans = max(ans, n - elem)
        return ans
