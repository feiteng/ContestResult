class Solution:
    def getLastMoment(self, n: int, left: List[int], right: List[int]) -> int:
        left.sort()
        right.sort()
        
        time = 0
        if left:
            time = max(time, left[-1])
        if right:
            time = max(time, n-right[0])
        return time
