class Solution:
    def minInteger(self, num: str, k: int) -> str:
        num = list(num)
        
        n = len(num)
        
        if k >= n * (n + 1) // 2:
            return ''.join(sorted(num))
        
        i = 0
        
        while i < n - 1 and k > 0:
            cur 
= i
            
            for j in range(i+1, n):
                if k < j - i:
                    break
                
                if num[j] < num[cur]:
                    cur = j
            
            num[i:cur+1] = [num[cur]] + num[i:cur]
            
            k -= cur - i
            i += 1
            # print(i, cur, k)
        
        return ''.join(num)
