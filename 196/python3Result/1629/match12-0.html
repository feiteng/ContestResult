class Solution:
    def minInteger(self, num: str, k: int) -> str:
        if k >= len(num)*(len(num)-1)//2: return "".join(sorted(num))
        
        num = list(num)
        i = 0
        while k and i < len(num):
            mn = num[i]
            ii = i
            for j in range(i+1, min(len(num), i+k+1)):
                if num[j] < mn: 
                    ii = j
                    mn = num[j]
            if ii != i: 
                k -= ii-i
                for j in range(ii, i, -1):
                    num[j-1], num[j] = num[j], num[j-1]
            i += 1
        return "".join(num)
