class Solution:
    def minInteger(self, num: str, k: int) -> str:
        if k >= len(num)*(len(num)+1)//2:
            return "".join(sorted(list(num)))
        def dfs(num: List[num], k: int):
            if k == 0 or len(num) == 1:
                return num
            min_idx = min(range(min(k+1, len(num))), key=lambda x: num[x])

            res = dfs(num[:min_idx] + num[min_idx+1:], k-min_idx)
            return [num[min_idx]] + res
        return "".join(dfs(list(num), k))
