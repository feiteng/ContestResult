class Solution:
    def minInteger(self, arr: str, k: int) -> str:
        if k>=(len(arr)+1)*len(arr)/2:
            return ''.join(sorted(arr))
        arr = list(arr)
        def swap(A, target, cur):
            tmp = '99'
            for i in range(cur,target, -1):
                tmp = A[i-1]
                arr[i-1] = arr[i]
                arr[i] = tmp
        if k==0:
            return ''.join(arr)
        n = len(arr)
        for i in range(n):
            max_idx = 0
            min_ = ('99')
            if (k+i)>n:
                limit = n
            else:
                limit = k+i+1
            for j in range(i, min(limit,n)):
                if arr[j]<min_:
                    min_ = arr[j]
                    max_idx = j
            k-=(max_idx-i)
            swap(arr, i, max_idx)
            if k==0:
                break
        return ''.join(arr)
