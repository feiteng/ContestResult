class Solution:
    def minInteger(self, num: str, k: int) -> str:
        # https://blog.csdn.net/qq_28468707/article/details/104392619
        def isbig(i, array, K):


            if (K + 1) < len(array):
                m=K+1
            else:
                m=len(array)
            if array[i] == min(array[0:m]):
                return True
            else:
                return False

        def new_trans_k(array, index_max, k):
            for i in range(k):
                t = array[index_max]
                array[index_max] = array[index_max - 1]
                array[index_max - 1] = t
                index_max -= 1
            return array

        def trans(array, K):
            for i in range(len(array)):
                if isbig(i, array, K):
                    continue
                else:
                    if (i + K)<len(array):
                        m=i + K + 1
                    else:
                        m=len(array)
                    max_num = min(array[i:m])
                    index_max = array[i:m].index(max_num) + i
                    
                    c = index_max - i
                    K = K - c
                    array = new_trans_k(array, index_max, c)
                    if K==0:
                        break
            return array
        
        n = len(num)
        array = [0] * n
        for i, c in enumerate(num):
            array[i] = int(c)
            
        if k >= n*(n-1):
            return "".join(map(str, sorted(array)))
        
        return "".join(map(str, trans(array, k)))
