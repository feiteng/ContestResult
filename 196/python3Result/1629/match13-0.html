class Solution:
    def minInteger(self, num: str, k: int) -> str:
        n = len(num)
        if k>=n*(n-1):
            return "".join(sorted(num))
        
        def small(ls, i, k):
            m = min(k+1, len(ls))
            return ls[i] == min(ls[:m])
        
        def tran(ls, midx, c):
            for i in range(c):
                t = ls[midx]
                ls[midx] = ls[midx - 1]
                ls[midx - 1] = t
                midx -= 1
            return ls
        
        ls = list(num)
                
        for i, c in enumerate(ls):
            if not small(ls, i, k):
                m = min(i+k+1, len(ls))
                mc = min(ls[i:m])
                midx = ls[i:m].index(mc) + i
                k -= (midx - i)
                ls = tran(ls, midx, midx-i)
                # ls = ls[:i] + ls[midx:midx+1] + ls[i:midx] + ls[midx:]
                if k == 0:
                    break
        return ''.join(ls)
