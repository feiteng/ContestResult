class Solution:
    
    def minInteger(self, num: str, k: int) -> str:
        def ismin(i, array, K):
            if (K + 1) < len(array):
                m=K+1
            else:
                m=len(array)
            if array[i] == min(array[0:m]):
                return True
            else:
                return False

        def new_trans_k(array, index_min, k):
            for i in range(k):
                t = array[index_min]
                array[index_min] = array[index_min - 1]
                array[index_min - 1] = t
                index_min -= 1
            return array



        def trans(array, K):
            for i in range(len(array)):
                if ismin(i, array, K):
                    continue
                else:
                    if (i + K)<len(array):
                        m=i + K + 1
                    else:
                        m=len(array)
                    m_num = min(array[i:m])
                    index_min = array[i:m].index(m_num) + i
                    c = index_min - i
                    K = K - c
                    array = new_trans_k(array, index_min, c)
                    if K==0:
                        break
            return array
    
        n = len(num)
        if k >= n*(n-1):
            return "".join(sorted(num))

        return "".join(trans(list(num), k))
