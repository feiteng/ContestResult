class Solution:
    def minInteger(self, num: str, k: int) -> str:
        
        idxs = [deque() for _ in range(10)]
        for i, digit in enumerate(num):
            idxs[ord(digit) - ord('0')].append(i)
            
        def ascending():
            max_idx = 0
            for l in idxs:
                if l and max_idx > l[0]: return False
                if l: max_idx = l[-1]
            return True
        
        ans = ''
        seen = []
        while k and not ascending():
            for i, l in enumerate(idxs):
                if l:
                    idx = bisect.bisect_left(seen, l[0])
                    if l[0] - idx <= k:
                        ans += chr(ord('0') + i)
                        bisect.insort(seen, l[0])
                        k -= l[0] - idx
                        idxs[i].popleft()
                        break
                        
        remain = [(idx, i) for i, l in enumerate(idxs) for idx in l]
        return ans + ''.join([chr(ord('0') + i) for _, i in sorted(remain)])
