class Solution {
public:
    string minInteger(string s, int k) {
        int n=s.size();
        if (k>=n*(n-1)/2) {
            string o(s);
            sort(o.begin(),o.end());
            return o;
        }
        int pre=0;
        while (k) {
            if (k>=(n-pre)*(n-pre-1)/2) {
                string a(s.begin(),s.begin()+pre);
                string b(s.begin()+pre,s.begin()+n);
                sort(b.begin(),b.end());
                return a+b;
            }
            int x=min_element(s.begin()+pre,s.begin()+min(pre+k+1,n))-s.begin();
            char c=s[x];
            for (int i=x;i>pre;i--) {
                s[i]=s[i-1];
            }
            //printf("%d %d %d\n",pre,x,k);
            k-=x-pre;
            s[pre++]=c;
            //cout<<s<<endl;
        }
        return s;
    }
};
