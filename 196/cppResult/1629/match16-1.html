class Solution {
public:
    string minInteger(string num, int k) {
        int n = num.size();
        if (k >= n * (n - 1) / 2) {
            sort(num.begin(), num.end());
            return num;
        }
        int pos = 0;
        while (pos < n && k) {
            int t = pos;
            for (int i = pos; i <= min(pos + k, n - 1); ++i)
                if (num[i] < num[t])
                    t = i;
            for (int idx = t; idx > pos; --idx)
                swap(num[idx], num[idx - 1]);
            k -= t - pos;
            ++pos;
            // cout << k << " " << num << endl;
        }
        return num;
    }
};
