class Solution {
public:
    

    
    int read(int idx, const vector<int>& tree){
        int sum = 0;
        while (idx < tree.size()){
            sum += tree[idx];
            idx += (idx & -idx);
        }
        return sum;
    }

    void update(int idx , vector<int>* tree){
        while (idx > 0){
            (*tree)[idx] ++;
            idx -= (idx & -idx);
        }

    }    
    
    string minInteger(string num, int k) {
        int N=num.size();
        vector<int> poss[10];
        vector<int> curr(10, 0);
        for(int i=0;i<N;i++) {
            poss[num[i]-'0'].push_back(i);
        }
        vector<bool> done(N, false);
        string ans="";
        
        vector<int> tree(N+2, 0);
        

        for(int i=0;i<N;i++) {
            
            for(int d=0;d<=9;d++) {
                if (poss[d].size() == curr[d]) continue;
                int reqd = poss[d][curr[d]];
                int ct = poss[d][curr[d]]-i;
                                
                ct = poss[d][curr[d]]-i+read(reqd+1, tree);
                if (ct <= k) {
                    update(reqd+1, &tree);
                    curr[d]++;
                    k-=ct;
                    ans.push_back(d+'0');
                    break;
                }
            }
        }
        return ans;
    }
};
