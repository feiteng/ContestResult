class Solution {
public:
    string minInteger(string num, int k) {
      string v = num;
      int n = num.size();
      if (k >= (n * (n + 1) / 2)) {
        sort(num.begin(), num.end());
        return num;
      }
      int index = 0;
      while (index < n and k > 0) {
        int minIndex = index;
        for (int i = index; i < min(n, index + k + 1); ++i) {
          if (v[i] < v[minIndex])
            minIndex = i;
        }
        int i = minIndex - 1;
        while (i >= index and k) {
          swap(v[i], v[i+1]);
          i--;
          k--;
        }
        index++;
      }
      return v;
    }
};
