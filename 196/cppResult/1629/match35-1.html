class Solution {
public:
    string minInteger(string num, int k) {
        int len = num.length();
        if(k >= len*(len-1)/2){
            sort(num.begin(), num.end());
            return num;
        }
        int loc = 0;
        while(k > 0){
            if(loc >= len-1) break;

            int end = min(loc+k+1, len);
            int tmp = loc;
            for(int i=loc+1; i<end; i++){
                if(num[i] < num[tmp]){
                    tmp = i;
                    if(num[i] == '0') break;
                }
            }
            if(tmp != loc){
                int curnum = num[tmp];
                for(int i=tmp; i>loc; i--){
                    num[i] = num[i-1];
                }
                num[loc] = curnum;
            }
            k -= (tmp - loc);
            loc++;
        }
        return num;
    }
};
