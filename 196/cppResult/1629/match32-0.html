class Solution {
public:
    string minInteger(string num, int k) {
        if (k >= 1000000000) {
            sort(num.begin(), num.end());
            return num;
        }
        for (int i = 0; i < num.size(); i++) { 
            int min_index = i, min = INT_MAX;
            int limit = (k + i) > num.size() - 1 ? num.size() - 1 : k + i;
            for (int j = i; j <= limit; j++) {
                if (num[j] < min) {
                    min = num[j];
                    min_index = j;
                }
            }
            k -= (min_index - i); 
            swapMin(num, i, min_index); 
            if (k == 0) break; 
        }
        return num;
    }

    void swapMin(string& arr, int target_position, int current_position) {
        for (int i = current_position; i > target_position; i--) {
            swap(arr[i - 1], arr[i]);
        }
    }
};
