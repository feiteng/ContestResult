class Solution {
public:
    string minInteger(string num, int k) {
        
        int n=num.length();
        if(k*2>=n*(n-1)){
            sort(num.begin(),num.end());
            return num;
        }
        int cur=0;
        vector<int> dd(n);
        for(int i=0;i<n;i++){
            dd[i]=num[i]-'0';
        }
        vector<int> mark(n,0);
        queue<int> tmp;
        
        while(k>0){
            int mini=10;
            int idx=-1;
            int tmpcnt=0;
            int cnt=0;
            for(int i=cur;i<n && cnt<k+1;i++){
                if(mark[i]==0){
                    cnt++;
                    if(dd[i]<mini){
                        mini=dd[i];
                        idx=i;
                        tmpcnt=cnt;
                    }
                }
            }
            if(idx!=-1){
                k-=tmpcnt-1;
                tmp.push(dd[idx]);
                mark[idx]=1;
                if(idx==cur)
                    cur++;
                
            }
            else{
                break;
            }
        }
        string ans="";
        while(!tmp.empty()){
            ans+=tmp.front()+'0';
            tmp.pop();
        }
        for(int i=0;i<n;i++){
            if(mark[i]==0){
                ans+=dd[i]+'0';
            }
        }
        return ans;
    }
};
