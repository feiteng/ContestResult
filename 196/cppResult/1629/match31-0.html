class Solution {
public:
    int sum[30009];
    void add(int pos,int v){
      for(;pos<30009;pos+=pos&-pos)sum[pos]+=v;
    }
    int cal(int pos){
      int res=0;
      for(;pos;pos-=pos&-pos)res+=sum[pos];
      return res;
    }
    string minInteger(string num, int k) {
      queue<int> V[10];
      for(int i=0;i<num.size();++i){
        V[num[i]-'0'].push(i+1);
        add(i+1,1);
      }
      string t;
      for(int i=1;i<=num.size();++i){
        for(int j=0;j<=9;++j){
          if(V[j].size()==0)continue;
          if(cal(V[j].front())-1<=k){
            t+=j+'0';
            k-=cal(V[j].front())-1;
            add(V[j].front(),-1);
            V[j].pop();
            break;
          }
        }
      }
      return t;
    }
};
