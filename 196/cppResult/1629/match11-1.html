class Solution {
public:
    void print(vector<int>& arr, int n) 
    { 
        for (int i = 0; i < n; i++) { 
            cout << arr[i] << " "; 
        } 
        cout << endl; 
    } 
  

    void swapMax(vector<int>& arr, int target_position, int current_position) 
    { 
        int aux = 0; 
        for (int i = current_position; 
             i > target_position; i--) { 
            aux = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = aux; 
        } 
    } 

    // Function to return the 
    // maximum number array 
    void maximizeArray(vector<int>& arr, int length, int swaps) 
    { 
        // Base condition 
        if (swaps == 0) 
            return; 

        // Start from the first index 

        for (int i = 0; i < length; i++) { 
            int max_index = 0, max = INT_MAX; 

            // Search for the next K elements 
            int limit = (swaps + i) > length ?  
                            length : swaps + i; 

            // Find index of the maximum 
            // element in next K elements 
            // cout<<limit<<" "<<i<<"\n";
            for (int j = i; j < length && j <= limit; j++) { 
                if (arr[j] < max) { 
                    max = arr[j]; 
                    max_index = j; 
                } 
            } 

            // Update the value of 
            // number of swaps 
            swaps -= (max_index - i); 

            // Update the array elements by 
            // swapping adjacent elements 
            swapMax(arr, i, max_index); 

            if (swaps == 0) 
                break; 
        } 
    } 
  
    string minInteger(string num, int k) {
        int n = num.size();
        int cnt = n*(n-1)/2;
        
        if(k >= cnt){
            sort(num.begin(), num.end());
            return num;
        }
        vector<int > arr;
        for(auto x: num){
            int v = x - '0';
            arr.push_back(v);
        }
        maximizeArray(arr, n, k);
        string ans;
        for(auto x: arr){
            ans += (x + '0');
        }
        return ans;
    }
};
