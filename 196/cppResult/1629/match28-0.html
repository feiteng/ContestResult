class Solution {
public:
    
    void mdf(vector<int>& fen, int w, int x) {
        for(; w < fen.size(); w += w&-w) fen[w] += x;
    }
    
    int qry(const vector<int>& fen, int w) {
        int ret = 0;
        for(; w > 0; w -= w&-w) ret += fen[w];
        return ret;
    }
    
    string minInteger(string num, int k) {
        int n = num.size();
        vector<int> fen(n + 1);
        vector<queue<int>> vcq(10);
        for(int i = 0; i < n; i++) {
            mdf(fen, i + 1, 1);
            vcq[num[i] - '0'].push(i);
        }
        
        string ans = "";
        for(int i = 0; i < n && k; i++) {
            for(int j = 0; j < 10; j++) {
                if(vcq[j].empty()) continue;
                int elm = vcq[j].front();
                int costo = qry(fen, elm + 1) - 1;
                if(costo <= k) {
                    k -= costo;
                    mdf(fen, elm + 1, -1);
                    ans += char('0' + j);
                    vcq[j].pop();
                    break;
                }
            }
        }
        for(int i = 0; i < n; i++)
            if(qry(fen, i + 1) != qry(fen, i)) ans += num[i];
        return ans;
    }
};
