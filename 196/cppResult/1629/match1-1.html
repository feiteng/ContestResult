class Solution {
public:
    string minInteger(string num, int k) {
        int n=num.length();

        if(n*(n-1/2) <k){sort(num.begin(),num.end());
                        return num;
                        }
         for (int i = 0; i<n-1 && k>0; i++) 
    { 
        int p = i; 
        for (int j = i+1; j<n ; j++) 
        { 
            if (j-i > k) 
                break; 
            if (num[j] < num[p]) 
                p = j; 
        } 
          for (int j = p; j>i; j--) 
            swap(num[j], num[j-1]); 
        k -=  p-i; 
    } 
    return num;
    }
};
