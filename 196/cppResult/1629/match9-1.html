class Solution {
public:
    string minInteger(string num, int k) {
    	
    	int idx = 0,n = num.length();
    	
    	if(n*(n-1)/2 <= k){
    		sort(num.begin(),num.end());
    		return num;
		}
    	
    	while(k > 0 && idx < n){
    		char cur = num[idx],mi = '9';
    		if(cur == '0')	continue;

    		int pos = -1;
    		for(int i = idx+1;i < min(idx+k+1,n);i++){
    			if(num[i] < mi){
    				mi = num[i];
    				pos = i;
				}
			}
			
			if(mi < cur){
				k -= (pos - idx);
				char t = num[pos];
				//cout << num << " " << t << "\n";
				num.erase(pos,1); 
				//cout << num << " " << t << "\n";
				num.insert(num.begin()+idx,t);
				//cout << num << "\n";
			}
			idx++;
			cout << idx << "\n";
		}
		
		return num;
    	
    }
};
