class Solution {
public:
    void swapMin(int* arr, int target_position, int current_position) 
    { 
        int aux = 0; 
        for (int i = current_position; i > target_position; i--) { 
            aux = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = aux; 
        } 
    } 

    
    string minInteger(string s, int swaps) {
        if (swaps == 0) 
        return s; 
        
        if(swaps>=(s.length()*s.length())){
            sort(s.begin(), s.end());
            return s;
        }

        int length = s.length();
        int arr[length];
        for(int i=0;i<length;i++)arr[i] = s[i]-'0';
        for (int i = 0; i < length; i++) { 
            int min_index = 0, min = INT_MAX; 
            int limit = (swaps + i) >= length ?  length-1 : swaps + i; 
            for (int j = i; j <= limit; j++) { 
                if (arr[j] < min) { 
                    min = arr[j]; 
                    min_index = j; 
                } 
            } 
            swaps -= (min_index - i);  
            swapMin(arr, i, min_index); 
            if (swaps == 0) break; 
        } 
        string ans;
        for(int i=0;i<length;i++)ans+=to_string(arr[i]);
        return ans;
    }
};
