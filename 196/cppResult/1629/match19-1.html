class Solution {
 public:
  string minInteger(string num, int k) {
    vector<int> cnt(10);
    vector<vector<int>> before_big(10, vector<int>());
    vector<vector<int>> idx(10, vector<int>());
    int index = 0;
    for (auto x : num) {
      int a = x - '0';
      int b = 0;
      for (int i = a + 1; i < 10; i++) b += cnt[i];
      before_big[a].push_back(b);
      idx[a].push_back(index++);
      cnt[a]++;
    }

    int n = num.size();
    string ret(n, '-');
    vector<int> used(10);
    vector<int> before_used_big(10);
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < 10; j++) {
        if (used[j] < before_big[j].size()) {
          if (k >= before_big[j][used[j]] - before_used_big[j]) {
            ret[i] = j + '0';
            k -= before_big[j][used[j]] - before_used_big[j];
            used[j]++;
            for (int k = 0; k < j; k++) before_used_big[k]++;
            break;
          }
        }
      }
    }
    return ret;
  }
};
