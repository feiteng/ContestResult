class Solution {
public:
    string minInteger(string num, int k) {
        int n = num.length();
        queue<int> index[256];
        for (int i = 0; i < n; i++) {
            index[num[i]].push(i);
        }
        
        string ans;
        deque<int> picked;

        for (int i = 0; i < n; i++) {
            for (char d = '0'; d <= '9'; d++) {
                if (index[d].empty())
                    continue;
                int first = index[d].front();
                auto insert_at = lower_bound(picked.begin(), picked.end(), first);

                first += picked.end() - insert_at;
                if (first - i <= k) {
                    k -= first - i;
                    picked.insert(insert_at, index[d].front());
                    index[d].pop();
                    ans += d;
                    break;
                }
            }
        }
        return ans;
    }
};
