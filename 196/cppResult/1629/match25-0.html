class Solution {
public:
    string minInteger(string num, int k) {
        if(k==0) return num;
        int n = num.size();
        if(n*(n-1)/2 <=k){
            sort(num.begin(), num.end());
            return num;
        }
        int left = 0, right=0;
        int minn = 1000000,idx;
        while(k && left < n-1){
            right = left + 1;
            minn = num[left];
            idx = left;
            while(k && right < n){
                if(num[right]< minn){
                    minn = num[right];
                    idx = right;
                }
                right++;
                k--;
                
            }
            k += right - idx -1;
            num = num.substr(0, left) + num[idx] + num.substr(left, idx - left) + num.substr(idx + 1, n-idx-1);
            left++;
    }
        return num;
    }
};
