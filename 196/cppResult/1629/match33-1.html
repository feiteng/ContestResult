class Solution {
public:
  string minInteger(string num, int k) {
    int n = num.size();
    vector<set<int>> a(10);
    for (int i = 0; i < n; ++i) a[num[i] - '0'].insert(i);
    vector<int> c(n + 1);
    string ret;

    for (int i = 0; i < n; ++i) {
      for (int d = 0; d < 10; ++d) {
        if (a[d].empty()) continue;
        int x = *a[d].begin();
        int y = x;
        for (int u = x + 1; u > 0; u -= u & -u) y += c[u];
        if (y - i <= k) {
          ret += (char)(d + '0');
          a[d].erase(a[d].begin());
          k -= y - i;
          for (int u = 1; u <= n; u += u & -u) ++c[u];
          for (int u = x + 1; u <= n; u += u & -u) --c[u];
          break;
        }
      }
    }
    return ret;
  }
};
