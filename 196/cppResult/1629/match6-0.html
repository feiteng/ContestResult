class Solution {
public:
    string minInteger(string num, int k) {
        int size = num.size();
        if (k >= size * (size - 1) / 2)
        {
            sort(num.begin(), num.end());
            return num;
        }
        int high = 0;
        while (k > 0 && high < size)
        {
            int index = high;
            for (int i = high; i - high <= k && i < size; ++i)
            {
                if (num[index] > num[i])
                    index = i;
            }
            if (index == high)
                ++high;
            else
            {
                num.insert(num.begin() + high, num[index]);
                num.erase(num.begin() + index + 1);
                k -= index - high;
                ++high;
            }
        }
        return num;
    }
    
};
