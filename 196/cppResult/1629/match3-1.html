class Solution {
public:
    string minInteger(string num, int k) 
    {
        int n=num.size();
        if(k>=n*(n-1)/2)
        {
            sort(num.begin(),num.end());
            return num;
        }
        for(int i=1;i<=n;i++)
        {
            int mn=i-1;
            for(int j=i;j<n&&j<k+i;j++)
                if(num[mn]>num[j]) mn=j;
            if(mn!=i-1)
            {
                int t=num[mn];
                for(int j=mn;j>=i;j--,k--)
                    num[j]=num[j-1];
                num[i-1]=t;
            }
        }
        return num;
    }
};
