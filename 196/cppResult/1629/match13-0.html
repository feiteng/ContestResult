class Solution {
public:
    string minInteger(string arr, int k) {
        if (k >= 30000*30000) {
            sort(arr.begin(), arr.end());
            return arr;
        }
        for (int i = 0; i<arr.size()-1 && k>0; ++i) 
        { 
            int pos = i; 
            for (int j = i+1; j<arr.size() ; ++j) 
            { 
                if (j-i > k) 
                    break; 

                if (arr[j] < arr[pos]) 
                    pos = j; 
            } 
            for (int j = pos; j>i; --j) 
                swap(arr[j], arr[j-1]); 

            k -=  pos-i; 
        }
        return arr;
    }
};
