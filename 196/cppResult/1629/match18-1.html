class Solution {
public:
    
    int a[40000];

    void init(int n) {
        for (int i = 0;i <= n;i++) {
            a[i] = 0;
        }
    }
    int lowbit(int x) {
        return x & (-x);
    }
    void add(int x, int v) {
        while (x <= 30000) {
            a[x] += v;
            x += lowbit(x);
        }
    }
    int sum(int x) {
        int ret = 0;
        while (x != 0) {
            ret += a[x];
            x -= lowbit(x);
        }
        return ret;
    }
    
    string minInteger(string num, int k) {
        init(num.size());
        string ret = "";
        vector<int> data[10];
        int next[10];
        memset(next, 0, sizeof(next));

        for (int i = 0;i < num.size();i++) {
            data[num[i] - '0'].push_back(i);
        }
        for (int i = 0;i < num.size();i++) {
            int start = 0;

            for (int j = start;j < 10;j++) {
                if (next[j] == data[j].size()) {
                    continue;
                }
                int p = data[j][next[j]];
                if (k >= p + sum(p + 1) - i) {
                    k -= p + sum(p + 1) - i;
                    next[j]++;
                    ret.push_back(char('0' + j));
                    add(1, 1);
                    add(p + 1, -1);
                    break;
                }
            }
        }
        return ret;
    }
};
