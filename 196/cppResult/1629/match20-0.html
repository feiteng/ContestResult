#include<cstdio>
#include<algorithm>
#include<vector>
#include<iostream>
using namespace std;

class Solution {
public:
    int a[300005];
    std::vector<int>pos[15];
    int c[300005];
    
    void change(int p,int v,int n){
        while(p<=n){
            c[p] += v;
            p += (p&-p);
        }
    }
    
    int ask(int p){
        int res = 0;
        while(p!=0){
            res += c[p];
            p -= (p&-p);
        }
        return res;
    }
    
    string minInteger(string num, int k) {
        string res = "";
        int n = num.size();
        for(int i = 1; i <= n; i++){ 
            a[i] = (num[i-1]-'0');
            change(i,1,n);
        }
        for(int i = n; i >= 1; i--){
            int u = a[i];
            pos[u].push_back(i);
        }
        for(int i = 0; i < n; i++){
            int find = -1;
            for(int j = 0; j <= 9; j++){
                if(pos[j].size()!=0){
                    int pp = pos[j][pos[j].size()-1];
                    if(ask(pp-1)<=k){
                        find = pp;
                        break;
                    }
                }
            }
            int u = a[find];
            res.push_back(u+'0');
            pos[u].pop_back();
            k -= ask(find-1);
            change(find,-1,n);
        }
        return res;
    }
};
