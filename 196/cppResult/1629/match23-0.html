class Solution {
public:
    void bitSet(vector<int>& bit, int x) {
        ++x;
        while (x<bit.size()) {
            ++bit[x];
            x+=x&(-x);
        }
    }
    int bitCount(vector<int>&  bit, int x) {
        ++x;
        int ans=0;
        while (x) {
            ans+=bit[x];
            x-=x&(-x);
        }
        return ans;
    }
    string minInteger(string num, int k) {
        int nz=num.size();
        vector<int> bit(nz+10);
        map<char, set<int>> m;
        for (int i=0;i<nz;++i) {
            m[num[i]].insert(i);
        }
        string ans;
        for (int i=0;i<nz;++i) {
            for (char c='0';c<='9';++c) {
                if (m[c].empty())continue;
                int front=*m[c].begin();
                int newFront=front+i-bitCount(bit, front);
                if (newFront-i<=k) {
                   ans+=c;
                    m[c].erase(front);
                    bitSet(bit, front);
                    k-=newFront-i;
                    break;
                }
            }
        }
        return ans;
            
        
        
    }
};
