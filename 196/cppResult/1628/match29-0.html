class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int r = mat.size(), c= mat[0].size();
        vector<vector<int>> p_arr(r, vector<int>(c,0));
        
        for(int i=0;i<r;i++)
        {
            for(int j=c-1;j>=0;j--)
            {
                if(mat[i][j]==1)
                {
                    if(j!=c-1)
                        p_arr[i][j]+=p_arr[i][j+1];
                    p_arr[i][j]+=1;
                }
            }
        }
        
        int ans = 0;
        for(int j=0;j<c;j++)
        {
            int i = r-1;
            stack<pair<int,int>> q;
            int to_sum = 0;
            
            while(i>=0)
            {   
                int co = 0;
                 while (q.size() != 0 and q.top().first > p_arr[i][j]) { 
  
                    to_sum -= (q.top().second + 1) *  
                                (q.top().first - p_arr[i][j]); 

                    co += q.top().second + 1; 
                    q.pop(); 
                } 

                to_sum += p_arr[i][j]; 

                ans += to_sum; 

                q.push({ p_arr[i][j], co }); 

                i--; 
                }
        }
        return ans;
    }
};
