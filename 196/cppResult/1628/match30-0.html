class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        vector<vector<int>> right = mat, down = mat;
        for (int i = mat.size() - 1; i >= 0; --i) {
            for (int j = mat[0].size() - 1; j >= 0; --j) {
                if (mat[i][j] == 1) {
                    right[i][j] = j == mat[0].size() - 1 ? 1 : (right[i][j + 1] + 1);
                    down[i][j] = i == mat.size() - 1 ? 1 : (down[i + 1][j] + 1);
                } else {
                    right[i][j] = 0;
                    down[i][j] = 0;
                }
            }
        }
        int result = 0;
        for (int i = 0; i != mat.size(); ++i) {
            for (int j = 0; j != mat[0].size(); ++j) {
                int accWidth = right[i][j];
                for (int di = 0; di != down[i][j]; ++di) {
                    result += (accWidth = min(right[i + di][j], accWidth));
          //          cout << i << ", " << j << " with " << di << " has " << accWidth << '\n';
                }
            }
        }
        return result;
    }
};
