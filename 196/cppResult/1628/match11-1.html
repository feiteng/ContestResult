class Solution {
public:
    // Taken from https://www.geeksforgeeks.org/number-of-submatrices-with-all-1s/
    int numSubmat(vector<vector<int>>& arr) {
        
        int n = arr.size();
        int m = arr[0].size();
        int p_arr[n][m];
        memset(p_arr, 0, sizeof p_arr);

        for (int i = 0; i < n; i++) { 
            for (int j = m - 1; j >= 0; j--) { 

                if (!arr[i][j]) 
                    continue; 

                if (j != m - 1) 
                    p_arr[i][j] += p_arr[i][j + 1]; 

                p_arr[i][j] += (int)arr[i][j]; 

            } 
        } 

       
        int ans = 0; 

      
        for (int j = 0; j < m; j++) { 

            int i = n - 1; 

           
            stack<pair<int, int> > q; 

           
            int to_sum = 0; 

            while (i >= 0) { 

                int c = 0; 

                while (q.size() != 0 and q.top().first > p_arr[i][j]) { 


                    to_sum -= (q.top().second + 1) *  
                                (q.top().first - p_arr[i][j]); 

                    c += q.top().second + 1; 
                    q.pop(); 
                } 

                to_sum += p_arr[i][j]; 

                ans += to_sum; 

                q.push({ p_arr[i][j], c }); 

                i--; 
            } 
        } 

        return ans; 
    }
};
