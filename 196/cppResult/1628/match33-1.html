class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> left(m, vector<int>(n,0));
        

        int total = 0;
        
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j++)
            {
                if(mat[i][j] == 1)
                {
                    left[i][j] = (j == 0) ? 1 : 1 + left[i][j-1];
                    

                    int width = left[i][j];
                    for(int k = i; k >= 0; k--)
                    {
                        if(mat[k][j] == 0)
                        {
                            break;
                        }
                        width = min(left[k][j], width);
                        total += width;
                    }
                }
            }
        }
        
        return total;
    }
};
