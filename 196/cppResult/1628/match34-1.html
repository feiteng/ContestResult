class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        if (mat.size() == 0 || mat[0].size() == 0) return 0;
        int n = mat.size(), m = mat[0].size();
        vector<vector<int>> dp(n, vector<int>(m));
        for(int i=0; i<n; i++) {
            int pre = 0;
            for(int j=0; j<m; j++) {
                if (mat[i][j] == 1) pre++;
                else pre = 0;

                dp[i][j] = pre;
            }
        }
        int ret = 0;
        for(int i=0; i<n; i++) {
            for(int j=0; j<m; j++) {
                int mp = dp[i][j];
                for(int k=i; k<n; k++) {
                    mp = min(mp, dp[k][j]);
                    ret += mp;
                }
            }
        }
        return ret;
    }
};
