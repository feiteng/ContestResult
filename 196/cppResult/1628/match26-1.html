class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<int>> tmp = mat;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (j > 0 && tmp[i][j]) {
                    tmp[i][j] += tmp[i][j-1];
                }

            }
        }
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {

                int width = n;
                for (int k = i; k < m; k++) {
                    width = min(width, tmp[k][j]);
                    ans += width;
                }
            }
        }
        return ans;
    }
};
