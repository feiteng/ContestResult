class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        vector<vector<int>>num(n+1,vector<int>(m+1,0));

        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                if(mat[i-1][j-1]==0) num[i][j]=0;
                else num[i][j]=num[i-1][j]+1;
            }
        }
        int ans=0;
        for(int i=1;i<=n;i++){

            for(int l=1;l<=m;l++){
                int h=num[i][l];
                for(int r=l;r<=m;r++){
                    h=min(h,num[i][r]);
                    ans+=h;
                }
            }
        }
        return ans;
    }
};
