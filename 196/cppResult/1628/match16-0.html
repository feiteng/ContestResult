class Solution {
public:
    
     
    int numSubmat(vector<vector<int>>& arr) {
        int R = arr.size();
        int C = arr[0].size();
        
        
        vector<vector<int> > p_arr(R, vector<int> (C, 0));
        
        for(int i = 0; i < R; i++) { 
            for(int j = C - 1; j >= 0; j--) { 
                if(!arr[i][j]) 
                    continue; 
                if (j != C - 1) 
                    p_arr[i][j] += p_arr[i][j + 1]; 
                p_arr[i][j] += arr[i][j]; 
            } 
        }
        
        int ans = 0; 

        for (int j = 0; j < C; j++) { 
            int i = R - 1; 
            stack<pair<int, int> > q; 
            int to_sum = 0; 
            while (i >= 0) { 
                int c = 0; 
                while (q.size() != 0 and q.top().first > p_arr[i][j]) { 
                    to_sum -= (q.top().second + 1) *  
                                (q.top().first - p_arr[i][j]); 

                    c += q.top().second + 1; 
                    q.pop(); 
                } 
                to_sum += p_arr[i][j]; 
                ans += to_sum; 
                q.push({ p_arr[i][j], c }); 
                i--; 
            } 
        } 

        return ans; 
    }
};
