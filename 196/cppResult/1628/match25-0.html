class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m=mat.size(),n=mat[0].size();
        int res=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                int len=n;
                for(int k=i;k<m;k++){
                    for(int t=j;t<n;t++){
                        if(t<len&&mat[k][t]==1){
                            res++;
                        }
                        else{
                            len=min(len,t);
                            break;
                        }
                    }
                }
            }
        }
        return res;
    }
};
