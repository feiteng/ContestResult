class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        int ans = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == 0) continue;

                int right = n;
                for (int p = i; p < m; ++p) {
                    for (int q = j; q < right; ++q) {
                        if (mat[p][q] == 1) ++ans;

                        else {
                            right = q;
                            break;
                        }
                    }
                }
                
            }
        }
        return ans;
    }
};
