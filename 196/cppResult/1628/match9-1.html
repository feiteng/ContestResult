class Solution {
public:
    void findPrefixCount(vector<vector<int>>&p_arr,vector<vector<int>>mat,int m,int n) 
{ 
    for (int i = 0; i < m; i++) { 
        for (int j = n - 1; j >= 0; j--) { 
  
            if (!mat[i][j]) 
                continue; 
  
            if (j != n - 1) 
                p_arr[i][j] += p_arr[i][j + 1]; 
  
            p_arr[i][j] += mat[i][j]; 
        } 
    } 
} 

    int numSubmat(vector<vector<int>>& mat) {
        int m=mat.size(),n=mat[0].size();

        vector<vector<int>>p_arr(m,vector<int>(n,0)); 
  
    findPrefixCount(p_arr, mat,m,n); 
   
    int g = 0; 
 
    for (int j = 0; j < n; j++) { 
  
        int i = m - 1; 
  
        stack<pair<int, int> > que;  
        int to_sum = 0; 
  
        while (i >= 0) { 
  
            int c = 0; 
  
            while (que.size() != 0 and que.top().first > p_arr[i][j]) { 
  
                to_sum -= (que.top().second + 1) *  
                            (que.top().first - p_arr[i][j]); 
  
                c += que.top().second + 1; 
                que.pop(); 
            } 
  
            to_sum += p_arr[i][j]; 
  
            g += to_sum; 
  
            que.push({ p_arr[i][j], c }); 
  
            i--; 
        } 
    } 
  
    return g;
    }
};
