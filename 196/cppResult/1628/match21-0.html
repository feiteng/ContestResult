class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int psum[n+1][m+1];
        memset(psum, 0, sizeof psum);
        
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++) psum[i+1][j+1] = psum[i+1][j]+psum[i][j+1]-psum[i][j]+mat[i][j];
        
        int res = 0;
        
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= m; j++)
                for (int k = i; k <= n; k++)
                    for (int l = j; l <= m; l++)
                        if (psum[k][l]-psum[i-1][l]-psum[k][j-1]+psum[i-1][j-1] == ((k-i+1)*(l-j+1)))
                            res++;
        return res;
    }
};
