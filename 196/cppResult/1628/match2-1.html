class Solution
{
public:
    int numSubmat(vector<vector<int>> mat)
    {
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> pref(m, vector<int>(n));
        for (int i = 0; i < m; i++)
        {
            for (int j = n - 1; j >= 0; j--)
            {

                if (!mat[i][j])
                    continue;

                if (j != n - 1)
                    pref[i][j] += pref[i][j + 1];


                pref[i][j] += mat[i][j];
            }
        }
        int ans = 0;
        for (int j = 0; j < n; j++)
        {
            int i = m - 1;
            stack<pair<int, int>> st;
            int sum = 0;
            while (i >= 0)
            {

                int c = 0;

                while (st.size() != 0 and st.top().first > pref[i][j])
                {

                    sum -= (st.top().second + 1) *
                              (st.top().first - pref[i][j]);

                    c += st.top().second + 1;
                    st.pop();
                }

                sum += pref[i][j];

                ans += sum;

                st.push({pref[i][j], c});

                i--;
            }
        }

        return ans;
    }
};
