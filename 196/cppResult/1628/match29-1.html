class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<int>> p(m, vector<int>(n));
        for (int i = 0; i < m; i++) { 
            for (int j = n - 1; j >= 0; j--) { 
                if (!mat[i][j]) 
                    continue; 
                p[i][j] = mat[i][j]; 
                if (j != n - 1) 
                    p[i][j] += p[i][j + 1]; 
                

            } 
        }
        int ans = 0;
        for (int j = 0; j < n; j++) { 
            int i = m - 1; 
            stack<pair<int, int> > q; 
            int to_sum = 0; 
            while (i >= 0) { 
                int c = 0; 
                while (q.size() != 0 and q.top().first > p[i][j]) { 
                    to_sum -= (q.top().second + 1) * (q.top().first - p[i][j]); 
                    c += q.top().second + 1; 
                    q.pop(); 
                } 
                to_sum += p[i][j]; 
                ans += to_sum; 
                q.push({ p[i][j], c }); 
                i--; 
            } 
        } 
        return ans;
    }
    
};
