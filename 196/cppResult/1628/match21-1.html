class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int a[200][200];
        memset(a, 0, sizeof a);
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j){
                a[i+1][j+1] = a[i][j+1]  + a[i+1][j] - a[i][j] + mat[i][j];
            }


        int ans = 0;
        for (int i = 1; i <= n; ++i)
            for (int j = 1; j <= m; ++j)
                for (int k = 0; k < i; ++k)
                    for (int l = 0; l < j; ++l)
                        if (a[i][j] - a[k][j] - a[i][l] + a[k][l] == (i - k) * (j -l)) ans++;

        return ans;
    }
};
