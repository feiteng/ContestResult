class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        vector<vector<int>>pre(n+1,vector<int>(m+1,0));
        
        // for(int i=0;i<n;i++){
        //     for(int j=0;j<m;j++){
        //         pre[i+1][j+1]=pre[i+1][j]+pre[i][j+1]-pre[i][j] + mat[i][j];
        //     }
        // }
        // int ans=0;
        // for(int height=1;height<=n;height++){
        //     for(int width=1;width<=m;width++){
        //         for(int i=0;i<n;i++){
        //             for(int j=0;j<m;j++){
        //                 if(i+height<=n && j+width<=m){
        //                     if(pre[i+height][j+width]-pre[i+height][j]-pre[i][j+width]+pre[i][j]==height*width)
        //                         ans++;
        //                 }            
        //             }
        //         }
        //     }
        // }
        for(int i=0;i<n;i++){
            for(int j=m-1;j>=0;j--){
                if(!mat[i][j])continue;
                pre[i][j]=pre[i][j+1]+mat[i][j];
            }
        }
        int ans = 0; 
  
        for (int j = 0; j < m; j++) { 
            
            int i=n-1;
            stack<pair<int,int>>s; 

            int sum = 0; 

            while (i >= 0) {

                int c = 0; 

                while (s.size() != 0 and s.top().first > pre[i][j]) { 

                    sum -= (s.top().second + 1)*(s.top().first - pre[i][j]); 

                    c += s.top().second + 1; 
                    s.pop(); 
                } 

                sum += pre[i][j]; 

                ans += sum; 

                s.push({pre[i][j], c}); 

                i--; 
            } 
        } 
        return ans;
    }
};
