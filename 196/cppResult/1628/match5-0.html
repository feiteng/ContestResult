class Solution {
public:
    int numSubmat(vector<vector<int>>& matrix) {
        int mm = matrix.size(), nn = matrix[0].size();
        vector<vector<int>> a(mm, vector<int>(nn, 0)), b(mm, vector<int>(nn, 0));
        int ans = 0;
        for (int i = 0; i < mm; ++i) {
            for (int j = 0; j < nn; ++j) {
                if(i == 0 && j == 0){
                    a[i][j] = matrix[i][j];
                    b[i][j] = matrix[i][j];
                    ans += a[i][j] * b[i][j];
                    continue;
                }
                if (i == 0){
                    b[i][j] = matrix[i][j];
                    if(matrix[i][j] == 0){
                        a[i][j] = 0;
                    }else{
                        a[i][j] = a[i][j - 1] + 1;
                    }
                }else if(j == 0){
                    a[i][j] = matrix[i][j];
                    if(matrix[i][j] == 0){
                        b[i][j] = 0;
                    }else{
                        b[i][j] = b[i - 1][j] + 1;
                    }
                }
                else if (matrix[i][j] == 0) {
                    a[i][j] = 0;
                    b[i][j] = 0;
                }
                else {
                    a[i][j] = a[i][j - 1] + 1;
                    b[i][j] = b[i - 1][j] + 1;
                }
                int mm1 = a[i][j];
                for(int k = 0; k < b[i][j]; ++k){
                    mm1 = min(mm1, a[i - k][j]);
                    ans += min(mm1, a[i - k][j]);
                }
            }
        }
        return ans;
    }
};
