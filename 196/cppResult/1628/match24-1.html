class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        vector<vector<int>>p(n,vector<int>(m,0));
        findPre(p,mat,n,m);
        int ans=0;
        for(int j=0;j<m;j++)
        {
            int i=n-1;
            stack<pair<int,int>>q;
            int sum=0;
            while(i>=0)
            {
                int c=0;
                while(q.size()!=0&& q.top().first > p[i][j])
                {
                sum-=(q.top().second+1)*(q.top().first-p[i][j]);
                    c+=q.top().second+1;
                    q.pop();
                }
                sum+=p[i][j];
                ans+=sum;
                q.push({p[i][j],c});
                i--;
            }
        }
        return ans;
    }
    void findPre(vector<vector<int>>&p,vector<vector<int>>mat,int n,int m)
    {

        for(int i=0;i<n;i++){
            for(int j=m-1;j>=0;j--){
                if(!mat[i][j])
                    continue;
                if(j!=m-1)
                    p[i][j]+=p[i][j+1];
                p[i][j]+=mat[i][j];
            }
        }
    }
    
    
};
