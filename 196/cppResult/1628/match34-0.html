class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        if(mat.size() == 0)
            return 0;
        int H = mat.size(), W = mat[0].size();
        vector<vector<int>> M(H, vector<int>(W));
        for(int r=0;r<H;++r) {
            int m = 0;
            for(int c=0;c<W;++c) {
                if(mat[r][c] == 1) {
                    m++;
                    M[r][c] = m;
                }
                else {
                    M[r][c] = 0;
                }
                m = M[r][c];
            }
        }
        
        int ans = 0;
        for(int c=0;c<W;++c) {
            for(int r1=0;r1<H;++r1) {
                int min_val = M[r1][c];
                for(int r2=r1;r2<H;++r2) {
                    min_val = min(min_val, M[r2][c]);
                    ans += min_val;
                }
            }
        }
        
        return ans;
    }
};
