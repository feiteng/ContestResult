class Solution {
public:
    int r,c;
    void prefix(vector<vector<int>>& dp, vector<vector<int>>& mat) {
        for (int i=0; i<r; i++) {
            for (int j=c-1; j >= 0; j--) {
                if (!mat[i][j]) continue;
                if (j != (c-1))
                    dp[i][j] += dp[i][j+1];
                dp[i][j] += (int)mat[i][j];
            }       
        }
    }
    int numSubmat(vector<vector<int>>& mat) {
        r = mat.size();
        c = mat[0].size();
        vector<vector<int>> dp(r , vector<int>(c,0));
        prefix(dp,mat);
        int ans = 0;
        
        for(int j=0; j<c; j++) {
            
            int i = r-1;
            stack<pair<int,int>> st;
            int to_sum = 0;
            
            while (i>=0) {
                int c = 0;
                while ( !st.empty() and st.top().first > dp[i][j]) {
                    to_sum -= (st.top().second + 1) * (st.top().first - dp[i][j]);
                    c += st.top().second + 1;
                    st.pop();
                }
                
                to_sum += dp[i][j];
                ans += to_sum;
                st.push({dp[i][j] , c});
                i--;
            }
            
        }
        
        return ans;
        
    }
};
