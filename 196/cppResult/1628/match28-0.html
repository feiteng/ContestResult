class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int r = mat.size();
        int c = mat[0].size();
        
        int prefix[r][c];
        memset(prefix, 0, sizeof(prefix));
        
        for (int i = 0; i < r; i++) {
            for (int j = c-1; j >= 0; j--) {
                
                if (!mat[i][j]) continue;
                if (j != c-1) prefix[i][j] += prefix[i][j+1];
                prefix[i][j]++;
            }
        }
        
        int ans = 0;
        
        for (int j = 0; j < c; j++) {
            int i = r-1;
            stack<pair<int,int>> stack;
            
            int sum = 0;
            while (i >= 0) {
                int cnt = 0;
                while (!stack.empty() && stack.top().first > prefix[i][j]) {
                    auto elem = stack.top();
                    stack.pop();
                    
                    sum -= (elem.second + 1) * (elem.first - prefix[i][j]);
                    cnt += elem.second + 1;
                }
                
                sum += prefix[i][j];
                ans += sum;
                stack.push({prefix[i][j], cnt});
                i--;
            }
        }
        return ans;
    }
};
