class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> p_mat (m, vector<int> (n, 0));
        findPrefixCount(p_mat, mat); 


        int ans = 0; 

        for (int j = 0; j < n; j++) { 
            int i = m - 1; 
            stack<pair<int, int> > q; 
            int to_sum = 0; 

            while (i >= 0) { 
                int c = 0; 

                while (q.size() != 0 and q.top().first > p_mat[i][j]) { 

                    to_sum -= (q.top().second + 1) *  
                                (q.top().first - p_mat[i][j]); 

                    c += q.top().second + 1; 
                    q.pop(); 
                } 

                to_sum += p_mat[i][j]; 

                ans += to_sum; 

                q.push({ p_mat[i][j], c }); 

                i--; 
            } 
        } 

        return ans; 
    }
    void findPrefixCount(vector<vector<int>>& p_mat, vector<vector<int>>& mat) 
    { 
        int m = mat.size();
        int n = mat[0].size();

        for (int i = 0; i < m; i++) { 
            for (int j = n - 1; j >= 0; j--) { 

                if (!mat[i][j]) 
                    continue; 

                if (j != n - 1) 
                    p_mat[i][j] += p_mat[i][j + 1]; 

                p_mat[i][j] += mat[i][j]; 
            } 
        } 
    }
};
