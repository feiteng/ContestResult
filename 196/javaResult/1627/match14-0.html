class Solution {
    public int getLastMoment(int n, int[] left, int[] right) {
        int res = 0;
        for (int val : left) {
            res = Math.max(res, val);
        }
        for (int val : right) {
            res = Math.max(res, n - val);
        }
        return res;
    }
}
