class Solution {
    public int getLastMoment(int n, int[] left, int[] right) {
        
        int min = Integer.MIN_VALUE;
        for (int i : left) {
            min = Math.max(i, min);
        }
        
        for (int i : right) {
            min = Math.max(n - i, min);
        }
        
        return min;
    }
}
