class Solution {
    public int getLastMoment(int n, int[] left, int[] right) {
        int max = Integer.MIN_VALUE;
        for (int i : left) {
            max = Math.max(i, max);
        }
        for (int i : right) {
            max = Math.max(max, n - i);
        }
        return max;
    }
}
