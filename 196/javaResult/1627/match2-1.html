class Solution {
    public int getLastMoment(int n, int[] left, int[] right) {
        // Set<Integer> toLeft = new HashSet<>(Arrays.toList(left));
        // Set<Integer> toRight = new HashSet<>(Arrays.toList(right));
        // int time = 0;
        // while (true) {
        //     Set<Integer> tmp = new HashSet<>();
        //     for (int ant : toLeft) {
        //         if (toRight.contains(ant)) {
        //             tmp.add(ant);
        //         } 
        //     }
        // }
        int res = 0;
        for (int l : left) {
            res = Math.max(res, l);
        }
        
        for (int r : right) {
            res = Math.max(res, n - r);
        }
        return res;
    }
}
