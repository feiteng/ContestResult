class Solution {
    public String minInteger(String num, int k) {
        // Base condition 
        if (k == 0) 
            return num;
        
        int n = num.length();
        
        
char[] arr = num.toCharArray();

        for (int i = 0; i < n; i++) { 
            int min_index = 0, min = Integer.MAX_VALUE; 
            int limit = (k + i) >= n ? n - 1 : k + i; 

            for (int j = i; j <= limit; j++) { 
                if (arr[j] < min) { 
                    min = arr[j]; 
                    min_index = j; 
                }
            } 
            
            k -= (min_index - i); 

            swapMin(arr, i, min_index); 

            if (k == 0) 
                break; 
        }
        
        return new String(arr);
    }
    
    static void swapMin(char[] arr, int target, 
int cur) { 
        char tmp = 0; 
        for (int i = cur; i > target; i--) { 
            tmp = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = tmp; 
        } 
    } 
}
