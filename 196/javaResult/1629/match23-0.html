class Solution {
    public String minInteger(String num, int k) {
        // find the minimum value that is within k swaps
        char[] cs = num.toCharArray();
        for (int i = 0; i < cs.length && k > 0; i += 1) {
            char cur = cs[i];
            int iSwap = i;
            for (int j = i + 1; j - i <= k && j < cs.length; j += 1) {
                if (cs[j] < cs[iSwap]) {
                    iSwap = j;
                }
            }
            k -= iSwap - i;
            char newVal = cs[iSwap];
            for (int j = iSwap - 1; j >= i; j -= 1) {
                cs[j + 1] = cs[j];
            }
            cs[i] = newVal;
        }
        return new String(cs);
    }
}
