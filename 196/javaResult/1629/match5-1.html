class Solution {
    
public String minInteger(String num, int k) {
        if (k == 0) return num; 
        char[] charr = num.toCharArray();
        for (int i = 0; i < charr.length; i++)  
        { 
            int minIdx = 0, min = Integer.MAX_VALUE; 
            int lmt = (k + i) >= charr.length ? charr.length - 1 : k + i; 
            for (int j = i; j <= lmt; j++)  
            { 
                
if ((int)(charr[j] - '0') < min)  
                { 
                    min = (int)(charr[j] - '0'); 
                    minIdx = j; 
                } 
            } 
            k -= (minIdx - i); 
            swapMin(charr, i, minIdx); 

            if (k == 0) break;         
        }
        return new String(charr);
    }

    
private void swapMin(char[] arr, int tp, int cp) { 
        char aux = 0; 
        for (int i = cp; i > tp; i--) { 
            aux = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = aux; 
        } 
    } 
}
