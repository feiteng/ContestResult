class Solution {
    public String minInteger(String num, int k) {
        int nextDigit = 0;
        char[] c = num.toCharArray();
        while(nextDigit < c.length - 1 && k > 0) {
            int min = c[nextDigit];
            int minIndex = nextDigit;
            for(int i = nextDigit + 1; i < c.length; i++) {
                if(c[i] < min) {
                    min = c[i];
                    minIndex = i;
                }
            }
            if(minIndex - nextDigit <= k) {
                putToNextDigit(c, minIndex, nextDigit);
                k -= minIndex - nextDigit;
                nextDigit++;
            } else {
                return best(c, nextDigit, k);
            }
        }
        return result(c);
    }
    
    private String best(char[] c, int nextDigit, int k) {
        while(k > 0) {
            int min = c[nextDigit];
            int minIndex = nextDigit;
            for(int i = nextDigit + 1; i <= nextDigit + k; i++) {
                if(c[i] < min) {
                    min = c[i];
                    minIndex = i;
                }
            }
            putToNextDigit(c, minIndex, nextDigit);
            k -= minIndex - nextDigit;
            nextDigit++;
        }
        return result(c);
    }
    
    private String result(char[] c) {
        StringBuilder sb = new StringBuilder();
        for(char ch : c) {
            sb.append(ch);
        }
        return sb.toString();
    }
    
    private void putToNextDigit(char[] c, int minIndex, int nextDigit) {
        char tmp = c[minIndex];
        for(int i = minIndex - 1; i >= nextDigit; i--) {
            c[i + 1] = c[i];
        }
        c[nextDigit] = tmp;
    }
}
