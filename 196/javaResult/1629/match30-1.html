class Solution {
    public String minInteger(String num, int k) {
        int l = num.length();
        char[] arr = num.toCharArray();
        int pt = 0;
        while (k > 0 && pt < l) {
            int top = Integer.min(l, pt + k + 1);
            
char min = '9' + 1;
            int minIndex = pt;
            for (int i = pt; i < top; i++) {
                if (arr[i] < min) {
                    min = arr[i];
                    minIndex = i;
                }
            }
            swap(arr, pt, minIndex);
            k -= (minIndex - pt);
            pt++;
        }

        return new String(arr);
    
}

    private void swap(char[] arr, int pt, int minIndex) {
        char temp = arr[minIndex];
        for (int i = minIndex; i > pt; i--) {
            arr[i] = arr[i-1];
        }
        arr[pt] = temp;
    }
}
