class Solution {
    public void swapMin(int[] arr, int target_position, 
                    int current_position) 
        { 
            int aux = 0; 
            for (int i = current_position; 
                i > target_position; i--)  
            { 
                aux = arr[i - 1]; 
                arr[i - 1] = arr[i]; 
                arr[i] = aux; 
            } 
        } 
    public String minInteger(String num, int k) {
        if(k == 0)
            return num;
        char[] ch = num.toCharArray();
        int length = num.length();
        int[] arr = new int[length];
        for(int i = 0; i < length; i++){
            arr[i] = ch[i] - '0';
        }
        
for(int i = 0; i < length; i++){
            int min_index = 0, min = Integer.MAX_VALUE;
            int limit = (k + i) >= length ? length : k + i + 1;
            
            for(int j = i; j < limit; j++){
                if(arr[j] < min){
                    min = arr[j];
                    min_index = j;
                }
            }
            k -= (min_index - i);
            swapMin(arr, i, min_index);
            if(k == 0)
                break;
        }
        for(int i = 0; i < length; i++){
            ch[i] = (char)(arr[i] + '0');
        }
        return String.valueOf(ch);
    }
}
