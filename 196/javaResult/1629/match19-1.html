class Solution {
    
void swapMax(char[] arr, int target_position, 
                    int current_position) 
    { 
    char aux = '\0'; 
    for (int i = current_position; i > target_position; i--)
          
        { 
            aux = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = aux; 
        } 
    } 
    public String minInteger(String num, int k) {
        if(k==0){
            return num;
        }
        
int n = num.length()-1;
        char arr[] = num.toCharArray();
        for (int i = 0; i <n+1; i++)  
        { 
        int maxindex = 0, max = Integer.MAX_VALUE; 
        int limit = (k + i) > n ?  
                        n: k + i; 
        for (int j = i; j <= limit; j++)  
        { 
            
if (arr[j] < max)  
            { 
                max = arr[j]; 
                maxindex = j; 
            } 
        } 
        k -= (maxindex - i); 
        swapMax(arr, i, maxindex); 
  
        if (k == 0) 
            break; 
        }
        String ans = new String(arr);
        return ans;
    }
}
