class Solution {
    
    static void swaper(char[] arr, int tar,int cur) 
    { 
	    char aux = 0; 
	    for (int i = cur;i > tar; i--) 
	    { 
		    aux = arr[i - 1]; 
		    arr[i - 1] = arr[i]; 
		    arr[i] = aux; 
	    } 
    } 
    
    static String cal(String arr1,int length, int swaps) 
    { 
	    if (swaps == 0) 
		    return arr1; 

        char arr[] = arr1.toCharArray();
        
	    for (int i = 0; i < length; i++) 
	    { 
		    int max_index = 0, min = Integer.MAX_VALUE; 
            int limit = (swaps + i) > length  ? length  : swaps + i; 

		    for (int j = i; j < limit; j++) 
		    { 
			    if (arr[j] < min) 
			    { 
				    min = arr[j]; 
				    max_index = j; 
			    } 
		    } 

	        swaps -= (max_index - i);   
            swaper(arr, i, max_index);  
            
		    if (swaps == 0) 
			    break; 
	    } 
        return new String(arr);
    } 

    
    public String minInteger(String num, int k) {
        return cal(num,num.length(),k+1);
    }
}
