class Solution {
    public String minInteger(String num, int k) {
        char[] cc = num.toCharArray();
        int n = cc.length;
        int curr = 0;
        while(k != 0 && curr < n){
            int localMin = curr;
            for(int i = curr;i<Math.min(n,curr+k+1);i++){
                if(cc[i]-'0' < cc[localMin]-'0'){
                    localMin = i;
                }
            }
            if(localMin == curr){
                curr++;
                continue;
            }
            k-=localMin - curr;
            swap(curr++,localMin,cc);
            
        }
        return new String(cc);
    }
    
    public void swap(int a, int b, char[] c){
        for(int i = b;i>a;i--){
            char temp = c[i];
            c[i] = c[i-1];
            c[i-1] = temp;
        }
    }
}
