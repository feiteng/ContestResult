class Solution {
    public String minInteger(String num, int k) {
        char[] ch = num.toCharArray();
        recur(ch,k,0);
        return String.valueOf(ch);
    }
    
    public void recur(char[] ch, int k, int start) {
        if(k == 0 || start >= ch.length) return;
        int end;
        if(ch.length-1- start <= k) {
            end = ch.length - 1;
        }else {
            end = start + k;
        }
        
        int index = start;
        
        for(int i = start+1; i <= end; i++) {
            if(Integer.valueOf(ch[i]) < Integer.valueOf(ch[index])) {
                index = i;
            }
        }
        
        char temp = ch[index];
        for(int i = index - 1; i >= start; i--) {
            ch[i+1] = ch[i];
        }
        ch[start] = temp;
        
        k = k-index + start;
        
        recur(ch,k,start+1);
    }
}
