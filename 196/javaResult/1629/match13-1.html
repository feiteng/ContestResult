class Solution {
    
public void minInt(char[] n, int start, int k) {
        if (start >= n.length || k <= 0) return;
        char minDigit = 'a';
        int minIndex = -1;
        for (int i = start; i < Math.min(start + k + 1, n.length); i++) 
{
            // System.out.println(minDigit + " " + n[i]);
            if (minDigit > n[i]) {
                minDigit = n[i];
                minIndex = i;
            }
        }
        char curr = '0';
        for (int i = minIndex; i > start; i--) {
            curr = n[i-1];
            n[i-1] = n[i];
            n[i] = curr;
        }
        minInt(n, start+1, k - (minIndex - start));
    }
    
    
public String minInteger(String num, int k) {
        char[] n = num.toCharArray();
        minInt(n, 0, k);
        return new String(n);
    }
}
