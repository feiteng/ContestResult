class Solution {
    public String minInteger(String num, int k) {
        char[] t = num.toCharArray();
        
        for (int i = 0; i < t.length; ++i) {
            int min = i;
            for (
int j = i+1, l = 0; j < t.length && l < k; ++j, ++l) {
                if (t[j] < t[min]) {
                    min = j;
                }
            }
            char tt = t[min];
            for (int j = min; j > i; --j) {
                t[j] = t[j-1];
            }
            t[i] = tt;
            
            k -= (min -i);
        }
        StringBuilder sb = new StringBuilder();
        for (char c: t) {
            sb.append(c);
        }
        return sb.toString();
    }
}
