class Solution {
    private void swap(char[] chars, int i, int index) {
        char temp = '0';
        for (int k = index; k > i; k--) {
            temp = chars[k-1];
            chars[k-1] = chars[k];
            chars[k] = temp;
        }
    }
    public String minInteger(String num, int k) {
        if (k == 0) return num;
        char[] chars = num.toCharArray();
        int n = chars.length;
        for (int i=0; i<n; i++) {
            int index = 0, min = Integer.MAX_VALUE;
            int limit = Math.min(k + i, n - 1);
            for (int j=i; j<=limit; j++) {
                int v = chars[j] - '0';
                if (v < min) {
                    min = v;
                    index = j;
                }
            }
            k -= (index - i);
            swap(chars, i, index);
            if (k == 0) break;
        }
        return new String(chars);
    }
}
