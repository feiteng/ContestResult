class Solution {
    public String minInteger(String num, int k) {
        char[] arr = num.toCharArray();

        int start = 0;
        int end = arr.length - 1;
        while (start < end) {
            if (k == 0) break;
                int min = start;
                for (int j = start + 1; j <= end && j <= start + k; j++) {
                    if(arr[j] < arr[min]) min = j;
                }
                if (min != start) {
                    k -= min - start;
                    swap(arr, min, start);
                }
                start++;
            
        
}
        return new String(arr);
           
    }
    
    public void swap(char[] arr, int min, int start) {
        for (int i = min; i > start; i--) {
            char temp = arr[i];
            arr[i] = arr[i - 1];
            arr[i - 1] = temp;
        }
    }
}
