class Solution {
    public String minInteger(String num, int k) {
        char[] arr = num.toCharArray();
        
        for (int i = 0; i < arr.length; i++) {
            int idx = i;            
            for (int j = 1; j <= k && i + j < arr.length; j++) {
                if (arr[i + j] - '0' < arr[idx] - '0') {
                    idx = i + j;
                }
                
                if (arr[idx] == '0') break;
            }
            
            char temp = arr[idx];
            for (int j = idx; j > i; j--) {
                arr[j] = arr[j - 1];
            }
            
            arr[i] = temp;
            
            k -= idx - i;
        }
        
        return String.valueOf(arr);
    }
}
