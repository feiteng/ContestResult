class Solution {
    int find(char[] arr, int i, int k) {
        
int candidate = i;
        for (int j = 1; j <= k && i + j < arr.length; j++) {
            if (arr[candidate] > arr[i + j]) {
                candidate = i + j;
            }
        }
        return candidate;
    }
    
    void move(char[] arr, int i, int j) {
        
char temp = arr[j];
        for (int k = j; k > i; k--) {
            arr[k] = arr[k - 1];
        }
        arr[i] = temp;
    }
    
    
public String minInteger(String num, int k) {
        char[] arr = num.toCharArray();
        for (int i = 0; i < arr.length && k > 0; i++) {
            int index = find(arr, i, k);
            move(arr, i, index);
            k -= (index - i);
        }
        return new String(arr);
    }
}
