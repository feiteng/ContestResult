class Solution {
    public String minInteger(String num, int k) {
        char[] charNum = num.toCharArray();
      
        int lastK = k;
        for(int i=0;i<charNum.length;++i){
            int bestChoice =i;
            char bestC = charNum[i];
            for(int j=i+1;j<charNum.length && j<=i+lastK;++j){
                if(charNum[j]<bestC){
                    bestC = charNum[j];
                    bestChoice = j;
                }
                if(bestC=='0')break;
            }
            for(int j=bestChoice;j>i;--j){
                changeP(charNum,j,j-1);
            }
            lastK-=(bestChoice-i);
            
        }
        return String.valueOf(charNum);
    }
    private void changeP(char[] c, int a,int b){
        char t=c[a];
        c[a]=c[b];
        c[b]=t;
    }
}
