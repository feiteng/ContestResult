class Solution {
    
    public String minInteger(
String s, int k) {
        if (k == 0) return s; 
        char[] cc = s.toCharArray();
        for (int i = 0; i < cc.length; i++)  
        { 
            int minIdx = 0;
            int min = Integer.MAX_VALUE; 
            int count = 0;
            if ((k + i) >= cc.length) {
                count = cc.length - 1;
                
            } else {
                count = k + i;
            }
            
for (int j = i; j <= count; j++)  
            { 
                if ((cc[j] - '0') < min)  
                { 
                    
                    minIdx = j; 
                    min = (cc[j] - '0'); 
                    
                } 
            } 
            
            k -= (minIdx - i); 
            swap(cc, i, minIdx); 

            if (k == 0) break;         
        }
        return new String(cc);
    }

    public void swap(char[] arr, int t, 
int s) { 
        char tmp = 0; 
        for (int i = s; i > t; i--) { 
            tmp = arr[i - 1]; 
            arr[i - 1] = arr[i]; 
            arr[i] = tmp; 
        } 
    } 
}
