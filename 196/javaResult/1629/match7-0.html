class Solution {
    public String minInteger(String num, int k) {
        char[] numArr = num.toCharArray();
        int n = num.length();
        for (int i = 0; i < n; i++) {
            int minIdx = 0;
            int min = Integer.MAX_VALUE;
            int limit = (k + i) > n - 1 ? n - 1: k + i;
            for (int j = i; j <= limit; j++) {
                int digit = (int)(numArr[j] - '0');
                if (digit < min) {
                    min = digit;
                    minIdx = j;
                }
            }
            k -= (minIdx - i);
            swapMin(numArr, minIdx, i);
            if (k == 0) break;
        }
        return new String(numArr);
    }
    
    public void swapMin(char[] c, int current, int target) {
        char tmp = c[current];
        for (int i = current; i > target; i--) {
            tmp = c[i - 1];
            c[i - 1] = c[i];
            c[i] = tmp;
        }
    }
}
