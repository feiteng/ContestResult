        int [][]dp=new int[mat.length+1][mat[0].length+1];
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
               if(mat[i][j]==1){
                dp[i+1][j+1]=dp[i][j+1]+1;
               
}
            }
        }
        int sum=0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                int min=Integer.MAX_VALUE;
                for(int k=j;k>=0;k--){
                 min=Math.min(min,dp[i+1][k+1]);
                sum+=min;
                }
            }
        }
        return sum;
    }
