class Solution {
    public int numSubmat(int[][] mat) {
        int r = mat.length, c = mat[0].length;
        int[][] pmat = new int[r][c];
        
        for (int i = 0; i < r; i++) {
            for (int j = c-1; j >= 0; j--) {
                if (mat[i][j] == 0) continue;
                if (j != c-1)
                    pmat[i][j] += pmat[i][j+1];
                pmat[i][j] += mat[i][j];
            }
        }
        
        int res = 0;
        for (int j = 0; j < c; j++) {
            int i = r-1;
            Stack<int[]> stack = new Stack<>();
            int curr = 0;
            
            while (i >= 0) {
                int temp = 0;
                while (!stack.isEmpty() && stack.peek()[0] > pmat[i][j]) {
                    curr -= (stack.peek()[1] + 1) * (stack.peek()[0] - pmat[i][j]);
                    temp += stack.pop()[1] + 1;
                }
                curr += pmat[i][j];
                res += curr;
                stack.push(new int[] {pmat[i--][j], temp});
            }
        }
        return res;
    }
}
