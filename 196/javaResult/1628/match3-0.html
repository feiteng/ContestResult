class Solution {
    public int numSubmat(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        
        
        //how many 1s in the same row are at the right of right[i][j]
        int[][] right = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = n-1; j >= 0; j--) {
                if (mat[i][j] == 0) {
                    continue;
                }
                if (j != n-1) {
                    right[i][j] += right[i][j+1];
                }
                right[i][j] += mat[i][j];
            }
        }
        
        int ans = 0;
        for (int j = 0; j < n; j++) {
            int i = m-1;
            Stack<int[]> stack = new Stack<>();
            int sum = 0;
            while(i >= 0) {
                int c = 0;
                while(!stack.isEmpty() && stack.peek()[0] > right[i][j]) {
                    sum -= (stack.peek()[1] + 1) * (stack.peek()[0] - right[i][j]);
                    c += stack.peek()[1] + 1;
                    stack.pop();
                }
                sum += right[i][j];
                ans += sum;
                stack.push(new int[] {right[i][j], c});
                i--;
            }
        }
        return ans;
    }
}
