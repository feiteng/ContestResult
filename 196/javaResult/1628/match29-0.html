class Solution {
    public int numSubmat(int[][] mat) {
        int m=mat.length;
        int n=mat[0].length;
        int[][] sum=new int[m+1][n+1];
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                sum[i][j]=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];
                if(mat[i-1][j-1]==1) sum[i][j]++;
            }
        }
        // System.out.println(Arrays.deepToString(sum));
        int ans=0;
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                for(int ri=i;ri<=m;ri++){
                    for(int rj=j;rj<=n;rj++){
                        int cnt=sum[ri][rj]-sum[i-1][rj]-sum[ri][j-1]+sum[i-1][j-1];
                        int di=ri-i+1;
                        int dj=rj-j+1;
                        // System.out.format("i is %d, j is %d, ri is %d, rj is %d, one is %d, cnt is %d\n",i,j,ri,rj,cnt,di*dj);
                        if(di*dj==cnt) ans++;
                    }
                }
            }
        }
        return ans;
    }
}
