class Solution {
    public int numSubmat(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] sum = new int[m + 1][n + 1];
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                sum[i+1][j+1] = sum[i+1][j] + sum[i][j+1] - sum[i][j]; 
                if(mat[i][j] == 1) {
                    sum[i+1][j+1]++;
                }
            }
        }
        int count = 0;
        for(int i = 0; i <= m; i++) {
            for(int j = 0; j <= n; j++) {
                for(int k = i+1; k <= m; k++) {
                    for(int l = j+1; l <= n; l++) {
                        if(sum[k][l] - sum[i][l] - sum[k][j] + sum[i][j] == (k - i) * (l - j)) {
                            count++;
                        }
                    }
                }
            }
        }
        return count;
    }
}
