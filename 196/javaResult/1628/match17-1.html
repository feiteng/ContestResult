class Solution {
    public int numSubmat(int[][] mat) {
        int row = mat.length;
        int col = mat[0].length;
        for (int i = 0; i < row; i++) {
            for (int j = 1; j < col; j++) {
                if (mat[i][j] == 1 && mat[i][j - 1] > 0) {
                    mat[i][j] = mat[i][j - 1] + 1;
                
}
            }
        }
        int sum = 0;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                int curr = mat[i][j];
                for (int k = i; k >= 0 && curr > 0; k--) {
                    curr = Math.min(curr, mat[k][j]);
                    sum += curr;
                }
            }
        }
        return sum;
    }
}
