class Solution {
        public int numSubmat(int[][] mat) {
            int row = mat.length;
            int col = mat[0].length;
            int[][] sum = new int[row+1][col+1];
            for (int i=0;i<row;i++) {
                for (int j=0;j<col;j++) {
                    sum[i+1][j+1] = sum[i][j+1] + sum[i+1][j] - sum[i][j] + mat[i][j];
                }
            }
//            for (int i=0;i<=row;i++) {
//                for (int j=0;j<=col;j++) {
//                    System.out.printf("%d ", sum[i][j]);
//                }
//                System.out.println();
//            }
            int count = 0;
            for (int i1=1;i1<=row;i1++) {
                for (int i2=i1;i2<=row;i2++) {
                    for (int j1=1;j1<=col;j1++) {
                        for (int j2=j1;j2<=col;j2++) {
                            
if (sum[i2][j2] - sum[i2][j1-1] - sum[i1-1][j2] + sum[i1-1][j1-1] == (i2-i1+1) * (j2-j1+1)) {
                                count++;
                            }
                        }
                    }
                }
            }
//            System.out.println(count);
            return count;
        }
    }
