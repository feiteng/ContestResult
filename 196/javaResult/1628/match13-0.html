class Solution {
    public int numSubmat(int[][] mat) {
        int n = mat.length;
        int m = mat[0].length;
        int[][] pref = new int[1 + n][1 + m];
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                pref[i][j] = pref[i - 1][j] + pref[i][j - 1] - pref[i - 1][j - 1] + mat[i - 1][j - 1];
            }
        }

        int res = 0;
        for (int i1 = 0; i1 < n; i1++) {
            for (int j1 = 0; j1 < m; j1++) {
                for (int i2 = i1; i2 < n; i2++) {
                    for (int j2 = j1; j2 < m; j2++) {
                        // [i1:i2][j1:j2]
                        int sum = pref[i2 + 1][j2 + 1] - pref[i2 + 1][j1] - pref[i1][j2 + 1];
                        sum += pref[i1][j1];

                        if (sum == (i2 - i1 + 1) * (j2 - j1 + 1)) {
                            res++;
                        }
                    }
                }
            }
        }
        return res;
    }
}
