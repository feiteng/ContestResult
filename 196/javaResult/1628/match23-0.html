class Solution {
    public int numSubmat(int[][] mat) {
        
        int l = mat.length;
        int  ll = mat[0].length;
        
        int s[][]  = new int[l+1][ll+1];
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                s[i+1][j+1] = mat[i][j] +  s[i][j+1] + s[i+1][j] - s[i][j];
            }
        }
        
        int ct = 0;
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                for(int ii=0;ii<=i;++ii){
                    for(int jj=0;jj<=j;++jj){
                    
                        int all =  s[i+1][j+1] - s[ii][j+1] - s[i+1][jj] + s[ii][jj];
                        int v = Math.abs(i-ii+1) * Math.abs(j-jj+1);
                        
                        if(v==all){
                            ct++;
                        }
                        
                    }
                }
            }
        }
        
        return ct;
        
        
        
    }
}
