class Solution {
    public int numSubmat(int[][] mat) {

         int m = mat.length,n=mat[0].length;
        int[][] pre = new int[m+1][n+1];
        for (int i=1;i<=m;i++){
            for (int j=1;j<=n;j++){
                pre[i][j] = -pre[i-1][j-1]+pre[i-1][j]+pre[i][j-1]+mat[i-1][j-1];
            }
        }
        for (int[] a: pre) System.out.println(Arrays.toString(a));
        int res =0 ;
        for (int i=1;i<=m;i++){
            for (int j=1;j<=n;j++){
                for (int x=i;x<=m;x++){
                    //boolean b = false;
                    for (int y = j;y<=n;y++){
                       int cur =  pre[x][y]+pre[i-1][j-1]-pre[x][j-1]-pre[i-1][y];
                        //System.out.println(i+"  "+j+"  "+x+"  "+y+"      "+cur+"  "+(x-i+1)*(y-j+1));
                      if (cur == (x-i+1)*(y-j+1)) res++;
                      else break;
                    }
                }
            }
        }
        return res;
    }
}
