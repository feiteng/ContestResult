class Solution {
    private class pair {  
        int first, second;  
        public pair(int first, int second)  
        {  
            this.first = first;  
            this.second = second;  
        }  
    } 
    public int numSubmat(int[][] arr) {
        int R = arr.length, C = arr[0].length;
        int[][] p_arr = new int[R][C]; 
        findPrefixCount(p_arr, arr); 
  
        int ans = 0; 
        for (int j = 0; j < C; j++) { 
            int i = R - 1;
            Stack<pair> q = new Stack<pair>(); 
            int to_sum = 0; 
            while (i >= 0) { 
                int c = 0; 
                while (q.size() != 0 && q.peek().first > p_arr[i][j]) { 
                    to_sum -= (q.peek().second + 1) *  
                              (q.peek().first - p_arr[i][j]); 

                    c += q.peek().second + 1; 
                    q.pop(); 
                } 
                to_sum += p_arr[i][j]; 
                ans += to_sum; 
                q.add(new pair(p_arr[i][j], c)); 
                i--; 
            } 
        } 
        return ans;
    }
    
    private void findPrefixCount(int p_arr[][], int arr[][]) {
        int R = arr.length, C = arr[0].length;
        for (int i = 0; i < R; i++) { 
            for (int j = C - 1; j >= 0; j--)  { 
                if (arr[i][j] == 0) continue; 
                if (j != C - 1) p_arr[i][j] += p_arr[i][j + 1]; 
                p_arr[i][j] += (arr[i][j] == 1) ? 1 : 0; 
            }
        }
    }
}
