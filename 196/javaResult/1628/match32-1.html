class Solution {
    
public void solve(int dp[][], int mat[][]){
        int n=mat.length;
        int m=mat[0].length;
        for (int i = 0; i < n; i++) 
        { 
            for (int j=m-1; j >= 0; j--)  
            { 
                if (mat[i][j]==0) 
                    continue; 
  
                if (j != m - 1) 
                    dp[i][j] += dp[i][j + 1]; 
  
                dp[i][j] += mat[i][j] == 1 ? 1 : 0; 
            } 
        } 
    }
    public int numSubmat(int[][] mat) {
        int n = mat.length;
        int m = mat[0].length;
        int dp[][] = new int[n][m];
        solve(dp,mat);
        int ans=0;
        for(int j=0;j<m;j++){
            int i = n - 1;
            Stack<Pair> q = new Stack<Pair>(); 
            int tsum = 0; 
            while (i >= 0)  
            { 
                int c = 0; 
  
                while (q.size() != 0 && q.peek().first > dp[i][j])  
                { 
                    tsum -= (q.peek().second + 1) *  
                          (q.peek().first - dp[i][j]); 
  
                    c += q.peek().second + 1; 
                    q.pop(); 
                } 
  
                tsum += dp[i][j]; 
  
                ans += tsum; 
  
                q.add(new Pair(dp[i][j], c)); 
  
                i--; 
            } 
        }
        return ans;
    }
}

class Pair 
{  
    int first, second;  
    Pair(int fi, int se)  
    {  
        first = fi;  
        second = se;  
    }  
}
