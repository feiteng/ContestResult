class Solution {
    public int numSubmat(int[][] mat) {
        
        int[][] rects = new int[mat.length][mat[0].length];        
        for (int i = 0; i < mat.length; i += 1) {
            for (int j = 0; j < mat[0].length; j += 1) {
                if (mat[i][j] == 1) {
                    rects[i][j] = 1;
                    if (i > 0) {
                        rects[i][j] += rects[i - 1][j];
                    }
                }
            }
        }
        int numSubmat = 0;
        for (int i = 0; i < mat.length; i += 1) {
            for (int j = 0; j < mat[0].length; j += 1) {
                int height = rects[i][j];
                numSubmat += height;
                for (int k = j - 1; height > 0 && k >= 0; k -= 1) {
                    height = Math.min(rects[i][k], height);
                    numSubmat += height;
                }
            }
        }
        return numSubmat;
    }
}
