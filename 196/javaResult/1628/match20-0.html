class Solution {
    int m, n;
    public int numSubmat(int[][] mat) {
        m = mat.length;
        n = mat[0].length;
        boolean[][] arr = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                arr[i][j] = mat[i][j] == 1;
            }
        }
        return matrixAllOne(arr);
    }
    class pair {  
        int first, second;  
        public pair(int first, int second)  
        {  
            this.first = first;  
            this.second = second;  
        }  
    }  


    private void findPrefixCount(int p_arr[][],  
                                boolean arr[][]) { 
        for (int i = 0; i < m; i++) 
        { 
            for (int j = n - 1; j >= 0; j--)  
            { 
                if (!arr[i][j]) 
                    continue; 

                if (j != n - 1) 
                    p_arr[i][j] += p_arr[i][j + 1]; 

                p_arr[i][j] += arr[i][j] == true ? 1 : 0; 
            } 
        } 
    } 


    private int matrixAllOne(boolean arr[][]) {
        int [][]p_arr = new int[m][n]; 
        findPrefixCount(p_arr, arr); 
        int ans = 0; 
        for (int j = 0; j < n; j++)  
        { 
            int i = m - 1; 

            Stack<pair> q = new Stack<pair>(); 
            int to_sum = 0; 

            while (i >= 0)  
            { 
                int c = 0; 

                while (q.size() != 0 && 
                       q.peek().first > p_arr[i][j])  
                { 
                    to_sum -= (q.peek().second + 1) *  
                              (q.peek().first - p_arr[i][j]); 

                    c += q.peek().second + 1; 
                    q.pop(); 
                } 

                to_sum += p_arr[i][j]; 

                ans += to_sum; 

                q.add(new pair(p_arr[i][j], c)); 

                i--; 
            } 
        } 
        return ans; 
    } 
}
