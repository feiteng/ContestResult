class Solution {
    public int numSubmat(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[][] preArr = new int[m][n]; 
        for (int i = 0; i < m; i++) {
            for (int j = n - 1; j >= 0; j--) { 
                if (mat[i][j] == 0) 
                    continue; 

                if (j != n - 1) {
                    preArr[i][j] += preArr[i][j + 1]; 
                
}

                preArr[i][j] += mat[i][j]; 
            } 
        }
        int res = 0; 

        for (int j = 0; j < n; j++)  
        { 
            int i = m - 1; 
            Stack<int[]> q = new Stack<int[]>(); 
            int sum = 0; 

            while (i >= 0)  
            { 
                int c = 0; 
                while (q.size() != 0 && q.peek()[0] > preArr[i][j]) { 
                    sum -= (q.peek()[1] + 1) * (q.peek()[0] - preArr[i][j]); 
                    c += q.peek()[1] + 1; 
                    q.pop(); 
                
} 

                sum += preArr[i][j]; 
                res += sum; 
                q.add(new int[]{preArr[i][j], c}); 
                i--; 
            } 
        } 
        return res; 
    }
}
