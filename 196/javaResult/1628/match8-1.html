class Solution {
    public int numSubmat(int[][] mat) {
        int r = mat.length;
        int c = mat[0].length;
        int[][] prefix = new int[r + 1][c + 1];
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                prefix[i + 1][j + 1] = prefix[i + 1][j] + prefix[i][j + 1] - prefix[i][j] + mat[i][j];   
            
}
        }
        
        int count = 0;
        for (int x = 0; x < r; x++) {
            for (int y = x; y < r; y++) {
                for (int p = 0; p < c; p++) {
                    for (int q = p; q < c; q++) {
                        int target = (q - p + 1) * (y - x + 1);
                        int actual = prefix[y + 1][q + 1] + prefix[x][p] - prefix[y + 1][p] - prefix[x][q + 1];
                        
if (target == actual) {
                            count++;
                        }                        
                    }
                }
            }            
        }
        return count;
    }
}
