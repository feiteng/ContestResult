class Solution {
    public int numSubmat(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        boolean[][] A = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 1) A[i][j] = true;
            }
        }
        return matrixAllOne(A);
    }

    class Pair {
        int first, second;

        public Pair(int first, int second) {
            this.first = first;
            this.second = second;
        }
    }

    void findPrefixCount(int[][] p_arr, boolean[][] arr) {
        int m = arr.length, n = arr[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = n - 1; j >= 0; j--) {
                if (!arr[i][j])
                    continue;

                if (j != n - 1)
                    p_arr[i][j] += p_arr[i][j + 1];

                p_arr[i][j] += arr[i][j] ? 1 : 0;
            }
        }
    }

    int matrixAllOne(boolean[][] arr) {
        int m = arr.length, n = arr[0].length;
        int[][] p_arr = new int[m][n];
        findPrefixCount(p_arr, arr);
        int ans = 0;
        for (int j = 0; j < n; j++) {
            int i = m - 1;
            Stack<Pair> q = new Stack<Pair>();
            int to_sum = 0;
            while (i >= 0) {
                int c = 0;
                while (q.size() != 0 &&
                        q.peek().first > p_arr[i][j]) {
                    to_sum -= (q.peek().second + 1) *
                            (q.peek().first - p_arr[i][j]);

                    c += q.peek().second + 1;
                    q.pop();
                }
                to_sum += p_arr[i][j];
                ans += to_sum;
                q.add(new Pair(p_arr[i][j], c));
                i--;
            }
        }
        return ans;
    }
}
