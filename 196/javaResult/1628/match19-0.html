class Solution {
    public int numSubmat(int[][] mat) {
        return matrixWithAllOne(mat);
    }
    
    void findPrefixCount(int prefixArray[][], int arr[][]) { 
         int m = arr.length;
        int n = arr[0].length;
    for (int i = 0; i < m; i++) 
    { 
        for (int j = n - 1; j >= 0; j--)  
        { 
            if (arr[i][j] == 0) 
                continue; 
  
            if (j != n - 1) 
                prefixArray[i][j] += prefixArray[i][j + 1]; 
  
            prefixArray[i][j] += arr[i][j] == 1 ? 1 : 0; 
        } 
    } 
} 
    
    int matrixWithAllOne(int arr[][]) { 
        int m = arr.length;
        int n = arr[0].length;
    int [][]prefixArray = new int[m][n]; 
  
    findPrefixCount(prefixArray, arr); 
    int ans = 0; 
    for (int j = 0; j < n; j++)  
    { 
        int i = m - 1; 
        Stack<Pair> q = new Stack<>(); 
        int sum = 0; 
  
        while (i >= 0)  
        { 
            int c = 0; 
  
            while (q.size() != 0 && q.peek().first > prefixArray[i][j])  
            { 
                sum = sum - ((q.peek().second + 1) * (q.peek().first - prefixArray[i][j])); 
  
                c += q.peek().second + 1; 
                q.pop(); 
            } 
  
            sum += prefixArray[i][j]; 
  
            ans += sum; 
  
            q.add(new Pair(prefixArray[i][j], c)); 
  
            i--; 
        } 
    } 
    return ans; 
} 
  
}
public class Pair 
{  
    int first, second;  
    public Pair(int first, int second)  
    {  
        this.first = first;  
        this.second = second;  
    }  
}
