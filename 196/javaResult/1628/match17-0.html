class Solution {
    public int numSubmat(int[][] mat) {
        int N = mat.length;
        int M = mat[0].length;
        for (int i = 0; i < N; i++) {
            for (int j = M - 1; j >= 0; j--) {
                if (mat[i][j] > 0) {
                    mat[i][j] = (j + 1 < M ? mat[i][j + 1] : 0) + mat[i][j];
                }
            }
        }
        
        int ans = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                int min = mat[i][j];
                for (int k = i; k >= 0; k--) {
                    min = Math.min(min, mat[k][j]);
                    ans += min;
                }
            }
        }
        
        return ans;
    }
}
