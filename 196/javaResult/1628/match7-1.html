class Solution {
    public int numSubmat(int[][] mat) {
        final int N = mat.length;
        final int M = mat[0].length;
        final int[][] sum = new int[N + 1][M + 1];
        for (int r = 0; r < N; ++r) {
            for (int c = 0; c < M; ++c) {
                sum[r + 1][c + 1] = mat[r][c] + sum[r + 1][c] + sum[r][c + 1] - sum[r][c];
            }
        }
        int ans = 0;
        for (int r0 = 0; r0 < N; ++r0) {
            for (int c0 = 0; c0 < M; ++c0) {
                for (int rf = r0 + 1; rf <= N; ++rf) {
                    for (int cf = c0 + 1; cf <= M; ++cf) {
                        int want = (rf - r0) * (cf - c0);
                        int count = sum[rf][cf] - sum[rf][c0] - sum[r0][cf] + sum[r0][c0];
                        if (want == count) {
                            ++ans;
                        } else {
                            break;
                        
}
                    }
                }
            }
        }
        return ans;
    }
}
