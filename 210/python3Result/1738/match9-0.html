class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        G = defaultdict(set)
        for v,w in roads:
            G[v].add(w)
            G[w].add(v)
        
        # print(G)
        
        ans = 0
        for i in range(n):
            for j in range(i+1,n):
                cnt = len(G[i]) + len(G[j])
                if j in G[i]:
                    cnt -= 1
                ans = max(ans, cnt)
        return ans
