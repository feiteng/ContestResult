class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        road = [[0] * n for _ in range(n)]
        cnt = [0] * n
        for x, y in roads:
            road[x][y] = 1
            road[y][x] = 1
            cnt[x] += 1
            cnt[y] += 1
        
        ans = 0
        for i in range(n - 1):
            for j in range(i + 1, n):
                ans = max(ans, cnt[i] + cnt[j] - road[i][j])
        return ans
