class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        nei = collections.defaultdict(set)
        for i, j in roads:
            nei[i].add(j)
            nei[j].add(i)
        sol = 0
        for i in range(n):
            for j in range(n):
                if j == i:
                    continue

                rank = len(nei[i]) + len(nei[j])
                if i in nei[j]:
                    rank -= 1
                sol = max(sol, rank)
        return sol
