class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        nei = [set() for i in range(n)]
        for u, v in roads:
            nei[u].add(v)
            nei[v].add(u)
        ret = 0
        for i in range(n):
            for j in range(i + 1, n):
                ret = max(ret, len(nei[i]) + len(nei[j]) - (1 if j in nei[i] else 0))
        return ret
