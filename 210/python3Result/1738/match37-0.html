class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        G = [[] for _ in range(n)]
        for p,q in roads:
            G[p].append(q)
            G[q].append(p)
        ans = 0
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                temp = len(G[i])+len(G[j])
                if j in G[i]:
                    temp -= 1
                #print(temp)
                ans = max(ans,temp)
        return ans
