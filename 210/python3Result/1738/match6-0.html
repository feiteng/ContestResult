class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        g = [[] for _ in range(n)]
        for u, v in roads:
            g[u].append(v)
            g[v].append(u)
        ans = 0
        for u in range(n):
            for v in range(u + 1, n):
                cnt = len(g[u]) + len(g[v])
                if u in g[v]:
                    cnt -= 1
                ans = max(cnt, ans)
        return ans
