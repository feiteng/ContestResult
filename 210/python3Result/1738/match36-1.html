class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        E = [[] for _ in range(n)]
        for u, v in roads:
            E[u].append(v)
            E[v].append(u)
        res = 0
        for u in range(n):
            for v in range(u + 1, n):
                cur 
= len(E[u]) + len(E[v])
                if v in E[u]:
                    cur -= 1
                res = max(res, cur)
        return res
