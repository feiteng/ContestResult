class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        graph = defaultdict(list)
        for u, v in roads:
            graph[u].append(v)
            graph[v].append(u)

        ans = 0
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                cnt = len(graph[i]) + len(graph[j])
                if j in graph[i]:
                    cnt -= 1
                ans = max(ans, cnt)

        return ans
