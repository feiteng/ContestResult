class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        d = collections.defaultdict(set)
        for a,b in roads:
            d[a].add(b)
            d[b].add(a)
            
        maxres = 0
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                cnt 
= len(d[i]) + len(d[j])
                if j in d[i]:
                    cnt -= 1
                maxres = max(maxres, cnt)
        return maxres
