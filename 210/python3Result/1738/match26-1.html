class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        e = collections.defaultdict(set)
        for x in roads:
            e[x[0]].add(x[1])
            e[x[1]].add(x[0])
        ret = 0
        for i in range(n):
            for j in range(i+1,n):
                ret = max(ret, len(e[i])+len(e[j])-(1 if j in e[i] else 0))
        return ret
