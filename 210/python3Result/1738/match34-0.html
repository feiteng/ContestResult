class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        graph = collections.defaultdict(set)
        for i, j in roads:
            graph[i].add(j)
            graph[j].add(i)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                total = len(graph[i]) + len(graph[j])
                if i in graph[j]:
                    total -= 1
                res = max(res, total)
        return res
