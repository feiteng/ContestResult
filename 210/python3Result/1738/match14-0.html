class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        g = [[] for i in range(n)]
        for u, v in roads:
            g[u].append(v)
            g[v].append(u)
            
        res = 0
        for u in range(n - 1):
            for v in range(u + 1, n):
                d = len(g[u]) + len(g[v])
                if u in g[v]:
                    d -= 1
                    
                res = max(res, d)
                
        return res
