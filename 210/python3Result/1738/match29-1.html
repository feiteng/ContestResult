class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        conn=[0 for _ in range(n)]
        memo=set()
        for x,y in roads:
            conn[x]+=1
            conn[y]+=1
            memo.add((x,y))
        ret=0
        for i in range(n):
            for j in range(i+1,n):
                temp=conn[i]+conn[j]
                if (i,j) in memo or (j,i) in memo:
                    temp-=1
                ret=max(ret,temp)
        return ret
