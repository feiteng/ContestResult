class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        
        g = defaultdict(list)
        
        for a, b in roads:
            g[a].append(b)
            g[b].append(a)
        
        
        ans = 0
        
        for i in range(0, n + 1):
            for j in range(0, n + 1):
                if i == j:continue
                curr = len(g[i]) + len(g[j])
                if i in g[j]:curr -= 1
                ans = max(ans, curr)
                
        return ans
