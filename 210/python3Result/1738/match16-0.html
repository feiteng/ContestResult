class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        i_js = defaultdict(set)
        for i, j in roads:
            i_js[i].add(j)            
            i_js[j].add(i)
        ans = 0
        for i in range(n-1):
            for j in range(i+1, n):
                v = len(i_js[i]) + len(i_js[j])
                if j in i_js[i]:
                    v -= 1
                ans = max(ans, v)
        return ans
