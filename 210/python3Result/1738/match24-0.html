class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        gr = [set() for _ in range(n)]
        for x, y in roads:
            gr[x].add(y)
            gr[y].add(x)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                res = max(res, len(gr[i]) + len(gr[j]) - (1 if i in gr[j] else 0))
        return res
