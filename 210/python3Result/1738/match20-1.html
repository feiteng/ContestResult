class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        c = collections.Counter()
        s = set()
        for a, b in roads:
            c[a] += 1
            c[b] += 1
            s.add((a,b))
            s.add((b,a))
        ans = 0
        for i in range(n-1):
            for j in range(i+1, n):
                t = c[i] + c[j] - ((i, j) in s)
                ans = max(ans, t)
        return ans
