class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        cache = collections.defaultdict(set)
        for a, b in roads:
            cache[a].add(b)
            cache[b].add(a)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                v = len(cache[i]) + len(cache[j])
                if j in cache[i]:
                    v -= 1
                res = max(res, v)
        return res
