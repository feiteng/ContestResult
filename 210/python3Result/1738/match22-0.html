class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        G = [[0]*n for _ in range(n)]
        deg = [0]*n
        
        for a, b in roads:
            G[a][b] = 1
            G[b][a] = 1
            deg[a] += 1
            deg[b] += 1
        
        ans = 0
        
        for i in range(n):
            for j in range(i+1, n):
                ans = max(ans, deg[i]+deg[j]+(-1 if G[i][j]==1 else 0))
        
        return ans
