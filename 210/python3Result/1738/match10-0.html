class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        graph = defaultdict(list)
        
        for i, j in roads:
            graph[i].append(j)
            graph[j].append(i)
        
        result = 0
        for i in range(n):
            for j in range(i+1, n):
                rank = len(graph[i])+len(graph[j])
                if j in graph[i]:
                    rank -= 1
                result = max(result, rank)
                
        return result
