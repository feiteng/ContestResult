class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        adj = collections.defaultdict(set)
        for u, v in roads:
            adj[u].add(v)
            adj[v].add(u)
        mxrank = 0
        for i in range(n):
            for j in range(i+1, n):
                rank = len(adj[i]) + len(adj[j])
                if i in adj[j]:
                    rank -= 1
                mxrank = max(mxrank, rank)
        return mxrank
