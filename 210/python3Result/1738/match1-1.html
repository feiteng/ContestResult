class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        graph = collections.defaultdict(set)
        
        for a, b in roads:
            graph[a].add(b)
            graph[b].add(a)
            
        max_rank = 0      
        for i in range(n):
            for j in range(n):
                if i == j: continue
                    
                rank = len(graph[i])+len(graph[j])
                if i in graph[j]:
                    rank -= 1
                max_rank = max(max_rank, rank)
        return max_rank
