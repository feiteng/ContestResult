class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        # for each edge count incident edges on its vertices - 1
        # oops! cities don't have to be counted
        degree = collections.Counter()
        direct = set()
        for u, v in roads:
            degree[u] += 1
            degree[v] += 1
            direct.add((u,v))
            direct.add((v,u))
            
        ans = 0
        for u in range(n):
            for v in range(u+1, n):
                rank = degree[u] + degree[v] - (1 if (u,v) in direct else 0)
                ans = max(ans, rank)
            
        return ans
