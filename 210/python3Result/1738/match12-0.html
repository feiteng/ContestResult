class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        tmp = defaultdict(set)
        for i, p in roads:
            tmp[i].add(p)
            tmp[p].add(i)
        ans = 0
        for i in range(n):
            for p in range(i+1, n):
                tmpans = len(tmp[i]) + len(tmp[p])
                if p in tmp[i]:
                    tmpans -= 1
                ans = max(ans, tmpans)
        return ans
