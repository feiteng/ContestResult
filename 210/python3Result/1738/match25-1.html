class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        connects = [set() for i in range(n)]
        for a, b in roads:
            connects[a].add(b)
            connects[b].add(a)
        answer = 0
        for i in range(n-1):
            for j in range(i+1, n):
                answer = max(answer, len(connects[i])+len(connects[j])-(j in connects[i]))
        return answer
