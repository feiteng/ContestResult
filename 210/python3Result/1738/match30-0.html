class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        graph = collections.defaultdict(set)
        for a, b in roads:
            graph[a].add(b)
            graph[b].add(a)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                cur = len(graph[i]) + len(graph[j])
                if i in graph[j]:
                    cur -= 1
                res = max(res, cur)
        return res
