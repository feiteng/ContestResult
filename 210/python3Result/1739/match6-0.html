class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        if a == a[::-1] or b == b[::-1]:
            return True
        
        def make(c: str, d: str) -> bool:
            i, j = 0, len(d) - 1
            while i <= j and c[i] == d[j]:
                i += 1
                j -= 1
            c1 = c[i:j + 1]
            d1 = d[i:j + 1]
            return i >= j or c1 == c1[::-1] or d1 == d1[::-1]
