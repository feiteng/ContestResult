class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        
        def isPal(s):
            n = len(s)
            for i in range(n // 2):
                if s[i] != s[n - 1 - i]:
                    return False
            return True
        
        n = len(a)
        for i in range(n):
            if a[i] != b[n - 1 - i]:
                if isPal(a[i:n-i]) or isPal(b[i:n-i]):
                    return True
                break
        for i in range(n):
            if b[i] != a[n - 1 - i]:
                if isPal(a[i:n-i]) or isPal(b[i:n-i]):
                    return True
                break
        return False
