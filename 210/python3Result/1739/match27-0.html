class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def isPalinerome(s):
            i = 0
            j = len(s) - 1
            while i<j:
                if s[i]!=s[j]:
                    return False
                i += 1
                j -= 1
            return True

            return True

        if len(a) == 1:
            return True
        if isPalinerome(a) or isPalinerome(b):
            return True

        j = len(a) - 1
        while i<j:

        if flag == 0:
            return True
        flag = 0
        i = index_i
        j = index_j
        while i<j:
            if  b[i] == b[j]:

            else:
                flag = 1
                break
        if flag == 0:
            return True
