class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        if a==a[::-1] or b==b[::-1]:
            return True
        
        l = len(a)
        
        head, tail = 0,l-1
        while 0<=head<l and 0<=tail<l and a[head] == b[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        h,t = head,tail
        while 0<=head<l and 0<=tail<l and b[head] == b[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        head,tail = h,t
        while 0<=head<l and 0<=tail<l and a[head] == a[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        head,tail = 0,l-1
        while 0<=head<l and 0<=tail<l and b[head] == a[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        h,t=head,tail
        while 0<=head<l and 0<=tail<l and a[head] == a[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        head,tail = h,t
        while 0<=head<l and 0<=tail<l and b[head] == b[tail]:
            head+=1
            tail-=1
        if tail<=head:
            return True
        
        return False
