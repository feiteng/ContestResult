class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        
        def isPal(x):
            n = len(x) // 2            
            return x[:n] == x[-n:][::-1]
        
        if len(a) < 2 or isPal(a) or isPal(b):
            return True
        
        if a[0] == b[-1] or a[-1] == b[0]:
            return True
        
        return False
