class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def checkp(s):
            tot = len(s)
            n = math.floor(len
(s) / 2)
            for i in range(n):
                if s[i] != s[tot - i - 1]:
                    return False
                
            return True
        
        tot = len(a)
        n = math.floor(len(a) / 2)
        

        flag = True
        
for i in range(n):
            if a[i] != b[tot - i - 1]:
                if checkp(a[i: (tot - i)]) or checkp(b[i : (tot - i)]):
                    return True
                
                flag = False
                break
        

        if flag:
            return True
        
        flag = True
        
for i in range(n):
            if b[i] != a[tot - i - 1]:
                if checkp(a[i: (tot - i)]) or checkp(b[i : (tot - i)]):
                    return True
                
                flag = False
                break
        
        return flag
