class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        return self.helper(a, b) or self.helper(b, a)
        
    def helper(self, a, b):
        left, right 
= 0, len(a)-1
        while left < right:
            if a[left] == b[right]:
                left += 1
                right -= 1
            else:
                break
        if left >= right:
            return True
        return a[left:right+1] == a[left:right+1][::-1] or b[left:right+1] == b[left:right+1][::-1]
