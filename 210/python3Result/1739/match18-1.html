class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        # Write your code here
        def helper(a,b):
            l, r = 0, len(a)-1
            while l < r and a[l] == b[r]:
                l, r 
= l+1, r-1
            return a[l:r+1] == a[l:r+1][::-1] or b[l:r+1] == b[l:r+1][::-1]
        return helper(a,b) or helper(b,a)
