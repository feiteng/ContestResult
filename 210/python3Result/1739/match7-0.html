class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def solve(a, b):
            i, j = 0, len(a) - 1
            while i < j and a[i] == b[j]:
                i += 1
                j -= 1
            if i >= j:
                return True
            bi, bj = i, j
            while bi < bj and b[bi] == b[bj]:
                bi += 1
                bj -= 1
            if bi >= bj:
                return True
            ai, aj = i, j
            while ai < aj and a[ai] == a[aj]:
                ai += 1
                aj -= 1
            if ai >= aj:
                return True
            return False
        
        return solve(a, b) or solve(a[::-1], b[::-1])
