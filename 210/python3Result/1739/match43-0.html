class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def isPalindrome(s: str) -> bool:
            i, j = 0, len(s) - 1
            while i < j:
                if s[i] != s[j]:
                    return False
                i += 1
                j -= 1
            return True

        if isPalindrome(a) or isPalindrome(b):
            return True

        n = len(a)
        return a[0] == b[n - 1] or a[n - 1] == b[0]
