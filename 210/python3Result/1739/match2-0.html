class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def helper(a, b):
            i, n = 0, len(a)
            while a[i] == b[-i-1] and i < n - i - 1:
                i += 1
            return a[i:n-i] == a[i:n-i][::-1] or b[i:n-i] == b[i:n-i][::-1]
        return helper(a, b) or helper(b, a)
