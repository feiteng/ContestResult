class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def check(a, b):
            la, lb = len(a), len(b)
            i, j = 0, la-1
            while j-i > 1 and a[i]==b[j]:
                i += 1
                j -= 1
            if j-i==1 and la & 1 == 0:
                return True
            if j-i==0 and la & 1:
                return True
            bi, bj = i, j
            while bj>bi and b[bi]==b[bj]:
                bi += 1
                bj -= 1
            if bj<=bi:
                return True

            ai, aj = i, j
            while aj>ai and a[ai]==a[aj]:
                ai += 1
                aj -= 1
            if aj<=ai:
                return True
            return False
        
        return check(a, b) or check(b, a)
