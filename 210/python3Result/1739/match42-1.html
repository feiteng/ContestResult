class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        
        def isp(s):
            n = len(s)
            for i in range(len(s) // 2):
                if s[i] != s[n - i - 1]:
                    return False
            return True
        
        if isp(a) or isp(b):
            return True
        
        def maxMidPLen(s):
            n = len(s)
            thisres = 0
            if n % 2 == 1:
                thisres = 1
            
            
for i in range(n // 2 - 1, -1, -1):
                if s[i] == s[n - i - 1]:
                    thisres += 2
                else:
                    break
            return thisres
        
        maxmidpa = maxMidPLen(a)
        maxmidpb = maxMidPLen(b)
        
        mina = (len(a) - maxmidpa) // 2
        maxa = len(a) - mina - 1
        minb = (len(b) - maxmidpb) // 2
        maxb = len(b) - minb - 1
        
        def maxlen(s1, s2):
            n = len(s1)
            thisres 
= 0
            for i in range(n // 2):
                if s1[i] == s2[n - i - 1]:
                    thisres += 1
                else:
                    break
            return thisres
        
        
        maxlen1 = maxlen(a, b)
        maxlen2 = maxlen(b, a)
        # print(maxmidpa, maxmidpb, mina, minb, maxa, maxb, maxlen1, maxlen2)
        # print(len(a))
        # print(maxlen1, maxlen2)
        # print(maxlen1, maxb, len(a))
        if maxlen1 >= mina or maxlen1 + maxb >= len(a) - 1:
            # print("here")
            return True
        
        
        if maxlen2 >= minb or maxlen2 + maxa >= len(a) - 1:
            return True
        return False
