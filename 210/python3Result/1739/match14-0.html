class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        length=len(a)
        if length==1:
            return True
        left,right=0,length-1
        while left<right:
            if a[left]!=b[right]:
                break
            left+=1
            right-=1
        if left>=right:
            return True
        if a[left:right+1]==a[left:right+1][::-1] or b[left:right+1]==b[left:right+1][::-1]:
            return True
        left,right=0,length-1
        while left<right:
            if b[left]!=a[right]:
                break
            left+=1
            right-=1
        if left>=right:
            return True
        if a[left:right+1]==a[left:right+1][::-1] or b[left:right+1]==b[left:right+1][::-1]:
            return True
        return False
