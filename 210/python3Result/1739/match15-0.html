class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        i,j = 0,len(a)-1
        while a[i] == b[j] and i<j:
            i+=1
            j-=1
        if i>=j:
            return True
        x,y = i,j
        while a[x]==a[y] and i<j:
            x+=1
            y-=1
        if x>=y:
            return True
        while b[i]==b[j] and i<j:
            i+=1
            j-=1
        if i>=j:
            return True
        a,b = b,a
        i,j = 0,len(a)-1
        while a[i] == b[j] and i<j:
            i+=1
            j-=1
        if i>=j:
            return True
        x,y = i,j
        while a[x]==a[y] and i<j:
            x+=1
            y-=1
        if x>=y:
            return True
        while b[i]==b[j] and i<j:
            i+=1
            j-=1
        if i>=j:
            return True
        return False
