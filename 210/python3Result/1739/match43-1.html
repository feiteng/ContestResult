class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def valid(s):
            i, j = 0, len(s)-1
            while i < j:
                if s[i] != s[j]:
                    return False 
                i += 1 
                j -= 1
            return True     
        
        if valid(a) or valid(b):
            return True
        
        def match(a, b):
            # find palin from b's center
            left = n = (len(b)-1) // 2
            right = left if len(b) % 2 == 1 else left+1
            
            palin = [False for _ in range(len(b))]
            while left > 0:
                if b[left] == b[right]:
                    palin[left] = palin[right] = True
                else:    
                    break
                
                left -= 1 
                right += 1
            
            left, right = 0, len(a)-1
            while left < right:
                if a[left] != b[right]:
                    return False
                
                if palin[left+1] or left+1 == right:
                    return True
                
                left += 1
                right -= 1
                
        return match(a, b) or match(b, a) or match(a[::-1], b[::-1]) or match(b[::-1], a[::-1])
