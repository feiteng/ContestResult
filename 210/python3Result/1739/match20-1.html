class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        def isPal(c, i, j):
            while i <= j and c[i] == c[j]:
                i += 1
                j -= 1
            return i > j
        if isPal(a, 0, n - 1) or isPal(b, 0, n - 1):
            return True
        i, j 
= 0, n - 1
        while i <= j and a[i] == b[j]:
            i += 1
            j -= 1
        if isPal(a, i, j) or isPal(b, i, j):
            return True
        i, j = 0, n - 1
        while i <= j and b[i] == a[j]:
            i += 1
            j -= 1
        if isPal(a, i, j) or isPal(b, i, j):
            return True
        return False
