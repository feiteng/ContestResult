class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def helper(a, b):
            if len(a) <= 1:
                return True
            #step1
            i = 0
            n = len(a)
            while i <= n // 2:
                if a[i] == b[n - i - 1]:
                    i += 1
                else:
                    break
            #print(i)
            if n % 2 == 0:
                mid 
= n // 2
                l, r = mid - 1, mid
                cur_len = 0
                while l >= 0 and r < n:
                    if a[l] == a[r]:
                        l -= 1
                        r += 1
                        cur_len += 2
                    else:
                        break
                if 2 * (i + 1) + cur_len >= n:
                    return True

                l, r = mid - 1, mid
                cur_len = 0
                
while l >= 0 and r < n:
                    if b[l] == b[r]:
                        l -= 1
                        r += 1
                        cur_len += 2
                    else:
                        break
                if 2 * (i + 1) + cur_len >= n:
                    return True
                return False
            else:
                mid 
= n // 2
                l, r = mid - 1, mid + 1
                cur_len = 1
                while l >= 0 and r < n:
                    if a[l] == a[r]:
                        l -= 1
                        r += 1
                        cur_len += 2
                    else:
                        break
                if 2 * (i + 1) + cur_len >= n:
                    return True

                l, r = mid - 1, mid + 1
                cur_len = 1
                
while l >= 0 and r < n:
                    if b[l] == b[r]:
                        l -= 1
                        r += 1
                        cur_len += 2
                    else:
                        break
                if 2 * (i + 1) + cur_len >= n:
                    return True
                return False
            return False
        return helper(a, b) or helper(b, a)
