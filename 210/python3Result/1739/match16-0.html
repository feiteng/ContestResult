class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        l, r = (len(a) - 1) // 2, len(a) // 2
        
        while l >= 0 and a[l] == a[r]:
            l -= 1
            r += 1

        l1, r1 = l, r

        
        while l1 >= 0 and a[l1] == b[r1]:
            l1 -= 1
            r1 += 1

        if l1 < 0:
            return True
        
        l1, r1 = l, r
        
        while l1 >= 0 and b[l1] == a[r1]:
            l1 -= 1
            r1 += 1

        if l1 < 0:
            return True
        

        l, r = (len(b) - 1) // 2, len(b) // 2
        
        
        while l >= 0 and b[l] == b[r]:
            l -= 1
            r += 1
        
        l1, r1 = l, r
        
        while l1 >= 0 and b[l1] == a[r1]:
            l1 -= 1
            r1 += 1
            
        if l1 < 0:
            return True
        
        l1, r1 = l, r
        
        while l1 >= 0 and a[l1] == b[r1]:
            l1 -= 1
            r1 += 1
            
        
        if l1 < 0:
            return True
        
        return False
