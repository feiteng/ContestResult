class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        if a == a[::-1] or b == b[::-1]:
            return True
        
        def check(a, b):
            l, r = 0, len(b) - 1
            while l < r:
                if a[l] != b[r]:
                    break
                l 
+= 1
                r -= 1
            return a[l:r+1] == a[l:r+1][::-1] or b[l:r+1] == b[l:r+1][::-1]
        
        return check(a,b) or check(b,a)
