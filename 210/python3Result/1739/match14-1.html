class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        i, j = 0, n - 1
        while i < j and a[i] == b[j]:
            i 
+= 1
            j -= 1
        if i > (n - 1) // 2:
            return True
        if a[i: j + 1] == a[i: j + 1][::-1] or b[i: j + 1] == b[i: j + 1][::-1]:
            return True
        i, j = 0, n - 1
        while i < j and b[i] == a[j]:
            i 
+= 1
            j -= 1
        if i > (n - 1) // 2:
            return True
        if a[i: j + 1] == a[i: j + 1][::-1] or b[i: j + 1] == b[i: j + 1][::-1]:
            return True
        return False
