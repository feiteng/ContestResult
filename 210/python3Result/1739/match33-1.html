class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        # check 0 split
        i, j = 0, len(a) - 1
        while i < j and a[i] == a[j]:
            i += 1
            j -= 1
        if i >= j:
            return True
        
        i, j = 0, len(b) - 1
        while i < j and b[i] == b[j]:
            i += 1
            j -= 1
        if i >= j:
            return True
        
        # check a with b
        i, j = 0, len(b) - 1
        while a[i] == b[j] and i <= j:
            i += 1
            j -= 1
        if i >= j:
            return True
        p, q = i, j
        while b[p] == b[q] and p <= q:
            p += 1
            q -= 1
        if p >= q:
            return True
        p, q = i, j
        while a[p] == a[q] and p <= q:
            p += 1
            q -= 1
        if p >= q:
            return True            
        
        # check b with a
        i, j 
= 0, len(b) - 1
        while b[i] == a[j] and i <= j:
            i += 1
            j -= 1
        if i >= j:
            return True
        p, q = i, j
        while b[p] == b[q] and p <= q:
            p += 1
            q -= 1
        if p >= q:
            return True
        p, q = i, j
        while a[p] == a[q] and p <= q:
            p += 1
            q -= 1
        if p >= q:
            return True
        
        return False
