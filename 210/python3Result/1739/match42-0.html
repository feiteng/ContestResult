class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        
        if n == 1:
            return True
        
        def calc(a1, b1):
            res1 = -1
            for i in range(n//2):
                if a1[i] != b1[n-1-i]:
                    res1 = i
                    break
            return res1 if res1 != -1 else n
        
        def calcMid(s):
            x = (n-1)//2
            c = 0
            for i in range(x, -1, -1):
                if s[i] == s[n-1-i]:
                    c += 1
                else:
                    break
            return c
        
        x1 = calc(a, b)
        x2 = calc(b, a)
        x3 = calcMid(a)
        x4 = calcMid(b)
        
        x = max(x1, x2)
        
        return x1 >= n//2 or x2 >= n//2 or x + x3 >= n//2 or x + x4 >= n//2
