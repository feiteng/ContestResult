class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def checkk(a):
            x, y = 0, len(a) - 1
            while x < y:
                if a[x] != a[y]:
                    return 
                x += 1
                y -= 1
            return True 
        
        def check(a, b):
            x, y = 0, len(a) - 1
            while x < y:
                print(f'{a[x]=}, {b[y]=}')
                if a[x] != b[y]:
                    return checkk(a[x:y+1]) or checkk(b[x:y+1])
                x += 1
                y -= 1
            return True
        return checkk(a) or checkk(b) or check(a, b) or check(b, a)
