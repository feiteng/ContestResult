class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def is_palindrome(s, i, j):
            while i < j and s[i] == s[j]:
                i += 1
                j -= 1
            return i >= j
        
        n = len(a)
        
        i, j = 0, n-1
        while a[i] == b[j]:
            i += 1 
            j -= 1
        if is_palindrome(a, i, j) or is_palindrome(b, i, j):
            return True
        
        i, j = 0, n-1
        while b[i] == a[j]:
            i += 1 
            j -= 1
        if is_palindrome(a, i, j) or is_palindrome(b, i, j):
            return True
        
        return False
