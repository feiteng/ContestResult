class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        ifa = [0 for i in range(n)]
        ifb = [0 for i in range(n)]
        mid = n//2
        if n%2:
            ifa[mid] = 1
            ifb[mid] = 1
            for i in range(mid-1,-1,-1):
                if a[i] == a[n - i - 1]:
                    ifa[i] = ifa[n - i - 1] = 1
                else:
                    break
            for i in range(mid-1,-1,-1):
                if b[i] == b[n - i - 1]:
                    ifb[i] = ifb[n - i - 1] = 1
                else:
                    break
        else:
            if a[mid] == a[mid-1]:
                ifa[mid] = ifa[mid-1] = 1
                for i in range(mid-1,-1,-1):
                    if a[i] == a[n - i - 1]:
                        ifa[i] = ifa[n - i - 1] = 1
                    else:
                        break
            if b[mid] == b[mid-1]:
                ifb[mid] = ifb[mid-1] = 1
                for i in range(mid-1,-1,-1):
                    if b[i] == b[n - i - 1]:
                        ifb[i] = ifb[n - i - 1] = 1
                    else:
                        break
        if ifa[0] or ifb[0]:
            return True
        i 
= 0
        j = len(a) - 1
        while i<j:
            if a[i] == b[j]:
                i+=1
                j-=1
            else:
                break
        if ifa[i] or ifb[j] or i>=j:
            return True
        i = 0
        j = len(a) - 1
        while i<j:
            if b[i] == a[j]:
                i+=1
                j-=1
            else:
                break
        if ifb[i] or ifa[j] or i>=j:
            return True
        i = 0
        j = len(a) - 1
        
while i<j:
            if a[i] == a[j]:
                i+=1

                break
        if i>=j:
            return True
        i = 0

while i<j:
