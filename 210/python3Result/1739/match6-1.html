class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        
def helper(a, b):
            i, j = 0, len(a) - 1
            while i < j and a[i] == b[j]:
                i += 1
                j -= 1
            
            a = a[i:j+1]
            b = b[i:j+1]
            
            return a == a[::-1] or b ==b[::-1]
        
        return helper(a, b) or helper(b, a)
