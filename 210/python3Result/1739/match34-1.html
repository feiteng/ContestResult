class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def check(a, b):
            n = len(a)
            i, j = 0, n-1
            
while i < j and a[i] == b[j]:
                i += 1
                j -= 1
            if i >= j:
                return True
            
            i0, j0 = i, j
            while i0 < j0 and a[i0] == a[j0]:
                i0 += 1
                j0 -= 1
            if i0 >= j0:
                return True
            
            i0, j0 = i, j
            while i0 < j0 and b[i0] == b[j0]:
                i0 += 1
                j0 -= 1
            if i0 >= j0:
                return True
            
            return False
        
        return check(a, b) or check(b, a)
