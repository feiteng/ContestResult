class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        
        def is_palindrome(s):
            i, j = 0, len(s) - 1
            while i <= j and s[i] == s[j]:
                i += 1
                j -= 1
            if i > j:
                return True
            else:
                return False
        
        def check(a, b):
            '''
            check if prefix_a and suffix_b form a palindrome string.
            '''
            i = 0
            j = len(b) - 1
            while i <= j and a[i] == b[j]:
                i += 1
                j -= 1
            
            if i > j:
                return True
            
            return is_palindrome(a[i: j + 1]) or is_palindrome(b[i: j + 1])
            
        return check(a, b) or check(b, a)
