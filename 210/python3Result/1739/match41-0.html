class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        start = 0
        end = len(a) - 1
        while start < end:
            if a[start] == b[end]:
                start += 1
                end -= 1
            else:
                break
        if start >= end:
            return True
        if a[start:end+1] == a[start:end+1][::-1] or b[start:end+1] == b[start:end+1][::-1]:
            return True
        start = 0
        end = len(a) - 1
        while start < end:
            if b[start] == a[end]:
                start += 1
                end -= 1
            else:
                break
        if start >= end:
            return True
        return a[start:end+1] == a[start:end+1][::-1] or b[start:end+1] == b[start:end+1][::-1]
