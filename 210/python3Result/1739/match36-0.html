class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        l = len(a)
        half = l  // 2
        
        sliced = False
        def check(a, b):
            for i in range(half):
                if a[i] != b[-i-1]:
                    if a[i] == a[-i-1] and a[i:half] == a[-i-1:-half-1:-1]:
                        return True
                    if b[i] == b[-i-1] and b[i:half] == b[-i-1:-half-1:-1]:
                        return True
                    return False
            return True
        
        return check(a, b) or check(b, a)
