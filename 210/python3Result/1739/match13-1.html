class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        return self.check(a, b) or self.check(b, a) or self.check(a, a) or self.check(b, b)
    def check(self, a, b):
        i = 0
        j 
= len(a) - 1
        
        while i < j:
            if a[i] == b[j]:
                i += 1
                j -= 1
            else:
                break
        x, y = i, j
        while x < y:
            if a[x] == a[y]:
                x += 1
                y -= 1
            else:
                break
        if x == y or x > y:
            return True
        
        x, y = i, j
        
while x < y:
            if b[x] == b[y]:
                x += 1
                y -= 1
            else:
                break
        if x == y or x > y:
            return True
        
        return False
