class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        def check_palindrome(x, y):
            i, j = 0, len(y) - 1

            while i < j:
                if x[i] == y[j]:
                    i += 1
                    j -= 1
                else:
                    break

            l, r = i, j

            while l < r:
                if y[i] == y[j]:
                    i += 1
                    j -= 1
                else:
                    break
            if l >= r:
                return True

            l, r = i, j
            l, r = i, j

            while l < r:
                if x[i] == x[j]:
                    i += 1
                    j -= 1
                else:
                    break
            if l >= r:
                return True

            return i >= j
