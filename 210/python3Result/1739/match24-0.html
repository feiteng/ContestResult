class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        if self.checkSingle(a) or self.checkSingle(b):
            return True
        
        return self.checkDouble(a, b) or self.checkDouble(b, a)
    
    
    def checkDouble(self, s, t):
        i, j = 0, len(s) - 1
        
        while i < j:
            if s[i] != t[j]:
                break
            i += 1
            j -= 1
        return self.checkSingle(s[i:j+1]) or self.checkSingle(t[i:j+1])
    
    def checkSingle(self, s):
        i, j = 0, len(s)-1
        while i < j:
            if s[i] != s[j]:
                return False
            i += 1
            j -= 1
            
        return True
