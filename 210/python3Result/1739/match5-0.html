class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        if n < 2:
            return True
        
        def isPalindrome(s):
            n = len(s)
            i, j = 0, n - 1
            while i < j:
                if s[i] != s[j]:
                    return False
                i += 1
                j -= 1
            return True
        def check(s, t):
            n = len(s)
            i, j = 0, n - 1
            while i < j:
                if s[i] != t[j]:
                    return isPalindrome(t[i+1:j]) or isPalindrome(s[i+1:j])
                i += 1
                j -= 1
            #print(s, t, i, j)
            return True
        if isPalindrome(a) or isPalindrome(b):
            return True
        if check(a, b) or check(b, a):
            return True
        return False
