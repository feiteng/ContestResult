class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:

        def ispal(s):
            i, j = 0, len(s)-1
            
while i < j:
                if s[i] != s[j]:
                    return False
                
                i += 1
                j -= 1
            return True

        
        def check(a, b):
            i, j = 0, len(a) - 1
            
while i < j:
                if a[i] != b[j]:
                    return ispal(a[i:j+1]) or ispal(b[i:j+1])
                i += 1
                j -= 1
            return True
        

        return check(a,b) or check(b,a)
