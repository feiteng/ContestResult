class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        if a == a[::-1] or b == b[::-1]:
            return True
        if a[0] != b[-1] and b[0] != a[-1]:
            return False
        n = len(a)
        i = (n - 1) // 2
        while i >= 0:
            if a[i] != a[n-1-i]:
                break
            i -= 1
        while i >= 0:
            if a[i] != b[n-1-i] and b[i] != a[n-1-i]:
                break
            i -= 1
        if i == -1:
            return True
        i = (n - 1) // 2
        while i >= 0:
            if b[i] != b[n-1-i]:
                break
            i -= 1
        while i >= 0:
            if a[i] != b[n-1-i] and  b[i] != a[n-1-i]:
                break
            i -= 1
        return i == -1
