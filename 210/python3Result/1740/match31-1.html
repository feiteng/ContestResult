class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        self
.adj = defaultdict(list)
        
        for x, y in edges:
            self.adj[x - 1].append(y - 1)
            self.adj[y - 1].append(x - 1)
        ans = [0] * (n - 1)
        for op in range(1, 1 << n):
            ret = self.dia(op, n)
            if ret != -1 and ret != 0:
                ans[ret - 1] += 1
        return ans
    
    def dia(self, op, n):
        visited = [False] * n
        for i in range(n):
            if (op & (1 << i)):
                break
        q = deque([i])
        visited
[i] = True
        while q:
            i = q.popleft()
            for j in self.adj[i]:
                if op & (1 << j) and not visited[j]:
                    visited[j] = True
                    q.append(j)
        # start from i
        q = deque([i])
        d = [0] * n
        visited = [False] * n
        visited[i] = True
        while q:
            i = q.popleft()
            for j in self.adj[i]:
                if op & (1 << j) and not visited[j]:
                    q.append(j)
                    visited[j] = True
                    d[j] = d[i] + 1
        ret = d[i]
        for i in range(n):
            if (op & (1 << i)) and not visited[i]:
                return -1
        return ret
