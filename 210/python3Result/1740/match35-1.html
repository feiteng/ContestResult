from collections import defaultdict, deque
class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        
        d = defaultdict(set)
        for u, v in edges:
            d[u-1].add(v-1)
            d[v-1].add(u-1)
            
        dist_d = {}
        for i in range(n):
            q, seen = deque([(i, 0)]), set([i])
            while q:
                j, dist = q.popleft()
                dist_d[i, j] = dist
                for nj in d[j]:
                    if nj not in seen:
                        q.append((nj, dist + 1))
                        seen.add(nj)
                        
        print(dist_d)
                        
        res = [0] * (n - 1)
            
        q, seen = deque(), set()
        for i in range(n):
            q.append((0, 1 << i))
            
        while q:
            dist, state = q.popleft()
            if dist != 0:
                res[dist - 1] += 1
            
            cur = set()
            for i in range(n):
                if (state >> i) & 1:
                    cur.add(i)
                    
            for i in range(n):
                if i not in cur and d[i] & cur:
                    nstate = state | (1 << i)
                    if nstate not in seen:
                        
                        ndist = dist
                        for j in cur:
                            ndist = max(ndist, dist_d[i, j])
                        
                        seen.add(nstate)
                        q.append((ndist, nstate))
                        
        return res
