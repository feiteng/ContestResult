class Solution:
    def treeDiameter(self, vertices, graph) -> int:
        if len(vertices) == 1: return 0
            

        def bfs(source):
            d = 0
            queue = collections.deque([source])
            visited = set(queue)
            last_node = None
            
            while queue:
                d += 1
                for _ in range(len(queue)):
                    u = queue.popleft()
                    last_node = u
                    for v in graph[u]:
                        if v in vertices and v not in visited:
                            queue.append(v)
                            visited.add(v)
                            master_visited.add(v)
            
            return (d - 1, last_node)
        
        first_node = vertices[0]
        vertices = set(vertices)
        master_visited = set([first_node])
        
        
        _, last_node = bfs(first_node)
        if len(master_visited) != len(vertices):
            return 0
        ret , _ = bfs(last_node)
        return ret
    
    def countSubgraphsForEachDiameter(self, N: int, edges: List[List[int]]) -> List[int]:
        ret = [0] * (N-1)
        graph = [[] for _ in range(N)]
        
        for u, v in edges:
            graph[u-1].append(v-1)
            graph[v-1].append(u-1)
        
        
        for x in range(1, 1 << N):
            vertices = [u for u in range(N) if x & (1 << u)]
            d = self.treeDiameter(vertices, graph)
            #print(d, vertices)
            if d > 0:
                #print(vertices, d)
                ret[d-1] += 1
        
        return ret
