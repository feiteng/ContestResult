class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        def judgeConnect(comb):
            first = comb[0]
            comb = set(comb[1:])
            dq = collections.deque()
            dq.append(first)

            while dq:
                node = dq.popleft()
                for nxt in graph[node]:
                    if nxt in comb:
                        dq.append(nxt)
                        comb.remove(nxt)

            return len(comb) == 0

        def calMaxDistance(comb):
            if not judgeConnect(comb): return -1

            combSet = set(comb)
            # start = -1

            # for node in comb:
            #     nxtList = graph[node]
            #     cnt = 0
            #     for nxt in nxtList:
            #         if nxt in combSet:
            #             cnt += 1
            #             if cnt > 1: break
            #
            #     if cnt == 1:
            #         start = node
            #         break

            res = 0
            for start in comb:
                newComb = set(comb)
                dq = collections.deque()
                dq.append(start)
                newComb.remove(start)
                cnt = 0

                while dq:
                    size = len(dq)
                    cnt += 1

                    for i in range(size):
                        node = dq.popleft()
                        for nxt in graph[node]:
                            if nxt in newComb:
                                dq.append(nxt)
                                newComb.remove(nxt)

                res = max(res, cnt)

            return res


        res = [0 for i in range(n - 1)]
        graph = collections.defaultdict(list)
        for x, y in edges:
            graph[x].append(y)
            graph[y].append(x)

        for i in range(2, n + 1):
            combs = itertools.combinations(range(1, n + 1), i)
            for comb in combs:
                dist = calMaxDistance(comb)
                if dist != -1:
                    res[dist - 2] += 1

        return res
