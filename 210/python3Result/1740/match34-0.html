import itertools
class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        graph = defaultdict(set)
        for a,b in edges:
            graph[a].add(b)
            graph[b].add(a)
        print(graph)
        result = [0]*(n-1)
        for i in range(2, n+1):
            for comb in itertools.combinations(range(1,n+1), i):
                unseen=set(comb)
                def dist(node):
                    unseen.remove(node)
                    max_dist = 0
                    terminal_dist=0
                    for i in graph[node]:
                        if i in unseen:
                            c = dist(i)
                            max_dist = max(max_dist, max(c['max'], 1+c['terminal']+terminal_dist))
                            terminal_dist = max(terminal_dist, c['terminal']+1)
                    max_dist = max(max_dist, terminal_dist)
                    return {'terminal':terminal_dist, 'max':max_dist}
                d = dist(comb[0])
                if len(unseen)==0:
                    result[d['max']-1]+=1
        return result
