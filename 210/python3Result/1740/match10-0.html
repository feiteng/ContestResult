class Solution:        
        
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        arr = [[1e9 for i in range(n+1)] for i in range(n+1)]
        for i,j in edges:
            arr[i][j] = arr[j][i] = 1
        pa = [[1e9 for i in range(n+1)] for i in range(n+1)]
        for k in range(1,n+1):
            for i in range(1,n+1):
                for j in range(1,n+1):
                    if arr[i][k] + arr[k][j]<arr[i][j]:
                        arr[i][j] = arr[j][i] = arr[i][k] + arr[k][j]
                        pa[i][j] = pa[j][i] = k
        ans = [0 for i in range(n)]
        t = 0
        for t in range(2**n):
            temp = 0
            loc = []
            for i in range(n):
                if 1<<i&t:
                    loc.append(i+1)
            if len(loc)<2:
                continue
            vis = [0 for i in range(n)]
            flag = 0
            for i in range(len(loc)):
                for j in range(i+1,len(loc)):
                    if arr[loc[i]][loc[j]]>1 and not pa[loc[i]][loc[j]] in loc:
                        flag = 1
                    temp = max(temp,arr[loc[i]][loc[j]])
            if not flag and temp<n:
                ans[temp]+=1
                # print(loc,temp)
        return ans[1:]
