from typing import List, Dict, Tuple
import collections

class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        ans = [0] * (n - 1)

        new_edges = [[] for j in range(n + 1)]

        for a, b in edges:
            new_edges[a].append(b)
            new_edges[b].append(a)

        for i in range(1, 2 ** n):
            selected = []
            for j in range(n):
                if i >> j & 1:
                    selected.append(j + 1)

            setselected = set(selected)

            if len(selected) == 1:
                continue

            q = collections.deque()
            q.append(selected[0])
            v = set()
            v.add(selected[0])

            while q:
                ci = q.pop()
                for next in new_edges[ci]:
                    if next in setselected and next not in v:
                        v.add(next)
                        q.append(next)

            if len(v) != len(selected):
                continue

            ma = 0
            for start in selected:
                q = collections.deque()
                q.append((start, 0))
                v = set()
                v.add(start)

                while q:
                    ci, cd = q.pop()

                    ma = max(ma, cd)

                    for next in new_edges[ci]:
                        if next in setselected and next not in v:
                            v.add(next)
                            q.append((next, cd + 1))

            if ma != 0:
                ans[ma - 1] += 1

        return ans
