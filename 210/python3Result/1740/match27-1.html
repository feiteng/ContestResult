class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        ans = collections.Counter()
        e 
= collections.defaultdict(set)
        for u,v in edges:
            e[u].add(v)
            e[v].add(u)
            
        def solve(s):
            # print("s", s)
            if len(s) == 1: return 0
            ans = 0
            l = list(s)
            for a in l:
                for b in l:
                    if a == b: continue
                    # print(a,b)
                    seen = set()
                    q = collections.deque([(a,0)])
                    seen = {
a}
                    found = None
                    while q:
                        v, dist = q.popleft()
                        for nei in e[v]:
                            if nei not in seen and nei in s:
                                seen.add(nei)
                                q.append((nei,dist+1))
                                if nei == b:
                                    found = dist+1
                        if found is not None: break
                    # print(found)
                    if found is None: return 0
                    ans = max(ans, found)
            return ans
            
        for bm in range(1, 2**n):
            s = set()
            for i in range(n):
                if (bm >> i) & 1 == 1:
                    s.add(i+1)
            cnt = solve(s)
            ans[cnt] += 1
        return [ans[cnt] for cnt in range(1,n)]
