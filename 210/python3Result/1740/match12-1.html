from itertools import combinations
class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        a,b,s=[[] for _ in range(n)],[[0]*n for _ in range
(n)],[0]*n
        for i,j in edges:
            a[i-1].append(j-1)
            a[j-1].append(i-1)
        for i in range(n):
            aa,d=[i],[1]*n
            d[i],t=0,0
            while aa:
                bb,t=[],t+1
                for ii in aa:
                    for jj in a[ii]:
                        if d[jj]:
                            d[jj]=0
                            b[i][jj]=t
                            bb.append(jj)
                aa=bb
        for k in range(2,n+1):
            for x in combinations([i for i in range(n)],k):
                st,d=[x[0]],set(x)
                d.remove(x[0])
                
while st:
                    i=st.pop()
                    for j in a[i]:
                        if j in d:
                            d.remove(j)
                            st.append(j)
                if not d: s[max([b[i][j] for i in x for j in x])]+=1
        return s[1:]
