class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        graph = defaultdict(set)
        for i, j in edges:
            graph[i].add(j)
            graph[j].add(i)
            
        distance = {}
        def dfs(start, i, used, d):
            distance[(start, i)] = d
            for j in graph[i]:
                if not (1 << j & used):
                    dfs(start, j, used | (1 << j), d + 1)
        for i in range(1, n + 1):
            dfs(i, i, 1 << i, 0)
            
        
        def get_nodes(tree):
            result = []
            i = 1
            while tree:
                if tree & 1:
                    result.append(i)
                tree >>= 1
                i += 1
            return result
        
        def dfs(i, tree):
            visited.add(i)
            for j in graph[i]:
                if j not in visited and tree & 1 << (j - 1):
                    dfs(j, tree)
                    
        
        result = [0] * (n - 1)
        for tree in range(1, 2 ** n):
            nodes = get_nodes(tree)
            if len(nodes) == 1:
                continue
            visited = set()
            dfs(nodes[0], tree)
            if len(visited) < len(nodes):
                continue
            max_distance = 0
            for i in range(len(nodes)):
                for j in range(i + 1, len(nodes)):
                    max_distance = max(max_distance, distance[(nodes[i], nodes[j])])
            result[max_distance - 1] += 1
        return result
