class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        res = [0] * (n-1)
        
        def dfs(u,w,g) :
            r = 0
            for v in g[u] :
                if v==w :
                    continue
                r = max(dfs(v,u,g)+1,r)
            return r
            
        def go(s,g) :
            r = 0
            for u in s :
                r = max(dfs(u,-1,g),r)
            return r
        
        for mask in range(1<<n) :
            s = set()
            for u in range(n) :
                if (mask>>u) & 1 == 1 :
                    s.add(u)
            if len(s) <= 1 :
                continue
            g = [[] for i in range(n)]
            tot = 0
            for u,v in edges :
                u -= 1
                v -= 1
                if u in s and v in s :
                    g[u].append(v)
                    g[v].append(u)
                    tot += 1
            if tot < len(s) - 1 :
                continue
            
            ml = go(s,g)
            # print(ml,s,g)
            res[ml-1] += 1
        return res
