class Solution:
    
    def longest(self, node, nodes, g):  
        max_dist = 0
        max_node = node
        #print(node, nodes)
        for nei in g[node]:
            if nei in nodes:
                nodes.remove(nei)
                dist, ans_node = self.longest(nei, nodes, g)
                dist += 1
                if dist > max_dist:
                    max_dist = dist
                    max_node = ans_node
        return max_dist, max_node
    
    def longest_path(self, node, nodes, g):
        #print(nodes)
        n = len(nodes)
        nodes_copy = nodes.copy()
        nodes.remove(node)
        _, far_node = self.longest(node, nodes, g)
        if far_node in nodes_copy:
            nodes_copy.remove(far_node)
        else:
            return 0
        ans, _ = self.longest(far_node, nodes_copy, g)
        if not nodes and not nodes_copy:
            return ans
        
return 0
    
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        ans = [0]*(n-1)
        g = defaultdict(list)
        for u, v in edges:
            g[u].append(v)
            g[v].append(u)
            
        
        for mask in range(1, 2**n):
            nodes = set()
            start_node = -1
            for i in range(n):
                if (1 << i) & mask:
                    nodes.add(i+1)
                    start_node = i+1
            longest_path = self.longest_path(start_node, nodes, g)
            #print(longest_path)
            if longest_path > 0:
                ans[longest_path-1] += 1
        return ans
