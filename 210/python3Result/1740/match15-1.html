class Solution:
    def countSubgraphsForEachDiameter(self, n: int, E: List[List[int]]) -> List[int]:
        G = [i for i in range(n)]
        G_cnt = [1] * n
        def get_group(x) :
            if G[x] == x :
                return x
            G[x] = get_group(G[x])
            return G[x]

        def merge_group(x, y) :
            gx = get_group(x)
            gy = get_group(y)
            if gx == gy :
                return
            if gx > gy :
                gx, gy = gy, gx
            G[gx] = G[gy]
            G_cnt[gy] += G_cnt[gx]
        
        D = [[float('inf')] * n for _ in range(n)]
        for i in range(n) :
            D[i][i] = 0
        for e in E :
            u, v = e[0]-1, e[1]-1
            D[u][v] = D
[v][u] = 1
        
        for k in range(n) :
            for i in range(n) :
                for j in range(n) :
                    if D[i][k] + D[k][j] < D[i][j] :
                        D[i][j] = D[i][k] + D[k][j]
        
        ret = [0] * (n-1)
        for subset in range(1, 1 << n) :
            G = [i for i in range(n)]
            G_cnt = [1] * n
            for e in E :
                u, v = e[0]-1, e[1]-1
                if subset & (1 << u) and subset & (1 << v) :
                    merge_group(u, v)
            one_cnt = bin(subset).count('1')
            if one_cnt <= 1 :
                continue
            if max(G_cnt) != one_cnt :
                #print(bin(subset), G_cnt)
                continue
            
            max_val = 0
            for i in range(n) :
                if subset & (1 << i) :
                    for j in range(i+1, n) :
                        if subset & (1 << j) :
                            max_val = max(max_val, D[i][j])

            
            ret[max_val-1] += 1
        return ret
