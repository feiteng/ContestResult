class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        d = [[100] * n for _ in range(n)]
        p = [[[]for _ in range(n)]for _ in range(n)]
        for i,j in edges:
            d[i-1][j-1] = 1
            d[j-1][i-1] = 1
            p[i-1][j-1] = [i-1,j-1]
            p[j-1][i-1] = [i-1,j-1]
            
        for i in range(n):
            d[i][i] = 0
            p
[i][i] = [i]
            
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if d[i][j] > d[i][k] + d[k][j]:
                        d[i][j] = d[i][k] + d[k][j]
                        p
[i][j] = p[i][k] + p[k][j]
                        
        ret = [0] * (n-1)
        
        for i in range(1, 2**n):
            l = []
            for j in range(16):
                if i & (1 << j):
                    l.append(j)
            if len(l) <= 1:
                continue
                
            rm = 0
            f = 1
            for u in l:
                for v in l:
                    if u != v and f:
                        rm = max(rm, d[u][v])
                        for c in p[u][v]:
                            if not c in l:
                                f = 0
            if not f:
                continue
            
            # print(i,l,rm)
            
            if rm:
                ret[rm-1] += 1
        return ret
