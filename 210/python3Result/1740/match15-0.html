class Solution:
    def countSubgraphsForEachDiameter(self, n: int, edges: List[List[int]]) -> List[int]:
        res = [0]*(n-1)
        dd = [[10000000]*n for _ in range(n)]
        inc = [[] for _ in range(n)]
        for edge in edges:
            x,y = edge[0]-1,edge[1]-1
            dd[x][y] = 1
            dd[y][x] = 1
            inc[y].append(x)
            inc[x].append(y)
        for i in range(n):
            dd[i][i] = 0
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if dd[i][j]>dd[i][k]+dd[k][j]:
                        dd[i][j] = dd[i][k]+dd[k][j]
        # for d in dd:
        #     print(d)
        def bitmask(arr):
            res = 0
            for nu in arr:
                res+=(1<<nu)
            return res
        to = [[a[0]-1,a[1]-1] for a in edges]
        visi = set()
        while to:
            te = to.pop()
            mate = bitmask(te)
            if mate in visi:
                continue
            visi.add(mate)
            for tu in te:
                for ti in inc[tu]:
                    if ((1<<ti)&mate)==0:
                        to.append(te+[ti])
            mad = 0
            for i in range(1,len(te)):
                for j in range(i):
                    if dd[te[i]][te[j]]>mad:
                        mad = dd[te[i]][te[j]]
            # print(mad)
            res[mad-1] += 1
        return res
