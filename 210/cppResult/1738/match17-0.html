class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<int>> adjMat(n, vector<int>(n));
        vector<int> cnts(n);
        for (const auto &road : roads) {
            ++adjMat[road[0]][road[1]];
            ++adjMat[road[1]][road[0]];
            ++cnts[road[0]];
            ++cnts[road[1]];
        }
        int result = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                result = max(result, cnts[i] + cnts[j] - adjMat[i][j]);
            }
        }
        return result;
    }
};
