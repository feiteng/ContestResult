class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<set<int>> neighbors(n);
        for (auto &v: roads) {
            neighbors[v[0]].insert(v[1]);
            neighbors[v[1]].insert(v[0]);
        }
        
        int maxRank = 0;
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int rank = neighbors[i].size() + neighbors[j].size();
                if (neighbors[i].count(j)) rank -= 1;
                maxRank = max(maxRank, rank);
            }
        }
        
        return maxRank;
        
    }
};
