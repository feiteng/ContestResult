class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<set<int>> connected(n); 
        for (const auto& road : roads) {
            connected[road[0]].insert(road[1]);
            connected[road[1]].insert(road[0]);
        }
        int max_rank = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                int pair_rank = connected[i].size() + connected[j].size() - (connected[i].count(j) ? 1 : 0);
                max_rank = max(max_rank, pair_rank);
            }
        }
        return max_rank;
    }
};
