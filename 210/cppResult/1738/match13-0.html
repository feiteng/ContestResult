class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<unordered_set<int>> graph(n);
        for (auto &v: roads) {
            graph[v[0]].insert(v[1]);
            graph[v[1]].insert(v[0]);
        }
        
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                int count = graph[i].size() + graph[j].size();
                if (graph[i].count(j)) count--;
                res = max(count, res);
            }
        }
            
        return res;
    }
};
