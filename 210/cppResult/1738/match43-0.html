class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> d(n + 1);
        map<int, set<int> > g;
        for(auto road : roads) {
            int a = road[0], b = road[1];
            g[a].insert(b);
            g[b].insert(a);
            ++d[a]; ++d[b];
        }
        int ans = 0;
        for(int i = 0; i < n; ++i) {
            for(int j = i + 1; j < n; ++j) {
                ans = max(ans, d[i] + d[j] - (g[i].count(j) > 0));
            }
        }
        return ans;
    }
};
