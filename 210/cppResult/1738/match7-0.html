class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        unordered_map<int, unordered_set<int>> e;
        for(auto vec: roads) {
            e[vec[0]].insert(vec[1]);
            e[vec[1]].insert(vec[0]);
        }
        int ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                int tmp = e[i].size() + e[j].size();
                if (e[i].count(j) > 0)
                    tmp -= 1;
                ans = max(ans, tmp);
            }
        }
        return ans;
    }
};
