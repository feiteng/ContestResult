class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        unordered_map<int,set<int>> mp;
        for(auto v:roads){
            mp[v[0]].insert(v[1]);
            mp[v[1]].insert(v[0]);
        }
        int ans=INT_MIN;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int cur=mp[i].size()+mp[j].size();
                if(mp[i].count(j))
                    cur--;
                ans=max(ans,cur);
            }
        }
        return ans;
    }
};
