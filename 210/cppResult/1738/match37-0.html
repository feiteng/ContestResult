class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<int>> connected(n, vector<int>(n));
        vector<int> cnt(n);
        for (auto & road : roads) {
            connected[road[0]][road[1]] = connected[road[1]][road[0]] = -1;
            cnt[road[0]] ++;
            cnt[road[1]] ++;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                ans = max(ans, cnt[i] + cnt[j] + connected[i][j]);
            }
        }
        return ans;
    }
};
