class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<int>> link(n);
        vector<unordered_set<int>> l2(n);
        for(auto v:roads){
            link[v[0]].push_back(v[1]);
            link[v[1]].push_back(v[0]);
            l2[v[0]].insert(v[1]);
            l2[v[1]].insert(v[0]);
        }
        int ans=0;
        for(int i=0;i+1<n;++i){
            for(int j=i+1;j<n;++j){
                int cur=link[i].size()+link[j].size();
                if(l2[i].find(j)!=l2[i].end())
                    --cur;
                ans=max(ans,cur);
            }
        }
        return ans;
    }
};
