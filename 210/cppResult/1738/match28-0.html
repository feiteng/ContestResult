class Solution {
public:
    int degree[105];
    bool g[105][105];
    int maximalNetworkRank(int n, vector<vector<int>>& r) {
        memset(degree, 0, sizeof(degree));
        memset(g, false, sizeof(g));
        
        for (int i = 0; i < r.size(); i++) {
            int a, b;
            a = r[i][0];
            b = r[i][1];
            
            degree[a]++;
            degree[b]++;
            g[a][b] = g[b][a] = true;
        }
        
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int total = degree[i] + degree[j];
                if (g[i][j]) total--;
                res = max(res, total);
            }
        }
        return res;
    }
};
