class Solution {
public:
    int a[105][105],b[105];
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
        for(int i = 0;i < roads.size();i++) {
            int x = roads[i][0],y = roads[i][1];
            b[x]++;b[y]++;
            a[x][y] = a[y][x] = 1;
        }
        int ans = 0;
        for(int i = 0;i < n;i++) {
            for(int j = i + 1;j < n;j++) {
                int num = b[i] + b[j];
                if(a[i][j]) num--;
                ans = max(ans,num);
            }
        }
        return ans;
    }
};
