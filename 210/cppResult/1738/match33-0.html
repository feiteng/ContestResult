class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> count(n);
        vector<vector<int>> has_road(n, vector<int>(n, 0));
        for(auto &r: roads) {
            if(r[0] > r[1]) {
                swap(r[0], r[1]);
            }
            count[r[0]]++;
            count[r[1]]++;
            has_road[r[0]][r[1]] = 1;
            has_road[r[1]][r[0]] = 1;
        }
        
        int ans = 0;
        for(int i=0;i<n;++i) {
            for(int j=i+1;j<n;++j) {
                int rank = count[i] + count[j] - has_road[i][j];
                ans = max(rank, ans);
            }
        }
        return ans;
    }
};
