class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> dp(n, 0);
        vector<vector<int>> mp(n, vector<int>(n, 0));
        for (auto temp: roads) {
            dp[temp[0]]++;
            dp[temp[1]]++;
            mp[temp[0]][temp[1]] = 1;
            mp[temp[1]][temp[0]] = 1;
        }
        int ans = 0;
        for (int i = 0;i < n - 1;i++) {
            for (int j = i + 1;j < n;j++) {
                int temp = dp[i] + dp[j] - mp[i][j];
                ans = max(ans, temp);
            }
        }
        return ans;
    }
};
