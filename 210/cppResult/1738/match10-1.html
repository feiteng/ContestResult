class Solution {
 public:
  int maximalNetworkRank(int n, vector<vector<int>>& roads) {
    vector<vector<int>> edges(n, vector<int> (n, 0));
    vector<int> rank(n, 0);
    for (auto &road : roads) {
      int a = road[0], b = road[1];
      edges[a][b] = 1;
      edges[b][a] = 1;
      rank[a]++;
      rank[b]++;
    }
    int res = 0;
    for (int i = 0; i < n; i++) {
      for (int j = i + 1; j < n; j++) {
        res = max(rank[i] + rank[j] - edges[i][j], res);
      }
    }
    return res;
  }
};
