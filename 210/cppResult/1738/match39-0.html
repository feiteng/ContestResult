class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& a) {
        vector<int> degree(n);
        vector<vector<int>> adj(n,vector<int>(n));
        for(auto &i:a){
            degree[i[0]]++;
            degree[i[1]]++;
            adj[i[0]][i[1]]=1;
            adj[i[1]][i[0]]=1;
        }
        int max1=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                max1=max(max1,degree[i]+degree[j]-adj[i][j]);
            }
        }
        return max1;
    }
};
