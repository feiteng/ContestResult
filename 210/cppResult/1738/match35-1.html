class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        set<int> G[105];
        int deg[105];
        memset(deg, 0, sizeof deg);
        for(auto& e:roads){
            G[e[0]].insert(e[1]);
            G[e[1]].insert(e[0]);

            deg[e[0]]++; deg[e[1]]++;
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            for(int j = i+1; j < n; j++){
                int s = deg[i] + deg[j] + (G[i].find(j) == G[i].end() ? 0:-1);
                res = max(res, s);
            }
        }
        return res;
    }
};
