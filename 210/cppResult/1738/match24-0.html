class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> cnt(n); vector<vector<bool>> g(n, vector<bool>(n));
        for (auto i : roads) {
            cnt[i[0]]++; cnt[i[1]]++; g[i[0]][i[1]] = g[i[1]][i[0]] = 1;
        }
        int r = 0;
        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                r = max(r, cnt[i] + cnt[j] - g[i][j]);
        return r;
    }
};
