class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<bool>> matrix(n, vector<bool>(n));
        vector<int> deg(n);
        for (auto &road : roads) {
            deg[road[0]]++;
            deg[road[1]]++;
            matrix[road[0]][road[1]] = true;
            matrix[road[1]][road[0]] = true;
        }
        
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                int rank = deg[i] + deg[j];
                if (matrix[i][j]) {
                    --rank;
                }
                res = max(res, rank);
            }
        }
        return res;
    }
};
