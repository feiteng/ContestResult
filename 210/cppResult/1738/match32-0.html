class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> cnt(n);
        vector<vector<int>> bet(n, vector<int>(n));
        for (auto x : roads) {
            int a = x[0], b = x[1];
            ++cnt[a], ++cnt[b];
            ++bet[a][b], ++bet[b][a];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            //cout << cnt[i] << " ";
            for (int j = i + 1; j < n; ++j) {
                int cur = cnt[i] + cnt[j] - bet[i][j];
                ans = max(ans, cur);
            }
        }
        return ans;
    }
};
