class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<int>> edge(n,vector<int>(n,0)); vector<int> degree(n,0);
        for(auto road:roads){
            degree[road[0]]++; degree[road[1]]++; 
            edge[road[0]][road[1]]=1; edge[road[1]][road[0]]=1;
        }
        int ans=0;
        for(int i=0; i<n-1; i++){
            for(int j=i+1; j<n; j++){
                int rank = degree[i]+degree[j];
                if(edge[i][j]) rank--;
                ans = max(ans,rank);
            }
        }
        return ans;
    }
};
