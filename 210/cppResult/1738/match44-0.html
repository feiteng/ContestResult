class Solution {
public:
	int in[101];
	int link[101][101];

    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        memset(in, 0, sizeof(in));
        memset(link, 0, sizeof(link));
        for(auto& e : roads) {
        	int a = e[0], b = e[1];
        	in[a]++;
        	in[b]++;
        	link[a][b] = 1;
        	link[b][a] = 1;
        }
        int res = 0;
        for(int i = 0; i < n; i++)
        	for(int j = i+1; j < n; j++)
        		res = max(res, in[i] + in[j] - link[i][j]);
        return res;
    }
};
