class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> cityRoadCount(n, 0);
        vector<vector<int>> betweenCitiesCount(n, vector<int>(n, 0));
        for (auto &road : roads) {
            int from = road[0];
            int to = road[1];
            cityRoadCount[from]++;
            cityRoadCount[to]++;
            betweenCitiesCount[from][to]++;
            betweenCitiesCount[to][from]++;
        }
        int result = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int nowResult = cityRoadCount[i] + cityRoadCount[j] - betweenCitiesCount[i][j];
                result = max(result, nowResult);
            }
        }
        return result;
    }
};
