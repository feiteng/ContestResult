class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        int cnt[105];
        fill(cnt,cnt+n+1,0);
        set<pair<int,int> > s;
        for(auto it:roads){
            s.insert(make_pair(it[0],it[1]));
            cnt[it[0]]++;
            cnt[it[1]]++;
            s.insert(make_pair(it[1],it[0]));
        }
        int ans=0;
        for(int i=0;i<n;i++){
            for(int j = i+1;j<n;j++){
                int tot=cnt[i]+cnt[j];
                if(s.find(make_pair(i,j))!=s.end())tot--;
                ans=max(ans,tot);
            }
        }
        return ans;
    }
};
