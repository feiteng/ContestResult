class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<vector<bool>> c(n,vector<bool>(n));
        vector<int> a(n);
        for(auto& q:roads){
            c[q[0]][q[1]]=true;
            c[q[1]][q[0]]=true;
            ++a[q[0]];
            ++a[q[1]];
        }
        int res=0;
        for(int i=0;i<n;++i)
            for(int j=i+1;j<n;++j)
                res=max(res,a[i]+a[j]-c[i][j]);
        return res;
    }
};
