class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int>inCounts(n,0);
        unordered_map<int,unordered_set<int>>memo;
        for(auto road:roads)
        {

            inCounts[road[0]]++;
            inCounts[road[1]]++;
            memo[road[0]].insert(road[1]);
            memo[road[1]].insert(road[0]);
        }
        int maxResult = 0;
        for(int i = 0;i < n;i++)
        {
            for(int j =i+1; j < n;j++)
            {
                int temp = inCounts[i] + inCounts[j];
                if(memo.count(i) >0 && memo[i].count(j)>0)
                    temp-=1;
                maxResult = max(maxResult,temp);
            }
        }
        
        return maxResult;
    }
};
