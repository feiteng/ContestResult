class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> g(n, 0);
        vector<vector<bool>> connected(n, vector<bool>(n, false));
        for(auto &v : roads){
            ++g[v[0]];
            ++g[v[1]];
            connected[v[0]][v[1]] = true;
            connected[v[1]][v[0]] = true;
        }
        int ans = 0;
        for(int i = 0; i < n; ++i){
            for(int j = i+1; j < n; ++j){
                int cnt = g[i] + g[j];
                if(connected[i][j]){
                    --cnt;
                }
                ans = max(cnt, ans);
            }
        }
        return ans;
    }
};
