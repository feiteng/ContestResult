class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> nr(n, 0);
        vector<vector<int>> binr(n, vector<int>(n, 0));
        for (const auto& r : roads) {
            nr[r[0]]++;
            nr[r[1]]++;
            binr[r[0]][r[1]]++;
            binr[r[1]][r[0]]++;
        }
        int max_r = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                max_r = max(nr[i] + nr[j] - binr[i][j], max_r);
            }
        }
        return max_r;
    }
};
