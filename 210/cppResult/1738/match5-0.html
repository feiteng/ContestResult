class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        set<int> adj[n];
        for(auto it : roads)
        {
            adj[it[0]].insert(it[1]);
            adj[it[1]].insert(it[0]);
        }
        
        int ans = 0;
        
        for(int i = 0; i + 1<n; i++)
        {
            for(int j = i + 1; j<n; j++)
            {
                int cur = adj[i].size() + adj[j].size();
                if (adj[i].find(j) != adj[i].end())
                    cur--;
                
                ans = max(ans, cur);
            }
        }
        
        return ans;
    }
};
