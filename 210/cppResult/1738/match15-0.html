class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<int> edges(n, 0);
        vector<vector<int>> graph(n, vector<int>(n, 0));
        for (const auto& r : roads) {
            edges[r[0]]++;
            edges[r[1]]++;
            graph[r[0]][r[1]]++;
            graph[r[1]][r[0]]++;
        }
        int ret = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                ret = max(ret, edges[i] + edges[j] - graph[i][j]);
            }
        }
        return ret;
    }
};
