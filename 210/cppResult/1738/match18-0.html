class Solution {
public:
    // Store the edge and calculate all pairs brute-force
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<set<int>> edges(n);
        for (int i = 0; i < roads.size(); ++i) {
            edges[roads[i][0]].insert(roads[i][1]);
            edges[roads[i][1]].insert(roads[i][0]);
        }
        int answer = 0;
        for (int i = 0; i < n-1; ++i) {
            for (int j = i+1; j < n; ++j) {
                int temp = edges[i].size() + edges[j].size();
                if (edges[i].find(j) != edges[i].end()) temp--;
                answer = max(answer, temp);
            }
        }
        return answer;
    }
};
