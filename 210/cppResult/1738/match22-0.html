class Solution {
public:
    int maximalNetworkRank(int n, vector<vector<int>>& roads) {
        vector<unordered_set<int>> adj_list(n);
        for (auto it : roads) {
            adj_list[it[0]].insert(it[1]);
            adj_list[it[1]].insert(it[0]);
        }
        int best = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                int count = adj_list[i].size() + adj_list[j].size();
                if (adj_list[i].find(j) != adj_list[i].end())
                    count--;
                best = max(best, count);
            }
        }
        return best; 
    }
};
