class Solution {
public:
  int maximalNetworkRank(int n, vector<vector<int>>& roads) {
    vector<int> s(n, 0);
    vector<vector<int>> p(n, vector<int>(n, 0));
    for (auto r : roads) {

      ++s[r[0]];
      ++s[r[1]];
      ++p[r[0]][r[1]];
      ++p[r[1]][r[0]];
    }
    int best =0;
    for (int i=0; i<n; ++i) {
      for (int j=i+1; j<n; ++j) {
        int tmp = s[i] + s[j] - p[i][j];
        best = max(best, tmp);
      }
    }
    return best;
  }
};
