class Solution {
public:
    int maxDepth(string s) {
        int len = s.length();
        int ans = 0;
        int cnt = 0;
        for (int i = 0; i < len; ++i) {
            if (s[i] == '(') cnt += 1;
            else if (s[i] == ')') cnt -= 1;
            ans = max(ans, cnt);
        }
        return ans;
    }
};
