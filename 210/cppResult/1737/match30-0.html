class Solution {
public:
    int maxDepth(string s) {
        int res = 0;
        int cnt = 0;
        
        for (auto ch : s)
        {
            if (ch == '(')
            {
                ++cnt;
                res = max(res, cnt);
            }
            else if (ch == ')')
            {
                --cnt;
            }
        }
        return res;
    }
};
