class Solution {
public:
    int maxDepth(string s) {
        int result = 0;
        int counter = 0;
        for (auto ch: s) {
            if (ch == '(') {
                counter++;
                result = max(result, counter);
            } else if (ch == ')') {
                counter--;
            }
        }
        
        return result;
    }
};
