class Solution {
public:
    bool isPalindrome(string &str)
    {
        int i=0, j=str.length()-1;
        while(i<j)
        {
            if(str[i]!=str[j])
                return false;
            i++, j--;
        }
        return true;
    }
    
    bool checkPalindromeFormation(string a, string b) {
        if(isPalindrome(a) || isPalindrome(b))
            return true;
        
        int i=0, j=b.length()-1;
        if(a[i] == b[j])
            return true;
        
        i=a.length()-1, j=0;
        if(a[i] == b[j])
            return true;
        
        return false;
    }
};
