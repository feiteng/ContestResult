class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int n = a.size();
        bool a1 = true, b1 = true;
        for(int i=0; i<n/2; i++) {
            if (a[i] != a[n-i-1]) a1 = false;
            if (b[i] != b[n-i-1]) b1 = false;
        }
        if (a1 || b1) return true;
        if (a[0] == b[n-1] || a[n-1] == b[0]) return true;
        return false;
    }
};
