class Solution {
    bool ok(string a,int l,int r)
    {

        while(l<r)
        {
            if(a[l]!=a[r])
                return 0;
            l++;
            r--;
        }
        return 1;
    }
    bool check(string a,string b)
    {
        int l=0;

        int r=a.size()-1;
        while(l<r)
        {
            if(a[l]!=b[r])
                break;
            l++;
            r--;
        }
        if(l>=r)return 1;
        return ok(a,l,r)||ok(b,l,r);
    }
public:
    bool checkPalindromeFormation(string a, string b) {
        return check(a,b)||check(b,a);
    }
};
