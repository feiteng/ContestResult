class Solution {
public:
    bool isPalindrome(string s){
      int i = 0;
      int j = s.size() - 1;
      while(i < j){
          if(s[i] != s[j]) return false;
          //cout << s << " " << i << " " << j << en
          i++;
          j--;
      }
      //cout << s << endl;
      return true;
  }
    bool solve(string a, string b){
        if(isPalindrome(b) || isPalindrome(a)) return true;
      int n = a.size();
      int i = 0;
      int j = n - 1;
      while(i < j){
          if(a[i] != b[j]) break;
          i++;
          j--;
      }
      //cout << i << " -- " << j << endl;
      if(i >= j) return true;
      if(isPalindrome(a.substr(i, n - (i +(n - j) - 1)))) return true;
      if(isPalindrome(b.substr(i, n - (i +(n - j) - 1)))) return true;
      return false;
        
    }
    bool checkPalindromeFormation(string a, string b) {
        return solve(a, b) || solve(b, a);
         
    }
};
