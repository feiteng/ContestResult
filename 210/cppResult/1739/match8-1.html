class Solution {
public:
    bool check(string& s) {
        int i = 0, j = s.size()-1;
        while (i < j) {
            if (s[i] != s[j]) return false;
            i++, j--;
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        if (check(a) || check(b)) return true;
        int len1 = a.size(), len2 = b.size();
        if (a[0] == b[len2-1] || a[len1-1] == b[0]) return true;
        return false;
    }
};
