class Solution {
public:
    bool isPalindrome(string &s) {
        int n = s.length();
        int i = 0;
        int j = n-1;
        while(i <= j) {
            if(s[i] == s[j]) {
                i++;
                j--;
            } else {
                return false;
            }
        }
        return true;
    }
    
    bool myPalindrome(string &a, string &b) {
        int n = a.length();
        int i = 0;
        int j = n-1;
        while(i <= j) {
            if(a[i] == b[j]) {
                i++;
                j--;
            } else {
                break;
            }
        }
        // either rest in b or a
        int t_i = i;
        int t_j = j;
        while(i<=j) {
            if(b[i] == b[j]) {
                i++;
                j--;
            } else {
                break;
            }
        }
        if(i > j) return true;
        i = t_i;
        j = t_j;
        while(i <= j) {
            if(a[i] == a[j]) {
                i++;
                j--;
            } else {
                return false;
            }
        }
        return true;
    }
