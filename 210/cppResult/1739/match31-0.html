class Solution {
public:
    bool check(string str){
        int n = str.size();
        for(int i=0;i<n/2;++i){
            if(str[i]!=str[n-1-i])
                return false;
        }
        return true;
    }
    bool hlp(string a, string b){
        int n = a.size();
        int idx1=-1;
        int idx2=-1;
        for(int i=(n/2)-1;i>=0;--i){
            if(a[i]!=a[n-1-i]){
                idx1 = i;
                break;
            }
        }
        for(int i=(n/2)-1;i>=0;--i){
            if(b[i]!=b[n-1-i]){
                idx2 = i;
                break;
            }
        }
        int idx = min(idx1,idx2);
        if(idx==-1)
            return true;
        bool ok1 = true;
        bool ok2 = true;
        for(int i=idx1;i>=0;--i){
            if(a[i]!=b[n-1-i])
                ok1 = false;
        }
        for(int i=idx2;i>=0;--i){
            if(b[i]!=a[n-1-i])
                ok2 = false;
        }
        if(ok1 || ok2)
            return true;
        return false;
    }
    bool checkPalindromeFormation(string a, string b) {
        if(hlp(a,b))
            return true;
        reverse(a.begin(),a.end());
        reverse(b.begin(),b.end());
        if(hlp(a,b))
            return true;
        return false;
    }
};
