class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(a.size() == 1) return 1;
        int tot=0,n=a.size()-1;
        for(int i=0;i<=n;i++){
            if(a[i]==b[n-i])
                tot++;
            else 
                break;
        }
        if(tot>0) return true;
        tot=0;
        for(int i=0;i<=n;i++){
            if(b[i]==a[n-i])
                tot++;
            else 
                break;
        }
        if(tot>0) return 1; 
        return 0;
    }
};
