class Solution {
public:
    bool pal(string s)
    {
        int i=0,j=s.length()-1;
        while(i<j)
        {
            if(s[i]!=s[j])
                return false;
            i++,j--;
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        int n = a.length(), m = b.length();
        
        if(n<2 || m<2)
            return true;
        if(a[0] == b[m-1] || a[n-1] == b[0])
            return true;
        if(pal(a) || pal(b))
            return true;
        return false;
    }
};
