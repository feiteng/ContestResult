class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int n = a.length();
        bool isA, isB;
        isA = isB = true;
        for (int i=0; i<n/2; i++) {
            if (a[i] != a[n-i-1]) isA = false;
            if (b[i] != b[n-i-1]) isB = false;
        }
        if (isA || isB) return true;
        if (a[0]==b[n-1]) return true;
        if (a[n-1]==b[0]) return true;
        return false;
    }
};
