class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
      if (check(a) || check(b)) return true;
      int i = 0, j = b.length()-1;
      for (; i < j && a[i] == b[j]; ++i, --j);
      if (i >= j) return true;
      if (check(a.substr(i, j-i+1)) || check(b.substr(i, j-i+1))) return true;
      i = 0, j = a.length()-1;
      for (; i < j && b[i] == a[j]; ++i, --j);
      if (i >= j) return true;
      if (check(a.substr(i, j-i+1)) || check(b.substr(i, j-i+1))) return true;
      return false;
    }
private:
    bool check(const string& s) {
      int i = 0, j = s.length()-1;
      while (i < j) {
        if (s[i] != s[j]) return false;
        ++i; --j;
      }
      return true;
    }
};
