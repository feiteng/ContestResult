class Solution {
public:
    bool check1(string a){
        int l = a.length();
        for(int i=0;i<=l/2;i++)
            if(a[i]!=a[l-i-1]) return false;
        return true;
    }
    bool check(string a,string b){
        if(a[0]==b[b.length()-1]) return true;
        return false;
    }
    
    bool checkPalindromeFormation(string a, string b) {
        if(check1(a)||check1(b)) return true;
        if(check(a,b) ||check(b,a)) return true;
        return false;
    }
};
