class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        bool f = false, f1 = true, f2 = true, f3, f4;
        int len = a.size();
        for (int i = 0; i < len; i++) {
            if(a[i] != a[len - 1 - i]) f1 = false;
            if(b[i] != b[len - 1 - i]) f2 = false;
        }
        if(f1 || f2) f = true;
        if(f) return true;
        int pos = 0;
        f1 = true; f2 = true, f3 = true, f4 = true;
        for (int i = 0; i < len / 2; i++) {
//            printf("%c %c\n", a[i], b[len - i - 1]);
            if(a[i] != b[len - i - 1]) {
                pos = i;
                break;
            }
            pos++;
        }
//        printf("pos = %d\n", pos);
        for (int i = pos; i < len / 2; i++) {
            if(b[i] != b[len - i - 1]) f1 = false;
        }
        for (int i = pos; i < len / 2; i++) {
            if(a[i] != a[len - i - 1]) f2 = false;
        }
        pos = 0;
        for (int i = 0; i < len / 2; i++) {
            if(b[i] != a[len - i - 1]) {
                pos = i;
                break;
            }
            pos++;
        }
        for (int i = pos; i < len / 2; i++) {
            if(a[i] != a[len - i - 1]) f3 = false;
        }
        for (int i = pos; i < len / 2; i++) {
            if(b[i] != b[len - i - 1]) f4 = false;
        }
        if(f1 || f2 || f3 || f4) f = true;
        return f;

    }
}pp;
