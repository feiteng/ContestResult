class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        return check(a, b) || check(b, a); 
    }
private:

    bool check(string& a, string& b) {
        int i = 0, j = a.size() - 1, sz = a.size();
        while (a[i] == b[j]) ++i, --j;
        if (i >= sz / 2) return true;
        if (isPal(a.substr(i, j - i + 1))) return true;
        if (isPal(b.substr(i, j - i + 1))) return true;
        return false;
    }
    

    bool isPal(string s) {
        int i = 0, j = s.size() - 1;
        while (i <= j) {
            if (s[i] != s[j]) return false;
            ++i, --j;
        } 
        return true;
    }
};
