class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {

        int l=0,r=a.size()-1,pass=1;
        while(l<r){
            if(a[l]!=a[r]){
                pass=false;break;
            }
            l++;
            r--;
        }

        if(pass) return true;
        l=0,r=b.size()-1,pass=1;
        while(l<r){
            if(b[l]!=b[r]){
                pass=false;break;
            }
            l++;
            r--;
        }
        if(pass) return true;
        if(a[0]==b.back())return true;
        if(a.back()==(*b.begin())) return true;
        return false;
    }
};
