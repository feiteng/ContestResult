class Solution {
public:
    bool check(string s)
    {
        int left=0,right=s.size()-1;
        while (left<right)
        {
            if (s[left]!=s[right])
                return false;
            left++;
            right--;
        }
        return true;
    }
    bool helper(string a, string b)
    {
        int left=0,right=a.size()-1;
        while (left<right)
        {
            if (a[left]!=b[right])
                break;
            left++;
            right--;
        }
        if (check(a.substr(left,right-left+1)) || check(b.substr(left,right-left+1)))
            return true;
        return false;
    }
    bool checkPalindromeFormation(string a, string b) {
        if (check(a) || check(b))
            return true;
        return helper(a,b) || helper(b,a);
    }
};
