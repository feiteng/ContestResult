int is_ok(string &s, int i, int j) {
    for (int k = i; k <= j; k++) {
        if (s[k] != s[j - (k - i)]) {
            return false;
        }
    }
    return true;
}

class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int len = a.size();
        int m = len / 2;
        
        // printf("%d\n", len);
        
        // a p, b s
        int i;
        for (i = 0; i < m; i++) {
            if (a[i] != b[len - i - 1]) {
                break;
            }
        }

        if (is_ok(a, i, len - i - 1) || is_ok(b, i, len - i - 1)) {
            return true;
        }
        
        // b p, a s
        // int i;
        for (i = 0; i < m; i++) {
            // printf("%d\n", i);
            if (b[i] != a[len - i - 1]) {
                break;
            }
        }
        
        if (is_ok(a, i, len - i - 1) || is_ok(b, i, len - i - 1)) {
            return true;
        }

        return false;
    }
