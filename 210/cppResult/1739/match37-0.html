class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if (isPalin(a) || isPalin(b)) return true;
        if (a[0] == b[b.size() - 1] || a[a.size()-1] == b[0]) return true;
        return false;
    }
    bool isPalin(string& a) {
        int i = 0, j = a.size() - 1;
        while (i < j) {
            if (a[i++] != a[j--]) return false;
        }
        return true;
    }
};
