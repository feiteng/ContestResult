class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(a.length() == 1) return true;
        bool A = true, B = true;
        int N = a.length();
        cout << a << endl;
        for(int i = 0 ; i < N ; i++)
        {
            if(a[i] != b[N - i - 1])
            {
                for(int j = i ; j <= N - i - 1 ; j++) if(b[j] != b[N - j - 1]) {B = false; break;}
                if(B) return true;
                B = true;
                for(int j = i ; j <= N - i - 1 ; j++) if(a[j] != a[N - j - 1]) {B = false; break;}
                if(B) return true;
                break;
            }
        }
        for(int i = 0 ; i < N ; i++)
        {
            if(b[i] != a[N - i - 1])
            {
                for(int j = i ; j <= N - i - 1 ; j++) if(a[j] != a[N - j - 1]) {A = false; break;}
                if(A) return true;
                A = true;
                for(int j = i ; j <= N - i - 1 ; j++) if(b[j] != b[N - j - 1]) {A = false; break;}
                if(A) return true;
                break;
            }
        }
        return false;
    }
};
