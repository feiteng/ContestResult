const int maxn=1e5+5;
class Solution {
public:
    int x[maxn], y[maxn];
bool checkPalindromeFormation(string a, string b) {
	int n = a.size();
	int flag = 1;
	for (int i = 0;i<n;i++) {
		if (a[i] == b[n - 1 - i] && flag) x[i] = 1, y[n - 1 - i] = 1;
		if (a[i] != b[n - 1 - i]) {
			flag = 0;
			break;
		}
	}
	flag = 1;
	for (int i = 0;i<n;i++) {
		if (b[i] == a[n - 1 - i] && flag) x[i] = 1, y[n - 1 - i] = 1;
		if (b[i] != a[n - 1 - i]) {
			flag = 0;
			break;
		}
	}
	bool ans = false;
	int f1 = 1, f2 = 1;
	string a2 = a, b2 = b;
	reverse(a.begin(), a.end());
	reverse(b.begin(), b.end());
	for (int i = 0;i<n;i++) {
		if (a[i] != a2[i]) {
			f1 = 0;
		}
		if (b[i] != b2[i]) {
			f2 = 0;
		}
	}
	if (f1 || f2) ans = true;
	for (int i = 0;i<n;i++) {
		if (x[i] == 1 && y[n - 1 - i] == 1) {
			ans = true;
		}
	}
	return ans;
}
};
