class Solution {
public:
    bool isPalindrome(string &a, int i, int j)
    {
        while( i < j )
        {
            if(a[i] != a[j])
                return false;
            
            i++;
            j--;
        }
        
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        if(isPalindrome(a,0,a.size()-1) || isPalindrome(b,0,b.size()-1))
            return true;
        
        
        int i=0, j= b.size()-1;
        bool palindrome = true;
        while( i<j )
        {
            if(a[i]!= b[j])
            {
                palindrome = false;
                break;
            }
            i++;
            j--;
        }
        
        if(palindrome == true )
            return true;
        
        if(isPalindrome(a, i, j) || isPalindrome(b,i,j))
            return true;
        
        
        i =0, j = a.size()-1;
        palindrome = true;
        while( i < j )
        {
            if(b[i] != a[j])
            {
                palindrome = false;
                break;
            }
            i++;
            j--;
        }
        
        if(palindrome == true)
            return true;
        
        if(isPalindrome(a, i, j ) || isPalindrome(b,i,j))
            return true;
        
        return false;
    }
};
