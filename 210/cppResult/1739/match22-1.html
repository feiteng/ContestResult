class Solution {
public:
    bool fun(string &a,int i,int j){
          while(i<j)
          { if(a[i]!=a[j])
              return 0;
           i++;j--;
          }
        return 1;
    }
    bool checkPalindromeFormation(string a, string b) {

          int i,j,n=a.size(),f=0;
           i=0;j=n-1;
           while(i<j)
           {   if(a[i]==b[j]&&f==0)
           {  i++;
               j--;
           }
            else {
                 if(fun(a,i,j)||fun(b,i,j))
                     return 1;
                break;
            }
           }
           if(i>=j)
               return 1;
        f=0;i=0;j=n-1;

          while(i<j)
          {  if(f==0&&b[i]==a[j])
          {  i++;
              j--;
          }
           else{
                if(fun(a,i,j)||fun(b,i,j))
                    return 1;
               break;
           }
          }
         if(i>=j)
             return 1;
        return 0;
    }
};
