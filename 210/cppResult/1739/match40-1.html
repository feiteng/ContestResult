class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {

        int len = a.length();
        if(a[0]==b[len-1]||a[len-1]==b[0])return true;
        return isPal(a)||isPal(b);
    }
    bool isPal(string &x){
        int len = x.length();
        for(int i = 0;i<len;++i){
            if(x[i]!=x[len-i-1])return false;
        }
        return true;
    }
};
