class Solution {
public:
    bool ispalin(string a){
        int i=0,j=a.length()-1;
        while(i<j){
            if(a[i]!=a[j])
                return false;
            i++;
            j--;
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        if(ispalin(a) or ispalin(b))
            return true;
        int i=0,j=a.length()-1;
        bool ans=true;
        while(i<j){
            if(a[i]!=b[j] and b[i]!=b[j]){
                ans=false;
                break;
            }
            i++;
            j--;
        }
        if(ans)
            return ans;
        ans=true;
        i=0,j=a.length()-1;
        while(i<j){
            if(a[i]!=b[j] and a[i]!=a[j]){
                ans=false;
                break;
            }
            i++;
            j--;
        }
        if(ans)
            return ans;
        ans=true;
        i=0,j=a.length()-1;
        while(i<j){
            if(b[i]!=a[j] and b[i]!=b[j]){
                ans=false;
                break;
            }
            i++;
            j--;
        }
        if(ans)
            return ans;
        ans=true;
        i=0,j=a.length()-1;
        while(i<j){
            if(b[i]!=a[j] and a[i]!=a[j]){
                ans=false;
                break;
            }
            i++;
            j--;
        }
        return ans;
    }
};
