class Solution {
public:
    bool help(string a)
    {

        int x=0;
        int y = a.length() - 1;
        while(x <= y)
        {
            if(a[x] == a[y])
            {
                x++;
                y--;
            }
            else
            {
                return false;
            }
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        int len = a.length();
        if(len == 1) return true;
        int left = 0;
        int right = len-1;
        if(help(a)||help(b)) return true;
        

        while(left <= right)
        {
            if(a[left] == b[right])
            {
                left ++;
                right --;
            }
            else
            {
                break;
            }
        }
        if(left > right) return true;
        string na = a.substr(0, left) + b.substr(left, len);
        string na2 = a.substr(0, right+1) + b.substr(right+1, len);
        if(help(na)||help(na2)) return true;
        
        left = 0;

        right = len-1;
        while(left <= right)
        {
            if(b[left] == a[right])
            {
                left ++;
                right --;
            }
            else
            {
                break;
            }
        }
        
        string nb = b.substr(0, left)+a.substr(left, len);
        string nb2 = b.substr(0, right+1) + a.substr(right+1, len);

        if(help(nb)||help(nb2)) return true;
        return false;
        
    }
};
