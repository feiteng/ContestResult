class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        bool win = false;
        bool aa = true;
        bool bb = true;
        int n = a.length();
        for (int i = 0; i < n/2; i ++)
        {
            if (a[i] != a[n-1-i])
                aa = false;
            if (b[i] != b[n-1-i])
                bb = false;
        }
        if (aa or bb)
            return true;
        return a[0] == b[n-1] or a[n-1] == b[0];
    }
};
