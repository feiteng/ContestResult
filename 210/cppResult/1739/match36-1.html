class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if (a.length() == 1) return true;
        
        string a_reverse = a;
        reverse(a_reverse.begin(), a_reverse.end());
        //cout << a << endl;
        //cout << a_reverse << endl;
        
        string b_reverse = b;
        reverse(b_reverse.begin(), b_reverse.end());
        //cout << b << endl;
        //cout << b_reverse << endl;
        
        int index = 0;
        while (index < a.length() / 2 && a[index] == b_reverse[index]) {
            if (a[index] == b_reverse[index]) index++;
        }
        if (index == a.length() / 2) return true;
        
        auto subA = a.substr(index, a.length() - 2 * index);
        auto subB = b.substr(index, a.length() - 2 * index);
        if (isPalindrome(subA) || isPalindrome(subB)) return true;
        
        
        
        
        index = 0;
        while (index < a.length() / 2 && b[index] == a_reverse[index] ) {
            index++;
        }
        if (index == a.length() / 2) return true;
        
        subA = a.substr(index, a.length() - 2 * index);
        subB = b.substr(index, a.length() - 2 * index);

        if (isPalindrome(subA) || isPalindrome(subB)) return true;
        
        
        return false;
    }
    
    bool isPalindrome(string a) {
        int left = 0, right = a.length() - 1;
        
        while (left < right) {
            if (a[left] != a[right]) return false;
            left ++;
            right --;
        }
        
        return true;
    }
};
