class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int L = 0,R = a.size() - 1;
        while(L <= R and a[L] == b[R]){
            ++L,--R;
        }
        int l=L,r=R;

        while(L <= R and a[L] == a[R]){
            ++L,--R;
        }
        if(L>R) return true;
        L=l,R=r;
        while(L <= R and b[L] == b[R]){
            ++L,--R;
        }
        if(L>R) return true;
        L = 0, R= a.size()-1;
        while(L <= R and a[R] == b[L]){
            ++L,--R;
        }
        l=L,r=R;
        while(L <= R and a[L] == a[R]){
            ++L,--R;
        }
        if(L>R) return true;
        L=l,R=r;
        while(L <= R and b[L] == b[R]){
            ++L,--R;
        }
        if(L>R) return true;
        return false;
    }
};
