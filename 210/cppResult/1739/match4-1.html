class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        bool f = false, f1 = true, f2 = true, f3, f4;
        int len = a.size();
        for (int i = 0; i < len; i++) {
            if(a[i]!=a[len-1-i]) 
                f1 = false;
            if(b[i]!=b[len-1-i])
                f2 = false;
        }
        if(f1 || f2){
            f=true;
            return true;
        }

        int t = 0;
        f1 = true; f2 = true, f3 = true, f4 = true;
        for (int i = 0; i < len / 2; i++) {
            if(a[i] != b[len - i - 1]) {
                t = i;
                break;
            }
            t++;
        }
        for (int i=t; i<len/2;i++)
        {
            if(b[i]!=b[len-i-1])
                f1 = false;
        }
        for (int i=t; i<len/2;i++)
        {
            if(a[i]!=a[len-i-1])
                f2 = false;
        }
        t = 0;
        for (int i=0;i<len/2;i++)
        {
            if(b[i] != a[len - i - 1]) {
                t = i;
                break;
            }
            t++;
        }
        for (int i=t;i<len/2;i++) {
            if(a[i]!=a[len-i-1])
                f3 = false;
        }
        for (int i=t;i<len/2; i++) {
            if(b[i]!= b[len-i-1])
                f4 = false;
        }
        if(f1 || f2 || f3 || f4){
            return true;
        }
        return false;
    }
};
