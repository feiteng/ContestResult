class Solution {
public:
    bool iss(string a,string b){
        int l=0,r=b.size()-1;
        while(l<r && a[l]==b[r]){
            l++;
            r--;
        }
        return ispalin(a,l,r) or ispalin(b,l,r);
    }
    
    bool ispalin(string a,int l,int h){
        while(l<h){
            if(a[l]!=a[h])
                return false;
            l++;
            h--;
        }
        return true;
    }
    
    bool checkPalindromeFormation(string a, string b) {
        if(a.size()<2)
            return true;
        return iss(a,b) or iss(b,a);    
    }
};
