class Solution {
public:
    
    bool doit(string a,string b){
        int n = a.size();
        int idx = 0;
        int cnt=0;
        while(idx<n){
            if(a[idx]==b[idx]){
                cnt+=2;
            }
            else{
                break;
            }
            idx++;
        }
        if(cnt>=n){
            return true;

        }
        
        int left = idx;
        int right = n-idx-1;
        while(left<=right){
            if(a[left]==a[right]){
                left++;
                right--;
            }
            else{
                break;
            }
        }
        if(left>right){
            return true;
        }

        left = idx;
        right = n-idx-1;
         while(left<right){
            if(b[left]==b[right]){
                left++;
                right--;
            }
            else{
                return false;
            }
        }
        return true;
    }
    
    bool checkPalindromeFormation(string a, string b) {
        string c = a;
        string d = b;
        
        
        reverse(b.begin(),b.end());
        reverse(c.begin(),c.end());
        if(doit(a,b)||doit(d,c)){
            
            return true;
        }
        return false;
    }
};
