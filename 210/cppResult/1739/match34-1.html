class Solution {
public:
    bool ispal(string str)
    {
        int s=0;
        int e=str.size()-1;
        while (s<e) {
            if (str[s]!=str[e]) return false;
            s++;
            e--;
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        return (ispal(a) || ispal(b) || a[0]==b.back() || b[0]==a.back());
    }
};
