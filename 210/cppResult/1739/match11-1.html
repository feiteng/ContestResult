class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int n = (int)a.size();
        int ma = 0, mb = 0;
        if(n%2) {
            ma = 1;
            while(ma<(n+1)/2 && a[n/2 - ma] == a[n/2 + ma]) ma++;
            mb = 1;
            while(mb<(n+1)/2 && b[n/2 - mb] == b[n/2 + mb]) mb++;
        } else {
            while(ma<n/2 && a[n/2 - 1 - ma] == a[n/2 + ma]) ma++;
            while(mb<n/2 && b[n/2 - 1 - mb] == b[n/2 + mb]) mb++;
        }

        if(2*ma - n%2 == n || 2*mb- n%2 == n) return true;
        int lena = 0, lenb = 0;
        for(int i = 0; i < n; i++) {
            if(a[i] == b[n - i - 1]) lena ++;
            else break;
        }

        for(int i = 0; i < n; i++) {
            if(b[i] == a[n - i - 1]) lenb ++;
            else break;
        }
        
        if(ma + lena >= n/2 + n%2 ||
           ma + lenb >= n/2 + n%2 || 
           mb + lena >= n/2 + n%2 || 
           mb + lenb >= n/2 + n%2) return true;
        return false;
    }
};
