class Solution {
public:
    bool isPalindrome(string &a){
        int i=0,j=a.size()-1;
        while(i<j){
            if(a[i]!=a[j])
                return false;
          i++;
            j--;
        }
        return true;
    }
    bool checkPalindromeFormation(string a, string b) {
        int i=0,j=b.size()-1;
        if(isPalindrome(a)||isPalindrome(b))
            return true;
        while(i<a.size() && j >=0 && a[i]==b[j]){
            return true;
        }
        i=a.size()-1,j=0;
        while(i>=0 && j <b.size() && a[i]==b[j]){
            return true;
        }
        return false;
        
    }
};
