class Solution {
public:
    bool ispal(string x){
        int n=x.length();
        for(int i=0;i<n;i++){
            if(x[i]!=x[n-i-1])return false;
        }
        return true;
    }
    bool check(string a, string b){
        int match=0;
        int n=a.length();
        while(a[match]==b[n-1-match])match++;

        if(match==n/2)return true;
        return ispal(b.substr(match,n-match*2)) || ispal(a.substr(match,n-match*2));
    }
    
    bool checkPalindromeFormation(string a, string b) {
        if(ispal(a) || ispal(b))return true;
        if(a[0]!=b.back() && b[0]!=a.back())return false;
        return check(a,b) || check(b,a);
    }
};
