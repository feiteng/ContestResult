class Solution {
public:
    bool check(string s) {
        int i = 0, j = s.size() - 1;
        while (i <= j) {
            if (s[i] != s[j]) return 0;
            ++i, --j;
        }
        return 1;
    }
    
    bool solve(string& A, string& B) {
        int i = 0, j = A.size() - 1, sz = A.size();
        while (A[i] == B[j]) ++i, --j;
        if (i >= sz / 2) return 1;
        if (check(A.substr(i, j - i + 1)) || check(B.substr(i, j - i + 1))) return 1;
        return 0;
    }

    bool checkPalindromeFormation(string a, string b) {
        return solve(a, b) || solve(b, a);
    }
};
