class Solution {
public:
    int mxa,mxb;
    bool checkPalindromeFormation(string a, string b) {
        int slen=a.length();
        int pos=slen/2-1,ok=1;
        for(;pos>=0;pos--){
            if(a[pos]!=a[slen-pos-1]){
                ok=0;
                break;
            }
        }
        if(ok) return true;

        ok=1;
        for(int i=0;i<=pos;i++){
            if(a[i]!=b[slen-1-i]){
                ok=0;
                break;
            }
        }
        if(ok) return true;
        ok=1;
        for(int i=0;i<=pos;i++){
            if(b[i]!=a[slen-1-i]){
                ok=0;
                break;
            }
        }
        if(ok) return true;
        ok=1,pos=slen/2-1;
         for(;pos>=0;pos--){
            if(b[pos]!=b[slen-pos-1]){
                ok=0;
                break;
            }
        }
        if(ok) return true;

        ok=1;
        for(int i=0;i<=pos;i++){
            if(b[i]!=a[slen-1-i]){
                ok=0;
                break;
            }
        }
        if(ok) return true;
        ok=1;
        for(int i=0;i<=pos;i++){
            if(a[i]!=b[slen-1-i]){
                ok=0;
                break;
            }
        }
        if(ok) return true;
        return false;
    }
};
