class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(a.size() == 1) return true;
        int possibleA = 0;

        int possibleB = 0;
        string revb(b);
        string reva(a);
        reverse(revb.begin(),revb.end());
        for(int i = 0;!possibleA&&i<a.size();i++)
        {
            if(a[i] == revb[i])
                possibleA++;
            else break;
        }
        if(possibleA) return true;
        reverse(reva.begin(),reva.end());

        for(int i = 0;!possibleB&&i<a.size();i++)
        {
            if(reva[i] == b[i])
                possibleB++;
            else break;
        }
        
        if(possibleB) return true;
        return false;
    }
};
