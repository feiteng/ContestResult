class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int l = 0, r = a.size() - 1;
        while (l <= r && a[l] == b[r]) {
            l++;
            r--;
        }
        int templ = l;
        int tempr = r;
        while (l <= r && b[l] == b[r]) {
            l++;
            r--;
        }
        if (l > r) return true;
        l = templ;
        r = tempr;
        while (l <= r && a[l] == a[r]) {
            l++;
            r--;
        }
        if (l > r) return true;
        l = 0;
        r = a.size() - 1;
        while (l <= r && b[l] == a[r]) {
            l++;
            r--;
        }
        templ = l;
        tempr = r;
        while (l <= r && b[l] == b[r]) {
            l++;
            r--;
        }
        if (l > r) return true;
        l = templ;
        r = tempr;
        while (l <= r && a[l] == a[r]) {
            l++;
            r--;
        }
        if (l > r) return true;
        return false;
    }
};
