class Solution {

private:
    bool palin(const string& s, int l, int r) {
        int n = s.length();

        for (int i = 0; l+i < r-i; ++i) {
            if (s[l+i] != s[r-i]) {
                return false;
            }
        }
        return true;
    }
    
public:
    bool checkPalindromeFormation(string a, string b) {
        int n = a.length();
        if (n == 1) {
            return true;
        }
        
        if (palin(a, 0, n-1) || palin(b, 0, n-1)) {
            return true;
        }
        

        int i;
        for (i = 0; i < n && i < n-1-i; ++i) {
            if (a[i] != b[n-1-i]) {
                break;
            }
        }
        
        if (palin(a, i, n-1-i) || palin(b, i, n-1-i)) {
            return true;
        }
        
        for (i = 0; i < n && i < n-1-i; ++i) {
            if (b[i] != a[n-1-i]) {
                break;
            }
        }
        
        if (palin(a, i, n-1-i) || palin(b, i, n-1-i)) {
            return true;
        }
        
        return false;
    }
};
