class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(isPalindrome(a) || isPalindrome(b))
            return true;
        if(a[0]==b[b.size()-1] || a[a.size()-1] ==b[0])
            return true;
        return false;
        
    }
 
    bool isPalindrome(string s){
        int start = 0, end = s.size()-1;
        while(start<end){
            if(s[start]!=s[end])
                return false;
            start++;
            end--;
        }
        return true;
    }
};
