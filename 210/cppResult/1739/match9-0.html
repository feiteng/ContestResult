class Solution {
public:
    bool judge(const string &a, const string &b) {
        int len = a.length();
        int prelen = 0;
        for (int i = 0; i < len; i++) {
            if (a[i] == b[len - i - 1]) {
                ++prelen;
            } else
                break;
        }
        int huilen = len - prelen * 2;
        bool flag = true;
        for (int i = prelen; i < prelen + huilen; i++) {
            if (a[i] != a[huilen - i + prelen + prelen - 1]) {
                flag = false;
                break;
            }
        }
        if (flag)
            return true;
        flag = true;
        for (int i = prelen; i < prelen + huilen; i++) {
            if (b[i] != b[huilen - i + prelen + prelen - 1]) {
                flag = false;
                break;
            }
        }
        if (flag)
            return true;
        return false;
    }
    bool checkPalindromeFormation(string a, string b) {
        return judge(a, b) || judge(b, a);
    }
};
