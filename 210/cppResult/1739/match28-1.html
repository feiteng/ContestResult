class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        int n = a.size();
        // if(n == 1)
        //     return true;
        int r = n;
        while(b[r - 1] == a[n - r])--r;
        bool v = true;

        for(int i = n - r; i < r; ++i)
            if(a[i] != a[n - 1 - i]){
                v = false;
                break;
            }
        if(v)
            return true;
        v = true;
        for(int i = n - r; i < r; ++i)
            if(b[i] != b[n - 1 - i]){
                v = false;
                break;
            }
        if(v)
            return true;
        r = n;
        while(a[r - 1] == b[n - r])--r;
        v = true;

        for(int i = n - r; i < r; ++i)
            if(b[i] != b[n - 1 - i]){
                v = false;
                break;
            }
        if(v)
            return true;
        v = true;
        for(int i = n - r; i < r; ++i)
            if(a[i] != a[n - 1 - i]){
                v = false;
                break;
            }
        if(v)
            return true;
        return false;
    }
};
