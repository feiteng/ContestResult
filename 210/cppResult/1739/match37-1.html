class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(isPalin(a) || isPalin(b))
            return true;
        return check(a,b) || check(b,a);
    }
    
    bool check(string &a, string &b){
        int n=a.size(), i=0, j=n-1;
        while(i<j){
            if(a[i]==b[j])
                ++i, --j;
            else
                break;
        }

        if(i>=j)
            return true;
        if(isPalin(a.substr(i,j-i+1)) || isPalin(b.substr(i,j-i+1)))
            return true;
        return false;
    }
    
    bool isPalin(string s){
        int i=0, j=s.size()-1;
        while(i<j){
            if(s[i++]!=s[j--])
                return false;
        }
        return true;
    }
};
