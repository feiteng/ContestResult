bool isPaline(string a)
{
  int i,j;
  i=0;
  j=a.length()-1;
  while(i<j)
  {
    if(a[i]!=a[j])
      return(false);
    
    i++;
    j--;
  }
  return(true);
}

class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        if(isPaline(a) || isPaline(b))
          return(true);
      int i,j,n=a.length();
      i=0;
      j=n-1;
      while(i<j)
      {
        if(a[i]==b[j])
        {
          i++;
          j--;
        }
        else
        {
          if(isPaline(a.substr(i,n-2*i)) || isPaline(b.substr(j-(n-2*i)+1,n-2*i)))
            return(true);
          break;
        }
      }
      if(i==j || i-j==1)
        return(true);
      
      i=n-1;
      j=0;
      while(i>j)
      {
        if(a[i]==b[j])
        {
          i--;
          j++;
        }
        else
        {
          if(isPaline(a.substr(i-(n-2*j)+1,n-2*j)) || isPaline(b.substr(j,n-2*j)))
            return(true);
          break;
        }
      }
      if(i==j || j-i==1)
        return(true);
      
      return(false);
    }
};
