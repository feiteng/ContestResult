class Solution {
public:
    bool check(string s){
        int l=0,r=s.size()-1;
        while(l<=r){
            if(s[l]!=s[r]) return false;
            l++; r--;
        }
        return true;
    }
    
    bool checkPalindromeFormation(string a, string b) {
        if(check(a) || check(b)) return true;
        if(a[0]==b[b.size()-1] || b[0]==a[a.size()-1]) return true;
        return false;
    }
};
