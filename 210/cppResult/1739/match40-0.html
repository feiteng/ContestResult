class Solution {
public:
    bool checkPalindromeFormation(string a, string b) {
        bool ans1 = false;
        bool ans2 = false;
        
        if(isPel(a) || isPel(b)){
            return true;
        }
        
        int n = a.length();
        
        int ind1=0;
        while(a[ind1] == b[n-ind1-1])
            ind1++;
        
        if(ind1 > 0)
            return true;
        
        int ind2 = 0;
        while(b[ind2] == a[n-ind2-1])
            ind2++;
        
        if(ind2 > 0)
            return true;
        
        
        return false;
        
        
    }
    
    bool isPel(string s){
        int n = s.length();
        for(int i=0;i<n/2;i++){
            if(s[i] != s[n-i-1])
                return false;
        }
        return true;
    }
};
