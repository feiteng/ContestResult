class Solution {
public:
    bool checkok(int n, vector <vector <int> > &G, vector <bool> &res) {
        vector <bool> vis;
        vis.resize(n + 1);

        int start = 0;
        for (int i = 1; i <= n; ++ i) {
            if (res[i] == true) {
                start = i;
                break;
            }
        }
        vis[start] = true;
        queue<int> q;
        q.push(start);
        while(!q.empty()) {
            int top = q.front(); q.pop();
            for (int i = 0; i < G[top].size(); ++ i) {
                int y = G[top][i];
                if (res[y] == false) continue;
                if (vis[y] == true) continue;
                vis[y] = true;
                q.push(y);
            }
        }
        for (int i = 1; i <= n; ++ i) if(res[i] && !vis[i]) return false;
        return true;
    }
    vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges) {
        vector <vector <int> > G, dis;
        vector <bool> res;
        G.resize(n + 1);
        res.resize(n + 1);
        dis.resize(n + 1);
        for (int i = 1; i <= n; ++ i) dis[i].resize(n + 1);

        for (int i = 1; i <= n; ++ i)
            for (int j = 1; j <= n; ++ j)
                dis[i][j] = 1e9;
        for (int i = 1; i <= n; ++ i) dis[i][i] = 0;
        for (auto edge:edges) {

            G[edge[0]].push_back(edge[1]);
            G[edge[1]].push_back(edge[0]);
            dis[edge[0]][edge[1]] = dis[edge[1]][edge[0]] = 1;
        }
        for (int k = 1; k <= n; ++k)
            for (int i = 1; i <= n; ++ i)
                for (int j = 1; j <= n; ++ j)
                    dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j]);
        vector <int> ans; ans.resize(n - 1);
        for (int s = 1; s < (1 << n); ++ s) {
            int cnt = 0;
            for (int i = 1; i <= n; ++ i)
                if (s & (1 << i - 1)) res[i] = true, ++ cnt;
                else res[i] = false;
            if(cnt == 1) continue;
            if (checkok(n, G, res) == false) continue;
            int mx = 0;
            for (int i = 1; i <= n; ++ i) {
                if (~s & (1 << i - 1)) continue;
                for (int j = i + 1; j <= n; ++ j) {
                    if(~s & (1 << j - 1)) continue;
                    mx = max(mx, dis[i][j]);   
                }
            }
            ans[mx - 1] ++;
        }
        return ans;
    }
};
