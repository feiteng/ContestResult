class Solution {
 public:
  vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges) {
    vector<int> res(n - 1, 0);
    for (int i = 1; i < 1 << (n - 1); i++) {
      vector<vector<int>> adj_list(n);

      int cnt = 0, start = -1, d = -1;
      for (int j = 0; j < n - 1; j++) {
        if (((1 << j) & i) != 0) {
          cnt++;
          int u = edges[j][0], v = edges[j][1];
          adj_list[u - 1].push_back(v - 1);
          adj_list[v - 1].push_back(u - 1);
          start = u - 1;
        }
      }
      getDepth(adj_list, start, -1, d, cnt);
      if (cnt == -1) {
        res[d - 1]++;
      }
    }
    return res;
  }
  int getDepth(vector<vector<int>> &edges, int v, int parent, int &d, int &cnt) {
    int max_depth = 0;
    cnt--;
    for (int next : edges[v]) {
      if (next != parent) {
        int depth = getDepth(edges, next, v, d, cnt);
        d = max(d, depth + max_depth);
        max_depth = max(max_depth, depth);
      }
    }
    return max_depth + 1;
  }
};
