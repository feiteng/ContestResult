class Solution {
public:
    vector<int>adj[16];
    int dist[16][16];
    vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges) {
        for(int i=0;i<16;i++)
            for(int j=0;j<16;j++)
                dist[i][j]=1000;
        for(int i=0;i<16;i++)
            dist[i][i]=0;
        for(int i=0;i<edges.size();i++){
            dist[edges[i][0]][edges[i][1]]=1;
            dist[edges[i][1]][edges[i][0]]=1;
        }
        for(int k=0;k<16;k++)
            for(int i=0;i<16;i++)
                for(int j=0;j<16;j++)
                    dist[i][j]=min(dist[i][j],dist[i][k]+dist[k][j]);
        
        vector<int>cnt(n+1);
        
        for(int i=0;i<(1<<n);i++){
            vector<int>tmp;
            for(int j=0;j<n;j++){
                if(i&(1<<j))
                    tmp.push_back(j+1);
            }
            if(tmp.size()<=1)
                continue;
            int maxi=0;
            int cnt1=0;
            for(int j=0;j<tmp.size();j++)
                for(int k=j+1;k<tmp.size();k++){
                    maxi=max(maxi,dist[tmp[j]][tmp[k]]);
                    if(dist[tmp[j]][tmp[k]]==1)
                        cnt1++;}
            if(cnt1==tmp.size()-1)
            cnt[maxi]++;
        }
        
        vector<int>res;
        for(int i=1;i<n;i++)
            res.push_back(cnt[i]);
        return res;
    }
};
