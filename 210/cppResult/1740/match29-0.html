class Solution {
public:
    vector<int> bits(int n) {
        int cur = 0;
        vector<int> ret;
        while ((1<<cur) <= n) {
            if (n&(1<<cur)) ret.push_back(cur);
            cur++;
        }
        return ret;
    }
    vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges) {
        set<int> cur;
        vector<int> ans(n-1);
        unordered_map<int, vector<int>> es;
        vector<vector<int>> dist(n, vector<int>(n, 18));
        for (auto e : edges) {
            dist[e[0]-1][e[1]-1] = 1;
            dist[e[1]-1][e[0]-1] = 1;
        }
        for (int k=0; k<n; k++) {
            for (int i=0; i<n; i++) {
                for (int j=0; j<n; j++) {
                    dist[i][j] = min(dist[i][j], dist[i][k]+dist[k][j]);
                }
            }
        }
        for (auto e : edges) {
            cur.insert((1<<(e[0]-1))|(1<<(e[1]-1)));
            es[e[0]-1].push_back(e[1]-1);
            es[e[1]-1].push_back(e[0]-1);
        }
        while (cur.size()) {
            set<int> ns;
            for (auto c : cur) {
                auto bs = bits(c);
                int N = bs.size();
                int mx = 0;
                for (int j=0; j<N; j++) {
                    for (int k=j+1; k<N; k++) {
                        mx = max(mx, dist[bs[j]][bs[k]]);
                    }
                }
                ans[mx-1]++;
                for (auto b : bs) {
                    for (auto nv : es[b]) {
                        if (c != (c|(1<<nv))) {
                            ns.insert(c|(1<<nv));
                        }
                    }
                }
            }
            swap(cur, ns);
        }
        return ans;
