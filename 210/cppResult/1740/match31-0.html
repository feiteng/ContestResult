int n;
int e[16][16];
vector<int> ans;

class Solution
{
    void dfs(int i, vector<int>& nums)
    {
        if (i > n)
        {
            if (nums.size() >= 2)
            {
                int cnt = 0;
                
                for (int i = 0; i < nums.size(); ++i)
                {
                    for (int j = i + 1; j < nums.size(); ++j)
                    {
                        cnt += e[nums[i]][nums[j]] == 1;
                    }
                }
                
                if (cnt != nums.size() - 1)
                {
                    return;
                }
                
                for (auto& num : nums)
                {
                    //cout << num << " ";
                }
                //cout << endl;
                
                int d = 0;
                
                for (int i = 0; i < nums.size(); ++i)
                {
                    for (int j = i + 1; j < nums.size(); ++j)
                    {
                        d = max(d, e[nums[i]][nums[j]]);
                    }
                }
                
                ++ans[d - 1];
            }
        }
        else
        {
            nums.push_back(i);
            dfs(i + 1, nums);
            nums.pop_back();
            
            dfs(i + 1, nums);
        }
    }

public:

    vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges)
    {
        ::n = n;
        ans = vector<int>(n - 1, 0);
        memset(e, 0x3f3f3f3f, sizeof(e));
        
        for (auto& edge : edges)
        {
            e[edge[0]][edge[1]] = e[edge[1]][edge[0]] = 1;
        }
        
        for (int k = 1; k <= n; ++k)
        {
            for (int i = 1; i <= n; ++i)
            {
                for (int j = 1; j <= n; ++j)
                {
                    if (e[i][j] > e[i][k] + e[k][j])
                    {
                        e[i][j] = e[i][k] + e[k][j];
                    }
                }
            }
        }
        
        for (int i = 1; i <= n; ++i)
        {
            for (int j = 1; j <= n; ++j)
            {
                //cout << e[i][j] << ' ';
            }
            //cout << endl;
        }
        
        //cout << endl;
        
        
        
        vector<int> nums;
        
        dfs(1, nums);
        
        return ans;
    }
};
