class Solution {
public:
    
    const int INF = 1e9 + 7;    
    int N, dis[20][20];
    vector<int> g[20];
    bool vis[20];
    
    void dfs(int v, int p , int mask){
        vis[v] = 1; for(int u : g[v]) if(!vis[u] && (mask & (1 << (u - 1)))) dfs(u , v , mask);
    }
    
    vector<int> countSubgraphsForEachDiameter(int n, vector<vector<int>>& edges) {
        N = n;
        for(int i = 1; i <= n; ++i)
            for(int j = 1; j <= n; ++j)
                if (i == j) 
                    dis[i][j] = 0;
                else 
                    dis[i][j] = INF;
        for(vector<int>& edge : edges){
            g[edge[0]].push_back(edge[1]);
            g[edge[1]].push_back(edge[0]);
            dis[edge[0]][edge[1]] = dis[edge[1]][edge[0]] = 1;
        }

        for(int k = 1; k <= n; ++k)
            for(int i = 1; i <= n; ++i)
                for(int j = 1; j <= n; ++j)
                    dis[i][j] = min(dis[i][j] , dis[i][k] + dis[k][j]);
        vector<int> ret(n - 1);
        for(int mask = 1; mask < (1 << n); ++mask) {
            int cnt = 0;
            for(int i = 1; i <= n; ++i) vis[i] = 0;
            for(int i = 1; i <= n; ++i){
                if(mask & (1 << (i - 1))){
                    if(!vis[i]){
                        ++cnt;
                        dfs(i , i , mask);
                    }
                }
            }

            if(cnt != 1) continue;
            int md = 0;
            for(int i = 1; i <= n; ++i)
                if (mask & (1 << (i - 1)))
                    for(int j = i; j <= n; ++j)
                        if(mask & (1 << (j - 1)))
                            md = max(md , dis[i][j]); 
            if(md == 0) continue;
            vector<int> path;
            for(int i = 1; i <= n; ++i) if(mask & (1 << (i - 1))) path.push_back(i);
            if(md >= 1) ret[md - 1] += 1;
        }
        return ret;
    }
};
