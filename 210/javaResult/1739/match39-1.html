class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        // check if a or b is palindrome
        
        if (isP(a) || isP(b)) {
            return true;
        }
        
        if (check(a, b) || check(b, a)) {
            return true;
        }
        return false;

    }
    
    private boolean check(String a, String b) {
        
        int l = 0;
        int r = a.length() - 1;
        
        
        // l is a, r is b
        
        while (l < r) {
            if (a.charAt(l) != b.charAt(r)) {
                break;
            }
            ++l;
            --r;
        }
        
        if (l >= r) {
            return true;
        }
        
        int firstl = l;
        int firstr = r;
        while (firstl < firstr) {
            if (a.charAt(firstl) != a.charAt(firstr)) {
                break;
            }
            ++firstl;
            --firstr;
        }
        
        if (firstl >= firstr) {
            return true;
        }
        
        int sl = l;
        int sr = r;
        while (sl < sr) {
            if (b.charAt(sl) != b.charAt(sr)) {
                break;
            }
            ++sl;
            --sr;
        }
        if (sl >= sr) 
{
            return true;
        }
        
        return false;
        
        
        
        
        
    }
    
    private boolean isP(String a) {
        int l = 0;
        int r = a.length() - 1;
        while (l < r) {
            if (a.charAt(l) != a.charAt(r)) {
                return false;
            }
            ++l;
            --r;
        }
        return true;
    }
}
