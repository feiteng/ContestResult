class Solution {
    public boolean checkPalindromeFormation(String a, String b)
	{
		return isValid(a, b) || isValid(b, a);
	}

	private boolean isValid(String a, String b)
	{
		int len = a.length();
		int matchCount = 0;
		for (int i = 0; i < len / 2; i++)
		{
			if (a.charAt(i) == b.charAt(len - i - 1))
				matchCount++;
			else
				break;
		}

		if (matchCount == len / 2)
			return true;

		return isPal(a.substring(matchCount, len - matchCount))
				|| isPal(b.substring(matchCount, len - matchCount));
	}

	
	private boolean isPal(String a)
	{
		int len = a.length();
		for (int i = 0; i < len / 2; i++)
		{
			if (a.charAt(i) != a.charAt(len - i - 1))
				return false;
		}

		return true;
	}
}
