class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (check(a) || check(b))
            return true;

        int len = a.length();

        int x = 0;

        while (x < len / 2) {
            if (a.charAt(x) != b.charAt(len - 1 - x))
                break;

            x++;
        }

        if (x == len / 2)
            return true;
        if (check(b.substring(x, len - x)) || check(a.substring(x, len - x)))
            return true;

        int y = 0;
        while (y < len / 2) {
            if (b.charAt(y) != a.charAt(len - 1 - y))
                break;

            y++;
        }
        if (y == len / 2)
            return true;
        if (check(
a.substring(y, len - y)) || check(b.substring(y, len - y)))
            return true;

        return false;
    }

    boolean check(String a) {
        int l = a.length();
        for (int i = 0; i < l / 2; i++) {
            if (a.charAt(i) != a.charAt(l - 1 - i))
                return false;
        }
        return true;
    }
}
