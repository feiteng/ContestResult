class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return helper(a, b) || helper(b, a);
    }
    
    private boolean helper(String a, String b) {
                //two pointers
        int n = a.length();
        
        int l = 0;
        int r = n - 1;
        while (l < r) {
            if (a.charAt(l) == b.charAt(r)) {
                ++l;
                --r;
            } else {
                break;
            }
        }
        if (l >= r) {
            return true;
        }
        
        //check a
        int la = l;
        int ra = r;
        while (la < ra) {
            if (a.charAt(la) == a.charAt(ra)) {
                ++la;
                --ra;
            } else {
                break;
            }
        }
        if (la >= ra) {
            return true;
        }
        
        int lb = l;
        int rb = r;
        while (lb < rb) {
            if (b.charAt(lb) == b.charAt(rb)) {
                ++lb;
                --rb;
            } else {
                break;
            }
        }
        if (lb >= rb) {
            return true;
        }
        return false;
    }
}
