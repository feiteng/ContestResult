class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (isPalindrome(a) || isPalindrome(b)) {
            return true;
        }
        
        int len = a.length();
        int idxA = 0, idxB = len - 1;
        boolean result = false;
        /*
            "pvhm upgqeltozftlmfjjde"
                  | idxB
            "yjgp zbezspnnpszebz mhvp"
                               | idxA
        
        */
        while (idxA < idxB) {
            if (a.charAt(idxA) == b.charAt(idxB)) {
                idxA ++;
                idxB --;
            } else {
                break;
            }
        }
        
        result = isPalindrome(a.substring(0, idxA) + b.substring(idxA)) ||
                 isPalindrome(a.substring(0, idxB + 1) + b.substring(idxB + 1));
        
        if (result) {
            return true;
        }
        
        idxA = len - 1;
        idxB = 0;
        while (idxB < idxA) {
            if (a.charAt(idxA) == b.charAt(idxB)) {
                idxA --;
                idxB ++;
            } else {
                break;
            }
        }
        
        result = isPalindrome(b.substring(0, idxB) + a.substring(idxB)) ||
                 isPalindrome(b.substring(0, idxA + 1) + a.substring(idxA + 1));
        
        return result;
    }
    
    private boolean isPalindrome(String s) {
        int start = 0, end = s.length() - 1;
        while (start < end) {
            if (s.charAt(start) != s.charAt(end)) {
                return false;
            }
            start ++;
            end --;
        }
        return true;
    }
}
