class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return checkPalindromeSplit(a, b) || checkPalindromeSplit(b, a);    
    }
    
    private boolean checkPalindromeSplit(String a, String b) {
        int i = 0;
        int j = b.length() - 1;
        while (i < a.length() / 2 && a.charAt(i) == b.charAt(j)) {
            i++;
            j--;
        
}
        return checkPalindromeSegment(a, i, j) || checkPalindromeSegment(b, i, j);
    }
    
    private boolean checkPalindromeSegment(String str, int i, int j) {
        while (i < j) {
            if (str.charAt(i) != str.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
