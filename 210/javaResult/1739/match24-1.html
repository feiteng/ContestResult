class Solution {
    public boolean checkPalindromeFormation(String s, String t) {
        return isPalindrome(s, t) || isPalindrome(t, s);
    }

    private boolean isPalindrome(String s, String t) {
        int l = 0;
        int r = t.length() - 1;
        for (; l <= r; ++l, --r) {
            if (s.charAt(l) != t.charAt(r)) {
                break;
            }
        }
        return isPalindrome(s, l, s.length() - 1 - l)
            || 
isPalindrome(t, l, s.length() - 1 - l);
    }

    private boolean isPalindrome(String s, int start, int end) {
        while (start < end) {
            if (s.charAt(start++) != s.charAt(end--)) {
                return false;
            }
        }
        return true;
    }
}
