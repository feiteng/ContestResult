class Solution {
    
public boolean checkPalindromeFormation(String a, String b) {
        int n = a.length();
        int al = (n-1)/2;
        int ar = n/2;
        while(al>=0 && ar<n){
            if(a.charAt(al)==a.charAt(ar)){
                al--;
                ar++;
            }    
            else{
                break;
            }
        }
        int bl = (n-1)/2;
        int br = n/2;
        while(bl>=0 && br<n){
            if(b.charAt(bl)==b.charAt(br)){
                bl--;
                br++;
            }     
            else{
                break;
            }
        }
        
        
int as = 0;
        int be = n-1;
        // System.out.println(al+","+ar);
        // System.out.println(bl+","+br);
        while(as<=be){
            if(a.charAt(as)==b.charAt(be)){
                as++;
                be--;
            }
            else{
                break;
            }
        }
        // System.out.println(as+","+be);
        if(as>=al || as>=bl){
            return true;
        }
        int bs = 0;
        int ae = n-1;
        while(bs<=ae){
            if(b.charAt(bs)==a.charAt(ae)){
                bs++;
                ae--;
            }
            else{
                break;
            }
        }
        if(bs>=al || bs>=bl){
            return true;
        }
        return false;
    }
}
