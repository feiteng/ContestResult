public class Solution {
	private int n;
	public boolean checkPalindromeFormation(String a, String b) {
		n = a.length();
		return search(
a, b) || search(b, a);
	}

	private boolean search(String a, String b) {
		int start = 0;
		while (2 * start < n - 1 && a.charAt(start
) == b.charAt(n - 1 - start)) {
			start++;
		}
		return isPalindrome(a, start) || isPalindrome(b, start);
	}

	private boolean isPalindrome(String 
b, int start) {
		while (2 * start < n - 1) {
			if (b.charAt(start) != b.charAt(n - 1 - start)) {
				return false;
			}
			start++;
		}
		return true;
	}
}
