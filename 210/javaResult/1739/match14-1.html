class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return helper(a, b, a.length()/2) || helper(b, a, a.length()/2);
    }
    
    private boolean helper(String a, 
String b, int len) {
        int iA = 0, iB = a.length()-1;
        while(true) {
            // System.out.println(iA + "," + iB);
            // System.out.println(a.charAt(iA) + "," + b.charAt(iB));
            if(a.charAt(iA) == b.charAt(iB)) {
                len--;
                iA ++;
                iB --;
            }
            else break;
        }
        return helper1(a, iA, iB) || helper1(b, iA, iB);
    
}
    
    private boolean helper1(String a, int iA, int iB) {
        while(iA < iB) {
            if(a.charAt(iA++) != a.charAt(iB--)) return false;
        }
        return true;
    }
}
