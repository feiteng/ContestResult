class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (check(a) || check(b)) {
            return true;
        }
        if (a.charAt(0) == b.charAt(b.length() - 1) || a.charAt(a.length() - 1) == b.charAt(0)) {
            return true;
        }
        return false;
    }
    
    public boolean check (String a) {
        int l = 0;
        int r = a.length() - 1;
        while (l < r) {
            if (a.charAt(l++) != a.charAt(r--)) {
                return false;
            }
        }
        return true;
    }
}
