class Solution {
 public boolean checkPalindromeFormation(String a, String b) {
    if (isPalindrome(a) || isPalindrome(b)) return true;
    int i = 0, j = a.length()-1;

    if (a.charAt(i) == b.charAt(j) || a.charAt(j) == b.charAt(i))
      return true;
    return false;
  }
  private boolean isPalindrome(String a) {
    int i = 0;
    int j = a.length()-1;
    while (i < j)
    {
      if (a.charAt(i) != a.charAt(j))
        return false;
      i++;
      j--;
    }
    return true;
  }
}
