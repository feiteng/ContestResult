class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return isPalindrome(a, 0, a.length() - 1) || isPalindrome(b, 0, b.length() - 1) || check(a, b) || check(b, a);
    }
    
    public boolean check(String str1, String str2) {
        int lo = 0;
        int hi = str2.length() - 1;
        
        while(lo < hi) {
            if (str1.charAt(lo) == str2.charAt(hi)) {
                lo++;
                hi--;
            } else {
                return isPalindrome(str1, lo, hi) || isPalindrome(str2, lo, hi);
            }
        }
        return true;
    }
    
    protected boolean isPalindrome(String str, int lo, int hi) {
        while(lo < hi) {
            if (str.charAt(lo++) != str.charAt(hi--)) return false;
        }
        return true;
    }
}
