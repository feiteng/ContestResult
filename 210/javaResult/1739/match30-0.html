class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (isPal(a) || isPal(b)) {
            return true;
        }
        
        if (isPal(a, b) || isPal(b, a)) {
            return true;
        }
        return false;
    }
    
    private boolean isPal(String s1, String s2) {
        int left = 0, right = s1.length() - 1;
        while (left <= right) {
            if (s1.charAt(left) != s2.charAt(right)) {
                break;
            }
            left++; right--;
        }
        
        System.out.println("left="+left + " right="+right);
        
        if (left > right) { return true; }
        
        int l1 = left, r1 = right;
        while (l1 <= r1) {
            if (s1.charAt(l1) != s1.charAt(r1)) {
                break;
            }
            l1++; r1--;
        }
        System.out.println("l1="+l1 + " r1="+r1);
        if (l1 > r1) { return true; }
        
        int l2 = left, r2 = right;
        while (l2 <= r2) {
            if (s2.charAt(l2) != s2.charAt(r2)) {
                break;
            }
            l2++; r2--;
        }
        System.out.println("l2="+l2 + " r2="+r2);
        if (l2 > r2) { return true; }
        return false;
    }
    
    private boolean isPal(String s) {
        int left = 0, right = s.length() - 1;
        while (left < right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++; right--;
        }
        return true;
    }
}
