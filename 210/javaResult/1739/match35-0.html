class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (isPalindrome(a) || isPalindrome(b)) {
            return true;
        }
        if (check(a, b) || check(b, a)) {
            return true;
        }
        return false;
        
    }
    
    private boolean check(String s1, String s2) {
        int left = 0, right = s1.length() - 1;
        int mid = s1.length() / 2;
        while (left <= mid) {
            right = s1.length() - 1 - left;
            if (s1.charAt(left) == s2.charAt(right)) {
                left++;
                right--;
            } else {
                break;
            }
        }
        if (left > mid) return true;
        int length = s1.length() - left * 2;
        if (isPalindrome(s1.substring(left, left + length))) return true;
        if (isPalindrome(s2.substring(right + 1 - length,right + 1))) return true;
        
        return false;
        
    }
    
    private boolean isPalindrome(String s) {
        int left = 0, right = s.length() - 1;
        while (left < right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
