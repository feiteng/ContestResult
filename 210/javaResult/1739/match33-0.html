class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (isPanli(a) || isPanli(b)) {
            return true;
        }
        return helper(a, b) || helper(b, a);
        
    } 
    private boolean helper(String a, String b) {
        int len = a.length();
        boolean isOdd = len % 2 == 1;
        int i = (len  - 1) / 2, j = isOdd ? i : (len / 2);
        int p1 = i, p2 = j;
        while (p1 >= 0 && p2 < len) {
            if (a.charAt(p1) == a.charAt(p2)) {
                p1--;p2++;
            } else {
                break;
            }
        }
        int k = 0;
        for (;k <= p1;) {
            if (a.charAt(k) == b.charAt(len - 1 - k)) {
                k++;
            } else {
                break;
            }
        }
        if (k == p1 + 1) return true;
        
        
        p1 = i; p2 = j;
        while (p1 >= 0 && p2 < len) {
            if (b.charAt(p1) == b.charAt(p2)) {
                p1--;p2++;
            } else {
                break;
            }
        }
        // System.out.println(p1);
        k = 0;
        for (;k <= p1;) {
            if (a.charAt(k) == b.charAt(len - 1 - k)) {
                k++;
            } else {
                break;
            }
        }
        if (k == p1 + 1) return true;
        return false;
        
    }
    private boolean isPanli(String s) {
        int i = 0, j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;j--;
        }
        return true;
    }
}
