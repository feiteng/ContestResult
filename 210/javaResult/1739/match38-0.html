class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return a.charAt(0) == b.charAt(b.length()-1) || a.charAt(a.length()-1) == b.charAt(0) || isPalindrome(a) || isPalindrome(b);
    }
    
    private boolean isPalindrome(String a) {
        int i = 0, j = a.length()-1;
        while(i < j) {
            if(a.charAt(i) != a.charAt(j))
                return false;
            i++;
            j--;
        }
        return true;
    }
}
