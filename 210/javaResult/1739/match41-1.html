class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if(isPalindrome(a) || isPalindrome(b))
            return true;
        
        int i = 0, j = a.length() - 1, n = a.length();
        StringBuilder sb1 = new StringBuilder();
        StringBuilder sb2 = new StringBuilder();
        StringBuilder s = new StringBuilder();
        
        while(i < n / 2) {
            if(a.charAt(i) == b.charAt(j)) {
                sb1.append(a.charAt(i));
                sb2.append(b.charAt(j));
                s.append(a.charAt(i));
            }
            else {
                break;
            }
            
            ++i;
            --j;
        }
        
        
        while(i <= j) {
            sb1.append(a.charAt(i++));
        }
        
        
        sb1.append(sb2.reverse());
        
        if(isPalindrome(sb1.toString()))
            return true;
        
        i = s.length();
        
        while(i <= j) {
            s.append(b.charAt(i++));
        }
        
        s.append(sb2);
        
        if(isPalindrome(s.toString()))
            return true;
        
        s = new StringBuilder();
        sb1 = new StringBuilder();
        sb2 = new StringBuilder();
        i = 0;
        j = a.length() - 1;
        n = a.length();
        
        while(i < n / 2) {
            if(b.charAt(i) == a.charAt(j)) {
                sb1.append(b.charAt(i));
                sb2.append(a.charAt(j));
                s.append(b.charAt(i));
            }
            else {
                break;
            }
            
            ++i;
            --j;
        }
        
        
        while(i <= j) {
            sb1.append(b.charAt(i++));
        }
        
        sb1.append(sb2.reverse());
        
        if(isPalindrome(sb1.toString()))
            return true;
        
        i = s.length();
        
        while(i <= j) {
            s.append(a.charAt(i++));
        }
        
        s.append(sb2);
        
        if(
isPalindrome(s.toString()))
            return true; 
            
        return false;     
    }
    
    private boolean isPalindrome(String s) {
        int i = 0, j = s.length() - 1;
        
        while(i < j) {
            if(s.charAt(i) != s.charAt(j))
                return false;
            
            ++i;
            --j;
        }
        
        return true;
    }
}
