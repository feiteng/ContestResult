class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return helper(a, b) | helper(b, a);
    }
    
    private boolean helper(String a, String b) {
        int N = a.length();
        int le = 0, ri = N - 1;
        while (le < ri) {
            if (a.charAt(le) == b.charAt(ri)) {
                le++;
                ri--;
            } else {
                break;
            }
        }
        if (le >= ri) {
            return true;
        }
        int i = le, j = ri;
        while (i < j) {
            if (a.charAt(i) == a.charAt(j)) {
                i++;
                j--;
            } else {
                break;
            }
        }
        if (i >= j) {
            return true;
        }
        
while (le < ri) {
            if (b.charAt(le) == b.charAt(ri)) {
                le++;
                ri--;
            } else {
                break;
            }
        }
        return le >= ri;
    }
}
