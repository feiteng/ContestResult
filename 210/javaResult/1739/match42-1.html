class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        char[] arr1=a.toCharArray();
        char[] arr2=b.toCharArray();
        int hi=arr1.length-1;
        if(checkPalindrome(arr1) || checkPalindrome(arr2))
            
return true;
        
        
       
        
        return arr1[0]==arr2[hi] || arr2[0]==arr1[hi];
        
    }
    
    public boolean checkPalindrome(char[] s){
        int lo=0;
        int hi=s.length-1;
        
        while(lo<hi){
            if(s[lo]!=s[hi])
                return false;
            lo++;
            hi--;
        }
        return true;
    }
}
