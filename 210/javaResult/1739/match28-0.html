class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        int aCommonLen = findCommonLen(a, b);
        int bCommonLen = findCommonLen(b, a);
        
        int aMiddleCommon = findMiddle(a);
        int bMiddleCommon = findMiddle(b);
        
        return 2 * aCommonLen + aMiddleCommon >= a.length() ||
               2 * aCommonLen + bMiddleCommon >= a.length() ||
               2 * bCommonLen + aMiddleCommon >= a.length() ||
               2 * bCommonLen + bMiddleCommon >= a.length();
    }
    
    int findCommonLen(String a, String b) {
        int aLeft = 0;
        int bRight = b.length() - 1;
        while (aLeft < a.length() / 2 && a.charAt(aLeft) == b.charAt(bRight)) {
            aLeft++;
            bRight--;
        }
        
        return aLeft;
    }
    
    int findMiddle(String a) {
        int left = 0;
        int right = 0;
        if (a.length() % 2 == 0) {
            left = a.length() / 2 - 1;
            right = a.length() / 2;
        } else {
            left = a.length() / 2 - 1;
            right = a.length() / 2 + 1;
        }
        
        while (left >= 0 && right < a.length() && a.charAt(left) == a.charAt(right)) {
            left--;
            right++;
        }
        
        return right - left - 1;
    }
}
