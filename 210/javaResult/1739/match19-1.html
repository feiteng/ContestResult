class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return prePost(a, b) || prePost(b, a);
    }
    private boolean prePost(String a, String b) {
        char[] ac = a.toCharArray();
        char[] bc = b.toCharArray();
        int n = ac.length;
        int e = n - 1; 
        int i = 0;
        while (i < e && ac[i] == bc[e]) {
            i++;
            e--;
        }
        
if (i > e) return true;
        return isPalindrome(a.substring(i, e+1)) || isPalindrome(b.substring(i, e+1));
    }
    private boolean isPalindrome(String s) {
        int n = s.length();
        for (int i = 0; i < n / 2; i++) {
            if (s.charAt(i) != s.charAt(n-i-1)) return false;
        }
        return true;
    }
}
