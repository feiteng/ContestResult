class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        return check(a,b) || check(b,a);
    }
    
    public boolean check(String a, String b){
        int n = a.length();
        int i=0, j=n-1;
        while(i<j){
            if(a.charAt(i)==b.charAt(j)){
                i++;
                j--;
            }else{
                break;
            }
        }
        int x = i, y = j;
        boolean fa = true;
        while(x<y){
            if(a.charAt(x)==a.charAt(y)){
                x++;
                y--;
            }else{
                fa = false;
                break;
            }
        }
        x = i;
        y = j;
        boolean fb = true;
        while(x<y){
            if(b.charAt(x)==b.charAt(y)){
                x++;
                y--;
            }else{
                fb = false;
                break;
            }
        }
        return fa || fb;
    }
}
