class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        if (isHuiWen(a) || isHuiWen(b)) {
            return true;
        }
        int len = a.length();
        return 
a.charAt(0) == b.charAt(len - 1) || a.charAt(len - 1) == b.charAt(0);
    }

    private boolean isHuiWen(String a) {
        int len = a.length();
        if (len == 1) {
            return true;
        }
        for (int i = 0; i < len / 2; i++) 
{
            if (a.charAt(i) != a.charAt(len - 1 - i)) {
                return false;
            }
        }
        return true;
    }
}
