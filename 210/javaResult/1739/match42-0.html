class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        char c1[] = a.toCharArray();
        char c2[] = b.toCharArray();
        if(isString(c1) || isString(c2)) return true;
        int n = c1.length;
        int m = c2.length;
        int i = 0;
        int j = m - 1;
        while(i < n && j >= 0){
            if(c1[i] != c2[j]) break;
            ++i;
            --j;
        }
        if(i == m - j - 1 && i != 0) return true;
        i = n - 1;
        j = 0;
        while(i >= 0 && j < m){
            if(c1[i] != c2[j]) break;
            --i;
            ++j;
        }
        if(j == m - i - 1 && j != 0) return true;
        return false;
        
    }

    public boolean isString(char ch[]){
        int low = 0,high = ch.length - 1;
        while(low < high){
            if(ch[low] != ch[high]) return false;
            ++low;
            --high;
        }
        return true;
    }
}
