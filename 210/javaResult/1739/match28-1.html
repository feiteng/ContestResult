class Solution {
    public boolean checkPalindromeFormation(String a, String b) {
        int matchAB = findMatchIndex(a, b);
        int matchBA = findMatchIndex(b, a);
        int palinA = findPalin(a);
        int palinB = findPalin(b);
        return matchAB >= palinA || matchAB >= palinB || matchBA >= palinA || matchBA >= palinB;
    
}
    private int findMatchIndex(String s1, String s2) {
        int i = 0;
        int j = s1.length() - 1;
        while (i <= j && s1.charAt(i) == s2.charAt(j)) 
{
            i++;
            j--;
        }
        return i - 1;
    }
    private int findPalin(String s) {
        int i = (s.length() - 1) / 2;
        int j = s.length() / 2;
        while (i >= 0 && 
s.charAt(i) == s.charAt(j)) {
            i--;
            j++;
        }
        return i;
    }
}
