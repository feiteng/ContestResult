class Solution {

  public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
    boolean[][] conn = new boolean[n + 1][n + 1];
    List<Integer>[] adj = new List[n + 1];
    for (int i = 1; i <= n; i++) {
      adj[i] = new ArrayList<>();
    }
    for (int[] edge : edges) {
      conn[edge[0]][edge[1]] = true;
      conn[edge[1]][edge[0]] = true;
      adj[edge[0]].add(edge[1]);
      adj[edge[1]].add(edge[0]);
    }
    int[] res = new int[n - 1];
    for (int i = 1; i < 1 << n; i++) {
//      System.out.println(Integer.toBinaryString(i));
      int first = 1;
      for (int j = 1; j <= n; j++) {
        if ((i & (1 << (j - 1))) != 0) {
          first = j;
          break;
        }
      }
      if (isConn(conn, n, i)) {
        int maxDepth = getMaxDepth(adj, i, first, -1);
        if (maxDepth > 0) {
          res[maxDepth - 1]++;
        }
      }
    }
    return res;
  }

  private int getMaxDepth(List<Integer>[] adj, int mask, int cur, int p) {
    int max1 = 0, max2 = 0, res = 0;
    for (int next : adj[cur]) {
      if (next != p && (mask & (1 << (next - 1))) != 0) {
        int num = getDepth(adj, mask, next, cur);
        if(num >= max1){
          max2 = max1;
          max1 = num;
        } else if(num >= max2){
          max2 = num;
        }
        res = Math.max(res, getMaxDepth(adj, mask, next, cur));
      }
    }
    return Math.max(max1 + max2, res);
  }

  private int getDepth(List<Integer>[] adj, int mask, int cur, int p) {
    int max = 0;
    for (int next : adj[cur]) {
      if (next != p && (mask & (1 << (next - 1))) != 0) {
        max = Math.max(max, getDepth(adj, mask, next, cur));
      }
    }
    return 1 + max;
  }

  private int[] parents;

  private boolean isConn(boolean[][] conn, int n, int cur) {
    parents = new int[n + 1];
    for (int i = 1; i <= n; i++) {
      parents[i] = i;
    }
    for (int i = 1; i <= n; i++) {
      if ((cur & (1 << (i - 1))) == 0) {
        continue;
      }
      for (int j = 1; j <= n; j++) {
        if ((cur & (1 << (j - 1))) == 0) {
          continue;
        }
        if (conn[i][j]) {
          union(i, j);
        }
      }
    }
    int res = -1;
    for (int i = 1; i <= n; i++) {
      if ((cur & (1 << (i - 1))) == 0) {
        continue;
      }
      int p = find(i);
      if (res == -1) {
        res = p;
      } else if (res != p) {
        return false;
      }
    }
    return true;
  }

  private void union(int u, int v) {
    int pu = find(u), pv = find(v);
    if (pu == pv) {
      return;
    }
    parents[pu] = pv;
  }

  private int find(int u) {
    if (parents[u] != u) {
      parents[u] = find(parents[u]);
    }
    return parents[u];
  }


}
