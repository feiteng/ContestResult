class Solution {
    public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
        int[][] dist = new int[n][n];
        int[] g = new int[n];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                dist[i][j] = 1000;
            }
            dist[i][i] = 0;
        }
        for (int[] uv : edges) {
            int u = uv[0]-1;
            int v = uv[1]-1;
            dist[u][v] = 1;
            dist[v][u] = 1;
            g[v]|=(1<<u);
            g[u]|=(1<<v);
        }
        for (int k = 0; k < n; ++k) {
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    dist[i][j] = Math.min(dist[i][j],dist[i][k]+dist[k][j]);
                }
            }
        }
        ArrayDeque<Integer> q = new ArrayDeque<>();
        int[] dp = new int[1<<n];
        for (int i = 0; i < n; ++i) {
            q.add(1<<i);
        }
        int[] ans = new int[n-1];
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            for (int j = 0; j < n; ++j) {
                int v = u|(1<<j);
                //System.out.println(u+" "+v);
                if (u==v)
                    continue;
                if ((g[j]&u)==0)
                    continue;
                if (dp[v]>0)
                    continue;
                dp[v] = dp[u];
                for (int k = 0; k < n; ++k) {
                    if (((1<<k)&u)>0)
                        dp[v] = Math.max(dp[v],dist[j][k]);
                }
                ans[dp[v]-1]++;
                q.add(v);
            }
        }
        return ans;
    }
}
