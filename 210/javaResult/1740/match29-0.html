class Solution {
    public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
        int[] sol = new int[n-1];
        int[][] dis = new int[n][n];
        boolean[][] conn = new boolean[n][n];
        
        
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < n; i++) adj.add(new ArrayList<>());
        for (int i = 0; i < edges.length; i++) {
            int a = edges[i][0] - 1;
            int b = edges[i][1] - 1;
            adj.get(a).add(b);
            adj.get(b).add(a);
            conn[a][b] = conn[b][a] = true;
        }
        
        for (int i = 0; i < n; i++) {
            Queue<Integer> q = new LinkedList<>();
            q.add(i);
            Set<Integer> set = new HashSet<>();
            set.add(i);
            int val = 0;
            while (!q.isEmpty()) {
                int len = q.size();
                for (int j = 0; j < len; j++) {
                    int cur = q.poll();
                    dis[i][cur] = val;
                    //System.out.println("dis_" + i + "_" + cur + " is " + val);
                    for (int next : adj.get(cur)) {
                        if (set.add(next)) {
                            q.add(next);
                        }
                    }
                }
                val++;
            }
        }
        
        for (int i = 0; i < (1<<n); i++) {
            List<Integer> cur = new ArrayList<>();
            for (int j = 0; j < n; j++) {
                if (((i>>j)&1) != 0) cur.add(j);
            }
            
            int ct = 0;
            int max = 0;
            //System.out.println("current bit value is " + i);
            for (int j = 0; j < cur.size(); j++) {
                int a = cur.get(j);
                //System.out.print(a);
                //System.out.print(" ");
                for (int k = j + 1; k < cur.size(); k++) {
                    int b = cur.get(k);
                    if (conn[a][b]) ct++;
                    max = Math.max(max, dis[a][b]);
                }
            }
            if (ct == cur.size() - 1 && max > 0) {
                sol[max-1]++;
            }
            //System.out.println("");
            //System.out.println("max dis is " + max);
        }
        return sol;
    }
    
   
}
