class Solution {
    public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
        int[] ret = new int[n-1];
        int[][] graph = new int[n][n];
        
        int[][] graph2 = new int[n][n];
        for (int[] edge : edges) {
            edge[0]--;
            edge[1]--;
            graph2[edge[0]][edge[1]] = 1;
            graph2[edge[1]][edge[0]] = 1;
        }
        for (int[] edge : edges) {
                graph[edge[0]][edge[1]] = 1;
                graph[edge[1]][edge[0]] = 1;
            }
            for (int k = 0; k < n; k++) {
                for (int i = 0; i < n; i++) {
                    for (int l = 0; l < n; l++) {
                        if (i == l || i == k || k == l)
                            continue;
                        if (graph[i][k] > 0 && graph[k][l] > 0) {
                            if (graph[i][l] == 0) {
                                graph[i][l] = graph[i][k] + graph[k][l];
                            } else {
                                graph[i][l] = Math.min(graph[i][k] + graph[k][l], graph[i][l]);
                            }
                        }
                    }
                }
            }
        for (int j = 1; j < 1<<n; j++) {
            
            boolean[] seen = new boolean[n];
            int num = 0;
            Queue<Integer> q = new LinkedList<>();
            for (int k = 0; k < n; k++) {
                if (((1<<k)&j) != 0 && q.size() == 0) {
                    q.add(k);
                    seen[k] = true;
                    num++;
                }
                if (((1<<k)&j) != 0) {
                    num--;
                }
            }
            while (q.size() > 0) {
                int yo = q.poll();
                for (int k = 0; k < n; k++) {
                    if (((1<<k)&j) != 0 && graph2[yo][k] == 1 && !seen[k]) {
                        num++;
                        seen[k] = true;
                        q.add(k);
                    }
                }
            }
            int max = 0;
            boolean conn = num == 0;
            for (int k = 0; k < n; k++) {
                for (int l = 0; l < n; l++) {
                    if (((1<<k)&j) != 0 && ((1<<l)&j) != 0 && k != l) {
                    max = Math.max(max, graph[k][l]);
                    }
                }
            }
            if (conn && max > 0)
                ret[max-1]++;
            
            //System.out.printf("max: %d, conn: %d, j: %d\n", max, conn ? 1 : 0, j);
        }
        return ret;
    }
}
