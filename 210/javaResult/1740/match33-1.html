class Solution {
    
    int ct = 0;
    int res= 0;
    Map<Integer,Integer> dep = new HashMap<>();
    public void dfs(int root,Set<Integer> st,int fa){
        int fir = 0;
        int se = 0;
        for(Object go1:g[root]){
            int go  = (int) go1;
            if(go==fa) continue;
            if(st.contains(go)){
                ct++;
                dfs(go,st,root);
                
int cur =  dep.get(go);
                if(cur>fir){
                    se = fir;
                    fir = cur;
                }else if(cur>se){
                    se = cur;
                }
            }
        }
        res  = Math.max(res, fir+se);
        dep.put(root, fir+1);
    }
    
    List g[] ;
    public int[] countSubgraphsForEachDiameter(int n, int[][] es) {
        
        g = 
new List[n+1];
        
        for(int i=0;i<=n;++i){
            g[i] = new ArrayList();
        }
        
        for(int e[]:es){
            
g[e[0]].add(e[1]);
            g[e[1]].add(e[0]);
        }
        
        int d[] = new int[n-1];
        
        for(int st = 1 ;st< 1<<n ;++st){
            List<Integer> li  = new ArrayList<>();
            Set<Integer> st1 = new HashSet<>();
            
            for(int j=0;j<n;++j){
                if((st&(1<<j))>0){
                    li.add(j+1);
                    st1.add(j+1);
                }
            }
            ct = 1;
            dep = new HashMap<>();
            dep.put(li.get(0),0);
            res = 0;
            dfs(li.get(0),st1,-1);
            
            if(ct==li.size()){
                
if(res>=1){
                    d[res-1] ++;
                
                
                }
            }
            
            
        }
        return d;
    }
}
