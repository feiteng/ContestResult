class Solution {
    public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
        boolean[][] map = new boolean[n][n];
        for(int[] edge : edges){
            map[edge[0] - 1][edge[1] - 1] = true;
            map[edge[1] - 1][edge[0] - 1] = true;
        }
        
        int[][] dis = new int[n][n];
        
        // get distance
        for(int i = 0; i < n; i++){
            Queue<Integer> q = new LinkedList<>();
            q.add(i);
            boolean[] record = new boolean[n + 1];
            record[i] = true;
            int curDis = 0;
            
while(!q.isEmpty()){
                int size = q.size();
                for(int j = 0; j < size; j++){
                    int cur = q.poll();
                    dis[i][cur] = curDis;
                    for(int k = 0; k < n; k++){
                        if(map[cur][k] && !record[k]){
                            record[k] = true;
                            q.add(k);
                        
}
                    }
                }
                curDis++;
            }
        }
        
//         for(int i = 0; i < n; i++){
//             System.out.println(Arrays.toString(dis[i]));
//         }
        
        int[] mask = new int[1 << n];
        int[] res = new int[n - 1];
        Queue<Integer> q = new LinkedList<>();
        for(int i = 0; i < n; i++){
            int cur = 1 << i;
            q.add(cur);
        
}
        
        while(!q.isEmpty()){
            int cur = q.poll();
            for(int i = 0; i < n; i++){
                if((cur & (1 << i)) > 0){
                    for(int j = 0; j < n; j++){
                        // new dege
                        if((cur & (1 << j)) == 0 && map[i][j]){
                            int key = (cur | (1 << j));
                            // new group
                            if(mask[key] == 0){
                                int curDis = mask[cur];
                                for(int k = 0; k < n; k++){
                                    if((cur & (1 << k)) > 0){
                                        curDis = Math.max(curDis, dis[j][k]);
                                    }
                                }
                                
                                mask[key] = curDis;
                                res[curDis - 1]++;
                                q.add(key);
                            }
                        }
                    }
                
}
            }
        }
        
        
        return res;
    }
    
}
