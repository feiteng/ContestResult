class Solution {
    public int[] countSubgraphsForEachDiameter(int n, int[][] edges) {
        boolean[][] g = new boolean[n][n];
        int[][] dis = new int[n][n];
        for(int j=0;j<n;j++){
            Arrays.fill(dis[j],100);
            dis[j][j]=0;
        }
        for(int[] e: edges){
            
dis[e[0]-1][e[1]-1] = 1;
            dis[e[1]-1][e[0]-1] = 1;
        }
        for(int k=0;k<n;k++){
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    dis[i][j] = Math.min(dis[i][j],dis[i][k]+dis[k][j]);
                }
            }
        }
        
        int[] count = new int[n-1];
        Queue<Integer> q = new ArrayDeque<>();
        Set<Integer> set = new HashSet<>();
        for(int i=0;i<n;i++){
            q.add(1<<i);
            set.add(1<<i);
        }
        while(!q.isEmpty()){
            int s = q.remove();
            int max = 0;
            
for(int i=0;i<n;i++){
                if((s>>i)%2==0) continue;
                for(int j=0;j<n;j++){
                    if((s>>j)%2==0){
                        if(!set.contains(s | (1<<j))  && dis[i][j]==1){
                            set.add(s | (1<<j));
                            q.add(s | (1<<j));
                        }
                    
}else {
                        max = Math.max(max,dis[i][j]);
                    }
                }
            }
            if(max==0) continue;
            count[max-1] ++;
        }
        
        
        // for(int i=0;i<n;i++){
        //     for(int j=0;j<n;j++){
        //         System.out.printf("%d,%d: %d\n",i,j,dis[i][j]);
        //     }
        // }
        //     System.out.println(Integer.toBinaryString(s)+" "+max);
            
        // }
        return count;
    }
}
