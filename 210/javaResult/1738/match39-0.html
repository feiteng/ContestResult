class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        boolean[][] grid = new boolean[n][n];
        int[] t = new int[n];
        for (int i = 0; i < roads.length; i++) {
            grid[roads[i][0]][roads[i][1]] = true;
            grid[roads[i][1]][roads[i][0]] = true;
            t[roads[i][0]]++;
            t[roads[i][1]]++;
        }
        int max = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int z = t[i] + t[j];
                if (grid[i][j]) {
                    z--;
                }
                max = Math.max(z, max);
            }
        }
        return max;
    }
}
