class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[][] g = new int[n][n];
        int[] d = new int[n];
        for (int[] r : roads) {
            g[r[0]][r[1]] = 1;
            g[r[1]][r[0]] = 1;
            d[r[0]]++;
            d[r[1]]++;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                ans = Math.max(ans, d[i] + d[j] - g[i][j]);
            }
        }
        return ans;
    }
}
