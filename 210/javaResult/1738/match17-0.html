class Solution {
   public int maximalNetworkRank(int n, int[][] roads) {
      boolean [][]graph = new boolean[n][n];
      int []dp = new int[n];
      for (int []road : roads) {
        graph[road[0]][road[1]] = true;
        graph[road[1]][road[0]] = true;
        dp[road[0]]++;
        dp[road[1]]++;
      }
      int max = 0;
      for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
          max = Math.max(max, dp[i] + dp[j] - (graph[i][j] ? 1 : 0));
        }
      }
      return max;
  }
}
