class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int [][] edge = new int[n][n];
        int [] du = new int [n];
        for(int [] e : roads){
            du[e[0]] ++;
            du[e[1]] ++;
            edge[e[0]][e[1]] = 1;
            edge[e[1]][e[0]] = 1;
        }
        int ans = 0;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                ans = Math.max(ans, du[i] + du[j] - edge[i][j]);
            }
        }
        return ans;
    }
}
