class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[][] conn = new int[n][n];
        int[] con = new int[n];
        for (int[] road : roads) {
            con[road[0]]++;
            con[road[1]]++;
            conn[road[0]][road[1]] = 1;
            conn[road[1]][road[0]] = 1;
        }
        
        int max = 0;
        
        for (int i = 0; i < n-1; i++) {
            for (int j = i+1; j < n; j++) {
                max = Math.max(con[i]+con[j] - conn[i][j], max);
            }
        }

        return max;
    }
}
