class Solution {
    public int maximalNetworkRank(int n, int[][] rs) {
        int[] cs = new int[n];
        
        boolean[][] connect = new boolean[n][n];
        for (int[] r: rs) {
            cs[r[0]] ++;
            cs[r[1]] ++;
            connect[r[0]][r[1]] = true;
            connect[r[1]][r[0]] = true;
        }
        int max = 0;
        for (int i = 0; i < n; i ++) {
            for (int j = i + 1; j < n; j ++) {
                max = Math.max(cs[i] + cs[j] - (connect[i][j]? 1:0), max);
            }
        }
        return max;
    }
}
