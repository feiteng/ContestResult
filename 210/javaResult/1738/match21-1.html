class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {

        boolean[][] hasRoad = new boolean[n][n];
        int[] cnt = new int[n];
        for (int[] road : roads) {
            int v = road[0], w = road[1];
            cnt[v]++;
            cnt[w]++;
            hasRoad[v][w] = true;
            hasRoad[w][v] = true;
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int roadCnt = cnt[i] + cnt[j];
                if (hasRoad[i][j]) {
                    roadCnt--;
                }
                ans = Math.max(ans, roadCnt);
            }
        }
        return ans;

    }
}
