class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] degrees = new int[n];
        int[][] edges = new int[n][n];
        
        for(int[] road : roads){
            degrees[road[0]] += 1;
            degrees[road[1]] += 1;
            edges[road[0]][road[1]] = 1;
            edges[road[1]][road[0]] = 1;
        }
        
        int res = 0;
        int count = 0;
        
for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                res = Math.max(res, degrees[i] + degrees[j] - edges[i][j]);
            }
        }
        return res;
    }
}
