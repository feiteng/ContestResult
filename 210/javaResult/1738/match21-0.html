class Solution {
public int maximalNetworkRank(int n, int[][] roads) {
        int[] degree = new int[n];
        boolean[][] con = new boolean[n][n];
        for (int i = 0; i < roads.length; i++) {
            int x = roads[i][0];
            int y = roads[i][1];
            degree[x]++;
            degree[y]++;
            con[x][y] = con[y][x] = true;

        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int sum = degree[i] + degree[j];
                if (con[i][j]) {
                    sum--;
                }
                ans = Math.max(ans, sum);
            }
        }
        return ans;
    }

}
