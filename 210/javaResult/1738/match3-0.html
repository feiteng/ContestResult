class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[][] u = new int[n][n];
        int[] count = new int[n];
        for(int[] road : roads) {
            count[road[0]]++;
            count[road[1]]++;
            u[road[0]][road[1]]++;
            u[road[1]][road[0]]++;
        }
        int ans = 0;
        for(int i =0 ; i< n; i++) {
            for(int j = i + 1;j < n; j++) {
                ans = Math.max(count[i]+count[j] - u[i][j],ans);
            }
        }
        return ans;
        
    }
}
