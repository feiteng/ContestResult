class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] d = new int[n];
        boolean[][] c = new boolean[n][n];
        for (int[] r : roads) {
            d[r[0]]++;
            d[r[1]]++;
            c[r[0]][r[1]] = true;
            c[r[1]][r[0]] = true;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                res = Math.max(res, d[i] + d[j] - (c[i][j] ? 1 : 0));
            }
        }
        return res;
    }
}
