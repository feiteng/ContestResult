	class Solution {
		public int maximalNetworkRank(int n, int[][] roads) {
			int[] deg = new int[n];
			boolean[][] has = new boolean[n][n];
			for(int[] r : roads){
				deg[r[0]]++;
				deg[r[1]]++;
				has[r[0]][r[1]] = has[r[1]][r[0]] = true;
			}
			int ret = 0;
			for(int i = 0;i < n;i++){
				for(int j = i+1;j < n;j++){
					int v = deg[i] + deg[j];
					if(has[i][j])v--;
					ret = Math.max(ret, v);
				}
			}
			return ret;
		}
	}
