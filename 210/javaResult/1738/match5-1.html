class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] count = new int[n];
        boolean[][] connected = new boolean[n][n];
        for (int[] r : roads) {
            count[r[0]]++;
            count[r[1]]++;
            connected[r[0]][r[1]] = true;
            connected[r[1]][r[0]] = true;
        }
        int max = 0;
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                max = Math.max(max, count[i] + count[j] - (connected[i][j] ? 1 : 0));
            }
        }
        return max;
    }
}
