class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int dis[] = new int[n];
        boolean matrix[][] = new boolean[n][n];
        for(int i = 0;i < roads.length;++i){
            dis[roads[i][0]] += 1;
            dis[roads[i][1]] += 1;
            matrix[roads[i][0]][roads[i][1]]  = true;
            matrix[roads[i][1]][roads[i][0]]  = true;
        }
        int max = 0;
        for(int i = 0;i < n;++i){
            for(int j = i+1;j < n ;++j){
                int ans = dis[i] + dis[j];
                if(matrix[i][j]) ans -= 1;
                max = Math.max(max,ans);
            }
        }
        
        return max;
    }
}
