class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] deg = new int[n];
     
        boolean[][] connected = new boolean[n][n];
        for (int i = 0; i < roads.length; i++) {
            int a = roads[i][0];
            int b = roads[i][1];
            deg[a]++;
            deg[b]++;
            connected[a][b] = connected[b][a] = true;
        }
        
        int sol = 0;
        for (int i = 0; i < n; i++){
            for (int j = i + 1; j < n; j++) {
                int cur = deg[i] + deg[j];
                if (connected[i][j]) cur--;
                sol = Math.max(sol, cur);
            }
        }
        return sol;
      
    }
}
