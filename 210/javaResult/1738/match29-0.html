class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        Map<Integer, HashSet<Integer>> mp = new HashMap<>();
        for(int i = 0; i < n; i++) mp.put(i, new HashSet<Integer>());
        for(int[] road : roads) {
            mp.get(road[0]).add(road[1]);
            mp.get(road[1]).add(road[0]);
        }
        int ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                int tmp = mp.get(i).size() + mp.get(j).size();
                if(mp.get(i).contains(j)) tmp--;
                ans = Math.max(ans, tmp);
            }
        }
        return ans;
    }
}
