class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] counts = new int[n];
        Set<Integer>[] sets = new Set[n];
        for (int i = 0;i < n;i++) {
            sets[i] = new HashSet<>();
        }
        for (int[] r : roads) {
            sets[r[0]].add(r[1]);
            sets[r[1]].add(r[0]);
            counts[r[0]]++;
            counts[r[1]]++;
        }
        int res = 0;
        for (int i = 0;i < n;i++) {
            for (int j = i + 1;j < n;j++) {
                int cur = counts[i] + counts[j];
                if (sets[i].contains(j)) {
                    cur--;
                }
                res = Math.max(res, cur);
            }
        }
        return res;
    }
}
