class Solution {
    public int maximalNetworkRank(int n, int[][] rs) {
        int d[]  = new int[n+1];
         int in[][]  = new int[n+1][n+1];
        for(int ps[]:rs){
            d[ps[0]]++;d[ps[1]]++;
            in[ps[0]][ps[1]]++;
            in[ps[1]][ps[0]]++;
        
}
        int r = 0;
        for(int i=0;i<n;++i){
            for(int j=i+1;j<n;++j){
               r = Math.max(r, d[i]+d[j]-in[i][j]);
                
            }
        }
        return r;
    }
}
