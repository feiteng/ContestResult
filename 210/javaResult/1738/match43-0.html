class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] count = new int[n];
        boolean[][] isConnected = new boolean[n][n];
        
        for (int[] road : roads) {
            count[road[0]]++;
            count[road[1]]++;
            isConnected[road[0]][road[1]] = true;
            isConnected[road[1]][road[0]] = true;
        }
        
        int max = 0;
        for (int i=0; i<n; i++) {
            for (int j=i+1; j<n; j++) {
                int connect = count[i] + count[j] - (isConnected[i][j] ? 1 : 0);
                max = Math.max(max, connect);
            }
        }
        return max;
    }
}
