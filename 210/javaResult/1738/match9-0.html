class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] indegree = new int[n];
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] road : roads) {
            indegree[road[0]]++;
            indegree[road[1]]++;
            map.computeIfAbsent(road[0], k -> new HashSet<>()).add(road[1]);
            map.computeIfAbsent(road[1], k -> new HashSet<>()).add(road[0]);
        }
        //Arrays.sort(indegree);
        int res = 0;
        for (int i = 0; i < indegree.length; i++) {
            for (int j = i + 1; j < indegree.length; j++) {
                int cur = indegree[i] + indegree[j];
                if (map.get(i) != null && map.get(i).contains(j)) {
                    cur--;
                }
                res = Math.max(res, cur);
            }
        }
        return res;
    }
}
