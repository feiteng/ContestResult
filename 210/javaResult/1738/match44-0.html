class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[][] e = new int[n][n];
        int[] deg = new int[n];
        for (int[] r : roads) {
            int v1 = r[0];
            int v2 = r[1];
            e[v1][v2] = 1;
            e[v2][v1] = 1;
            deg[v1]++;
            deg[v2]++;
        }

        int max = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int rank = deg[i] + deg[j] - e[i][j];
                max = Math.max(max, rank);
            }
        }
        return max;
    }
}
