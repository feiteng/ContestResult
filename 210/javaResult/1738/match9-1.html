class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] count = new int[n];
        HashMap<Integer, HashSet<Integer>> pair = new HashMap<>();
        
        for(int[] road : roads){
            count[road[0]]++;
            count[road[1]]++;
            pair.computeIfAbsent(road[0], id -> new HashSet<>()).add(road[1]);
            pair.computeIfAbsent(road[1], id -> new HashSet<>()).add(road[0]);
        }
        
        
        int max = 0;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                int cur = count[i] + count[j];
                if(pair.containsKey(i) && pair.get(i).contains(j)){
                    cur--;
                }
                
                max = Math.max(max, cur);
            }
        }
        
        return max;
    }
}
