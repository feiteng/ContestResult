class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[][] con = new int[n][n];
        int[] in = new int[n];
        for (int[] r : roads) {
            con[r[0]][r[1]]++;
            con[r[1]][r[0]]++;
            in[r[0]]++;
            in[r[1]]++;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                res = Math.max(res, in[i] + in[j] - con[i][j]);
            }
        }
        
        return res;
    }
}
