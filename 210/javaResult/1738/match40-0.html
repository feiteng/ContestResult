class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        boolean[][] graph = new boolean[n][n];
        int[] count = new int[n];
        for (int[] r : roads) {
            graph[r[1]][r[0]] = graph[r[0]][r[1]] = true;
            count[r[0]]++;
            count[r[1]]++;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                int sum = count[i] + count[j];
                if (graph[i][j]) sum--;
                ans = Math.max(ans, sum);
            }
        }
        return ans;
    }
}
