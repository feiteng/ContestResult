class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        boolean [][] con = new boolean [n][n];
        int [] deg = new int [n];
        for (int i = 0; i<roads.length; ++i) {
            int a = roads[i][0];
            int b = roads[i][1];
            con[a][b] = true;
            con[b][a] = true;
            deg[a]++;
            deg[b]++;
        }
        int best = 0;
        for (int i = 0; i<n; ++i) {
            for (int j = i+1; j<n; ++j) {
                int d1 = deg[i];
                int d2 = deg[j];
                int tot = d1 + d2;
                if (con[i][j]) tot--;
                best = Math.max(tot, best);
            }
        }
        return best;
            
        
    }
}
