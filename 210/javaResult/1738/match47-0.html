class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] nums = new int[n];
        boolean[][] store = new boolean[n][n];
        for (int[] road : roads) {
            nums[road[0]]++;
            nums[road[1]]++;
            store[road[0]][road[1]] = true;
            store[road[1]][road[0]] = true;
        }
        int max = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                int now = nums[i] + nums[j];
                if (store[i][j]) {
                    now--;
                }
                max = Math.max(max, now);
            }
        }
        
        return max;
    }
}
