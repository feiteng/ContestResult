class Solution {
    public int maximalNetworkRank(int n, int[][] roads) {
        int[] in = new int[n];
        boolean[][] connected = new boolean[n][n];
        for(int[] r : roads){
            int u= r[0];
            int v= r[1];
            in[u]++;
            in[v]++;
            connected[u][v]=connected[v][u]=true;
        }
        int ans =0;
        for(int i=0; i<n;++i){
            for(int j=i+1;j<n;++j){
                int cur = in[i]+in[j]-(connected[i][j]? 1 :0);
                ans=Math.max(ans, cur);
            }
        }
        return ans;
    }
}
