class Solution {
    public int maxDepth(String s) {
        int d = 0, maxD = 0;
        for (char c : s.toCharArray()) {
            if (c == '(') {
                d++;
            } else if (c == ')') {
                d--;
            }
            maxD = Math.max(d, maxD);
        }
        return maxD;
    }
}
