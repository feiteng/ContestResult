class Solution {
  public int maxDepth(String s) {
    int best = 0;
    int curr = 0;
    for (char ch : s.toCharArray()) {
      if (ch == '(') {
        curr++;
      } else if (ch == ')') {
        curr--;
      }
      best = Math.max(best, curr);
    }
    return best;
  }
}
