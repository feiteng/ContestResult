class Solution {
    public int maxDepth(String s) {
        int cur = 0;
        int max = 0;
        
        for(char ch : s.toCharArray()) {
            if(ch == '(')
                ++cur;
            else if(ch == ')')
                --cur;
            
            max = Math.max(cur, max);
        }
        
        return max;
    }
}
