class Solution {
    public int maxDepth(String s) {
        int res = 0;
        int cnt = 0;
        for(int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if(ch == '(')
                cnt++;
            else if(ch == ')')
                cnt--;
            res = Math.max(res, cnt);
        }
        return res;
    }
}
