class Solution {
    public int maxDepth(String s) {
        int d = 0;
        int res = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '(') {
                d++;
                res = Math.max(res, d);
            } else if (c == ')') {
                d--;
            }
        }
        
        return res;
    }
}
