class Solution {
    public int maxDepth(String s) {
        int d = 0;
        int dMax = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '(') {
                d += 1;
                dMax = Math.max(dMax, d);
            } else if (c == ')') {
                d -= 1;
            }
        }
        return dMax;
    }
}
