class Solution {
    public int maxDepth(String s) {
        int ans = 0,l = 0;
        for(char c: s.toCharArray()){
            if(c=='('){
                l++;
            }else if(c==')'){
                l--;
            }
            ans = Math.max(l,ans);
        }
        return ans;
    }
}
