class Solution {
    public int maxDepth(String s) {
        int ret = 0, curr = 0;
        for(char c : s.toCharArray()) {
            if(c == '(') {
                curr++;
            }
            else if(c == ')') curr--
;
            ret = Math.max(ret, curr);
        }
        return ret;
    }
}
