class Solution {
    public int maxDepth(String s) {
        int p  =0;int r = 0;
        for(char c:s.toCharArray()){
            if(c=='('){
                p++;
            }else if(c==')'){
                p--;
            }
            r =Math.max(r,p);
        }
        return r;
    }
}
