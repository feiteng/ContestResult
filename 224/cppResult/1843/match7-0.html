class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int maxx=0;
        for(auto it:rectangles)maxx=max(maxx,min(it[0],it[1]));
        int ct=0;
        for(auto it:rectangles){
            if(min(it[0],it[1])==maxx)ct++;
        }
        return ct;
    }
};
