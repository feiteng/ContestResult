class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rect) {
        int mx = 0;
        for (auto v: rect)
            mx = max(mx, min(v[0], v[1]));
        
        int res = 0;
        for (auto v: rect) {
            if (mx == min(v[0], v[1]))
                res += 1;
        }
        return res;
    }
};
