class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) 
    {
        int temp=INT_MIN;
        for(auto i:rectangles)
            temp=max(min(i[0],i[1]),temp);
        int count=0;
        for(auto i:rectangles)
        {
            if(temp<=min(i[0],i[1]))
                count++;
        }
        return count;
    }
};
