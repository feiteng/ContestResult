class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int n = rectangles.size();
        int mx = 0;
        for(int i=0;i<n;++i){
            int curr = min(rectangles[i][0],rectangles[i][1]);
            mx = max(curr,mx);
        }
        
        int ans = 0;

        for(int i=0;i<n;++i){
            int curr = min(rectangles[i][0],rectangles[i][1]);
            if(curr == mx) ans++;
        }
        
        return ans;
        
    }
};
