class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int mx = 0;
        for (auto r : rectangles) mx = max(mx, min(r[0], r[1]));
        int res = 0;
        for (auto r : rectangles) {
            if (min(r[0], r[1]) == mx) res++;
        }
        return res;
    }
};
