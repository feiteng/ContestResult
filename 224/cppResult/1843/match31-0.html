class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rec) {
        int n = rec.size();
        int l = 0;
        for(int i = 0; i < n; ++i) {
            l = max(min(rec[i][0], rec[i][1]), l);
        }
        int ans = 0;
        for(int i = 0; i <n; ++i) {
            if(min(rec[i][0], rec[i][1]) == l) ans ++;
        }
        return ans;
    }
};
