class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& r) {
        int n = r.size();
        int sz = 0;
        for(int i=0; i<n; i++) {
            sz = max(sz, min(r[i][0], r[i][1]));
        }
        //cout << sz << endl;
        int ans = 0;
        for(int i=0; i<n; i++) {
            if(r[i][0] >= sz && r[i][1] >= sz) ans++;
        }
        return ans;
    }
};
