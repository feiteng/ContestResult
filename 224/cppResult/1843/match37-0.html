class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int max_len = 0;
        int max_count = 0;
        
        for (int i = 0; i < rectangles.size(); i++) {
            int len = min(rectangles[i][0], rectangles[i][1]);
            
            if (len > max_len) {
                max_len = len;
                max_count = 1;
            } else if (len == max_len) {
                max_count++;
            }
        }
        
        return max_count;
    }
};
