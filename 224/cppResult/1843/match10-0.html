class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int max_len = 0, max_count = 0;
        for (const auto& rec : rectangles)
        {
            int max_sq = min(rec[0], rec[1]);
            if (max_sq > max_len)
            {
                max_len = max_sq;
                max_count = 1;
            }
            else if (max_sq == max_len)
            {
                ++max_count;
            }
        }
        return max_count;
    }
};
