class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int max_num = 0;
        int max_long = 0;
        for(int i = 0; i < rectangles.size(); i++){
            int a = min(rectangles[i][0], rectangles[i][1]);
            if(a > max_long){
                max_long = a;
                max_num = 1;
            }
            else if(a == max_long){
                max_num++;
            }
        }
        return max_num;
    }
};
