class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& r) {
        int maxval = 0;
        int cnt = 0;
        for(int i = 0 ;i < r.size();i++ ){
            int m  = min(r[i][0],r[i][1]);
            if(m>maxval){
                maxval = m;
                cnt =1;
            }else if(m == maxval){
                cnt++;
            }
        }
        return cnt;    
    }
};
