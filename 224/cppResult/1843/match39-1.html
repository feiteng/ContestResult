class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int max = -1;
        int c = 0;
        for (int i = 0; i < rectangles.size(); ++i) {
            int l = rectangles[i][0] <rectangles[i][1] ? rectangles[i][0] : rectangles[i][1];

            if (l > max) {
                max = l;
                c = 1;
            } else if (l == max) {
                ++c;
            }
        }
        return c;
    }
};
