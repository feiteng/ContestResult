class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int res = 0, cur = 0;
        for (const auto& r: rectangles) {
            int t = min(r[0], r[1]);
            if (t > cur) {
                cur = t, res = 1;
            } else if (t == cur) {
                res += 1;
            }
        }
        return res;
    }
};
