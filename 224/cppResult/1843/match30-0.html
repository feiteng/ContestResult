class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        
        int max_len=-1;
        
        int count=0;
        
        for(int i=0;i<rectangles.size();i++)
            max_len=max(max_len,min(rectangles[i][0],rectangles[i][1]));
        
        for(int i=0;i<rectangles.size();i++)
        {
            if(max_len==min(rectangles[i][0],rectangles[i][1]))
                count++;
        }
        
        return count;
    }
};
