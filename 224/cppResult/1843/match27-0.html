class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int ma=0;
        for(auto it:rectangles) ma=max(ma,min(it[0],it[1]));
        int cnt=0;
        for(auto it:rectangles) if(min(it[0],it[1])>=ma) cnt++;
        return cnt;
    }
};
