class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles)
    {
        int sz = 0;
        for (auto& r : rectangles)
            sz = max( sz, min(r[0], r[1]));
        
        int res = 0;
        for (auto& r : rectangles)
        {
            if (r[0] >= sz && r[1] >= sz)
                ++res;
        }
        return res;
    }
};
