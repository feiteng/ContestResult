class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        int res = 0, m = 0;
        for (const auto& r : rectangles) {
            int x = min(r[0], r[1]);
            if (x > m) {
                res = 1;
                m = x;
            } else if (x == m) {
                res ++;
            }
        }
        return res;
    }
};
