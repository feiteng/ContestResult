class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& A) {
        int best = 0;
        int n= A.size();
        for (int i = 0; i < n; i++) {
            best = max(best, min(A[i][0], A[i][1]));
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int k = min(A[i][0], A[i][1]);
            if (k==best) ans++;
        }
        return ans;
    }
};
