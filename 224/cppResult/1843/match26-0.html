class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& a) {
        int n = a.size();
        int side = 0;
        for(int i=0; i<n; ++i){
            int cur = min(a[i][0], a[i][1]);
            side = max(side, cur);
        }
        int cnt = 0;
        for(int i=0; i<n; ++i){
            int cur = min(a[i][0], a[i][1]);
            if(cur==side) ++cnt;
        }
        return cnt;
    }
};
