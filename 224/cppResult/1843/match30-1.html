class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& r) {
        int n=r.size();
        int mx=0;
        for(int i=0;i<n;i++){
            r[i][0] = min(r[i][0],r[i][1]);
            mx=max(mx,r[i][0]);
        }

        int cnt=0;
        for(int i=0;i<n;i++){
            if(mx == r[i][0])cnt++;
        }
        return cnt;
        
    }
};
