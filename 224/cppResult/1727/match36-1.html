inline LL fastpo(LL x, LL n, LL mod) {
	LL res(1);
	while(n) {
		if(n & 1) {
			res = res * (LL)x % mod;
		}
		x = x * (LL) x % mod;
		n /= 2;
	}
	return res;
}
LL c(int a, int b) {
    LL res = 1;
    for(int i = 1; i <= b; i++) {
        res = res * (a - i + 1);
        res /= i;
    }
    return res;
}

inline string itoa(int x, int width = 0) {
  string res;
  if(x == 0) res.push_back('0');
  while(x) {
    res.push_back('0' + x % 10);
    x /= 10;
  }
  while((int)res.size() < width) res.push_back('0');
  reverse(res.begin(), res.end());
  return res;
}

int getr(int x) {
    int p = x;
    while(rela[p] != p) p = rela[p];
    int p1 = p; p = x;
    while(rela[p] != p) {
        int p2 = rela[p];
        rela[p] = p1;
        p = p2;
    }
    return p1;
}

vector<string> a;
int cj, mj;
int _;
int n, m;
int mp[11][11][11][11][2];
bool inRange(int x, int y) {
    return 0 <= x && x < n && 0 <= y && y < m;
}
class Solution {
public:
    int g(int mx, int my, int cx, int cy, int o) {
        if(mx == cx && my == cy) return false;
        if(a[mx][my] == 'F') return true;
        if(a[cx][cy] == 'F') return false;
        return mp[mx][my][cx][cy][o];
    }
    int f(int mx, int my, int cx, int cy, int o) {
        if(mx == cx && my == cy) return false;
        if(a[mx][my] == 'F') return true;
        if(a[cx][cy] == 'F') return false;
        if(-1 != mp[mx][my][cx][cy][o]) return mp[mx][my][cx][cy][o];
        int & res = mp[mx][my][cx][cy][o];
        if(o == 0) {

            int cnt = 0;
            for(int d = 0; d < 4; d++) {
                for(int len = 0; len <= mj; len++) {
                    int x = mx + dx[d] * len;
                    int y = my + dy[d] * len;
                    if(inRange(x, y) && a[x][y] != '#') {
                        if(g(x, y, cx, cy, o ^ 1) == 1) return res = true;
                        else if(g(x, y, cx, cy, o ^ 1) == -1) cnt ++;
                    }else break;
                }
            }
            if(cnt)
                return -1;
            else return res = false;
        }else {

            int cnt = 0;
            for(int d = 0; d < 4; d++) {
                for(int len = 0; len <= cj; len++) {
                    int x = cx + dx[d] * len;
                    int y = cy + dy[d] * len;
                    if(inRange(x, y) && a[x][y] != '#') {
                        if(g(mx, my, x, y, o ^ 1) == false) return res = false;
                        else if(g(mx, my, x, y, o ^ 1) == -1) cnt++;
                        
                    }else break;
                }
            }
            if(cnt)
                return -1;
            else return res = true;
        }
    }
    bool canMouseWin(vector<string>& a, int cj, int mj) {
        n = a.size(), m = a[0].size();
        ::a = a; :: cj = cj; :: mj = mj;
        _ = 0;
        memset(mp, -1, sizeof(mp));

        int mx, my, cx, cy;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(a[i][j] == 'M') {
                    mx = i; 
                    my = j;
                    a[i][j] = '.';
                }else if(a[i][j] == 'C') {
                    cx = i; cy = j;
                    a[i][j] = '.';
                }
            }
        }
        bool flag = true;
        while(flag) {
            flag = false;
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < m; j++) {
                    for(int k = 0; k < n; k++) {
                        for(int l = 0; l < m; l++) {
                            for(int d = 0; d < 2; d++) {
                                int bak = g(i, j, k, l, d);
                                int nw = f(i, j, k, l, d);
                                assert(nw == g(i, j, k, l, d));
                                //if(i == 2 && j == 4 && k == 2 && l == 3 && d == 0) cout << 'g' << g(i, j, k, l, d) << endl;;
                                if(nw != bak) {
                                    //if(i == 2) cout << i << j << k << l << d << nw << endl;
                                    flag = true;
                                }
                            }
                        }
                    }
                }
            }
            if(flag == false) break;
        }
        return g(mx, my, cx, cy, 0) == 1;
    }
};
