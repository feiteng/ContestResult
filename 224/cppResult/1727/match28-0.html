class Solution {
    int F[1010][8][8][8][8];
    vector<string> A;
    int n,m,c,d,p_F_x,p_F_y;
    int dx[4]={1,-1,0,0},dy[4]={0,0,1,-1};
    bool flag[100];
    int top,Q[100];
    bool check(int x,int y)
    {
        if (x<0 || x>=n || y<0 || y>=m) return false;
        if (A[x][y]=='#') return false;
        return true;
    }
public:
    bool canMouseWin(vector<string>& grid, int catJump, int mouseJump) {
        A=grid;
        n=A.size();
        m=A[0].size();
        c=catJump;
        d=mouseJump;
        memset(F,0,sizeof F);
        int p_C_x,p_C_y,p_M_x,p_M_y;
        for (int i=0;i<n;++i)
            for (int j=0;j<m;++j)
            {
                if (A[i][j]=='C')
                {
                    p_C_x=i;
                    p_C_y=j;
                }
                if (A[i][j]=='M')
                {
                    p_M_x=i;
                    p_M_y=j;
                }
                if (A[i][j]=='F')
                {
                    p_F_x=i;
                    p_F_y=j;
                }
            }
        for (int f=99;f>=0;--f)
            for (int cx=0;cx<n;++cx)
                for (int cy=0;cy<m;++cy)
                    for (int mx=0;mx<n;++mx)
                        for (int my=0;my<m;++my)
        {
            if (!check(cx,cy) || !check(mx,my)) continue;
            top=0;
            if (!(f&1))
            {
                if (cx==mx && cy==my)
                {
                    F[f][cx][cy][mx][my]=0;
                    continue;
                }
                if (cx==p_F_x && cy==p_F_y)
                {
                    F[f][cx][cy][mx][my]=0;
                    continue;
                }
                for (int i=0;i<4;++i)
                {
                    for (int j=0;j<=d;++j)
                    {
                        int tx=mx+j*dx[i];
                        int ty=my+j*dy[i];
                        if (check(tx,ty))
                        {
                            Q[++top]=F[f+1][cx][cy][tx][ty];
                        }
                        else break;
                    }
                }
            }
            else
            {
                if (mx==p_F_x && my==p_F_y)
                {
                    F[f][cx][cy][mx][my]=0;
                    continue;
                }
                for (int i=0;i<4;++i)
                {
                    for (int j=0;j<=c;++j)
                    {
                        int tx=cx+j*dx[i];
                        int ty=cy+j*dy[i];
                        if (check(tx,ty))
                        {
                            Q[++top]=F[f+1][tx][ty][mx][my];
                        }
                        else break;
                    }
                }
            }
            for (int i=1;i<=top;++i) flag[Q[i]]=true;
            int p=0;
            while (flag[p]) ++p;
            F[f][cx][cy][mx][my]=p;
            for (int i=1;i<=top;++i) flag[Q[i]]=false;
        }
        return F[0][p_C_x][p_C_y][p_M_x][p_M_y];
    }
};
