class Solution {
public:
    int tupleSameProduct(vector<int>& a) {
        sort(a.begin(),a.end());
        int ans=0,n=a.size();
        map<int,int> mp;
        for ( int i=0; i<n; i++ ) {
            for ( int j=i-1; j>=0; j-- ) {
                if ( mp.count(a[i]*a[j]) ) {
                    ans+=mp[a[i]*a[j]]*2*2*2;
                }
                mp[a[i]*a[j]]++;
            }
        }
        return ans;
    }
};
