class Solution {
public:
    int tupleSameProduct(vector<int>& A) {
        sort(A.begin(), A.end());
        unordered_map<int,int> ppm;  // product -> pair map
        int n = A.size();
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                // pair i,j
                int prod = A[i] * A[j];
                auto itr = ppm.find(prod);
                if (itr != ppm.end()) {
                    int prev = itr->second;
                    cnt += prev * 8;
                    ++itr->second;
                } else {
                    ppm[prod] = 1;
                }
            }
        }
        return cnt;
    }
};
