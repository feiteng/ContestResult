int res = 0;

class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int len = nums.size();
        unordered_map<int,int> m;
        for(int i =0;i<len;i++){
            for(int j = i+1;j<len;j++){
                m[nums[i]*nums[j]]++;
            }
        }
        int res  = 0;
        for(auto p:m){
            int v = p .second;
            res+=(8*v*(v-1)/2);
        }
        return res;
    }
};
