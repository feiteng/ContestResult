class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int, int> m;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = i + 1; j < nums.size(); j++) {
                m[nums[i] * nums[j]]++;
            }
        }
        
        int ans = 0;
        for (auto val : m) {
            if (val.second < 2)
                continue;
            
            ans += 4 * (val.second - 1) * val.second;
        }
        
        return ans;
    }
};
