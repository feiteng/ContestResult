class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        map<int, int> count;
        
        int n = nums.size();
        for (int i = 0 ; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                count[nums[i]*nums[j]]++;
            }
        }
        
        int ret  =0 ;
        for (auto &cnt :count)
        {
            int num = cnt.second;
            
            ret += num*(num-1)*4;
        }
        
        return ret;
    }
};
