class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int, int> m;
        int n = nums.size();
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                m[nums[i] * nums[j]]++;
            }
        }
        int ans = 0;
        for (auto &[f, s] : m) {
            int cn2 = s * (s - 1) / 2;
            ans += cn2 * 8;
        }
        return ans;
    }
};
