class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        int n = nums.size();
        map<int, int> dp;
        for(int i=0;i<n;i++)
        {
            for(int j=i+1;j<n;j++)
            {
                dp[nums[i]*nums[j]] ++;
            }
        }
        int res = 0;
        for(auto& it:dp)
        {
            int cur = it.second;
            res += cur*(cur-1)*4;
        }
        return res;
    }
};
