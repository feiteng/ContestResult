class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int,int>mp;
        int ans = 0;
        for (int i = 0; i < nums.size(); i++) {
            // int temp = nums[i] * nums[j];
            for (int j = i + 1; j < nums.size(); j++) {
                int temp = nums[i] * nums[j];
                mp[temp]++;
            }
        }
        for (auto x : mp) {
            // if (x.second > 1)
            int temp = ((x.second) * (x.second - 1)) / 2;
            ans += (temp) * 8;
            // cout <<x.first << " " <<  x.second << endl;
        }
        return ans;
    }
};
