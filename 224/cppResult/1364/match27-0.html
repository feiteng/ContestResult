class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int, int> feq;
        for(int i=0;i<nums.size();i++){
            for(int j=i+1;j<nums.size();j++){
                feq[nums[i]*nums[j]]++;
            }
        }
        int ans = 0;
        for(auto p : feq){
            if(p.second>=2){
                ans += 4*(p.second-1)*(p.second);
            }
        }
        return ans;
    }
};
