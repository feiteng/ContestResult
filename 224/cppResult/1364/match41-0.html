class Solution {
public:

int tupleSameProduct(vector<int>& nums) {
    map<int,int>help;
    int len=nums.size();
    for(int i=0;i<len;++i){
        for(int j=i+1;j<len;++j){
            help[nums[i]*nums[j]]++;
        }
    }
    int res=0;
    for(auto v:help){
        int count=v.second;
        res+=count*(count-1)*4;
    }
    return res;
    
}
};
