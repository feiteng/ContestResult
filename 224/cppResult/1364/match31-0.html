class Solution {
public:
	int tupleSameProduct(vector<int>& nums) {
		unordered_map<int, vector<int>> M;
		int len = nums.size();
		for (int i = 0; i < len; ++i) {
			for (int j = i + 1; j < len; ++j) {
				int res = nums[i] * nums[j];
				M[res].push_back(min(nums[i], nums[j]));
			}
		}
		int ans = 0;
		for (auto i : M) {
			int n = i.second.size();
			ans += n * (n - 1) / 2;
		}
		return ans * 8;
	}
};
