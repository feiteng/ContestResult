class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int, int> c;
        int l = nums.size();
        int ans = 0;
        for (int i = 0; i < l-1; ++i) {
            for (int j = i+1; j < l; ++j) {
                int v = nums[i] * nums[j];
                auto it = c.find(v);
                if (it != c.end()) {
                    int p = it->second;
                    ans += p * 8;
                    c[v] = p+1;
                } else {
                    c[v] = 1;
                }
            }
        }
        return ans;
    }
};
