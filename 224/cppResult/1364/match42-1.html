class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        map <int, int> cnt;
        int n = nums.size();
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                int mul = nums[i] * nums[j];

                cnt[mul]++;
            }
        }
        int ret = 0;
        for(auto& it : cnt){
            // cout << it.first << " " << it.second << endl;
            int x = (it.second * (it.second - 1)) / 2;
            ret += 8 * x;
        }
        return ret;
    }
};
