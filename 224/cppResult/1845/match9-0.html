class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        vector<vector<int>>sum=matrix;
        int n=matrix.size();
        int m=matrix[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(matrix[i][j]==0){
                    sum[i][j]=0;
                }else{
                    if(i==0) sum[i][j]=1;
                    else sum[i][j]=sum[i-1][j]+1;
                }
            }
        }
        int ans=0;
        for(int i=0;i<n;i++){
            sort(sum[i].begin(),sum[i].end());
            reverse(sum[i].begin(),sum[i].end());
            for(int j=0;j<m;j++){
                ans=max(ans,(j+1)*sum[i][j]);
            }
        }
        return ans;
    }
};
