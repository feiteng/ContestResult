class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size(); 
        int m = matrix[0].size(); 
        vector<vector<int> > acc(n+1, vector<int>(m+1, 0)); 
        for(int j = 0; j < m; j++) {
            for(int i = n-1; i>=0; i--) {
                if(matrix[i][j] == 0) continue; 
                acc[i][j] = acc[i+1][j]+1; 
            }
        }
        int res = 0; 
        for(int i = 0; i < n; i++) {
            vector<int> vec;
            for(int j = 0; j < m; j++) {
                vec.push_back(acc[i][j]); 
                
            }
            sort(vec.begin(), vec.end()); 
            for(int j = m-1; j>=0; j--) {
                res = max(res, vec[j] * (m-j)); 
            }
        }
        return res; 
    }
};
