class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        
        int m=matrix.size();
        int n=matrix[0].size();
        
        vector<vector<int>> run_sum(m,vector<int> (n,0));
        
        run_sum[0]=matrix[0];
        int ans=0;

        for(int j=0;j<n;j++)
        {
            for(int i=1;i<m;i++)
            {
                if(matrix[i][j]==1)
                {
                    run_sum[i][j]=run_sum[i-1][j]+1;
                }
                else
                {
                    run_sum[i][j]=0;
                }
            }
        }
        

        for(int i=0;i<m;i++)
        {
            sort(run_sum[i].begin(),run_sum[i].end(),greater<int>());
            for(int j=0;j<n;j++)
            {
                ans=max(ans,run_sum[i][j]*(j+1));
            }
        }
        
        return ans;
    }
};
