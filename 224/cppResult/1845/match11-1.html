class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
int n=matrix.size(),m=matrix[0].size();
    vector<vector<int>> dp(n, vector<int>(m, 0));
    for(int j=0;j<m;j++)
    {
        for(int i=n-1;i>=0;i--)
        {
            if(matrix[i][j]==0)
                dp[i][j]=0;
            else
            {
                if(i==n-1)
                    dp[i][j]=1;
                else
                    dp[i][j]= 1 +dp[i+1][j];
            }
        }
    }
    int ans=0;
    for(int i=0;i<n;i++)
    {
        vector<int> tmp;
        for(int j=0;j<m;j++)
            tmp.push_back(dp[i][j]);
        sort(tmp.begin(),tmp.end());

        int now_len=0x3f3f3f3f;
        for(int j=m-1;j>=0;j--)
        {
            now_len=min(now_len,tmp[j]);
            ans=max(ans,now_len*(m-j));
        }
    }
    return ans;
    }
};
