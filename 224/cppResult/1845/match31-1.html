class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<vector<int>> col_sum = matrix;

        for (int i = 1; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (matrix[i][j] == 0) {
                    col_sum[i][j] = 0;
                } else {
                    col_sum[i][j] = col_sum[i - 1][j] + 1;
                }
            }
        }
        int res = 0;

        for (int i = 0; i < m; ++i) {
            sort(col_sum[i].begin(), col_sum[i].end());
            for (int j = n - 1; j >= 0; --j) {
                res = max(res, col_sum[i][j] * (n - j));
            }
        }
        return res;
    }
};
