class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& mat) {
        int r = mat.size();
        int c = mat[0].size();
        if(c==1)
        {
            int co=0;
            for(int i=0;i<r;i++)
            {
                for(int j=0;j<c;j++)
                {
                    if(mat[i][j]==1)
                        co++;
                }
            }
            return co;
        }
        int count[r+1][c+1];
        for (int i = 0; i < c; i++)
        { 
            count[0][i] = mat[0][i]; 
            for (int j = 1; j < r; j++) 
                count[j][i] = (mat[j][i] == 0) ? 0 : count[j - 1][i] + 1; 
        } 
        
        for (int i = 0; i < r; i++) { 
            vector<int>counter(r+1,0); 
            
            for (int j = 0; j < c; j++) 
                counter[count[i][j]]++; 

            int col = 0; 
            for (int j = r; j >= 0; j--) { 
                if (counter[j] > 0) { 
                    for (int k = 0; k < counter[j]; k++) { 
                        count[i][col] = j; 
                        col++; 
                    } 
                } 
            } 
        } 
        int curr_area, max_area = 0; 
        for (int i = 0; i < r; i++) { 
            for (int j = 0; j < c; j++) { 
                curr_area = (j + 1) * count[i][j]; 
                if (curr_area > max_area) 
                    max_area = curr_area; 
            } 
        } 
        return max_area;
    }
};
