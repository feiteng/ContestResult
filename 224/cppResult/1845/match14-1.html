class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& a) {
      int n = a.size(), m = a[0].size();
      vector<vector<int>> g(n, vector<int>(m, 0));
      for (int i = 0; i < m; i++) {
        int cnt = 0;

        for (int j = n - 1; j >= 0; j--) {
          if (a[j][i] == 1) {
            cnt += 1;
          } else {
            cnt = 0;
          }
          g[j][i] = cnt;

        }
      }
      int ans = 0;
      for (int i = 0; i < n; i++) {
        vector<int> p;
        for (int j = 0; j < m; j++) {
          p.push_back(g[i][j]);

        }
        sort(p.begin(), p.end());
        reverse(p.begin(), p.end());
        int c = n;
        for (int j = 0; j < m; j++) {
          c = min(c, p[j]);
          ans = max(ans, (j + 1) * c);
        }
      }

      return ans;
    }
};
