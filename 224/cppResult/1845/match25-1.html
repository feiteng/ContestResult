class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& mx) {
        int m=mx.size();
        int n=mx[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(mx[j][i]==1&&j!=0){
                    mx[j][i]=mx[j-1][i]+1;
                }
            }
        }
        int ans=0;
        for(auto& vec:mx){

            sort(vec.begin(),vec.end());
            for(int i=0;i<n;i++){
                ans=max(vec[i]*(n-i),ans);
            }
        }
        return ans;
    }
};
