class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<int> cur(n);
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (matrix[i][j]) {
                    cur[j]++;
                } else {
                    cur[j] = 0;
                }
            }
            
            vector<int> temp = cur;
            sort(temp.begin(), temp.end(), greater<>());
            for (int k = 0; k < n; k++) {
                ans = max(ans, (k + 1) * temp[k]);
            }
        }
        
        return ans;
    }
};
