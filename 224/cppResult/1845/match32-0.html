class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<vector<int>> up(n, vector<int>(m, 0));
        for(int i = 0; i < m; ++i) up[0][i] = matrix[0][i];
        for(int i = 1; i < n; ++i) {
            for(int j = 0; j < m; ++j) {
                if(matrix[i][j]) up[i][j] = up[i - 1][j] + 1;
                else up[i][j] = 0;
            }
        }
        int ans = 0;
        vector<int> res;
        for(int i = n - 1; i >= 0; --i) {
            for(int j = 0; j < m; ++j) res.eb(up[i][j]);
            sort(all(res));
            for(int j = m - 1; j >= 0; --j) {
                ans = max(ans, res[j] * (m - j));
            }
            res.clear();
        }
        return ans;
    }
};
