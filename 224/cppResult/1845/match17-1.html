class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int>> ones (n, vector<int> (m,0));
        for(int j = 0; j < m; j++){
            for(int i = 0; i < n; i++){
                if(!matrix[i][j]){
                    ones[i][j] = matrix[i][j];
                }

                else if(!i){
                    ones[i][j] = matrix[i][j];
                }
                else{
                    ones[i][j] = matrix[i][j] + ones[i-1][j];
                }
            }
            
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            sort(ones[i].begin(), ones[i].end(),greater<int>());
            for(int j = 0; j < m; j++){
                res = max(res,(j+1)*ones[i][j]);
            }
        }
        // for(auto a:ones){
        //     for(auto b:a){
        //         cout << b << " ";
        //     }
        //     cout << "\n";
        // }
        return res;
    }
};
