class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<vector<int> > down(n, vector<int>(m, 0));
        for (int i=n-1;i>=0;i--) {
            for (int j=0;j<m;j++) {
                if (matrix[i][j]) {
                    down[i][j] = 1;
                    if (i < n - 1) down[i][j] += down[i + 1][j];
                }

            }
        }
        int ans = 0;
        for (int i=0;i<n;i++) {
            vector<int>fuck(m);
            for (int j=0;j<m;j++) {
                fuck[j] = down[i][j];

            }
            sort(fuck.begin(), fuck.end());
            reverse(fuck.begin(), fuck.end());
            for (int j=0;j<m;j++) {
                ans = max(ans, (j + 1) * fuck[j]);
            }
        }
        return ans;
    }
};
