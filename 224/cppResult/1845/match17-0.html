class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> len(m, vector<int>(n, 0));
        for (int c = 0; c < n; ++c) {
            for (int r = 0, start = 0; r < m; ++r) {
                if (matrix[r][c] == 0) {
                    start = r + 1;
                }
                else {
                    len[r][c] = r - start + 1;
                }
            }
        }
        int result = 0;
        for (int r = 0; r < m; ++r) {
            vector<int> columns(len[r].begin(), len[r].end());
            sort(columns.begin(), columns.end());
            for (int c = 0; c < n; ++c) {
                result = max(result, columns[c] * (n - c));
            }
        }
        return result;
    }
};
