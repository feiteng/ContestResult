class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n=matrix.size();
        int m=matrix[0].size();
        int ans=0;
        vector<vector<int>> height(n,vector<int>(m));
        for(int i=0;i<n;++i){
            vector<int> row(m);
            for(int j=0;j<m;++j){
                if(0==i){
                    height[i][j]=matrix[i][j];
                }
                else{
                    if(matrix[i][j]) height[i][j]=height[i-1][j]+1;
                    else height[i][j]=0;
                }
                row[j]=height[i][j];
            }
            sort(row.begin(),row.end(),greater<int>());
            for(int j=0;j<m;++j){
                ans=max(ans,row[j]*(j+1));
            }
        }
        return ans;
    }
};
