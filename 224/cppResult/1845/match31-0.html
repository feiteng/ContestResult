class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& a) {
        int ans = 0;
        int n = a.size();
        int m = a[0].size();
        int A[n + 2][m + 2];
        int B[n + 2][m + 2];
        for (int i = n - 1; i >= 0; --i) {
            for (int j = m - 1; j >= 0; --j) {
                A[i + 1][j + 1] = a[i][j];
            }
        }
        memset(B, 0, sizeof(B));
        for (int i = n; i >= 1; --i) {
            for (int j = m; j >= 1; --j) {
                if (A[i][j]) {
                    B[i][j] = B[i + 1][j] + 1;
                } else {
                    B[i][j] = 0;
                }
            }
        }
        int tmp[m + 2];
        for (int i = 1; i <= n; ++i) {
            memcpy(tmp, B[i], sizeof(tmp));
            sort(tmp + 1, tmp + m + 1);
            reverse(tmp + 1, tmp + m + 1);
            int cur = tmp[1];
            for (int j = 1; j <= m; ++j) {
                cur = min(cur, tmp[j]);
                ans = max(ans, cur * j);
            }
        }
        return ans;
    }
};
