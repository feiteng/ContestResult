class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& M) {
        int m = M.size(), n = M[0].size();
        
        vector<int> cnt(n, 0);
        int res = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                cnt[j] = M[i][j] == 0? 0: cnt[j] + 1;
            }
            
            vector<int> tmp = cnt;
            sort(tmp.rbegin(), tmp.rend());
            for(int j = 0; j < n; ++j) {
                res = max(res, tmp[j] * (j + 1));
            }
        }
        return res;
    }
};
