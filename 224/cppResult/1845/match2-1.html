class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& g) {
        int n = g.size(), m = g[0].size();
        vector<int> f(m);
        int res = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m ; j++){
                f[j] = g[i][j] ? f[j] + 1 : 0;
            }
            vector<int> t = f;
            sort(t.begin(), t.end());
            for(int j = 0; j < m; j++){
                res = max(res, t[j]*(m-j));
            }
        }
        return res;
    }
};
