class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<vector<int>> nums(n,vector<int>(m,0));
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(matrix[j][i] == 1){
                    nums[j][i] = (j>0?nums[j-1][i]:0)+1;
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < n; i++){
            sort(rbegin(nums[i]),rend(nums[i]));
            int cur = INT_MAX;
            for(int j = 0; j < m; j++){
                cur = min(cur,nums[i][j]);
                ans = max(ans,cur*(j+1));
            }
        }
        return ans;
    }
};
