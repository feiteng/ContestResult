class Solution {
public:
    
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m=matrix.size();
        int n=matrix[0].size();
        //cout<<m<<" "<<n<<endl;
        vector<vector<int>> pre(m,vector<int>(n,0));
        for (int i=0;i<n;i++)
        {
            if (matrix[m-1][i]==0)
            {
                pre[m-1][i]=0;
            }else
                pre[m-1][i]=1;
        }
        if (m>1)
            for (int j=m-2;j>=0;j--)
            {
                for (int i=0;i<n;i++)
                {
                    if (matrix[j][i]==1)
                    {
                        pre[j][i] = pre[j+1][i]+1;
                    }else
                    {
                        pre[j][i] = 0;
                    }
                }
            }
        int ans=0;
        for (int i=0;i<m;i++)
        {
            //for (int j=0;j<n;j++)
            //{
                sort(pre[i].begin(),pre[i].end(),greater<int>());
            //}
            for (int j=0;j<n;j++)
            {
                ans=max(ans,pre[i][j]*(j+1));
            }
        }
        return ans;

    }
};
