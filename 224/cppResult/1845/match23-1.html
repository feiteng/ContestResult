class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& q) {
        int n = q.size(), m = q[0].size();
        for (int i = 1; i < n; i ++) {
            for (int j = 0; j < m; j ++) {
                if (q[i][j]) {
                    q[i][j] = q[i - 1][j] + 1;
                }
            }

        }
        
        int res = 0;
        for (int i = 0; i < n; i ++) {
            sort(q[i].begin(), q[i].end(), greater<int>());
            for (int j = 0; j < m && q[i][j]; j ++) {
                res = max(res, q[i][j] * (j + 1));
            }
        }
        
        return res;
    }
};
