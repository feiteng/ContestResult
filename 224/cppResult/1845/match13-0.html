class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& mat) {
      int m=mat.size(),n=mat[0].size();
        for(int c=0;c<n;c++)
        {
            for(int r=1;r<m;r++)
            {
                if(mat[r][c])
                {
                    mat[r][c]+=mat[r-1][c];
                }
            }
        }
        int res=0;
        
        for(int i=0;i<m;i++)
        {
            sort(mat[i].begin(),mat[i].end());
            for(int col=0;col<mat[0].size();col++)
            {
                    res=max(res,mat[i][col]*(n-col));
            }
        }
        return res;
    }
};
