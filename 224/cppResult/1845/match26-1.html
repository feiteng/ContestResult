class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int>> sds(n, vector<int>(m, 0));
        for(int i = n - 1; i >= 0; i--)
            for(int j = 0; j < m; j++)
                sds[i][j] = (matrix[i][j] == 0? 0: 1 + (i + 1 == n? 0: sds[i + 1][j]));

        int ret = 0;
        for(auto& vc: sds) {
            sort(vc.begin(), vc.end(), greater<int>());
            for(int i = 0; i < m; i++) 
                ret = max(ret, (i + 1) * vc[i]);
        }
        return ret;
    }
};
