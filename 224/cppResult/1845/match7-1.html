class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m=matrix.size(),n=matrix[0].size();
        vector<vector<int>> dp(m,vector<int>(n,9));
        for(int j=0;j<n;j++){
            for(int i=m-1;i>=0;i--){
                if(matrix[i][j]==0) dp[i][j]=0;
                else{
                    if(i<m-1) dp[i][j]=dp[i+1][j]+1;
                    else dp[i][j]=1;

                }
            }
        }
        int ans=0;
        for(int i=0;i<m;i++){
            sort(dp[i].begin(),dp[i].end());
            for(int j=n-1;j>=0;j--){
                ans=max(ans,dp[i][j]*(n-j));
            }
        }
        return ans;
    }
};
