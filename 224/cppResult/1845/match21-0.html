class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> presum(m+1, vector<int>(n));
        for (int i = 0; i < m; i++) {
            for (int j =0; j < n; j++) {
                if (matrix[i][j] == 1){
                    presum[i+1][j] = presum[i][j] + matrix[i][j];
                }
                else {
                    presum[i+1][j] = 0;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            sort(presum[i+1].begin(), presum[i+1].end());
            for (int j = 0; j < n; j++) {
                res = max(res, presum[i+1][j] * (n-j));
            }
        }
        return res;
    }
};

/**
presum[i][j]表示第j列以第i行为结尾的连续i的个数
有多少个宽度为x的列。

map记录每一行上为1的位置。
滑动窗口
