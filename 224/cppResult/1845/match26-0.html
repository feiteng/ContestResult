class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int>> w(n + 1, vector<int>(m));
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {
                if (matrix[i][j]) w[i + 1][j] = w[i][j] + 1;
                else w[i + 1][j] = 0;
            }
        
        int ans = 0;
        for (auto & cols : w) {
            sort(cols.begin(), cols.end(), greater<int>());
            for (int k = 0; k < m; ++k) ans = max(ans, (k + 1) * cols[k]);
        }
        return ans;
    }
};
