class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> left(m, vector<int>(n));
        for (int j = 0; j < n; j++) {
            for (int i = 0; i < m; i++) {
                if (matrix[i][j] == 0) left[i][j] = i;
                else {
                    if (i == 0) left[i][j] = -1;
                    else left[i][j] = left[i-1][j];
                }
            }
        }
        // for (int i = 0; i < m; i++) {
        //     for (int j = 0; j < n; j++) cout << left[i][j] << " ";
        //     cout << "\n";
        // }
        int mx = 0;
        for (int i = 0; i < m; i++) {
            int mxleft = -1;
            sort(left[i].begin(), left[i].end());
            for (int j = 0; j < n; j++) {
                mxleft = max(mxleft, left[i][j]);
                mx = max(mx, (i-mxleft)*(j+1));
            }
        }
        return mx;
    }
};
