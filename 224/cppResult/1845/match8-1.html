class Solution {
public:
    // Compute the height for each cell, then sort them for each row and compute the submattrix size (dp)
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int row = matrix.size(), col = matrix[0].size();
        vector<vector<int>> height(row, vector<int> (col, 0));
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                if (i == 0) height[0][j] = matrix[0][j];
                else {

                    if (matrix[i][j] == 1) height[i][j] = height[i-1][j]+1;
                }
            }
        }
        int answer = 0;
        for (int i = 0; i < row; ++i) {
            sort(height[i].begin(), height[i].end());
            for (int j = 0; j < col; ++j) {
                answer = max(answer, height[i][j]*(col-j));
            }
        }
        return answer;
    }
};
