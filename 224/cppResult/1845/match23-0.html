#ifdef DEBUG
#define db(x) cout << #x << "=" << x << endl
#define db2(x, y) cout << "(" << #x << ", " << #y << ")=(" << x << ", " << y << ")\n";
#define db3(x, y, z) cout << "(" << #x << ", " << #y << ", " << #z << ")=(" << x << ", " << y << ", " << z << ")\n"
#define dbv(a) cout << #a << "=["; for (auto xxxx: a) cout << xxxx << ","; cout << "]" << endl
#else
#define db(x) 
#define db2(x, y) 
#define db3(x, y, z)
#define dbv(a)
#endif

class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int rows = matrix.size(), cols = matrix[0].size();
        for (int r = rows - 2; r >= 0; --r) {
            for (int c = 0; c < cols; ++c) {
                if (matrix[r][c] != 0) {
                    matrix[r][c] = matrix[r + 1][c] == 0 ? 1 : (matrix[r + 1][c] + 1);
                }
            }
        }
        db("========");
        for (auto& row: matrix) {
            dbv(row);
        }
        int ans = 0;
        for (int r = 0; r < rows; ++r) {
            sort(begin(matrix[r]), end(matrix[r]), greater<int>());
            dbv(matrix[r]);
            for (int c = 0; c < cols; ++c) {
                ans = max(ans, matrix[r][c] * (c + 1));
            }
        }
        return ans;
    }
};
