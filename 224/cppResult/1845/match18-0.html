class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int n=matrix.size(); int m=matrix[0].size();
        for(int i=1;i<n;i++)
        {
            for(int j=0;j<m;j++) if(matrix[i][j]==1) matrix[i][j]+=matrix[i-1][j];
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            map<int , int ,greater<int> >cal;
            
            for(int j=0;j<m;j++)
            {
                cal[matrix[i][j]]++;
            }
            int cnt=0;
            for(auto it:cal)
            {
                cnt+=it.second;
                ans=max(ans,it.first*cnt);
            }
            
        }
        return ans;
    }
};
