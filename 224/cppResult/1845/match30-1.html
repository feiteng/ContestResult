class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> c(m, vector<int>(n,0));
        c[0 ] = matrix[0];
        
        for(int i = 1; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(matrix[i][j] == 0) {
                    c[i][j] = 0;
                } else {
                    c[i][j] = c[i-1][j] + 1;
                }
            }
        }
        
        int res = 0;
        for(int i = 0; i < m; ++i) {

            vector<int> cur = c[i];
            sort(cur.rbegin(), cur.rend());
            for(int j = 0; j < n; ++j) {
                res = max(res, cur[j] * (j+1));
            }
        }
        
        return res;
    }
};
