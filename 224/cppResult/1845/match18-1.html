class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& m) {
        int r=m.size();
        int c=m[0].size();
        
        for(int i=1;i<r;i++){
            for(int j=0;j<c;j++){
                if(m[i][j]==1){
                    m[i][j]+=m[i-1][j];
                }
            }

        }
        
        int res=0;
        for(int i=0;i<r;i++){
            map<int,int,greater<int>> freq;
            for(int j=0;j<c;j++){
               freq[m[i][j]]++;
            }
            int val=0;
            for(auto l:freq){
                res=max(res,(l.second+val)*l.first);
                val+=l.second;
            }
        }
        
        return res;
        
    }
};
