class Solution {
public:
    int largestSubmatrix(vector<vector<int>>& d) {
        int m = d.size();
        int n = d[0].size();
        vector<vector<int>> dp(m, vector<int>(n, 0));
        for(int i=m-1;i>=0;i--)
        {
            for(int j=0;j<n;j++)
            {
                if(d[i][j])
                {
                    dp[i][j] = (i==m-1?0:dp[i+1][j])+1;
                }
                else
                {
                    dp[i][j] = 0;
                }

            }
        }
        vector<int> tmp(n, 0);
        int res = 0;
        for(int i=0;i<m;i++)
        {
            
            for(int j=0;j<n;j++)
            {
                tmp[j] = dp[i][j];
            }
            sort(tmp.begin(), tmp.end());
            for(int j=0;j<n;j++)
            {
                res = max(res, tmp[j]*(n-j));
            }
            
        }
        return res;
    }
};
