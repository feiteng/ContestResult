class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        count = 0
        max_line = 0
        for i,j in rectangles:
            min_n = min(i,j)
            if min_n > max_line:
                count = 1
                max_line = min_n
            elif min_n ==max_line:
                count += 1
        
        return count
