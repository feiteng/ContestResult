class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        a = 0
        cnt = 0
        
        for x, y in rectangles:
            z = min(x, y)
            if z > a:
                a = z
                cnt = 1
            elif z == a:
                cnt += 1
        
        return cnt
