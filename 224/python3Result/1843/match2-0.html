class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        maxlen = 0
        count = 0
        for a, b in rectangles:
            l = min(a,b)
            if l > maxlen:
                maxlen = l
                count = 1
            elif l == maxlen:
                count += 1
        return count
