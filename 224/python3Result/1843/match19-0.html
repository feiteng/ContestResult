class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        max_len = -1
        max_count = 0
        
        for rect in rectangles:
            rect_width = min(rect)
            if rect_width > max_len:
                max_len = rect_width
                max_count = 1
            elif rect_width == max_len:
                max_count += 1
            
        return max_count
