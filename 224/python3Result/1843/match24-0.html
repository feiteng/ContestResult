class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        ma = -1
        cnt = 0
        for a,b in rectangles:
            t = min(a,b)
            if t>ma:
                ma = t
                cnt=1
            elif t==ma:
                cnt+=1
        return cnt
