class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        mapping = defaultdict(int)
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                mapping[nums[i] *nums[j]] += 1
        res = 0
        for k, v in mapping.items():
            if v >= 2:
                res += v * (v-1) * 4
        return res
