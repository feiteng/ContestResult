class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        dict_ = collections.defaultdict(int)
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                dict_[nums[i] * nums[j]] += 1
        ans = 0
        for i in dict_.values():
            ans += i*(i-1)*4
        return ans
