class Solution:
  def tupleSameProduct(self, arr: List[int]) -> int:
    arr.sort()
    n = len(arr)
    count = defaultdict(int)
    for i in range(n):
      for j in range(i+1, n):
        count[arr[i] * arr[j]] += 1
        
    ret = 0
    for t in count.values():
      ret += t * (t-1) * 4
      
    return ret
