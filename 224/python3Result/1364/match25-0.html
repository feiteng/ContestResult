class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        cnt = collections.defaultdict(int)
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                cnt[nums[i]*nums[j]] += 1
        ans = 0
        for key in cnt:
            if cnt[key] > 1:
                ans += (8)*(cnt[key]*(cnt[key]-1))//2
        print(cnt)
        return ans
