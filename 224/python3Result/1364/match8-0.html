class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        n = len(nums)
        count = Counter()
        for i in range(n):
            for j in range(i + 1, n):
                count[nums[i] * nums[j]] += 1
        ans = 0
        for x, c in count.items():
            if c > 1:
                ans += c * (c - 1) * 4
        return ans
