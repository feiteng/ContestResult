class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        from collections import defaultdict
        dic 
= defaultdict(int)
        L = len(nums)
        for i in range(L):
            for j in range(i):
                dic[nums[i]*nums[j]] += 1
        res = 0
        for n in dic.values():
            res += (n-1)*n//2
        return res*8
