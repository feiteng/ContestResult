class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        dic = Counter()
        ln = len(nums)
        for i in range(ln):
            for j in range(i+1, ln):
                dic[nums[i] * nums[j]] += 1
        
        ans = 0
        #print(dic)
        for v in dic.values():
            ans += 2*2* v*(v-1)
        
        return ans
