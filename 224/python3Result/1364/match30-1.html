class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        nums.sort()
        N 
= len(nums)

        counter = Counter()
        for i in range(N - 1):
            for j in range(i + 1, N):
                counter[nums[i] * nums[j]] += 1
        
        res = 0
        for k, v in counter.items():
            res += (v - 1) * v
        
        return res * 4
