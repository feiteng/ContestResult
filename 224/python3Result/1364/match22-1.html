from collections import defaultdict

class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        nums.sort()
        d = defaultdict(list)
        n = len(nums)
        for i in range(n):
            for j in range(i + 1, n):  
                a, b = nums
[i], nums[j]
                d[a * b].append((a, b))
        
        res = 0
        for v in d.values():
            l = len(v)
            res += l * (l - 1) * 4
        return res
