class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        d = collections.defaultdict(int)
        for i in range(len(nums)):
            for j in range(i+1,len(nums)):
                d[nums[i]*nums[j]]+=1
        # print(d)
        ans = 0
        for k in d:
            if d[k]>1:
                ans+=(2*d[k])*(2*(d[k]-1))
        return ans
