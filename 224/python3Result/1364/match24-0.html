class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        c = Counter()
        n = len(nums)
        for i in range(n):
            for j in range(i+1,n):
                c[nums[i]*nums[j]]+=1
        ans = 0
        for k,v in c.items():
            if v > 1:
                ans += v*(v-1)*4
        return ans
