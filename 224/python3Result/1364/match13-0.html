class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        counter = defaultdict(int)
        n = len(nums)
        for i in range(n - 1):
            for j in range(i + 1, n):
                counter[nums[i] * nums[j]] += 1
        
        res = 0
        for k, m in counter.items():
            res += 2 * m * 2 * (m - 1)
        return res
