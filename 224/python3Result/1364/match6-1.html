class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        prod_pairs = collections.defaultdict(int)
        n = len(nums)
        for i in range(n):
            for j in range(i + 1, n):
                prod_pairs[nums[i] * nums[j]] += 1
                
        res = 0
        for val in prod_pairs.values():
            res += (val * (val - 1))
            
        return res * 4
