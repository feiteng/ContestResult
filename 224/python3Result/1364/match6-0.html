class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        mul=collections.defaultdict(int)
        n=len(nums)
        for i in range(n):
            for j in range(i+1,n):
                mul[nums[i]*nums[j]]+=1
        re=0
        # print(mul)
        for k in  mul.values():
            re+=(k*(k-1))//2
        return 8*re
