from collections import defaultdict

class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        m = Counter()
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                m[nums[i]*nums[j]] += 1
        res = 0
        for v in m.values():
            if v > 1:
                res += 4 * v * (v - 1)
        return res
