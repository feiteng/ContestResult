class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        # distinct group of 4 * 8
        # count distinct group of 4
        size 
= len(nums)
        product = defaultdict(list)
        for i in range(size-1):
            for j in range(i+1, size):
                a,b = nums[i], nums[j]
                product[a*b].append((a,b))
                
        count = 0
        # print(product)
        for k, v in product.items():
            n = len(v)
            count += n * (n - 1) // 2
            
        return count * 8
