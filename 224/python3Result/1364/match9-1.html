class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        x = collections.defaultdict(int)
        
        for i in range(len(nums)):
            for j in range(i+1,len(nums)):
                x[nums[i]*nums[j]] += 1
        c = 0
        for k in x:
            c += x[k]*(x[k]-1)//2
        return c*8
