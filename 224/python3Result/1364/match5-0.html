class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        d = defaultdict(int)
        n = len(nums)
        for i in range(n-1):
            for j in range(i+1, n):
                d[nums[i]*nums[j]] += 1
        ans = 0
        for k in d:
            if d[k] > 1:
                ans += (d[k] * (d[k]-1)) // 2
        return ans * 8
