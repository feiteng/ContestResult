class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        nums.sort()
        se 
= {}
        n = len(nums)
        ans = 0
        for i in range(n):
            for j in range(i+1, n):
                cur = nums[i] * nums[j]
                if cur in se:
                    ans += se[cur]
                    se[cur] += 1
                else:
                    se[cur] = 1
        return ans * 8
