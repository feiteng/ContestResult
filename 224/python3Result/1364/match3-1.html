class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        ct = collections.Counter()
        n = len(nums)
        for i in range(n):
            for j in range(i+1,n):
                ct[nums[i]*nums[j]]+=1
        return sum([ v*(v-1)//2 for x,v in ct.items() if v>=2])*8
