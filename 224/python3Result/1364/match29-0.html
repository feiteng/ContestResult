class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        cnt = {}
        n = len(nums)
        ans = 0
        
        for i in range(n):
            for j in range(i + 1,n):
                curr = nums[i] * nums[j]
                # print(nums[i],nums[j],curr)
                if curr in cnt:
                    ans += 8 * cnt[curr]
                    cnt[curr] += 1
                else:
                    cnt[curr] = 1
        
        # print(cnt)
        return ans
