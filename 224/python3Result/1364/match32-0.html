class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        counter = defaultdict(int)
        n = len(nums)
        for i in range(n):
            for j in range(i + 1, n):
                counter[nums[i] * nums[j]] += 1
        count = 0
        for key, value in counter.items():
            count += 4 * value * (value - 1)
        return count
