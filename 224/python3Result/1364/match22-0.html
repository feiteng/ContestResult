class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        nums.sort()
        n = len(nums)
        d = defaultdict(set)
        for i in range(n):
            for j in range(i + 1, n):
                a, b = nums[i], nums[j]
                if a > b:
                    a,b = b, a
                d[nums[i] * nums[j]].add((a, b))
        ans = 0
        for k, v in d.items():
            m = len(v)
            ans += 8 * m * (m - 1) // 2
        return ans
