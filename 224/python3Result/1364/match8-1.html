class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        dic = defaultdict(int)
        n = len(nums)
        
        for i in range(n):
            for j in range(i+1,n):
                dic[nums[i]*nums[j]] += 1
        # print(dic)
        ans = 0
        for i in dic.values():
            if i >= 2:
                ans += (i-1) * i /2
        
        return int(ans*8)
