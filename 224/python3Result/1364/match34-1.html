class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        ht 
= collections.defaultdict(int)
        
        n = len(nums)
        for i in range(n):
            for j in range(i+1, n):
                ht[nums[i] * nums[j]] += 1
        
        ans 
= 0
        
        for key in ht:
            if ht[key] > 1:
                ans += ht[key] * (ht[key]-1) * 4
        
        return ans
