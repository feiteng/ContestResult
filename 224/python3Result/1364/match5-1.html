class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        cnt = defaultdict(int)
        n = len(nums)
        for i in range(n):
            for j in range(i+1, n):
                cnt[nums[i]*nums[j]] += 1
        ans = 0
        for e in cnt:
            if cnt[e] > 1:
                ans += cnt[e]*(cnt[e]-1) // 2
        return ans*8
