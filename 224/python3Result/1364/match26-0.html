class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        from collections import defaultdict
        n = len(nums)
        dic = defaultdict(int)
        for i in range(n):
            for j in range(i + 1, n):
                dic[nums[i]*nums[j]] += 1
        # print(dic)
        res = 0
        for x in dic:
            val = dic[x]
            if val > 1:
                res += (val * (val - 1) // 2) * 8
        return res
