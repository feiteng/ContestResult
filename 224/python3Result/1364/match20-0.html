class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        all_products = defaultdict(int)
        
        for i in range(len(nums)-1):
            for j in range(i+1, len(nums)):
                all_products[nums[i] * nums[j]] += 1
        
        res = 0
        
        for val in all_products.values():
            res += (val-1) * 4 * val
        
        return res
