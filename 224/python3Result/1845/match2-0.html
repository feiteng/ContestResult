class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m = len(matrix)
        n = len(matrix[0])
        for i in range(1, m):
            for j in range(n):
                if matrix[i - 1][j] and matrix[i][j]:
                    matrix[i][j] = matrix[i - 1][j] + 1
        maxcover = 0
        for line in matrix:
            temp = line
            temp.sort()
            for i in range(n - 1, -1, -1):
                maxcover = max(maxcover, temp[i] * (n - i))
        return maxcover
