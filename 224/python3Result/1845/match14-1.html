class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        
        for i in range(1, m):
            for j in range(n): 
                if matrix[i][j] == 1:
                    matrix[i][j] = matrix[i - 1][j] + 1
        
        res = 0
        for i in range(m):
            nums = sorted(matrix[i])
            for j in range(n):
                if nums[j] > 0:
                    res = max(res, (n - j) * nums[j])
                    
        return res
