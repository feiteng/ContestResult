class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        R = len(matrix)
        C = len(matrix[0])
        for i in range(1,R):
            for j in range(C):
                if matrix[i][j]==1:
                    matrix[i][j]+=matrix[i-1][j]
        ans = 0
        for i in range(R):
            arr = sorted([matrix[i]
[j] for j in range(C)])
            for k in range(C):
                ans = max(ans,arr[k]*(C-k))
        return ans
