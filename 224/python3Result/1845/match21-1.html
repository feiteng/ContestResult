class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        n, m = len(matrix), len(matrix[0])
        h = [0] * m
        ans = 0
        for i in range(n):
            for j in range(m):
                if matrix[i][j] == 1:
                    h[j] += 1
                else:       
                    h[j] = 0
            tmp = h[::]            
            tmp.sort()
            for j in range(m):
                ans = max(ans, tmp[j] * (m - j))
        return ans
