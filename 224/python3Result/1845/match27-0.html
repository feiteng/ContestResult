class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        R = len(matrix)
        C = len(matrix[0])
        best = 0
        histo = [0] * C
        for row in matrix:
            for c, x in enumerate(row):
                if x:
                    histo[c] += 1
                else:
                    histo[c] = 0
            sHisto = sorted(histo, reverse=True)
            for c, h in enumerate(sHisto):
                best = max(best, h * (c + 1))
            
                
        return best
