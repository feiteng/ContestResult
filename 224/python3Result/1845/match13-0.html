class Solution:
    def largestSubmatrix(self, M: List[List[int]]) -> int:
        n,m= len(M), len(M[0])
        #transform to colSum
        for i in range(1, n):
            for j in range(m):
                if M[i][j] == 1:
                    M[i][j] += M[i-1][j]
        #now M[i][j] is the number of consecutive 1's above

        res = 0
        for i in range(n):
            vals = sorted(M[i], reverse=True)
            for j in range(m):
                res = max(res, (1+j)*vals[j])
        return res
