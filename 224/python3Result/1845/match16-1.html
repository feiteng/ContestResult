class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        n = len(matrix)
        m = len(matrix[0])
        dp = [0] * m
        res = 0
        
        for i in range(n):
            for j in range(m):
                dp[j] = dp[j] + 1 if matrix[i][
j] == 1 else 0
                
            heis = sorted(dp, reverse=True)
            for i, hei in enumerate(heis):
                res = max(res, (i + 1) * hei)
                
        return res
