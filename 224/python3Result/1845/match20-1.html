class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        if not matrix or not matrix[0]: 
return 0
        m, n = len(matrix), len(matrix[0])
        colOnes = [[0] * n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if matrix[i][j] == 1:
                    colOnes[i][j] = colOnes[i-1][j] + 1
        
        maxarea = 0
        for i in range(m):
            tmp = sorted(colOnes[i], reverse 
= True)
            for j in range(n):
                if tmp[j] == 0:
                    break
                maxarea = max(maxarea, tmp[j] * (j+1))
        
        return maxarea
