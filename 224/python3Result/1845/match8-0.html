class Solution:
    def largestSubmatrix(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        prev = [0] * n
        mx = 0
        for i in range(m):
            cur = prev[::]
            for j in range(n):
                if A[i][j] == 1:
                    cur[j] += 1
                else:
                    cur[j] = 0
            sortedCur = sorted(cur)
            for j in range(n):
                mx = max(mx, sortedCur[j] * (n-j))
            prev = cur
        return mx
