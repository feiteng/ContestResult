class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        n = len(matrix)
        m = len
(matrix[0])
        h = [0] * m
        result = 0
        for i in range(n):
            for j in range(m):
                if matrix[i][j] == 1:
                    h[j] += 1
                else:
                    h[j] = 0
            k = sorted(h, reverse=True)
            for j in range(m):
                result = max(result, k[j] * (j + 1))
        return result
