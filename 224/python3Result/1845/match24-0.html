class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m = len(matrix)
        n = len(matrix[0])

        down = [[0 for j in range(n)] for i in range(m)]

        for j in range(n):
            for i in range(m - 1, -1, -1):
                if matrix[i][j] == 0:
                    down[i][j] = 0
                else:
                    down[i][j] = 1 + (down[i + 1][j] if i < m - 1 else 0)
        
        ans = 0
        for i in range(m):
            heights = sorted((down[i][j] for j in range(n)), reverse=True)
            for j in range(n):
                ans = max(ans, heights[j] * (j + 1))
        return ans
