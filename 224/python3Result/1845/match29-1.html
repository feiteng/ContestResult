class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        cnter = [0] * n
        res = 0
        for i in range(m):
            for j in range(n):
                if matrix[i][j] == 1:
                    cnter[j] += 1
                else:
                    cnter[j] = 0
            
            sted = list(sorted(cnter, reverse=True))
            for k in range(n):
                res = max(res, sted[k] * (k + 1))
                
        return res
