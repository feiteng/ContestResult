class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        sub = [0] * n
        res = 0
        for i in range(m):
            for j in range(n):
                if matrix[i][j]==1:
                    sub[j] += 1
                else:
                    sub[j] =0
            one = sub[:]
            one.sort()
            for k in range(n):
                if one[k] == 0:continue
                res = max(res, (n-k)*one[k])
            
        return res
