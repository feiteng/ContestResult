class Solution:
    def largestSubmatrix(self, A: List[List[int]]) -> int:
        m = len(A)
        n = len(A[0])
        for i in range(1,m):
            for j in range(n):
                if A[i][j] == 1:
                    A[i][j] = A[i-1][j] + 1
        # print(A)
        ans = 0
        for row in A:
            row.sort()
            for i in range(n-1,-1,-1):
                ans = max(ans,(n-i)*row[i])
        return ans
