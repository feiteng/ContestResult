class Solution:
    def largestSubmatrix(self, mat: List[List[int]]) -> int:
        R = len(mat)
        C = len(mat[0])

        run = [[0 for i in range(C + 1)] for i in range(R + 1)]

        for i in range(C):
            run[0][i] = mat[0][i]
            for j in range(1, R):
                if mat[j][i] == 0:
                    run[j][i] = 0
                else:
                    run[j][i] = run[j - 1][i] + 1

        for i in range(R):
            run[i].sort(reverse=True)

        max_area = 0
        for i in range(R):
            for j in range(C):
                curr_area = (j + 1) * run[i][j]
                max_area = max(curr_area, max_area)

        return max_area
