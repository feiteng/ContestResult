class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        
for i in range(1, m):
            for j in range(n):
                if matrix[i][j]:
                    matrix[i][j] = matrix[i - 1][j] + 1
        for i in range(m):
            matrix[i] = sorted(matrix[i], reverse = True)
        largest = 0
        for i in range(m):
            for j in range(n):
                largest = max(largest, matrix[i][j] * (j + 1))
        return largest
