class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        n = len(matrix)
        m = len(matrix[0])
        
        continousOneInCol = [[0 for _ in range(m)] for _ in range(n)]
        for j in range(m):
            continousOneInCol[n-1][j] = matrix[n-1][j]
        for i in range(n-2, -1, -1):
            for j in range(m):
                if matrix[i][j] == 0:
                    continousOneInCol[i][j] = 0
                else:
                    continousOneInCol[i][j] = continousOneInCol[i+1][j] + 1
        
        result = 0
        for i in range(n):
            result = max(result, self.getMaxArea(continousOneInCol[i]))
        return result
            
    def getMaxArea(self, nums):
        nums.sort()
        n = len(nums)
        dp = [0 for _ in range(n)]
        dp[n-1] = nums[n-1]
        for i in range(n-2, -1, -1):
            dp[i] = (n-i) * nums[i]
        return max(dp)
