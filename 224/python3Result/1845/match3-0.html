class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        row, col = len(matrix), len(matrix[0])
        for r in range(1,row):
            for c in range(col):
                if matrix[r][c] != 0:
                    matrix[r][c] += matrix[r-1][c]
        
        ans = 0
        for row in matrix:
            row.sort()
            for i,it in enumerate(row):
                ans = max(ans, it*(col-i))
        return ans
