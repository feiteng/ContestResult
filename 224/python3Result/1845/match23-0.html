class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        dp = [[0 for _ in range(n)] for _ in range(m)]
        for i in range(n):
            dp[0][i] = matrix[0][i]
        for i in range(1, m):
            for j in range(n):
                if matrix[i][j]: dp[i][j] = dp[i-1][j] + 1
        for i in range(m):
            dp[i] = sorted(dp[i], reverse = True)
        ans = 0
        for i in range(m):
            for j in range(n):
                ans = max(ans, dp[i][j] * (j+1))
        return ans
