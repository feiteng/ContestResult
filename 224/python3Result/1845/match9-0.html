class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        R, C = len(matrix), len(matrix[0])
        
        flag = True
        for r in range(R):
            for c in range(C):
                if matrix[r][c] == 0:
                    flag = False
                    break
        if flag:
            return R * C
        
        mem = [[0] * C for r in range(R)]
        mem[0][:] = matrix[0][:]
        for c in range(C):
            for r in range(1, R):
                if matrix[r][c] == 0:
                    mem[r][c] = 0
                else:
                    mem[r][c] = mem[r-1][c] + 1
        
        for r in range(R):
            cnt = [0] * (R+1)
            for c in range(C):
                cnt[mem[r][c]] += 1
            
            curr = 0
            j = R
            while j >= 0:
                if cnt[j] > 0:
                    for p in range(cnt[j]):
                        mem[r][curr] = j
                        curr += 1
                j -= 1
                
        maxx = 0
        for r in range(R):
            for c in range(C):
                area = (c + 1) * mem[r][c]
                maxx = max(maxx, area)
        
        return maxx
