class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        for j in range(n):
            for i in range(1,m):
                if matrix[i][j]>0:
                    matrix[i][j] += matrix[i-1][j]
        for row in matrix:
            row.sort(reverse=True)
        return max(max(h*l for l,h in enumerate(row,1)) for row in matrix)
