class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        r, c = len(matrix), len(matrix[0])
        ans = 0
        
for i in range(1,r):
            for j in range(c):
                if matrix[i][j]!=0: matrix[i][j] += matrix[i-1][j]
        
for i in range(r):
            a =  sorted(matrix[i], reverse=True)
            #print(a)
            for j in range(c):
                ans = max(a[j]*(j+1), ans)
        return ans
