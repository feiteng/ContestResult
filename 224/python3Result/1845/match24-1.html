class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m = len(matrix)
        n = len(matrix[0])
        dp = [[0 for j in range(n)] for i in range(m)]
        for i in range(m):
            for j in range(n):
                if matrix[i][j] == 1:
                    dp[i][j] = 1
                    if i > 0:
                        dp[i][j] += dp[i - 1][j]
        ans = 0
        for i in range(m):
            a = dp
[i]
            a.sort(reverse=True)
            for j in range(n):
                ans = max(ans, a[j] * (j + 1))
        return ans
