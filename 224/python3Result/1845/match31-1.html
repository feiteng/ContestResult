class Solution:
    def largestSubmatrix(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        
        # print(m, n)
        count = 0
        for i in range(m):
            for j in range(n):
                if mat[i][j] == 1:
                    count += 1
        
        if count == 0:
            return 0
    
        if count == m * n:
            return m * n
        
        if n == 1:
            return count
        
        hist 
= [[0 for i in range(n + 1)]  for i in range(m + 1)] 

        for i in range(n): 
            hist[0][i] = mat[0][i] 

            for j in range(1, m): 
                if ((mat[j][i] == 0)): 
                    hist[j][i] = 0
                else: 
                    hist[j][i] = hist[j - 1][i] + 1

        for i in range(m): 
            count = [0 for i in range(m + 1)] 
 
            for j in range(n): 
                count[hist[i][j]] += 1
 
            cur_col = 0
            j = m 
            while(j >= 0): 
                if (count[j] > 0): 
                    for k in range(0, count[j], 1): 
                        hist[i][cur_col] = j 
                        cur_col += 1

                j -=
 1

        res = 0
        for i in range(m): 
            for j in range(n): 
                cur = (j + 1) * hist[i][j] 
                res = max(res, cur)

        return res
