from typing import List


class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        for i in range(1, m):
            for j in range(n):
                if matrix[i][j]:
                    matrix[i][j] = matrix[i - 1][j] + 1 if matrix[i - 1][j] else 1
        ans = 0
        # print(matrix)
        for i in range(m):
            cur = [matrix[i][j] for j in range(n)]
            cur.sort(reverse=True)
            for j, t in enumerate(cur):
                if t > 0:
                    ans = max(ans, (j + 1) * t)
        return ans
