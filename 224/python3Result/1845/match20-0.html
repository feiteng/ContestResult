class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        ans = 0
        if not matrix:
            return 0
        m,n = len(matrix),len(matrix[0])
        
        v = [[0]*n for _ in range(m)]
        
        # for i in range(m):
        #     for j in range(n):
        #         if matrix[i][j] == 1:
        #             count = 0
        #             k = i
        #             while k < m and matrix[k][j] == 1:
        #                 count += 1
        #                 k += 1
        #             v[i][j] = count
        # print(v)
        
        for j in range(n):
            for i in range(m-1,-1,-1):
                if matrix[i][j] == 1:
                    v[i][j] = v[i+1][j] + 1 if i != m-1 else 1
        
        for i in range(m):
            # for j in range(n):
            temp = v[i][:]
            temp.sort(reverse = True)
            for j in range(n):
                if temp[j] == 0:
                    break
                # print(i,j,'kk',(j+1)*temp[j])
                ans = max(ans, (j+1)*temp[j])
                
        return ans
