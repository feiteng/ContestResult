class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        R,C=len(matrix),len(matrix[0])
        cols=[[] for i in range(R)]

        for c in range(C):
            count=0
            for r in range(R-1,-1,-1):
                if matrix[r][c]==0:
                    count=0
                else:
                    count+=1
                cols[r].append(count)

        res=0

        for nums in cols:
            nums
.sort()
            for i in range(C):
                res=max(res,nums[i]*(C-i))
        return res
