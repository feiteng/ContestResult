class Solution:
    def largestSubmatrix(self, A: List[List[int]]) -> int:
        R, C = len(A), len(A[0])
        if sum([sum(row) for row in A]) == R*C: return R*C
        
        dp = [[0 for i in range(C + 1)]  
                   for i in range(R + 1)] 

        for i in range(0, C, 1): 

            dp[0][i] = A[0][i] 

            for j in range(1, R, 1): 
                if ((A[j][i] == 0)): 
                    dp[j][i] = 0
                else: 
                    dp[j][i] = dp[j - 1][i] + 1

        for i in range(0, R, 1): 
            count = [0 for i in range(R + 1)] 

            for j in range(0, C, 1): 
                count[dp[i][j]] += 1

            col_no = 0
            j = R 
            while(j >= 0): 
                if (count[j] > 0): 
                    for k in range(0, count[j], 1): 
                        dp[i][col_no] = j 
                        col_no += 1
                j -= 1

        max_area = 0
        for i in range(0, R, 1): 
            for j in range(0, C, 1): 
                curr_area = (j + 1) * dp[i][j] 
                if (curr_area > max_area): 
                    max_area = curr_area 

        return max_area
