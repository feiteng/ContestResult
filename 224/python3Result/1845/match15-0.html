class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m,n,s=len(matrix),len(matrix[0]),0
        for j in range(n):
            for i in range(1,m):
                if matrix[i][j]: matrix[i][j]+=matrix[i-1][j]
        for a in matrix:
            a.sort(reverse=True)
            s=max(s,max([i*x for i,x in enumerate(a,1)]))
        return s
