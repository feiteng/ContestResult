class Solution:
    def largestSubmatrix(self, M: List[List[int]]) -> int:
        h, w = len(M), len(M[0])
        for c in range(w):
            for r in range(1, h):
                if M[r][c] != 0:
                    M[r][c] = M[r-1][c]+1
        # print(M)
        ans = 0
        for r in range(h):
            temp = [M[r][c] for c in range(w)]
            temp.sort()
            for i in range(w):
               ans = max(ans, temp[i]*(w-i))
        return ans
