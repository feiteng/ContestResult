class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m,n=len(matrix),len(matrix[0])
        res=0
        arr=[0]*(n)
        for i in range(m):
            for j in range(n):
                if matrix[i][j]==1:
                    arr[j]+=1
                else:
                    arr[j]=0
            tmp=sorted(arr)+[0]
            stack=[-1]
            for k,num in enumerate(tmp):
                while stack[-1]>=0 and tmp[stack[-1]]>num:
                    index=stack.pop()
                    v=tmp[index]
                    res=max(res,(k-stack[-1]-1)*v)
                stack.append(k)
        return res
