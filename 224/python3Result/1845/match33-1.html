class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        h=len(matrix)
        w=len(matrix[0])
        
for j in range(w):
            for i in range(h-1):
                if matrix[i+1][j]:matrix[i+1][j]=matrix[i][j]+1
        ans=0
        for i,row in enumerate(matrix):
            row
.sort(reverse=True)
            for j,a in enumerate(row):
                ans=max(ans,a*(j+1))
        return ans
