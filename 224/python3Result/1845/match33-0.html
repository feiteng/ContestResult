class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        up = [None] * m
        for i in range(m):
            up[i] = [0] * n

        for j in range(n):
            up[0][j] = matrix[0][j]
        for i in range(1, m):
            for j in range(n):
                if matrix[i][j] == 1:
                    up[i][j] = up[i - 1][j] + 1
        max_size = 0
        for i in range(m):
            x = [up[i][j] for j in range(n)]
            x.sort(reverse=True)
            for idx, height in enumerate(x):
                max_size = max(max_size, (idx + 1) * height)
        return max_size
