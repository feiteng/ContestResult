class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        row = [0] * n
        res = 0
        for i in range(m):
            for j in range(n):
                row[j] = row[j] + matrix[i][j] if matrix[i][j] else 0
            sr = sorted(row)
            for j in range(n):
                if sr[j] == 0:
                    continue
                res = max(res, (n - j) * sr[j])
        return res
