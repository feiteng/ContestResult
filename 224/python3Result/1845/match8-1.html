class Solution:
    def largestSubmatrix(self, matrix: List[List[int]]) -> int:
        rows, cols = len(matrix), len(matrix[0])
        heights = [0] * cols
        ans = 0 
        for r in range(rows):
            
for c in range(cols):
                if matrix[r][c] == 1:
                    heights[c] += 1 
                else:
                    heights[c] = 0 

            
            temp = sorted(heights, reverse = True)
            
            for i in range(cols):
                ans = max(ans, temp[i] * (i + 1))
                
        return ans
