import java.util.ArrayList;
import java.util.Arrays;
import java.util.Comparator;
import java.util.List;
import java.util.stream.IntStream;

public class Solution {
    char[][] grid;
    int[][][][][] cat;
    int[][][][][] mouse;
    int cj, mj;
    int n;
    int m;

    
int[][] dirs = new int[][]{
            {-1, 0},
            {1, 0},
            {0, 1},
            {0, -1}
    };

    public boolean valid(int i, int j) {
        return i >= 0 && i < n && j >= 0 && j < m && grid[i][j] != '#';
    }

    public int cat(int r, int i, int j, int t, int k) {
        if (cat[r][i][j][t][k] == -1) {
            if (i == t && j == k) {
                return cat[r][i][j][t][k] = 1;
            }
            if (grid[i][j] == 'F') {
                return cat[r][i][j][t][k] = 0;
            }
            if (grid[t][k] == 'F') {
                return cat[r][i][j][t][k] = 1;
            }
            for (int[] d : dirs) {
                for (int jump = 0; jump <= cj; jump++) {
                    int x = t + d[0] * jump;
                    int y = k + d[1] * jump;
                    
if (!valid(x, y)) {
                        break;
                    }
                    if (mouse(r + 1, i, j, x, y) == 0) {
                        return cat[r][i][j][t][k] = 1;
                    }
                }
            }
            cat[r][i][j][t][k] = 0;
        }
        return cat[r][i][j][t][k];
    }

    public int mouse(
int r, int i, int j, int t, int k) {
        if (r >= 200) {
            return 0;
        }
        if (mouse[r][i][j][t][k] == -1) {
            if (i == t && j == k) {
                return mouse[r][i][j][t][k] = 0;
            }
            if (grid[i][j] == 'F') {
                return mouse[r][i][j][t][k] = 1;
            }
            if (grid[t][k] == 'F') {
                return mouse[r][i][j][t][k] = 0;
            }
            int ans = 0;
            for (int[] d : dirs) {
                for (int jump = 0; jump <= mj; jump++) {
                    int x = i + d[0] * jump;
                    int y = j + d[1] * jump;
                    if (!valid(x, y)) {
                        break;
                    }
                    if (cat(r, x, y, t, k) == 0) {
                        return mouse[r][i][j][t][k] = 1;
                    }
                }
            }
            mouse[r][i][j][t][k] = 0;
        }
        return mouse[r][i][j][t][k];
    }

    public boolean canMouseWin(String[] grid, int catJump, int mouseJump) {
        n = grid.length;
        m = grid[0].length();
        this.grid = new char[n][m];
        int mi = -1;
        int mj = -1;
        int ci = -1;
        int cj = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                this.grid[i][j] = grid[i].charAt(j);
                if (this.grid[i][j] == 'C') {
                    ci = i;
                    cj = j;
                }
                if (this.grid[i][j] == 'M') {
                    mi = i;
                    mj = j;
                }
            }
        }
        this.cj = catJump;
        this.mj = mouseJump;
        cat = new int[200][n][m][n][m];
        mouse = new int[200][n][m][n][m];
        for (int r = 0; r < 200; r++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    for (int t = 0; t < n; t++) {
                        for (int k = 0; k < m; k++) {
                            cat[r][i][j][t][k] = -1;
                            mouse[r][i][j][t][k] = -1;
                        }
                    }
                }
            
}
        }
        int ans = mouse(0, mi, mj, ci, cj);
        return ans == 1;
    }
}
