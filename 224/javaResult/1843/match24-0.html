class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int maxLen = Integer.MIN_VALUE;
        int count = 0;
        for (int[] p:rectangles) {
            int len = Math.min(p[0], p[1]);
            if (len > maxLen) {
                maxLen = len;
                count = 1;
            } else if (len == maxLen) {
                count++;
            }
        }
        return count;
    }
}
