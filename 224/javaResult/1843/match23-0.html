class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int ml = 0,mc = 0;
        for(int[] r : rectangles){
            int cml = Math.min(r[0],r[1]);
            if(cml > ml){
                ml = cml;
                mc = 1;
            }else if(cml == ml){
                mc++;
            }
        }
        return mc;
    }
}
