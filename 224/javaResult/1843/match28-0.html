class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        
        int res = 0;
        int max = Integer.MIN_VALUE;
        for (int [] rect: rectangles) {
            int min = Math.min(rect[0], rect[1]);
            if (min > max) {
                max = min;
                res = 1;
            } else if (min == max) {
                res++;
            }
        }
        return res;
        
    }
}
