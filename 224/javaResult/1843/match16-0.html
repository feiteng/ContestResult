class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int ret = 0, maxLen = 0;
        for(int[] r : rectangles) {
            int cur = Math.min(r[0], r[1]);
            if(cur > maxLen) {
                maxLen = cur;
                ret = 1;
            } else if(cur == maxLen) {
                ++ret;
            }
        }
        return ret;
    }
}
