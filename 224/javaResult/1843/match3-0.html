class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        HashMap<Integer, Integer> len2counts = new HashMap<>();
        int max = 0;
        for (int[] rect: rectangles) {
            int len = Math.min(rect[0], rect[1]);
            len2counts.put(len, len2counts.getOrDefault(len, 0)+1);
            max = Math.max(max, len);
        }
        
        return len2counts.getOrDefault(max, 0);
    }
}
