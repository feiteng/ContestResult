class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int best = 0;
        int count = 0;
        for (int[] r : rectangles) {
            int s = Math.min(r[0], r[1]);
            if (s > best) {
                count = 0;
                best = s;
            }
            if (s == best) {
                ++count;
            }
        }
        return count;
    }
}
