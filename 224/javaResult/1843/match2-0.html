class Solution {
    public int countGoodRectangles(int[][] A) {
        Map<Integer, Integer> map = new HashMap<>();
        int max = -1;
        for (int[] r : A) {
            int cur = Math.min(r[0], r[1]);
            map.put(cur, map.getOrDefault(cur, 0) + 1);
            max = Math.max(cur, max);
        }
        return map.get(max);
    }
}
