class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int max = 0;
        int cnt = 0;
        for(int[] hw : rectangles){
            int size = Math.min(hw[0], hw[1]);
            if(size > max){
                max = size;
                cnt = 0;
            }
            if(size == max){
                cnt++;
            }
        }
        return cnt;
    }
}
