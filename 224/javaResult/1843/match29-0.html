class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int maxLen = 0;
        int cnt = 0;
        for (int[] rectangle : rectangles) {
            int edge = Math.min(rectangle[0], rectangle[1]);
            if (edge > maxLen) {
                cnt = 1;
                maxLen = edge;
            } else if (edge == maxLen) {
                cnt++;
            }
        }
        return cnt;
    }
}
