class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        Map<Integer, Integer> map = new HashMap<>();
        int ml = 0;
        for (int[] rec: rectangles) {
            int l = Math.min(rec[0], rec[1]);
            map.put(l, map.getOrDefault(l, 0) + 1);
            ml = Math.max(ml, l);
        }
        return map.get(ml);
    }
}
