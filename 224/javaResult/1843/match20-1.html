class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int res = 0;
        int max = 0;
        for (int[] rec : rectangles) {
            int len = Math.min(rec[0], rec[1]);
            
            if (len > max) {
                res = 0;
                max = len;
                res++;
            }
            else if (len == max) {
                res++;
            }
        }
        
        return res;
    }
}
