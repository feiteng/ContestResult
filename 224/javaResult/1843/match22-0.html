class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int max = 0, maxCount = 0;
        for (int[] rectangle : rectangles){
            int cur = Math.min(rectangle[0], rectangle[1]);
            if (cur > max){
                max = cur;
                maxCount = 1;
            }else if (max == cur){
                maxCount ++;
            }
        }
        return maxCount;
    }
}
