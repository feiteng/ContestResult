class Solution {
    public int countGoodRectangles(int[][] rect) {
        int res = 0, max = -1;
        for (int[] i : rect) {
            int min = Math.min(i[0], i[1]);
            if (min > max) {
                max = min;
                res = 1;
            } else if (min == max) {
                res += 1;
            }
        }
        return res;
    }
}
