class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int max = 0;
        for(int[] r : rectangles) {
            int side = Math.min(r[0], r[1]);
            map.put(side, map.getOrDefault(side, 0) + 1);
            max = Math.max(max, side);
        }
        return map.get(max);
    }
}
