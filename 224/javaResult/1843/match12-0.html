class Solution {
    
    public int countGoodRectangles(int[][] rectangles) {

        int ans = 0;
        
        int maxLen = 0;
        for (int[] one : rectangles) {
            
            int len = Math.min(one[0], one[1]);
            if (maxLen < len) {
                
                ans = 0;
                maxLen = len;
            }
            
            if (maxLen == len) {
                
                ++ans;
            }
        }
        
        return ans;
    }
}
