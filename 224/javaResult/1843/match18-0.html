class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int c = 0, m = 0;
        for (int[] rect : rectangles) {
            int s = Math.min(rect[0], rect[1]);
            if (s > m) {
                m = s;
                c = 1;
            } else if (s == m) {
                c++;
            }
        }
        return c;
    }
}
