class Solution {
    public int countGoodRectangles(int[][] rectangles) {
        int ctr = 0, maxLen = 0;
        for (int[] rec: rectangles) {
            int lenSQ = Math.min(rec[0], rec[1]);
            if (lenSQ > maxLen) {
                maxLen = lenSQ;
                ctr = 0;
            }
            if (lenSQ == maxLen) {
                ctr++;
            }
        }
        return ctr;
    }
}
