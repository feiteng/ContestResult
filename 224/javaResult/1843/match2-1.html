class Solution {
    public int countGoodRectangles(int[][] recs) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int max = 0;
        for(int[] rec : recs){
            int min = Math.min(rec[0], rec[1]);
            map.put(min, map.getOrDefault(min, 0) + 1);
            max = Math.max(min, max);
        } 
        return map.get(max);
    }
}
