	class Solution {
		public int tupleSameProduct(int[] nums) {
			Map<Integer, Integer> map = new HashMap<>();
			for(int i = 0;i < nums.length;i++){
				for(int j = i+1;j < nums.length;j++){
					int x = nums[i]*nums[j];
					map.merge(x, 1, Integer::sum);
				}
			}
			int ans = 0;
			for(int v : map.values()){
				ans += v*(v-1)/2;
			}
			return ans*8;
		}
	}
