class Solution {
    public int tupleSameProduct(int[] nums) {
        Map<Integer,Integer> map = new HashMap<>();
        int n = nums.length;
        for(int i = 1;i < n;i++){
            for(int j = 0;j < i;j++){
                int key = nums[i] * nums[j];
                map.put(key,map.getOrDefault(key,0) + 1);
            }
        }
        int ans = 0;
        for(Map.Entry<Integer,Integer> entry : map.entrySet()){
            int cnt = entry.getValue();
            if(cnt > 1){
                ans += cnt * (cnt - 1) / 2;
            }
        }
        return ans * 8;
    }
}
