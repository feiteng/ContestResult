class Solution {
    public int tupleSameProduct(int[] nums) {
        //two sum idea ? 
        Arrays.sort(nums);
        int res = 0;
        for(int i = 0;i<nums.length - 3;i++){
            for(int j = nums.length - 1; j > i + 2; j--){
                int target = nums[i] * nums[j];
                int left = i + 1, right = j - 1;
                while(left < right) {
                    int square = nums[left] * nums[right];
                    if(square == target) {
                        res++;
                        left++;
                        right--;
                    } else if(square < target) {
                        left++;
                    } else {
                        right--;
                    }
                }
            }
        }
        return res * 8;
    }
}
