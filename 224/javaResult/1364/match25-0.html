class Solution {
    public int tupleSameProduct(int[] nums) {
        Map<Integer, Integer> mapping = new HashMap<>();
        int n = nums.length;
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int tmp = nums[i] * nums[j];
                mapping.put(tmp, mapping.getOrDefault(tmp, 0) + 1);
            }
        }
        
        int res = 0;
        
        for (Integer key : mapping.keySet()) {
            int v = mapping.get(key);
            
            if (v > 1) {
                res += v * (v - 1) * 4;
            }
        }
        
        return res;
    }
}
