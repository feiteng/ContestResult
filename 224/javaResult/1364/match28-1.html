class Solution {
    public int tupleSameProduct(int[] nums) {
        Arrays.sort(nums);
        int res = 0;
        for (int i = 0; i < nums.length; i ++){
            for (int j = nums.length - 1; j >= 0; j --){
                int product = nums[i] * nums[j];
                int l = i + 1, r = j - 1;
                while (l < r){
                    if (nums[l] * nums[r] == product){
                        res += 8;
                        l ++;
                        r --;
                    }else if (nums[l] * nums[r] < product){
                        l ++;
                    }else{
                        r --;
                    }
                }
            }
        }
        return res;
    }
}
