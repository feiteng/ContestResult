class Solution {
    public int tupleSameProduct(int[] nums) {
        int ans = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 3; i++) {
            for (int j = nums.length - 1; j >= 3; j--) {
                int mul1 = nums[i] * nums[j];
                int start = i + 1;
                int end = j - 1;
                while (start < end) {
                    int mul2 = nums[start] * nums[end];
                    
if (mul1 == mul2) {
                        ans += 8;
                        start++;
                        end--;
                    } else if (mul1 < mul2) {
                        end--;
                    } else {
                        start++;
                    }
                }
            }
        }
        return ans;
    }
}
