class Solution {
    public int tupleSameProduct(int[] nums) {
        // <products, count>
        HashMap<Long, Integer> products = new HashMap<Long, Integer>();
        for (int i = 0; i < nums.length-1; i++) {
            for (int j = i+1; j < nums.length; j++) {
                long curProduct = nums[i] * nums[j]; 
                products.put(curProduct, products.getOrDefault(curProduct, 0)+1);
            }
        }
        
        int set = 0;
        
        for (long prod: products.keySet()) {
            int count = products.get(prod);
            if (count >= 2) {
                set += count * (count-1)/2;
            }
        }
        
        return 8*set;
    }
}
