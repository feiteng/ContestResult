class Solution {
    public int tupleSameProduct(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        int len = nums.length;
        for (int i = 0;i < len;i++) {
            for (int j = 0;j < i;j++) {
                int n = map.getOrDefault(nums[i] * nums[j], 0);
                map.put(nums[i] * nums[j], n + 1);
            }
        }
        int res = 0;
        for (int key : map.keySet()) {
            int v = map.get(key);
            res += v * (v - 1) * 2 * 2;
        }
        return res;
    }
}
