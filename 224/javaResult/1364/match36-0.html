class Solution {
    public int tupleSameProduct(int[] nums) {
        int ans = 0;
        Arrays.sort(nums);
        for (int beg = 0; beg < nums.length; beg++) {
            for (int end = beg + 3; end < nums.length; end++) {
                int prod = nums[beg] * nums[end];
                int i = beg + 1, j = end - 1;
                while (i < j) {
                    if (nums[i] * nums[j] == prod) {
                        ans += 8;
                        i++;
                        j--;
                    } else if (nums[i] * nums[j] < prod) {
                        i++;
                    } else {
                        j--;
                    }
                }
            }
        }
        return ans;
    }
}
