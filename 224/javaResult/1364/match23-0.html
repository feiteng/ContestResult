class Solution {
    public int tupleSameProduct(int[] nums) {
        Arrays.sort(nums);
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                int multi = nums[i] * nums[j];
                int ll = i + 1;
                int rr = j - 1;
                while (ll < rr) {
                    int m = nums[ll] * nums[rr];
                    if (m > multi) {
                        rr--;
                    } else if (m < multi) {
                        ll++;
                    } else {
                        res += 8;
                        ll++;
                        rr--;
                    }
                }
            }
        }
        return res;
    }
}
