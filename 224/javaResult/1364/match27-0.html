class Solution {
    public int tupleSameProduct(int[] nums) {
        Map<Integer,Integer> cnt = new HashMap<>();
        int n = nums.length;
        for(int i = 0;i < n;++i) {
            for(int j = i+1;j < n;++j) {
                int val = nums[i] * nums[j];
                cnt.put(val,cnt.getOrDefault(val,0)+1);
            }
        }
        int res = 0;
        for(Integer key : cnt.keySet()) {
            int tot = cnt.get(key);
            if(tot>1) {
                res+=tot*(tot-1)/2*8;
            }
        }
        return res;
    }
}
