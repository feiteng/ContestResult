class Solution {
    public int tupleSameProduct(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int p = nums[i] * nums[j];
                
map.put(p, map.getOrDefault(p, 0) + 1);
            }
        }
        
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            int k = e.getValue();
            
if (k > 1) {
                res += k * (k - 1) * 4;
            }
        }
        
        return res;

    }
}
