class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[][] sum = new int[n][m];
        for(int j = 0; j < m; j++){
            sum[0][j] = matrix[0][j];
        }
        for(int i = 1; i < n; i++){
            for(int j = 0; j < m; j++){
                if(matrix[i][j] == 1)
{
                    sum[i][j] = sum[i - 1][j] + 1;
                }
            }
        }
        
        int res = 0;
        for(int i = 0; i < n; i++){
            Arrays.sort(sum[i]);
            for(int j = 0; j < m; j++){
                res = Math.max(res, sum[i][j] * (m - j));
            }
        }
        return res;
    }
}
