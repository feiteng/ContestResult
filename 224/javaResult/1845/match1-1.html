class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        int[] cur = new int[n];
        
        int res = 0;
        
        for (int i = 0; i < m; i ++) {
            for (int j = 0; j < n; j ++) {
                if (matrix[i][j] == 0) {
                    cur[j] = 0;
                } else {
                    cur[j] ++;
                }
            }
            
            int[] tmp = Arrays.copyOfRange(cur, 0, n);
            Arrays.sort(tmp);
            for (int k = n - 1; k >= 0 && tmp[k] > 0; k --) {
                res = Math.max(res, (n - k) * tmp[k]);
            }
        }
        
        return res;
    }
}
