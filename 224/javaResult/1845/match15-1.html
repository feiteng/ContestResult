class Solution {
    public int largestSubmatrix(int[][] matrix) {
         int m = matrix.length;
        int n = matrix[0].length;
        int[] height = new int[n];
        int area = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (matrix[i][j] == 1) {
                    height[j] += 1;
                }else{
                    height[j] = 0;
                }
            }
            int[] tmp = Arrays.copyOf(height, n);
            Arrays.sort(tmp);
            for (int k = 0; k < n; k++) {
                int cur = tmp[k] * (n - k);
                
area = Math.max(cur, area);
            }
        }

        return area;
    }
}
