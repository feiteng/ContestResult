class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] dp = new int[m][n];
        for (int j = 0; j < n; j++) {
            int curr = 0;
            
for (int i = 0; i < m; i++) {
                if (matrix[i][j] == 1) {
                    curr++;
                    dp[i][j] = curr;
                } else {
                    curr = 0;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            Arrays.sort(dp[i]);
            for (int j = 0; j < n; j++) {
                res = Math.max(res, (n-j)*dp[i][j]);
            }
        }
        return res;
    }
}
