class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] arr = new int[m][n];
        
        for (int i = 0; i < n; i++) {
            arr[0][i] = matrix[0][i];
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (matrix[i][j] == 0) {
                    arr[i][j] = 0;
                } else {
                    arr[i][j] = arr[i - 1][j] + 1;
                }
            }
        }
        
        int result = 0;
        for (int i = 0; i < m; i++) {
            Arrays.sort(arr[i]);
            for (int j = n - 1; j >= 0; j--) {
                result = Math.max(result, arr[i][j] * (n - j));
            }
        }
        
        return result;
    }
}
