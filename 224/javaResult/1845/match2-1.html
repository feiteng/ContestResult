class Solution {
    public int largestSubmatrix(int[][] matrix) {
        //按行处理
        int m = matrix.length, n = matrix[0].length;
        int[] high = new int[n];
        int ret = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(matrix[i][j] == 0) {
                    high[j] = 0;
                } else {
                    ++high[j];
                }
            }
            int[] t = Arrays.copyOf(high, n);
            Arrays.sort(t);
            //System.out.println(Arrays.toString(t));
            for(int j = 0; j < n; ++j) {
                ret = Math.max(ret, (n - j) * t[j]);
            }
        }
        return ret;
    }
}
