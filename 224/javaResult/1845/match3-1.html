class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[] sum = new int[m];
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (matrix[i][j] == 0) {
                    sum[j] = 0;
                }
                else {
                    sum[j] += 1;
                }
            }
            int[] curr = sum.clone();
            Arrays.sort(curr);
            for (int j = 0; j < m; j++) {
                res = Math.max(res, curr[j] * (m - j));
            }
        }
        
        return res;

    }
}
