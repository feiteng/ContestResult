class Solution {
    public int largestSubmatrix(int[][] A) {
        int m = A.length, n = A[0].length;
        for(int j = n - 1; j >= 0; j --) {
            for(int i = m - 2; i >= 0 ; i --) {
                if(A[i + 1][j] > 0 && A[i][j] > 0) {
                    A[i][j] = A[i + 1][j] + 1;
                
}
            }
        }
        int max = 0;
        for(int i = 0; i < m; i ++) {
            Arrays.sort(A[i]);
            for(int j = n - 1; j >= 0; j --) {
                max = Math.max(max, (n - j) * A[i][j]);
            }
        }
        return max;
    }
}
