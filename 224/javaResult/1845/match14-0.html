class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        
        for(int i = 1; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(matrix[i][j] == 1) {
                    matrix[i][j] += matrix[i - 1][j];
                }
            }
        }
        
        int ans = 0;
        for(int i = 0; i < m; i++) {
            //int[] tmp = Arrays.copyOf(matrix[i], n);
            Arrays.sort(matrix[i]);
            int temp = 0;
            for(int j = 0; j < n; j++) {
                temp = Math.max(temp, matrix[i][j]*(n-j));
            }
            ans = Math.max(ans, temp);
        }
        
        return ans;
    }
}
