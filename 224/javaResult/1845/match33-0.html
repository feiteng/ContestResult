class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int row=matrix.length,col=matrix[0].length;
        int[][] sum=new int[row][col];
        
        for(int i=0;i<row;i++){
            for(int j=0;j<col;j++){
                if(i>0){
                    if(matrix[i][j]==0)
                        sum[i][j]=0;
                    else
                        sum[i][j] =sum[i-1][j]+1;
                }else{
                    sum[i][j]=matrix[i][j];
                }
            }
        }
        
        int res=0;
        for(int i=0;i<row;i++){
            int[] arr=new int[col];
            for(int j=0;j<col;j++){
                arr[j]=sum[i][j];
            }
            Arrays.sort(arr);
            for(int k=1;k<=col;k++){
                res=Math.max(res,k*arr[col-k]);
            }
        }
        
        return res;
    }
}
