class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        for(int i=1; i<m; i++){
            for(int j=0; j<n; j++){
                if(matrix[i][j]==1){
                    matrix[i][j] += matrix[i-1][j];
                }
            }
        }
        
        
        int ans = 0;
        for(int[] h : matrix){
            
            Arrays.sort(h);
            for(int i=0; i<n; i++){
                ans = Math.max(ans, h[i] * (n-i));
            }
        }
        
        return ans;
    }
}
