class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int row = matrix.length, col = matrix[0].length;
        Integer[][] count = new Integer[row][col];
        for(int j = 0; j<col; j++){
            for(int i = 0; i<row; i++){
                if(matrix[i][j]==0) count[i][j] = 0;
                else {
                    count[i][j] = (i==0? 0 : count[i-1][j]) + 1;
                }
            }
        }
        int res = 0;
        for(int i = 0; i<row; i++){
            Integer[] cur = count[i];
            Arrays.sort(cur, Collections.reverseOrder());
            for(int j = 0; j<col; j++){
                res = Math.max(res, count[i][j] * (j+1));
            }
        }
        return res;
    }
}
