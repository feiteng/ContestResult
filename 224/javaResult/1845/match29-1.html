class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int max = 0;
        int row = matrix.length;
        int col = matrix[0].length;
        int[] prefix = new int[col];
        for (int i = 0; i < row; i++) {
            for (int j = 0;  j < col; j++) {
                if (matrix[i][j] == 0) {
                    prefix[j] = 0;
                } else {
                    prefix[j]++;
                }
            }
            max = Math.max(max, runMaxRect(prefix));
        }
        return max;
    }
    
    private int runMaxRect(int[] prefix) {
        
        int max = 0;
        int len = prefix.length;
        int[] array = Arrays.copyOf(prefix, len);
        
        Arrays.sort(array);
        
for (int i = 0; i < prefix.length; i++) {
            max = Math.max(max, (len - i) * array[i]);
        }
        return max;
    }
}
