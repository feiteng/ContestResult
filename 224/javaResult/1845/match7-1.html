class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        
        for (int j = 0; j < n; j++) {
            for (int i = m - 1; i >= 0; i--) {
                if (matrix[i][j] > 0) {
                    matrix[i][j] += (i + 1 < m ? matrix[i + 1][j] : 0);
                
}
            }
        }
        
        int ans = 0;
        for (int r = 0; r < m; r++) {
            Arrays.sort(matrix[r]);
            for (int c = n - 1; c >= 0; c--) {
                int w = n - c;
                int h = matrix[r][c];
                
ans = Math.max(ans, w * h);
            }
        }
        
        return ans;
    }
}
