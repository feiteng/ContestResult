class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        for(int  i= 0; i< n; i++){
            for(int j = 0; j<m; j++){
                if(i > 0){
                    if(matrix[i][j] == 1){
                        matrix[i][j] += matrix[i-1][j];
                    } 
                }
            }
        }
        int ans = 0;
        for(int  i= 0; i< n; i++){
            Arrays.sort(matrix[i]);
            for(int j = 0; j<m; j++){
                ans = Math.max(ans, matrix[i][j] * (m-j));
            }
        }
        return ans;
    }
}
