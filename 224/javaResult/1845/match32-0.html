	class Solution {
		public int largestSubmatrix(int[][] a) {
			int n = a.length, m = a[0].length;
			int[] os = new int[m];
			int max = 0;
			for(int i = n-1;i >= 0;i--){
				for(int j = 0;j < m;j++){
					if(a[i][j] == 1){
						os[j]++;
					}else{
						os[j] = 0;
					}
				}
				int[] temp = Arrays.copyOf(os, m);
				Arrays.sort(temp);
				for(int j = 0;j < m;j++){
					max = Math.max(max, temp[j]*(m-j));
				}
			}
			return max;
		}
	}
