class Solution {
    
public int maximumScore(int[] nums, int[] multipliers) {
        //Map<String, Integer> hm = new HashMap<>();
        int[][] arr = new int[multipliers.length+1][multipliers.length+2];
        for(int[]ar: arr) {
            Arrays.fill(ar, Integer.MIN_VALUE);
        
}
        return check(nums, multipliers, arr, 0, nums.length - 1, 0);
    }
    
    int check(int[] nums, int[] mul, int[][]arr, int start, int end, int m_s) {
        if(m_s == mul.length) {
            return 0;
        }
        //String cur = String.valueOf(start) + "," + String.valueOf(end);
        int back_index = nums.length - end;
        
if(arr[start][back_index] != Integer.MIN_VALUE) {
            return arr[start][back_index];
        }
        int out = Math.max(nums[start]*mul[m_s] + check(nums, mul, arr, start+1, end, m_s+1), nums[end]*mul[m_s] + check(nums, mul, arr, start, end-1, m_s+1));
        arr[start][back_index] = out;
        return out;
    }
}
