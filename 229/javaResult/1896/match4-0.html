class Solution {

    public int maximumScore(int[] nums, int[] multipliers) {
        int N = multipliers.length;
        int[][] f = new int[N + 1][N + 1];
        for (int i = N - 1; i >= 0; i--) {
            for (int ln = 0; ln <= i; ln += 1) {
                int rn = i - ln;
                f[i][ln] = Math.max(f[i + 1][ln + 1] + nums[ln] * multipliers[i],
                        f[i + 1][ln] + nums[nums.length - rn - 1] * multipliers[i]);
            }
        }
        return f[0][0];
    }
}
