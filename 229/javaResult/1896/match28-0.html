class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        Map<Integer, Map<Integer, Integer>> mem = new HashMap<Integer, Map<Integer, Integer>>();
        return traverse(nums, multipliers, 0, nums.length - 1, 0, mem);
    }
    
    private int traverse(int[] nums, int[] multipliers, int i, int j, int k, Map<Integer, Map<Integer, Integer>> mem) {
        if (k == multipliers.length) {
            return 0;
        }
        if (mem.containsKey(i) && mem.get(i).containsKey(j)) {
            return mem.get(i).get(j);
        }
        int front = traverse(nums, multipliers, i + 1, j, k + 1, mem);// cur + multipliers[k] * nums[i],
        int back = traverse(nums, multipliers, i, j - 1, k + 1, mem);// cur + multipliers[k] * nums[j],
        int max = Math.max(front + multipliers[k] * nums[i], back + multipliers[k] * nums[j]);
        if (!mem.containsKey(i)) {
            mem.put(i, new HashMap<Integer, Integer>());
        }
        mem.get(i).put(j, max);
        return max;
    }
}
