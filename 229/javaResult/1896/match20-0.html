class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int m = multipliers.length;
        int[][] memo = new int[m + 1][m + 1];
        return helper(0, nums.length - 1, nums, 0, multipliers, memo);
    }
    
    int helper(int left, int right, int[] nums, int index, int[] multipliers, int[][] memo) {
        if (index == multipliers.length) {
            return 0;
        }
        if (memo[left][nums.length - right - 1] != 0) {
            return memo[left][nums.length - right - 1];
        }
        
        int result1 = nums[left] * multipliers[index] + helper(left + 1, right, nums, index + 1, multipliers, memo);
        int result2 = nums[right] * multipliers[index] + helper(left, right - 1, nums, index + 1, multipliers, memo);
        
        memo[left][nums.length - right - 1] = Math.max(result1, result2);
        return memo[left][nums.length - right - 1];
    }
}
