class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n = nums.length;
        int m = multipliers.length;
        int[][] dp = new int[m + 1][m + 1];
        for (int[] row : dp) Arrays.fill(row, Integer.MIN_VALUE);
        return helper(dp, n, m, nums, multipliers, 0, 0);
    }
    
    private int helper(int[][] dp, int n, int m, int[] nums, int[] mul, int l, int r) {
        if (l + r == m) return 0;
        if (dp[l][r] != Integer.MIN_VALUE) return dp[l][r]
;
        int tmp1 = helper(dp, n, m, nums, mul, l + 1, r);
        int tmp2 = helper(dp, n, m, nums, mul, l, r + 1);
        dp[l][r] = Math.max(tmp1 + nums[l] * mul[l + r], tmp2 + nums[n - r - 1] * mul[l + r]);
        return dp[l][r];
    }
}
