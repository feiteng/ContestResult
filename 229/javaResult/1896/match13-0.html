class Solution {
    public int maximumScore(int[] nums, int[] mult) {
        int n = nums.length, m = mult.length;
        int[][] dp = new int[m+1][m+1];

        
        for (int i = 1; i <= m; i++) {
            dp[0][i] = dp[0][i-1] + nums[n-i] * mult[i-1];
            dp[i][0] = dp[i-1][0] + nums[i-1] * mult[i-1];
        }
        
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; i + j <= m; j++) {
                dp[i][j] = Math.max(dp[i-1][j] + mult[i+j-1] * nums[i-1], dp[i][j-1] + mult[i+j-1]*nums[n-j]);
            }
        }
        
        int sol = dp[0][m];
        for (int i = 1; i <= m; i++) {
            sol = Math.max(sol, dp[i][m-i]);
        }
        return sol;
    }
}
