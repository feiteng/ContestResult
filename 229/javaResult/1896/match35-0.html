class Solution {
    private class Key{
        int start;
        int end;
        
        public Key(int start, int end) {
            this.start = start;
            this.end = end;
        }
    }
    
    public int maximumScore(int[] nums, int[] multipliers) {
        int[][] mem = new int[multipliers.length][multipliers.length];
        
        for(int i = 0; i < mem.length; i++) {
            for(int j = 0; j < mem[0].length; j++) {
                mem[i][j] = -1;
            }
        }
        
        return getMaxScore(nums, multipliers, 0, nums.length - 1, 0, mem);
    }
    
    private int getMaxScore(int[] nums, int[] multipliers, int start, int end, int index, int[][] mem) {
        if(index >= multipliers.length) {
            return 0;
        }
        
        if(mem[start][index] != -1) {
            return mem[start][index];
        }
                
        int out = Math.max(nums[start] * multipliers[index] + getMaxScore(nums, multipliers, start+1, end, index+1, mem),
                           nums[end] * multipliers[index] + getMaxScore(nums, multipliers, start, end-1, index+1, mem));
        
        mem[start][index] = out;
        
        return out;                                                                                                        
    }
}
