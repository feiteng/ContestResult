class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n=nums.length, m=multipliers.length;
        int[][] cache = new int[m][m];
        for (int[] row: cache) Arrays.fill(row,Integer.MAX_VALUE);
        return navigate(nums,multipliers,0,0,n-1,m,n,cache);
    }
    private int navigate(int[] ns, int[] ms, int i, int a, int b, int m, int n, int[][] cache) {
        if (i==m) return 0;
        if (cache[a][n-b-1]<Integer.MAX_VALUE) return cache[a][n-b-1];
        int ans = Math.max(ns[a]*ms[i]+navigate(ns,ms,i+1,a+1,b,m,n,cache),ns[b]*ms[i]+navigate(ns,ms,i+1,a,b-1,m,n,cache));
        cache[a][n-b-1] = ans;
        return ans;
    }
}
