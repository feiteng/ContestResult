class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        // l, m
        int[][] cache = new int[multipliers.length][multipliers.length];
        for (int i = 0; i < cache.length; i++) {
            Arrays.fill(cache[i], Integer.MIN_VALUE);
        }
        return dfs(nums, 0, nums.length - 1, multipliers, 0, cache);
    }

	private int dfs(int[] nums, int l, int r, int[] m, int curM, int[][] cache) {
	    if (curM == m.length) {
	        return 0;
	    }
	    if (cache[l][curM] != Integer.MIN_VALUE){
	        return cache[l][curM];
	    }
	    
	    int max = Math.max(
	            dfs(nums, l + 1, r, m, curM + 1, cache) + nums[l] * m[curM],
	            dfs(nums, l, r - 1, m, curM + 1, cache) + nums[r] * m[curM]);
	    
        return cache[l][curM] = max;
    }
}
