class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n=nums.length;
        int m=multipliers.length;
        int[][] dp=new int[m+1][m+1];
        int res=Integer.MIN_VALUE;
        for(int k=1;k<=m;k++){
            for(int l=0;l<=k;l++){
                dp[l][k]=Math.max(l>0?dp[l-1][k-1]+nums[l-1]*multipliers[k-1]:Integer.MIN_VALUE,l<k?dp[l][k-1]+nums[n-(k-l)]*multipliers[k-1]:Integer.MIN_VALUE);
                if(k==m) res=Math.max(res,dp[l][k]);
            }
        }
        return res;
    }
}
