class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        
int[][] dp = new int[multipliers.length + 1][multipliers.length + 1];
        for (int i = 0; i < dp.length; ++i)
            Arrays.fill(dp[i], -1);
        return rec(0, 0, nums.length - 1, multipliers.length, nums, multipliers, dp);
    }
    
    int rec(
int i, int left, int right, int movesLeft, int[] nums, int[] m, int[][] dp) {
        if (left > right || movesLeft == 0 || i >= nums.length)
            return 0;
        if (dp[left][nums.length - right] != -1)
            return dp[left][nums.length - right];
        int res = Math.max(nums[left] * m[i] + rec(i + 1, left + 1, right, movesLeft - 1, nums, m, dp),
                        nums[right] * m[i] + rec(i + 1, left, right - 1, movesLeft - 1, nums, m, dp));
        dp[left][nums.length - right] = res;
        return dp[left][nums.length - right];
    }
}
