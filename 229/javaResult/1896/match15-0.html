class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int[][] dp = new int[multipliers.length + 1][multipliers.length + 1];        
        for (int i = 0; i < multipliers.length; i++) {
            dp[0][i + 1] = dp[0][i] + multipliers[i] * nums[nums.length - 1 - i];
            dp[i + 1][0] = dp[i][0] + multipliers[i] * nums[i];            
            for (int j = 1; j <= i; j++) {
                dp[j][i - j + 1] = Math.max(
                        dp[j - 1][i - j + 1] + multipliers[i] * nums[j-1] ,
                        dp[j][i - j] + multipliers[i] * nums[nums.length - 1 - i + j] 
                        );
            }
            // Arrays.stream(dp).forEach(line-> System.out.println(Arrays.toString(line)));
        }
        int ans = Integer.MIN_VALUE;
        for (int i = 0; i <= multipliers.length ; i++) {
            ans = Math.max(ans, dp[i][multipliers.length - i]);
        }
        return ans;
    }
}
