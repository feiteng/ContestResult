class Solution {
    public int maximumScore(int[] nums, int[] mul) {
        
int m = mul.length, n = nums.length;
        Integer[][] dp = new Integer[m + 1][m + 1];
        return dfs(nums, mul, 0, nums.length - 1, dp, 0);
    }
    
    private int dfs(int[] nums, int[] mul, int i, int j, Integer[][] dp, int count) {
        if (count == mul.length) {
            return 0;
        }
        if (dp[i][nums.length - j] != null) {
            return dp[i][nums.length - j];
        }
        int left = nums[i] * mul[count] + dfs(nums, mul, i + 1, j, dp, count + 1);
        int right = nums[j] * mul[count] + dfs(nums, mul, i, j - 1, dp, count + 1);
        dp[i][nums.length - j] = Math.max(left, right);
        return dp[i][nums.length - j];
    }
}
