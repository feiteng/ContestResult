class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n = nums.length, m = multipliers.length;
        int[][] dp = new int[m + 1][m + 1];
        
        for(int i = 0; i <= m; i++) {
            dp[i][i] = 0;
        }

        for(int j = 1; j <= m; j++) {
            for(int i = j - 1; i >= 0; i--) {
                int index = m - (j - i);
                dp[i][j] = Math.max(dp[i + 1][j] + nums[i] * multipliers[index], dp[i][j - 1] + nums[j + n - m - 1] * multipliers[index]);
                //System.out.println(dp[i][j]);
            }
        }
        return dp[0][m];
    }
}
