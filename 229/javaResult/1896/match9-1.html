class Solution {
    public int maximumScore(int[] nums, int[] mul) {
        //Map<Integer, Map<Integer, Integer>> cache = new HashMap<>();
        int n = nums.length;
        int m = mul.length;
        int[][] cache = new int[m][m];
        for (int i = 0; i < m; i++) {
            Arrays.fill(cache[i], Integer.MIN_VALUE);
        }
        return dp(0, 0, nums, mul, cache);
    }
    
    public int dp(int op, int start, int[] nums, int[] mul, int[][] cache) {
        int n = nums.length;
        int m = mul.length;
        
        
if (op == m) {
            return 0;
        }
        
        /*
        Map<Integer, Integer> map = cache.get(op);
        if (map == null) {
            map = new HashMap<>();
            cache.put(op, map);
        }
        if (map.containsKey(start)) {
            return map.get(start);
        }
        */
        if (cache[op][start] != Integer.MIN_VALUE) {
            return cache[op][start];
        }
        int res1 = nums[start] * mul[op] + dp(op + 1, start + 1, nums, mul, cache);
        int res2 = nums[start+n-op-1] * mul[op] + dp(op + 1, start, nums, mul, cache);
        
        cache[op][start] = Math.max(res1, res2);
        return cache[op][start];
    }
}
