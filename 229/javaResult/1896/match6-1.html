class Solution {
  public int maximumScore(int[] nums, int[] mul) {
    final int n = nums.length;
    final int m = mul.length;
    int[][] M = new int[m + 1][m + 1];
    for (int j = m - 1; j >= 0; j--) {
      
for (int i = 0; i <= j; i++) {
        M[i][j] = Math.max(nums[i] * mul[j] + M[i + 1][j + 1],
                           nums[nums.length - (j - i) - 1] * mul[j] + M[i][j + 1]);
      }
    }
    return M[0][0];
  }
}
