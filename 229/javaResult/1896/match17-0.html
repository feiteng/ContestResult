class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int m = multipliers.length;
        int n = nums.length;
        int[][] dp = new int[m + 1][m + 1];
        for (int i = 1; i <= m; i++) {
            dp[i][0] = dp[i - 1][0] + nums[n - i] * multipliers[i - 1];
            for (int j = 1; j < i; j++) {
                dp[i][j] = Integer.max(dp[i - 1][j - 1] + nums[j - 1] * multipliers[i - 1], dp[i - 1][j] + nums[n - i + j] * multipliers[i - 1]);
            }
            dp[i][i] = dp[i - 1][i - 1] + nums[i - 1] * multipliers[i - 1];
        }
        int res = Integer.MIN_VALUE;
        for (int i = 0; i <= m; i++) {
            res = Integer.max(res, dp[m][i]);
        }
        return res;
    }
}
