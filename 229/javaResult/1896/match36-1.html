class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int m = multipliers.length;
        int n = nums.length;
        int[] dp = new int[n + 1];
        int max =Integer.MIN_VALUE;
        for (int i = 1; i <= m; i++) {
            for (int l = i; l >= 0; l--) {
                int r = n - i + l;
                dp[l] = Math.max(l > 0 ? dp[l - 1] + nums[l - 1] * multipliers[i - 1] : Integer.MIN_VALUE,
                    r < n ? dp[l] + nums[r] * multipliers[i - 1] : Integer.MIN_VALUE);
                
if(i==m){
                    max =Math.max(max,dp[l]);
                }
            }
        }
        return max;
    }
}
