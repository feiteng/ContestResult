class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n = nums.length, m = multipliers.length;
        int[][] dp = new int[2][n];
        int ans = Integer.MIN_VALUE;
        for(int i = 1; i <= m; i++){
            for(int left = 0; left < n; left++){
                int x = i%2, y = (x+1)%2;
                int right = n - 1 + left - i;
                if(right >= n) break;
                if(left == 0) dp[x][left] = dp[y][left] + multipliers[i-1]*nums[right+1];
                else if(right == n-1) dp[x][left] = dp[y][left-1] + multipliers[i-1]*nums[left-1];
                else{
                    int a = dp[y][left] + multipliers[i-1]*nums[right+1];
                    int b = dp[y][left-1] + multipliers[i-1]*nums[left-1];
                    dp[x][left] = Math.max(a,b);
                }
                if(i == m) {
                    ans = Math.max(ans,dp[x][left]);
                }
            }
        }
        return ans;
    }
}
