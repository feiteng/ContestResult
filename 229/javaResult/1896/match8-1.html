	class S
olution {
		public int maximumScore(int[] nums, int[] multipliers) {
			int n = nums.length, m = multipliers.length;
			int[][] dp = new int[m+1][m+1];
			for(int i = 0;i < m+1;i++)Arrays.fill(dp[i], Integer.MIN_VALUE / 2);

			dp[0][0] = 0;
			int ans = Integer.MIN_VALUE;
			for(int u = 1;u <= m;u++){
				for(int i = 0;i <= u;i++){
					int j = u-i;
					if(i > 0){
						dp[i][j] = Math.max(dp[i][j], dp[i-1][j] + nums[i-1] * multipliers[u-1]);
					}
					if(j > 0){
						dp[i][j] = Math.max(dp[i][j], dp[i][j-1] + nums[n-j] * multipliers[u-1]);
					}
					if(u == m){
						ans = Math.max(ans, dp[i][j]);
					}
				}
			}
			return ans;
		}
	}
