class Solution {
    public int maximumScore(int[] nums, int[] muls) {
        int n = nums.length, m = muls.length;
        int[][] dp = new int[m + 1][m + 1];
        for(int sum = 1; sum <= m; sum++) {
            dp[0][sum] = dp[0][sum - 1] + muls[sum - 1] * nums[n - sum];
            dp[sum][0] = dp[sum - 1][0] + muls[sum - 1] * nums[sum - 1];
            for(int i = 1; i < sum; i++) {
                dp[i][sum - i] = Math.max(dp[i - 1][sum - i] + muls[sum - 1] * nums[i - 1], dp[i][sum - i - 1] + muls[sum - 1] * nums[n - sum + i]);
                //dp[i][sum - i] = Math.max(dp[i][sum - i], dp[i - 1][sum - i] + muls[sum - 1] * nums[i - 1]);
            }
        }
        //for(int i = 0; i <= m; i++) System.out.println(Arrays.toString(dp[i]));
        int res = Integer.MIN_VALUE;
        for(int i = 0; i <= m; i++) res = Math.max(res, dp[i][m - i]);
        return res;
    }
}
