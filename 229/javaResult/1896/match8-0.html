class Solution {
    // dp[i][j] = dp[i - 1][j] + nums[i] * mult, dp[i][j - 1] + nums[n - j] * mult
    
    
    public int maximumScore(int[] nums, int[] multipliers) {
        int n = nums.length;
        int m = multipliers.length;
        int[][] dp = new int[m + 1][m + 1];
        for (int i = 0; i <= m; i++) {
            Arrays.fill(dp[i], Integer.MIN_VALUE);
        }
        dp[0][0] = 0;
        int totalScore = Integer.MIN_VALUE;
        for (int len = 1; len <= m; len++) {
            for (int front = 0; front <= len; front++) {
                int back = len - front;
                if (front > 0) {
                    dp[front][back] = Math.max(dp[front][back], dp[front - 1][back] + nums[front - 1] * multipliers[len - 1]);
                }
                if (back > 0) {
                    dp[front][back] = Math.max(dp[front][back], dp[front][back - 1] + nums[n - back] * multipliers[len - 1]);
                }
                if (len == m) {
                    totalScore = Math.max(totalScore, dp[front][back]);
                }
            } 
        }
        return totalScore;     
    }
    
}
