class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int n = nums.length, m = multipliers.length;
        int[][] dp = new int[n][2];
        for (int j = 0; j + n - m < n; j++) {
            dp[j][0] = Math.max(nums[j] * multipliers[m - 1], nums[j + n - m] * multipliers[m - 1]);
        }
        for (int i = 2; i <= m; i++) {
            int r = n - m + i - 1;
            for (int j = 0; j + r < n; j++) {
                dp[j][(i - 1) % 2] = Math.max(nums[j] * multipliers[m - i] + dp[j + 1][i % 2], nums[j + r] * multipliers[m - i] + dp[j][i % 2]);
            }
        }
        return dp[0][(m + 1) % 2];
    }
}
