class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int m = multipliers.length;
        int n = nums.length;
        
        int [] dp = new int [m+1];
        for (int j = m-1; j>=0; --j) {
            int mulj = multipliers[j];
            for (int i = 0; i<=j; ++i) {
                int k = n - (j - i) - 1;
                
int Up = dp[i+1] + nums[i]*mulj;
                int Over = dp[i] + nums[k]*mulj;
                dp[i] = Math.max(Up, Over);
            }
        }
        return dp[0];
    }
}
