class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        int m = multipliers.length;
        int n = nums.length;
        int[][] f = new int[m + 1][m + 1];
        int res = Integer.MIN_VALUE;
        for (int k = 1; k <= m; k ++) {
            for (int i = 0; i <=k ; i ++) {
                int j = k  - i;
                f[i][j] = Integer.MIN_VALUE;
                if (i > 0) f[i][j] = Math.max(f[i][j], f[i - 1][j] + nums[i - 1] * multipliers[k - 1]);
                if (j > 0) f[i][j] = Math.max(f[i][j], f[i][j - 1] + nums[n - j] * multipliers[k - 1]);
                
if (k == m) res = Math.max(res, f[i][j]);
            }
        }
        return res;
    }
}
