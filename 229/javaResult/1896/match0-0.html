class Solution {
    public int maximumScore(int[] nums, int[] mps) {
        int n = nums.length;
        int m = mps.length;
        int[][] dp = new int[m + 1][m + 1];
        for (int k = m - 1; k >= 0; k--) {
            for (int i = 0; i < k + 1; i++) {
                dp[k][i] = Math.max(dp[k + 1][i] + nums[i + n - k - 1] * mps[k],
                                    dp[k + 1][i + 1] + nums[i] * mps[k]);
            }
        }
        return dp[0][0];
    }
}
