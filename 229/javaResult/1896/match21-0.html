class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        //n>=m
        int n = nums.length, m = multipliers.length;
        int[][] dp = new int[m+1][m+1];
        for(int[] row:dp) Arrays.fill(row,Integer.MAX_VALUE);
        
        return dfs(nums,multipliers,0,n-1,m,n,dp);
    }
    
    public int dfs(int[] nums, int[] multi, int i, int j,int m,int n,int[][] dp){
        if(dp[i][n-1-j]!=Integer.MAX_VALUE) return dp[i][n-1-j];
        if(n-(j-i+1)==m){
            dp[i][n-1-j]=0;
            return dp[i][n-1-j];
        }
        int index = n-(j-i+1);
        if(i==j){
            dp[i][n-1-j] = nums[i]*multi[index];
            return dp[i][n-1-j];
        }
        
         
        int takeLeft = dfs(nums,multi,i+1,j,m,n,dp)+nums[i]*multi[index];
        int takeRight = dfs(nums,multi,i,j-1,m,n,dp)+nums[j]*multi[index];
        
        dp[i][n-1-j] = Math.max(takeLeft,takeRight);
        return dp[i][n-1-j];
    }
}
