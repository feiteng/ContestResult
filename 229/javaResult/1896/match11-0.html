class Solution {
  // dp[i][j] i is total used, j is used from left part
  public int maximumScore(int[] nums, int[] multipliers) {
    int cnt = multipliers.length, n = nums.length, dp[][] = new int[cnt + 1][nums.length + 1], res = Integer.MIN_VALUE;
    
    dp[1][1] = multipliers[0] * nums[0];
    dp[1][0] = multipliers[0] * nums[n - 1];
    for(int i = 2; i <= cnt; i++) {
      dp[i][0] = dp[i-1][0] + multipliers[i-1] * nums[n - i];
      dp[i][i] = dp[i-1][i-1] + multipliers[i-1] * nums[i - 1];
      
      for(int j = 1; j < i; j++) {
        dp[i][j] = Math.max(dp[i-1][j] + multipliers[i - 1] * nums[n - (i - j)], dp[i-1][j-1] + multipliers[i - 1] * nums[j - 1]);
      }
    }
    for(int i = 0; i <= cnt; i++) res = Math.max(res, dp[cnt][i]);
    
    return res;
  }
}
