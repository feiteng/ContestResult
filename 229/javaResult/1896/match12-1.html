class Solution {
    public int maximumScore(int[] nums, int[] multipliers) {
        return sln1(nums, multipliers);
    }
    
    
private int sln1(int[] nums, int[] mult){
        int n = nums.length;
        int m =mult.length;
        int[][] dp = new int[m+1][m+1];
        for(int i = 1;i<=m;i++){
            dp[i][i] = dp[i-1][i-1]+nums[i-1]*mult[i-1];
            dp[i][0] = dp[i-1][0]+nums[n-i]*mult[i-1];
        }
        for(int i = 1;i<=m;i++){
            
for(int j = 1; j<=i-1;j++){
                dp[i][j] = Math.max(
                    dp[i-1][j-1]+nums[j-1]*mult[i-1],
                    dp[i-1][j] + nums[n-(i-j)]*mult[i-1]
                );
            }
        }
        int max = Integer.MIN_VALUE;
        for(int i : dp[m]) max = Math.max(max, i);
        return max;
    }
}
