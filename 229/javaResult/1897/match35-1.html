class Solution {
    public int longestPalindrome(String word1, String word2) {

    return helper(word1,word2);

}
    
private int helper(String word1,String word2){
        char[] c1 = word1.toCharArray();
        char[] c2 = word2.toCharArray();
        int m = c1.length;
        int n = c2.length;
//        [i,m)+[0,j)
        int[][] dp111 = longestPalindromeSubseq(word1);
        int[][] dp222 = longestPalindromeSubseq(word2);
        int[][] dp = new int[m+1][n+1];
        for(int i = m-1; i >= 0; i --){
            dp[i][0] = dp111[i][m-1];
        }
        
for(int j = 1; j < n; j ++){
            dp[m][j] = dp222[0][j-1];
        }
        int max = 0;
        for(int i = m-1; i >= 0; i --){
            for(int j = 1; j <= n; j ++){
                if(c1[i] == c2[j-1]){
                    dp[i][j] = dp[i+1][j-1]+2;
                    max = Math.max(dp[i][j],max);
                
}else{
                    dp[i][j] = Math.max(Math.max(dp[i+1][j],dp[i][j-1]),dp[i+1][j-1]);
                }

            }
        }
        // if(dp[0][n] == dp111[0][m-1] || dp[0][n] == dp222[0][n-1]) return 0;

        return max;
    }
    public int[][] longestPalindromeSubseq(String s) {
//
//         bbbab
//         OPT(i,j) =   OPT(i+1,j-1)+2                  if s.char(i) == s.char(j)
//                  =  max{OPT(i+1,j) ,OPT(i,j-1)}      o.w.

        int[][] dp = new int[s.length()][s.length()];

        for(int i = dp.length-1; i >= 0; i --)
        {
            for(int j = i; j < dp[0].length; j++)
            {
                if(i==j)
                {
                    dp[i][j] = 1;
                }else if(i == j-1 && s.charAt(i) == s.charAt(j))
                {
                    dp[i][j] = 2;
                }else if(s.charAt(i) == s.charAt(j))
                {
                    dp[i][j] = dp[i+1][j-1] +2;
                }else
                {
                    dp[i][j] = Math.max(dp[i+1][j],dp[i][j-1]);
                }
            }
        }
        return dp;
    }
}
