class Solution {
    public int longestPalindrome(String w1, String w2) {
        String w = new StringBuilder(w1 + w2).toString();
        
int k = w.length(), n = w2.length(), m = w1.length();
        int res = 0;
        int[][] dp = new int[k][k];
        for (int l = 1; l <= k; l++) {
            for (int ll = 0; ll <= k - l; ll++) {
                int rr = ll + l - 1;
                
int ss = 0;
                if (w.charAt(ll) == w.charAt(rr)) {
                    if (ll + 2 >= rr) {
                        ss = rr - ll + 1;
                    } else {
                        ss = dp[ll + 1][rr - 1] + 2;
                    }
                    if (ll < m && rr >= m) {
                        res = Math.max(res, ss);
                    }
                } else {
                    ss = Math.max(dp[ll + 1][rr], dp[ll][rr - 1]);
                }
                dp[ll][rr] = ss;
            }
        }
        return res;
    }
}
