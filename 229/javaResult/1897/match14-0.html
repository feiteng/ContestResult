class Solution {
    public int longestPalindrome(String word1, String word2) {
        char[] s = (word1 + word2).toCharArray();
        int m = word1.length(), n = s.length, max = 0;
        int[][] maxPal = new int[n][n];
        for (int i = 0; i < n; i++) {
            maxPal[i][i] = 1;
            for (int j = i - 1; j >= 0; j--) {
                if (s[j] == s[i]) {
                    maxPal[j][i] = maxPal[j + 1][i - 1] + 2;
                    if (j < m && i >= m) {
                        max = Math.max(max, maxPal[j][i]);
                    }
                } else {
                    maxPal[j][i] = Math.max(maxPal[j][i - 1], maxPal[j + 1][i]);
                }
            }
        }
        return max;
    }
}
