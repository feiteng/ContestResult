class Solution {
    public int longestPalindrome(String w1, String w2) {
        int l1 = w1.length();
        int l2 = w2.length();
        w1 = w1+w2;
        char c1[] = w1.toCharArray();

        int dp1[][] = new int[c1.length+1][c1.length+1];
        

        
        
int rs = 0;
        for(int i=c1.length-1;i>=0;--i){
   
            for(int j=i;j<c1.length;++j){
                if(i==j){
                    dp1[i][i] = 1; continue;
                }
                if(c1[i]==c1[j]){
                    dp1[i][j] = 2 + dp1[i+1][j-1];
                    
                    
if(i<l1&&j>=l1){
                        rs = Math.max(rs,dp1[i][j]);
                    }
                    
                }else{
                    dp1[i][j] = Math.max(dp1[i+1][j], dp1[i][j-1]);
                }
                
            }
           
            
        }
        
        
        return rs;
        
    }
}
