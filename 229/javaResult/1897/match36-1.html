class Solution {
    public int longestPalindrome(String word1, String word2) {
        return longestPalindromeSubseq(word1 + word2, word1.length());
    }
    
    public int longestPalindromeSubseq(String s, int length) {
        int n = s.length();
        int max = 0;
        if (n < 2) return n;
        int[][][] dp = new int[n][n][3];
        
for (int len = 0; len <= n - 1; len++) {
            for (int left = 0; left + len < n ; left++) {
                if (len == 0) {
                    dp[left][left] = new int[]{left, left, 1};
                    continue;
                }
                int right = left + len;
                if (s.charAt(left) == s.charAt(right)) {
                    dp[left][right] = new int[]{left, right, dp[left + 1][right - 1][2] + 2};
                    if (left < length && right >= length) {
                        max = Math.max(max, dp[left][right][2]);
                    }
                } else {
                    int d1 = dp[left + 1][right][2], d2 = dp[left][right - 1][2];
                    if (d1 > d2) {
                        dp[left][right] = new int[]{dp[left + 1][right][0], dp[left + 1][right][1], d1};
                        if (dp[left + 1][right][0] < length && dp[left + 1][right][1] >= length) {
                            max = Math.max(max, dp[left][right][2]);
                        }
                    } else if (d1 < d2) {
                        dp[left][right] = new int[]{dp[left][right - 1][0], dp[left][right - 1][1], d2};
                        
if (dp[left][right - 1][0] < length && dp[left][right - 1][1] >= length) {
                            max = Math.max(max, dp[left][right][2]);
                        }                        
                    } else {
                        dp[left][right] = new int[]{dp[left + 1][right][0], dp[left + 1][right][1], d1};
                        if (dp[left + 1][right][0] < length && dp[left + 1][right][1] >= length) {
                            max = Math.max(max, dp[left][right][2]);
                        }
                        if (dp[left][right - 1][0] < length && dp[left][right - 1][1] >= length) {
                            max = Math.max(max, dp[left][right][2]);
                        }  
                    }
                
}
            }
        }
        return max;
    }

}
