class Solution {
    public int longestPalindrome(String word1, String word2) {
        int ans = 0;
        int l1 = word1.length(), l2 = word2.length();
        String s = word1 + word2;
        
        int[][] f = new int[l1 + l2][l1 + l2];
        for(int i = l1 + l2 - 1; i >= 0; -- i) {
            f[i][i] = 1;
            for(int j = i + 1; j < l1 + l2; ++ j) {
                if(s.charAt(i) == s.charAt(j)) {
                    f[i][j] = f[i + 1][j - 1] + 2;
                    if(i < l1 && j >= l1) ans = Math.max(ans, f[i][j]);
                } else {
                    f[i][j] = Math.max(f[i + 1][j], f[i][j - 1]);
                }
            }
        }
        
        // for(int len = 1; len <= l1 + l2; ++ len) {

        //         int j = i + len - 1;
        //         if(i == j) {
        //             f[i][j] = 1;
        //             continue;
        //         }
        //         if(i + 1 == j) {
        //             if(s.charAt(i) == s.charAt(j)) {
        //                 f[i][j] = 2;
        //                 if(i < l1 && j >= l1) ans = Math.max(ans, f[i][j]);
        //             } else {
        //                 f[i][j] = 0;
        //             }
        //             continue;
        //         }
        //         if(s.charAt(i) == s.charAt(j)) {
        //             f[i][j] = f[i + 1][j - 1] + 2;
        //             if(i < l1 && j >= l1) ans = Math.max(ans, f[i][j]);
        //         } else {
        //             f[i][j] = Math.max(f[i + 1][j], f[i][j - 1]);
        //         }
        //     }
        // }
        System.out.println(f[0][l1 + l2 - 1]);
        return ans;
    }
}
