class Solution {
    public int longestPalindrome(String word1, String word2) {
        int l1 = word1.length();
        int l2 = word2.length();
        int l = l1 + l2;
        String s = word1 + word2;
        char[] arr = s.toCharArray();
        int[][] dp = new int[l][l];
        for (int i = 0; i < l; i ++) {
            dp[i][i] = 1;
        }
        int res = 0;
        for (int i = l - 1; i >= 0; i --) {
            for (int j = i + 1; j < l; j ++) {
                if (arr[i] == arr[j]) {
                    dp[i][j] = 2 + dp[i + 1][j - 1];
                    if (i < l1 && j >= l1) {
                        res = Math.max(res, dp[i][j]);
                    }
                } else {
                    dp[i][j] = Math.max(Math.max(dp[i + 1][j], dp[i][j - 1]), dp[i + 1][j - 1]);
                }
                
            }
        }
        return res;
    }
}
