class Solution {
    public int longestPalindrome(String word1, String word2) {
        String s=word1+word2;
        int len1=word1.length();
        int n = s.length();
        int ans=0;
        int[][] f = new int[n][n];
        for (int i = n - 1; i >= 0; i--) {
            f[i][i] = 1;
            for (int j = i + 1; j < n; j++) {
                if (s.charAt(i) == s.charAt(j)) {
                    f[i][j] = f[i + 1][j - 1] + 2;
                    if(i<len1&&j>=len1)
                        ans=Math.max(f[i][j],ans);
                } else {
                    f[i][j] = Math.max(f[i + 1][j], f[i][j - 1]);
                }
            }
        }
        return ans;
    }
}
