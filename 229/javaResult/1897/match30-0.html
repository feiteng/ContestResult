class Solution {
    public int longestPalindrome(String word1, String word2) {
        int m = word1.length();
        int n = word2.length();
        int[][] dp1 = calculatePalindrome(word1);
        int[][] dp2 = calculatePalindrome(word2);
        
        int[][] dp = new int[m][n];
        dp[0][n-1] = word1.charAt(0) == word2.charAt(n-1) ? 2 : 0;
        
        int res = 0;
        for (int j = n-2; j >= 0; j--) {
            dp[0][j] = word1.charAt(0) == word2.charAt(j) ? 2 : dp[0][j+1];
        }
        for (int i = 1; i < m; i++) {
            dp[i][n-1] = word1.charAt(i) == word2.charAt(n-1) ? 2 : dp[i-1][n-1];
        }
        for (int i = 1; i < m; i++) {
            for (int j = n-2; j >= 0; j--) {
                if (word1.charAt(i) == word2.charAt(j)) {
                    dp[i][j] = 2 + dp[i-1][j+1];
                } else {
                    dp[i][j] = Math.max(dp[i-1][j], dp[i][j+1]);
                }
            }
        }
        
        // System.out.println(Arrays.deepToString(dp));
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (dp[i][j] != 0) {
                    res = Math.max(res, dp[i][j]);
                    if (i+1 < m) res = Math.max(res, dp[i][j] + dp1[i+1][m-1]);
                    if (j-1 >= 0) res = Math.max(res, dp[i][j] + dp2[0][j-1]);
                }
            }
        }
        return res;
    }
    
    private int[][] calculatePalindrome(String s) {
        int n = s.length();
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++) dp[i][i] = 1;
        for (int j = 1; j < n; j++) {
            for (int i = j-1; i >= 0; i--) {
                if (s.charAt(i) == s.charAt(j)) dp[i][j] = 2 + dp[i+1][j-1];
                else dp[i][j] = Math.max(dp[i+1][j], dp[i][j-1]);
            }
        }
        return dp;
    }
}
