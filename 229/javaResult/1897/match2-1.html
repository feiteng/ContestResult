class Solution {
    public int longestPalindrome(String word1, String word2) {
        String common = word1 + word2;
        int n = common.length();
        int border = word1.length();
        int mx = 0;
        int[][] lps = new int[n][n];
        for(int i = n - 1; i >= 0; i--) {
            lps[i][i] = 1;
            for(int j = i + 1; j < n; j++) {
                if(common.charAt(i) == common.charAt(j)) {
                    lps[i][j] = 2 + lps[i + 1][j - 1];
                    if(i < border && j >= border) mx = Math.max(mx, lps[i][j]);
                }
                else {
                    lps[i][j] = Math.max(lps[i + 1][j], lps[i][j - 1]);
                } 
            }
        }
        
        return mx;
        
    }
}
