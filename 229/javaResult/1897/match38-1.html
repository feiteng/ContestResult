class Solution {
    public int longestPalindrome(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        sb.append(word1);
        sb.append(word2);
        int s1 = word1.length();
        int s2 = word2.length();
        char[] ca = sb.toString().toCharArray();
        int n = ca.length;
        
int[][] dp = new int[n][n];
        for (int d = 0; d < n; d++) {
            for (int i = 0; i + d < n; i++) {
                int j = i + d;
                if (i == j) dp[i][j] = 1;
                else if (ca[i] == ca[j]) dp[i][j] = dp[i + 1][j - 1] + 2;
                else dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1])
;
            }
        }
        
        int ans = 0;
        for (int i = 0; i < s1; i++) {
            for (int j = n - 1; j >= n - s2; j--) {
                if (ca[i] == ca[j]) {
                    ans = Math.max(ans, dp[i][j]);
                    break;
                
}
            }
        }
        
        // System.out.println(Arrays.deepToString(dp));
        return ans;
    }
}
