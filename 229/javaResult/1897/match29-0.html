class Solution {
    public int longestPalindrome(String word1, String word2) {
        String w = word1 + word2;
        int k = w.length();
        int m = word1.length();
        int n = word2.length();
        int res = 0;
        int[][] dp = new int[k][k];
        for (int len = 1; len <= k; len++) {
            for (int left = 0; left <= k - len; left++) {
                int right = left + len - 1;
                if (w.charAt(left) == w.charAt(right)) {
                    if (left + 2 >= right) {
                        dp[left][right] = right - left + 1;
                    } else {
                        dp[left][right] = dp[left + 1][right - 1] + 2;
                    }
                    if (left < m && right >= m) {
                        res = Math.max(res, dp[left][right]);
                    }
                } else {
                    dp[left][right] = Math.max(dp[left + 1][right], dp[left][right - 1]);
                }
                
            }
        }
        return res;
    }
}
