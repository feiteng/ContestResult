class Solution {
    public String mergeAlternately(String word1, String word2) {
        int N1 = word1.length();
        int N2 = word2.length();
        int i1 = 0;
        int i2 = 0;
        StringBuilder sb = new StringBuilder();
        while (i1 < N1 && i2 < N2) {
            sb.append(word1.charAt(i1++));
            sb.append(word2.charAt(i2++));
        }
        while (i1 < N1) {
            sb.append(word1.charAt(i1++));
        }
        while (i2 < N2) {
            sb.append(word2.charAt(i2++));
        }
        return sb.toString();
    }
}
