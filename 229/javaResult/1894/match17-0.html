class Solution {
    public String mergeAlternately(String word1, String word2) {
        int i = 0, j = 0;
        int m = word1.length(), n = word2.length();
        StringBuffer sb = new StringBuffer();
        while(i < m || j < n) {
            if(i < m) {
                sb.append(word1.charAt(i));
                i++;
            }
            if(j < n) {
                sb.append(word2.charAt(j));
                j++;
            }
        }
        return sb.toString();
    }
}
