class Solution {
    public String mergeAlternately(String w1, String w2) {
        
        int p1 = 0;
        int p2 = 0;
        int l1 = w1.length();
        int l2 = w2.length();
        StringBuilder sb = new StringBuilder();
        while(p1<l1||p2<l2){
            if(p1<l1){
                sb.append(w1.charAt(p1++));
            }
            if(p2<l2){
                sb.append(w2.charAt(p2++));
            }
            
            
        }
        return sb.toString();
        
        
    }
}
