class Solution {
    public String mergeAlternately(String word1, String word2) {
        int l1 = word1.length();
        int l2 = word2.length();
        StringBuilder sb = new StringBuilder();
        int i1 = 0, i2 = 0;
        while(i1 < l1 && i2 < l2) {
            sb.append(word1.charAt(i1++));
            sb.append(word2.charAt(i2++));
        }
        while(i1 < l1) {
            sb.append(word1.charAt(i1++));
        }
        while(i2 < l2) {
            sb.append(word2.charAt(i2++));
        }
        return sb.toString();
    }
}
