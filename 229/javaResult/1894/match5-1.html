class Solution {
    public String mergeAlternately(String word1, String word2) {
        int l1 = word1.length();
        int l2 = word2.length();
        int a = 0;
        int b = 0;
        StringBuilder sb = new StringBuilder();
        while (a < l1 && b < l2) {
            sb.append(word1.charAt(a));
            sb.append(word2.charAt(b));
            a++;
            b++;
        }
        if (a < l1) {
            sb.append(word1.substring(a));
        }
        if (b < l2) {
            sb.append(word2.substring(b));
        }
        return sb.toString();
    }
}
