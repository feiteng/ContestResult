class Solution {
    public String mergeAlternately(String w1, String w2) {
      int n = w1.length(), m = w2.length();
        StringBuffer x = new StringBuffer();
      int i = 0, j= 0;
      while(i < n && j < m) {
        x.append(w1.charAt(i)); i++;
        x.append(w2.charAt(j)); j++;
      }
      while(i < n) {x.append(w1.charAt(i)); i++;}
      while(j < m) {x.append(w2.charAt(j)); j++;}
      return x.toString();
    }
}
