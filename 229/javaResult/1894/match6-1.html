class Solution {
    public String mergeAlternately(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int l=0,r=0;
        StringBuilder sb = new StringBuilder();
        while(l<m && r<n){
            sb.append(word1.charAt(l++));
            sb.append(word2.charAt(r++));
        }
        while(l<m){
            sb.append(word1.charAt(l++));
        }
        while(r<n){
            sb.append(word2.charAt(r++));
        }
        return sb.toString();
    }
}
