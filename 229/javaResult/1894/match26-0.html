class Solution {
    public String mergeAlternately(String word1, String word2) {
        int l1 = word1.length();
        int l2 = word2.length();
        
        StringBuilder result = new StringBuilder();
        
        int p1=0;
        int p2=0;
        while(p1<l1 && p2<l2){
            result.append(word1.charAt(p1++));
            result.append(word2.charAt(p2++));
        }
        
        while(p1<l1){
            result.append(word1.charAt(p1++));
        }
        
        while(p2<l2){
            result.append(word2.charAt(p2++));
        }
        return result.toString();
        
    }
}
