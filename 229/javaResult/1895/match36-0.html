class Solution {
    public int[] minOperations(String box) {
        char[] boxes = box.toCharArray();
        int[] out = new int[boxes.length];
        for(int i = 0;i<boxes.length;i++) {
            int sum = 0;
            for(int j = 0;j<boxes.length;j++) {
                if(boxes[j] == '1') {
                    sum+=(Math.abs(i-j));
                }
            }
            out[i] = sum;
        }
        return out;
    }
}
