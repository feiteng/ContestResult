class Solution {
    public int[] minOperations(String boxes) {
        int N = boxes.length();
        int[] res = new int[N];
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (i != j && boxes.charAt(j) == '1') {
                    int dist = Math.abs(i - j);
                    res[i] += (dist);
                }
            }
        }
        
        return res;
    }
}
