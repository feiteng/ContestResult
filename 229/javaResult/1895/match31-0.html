class Solution {
    public int[] minOperations(String boxes) {
        char[] chs = boxes.toCharArray();
        int[] res = new int[chs.length];
        for(int i = 0; i < chs.length; ++i) {
            for(int j = 0; j < chs.length; ++j) {
                if(chs[j] == '1') res[i] += Math.abs(i - j);
            }
        }
        return res;
    }
}
