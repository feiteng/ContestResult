class Solution {
    public int[] minOperations(String boxes) {
        int[] left = new int[boxes.length()];
        int[] right = new int[boxes.length()];
        int curr = 0;
        for(int i = 0; i < boxes.length(); i++) {
            left[i] = ((i > 0) ? left[i - 1] : 0) + curr;
            if(boxes.charAt(i) == '1') {
                curr++;
            }
        }
        curr = 0;
        for(int i = boxes.length() - 1; i >= 0; i--) {
            right[i] = ((i < boxes.length() - 1) ? right[i + 1] : 0) + curr;
            if(boxes.charAt(i) == '1') {
                curr++;
            }
        }
        int[] output = new int[boxes.length()];
        for(int i = 0; i < boxes.length(); i++) {
            output[i] = left[i] + right[i];
        }
        return output;
    }
}
