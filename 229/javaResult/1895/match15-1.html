class Solution {
    public int[] minOperations(String b) {
        char[] boxes = b.toCharArray();
        int n = boxes.length;
        int[] ans = new int[n];
        
        for(int i = 0 ; i < n ; i++) {
            int temp = 0;
            for(int j = 0 ;j < n ; j++) {
                if(boxes[j] == '1' && i != j) {
                    temp += Math.abs(i-j);
                }
            }
            ans[i] = temp;
        }
        
        return ans;
    }
}
