class Solution {
    // 解法1
    // 两遍扫描
    public int[] minOperations(String boxes) {
        int len = boxes.length();
        int[] res = new int[len];
        int acc = 0, count = 0;
        for (int i = 0; i < len; i++) {
            res[i] += acc;
            if (boxes.charAt(i) == '1') {
                count++;
            }
            acc += count;
        }
        acc = 0;
        count = 0;
        for (int i = len - 1; i >= 0; i--) {
            res[i] += acc;
            if (boxes.charAt(i) == '1') {
                count++;
            }
            acc += count;
        }
        return res;
    }
}
