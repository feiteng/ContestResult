class Solution {
    public int[] minOperations(String boxes) {
        int n = boxes.length();
        int[] answer = new int[n];
        
        for (int i = 0; i < n; i++) {
            int total = 0;
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                if (boxes.charAt(j) == '1') 
{
                    total += Math.abs(i - j);
                }
            }
            answer[i] = total;
        }
        return answer;
    }
}
