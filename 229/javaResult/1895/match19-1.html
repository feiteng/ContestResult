class Solution {
    public int[] minOperations(String boxes) {
        char[] bs = boxes.toCharArray();
        int n = bs.length;
        int[] left = new int[n];
        int[] right = new int[n];
        int balls = bs[0] - '0';
        for (int i = 1; i < n; i++) {
            left[i] = balls + left[i - 1];
            balls += bs[i] - '0';
        }
        balls = bs[n - 1] - '0';
        for (int i = n - 2; i >= 0; i--) {
            right[i] = balls + right[i + 1];
            balls += bs[i] - '0';
        }
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            res[i] = left[i] + right[i];
        }
        return res;
    }
}
