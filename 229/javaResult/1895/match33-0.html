class Solution {
    public int[] minOperations(String boxes) {
        char[] ca = boxes.toCharArray();
        int[] result = new int[ca.length];
        for (int i = 0; i < ca.length; i++) {
            for (int j = 0; j < ca.length; j++) {
                if (ca[j] == '1') {
                    result[i] += Math.abs(j - i);
                }
            }
        }
        return result;
    }
}
