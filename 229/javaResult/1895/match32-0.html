class Solution {
    public int[] minOperations(String boxes)
	{
		char[] b = boxes.toCharArray();
		int[] result = new int[b.length];
		for(int x = 0; x < b.length; x++)
		{
			for(int y = 0; y < b.length; y++)
			{
				if(b[y] == '1')
				{
					result[x] += Math.abs(y - x);
				}
			}
		}
		
		return result;
	}
}
