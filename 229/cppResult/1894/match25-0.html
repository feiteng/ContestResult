class Solution {
public:
    string mergeAlternately(string s1, string s2) {
        int l=0,r=0;
        string res;
        while((l<s1.size())&&(r<s2.size()))
        {
            res+=s1[l];
            res+=s2[r];
            l++;
            r++;
        }
        while(l<s1.size())
        {
            res+=s1[l];
            l++;
        }
        while(r<s2.size())
        {
            res+=s2[r];
            r++;
        }
        return res;
    }
};
