class Solution {
public:
    
    string mergeAlternately(string word1, string word2) {
        string r;
        int i = 0, j = 0;
        while (i < word1.S && j < word2.S) {
            r += word1[i]; r += word2[j]; i++; j++;
        }
        if (i < word1.S) r += string(word1.B + i, word1.E);
        if (j < word2.S) r += string(word2.B + j, word2.E);
        return r;
    }
};
