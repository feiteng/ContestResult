class Solution {
public:
    string mergeAlternately(string a, string b) {
        int i = 0, j = 0;
        string ans;
        while(i<a.length() && j<b.length()){
            ans+=a[i++];
            ans+=b[j++];
        }
        while(i<a.length())ans+=a[i++];
        while(j<b.length())ans+=b[j++];
        return ans;
    }
};
