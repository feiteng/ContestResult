class Solution {
public:
string mergeAlternately(string word1, string word2) {
    int i1=0,i2=0,n1=word1.size(),n2=word2.size();
    string res;
    while(i1<n1&&i2<n2){
        res+=word1[i1++];
        res+=word2[i2++];
    }
    while(i1<n1) res+=word1[i1++];
    while(i2<n2) res+=word2[i2++];
    return res;
}
};
