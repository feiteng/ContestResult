class Solution {
public:
    string mergeAlternately(string word1, string word2) {
        int l=0,r=0;
        string ret;
        while(l<word1.length() && r < word2.length())
        {

            ret+=word1[l];
            ret += word2[r];
            l++,r++;
        }
        while(l<word1.length()){
            ret+=word1[l];
            l++;
        }
        while(r<word2.length()){
            ret+=word2[r];
            r++;
        }

        return ret;
    }
};
