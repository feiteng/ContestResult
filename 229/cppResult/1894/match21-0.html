class Solution {
public:
    string mergeAlternately(string word1, string word2) {
        int i1 = 0, i2 = 0;
        string merged;
        while (i1 < word1.size() and i2 < word2.size()) {
            merged += word1[i1++];
            merged += word2[i2++];
        }   
        if (i1 < word1.size()) merged += word1.substr(i1);
        if (i2 < word2.size()) merged += word2.substr(i2);
        return merged;
    }
};
