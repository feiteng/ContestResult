class Solution {
public:
    string mergeAlternately(string l, string r) {
        int left = 0, right = 0;
        string ans;
        while(left < l.size() && right < r.size()){
            ans += l[left++];
            ans += r[right++];
        }
        // cout << ans << endl;
        while(left < l.size()) ans += l[left++];
        while(right < r.size()) ans += r[right++];
        return ans;
    }
};
