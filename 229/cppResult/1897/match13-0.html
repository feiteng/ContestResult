class Solution {
public:
    
    int longestPalindrome(string w1, string w2) {
        map<int,int> mapp;
        string s=w1+w2;
        int ok=0;
        int start=w1.size();
        int end=0;
        int index=0;
        for(char c:w1){
            if(w2.find(c)!=-1){
                ok=1;
                
                mapp[index]=w2.find_last_of(c)+w1.size();
            } 
            index++;
        }
        if(ok==0)return 0;
        vector<vector<int>> dp(s.size(),vector<int>(s.size(),0));
        for(int j=0;j<s.size();j++){
            dp[j][j]=1;
            for(int i=j-1;i>=0;i--){
                if(s[i]==s[j]){
                    dp[i][j]=max(dp[i][j],2+dp[i+1][j-1]);
                }else{
                    dp[i][j]=max(dp[i+1][j],dp[i][j-1]);
                }
            }
        }
        // cout<<start<<" "<<end;
        int ans=0;
        for(auto a:mapp){
            ans=max(ans,dp[a.first][a.second]);
        }
        return ans;

    }
};
