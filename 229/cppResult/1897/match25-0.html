class Solution {
public:
    int lps(string& word, int limit) {
        int n = word.size();
        vector<vector<int> > L(n, vector<int>(n));
        for(int i = 0; i < n; i++) {
            L[i][i] = 1;
        }
        int total = 0;
        for(int cl = 2; cl <= n; cl++) {
            for(int i = 0; i < n-cl+1; i++) {
                int j = i + cl -1;
                if (word[i] == word[j] && cl == 2) {
                    L[i][j] = 2;
                    if (i < limit && j >= limit)  {
                        // cout<<j<<"|"<<limit<<endl;
                        total = max(total, L[i][j]);
                    }
                } else if (word[i] == word[j]) {
                    L[i][j] = L[i+1][j-1] + 2;
                    if (i < limit && j >= limit) {
                        // cout<<j<<"|"<<limit<<endl;
                        total = max(total, L[i][j]);
                    }
                } else {
                    L[i][j] = max(L[i][j-1], L[i+1][j]);
                }
            }
        }
        return total;
    }
    int longestPalindrome(string word1, string word2) {
        string word = word1 + word2;
        return lps(word, word1.size());
    }
};
