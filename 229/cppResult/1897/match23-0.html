class Solution {
public:
    int longestPalindrome(string s1, string s2) {
        string str = s1 + s2;
        int n = str.size();
        int a[n][n];
        memset(a, 0, sizeof(a));
        for(int i=0; i<n; i++) a[i][i] = 1;
        for(int i=n-1; i>=0; i--) {
            for(int j=i+1; j<n; j++) {
                if(str[i] == str[j])
                    a[i][j] = a[i+1][j-1] + 2;
                else
                    a[i][j] = max(a[i+1][j], a[i][j-1]);
            }
        }
        int ans = a[0][n-1], ans1 = a[0][s1.size()-1], ans2 = a[s1.size()][n-1];
        if(ans1==ans || ans2==ans) {
            ans = 0;
            for(int i=0; i<s1.size(); i++) {
                for(int j=0; j<s2.size(); j++) {
                    if(s1[i] == s2[j]) {
                        ans = max(ans, a[i][j+s1.size()]);
                    }
                }
            }
        }
        return ans;
    }
};
