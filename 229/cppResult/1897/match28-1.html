class Solution {
public:
    
    int longestPalindrome(string word1, string word2) {

        int len1 = word1.size();
        int len2 = word2.size();
        string word = word1 + word2;
        int len = word.size();
        vector<vector<int>> dp(len, vector<int>(len));
        for (int i = 0; i < len; ++i) {
            dp[i][i] = 1;
        }

        int ret = 0;
        for (int i = len - 1; i >= 0; --i) {
            for (int j = i + 1; j < len; ++j) {
                if (word[i] == word[j]) {
                    dp[i][j] = 2 + dp[i + 1][j - 1];
                    if (i < len1 && j >= len1) {
                        ret = max(ret, dp[i][j]);
                    }
                } else {
                    dp[i][j] = max(dp[i][j - 1], dp[i + 1][j]);
                }
            }
        }
        return ret;
    }
};
