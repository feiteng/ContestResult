#define sz(a) int((a).size())
using ll = long long;


class Solution {
public:
    int longestPalindrome(string word1, string word2) {
		string s = word1 + word2;
		int l1 = sz(word1);
		int n = sz(s);
		vector<vector<int>> dp(n + 1, vector<int>(n));
		for(int i = 1;i <= n;i ++) {
			for(int j = 0;j <= n - i;j ++) {
				dp[i][j] = 0;
				if(i == 1) {
					dp[i][j] = 1;

				} else if(i == 2) {
					if(s[j] == s[j + 1])
						dp[i][j] = 2;
					else
						dp[i][j] = 1;
				} else {
					if(s[j] == s[j + i - 1])
						dp[i][j] = 2 + dp[i - 2][j + 1];
					dp[i][j] = max(dp[i][j], max(dp[i - 1][j + 1], dp[i - 1][j]));
				}
			}
		}

		int ans = 0;
		for(int i = 0;i < l1;i ++) {
			for(int j = l1;j < n;j ++) {
				if(s[i] == s[j]) {
					if(abs(i - j) == 1) ans = max(ans, 2);
					else ans = max(ans, 2 + dp[j - i - 1][i + 1]);
				}
			}
		}
		return ans;
    }
};
