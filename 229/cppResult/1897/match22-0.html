class Solution {
public:
    vector<vector<int> > dp;
    int longestPalindrome(string word1, string word2) {
        string s = word1 + word2;
        dp = vector<vector<int> > (s.size(), vector<int>(s.size(), 0));
        for(int i = 0; i < s.size(); i++){
            for(int j = i; j >= 0; j--){
                if(j==i) dp[j][i] = 1;
                else{
                    if(s[i] == s[j]){
                        if(j+1==i) dp[j][i] = 2;
                        else dp[j][i] = 2 + dp[j+1][i-1];
                    } else dp[j][i] = max(dp[j+1][i], dp[j][i-1]);
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < word1.size(); i++)
            for(int j = 0; j < word2.size(); j++){
                if(word1[i] == word2[j]){
                    ans = max(ans, 2 + dp[i+1][word1.size() + j - 1]);
                }
            }
        return ans;
    }
};
