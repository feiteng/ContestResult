class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string s = word1 + word2;
        int dp[s.size()][s.size()];
        for(int i = 0; i < s.size(); ++i)
            dp[i][i] = 1;
        for(int i = 1; i < s.size(); ++i)
            dp[i][i-1] = 0;
        for(int sp = 1; sp < s.size(); ++sp) {
            for(int l = 0; l + sp < s.size(); ++l) {
                int r = l + sp;
                if(s[l] == s[r])
                    dp[l][r] = dp[l+1][r-1] + 2;
                else
                    dp[l][r] = max(dp[l+1][r],dp[l][r-1]);
            }
        }
        int m = word1.size(), n = word2.size();
        int res = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = m; j < m + n; ++j) {
                if(s[i] == s[j]) {
                    res = max(res, dp[i][j]);
                }
            }
        }
        return res;
    }
};
