class Solution {
public:
    int dp[2010][2020];
 
    int longestPalindrome(string word1, string word2) {
        string ans = word1;
        for(int i = 0; i < word2.size(); ++i) ans += word2[i];
        for(int i  = ans.size(); i >= 1; --i){
            dp[i][i] = 1;
            for(int j = i; j <= ans.size(); ++j){
                if(i == j) continue;
                if(ans[i-1] == ans[j-1]) {
                    dp[i][j] = dp[i+1][j-1]+2;
                }
                // else if(j > 1 && ans[i-1] == ans[j-2]){
                    dp[i][j] = max(dp[i][j], dp[i][j-1]);
                // }
                // else if(i < ans.size()&& ans[i] == ans[j-1]){
                    dp[i][j] = max(dp[i][j], dp[i+1][j]);
                // }
            }
        }
        int s = 0;
        for(int i = 1; i <= word1.size(); ++i){
            for(int j = word1.size()+1; j <= ans.size(); ++j){
               if(ans[i-1] == ans[j-1])
                    s = max(s, dp[i][j]);
            }
        }
        return s;
    }
};//aabb
