class Solution {
public:
    
    int longestPalindrome(string a, string b) {
        
        int m = a.length();
        int n = a.length() + b.length();
        string s = a+b;
        
        int dp[n][n];
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++)dp[i][i]=1;
        int ans = 0;
        for(int i=0;i+1<n;i++){
            int j = i+1;
            if(s[i]==s[j]){
                dp[i][j]=2;
                if(i<m && j>=m)ans=max(ans,dp[i][j]);
            }
            else dp[i][j]=1;
        }
        for(int l=2;l<n;l++){
            for(int i=0,j=l;j<n;i++,j++){
                if(s[i]==s[j]){
                    dp[i][j]=2+dp[i+1][j-1];
                    if(i<m && j>=m)ans=max(ans,dp[i][j]);
                }
                else dp[i][j]=max(dp[i+1][j],dp[i][j-1]);
            }
        }
        
        return ans;
    }
};
