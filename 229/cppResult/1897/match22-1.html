class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string word = word1 + word2;

        int n = word.size();
        vector<vector<int>> dp(2005, vector<int>(2005));
        for (int i = n-1; i >= 0; --i) {
            for (int j = i; j < n; ++j) {
                if (i == j) dp[i][j] = 1;
                else if (i+1 == j) dp[i][j] = word[i] == word[j] ? 2 : 1;
                else dp[i][j] = max(dp[i+1][j-1] + 2*(word[i] == word[j]), max(dp[i+1][j], dp[i][j-1]));

            }
        }
        int ans = 0;
        for (int i = 0; i < word1.size(); ++i)
            for (int j = 0; j < word2.size(); ++j) {
                if (word1[i] == word2[j]) {
                    ans = max(ans, dp[i+1][word1.size()+j-1]+2);
                }
            }
        return ans;
    }
};
