class Solution {
public:
    int longestPalindrome(string s1, string s2) {
        string s3 = s1 + s2;
        
        int n = s3.size();
        if(!n)
            return 0;

        vector<vector<int>> dp(n, vector<int>(n, 0));

        for(int i = 0; i < n; i++)
            dp[i][i] = 1;

        for(int i = n - 2; i >= 0; i--) {
            for(int j = i + 1; j < n; j++) {
                if(s3[i] == s3[j]) {
                    dp[i][j] = 2 + dp[i + 1][j - 1];
                } else {
                    dp[i][j] = max(dp[i][j - 1], dp[i + 1][j]);
                }
            }
        }

        int i = 0;
        int j = 0;
        n = s1.size();
        int m = s2.size();

        int ans = 0;
        
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(s1[i] == s2[j]) {
                    ans = max(ans, 2 + dp[i + 1][n + j - 1]);
                }
            }
        }
        
        return ans;
    }
};
