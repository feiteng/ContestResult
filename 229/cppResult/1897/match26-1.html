class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string s=word1+word2;
        int n=s.size();
        int x=word1.size();
        vector<vector<int>> dp(n,vector<int>(n,1));

        int ans=0;
        for (int cl=2; cl<=n; cl++) 
        { 
            for (int i=0; i<n-cl+1; i++) 
            { 
                int j = i+cl-1; 
                if (s[i] == s[j] && cl == 2) {
                    dp[i][j] = 2; 
                    if(i<x&&j>=x){
                        ans=max(ans,dp[i][j]);
                    }
                }
                else if (s[i] == s[j]) {
                    dp[i][j] = dp[i+1][j-1] + 2;
                    if(i<x&&j>=x){
                        ans=max(ans,dp[i][j]);
                    }
                }
                else{
                   dp[i][j] = max(dp[i][j-1], dp[i+1][j]); 
                }
            } 
        }
        
        
        return ans;
    }
};
