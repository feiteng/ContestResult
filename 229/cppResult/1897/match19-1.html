class Solution {
public:
    int longestPalindrome(string w1, string w2) {
        string s = w1 + w2;
        int n = s.length();
        vector<vector<int>> dp(n+1, vector<int>(n+1));
        for (int i = 0; i < n; ++i) {
            dp[i][i+1] = 1;
        }
        
        for (int len = 2; len <= n; ++len) {
            for (int i = 0; i + len <= n; ++i) {
                dp[i][i+len] = dp[i+1][i+len-1];
                if (s[i] == s[i+len-1]) {
                    dp[i][i+len] += 2;
                }

                dp[i][i+len] = max(
                    dp[i][i+len],
                    max(dp[i+1][i+len], dp[i][i+len-1])
                );
                //cout << "dp " << i << " " << i+len << ": " << dp[i][i+len] << endl;
            }
        }
        
        int best = 0;
        for (int i = 0; i < w1.length(); ++i) {
            for (int j = 0; j < w2.length(); ++j) {
                if (w1[i] == w2[j]) {
                    best = max(best, dp[i][w1.length()+j+1]);
                }
            }
        }
        return best;
    }
};
