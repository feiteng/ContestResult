class Solution {
public:
    string str;
    int lim;
    int dp[2048 * 2048 + 1];
    
    int search(int s, int t) {
        int pos = s * 2048 + t;
        if(dp[pos] != -1) return dp[pos];
        int& ret = dp[pos];
        if(s == t) return ret = 1;
        if(s > t) return ret = 0;
        
        if(str[s] == str[t]) {
            return ret = 2 + search(s+1, t-1);
        }
        return  ret  = max(search(s+1, t) , search(s, t-1));

    }
    int longestPalindrome(string word1, string word2) {
        str = word1 + word2;
        lim = word1.size();
        memset(dp, -1, sizeof(dp));

        search(0, str.size() - 1);
        int ans = 0;
        for(int i = 0; i < word1.size(); i++) {
            for(int j = 0; j < word2.size(); j++) {
                if(word1[i] != word2[j]) {
                    continue;
                }
                int idx = i * 2048 + j + lim;
                ans = max<int>(dp[idx], ans);
            }
        }
        
        return ans;
    }
};
