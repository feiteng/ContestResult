class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        int n = word1.size();
        int m = word2.size();
        string w = word1 + word2;
        
        vector<vector<int>> f(n + m + 1, vector<int>(n + m + 1));
        for (int i = 0; i < n + m; ++i) {
            f[i][i] = 1;
        }
        for (int len = 2; len <= n + m; ++len) {
            for (int i = 0; i + len <= n + m; ++i) {
                int j = i + len - 1;
                f[i][j] = max(f[i + 1][j], f[i][j - 1]);
                if (w[i] == w[j]) {
                    f[i][j] = max(f[i][j], f[i + 1][j - 1] + 2);
                }
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (word1[i] == word2[j]) {
                    ans = max(ans, 2 + f[i + 1][n + j - 1]);
                }
            }
        }
        return ans;
    }
};
