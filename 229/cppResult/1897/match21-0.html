class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string str=word1+word2;
        int n=str.length();
        int m=word1.length();
        vector<vector<int> > dp(n,vector<int>(n));
        int ans=0;
        for(int i=n-1;i>=0;i--){
            dp[i][i]=1;
            for(int j=i+1;j<n;j++){
                if(str[i]==str[j]){
                    dp[i][j]=dp[i+1][j-1]+2;
                    if(j>=m&&i<m)
                        ans=max(ans,dp[i][j]);
                }
                else
                    dp[i][j]=max(dp[i+1][j],dp[i][j-1]);
            }
        }
        return ans;
        return dp[0][n-1];

    }
};
