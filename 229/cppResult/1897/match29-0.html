class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string w = word1 + word2;
        int W1 = word1.size();
        int W = w.size();
        vector<vector<int>> dp(W, vector<int>(W));
        // cout << "here" << endl;
        int ans = 0;
        for (int i=0; i<W; i++) dp[i][i] = 1;
        for (int l=2; l<=W; l++) {
            for (int i=0; i+l<=W; i++) {
                int j = i + l - 1;
                // cout << i << " " << j << endl;
                if (w[i] == w[j]) {
                    dp[i][j] = i+1 > j-1 ? 2 : 2 + dp[i+1][j-1]; 
                    if (i<W1 and j>=W1) ans = max(ans, dp[i][j]);
                } else {
                    dp[i][j] = max(dp[i][j-1], dp[i+1][j]);
                }
            }
        }
        // for (auto r : dp) {
        //     for (auto c : r) cout << c << " ";
        //     cout << endl;
        // }
        return ans;
    }
};
