class Solution {
public:
    string str;
    int lim;
    int dp[2048 * 2048 * 2 + 1];
    
    int search(int s, int t) {
        
        int pos = s * 2048 + t;
        if(dp[pos] != -1) return dp[pos];
        int& ret = dp[pos];
        if(s > t) return ret = 0;
        if(s == t) return ret = 1;
        if(str[s] == str[t]) {
           // cout<<s<<" "<<t<<" "<<ret<<endl;
            return ret = 2 + search(s+1, t-1);
        }
         ret = max(search(s+1, t) , search(s, t-1));
       // cout<<s<<" "<<t<<" "<<ret<<endl;
        return ret;
    }
    int longestPalindrome(string word1, string word2) {
        str = word1 + word2;
        lim = word1.size();
        memset(dp, -1, sizeof(dp));

        search(0, str.size() - 1);
        int ans = 0;
        for(int i = 0; i < word1.size(); i++) {
            for(int j = 0; j < word2.size(); j++) {
                if(word1[i] == word2[j]) {
                //    cout<<i<<" "<<j<<" "<< dp[i * 2048 + j]<<endl;
                    ans = max<int>(dp[i * 2048 + j + word1.size()], ans);
                }
            }
        }
        
        return ans;
    }
};
