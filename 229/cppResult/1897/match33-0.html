class Solution {
public:
    
    int longestPalindrome(string a, string b) {
        string c = a + b;
        int n = c.size();
        int split = a.size();
    
        
        vector<vector<int>> f(n+2, vector<int>(n+2,1));
        int ans = 0;
        for(int len = 2; len <= n; len++){
            for(int l = 0; l + len - 1 < n; l++){
                int r = l + len - 1;
                if(len == 2){
                    if(c[l] == c[r]) f[l][r] = 2;
                    else f[l][r] = 1;
                }else{
                    if(c[l] == c[r]) f[l][r] = 2 + f[l+1][r-1];
                    f[l][r] = max(f[l][r], max(f[l+1][r], f[l][r-1]));
                }
                if(l < split && r >= split && c[l] == c[r]){
                    ans = max(ans, f[l][r]);
                } 
            }
        }
        return ans;

    }
};
