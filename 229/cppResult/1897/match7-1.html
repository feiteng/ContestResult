const int N = 1e3 + 10;

int dp1[N][N], dp2[N][N], dp[N][N];

class Solution {
public:

    int longestPalindrome(string str1, string str2) {
      int n = str1.size(), m = str2.size();
      reverse(str2.begin(), str2.end());
      str1 = " " + str1, str2 = " " + str2;
      for (int i = n; i >= 1; i--) {
        dp1[i][i] = 1;
        for (int j = i + 1; j <= n; j++) {
          if (str1[i] == str1[j]) {
            dp1[i][j] = dp1[i + 1][j - 1] + 2;
          }
          else {
            dp1[i][j] = max(dp1[i + 1][j], dp1[i][j - 1]);
          }
        }
      }
      for (int i = m; i >= 1; i--) {
        dp2[i][i] = 1;
        for (int j = i + 1; j <= m; j++) {
          if (str2[i] == str2[j]) {
            dp2[i][j] = dp2[i + 1][j - 1] + 2;
          }
          else {
            dp2[i][j] = max(dp2[i + 1][j], dp2[i][j - 1]);
          }
        }
      }
      for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
          if (str1[i] == str2[j]) {
            dp[i][j] = dp[i - 1][j - 1] + 1;
          }
          else {
            dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
          }
        }
      }
      int ans = 0;
      for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
          if (dp[i][j]) {
            ans = max(ans, dp[i][j] * 2 + max(dp1[i + 1][n], dp2[j + 1][m]));
          }
          // cout << dp[i][j] << " \n"[j == m];
        }
      }
      return ans;
    }
};
