class Solution {
public:
    int f[1005][1005], s[1005][1005], dp[1005][1005];
    int longestPalindrome(string word1, string word2) {
        int n = word1.size(), m = word2.size();
        reverse(word2.begin(), word2.end());
        string a = ' ' + word1, b = ' ' + word2;
        for (int i = n; i >= 1; i--) {
            f[i][i] = 1;
            for (int j = i + 1; j <= n; j++) {
                if (a[i] == a[j]) {
                    f[i][j] = f[i + 1][j - 1] + 2;
                } else {
                    f[i][j] = max(f[i + 1][j], f[i][j - 1]);
                }
            }
        }
        for (int i = m; i >= 1; i--) {
            s[i][i] = 1;
            for (int j = i + 1; j <= m; j++) {
                if (b[i] == b[j]) {
                    s[i][j] = s[i + 1][j - 1] + 2;
                } else {
                    s[i][j] = max(s[i + 1][j], s[i][j - 1]);
                }
            }
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (a[i] == b[j]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        int ans = -1e9;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (dp[i][j] > 0) {
                   ans = max(ans, dp[i][j] * 2 + max(f[i + 1][n], s[j + 1][m]));        
                }
            }
        }
        if (ans == -1e9) {
            ans = 0;
        }
        return ans;
    }
};
