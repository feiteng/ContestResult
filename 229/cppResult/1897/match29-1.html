class Solution {
public:
    int longestPalindrome(string X, string Y) {
        const int m = X.size(), n = Y.size();

        string s = X+Y;
        vector<vector<int>> L(m+n+1, vector<int>(n+m));
        int ans = 0;
        for (int i = 0; i < s.size(); ++i) L[1][i] = 1;
        for (int l = 2; l <= m+n; ++l) {
            for (int i = 0; i+l-1 < s.size(); ++i) {
                const int j = i+l-1;
                if (s[i] == s[j]) {
                    L[l][i] = 2 + L[l-2][i+1];

                    if (i < m && j >= m) ans = max(ans, L[l][i]);
                } else {
                    L[l][i] = max(L[l-1][i+1], L[l-1][i]);
                }
            }
        }
        return ans;
    }
    
};
