class Solution {
public:
    int longestPalindrome(string s1, string s2) {
        int res=0;

        int l1=s1.size();
        int l2=s2.size();
        int n=l1+l2;
        string s=s1+s2;
        vector<vector<int>> f(n,vector<int>(n,0));
        for(int i=n-1;i>=0;i--){
            f[i][i]=1;
            for(int j=i+1;j<n;j++){
                if(s[i]==s[j]){
                    f[i][j]=f[i + 1][j - 1] + 2;
                    if(i<l1&&j>=l1){
                        res=max(res,f[i][j]);
                    }
                }else{
                    f[i][j] = max(f[i + 1][j], f[i][j - 1]);
                }
            }
        }
        return res;
    }
};
