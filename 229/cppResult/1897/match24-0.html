class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string ws = word1 + word2;
        return longestPalindromeSubseq(ws, word1.size());
    }
    
    int longestPalindromeSubseq(string s, int mid) {
      int n = s.size(), res = 0;
      // dp 数组全部初始化为 0
      vector<vector<int>> dp(n, vector<int>(n, 0));
      // base case
      for (int i = 0; i < n; i++)
        dp[i][i] = 1;
      // 反着遍历保证正确的状态转移
      for (int i = n - 1; i >= 0; i--) {
        for (int j = i + 1; j < n; j++) {
            // 状态转移方程
            if (s[i] == s[j]){
                dp[i][j] = dp[i + 1][j - 1] + 2;
                //非空要求
                if(j >= mid && i < mid) res = max(res, dp[i][j]);
            }
            else
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
        }
       }
       // 整个 s 的最长回文子串长度
       return res;
    }

};
