class Solution {
public:
        
    int m1, m2;
    int maxLength(string s) {
        int n = s.size();
        int dp[n][n];
        int ans = 0;
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < n; i++) {
            dp[i][i] = 1;
        }
        for(int l = 2; l <= n; l++) {
            for (int i = 0; i <= n - l; i++) {
                int j = i + l - 1;
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
                if (s[i] == s[j] && dp[i][j] <= dp[i + 1][j - 1] + 2) {
                    dp[i][j] = dp[i + 1][j - 1] + 2;
                    if (j >= m1 && i < m1) {
                        ans = max(ans, dp[i][j]);   
                    }
                }
            }
        }
        return ans;
    }
    
    int longestPalindrome(string word1, string word2) {
        string s = word1 + word2;
        m1 = word1.size(), m2 = word2.size();
        return maxLength(s);
    }
};
