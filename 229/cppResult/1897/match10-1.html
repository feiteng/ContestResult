class Solution {
public:
    int longestPalindrome(string word1, string word2) {

        string word = word1 + word2;
        int res = 0;
        
        vector<vector<int>> record(word.size() + 1, vector<int>(word.size(), 0));
        
        for(int i = word.size() - 1; i >= 0; --i) {
            record[i][i] = 1;
            for(int j = i + 1; j < word.size(); ++j) {
                if(word[i] == word[j]) record[i][j] = record[i + 1][j - 1] + 2;
                else record[i][j] = max(record[i + 1][j], record[i][j - 1]);
            }
        }
        

        for(int i = 0; i < word1.size(); ++i) {
            for(int j = word1.size(); j < word.size(); ++j) {
                if(word[i] == word[j]) res = max(res, record[i][j]);
            }
        }
        
       return res;
    }
};
