class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        int n1 = word1.size();
        int n2 = word2.size();
        string ss = word1 + word2;

        int res = 0;
        vector<vector<int>> dp(n1+n2,vector<int>(n1+n2,0));
        for(int i=0;i<n1+n2;i++)
            dp[i][i] = 1;
        for(int l=2;l<=n1+n2;l++){
            for(int i=0;i<n1+n2-l+1;i++){
                int j = i+l-1;

                if(ss[i]==ss[j]){
                    dp[i][j] = dp[i+1][j-1] + 2;
                    if(i<n1&&j>=n1)
                        res = max(res,dp[i][j]);
                }
                else
                    dp[i][j] = max(dp[i+1][j],dp[i][j-1]);
            }
        }
        return res;
    }
};
