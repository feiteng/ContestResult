class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        string word = word1 + word2;
        int m = word1.length(), n = word2.length();
        int l = word.length();
        vector<vector<int>> dp(l, vector<int>(l));
        for (int i = 0; i < l; i++) {
            dp[i][i] = 1;
            for (int j = i - 1; j >= 0; j--) {
                if (word[j] == word[i]) dp[j][i] = 2 + dp[j + 1][i - 1];
                else dp[j][i] = max(dp[j + 1][i], dp[j][i - 1]);
            }
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = m; j < m + n; j++) {
                if (word[i] == word[j]) res = max(res, 2 + dp[i + 1][j - 1]);
            }
        }
        return res;
    }
};
