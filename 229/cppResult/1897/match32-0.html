class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        int n=word1.size();
        int m = word2.size();
        vector<vector<int>>lps(m+n,vector<int>(m+n,0));
        string conc = word1+word2;
    
        for(int i = m+n-1;i>=0;i--){
            for(int j=i;j<m+n;j++){
                if(j-i+1 == 1)lps[i][j] = 1;
                else if(j-i+1 == 2){
                    if(conc[i] == conc[j])lps[i][j] = 2;
                    else lps[i][j] = 1;
                }
                else{
                    if(conc[i] == conc[j]){
                        lps[i][j] = 2+lps[i+1][j-1];
                    }
                    else {
                        lps[i][j] = max(lps[i+1][j],lps[i][j-1]);
                    }
                }
            }
        }
        // for(int i=0;i<m+n;i++){
        //     for(int j = 0;j<m+n;j++)cout<<lps[i][j]<<" ";
        //     cout<<endl;
        // }
        int ans = 0;
        for(int i=0;i<n;i++){
            for(int j=m+n-1;j>=n;j--){
                if(conc[i] == conc[j]){
                    ans = max(ans,2+lps[i+1][j-1]);
                }
            }
        }
        return ans;
    }
};
