class Solution {
public:
    int longestPalindrome(string a, string b) {
        int n=a.size(),m=b.size();
        reverse(b.begin(),b.end());
        a="#"+a;
        b='#'+b;

        vector<vector<int>>dp(n+2,vector<int>(m+2,0)),x(n+2,vector<int>(n+2,0)),y(m+2,vector<int>(m+2,0));
        for(int i=1;i<=n;i++) x[i][i]=1;
        for(int i=1;i<=m;i++) y[i][i]=1;
        for(int l=2;l<=n;l++) {
            for(int i=1;i+l-1<=n;i++) {
                int j=i+l-1;
                if(a[i]==a[j]) x[i][j]=x[i+1][j-1]+2;
                else x[i][j]=max(x[i+1][j],x[i][j-1]);
            }
        }
        for(int l=2;l<=m;l++) {
            for(int i=1;i+l-1<=m;i++) {
                int j=i+l-1;
                if(b[i]==b[j]) y[i][j]=y[i+1][j-1]+2;
                else y[i][j]=max(y[i+1][j],y[i][j-1]);
            }
        }

        int ans=0;
        for(int i=1;i<=n;i++) {
            for(int j=1;j<=m;j++) {
                if(a[i]==b[j]) {
                    dp[i][j]=dp[i-1][j-1]+1;
                }
                dp[i][j]=max(dp[i][j],dp[i-1][j]);
                dp[i][j]=max(dp[i][j],dp[i][j-1]);
                if(dp[i][j])ans=max(ans,dp[i][j]*2+max(x[i+1][n],y[j+1][m]));
                
            }
        }
        return ans;
    }
};
