class Solution {
public:
    int longestPalindrome(string word1, string word2) {
        int size1 = word1.size(), size2 = word2.size();
        string s = word1 + word2;

        int n = s.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        int ret = 0;
        for (int i=n-1;i>=0;--i) {
            dp[i][i] = 1;
            for (int j=i+1;j<n;++j) {
                if (s[i] == s[j]) {
                    dp[i][j] = dp[i+1][j-1] + 2;
                    if (i < size1 && j >= size1) {
                        ret = max(ret, dp[i][j]);
                    }
                } else {
                    dp[i][j] = max(dp[i+1][j], dp[i][j-1]);
                }
            }
        }
        return ret;
    }
};
