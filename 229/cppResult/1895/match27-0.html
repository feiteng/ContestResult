class Solution {
public:
    vector<int> minOperations(string boxes) {
        int n = boxes.size();
        vector<int> ans(n,0);
        for (int i = 0; i < n; i++) {
            int tmp = 0;
            for (int l = 0; l < i; l++) {
                if (boxes[l] == '1') {
                    tmp += i - l;
                }
            }
            for (int r = i+1; r < n; r++) {
                if (boxes[r] == '1') {
                    tmp += r - i;
                }
            }
            ans[i] = tmp;
        }
        return ans;
    }
};
