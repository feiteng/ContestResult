class Solution {
public:
    vector<int> minOperations(string boxes) {
      //  boxes = string(2000, '1');
        vector<int> ret(boxes.size(), 0);
        for (int i = 0; i < boxes.size(); ++i) {
            for (int j = 0; j < boxes.size(); ++j) {
                if (boxes[j] == '1') ret[i] += abs(j - i); 
            }
        }
        return ret;
    }
};
