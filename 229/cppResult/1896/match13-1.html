class Solution {
public:
    int maximumScore(vector<int>& v,vector<int>& p){
        int n1=v.size();
        int n2=p.size();
        vector<vector<int>> g(n2+1,vector<int> (n2+1,0));

        for(int i=n1-n2+1;i<=n1;i++)
            for(int j=0;j+i<=n1;j++){
                g[j][j+i-n1+n2]=max(g[j+1][j+i-n1+n2]+v[j]*p[n1-i],g[j][j+i-1-n1+n2]+v[j+i-1]*p[n1-i]);
            }
        return g[0][n2];
    }
};
