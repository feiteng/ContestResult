class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int m = mul.size(), n = nums.size();
        
        vector<vector<int>> dp(m + 2, vector<int>(m + 2, 0));
        
        dp[0][0] = 0;
        for (int i = 1; i <= m; i++) {
            dp[i][0] = dp[i - 1][0] + mul[i - 1] * nums[i - 1];
            dp[0][i] = dp[0][i - 1] + mul[i - 1] * nums[n - i];
        }
        

        for (int l = 1; l <= m; l++) {
            for (int r = 1; l + r <= m ; r++) {
                int i = l + r - 1;
                dp[l][r] = max(dp[l - 1][r] + mul[i] * nums[l - 1], dp[l][r - 1] + mul[i] * nums[n - r]);

            }
        }
        
        int ret = dp[0][m];
        for (int i = 0; i <= m; i++) {
            ret = max(ret, dp[i][m - i]);
        }
        
        return ret;
    }
};
