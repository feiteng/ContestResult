#define INF 0x3f3f3f3f


class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m + 1, vector<int>(m + 1, -INF));
        dp[0][0] = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j <= i; ++j) {

                dp[j + 1][i - j] = max(dp[j + 1][i - j], dp[j][i - j] + nums[j] * multipliers[i]);
                dp[j][i - j + 1] = max(dp[j][i - j + 1], dp[j][i - j] + nums[n - 1 - (i - j)] * multipliers[i]);
            }
        }
        int ans = INT_MIN;
        for (int i = 0; i <= m; ++i)
            ans = max(ans, dp[i][m - i]);
        return ans;
    }
};
