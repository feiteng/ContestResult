class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int m = multipliers.size(), n = nums.size();
        vector<vector<int>> dp(m + 1, vector<int>(m + 1, -1e9));
        dp[0][0] = 0;
        for(int i = 1; i <= m; i++)
        for(int l = 0; l <= i; l++){
            int r = n - (i - l) - 1;
//            cout<<i<<'\t'<<l<<'\t'<<r<<endl;
            if(l > 0) dp[i][l] = max(dp[i][l], dp[i - 1][l - 1] + nums[l - 1] * multipliers[i - 1]);
            if(r + 1 < n) dp[i][l] = max(dp[i][l], dp[i - 1][l] + nums[r + 1] * multipliers[i - 1]);
//            cout<<i<<'\t'<<l<<'\t'<<r<<'\t'<<dp[i][l]<<endl;
        }
        int best = INT_MIN;
        for(int i = 0; i <= m; i++) best = max(best, dp[m][i]);
        return best;
    }
};
