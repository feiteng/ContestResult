int dp[1001][1001];
class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n=nums.size(),m=multipliers.size();
        for(int i=1;i<=m;i++)
        {
            for(int j=0;j<=i;j++)
            {
                if(j==0)dp[i][j]=dp[i-1][j]+nums[n-i+j]*multipliers[i-1];
                else if(j==i)dp[i][j]=dp[i-1][j-1]+nums[j-1]*multipliers[i-1];
                else dp[i][j]=max(dp[i-1][j]+nums[n-i+j]*multipliers[i-1],dp[i-1][j-1]+nums[j-1]*multipliers[i-1]);
            }
        }
        int ans=-1000000007;
        for(int i=0;i<=m;i++)
        {
            ans=max(ans,dp[m][i]);
        }
        return ans;
    }
};
