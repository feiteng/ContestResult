class Solution {
    int dfs(int pos1, int pos2, vector<int>& nums, vector<int>& multipliers, vector<vector<int>>& dp) {
        if (pos1 == multipliers.size())
            return 0;
        if (dp[pos1][pos2] != INT_MIN)
            return dp[pos1][pos2];
        int n = nums.size();
    
        return dp[pos1][pos2] = max(dfs(pos1+1, pos2, nums, multipliers, dp) + multipliers[pos1]*nums[pos2+n-pos1-1], 
                                    dfs(pos1+1, pos2+1, nums, multipliers, dp) + multipliers[pos1]*nums[pos2]);
    }
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m, vector<int>(m + 1, INT_MIN));
        return dfs(0, 0, nums, multipliers, dp);
    }
};
