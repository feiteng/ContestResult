class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int m = multipliers.size();
        int n = nums.size();
        vector<vector<int>> dp(m + 1, vector<int>(m + 1, -1e9));
        dp[0][0] = 0;

        for (int i = 1; i <= m; i++) {
            dp[i][0] = dp[i - 1][0] + nums[i - 1] * multipliers[i - 1];
            dp[0][i] = dp[0][i - 1] + nums[n - i] * multipliers[i - 1];
        }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; i + j <= m; j++) {
                dp[i][j] = max(dp[i - 1][j] + nums[i - 1] * multipliers[i + j - 1], dp[i][j - 1] + nums[n - j] * multipliers[i + j - 1]);
            }
        }
        /*for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= m; j++) {
                cout << dp[i][j] << " ";
            }
            cout << endl;
        }*/
        int ans = -1e9;
        for (int i = 0; i <= m; i++) {
            ans = max(ans, dp[i][m - i]);
        }
        return ans;
    }
};
