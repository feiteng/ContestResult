class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int n = nums.size(), m = mul.size();
        vector<vector<int>> dp(m+1, vector<int>(m+1, 0));
        dp[1][0] = nums[0] * mul[0];
        dp[0][1] = nums[n-1] * mul[0];
        for(int cur = 2; cur <= m; ++cur){
                cout << cur << endl;
            dp[0][cur] = dp[0][cur-1] + nums[n-cur] * mul[cur-1];
            dp[cur][0] = dp[cur-1][0] + nums[cur-1] * mul[cur-1];
            for(int i = 1; i < cur; ++i){
                //cout << i << " " << cur << endl;
                dp[i][cur-i] = max(dp[i-1][cur-i] + mul[cur-1]*nums[i-1], dp[i][cur-i-1] + mul[cur-1]*nums[n-cur+i]);
            }
        }
        int ans = INT_MIN;
        for(int i = 0; i <= m; ++i) ans = max(ans, dp[i][m-i]);
        return ans;
    }
};
