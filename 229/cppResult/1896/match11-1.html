class Solution {

public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n=nums.size(),m=multipliers.size();
        vector<vector<int>> dp(m,vector<int>(m,INT_MIN));
        return fff(dp,0,n-1,nums,multipliers,0);
    }

    int fff(vector<vector<int>>& dp,int l,int r,vector<int>& nums, vector<int>& multipliers,int cnt){
        if(cnt>=multipliers.size())
            return 0;
        if(dp[l][nums.size()-1-r]!=INT_MIN)
            return dp[l][nums.size()-1-r];
        return dp[l][nums.size()-1-r]=max(fff(dp,l+1,r,nums,multipliers,cnt+1)+nums[l]*multipliers[cnt],fff(dp,l,r-1,nums,multipliers,cnt+1)+nums[r]*multipliers[cnt]);
    }
};
