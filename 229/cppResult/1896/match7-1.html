class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> f(2, vector<int>(m + 1, 0));
        for (int i = 1; i <= m; ++i)
        {
            f[i % 2][0] = f[1 - (i % 2)][0] + multipliers[i - 1] * nums[n - i];
            f[i % 2][i] = f[1 - (i % 2)][i - 1] + multipliers[i - 1] * nums[i - 1];

            for (int j = 1; j < i; ++j)
            {
            
                f[i % 2][j] = max(f[1 - (i % 2)][j] + multipliers[i - 1] * nums[n - (i - j)], f[1 - (i % 2)][j - 1] + multipliers[i - 1] * nums[j - 1]);
            }
        }
        int ans = -1e9;
        for (int i = 0; i <= m; ++i)
        {
            ans = max(ans, f[m % 2][i]);    
        }
        return ans;
    }
};
