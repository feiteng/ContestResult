class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int N = nums.size(), M = multipliers.size();
        vector<vector<int>> dp(M+1, vector<int>(M+1));

        for (int t = 1; t <= M; t++) {
            for (int i = 0; i <= t; i++) {
                //cout << t << " " << i << " " << N-(t-i) << endl;
                if (N-(t-i) < N) {
                    dp[t][i] = dp[t-1][i] + nums[N-(t-i)] * multipliers[t-1];
                    if (i > 0) dp[t][i] = max(dp[t][i], dp[t-1][i-1] + nums[i-1] * multipliers[t-1]);
                }
                if (i > 0) {
                    dp[t][i] = dp[t-1][i-1] + nums[i-1] * multipliers[t-1]; 

                    if (N-(t-i) < N) dp[t][i] = max(dp[t][i], dp[t-1][i] + nums[N-(t-i)] * multipliers[t-1]);
                }
                
                
            }
        }
        int ans = INT_MIN;
        for (int i = 0; i <= M; i++) {
            ans = max(ans, dp[M][i]);
        }
        return ans;
    }
};
