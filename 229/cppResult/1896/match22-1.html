class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size();
        int m = multipliers.size();
        int ans = INT_MIN;
        vector<vector<int>> dp(m+1,vector<int>(m+1,0));
        
        for(int i = 1; i <= m; ++i){
            dp[i][0] = dp[i-1][0] + nums[n-i]*multipliers[i-1];

            dp[i][i] = dp[i-1][i-1] + nums[i-1]*multipliers[i-1];
        }
        for(int i = 1; i <= m; ++i){
            for(int j = 1; j < i; ++j){
                dp[i][j] = max(dp[i-1][j] + nums[n-(i-j)]*multipliers[i-1],dp[i-1][j-1] + nums[j-1]*multipliers[i-1]);
            }
        }
        for(int i = 0; i <= m; ++i){
            ans = max(ans,dp[m][i]);
        }
            
        return ans;
    }
};
