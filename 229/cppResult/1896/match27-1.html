class Solution {
public:
    int maximumScore(vector<int>& a, vector<int>& g) {

        const int inf = 1e9 + 10;
        int n = g.size();
        vector<vector<int>> f(n + 1, vector<int>(n + 1, -inf));
        f[0][0] = 0;
        for(int L = 0;  L < n; L++) {

            for(int i = 0; i <= L; i++) {
                int j = L - i;
                f[i+1][j] = max(f[i+1][j],f[i][j] + a[i] * g[L]);
                f[i][j+1] = max(f[i][j+1] , f[i][j] + a[a.size() - 1 - j] * g[L]);
            }
        }
        int ret = -inf;
        for(int i = 0; i <= n; i++) {
            ret = max(ret, f[i][n-i]);
        }
        return ret;
    }
};
