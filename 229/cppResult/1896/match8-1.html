const int maxn=1e3+5;

int f[maxn][maxn];
class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& g) {
        int n=nums.size(),m=g.size();
        for(int i=1;i<=m;i++){
            for(int j=0;j<=i;j++){
                if(j==0) f[i][j]=f[i-1][0]+g[i-1]*nums[n-i];
                else if(j==i) f[i][j]=f[i-1][j-1]+g[i-1]*nums[j-1];
                else f[i][j]=max(f[i-1][j]+g[i-1]*nums[n-i+j],f[i-1][j-1]+g[i-1]*nums[j-1]);
                // cout<<f[i][j]<<" ";
            }
            // cout<<endl;
        }
        int ret=-0x3f3f3f3f;
        for(int i=0;i<=m;i++){
            ret=max(ret,f[m][i]);
        }
        return ret;
    }
};
