class Solution {
    //static const int INF = 2e9;
    int getans(vector<int>& nums, vector<int>& multipliers, int l, int r, int ind, vector<vector<int>>& dp) {
        if (ind == multipliers.size()) return 0;
        if (dp[l][r] != 2e9) return dp[l][r];
        return dp[l][r] = max(nums[l] * multipliers[ind] + getans(nums, multipliers, l + 1, r, ind + 1, dp), nums[nums.size() - 1 - r] * multipliers[ind] + getans(nums, multipliers, l, r + 1, ind + 1, dp));
    }
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m + 5, vector<int>(m + 5, 2e9));
        int l = 0, r = 0;
        return getans(nums, multipliers, l, r, 0, dp);
    }
};
