class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        int dp[m + 1][m + 1];
        memset(dp, 0, sizeof(dp));
        for (int i = 1; i <= m; i++) {
            for (int j = 0; j <= i; j++) {
                dp[i][j] = INT_MIN;
                if (j > 0)

                    dp[i][j] = max(dp[i][j], multipliers[i - 1] * nums[j - 1] + dp[i - 1][j - 1]);
                if (j < i)
                    dp[i][j] = max(dp[i][j], multipliers[i - 1] * nums[n - (i - j)] + dp[i - 1][j]);
            }
        }
        int res = INT_MIN;
        for (int i = 0; i <= m; i++)
            res = max(res, dp[m][i]);
        return res;
    }
};
