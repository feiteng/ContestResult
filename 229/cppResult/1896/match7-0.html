class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n=nums.size(),m=multipliers.size();
        vector<vector<int>>v(m+1,vector<int>(m+1));
        for(int i=1;i<=m;++i){
            v[i][0]=v[i-1][0]+multipliers[i-1]*nums[i-1];
            v[0][i]=v[0][i-1]+multipliers[i-1]*nums[n-i];
        }
        for(int i=2;i<=m;++i){
            for(int j=1;j<i;++j){
                v[j][i-j]=max(nums[j-1]*multipliers[i-1]+v[j-1][i-j],v[j][i-j-1]+nums[n-(i-j)]*multipliers[i-1]);
            }
        }
        int ans=INT_MIN;
        for(int i=0;i<=m;++i){
            ans=max(ans,v[i][m-i]);
        }
        return ans;
    }
};
