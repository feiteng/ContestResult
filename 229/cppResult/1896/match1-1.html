class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> f(m + 1, vector<int>(m + 1, INT_MIN));

        f[0][0] = 0;
        for (int i = 0; i < m; ++i) {
            for (int left = 0; left <= i + 1; ++left) {
                int right = i + 1 - left;
                if (left > 0) {
                    f[left][right] = max(f[left][right], f[left - 1][right] + nums[left - 1] * multipliers[i]);
                }
                if (right > 0) {
                    f[left][right] = max(f[left][right], f[left][right - 1] + nums[n - right] * multipliers[i]);
                }
            }
        }
        int result = INT_MIN;
        for (int left = 0; left <= m; ++left) {
            result = max(result, f[left][m - left]);
        }
        return result;
    }
};
