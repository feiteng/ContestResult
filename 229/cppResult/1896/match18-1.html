class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n=nums.size(),m=multipliers.size();
        vector <vector<int> > f(m,vector<int>(m+1,0));
        f[0][0]=nums[n-1]*multipliers[0];
        f[0][1]=nums[0]*multipliers[0];
        for(int i=1;i<m;i++){
            f[i][0]=f[i-1][0]+nums[n-i-1]*multipliers[i];
            for(int j=1;j<=i;j++){
                f[i][j]=max(f[i-1][j-1]+nums[j-1]*multipliers[i],f[i-1][j]+nums[n-(i-j)-1]*multipliers[i]);
            }
            f[i][i+1]=f[i-1][i]+nums[i]*multipliers[i];
        }
        int ans=f[m-1][0];
        for(int i=1;i<=m;i++){
            ans=max(f[m-1][i],ans);
        }
        return ans;
    }
};
