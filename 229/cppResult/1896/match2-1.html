class Solution {
public:
    const long long MIN_SCORE = -1E9;

    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> max_score(m + 1, vector<int>(m + 1, MIN_SCORE));
        max_score[0][0] = 0;
        for (int i = 1; i <= m; ++i) {
            for (int left = 0; left <= i; ++left) {
                int right = i - left;
                if (left > 0) {
                    max_score[left][right] = max(max_score[left - 1][right] + nums[left - 1] * multipliers[i - 1], max_score[left][right]);
                }
                if (right > 0) {
                    max_score[left][right] = max(max_score[left][right - 1] + nums[n - right] * multipliers[i - 1], max_score[left][right]);
                }
            }
        }
        int ans = MIN_SCORE;
        for (int left = 0; left <= m; ++left) {
            ans = max(max_score[left][m - left], ans);
        }
        return ans;
    }
};
