class Solution {
public:
    int inf=-111111111,res=-111111111,n,m;

    vector<vector<int>>dp;
    int dfs(int l,int r,int now,vector<int>& a, vector<int>& s){
    if(r-l+1==n-m)return now;
    if(dp[l][n-1-r]!=inf)return now+dp[l][n-1-r];
    dp[l][n-1-r]=max(dfs(l+1,r,now,a,s)+a[l]*s[l+n-1-r],dfs(l,r-1,now,a,s)+a[r]*s[l+n-1-r]);
    return dp[l][n-1-r];
    }
    int maximumScore(vector<int>& a, vector<int>& s){
    n=a.size(),m=s.size();
    dp=vector<vector<int>>(m+1,vector<int>(m+1));

    for(int i=0;i<=m;i++)for(int j=0;j<=m;j++)dp[i][j]=inf;
    return dfs(0,n-1,0,a,s);
    }
};
