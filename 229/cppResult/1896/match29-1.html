class Solution {
    int helper(vector<int> &nums, vector<int> &multipliers, int left, int right, vector<vector<int>> &dp) {
        if (left + right >= multipliers.size())
            return 0;
        if (dp[left][right] == -1) {
            int a = nums[left] * multipliers[left+right] + helper(nums, multipliers, left+1, right, dp);
            int b = nums[nums.size()-right-1] * multipliers[left+right] + helper(nums, multipliers, left, right+1, dp);
            dp[left][right] = max(a, b);
        }
        return dp[left][right];

    }
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int m = multipliers.size(), n = nums.size();
        vector<vector<int>> dp(m, vector<int>(m, -1));
        return helper(nums, multipliers, 0, 0, dp);
    }
};
