class Solution {
public:
    
    
        //     m, n = len(multipliers), len(nums)

        // _dp = [[-1]*1000 for i in range(1000)]
        // def dp(l, r):
        //     if l+r == m:
        //         return 0
        //     if _dp[l][r] != -1:
        //         return _dp[l][r]
        //     _dp[l][r] = max(dp(l+1, r)+multipliers[l+r]*nums[l], 
        //               dp(l, r+1)+multipliers[l+r]*nums[n-r-1])
        //     return _dp[l][r]
        // return dp(0, 0)
    int _dp[1000][1000];
    int m, n;
    int dp(int l, int r, vector<int>& multipliers, vector<int>& nums){
        if (l+r==m) return 0;
        if (_dp[l][r] != -1) return _dp[l][r];
        _dp[l][r] = max(dp(l+1, r, multipliers, nums)+multipliers[l+r]*nums[l], 
                       dp(l, r+1, multipliers, nums)+multipliers[l+r]*nums[n-r-1]);
        return _dp[l][r];
    }
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        m = multipliers.size();
        n = nums.size();
        // int _dp[1000][1000];
        for (int i=0; i<1000;i++) for (int j=0; j<1000; j++) _dp[i][j] = -1;
        return dp(0, 0, multipliers, nums); 
    }
};
