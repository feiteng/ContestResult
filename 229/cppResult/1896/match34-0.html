class Solution {
public:
    
    int solve(vector<int>& nums, vector<int>& multipliers, int l, int index, vector<vector<int>> &dp)
    {
        int r = nums.size() + l - index - 1;
        if(index == multipliers.size() || l > r)
            return 0;
        if(dp[l][index] != -1)
            return dp[l][index];
        
        return dp[l][index] = max( multipliers[index]*nums[l] + solve(nums, multipliers, l+1, index+1, dp), 
                                multipliers[index]*nums[r] + solve(nums, multipliers, l,index+1, dp) );
    }
    
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m,vector<int>(m,-1));
        return solve(nums, multipliers, 0, 0, dp);
    }
};
