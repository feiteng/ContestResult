class Solution {
public:
    int dp[1001][10001];
    int solve(int i,int l,int r,vector<int>& a, vector<int>& m){
        if(i>=m.size()) return 0;
        if(dp[i][l]!=-1) return dp[i][l];
        int ans1=m[i]*a[l]+solve(i+1,l+1,r,a,m);
        int ans2=m[i]*a[r]+solve(i+1,l,r-1,a,m);
        return dp[i][l]=max(ans1,ans2);
    }
    int maximumScore(vector<int>& a, vector<int>& m) {
        memset(dp,-1,sizeof(dp));
        return solve(0,0,a.size()-1,a,m);
    }
};
