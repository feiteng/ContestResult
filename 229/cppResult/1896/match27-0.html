class Solution {
public:
    int maximumScore(vector<int>& a, vector<int>& b) {
        int n = a.size();
        int m = b.size();
        
        vector<vector<int> > f(m+2, vector<int> (m+2,-1e9-7));
        
        f[0][0] = 0;
        
        for(int i = 1; i <= m; i++){
            int x = b[i-1];
            for(int j = 0; j <= i-1; j++){
                int tail = n - 1 - (i-1-j);
                f[i][j] = max(f[i][j], f[i-1][j] + (x * a[tail]));
                f[i][j+1] = max(f[i][j+1], f[i-1][j] + (x * a[j]));
            }            
        }
        
        int ans = -1e9-7;
        for(int i = 0; i <= m; i++){
            ans = max(ans, f[m][i]);
        }
        return ans;
    }
};
