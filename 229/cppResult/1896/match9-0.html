class Solution {
public:
    int maximumScore(vector<int>& a, vector<int>& b) {
        int n = a.size(), m = b.size();
        int f[m+10][m+10];
        memset(f, 0, sizeof f);
        for(int i = 1; i <= m; i++) {
            for(int j = 0; j <= i; j++) {
                if(j == 0) {
                    f[i][j] = f[i-1][0] + b[i-1] * a[n-i];
                    continue;
                }
                f[i][j] = f[i-1][j-1] + b[i-1]*a[j-1];
                if(j < i) f[i][j] = max(f[i][j], f[i-1][j] + b[i-1] * a[n - (i - j)]);
            }
        }
        int ans = INT_MIN;
        for(int i = 1; i <= m; i++) ans = max(f[m][i], ans);
        return ans;
    }
};
