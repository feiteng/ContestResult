class Solution {
public:
	int maximumScore(vector<int>& nums, vector<int>& multipliers) {
		int n = nums.size();
		int m = multipliers.size();
		vector<vector<int>> dp(m + 1, vector<int>(m + 1, 0));

		dp[0][0] = 0;
		for (int k = 1; k <= m; k++) {
			for (int l = 0; l <= k; l++) {
				int r = k - l;
				//cout << l << " " << r << endl;
				if (l == 0) {
					dp[l][r] = dp[l][r - 1] + nums[n - r] * multipliers[k-1];
				}
				else if (r == 0) {
					dp[l][r] = dp[l - 1][r] + nums[l - 1] * multipliers[k-1];
				}
				else {
					dp[l][r] = max(dp[l - 1][r] + nums[l - 1] * multipliers[k-1],
						dp[l][r - 1] + nums[n - r] * multipliers[k-1]);
				}
			}
		}
		int ans = -1000000007;
		for (int l = 0; l <= m; l++) ans = max(ans, dp[l][m - l]);
		return ans;
	}
};
