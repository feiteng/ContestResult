class Solution {
public:
    int dfs(int up,int down,vector<vector<int>>&dp,vector<int>&nums,vector<int>&mul)
    {
        int n=nums.size(),m=mul.size();
        if(up>=n||down>=m){return 0;}
        if(dp[up][down]!=-1){return dp[up][down];}
        int x=mul[down]*nums[up]+dfs(up+1,down+1,dp,nums,mul);
        int y=INT_MIN;
        int left1=n-up,left2=m-down;
        if(left1>1)
        {
            int last=n-1-(down-up);
            y=mul[down]*nums[last]+dfs(up,down+1,dp,nums,mul);
            
        }

        return dp[up][down]=max(x,y);
        
        
    }
    int maximumScore(vector<int>& nums, vector<int>& v) {
        
        int n=nums.size(),m=v.size();
        vector<vector<int>>dp(m,vector<int>(m,-1));
        return dfs(0,0,dp,nums,v);
    }
};
