class Solution {
public:
    int getans(int i,int j,int cnt,vector<int>&arr,vector<int>&multipliers,vector<vector<int>>&dp)
    {
        if(cnt==multipliers.size())
            return 0;
        if(dp[i][j]!=INT_MIN)
            return dp[i][j];
        int ans1=arr[i]*multipliers[cnt]+getans(i+1,j,cnt+1,arr,multipliers,dp);
        int ans2=arr[j]*multipliers[cnt]+getans(i,j-1,cnt+1,arr,multipliers,dp);
        return dp[i][j]=max(ans1,ans2);
    }
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        
        int m=multipliers.size();
        vector<int> arr;
        for(int i=0;i<m;i++)
        {
            arr.push_back(nums[i]);
        }
        for(int j=nums.size()-m;j<nums.size();j++)
        {
            arr.push_back(nums[j]);
        }
        vector<vector<int>> dp(arr.size(),vector<int>(arr.size(),INT_MIN));
        return getans(0,arr.size()-1,0,arr,multipliers,dp);
    }
};
