class Solution {
    
    int maximumScore(vector<int>& nums, vector<int>& multipliers, int i, int j, int k, vector<vector<int>>& dp) {
        if (k == multipliers.size())
            return 0;
        if (dp[k][i] != -1)
            return dp[k][i];
        return dp[k][i] = max(
            nums[i] * multipliers[k] + maximumScore(nums, multipliers, i + 1, j, k + 1, dp),
            nums[j] * multipliers[k] + maximumScore(nums, multipliers, i, j - 1, k + 1, dp)
        );
    }
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m, vector<int>(m, -1));
        return maximumScore(nums, multipliers, 0, n - 1, 0, dp);
    }
};
