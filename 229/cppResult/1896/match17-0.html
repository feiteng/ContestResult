class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int n = nums.size(), m = mul.size();
        vector<vector<int>> dp(m+1, vector<int>(m+1, INT_MIN));
        dp[0][0] = nums[n-1]*mul[0], dp[0][1] = nums[0]*mul[0];
        for(int i = 1; i < m; i++) {
            for(int j = 0; j <= i+1; j++) {
                if(n-1-(i-j) >= 0 && n-1-(i-j) < n && dp[i-1][j] != INT_MIN) {
                    dp[i][j] = max(dp[i][j], dp[i-1][j]+mul[i]*nums[n-1-(i-j)]);
                }
                if(j-1 >= 0 && j-1 < n && dp[i-1][j-1] != INT_MIN) {
                    dp[i][j] = max(dp[i][j], dp[i-1][j-1]+mul[i]*nums[j-1]);
                }
            }
        }
        int ans = INT_MIN;
        for(int i = 0; i <= m; i++) {
            ans = max(ans, dp[m-1][i]);
        }
        return ans;
    }
};
