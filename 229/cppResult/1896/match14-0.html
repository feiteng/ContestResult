int dp[1001][1001];
class Solution {
public:
    
    int maximumScore(vector<int>& nums, vector<int>& ms) {
        int m=ms.size();
        memset(dp,8,sizeof dp);
        dp[0][0]=0;
       
        for(int i=1;i<=m;i++){
            dp[i][0]=dp[i-1][0]+nums[i-1]*ms[i-1];
        }
        for(int i=1;i<=m;i++){
            dp[0][i]=dp[0][i-1]+nums[nums.size()-i]*ms[i-1];
        }
        for(int i=1;i<=m;i++){
            for(int j=1;j<=m-i;j++){
            
                dp[i][j]=max(dp[i][j-1]+ms[i+j-1]*nums[nums.size()-j],dp[i-1][j]+ms[i+j-1]*nums[i-1]);
            }
        }
        int ans=1<<31;
        for(int i=0;i<=m;i++){
            ans=max(ans,dp[i][m-i]);
        }
        return ans;
        
    }
};
