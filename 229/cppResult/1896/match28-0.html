class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int n = nums.size(), m = mul.size(); 
        vector<vector<int>> dp(m+1, vector<int>(m+1, -2000000000)); // left take, right take
        dp[0][0] = 0;
        dp[0][1] = nums[n-1] * mul[0]; 
        dp[1][0] = nums[0] * mul[0];
        for(int s = 2; s <= m; s ++) {
            for(int l = 0; l <= s; l ++) {
                int r = s - l; 
                if(l > 0) {
                    dp[l][r] = max(dp[l][r], dp[l-1][r] + nums[l-1] * mul[s-1]); 
                }
                if(r > 0) {
                    dp[l][r] = max(dp[l][r], dp[l][r-1] + nums[n-r] * mul[s-1]); 
                }
                
            }
        }
        
        // for(auto v: dp) {
        //     for(auto e: v) cout << e << "\t";
        //     cout << endl;
        // }
        
        int ans = -2000000000; 
        for(int i = 0; i <= m; i ++) {
            ans = max(ans, dp[i][m-i]);
        }
        
        return ans;
    }
};
