class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        
        
      
        
        int result =INT_MIN;
        
        int start =0;
        int end =nums.size()-1;
        
        int len = multipliers.size();
        
        vector<vector<int>> dp(len+1,vector<int>(len+1,0));
        
        dp[0][0]=0;
        
        for(int i=1;i<=len;i++)
        {
            dp[i][0]= dp[i-1][0]+nums[i-1]*multipliers[i-1];
        }
        for(int j=1;j<=len;j++)
        {
            dp[0][j]=dp[0][j-1]+nums[nums.size()-j]*multipliers[j-1];
        }
        
        for(int i=1;i<=len;i++)
        {
            for(int j=1;j+i<=len;j++)
            {
                dp[i][j]=max(dp[i-1][j]+nums[i-1]*multipliers[i+j-1],dp[i][j-1]+nums[nums.size()-j]*multipliers[i+j-1]);
            }
        }
        
        
        for(int i=0;i<=len;i++)
        {
           result = max(result,dp[i][len-i]);
        }
        return result;
                
        
        
        
        
    }
};
