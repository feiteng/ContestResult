class Solution {
public:
    const int INF = 1e9 + 5;
    

    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size(), m = multipliers.size();
        vector<vector<int>> dp(m + 1, vector<int>(m + 1, -INF));
        // amount taken from left, number of multipliers
        dp[0][0] = 0;
        for (int j=0; j<m; j++)
            for (int i=0; i<=j; i++) {
                int r = n - (j - i) - 1;
                // take from left
                dp[i+1][j+1] = max(dp[i+1][j+1], dp[i][j] + nums[i] * multipliers[j]);
                // take from right

                dp[i][j+1] = max(dp[i][j+1], dp[i][j] + nums[r] * multipliers[j]);
            }
        int ret = -INF;
        for (int i=0; i<=m; i++)
            ret = max(ret, dp[i][m]);
        return ret;
    }
};
