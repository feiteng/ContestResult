class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& multipliers) {
        int n = nums.size();
        int m = multipliers.size();
        int ans[m + 1][m + 1];
        for (int i = 0; i != m + 1; ++i)
            ans[m][i] = 0;
        for (int mi = m - 1; mi != -1; --mi)
            for (int ni = 0; ni != mi + 1; ++ni) {
                ans[mi][ni] = max(multipliers[mi] * nums[ni] + ans[mi + 1][ni + 1], multipliers[mi] * nums[n - (mi - ni) - 1] + ans[mi + 1][ni]);
            }
        return ans[0][0];
    }
};
