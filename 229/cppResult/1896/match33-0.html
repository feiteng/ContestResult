class Solution {
public:
    int INF = 1e9;
    int maximumScore(vector<int>& nums, vector<int>& mul)
    {
        int n = nums.size(), m = mul.size();
        vector<vector<int>> dp(m + 1, vector<int>(m + 1, 0));
        //dp[1][0] = nums[0] * mul[0];
        //dp[0][1] = nums[n - 1] * mul[0];
        for(int i = 0; i <= m; i++)
        {
            for(int j = 0; j <= m; j++)
            {
                if(i + j - 1 >= m)
                    continue;
                if(i == 0 && j == 0)
                    continue;
                int can1 = -INF;
                if(i - 1 >= 0)
                    can1 = dp[i - 1][j] + nums[i - 1] * mul[i + j - 1];
                int can2 = -INF;
                if(j - 1 >= 0)
                    can2 = dp[i][j - 1] + nums[n - j] * mul[i + j - 1];
                dp[i][j] = max(can1, can2);
                //cout << i << " " << j << " " << can1 << " " << can2 << endl;
            }
        }
        
        int ret = -1e9;
        for(int i = 0; i <= m; i++)
        {
            int can = dp[i][m - i];
            ret = max(ret, can);
        }
        return ret;
    }
};
