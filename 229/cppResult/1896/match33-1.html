class Solution {
public:

    int maximumScore(vector<int>& a, vector<int>& mul) {
        int n = a.size(), m = mul.size();
        vector<vector<int>> f(m + 1, vector<int>(m + 1, 0));
        for(int k = m - 1; ~k; --k) for(int l = 0; l <= k; ++l)

        {
            int r = l + n - 1 - k;
            f[l][k] = max(f[l][k+1] + mul[k] * a[r], f[l+1][k+1] + mul[k] * a[l]);
        }
        return f[0][0];
    }
};
