class Solution {
public:
    int maximumScore(vector<int>& a, vector<int>& mul) {
        int n = a.size() , m = mul.size();
        vector<vector<int>> dp(m + 2, vector<int>(m + 2, -1e9));
        a.insert(a.begin(), 0);
        mul.insert(mul.begin(), 0);
        dp[0][1] = 0;
        for(int j = 0; j < m; ++j){
            for(int i = 1; i <= m; ++i){
                if(dp[j][i] == -1e9) continue;
                dp[j + 1][i] = max(dp[j + 1][i] , dp[j][i] + a[n - (j - (i - 1))] * mul[j + 1]);
                dp[j + 1][i + 1] = max(dp[j + 1][i + 1] , dp[j][i] + a[i] * mul[j + 1]);
            }
        }
        int res = -1e9;
        for(int i = 1; i <= m; ++i) res = max(res , dp[m][i]);
        return res;
    }
};
