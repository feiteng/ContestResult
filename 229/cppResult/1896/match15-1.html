class Solution {
public:
    int maximumScore(vector<int>& a, vector<int>& b) {
        int n = a.size();
        int m = b.size();
        vector<vector<int>> dp(m+1, vector<int>(m+1, -1e9));
        dp[0][0] = 0;
        for(int i=1; i<=m; i++){
            for(int j=0; j<=i; j++){

                if (j == 0) {
                    dp[i][j] = dp[i - 1][j] + b[i-1] * a[n - i + j];
                }
                else if(j==i){
                    dp[i][j] = dp[i - 1][j - 1] + b[i-1] * a[j - 1];
                }
                else {
                    dp[i][j] = max(dp[i - 1][j] + b[i-1] * a[n - i + j], dp[i - 1][j - 1] + b[i-1] * a[j - 1]);
                }
            }
        }
        int ans = -1e9;
        for(int j=0; j<=m; j++){
            ans = max(ans, dp[m][j]);
        }
        return ans;
    }
};
