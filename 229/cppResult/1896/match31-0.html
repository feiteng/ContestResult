class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int n=nums.size();
        int m=mul.size();
        int a[m+1][m+1];
        for(int i=0;i<=m;++i)
            a[i][m-i]=0;
        for(int s=m-1;s>=0;--s){
            for(int i=0;i<=s;++i){
                a[i][s-i]=max(a[i+1][s-i]+mul[s]*nums[i],a[i][s-i+1]+mul[s]*nums[n-1-s+i]);
            }
        }
        
        return a[0][0];
    }
};
