class Solution:
    def maximumScore(self, nums: List[int], mul: List[int]) -> int:
        n=len(nums)
        m=len(mul)
        dp=[[-inf]*(m+1)for i in range(m+1)]
        dp[0][0]=0
        for i in range(1,m+1):
            dp[i][0]=dp[i-1][0]+nums[i-1]*mul[i-1]
            dp[0][i]=dp[0][i-1]+nums[-i]*mul[i-1]
        for i in range(1,m+1):
            for j in range(1,min(m+1,m-i+1)):
                #print(i,j)
                dp[i][j]=max(dp[i-1][j]+nums[i-1]*mul[i+j-1],dp[i][j-1]+nums[-j]*mul[i+j-1])
        ans=max(dp[i][m-i] for i in range(m+1))
        return ans
