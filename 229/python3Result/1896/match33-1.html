class Solution:
    def maximumScore1(self, nums: List[int], multipliers: List[int]) -> int:
        nm = len(nums),len(multipliers)
        
        cache = dict()
        
        def func(i,j):
            if nm[0] - (j-i+1) >= nm[1]:
                return 0
            
            result = cache.get((i,j))
            if result is not None:
                return result
            k = nm[0] - (j-i+1)
            result = max(multipliers[k]*nums[i] + func(i+1,j), multipliers[k]*nums[j] + func(i,j-1))
            cache[(i,j)] = result
            return result
        
        return func(0, nm[0]-1)

    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n,m = len(nums),len
(multipliers)
        dp = [[0]*(m+1-i) for i in range(m+1)]
        for i in range(m-1,-1,-1):
            for j in range(m-i-1,-1,-1):
                if i + j == m-1:
                    dp[i][j] = max(multipliers[i+j]*nums[i], multipliers[i+j]*nums[n-1-j])
                else:
                    dp[i][j] = max(multipliers[i+j]*nums[i] + dp[i+1][j], multipliers[i+j]*nums[n-1-j] + dp[i][j+1])
        return dp[0][0]
