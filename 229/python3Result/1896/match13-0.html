class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:        
        out = 0
        
        @lru_cache(None)
        def dp(i, j, k):
            if k >= len(multipliers):
                return 0
            
            return max(nums[i]*multipliers[k]+dp(i+1, j, k+1), nums[j]*multipliers[k]+dp(i, j-1, k+1))
        x = dp(0, len(nums)-1, 0)
        dp.cache_clear()
        return x
