class Solution:
    def maximumScore(self, nums: List[int], muls: List[int]) -> int:
        n = len(nums)
        m = len(muls)
        
        dp = [[0] * m for _ in range(m)]
        
        for j in range(m-1, -1, -1):
            for i in range(j+1):
                if j == m-1:
                    dp[i][j] = max(muls[j] * nums[i], muls[j] * nums[n-1-(j-i)])
                else:
                    dp[i][j] = max(muls[j] * nums[i] + dp[i+1][j+1], muls[j] * nums[n-1-(j-i)] + dp[i][j+1])
        
        return dp[0][0]
