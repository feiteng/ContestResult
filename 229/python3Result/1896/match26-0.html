class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        d = {}
        n = len(nums)
        m = len(multipliers)
        d[(0, 1)] = multipliers[0] * nums[-1]
        d[(1, 0)] = multipliers[0] * nums[0]
        for k in range(2, m+1):
            d[(0, k)] = multipliers[k-1] * nums[-k] + d[(0, k-1)]
            d[(k, 0)] = multipliers[k-1] * nums[k-1] + d[(k-1, 0)]
            for i in range(1, k):
                j = k-i
                d[(i,j)] = max(multipliers[k-1]*nums[i-1]+d[(i-1,j)], multipliers[k-1]*nums[-j]+d[(i,j-1)])
        ans = d[(m, 0)]
        for i in range(m):
            ans = max(ans, d[(i, m-i)])
        return ans
