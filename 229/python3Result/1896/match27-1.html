class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n, m = len(nums), len(multipliers)
        dp = [[0 for j in range(m+1)] for i in range(m+1)]
        for k in range(1, m+1):
            for i in range(k+1):
                j = k-i
                if i == 0:
                    dp
[i][j] = dp[i][j-1] + nums[-j] * multipliers[k-1]
                elif j == 0:
                    dp[i][j] = dp[i-1][j] + nums[i-1] * multipliers[k-1]
                else:
                    dp[i][j] = max(dp[i-1][j] + nums[i-1] * multipliers[k-1], dp[i][j-1] + nums[-j] * multipliers[k-1])
        result = dp[m][0]
        for i in range(1, m):
            j = m-i
            result = max(result, dp[i][j])
        return result
