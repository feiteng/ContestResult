class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        
        m = len(multipliers)
        n = len(nums)
        dp = [[0] * (m) for i in range(m)]
        
        for i in range(m):
            #左边选了i个(该[i]) 右边选m-i-1个(该[n-1-m+i+1]) 总共选了m-1个 该选第m个时的最佳结果
            dp[i][m-i-1] = max(nums[i] * multipliers[-1], nums[n-m+i] * multipliers[-1])
        
        for j in range(m-1,-1,-1):
            for i in range(j):
                dp[i][j-i-1] = max(dp[i][j-i] + nums[n-j+i]*multipliers[j-1], dp[i+1][j-i-1] + nums[i]*multipliers[j-1])
        
        return dp[0][0]
