class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n = len(nums)
        m = len(multipliers)
        dp = [[0]*(m+1) for _ in range(m+1)]
        
        for i in range(m):
            dp
[i+1][0] = dp[i][0] + nums[i]*multipliers[i]
            dp[i+1][i+1] = dp[i][i] + nums[n-i-1]*multipliers[i]
            for j in range(1, i+1):
                dp
[i+1][j] = max(dp[i][j] + nums[i-j]*multipliers[i],
                                dp[i][j-1] + nums[n-j]*multipliers[i])
        # print(dp)
        
        return max(dp[m])
