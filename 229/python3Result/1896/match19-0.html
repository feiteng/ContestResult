class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n,m=len(nums),len(multipliers)
        dp0=[0 for _ in range(m+2)]
        for i in range(m-1,-1,-1):
            dp1=[0 for _ in range(m+2)]
            for left in range(i,-1,-1):
                right=left+n-1-i
                dp1[left]=max(multipliers[i]*nums[left]+dp0[left+1],multipliers[i]*nums[right]+dp0[left])
            dp0=dp1
        return dp0[0]
