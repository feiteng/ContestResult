class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        
        n, m = len(nums), len(multipliers)
        
        dp = [[-float('inf')] * (m+1) for _ in range(m+1)]
        
        for i in range(m+1):
            for j in range(m+1):
                if i + j > m:
                    break
                if i == 0 and j == 0:
                    dp[i][j] = 0
                elif i == 0:
                    dp[i][j] = dp[i][j-1] + nums[-j] * multipliers[j-1]
                elif j == 0:
                    dp[i][j] = dp[i-1][j] + nums[i-1] * multipliers[i-1]
                else:
                    dp[i][j] = max(dp[i][j-1] + nums[-j] * multipliers[i+j-1], dp[i-1][j] + nums[i-1] * multipliers[i+j-1])
                    
        
#         @lru_cache(None)
#         def dp(i, j):
#             #前面用了i，后面用了j，一共用了i+j
#             if i == 0 and j == 0:
#                 return 0
#             if i == 0:
#                 return dp(i, j-1) + nums[-j] * multipliers[j-1]
#             if j == 0:
#                 return dp(i-1, j) + nums[i-1] * multipliers[i-1]
#             return max(dp(i, j-1) + nums[-j] * multipliers[i+j-1], dp(i-1, j) + nums[i-1] * multipliers[i+j-1])
        
        res = -float('inf')
        for i in range(len(multipliers) + 1):
            res = max(res, dp[i][len(multipliers)-i])
        return res
