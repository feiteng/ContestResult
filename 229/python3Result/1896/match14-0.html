class Solution:
    def maximumScore(self, nums: List[int], mult: List[int]) -> int:
        
        @functools.lru_cache(None)
        def helper(i, k):
            
            if k == len(mult):
                return 0
            
            return max(mult[k]*nums[i] + helper(i+1, k+1), mult[k]*nums[n - k + i] + helper(i, k+1))
        
        n = len(nums)-1
        res = helper(0, 0)
        helper.cache_clear()
        return res
