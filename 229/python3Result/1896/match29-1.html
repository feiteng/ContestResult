class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        m = len(multipliers)
        dp = [[0]*m for i in range(m)]
        nums_ = nums[::-1]
        
for i in range(m):
            dp[i][m-i-1] = max(nums[i]*multipliers[-1],nums_[m-i-1]*multipliers[-1])
        for i in range(m-2,-1,-1):
            for j in range(m-i-2,-1,-1):
                dp[i][j] = max(nums[i]*multipliers[i+j]+dp[i+1][j],nums_[j]*multipliers[i+j]+dp[i][j+1])
        return dp[0][0]
