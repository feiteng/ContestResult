class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n=len(nums)
        m=len(multipliers)
        dp=[[0]*(m+1) for i in range(m+1)]
        for i in range(m):
            j=m-1-i
            dp[i][j]=max(multipliers[m-1]*nums[i], multipliers[m-1]*nums[n-1-j])
        for s in range(m-2,-1,-1):
            for i in range(s+1):
                j=s-i
                dp[i][j]=max(dp[i+1][j]+multipliers[s]*nums[i], dp[i][j+1]+multipliers[s]*nums[n-1-j])
        return dp[0][0]
