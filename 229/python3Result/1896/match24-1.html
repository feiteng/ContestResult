class Solution:
    def maximumScore(self, nums: List[int], mul: List[int]) -> int:
        d = collections.defaultdict(int)
        n = len(nums)
        m = len(mul)
        ans = [[-float('inf')] * (m+1) for i in range
(m+1)]
        ans[0][0] = 0
        for i in range(m):
            for j in range(i+1):
                ans[i+1][j] = max(ans[i+1][j], ans[i][j] + mul[i] * nums[-1-i+j])
                ans[i+1][j+1] = max(ans[i+1][j+1], ans[i][j] + mul[i] * nums[j])
                
        return max(ans[m])
