class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n = len(nums)
        m = len(multipliers)
        d = [[0 for _ in range(m+1)] for _ in range(m+1)] 
        # d[mm,i] means max amount of chosen mm number, and i from the front side, then mm - i from the end side
        # d[mm,i] = max(d[mm-1,i] + nums[-(mm-i)]*multipliers[mm],d[mm-1,i-1] + nums[i] * multipliers[mm]), 0<= i <= mm, 1<= mm <= m
        for mm in range(1,m+1):
            d
[mm][0] = d[mm-1][0] + nums[-mm] * multipliers[mm-1]
            d[mm][mm] = d[mm-1][mm-1] + nums[mm-1] * multipliers[mm-1]
            for i in range(1,mm):
                    d
[mm][i] = max(d[mm-1][i] + nums[-(mm-i)]*multipliers[mm-1],d[mm-1][i-1] + nums[i-1] * multipliers[mm-1])     
        return max(d[mm])
