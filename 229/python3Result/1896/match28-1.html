class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        mult = multipliers
        
        n, m = len(nums), len(multipliers)
        
        #opt[i, j] means subproblem on nums[i:n-j]
        opt = [[0 for _ in range(m+1)]
               for _ in range(m+1)]
        
        for i in range(m):
            j = m - 1 - i
            k 
= i + j
            opt[i][j] = max(mult[k] * nums[i], mult[k] * nums[n-j-1])
            
        for l in range(2, m+1):
            k = m - l
            
for i in range(k+1):
                j = k - i
                opt[i][j] = max(opt[i+1][j] + mult[k] * nums[i],
                               opt[i][j+1] + mult[k] * nums[n-j-1])
        return opt[0][0]
