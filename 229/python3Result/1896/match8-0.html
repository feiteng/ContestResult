class Solution:
    def maximumScore(self, nums, multipliers) -> int:
        N = len(nums)
        M = len(multipliers)
        dp = [[0 for j in range(M)] for i in range(M)]
        for k in range(M - 1, -1, -1):
            for i in range(k + 1):
                j = k - i
                if k == M - 1:
                    dp[i][j] = max(multipliers[k] * nums[i],
                                   multipliers[k] * nums[(N - 1) - j])
                else:
                    dp[i][j] = max(
                        multipliers[k] * nums[i] + dp[i + 1][j],
                        multipliers[k] * nums[(N - 1) - j] + dp[i][j + 1])
        return dp[0][0]
