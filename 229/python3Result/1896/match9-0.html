class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        # dp, dp[i][j] is the maximum multiplication sum when nums[:i+1] and nums[n-j:] have been used
        # notice the fact that i <= m and j >= n-m

        n, m = len(nums), len(multipliers)
        
        dp = [[0 for _ in range(m)] for _ in range(m)]
        
        for k in range(1, m+1):
            # k is number of elements remaining in multipliers
            # cosumed elements in nums is k
            for i in range(m-k+1):
                j = m - k - i # right end of nums
                if k == 1:
                    dp[i][j] = max(nums[i]*multipliers[-k], nums[-j-1]*multipliers[-k])
                else:
                    dp[i][j] = max(nums[i]*multipliers[-k]+dp[i+1][j], nums[-j-1]*multipliers[-k]+dp[i][j+1])
        return dp[0][0]
