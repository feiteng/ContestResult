class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n = len(nums)
        m = len(multipliers)
        dp = [[0 for i in range(m + 1)] for j in range(m + 1)]
        
        for l in range(1, m + 1):
            for head in range(l+1):
                if head == 0:
                    dp[l][head] = dp[l-1][head] + multipliers[l-1]*nums[n-l]
                elif head == l:
                    dp[l][head] = dp[l-1][head-1] + multipliers[l-1]*nums[l-1]
                else:
                    dp[l][head] = max(dp[l-1][head-1] + multipliers[l-1]*nums[head-1], dp[l-1][head] + multipliers[l-1]*nums[n-l+head])
        return max(dp[-1])
