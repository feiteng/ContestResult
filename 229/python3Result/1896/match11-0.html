class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        ans = -float('inf')
        dp = [[0] * (len(multipliers)+1) for i in range(len(multipliers)+1)]
        for i in range(1, len(multipliers)+1):
            for p in range(i+1):
                if p == 0 and i-p == 0:
                    dp[p][i-p] = 0
                elif p == 0:
                    dp[p][i-p] = dp[p][i-p-1] + multipliers[i-1]*nums[p-i]
                elif i-p == 0:
                    dp[p][i-p] = dp[p-1][i-p] + multipliers[i-1]*nums[p-1]
                else:
                    dp[p][i-p] = max(dp[p-1][i-p] + multipliers[i-1]*nums[p-1], dp[p][i-p-1] + multipliers[i-1]*nums[p-i])
        for i in range(len(multipliers)+1):
            ans = max(ans, dp[i][len(multipliers)-i])
        return ans
