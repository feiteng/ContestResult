class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n = len(nums)
        m = len(multipliers)
        if 2*m < n:
            nums = nums[:m] + nums[-m:]
            n = len(nums)
            
        dp = [[0] * n for i in range(n)]
        
        
        for i in range(0, n-(n-m)):
            dp[i][i+n-m] = max(multipliers[-1]*nums[i], multipliers[-1] * nums[i+n-m])
        
        for size in range(n-m+1, n):
            for i in range(0, n-size):
                dp[i][i+size] = max(dp[i+1][i+size]+nums[i]*multipliers[n-size-1], 
                                    dp[i][i+size-1]+nums[i+size]*multipliers[n-size-1])
        
        return dp[0][n-1]
