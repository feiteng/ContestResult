class Solution:
    def maximumScore(self, A: List[int], M: List[int]) -> int:
        n = len(A)
        m = len(M)
        dp = [[0] * (m+1) for _ in range(m+1)]
        dp[1][0] = M[0] * A[0]
        dp[0][1] = M[0] * A[-1]
        
for i in range(2, m+1):
            dp[i][0] = dp[i-1][0] + M[i-1] * A[i-1]
            dp[0][i] = dp[0][i-1] + M[i-1] * A[-i]
            for j in range
(1, i):
                dp[j][i-j] = max(dp[j-1][i-j] + M[i-1] * A[j-1], dp[j][i-j-1] + M[i-1] * A[-(i-j)])
        return max(dp[i][m-i] for i in range(m+1))
