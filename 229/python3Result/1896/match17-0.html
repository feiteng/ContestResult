class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n = len(nums)
        m = len(multipliers)
        #dp # dp[i, j] 1<=i<=m, 0<=j <=i = take multipliers[:i] and nums[:j] + nums[-(i-j):]
        dp = [[0]*(m+1) for _ in range(m+1)]
        for i in range(1,m+1):
            for j in range(i+1):
                if j == 0: # i-j==i
                    dp[i][j] = dp[i-1][j] +  multipliers[i-1]*nums[-i]
                elif j == i:
                    dp[i][j] = dp[i-1][j-1] +  multipliers[i-1]*nums[j-1]
                else:
                    dp[i][j] = max(dp[i-1][j-1] + multipliers[i-1]*nums[j-1],
                                  dp[i-1][j] + multipliers[i-1]*nums[-(i-j)],
                                  )
        #print(dp)
        return max(dp[m])
