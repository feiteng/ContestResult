class Solution:
    def maximumScore(self, A: List[int], B: List[int]) -> int:
        n, m = len(A), len(B)
        
        dp = [[0 for _ in range(i + 1)] for i in range(m + 1)]
        
        for i in range(m - 1, -1, -1):
            length = n - i
            for j in range(i + 1):
                k = j + length - 1
                dp[i][j] = max(dp[i + 1][j] + B[i] * A[k], dp[i + 1][j + 1] + B[i] * A[j])
            
        
        return dp[0][0]
