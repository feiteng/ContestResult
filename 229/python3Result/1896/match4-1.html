class Solution:
    def maximumScore(self, ns: List[int], ms: List[int]) -> int:
        m, n = len(ms), len(ns)

        # @functools.cache
        # def kg(i, ls, rs):

        #         ms[i] * ns[~rs] + kg(i + 1, ls, rs + 1),
        #     )
        # return kg(0, 0, 0)

        """
        [123]
        [321]
        14 5 1
         8 2 0
         3 0 0
        """
        dp = [[0] * (m + 1) for _ in range
(m + 1)]
        for k in range(m - 1, -1, -1):
            for i in range(k + 1):
                j = k - i
                dp[i][j] = max(
                    dp[i][j + 1] + ns[~j] * ms[k],
                    dp[i + 1][j] + ns[i] * ms[k],
                )
        return dp[0][0]
