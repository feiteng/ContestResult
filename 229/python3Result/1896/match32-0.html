class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n, m = len(nums), len(multipliers)
        dp = [[None for _ in range(m + 1)] for _ in range(m + 1)]
        
        for i in range(0, m + 1):
            for j in range(0, m + 1):
                if i + j > m:
                    break
                    
                if i == 0 and j == 0:
                    dp[i][j] = 0
                elif i == 0:
                    dp[i][j] = dp[i][j - 1] + nums[-j] * multipliers[j - 1]
                elif j == 0:
                    dp[i][j] = dp[i - 1][j] + nums[i - 1] * multipliers[i - 1]
                else:
                    dp[i][j] = max(dp[i - 1][j] + nums[i - 1] * multipliers[i + j - 1], \
                                  dp[i][j - 1] + nums[-j] * multipliers[i + j - 1])
                    
        result = float('-inf')
        for i in range(m + 1):
            result = max(result, dp[i][m - i])
            
        return result
