class Solution:
    def maximumScore(self, nums: List[int], multipliers: List[int]) -> int:
        n,m = len(nums),len(multipliers)
        # @lru_cache(None)
        # def dp(left,k):

        # return dp(0,0)
        dp = [0 for _ in range(m+1)]
        for k in range(m-1,-1,-1):
            ndp = [float('-inf') for _ in range(m+1)]
            for left in range(k+1):
                ndp
[left] = max(dp[left+1]+nums[left]*multipliers[k],dp[left]+nums[n-1+left-k]*multipliers[k])
            dp = ndp
        return max(dp)
