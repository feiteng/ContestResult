class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        n = len(boxes)
        ans = [0] * n

        tmp = 0
        sum = 0
        for i in range(0, n):
            ans[i] += sum
            tmp += boxes[i] == '1'
            sum += tmp

        tmp = 0
        sum = 0
        for i in range(n - 1, -1, -1):
            ans[i] += sum
            tmp += boxes[i] == '1'
            sum += tmp

        return ans
