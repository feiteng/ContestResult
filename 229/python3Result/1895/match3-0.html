class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        res = []
        b = []
        for i in range(len(boxes)):
            if boxes[i] == '1':
                b.append(i)
        for i in range(len(boxes)):
            t = 0
            for j in b:
                t += abs(j-i)
            res.append(t)
        return res
