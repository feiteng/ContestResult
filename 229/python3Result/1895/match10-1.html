class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        pos=[]
        for i,c in enumerate(boxes):
            if c=="1":pos.append(i)
        
        ans=[0]*len(boxes)
        for i in range(len(boxes)):
            for j in pos:
                ans[i]+=abs(i-j)
        
        return ans
