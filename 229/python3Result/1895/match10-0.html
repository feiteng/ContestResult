class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        onesIndexes = []
        for idx, b in enumerate(boxes):
            if b == '1':
                onesIndexes.append(idx)
        
        results = [0] * len(boxes)
        for i in range(len(boxes)):
            for oi in onesIndexes:
                results[i] += abs(i - oi)
        
        return results
