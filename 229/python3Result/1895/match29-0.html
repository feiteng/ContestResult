class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        N = len(boxes)
        res = [0] * N
        
        left = 0
        leftCount = 0
        for i in range(N):
            res[i] = left
            leftCount += boxes[i] == '1'
            left += leftCount
            
        right = 0
        rightCount = 0
        for i in range(N - 1, -1, -1):
            res[i] += right
            rightCount += boxes[i] == '1'
            right += rightCount
            
        return res
