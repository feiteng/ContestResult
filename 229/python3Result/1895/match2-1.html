class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        
        result = []
        
        for i in range(len(boxes)):
            temp_result = 0
            for j in range(len(boxes)):
                if boxes[j] == "1":
                    temp_result += abs(j - i)
                    
            result.append(temp_result)
            
        return result
