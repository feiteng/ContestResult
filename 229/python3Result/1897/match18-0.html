class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        L1 = len(word1)
        L2 = len(word2)
        word = word1 + word2
        dp = [[0] * (L1 + L2) for _ in range(L1 + L2)]
        res = 0
        for i in range(L1 + L2 - 1, -1, -1):
            dp[i][i] = 1
            for j in range(i + 1, L1 + L2):
                if word[i] == word[j]:
                    dp[i][j] = dp[i + 1][j - 1] + 2
                    if i < L1 and j >= L1:
                        res = max(res, dp[i][j])
                else:
                    dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])
        return res
