class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        m = len(word1)
        mem = {}
        
        s = word1 + word2
        res = 0
        
        def dp(i, j):
            nonlocal res
            if i > j:
                return 0
            if i == j:
                return 1
            if (i,j) in mem:
                return mem[(i,j)]
            
            ans = 0
            if s[i] == s[j]:
                ans = dp(i+1, j-1) + 2
                if i < m and j >= m:
                    res = max(res, ans)
            else:
                ans = max(dp(i, j-1), dp(i+1, j))
                
            
                
            mem[(i, j)] = ans
            return ans
            
        dp(0, len(s) - 1)
        return res
