class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        
        w = word1+word2
        
        n = len(w)
        dp 
= [[0]*n for _ in range(n)]

        for i in range(n):
            dp[i][i] = 1
        
        res = 0
        for d in range(1,n):
            for i in range(n-d):
                j = i+d
                
if w[i]==w[j]:
                    dp[i][j] = 2+dp[i+1][j-1]
                    if i < len(word1) and j >= len(word1):
                        res = max(res,dp[i][j])
                else:
                    dp[i][j] = max(dp[i][j-1],dp[i+1][j])
        
        return res
