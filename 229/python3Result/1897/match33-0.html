class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        w = word1 + word2
        @lru_cache(None)
        def helper(i, j):
            if i>=j: return 1 if i==j else 0
            if w[i]==w[j]: return 2 + helper(i+1, j-1)
            return max(helper(i+1, j), helper(i, j-1))
        ans = 0
        for i in range(len(word1)):
            for j in range(len(word1), len(word1)+len(word2)):
                if w[i]==w[j]: ans = max(ans, 2 + helper(i+1, j-1))
        return ans
