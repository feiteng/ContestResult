class Solution:
    def longestPalindrome(self, w1: str, w2: str) -> int:
        m, n = len(w1), len(w2)
        w = w1 + w2
        t = m+n
        dp = [[1 for _ in range(i, t)] for i in range(t)]
        for i in range(t-1):
            if w[i]==w[i+1]:
                dp[i][1] = 2
        for j in range(2, t):
            for i in range(t-j):
                if w[i]==w[i+j]:
                    dp[i][j] = dp[i+1][j-2] + 2
                else:
                    dp[i][j] = max(dp[i+1][j-1], dp[i][j-1])
        res = 0
        for i in range(m):
            for j in range(n):
                if w1[i]==w2[j]:
                    res = max(res, dp[i][m+j-i])
        return res
