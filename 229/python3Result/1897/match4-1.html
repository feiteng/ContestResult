class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        word = word1 + word2
        n2 = len(word1)
        n = len(word)
        dp = [[0] * n for i in range(n)]
        
        best = 0
        for i in reversed(range
(n)):
            dp[i][i] = 1
            for j in range(i+1, n):
                if word[i] == word[j]:
                    dp[i][j] = dp[i+1][j-1]+2
                    # print(i, j, dp[i][j])
                    if i < n2 and j >= n2:
                        best = max(best, dp[i][j])
                else:
                    dp[i][j] = max(dp[i+1][j], dp[i][j-1])
        
        return best
