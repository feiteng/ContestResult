class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        combine = word1 + word2
        dp = [[0 for i in range(m+n)] for j in range
(m+n)]
        for i in range(m+n):
            dp[i][i] = 1
        for i in range(m+n-2, -1, -1):
            for j in range(i+1, m+n):
                if combine[i] == combine[j]:
                    if j == i+1:
                        dp[i][j] = 2
                    else:
                        dp[i][j] = 2 + dp[i+1][j-1]
                else:
                    dp
[i][j] = max(dp[i+1][j], dp[i][j-1])
        res = 0
        for i in range(m):
            for j in range(n):
                if word1[i] == word2[j]:
                    res = max(res, dp[i][j+m])
        return res
