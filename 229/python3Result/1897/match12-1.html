class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        L = len(word1)
        s = word1 + word2
        
        ans = 0
        
        mem = {}
        
        def dp(i, j):
            nonlocal ans
            if i > j:
                return 0
            if i == j:
                return 1
            elif (i, j) in mem:
                return mem[(i, j)]
            elif s
[i] == s[j]:
                res = dp(i+1, j-1) + 2
                if i < L and j >= L:
                    ans = max(ans, res)
            else:
                res = max(dp(i+1, j), dp(i, j-1))
            mem[(i, j)] = res
            return res
            
        dp(0, len(s) - 1)
        return ans
