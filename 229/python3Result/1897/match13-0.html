class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        l1 = len(word1)
        l2 = len(word2)
        s = word1 + word2
        n = len(s)
        dp = [[0] * n for _ in range(n)]
        for i in range(n):
            dp[i][i] = 1
        for i in range(n):
            for j in range(i)[::-1]:
                if s[i] == s[j]:
                    dp[j][i] = 2 + dp[j+1][i-1]
                else:
                    dp[j][i] = max(dp[j+1][i], dp[j][i-1])
        
        ans = 0
        # print(dp)
        for i in range(l1):
            for j in range(l1,n)[::-1]:
                # print(j)
                if s[i] == s[j]:
                    ans = max(ans, dp[i][j])
                
        return ans
