class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        word = word1 + word2
        l1, l2, l = len(word1), len(word2), len(word)
        dp = [[0] * l for _ in range(l)]
        for i in range(l):
            dp[i][i] = 1
        res = 0
        for le in range(2, l+1):
            for s in range(l - le+1):
                e = s + le - 1
                if word[s] == word[e]:
                    dp[s][e] = dp[s+1][e-1] + 2
                    if s < l1 and e >=l1:
                        res = max(res, dp[s][e])
                else:
                    dp[s][e]= max(dp[s+1][e], dp[s][e-1])
        #print(dp)
        #print(dp[0][l-1])
        return res
