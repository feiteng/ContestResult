class Solution:
    def longestPalindrome(self, w1: str, w2: str) -> int:
        w = w1 + w2
        @functools.lru_cache(None)
        def f(l, r):
            if l == r: return 1
            if l > r: return 0
            if w[l] == w[r]: return 2 + f(l + 1, r - 1)
            return max(f(l, r-1), f(l+1, r))
        
        l1, l2 = len(w1), len(w2)
        mp = {}
        for i in range(l2):
            mp[w2[i]] = i
        ans = 0
        for i in range
(l1):
            if w1[i] in mp:
                ans = max(ans, 2 + f(i + 1, l1 + mp[w1[i]] - 1))
        return ans
