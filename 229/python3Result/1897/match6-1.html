class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        n,m = len(word1),len
(word2)
        temp = word1+word2
        ll = len(temp)
        dp = [[0]*ll for _ in range(ll)]
        for i in range(ll):
            dp[i][i] = 1
        for l in range(2,ll+1):
            for i in range(ll-l+1):
                j = i+l-1
                
if temp[i]==temp[j]:
                    dp[i][j] = 2+dp[i+1][j-1]
                else:
                    dp[i][j] = max(dp[i+1][j],dp[i][j-1])
        res = 0
        for i in range(ll):
            for j in range(i+1,ll):
                
if temp[i]==temp[j] and i<n and j>=n:
                    res = max(res,dp[i][j])
        return res
