class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        k = len(word1)
        s = word1 + word2
        n = len(s)
        dp = [[0]*n for _ in range(n)] # longest for range(i,j)
        res = 0
        for l in range(1, n+1):
            for i in range(n):
                j = i+l-1
                if i == j:
                    dp[i][j] = 1
                elif j>=n:
                    break
                else:
                    if s[i] == s[j]:
                        dp[i][j] = dp[i+1][j-1] + 2
                        if i < k and j >=k:
                            res = max(res, dp[i][j])
                    else:
                        dp[i][j] = max(dp[i+1][j], dp[i][j-1])
        return res
