class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        word2 = word2[::-1]
        n1 = len(word1)
        n2 = len(word2)
        dp = [[0 for i in range(n2)] for j in range(n1)]
        
        for i in (range(n1)):
            for j in (range(n2)):
                if word1[i] == word2[j]:
                    dp[i][j] = 1
                    if i > 0:
                        dp[i][j] = max(dp[i][j], dp[i-1][j])
                    if j > 0:
                        dp[i][j] = max(dp[i][j], dp[i][j-1])
                    if i > 0 and j > 0:
                        dp[i][j] = max(dp[i][j], dp[i-1][j-1]+1)
                else:
                    if i > 0:
                        dp[i][j] = max(dp[i][j], dp[i-1][j])
                    if j > 0:
                        dp[i][j] = max(dp[i][j], dp[i][j-1])
        
        dp2 = [[0 for i in range(n2)] for j in range(n1)]
        def longestPalindromeSubseq(s):
            n = len(s)
            dp = [[0]*(n) for _ in range(n) ]

            for i in range(n):
                dp[i][i] = 1


            for i in range(n-1, -1, -1):
                for j in range(i+1, n):                    
                    if s[i]==s[j]:
                        dp[i][j] = dp[i+1][j-1]+2
                    else:
                        dp[i][j] = max(dp[i][j-1], dp[i+1][j])

            return dp[0]
        
        memo = {}
        l1 = longestPalindromeSubseq(word1[::-1])[::-1]
        l2 = longestPalindromeSubseq(word2[::-1])[::-1]
        l1.pop(0)
        l1.append(0)
        l2.pop(0)
        l2.append(0)
        for i in range(n1):
            for j in range(n2):
                dp2[i][j] = max(l1[i], l2[j])
        
        res = 0
        for i in range(n1):
            for j in range(n2):
                if dp[i][j] != 0:
                    res = max(res, dp[i][j] * 2 + dp2[i][j])
        
        return res
