class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        n1 = len(word1)
        n2 = len
(word2)
        w = word1 + word2
        length = n1 + n2

        ans = 0
        dp = [[0] * length for _ in range(length)]
#         for i in range(n1 + n2):
#             dp[i][i] = 1
            
#         for r in range(n1, n2 + n1):
#             for l in range(n1 - 1, -1, -1):
#                 if w[l] == w[r]:
#                     # if l + 1 < n1 and r - 1 >= n1:
#                     dp[l][r] = dp[l + 1][r - 1] + 2
#                     # elif l == n1 - 1 and r == n1:

#                 else:
#                     dp[l][r] = max(dp[l + 1][r], dp[l][r - 1])
#                 ans = max(ans, dp[l][r])
#         return ans
                    
        for i in range(length - 1, -1, -1):
            dp[i][i] = 1
            for j in range(i + 1, length):
                if w[i] == w[j]:
                    dp[i][j] = dp[i + 1][j - 1] + 2
                    if i < n1 <= j:
                        ans = max(ans, dp[i][j])
                else:
                    dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])
        return ans
