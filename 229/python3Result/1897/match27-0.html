class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        merged = word1 + word2
        
        @functools.lru_cache(None)
        def get_max_palindrome(i, j):
            if i > j:
                return 0
            if i == j:
                return 1
            if merged[i] == merged[j]:
                return 2 + get_max_palindrome(i + 1, j - 1)
            return max(get_max_palindrome(i + 1, j), get_max_palindrome(i, j - 1))
            
        res = 0
        rindex = {}
        for j in range(len(merged) - 1, len(merged) - len(word2) - 1, -1):
            if merged[j] not in rindex:
                rindex[merged[j]] = j
        # print(rindex)
        for i in range(len(word1)):
            if word1[i] in rindex:
                res = max(res, get_max_palindrome(i, rindex[word1[i]]))
        return res
