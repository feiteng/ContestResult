class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        l1 = len(word1)
        s = word1 + word2
        n = len(s)
        dp = [[0]*n for _ in range(n)]
        ans = 0
        for i in range(n):
            dp[i][i] = 1
        for k in range(2, n+1):
            for i in range(n-k+1):
                j = i + k - 1
                
if s[i] == s[j]:
                    dp[i][j] = (2 if k == 2 else dp[i+1][j-1] + 2)
                    if i < l1 and j >= l1:
                        ans = max(ans, dp[i][j])                    
                else:
                    dp[i][j] = max(dp[i+1][j], dp[i][j-1])
        return ans
