class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        n1, n2 = len(word1), len(word2)
        n = n1 + n2
        word = word1 + word2
        ans = 0
        dp = [[0] * n for _ in range(n)]
        for i in range(n - 1, -1, -1):
            dp[i][i] = 1
            for j in range(i + 1, n):
                if word[i] == word[j]:
                    dp[i][j] = dp[i + 1][j - 1] + 2
                    if i < n1 and j >= n1:
                        ans = max(ans, dp[i][j])
                else:
                    dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])
        return ans
