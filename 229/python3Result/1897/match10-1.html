class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        m = len(word1)
        s = word1 + word2
        n = len(s)
        # print(s)
        # print(n)
        dp = [[0 for _ in range(n)] for _ in range
(n)]
        for i in range(n):
            dp[i][i] = 1
        # flag = False
        res = 0
        for r in range(n):
            # if r < m:
            # for l in range(min(r-1, m-1), -1, -1):
            for l in range(r-1, -1, -1):
                
if s[l] == s[r]:
                    dp[l][r] = dp[l+1][r-1] + 2
                    if r >= m and l < m:
                        # flag = True
                        res = max(res, dp[l][r])
                    # if dp[l][r] == 5:
                    #     print((l, r))
                else:
                    dp[l][r] = max(dp[l+1][r], dp[l][r-1])
            # else: # r in word2
            #     for l in range(m-1, -1, -1):
        # print(dp)
                
        # return 0 if not flag else dp[0][n-1]
        return res
