class Solution:
    def longestPalindrome(self, word1: str, word2: str) -> int:
        n1, n2 = len(word1), len(word2)
        word = word1 + word2
        n = len(word)
        
        # L[i][j] is the longest palindrome length of word[i:j+1]
        L = [[0 for x in range(n)] for x in range
(n)] 
  
        # Strings of length 1 are palindrome of length 1 
        for i in range(n): 
            L[i][i] = 1
        
        ret = 0
        for k in range(2, n+1): 
            for i in range(n-k+1,): 
                j = i+k-1
                if word[i] == word[j] and k == 2: 
                    L[i][j] = 2
                    if i < n1 and j >= n1:
                        ret = max(ret, L[i][j])
                elif word
[i] == word[j]: 
                    L[i][j] = L[i+1][j-1] + 2
                    if i < n1 and j >= n1:
                        ret = max(ret, L[i][j])                    
                else: 
                    L[i][j] = max(L[i][j-1], L[i+1][j]); 
  
        return ret
