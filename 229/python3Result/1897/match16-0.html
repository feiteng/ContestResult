class Solution:
    def longestPalindrome(self, s1: str, s2: str) -> int:
        s = s1 + s2
        

        n = len(s)
        if s == s[::-1]: return n
        dp = [[0] * n for _ in range(n)]
        for i in range(n):
            dp[i][i] = 1

        res = 0
        for le in range(2, n + 1):
            for l in range(n):
                r = l + le - 1
                if r >= n: continue
                if s[l] == s[r]:
                    dp[l][r] = dp[l + 1][r - 1] + 2
                    if l < len(s1) and r >= len(s1):
                        res = max(res, dp[l][r])
                else:
                    dp[l][r] = max(dp[l + 1][r], dp[l][r - 1])
        

        return res
