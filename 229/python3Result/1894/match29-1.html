class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        a = 0
        ans = ''
        while a<len(word1) and a<len(word2):
            ans += word1[a]
            ans += word2[a]
            a += 1
        if a<len(word1):
            ans += word1[a:]
        if a<len(word2):
            ans += word2[a:]
        return ans
