class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        m = min(len(word1), len(word2))
        lst = []
        for i in range(m):
            lst.append(word1[i])
            lst.append(word2[i])
        if len(word1) > m:
            lst.append(word1[m:])
        if len(word2) > m:
            lst.append(word2[m:])
        return ''.join(lst)
