class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        index1, index2 = 0, 0
        res = []
        turn = 0
        while index1 < len(word1) and index2 < len(word2):
            if turn == 0:
                res.append(word1[index1])
                index1 += 1
            else:
                res.append(word2[index2])
                index2 += 1
            turn = 1-turn
        while index1 < len(word1):
            res.append(word1[index1])
            index1 += 1
        while index2 < len(word2):
            res.append(word2[index2])
            index2 += 1
        return ''.join(res)
