class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        p1 = 0
        p2 = 0
        merged_word = ""
        while p1 < len(word1) and p2 < len(word2):
            merged_word += word1[p1]
            merged_word += word2[p2]
            p1 += 1
            p2 += 1
            
        if p1 < len(word1):
            merged_word += word1[p1:]
        
        if p2 < len(word2):
            merged_word += word2[p2:]
            
        return merged_word
