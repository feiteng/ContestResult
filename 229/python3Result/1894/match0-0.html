class Solution:
    def mergeAlternately(self, w1: str, w2: str) -> str:
        n=len(w1)
        m=len(w2)
        i=j=0
        ans=''
        while i<n and j<m:
            ans+=w1[i]+w2[j]
            i+=1
            j+=1
        if i<n:
            ans+=w1[i:]
        if j<m:
            ans+=w2[j:]
        return ans
