class Solution {
public:
    bool check(vector<int>& nums) {
        int n=nums.size();
        for(int i=0;i<nums.size();i++){
            bool can=true;
            for(int j=0;j<nums.size()-1;j++){
                if(nums[(i+j)%n]>nums[(i+j+1)%n])can=false;
            }
            if(can)return true;
        }
        return false;
    }
};
