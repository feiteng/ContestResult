class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        vector<int> v = nums;
        sort(v.begin(), v.end());
        
        for(int i = 1; i <= n; i ++) {
            int l = 1, r = n - 1;
            while(l < r) {
                swap(nums[l], nums[r]);
                l ++;
                r --;
            }
            l = 0, r = n - 1;
            while(l < r) {
                swap(nums[l], nums[r]);
                l ++;
                r --;
            }
            bool flag = true;
            for(int i = 0; i < n; i ++) {
                if(v[i] != nums[i]) {
                    flag = false;
                    break;
                }
            }
            if(flag) return true;
        }
        
        return false;
    }
};
