class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for(int st = 0; st < n; ++st) {

            bool ok = true;
            for(int i = 0; i + 1 < n; ++i) {
                if(nums[(st + i + 1) % n] < nums[(st + i) % n]) {
                    ok = false;
                    break;
                }
            }
            if(ok) {
                return true;
            }
        }
        return false;
    }
};
