class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for(int st = 0;st<n;st++){
            int prev = nums[st];
            bool foo = true;
            for(int c=0;c<n;c++){
                int cur = nums[(c+st)%n];
                if(prev>cur) {
                    foo = false;
                    break;
                }
                prev = cur;
            }
            if(foo) return true;
        }
        return false;
    }
};
