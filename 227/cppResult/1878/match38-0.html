class Solution {
public:
    bool check(vector<int>& nums) {
        int L=nums.size();
        rep(i,L){
            bool ok = true;
            rep(j,L-1){
                if (nums[(i+j)%L] > nums[(i+j+1)%L]) { ok = false; break; }
            }
            if (ok) return true;
        }
        return false;
    }
};
