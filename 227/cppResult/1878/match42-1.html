class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();        
        for (int i = 0; i < n; ++i)
        {
            int start = i;

            bool ret = true;
            for (int t = 1; t < n; ++t)
            {
                int index = (i + t) % n;
                int prev = (i + t - 1 + n) % n;
                if (nums[index] < nums[prev]) 
                {
                    ret = false;
                    break;
                }
            }
            if (ret == true) return true;
        }
        return false;
    }
};
