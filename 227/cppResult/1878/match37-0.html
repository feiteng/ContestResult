class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for (int rep=0; rep<n; rep++) {
            bool ok = true;
            for (int i=1; i<n; i++)
                ok &= nums[i] >= nums[i-1];
            if (ok)
                return true;
            rotate(nums.begin(), nums.begin() + 1, nums.end());
        }
        return false;
    }
};
