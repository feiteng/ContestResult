class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            nums.insert(nums.begin(), nums.back());
            nums.pop_back();
            bool flag = true;
            for (int j = 1; j < n; j++) {
                if (nums[j] < nums[j - 1]) {
                    flag = false;
                    break;
                }
            }
            if (flag) return true;
        }
        return false;
    }
};
