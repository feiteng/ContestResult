#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=1e6+5;
class Solution {
public:
    bool check(vector<int>& nums) {
        int n=nums.size();
        for(int i=0;i<n;i++){
            vector<int>ans;    for(int j=i;j<n;j++)ans.push_back(nums[j]);
            for(int j=0;j<i;j++)ans.push_back(nums[j]);
            
            int f=0;
            for(int j=0;j<n;j++){
                if(j+1<n&&ans[j]>ans[j+1])f=1;
            }
            if(f==0)return 1;
        }
        return 0;
        

    }
};
