class Solution {
public:
    bool check(vector<int>& nums) {
        int len = nums.size();
        for(int i = 0;i<len;i++){
            bool flag = true;
            for(int j = i+1;(j%len)!=i;j++){
                if(nums[j%len]<nums[(j-1)%len]) {
                    flag =false;
                    break;
                }                
            }
            if(flag) return true;
        }
        return false;
    }
};
