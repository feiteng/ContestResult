class Solution {
public:
    bool check(vector<int>& w) {
        vector<int> b = w;
        sort(b.begin(), b.end());
        int n = b.size();
        for(int x=0; x<n; x++){
            bool flag = true;
            for(int i=0; i<n; i++){
                if(b[(i+x)%n] != w[i]){
                    flag = false;
                    break;
                }
            }
            if(flag) return true;
        }
        return false;
    }
};
