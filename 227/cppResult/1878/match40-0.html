class Solution {
public:
    bool check(vector<int>& nums) 
    {
        int idx = 0;
        int n = nums.size();
        for(int idx=0;idx<n;idx++)
        {
            int flag = 1;
            for(int i=1;i<n;i++)
            {
                if(nums[(idx+i)%n] < nums[(idx+i-1)%n])
                {
                    flag = 0;
                    break;
                }
            }
            if(flag == 1)
                return true;
        }
        
        return false;
    }
};
