class Solution {
public:
    bool check(vector<int>& a) {
        int n = a.size();
        for (int k = 0; k < n; ++k) {
            vector<int> b(n);
            for (int i = 0; i < n; ++i) {
                b[i] = a[(k + i) % n];
            }
            bool found = false;
            for (int i = 1; i < n; ++i) {
                if (b[i] < b[i - 1]) found = true;
            }
            if (!found) return true;
        }
        return false;

    }
};
