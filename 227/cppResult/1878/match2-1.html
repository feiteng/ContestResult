class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            bool good = 1;
            for (int j = (i+1)%n; j != i && good; j = (j+1)%n) {
                if (nums[(j-1+n)%n] > nums[j]) good = 0;
            }
            if (good) return 1;
        }
        return 0;
    }
};
