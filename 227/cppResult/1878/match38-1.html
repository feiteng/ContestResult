class Solution {
#define rep(i,a,b) for(int i = (a);i <= (b);++i)
#define re_(i,a,b) for(int i = (a);i < (b);++i)
#define dwn(i,a,b) for(int i = (a);i >= (b);--i)
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        re_(i,0,n){
            bool fl = true;
            re_(j,0,n-1){
                if(nums[(i+j)%n] > nums[(i+j+1)%n]){
                    fl = false;break;
                }
            }
            if(fl) return true;
        }
        return false;
    }
};
