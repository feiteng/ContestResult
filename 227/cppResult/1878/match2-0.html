class Solution {
public:
    bool check(vector<int>& nums) {
        int n=nums.size();
        for ( int i=0; i<n; i++ ) {
            bool ok=true;
            for ( int j=1; j<n; j++ ) {
                if ( nums[(i+j)%n]<nums[(i+j-1)%n] ) ok=false;
            }
            if ( ok ) return ok;
        }
        return false;
    }
};
