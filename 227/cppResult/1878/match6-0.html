class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for (int st = 0; st < n; ++st) {
            bool ok = true;
            for (int i = 0; i < n - 1; ++i) {
                if (nums[(st + i) % n] > nums[(st + i + 1) % n])
                    ok = false;
            }
            if (ok) return true;
        }
        return false;
    }
};
