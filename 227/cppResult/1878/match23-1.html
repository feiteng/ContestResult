class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        for (int s = 0; s < n; ++s) {
            bool sorted = true;
            for (int i = 0; i < n - 1; ++i) {
                if (nums[(s+i)%n] > nums[(s+i+1)%n]) {
                    sorted = false;
                    break;
                }
            }
            if (sorted) return true;
        }
        return false;
    }
};
