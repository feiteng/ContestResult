class Solution {
public:
    bool check(vector<int>& nums) {
        int n = nums.size();
        vector<int> tmp(nums.begin(), nums.end());
        sort(tmp.begin(), tmp.end());
        for (int x = 0; x < n; ++x) {
            bool flag = true;
            for (int i = 0; i < n; ++i) {
                if (tmp[i] != nums[(i + x) % n]) {
                    flag = false;
                    break;
                }
            }
            if (flag) return true;
        }
        return false;
    }
};
