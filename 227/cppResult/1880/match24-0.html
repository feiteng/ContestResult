class Solution {
public:
    bool compare(string w1, string w2){
        return w1 > w2;
    }
    string largestMerge(string word1, string word2) {
        int l = 0, r = 0;
        int n1 = word1.size(), n2 = word2.size();
        string res;
     //   res += 'a';
        while(l < n1 && r < n2){
            if(word1[l] > word2[r]){
                res += word1[l];
                l++;
            }else if(word1[l] < word2[r]){
                res += word2[r];
                r++;
            }else{
                string s1 = word1.substr(l, n1 - l), s2 = word2.substr(r, n2 - r) ;
                if(compare(s1, s2)){
                    res += word1[l];
                    l++;
                }else{
                    res += word2[r];
                    r++;
                }
            }
            
        }
        while(l < n1){
            res += word1[l];
            l++;
            
        }
        while(r < n2){
            res += word2[r];
            r++;
        }
        return res;
        

    }
};
