class Solution {
public:
    string largestMerge(string word1, string word2) {
        int p1 = 0;
        int p2 = 0;

        int len1 = word1.size();
        int len2 = word2.size();
        string ans = "";
        while (p1 < len1 && p2 < len2) {
            if (word1[p1] > word2[p2]) {
                ans += word1[p1++];
            }
            else if (word1[p1] < word2[p2]) {
                ans += word2[p2++];
            }
            else {
                int t1 = p1;

                int t2 = p2;
                while (t1 < len1 && t2 < len2 && word1[t1] == word2[t2]) {
                    ++t1;
                    ++t2;
                }
                if (t1 == len1) {
                    ans += word2[p2++];
                }
                else if (t2 == len2) {
                    ans += word1[p1++];
                }
                else if (word1[t1] > word2[t2]) {
                    ans += word1[p1++];
                }
                else {
                    ans += word2[p2++];
                }
            }
        }
        while (p1 < len1) { ans += word1[p1++]; }
        while (p2 < len2) { ans += word2[p2++]; }
        return ans;
    }
};
