class Solution {
public:
    string largestMerge(string word1, string word2) {
        string merge = "";
        int idx1, idx2;
        
        idx1 = 0, idx2 = 0;
        while (idx1 != word1.size() && idx2 != word2.size())
        {
            if (word1[idx1] > word2[idx2])
            {
                merge += word1[idx1];
                idx1++;
            }
            else if (word1[idx1] < word2[idx2])
            {
                merge += word2[idx2];
                idx2++;
            }
            else  // 相等的时候，移动更大的那个指针，更可能遇到更大的字符
            {
                if (strcmp(word1.c_str() + idx1, word2.c_str() + idx2) >= 0)
                {
                    merge += word1[idx1];
                    idx1++;
                }
                else
                {
                    merge += word2[idx2];
                    idx2++;
                }
            }
        }
        while (idx1 != word1.size())
        {
            merge += word1[idx1++];
        }
        while (idx2 != word2.size())
        {
            merge += word2[idx2++];
        }
        return merge;
    }
};
