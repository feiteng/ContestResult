class Solution {
public:
    string largestMerge(string word1, string word2) {
        string ans = "";
        int i = 0, j = 0;
        while(i < word1.size() && j < word2.size()){
            if(word1[i] < word2[j]) {
                ans += word2[j];
                j++;
            } else if(word1[i] > word2[j]){
                ans += word1[i];
                i++;
            } else {
                int diff = 1;
                while(diff+i < word1.size() && diff+j < word2.size() && word1[diff+i] == word2[diff+j]) diff++;
                if(diff+i == word1.size() || word1[diff+i] < word2[diff+j]) {
                    ans += word2[j];
                    j++;
                } else {
                    ans += word1[i];
                    i++;   
                }
            }
        }
        while(i < word1.size()){
            ans += word1[i];
            i++;
        }
        while(j < word2.size()){
            ans += word2[j];
            j++;
        }
        return ans;
    }
};
