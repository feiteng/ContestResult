class Solution {
public:
    string largestMerge(string word1, string word2) {
        int poz1= 0, poz2 = 0;
        string ret;
        while (poz1 < word1.size() || poz2 < word2.size()){
            if (word1.substr(poz1) < word2.substr(poz2))
                ret.push_back(word2[poz2++]);
            else
                ret.push_back(word1[poz1++]);
        }
        return ret;
    }
};
