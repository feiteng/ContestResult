class Solution {
public:
    string largestMerge(string word1, string word2) {
        int n=word1.size(),m=word2.size();

        int i=0,j=0;
        string ans="";
        while(i<n && j<m){
            if(word1[i]<word2[j]){
                ans += word2[j];
                ++j;
            }
            else if(word1[i]>word2[j]){
                ans += word1[i];
                ++i;
            }
            else{
                int i1=i,j1=j;
                while(i1<n && j1<m && word1[i1]==word2[j1]){
                    ++i1;++j1;
                }
                if(i1==n){
                    ans += word2[j];
                    ++j;
                }
                else if(j1==m){
                    ans += word1[i];
                    ++i;
                }
                else{
                    if(word1[i1]<word2[j1]){
                        ans += word2[j];
                        ++j;
                    }
                    else if(word1[i1]>word2[j1]){

                        ans += word1[i];
                        ++i;
                    }       
                }
            }
        }
        while(i<n){
            ans+=word1[i];
            ++i;
        }
        while(j<m){
            ans+=word2[j];
            ++j;
        }
        return ans;
    }
};
