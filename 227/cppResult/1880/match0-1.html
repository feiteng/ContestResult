class Solution {
public:
    string largestMerge(string s1, string s2) {
        string s;
        while(s1.size()>0||s2.size()>0){
            if(s1>=s2){
                s+=s1[0];
                s1=s1.substr(1);
            }
            else{
                s+=s2[0];
                s2=s2.substr(1);
            }
        }
        return s;
    }
};
