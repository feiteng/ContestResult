class Solution {
public:
    string largestMerge(string word1, string word2) {
		
		string ans;
		
		int index1 = 0;
		int index2 = 0;
		
		int length1 = word1.size();
		int length2 = word2.size();
		
		while(index1<length1 && index2<length2){
			
			if(word1[index1] < word2[index2]){
				ans.push_back(word2[index2]);
				++index2;
			}else if(word1[index1] > word2[index2])
			{
				ans.push_back(word1[index1]);
				++index1;
			}else{
				int tmpIndex1 = index1+1;
				int tmpIndex2 = index2+1;
				
				bool isFirLarge = true;
				
				while(tmpIndex1<length1&& tmpIndex2<length2){
					
					if(word1[tmpIndex1]<word2[tmpIndex2]){
						isFirLarge = false;
						break;
					}else if(word1[tmpIndex1]>word2[tmpIndex2]){
						break;
					}else{
						++tmpIndex1;
						++tmpIndex2;
					}
				}
				
				//word1首先结束，则其比较小
				if(tmpIndex1 == length1){
					isFirLarge = false;
				}
				
				if(isFirLarge){
					ans.push_back(word1[index1]);
					++index1;
				}else{
					ans.push_back(word2[index2]);
					++index2;
				}
			}
		}
		
		//cout<<ans<<":"<<index1<<" "<<index2<<endl;
		
		while(index1<length1){
			ans.push_back(word1[index1]);
			++index1;
		}
		
		while(index2<length2){
			ans.push_back(word2[index2]);
			++index2;
		}
		
		return ans;
    }
};
