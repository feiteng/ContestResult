class Solution {
public:
    string largestMerge(string word1, string word2) {
        string ans="";
        int i1=0,i2=0;
        while(i1<word1.size() && i2<word2.size()){
            if(word1[i1]<word2[i2])
                ans+=word2[i2++];
            else if(word2[i2]<word1[i1])
                ans+=word1[i1++];
            else{
                int bigger1=0;
                for(int i=i1;i<word1.size() && i-i1+i2<word2.size() && i-i1+i2>=0;i++){
                    if(word2[i-i1+i2]>word1[i]){
                        bigger1=-1;
                        break;
                    }
                    else if(word1[i]>word2[i-i1+i2]){
                        bigger1=1;
                        break;
                    }
                }
                if(bigger1==0){
                    if(int(word1.size()-i1)>int(word2.size())-i2)
                        bigger1=1;
                    else
                        bigger1=-1;
                }

                if(bigger1==-1)
                    ans+=word2[i2++];
                else
                    ans+=word1[i1++];
            }
        }
        while(i1<word1.size())
            ans+=word1[i1++];
        while(i2<word2.size())
            ans+=word2[i2++];
        return ans;
    }
};
