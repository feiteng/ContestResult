class Solution {
public:
    string largestMerge(string word1, string word2) {
        string res;

        int n1 = word1.length(), n2 = word2.length();
        int i = 0, j = 0;
        
        while (i < n1 && j < n2)
        {
            if (word1[i] < word2[j]) {
                res += word2[j ++ ];
            } else if (word1[i] > word2[j]) {
                res += word1[i ++ ];
            } else {
                if (word1.substr(i) > word2.substr(j)) {
                    res += word1[i ++ ];
                } else {
                    res += word2[j ++ ];
                }
            }
        }
        
        if (i < n1) res += word1.substr(i);
        if (j < n2) res += word2.substr(j);
         
        return res;
    }
};
