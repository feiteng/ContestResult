class Solution {
public:
    string largestMerge(string word1, string word2) {
        string ret;
        int a = 0, b = 0;
        while(a < word1.size() || b < word2.size()) {
            if(a == word1.size()) {
                ret.push_back(word2[b]);
                b++;
            } else if(b == word2.size()) {
                ret.push_back(word1[a]);
                a++;
            } else if(word1[a] > word2[b]) {
                ret.push_back(word1[a]);
                a++;
            } else if(word1[a] < word2[b]) {
                ret.push_back(word2[b]);
                b++;
            } else {
                if(word1.substr(a) < word2.substr(b)) {
                    ret.push_back(word2[b]);
                b++;
                } else {
                    ret.push_back(word1[a]);
                a++;
                }
            }
        }
        return ret;
    }
};
