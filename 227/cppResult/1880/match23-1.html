class Solution
{
public:
    string largestMerge(string word1, string word2)
    {
        string ans;
        int m=word1.size(),n=word2.size(),i=0,j=0,k,l;
        while(i<m&&j<n)
        {
            if(word1[i]>word2[j])
            {
                ans.push_back(word1[i++]);
            }
            else if(word2[j]>word1[i])
            {
                ans.push_back(word2[j++]);
            }
            else
            {
                k=i;l=j;
                while(k<m&&l<n&&word1[k]==word2[l])
                {
                    k++;
                    l++;
                }
                if(k==m)
                {
                    ans.push_back(word2[j++]);
                }
                else if(l==n)
                {
                    ans.push_back(word1[i++]);
                }
                else if(word1[k]>word2[l])
                {
                    ans.push_back(word1[i++]);
                }
                else

                {
                    ans.push_back(word2[j++]);
                }
            }
        }
        while(i<m)
        {
            ans.push_back(word1[i++]);
        }
        while(j<n)
        {
            ans.push_back(word2[j++]);
        }
        return ans;
    }
};
