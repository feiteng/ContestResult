class Solution {
public:
    string largestMerge(string word1, string word2) {
        int n1=word1.size();
        int n2=word2.size();
        int i=0;
        int j=0;
        string ans;
        while(i<n1 && j<n2){
            if(word1[i]==word2[j]){
                string t1=word1.substr(i+1);
                string t2=word2.substr(j+1);
                if(t1>=t2){
                    ans.push_back(word1[i]);
                    i++;
                }
                else{
                    ans.push_back(word2[j]);
                    j++;
                }
            }

            else if(word1[i]>word2[j]){
                ans.push_back(word1[i]);
                i++;
            }
            else{
                ans.push_back(word2[j]);
                j++;
            }
        }
        while(i<n1){
            ans.push_back(word1[i]);
            i++;
        }
        while(j<n2){
            ans.push_back(word2[j]);
            j++;
        }
        return ans;
    }
};
