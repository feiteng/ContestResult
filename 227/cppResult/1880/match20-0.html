class Solution {
public:
    string largestMerge(string word1, string word2) {
        string ans;
        int n1 = word1.size(), n2 = word2.size();
        int i1 = 0, i2 = 0;
        while (i1 < n1 && i2 < n2) {
            if (word1[i1] > word2[i2]) {
                ans += word1[i1];
                i1++;
            } else if (word1[i1] < word2[i2]) {
                ans += word2[i2];
                i2++;
            } else {
                if (word1.substr(i1) > word2.substr(i2)) {
                    ans += word1[i1];
                    i1++;
                } else {
                    ans += word2[i2];
                    i2++;
                }
            }
        }
        while (i1 < n1) {
            ans += word1[i1];
            i1++;
        }
        while (i2 < n2) {
            ans += word2[i2];
            i2++;
        }
        return ans;
    }
};
