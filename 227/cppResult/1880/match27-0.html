class Solution {
public:
    string largestMerge(string word1, string word2) {
int n1 = word1.size(), n2 = word2.size();
    string merge = "";
    int i = 0, j = 0;
    while (i < n1 && j < n2) {
      if (word1[i] <word2[j]) {
        merge += word2[j];
        ++j;
      }
      else if (word1[i] > word2[j]) {
        merge += word1[i];
        ++i;
      }
      else {
        if (word1.substr(i + 1) > word2.substr(j + 1)) {
          merge += word1[i];
          ++i;
        }
        else {
          merge += word2[j];
          ++j;
        }
      }
    }
    if (i < n1) {
      while (i < n1) {
        merge += word1[i];
        ++i;
      }
    }
    if (j < n2) {
      while (j < n2) {
        merge += word2[j];
        ++j;
      }
    }
    return merge;
    }
};
