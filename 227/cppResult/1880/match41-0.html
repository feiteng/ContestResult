class Solution {
public:
    string largestMerge(string word1, string word2) {
        int i=0,j=0,ni=word1.size(),nj=word2.size();
        string res;
        while(i<ni&&j<nj){
            if(word1.substr(i,ni-i+1)>word2.substr(j,nj-j+1)){
                res+=word1[i++];
            }else{
                res+=word2[j++];
            }
        }
        while(i<ni){
            res+=word1[i++];
        }
        while(j<nj){
            res+=word2[j++];
        }
        return res;
    }
};
