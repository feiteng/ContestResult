class Solution {
public:
  string largestMerge(string word1, string word2) {
    string res;
    int l1 = 0, l2 = 0;
    int n1 = word1.size(), n2 = word2.size();
    while (l1 < n1 && l2 < n2) {

      if (word1.substr(l1) > word2.substr(l2)) res.push_back(word1[l1++]);
      else res.push_back(word2[l2++]);
    }
    while (l1 < n1) res.push_back(word1[l1++]);
    while (l2 < n2) res.push_back(word2[l2++]);
    return res;
  }
};
