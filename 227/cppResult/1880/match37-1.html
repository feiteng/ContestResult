class Solution {
public:
    string largestMerge(string word1, string word2) {
        string output = "";
        int idx1 = 0, idx2 = 0;
        while (idx1 < word1.length() && idx2 < word2.length())
        {
            if (word1[idx1] < word2[idx2])
            {
                output += word2[idx2]; idx2++;
            }
            else if (word1[idx1] > word2[idx2])
            {
                output += word1[idx1]; idx1++;
            }
            else
            {
                int idx1_copy = idx1, idx2_copy = idx2;
                while (idx1_copy < word1.length() && idx2_copy < word2.length() && word1[idx1_copy] == word2[idx2_copy])
                {
                    idx1_copy++;
                    idx2_copy++;
                }
                if (idx1_copy == word1.length())
                {
                    output += word2[idx2]; idx2++;
                }
                else if (idx2_copy == word2.length())
                {
                    output += word1[idx1]; idx1++;
                }

                else if (word1[idx1_copy] < word2[idx2_copy])
                {
                    output += word2[idx2]; idx2++;
                }
                else
                {
                    output += word1[idx1]; idx1++;
                }
            }
        }
        while (idx1 < word1.length())
        {
            output += word1[idx1];
            idx1++;
        }
        while (idx2 < word2.length())
        {
            output += word2[idx2]; idx2++;
        }
        return output;
    }
};
