class Solution {
public:
    string largestMerge(string word1, string word2) {
        string merge;
        int n1 = word1.size();
        int n2 = word2.size();
        int i=0,j=0,k=0;
        while(i<n1 && j<n2){
            if (word1[i] > word2[j]){
                merge += word1[i];
                i ++;
            }
            else if (word1[i] < word2[j]){
                merge += word2[j];
                j ++;
            }
            else if(word1[i] == word2[j]) {
                int t1 = i,t2=j;

                while(t1<n1 && t2<n2 && word1[t1] == word2[t2]){
                    // cout<<t1<<" "<<t2<<endl;
                    // cout<<word1[t1]<<" "<<word2[t2]<<endl;
                    t1 ++;
                    t2 ++;
                }
                // cout<<word1[t1]<<" "<<word2[t2]<<endl;
                if (t1 == n1){
                    merge+= word2[j];
                    j ++;
                }
                else if (t2 == n2){
                    merge+= word1[i];
                    i ++;
                }
                else if (word1[t1] > word2[t2]){
                    merge += word1[i];
                    i ++;
                }
                else{
                    merge += word2[j];
                    j ++;
                }
            }
            
        }
        
        while (i<n1){
            merge += word1[i];
            i ++;
        }
        while(j<n2){
            merge += word2[j];
            j ++;
        }
        
        return merge;
    }
};
