class Solution {
public:
    string largestMerge(string word1, string word2) {
        int len1=word1.length(),len2=word2.length();
        int i=0,j=0;
        string ans="";
        while(i<len1&&j<len2){
            if(word1.substr(i,len1-i)>word2.substr(j,len2-j)) {ans+=word1[i];i++;}
            else {ans+=word2[j];j++;}
        }
        while(i<len1){
            ans+=word1[i];i++;   
        }
        while(j<len2){
            ans+=word2[j];j++;
        }
        return ans;
    }
};
