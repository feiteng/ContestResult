class Solution {
public:
    string largestMerge(string word1, string word2) {
        stringstream ss;
        while (word1.size()>0 && word2.size()>0){
            if (word1 >= word2) {
                ss << word1[0];
                word1 = word1.substr(1);
            } else {
                ss << word2[0];
                word2 = word2.substr(1);
            }
        }
        ss << word1 << word2;
        return ss.str();
    }
};
