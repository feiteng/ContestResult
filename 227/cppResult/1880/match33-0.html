class Solution {
public:
    string largestMerge(string word1, string word2) {
        string ans = "";
        int i = 0, j = 0;
        while(i < word1.size() && j < word2.size()){
            if(word1[i] > word2[j]) ans += word1[i++];
            else if(word1[i] < word2[j]) ans += word2[j++];
            else{
                int a = i, b = j;
                while(a < word1.size() && b < word2.size()){
                    if(word1[a] == word2[b]) a++, b++;
                    else break;
                }
                if(a < word1.size() && b < word2.size()){
                    if(word1[a] > word2[b]) ans += word1[i++];
                    else if(word1[a] < word2[b]) ans += word2[j++];
                }
                else if(a == word1.size())
                    ans += word2[j++];
                else ans += word1[i++];
            }
        }
        while(i < word1.size()) ans += word1[i++];
        while(j < word2.size()) ans += word2[j++];
        return ans;
    }
};
