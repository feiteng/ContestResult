class Solution {
public:
    string largestMerge(string word1, string word2) {
        string merge="";
        while(!word1.empty()||!word2.empty())
        {
            if(word1.empty())
            {
                merge+=word2;
                word2.clear();
            }
            else if(word2.empty())
            {
                merge+=word1;
                word1.clear();
            }
            else
            {
                if(word1>=word2)
                {
                    merge+=word1[0];
                    word1.erase(word1.begin());
                }
                else
                {
                    merge+=word2[0];
                    word2.erase(word2.begin());
                }
            }
        }
        return merge;
    }
};
