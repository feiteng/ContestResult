class Solution {
public:
    string largestMerge(string word1, string word2) {
      string_view a(word1);
      string_view b(word2);
      string c;
      while (!a.empty() || !b.empty()) {
        if (a > b) {
          c += a[0];
          a = a.substr(1);
        } else {
          c += b[0];
          b = b.substr(1);
        }
      }
      return c;
    }
};
