class Solution {
public:
    string largestMerge(string word1, string word2) {
        int i,j,k,l,m,n,r;
        n=word1.size();
        m=word2.size();
        l=r=0;
        string merge;
        while(l<n && r<m)
        {
            if(word1[l]>word2[r])
            {
                merge.push_back(word1[l]);
                l++;
            }
            else if(word1[l]<word2[r])
            {
                merge.push_back(word2[r]);
                r++;
            }
            else
            {
                i=l;
                j=r;
                while(i<n && j<m && word1[i]==word2[j])
                {
                    i++;
                    j++;
                }
                if(word1[i]>word2[j])
                {
                    merge.push_back(word1[l]);
                    l++;
                }
                 else
                 {
                     merge.push_back(word2[r]);
                     r++;
                 }
            }
        }
        while(l<n)
        {
            merge.push_back(word1[l]);
            l++;
        }
        while(r<m)
        {
            merge.push_back(word2[r]);
            r++;
        }
        return merge;
    }
};
