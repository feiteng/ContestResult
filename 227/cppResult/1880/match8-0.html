class Solution {
public:
    string largestMerge(string word1, string word2) {
        int n=word1.size();
        int m=word2.size();
        string ans;
        int i=0,j=0;
        while(i<n&&j<m){
            if(word1.substr(i)<word2.substr(j)){
                ans.push_back(word2[j]);
                ++j;
            }else{
                ans.push_back(word1[i]);
                ++i;
            }
        }
        while(i<n){
            ans.push_back(word1[i]);
            ++i;
        }
        while(j<m){
            ans.push_back(word2[j]);
            ++j;
        }
        return ans;
    }
};
