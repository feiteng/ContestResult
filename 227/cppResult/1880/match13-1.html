class Solution {
public:
    string largestMerge(string w1, string w2) {
        string ans="";
        int n1=w1.size();

        int n2=w2.size();
        int i=0,j=0;
        while(i<n1&&j<n2){
            if(w1[i]>w2[j]){
                ans+=w1[i];
                i++;
            }else if(w1[i]<w2[j]){
                ans+=w2[j];
                j++;
            }else{
                string tmp1=w1.substr(i);
                string tmp2=w2.substr(j);
                if(tmp1>=tmp2){
                    ans+=w1[i];
                    i++;
                }else{
                    ans+=w2[j];
                    j++;
                }
            }
        }
        while(i<n1){
            ans+=w1[i];
            i++;
        }
        while(j<n2){
            ans+=w2[j];
            j++;
        }
        return ans;
    }
};
