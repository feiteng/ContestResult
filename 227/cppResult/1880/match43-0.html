class Solution {
public:
    string largestMerge(string word1, string word2) {
        int m=word1.size(),n=word2.size();
        int i=0,j=0;
        string ret;
        while(i<m&&j<n){
            if(word1[i]>word2[j]){
                ret+=word1[i];
                i++;
            }else if(word1[i]<word2[j]){
                ret+=word2[j];
                j++;
            }else{
                int t1=i+1,t2=j+1,phase=-1;
                while(t1<m&&t2<n){
                    if(word1[t1]<word2[t2]){
                        phase=0;
                        break;
                    }else if(word1[t1]>word2[t2]){
                        phase=1;
                        break;
                    }
                    t1++;
                    t2++;
                }
                if(phase==-1){
                    if(t1<m)
                        phase=1;
                    else
                        phase=0;
                }
                if(phase==0){
                    ret+=word2[j];
                    j++;
                }else{
                    ret+=word1[i];
                    i++;
                }
            }
        }
        while(i<m){
            ret+=word1[i];
            i++;
        }
        while(j<n){
            ret+=word2[j];
            j++;
        }
        return ret;
    }
};
