class Solution {
public:
    string largestMerge(string word1, string word2) {
        
        int i=0;
        int j=0;
        string ans="";
        while(i<word1.length() and j<word2.length())
        {
            if(word1[i]>word2[j])
            {
                ans+=word1[i];
                i++;
            }
            else if(word1[i]<word2[j])
            {
                ans+=word2[j];
                j++;
            }
            else
            {
                string p1=word1.substr(i+1,word1.length()-1-(i+1)+1);
                string p2=word2.substr(j+1,word2.length()-1-(j+1)+1);
                if(p1>=p2)
                {
                    ans+=word1[i];
                    i++;
                }
                else
                {
                    ans+=word2[j];
                    j++;
                }
            }
        }
        while(i<word1.length())
        {
            ans+=word1[i];
            i++;
        }
        while(j<word2.length())
        {
            ans+=word2[j];
            j++;
            
        }
        return ans;
    }
};
