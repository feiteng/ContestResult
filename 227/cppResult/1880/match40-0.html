class Solution {
public:
    string largestMerge(string word1, string word2) {
	string ans;
	while (word1.size() || word2.size()) {
	    if (word1 < word2) {
		ans.push_back(word2[0]);
		word2 = word2.substr(1);
	    } else {
		ans.push_back(word1[0]);
		word1 = word1.substr(1);
	    }
	}
	return ans;
    }
};
