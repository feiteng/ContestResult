class Solution {
public:
    string largestMerge(string a, string b) {
        string res="";
        int i=0,j=0;
        while(i<a.length()&&j<b.length())
        {
            if(a[i]>b[j])
                res+=a[i++];
            else if(a[i]<b[j])
                res+=b[j++];
            else
            {

                if(a.substr(i,a.length()-i)>b.substr(j,b.length()-j))
                    res+=a[i++];
                else
                    res+=b[j++];
            }
        }
        while(i<a.length())
            res+=a[i++];
        while(j<b.length())
            res+=b[j++];
        return res;
    }
};
