class Solution {
public:
    string largestMerge(string word1, string word2) {
        string lM = "";
        while (!(word1.empty() && word2.empty())) {
            if (word1.empty()) {
                lM += word2;
                word2.clear();
            }
            else if (word2.empty()) {
                lM += word1;
                word1.clear();
            }
            else {
                if (word1 > word2) {
                    lM += word1[0];
                    word1.erase(0, 1);
                }
                else {
                    lM += word2[0];
                    word2.erase(0, 1);
                }
            }
        }
        return lM;
    }
};
