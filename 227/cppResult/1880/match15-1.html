class Solution {
public:
    string largestMerge(string word1, string word2) {

      int n = word1.length(), m = word2.length();
      int l = 0, r = 0;
      // abd 
      // abc 
      string res;
      while (l < n && r < m) {
        if (word1[l] > word2[r]) {
          res += word1[l++];
        } else if (word2[r] > word1[l]) {
          res += word2[r++];
        } else {
          bool found = false;

          string s1 = word1.substr(l);
          string s2 = word2.substr(r);
          if (s1 > s2) {
            res += word1[l++];
          } else {
            res += word2[r++];
          }
        }
      }
      while (l < n) {
        res += word1[l++];
      }
      while (r < m) {
        res += word2[r++];
      }
      return res;
    }
};
