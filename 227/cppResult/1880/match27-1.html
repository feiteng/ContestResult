class Solution {
public:
    string largestMerge(string word1, string word2) {

        string result = "";
        int i = 0, j = 0;
        
        while(i < word1.size() && j < word2.size()) {
            if(word1[i] > word2[j]) {
                result += word1[i++];
            } else if(word1[i] < word2[j]) {
                result += word2[j++];
            } else {
                if(word1.substr(i) > word2.substr(j)) {
                    result += word1[i++];
                } else {
                    result += word2[j++];
                }
            }
        }
        
        if(i < word1.size()) {
            while(i < word1.size()) {
                result += word1[i++];
            }
        }
        
        if(j < word2.size()) {
            while(j < word2.size()) {
                result += word2[j++];
            }
        }
        
        return result;
    }
};
