class Solution {
public:
    string largestMerge(string word1, string word2) {
        int n = word1.size(), m = word2.size();
        
        int i=0, j=0;
        
        string res = "";
        while(i<n && j<m) {
            if(word1[i] == word2[j]) {
                int x=0;
                while(i+x<n && j+x<m && word1[i+x] == word2[j+x]) x++;
                
                if(i+x >= n) res += word2[j++];
                else if(j+x >= m) res += word1[i++];
                else if(word1[i+x] > word2[j+x]) res += word1[i++];
                else res += word2[j++];
            }

            else if(word1[i] > word2[j]) {
                res += word1[i++];
            }
            else {
                res += word2[j++];
            }
        }
        
        while(i<n) {
            res += word1[i++];
        }
        while(j<m) {
            res += word2[j++];
        }
        
        
        return res;
    }
};
