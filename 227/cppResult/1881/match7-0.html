#define SZ(x) ((int)(x).size())
#define all(x) (x).begin(), (x).end()
#define rep(i, l, r) for (int i = l; i <= r; ++i)
#define per(i, l, r) for (int i = l; i >= r; --i)

#define INF 0x3f3f3f3f

class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
      int n = SZ(nums);
      int k = n / 2;
      vector<int> a,b;
      for(int i = 0 ; i < (1 << k) ; i++){
        int sum = 0;
        for(int j = 0 ; j < k ; j++) if((i >> j) & 1) sum += nums[j];
        a.push_back(sum);
      }
      int z = n - k;
      for(int i = 0 ; i < (1 << z) ; i++){
        int sum = 0;
        for(int j = 0 ; j < z ; j++) if((i >> j) & 1) sum += nums[k + j];
        b.push_back(sum);
      }
      sort(all(a));
      sort(all(b));
      int ans = INF;
      for(int i = 0 ; i < a.size() ; i++){
        int z1 = goal - a[i];
        int pos1 = lower_bound(all(b),z1) - b.begin();
        for(int j = max(0,pos1 - 2); j < min((int)b.size(),pos1 + 2) ; j++) {
          int res = abs((a[i] + b[j]) - goal);
          ans = min(ans,res);
        }
      }
      return ans;
    }
};
