class Solution {
public:
    int n,m,a[22],b[22];
    int buf[1100000],tot;
    
    int minAbsDifference(vector<int>& nums, int goal) {
        int n=nums.size()/2,m=nums.size()-n;
        for(int i=0;i<n;++i) a[i]=nums[i];

        for(int i=0;i<m;++i) b[i]=nums[i+n];
        
        for(int s=0;s<(1<<m);++s){
            int sum=0;
            for(int i=0;i<m;++i){
                if(s>>i&1) sum+=b[i];
            }
            buf[tot++]=sum;
        }
        sort(buf,buf+tot);

        tot=unique(buf,buf+tot)-buf;
        
        int ans=INT_MAX;
        for(int s=0;s<(1<<n);++s){
            int x=0;
            for(int i=0;i<n;++i){
                if(s>>i&1) x+=a[i];
            }
            int y=goal-x;
            int pos=lower_bound(buf,buf+tot,y)-buf;
            if(pos<tot){
                ans=min(ans,abs(x+buf[pos]-goal));
            }
            if(pos>0){
                ans=min(ans,abs(x+buf[pos-1]-goal));
            }
        }
        return ans;
    }
};
