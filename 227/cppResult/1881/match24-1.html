class Solution {
public:
    int minAbsDifference(vector<int>& v, int goal) {
        set<int>s;

        int n=v.size(),nl=n/2,nr=n-nl;
        for(int i=0; i<1<<nl; i++){
            int now = 0;
            for(int j=0; j<21; j++)if(i&(1<<j)){
                now += v[j];
            }
            s.insert(now);
        }

        int ans = 1e9+7;
        for(int i=0; i<1<<nr; i++){
            int now = 0;
            for(int j=0; j<21; j++)if(i&(1<<j)){
                now += v[j+nl];
            }
            auto it=s.lower_bound(goal-now);
            if(it!=s.end()){
                if(*it==goal-now)return 0;
                ans = min(ans, *it-(goal-now));
            }
            if(it!=s.begin()){
                it--;
                ans = min(ans, (goal-now)-*it);
            }
        }
        return ans;
    }
};
