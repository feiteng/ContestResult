vector<int> calc(vector<int> a) {
    vector<int> res;

    for (int m = 0; m < (1 << a.size()); ++ m) {
        int s = 0;
        for (int i = 0; i < a.size(); ++ i) {
            if (m >> i & 1) {
                s += a[i];
            }
        }
        res.push_back(s);
    }
    sort(res.begin(), res.end());
    return res;
}

class Solution {
public:
    int minAbsDifference(vector<int>& A, int s) {
        int B = A.size() / 2;
        vector<int> a, b;
        for (int i = 0; i < B; ++ i) a.push_back(A[i]);
        for (int i = B; i < A.size(); ++ i) b.push_back(A[i]);
        auto t1 = calc(a);
        auto t2 = calc(b);

        int ans = 2e9;
        for (int x : t1) {
            int i = upper_bound(t2.begin(), t2.end(), s - x) - t2.begin();
            if (i > 0) {
                ans = min(ans, abs(t2[i - 1] + x - s));
            }
            if (i < t2.size()) {
                ans = min(ans, abs(t2[i] + x - s));
            }
        }
        return ans;
    }
};
