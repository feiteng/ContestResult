class Solution {
public:
	int minAbsDifference(vector<int>& nums, int goal) {
		int n = nums.size();
		vector<int>a,b;
		for (int i = 0; i < n; i++) {
			if (i < n / 2)a.push_back(nums[i]);
			else b.push_back(nums[i]);
		}
		vector<int>x, y;
		int na = a.size(), nb = b.size();
		for (int i = 0; i < (1 << na); i++) {
			int sum = 0;
			for (int j = 0; j < na; j++) {
				if (i & (1 << j))sum += a[j];
			}
			x.push_back(sum);
		}
		for (int i = 0; i < (1 << nb); i++) {
			int sum = 0;
			for (int j = 0; j < nb; j++) {
				if (i & (1 << j))sum += b[j];
			}
			y.push_back(sum);
		}
		sort(x.begin(), x.end());
		sort(y.begin(), y.end());

		int minn = 2e9;
		int ret = -1;
		for (int i = 0; i < x.size(); i++) {
			int need = goal - x[i];
			auto it = upper_bound(y.begin(), y.end(), need - 1);
			if (it != y.end()) {
				int val = y[it - y.begin()];
				if (abs(goal - (x[i] + val)) < minn) {
					ret = x[i] + val;
					minn = min(minn, abs(goal - ret));
				}
			}
			if (it != y.begin()) {
				it--;
				int val = y[it - y.begin()];
				if (abs(goal - (x[i] + val)) < minn) {
					ret = x[i] + val;
					minn = min(minn, abs(goal - ret));
				}

			}
		}
		return minn;
	}
};
