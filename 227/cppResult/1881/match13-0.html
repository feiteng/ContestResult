int s[1<<20];
class Solution {
public:
    int find(int l, int r, int x){
        int mid;
        while(l <= r){
            mid = (l + r) >> 1;
            if(s[mid] <= x) l = mid + 1;
            else r = mid - 1;
        }
        return r;
    }
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        if (n == 1) {
            return max(goal - nums[0], nums[0] - goal);
        }
        int m = n / 2;
        for (int i = 0; i < (1<<m); ++i) {
            int t = 0;
            for (int j = 0; j < m; ++j) {
                if (i >> j & 1) t += nums[j];
            }
            s[i] = t;
        }
        sort(s, s + (1<<m));
        int w = n - m;
        int ans = 2e9;
        for (int i = 0; i < (1<<w); ++i) {
            int t = 0;
            for (int j = 0; j < w; ++j) {
                if (i >> j & 1) t += nums[n - j - 1];
            }
            int need = goal - t;
            int idx = find(0, (1<<m) - 1, need);
            if (idx >= 0) {
                ans = min(ans, max(need - s[idx], s[idx] - need));
            }
            if (idx + 1 < (1<<m)) {
                ans = min(ans, max(need - s[idx+1], s[idx+1] - need));
            }
        }
        return ans;
    }
};
