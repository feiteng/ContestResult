class Solution {
    set<int> f(vector<int> a) {
        set<int> r;

        for (int i = 0; i < (1 << a.size()); ++i) {
            int s = 0;
            for (int j = 0; j < a.size(); ++j) {
                if ((i >> j) & 1) s += a[j];
            }
            r.insert(s);
        }
        return r;
    }
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int ans = abs(goal);
        ans = min(ans, abs(goal - nums[0]));
        auto it = nums.begin() + nums.size() / 2;
        vector<int> a(nums.begin(), it);
        vector<int> b(it, nums.end());

        // auto sa = f(a);
        auto sb = f(b);


        for (int i = 0; i < (1 << a.size()); ++i) {
            int s = 0;
            for (int j = 0; j < a.size(); ++j) {
                if ((i >> j) & 1) s += a[j];
            }
            int x = s;
            int diff = goal - x;

            auto it = sb.lower_bound(diff);
            if (it != sb.end()) ans = min(ans, abs(x + *it - goal));
            if (it != sb.begin()) {
                --it;
                ans = min(ans, abs(x + *it - goal));
            }
            if (ans == 0) return ans;
        }
        // for (int x : sa) {
        //     int diff = goal - x;
        //     auto it = sb.lower_bound(diff);
        //     if (it != sb.end()) ans = min(ans, abs(x + *it - goal));
        //     if (it != sb.begin()) {
        //         --it;
        //         ans = min(ans, abs(x + *it - goal));
        //     }
        // }
        return ans;
    }
};
