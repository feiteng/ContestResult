class Solution {
public:
    int minAbsDifference(vector<int>& a, int goal) {
        int n = a.size();
        int m = n / 2;
        vector<int> v;
        for (int mask = 0; mask < (1 << m); mask++) {
            int cur = 0;
            for (int bit = 0; bit < m; bit++) if ((mask >> bit) & 1) cur += a[bit];
            v.push_back(cur);
        }
        sort(v.begin(), v.end());

        int k = n - m;
        int ans = INT_MAX;
        for (int mask = 0; mask < (1 << k); mask++) {
            int cur = 0;
            for (int bit = 0; bit < k; bit++) if ((mask >> bit) & 1) cur += a[n / 2 + bit];
            
            int l = goal - cur;
            int p = lower_bound(v.begin(), v.end(), l) - v.begin();
            if (p >= 0 && p < v.size()) ans = min(ans, abs(goal - cur - v[p]));
            p--;
            if (p >= 0 && p < v.size()) ans = min(ans, abs(goal - cur - v[p]));
        }
        return ans;
    }
};
