class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n=nums.size();int k=n/2;
        vector<int>possible;
        for(int i=0;i<(1<<k);i++){
            int temp=0;
            for(int j=0;j<k;j++){
                if((i&(1<<j))>=1)
                    temp=temp+nums[j];
            }
            possible.push_back(temp);
        }
        k=n-k;sort(possible.begin(),possible.end());
        int min_diff=INT_MAX;
        for(int i=0;i<(1<<k);i++){
            int temp=0;
            for(int j=0;j<k;j++){
                if((i&(1<<j))>=1)
                    temp=temp+nums[j+(n/2)];
            }
            int req_sum=goal-temp;
            int pos=lower_bound(possible.begin(),possible.end(),req_sum)-possible.begin();
            if(pos==possible.size())pos--;
            if(possible[pos]==req_sum)min_diff=0;
            else{
                min_diff=min(min_diff,abs(possible[pos]+temp-goal));
                if(pos-1>=0){
                    pos=pos-1;
                    min_diff=min(min_diff,abs(possible[pos]+temp-goal));
                }
            }
        }
        return min_diff;
    }
};
