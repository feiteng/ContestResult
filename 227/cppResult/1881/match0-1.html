class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        if (nums.size() == 1) {
            return min(abs(goal), abs(nums[0] - goal));
        }

        int n = nums.size(), m = n / 2;
        vector<int> vec0, vec1;
        for (int st = 0; st < (1 << m); ++st) {
            int sum = 0;
            for (int k = 0; k < m; ++k)
                if (st >> k & 1)
                    sum += nums[k];
            vec0.push_back(sum);
        }
        for (int st = 0; st < (1 << (n - m)); ++st) {
            int sum = 0;
            for (int k = 0; k < n - m; ++k)
                if (st >> k & 1)
                    sum += nums[m + k];
            vec1.push_back(sum);
        }
        sort(vec0.begin(), vec0.end());
        sort(vec1.begin(), vec1.end());
        /*
        for (auto x : vec0)
            cout << x << " ";
        cout << endl;
        for (auto x : vec1)
            cout << x << " ";
        */
        int ans = abs(goal);
        for (int x : vec0) {

            int pos = lower_bound(vec1.begin(), vec1.end(), goal - x) - vec1.begin();
            if (pos > 0) ans = min(ans, abs(x + vec1[pos - 1] - goal));
            if (pos < vec1.size()) ans = min(ans, abs(x + vec1[pos] - goal));
            if (pos + 1 < vec1.size()) ans = min(ans, abs(x + vec1[pos + 1] - goal));
            
        }
        return ans;
    }
};
