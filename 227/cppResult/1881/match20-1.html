class Solution {
 public:
  int minAbsDifference(vector<int>& nums, int goal) {
    int n = (int) nums.size();
    int m = (n >> 1);
    set<int> st;

    for (int i = 0; i < (1 << m); i++) {
      int s = 0;
      for (int j = 0; j < m; j++) 
        if (i & (1 << j)) 
          s += nums[j];
      st.insert(s);
    }
    int ans = INT_MAX;
    for (int i = 0; i < (1 << (n - m)); i++) {
      int cur = 0;
      for (int j = 0; j < (n - m); j++) 
        if (i & (1 << j)) 
          cur += nums[m + j];
      int x = goal - cur;
      auto target = st.upper_bound(x);
      if (target != st.end()) {
        int val = cur + *target;
        ans = min(ans, abs(goal - val));
      }
      if (target != st.begin()) {
        --target;
        int val = cur + *target;
        ans = min(ans, abs(goal - val));
      }
    }
    return ans;
  }
};
