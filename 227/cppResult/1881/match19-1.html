#define ll long long
#define MIN(x,y) ((x)<(y)?(x):(y))


class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int N = nums.size();
        
        int N1 = N/2;
        int N2 = N - N1;
        
        vector<ll> arr1,arr2;

        for(int i=0;i<(1<<N1);i++){
            ll sum = 0;
            for(int j=0;j<N1;j++){
                if(i & (1<<j))
                    sum += nums[j];
            }
            arr1.push_back(sum);
        }
        
        for(int i=0;i<(1<<N2);i++){
            ll sum = 0;
            for(int j=0;j<N2;j++){
                if(i & (1<<j))
                    sum += nums[j+N1];
            }
            arr2.push_back(sum);
        }
        
        sort(arr1.begin(),arr1.end());
        sort(arr2.begin(),arr2.end());
        
        int M = arr1.size();
        N = arr2.size();
        /*
        for(auto x:arr1)
            cout<<x<<" ";
        cout<<"\n";
        
        for(auto x:arr2)
            cout<<x<<" ";
        cout<<"\n";*/
        
        int ans = 2000000000;
        
        for(int i=0;i<M;i++){
            ll lo,ho,mid;
            
            lo = 0;
            ho = N;
            
            while(lo < ho){
                mid = (lo + ho)/2;
                
                if(arr1[i] + arr2[mid] > goal)
                    ho = mid;
                else
                    lo = mid+1;
            }
            
            //cout<<i<<" : "<<lo<<"\n";
            if(lo > 0 and lo <= N)
                ans = MIN(ans,abs(goal - arr1[i] - arr2[lo-1]));
            
            if(lo < N)
            ans = MIN(ans,abs(goal - arr1[i] - arr2[lo]));
        }
        
        return ans;
    }
};
