class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
      int n = nums.size(), m = (n >> 1);
      set<int> s1;
      for (int i = 0; i < (1 << m); i++) {
        int s = 0;
        for (int j = 0; j < m; j++) {
          if (i & (1 << j)) {
            s += nums[j];
          }
        }
        s1.insert(s);
      }
      int rem = n - m;
      int res = numeric_limits<int>::max();
      for (int i = 0; i < (1 << rem); i++) {
        int cur = 0;
        for (int j = 0; j < rem; j++) {
          if (i & (1 << j)) {
            cur += nums[m + j];
          }
        }
        int need = goal - cur;
        auto ub = s1.upper_bound(need);
        if (ub != s1.end()) {
          int here = cur + *ub;
          res = min(res, abs(goal - here));
        }
        if (ub != s1.begin()) {
          --ub;
          int here = cur + *ub;
          res = min(res, abs(goal - here));
        }
      }
      return res;
    }
};
