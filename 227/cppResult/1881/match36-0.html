class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        if (n == 1) {
            return abs(goal - nums[0]);
        }
        
        int l = n / 2;
        int r = n - l;
        int L = 1 << l;
        int R = 1 << r;
        
        vector<int> LA;
        LA.reserve(L);
        vector<int> RA;
        RA.reserve(R);
        
        int res = 2e9;
        for (int i = 0; i < L; ++i) {
            int sum = 0;
            for (int j = 0; j < l; ++j) {
                if ((i >> j) & 1) {
                    sum += nums[j];
                }
            }
            res = min(res, abs(sum - goal));
            LA.push_back(sum);
        }
        for (int i = 0; i < R; ++i) {
            int sum = 0;
            for (int j = 0; j < r; ++j) {
                if ((i >> j) & 1) {
                    sum += nums[l + j];
                }
            }
            res = min(res, abs(sum - goal));
            RA.push_back(sum);
        }
        
        sort(LA.begin(), LA.end());
        LA.resize(unique(LA.begin(), LA.end()) - LA.begin());
        sort(RA.begin(), RA.end());
        RA.resize(unique(RA.begin(), RA.end()) - RA.begin());
        
        for (int i = 0, j = R - 1; i < L; ++i) {
            while (j >= 0 && LA[i] + RA[j] > goal) {
                --j;
            }
            if (j >= 0) {
                res = min(res, abs(LA[i] + RA[j] - goal));
            }
            if (j + 1 < R) {
                res = min(res, abs(LA[i] + RA[j + 1] - goal));
            }
        }
        return res;
    }
};
