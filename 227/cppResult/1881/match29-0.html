class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        int p = n / 2;
        int t = n - p;
        vector<int> p1, p2;
        for (int i = 0; i < (1 << p); i++) {
            int temp = 0;
            for (int j = 0; j < p; j++) {
                if (i & (1 << j)) {
                    temp += nums[j];
                }
            }
            p1.push_back(temp);
        }
        for (int i = 0; i < (1 << t); i++) {
            int temp = 0;
            for (int j = 0; j < t; j++) {
                if (i & (1 << j)) {
                    temp += nums[p + j];
                }
            }
            p2.push_back(temp);
        }
        sort(p1.begin(), p1.end());
        sort(p2.begin(), p2.end());
        int ans = abs(goal);
        int id1 = 0, id2 = p2.size() - 1;
        while (id1 < p1.size() && id2 >= 0) {
            int ss = p1[id1] + p2[id2];
            ans = min(ans, abs(goal - ss));
            if (ss < goal) {
                id1++;
            }
            else {
                id2--;
            }
        }
        return ans;
    }
};
