class Solution {
public:
    int minAbsDifference(vector<int>& a, int goal) {
        int n = a.size();
        int L = n / 2, R = n - L;
        set<int> b;
        for (int k = 0; k < (1 << L); ++k) {
            int sum = 0;
            for (int i = 0; i < L; ++i) {
                if ((k >> i) & 1) sum += a[i];
            }
            b.insert(sum);
        }
        int ret = 2e9;
        for (int k = 0; k < (1 << R); ++k) {
            int sum = 0;
            for (int i = 0; i < R; ++i) {
                if ((k >> i) & 1) sum += a[L + i];
            }
            int target = goal - sum;
            auto it = b.lower_bound(target);
            if (it != b.end()) ret = min(ret, abs(sum + *it - goal));
            if (it != b.begin()) {
                --it;
                ret = min(ret, abs(sum + *it - goal));
            }
        }
        return ret;
    }
};
