class Solution {
public:
    int minAbsDifference(vector<int>& nums, int g) {
        vector<int> a,b;
        int n=nums.size();
        for(int i=0;i<n/2;i++)
        {
        a.push_back(nums[i]);
        }
        for(int i=n/2;i<n;i++)
        {
        b.push_back(nums[i]);
        }
        int x=a.size(),y=b.size();
        vector<int> p,q;
        for(int i=0;i<(1<<x);i++)
        {
        int sm=0;
            for(int o=0;o<x;o++)
            {
            if((i & (1<<o)))
            {
            sm+=a[o];
            }
            }
            p.push_back(sm);
        }
        sort(p.begin(),p.end());
        for(int i=0;i<(1<<y);i++)
        {
        int sm=0;
            for(int o=0;o<y;o++)
            {
            if((i & (1<<o)))
            {
            sm+=b[o];
            }
            }
            q.push_back(sm);
        }
        sort(q.begin(),q.end());
        int ans=1e9;
        for(int i=0;i<p.size();i++)
        {
        int ps=upper_bound(q.begin(),q.end(),g-p[i])-q.begin();
            if(ps<q.size())
            {
            ans=min(ans,q[ps]+p[i]-g);
            }
            if(ps-1>=0)
            {
            ans=min(ans,g-q[ps-1]-p[i]);
            }
            
        }
        return ans;
    }
};
