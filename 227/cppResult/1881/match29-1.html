class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int ans=abs(goal);
        int n=nums.size();
        int m=n/2;
        vector<int> v;
        for ( int s=0; s<1<<m; s++ ) {
            int sum=0;
            for ( int i=0; i<m; i++ ) {
                if ( s>>i&1 ) sum+=nums[i];
            }
            ans=min(ans,abs(sum-goal));
            v.push_back(sum);

        }
        sort(v.begin(),v.end());
        for ( int s=0; s<1<<(n-m); s++ ) {
            int sum=0;
            for ( int i=0; i<(n-m); i++ ) {
                if ( s>>i&1 ) sum+=nums[i+m];
            }
            ans=min(ans,abs(sum-goal));
            {
                auto it=lower_bound(v.begin(),v.end(),goal-sum);
                if ( it!=v.end() ) {
                    ans=min(ans,abs(sum+*it-goal));
                }
                if ( it!=v.begin() ) {
                    --it;
                    ans=min(ans,abs(sum+*it-goal));
                }
            }
            // {
            //     auto it=lower_bound(v.begin(),v.end(),-goal+sum);
            //     if ( it!=v.end() ) {
            //         ans=min(ans,abs(sum+*it-goal));
            //     }
            //     if ( it!=v.begin() ) {
            //         --it;
            //         ans=min(ans,abs(sum+*it-goal));
            //     }
            // }
        }
        return ans;
    }
};
