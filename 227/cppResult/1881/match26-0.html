class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int i,n=nums.size(),res=abs(goal),m=n>>1;
        int N=1<<m,M=1<<(n-m);
        set<int>s;
        for(i=0;i<N;++i){
            int sum = 0;
            for(int j=0;j<m;++j){
                if((1<<j)&i)sum += nums[j];
            }
            //cout<<sum<<" "<<goal;
            if(sum==goal)return 0;
            res = min(res,abs(sum-goal));
            s.insert(sum);
        }
        for(i=0;i<M;++i){
            int sum = 0;
            for(int j=m;j<n;++j){
                if((1<<(j-m))&i)sum += nums[j];
            }
            auto p = s.lower_bound(goal-sum);
            if(p!=s.end())res = min(res,abs(goal-sum-*p));
            if(p!=s.begin()){
                p--;
                res = min(res,abs(goal-sum-*p));
            }
            //cout<<sum<<" "<<res<<endl;
            if(res==0)return 0;
        }
        return res;
    }
};
