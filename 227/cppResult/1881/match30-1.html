#pragma GCC optimize("Ofast")
#pragma GCC optimization("unroll-loops")
#pragma GCC optimize("unroll-loops")
#pragma GCC optimize("fast-math")
#pragma GCC optimize("no-stack-protector")


class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        vector<int> v;
        for(int i=0;i<(1 << (n/2));i++){
            int sum = 0;
            for(int j=0;j<n/2;j++){
                if((i >> j) & 1){
                    sum += nums[j];
                }
            }
            v.push_back(sum);
        }
        sort(v.begin(),v.end());
        int left = n - n/2;
        int ans = 1e9;

        for(int i=0;i<(1 << left);i++){
            int sum = 0;
            for(int j=0;j<left;j++){
                if((i >> j) & 1){
                    sum += nums[n/2+j];
                }
            }
            int req = sum - goal;
            int idx = upper_bound(v.begin(),v.end(),-req) - v.begin() - 1;
            if(idx >= 0){
                ans = min(ans,abs(sum+v[idx] - goal));
            }
            if(idx+1 < v.size()){
                ans = min(ans,abs(sum+v[idx+1] - goal));
            }
        }
        return ans;
    }
};
