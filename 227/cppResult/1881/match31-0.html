class Solution {
public:
    set<int>s;
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        int tp = 1<<(n/2),hf = n-n/2;
        for(int i=0;i<tp;i++){
            int sum = 0;
            for(int j=0;j<n/2;j++)if(i>>j&1){
                sum+=nums[j];
            }
            s.insert(sum);
        }
        tp = 1<<hf;
        int ans = 1e9;
        for(int i=0;i<tp;i++){
            int sum = 0;
            for(int j=0;j<hf;j++)if(i>>j&1){
                sum += nums[j+n/2];
            }
            auto it = s.lower_bound(goal-sum);
            if(it!=s.end())ans = min(ans,abs(*it+sum-goal));
            if(it!=s.begin()){
                --it;
                ans = min(ans,abs(*it+sum-goal));
            }
        }
        return ans;
    }
};
