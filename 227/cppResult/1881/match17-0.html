class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        //TODO: n=1;
        int n=nums.size();
        vector<int> left, right;
        int p1=n/2, p2=n-p1;
        
        left.push_back(0);
        for(int i=0; p1 && i<(1<<p1); ++i){
            int sum=0;
            for(int j=0; j<p1; ++j){
                if(i&(1<<j)) sum+=nums[j];
            }
            left.push_back(sum);
        }
        
        for(int i=0; i<(1<<p2); ++i){
            int sum=0;
            for(int j=0; j<p2; ++j){
                if(i&(1<<j)) sum+=nums[p1+j];
            }
            right.push_back(sum);
        }
        
        sort(right.begin(), right.end());
                
        int res=INT_MAX;
        for(int v:left){
            vector<int>::iterator it=lower_bound(right.begin(), right.end(), goal-v);
            if(it!=right.end()) res=min(res, abs(v+*it-goal));
            if(it!=right.begin()){
                it--;
                res=min(res, abs(v+*it-goal));
            }
        }
        
        return res;
    }
};
