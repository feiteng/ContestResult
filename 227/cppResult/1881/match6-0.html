class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        int t = n>>1; 
        vector<int> A, B;
        int N = 1<<t;
        for(int x = 0; x < N; x++)
        {
            int s = 0;
            for(int i = 0; i < t; i++) if(x>>i&1)
                s += nums[i];
            A.push_back(s);
        }
        N = 1<<(n-t);
        for(int x = 0; x < N; x++)
        {
            int s = 0;
            for(int i = t; i < n; i++) if(x>>(i-t)&1)
                s += nums[i];
            B.push_back(s);
        }
        sort(B.begin(), B.end());
        int diff = 0x7fffffff;
        n = B.size();
        for(int x : A)
        {
            int y = goal-x;
            int i = lower_bound(B.begin(), B.end(), y) - B.begin();
            if(i) diff = min(diff, abs(B[i-1]-y));
            if(i < n) diff = min(diff, abs(B[i]-y));
            if(!diff) return 0;
        }
        return diff; 
    }
};
