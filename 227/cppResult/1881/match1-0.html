const int N = 45, INF = ~0U >> 2;

class Solution {
public:
    int minAbsDifference(vector<int>& a, int g) {
        int n = a.size(), m = n / 2;
        vector<int> f, h;
        for (int st = 1; st < (1 << m); st++) {
            int tmp = 0;
            for (int i = 0; i < m; i++) if ((st >> i) & 1) tmp += a[i];
            f.push_back(tmp);
        }
        for (int st = 1; st < (1 << (n - m)); st++) {
            int tmp = 0;
            for (int i = 0; i < n - m; i++) if ((st >> i) & 1) tmp += a[i + m];
            h.push_back(tmp);
        }
        int ans = abs(g);
        sort(h.begin(), h.end());
        for (auto x: f) ans = min(ans, abs(g - x));// , cout << x << ' ';
        // puts("");
        for (auto x: h) ans = min(ans, abs(g - x));// , cout << x << ' ';
        // puts("");
        for (auto x: f) {
            int id = lower_bound(h.begin(), h.end(), g - x) - h.begin();
            // cout << x << ' ' << id << '\n';
            if (id < h.size()) ans = min(ans, abs(g - (x + h[id])));
            if (id + 1 < h.size()) ans = min(ans, abs(g - (x + h[id + 1])));
            if (id - 1 >= 0) ans = min(ans, abs(g - (x + h[id - 1])));
        }
        return ans;
    }
};
