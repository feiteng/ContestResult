class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        set<int> s;
        int n = nums.size();
        int bits = n / 2;
        int ans = 1e9 + 5;
        

        for(int i = 0;i < (1 << bits);i++) {
            int now = 0;
            for(int b = 0; b < bits;b++) {
                if((i >> b) & 1) {
                    now += nums[b];
                }
            }
            s.insert(now);
        }
        int left = n - bits;

        for(int i = 0;i < (1 << left);i++) {
            int now = 0;
            for(int b = 0;b < left;b++) {
                if((i >> b) & 1) {
                    now += nums[b + bits];
                }
            }
            int target = goal - now;
            auto iter = s.upper_bound(target);
            if(iter != s.end())
                ans = min(ans, abs(*iter + now - goal));
            if(iter != s.begin()) {
                iter--;
                ans = min(ans, abs(*iter + now - goal));
            }
        }
        return ans;
    }
};
