class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int N = nums.size();
        vector<int> v1, v2;
        for(int i = 0; i < N/2; i++){
            v1.push_back(nums[i]);
        }
        for(int i = N/2; i < N; i++){
            v2.push_back(nums[i]);
        }
        
        vector<int> p1, p2;
        genAllPossible(v1, p1);
        genAllPossible(v2, p2);
        
        sort(p2.begin(), p2.end()); 
        
        int res = 2000000001;
        for(int n: p1){
            int ptr = lower_bound(p2.begin(), p2.end(), goal - n) - p2.begin();
            if(ptr < p2.size()){
                res = min(res, abs(p2[ptr] + n - goal));
            }
            if(ptr > 0){
                res = min(res, abs(p2[ptr-1] + n - goal));
            }
        }
        
        return res;
    }
    
    void genAllPossible(vector<int>& origin, vector<int>& allP){
        int N = origin.size();
        for(int x = 0; x < (1 << N); x++){
            int sum = 0;
            for(int i = 0; i < N; i++){
                if((x & (1 << i)) > 0) sum += origin[i];
            }
            allP.push_back(sum);
        }
    }
};
