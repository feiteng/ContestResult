class Solution {
public:
    vector<int> a, b;
    int n;
    int x, y;
    
    int minAbsDifference(vector<int>& nums, int goal) {
        n = nums.size();

        x = n / 2, y = n - n / 2;
        
        for (int i = 0; i < (1 << x); i++) {
            int sum = 0;
            for (int j = 0; j < x; j++) {
                if (i >> j & 1) sum += nums[j];
            }
            a.push_back(sum);
        }      

        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
        
        for (int i = 0; i < (1 << y); i++) {
            int sum = 0;
            for (int j = 0; j < y; j++) {
                if (i >> j & 1) sum += nums[x + j];
            }
            b.push_back(sum);
        }
        sort(b.begin(), b.end());
        b.erase(unique(b.begin(), b.end()), b.end());
        
        
        int ptr = b.size() - 1;
        int ans = 2e9 + 10;
        // for (int i = 0; i < x; i++) cout << a[i] << " ";
        // cout << endl;
        //         for (int i = 0; i < y; i++) cout << b[i] << " ";
        // for (int i = 0; i < a.size(); i++) cout << a[i] << " "; cout << endl;
        // for (int i = 0; i < b.size(); i++) cout << b[i] << " "; cout << endl;
        
        for (int i = 0; i < a.size(); i++) {
            while (ptr >= 0 && b[ptr] + a[i] > goal) ptr -= 1;
            if (ptr >= 0) ans = min(ans, abs(b[ptr] + a[i] - goal));
            if (ptr + 1 >= 0 && ptr + 1 < b.size()) ans = min(ans, abs(b[ptr + 1] + a[i] - goal));
        }
        
        return ans;
    }
};
