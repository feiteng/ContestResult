class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        set<int> s1,s2;
        if(nums.size()<=20){

            int ans=1e9;
            for(int i = 0;i<(1<<nums.size());i++){
                int sum=0;
                for(int j = 0;j<nums.size();j++){
                    if(i&(1<<j))sum+=nums[j];
                }
                ans=min(ans,abs(goal-sum));
            }
            return ans;
        }
        else{
            int mid=nums.size()/2;

            for(int i = 0;i<(1<<mid);i++){
                int sum=0;
                for(int j = 0;j<mid;j++){
                    if(i&(1<<j))sum+=nums[j];
                }
                s1.insert(sum);
            }
            int ans=1e9;
            for(int i =0;i<(1<<(nums.size()-mid));i++){
                int sum=0;
                for(int j = 0;j+mid<nums.size();j++){
                    if(i&(1<<j))sum+=nums[mid+j];
                }
             //   s2.insert(sum);

                auto it2=s1.lower_bound(goal-sum);
                if(it2!=s1.end())ans=min(ans,sum+*it2-goal);
                if(it2!=s1.begin()){
                    it2--;
                    ans=min(ans,goal-sum-*it2);
                }
            }
            /*
            int ans=1e9;
            for(auto it:s1){
                auto it2=s2.lower_bound(goal-it);
                if(it2!=s2.end())ans=min(ans,it+*it2-goal);
                if(it2!=s2.begin()){
                    it2--;
                    ans=min(ans,goal-it-*it2);
                }
            }*/
            return ans;
        }
    }
};
