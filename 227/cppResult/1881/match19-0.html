class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        int mid = n / 2;
        //[0,mid-1],[mid,n-1]
        int lsz = mid,rsz = n - mid;
        
        typedef long long ll;
        vector<ll>L,R;
        
        for(int i = 0;i < 1 << lsz;i++){
            ll ans = 0;
            for(int j = 0;j < lsz;j++){
                if((i >> j) & 1)ans += nums[j];
            }
            L.push_back(ans);
        }
        
        for(int i = 0;i < 1 << rsz;i++){
            ll ans = 0;
            for(int j = 0;j < rsz;j++){
                if((i >> j) & 1)ans += nums[mid + j];
            }
            R.push_back(ans);
        }
        
        sort(L.begin(),L.end());
        sort(R.begin(),R.end());
        
        int nl = L.size(),nr = R.size();
        ll ans = 1e18;
        for(int i = 0;i < nl;i++){
            ll v = goal - L[i];
            int p = lower_bound(R.begin(),R.end(),v) - R.begin();
            
            for(int idx = p - 1;idx <= p + 1;idx++){
                if(idx < nr and idx >= 0)ans = min(ans, abs(L[i] + R[idx] - goal));
            }
        }
        return ans;
    }
};
