class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        vector<int>a,b;
        int n1 = (n-1)/2;
        for(int i=0; i<(1<<n1); ++i){
        	int sum = 0;
        	for(int j=0; j<n1; ++j)
        		if((1<<j) & i)
        			sum += nums[j];
        	a.push_back(sum);
        }

        int n2 = n-n1;
        for(int i=0; i<(1<<n2); ++i){
        	int sum = 0;
        	for(int j=0; j<n2; ++j)
        		if((1<<j) & i)
        			sum += nums[n1+j];
        	b.push_back(sum);
        }

        sort(b.begin(), b.end());
        int m = b.size();

        int ans = INT_MAX;
        for(auto x : a){
        	int idx = lower_bound(b.begin(), b.end(), goal-x) - b.begin();
        	if(idx != m)
        		ans = min(ans, abs(x+b[idx]-goal));
        	if(idx)
        		ans = min(ans, abs(x+b[idx-1]-goal));

        	idx = lower_bound(b.begin(), b.end(), -goal-x) - b.begin();
        	if(idx != m)
        		ans = min(ans, abs(x+b[idx]-goal));
        	if(idx)
        		ans = min(ans, abs(x+b[idx-1]-goal));
        }
        return ans;
    }
};
