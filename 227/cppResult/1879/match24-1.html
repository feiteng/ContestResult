class Solution {
public:
    int maximumScore(int a, int b, int c) {
		priority_queue<int> pri_que;
		pri_que.push(a);
		pri_que.push(b);
		pri_que.push(c);
		int ans = 0;
		while(pri_que.size() >= 2){
			int v = pri_que.top(); pri_que.pop();
			int u = pri_que.top(); pri_que.pop();
			v--, u--;
			ans++;
            if (u > 0)
			    pri_que.push(v);
			if (v > 0)
                pri_que.push(u);
		}
        return ans;
    }
};
