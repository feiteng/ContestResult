class Solution {
public:
    int maximumScore(int a, int b, int c) {
        int C[3]={a,b,c};
        int num=0;
        while(1) {
			sort(C,C+3);
			if(C[1]==0) break;
			C[1]--;
			C[2]--;
			num++;
		}
		return num;
    }
};
