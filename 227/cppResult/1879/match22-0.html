class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> q;
        q.push(a);
        q.push(b);
        q.push(c);
        int ans = 0;
        while (q.size() > 1){
            int num1 = q.top(); q.pop();
            int num2 = q.top(); q.pop();
            ++ans;
            --num1; --num2;
            if (num1 > 0) q.push(num1);
            if (num2 > 0) q.push(num2);
        }
        return ans;
    }
};
