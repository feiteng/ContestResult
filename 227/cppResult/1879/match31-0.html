class Solution {
public:
    int maximumScore(int a, int b, int c) {
        vector<int> A = {a, b, c};
        int ret = 0;
        while (true) {
            sort(A.rbegin(), A.rend());
            if (A[1] == 0) break;
            A[0]--;
            A[1]--;
            ++ret;
        }
        return ret;
    }
};
