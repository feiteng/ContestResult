class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> pq;
        pq.push(a);
        pq.push(b);
        pq.push(c);
        
        int ans = 0;
        while(pq.size() > 1){
            auto t1 = pq.top();
            pq.pop();
            auto t2 = pq.top();
            pq.pop();
            ans++, t1--, t2--;
            if(t1)  pq.push(t1);
            if(t2)  pq.push(t2);
        }
        
        return ans;
    }
};
