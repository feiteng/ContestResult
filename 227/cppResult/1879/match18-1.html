class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> pq;
        pq.push(a);
        pq.push(b);
        pq.push(c);
        int res=0;
        while (pq.size()>1){
            a=pq.top();
            pq.pop();
            b=pq.top();
            pq.pop();
            a--;
            b--;
            res++;
            if (a) pq.push(a);
            if (b) pq.push(b);
        }
        return res;
    }
};
