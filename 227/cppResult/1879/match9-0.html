class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> q;
        q.push(a);
        q.push(b);
        q.push(c);
        int ans=0;
        while(q.size()>1){
            int a=q.top();q.pop();
            int b=q.top();q.pop();
            a--;
            b--;
            if(a>0) q.push(a);
            if(b>0) q.push(b);
            ans++;
        }
        return ans;
    }
};
