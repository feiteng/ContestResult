class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> pq;
        pq.push(a);
        pq.push(b);
        pq.push(c);
        int ans = 0, top, top2;
        while(pq.size() >= 2) {
            top = pq.top();
            pq.pop();
            top2 = pq.top();
            pq.pop();
            ans++;
            top--; top2--;
            if(top)
                pq.push(top);
            if(top2)
                pq.push(top2);
        }
        return ans;
    }
};
