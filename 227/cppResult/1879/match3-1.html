class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> q;
        q.push(a);
        q.push(b);
        q.push(c);
        int res = 0;
        while(q.size() >= 2) {
            auto t1 = q.top();
            q.pop();
            auto t2 = q.top();
            q.pop();
            t1--;
            t2--;
            res += 1;
            if(t1) q.push(t1);
            if(t2) q.push(t2);
           //out << t1 << ' ' << t2 << '\n';
        }
        return res;
    }
};
