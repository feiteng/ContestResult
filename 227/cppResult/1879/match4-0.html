class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> q;
        q.emplace(a);
        q.emplace(b);
        q.emplace(c);
        int ret = 0;
        while(q.size() > 1) {
            auto a = q.top(); q.pop();
            auto b = q.top(); q.pop();
            --a; --b;++ret;
            if(a > 0)
                q.push(a);
            if(b > 0)
                q.push(b);
        }
        return ret;
    }
};
