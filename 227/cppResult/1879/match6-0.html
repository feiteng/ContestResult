class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int>q;
        q.push(a),q.push(b),q.push(c);
        int ans =0 ;
        while(q.size()>1){
            int x = q.top();
            q.pop();
            int y = q.top();
            q.pop();
            x--,y--;
            if(x)q.push(x);
            if(y)q.push(y);
            ans++;
        }
        return ans;
    }
};
