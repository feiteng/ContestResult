class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> que;
        que.push(a);
        que.push(b);
        que.push(c);
        int ans=0;
        while(que.size()>=2){
            int x=que.top();
            que.pop();
            int y=que.top();
            que.pop();
            ++ans;
            --x;
            --y;
            if(x) que.push(x);
            if(y) que.push(y);
        }
        return ans;
    }
};
