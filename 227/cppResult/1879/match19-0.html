class Solution {
public:
    int maximumScore(int a, int b, int c) {
        priority_queue<int> pq;
        pq.push(a);
        pq.push(b);
        pq.push(c);
        int ans = 0;
        while (pq.size() > 1) {
            int i = pq.top();
            pq.pop();
            int j = pq.top();
            pq.pop();
            --i;
            --j;
            ++ans;
            if (i != 0) pq.push(i);
            if (j != 0) pq.push(j);
        }
        return ans;
    }
};
