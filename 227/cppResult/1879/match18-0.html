class Solution {
public:
	int maximumScore(int a, int b, int c) {
		priority_queue<int> pq;
		pq.push(a);
		pq.push(b);
		pq.push(c);

		int ans = 0, aa, bb;
		while (pq.size() >= 2) {
			aa = pq.top();
			pq.pop();
			bb = pq.top();
			pq.pop();
			ans++;
			aa--;
			bb--;

			if (aa > 0) pq.push(aa);
			if (bb > 0) pq.push(bb);
		}
		return ans;
	}
};
