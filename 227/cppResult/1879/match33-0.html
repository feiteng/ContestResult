using namespace std;
using ll = long long;
using ld = double;
typedef pair<int, int> II;
typedef pair<II, int> III;
typedef complex<ld> cd;
typedef vector<cd> vcd;

class Solution {
public:
    int maximumScore(int a, int b, int c) {
        vector<int> d = {a, b, c};
        sort(ALL(d));
        if (d[0] + d[1] <= d[2]) return d[0] + d[1];
        return (d[0] + d[1] + d[2]) / 2;
    }
};
