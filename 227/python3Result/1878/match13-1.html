from typing import List
import bisect
class Solution:
    def check(self, nums: List[int]) -> bool:
        def increase(nums):
            for i in range(1, len(nums)):
                if nums[i] < nums[i - 1]:
                    return False
            return True

        loc = 0
        for i in range(1, len(nums)):
            if nums[i] < nums[i - 1]:
                loc = i
                break
        return increase(nums[loc:] + nums[:loc])
