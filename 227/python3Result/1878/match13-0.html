from typing import List
import bisect

class Solution:
    
    def check(self, nums: List[int]) -> bool:
        def _order(nums):
            for i in range(1, len(nums)):
                if nums[i] < nums[i - 1]:
                    return False
            return True

        loc = 0
        for i in range(1, len(nums)):
            if nums[i] < nums[i - 1]:
                loc = i
                break
        return _order(nums[loc:] + nums[:loc])
