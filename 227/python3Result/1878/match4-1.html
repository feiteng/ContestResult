class Solution:
    def check(self, nums: List[int]) -> bool:
        rotated = False
        
        for i in range(1, len(nums)):
            if nums[i] < nums[i-1]:
                if rotated: 
                    return False
                rotated = True
            
                
        return (not rotated) or (rotated and nums[-1] <= nums[0])
