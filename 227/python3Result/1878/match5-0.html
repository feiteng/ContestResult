class Solution:
    def check(self, nums: List[int]) -> bool:
        s = False
        # t = float('inf')
        for i in range(len(nums) - 1):
            if nums[i + 1] < nums[i]:
                if s:
                    return False
                s = True
                # t = nums[i]
        return nums[0] >= nums[-1] if s else True
