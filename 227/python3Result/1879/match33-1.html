class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        ans = 0
        pool = [-a, -b, -c]
        heapq.heapify(pool)
        while len(pool) > 1:
            x, y = -heapq.heappop(pool), -heapq.heappop(pool)
            ans += 1
            
if x - 1 > 0:
                heapq.heappush(pool, -(x - 1))
            if y - 1 > 0:
                heapq.heappush(pool, -(y - 1))
        return ans
