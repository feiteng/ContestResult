class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        maxHeap = []
        heapq.heappush(maxHeap, -a)
        heapq.heappush(maxHeap, -b)
        heapq.heappush(maxHeap, -c)
        score = 0
        while len(maxHeap) >= 2:
            top1 = -heapq.heappop(maxHeap)
            top2 = -heapq.heappop(maxHeap)
            score += 1
            if top1-1 > 0:
                heapq.heappush(maxHeap, 1-top1)
            if top2-1 > 0:
                heapq.heappush(maxHeap, 1-top2)
        return score
