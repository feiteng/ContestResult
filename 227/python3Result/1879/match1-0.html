class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        heap = []
        
        heapq.heappush(heap, -a)
        heapq.heappush(heap, -b)
        heapq.heappush(heap, -c)
        
        ans = 0
        
        while(len(heap) >= 2):
            x = heapq.heappop(heap)+1
            y = heapq.heappop(heap)+1
            
            ans += 1
            if x != 0:
                heapq.heappush(heap, x)
            if y != 0:
                heapq.heappush(heap, y)
        
        return ans
