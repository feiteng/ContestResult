class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        
        arr = [a, b, c]
        arr.sort()
        
        cnt = 0
        while arr[1] != 0:
            arr[1] -= 1
            arr[2] -= 1
            cnt += 1
            arr.sort()
        
        return cnt
