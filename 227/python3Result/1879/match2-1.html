class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        h = []
        
        heapq.heappush(h, -a)
        heapq.heappush(h, -b)
        heapq.heappush(h, -c)
        
        ans = 0
        while len(h) > 1:
            x = -heapq.heappop(h) - 1
            y = -heapq.heappop(h) - 1
            ans += 1
            if x > 0:
                heapq.heappush(h, -x)
            if y > 0:
                heapq.heappush(h, -y)
                
        return ans
