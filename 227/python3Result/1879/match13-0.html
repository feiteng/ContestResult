class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        heap = [-a, -b, -c]
        heapq.heapify(heap)
        res = 0
        while len(heap) >= 2:
            x, y = -heapq.heappop(heap), -heapq.heappop(heap)
            x -= 1 ; y -= 1
            res += 1
            if x:
                heapq.heappush(heap, -x)
            if y:
                heapq.heappush(heap, -y)
        return res
