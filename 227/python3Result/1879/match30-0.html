class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        
        nums = [-a, -b, -c]
        heapq.heapify(nums)
        ans = 0
        while len(nums) > 1:
            a = heapq.heappop(nums)
            b = heapq.heappop(nums)
            ans += 1
            if a < -1:
                heapq.heappush(nums, a + 1)
                
            if b <-1 :
                heapq.heappush(nums, b + 1)
        return ans
