class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        pq = [-a,-b,-c]
        heapq.heapify(pq)
        cnt = 0
        while len(pq) > 1:
            a, b = heapq.heappop(pq), heapq.heappop(pq)
            a+=1
            b+=1
            cnt+=1
            
if a: heapq.heappush(pq, a)
            if b: heapq.heappush(pq, b)
        return cnt
