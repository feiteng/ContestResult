from heapq import heappush, heappop, heapify

class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        res = 0
        heap = [-a, -b, -c]
        heapify(heap)
        
        while len(heap) > 1:
            res += 1
            
            v = heappop(heap) 
            #print(v)
            v += 1
            
            v2 = heappop(heap) 
            #print(v)
            v2 += 1
            if v != 0:
                heappush(heap, v)
            if v2 != 0:
                heappush(heap, v2)
        return res
