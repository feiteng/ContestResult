import heapq

class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        cnt = 0
        h = [-a,-b,-c]
        heapq.heapify(h)
        while len(h) > 1:
            a = -heapq.heappop(h)
            b = -heapq
.heappop(h)
            cnt += 1
            a -= 1
            b -= 1
            if a:
                heapq.heappush(h, -a)
            if b:
                heapq.heappush(h, -b)
        return cnt
