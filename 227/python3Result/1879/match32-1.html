import heapq


class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        lst = [-a,-b,-c]
        heapq.heapify(lst)
        score = 0
        while 1:
            i, j = heapq.heappop(lst), heapq.heappop(lst)
            if not j:
                break
            score += 1
            heapq.heappush(lst, i + 1)
            heapq.heappush(lst, j + 1)
        return score
