class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        heap = [-a, -b, -c]
        heapq.heapify(heap)
        score = 0
        while len(heap) >= 2:
            pile1, pile2 = heapq.heappop(heap) + 1, heapq.heappop(heap) + 1
            score += 1
            if pile1 != 0:
                heapq.heappush(heap, pile1)
            if pile2 != 0:
                heapq.heappush(heap, pile2)
        return score
