import heapq
class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        lst = [-a,-b,-c]
        heapq.heapify(lst)
        ret = 0
        while True:
            m, n = heapq.heappop(lst), heapq.heappop(lst)
            if n == 0:
                break
            ret += 1
            heapq.heappush(lst, m + 1)
            heapq.heappush(lst, n + 1)
        return ret
