class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        cnt = 0
        h = [-a, -b, -c]
        heapq.heapify(h)
        while len(h) > 1:
            i, j = heapq.heappop(h), heapq.heappop(h)
            i += 1
            j += 1
            cnt += 1
            if i != 0:
                heapq.heappush(h, i)
            if j != 0:
                heapq.heappush(h, j)
        return cnt
