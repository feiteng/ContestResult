class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        heap = [-a,-b,-c] 
        heapq.heapify(heap)
        s = 0
        while True:
            ma, mi = heapq.heappop(heap), heapq.heappop(heap)
            if mi == 0:
                break
            s += 1
            heapq.heappush(heap, ma+1)
            heapq.heappush(heap, mi+1)
        return s
