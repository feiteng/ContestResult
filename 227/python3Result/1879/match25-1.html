class Solution:
    def maximumScore(self, a: int, b: int, c: int) -> int:
        res = [a,b,c]
        res.sort()
        ans = 0
        while res[1] > 0:
            res[1] -= 1
            res[2] -= 1
            ans += 1
            res.sort()
            
        return ans
