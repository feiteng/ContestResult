import bisect
class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        #@lru_cache(None)
#         def backtrack(pos, currsum):
#             if pos >= n:
#                 return currsum
            
#             take = backtrack(pos+1, currsum+nums[pos])
#             leave = backtrack(pos+1, currsum)
#             #print(take)
#             if abs(take - goal) < abs(leave - goal):
#                 return take
#             else:
#                 return leave
            
        n = len(nums)
#         nums.sort()
#         return abs(backtrack(0, 0)-goal)
        def gen(i, e, cs):
            if i == e:
                return [cs]
            take = gen(i+1, e, cs+nums[i])
            leave = gen(i+1, e, cs)
            return take+leave
        
        f = gen(0, n//2, 0)
        s = gen(n//2, n, 0)
        #print(n//2)
        s.sort()
        res = abs(goal)
        #print(f, s)
        for i in range(len(f)):
            c1 = bisect.bisect_left(s, goal - f[i])
            #print(rem)
            #c2 = bisect.bisect_right(s, goal - f[i])
            #print(c1, c2)
            res = min(res, abs(goal - (f[i]+s[min(c1, len(s)-1)])), abs(goal - (f[i]+s[min(c1+1, len(s)-1)])), abs(goal - (f[i]+s[min(c1-1, len(s)-1)])))
            
        return res
