class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        res = float('inf')
        
        def dfs(idx, end, ans, total=0):
            ans.add(total)
            if idx >= end: return
            for i in range(idx, end):
                dfs(i+1, end, ans, total + nums[i])
        
        ans = set()
        mid = len(nums) // 2
        dfs(0, mid, ans)
        ans2 = set()
        dfs(mid, len(nums), ans2)
        ans2 = sorted(list(ans2))
        
        print(ans, ans2)
        
        for num in ans:
            idx1 = bisect.bisect_left(ans2, goal - num)
            idx2 = bisect.bisect_right(ans2, goal - num) - 1
            idx3 = bisect.bisect_left(ans2, goal + num)
            idx4 = bisect.bisect_right(ans2, goal + num) - 1
            if 0 <= idx1 < len(ans2):
                res 
= min(res, abs(num + ans2[idx1] - goal))
            if 0 <= idx2 < len(ans2):
                res = min(res, abs(num + ans2[idx2] - goal))
            if 0 <= idx3 < len(ans2):
                res = min(res, abs(num + ans2[idx3] - goal))
            if 0 <= idx4 < len(ans2):
                res = min(res, abs(num + ans2[idx4] - goal))
            
        return res
