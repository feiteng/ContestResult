class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        def getAllSums(nums):
            vset=set([0])
            for x in nums:
                for e in list(vset):
                    vset.add(x+e)
            return sorted(list(vset))
        n=len(nums)
        md=n//2
        p1sums=getAllSums(nums[:md])
        p2sums=getAllSums(nums[md:])
        m1=len(p1sums)
        m2=len(p2sums)
        i1=0
        i2=m2-1
        ans=abs(goal)
        while i1<m1 and i2>=0:
            diff=p1sums[i1]+p2sums[i2]-goal
            if diff==0:
                return 0
            ans=min(ans,abs(p1sums[i1]+p2sums[i2]-goal))
            if diff<0:
                i1+=1
            else:
                i2-=1
        return ans
