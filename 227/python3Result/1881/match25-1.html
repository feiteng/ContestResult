class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        s = {0}
        t = {0}
        n = len(nums)
        for i in range(n // 2):
            s |= {j + nums[i] for j in s}
        for i in range(n // 2, n):
            t |= {j + nums[i] for j in t}
        t = sorted(t)
        ret = abs(goal)
        m = len(t)
        for i in s:
            p = bisect_left
(t, goal - i)
            if p > 0:
                ret = min(ret, abs(goal - i - t[p - 1]))
            if p < m:
                ret = min(ret, abs(goal - i - t[p]))
        return ret
