class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        n = len(nums)
        ans = float("inf")
        def subset_sum(a):
            res = []
            for bit in range(1 << len(a)):
                cnt = 0
                for i in range(len
(a)):
                    if (1 << i) & bit:
                        cnt += a[i]
                res.append(cnt)
            return res
        first_half = subset_sum(nums[: (n + 1) // 2])
        second_half = subset_sum(nums[(n + 1) // 2:])
        second_half.sort()
        for s1 in first_half:
            idx = bisect.bisect_left(second_half, goal - s1)
            
if idx > 0:
                ans = min(ans, abs(goal - s1 - second_half[idx - 1]))
            if idx < len(second_half):
                ans = min(ans, abs(goal - s1 - second_half[idx]))
            if idx + 1 < len(second_half):
                ans = min(ans, abs(goal - s1 - second_half[idx + 1]))
        return ans
