class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        if goal < 0:
            goal = -goal
            nums = list(map(lambda x: -x, nums))
        nums.sort()
        N = len(nums)
        left = nums[:N//2]
        right = nums[N//2:]
        
        # print(left, right)
        lst = set()
        rst = set()
        for x in left:
            lst |= {x} | {x + y for y in lst}
        for x in right:
            rst |= {x} | {x + y for y in rst}
        
        lst.add(0)
        # print(lst, rst)
        rst = list(sorted(rst))
        # print(rst)
        ans = goal
        for x in lst:
            idx = bisect.bisect_left(rst, goal - x)
            # print(x, idx, goal - x)
            if idx > 0:
                ans = min(ans, abs(rst[idx - 1] + x - goal))
            if idx < len(rst):
                ans = min(ans, abs(rst[idx] + x - goal))
                
        return ans
