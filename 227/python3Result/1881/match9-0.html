class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        n = len(nums)
        def csa(arr):
            x = []
            nn = len(arr)
            for i in range(1<<nn):
                bin_i = bin(i)[2:]
                bin_i = '0' * (nn-len(bin_i)) + bin_i
                s = 0
                for j, c in enumerate(bin_i):
                    if c == '1':
                        s += arr[j]
                x.append(s)
            return x

        a = csa(nums[:n//2])
        b = csa(nums[n//2:])
        a.sort()
        b.sort()

        res = abs(goal)
        cmax = len(b)
        for x in a:
            target = goal - x
            loc = bisect.bisect_left(b,target,hi=cmax)
            if loc == len(b):
                res = min(res, abs(x+b[-1]-goal))
            elif b[loc] == target:
                return 0
            else:
                res = min(res,abs(x+b[loc]-goal))
                if loc:
                    res = min(res,abs(x+b[loc-1]-goal))
                if loc < len(b)-1:
                    res = min(res,abs(x+b[loc+1]-goal))
                if res == 0:
                    return 0
            cmax = min(cmax, loc+1)
                
        return res
