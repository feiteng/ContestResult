class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        nums.sort()
        n = len(nums)
        first = nums[:n // 2]
        last = nums[n // 2:]
        
        def findAll(arr):
            res = set()
            for num in arr:
                tmp = set()
                for sSum in res:
                    tmp.add(sSum + num)
                res.add(num)
                res = res.union(tmp)
            return res
        
        first = findAll(first)
        first.add(0)
        last = list(findAll(last))
        last.sort()
        lastLen = len(last)
        res = abs(goal)
        
        for each in first:
            idx = bisect.bisect_left(last, goal - each)
            res = min(res, abs(goal - each))
            if idx - 1 >= 0:
                res = min(res, abs(each + last[idx - 1] - goal))
            if idx < lastLen:
                res = min(res, abs(each + last[idx] - goal))
        
        return res
