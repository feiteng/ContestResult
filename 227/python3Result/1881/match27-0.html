class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        n = len(nums); n1 = n // 2
        if n & 1:
            n2 = n1 + 1
        else:
            n2 = n1
        
        nums1 = [0] * (1 << n1)
        nums2 = [0] * (1 << n2)
        for state in range(1<<n1):
            s = 0
            for i in range(n1):
                if state & (1<<i):
                    s += nums[i]
            nums1[state] = s
        for state in range(1<<n2):
            s = 0
            for i in range(n2):
                if state & (1<<i):
                    s += nums[n1+i]
            nums2[state] = s
        nums1.sort()
        res = abs(goal)
        for item in nums2:
            diff = goal - item
            idx = bisect.bisect_left(nums1, diff)
            if idx > 0:
                res = min(res, abs(goal - nums1[idx-1] - item))
            if idx < len(nums1):
                res = min(res, abs(goal - nums1[idx] - item))
        return res
