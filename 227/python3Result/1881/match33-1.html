class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        def gen(i, j, tot):
            if i == j:
                return [tot]
            first = gen(i+1, j, nums[i]+tot)
            sec = gen(i+1, j, tot)
            return first + sec
        
        n = len(nums)
        first = gen(0, n//2, 0)
        sec = gen(n//2, n, 0)
        sec.sort()
        
        res = inf
        for i in range(len(first)):
            target = goal - first[i] 
            idx = bisect_left(sec, target)
            if idx < len(sec):
                res 
= min(res, abs(sec[idx] - target))
            if idx - 1 >= 0:
                res = min(res, abs(sec[idx-1] - target))
        return res
