class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        
        max_sub = sum( [data  for data in nums if data > 0])
        min_sub = sum( [data  for data in nums if data < 0])
        if goal >= max_sub:
            return goal - max_sub
        if goal <= min_sub:
            return min_sub - goal
        
        ans = float('inf')
        unique = set([0])
        for num in nums:
            next = set()
            for data in unique:
                if data + num == goal:
                    return 0
                next.add(data + num)
            unique.update(next)
        for data in unique:
            ans = min(ans, abs(data - goal))
        return ans
