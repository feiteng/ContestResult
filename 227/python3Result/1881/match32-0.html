class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        negsum = sum([n for n in nums if n < 0])
        possum = sum([n for n in nums if n > 0])
        if possum <= goal:
            return goal - possum
        if negsum >= goal:
            return negsum - goal
        d = set([-goal])
        mindiff = abs(goal)
        for n in nums:
            dd = d.copy()
            for key in d:
                v = key + n
                dd.add(v)
                mindiff = min(mindiff, abs(v))
                if mindiff == 0:
                    return 0
            d = dd
        return mindiff
