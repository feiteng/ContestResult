class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        def getAns(ns):
            aSet = set([0])
            for a in ns:
                for b in list(aSet):
                    aSet.add(b + a)
            return list(aSet)
        
        n = len(nums)
        n2 = n // 2
        ans1 = getAns(nums[:n2])
        ans2 = getAns(nums[n2:])
        ans1.sort()
        ans2.sort()
        n1, n2 = len(ans1), len(ans2)

        ans = abs(goal)
        i1, i2 = 0, n2-1 
        while i1 < n1 and i2 >= 0:
            g = ans1[i1] + ans2[i2]
            ans = min(ans, abs(g - goal))
            if g < goal:
                i1 += 1
            else:
                i2 -= 1
        return ans
