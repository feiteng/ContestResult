class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        
        def dfs(i,c_sum,arr,a_sums):
            a_sums.add(c_sum)
            if i==len(arr):
                return
            dfs(i+1,c_sum,arr,a_sums)
            dfs(i+1,c_sum+arr[i],arr,a_sums)
        
        n=len(nums)
        a_sums1=set()
        a_sums2=set()
        dfs(0,0,nums[:n//2],a_sums1)
        dfs(0,0,nums[n//2:],a_sums2)
        
        a2=sorted(a_sums2)
        ans=10**10
        for v1 in a_sums1:
            i1=bisect_left
(a2,goal-v1)
            if i1<len(a2):
                ans=min(ans,abs(goal-v1-a2[i1]))
            if i1>=1:
                ans=min(ans,abs(goal-v1-a2[i1-1]))
        return ans
