class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        
        n = len(nums)
        
        def generate(i, end, s):
            
            if i == end:
                return s
            
            new = set()
            for prev in s:
                new.add(prev + nums[i])
            return generate(i+1, end, s | new)
        
        ls, rs = generate(0, n // 2, set([0])), generate(n // 2, n, set([0]))
        ls, rs = list(sorted(ls)), list(sorted(rs))
        
        res = abs(goal)
        
        idx = bisect.bisect_left(ls, goal)
        if idx > 0:
            res = min(res, abs(goal - ls[idx - 1]))
        if idx < len(ls):
            res = min(res, abs(goal - ls[idx]))
            
        idx = bisect.bisect_left(rs, goal)
        if idx > 0:
            res = min(res, abs(goal - rs[idx - 1]))
        if idx < len(rs):
            res = min(res, abs(goal - rs[idx]))
            
        for l in ls:
            idx = bisect.bisect_left(rs, goal - l)
            if idx > 0:
                res = min(res, abs(goal - l - rs[idx - 1]))
            if idx < len(rs):
                res = min(res, abs(goal - l - rs[idx]))
        return res
