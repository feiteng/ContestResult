from bisect import bisect_left
class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        def helper(l, r, S):
            if l == r: return {S}
            pick = helper(l + 1, r, S + nums[l])
            leave = helper(l + 1, r, S)
            return pick | leave
        
        n = len(nums)
        H1 = helper(0, n // 2, 0)
        H2 = helper(n // 2, n, 0)
        if len(H1) < len(H2):
            H1, H2 = H2, H1
        H2, ans = sorted(H2), abs(goal)
        for x in H1:
            j = bisect_left(H2, goal - x)
            if j == len(H2):
                ans = min(ans, abs(x + H2[-1] - goal))
            else:
                ans = min(ans, abs(x + H2[j] - goal))
                ans = min(ans, abs(x + H2[j - 1] - goal))
        return ans
