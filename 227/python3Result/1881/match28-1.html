class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        n=len(nums)

        def calall(aa):
            ss=set([0])
            for a in aa:
                nss=set()
                for s in ss:
                    nss.add(s+a)
                ss|=nss
            return sorted(ss)

        pre=calall(nums[:n//2])
        pos=calall(nums[n//2:])

        ans=10**16
        j=len(pos)-1
        for i,a in enumerate(pre):
            while j>=0 and a+pos[j]>goal:
                j-=1
            
if j>=0:ans=min(ans,goal-(a+pos[j]))
            if j+1<len(pos):ans = min(ans, a+pos[j+1]-goal)

        return ans
