class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        if len(nums) == 1:
            return min(abs(goal), abs(goal-nums[0]))
        k = len(nums)//2
        
        
        def allcomb(n):
            res = [0, n[0]]
            for i in range(1, len(n)):
                res.extend([q+n[i] for q in res])
            return res
        
        
        n1 = allcomb(nums[:k])
        n2 = sorted(allcomb(nums[k:]))
        
        res = abs(goal)
        
        for n in n1:
            target = goal - n
            i = bisect.bisect_left(n2, target)
            if i != len(n2):
                res 
= min(res, abs(target-n2[i]))
            if i != 0:
                res = min(res, abs(target-n2[i-1]))
                
        return res
