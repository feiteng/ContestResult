class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        if len(nums) == 1:
            return min(abs(nums[0]-goal), abs(goal))
        nums = sorted(nums,key=lambda x:abs(x))[::-1]
        def sumset(nums):
            s = {0}
            for n in nums:
                new_s = s.copy()
                for i in s:
                    new_s.add(i+n)
                s = new_s
            return s
        s1 = sorted(list(sumset(nums[:len(nums)//2])))
        s2 = sorted(list(sumset(nums[len(nums)//2:])))
        #print(s1, s2)
        res = abs(goal)
        for s in s1[::-1]:
            p = bisect_left(s2, goal-s)
            if p == 0:
                res = min(res, abs(goal-s-s2[0]))
            if p - 1 >= 0:
                res = min(res, abs(goal-s-s2[p-1]))
            if p + 1 < len(s2):
                res = min(res, abs(goal-s-s2[p+1]))
            if p < len(s2):
                res = min(res, abs(goal-s-s2[p]))
            if p == len(s2):
                res = min(res,abs(goal-s-s2[-1]))
                break
        return res
