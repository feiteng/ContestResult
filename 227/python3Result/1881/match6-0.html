from typing import List
import bisect
class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        
        n = len(nums)
        mid = n // 2
        
        nums1 = nums[:mid]
        nums2 = nums[mid:]
        
        lst1, lst2 = [], []
        ret = float('inf')
        
        for i in range(2 ** mid):
            this = 0
            for j in range(mid):
                if i & 1 << j:
                    this += nums1[j]
            lst1.append(this)
        for i in range(2 ** (n - mid)):
            this = 0
            for j in range(n - mid):
                if i & 1 << j:
                    this += nums2[j]
            lst2.append(this)
            
        lst1.sort()
        lst2.sort()
        
        for i in range(len(lst1)):
            if i and lst1[i] == lst1[i - 1]:
                continue
            idx = bisect.bisect_left(lst2, goal - lst1[i])
            if 0 <= idx < len(lst2):
                ret = min(ret, abs(goal - lst2[idx] - lst1[i]))
            if idx:
                ret = min(ret, abs(goal - lst2[idx - 1] - lst1[i]))
                
        return ret
