class Solution:
    def minAbsDifference(self, nums: List[int], goal: int) -> int:
        posSum=sum(num for num in nums if num>0)
        negSum=sum(num for num in nums if num<0)
        maxSum=max(posSum,abs
(negSum))
        if posSum<=goal:
            return goal-posSum
        if negSum>=goal:
            return negSum-goal
        cur=set([0])
        res=abs(goal)
        for i,num in enumerate(sorted(nums)):
            cur={num+v for v in cur}|cur|set([num])
            sorted1=sorted(cur)
            idx=bisect.bisect_left(sorted1,goal)
            val=sorted1[idx] if idx<len(sorted1) else math.inf
            prevVal=sorted1[idx-1] if idx>0 else -math.inf
            nxtNum=nums[i+1] if i<len(nums)-1 else math.inf
            if nxtNum>0:
                cur=set(sorted1[:idx+1])
            res=min(res,val-goal,goal-prevVal)
            if not res:
                return 0
        return res
