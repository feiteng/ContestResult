class Solution:
    
    def largestMerge(self, word1: str, word2: str) -> str:
        n1,n2 = len(word1), len(word2)
        i1,i2 = 0,0
        result = []
        while i1 < n1 and i2 < n2:
            if word1[i1] > word2[i2]:
                result.append(word1[i1])
                i1 += 1
            elif word2[i2] > word1[i1]:
                result.append(word2[i2])
                i2 += 1
            else:
                # equal
                
if word1[i1:] > word2[i2:]:
                    result.append(word1[i1])
                    i1 += 1
                else:
                    result.append(word2[i2])
                    i2 += 1
                
                        
        result.append(word1[i1:])
        result.append(word2[i2:])
        return ''.join(result)
