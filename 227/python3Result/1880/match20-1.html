class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        i, j = 0, 0
        # 插入标兵
        word1 
+= 'A'
        word2 += 'A'
        merge = ''
        # 由于插入了标兵，所以长度判断的时候需要-1
        while i < len(word1) - 1 or j < len(word2) - 1:
            if word1[i:] >= word2[j:]:
                merge += word1[i]
                i += 1
            else:
                merge += word2[j]
                j += 1
        return merge
