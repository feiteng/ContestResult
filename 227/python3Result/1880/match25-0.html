class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        # merge two sorted words
        merge=[]
        w1=0
        w2=0
        while w1<len(word1) and w2<len(word2):
            if word1[w1]>word2[w2]:
                merge.append(word1[w1]) 
                w1+=1
            elif word1[w1]<word2[w2]:
                merge.append(word2[w2]) 
                w2+=1
            else:
                # look for first point difference
                if word1[w1:]>word2[w2:]:
                    merge.append(word1[w1]) 
                    w1+=1
                else:
                    merge.append(word2[w2]) 
                    w2+=1        
        while w1<len(word1):
            merge.append(word1[w1]) 
            w1+=1
        while w2<len(word2):
            merge.append(word2[w2]) 
            w2+=1
        return "".join(merge)
