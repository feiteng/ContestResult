class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        index1, index2 = 0, 0
        n1, n2 = len(word1), len(word2)
        res = ''
        while index1 < n1 and index2 < n2:
            if word1[index1] < word2[index2]:
                res += word2[index2]
                index2 += 1
            elif word1[index1] > word2[index2]:
                res += word1[index1]
                index1 += 1
            elif word1[index1] == word2[index2]:
                preIndex1, preIndex2 = index1, index2
                # print(index1, index2)
                flag = False
                while preIndex1 < n1 and preIndex2 < n2:
                    if word1[preIndex1] > word2[preIndex2]:
                        res += word1[index1]
                        index1 += 1
                        flag = True
                        break
                    elif word1[preIndex1] < word2[preIndex2]:
                        res += word2[index2]
                        index2 += 1
                        
                        flag = True
                        break
                    else:
                        preIndex1 += 1
                        preIndex2 += 1
                if not flag:
                    if preIndex1 == n1:
                        res += word2[index2]
                        index2 += 1
                        
                    elif preIndex2 == n2:
                        res += word1[index1]
                        index1 += 1
                        
        if index1 < n1:
            res += word1[index1:]
        elif index2 < n2:
            res += word2[index2:]
        return res
