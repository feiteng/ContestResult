class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        i = 0
        j = 0
        m = ""
        while i < len(word1) and j < len(word2):
            if word1[i] > word2[j]:
                m += word1[i]
                i += 1
            elif word1[i] < word2[j]:
                m += word2[j]
                j += 1
            else:
                i1 = i
                j1 = j
                while i1 < len(word1) and j1 < len(word2) and word1[i1] == word2[j1]:
                    i1 += 1
                    j1 += 1
                if i1 == len(word1):
                    m += word2[j]
                    j += 1
                elif j1 == len(word2):
                    m += word1[i]
                    i += 1
                elif word1
[i1] < word2[j1]:
                    m += word2[j]
                    j += 1
                else:
                    m += word1[i]
                    i += 1
        m += word1[i:] + word2[j:]
        return m
