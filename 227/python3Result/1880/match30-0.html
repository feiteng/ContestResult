class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        m, n =len(word1), len(word2)
        ans = []
        i, j = 0, 0
        while i < m and j < n:
            if word1[i] > word2[j]:
                ans.append(word1[i])
                i += 1
            elif word1[i] < word2[j]:
                ans.append(word2[j])
                j += 1
            else:
                incr = 1
                while i +incr < m and j +incr < n and word1[i+incr] == word2[j+incr]:
                    incr += 1
                if i +incr == m:
                    ans.append(word2[j])
                    j += 1
                    continue
                if j +incr == n:
                    ans.append(word1[i])
                    i += 1
                    continue
                if word1[i+incr] > word2[j+incr]:
                    ans.append(word1[i])
                    i += 1
                else:
                    ans.append(word2[j])
                    j += 1
        ans.extend(word1[i:m])
        ans.extend(word2[j:n])
        return "".join(ans)
