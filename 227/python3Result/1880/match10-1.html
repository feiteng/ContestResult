class Solution:
    def largestMerge(self, w1: str, w2: str) -> str:
        ans = ''
        while w1 or w2:
            if not w1:
                ans += w2
                break
                
            if not w2:
                ans += w1
                break
                
            if w1 >= w2:
                ans += w1[0]
                w1 = w1[1:]
            else:
                ans += w2[0]
                w2 = w2[1:]
                
        return ans
