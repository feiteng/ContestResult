class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        wd1 = wd2 = 0
        merge = ""
        while wd1 < len(word1) and wd2 < len(word2):
            if word1[wd1] > word2[wd2]:
                merge += word1[wd1]
                wd1 += 1
            elif word1[wd1] < word2[wd2]:
                merge += word2[wd2]
                wd2 += 1
            else:
                if word1[wd1:] < word2[wd2:]:
                    merge += word2[wd2]
                    wd2 += 1
                else:
                    merge += word1[wd1]
                    wd1 += 1
        if wd1 < len(word1):
            merge += word1[wd1:]
        if wd2 < len(word2):
            merge += word2[wd2:]
        return merge
