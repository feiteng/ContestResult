class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        result = []
        i = j = 0
        while i < len(word1) and j < len(word2):
            if word1[i] > word2[j]:
                result.append(word1[i])
                i += 1
            elif word1[i] < word2[j]:
                result.append(word2[j])
                j += 1
            else:
                if word1[i:] >= word2[j:]:
                    result.append(word1[i])
                    i += 1
                else:
                    result.append(word2[j])
                    j += 1
        while i < len(word1):
            result.append(word1[i])
            i += 1
        while j < len(word2):
            result.append(word2[j])
            j += 1
        return "".join(result)
