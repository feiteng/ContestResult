class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        head1 = head2 = 0
        m = ""
        while head1<len(word1) or head2<len(word2):
            #print(m , head1, head2)
            if word1[head1:]>word2[head2:]:
                m+=word1[head1]
                head1+=1
            else:
                m+=word2[head2]
                head2+=1
        return m
