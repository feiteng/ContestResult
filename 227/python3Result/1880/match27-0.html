class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        s = ''
        i, j = 0, 0
        while i < len(word1) and j < len(word2):
            if word1[i] < word2[j]:
                s += word2[j]
                j += 1
            elif word1[i] > word2[j]:
                s += word1[i]
                i += 1
            else:
                if word1[i:] < word2[j:]:
                    s += word2[j]
                    j += 1
                else:
                    s += word1[i]
                    i += 1                    
        if i < len(word1):
            s += word1[i:]
        if j < len(word2):
            s += word2[j:]
        
        return s
