class Solution:
    def largestMerge(self, a: str, b: str) -> str:
        res = []
        i,j = 0, 0
        m,n = len(a), len(b)
        while i < m and j < n:
            if a[i] > b[j]:
                res.append(a[i])
                i += 1
            elif b[j] > a[i]:
                res.append(b[j])
                j += 1
            else:
                x,y = i,j
                while x < m and y < n and a[x] == b[y]:
                    x,y = x+1, y+1
                if x == m and y == n:
                    res.append(a[i])
                    i += 1
                elif x == m:
                    res.append(b[j])
                    j += 1
                elif y == n:
                    res.append(a[i])
                    i += 1
                else:
                    if a[x] > b[y]:
                        res.append(a[i])
                        i += 1
                    else:
                        res.append(b[j])
                        j += 1
        while i < m:
            res.append(a[i])
            i += 1
        while j < n:
            res.append(b[j])
            j += 1
        return ''.join(res)
