class Solution:
    def largestMerge(self, word1: str, word2: str) -> str:
        res = []
        n1,n2 = len(word1), len(word2)
        i1,i2 = 0,0        
        while i1 < n1 and i2 <n2:
            if word1[i1] > word2[i2]:
                res.append(word1[i1])
                i1 += 1
            elif word1[i1] < word2[i2]:
                res.append(word2[i2])
                i2 += 1
            else:
                cur1,cur2=i1,i2
                while cur1<n1 and cur2<n2 and word1[cur1] == word2[cur2]:
                    cur1+=1
                    cur2+=1
                if cur1 == n1 and cur2 == n2:
                    c = word1[i1]
                    while i1 < n1 and word1[i1] == c:
                        res.append(c)
                        i1+=1
                elif cur1 == n1:
                    c = word2[i2]
                    while i2 < n2 and word2[i2] == c:
                        res.append(c)
                        i2+=1
                elif cur2==n2:
                    c = word1[i1]
                    while i1 < n1 and word1[i1] == c:
                        res.append(c)
                        i1+=1
                elif word1[cur1] > word2[cur2]:
                    c = word1[i1]
                    while i1 < n1 and word1[i1] == c:
                        res.append(c)
                        i1+=1
                elif word1[cur1] < word2[cur2]:
                    c = word2[i2]
                    while i2 < n2 and word2[i2] == c:
                        res.append(c)
                        i2+=1
                else:
                    return -2342
