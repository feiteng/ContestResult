class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length;
        int m = n / 2;
        long ans = Integer.MAX_VALUE;
        TreeSet<Integer> set = new TreeSet<>();
        for (int i = 0; i < (1 << m); i++) {
            int sum = 0;
            for (int j = 0; j < m; j++) {
                if (((i >> j) & 1) != 0) {
                    sum += nums[j];
                }
            }
            set.add(sum);
        }
        // System.out.println(set);
        for (int i = 0; i < (1 << (n - m)); i++) {
            int sum = 0;
            for (int j = 0; j < n - m; j++) {
                if (((i >> j) & 1) != 0) {
                    sum += nums[m + j];
                }
            }
            // System.out.println(sum);
            int k = goal - sum;
            Integer floor = set.floor(k);
            Integer ceiling = set.ceiling(k);
            if (floor != null && ans > Math.abs(goal - (floor + sum))) ans = Math.abs(goal - (floor + sum));
            if (ceiling != null && ans > Math.abs(goal - (ceiling + sum))) ans = Math.abs(goal - (ceiling + sum));
                
        }
        return (int)ans;
    }
}
