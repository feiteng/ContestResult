class Solution {
    public int minAbsDifference(int[] nums, int goal)
	{
		int n = nums.length;
		if(n == 1)
		{
			return Math.abs(nums[0] - goal);
		}
		else
		{
			TreeSet<Long> left = new TreeSet<Long>();
			int ln = n / 2;
			int rn = n - ln;
			for(int x = 0; x < (1
 << ln); x++)
			{
				long current = 0;
				for(int y = 0; y < ln; y++)
				{
					if((x & (1 << y)) > 0)
					{
						current += nums[y];
					}
				}
				
				left.add(current);
			}
			
			long result = Long.MAX_VALUE;
			for(int z = 0; z < (1
 << rn); z++)
			{
				long current = 0;
				for(int a = 0; a < rn; a++)
				{
					if((z & (1 << a)) > 0)
					{
						current += nums[ln + a];
					}
				}
				
				long curGoal = goal - current;
				Long low = left.floor(curGoal);
				Long high = left.ceiling(curGoal);
				if(low != null)
				{
					result = Math.min(result, Math.abs(goal - (current + low)));
				}
				if(high != null)

				{
					result = Math.min(result, Math.abs(goal - (current + high)));
				}
			}
			
			return (int)result;
		}
	}
}
