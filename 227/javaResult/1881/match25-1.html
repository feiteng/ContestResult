class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length;
        int len = 1 << n / 2;
        long ans = 5000000000L;
        List<Long> list = new ArrayList<>();
        for (
int i = 0; i < (1 << n / 2); i++) {
            long sum = 0;
            for (int j = 0; j < n / 2; j++) {
                if ((i >> j & 1) != 0)
                    sum += nums[j];
            }
            list.add(sum);
        }
        int size = list.size();
        Collections.sort(list);
        for (int i = 0; i < (n + 1) / 2; i++)
            nums[i] = nums[i + n / 2];
        
for (int i = 0; i < (1 << (n + 1) / 2); i++) {
            long sum = 0;
            for (int j = 0; j < (n + 1) / 2; j++) {
                if ((i >> j & 1) != 0)
                    sum += nums[j];
            }
            int l = 0, r = size - 1, p = -1;
            while (l <= r) {
                int mid = (l + r) / 2;
                if (goal - sum >= list.get(mid)) {
                    p = mid;
                    l = mid + 1;
                } else
                    r = mid - 1;
            }
            if (p != -1)
                ans = Math.min(ans, Math.abs(goal - (sum + list.get(p))));
            if (p + 1 < size)
                ans = Math.min(ans, Math.abs(goal - (sum + list.get(p + 1))));
        }
        return (int) ans;
    }
}
