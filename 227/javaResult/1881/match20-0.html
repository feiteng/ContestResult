class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n1 = nums.length / 2;
        int n2 = nums.length - n1;
        List<Integer> list = Arrays.stream(nums).boxed().collect(Collectors.toList());
        int[] list1 = sumArray(list.subList(0, n1));
        TreeSet<Integer> set2 = toSet(sumArray(list.subList(n1, nums.length)));
        int out = Math.abs(goal);
        for (int i : list1) {
            int diff = goal - i;
            Integer c = set2.ceiling(diff);
            if (c != null) {
                out = Math.min(out, c - diff);
            }
            Integer f = set2.floor(diff);
            if (f != null) {
                out = Math.min(out, diff - f);
            }
            if (out == 0) {
                return 0;
            }
        }
        return out;
    }

    private int[] sumArray(List<Integer> nums) {
        int[] a = new int[1 << nums.size()];
        int i = 1;
        for (int n : nums) {
            int size = i;
            for (int j = 0; j < size; j++) {
                a[i++] = a[j] + n;
            }
        }
        return a;
    }

    private TreeSet<Integer> toSet(int[] a) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int i : a) {
            set.add(i);
        }
        return set;
    }
}
