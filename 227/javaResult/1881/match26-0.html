class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int N = nums.length;
        List<Integer> l1 = new ArrayList<>();
        List<Integer> l2 = new ArrayList<>();
        for (int i = 0; i < N; i++) {
            if (i % 2 == 0) {
                l1.add(nums[i]);
            } else {
                l2.add(nums[i]);
            }
        }
        TreeSet<Integer> set1 = new TreeSet<>();
        set1.add(0);
        TreeSet<Integer> set2 = new TreeSet<>();
        set2.add(0);
        dfs(l1, 0, set1);
        dfs(l2, 0, set2);
        int res = Integer.MAX_VALUE;
        for (int v1 : set1) {
            int t = goal - v1;
            Integer ceil = set2.ceiling(t);
            if (ceil != null) {
                res = Integer.min(Math.abs(ceil - t), res);
            }
            Integer floor = set2.floor(t);
            if (floor != null) {
                res = Integer.min(Math.abs(floor - t), res);
            }
        }
        return res;
    }

    private void dfs(List<Integer> list, int i, TreeSet<Integer> set) {
        if (i == list.size()) {
            return;
        }
        TreeSet<Integer> nSet = new TreeSet<>();
        for (int v : set) {
            int vv = list.get(i);
            nSet.add(v + vv);
        }
        set.addAll(nSet);
        dfs(list, i + 1, set);
    }
}
