class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        TreeSet<Integer> resultSet1 = new TreeSet<>();
        HashSet<Integer> tmpSet = new HashSet<>();
        resultSet1.add(0);
        long target = goal;
        long ans = Math.abs(goal);
        int n = nums.length;
        for (int i = 0; i < n / 2; i++) {
            tmpSet.clear();
            for (int pre : resultSet1) {
                int validSum = pre + nums[i];
                ans = Math.min(ans, Math.abs(target - validSum));
                
tmpSet.add(validSum);
            }
            resultSet1.addAll(tmpSet);
        }

        HashSet<Integer> resultSet2 = new HashSet<>();
        resultSet2.add(0);
        for (int i = n / 2; i < n; i++) {
            tmpSet.clear();
            for (int pre : resultSet2) {
                int validSum = pre + nums[i];
                ans = Math.min(ans, Math.abs(target - validSum));
                
tmpSet.add(validSum);
            }
            resultSet2.addAll(tmpSet);
        }

        for (int halfNum : resultSet2) {
            long distance = target - halfNum;
            if (distance > Integer.MAX_VALUE || distance < Integer.MIN_VALUE) {
                continue;
            }
            int d = (int)distance;
            Integer tmp = resultSet1.floor(d);
            if (tmp != null) {
                ans = Math.min(ans, Math.abs(d - tmp));
            }

            tmp = resultSet1.ceiling(d);
            if (tmp != null) {
                ans = 
Math.min(ans, Math.abs(d - tmp));
            }
        }

        return (int)ans;
    }
}
