class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length, m = n >> 1, res = Math.abs(goal);
        TreeSet<Integer> set1 = new TreeSet<>(), set2 = new TreeSet<>();
        set1.add(0);
        set2.add(0);
        
        for (int i = 1, len = 1 << m, sum; i < len; i++) {
            sum = 0;
            for (int j = 0; (1 << j) <= i; j++) {
                if (((1 << j) & i) != 0) {
                    sum += nums[j];                 
                } 
            }
            set1.add(sum);
        }
        // System.out.println(set1);
        Integer floor = set1.floor(goal), ceiling = set1.ceiling(goal);
        if (floor != null) res = Math.min(res, goal - floor);
        if (ceiling != null) res = Math.min(res, ceiling - goal);
        
        for (int i = 1, len = 1 << (n - m), sum; i < len; i++) {
            sum = 0;
            for (int j = 0; (1 << j) <= i; j++) {
                if (((1 << j) & i) != 0) {
                    sum += nums[m + j];                 
                } 
            }
            set2.add(sum);
        }
        // System.out.println(set2);
        floor = set2.floor(goal);
        ceiling = set2.ceiling(goal);
        if (floor != null) res = Math.min(res, goal - floor);
        if (ceiling != null) res = Math.min(res, ceiling - goal);
        
        for (int g : set1) {
            int newGoal = goal - g;
            floor = set2.floor(newGoal);
            ceiling = set2.ceiling(newGoal);
            if (floor != null) res = Math.min(res, newGoal - floor);
            if (ceiling != null) res = Math.min(res, ceiling - newGoal);
        }     
        
        return res;
    }
}
