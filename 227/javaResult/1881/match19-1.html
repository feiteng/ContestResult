class Solution {
    /**
    Generate all possible sum 0 -> n /2, first
    Generate all posssible sum n/2 -> n, second
    sort second 
    for each sum in first, |goal - sum|, binary search the first number in the second, find firstNumber <
    **/
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length;
        TreeSet<Integer> firstSums = generate(nums, 0, n / 2);
        TreeSet<Integer> treeSet = 
generate(nums, n / 2, n);
        
        int ans = Integer.MAX_VALUE;
        
        for (int sum: firstSums) {
            int target = goal - sum;
            Integer bigger = treeSet.ceiling(target);
            if (bigger != null) {
                ans = Math.min(ans, Math.abs(bigger - target));
            }
            
            Integer smaller = treeSet.floor(target);
            if (smaller != null) {
                ans = Math.min(ans, Math.abs(target - smaller));
            }
        }
        
        return ans;
    }
    
    private TreeSet<Integer> generate(int[] nums, int start, int end) {
        if (start == end) {
            return new TreeSet<>(Collections.singleton(0));
        }
        
        TreeSet<Integer> subset = generate(nums, start + 1,  end);
        TreeSet<Integer> res = new TreeSet<>(subset);
        for (int sum: subset) {
            res.add(nums[start] + sum);
        }
        return res;
    }
}
