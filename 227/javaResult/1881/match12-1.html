class Solution {
  void pre(int n, int nn, int[] v, int[] res) {
    for(int x = 0; x < nn; x++) {
      
for(int i = 0; i < n; i++) {
        if( ((1<<i) & x) > 0) {
          res[x] += v[i];
        }
      }
    
}
  }
  public int minAbsDifference(int[] nums, int goal) {
    int n = nums.length;
    if(n == 1) {
      int v1 = Math.abs(0 - goal);
      int v2 = Math.abs(nums[0] - goal);
      return 
Math.min(v1, v2);
    }
    int n1 = n/2, n2 = n-n/2;
    int[] v1 = new int[n1], v2 = new int[n2];
    for(int i = 0; i < n1; i++) v1[i] = nums[i];
    for(int j = 0; j < n2; j++) v2[j] = nums[n1 + j];
    int nn1 = (1<<n1), nn2 = (1<<n2);
    int[] A = new int[nn1], B = new int[nn2];
    pre(n1, nn1, v1, A);
    pre(n2, nn2, v2, B);
    Arrays.sort(A);
    Arrays.sort(B);
    int ans = Math.abs(goal);
    int j = nn2-1;
    long gg = goal;
    for(int i = 0; i < nn1; i++) {
      while(j >= 0 && A[i] + B[j] >= goal) {
        if(A[i] + B[j] == goal) return 0;
        if(A[i] + B[j] > goal) {
          ans = Math.min(ans, (A[i] + B[j] - goal));
          j--;
        }
      }
      
if(j >= 0) {
          ans = Math.min(ans, Math.abs(A[i] + B[j] - goal));        
      }
    }
    return ans;
  }
}
