class Solution {
    public int minAbsDifference(int[] nums, int goal) {
       int n = nums.length;
    int ans = Math.abs(nums[0] - goal);
    if (n == 1) {
      return ans;
    }
    int n1 = n / 2;
    int n2 = n - n1;
    int[] nums1 = new int[n1];
    for (int i = 0; i < n1; i++) {
      nums1[i] = nums[i];
    }
    int[] nums2 = new int[n2];
    for (int i = 0; i < n2; i++) {
      nums2[i] = nums[n1 + i];
    }
    
List<Long> sums1 = new ArrayList<>();
    for (int i = 0; i < (1 << n1); i++) {
      long sum = 0;
      for (int j = 0; j < n1; j++) {
        if ((i & (1 << j)) != 0) {
          sum += nums1[j];
        }
      }
      ans = (int) Math.min(ans, Math.abs(sum - goal));
      if (ans == 0) return 0;
      sums1.add(sum);
    }
      
    // Collections.sort(sums1);

    TreeSet<Long> sums2 = new TreeSet<>();
    
for (int i = 0; i < (1 << n2); i++) {
      long sum = 0;
      for (int j = 0; j < n2; j++) {
        if ((i & (1 << j)) != 0) {
          sum += nums2[j];
        }
      }
      ans = (int) Math.min(ans, Math.abs(sum - goal));
      if (ans == 0) return 0;
      sums2.add(sum);
    }

    // System.out.format("sums1: %s\n", Utils.traceListInt(sums1));
    // System.out.format("  ans:%d\n", ans);

    for (long v1 : sums1) {
      long k = goal - v1;
      Long ck = sums2.ceiling(k);
      if (ck != null) {
        ans = (int) Math.min(ans, ck - k);
      }

      Long fk = sums2.floor(k);
      if (fk != null) {
        ans = (int) Math.min(ans, k - fk);
      }
      if (ans == 0) return 0;
    }
    return ans; 
    }
}
