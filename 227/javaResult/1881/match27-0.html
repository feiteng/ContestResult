class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length;
        int ans = Math.abs(goal);
        TreeSet<Integer> lsums = new TreeSet<>();
        lsums.add(0);
        for (int i = 0; i < n/2; ++i) {
            int x = nums[i];
            ArrayList<Integer> ta = new ArrayList<>();
            for (int y : lsums) {
                if (lsums.contains(x+y))
                    continue;
                ta.add(x+y);
            }
            lsums.addAll(ta);
        }
        HashSet<Integer> rsums = new HashSet<>();
        rsums.add(0);
        for (int i = n/2; i < n; ++i) {
            int x = nums[i];
            ArrayList<Integer> ta = new ArrayList<>();
            for (int y : rsums) {
                if (rsums.contains(x+y))
                    continue;
                ta.add(x+y);
            }
            rsums.addAll(ta);
        }
        for (int rs : rsums) {
            int left = goal-rs;
            Integer low = lsums.floor(left);
            Integer hi = lsums.ceiling(left);
            if (low != null) {
                ans = Math.min(ans, Math.abs(goal-rs-low.intValue()));
            }
            if (hi != null) {
                ans = Math.min(ans, Math.abs(goal-rs-hi.intValue()));
            }
        }
        return ans;
    }
}
