class Solution {
  public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length;
        if (n == 1) {
            return Math.min(Math.abs(goal), Math.abs(goal - nums[0]));
        }

        int m = n / 2;
        TreeSet<Integer> set = new TreeSet<>((x, y) -> Integer.compare(x, y));

        for (int s = 0; s < (1 << m); s++) {
            int sum = 0;
            for (int i = 0; i < m; i++) {
                if (((s >>> i) & 1) != 0) {
                    sum += nums[i];
                }
            }
            set.add(sum);
        }

        int ans = Integer.MAX_VALUE;
        int l = n - m;
        for (int s = 0; s < (1 << l); s++) {
            int sum = 0;
            for (int i = 0; i < l; i++) {
                if (((s >>> i) & 1) != 0) {
                    sum += nums[i + m];
                }
            }
            Integer ceil = set.ceiling(goal - sum);
            if (ceil != null) {
                ans = Math.min(ans, Math.abs((ceil + sum) - goal));
            }
            
            Integer floor = set.floor(goal - sum);
            if (floor != null) {
                ans = Math.min(ans, Math.abs((floor + sum) - goal));
            }
        }
        return ans;
    }
}
