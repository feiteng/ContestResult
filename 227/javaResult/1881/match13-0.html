class Solution {
public void getResArr(int n, int[] pre, int[] res, int tot) {
        for (int i = 0; i < tot; ++i) {
            int x = 0;
            for (int j = 0; j < n; ++j) {
                if ((i & (1 << j)) > 0) {
                    x += pre[j];
                }
            }
            res[i] = x;
        }
    }
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length, ans = Math.abs(goal);
        if (n == 0) {
            return ans;
        }
        if (n == 1) {
            return Math.min(ans, Math.abs(nums[0] - goal));
        }
        int n1 = (n + 1) / 2, n2 = n - n1;
        int[] pre = new int[n1], suf = new int[n2];
        for (int i = 0; i < n1; ++i) {
            pre[i] = nums[i];
        }
        for (int i = 0; i < n2; ++i) {
            suf[i] = nums[i+n1];
        }

        int[] res1 = new int[1200000], res2 = new int[1200000];
        int tot1 = 1 << n1, tot2 = 1 << n2;
        getResArr(n1, pre, res1, tot1);
        getResArr(n2, suf, res2, tot2);
        Arrays.sort(res2, 0, tot2);
        for (int i = 0; i < tot1; ++i) {
            int val = goal - res1[i];
            int L = 0, R = tot2;
            while (L < R) {
                int mid = (L + R) >> 1;
                if (res2[mid] >= val) {
                    R = mid;
                } else {
                    L = mid + 1;
                }
            }
            ans = Math.min(ans, Math.abs(res2[L] - val));
            if (L > 0) {
                ans = Math.min(ans, Math.abs(res2[L-1] - val));
            }
        }
        return ans;
    }
}
