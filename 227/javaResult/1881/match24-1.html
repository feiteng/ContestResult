class Solution {
    public int minAbsDifference(int[] nums, int g) {
        int len =  nums.length;
        int l1 =  len/2;
        int l2 = len - l1;
        
        
        TreeSet<Integer> st = new TreeSet<>();
        
for(int j=0;j<1<<l1;++j){
            int s = 0;
            for(int k=0;k<l1;++k){
                if((j&(1<<k))!=0){
                    s += nums[k];
                }
            }
            st.add(s);
        }
        int sz =  st.size();
        
int[] li = new int[sz];
        int p = 0;
        for(int u:st){
            li[p++] = u;
        }
        
int r=  2000000000;
        for(int j=0;j<1<<l2;++j){
            int s = 0;
            for(int k=0;k<l2;++k){
                if((j&(1<<k))!=0){
                    s += nums[l1+k];
                }
            }
            int ck =  g - s;
            
            int lo = 0;
            int hi = sz-1;
            while(lo<hi){
                int mid = lo+hi;mid /=2;
                if(li[mid]>=ck)
{
                    hi = mid;
                }else{
                    lo = mid+1;
                }
            }
            r = Math.min(r, Math.abs(ck-li[lo]));
            
            
             lo = 0;
             hi = sz-1;
            while(lo<hi){
                int mid = lo+hi + 1;mid /=2;
                if(li[mid]>ck){
                    hi = mid - 1;
                }else{
                    lo = mid;
                }
            }
            r = Math.min(r, Math.abs(ck-li[lo]));
            
            
            
            
            
        }
        return r;
        
        
        
        
        
        
    }
}
