class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int n = nums.length, m1 = n >> 1, m2 = n - m1;
        long res = Long.MAX_VALUE;
        long[] data = new long[1 << m1];
        for (int i = 1, k = 1; i < 1 << m1; ++i) {
            long sum = 0;
            for (int j = 0; j < m1; ++j) {
                if (1 == ((i >> j) & 1)) sum += nums[j];
            }
            data[k++] = sum;
        }
        Arrays.sort(data);
        for (int i = 0; i < 1 << m2; ++i) {
            long sum = 0;
            for (int j = 0; j < m2; ++j) {
                if (1 == ((i >> j) & 1)) sum += nums[j + m1];
            }            
            long target = (long) goal - sum;
            int j = Arrays.binarySearch(data, target);
            if (0 <= j) {
                res = 0;
                break;
            } else {
                j = -j - 1;
                if (j < data.length) res = Math.min(res, Math.abs(target - data[j]));
                if (0 < j) res = Math.min(res, Math.abs(target - data[j - 1]));
            }
        }
        return (int) res;
    }
}
