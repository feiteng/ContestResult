class Solution {
    public int minAbsDifference(int[] nums, int goal) {
        int N = nums.length;
        int mid = (N - 1) / 2;
        int size1 = mid + 1;
        int size2 = N - 1 - mid;
        long[] s1 = new long[1 << size1];
        long[] s2 = new long[1 << size2];
        
        int ind1 = 1;
        int ind2 = 1;

        for (int i = 0; i <= mid; i++) {
            int time = nums[i];

            for (int j = 0; j < (1 << i); j++) {
                s1[ind1++] = s1[j] + time;
            }
        }

        for (int i = mid + 1; i < N; i++) {
            int time = nums[i];

            for (int j = 0; j < 1 << (i - (mid + 1)); j++) {
                s2[ind2++] = s2[j] + time;
            }
        }

        Arrays.sort(s2);

        long max = Long.MAX_VALUE;

        for (long sum1 : s1) {
            int ind = bs(s2, goal - sum1);
            
            if (ind - 1 >= 0) {
                max = Math.min(max, Math.abs(goal - (sum1 + s2[ind - 1])));
            }
            
            max = Math.min(max, Math.abs(goal - (sum1 + s2[ind])));
            
            if (ind + 1 < s2.length) {
                max = Math.min(max, Math.abs(goal - (sum1 + s2[ind + 1])));
            }
        }

        return (int)max;
    }
    
    private int bs(long[] s, long target) {
        int low = 0;
        int high = s.length - 1;
        
        while (low < high) {
            int mid = low + (high - low) / 2;
            
            if (s[mid] >= target) {
                high = mid;
            }
            else {
                low = mid + 1;
            }
        }
        
        return low;
    }
}
