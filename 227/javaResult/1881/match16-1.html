class Solution {
    public int minAbsDifference(int[] nums, int goal) {
    int n = nums.length;
    int hf = n / 2;
    List<Integer> first = new ArrayList<>();
    for (int i = 0; i < hf; i++) {
      
first.add(nums[i]);
    }
    List<Integer> second = new ArrayList<>();
    for (int i = hf; i < n; i++) {
      second.add(nums[i]);
    }
    List<Integer> p1 = poss(first);
    List<Integer> p2 = poss(second);
    int res = Integer.MAX_VALUE;
    for (int v : p1) {
      int target = goal - v;
      // Find the first >= target.
      int low = 0;
      int high = p2.size();
      while (low < high) {
        int mid = (low + high) / 2;
        int val = p2.get(mid);
        if (val < target) {
          low = mid + 1;
        } else {
          high = mid;
        }
      }
      
if (low > 0) {
        res = Math.min(res, Math.abs(goal - v - p2.get(low - 1)));
      }
      if (low < p2.size()) {
        res = Math.min(res, Math.abs(goal - v - p2.get(low)));
      }
    }
    return res;
  }

  List<Integer> poss(List<Integer> ls) {
    int n = ls.size();
    
List<Integer> has = new ArrayList<>();
    for (int m = 0; m < (1 << n); m++) {
      int sum = 0;
      for (int i = 0; i < n; i++) {
        if ((m & (1 << i)) != 0) {
          sum += ls.get(i);
        }
      }
      has.add(sum);
    }
    Collections.sort(has);
    return has;
  }
}
