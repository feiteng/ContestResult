class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int i = 0;
        int j = 0;
        while(i<word1.length()&&j<word2.length()){
            int cur1 = i;
            int cur2 = j;
            while(cur1<word1.length()&&cur2<word2.length()&&word1.charAt(cur1)==word2.charAt(cur2)){
                cur1++;
                cur2++;
            }
            
            if(cur1==word1.length()){
                // sb.append(word2.substring(j,cur2));
                // j = cur2;
                sb.append(word2.charAt(j));
                j++;
            } else if(cur2==word2.length()){
                // sb.append(word1.substring(i,cur1));
                // i = cur1;
                sb.append(word1.charAt(i));
                i++;
            } else {
                if(word1.charAt(cur1)<word2.charAt(cur2)){
                    sb.append(word2.charAt(j));
                    j++;
                } else {
                    sb.append(word1.charAt(i));
                    i++;
                }
            }
            
        }
        
        if(j<word2.length()){
            sb.append(word2.substring(j));
        } else if(i<word1.length()){
            sb.append(word1.substring(i));
        }
        return sb.toString();
    }
}
