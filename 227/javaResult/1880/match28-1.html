class Solution {
  public static String largestMerge(String word1, String word2) {
    char[] ca1 = word1.toCharArray();
    char[] ca2 = word2.toCharArray();
    int e1 = ca1.length;
    int e2 = ca2.length;
    
StringBuilder sb = new StringBuilder();
    int i = 0;
    int j = 0;
    while (i < e1 && j < e2) {
      if (compare(ca1, i, ca2, j) < 0) {
        sb.append(ca2[j++]);
      } else {
        sb.append(ca1[i++]);
      }
    }
    while (i < e1) {
      sb.append(ca1[i++]);
    }
    while (j < e2) {
      sb.append(ca2[j++]);
    }
    
return sb.toString();
  }

  static int compare(char[] ca1, int i, char[] ca2, int j) {
    int e1 = ca1.length;
    int e2 = ca2.length;
    while (i < e1 && j < e2) {
      if (ca1[i] != ca2[j]) {
        return Character.compare(ca1[i], ca2[j]);
      }
      i++;
      j++;
    
}
    if (i < e1) {
      return 1;
    } else if (j < e2) {
      return -1;
    } else {
      return 0;
    }
  }
}
