class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb1 = new StringBuilder(word1);
        StringBuilder sb2 = new StringBuilder(word2);
        StringBuilder ans = new StringBuilder();
        
        while(sb1.length() > 0 && sb2.length() > 0){
            if (sb1.compareTo(sb2) >= 0){
                ans.append(sb1.charAt(0));
                sb1.deleteCharAt(0);
            }else{
                ans.append(sb2.charAt(0));
                sb2.deleteCharAt(0);
            }
        }
        
        if (sb1.length() > 0) ans.append(sb1);
        if (sb2.length() > 0) ans.append(sb2);
        
        
        return ans.toString();
    }
}
