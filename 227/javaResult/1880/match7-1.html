class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb1 = new StringBuilder(word1);
        StringBuilder sb2 = new StringBuilder(word2);
        StringBuilder sb3 = new StringBuilder();
        while(sb1.length() > 0 || sb2.length() > 0) {
            if(sb1.length() == 0) {
                sb3.append(sb2.charAt(0));
                sb2.deleteCharAt(0);
            } else if(sb2.length() == 0) {
                sb3.append(sb1.charAt(0));
                sb1.deleteCharAt(0);
            } else if(sb1.toString().compareTo(sb2.toString()) > 0) {
                sb3.append(sb1.charAt(0));
                sb1.deleteCharAt(0);
            } else {
                sb3.append(sb2.charAt(0));
                sb2.deleteCharAt(0);
            }
        }
        return sb3.toString();
    }
}
