class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int word1Ptr = 0;
        int word2Ptr = 0;
        while (word1Ptr < word1.length() || word2Ptr < word2.length()) {
            if (word1Ptr >= word1.length()) {
                sb.append(word2.charAt(word2Ptr++));
            } else if (word2Ptr >= word2.length()) {
                sb.append(word1.charAt(word1Ptr++));
            } else if (word1.charAt(word1Ptr) > word2.charAt(word2Ptr)) {
                sb.append(word1.charAt(word1Ptr++));
            } else if (word2.charAt(word2Ptr) > word1.charAt(word1Ptr)) {
                sb.append(word2.charAt(word2Ptr++));
            } else {
                String s = word1.substring(word1Ptr, word1.length());
                String t = word2.substring(word2Ptr, word2.length());
                if (s.compareTo(t) > 0) {
                    sb.append(word1.charAt(word1Ptr++));
                } else {
                    sb.append(word2.charAt(word2Ptr++));
                }
            }
        }
        
        return sb.toString();
    }
}
