class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int i1 = 0, i2 = 0;
        while (i1 < word1.length() && i2 < word2.length()) {
            if (word1.charAt(i1) > word2.charAt(i2) || word1.charAt(i1) == word2.charAt(i2) && prioritizeOne(word1, word2, i1, i2)) {
                sb.append(word1.charAt(i1++));
            } else {
                sb.append(word2.charAt(i2++));
            }
        }
        sb.append(word1.substring(i1, word1.length()));
        sb.append(word2.substring(i2, word2.length()));
        return sb.toString();
    }
    
    private boolean prioritizeOne(String word1, String word2, int i1, int i2) {
        while (i1 < word1.length() && i2 < word2.length() && word1.charAt(i1) == word2.charAt(i2)) {
            i1++;
            i2++;
        }
        if (i1 == word1.length()) {
            return false;
        }
        if (i2 == word2.length()) {
            return true;
        }
        return word1.charAt(i1) > word2.charAt(i2);
    }
}
