class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int index1 = 0;
        int index2 = 0;
        
        while (index1 < word1.length() && index2 < word2. length()) {
            if (word1.charAt(index1) > word2.charAt(index2)) {
                sb.append(word1.charAt(index1++));
            } else if (word1.charAt(index1) < word2.charAt(index2)) {
                sb.append(word2.charAt(index2++));
            } else {
                int i1 = index1; int i2 = index2;
                
                while (i1 < word1.length() && i2 < word2.length() && word1.charAt(i1) == word2.charAt(i2)) {
                    i1++;
                    i2++;
                }
                if (i1 < word1.length() && i2 <word2.length()) {
                    if (word1.charAt(i1) > 
word2.charAt(i2)) {
                        sb.append(word1.charAt(index1++));
                    } else {
                        sb.append(word2.charAt(index2++));
                       
                    }
                } else if (i1 < word1.length()) {
                     sb.append(word1.charAt(index1++));
                    
                } else {
                      sb.append(word2.charAt(index2++));
                }
            
}
        }
        while (index1 < word1.length()) {
            sb.append(word1.charAt(index1++));
        }
        
        while (index2 < word2.length()) {
            sb.append(word2.charAt(index2++));
        }
        
        return sb.toString();
    }
}
