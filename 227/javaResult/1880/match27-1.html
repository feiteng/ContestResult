class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder w1 = new StringBuilder(word1);
        StringBuilder w2 = new StringBuilder(word2);
        StringBuilder result = new StringBuilder();
        
        while (w1.length() > 0 && w2.length() > 0) {
            if (w1.compareTo(w2) > 0) {
                result.append(w1.charAt(0));
                w1.deleteCharAt(0);
            } else {
                result.append(w2.charAt(0));
                w2.deleteCharAt(0);
            }
        }
        
        if (w1.length() > 0)
            result.append(w1.toString());
        if (w2.length() > 0)
            result.append(w2.toString());
        
        return result.toString();
    }
}
