class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder result = new StringBuilder();
        
        int i = 0;
        int j = 0;
        
        while (i < word1.length() && j < word2.length()) {
            int ii = i;
            int jj = j;

            while (ii < word1.length() && jj < word2.length() && word1.charAt(ii) == word2.charAt(jj)) {
                ii++;
                jj++;
            }
            
            if ((ii >= word1.length() && jj >= 
word2.length()) || jj >= word2.length()) {
                result.append(word1.charAt(i));
                i++;
                
            } else if (ii >= word1.length()) {
                result.append(word2.charAt(j));
                j++;
                
            } else {
                if (
word1.charAt(ii) > word2.charAt(jj)) {
                    result.append(word1.charAt(i));
                    i++;
                } else {
                    result.append(word2.charAt(j));
                    j++;
                }
            }
        }
        
        if (i < word1.length()) {
            result.append(word1.substring(i));
        }
        
        if (j < word2.length()) {
            result.append(word2.substring(j));
        }
        
        return result.toString();
    }
}
