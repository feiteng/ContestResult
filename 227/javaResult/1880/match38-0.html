class Solution {
     public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int len1 = word1.length();
        int len2 = word2.length();
        int idx1 = 0;
        int idx2 = 0;
        while (idx1<len1||idx2<len2){
            if(idx1==len1){
                sb.append(word2.charAt(idx2));
                idx2++;
                continue;
            }
            if(idx2==len2){
                sb.append(word1.charAt(idx1));
                idx1++;
                continue;
            }
            if(word1.charAt(idx1)>word2.charAt(idx2)){
                sb.append(word1.charAt(idx1));
                idx1++;
            }else if(word1.charAt(idx1)<word2.charAt(idx2)){
                sb.append(word2.charAt(idx2));
                idx2++;
            }else{
                if(select(word1,word2,idx1,idx2)){
                    sb.append(word1.charAt(idx1));
                    idx1++;
                }else{
                    sb.append(word2.charAt(idx2));
                    idx2++;
                }
            }
        }
        return sb.toString();
    }

    public boolean select(String word1, String word2,int idx1,int idx2){
        int len1 = word1.length();
        int len2 = word2.length();
        while (idx1<len1&&idx2<len2){
            if(word1.charAt(idx1)==word2.charAt(idx2)){
              idx1++;
                idx2++;
                continue;
            }
            return word1.charAt(idx1)>word2.charAt(idx2);
        }
        return  idx1<len1;
    }
}
