class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int i1 = 0, i2 = 0;
        
while (i1 < word1.length() && i2 < word2.length()) {
            char c1 = word1.charAt(i1), c2 = word2.charAt(i2);
            if (c1 > c2) {
                sb.append(c1);
                i1++;
            } else if (c2 > c1) {
                sb.append(c2);
                i2++;
            } else {
                
if (word1EqualLarger(i1, word1, i2, word2)) {
                    sb.append(c1);
                    i1++;
                } else {
                    sb.append(c2);
                    i2++;
                }
            }
        }
        while (i1 < word1.length()) {
            sb.append(word1.charAt(i1));
            i1++;
        }
        while (i2 < word2.length()) {
            sb.append(word2.charAt(i2));
            i2++;
        }
        return sb.toString();
    }
    
    private boolean word1EqualLarger(int i1, String word1, int i2, String word2) {
        while (i1 < word1.length() && i2 < word2.length() && word1.charAt(i1) == word2.charAt(i2)) {
            i1++;
            i2++;
        } 
        if (i2 == word2.length()) {
            return true;
        }
        if (i1 == word1.length()) {
            return false;
        }
        return word1.charAt(i1) > word2.charAt(i2);
    }
}
