class Solution {
    public String largestMerge(String word1, String word2) {
        int word1Pos = 0;
        int word2Pos = 0;
        StringBuilder mergerStrBuilder = new StringBuilder();
        while(word1Pos < word1.length() || word2Pos < word2.length()) {
            if (word1Pos < word1.length() && word2Pos < word2.length()) 
{
                if (word1.charAt(word1Pos) > word2.charAt(word2Pos)) {
                    mergerStrBuilder.append(word1.charAt(word1Pos));
                    word1Pos++;
                } else if (word1.charAt(word1Pos) == word2.charAt(word2Pos)) {
                    // quick check whoe is larger
                    if (word1.substring(word1Pos, 
word1.length()).compareTo(word2.substring(word2Pos, word2.length())) > 0) {
                        mergerStrBuilder.append(word1.charAt(word1Pos));
                        word1Pos++;
                    } else {
                        mergerStrBuilder.append(word2.charAt(word2Pos));
                        word2Pos++;
                    }
                } else {
                    mergerStrBuilder.append(word2.charAt(word2Pos));
                    word2Pos++;
                }
            
} else if (word1Pos >= word1.length()) {
                mergerStrBuilder.append(word2.charAt(word2Pos));
                word2Pos++;
            } else {
                mergerStrBuilder.append(word1.charAt(word1Pos));
                word1Pos++;
            
}
        }
        return mergerStrBuilder.toString();
    }
}
