class Solution {
    public String largestMerge(String s1, String s2) {
        StringBuilder res = new StringBuilder();
        int i = 0, j = 0;
        while (i < s1.length() || j < s2.length()) {
           if (s1.substring(i).compareTo(s2.substring(j)) > 0) {
               res.append(s1.charAt(i++));
           } else {
               res.append(s2.charAt(j++));
           }
        }
        while (j < s2.length()) {
            res.append(s2.charAt(j++));
        }
        while (i < s1.length()) {
            res.append(s1.charAt(i++));
        }
        return res.toString();
    }
}
