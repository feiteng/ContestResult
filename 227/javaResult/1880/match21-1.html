class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int p1 = 0;
        int p2 = 0;
        
while (p1 < word1.length() && p2 < word2.length()) {
            char c1 = word1.charAt(p1);
            char c2 = word2.charAt(p2);
            if (c1 > c2) {
                sb.append(c1);
                p1++;
            }
            else if (c1 < c2) {
                sb.append(c2);
                p2++;
            }
            else {
                if (word1.substring(p1).compareTo(
word2.substring(p2)) >= 0) {
                    sb.append(c1);
                    p1++;
                }
                else {
                    sb.append(c2);
                    p2++;
                }
            }
        }
        while (p1 < word1.length()) {
            char c1 = word1.charAt(p1);
            sb.append(c1);
            p1++;
        }
        while (p2 < word2.length()) {
            char c2 = 
word2.charAt(p2);
            sb.append(c2);
            p2++;
        }
        return sb.toString();
    }
}
