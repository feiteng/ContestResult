class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb=new StringBuilder();
        //StringBuilder sb1=new StringBuilder(word1);
        //StringBuilder sb2=new StringBuilder(word2);
        int temp1=0,temp2=0;
        while(temp1<word1.length()&&temp2<word2.length()){
            if(word1.charAt(temp1)<word2.charAt(temp2)){
                sb.append(word2.charAt(temp2));
                temp2++;
            }else if(word1.charAt(temp1)>word2.charAt(temp2)){
                sb.append(word1.charAt(temp1));
                temp1++;
            }else{
                if(word1.substring(temp1,word1.length()).compareTo(word2.substring(temp2,word2.length()))<=0)
{
                    sb.append(word2.charAt(temp2));
                    temp2++;
                }else{
                    sb.append(word1.charAt(temp1));
                    temp1++; 
                }
            }
        }
        if(temp1==word1.length()){
            sb.append(word2.substring(temp2,word2.length()));
        }else{
            sb.append(
word1.substring(temp1,word1.length()));
        }
        return sb.toString();
    }
}
