class Solution {
    public String largestMerge(String word1, String word2) {
        StringBuilder sb = new StringBuilder();
        int index1 = 0;
        int index2 = 0;
        while(index1<word1.length() || index2<word2.length() ){
            // System.out.println("index1: " + word1.charAt(index1));
            // System.out.println("index2: " + word2.charAt(index2));
            if(compare(word1, index1, word2, index2)){
                sb.append(word1.charAt(index1++));
            }else {
                sb.append(word2.charAt(index2++));
            }
        }
        return sb.toString();
    }
    
    private boolean compare(String w1, int index1, String w2, int index2){
        while(index1<w1.length() && index2<w2.length()){
             // System.out.println("index1: " + w1.charAt(index1));
             // System.out.println("index2: " + w2.charAt(index2));
            if(w1.charAt(index1) > w2.charAt(index2)){
                return true;
            }
            if(w1.charAt(index1) < w2.charAt(index2)){
                return false;
            }
            index1++;
            index2++;
        }
        
        if(index1==w1.length())
            return false;
        return true;
    }
}
