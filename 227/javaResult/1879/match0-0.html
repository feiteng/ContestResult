class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<> ((m, n) -> n -m);
        pq.offer(a);
        pq.offer(b);
        pq.offer(c);
        
        int res = 0;
        while (pq.size() >= 2) {
            int x = pq.poll();
            int y = pq.poll();
            
            res++;
            if (x > 1) {
                pq.offer(x-1);
            }
            if (y > 1) {
                pq.offer(y-1);
            }
        }
        return res;
    }
}
