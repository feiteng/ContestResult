class Solution {
    public int maximumScore(int a, int b, int c) {
        
        
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        
        
        pq.offer(a);
        pq.offer(b);
        pq.offer(c);
        
        
        int score =0;
        
        while(pq.size()>1){
            
            
            int o = pq.poll();
            int t = pq.poll();
            
            
            if(o-1!=0){
                pq.offer(o-1);
            }
            
            
            if(t-1!=0){
                pq.offer(t-1);
            }
            
            score++;
            
            
        }
        return score;
    }
}
