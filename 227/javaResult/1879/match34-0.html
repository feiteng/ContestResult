class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> queue = new PriorityQueue<>((o1, o2) -> o2 - o1);
        queue.offer(a);
        queue.offer(b);
        queue.offer(c);
        int res = 0;
        while (queue.size() > 1) {
            int first = queue.poll();
            int second = queue.poll();
            res++;
            if (--first > 0) queue.offer(first);
            if (--second > 0) queue.offer(second);
        }
        return res;
    }
}
