class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq=new PriorityQueue<>((x1,x2)->(x2-x1));
        pq.add(a);
        pq.add(b);
        pq.add(c);
        int ans=0;
        while(pq.size()>1){
            int max=pq.poll();
            int second=pq.poll();
            max--;
            second--;
            ans++;
            if(max>0) pq.add(max);
            if(second>0) pq.add(second);
        }
        return ans;
    }
}
