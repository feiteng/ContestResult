class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> queue = new PriorityQueue<>(Collections.reverseOrder());
        queue.add(a);
        queue.add(b);
        queue.add(c);
        int result=0;
        while(queue.size()>=2){
            int x=queue.poll()-1;
            int y=queue.poll()-1;
            if(x>0){
                queue.add(x);
            }
            
            if(y>0){
                queue.add(y);
            }
            result++;
        }
        return result;        
    }
}
