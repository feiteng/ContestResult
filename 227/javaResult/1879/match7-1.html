class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> que = new PriorityQueue<>((i,j)->j-i);
        que.add(a);
        que.add(b);
        que.add(c);
        int res = 0;
        while(que.size() > 1){
            int first = que.poll();
            int second = que.poll();
            res++;
            if(first > 1)
                que.add(first - 1);
            if(second > 1)
                que.add(second - 1);
        }
        
        return res;
    }
}
