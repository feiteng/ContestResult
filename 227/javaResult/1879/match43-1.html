class Solution {
    public int maximumScore(int a, int b, int c) {
        
        PriorityQueue<Integer> queue = new PriorityQueue<>((x, y) ->  y - x);
        queue.offer(a);
        queue.offer(b);
        queue.offer(c);
        
        int count = 0;
        while(queue.peek() != 0)
{
            int first = queue.poll();
            int secornd = queue.poll();
            if(secornd == 0) break;
            queue.offer(first - 1);
            queue.offer(secornd - 1);
            count++;
        }
        
        return count;
        
    }
}
