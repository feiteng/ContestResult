class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> q=new PriorityQueue<>((o1, o2) -> o2-o1);
        q.add(a);
        q.add(b);
        q.add(c);
        int sum=0;
        while(true){
            int t1=q.poll();
            int t2=q.poll();
            if(t2==0)break;
            sum++;
            t1--;
            t2--;
            q.add(t1);
            q.add(t2);
        }
        return sum;
    }
}
