class Solution {
    public int maximumScore(int a, int b, int c) {
        Queue<Integer> pq = new PriorityQueue<>((x, y) -> y - x);
        pq.offer(a);
        pq.offer(b);
        pq.offer(c);
        int res = 0;
        while (pq.size() > 1){
            int k  =pq.poll();
            int m = pq.poll();
            k--;
            m--;
            res++;
            if (k > 0) pq.offer(k);
            if (m > 0) pq.offer(m);
        }
        return res;
    }
}
