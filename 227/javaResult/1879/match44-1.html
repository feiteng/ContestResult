class Solution {
    public int maximumScore(int a, int b, int c) {
        int[] arr = {a, b, c};
        Arrays.sort(arr);
        a = arr[0];
        b = arr[1];
        c = arr[2];
        
int res = 0;
        while (a + b > c) {
            a--;
            b--;
            res++;
        }
        return res + a + b;
    }
}
