class Solution {
 public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> q = new PriorityQueue<>((x, y) -> Integer.compare(y, x));

        q.add(a);
        q.add(b);
        q.add(c);
        
        int ans = 0;
        while (q.size() >= 2) {
            int a1 = q.poll();
            int a2 = q.poll();
            
            ans++;
            a1--;
            a2--;
            if (a1 > 0) q.add(a1);
            if (a2 > 0) q.add(a2);
        }
        return ans;
    }
}
