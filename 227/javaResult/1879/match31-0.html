class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((x, y) -> Integer.compare(y, x));
        pq.offer(a);
        pq.offer(b);
        pq.offer(c);
        int ans = 0;
        while (pq.size() > 1) {
            int i = pq.poll(), j = pq.poll();
            if (--i > 0) pq.offer(i);
            if (--j > 0) pq.offer(j);
            ++ans;
        }
        return ans;
    }
}
