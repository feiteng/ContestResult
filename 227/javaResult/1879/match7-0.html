class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(3, (o1, o2) -> o2 - o1);
        pq.offer(a);
        pq.offer(b);
        pq.offer(c);
        int res = 0;
        while (pq.size() > 1) {
            int t1 = pq.poll(), t2 = pq.poll();
            res ++;
            t1 --;
            t2 --;
            if (t1 > 0)
                pq.offer(t1);
            if (t2 > 0)
                pq.offer(t2);
        }
        
        return res;
    }
}
