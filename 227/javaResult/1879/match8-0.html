class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((Integer m,Integer n)->n-m);
        pq.add(a);
        pq.add(b);
        pq.add(c);
        int res = 0;
        while(pq.size()>1){
            int k = pq.poll();
            int l = pq.poll();
            res++;
            if(--k!=0)
                pq.add(k);
            if(--l!=0)
                pq.add(l);
        }
        return res;
    }
}
