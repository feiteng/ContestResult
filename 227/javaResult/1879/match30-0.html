class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((x,y)->y-x);
        pq.add(a);
        pq.add(b);
        pq.add(c);
        int score=0;
        while(!pq.isEmpty() && pq.size()>=2) {
            int top1 = pq.remove();
            top1--;
            int top2 = pq.remove();
            top2--;
            score++;
            if(top1 > 0) {
                pq.add(top1);
            }
            if(top2>0) {
                pq.add(top2);
            }
        }
        
        return score;
    }
}
