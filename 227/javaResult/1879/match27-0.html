class Solution {
    public int maximumScore(int a, int b, int c) {
        int[] array = new int[3];
        array[0] = a;
        array[1] = b;
        array[2] = c;
        Arrays.sort(array);
        
        int res = 0;
        while(array[1] > 0){
            array[1]--;
            array[2]--;
            Arrays.sort(array);
            res++;
        }
        
        return res;
    }
}
