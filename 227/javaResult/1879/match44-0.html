class Solution {
    // a, b, c
    
    // a+b == c
    // return c
    
    // a+b < c
    // return a+b
    
    // a+b > c
    // return 
    // a,c
    // b, c-a
    // 
    // a = 4, b = 4, c = 6
    // 0 4 2
    // 1 2 4
    // 
    // 3 4 6
    // 0 4 3  3+3
    // 2 3 6  1+5
    
    // 4 4 6
    // 0 4 2   4+2
    // 3 3 6   1+6=7
    public int maximumScore(int a, int b, int c) {
        int[] piles = {a, b, c};
        Arrays.sort(piles);
        
        int res = 0;
        
        while (piles[0]+piles[1] > piles[2]) {
            piles[0]--;
            piles[1]--;
            res++;
        }
        
        return res+piles[0]+piles[1];
    }
}
