class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> q = new PriorityQueue<>((i, j) -> j - i);
        q.offer(a);
        q.offer(b);
        q.offer(c);
        
        int res = 0;
        while (q.size() >= 2) {
            int i = q.poll();
            int j = q.poll();
            res++;
            
            if (i - 1 > 0) q.offer(i - 1);
            if (j - 1 > 0) q.offer(j - 1);
        }
        
        return res;
    }
}
