class Solution {
    public int maximumScore(int a, int b, int c) {
        int[] nums = new int[3];
        nums[0] = a;
        nums[1] = b;
        nums[2] = c;
        Arrays.sort(nums);
        
        if (nums[0] + nums[1] <= nums[2]) {
            return nums[0] + nums[1];
        } else {
            return (nums[0] + nums[1] + nums[2]) / 2;
        }
    }
}
