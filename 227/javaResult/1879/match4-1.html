class Solution {
    public int maximumScore(int a, int b, int c) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((x, y) -> (y - x));
        pq.add(a);
        pq.add(b);
        pq.add(c);
        int cnt = 0;
        while(pq.size() > 1) {
            int x = pq.poll();
            int y = pq.poll();
            x -= 1;
            y -= 1;
            cnt++;
            if(x > 0) {
                pq.add(x);
            }
            if(y > 0) {
                pq.add(y);
            }
        }
        return cnt;
    }
}
