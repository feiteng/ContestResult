class Solution {
    public int maximumScore(int a, int b, int c) 
    {
        PriorityQueue<Integer> pq = new PriorityQueue<>((x, y) -> Integer.compare(y, x));
        pq.offer(a); pq.offer(b); pq.offer(c);
        int ans = 0;
        while (pq.size() > 1) {
            int maxi = pq.poll();
            int smaxi = pq.poll();
            // System.out.println(maxi + " " + smaxi);
            ans++;
            if (maxi-1>0) pq.offer(maxi-1);
            if (smaxi-1>0) pq.offer(smaxi-1);
        }
        return ans;
