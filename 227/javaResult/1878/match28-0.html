class Solution {
    public boolean check(int[] nums) {
        int mini = 0;

        for (int i = 1; i < nums.length; i++) {
            if (nums[i] <= nums[mini] && nums[i] < nums[i-1]) {
                mini = i;
            }
        }
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[(mini+i + 1) % nums.length] < nums[(mini + i) % nums.length] ) return false;
        }
        return true;
    }
}
