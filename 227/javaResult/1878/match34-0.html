class Solution {
    // [3,4,5,1,2]
    // 
    // [2,1,3,4]
    public boolean check(int[] nums) {
        int num = 0;
        int index = -1;
        
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] < nums[i-1]) {
                num++;
                index = i;
                if (num > 1) {
                    return false;
                }
            }
        }
        
        if (num == 0) {
            return true;
        }
        
        if (index == -1) {
            return false;
        }
        
        
        return nums[nums.length-1] <= nums[0];
        
        
        // return num == 1;
    }
}
