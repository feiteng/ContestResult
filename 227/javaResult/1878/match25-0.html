class Solution {
    public boolean check(int[] nums) {
        int sum = 0;
        for(int i = 1; i < nums.length; i++) {
            if(nums[i] < nums[i - 1]) {
                sum++;
                if(sum > 1)
                    return false;
            }    
        }
        if(sum == 0)
            return true;
        if(nums[nums.length - 1] <= nums[0])
            return true;
        return false;
    }
}
