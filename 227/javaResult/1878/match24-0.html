class Solution {
    public boolean check(int[] nums) {
        int N = nums.length;
        for (int i = 0; i < N; i++) {
            boolean ans = true;
            for (int j = i + 1; j < i + N; j++) {
                ans &= nums[j % N] >= nums[(j - 1) % N];
            }
            
            if (ans) return true;
        }
        
        return false;
    }
}
