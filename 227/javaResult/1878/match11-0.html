class Solution {
    public boolean check(int[] a) 
    {
        int n = a.length;
        for (int i = 0; i < n; i++) {
            boolean ok = true;
            int prev = -1;
            for (int j = 0; j < n; j++) {
                int current = a[(i+j)%n];
                if (prev != -1 && current < prev) {
                    ok = false;
                    break;
                }
                prev = current;
            }
            if (ok) return true;
        }
        return false;
    }
}
