class Solution {

    
    public boolean check(int[] nums) {
        int n = nums.length;
        for(int i = 0; i < n; i++){
            boolean ok = true;
            for(int j = 0; j < n - 1; j++){
                int e = nums[(j + i) % n];
                int next = nums[(j + i + 1) % n];
                if(e > next){
                    ok = false;
                
}
            }
            if(ok){
                return true;
            }
        }
        return false;
    }
}
