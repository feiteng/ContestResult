class Solution {
    public boolean check(int[] nums) {
        int len = nums.length;
        for (int i = 0; i < len; i++){
            boolean isFind = true;
            for (int j = i+1; j < i + len; j++){
                int ind1 = j % len;
                int ind2 = (j - 1) % len;
                // System.out.println(ind1);
                // System.out.println(ind2);
                // System.out.println("j: "+j);
                if (nums[ind1] < nums[ind2]){
                    // System.out.println("wrong: " + ind1);
                    isFind = false;
                    break;
                }
            }
            if (isFind) return true;
        }
        
        return false;
    }
}
