class Solution {
    public boolean check(int[] nums) {        
        int NumberOfTurn = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i != nums.length - 1 && nums[i] > nums[i + 1]) {
                NumberOfTurn++;
            }
        }
        
        return (NumberOfTurn == 1 && nums[0] >= nums[nums.length - 1]) || NumberOfTurn == 0;
    }
}
