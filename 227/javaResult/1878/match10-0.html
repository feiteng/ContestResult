class Solution {
    public boolean check(int[] nums) {
        int len = nums.length;
        for (int i = 0; i < len; i++) {
            boolean result = true;
            for (int j = i + 1, k = 0; k < len - 1; j++, k++) {
                j %= len;
                if (nums[j] - nums[(j - 1 + len) % len] < 0) {
                    result = false;
                    break;
                }
            }
            if (result) {
                return true;
            }
        }
        return false;
    }
}
