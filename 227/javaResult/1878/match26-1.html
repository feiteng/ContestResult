class Solution {
    public boolean check(int[] nums) {
        int minv = nums[0];
    int mini = 0;
    int n = nums.length;
    for (int i = 0; i < n; i++) {
      if (nums[i] <= minv) {
        mini = i;
        minv = nums[i];
      }
    }
      while (mini > 0 && nums[mini-1] == minv) {
      mini--;
    }
    
for (int i = mini + 1; i < mini + n; i++) {
      if (nums[i % n] < nums[(i-1) % n]) {
        return false;
      }
    }
    return true;
    }
}
