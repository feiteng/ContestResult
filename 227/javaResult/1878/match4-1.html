class Solution {
    public boolean check(int[] nums) {
        int n = nums.length;
        for (int start = 0; start < n; start++) {
            boolean valid = true;
            for (int i = 1; i < n; i++) {
                int i1 = (i - 1 + start) % n;
                int i2 = (i + start) % n;
                if (nums[i1] > nums[i2]) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                return true;
            }
        }
        return false;
    }
}
