class Solution {
    public boolean check(int[] nums) {
        int[] sources = new int[nums.length];
        for (int i =0; i < nums.length; i ++){
            sources[i] = nums[i];
        }
        Arrays.sort(sources);
        for (int k = 0; k < nums.length; k ++){
            // System.out.println(k);
            // 旋转k次
            boolean success = true;
            for (int i = 0; i < nums.length; i ++){
                if (nums[i] != sources[(i + k) % nums.length]){
                    success = false;
                    break;
                }
            }
            if (success){
                return true;
            }
        }
        return false;
    }
}
