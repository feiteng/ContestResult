class Solution {
    public boolean check(int[] nums) {
        int len = nums.length;
        int mark = 0;
        for(int i = 0 ; i < len -1; i++){
            if(nums[i]>nums[i+1])
                ++mark;
            if(mark>1 || (mark==1 && nums[len-1]>nums[0]) )
                return false;
        }
        return true;
    }
}
