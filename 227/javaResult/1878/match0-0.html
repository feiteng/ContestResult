class Solution {
    public boolean check(int[] nums) {
        // 无逆序, 或逆序一次, 且头大于等于尾(即保证逆序仅一次)
        int n = nums.length;
        int cnt = 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] < nums[i - 1]) {
                cnt++;
            }
        }
        if (cnt > 1) {
            return false;
        }
        if (cnt == 1 && nums[0] < nums[n - 1]) {
            return false;
        }
        return true;
    }
}
