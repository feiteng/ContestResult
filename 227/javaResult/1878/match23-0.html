class Solution {
    public boolean check(int[] nums) {

        int n = nums.length;
        for (int i = 0; i < n; i++) {
            int num = 0;
            for (int j = 1; j < n; j++) {
                if (nums[(i + j) % n] >= nums[(i + j - 1) % n])
                    num++;
                else
                    break;
            }
            if (num == n - 1)
                return true;
        }

        return false;
    }
}
