class Solution {
    public boolean check(int[] nums) {
        int len = nums.length;
        for (int i = 0; i < nums.length; i++) {
            boolean find = true;
            for (int j = 0; j < len - 1; j++) {
                int preIndex = (i + j + 1) % len;
                int index = (i + j) % len;
                
if (nums[preIndex] < nums[index]) {
                    find = false;
                    break;
                }
            }
            if (find) {
                return true;
            }
        }
        return false;
    }
}
