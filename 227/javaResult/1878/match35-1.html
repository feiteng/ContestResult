class Solution {
    public boolean check(int[] nums) {
        int min = nums[0];
        int a = 0;
        for(int i = 1; i < nums.length; i++) {
            if(nums[i] < nums[i - 1]) {
                if(a == 0) {
                    a = 1;
                } else {
                    return false;
                }
            
}
            if(a == 1 && nums[i] > min) return false;
        }
        return true;
    }
}
