class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
       /* nums = l = r = vector<int>(500, 0);
        for (int i = 0; i < 500; ++i) {
            nums[i] = i;
            l[i] = 0;
            r[i] = 499;
        }*/
        
        vector<bool> res;
        
        for (int i = 0; i < l.size(); ++i) {
            vector<int> sub(nums.begin() + l[i], nums.begin() + r[i] + 1);
            sort(sub.begin(), sub.end());
            int d = sub[1] - sub[0];
            bool flag = true;

            for (int j = 1; j + 1 < sub.size(); ++j) {
                if (sub[j + 1] - sub[j] != d) {
                    flag = false;
                    break;
                }
            }
            res.push_back(flag);
        }
        
        return res;
        
    }
};
