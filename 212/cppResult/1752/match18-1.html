class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        // 500 * 500 
        int n = l.size();
        vector<bool> ans(n, true);
        for (int i = 0; i < n; ++i) {
            vector<int> sa(nums.begin() + l[i], nums.begin() + r[i] + 1);
            //cout << '5';
            sort(sa.begin(), sa.end());
            int d = sa[1] - sa[0];
            for (int j = 1; j + 1 < sa.size(); ++j) {
                if (sa[j + 1] - sa[j] != d) {
                    ans[i] = false;
                    break;
                }
                    
            }
        }
        return ans;
    }
};
