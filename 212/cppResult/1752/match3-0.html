class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        int q = l.size();
        vector<bool> ans(q, 0);
        for(int i = 0; i < q; ++i) {
            vector<int> v; v.clear();
            for(int j = l[i]; j <= r[i]; ++j) v.push_back(nums[j]);
            
            sort(v.begin(), v.end());
            ans[i] = 1;
            for(int j = 1; j < v.size(); ++j) {
                if(v[j] - v[j - 1] != v[1] - v[0]) {
                    ans[i] = 0;
                    break;
                }
            }
        }
        return ans;
    }
};
