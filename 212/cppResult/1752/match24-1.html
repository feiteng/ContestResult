class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        int m = l.size();
        int n = nums.size();
        vector<bool> ans(m, true);
        for (int i = 0; i < m; ++i) {
            vector<int> tmp(nums.begin() + l[i], nums.begin() + r[i] + 1);
            sort(tmp.begin(), tmp.end());
            
            int d = tmp[1] - tmp[0];

            for (int j = 2; j < tmp.size(); ++j) {
                if (tmp[j] - tmp[j-1] != d) {
                    ans[i] = false;
                    break;
                }
            }
        }
        return ans;
    }
};
