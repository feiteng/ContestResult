class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for(int i=0;i<l.size();i++)
        {
            int l1=l[i],r1=r[i];
            vector<int> v1;
            for(int j=l1;j<=r1;j++)v1.push_back(nums[j]);
            sort(v1.begin(),v1.end());
            int diff=v1[1]-v1[0];
            bool ok=true;
            for(int j=1;j<v1.size();j++)
            {
                if(v1[j]-v1[j-1]!=diff)
                {
                    ok=false;
                    break;
                }
            }
            ans.push_back(ok);
        }
        return ans;
    }
};
