class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for (int i = 0 ; i < l.size() ; i++) {
            vector<int> V;
            for (int j = l[i] ; j <= r[i] ; j++)
                V.push_back(nums[j]);
            sort(V.begin(), V.end());

            bool flag = true;
            for (int i = 2 ; i < V.size() ; i++) {
                if (V[i] - V[i - 1] != V[1] - V[0])
                    flag = false;
            }
            ans.push_back(flag);
        }
        return ans;
    }
};
