class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        int n = nums.size();
        int m = l.size();
        for(int i = 0; i < m; ++i) {
            vector<int> tmp;
            tmp.assign(nums.begin() + l[i], nums.begin() + r[i] + 1);
            sort(tmp.begin(), tmp.end());
            int len = tmp[1] - tmp[0];
            bool flag = true;
            for(int j = 1; j < tmp.size(); ++j) {
                if(tmp[j] - tmp[j - 1] != len) {
                    flag = false;
                    break;
                }
            }
            ans.push_back(flag);
        }
        return ans;
    }
};
