class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for(int i = 0; i < l.size(); ++i) {
            vector<int> arr;
            for(int j = l[i]; j <= r[i]; ++j) {
                arr.push_back(nums[j]);
            }
            sort(arr.begin(), arr.end());
            int d = arr[1] - arr[0];
            bool f = true;
            for(int j = 1; j < arr.size(); ++j) {
                if(d != arr[j] - arr[j - 1]) {
                    f = false;
                    break;
                }
            }
            ans.push_back(f);
        }
        return ans;
    }
};
