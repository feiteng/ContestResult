class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& A, vector<int>& l, vector<int>& r) {
        vector<bool> v;
        for(int i = 0; i < l.size(); i++) {
            vector<int> v1;
            for(int j = l[i]; j <= r[i]; j++) {
                v1.push_back(A[j]);
            }
            if(v1.size() < 2) {
                v.push_back(0);
                continue;
            }
            sort(v1.begin(), v1.end());
            int d = v1[1] - v1[0];
            bool res = true;
            for(int j = 0; j + 1 < v1.size(); j++) {
                if(d != v1[j + 1]  - v1[j]) {
                    res = false;
                    break;
                }
            }
            v.push_back(res);
        }
        return v;
    }
};
