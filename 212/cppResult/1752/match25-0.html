class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        int n = nums.size();
        int m = l.size();
        vector<bool> res;
        
        for (int i = 0; i < m; ++i) {
            vector<int> tmp(nums.begin() + l[i], nums.begin() + r[i] + 1);
            sort(tmp.begin(), tmp.end());
            bool flag = true;
            int d = tmp[1] - tmp[0];
            for (int i = 2; i < tmp.size(); ++i) {
                if (tmp[i] - tmp[i-1] != d) {
                    flag = false;
                    break;
                }
            }
            res.push_back(flag);
        }
        return res;
    }
};
