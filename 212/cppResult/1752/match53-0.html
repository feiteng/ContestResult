class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> res;
        for (int i=0;i<l.size();i++){
            int L = l[i];
            int R = r[i];
            vector<int> temp;
            for (int i=L;i<=R;i++) temp.emplace_back(nums[i]);
            stable_sort(temp.begin(),temp.end());
            bool flag = 1;
            for (int i=2;i<temp.size();i++){
                if (temp[i]-temp[i-1]!=(temp[1]-temp[0])){
                    flag=0;
                    break;
                }
            }
            res.push_back(flag);
        }
        return res;
    }
    
};
