class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        int n = nums.size(), m = l.size();
        
        vector<bool> res(m);
        for (int i = 0; i < m; ++i) {
            // int ll = l[i], rr = r[i];
            // if (rr - ll < 1) {
            //     res[i] = false;
            //     continue;
            // }
            
            vector<int> v(nums.begin() + l[i], nums.begin() + r[i] + 1);
            sort(v.begin(), v.end());
            
            int diff = v[1] - v[0];
            res[i] = true;
            for (int j = 2; j < v.size(); ++j) {
                if (v[j] - v[j-1] != diff) {
                    res[i] = false;
                    break;
                }
            }
        }
        
        return res;
    }
};
