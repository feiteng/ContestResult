class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector <bool> ans;
        for(int i = 0; i < l.size(); i++){
            int x = l[i];
            int y = r[i];
            vector <int> v;
            for(int j = x; j <= y; j++){
                v.push_back(nums[j]);
            }
            if(v.size() == 1){
                ans.push_back(false);
                continue;
            }
            sort(v.begin(), v.end());
            int d = v[1] - v[0];
            int ok = true;
            for(int j = 2; j < v.size(); j++){
                if(v[j] - v[j - 1] != d){
                    ok = false;
                    break;
                }
            }
            ans.push_back(ok);
        }
        return ans;
    }
};
