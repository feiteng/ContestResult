class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        int n = l.size();
        
        for (int i=0; i<n; i++) {
            vector<int> data;
            for (int j=l[i]; j<=r[i]; j++) data.push_back(nums[j]);
            sort(data.begin(), data.end());
            
            bool valid = true;
            for (int i=2; i<data.size(); i++) {
                if (data[i]-data[i-1] != data[1]-data[0]) {
                    valid = false;
                    break;
                }
            }
            ans.push_back(valid);
        }
        
        return ans;
    }
};
