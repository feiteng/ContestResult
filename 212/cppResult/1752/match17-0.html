class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        
        vector<bool> ret;
        
        for(int i=0;i<l.size();i++) {
            vector<int> V;
            for(int j=l[i];j<=r[i];j++)
                V.push_back(nums[j]);
            
            sort(V.begin(),V.end());
            
            bool ok = true;
            for(int j=1;j<V.size();j++)
                ok &= (V[j]-V[j-1]==V[1]-V[0]);
            ret.push_back(ok);
        }
        return ret;
        
        
        
    }
};
