class Solution {
    bool check(vector<int> &A, int L, int R) {
        vector<int> c(begin(A) + L, begin(A) + R + 1);
        sort(begin(c), end(c));
        int d = c[1] - c[0];
        for (int i = 1; i < c.size(); ++i) {
            if (c[i] - c[i - 1] != d) return false;
        }
        return true;
    }
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& A, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for (int i = 0; i < l.size(); ++i) {
            ans.push_back(check(A, l[i], r[i]));
        }
        return ans;
    }
};
