class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ret;
        for (int t = 0; t < l.size(); ++t) {
            vector<int> a;
            for (int i = l[t]; i <= r[t]; ++i)
                a.push_back(nums[i]);
            sort(a.begin(), a.end());
            bool f = true;
            for (int i = 1; i < a.size(); ++i)
                f = f && a[i] - a[i - 1] == a[1] - a[0];
            ret.push_back(f);
        }
        return ret;
    }
};
