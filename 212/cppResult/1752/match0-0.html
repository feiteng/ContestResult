class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for(int i = 0;i < l.size(); ++i){
            vector<int> vec;
            for(int j = l[i];j <= r[i]; ++j) vec.push_back(nums[j]);
            sort(vec.begin(),vec.end()); bool flag = true;
            for(int j = 2;j < vec.size(); ++j){
                if(vec[j]-vec[j-1] != vec[1]-vec[0]) flag = false;
            }
            ans.push_back(flag);
        }
        return ans;
    }
};
