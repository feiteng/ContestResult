class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> res;
        for (int i = 0; i < l.size(); i++) {
            vector<int> v;
            for (int j = l[i]; j <= r[i]; j++)
                v.push_back(nums[j]);
            sort(v.begin(), v.end());
            bool yes = true;
            int dif = v[1] - v[0];
            for (int i = 2; i < v.size() && yes; i++) {
                if (v[i] - v[i-1] != dif) yes = false;
            }
            res.push_back(yes);
        }
        return res;
    }
};
