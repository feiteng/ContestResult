class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for(int i = 0; i < l.size(); i++){
            int ll = l[i];
            int rr = r[i];
            vector<int> tmp;
            for(int j = ll; j <= rr; j++) tmp.push_back(nums[j]);
            sort(tmp.begin(), tmp.end());
            bool f = true;
            // if(tmp.size() < 3){
            //     ans.push_back(false);
            //     continue;
            // }
            for(int j = 1; j < tmp.size(); j++){
                if(j + 1 < tmp.size() && tmp[j+1] - tmp[j] != tmp[j] - tmp[j-1]){
                    f = false;
                    break;
                }
            }
            ans.push_back(f);
        }
        return ans;
    }
};
