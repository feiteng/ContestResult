class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ret;
        for(int q = 0; q < l.size(); q++){
            vector<int> t;
            for(int i = l[q]; i<=r[q]; i++){
                t.push_back(nums[i]);
            }
            sort(t.begin(), t.end());
            bool b = 1;
            for(int i = 1; i < t.size(); i++){
                if(t[i] - t[i-1] != t[1] - t[0]){
                    b = 0;
                    break;
                }
            }
            ret.push_back(b);
        }
        return ret;
    }
};
