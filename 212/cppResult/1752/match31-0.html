class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& a, vector<int>& l, vector<int>& r) {
        vector<bool> ans(l.size());
        for (int i = 0; i < l.size(); ++i) {
            vector<int> b(a.begin() + l[i], a.begin() + r[i] + 1);
            sort(b.begin(), b.end());
            ans[i] = true;
            for (int j = 1; j < b.size(); ++j) {
                if (b[j] - b[j - 1] != b[1] - b[0]) {
                    ans[i] = false;
                    break;
                }
            }
        }
        return ans;
    }
};
