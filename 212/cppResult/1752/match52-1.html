class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> ans;
        for(int i = 0; i < l.size(); i ++) {
            int L = l[i];
            int R = r[i];
            
            vector<int> s;
            for(int j = L; j <= R; j ++) {
                s.push_back(nums[j]);
            }
            
            sort(s.begin(), s.end());
            
            bool f = true;
            

            for(int j = 1; j < s.size(); j ++) {
                if(s[j]-s[j-1] != s[1]-s[0]) {
                    f = false;
                    break;
                }
            }
            
            ans.push_back(f);
        }
        
        return ans;
    }
};
