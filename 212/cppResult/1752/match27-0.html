class Solution {
public:
    vector<bool> checkArithmeticSubarrays(vector<int>& nums, vector<int>& l, vector<int>& r) {
        vector<bool> output(l.size());
        for (int query = 0; query < l.size(); query++)
        {
            int begin = l[query], end = r[query];
            vector<int> data(nums.begin() + begin, nums.begin() + end + 1);
            sort(data.begin(), data.end());
            bool flag = true;
            int interval = data[1] - data[0];
            for (int i = 1; i < data.size(); i++)
            {
                if (data[i] - data[i - 1] != interval)
                {
                    flag = false;
                    break;
                }
            }
            output[query] = flag;
        }
        return output;
    }
};
