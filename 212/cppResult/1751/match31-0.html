class Solution {
public:
    char slowestKey(vector<int>& r, string key) {
        char ans = key[0] ;
        int ma = r[0] ;
        for (int i = 1 ; i < r.size() ; i++) {
            if (r[i] - r[i - 1] > ma) {
                ma = r[i] - r[i - 1] ;
                ans = key[i] ;
            } else if (ma == r[i] - r[i - 1]) {
                ans = max(ans , key[i]) ;
            }
        }
        return ans ;
    }
};
