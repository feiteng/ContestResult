class Solution {
public:
    char slowestKey(vector<int>& rt, string kp) {
        int prev = 0;
        int letters[26] = {};
        for (int i = 0; i < rt.size(); ++i) {
            letters[kp[i] - 'a'] = max(letters[kp[i] - 'a'], rt[i] - prev);
            prev = rt[i];
        }
        int mt = *max_element(letters, letters + 26);
        for (int i = 25; i >= 0; --i) {
            if (mt == letters[i]) return 'a' + i;
        }
        return 'a';
    }
};
