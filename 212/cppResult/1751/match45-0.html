class Solution {
public:
    char slowestKey(vector<int>& a, string s) {
        a.insert(a.begin(),0);
        char ans = s[0];
        int t = a[1];
        for(int i=1; i<=s.size(); i++){
            if(a[i]-a[i-1]>t){
                t = a[i]-a[i-1];
                ans = s[i-1];
            }
            else if(a[i]-a[i-1]==t){
                ans = max(ans,s[i-1]);
            }
        }
        return ans;
    }
};
