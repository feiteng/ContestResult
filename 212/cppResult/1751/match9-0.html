class Solution {
public:
    char slowestKey(vector<int>& t, string keysPressed) {
        int len = t.size();
        char res= keysPressed[0];
        int time=t[0];
        for (int i = 1; i < len; i++)
        {
            if (t[i]-t[i-1]>time)
            {
                time = t[i] - t[i - 1];
                res = keysPressed[i];
            }
            else if (t[i] - t[i - 1] == time && keysPressed[i]>time) {
                res = keysPressed[i];
            }
        }
        return res;
    }
};
