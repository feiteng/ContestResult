class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        char res = keysPressed[0];
        int mx = releaseTimes[0];
        for (int i=1; i<releaseTimes.size(); ++i) {
            int t = releaseTimes[i] - releaseTimes[i-1];
            if (t >= mx) {
                mx = t;
                res = keysPressed[i];
            }
        }
        
        return res;
    }
};
