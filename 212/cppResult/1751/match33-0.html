class Solution {
public:
    char slowestKey(vector<int>& r, string k) {
        char ans = k[0];
        int tmp = r[0];
        for (int i = 1; i < k.length(); i++) {
            if (r[i] - r[i-1] > tmp) {
                tmp = r[i] - r[i-1];
                ans = k[i];
            } else if (r[i] - r[i-1] == tmp) {
                ans = max(ans, k[i]);
            }
        }
        return ans;
    }
};
