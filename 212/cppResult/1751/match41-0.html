class Solution {
public:
    char slowestKey(vector<int>& t, string s) {
        char res = s[0];
        int best = t[0];
        for (int i = 1; i < t.size(); ++i) {
            if (t[i] - t[i-1] > best) {
                res = s[i];
                best = t[i] - t[i-1];
            } else if (t[i] - t[i-1] == best && s[i] > res) {
                res = s[i];
            }
            // cout << res << " " << best << endl;
        }
        return res;
    }
};
