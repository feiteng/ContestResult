class Solution {
public:
    char slowestKey(vector<int>& v, string s) {
        char ch=s[0];
        int dur=v[0];
        for(int i=1;i<v.size();i++)
        {
            if(v[i]-v[i-1]>dur)
            {
                dur=v[i]-v[i-1];
                ch=s[i];
            }
            else if(v[i]-v[i-1]==dur && s[i]>ch)
            {
                ch=s[i];
            }
        }
        return ch;
        
    }
};
