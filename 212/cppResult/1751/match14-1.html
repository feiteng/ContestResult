class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        char ans;
        int len = 0;
        for (int i = 0; i < releaseTimes.size(); ++ i) {
            int dur = (i == 0) ? releaseTimes[i] : (releaseTimes[i] - releaseTimes[i - 1]);
            if (dur > len || (dur == len && ans < keysPressed[i])) {
                len = dur;
                ans = keysPressed[i];
            }
        }
        return ans;
    }
};
