class Solution {
public:
    char slowestKey(vector<int>& t, string s) {
        char res = s[0];
        int maxt = t[0];
        for (int i = 1; i < s.size(); i++) {
            if (t[i]-t[i-1] > maxt) {
                maxt = t[i]-t[i-1];
                res = s[i];
            } else if (t[i]-t[i-1] == maxt) {
                res = max(res, s[i]);
            }
        }
        return res;
    }
};
