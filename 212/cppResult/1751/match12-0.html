class Solution {
public:
  char slowestKey(vector<int>& releaseTimes, string keysPressed) {
    int max_duration = releaseTimes[0];
    char key = keysPressed[0];
    for (int i=1; i<releaseTimes.size(); i++) {
      int tmp = releaseTimes[i] - releaseTimes[i-1];
      if (tmp > max_duration) {
        max_duration = tmp;
        key = keysPressed[i];
      } else if (tmp == max_duration) {
        key = max(key, keysPressed[i]);
      }
    }
    return key;
  }
};
