class Solution {
public:
    char slowestKey(vector<int>& r, string k) {
        char ret = k[0];
        int mx = r[0];
        for (int i = 1; i < r.size(); ++i) {
            if (r[i] - r[i - 1] > mx) {
                mx = r[i] - r[i - 1];
                ret = k[i];
            } 
            else if (r[i] - r[i - 1] == mx) {
                ret = max(k[i], ret);
            }
        }
        return ret;
    }
};
