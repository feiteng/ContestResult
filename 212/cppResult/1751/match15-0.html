class Solution {
 public:
  char slowestKey(vector<int>& releaseTimes, string keysPressed) {
    char res = 'a';
    int max_d = 0;
    int prev = 0;
    for (int i = 0; i < releaseTimes.size(); i++) {
      int cur_d = releaseTimes[i] - prev;
      if ((max_d < cur_d) || (max_d == cur_d && res < keysPressed[i])) {
        max_d = cur_d;
        res = keysPressed[i];
      }
      prev = releaseTimes[i];
    }
    return res;
  }
};
