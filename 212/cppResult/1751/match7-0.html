class Solution {
public:
    char slowestKey(vector<int>& r, string p) {
        char k = p[0];
        int t = r[0];
        for (int i = 1; i < r.size(); ++i)
        {
            int x = r[i] - r[i - 1];
            if (x > t || (x == t && k < p[i]))
            {
                k = p[i];
                t = x;
            }
        }
        return k;
    }
};
