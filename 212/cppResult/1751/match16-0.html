class Solution {
public:
    char slowestKey(vector<int>& rt, string kp) {
        char res = kp[0];
        int mx = rt[0];
        for (int i = 1; i < rt.size(); ++i) {
            if (rt[i] - rt[i-1] > mx) {
                mx = rt[i] - rt[i-1];
                res = kp[i];
            } else if (rt[i] - rt[i-1] == mx) {
                res = max(kp[i], res);
            }
        }
        return res;
    }
};
