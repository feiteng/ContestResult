class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        char ret = 'a'; int res = 0;
        for (int i = 0; i < releaseTimes.size(); i++) {
            int t = 0;
            if (i) t = releaseTimes[i] - releaseTimes[i - 1];
            else t = releaseTimes[i];


            if (t > res) {
                ret = keysPressed[i];
                res = t;
            }
            else if (t == res) {
                ret = max(ret, keysPressed[i]);
            }
        }
        return ret;
    }
};
