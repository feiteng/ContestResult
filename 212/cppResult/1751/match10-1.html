class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        int mxtime = releaseTimes[0];
        char ret = keysPressed[0];
        int N = releaseTimes.size();
        for (int i=1;i<N;i++) {
            if (releaseTimes[i] - releaseTimes[i - 1] > mxtime) {
                mxtime = releaseTimes[i] - releaseTimes[i - 1];
                ret = keysPressed[i];
            } else if (releaseTimes[i] - releaseTimes[i - 1] == mxtime) {
                ret = max(ret, keysPressed[i]);
            }
        }
        return ret;
    }
};
