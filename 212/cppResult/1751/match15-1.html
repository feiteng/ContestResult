class Solution
{
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed)
    {
        char key = 0;
        int duration = 0;
        int time = 0;
        for (int i = 0; i < keysPressed.size(); i++) {
            int d = releaseTimes[i] - time;
            if (d > duration || d == duration && keysPressed[i] > key) {
                key = keysPressed[i];
                duration = d;
            }
            time = releaseTimes[i];
        }
        return key;
    }
};
