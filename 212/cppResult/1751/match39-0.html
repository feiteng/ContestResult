class Solution {
public:
    char slowestKey(vector<int>& rt, string kp) {
        char c = kp.front();
        int time = rt.front();
        for (int i = 1; i < rt.size(); ++i) {
            int t = rt[i] - rt[i-1];
            if (t > time) {
                time = t;
                c = kp[i];
            } else if (t == time && kp[i] > c) {
                c = kp[i];
            }
        }
        
        return c;
    }
};
