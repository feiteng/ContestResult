class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        int len = releaseTimes.size();
        char res = keysPressed[0];
        int restime = releaseTimes[0];
        for (int i = 1; i < len; i++) {
            if (releaseTimes[i] - releaseTimes[i - 1] > restime) {
                res = keysPressed[i];
                restime = releaseTimes[i] - releaseTimes[i - 1];
            } else if (releaseTimes[i] - releaseTimes[i - 1] == restime && keysPressed[i] > res) {
                res = keysPressed[i];
            }
        }
        return res;
    }
};
