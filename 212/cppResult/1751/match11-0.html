class Solution {
public:
    char slowestKey(vector<int>& v, string keysPressed) {
        int maxi = v[0];
        char ch = keysPressed[0];
        
        for (int i = 1; i < v.size(); i++) {
            int dif =  v[i] - v[i-1];
            if ( dif > maxi) {
                maxi = dif;
                ch = keysPressed[i];
            }else if (dif == maxi) {
                ch = max(ch , keysPressed[i]);
            }
        }
        
        return ch;
    }
};
