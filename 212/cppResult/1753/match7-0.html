const int d4[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        int low = 0, high = 1e6;
        vector<bool> flag(n * m);
        while (low != high) {
            int mid = (low + high) / 2;
            fill(flag.begin(), flag.end(), false);
            flag[0] = true;
            queue<int> Q;
            Q.push(0);
            while (!Q.empty()) {
                int pos = Q.front();
                Q.pop();
                int x = pos / m, y = pos % m;
                for (int k = 0; k < 4; ++k) {
                    int nx = x + d4[k][0];
                    int ny = y + d4[k][1];
                    if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                    if (abs(a[nx][ny] - a[x][y]) > mid) continue;
                    int npos = nx * m + ny;
                    if (flag[npos]) continue;
                    flag[npos] = true;
                    Q.push(npos);
                }
            }
            if (!flag[n * m - 1]) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return high;
    }
};
