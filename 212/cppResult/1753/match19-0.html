class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int n = heights.size();
        int m = heights[0].size();
        vector<vector<bool>> visit(n,vector<bool>(m,false));
        priority_queue<pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>> q;
        q.push({0,0});
        vector<int> dirs = {0,1,0,-1,0};
        while(!q.empty()){
            auto cur = q.top();
            q.pop();
            int i = cur.second/m;
            int j = cur.second%m;
            if(visit[i][j])continue;
            //cout<<i<<j<<endl;
            if(i==n-1&&j==m-1) return cur.first;
            visit[i][j] = true;
            for(int k = 0; k < 4; k++){
                int ni = i+ dirs[k];
                int nj = j+dirs[k+1];
                //cout<<ni<<nj<<endl;
                if(ni<0||ni>=n||nj<0||nj>=m)continue;
                if(visit[ni][nj])continue;
                q.push({max(cur.first,abs(heights[ni][nj]-heights[i][j])),ni*m+nj});
            }
        }
        return -1;
    }
};
