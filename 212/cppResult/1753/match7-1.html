class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& he) {
        int m=he.size(),n=he[0].size();
        int l=-1,r=1e6;
        while(l+1<r){
            int mid=(l+r)/2;
            vector<vector<bool>> vis(m,vector<bool>(n,false));
            vis[0][0]=true;
            queue<pair<int,int>> qq;
            qq.push({0,0});

            int dd[][2]={{0,1},{0,-1},{1,0},{-1,0}};
            while(!qq.empty()){
                int x,y;

                tie(x,y)=qq.front();qq.pop();
                for(int i=0;i<4;i++){
                    int xx=x+dd[i][0];
                    int yy=y+dd[i][1];
                    if(xx<0||xx>=m||yy<0||yy>=n) continue;
                    if(abs(he[x][y]-he[xx][yy])>mid) continue;
                    if(vis[xx][yy]) continue;
                    vis[xx][yy]=true;
                    qq.push({xx,yy});
                }
            }
            if(vis[m-1][n-1]) r=mid;
            else l=mid;
        }
        return r;
    }
};
