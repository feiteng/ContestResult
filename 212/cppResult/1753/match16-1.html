#define pii pair<int,int>

int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};


class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<int>> dist(n, vector<int>(m, 1e6));
        vector<vector<bool>> vis(n, vector<bool>(m));
        priority_queue<pair<int,pii>, vector<pair<int,pii>>, greater<pair<int,pii>>> pq;
        pq.push({0, {0, 0}});
        dist[0][0] = 0;
        while (!pq.empty()) {
            int cx = pq.top().second.first;
            int cy = pq.top().second.second;
            pq.pop();
            if (vis[cx][cy]) continue;
            vis[cx][cy] = 1;
            if (cx == n-1 && cy == m-1) {
                return dist[cx][cy];
            }

            for (int d=0; d<4; d++) {
                int x = cx + dir[d][0];
                int y = cy + dir[d][1];
                if (x<0 || x>=n || y<0 || y>=m) continue;
                int nxt = max(abs(a[x][y]-a[cx][cy]), dist[cx][cy]);
                if (nxt < dist[x][y]) {
                    pq.push({nxt, {x, y}});
                    dist[x][y] = nxt;
                }
            }
        }
        return 1e6;
    }
};
