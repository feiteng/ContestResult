class Solution {
public:
    vector<vector<int>> dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    int minimumEffortPath(vector<vector<int>>& heights) {
        priority_queue<tuple<int, int, int>> pq; // dist, y, x
        vector<vector<int>> dists(heights.size(), vector<int>(heights[0].size(), 1234567));
        pq.push({0, 0, 0});
        dists[0][0] = 0;
        while (!pq.empty()) {
            auto [curD, curY, curX] = pq.top(); pq.pop();
            curD = -curD;
            if (dists[curY][curX] < curD)
                continue;
            for (int i=0; i<4; i++) {
                int nextY = curY+dirs[i][0], nextX = curX+dirs[i][1];
                int dFromCur = distance(heights, curY, curX, nextY, nextX);
                if (dFromCur == -1)
                    continue;
                int nextD = max(curD, dFromCur);
                if (dists[nextY][nextX] <= nextD)
                    continue;
                dists[nextY][nextX] = nextD;
                pq.push({-nextD, nextY, nextX});
            }
        }
        return dists[heights.size()-1][heights[0].size()-1];
    }
    int distance(vector<vector<int>>& heights, int y1, int x1, int y2, int x2) {
        if (y2<0 || heights.size() <= y2 || x2<0 || heights[0].size() <=x2)
            return -1;
        int d = heights[y2][x2] - heights[y1][x1];
        return d<0 ? -d : d;
    }
};
