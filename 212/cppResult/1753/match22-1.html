struct box
{
    int i;
    int j;
    int e;
    
    bool operator>(const box& other) const
    {
        return e > other.e;
    }
};

class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& H) {
        int m = H.size();
        int n = H[0].size();
        
        vector<vector<int>> E(m, vector<int>(n,1e9));
        E[0][0] = 0;
        
        priority_queue<box, vector<box>, std::greater<box>> pq;
        pq.push({0,0,0});
        
        while(!pq.empty())
        {
            int i = pq.top().i;
            int j = pq.top().j;
            int e = pq.top().e;
            pq.pop();
            
            if(i == m-1 && j == n-1)
            {
                return e;
            }
            

            pair<int,int> dir[4] = {{i+1,j}, {i-1,j}, {i,j+1}, {i,j-1}};

            int lev = H[i][j];
            for(int k = 0; k < 4; k++)
            {
                int a = dir[k].first;
                int b = dir[k].second;
                if(0 <= a && a < m && 0 <= b && b < n)
                {
                    int ne = max(e, abs(H[a][b]-lev));
                    if(ne < E[a][b])
                    {
                        E[a][b] = ne;
                        pq.push({a,b,ne});
                    }
                }
            }
        }
        
        return -1;
    }
};
