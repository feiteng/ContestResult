class Solution {
private:
    vector<vector<int>> directions = {{1,0},{0,1},{-1,0},{0,-1}};
    bool hasPath(vector<vector<int>>& heights, int i, int j,  int d, vector<vector<bool>>& visited) {
        if (i==heights.size()-1 && j==heights[0].size()-1)  return true;
        visited[i][j] = true;
        
        for (auto& di: directions) {
            int i2 = i + di[0], j2 = j + di[1];
            if (i2<0 || i2>=heights.size() || j2<0 || j2>=heights[0].size() || visited[i2][j2] || abs(heights[i2][j2]-heights[i][j])>d) continue;
            if (hasPath(heights, i2, j2, d, visited)) return true;
        }
        return false;
    }
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int l=0, r=pow(10, 6);
        
        while (l<r) {
            int mid = (l+r)/2;
            vector<vector<bool>> visited(heights.size(), vector<bool>(heights[0].size(), false));
            if (hasPath(heights, 0, 0, mid, visited)) {
                r = mid;
            } else {
                l = mid+1;
            }
        }
        return l;
    }
};
