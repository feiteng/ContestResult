template<class T> using MinHeap = priority_queue<T, vector<T>, greater<T>>;

const int dx[]={1,0,-1,0};
const int dy[]={0,1,0,-1};
class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& h) {
        int n=h.size(), m=h[0].size();
        auto f=[&]( int i, int j ) { return i*m+j; };
        MinHeap<pair<int,int>> q;
        q.emplace(0,0);
        vector<int> d(n*m, 1e9);
        d[0]=0;
        while ( q.size() ) {
            auto [w,u]=q.top();
            q.pop();
            if ( w>d[u] ) continue;

            int a=u/m, b=u%m;
            for ( int i=0; i<4; i++ ) {
                int x=a+dx[i],y=b+dy[i];
                if ( x>=0&&x<n&&y>=0&&y<m&&max(w,abs(h[a][b]-h[x][y]))<d[f(x,y)] ) {
                    d[f(x,y)]=max(w,abs(h[a][b]-h[x][y]));
                    q.emplace(d[f(x,y)],f(x,y));
                }
            }
        }
        return d[f(n-1,m-1)];
    }
};
