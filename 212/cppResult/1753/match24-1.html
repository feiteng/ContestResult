class Solution {
public:
    const int dx[4] = {1, -1, 0, 0}, dy[4] = {0, 0, 1, -1};
    int n, m;
    vector<vector<int>> a;
    vector<vector<bool>> vis;
    bool dfs(int i, int j, int x) {

        vis[i][j] = 1;
        if (i == n - 1 && j == m - 1) return 1;
        for (int k = 0; k < 4; ++k) {
            int ni = i + dx[k], nj = j + dy[k];
            if (0 <= ni && ni < n && 0 <= nj && nj < m && !vis[ni][nj] && abs(a[i][j] - a[ni][nj]) <= x) {
                if (dfs(ni, nj, x)) return 1;
            }
        }

        return 0;
    }
    int minimumEffortPath(vector<vector<int>>& h) {
        a = h;
        n = a.size(), m = a[0].size();
        int l = 0, r = 1000000;
        while(l < r) {
            int mid = (l + r) / 2;
            vis.assign(n, vector<bool>(m));
            if (dfs(0, 0, mid)) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
