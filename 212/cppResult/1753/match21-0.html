class Solution {
public:
    int dir[4][2] = {{1,0},{0,1},{-1,0},{0,-1}};
    int minimumEffortPath(vector<vector<int>>& heights) {
        auto cmp = [](vector<int>&a, vector<int>&b){
            return a[2]>b[2];  
        };
        priority_queue<vector<int>, vector<vector<int>>, decltype(cmp)> q(cmp);
        q.push({0,0,0});
        int dp[200][200] = {false};
        memset(dp,0x3f,sizeof(dp));
        dp[0][0] = 0;
        while(q.size()>0){
            auto cur = q.top(); q.pop();
            if ( cur[0]+1==heights.size() && cur[1]+1 ==heights[0].size() ){
                return cur[2];
            }
            for(int k=0; k<4; ++k){
                int y = dir[k][0] + cur[0];
                int x = dir[k][1] + cur[1];
                if ( x>=0 && y>=0 && x<heights[0].size() && y<heights.size()  ){
                    int c = max( cur[2], abs(heights[y][x] - heights[cur[0]][cur[1]]) );
                    if ( q.size()==0 || dp[y][x]>c ){
                        dp[y][x] = c;
                        q.push({y,x,c});
                    }
                    
                }
            }
        }
        return -1;
    }
};
