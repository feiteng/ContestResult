class Solution {
private:
    int m, n;
    int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

    
    
public:
    int minimumEffortPath(vector<vector<int>>& h) {
        m = h.size();
        n = h[0].size();
        vector<vector<int>> dp(m, vector<int>(n, -1));     
        
        
        dp[0][0] = 0;
        
        queue<pair<int, int>> q;
        q.push({0, 0});
        while (!q.empty()) {
            auto c = q.front();
            q.pop();
            int i = c.first, j = c.second;
            for (int k = 0; k < 4; k++) {
                int x = i + dir[k][0];
                int y = j + dir[k][1];
                if (x>=0 && x<m && y>=0 && y<n) {
                    int next = max(dp[i][j], abs(h[x][y] - h[i][j]));
                    if (dp[x][y] == -1 || dp[x][y] > next) {
                        dp[x][y] = next;
                        q.push({x, y});
                    }
                }
            }
        }
        
        
        
        return dp[m-1][n-1];
    }
};
