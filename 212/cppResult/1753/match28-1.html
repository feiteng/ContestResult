class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights[0].size();
        vector<vector<int>> dp(m, vector<int>(n, INT_MAX));
        queue<pair<int,int>> vst;
        vst.push(make_pair(0, 0));
        dp[0][0] = 0;
        while(!vst.empty()){

            auto tmp = vst.front();
            vst.pop();
            if(tmp.first > 0){
                int cur = max(dp[tmp.first][tmp.second], abs(heights[tmp.first][tmp.second] - heights[tmp.first - 1][tmp.second]));
                if(cur < dp[tmp.first - 1][tmp.second]){
                    dp[tmp.first - 1][tmp.second] = cur;
                    vst.push(make_pair(tmp.first - 1, tmp.second));
                }
            }
            if(tmp.first < m - 1){
                int cur = max(dp[tmp.first][tmp.second], abs(heights[tmp.first][tmp.second] - heights[tmp.first + 1][tmp.second]));
                if(cur < dp[tmp.first + 1][tmp.second]){
                    dp[tmp.first + 1][tmp.second] = cur;
                    vst.push(make_pair(tmp.first + 1, tmp.second));
                }
            }
            if(tmp.second > 0){
                int cur = max(dp[tmp.first][tmp.second], abs(heights[tmp.first][tmp.second] - heights[tmp.first][tmp.second - 1]));
                if(cur < dp[tmp.first][tmp.second - 1]){
                    dp[tmp.first][tmp.second - 1] = cur;
                    vst.push(make_pair(tmp.first, tmp.second - 1));
                }
            }
            if(tmp.second < n - 1){
                int cur = max(dp[tmp.first][tmp.second], abs(heights[tmp.first][tmp.second] - heights[tmp.first][tmp.second + 1 ]));
                if(cur < dp[tmp.first][tmp.second + 1]){
                    dp[tmp.first][tmp.second + 1] = cur;
                    vst.push(make_pair(tmp.first, tmp.second + 1));
                }
            }
        }
        return dp[m - 1][n - 1];
    }
};
