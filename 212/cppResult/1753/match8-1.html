int dir[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<int>>f=vector<vector<int>>(n, vector<int>(m, -1));
        f[0][0] = 0;
        queue<int>qx,qy;
        vector<vector<bool>>v=vector<vector<bool>>(n, vector<bool>(m, false));
        qx.push(0);
        qy.push(0);
        v[0][0] = 1;

        while(!qx.empty()){
            int x = qx.front();
            qx.pop();
            int y = qy.front();
            qy.pop();
            v[x][y] = 0;
            for(int i=0;i<4;i++){
                int nx = x +dir[i][0];
                int ny = y + dir[i][1];
                if(nx<0||nx>=n||ny<0||ny>=m)continue;
                if(f[nx][ny]==-1||f[nx][ny]>max(f[x][y], abs(a[x][y]-a[nx][ny]))){
                    f[nx][ny]=max(f[x][y], abs(a[x][y]-a[nx][ny]));
                    if(!v[nx][ny]){
                        v[nx][ny]=1;
                        qx.push(nx);
                        qy.push(ny);
                    }
                }
            }
        }
        return f[n-1][m-1];
    }
};
