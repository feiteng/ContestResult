class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int n = heights.size();
        int m = heights[0].size();
        vector<vector<int>> minEffort(n, vector<int>(m, INT_MAX));
        
        minEffort[0][0] = 0;
        queue<pair<int,int>> q;
        q.push({0, 0});
        
        int dr[4] = {0, 1, 0, -1};
        int dc[4] = {1, 0, -1, 0};
        
        while (!q.empty()) {
            auto pos = q.front();
            q.pop();
            
            for (int i = 0; i < 4; i++) {
                int r = pos.first + dr[i];
                int c = pos.second + dc[i];
                if (r >= 0 && r < n && c >= 0 && c < m) {
                    int eff = abs(heights[r][c] - heights[pos.first][pos.second]);
                    int k = max(eff, minEffort[pos.first][pos.second]);
                    if (k < minEffort[r][c]) {
                        minEffort[r][c] = k;
                        q.push({r, c});
                    }
                }
            }
        }
        
        // for (int i = 0; i < n; i++) {
        //     for (int j = 0; j < m; j++) {
        //         cout << minEffort[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        // cout << endl;
        
        return minEffort[n-1][m-1];
    }
};
