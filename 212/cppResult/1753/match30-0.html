class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m=heights.size();
        int n=heights[0].size();
        vector<vector<int>> dp(m,vector<int>(n,-1));
        dp[0][0]=0;
        queue<pair<int,int>> m_queue;
        m_queue.push(make_pair(0,0));
        vector<pair<int,int>> paths={{1,0},{0,1},{-1,0},{0,-1}};
        while(!m_queue.empty()){
            auto node = m_queue.front();
            m_queue.pop();
            for(int i=0;i<4;i++){
                int x=node.first+paths[i].first;
                int y=node.second+paths[i].second;
                if(x<0||y<0||x>=m||y>=n){
                    continue;
                }
                int diff = abs(heights[x][y]-heights[node.first][node.second]);
                int val = max(diff,dp[node.first][node.second]);
                if(dp[x][y]==-1 || dp[x][y]>val){
                    dp[x][y]=val;
                    m_queue.push(make_pair(x,y));
                }
            }
        }
        return dp[m-1][n-1];
        
    }
};
