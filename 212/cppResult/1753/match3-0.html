class Solution {
public:
    int n, m;
    int dx[4] = {0,0,-1,1}, dy[4]={-1,1,0,0};
    bool check(vector<vector<int>>& h, int maxd) {
        queue<pair<int, int>> q;
        vector<vector<int>> dis(h);
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                dis[i][j] = -1;
        dis[0][0] = 0;
        q.push(make_pair(0, 0));
        while (!q.empty())  {
            auto p = q.front();
            q.pop();
            int x = p.first, y = p.second;
            if (x == m-1 && y == n-1)
                return true;
            for (int k = 0 ;k  < 4; k++){
                int xt = x + dx[k], yt = y + dy[k];
                if (xt >= 0 && xt < m && yt >= 0 && yt < n && abs(h[xt][yt] - h[x][y]) <= maxd && dis[xt][yt]==-1) {
                    q.push(make_pair(xt, yt));
                    dis[xt][yt] = dis[x][y]+1;
                }
            }
        }
        return false;
    }
    
    int minimumEffortPath(vector<vector<int>>& heights) {
        m = heights.size(), n = heights[0].size();
        int lo = 0, hi = 1e6;
        while (lo < hi) {
            int mi = (lo + hi) / 2;
            if (check(heights, mi))
                hi = mi;
            else
                lo = mi + 1;
        }
        return lo;
    }
};
