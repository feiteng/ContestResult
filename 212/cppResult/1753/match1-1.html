class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int row = heights.size(), col = heights[0].size();
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>pq;
        pq.push({0, 0});
        vector<vector<int>>effort(row, vector<int>(col, INT_MAX));
        effort[0][0] = 0;
        vector<vector<int>>dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

        
        while(!pq.empty()) {
            auto f = pq.top();
            pq.pop();
            int ef = f.first, r = f.second/col, c = f.second%col;
        //    cout << "sdad   " << r << ", " << c << ", " << ef << endl;
            if (r==row-1&&c==col-1)
                return ef;
            
            for (auto dir: dirs) {

                int x = dir[0]+r, y = dir[1]+c;
                if (x>=0&&x<row&&y>=0&&y<col&&effort[x][y]>max(ef, abs(heights[r][c]-heights[x][y]))) {
                    effort[x][y] = max(ef, abs(heights[r][c]-heights[x][y]));
                 //   cout <<x << ", " << y << ", " << effort[x][y] << endl;
                    pq.push({effort[x][y], x*col+y});
                }
            }
            
        }
        return -1;
    }
};
