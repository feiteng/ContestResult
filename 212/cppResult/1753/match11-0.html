class Solution {
public:
    vector<vector<int>> dirs = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
    
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights.front().size();
        vector<vector<int>> distances(m, vector<int>(n, INT_MAX));
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> mqueue;
        
        mqueue.push({0, 0, 0});
        distances[0][0] = 0;
        
        while (!mqueue.empty()) {
            vector<int> curr = mqueue.top();
            mqueue.pop();
            if (curr[1] == m-1 && curr[2] == n-1)
                return curr[0];
            for (auto &dir: dirs) {
                int x = curr[1] + dir[0], y = curr[2] + dir[1];
                if (x >= 0 && x < m && y >= 0 && y < n) {
                    int dist = max(curr[0], abs(heights[x][y] - heights[curr[1]][curr[2]]));
                    if (dist < distances[x][y]) {
                        distances[x][y] = dist;
                        mqueue.push({dist, x, y});
                    }
                }
            }
        }
        return -1;
    }
};
