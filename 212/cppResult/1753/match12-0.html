class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) const {
        return foo(heights);
    }
    
    int foo(vector<vector<int>> &arr) const {
        const int n = arr.size();
        const int m = arr[0].size();
        vector<vector<int>> dist(n, vector<int>(m, INT_MAX));
        queue<pair<int, int>> q;
        q.push({0, 0});
        dist[0][0] = 0;
        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            
            if (x - 1 >= 0) {
                int w = max(dist[x][y], abs(arr[x - 1][y] - arr[x][y]));
                if (w < dist[x - 1][y]) {
                    dist[x - 1][y] = w;
                    q.push({x - 1, y});
                }
            }
            
            if (x + 1 < n) {
                int w = max(dist[x][y], abs(arr[x + 1][y] - arr[x][y]));
                if (w < dist[x + 1][y]) {
                    dist[x + 1][y] = w;
                    q.push({x + 1, y});
                }
            }
            
            if (y - 1 >= 0) {
                int w = max(dist[x][y], abs(arr[x][y - 1] - arr[x][y]));
                if (w < dist[x][y - 1]) {
                    dist[x][y - 1] = w;
                    q.push({x, y - 1});
                }
            }
            
            if (y + 1 < m) {
                int w = max(dist[x][y], abs(arr[x][y + 1] - arr[x][y]));
                if (w < dist[x][y + 1]) {
                    dist[x][y + 1] = w;
                    q.push({x, y + 1});
                }
            }
        }
        
        return dist[n - 1][m - 1];
    }
};
