typedef pair<int, int> pii;
typedef pair<int, pii> piii;
const int MAXN = 123;
int dp[MAXN][MAXN];
int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, -1, 1};
class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        memset(dp, -1, sizeof(dp));
        int n = heights.size();
        int m = heights[0].size();
        dp[0][0] = 0;
        priority_queue<piii, vector<piii>, greater<piii>> pq;
        pq.emplace(0, make_pair(0, 0));
        while (!pq.empty()) {
            auto item = pq.top(); pq.pop();
            int val = item.first;
            int x = item.second.first;
            int y = item.second.second;
            if (dp[x][y] != -1 && dp[x][y] < val) continue;
            for (int k = 0; k < 4; k++) {
                int xx = x + dx[k];
                int yy = y + dy[k];
                if (xx < 0 || xx >= n || yy < 0 || yy >= m) continue;
                int tmp = max(abs(heights[xx][yy] - heights[x][y]), val);
                if (dp[xx][yy] == -1 || dp[xx][yy] > tmp) {
                    dp[xx][yy] = tmp;
                    pq.emplace(tmp, make_pair(xx, yy));
                }
            }
        }
        return dp[n-1][m-1];
        
    }
};
