int dx[] = {-1, 0, 1, 0, -1, 1, 1, -1};
int dy[] = {0, 1, 0, -1, 1, 1, -1, -1};

class Solution {
public:
    void dfs(int x, int y, int n, int m, vector<vector<int>>& a, vector<vector<bool>>& vis, int d) {
        vis[x][y] = 1;
        
        for (int k = 0; k < 4; k++) {
            int xx = x + dx[k];
            int yy = y + dy[k];
            if (0 <= xx and xx < n and 0 <= yy and yy < m and !vis[xx][yy] and abs(a[x][y] - a[xx][yy]) <= d) {
                dfs(xx, yy, n, m, a, vis, d);
            }
        }
    }
    
    bool can(vector<vector<int>>& a, int d) {
        int n = a.size(), m = a[0].size();
        vector<vector<bool>> vis(n, vector<bool>(m));
        dfs(0, 0, n, m, a, vis, d);
        return vis[n-1][m-1];
    }
    
    int minimumEffortPath(vector<vector<int>>& a) {
        int l = -1, r = 1e6;
        while (l + 1 < r) {
            // cout << "l = " << l << ", r = " << r << "\n";
            int m = (l + r) / 2;
            if (can(a, m)) {
                r = m;
            } else {
                l = m;
            }
        }
        return r;
    }
};
