class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        const int MAXI = 1e7;
        int m = heights.size(), n = heights[0].size();
        vector<vector<int>> dist(m, vector<int>(n, MAXI));
        vector<vector<int>> visited(m, vector<int>(n, false));
        dist[0][0] = 0;
        
        int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        pq.push({0, 0});
        
        while (!pq.empty()) {
            auto [cost, cur_state] = pq.top();
            pq.pop();
            
            int cur_i = cur_state >> 8;
            int cur_j = cur_state & 0xFF;
            
            if (cur_i == m - 1 && cur_j == n - 1) break;
            if (visited[cur_i][cur_j]) continue;
            visited[cur_i][cur_j] = true;
            
            for (const auto& dir: dirs) {
                int new_i = cur_i + dir[0];
                int new_j = cur_j + dir[1];
                if (new_i >= 0 && new_i < m && new_j >= 0 && new_j < n) {
                    if (max(cost, abs(heights[new_i][new_j] - heights[cur_i][cur_j])) < dist[new_i][new_j]) {
                        dist[new_i][new_j] = max(cost, abs(heights[new_i][new_j] - heights[cur_i][cur_j]));
                        pq.push({dist[new_i][new_j], (new_i << 8) | new_j});
                    }
                }
            }
            
        }
        return dist[m-1][n-1];
    }
};
