class Solution {
public:

    const int dir[4][2] = {{1,0}, {0,1}, {-1,0}, {0, -1}};
    
    int minimumEffortPath(vector<vector<int>>& heights) {
        queue<int> row;
        queue<int> col;
        int val[110][110];
        bool check[110][110];
        
        memset(check, true, sizeof(check));
        memset(val, 0xf, sizeof(val));
        val[0][0] = 0;
        check[0][0] = false;
        row.push(0);
        col.push(0);
        
        while (!row.empty()) {
            int x = row.front(), y = col.front();
            row.pop();
            col.pop();

            check[x][y] = true;
            
            for (int i = 0; i < 4; i++) {
                int a = x+dir[i][0];
                int b = y+dir[i][1];
                
                if (0 <= a && a < heights.size() && 0 <= b && b < heights[0].size()) {
                    if (val[a][b] > max(val[x][y], abs(heights[a][b] - heights[x][y]))) {
                        val[a][b] = max(val[x][y], abs(heights[a][b] - heights[x][y]));
                        
                        if (check[a][b]) {
                            check[a][b] = false;
                            row.push(a);
                            col.push(b);
                        }
                    }
                }
            }
        }
        
        return val[heights.size() -1][heights[0].size() -1];
    }
};
