class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> q;
        q.push({0,0});//md->p
        unordered_map<int,int> um;
        um[0]=0;//p->md
        int m=heights.size(), n=heights[0].size();
        vector<pair<int,int>> vp{{-1,0},{1,0},{0,-1},{0,1}};
        while (!q.empty()) {
            auto p=q.top();
            int md=p.first, x=p.second/1000, y=p.second%1000;
            q.pop();
            for (auto [dx,dy]: vp) {
                // for (int dy: {-1,1}) {
                    int nx=x+dx,ny=dy+y;
                    if (nx<0||nx>=m||ny<0||ny>=n) continue;
                    int nd=max(md,abs(heights[x][y]-heights[nx][ny]));
                    int p=nx*1000+ny;
                    if (um.count(p)==0||um[p]>nd) {
                        um[p]=nd;
                        q.push({nd,p});
                        // cout<<nd<<" "<<p<<endl;
                    }
                // }
            }
        }
        // cout<<endl;
        return um[(m-1)*1000+n-1];
    }
};
