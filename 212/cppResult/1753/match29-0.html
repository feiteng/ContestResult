class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
		int M = heights.size();
		int N = heights[0].size();
        vector<vector<int> > Map(M, vector<int>(N, 1000000));
		
		int dx[] = {0, 1, 0, -1};
		int dy[] = {1, 0, -1, 0};
		
		queue<int> Q;
		Q.push(0);
		Map[0][0] = 0;
		
		while(Q.size())
		{
			int i = Q.front() >> 16;
			int j = (Q.front() & 0xFFFF);
			Q.pop();
			
			for(int l = 0; l < 4; l++)
			{
				if(((i + dx[l]) >= 0) && ((i + dx[l]) < M) && ((j + dy[l]) >= 0) && ((j + dy[l]) < N))
				{
					int T = max(Map[i][j], abs(heights[i][j] - heights[(i + dx[l])][(j + dy[l])]));
					
					if(T < Map[(i + dx[l])][(j + dy[l])])
					{
						Map[(i + dx[l])][(j + dy[l])] = T;
						Q.push(((i + dx[l]) << 16) + (j + dy[l]));
					}
				}
			}
		}
	
		return Map[(M - 1)][(N - 1)];
    }
};
