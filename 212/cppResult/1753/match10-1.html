class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int di[] = {0, 1, 0, -1}, dj[] = {1, 0, -1, 0};

        int n = heights.size(), m = heights[0].size();
        priority_queue<array<int, 3>, vector<array<int, 3>>, greater<array<int, 3>>> pq; // current cost, i, j
        vector<vector<int>> res(n, vector<int>(m, 1e8));
        pq.push({0, 0, 0});
        res[0][0] = 0;
        
        while (!pq.empty()) {
            auto [cost, i, j] = pq.top();
            pq.pop();
            

            for (int k = 0; k < 4; ++k) {
                int ni = i + di[k], nj = j + dj[k];
                if (ni >= 0 && nj >= 0 && ni < n && nj < m) {
                    int ncost = max(abs(heights[ni][nj] - heights[i][j]), cost);
                    if (ncost < res[ni][nj]) {
                        res[ni][nj] = ncost;
                        pq.push({ncost, ni, nj});
                    }
                }
            }
        }
        
        return res[n-1][m-1];
        
    }
};
