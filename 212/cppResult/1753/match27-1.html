class Solution {
public:
    vector<vector<int>> g;
    int n;
    int m;
    vector<vector<int>> dp;

    vector<pair<int,int>> moves = {{-1,0},{1,0},{0,-1},{0,1}};
    
    void bfs() {
        dp[0][0] = 0;
        

        queue<pair<int,int>> q;
        q.push({0,0});
        
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            int x = cur.first;
            int y = cur.second;
            // cout << "\n";
            // cout << "(x,y) = " << x << ", " << y << "\n";

            for (auto move: moves) {
                int dx = x + move.first;
                int dy = y + move.second;
                // cout << "(dx,dy) = " << dx << ", " << dy << "\n";
                if (dx < 0 || dx >= n) continue;
                if (dy < 0 || dy >= m) continue;
                int val = max(dp[x][y], abs(g[x][y]-g[dx][dy]));
                if (val < dp[dx][dy]) {
                    dp[dx][dy] = val;
                    q.push({dx,dy});
                }
            }
        }
        
    }
    
    int minimumEffortPath(vector<vector<int>>& heights) {
        g = heights;
        n = (int)g.size();
        m = (int)g[0].size();
        
        dp.assign(n, vector<int>(m,1e9));
        
        bfs();
        
        return dp[n-1][m-1];
        
    }
};
