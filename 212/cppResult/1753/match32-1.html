class Solution {
public:
    int dx[4]={0,0,1,-1};
    int dy[4]={1,-1,0,0};
    int minimumEffortPath(vector<vector<int>>& h) {
        int n=h.size(),m=h[0].size();
        vector<vector<int>>dp(n,vector<int>(m,1e9));
        dp[0][0]=0;
        int flag=0;
        do{
            flag=0;

            for(int i=0;i<n;i++){
                for(int j=0;j<m;j++){
                    for(int k=0;k<4;k++){
                        int ni=i+dx[k],nj=j+dy[k];
                        if(ni>=0&&ni<n&&nj>=0&&nj<m){
                            int t=max(abs(h[ni][nj]-h[i][j]),dp[i][j]);
                            if(dp[ni][nj]>t){
                                flag=1;
                                dp[ni][nj]=t;
                            }
                        }
                    }
                }
            }
        }while(flag);
        return dp[n-1][m-1];
    }
};
