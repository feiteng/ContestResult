bool vis[100][100];
const int dx[] = {-1, 1, 0, 0};
const int dy[] = {0, 0, -1, 1};
class Solution {
    
public:
    bool ok(vector<vector<int>>& h, int d)
    {
        int n = h.size(), m = h[0].size();
        memset(vis, 0, sizeof vis);
        queue<pair<int, int>> q;
        vis[0][0] = 1;
        q.emplace(0, 0);
        while (q.size()) {
            int i, j;
            tie(i, j) = q.front();
            q.pop();
            for (int k = 0; k < 4; ++k) {
                int a = i + dx[k], b = j + dy[k];
                if (min(a, b) >= 0 && a < n && b < m && !vis[a][b] && abs(h[a][b] - h[i][j]) <= d) {
                    vis[a][b] = 1;
                    q.emplace(a, b);
                }
            }
        }
        return vis[n - 1][m - 1];
    }
    int minimumEffortPath(vector<vector<int>>& heights) {
        int lo = 0, hi = 1e6;
        while (lo <= hi) {
            int mi = (lo + hi) / 2;
            if (ok(heights, mi))
                hi = mi - 1;
            else
                lo = mi + 1;
        }
        return lo;
    }
};
