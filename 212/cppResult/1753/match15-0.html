class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& h) {
        int m = h.size();
        int n = h[0].size();
        vector<vector<int>> cost(m, vector<int>(n, INT_MAX /2));

        cost[0][0] = 0;
        deque<pair<int,int>> deq;
        deq.emplace_back(0,0);
        int dir[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
        while (!deq.empty()){
            auto p = deq.front();
            int x = p.first ;
            int y = p.second ;
            //printf("search %d %d\n", x, y);
            deq.pop_front();
            for (int k = 0; k < 4; k++){
                int xx = x + dir[k][0];
                int yy = y + dir[k][1];
                if (xx < 0 || yy < 0 || xx >= m || yy >= n)
                    continue;
                int d = max(cost[x][y], abs(h[x][y] - h[xx][yy]));
                if (d < cost[xx][yy]){
                    cost[xx][yy] = d;
                    deq.emplace_back(xx, yy);
                }
            }
            
        }
        return cost[m-1][n-1];
    }
};
