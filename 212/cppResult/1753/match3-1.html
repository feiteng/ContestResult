int f[111][111];
typedef pair<int, int> PII;
int dx[4] = {0, 0, -1, 1};
int dy[4] = {-1, 1, 0, 0};

class Solution {
public:
    int n, m;
    int check(int D, const vector<vector<int>>& h) {

        for (int i = 0; i < n; i++) for (int j = 0; j < m; j++) f[i][j] = 0;
        f[0][0] = 1;
        queue<PII> Q;
        Q.push(PII(0, 0));
        while (!Q.empty()) {
            auto nico = Q.front();
            Q.pop();
            int x = nico.first, y = nico.second;

            for (int k = 0; k < 4; k++) {
                int p = x + dx[k], q = y + dy[k];
                if (p >= 0 && q >= 0 && p < n && q < m && f[p][q] == 0 && abs(h[p][q] - h[x][y]) <= D) {
                    f[p][q] = 1;
                    Q.push(PII(p, q));
                }
            }
        }
        return f[n - 1][m - 1];
    }
    int minimumEffortPath(vector<vector<int>>& h) {
        n = h.size(), m = h[0].size();
        int L = -1, R = 1000000;
        while (L + 1 < R) {
            int M = (L + R) / 2;
            if (check(M, h)) R = M;
            else L = M;
        }
        return R;
    }
};
