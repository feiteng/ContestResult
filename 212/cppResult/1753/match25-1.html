class Solution {
public:
    

    const vector<vector<int>> WALKS = {{0,1},{0,-1},{-1,0},{1,0}};
    
    bool good(vector<vector<int>>& a, int max_diff) {
        int n = a.size(), m = a[0].size();
        
        vector<vector<bool>> visited(n, vector<bool>(m, false));
        visited[0][0] = true;
        queue<pair<int, int>> q;
        q.emplace(0, 0);
        
        while (!q.empty()) {
            pair<int, int> cur = q.front();
            q.pop();
            
            for (auto const& walk : WALKS) {
                int next_x = cur.first + walk[0], next_y = cur.second +walk[1];
                
                if (next_x < 0 || next_x >= n || next_y < 0 || next_y >= m || visited[next_x][next_y] ||
                   abs(a[next_x][next_y] - a[cur.first][cur.second]) > max_diff) {
                    continue;
                }
                
                visited[next_x][next_y] = true;
                q.emplace(next_x, next_y);
            }
        }
        
        return visited[n - 1][m - 1];
    }
    

    int minimumEffortPath(vector<vector<int>>& a) {
        int l = 0, r = 1e6;
        
        while (l <= r) {
            int mid = (l + r) / 2;
            
            if (good(a, mid)) {
                r = mid - 1;
            }
            else {
                l = mid + 1;
            }
        }
        
        return l;
    }
};
