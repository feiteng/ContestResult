class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        
        int m = heights.size();
        int n = heights[0].size();
        
        vector< vector<int> > dp(m, vector<int>(n, INT_MAX));
        deque<pair<int,int>> Q;
        Q.push_back( {0, 0} );
        dp[0][0] = 0;
        
        vector<int> dr = {-1, 0, 1, 0};
        vector<int> dc = {0, 1, 0, -1};
        
        while (Q.size() > 0)
        {

            pair<int, int> cur = Q.front();
            Q.pop_front();
            int r = cur.first;
            int c = cur.second;
            
            for (int a = 0; a < 4; ++a)
            {
                int nr = r + dr[a];
                int nc = c + dc[a];
                if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;
                int curDiff = abs(heights[nr][nc] - heights[r][c]);
                if (dp[nr][nc] > max(curDiff, dp[r][c]) )
                {
                    dp[nr][nc] = max(curDiff, dp[r][c]);
                    Q.push_back( {nr, nc} );
                }
            }
        }
        
        return dp[m - 1][n - 1];
    }
};
