class Solution {
public:

    vector<vector<int>> dir = {{1, 0}, {-1, 0}, {0, -1}, {0, 1}};
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights[0].size();
        queue<pair<int, int>> q;
        vector<vector<int>> rec(m, vector<int>(n, 0x7fffffff));
        q.push({0, 0});
        rec[0][0] = 0;
        while (!q.empty()) {
            auto e = q.front();
            q.pop();
            int x = e.first;
            int y = e.second;

            for (auto& d : dir) {
                int n_x = x + d[0];
                int n_y = y + d[1];
                
                if (n_x >= 0 && n_x < m && n_y >= 0 && n_y < n) {
                    //cout << n_x << " " << n_y << endl;
                    if (rec[n_x][n_y] > max(rec[x][y], abs(heights[x][y] - heights[n_x][n_y]))) {
                        rec[n_x][n_y] = max(rec[x][y], abs(heights[x][y] - heights[n_x][n_y]));
                        //cout << "push " << n_x << " " << n_y << " " << rec[n_x][n_y] << endl;
                        q.push({n_x, n_y});
                    }
                }
            }
        }
        return rec[m - 1][n - 1];
    }
};
