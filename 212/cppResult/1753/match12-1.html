class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights[0].size();
        vector<vector<int>> dp(m, vector<int>(n, INT_MAX / 2));
        dp[0][0] = 0;
        queue<array<int, 3>> q;
        q.push({0, 0, 0});
        while (!q.empty()) {
            auto curr = q.front();
            q.pop();
            int x = curr[0], y = curr[1], e = curr[2];
            // cout << x << " " << y << " " << e << endl;
            if (dp[x][y] < e) continue;

            if (x - 1 >= 0 && x - 1 < m) {
                int newE = max(abs(heights[x-1][y] - heights[x][y]), dp[x][y]);
                if (newE < dp[x-1][y]) {
                    dp[x-1][y] = newE;
                    q.push({x-1, y, newE});
                }
            }
            if (x + 1 >= 0 && x + 1 < m) {
                int newE = max(abs(heights[x+1][y] - heights[x][y]), dp[x][y]);
                if (newE < dp[x+1][y]) {
                    dp[x+1][y] = newE;
                    q.push({x+1, y, newE});
                }
            }
            if (y - 1 >= 0 && y - 1 < n) {
                int newE = max(abs(heights[x][y-1] - heights[x][y]), dp[x][y]);
                if (newE < dp[x][y-1]) {
                    dp[x][y-1] = newE;
                    q.push({x, y-1, newE});
                }
            }
            if (y + 1 >= 0 && y + 1 < n) {
                int newE = max(abs(heights[x][y+1] - heights[x][y]), dp[x][y]);
                if (newE < dp[x][y+1]) {
                    dp[x][y+1] = newE;
                    q.push({x, y+1, newE});
                }
            }
        }
        return dp.back().back();
    }
};
