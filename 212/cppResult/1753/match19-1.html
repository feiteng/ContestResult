#define ll long long
struct Data {
    int x,y,c;
    Data(int x, int y, int c): x(x),y(y),c(c){}
    bool operator<(const Data& a) const {
    return c > a.c;
    }
};




class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& h) {
        int n = h.size();
        int m = h[0].size();
        vector<vector<int>> v(n,vector<int>(m,1001001001));
        priority_queue<Data> pq;
        pq.push({0,0,0});

        vector<int> dx = {0,0,1,-1};
        vector<int> dy = {1,-1,0,0};
        while(pq.size()){
            auto cur = pq.top();
            pq.pop();
            
            int x = cur.x;
            int y = cur.y;
            int c = cur.c;
            if (x == n-1 && y == m-1){
                return c;
            }
            if (v[cur.x][cur.y] <= c) continue;

            v[cur.x][cur.y] = c;
            
            for(int i = 0; i < 4; i++){
                int nx = x + dx[i];
                int ny = y + dy[i];
                if (nx < 0 || ny < 0 || nx >= n || ny >= m) continue;
                int nc = abs(h[x][y]-h[nx][ny]);
                pq.push({nx,ny,max(nc,c)});
            }
        }
        
        return -1;
        
    }
};
