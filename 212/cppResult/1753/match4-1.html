class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size();
        int n = heights[0].size();
        vector<vector<int>> v(m, vector<int>(n, INT_MAX));
        queue<pair<int,int>> q;
        v[0][0] = 0;
        q.emplace(0, 0);

        int di[4] = {0, 0, -1, 1};
        int dj[4] = {-1, 1, 0, 0};
        while(!q.empty()) {
            auto [thei, thej] = q.front();
            q.pop();
            for (int t = 0; t < 4; t++) {
                int i = thei + di[t];
                int j = thej + dj[t];
                if (i >= 0 && i < m && j >= 0 && j < n) {
                    int newV = max(v[thei][thej], abs(heights[thei][thej] - heights[i][j]));
                    if (newV < v[i][j]) {
                        v[i][j] = newV;
                        // cout << "v " << i << "," << j << ": " << v[i][j] << endl;
                        q.emplace(i, j);
                    }
                    
                }
            }
            
        }
        return v[m - 1][n - 1];
    }
};
