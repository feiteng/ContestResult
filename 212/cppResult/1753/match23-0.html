class Solution {
public:
    int nxt[4][2]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    int d[105][105];
    bool f[105][105];
    int minimumEffortPath(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        memset(d, 1, sizeof(d));
        memset(f, true, sizeof(f));
        queue<pair<int, int>> q;
        q.push(make_pair(0, 0));
        f[0][0] = false;
        d[0][0] = 0;
        while (!q.empty()){
            auto xx = q.front();
            q.pop();
            int x = xx.first;
            int y = xx.second;
            for (int i = 0; i < 4; i++){
                int nx = x + nxt[i][0];
                int ny = y + nxt[i][1];
                if (nx >= 0 && nx < n && ny >= 0 && ny < m){
                    if (d[nx][ny] > max(d[x][y], abs(a[nx][ny] - a[x][y]))){
                        d[nx][ny] = max(d[x][y], abs(a[nx][ny] - a[x][y]));
                        if (f[nx][ny]){
                            q.push(make_pair(nx, ny));
                            f[nx][ny] = false;
                        }
                    }
                }
            }
            f[x][y] = true;
        }
        return d[n - 1][m - 1];
    }
};
