int dx[] = {0, 0, -1, +1};
int dy[] = {+1, -1, 0, 0};
const  int inf = 1e9;
    
class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& h) {
        int n = h.size();
        int m = h[0].size();
        priority_queue < array<int,3> ,vector<array<int,3>> ,greater<array<int,3>> > q;
        if(n == 1 && m == 1)
            return 0;
        vector < vector < int > > dp(n, vector < int >(m, inf));
        dp[0][0] = 0;
        if(n > 1) 
            q.push({abs(h[1][0] - h[0][0]), 1, 0});
        if(m > 1)
            q.push({abs(h[0][1] - h[0][0]), 0, 1});
        while(!q.empty()) {
            auto top = q.top();
            q.pop();
            for(int k = 0; k < 4; k++) {
                int x = dx[k] + top[1];
                int y = dy[k] + top[2];
                if(x < 0 or y < 0 or x >= n or y >= m )
                    continue;
                int cost = max(top[0] , abs(h[x][y] - h[top[1]][top[2]]));
                if(cost < dp[x][y]) {
                    dp[x][y] = cost;
                    q.push({cost, x, y});
                }
            }
        }
        return dp[n - 1][m - 1];
    }
};
