class UF {
public:
    vector<int> fa;
    vector<int> sz;
    int n;
    int comp_cnt;
    
public:
    UF(int _n): n(_n), comp_cnt(_n), fa(_n), sz(_n, 1) {
        iota(fa.begin(), fa.end(), 0);
    }
    
    int findset(int x) {
        return fa[x] == x ? x : fa[x] = findset(fa[x]);
    }
    
    void unite(int x, int y) {
        x = findset(x);
        y = findset(y);
        if (x == y) {
            return;
        }
        if (sz[x] < sz[y]) {
            swap(x, y);
        }
        fa[y] = x;
        sz[x] += sz[y];
        --comp_cnt;
    }
    
    bool connected(int x, int y) {
        x = findset(x);
        y = findset(y);
        return x == y;
    }
};

using PII = pair<int, int>;

class Solution {
public:
    vector<vector<int>> matrixRankTransform(vector<vector<int>>& a) {
        int m = a.size();

        int n = a[0].size();
        
        UF uf(m * n);
        
        vector<PII> v;
        for (int i = 0; i < m; ++i) {
            v.clear();
            for (int j = 0; j < n; ++j) {
                v.emplace_back(a[i][j], j);
            }
            sort(v.begin(), v.end());
            for (int j = 0; j < n - 1; ++j) {
                if (v[j].first == v[j + 1].first) {
                    uf.unite(i * n + v[j].second, i * n + v[j + 1].second);

                }
            }
        }
        
        for (int j = 0; j < n; ++j) {
            v.clear();
            for (int i = 0; i < m; ++i) {
                v.emplace_back(a[i][j], i);
            }
            sort(v.begin(), v.end());
            for (int i = 0; i < m - 1; ++i) {
                if (v[i].first == v[i + 1].first) {
                    uf.unite(v[i].second * n + j, v[i + 1].second * n + j);
                }
            }
        }
        
        vector<vector<int>> edges(m * n);
        vector<int> indeg(m * n);
        vector<int> seen(m * n);
        for (int i = 0; i < m; ++i) {
            v.clear();
            for (int j = 0; j < n; ++j) {
                v.emplace_back(a[i][j], j);
                seen[uf.findset(i * n + j)] = 1;
            }
            sort(v.begin(), v.end());
            for (int j = 0; j < n - 1; ++j) {
                if (v[j].first != v[j + 1].first) {
                    int p = uf.findset(i * n + v[j].second);
                    int q = uf.findset(i * n + v[j + 1].second);
                    edges[p].push_back(q);

                    ++indeg[q];
                }
            }
        }
        
        for (int j = 0; j < n; ++j) {
            v.clear();
            for (int i = 0; i < m; ++i) {
                v.emplace_back(a[i][j], i);
            }
            sort(v.begin(), v.end());
            for (int i = 0; i < m - 1; ++i) {
                if (v[i].first != v[i + 1].first) {
                    int p = uf.findset(v[i].second * n + j);
                    int q = uf.findset(v[i + 1].second * n + j);
                    edges[p].push_back(q);
                    ++indeg[q];
                }
            }
        }
        
        vector<int> rep(m * n);
        queue<int> q;
        for (int i = 0; i < m * n; ++i) {
            if (seen[i] && !indeg[i]) {
                rep[i] = 1;
                q.push(i);
            }
        }
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v: edges[u]) {
                rep[v] = max(rep[v], rep[u] + 1);
                --indeg[v];
                if (!indeg[v]) {
                    q.push(v);
                }
            }
        }
        
        vector<vector<int>> ans(m, vector<int>(n));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int id = i * n + j;
                ans[i][j] = rep[uf.findset(id)];
            }
        }
        return ans;
    }
};
