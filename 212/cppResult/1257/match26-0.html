class Num {
public:
    int x, y, a;
    Num(int _x, int _y, int _a): x(_x), y(_y), a(_a) {}
};

class Nums {
	public:
	int idx, x;
	Nums(int _idx, int _x): x(_x), idx(_idx) {}
};

bool cmp(const Nums &a, const Nums &b) {
	return a.x < b.x;
}

vector<int> discretize(vector<int> &x, int &maxCur) {
	// cout << endl;
	vector<Nums> a;
	a.clear();
	for (int i = 0; i < x.size(); ++i) {
		a.push_back(Nums(i, x[i]));
	}
	sort(a.begin(), a.end(), cmp);
	int cur = 0;
	vector<int> res(a.size());
	for (int i = 0; i < a.size(); ++i) {
		// cout << a[i].idx << ' ' << a[i].x << endl;
		if (i != 0 && a[i].x != a[i - 1].x) {
			++cur;
		}
		res[a[i].idx] = cur;
	}
	maxCur = cur;
	return res;
}

bool cmp1(const Num &a, const Num &b) {
    return a.a < b.a;
}
class Solution {
public:
    int n, m;
    vector<int> f;
    int mc;
    int gf(int x) {
        if (f[x] == x) {
            return x;
        }
        return (f[x] = gf(f[x]));
    }
    int getn(int x, int y) {
        return x * m + y;
    }
    void makeLink(vector<vector<int> > &b) {
        vector<int> lk(mc + 1, -1);
        vector<int> app;
        for (int x = 0; x < n; ++x) {
            app.clear();
            for (int y = 0; y < m; ++y) {
                int a = b[x][y];
                if (lk[a] == -1) {
                    lk[a] = getn(x, y);
                    app.push_back(a);
                } else {
                    f[gf(getn(x, y))] = gf(lk[a]);
                    // cout << "link1 " << getn(x, y) << ' ' << lk[a] << endl;
                }
            }
            for (int i = 0; i < app.size(); ++i) {
                lk[app[i]] = -1;
            }
        }
        for (int y = 0; y < m; ++y) {
            app.clear();
            for (int x = 0; x < n; ++x) {
                int a = b[x][y];
                if (lk[a] == -1) {
                    lk[a] = getn(x, y);
                    app.push_back(a);
                } else {
                    f[gf(getn(x, y))] = gf(lk[a]);
                    // cout << "link2 " << getn(x, y) << ' ' << lk[a] << endl;
                }
            }
            for (int i = 0; i < app.size(); ++i) {
                lk[app[i]] = -1;
            }
        }
    }
    vector<vector<int>> matrixRankTransform(vector<vector<int>>& matrix) {
        vector<Num> a;
        a.clear();
        vector<int> ax;
        ax.clear();
        n = matrix.size(); m = matrix[0].size();
        for (int x = 0; x < n; ++x) {
            for (int y = 0; y < m; ++y) {
                ax.push_back(matrix[x][y]);
            }
        }
        ax = discretize(ax, mc);
        int cur = 0;
        for (int x = 0; x < n; ++x) {
            for (int y = 0; y < m; ++y) {
                a.push_back(Num(x, y, ax[cur]));
                matrix[x][y] = ax[cur];
                ++cur;
            }
        }
        sort(a.begin(), a.end(), cmp1);
        vector<vector<int> > res(n);
        for (int i = 0; i < n; ++i) {
            res[i] = vector<int>(m, -1);
        }
        vector<int> rowR(n, 0), colR(m, 0);
        vector<int> rowV(n, -1), colV(m, -1);
        
        f = vector<int>(n * m, 0);
        int tot = n * m; 
        for (int i = 0; i < tot; ++i) {
            f[i] = i;
        }
        makeLink(matrix);
        // for (int i = 0; i < n * m; ++i) {
        //     cout << gf(i) << ' ';
        // }
        // cout << endl;
        vector<int> rk(n * m, 1);
        
        int currr = 0;
        for (int i = 0; i < a.size(); ++i) {
            int x = a[i].x, y = a[i].y, aa = a[i].a;
            int nn = getn(x, y);
            currr = rk[gf(nn)] - 1;
            
            if (rowV[x] >= 0 && gf(rowV[x]) != gf(nn)) {
                currr = max(currr, rk[gf(rowV[x])]);
            }
            if (colV[y] >= 0 && gf(colV[y]) != gf(nn)) {
                currr = max(currr, rk[gf(colV[y])]);
            }
            
            // if (aa > rowV[x] && rowV[x] >= 0) {
            //     currr = max(currr, rk[rowV[x]]);
            // }
            // if (aa > colV[y] && colV[y] >= 0) {
            //     currr = max(currr, rk[colV[y]]);
            // }
            rowV[x] = colV[y] = gf(nn);
            rk[gf(nn)] = currr + 1;
        }
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                res[i][j] = rk[gf(getn(i, j))];
            }
        }
        return res;
    }
};
