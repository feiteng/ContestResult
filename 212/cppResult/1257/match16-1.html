class Solution {
private:
    int n, m;
    vector<int> fa;

    int id(int x, int y) {
        return x * m + y + 1;
    }

    int fis(int x) {
        return fa[x] == x ? x : fa[x] = fis(fa[x]);
    }

public:
    vector<vector<int>> matrixRankTransform(vector<vector<int>> &matrix) {

        n = matrix.size(), m = matrix[0].size();
        fa.resize(n * m + 1);
        for (int i = 1; i <= n * m; i++)
            fa[i] = i;
        for (int i = 0; i < n; i++) {
            map<int, int> v;
            for (int j = 0; j < m; j++) {
                if (v.count(matrix[i][j])) {
                    auto p = v[matrix[i][j]];
                    auto q = id(i, j);
                    fa[fis(p)] = fis(q);
                }
                v[matrix[i][j]] = id(i, j);

            }
        }
        for (int j = 0; j < m; j++) {
            map<int, int> v;
            for (int i = 0; i < n; i++) {
                if (v.count(matrix[i][j])) {
                    auto p = v[matrix[i][j]];
                    auto q = id(i, j);
                    fa[fis(p)] = fis(q);
                }
                v[matrix[i][j]] = id(i, j);
            }
        }
        vector<vector<int> > E(n * m + 1);

        for (int i = 0; i < n; i++) {
            vector<pair<int, int> > tmp;
            for (int j = 0; j < m; j++)
                tmp.emplace_back(matrix[i][j], j);
            sort(tmp.begin(), tmp.end());
            for (int j = 1; j < m; j++)
                if (tmp[j].first != tmp[j - 1].first) {
                    auto p = fis(id(i, tmp[j].second));
                    auto q = fis(id(i, tmp[j - 1].second));
                    E[q].emplace_back(p);

                }
        }
        for (int j = 0; j < m; j++) {
            vector<pair<int, int> > tmp;
            for (int i = 0; i < n; i++)
                tmp.emplace_back(matrix[i][j], i);
            sort(tmp.begin(), tmp.end());
            for (int i = 1; i < n; i++)
                if (tmp[i].first != tmp[i - 1].first) {
                    auto p = fis(id(tmp[i].second, j));
                    auto q = fis(id(tmp[i - 1].second, j));
                    E[q].emplace_back(p);
                }
        }
        vector<int> ind(n * m + 1);
        for (int i = 1; i <= n * m; i++) {
            sort(E[i].begin(), E[i].end());
            E[i].resize(unique(E[i].begin(), E[i].end()) - E[i].begin());
            for (auto j:E[i])
                ind[j]++;
        }
        queue<int> Q;
        auto ans = matrix;
        for (int i = 1; i <= n * m; i++)
            if (fa[i] == i && ind[i] == 0) {
                Q.push(i);
                ans[(i - 1) / m][(i - 1) % m] = 1;
            }
        while (!Q.empty()) {
            int u = Q.front();
            Q.pop();
            for (auto i:E[u])
                if (--ind[i] == 0) {
                    ans[(i - 1) / m][(i - 1) % m] = ans[(u - 1) / m][(u - 1) % m] + 1;
                    Q.push(i);
                }
        }
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++) {
                int tmp = fis(id(i, j)) - 1;
                ans[i][j] = ans[tmp / m][tmp % m];
            }
        return ans;
    }
};
