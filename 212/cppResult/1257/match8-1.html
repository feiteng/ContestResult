class Solution {
public:
  int N;
  int M;
  vector<vector<int>> G;
  vector<int> indeg;
  vector<int> ans;
  int id(int x, int y) {
    return find(x*M+y);
  }
  vector<int> P;
  int find(int x) {
    return P[x] == x? x: (P[x] = find(P[x]));
  }
  
  vector<vector<int>> matrixRankTransform(vector<vector<int>>& matrix) {
    G.clear();
    indeg.clear();
    ans.clear();
    N = matrix.size();
    M = matrix[0].size();
    G.resize(N*M);
    indeg.resize(N*M);
    ans.resize(N*M);
    P.resize(N*M);

    for (int i = 0; i < N*M; i++) P[i] = i;
    for (int i = 0; i < N; i++) {
      map<int, vector<int>> MM;
      for (int j = 0; j < M; j++) MM[matrix[i][j]].push_back(i*M+j);
      for (auto [k, v]: MM) {
        for (int i = 1; i < v.size(); i++) {
          int px = find(v[i]);
          int py = find(v[i-1]);
          if (px != py) P[px] = py;
        }

      }
    }
    for (int i = 0; i < M; i++) {
      map<int, vector<int>> MM;
      for (int j = 0; j < N; j++) MM[matrix[j][i]].push_back(j*M+i);
      for (auto [k, v]: MM) {
        for (int i = 1; i < v.size(); i++) {
          int px = find(v[i]);
          int py = find(v[i-1]);
          if (px != py) P[px] = py;
        }
      }
    }
    for (int i = 0; i < N; i++) {
      map<int, vector<int>> MM;
      for (int j = 0; j < M; j++) MM[matrix[i][j]].push_back(id(i, j));
      for (auto it = MM.begin(), it1 = it; it != MM.end(); it++) {
        if (it1 == it) continue;
        for (auto s: it1->second)  for (auto t: it->second)  G[s].push_back(t);

        it1 = it;
      }
    }
    for (int i = 0; i < M; i++) {
      map<int, vector<int>> MM;
      for (int j = 0; j < N; j++) MM[matrix[j][i]].push_back(id(j, i));
      for (auto it = MM.begin(), it1 = it; it != MM.end(); it++) {
        if (it1 == it) continue;
        for (auto s: it1->second)  for (auto t: it->second)  G[s].push_back(t);
        it1 = it;
      }
    }
    for (int i = 0; i < M*N; i++) {
      for ( auto t: G[i]) indeg[t]++;
    }
    queue<int> Q;
    for (int i = 0; i < N*M; i++) if (indeg[i] == 0) {
      Q.push(i);
      ans[i] = 1;
    }
    while (Q.size()) {
      auto s = Q.front(); Q.pop();
    //  cout << "# " << s/M << " " << s % M << ":" << ans[s] << endl;
      for (auto t: G[s]) {
        ans[t] = max(ans[t], ans[s]+1);
     //   cout << " " << t/M << " " << t % M << ":" << ans[t] << endl;
        if (--indeg[t] == 0)  Q.push(t);
      }
    }
    vector<vector<int>> ans2(N, vector<int>(M));
    for (int i = 0; i < N; i++) {
      for(int j = 0; j < M; j++) ans2[i][j] = ans[id(i, j)];
    }
    return ans2;
  }
};
