class Solution {
    //dijkstra 
    
    int[][] DIRS = {
        {-1,0},{1,0},{0,-1},{0,1}
    };

    
    public int minimumEffortPath(int[][] heights) {
        final int R = heights.length, C = heights[0].length; 
        
        boolean[][] vis = new boolean[R][C]; 
        int[][] bests = new int[R][C]; 
        for(int[] row : bests)
            Arrays.fill(row, Integer.MAX_VALUE);
        bests[0][0] = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[2] - b[2]); // row #, col #, effort
        pq.offer(new int[]{0, 0, 0});
        
        while(!pq.isEmpty()){
            int[] cur = pq.poll(); 
            if (vis[cur[0]][cur[1]])
                continue;
            vis[cur[0]][cur[1]] = true;
            
            for(int[] DIR : DIRS){
                int r = cur[0] + DIR[0], c = cur[1] + DIR[1]; 
                if(r == -1 || r == R || c == -1 || c == C)
                    continue; 
                int effort = Math.max(cur[2], Math.abs(heights[r][c] - heights[cur[0]][cur[1]]));
                if(bests[r][c] <= effort)
                    continue;
                bests[r][c] = effort;
                pq.offer(new int[]{r, c, effort});
            }
        }
        return bests[R-1][C-1];
    }
}
