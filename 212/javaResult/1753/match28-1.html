class Solution {
    
public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        int[][] mins = new int[m][n];
        for (int i = 0; i < m; i++) {
            Arrays.fill(mins[i], Integer.MAX_VALUE);
        }
        mins[0][0] = 0;
        Queue<int[]> queue = new LinkedList<>();
        
int[][] dirs = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        queue.add(new int[]{0, 0, 0});
        while (!queue.isEmpty()) {
            int[] tmp = queue.poll();
            
int x = tmp[0], y = tmp[1], cur = heights[x][y], curmax = tmp[2];
            for (int[] dir : dirs) {
                int newx = x + dir[0], newy = y + dir[1];
                if (newx < 0 || newx >= m || newy < 0 || newy >= n) continue;
                int gap = Math.abs(cur - heights[newx][newy]);
                int routeMax = Math.max(curmax, gap);
                if (routeMax < mins[newx][newy]) {
                    mins[newx][newy] = routeMax;
                    queue.offer(new int[]{newx, newy, routeMax});
                }
            }
        }
        return mins[m - 1][n - 1];
    }
}
