class Solution {
    public int minimumEffortPath(int[][] heights) {
        final int R = heights.length;
        final int C = heights[0].length;
        
        int[][] dp = new int[R][C];
        for (int r = 0; r < R; r++) {
            Arrays.fill(dp[r], Integer.MAX_VALUE);
        }
        dp[0][0] = 0; 
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[2] - b[2]);
        pq.add(new int[] {0,0,0});
        while (pq.size() > 0) {
            int[] p = pq.poll();
            for (int[] d : DIRS) {
                int nr = p[0] + d[0];
                int nc = p[1] + d[1];
                if (nr >= 0 && nr < R && nc >= 0 && nc < C && dp[nr][nc] > Math.max(p[2], Math.abs(heights[p[0]][p[1]] - heights[nr][nc]))) {
                    dp[nr][nc] = Math.max(p[2], Math.abs(heights[p[0]][p[1]] - heights[nr][nc]));
                    pq.add(new int[] {nr, nc, dp[nr][nc]});
                }
            }
        }

        return dp[R - 1][C - 1];
    }

    int[][] DIRS = {{1,0},{-1,0},{0,1},{0,-1}};
}
