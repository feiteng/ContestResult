class Solution {
  private static final int[] DIRS = {-1, 0, 1, 0, -1};

  public int minimumEffortPath(int[][] points) {
    final int R = points.length, C = points[0].length;

    boolean[][] vis = new boolean[R][C];
    int[][] bests = new int[R][C];
    for (int[] row : bests)
      Arrays.fill(row, Integer.MAX_VALUE);
    bests[0][0] = 0;
    PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[2] - b[2]));
    pq.offer(new int[] {0, 0, 0}); // r, c, max cost
    while (!pq.isEmpty()) {
      int[] now = pq.poll();
      if (vis[now[0]][now[1]])
        continue;
      vis[now[0]][now[1]] = true;
      for (int j = 0; j != 4; j++) {
        int nr = now[0] + DIRS[j], nc = now[1] + DIRS[j + 1];
        if (nr == R || nr == -1 || nc == C || nc == -1 || vis[nr][nc])
          continue;
        int would = Math.max(now[2], Math.abs(points[now[0]][now[1]] - points[nr][nc]));
        if (bests[nr][nc] <= would)
          continue;
        bests[nr][nc] = would;
        pq.offer(new int[] {nr, nc, would});
      }
    }
    return bests[R - 1][C - 1];
  }
}
