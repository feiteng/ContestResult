class Solution {
    
    class Node {
        int x;
        int y; 
        int e;
        public Node(int x, int y, int e) {
            this.x = x;
            this.y = y;
            this.e = e;
        }
    }
    
    // use best first search
    
    public int minimumEffortPath(int[][] heights) {
        int[][] dir = new int[][]{{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
        int m = heights.length, n = heights[0].length;
        boolean[][] visited = new boolean[m][n];
        // for (int i = 0; i < m; i++) {
        //     Arrays.fill(effort[i], Integer.MAX_VALUE);
        // }
        PriorityQueue<Node> minHeap = new PriorityQueue<>((e1, e2) -> Integer.compare(e1.e, e2.e));
        minHeap.offer(new Node(0, 0, 0));
        while (!minHeap.isEmpty()) {
            Node cur = minHeap.poll();
            if (cur.x == m - 1 && cur.y == n - 1) return cur.e;
            if (visited[cur.x][cur.y]) continue;
            visited[cur.x][cur.y] = true;
            for (int i = 0; i < 4; i++) {
                int nextx = cur.x + dir[i][0];
                int nexty = cur.y + dir[i][1];
                if (nextx >= 0 && nextx < m && nexty >= 0 && nexty < n) {
                    minHeap.offer(new Node(nextx, nexty, Math.max(cur.e, Math.abs(heights[nextx][nexty] - heights[cur.x][cur.y]))));
                }
            }
            
        }
        return 0;
    }
}
