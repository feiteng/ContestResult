class Solution {
    int[][] dir = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length;
        int n = heights[0].length;
        int[][] cost = new int[m][n];
        for (int[] row: cost) Arrays.fill(row, Integer.MAX_VALUE);
  
        // Contains {cost, r, c}
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[0]-b[0]));
        pq.offer(new int[]{0,0, 0});
        //cost[0][0] = 0;
        
        while(!pq.isEmpty()){
            int[] cur = pq.poll();
            
            if (cur[1] == (m-1) && (cur[2] == (n-1))) return cur[0];
            
            // check cost for u
            if (cost[cur[1]][cur[2]] <= cur[0]) continue; // This PQ has duplciate entries.
            cost[cur[1]][cur[2]] = cur[0];
            for(int i = 0; i < 4; i++){
                int r = cur[1]+dir[i][0];
                int c = cur[2]+dir[i][1];
                if (r < 0 || r >= m || c < 0 || c >= n) continue;
                int newCost = Math.abs(heights[r][c]-heights[cur[1]][cur[2]]);
                newCost = Math.max(newCost, cur[0]);
                pq.offer(new int[]{newCost, r, c});
            }
        }
        
        return 0;
    }
}
