class Solution {
    
public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[2] - b[2]);
        pq.offer(new int[] {0, 0, 0});
        int[][] diff = new int[m][n];
        for (int i = 0; i < m; i++) {
            Arrays.fill(diff[i], -1);
        }
        
int[][] dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
while (!pq.isEmpty()) {
            int[] temp = pq.poll();
            if (temp[0] == m - 1 && temp[1] == n - 1) {
                return temp[2];
            }
            for (int[] dir : dirs) {
                int x = dir[0] + temp[0];
                int y = dir[1] + temp[1];
                if (x < m && x >= 0 && y < n && y >= 0) {
                    int curDiff = Math.max(temp[2], Math.abs(heights[temp[0]][temp[1]] - heights[x][y]));
                    if (diff[x][y] != -1 && curDiff >= diff[x][y]) continue;
                    diff[x][y] = curDiff;
                    pq.offer(new int[] {x, y, curDiff});
                
}
            }
        }
        return -1;
    }
}
