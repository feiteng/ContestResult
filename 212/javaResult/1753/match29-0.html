class Solution {
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        int[][] dp = new int[m][n];
        boolean[][] visited = new boolean[m][n];
        visited[0][0] = true;
        Queue<int[]> q = new PriorityQueue<>((a, b) -> Integer.compare(a[0], b[0]));
        if (m > 1) q.offer(new int[] { Math.abs(heights[1][0] - heights[0][0]), n });
        if (n > 1) q.offer(new int[] { Math.abs(heights[0][1] - heights[0][0]), 1 });
        while (!q.isEmpty() && !visited[m - 1][n - 1]) {
            int[] path = q.poll();
            int w = path[0], ser = path[1];
            int i = ser / n, j = ser % n;
            if (visited[i][j]) continue;
            visited[i][j] = true;
            dp[i][j] = w;
            if (i > 0 && !visited[i - 1][j]) {
                q.offer(new int[] { Math.max(w, Math.abs(heights[i][j] - heights[i - 1][j])), (i - 1) * n + j });
            }
            if (j > 0 && !visited[i][j - 1]) {
                q.offer(new int[] { Math.max(w, Math.abs(heights[i][j] - heights[i][j - 1])), i * n + j - 1 });
            }
            if (i + 1 < m && !visited[i + 1][j]) {
                q.offer(new int[] { Math.max(w, Math.abs(heights[i][j] - heights[i + 1][j])), (i + 1) * n + j });
            }
            if (j + 1 < n && !visited[i][j + 1]) {
                q.offer(new int[] { Math.max(w, Math.abs(heights[i][j] - heights[i][j + 1])), i * n + j + 1 });
            }
        }
        return dp[m - 1][n - 1];
    }
}
