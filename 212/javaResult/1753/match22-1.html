class Solution {
    private class State {
        int i, j, e;
        public State(int i, int j, int e) {
            this.i = i;
            this.j = j;
            this.e = e;
        }
    }
    
    public int minimumEffortPath(int[][] heights) {
        PriorityQueue<State> pq = new PriorityQueue<>(new Comparator<State>(){
            @Override
            public int compare(State s1, State s2) {
                return s1.e - s2.e;
            }
        });
        
        int m = heights.length, n = heights[0].length;
        boolean[][] visited = new boolean[m][n];
        int[][] dirs = 
new int[][]{{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        
pq.offer(new State(0, 0, 0));
        while (!pq.isEmpty()) {
            State s = pq.poll();
            if (s.i == m - 1 && s.j == n - 1) return s.e
;
            visited[s.i][s.j] = true;
            for (int[] dir : dirs) {
                int x = s.i + dir[0], y = s.j + dir[1];
                if (x >= 0 && x < m && y >= 0 && y < n && !visited[x][y]) {
                    pq.offer(new State(x, y, Math.max(s.e, Math.abs(heights[x][y] - heights[s.i][s.j]))));
                }
            }
        }
        
        return -1;
    }
}
