class Solution {
    public int minimumEffortPath(int[][] heights) {
        int R = heights.length, C = heights[0].length;
        int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        int[][] effort = new int[R][C];
        for (int[] e : effort) {
            Arrays.fill(e, Integer.MAX_VALUE);
        }
        Queue<int[]> q = new LinkedList();
        q.offer(new int[]{0, 0, 0});
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int r0 = cur[0], c0 = cur[1], ef = cur[2];
            if (ef > effort[r0][c0]) {
                continue;
            } else {
                effort[r0][c0] = ef;
            }
            for (int[] d : dirs) {
                int r = r0 + d[0], c = c0 + d[1];
                if (r >= 0 && c >= 0 && r < R && c < C) {
                    int nEf = Math.max(ef, Math.abs(heights[r][c] - heights[r0][c0]));
                    if (nEf < effort[r][c]) {
                        effort[r][c] = nEf;
                        q.offer(new int[]{r, c, nEf});
                    }
                }
            }
        }
        return effort[R - 1][C - 1];
    }
}
