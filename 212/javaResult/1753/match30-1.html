class Solution {
    private static final int[][] moves = {{0,1},{1,0},{0,-1},{-1,0}};

    public int minimumEffortPath(int[][] heights) {
        int row = heights.length, col = heights[0].length;
        Integer[][] dp = new Integer[row][col];
        dp[0][0] = 0;
        Queue<Integer> queue = new LinkedList<>();
        queue.add(0);
        
while (!queue.isEmpty()) {
            int tmp = queue.poll();
            int x = tmp / col, y = tmp % col;
            for (int[] mv : moves) {
                int xt = x + mv[0], yt = y + mv[1];
                if (xt >= 0 && xt < row && yt >= 0 && yt < col) {
                    int diff = Math.max(dp[x][y], Math.abs(heights[xt][yt] - heights[x][y]));
                    if (dp[xt][yt] == null || dp[xt][yt] > diff) {
                        dp[xt][yt] = diff;
                        queue.add(xt * col + yt);
                    
}
                }
            }
        }
        return dp[row - 1][col - 1];
    }
}
