class Solution {
    
    int[][] dirs = new int[][]{{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length;
        int n = heights[0].length;
        int[][] cache = new int[m][n];
        for (int[] cachei : cache) {
            Arrays.fill(cachei, -1);
        }
        int ans = dfs(0, 0, new boolean[m][n], cache, heights);
        //System.out.println("ans is: " + ans);
        return ans;
    }
    
    private int dfs(int sx, int sy, boolean[][] visited, int[][] cache, int[][] heights) {
        int m = heights.length;
        int n = heights[0].length;
        int[][] md = new int[m][n];
        for (int[] mdi : md) {
            Arrays.fill(mdi, Integer.MAX_VALUE);
        }
        md[0][0] = 0;
        LinkedList<int[]> queue = new LinkedList<>();
        queue.add(new int[]{0, 0});
        while (! queue.isEmpty()) {
            int[] f = queue.poll();
            int x = f[0];
            int y = f[1];
            for (int[] dir : dirs) {
                int nx = x + dir[0];
                int ny = y + dir[1];
                if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                    int cd = Math.max(md[x][y], Math.abs(heights[nx][ny] - heights[x][y]));
                    if (cd < md[nx][ny]) {
                        md[nx][ny] = cd;
                        queue.add(new int[]{nx, ny});
                    }
                }
            }
        }
        
        return md[m-1][n-1];
    }
}
