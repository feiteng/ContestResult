class Solution {
   
    public int minimumEffortPath(int[][] heights) {

        int row = heights.length;
        int col = heights[0].length;
        boolean visit[][] = new boolean[row][col];
        PriorityQueue<int[]> queue = new PriorityQueue<>(Comparator.comparingInt(o -> o[2]));
        queue.add(new int[]{0, 0, 0});

        
while (!queue.isEmpty()) {
            int[] poll = queue.poll();
            int i = poll[0];
            int j = poll[1];
            int k = poll[2];
            int h = heights[i][j];
            visit[i][j] = true;
            if (i == row - 1 && j == col - 1) return k;
            
if (i > 0 && !visit[i - 1][j]) {
                queue.add(new int[]{i - 1, j, Math.max(k, Math.abs(heights[i - 1][j] - h))});
            }
            if (j > 0 && !visit[i][j - 1]) {
                queue.add(new int[]{i, j - 1, Math.max(k, Math.abs(heights[i][j - 1] - h))});
            }
            if (i + 1 < row && !visit[i + 1][j]) {
                queue.add(new int[]{i + 1, j, Math.max(k, Math.abs(heights[i + 1][j] - h))});
            }
            if (j + 1 < col && !visit[i][j + 1]) {
                queue.add(new int[]{i, j + 1, Math.max(k, Math.abs(heights[i][j + 1 ]-h))});
            }
        }
        return 0;
    }
}
