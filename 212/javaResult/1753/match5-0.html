class Solution {
    int[][] dirs = {{1,0},{-1,0},{0,1},{0,-1}};
    public int minimumEffortPath(int[][] A) {
        int a = A.length;
        int b = A[0].length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((x,y)->x[2]-y[2]);
        pq.add(new int[]{0,0,0});
        int[][] visited = new int[a][b];
        for(int i = 0; i < a; i++){
            Arrays.fill(visited[i], Integer.MAX_VALUE);
        }
        while(!pq.isEmpty()){
            int[] e = pq.poll();
            if(e[0] == a-1 && e[1] == b-1){
                return e[2];
            }
            for(int[] dir: dirs){
                int x = e[0]+dir[0];
                int y = e[1] + dir[1];
                if(x >= 0 && y >=0 && x < a && y < b){
                    int newMin = Math.max(e[2], Math.abs(A[x][y]-A[e[0]][e[1]]));
                    if(newMin < visited[x][y]){
                        visited[x][y] = newMin;
                        pq.add(new int[]{x,y,newMin});
                    }
                }
            }
        }
        return -1;

        // }

        //     for(int j = 0; j < b; j++){
        //         int cur = A[i][j];

        //             dp[i][j] = Math.min(dp[i][j], dp[i-1][j] + Math.abs(cur-A[i-1][j]));
        //         }
        //         if(j > 0){
        //             dp[i][j] = Math.min(dp[i][j], dp[i][j-1] + Math.abs(cur-A[i][j-1]));
        //         }
        //         if(i < a){}
        //     }
        // }
    }
}
