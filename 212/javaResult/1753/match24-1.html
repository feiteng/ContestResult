class Solution {
    private final int[][] dirs = new int[][] {{0,1},{1,0},{-1,0},{0,-1}};
    public int minimumEffortPath(int[][] heights) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2) -> o1[2] - o2[2]);
        boolean[][] visited = new boolean[heights.length][heights[0].length];
        
pq.offer(new int[] {0,0,0});
        
        while (!pq.isEmpty()) {
            int[] curr = pq.poll();
            int x = curr[0];
            int y = curr[1];
            int currMax = curr[2];
            int height = heights[x][y];
            
            
if (x == heights.length-1 && y == heights[0].length-1) {
                return currMax;
            }
            
            if (visited[x][y]) {
                continue;
            }
            visited[x][y] = true;
            
            for (int[] dir : dirs) {
                int nx = x + dir[0];
                int ny = y + dir[1];
                if (nx >= 0 && nx < heights.length && ny >= 0 && ny < heights[0].length && !visited[nx][ny]) {
                    int nHeight = heights[nx][ny];
                    int diff = Math.abs(nHeight - height);
                    pq.offer(new int[] {nx, ny, 
Math.max(diff, currMax)});
                }
            }
        }
        return 0;
    }
    
}
