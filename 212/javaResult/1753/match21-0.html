class Solution {
    public int minimumEffortPath(int[][] heights) {
        int R = heights.length, C = heights[0].length;
        int[][] dp = new int[R][C];
        for (int i = 0; i < dp.length; i++) Arrays.fill(dp[i], 2_000_000);
        PriorityQueue<int[]> q = new PriorityQueue<>(1, (a, b) -> Integer.compare(a[2], b[2]));
        q.offer(new int[]{0, 0, 0});
        dp[0][0] = 0;
        int[] di = new int[]{0, 0, 1, -1},
              dj = new int[]{1, -1, 0, 0};
        while (!q.isEmpty()) {
            int[] e = q.poll();
            int i = e[0], j = e[1], f = e[2];
            // System.out.println(i + " " + j + " " + f);
            if (f > dp[i][j]) continue;
            if (i == R - 1 && j == C - 1) return f;
            for (int k = 0; k < 4; k++) {
                int ni = i + di[k], nj = j + dj[k];
                if (ni < 0 || ni >= R || nj < 0 || nj >= C) continue;
                int nf = Math.max(f, Math.abs(heights[ni][nj] - heights[i][j]));
                if (nf >= dp[ni][nj]) continue;
                dp[ni][nj] = nf;
                q.offer(new int[]{ni, nj, nf});
            }
        }
        return 0;
    }
}
