class Solution {
    
private static class Point {
        int x;
        int y;
        int diff;
        Point(int x, int y, int diff) {
            this.x = x;
            this.y = y;
            this.diff = diff;
        }
    }
    
public int minimumEffortPath(int[][] h) {
        int[][] c = new int[h.length][h[0].length];
        for (int i = 0; i < c.length; i++) {
            for (int j = 0; j < c[i].length; j++) {
                c[i][j] = Integer.MAX_VALUE;
            }
        }
        c[0][0] = 0;
        Queue<Point> q = new LinkedList<>();
        q.add(new Point(0, 0, 0));
        while (!q.isEmpty()) {
            Point p = q.remove();
            
if (p.x + 1 < h.length && Math.max(p.diff, Math.abs(h[p.x + 1][p.y] - h[p.x][p.y])) < c[p.x + 1][p.y]) {
                c[p.x + 1][p.y] = Math.max(p.diff, Math.abs(h[p.x + 1][p.y] - h[p.x][p.y]));
                q.add(new Point(p.x + 1, p.y, c[p.x + 1][p.y]));
                
            }
            if (p.x - 1 >= 0 && Math.max(p.diff, Math.abs(h[p.x - 1][p.y] - h[p.x][p.y])) < c[p.x - 1][p.y]) {
                c[p.x - 1][p.y] = Math.max(p.diff, Math.abs(h[p.x - 1][p.y] - h[p.x][p.y]));
                q.add(new Point(p.x - 1, p.y, c[p.x - 1][p.y]));
                
            }
            if (p.y + 1 < h[0].length && Math.max(p.diff, Math.abs(h[p.x][p.y + 1] - h[p.x][p.y])) < c[p.x][p.y + 1]) {
                c[p.x][p.y + 1] = Math.max(p.diff, Math.abs(h[p.x][p.y + 1] - h[p.x][p.y]));
                q.add(new Point(p.x, p.y + 1, c[p.x][p.y + 1]));
                
            }
            if (p.y - 1 >= 0 && Math.max(p.diff, Math.abs(h[p.x][p.y - 1] - h[p.x][p.y])) < c[p.x][p.y - 1]) {
                c[p.x][p.y - 1] = Math.max(p.diff, Math.abs(h[p.x][p.y - 1] - h[p.x][p.y]));
                q.add(new Point(p.x, p.y - 1, c[p.x][p.y - 1]));
                
            }
        }
        return c[h.length - 1][h[0].length - 1];
    }
}
