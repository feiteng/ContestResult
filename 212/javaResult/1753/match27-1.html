class Solution {
    
int[][] dir = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    
public int minimumEffortPath(int[][] heights) {
       int m = heights.length, n = heights[0].length;
       int[][] dp =new int[m][n];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                dp[i][j] = 10000000;
            }
        }
       PriorityQueue<int[]> pq = new  PriorityQueue<>((a,b)-> a[0]- b[0]);
        
pq.offer(new int[]{0,0,0});
        
        while(pq.size() > 0){
            int[] cur = pq.poll();
            int w = cur[0];
            int x = cur[1];
            int y = cur[2];
            //System.out.println(w +" "+x + " "+ y);
            if(dp[x][y] < w)continue;//no need to propagate
            dp[x][y] = w;
            if(x == m -1 && y == n -1) break;
            
for(int k =0; k < 4; k++){
                int nx = x + dir[k][0];
                int ny = y + dir[k][1];
                if(nx < 0 || ny < 0 || nx >= m || ny >= n)continue;//out of bound
                int diff = Math.abs(heights[x][y] - heights[nx][ny]);
                int nw = w > diff? w: diff;
                if(nw < dp[nx][ny]){
                    
pq.offer(new int[]{nw, nx, ny});
                }
            }
        }
        return dp[m - 1][n - 1];
        
    }
}
