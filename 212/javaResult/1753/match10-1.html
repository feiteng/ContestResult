class Solution {
    final int[][] direction = new int[][] {{1,0}, {-1,0}, {0, 1}, {0, -1}};
    public int minimumEffortPath(int[][] heights) {
        Deque<int[]> que = new ArrayDeque<>();
        int row = heights.length;
        int column = heights[0].length;
        int[][] count = new int[row][column];
        for (int i = 0; i < row; i++) {
            Arrays.fill(count[i], Integer.MAX_VALUE);
        }
        
        count[0][0] = 0;
        
        
        que.add(new int[] {0, 0});
        
        
while (!que.isEmpty()) {
            int[] now = que.pollFirst();
            for (int[] dir : direction) {
                int newX = now[0] + dir[0];
                int newY = now[1] + dir[1];
                
                if (newX >= 0 && newX < row && newY >= 0 && newY < column) {
                    int t = Math.max(count[now[0]][now[1]], Math.abs(heights[now[0]][now[1]] - heights[newX][newY]));
                    if (t < count[newX][newY]) {
                        count[newX][newY] = t;
                        que.add(new int[] {newX, newY});
                    }
                }
            }
        }
        
        return count[row - 1][column - 1];
    }
}
