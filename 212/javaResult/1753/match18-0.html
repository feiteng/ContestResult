class Solution {
    public int minimumEffortPath(int[][] heights) {
        int[][] minPath = new int[heights.length][heights[0].length];
        for (int i = 0; i < minPath.length; i++) {
            for (int j = 0; j < minPath[0].length; j++) {
                minPath[i][j] = Integer.MAX_VALUE;
            }
        }
        minPath[0][0] = 0;
        PriorityQueue<Point> points = new PriorityQueue<>(Comparator.comparingInt(o -> o.distance));
        Point start = new Point(0,0,minPath[0][0]);
        points.offer(start);

        while (!points.isEmpty()) {
            Point poll = points.poll();
            //已无效
            int x = poll.x;
            int y = poll.y;
            if (poll.distance > minPath[x][y]) {
                continue;
            }
            //左
            if (x > 0 && minPath[x -1][y] > Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x-1][y]))) {
                minPath[x -1][y] = Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x-1][y]));
                points.offer(new Point(x-1,y,minPath[x-1][y]));
            }
            //右
            if (x < heights.length-1 && minPath[x +1][y] > Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x+1][y]))) {
                minPath[x +1][y] = Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x+1][y]));
                points.offer(new Point(x+1,y,minPath[x+1][y]));
            }
            //上
            if (y > 0 && minPath[x][y-1] > Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x][y-1]))) {
                minPath[x][y-1] = Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x][y-1]));
                points.offer(new Point(x,y-1,minPath[x][y-1]));
            }
            //下
            if (y < heights[0].length-1 && minPath[x ][y+1] > Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x][y+1]))) {
                minPath[x ][y+1] = Math.max(minPath[x][y] , Math.abs(heights[x][y] - heights[x][y+1]));
                points.offer(new Point(x,y+1,minPath[x][y+1]));
            }
        }
        return minPath[minPath.length-1][minPath[0].length-1];
    }
    class Point{
        int x;
        int y;
        int distance;

        public Point(int x, int y, int distance) {
            this.x = x;
            this.y = y;
            this.distance = distance;
        }
    }
}
