class Solution {
    
private static final int[][] MOVES = {{0, -1}, {1, 0}, {0, 1}, {-1, 0}};
    
    
public int minimumEffortPath(int[][] heights) {
        int[][] visited = new int[heights.length][heights[0].length];
        for (int i = 0; i < heights.length; i++) {
            Arrays.fill(visited[i], Integer.MAX_VALUE);
        }
        PriorityQueue<int[]> minHeap = new PriorityQueue<>((i1, i2) -> i1[2] - i2[2]);
        minHeap.offer(new int[]{0, 0, 0});
        visited[0][0] = 0;
        while (!minHeap.isEmpty()) {
            int[] current = minHeap.poll();
            
for (int[] move : MOVES) {
                int nextX = current[0] + move[0];
                int nextY = current[1] + move[1];
                if (nextX >= 0 && nextX < heights.length && nextY >= 0 && nextY < heights[0].length) {
                    int effort = Math.max(current[2], 
Math.abs(heights[nextX][nextY] - heights[current[0]][current[1]]));
                    if (effort < visited[nextX][nextY]) {
                        visited[nextX][nextY] = effort;
                        minHeap.offer(new int[]{nextX, nextY, effort});
                    }
                }
            }
        }
        return visited[heights.length - 1][heights[0].length - 1];
    }
}
