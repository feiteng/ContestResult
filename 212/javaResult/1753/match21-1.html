class Solution {
    public int minimumEffortPath(int[][] heights) {
        
        int M = heights.length, N = heights[0].length;
        
        int[][] efforts = new int[M][N];
        for (int i = 0; i < M; ++i) Arrays.fill(efforts[i], Integer.MAX_VALUE);
        efforts[0][0] = 0;
        
        // 0: i, 1: j, 2: max effort in the path
        Queue<int[]> queue = new LinkedList<>();
        queue.add(
new int[] {0, 0, 0});
        
        int[] dirs = new int[] {1, 0, -1, 0, 1};
        
        while (!queue.isEmpty()) {
            
            int[] node = queue.poll();
            int x = node[0], y = node[1], e = node[2];
            
for (int i = 0; i < 4; ++i) {
                int nx = x + dirs[i];
                int ny = y + dirs[i + 1];
                
                if (nx < 0 || nx == M || ny < 0 || ny == N) continue;
                int ne = Math.max(e, Math.abs(heights[nx][ny] - heights[x][y]));
                
                if (efforts[nx][ny] <= ne) continue;
                
                efforts[nx][ny] = ne;
                queue.add(new int[] {nx, ny, ne});
            }
            
        }
        
        return efforts[M - 1][N - 1];
    }
}
