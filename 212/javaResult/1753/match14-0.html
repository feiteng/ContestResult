class Solution {
    public int minimumEffortPath(int[][] heights) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->a[2]-b[2]);
        pq.add(new int[]{0,0,0});
        int n = heights.length, m = heights[0].length;
        int[][] min = new int[n][m];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                min[i][j] = Integer.MAX_VALUE;
            }
        }
        min[0][0] = 0;
        int[][] dirs = new int[][]{{-1,0},{1,0},{0,-1},{0,1}};
        while(true){
            int[] now = pq.remove();
            //System.out.println(now[0]+" "+now[1]+" "+now[2]);
            if(now[0]==n-1 && now[1]==m-1) return now[2];
            for(int[] dir: dirs){
                int x = now[0]+dir[0], y = now[1]+dir[1];
                if(x<0 || x>=n || y<0 || y>=m) continue;
                int val = Math.max(now[2],(int)Math.abs(heights[x][y]-heights[now[0]][now[1]]));
                if(val<min[x][y]){
                    min[x][y] = val;
                    pq.add(new int[]{x,y,val});
                }
            }
        }
    }
}
