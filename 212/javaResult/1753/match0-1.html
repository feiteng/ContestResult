class Solution {
    public int minimumEffortPath(int[][] heights) {
        int r=heights.length;
        int c=heights[0].length;
        
int[][] dir={{-1,0},{1,0},{0,1},{0,-1}};
        
        
int[][] dp=new int[r][c];
        for(int i=0;i<r;i++){
           Arrays.fill(dp[i],Integer.MAX_VALUE);
        }
        dp[0][0]=0;
        
        PriorityQueue<int[]> pq=new PriorityQueue<>((a,b)->dp[a[0]][a[1]]-dp[b[0]][b[1]]);
        pq.add(new int[]{0,0});
        
        while(!pq.isEmpty()){
            int[] val=pq.poll();
            int dist=dp[val[0]][val[1]];
            
for(int[] d:dir){
                int nr=d[0]+val[0];
                int nc=d[1]+val[1];
                if(nr>=0 && nc>=0 && nr<r && nc<c && dp[nr][nc]>Math.max(dist,Math.abs(heights[nr][nc]-heights[val[0]][val[1]])))                 {
                    dp[nr][nc]=Math.max(dist,Math.abs(heights[nr][nc]-heights[val[0]][val[1]]));
                    pq.add(new int[]{nr,nc});
                }
                    
            }
        }
        
        
        /*for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(i==0 && j==0)
                    continue;
                if(i==0)
                    dp[i][j]=Math.max(dp[i][j-1],Math.abs(heights[i][j-1]-heights[i][j]));
                else if(j==0)
                    dp[i][j]=Math.max(dp[i-1][j],Math.abs(heights[i-1][j]-heights[i][j]));
                else
                    dp[i][j]=Math.min(Math.max(dp[i-1][j],Math.abs(heights[i-1][j]-heights[i][j])),
                                      Math.max(dp[i][j-1],Math.abs(heights[i][j-1]-heights[i][j])));
                System.out.println(i+","+j+","+dp[i][j]);
            }
        }*/
        
        
        return dp[r-1][c-1];
    }
}
