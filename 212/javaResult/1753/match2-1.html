class Solution {
    
static int[][] ops = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    
public int minimumEffortPath(int[][] h) {
        int m = h.length, n = h[0].length;
        int[][] dist = new int[m][n];
        for (int i = 0; i < m; ++i) Arrays.fill(dist[i], Integer.MAX_VALUE);
        dist[0][0] = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>(1, (a, b) -> a[4] - b[4]);
        pq.add(new int[5]);
        
while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            for (int[] op : ops) {
                int i = cur[2] + op[0], j = cur[3] + op[1];
                if (0 <= i && i < m && 0 <= j && j < n && Math.max(Math.abs(h[i][j] - h[cur[2]][cur[3]]), cur[4]) < dist[i][j]) {
                    dist[i][j] = Math.max(Math.abs(h[i][j] - h[cur[2]][cur[3]]), cur[4]);
                    pq.add(new int[]{cur[2], cur[3], i, j, dist[i][j]});             
                }
            }
        }
        return dist[m - 1][n - 1];
    }
}
