class Solution {
    public int minimumEffortPath(int[][] heights) {
        int[][] mtr = new int[heights.length][heights[0].length];
		for (int[] row : mtr)
		{
			Arrays.fill(row, Integer.MAX_VALUE);
		}

		mtr[0][0] = 0;
		int[][] dir = 
new int[][] { { 1, 0 }, { -1, 0 }, { 0, 1 }, { 0, -1 } };

		Queue<int[]> q = new LinkedList
<int[]>();
		q.offer(new int[] { 0, 0 });
		while (!q.isEmpty())
		{
			int[] head = q.poll();
			int effort = mtr[head[0]][head[1]];
			for (int[] d : dir)

			{
				int x = head[0] + d[0];
				int y = head[1] + d[1];
				if (x >= 0 && x < heights.length && y >= 0
						&& y < heights[0].length)
				{
					int absDiff = Math
							.abs(heights[x][y] - heights[head[0]][head[1]]);
					int newEffort = Math.max(absDiff, effort);

					if (newEffort < mtr[x][y])
					{
						mtr[x][y] = newEffort;
						q.offer(new int[] { x, y });
					}
				}
			}
		}

		return mtr[heights.length - 1][heights[0].length - 1];
    }
}
