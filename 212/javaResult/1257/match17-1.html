class Solution {

    public int[][] matrixRankTransform(int[][] matrix) {
        TreeMap<Integer, ArrayList<int[]>> map = new TreeMap<>();
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                
if (!map.containsKey(matrix[i][j])) {
                    map.put(matrix[i][j], new ArrayList<>());
                }
                map.get(matrix[i][j]).add(new int[] { i, j });
            }
        }
        int[] ans[] = new int[matrix.length][matrix[0].length], r = new int[matrix.length],
                c = new int[matrix[0].length];
        for (int i : map.keySet()) {
            int[] max1 = new int[matrix.length], max2 = new int[matrix[0].length];
            for (int[] j : map.get(i)) {
                max1[j[0]] = Math.max(max1[j[0]], r[j[0]]);
                max2[j[1]] = Math.max(max2[j[1]], c[j[1]]);
            }
            for (int[] j : map.get(i)) {
                ans[j[0]][j[1]] = Math.max(max1[j[0]], max2[j[1]]) + 1;
                
r[j[0]] = Math.max(r[j[0]], ans[j[0]][j[1]]);
                c[j[1]] = Math.max(c[j[1]], ans[j[0]][j[1]]);
            }
            boolean flag = true;
            while (flag) {
                flag = false;
                for (int[] j : map.get(i)) {
                    if (r[j[0]] != c[j[1]]) {
                        flag = true;
                    }
                    ans[j[0]][j[1]] = Math.max(r[j[0]], c[j[1]]);
                    r[j[0]] = Math.max(r[j[0]], ans[j[0]][j[1]]);
                    c[j[1]] = Math.max(c[j[1]], ans[j[0]][j[1]]);
                }
            }
        }
        return ans;
    }
}
