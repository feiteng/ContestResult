class Solution {
    public int[][] matrixRankTransform(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        Node[][] nodes = new Node[m][n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                nodes[i][j] = new Node(i, j, matrix[i][j]);
            }
        }
        // row dependency.
        for(int i = 0; i < m; ++i) {
            List<Node> row = new ArrayList<>();
            for(int j = 0; j < n; ++j) {
                row.add(nodes[i][j]);
            }
            Collections.sort(row, (a, b) -> Integer.compare(a.value, b.value));
            for(int j = 1; j < n; ++j) {
                Node p = row.get(j-1);
                Node current = row.get(j);
                if (current.value > p.value) {
                    current.find().prev.add(p);
                } else {
                    current.union(p);
                }
            }
        }
        // col dependency.
        for(int j = 0; j < n; ++j) {
            List<Node> col = new ArrayList<>();
            for(int i = 0; i < m; ++i) {
                col.add(nodes[i][j]);
            }
            Collections.sort(col, (a, b) -> Integer.compare(a.value, b.value));
            for(int i = 1; i < m; ++i) {
                Node p = col.get(i-1);
                Node current = col.get(i);
                if (current.value > p.value) {
                    current.find().prev.add(p);
                } else {
                    current.union(p);
                }
            }
        }
        // result.
        int[][] result = new int[m][n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                result[i][j] = nodes[i][j].find().getRank();
            }
        }
        return result;
    }
    
    static class Node {
        
        Node parent = this;
        int size = 1;
        
        int x;
        int y;
        int value;
        int rank = -1;
        Set<Node> prev = new HashSet<>();
        
        Node(int x, int y, int value) {
            this.x = x;
            this.y = y;
            this.value = value;
        }
        
        Node find() {
            if (parent != this) {
                parent = parent.find();
            }
            return parent;
        }
        
        void union(Node that) {
            Node x = this.find();
            Node y = that.find();
            if (x == y) {
                return;
            }
            if (x.size < y.size) {
                Node tmp = x;
                x = y;
                y = tmp;
            }
            y.parent = x;
            x.size += y.size;
            x.prev.addAll(y.prev);
        }
        
        int getRank() {
            if (rank == -1) {
                if (prev.isEmpty()) {
                    rank = 1;
                } else {
                    int max = 0;
                    for(Node node : prev) {
                        max = Math.max(max, node.find().getRank());
                    }
                    rank = max + 1;
                }
            }
            return rank;
        }
    }
}
