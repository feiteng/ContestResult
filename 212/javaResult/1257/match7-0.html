class Solution {
    
    List<int[]> li = new ArrayList<>();
    public void get(int x,int y){
        li.add(new int[]{x,y});
        vis[x][y] = true;
        int v = m[x][y];
        for(int i=0;i<m.length;++i){
            if(i!=x&&m[i][y]==v&&!vis[i][y]){
                get(i,y);
            }
        }
        for(int j=0;j<m[0].length;++j){
            if(j!=y&&m[x][j]==v&&!vis[x][j]){
                get(x,j);
            }
        }    
    }
    
    
    
    
    int m[][];
    boolean vis[][];int rs[];int cs[];int rk[][];
    public int[][] matrixRankTransform(int[][] mm) {
        m = mm;
        int l = m.length;
        int ll = m[0].length;
        vis = new boolean[l][ll];

        PriorityQueue<int[]> q = new PriorityQueue<>((x,y)->{return Integer.compare(x[0],y[0]);});
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                q.offer(new int[]{m[i][j],i,j});
            }
        }

        rs = new int[l];
        cs = new int[ll];
        rk = new int[l][ll];

        while(q.size()>0){
            int cur[] = q.poll();
            int x =  cur[1];
            int y =  cur[2];
            if(vis[x][y]) continue;
            li.clear();
            get(x,y);
            int v = 0;
            for(int r[]:li){
                v = Math.max(v,  rs[r[0]]);
                v = Math.max(v,  cs[r[1]]);
            }
            v++;
            
            for(int r[]:li){
                rk[r[0]][r[1]] =v;
                rs[r[0]] = v;
                cs[r[1]] = v;
            }
            
            
            
            
        }
        
        return rk;
        
        
    }
}
