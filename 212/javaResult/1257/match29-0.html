class Solution {
    public int[][] matrixRankTransform(int[][] matrix) {
        Map<Integer, List<int[]>> map = new HashMap<>();
        TreeSet<Integer> valueSet = new TreeSet<>();
        int m = matrix.length;
        int n = matrix[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int value = matrix[i][j];
                valueSet.add(value);
                List<int[]> list = map.getOrDefault(value, new ArrayList<>());
                list.add(new int[] {i, j});
                map.put(value, list);
            }
        }
        int[][] ret = new int[m][n];
        int[] rowNum = new int[m];
        int[] colNum = new int[n];
        int[] rowRank = new int[m];
        int[] colRank = new int[n];
        Arrays.fill(rowNum, -1000000000-1);
        Arrays.fill(colNum, -1000000000-1);
        
        for (int value: valueSet) {
            List<int[]> list = map.get(value);
            boolean updated = true;
            while (updated) {
                updated = false;
                for (int[] pair: list) {
                    int x = pair[0];
                    int y = pair[1];
                    int rowR = (rowNum[x] == matrix[x][y]) ? rowRank[x] : rowRank[x] + 1;
                    int colR = (colNum[y] == matrix[x][y]) ? colRank[y] : colRank[y] + 1;
                    int newRank = Math.max(rowR, colR);
                    if (newRank != ret[x][y]) {
                        updated = true;
                        ret[x][y] = newRank;
                        rowNum[x] = matrix[x][y];
                        colNum[y] = matrix[x][y];
                        rowRank[x] = newRank;
                        colRank[y] = newRank;
                    }
                }
            }
        }
            
            return ret;
    }
}
