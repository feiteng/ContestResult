class Solution {
    class Block {
        int x, y, num;

        public Block(int x, int y, int num) {
            this.x = x;
            this.y = y;
            this.num = num;
        }


    }

    public void show(int [][]a){
        for (int[] ints : a) {
            for (int aa : ints) {
                System.out.print(aa);
            }
            System.out.println();
            System.out.println();
        }
    }
    public int[][] matrixRankTransform(int[][] matrix) {
        int[][] result = new int[matrix.length][matrix[0].length];
        int[] pi = new int[matrix.length];
        int[] pj = new int[matrix[0].length];
        ArrayList<Block> blocks = new ArrayList<>();
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                blocks.add(new Block(i, j, matrix[i][j]));
            }
        }
        blocks.sort((a, b) -> a.num - b.num);

        int i = 0;
        while (i < blocks.size()) {
            int j = i;
            while (j + 1 < blocks.size() && blocks.get(i).num == blocks.get(j+1).num)
                j++;

            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }            
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }            
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = Math.max(pi[x], pj[y]);
                pj[y] = Math.max(pi[x], pj[y]);
            }
            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                result[x][y] = Math.max(pi[x], pj[y])+1;
            }

            for (int k = i; k <= j; k++) {
                int x = blocks.get(k).x;
                int y = blocks.get(k).y;
                pi[x] = result[x][y];
                pj[y] = result[x][y];
            }
            i = j + 1;
        }

        return result;
    }
}
