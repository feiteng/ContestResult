class Solution {
    public char slowestKey(int[] rel, String keysPressed) {
        char[] str = keysPressed.toCharArray();
        char mc = str[0];
        int mx = rel[0];
        
        for (int i = 1; i < rel.length; i++) {
            int d = rel[i] - rel[i - 1];
            if (d > mx || (d == mx && str[i] > mc)) {
                mc = str[i];
                mx = d;
            }
        }
        
        return mc;
    }
}
