class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        int maxCount = 0;
        char maxChar = 'a';
        int prev = 0;
        for(int i=0; i<releaseTimes.length; i++) {
            if(releaseTimes[i] - prev > maxCount || (releaseTimes[i] - prev == maxCount && keysPressed.charAt(i) > maxChar)) {
                maxCount = releaseTimes[i] - prev;
                maxChar = keysPressed.charAt(i);
            }
            prev = releaseTimes[i];
        }
        return maxChar;
    }
}
