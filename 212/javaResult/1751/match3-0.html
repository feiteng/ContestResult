class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        int prev = 0;
        int n = releaseTimes.length;
        int max = 0;
        char key = 'a';
        for (int i = 0; i < n; i++) {
            int d = releaseTimes[i] - prev;
            if (d > max || (d == max && (keysPressed.charAt(i) - key > 0))) {
                max = d;
                key = keysPressed.charAt(i);
            }
            prev = releaseTimes[i];
        }
        return key;
    }
}
