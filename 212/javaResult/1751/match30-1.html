class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        int[] a = new int[26];
        for (int i = 0; i < keysPressed.length(); i++) 
{
            int cur = keysPressed.charAt(i) - 'a';
            a[cur] =Math.max(a[cur],  i == 0 ? releaseTimes[i] : releaseTimes[i] - releaseTimes[i - 1]);
        }
        int t = 0;
        for (int i = 0; i < 26; i++) {
            if (a[i] >= a[t]) t = i;
        }
        return (char)('a' + t);
    }
}
