class Solution {
    public char slowestKey(int[] r, String k) {
        char ans = k.charAt(0); int max = r[0];
        for (int i=1; i<r.length; i++) {
            if (max < r[i]-r[i-1]) {
                max = r[i]-r[i-1];
                ans = k.charAt(i);
            } else if (max == r[i]-r[i-1] && ans < k.charAt(i)) {
                ans = k.charAt(i);
            }
        }
        return ans;
    }
}
