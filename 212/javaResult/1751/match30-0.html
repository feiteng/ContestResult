class Solution {
    public char slowestKey(int[] r, String k) {
        long[] cnt = new long[26];
        for (int i = 0; i < r.length; i++) {
            char c = k.charAt(i);
            cnt[c - 'a'] = Math.max(r[i] - (i > 0 ? r[i - 1] : 0), cnt[c - 'a']);
        }
        
        int max = 25;
        for (int i = 24; i >= 0; i--) {
            if (cnt[i] > cnt[max]) max = i;
        }
        
        return (char) ('a' + max);
    }
}
