class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        char ret = 'a';
        int num = 0;
        for (int i = 0; i < releaseTimes.length; i += 1) {
            int start = i == 0 ? 0 : releaseTimes[i - 1];
            int end = releaseTimes[i];
            int cur = end - start;
            
char c = keysPressed.charAt(i);
            if (cur > num || (cur == num && c - ret > 0)) {
                ret = c;
                num = cur;
            }
        }
        return ret;
    }
}
