class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        int n=releaseTimes.length;
        int max=-1;
        int last=0;
        char ans='0';
        for(int i=0;i<n;i++){
            char c=keysPressed.charAt(i);
            int t=releaseTimes[i]-last;
            last=releaseTimes[i];
            if(t>max||(t==max&&c>ans)){
                max=t;
                ans=c;
            }
        }
        return ans;
    }
}
