class Solution {
    public char slowestKey(int[] rt, String k) {
        final int len = rt.length;
        
        int r = rt[0];
        char c = k.charAt(0);
        
        for (int i = 1; i < len; i++) {
            int cur = rt[i] - rt[i - 1];
            if (cur > r) {
                r = cur;
                c = k.charAt(i);
            } else if (cur == r) {
                if (k.charAt(i) > c) {
                    c = k.charAt(i);
                }
            }
        }
        return c;
    }
}
