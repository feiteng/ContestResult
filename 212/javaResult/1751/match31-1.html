class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        int max_dur = releaseTimes[0];
        char res = keysPressed.charAt(0);
        for (int i = 1; i < releaseTimes.length; i++) 
{
            int cur_dur = releaseTimes[i] - releaseTimes[i - 1];
            char ch = keysPressed.charAt(i);
            if (cur_dur > max_dur) {
                max_dur = cur_dur;
                res = ch;
            } else if (cur_dur == max_dur) {
                if (ch > res)
                    res = ch;
            }
        }
        return res;
    }
}
