class Solution {
    public char slowestKey(int[] r, String k) {
        int maxD = r[0], maxC = 0;
        for (int i = 1; i < r.length; i++) {
            int curD = r[i] - r[i - 1];
            if (curD > maxD || (curD == maxD && k.charAt(i) > k.charAt(maxC))) {
                maxD = curD;
                maxC = i;
            }
        }
        return k.charAt(maxC);
    }
}
