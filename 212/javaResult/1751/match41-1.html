class Solution {
    public char slowestKey(int[] A, String s) {
        char key = s.charAt(0);
        int result = A[0];
        
        for (int i = 1; i < A.length; i++) {
            
if (A[i] - A[i - 1] > result) {
                result = A[i] - A[i - 1];
                key = s.charAt(i);
            } else if (A[i] - A[i - 1] == result) {
                key = s.charAt(i) < key ? key : s.charAt(i);
            }
        }
        
        return key;
    }
}
