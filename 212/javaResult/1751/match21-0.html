class Solution {
    public char slowestKey(int[] R, String S) {
        char c = S.charAt(0);
        int cur = R[0];
        for (int i = 1; i < R.length; i++) {
            int curTime = R[i] - R[i - 1];
            if (curTime > cur) {
                cur = curTime;
                c = S.charAt(i);
            } else if (curTime == cur && cur < S.charAt(i)) {
                c = S.charAt(i);
            } 
        }
        return c;
    }
}
