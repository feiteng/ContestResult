class Solution {
    public char slowestKey(int[] releaseTimes, String keysPressed) {
        char res = 0;
        int max = 0, last = 0;
        for (int i = 0; i <releaseTimes.length; ++i) {
            int dur = releaseTimes[i] - last;
            if (dur > max) {
                max = dur;
                res = keysPressed.charAt(i);
            } else if (dur == max && keysPressed.charAt(i) > res) {
                res = keysPressed.charAt(i);
            }
            last = releaseTimes[i];
        }
        return res;
    }
}
