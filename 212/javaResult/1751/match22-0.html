class Solution {
    public char slowestKey(int[] r, String s) {
        char res = s.charAt(0);
        int max = r[0];
        for(int i = 1;i < r.length;i++){
            int diff = r[i]-r[i-1];
            if(diff > max){
                max = diff;
                res = s.charAt(i);
            }else if(diff == max && s.charAt(i)>res){
                res = s.charAt(i);
            }
        }
        return res;
    }
}
