class Solution {
  public char slowestKey(int[] releaseTimes, String keysPressed) {
    char bestKey = keysPressed.charAt(0);
    int bestDuration = 0;
    int prev = 0;
    for (int i = 0; i < releaseTimes.length; i++) {
      int curr = releaseTimes[i];
      char key = keysPressed.charAt(i);
      if (curr - prev > bestDuration) {
        bestDuration = curr - prev;
        bestKey = key;
      } else if (curr - prev == bestDuration && key > bestKey) {
        bestKey = key;
      }
      prev = curr;
    }
    
    return bestKey;
  }
}
