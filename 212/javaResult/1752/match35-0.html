class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> ans = new ArrayList<>();
        for (int q = 0; q < l.length; ++q) {
            int[] a = Arrays.copyOfRange(nums, l[q], r[q] + 1);
            Arrays.sort(a);
            int k = a[1] - a[0];
            boolean seq = true;
            for (int i = 2; i < a.length; ++i) {
                if (a[i] - a[i-1] != k) {
                    seq = false;
                    break;
                }
            }
            ans.add(seq);
        }
        
        return ans;
    }
}
