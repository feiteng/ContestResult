class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> list = new ArrayList<>();
        for (int i=0; i<l.length; i++) {
            list.add(isArithmetic(nums, l[i], r[i]));
        }
        return list;
    }
    
    private Boolean isArithmetic(int[] nums, int s, int e) {
        int[] tmp = new int[e - s + 1];
        for (int i=0; i<=e-s; i++) {
            tmp[i] = nums[s+i];
        }
        Arrays.sort(tmp);
        int diff = tmp[1] - tmp[0];
        for (int i=1; i<=e-s; i++) {
            if (tmp[i] - tmp[i-1] != diff)
                return false;
        }
        return true;
    }
}
