class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        int m=l.length;
        List<Boolean> res = new ArrayList<>();
        for(int i=0;i<m;++i){
            int[] sub = Arrays.copyOfRange(nums,l[i],r[i]+1);
            Arrays.sort(sub);
            int d=sub[1]-sub[0];
            boolean f=true;
            for(int j=2;j<sub.length;++j){
                if(sub[j]-sub[j-1]!=d){
                    f=false;
                    break;
                }
            }
            res.add(f);
        }
        return res;
    }
}
