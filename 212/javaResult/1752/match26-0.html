class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> res = new ArrayList<>();
        for (int i = 0; i < l.length; i++) {
            res.add(check(nums, l[i], r[i]));
        }
        return res;
    }
    private boolean check(int[] nums, int start, int end) {
        if (end - start + 1 < 2) return false;
        int[] ns = Arrays.copyOfRange(nums, start, end+1);
        Arrays.sort(ns);
        int diff = ns[1] - ns[0];
        for (int i = 2; i < ns.length; i++) {
            if (ns[i] - ns[i-1] != diff) return false;
        }
        return true;
    }
}
