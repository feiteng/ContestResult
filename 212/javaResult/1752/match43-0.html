class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> list = new ArrayList<>(l.length);
        for (int i = 0; i < l.length; i++) {
            list.add(subCheck(Arrays.copyOfRange(nums, l[i], r[i] + 1)));
        }
        return list;
    }
    public boolean subCheck(int[] nums){
        Arrays.sort(nums);
        int d = nums[1] - nums[0];
        for (int i = 2; i < nums.length; i++) {
            if(nums[i] - nums[i - 1] != d) return false;
        }
        return true;
    }
}
