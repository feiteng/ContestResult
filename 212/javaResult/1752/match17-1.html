class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> res = new ArrayList<>();
        for (int i = 0; i < l.length; i++) {
            int len = r[i] - l[i] + 1;
            
int[] num = new int[len];
            for (int j = l[i]; j <= r[i]; j++) {
                num[j-l[i]] = nums[j];
            }
            Arrays.sort(num);
            boolean ok = true;
            int d = num[1] - num[0];
            for (int j = 2; j < num.length; j++) {
                if (num[j] - num[j-1] != d) {
                    ok = false;
                    break;
                
}
            }
            res.add(ok);
        }
        return res;
    }
}
