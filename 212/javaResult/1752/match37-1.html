class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> list = new LinkedList<>();
        for(int i = 0; i < l.length; ++i) {
            int [] temp = new int[r[i] - l[i] + 1];
            if(temp.length == 2) {
                list.add(true);
                continue;
            }
            if(temp.length < 2) {
                list.add(false);
                continue;
            }
            
for(int j = 0; j < temp.length; ++j) {
                temp[j] = nums[l[i] + j];
            }
            Arrays.sort(temp);
            int pre = temp[1] - temp[0];
            boolean can = true;
            for(int j = 2; j < temp.length; ++j) {
                if(temp[j] - temp[j - 1] != pre) {
                    can = false;
                    break;
                
}
            }
            list.add(can);
        }
        return list;
    }
}
