class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        ArrayList<Boolean> result = new ArrayList<Boolean>(l.length);
        out:for (int i = 0; i < l.length; i++) {
            int len = r[i] - l[i] + 1;
            if(len <= 2) {
                result.add(true);
                continue;
            }
            
int[] copy = new int[len];
            System.arraycopy(nums,l[i],copy,0,len);
            Arrays.sort(copy);
            int distance = copy[1] - copy[0];
            for (int j = 2; j < copy.length; j++) {
                if (copy[j] - copy[j-1] != distance) {
                    result.add(false);
                    continue out;
                }
            }
            result.add(true);
        }
        return result;
    }
}
