class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        int m = nums.length, n = l.length;
        List<Boolean> res = new ArrayList<>();
        for (int i = 0;i < n;i++) {
            int[] A = Arrays.copyOfRange(nums, l[i], r[i] + 1);
            Arrays.sort(A);
            int d = A[1] - A[0];
            boolean flag = true;
            for (int j = 2;j < A.length;j++) {
                if (A[j] - A[j - 1] != d) {
                    flag = false;
                    break;
                }
            }
            res.add(flag);
        }
        return res;
    }
}
