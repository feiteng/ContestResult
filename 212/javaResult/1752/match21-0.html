class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        //if sorted subarray can be a subsequence
        //2500000
        List<Boolean> answ = new ArrayList<>();
        
        for (int i = 0; i < l.length; i++) {
            int[] copy = new int[r[i]-l[i]+1];
            System.arraycopy(nums, l[i], copy, 0, r[i]-l[i]+1);
            Arrays.sort(copy);
            int desiredDiff = copy[1]-copy[0];
            boolean wrong = false;
            for (int j = 2; j < copy.length; j++) {
                if (copy[j]-copy[j-1] != desiredDiff) {
                    wrong = true;
                    break;
                }
            }
            answ.add(!wrong);
        }
        
        return answ;
    }
}
