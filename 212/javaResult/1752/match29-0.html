class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        int n = nums.length, m = l.length;
        List<Boolean> res = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            int[] a = new int[r[i] - l[i] + 1];
            for (int j = l[i]; j <= r[i]; j++) {
                a[j - l[i]] = nums[j];
            }
            Arrays.sort(a);
            boolean b = true;
            int d = a[1] - a[0];
            for (int k = 2; k < a.length; k++) {
                if (a[k] - a[k - 1] != d) {
                    b = false;
                    break;
                }
            }
            res.add(b);
        }
        return res;
    }
}
