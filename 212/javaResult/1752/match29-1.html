class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        int n = nums.length, m = l.length;
        List<Boolean> res = new ArrayList<>();
        for(int i =0; i < m; i++){
            int st = l[i], ed = r[i], len = ed - st + 1;
            
int[] ps = new int[len];
            for(int j = 0; j <len; j++){
                ps[j] = nums[st+ j];
            }
            Arrays.sort(ps);
            boolean flag = true;
            int diff = ps[1] - ps[0];
            for(int j = 2; j < len; j++){
                if(diff != ps[j] - ps[j - 1]){
                    flag = false;
                    break;
                }
            }
            res.add(flag);
            
        }
        return res;
    }
}
