class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> res = new ArrayList<>();
        for (int i = 0; i < l.length; i++) {
            res.add(is(nums,l[i], r[i]));
        }
        
        return res;
    }
    
    boolean is(int[] nums, int l, int r) {
        if (r - l == 1) {
            return true;
        }
        
List<Integer> copy = new ArrayList<>();
        
        for (int i = l; i<=r; i++) {
            copy.add(nums[i]);
        }
        
        Collections.sort(copy);
        
        for (int i = 1; i < copy.size() - 1; i++) {
            if (copy.get(i+1) - copy.get(i) != copy.get(1) - copy.get(0)) {
                return false;
            }
        }
        
        return true;
    }
}
