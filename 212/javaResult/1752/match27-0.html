class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> result = new ArrayList<>();
        for (int i = 0; i < l.length; i++) {
            if (r[i] - l[i] < 1) {
                result.add(false);
                continue;
            }
            int[] copied = Arrays.copyOfRange(nums, l[i], r[i] + 1);
            Arrays.sort(copied);
            int delta = copied[1] - copied[0];
            boolean isArithmetic = true;
            for (int j = 2; j < copied.length; j++) {
                if (copied[j] - copied[j-1] != delta) {
                    isArithmetic = false;
                    break;
                }
            }
            result.add(isArithmetic);
        }
        return result;
    }
}
