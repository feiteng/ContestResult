class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        // for each query, call check
        List<Boolean> ans = new ArrayList<>();
        for (int i = 0; i < l.length; i++) {
            ans.add(check(nums, l[i], r[i]));
        }
        return ans;
    }
    
    private boolean check(int [] nums, int start, int end) {
        // sort, then linear scan check dif
        if (end <= start) return false;
        int[] array = new int[end - start + 1];
        for (int i = start; i <= end; i++) array[i - start] = nums[i];
        Arrays.sort(array);
        // -12, -12, -9, -6, -3
        int dif = array[1] - array[0];
        for (int i = 2; i < array.length; i++) {
            if (array[i] - array[i - 1] != dif) return false;
        }
        return true;
    }
}
