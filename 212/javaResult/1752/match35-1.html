class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> ans = new ArrayList<>();
        for(int i = 0;i<l.length;i++){
            int l1 = l[i], r1 = r[i];
            if(r1-l1<1){
                ans.add(false);
                continue;
            }
            if(r1-l1==1){
                ans.add(true);
                continue;
            }
            
int copy [] = Arrays.copyOfRange(nums,l1,r1+1);
            Arrays.sort(copy);
            int d = copy[1]-copy[0];
            boolean f = false;
            for(int j = 2;j<copy.length;j++){
                if(copy[j]-copy[j-1]!=d){
                    f= true;
                    break;
                }
            }
            ans.add(!f);
        }
        return ans;
    }
}
