class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        int n = l.length;
        List<Boolean> res = new ArrayList();
        for(int i=0; i<n; i++){
            int len = r[i]-l[i]+1;
            if(len<2){
                res.add(false);
                continue;
            }
            int[] arr = new int[len];
            System.arraycopy(nums, l[i], arr, 0, r[i]-l[i]+1); 
            Arrays.sort(arr);
            boolean isValid = true;
            for(int j=1; j<len; j++){
                if(arr[j]-arr[j-1] != arr[1]-arr[0]){
                    isValid = false;
                    break;                   
                }
            }
            res.add(isValid);
        }
        return res;
    }
}
