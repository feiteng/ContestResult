class Solution {
    public List<Boolean> checkArithmeticSubarrays(int[] nums, int[] l, int[] r) {
        List<Boolean> res = new ArrayList<>();
        for(int i=0;i<l.length;i++){
            res.add(canForm(nums, l[i],r[i]));
        }
        return res;
    }
    
    
    private boolean canForm(int[] nums, int l, int r){
        if(r-l+1<=2){
            return true;
        }
        int[] t = new int[r-l+1];
        int idx = 0;
        for(int i=l;i<=r;i++){
            t[idx++] = nums[i];
        }
        Arrays.sort(t);
        int d = t[1]-t[0];
        for(int i=2;i<t.length;i++){
            if(t[i]-t[i-1]!=d){
                return false;
            }
        }
        return true;
    }
}
