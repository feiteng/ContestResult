class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m, n = len(heights), len(heights[0])
        effort = [[float('inf')] * n for i in range(m)]
        effort[0][0] = 0
        dirs = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        queue = [(0, 0)]
        while queue:
            i, j = queue.pop(0)
            for ki, kj in dirs:
                ni, nj = i + ki, j + kj
                
if 0 <= ni < m and 0 <= nj < n and effort[ni][nj] > max(effort[i][j], abs(heights[ni][nj] - heights[i][j])):
                    effort[ni][nj] = max(effort[i][j], abs(heights[ni][nj] - heights[i][j]))
                    queue.append((ni, nj))
        return effort[-1][-1]
