class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        pq = [(0, 0, 0)]
        m, n = len(heights), len(heights[0])
        dist = [[10000000000] * n for i in range(m)]
        dist[0][0] = 0
        
        while pq:
            d, x, y = heapq.heappop(pq)
            for dx, dy in [(1,0), (-1,0), (0,1), (0, -1)]:
                nx = x + dx
                ny = y + dy
                if 0<= nx < m and 0 <= ny < n:
                    effort 
= max(dist[x][y], abs(heights[nx][ny] - heights[x][y]))
                    if dist[nx][ny] > effort:
                        dist[nx][ny] = effort
                        heapq.heappush(pq, (effort, nx, ny))
        print(dist)
        return dist[m-1][n-1]
