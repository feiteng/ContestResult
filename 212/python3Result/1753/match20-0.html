class Solution:
    def minimumEffortPath(self, hts: List[List[int]]) -> int:
        m, n = len(hts), len(hts[0])
        cost = [[-1] * n for _ in range(m)]
        
        pq = [(0, 0, 0)]
        
        while pq:
            c, x, y = heappop(pq)
            
            if x == m-1 and y == n-1: return c
            if cost[x][y] >= 0: continue
                
            cost[x][y] = c
            
            for i, j in [(x-1, y), (x+1, y), (x, y-1), (x, y+1)]:
                if 0 <= i < m and 0 <= j < n and cost[i][j] < 0:
                    heappush(pq, (max(c, abs(hts[i][j] - hts[x][y])), i, j))
        
        return -1
