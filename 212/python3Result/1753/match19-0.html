class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        R,C = len(heights),len(heights[0])
        BFS= [(0,0,0)]
        been = {(0,0):None}
        res = 0
        while(BFS):
            cost,r,c = heapq.heappop(BFS)
            if cost>res:res=cost
            if r==R-1 and c==C-1:
                return res
            been[r,c] = cost
            for nr,nc in [(r+1,c),(r-1,c),(r,c+1),(r,c-1)]:
                if 0<=nr<R and 0<=nc<C and (nr,nc) not in been:
                    ncost = abs(heights[nr][nc]-heights[r][c])
                    heapq.heappush(BFS,(ncost,nr,nc))
        return -1
