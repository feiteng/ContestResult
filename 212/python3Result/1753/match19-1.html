class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m,n = len(heights), len(heights[0])
        
        heap = [(0,0,0)]
        effort = {}
        
        while heap:
            eff, i, j = heapq.heappop(heap)
            if (i,j) == (m-1,n-1):
                return eff
            if (i,j) not in effort:
                effort[(
i,j)] = eff
                for a,b in [(i-1,j),(i+1,j),(i,j-1),(i,j+1)]:
                    if 0<=a<m and 0<=b<n and (a,b) not in effort:
                        plus = abs(heights[i][j] - heights[a][b])
                        heapq.heappush(heap, (max(eff,plus),a,b))
