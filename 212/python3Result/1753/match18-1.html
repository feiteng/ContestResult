from collections import deque

class Solution:
    def minimumEffortPath(self, h: List[List[int]]) -> int:
        n, m = len(h), len(h[0])
        que = deque()
        dp = [[100000000]*m for _ in range(n)]
        dp[0][0] = 0
        que.append((0,0))
        
while que:
            i,j = que.popleft()
            if i > 0 and max(dp[i][j], abs(h[i][j] - h[i-1][j])) < dp[i-1][j]:
                dp[i-1][j] = max(dp[i][j], abs(h[i][j] - h[i-1][j]))
                que.append((i-1,j))
            if i < n-1 and max(dp[i][j], abs(h[i][j] - h[i+1][j])) < dp[i+1][j]:
                dp[i+1][j] = max(dp[i][j], abs(h[i][j] - h[i+1][j]))
                que.append((i+1,j))
            if j > 0 and max(dp[i][j], abs(h[i][j] - h[i][j-1])) < dp[i][j-1]:
                dp[i][j-1] = max(dp[i][j], abs(h[i][j] - h[i][j-1]))
                que.append((i,j-1))
            if j < m-1 and max(dp[i][j], abs(h[i][j] - h[i][j+1])) < dp[i][j+1]:
                dp[i][j+1] = max(dp[i][j], abs(h[i][j] - h[i][j+1]))
                que.append((i,j+1))
        return dp[n-1][m-1]
