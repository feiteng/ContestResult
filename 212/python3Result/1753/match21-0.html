class Solution:
    def minimumEffortPath(self, h: List[List[int]]) -> int:
        
        R, C = len(h), len(h[0])
        tr, tc = R-1, C-1
        
        
        q = [(0, 0, 0, set())] # cost, i, j, v
        res = float('inf')
        
        while q:
            
            c, i, j, v = heapq.heappop(q)
            
            if c >= res:
                continue
            
            if i == tr and j == tc:
                res = min(res, c)
                continue
            
            m = (i, j)
            if m in v:
                continue
            v.add(m)
            #m = 1 << (i*C + j)
            #if m & v:
            #    continue
            #v |= m
            
            if i != tr:
                heapq.heappush(q, (max(abs(h[i+1][j] - h[i][j]), c), i+1, j, v))
            
            if j != tc:
                heapq.heappush(q, (max(abs(h[i][j+1] - h[i][j]), c), i, j+1, v))
                
            if i != 0:
                heapq.heappush(q, (max(abs(h[i-1][j] - h[i][j]), c), i-1, j, v))
            
            if j != 0:
                heapq.heappush(q, (max(abs(h[i][j-1] - h[i][j]), c), i, j-1, v))
            
        return res
