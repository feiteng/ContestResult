class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        r, c = len(heights), len(heights[0])
        heap = [(0, 0, 0)]
        visited = set()

        while heap:
            cost, i, j = heappop(heap)

            if (i, j) == (r-1, c-1):
                print(cost)
                return cost

            if (i, j) not in visited:
                visited.add((i, j))
                for a, b in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                    if 0 <= a < r and 0 <= b < c and (a, b) not in visited:
                        next_height = abs(heights[i][j] - heights[a][b])
                        heappush(heap, (max(cost, next_height), a, b))
