class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m, n = len(heights), len(heights[0])
        pq = [(0, 0, 0)]
        direct = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        dic = {(0, 0): 0}
        while pq:
            cost, r, c  = heapq.heappop(pq)
            if r == m - 1 and c == n - 1:
                return cost
            for dr, dc in direct:
                nr, nc = r + dr, c + dc
                if 0 <= nr < m and 0 <= nc < n and ((nr, nc) not in dic or dic[(nr, nc)] > abs(heights[nr][nc] - heights[r][c])):
                    dic[(nr, nc)] = abs(heights[nr][nc] - heights[r][c])
                    heapq.heappush(pq, (max(cost, abs(heights[nr][nc] - heights[r][c])), nr, nc))
