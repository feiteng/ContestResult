class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        R = len(heights)
        C = len(heights[0])
        
        def valid(effort):
            queue = deque([(0,0)])
            seen = set([(0,0)])
            
            while queue:
                r, c = queue.popleft()
                
                if r == R - 1 and c == C - 1:
                    return True
                
                for nr, nc in [(r + 1, c), (r - 1, c), (r, c + 1), (r, c - 1)]:
                    if (nr, nc) in seen:
                        continue
                    if 0 <= nr < R and 0 <= nc < C:
                        if abs(heights[nr][nc] - heights[r][c]) <= effort:
                            seen.add((nr, nc))
                            queue.append((nr, nc))
            
            return False
        
        left = 0
        right = 1000001
        
        while left < right:
            mid = (left + right) // 2
            
            if valid(mid):
                right = mid
            else:
                left = mid + 1
        
        return left
