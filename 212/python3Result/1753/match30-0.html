class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        import heapq
        m, n = len(heights), len(heights[0])
        DP = [[-1]*n for _ in range(m)]
        DP[0][0] = 0
        h = [] # effort, x, y
        heapq.heappush(h, (0, 0, 0))
        visited = set()
        while h:
            e, x, y  = heapq.heappop(h)
            if x == m - 1 and y == n - 1:
                return e
            if (x, y) in visited:
                continue
            visited.add((x, y))
            if x > 0 and (x-1, y) not in visited:
                heapq.heappush(h, (max(e, abs(heights[x][y] - heights[x-1][y])), x - 1, y))
            if x < m - 1 and (x + 1, y) not in visited:
                heapq.heappush(h, (max(e, abs(heights[x][y] - heights[x+1][y])), x + 1, y))
            if y > 0 and (x, y-1) not in visited:
                heapq.heappush(h, (max(e, abs(heights[x][y] - heights[x][y-1])), x, y - 1))
            if y < n - 1 and (x, y + 1) not in visited:
                heapq.heappush(h, (max(e, abs(heights[x][y] - heights[x][y+1])), x, y + 1))
        return -1
