class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        # dijsktras
        # priority queue with maximum hight diff so far, max(cur, new diff)
        m, n = len(heights), len(heights[0])
        q = [(0, 0, 0)]  # max height diff, r, c
        visited = set()
        while q:
            h, r, c = heapq.heappop(q)
            if r == m-1 and c == n-1:
                return h
            if (r,c) in visited:
                continue
            visited.add((r,c))
            for i, j in ((r+1, c), (r-1,c), (r,c+1), (r,c-1)):
                if 0 <= i < m and 0 <= j < n and (i,j) not in visited:
                    next_h = max(h, abs(heights[i][j] - heights[r][c]))
                    heapq.heappush(q, (next_h, i, j))
