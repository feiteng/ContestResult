class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        l = 0
        r = 10 ** 6
        n = len(heights)
        m = len(heights[0])
        
        def check(mid):
            seen = set()
            seen.add((0, 0))
            bfs = [(0, 0)]
            
            while bfs:
                nbfs = []
                for i, j in bfs:
                    if i == n - 1 and j == m - 1:
                        return True
                    for ni, nj in ((i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)):
                        if 0 <= ni < n and 0 <= nj < m and (ni, nj) not in seen and abs(heights[ni][nj] - heights[i][j]) <= mid:
                            nbfs.append([ni, nj])
                            seen.add((ni, nj))
                bfs = nbfs[:]
            
            return False
        
        while l < r:
            mid = (l + r) // 2
            if check(mid):
                r = mid
            else:
                l = mid + 1
        
        return l
