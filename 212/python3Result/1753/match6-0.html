class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        if not heights or not heights[0]:
            return 0
        n, m = len(heights), len(heights[0])
        hq = [(0, 0, 0)] # effor, x, y
        visited = set()
        while hq:
            effort, r, c = heapq.heappop(hq)
            if (r, c) == (n - 1, m - 1):
                return effort
            elif (r, c) in visited:
                continue
            visited.add((r, c))
            for row, col in [(r+1, c), (r-1, c), (r, c+1), (r, c-1)]:
                if 0 <= row < n and 0 <= col < m and (row, col) not in visited:
                    new_effort = max(abs(heights[r][c] - heights[row][col]), effort)
                    heapq.heappush(hq, (new_effort, row, col))
        return -1
