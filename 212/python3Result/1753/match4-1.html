class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        R = len(heights)
        C = len
(heights[0])
        que = [(0, 0, 0)]
        vis = set()
        while que:
            e, r, c = heapq.heappop(que)
            # print(r, c, e)
            if (r, c) == (R-1, C-1):
                
return e
            if (r, c) in vis:
                continue
            vis.add((r, c))
            
            for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                nr, nc = r + dr, c + dc
                
if nr >= 0 and nr < R and nc >= 0 and nc < C and (nr, nc) not in vis:
                    heapq.heappush(que, (max(e, abs(heights[nr][nc] - heights[r][c])), nr, nc))
                    
        return -1
