class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        rows = len(heights)
        cols = len(heights[0])
        mp = [] #mp[i][j] 表示从[i][j]点
        for _ in range(rows):
            mp.append([float('inf')]*cols)
          
        mp[0][0] = 0
        cur = [[0,0]]
        
        while len(cur)!=0:
            ncur = []
            for c in cur:
                X = c[0]
                Y = c[1]
                if X>0:
                    delta = max(mp[X][Y],abs(heights
[X][Y]-heights[X-1][Y]))
                    if delta<mp[X-1][Y]:
                        mp[X-1][Y] = delta
                        ncur.append([X-1,Y])
                if Y>0:
                    delta = max(mp[X][Y],abs(heights
[X][Y]-heights[X][Y-1]))
                    if delta<mp[X][Y-1]:
                        mp[X][Y-1] = delta
                        ncur.append([X,Y-1])
                if X<rows-1:
                    delta = max(mp[X][Y],abs(heights
[X][Y]-heights[X+1][Y]))
                    if delta<mp[X+1][Y]:
                        mp[X+1][Y] = delta
                        ncur.append([X+1,Y])
                if Y<cols-1:
                    delta = max(mp[X][Y],abs(heights
[X][Y]-heights[X][Y+1]))
                    if delta<mp[X][Y+1]:
                        mp[X][Y+1] = delta
                        ncur.append([X,Y+1])
            cur = ncur
            
        return mp[-1][-1]
