class Solution:
  def minimumEffortPath(self, M: List[List[int]]) -> int:
    frontier = [(0, 0)] 
    Y, X = len(M), len(M[0])
    E = [[float('inf')] * X for _ in range(Y)]
    E[0][0] = 0

    while frontier:
      new_frontier = []
      for y, x in frontier:
        for y_, x_ in [(y + 1, x), (y - 1, x), (y, x + 1), (y, x - 1)]:
          if y_ < 0 or x_ < 0 or y_ >= Y or x_ >= X:
            continue

          effort = max(E[y][x], abs(M[y][x] - M[y_][x_]))
          if effort < E[y_][x_]:
            E[y_][x_] = effort
            new_frontier.append((y_, x_))
      
      frontier = new_frontier
    
    return E[-1][-1]
