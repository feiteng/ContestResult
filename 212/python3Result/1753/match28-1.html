class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        
        
        m, n = len(heights), len(heights[0])
        
        arr = [[float('inf')] * n for _ in range(m)]
        
        arr[0][0] = 0
        
        # visit = {(0, 0)}
        
        visited = set()
        q = [(0, 0)]
        
        pre = {}
        
        for x, y in q:
            
            for u, v in ( (x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1) ):
                
                if 0 <= u < m and 0 <= v < n:
                    
                    temp = max( abs(heights[x][y] - heights
[u][v]), arr[x][y] )
                    
                    if temp < arr[u][v]:
                        
                        arr[u][v] = temp
                        q.append((u, v))
                        # pre[(u, v)] = (x, y)
                        
                        # visited.add((x, y, u, v))

        return arr[m - 1][n - 1]
