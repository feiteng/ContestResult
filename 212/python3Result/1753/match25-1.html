class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m,n=len(heights),len(heights
[0])
        ans=[[inf]*n for _ in range(m)]
        ans[0][0]=0
        q=deque([[0,0]])
        while q:
            i,j=q.popleft()
            for i1,j1 in [[i+1,j],[i-1,j],[i,j+1],[i,j-1]]:
                if 0<=i1<m and 0<=j1<n:
                    new
=max(ans[i][j],abs(heights[i][j]-heights[i1][j1]))
                    if new<ans[i1][j1]:
                        ans[i1][j1]=new
                        q.append([i1,j1])
        return ans[-1][-1]
