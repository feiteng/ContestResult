import heapq

class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        M,N=len(heights),len(heights[0])
        visited=set()
        # visited.add((0,0))
        cur=[(0,0,0)]
        while True:
            dist,i,j=heapq.heappop(cur)
            if i==M-1 and j==N-1:
                return dist
            else:
                if (i,j) in visited:
                    continue
                else:
                    visited.add((i,j))
                    for a,b in zip([
1,-1,0,0],[0,0,1,-1]):
