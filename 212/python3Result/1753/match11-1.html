class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        
        r = len(heights)
        c = len(heights[0])
        
        gaps = [ [float("inf")] * c for i in range(r) ]
        gaps[0][0] = 0
        ques = collections.deque([(0, 0, 0)])
        
        while ques:
            i, j, gap = ques.popleft()
            if gaps[i][j] == gap:
                for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    if 0 <= di+i < r and 0 <= dj+j < c:
                        newGap = max(gap, abs
(heights[i][j] - heights[i+di][j+dj]))
                        if gaps[di+i][dj+j] > newGap:
                            gaps[di+i][dj+j] = newGap
                            ques.append((di+i, dj+j, newGap))
        return gaps[-1][-1]
