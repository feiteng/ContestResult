class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        N,M = len(heights), len
(heights[0])
        def chk(k):
            s = set()
            s.add((0,0))
            l = [(0,0)]
            while l:
                x,y = l.pop()
                
if x == N-1 and y == M-1:
                    return True
                for u,v in [[x+1,y],[x-1,y],[x,y+1],[x,y-1]]:
                    if 0 <= u < N and 0 <= v < M and not (u,v) in s and abs(heights[x][y] - heights[u][v]) <= k:
                        l.append((u,v))
                        s.add((u,v))
            return False
        l,r = 0, 10** 6 + 1
        while l < r:
            m = (l + r) // 2
            if chk(m):
                r = m
            else:
                l = m + 1
        return l
