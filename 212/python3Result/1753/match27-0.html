class Solution:
    def minimumEffortPath(self, h: List[List[int]]) -> int:
        m,n=len(h),len(h[0])
        dis=[[inf]*n for i in range(m)]
        dis[0][0]=0
        pq=[(0,0,0)]
        while pq:
            d,x,y=heappop(pq)
            for i,j in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx,ny=x+i,y+j
                if 0<=nx<m and 0<=ny<n and max(abs(h[nx][ny]-h[x][y]),d)<dis[nx][ny]:
                    dis[nx][ny]=max(abs(h[nx][ny]-h[x][y]),d)
                    heappush(pq,(dis[nx][ny],nx,ny))
        return dis[-1][-1]
