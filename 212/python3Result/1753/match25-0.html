class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m = len(heights)
        n = len(heights[0])
        shift = [(1, 0), (0, 1), (-1, 0), (0, -1)]

        ef = [[1048576] * n for i in range(m)]
        ef[0][0] = 0
        queue = deque([(0, 0)])

        while queue:
            x, y = queue.popleft()
            for sx, sy in shift:
                i, j = x + sx, y + sy
                if i < 0 or j < 0 or i >= m or j >= n:
                    continue
                new = max(ef[x][y], abs(heights[i][j] - heights[x][y]))
                if new < ef[i][j]:
                    ef[i][j] = new
                    queue.append((i, j))

        return ef[m-1][n-1]
