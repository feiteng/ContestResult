from heapq import heappush, heappop


class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        m,n = len(heights), len(heights[0])
        q = [(0,0,0)]
        dir = [[0,-1],[0,1],[-1,0],[1,0]]
        d = [[-1] * n for _ in range(m)]
        while q:
            c, x, y = heappop(q)
            if x == m-1 and y == n-1: return c
            if d[x][y] == -1:
                d[x][y] = c
                for x0,y0 in dir:
                    x1, y1 = x+x0, y+y0
                    if 0<=x1<m and 0<=y1<n and d
[x1][y1] == -1:
                        heappush(q, (max(c, abs(heights[x1][y1] - heights[x][y])), x1, y1))
        return -1
