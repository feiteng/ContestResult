class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        rows = len(heights)
        cols = len(heights[0])
        temp = [1e6+1 for k in range(cols)]
        dis = [temp.copy() for k in range(rows)]
        
        dis[0][0] = 0
        que = [(0,0)]
        while len(que) > 0:
            rr, cc = que.pop(0)
            if cc > 0:
                nr = rr
                nc = cc - 1
                d0 = max(dis[rr][cc], abs
(heights[rr][cc] - heights[nr][nc]))
                if d0 < dis[nr][nc]:
                    dis[nr][nc] = d0
                    que.append((nr, nc))
            if cc < cols-1:
                nr = rr
                nc = cc + 1
                d0 = max(dis[rr][cc], abs
(heights[rr][cc] - heights[nr][nc]))
                if d0 < dis[nr][nc]:
                    dis[nr][nc] = d0
                    que.append((nr, nc))
            if rr > 0:
                nr = rr - 1
                nc = cc
                d0 = max(dis[rr][cc], abs
(heights[rr][cc] - heights[nr][nc]))
                if d0 < dis[nr][nc]:
                    dis[nr][nc] = d0
                    que.append((nr, nc))
            if rr < rows-1:
                nr = rr + 1
                nc = cc
                d0 = max(dis[rr][cc], abs
(heights[rr][cc] - heights[nr][nc]))
                if d0 < dis[nr][nc]:
                    dis[nr][nc] = d0
                    que.append((nr, nc))
                    
        return dis[rows-1][cols-1]
