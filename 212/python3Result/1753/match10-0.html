class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        r, c = len(heights), len(heights[0])
        dp = [[1e18 for _ in range(c)] for _ in range(r)]
        dp[0][0] = 0
        
        h = [(0, 0, 0)]
        while h:
            cur = heapq.heappop(h)
            i, j, cost = cur
            for dx, dy in [[-1, 0], [1, 0], [0, 1], [0, -1]]:
                nx, ny = i+dx, j+dy
                if nx >= 0 and nx < r and ny >= 0 and ny < c:
                    new_cost = max(cost, abs(heights[nx][ny] - heights[i][j]))
                    if new_cost < dp[nx][ny]:
                        dp[nx][ny] = new_cost
                        heapq.heappush(h, (nx, ny, new_cost))
        return dp[r-1][c-1]
