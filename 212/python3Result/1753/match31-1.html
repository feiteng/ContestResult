class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        if not heights or not heights[0]:
            return 0
        
        m, n = len(heights), len(heights[0])
        dists = {(0, 0): 0}
        q = [(0, 0, 0)]
        heapq.heapify(q)
        to_delete = set()
        print(q)
        while q:
            e, x, y = heapq.heappop(q)
            # print(q)
            if (e, x, y) in to_delete:
                continue
            if x == m-1 and y == n-1:
                return e
            
for dx, dy in [(0, -1), (0, 1), (-1, 0), (1, 0)]:
                nx, ny = x+dx, y+dy
                if 0 <= nx < m and 0 <= ny < n:
                    ne = max(e, abs(heights[nx][ny] - heights[x][y]))
                    if (nx, ny) not in dists or ne < dists[(nx, ny)]:
                        if (nx, ny) in dists:
                            to_delete.add((dists[(nx, ny)], nx, ny))                        
                        dists[(
nx, ny)] = ne
                        heapq.heappush(q, (ne, nx, ny))
        return dists[(m-1, n-1)]
