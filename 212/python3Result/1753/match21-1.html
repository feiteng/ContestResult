class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        q = [(0, 0, 0)]
        visited = set()
        m, n = len(heights), len(heights[0])
        while q:
            effort, i, j = heapq.heappop(q)
            if i == m - 1 and j == n - 1:
                return effort
            if (i, j) in visited or i >= m or j >= n:
                continue
            visited.add((i, j))
            
if i > 0:
                heapq.heappush(q, (max(effort, abs(heights[i][j] - heights[i - 1][j])), i - 1, j))
            if j > 0:
                heapq.heappush(q, (max(effort, abs(heights[i][j] - heights[i][j - 1])), i, j - 1))
            if i + 1 < m:
                heapq.heappush(q, (max(effort, abs(heights[i][j] - heights[i + 1][j])), i + 1, j))
            if j + 1 < n:
                heapq.heappush(q, (max(effort, abs(heights[i][j] - heights[i][j + 1])), i, j + 1))
        return 0
