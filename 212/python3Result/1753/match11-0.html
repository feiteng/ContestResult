class Solution:
    def minimumEffortPath(self, heights: List[List[int]]) -> int:
        n = len(heights)
        m = len(heights[0])
        f = [[1e10 for i in range(m)] for j in range(n)]
        f[0][0] = 0
        q = [(0, 0)]
        while (q):
            x, y = q.pop(0)
            for i, j in [[x + 1, y], [x - 1, y], [x, y - 1], [x, y + 1]]:
                if (0 <= i < n) and (0 <= j < m):
                    new_dist = max(f[x][y], abs(heights[i][j] - heights[x][y]))
                    if (f[i][j] > new_dist):
                        # print(i, j, new_maxn, new_minn)
                        f[i][j] = new_dist
                        q.append((i, j))
        # print(f)            
        return f[n - 1][m - 1]
