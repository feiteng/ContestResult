class Solution:
    def minimumEffortPath(self, h: List[List[int]]) -> int:
        n,m=len(h),len(h[0])
        dp=[[float('inf')]*m for _ in range(n)]


        dp[0][0]=0
        stack={(0,0)}

        while stack:
            tmp=set()
            for _ in range(len(stack)):
                (x,y)=stack.pop()


                for dx,dy in [[1,0],[0,1],[0,-1],[-1,0]]:
                    if 0<=x+dx<n and 0<=y+dy<m and dp[x+dx][y+dy]>max(dp[x][y],abs(h[x+dx][y+dy]-h[x][y])):
                        dp[x+dx][y+dy]=max(dp[x][y],abs(h[x+dx][y+dy]-h[x][y]))
                        tmp.add((x+dx,y+dy))

            stack=tmp
            
        return dp[n-1][m-1]
