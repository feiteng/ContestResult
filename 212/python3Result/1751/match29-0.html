class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        maxTime = 0
        maxChar = chr(ord('a') - 1)
        for i in range(len(keysPressed)):
            time = releaseTimes[i] - releaseTimes[i - 1] if i else releaseTimes[i]
            if time > maxTime or (time == maxTime and keysPressed[i] > maxChar):
                maxTime = time
                maxChar = keysPressed[i]
        return maxChar
