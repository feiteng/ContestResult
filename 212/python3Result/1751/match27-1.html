class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        duration = [releaseTimes[0]]
        n = len(releaseTimes)
        
for i in range(1, n):
            duration.append(releaseTimes[i] - releaseTimes[i-1])
        cnt = collections.Counter()
        for c, d in zip(keysPressed, duration):
            cnt[c] = max
(cnt[c], d)
        mv = max(cnt.values())
        return max([k for k in cnt if cnt[k] == mv])
