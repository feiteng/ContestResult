class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        maxd = releaseTimes[0]
        res = keysPressed[0]
        for i in range(1, len(keysPressed)):
            newd = releaseTimes[i] - releaseTimes[i-1]
            if newd == maxd:
                res = max(res, keysPressed[i])
            elif newd > maxd:
                res = keysPressed[i]
                maxd = newd
                
        return res
