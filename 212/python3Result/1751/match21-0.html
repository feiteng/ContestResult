class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        prev = 0
        res = ''
        t = 0
        for i,c in zip(releaseTimes,keysPressed):
            if i-prev==t:
                res = max(res,c)
            elif i-prev>t:
                res = c
                t = i-prev
            prev = i
        return res
