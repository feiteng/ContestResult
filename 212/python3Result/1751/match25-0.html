class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        releaseTimes = [0] + releaseTimes
        max_diff, letter = 0, '_'
        
        for x in range(len(releaseTimes)-1):
            diff = releaseTimes[x+1] - releaseTimes[x]
            
            if diff == max_diff:
                letter = max(letter, keysPressed[x])
            elif diff > max_diff:
                letter = keysPressed[x]
                max_diff = diff
                
        return letter
