class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        prev=0
        mintime=-math.inf
        res
=None
        for r,k in zip(releaseTimes,keysPressed):
            cur=r-prev
            if cur>mintime:
                mintime=cur
                res=k
            elif cur==mintime:
                if res==None or k>res:
                    res=k
            prev=r
        return res
