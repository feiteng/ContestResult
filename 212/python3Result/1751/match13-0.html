class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        m = 0
        pv = 0
        x = 'a'
        for k,v in zip(keysPressed, releaseTimes):
            d = v-pv
            if d > m:
                m = d
                x = k
            elif d == m:
                x = max(k, x)
            pv = v
        return x
