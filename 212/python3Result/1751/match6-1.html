class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        prev=0
        ret=None
        best=-inf
        for i in range(len(releaseTimes)):
            me=releaseTimes[i]-prev
            if me>best or (me==best and keysPressed[i]>ret):
                best=me
                ret=keysPressed[i]
            prev=releaseTimes[i]
        return ret
