class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        
        
        durations = collections.defaultdict(int)
        pre = 0
        for i in range(len(releaseTimes)):
            durations[keysPressed[i]] = max(durations[keysPressed[i]], releaseTimes[i] - pre)
            pre = releaseTimes[i]
            
        maxV = max(durations.values())
        possible = [key for key in durations if durations[key] == maxV]
        possible.sort()
        return possible.pop()
