class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        md = 0
        ans = 'a'
        
        for i in range(len(keysPressed)):
            d = releaseTimes[i] - (0 if i == 0 else releaseTimes[i-1])
            if d > md or (d == md and keysPressed[i] > ans):
                ans = keysPressed[i]
                md = d
        return ans
