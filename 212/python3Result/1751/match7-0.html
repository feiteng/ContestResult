class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        ti = 0
        ans = ''
        for i in range(len(releaseTimes)):
            if i == 0:
                temp = releaseTimes[0]
            else:
                temp = releaseTimes[i] - releaseTimes[i-1]
            if temp > ti:
                ti = temp
                ans = keysPressed[i]
            elif temp == ti:
                ans = max(ans, keysPressed[i])
        return ans
