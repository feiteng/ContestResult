class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        pre = 0
        res = 0
        s = ''
        for x, time in zip(keysPressed, releaseTimes):
            t = time - pre
            if t > res:
                res = t
                s = x
            elif t == res:
                s = max(x, s)
            pre = time
                
        return s
