class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        prev = 0
        long = 0
        longCh = None
        for i in range(len(releaseTimes)):
            l = releaseTimes[i]-prev
            if l > long or (l==long and keysPressed[i]>longCh):
                long = l
                longCh = keysPressed[i]
            
            prev = releaseTimes[i]
        return longCh
