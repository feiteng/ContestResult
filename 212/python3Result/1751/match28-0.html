class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        press = 0
        longestTime = 0
        longestChar = ''
        for c, time in zip(keysPressed, releaseTimes):
            length = time - press
            if length > longestTime:
                longestTime = length
                longestChar = c
            elif length == longestTime:
                if c > longestChar:
                    longestChar = c
            press = time
        return longestChar
