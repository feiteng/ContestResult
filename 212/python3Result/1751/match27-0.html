class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        n = len(releaseTimes)
        times = [releaseTimes[0]]
        for i in range(n-1):
            times.append(releaseTimes[i+1] - releaseTimes[i])
        count = collections.defaultdict(list)
        for i in range(n):
            count[times[i]].append(keysPressed[i])
        shortest = max(count.keys())
        return max(count[shortest])
