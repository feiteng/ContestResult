class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        key = keysPressed[0]
        dur = releaseTimes[0]
        for i in range(1, len(releaseTimes)):
            d = releaseTimes[i] - releaseTimes[i - 1]
            if d > dur:
                key = keysPressed[i]
                dur = d
            elif d == dur and keysPressed[i] > key:
                key = keysPressed[i]
        return key
