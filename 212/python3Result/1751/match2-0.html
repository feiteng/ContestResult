class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        m = releaseTimes[0]
        s = keysPressed[0]
        for i in range(1, len(releaseTimes)):
            delta = releaseTimes[i] - releaseTimes[i - 1]
            if delta > m or (delta == m and keysPressed[i] > s):
                m = delta
                s = keysPressed[i]
        return s
