class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        t = [0]+ releaseTimes
        ans = 0
        ret = ''
        for i in range(len(t)-1):
            if t[i+1]-t[i]>ans:
                ans = t[i+1]-t[i]
                ret = keysPressed[i]
            elif t[i+1]-t[i]==ans:
                if keysPressed[i]> ret:
                    ret = keysPressed[i]
                    
        return ret
