class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        ct = 0
        cc = ''
        st = 0
        for i in range(len(keysPressed)):
            tt = releaseTimes[i] - st
            if tt > ct or (tt == ct and keysPressed[i] > cc):
                ct = tt
                cc = keysPressed[i]
            st = releaseTimes[i]
        return cc
