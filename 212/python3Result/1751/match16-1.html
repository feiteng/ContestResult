class Solution:
    def slowestKey(self, r: List[int], k: str) -> str:
        ma=r[0]
        ans=k[0]
        
        for i in range(1,len(r)):
            if r[i]-r[i-1]>ma:
                ma=r[i]-r[i-1]
                ans=k
[i]
            elif r[i]-r[i-1]==ma:
                if k[i]>ans:
                    ans=k[i]
        return ans
