class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        
        ans = []
        maxduration = -1
        for i in range(0, len(releaseTimes)):
            duration = releaseTimes[i] - (releaseTimes[i - 1] if i >= 1 else 0)
            if duration > maxduration:
                ans = [keysPressed[i]]
                maxduration = duration
            elif duration == maxduration:
                ans.append(keysPressed[i])
        
        return max(ans)
