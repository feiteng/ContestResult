class Solution:
    def slowestKey(self, r: List[int], k: str) -> str:
        time = 0
        ans = 'a'
        r = [0] + r
        m = {x:0 for x in set(k)}
        
for i in range(len(k)):
            if r[i+1] - r[i] > time:
                ans = k[i]
                time = r[i+1] - r[i]
            elif r[i+1] - r[i] == time:
                if k[i] > ans:
                    ans = k[i]
