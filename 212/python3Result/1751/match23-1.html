class Solution:
    def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        ret = keysPressed[0]
        best = releaseTimes[0]
        
        for i in range(len(releaseTimes)):
            this = releaseTimes[i] - releaseTimes[i - 1]
            
            if this > best or (this == best and ret < keysPressed[i]):
                ret, best = keysPressed[i], this
                
        return ret
