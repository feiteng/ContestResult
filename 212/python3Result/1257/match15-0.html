from collections import defaultdict
class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        n = len(matrix)
        m = len(matrix[0])
        ranks = [[0] * m for _ in range(n)]
        idx_by_val = defaultdict(list)
        all_values = []
        for i in range(n):
            for j in range(m):
                val = matrix[i][j]
                idx_by_val[val].append((i, j))
                all_values.append(val)
                
        max_by_row = [0] * n
        max_by_col = [0] * m
        all_values = sorted(list(set(all_values)))
        for val in all_values:
            changed = True
            while changed:
                changed = False
                for i, j in idx_by_val[val]:
                    r = max(max_by_row[i], max_by_col[j])
                    if r > max_by_row[i] or r > max_by_col[j]:
                        max_by_row[i] = r
                        max_by_col[j] = r
                        changed = True
                
            for i, j in idx_by_val[val]:
                ranks[i][j] = max(max_by_row[i], max_by_col[j]) + 1
            for i, j in idx_by_val[val]:
                max_by_row[i] = max_by_col[j] = ranks[i][j]
        
        return ranks
