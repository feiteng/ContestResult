from heapq import heappop, heappush
from collections import defaultdict

class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        def dfs(i, j):
            nonlocal k
            st.add((i,j))
            ans.append((i,j))
            k = max(k, max(r[i], c
[j]) + 1)
            for j1 in mpi[i]:
                if (i,j1) not in st:
                    dfs(i, j1)
            for i1 in mpj[j]:
                if (i1,j) not in st:
                    dfs(i1, j)
        m,n = len(matrix), len(matrix[0])
        q = []
        for i in range(m):
            for j in range(n):
                heappush(q, (matrix[i][j],i,j))
        r, c = [0] * m, [0] * n
        res = [[0] * n for _ in range(m)]
        while q:
            v = q[0][0]
            ls = []
            mpi = defaultdict(list)
            mpj = defaultdict(list)
            while q and q[0][0] == v:
                v,i,j = heappop(q)
                ls.append((i,j))
                mpi[i].append(j)
                mpj[j].append(i)
            st = set()
            for i,j in ls:
                if (i,j) not in st:
                    k = 0
                    ans = []
                    dfs(i,j)
                    for x,y in ans:
                        res[x][y] = k
                        r[x] = k
                        c[y] = k
        return res
