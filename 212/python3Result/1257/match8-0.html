class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        m, n = len(matrix), len(matrix[0])
        
        visited = collections.defaultdict(lambda: False)
        groups = []
        
        rows = [collections.defaultdict(list) for i in range(m)]
        cols = [collections.defaultdict(list) for i in range(n)]
        for i in range(m):
            for j in range(n):
                rows[i][matrix[i][j]].append(j)
                cols[j][matrix[i][j]].append(i)
        
        def dfs(i, j):
            grp.append((i, j))
            visited[(i, j)] = True
            
            for z in cols[j][matrix[i][j]]:
                if z != i and not visited[(z, j)]:
                    dfs(z, j)
                
            for z in rows[i][matrix[i][j]]:
                if z != j and not visited[(i, z)]:
                    dfs(i, z)
            
        
        for i in range(m):
            for j in range(n):
                if not visited[(i, j)]:
                    grp = []
                    dfs(i, j)
                    groups.append((matrix[i][j], grp))
            
        groups.sort()
        rows = [0] * m
        cols = [0] * n
        res = [[0] * n for i in range(m)]
        
        for v, grp in groups:
            rank = 0
            for i, j in grp:
                rank = max(rank, rows[i]+1, cols[j]+1)
                
            for i, j in grp:
                res[i][j] = rank
                rows[i] = rank
                cols[j] = rank
                
        return res
