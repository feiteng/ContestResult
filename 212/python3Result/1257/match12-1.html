class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        heap = []
        n = len(matrix)
        m = len(matrix[0])
        res = [[0] * m for _ in range(n)]
        row 
= [0] * n
        col = [0] * m
        
        for i in range(n):
            for j in range(m):
                heapq.heappush(heap, (matrix[i][j], i, j))
        
        while heap:
            cur = heap[0][0]
            curl = []
            
            while heap and heap[0][0] == cur:
                curl.append((heap[0][1], heap[0][2]))
                heapq.heappop(heap)
            
            f = {}
            mm = collections.defaultdict(int)
            def find(a):
                if a not in f:
                    f[a] = a
                    r
eturn a
                if f[a] != a:
                    f[a] = find(f[a])
                return f[a]
            def merge(a, b):
                fa = find(a)
                fb = find(b)
                f[fa] = fb
                mm[a] = max(max(row[a], col[b - n]) + 1, mm[a])
                mm[b] = max(max(row[a], col[b - n]) + 1, mm[b])
                mm[a] = max(mm[a], mm[b])
                mm[b] = max(mm[a], mm[b])
                mm[fb] = max(mm[fb], mm[a], mm[fa])
                mm[fa] = max(mm[fa], mm[a], mm[fb])
            
            for i, j in curl:
                merge(i, j + n)
            
            
            for i, j in curl:
                res[i][j] = max(mm[find(i)], mm[find(j + n)])
                row[i] = max(res[i][j], row[i])
                col[j] = max(res[i][j], col[j])
            

        
        return res
