class UF:
    def __init__(self, n):
        self.n = n
        self.data = list(range(n))
        
    def union(self, i, j):
        pi, pj = self.find(i), self.find(j)
        if pi != pj:
            self.data[pi] = self.data[p
j]

    def find(self, i):
        if self.data[i] != i:
            self.data[i] = self.find(self.data[i])
        return self.data[i]
    
    def reset(self):
        self.data = list(range(self.
n))
