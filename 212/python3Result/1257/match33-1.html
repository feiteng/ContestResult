class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        row, col = len(matrix), len(matrix[0])
        num2pos = {}
        for i in range(row):
            for j in range(col):
                val = matrix[i][j]
                if val not in num2pos:
                    num2pos[val] = set()
                num2pos[val].add((i, j))
        row_min = [1] * row
        col_min = [1] * col
        res = [[0] * col for _ in range(row)]
        for k in sorted(num2pos.keys()):
            points = num2pos[k]
            row_change = set()
            col_change = set()
            while True:
                flag = 0
                for i, j in points:
                    res[i][j] = max(row_min[i], col_min[j])
                    if row_min[i] != res[i][j] or col_min[j] != res[i][j]:
                        row_min[i], col_min[j] = res[i][j], res[i][j]
                        flag = 1
                    row_change.add(i)
                    col_change.add(j)
                if flag == 0:
                    break
            for i in row_change:
                row_min[i] += 1
            for j in col_change:
                col_min[j] += 1
        return res
