class Solution:
    def matrixRankTransform(self, mat: List[List[int]]) -> List[List[int]]:
        h,w=len(mat),len(mat[0])
        MIN=float("-inf")
        rmx=[(MIN, 0)]*h
        cmx=[(MIN, 0)]*w
        arr=[]
        
for y in range(h):
            for x in range(w):
                arr.append([mat[y][x],y,x])
        arr.sort()
        
        ans=[[0]*w for _ in range(h)]
        for i, (v,y,x) in enumerate(arr):
            if v==rmx[y][0]:
                cur=rmx[y][1]
            else:
                cur=rmx
[y][1]+1
            if v==cmx[x][0]:
                cur=max(cmx[x][1],cur)
            else:
                cur=max(cmx[x][1]+1,cur)
            ans[y][x]=cur
            rmx[y]=(v,cur)
            cmx[x]=(v,cur)
            if i+1==len(arr) or arr[i+1][0]!=v:
                beg=i
                while beg>0 and arr[beg-1][0]==v:
                    beg-=1
                updated = True
                while updated:
                    updated = False
                    for j in range(beg,i+1):
                        v1,y1,x1=arr[j]
                        
                        if v1==rmx[y1][0]:
                            cur=rmx[y1][1]
                        else:
                            cur=rmx[y1][1]+1
                        if v==cmx[x1][0]:
                            cur=max(cmx[x1][1],cur)
                        else:
                            cur=max(cmx[x1][1]+1,cur)
#                         if j != i:
#                             print("v1,y1,x1:%d,%d,%d,%s,%s,cur:%d"%(v1,y1,x1, rmx[y1],cmx[x1],cur))
                        
                        if cur > ans[y1][x1]:
                            ans[y1][x1]=cur
                            rmx[y1]=(v,cur)
                            cmx[x1]=(v,cur)
                            updated = True
        return ans
