class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        m, n = len(matrix), len(matrix[0])
        answer = [[0 for j in range(n)] for i in range(m)]
        rows = [0 for i in range(m)]
        cols 
= [0 for j in range(n)]
        x = []
        for i in range(m):
            for j in range(n):
                x.append((matrix[i][j], i, j))
        x.sort()
        rowdict = [{} for i in range(m)]
        for i in range(m):
            for j in range(n):
                if matrix[i][j] in rowdict[i]:
                    rowdict[i][matrix[i][j]].append(j)
                else:
                    rowdict[i][matrix[i][j]] = [j]
        coldict = [{} for j in range(n)]
        for j in range(n):
            for i in range(m):
                if matrix[i][j] in coldict[j]:
                    coldict[j][matrix[i][j]].append(i)
                else:
                    coldict[j][matrix[i][j]] = [i]
                    
                    
        for val, i, j in x:
            if answer[i][j]:
                continue
            a, b = {i}, {j}
            cands = [(i, j)]
            have = {(i, j)}
            while cands:
                candsnext = []
                for i, j in cands:
                    for c in coldict[j][val]:
                        if (c, j) in have:
                            continue
                        have.add((c, j))
                        candsnext.append((c, j))
                        a.add(c)
                    for d in rowdict[i][val]:
                        if (i, d) in have:
                            continue
                        have.add((i, d))
                        candsnext.append((i, d))
                        b.add(d)
                cands = candsnext
            e = 0
            for i in a:
                e = max(e, rows[i])
            for j in b:
                e = max(e, cols[j])
            e += 1
            
for i in a:
                rows[i] = e
            for j in b:
                cols[j] = e
            for i, j in have:
                answer[i][j] = e
            
            
        return answer
