class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        from collections import defaultdict as dc
        n = len(matrix)
        m = len(matrix[0])
        dic 
= dc(list)
        for i in range(n):
            for j in range(m):
                dic[matrix[i][j]].append((i, j))
        row = [1]*n
        line = [1]*m
        ans = [[0]*m for _ in range(n)]

        def sol(g):
            val = 1
            for q in g:
                if row[q[0]] > val:
                    val = row[q[0]]
                if line[q[1]] > val:
                    val = line[q[1]]
            for q in g:
                ans[q[0]][q[1]] = val
                row[q[0]] = val+1
                line[q[1]] = val+1

        def solve(keys):
            dr = dc(list)
            dl = dc(list)
            vis = [0]*len(keys)
            for q in range(len(keys)):
                dr[keys[q][0]].append(q)
                dl[keys[q][1]].append(q)
            for q in range(len(keys)):
                if not vis[q]:
                    vis[q] = 1
                    cg = [q]
                    st = [keys[q]]
                    while st:
                        cx, cy = st.pop()
                        for p in dr[cx]:
                            if not vis[p]:
                                cg.append(p)
                                vis[p] = 1
                                st
.append(keys[p])
                        for p in dl[cy]:
                            if not vis[p]:
                                cg.append(p)
                                vis[p] = 1
                                st.append(keys[p])
                    sol([keys[q]for q in cg])

        for i in sorted(dic):
            solve(dic[i])
        return ans
