class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        from math import inf
        from collections import defaultdict
        
        def dfs(x, y):
            cur.add((x, y))
            for y2 in dr[x]:
                if (x, y2) not in cur:
                    dfs(x, y2)
            for x2 in dc[y]:
                if (x2, y) not in cur:
                    dfs(x2, y)
        
        m, n, i = len(matrix), len(matrix[0]), 0
        ans, pool, bound = [[None] * n for i in range(m)], sorted([[matrix[i][j], i, j] for i in range(m) for j in range(n)]), m * n
        row, col = [0] * m, [0] * n
        while i < bound:
            j = i + 1
            while j < bound and pool[j][0] == pool[i][0]:
                j += 1
            dr, dc, visited = defaultdict(list), defaultdict(list), set()
            for k in range(i, j):
                dr[pool[k][1]].append(pool[k][2])
                dc[pool[k][2]].append(pool[k][1])
            for k in range(i, j):
                if (pool[k][1], pool[k][2]) not in visited:
                    cur = set()
                    dfs(pool[k][1], pool[k][2])
                    value = max(max(row[x], col[y]) for x, y in cur) + 1
                    for x, y in cur:
                        ans[x][y] = row[x] = col[y] = value
                    visited |= cur
            i = j
        return ans
