class Solution:
    def matrixRankTransform(self, mat: List[List[int]]) -> List[List[int]]:
        n = len(mat)
        m = len(mat[0])
        arr = [[mat[i][j], i, j] for i in range(n) for j in range(m)]
        ans = [[0 for i in range(m)] for j in range(n)]
        arr.sort(key = lambda x:x[0])
        mi = [0 for i in range(n)]
        mj = [0 for i in range(m)]
        fa = [i for i in range(n*m)]
        
        def getfa(x):
            if not fa[x]==x:
                fa[x] = getfa(fa[x])
            return fa[x]

        def merge(a, b):
            a = getfa(a)
            b = getfa(b)
            fa[a] = b
            
        for i in range(n):
            tmp = sorted([[mat[i][j], j] for j in range(m)])
            for j in range(len
(tmp)-1):
                if tmp[j][0]==tmp[j+1][0]:
                    merge(m*i+tmp[j][1], m*i+tmp[j+1][1])
        for i in range(m):
            tmp = sorted([[mat[j][i], j] for j in range(n)])
            for j in range(len
(tmp)-1):
                if tmp[j][0]==tmp[j+1][0]:
                    merge(m*tmp[j][1]+i, m*tmp[j+1][1]+i)
        dic = dict()
        for i in range(n*m):
            t = getfa(i)
            if t in dic.keys():
                dic[t].append(i)
            else:
                dic[t] = [i]
        
        for t in arr:
            v, x, y = t
            if ans[x][y]:
                continue
            tmp = dic[fa[x*m+y]]
            mx = 0
            for tt in tmp:
                mx = max(mx, mi[tt//m], mj[tt%m])
            for tt in tmp:
                ans[tt//m][tt%m] = mx+1
                mi[tt//m] = mx+1
                mj[tt%m] = mx+1
        return ans
