class Solution:
    def matrixRankTransform(self, matrix: List[List[int]]) -> List[List[int]]:
        m,n=len(matrix),len(matrix[0])
        d=defaultdict(set)
        #row=defaultdict(set)
        rf=[defaultdict(set) for _ in range(m)]
        cf=[defaultdict(set) for _ in range(n)]
        #col=defaultdict(set)
        ans=[[0]*n for i in range(m)]
        for i in range(m):
            for j in range(n):
                d[matrix[i][j]].add((i,j))
                #row[matrix[i][j]].add(i)
                rf[i][matrix[i][j]].add(j)
                #col[matrix[i][j]].add(j)
                cf[j][matrix[i][j]].add(i)
        r=[1 for _ in range(m)]
        c=[1 for _ in range(n)]
        #print(row,col)
        res={}
        def dfs(i,j,val):
            seen.add((i,j))
            for nj in rf[i][val]:
                if (i,nj) not in seen:
                    dfs(i,nj,val)
            for ni in cf[j][val]:
                if (ni,j) not in seen:
                    dfs(ni,j,val)
        for k in sorted(d.keys()):
            cur=1
            tmp=d[k]
            while tmp:
                seen=set()
                i,j=tmp.pop()
                dfs(i,j,matrix[i][j])
                tmp-=seen
                cur=1
                for i,j in seen:
                    cur=max(cur,r[i],c[j])
                for i,j in seen:
                    r[i]=cur+1
                    c[j]=cur+1
                    ans[i][j]=cur

        return ans
