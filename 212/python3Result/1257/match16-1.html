from collections import *

class Solution:
    def matrixRankTransform(self, m: List[List[int]]) -> List[List[int]]:
        def find_f(s):
            if fa[s] == -1:
                return s
            fa[s] = find_f(fa[s])
            return fa[s]
        
        R, C = len(m), len(m[0])
        ret = [[1] * C for _ in range(R)]
        arr 
= []
        for i in range(R):
            for j in range(C):
                arr.append((m[i][j], i, j))
        arr.sort()
        col = [0] * (C + R)
        row = [0] * (R + C)
        st, en = 0, 0
        while en < len(arr):
            st = en
            tc = [col[i] for i in range(C)]
            tr = [row[i] for i in range(R)]
            fa 
= [-1] * (R + C)
            while en < len(arr) and arr[en][0] == arr[st][0]:
                r, c = arr[en][1], arr[en][2]
                row[r] = max(row[r], tr[r] + 1)
                col[c] = max(col[c], tc[c] + 1)
                fr = find_f(r)
                fc = find_f(R + c)
                if fr != fc:
                    fa[fc] = fr
                en += 1
            for i in range(st, en):
                r, c = arr[i][1], arr[i][2]
                fc = find_f(R + c)
                row[fc] = max(row[fc], max(row[r], col[c]))
            for i in range(st, en):
                r, c = arr[i][1], arr[i][2]
                fc = find_f(R + c)
                row[r] = col[c] =row[fc]
                ret[r][c] = row[fc]
     
        return ret
