class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        result = []
        for i in range(len(l)):
            subarray = sorted(nums[l[i]:r[i]+1])
            if len(subarray) <= 1:
                result.append(True)
            else:
                const = subarray[1] - subarray[0]
                isArith = True
                for j in range(2,len(subarray)):
                    if subarray[j] - subarray[j-1] != const:
                        isArith = False
                        break
                result.append(isArith)
        return result
