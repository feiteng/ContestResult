class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def isvalid(cl, cr):
            cnums = sorted(nums[cl: cr + 1])
            delta = cnums[1] - cnums[0]
            for i in range(1, cr - cl):
                cd = cnums[i + 1] - cnums[i]
                if cd != delta:
                    
return False
            return True
        ret = []
        for cl, cr in zip(l, r):
            ret.append(isvalid(cl, cr))
        return ret
