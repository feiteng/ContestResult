class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def check(s, e):
            if e-s < 1:
                return False
            cur = sorted(nums[s:e+1])
            delta = cur[1]-cur[0]
            for i in range(2, len(cur)):
                if cur[i]-cur[i-1] != delta:
                    return False
            return True

        ans = []
        for s, e in zip(l, r):
            ans.append(check(s, e))
        return ans
