class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        answer = []
        for p, q in zip(l, r):
            cut = sorted(nums[p: q+1])
            if len(cut) < 2:
                answer.append(False)
                continue

            diff 
= cut[1] - cut[0]
            flag = True
            for i in range(2, len(cut)):
                if cut[i] - cut[i-1] != diff:
                    flag = False
                    break
            answer.append(flag)

        return answer
