class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for i in range(len(l)):
            curr = nums[l[i]:r[i]+1]
            curr.sort()
            diff = curr[1]-curr[0]
            isArith = True
            for i in range(2,len(curr)):
                if curr[i] - curr[i-1] != diff:
                    isArith = False
                    break
            if isArith == False:
                res.append(False)
            else:
                res.append(True)
        return res
