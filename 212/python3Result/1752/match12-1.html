class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def check(a):
            if len(a)<2:
                return True
            d=a[1]-a[0]
            for m in range(1,len(a)):
                if a[m]-a[m-1]!=d:
                    return False
            return True
        res=[]
        for m,i in enumerate(l):
            curr
=nums[l[m]:r[m]+1]
            curr.sort()
            res.append(check(curr))
        return res
