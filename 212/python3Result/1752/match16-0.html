class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        out = []
        for i in range(len(l)):
            lower, upper = l[i], r[i]
            cand = sorted(nums[lower:upper+1])
            if len(cand) < 2:
                out.append(False)
                continue
                
            diff = cand[1] - cand[0]
            is_arith = True
            for j in range(2, len(cand)):
                if cand[j] - cand[j-1] != diff:
                    is_arith = False
                    break
            out.append(is_arith)        
        return out
