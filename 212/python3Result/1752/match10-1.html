class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for a, b in zip(l, r):
            t = sorted(nums[a: b + 1])
            if len(t) <= 1:
                res.append(True)
                continue
            s = t[1] - t[0]
            f = True
            for i in range(2, len(t)):
                if t[i] - t[i - 1] != s:
                    f = False
                    break
            res.append(f)
        return res
