class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def OK(i,j):
            if i==j: return True
            L = sorted(nums[i:j+1])
            diff = L[1]-L[0]
            for i in range(2, len(L)):
                if diff!=L[i]-L[i-1]: return False
            return True
        
        return [OK(i,j) for i,j in zip(l,r)]
