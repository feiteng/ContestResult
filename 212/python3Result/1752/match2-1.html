class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def chk(u,v):
            A = sorted(nums[u:v+1])
            t = A[1] - A[0]
            for i in range(len(A) - 1):
                if A[i+1] - A[i] != t:
                    return False
            return True
        ret = []
        for u,v in zip(l,r):
            ret.append(chk(u,v))
        return ret
