class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for left, right in zip(l, r):
            temp = sorted(nums[left:right + 1])
            length = len(temp)
            if length == 1:
                res
.append(True)
            else:
                dx = temp[1] - temp[0]
                for i in range(2, length):
                    if temp[i] - temp[i - 1] != dx:
                        res.append(False)
                        break
                else:
                    res.append(True)
                    
        return res
