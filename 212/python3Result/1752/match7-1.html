class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def isar(i, j):
            if i == j:
                return True

            l = sorted(nums[i: j + 1])
            diff = l[1] - l[0]
            for i in range(2, len(l)):
                if l[i] - l[i-1] != diff:
                    return False

            return True

        res = []
        for i, j in zip(l, r):
            res.append(isar(i, j))

        return res
