class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        
        for li, ri in zip(l, r):
            ll = nums[li:ri+1]
            ll.sort()
            
            if len(ll)==1:
                ans.append(True)
                continue
            
            d = ll[1]-ll[0]
            
            for i in range(len(ll)-1):
                if ll[i+1]-ll[i]!=d:
                    ans.append(False)
                    break
            else:
                ans.append(True)
        
        return ans
