class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for left, right in zip(l, r):
            res.append(self.check(nums[left: right + 1]))
        
        return res
        
    def check(self, nums):
        if len(nums) <= 2:
            return True
        
        nums.sort()
        diff = nums[1] - nums[0]
        for i in range(2, len(nums)):
            if nums[i] - nums[i - 1] != diff:
                return False
        
        return True
