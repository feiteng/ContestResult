from typing import List, Dict, Tuple


class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []

        for i in range(len(l)):
            x = nums[l[i]:r[i] + 1]
            x.sort()
            if len(x) == 1:
                ans.append(True)
                continue
            sa 
= x[1] - x[0]
            f = True
            for j in range(len(x) - 1):
                if sa != x[j + 1] - x[j]:
                    f = False
                    break
            if f:
                ans.append(True)
            else:
                ans.append(False)



        return ans
