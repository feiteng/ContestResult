class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        
        
        def t(l,r):
            
            tmp=sorted(nums[l:r+1])
            dif=tmp[1]-tmp[0]
            s=tmp[1]
            for i in tmp[2:]:
                s+=dif
                if s!=i:
                    return False
            return True
        
        
        return  [t(l1,r1) for l1,r1 in zip(l,r)]
