class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for a, b in zip(l, r):
            seg = nums[a:b+1]
            seg.sort()
            for i in range(1, len(seg)):
                if (seg[i] - seg[i-1] != seg[1] - seg[0]):
                    res.append(False)
                    break
            else:
                res.append(True)
        return res
