class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def check(l,r):
            if l==r:
                return True
            lst = sorted(nums[l:r+1])
            diff=lst[1]-lst[0]
            for i in range(1,len(lst)):
                if lst[i]-lst[i-1]!=diff:
                    return False
            return True
        return [check(le,re) for le,re in zip(l,r)]
