class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        result = []
        for i in range(len(l)):
            valid = True
            num = sorted(nums[l[i]:r[i]+1])
            for j in range(1, len(num)):
                if num[j]-num[j-1]!=num[1]-num[0]:
                    valid = False
                    break
            result.append(valid)
        return result
