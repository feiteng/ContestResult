class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def check(left, right):
            if right - left <= 2:
                return True
            a = sorted(nums[left:right])
            n = len(a)
            for i in range(2, n):
                if a[i] - a[i - 1] != a
[1] - a[0]:
                    return False
            return True
        
        ret = []
        
        for left, right in zip(l, r):
            ret.append(check(left, right + 1))
            
        return ret
