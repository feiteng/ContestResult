class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        answer = []
        for x, y in zip(l, r):
            a = nums[x:y+1]
            a.sort()
            d 
= a[1] - a[0]
            for i in range(1, len(a)-1):
                if a[i+1]-a[i] != d:
                    answer.append(False)
                    break
            else:
                answer.append(True)
        return answer
