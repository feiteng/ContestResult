class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def checkArithmetic(left, right):
            tmp = nums[left: right+1]
            tmp.sort()
            d = tmp[1] - tmp[0]
            for i in range
(2, right+1-left):
                if tmp[i] - tmp[i-1] != d:
                    return False
            return True
        
        ans = []
        
        for i in range(len(l)):
            ans.append(checkArithmetic(l[i], r[i]))
        return ans
