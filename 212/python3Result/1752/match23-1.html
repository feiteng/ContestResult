class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def check(aa):
            aa.sort()
            d=aa[1]-aa[0]
            for a0,a1 in zip(aa,aa[1:]):
                if a1-a0!=d:
return False
            return True

        ans=[]
        for li,ri in zip(l,r):
            ans.append(check(nums[li:ri+1]))
        return ans
