class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans=[]
        for i in range(len(l)):
            li,ri=l[i],r[i]
            cur=sorted(nums[li:ri+1])
            
if all(cur[j]-cur[j-1]==cur[1]-cur[0] for j in range(1,len(cur))):
                ans.append(True)
            else:
                ans.append(False)
        return ans
