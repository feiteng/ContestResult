class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def helper(left,right):
            arr = sorted(nums[left:right+1])
            d = arr[1]-arr[0]
            for a,b in zip(arr[2:],arr[1:]):
                
if a-b!=d:
                    return False
            return True
        return [helper(left,right) for left,right in zip(l,r)]
