class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        
        res = []
        
        for i in range(len(l)):
            lw, up = l[i], r[i]
            if up - lw + 1 < 2: 
                res.append(False)
                continue
            if up - lw + 1 == 2:
                res.append(True)
                continue                
            temp 
= nums[lw:(up+1)]
            temp.sort()
            res.append(self.check(temp))
            
            
        return res
            
        
                
    def check(self, ls):
        # len(ls) >= 3
        diff 
= ls[1] - ls[0]
        for i in range(2, len(ls)):
            if ls[i] - ls[i-1] != diff: return False
        return True
