class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ret = []
        for i in range(len(l)):
            begin = l[i]
            end = r[i]
            if end-begin<1:
                ret.append(False)
                continue
            tmp = []
            for j in range(begin,end+1):
                tmp.append(nums[j])
            tmp
.sort()
            cha = tmp[1]-tmp[0]
            isDC = True
            for k in range(1,len(tmp)-1):
                if tmp[k+1]-tmp[k]!=cha:
                    isDC = False
                    break
            ret.append(isDC)
            
        return ret
