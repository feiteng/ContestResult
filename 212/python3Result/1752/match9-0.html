class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        output = []
        for i in range(len(l)):
            seq = nums[(l[i]):(r[i] + 1)]
            seq.sort()
            isArith = True
            for j in range(2, len(seq)):
                if seq[j] - seq[j - 1] != seq[1] - seq[0]:
                    isArith = False
                    break
                    
            output.append(isArith)
            
        return output
