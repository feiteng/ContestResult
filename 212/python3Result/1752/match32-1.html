class Solution:
  
  def is_arithmetic(self, nums):
    nums
.sort()
    d = nums[1] - nums[0]
    for i in range(1, len(nums)):
      if nums[i] - nums[i - 1] != d:
        return False
    return True
  
  
def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
    return [self.is_arithmetic(nums[v1:v2 + 1]) for v1, v2 in zip(l, r)]
