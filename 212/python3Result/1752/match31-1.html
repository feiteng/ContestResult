class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ret = []
        for i in range(len(l)):
            tmp = nums[l[i] : (r[i] + 1)]
            tmp.sort()
            if len(tmp) <= 1:
                ret.append(False)
                continue
            d 
= tmp[1] - tmp[0]
            f = True
            for i in range(2, len(tmp)):
                if tmp[i] - tmp[i - 1] != d:
                    f = False
                    break
            ret.append(f)
        return ret
