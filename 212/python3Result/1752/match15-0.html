class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        for l0, r0 in zip(l, r):
            sortedNums = sorted(nums[l0:r0+1])
            length = (r0 - l0 + 1)
            if (sortedNums[-1] - sortedNums[0])%(length - 1) != 0:
                ans.append(False)
            else:
                delta = (sortedNums[-1] - sortedNums[0])//(length - 1) 
                for i in range(1, length):
                    if sortedNums[i] != sortedNums[i-1] + delta:
                        ans.append(False)
                        break
                else:
                    ans.append(True)
        return ans
