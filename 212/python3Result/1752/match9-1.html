class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        for i in range(len(l)):
            v = nums[l[i]:r[i] + 1]
            v.sort()
            ok = True
            for i in range(1, len(v)):
                if v[i] - v[i - 1] != v[1] - v[0]:
                    ok = False
                    break
            ans.append(ok)
        return ans
