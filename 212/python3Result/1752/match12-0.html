class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def test(x):
            if len(x) <= 2:
                return True
            
            d = x[1] - x[0]
            
            for i in range(2, len(x)):
                if x[i] - x[i-1] != d:
                    return False
            
            return True
        
        res = []
        for i in range(len(l)):
            x = nums[l[i]:r[i]+1]
            x.sort()
            # print(x)
            res.append(test(x))
        
        return res
