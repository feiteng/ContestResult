class Solution:
  def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
    return [self.check(nums[start:end+1]) for start, end in zip(l, r)]
    
  def check(self, nums):
    if len(nums) < 2:
      return False
    if len(nums) == 2:
      return True
  
    nums.sort()
    dis = nums[1] - nums[0]
    for i in range(1, len(nums)):
      if nums[i] - nums[i-1] != dis:
        return False
    return True
