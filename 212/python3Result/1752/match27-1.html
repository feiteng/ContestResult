class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for i in range(len(l)):
            s = sorted(nums[l[i]:r[i]+1])
            if len(s) <= 2:
                res.append(True)
                continue
            d 
= s[1] - s[0]
            flag = True
            for j in range(len(s)-1):
                if s[j+1] - s[j] != d:
                    flag = False
                    break
            res.append(flag)
        return res
