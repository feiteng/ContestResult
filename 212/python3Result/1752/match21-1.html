class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        for i in range(len(l)):
            ll = l[i]
            rr = r[i]
            if rr - ll < 2:
                ans.append(True)
                continue
            x = nums[ll:rr + 1][::]
            x
.sort()
#             print(rr, ll, x)
            d = x[1] - x[0]
            f = True
            for j in range(2, len(x)):
                if d != x[j] - x[j-1]:
                    f = False
                    break
            ans.append(f)
        return ans
