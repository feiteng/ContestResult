class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        def canRearrange(l, r):
            if l + 1 >= r:
                return True
            array = sorted(nums[l:r+1])
            diff = array[1] - array[0]
            for i in range(1, len(array)-1):
                if array[i+1] - array[i] != diff:
                    return False
            return True
        return [canRearrange(left, right) for left, right in zip(l, r)]
