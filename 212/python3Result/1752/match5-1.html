class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        for s, e in zip(l, r):
            sub = nums[s:e+1]
            sub.sort()
            for i in range(2, len(sub)):
                if sub[i] - sub[i-1] != sub[1]-sub[0]:
                    ans.append(False)
                    break
            else:
                ans.append(True)
        return ans
