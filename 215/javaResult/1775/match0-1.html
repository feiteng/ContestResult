class OrderedStream {
    int ptr;
    String arr[];
    public OrderedStream(int n) {
        arr=new String[n];
        ptr=0;
    }
    
    public List<String> insert(int id, String value) {
        
        id-=1;
        arr[id]=value;
        ArrayList<String> list=new ArrayList<>();
        while(ptr<arr.length&&arr[ptr]!=null)
        {
            list.add(arr[ptr]);
            ptr++;
        }
        return list;
    }
}
