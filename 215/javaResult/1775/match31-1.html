class OrderedStream {
    
    int ptr = 1;
    String s[];
    int n;
    public OrderedStream(int n) {
        s = new String[n+1];
        
this.n = n;
    }
    
    public List<String> insert(int id, String value) {
        s[id] = value;
        List<String> res = new ArrayList<>();
        while(ptr<=n&&s[ptr]!=null){
            res.add(s[ptr]);ptr++;
        }
        
        return res;
    }
}
