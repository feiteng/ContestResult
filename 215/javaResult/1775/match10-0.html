class OrderedStream {
    String[] arrays;
    int ptr;

    public OrderedStream(int n) {
        arrays = new String[n + 2];
        ptr = 1;
    }
    
    public List<String> insert(int id, String value) {
        arrays[id] = value;
        List<String> res = new ArrayList<>();
        while (arrays[ptr] != null) {
            res.add(arrays[ptr++]);
        }
        return res;
    }
}
