class OrderedStream {

    String[] inputs;
    int index = 0;
    
    public OrderedStream(int n) {
        inputs = new String[n];
    }
    
    public List<String> insert(int id, String value) {
        inputs[id - 1] = value;
        List<String> output = new ArrayList<>();
        while(index < inputs.length && inputs[index] != null) {
            output.add(inputs[index]);
            index++;
        }
        return output;
    }
}
