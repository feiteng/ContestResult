class OrderedStream {
    
    private int idx = 1;
    private String[] data;

    public OrderedStream(int n) {
        data = new String[n + 1];
    }
    
    public List<String> insert(int id, String value) {
        data[id] = value;
        List<String> list = new ArrayList<>();
        while (idx < data.length && data[idx] != null) {
            list.add(data[idx++]);
        }
        return list;
    }
}
