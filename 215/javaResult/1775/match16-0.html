class OrderedStream {

    String[] values;
    int idx = 0;
    public OrderedStream(int n) {
        this.values = new String[n];
        this.idx = 0;
    }
    
    public List<String> insert(int id, String value) {
        this.values[id-1] = value;
        List<String> res = new ArrayList<String>();
        if(this.idx + 1 == id) {
            while(this.idx < this.values.length && this.values[this.idx] != null) {
                res.add(this.values[this.idx++]);
            }
        }
        
        return res;
    }
}
