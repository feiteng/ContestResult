class OrderedStream {
    Map<Integer, String> map;
    int lastUnvisited;
    public OrderedStream(int n) {
        map = new HashMap<>();
        lastUnvisited = 1;
    }
    
    public List<String> insert(int id, String value) {
        map.put(id, value);
        List<String> res = new ArrayList<>();
        if (id == lastUnvisited) {
            while (map.containsKey(lastUnvisited)) {
                res.add(map.get(lastUnvisited));
                lastUnvisited++;
            }
        }
        
        return res;
    }
}
