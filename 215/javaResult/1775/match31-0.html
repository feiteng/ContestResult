class OrderedStream {
    String[] strs;
    int ptr;
    int n;
    public OrderedStream(int n) {
        strs = new String[n];
        ptr = 0;
        this.n = n;
    }
    
    public List<String> insert(int id, String value) {
        strs[id - 1] = value;
        List<String> ans = new ArrayList<>();
        while (ptr < n && strs[ptr] != null) {
            ans.add(strs[ptr++]);
        }
        return ans;
    }
}
