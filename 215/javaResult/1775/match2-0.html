class OrderedStream {

    int ptr;
    String[] stream;
    
    public OrderedStream(int n) {
        stream = new String[n];
        ptr = 0;
    }
    
    public List<String> insert(int id, String value) {
        stream[id - 1] = value;
        List<String> res = new ArrayList<>();
        while(ptr < stream.length && stream[ptr] != null){
            res.add(stream[ptr]);
            ptr++;
        }
        return res;
    }
}
