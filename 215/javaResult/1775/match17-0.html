class OrderedStream {

    private final String[] strings;
    private int ptr;

    public OrderedStream(int n) {
        strings = new String[n];
        ptr = 0;
    }

    public List<String> insert(int id, String value) {
        id--;
        strings[id] = value;
        List<String> res = new ArrayList<>();
        for (; ptr < strings.length && strings[ptr] != null; ptr++) {
            res.add(strings[ptr]);
        }
        return res;
    }
}
