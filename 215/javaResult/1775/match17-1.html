class OrderedStream {
    private String[] strings;
    private int i;

    public OrderedStream(int n) {
        strings = new String[n];
        i = 0;
    }

    public List<String> insert(int id, String value) {
        strings[id - 1] = value;
        ArrayList<String> list = new ArrayList<>();
        for (; i < strings.length && strings[i] != null; i++) {
            list.add(strings[i]);
        }
        return list;
    }
}
