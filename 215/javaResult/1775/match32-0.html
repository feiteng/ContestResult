class OrderedStream {
    String[] strs;
    int ptr = 0;

    public OrderedStream(int n) {
        strs = new String[n];
    }
    
    public List<String> insert(int id, String value) {
        id--;
        strs[id] = value;
        List<String> ans = new ArrayList<>();
        while (ptr < strs.length && strs[ptr] != null)
            ans.add(strs[ptr++]);
        return ans;
    }
}
