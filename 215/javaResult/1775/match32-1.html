class OrderedStream {
    int ptr = 0;
    String[] strs;
    public OrderedStream(int n) {
        strs = new String[n];
    }
    
    public List<String> insert(int id, String value) {
        id--;
        strs[id] = value;
        if(id != ptr) {
            return new ArrayList<>();
        }
        
        
List<String> ans = new LinkedList<>();
        while(ptr<strs.length && strs[ptr]!=null) ans.add(strs[ptr++]);
        return ans;
    }
}
