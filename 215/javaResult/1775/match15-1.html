class OrderedStream {

    private int ptr;
    private String[] data;

    public OrderedStream(int n) {
        this.data = new String[n + 1];
        ptr = 1;
    }
    
    public List<String> insert(int id, String value) {
        data[id] = value;
        List<String> ret = new ArrayList();
        if (id == this.ptr) {
            while (this.ptr < data.length && data[this.ptr] != null) {
                ret.add(data[ptr++]);
            }
        }
        return ret;
    }
}
