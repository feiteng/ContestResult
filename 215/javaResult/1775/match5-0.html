class OrderedStream {
    private String[] mem;
    private int ptr;

    public OrderedStream(int n) {
        mem = new String[n];
        ptr = 0;
    }

    public List<String> insert(int id, String value) {
        mem[id - 1] = value;
        List<String> res = new ArrayList<>();
        while (ptr < mem.length && mem[ptr] != null) {
            res.add(mem[ptr]);
            ++ptr;
        }
        return res;
    }
}
