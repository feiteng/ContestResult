class OrderedStream {

    private String[] arr;
    private int idx;
    
    public OrderedStream(int n) {
        arr = new String[n];
        idx = 0;
    }
    
    public List<String> insert(int id, String value) {
        arr[id - 1] = value;
        List<String> res = new ArrayList<>();
        while (idx < arr.length && arr[idx] != null) {
            res.add(arr[idx++]);
        }
        return res;
    }
}
