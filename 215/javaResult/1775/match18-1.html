class OrderedStream {

    int ptr = 1;
    HashMap<Integer, String> db = new HashMap<>();
    public OrderedStream(int n) {
        
    }
    
    public List<String> insert(int id, String value) {
        List<String> ans = new LinkedList<>();
        db.put(id, value);
        while (db.containsKey(ptr)) {
            ans.add(db.get(ptr));
            ptr++;
        }
        return ans;
    }
}
