import java.util.ArrayList;
import java.util.List;

class OrderedStream {
    private String[] seq;
    private int pos;

    public OrderedStream(int n) {
        seq = new String[n];
        pos = 0;
    }

    public List<String> insert(int id, String value) {
        seq[id-1] = value;
        List<String> ans = new ArrayList<>();
        while (pos < seq.length && seq[pos] != null) {
            ans.add(seq[pos]);
            pos++;
        }
        return ans;
    }
}
