class OrderedStream {

    int p, q = 0;
    String[] strs;
    public OrderedStream(int n) {
        p = 0; q = 1;
        strs = new String[n + 1];
    }
    
    public List<String> insert(int id, String value) {
        List<String> list = new ArrayList<>();
        strs[id] = value;
        p++;
        while (q <= p && strs[q] != null) {
            list.add(strs[q++]);
        }
        return list;
        
    }
}
