class OrderedStream {
    String[] arr;
    int ptr;
    public OrderedStream(int n) {
        arr = new String[n+1];
        ptr=1;
    }
    
    public List<String> insert(int id, String value) {
        arr[id]=value;
        List<String> ret = new ArrayList<>();
        while(ptr<arr.length&&arr[ptr]!=null){
            ret.add(arr[ptr++]);
        }
        return ret;
    }
}
