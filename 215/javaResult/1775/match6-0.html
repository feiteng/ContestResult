class OrderedStream {
    int ptr;
    String[] stream;

    public OrderedStream(int n) {
        stream = new String[n + 1];
        ptr = 1;
    }
    
    public List<String> insert(int id, String value) {
        stream[id] = value;
        List<String> result = new ArrayList<String>();
        
        while (ptr < stream.length && stream[ptr] != null) {
            result.add(stream[ptr++]);
        } 
        
        return result;
    }
}
