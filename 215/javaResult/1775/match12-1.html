class OrderedStream {
    int curIdx;
    int n;
    String[] arr;
    public OrderedStream(int n) {
      curIdx = 0;
      this.n = n;
      arr = new String[n];
    }

    public List<String> insert(int id, String value) {
      id--;
      List<String> res = new ArrayList<>();
      arr[id] = value;
      while (curIdx < n && arr[curIdx] != null) {
        res.add(arr[curIdx]);
        curIdx++;
      }
      return res;
    }
  }
