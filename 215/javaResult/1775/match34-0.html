class OrderedStream {
    HashMap<Integer, String> map;
    int ptr;
    public OrderedStream(int n) {
        map = new HashMap<>();
        ptr = 1;
    }
    
    public List<String> insert(int id, String value) {
        map.put(id, value);
        List<String> ans = new ArrayList<>();
        if (map.containsKey(ptr)) {
            while (map.containsKey(ptr)) {
                ans.add(map.get(ptr));
                ++ptr;
            }
                
                
                
            // ptr = id;
        } 
        return ans;
        
            
   
        
    }
}
