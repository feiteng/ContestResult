class OrderedStream {
    
    String[] strs;
    int p;

    public OrderedStream(int n) {
        strs = new String[n];
    }
    
    public List<String> insert(int id, String value) {
        --id;
        strs[id] = value;
        List<String> result = new ArrayList<>();
        while(p < strs.length && strs[p] != null) {
            result.add(strs[p++]);
        }
        
        return result;
    }
}
