class Solution {
   public int minOperations(int[] nums, int x) {
    	int sum =0;
    	for(int num:nums) {
    		sum+=num;
    	}
    	int target = sum-x;
    	int start =0;
    	int end =0;
    	int cur =0;
    	int len = -1;
    	while(end<nums.length) {
    		cur+=nums[end];
    		while(start<=end&&cur>target) {
    			cur-=nums[start++];
    		}
    		if(cur==target) {
    			len = Math.max(len, end-start+1);
    		}
    		end++;
    	}
		return len==-1?-1:nums.length-len;
    }
}
