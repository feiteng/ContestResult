class Solution {
    public int minOperations(int[] nums, int x) {
        int total = 0;
        for (int i : nums) {
            total += i;
        }
        if (total < x){
            return -1;
        } else if (total == x) {
            return nums.length;
        }
        
        int tar = total - x;
        Map<Long, Integer> preSum = new HashMap<>();
        preSum.put(0l, -1);
        long sum = 0;
        
int max = -1;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (preSum.containsKey(sum - tar)) {
                max = Math.max(max, i - preSum.get(sum - tar));
            }
            preSum.put(sum, i);
        }
        return max == -1 ? -1 : nums.length - max;
    }
}
