class Solution {
    public int minOperations(int[] nums, int x) {
        int total = 0;
        int n = nums.length;
        for(int tmp:nums) total+=tmp;
        
        if(total<x) return -1;
        else if(total == x) return n;
        
        int target = total - x;
        int max = -1;
       
        
        
       
        for(int head = 0,c = 0, tail = 0;head<n;head++){
            c+=nums[head];
            while(c>target && tail<head){
                c-=nums[tail++];
            }
            if(c == target){
                max = Math.max(max, head -tail+1);
            }
        }
        
        return max==-1?-1:n-max;
        
    }
}
