class Solution {
    public int minOperations(int[] nums, int x) {
        Map<Integer, Integer> left = new HashMap<>(), right = new HashMap<>();
        int n = nums.length, l = 0, r = 0;
        if(nums[0] > x && nums[n-1] > x) return -1;
        for(int i = 0; i < n; i++) {
            l += nums[i];
            left.put(l, i+1);
        }
        for(int i = n - 1; i >= 0; i--) {
            r += nums[i];
            right.put(r, n - i);
        }
        if(r < x) return -1;
        
int target = x, min = n;
        for(int i = 0; i < n; i++) {
            target -= nums[i];
            if(target == 0) {
                min = Math.min(min, i + 1);
            } else if(target < 0) {
                break;
            }
            if(right.containsKey(target) && right.get(target) < n - i) {
                min = Math.min(min, i + 1 + right.get(target));
            }
        
}
        
        target = x;
        for(int i = n - 1; i >= 0; i--) {
            target -= nums[i]; 
            if(target == 0) {
                min = Math.min(min, n - i);
            } else if(target < 0) {
                break;
            
}
            if(left.containsKey(target) && left.get(target) < i) {
                min = Math.min(min, n - i + left.get(target));
            }
        }
        return min;
    }
}
