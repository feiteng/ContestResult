class Solution {
    public int minOperations(int[] nums, int x) {
        //min to pick exactly x
        //= widest window containing sum - x
        //expand to right, when starts overflowing, move left
        int total = 0;
        for (int i : nums) total += i;
        int desired = total - x;
        int left = 0;
        int currSum = 0;
        int answ = Integer.MAX_VALUE;
        for (int i  = 0; i < nums.length; i++) {
            currSum += nums[i];
            while (currSum > desired && left <= i) {
                currSum -= nums[left++];
            }
            if (currSum == desired) {
                int otherLength = left + (nums.length - i-1);
                answ = Math.min(answ, otherLength);
            } 
        }
        return answ == Integer.MAX_VALUE ? -1 : answ;
    }
}
