class Solution {
    public int minOperations(int[] nums, int x) {
        int sum = 0;
        for(Integer i: nums) sum += i;
        if(sum < x) return -1;
        if(sum == x) return nums.length;
        
int j = 0;
        int s = 0;
        int l = 0;
        for(int i = 0; i < nums.length; i++){
            s += nums[i];
            while(s > sum - x){
                s -= nums[j++];
            }
            if(s == sum - x){
                
l = Math.max(l, i - j + 1);
            }
        }
        return l == 0 ? -1: nums.length - l;
    }
}
