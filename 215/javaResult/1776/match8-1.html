class Solution {
    public int minOperations(int[] nums, int x) {
        int arrSum = 0;
        for (int num : nums){
            arrSum += num;
        }
        
        int expSum = arrSum - x, maxLen = -1, sum = 0;
        if (expSum == 0){
            return nums.length;
        }
        
Map<Integer, Integer> leftSumToEndIndex = new HashMap<>();
        leftSumToEndIndex.put(0, -1);
        
        for (int i = 0; i < nums.length; i++){
            sum += nums[i];
            if (leftSumToEndIndex.containsKey(sum - expSum)){
                maxLen = Math.max(maxLen, i - leftSumToEndIndex.get(sum - expSum));
            }
            leftSumToEndIndex.put(sum, i);
        }
        
        return maxLen == -1 ? -1 : nums.length - maxLen;
    }
}
