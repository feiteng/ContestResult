class Solution {
    public int minOperations(int[] nums, int x) {
        Map<Integer,Integer> map=new HashMap<Integer,Integer>();
        int n=nums.length;
        int sum=0;
        int res=n+1;
        for(int i=0;i<n;i++){
            sum+=nums[i];
            if(sum==x){
                res=Math.min(res,i+1);
            }
            map.put(sum,i);
        }
        sum=0;
        for(int i=n-1;i>=0;i--){
            sum+=nums[i];
            if(sum==x){
                res=Math.min(res,n-i);
            }
            if(map.containsKey(x-sum)&&map.get(x-sum)<i){
                res=Math.min(res,map.get(x-sum)+1+n-i);
            }
        }
        return res==n+1?-1:res;
    }
}
