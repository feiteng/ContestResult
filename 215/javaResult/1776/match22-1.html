class Solution {
    public int minOperations(int[] nums, int x) {
        var sumFromLeft = new HashMap<Integer, Integer>();
        int sum = 0;
        sumFromLeft.put(0, 0);
        
for (int i=0; i<nums.length; i++) {
            sum += nums[i];
            sumFromLeft.put(sum, i+1);
        }
        
        int sol = sumFromLeft.containsKey(x) ? sumFromLeft.get(x) : Integer.MAX_VALUE;
        int sumFromRight = 0;
        
        
for (int j=1; j<=nums.length; j++) {
            sumFromRight += nums[nums.length-j];
            if (sumFromLeft.containsKey(x - sumFromRight))
                sol = Math.min(sol, sumFromLeft.get(x - sumFromRight) + j);
        }
        return sol > nums.length ? -1 : sol;
    }
}
