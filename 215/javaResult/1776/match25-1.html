class Solution {
    public int minOperations(int[] nums, int x) {
        Map<Integer, Integer> preSumMap = new HashMap();
        int preSum = 0;
        int n = nums.length;
        for(int i=0; i<n; i++){
            preSum += nums[i];
            preSumMap.putIfAbsent(preSum, i);
        }
        int res = preSumMap.containsKey(x) ? (preSumMap.get(x)+1) : Integer.MAX_VALUE;
        preSum = 0;
        for(int i=n-1; i>=0; i--){
            preSum += nums[i];
            int target = x-preSum;
            
if(target==0){
                res = Math.min(res, n-i);
            }
            if(preSumMap.containsKey(target)){
                int index = preSumMap.get(target);
                if(i>index){
                    res = Math.min(res, index+1+n-i);
                
}
            }
        }
        return res==Integer.MAX_VALUE ? -1 : res;
    }
}
