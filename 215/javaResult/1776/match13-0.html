// DP X -> space complexity too high
// Binary Search X -> unable to find a way fast to determine whether to go left or right
class Solution {
    public int minOperations(int[] nums, int x) {
        if (x < nums[0] && x < nums[nums.length - 1]) {
            return -1;
        }
        
        int sum = 0;
        for (var n : nums){
            sum += n;
        }
        
        int target = sum - x;
        if (target < 0) {
            return -1;
        }
        if (target == 0) {
            return nums.length;
        }
        
        int left = 0, right = 0;
        int window = 0;
        int maxWindow = -1;
        while (right < nums.length) {
            window += nums[right];
            while (left <= right && window > target) {
                window -= nums[left];
                left++;
            }
            if (window == target) {
                maxWindow = Math.max(maxWindow, right - left + 1);
            }
            right++;
        }
        
        if (maxWindow < 0) {
            return -1;
        }
        return nums.length - maxWindow;
    }
}
