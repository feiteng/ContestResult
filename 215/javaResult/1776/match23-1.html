class Solution {
    public int minOperations(int[] nums, int x) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        int n = nums.length;
        int sum = 0;
        mp.put(sum, 0);
        int ans = Integer.MAX_VALUE;
        for(int i = 0; i < n; ++i){
            sum += nums[i];
            mp.put(sum, i+1);
            if(sum == x){
                ans =  Math.min(ans, i + 1);
            
}
        }
        sum = 0;
        
        for(int i = n - 1; i >= 0; --i){
            sum += nums[i];
            if(mp.get(x - sum) != null){
                int t = mp.get(x - sum);
                // System.out.println(x - sum + " " + t);
                if(t <= i){
                    ans = Math.min(ans, t + n - i);
                }
            
}
        }
        return ans == Integer.MAX_VALUE? -1 : ans;
    }
}
