class Solution {
    public int minOperations(int[] nums, int x) {
        int N = nums.length;
        int res = Integer.MAX_VALUE;
        // step1:
        int sum = 0;
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        
for (int i = 0; i < N; i++) {
            sum += nums[i];
            map.put(sum, i);
            if (sum == x) {
                res = i + 1;
            }
        }
        // step2:
        sum = 0;
        for (int i = N - 1; i >= 0; i--) {
            sum += nums[i];
            if (map.containsKey(x - sum) && i > map.get(x - sum)) {
                res = Math.min(res, N - i + map.get(x - sum) + 1);
            }
        }
        return res == Integer.MAX_VALUE ? -1 : res;
    }
}
