class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 0);
        int sum = 0;
        for (int i=0; i<n; i++) {
            sum += nums[i];
            map.put(sum, i+1);
        }
        int ans = map.containsKey(x) ? map.get(x) : n+1;
        sum = 0;
        for (int i=n-1; i>=0; i--) {
            sum += nums[i];
            if (map.containsKey(x - sum))
                ans = Math.min(ans, (n-i) + map.get(x - sum));
        }
        return ans > n ? -1 : ans;
    }
}
