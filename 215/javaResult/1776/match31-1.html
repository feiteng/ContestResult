class Solution {
    public int minOperations(int[] nums, int x) {
        int sum = 0;
        for(int k : nums){
            sum+=k;
        }
        int n = nums.length;
        int[] pref = new int[n+1];
        for(int i = 1; i <= n; i ++){
            pref[i] = pref[i-1]+nums[i-1];
        }
        int min = Integer.MAX_VALUE;
//        for(int i = 0; i < n; i ++){
//            for(int j = 0; j <= i; j ++){
//                int cur = pref[i+1]-pref[j];
//                if(cur == sum-x){
//                    min = Math.min(min,n-(i-j+1));
//                }
//            }
//        }
        
        int cur = 0;
        int target = sum-x;
        
HashMap<Integer,Integer> map = new HashMap<>();
        map.put(0,-1);
        for(int i = 0; i < n; i ++){
            cur+=nums[i];
            if(map.containsKey(cur-target)){
                min = Math.min(min,n-(i-map.get(cur-target)));
            }
            map.put(cur,i);
        }
        if(sum == x) {
            min = Math.min(min,n);
        }
        return min == Integer.MAX_VALUE ? -1 : min;
    }
}
