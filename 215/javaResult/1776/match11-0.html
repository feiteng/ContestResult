class Solution {
    public int minOperations(int[] nums, int x) {
        int s = 0;
        for (int num : nums) {
            s += num; 
        }
        if (s < x) {
            return -1;
        } else if (s == x) {
            return nums.length;
        }
        int diff = s - x;
        int accum = 0;
        // int[] preSum = new int[nums.length + 1];
        // for (int i = 0; i< nums.length; i++) {
        //     preSum[i + 1] = preSum[i] + nums[i];
        // }
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 0);
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            accum += nums[i];
            if (map.containsKey(accum - diff)) {
                res= Math.max(res, i - map.get(accum - diff) + 1);
            }
            map.put(accum, i + 1);
        }
        return res == 0 ? -1 : nums.length - res;
        
    }
}
