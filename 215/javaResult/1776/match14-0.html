class Solution {
    public int minOperations(int[] nums, int x) {
        
        int n = nums.length;
        
        // let's compute a prefix sum -- with the number of nums used
        HashMap<Integer, Integer> pfsMap = new HashMap<>();
        int pfs = 0;
        
        pfsMap.put(0, 0);
        for (int i = 0; i < n; i++) {
            pfs += nums[i];
            pfsMap.put(pfs, i+1);
        }
        
        // System.out.println(pfsMap);
        

        
        // for a suffix sum, check if the corresponding prefix sum exists
        int sfs = 0;
        int best = n+1;
        
        if (pfsMap.containsKey(x)) {
            best = pfsMap.get(x);
        }
        
        for (int j = n-1; j >= 0; j--) { // j is sort of the "right count", or rCount
            sfs += nums[j];
            int pfsNeeded = x - sfs;
            if (pfsMap.containsKey(pfsNeeded)) {
                int lCount = pfsMap.get(pfsNeeded); // 0 ... lCount-1
                if (lCount < j) {
                    best = Math.min(best, lCount + (n-j));
                }
            }
        }
        
        return (best == n+1) ? -1 : best;
    }
}
