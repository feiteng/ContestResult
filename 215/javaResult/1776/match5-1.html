class Solution {
    public int minOperations(int[] nums, int x) {
        HashMap<Integer, Integer> map = new HashMap(); //key: Sum, value: Index, don't override
        
int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        if (sum == x) {
            return nums.length;
        }
        int target = sum - x;
        
        
map.put(0, -1);
        int runningSum = 0;
        int maxDifference = -1;
        for (int i = 0; i < nums.length; i++) {
            runningSum += nums[i];
            // System.out.println(runningSum - target);
            // System.out.println(map.containsKey(runningSum - target));
            if (map.containsKey(runningSum - target)) {
                // System.out.println("HERE");
                int index = map.get(runningSum - target);
                maxDifference = 
Math.max(maxDifference, i - index);
            }
            map.put(runningSum, i);
        }
        
        return maxDifference == -1 ? -1 : nums.length - maxDifference;
        
        
    }
}
