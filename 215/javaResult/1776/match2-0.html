class Solution {
    public int minOperations(int[] nums, int x) {
        int sum = 0;
        for (int i : nums) sum += i;
        sum -= x;
        if (sum < 0) return -1;
        
        int i = 0, res = -1;
        int temp = 0;
        for (int j = 0; j < nums.length; j++) {
            temp += nums[j];
            while (i < nums.length && temp > sum) {
                temp -= nums[i];
                i++;
            }
            if (temp == sum) {
                res = Math.max(j - i + 1, res);
            }
        }
        return res == -1? -1 : nums.length - res;
    }
}
