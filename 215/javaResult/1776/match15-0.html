class Solution {
    public int minOperations(int[] nums, int x) {
        int[] preSum = new int[nums.length + 1];
        Map<Integer, Integer> pre = new HashMap<>();
        int[] postSum = new int[nums.length + 1];
        Map<Integer, Integer> post = new HashMap<>();
        pre.put(0, 0);
        for (int i = 1; i <= nums.length; i++) {
            preSum[i] = preSum[i - 1] + nums[i - 1];
            pre.put(preSum[i], i);
        }
        post.put(0, 0);
        for (int i = 1; i <= nums.length; i++) {
            postSum[i] = postSum[i - 1] + nums[nums.length - i];
            post.put(postSum[i], i);
        }
        if (preSum[nums.length] < x || (preSum[1] > x && postSum[1] > x)) return -1;
        int res = Integer.MAX_VALUE;
        for (int i = 0; i <= nums.length; i++) {
            int cur = x - postSum[i];
            if (pre.containsKey(cur)) {
                int l = pre.get(cur);
                res = Math.min(res, l + i);
            }
        }
        for (int i = 0; i <= nums.length; i++) {
            int cur = x - preSum[i];
            if (post.containsKey(cur)) {
                int r = post.get(cur);
                res = Math.min(res, r + i);
            }
        }
        return res == Integer.MAX_VALUE ? -1 : res;
    }
}
