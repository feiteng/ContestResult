class Solution {
    public int minOperations(int[] nums, int x) {
        
int sum = 0;
        for(int i:nums){
            sum+=i;
        }
        
        int target = sum-x;
        if(target<0){
            return -1;
        }
        if(target==0){
            return nums.length;
        }
        
        int i = 0;
        int j = 0;
        int len = 0;
        int cur =0;
        
        while(j<nums.length){
            cur+=nums[j++];
            if(cur==target){
                len = Math.max(len,j-i);
            }
            
while(cur>target){
                cur-=nums[i++];
            }
            if(cur==target){
                len = Math.max(len,j-i);
            }
        }
        if(len==0){
            return -1;
        }
        return nums.length-len;
        
    }
    
    
}
