import java.util.Arrays;

class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        int[] pre = new int[n];
        pre[0] = nums[0];
        for (int i = 1; i < n; i++) {
            pre[i] = pre[i - 1] + nums[i];
        }
        int ans = n + 1;
        int sum = 0;
        int p = Arrays.binarySearch(pre, x);
        if (p >= 0) {
            ans = Math.min(ans, p + 1);
        }
        for (int i = n - 1; i >= 0; i--) {
            sum += nums[i];
            if (sum == x) {
                ans = Math.min(ans, n - i);
            } else if (sum < x) {
                p = Arrays.binarySearch(pre, x - sum);
                if (p >= 0) {
                    ans = Math.min(ans, n - i + p + 1);
                }
            }
        }
        return ans == n + 1 ? -1 : ans;
    }
}
