class Solution {
    public int minOperations(int[] nums, int x) {
        
        
        
        int sum = 0;
        
        //System.out.println(nums.length);
        for (int num : nums) {
            sum += num;
        }
        if (sum < x) {
            return -1;
        }
        
        ///
        // for (int i = 0; i < 123 - 10; ++i) {
        //     int testSum = 0;
        //     for (int j = i; j < i + 10; ++j) {
        //         testSum += nums[j];
        //     }
        //        if (sum - testSum - x == 0) System.out.println(i);
        // }
        // ///
        
        int i = 0;
        while (x > 0) {
            x -= nums[i];
            i += 1;
        }
        int ret = 999999999;
        if (x == 0) {
            ret = i;
        }
        int curr = i;
        int j = nums.length - 1;
        
        while (i < j && i > 0) {
            while (i > 0 && x < 0) {
                i -= 1;
                x += nums[i];
                curr -= 1;
            }
            if (x == 0 && i > 0) {
                ret = Math.min(curr, ret);
                i -= 1;
                x += nums[i];
                curr -= 1;
            }
            while (j > i && x > 0) {
                x -= nums[j];
                j -= 1;
                curr += 1;
            }
            //System.out.println("i " + i + " j " + j + " x " + x);
            if (x == 0) {
                ret = Math.min(curr, ret);
            }
        }
        
        
        if (ret == 999999999) {
            return -1;
        }
        return ret;
    }
}
