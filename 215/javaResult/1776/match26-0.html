class Solution {
    public int minOperations(int[] nums, int x) {
        int n=nums.length;
        int tot=0;
        HashMap<Integer,Integer> map=new HashMap<>();
        for(int i=0;i<n;i++){
            tot+=nums[i];
        }
        if(tot==x) return n;
        int sum=0;
        map.put(0,0);
        int ans=n+32;
        for(int i=1;i<=n;i++){
            sum+=nums[i-1];
            if(map.containsKey(sum-tot+x)){
                ans=Math.min(ans,n-(i-map.get(sum-tot+x)));
            }
            map.put(sum,i);
        }
        if(ans>n) return -1;
        return ans;
        
    }
}
