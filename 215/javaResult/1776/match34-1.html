class Solution {
    public int minOperations(int[] nums, int x) {
        int sum = 0;
        for(int num : nums) {
            sum += num;
        }
        int n = nums.length;
        int target = sum - x;
        if (target == 0) {
            return n;
        } else if (target < 0) {
            return -1;
        }
        int result = 0;
        for(int left = 0, right = 0, s = 0; right < n; ) 
{
            s += nums[right++];
            while(s > target) {
                s -= nums[left++];
            }
            if (s == target) {
                result = Math.max(result, right - left);
            }
        }
        return result == 0 ? -1 : n - result;
    }
}
