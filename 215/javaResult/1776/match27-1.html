class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, 0);
        int sum = 0;
        for (int i = n-1; i >= 0; i--) {
            sum += nums[i];
            if (sum > x)
                break;
            map.put(sum, n-i);
        }
        
        sum = 0;
        int ans = Integer.MAX_VALUE;
        if (map.containsKey(x))
            ans = map.get(x);
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (sum > x)
                break;
            
if (map.containsKey(x-sum) && map.get(x-sum) + i + 1 <= n) {
                ans = Math.min(ans, map.get(x-sum) + i + 1);
            }
        }
        return (ans == Integer.MAX_VALUE ? -1 : ans);
    }
}
