class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        int []c = new int[n];
        c[0] = nums[0];
        for(int i=1;i<n;i++){
            c[i] = nums[i] + c[i-1];
        }
        int ans = n+1;
        int val = 0;
        int ind = Arrays.binarySearch(c, x-val);
        if(ind >= 0){
            ans = Math.min(ans, ind+1);
        }
        for(int i=n-1;i>=0;i--){
            val += nums[i];
            if(val == x){
                ans = Math.min(ans, n-i);
            }
            
ind = Arrays.binarySearch(c, x-val);
            if(ind >= 0){
                ans = Math.min(ans, n-i+ind+1);
            }
        }
        return ans == n+1 ? -1 : ans;
    }
}
