class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        int pre = 0, post = 0, ans = Integer.MAX_VALUE;
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i < n; i++) {
            pre += nums[i];
            map.put(pre, i + 1);
        }
        map.put(0, 0);
        if(map.containsKey(x)) ans = map.get(x);
        
for(int i = n - 1; i >= 0; i--) {
            post += nums[i];
            if(map.containsKey(x - post) && map.get(x - post) <= i) {
                ans = Math.min(ans, n - i + map.get(x - post));
            }
        }
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
}
