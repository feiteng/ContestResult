class Solution {

    public int minOperations(int[] nums, int x) {
        int sum = 0; 
        for(int i : nums) sum += i; 
        int target = sum - x; 
        
        int window = 0; 
        int len = 0;
    
        
        int i = 0, j = 0; 
        while(i < nums.length && j < nums.length) { 
            if(window > target){
                window -= nums[i];
                i++;
            } else {
                window += nums[j];
                j++;
            }
            //System.out.println(i + " " + j + " " + window + " " + target);
            if(window == target) {
                if(j - i + 1 > len) {
                    len = j - i ;
                }
                
            } 
          
        }
        if(len == 0) return sum == x ? nums.length : -1;
        
        return nums.length - len;
    }
}
