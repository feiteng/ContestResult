class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        int total = 0;
        for (int num : nums) {
            total += num;
        }
        
if (x > total) {
            return -1;
        }
        int res = n + 1;
        int target = total - x;
        int sum = 0;
        for (
int i = 0, j = 0; j < nums.length; j++) {
            sum += nums[j];
            while (sum > target && i <= j) {
                sum-=nums[i];
                i++;
            }
            if (sum == target) {
                res = Math.min(res, n - (j - i + 1));
            }
        }
        return res == n + 1 ? -1 : res;
    }
}
