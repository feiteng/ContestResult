class Solution {
    public int minOperations(int[] nums, int x) {
        int len = nums.length;
        int sum = 0;
        Map<Integer,Integer> map = new HashMap<>();
        for(int num : nums){
            sum += num;
        }
        if(sum == x) return len;
        int cur = 0;
        int ans = 9999999;
        map.put(0 , -1);
        
for(int i = 0 ; i < len ; i++){
            cur += nums[i];
            if(map.containsKey(cur - (sum - x))){
                ans = Math.min(ans , len - (i - map.get(cur - (sum - x))));
            }
            if(!map.containsKey(cur)){
                map.put(cur , i);
            }
        }
        return ans == 9999999 ? -1 : ans;
        
    }
}
