class Solution {
    public int minOperations(int[] nums, int x) {
        int n = nums.length;
        Map<Integer,Integer> map = new HashMap<>();
        map.put(0,0);
        int sum = 0;
        for(int i = 0; i < nums.length; i++){
            sum += nums[i];
            map.put(sum, i + 1);
        }
        sum = 0;
        int res = 100001;
        for(int i = n - 1; i >= 0; i--){
            if(map.containsKey(x - sum) && map.get(x - sum) - 1 <= i){
                res = Math.min(res, n - i - 1 + map.get(x - sum));
            }
            sum += nums[i];
        }
        
        return res == 100001 ? -1 : res;
    }
}
