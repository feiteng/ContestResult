class Solution {
    public boolean closeStrings(String word1, String word2) {
		int[] a = new int[26], b = new int[26];
		for (char c : word1.toCharArray()) {
			a[c - 'a']++;
		}
		for (char c : word2.toCharArray()) {
			b[c - 'a']++;
		}
		for (int i = 0; i < a.length; i++) {
			if (a[i] != b[i] && a[i] * b[i] == 0)
				return false;
		}
		Arrays.sort(a);
		Arrays.sort(b);
		for (int i = 0; i < a.length; i++) {
			if (a[i] != b[i])
				return false;
		}
		return true;
    }
}
