class Solution {
    public boolean closeStrings(String word1, String word2) {
        int[] one = new int[26];
        int[] two = new int[26];
        for (char c : word1.toCharArray()) {
            one[c - 'a']++;
        }
        for (char c : word2.toCharArray()) {
            two[c - 'a']++;
        }
        for (int i = 0; i < 26;i++) {
            if (one[i] == 0 && two[i] != 0 || two[i] == 0 && one[i] != 0) {
                return false;
            }
        }
        Arrays.sort(one);
        Arrays.sort(two);
        for (int i = 0; i < 26;i++) {
            if (one[i] != two[i]) {
                return false;
            }
        }
        return true;
    }
}
