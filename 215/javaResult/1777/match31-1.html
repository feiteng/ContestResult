class Solution {
    public boolean closeStrings(String word1, String word2) {
        if(word1.length() != word2.length()) {
            return false;
        }
        
        int[] signature1 = new int[26], signature2 = new int[26];
        for(int i = 0; i < word1.length(); ++i) {
            ++signature1[word1.charAt(i)-'a'];
            ++signature2[word2.charAt(i)-'a'];
        }
        
        for(int i = 0; i < signature1.length; ++i) {
            if(signature1[i] > 0 && signature2[i] == 0 || 
              signature1[i] == 0 && signature2[i] > 0) {
                
return false;
            }
        }
        
        Arrays.sort(signature1);
        Arrays.sort(signature2);
        for(int i = 0; i < signature1.length; ++i) {
            if(signature1[i] != signature2[i]) {
                return false;
            }
        }
        
        return true;
    }
}
