class Solution {
    public boolean closeStrings(String word1, String word2) {
        int[] hist1 = new int[26];
        int[] hist2 = new int[26];
        for (char c : word1.toCharArray()) {
            hist1[c - 'a']++;
        }
        for (char c : word2.toCharArray()) {
            hist2[c - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if ((hist1[i] > 0 && hist2[i] == 0) || (hist2[i] > 0 && hist1[i] == 0)) {
                return false;
            }
        }
        Arrays.sort(hist1);
        Arrays.sort(hist2);
        for (int i = 0; i < 26; i++) {
            if (hist1[i] != hist2[i]) {
                return false;
            }
        }
        return true;
    }
}
