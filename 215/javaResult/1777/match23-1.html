class Solution {
    public boolean closeStrings(String w1, String w2) {
        int arr[] = new int[26];
        int brr[] = new int[26];
        for(int i =0; i<w1.length();i++){
            arr[w1.charAt(i)-'a']++;
        }
        for(int i =0; i<w2.length();i++){
            brr[w2.charAt(i)-'a']++;
        }
        for(int i =0;i<26;i++){
            if(arr[i]>0 && brr[i]==0 || brr[i]>0 && arr[i]==0){
                //System.out.println(i);
                return false;
            }
        }
        Arrays.sort(arr);
        Arrays.sort(brr);
        for(int i =0;i<26;i++){
            if(arr[i]!=brr[i]){
                return false;
            }
        }
        return true;
    }
}
