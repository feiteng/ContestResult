class Solution {
    static int n;
    static int m;
    static int[][]dir=new int[][]{{0,1},{0,-1},{1,0},{-1,0}};
    static int[][]mat;
    static HashMap<String,Integer>dp;
    static int calc(int r,int c,int intro,int extro)
    {
        if(r>=n)
            return 0;
        if(c>=m)
            return calc(r+1,0,intro,extro);
        String s=r+" "+c+" "+intro+" "+extro;
        if(r>=1)
        {
            for(int i=c;i<m;i++)
            {
                s=s+" "+mat[r-1][i];
            }
        }
        for(int i=0;i<c;i++)
        {
            s=s+" "+mat[r][i];
        }
        
        if(dp.containsKey(s))
        {
            //System.out.println(s);
            return dp.get(s);
        }
        int res=calc(r,c+1,intro,extro);
        if(intro>0)
        {
            int req=120;
            for(int i=0;i<4;i++)
            {
                int r1=r+dir[i][0];
                int c1=c+dir[i][1];
                if(r1<0 || c1<0 || r1>=n || c1>=m)
                    continue;
                if(mat[r1][c1]!=0)
                {
                    req-=30;
                    if(mat[r1][c1]==2)
                        req+=20;
                    else
                        req-=30;
                }
            }
            mat[r][c]=1;
            res=Math.max(res,req+calc(r,c+1,intro-1,extro));
            mat[r][c]=0;
        }
        if(extro>0)
        {
            int req=40;
            for(int i=0;i<4;i++)
            {
                int r1=r+dir[i][0];
                int c1=c+dir[i][1];
                if(r1<0 || c1<0 || r1>=n || c1>=m)
                    continue;
                if(mat[r1][c1]!=0)
                {
                    req+=20;
                    if(mat[r1][c1]==2)
                        req+=20;
                    else
                        req-=30;
                }
            }
            mat[r][c]=2;
            res=Math.max(res,req+calc(r,c+1,intro,extro-1));
            mat[r][c]=0;
        }
        dp.put(s,res);
        
return res;
    }
    public int getMaxGridHappiness(int m1, int n1, int intro, int extro) {
        n=m1;
        m=n1;
        mat=new int[n][m];
        dp=new HashMap<>();
        return calc(0,0,intro,extro);
    }
}
