	class Solution {
		public int getMaxGridHappiness(int n, int m, int ic, int ec) {
			int pm = (int)Math.pow(3, m);
			int I = Integer.MIN_VALUE / 2;
			int[][][] dp = new int[ic+1][ec+1][pm];
			for(int i = 0;i <= ic
;i++){
				for(int j = 0;j <= ec;j++){
					Arrays.fill(dp[i][j], I);
				}
			}
			dp[ic][ec][0] = 0;
			int[] E = {0, 120, 40};
			int[] N = {0, -30, 20};
			for(int z = 0;z < n;z++){
				int[] b = new int[m];
				int[][][] ndp = new int[ic+1][ec+1][pm];
				for(int i = 0;i <= ic;i++){
					for(int j = 0;j <= ec;j++){
						Arrays.fill(ndp[i][j], I);
					}
				}
				int j = 0;
				do{
					int sc = 0;
					int uic = 0, uec = 0;
					for(int k = 0;k < m;k++){
						sc += E[b[k]];
						if(b[k] == 1)uic++;
						if(b[k] == 2)uec++;
						if(k+1 < m && b[k+1] != 0){
							sc += N[b[k]];
						}
						if(k-1 >= 0 && b[k-1] != 0){
							sc += N[b[k]];
						}
					}

					// f -> j
					int[] ob = new int[m];
					int f = 0;
					int maxsc = I;
					do{
						for(int fic = uic;fic <= ic;fic++){
							for(int fec = uec;fec <= ec;fec++){
								int lsc = sc + dp[fic][fec][f];
								for(int k = 0;k < m;k++){
									if(ob[k] != 0 && b[k] != 0){
										lsc += N[ob[k]] + N[b[k]];
									}
								}
								ndp[fic-uic][fec-uec][j] = Math.max(
										ndp[fic-uic][fec-uec][j], lsc
								);
							}
						}

						f++;
					}while(inc(ob, 3));

					j++;
				}while(inc(b, 3));
				dp = ndp;
			}


			int ans = I;
			for(int i = 0;i <= ic;i++){
				for(int j = 0;j <= ec;j++){
					for(int k = 0;k < pm;k++){
						ans = Math.max(ans, dp[i][j][k]);
					}
				}
			}

			return ans;
		}

		public boolean inc(int[] a, int base)
		{
			int n = a.length;
			int i;
			for(i = n - 1;i >= 0 && a[i] == base - 1;i--);
			if(i == -1)return false;

			a[i]++;
			Arrays.fill(a, i + 1, n, 0);
			return true;
		}

	}
