class Solution {
    public int getMaxGridHappiness(int m, int n, int ic, int ec) {
        int rc = 1;
        for (int i = 0; i < n; ++i)
            rc*=3;
        int[] ric = new int[rc];
        int[] rec = new int[rc];
        int[] rs = new int[rc];
        for (int i = 0; i < rc; ++i) {
            int last = 0;
            int num = i;
            for (int j = 0; j < n; ++j) {
                int cur = num%3;
                if (cur==1) {
                    ric[i]++;
                    rs[i]+=120;
                    if (last==1)
                        rs[i]-=60;
                    if (last==2)
                        rs[i]-=10;
                }
                if (cur==2) {
                    rec[i]++;
                    rs[i]+=40;
                    if (last==1)
                        rs[i]-=10;
                    if (last==2)
                        rs[i]+=40;
                }
                last=cur;
                num/=3;
            }
        }
        int[][] conn = new int[rc][rc];
        for (int i = 0; i < rc; ++i) {
            for (int j = 0; j < rc; ++j) {
                int ni = i;
                int nj = j;
                for (int k = 0; k < n; ++k) {
                    int ci = ni%3;
                    int cj = nj%3;
                    if (ci==1&&cj==1)
                        conn[i][j]-=60;
                    if (ci==1&&cj==2)
                        conn[i][j]-=10;
                    if (ci==2&&cj==1)
                        conn[i][j]-=10;
                    if (ci==2&&cj==2)
                        conn[i][j]+=40;
                    ni/=3;
                    nj/=3;
                }
            }
        }
        
        int[][][][] dp = new int[m][rc][ic+n+1][ec+n+1];
        for (int i = 0; i < rc; ++i) {
            dp[0][i][ric[i]][rec[i]]=rs[i];
        }
        for (int i = 1; i < m; ++i) {
            for (int ci = 0; ci<=ic; ++ci) {
                for (int ce = 0; ce<=ec; ++ce) {
                    for (int j = 0; j < rc; ++j) {
                        if (dp[i-1][j][ci][ce]==0)
                            continue;
                        for (int k = 0; k < rc; ++k) {
                            dp[i][k][ci+ric[k]][ce+rec[k]] = Math.max(dp[i][k][ci+ric[k]][ce+rec[k]], dp[i-1][j][ci][ce]+rs[k]+conn[j][k]);
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < rc; ++j) {
                for (int ci = 0; ci<=ic; ++ci) {
                    for (int ce = 0; ce<=ec; ++ce) {
                        ans = Math.max(ans,dp[i][j][ci][ce]);
                    }
                }
            }
        }
        return ans;
    }
}
