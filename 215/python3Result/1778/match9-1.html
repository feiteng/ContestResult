class Solution:
    def getMaxGridHappiness(self, m: int, n: int, introvertsCount: int, extrovertsCount: int) -> int:
        dpp = [[0 for _ in range(m)] for _ in range(n)]
        
        def ival(dp, pp):
            i, j = pp
            if dp[i][j] != 0:
                return -10000
                
            v 
= 120
            if i == 0:
                v += 2
            if i == n - 1:
                v += 2
            if j == 0:
                v += 1
            if j == m - 1:
                v += 1
                
            for ii, jj in [[i+1,j],[i-1,j],[i,j+1],[i,j-1]]:
                if 0 <= ii < n and 0 <= jj < m:
                    if dp[ii][jj] == 0:
                        pass
                    elif dp[ii][jj] == 1:
                        v -= 60
                    elif dp[ii][jj] == 2:
                        v = v - 30 + 20
            return v
        
        def eeval(dp, pp):
            i, j = pp
            if dp[i][j] != 0:
                return -10000
            
            v = 49
            if i == 0:
                v -= 1
            if i == n - 1:
                v -= 1
            if j == 0:
                v -= 2
            if j == m - 1:
                v -= 2
           
            for ii, jj in [[i+1,j],[i-1,j],[i,j+1],[i,j-1]]:
                if 0 <= ii < n and 0 <= jj < m:
                    if dp[ii][jj] == 0:
                        pass
                    elif dp[ii][jj] == 1:
                        v = v - 30 + 20
                    elif dp[ii][jj] == 2:
                        v += 40
            return v
        
        poss = [(i, j) for i in range(n) for j in range(m)]
        def tt(rr):
            return tuple([tuple(n) for n in rr])
        def dd(rr):
            return list([list(n) for n in rr])
        
        ans = 0
        cc = {}
        def f(ic, ec, dp):
            if dp in cc:
                return cc[dp]
            
            ret = 0
            odp =dp
            dp = dd(dp)
            if ic > 0:
                cand = {ival(dp, (i, j)): (i, j) for i, j in poss}
                for v, (i, j) in list(sorted(cand.items(), reverse=True))[:2]:
                    if v > 0:
                        v = (v // 10) * 10
                        dp[i][j] = 1
                        ret = max(ret, f(ic - 1, ec, tt(dp)) + v)
                        dp[i][j] = 0
            if ec > 0:
                cand = {eeval(dp, (i, j)): (i, j) for i, j in poss}
                for v, (i, j) in list(sorted(cand.items(), reverse=True))[:2]:
                    if v > 0:
                        v = (v // 10) * 10
                        dp[i][j] = 2
                        ret = max(ret, f(ic, ec - 1, tt(dp)) + v)
                        dp[i][j] = 0
                        
            cc[odp] = ret
            return ret
                        
        return f(introvertsCount, extrovertsCount, tt(dpp))
