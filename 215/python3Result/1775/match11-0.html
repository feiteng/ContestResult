class OrderedStream:

    def __init__(self, n: int):
        self.mm = {}
        self.ptr = 1

    def insert(self, id: int, value: str) -> List[str]:
        self.mm[id] = value
        sol = []
        while self.ptr in self.mm:
            sol.append(self.mm[self.ptr])
            self.ptr += 1
        return sol
