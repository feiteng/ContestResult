class OrderedStream:

    def __init__(self, n: int):
        self.queue = [None]*(n+1)
        self.n = 1

    def insert(self, id: int, value: str) -> List[str]:
        self.queue[id]=value
        ans = []
        while self.n<len(self.queue) and self.queue[self.n]:
            ans.append(self.queue[self.n])
            self.n+=1
        return ans
