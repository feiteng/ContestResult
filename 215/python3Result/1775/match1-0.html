class OrderedStream:

    def __init__(self, n: int):
        self.cache = [None] * (n+1)
        self.cur = 0

    def insert(self, id: int, value: str) -> List[str]:
        self.cache[id-1] = value
        res = []
        while self.cache[self.cur] is not None:
            res.append(self.cache[self.cur])
            self.cur += 1
        return res
