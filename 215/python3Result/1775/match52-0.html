class OrderedStream:

    def __init__(self, n: int):
        self.data = {}
        self.ptr = 1


    def insert(self, id: int, value: str) -> List[str]:
        self.data[id] = value
        ans = []
        while self.data.get(self.ptr, ""):
            ans.append(self.data[self.ptr])
            self.ptr += 1
        return ans
