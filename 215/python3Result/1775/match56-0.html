class OrderedStream:

    def __init__(self, n: int):
        self.array = [None] * (n+2)
        self.next = 1
        

    def insert(self, id: int, value: str) -> List[str]:
        ret = []
        self.array[id] = value
        if self.next == id:
            while self.array[self.next] is not None:
                ret.append(self.array[self.next])
                self.next += 1
        return ret
