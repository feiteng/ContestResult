class OrderedStream:

    def __init__(self, n: int):
        self.ans = n*[0]
        self.index = 0


    def insert(self, id: int, value: str) -> List[str]:
        self.ans[id-1] = value
        ans = []
        while(self.index < len(self.ans) and self.ans[self.index]):
            ans.append(self.ans[self.index])
            self.index+=1
        return ans
