class OrderedStream:

    def __init__(self, n: int):
        self.ptr = 1
        self.vals = [None] *(n + 1)


    def insert(self, id: int, value: str) -> List[str]:
        self.vals[id] = value
        ret = []
        while self.ptr < len(self.vals) and self.vals[self.ptr] is not None:
            ret.append(self.vals[self.ptr])
            self.ptr += 1
        return ret
