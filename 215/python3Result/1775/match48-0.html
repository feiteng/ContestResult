class OrderedStream:

    def __init__(self, n: int):
        self.li = [None] * n
        self.ptr = 0
        

    def insert(self, idx: int, value: str) -> List[str]:
        self.li[idx-1] = value
        res = []
        while self.ptr < len(self.li) and self.li[self.ptr] != None:
            res.append(self.li[self.ptr])
            self.ptr += 1
        return res
