class OrderedStream:

    def __init__(self, n: int):
        self.cnt = 1
        self.record = {}

    def insert(self, id: int, value: str) -> List[str]:
        self.record[id] = value
        target = []
        
        while self.cnt in self.record:
            target.append(self.record[self.cnt])
            self.cnt += 1
            
        return target
