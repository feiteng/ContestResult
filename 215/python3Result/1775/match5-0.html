class OrderedStream:

    def __init__(self, n: int):
        self.l = [None]*(n+1)
        self.ptr = 0


    def insert(self, id: int, value: str) -> List[str]:
        self.l[id-1]=value
        
        ans=[]
        while self.l[self.ptr] is not None:
            ans.append(self.l[self.ptr])
            self.ptr+=1
        
        return ans
