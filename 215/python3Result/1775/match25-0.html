class OrderedStream:

    def __init__(self, n: int):
        self.date = [""]*(n+1)
        self.ptr = 1


    def insert(self, idx: int, value: str) -> List[str]:
        self.date[idx] = value
        ans = []
        while self.ptr<len(self.date) and self.date[self.ptr]:
            ans.append(self.date[self.ptr])
            self.ptr+=1
        return ans
