class OrderedStream:

    def __init__(self, n: int):
        self.ptr = 1
        self.d = {}

    def insert(self, ID: int, value: str) -> List[str]:
        
        self.d[ID] = value
        
        ret = []
        
        while self.ptr in self.d:
            ret.append(self.d[self.ptr])
            self.ptr += 1
        
        return ret
