class OrderedStream:

    def __init__(self, n: int):
        self.d = {}
        self.p = 1

    def insert(self, id: int, value: str) -> List[str]:
        self.d[id] = value
        ret = []
        while self.p in self.d:
            ret.append(self.d[self.p])
            self.p += 1
        return ret
