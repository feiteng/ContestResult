class OrderedStream:

    def __init__(self, n: int):
        self.array = [""] * n
        self.ptr = 0

    def insert(self, id: int, value: str) -> List[str]:
        self.array[id - 1] = value
        res = []
        while self.ptr < len(self.array) and self.array[self.ptr] != "":
            res.append(self.array[self.ptr])
            self.ptr += 1
        return res
