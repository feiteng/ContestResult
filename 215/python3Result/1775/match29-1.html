class OrderedStream:

    def __init__(self, n: int):
        self.n = n
        self.ptr  = 1
        self.table = {}

    def insert(self, id: int, value: str) -> List[str]:
        self.table[id] = value
        if id == self.ptr:
            ans = []
            while self.ptr in self.table:
                ans.append(self.table[self.ptr])
                self.ptr+=1
            return ans
        else:
            return []
