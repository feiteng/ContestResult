class OrderedStream:

    def __init__(self, n: int):
        self.cur = 1
        self.dp = {}
        

    def insert(self, id: int, value: str) -> List[str]:
        self.dp[id] = value
        ans = []
        while self.cur in self.dp:
            ans.append(self.dp[self.cur])
            self.cur += 1
        return ans
