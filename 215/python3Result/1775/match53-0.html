class OrderedStream:

    def __init__(self, n: int):
        self.list = [None] * n
        self.pointer = 0

    def insert(self, id: int, value: str) -> List[str]:
        self.list[id-1] = value
        
        ans = []
        while self.pointer < len(self.list) and self.list[self.pointer] is not None:
            ans.append(self.list[self.pointer])
            self.pointer += 1
        return ans
