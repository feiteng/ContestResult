from bisect import bisect
class OrderedStream:

    def __init__(self, n: int):
        self.q = [None] * (n + 1)
        self.ptr = 1

    def insert(self, id: int, value: str) -> List[str]:
        self.q[id] = value
        res = []
        while self.ptr < len(self.q) and self.q[self.ptr] != None:
            res.append(self.q[self.ptr])
            self.ptr += 1
        return res
