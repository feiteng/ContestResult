class OrderedStream:

    def __init__(self, n: int):
        self.l = [-1]*n
        self.now = 0

    def insert(self, id: int, value: str) -> List[str]:
        self.l[id-1] = value
        res = []
        
        while self.now<len(self.l) and self.l[self.now]!=-1:
            res.append(self.l[self.now])
            self.now += 1
        
        return res
