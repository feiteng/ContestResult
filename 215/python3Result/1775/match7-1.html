class OrderedStream:

    def __init__(self, n: int):
        self.pos = 1
        self.data = {}


    def insert(self, id: int, value: str) -> List[str]:
        self.data[id] = value
        res = []
        while self.pos in self.data:
            res.append(self.data[self.pos])
            self.pos += 1
        return res
