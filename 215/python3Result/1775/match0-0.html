class OrderedStream:

    def __init__(self, n: int):
        self.data = [None]*(n+2)
        self.ptr = 1

    def insert(self, id: int, value: str) -> List[str]:
        self.data[id] = value
        res = []
        while(self.data[self.ptr]):
            res.append(self.data[self.ptr])
            self.ptr+=1
        return res
