class Solution:
    def closeStrings(self, A: str, B: str) -> bool:
        
        if len(A) != len(B):
            return False
        
        frequency_A = collections.Counter(A)
        frequency_B = collections.Counter(B)
        
        if set(frequency_A.keys()) != set(frequency_B.keys()):
            return False
        
        return sorted(frequency_A.values()) == sorted(frequency_B.values())
