class Solution:
    def closeStrings(self, word1: str, word2: str) -> bool:
        if len(word1) != len(word2): return False # edge case 
        cnt1 = Counter(word1)
        cnt2 = Counter(word2)
        if sorted(cnt1) != sorted(cnt2): return False 
        if sorted(cnt1.values()) != sorted(cnt2.values()): return False 
        return True
