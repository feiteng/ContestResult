class Solution:
    def closeStrings(self, word1: str, word2: str) -> bool:
        C1 = Counter(word1)
        C2 = Counter(word2)
        v1 = sorted(C1.values())
        v2 = sorted(C2.values())
        k1 = set(C1.keys())
        k2 = set(C2.keys())
        return v1 == v2 and k1 == k2
