class Solution:
    def closeStrings(self, word1: str, word2: str) -> bool:
        cnt1 = collections.Counter(word1)
        cnt2 = collections.Counter(word2)
        
        return list(sorted(cnt1.keys())) == list(sorted(cnt2.keys())) and sorted(list(cnt1.values())) == sorted(list(cnt2.values()))
