class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        m = float('inf')
        prefix = {0: -1}
        s = 0
        for i, num in enumerate(nums):
            s += num
            prefix[s] = i
        s = 0
        nums.append(0)
        for i in range(len(nums)-1,-1,-1):
            s += nums[i]
            
            if x-s in prefix:
                
                if i > prefix[x-s]:
                    m = min(m, 1+prefix[x-s]+len(nums)-1-i)
        if m == float('inf'):
            return -1
        else:
            return m
