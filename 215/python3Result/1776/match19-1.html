class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        n = len(nums)
        nums = nums * 2
        
        i, res, cur = 0, float('inf'), 0
        for j, num in enumerate(nums):
            cur 
+= num
            
            while cur > x:
                cur -= nums[i]
                i += 1
                
            if cur == x and i <= n and n - 1 <= j:
                res = min(res, j - i + 1)
        return res if 0 < res <= n else -1
