class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        s = sum(nums)
        s -= x
        if s == 0:
            return len(nums)
        if s < 0:
            return -1
        i 
= 0
        cur = 0
        ans = -1
        for j, num in enumerate(nums):
            cur += num
            while cur > s:
                cur -= nums[i]
                i += 1
            if cur == s:
                ans = max(ans, j - i + 1)
        return len(nums) - ans if ans > -1 else ans
