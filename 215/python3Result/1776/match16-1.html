import bisect
class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        tmp1 = [0]
        for i in nums:
            tmp1.append(tmp1[-1]+i)
        tmp2 = [0]
        for i in nums[::-1]:
            tmp2.append(tmp2[-1]+i)
        ans = float("inf")
        for i in range(len(tmp1)):
            tmpindex = i
            index = bisect.bisect_left(tmp2, x-tmp1[i])
            if tmpindex + index <= len(nums) and index != len(tmp2) and tmp2[index] == x-tmp1[i]:
                ans = min(ans, index+i)
        if ans != float("inf"):
            return ans
        else:
            return -1
