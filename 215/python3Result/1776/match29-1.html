class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        total = sum(nums)
        if x == total: return n
        tar = total - x
        psum = 0
        dic = {0:-1}
        win 
= 0
        for i, num in enumerate(nums):
            psum += num
            if psum - tar in dic:
                win = max(win, i - dic[psum - tar])
            if psum not in dic:
                dic[psum] = i
        return n - win if win != 0 else -1
