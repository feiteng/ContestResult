class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        t = sum(nums)
        if t < x:
            return -1
        if t == x:
            return len(nums)
        
        k = t - x
        mp = {}
        sumc = maxL = 0
        for i in range(len(nums)):
            sumc += nums[i]
            if sumc == k:
                maxL = i + 1
            elif sumc - k in mp:
                maxL = max(maxL, i-mp[sumc-k])
            if sumc not in mp:
                mp[sumc] = i
        return len(nums) - maxL if maxL != 0 else -1
