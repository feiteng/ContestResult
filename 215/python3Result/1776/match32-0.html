from itertools import accumulate

class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        N, total = len(nums), sum(nums)
        rightPtr, ans = N, N
        
        if total <= x:
            return N if total == x else -1
        
        for leftPtr in range(N-1, -1, -1):
            total -= nums[leftPtr]
            while leftPtr < rightPtr-1 and total < x:
                total += nums[rightPtr-1]
                rightPtr -= 1
            if total == x:
                ans = min(ans, N - (rightPtr - leftPtr))
        
        return ans if ans < N else -1
