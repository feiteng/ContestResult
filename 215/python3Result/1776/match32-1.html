class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        r = 0
        lsum = 0
        rsum = sum(nums)
        for i in range(n):
            if rsum <= x:
                break
            rsum -= nums[i]
            r += 1
        ans = 10 ** 9
        if rsum == x:
            ans =
 n - r
        for l in range(n):
            lsum += nums[l]
            while (rsum + lsum > x or r <= l) and r < n:
                rsum -= nums[r]
                r += 1
            # print(lsum, rsum)
            if rsum + lsum == x:
                ans = min(ans, l + 1 + n - r)
        if ans == 10 ** 9:
            return -1
        return ans
