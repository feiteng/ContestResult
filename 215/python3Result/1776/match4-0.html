class Solution:
    # find longest subarray equal to sum-x
    # len(arr)-solution of above
    # sliding window
    
    # [1,1,4,2,3], 5 -> 2
    def minOperations(self, nums: List[int], x: int) -> int:
        if x == 0:
            return 0
        elif not nums:
            return -1
        
        tot = sum(nums)
        if tot == x:
            return len(nums)
        elif tot < x:
            return -1
        
        left = 0; right = 0; target = tot-x; curSum = 0; maxLength = 0
        # len = 5
        # 2,3,6,6,3
        while right < len(nums):
            curSum += nums[right]
            while curSum > target:
                curSum -= nums[left]
                left += 1
                
            if curSum == target:
                maxLength = max(maxLength, right-left+1)
            right += 1
            
        return -1 if maxLength == 0 else len(nums)-maxLength
