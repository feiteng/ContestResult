class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        if min(nums) > x:
            return -1
    
        total_sum = sum(nums)
        if total_sum == x:
            return len(nums)

        target = total_sum - x
        prefix_dict = {0: -1}
        prefix_sum 
= 0
        max_length = -1
        for i, num in enumerate(nums):
            prefix_sum += num
            # print(prefix_sum, prefix_dict, prefix_sum - target)
            
            if prefix_sum - target in prefix_dict:
                max_length = max(max_length, i - prefix_dict[prefix_sum - target])
            if not prefix_sum in prefix_dict:
                prefix_dict[prefix_sum] = i
        
        if max_length == -1:
            return -1
        return len(nums) - max_length
