class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        if min(nums) > x: return -1
        target = sum(nums) - x
        if target == 0: return len(nums)
        # @lru_cache(None)
        # def dfs(l, r, c):
        #     if c == x: return 0
        #     if c > x or l>r: return math.inf
        #     return 1 + min(dfs(l+1,r,c+nums[l]), dfs(l,r-1,c+nums[r]))
        # res = dfs(0,len(nums)-1,0)
        # return res if res != math.inf else -1
        seen = {}
        seen[0] = -1
        res = 0
        cur = 0
        for i, n in enumerate(nums):
            cur += n
            if (cur - target) in seen: res = max(res, i-seen[cur-target])
            if cur not in seen: seen[cur] = i
        return (len(nums) - res) if res else -1
