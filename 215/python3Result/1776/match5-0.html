class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        total = sum(nums)
        
        if total < x:
            return -1
        elif total == x:
            return n
        
        target = total - x
        l, r = 0, 0
        ans = float('inf')
        temp = 0
        
        while r < n:
            temp += nums[r]
            while temp > target:
                temp -= nums[l]
                l += 1
                
            if temp == target:
                ans = min(ans, n - (r - l + 1))
                
            r += 1
            
        return ans if ans < float('inf') else -1
