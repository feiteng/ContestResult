class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        def helper(nums, target):   
            # longest subarray with sum = target
            n = len(nums)
            d = dict() 
            s = 0
            max_len = 0
            for i in range(n): 
                s += nums[i] 
                if (s == target): 
                    max_len = i + 1
                elif (s - target) in d: 
                    max_len = max(max_len, i - d[s-target]) 
                if s not in d: 
                    d[s] = i 

            return max_len 

        n = len(nums)
        sum_nums = sum(nums)
        if sum_nums < x:
            return -1
        if sum_nums == x:
            return n
        max_len = helper(nums, sum_nums-x)
        if max_len == 0:
            return -1
        return n - max_len
