class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        '''
        Numbers in array are strictly positive
        X is strictly positive
        
        if nums[0] > x and nums[-1] > x, return -1
        
        Find the longest subarray that sums to sum(array) - 5
        Use sliding window
        11 - 5 = 6
        '''
        
        target = sum(nums) - x
        if target < 0:
            return -1
        if target == 0:
            return len(nums)
        
        cur_sum = 0
        start = 0
        max_length = -float("inf")
        
        
        for i in range(len(nums)):
            cur_sum += nums[i]
            while cur_sum > target and start != i:
                cur_sum -= nums[start]
                start += 1
            if cur_sum == target:
                max_length = max(max_length, i - start + 1)
                
        return len(nums) - max_length if max_length != -float("inf") else -1
