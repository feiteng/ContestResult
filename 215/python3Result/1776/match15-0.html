import bisect
class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        presum = [0]
        for num in nums:
            presum.append(num + presum[-1])
    
        postsum = [0]
        for num in nums[::-1]:
            postsum.append(num + postsum[-1])
        
        res = float('inf')
        for i in range(len(presum)):
            t = x - presum[i]
            ind = bisect.bisect_left(postsum, t)
            if ind != len(postsum) and postsum[ind] == t and i + ind <= len(nums):
                res = min(res, i + ind)
        return res if res != float('inf') else -1
