class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        total = sum(nums)
        # find the longest subarray whose sum is total - x
        target = total - x
        ans = -1
        left = 0
        cur = 0
        for i, a in enumerate(nums):
            cur += a
            while left <= i and cur > target:
                cur -= nums[left]
                left += 1
            if cur == target:
                ans = max(ans, i - left + 1)
        return len(nums) - ans if ans != -1 else -1
