class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        if sum(nums) < x:
            return -1
        i = 0
        j = len(nums)-1
        q = 0
        best = 1e9
        while i < len(nums) and q < x:
            q += nums[i]
            i += 1
        if q == x:
            best = i
        for i in range
(i-1, -1, -1):
            q -= nums[i]
            while j > 0 and q < x:
                q += nums[j]
                j -= 1
            if q == x:
                best = min(best, i + len(nums) - j - 1)
        return -1 if best == 1e9 else best
