class Solution:
    def minOperations(self, A: List[int], x: int) -> int:
        n = len(A)
        pre, suf = [0]*(n+1), [0]*(n+1)
        for i in range(n):
            pre[i+1] = pre[i] + A[i]
            suf[i+1] = suf[i] + A[-1-i]
        ans = inf
        for i in range(n+1):
            j = bisect_left(suf, x-pre[i])
            if i + j <= n and suf[j] + pre[i] == x:
                ans = min(ans, i+j)
        return ans if ans != inf else -1
