class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        if sum(nums) < x:
            return -1
        
        temp = 0
        ans = len(nums) + 1
        left = 0
        
        while left < len(nums) and temp < x:
            temp += nums[left]
            left += 1
        
        if temp == x:
            ans = left
        
        left -= 1
        
        for right in range(len(nums) - 1, -1, -1):
            temp += nums[right]
            while temp > x and left >= 0:
                temp -= nums[left]
                left -= 1
            
            if temp == x:
                ans = min(ans, left + 1 + len(nums) - right)
            
            if left < 0:
                break
        
        return ans if ans <= len(nums) else -1
