class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        pre, suf = [0] * (n + 1), [0] * (n + 1)
        for i in range(n):
            pre[i + 1] = pre[i] + nums[i]
            suf[n - 1 - i] = suf[n - i] + nums[n - 1 - i]
        j = 0
        res = n + 1
        for i in range(n + 1):
            while j <= n and pre[i] + suf[j] > x:
                j += 1
            if i <= j <= n and pre[i] + suf[j] == x:
                print(i, j)
                res = min(res, i + n - j)
        return res if res <= n else -1
