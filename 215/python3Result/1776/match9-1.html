class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        array = nums + nums
        # print(array)
        accu = 0
        left = 0
        res = n + 1
        for i, num in enumerate(array):
            accu += num
            while accu > x:
                accu -= array[left]
                left += 1
            if accu == x:
                if left <= n and i >= n - 1:
                    res = min(res, i - left + 1)
        if res <= n:
            return res
        return -1
