class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        dic={0:-1}
        
        target=sum(nums)-x
        if target<=-1:
            return -1
        elif target==0:
            return len(nums)
        #print(len(nums))
        
        cur
=0
        res=-1
        for i,num in enumerate(nums):
            
            cur+=num
            
            if cur-target in dic:
                
                res=max(res,i-dic[cur-target])
            
            if cur not in dic:
                dic[cur]=i
                
        return -1 if res==-1 else len(nums)-res
