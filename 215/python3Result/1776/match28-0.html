class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        length = len(nums)
        left = 0
        right = length - 1
        
        total = 0
        INF = 987654321
        result = INF
        while left < length and total < x:
            total += nums[left]
            left += 1
        left -= 1
        
        if total == x:
            result = min(result, left + (length - right))
        while left >= 0:
            total -= nums[left]
            left -= 1
            while total < x and right > left:
                total += nums[right]
                right -= 1
            
            if total == x:
                result = min(result, left + (length - right))
        return result if result != INF else -1
