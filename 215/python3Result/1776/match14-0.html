class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        def lenOfLongSubarr(nums, k):
            n = len(nums)
            mydict = dict() 
            sum = 0
            maxLen = 0
            for i in range(n): 
                sum += nums[i] 
                if (sum == k): 
                    maxLen = i + 1
                elif (sum - k) in mydict: 
                    maxLen = max(maxLen, i - mydict[sum - k]) 
                if sum not in mydict: 
                    mydict[sum] = i 
            return maxLen 
        
        midLen = lenOfLongSubarr(nums, sum(nums) - x)
        # print(midLen)
        if midLen and midLen != len(nums):
            return len(nums) - midLen
        elif midLen == 0:
            if sum(nums) == x:
                return len(nums)
        return -1
