class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        left = 0
        tmp = 0
        ans = inf
        target = sum(nums) - x
        
        
for right in range(len(nums)):
            tmp += nums[right]
            
            while left <= right and tmp > target:
                tmp -= nums[left]
                left += 1
                
            if tmp == target:
                ans = min(ans, len(nums) - (right - left + 1))
                
        return -1 if ans == inf else ans
