class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        
        ## 中间 sum == sum_- x
        
        target = sum(nums) - x
        n = len(nums)
        if target < 0 : return -1
        elif target == 0 : return n
        l, r = 0, 0 
        cur = 0
        res = 0
        
while r < n:
            cur += nums[r]
            r += 1
            while cur > target:
                cur -= nums[l]
                l += 1
            if cur == target:
                res = max(res, r - l)
        # print(l,r,res)
        return n - res if res else - 1
