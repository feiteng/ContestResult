class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        n=len(nums)
        front=[0]*(n+1)
        back=[0]*(n+1)
        front[0]=0
        back[0]=0
        for i in range(1,n+1):
            front[i]=front[i-1]+nums[i-1]
            back[i]=back[i-1]+nums[n-i]
        result=[]
        j=n
        for i in range(n+1):
            while j>=1 and front[i]+back[j-1]>=x:
                j=j-1
            if front[i]+back[j]==x:
                if i+j>n:
                    continue
                if not result:
                    result.append(i+j)
                else:
                    result[0]=min(result[0],i+j)
        if not result:
            return -1
        return result[0]
