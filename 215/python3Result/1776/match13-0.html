class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        target = sum(nums) - x
        if target == 0:
            return len(nums)
        
        smallest = {0: 0}
        best = 0
        s = 0
        for i, x in enumerate(nums):
            s += x
            if s - target in smallest:
                best = max(best, i + 1 - smallest[s - target])
            
            if s not in smallest:
                smallest[s] = i + 1
            
        if best == 0:
            return -1
        
        return len(nums) - best
