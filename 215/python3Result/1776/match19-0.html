class Solution:
    # longest subarry with sum equal to k, time O(n), space O(1)
    def minOperations(self, nums: List[int], x: int) -> int:
        n = len(nums)
        k = sum(nums) - x
        start = end = 0
        cur_sum = 0
        max_len = -1
        for num in nums:
            cur_sum += num
            end += 1
            while start < n and cur_sum > k:
                cur_sum -= nums[start]
                start += 1
            if cur_sum == k:
                max_len = max(max_len, end - start)
        return n - max_len if max_len != -1 else -1
