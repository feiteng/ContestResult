class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        out = float('inf')
        s = sum(nums)
        if s <x:
            return -1
        cur_sum = 0
        cur_len = 0
        L = len(nums)
        target = s-x
        #print(target)
        for i,n in enumerate(nums):
            cur_sum +=n
            cur_len +=1
            while cur_sum > target:
                #print(cur_sum, nums[i-cur_len+1])
                cur_sum -= nums[i-cur_len+1]
                cur_len -=1
            #print("i=",i)
            #print("cur_len=", cur_len)
            if cur_sum == target:
                out = min(out, L-cur_len)
            #print(cur_sum)
        return out if out != float('inf') else -1
