class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        target = sum(nums) - x
        if target < 0:
            return -1
        if target == 0:
            return len(nums)
        l, r = 0, 0
        csum = 0
        ret = -1
        while r < len(nums):
            csum 
+= nums[r]
            while l < r and csum > target:
                csum -= nums[l]
                l += 1
            if csum == target:
                ret = max(ret, r - l + 1)
            r += 1
        return len(nums) - ret if ret != -1 else -1
