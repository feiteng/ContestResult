class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        #find the max length subarray with sum of total - x
        total = sum(nums)
        target = total - x
        if target == 0:
            return len(nums)
        
        res = -1
        left = 0
        
        acc = 0
        for right in range(len(nums)):
            acc += nums[right]
            
while left < right and acc > target:
                acc -= nums[left]
                left += 1
            if acc == target:
                res = max(res, right - left + 1)
            
        return len(nums) - res if res != -1 else -1
