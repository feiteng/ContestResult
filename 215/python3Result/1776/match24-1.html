class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        k = sum(nums) - x
        if k<0:
            return -1
        elif k==0:
            return len(nums)
        hmap 
= {}
        m = 0
        s = 0
        for i in range(len(nums)):
            s+=nums[i]
            if s == k:
                m = i+1
            elif s-k in hmap:
                m = max(m, i-hmap[s-k])
            if s not in hmap:
                hmap[s] = i
        if m==0:
            return -1
        return len(nums) - m
