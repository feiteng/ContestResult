class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
            tot=sum(nums)

            if tot<x:
                return -1

            
if tot==x:
                return len(nums)

            memo={0:-1}
            res=tot-x
            tmp=0
            ans=-1
            for i,j in enumerate(nums):
                tmp+=j

                if tmp-res in memo:
                    ans=max(ans,i-memo[tmp-res])


                memo[tmp]=i

            return len(nums)-ans if ans>=0 else -1
