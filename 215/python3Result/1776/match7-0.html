class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        
        # count = collections.defaultdict(list)
        n = len(nums)
        l = [0]*n
        r = [0]*n
        cl, cr = {}, {}
        ans = float('inf')
        cuml, cumr = 0, 0
        for i in range(len(nums)):
            cuml += nums[i]
            cumr += nums[n-1-i]
            l[i] = cuml
            r[n-1-i] = cumr
            cl[cuml] = i
            cr[cumr] = n-1-i
        # print(n, cl)
        # print(cr)
        for i in range(n):
            j = n-1-i
            if l[i] == x:
                ans = min(ans, i+1)
            if r[j] == x:
                ans = min(ans, i+1)
            if x-l[i] in cr and cr[x-l[i]] > i:
                # print(i, cr[x-l[i]])
                ans = min(ans, i+1 + n-cr[x-l[i]])
            if x-r[j] in cl and cl[x-r[j]] < j:
                # print(r[j], j, i, cl[x-r[j]])
                ans = min(ans, i+1 + cl[x-r[j]]+1)
        return ans if ans <= n else -1
