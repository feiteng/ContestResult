class Solution:
    def minOperations(self, nums: List[int], x: int) -> int:
        if x < min(nums) or x > sum(nums) : return -1
        n = len(nums)
        mn = float('inf')
        l,r,sl,sr,al,ar = {},{},0,0,[0 for _ in range(n)],[0 for _ in range(n)]
        for i in range
(n):
            sl += nums[i]
            sr += nums[n-i-1]
            al[i] = sl
            ar[n-i-1] = sr
            l[sl] = i
            r[sr] = n-i-1
        
        for i in range(n):
            
if al[i] == x: 
                mn = min(mn, i+1)
            if ar[i] == x:
                mn = min(mn,n-i)
            if x - al[i] in r and r[x - al[i]] > i: 
                mn = min(mn, i + 1 + n - r[x-al[i]])
            if x - ar[i] in l and l[x - ar[i]] < i: 
                mn = min(mn, l[x - ar[i]] + 1 + n - i )
        
        return -1 if mn == float('inf') else mn
