class Solution {
public:
    bool closeStrings(string word1, string word2) {
        int fr1[26] = {0};
        int fr2[26] = {0};
        int l1 = word1.length();
        int l2 = word2.length();

        for(int i = 0;i < l1;i ++){
            fr1[word1[i] - 'a'] ++;
        }
        for(int i = 0;i < l2;i ++){
            fr2[word2[i] - 'a'] ++;
        }
        for(int i = 0;i < 26;i ++){
            if(fr1[i] == 0 and fr2[i] != 0)
                return false;
            if(fr1[i] != 0 and fr2[i] == 0)
                return false;
        }
        sort(fr1,fr1 + 26);
        sort(fr2,fr2 + 26);
        for(int i = 0;i < 26;i ++){
            if(fr1[i] != fr2[i])
                return false;
        }
        return true;
    }
};
