class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if (word1.size() != word2.size()) return false;
        vector<int> n1(26), n2(26);
        for (int i = 0; i < word2.size(); i++) {
            n1[word1[i]-'a']++;
            n2[word2[i]-'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (n1[i] > 0 && n2[i] == 0) return false;
            if (n1[i] == 0 && n2[i] > 0) return false;
        }
        sort(n1.begin(), n1.end());
        sort(n2.begin(), n2.end());
        return n1 == n2;
    }
};
