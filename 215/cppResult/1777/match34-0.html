class Solution {
public:
    bool closeStrings(string word1, string word2) {
      vector<int> a(26, 0), b(26, 0);
      for (auto ch : word1) {
        a[ch - 'a'] += 1;
      }
      for (auto ch : word2) {
        b[ch - 'a'] += 1;
      }
      for (int i = 0; i < 26; i++) {
        if ((a[i] == 0 && b[i] > 0) || (a[i] > 0 && b[i] == 0)) {
          return false;
        }
      }
      sort(a.begin(), a.end());
      sort(b.begin(), b.end());
      
      for (int i = 0; i < 26; i++) {
        if (a[i] != b[i]) {
          return false;
        }
      }
      return true;
    }
};
