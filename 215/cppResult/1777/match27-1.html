class Solution {
public:
    bool closeStrings(string word1, string word2) {

        vector<int> A(26,0), B(26,0);
        for(char c:word1)
            A[c-'a']++;
        for(char c:word2)
            B[c-'a']++;
        
        for(int i=25; i>=0; i--){
            if((A[i] != 0 && B[i] ==0) || (A[i] == 0 && B[i] !=0))
                return false;
        }
        sort(A.begin(),A.end());
        sort(B.begin(),B.end());
        for(int i=25; i>=0; i--){
            if(A[i] != B[i])
                return false;
        }
        return true;
    }
};
