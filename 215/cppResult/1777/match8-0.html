class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> freqs1(26,0), freqs2(26,0);
        for (char c: word1){
            freqs1[c-'a']++;
        }
        for (char c: word2){
            freqs2[c-'a']++;
        }
        for (int i=0; i<26; ++i){
            if ((freqs1[i]!=0&&freqs2[i]==0) || (freqs1[i]==0&&freqs2[i]!=0))
                return false;
        }
        sort(freqs1.begin(), freqs1.end());
        sort(freqs2.begin(), freqs2.end());
        for (int i=0; i<26; ++i){
            if (freqs1[i]!=freqs2[i])
                return false;
        }
        return true;
    }
};
