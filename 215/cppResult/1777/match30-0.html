class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if(word1.size()!=word2.size()){
            return false; 
        }
        vector<int> a(26);
        vector<int> b(26);
        for(char ch : word1) {
            a[ch-'a']++;
        }
        for(char ch : word2){
            b[ch-'a']++;
        }
        for(int i = 0;i<26;i++) {
            if((a[i]&&!b[i])||(b[i]&&!a[i])){
                return false; 
            }
        }
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        return a == b; 
    }
};
