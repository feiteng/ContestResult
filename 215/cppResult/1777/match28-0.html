class Solution {
public:
    bool closeStrings(string word1, string word2) {
        int a[26] = {};
        int b[26] = {};
        for(char c: word1) {
            a[c-'a']++;
        }
        for(char c: word2) {
            b[c-'a']++;
        }
        for(int i=0; i<26; i++) {
            if((not a[i] and b[i]) or (a[i] and not b[i]))
                return false;
        }
        
        sort(a, a+26);
        sort(b, b+26);
        for(int i=0; i<26;i++) {
            if(a[i] != b[i])
                return false;
        }
        return true;
    }
};
