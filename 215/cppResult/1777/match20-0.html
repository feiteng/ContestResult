class Solution {
public:
    bool closeStrings(string word1, string word2) {
        int xx[27]={0};
        int xy[27]={0};
        for(int i=0;i<word1.size();i++) {
            xx[word1[i]-'a']++;
        }
        for(int i=0;i<word2.size();i++) {
            xy[word2[i]-'a']++;
        }
        // 判断a b里是否现有字符不同
        for(int i=0;i<26;i++) {
            if(xx[i]>0&&xy[i]==0) return false;
            if(xy[i]>0&&xx[i]==0) return false;
        }
        sort(xx,xx+26);
        sort(xy,xy+26);
        for(int i=0;i<26;i++) {
            if(xx[i]!=xy[i]) return false;
        }
        return true;
    }
};
