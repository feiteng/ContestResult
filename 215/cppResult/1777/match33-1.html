class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> num1(26), num2(26);
        set<int> st1, st2;
        for(auto c: word1) {
            num1[c-'a']++;
            st1.insert(c);
        }
        for(auto c: word2) {
            num2[c-'a']++;
            st2.insert(c);
        }
        sort(begin(num1), end(num1));
        sort(begin(num2), end(num2));
        for(int i = 0; i < 26; i++) {
            if(num1[i] != num2[i]) {
                return false;
            }
        }
        return st1 == st2;
    }
};
