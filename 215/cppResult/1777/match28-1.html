class Solution {
public:
    bool closeStrings(string word1, string word2) {
        const int N = size(word1);
        if (size(word2) != N) return false;

        int cnt1[26] = {};
        int cnt2[26] = {};
        for (char ch : word1) {
            ++cnt1[ch-'a'];
        }
        for (char ch : word2) {
            ++cnt2[ch-'a'];
        }
        for (int i = 0; i < 26; ++i) {
           if (cnt1[i] && !cnt2[i] || cnt2[i] && !cnt1[i]) return false;
        }
        sort(begin(cnt1), end(cnt1));
        sort(begin(cnt2), end(cnt2));
        for (int i = 0; i < 26; ++i) {
            if (cnt1[i] != cnt2[i]) return false;
        }
        return true;
    }
};
