class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if(word1.length() != word2.length()) return false;
        vector<int> dp1(26);

        vector<int> dp2(26);
        for(auto p : word1){
            dp1[p - 'a']++;
        }
        for(auto p : word2){
            dp2[p - 'a']++;
        }
        for(int i = 0; i < 26; i++){
            if(dp1[i] > 0 && dp2[i] == 0){
                return false;
            }
            if(dp1[i] == 0 && dp2[i] > 0){
                return false;
            }

        }
        sort(dp1.begin(), dp1.end());
        sort(dp2.begin(), dp2.end());
        for(int i = 0; i < 26; i++){
            if(dp1[i] != dp2[i]){
                return false;
            }
        }
        return true;
    }
};
