class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> c1(26);
        vector<int> c2(26);
        for (auto & v : word1) c1[v - 'a']++;
        for (auto & v : word2) c2[v - 'a']++;
        for (int i = 0; i < 26; ++i) {
            if (c1[i] != 0 && c2[i] == 0) return false;
            if (c2[i] != 0 && c1[i] == 0) return false;

        }
        sort(c1.begin(), c1.end());
        sort(c2.begin(), c2.end());
        
        for (int k = 0; k < 26; ++k) {
            if (c1[k] != c2[k]) return false;
        }
        return true;
    }
};
