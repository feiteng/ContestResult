class Solution {
public:
    bool closeStrings(string word1, string word2) {
        int size1 = word1.size();

        int size2 = word2.size();
        if(size1 != size2) {
            return false;
        }
        vector<int> v1(26, 0);
        vector<int> v2(26, 0);
        for(int i = 0; i < size1; ++i) {
            ++v1[word1[i]-'a'];
            ++v2[word2[i]-'a'];
        }
        for(int i = 0; i < 26; ++i) {
            if(v1[i] != 0 && v2[i] == 0) {
                return false;
            }

            if(v1[i] == 0 && v2[i] != 0) {
                return false;
            }
            
        }
        sort(v1.begin(), v1.end());
        sort(v2.begin(), v2.end());
        for(int i = 0; i < 26; ++i) {
            if(v1[i] != v2[i]) {
                return false;
            }
        }
        return true;
    }
};
