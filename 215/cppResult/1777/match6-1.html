class Solution {
public:
    bool closeStrings(string w1, string w2) {
        if (w1.length() != w2.length()) {
            return false;
        }
        vector<int> c1(26, 0), c2(26, 0);
        for (auto &c : w1) {
            c1[c - 'a']++;
        }
        for (auto &c : w2) {
            c2[c - 'a']++;
        }
        
        for (int i = 0; i < 26; ++i) {
            if (c1[i] == 0 ^ c2[i] == 0) {
                return false;
            }
        }
        
        sort(c1.begin(), c1.end());
        sort(c2.begin(), c2.end());
        
        for (int i = 0; i < 26; ++i) {
            if (c1[i] != c2[i]) {
                return false;
            }
        }
        return true;
    }
};
