class Solution {
public:
    bool closeStrings(string wd1, string wd2) {
        int n1 = wd1.size(), n2 = wd2.size();
        if(n1 != n2) return false;
        vector<int> c1(26, 0), c2(26, 0);
        for(int i = 0; i < n1; i++){
            c1[wd1[i]-'a']++;
            c2[wd2[i]-'a']++;
        }
        for(int i = 0; i < c1.size(); i++){
            if((c1[i] == 0 && c2[i] != 0) || (c1[i] != 0 && c2[i] == 0))
                return false;
        }
        sort(c1.begin(), c1.end());
        sort(c2.begin(), c2.end());
        for(int i = 0; i < c1.size(); i++){
            if(c1[i] != c2[i])
                return false;
        }
        return true;
    }
};
