class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> c1(26), c2(26);
        for (char c : word1)
            c1[c-'a'] += 1;
        for (char c : word2)
            c2[c-'a'] += 1;
        for (int i = 0; i < 26; ++i)
            if (!c1[i] != !c2[i])
                return false;
        sort(c1.begin(), c1.end());
        sort(c2.begin(), c2.end());
        return c1 == c2;
    }
};
