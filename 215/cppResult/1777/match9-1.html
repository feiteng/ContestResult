class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> charCnt1(26, 0), charCnt2(26, 0);
        for (char c : word1)
            charCnt1[c - 'a']++;
        for (char c : word2)
            charCnt2[c - 'a']++;
        for (int i = 0; i < 26; i++)
            if (charCnt1[i] && !charCnt2[i] || !charCnt1[i] && charCnt2[i])
                return false;
        
        sort(charCnt1.begin(), charCnt1.end());
        sort(charCnt2.begin(), charCnt2.end());
        
        return charCnt1 == charCnt2;
    }
};
