class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if (word1.length() != word2.length()) 
            return false;
        
        vector<int> c1(26, 0);
        vector<int> c2(26, 0);
        
        for (int i=0; i<word1.length(); ++i) {
            c1[word1[i] - 'a']++;
            c2[word2[i] - 'a']++;
        }
        
        for (int k=0; k<26; ++k) {
            if (c1[k] == c2[k]) {
                c1[k] = 0;
                c2[k] = 0;
                continue;
            }
            if (c1[k]==0 || c2[k]==0) {
                return false;
            }
        }
        sort(c1.begin(), c1.end());
        sort(c2.begin(), c2.end());
        
        for (int k=0; k<26; ++k) {
            if (c1[k] != c2[k]) 
                return false;
        }
        return true;
    }
};
