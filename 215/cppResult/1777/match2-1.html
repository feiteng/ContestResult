class Solution {
public:
    bool closeStrings(string word1, string word2) {
        map<char,int> a,b;
        for(char i:word1)a[i]=a[i]+1;
        for(char i:word2)b[i]=b[i]+1;
        vector<int> aa,bb;
        vector<char> ta,tb;
        for(auto i:a)aa.push_back(i.second),ta.push_back(i.first);
        for(auto i:b)bb.push_back(i.second),tb.push_back(i.first);
        sort(aa.begin(),aa.end());
        sort(bb.begin(),bb.end());
        sort(ta.begin(),ta.end());
        sort(tb.begin(),tb.end());
        return aa==bb && ta==tb;
    }
};
