class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if (word1.size() != word2.size()) return false;
        vector<int> map1(26), map2(26);
        for (int i = 0; i < word1.size(); ++i) {
            ++map1[word1[i]-'a'];
            ++map2[word2[i]-'a'];
        }
        vector<int> cnts1, cnts2;
        for (int i = 0; i < 26; ++i) {
            if ((map1[i]>0 && map2[i]==0) || (map1[i]==0 && map2[i]>0)) return false;
            cnts1.push_back(map1[i]);
            cnts2.push_back(map2[i]);
        }
        sort(cnts1.begin(), cnts1.end());
        sort(cnts2.begin(), cnts2.end());
        for (int i = 0; i < cnts1.size(); ++i) {
            if (cnts1[i] != cnts2[i]) return false;
        }
        return true;
    }
};
