class Solution {
public:
    bool closeStrings(string word1, string word2) {
        
        int size1 = word1.size();
        int size2 = word2.size();
        
        if (size1 != size2) {
            return false;
        }
        
        vector<int> counts1(26, 0);
        vector<int> counts2(26, 0);
        
        for (int i = 0; i < size1; ++i) {
            counts1[word1[i] - 'a']++;
            counts2[word2[i] - 'a']++;
        }
        
        vector<int> freq1, freq2;
        
        for (int i = 0; i < 26; ++i) {
            if (counts1[i] == 0 && counts2[i] != 0) {
                return false;
            }
            if (counts1[i] != 0 && counts2[i] == 0) {
                return false;
            }
            freq1.push_back(counts1[i]);
            freq2.push_back(counts2[i]);
        }
        
        sort(freq1.begin(), freq1.end());
        sort(freq2.begin(), freq2.end());
        
        for (int i = 0; i < 26; ++i) {
            if (freq1[i] != freq2[i]) {
                return false;
            }
        }
        
        return true;
    }
};
