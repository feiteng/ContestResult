class Solution {
public:
    bool closeStrings(string word1, string word2) {
        vector<int> tab1(26);
        vector<int> tab2(26);
        for (char c: word1) 
            tab1[c-'a']++;
        for (char c: word2) 
            tab2[c-'a']++;
        for (int i=0; i<26; i++) {
            if ((tab1[i]==0 && tab2[i]!=0) || (tab1[i]!=0 && tab2[i]==0))
                return false;
        }
        sort(tab1.begin(), tab1.end());
        sort(tab2.begin(), tab2.end());
        
        for (int i=0; i<26; i++) {
            if (tab1[i]!=tab2[i])
                return false;
        }
        return true;
    }
};
