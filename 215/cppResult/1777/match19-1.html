class Solution {
public:
    bool closeStrings(string word1, string word2) {
        if(word1.length() != word2.length())
            return false;
        vector<int> freq1(26,0), freq2(26,0);
        vector<bool> occr1(26,false), occr2(26,false);
        for(int i = 0;i < word1.length();i++){
            freq1[word1[i] - 'a']++;
            freq2[word2[i] - 'a']++;

            occr1[word1[i] - 'a'] = true;
            occr2[word2[i] - 'a'] = true;
        }
        
        sort(freq1.begin(),freq1.end());
        sort(freq2.begin(),freq2.end());
        
        for(int i = 0;i < 26;i++){
            if(freq1[i] != freq2[i])
                return false;
            if(occr1[i] != occr2[i])
                return false;
        }
        return true;
    }
};
