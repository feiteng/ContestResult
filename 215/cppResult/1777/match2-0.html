class Solution {
public:
    bool closeStrings(string word1, string word2) {
        map<char,int>p1,p2;
        for(char c : word1)p1[c]++;
        for(char c : word2)p2[c]++;
        
        vector<int>l1,l2;
        set<char>s1,s2;
        for(auto i : p1)l1.push_back(i.second),s1.insert(i.first);
        for(auto i : p2)l2.push_back(i.second),s2.insert(i.first);
        sort(l1.begin(),l1.end());
        sort(l2.begin(),l2.end());
        return l1 == l2 and s1 == s2;
    }
};
