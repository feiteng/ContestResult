class Solution {
public:
    bool closeStrings(string word1, string word2) {
         if (word1.length() != word2.length()) {
            return false;
        }
        int n = word1.length();
        if (n == 0) {
            return true;
        }
        vector<int> cnt1(26, 0);
        vector<int> cnt2(26, 0);
        for (int i = 0; i < n; i++) {
            cnt1[word1.at(i) - 'a']++;
            cnt2[word2.at(i) - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (cnt1[i] == 0 && cnt2[i] > 0) {
                return false;
            } else if (cnt1[i] > 0 && cnt2[i] == 0) {
                return false;
            }
        }
        sort(cnt1.begin(),cnt1.end());
        sort(cnt2.begin(),cnt2.end());
        for (int i = 0; i < 26; i++) {
            if (cnt1[i] != cnt2[i]) {
                return false;
            }
        }
        return true;
    }
};
