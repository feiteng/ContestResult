class Solution {
public:
    int dp[6][7][7][250],A[5],B[5];
    int getMaxGridHappiness(int n, int m, int in, int ex) {
        memset(dp,-0x3f,sizeof(dp));
        dp[0][0][0][0] = 0;
        int tp = 1;
        int ans = 0;

        for(int i=0;i<m;i++)tp *= 3;
        for(int i=1;i<=n;i++){
            for(int j=0;j<=in;j++){
                for(int k=0;k<=ex;k++){
                    for(int l=0;l<tp;l++)if(dp[i-1][j][k][l]>=0){
                        for(int p=0;p<tp;p++){
                            int _l=l,_p=p,_1=0,_2=0;
                            for(int x=0;x<m;x++){
                                A[x] = _l%3; _l/=3;
                                B[x] = _p%3; _p/=3;
                                if(B[x]==1)_1++;
                                if(B[x]==2)_2++;
                            }
                            if(_1+j>in||_2+k>ex)continue;
                            int v = 0;
                            for(int x=0;x<m;x++){
                                if(!B[x])continue;
                                if(B[x]==1){
                                    v += 120;
                                    if(A[x]==1)v-=60;

                                    else if(A[x]==2)v-=10;
                                    if(x>0&&B[x-1]==1)v-=60;
                                    if(x>0&&B[x-1]==2)v-=10;
                                }
                                if(B[x]==2){
                                    v += 40;
                                    if(A[x]==1)v-=10;

                                    else if(A[x]==2)v+=40;
                                    if(x>0&&B[x-1]==1)v-=10;
                                    if(x>0&&B[x-1]==2)v+=40;
                                }
                            }
                            dp[i][j+_1][k+_2][p] = max(dp[i][j+_1][k+_2][p],dp[i-1][j][k][l]+v);
                        }
                    }
                }
            }
        }

        for(int i=0;i<=in;i++)for(int j=0;j<=ex;j++)for(int k=0;k<tp;k++)ans = max(ans,dp[n][i][j][k]);
        return ans;
    }
};
