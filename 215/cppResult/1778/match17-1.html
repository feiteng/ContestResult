class Solution {
    int ii,ee;
    int m,n,mn;
    int room[25];
    int ans=0;
public:
    void fun(int cur,int _score){
        if(cur==mn)return;
        if(ii){
            int score=_score;
            room[cur]=1;

            score+=120;
            if(cur-n>=0){
                if(room[cur-n]==1)score-=60;
                else if(room[cur-n]==2)score-=10;
            }
            if(cur&&cur%n){
                if(room[cur-1]==1)score-=60;
                else if(room[cur-1]==2)score-=10;
            }
            ans=max(ans,score);
            ii--;
            fun(cur+1,score);
            ii++;
        }
        if(ee){
            int score=_score;
            room[cur]=2;

            score+=40;
            if(cur-n>=0){
                if(room[cur-n]==1)score-=10;
                else if(room[cur-n]==2)score+=40;
            }
            if(cur&&cur%n){
                if(room[cur-1]==1)score-=10;
                else if(room[cur-1]==2)score+=40;
            }
            ans=max(ans,score);
            ee--;
            fun(cur+1,score);
            ee++;
        }
        {
            int score=_score;
            room[cur]=0;
            fun(cur+1,score);
        }
    }
    int getMaxGridHappiness(int _m, int _n, int introvertsCount, int extrovertsCount) {
        ii=introvertsCount,ee=extrovertsCount;
        m=_m,n=_n;
        mn=m*n;
        int i,j,k;
        
        if(mn>16){
            int aa[7]={0,40,120,200,320,400,520};
            ans=120*ii+aa[ee];
            if(mn==20&&ii%2==0&&(ii>5&&ee>5))ans-=10;
        }
        else fun(0,0);
        return ans;
    }
};
