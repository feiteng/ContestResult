class OrderedStream {
public:
    OrderedStream(int n) {
        strs.resize(n + 1);
        ptr = 0;
    }
    
    vector<string> insert(int id, string value) {
        vector<string> ans;
        strs[id-1] = value;
        while (!strs[ptr].empty())  {
            ans.push_back(strs[ptr]);
            ptr ++;
        }
        return ans;
    }

private:
    vector<string> strs;
    int ptr;
};

 * OrderedStream* obj = new OrderedStream(n);
 * vector<string> param_1 = obj->insert(id,value);
