class OrderedStream {
public:
    vector<string> v;
    int ptr=0;
    OrderedStream(int n) {
        v = vector<string>(n,"");
    }
    
    vector<string> insert(int id, string value) {
        vector<string> res;
        v[id-1]=value;
        while (ptr<S(v)&&v[ptr]!="") {
            res.emplace_back(v[ptr]);
            ptr++;
        }
        return res;
    }
};
