class OrderedStream
{
public:
    vector<string> v;
    int p;
    OrderedStream(int n)
    {
        v.resize(n); p = 0;
    }
    
    vector<string> insert(int id, string value)
    {
        vector<string> res;
        v[id - 1] = value;
        while (p < v.size() && v[p] != "")
        {
            res.push_back(v[p++]);
        }
        return res;
    }
};
