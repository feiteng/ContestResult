class OrderedStream {
public:
    OrderedStream(int n) {
        stat.resize(n + 1, "special");
        ptr = 1;
    }
    
    vector<string> insert(int id, string value) {
        vector<string> res;
        stat[id] = value;
        while (ptr < stat.size() && stat[ptr] != "special") {
            res.push_back(stat[ptr]);
            ++ptr;
        }
        return res;
    }
    
    int ptr;
    vector<string> stat;
};

 * OrderedStream* obj = new OrderedStream(n);
