class OrderedStream {
public:
    vector<string> str;
    int ptr;
    OrderedStream(int n) {
        str = vector<string>(n);
        ptr = 0;
    }
    
    vector<string> insert(int id, string value) {
        str[id-1] = value;
        vector<string> ret;
        while(ptr < str.size() && !str[ptr].empty()) {
            ret.push_back(str[ptr++]);
        }
        return ret;
    }
};
