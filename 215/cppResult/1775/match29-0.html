class OrderedStream {
public:
    vector<string> a;
    vector<bool> f;
    int ptr;
    OrderedStream(int n) {
        a.assign(n, "");
        f.assign(n, false);
        ptr = 0;
    }
    
    vector<string> insert(int id, string value) {
        a[id - 1] = value;
        f[id - 1] = true;
        vector<string> ans;
        while(ptr < a.size() && f[ptr]) {
            ans.push_back(a[ptr]);
            ptr++;
        }
        return ans;
    }
};
