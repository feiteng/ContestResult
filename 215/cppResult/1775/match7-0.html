class OrderedStream {
public:
    vector<string> arr;
    int cur;
    OrderedStream(int n) {
        arr.resize(n, "");
        cur = 0;
    }

    vector<string> insert(int id, string value) {
        vector<string> ret;
        arr[id - 1] = value;
        while (arr[cur] != "" && cur < arr.size()) {
            ret.push_back(arr[cur]);
            cur++;
        }
        return ret;
    }
};
