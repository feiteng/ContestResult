class OrderedStream {
    vector<string> v;
    int n;
    int ptr;
public:
    OrderedStream(int nn) {
        ptr = 1;
        n = nn;
        v.resize(n + 2);
    }
    
    vector<string> insert(int id, string value) {
        v[id] = value;
        vector<string> ret;
        while (v[ptr] != "") {
            ret.push_back(v[ptr]);
            ptr++;
        }
        return ret;
    }
};
