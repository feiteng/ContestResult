class OrderedStream {
public:
    int ptr;
    vector<string> v;
    OrderedStream(int n) {
        ptr=1;
        v.resize(1005,"");
    }
    
    vector<string> insert(int id, string value) {
        v[id]=value;
        vector<string> ans;
        while(v[ptr]!=""){
            ans.push_back(v[ptr]);
            ptr++;
        }
        return ans;
    }
};
