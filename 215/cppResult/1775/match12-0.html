class OrderedStream {
public:
    vector<string> bank;
    int op;
    OrderedStream(int n) {
        bank = vector<string>(n+1,"");
        op=1;
    }
    
    vector<string> insert(int id, string value) {
        vector<string> res;
        bank[id] = value;
        while(op<bank.size()&&bank[op]!="")
        {
            res.emplace_back(bank[op]);
            op++;
        }
        return res;
    }
};

 * OrderedStream* obj = new OrderedStream(n);
 * vector<string> param_1 = obj->insert(id,value);
