class OrderedStream {
public:
    int ptr;
    vector<string> vs;
    OrderedStream(int n) {
        ptr = 1;
        vs.resize(n + 2);
    }
    
    vector<string> insert(int id, string value) {
        vs[id] = value;
        vector<string> res;
        while(not vs[ptr].empty()){
            res.push_back(vs[ptr]);
            ptr += 1;
        }
        return res;
    }
};

 * OrderedStream* obj = new OrderedStream(n);
