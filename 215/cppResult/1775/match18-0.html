class OrderedStream {
public:
    int ptr;
    vector <string> v;
    
    OrderedStream(int n) {
        ptr = 1;
        v.resize(n + 5);
    }
    
    vector<string> insert(int id, string value) {
        vector <string> ret;
        v[id] = value;
        while (v[ptr] != "") {
            ptr++;
            ret.push_back(v[ptr - 1]);
        }
        return ret;
    }
};
