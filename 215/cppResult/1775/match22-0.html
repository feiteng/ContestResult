class OrderedStream {
public:
    vector<string> vec;
    int ptr;
    OrderedStream(int n) {
        vec.clear();
        vec.resize(n);
        ptr = 0;
    }
    
    vector<string> insert(int id, string value) {
        vec[id - 1] = value;
        vector<string> ret;
        while (ptr < vec.size() && !vec[ptr].empty()) {
            ret.push_back(vec[ptr]);
            ptr++;
        }
        return ret;
    }
};
