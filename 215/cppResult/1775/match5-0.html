class OrderedStream {
public:
    vector <string> words;
    int ptr;
    OrderedStream(int n) {
        words.resize(n + 5, "$");
        ptr = 1;
    }
    
    vector<string> insert(int id, string value) {
        words[id] = value;
        vector <string> res;
        while(words[ptr] != "$") {
            res.push_back(words[ptr++]);
        }
        return res;
    }
};
