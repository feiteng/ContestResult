class OrderedStream {
public:
    string val[1010];
    int ptr;
    OrderedStream(int n) {
        ptr = 1;
        fill(val, val+n, "");
    }
    
    vector<string> insert(int id, string value) {
        val[id] = value;
        vector<string> ret;
        while(val[ptr]!=""){
            ret.push_back(val[ptr]);
            ++ptr;
        }
        return ret;
    }
};
