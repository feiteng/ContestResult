class OrderedStream {
public:
    vector<string> s;
    int ptr;

    OrderedStream(int n) {
        s.resize(n);
        ptr = 0;
    }

    vector<string> insert(int id, string value) {
        s[id - 1] = value;
        vector<string> res;
        while (ptr < s.size() && s[ptr] != "")
        {
            res.push_back(s[ptr]);
            ++ptr;
        }
        return res;
    }
};
