class OrderedStream {
public:
    vector<string> arr;
    int pointer;
    OrderedStream(int n) {
        arr.resize(n, "");
        pointer = 0;
    }
    
    vector<string> insert(int id, string value) {
        arr[id-1] = value;
        vector<string> ans;
        while(pointer < arr.size() && arr[pointer] != ""){
            ans.push_back(arr[pointer]);
            pointer++;
        }
        return ans;
    }
};
