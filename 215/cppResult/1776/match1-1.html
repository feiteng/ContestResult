class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        int ans = 1e9;
        int lsum = 0, r = n - 1;
        while (r >= 0 && lsum < x) {
            lsum += nums[r];
            r--;
        }
        if (lsum == x) {
            ans = min(ans, (n - 1) - r);
        }
        if (lsum < x) {
            return -1;

        }
        for (int l = 1; l < n; l++) {
            //cout << l << lsum << r << endl;
            x -= nums[l - 1];
            while (r < n - 1 && lsum > x) {
                lsum -= nums[r + 1];
                r++;
            }
            if (lsum == x) {
                ans = min(ans, l + (n - 1) - r);
            }
        }
        if (ans == 1e9) return -1;
        return ans;
    }
};
