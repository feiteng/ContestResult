class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> sum(n + 1, 0);   
        unordered_map<int, int> pos;
        for(int i = 1; i <= n; ++i) {
            sum[i] = sum[i - 1] + nums[i - 1];
        } 
        if(sum[n] < x) return -1;
        for(int i = 0; i <= n; ++i) {
            pos[sum[n] - sum[i]] = i;
        }
        int ans = INT_MAX;
        for(int i = 0; i <= n; ++i) {
            if(pos.count(x - sum[i]) == 0) continue;
            ans = min(ans, i + n - pos[x - sum[i]]);    
        } 
        if(ans == INT_MAX) return -1;
        return ans;
    }
};
