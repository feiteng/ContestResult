class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n=nums.size();
        int ret=n+1;
        vector<int> sum(n+1,0);

        for (int i=1;i<=n;i++)
        {
            sum[i]=nums[i-1]+sum[i-1];
            if (sum[i]==x && nums[i-1]!=0) ret = i;
        }
        
        
        int now=0;
        int cnt=0;
        for (int i=n-1;i>=0;i--)
        {
            cnt++;
            now+=nums[i];
            int l = lower_bound(sum.begin(),sum.begin()+i+1, x-now) - sum.begin();
            //cout<<sum[l]<<' '<<l<<endl;
            //cout<<l+1<<' '<<cnt<<endl;
            //cout<<"----\n";
            if (sum[l]==x-now && l+cnt<ret)
            {
                ret=l+cnt;
            }
        }
        if (ret==n+1) return -1;
        else return ret;
    }
};
