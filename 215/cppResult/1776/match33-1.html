class Solution {
public:
    
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> left(n+1,0);
        vector<int> right(n+1,0);
        for(int i = 1;i<=n;i++)
            left[i] +=left[i-1]+nums[i-1];
        for(int i = 1;i<=n;i++)

            right[i] += right[i-1] + nums[n-i];
        int res = INT_MAX;
        for(int i = 0;i<=n;i++){
            int temp = x -left[i];
            int idx = lower_bound(right.begin(),right.end(),temp) - right.begin();
            if( idx +i<=n && right[idx] == temp){
                res = min(res,idx +i );
            }
        }
        return res == INT_MAX? -1: res;
    }
};
