class Solution {
public:
    int minOperations(vector<int>& A, int x) {
        unordered_map<int, int> S;
        S[0] = 0;
        int s = 0;
        int n = A.size();
        
        int ans = n + 1;
        
        for (int i = 0; i < n; i++) {
            s += A[i];
            S[s] = i + 1;
            if (s == x) ans = min(ans, i+1);
        }
        
        s= 0;

        for (int i = n - 1; i >= 0; i--) {
            s += A[i];
            
            if (s <= x && S.count(x - s) > 0) {
                int y = S[x-s];
                if (y - 1 >= i) continue;
                
                ans = min(ans, n - i + y);
            }
        }
        if (ans <=n) return ans;
        else return -1;
    }
};
