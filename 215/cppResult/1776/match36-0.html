class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int Z=333444555;
        int sm=accumulate(nums.begin(),nums.end(),0);
        unordered_map<int,int> mm;
        mm[0]=0;
        int ps=0;
        for(int i=1;i<=nums.size();i++) {
            ps+=nums[i-1];
            int tg=x-sm+ps;
            if(!mm.count(ps)) mm[ps]=i;
            if(mm.count(tg)) Z=min(Z,mm[tg]+int(nums.size())-i);
        }
        if(Z>nums.size()) Z=-1;
        return Z;
    }
};
