class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
		int ret=INT_MAX;
		int n=nums.size();
		int l=0,r=nums.size()-1;
		while(r>=0&&x-nums[r]>=0){
			x-=nums[r--];
		}
		if(r==-1){
			if(x==0)return n;
			return -1;
		}
		while(x-nums[l]>=0){
			x-=nums[l++];
		}
		if(x==0){
			ret=min(ret,n-(r-l+1));
		}
		while(r<nums.size()-1){
			x+=nums[++r];
			while(x-nums[l]>=0){
				x-=nums[l++];
			}
			if(x==0){
				ret=min(ret,n-(r-l+1));
			}
		}
		if(ret==INT_MAX)return -1;
		return ret;

	}
};
