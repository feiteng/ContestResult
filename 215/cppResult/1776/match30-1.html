class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int sum=accumulate(nums.begin(),nums.end(),0);
        if(x-sum>0)return -1;
        int n=nums.size();
        vector<int> pref(n,0);
        pref[0]=nums[0];
        for(int i=1;i<n;i++)pref[i]=pref[i-1]+nums[i];
        int ans=n+1;

        for(int i=0;i<n;i++)if(x==pref[i])ans=min(ans,i+1);
        int cur=x;
        for(int i=n-1;i>0;i--)
        {
            cur-=nums[i];
            if(cur==0){ans=min(ans,n-i);break;}
            if(cur<0)break;
            int l=0,r=i-1;
            int ans1=r+1;
            while(l<=r)
            {
                int mid=(l+r)>>1;
                if(cur>pref[mid])
                {
                    l=mid+1;
                }
                else if(cur<pref[mid])
                {
                    r=mid-1;
                }
                else
                {
                     ans1=mid+1;
                     break;
                }
            }
            if(ans1!=i)
            {
                 ans=min(ans,ans1+n-i);
            }
        }
        if(ans==n+1)return -1;
        return ans;
    }
};
