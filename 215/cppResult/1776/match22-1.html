class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
      int n = nums.size();

      int tot = accumulate(nums.begin(), nums.end(), 0);
      
      map<int, int> mp;
      mp[0] = 0;
      int sum = 0;
      for (int i = 0; i < n; i++) {
        sum += nums[i];
        mp[sum] = i+1;
      }
      

      int res = n+10;
      if (mp.count(x)) res = min(res, mp[x]);
      
      sum = 0;
      for (int i = n-1; i >= 0; i--) {
        sum += nums[i];
        if (mp.count(x-sum)) {
          res = min(res, mp[x-sum] + (n-i));
        }
      }
      
      if (res > n) res = -1;
      return res;
    }
};
