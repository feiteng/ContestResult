class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int,int> suf;
        suf[0] = nums.size();

        int sum = 0;
        for(int i = nums.size()-1;i >= 0;i--){
            sum += nums[i];
            if (suf.find(sum) == suf.end()){
                suf[sum] = i;
            }
        }
        int ret = nums.size() + 1;
        
        if (sum == x)return nums.size();

        if (suf.find(x) != suf.end())ret = (int)nums.size() - suf[x];
        
        sum = 0;
        for(int i = 0;i < nums.size();i++){
            sum += nums[i];
            if (suf.find(x-sum) != suf.end() && suf[x-sum] > i){
                ret = min(ret, i+1 + (int)nums.size() - suf[x-sum]);
            }
        }
        
        return ret == nums.size() + 1 ? -1 : ret;
    }
};
