class Solution {
public:
    
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> l(1, 0), r(1, 0);
        unordered_map<int,int> mp;
        int cur = 0, i = 1;
        mp[0] = 0;
        for(int x : nums) {
            cur += x;
            l.push_back(cur);
            mp[cur] = i;
            i++;
        }
        cur = 0;

        for(int i = n-1; i >=0; --i) {
            cur+= nums[i];
            r.push_back(cur);
        }
        reverse(r.begin(), r.end());
        
        int res = INT_MAX;
        for(int i = n; i >=0; --i) {
            int t = x - r[i];
            if( mp.count(t) > 0 && mp[t] <= i) {
                res = min(res, n-i + mp[t]);
            }
        }
        
        return res == INT_MAX ? -1 : res;
    }
};
