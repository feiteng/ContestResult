class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int r = nums.size();
        int lsum = 0, rsum = 0;
        while(r>0&&rsum+nums[r-1]<=x) {
            r--;
            rsum += nums[r];
        }
        int ans = INT_MAX;
        if (rsum == x) ans = nums.size()-r; 
        for (int l = 0; l < nums.size(); l++) {
            while(r<=l) {
                rsum -= nums[r];
                r++;
            }

            lsum += nums[l];
            while(r<nums.size()&&lsum+rsum>x) {
                rsum -= nums[r];
                r++;
            }
            if (lsum+rsum==x) {
                ans = min(ans,l+1+(int)nums.size()-r);
            }
        }
        return ans==INT_MAX?-1:ans;
    }
};
