class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int dp1[100005],dp2[100005];
        dp1[0] = 0;
        dp2[0] = 0;
        for(int i=1;i<=nums.size();i++)
        {
            dp1[i] = dp1[i-1]+nums[i-1];
            dp2[i] = dp2[i-1]+nums[nums.size()-i];
        }
        if(dp1[nums.size()]<x)
            return -1;
        int ans = 100005;
        for(int i=0;i<=nums.size();i++)
        {
            if(dp1[i]>x)
                break;
            int k = lower_bound(dp2,dp2+nums.size()+1,x-dp1[i])-dp2;
            if(k!=nums.size()+1&&dp2[k]==x-dp1[i])
            {
                ans = min(i+k,ans);
            }
        }
        if(ans==100005)
            return -1;
        return ans;
    }
};
