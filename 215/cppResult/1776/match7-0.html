class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int target = -x;
        vector<int> cumsum(nums.size() + 1, 0);
        unordered_map<int, int> record;
        int res = -1;
        
        record[0] = -1;
        for(int i = 0; i < nums.size(); ++i) target += nums[i];
        
        for(int i = 0; i < nums.size(); ++i) {
            cumsum[i + 1] = cumsum[i] + nums[i];
            if(record.count(cumsum[i + 1]) == 0) record[cumsum[i + 1]] = i;
            if(record.count(cumsum[i + 1] - target)) res = max(res, i - record[cumsum[i + 1] - target]);
        }
        
        if(res != -1) res = nums.size() - res;
        
        return res;
    }
};
