class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        unordered_map<int,int> mp;
        int pre = 0;
        mp[0] = -1;
        for (int i=0; i<n; i++) {
            pre+=nums[i];
            mp[pre] = i;
            //cout << pre << ":" << i << endl;
        }
        // [0:i]
        if (pre<x)
            return -1;
        int cur = x;
        int ret = INT_MAX;
        for (int i=n-1; i>=-1; i--) {
            if (cur<0)
                break;
            //cout << cur << endl;
            if (mp.count(cur)) {
                //cout << "find" << cur  <<" " << mp[cur]+n-i<< endl;
                ret = min(ret, mp[cur]+n-i);
                //cout << ret << endl;
            }
            if (i==-1)
                break;
            cur -= nums[i];
        }

        return ret==INT_MAX?-1:ret;
    }
};
