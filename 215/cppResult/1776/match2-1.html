class Solution {
public:
    int minOperations(vector<int>& a, int x) {
        int n = a.size();
        int ret = n+1;
        int sum = 0;
        for(int i=0;i<n;i++){
            sum+=a[i];
            if(sum==x)ret=min(ret, i+1);
        }
        if(sum<x)return -1;
        sum=0;
        for(int i=n-1;i>=0;i--){
            sum+=a[i];
            if(sum==x)ret=min(ret, n-i);
        }
        int r = 0;

        for(int i=0;i<n;i++){
            sum+=a[i];
            while(r<n&&sum>x){
                sum-=a[r];
                r++;
            }
            if(r<n&&sum==x){
                ret=min(ret,i+1+(n-r));
            }
        }
        if(ret==n+1)return -1;
        return ret;
    }
};
