class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        map<int, int> mp;
        mp[0] = 0;
        int sum = 0;
        for (int i = 0; i < n; ++i){
            sum += nums[i];
            mp[sum] = i + 1;
        }
        if (sum < x) return -1;
        int ans = 0x3f3f3f3f;
        if (mp.find(x) != mp.end()) ans = mp[x];
        sum = 0;
        for (int i = n-1; i > 0; --i){
            sum += nums[i];
            if (sum > x) break;
            if (mp.find(x-sum) != mp.end()){
                ans = min(ans, mp[x-sum]+(n-i));
            }
        }
        if (ans == 0x3f3f3f3f) return -1;
        return ans;
    }
};
