class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n=nums.size();
        int sum=0;
        
        for(int i:nums) sum+=i;
        if(sum==x) return n;
        int aim=sum-x;
        unordered_map<int,int> pos;
        pos[0]=-1;

        int presum=0;
        int res=-1;
        //cout<<aim<<endl;
        for(int i=0;i<n;i++){
            presum+=nums[i];
            if(pos.find(presum-aim)!=pos.end()){
                res=max(res,i-pos[presum-aim]);
            }
            if(pos.find(presum)==pos.end()){
                pos[presum]=i;
            }
        }
        return res==-1?-1:n-res;
    }
};
