class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> prefix(n, 0); // prefix sum
        int acc = 0;
        int ans = INT_MAX;
        for(int i = 0; i < n; i ++) {
            acc += nums[i];
            prefix[i] = acc;
            if(acc == x) ans = i + 1; // trying with left only
        }
        acc = 0;
        for(int r = n - 1; r >= 0; r --) {
            acc += nums[r];
            if(x - acc == 0) {
                ans = min(ans, n - r);
                break;
            }
            else if(x - acc < 0) break;
            int pos = lower_bound(prefix.begin(), prefix.begin() + r + 1, x - acc) - prefix.begin();
            if(pos >= r) continue;
            if(prefix[pos] + acc == x) ans = min(ans, pos + 1 + (n - r)); 
        }
        return (ans == INT_MAX)?-1:ans;
    }
};
