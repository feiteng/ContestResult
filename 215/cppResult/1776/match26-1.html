class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        unordered_map<int, int> rightsum;
        int curr_sum = 0;
        rightsum[0] = 0;
        for (int i = 0; i < n; i++) {
            curr_sum += nums[n-i-1];
            rightsum[curr_sum] = i+1;
        }
        if (curr_sum < x)
            return -1;
        // else if (curr_sum == x)
        //     return n;
        int ans = -1;
        curr_sum = 0;
        for (int i = 0; i < n; i++) {
            if (rightsum.count(x-curr_sum)) {
                int curr = i + rightsum[x-curr_sum];
                if (ans < 0 || curr < ans)
                    ans = curr;
            }
            curr_sum += nums[i];
        }
        return ans;
    }
};
