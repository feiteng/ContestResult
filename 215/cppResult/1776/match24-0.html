class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int, int> mp;
        int sum = 0;
        mp[0] = -1;
        for (int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            if (!mp.count(sum)) {
                mp[sum] = i;
            }
        }
        int tar = sum - x;
        sum = 0;
        int ret = -1;
        for (int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            if (mp.count(sum - tar) && mp[sum - tar] <= i) {
                ret = max(ret, i - mp[sum - tar]);
            }
        }
        return ret == -1 ? ret : nums.size() - ret;
    }
};
