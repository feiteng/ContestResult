class Solution {
public:
    int minOperations(vector<int>& nums, int x) {

        int n=nums.size(),ans=INT_MAX-100000;
        vector<int> left(n+1,0),right(n+1,0);
        for(int i=1;i<=n;++i){
            left[i]=left[i-1]+nums[i-1];
        }
        for(int i=1;i<=n;++i){
            right[i]=right[i-1]+nums[n-i];
        }
        for(int i=0;i<n+1;++i){

            int k=lower_bound(left.begin(),left.end(),x-right[i])-left.begin();
            if(k<n&&left[k]+right[i]==x){
                ans=min(ans,k+i);
            }
        }
        if(ans>n)
            return -1;
        return ans;
    }
};
