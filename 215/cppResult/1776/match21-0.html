class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int,int> sumToPos;
        sumToPos[0] = -1;
        int sum = 0;
        for (int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            sumToPos[sum] = i;
        }
        
        int ret = INT_MAX;
        sum = 0;
        if (sumToPos.count(x)) {
            ret = min(ret,sumToPos[x]+1);
        }
        for (int i = nums.size()-1; i >= 0; --i) {
            sum += nums[i];
            if (sumToPos.count(x-sum)) {
                int steps = sumToPos[x-sum] + 1 + (nums.size()-i);
                if (steps > nums.size()) continue;
                ret = min(ret,steps);
            }
        }
        if (ret == INT_MAX) ret = -1;
        return ret;
        
        
    }
};
