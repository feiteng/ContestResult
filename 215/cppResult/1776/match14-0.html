class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map <int, int> pos;
        int s = 0;
        pos[0] = 0;
        for (int i = 0; i < nums.size(); i++) {
            s += nums[i];
            pos[s] = i + 1;
        }
        if (s < x) {
            return -1;
        }
        int ans = 1e9;
        int n = nums.size(); 
        s = 0;
        int cnt = 0;
        if (pos.find(x) != pos.end()) {
            ans = pos[x];
        }
        
        for (int i = n - 1; i >= 0; i--) {
            s += nums[i];
            cnt++;
            if (s > x) {
                break;
            }
            if (pos.find(x - s) != pos.end()) {
                ans = min(ans, pos[x - s] + cnt);
            }
        }
        if (ans > n + 5) {
            return -1;
        }
        return ans;
    }
};
