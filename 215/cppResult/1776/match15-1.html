class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int, int> m1, m2;

        int tmp = 0;
        m1[0] = 0;
        for (int i = 0; i < nums.size(); ++i) {
            tmp += nums[i];
            m1[tmp] = i + 1;
        }
        
        tmp = 0;
        m2[0] = 0;
        for (int i = 0; i < nums.size(); ++i) {
            tmp += nums[nums.size() - 1 - i];
            m2[tmp] = i + 1;
        }
        
        int res = INT_MAX;
        for (const auto &it : m1) {
            if (x < it.first) {
                continue;

            }
            if (m2.find(x - it.first) != m2.end()) {
                res = min(res, it.second + m2[x - it.first]);
            }
        }
        
        if (res > nums.size()) {
            return -1;
        }
        return res;
    }
};
