class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int cursum = 0;
        int n = nums.size();
        map<int,int> need;
        need[cursum] = 0;
                
        
        int ans = 1001001001;
        for(int i = n-1; i>= 0; i--){
            cursum += nums[i];
            if (need.find(cursum)==need.end()) need[cursum] = n-i;
            if (cursum == x) ans = min(ans,n-i);
        }

        
        cursum = 0;
        
        for(int i = 0; i < n; i++){
            cursum += nums[i];
            int left = x-cursum;
            if (need.find(left) == need.end()) continue;
            int ne = need[left];
            int now = i+1;
            if (ne + now > n) continue;
            ans = min(ans,ne+now);
        }
        
        if (ans == 1001001001) return -1;
        return ans;
    }
};
