class Solution {
public:
    int pre[100010],suf[100010];

    int minOperations(vector<int>& nums, int x) {
        int n=nums.size();
        pre[0]=0;
        for (int i=1;i<=n;i++){
            pre[i]=pre[i-1]+nums[i-1];
        }
        suf[n+1]=0;
        for (int i=n;i>=1;i--){
            suf[i]=suf[i+1]+nums[i-1];
        }
        int Min = 1000000;
        for (int i=n+1;i>=1;i--){
            int sum1=suf[i];
            int diff=x-sum1;

            if (diff<0) break;
            int pos=lower_bound(pre,pre+i,diff)-pre;
            if (pre[pos]==diff&&pos<i){
                Min = min(Min,pos+(n+1)-i);
            }
        }
        if (Min==1000000) return -1;
        return Min;
    }
};
