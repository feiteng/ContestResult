class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int, int> exist;
        int sum = 0;
        for(auto n : nums) {
            sum += n;
        }
        
        sum -= x;
        if(sum == 0) {
            return nums.size();
        }
        
        int cur = 0;
        int maxLen = -1;
        exist[0] = -1;
        for(int i = 0; i < nums.size(); i++) {
            cur += nums[i];
            if(exist.find(cur) == exist.end()) {
                exist[cur] = i;
            }
            
            if(exist.find(cur - sum) != exist.end()) {
                if(i - exist[cur - sum] > maxLen) {
                    maxLen = i - exist[cur - sum];
                }
            }
        }
        return maxLen == -1 ? -1 : (nums.size() - maxLen);
    }
};
