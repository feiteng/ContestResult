class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size(), result = n + 1, sum = 0;
        for (int x : nums) {
            sum += x;
        }

        for (int l = 0, r = 0; l <= nums.size(); ++l) {
            while (r < nums.size() && sum > x) {
                sum -= nums[r++];
            }
            if (sum == x) {
                result = min(result, l + (int)nums.size() - r);
            }
            if (l < nums.size()) {
                sum += nums[l];
            }
        }
        return result == n + 1 ? -1 : result;
    }
};
