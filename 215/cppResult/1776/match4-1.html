class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> a(n+1,0),b(n+1,0);
        for(int i = 1; i <= n; i++) a[i] = a[i-1] + nums[i-1];
        reverse(nums.begin(),nums.end());
        for(int j = 1; j <= n; j++) b[j] = b[j-1] + nums[j-1];
        int res = -1;
        for(int i = 0; i <= n; i++) {
            int y = x-a[i];
            int z = lower_bound(b.begin(),b.end(),y) - b.begin();
            int op = i+z;
            if(z==n+1 || b[z]!=y || op>n) continue;
            if(res==-1 || res > op) res = op; //res = min(res,op);
        }
        return res;
    }
};
