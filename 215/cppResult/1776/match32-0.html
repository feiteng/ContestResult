const int N = 100005;

const int INF = 0x3f3f3f3f;

class Solution {
public:
    int pre[N], suf[N];
    int minOperations(vector<int>& nums, int x) {
        memset(pre,0,sizeof pre);
        memset(suf,0,sizeof suf);
        
        int n=(int)nums.size();
        for(int i=1;i<=n;++i){
            pre[i]=pre[i-1]+nums[i-1];
        }
        for(int i=1;i<=n;++i){
            suf[i]=suf[i-1]+nums[n-i];
        }
        
        int res=INF;
        for(int i=1;i<=n;++i){
            if(suf[i]==x)res=i;
        }
        
        for(int i=1;i<=n;++i){
            int p=pre[i];
            if(p==x)res=min(res,i);
            if(p>x)break;
            int idx=lower_bound(suf+1,suf+n+1,x-p)-suf;
            //cout<<idx<<endl;
            if(n-idx+1>i&&suf[idx]==x-p){
                res=min(res,i+idx);
            }
        }
        
        if(res==INF)return -1;
        return res;
    }
};
