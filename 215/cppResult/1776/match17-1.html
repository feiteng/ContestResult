class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> p(n + 1), s(n + 1);
        for (int i = 0; i < n; ++i) p[i + 1] = p[i] + nums[i];
        for (int i = 0; i < n; ++i) s[i + 1] = s[i] + nums[n - 1 - i];
        int ans = -1;
        for (int k = 0; k <= n; ++k) {
            int key = x - p[k];
            auto it = lower_bound(s.begin(), s.end(), key) - s.begin();
            if (it > n || s[it] != key) continue ;
            if (n - it <= k - 1) continue ;
            int r = k + it;
            if (ans == -1 || r < ans) ans = r;
        }
        return ans;
    }
};
