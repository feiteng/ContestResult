class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int res = INT_MAX;
        int n = nums.size();
        
        int sum = accumulate(nums.begin(), nums.end(), 0);
        if (sum < x) return -1;
        
        int presum = 0;
        unordered_map<int, int> left;
        left[0] = 0;
        for (int i = 1; i <= 2 * n; ++i) {
            presum += nums[(i - 1) % n];
            if (i <= n) {
                left[presum] = i;
            }
            if (i >= n) {
                if (left.count(presum - x) != 0) {
                    res = min(res, i - left[presum - x]);
                }
            }
        }
        
        if (res == INT_MAX) return -1;
        
        return res;
    }
};
