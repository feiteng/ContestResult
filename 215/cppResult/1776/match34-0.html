class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        
        int sum = 0;
        
        
        int n = nums.size();
        
        for(int i = 0; i < n; i++)sum += nums[i];
        
        if(x > sum)return -1;
        vector<int> v1(n),v2(n);
        
        v1[0] = nums[0];
        map<int,int> mp;
        mp[v1[0]] = 1;
        
        for(int i = 1; i < n; i++)
        {
            v1[i] = v1[i-1] + nums[i];
            mp[v1[i]] = i + 1;
            
        }
        
       
        
        v2[n-1] = nums[n-1];
        
        for(int i = n-2; i >= 0; i--)
        {
            v2[i] = v2[i+1] + nums[i];
        }
        
        int ans = 1000000000;
            
        int j = 0;
        
        for(int i = n-1; i >=0; i--)
        {
            j++;
            if(mp.find(x-v2[i]) != mp.end())
            {
                ans = min(ans,mp[x-v2[i]] + j);
            }
        }
        
        if(mp.find(x) != mp.end())
        {
            ans = min(ans, mp[x]);
        }
        
        j = 0;
        
        for(int i = n-1; i >=0;i--)
        {
            j++;
            
            if(v2[i] == x)
            {
                ans = min(ans,j);
            }
        }
        
        
        if(ans == 1000000000)
        {
            ans = -1;
        }
        
        return ans;
        
    }
};
