class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int ret = -1;
        int l = 0;
        int r = nums.size()-1;
        int count = 0;
        while(x > 0 && l <= r){
            x -= nums[l];
            l++;
            count++;
        }
        if(x == 0)
            ret = count;
        if(l > r)
            return ret;
        l--;
        while(l >= 0){
            count--;
            x += nums[l];
            l--;
            while(r >= l && x > 0){
                x -= nums[r];

                r--;
                count++;
            }
            if(x == 0){
                if(ret == -1)
                    ret = count;
                else
                    ret = min(ret,count);
            }
        }
        return ret;
    }
};
