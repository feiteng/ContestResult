const int maxn=1e5+10;
#define ll long long
ll sum[maxn];

class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n=nums.size();
        for (int i=1; i<=n; i++) {
            sum[i]=sum[i-1]+nums[i-1];
        }
        int res=1e9;
        if (sum[n]<x) return -1;
        else if (sum[n]==x) return n;
        ll tem=sum[n]-x; //剩余
        map<ll,int> M;

        M[0]=0;
        for (int i=1; i<=n; i++) {
            if (M.count(sum[i]-tem)) res=min(res,n-i+M[sum[i]-tem]);
            if (!M.count(sum[i])) M[sum[i]]=i;
        }
        if (res==1e9) res=-1;
        return res;
    }
};
