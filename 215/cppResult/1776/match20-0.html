#define INF 0x3f3f3f3f

class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
      unordered_map<int,int> pre;
      int sum = 0;
      pre[0] = 0;
      for(int i = 0 ; i < nums.size() ; i++) {
        sum += nums[i];
        if (!pre.count(sum)) pre[sum] = i + 1;
      }
      sum = 0;
      int res = INF;
      if(pre.count(x)) res = pre[x];
      for(int j = nums.size() - 1 ; j >= 0 ; j--){
        sum += nums[j];
        int now = x - sum;
        if(pre.count(now)) {
          if(pre[now] < j) {
            int tmp = nums.size() - j + pre[now];
            res = min(res,tmp);
          }
        }
      }
      if(res == INF) return -1;
      else return res;
    }
};
