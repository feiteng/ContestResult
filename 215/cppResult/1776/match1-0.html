class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        int r = n - 1;
        int l_sum = 0, r_sum = 0;
        int result = 0x7fffffff;
        for (int i = 0; i < n; i++) {
            l_sum += nums[i];
        }
        if (l_sum == x) {
            result = min(result, n);
        }
        for (int i = n - 1; i >= 0; i--) {
            l_sum -= nums[i];
            while (r >= i && l_sum + r_sum < x) {
                r_sum += nums[r];
                r--;
            }
            if (l_sum + r_sum == x) {
                result = min(result, i + n - 1 - r);
            }
        }
        return (result == 0x7fffffff) ? -1 : result;
    }
};
