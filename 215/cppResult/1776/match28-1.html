class Solution {
public:
    int minOperations(vector<int>& v, int x) {
        unordered_map<int,int>mp;
        int sum = 0;
        for (auto &i : v) {
            sum += i;
        }
        int tar = sum - x;
        mp[0] = -1;
        int mx = -1;
        int cur = 0;
        if (tar == 0) {
            mx = 0;
        }

        for (int i = 0 ; i < v.size() ; i++) {
            cur += v[i];
            if (mp.count(cur - tar)) {
                mx = max(mx, i - mp[cur-tar]);
            }
            if (!mp.count(cur)) {
                mp[cur] = i;
            }
        }
        if (mx == -1) {
            return -1;
        }
        else {
            return v.size() - mx;
        }
    }
};
