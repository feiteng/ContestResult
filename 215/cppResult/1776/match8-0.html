class Solution {
public:
    int minOperations(vector<int> &v, int x) {
        unordered_map<int, int> sums;
        int ret = numeric_limits<int>::max();
        for (int i = 0, sum = 0; i < v.size(); ++i) {
            sum += v[i];
            sums[sum] = i + 1;
            if (sum == x)ret = i + 1;
        }
        for (int i = v.size() - 1, sum = 0; i >= 0; --i) {
            sum += v[i];
            if (sum == x)ret = min(ret, (int) v.size() - i);
            auto it = sums.find(x - sum);
            if (it != sums.end() and it->second <= i) {
                ret = min(ret, int(it->second + v.size() - i));
            }
        }
        return ret == numeric_limits<int>::max() ? -1 : ret;
    }
};
