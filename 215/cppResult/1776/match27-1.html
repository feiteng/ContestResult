class Solution {
public:
    int minOperations(vector<int>& a, int x) {
        const int n = a.size();
        a.insert(a.end(), a.begin(), a.end());
        int64_t s = 0;

        int ans = n * 3;
        for (int i = 0, j = 0; i < n + n; ++i) {
            s += a[i];
            while ((j < i && s > x) || i - j >= n) {
                s -= a[j++];

            }
            if (s == x && (j == 0 || i == n - 1 || (i / n != j / n))) {
                ans = min(ans, i - j + 1);
            }
        }
        if (ans == n * 3) ans = -1;
        return ans;
    }
};
