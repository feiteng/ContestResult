class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        vector<int>l,r;
        int n=nums.size();
        int sl=0;
        l.push_back(0);
        for(int i=0;i<n;i++) {
            sl+=nums[i];
            l.push_back(sl);
        }
        int sr=0;
        r.push_back(0);
        for(int i=n-1;i>=0;i--) {
            sr+=nums[i];
            r.push_back(sr);
        }
        int ans=0x3f3f3f3f;
        for(int lp=0;lp<=n;lp++) {
            int fin=x-l[lp];
            if(fin<0) {
                break;
            }
            int low=lower_bound(r.begin(),r.end(),fin)-r.begin();
            if(low+lp<=n&&r[low]==fin) {
                ans=min(ans,lp+low);
            }
        }
        if(ans==0x3f3f3f3f) {
            return -1;
        }
        return ans;
    }
};
