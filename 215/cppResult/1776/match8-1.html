class Solution {
public:
    int minOperations(vector<int>& A, int x) {
        map<int,int> pos;
        int n=A.size();
        int Ans=-1;
        pos[0]=-1;

        for (int i=0,sum=0;i<n;++i)
        {
            sum+=A[i];
            pos[sum]=i;
            if (sum==x) Ans=i+1;
        }
        for (int i=n-1,sum=0;i>=0;--i)
        {
            sum+=A[i];
            if (sum>x) break;
            if (pos.find(x-sum)!=pos.end())
            {
                if (pos[x-sum]>=i) continue;
                int v=pos[x-sum]+1+n-i;
                if (Ans!=-1) Ans=min(Ans,v);
                else Ans=v;
            }
        }
        return Ans;
    }
};
