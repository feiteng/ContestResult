class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n=nums.size();
        int sum=0;
        int l=0,r=n;
        while(l<n&&sum<x)
        {
            sum=sum+nums[l];
            l++;
        }
        if(sum<x)return -1;
        int ans=n+1;
        if(sum==x)ans=l;

        while(l)
        {
            l--;
            sum=sum-nums[l];
            while(sum<x)
            {
                r--;
                sum=sum+nums[r];
            }
            if(sum==x)ans=min(ans,l+n-r);
        }
        if(ans==n+1)ans=-1;
        return ans;
    }
};
