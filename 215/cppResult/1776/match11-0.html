class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int s=0,mid=0,ans=-1;
        for(int i=0;i<nums.size();i++)s+=nums[i];
        
        int i,j;
        for(i=0,j=0;i<=j && j<nums.size();i++)
        {
            while(mid<s-x&&j<nums.size())
            {
                mid+=nums[j++];
            }
            if(mid==s-x && ans<j-i)
            {
                ans=j-i;
            }
            mid-=nums[i];
        }
        // return ans;
        if(ans==-1)return -1;
        return nums.size()-ans;
    }
};
