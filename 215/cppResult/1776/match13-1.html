class Solution {
  public:
    int minOperations(vector<int>& nums, int x) 
    {
      int N = nums.size();
      vector<int> prefix(N + 1, 0); // 0: empty, 1: 0, N: all
      for (int i = 0; i < N; ++i) prefix[i + 1] = prefix[i] + nums[i];

      unordered_map<int, int> suffix;
      int suffix_sum = 0;
      for (int i = N - 1; i >= 0; --i) {
        suffix_sum += nums[i];
        suffix[suffix_sum] = N - i;
      }
      
      int min = N + 1;
      for (int i = 0; i < N; ++i) {
        if (prefix[i] == x) {
          min = std::min(min, i);
        } else if (prefix[i] < x) {
          int d = x - prefix[i];
          if (suffix.find(d) != suffix.end()) {
            min = std::min(min, i + suffix[d]);
          } 
        } else {
          break;
        }
      }
      return min == N + 1 ? -1 : min;
    }
};
