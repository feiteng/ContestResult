class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int sum[100020];
        sum[0] = 0;
        int n = nums.size();
        int ans = -1;
        for(int i=0;i<n;++i){
            sum[i+1] = sum[i] + nums[i];
            if(sum[i+1] == x) ans = i+1;
        }
        
        int aug = 0;

        int y = n-1;
        for(int i=n;i>0;--i){
            aug += nums[i-1];
            if(y == i) --y;
            while(y > 0 && aug + sum[y] > x) --y;
            if(aug + sum[y] == x){
                if(ans == -1)ans = y + n-i+1;
                else ans = min(ans, y+n-i+1);
            }
        }
        return ans;
    }
};
