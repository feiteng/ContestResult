class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        for (int i=1; i<n; i++)
            nums[i]+=nums[i-1];
        unordered_map<int, int>m;
        m[0] = -1;
        int res=-1, target=nums[n-1]-x;
        if (target==0)
            res=0;
        for (int i=0; i<n; i++) {
          //  cout << i << ", " << nums[i] << ", " << target << endl;
            if (m.find(nums[i]-target)!=m.end())
                res = max(res, i-m[nums[i]-target]);
            if (m.find(nums[i])==m.end())
                m[nums[i]] = i;
        }
        if (res==-1)
            return -1;
        return n-res;
        
    }
};
