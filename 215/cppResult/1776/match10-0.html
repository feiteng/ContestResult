const int INF = 1000000000;
class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int N = nums.size();
        
        // initialize
        int left = 0;
        int right = 0;
        while (left<N and x>0)
            x -= nums[left++];
        
        // search by greedy way
        int min_operation = (x==0)? left: INF;
        while (left+right<=N and left>0) {
            x += nums[--left];
            while (right<N and x>0) {
                x -= nums[N-right-1];
                ++right;                
            }
            // printf("left: %d, right: %d, x: %d\n", left, right, x);
            if (left+right<=N and x==0) min_operation = min(min_operation, left+right);
        }
        
        if (min_operation == INF)
            return -1;
        else
            return min_operation;
    }
};
