class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        // 
        set<pair<int, int>>se;
        unordered_map<int, int> umap;
        int s = 0;
        int ans = INT_MAX;
        int sum = 0;

        for (auto num : nums) sum += num;
        x = sum - x;
        // cout << s << endl;
        umap[0] = -1;
        for (int i = 0; i < n; i ++) {
            s += nums[i];
            // se.insert({s, i});
            if (!umap.count(s))
                umap[s] = i;
            // cout << s << " " << i << endl;
            if (umap.count(s - x)) {
                // cout << umap[s - x] << endl;
                ans = min(ans, umap[s - x] - 0 + 1 + n - 1 - i);
            }
        }
        return ans == INT_MAX ? -1 : ans;
    }
};
