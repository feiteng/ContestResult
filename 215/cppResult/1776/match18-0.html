class Solution {
 public:
  int minOperations(vector<int>& nums, int x) {
    int res = 1e9;
    int r = nums.size() - 1, sum = nums[r];
    while (sum < x && r != 0) {
      --r;
      sum += nums[r];
    }
    if (sum == x) res = min(res, (int)nums.size() - r);
    if (r == 0) {
      if (res == 1e9) return -1;
      return res;
    }
    for (int i = 0; i < nums.size(); ++i) {
      sum += nums[i];
      while (r != nums.size() && sum - nums[r] >= x) {
        sum -= nums[r];
        r++;
      }
      if (sum == x) {
        res = min(res, i + 1 + (int)nums.size() - r);
      }
    }
    if (res == 1e9) return -1;
    return res;
  }
};
