class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> orderSum(n+1);
        vector<int> inverseSum(n+1);
        for (int i=0;i<n;i++) {
            orderSum[i+1] = orderSum[i] + nums[i];
            inverseSum[i+1] = inverseSum[i] + nums[n-1-i];
        }
        bool can = false;
        int ret = n;
        for (int i=0;i<=n;i++) {
            int target = x-orderSum[i];
            int l = 0, r = n, mid;
            bool find = false;
            while (l<=r) {
                mid = (l+r)>>1;
                if (inverseSum[mid]==target) {
                    find = true;
                    break;
                } else if (inverseSum[mid]>target) {
                    r = mid-1;
                } else {
                    l = mid+1;
                }
            }
            if (find&&i+mid<=n) {
                can = true;
                ret = min(ret, i+mid);
            }
        }
        if (!can) ret = -1;
        return ret;
    }
};
