class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        // sum(nums) - x = target
        // longest subarray sum == target
        int S = accumulate(nums.begin(), nums.end(), 0);
        // cout << "S = "  << S << endl;
        if (S == x) return nums.size();
        int target = S - x;
        int acc = 0;
        unordered_map<int, int> dp;
        dp[0] = -1;

        int ans = -1;
        for(int i = 0; i < nums.size(); i++) {
            acc += nums[i];
            if (dp.count(acc - target) > 0) {
                ans = max(ans, i - dp[acc-target]);
            }
            if (dp.count(acc) == 0)
                dp[acc] = i;            
        }
        if (ans == -1) {
            return -1;
        }
        return nums.size() - ans;
        
    }
};
