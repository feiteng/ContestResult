class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int,int> occur;
        int sum = 0;
        occur[0] = -1;
        for(int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            occur[sum] = i;
        }
        int res = 1e9;
        if(sum == x) {
            res = nums.size();
        }
        if(occur.count(sum - x)) {
            res = min(res, (int)nums.size() - occur[sum - x] - 1);
        }
        for(int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            if(occur.count(sum - x)) {
                res = min(res, i - occur[sum - x] + (int)nums.size());
            }
        }
        if(res > nums.size()) {
            return -1;
        }
        return res;
    }
};
