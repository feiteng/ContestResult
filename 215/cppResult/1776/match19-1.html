class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int sum = accumulate(nums.begin(), nums.end(), 0);
        //if(sum == x) return nums.size();
        int target = sum - x;
        int n = nums.size();
        unordered_map<int,int> store;
        store[0] = -1;
        int curSum = 0;
        int res = -1;

        if(target == 0) res = 0;
        for(int i = 0; i < nums.size(); ++i){
            curSum += nums[i];
            if(store.count(curSum - target) != 0){
                res = max(res, i - store[curSum - target]);
            }
            if(store.count(curSum) == 0) store[curSum] = i;
        }
        return res == -1 ? -1 : n - res;
    }
};
