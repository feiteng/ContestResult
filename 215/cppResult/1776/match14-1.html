#include <bits/stdc++.h>
using namespace std;
using ll = long long;



class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        unordered_map<int, int> mp;
        int n = nums.size(), run = 0;
        for (int i = n-1; i >= 0; --i) {
            run += nums[i];
            mp[run] = i;
        }      

        mp[0] = n;
        run = 0;
        int ans = -1;
        if (mp.count(x)) {
            ans = n - mp[x];
        }
        for (int i = 0; i < n; ++i) {
            run += nums[i];
            int target = x - run;
            if (mp.count(target) && mp[target] > i) {
                int cur = (i+1) + (n-mp[target]);
                if (ans == -1 || cur < ans) {
                    ans = cur;
                }
            }
        }
        return ans;
    }
};
