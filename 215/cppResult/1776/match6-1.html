class Solution {
private:
    int pre[100005];
    unordered_map<int, int> M;
public:
    int minOperations(vector<int>& nums, int x) {
        pre[0] = nums[0];

        M[0] = -1;
        M[ pre[0] ] = 0;
        for (int i=1; i<nums.size(); i++) {
            pre[i] = pre[i-1] + nums[i];
            M[ pre[i] ] = i;
        }
        int ans = nums.size() + 1, sum = 0;
        
        if (M.count(x)) {
            ans = M[x] + 1;

        }
        
        for (int i=nums.size()-1; i>=0; i--) {
            M.erase(pre[i]);
            sum += nums[i];
            
            if (M.count(x - sum)) {
                ans = min(ans, (int) nums.size() - i + M[x - sum] + 1);
            }
        }
        if (ans == nums.size() + 1) {
            return -1;
        }
        return ans;
    }
};
