class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();
        vector<int> psum(n);
        int sum = 0;
        int res = INT_MAX;
        for(int i = 0; i < n; i++) {
            sum+=nums[i];
            psum[i]=sum;
            if(sum==x) res = min(res,i+1);
        }
        sum=0;
        for(int i = n-1, j=1; i>=0; i--,j++) {
            sum+=nums[i];
            if(x==sum) res = min(res,j);
            int y = x-sum;
            if(y<0) break;
            int id = lower_bound(psum.begin(), psum.begin()+i,y)-psum.begin();
            if(id<i && psum[id]==y) {
                res = min(res,id+1+j);
            }
        }
        if(res==INT_MAX) res=-1;
        return res;
    }
};
