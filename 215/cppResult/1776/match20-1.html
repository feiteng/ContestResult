class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();

        unordered_map<int, int> dict;
        int sum = 0;
        dict[0] = 0;
        for (int i = 0; i < nums.size(); ++i)
        {
            sum += nums[i];
            if (dict.count(sum) == 0) dict[sum] = i + 1;
        }
      
        sum = 0;
        int res = INT_MAX;
        if (dict.count(x) != 0) res = dict[x];        
        for (int i = n - 1; i >= 0; --i)
        {
            sum += nums[i];
            int left = x - sum;            
            
            if (dict.count(left) != 0 && dict[left] + (n - i) <= n) 
            {
                res = min(res, (n - i) + dict[left]);
               // cout << i << "," << left << ","  << dict[left] << "," << (n - i) << "," << res << endl;
            }
        }
        if (res == INT_MAX) return -1;
        
        return res;
    }
};
