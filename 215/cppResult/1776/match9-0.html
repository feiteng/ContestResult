class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int total = accumulate(nums.begin(), nums.end(), 0);
        //cout<<total<<endl;
        if (total < x)
            return -1;
        int res = INT_MAX;
        unordered_map<int, int> prefix;
        int target = total - x;
        //cout<<target<<endl;
        int cur = 0;
        prefix[0] = 0;
        int n = nums.size();
        prefix[total] = n;
        for (int i = 0; i < n; i++) {
            cur += nums[i];
            //cout<<i<<" "<<cur<<endl;
            if (prefix.count(cur - target) > 0) {
                //cout<<i<<" "<<prefix[cur - target]<<endl;
                res = min(res, n - (i + 1 - prefix[cur - target]));
            }
            if (prefix.count(cur) == 0) {
                prefix[cur] = i + 1;
            } 
        }
        return res == INT_MAX ? -1 : res;
    }
};
