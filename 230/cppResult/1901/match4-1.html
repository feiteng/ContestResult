class Solution {
public:
    int minOperations(vector<int>& A, vector<int>& B) {
        vector<int> cnt1(7), cnt2(7);
        int sum1 = 0, sum2 = 0;
        for (int a : A) {
            cnt1[a]++;
            sum1 += a;
        }
        for (int b : B) {
            cnt2[b]++;
            sum2 += b;
        }
        if (sum1 == sum2) return 0;
        if (sum1 < sum2) {
            swap(sum1, sum2);
            cnt1.swap(cnt2);
        }
        int diff = sum1 - sum2, res = 0;
        for (int i = 5; i >= 1; i--) {
            int roll = cnt1[1 + i] + cnt2[6 - i];
            if (roll * i >= diff) {
                res += (diff + i - 1) / i;
                return res;
            }
            diff -= i * roll;
            res += roll;
        }
        return -1;
    }
};
