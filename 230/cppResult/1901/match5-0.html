class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int s1 = 0;
        int s2 = 0;
        int c1[7];
        int c2[7];
        for (int i = 1; i < 7; ++i) {
            c1[i] = c2[i] = 0;
        }
        for (int i = 0; i < nums1.size(); ++i) {
            s1 += nums1[i];
            c1[nums1[i]]++;
        }
        for (int i = 0; i < nums2.size(); ++i) {
            s2 += nums2[i];
            c2[nums2[i]]++;
        }
        if (s1 == s2)
            return 0;
        if (s1 < s2) {
            int t = 0;
            int flag = 0;
            for (int i = 1; i < 6; ++i) {
                if (s2-s1 <= (6-i)*(c1[i]+c2[7-i])) {
                    t += (s2-s1+5-i)/(6-i);
                    flag = 1;
                    break;
                } else {
                    t += c1[i]+c2[7-i];
                    s2 -= (6-i)*(c1[i]+c2[7-i]);
                }
            }
            if (flag == 1)
                return t;
            else
                return -1;
        } else {
            int t = 0;
            int flag = 0;
            for (int i = 1; i < 6; ++i) {
                if (s1-s2 <= (6-i)*(c2[i]+c1[7-i])) {
                    t += (s1-s2+5-i)/(6-i);
                    flag = 1;
                    break;
                } else {
                    t += c2[i]+c1[7-i];
                    s1 -= (6-i)*(c2[i]+c1[7-i]);
                }
            }
            if (flag == 1)
                return t;
            else
                return -1;
        }
        return -1;
    }
};
