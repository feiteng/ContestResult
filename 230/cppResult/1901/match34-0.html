class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size();
        if(m>6*n || n>6*m)
            return -1;
        int sum1=0,sum2=0,ans=0;
        int sum[2][6]={0};
        int max[6]={0};
        for(auto n:nums1)
        {
            sum1+=n;
            sum[0][n-1]++;
        }
        for(auto n:nums2)
        {
            sum2+=n;
            sum[1][n-1]++;
        }
        if(sum1>sum2)
        {
            for(int i=0;i<6;i++)
            {
                max[i]=sum[0][i]+sum[1][5-i];
            }
        }
        else
        {
            for(int i=0;i<6;i++)
            {
                max[i]=sum[0][5-i]+sum[1][i];
            }
        }
        cout<<sum1<<" "<<sum2<<endl;
        for(int i=0;i<6;i++)
            cout<<max[i]<<" ";
        cout<<endl;
        sum1=abs(sum1-sum2);
        int t=5;
        while(sum1>0)
        {
            if(t*max[t]<sum1)
            {
                sum1-=t*max[t];
                ans+=max[t];
            }
            else
            {
                ans+=(sum1+t-1)/t;
                sum1=0;
            }
            cout<<ans<<" "<<sum1<<endl;
            t--;
        }
        return ans;
    }
};
