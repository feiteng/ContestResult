class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        auto sum1 = accumulate(nums1.begin(), nums1.end(), 0);
        auto sum2 = accumulate(nums2.begin(), nums2.end(), 0);
        unordered_map<int, int> one, two, big, small;
        int bigsum, smallsum;
        for(auto num : nums1)
            one[num]++;
        for(auto num : nums2)
            two[num]++;

        if(sum1 > sum2) {
            big = one;
            small = two;
            bigsum = sum1;
            smallsum = sum2;
        } else if(sum2 > sum1) {
            small = one;
            big = two;
            bigsum = sum2;
            smallsum = sum1;
        } else
            return 0;
        auto bi = 6;
        auto si = 1;
        auto res = 0;
        while(bigsum > smallsum && bi > 1 && si < 6) {
            while(big[bi] && bigsum - smallsum > 0) {
                bigsum -= bi - 1;
                res++;
                big[bi]--;
            }
            bi--;
            while(small[si] && bigsum - smallsum > 0) {
                smallsum += 6 - si;
                res++;
                small[si]--;
            }
            si++;
        }

       return smallsum >= bigsum ? res : -1;
    }
};
