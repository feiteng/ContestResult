class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = 0, sum2 = 0;
        for(auto x : nums1) sum1 += x;
        for(auto x : nums2) sum2 += x;
        if(sum1 > sum2) return minOperations(nums2, nums1);
        priority_queue<int, vector<int>, less<int> > MAX;
        priority_queue<int, vector<int>, greater<int> > MIN;
        
        for(auto x : nums1) MIN.push(x);
        for(auto x : nums2) MAX.push(x);
        
        int res = 0;
        int cnt = sum2 - sum1;
        while(cnt > 0){
            if(MIN.top() == 6 && MAX.top() == 1) return -1;
            res ++;
            if(6 - MIN.top() > MAX.top() - 1){
                cnt -= (6 - MIN.top());
                MIN.pop();
                MIN.push(6);
            }else{
                cnt -= (MAX.top() - 1);
                MAX.pop();
                MAX.push(1);
            }
        }
        
        return res;
    }
};
