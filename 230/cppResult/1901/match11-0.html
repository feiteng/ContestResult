class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1, sum2;
        sum1 = sum2 = 0;
        for (int e : nums1) sum1 += e;
        for (int e : nums2) sum2 += e;
        if (sum1 > sum2) swap(nums1, nums2), swap(sum1, sum2);
        int target = sum2 - sum1;
        if (target == 0) return 0;
        vector<int> foo;
        for (int e : nums1) {
            foo.push_back(6 - e);
        }
        for (int e : nums2) {
            foo.push_back(e - 1);
        }
        sort(foo.rbegin(), foo.rend());
        int ans = -1;
        int sum = 0;
        for (int i = 0; i < foo.size(); i++) {
            sum += foo[i];
            if (sum >= target) {
                ans = i + 1;
                break;
            }
        }
        return ans;
    }
};
