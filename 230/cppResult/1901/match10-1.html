class Solution {
public:
    int minOperations(vector<int>& a, vector<int>& b) {
        int sa = accumulate(a.begin(), a.end(), 0);
        int sb = accumulate(b.begin(), b.end(), 0);
        if(sa > sb) {
            swap(sa, sb);
            swap(a, b);
        }
        
        int ans = 0;
        sort(a.begin(), a.end());
        sort(b.begin(), b.end(), greater<int>());
        
        int n = a.size(), m = b.size(), i = 0, j = 0;
        
        while(sa < sb and (i < n or j < m)) {

            if(j == m or i < n and 6 - a[i] > b[j] - 1) {
                sa += 6 - a[i ++];
            }
            else {
                sb -= b[j ++] - 1;
            }
            ans ++;
        }
        return sa < sb ? -1 : ans;
    }
};
