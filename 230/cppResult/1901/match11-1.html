class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = 0, sum2 = 0;
        for (int& i: nums1) sum1 += i;
        for (int& i: nums2) sum2 += i;

        if (sum1 > sum2) return minOperations(nums2, nums1);
        
        vector<int> v;
        for (int& i: nums1) {
            v.push_back(6 - i);
        }
        for (int& i: nums2) {
            v.push_back(i - 1);
        }
        sort(v.rbegin(), v.rend());
        int diff = sum2 - sum1;
        int ans = 0;
        for (int& i: v) {
            if (diff <= 0) return ans;
            diff -= i;
            ++ans;
        }
        if (diff <= 0) return ans;
        return -1;
    }
};
