class Solution {
public:
    int minOperations(vector<int>& num1, vector<int>& num2) {
        int s1 = 0;
        int s2 = 0;
        for(auto n:num1) s1+=n;
        for(auto n:num2) s2+=n;
        if(s1>s2) {
            swap(num1, num2); // make sure sum1 < sum2;
            swap(s1,s2);
        }
        // can add,
        // can down 
        vector<int > dlt;
        for(auto x:num1) {
            if(x!=6) dlt.push_back(6-x);
        }
        for(auto x:num2) {
            if(x!=1) dlt.push_back(x-1);
        }
        sort(dlt.begin(), dlt.end());
        
        int op = 0;
        int need_ = s2 - s1;
        if(need_==0 ) return 0;

        for(int i = dlt.size()-1; i>=0; i--) {
            op++;
            need_-=dlt[i];
            if(need_<=0) return op;
        }
        return -1;
        
    }
};
