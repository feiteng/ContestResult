class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size(), m = nums2.size();
        if (n*6 < m || m*6 < n) return -1;
        
        int delta = accumulate(nums1.begin(), nums1.end(), 0) - accumulate(nums2.begin(), nums2.end(), 0);
        //cout << delta << endl;
        int cnt = 0;
        if (delta < 0) {
            for (int i = 0; i < n; i++) {
                nums1[i] = 6 - nums1[i];
            }
            for (int i = 0; i < m; i++) {
                nums1.push_back(nums2[i]-1);
            }
            
            sort(nums1.begin(), nums1.end(), greater<int>());
            for (int i : nums1) {
                delta += i;
                cnt++;
                if (delta >= 0) break;
            }
        } else if (delta > 0) {
            for (int i = 0; i < n; i++) {
                nums1[i] = nums1[i] - 1;
            }
            for (int i = 0; i < m; i++) {
                nums1.push_back(6 - nums2[i]);
            }
            sort(nums1.begin(), nums1.end(), greater<int>());
            for (int i : nums1) {
                delta -= i;
                cnt++;
                if (delta <= 0) break;
            }
        }
        
        return cnt;
    }
};
