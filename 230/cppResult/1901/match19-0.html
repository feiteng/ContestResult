#define FOR(i,a,n) for (int i = (a); i < (n); ++i)
#define FORE(i,a,n) for (int i = (a); i <= (n); ++i)
#define FORD(i,a,b) for (int i = (a); i >= (b); --i)
#define REP(i,n) FOR(i,0,n)
#define REPE(i,n) FORE(i,0,n)
#define LL long long
#define FIR(n) REP(i,n)
#define FJR(n) REP(j,n)
#define ALL(v) v.begin(), v.end()

#define FI FIR(n)
#define FJ FJR(n)
#define FR(i,a) FOR(i,a,n)
#define REPN(i) REP(i,n)

class Solution {
public:
	int minOperations(vector<int>& v1, vector<int>& v2) {
		int s1 = accumulate(ALL(v1), 0);
		int s2 = accumulate(ALL(v2), 0);
		if (s1 == s2)
			return 0;
		if (s1 > s2)
			return minOperations(v2, v1);

		if (v1.size() * 6 < v2.size())
			return -1;

		sort(ALL(v1));
		sort(ALL(v2));
		int res = 0;

		int p1 = 0, p2 = v2.size() - 1;
		while (s1 < s2) {
            ++res;
			int w1 = 0;
			if (p1 < v1.size()) w1 = 6 - v1[p1];
			int w2 = 0;
			if (p2 >= 0) w2 = v2[p2] - 1;

			if (w1 > w2) {
				s1 += w1;
				++p1;
			}
			else {
				s2 -= w2;
				--p2;
			}
				
		}

		return res;
	}
};
