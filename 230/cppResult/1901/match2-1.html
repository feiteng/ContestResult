class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int res = 0;

        int sum1 = 0, sum2 = 0;
        for (int i = 0; i < nums1.size(); ++i) sum1 += nums1[i];
        for (int i = 0; i < nums2.size(); ++i) sum2 += nums2[i];
        
        if (sum2 < sum1) {
            swap(nums1, nums2);
            swap(sum1, sum2);
        }
        
        priority_queue<int> que;
        for (int i = 0; i < nums1.size(); ++i) que.push(6 - nums1[i]);
        for (int i = 0; i < nums2.size(); ++i) que.push(nums2[i] - 1);
        
        int diff = sum2 - sum1;

        while(que.size() && 0 != diff) {
            int t = que.top();
            diff -= min(diff, t);
            que.pop();
            ++res;
        }
        if (0 != diff) res = -1;
        return res;
        
    }
};
