class Solution {
public:
    int minOperations(vector<int>& n1, vector<int>& n2) {
        int s1 = accumulate(n1.begin(), n1.end(), 0);
        int s2 = accumulate(n2.begin(), n2.end(), 0);
        if(s1 == s2) return 0;
        if(s1 > s2) {
            return minOperations(n2,n1);
        }
        
        int c1 = n1.size(), c2 = n2.size();
        if(c1 * 6 < c2 * 1) return -1;
        sort(n1.begin(), n1.end());
        sort(n2.begin(), n2.end());
        
        int i = 0, j = c2-1;
        int diff = s2 -s1;

        int res = 0;
        while(i<c1 && j>=0) {
            if(6-n1[i] < n2[j]-1) {
                diff -= n2[j] - 1;
                j--;
            } else {
                diff -= 6-n1[i];
                i++;
            }
            res++;
            if(diff<= 0) return res;
        } 
        
        while(i<c1) {
             diff -= 6-n1[i];
                i++;
            res++;
            if(diff<= 0) return res;
        }
        while(j>=0) {
            diff -= n2[j] - 1;
                j--;
            res++;
            if(diff<= 0) return res;
        }
        
        return -1;
    }
};
