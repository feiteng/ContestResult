class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2 = nums2.size();
        int s1 = 0, s2 = 0;

        vector<int> a1 = nums1, a2 = nums2;
        for (int x: a1) {
            s1 += x;
        }
        for (int x: a2) {
            s2 += x;
        }
        if (s1 < s2) {
            swap(s1, s2); swap(a1, a2); swap(n1, n2);
        }
        // we want s1 - s2, and we have
        vector<int> cs(6, 0);

        for (int x: a1) {
            cs[x - 1] += 1;
        }
        for (int x: a2) {
            cs[6 - x] += 1;
        }
        int curr = 0;
        int ans = 0;
        int need, now;

        for (int i = 5; i >= 1; --i) {
            if (curr + i * cs[i] < s1 - s2) {
                curr += i * cs[i];
                ans += cs[i];
            } else {
                need = s1 - s2 - curr;
                now = (need + i - 1) / i;
                curr += now * i; ans += now;
                break;
            }
        }
        if (curr < s1 - s2) {
            return -1;
        } else {
            return ans;
        }
    }
};
