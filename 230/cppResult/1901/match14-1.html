class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int m=nums1.size(), n=nums2.size();
        int sum1=accumulate(nums1.begin(), nums1.end(), 0);        
        int sum2=accumulate(nums2.begin(), nums2.end(), 0);
        sort(nums1.begin(), nums1.end());        
        sort(nums2.begin(), nums2.end());
        int res=0;
        if(sum1==sum2) return 0;
        else if(sum1<sum2) {
            int i=0, j=n-1;
            while(i<m || j>=0) {
                if(i<m && j>=0) {
                    int x=6-nums1[i], y=nums2[j]-1;

                    if(x>=y) {
                        sum1 += x;
                        ++i;
                    } else {
                        sum2 -= y;
                        --j;
                    }
                } else if(i<m) {
                    sum1+=6-nums1[i];
                    ++i;
                } else {
                    sum2-=nums2[j]-1;
                    --j;
                }
                ++res;
                if(sum1>=sum2) return res;
            }
        } else {
            int i=m-1, j=0;
            while(i>=0 || j<n) {
                if(i>=0 && j<n) {
                    int x=nums1[i]-1, y=6-nums2[j];

                    if(x>=y) {
                        sum1 -= x;
                        --i;
                    } else {
                        sum2 += y;
                        ++j;
                    }
                } else if(i>=0) {
                    sum1-=nums1[i]-1;
                    --i;
                } else {
                    sum2+=6-nums2[j];
                    ++j;
                }
                ++res;
                if(sum1<=sum2) return res;
            }
        }
        
        return -1;
    }
};
