class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        vector<int> cn1(6, 0), cn2(6, 0);
        int sum1 = 0, sum2 = 0;
        for (const auto each : nums1)
        {
            ++cn1[each - 1];
            sum1 += each;
        }
        for (const auto each : nums2)

        {
            ++cn2[each - 1];
            sum2 += each;
        }
        if (sum1 > sum2)
        {
            swap(cn1, cn2);
            swap(sum1, sum2);
        }
        int del = sum2 - sum1, ans = 0;
       
        for (int i = 0; i < 5; ++i)
        {
            int c = cn1[i] + cn2[5 - i], rdc = 5 - i;
            if (rdc * c >= del)
            {
                ans += del / rdc;
                del = del - (del / rdc * rdc);
                if (del != 0)
                {
                    del = 0;
                    ++ans;
                }
            }
            else{
                ans += c;
                del -= c * rdc;
            }
        }
        if (del > 0) return -1;
        return ans;
    }
};
