class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        sort(nums1.begin(),nums1.end());
        sort(nums2.begin(),nums2.end());
        int ans = 0 ;
        int s1 = 0 , s2 = 0;
        for(int i : nums1)
            s1+=i;
        for(int i : nums2)
            s2+=i;
        cout<<s1<<" "<<s2<<endl;
        if(s1==s2)
            return 0;
        if(s1<s2){
            swap(nums1,nums2);
            swap(s1,s2);
        }

        int s = 0, e = nums1.size()-1 , n = nums2.size();
        while(s<n && e>-1){
            int val1 = nums1[e]-1;
            int val2 = 6-nums2[s];
            if(val1>val2){
                s1-=val1;
                e--;
            }
            else{
                s1-=val2;
                s++;
            }
            ans++;
            if(s1<=s2)
                    return ans;
        }
        while(s<n){
            int val2 = 6-nums2[s];
            s1-=val2;
            s++;
            ans++;
            if(s1<=s2)
                    return ans;
        }
        while(e>-1){
            int val1 = nums1[e]-1;
            s1-=val1;
            e--;
            ans++;
            if(s1<=s2)
                    return ans;
        }
        return -1;
    }
};
