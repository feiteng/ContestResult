class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = 0, sum2 = 0;
        for (int i : nums1) {
            sum1 += i;
        }
        for (int i : nums2) {
            sum2 += i;
        }
        if (sum2 > sum1) {
            swap(nums1, nums2);
            swap(sum2, sum1);
        }
        int diff = sum1 - sum2;
        if (diff == 0) {
            return 0;
        }
        int n = nums1.size();
        int m = nums2.size();
        if (n > m * 6) {
            return -1;
        }
        vector<int> add(6, 0);
        for (int i : nums1) {
            add[i - 1]++;
        }
        for (int i : nums2) {
            add[6 - i]++;
        }
        int ans = 0;
        for (int i = 5; i > 0; i--) {
            if (diff > add[i] * i) {
                diff -= add[i] * i;
                ans += add[i];
            }
            else {
                ans += (diff - 1) / i + 1;
                break;
            }
        }
        return ans;
    }
};
