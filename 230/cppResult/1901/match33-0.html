class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        sort(nums1.begin(), nums1.end());
        sort(nums2.begin(), nums2.end());
        int s1 = 0, s2 = 0;
        for (int x : nums1) s1 += x;
        for (int x : nums2) s2 += x;
        //cout << s1 << ' ' << s2 << endl;
        if (s1 == s2) return 0;
        if (s1 < s2) { swap(nums1, nums2); swap(s1, s2); }
        //s1 > s2 把s1的数变小 把s2的数变大
        int n = nums1.size(), m = nums2.size();
        if (n * 1 > m * 6) return -1;
        int ans = 0, p1 = n - 1, p2 = 0;
        while (s1 > s2) {
            int d = 0;
            if (p1 >= 0 && p2 < m) {
                if (nums1[p1] - 1 >= 6 - nums2[p2]) {
                    d = nums1[p1] - 1;
                    --p1;
                } else {
                    d = 6 - nums2[p2];
                    ++p2;
                }
            } else if (p1 >= 0) {
                d = nums1[p1] - 1;
                --p1;
            } else {
                d = 6 - nums2[p2];
                ++p2;
            }
            s1 -= d;
            ++ans;
           // cout << p1 << ' ' << p2 << endl;
        }
        return ans;
    }
};
