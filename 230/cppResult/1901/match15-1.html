class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        sort(nums1.begin(), nums1.end());
        sort(nums2.begin(), nums2.end());
        int sum1 = accumulate(nums1.begin(), nums1.end(), 0); 
        int sum2 = accumulate(nums2.begin(), nums2.end(), 0);
        if (sum1 < sum2) {
            swap(nums1, nums2);
            swap(sum1, sum2);
        }
        //if (nums1.size() > nums2.size() * 6) return -1; 
        int i = nums1.size()-1; 
        int j = 0; 
        int ans = 0;
        while (sum1 > sum2) {
            if (i < 0 && j >= nums2.size()) {
                ans = -1;
                break; 
            }
            if (i < 0) {
                sum2 += 6 - nums2[j]; 
                j++; 
                ans++;
                continue; 
            }
            if (j >= nums2.size()) {
                sum1 -= nums1[i] - 1; 
                i--;  
                ans++; 
                continue; 
            }
            

            if (nums1[i]-1 > 6 - nums2[j]) {
                sum1 -= nums1[i] - 1; 
                i--; 
            }
            else {
                sum2 += 6 - nums2[j]; 
                j++; 
            }
            ans++; 
        }
        return ans; 
    }
};
