class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = accumulate(nums1.begin(), nums1.end(), 0), sum2 = accumulate(nums2.begin(), nums2.end(), 0);
        if (sum1 > sum2)
            return minOperations(nums2, nums1);
        if (sum1 == sum2)
            return 0;
        int op = 0, dif = sum2 - sum1, l1 = 0, l2 = 0;
        sort(nums1.begin(), nums1.end());
        sort(nums2.begin(), nums2.end(), greater<int>());
        while (l1 < nums1.size() && l2 < nums2.size() && dif) {
            if ((6 - nums1[l1]) > (nums2[l2] - 1)) {
                dif = max(0, dif - (6 - nums1[l1]));
                op++;
                l1++;
            }
            else {
                dif = max(0, dif - (nums2[l2] - 1));
                op++;
                l2++;
            }
        }
        while (l1 < nums1.size() && dif) {
            dif = max(0, dif - (6 - nums1[l1]));
            op++;
            l1++;
        }
        while (l2 < nums2.size() && dif) {
            dif = max(0, dif - (nums2[l2] - 1));
            op++;
            l2++;
        }
        return dif == 0 ? op : -1;
    }
};
