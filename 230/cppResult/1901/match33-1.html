class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = 0, sum2 = 0; 
        for (int n : nums1) sum1 += n; 
        for (int n : nums2) sum2 += n; 
        
        if (sum1 == sum2) return 0;
        
        sort(nums1.begin(), nums1.end()); 
        sort(nums2.begin(), nums2.end());
        
        int ans = 0;
        if (sum1 < sum2) {
            int i = 0, j = nums2.size() - 1; 
            while (i < nums1.size() || j >= 0) {

                if (i < nums1.size() && j >= 0) {
                    if (6 - nums1[i] > (nums2[j] - 1)) {
                        sum1 += 6 - nums1[i++];
                    } else {
                        sum2 -= nums2[j--] - 1; 
                    }
                } else if (i < nums1.size()) {
                     sum1 += 6 - nums1[i++];
                } else {
                    sum2 -= nums2[j--] - 1; 
                }
                ++ans; 
                if (sum1 >= sum2) return ans; 
            }
        } else {
            int i = nums1.size() - 1, j = 0; 
            while (i >= 0 || j < nums2.size()) {
                if (i >= 0 && j < nums2.size()) {
                    if (nums1[i] - 1 > (6 - nums2[j])) {
                        sum1 -= nums1[i--] - 1;
                    } else {
                        sum2 += 6 - nums2[j++]; 
                    }
                } else if (i >= 0) {
                     sum1 -= nums1[i--] - 1;
                } else {
                    sum2 += 6 - nums2[j++]; 
                }
                ++ans; 
                if (sum1 <= sum2) return ans; 
            }
        }
        return -1; 
    }
};
