class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int tot1 = accumulate(nums1.begin(), nums1.end(), 0);
        int tot2 = accumulate(nums2.begin(), nums2.end(), 0);
        if (tot1 > tot2) {
            swap(nums1, nums2);
        }
        int delta = abs(tot1 - tot2);
        int cnt = 0;
        sort(nums1.begin(), nums1.end());
        sort(nums2.rbegin(), nums2.rend());
        int idx1 = 0;
        int idx2 = 0;
        while (!(idx1 == nums1.size() and idx2 == nums2.size()) and delta > 0) {
            if (idx1 < nums1.size() and idx2 < nums2.size()) {
                if (6 - nums1[idx1] > nums2[idx2] - 1) {
                    delta -= 6 - nums1[idx1++];
                } else {
                    delta -= nums2[idx2++] - 1;
                }
            } else if (idx1 < nums1.size()) {
                delta -= 6 - nums1[idx1++];
            } else {
                delta -= nums2[idx2++] - 1;
            }
            ++cnt;
        }
        return delta > 0 ? -1 : cnt;
    }
};
