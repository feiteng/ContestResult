class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size() , m = nums2.size() ; 
        if( m*6 < n || n*6 < m ) 
            return -1 ; 
        int sum1 = 0 , sum2 = 0 ; 
        for( int i = 0 ; i < n ; i++ )
            sum1 += nums1[i] ; 
        for( int i = 0 ; i < m ; i++ )
            sum2 += nums2[i] ; 
        sort( nums1.begin() , nums1.end() ) ; 
        sort( nums2.begin() , nums2.end() ) ; 
        int ans = 0 ; 
        if( sum1 < sum2 ){
            int ind1 = 0 , ind2 = m-1 ; 
            while( sum1 < sum2 && ind1 < n && ind2 >= 0 ){
                int change1 = 6 - nums1[ind1] ; 
                int change2 = nums2[ind2] - 1 ; 

                if( change1 > change2 ){
                    sum1 += change1 ; 
                    ans++ ; 
                    ind1++ ; 
                }
                else{
                    sum2 -= change2 ; 
                    ans++ ; 
                    ind2-- ; 
                }
            }
            while( sum1 < sum2 && ind1 < n ){
                sum1 += ( 6 - nums1[ind1] ) ; 
                ind1++ ; 
                ans++ ; 
            }
            while( sum1< sum2 && ind2 >= 0 ){
                sum2 -= ( nums2[ind2] - 1 ) ; 
                ind2-- ; 
                ans++ ; 
            }
        }
        else{
            int ind1 = n-1 , ind2 = 0 ; 
            while( sum1 > sum2 && ind1 >= 0 && ind2 < m ){
               // cout << sum1 << ' ' << sum2 << endl ; 
                int c1 = 6 - nums2[ind2]; 
                int c2 = nums1[ind1] - 1 ; 
                //cout << c1 << ' ' << c2 << endl ; 
                if( c1 > c2 ){
                    ans++ ; 
                    sum2 += c1 ; 
                    ind2++ ; 
                }
                else{
                    sum1 -= c2 ; 
                    ans++ ; 
                    ind1-- ; 
                }
            }
            while( sum1 > sum2 && ind2 < m ){
                sum2 += ( 6 - nums2[ind2] ) ; 
                ind2++ ;
                ans++ ; 
            }
            while( sum1 > sum2 && ind1 >= 0 ){
                sum2 -= ( nums1[ind1] - 1 ) ; 
                ind1-- ; 
                ans++ ; 
            }
        }
        return ans ; 
    }
};
