class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size();
        int n = nums2.size();
        if (m <= n && m*6 < n)
            return -1;
        if (n <= m && n*6 < m)
            return -1;
        int s1 = 0;
        for (const auto& i : nums1)
            s1 += i;
        int s2 = 0;
        for (const auto& i : nums2)
            s2 += i;
        if (s1 == s2)
            return 0;
        sort(nums1.begin(), nums1.end());
        sort(nums2.begin(), nums2.end());
        vector<int>& small = s1 < s2 ? nums1 : nums2;
        vector<int>& big = s1 < s2 ? nums2 : nums1;
        int l = 0;
        int r = big.size()-1;
        int s = abs(s1-s2);
        int res = 0;
        while (s > 0) {
            if (l >= small.size() || (r >= 0 && 6 - small[l] <= big[r] - 1)) {
                s -= big[r] - 1;
                r--;
            } else {
                s -= 6 - small[l];
                l++;
            }
            res++;
        }
        return res;
    }
};
