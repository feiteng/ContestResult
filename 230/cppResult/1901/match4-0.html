class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        vector<int> a1(7),a2(7);
        int s1=0,s2=0;
        for(int q:nums1){
            s1+=q;
            ++a1[q];
        }
        for(int q:nums2){
            s2+=q;
            ++a2[q];
        }
        if(s1==s2)
            return 0;
        if(s1<s2){
            swap(s1,s2);
            swap(a1,a2);
        }
        int d=s1-s2;
        int res=0;
        for(int i=6;d>0&&i>=2;--i){
            int t=a1[i]+a2[7-i];
            int s=t*(i-1);
            if(s<d){
                d-=s;
                res+=t;
            }
            else{                
                res+=d/(i-1);
                res+=(d%(i-1))!=0;
                d=0;
            }                
        }
        if(d==0)
            return res;
        return -1;
    }
};
