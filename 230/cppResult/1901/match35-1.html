class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2= nums2.size();
        if(n1>n2*6||n2>n1*6) return -1;
        int res = 0;
        int h[7];
        memset(h,0,sizeof h);
        vector<int> s1(n1+1);
        vector<int> s2(n2+1);
        for(int i=1;i<=n1;i++) s1[i]=s1[i-1]+nums1[i-1];

        for(int i=1;i<=n2;i++) s2[i]=s2[i-1]+nums2[i-1];
        
        if(s1[n1]>s2[n2]){
            for(int i=0;i<n1;i++){
                h[nums1[i]]++;
            }
            for(int i=0;i<n2;i++){
                h[7-nums2[i]]++;
            }
            int x = s1[n1]-s2[n2];
            int delta = 5;
            while(x>0&&delta>=1){
                if(x<=delta*h[delta+1]){
                    res+=x/delta;
                    if(x%delta!=0) res++;
                    break;
                }
                else{
                    res+=h[delta+1];
                    x-=delta*h[delta+1];
                    delta--;
                }

            }
        }
        else if(s1[n1]<s2[n2]){
            for(int i=0;i<n2;i++){
                h[nums2[i]]++;
            }
            for(int i=0;i<n1;i++){
                h[7-nums1[i]]++;
            }
            int x = -s1[n1]+s2[n2];
            int delta = 5;
            while(x>0&&delta>=1){
                if(x<=delta*h[delta+1]){
                    res+=x/delta;
                    if(x%delta!=0) res++;
                    break;
                }
                else{
                    res+=h[delta+1];
                    x-=delta*h[delta+1];
                    delta--;
                }
            }
        }
        
        return res;     
    }
};
