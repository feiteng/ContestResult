class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int len1 = nums1.size();
        int len2 = nums2.size();
        int a[6] = { 0,0,0,0,0,0 };
        int b[6] = { 0,0,0,0,0,0 };
        int sum1 = 0;
        int sum2 = 0;

        for (int i = 0;i < len1;i++) {
            a[nums1[i] - 1]++;
            sum1 = sum1 + nums1[i];
        }
        for (int i = 0;i < len2;i++) {
            b[nums2[i] - 1]++;
            sum2 = sum2 + nums2[i];
        }
        if (sum1 == sum2) {
            return 0;
        }

        else if (sum1 > sum2) {
            int gap = sum1 - sum2;
            int result = 0;
            for (int i = 0;i < 6;i++) {
                if ((a[5 - i] + b[i])*(5-i) >= gap) {
                    result = result + (gap-1) / (5 - i) + 1;
                    gap = 0;
                    break;
                }
                else {
                    gap = gap - (a[5 - i] + b[i]) * (5 - i);
                    result = result + (a[5 - i] + b[i]);
                }
            }
            if (gap == 0) {
                return result;
            }
            else {
                return -1;
            }

        }
        else {
            int gap = sum2 - sum1;
            int result = 0;
            for (int i = 0;i < 6;i++) {
                if ((a[i] + b[5-i]) * (5 - i) >= gap) {
                    result = result + (gap - 1) / (5 - i) + 1;
                    gap = 0;
                    break;
                }
                else {
                    gap = gap - (a[i] + b[5-i]) * (5 - i);
                    result = result + (a[i] + b[5-i]);
                }
            }
            if (gap == 0) {
                return result;
            }
            else {
                return -1;
            }
        }
    }
};
