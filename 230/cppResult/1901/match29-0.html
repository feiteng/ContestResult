class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        vector<int> cnt1(7), cnt2(7);
        int sum1 = 0, sum2 = 0;
        for(int x : nums1){
            cnt1[x] += 1;
            sum1 += x;
        }
        for(int x : nums2){
            cnt2[x] += 1;
            sum2 += x;
        }
        if(sum1 > sum2){
            swap(sum1, sum2);
            swap(cnt1, cnt2);
        }
        int ans = 0, d = sum2 - sum1;
        for(int i = 5; i >= 1 and d > 0; i -= 1){
            int x = cnt1[6 - i] + cnt2[i + 1];
            int h = min(x, (d + i - 1) / i);
            ans += h;
            d -= h * i;
        }
        if(d > 0) return -1;
        return ans;
    }
};
