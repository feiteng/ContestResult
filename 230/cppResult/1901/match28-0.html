class Solution {
public:
    int minOperations(vector<int>& A, vector<int>& B) {
        int n = A.size();
        int m = B.size();
        if (n > m * 6 || m > n * 6) return -1;
        int sa = 0;
        int sb = 0;
        for (auto k : A) sa += k;
        for (auto k : B) sb += k;
        if (sa > sb) {
            swap(A,B);
            swap(n,m);
            swap(sa,sb);
        }
        
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
        
        reverse(B.begin(), B.end());
        A.push_back(100);
        B.push_back(-100);
        
      //  cout << sa << " " << sb << endl;
        int ans = 0;
        int d = sb - sa;
        int i = 0, j = 0;
        while (d > 0) {
            ans++;
            int x = 6 - A[i];
            int y = B[j] - 1;
            if (x > y && i < n) {
                d -= x;
                i++;
            } else {
                d -= y;
                j++;
            }
        //    cout << d << " " << i << " " << j << endl;
        }
        return ans;
    }
};
