class Solution {
public:
int minOperations(vector<int>& nums1, vector<int>& nums2) {
	int n1 = nums1.size(), n2 = nums2.size();
	if (n2 > 6 * n1 || n1 > 6 * n2)return -1;
	int s = 0;
	for (auto x : nums1)s += x;
	for (auto x : nums2)s -= x;
	if (s == 0)return 0;
	if (s > 0) {
		swap(n1,n2);
		swap(nums1, nums2);
		s = -s;
	}
	sort(nums1.begin(), nums1.end());
	sort(nums2.begin(), nums2.end());
	reverse(nums2.begin(), nums2.end());
	s = -s;
	int i = 0, j = 0;
	int res = 0;
	while (i < n1 && j < n2) {
		int x = 6 - nums1[i];
		int y = nums2[j] - 1;
		if (x > y) {
			s -= x;
			i++;
		}
		else {
			s -= y;
			j++;
		}
		res++;
		if (s <= 0)return res;
	}
	while (i < n1) {
		int x = 6 - nums1[i];
		s -= x;
		i++;
		res++;
		if (s <= 0)return res;
	}
	while (i < n2) {
		int y = nums2[j] - 1;
		s -= y;
		j++;
		res++;
		if (s <= 0)return res;
	}
	return res;
}
};
