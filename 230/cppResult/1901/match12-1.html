class Solution {
public:
    int minOperations(vector<int>& a, vector<int>& b) {
        int n1=a.size(),n2=b.size();
        if(min(n1,n2)*6 < max(n1,n2)) return -1;

        int asum=0,bsum=0;
        for(int i:a) asum+=i;
        for(int i:b) bsum+=i;
        // cout<<asum<<" "<<bsum;
        if(asum==bsum) return 0;
        if(asum>bsum) {a.swap(b);swap(asum,bsum);}
        sort(a.begin(),a.end());
        sort(b.begin(),b.end(),greater<int>());
        int i=0,j=0,res=0,dif=bsum-asum;
        while(dif>0){

            if(i<a.size() && j<b.size()){
                if((6-a[i])>(b[j]-1)){
                    asum+=(6-a[i]);
                    i++;
                }else{
                    bsum-=(b[j]-1);
                    j++;
                }
                // cout<<"1"<<endl;
            }else if(i<a.size()){
                asum+=(6-a[i]);
                i++;
                // cout<<"2"<<endl;
            }else{
                bsum-=(b[j]-1);
                j++;
                // cout<<"3"<<endl;
            }
            res++;
            dif=bsum-asum;
        }
        return res;
    }
};
