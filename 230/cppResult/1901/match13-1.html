class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int cnt1[7] = {};
        int cnt2[7] = {};

        int sum1 = 0, sum2 = 0;
        for (int num : nums1)
        {
            sum1 += num;
            ++cnt1[num];
        }
        for (int num : nums2)
        {
            sum2 += num;
            ++cnt2[num];
        }
        if (sum1 == sum2) return 0;

        else if (sum1 < sum2)
        {
            swap(sum1, sum2);
            swap(cnt1, cnt2);
        }
        
        int res = 0;
        int diff = sum1 - sum2;
        for (int i = 6; diff && i > 1; --i)
        {
            int required = (diff - 1) / (i - 1) + 1;
            int available = cnt1[i] + cnt2[7 - i];
            if (diff > available * (i - 1)) {
                diff -= available * (i - 1);
                res += available;
            }
            else {
                diff = 0;
                res += required;
            }
        }
        if (diff == 0) return res;
        else return -1;
        
    }
};
