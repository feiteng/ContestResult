class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = 0;
        int sum2 = 0;
        for (auto it : nums1) sum1 += it;
        for (auto it : nums2) sum2 += it;
        if (sum1 == sum2) return 0;
        int del = abs(sum1 - sum2);
        if (sum1 > sum2) nums1.swap(nums2);
        // 1 want big
        vector<int> vec;
        for (auto it : nums1) {
            vec.push_back(6 - it);
        }

        for (auto it : nums2) {
            vec.push_back(it - 1);
        }

        sort(vec.begin(), vec.end());

        int cnt = 0;
        int sum = 0;
        for (int i = vec.size() - 1; i >= 0; i--) {
            sum += vec[i];
            cnt++;
            if (sum >= del) return cnt;
        }

        return -1;
    }
};
