class Solution {
public:
    int minOperations(vector<int>& a, vector<int>& b) {
        
        long long diff = 0;
        for (auto aa : a) diff += aa;
        for (auto bb : b) diff -= bb;
        
        if (diff < 0) {
            vector<int> temp = a;
            a = b;
            b = temp;
            diff = -diff;
        }
        
        if (a.size() * 1 > b.size() * 6) return -1;
        
        
        sort(a.rbegin(), a.rend());
        sort(b.begin(), b.end());
        
        int i = 0;
        int j = 0;
        

        int n = a.size();
        int m = b.size();

        int ans = 0;
        while (diff > 0) {
            // cout << i << " " << j << endl;
            ans++;
            if (i == n) {
                diff -= 6 - b[j++];
            } else if (j == m) {
                diff -= a[i++] - 1;
            } else {

                if (a[i] - 1 > 6 - b[j]) {
                    diff -= a[i++] - 1;
                } else {
                    diff -= 6 - b[j++];
                }
            }   
            
        }
        
        return ans;
        
    }
};
