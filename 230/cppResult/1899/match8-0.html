class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int cnt = 0;
        for (auto x : items) {
            if (x[0] == ruleValue && ruleKey == "type") cnt ++;
            if (x[1] == ruleValue && ruleKey == "color") cnt ++;
            if (x[2] == ruleValue && ruleKey == "name") cnt ++;
        }
        return cnt;
    }
};
