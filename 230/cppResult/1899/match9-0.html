class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans = 0;
        for(auto &v : items) {
            if(ruleKey == "type" and ruleValue == v[0]) ans ++;
            if(ruleKey == "color" and ruleValue == v[1]) ans ++;
            if(ruleKey == "name" and ruleValue == v[2]) ans ++;
        }
        return ans;
    }
};
