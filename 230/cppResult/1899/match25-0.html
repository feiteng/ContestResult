class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int cnt = 0;
        for (auto& item : items) {
            if (ruleKey == "type" && ruleValue == item[0]) ++cnt;
            else if (ruleKey == "color" && ruleValue == item[1]) ++cnt;
            else if (ruleKey == "name" && ruleValue == item[2]) ++cnt;
        }
        return cnt;
    }
};
