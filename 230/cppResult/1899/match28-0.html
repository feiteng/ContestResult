class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int cnt = 0;
        int t = 0;
        if (ruleKey == "color") t = 1;
        else if (ruleKey == "name") t = 2;
        for (auto i : items) {
            if (i[t] == ruleValue) cnt++;
        }
        return cnt;
    }
};
