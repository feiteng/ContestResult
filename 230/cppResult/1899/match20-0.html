class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans = 0;
        for (auto &item : items) {
            if (ruleKey == "type") ans += (ruleValue == item[0]);
            if (ruleKey == "color") ans += (ruleValue == item[1]);
            if (ruleKey == "name") ans += (ruleValue == item[2]);
        }
        
        return ans;
    }
};
