class Solution {
public:
    int countMatches(vector<vector<string>>& items, string rl, string rv) {
        int cnt = 0;
        for(auto v : items) {
            if(rl == "type" && rv == v[0]) cnt ++;
            else if(rl == "color" && rv == v[1]) cnt ++;
            else if(rl == "name" && rv == v[2]) cnt ++;
        }
        
        return cnt;
    }
};
