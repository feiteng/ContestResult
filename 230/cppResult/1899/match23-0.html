class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans = 0;
        for(auto& s:items)
        {
            if(ruleKey == "type" && s[0] == ruleValue) ans++;
            else if(ruleKey == "color" && s[1] == ruleValue) ans++;
            else if(ruleKey == "name" && s[2] == ruleValue) ans++;
        }
        return ans;
    }
};
