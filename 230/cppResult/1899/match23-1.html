class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans=0;
        for( auto p: items) {
            if (ruleKey=="type" && ruleValue == p[0]) ans++;
            else if (ruleKey=="color" && ruleValue == p[1]) ans++;
            else if (ruleKey=="name" && ruleValue==p[2]) ans++;
        }
        return ans;
    }
};
