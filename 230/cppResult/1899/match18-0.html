class Solution {
public:
    int countMatches(vector<vector<string>>& items, string s, string t) {
        int ret = 0;
        for (auto& v : items) {
            if (s == "type" && t == v[0]) ++ret;
            if (s == "color" && t == v[1]) ++ret;
            if (s == "name" && t == v[2]) ++ret;
        }
        return ret;
    }
};
