class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans = 0;
        for(auto &item : items){
            if(ruleKey == "type" && ruleValue == item[0]) ans += 1;
            if(ruleKey == "color" && ruleValue == item[1]) ans += 1;
            if(ruleKey == "name" && ruleValue == item[2]) ans += 1;
        }
        return ans;
    }
};
