class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int res=0;
        for(auto& v: items) {
            if(ruleKey=="type" && ruleValue==v[0]) ++res;            
            if(ruleKey=="color" && ruleValue==v[1]) ++res;
            if(ruleKey=="name" && ruleValue==v[2]) ++res;
        }
        return res;
    }
};
