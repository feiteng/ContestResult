class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans = 0;
        for (auto v : items) {
            if (ruleKey == "type")
                ans += v[0] == ruleValue;
            if (ruleKey == "color")
                ans += v[1] == ruleValue;
            if (ruleKey == "name")
                ans += v[2] == ruleValue;
        }
        return ans;
    }
};
