#include<cstring>
#include<iostream>

class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int ans=0;
        for (auto v:items) {
            if (ruleKey.compare("type")==0&&ruleValue.compare(v[0])==0)++ans;
            if (ruleKey.compare("color")==0&&ruleValue.compare(v[1])==0)++ans;
            if (ruleKey.compare("name")==0&&ruleValue.compare(v[2])==0)++ans;
        }
        return ans;
    }
};
