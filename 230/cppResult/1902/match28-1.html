class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& v) {
        int n = v.size();
        vector<double> ans(n, -1);
        stack<int> stk;
        for(int i = n - 1; i >= 0; i--) {
            while(stk.size()) {
                int idx = stk.top();
                if(v[i][1] <= v[idx][1]) {
                    stk.pop();
                    continue;
                }
                // cout<<i<<" "<<idx<<endl;
                double dist = v[idx][0] - v[i][0];
                double sp = v[i][1] - v[idx][1];
                double t = (dist / sp);
                // cout<<t<<endl;
                if(ans[idx] == -1 || t <= ans[idx]) {
                    break;
                }
                if(ans[idx] > 0)
                    stk.pop();
            }
            if(stk.size()) {
                int idx = stk.top();
                double dist = v[idx][0] - v[i][0];
                double sp = v[i][1] - v[idx][1];
                double t = (dist / sp);
                // cout<<t<<endl;
                ans[i] = t;
            }
            stk.push(i);
        }
        
        return ans;
    }
};
