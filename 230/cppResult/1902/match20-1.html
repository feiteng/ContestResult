class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n = cars.size();
        vector<double> ans(n, -1);
        vector<int> tree(n, -1);
        tree[n - 1] = -1;
        for (int i = n - 2; i >= 0; i--) {
            int id = i + 1;
            double s = -1;
            double t = 1e18;
            while (id != -1) {
                if (cars[i][1] <= cars[id][1]) {
                    id = tree[id];
                    continue;
                }
                else {
                    double temp = ((double)(cars[id][0] - cars[i][0])) / (cars[i][1] - cars[id][1]);
                    if (temp > t) {
                        break;
                    }
                    else {
                        t = temp;
                        tree[i] = id;
                        s = temp;
                    }
                }
                id = tree[id];
            }
            ans[i] = s;
        }
        return ans;
    }
};
