class Solution {
public:
    
    double cal(vector<int>& a, vector<int>& b) {
        return -((double)a[0] - b[0]) / (a[1] - b[1]);
    }
    
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        vector<int> Q;
        
        vector<double> ret(cars.size(), 0);
        int n = cars.size();
        for (int i = n - 1;i >= 0;i--) {
            while (Q.size() > 0) {
                int x = Q[Q.size() - 1];
                if (cars[i][1] <= cars[x][1]) {
                    Q.pop_back();
                } else {
                    break;
                }
            }
            while (Q.size() >= 2) {

                if (cal(cars[i], cars[Q[Q.size() - 1]]) >= cal(cars[i], cars[Q[Q.size() - 2]])) {
                    Q.pop_back();
                } else {
                    break;
                }
            } 
            if (Q.size() == 0) {
                ret[i] = -1;
            } else {
                ret[i] = cal(cars[i], cars[Q[Q.size() - 1]]);
            }
            Q.push_back(i);
        }
        return ret;
    }
};
