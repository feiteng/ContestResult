class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n=cars.size();
        vector<double> ret(n);
        vector<int> cach(n);
        stack<int> st;

        for(int i=n-1;i>=0;i--){
            while(!st.empty()&&cars[i][1]<=cars[st.top()][1]){
                st.pop();
            }
            if(st.empty())
                ret[i]=-1;
            else{
                int last=st.top();
                double t=(double)(cars[last][0]-cars[i][0])/(cars[i][1]-cars[last][1]);
                while(ret[last]!=-1&&ret[last]<(double)(cars[last][0]-cars[i][0])/(cars[i][1]-cars[last][1])){
                    last=cach[last];
                    t=(double)(cars[last][0]-cars[i][0])/(cars[i][1]-cars[last][1]);
                }
//                if(cars[i][1]>cars[i+1][1])
//                    ret[i]=(double)(cars[i+1][0]-cars[i][0])/(cars[i][1]-cars[i+1][1]);
//                else
//                    ret[i]=ret[i+1]+(double)(cars[i+1][0]-cars[i][0]+ret[i+1]*(cars[i+1][1]-cars[i][1]))/(cars[i][1]-speed[i+1]);
                cach[i]=last;
                ret[i]=t;
            }
            st.push(i);

        }return ret;
    }

};
