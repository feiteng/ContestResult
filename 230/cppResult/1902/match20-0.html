class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n=cars.size();
        vector<double>ans(n);
        stack<int>stk;
        ans[n-1]=-1;
        stk.push(n-1);
        for(int i=n-2;i>=0;i--){
            while(!stk.empty()&&cars[stk.top()][1]>=cars[i][1])stk.pop();
            if(stk.empty()){
                stk.push(i);
                ans[i]=-1;
                continue;
            }
            double tmp=1.0*(cars[stk.top()][0]-cars[i][0])/(cars[i][1]-cars[stk.top()][1]);
            while(!stk.empty()&&ans[stk.top()]>0&&tmp>=ans[stk.top()]){
                stk.pop();
                tmp=1.0*(cars[stk.top()][0]-cars[i][0])/(cars[i][1]-cars[stk.top()][1]);
            }
            stk.push(i);
            ans[i]=tmp;
        }
        return ans;
    }
};
