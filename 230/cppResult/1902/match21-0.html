using LL = long long;

class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n = cars.size();
        vector<double> ans(n);
        ans[n - 1] = -1.0;
        
        vector<int> conv = {n - 1};
        
        for (int i = n - 2; i >= 0; --i) {
            while (conv.size() > 1) {
                int m = conv.size();
                LL x1 = cars[conv[m - 1]][1] - cars[conv[m - 2]][1];
                LL y1 = cars[conv[m - 1]][0] - cars[conv[m - 2]][0];
                LL x2 = cars[i][1] - cars[conv[m - 1]][1];
                LL y2 = cars[i][0] - cars[conv[m - 1]][0];
                LL det = x1 * y2 - x2 * y1;
                if (det <= 0) {
                    conv.pop_back();
                }
                else {
                    break;
                }
            }
            if (cars[i][1] > cars[conv.back()][1]) {
                ans[i] = 1.0 * (cars[conv.back()][0] - cars[i][0]) / (cars[i][1] - cars[conv.back()][1]);
            }
            else {
                ans[i] = -1.0;
            }
            conv.push_back(i);
        }
        
        return ans;
    }
};
