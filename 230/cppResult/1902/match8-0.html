class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int v = INT_MAX;
        vector<double> res(cars.size(), INT_MAX);
        vector<int> fleet(cars.size(), -1);
        vector<int> stack;
        for (int i = cars.size() - 1; i > -1; i--) {
            while (!stack.empty() && cars[i][1] <= cars[stack.back()][1]) {
                stack.pop_back();
            }
            if (cars[i][1] <= v) {
                v = cars[i][1];
                res[i] = -1;
            }
            else {
                cout << i << " " << stack.size() << endl;
                // cout << stack.size()
                int j = stack.back();
                while (true) {
                    double t = (cars[j][0] - cars[i][0]) * 1.0 / (cars[i][1] - cars[j][1]);
                    if (res[i] > t) {
                        res[i] = t;
                        fleet[i] = j;
                    }
                    else break;
                    if (fleet[j] != -1) j = fleet[j];
                }
                
            }
            stack.push_back(i);
        }
        return res;
    }
};
