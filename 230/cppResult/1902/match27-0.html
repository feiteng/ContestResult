class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int N = cars.size();
        vector<double> ret(N, -1);
        vector<int> nextCar(N, -1);
        for (int i = N - 2; i >= 0; i--) {
            int nxt = i + 1;
            while (nxt >= 0) {
                int spdDiff = cars[i][1] - cars[nxt][1];
                if (spdDiff > 0) {
                    int posDiff = cars[nxt][0] - cars[i][0];
                    double time = (double)posDiff / spdDiff;
                    if (ret[i] < 0 || time < ret[i]) {
                        ret[i] = (double)posDiff / spdDiff;
                        nextCar[i] = nxt;
                    } else {
                        break;
                    }
                }
                nxt = nextCar[nxt];
            }
        }
        return ret;
    }
};
