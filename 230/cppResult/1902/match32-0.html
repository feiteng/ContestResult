class Solution {
public:
    #define maxn 100010
    int sx[maxn],sy[maxn],idx=0;
    vector<double> getCollisionTimes(vector<vector<int>>& a) {
        int n=a.size();
        vector<double>res(n,-1);
        for(int i=n-1;i>=0;--i) {
            while(idx>=2) {
                if(((sx[idx]-a[i][0])*(sy[idx-1]-sy[idx])<=(sx[idx-1]-sx[idx])*(sy[idx]-a[i][1])))
                    --idx;
                else break;
            }
            if(idx) res[i]=-1.0*(sx[idx]-a[i][0])/(sy[idx]-a[i][1]);
            sx[++idx]=a[i][0];sy[idx]=a[i][1];
            if(res[i]<0) res[i]=-1;
        }
        return res;
    }
};
