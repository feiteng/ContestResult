class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& A) {
        int n = A.size();
        set<pair<double,int>> S;
        set<int> B;
        for (int i = 0; i < n - 1; i++) {
            int d = A[i+1][0] - A[i][0];
            double delta = A[i][1] - A[i+1][1];
            
            if (delta > 0) {
              S.insert(make_pair(d/delta, i));
         //       cout << d/delta << " " << i << " " << i + 1 << endl;
            }
            
            
            B.insert(i);
        }
        B.insert(n-1);
        
        vector<double> ans(n, -1);
        
        while (!S.empty()) {
            auto t = S.begin();
            double cost = t->first;
            int x = t->second;
            S.erase(t);
            if (ans[x] >= 0) continue;
            ans[x] = cost;
            
            B.erase(x);
            
            auto tt = B.lower_bound(x);
            if (tt == B.begin() || tt == B.end()) continue;
            tt--;
            
            x = *tt;
            tt++;
            int y = *tt;
            
            int d = A[y][0] - A[x][0];
            double delta = A[x][1] - A[y][1];
            
            if (delta > 0) {
              S.insert(make_pair(d/delta, x));
         //       cout << d/delta << " " << x << " " << y << endl;
            }
        }
        
        return ans;
        
    }
};
