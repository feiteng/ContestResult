class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
	vector<pair<LL, LL> > cht;
	vector<double> ans(cars.size());
	for (int i=cars.size(); i--;) {
	    LL pos = cars[i][0];
	    LL speed = cars[i][1];
	    while (cht.size() && cht.back().first >= speed) cht.pop_back();

	    while (cht.size() >= 2u) {
		int len = cht.size();
		LL Pos1 = cht[len-1].second - pos;
		LL Sp1 = speed - cht[len-1].first;
		LL Pos0 = cht[len-2].second - pos;
		LL Sp0 = speed - cht[len-2].first;

		if (Pos0 * Sp1 <= Pos1 * Sp0) {
		    cht.pop_back();
		} else {
		    break;
		}
	    }

	    if (cht.empty()) {
		ans[i] = -1;
	    } else {
		int len = cht.size();
		LL Pos1 = cht[len-1].second - pos;
		LL Sp1 = speed - cht[len-1].first;
		ans[i] = (double)Pos1 / Sp1;
	    }
	    cht.emplace_back(speed, pos);
	}
        
	return ans;
    }
};
