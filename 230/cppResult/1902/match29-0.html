class Solution {
public:
    typedef long long LL;

bool isRight(pair<LL, LL>& A, pair<LL, LL>& B, pair<LL, LL>& C)
{
    return A.first * B.second - A.second * B.first +
    B.first * C.second - B.second * C.first +
    C.first * A.second - C.second * A.first <= 0;
}

vector<double> getCollisionTimes(vector<vector<int> >& cars)
{
    int N = (int)cars.size();
    vector<double> ans(N, -1);
    vector<pair<LL, LL> > stk; //-speed, position
    stk.push_back({-cars.back()[1], cars.back()[0]});
    for(int i = N - 2; i >= 0; i--)
    {
        pair<LL, LL> cur = {-cars[i][1], cars[i][0]};
        int L = (int)stk.size();
        while(!stk.empty() && (cur.first >= stk.back().first
                               || (L >= 2 && isRight(cur, stk[L - 1], stk[L - 2]))))
        {
            stk.pop_back();
            L--;
        }
        if(!stk.empty())
            ans[i] = 1.0 * (stk.back().second - cur.second) / (stk.back().first - cur.first);
        stk.push_back(cur);
    }
    return ans;
}
};
