class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n = cars.size();
        vector<double> ans(n, -1);
        vector<int> r(n+1, n);
        stack<int> st;
        for (int i = n-1; i >= 0; i--){
            while (!st.empty() && cars[st.top()][1] >= cars[i][1]){
                st.pop();
            }
            if (!st.empty()) r[i] = st.top(); 
            st.push(i);
        }
        for (int j = n-1; j >= 0; j--){
            int i = j;
            if (r[i] == n) continue;
            double dis = cars[r[i]][0] - cars[i][0];
            double speed = cars[i][1] - cars[r[i]][1];
            ans[i] = dis/speed;
            while (ans[r[i]] != -1 && ans[i] > ans[r[i]]){
                r[i] = r[r[i]];
                dis = cars[r[i]][0] - cars[i][0];
                speed = cars[i][1] - cars[r[i]][1];
                ans[i] = dis/speed;
            }
        }
        return ans;
    }
};
