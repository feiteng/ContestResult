class Solution {
public:
struct trip{
    int a,b;
    double r;
    trip(){}
    trip(int a,int b,double r):a(a),b(b),r(r){}
};

vector<double> getCollisionTimes(vector<vector<int>>& cars) {
    int n = cars.size();
    vector<double> res(n);
    stack<trip> st;
    for(int i=n-1;i>=0;i--){
        while(!st.empty()&&
            ((cars[i][1]<=st.top().a)||
            (1.0*(st.top().b-cars[i][0])/(cars[i][1]-st.top().a)>st.top().r))){
            st.pop();
        }
        if(st.empty()){
            res[i]=-1.0;
            st.push(trip(cars[i][1],cars[i][0],1.0/0));

        } else {
            res[i]=1.0*(st.top().b-cars[i][0])/(cars[i][1]-st.top().a);
            st.push(trip(cars[i][1],cars[i][0],res[i]));
        }
    }
    return res;
}
};
