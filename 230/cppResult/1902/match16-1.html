class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n = cars.size();
        vector<double> ans(n, 1.0e20);
        stack<int> s;
        for(int i = n - 1; i >= 0; --i){
            while(s.size() && ans[s.top()] > 0 && (cars[i][1] <= cars[s.top()][1] || (double)(cars[s.top()][0] - cars[i][0]) / (double)(cars[i][1] - cars[s.top()][1]) > ans[s.top()])){ // cannot catch up
                s.pop();
            }
            if(s.empty() || cars[i][1] <= cars[s.top()][1]){
                ans[i] = -1.0;
            } else {
                ans[i] = (double)(cars[s.top()][0] - cars[i][0]) / (double)(cars[i][1] - cars[s.top()][1]);
            }
            s.push(i);
        }
        return ans;
    }
};
