class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& a) {
        const int n = a.size();
        vector<double> ans(n, 1e9);
        const double inf = ans[0];
        
        auto f = [&](int i, int j) {
            if (a[i][1] <= a[j][1]) return inf;
            return 1.0 * (a[j][0] - a[i][0]) / (a[i][1] - a[j][1]);
        };
        
        vector<int> q(n + 10);
        int s = 0, e = 0;
        q[e++] = n - 1;



        for (int i = n - 2; i >= 0; --i) {
            while (e - s >= 2) {
                if (f(i, q[e - 1]) >= f(i, q[e - 2])) --e;
                else break;
            }
            ans[i] = f(i, q[e - 1]);
            while (e - s > 0 && a[q[e-1]][1] >= a[i][1]) --e; 
            q[e++] = i;
        }
        for (auto &x : ans) if (x == inf) x = -1;
        return ans;
    }
};
