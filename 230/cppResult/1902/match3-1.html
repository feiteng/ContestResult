class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) 
    {
        int N = cars.size();
        vector<double> ret; ret.resize(N, -1);
        vector<int> pre; pre.resize(N);
        for (int i = 0; i < N; i ++) pre[i] = i-1;
        /*
        vector<pair<int, int> > a; a.resize(N);
        for (int i = 0; i < N; i ++)
            a[i] = make_pair(cars[i][1], i);
        sort(a.begin(), a.end());
        for (int i = 0; i < N; i ++)
        {
            int x = a[i].second;
            while (x > 0 && ret[x-1] == -1)
            {

        }
        
        stack<pair<double, pair<int, int> > > s;

        {
            if (!)
        }
        */
        priority_queue <pair<double, pair<int, int> > >h;
        for (int i = 0; i+1 < N; i ++)
            if (cars[i][1] > cars[i+1][1])
                h.push(make_pair(1.0*(cars[i][0]-cars[i+1][0])/(cars[i][1]-cars[i+1][1]), make_pair(i, i+1)));
        while (!h.empty())
        {
            auto t = h.top(); h.pop();
            int x = t.second.first, y = t.second.second;
            if (ret[x] == -1)
            {
                ret[x] = -t.first;
                pre[y] = pre[x];
                if (pre[x] >= 0 && cars[pre[x]][1] > cars[y][1])
                    h.push(make_pair(1.0*(cars[pre[x]][0]-cars[y][0])/(cars[pre[x]][1] - cars[y][1]), make_pair(pre[x], y)));
            }
        }
        return ret;
    }
};
