class Solution {
public:
    vector<double> getCollisionTimes(vector<vector<int>>& cars) {
        int n = cars.size();
        vector<double> ans(n, -1);
        vector<int> now(n, -1);
        reverse(cars.begin(), cars.end());
        for (int i = 1; i < n; ++i) {
            int t = i-1;
            while (t != -1 && cars[i][1] <= cars[t][1]) {
                t = now[t];
            }
            if (t == -1) continue;
            double t1 = (double)(cars[t][0] - cars[i][0]) / (double)(cars[i][1] - cars[t][1]);
            while (ans[t] != -1.0 && t1 > ans[t]) {
                t = now[t];
                t1 = (double)(cars[t][0] - cars[i][0]) / (double)(cars[i][1] - cars[t][1]);
            }
            ans[i] = t1;
            now[i] = t;
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
