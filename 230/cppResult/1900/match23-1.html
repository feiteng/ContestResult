class Solution {
public:
    vector<int> build(vector<int>& toppingCosts) {
        vector<int> ret;
        ret.push_back(0);

        for (int i = 0;i < toppingCosts.size();i++) {
            int x = toppingCosts[i];
            int n = ret.size();
            for (int j = 0;j < n;j++) {
                for (int k = 1;k <= 2;k++) {
                    ret.push_back(ret[j] + k * x);
                }
            }
        }
        return ret;
    }
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {

        int ret = -1000000000;
        vector<int> t = build(toppingCosts);
        for (int i = 0;i < baseCosts.size();i++) {
            int temp = 0;
            
            for (int &x: t) {

                temp = baseCosts[i] + x;
                if (abs(ret - target) > abs(target - temp)) {
                    ret = temp;
                } else if (abs(ret - target) == abs(target - temp) && ret > temp) {
                    ret = temp;
                }
            }
        }
        return ret;
    }
};
