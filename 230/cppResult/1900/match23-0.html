class Solution {
public:
    int closestCost(vector<int>& b, vector<int>& c, int target) {
        int n = b.size();
        int m = c.size();
        int ans = -1;
        for (int i = 0; i < n; ++i) {
            int mx = 1;
            for (int j = 0; j < m; ++j) mx *= 3;
            for (int j = 0; j < mx; ++j) {
                int bb = j;
                int cur = b[i];
                for (int k = 0; k < m; ++k) {
                    cur += c[k] * (bb % 3);
                    bb /= 3;
                }
                if (ans < 0) ans = cur;
                if (abs(ans-target) > abs(cur-target)) {
                    ans = cur;
                    //printf("update1 i=%d j=%d cur=%d\n", i, j, cur);
                } else if (abs(ans-target) == abs(cur-target) && cur < ans) {
                    ans = cur;
                    //printf("update2 i=%d j=%d cur=%d\n", i, j, cur);
                }
            }
        }
        return ans;
    }
};
