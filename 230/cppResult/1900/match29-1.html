class Solution {
public:
    int ret = 0;

    int n,m;
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        ret = baseCosts[0];
        n = baseCosts.size();
        m = toppingCosts.size();
        DFS(baseCosts,toppingCosts,target,0,0);
        return ret;
    }
    void DFS(vector<int>& baseCosts, vector<int>& toppingCosts, int target, int cur, int sum){
        //cout << m << cur << sum << endl;
        if(cur >= m){
            int temp = abs(ret - target); 
            for(int& i: baseCosts){

                int t = abs(i + sum -target);
                //cout << t << temp << "DFGH" << endl;
                if(t < temp){
                    ret = i+sum;
                    temp = abs(ret - target); 
                } 
                else if(t == temp && i + sum < target){
                    ret = i + sum;
                    abs(ret - target); 
                }
            }
            //cout << ret << endl;
            return;
        }
        DFS(baseCosts,toppingCosts,target,cur+1,sum);
        DFS(baseCosts,toppingCosts,target,cur+1,sum + toppingCosts[cur]);
        DFS(baseCosts,toppingCosts,target,cur+1,sum + 2 * toppingCosts[cur]);
    }
};
