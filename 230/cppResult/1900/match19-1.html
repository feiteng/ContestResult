class Solution {
public:
    vector<int> temp;
    void gen(int idx, vector<int> &a, int sum){
        if(idx == a.size()){
            temp.push_back(sum);
            return;
        }

        for(int i=0; i<=2; i++)
            gen(idx+1,a,sum+i*a[idx]);
    }
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int best = 0x3f3f3f3f;
        temp.clear();
        gen(0,toppingCosts,0);
        for(int i=0; i<baseCosts.size(); i++){
            for(int check:temp){

                int tot = check+baseCosts[i];
                if(abs(tot-target) < abs(best-target)) best = tot;
                else if(abs(tot-target) == abs(best-target) && tot < best) best = tot;
            }
        }
        return best;
    }
};
