class Solution {
public:
    int closestCost(vector<int>& bc, vector<int>& tc, int t) {
        vector<int> pow3(11);
        pow3[0] = 1;
        for(int i =0; i < 10; i++) {
            pow3[i+1] = pow3[i] * 3;
        }
        int ret = -1;
        for(auto b : bc) {
            for(int j = 0; j < pow3[tc.size()]; j++) {
                int tmp = b;

                int x = j;
                for(int k = 0; k < tc.size(); k++) {
                    tmp += x % 3 * tc[k];
                    x /=3;
                }
                if (ret < 0 || abs(tmp-t) < abs(ret-t) || tmp < ret && abs(tmp-t) == abs(ret-t)) {
                    ret = tmp;
                }
            }
        }
        return ret;
    }
};
