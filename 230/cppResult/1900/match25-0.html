class Solution {
    vector<int> toppingCosts;
    int ans, tar;
    void dfs(int step, int now) {
        if (step == toppingCosts.size()) {
            if (abs(now - tar) < abs(ans - tar))
                ans = now;
            else if (abs(now - tar) == abs(ans - tar) && now < ans)
                ans = now;
            return;
        }
        dfs(step + 1, now);
        dfs(step + 1, now + toppingCosts[step]);
        dfs(step + 1, now + toppingCosts[step] * 2);
    }
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        this->toppingCosts = toppingCosts;
        ans = baseCosts[0];
        tar = target;
        for (auto &i : baseCosts)
            dfs(0, i);
        return ans;
    }
};
