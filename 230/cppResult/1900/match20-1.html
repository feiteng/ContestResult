class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        
        int m = baseCosts.size();
        int n = toppingCosts.size();
        

        vector<vector<int>> dp(n + 1, vector<int> (10001, 0));
        dp[0][0] = 1;
        
        for(int i = 1; i <= n; i++){
        
            int c = toppingCosts[i - 1];
            
            for(int j = 0; j <= 10000; j++){
                
            
                dp[i][j] = dp[i - 1][j];
                
                if(j - c >= 0){
                    dp[i][j] = max(dp[i][j], dp[i - 1][j - c]);    
                }
                
                if(j - 2 * c >= 0){
                    dp[i][j] = max(dp[i][j], dp[i - 1][j - 2 * c]);    
                }
                
                
            }
            
            
        }
        
         int res = baseCosts[0];
        

        for(int i = 0; i < baseCosts.size(); i++){
            
            int b = baseCosts[i];
            
            for(int j = 0; j <= 10000; j++){
                
                if(dp[n][j] && abs(b + j - target) < abs(res - target)){
                    res = b + j;
                }
                
            }
        }
        
        return res;
        
    }
};
