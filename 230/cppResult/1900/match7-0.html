class Solution {
    int power(int a , int b){
        int prod = 1;
        while(b){
            if(b & 1){
                prod *= a;
            }
            a *= a;
            b >>= 1;
        }
        
        return prod;
    }
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int n = baseCosts.size();
        int m = toppingCosts.size();
        
        int ans = (int)(1e9) , min_diff = (int)(1e9);
        
        for(int x = 0 ; x < n ; x++){
            for(int i = 0 ; i < power(3 , m) ; i++) {
                int tmp = i;
                int sum = baseCosts[x];
                for(int j = 0 ; j < m ; j++){
                    sum += toppingCosts[j] * (tmp%3);
                    tmp /= 3;
                }
                
                if(abs(target - sum) < min_diff) {
                    min_diff = abs(sum - target);
                    ans = sum;
                }
                else if(abs(target - sum) == min_diff) {
                    ans = min(ans , sum);
                }
            }
        }
        
        return ans;
    }
};
