class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int res = baseCosts[0];
        for(int basecost: baseCosts){
            int tmp = basecost;
            addtopping(0, tmp, toppingCosts, target, res);
        }
        return res;
    }
private:
    void addtopping(int i, int &tmp, vector<int>& toppingCosts, int &target, int &res){
        if(i == toppingCosts.size()){
            if(abs(tmp - target) < abs(res - target)){
                res = tmp;
            }else if(abs(tmp - target) == abs(res - target)){
                res = min(res, tmp);
            }
            return;
        }
        for(int cnt = 0; cnt < 3; ++cnt){
            tmp += toppingCosts[i] * cnt;
            addtopping(i + 1, tmp, toppingCosts, target, res);
            tmp -= toppingCosts[i] * cnt;
        }
        return;
    }
};
