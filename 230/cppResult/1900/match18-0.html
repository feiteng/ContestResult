class Solution {
public:
    int closestCost(vector<int>& b, vector<int>& a, int t) {
        int n = b.size(), m = a.size();
        pair<int, int> ans(1e9, 1e9);
        int mx = 1;
        for (int i = 0; i < m; ++i)
            mx *= 3;
        for (int i = 0; i < n; ++i) {
            for (int S = 0; S < mx; ++S) {
                int T = S, w = b[i];
                for (int j = 0; j < m; ++j, T /= 3) {
                    w += T % 3 * a[j];
                }
                ans = min(ans, make_pair(abs(w-t), w));
            }
        }
        return ans.second;
    }
};
