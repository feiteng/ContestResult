class Solution {
public:
    vector<int> tops;
    int best = INT_MAX;
    int res = -1;

    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        
        int n = baseCosts.size();
        int m = toppingCosts.size();
        tops = toppingCosts;
        for (int i = 0; i < n; ++i)
        {
            DFS(0, target, baseCosts[i]);
        }
        
        return res;
    }
    

    void DFS(int index, int target, int curCost)
    {
        if (index >= tops.size())
        {
            if (abs(curCost - target) < best)
            {
                best = abs(curCost - target);
                res = curCost;
            }
            else if (abs(curCost - target) == best && curCost < res)
            {
                res = curCost;
            }
            return;
        }
        
        DFS(index + 1, target, curCost);
        DFS(index + 1, target, curCost + tops[index]);
        DFS(index + 1, target, curCost + tops[index] * 2);
    }
};
