class Solution {
public:
    int ans = -1e9;
    void dfs(vector<int>& toppingCosts, int target, int u, int sum){
        if(u == toppingCosts.size()){
            //cout << sum << endl;
            if(abs(target - sum) < abs(target - ans)) ans = sum;
            else if(abs(target - sum) == abs(target - ans)) ans = min(ans, sum);
            return;
        }
        dfs(toppingCosts, target, u + 1, sum);
        dfs(toppingCosts, target, u + 1, sum + toppingCosts[u]);
        dfs(toppingCosts, target, u + 1, sum + toppingCosts[u] * 2);
    }
    
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        for(int i = 0; i < baseCosts.size(); i++)
            dfs(toppingCosts, target, 0, baseCosts[i]);
        
        return ans;
    }
};
