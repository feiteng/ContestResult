class Solution {
public:
    void dfs(vector<int> tc, int j, int temp, int &fd, int &ans, int target) {
        if(fd >= abs(target - temp)) {
            if(fd == abs(target - temp)) {
                ans =  min(ans, temp);
            } else {
                ans = temp;
            }
            fd = abs(target -  temp);
        }
        if(j == tc.size()) return;
        for(int k = 0; k <= 2; k++) {
            dfs(tc, j + 1, temp + tc[j]*k, fd, ans, target);
        }
    }
    int closestCost(vector<int>& bc, vector<int>& tc, int target) {
        int finalclose = 1e7;
        int ans = 1e7;
        for(int i = 0; i < bc.size(); i++) {
            dfs( tc, 0 ,bc[i],  finalclose, ans, target);
        }
        return ans;
    }
};
