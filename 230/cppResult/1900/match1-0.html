class Solution {
public:

    vector<int> tc;
    vector<int> bc;
    int delta = INT_MAX;
    int res;
    int t;
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        t = target;
        bc = baseCosts;
        tc = toppingCosts;
        dfs(0, 0);
        return res;
    }

    void dfs(int index, int sum) {
        if(index >= tc.size()) {
            for(auto b :  bc) {
                auto tmp = b + sum;
                if(abs(tmp - t) < delta) {
                    delta = abs(tmp - t);
                    res = tmp;
                } else if(abs(tmp - t) == delta && tmp < res) {
                    res = tmp;
                }
            }
            return;
        }
        dfs(index + 1, sum);
        dfs(index + 1, sum + tc[index]);
        dfs(index + 1, sum + 2 * tc[index]);
    }
};
