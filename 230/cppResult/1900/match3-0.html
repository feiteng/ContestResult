class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int diff = -1, res = -1;
        for (const int& base : baseCosts) {
            dfs(toppingCosts, target, diff, res, base, 0, 0);
        }
        return res;
    }
    
    void dfs(vector<int>& toppingCosts, int target, int& diff, int& res, int cur, int depth, int idx) {
        if (diff == -1 || diff > abs(target - cur)) {
            diff = abs(target - cur);
            res = cur;
        } else if (diff == abs(target - cur)) {
            res = min(res, cur);
        }
        if (idx >= toppingCosts.size())
            return;
        for (int i = 0; i < 3; ++i) {
            dfs(toppingCosts, target, diff, res, cur + i * toppingCosts[idx], depth + 1, idx + 1);
        }
    }
};
