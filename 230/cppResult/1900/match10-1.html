class Solution {
public:
void findBestCost(vector<int>& toppingCosts, int curr_topping, int target, int curr_cost, int* best_cost) {
   if (abs(curr_cost - target) < abs(*best_cost - target) || (abs(curr_cost - target) == abs(*best_cost - target) && curr_cost < *best_cost)) {

      *best_cost = curr_cost;
   }
   if (curr_cost >= target || curr_topping >= toppingCosts.size()) {
      return;
   }
   for (int j = 0; j <= 2; j++) {
      findBestCost(toppingCosts, curr_topping + 1, target, curr_cost + j * toppingCosts[curr_topping], best_cost);
   }
}

int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {


   int best_cost = baseCosts[0];
   for (int i = 0; i < baseCosts.size(); i++) {
      int curr_cost = baseCosts[i];
      findBestCost(toppingCosts, 0, target, curr_cost, &best_cost);
   }
   return best_cost;
}


};
