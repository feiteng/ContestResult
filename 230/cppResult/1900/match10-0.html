#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define ULL unsigned long long
class Solution {
public:
    int ans;
    int gap;
    void dfs(int i, vector<int>& toppingCosts, int target, int curSum) {
        if (abs(target - curSum) < gap || abs(target - curSum) == gap
            && curSum < gap) {
            ans = curSum;
            gap = abs(target - curSum);
        }
        if (i >= toppingCosts.size()) {
            return;
        }
        for (int j = 0; j <= 2; j++) {
            dfs(i + 1, toppingCosts, target, curSum + toppingCosts[i] * j);
        }
    }
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        gap = 0x3FFFFFFF;
        ans = 0;
        for (int i = 0; i < baseCosts.size(); i++) {
            int cur = baseCosts[i];
            dfs(0, toppingCosts, target, cur);
        }
        return ans;
    }
};
