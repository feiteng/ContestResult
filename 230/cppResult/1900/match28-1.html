class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        const int N = 2*(1e4);
        vector<bool> dp(N+1,  false);
        for (auto b : baseCosts) {
            dp[b] = true;
        }

        for (auto t : toppingCosts) {
            for (int i = N; i >= t; i--) {
                if (dp[i-t]) dp[i] = true;
            }
            for (int i = N; i >= t; i--) {
                if (dp[i-t]) dp[i] = true;
            }
        }
        
        int ans = INT_MAX;
        for (int i = 1; i <= N; i++) {
            if (dp[i] && abs(ans-target) > abs(i-target)) ans = i;
        }
        
        
        return ans;
    }
};
