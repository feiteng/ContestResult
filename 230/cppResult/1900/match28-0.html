class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        vector<bool> dp((int)3e4+10,false);
        for(int x:baseCosts) dp[x]=true;
        for(int x:toppingCosts) {
            for(int i=dp.size()-1;i>=x;i--) {
                if(dp[i-x]) dp[i]=true;
            }
            for(int i=dp.size()-1;i>=x;i--) {
                if(dp[i-x]) dp[i]=true;
            }
        }
        int ans=1e6;
        for(int i=0;i<dp.size();i++) {
            if(dp[i]&&abs(target-ans)>abs(target-i)) ans=i;
        }
        return ans;
    }
};
