class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int ans = 0  ;
        int n = toppingCosts.size();
        sort(baseCosts.begin(),baseCosts.end());
        ans = baseCosts.front();
        int mn = INT_MAX;
        for(int i = 0 ;i < 1<<n ;i ++){
            int pre = 0;
            for(int k= 0 ;k < n ;k++){
                if(i&(1<<k))
                    pre+=toppingCosts[k];
            }
            for(int j = 0 ;j < 1<<n ;j ++){
                int temp = 0;
                for(int k = 0 ;k < n ; k++){
                    if(j&(1<<k)){
                        temp+=toppingCosts[k];
                    }
                }
                temp+=pre;
                for(int itr : baseCosts){
                    if(mn>abs(target-temp-itr) || mn==abs(target-temp-itr) && ans>temp+itr){
                        ans = temp+itr;
                        mn= abs(target-temp-itr);
                    }
                        
                }
            }
        }
        return ans;
    }
};
