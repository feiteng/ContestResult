class Solution {
public:
    int closestCost(vector<int>& b, vector<int>& t, int C) {

        int ans = 1e8;

        int n = b.size();
        int m = t.size();


        int M=1;
        for(int i=1;i<=m;i++)
            M*=3;
        //M-=1;// M/=2;
        
        //d3(n,m,M);

            for(int j=0;j<M;j++)
            {
                int x = j;int tmp = 0;
                for(int k=0;k<m;k++)
                {
                    tmp += t[k]*(x%3);
                    x/=3;
                }
               // d2(j,tmp);
                for(int i=0;i<n;i++)
                {
                    int tmp1 = tmp + b[i];
                    if(abs(C-tmp1)<abs(C-ans)||(abs(C-tmp1)==abs(C-ans) && tmp1<ans))
                        ans = tmp1;
                }
            }
        
        return ans;
    }
};
