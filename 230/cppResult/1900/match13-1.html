class Solution {
public:
    int gap = INT_MAX;
    int res = 0;
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        for (auto base : baseCosts) {

            dfs(toppingCosts, 0, target, base);    
        }
        return res;
    }
    
    void dfs(vector<int>& toppingCosts, int idx, int target, int cur) {
        if (idx==toppingCosts.size()) {
            if (abs(cur -target) < gap) {
                gap = abs(cur -target);
                res = cur;
            }    
            return;
        }
        
        int topping = toppingCosts[idx];
        dfs(toppingCosts, idx+1, target, cur);
        dfs(toppingCosts, idx+1, target, cur+topping);
        dfs(toppingCosts, idx+1, target, cur+2*topping);
        
    }
    
    
};
