class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
    	int n = baseCosts.size(), m = toppingCosts.size();
    	int s = 1; for(int i=0;i<m;i++) s *= 3;

    	int ans = 1e9, res;
    	for(int i=0;i<s;i++){
    		int x = i, y = 0;
    		for(int j=0;j<m;j++){
    			y += x % 3 * toppingCosts[j];
    			x /= 3;
    		}
    		for(int j=0;j<n;j++){

    			int t = y + baseCosts[j];
    			if(abs(t-target)<ans||abs(t-target)==ans&&t<res){
    				ans = abs(t-target);
    				res = t;
    			}
    		}
    	}
    	return res;
    }
};
