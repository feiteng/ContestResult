class Solution {
public:
    int ret = 1000000;
    
    void dfs(vector<int>& toppingCosts, int price, int p, int target) {
        if (p == toppingCosts.size()) return;
        
        for (int i = 0; i <= 2; ++i) {
            int new_price = price + toppingCosts[p] * i;
            if (abs(target - new_price) < abs(target - ret)) {
                ret = new_price;
            }
            
            dfs(toppingCosts, new_price, p + 1, target);
        }
    }
    
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        for (int base_idx = 0; base_idx < baseCosts.size(); ++base_idx) {
            int price = baseCosts[base_idx];
            
            dfs(toppingCosts, price, 0, target);
        }
        return ret;
    }
};
