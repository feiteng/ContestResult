class Solution {
public:
    int diff=INT_MAX;
    int res=INT_MAX;
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        for(int i=0;i<baseCosts.size();++i){
            dfs(toppingCosts,target,0,baseCosts[i]);
        }
        
        return res;
    }
    
    void dfs(vector<int>& topping, int target, int idx, int sum){
        if(idx == topping.size()){
            if(abs(sum-target) < diff){
                diff=abs(sum-target);
                res=sum;
            }
            return ;
        }
        
        for(int i=idx;i<topping.size();++i){
            for(int k=0;k<=2;++k){
                dfs(topping,target,i+1,sum+k*topping[i]);
            }
        }
    }
};
