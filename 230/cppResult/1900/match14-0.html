class Solution {
public:
    void dfs(vector<int>& toppingCosts, int start, int cur, unordered_set<int>& sum) {
        if (start == toppingCosts.size()) {
            sum.emplace(cur); 
            return; 
        }
        dfs(toppingCosts, start + 1, cur, sum); 
        dfs(toppingCosts, start + 1, cur + toppingCosts[start], sum); 
        dfs(toppingCosts, start + 1, cur + toppingCosts[start] + toppingCosts[start], sum); 
    }
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        unordered_set<int> sum; 
        dfs(toppingCosts, 0, 0, sum); 
        for (int tmp : sum) {
            cout << tmp << endl; 
        }
        int ans = 1e9; 
        for (int bc : baseCosts) {
            for (int tmp : sum) {
                int temp = tmp + bc; 
                if (temp == target) {
                    return target; 
                }
                if (abs(temp - target) < abs(ans - target) || (abs(temp - target) == abs(ans - target) && temp < ans)) {
                    ans = temp; 
                }
            }
        }
        return ans; 
    }
};
