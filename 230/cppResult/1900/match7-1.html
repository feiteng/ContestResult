class Solution {
    
    vector<vector<int>> prices;
    

public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int n = baseCosts.size();
        int m = toppingCosts.size();
        int diff = 1e9+7;
        int ans = 1e9+7;

        for (int base = 0; base < n; ++base) {
            for (int top = 0; top < pow(3, m); ++top) {
                int price = baseCosts[base];
                int cur = top;
                for (int ind = 0; ind < m; ++ind) {
                    int rem = cur % 3;

                    cur /= 3;
                    price += toppingCosts[ind] * rem;
                }
                if (abs(target - price) < diff) {
                    diff = abs(target - price);
                    ans = price;
                }
                else if (abs(target - price) == diff && price < ans) {
                    ans = price;
                }
            }
        }
        return ans;
    }
};
