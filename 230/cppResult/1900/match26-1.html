class Solution {
public:

    int closestCost(vector<int>& base, vector<int>& top, int target) {
        unordered_set<int> all_comb;
        all_comb.insert(0);
        for (auto& t:top) {
            vector<int> tmp;
            for (auto& item:all_comb) {
                tmp.push_back(item + t);
                tmp.push_back(item + 2*t);
            }
            for (auto& item:tmp) {
                all_comb.insert(item);
            }
        }
        int ans = INT_MAX;
        int diff = INT_MAX;
        for (auto& b:base) {
            for (auto& item:all_comb) {

                int sum = b + item;
                if (abs(sum - target) < diff) {
                    diff = abs(sum - target);
                    ans = sum;
                } else if (abs(sum - target) == diff && sum < ans) {
                    ans = sum;
                }
            }
        }
        return ans;
    }
};
