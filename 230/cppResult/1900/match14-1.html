class Solution {
public:
    void ps(vector<int>& pos, int i, int cur, vector<int>& t) {
        if(i >=t.size()) {
            pos.push_back(cur);
            return;
        }
        ps(pos, i+1, cur, t);
        ps(pos, i+1, cur + t[i], t);
        ps(pos, i+1, cur + 2 * t[i], t);
    }
    int closestCost(vector<int>& b, vector<int>& t, int tar) {
        sort(b.begin(), b.end());
        sort(t.begin(), t.end());
        vector<int> pos;
        ps(pos, 0,0,t);
        

        int res = b[0];
        // for(int x : pos) cout << x << "->";
        // cout<<endl;
        for(int x : pos) {
            for(int i : b) {
                int tmp = x + i;
                if(abs(tmp - tar) < abs(tar-res) || (abs(tmp - tar) == abs(tar-res)  && tmp<res)) {
                    res = tmp;
                }
            }
        }
        return res;
    }
};
