class Solution {
public:
    int res = -1;
    int d = INT_MAX;
    //int dp[10010];
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        unordered_set<int> cnt;
        cnt.insert(0);
        for (auto &top : toppingCosts) {
            vector<int> bar;
            for (auto& x : cnt) {
                bar.emplace_back(x + top);
                bar.emplace_back(x + top + top);
            }
            for (auto& x : bar)
                cnt.insert(x);
        }
        for (auto& base : baseCosts) {
            for (auto& top : cnt) {
                int sum = base + top;
                int dd = abs(sum - target);
                if (dd < d) {
                    d = dd;
                    res = sum;
                }else if (dd == d && sum < target) {
                    res = sum;
                } 
            }
        }
        return res;
    }
};
