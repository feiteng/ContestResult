class Solution {
public:

    void fun(vector<int>&t,int i,int s,int &ans,int target) {
        if(i==t.size()) {
            if(abs(s-target)<abs(ans-target)) {
                ans=s;
            }
            else if(abs(s-target)==abs(ans-target)&&s<ans)
                ans=s;
        }
        else { 
            fun(t,i+1,s,ans,target);
            for(int c=1;c<=2;c++) {
                s+=t[i];
                fun(t,i+1,s,ans,target);
            }

        }
    }
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int ans=baseCosts[0];
        int n=baseCosts.size();
        for(int i=0;i<n;i++) {
            fun(toppingCosts,0,baseCosts[i],ans,target);
        }
        return ans;
    }
};
