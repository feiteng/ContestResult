class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int n = baseCosts.size();
        int m = toppingCosts.size();
        
        vector<int> topComb;
        int p3[11];
        p3[0] = 1;
        for(int i = 1; i <= 10; i++) p3[i] = 3 * p3[i-1];
        for(int S = 0; S < p3[m]; S++){
            int tot = 0;
            for(int j = 0; j < m; j++){
                tot += ((S / p3[j]) % 3) * toppingCosts[j];
            }
            topComb.push_back(tot);
        }
        sort(topComb.begin(), topComb.end());
        
        int res = 20000;
        for(int i = 0; i < n; i++){
            int base = baseCosts[i];
            int k = lower_bound(topComb.begin(), topComb.end(), target - base) - topComb.begin();
            if(k >= 0 && k < topComb.size()){
                if(abs(topComb[k] + base - target) < abs(res - target)){
                    res = topComb[k] + base;
                }else if(abs(topComb[k] + base - target) == abs(res - target)){
                    res = min(res, topComb[k] + base);
                }
            }
            if(k > 0){
                k--;
                if(abs(topComb[k] + base - target) < abs(res - target)){
                    res = topComb[k] + base;
                }else if(abs(topComb[k] + base - target) == abs(res - target)){
                    res = min(res, topComb[k] + base);
                }
            }
        }
        
        return res;
    }
};
