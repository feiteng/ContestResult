class Solution {
    
public:
    int closestCost(vector<int>& bc, vector<int>& tc, int t) {

        int diff = INT_MAX, ans = 0;
        int lt = tc.size();
        int limit = pow(3, lt);
        
        for (int i = 0; i < limit; i++) {
            int s = 0, k = i;
            for (int j = 0; k; j++) {
                s += tc[j] * (k%3);
                k /= 3;
            }
            for (int x : bc) {
                int now = s + x;
                int df = abs(now-t);
                if (df < diff || (df == diff && ans > now)) {
                    ans = now; diff = df;
                }
            }
        }
        
        return ans;
        
    }
};
