int Abs(int x) {
  return x >= 0 ? x : -x;
}

class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
      int min_cost = numeric_limits<int>::max();
      int j_max = 1;
      for (int i = 0; i < toppingCosts.size(); ++i) j_max *= 3;            
      for (int i = 0; i < baseCosts.size(); ++i) {
        for (int j = 0; j < j_max; ++j) {
          int cost = baseCosts[i];
          int bits = j;
          for (int k = 0; k < toppingCosts.size(); ++k) {
            cost += toppingCosts[k] * (bits % 3);
            bits /= 3;
          }
          if (Abs(target - cost) < Abs(target - min_cost) ||
              (Abs(target - cost) == Abs(target - min_cost) && cost < min_cost)) {
            min_cost = cost;
          }
        }
      }
      return min_cost;
    }
};
