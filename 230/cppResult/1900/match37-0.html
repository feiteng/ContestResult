class Solution {
public:
    int cur_res = INT_MAX;
    int res_t = INT_MAX;
    void dfs(vector<int>& tc, int index, int t, int cur){
        if(index == tc.size()){
            if(abs(t) < res_t){
                res_t = abs(t);
                cur_res = cur;
            }else if(abs(t) == res_t){
                cur_res = min(cur, cur_res);
            }
            return;
        }
        for(int i = 0; i < 3; i++){
            dfs(tc, index + 1, t - tc[index] * i, cur + tc[index] * i);
        }
    }
    int closestCost(vector<int>& bc, vector<int>& tc, int t) {
        int res = INT_MAX;
        int n = bc.size(), m = tc.size();
        for(int i = 0; i < n; i++){
            int cnt = bc[i];
            dfs(tc, 0, t-cnt, cnt);
        }
        return cur_res;
    }
};
