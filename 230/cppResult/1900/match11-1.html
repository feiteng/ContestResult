class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int ans = baseCosts[0];
        for (int i = 0; i < baseCosts.size(); ++i) {
            Dfs(baseCosts[i], 0, toppingCosts, target, ans);
        }
        return ans;
    }
    

    void Dfs(int tmp, int id, vector<int>& toppingCosts, int target, int& ans) {
        if (id == toppingCosts.size()) {
            if (abs(tmp - target) < abs(ans - target)) {
                ans = tmp;
            } else if (abs(tmp - target) == abs(ans - target) && tmp < ans) {
                ans = tmp;
            }
            return;
        }
        for (int i = 0; i < 3; ++i) {
            Dfs(tmp + i * toppingCosts[id], id + 1, toppingCosts, target, ans);   
        }
    }
};
