class Solution {
public:
    int res=100000;
    void dfs(vector<int>&t,int start,int cur,int target){
        if(cur>target&&cur-target>abs(res-target))
            return;
        if(abs(cur-target)<abs(res-target)){
            res=cur;
        }
        if(start>=t.size())
            return;
        for(int i=start;i<t.size();i++){
            dfs(t,i+1,cur,target);
            dfs(t,i+1,cur+t[i],target);
            dfs(t,i+1,cur+2*t[i],target);
        }
    }
    int closestCost(vector<int>& b, vector<int>& t, int target) {
        for(int i=0;i<b.size();i++){
            if(abs(b[i]-target)<abs(res-target)){
                res=b[i];
            }
            dfs(t,0,b[i],target);
        }
        return res;
    }
};
