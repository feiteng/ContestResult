class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int n = baseCosts.size();
        int m = toppingCosts.size();
        
        //Price array
        int poss_comb=1;
        for(int i=0;i<m;i++)poss_comb*=3;
        int price[n][poss_comb];
        
        //Calculate the price array
        for(int i=0;i<poss_comb;i++){
            
            //First find the coefficient array
            int coeff[m];

            int temp=i;
            for(int j=0;j<m;j++){
                coeff[j]=temp%3;
                temp/=3;
            }
            
            //Now find the price for each n
            for(int j=0;j<n;j++){
                price[j][i]=baseCosts[j];
                
                for(int k=0;k<m;k++){
                    price[j][i]+=(coeff[k]*toppingCosts[k]);
                }
            }
        }
        
        //Npw find the answer price
        int ans[2]={1000000,1000000};
        for(int i=0;i<n;i++){
            for(int j=0;j<poss_comb;j++){

                if(abs(price[i][j]-target)<ans[1]){
                    ans[1]=abs(price[i][j]-target);
                    ans[0]=price[i][j];
                }
                else
                if(abs(price[i][j]-target)==ans[1]){
                    ans[0]=min(ans[0],price[i][j]);
                }
            }
        }
        
        return ans[0];
    }
};
