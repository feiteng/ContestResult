class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int ans = -1e9;
        int n = baseCosts.size();
        int m = toppingCosts.size();
        int p3 = 1;
        for (int i = 0; i < m; i++) p3 *= 3;
        vector<int> costs;

        for (int i = 0; i < p3; i++) {
            int foo = i;
            int cost = 0;
            for (int j = 0; j < m; j++) {
                int u = foo % 3;
                foo /= 3;
                cost += u * toppingCosts[j];
            }
            costs.push_back(cost);
        }
        for (int i = 0; i < n; i++) {
            for (int e : costs) {
                int foo = baseCosts[i] + e;
                if (abs(target - foo) < abs(target - ans)) {

                    ans = foo;
                } else if (abs(target - foo) == abs(target - ans)) {
                    ans = min(ans, foo);
                }
            }
        }
        return ans;
    }
};
