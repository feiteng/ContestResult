class Solution {
public:
    int ans = 0x3fffffff;
    void dfs(vector<int> & top,int st,int v,int target)
    {
        if(st == top.size())
        {
            if(abs(target - v) < abs(ans - target) || (abs(target - v) == abs(ans - target) && target < ans))
                ans = v;

            return;
        }
        dfs(top,st+1,v,target);
        dfs(top,st+1,v+top[st],target);
        dfs(top,st+1,v+2*top[st],target);
    }
    int closestCost(vector<int>& base, vector<int>& top, int target) {
        for(int i = 0;i<base.size();i++)
        {
            dfs(top,0,base[i],target);
        }
        return ans;
    }
};
