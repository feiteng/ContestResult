class Solution {
    vector<int> B;
    int n,m,k,ans;
    void dfs(int x,int sum)
    {
        if (x==m)
        {
            if (ans==-1) ans=sum;
            else if (abs(ans-k)>abs(sum-k)) ans=sum;
            return ;
        }
        dfs(x+1,sum);
        dfs(x+1,sum+B[x]);
        dfs(x+1,sum+B[x]*2);
    }
public:
    int closestCost(vector<int>& A, vector<int>& toppingCosts, int target) {
        B=toppingCosts;
        n=A.size();
        m=B.size();
        k=target;
        ans=-1;
        for (int i=0;i<n;++i)
        {
            dfs(0,A[i]);
        }
        return ans;
    }
};
