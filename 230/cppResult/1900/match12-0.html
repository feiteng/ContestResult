class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target)
    {
        int answer = INT_MAX;
        for (int i = 0; i < baseCosts.size(); ++i)
        {
            for (int j = 0; j < pow(3, toppingCosts.size()); ++j)
            {
                int temp = baseCosts[i], num = j;
                for (int k = 0; k < toppingCosts.size(); ++k)
                {
                    temp += (num % 3) * toppingCosts[k];
                    num /= 3;
                }
                if (abs(temp - target) < abs(answer - target) || abs(temp - target) == abs(answer - target) && temp < answer)
                {
                    answer = temp;
                }
            }
        }
        return answer;
    }
};
