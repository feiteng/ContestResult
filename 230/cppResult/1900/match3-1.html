class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        int res = 0, diff = INT_MAX;
        
        for (auto c: baseCosts)
            dfs(0, c, toppingCosts, target, res, diff);
        
        return res;
    }
    

    void dfs(int index, int cost, vector<int>& toppingCosts, int target, int& res, int& diff) {
        if (abs(cost - target) < diff) {
            diff = abs(cost - target);
            res = cost;
        } else if (abs(cost - target) == diff && cost < res) {
            res = cost;
        }
        if (index == toppingCosts.size()) return;
        
        for (int i = 0; i <= 2; ++i) {
            dfs(index+1, cost + toppingCosts[index] * i, toppingCosts, target, res, diff);
        }
    }
};
