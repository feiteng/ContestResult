class Solution {
public:
    int n,m;

    int result = 1e8;
    
    void dfs(int sum, int idx, vector<int>& toppingCosts, int target) {
        if (sum >= target || idx == m) {
            if (abs(sum-target) < abs(result-target)) result = sum;
            else if (abs(sum-target) == abs(result-target)) result = min(result, sum);
            return;
        }
        dfs(sum, idx+1, toppingCosts, target);
        dfs(sum+toppingCosts[idx], idx+1, toppingCosts, target);
        dfs(sum+2*toppingCosts[idx], idx+1, toppingCosts, target);
    }
    
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
        sort(baseCosts.begin(), baseCosts.end());
        sort(toppingCosts.begin(), toppingCosts.end());
        n = baseCosts.size(), m = toppingCosts.size();
        for (auto x : baseCosts) {
            dfs(x, 0, toppingCosts, target);
        }
        return result;
    }
};
