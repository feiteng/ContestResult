class Solution {
public:
    int closestCost(vector<int>& baseCosts, vector<int>& toppingCosts, int target) {
    	int m = toppingCosts.size();
    	int n = baseCosts.size();
    	int all = 1;
    	int a[20],ans,mn = 10000000;
    	for (int i = 1; i <= m; ++i)
    		all *= 3;
    	for (int s = 0; s < all; ++s){
    		int tmp = s;

    		for (int i = 0; i < m; ++i){
    			a[i] = tmp % 3;
    			tmp /= 3;
    		}
    		for (int pos = 0; pos < n; ++pos){
    			int now = baseCosts[pos];
    			for (int i = 0; i < m; ++i)
    				now += a[i] * toppingCosts[i];
    			if (abs(now - target) < abs(mn - target) ){
    				ans = now;
    				mn = now;
    			} else if (abs(now - target) == abs(mn - target)){
    				ans = min(ans, now);
    			}
    		}
    	}
    	return ans;
    }
};
