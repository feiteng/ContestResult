class Solution {
    
    int ans = Integer.MAX_VALUE;
    
    public void dfs(int[] tc, int t, int target, int current) {
        //if(current == 10) System.out.println(current + " " + ans);
        if(Math.abs(ans - target) > Math.abs(current - target)) {
            ans = current;
        }else if (Math.abs(ans - target) == Math.abs(current - target)){
            ans = Math.min(ans, current);
        }
        if(t >= tc.length) return;
        
        
        
        dfs(tc, t+1, target, current + tc[t]);
        dfs(tc, t+1, target, current + 2*tc[t]);
        dfs(tc, t+1, target, current);
    }
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        //Arrays.sort(toppingCosts);
        
        
for(int i = 0; i < baseCosts.length; i++) {
            dfs(toppingCosts, 0, target, baseCosts[i]);
        }
        return ans;
        
    }
}
