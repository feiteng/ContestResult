class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
            Set<Integer> toppingSet = new HashSet<>();
            toppingSet.add(0);
            for (int i = 0; i < toppingCosts.length; i++) {
                HashSet<Integer> newSet = new HashSet<>();
                for (int j = 1; j < 3; j++) {
                    for (Integer old : toppingSet) {
                        newSet.add(old + toppingCosts[i] * j);
                    }
                }
                toppingSet.addAll(newSet);
            }
            System.out.println(toppingSet);
            
int min = Integer.MAX_VALUE;
            int result = 0;
            for (int baseCost : baseCosts) {
                for (int topCost : toppingSet) {
                    int diff = baseCost + topCost - target;
                    int diffABS = Math.abs(diff);
                    if (diffABS < min || (diffABS == min && diff < 0)) {
                        min = diffABS;
                        result = baseCost + topCost;
                    
}
                }
            }
            return result;
        }
}
