class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        // base ; toppingCosts 
        int minDiff = Integer.MAX_VALUE;
        int closestCost = Integer.MAX_VALUE;
        // permutate all combinations of topping costs.
        
        HashSet<Integer> set = new HashSet<Integer>();
        set.add(0);
        for (int i = 0; i < toppingCosts.length; i++){
            int currentCost = toppingCosts[i];
            HashSet<Integer> tempSet = new HashSet<Integer>(set);
            for (int previous : tempSet){
                set.add(previous + currentCost);
                set.add(previous + 2 * currentCost);
            }
        
}
        for (int baseCost : baseCosts){
            for (int toppingCost : set){
                int diff = Math.abs(target - toppingCost - baseCost);
                int cost = toppingCost + baseCost;
                
if (diff < minDiff){
                    minDiff = diff;
                    closestCost = cost;
                }
                else if (diff == minDiff){
                    closestCost = Math.min(closestCost, cost);
                }
            }
        }
        return closestCost;
    }
}
