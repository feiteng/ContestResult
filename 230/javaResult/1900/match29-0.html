class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
    int diff = Integer.MAX_VALUE;
    int res = -1;
    int total = 1;
    for (int i = 0; i < toppingCosts.length; i++) {
      total = 3 * total;
    }
    List<Integer> poss = new ArrayList<>();
    for (int mask = 0; mask < total; mask++) {
      int want = 0;
      int tmp = mask;
      for (int i = 0; i < toppingCosts.length; i++) {
//        want.add(mask % 3);
        want += toppingCosts[i] * (tmp % 3);
        tmp = tmp / 3;
      }
      poss.add(want);
    }
    for (int v1 : baseCosts) {
      for (int v2 : poss) {
        int d = Math.abs(v1 + v2 - target);
        if (d < diff) {
          diff = d;
          res = v1 + v2;
        } else if (d == diff) {
          res = Math.min(res, v1 + v2);
        }
      }
    }
    return res;
  }
}
