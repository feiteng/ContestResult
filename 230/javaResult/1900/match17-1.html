class Solution {
    int best;
    public int closestCost(int[] a, int[] b, int target) {
        best = a[0];
        for (int i = 0; i < a.length; i++) {
            recur(a[i],0,b,target);
        }
        return best;
    }
    public void recur(int s, int ind, int[] b, int target) {
        if (Math.abs(target-s)<Math.abs(target-best) || Math.abs(target-s)==Math.abs(target-best)&&s<best) {
            best = s;
        }
        if (ind==b.length) return;
        recur(s,ind+1,b,target);
        int c = 0;
        while (true) {
            s += b[ind];
            recur(s,ind+1,b,target);
            if (s>target || c==1)  break;
            c++;
        }
    }
}
