class Solution {
    int v = Integer.MAX_VALUE;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        for (int cost : baseCosts) {
            int[] nums = new int[toppingCosts.length];
            dfs(toppingCosts, nums, target, cost);
            if (v == target) {
                break;
            }
        
}
        return v;
    }
    
    public void dfs(int[] toppingCosts, int[] nums, int target, int ans) {
        if ((Math.abs(ans - target) < Math.abs(v - target)) || (Math.abs(ans - target) == Math.abs(v - target) && ans < v)) {
            v = ans;
        
}
        if (ans > target) {
            return;
        }
        for (int i = 0; i < toppingCosts.length; i++) {
            if (nums[i] == 2) {
                continue;
            }
            nums[i]++;
            dfs(toppingCosts, nums, target, ans + toppingCosts[i]);
            nums[i]--;
        }
    }
}
