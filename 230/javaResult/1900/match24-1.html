class Solution {
    
public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        Set<Integer> set = new HashSet<>();
        set.add(0);
        for (int val : toppingCosts) {
            Set<Integer> next = new HashSet<>();
            for (int c : set) {
                next.add(c);
                next.add(c + val);
                next.add(c + val + val);
            }
            set = next;
        }
        int diff = Integer.MAX_VALUE;
        int result = Integer.MAX_VALUE;
        for (int val : set) {
            for (int b : baseCosts) {
                int t = val + b;
                int d = Math.abs(target - t);
                if (d == 0) {
                    return target;
                }
                if (d < diff) {
                    diff = d;
                    result = t;
                } else if (d == diff && t < result) {
                    result = t;
                
}
            }
        }
        return result;
    }
}
