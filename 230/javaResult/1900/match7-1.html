class Solution {
    int closest = Integer.MAX_VALUE;
    
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        
for(int base : baseCosts) {
            backtrack(base,toppingCosts,0, target);
        }
        return closest;
    }
    private void backtrack(int cur, int[] toppings, int i, int target) {
        if(i >= toppings.length || cur > target) {
            int curDif = Math.abs(cur - target);
            int prevDif = Math.abs(closest - target);
            if(curDif < prevDif) closest = cur;
            else if(curDif == prevDif) closest = Math.min(closest, cur)
;
            return;
        }
        backtrack(cur, toppings, i + 1, target);
        backtrack(cur + toppings[i],toppings, i + 1, target);
        backtrack(cur + 2 * toppings[i],toppings, i + 1, target);
    }
}
