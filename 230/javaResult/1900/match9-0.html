class Solution {
    int so = Integer.MAX_VALUE  / 2;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
      //dfs
      for (int i = 0; i < baseCosts.length; i++) {
          dfs(baseCosts[i], toppingCosts, target, 0);
      }
      return so;
    }
    
    private void dfs(int cur, int[] toppingCosts, int target, int index) {
      if (index == toppingCosts.length) {
          if (Math.abs(cur - target) < Math.abs(so - target)) {
              so = cur;
          } else if (Math.abs(cur - target) == Math.abs(so - target) && cur < so) {
              so = cur;
          }
          return;
      }
      for (int i = 0; i <= 2; i++) {
          dfs(cur + i * toppingCosts[index], toppingCosts, target, index + 1);
      }
    }
}
