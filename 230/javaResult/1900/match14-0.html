class Solution {
    int bestDiff;
    int bestNum;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        bestNum = -1;
        bestDiff = Integer.MAX_VALUE;
        
        for (int i = 0; i < baseCosts.length; i++) {
            recurse(0, baseCosts[i], target, toppingCosts);
        }
        
        return bestNum;
    }
    
    private void recurse(int idx, int partial, int target, int[] toppingCosts) {
        // update best
        int curDiff = Math.abs(partial-target);
        if (curDiff < bestDiff) {
            bestDiff = curDiff;
            bestNum = partial;
        } else if (curDiff == bestDiff) {
            bestNum = Math.min(bestNum, partial);
        }
        
        // return early if no further step can beat the optimal seen so far
        if (partial > target && (partial-target) >= bestDiff) return;
        
        // base case
        if (idx == toppingCosts.length) return;
        
        
        // consider taking topping[idx] 0,1,2 times
        recurse(idx+1, partial, target, toppingCosts);
        recurse(idx+1, partial + toppingCosts[idx], target, toppingCosts);
        recurse(idx+1, partial + 2*toppingCosts[idx], target, toppingCosts);
        
    }
}
