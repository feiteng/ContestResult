class Solution {
    int res = Integer.MAX_VALUE;
    private void helper (int[] toppingCosts, int idx, int target, int cur) {
        if (Math.abs(target - cur) < Math.abs(target - res)) {
            res = cur;
        } else if (Math.abs(target - cur) == Math.abs(target - res)) {
            res = Math.min(cur, res);
        }
        if (target < cur) {
            return;
        }
        for (; idx < toppingCosts.length; idx++) {
            helper(toppingCosts, idx + 1, target, cur + toppingCosts[idx]);
            helper(toppingCosts, idx + 1, target, cur + toppingCosts[idx] * 2);
        }
    }
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        Arrays.sort(toppingCosts);
        for (int i = 0; i < baseCosts.length; i++) {
            helper(toppingCosts, 0, target, baseCosts[i]);
        }
        return res;
    }
}
