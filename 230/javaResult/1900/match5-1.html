class Solution {
    private int ans;

    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {

        ans = Integer.MAX_VALUE;


        for (int base : baseCosts) {

            dfs(base, toppingCosts, 0, target);
        }

        return ans;
    }

    private void dfs(int curr, int[] arr, int k, int target) {


        if (Math.abs(curr - target) < Math.abs(ans - target) ||
                Math.abs(curr - target) == Math.abs(ans - target) && curr < ans) {
            ans = curr;
        }
        if (k == arr.length || curr > ans && curr > target) {
            
return;
        }


        dfs(curr + 2 * arr[k], arr, k + 1, target);
        dfs(curr + arr[k], arr, k + 1, target);
        dfs(curr, arr, k + 1, target);


    }
}
