class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        int result = -1;
        Arrays.sort(baseCosts);
        Arrays.sort(toppingCosts);
        int max = Math.max(baseCosts[baseCosts.length - 1], toppingCosts[toppingCosts.length - 1]);
        
boolean[] dp = new boolean[max + target + 1];
        for (int base : baseCosts) {
            dp[base] = true;
        }
        for (int top : toppingCosts) {
            for (int i = 0; i < 2; i++) {
                
for (int j = dp.length - 1; j >= top; j--) {
                    if (dp[j - top]) {
                        dp[j] = true;
                    }
                }
            }
        }
        
for (int i = 0; i < dp.length; i++) {
            if (dp[i] && (result < 0 || Math.abs(target - i) < Math.abs(target - result))) {
                result = i;
            }
        }
        
        return result;
    }
}
