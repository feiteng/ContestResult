class Solution {
    private int min = Integer.MAX_VALUE;
    private int res = -1;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        for(int i=0;i<baseCosts.length;i++) {
            dp(toppingCosts, 0, target, baseCosts[i]);
        }
        return res;
    }
    
    private void dp(int[] toppingCosts, int start, int target, int curCost) {
        if(Math.abs(target-curCost)<=min) {
            if(min==Math.abs(target-curCost)) {
                res = Math.min(curCost, res);
            } else res = curCost;
            min = Math.abs(target-curCost);
        }
        if(start==toppingCosts.length) return
;
        
        for(int i=start;i<toppingCosts.length;i++) {
            dp(toppingCosts, i+1, target, curCost);
            dp(toppingCosts, i+1, target, curCost+toppingCosts[i]);
            dp(toppingCosts, i+1, target, curCost+toppingCosts[i]*2);
        }
    }
}
