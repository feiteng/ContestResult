class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        Integer[] res = {null};
        for (int baseCost : baseCosts) {
            dfs(toppingCosts, target, baseCost, 0, res);    
        }        
        return res[0];
    }
    
    private void dfs(int[] toppingCosts, int target, int cost, int index, Integer[] res) {
        if (index == toppingCosts.length || cost >= target) {
            if (res[0] == null || Math.abs(res[0] - target) > Math.abs(cost - target)) {
                res[0] = cost;
            } else if (Math.abs(res[0] - target) == Math.abs(cost - target)) {
                res[0] = Math.min(res[0], cost);
            }
            return;
        }
        
        for (int i = 0; i <= 2; i++) {
            dfs(toppingCosts, target, cost + toppingCosts[index] * i, index + 1, res);    
        }        
    }
}
