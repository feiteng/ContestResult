class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        boolean[] doable = new boolean[20002];
        for (int b : baseCosts) {
            doable[b] = true;
        }
        for (int t : toppingCosts) {
            for (int c = doable.length - 1; c >= 0; --c) {
                if (doable[c]) {
                    if (c + t + t < doable.length) {
                        doable[c + t + t] = true;
                    }
                    if (c + t < doable.length) {
                        doable[c + t] = true;
                    }
                }
            
}
        }
        int best = baseCosts[0];
        for (int c = 0; c < doable.length; ++c) {
            if (doable[c] && Math.abs(target - c) < Math.abs(target - best)) {
                best = c;
            }
        }
        return best;
    }
}
