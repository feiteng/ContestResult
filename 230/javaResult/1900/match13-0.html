class Solution {
	private int max;

	public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
		max = 99999;
		for (int baseCost : baseCosts) {
			closestCost(baseCost, 0, toppingCosts, target);
		}
		return max;
	}

	private void closestCost(int cost, int index, int[] toppingCosts, int target) {
		if (Math.abs(cost - target) < Math.abs(max - target)) {
			max = cost;
		} else if (Math.abs(cost - target) == Math.abs(max - target) && cost < max) {
			max = cost;
		}
		if (index < toppingCosts.length && cost <= target) {
			closestCost(cost, index + 1, toppingCosts, target);
			closestCost(cost + toppingCosts[index], index + 1, toppingCosts, target);
			closestCost(cost + 2 * toppingCosts[index], index + 1, toppingCosts, target);
		}
	}
}
