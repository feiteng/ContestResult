import java.util.TreeSet;


class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        int dist = 99999;
        int ans = 0;
        TreeSet<Integer> set = new TreeSet<>();
        set.add(0);
        for (int i = 0; i < toppingCosts.length; i++) {
            TreeSet<Integer> next = new TreeSet<>();
            for (int curr : set) {
                next.add(curr);
                next.add(curr + toppingCosts[i]);
                next.add(curr + toppingCosts[i] * 2);
            }
            set = next;
        }
        for (int base : baseCosts) {
            int curr = target - base;
            for (int topping : set) {
                if (dist > Math.abs(curr - topping) || (dist == Math.abs(curr - topping) && curr - topping > 0)) {
                    dist = Math.abs(curr - topping);
                    
ans = topping + base;
                }
            }
        }
        return ans;
    }
}
