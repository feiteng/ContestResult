class Solution {
    int min = Integer.MAX_VALUE;
    int ans = 0;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        for (int i = 0; i < baseCosts.length; i++) {
            dfs(toppingCosts, 0, baseCosts[i], target);
        }
        return ans;
    }
    
    private void dfs(int[] toppingCosts, int idx, int cost, int target) {
        if (Math.abs(target - cost) < min) {
            min = Math.abs(target - cost);
            ans = cost;
        } else if (Math.abs(target - cost) == min) {
            ans = Math.min(ans, cost);
        }
        if (idx == toppingCosts.length) {
            return;
        }
        
        dfs(toppingCosts, idx + 1, cost, target);
        dfs(toppingCosts, idx + 1, cost + toppingCosts[idx], target);
        dfs(toppingCosts, idx + 1, cost + 2 * toppingCosts[idx], target);
    }
}
