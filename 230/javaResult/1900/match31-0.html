class Solution {
    int mn=Integer.MAX_VALUE;
    int res=-1;
    int target;
    public int closestCost(int[] A, int[] B, int target) {
        
        this.target=target;
        for(int k:A){
            dfs(B,k,0);
        }
        
        return res;
    }
    
   public void dfs(int A[],int cur,int i){
        
        //System.out.println(cur);
        int dif=Math.abs(cur-target);
        if(mn>=dif){
            if(mn>dif){
                mn=dif;
                res=cur;
            }
            else{
                if(res>cur){
                    res=cur;
                }
            }
        }
        
        if(i>=A.length){
            return;
        }
        if(cur>target){
            return;
        }
        dfs(A,cur+A[i]*2,i+1);
        dfs(A,cur+A[i],i+1);
        dfs(A,cur,i+1);
    }
    
}
