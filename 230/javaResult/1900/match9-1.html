class Solution {
    public int closestCost(int[] bc, int[] tc, int target) {
        int[] res = new int[1];
        res[0] = 1000000000;
        
        
for (int i = 0; i < bc.length; i++) {
            bt(bc[i], i, 0, bc, tc, target, res);
        }
        
        return res[0];
    }
    
    public void bt(int cur, int base, int top, int[] bc, int[] tc, int target, int[] res) {
        int m = tc.length;
        
if (top == m) {
            if (Math.abs(cur - target) < Math.abs(res[0] - target)) {
                res[0] = cur;
            } else if (Math.abs(cur - target) == Math.abs(res[0] - target) && cur < res[0]) {
                res[0] = cur;
            }
            return;
        }
        
        for (int i = 0; i <= 2; i++) {
            bt(cur + i * tc[top], base, top + 1, bc, tc, target, res);
        }
    }
}
