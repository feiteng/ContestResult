class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        boolean[] dp = new boolean[20005];
        for (int i : baseCosts) {
            dp[i] = true;
        }
        for (int j : toppingCosts) {
            for (int i = dp.length - 1; i >= 0; i--) {
                if (dp[i] && i + j < dp.length) {
                    dp[i + j] = true;
                }
            }
            for (int i = dp.length - 1; i >= 0; i--) {
                if (dp[i] && i + j < dp.length) {
                    dp[i + j] = true;
                }
            }
        }
        int res = 40005;
        for (int i = 0; i < dp.length; i++) {
            if (dp[i] && Math.abs(i - target) < Math.abs(res - target)) {
                res = i;
            }
        }
        return res;
    }
}
