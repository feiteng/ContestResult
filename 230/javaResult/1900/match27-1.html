class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        diff = Integer.MAX_VALUE;
        ans =  0;
        
for(int val : baseCosts){
            comp(0, toppingCosts, target, val);
        }
        return ans;
    }
    
    static int ans;
    static int diff;
    
    public static void comp(int ind, int arr[], int target, int sum){
        
        int y = sum-target;
        
if(Math.abs(y)<diff){
            diff = Math.abs(y);
            ans = sum;
        }else if(Math.abs(y)==diff && sum<=target){
            diff = Math.abs(y);
            ans = sum;
        
}
        
        if(ind==arr.length){
            return;
        }
        comp(ind+1, arr, target, sum);
        comp(ind+1, arr, target, sum+arr[ind]);
        comp(ind+1, arr, target, sum+arr[ind]+arr[ind]);
    }
}
