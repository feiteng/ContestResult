class Solution {

public int closestCost(int[] baseCosts, int[] toppingCosts, int target)
	{
		int n = baseCosts.length;
		int m = toppingCosts.length;
		int result = Integer.MAX_VALUE;
		for(int x = 0; x < n; x++)
		{
			for(int y = 0; y < Math.pow(3, m); y++)
			{
				int val = y;
				int current = baseCosts[x];

				for(int z = 0; z < m; z++)
				{
					current += (val % 3) * toppingCosts[z];
					val /= 3;
				}
				
				if(Math.abs(current - target) < Math.abs(result - target) || Math.abs(current - target) == Math.abs(result - target) && current < result)
				{
					result = current;
				}

			}
		}
		
		return result;
	}
}
