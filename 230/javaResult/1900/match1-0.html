class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        List<Integer> values = new ArrayList<>();
        values.add(0);
        
        for (int i = 0; i < toppingCosts.length; i++) {
            List<Integer> nextValues = new ArrayList<>(values.size() * 3);
            nextValues.addAll(values);
            
            for (Integer v : values) {
                nextValues.add(v + toppingCosts[i]);
                nextValues.add(v + toppingCosts[i] * 2);
            }
            
            values = nextValues;
        }
        
        //System.out.println(values);
        
        int bestDiff = Integer.MAX_VALUE;
        int best = Integer.MAX_VALUE;
        
        for (int i = 0; i < baseCosts.length; i++) {
            for (Integer v : values) {
                int cost = baseCosts[i] + v;
                int diff = Math.abs(target - cost);
                
                if (diff < bestDiff) {
                    bestDiff = diff;
                    best = cost;
                } else if (diff == bestDiff && best > cost) {
                    best = cost;
                }
            }
        }
        
        return best;
    }
}
