class Solution {
    int DIFF = Integer.MAX_VALUE;
    int ret = -1;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        for(int i=0; i<baseCosts.length; i++) {
            helper(baseCosts[i], toppingCosts, 0, target);
        }
        
        return ret;
    }
    
    private void helper(int cost, int[] toppingCosts, int start, int target) {
        if(start > toppingCosts.length || DIFF == 0) return;
        if(Math.abs(cost-target) == DIFF) ret = Math.min(ret, cost);
        else if(Math.abs(cost-target) < DIFF){
            DIFF = Math.abs(cost-target);
            ret = cost;
        }
        for(int i=start; i<toppingCosts.length; i++) {
            helper(cost, toppingCosts, i+1, target);
            helper(cost+toppingCosts[i], toppingCosts, i+1, target);
            helper(cost+2*toppingCosts[i], toppingCosts, i+1, target);
        }
    }
}
