class Solution {
    int closest;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        closest = -1;
        for (int type : baseCosts) {
            help(0, toppingCosts, type, target);
        }
        return closest;
    }
    
    public void help(int ind, int[] cost, int curr, int target) {
        if (ind == cost.length) {
            if (closest == -1 || (Math.abs(target - curr) < Math.abs(target - closest) || Math.abs(target-curr) == Math.abs(target-closest) && curr < closest)) {
                closest = curr;
            
}
            return;
        }
        help(ind+1, cost, curr, target);
        help(ind+1, cost, curr+cost[ind], target);
        help(ind+1, cost, curr+2*cost[ind], target);
    }
}
