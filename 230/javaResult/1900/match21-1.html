class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        List<Integer> topping = new ArrayList<>();
        topping.add(0);
        for (int i = 0; i < toppingCosts.length; i++) {
            int v = toppingCosts[i];
            
List<Integer> newTopping = new ArrayList<>(topping);
            for (int t : topping) {
                newTopping.add(t + v);
                newTopping.add(t + v * 2);
            }
            topping = newTopping;
        }
        int min = Integer.MAX_VALUE;
        int curr = 0;
        for (int i = 0; i < baseCosts.length; i++) {
            int b = baseCosts[i];
            
for (int t : topping) {
                int total = b + t;
                if (Math.abs(total - target) < min) {
                    min = Math.abs(total - target);
                    curr = total;
                } else if (curr > target && target - total == min) 
{
                    curr = total;
                }
            }
        }
        return curr;
    }
}
