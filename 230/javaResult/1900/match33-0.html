import java.util.HashSet;
import java.util.Set;

class Solution {
	public static void main(String[] args) {
		int[] baseCosts = {1,7};
		int[] toppingCosts = {3,4};
		System.out.println(new Solution().closestCost(baseCosts, toppingCosts, 10));
	}
	public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
		Set<Integer> set = new HashSet<>();
		set.add(0);
		for (int x : toppingCosts) {
			Set<Integer> set2 = new HashSet<>();
			for (int y : set) {
				set2.add(y);
				set2.add(y + x);
				set2.add(y + x * 2);
			}
			set = set2;
		}
		int ans = 0;
		int gap = Integer.MAX_VALUE;
		for (int x : baseCosts) {
			for (int y : set) {
				int z = x + y;
				int temp = Math.abs(z - target);
				if (temp < gap || temp == gap && ans > z) {
					gap = temp;
					ans = z;
				}
			}
		}
		return ans;
	}
}
