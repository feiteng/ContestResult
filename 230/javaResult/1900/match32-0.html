class Solution {
    static int closest;
    
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        closest = 100_000_000;
        for (int b = 0; b < baseCosts.length; ++b) {
            buildDessert(0, 0, toppingCosts, baseCosts[b], target);
        }
        return closest;
    }
    
    void buildDessert(int cost, int ind, int[] tops, int base, int target) {
        if (ind == tops.length) {
            cost += base;
            if (Math.abs(closest - target) >= Math.abs(cost - target)) {
                if (Math.abs(closest - target) > Math.abs(cost - target)) {
                    closest = cost;
                } else {
                    closest = Math.min(closest, cost);
                }
            }
            return;
        }
        buildDessert(cost, ind + 1, tops, base, target);
        buildDessert(cost + tops[ind], ind + 1, tops, base, target);
        buildDessert(cost + 2 * tops[ind], ind + 1, tops, base, target);
    }
    
    
}
