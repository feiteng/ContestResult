class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        return sln1(baseCosts, toppingCosts, target);
    }
    
    private int sln1(int[] bs, int[] ts, int t){
        int n = bs.length;
        int m = ts.length;
        
        int res = Integer.MAX_VALUE;;
        for(int i = 0;i< n;i++){
            
            for(int j = (1<<(2*m))-1; j>=0;j--){
                int sum = bs[i];
                for(int k = 0; k<2*m;k++){
                    sum += ts[k%m]*((j>>k)&1);
                    //System.out.println(":"+sum);
                }
                if(Math.abs(sum-t)<Math.abs(res-t)) res = sum;
                else if(Math.abs(sum-t)==Math.abs(res-t) && sum<res) res = sum;
            }
            
        }
        return res;
    }
}
