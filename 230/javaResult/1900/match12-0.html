class Solution {
    public int closestCost(int[] bcs, int[] tcs, int tg) {
        boolean[][] dp = new boolean[tcs.length*2+1][20001];
        for (int i = 0; i < bcs.length; ++i) {
            dp[0][bcs[i]] = true;
        }
        for (int i = 0; i < tcs.length; ++i) {
            for (int j = 0; j <= 10000; ++j) {
                dp[i+1][j] = dp[i+1][j]||dp[i][j];
                dp[i+1][j+tcs[i]] = dp[i+1][j+tcs[i]]||dp[i][j];
            }
        }
        for (int i = 0; i < tcs.length; ++i) {
            for (int j = 0; j <= 10000; ++j) {
                dp[i+1+tcs.length][j] = dp[i+1+tcs.length][j]||dp[i+tcs.length][j];
                dp[i+1+tcs.length][j+tcs[i]] = dp[i+1+tcs.length][j+tcs[i]]||dp[i+tcs.length][j];
            }
        }
        int ans = 1000000;
        for (int i = 0; i <= 20000; ++i) {
            if (dp[tcs.length*2][i] && Math.abs(i-tg)<Math.abs(ans-tg)) {
                ans = i;
            }
        }
        return ans;
    }
}
