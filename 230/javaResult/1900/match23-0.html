class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        int n = baseCosts.length;
    int m = toppingCosts.length;
    // You want to make a dessert with a total cost as close to target as possible.
    Set<Integer> curr = new HashSet<>();
    curr.add(0);
    for (int i = 0; i < m; i++) {
      Set<Integer> next = new HashSet<>();
      for (int v : curr) {
        next.add(v);
        next.add(v + toppingCosts[i]);
        next.add(v + toppingCosts[i] * 2);
      }
      curr = next;
    }

    int ans = baseCosts[0];
    for (int v : curr) {
      for (int i = 0; i < n; i++) {
        int dc = Math.abs(ans - target);
        int dn = Math.abs(baseCosts[i] + v - target);
        if (ans == 0 || dc > dn || (dc == dn && ans > baseCosts[i] + v)) {
          ans = baseCosts[i] + v;
        }
      }
    }
    return ans;
    }
}
