class Solution {
    
    int closest = 0;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        closest = baseCosts[0];
        
        for (int base : baseCosts) {
            dfs(toppingCosts, 0, base, target);
        }
        
        return closest;
    }
    
    private void dfs(int[] toppings, int index, int cost, int target) {
        if (index == toppings.length) {
            int oldDiff = Math.abs(closest - target);
            int diff = Math.abs(cost - target);
            if (diff < oldDiff || (diff == oldDiff && cost < closest)) closest = cost;
            return;
        }
        
        for (int i = index; i < toppings.length; i++) {
            for (int j = 0; j <= 2; j++) {
                dfs(toppings, i+1, cost+toppings[i]*j, target);
            }
        }
    }
}
