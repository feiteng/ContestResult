class Solution {
    int res = Integer.MAX_VALUE;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        // Arrays.sort(baseCosts);
        // Arrays.sort(toppingCosts);
        for (int b : baseCosts) {
            int[] visited = new int[toppingCosts.length];
            helper(toppingCosts, b, visited, 0, target);
        }
        return res;
    }
    public void helper(int[] toppingCosts, int cur, int[] visited, int index, int target) {
        if (Math.abs(cur - target) < Math.abs(res - target)) {
            res = cur;
        }
        if (Math.abs(cur - target) == Math.abs(res - target)) {
            res = Math.min(cur, res);
        }
        if (cur >= target) {
            return;
        }
        for (int i = index; i < toppingCosts.length; i++) {
            if (visited[i] == 2) {
                continue;
            }   
            visited[i] ++;
            helper(toppingCosts, cur + toppingCosts[i], visited, i, target);
            visited[i] --;
        }
    }
}
