class Solution {
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        TreeSet<Integer> topPoss = new TreeSet<Integer>();
        topPoss.add(0);
        topPoss.add(toppingCosts[0]);
        topPoss.add(toppingCosts[0] * 2);
        for (int i = 1; i < toppingCosts.length; i++) {
            HashSet<Integer> newSet = new HashSet<>();
            for (int j : topPoss) {
                newSet.add(j + toppingCosts[i]);
                newSet.add(j + toppingCosts[i] * 2);
            }
            topPoss.addAll(newSet);
        }
        
        Integer[] topArr = topPoss.toArray(new Integer[topPoss.size()]);
        
        int ans = 0;
        int minDiff = Integer.MAX_VALUE;
        
        for (int i : baseCosts) {
            for (int j : topArr) {
                int diff = target - i - j;
                if (Math.abs(diff) < minDiff) {
                    minDiff = Math.abs(diff);
                    ans = i + j;
                } else if (Math.abs(diff) == minDiff && diff > 0) {
                    ans = i + j;
                }
            }
            // int newTarget = target - i;
            // int l = 0, r = topArr.length - 1;
            // int mid;
            // while (l < r) {
            //     mid = (l + r) / 2;
            //     if (Math.abs(target - topArr[mid]) < minDiff) {
            //         ans = topArr[mid] + i;
            //         minDiff = Math.abs(target - topArr[mid]);
            //     }
            //     if (topArr[mid] > newTarget) {
            //         r = mid - 1;
            //     } else {
            //         l = mid + 1;
            //     }
            // }
        }
        
        return ans;
    }
}
