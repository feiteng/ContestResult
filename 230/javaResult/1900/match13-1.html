class Solution {
    int res;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        res = Integer.MIN_VALUE;
        for(int baseCost: baseCosts)
            helper(baseCost, 0, toppingCosts, target)
;
        return res;
    }
    public void helper(int value, int index, int[] toppingCosts, int target){
        if(Math.abs(value - target) < Math.abs(res - target)) res = value
;
        if(index < toppingCosts.length){
            helper(value, index + 1, toppingCosts, target);
            helper(value + toppingCosts[index], index + 1, toppingCosts, target);
            helper(value + toppingCosts[index] * 2, index + 1, toppingCosts, target);
        }
    }
}
