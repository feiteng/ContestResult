class Solution {
    private static int res;
    public int closestCost(int[] baseCosts, int[] toppingCosts, int target) {
        res = -1;
        for(int i = 0; i < baseCosts.length; i++) {
            dfs(baseCosts[i], 0, toppingCosts, target);
        }
        return res;
    }
    
    private void dfs(int curCost, int idx, int[] costs, int target) {
        if(idx == costs.length) {
            int diff1 = Math.abs(target - curCost);
            int diff2 = Math.abs(target - res);
            if(res == -1 || diff1 < diff2 || diff1 == diff2 && curCost < res) res = curCost;
            return;
        }
        
        dfs(curCost, idx + 1, costs, target);
        dfs(curCost + costs[idx], idx + 1, costs, target);
        dfs(curCost + costs[idx] * 2, idx + 1, costs, target);
    }
}
