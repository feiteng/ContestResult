class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int t = ruleKey.equals("type") ? 0 : (ruleKey.equals("color") ? 1 : 2), ans = 0;
        for (List<String> item : items) {
            if (item.get(t).equals(ruleValue)) ans ++;
        }
        return ans;
    }
}
