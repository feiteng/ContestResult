class Solution {
    public int countMatches(List<List<String>> items, String key, String val) {
        int idx = 0;
        if(key.equals("type")) {
            idx = 0;
        } else if (key.equals("color")) {
            idx = 1;
        } else {
            idx = 2;
        }
        int count = 0;
        for (List<String> str: items) {
            if (str.get(idx).equals(val)) {
                count ++;
            }
        }
        return count;
    }
}
