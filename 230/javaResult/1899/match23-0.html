class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int ret = 0;
        int checkIdx = 0;
        if (ruleKey.equals("color")) {
            checkIdx = 1;
        } else if (ruleKey.equals("name")){
            checkIdx = 2;
        }
        for (List<String> item : items) {
            if (item.get(checkIdx).equals(ruleValue)) {
                ret++;
            }
        }
        return ret;
    }
}
