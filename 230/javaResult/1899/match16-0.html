class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        Map<String, Integer> rk = new HashMap<>();
        rk.put("type", 0);
        rk.put("color", 1);
        rk.put("name", 2);
        int rki = rk.get(ruleKey), res = 0;
        for (List<String> i : items) {
            if (ruleValue.equals(i.get(rki))) res++;
        }
        return res;
    }
}
