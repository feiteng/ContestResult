class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int targetIndex = ruleKey.equals("type") ? 0
            : (ruleKey.equals("color") ? 1 : 2);
        int matches = 0;
        for (var item : items) {
            if (item.get(targetIndex).equals(ruleValue))
                matches++;
        }
        return matches;
    }
}
