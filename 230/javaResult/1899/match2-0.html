class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int ind = ruleKey.equals("type") ? 0 : ruleKey.equals("color") ? 1 : 2;
        int ret = 0;
        for(List<String> item : items){
            if(item.get(ind).equals(ruleValue))ret++;
        }
        return ret;
    }
}
