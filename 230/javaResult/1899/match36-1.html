class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int i = 0;
        if ("color".equals(ruleKey)) {
            i = 1;
        } else if ("name".equals(ruleKey)) 
{
            i = 2;
        }
        int ans = 0;
        for (List<String> item : items) {
            if (item.get(i).equals(ruleValue)) {
                ans++;
            }
        }
        return ans;
    }
}
