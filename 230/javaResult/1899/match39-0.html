class Solution {
    public int countMatches(List<List<String>> items, String rulekey, String ruleValue) {
        int ind = 0 ;
        if(rulekey.equals("type")){
            ind =  0;
        }else if(rulekey.equals("color")){
            ind = 1;
        }else{
            ind = 2;
        }
        int cnt =  0 ;
        for(List<String> val : items){
            if(val.get(ind).equals(ruleValue)){
                cnt++;
            }
        }
        return cnt;
    }
}
