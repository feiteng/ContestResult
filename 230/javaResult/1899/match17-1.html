class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int ruleIndex = ruleKey.equals("type") ? 0 : ruleKey.equals("color") ? 1 : 2;
        int count = 0;
        for(List<String> i: items) {
            if(i.get(ruleIndex).equals(ruleValue)) {
                count++;
            }
        }
        return count;
    }
}
