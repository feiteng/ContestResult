class Solution {
    public int countMatches(List<List<String>> items, String ruleKey, String ruleValue) {
        int key = ruleKey.equals("type")?0:ruleKey.equals("color")?1:2;
        int cnt = 0;
        for(List<String> item:items) {
            if(item.get(key).equals(ruleValue)) cnt++;
        }
        return cnt;
    }
}
