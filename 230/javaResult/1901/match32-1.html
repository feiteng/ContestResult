class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = 0;
        int sum2 = 0;
        for(int i : nums1){
            sum1 += i;
        }
        
        for(int i : nums2){
            sum2 += i;
        }
        
        if(sum1 == sum2){
            return 0;
        }
        
        return sum1 > sum2 ? helper(nums2, nums1, sum1 - sum2) : helper(nums1, nums2, sum2 - sum1);
    }
    
    private int helper(int[] small, int[] big, int diff){
        PriorityQueue<Integer> queue = new PriorityQueue<>(Collections.reverseOrder());
        
for(int i : small){
            queue.add(6 - i);
        }
        
        for(int i : big){
            queue.add(i - 1);
        }
        
        int count = 0;
        while(!queue.isEmpty()){
            diff -= queue.poll();
            count++;
            if(diff <= 0){
                return count;
            }
        }
        
        return -1;
    }
}
