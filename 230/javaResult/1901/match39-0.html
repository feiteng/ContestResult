class Solution {
    public int minOperations(int[] num1, int[] num2) {
        int a1=num1.length;
        int a2=num1.length*6;
        int b1=num2.length;
        int b2=num2.length*6;
        if(a1>b2||b1>a2){
            return -1;
        }
        
        int currs1=0,currs2=0;
        for(int i=0;i<num1.length;i++){
            currs1+=num1[i];
        }for(int i=0;i<num2.length;i++){
            currs2+=num2[i];
        }
        HashMap<Integer,Integer> map=new HashMap<>();
        
        if(currs1==currs2){
            return 0;
        }if(currs1>currs2){
            for(int i=0;i<a1;i++){
                if(map.containsKey(num1[i]-1)){
                    map.put(num1[i]-1,map.get(num1[i]-1)+1);
                }else{
                    map.put(num1[i]-1,1);
                }
            }for(int i=0;i<b1;i++){
                if(map.containsKey(6-num2[i])){
                    map.put(6-num2[i],map.get(6-num2[i])+1);
                }else{
                    map.put(6-num2[i],1);
                }
            }
        }else{
            for(int i=0;i<a1;i++){
                if(map.containsKey(6-num1[i])){
                    map.put(6-num1[i],map.get(6-num1[i])+1);
                }else{
                    map.put(6-num1[i],1);
                }
            }for(int i=0;i<b1;i++){
                if(map.containsKey(num2[i]-1)){
                    map.put(num2[i]-1,map.get(num2[i]-1)+1);
                }else{
                    map.put(num2[i]-1,1);
                }
            }
        }
        int[]arr=new int[6];
        for(int i=0;i<6;i++){
            if(map.containsKey(i)){
                arr[i]=map.get(i);
            }
        }
        int tar=Math.abs(currs1-currs2);
        
        int ans=0;
        for(int i=5;i>=0;i--){
            int curr=tar/i;
            if(tar%i!=0){
                curr++;
            }
            if(arr[i]>=curr){
                ans+=curr;
                break;
            }else{
                ans+=arr[i];
                tar-=arr[i]*i;
            }
        }
        return ans;
    }
}
