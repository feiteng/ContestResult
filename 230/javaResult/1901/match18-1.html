class Solution {
    public int minOperations(int[] num1, int[] num2) {
        int[] count1 = new int[7];
        int[] count2 = new int[7];
        
        int sum1=0;
        int sum2 = 0;
        
        for (int i = 0; i < num1.length; ++i) {
            count1[num1[i]]++;
            sum1 += num1[i];
        }
        for (int i = 0; i < num2.length; ++i) {
            count2[num2[i]]++;
            sum2 += num2[i];
        }
        
        if (sum1 > sum2) 
{
            int[] temp = count2;
            count2 = count1;
            count1 = temp;
        }
        
        int diff = Math.abs(sum2 - sum1);
        // System.out.println(diff);
        int tot = 0;
        int ans = 0;
        for (int i = 1; i < 6; ++i) {
            int last = 6 - i + 1;
            int poss = count1[i] + count2[last];
            tot = poss * (6 - i);
            if (tot > diff) {
                
                ans += ((diff + 5 - i) / (6 - i));
                diff = 0;
            } else {
                diff -= tot;
                ans += poss;
            }
            
            // System.out.println(diff);
        }
        if (diff != 0) return -1;
        return ans;
    }
}
