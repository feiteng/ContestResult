class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = 0, sum2 = 0;
        for (int num : nums1){
            sum1 += num;
        }
        for (int num : nums2){
            sum2 += num;
        }
        
        if (sum1 == sum2){
            return 0;
        }
        if (sum1 < sum2){
            return minOp(nums2, sum2, nums1, sum1);
        }
        return minOp(nums1, sum1, nums2, sum2);
    }
    
    private int minOp(int[] nums1, int sum1, int[] nums2, int sum2){
        Queue<Integer> maxHeap = new PriorityQueue<>(nums1.length, (Integer i1, Integer i2) -> i2.intValue() - i1.intValue());
        Queue<Integer> minHeap = new PriorityQueue<>(nums2.length);
        for (int num : nums1){
            maxHeap.offer(num);
        }
        for (int num : nums2){
            minHeap.offer(num);
        }
        int result = 0;
        
        while (sum1 > sum2){
            int diff = sum1 - sum2;
            int diff1 = maxHeap.peek() - 1, diff2 = 6 - minHeap.peek();
            if (diff1 == 0 && diff2 == 0){
                return -1;
            }
            if (diff <= Math.max(diff1, diff2)){
                return result + 1;
            } 
            if (diff1 >= diff2){
                sum1 = sum1 - maxHeap.peek() + 1;
                maxHeap.poll();
                maxHeap.offer(1);
            } else {
                sum2 = sum2 - minHeap.peek() + 6;
                minHeap.poll();
                minHeap.offer(6);
            }
            result++;
        }
        
        return result;
    }
}
