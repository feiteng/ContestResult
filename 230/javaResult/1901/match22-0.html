class Solution {
    public int minOperations(int[] nums1, int[] nums2) {

        if (nums1.length < nums2.length) {
            return minOperations(nums2, nums1);
        }
        if (nums1.length > nums2.length * 6) {
            return -1;
        }


        int s1 = 0, s2 = 0;
        int[] c1 = new int[7], c2 = new int[7];
        for (int n : nums1) {
            s1 += n;
            c1[n]++;
        }
        for (int n : nums2) {
            s2 += n;
            c2[n]++;
        }
        if (s1 == s2) {
            return 0;
        }

        if (s1 > s2) {
            return minOp(c1, c2, s1 - s2);
        } else {
            return minOp(c2, c1, s2 - s1);
        }
    }

    private int minOp(int[] cnt1, int[] cnt2, int target) {
        int ans = 0;
        int p1 = 6, p2 = 1, step = 5;

        while (true) {

            if ((cnt1[p1] + cnt2[p2]) * step >= target) {
                ans += (target + step - 1) / step;
                return ans;
            }
            target -= (cnt1[p1] + cnt2[p2]) * step;
            ans+= cnt1[p1] + cnt2[p2];
            p1--;
            p2++;
            step--;

        }


    }

}
