class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = helper(nums1);
        int sum2 = helper(nums2);
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
        int res = 0;
        if (sum1 > sum2) {
            int[] temp = nums1;
            nums1 = nums2;
            nums2 = temp;
            int v = sum1;
            sum1 = sum2;
            sum2 = v;
        }
        int i = 0;
        int j = nums2.length-1;
        
        while (sum1< sum2) {
            if (i>=nums1.length && j<0) {
                return -1;
            }
            else if (i>=nums1.length) {
                sum2 -= (nums2[j--]-1);
                res++;
            }
            else if (j<0) {
                sum1 += (6-nums1[i++]);
                res++;
            }
            else {
                int left = (6-nums1[i]);
                int right = (nums2[j]-1);
                if (left >= right) {
                    sum1 += left;
                    i++;
                }
                else {
                    sum2 -= right;
                    j--;
                }
                res++;
            }
        }
        
        return res;
    }
    
    int helper(int[] arr) {
        int res = 0;
        for (int val: arr) 
            res += val;
        return res;
    }
}
