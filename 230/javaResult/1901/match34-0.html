class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = add(nums1), sum2 = add(nums2);
        if (sum1 == sum2) return 0;
        if (sum1 > sum2) {
            int[] temp = nums1;
            nums1 = nums2;
            nums2 = temp;
            int t = sum1;
            sum1 = sum2;
            sum2 = t;
        }
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0, j = nums2.length - 1, ops = 0;
        //System.out.print(sum1 + "," + sum2);
        while (sum1 < sum2) {
            if ((i == nums1.length || nums1[i] == 6) && (j == -1 || nums2[j] == 1)) break;
            if (i == nums1.length || nums1[i] == 6) sum2 -= nums2[j--] - 1;
            else if (j == -1 || nums2[j] == 1) sum1 += 6 - nums1[i++];
            else if (6 - nums1[i] < nums2[j] - 1) sum2 -= nums2[j--] - 1;
            else sum1 += 6 - nums1[i++];
            ops++;
        }
        return sum1 < sum2 ? -1 : ops;
    }
    int add(int[] nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        return sum;
    }
}
