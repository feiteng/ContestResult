class Solution {
    public Integer[] getTrans(int[] hig,int low[]){
        Integer trans[]=new Integer[hig.length+low.length];
        int index=0;
        for(int i:hig){
            trans[index++]=i-1;
        }
        for(int i:low){
            trans[index++]=6-i;
        }
        Arrays.sort(trans,Collections.reverseOrder());
        return trans;
    }
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1=0,sum2=0;
        for(int i:nums1){
            sum1+=i;
        }
        for(int i:nums2){
            sum2+=i;
        }
        int diff=Integer.max(sum1,sum2)-Integer.min(sum1,sum2);
        Integer trans[];
        if(sum1>sum2){
            trans=getTrans(nums1,nums2);
        }else{
            trans=getTrans(nums2,nums1);
        }

        for(int i=0;i<trans.length;i++){
            if(diff<=0)return i;
            diff-=trans[i];
        }
        if(diff<=0)
            return trans.length;
        return -1;
        
    }
}
