class Solution {
    public int minOperations(int[] nu1, int[] nu2) {
        int n1 = nu1.length, n2 = nu2.length;
        int s1 = 0, s2 = 0;
        for (int i = 0; i < n1; ++i) s1 += nu1[i];
        for (int i = 0; i < n2; ++i) s2 += nu2[i];
        
        if (s1 == s2) return 0;
        
        if (s1 > s2) {
            int[] tmp = nu1;
            nu1 = nu2;
            nu2 = tmp;
            
            int tt = n1; n1 = n2; n2 = tt;
            tt = s1; s1 = s2; s2 = tt;
        }
        
        if (n1 * 6 < n2) return -1;
        
        Arrays.sort(nu1);
        Arrays.sort(nu2);
        
        int ans = 0;
        int p1 = 0, p2 = n2 - 1;
        while (s1 < s2) {
            ++ans;
            if (p2 < 0 || (p1 < n1 && 6 - nu1[p1] > nu2[p2] - 1)) {
                s1 += 6 - nu1[p1++];
            }
            else {
                s2 -= nu2[p2--] - 1;               
            }
        }
        
        return ans;
    }
}
