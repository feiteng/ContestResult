class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = sum(nums1);
        int sum2 = sum(nums2);
        
        if(sum1 == sum2) return 0;
        if(sum1 > sum2) return minOperations(nums2, nums1);
        
        // sum1 < sum2
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
        int diff = sum2 - sum1;
        int ops = 0;
        // Make them closer and closer. Make diff 
        int idx1 = 0, idx2 = nums2.length-1;
        while(diff > 0 && (idx1 < nums1.length || 0 <= idx2)) {
            ops++;
            if(idx1 == nums1.length) {
                // move 2
                diff -= dist(nums2[idx2], 1);
                idx2--;
                continue;
            } else if(idx2 < 0) {
                // move 1
                diff -= dist(nums1[idx1], 6);
                idx1++;
                continue;
            }
            
            // Find the biggest diff
            int dist1 = dist(nums1[idx1], 6);
            int dist2 = dist(nums2[idx2], 1);
            if(dist1 <= dist2) {
                // move 2
                diff -= dist2;
                idx2--;
            } else {
                // move 1
                diff -= dist1;
                idx1++;
            }
        }
        
        if(diff > 0) return -1;
        return ops;
    }
    
    int dist(int a, int b) {
        if(a > b) return a-b;
        return b-a;
    }
    
    int sum(int[] nums) {
        int a = 0;
        for(int n: nums) a += n;
        return a;
    }
}
