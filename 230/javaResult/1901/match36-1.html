class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = calculateSum(nums1);
        int sum2 = calculateSum(nums2);
        if (sum1 == sum2) return 0;
        // make nums1 > nums2
        if (sum1 < sum2) {
            int[] temp = nums1;
            nums1 = nums2;
            nums2 = temp;
            int t = sum1;
            sum1 = sum2;
            sum2 = t;
        }
        
        int res = 0;
        int diff = sum1 - sum2;
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int n = nums1.length;
        int m = nums2.length;
        int i = n-1;
        int j = 0;
        while (diff > 0) {
            if (i < 0 && j == m) return -1;
            if (i < 0) {
                if (nums2[j] == 6) return -1;
                int d = 6 - nums2[j];
                if (d >= diff) return res + 1;
                diff -= d;
                j++;
            } else if (j == m) {
                if (nums1[i] == 1) return -1;
                int d = nums1[i] - 1;
                if (d >= diff) return res + 1;
                diff -= d;
                i--;
            } else {
                int d1 = 6 - nums2[j];
                int d2 = nums1[i] - 1;
                if (d1 >= d2) {
                    if (d1 >= diff) return res + 1;
                    j++;
                    diff -= d1;
                } else {
                    if (d2 >= diff) return res + 1;
                    i--;
                    diff -= d2;
                }
            }
            res++;
        
}
        return res;
    }
    
    private int calculateSum(int[] arr) {
        int sum = 0;
        for (int num: arr) sum += num;
        return sum;
    }
}
