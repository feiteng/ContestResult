class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int max1 = nums1.length * 6;
        int max2 = nums2.length * 6;
        
int min1 = nums1.length;
        int min2 = nums2.length;
        if (max1 < min2 || max2 < min1) {
            return -1;
        }
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int sum1 = 0;
        int sum2 = 0;
        for (int i : nums1) {
            sum1 += i;
        }
        for (int i : nums2) {
            sum2 += i;
        }
        int dis = Math.abs(sum1 - sum2);
        
int ans = 0;
        int l1 = 0, l2 = 0, r1 = nums1.length - 1, r2 = nums2.length - 1;
        while (dis > 0) {
            if (sum1 > sum2) {
                int item1 = r1 < 0 ? 0 : (nums1[r1] - 1);
                int item2 = l2 >= nums2.length ? 0 : (6 - nums2[l2]);
                if (item1 > item2) {
                    r1--;
                    dis -= item1;
                } else {
                    l2++;
                    dis -= item2;
                }
            } else {
                int item2 = r2 < 0 ? 0 : (nums2[r2] - 1);
                int item1 = l1 >= nums1.length ? 0 : (6 - nums1[l1]);
                if (item1 > item2) {
                    l1++;
                    dis -= item1;
                } else {
                    r2--;
                    dis -= item2;
                }
            }
            ans++;
        }

        return ans;
    }
}
