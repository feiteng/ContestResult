class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int minLen = Math.min(len1, len2);
        int maxLen = Math.max(len1, len2);
        if (minLen * 6 < maxLen*1) {
            return -1;
        }
        
        int[] count1 = new int[7];
        int[] count2 = new int[7];
        
        int sum1 = 0;
        for (int num: nums1) {
            count1[num]++;
            sum1 += num;
        }
        
        int sum2 = 0;
        for (int num : nums2) {
            count2[num]++;
            sum2 += num;
        }
        // 默认1是小的
        int cha = Math.abs(sum1-sum2);
        if (cha == 0) {
            return 0;
        }
        int[] temp;
        if (sum1 > sum2) {
            temp = count1;
            count1 = count2;
            count2 = temp;
        }
        int index1 = 1;
        int index2 = 6;
        int dis = 5;
        int result = 0;
        while (cha > 0 && dis > 0) {
            if (count1[index1] > 0) {
                cha -= dis;
                count1[index1]--;
                result++;
            } else if (count2[index2] > 0) {
                cha -= dis;
                count2[index2]--;
                result++;
            } else {
                dis--;
                index1++;
                index2--;
            }
        }
        return result;
    }
}
