class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int a = getSum(nums1);
        int b = getSum(nums2);
        if (a == b) return 0;
        if (a > b) return minOperations(nums2, nums1);
        
        int diff = b - a;
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
        int count = 0;
        int i = 0;
        int j = nums2.length-1;
        while (diff > 0 && (i < nums1.length || j >= 0)) {
            int c = i < nums1.length ? nums1[i] : 6;
            int d = j >= 0 ? nums2[j] : 1;
            
            int diff1 = 6 - c;
            int diff2 = d - 1;
            
            if (diff1 == diff2 && diff1 == 0) return -1
;
            
            if (diff1 > diff2) {
                diff -= diff1;
                i++;
            } else {
                diff -= diff2;
                j--;
            } 
            count++;
        }
        
        return count;
    
}
    
    private int getSum(int[] nums) {
        int sum = 0;
        for (int n : nums) sum += n;
        return sum;
    }
}
