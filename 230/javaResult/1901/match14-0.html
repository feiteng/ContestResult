class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1=getSum(nums1);
        int sum2=getSum(nums2);
        if (sum1==sum2)return 0;
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        if (sum1<sum2){
            int[] temp=nums1;
            nums1=nums2;
            nums2=temp;
        }
        int ans=0;
        int diff=Math.abs(sum1-sum2);
        int i=nums1.length-1;
        int j=0;
        while (i>=0||j<nums2.length){
            if (diff<=0)break;
            if (i<0){
                if (nums2[j]==6) {
                    j++;
                    continue;
                }
                int diff2=6-nums2[j];
                diff-=diff2;
                ans++;
                j++;
                continue;
            }
            if (j==nums2.length){
                if (nums1[i]==1){
                    i--;
                    continue;
                }
                int diff1=nums1[i]-1;
                diff-=diff1;
                ans++;
                i--;
                continue;
            }
            int diff2=6-nums2[j];
            int diff1=nums1[i]-1;
            if (diff1==0&&diff2==0){
                i--;
                j++;
                continue;
            }
            ans++;
            if (diff1>diff2){
                diff-=diff1;
                i--;
            }else{
                diff-=diff2;
                j++;
            }
        }
        return diff<=0?ans:-1;
    }
    private int getSum(int[] nums){
        int ans=0;
        for(int num:nums)ans+=num;
        return ans;
    }
}
