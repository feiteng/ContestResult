class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int[] a = new int[6];
        int[] b = new int[6];

        int sum1 = 0;
        for (int i : nums1) {
            a[i - 1]++;
            sum1 += i;
        }

        int sum2 = 0;
        for (int i : nums2) {
            b[i - 1]++;
            sum2 += i;
        }

        int ans = 0;
        if (sum1 == sum2) {
            return ans;
        } else if (sum1 > sum2) {
            int[] t = a;
            a = b;
            b = t;

            int tt = sum1;
            sum1 = sum2;
            sum2 = tt;
        };

        int cha = sum2 - sum1;
        // sum1小 sum2大

        for (int i = 0; i < 6; i++) {
            int s = a[i] + b[5 - i];
            if (s * (5 - i) >= cha) {
                int tt = cha / (5 - i);
                if (tt * (5 - i) < cha) {
                    tt = tt + 1;
                }
                ans += tt;
                return ans;
            } else {
                ans += s;
                cha -= (s * (5 - i));
            }
        }

        return -1;

    }
}
