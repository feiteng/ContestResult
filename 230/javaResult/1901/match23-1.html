class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        
if (nums1 == null || nums2 == null) {
            return -1;
        }
        
        int sum1 = 0, sum2 = 0;
        for (int i = 0; i < nums1.length; i++) {
            sum1 += nums1[i];
        }
        for (int i = 0; i < nums2.length; i++) {
            sum2 += nums2[i];
        }
        
        if (sum1 == sum2) {
            return 0;
        }
        
        // nums1 -> large, nums2 -> small
        if (sum1 < sum2) {
            int[] tmp = nums1;
            
nums1 = nums2;
            nums2 = tmp;
        }
        
        int target = Math.abs(sum1 - sum2);
        Queue<Integer> candidates = new PriorityQueue<Integer>((x1, x2) -> x2 - x1);
        for (int i = 0; i < nums1.length; i++) {
            candidates.offer(nums1[i] - 1);
        }
        for (int i = 0; i < nums2.length; i++) {
            candidates.offer(6 - nums2[i]);
        }
        
        int result = 0;
        while (target > 0 && !candidates.isEmpty()) {
            int peek = candidates.poll();
            target -= peek;
            result++;
        }
        if (target > 0) {
            return -1;
        }
        return result;
    }
}
