class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = 0 ;
        for(int val : nums1){
            sum1+=val;
        }
        int sum2 = 0  ;
        for(int val : nums2){
            sum2+=val;
        }
        int gre[];
        int sma[];
        if(sum1>sum2){
            gre = new int[nums1.length];
            for(int i=0;i<nums1.length;i++){
                gre[i] = nums1[i];
            }
            sma = new int[nums2.length];
            for(int i=0;i<nums2.length;i++){
                sma[i] = nums2[i];
            }
        }else{
            gre = new int[nums2.length];
            for(int i=0;i<nums2.length;i++){
                gre[i] = nums2[i];
            }
            sma = new int[nums1.length];
            for(int i=0;i<nums1.length;i++){
                sma[i] = nums1[i];
            }
        }
        sum1 = 0 ; 
        sum2 = 0 ; 
        for(int val : gre)
            sum1+=val;
        for(int val : sma)
            sum2+=val;
        Arrays.sort(sma);
        Arrays.sort(gre);
        for(int i=0,j=gre.length-1;i<j;i++,j--){
            int temp = gre[i];
            gre[i] = gre[j];
            gre[j] = temp;
        }
        int ind1 = 0 ; 
        int ind2 =0 ; 
        int cnt = 0;
        while(ind1<gre.length && ind2<sma.length){
            if(sum1==sum2){
                return cnt;
            }
            int pro1 = Math.min(gre[ind1]-1,sum1-sum2);
            int pro2 = Math.min(6-sma[ind2], sum1-sum2);
            if(pro1>pro2){
                sum1-=pro1;
                if(pro1!=0)
                    cnt++;
                ind1++;
            }else{
                sum2+=pro2;
                if(pro2!=0)
                    cnt++;
                ind2++;
            }
        }
        
        while(ind1<gre.length){
            if(sum1==sum2){
                return cnt;
            }
            int pro1 = Math.min(gre[ind1]-1,sum1-sum2);
            sum1-=pro1;
            if(pro1!=0)
                cnt++;
            ind1++;
        }
        
        while(ind2<sma.length){
            if(sum1==sum2){
                return cnt;
            }
            int pro1 = Math.min(6-sma[ind2],sum1-sum2);
            sum2+=pro1;
            if(pro1!=0)
                cnt++;
            ind2++;
        }
        
        if(sum1==sum2){
            return cnt;
        }
        return -1;
        
    }
}
