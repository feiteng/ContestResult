class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length * 6 < nums2.length || nums1.length > nums2.length * 6) {
            return -1;
        }
        int sum1 = 0;
        int sum2 = 0;
        for (int i : nums1) sum1 += i;
        for (int i : nums2) sum2 += i;
        if (sum1 < sum2) {
            return help(nums1, nums2, sum2 - sum1);
        }
        return help(nums2, nums1, sum1 - sum2);
    }
    
    private int help(
int[] nums1, int[] nums2, int diff) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0;
        int j = nums2.length - 1;
        int res = 0;
        while (diff > 0) {
            int a = i < nums1.length ? 6 - nums1[i] : 0;
            int b = j >= 0 ? nums2[j] - 1 : 0;
            
if (a > b) {
                diff -= a;
                i++;
            } else {
                diff -= b;
                j--;
            }
            res++;
        }
        return res;
    }
}
