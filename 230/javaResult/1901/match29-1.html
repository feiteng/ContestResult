class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length * 6 < nums2.length || nums2.length * 6 < nums1.length) {
            return -1;
        }
        
        
int sum1 = 0;
        for (int num : nums1) {
            sum1 += num;
        }
        
        int sum2 = 0;
        for (int num : nums2) {
            sum2 += num;
        }
        
        if (sum1 == sum2) {
            return 0;
        }
        
        if (sum1 > sum2) {
            int[] temp = nums1;
            nums1 = nums2;
            nums2 = temp;
            int t = sum1;
            sum1 = sum2;
            sum2 = t;
        }
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
int i = 0;
        int j = nums2.length - 1;
        
        int res = 0;
        while (sum1 < sum2) {
            if (i >= nums1.length) {
                sum2 -= nums2[j] - 1;
                j--;
            } else if (j < 0) {
                sum1 += 6 - nums1[i];
                i++;
            } else {
                if (6 - nums1[i] > nums2[j] - 1) {
                    sum1 += 6 - nums1[i];
                    i++;
                } else {
                    sum2 -= nums2[j] - 1;
                    j--;
                }
            
}
            res++;
        }
        
        return res;
    }
}
