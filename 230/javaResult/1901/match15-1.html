class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length > 6 * nums2.length || nums2.length > 6 * nums1.length) {
            return -1;
        }
        
int[] cnt1 = new int[7];
        int[] cnt2 = new int[7];
        int sum1 = 0;
        int sum2 = 0;
        for (int n : nums1) {
            cnt1[n]++;
            sum1 += n;
        }
        for (int n : nums2) {
            cnt2[n]++;
            sum2 += n;
        }
        if (sum1 == sum2) {
            return 0;
        }
        if (sum1 > sum2) {
            return minOperations(cnt1, cnt2, sum1 - sum2);
        } else {
            return minOperations(cnt2, cnt1, sum2 - sum1);
        
}    
    }
    private int minOperations(int[] large, int[] small, int diff) {
        int ops = 0;
        for (int d : new int[]{5, 4, 3, 2, 1}) {
            int cnt = small[6 - d] + large[1 + d];
            int total = cnt * d;
            if (total >= diff) {
                if (diff % d == 0) {
                    ops += diff / d;
                } else {
                    ops += diff / d + 1;
                }
                return ops;
            
} else {
                diff -= total;
                ops += cnt;
            }
        }
        return ops;
    }
}
