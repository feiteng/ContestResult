class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        
int sum1 = 0, sum2 = 0;
        
        for(int i = 0; i < nums1.length; i++) {
            sum1 += nums1[i];
        }
        for(int i = 0; i < nums2.length; i++) {
            sum2 += nums2[i];
        }
        if(sum1 == sum2) {
            return 0;
        }
        int remain = Math.abs(sum1 - sum2);
        if(remain == 0) return 0;
        
PriorityQueue<Integer> queue = new PriorityQueue<>((a, b) -> b - a);
        if(sum1 < sum2) {
            for(int i = 0; i < nums2.length; i++) {
                queue.offer(nums2[i] - 1);
            }
            for(int i = 0; i < nums1.length; i++) {
                queue.offer(6 - nums1[i]);
            }
        } else {
            for(int i = 0; i < nums1.length; i++) {
                queue.offer(nums1[i] - 1);
            }
            for(int i = 0; i < nums2.length; i++) {
                queue.offer(6 - nums2[i]);
            }
        }
        int step = 0;
        while(!queue.isEmpty()) {
            int val = queue.poll();
            remain -= val;
            step++;
            if(remain <= 0) {
                return step;
            }
        }
        return remain > 0 ? -1 : step;
    }
}
