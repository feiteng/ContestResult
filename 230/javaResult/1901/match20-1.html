class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int m = nums1.length;
    int n = nums2.length;

    int[] c1 = new int[7];
    int[] c2 = new int[7];
    
int sum1 = 0;
    for (int v : nums1) {
      c1[v]++;
      sum1 += v;
    }
    int sum2 = 0;
    for (int v : nums2) {
      c2[v]++;
      sum2 += v;
    }
    if (sum1 == sum2) {
      return 0;
    }
    
if (m * 6 < n || n * 6 < m) {
      return -1;
    }

    if (sum1 > sum2) {
      int t = sum1;
      sum1 = sum2;
      sum2 = t;
      int[] c = c1;
      c1 = c2;
      c2 = c;
    
}

    int diff = sum2 - sum1;
    // System.out.format("c1:%s c2:%s diff:%d\n", Utils.trace(c1), Utils.trace(c2), diff);

    // Now always sum1 < sum2
    // Turn 1 to 6 in A and 6 to 1 in B as much as possible
    int ans = 0;
    int d = 5;
    while (diff > 0) {
      // turn l in A to be 6, turn h in B to be 1 as much as possible
      if (c1[6 - d] > 0) {
        if (c1[6 - d] * d >= diff) {
          ans += (diff + d - 1) / d;
          return ans;
        } else {
          ans += c1[6 - d];
          diff -= c1[6 - d] * d;
        }
      }

      if (c2[d + 1] > 0) 
{
        if (c2[d + 1] * d >= diff) {
          ans += (diff + d - 1) / d;
          return ans;
        } else {
          ans += c2[d + 1];
          diff -= c2[d + 1] * d;
        }
      }

      // System.out.format("  d:%d ans:%d diff:%d\n", d, ans, diff);
      d--;
    }
    return ans;
    }
}
