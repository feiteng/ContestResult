class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int n = nums1.length, m = nums2.length, sum1 = 0, sum2 = 0;
        if (n > 6 * m || m > 6 * n)
            return -1;
        
for (int num1: nums1)
            sum1 += num1;
        for (int num2 : nums2)
            sum2 += num2;
        if (sum1 == sum2)
            return 0;
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int ans = 0;
        if (sum1 > sum2) {
            
int i = n - 1, j = 0;
            while (sum1 > sum2) {
                if (j == m || nums1[i] - 1 > 6 - nums2[j]) {
                    sum1 -= nums1[i--] - 1;
                }
                else {
                    sum2 += 6 - nums2[j++];
                }
                ans++;                   
            }
        } else {
           int i = 0, j = m - 1;
            while (sum2 > sum1) {
                if (j == -1 || 6 - nums1[i] > nums2[j] - 1) {
                    sum1 += 6 - nums1[i++];
                }
                else {
                    sum2 -= nums2[j--] - 1;
                }
                ans++;                   
            } 
        }  
        return ans;      
    }
}
