class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
        int sum1 = 0;
        for (int i = 0; i < nums1.length; i++) {
            sum1 += nums1[i];
        }
        
        int sum2 = 0;
        for (int i = 0; i < nums2.length; i++) {
            sum2 += nums2[i];
        }
        
        // System.out.println(sum1);
        // System.out.println(sum2);
        
        int diff = sum1 - sum2;
        // System.out.println(diff);
        
        if (diff == 0) {
            return 0;
        }
        
        if (diff < 0) {
            int[] tmp = nums1;
            nums1 = nums2;
            nums2 = tmp;
            diff = -diff;
        }
        
        if (nums1[nums1.length - 1] == 1 && nums2[0] == 6) {
            return -1;   
        }
        
        int steps = 0;
        int i = nums1.length - 1;        
        int j = 0;
        
        while (diff > 0 && i >= 0 && j < nums2.length) {
            steps++;
            if (nums1[i] - 1 >= 6 - nums2[j]) {
                if (diff <= nums1[i] - 1) {
                    return steps;
                }
                else {
                    diff -= nums1[i] - 1;                    
                }
                i--;
            }
            else {
                if (diff <= 6 - nums2[j]) {
                    return steps;
                }
                else {
                    diff -= 6 - nums2[j];
                }
                j++;
            }
        }
        while (diff > 0 && i >= 0) {
            steps++;
            if (nums1[i] - 1 > 0) {
                if (diff <= nums1[i] - 1) {
                    return steps;
                }
                else {
                    diff -= nums1[i] - 1;                    
                }
                i--;
            }
            else {
                return -1;
            }            
        }
        while (diff > 0 && j < nums2.length) {
            steps++;
            if (6 - nums2[j] > 0) {
                if (diff <= 6 - nums2[j]) {
                    return steps;
                }
                else {
                    diff -= 6 - nums2[j];
                }
                j++;
            }
            else {
                return -1;
            }
        }
        
        if (diff <= 0) {
            return steps;
        }
        else {
            return -1;
        }
        
    }
}
