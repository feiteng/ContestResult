class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length > nums2.length * 6 || nums2.length > nums1.length * 6) {
            return -1;
        }
        
        int sum1 = 0, sum2 = 0;
        
        for (int i : nums1) {
            sum1 += i;
        }
        
        for (int i : nums2) {
            sum2 += i;
        }
        
        if (sum1 > sum2) {
            return minOp(nums1, nums2, sum1, sum2);
        } else {
            return minOp(nums2, nums1, sum2, sum1);
        }
    }
    
    public int minOp(int[] nums1, int[] nums2, int sum1, int sum2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = nums1.length - 1;
        int j = 0;
        
        int ans = 0;
        
        while (sum1 > sum2) {
            ans++;
            if (i == -1) {
                sum2 += 6 - nums2[j++];
                continue;
            } else if (j == nums2.length) {
                sum1 -= nums1[i--] - 1;
                continue;
            }
            
            if (nums1[i] - 1 >= 6 - nums2[j]) {
                sum1 -= nums1[i--] - 1;
            } else {
                sum2 += 6 - nums2[j++];
            }
        }
        
        return ans;
    }
}
