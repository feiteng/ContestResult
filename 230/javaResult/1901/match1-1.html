class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int[] count1 = new int[7];
        for (int i = 0; i < nums1.length; i++) {
            count1[nums1[i]]++;
        }
        int[] count2 = new int[7];
        for (int i = 0; i < nums2.length; i++) {
            count2[nums2[i]]++;
        }

        
int sum1 = 0;
        for (int i = 0; i < count1.length; i++) {
            sum1 += i * count1[i];
        }

        int sum2 = 0;
        for (int i = 0; i < count2.length; i++) {
            sum2 += i * count2[i];
        }
        
if (sum1 > sum2) {
            int[] tmp = count1;
            count1 = count2;
            count2 = tmp;
            int fake = sum1;
            sum1 = sum2;
            sum2 = fake;
        }
        int diff = sum2 - sum1;
        
int res = 0;
        for (int i = 1; i < count1.length - 1; i++) {
            int val = count1[i] * (6 - i) + count2[7 - i] * (6 - i);
            if (diff <= val) {
                int cur = diff / (6 - i);
                diff -= (6 - i) * cur;
                res += cur;
                if (diff > 0) {
                    res++;
                    
                
}
                return res;
            } else {
                diff -= val;
                res += val / (6 - i);
            }

        }
        return -1;
    }
}
