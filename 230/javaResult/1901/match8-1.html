class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = sum(nums1), sum2 = sum(nums2);
        if(sum1 < sum2) {
            return minOperations(nums2, nums1);
        }
        // 1 > 2
        int[] n1 = new int[6], n2 = new int[6];
        for(int v : nums1) {
            n1[v - 1]++;
        }
        for(int v : nums2) {
            n2[6 - v]++;
        }
        
        int diff = sum1 - sum2, cnt = 0;
        int res = 0;
        for(int i = 5; i > 0; i--) {
            cnt += n1[i] + n2[i];
            while(diff >= i && cnt > 0) {
                diff -= i;
                cnt--;
                res++;
                //System.out.println(i);
            }
        
}
        
        return diff == 0 ? res : -1;
    }
    
    private int sum(int[] nums) {
        int sum = 0;
        for(int v : nums) {
            sum += v;
        }      
        return sum;
    }
}
