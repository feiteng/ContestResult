class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int s1 = 0, s2 = 0, m = nums1.length, n = nums2.length ; 
        for ( int i = 0 ; i < m ; i ++ ) s1 += nums1[i] ; 
        for ( int j = 0 ; j < n ; j ++ ) s2 += nums2[j] ; 
        if ( s1 == s2 ) return 0 ; 
        if ( s1 < s2 ) {
            int[] t = nums1 ; 
            nums1 = nums2 ; 
            nums2 = t ; 
            int st = s1 ; 
            s1 = s2 ; 
            s2 = st ; 
        }
        // now s1 > s2 
        Arrays.sort(nums1) ; 
        Arrays.sort(nums2) ; 
        m = nums1.length ; 
        n = nums2.length ; 
        
        int ans = 0 ; 
        int iA = m-1, iB = 0, cA, cB ; 
        while (s1 > s2 ) {
            if ( iA >= 0 ) cA = nums1[iA] ; else cA = 1 ; 
            if ( iB < n ) cB = nums2[iB] ; else cB = 6 ; 
            if ( cA == 1 && cB == 6 ) break ; 
            ans ++ ; 
            if ( cA - 1 > 6 - cB ) {
                iA -- ; 
                s1 += (1 - cA) ; 
            }
            else {
                iB ++ ; 
                s2 += (6 - cB) ; 
            }
        }
        if ( s1 > s2 ) ans = -1 ; 
        return ans ; 
    }
}
