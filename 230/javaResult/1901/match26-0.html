class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int[] count1 = new int[7];
        int[] count2 = new int[7];
        int sum1 = 0;
        int sum2 = 0;
        for (int i1 : nums1) {
            count1[i1]++;
            sum1 += i1;
        }
        
        for (int i2 : nums2) {
            count2[i2]++;
            sum2 += i2;
        }
        
        if (sum1 == sum2) {
            return 0;
        } else if (sum1 > sum2) {
            return find(Math.abs(sum1 - sum2), count1, count2);
        } else {
            return find(Math.abs(sum1 - sum2), count2, count1);
        }
    }
    
    int find(int diff, int[] countL, int[] countS) {
        int gap = 5;
        int result = 0;
        for (int i = 1; i <= 5; i++) {
            if ((countS[i] + countL[7 - i]) * gap >= diff) {
                return result + (diff / gap) + ((diff % gap == 0) ? 0 : 1);
            } else {
                result += countS[i] + countL[7 - i];
                diff -= (countS[i] + countL[7 - i]) * gap;
            }
            gap--;
        }
        
        return -1;
    }
}
