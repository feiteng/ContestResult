class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length > nums2.length * 6 || nums1.length * 6 < nums2.length) return -1;
        int sum1 = 0;
        int sum2 = 0;
        for (int num : nums1){
            sum1 += num;
        }
        for (int num : nums2){
            sum2 += num;
        }
        if (sum1 == sum2) return 0;
        
        PriorityQueue<Integer> minQ = new PriorityQueue<Integer>();
        PriorityQueue<Integer> maxQ = new PriorityQueue<Integer>((a, b) -> (b - a));
        if (sum1 > sum2) {
            for (int num : nums1){
                maxQ.add(num);
            }
            for (int num : nums2){
                minQ.add(num);
            }
        }
        else {
            for (int num : nums2){
                maxQ.add(num);
            }
            for (int num : nums1){
                minQ.add(num);
            }
        }
        
        int operations = 0;
        int diff = Math.abs(sum1 - sum2);
        while(diff != 0){
            // max gain
            //System.out.println(diff);
            operations++;
            int max = maxQ.peek();
            int min = minQ.peek();
            if (max  - 1 < 6 - min){
                // by increase minQ head will make the biggest differnece
                int adjustment = 6 - min;
                if (adjustment >= diff) {
                    return operations;
                }
                else {
                    diff -= adjustment;
                    minQ.poll();
                    minQ.add(6);
                }
            }
            else {
                int adjustment = max - 1;
                if (adjustment >= diff) {
                    return operations;
                }
                else {
                    diff -= adjustment;
                    maxQ.poll();
                    maxQ.add(1);
                }
            }
        }
        
        
        return operations;

    }
}
