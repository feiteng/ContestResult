class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int v1 = 0, v2 = 0;
        for (int v : nums1) {
            v1 += v;
        }
        for (int v : nums2) {
            v2 += v;
        }
        // check the range
        if (v1 == v2) {
            return 0;
        }
        // make sum of num1 smaller.
        if (v1 > v2) {
            int[] t = nums1;
            nums1 = nums2;
            nums2 = t;
        }
        int n1 = nums1.length;
        int n2 = nums2.length;
        int min1 = n1, max1 = 6 * n1;
        int min2 = n2, max2 = 6 * n2;
        if (min1 > max2 || min2 > max1) {
            return -1;
        }
        int[] count1 = new int[7];
        int[] count2 = new int[7];
        for (int v : nums1) {
            count1[v]++;
        }
        for (int v : nums2) {
            count2[v]++;
        }
        int diff = Math.abs(v2 - v1);
        int step = 0;
        for (int i = 5; i >= 1; i--) {
            int c = count2[i + 1] + count1[6 - i];
            // could be finished in the current move.
            if (diff <= c * i) {
                int v = diff % i == 0 ? 0 : 1;
                v += diff / i;
                step += v;
                return step;
            } else {
                step += c;
                diff -= c * i; 
            }
        }
        return step;
    }
}
