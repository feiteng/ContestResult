class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        if (n1 > n2 * 6 || n2 > n1 * 6) {
            return -1;
        }
        
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);

        int sum1 = 0;
        int sum2 = 0;

        for (int num : nums1) {
            sum1 += num;
        }

        for (int num : nums2) {
            sum2 += num;
        }

        int ans = 0;
        if (sum1 < sum2) {
            int[] tmp = nums1;
            nums1 = nums2;
            nums2 = tmp;

            sum1 ^= sum2;
            sum2 ^= sum1;
            sum1 ^= sum2;
        }
        
        int minConst = 1;
        int maxCoust = 6;
        
        int diff = sum1 - sum2;
        int cur1 = nums1.length - 1;
        int cur2 = 0;

        while (diff > 0) {
            ans++;
            if (cur1 >= 0 && nums1[cur1] - 1 >= 6 - nums2[cur2]) {
                int delta = nums1[cur1] - 1;
                diff -= delta;
                cur1--;
            } else {
                int delta = 6 - nums2[cur2];
                diff -= delta;
                cur2++;
            }
        }
        
        return ans;
    }
}
