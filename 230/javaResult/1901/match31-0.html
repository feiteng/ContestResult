class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int longw = Math.max(nums1.length, nums2.length);
        int shortw = Math.min(nums1.length, nums2.length);
        if (shortw * 6 < longw) {
            return -1;
        }
        int sum1 = 0;
        int sum2 = 0;
        for (int i : nums1) {
            sum1 += i;
        }
        for (int i : nums2) {
            sum2 += i;
        }   
        if (sum1 == sum2) {
            return 0;
        }
        int bigger = 0;
        int smaller = 0;
        int[] bigA = new int[1];
        int[] smallA = new int[1];
        if (sum1 < sum2) {
            bigger = sum2;
            bigA = nums2;
            smaller = sum1;
            smallA = nums1;
        } else {
            bigger = sum1;
            bigA = nums1;
            smaller = sum2;
            smallA = nums2;
        }
        return helper(bigA, bigger, smallA, smaller);                
    }
    public int helper(int[] arr1, int sum1, int[] arr2, int sum2) {
        int diff = sum1 - sum2;
        int res = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>((o1 , o2) -> {
            if (Math.abs(o1) == Math.abs(o2)) {
                return 0;
            }
            if (Math.abs(o1) > Math.abs(o2)) {
                return -1;
            }
            return 1;
        });
        for (int i = 0; i < arr1.length; i++) {
            pq.offer(1 - arr1[i]);
        }
        for (int i = 0; i < arr2.length; i++) {
            pq.offer(6 - arr2[i]);
        }        
        while (pq.size() > 0) {
            int cur = Math.abs(pq.poll());
            diff = diff - cur;
            res++;
            if (diff <= 0) {
                return res;
            }
        }
        return -1;
    }
}
