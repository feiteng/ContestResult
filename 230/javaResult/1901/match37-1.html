class Solution {
  
int doit(int[] cnt, int sum, int target) {
    if(sum == target) return 0;
    int v = 0;
    if(sum < target) {
      for(int j = 1; j < 6; j++) {
        if(sum + (6-j) * cnt[j] >= target) {
          v += (target - sum + (6-j-1)) / (6-j);
          return v;
        
}
        v += cnt[j];
        sum += (6-j)*cnt[j];
      }
    }
    else {
      for(int j = 6; j > 1; j--) {
        if(target + (j-1) * cnt[j] >= sum) {
          v += (sum - target + (j-2)) / (j-1);
          return v;
        }
        v += cnt[j];
        target += (j-1)*cnt[j];
      }      
    }
    return 100_000_000;
  }
  
    
public int minOperations(int[] nums1, int[] nums2) {
      int n1 = nums1.length, n2 = nums2.length;
      int[] c1 = new int[7], c2 = new int[7];
      int s1 = 0, s2 = 0;
      
for(int x : nums1){ c1[x]++; s1 += x; }
      for(int y : nums2){ c2[y]++; s2 += y; }
      if(s1 == s2) return 0;
      int ans = 100_000_000;
      for(int tar = Math.min(s1, s2); tar <= Math.max(s1, s2); tar++) {
        int v1 = doit(c1, s1, tar);
        int v2 = doit(c2, s2, tar);
        ans = Math.min(ans, v1+v2);
        // System.out.format("tar = %d, s1 = %d, v1 = %d,   s2 = %d, v2 = %d\n", tar, s1, v1, s2, v2);
      }
      return ans == 100_000_000 ? -1 : ans;
    }
}
