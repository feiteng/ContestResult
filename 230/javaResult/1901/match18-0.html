class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int[] count1=new int[7], count2=new int[7];
        int sum1=0, sum2=0;
        for(int i=0; i<nums1.length; i++){
            sum1+=nums1[i];
            count1[nums1[i]]++;
        }
        for(int i=0; i<nums2.length; i++){
            sum2+=nums2[i];
            count2[nums2[i]]++;
        }
        
        if(sum1==sum2) return 0;
        if(sum1>sum2){
            int[] temp=count1;
            count1=count2;
            count2=temp;
        }
        int diff=Math.abs(sum1-sum2);
        
        int ops=0;
        for(int i=1; i<6; i++){
            //count 1 from count1 and 6 from count2 - contribution=5
            int count=count1[i]+count2[6-i+1];
            int contribution=6-i;
            if(count*contribution>=diff){
                ops+=diff/contribution;
                if(diff%contribution!=0) ops++;
                diff=0;
            }
            else{ 
                ops+=count;
                diff-=count*contribution;
            }
            if(diff==0) break;            
        }
        return diff>0?-1:ops;
    }
}
