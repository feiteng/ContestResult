class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        
        int[] big;
        int[] small;
        if (n1 >= n2) {
            big = nums1;
            small = nums2;
        } else {
            big = nums2;
            small = nums1;
        }
        
        int bigMin = 1 * big.length;
        int smallMax = 6 * small.length;
        if (bigMin > smallMax) {
            return -1;
        }
        
        int bigSum = 0;
        int smallSum = 0;
        for (int i = 0; i < big.length; i++) {
            bigSum += big[i];
        }
        for (int i = 0; i < small.length; i++) {
            smallSum += small[i];
        }
        
        int[] big2;
        int[] small2;
        if (bigSum >= smallSum) {
            big2 = big;
            small2 = small;
        } else {
            big2 = small;
            small2 = big;
        }
        
        int gap = Math.abs(bigSum - smallSum);
        
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> (b-a));
        
        //turn smaller -> change to 1
        for (int temp : big2) {
            pq.add(temp - 1);
        }
        //turn biger -> change to 6
        for (int temp : small2) {
            pq.add(6 - temp);
        }
        
        int ans = 0;
        while(!pq.isEmpty()) {
            
            if (gap <= 0) {
                return ans;
            }
            
            int curr = pq.remove();
            ans++;
            gap -= curr;
            
        }
        
        if (gap <= 0) {
            return ans;
        } else {
            return -1;
        }
    }
}
