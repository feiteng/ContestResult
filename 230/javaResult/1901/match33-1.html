class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length > 6 * nums2.length || nums2.length > 6 * nums1.length) {
            return -1;
        }
        
        int[] countsA = new int[7];
        int[] countsB = new int[7];
        
        int sumA = 0;
        int sumB = 0;
        
        
for (int x : nums1) {
            countsA[x] ++;
            sumA += x;
        }
        for (int x : nums2) {
            countsB[x] ++;
            sumB += x;
        }
        

        int ops = 0;
        while (sumA != sumB) {
            if (sumA < sumB) {
                int t = sumA;
                sumA = sumB;
                sumB = t;
                int[] t2 = countsA;
                countsA = countsB;
                countsB = t2;
            }
            // sumA > sumB;
            
            int maxMoveA = 0;
            for (int i = 6; i > 0; i --) {
                if (countsA[i] != 0) {
                    maxMoveA = i - 1; 
                    break;
                }
            }
           
            int maxMoveB = 0;
            for (int i = 1; i < 7; i ++) {
                if (countsB[i] != 0) {
                    maxMoveB = 6 - i; 
                    break;
                }
            }
                        
            int maxMove = Math.max(maxMoveA, maxMoveB);

            if (sumA - sumB > maxMove) {
                // max move
                for (int diff = maxMove; diff > 0; diff --) {
                    if (countsA[1 + diff] != 0) {
                        countsA[1 + diff] --;
                        countsA[1] ++;
                        sumA -= diff;
                        ops ++;
                        break;
                    } else if (countsB[6 - diff] != 0) {
                        countsB[6 - diff] --;
                        countsB[6] ++;
                        sumB += diff;
                        ops ++;
                        break;
                    }
                }
            
} 
            else {
                ops ++;
                break;
            } 
            
            
            
        }
        return ops;
        
        
        
        
        
    }
    
    
    
}
