class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        // Arrays.sort(nums1);
        // Arrays.sort(nums2);
        int res = 0;
        
int s1 = 0;
        int s2 = 0;
        int[] c1 = new int[7];
        int[] c2 = new int[7];
        for(int x : nums1) {
            s1 += x;
            c1[x]++;
        }
        for(int x : nums2) {
            s2 += x;
            c2[x]++;
        }
        if (s1 == s2) return 0;
        if (nums1.length * 6 < nums2.length || nums2.length * 6 < nums1.length) return -1;
        if (s1 > s2){
            int dif = s1 - s2;
            int ct = 5;
            for(int i = 1, j = 6; i < 6 && j > 1; j --, i++, ct--){
                int ak = c1[j] + c2[i];
                //System.out.println(ak+" "+dif);
                if (ct * ak < dif){
                    res += ak;
                    dif -= ct * ak;
                }else{
                    int nd = (dif - 1) / ct + 1;
                    res += nd;
                    
return res;
                }
            }
        }else{
            int dif = s2 - s1;
            int ct = 5;
            for(int i = 1, j = 6; i < 6 && j > 1; j --, i++, ct--){
                int ak = c2[j] + c1[i];
                if (ct * ak < dif){
                    res += ak;
                    dif -= ct * ak;
                }else{
                    int nd = (dif - 1) / ct + 1;
                    res += nd;
                    
return res;
                }
            }
        }
        return res;
    }
}
