class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int len1 = nums1.length, len2 = nums2.length;
        
if (len1 * 6 < len2) {
            return -1;
        }
        int sum1 = 0, sum2 = 0;
        for (int i : nums1)
            sum1 += i;
        for (int i : nums2)
            sum2 += i;
        if (sum1 > sum2) {
            return minOperations(nums2, nums1);
        }
        int d = sum2 - sum1;
        int res = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                int v1 = o1 < nums1.length ? 6 - nums1[o1] : nums2[o1 - nums1.length] - 1;
                int v2 = o2 < nums1.length ? 6 - nums1[o2] : nums2[o2 - nums1.length] - 1;
                return v2 - v1;
            }
        });
        for (int i = 0; i < len1 + len2; i++) {
            pq.add(i);
        }
        while (d > 0) {
            int cur = pq.poll();
            
if (cur >= len1) {
                cur -= len1;
                d -= nums2[cur] - 1;
            } else {
                d -= 6 - nums1[cur];
            }
            res++;
        }
        return res;
    }
}
