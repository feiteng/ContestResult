class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int l1 = nums1.length;
        int l2 = nums2.length;
        int sum1 = 0;
        int sum2 = 0;
        for(int i = 0; i < l1; i++)
            sum1 += nums1[i];
        for (int i = 0; i < l2; i++) {
            sum2 += nums2[i];
        }
        int ans = 0;
        if(sum1 > sum2){
            if(l1 > l2*6) return -1;
            int dif = sum1 - sum2;
            Arrays.sort(nums1);
            Arrays.sort(nums2);
            int index1 = l1 - 1;
            int index2 = 0;
            int change = 0;
            while(change < dif){
                if(index1 >= 0 && index2 < l2) {
                    if (nums1[index1] - 1 > 6 - nums2[index2]) {
                        change += (nums1[index1] - 1);
                        index1--;

                        ans ++;
                    } else {
                        change += (6 - nums2[index2]);
                        index2++;

                        ans ++;
                    }
                }
                else if(index1 >= 0){
                    change += (nums1[index1] - 1);
                    index1--;

                    ans ++;
                }
                else if(index2 < l2){
                    change += (6 - nums2[index2]);
                    index2++;

                    ans ++;
                }
            }
        }else{
            if(l2 > l1*6) return -1;
            int dif = sum2 - sum1;
            Arrays.sort(nums1);
            Arrays.sort(nums2);
            int index2 = l2 - 1;
            int index1 = 0;
            int change = 0;
            while(change < dif){
                if(index2 >= 0 && index1 < l1) {
                    if (nums2[index2] - 1 > 6 - nums1[index1]) {
                        change += (nums2[index2] - 1);
                        index2--;

                        ans ++;
                    } else {
                        change += (6 - nums1[index1]);
                        index1++;

                        ans ++;
                    }
                }
                else if(index2 >= 0){
                    change += (nums2[index2] - 1);
                    index2--;

                    ans ++;
                }
                else if(index1 < l1){
                    change += (6 - nums1[index1]);
                    index1++;

                    ans ++;
                }
            }
        }
        return ans;
    }
}
