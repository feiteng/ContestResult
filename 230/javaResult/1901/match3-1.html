class Solution {
    
    public int solve(int a, int b, int[] nums1, int[] nums2) {
        
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        
int count = 0;
        
        int indexA = nums1.length-1;
        int indexB = 0;
        int candidate_A = 0;
        int candidate_B = 0;
        while(a > b) {
            if (indexA >= 0) candidate_A = nums1[indexA]; 
            else candidate_A = 1;
            if (indexB < nums2.length) candidate_B = nums2[indexB]; 
            else candidate_B = 6;
            if ((candidate_A == 1) && (candidate_B == 6))  break;
            count++;
            if ((candidate_A -1) > (6 - candidate_B)) {
                indexA--;
                a += (1 - candidate_A);
            } else {
                indexB++;
                b += (6 - candidate_B);
            }
        }
        if (a > b) count = -1;
        return count;
    }
    public int minOperations(int[] nums1, int[] nums2) {
        //Arrays.sort(nums1);
        //Arrays.sort(nums2);
        
        
int sum1 = 0;
        int sum2 = 0;
        
        for(int a : nums1) sum1 += a;
        for(int a: nums2) sum2 += a;
        
        if(sum1 == sum2) return 0;
        return (sum1 > sum2) ? solve(sum1, sum2, nums1, nums2) : solve(sum2, sum1, nums2, nums1);
        
    }
}
