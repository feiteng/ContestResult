class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if (nums1.length > nums2.length * 6 || nums2.length > nums1.length * 6) {
            return -1;
        }
        int s1 = Arrays.stream(nums1).sum(), s2 = Arrays.stream(nums2).sum();
        if (s1 < s2) {
            int[] temp = nums1;
            nums1 = nums2;
            nums2 = temp;
        }
        int[] arr = new int[6];
        for (int i : nums1) {
            arr[i - 1]++;
        }
        for (int i : nums2) {
            arr[6 - i]++;
        }
        int res = 0;
         int val =Math.abs( s1 - s2);
        int sss = 5;
        while (val > 0 && sss > 0) {
            if (val > arr[sss] * sss) {
                val -= arr[sss] * sss;
                res += arr[sss];
                sss--;
            } else {
                res += (val + sss - 1) / sss;
                break;
            }
        }
        return res;
    }
}
