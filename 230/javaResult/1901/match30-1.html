class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        final int N = nums1.length;
        final int M = nums2.length;
        
        
int sum1 = arraySum(nums1);
        int sum2 = arraySum(nums2);
        
        if (sum1 < sum2) {
            return minOperations(nums2, nums1);
        }
        
        // int[] freq1 = makeFreq(nums1);
        // int[] freq2 = makeFreq(nums2);
        
        
int[] deltaPower = new int[6];
        for (int x : nums1) {
            ++deltaPower[x - 1];
        }
        for (int x : nums2) {
            ++deltaPower[6 - x];
        }
        // System.out.println(Arrays.toString(deltaPower));
        
        int moves = 0;
        int diff = sum1 - sum2;
        int p = 5;
        while (diff > 0) {
            while (p > 0 && deltaPower[p] <= 0) {
                --p;
            }
            if (p <= 0) {
                break;
            }
            // System.out.format("move = %d, diff = %d, p = %d\n", moves, diff, p);
            ++moves;
            diff -= p;
            --deltaPower[p];
        }
        if (diff > 0) {
            return -1;
        }
        return moves;

//         ArrayList<int[]> opts2 = new ArrayList<>();
//         generate(new int[M], 0, opts2);
        
//         IntArrayList[] 
    
}
    
    private static int arraySum(int[] arr) {
        int s = 0;
        for (int x : arr) {
            s += x;
        }
        return s;
    }
    
    private static int[] makeFreq(int[] arr) {
        int[] freq = new int[7];
        for (int x : arr) {
            ++freq[x];
        }
        return freq;
    }
    
    private static void generate(int[] arr, int i, ArrayList<int[]> output) {
        if (i >= arr.length) {
            output.add(Arrays.copyOf(arr, arr.length));
            return;
        }
        for (int x = 1; x <= 6; ++x) {
            arr[i] = x;
            generate(arr, i + 1, output);
        }
    }
    
    private static class IntArrayList extends ArrayList<int[]> {}
}
