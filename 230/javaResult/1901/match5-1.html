class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int sum1 = 0, sum2 = 0;
        for(int k : nums1){
            sum1 += k;
        
}
        for(int k : nums2){
            sum2 += k;
        }
        
        int ans = 0;
        if(sum1 == sum2){
            return 0;
        }
        else if(sum1 < sum2){
            int idx1 = 0, idx2 = nums2.length-1;
            int diff = sum2-sum1;
            while(diff > 0){
                if(idx1 == nums1.length && idx2 < 0){
                    return -1;
                }
                
if(idx1 < nums1.length && idx2 >= 0){
                    if(6 - nums1[idx1] >= nums2[idx2] - 1){
                        diff -= 6 - nums1[idx1];
                        ++idx1;
                    }
                    else{
                        diff -= nums2[idx2] - 1;
                        --idx2;
                    }
                }
                else if(idx1 < nums1.length){
                    diff -= 6 - nums1[idx1];
                    ++idx1;
                }
                else if(idx2 >= 0){
                    diff -= nums2[idx2] - 1;
                    --idx2;
                }
                ++ans;
            }
        }
        else{
            int idx1 = nums1.length-1, idx2 = 0;
            int diff = sum1 - sum2;
            while(diff > 0){
                if(idx1 < 0 && idx2 == nums2.length){
                    return -1;
                }
                
if(idx1 >= 0 && idx2 < nums2.length){
                    if(6 - nums2[idx2] >= nums1[idx1] - 1){
                        diff -= 6 - nums2[idx2];
                        ++idx2;
                    }
                    else{
                        diff -= nums1[idx1] - 1;
                        --idx1;
                    }
                }
                else if(idx1 >= 0){
                    diff -= nums1[idx1] - 1;
                    --idx1;
                }
                else if(idx2 < nums2.length){
                    diff -= 6 - nums2[idx2];
                    ++idx2;
                }
                ++ans;
            }
        }
        
        
        return ans;
    }
}
