class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1=0, sum2=0, count=0, diff;
        int[] large,small;
        for(int n:nums1){
            sum1+=n;
        }
        for(int n:nums2){
            sum2+=n;
        }
        if(sum1>=sum2){
            large=nums1;
            small=nums2;
        }else{
            large=nums2;
            small=nums1;
        }
        diff=Math.abs(sum1-sum2);
        PriorityQueue<Integer> lQueue=new PriorityQueue<>((x,y)->y-x), sQueue=new PriorityQueue<>((x,y)->x-y);
        for(int n:large){
             lQueue.add(n);
        }
        for(int n:small){
             sQueue.add(n);
        }
        
        while(diff>0&&(!lQueue.isEmpty()||!sQueue.isEmpty())){
            if(lQueue.isEmpty()){
                diff-=6-sQueue.poll();
                count++;
            }else if(sQueue.isEmpty()){
                diff-=lQueue.poll()-1;
                count++;
            }else{
                if(6-sQueue.peek()>lQueue.peek()-1){
                    diff-=6-sQueue.poll();
                    count++;
                }else{
                    diff-=lQueue.poll()-1;
                    count++;
                }
            }
        }
        
        return diff>0?-1:count;
    }
}
