class Solution {
    
public int minOperations(int[] nums1, int[] nums2) {
        // sort
        int sum1 = 0;
        int sum2 = 0;
        for (int i : nums1) { sum1 += i;}
        for (int i : nums2) { sum2 += i;}

        int[] flip = sum1 > sum2 ? nums1 : nums2;
        for (int i = 0; i < flip.length; i++) {
            flip[i] = 7 - flip[i];
        }
        
        
int[] arr = new int[nums1.length + nums2.length];
        int z = 0;
        for (int j : nums1) { arr[z++] = j; }
        for (int j : nums2) { arr[z++] = j; }
        
        Arrays.sort(arr);
        int target = Math.abs(sum1 - sum2);
        
        int ret = 0;
        for (int i = 0; i < arr.length && target > 0; i++) {
            if (arr[i] == 6) continue;
            else {
                ret++;
                target -= (6 - arr[i]);
            }
        }
        return target <= 0 ? ret : -1;
    }
}
