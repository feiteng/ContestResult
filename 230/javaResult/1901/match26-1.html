class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int[] array1 = new int[7], array2 = new int[7];
        int sum1 = 0, sum2 = 0;
        for (int i : nums1) {
            array1[i]++;
            sum1 += i;
        }
        for (int i : nums2) {
            array2[i]++;
            sum2 += i;
        }
        if (sum1 == sum2) return 0;
        if (sum1 < sum2) {
            int temp = sum1;
            sum1 = sum2;
            sum2 = temp;
            int[] array = array1;
            array1 = array2;
            array2 = array;
        }
        
        for (int i = 1; i <= 6; i++) {
            array1[i] += array2[7 - i];
        }
        int res = 0;
        int dif = sum1 - sum2;
        for (int i = 6; i >= 1; i--) {
            if (array1[i] * (i - 1) >= dif) {
                int temp = dif / (i - 1);
                if (temp * (i - 1) < dif) {
                    temp++;
                } 
                res += temp;
                return res;
            
} else {
                dif -= array1[i] * (i - 1);
                res += array1[i];
            }
        }
        return -1;
    }
}
