class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        int sum1 = 0;
        int sum2 = 0;
        int[] arr1Smaller = new int[6];
        int[] arr2Smaller = new int[6];
        
        for (int n1 : nums1) {
            sum1 += n1;
            arr1Smaller[6 - n1]++;
            arr2Smaller[n1 - 1]++;
        }
        
        for (int n2 : nums2) {
            sum2 += n2;
            arr2Smaller[6 - n2]++;
            arr1Smaller[n2 - 1]++;
        }
        
        if (sum1 == sum2)
            return 0;
        
        int ops = 0;
        
        if (sum1 < sum2) {
            int diff = sum2 - sum1;
            for (int i = 5; i >= 0; i--) {
                if (diff > i * arr1Smaller[i]) {
                    ops += arr1Smaller[i];
                    diff -= i * arr1Smaller[i];
                }
                else {
                    ops += diff / i;
                    if (diff % i != 0) {
                        ops++;
                    }
                    return ops;
                }
            }
        }
        else {
            int diff = sum1 - sum2;
            for (int i = 5; i >= 0; i--) {
                if (diff > i * arr2Smaller[i]) {
                    ops += arr2Smaller[i];
                    diff -= i * arr2Smaller[i];
                }
                else {
                    ops += diff / i;
                    if (diff % i != 0) {
                        ops++;
                    }
                    return ops;
                }
            }
        }
        return -1;
    }
}
