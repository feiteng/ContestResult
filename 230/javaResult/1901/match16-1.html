class Solution {
    public int minOperations(int[] nums1, int[] nums2) {
        if(nums1.length * 6 < nums2.length || nums2.length * 6 < nums1.length){return -1;}
        int sum1 = 0, sum2 = 0;
        int[] count1 = new int[7], count2 = new int[7];
        
for(int num : nums1){
            sum1 += num;
            count1[num]++;
        }
        for(int num : nums2){
            sum2 += num;
            count2[num]++;
        }
        
if(sum1 > sum2){
            return helper(count1,count2,sum1-sum2);
        }else{
            return helper(count2,count1,sum2-sum1);
        }
    }
    
    //count1>count2
    public int helper(int[] count1,int[] count2,int diff){
        int res = 0;
        for(int i = 6;i > 1;i--){
            //change count1[i] and count2[7-i]
            if(count1[i] != 0){
                if((i-1)*count1[i] >= diff){
                    return res + (diff%(i-1)==0?diff/(i-1):(diff/(i-1)) + 1);
                }else{
                    res += count1[i];
                    diff -= (i-1)*count1[i];
                }
            }
            if(count2[7-i] != 0){
                if((i-1)*count2[7-i] >= diff){
                    return res + (diff%(i-1)==0?diff/(i-1) : (diff/(i-1) )+ 1);
                }else{
                    res += count2[7-i];
                    diff -= (i-1)*count2[7-i];
                }
            }
        }
        return -1;
    }
}
