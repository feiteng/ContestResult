class Solution {
  public int minOperations(int[] nums1, int[] nums2) {
    int sum1 = sum(nums1);
    int sum2 = sum(nums2);
    if (sum1 > sum2) {
      return minOperations(nums2, nums1);
    }
    int steps = 0;
    Arrays.sort(nums1);
    Arrays.sort(nums2);
    for (int i = 0, j = nums2.length - 1; sum1 < sum2 && (i < nums1.length || j >= 0); ++steps) {
      if (i == nums1.length) {
        sum2 -= (nums2[j--] - 1);
      } else if (j < 0) {
        sum1 += (6 - nums1[i++]);
      } else {
        if (7 > nums1[i] + nums2[j]) {
          sum1 += (6 - nums1[i++]);
        } else {
          sum2 -=(nums2[j--] - 1);
        }
      }
    }
    return sum1 >= sum2 ? steps : -1;
  }

  private int sum(int[] nums) {
    int sum = 0;
    for (int num : nums) {
      sum += num;
    }
    return sum;
  }
}
