class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        
        int n = cars.length;
        double[] ans = new double[n];
        double[] pos = new double[n];//crash pos, with the last, in actual
        int cap = 10000000, size = 0;
        int[] last = new int[n];
        
for(int i = n -1; i >= 0; i--){
            int psi = cars[i][0];
            int spi = cars[i][1];
            if(cap >= spi){
                ans[i] = -1;
                //clear the crash "history"
                size =0;
                last[size++] = i;
                cap = spi;
            } else {
                //spi > cap, crash


                                 
                    int lo = 0, hi = size - 1;
                    int mid = (lo + hi + 1)/2;
                    while(lo < hi){
                        int index = last[mid];
                        double tcrash = cal(psi,cars[index][0], spi, cars[index][1]);
                        double texist = ans[index];
                        if(tcrash < 0 || tcrash >=texist){
                            //not the one
                            hi = mid - 1;
                        } else {
                            lo = mid;
                        }
                        mid = (lo + hi + 1)/2;
                    }
                    int index = last[mid];
                    double tcrash = cal(psi,cars[index][0], spi, cars[index][1]);
                    //remove all cars after mid
                    size = mid + 1;
                    last[size++] = i;
                    ans[i] = tcrash;
                

            }
        }
        return ans;
    }
    
    double cal(int px, int py, int vx, int vy){

        double res = ((double)(py - px))/((double)(vx - vy));
        //System.out.println(px+" "+vx+" & "+py+" "+vy+ ": "+res);
        return res;
    }
}
