class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int N = cars.length;
        double[] ans = new double[N];
        Stack<Car> stack = new Stack();
        for (int i = N - 1; i >= 0; i--) {
            int p = cars[i][0], v = cars[i][1];
            while (!stack.isEmpty()) {
                Car next = stack.pop();
                if (next.v >= v) {
                    continue;
                } else if (stack.isEmpty()) {
                    stack.push(next);
                    break;
                } else {
                    Car nnext = stack.peek();
                    double t1 = (double) (next.p - p) / (double) (v - next.v);
                    double t2 = (double) (nnext.p - p) / (double) (v - nnext.v);
                    if (t2 <= t1) {
                        continue;
                    } else {
                        stack.push(next);
                        break;
                    }
                }
            }
            if (stack.isEmpty()) {
                ans[i] = -1.0;
            } else {
                Car next = stack.peek();
                ans[i] = (double) (next.p - p) / (double) (v - next.v);
            }
            stack.push(new Car(p, v));
            // System.out.println(stack.toString());
        }
        return ans;
    }
    
    class Car {
        int p, v;
        
        Car(int p, int v) {
            this.p = p;
            this.v = v;
        }
        
//         @Override
//         public String toString() {
//             return "[" + p + ", " + v + "]";
//         }
    }
}
