class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double[] res = new double[n];
        res[n - 1] = -1;
        int curSlow = cars[n - 1][1];
        for (int curCar = n - 2; curCar >= 0; curCar--)
        {
            int targetCar = curCar + 1;
            int curLoc = cars[curCar][0];
            int curSpeed = cars[curCar][1];
            if(curSpeed <= curSlow)
            {
                res[curCar] = -1;
                curSlow = Math.min(curSlow, curSpeed);
                continue;
            }
            while (true)
            {
                if (targetCar >= n)
                {
                    res[curCar] = -1;
                    break;
                }

                int targetSpeed = cars[targetCar][1];
                int targetLoc = cars[targetCar][0];
                double timeLimit = res[targetCar] == -1 ? 99999999 : res[targetCar];
                
if (curSpeed > targetSpeed)
                {
                    double time = (double)(targetLoc - curLoc) / (double)(curSpeed - targetSpeed);
                    if (time <= timeLimit)
                    { res[curCar] = time; break; }
                }
                targetCar++;
            }
            curSlow = Math.min(curSlow, curSpeed);
        }

        return res;
    }
}
