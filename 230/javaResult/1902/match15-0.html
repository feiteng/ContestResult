class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        // res[i]  res[i + 1]
        // cars[i][1] < min(cars[k][1]) (k > i): res[i] = -1
        // cars[i][1] < cars[i + 1][1]: res[i] = (cars[k][0] - cars[i][0]) / (cars[i][1] - cars[k][1]) k是右边第一个速度小于i的车
        // 首先找到i右边第一个 速度小于i 而且 自然情况i追上k的时间t <= res[k] || res[k] == -1 res[i] = t
        // 如果t > res[k], res[i] = 
        double[] res = new double[cars.length];
        res[res.length - 1] = -1.0;
        int minSpeed = cars[cars.length - 1][1];
        for (int i = res.length - 2; i >= 0; i--) {
            if (cars[i][1] < minSpeed) {
                res[i] = -1.0;
            } else {
                int j = i + 1;
                while (j < res.length) {
                    if (cars[j][1] < cars[i][1]) {
                        double t = (cars[j][0] - cars[i][0]) * 1.0 / (cars[i][1] - cars[j][1]);
                        if (t <= res[j] || res[j] == -1) {
                            res[i] = t;
                            break;
                        }
                    }
                    j++;
                }
                if (j == res.length) res[i] = -1.0;
            }
            minSpeed = Math.min(minSpeed, cars[i][1]);
        }
        return res;
    }
}
