class Solution {
    
    double times(int[] a, int[] b)
    {
        return 1.0 * (b[0]-a[0]) / (a[1]-b[1]);
    }
    
    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double[] ans = new double[n];
        

        
        Stack<int[]> stack = new Stack<>();
        for (int i = n-1; i >= 0 ; i-- )
        {
            while(!stack.isEmpty() && stack.peek()[1] >= cars[i][1]) stack.pop();
            if (stack.isEmpty())
            {
                ans[i] = -1;
                stack.add(cars[i]);
            }
            else
            {
                while(stack.size() >= 2)
                {
                    int[] cur = stack.pop();
                    if (times(cars[i], cur) >= times(cars[i], stack.peek()))
                    {
                        
                    }
                    else
                    {
                        stack.add(cur);
                        break;
                    }
                }
                ans[i] = times(cars[i], 
stack.peek());
                stack.add(cars[i]);
            }
        }
        return ans;
    }
}
