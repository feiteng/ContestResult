class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        Deque<Integer> s = new ArrayDeque<>();
        double[] res = new double[cars.length];
        for(int i = cars.length - 1; i >= 0; i--) {
            while(!s.isEmpty() && cars[i][1] <= cars[s.peek()][1]) s.pop();
            while(!s.isEmpty()) {
                int prevIdx = s.peek();
                double curTime = (cars[prevIdx][0] - cars[i][0]) * 1.0 / (cars[i][1] - cars[prevIdx][1]);
                if(res[prevIdx] != -1.0 && curTime > res[prevIdx]) s.pop();
                else break;
            }
            if(s.isEmpty()) res[i] = -1.0;
            else {
                res[i] = (cars[s.peek()][0] - cars[i][0]) * 1.0 / (cars[i][1] - cars[s.peek()][1]);
            }
            s.push(i);
        }
        return res;
    }
}
