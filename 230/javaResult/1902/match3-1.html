class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int size = cars.length;
        double[] res = new double[size];
        res[size - 1] = -1.0;
        int upperBound = size - 1;
        for(int i = size - 2; i >= 0; --i) {
            res[i] = -1;
            for(int j = i + 1; j <= upperBound; ++j) {
                int speed = cars[j][1];
                if(speed < cars[i][1]) {
                    double time = (cars[j][0] - cars[i][0]) * 1.0 / (cars[i][1] - speed);
                    if(res[j] == -1) {
                        res[i] = time;
                        break;
                    } else if(res[j] < time) continue;
                    else {
                        res[i] = time;
                        break;
                    }
                }
            }
            if(res[i] == -1) upperBound = i;
        }
        return res;
    }
}
