class Solution {
    static double INF = 1.0*Long.MAX_VALUE/3;
    public double[] getCollisionTimes(int[][] arr) {
        int N = arr.length;
        double[] res = new double[N];
        res[N-1] = -1.0;
        Stack<Integer> stack = new Stack<Integer>();
        stack.push(N-1);
        int slowest = arr[N-1][1];
        for(int i=N-2; i >= 0; i--)
        {
            if(slowest >= arr[i][1])
            {
                res[i] = -1.0;
                slowest = arr[i][1];
                stack = new Stack<Integer>();
            }
            else
            {
                int choice = -1;
                while(stack.size() > 0)
                {
                    choice = stack.pop();
                    if(stack.size() == 0)
                        break;
                    long num1 = arr[choice][0]-arr[i][0];
                    long denom1 = arr[i][1]-arr[choice][1];
                    int c = stack.peek();
                    long num2 = arr[c][0]-arr[i][0];
                    long denom2 = arr[i][1]-arr[c][1];
                    if(num1*denom2 < num2*denom1)
                        break;
                }
                stack.push(choice);
                long num1 = arr[choice][0]-arr[i][0];
                long denom1 = arr[i][1]-arr[choice][1];
                res[i] = 1.0*num1/denom1;
            
}
            stack.push(i);
        }
        return res;
