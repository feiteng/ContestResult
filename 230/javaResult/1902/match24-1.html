class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int N = cars.length;
        double[] res = new double[N];
        double currSpeed = Integer.MAX_VALUE;
        for(int i = N - 1; i >= 0; i--) {
            double p = cars[i][0], s = cars[i][1];
            if(s <= currSpeed) {
                res[i] = -1;
                currSpeed = s;
                continue;
            }
            
            for(int j = i+1; j < N; j++) {
                double dis = cars[j][0] - p;
                // System.out.println(res[j] + " " + s + "  " + cars[j][0]);
                if(res[j] < 0) {
                    res[i] = dis / (s - cars[j][1]);
                    break;
                } else {
                    if(s <= cars[j][1]) continue;
                    
double chase = dis / (s - cars[j][1]);
                    if(chase <= res[j]){
                        res[i] = chase;
                        break;
                    }
                }
            }
        }
        return res;
    }
}
