class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        if (cars == null || cars.length == 0) {
            return new double[]{};
        }
        
        for (int i = 0; i < cars.length; i++) {
            System.out.print("[" + cars[i][0] + "," + cars[i][1] + "]");
        }
        
        double[] result = new double[cars.length];
        result[cars.length - 1] = -1;
        for (int i = cars.length - 2; i >= 0; i--) {
            
double minTime = Integer.MAX_VALUE;
            for (int j = i + 1; j < cars.length; j++) {
                if (cars[i][1] > cars[j][1]) {
                    double currentTime = (cars[j][0] - cars[i][0]) / (double) (cars[i][1] - cars[j][1]);
                    if (currentTime < minTime) {
                        minTime = currentTime;
                    }
                }
                
                if (minTime <= result[j] || result[j] == (double) (-1)) {
                    break;
                }
            }
            if (minTime == (double) Integer.MAX_VALUE) {
                result[i] = -1;
            } else {
                result[i] = minTime;
            }
        }
        return result;
    }
}
