import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;

class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        double[] ans = new double[cars.length];
        ans[ans.length - 1] = -1;
        LinkedList<Speed> ss = new LinkedList<>();
        ss.add(new Speed(cars[cars.length - 1][1], 0, cars[cars.length - 1][0]));
        for (int i = cars.length - 1; i > 0; i--) {
            int[] curr = cars[i - 1];
            if (curr[1] <= ss.get(ss.size() - 1).speed) {
                ss = new LinkedList<>();
                ss.add(new Speed(curr[1], 0, curr[0]));
                ans[i - 1] = -1;
                continue;
            }

            double sub = cars[i][0] - curr[0];
            double time = 0;
            // System.out.println(ss);
            while (sub > 0.000001) {
                if (ss.size() > 1) {
                    double len = (curr[1] - ss.get(0).speed) * (ss.get(0 + 1).time - ss.get(0).time);
                    // System.out.println(len);
                    if (len < sub) {
                        time = ss.get(0 + 1).time;
                        sub -= len;
                        ss.pollFirst();
                    } else {
                        time += sub / (double)(curr[1] - ss.get(0).speed);
                        break;
                    }
                } else {
                    time += sub / (double)(curr[1] - ss.get(0).speed);
                    break;
                }
            }
            // System.out.println(time);
            ss.add(0, new Speed(ss.pollFirst().speed, time, curr[0]));
            ss.add(0, new Speed(curr[1], 0, curr[0]));
            ans[i - 1] = time;
        
}
        return ans;
    }

    class Speed {
        int speed;
        double time;
        double pos;
        public Speed() { }
        
public Speed(int speed, double time, int pos) {
            this.speed = speed;
            this.time = time;
            this.pos = pos;
        }
    }
}
