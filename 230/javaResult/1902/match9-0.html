class Solution {
    public double[] getCollisionTimes(int[][] cs) {
        int n = cs.length;
        double[] res = new double[n];
        Stack<Integer> stack = new Stack<>();
        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && cs[stack.peek()][1] >= cs[i][1]) {
                stack.pop();
            }
            
            if (stack.isEmpty()) {
                res[i] = -1;
            } else {
                for (int x = stack.size() - 1; x >= 0; x--) {
                    int idx = stack.get(x);
                    double time = ((double)cs[idx][0] - cs[i][0]) / (cs[i][1] - cs[idx][1]);
                    if (time <= res[idx] || res[idx] < 0) {
                        res[i] = time;
                        break;
                    }
                }
            }
            stack.push(i);
        }
        
        return res;
    }
}
