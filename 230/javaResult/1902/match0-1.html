class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double [] ans = new double [n];
        ans[n-1] = -1.0;
        int xprev = cars[n-1][0];
        int vprev = cars[n-1][1];
        
        LinkedList<Integer> Vs = new LinkedList<Integer>();
        LinkedList<Double> Ts = new LinkedList<Double>();
        Vs.add(vprev);
        Ts.add(-1.0);
        int [] xat = new int [1000001];
        xat[vprev] = xprev;
        
        for (int i = n-2; i>=0; --i) {
            int x = cars[i][0];
            int v = cars[i][1];
            xat[v] = x;
            
            while (!Vs.isEmpty() && v <= Vs.peek()) {
                Vs.pop();
                Ts.pop();
            }
            if (Vs.isEmpty()) {
                Vs.push(v);
                Ts.push(-1.0);
                ans[i] = -1.0;
                continue;
            }
            while (!Ts.isEmpty()) {
                double t2 = Ts.peek();
                int v2 = Vs.peek();
                double x2 = xat[v2];
                
                double t0 = (x2 - x) / (v - (double)v2 + 0.0);
                if (t0 <= t2 || t2 == -1.0) {
                    Vs.push(v);
                    Ts.push(t0);
                    ans[i] = t0;
                    break;
                } else {
                    Vs.pop();
                    Ts.pop();
                
}
            }
        }
        return ans;
    }
}
