class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double[] res = new double[n];
        res[n-1] = -1.0;
        int minSpd = cars[n-1][1];
        for (int i = n-2; i > -1; i--){
            int curLoc = cars[i][0], curSpd = cars[i][1];
            if (curSpd <= minSpd){
                res[i] = -1.0;
                minSpd = curSpd;
                continue;
            }
            
            int j = i + 1;
            // System.out.println(curLoc+" "+i+" "+j);
            while (j < n && curSpd > minSpd){
                int nxtLoc = cars[j][0], nxtSpd = cars[j][1];
                // System.out.println(curLoc+" "+nxtLoc);
                if ((double)res[j] == -1.0 || (double)(nxtLoc - curLoc) <= (double)res[j]*(curSpd - nxtSpd)){
                    // System.out.println(curLoc+" "+nxtLoc);
                    res[i] = (double)(nxtLoc - curLoc)/(double)(curSpd - nxtSpd);
                    break;
                }else{
                    j++;
                }
            }
        }
        return res;
    }
}
