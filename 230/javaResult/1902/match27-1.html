class Solution {


    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double[] res = new double[cars.length];
      
        Deque<int[]> cc = new LinkedList<>();
        Deque<Double> ct = new LinkedList<>();

        cc.addFirst(cars[n-1]);
        ct.addFirst(MAX);

        res[n-1] = -1;

        for(int i = n-2; i >= 0; --i){
            int[] cur = cars[i];

            double time = 0.0;
            while(cc.size() > 0){
                 time = isQuick(cur, cc.peekFirst());
                if(time > ct.peekFirst()){
                    cc.pollFirst();
                    ct.pollFirst();
                }else{
                    break;
                }
            }
            
            res[i] = time;
            if(time == MAX){
                res[i] = -1;
            }
            
            cc.addFirst(cur);
            ct.addFirst(time);

        }
        return res;

    }
    double MAX= 10000000;

    double isQuick(int[] cur, int[] next){

        
        if(cur[1] <= next[1]){
            return MAX;
        }
        return   (next[0] - cur[0])*1.0/(cur[1] - next[1]);

    }

}
