class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        if (cars == null || cars.length == 0) return new double[0];
        int n = cars.length;
        double[] res = new double[n];
        Arrays.fill(res, -1.0);
        
Stack<Integer> stack = new Stack<>();
        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && cars[i][1] <= cars[stack.peek()][1]) {
                stack.pop();
            }
            while (!stack.isEmpty()) {
                int dis = cars[stack.peek()][0] - cars[i][0];
                int spd = cars[i][1] - cars[stack.peek()][1];
                double time = (double) dis / spd;
                if (time < res[stack.peek()] || res[stack.peek()] == -1) {
                    res[i] = time;
                    break;
                }
                stack.pop();
            }
            stack.push(i);
        }
        return res;
    }
}
