class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int n = cars.length;
        double[] ans = new double[n];
        
        LinkedList<Integer> stk = new LinkedList<Integer>();
        int now;
        double tm;
        for (int i = n - 1; i >= 0; --i) {
            ans[i] = -1.0;
            while (!stk.isEmpty()) {
                now = stk.getFirst();
                if (cars[i][1] <= cars[now][1]) {
                    stk.removeFirst();
                    continue;
                }
                else {
                    tm = (cars[now][0] - cars[i][0]) / (double)(cars[i][1] - cars[now][1]);
                    if (ans[now] >= 0 && tm > ans[now]) stk.removeFirst();
                    else {
                        ans[i] = tm;
                        stk.addFirst(i);
                        break;
                    }
                    //if (ans[i] < 0 || ans[i] > tm) ans[i] = tm;
                }
            }
            if (ans[i] < 0) {
                stk.clear();
                stk.add(i);
            }
        }
        
        return ans;
    }
}
