class Solution {
    public double[] getCollisionTimes(int[][] cars) {
        int len = cars.length;
        double[] res = new double[len];
        Deque<double[]> qu = new LinkedList<>();
        for(int i = len - 1; i > -1; i--){
            double cur = 1000000;
            double pos = cars[i][0], speed = cars[i][1];
            double a = pos, b = speed;
            for(double[] it : qu){
                if(it[1] < speed){
                    double temp = (it[0] - pos) / (speed - it[1]);
                    if(cur < 0 || cur > temp) {
                        cur = temp;
                    }
                    a = it[0];
                    b = it[1];
                }
            }
            res[i] = cur == 1000000 ? -1 : cur;
            while(!qu.isEmpty() && (cur >= qu.getLast()[2] || speed <= qu.getLast()[1])){
                qu.pollLast();
            }
            qu.offerLast(new double[]{pos, speed, cur});
        }
        return res;
    }
}
