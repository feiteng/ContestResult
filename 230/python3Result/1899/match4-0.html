class Solution:
    def countMatches(self, items: List[List[str]], ruleKey: str, ruleValue: str) -> int:
        rule_map = {'type': 0, 'color':1, 'name': 2}
        idx = rule_map[ruleKey]
        
        cnt = 0
        
        for item in items:
            if item[idx] == ruleValue:
                cnt += 1
                
        return cnt
