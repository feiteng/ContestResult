class Solution:
    def countMatches(self, items: List[List[str]], ruleKey: str, ruleValue: str) -> int:
        cnt = 0
        for i, j, k in items:
            if ruleKey == "type":
                cnt += 1 if i == ruleValue else 0
            elif ruleKey == "color":
                cnt += 1 if j == ruleValue else 0
            else:
                cnt += 1 if k == ruleValue else 0
                
        return cnt
