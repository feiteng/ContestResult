class Solution:
    def countMatches(self, items: List[List[str]], rk: str, rv: str) -> int:
        ret = 0
        for t, c, n in items:
            if rk == "color" and c == rv: ret += 1
            elif rk == "type" and t == rv: ret += 1
            elif rk == "name" and n == rv: ret += 1
        return ret
