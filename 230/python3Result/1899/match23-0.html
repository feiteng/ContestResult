class Solution:
    def countMatches(self, items: List[List[str]], ruleKey: str, ruleValue: str) -> int:
        rst = 0
        for t, c, n in items:
            if ruleKey == 'type' and ruleValue == t:
                rst += 1
            elif ruleKey == 'color' and ruleValue == c:
                rst += 1
            elif ruleKey == 'name' and ruleValue == n:
                rst += 1
        return rst
