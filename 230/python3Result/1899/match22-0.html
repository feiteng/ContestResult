class Solution:
    def countMatches(self, items: List[List[str]], ruleKey: str, ruleValue: str) -> int:
        cnt = 0
        for t,c,n in items:
            if ruleKey == "type" and t == ruleValue:
                cnt += 1
            elif ruleKey == "color" and c == ruleValue:
                cnt += 1
            elif ruleKey == "name" and n == ruleValue:
                cnt += 1
        return cnt
