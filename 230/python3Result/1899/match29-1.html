class Solution:
    def countMatches(self, items: List[List[str]], ruleKey: str, ruleValue: str) -> int:
        # any rule lol
        
        ret = 0
        
        for typ,col,nam in items:
            if ruleKey == "type" and typ == ruleValue:
                ret += 1
            elif ruleKey == "color" and col == ruleValue:
                ret += 1
            elif ruleKey == "name" and nam == ruleValue:
                ret += 1
        
        return ret
