class Solution(object):
    def minOperations(self, nums1, nums2):
        nums1.sort()
        nums2.sort()
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1

        if len(nums1)*6 < len(nums2)*1:
            return -1

        l1 = 0
        r2 = len(nums2)-1
        cnt = 0
        target = s2-s1
        while target:
            if l1 < len(nums1) and (6-nums1[l1]) > (nums2[r2]-1):
                target -= min((6-nums1[l1]), target)
                l1 += 1
            else:
                target -= min((nums2[r2]-1), target)
                r2 -= 1
            cnt += 1
        return cnt
