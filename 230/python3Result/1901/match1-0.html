class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        
        if max(len(nums1), len(nums2)) > 6 * min(len(nums1), len(nums2)):
            return -1

        s1, s2 = sum(nums1), sum(nums2)
        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1

        counts = [0] * 6
        for n in nums1:
            counts[6 - n] += 1
        for n in nums2:
            counts[n - 1] += 1

        diff = s2 - s1
        changes = 0
        for x in reversed(range(6)):
            used = min(counts[x], ceil(diff / x))
            changes += used
            diff -= used * x
            if diff <= 0:
                break

        return changes
