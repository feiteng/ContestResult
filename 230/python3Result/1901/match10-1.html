class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        m, n = len(nums1), len(nums2)
        if n > 6 * m or m > 6 * n:
            return -1
        sum1, sum2 = sum(nums1), sum(nums2)
        if sum1 == sum2:
            return 0
        if sum1 > sum2:
            nums1, nums2 = nums2, nums1
        diff = abs(sum1 - sum2)
        count = [0] * 6
        for num in nums1:
            count[6 - num] += 1
        for num in nums2:
            count[num - 1] += 1
        res = 0
        for i in range(5, 0, -1):
            if count[i] * i >= diff:
                return res + (diff + i - 1) // i
            res += count[i]
            diff -= count[i] * i
        return -1
