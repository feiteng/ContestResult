class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if max(len(nums1), len(nums2)) > 6*min(len(nums1), len(nums2)):
            return -1

        sum1 = sum(nums1)
        sum2 = sum(nums2)
        if sum1 > sum2:
            sum1, sum2 = sum2, sum1
            nums1, nums2 = nums2, nums1

        n1 = Counter(nums1)
        n2 = Counter(nums2)
        count = 0
        while sum1 < sum2:
            n1c = 6
            for i in range(1, 7):
                if n1[i]:
                    n1c = i
                    break
            n2c = 1
            for i in range(6, 0, -1):
                if n2[i]:
                    n2c = i
                    break
            if 6-n1c > n2c - 1:
                n1[n1c] -= 1
                n1[6] += 1
                sum1 += 6-n1c
            else:
                n2[n2c] -= 1
                n2[1] += 1
                sum1 += n2c-1
            count += 1
        return count
