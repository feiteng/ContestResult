class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        m, n = len(nums1), len(nums2)
        
        if 6*m < n or 6*n < m: return -1
        
        s1, s2 = sum(nums1), sum(nums2)
        if s1 == s2: return 0
        if s1 > s2:
            nums1, nums2 = nums2, nums1
        
        delta = abs(s2 - s1)
        heap = []
        for e in nums1:
            if e < 6:
                heapq.heappush(heap, e-6)
        for e in nums2:
            if e > 1:
                heapq.heappush(heap, 1-e)

        res = 0
        while heap and delta > 0:
            cur = heapq.heappop(heap)
            if abs(cur) < delta:
                delta += cur
            else:
                delta = 0
            res += 1
        return res
