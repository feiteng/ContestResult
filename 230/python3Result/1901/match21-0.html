class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if len(nums1) > 6 * len(nums2) or len(nums2) > 6 * len(nums1):
            return -1
        
        s1, s2 = sum(nums1), sum(nums2)
        nums1.sort()
        nums2.sort()
        
        if s1 == s2:
            return 0
        elif s1 < s2:
            return self.find(nums1, nums2[::-1], s2 - s1)
        else:
            return self.find(nums2, nums1[::-1], s1 - s2)
        
    def find(self, nums1, nums2, diff):
        idx1, idx2, l1, l2 = 0, 0, len(nums1), len(nums2)
        res = 0
        
        while idx1 < l1 and idx2 < l2:
            n1, n2 = nums1[idx1], nums2[idx2]
            if (6 - n1) > (n2 - 1):
                diff -= (6 - n1)
                idx1 += 1
            else:
                diff -= (n2 - 1)
                idx2 += 1
            res += 1
            if diff <= 0:
                return res
        
        while idx1 < l1 and diff > 0:
            diff -= (6 - nums1[idx1])
            idx1 += 1
            res += 1
        
        while idx2 < l2 and diff > 0:
            diff -= (nums2[idx2] - 1)
            idx2 += 1
            res += 1
        return res
