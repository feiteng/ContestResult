class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if max(len(nums1), len(nums2)) >  6*min(len(nums1), len(nums2)): return -1
        
        s1, s2 = sum(nums1), sum(nums2)
        if s1 > s2: 
            s1, s2 = s2, s1
            nums1, nums2 = nums2, nums1
        nums1.sort()
        nums2.sort()
        
        out = 0
        id1 = 0
        id2 = len(nums2)-1
        while s1 < s2:
            out += 1
            if id1 == len(nums1):
                s2 -= nums2[id2]-1
                id2 -= 1
            elif id2==-1:
                s1 += 6-nums1[id1]
                id1 += 1
                
            elif 6-nums1[id1] > nums2[id2]-1:
                s1 += 6-nums1[id1]
                id1 += 1
            else:
                s2 -= nums2[id2]-1
                id2 -= 1

        return out
