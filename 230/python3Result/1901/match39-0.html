class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sm1 = sum(nums1)
        sm2 = sum(nums2)
        dec = sm2 - sm1
        if dec < 0:
            nums1, nums2 = nums2, nums1
            dec = -dec
        nums1.sort()
        nums2.sort()
        p1, p2 = 0, len(nums2) - 1
        n1, n2 = len(nums1), len(nums2)
        ans = 0
        while dec > 0 and (p1 < n1 or p2 >= 0):
            x1, x2 = 6 - nums1[p1] if p1 < n1 else 0, nums2[p2] - 1 if p2 >= 0 else 0
            x = max(x1, x2)
            if x == 0:
                break
            if x1 > x2 or p2 < 0:
                p1 += 1
                dec -= x1
            elif x1 < x2 or p1 >= n1:
                p2 -= 1
                dec -= x2
            else:
                p1 += 1
                dec -= x1
            ans += 1
        if dec > 0:
            return -1
        else :
            return ans
