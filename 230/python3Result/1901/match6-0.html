class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        
        if s1 == s2:
            return 0
        
        if s1 > s2:
            return self.minOperations(nums2, nums1)
        
        a = []
        for num in nums1:
            a.append(6 - num)
        
        for num in nums2:
            a.append(num - 1)
        
        a.sort()
        
        ans = 0
        diff = s2 - s1
        
        while a and diff > 0:
            diff -= a.pop()
            ans += 1
        
        if diff > 0:
            return -1
        return ans
