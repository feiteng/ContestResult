from itertools import chain
from collections import Counter

class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        if sum1 > sum2:
            nums1, nums2 = nums2, nums1
            sum1, sum2 = sum2, sum1
        diff = sum2 - sum1
        if diff == 0:
            return 0
        
        cnt1 = Counter(nums1)
        cnt2 = Counter(nums2)
        
        ops = 0
        
        for d, c in sorted(chain(map(lambda e: (6 - e[0], e[1]), cnt1.items()), map(lambda e: (e[0] - 1, e[1]), cnt2.items())), key=lambda e: e[0], reverse=True):
            # print(diff, (d, c))
            for _ in range(c):
                if diff > d:
                    diff -= d
                    ops += 1
                else:
                    diff = 0
                    ops += 1
                    break
            if diff == 0:
                break
        if diff == 0:
            return ops
        else:
            return -1
