class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        nums1.sort()
        nums2.sort()
        ans = 0
        s1, s2 = sum(nums1), sum(nums2)
        if s1 == s2:
            return ans
        if s1 > s2:
            s1, s2, nums1, nums2 = s2, s1, nums2, nums1
        if len(nums1) * 6 < len(nums2):
            return -1
        i, j = 0, len(nums2) - 1
        
while s1 != s2:
            if i >= len(nums1) or (0 <= j and (6 - nums1[i]) < (nums2[j] - 1)):
                s2 -= min(s2 - s1, nums2[j] - 1)
                j -= 1
            else:
                s1 += min(s2 - s1, 6 - nums1[i])
                i += 1
            ans += 1
        return ans
