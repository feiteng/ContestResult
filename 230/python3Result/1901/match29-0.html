class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1,s2=sum(nums1),sum(nums2)
        if s1==s2:
            return 0
        if s1>s2:
            nums1,nums2=nums2,nums1
            s1,s2=s2,s1
        
        c1,c2=Counter(nums1),Counter(nums2)
        diff=s2-s1
        # print(diff)
        count=0
        for i in range(5,0,-1):
            if i*(c1[6-i]+c2[i+1])>=diff:
                return count+diff//i+(diff%i>0)
            else:
                diff-=i*(c1[6-i]+c2[i+1])
                count+=c1[6-i]+c2[i+1]
        return -1
