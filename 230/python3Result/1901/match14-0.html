class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        diff = sum(nums1) - sum(nums2)
        if diff == 0:
            return 0 
        
        if diff > 0:
            diff = -diff
            nums1, nums2 = nums2, nums1
            
        takes = []      
        for num in nums1: takes.append(6-num)
        for num in nums2: takes.append(num-1)
        takes = [take for take in takes if take != 0]
        # if len(takes) == 0: return -1
        
        takes.sort()
        
        prev_diff = diff
        step = 0
        while takes:
            prev_diff = diff
            
            diff += takes.pop()
            step += 1
            
            if prev_diff * diff <= 0: return step
        return -1
