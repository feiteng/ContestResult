class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 == s2:
            return 0
        if s1 > s2:
            s1, s2 = s2, s1
            nums1, nums2 = nums2, nums1
        D = s2 - s1
        q = [x-6 for x in nums1] + [1-x for x in nums2]
        heapq.heapify(q)
        sol = 0
        while D > 0 and q:
            y = heapq.heappop(q)
            D += y
            sol += 1
        if D > 0:
            return -1
        return sol
