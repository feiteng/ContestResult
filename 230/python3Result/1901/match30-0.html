class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        
        if s1 == s2:
            return 0
        
        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
            
        pq1 = [i for i in nums1 if i != 6]
        pq2 = [-i for i in nums2 if i != 1]
        
        heapq.heapify(pq1)
        heapq.heapify(pq2)
        
        diff = s2 - s1
        cnt = 0
        while diff > 0 and pq1 and pq2:
            inc1 = 6 - pq1[0]
            dec2 = -pq2[0] - 1
            
            if inc1 >= dec2:
                gap = inc1
                heapq.heappop(pq1)
            else:
                gap = dec2
                heapq.heappop(pq2)
                
            if gap >= diff:
                diff = 0
            else:
                diff -= gap
            cnt += 1
        
        # print(diff, pq1, pq2, cnt)
        while diff > 0 and pq1:
            inc1 = 6 - pq1[0]
            gap = inc1
            heapq.heappop(pq1)
            if gap >= diff:
                diff = 0
            else:
                diff -= gap
            cnt += 1
            
        while diff > 0 and pq2:
            dec2 = -pq2[0] - 1
            heapq.heappop(pq2)
            gap = dec2
            if gap >= diff:
                diff = 0
            else:
                diff -= gap
            cnt += 1
            
            
        return cnt if diff == 0 else -1
