from heapq import *

class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        m = len(nums2)
        if n>6*m or m>6*n:
            return -1
        
        s1 = sum(nums1)
        s2 = sum(nums2)
        h1 = []
        if s1>s2:
            h2 = [-x for x in nums1]
            h1 = nums2
        else:
            h2 = [-x for x in nums2]
            h1 = nums1
            
        heapify(h1)
        heapify(h2)
            
        s1, s2 = min(s1,s2), max(s1,s2)
        ans = 0
        while s1<s2:
            ans += 1
            if 6-h1[0]>-h2[0]-1:
                k = heappop(h1)
                s1 += 6-k
                heappush(h1, 6)
            else:
                k = heappop(h2)
                s2 -= -k-1
                heappush(h2, -1)
            
        return ans
