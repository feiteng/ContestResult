class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        
        if s1 > s2:
            return self.minOperations(nums2, nums1)
    
        n1 = len(nums1)
        n2 = len(nums2)
        
        refund1 = [6 - x for x in nums1]
        refund2 = [x - 1 for x in nums2]
        r = refund1 + refund2
        r.sort(reverse = True)
        index = 0
        count = 0
        
        diff = s2 - s1
        while diff > 0 and index < len(r):
            diff -= r[index]
            index += 1
            count += 1
        
        if diff > 0:
            return -1
        
        return count
