class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        N1 = sum(nums1)
        N2 = sum(nums2)
        if N1>N2:
            nums1,nums2 = nums2,nums1
            N1,N2 = N2,N1
        diff = N2-N1
        if not diff:return 0
        print(diff, nums1,nums2)
        for i in nums2:
            nums1.append(7-i)
        nums1.sort()
        for i,n in enumerate(nums1):
            diff -= 6-n
            if diff<=0:return i+1
        return -1
