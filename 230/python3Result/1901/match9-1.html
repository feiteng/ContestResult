class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n1=len(nums1)
        n2=len(nums2)
        if n1>6*n2 or n2>6*n1: return -1
        s1=sum(nums1)
        s2=sum(nums2)
        if s1==s2: return 0
        if s1>s2:
            val1=[x-1 for x in nums1]
            val2=[6-x for x in nums2]
            vals=sorted(val1+val2)
            t=0
            ans=0
            while t<s1-s2:
                t+=vals.pop()
                ans+=1
            return ans
        else:
            val1=[6-x for x in nums1]
            val2=[x-1 for x in nums2]
            vals=sorted(val1+val2)
            t=0
            ans=0
            while t<s2-s1:
                t+=vals.pop()
                ans+=1
            return ans
