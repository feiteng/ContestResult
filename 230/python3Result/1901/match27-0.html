class Solution:
    def minOperations(self, a1: List[int], a2: List[int]) -> int:
        h1 = []
        h2 = []
        if sum(a1) > sum(a2):
            a1, a2 = a2, a1
        p = sum(a2) - sum(a1)
        for e in a1:
            heapq.heappush(h1, e)
        for e in a2:
            heapq.heappush(h2, -e)
        ans = 0
        while p > 0 and (h1 or h2):
            if h1:
                d1 = 6-h1[0]
            else:
                d1 = -1
            if h2:
                d2 = -h2[0]-1
            else:
                d2 = -1
            # print(h1[0], -h2[0], d1, d2, p)
            if d1 > d2:
                p -= d1
                heapq.heappop(h1)
            else:
                p -= d2
                heapq.heappop(h2)
            ans += 1
        # print('---')
        if p > 0:
            return -1
        return ans
