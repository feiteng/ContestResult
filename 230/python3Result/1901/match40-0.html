class Solution:
    def minOperations(self, A: List[int], B: List[int]) -> int:
        sa = sum(A)
        sb = sum(B)
        if sa == sb:
            return 0
        if sa > sb:
            sa, sb = sb, sa
            A, B = B, A
        A.sort()
        B.sort()
        mna = []
        mxa = []
        for a in A:
            if a > 1:
                mna.append(a - 1)
            if a < 6:
                mxa.append(6 - a)
        mnb = []
        mxb = []
        for b in B:
            if b > 1:
                mnb.append(b - 1)
            if b < 6:
                mxb.append(6 - b)
        
        if sa + sum(mxa) < sb - sum(mnb):
            return -1
        
        mxa.sort()
        mxb.sort()
        res = 0
        while sa < sb:
            # add to a or deduct from b
            if mnb and mxa:
                if mnb[-1] > mxa[-1]:
                    v = mnb.pop()
                    res += 1
                    sb -= v
                else:
                    v = mxa.pop()
                    res += 1
                    sa += v
            elif mnb:
                v = mnb.pop()
                res += 1
                sb -= v
            elif mxa:
                v = mxa.pop()
                res += 1
                sa += v
        return res
