class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        
        if sum1 == sum2:
            return 0
        
        if sum1 > sum2:
            nums1, nums2 = nums2, nums1
            sum1, sum2 = sum2, sum1
        
        delta = sum2 - sum1
        ans = 0
        
        counter1 = Counter(nums1)
        counter2 = Counter(nums2)
        
        for i in range(6):
            if delta <= 0:
                break
            
            value1 = counter1[i + 1]
            value2 = counter2[6 - i]
            
            while delta > 0 and value1 > 0:
                delta -= (5 - i)
                value1 -= 1
                ans += 1
            
            while delta > 0 and value2 > 0:
                delta -= (5 - i)
                value2 -= 1
                ans += 1
                
        
        if delta <= 0:
            return ans 
        else:
            return -1
