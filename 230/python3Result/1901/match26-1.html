class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if len(nums2) > 6 * len(nums1) or 6*len(nums2) < len(nums1):
            return -1
        curr1 = sum(nums1)
        curr2 = sum(nums2)
        
        if curr2 > curr1:
            steps = sorted([6 - num1 for num1 in nums1] + [num2 - 1 for num2 in nums2], reverse=True)
            count = 0
            for step in steps:
                curr1 += step
                count += 1
                if curr1 >= curr2:
                    return count
        elif curr2 < curr1:
            steps = sorted([6 - num2 for num2 in nums2] + [num1 - 1 for num1 in nums1], reverse=True)
            count = 0
            for step in steps:
                curr2 += step
                count += 1
                if curr2 >= curr1:
                    return count
        else:
            return 0
