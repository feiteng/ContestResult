class Solution:
    def minOperations(self, a: List[int], b: List[int]) -> int:
        s1=sum(a)
        s2=sum(b)
        if s1==s2:
            return 0
        if s1>s2:
            s1,s2=s2,s1
            a,b=b,a
        m,n=len(a),len(b)
        heapify(a)
        b=[-x for x in b]
        heapify(b)
        ans=0
        while s1<s2:
            if not a and not b:
                break
            if not a:
                v=-heappop(b)
                s2-=v-1
                ans+=1
            elif not b:
                v=heappop(a)
                s1+=6-v
                ans+=1
            else:
                d1=6-a[0]
                d2=-b[0]-1
                if d1<=d2:
                    heappop(b)
                    s2-=d2
                else:
                    heappop(a)
                    s1+=d1
                ans+=1
        if s1>=s2:
            return ans
        return -1
