class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n = sum(nums1)
        m = sum(nums2)
        if n == m:
            return 0
        if n < m:
            change = [6 - num for num in nums1] + [num - 1 for num in nums2]
        else:
            change = [6 - num for num in nums2] + [num - 1 for num in nums1]
        
        change.sort(reverse=True)
        
        tmp = 0
        tar = abs(n - m)
        for i, c in enumerate(change):
            if c == 0:
                return -1
            tmp += c
            if tmp >= tar:
                return i + 1
        return -1
