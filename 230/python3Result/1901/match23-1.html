class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        s1, s2 = sum(nums1), sum(nums2)
        if 6 * n1 < n2 or 6 * n2 < n1:
            return -1
        if s1 < s2:
            s1, s2 = s2, s1
            n1, n2 = n2, n1
            nums1, nums2 = nums2, nums1
        nums1.sort(reverse=True)
        nums2.sort()
        
        diff = s1 - s2
        ans = 0
        i, j = 0, 0
        while diff > 0:
            if i < n1 and nums1[i] - 1 > diff:
                    ans += 1
                    return ans
            if j < n2 and 6 - nums2[j] > diff:
                ans += 1
                return ans
            cur = 0
            
if i < n1 and j < n2:
                if nums1[i] - 1 >= 6 - nums2[j]:
                    diff -= nums1[i] - 1
                    i += 1
                    ans += 1
                else:
                    diff -= 6 - nums2[j]
                    j += 1
                    ans += 1
            elif i < n1:
                diff -= nums1[i] - 1
                i += 1
                ans += 1
            else:
                diff -= 6 - nums2[j]
                j += 1
                ans += 1
        return ans
