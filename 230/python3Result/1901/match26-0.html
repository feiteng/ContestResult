class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if 6*len(nums1) < len(nums2) or 6*len(nums2) < len(nums1): return -1 # impossible 
        
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 < s2: 
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
            
        pq1 = [-x for x in nums1] # max-heap 
        pq2 = nums2 # min-heap 
        heapify(pq1)
        heapify(pq2)
        
        ans = 0
        # s1 > s2 (1-2-3-4-5-6)
        while s1 > s2: 
            x1 = pq1[0]
            x2 = pq2[0]
            if -1-x1 > 6-x2: # change x1 to 1
                s1 += x1 + 1
                heapreplace(pq1, -1)
            else: 
                s2 += 6 - x2
                heapreplace(pq2, 6)
            ans += 1
        return ans
