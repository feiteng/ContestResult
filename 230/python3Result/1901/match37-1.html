class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n1,n2 = len(nums1),len(nums2)
        if n1*6<n2 or n2*6<n1:
            return -1
        s1,s2 = sum(nums1),sum(nums2)
        if s1 == s2:
            return 0
        elif s1 > s2:
            s1, s2 = s2, s1
            nums1,nums2 = nums2,nums1
        diff = s2 - s1
        d = sorted([6-i for i in nums1] + [i-1 for i in nums2],reverse=True)
        s = 0
        res = 0
        while s < diff:
            # print(d[res])
            s += d[res]
            res+=1
        return res
