def _cast_to_dict(A):
        initial_dict = {i: 0 for i in range(1, 7)}
        total_sum = 0
        for a in A:
            total_sum += a
            initial_dict[a] += 1
        return initial_dict, total_sum 

class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        return self.solution(nums1, nums2)

    def solution(self, nums1, nums2):
        if len(nums1) > 6 * len(nums2) or len(nums2) > 6 * len(nums1):
            return -1

        state_of_nums1, sum1 = _cast_to_dict(nums1)
        state_of_nums2, sum2 = _cast_to_dict(nums2)
    
        ans = 0
        if sum1 == sum2:
            return 0
        elif sum1 < sum2:
            state_of_nums1, state_of_nums2 = state_of_nums2, state_of_nums1
            sum1, sum2 = sum2, sum1
        different 
= sum1 - sum2
    
        possible_jumps_dict = {5: [(6, 1)], 4:[(6, 2), (5, 1)], 3:[(6, 3), (5, 2), (4, 1)], 2:[(6, 4), (5, 3), (4, 2), (3, 1)], 1:[(6, 5), (5, 4), (4, 3), (3, 2), (2, 1)]}
        while different > 0:
            max_diff_to_jump = min(5, different)
            for possible_jump in range(max_diff_to_jump, 0, -1):
                is_jump_realized = False
                for swap in possible_jumps_dict[possible_jump]:
                    if state_of_nums1[swap[0]] > 0:
                        state_of_nums1[swap[0]] -= 1
                        state_of_nums1[swap[1]] += 1
                        different -= possible_jump
                        ans += 1
                        is_jump_realized = True
                        break
                    elif state_of_nums2[swap[1]] > 0:
                        state_of_nums2[swap[1]] -= 1
                        state_of_nums2[swap[0]] += 1
                        different -= possible_jump
                        ans += 1
                        is_jump_realized = True
                        break                  
                if is_jump_realized:
                        break
        return ans
