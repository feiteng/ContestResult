class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
        d1 = collections.defaultdict(int)
        d2 = collections.defaultdict(int)
        for x in nums1:
            d1[x] += 1
        for y in nums2:
            d2[y] += 1
        d = s2 - s1
        res = 0
        for i in range(5, 0 , -1):
            a = d1[6-i] + d2[1+i]
            if a * i >= d:
                return res + (d-1) // i + 1
            d -= a*i
            res += a
        return -1
