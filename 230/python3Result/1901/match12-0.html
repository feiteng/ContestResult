class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        diff = sum(nums2) - sum(nums1) 
        if not diff:
            return 0
        if diff < 0:
            diff = -diff
            nums1, nums2 = nums2, nums1
        options = collections.Counter(6 - n for n in nums1) + collections.Counter(n - 1 for n in nums2)
        ans = 0
        for decre in reversed(range(1, 6)):
            c = options[decre]
            div, mod = divmod(diff, decre)
            div += bool(mod)
            used = min(c, div)
            ans += used
            diff -= used * decre
            if diff <= 0:
                return ans
        return -1
