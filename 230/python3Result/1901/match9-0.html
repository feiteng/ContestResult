class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        
        l1 = len(nums1)
        l2 = len(nums2)
        if l1 > l2 * 6 or l2 > l1 * 6: return -1
        
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 == s2: return 0
        if s1 < s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
        # s1 >= s2
        c1 = Counter(nums1)
        c2 = Counter(nums2)
        res = 0
        diff = s1 - s2
        for i in range(1, 7):
            n6 = c1[7-i] + c2[i]
            #print(diff, i, n6, res)
            if diff > n6 * (6-i):
                diff -= n6 * (6-i)
                res += n6
            else:
                if diff % (6-i) == 0:
                    res += diff // (6-i)
                    return res
                else:
                    res += (diff // (6-i)) + 1
                    return res
        return res
