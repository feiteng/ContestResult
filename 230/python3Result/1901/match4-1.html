class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
        
        cnt1 = collections.Counter(nums1)
        cnt2 = collections.Counter(nums2)
        
        res = 0
        diff = s2 - s1
        for i in range(1, 6):
            j = 7 - i
            cur = cnt1[i] + cnt2[j]
            if cur * (j - 1) >= diff:
                res += diff // (j - 1)
                if diff % (j - 1):
                    res += 1
                return res
            diff -= cur * (j - 1)
            res += cur
        
        return -1
