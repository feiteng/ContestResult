class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        m = len(nums2)
        if n * 6 < m or m * 6 < n:
            return -1
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        rst = 0
        if sum1 > sum2:
            l, s = nums1, nums2
            diff = sum1 - sum2
        elif sum1 < sum2:
            s, l = nums1, nums2
            diff = sum2 - sum1
        else:
            return rst
        
        diffs = []
        for c in l:
            diffs.append(c - 1)
        for c in s:
            diffs.append(6 - c)
        diffs.sort(reverse=True)
        
        for i, c in enumerate(diffs):
            diff -= c
            if diff <= 0:
                return i + 1
        return -1
