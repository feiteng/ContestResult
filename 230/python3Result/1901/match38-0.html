class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 < s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1
        if len(nums1) > len(nums2) * 6:
            return -1
        d = s1 - s2
        res = 0
        nums1 = sorted(nums1, reverse=True)
        nums2 = sorted(nums2)
        i = 0
        j = 0
        while i < len(nums1) and j < len(nums2):
            if d <= 0:
                return res
            if nums1[i] - 1 > 6 - nums2[j]:
                d -= nums1[i] - 1
                res += 1
                i += 1
            else:
                d -= 6 - nums2[j]
                res += 1
                j += 1
        if d <= 0:
            return res
        while i < len(nums1):
            d -= nums1[i] - 1
            res += 1
            i += 1
            if d <= 0:
                return res
        while j < len(nums2):
            d -= 6 - nums2[j]
            res += 1
            j += 1
            if d <= 0:
                return res
        return -1
