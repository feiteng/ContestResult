class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        diff = s1 - s2
        if diff < 0:
            nums1, nums2 = nums2, nums1
            diff = -diff
            
        l1, l2 = len(nums1), len(nums2)
        if l1 > l2 * 6:
            return -1
        nums1.sort()
        nums2.sort()
        res = 0
        i = l1 - 1
        j = 0
        while diff > 0 and (i >= 0 or j < l2):
            if i >= 0 and j < l2:
                if nums1[i] - 1 > 6 - nums2[j]:
                    diff -= nums1[i] - 1
                    res += 1
                    i -= 1
                else:
                    diff -= 6 - nums2[j]
                    res += 1
                    j += 1
            elif i >= 0:
                diff -= nums1[i] - 1
                res += 1
                i -= 1
            else:
                diff -= 6 - nums2[j]
                res += 1
                j += 1
                
        return res
