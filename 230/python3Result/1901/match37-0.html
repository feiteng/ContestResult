class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        if n1 > 6 * n2 or n2 > n1 * 6:
            return -1
        
        sum1, sum2 = sum(nums1), sum(nums2)
        if sum1 == sum2:
            return 0
        elif sum1 > sum2:
            largeNums, smallNums = nums1, nums2
            largeSums, smallSums = sum1, sum2
        else:
            largeNums, smallNums = nums2, nums1
            largeSums, smallSums = sum2, sum1
            
        largeNums.sort()
        smallNums.sort(reverse=True)
        res = 0
        while largeSums > smallSums:
            diff = largeSums - smallSums
            if largeNums:
                changeLarge = largeNums[-1] - 1
            else:
                changeLarge = -float('inf')
                
            if smallNums:
                changeSmall = 6 - smallNums[-1]
            else:
                changeSmall = -float('inf')
                
            if (largeNums and not smallNums) or changeLarge >= changeSmall:
                if changeLarge >= diff:
                    res += 1
                    return res
                else:
                    res += 1
                    largeSums -= changeLarge
                    largeNums.pop()
                continue
                
            if (smallNums and not largeNums) or changeLarge < changeSmall:
                if changeSmall >= diff:
                    res += 1
                    return res
                else:
                    res += 1
                    smallSums += changeSmall
                    smallNums.pop()
                continue
                
        return res
