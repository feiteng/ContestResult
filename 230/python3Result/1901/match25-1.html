class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = sum(nums1)
        sum2 = sum(nums2)

        if sum1 > sum2:
            return self.minOperations(nums2, nums1)
        
        nums1 = sorted(nums1)
        nums2 = sorted(nums2, reverse=True)
        m = len(nums1)
        n = len(nums2)
        i, j, ops = 0, 0, 0
        while sum1 != sum2 and (i < m or j < n):
            diff = sum2 - sum1
            if i < m and j < n:
                x = 6 - nums1[i]
                y = nums2[j] - 1
                if x >= y:
                    sum1 += min(x, diff)
                    i += 1
                else:
                    sum2 -= min(y, diff)
                    j += 1
            elif i < m:
                sum1 += min(6 - nums1[i], diff)
                i += 1 
            else:
                sum2 -= min(nums2[j] - 1, diff)
                j += 1
            ops += 1
        
        return -1 if sum1 != sum2 else ops
