class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        if sum1 > sum2:
            nums1, nums2 = nums2, nums1
            sum1, sum2 = sum2, sum1
        n1 = len(nums1)
        n2 = len(nums2)

        min1 = n1
        max1 = 6 * n1
        min2 = n2
        max2 = 6 * n2
        
        if max1 < min2 or max2 < min1:
            return -1

        diff = sum2 - sum1
        if not diff: return 0
        #print(diff)

        d 
= collections.defaultdict(int)
        for num in nums1:
            d[6-num] += 1
        #print(d)
        for num in nums2:
            d[num-1] += 1
        #print(d)
        ans = 0
        for change in range(5, 0, -1):
            if d[change] * change >= diff:
                ans += (diff + change - 1) // change
                return ans
            
            ans += d[change]
            diff -= change * d[change]
            
        return ans
