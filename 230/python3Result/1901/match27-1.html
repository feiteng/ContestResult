class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        a1, a2 
= [], []
        if sum(nums1) > sum(nums2):
            nums1, nums2 = nums2, nums1
        s1, s2 = sum(nums1), sum(nums2)
        for x in nums1:
            heapq.heappush(a1, x)
        for x in nums2:
            heapq.heappush(a2, -x)
        res = 0
        while s1 < s2:
            a, b = a1[0] if a1 else 6, -a2[0] if a2 else 1 # a min, b max
            if a == 6 and b == 1:
                return -1
            if 6-a > b-1:
                heapq.heappop(a1)
                s1 += 6-a
            else:
                heapq.heappop(a2)
                s2 -= b-1
            res += 1
        return res
