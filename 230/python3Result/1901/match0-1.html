class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if max(len(nums1), len(nums2)) > 6 * min(len(nums1), len(nums2)):
            return -1
        
        sum1 = sum(nums1)
        sum2 = sum
(nums2)
        if sum1 > sum2:
            nums1, nums2 = nums2, nums1
            sum1, sum2 = sum2, sum1
            
        swap = [0] * 6
        for u in nums1:
            swap[6 - u] += 1
        for v in nums2:
            swap[v - 1] += 1

        res = 0
        gap = sum2 - sum1
        for v in range(1, 6)[::-1]:
            curr 
= min(swap[v], ceil(gap / v))
            res += curr
            gap -= curr * v
            if gap <= 0:
                break
        return res
