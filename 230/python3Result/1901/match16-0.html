class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        if sum(nums1) > sum(nums2):
            nums1, nums2 = nums2, nums1
        
        s1, s2 = sum(nums1), sum(nums2)
        h1, h2 = list(filter(lambda x:x != 6, nums1)), list(filter(lambda y:y != -1, list(map(lambda x:-x, nums2))))
        heapq.heapify(h1)
        heapq.heapify(h2)
        
        print(h1, h2)
        ans = 0
        while s1 != s2 and (h1 or h2):
            if not h1 or (h2 and h1[0] - h2[0] >= 7):
                move = -1 - heapq.heappop(h2)
                move = min(s2 - s1, move)
                s2 = s2 - move
            else:
                move = 6 - heapq.heappop(h1)
                move = min(s2 - s1, move)
                s1 = s1 + move
            # print(move, s1, s2, h1, h2)
            ans += 1
        return ans if s1 == s2 else -1
