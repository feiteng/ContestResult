class Solution:
    def minOperations(self, a: List[int], b: List[int]) -> int:
        s1 = sum(a)
        s2 = sum(b)
        if s1 > s2:
            a, b = b, a
            s1, s2 = s2, s1
        if len(b) > len(a) * 6:
            return -1
        d = [6 - x for x in a]
        d.extend(x - 1 for x in b)
        d.sort(reverse=True)
        diff = s2 - s1
        ans = 0
        for x in d:
            if diff > 0 and x > 0:
                diff -= min(diff, x)
                ans += 1
        return ans
