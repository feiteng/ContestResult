class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1,s2 = sum(nums1),sum(nums2)
        if s1 == s2:
            return 0
        if s1>s2:
            s1,s2 = s2,s1
            nums1,nums2 = nums2,nums1
        n1,n2 = len(nums1),len(nums2)
        if 6*n1<n2 or 6*n2<n1:
            return -1
        def _func(arr):
            temp = [0] * 6
            for n in arr:
                temp[n-1] += 1
            return temp
        cnt = 0
        dis = s2-s1
        s1,s2 = _func(nums1),_func(nums2)
        # print(s1,s2)
        
        while dis:
            # print(dis,s1,s2)
            min0 = min([i for i in range(6) if s1[i]>0])
            max1 = max([i for i in range(6) if s2[i]>0])
            # print(min0,max1,5-min0)
            if 5-min0>max1:
                # print("=s1")
                if dis>s1[min0]*(5-min0):
                    dis -= s1[min0]*(5-min0)
                    cnt += s1[min0]
                    s1[-1] += s1[min0]
                    s1[min0] = 0
                else:
                    cnt += dis//(5-min0)
                    cnt += 1 if dis%(5-min0) else 0
                    return cnt
            else:
                # print("=s2")
                if dis>s2[max1]*max1:
                    dis -= s2[max1]*max1
                    cnt += s2[max1]
                    s2[0] += s2[max1]
                    s2[max1] = 0
                else:
                    cnt += dis//max1
                    cnt += 1 if dis%max1 else 0
                    return cnt
        return cnt
