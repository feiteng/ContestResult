class Solution:
    def minOperations(self, a: List[int], b: List[int]) -> int:
        
        asum = sum(a)
        bsum = sum(b)
        
        if asum == bsum:
            return 0
        
        if asum > bsum:
            a,b = b,a
            
        # asum <= bsum
        
        asum = sum(a)
        bsum = sum(b)
        
        diff = bsum-asum
        
        # a 6-x delta
        # b x-1 delta
        
        deltas 
= []
        
        for x in a:
            deltas.append(6-x)
        
        for x in b:
            deltas.append(x-1)
        
        deltas.sort(reverse=True)
        
        deltas = deque(deltas)
        
        ret = 0
        
        while deltas:
            diff -= deltas.popleft()
            ret += 1
            
            if diff <= 0:
                return ret
        
        return -1
        
        
        # greedy align
        
        # one will be lower
        
        # other will be higher
        
        # deduct until the difference becomes 0
        
        # the deltas depend on which one is higher or lower
        
        # if they are already equal then you already won
        
        # lower one == 6-x delta
        # higher one == x-1 delta
        
        # sort the deltas
        
        # greedily consume until you reduce the total difference down to <= 0
        
        # it might also be impossible to make them equal
        
        # return -1 if you exhausted all moves but still not equal
        
        # its fine if you accidentally use 0-ops
        # 0-ops implies that it is already GG game over anyways
        # you'll end up returning -1 in this case
