class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        min_sum1, min_sum2 = len(nums1), len(nums2)
        max_sum1, max_sum2 = 6 * len(nums1), 6 * len(nums2)
        if not (min_sum1 <= max_sum2 <= max_sum1 or min_sum2 <= max_sum1 <= max_sum2):
            return -1
        
        sum1, sum2 = sum(nums1), sum(nums2)
        ans = 0
        
if sum1 == sum2:
            return ans
        if sum1 < sum2:
            nums1, nums2 = nums2, nums1
        dif = abs(sum1 - sum2)
        ops = []
        for num in nums1:
            if num != 1:
                ops.append(num - 1)
        for num in nums2:
            if num != 6:
                ops.append(6 - num)
        ops.sort(reverse = True)
        for op in ops:
            dif -= op
            ans += 1
            if dif <= 0:
                return ans
        return -1
