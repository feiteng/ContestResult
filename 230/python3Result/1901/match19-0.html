class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        if sum1 == sum2:
            return 0
        if sum1 > sum2:
            sum1,sum2=sum2,sum1
            nums1,nums2 = nums2, nums1
        # have sum1 < sum2
        count = 0
        heap = []
        for num in nums1:
            if num != 6:
                heap.append((num-6,1))
        for num in nums2:
            if num != 1:
                heap.append((1-num,2))
        heapq.heapify(heap)
        while heap:
            value, index = heapq.heappop(heap)
            value = -value
            if value < sum2 - sum1:
                count += 1
                sum1 += value
            else:
                return count+1
        return -1
