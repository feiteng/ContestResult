class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 == s2:
            return 0
        if s1 < s2:
            s1, s2 = s2, s1
            nums1, nums2 = nums2, nums1
        dd = s1 - s2
        rr = list(map(lambda x: x-1, nums1)) + list(map(lambda x: 6-x, nums2))
        rr.sort(reverse= True)
        for i in range(len(rr)):
            if dd <= rr[i]:
                return i+1
            dd -= rr[i]
        return -1
