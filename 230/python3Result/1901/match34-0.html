class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
      
      #def helper(A, B):
        #sum(A) > sum(B)
        
      sm1 = sum(nums1)
      sm2 = sum(nums2)
      if sm1 == sm2:
        return 0
      if sm1 < sm2:
        nums1, nums2 = nums2, nums1
        sm1, sm2 = sm2, sm1
        
      #sum(nums1) > sum(nums2)
      diff = sm1 - sm2
      nums1.sort(reverse=True)
      nums2.sort()
      
      i = j = 0
      m, n = len(nums1), len(nums2)
      res = 0
      while (i < m or j < n) and diff > 0:
        d1 = nums1[i] - 1 if i < m else 0
        d2 = 6 - nums2[j] if j < n else 0
        if not d1 and not d2:
          break
        if d1 + d2 < diff:
          if d1:
            i += 1
            diff -= d1
            res += 1
          if d2:
            j += 1
            diff -= d2
            res += 1
        elif max(d1, d2) < diff:
          res += 2
          diff = 0
          break
        else:
          res += 1
          diff = 0
          break
      return res if diff == 0 else -1
