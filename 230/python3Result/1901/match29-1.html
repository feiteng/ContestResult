class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1, s2 = sum(nums1), sum(nums2)
        if s1 == s2:
            return 0
        if s1 < s2:
            s1, s2 = s2, s1
            nums1, nums2 = nums2, nums1
        # nums1.sort()
        # nums2.sort()
        target = s1 - s2
        print(nums1, nums2)
        res = []
        ret = 0
        for n1 in nums1:
            res.append(n1 - 1)
        for n2 in nums2:
            res.append(6 - n2)
        res.sort(reverse=True)
        s = 0
        for r in res:
            if s >= target:
                break
            s += r
            ret += 1
        if s < target: return -1
        return ret
