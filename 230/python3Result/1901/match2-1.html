class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)

        if s1 > s2:
            nums1, nums2 = nums2, nums1
            s1, s2 = s2, s1


        L1 = [0 for _ in range(7)]
        for e in nums1:
            L1[e] += 1

        L2 = [0 for _ in range(7)]
        
for e in nums2:
            L2[e] += 1


        ret = 0

        D = s2 - s1
        for i in range(1, 7):
            x = (6 - i) * (L1[i] + L2[7 - i])
            if x >= D:
                return ret + (D - 1) // (6 - i) + 1

            ret += L1[i] + L2[7 - i]
            D -= x

        return -1
