class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1, s2 = sum(nums1), sum(nums2)
        if s1 == s2: return 0
        if s1 < s2:
            nums1, nums2 = nums2, nums1
            s2, s1 = s1, s2
        d = s1 - s2
        import heapq
        pq = []
        for x in nums1:
            if x > 1:
                heapq.heappush(pq, 1-x)
        for x in nums2:
            if x < 6:
                heapq.heappush(pq, x-6)
        # print(d, pq)
        if -sum(pq) < d: return -1
        if -sum(pq) == d: return len(pq)
        cnt = 0
        while d > 0:
            d += heapq.heappop(pq)
            cnt += 1
        return cnt
