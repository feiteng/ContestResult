class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        sum1 = sum(nums1)
        sum2 = sum(nums2)
        if sum1 == sum2:
            return 0
        if sum1 > sum2: # let's swap these to simplify things
            sum1, sum2 = sum2, sum1
            nums1, nums2 = nums2, nums1
        nums1 = sorted(nums1)
        nums2 = sorted(nums2, reverse=True)
        ops = 0
        while (sum1 < sum2):
            prevDiff = sum2 - sum1
            if prevDiff == 0:
                return ops
            
            bestPlus = 6 - nums1[0] if len(nums1) > 0 else 0
            bestMinus = nums2[0] - 1 if len(nums2) > 0 else 0
            if bestPlus == 0 and bestMinus == 0:
                return -1
            if max(bestPlus, bestMinus) >= prevDiff:
                return ops + 1
            if bestPlus > bestMinus:
                sum1 += bestPlus
                nums1.pop(0)
            else:
                sum2 -= bestMinus
                nums2.pop(0)
            
            if sum2 - sum1 == prevDiff:
                return -1
            ops += 1
        return ops
