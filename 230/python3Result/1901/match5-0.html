class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        nums1.sort()
        nums2.sort()
        n, m = len(nums1), len(nums2)
        if 1 * n > 6 * m:
            return - 1
        if 1 * m > 6 * n:
            return - 1
        
        s1, s2= sum(nums1), sum(nums2)
        if s1 < s2:
            nums1, nums2 = nums2, nums1
            s1,s2 = s2, s1
        if s1 == s2:
            return 0
        statea = Counter(nums1)
        stateb = Counter(nums2)
        bal = s1 - s2
        ans = 0
        jumps = {5: [(6, 1)], 4: [(6, 2), (5, 1)], 3: [(6, 3), (5, 2), (4, 1)],
                      2: [(6, 4), (5, 3), (4, 2), (3, 1)], 1: [(6, 5), (5, 4), (4, 3), (3, 2), (2, 1)]}
        while bal > 0:
            maxjump  =  min(5, bal)
            for pos in range(maxjump, 0, -1):
                jumped = False
                for swap in jumps[pos]:
                    if statea[swap[0]] > 0:
                        statea[swap[0]] -= 1
                        
                        statea[swap[1]] += 1
                        bal -= pos
                        ans += 1
                        jumped = True
                        break
                    elif stateb[swap[1]] > 0:
                        stateb[swap[1]] -= 1
                        stateb[swap[0]] += 1
                        bal -= pos
                        ans += 1
                        jumped = True
                        break
                if jumped:
                    break
        return ans
