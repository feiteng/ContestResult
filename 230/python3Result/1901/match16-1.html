class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int]) -> int:
        s1 = sum(nums1)
        s2 = sum(nums2)
        if s1 == s2:
            return 0
        if s1 < s2:
            return self.minOperations(nums2, nums1)
        if len(nums1) > 6 * len(nums2): # nums1 all 1 and nums2 all 6 
            return -1
        hq1 = [-x for x in nums1]
        heapq.heapify(hq1)
        hq2 = [x for x in nums2]
        heapq.heapify
(hq2)
        ops = 0
        while s1 != s2:
            if not hq2 or ((hq1 and hq2) and (-hq1[0] - 1) >= (6 - hq2[0])):
                cur = -heapq.heappop(hq1)
                delta = min(s1 - s2, cur - 1)
                # cur -= delta
                s1 -= delta
            else:
                cur = heapq.heappop(hq2)
                delta = min(s1 - s2, 6 - cur)
                # cur += delta
                s2 += delta
            ops += 1
        return ops
