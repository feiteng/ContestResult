class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        st = [[cars[-1][0], cars[-1][1], sys.maxsize]]
        ans = [-1]
        for i in range(n-2, -1, -1):
            pi, si = cars[i]
            an = sys.maxsize
            found = False
            while st:
                if st[-1][1] >= si:
                    st.pop()
                else:
                    pj, sj, tj = st[-1]
                    t = (pj-pi)/(si-sj) 
                    if t <= tj:
                        st.append([pi, si, t])
                        ans.append(t)
                        found= True
                        break
                    else:
                        st.pop()
            if not found:
                st.append([pi, si, sys.maxsize])
                ans.append(-1)
        ans.reverse()
        return ans
