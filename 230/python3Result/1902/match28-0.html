class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        res = [-1] * n
        sp = [[0, cars[-1][1]]]
        #print(sp)
        for i in range(n - 2, -1, -1):
            p, s = cars[i]
            if s <= sp[-1][1]:
                res[i] = -1
                sp = [[0, s]]
                continue
            gap = cars[i+1][0] - p
            rt = 0
            nsp = []
            for j in range(len(sp)):
                ct, cs = sp[j]
                req_t = gap / (s - cs) if s != cs else inf
                ht = inf if j == len(sp) - 1 else sp[j+1][0] - ct
                if req_t < 0 or req_t > ht:
                    gap -= (ht * (s - cs))
                    rt += ht
                else:
                    
                    nsp = [[0, s], [rt + req_t, cs]]
                    res[i] = rt + req_t
                    for k in range(j, len(sp)):
                        if sp[k][0] > rt + req_t:
                            nsp.append(sp[k])
                        else:
                            nsp[-1][1] = sp[k][1]
                    break
            sp = nsp
            #print(sp)
                    
        return res
