class Solution:
    def getCollisionTimes(self, C: List[List[int]]) -> List[float]:
        n = len(C)
        ans = [-1.0] * n
        slowest = C[-1][1]
        que = collections.deque([n-1])
        
for i in range(n-2, -1, -1):
            p, v = C[i]
            while que and C[que[-1]][1] >= v: que.pop()
            if que:
                # ans[i] = C[-1][0]
                # for j in reversed(que):
                #     ans[i] = min(ans[i], (C[j][0] - p) / (v - C[j][1]))
                #     if ans[i] <= ans[j]: break
                        
                l, r = 0, len(que) - 1
                while r > 0 and l < r:
                    m = (l + r + 1) // 2
                    if (v - C[que[m]][1]) * ans[que[m]] + p < C[que[m]][0]: r = m - 1
                    else: l = m
                ans[i] = (C[que[l]][0] - p) / (v - C[que[l]][1])
                
            # print(i, C[i], [(C[j], ans[j]) for j in que], C[que[l]], ans[i])
            while que and (ans[que[-1]] != -1 and ans[que[-1]] <= ans[i]): que.pop() 
            que.append(i)
        return ans
