class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        ans = []
        stack = []
        for i in range(n)[::-1]:
            while stack and stack[-1][1] >= cars[i][1]:
                stack.pop()
            if not stack:
                ans.append(-1)
                stack.append([cars[i][0], cars[i][1], -1])
                continue
            j = len(stack) - 1
            while stack[j]
[2] != -1 and (stack[j][0] - cars[i][0]) / abs(stack[j][1] - cars[i][1]) > stack[j][2]:
                j -= 1

            ans.append((stack[j][0] - cars[i][0]) / abs(stack
[j][1] - cars[i][1]))
            stack.append([cars[i][0], cars[i][1], ans[-1]])
        return ans[::-1]
