class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        st = [[0, cars[-1][0], cars[-1][1]]]
        ret = [-1.0] * n        
        for i in range(n - 2, -1, -1):
            pos, speed = cars[i]
            p = None
            while st and st[-1][1] >= speed * (st[-1][0]) + pos:
                p = st.pop()
            if p is not None:
                x, y, k = p
                if speed > k:
                    xx = (y - k * x - pos) / (speed - k)
                    yy = speed * xx + pos
                    st.append([xx, yy, k])
                    ret[i] = xx
            st.append([0, cars[i][0], cars[i][1]])
        return ret
