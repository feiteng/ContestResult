class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        
        res = [0] * len(cars)
        res[-1] = -1
        slowSpeed = cars[-1][1]
        for i in range(len(cars)-2, -1, -1):
            if cars[i][1] <= slowSpeed:
                res[i] = -1
                slowSpeed = cars[i][1]
            else:
                for j in range(i+1, len(cars)):
                    if cars[i][1] <= cars[j][1]: continue
                    v = (cars[j][0] - cars[i][0]) / (cars[i][1] - cars[j][1])
                    if res[j] == -1 or v <= res[j]:
                        res[i] = v
                        break
                
        return res
