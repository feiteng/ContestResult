class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        res = [-1 for i in range(n)]
        aux = [-1 for i in range(n)]
        for i in range(n-2, -1, -1):
            # 前面是快车且追不上
            p = i+1
            while cars[i][1] <= cars[p][1] and aux[p] != -1:
                p = aux[p]
            # 前面的都是快车
            if cars[i][1] <= cars[p][1]:
                continue
            else:
                t = -(cars[p][0] - cars[i][0])/(cars[p][1] - cars[i][1])
                while aux[p] != -1 and t>res[p]:
                    p = aux[p]
                    t = -(cars[p][0] - cars[i][0])/(cars[p][1] - cars[i][1])
                aux[i] = p
                res[i] = t
            # 前面是慢车
        return res
