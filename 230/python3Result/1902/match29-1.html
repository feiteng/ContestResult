class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        times = []
        rst = [-1] * len(cars)
        fleet_firsts = [i for i in range(len(cars))]
        for i in range(len(cars) - 1):
            speed_diff = cars[i][1] - cars[i + 1][1]
            if speed_diff <= 0:
                continue
            dist = cars[i + 1][0] - cars[i][0]
            time = dist / speed_diff

            time, fleet_last, next_car = heappop(times)
            
            if rst[fleet_last] != -1:
                continue
            
            rst[fleet_last] = time
            
            fleet_first = fleet_firsts[fleet_last]
            fleet_firsts[next_car] = fleet_first
            
            if fleet_first == 0:
                continue
                
            new_fleet_last = fleet_first - 1
            
            new_speed_diff = cars[new_fleet_last][1] - cars[next_car][1]
            if new_speed_diff <= 0:
                continue
            
            new_dist = cars[next_car][0] - cars[new_fleet_last][0]
            new_time = time = new_dist / new_speed_diff
            heappush(times, (new_time, new_fleet_last, next_car))
