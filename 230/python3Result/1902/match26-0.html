class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        output = [-1] * n
        q = []
        for i in range(1, n):
            a, b = cars[i - 1], cars[i]
            if a[1] > b[1]:
                q.append((((b[0] - a[0]) / (a[1] - b[1])), (i - 1, i)))
        heapq.heapify(q)
        mapping = list(range(n))
        
        def find(x):
            if mapping[x] != x:
                mapping[x] = find(mapping[x])
            return mapping[x]
        
        while q:
            time, (a, b) = heapq.heappop(q)
            if output[a] == -1:
                output[a] = time
                if a > 0:
                    #print(a, find(a - 1), mapping)
                    tmp = a
                    a = find(a - 1)
                    mapping[tmp] = a
                    if cars[a][1] > cars[b][1]:
                        heapq.heappush(
                            q,
                            (
                                (cars[b][0] - cars[a][0]) / (cars[a][1] - cars[b][1]),
                                (a, b)
                            )
                        )
        return output
