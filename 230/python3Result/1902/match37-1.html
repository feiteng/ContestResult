class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        to_ret = [-1] * len(cars)
        next_less = {}
        stack = [(-1, -1e99)]
        for i, (p, s) in enumerate(cars) :
            while s < stack[-1][1] :
                next_less[stack[-1][0]] = i
                stack.pop(-1)
            stack.append([i, s])
        fathers = collections.defaultdict(lambda:-1)
            
        for idt in range(len(cars)-1, -1, -1) :
            if not idt in next_less :
                continue
            nlt = next_less[idt]
            to_ret[idt] = (cars[nlt][0]-cars[idt][0]) / (cars[idt][1]-cars[nlt][1])
            
            fathers[idt] = nlt
            father = fathers[nlt]
            while not father == -1 :
                temp 
= (cars[father][0]-cars[idt][0]) / (cars[idt][1]-cars[father][1])
                if temp < to_ret[idt] :
                    to_ret[idt] = temp
                    fathers[idt] = father
                    father = fathers[father]
                else :
                    father = -1
                
        return to_ret
