class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        s,st=[-1.00000]*len(cars),[]
        for i in range(len(cars)-1,-1,-1):
            while st and (cars[i][1]<=cars[st[-1]][1] or s[st[-1]]>0 and (cars[st[-1]][0]-cars[i][0])/(cars[i][1]-cars[st[-1]][1])>=s[st[-1]]): st.pop()
            if st: s[i]=(cars[st[-1]][0]-cars[i][0])/(cars[i][1]-cars[st[-1]][1])
            st.append(i)
        return s
