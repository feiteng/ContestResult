class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        res = [-1] * n
        if n == 1:
            return res
        carsToCollide = [(i + 1) for i in range(n)]
        carsColliding = [(i - 1) for i in range(n)]
        canCollide = [True] * n
        eventHeap = []
        for i in range(n - 1):
            if cars[i][1] <= cars[i + 1][1]: continue
            heapq.heappush(eventHeap, [(cars[i + 1][0] - cars[i][0]) / (cars[i][1] - cars[i + 1][1]), i, i + 1])
        
        while len(eventHeap):
            currTime, origCar, targetCar = heapq.heappop(eventHeap)
            print(currTime, origCar, targetCar)
            if carsColliding[targetCar] != origCar: continue
            if not canCollide[origCar] : continue
            canCollide[origCar] = False
            res[origCar] = currTime
            if origCar != 0:
                carToCollideOrigCar = carsColliding[origCar]
                carsToCollide[carToCollideOrigCar] = targetCar
                carsColliding
[targetCar] = carToCollideOrigCar
                if cars[carToCollideOrigCar][1] > cars[targetCar][1]:
                    heapq.heappush(eventHeap, [(cars[targetCar][0] - cars[carToCollideOrigCar][0]) / (cars[carToCollideOrigCar][1] - cars[targetCar][1]),carToCollideOrigCar, targetCar])
        
        return res
