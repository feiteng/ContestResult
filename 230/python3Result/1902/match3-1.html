from sortedcontainers import SortedList
from heapq import heappush, heappop

class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        ans = [-1] * n
        visited = set()
        
        sl = SortedList(list(range(n)))
        
        a = []
        for i in range(0, n-1):
            t = math
.inf if cars[i][1] <= cars[i+1][1] else (cars[i+1][0] - cars[i][0]) / (cars[i][1] - cars[i+1][1])
            heappush(a, (t, i, i+1))
        
        # print(a)  
        
        while a:
            t, x, y = heappop(a)
            if t == math.inf:
                break
            if x in visited or y in visited:
                continue
            ans[x] = t
            visited.add(x)
            sl.remove(x)
            idx = sl.bisect_left(x)
            if idx > 0:
                z = sl[idx-1]
                t = math.inf if cars[z][1] <= cars[y][1] else (cars[y][0] - cars[z][0]) / (cars[z][1] - cars[y][1])
                heappush(a, (t, z, y))
        
        return ans
