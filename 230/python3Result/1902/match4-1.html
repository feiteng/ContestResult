class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        ans = [inf]*n
        stack = [(cars[-1],n-1)]
        last = n-1
        for i in range(n-2,-1,-1):
            p, v = cars[i]
            while stack and (stack[-1][0][1]>=v or (stack[-1][0][0]-p)/(v-stack[-1][0][1])>ans[stack[-1][1]]):
                 stack.pop()
            if stack:
                ans[i] = (stack[-1][0][0]-p)/(v-stack[-1][0][1])
            stack.append((cars[i],i))
            
        for i, val in enumerate(ans):
            if val==inf:
                ans[i] = -1.0
        return ans
