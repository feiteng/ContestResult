class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        import heapq
        n = len(cars)
        events = []
        ans = [-1] * n
        prev = [i - 1 for i in range(n)]
        
        for i in range(1, n):
            x1, s1 = cars[i - 1]
            x2, s2 = cars[i]
            if s1 > s2:
                t = (x2 - x1)/(s1 - s2)
                heapq.heappush(events, (t, i - 1, i))
        
        while len(events):
            dt, i, j = heapq.heappop(events)
            if ans[i] < 0 and dt > 0:
                ans[i] = dt
                prev[j] = prev[i]
                idx = prev[j]
                if 0 < i < n and cars[idx][1] > cars[j][1]:
                    t = (cars[j][0] - cars[idx][0])/(cars[idx][1] - cars[j][1])
                    if t > 0:
                        heapq.heappush(events, (t, idx, j))
        return ans
