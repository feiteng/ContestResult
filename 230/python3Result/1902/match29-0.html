class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        prev = [i-1 for i in range(n)]
        nxt = [i+1 for i in range(n)]
        speed = [car[1] for car in cars]
        ps = [car[0] for car in cars]
        
        ts = []
        res = [-1] * n
        for i in range(n-1):
            if speed[i] > speed[i+1]:
                heappush(ts, [(ps[i+1] - ps[i]) / (speed[i] - speed[i+1]), i, i+1])
        
        while ts:
            t, carId, carId2 = heappop(ts)
            # print(t, carId, carId2)
            if res[carId] != -1:
                continue
            res[carId] = t
            before = prev[carId]
            if before != -1:
                nxt[before] = carId2
                prev[carId2] = before
                if speed[before] > speed[carId2]:
                    heappush(ts, [(ps[carId2] - ps[before]) / (speed[before] - speed[carId2]), before, carId2])
            # print(res)
        return res
