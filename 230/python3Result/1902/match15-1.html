class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        res = []
        stack = []
        
        for i in reversed(range(len(cars))):
            while stack and cars
[stack[-1]][1] >= cars[i][1]:
                stack.pop() # spead greater than this car, won't collide
            
            # won't collide with anyone
            if not stack:
                res.append(-1)
                stack.append(i)
            
            else:
                # collision time
                t = - (cars[i][0] - cars[stack[-1]][0]) / (cars[i][1] - cars[stack[-1]][1])
                while len(stack) > 1:
                    t1 = - (cars[i][0] - cars[stack[-2]][0]) / (cars[i][1] - cars[stack[-2]][1])
                    if t1 <= t:
                        t = t1
                        stack.pop()
                    else:
                        break
                res.append(t)
                stack.append(i)
    
        return res[::-1]
