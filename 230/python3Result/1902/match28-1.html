class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        res = [-1] * n
        st = [[0, cars[-1][0], cars[-1][1]]]  # time, start_pos, speed
        for i in range(n-2, -1, -1) :
            pos_i, speed_i = cars[i]
            t = 0
            while st :
                time, start_pos, speed = st[-1]
                if len(st) > 1 :
                    time2 = st[-2][0]
                    if speed_i > speed and (start_pos - pos_i - speed_i * time) <= (speed_i - speed) * (time2 - time) :
                        break
                else :
                    if speed_i > speed :
                        break
                st.pop()
            if st :
                time, start_pos, speed = st[-1]
                collide_time = (start_pos - pos_i - speed_i * time) / (speed_i - speed)
                #print(i, cars[i], st[-1], collide_time)
                st[-1][0] = res[i] = collide_time + time
                st[-1][1] = start_pos + speed * collide_time
            st.append([0, pos_i, speed_i])
        return res
