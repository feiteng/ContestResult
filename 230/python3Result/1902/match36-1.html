class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        final_speed = int(1e6+10)
        res = [-1.0]*len(cars)
        q = [(final_speed,int(1e10))]
        for i in range(len(cars)-1,-1,-1):
            if cars[i][1]<= q[0][0]:
                q =[(cars[i][1],int(1e10))]
                continue
            l,ti = cars[i+1][0]-cars[i][0],0

            while len(q):
                if l+q[-1][1]*q[-1][0]>cars[i][1]*(ti+q[-1][1]):
                    l+=q[-1][1]*q[-1][0]
                    ti+=q[-1][1]
                    q.pop()

                elif l+q[-1][1]*q[-1][0]==cars[i][1]*(ti+q[-1][1]):
                    l+=q[-1][1]*q[-1][0]
                    ti+=q[-1][1]
                    q.pop()
                    res [i]= ti
                    q.append((cars[i][1],ti))
                    break
                else:
                    x = (l-cars[i][1]*ti)/(cars[i][1]-q[-1][0])
                    l+=q[-1][0]*x
                    ti+=x
                    (tops,topt)= q[-1]
                    q.pop()
                    q.append((tops,topt-x))
                    q.append((cars[i][1],ti))
                    res [i]= ti
                    break
        return res
