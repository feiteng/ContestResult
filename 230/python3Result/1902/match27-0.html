class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        n = len(cars)
        ans = [-1.0] * n
        # find: 汇合的第一辆车
        find = [-1] * n
        # 用于找前一辆最慢的车
        stack = [n - 1]

        # 从后往前遍历
        for i in range(n - 2, -1, -1):
            pos, speed = cars[i]

            while stack and cars[stack[-1]][1] >= speed:
                stack.pop()
            if not stack:
                stack.append(i)
                continue

            # 撞
            last = stack[-1]
            while True:
                ans[i] = (cars[last][0] - pos) / (speed - cars[last][1])
                if ans[last] == -1.0 or ans[i] < ans[last]:
                    break
                last = find[last]
            find[i] = last
            stack.append(i)

        return ans
