import heapq

class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        heap = []
        result = [-1 for _ in cars]
        
        def get_time(car0, car1):
            return (car1[0] - car0[0]) / (car0[1] - car1[1])
        
        for i in range(len(cars) - 1):
            if cars[i][1] > cars[i + 1][1]:
                heapq.heappush(heap, (get_time(cars[i], cars[i + 1]), i, i + 1))
        
        behind = [i - 1 for i in range(len(cars))]
        while heap:
            time, i0, i1 = heapq.heappop(heap)
            if result[i0] != -1:
                continue
            result[i0] = time
            
            behind[i1] = behind[i0]
            if behind[i1] >= 0 and cars[behind[i1]][1] > cars[i1][1]:
                heapq.heappush(heap, (get_time(cars[behind[i1]], cars[i1]), behind[i1], i1))
            
            
            # i = i0
            # while i >= 0 and cars[i] == None:
            #     i -= 1
            # if i < 0:
            #     continue
            # if cars[i][1] > cars[i1][1]:
            #     heapq.heappush(heap, (get_time(cars[i], cars[i + 1]), i))
                
            
        
        return result
