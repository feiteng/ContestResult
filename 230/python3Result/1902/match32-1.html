class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        i = len(cars) - 1
        res = []
        stack = []
        while i >= 0:
            while stack and (cars[i][1] <= stack[-1][1] or (stack[-1][0] - cars[i][0]) / (cars[i][1] - stack[-1][1]) >= stack[-1][2]):
                stack.pop()
            if not stack:
                stack.append((cars[i][0], cars[i][1], math.inf))
                res.append(-1)
            else:
                sec 
= (stack[-1][0] - cars[i][0]) / (cars[i][1] - stack[-1][1])
                stack.append((cars[i][0], cars[i][1], sec))
                res.append(sec)
            i -= 1
        res.reverse()
        return res
