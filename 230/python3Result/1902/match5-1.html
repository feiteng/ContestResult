class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
      n = len(cars)
      t = [float('inf')] * n
      q = []
      inc = [n - 1]
      crash = {}
      for i in range(n - 1)[::-1]:
        while inc and cars[i][1] <= cars[inc[-1]][1]:
          inc.pop()
        if inc:
          idx = inc[-1]
          t[i] 
= (cars[idx][0] - cars[i][0]) / (cars[i][1] - cars[idx][1])
          while t[i] > t[idx]:
            idx = crash[idx]
            t[i] 
= (cars[idx][0] - cars[i][0]) / (cars[i][1] - cars[idx][1])
          crash[i] = idx
          heapq.heappush(q, (t[i], i))
        inc.append(i)
          
      res = [-1] * n
      while q:
        time, car = heapq.heappop(q)
        res[car] = time
      return res
