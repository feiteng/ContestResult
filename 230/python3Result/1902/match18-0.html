class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        team, n = [[0, cars[-1][1]]], len(cars)
        res = [-1.0] * n
        for i in range(n - 2, -1, -1):
            if cars[i][1] <= team[0][1]:
                team = [(0, cars[i][1])]
            else:
                dis = cars[i + 1][0] - cars[i][0]
                res[i] = 0
                while len(team) > 1 and dis > team[-1][0] * (cars[i][1] - team[-1][1]):
                    dis -= team[-1][0] * (cars[i][1] - team[-1][1])
                    res[i] += team[-1][0]
                    team.pop()
                if dis and len(team) == 1:
                    res[i] += dis / (cars[i][1] - team[0][1])
                    team.append([res[i], cars[i][1]])
                elif dis:
                    t = dis / (cars[i][1] - team[-1][1])
                    res[i] += t
                    team[-1][0] -= t
                    team.append([res[i], cars[i][1]])
        return res
