class Solution:
    def getCollisionTimes(self, cars: List[List[int]]) -> List[float]:
        h = []
        ans = [-1] * len(cars)
        prv = [-1] + list(range(len(cars)-1))
        nxt = list(range(1, len(cars))) + [-1]
        for i in range(len(cars)-1):
            a, b = cars[i]
            c, d = cars[i+1]
            if b > d:
                t = (c - a) / (b - d)
                heapq
.heappush(h, (t, i, i+1))
            
        while h:
            x, y, z = heapq.heappop(h)
            if ans[y] != -1:
                continue
            ans[y] = x
            
            a, b = prv[y], nxt[y]
            if b != -1:
                prv[b] = a
            if a != -1:
                nxt[a] = b
                
            if a != -1 and b != -1:
                y, z = a, b
                a, b = cars[y]
                c, d = cars[z]
                if b > d:
                    t = (c - a) / (b - d)
                    heapq.heappush(h, (t, y, z))
        return ans
