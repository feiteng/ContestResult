class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        sett = set([0])
        for toppingCost in toppingCosts:
            new_sett = set()
            for val in sett:
                new_sett.add(val + toppingCost)
                new_sett.add(val + toppingCost + toppingCost)
            sett.update(new_sett)
        
        new_sett = set()
        for baseCost in sorted(baseCosts):
            for val in sett:
                new_sett.add(val + baseCost)

        # print(new_sett)
        
        i = 0
        while True:
            if target - i in new_sett:
                return target - i
            if target + i in new_sett:
                return target + i
            i += 1
