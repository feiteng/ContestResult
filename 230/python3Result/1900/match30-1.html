class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        can_reach = set(baseCosts)
        for cost in toppingCosts:
            new_can_reach = set()
            for prev in can_reach:
                new_can_reach.add(prev + cost)
                new_can_reach.add(prev + 2*cost)
            can_reach |= new_can_reach
        return min(can_reach, key=lambda cost: (abs(cost - target), cost))
