class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        inf = 1000000000
        best = inf
        M = len(toppingCosts)

        def go(cost, index):
            if index == M:
                nonlocal best
                if abs(target - cost) < abs(target - best):
                    best = cost
                elif abs(target - cost) == abs(target - best) and best > cost:
                    best = cost
                return
            
            go(cost, index + 1)
            go(cost + toppingCosts[index], index + 1)
            go(cost + toppingCosts[index] * 2, index + 1)
        
        for x in baseCosts:
            go(x, 0)
                
        return best
