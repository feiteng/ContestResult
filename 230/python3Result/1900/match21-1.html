class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        ans = [10**10]
        n = len(toppingCosts)
        
        def dfs(cur, p):
            if p == n:
                if abs(cur-target) < abs(ans[0]-target):
                    ans[0] = cur
                elif abs(cur-target) == abs(ans[0]-target) and cur < ans
[0]:
                    ans[0] = cur
                return
            dfs(cur, p+1)
            dfs(cur+toppingCosts[p], p+1)
            dfs(cur+toppingCosts[p]*2, p+1)
        
        for i in baseCosts:
            dfs(i, 0)
        return ans[0]
