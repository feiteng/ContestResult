class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        res = 0
        
        cost = set()
        cost.add(0)
        
        for t in toppingCosts:
            new_cost = set()
            for c in cost:
                new_cost.add(c)
                new_cost.add(c+t)
                new_cost.add(c+t+t)
            cost = new_cost
        
        costs = sorted(list(cost))
        
        res = -float("Inf")
        for b in baseCosts:
            for c in costs:
                if abs(b + c - target) < abs(res - target):
                    res = b + c
                elif abs(b + c - target) == abs(res - target) and b+c < res:
                    res = b + c
                if b + c - target > res - target:
                    break
        return res
