class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        res = sys.maxsize
        
        def dfs(pos, cost):
            nonlocal res
            if abs(cost - target) < abs(res - target) or abs(cost - target) == abs(res - target) and cost < target:
                res = cost
            if pos >= len(toppingCosts):
                return
            dfs(pos+1, cost)
            dfs(pos+1, cost+toppingCosts[pos])
            dfs(pos+1, cost+2*toppingCosts[pos])
        
        for b in baseCosts:
            dfs(0, b)
            
        return res
