class Solution:
    def __init__(self):
            self.best = float('inf')
            
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        def dfs(i, cur, arr):
            if i==len(arr):
                if abs(target-cur) < abs(target-self.best): 
                    self.best = cur
                return
            dfs(i+1, cur, arr)
            dfs(i+1, cur+arr[i], arr)
            dfs(i+1, cur+2*arr[i], arr)
       
        for base in baseCosts:
            dfs(0, base, toppingCosts)
        
        return self.best
