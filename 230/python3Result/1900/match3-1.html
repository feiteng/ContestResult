class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        self.res = float('inf')
        def dfs(target, index):
            if abs(target) < abs(self.res):
                self.res = target
            elif abs
(target) == abs(self.res):
                self.res = max(self.res, target)
            if index == len(toppingCosts):
                return
            dfs(target - toppingCosts[index], index + 1)
            dfs(target - 2 * toppingCosts[index], index + 1)
            dfs(target, index + 1)
        for baseCost in baseCosts:
            newTarget = target - baseCost
            dfs(newTarget, 0)
        return target - self.res
