from functools import lru_cache
class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        n = len(baseCosts)
        m = len(toppingCosts)
        
        @lru_cache(None)
        def search(j, t):
            if j == m:
                return 0
            if t == 0:
                return 0
            ans = 100000
            for k in range(3):
                cost = k*toppingCosts[j]+search(j+1, max
(t-k*toppingCosts[j], 0))
                if abs(cost - t) < abs(ans-t):
                    ans = cost
                elif abs(cost - t) == abs(ans-t):
                    ans = min(cost, ans)
            return ans
                    
            
        ans = 10000000
        for k in baseCosts:
            cost = search(0, max
(target-k, 0))
            if abs(ans-target) > abs(cost+k-target):
                ans = cost+k
            elif abs(ans-target) == abs(cost+k-target):
                ans = min(ans, cost+k)
                
        return ans
