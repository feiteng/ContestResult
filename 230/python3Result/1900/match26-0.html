class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        dp = set(baseCosts)
        for i in toppingCosts:
            v = set()
            for j in dp:
                v.add(j + i)
                v.add(j + i + i)
            dp = dp.union(v)
        result = baseCosts[0]
        diff = abs(result - target)
        for i in dp:
            if diff > abs(i - target):
                result = i
                diff = abs(i - target)
            elif diff == abs(i - target) and i < result:
                result = i
        return result
