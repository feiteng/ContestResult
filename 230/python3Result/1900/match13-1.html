class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        cc = set()
        cc.add(0)
        for c in toppingCosts:
            ncc=set()
            for pc in cc:
                ncc.add(pc+c)
                ncc.add(pc+2*c)
            cc|=ncc
        
        mn=ans=10**16
        for bc in baseCosts:
            for tc in cc:
                cur=abs(bc+tc-target)
                if cur<mn or (cur==mn and bc+tc<ans):
                    mn=cur
                    ans=bc+tc
        
        return ans
