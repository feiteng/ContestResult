class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppings_set = set([0])
        for topping_cost in toppingCosts:
            for cost in list(toppings_set):
                toppings_set.add(cost + topping_cost)
                toppings_set.add(cost + 2 * topping_cost)
        
        ans = 0
        minn = 10**9
        for base_cost in baseCosts:
            for cost in list(toppings_set):
                diff = abs(cost + base_cost - target)
                if diff < minn or (diff == minn and cost + base_cost < target):
                    minn = diff
                    ans = cost + base_cost
        return ans
