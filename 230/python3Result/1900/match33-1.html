class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        t = [0]
        for i in toppingCosts:
            t = t + [j + i for j in t] + [j + i * 2 for j in t]
        # print(t)
        dif = float('inf')
        res = 0
        
for b in baseCosts:
            for c in t:
                tt = b + c
                # print(b, c, tt)
                if abs(target - tt) < dif or (abs(target - tt) == dif and tt < res):
                    dif = abs(target - tt)
                    res = tt
        return res
