class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        self.ret = min(baseCosts)
        
        def helper(curr, i):
            
            if abs(curr - target) < abs(self.ret - target):
                self.ret = curr
            if abs(curr - target) == abs(self.ret - target):
                self.ret = min(curr, self.ret)
            
            if i == len(toppingCosts):
                return 
            
            helper(curr, i+1)
            helper(curr + toppingCosts[i], i+1)
            helper(curr + 2 * toppingCosts[i], i+1)

        
        for base in baseCosts:
            helper(base, 0)
        
        return self.ret
