class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        self.ans = float('inf')
        def dfs(idx,cnt):
            if idx == len(toppingCosts):
                if abs(cnt-target) < abs(self.ans-target):
                    self.ans = cnt
                elif abs(cnt-target) == abs(self.ans-target):
                    self.ans = min(self.ans,cnt)
                return
            
            for amount in range(3):
                dfs(idx+1,cnt+amount*toppingCosts[idx])
            
        for baseCost in baseCosts:
            dfs(0,baseCost)
        
        return self.ans
