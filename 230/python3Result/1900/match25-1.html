class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        
        res = baseCosts[0]
        def dfs(j, cost):
            nonlocal res
            if abs
(cost - target) < abs(res - target):
                res = cost
            if cost >= target or j == len(toppingCosts):
                return

            dfs(j + 1, cost)
            dfs(j + 1, cost + toppingCosts[j])
            dfs(j + 1, cost + 2 * toppingCosts[j])
        
        for i in range(len(baseCosts)):
            dfs(0, baseCosts[i])
        
        return res
