class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        self.res = float('inf')
        def dfs(i, c):
            if i == len(toppingCosts):
                if abs(c - target) < abs(self.res - target) or (abs(c - target) == abs(self.res - target) and c < target):
                    self.res = c
                return
            dfs(i + 1, c)
            dfs(i + 1, c + toppingCosts[i])
            dfs(i + 1, c + toppingCosts[i] * 2)
        for b in baseCosts:
            dfs(0, b)
        return self.res
