class Solution:
    def closer(self, target: int, a: int, b: int) -> int:
        # returns the closer one to target
        ta = abs(target-a)
        tb = abs(target-b)
        if (ta < tb):
            return a
        elif (ta > tb):
            return b
        else:
            return min(a,b)
    def toppingSet(self, ts: List[int]): #set(int)
        ret = {0}
        tmp = set()
        for t in ts:
            tmp = set()
            for r in ret:
                tmp.add(r+t)
                tmp.add(r+t+t)
            ret |= tmp
        return ret
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        #brute force
        cloest = baseCosts[0]
        ts = self.toppingSet(toppingCosts) # topping set
        for b in baseCosts:
            cloest = self.closer(target, cloest, b)
            if b > target:
                continue
            for i in range(target):
                if target - i - b in ts:
                    cloest = self.closer(target, cloest, target - i)
                    break
                elif target + i - b in ts:
                    cloest = self.closer(target, cloest, target + i)
                    break
        return cloest
