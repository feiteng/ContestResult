import bisect

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        posTops = set()
        posTops.add(0)
        for top in toppingCosts:
            temp = set()
            for exist in posTops:
                temp.add(exist+top)
                temp.add(exist+2*top)
            posTops |= temp    
            
        out = float("inf")
        # tops = sorted(posToppings)
        for cost in baseCosts:
            for top in posTops:
                if abs(target - (cost+top)) < abs(out-target):
                    out = cost+top
                elif abs(target - (cost+top)) == abs(out-target) and cost+top < out:
                    out = cost + top 
            
        return out
