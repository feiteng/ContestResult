class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        cands = set()

        def _backtrack(i, acc):
            if i == len(toppingCosts):
                cands.add(acc)
                return

            _backtrack(i + 1, acc)
            _backtrack(i + 1, acc + toppingCosts[i])
            _backtrack(i + 1, acc + toppingCosts[i] * 2)

        _backtrack(0, 0)
        # print('cands:', cands)
        cands = sorted(list(cands))

        ext = float('inf')
        gap = float('inf')
        for base in baseCosts:
            for cand in cands:
                cur_gap = abs(base + cand - target)
                if cur_gap < gap:
                    gap = cur_gap
                    ext = base + cand
        return ext
