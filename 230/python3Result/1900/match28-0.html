class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        res = baseCosts[0]
        for baseCost in baseCosts:
            cost = self.dfs(toppingCosts, 0, target, baseCost)
            if abs(cost-target) < abs(res-target):
                res = cost
            elif abs(cost-target) == abs(res-target):
                res = min(res, cost)
        return res
    
    def dfs(self, toppingCosts, idx, target, curr):
        # if curr >= target:
            # return curr
        if idx == len(toppingCosts):
            return curr
        
        res = curr
        for i in range(3):
            cost = self.dfs(toppingCosts, idx + 1, target, curr + i*toppingCosts[idx])
            if abs(cost-target) < abs(res-target):
                res = cost
            elif abs(cost-target) == abs(res-target):
                res = min(res, cost)
        return res
