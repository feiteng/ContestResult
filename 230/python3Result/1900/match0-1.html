class Solution:
    def closestCost(self, base: List[int], topping: List[int], target: int) -> int:
        n, m = len(base), len(topping)
        res = base[0]
        
        def brute(i, curr):
            nonlocal res
            if i == m:
                if abs(target - curr) < abs(target - res) or (abs(target - curr) == abs(target - res) and curr < res):
                    res = curr
                return
            brute(i+1, curr)
            brute(i+1, curr + topping[i])
            brute(i+1, curr + topping[i]*2)
            
        for b in base:
            brute(0, b)
        
        return res
