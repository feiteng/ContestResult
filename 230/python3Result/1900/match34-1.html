class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        ts = [0]
        for c in toppingCosts: 
            ts.extend([c + x for x in ts])
            ts.extend([c + x for x in ts])
        ts 
= set(ts)
        ps = [x + b for x in ts for b in baseCosts]
        ps.sort(key=lambda x: (abs(target - x), x))
        return ps[0]
