class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        bset = set(baseCosts) # LEN<10
        for i in toppingCosts:
            for j in list(bset):
                bset.add(i+j)
                bset.add(i+i+j)
        return min(bset, key = lambda x: (abs(x-target), x))
