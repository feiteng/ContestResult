class Solution:
    def closestCost(self, costs: List[int], topp: List[int], target: int) -> int:
        vals = set()
        def rec(idx, current):
            if idx >= len(topp):
                vals.add(current)
                return 
            for i in range(3):
                rec(idx + 1, current + topp[idx] * i)
            return
        
        rec
(0, 0)
        ret = float('inf')
        min_d = float('inf')
        for c in costs:
            for v in vals:
                d = abs(target - c - v)
                if d == min_d and c + v < ret:
                    ret = c + v
                elif d < min_d:
                    ret = c +v 
                    min_d = d
        return ret
