class Solution:
    def closestCost(self, bc: List[int], tc: List[int], target: int) -> int:
        ret, diff = float("inf"), float("inf")
        def helper(p, i):
            nonlocal ret, diff
            if p > target and p - target >= diff: return
            if i == len(tc):
                if abs(target - p) < diff:
                    #print(p, i)
                    diff = abs(target - p) 
                    ret = p
                return
            helper(p, i + 1)
            helper(p + tc[i], i + 1)
            helper(p + tc[i] * 2, i + 1)
        
        for c in bc:
            helper(c, 0)
            #print("%%", ret, diff)
            
        return ret
