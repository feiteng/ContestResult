class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        t = {0}
        for topping in toppingCosts:
            t |= {c+topping for c in t} | {c
+2*topping for c in t}
        
        res = float("inf")
        for b in baseCosts:
            for tp in t:
                if abs(b+tp-target) < abs(res-target):
                    res = b+tp
                elif abs(b+tp-target) == abs(res-target) and b+tp < res:
                    res = b+tp
        return res
