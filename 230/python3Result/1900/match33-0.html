class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        base = set(baseCosts)
        
        #for i in range(len(baseCosts)):
        #    for j in range(i + 1, len(baseCosts)):
        #        base.add(baseCosts[i] + baseCosts[j])
        
        #top = set(toppingCosts)
        top = set()
        top.add(0)

        for t in toppingCosts:
            for topOld in list(top):
                top.add(topOld + t)
                top.add(topOld + 2 * t)

        #print(top, base)
        
        bestDiff = math.inf
        bestVal = math.inf
        for b in base:
            for t in top:
                total = b + t
                #print(total)
                if abs(total - target) < bestDiff or (abs(total - target) == bestDiff and total < bestVal):
                    bestDiff = abs(total - target)
                    bestVal = total
        return bestVal
