class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        m, n = len(baseCosts), len(toppingCosts)
        ans = baseCosts[0]
        def dfs(index, curCost):
            nonlocal ans
            if index == n:
                if abs(target - curCost) < abs(target-ans) or abs
(target - curCost) == abs(target-ans) and curCost < ans:
                    ans = curCost
                return
            dfs(index+1, curCost)
            dfs(index+1, curCost + toppingCosts[index])
            dfs(index+1, curCost + toppingCosts[index]*2)
        for baseCost in baseCosts:
            dfs(0, baseCost)
        return ans
