class Solution:
    def closestCost(self, bc: List[int], tc: List[int], target: int) -> int:
        self.res = inf
        self.diff = inf
        M = len(tc)
        def backtrack(i, cost):
            if i == M:
                # self.res = min(self.res)
                if abs(cost-target) < self.diff:
                    self.res = cost
                    self
.diff = abs(cost-target)
                elif abs(cost-target) == self.diff:
                    self.res = min(self.res, cost)
                return
            for j in range(3):
                backtrack(i+1, cost+tc[i]*j)
                
        for bcc in bc:
            backtrack(0, bcc)
            # print(self.res)
            
        return self.res
