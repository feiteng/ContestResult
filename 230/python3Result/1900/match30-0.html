class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        possibleCosts = set([0])
        for x in toppingCosts:
            new = set([])
            for old in possibleCosts:
                new.add(old + x)
                new.add(old + 2 * x)
            new = set([newCost for newCost in new if newCost <= 2 * target])
            possibleCosts = possibleCosts.union(new)
        minDist = float('inf')
        for toppingCost in possibleCosts:
            for baseCost in baseCosts:
                totalCost = baseCost + toppingCost
                if abs(target - totalCost) <= minDist:
                    if abs(target - totalCost) == minDist:
                        if totalCost <= target:
                            res = totalCost
                    else:
                        minDist = abs(target - totalCost)
                        res = totalCost
        return res
