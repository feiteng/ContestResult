class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        self.nearest = float('inf')
        def DFS(cur: int, i: int):
            if i == len(toppingCosts):
                if (abs(target - cur) < abs(target - self.nearest)) or \
                    ((abs(target - cur) == abs(target - self.nearest)) and cur < self.nearest):
                    self.nearest = cur
            else:
                DFS(cur, i + 1)
                DFS(cur + toppingCosts[i], i + 1)
                DFS(cur + toppingCosts[i] * 2, i + 1)
        for b in baseCosts:
            DFS(b, 0)
        return self.nearest
