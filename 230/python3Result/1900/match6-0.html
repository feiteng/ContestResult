class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        n, m = len(baseCosts), len(toppingCosts)
        ans = [None]

        def dfs(cost, t_type):
            if t_type == m:
                if ans[0] is None or abs(target - cost) < abs(target - ans[0]):
                    ans[0] = cost
                elif abs(target - cost) == abs(target - ans[0]):
                    ans[0] = min(cost, ans[0])
                return
            for i in range(3):
                dfs(cost + i * toppingCosts[t_type], t_type + 1)

        for i in range(n):
            dfs(baseCosts[i], 0)
        
        return ans[0]
