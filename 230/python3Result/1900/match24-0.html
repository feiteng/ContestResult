class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        costs = set(baseCosts)
        for toppingCost in toppingCosts:
            new_costs = set()
            for cost in costs:
                new_costs.add(cost + toppingCost)
                new_costs.add(cost + toppingCost * 2)
            costs.update(new_costs)
            
        best = math.inf
        for cost in costs:
            best_distance = abs(target - best)
            new_distance = abs(target - cost)
            if new_distance < best_distance:
                best = cost
            elif new_distance == best_distance and cost < best:
                best = cost
                
        return best
