class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        def p3(L):
            if len(L)==0:
                return [0]
            elif len(L)==1:
                return [0,L[0], 2*L[0]]
            else:
                n=len(L)
                A=p3(L[:n//2])
                B=p3(L[n//2:])
                return [x+y for x in A for y in B]
        Y=p3(toppingCosts)
        L=[x+y for x in baseCosts for y in Y]
        L.sort(key=lambda x:(abs(target-x),x))
        return L[0]
