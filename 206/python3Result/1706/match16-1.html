class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        
        costs = [[0] * len(points) for _ in range(len(points))]
        
        n = len(points)
        
        for i in range(n):
            for j in range(i+1, n):
                if i != j:
                    costs[i][j] 
= abs(points[i][0]-points[j][0]) + abs(points[i][1]-points[j][1])
                    costs[j][i] = abs(points[i][0]-points[j][0]) + abs(points[i][1]-points[j][1])
        
        def min_key(key, mstSet):
            min = float('inf')
            
for v in range(n):
                if key[v] < min and mstSet[v] == False:
                    min = key[v]
                    min_index = v
            return min_index

        key = [float('inf')] * n
        parent = [None] * n

        key[0] = 0
        mstSet 
= [False] * n

        parent[0] = -1

        for cout in range(n):
            u = min_key(key, mstSet)
            mstSet[u] = True

            for v in range(n):
                if costs[u][v] > 0 and mstSet[v] == False and key[v] > costs[u][v]:
                    key[v] = costs[u][v]
                    parent[v] = u

        ans = 0
        for i in range(1,n):
            ans += costs[i][parent[i]]
            # print(parent[i], "-", i, "\t", costs[i][ parent[i]])
            
        return ans
