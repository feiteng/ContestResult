class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        
        n = len(points)
        if n == 1:
            return 0
        
        q = []
        for i in range(n):
            for j in range(i + 1, n):
                x1, y1 = points[i]
                x2, y2 
= points[j]
                heapq.heappush(q, (abs(x1 - x2) + abs(y1 - y2), i, j))
                
        parent = list(range(n))
        
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        res, num = 0, 0
        while q:
            cost, i, j = heapq.heappop(q)
            rooti, rootj = find(i), find(j)
            if rooti != rootj:
                parent[rooti] = rootj
                res += cost
                num += 1
                
            if num == n-1:
                break
        return res
