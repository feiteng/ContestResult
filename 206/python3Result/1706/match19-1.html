class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        ds = []
        for i in range(n):
            ds.append([0 for _ in range(n)])
        
for i in range(n):
            for j in range(i):
                d = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])
                ds[i][j] = d
                ds[j][i] = d
        M = 1000000000
        dists = [0]
        used = [False]
        for i in range(1, n):
            dists.append(M)
            used.append(False)
        s = 0
        for _ in range(n):
            m, mi 
= M, -1
            for i in range(n):
                if dists[i] < m and not used[i]:
                    m = dists[i]
                    mi = i
            used[mi] = True
            s += dists[mi]
            
for i in range(n):
                if not used[i] and dists[i] > ds[mi][i]:
                    dists[i] = ds[mi][i]
        return s
