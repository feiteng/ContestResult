class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        edges = []
        for i in range(n):
            for j in range(i + 1, n):
                x0, y0 = points[i]
                x1, y1 = points[j]
                dist = abs(x0 - x1) + abs(y0 - y1)
                edges.append((dist, i, j))
        edges.sort()
        ids = list(range(n))
        def find(i):
            if ids[i] != i:
                ids[i] = find(ids[i])
            return ids[i]
        def union(i, j):
            ids[j] = i
        ans = 0
        for d, i, j in edges:
            i, j = find(i), find(j)
            if i != j:
                ans += d
                union(i, j)
        return ans
