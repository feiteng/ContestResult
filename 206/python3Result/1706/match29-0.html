class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        dist = []
        for i in range(n-1):
            for j in range(i+1,n):
                d = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1])
                dist.append((i,j,d))
        dist.sort(key=lambda x: -x[2])

        connected = list(range(n))
        def find(i):
            if i!=connected[i]:
                connected[i] = find(connected[i])
            return connected[i]
        def merge(i,j):
            i, j = find(i), find(j)
            connected[i] = j
            return
        count = 0
        res = 0
        while count<n-1:
            i, j, d = dist.pop()
            if find(i)==find(j):
                continue
            res += d
            count += 1
            merge(i,j)
        return res
