class Solution:
    def minCostConnectPoints(self, p: List[List[int]]) -> int:
        ans=0
        n
=len(p)
        edges=[]
        for i in range(n):
            for j in range(i+1,n):
                edges.append([i,j,abs(p[i][0]-p[j][0])+abs(p[i][1]-p[j][1])])
        edges.sort(key=lambda x:x[2])
        ufs={i:i for i in range
(n)}
        def find(x):
            if x==ufs[x]:
                return x
            ufs[x]=find(ufs[x])
            return ufs[x]
        for i,j,c in edges:
            ua,ub=find(i),find(j)
            if ua!=ub:
                ans+=c
                ufs[ua]=ub
        return ans
