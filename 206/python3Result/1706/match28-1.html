class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        dist = []
        for i in range(n):
            for j in range(i+1, n):
                x, y = points[i]
                x2, y2 = points[j]
                dist.append([abs(x-x2)+abs(
y-y2), i, j])

        dist.sort()
        parent = list(range(n))
        def find(x):
            if x != parent[x]:
                parent[x] = find(parent[x])
            return parent[x]
        
        total_dist = 0
        for tup in dist:
            d, i, j = tup
            pi, pj = find(i), find(j)
            if pi != pj:
                total_dist += d
                parent[pi] = pj
        return total_dist
