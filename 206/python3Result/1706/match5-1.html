class Solution:
    def minCostConnectPoints(self, ps: List[List[int]]) -> int:
        n = len(ps)
        ls 
= []
        for i in range(n) :
            for j in range(i+1,n) :
                dis = abs(ps[i][0]-ps[j][0]) + abs(ps[i][1]-ps[j][1])
                ls.append((dis,i,j))
        
        fa = [i for i in range
(n)]
        
        def ff(x) :
            if fa[x] == x :
                return x
            fa[x] = ff(fa[x])
            return fa[x]
        
        res = 0
        for d,u,v in sorted(ls) :
            x,y 
= ff(u),ff(v)
            if x != y :
                res += d
                fa[x] = y
            
        return res
