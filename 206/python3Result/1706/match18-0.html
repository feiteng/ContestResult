class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        le = len(points)
        li = []
        for i in range(le):
            for j in range(i):
                li.append((abs(points[i][0]-points[j][0])+ abs(points[i][1]-points[j][1]), i,j))
        li.sort()
        res = 0
        rep = [i for i in range(le)]
        def getRep(r):
            if rep[r]==r:
                return r
            rep[r]=getRep(rep[r])
            return rep[r]
        for d,a,b in li:
            if getRep(a)==getRep(b):
                continue
            rep[getRep(a)] = getRep(b)
            res+=d
        return res
