class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
            
        n = len(points)
        pq = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1]) 
                heapq.heappush(pq, (dist, i, j))
        parent 
= list(range(n))
        def find(pu):
            if parent[pu] != pu:
                parent[pu] = find(parent[pu])
            return parent[pu]
        visited = set()
        res = 0
        count = 0
        while pq:
            cur, u, v = heapq.heappop(pq)
            pu = find(u)
            pv = find(v)
            if pu == pv:
                continue
            parent[pu] = pv
            res += cur
            count += 1
            if count == n - 1:
                break
        return res
