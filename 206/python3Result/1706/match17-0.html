class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        m = len(points)
        graph=[[0 for i in range(m)] for j in range(m)]
        for i in range(m):
            for j in range(m):
                if (i!=j):
                    dist = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1])
                    graph[i][j]=dist
        
        def minnode(n, keyval, mstset): 
            mini = 999999999999
            mini_index = None

            for i in range(n):
                if (mstset[i] == False and keyval[i] < mini):
                    mini = keyval[i]
                    mini_index = i
            return mini_index
        
        parent = [None] * m
        keyval = [None] * m
        mstset = [None] * m
        for i in range(m): 
            keyval[i] = 9999999999999
            mstset[i] = False

        parent[0] = -1
        keyval[0] = 0

        for i in range(m - 1): 

            u = minnode(m, keyval, mstset)  

            mstset[u] = True

            for v in range(m): 
                if (graph[u][v] and mstset[v] == False and graph[u][v] < keyval[v]):  
                    keyval[v] = graph[u][v]  
                    parent[v] = u 

        cost = 0
        for i in range(1, m): 
            cost += graph[parent[i]][i]  
        return cost
