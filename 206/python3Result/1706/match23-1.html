class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        mst, edges = [], []  
        reps = [i for i in range(n)]
        for vi in range(n):
            for vj in range(n):
                if vi==vj:
                    continue
                edges
.append((abs(points[vi][0]-points[vj][0])+abs(points[vi][1]-points[vj][1]),vi,vj))
        edges.sort()
        ans = 0
        for weight, vi, vj in edges:
            if reps[vi] != reps
[vj]:
                ans+=weight
                repi, repj = reps[vi], reps[vj]
                for v in range(n):
                    if reps[v] == repj:
                        reps[v] = repi
        return ans
