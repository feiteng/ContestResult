class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        if not points:
            return 0
        n = len(points)
        dis_matrix = [[0 for _ in range(n)] for _ in range(n)]
        for i in range(n):
            for j in range(i + 1):
                dis = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])
                dis_matrix[i][j] = dis
                dis_matrix[j][i] = dis

        vis = [False] * n
        dis = [0x7fffffffff] * n
        dis[0] = 0
        vis[0] = True
        cur = 0

        for i in range(n):
            for j in range(n):
                if not vis[j] and dis[j] > dis_matrix[cur][j]:
                    dis[j] = dis_matrix[cur][j]

            min_dis = 0x7fffffffff
            for j in range(n):
                if not vis[j] and dis[j] < min_dis:
                    min_dis = dis[j]
                    cur = j
            vis[cur] = True
        return sum(dis)
