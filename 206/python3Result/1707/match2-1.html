class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        s = list(map(int, s))
        t = list(map(int, t))
        s_queues = defaultdict(deque)
        for i, num in enumerate(s):
            s_queues[num].append(i)
        t_counts = Counter(t)
        for num, count in t_counts.items():
            if len(s_queues[num]) != count:
                return False
        for idx, num in enumerate(t):
            rank 
= s_queues[num][0]
            for smaller_num in range(num):
                if s_queues[smaller_num] and s_queues[smaller_num][0] <= rank:
                    return False
            s_queues[num].popleft()
        return True
