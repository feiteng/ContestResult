class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        cs = Counter(s)
        ct = Counter(t)
        for i in cs.keys():
            if cs[i] != ct[i]:
                return False
        for i in ct.keys():
            if cs[i] != ct[i]:
                return False
        sx = [deque() for i in range(10)]
        for i, c in enumerate(s):
            pos = ord(c) - ord('0')
            sx[pos].append(i)
        
        for c in t:
            v = ord(c) - ord('0')
            for i in range(v):
                if sx[i] and sx[i][0] < sx[v][0]:
                    return False
            sx[v].popleft()
        return True
