class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        
        lists = collections.defaultdict(list)
        for i, c in enumerate(s):
            lists[c].append(i)
            
        for i in range(len(s)):
            if not lists[t[i]]:
                return False
            first = lists[t[i]][0]
            for c, l in lists.items():
                if c < t[i] and l and l[0] < first:
                    return False
            lists[t[i]].pop(0)
            
        return True
