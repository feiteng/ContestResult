class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        s_pos = defaultdict(deque)
        for i, c in enumerate(s):
            s_pos[int(c)].append(i)
        for i, c in enumerate(t):
            if not s_pos[int(c)]:
                return False
            pos = s_pos[int(c)].popleft()
            for x in range(int(c)):
                if s_pos[x] and s_pos[x][0] < pos:
                    return False
        return True
