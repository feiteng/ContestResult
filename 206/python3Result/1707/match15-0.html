class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        s = [int(c) for c in s]
        t = [int(c) for c in t]
        index = [[] for _ in range(10)]
        for (i, c) in enumerate(s):
            index[c].append(i)
        for i in range(10):
            index[i] = index[i][::-1]
        #print(s, t)
        for (i, c) in enumerate(t):
            #print(i, c, index)
            if index[c] == []:
                #print("A")
                return False
            k = index[c].pop()
            for j in range(c):
                if index[j] != [] and index[j][-1] < k:
                    #print("B", j)
                    return
        return True
