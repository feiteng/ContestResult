class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        idx = [[] for i in range(10)]
        for i, c in enumerate(s):
            idx[ord(c) - 48].append(i)
        for i, li in enumerate(idx):
            idx[i] = li[::-1]

        for c in t:
            i = ord(c) - 48
            if not idx[i]:
                return False
            tar = idx[i][-1]
            for j in range(i):
                if idx[j] and idx[j][-1] < tar:
                    return False
            idx[i].pop()
        return True
