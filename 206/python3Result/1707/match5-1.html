import collections


class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        ss = list(s)
        tt = list(t)
        tt.reverse()
        d = collections.defaultdict(list)
        for i in range(len(t)):
            d[ss[i]].append(i)
        for i in range(len(t)):
            c = tt[i]
            if not d[c]:
                return False
            for cc in "0123456789":
                
if cc > c and d[cc] and d[cc][-1] > d[c][-1]:
                    return False
            d[c].pop()
        return True
