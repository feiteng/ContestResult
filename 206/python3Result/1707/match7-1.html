class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        N = len(s)
        arr = [collections.deque() for _ in range(10)]
        
        for index, x in enumerate(s):
            arr[int(x)].append(index)
            
        for x in t:
            x = int(x)
            
            if len(arr[x]) == 0:
                return False
            
            for j in range(x):
                if len(arr
[j]) > 0 and arr[j][0] < arr[x][0]:
                    return False
            arr[x].popleft()
        
        return True
