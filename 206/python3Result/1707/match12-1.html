class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        if collections.Counter(s) != collections.Counter(t):
            return False
        
        s_chars = [collections.deque() for _ in range(10)]
        
for i, c in enumerate(s):
            s_chars[int(c)].append(i)
            
        for c in t:
            c = int(c)
            spot = s_chars[c].popleft()
            for c_i in range(c):
                if len(s_chars[c_i]) and s_chars[c_i][0] < spot:
                    return False
        
        return True
