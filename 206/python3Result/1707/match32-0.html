class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        if sorted(s) != sorted(t):
            return False
        n = len(s)
        cnt = [0 for i in range(10)]
        pos = [[] for i in range(10)]
        for i in range(n):
            pos[int(s[i])].append(i)
        for i in range(10):
            pos[i].append(n + 1)
        for i in range(n):
            x = int(t[i])
            need = pos[x][cnt[x]]
            cnt[x] += 1
            # print([pos[i][cnt[i]] for i in range(10)])
            for j in range(0, x):
                if pos[j][cnt[j]] < need:
                    return False
        # print()
        return True
