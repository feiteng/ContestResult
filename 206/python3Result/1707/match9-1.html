from collections import Counter,defaultdict,deque
class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        if Counter(s)!=Counter(t):
            return False
        
        nxt=defaultdict(deque)
        
        for i,d in enumerate(s):
            d=int(d)
            nxt[d]
.append(i)
        
        
        for d in t:
            d=int(d)
            limit=nxt[d].popleft()
            for i in range(d):
                if nxt[i] and nxt[i][0]<limit:
                    return False
        return True
