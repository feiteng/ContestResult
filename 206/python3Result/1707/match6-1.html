class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        sd=[[] for _ in range(10)]
        for i, c in enumerate(s):
            sd[int(c)].append(i)
        for i in reversed(range(len(t))):
            c = int(t[i])
            if not sd[c]:
                return False
            
for c1 in range(c+1, 10):
                if sd[c1] and sd[c1][-1] > sd[c][-1]: return False
            sd[c].pop()
        return True
