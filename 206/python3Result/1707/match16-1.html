class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        dic = [[] for _ in range(10)]
        for i, c in enumerate(s):
            dic[int(c)].append(i)
        N = len(s)
        for i in range(N - 1, -1, -1):
            ni = int(t[i])
            if not dic[ni]:
                return False
            idx 
= dic[ni].pop()
            for j in range(ni + 1, 10):
                if dic[j] and dic[j][-1] > idx:
                    return False
        return True
