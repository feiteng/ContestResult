class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        c1 
= [0] * 10
        c2 = [0] * 10
        for c in s:
            c1[int(c)] += 1
        for c in t:
            c2[int(c)] += 1
        
        for i in range(10):
            if c1[i] != c2[i]:
                return False
        
        def getRePair(c, s):
            ans = [[0 for _ in range(10)] for _ in range(10)]
        
            for ch in s:
                d = int(ch)
                c[d] -= 1
                for i in range(d):
                    ans[d][i] += c[i]
            
            return ans

        p1 = getRePair(c1, s)
        p2 = getRePair(c2, t)

        for i in range(10):
            
for j in range(10):
                if p1[i][j] < p2[i][j] :
                    return False
        
        return True
