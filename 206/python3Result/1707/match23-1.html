class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        sz = len(s)
        s, t = [ord(c) - ord('0') for c in s], [ord(c) - ord('0') for c in t]

        pos = [[] for i in range(10)]
        for i in range(sz):
            pos[s[i]].append(i)

        for i in range(sz-1, -1, -1):
            need = t[i]
            if len(pos[need]) <= 0:
                return False
            pv 
= pos[need][-1]

            for j in range(need+1, 10):
                if len(pos[j]) > 0 and pos[j][-1] > pv:
                    return False

            pos[need].pop()

        return True
