import bisect
from collections import defaultdict

class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        counter_s = defaultdict(int)
        for c in s:
            counter_s[int(c)] += 1
        counter_t = defaultdict(int)

        prev = 0
        is_sorted = True
        is_same = True

        for i, c in enumerate(t):
            counter_t[int(c)] += 1
            if int(c) < prev:
                is_sorted = False
            prev = int(c)
            if t[i] != s[i]:
                is_same = False
        
        for i in range(10):
            if counter_s[i] != counter_t[i]:
                # print("unequal num")
                return False

        if is_sorted or is_same:
            return True
        
        counts_t = [[0 for _ in range(10)] for _ in range(len(s))]
        prev_count = [0 for _ in range(10)]
        t_pos = defaultdict(list)
        for i, c in enumerate(t):
            counts_t[i] = prev_count[::]
            counts_t[i][int(c)] += 1
            prev_count = counts_t[i]
            t_pos[int(c)].append(i)

        count_s = [0 for _ in range(10)]
        for digit in s:
            d = int(digit)
            t_pos_d 
= t_pos[d][count_s[d]]
            count_s[d] += 1
            for smaller_d in range(d):
                if count_s[smaller_d] > counts_t[t_pos_d][smaller_d]:
                    # print("d", d, "count_s", count_s, "counts_t", counts_t)
                    return False
        
        return True
