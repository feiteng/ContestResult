class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        cnt = [0] * 10
        for ch1, ch2 in zip(s, t):
            cnt[int(ch1)] += 1
            cnt[int(ch2)] -= 1
        
        for i in range(10):
            if cnt[i] != 0:
                return False
        
        stks = [[] for _ in range(10)]
        for i, ch in enumerate(s):
            ch = int(ch)
            stks[ch].append(i)
        
        for ch in t[::-1]:
            ch = int(ch)
            idx = stks[ch].pop()
            for i in range(ch + 1, 10):
                if stks[i] and stks[i][-1] > idx:
                    return False
        return True
