class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        
        def check(begin, end) :
            if begin > end :
                return True
            if begin == end :
                return s[begin] == t[begin]
            
            for i in range(end, begin-1, -1) :
                k = ord(t[i]) - ord('0')
                if not a[k] :
                    return False
                pos = a[k][-1]
                for j in range(k+1, 10) :
                    if a[j] and a[j][-1] > pos :
                        return False
                a[k].pop()
            return True            
        
        start = 0
        n = len(s)
        diff = [0] * 10
        a = [[] for _ in range(10)]
        for i in range(n) :
            k1 = ord(s[i]) - ord('0')
            diff[k1] -= 1
            k2 = ord(t[i]) - ord('0')
            diff[k2] += 1
            
            a[k1].append(i)
            
            equal = True
            for j in range(10) :
                if diff[j] != 0 :
                    equal = False
                    break
            
            if equal :
                if not check(start, i):
                    return False
                else :
                    start = i+1
        
        for j in range(10) :
            if diff[j] != 0 :
                return False
        
        return check(start, n-1)
