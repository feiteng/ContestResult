class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        s = [ord(c) - ord('0') for c in s]
        t = [ord(c) - ord('0') for c in t]
        k = [[] for _ in range(10)]
        for i, n in enumerate(s):
            k[n].append(i)
        for n in t:
            if not k[n]:
                return False
            p = k[n][0]
            k[n] = k[n][1:]
            for i in range(n):
                if k[i] and k[i][0] < p:
                    return False
        return True
