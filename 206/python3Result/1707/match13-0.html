class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        c_count = collections.Counter(s)
        if c_count != collections.Counter(t):
            return False
        s_seq = collections.defaultdict(collections.deque)
        for i,c in enumerate(s):
            s_seq[int(c)].append(i)
        for c in t:
            c = int(c)
            cur = s_seq[c].popleft()
            for i in range(0,c):
                if s_seq[i] and s_seq[i][0]<cur:
                    return False
        return True
