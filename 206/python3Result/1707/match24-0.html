class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        s_positions = {str(i): deque([]) for i in range(10)}
        for i, c in enumerate(s): 
            s_positions[c].append(i)
        for c in t: 
            if len(s_positions[c]) == 0: 
                return False
            ic = int(c)
            for i in range(ic): 
                pos = s_positions[str(i)]
                if len(pos) > 0 and pos[0] < s_positions[c][0]: 
                    return False
            s_positions[c].popleft()
        return True
