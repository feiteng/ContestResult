class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        ord0 = ord('0')
        s = [ord(c) - ord0 for c in s]
        t = [ord(c) - ord0 for c in t]
        
        digitIdxList = [[] for i in range
(10)]
        for i, d in enumerate(s):
            digitIdxList[d].append(i)
        for i in range(len(s)-1, -1, -1):
            if not digitIdxList[t[i]]: return False
            dIdx = digitIdxList[t[i]][-1]
            #print(t[i], dIdx)
            for d in range(9, t[i], -1):
                if digitIdxList[d] and digitIdxList
[d][-1] > dIdx:
                    return False
            digitIdxList[t[i]].pop()
        return True
