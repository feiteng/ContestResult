class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        n = len(s)
        if not n == len(t):
            return False
        prefix = [[0] * 10 for _ in range(n + 1)]
        inds = [collections.deque() for _ in range(10)]
        used = [0] * 10
        for i in range(n):
            inds[int(s[i])].append(i)
            for j in range(10):
                prefix[i + 1][j] = prefix[i][j]
            prefix[i + 1][int(s[i])] += 1
        for i in range
(n):      
            c = int(t[i])
            if not inds[c]: 
                return False       
            for j in range(c):
                if prefix[inds[c][0] + 1][j] - used[j] > 0:
                    return False
            inds[c].popleft()      
            used[c] += 1      
        return True
