class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        t,s = s,t
        cntS = Counter(s)
        cntT = Counter(t)
        for x in cntS:
            if cntS[x] != cntT[x]:
                return False
        occur = [[] for i in range(10)]
        n = len(t)
        for i in range(n-1,-1,-1):
            num = ord(t[i]) - ord('0')
            occur[num].append(i)
        #print(occur)
        for ch in s:
            num = ord(ch) - ord('0')
            idx = occur[num][-1]
            for i in range
(num):
                if len(occur[i])>0 and occur[i][-1] < idx:
                    return False
            occur[num].pop()
        return True
