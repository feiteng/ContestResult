class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        qs = {i: [] for i in range(10)}
        for i, c in enumerate(s):
            c = int(c)
            qs[c].append(i)
        # now loop over t
        for c in t:
            c = int(c)
            if len(qs[c]) == 0:
                return False
            fv = qs[c][0]
            for tmpv in range(c):
                if len(qs[tmpv]) > 0 and qs[tmpv][0] < fv:
                    return False
            qs[c] = qs[c][1:]
        
        return True
