class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        
        level=[dict() for _ in range(n)]
        for i in range(n):
            for j in range(n-1):
                level[i][preferences[i][j]]=j
        result=0
        coup={}
        for a,b in pairs:
            coup[a]=b
            coup[b]=a
        for i in range(n):
            for j in range(level[i][coup[i]]):
                cur=preferences[i][j]
                if level[cur][i]<level[cur][coup[cur]]:
                    result+=1
                    break
        return result
