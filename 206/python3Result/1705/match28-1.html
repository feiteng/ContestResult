class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        value = [{} for _ in range(n)]
        
        for x, friends in enumerate(preferences):
            value[x] = {}
            for val, f in enumerate(friends):
                value
[x][f] = val
        
        partner = {}
        for x, y in pairs:
            partner[x] = y
            partner[y] = x
        
        ans = 0
        for x in range(n):
            for u in preferences[x]:
                if u == partner[x]:
                    break
                if value[u][x] < value[u][partner[u]]:
                    ans += 1
                    break
        return ans
