class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        prefer = {i:{n:j for j,n in enumerate(row)} for i,row in enumerate(preferences)}
        pairdict = {}
        for L,R in pairs:
            pairdict[L]=R
            pairdict[R]=L
        ans = 0
        for x,y in pairdict.items():
            for u,v in pairdict.items():
                if u==x:continue
                if u==y:continue
                if prefer[x][u]<prefer[x][y] and prefer[u][x]<prefer[u][v]:
                    ans += 1
                    break
        return ans
