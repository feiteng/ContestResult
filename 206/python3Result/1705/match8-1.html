class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        score = {i: {f: j for j, f in enumerate(tmp)} for i, tmp in enumerate(preferences)}
        graph = {}
        for a, b in pairs:
            graph[a] = b
            graph[b] = a
        ans = 0
        for a, b in graph.items():
            for i in range(score[a][b]):
                if score[preferences[a][i]]
[a] < score[preferences[a][i]][graph[preferences[a][i]]]:
                    # print(a, b, score[a][b], preferences[a][i])
                    ans += 1
                    break
        return ans
