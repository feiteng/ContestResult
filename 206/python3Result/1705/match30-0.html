class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        d = [{} for _ in range(n)]
        for friend1, p in enumerate(preferences):
            for friend2_rank, friend2 in enumerate(p):
                d[friend1][friend2] = friend2_rank
        
        d_pairs = {}
        for friend1, friend2 in pairs:
            d_pairs[friend1] = friend2
            d_pairs[friend2] = friend1
        
        ans = 0
        
        for f1, f2 in pairs:
            for preferred_friend in preferences[f1]:
                if preferred_friend == f2:
                    break
                their_friend = d_pairs[preferred_friend]
                if d[preferred_friend][their_friend] > d[preferred_friend][f1]:
                    ans += 1
                    break
            
            for preferred_friend in preferences[f2]:
                if preferred_friend == f1:
                    break
                their_friend = d_pairs[preferred_friend]
                if d[preferred_friend][their_friend] > d[preferred_friend][f2]:
                    ans += 1
                    break
                
        
        return ans
