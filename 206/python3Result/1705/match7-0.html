class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        rank = [[inf for _ in range(n)] for _ in range(n)]
        
        for i,pref in enumerate(preferences):
            for j,f in enumerate(pref):
                rank[i][f] = j
                
        def prefer(x, y, u, v):
            return rank[x][u] < rank[x][y] and rank[u][x] < rank[u][v]
           
        unhappy = set()
        for i,(x, y) in enumerate(pairs):
            for j,(u, v) in enumerate(pairs):
                if i == j:
                    continue
                if prefer(x, y, u, v) or prefer(x, y, v, u):
                    unhappy.add(x)
                if prefer(y, x, u, v) or prefer(y, x, v, u):
                    unhappy.add(y)
                    
        return len(unhappy)
