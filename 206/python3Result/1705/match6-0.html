class Solution:
    def unhappyFriends(self, n: int, A: List[List[int]], pairs: List[List[int]]) -> int:
      n2 = n // 2
      
      G = {}
      for i, a in enumerate(A):
        G[i] = {}
        for j, p in enumerate(a):
          G[i][p] = j  # person p is at j order of "i"'s friends
          
      seen = set()
      for i, (x, y) in enumerate(pairs):
        for j in range(i + 1, n2):
          u, v = pairs[j]
          if G[x][u] < G[x][y] and G[u][x] < G[u][v]:
            seen.add(x)
            seen.add(u)
          if G[x][v] < G[x][y] and G[v][x] < G[v][u]:
            seen.add(x)
            seen.add(v)
          if G[y][u] < G[y][x] and G[u][y] < G[u][v]:
            seen.add(y)
            seen.add(u)
          if G[y][v] < G[y][x] and G[v][y] < G[v][u]:
            seen.add(y)
            seen.add(v)
          
      return len(seen)
