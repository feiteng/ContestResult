import bisect

class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
#         However, this pairing may cause some of the friends to be unhappy. 
#         A friend x is unhappy if x is paired with y and there exists a friend u who is paired with v but:

#         x prefers u over y, and
#         u prefers x over v.
        
        preference_map = {}
        for i,row in enumerate(preferences):
            preference_map[i] = {}
            for j,cell in enumerate(row):
                preference_map[i][cell] = j
        
        pair_map = {}
        for x,y in pairs:
            pair_map[x] = y
            pair_map[y] = x
        
        res = 0
        pairs = [[x,y] for x,y in pairs] + [[y,x] for x,y in pairs]
        for x,y in pairs:
            for u in preferences[x]:  # for candidate friends
                if u == y:
                    break
                v = pair_map[u]
                # print(u,x,v)
                if preference_map[u][x] < preference_map[u][v]:
                    res += 1
                    break
                    
        return res
