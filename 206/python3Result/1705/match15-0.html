class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        pair_with = {}
        for x, y in pairs:
            pair_with[x] = y
            pair_with[y] = x
        ranks = [{} for _ in range(n)]
        for i, ps in enumerate(preferences):
            for rank, j in enumerate(ps):
                ranks[i][j] = rank
        ans = 0
        for x in range(n):
            y = pair_with[x]
            for u in range(n):
                if u == x or u == y:
                    continue
                v = pair_with[u]
                if ranks[x][u] < ranks[x][y] and ranks[u][x] < ranks[u][v]:
                    ans += 1
                    break
        return ans
