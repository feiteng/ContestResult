class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        memo = {}
        for u,v in pairs:
            memo[u] = v
            memo[v] = u
        res = 0
        for i in range(n):
            pair_n = memo[i]
            loc = preferences[i].index(pair_n)
            for prefer_n in preferences[i][:loc]:
                pair_m = memo[prefer_n]
                if preferences[prefer_n].index(i)<preferences[prefer_n].index(pair_m):
                    res += 1
                    break
        return res
