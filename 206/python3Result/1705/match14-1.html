class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        ranks=[]
        for i in range(n):
            ranks.append
({})
            for j in range(n-1):
                ranks[-1][preferences[i][j]]=j
        ret=0
        p=[0 for _ in range(n)]
        
for x,y in pairs:
            p[x]=y
            p[y]=x
        for i in range(n):
            for r in range(ranks[i][p[i]]):
                third=preferences[i][r]
                if ranks[third][i]<ranks[third][p[third]]:
                    ret+=1
                    break
        return ret
