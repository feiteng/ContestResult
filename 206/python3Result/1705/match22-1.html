from collections import *

class Solution:
    def unhappyFriends(self, n: int, l: List[List[int]], l1: List[List[int]]) -> int:
        
        
        d = defaultdict(dict)
        for i in range(n):
            for j in range(len(l[i])):
                d
[i][l[i][j]] = j
        
        pairs = {}
        for i in l1:
            pairs[i[0]] = i[1]
            pairs[i[1]] = i[0]
            
        
        ans = 0
        unhappy = {}
        for i in range(n):
            for j in range(n):
                if i == j or pairs[i] == j:continue
                
if d[i][pairs[i]] > d[i][j] and d[j][pairs[j]] > d[j][i]:
                    unhappy[i] = 1
                    # print(i, j)
                    ans += 1
        # print('\n\n')
        return len(unhappy)
