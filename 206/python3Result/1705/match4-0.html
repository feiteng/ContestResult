class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        preferences_dict = []
        for pt in preferences :
            preferences_dict.append({t:i for i, t in enumerate(pt)})
        # print(preferences_dict)
            
        not_happy = set()
        for i, (x, y) in enumerate(pairs) :
            for j in range(i+1, len(pairs)) :
                u, v = pairs[j] 
                if preferences_dict[x][u] < preferences_dict[x][y] and preferences_dict[u][x] < preferences_dict[u][v] :
                    not_happy.add(x)
                    not_happy.add(u)
                if preferences_dict[x][v] < preferences_dict[x][y] and preferences_dict[v][x] < preferences_dict[v][u] :
                    not_happy.add(x)
                    not_happy.add(v)
                if preferences_dict[y][u] < preferences_dict[y][x] and preferences_dict[u][y] < preferences_dict[u][v] :
                    not_happy.add(y)
                    not_happy.add(u)
                if preferences_dict[y][v] < preferences_dict[y][x] and preferences_dict[v][y] < preferences_dict[v][u] :
                    not_happy.add(y)
                    not_happy.add(v)
        # print(not_happy)
        return len(not_happy)
