class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        rank = {i:{} for i in range(n)}
        for i, pref in enumerate(preferences):
            for j, p in enumerate(pref):
                rank[i][p] = j
        friends = {}
        for x, y in pairs:
            friends[x] = y
            friends[y] = x
        res = 0 
        for x, y in pairs:
            for u in preferences[x]:
                if u == y: break 
                if rank[u][x] < rank[u][friends[u]]:
                    res += 1
                    break
            for u in preferences[y]:
                if u ==x: break
                if rank[u][y] < rank[u][friends[u]]:
                    res +=1 
                    break 
        return res
