class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        res = set()
        
        graph = [[0]*n for _ in range(n)]
        for i in range(len(preferences)):
            for j in range(len(preferences[i])):
                x, y = i, preferences[i][j]
                graph[x][y] = j
        
        # for i in graph:
        #     print(i)
        
        m = len(pairs)
        for i in range(m):
            for j in range(i+1, m):
                x, y = pairs[i]
                u, v 
= pairs[j]
                if graph[x][u] < graph[x][y] and graph[u][x] < graph[u][v]:
                    res.add(x); res.add(u)
                if graph[x][v] < graph[x][y] and graph[v][x] < graph[v][u]:
                    res.add(x); res.add(v)
                if graph[y][u] < graph[y][x] and graph[u][y] < graph[u][v]:
                    res.add(y); res.add(u)
                if graph[y][v] < graph[y][x] and graph[v][y] < graph[v][u]:
                    res.add(y); res.add(v)
        # print(res)
        return len(res)
