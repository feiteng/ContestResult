class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        rank=[[n]*n for i in range(n)]
        for i in range(n):
            for j in range(n-1):
                rank[i][preferences[i][j]]=j
        p=[0]*n
        for x in pairs:
            p[x[0]]=x[1]
            p[x[1]]=x[0]
        D=set()
        for i in range(n):
            for j in range(n):
                if rank[i][p[i]]>rank[i][j] and rank[j][p[j]]>rank[j][i]:
                    D.add(i)
                    D.add(j)
        return len(D)
