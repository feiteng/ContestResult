from collections import defaultdict


class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        
        pref = defaultdict(dict)
        
        for i in range(n):
            for j in range(n-1):
                pref[i][preferences[i][j]] = n - j
                
        p = {}
        for i,j in pairs:
            p[i] = j
            p[j] = i
                
        s = set()
                
        for x in range(n):
            for u in range(n):
                if x == u:
                    continue
                    
                
if pref[x][p[x]] < pref[x][u] and pref[u][p[u]] < pref[u][x]:
                    s.add(u)
                    s.add(x)
            
        return len(s)
