class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        pref = [{x:i for i,x in enumerate(p)} for p in preferences]
        happy = [0] * n
        np = len(pairs)
        
        
for i in range(np):
            for j in range(i):
                x,y = pairs[i]
                u,v = pairs[j]
                if pref[x][u] < pref[x][y] and pref[u][x] < pref[u][v]:
                    happy[x] = 1
                    happy[u] = 1

                y,x = pairs[i]
                u,v = pairs[j]
                
if pref[x][u] < pref[x][y] and pref[u][x] < pref[u][v]:
                    happy[x] = 1
                    happy[u] = 1

                x,y = pairs[i]
                v,u = pairs[j]
                
if pref[x][u] < pref[x][y] and pref[u][x] < pref[u][v]:
                    happy[x] = 1
                    happy[u] = 1

                y,x = pairs[i]
                v,u = pairs[j]
                
if pref[x][u] < pref[x][y] and pref[u][x] < pref[u][v]:
                    happy[x] = 1
                    happy[u] = 1

                    
        return sum(happy)
