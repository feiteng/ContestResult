class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        
        dic={}
        
        for i in range(n):
            for k,j in enumerate(preferences[i]):
                dic[(i,j)]=k+1
        
        #print(dic)
        
        S=set()
        
        for i in range(len(pairs)-1):
            for j in range(i+1,len(pairs)):
                
                pair1,pair2=pairs[i],pairs[j]
                
                x,y=pair1
                u,v=pair2
                
                
if dic[(x,y)]>dic[(x,u)] and dic[(u,v)]>dic[(u,x)]:
                    
                    S.add(x)
                    S.add(u)
                    
                if dic[(x,y)]>dic[(x,v)] and dic[(v,u)]>dic[(v,x)]:
                    
                    S.add(x)
                    S.add(v)
                    
                if dic[(y,x)]>dic[(y,u)] and dic[(u,v)]>dic[(u,y)]:
                    
                    S.add(y)
                    S.add(u)
                
                if dic[(y,x)]>dic[(y,v)] and dic[(v,u)]>dic[(v,y)]:
                    
                    S.add(y)
                    S.add(v)
        
        return len(S)
