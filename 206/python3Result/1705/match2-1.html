class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        kindness=[[0]*n for _ in range
(n)]
        for idy,pref in enumerate(preferences):
            for idx, friend in enumerate(pref):
                kindness[idy][friend]+=idx
        unhappy=set()
        for i in range(len(pairs)-1):
            for j in range(i+1,len(pairs)):
                x,y
=pairs[i]
                u,v=pairs[j]
                if kindness[x][u]<kindness[x][y] and kindness[u][x]<kindness[u][v]:
                    unhappy.add(x)
                    unhappy.add(u)
                if kindness[x][v]<kindness[x][y] and kindness[v][x]<kindness[v][u]:
                    unhappy.add(x)
                    unhappy.add(v)
                
if kindness[y][v]<kindness[y][x] and kindness[v][y]<kindness[v][u]:
                    unhappy.add(y)
                    unhappy.add(v)
                if kindness[y][u]<kindness[y][x] and kindness[u][y]<kindness[u][v]:
                    unhappy.add(y)
                    unhappy.add(u)
        return len(unhappy)
