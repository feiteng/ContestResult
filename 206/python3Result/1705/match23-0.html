class Solution:
    def unhappyFriends(self, n: int, preference: List[List[int]], pairs: List[List[int]]) -> int:
        unhappy = [0] * n
        m = len(pairs)
        pref = [[0] * n for _ in range(n)]
        for a in range(n):
            for i in range(n-1):
                b = preference[a][i]
                pref[a][b] = n - 1 - i
        for i in range(m):
            for j in range(i):
                x, y = pairs[j]
                u, v = pairs[i]
                if pref[x][u] > pref[x][y] and pref[u][x] > pref[u][v]:
                    unhappy[x] = unhappy[u] = 1
                if pref[x][v] > pref[x][y] and pref[v][x] > pref[v][u]:
                    unhappy[x] = unhappy[v] = 1
                if pref[y][v] > pref[y][x] and pref[v][y] > pref[v][u]:
                    unhappy[y] = unhappy[v] = 1
                if pref[y][u] > pref[y][x] and pref[u][y] > pref[u][v]:
                    unhappy[y] = unhappy[u] = 1
        return sum(unhappy)
