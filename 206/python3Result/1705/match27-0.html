class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        qj = {i:{} for i in range(n)}
        for i, lst in enumerate(preferences):
            for s, j in enumerate(lst):
                qj[i][j] = s
        
        pair = {}
        for i, j in pairs:
            pair[i] = j
            pair[j] = i
        
        cnt = 0
        for x in range(n):
            y = pair[x]
            for u in range(n):
                if u == x or u == y:
                    continue
                v = pair[u]
                if qj[x][u] < qj[x][y] and qj[u][x] < qj[u][v]:
                    cnt += 1
                    break
        return cnt
