class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        pref = [[-1 for i in range(n)] for j in range
(n)]
        for i, p in enumerate(preferences):
            for j, f in enumerate(p):
                pref[f][i] = j
        
        res = set()
        m = len(pairs)
        for i in range(m):
            for j in range(i+1, m):
                x, y = pairs[i]
                u, v = pairs[j]
                
if pref[u][x] < pref[y][x] and pref[x][u] < pref[v][u]:
                    #print('111', x, u)
                    res.add(x)
                    res.add(
u)
                y, x = pairs[i]
                v, u = pairs[j]
                if pref[u][x] < pref[y][x] and pref[x][u] < pref[v][u]:
                    #print('222', x, u)
                    res.add(x)
                    res.add(u)
                x, y = pairs[i]
                v, u = pairs[j]
                
if pref[u][x] < pref[y][x] and pref[x][u] < pref[v][u]:
                    #print('333', x, u)
                    res.add(x)
                    res.add(u)
                y, x = pairs[i]
                u, v = pairs[j]
                
if pref[u][x] < pref[y][x] and pref[x][u] < pref[v][u]:
                    #print('444', x, u)
                    res.add(x)
                    res.add(u)
        
        return len(res)
