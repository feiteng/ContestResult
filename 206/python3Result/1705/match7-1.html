class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        d = [[0 for _ in range(n)] for __ in range(n)]
        for i in range(n):
            for ind, j in enumerate(preferences[i]):
                d
[i][j] = n-ind

        def check(x, y, u, v):
            return d[x][u] > d[x][y] and d[u][x] > d[u][v]
        ans = set()
        for i, (x, y) in enumerate(pairs):
            for j, (u, v) in enumerate(pairs):
                if i != j:
                    
if check(x, y, u, v) or check(x, y, v, u):
                        ans.add(x)
                    if check(y, x, u, v) or check(y, x, v, u):
                        ans.add(y)

        return len(ans)
