class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs_list: List[List[int]]) -> int:
        pref_dict = [] 
        for pref_i in preferences:
            pref_dict.append({friend: pref for pref, friend in enumerate(
pref_i)})
        # print(pref_dict)     
        
        pairs = {}
        for x, y in pairs_list:
            pairs[x] = y
            pairs[y] = x
        # print(pairs)    
        
        cnt = 0
        for x in range(n):
            y = pairs[x]
            
            for u in range(n):
                if u == x or u == y: continue
                    
                v = pairs[u]
                # print(x, y, u, v)
                    
                if pref_dict[x][u] < pref_dict[x][y] and pref_dict[u][x] < pref_dict[u][v]:
                    cnt += 1
                    break
        return cnt
