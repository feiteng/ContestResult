class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        graph = collections.defaultdict(lambda: collections.defaultdict(int))
        for i, people in enumerate(preferences):
            for j, friend in enumerate(people):
                graph[i][friend] = j
        
        # now iterate pairs
        seen = set()
        cnt = 0
        m = len(pairs)
        for i in range(m):
            for j in range(m):
                if i == j: continue
                a, b = pairs[i]
                c, d = pairs[j]
                if graph[a][c] < graph[a][b] and graph[c][a] < graph[c][d]:
                    seen.add(a)
                    seen.add(c)
                if graph[b][d] < graph[b][a] and graph[d][b] < graph[d][c]:
                    seen.add(b)
                    seen.add(d)
                d, c = pairs[j]
                if graph[a][c] < graph[a][b] and graph[c][a] < graph[c][d]:
                    seen.add(a)
                    seen.add(c)
                if graph[b][d] < graph[b][a] and graph[d][b] < graph[d][c]:
                    seen.add(b)
                    seen.add(d)
        return len(seen)
