from collections import defaultdict

class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        
        d = {}
        for u, v in pairs:
            d[u] = v
            d[v] = u
            
        rank = defaultdict(dict)
        for i, pref in enumerate(preferences):
            for idx, j in enumerate(pref):
                rank[i][j] = idx
            
        res = 0
        for i, pref in enumerate(preferences):
            for j in pref:
                if j == d[i]:
                    break
                if rank
[j][d[j]] > rank[j][i]:
                    res += 1
                    break
        return res
