class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        
        prefer = defaultdict(set)
        for a,b in pairs:
            for c in preferences[a]:
                if c == b:
                    break
                prefer[a].add(c)
            for c in preferences[b]:
                if c == a:
                    break
                prefer[b].add(c)
                
        res = 0
        for a,b in prefer.items():
            for c in b:
                if c in prefer and a in prefer[c]:
                    res += 1
                    break
        
        return res
