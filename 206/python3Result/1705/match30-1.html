class Solution:
    def unhappyFriends(self, n: int, A: List[List[int]], pairs: List[List[int]]) -> int:
        tb = [[0]*n for _ in range(n)]
        for i in range(n):
            for j in range(n-1):
                tb[i][A[i][j]] 
= j
        ps = [0] * n
        for f1, f2 in pairs:
            ps[f1] = f2
            ps[f2] = f1
        ans = 0
        for i in range(n):
            for j in A[i]:
                if j == ps[i]: break
                if tb[j]
[ps[j]] > tb[j][i]:
                    ans += 1
                    break
        return ans
