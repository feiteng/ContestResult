class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        map_pre=[{} for i in preferences]
        for i,p in enumerate(preferences):
            for idx,j in enumerate(p):
                map_pre[i][j]=idx
        res=set()
        for i in range(len(pairs)):
            for j in range(i+1,len(pairs)):
                [x,y]=pairs[i]
                [u,v]=pairs[j]
                # print(x,y,u,v,map_pre[x][u],map_pre[x][y],map_pre[u][x],map_pre[u][v])
                if map_pre[x][u]<map_pre[x][y] and map_pre[u][x]<map_pre[u][v]:
                    res.add(x)
                    res.add(u)
                if map_pre[x][v]<map_pre[x][y] and map_pre[v][x]<map_pre[v][u]:
                    res.add(x)
                    res.add(v)
                if map_pre[y][v]<map_pre[y][x] and map_pre[v][y]<map_pre[v][u]:
                    res.add(y)
                    res.add(v)
                if map_pre[y][u]<map_pre[y][x] and map_pre[u][y]<map_pre[u][v]:
                    res.add(y)
                    res.add(u)
        return len(res)
