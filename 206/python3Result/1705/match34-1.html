class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        dic = {}
        for i,j in pairs:
            dic[i] = j
            dic[j] = i
        ans = 0
        for i,j in pairs:
            for u in preferences[i]:
                if u == j:
                    break
                v = dic[u]
                if preferences[u].index(i) < preferences[u].index(v):
                    ans += 1
                    break
            for u in preferences[j]:
                if u == i:
                    break
                v 
= dic[u]
                if preferences[u].index(j) < preferences[u].index(v):
                    ans += 1
                    break
        return ans
