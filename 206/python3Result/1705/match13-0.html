class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        p_list = [set() for _ in range(n)]
        for x, y in pairs:
            for z in preferences[x]:
                if z == y:
                    break
                p_list[x].add(z)
            for z in preferences[y]:
                if z == x:
                    break
                p_list[y].add(z)
        ans = 0
        for x, y_set in enumerate(p_list):
            for y in y_set:
                if x in p_list[y]:
                    ans += 1
                    break
        return ans
