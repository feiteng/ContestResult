class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        pref = [dict() for _ in range(n)]
        for i in range(n):
            for j in range(n - 1):
                pref[i][preferences[i][j]] = j
        
        unhappy = set()
        for i in range(len(pairs)):
            x = pairs[i][0]
            y = pairs[i][1]
            for j in range(i + 1, len(pairs)):
                u = pairs[j][0]
                v = pairs[j]
[1]
                
                if pref[x][u] < pref[x][y] and pref[u][x] < pref[u][v]:
                    unhappy.add(x)
                    unhappy.add(u)
                if pref[x][v] < pref[x][y] and pref[v][x] < pref[v][u]:
                    unhappy.add(x)
                    unhappy.add(v)
                if pref[y][u] < pref[y][x] and pref[u][y] < pref[u][v]:
                    unhappy.add(y)
                    unhappy.add(u)
                if pref[y][v] < pref[y][x] and pref[v][y] < pref[v][u]:
                    unhappy.add(y)
                    unhappy.add(v)
        return len(unhappy)
