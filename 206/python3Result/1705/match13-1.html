class Solution:
    def unhappyFriends(self, n: int, preferences: List[List[int]], pairs: List[List[int]]) -> int:
        acc = [set() for _ in range(n)]
        for x, y in pairs:
            for i in preferences[x]:
                if y == i:
                    break
                acc[x].add(i)
            for i in preferences[y]:
                if x == i:
                    break
                acc[y].add(i)
        
        res = 0
        for i in range(n):
            for t in acc[i]:
                if i in acc[t]:
                    res += 1
                    break
        return res
