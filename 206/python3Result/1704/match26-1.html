class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        m,n=len(mat),len(mat[0])
        counter_row=[0]*m
        counter_col=[0]*n
        for i in range(m):
            for j in range(n):
                if mat[i][j]==1:
                    counter_row[i]+=1
                    counter_col[j]+=1
        res=0
        for i in range(m):
            for j in range(n):
                if mat[i][j]==1 and counter_row[i]==1 and counter_col[j]==1:
                    res+=1
        return res
