class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        N = len(mat)
        M = len(mat[0])
        sumr = [sum(row) for row in mat]
        sumc = [sum([mat[i][c] for i in range(N)]) for c in range(M)]
        ct = 0
        for i in range(N):
            for j in range(M):
                if mat[i][j] == 1 and sumr[i] == 1 and sumc[j] == 1:
                    ct += 1
        return ct
