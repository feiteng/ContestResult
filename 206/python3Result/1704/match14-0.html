class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        rcnt=Counter()
        ccnt=Counter()
        for i, row in enumerate(mat):
            for j, n in enumerate(row):
                rcnt[i]+=n
                ccnt[j]+=n
        #print("rcnt:%s,ccnt:%s" %(rcnt, ccnt))
        ans=0
        for i, row in enumerate(mat):
            for j, n in enumerate(row):
                if rcnt[i]==1 and ccnt[j]==1 and n==1:
                    ans+=1
        return ans
