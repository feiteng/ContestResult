class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        R = len(mat)
        C = len(mat[0])
        row = [0] * R
        col = [0] * C
        for i in range(R):
            for j in range(C):
                if mat[i][j]:
                    row[i] += 1
                    col[j] += 1
        ans = 0
        for i in range(R):
            for j in range(C):
                if mat[i][j] and row[i] == 1 and col[j] == 1:
                    ans += 1
        return ans
