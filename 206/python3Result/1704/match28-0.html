class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        m = len(mat)
        n = len(mat[0])
        rows = [0] * m
        cols = [0] * n
        for i in range(m):
            for j in range(n):
                if mat[i][j]:
                    rows[i] += 1
                    cols[j] += 1
        res = 0
        for i in range(m):
            for j in range(n):
                if mat[i][j] and rows[i] == 1 and cols[j] == 1:
                    res += 1
        return res
