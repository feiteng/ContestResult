class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        n, m = len(mat), len(mat[0])
        
        row_sum = [0 for i in range(n)]
        col_sum = [0 for i in range(m)]
        for i in range(n):
            for j in range(m):
                row_sum[i] += mat[i][j]
                col_sum[j] += mat[i][j]
                
        cnt = 0
        for i in range(n):
            for j in range(m):
                if row_sum[i] == col_sum[j] == 1 and mat[i][j] == 1:
                    cnt += 1
        return cnt
