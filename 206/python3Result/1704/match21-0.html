class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        num_rows = len(mat)
        num_cols = len(mat[0])
        
        # compute number of ones in each row and col
        row_ones = [0]*num_rows
        col_ones = [0]*num_cols
        for row in range(num_rows):
            for col in range(num_cols):
                if mat[row][col] == 1:
                    row_ones[row] += 1
                    col_ones[col] += 1
        
        # for each 1: if its row count and col count == 1, +1 to ans
        ans = 0
        for row in range(num_rows):
            for col in range(num_cols):
                if mat[row][col] == 1 and row_ones[row] == 1 and col_ones[col] == 1:
                    ans += 1

        return ans
