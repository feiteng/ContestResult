class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        n,m = len(mat),len(mat[0])
        rc,cc = [0]*n,[0]*m
        for i in range(n) :
            for j in range(m) :
                rc[i] += mat[i][j]
                cc[j] += mat[i][j]
        
        # print(rc,cc)
        res = 0
        for i in range(n) :
            for j in range(m) :
                if rc[i] == 1 and cc[j] == 1 and mat[i][j]==1 :
                    res += 1
        return res
