class Solution:
    def numSpecial(self, grid: List[List[int]]) -> int:
        R = len(grid)
        C = len(grid[0])
        
        rowSum = [sum(row) for row in grid]
        colSum = [sum(grid[r][c] for r in range(R)) for c in range(C)]
        count = 0
        for r in range(R):
            for c in range(C):
                if grid[r][c] and colSum[c] == rowSum[r] == 1:
                    count +=1 
        return count
