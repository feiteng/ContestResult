class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        rsz, csz = len(mat), len(mat[0])
        rcnt, ccnt = [0] * rsz, [0] * csz

        for i in range(rsz):
            for j in range(csz):
                if mat[i][j] == 1:
                    rcnt[i] += 1
                    ccnt[j] += 1

        ans = 0
        for i in range(rsz):
            for j in range(csz):
                if mat[i][j] == 1 and rcnt[i] == 1 and ccnt[j] == 1:
                    ans += 1

        return ans
