class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        row_sums = Counter()
        col_sums = Counter()
        for i, row in enumerate(mat):
            for j, val in enumerate(row):
                row_sums[i] += val
                col_sums[j] += val
        ans = 0
        for i, row in enumerate(mat):
            for j, val in enumerate(row):
                if val == 1 and row_sums[i] == col_sums[j] == 1:
                    ans += 1
        return ans
