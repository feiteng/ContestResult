class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        R, C = len(mat), len(mat[0])
        col, row = [0] * C, [0] * R
        for r in range(R):
            for c in range(C):
                if mat[r][c] == 1:
                    row[r] += 1
                    col[c] += 1
        res = 0
        for r in range(R):
            for c in range(C):
                if mat[r][c] == 1 and row[r] == 1 and col[c] == 1:
                    res += 1
        return res
