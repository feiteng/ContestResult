class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        ans = 0
        for i in range(m):
            for j in range(n):
                if mat[i][j] == 1:
                    if sum(mat[i]) + sum
(mat[k][j] for k in range(m)) == 2:
                        ans += 1
        return ans
