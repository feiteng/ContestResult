class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        height ,width = len(mat), len(mat[0])
        x = [0] * height
        y = [0] * width 
        for i in range(height):
            for j in range(width):
                if mat[i][j]:
                    x[i] += 1
                    y[j] += 1
        rv = 0
        for i in range(height):
            for j in range(width):
                if mat[i][j] and x[i]==1 and y[j] == 1:
                    rv += 1
        return rv
