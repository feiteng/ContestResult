class Solution:
    def numSpecial(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        res = 0
        for i in range(m):
            for j in range(n):
                if mat[i][j] == 1:
                    if all(mat[ni][j] == 0 for ni in range(m) if ni != i) and all(mat[i][nj] == 0 for nj in range(n) if nj != j):
                        res += 1
        return res
