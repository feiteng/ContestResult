class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        HashSet<Integer>[] set = new HashSet[n];
        for (int i = 0; i < pairs.length; i++) {
            int person1 = pairs[i][0];
            int person2 = pairs[i][1];
            set[person1] = new HashSet<>();
            
set[person2] = new HashSet<>();
            for (int j = 0; j < preferences[person1].length; j++) {
                if (preferences[person1][j] == person2) {
                    break;
                }
                set[person1].add(preferences[person1][j]);
            }
            for (int j = 0; j < preferences[person2].length; j++) {
                if (preferences[person2][j] == person1) {
                    break;
                }
                set[person2].add(preferences[person2][j]);
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            
for (int person : set[i]) {
                if (set[person].contains(i)) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
}
