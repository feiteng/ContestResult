class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        TreeMap<Integer, Integer>[] adj = new TreeMap[n];
        for(int i = 0 ; i < n ; i++){
            adj[i] = new TreeMap<>();
            for(int j = 0 ; j < preferences[i].length; j++){
                adj[i].put(preferences[i][j], j);
            }
        
}
        Set<Integer> res = new HashSet<>();
        for(int i = 0; i < pairs.length; i++){
            for(int j = 0; j < pairs.length; j++){
                if(i == j){
                    continue;
                }
                if(judge(adj, pairs[i][0], pairs[i][1], pairs[j][0], pairs[j][1])){
                    res.add(pairs[i][0]);
                }
                if(judge(adj, pairs[i][0], pairs[i][1], pairs[j][1], pairs[j][0])){
                    res.add(pairs[i][0]);
                }
                if(judge(adj, pairs[i][1], pairs[i][0], pairs[j][0], pairs[j][1])){
                    res.add(pairs[i][1]);
                }
                if(judge(adj, pairs[i][1], pairs[i][0], pairs[j][1], pairs[j][0])){
                    
res.add(pairs[i][1]);
                }
            }
        }
        return res.size();
    }
    
    private boolean judge(TreeMap<Integer, Integer>[] adj, int x, int y, int u, int v){
        return adj[x].get(u) < adj[x].get(y) && adj[u].get(x) < adj[u].get(v);
    }
}
