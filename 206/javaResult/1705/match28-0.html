class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] paired = new int[n];
        for(int i = 0; i < pairs.length; i++) {
            paired[pairs[i][0]] = pairs[i][1];
            paired[pairs[i][1]] = pairs[i][0];
        }
        int[][] prefs = new int[n][n];
        for(int i = 0; i < n; i++) {
            prefs[i][i] = -1;
            for(int j = 0; j < n-1; j++) {
                prefs[i][preferences[i][j]] = j;
            }
        }
        int res = 0;
        for(int i = 0; i < n; i++) {
            int pairedFriend = paired[i];
            int prefOfCurr = prefs[i][pairedFriend];
            boolean flag = false;
            for(int j = 0; j < prefOfCurr && !flag; j++) {
                int candidate = preferences[i][j];
                int pairedOfCandidate = paired[candidate];
                int profInCandidate = prefs[candidate][pairedOfCandidate];
                int profSelfInCandidate = prefs[candidate][i];
                if(profSelfInCandidate < profInCandidate)
                    flag = true;
            }
            if(flag)
                res++;
        }
        return res;
    }
}
