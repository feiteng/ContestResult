class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        var like = new int[n][n];
        for (int i=0; i<n; i++) {
            for (int j=0; j<preferences[i].length; j++)
                like[i][preferences[i][j]] = j;
        }
        
        var currentLike = new int[n];
        for (int[] pair : pairs) {
            currentLike[pair[0]] = like[pair[0]][pair[1]];
            currentLike[pair[1]] = like[pair[1]][pair[0]];
        }
        
        var unhappy = new HashSet<>();
        for (int i=0; i<n; i++) {
            for (int j=i+1; j<n; j++) {
                if (like[i][j] < currentLike[i] &&
                    like[j][i] < currentLike[j]) {
                    unhappy.add(i);
                    unhappy.add(j);
                }
            }
        }
        return unhappy.size();
    }
}
