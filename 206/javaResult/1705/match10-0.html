class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {

        int res =0;
        int[] map =new int[n];
        for (int[] p: pairs){
            map[p[0]] = p[1];
            map[p[1]] = p[0];
        }
        int[][] dis = new int[n][n];
        for (int i=0;i<n;i++){
            for (int j=0;j<n-1;j++){
                dis[i][preferences[i][j]] = j+1;
            }
        }
        for (int x=0;x<n;x++){
            int y = map[x];
            for (int u = 0;u<n;u++){
                int v = map[u];
                if (u==x||u==y||v==x||v==y) continue;
                if (dis[x][u]<dis[x][y]&&dis[u][x]<dis[u][v]){
                    res++;
                    break;
                }
            }
            //System.out.println(x+"  "+res);
        }
        return res;
    }
}
