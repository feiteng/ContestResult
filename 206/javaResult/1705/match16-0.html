class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] map = new int[n];
        for (int[] p : pairs) {
            int u = p[0], v = p[1];
            map[u] = v;
            map[v] = u;
        }
        int count = 0;
        for (int x = 0;x < n;x++) {
            int y = map[x];
            int[] px = preferences[x];
            boolean found = false;
            for (int u : px) {
                if (found) break;
                if (u == y) break;
                int v = map[u];
                int[] pu = preferences[u];
                for (int xx : pu) {
                    if (xx == x) {
                        count++;
                        found = true;
                        break;
                    } else if (xx == v) break;
                }
            }            
        }
        
        return count;
    }
}
