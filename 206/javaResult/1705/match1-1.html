import java.util.*;

public class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        
int[][] inv = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - 1; j++) {
                inv[i][preferences[i][j]] = j;
            }
        }
        int[] partner = new int[n];
        
for (int[] p : pairs) {
            partner[p[0]] = p[1];
            partner[p[1]] = p[0];
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            boolean happy = true;
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    continue;
                }
                if (inv[i][partner[i]] > inv[i][j] &&
                        inv[j][partner[j]] > inv[j][i]) {
                    happy = false;
                
}
            }
            if(!happy){
                ans++;
            }
        }
        return ans;
    }
}
