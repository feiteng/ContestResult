class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] pos = new int[n][n];
        for (int i = 0; i < n; i++) {
            pos[i][i] = -1;
            for (int j = 0; j < n - 1; j++) {
                pos[i][preferences[i][j]] = j;
            }
        }
        int[] pair = new int[n];
        for (int[] pair2 : pairs) {
            pair[pair2[0]] = pair2[1];
            pair[pair2[1]] = pair2[0];
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            boolean valid = false;
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    continue;
                }
                if (pos[i][j] < pos[i][pair[i]] && pos[j][i] < pos[j][pair[j]]) {
                    valid = true;
                    break;
                }
            }
            if (valid) {
                res++;
            }
        }
        return res;
    }
}
