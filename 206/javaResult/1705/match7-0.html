class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {

        int[][] rank = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - 1; j++) {
                int k = preferences[i][j];
                rank[i][k] = j;
            }
        }

        int[] pairTo = new int[n];
        for (int[] pair : pairs) {
            int x = pair[0], y = pair[1];
            pairTo[x] = y;
            pairTo[y] = x;
        }


        int ans = 0;
        for (int i = 0; i < n; i++) {
            int r = rank[i][pairTo[i]];

            for (int[] pair : pairs) {

                int u = pair[0], v = pair[1];
                if (v == i || u == i) {
                    continue;
                }

                if (rank[i][u] < r && rank[u][i] < rank[u][v]) {
                    ans++;
                    break;
                }
                if (rank[i][v] < r && rank[v][i] < rank[v][u]) {
                    ans++;
                    break;
                }


            }


        }
        return ans;

    }
}
