class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] rev = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < preferences[i].length; j++) {
                rev[i][preferences[i][j]] = j;
            }
        }

        int[] pair = new int[n];
        for (int[] p : pairs) {
            pair[p[0]] = p[1];
            pair[p[1]] = p[0];
        }

        int out = 0;
        for (int i = 0; i < n; i++) {
            int x = i;
            int y = pair[i];
            int rank = rev[x][y];
            for (int j = 0; j < rank; j++) {
                int u = preferences[x][j];
                int v = pair[u];
                if (rev[u][x] < rev[u][v]) {
                    out++;
                    break;
                
}
            }
        }

        return out;
    }
}
