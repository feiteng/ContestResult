class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        Set<Integer> unhappy = new HashSet<>();
        
        int[][] pval = new int[n][n];
        
        for (int i = 0; i < preferences.length; i++) {
            for (int j = 0; j < preferences[i].length; j++) {
                pval[i][preferences[i][j]] = j;
            }
        }
        

        for (int i = 0; i < pairs.length; i++) {
            int x = pairs[i][0];
            int y = pairs[i][1];

            int xy = pval[x][y];
            int yx = pval[y][x];
            
            for (int j = i + 1; j < pairs.length; j++) {
                int u = pairs[j][0];
                int v = pairs[j][1];
                
                int xu = pval[x][u];
                int xv = pval[x][v];
                
                int yu = pval[y][u];
                int yv = pval[y][v];
                
                int uv = pval[u][v];
                int ux = pval[u][x];
                int uy = pval[u][y];
                
                int vu = pval[v][u];
                int vx = pval[v][x];
                int vy = pval[v][y];
                
                if (xy > xu && uv > ux) {
                    unhappy.add(x);
                    unhappy.add(u);
                }
                
                if (xy > xv && vu > vx) {
                    unhappy.add(x);
                    unhappy.add(v);
                }
                
                if (yx > yu && uv > uy) {
                    unhappy.add(y);
                    unhappy.add(u);
                }
                
                if (yx > yv && vu > vy) {
                    unhappy.add(y);
                    unhappy.add(v);
                }
            }
        }
        
        //log(unhappy);
        return unhappy.size();
    }
    
    private void log(Object o) {
        System.out.println(o);
    }
    
}
