class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] pres = new int[n][n];
        for (int i = 0; i < n; i++) {
            int cur = 0;
            for (int j = 0; j < n - 1; j++) {
                pres[i][preferences[i][j]] = cur;
                cur++;
            }
        }
        int k = pairs.length;
        Set<Integer> unhappy = new HashSet<>();
        
for (int i = 0; i < k; i++) {
            for (int j = 0; j < k; j++) {
                if (i == j) continue;
                int[] a = pairs[i], b = pairs[j];
                if (pres[a[0]][a[1]] > pres[a[0]][b[0]] && pres[b[0]][b[1]] > pres[b[0]][a[0]]
                   || pres[a[0]][a[1]] > pres[a[0]][b[1]] && pres[b[1]][b[0]] > pres[b[1]][a[0]]) unhappy.add(a[0]);
                if (pres[a[1]][a[0]] > pres[a[1]][b[0]] && pres[b[0]][b[1]] > pres[b[0]][a[1]]
                   || pres[a[1]][a[0]] > pres[a[1]][b[1]] && pres[b[1]][b[0]] > pres[b[1]][a[1]]) unhappy.add(a[1]);
            }
        }
        return unhappy.size();
    }
}
