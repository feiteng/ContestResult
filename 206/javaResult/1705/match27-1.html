class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        Map<Integer, Integer> mapPair = new HashMap<>();
        for(int[] p : pairs) {
            int x = p[0];
            int y = p[1];
            mapPair.put(x, y);
            mapPair.put(y, x);
        }
        
        int res = 0;
        
for(int[] p : pairs) {
            int x = p[0];
            int y = p[1];
            if(!isHappy(x, y, preferences, mapPair)) {
                res++;
            }
            if(!isHappy(y, x, preferences, mapPair)) {
                res++;
            }
        }
        
        return res;
    }
    
    boolean isHappy(int x, int y, int[][] preferences, Map<Integer, Integer> mapPair) {
        int[] prefer = preferences[x]; 
        for(int p : prefer) {
            
if(p == y) {
                break;
            }
            int v = mapPair.get(p);
            for(int t : preferences[p]) {
                if(t == x) {
                    return false;
                } else if(t == v) {
                    break;
                
}
            }
        }
        return true;
    }
}
