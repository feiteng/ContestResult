class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        
        HashMap <Integer, Integer> map = new HashMap<>();       
        int count = 0;
        int[][] pSort = new int[n][n];
        
        for (int i = 0; i < pairs.length; i++) {
            map.put(pairs[i][0], pairs[i][1]);
            map.put(pairs[i][1], pairs[i][0]);
        }
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; j++) {
                pSort[i][preferences[i][j]] = j;
            }
        }

        for (int[] p : pairs) {
            for (int i : preferences[p[0]]) {
                if (i == p[1]) {
                    break;
                } 
                else {
                    int v = map.get(i);
                    if (pSort[i][p[0]] < pSort[i][v]) {
                        count++;
                        break;
                    }
                }
            }

            for (int i : preferences[p[1]]) {
                if (i == p[0]) {
                    break;
                } 
                else {
                    int v = map.get(i);
                    if (pSort[i][p[1]] < pSort[i][v]) {
                        count++;
                        break;
                    }
                }
            }
        }
        return count;
    }
}
