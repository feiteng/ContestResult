class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
    	int[][] ref = new int[n][n];
    	for (int i=0; i<n; i++) {
    		int a = i;
    		for (int j=0; j<n-1; j++) {
    			int b = preferences[i][j];
    			ref[a][b] = j;
    		}
    	}
    	
    	int[] mate = new int[n];
    	for (int[] pair : pairs) {
    		int a = pair[0];
    		int b = pair[1];
    		mate[a] = b;
    		mate[b] = a;
    	}
    	
    	int ans = 0;
    	for (int i=0; i<n; i++) {
    		int j = mate[i];
    		if (ref[i][j] != 0) {
    			for (int k = ref[i][j]-1; k >=0; k--) {
    				int a = preferences[i][k];
    				int b = mate[a];
    				if (ref[a][b] > ref[a][i]) {
    					ans ++;
    					break;
    				}
    			}
    		}
    	}
    	return ans;
    }
}
