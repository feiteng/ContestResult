class Solution {
    public int unhappyFriends(int n, int[][] pr, int[][] pairs) {
        int[][] ranks = new int[n][n];
        for (int i = 0; i < pr.length; i++) {
            for (int j = 0; j < n - 1; j++) {
                ranks[i][pr[i][j]] = j;
            }
        }
        
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < pairs.length - 1; i++) {
            for (int j = i + 1; j < pairs.length; j++) {
                int[] p1 = pairs[i];
                int[] p2 = pairs[j];
                checkUnhappy(p1[0], p1[1], p2[0], p2[1], ranks, set);
                checkUnhappy(p1[1], p1[0], p2[0], p2[1], ranks, set);
                checkUnhappy(p1[0], p1[1], p2[1], p2[0], ranks, set);
                checkUnhappy(p1[1], p1[0], p2[1], p2[0], ranks, set);
                
                checkUnhappy(p2[0], p2[1], p1[0], p1[1], ranks, set);
                checkUnhappy(p2[1], p2[0], p1[0], p1[1], ranks, set);
                checkUnhappy(p2[0], p2[1], p1[1], p1[0], ranks, set);
                checkUnhappy(p2[1], p2[0], p1[1], p1[0], ranks, set);
            }
        }
        
        return set.size();
    }
    
    private void checkUnhappy(int x, int y, int u, int v, int[][] ranks, Set<Integer> set) {
        if (ranks[x][u] < ranks[x][y] && ranks[u][x] < ranks[u][v]) {
            set.add(x);
        }
    }
}
