import java.util.HashMap;
import java.util.HashSet;

class Solution {
  public static void main(String args[]) {
    int n = 4;
    int[][] preferences = {
        {1,2,3},{3,2,0},{3,1,0},{1,2,0}
    };
    int[][] pairs = {{0, 1},{2, 3}};
    Solution sol = new Solution();
    System.out.println(sol.unhappyFriends(n, preferences, pairs));
  }
  
  
public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
    HashMap<Integer,Integer>[] rankings = new HashMap[n];
    for (int i = 0; i < rankings.length; i++) {
      rankings[i] = new HashMap<>();
    }
    for (int i = 0; i < preferences.length; i++) {
      for (int j = 0; j < preferences[i].length; j++) {
        rankings[i].put(preferences[i][j], j);
      }
    }
    
    HashMap<Integer, Integer> pairMap = new HashMap<>();
    for (int[] pair : pairs) {
      int x = pair[0];
      int y = pair[1];
      pairMap.put(x, y);
      pairMap.put(y, x);
    
}
    
    HashSet<Integer> unHappy = new HashSet<>();
    for (int i = 0; i < n; i++) {
      for (int j = i + 1; j < n; j++) {
        if (pairMap.get(i) == j) {
          continue;
        }
        int rankI = rankings[i].get(pairMap.get(i));
        int rankJ = rankings[j].get(pairMap.get(j));
        
        int swapI = rankings[i].get(j);
        int swapJ = rankings[j].get(i);
        
        
if (swapI < rankI && swapJ < rankJ) {
          unHappy.add(i);
          unHappy.add(j);
        }
      }
    }

    return unHappy.size();
  }
}
