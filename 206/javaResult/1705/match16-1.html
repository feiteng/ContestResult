class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] pair = new int[n];
        for (int[] p : pairs) {
            int x = p[0], y = p[1];
            pair[x] = y;
            pair[y] = x;
        }
        int res = 0;
        for (int x = 0; x < n; x++) {
            int y = pair[x];
            boolean flag = true;
            for (int i = 0; flag && i < n - 1; i++) {
                int u = preferences[x][i];
                if (u == y) {
                    break;
                }
                
int v = pair[u];
                for (int j = 0; j < n - 1; j++) {
                    if (preferences[u][j] == x) {
                        res++;
                        flag = false;
                        break;
                    }
                    if (preferences[u][j] == v) {
                        break;
                    
}
                }
            }
        }
        return res;
    }
}
