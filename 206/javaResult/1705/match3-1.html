class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int mat[][] = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - 1; j++) {
                mat[i][preferences[i][j]] = j;
            }
        }
        
        int res = 0;
        
HashSet<Integer> unhappy = new HashSet<>();
        for (int i = 0; i < pairs.length; i++) {
            for (int j = 0; j < pairs.length; j++) {
                if (i == j) continue;
                
int x = pairs[i][0];
                int y = pairs[i][1];
                int u = pairs[j][0];
                int v = pairs[j][1];
                if (mat[x][y] > mat[x][u] && mat[u][v] > mat[u][x]) {
                    unhappy.add(x);
                    unhappy.add(u);
                }
                
                if (mat[x][y] > mat[x][v] && mat[v][u] > mat[v][x]) {
                    unhappy.add(x);
                    unhappy.add(v);
                }
                
                if (mat[y][x] > mat[y][u] && mat[u][v] > mat[u][y]) {
                    unhappy.add(y);
                    unhappy.add(u);
                }
                

                if (mat[y][x] > mat[y][v] && mat[v][u] > mat[v][y]) {
                    unhappy.add(y);
                    unhappy.add(v);
                }
            }
        }
        
        return unhappy.size();
    }
}
