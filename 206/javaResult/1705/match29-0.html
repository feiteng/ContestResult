class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] pairing = new int[n];
        for (int[] pair : pairs) {
            pairing[pair[0]] = pair[1];
            pairing[pair[1]] = pair[0];
        }
        
        int ret = 0;
        
        for (int i = 0; i < pairing.length; i++) {
            int val = i;
            int friend = pairing[val];
            int[] preference = preferences[val];
            for (int j = 0; j < preference.length; j++) {
                if (preference[j] == friend) {
                    break;
                }
                int u = preference[j]; //3
                int v = pairing[u]; //2
                //find if val is closer to v in u
                boolean found = false;
                for (int pos = 0; pos < preferences[u].length; pos++) {
                    if (val == preferences[u][pos]) {
                        ret++;
                        found = true;
                        break;
                    }
                    if (v == preferences[u][pos]) {
                        break;
                    }
                }
                
                if (found) {
                    break;
                }
                
            }
        }
        
        return ret;
        
        
        
    }
}
