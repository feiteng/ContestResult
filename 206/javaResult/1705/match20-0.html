class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] prefer = new int[n][n];
        Map<Integer, Integer> map = new HashMap<>();
        for(int i=0; i<n; i++){
            for(int j=0; j<n-1; j++){
                int to = preferences[i][j];
                prefer[i][to] = n-1-j;
            }
        }
        for(int i=0; i<n/2; i++){
            map.put(pairs[i][0], pairs[i][1]);
            map.put(pairs[i][1], pairs[i][0]);
        }
        int res = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(i == j) continue;
                int iP = map.get(i);
                int jP = map.get(j);
                if(j == iP) continue;
                if(prefer[i][j] > prefer[i][iP] && prefer[j][i] > prefer[j][jP]){
                    //System.out.println(i);
                    res ++;
                    break;
                }
            }
        }
        return res;
    }
}
