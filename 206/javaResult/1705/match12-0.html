class Solution {
    public int unhappyFriends(int n, int[][] pref, int[][] pairs) {
        int[] pair = new int[n];
        int res = 0;
        for (int[] p : pairs) {
            pair[p[0]] = p[1];
            pair[p[1]] = p[0];
        }
        int[][] prefs = new int[n][n];
        for (int i = 0; i < n; i++) {
            int[] single = pref[i];
            for (int j = 0; j < single.length; j++) {
                prefs[i][single[j]] = n - 1 - j;
            }
        }
        for (int i = 0; i < n; i++) {
            int cur = pair[i];
            int curPref = prefs[i][cur];
            int[] pre = pref[i];
            for (int k = 0; k < n - 1; k++) {
                if (pre[k] == cur) break;
                int target = pre[k];
                if (prefs[target][i] > prefs[target][pair[target]]) {
                    res++;
                    break;
                }
            }
        }
        return res;
    }
}
