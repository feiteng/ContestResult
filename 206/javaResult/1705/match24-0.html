/*
 n = 4, preferences = [[1, 2, 3], [3, 2, 0], [3, 1, 0], [1, 2, 0]], pairs = [[0, 1], [2, 3]]
 
 for i, pared with i2
 
 find i's prefer list, for any i1 before i2
 
 find i1's pair, i11
 
 if in i1's list, i is before i11, then ++
 

*/
class Solution {
    
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        
        Map<Integer, Integer> p = new HashMap<>();
        for(int[] pr : pairs){
            p.put(pr[0], pr[1]);
            p.put(pr[1], pr[0]);
        }
        
        int ans = 0; 
        for(int i = 0; i<n; i++){
           int i2 = p.get(i);
            
           for(int j= 0; j<preferences[i].length; j++){
               if(preferences[i][j] == i2) break;
               int i1 = preferences[i][j];
               
               int i11 = p.get(i1);
               
               if(isBefore(preferences[i1], i, i11)){
                   ans++;
                   break;
               }
           }
        }
        
        return ans;
        
    }
    
    boolean isBefore(int[] list, int a, int b){
        int pa = -1; 
        int pb = -1;
        for(int i = 0; i<list.length; i++){
            if(list[i] == a) pa = i;
            if(list[i] == b) pb = i;
        }
        return pa<pb;
    }
}
