class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] friends = new int[n];
        for (int[] pair : pairs) {
            int a = pair[0];
            int b = pair[1];
            friends[a] = b;
            friends[b] = a;
        }
        
        int[][] score = new int[n][n];
        for (int i = 0; i < n; i++) {
            int[] pref = preferences[i];
            int s = n;
            for (int j = 0; j < pref.length; j++) {
                score[i][pref[j]] = s;
                s--;
            
}
        }
        
        int unhappy = 0;
        for (int i = 0; i < n; i++) {
            int f = friends[i];
            int s = score[i][f];
            for (int j = 0; j < n; j++) {
                if (score[i][j] <= s) {
                    continue;
                }
                // friend j is a candidate
                int js = score[j][friends[j]];
                if (score[j][i] > js) 
{
                    unhappy++;
                    break;
                }
            }
        }
        
        return unhappy;
        
    }
}
