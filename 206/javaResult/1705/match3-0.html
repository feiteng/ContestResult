class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] rank = new int[n][n];
        for (int i = 0; i < n; ++i) {
            Arrays.fill(rank[i], n + 1);
        }
        
        for (int i = 0; i < n; ++i) {
            int pref = 0;
            for (int p: preferences[i]) {
                rank[i][p] = pref++;
            }
        }
        Set<Integer> potential = new HashSet<>();
        for (int i = 0; i < pairs.length; ++i) {
            for (int j = 0; j < pairs.length; ++j) {
                if (i == j) continue;
                int x = pairs[i][0], y = pairs[i][1];
                int u = pairs[j][0], v = pairs[j][1];
                boolean xPair = false, yPair = false;
                if (rank[x][y] > rank[x][u] && rank[u][v] > rank[u][x]) {
                    potential.add(x);
                    potential.add(u);
                }
                if (rank[x][y] > rank[x][v] && rank[v][u] > rank[v][x]) {
                    potential.add(x);
                    potential.add(v);
                }
                if (rank[y][x] > rank[y][u] && rank[u][v] > rank[u][y]) {
                    potential.add(y);
                    potential.add(u);
                }
                if (rank[y][x] > rank[y][v] && rank[v][u] > rank[v][y]) {
                    potential.add(y);
                    potential.add(v);
                }
                //if (xPair) cnt += 2;
                //if (yPair) cnt += 2;
            }
        }
        
        return potential.size();
    }
}
