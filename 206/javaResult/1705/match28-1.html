class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int [][] rank = new int[n][n];
        
int [] p = new int[n];
        for (int i = 0; i < pairs.length; i++) {
            p[pairs[i][0]] = pairs[i][1];
            p[pairs[i][1]] = pairs[i][0];
        }
        for (int i = 0; i < n; i++) {
            
for (int j = 0; j < n - 1; j++) {
                rank[i][preferences[i][j]] = j;
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            int cur = rank[i][p[i]]; boolean found = false;
            for (int j = 0; j < n; j++) {
                if (j == i || j == p[i]) continue;
                int myRank = rank[i][j];
                if (myRank > cur) continue;
                
int hisRank = rank[j][i];
                if (hisRank < rank[j][p[j]]) found = true;
            }
            if (found) ++res;
        }
        return res;
    }
}
