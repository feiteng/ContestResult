class Solution {
    public int unhappyFriends(int n, int[][] ps, int[][] pairs) {
        
        int g[][] = new int[n][n];
        
        for(int i=0;i<n;++i){
            int f = n+100;
            for(int j=0;j<ps[i].length;++j){
                
                g[i][ps[i][j]] = f;
                f--;
            }
            
        }
        
        int num = n/2;
        
        boolean f[] = new boolean[n];
        
        for(int i=0;i<num;++i){
            for(int j=0;j<num;++j){
                if(i==j) continue;
                
                int x = pairs[i][0];
                int y = pairs[i][1];
                
                int u = pairs[j][0];
                int v = pairs[j][1];
                
                if(g[x][u]>g[x][y] && g[u][x] > g[u][v]){
                    f[x] = true;
                    f[u] = true;
                }
                
                if(g[x][v]>g[x][y] && g[v][x] > g[v][u]){
                    f[x] = true;
                    f[v] = true;
                }
                
                if(g[y][u]>g[y][x] && g[u][y] > g[u][v]){
                    f[y] = true;
                    f[u] = true;
                }
                
                if(g[y][v]>g[y][x] && g[v][y] > g[v][u]){
                    f[y] = true;
                    f[v] = true;
                }
                
                
                
                
                
                
            }
        }
        int c= 0 ;
        for(int i=0;i<n;++i){
            if(f[i]){
                c++;
            }
        }
        return c;
        
        
        
        
    }
}
