class Solution {
    public int unhappyFriends(int n, int[][] pref, int[][] pairs) {
        Map<Integer, Integer> pmap = new HashMap<>();
        for (int[] p : pairs) {
            pmap.put(p[0], p[1]);
            pmap.put(p[1], p[0]);
        }
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            
            int other = pmap.get(i);
            for (int j = 0; j < pref[i].length && pref[i][j] != other; j++) {
                if (unhappy(pref, pref[i][j], 
pmap.get(pref[i][j]), i)) {
                    res++;
                    break;
                }
            }
        }
        return res;
    }
    
    private boolean unhappy(int[][] pref, int guy, int friend, int testf) {
        for (int i = 0; i < pref[guy].length && pref[guy][i] != friend; i++) {
            if (pref[guy][i] == testf) return true;
        }
        return false;
    }
}
