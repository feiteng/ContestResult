class Solution {
    public int unhappyFriends(int n, int[][] prefs, int[][] pairs) {
        int[][] rps = new int[n][n];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n-1; ++j) {
                rps[i][prefs[i][j]] = j;
            }
        }
        int[] fps = new int[n];
        for (int[] p : pairs) {
            fps[p[0]] = p[1];
            fps[p[1]] = p[0];
        }
        int ans = 0;
        for (int x = 0; x < n; ++x) {
            int y = fps[x];
            int uh = 0;
            for (int u = 0; u < n; ++u) {
                if (u==x)
                    continue;
                int v = fps[u];
                if (rps[x][u]<rps[x][y]&&rps[u][x]<rps[u][v])
                    uh = 1;
            }
            ans += uh;
        }
        return ans;
    }
}
