class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] other = new int[n];
        for(int[] pair : pairs) {
            int x = pair[0];
            int y = pair[1];
            other[x] = y;
            other[y] = x;
        }
        Set<Integer>[] over = new Set[n];
        for(int i = 0; i < n; ++i) {
            over[i] = new HashSet<>();
            for(int p : preferences[i]) {
                if (p == other[i]) {
                    break;
                } else {
                    over[i].add(p);
                }
            }
        }
        int result = 0;
        for(int i = 0; i < n; ++i) {
            Set<Integer> set = over[i];
            for(int y : set) {
                if (over[y].contains(i)) {
                    ++result;
                    break;
                }
            }
        }
        return result;
    }
}
