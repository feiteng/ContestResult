class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] pe = new int[n][n];
        int[] pa=new int[n];
        
        for(int i=0;i<n;i++){
            for(int j=0;j<n-1;j++){
                pe[i][preferences[i][j]]=j;
            }
        }
        for(int i=0;i<n/2;i++){
            pa[pairs[i][0]]=pairs[i][1];
            pa[pairs[i][1]]=pairs[i][0];
        }
        
        
        int ans = 0;
        for(int i=0;i<n;i++){
            boolean unHappy = false;
            for(int j=0;j<n;j++){
                if(j!=i&&pa[i]!=j){
                    if(pe[i][j]<pe[i][pa[i]] && pe[j][i]<pe[j][pa[j]]){
                        unHappy=true;
                        //System.out.println("j="+j);
                        break;
                    }
                }
            }
            if(unHappy){
                //System.out.println("i="+i);
                ans++;
            }
        }
        
        return ans;
    }
}
