class Solution {
    public int unhappyFriends(int n, int[][] pre, int[][] pairs) {
        int [] idx = new int[n];
        int res = 0;
        for(int i = 0; i < n/2; i++){
            idx[pairs[i][0]] = i;
            idx[pairs[i][1]] = i;
        }
        
        for(int i = 0; i < n; i++){
            int p = (pairs[idx[i]][0] == i ? pairs[idx[i]][1] : pairs[idx[i]][0]);
            
for(int u : pre[i]){
                if(u == p){
                    break;
                }
                boolean flag = false;
                int up = (pairs[idx[u]][0] == u ? pairs[idx[u]][1] : pairs[idx[u]][0]);
                
for(int x : pre[u]){
                    if(x == i){
                       res ++;
                        flag = true;
                        break;
                    }
                    if(x == up){
                        break;
                    }
                }
                if(flag){
                    break;
                }
            }
        }
        return res;
        
    }
    
    
}
