class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[] partners = new int[n];
        
for (int[] pair : pairs) {
            partners[pair[0]] = pair[1];
            partners[pair[1]] = pair[0];
        }
        int[][] values = new int[n][n];
        for (int i = 0; i < n; i++) {
            int[] preference = preferences[i];
            for (int j = 0; j < n - 1; j++) {
                values[i][preference[j]] = n - j;
            }
        }
        int unhappy = 0;
        for (int x = 0, y; x < n; x++) {
            y = partners[x];
            
for (int u : preferences[x]) {
                if (u == y) break;
                int v = partners[u];
                if (values[u][x] > values[u][v]) {
                    unhappy++;
                    break;
                }
            }
        }
        return unhappy;
    }
}
