class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int res = 0;
        HashMap<Integer, Integer> mapPair = new HashMap<>();
        for (int[] i : pairs) {
            mapPair.put(i[0], i[1]);
            mapPair.put(i[1], i[0]);
        }

        int[][] preferenceSort = new int[n][n];
        
for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                preferenceSort[i][preferences[i][j]] = j;
            }
        }

        for (int[] p : pairs) {
            for (int i : preferences[p[0]]) {
                if (i == p[1]) {
                    break;
                } else {
                    int v = mapPair.get(i);
                    if (preferenceSort[i][p[0]] < preferenceSort[i][v]) {
                        ++res;
                        break;
                    }
                }
            }

            for (int i : preferences[p[1]]) {
                if (i == p[0]) {
                    break;
                } else {
                    int v = mapPair.get(i);
                    if (preferenceSort[i][p[1]] < preferenceSort[i][v]) {
                        ++res;
                        break;
                    }
                }
            }
        }
        return res;
    }
}
