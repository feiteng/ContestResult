class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        //Map<Integer,Integer>[] mp = new HashMap[preferences.length];
    //    Map<Integer,Map<Integer,Integer>> mp = new HashMap<>();
        
int[][] mp = new int[n][n];
        for(int i=0;i<preferences.length;++i){
            for(int j=0;j<preferences[i].length;++j){
                mp[i][preferences[i][j]]=n-j;
            }
        }
        int ans = 0;
        Set<Integer> unHappy = new HashSet<>();
        for(int i=0;i<pairs.length;++i){
            int x = pairs[i][0];
            int y = pairs[i][1];
            for(int j=1;j<pairs.length;++j){
                
int u = pairs[j][0];
                int v = pairs[j][1];
                if(mp[x][u]>mp[x][y]&&mp[u][x]>mp[u][v]){
                    unHappy.add(x);
                    unHappy.add(u);
                }
                if(mp[x][v]>mp[x][y]&&mp[v][x]>mp[v][u]){
                    unHappy.add(x);
                    unHappy.add(v);
                }
                if(mp[y][u]>mp[y][x]&&mp[u][y]>mp[u][v]){
                    unHappy.add(y);
                    unHappy.add(u);
                }
                if(mp[y][v]>mp[y][x]&&mp[v][y]>mp[v][u]){
                    unHappy.add(y);
                    unHappy.add(v);
                }
            }
        }
        return unHappy.size();
    }
}
