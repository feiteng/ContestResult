class Solution {
    Map<Integer, Integer> map = new HashMap<>(); 
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int ret = 0; 
        
        for (int i = 0; i < pairs.length; i++) {
            map.put(pairs[i][0], pairs[i][1]);
            map.put(pairs[i][1], pairs[i][0]);
        }
        for (int i = 0; i < pairs.length; i++) {
            int x = pairs[i][0];
            int y = pairs[i][1]; 
            if (helper(preferences, pairs, x, y)) {
                ret++; 
            } 
            if (helper(preferences, pairs, y, x)) {
                ret++; 
            } 
            
            
        }
        return ret;
    }
    private boolean helper(int[][] preferences, int[][] pairs, int x, int y) {
            boolean f = false;
            for (int j = 0; j < preferences[x].length; j++) {
                int u = preferences[x][j];
                if (u == y) {
                    break;
                }
                int v = map.get(u);
                
                for (int k = 0; k < preferences[u].length; k++) {
                    if (preferences[u][k] == v) {
                        break;
                    }
                    if (preferences[u][k] == x) {
                        f = true;
                        break;
                    }
                }
            }
            return f;
        
    }
}
