class Solution {
    public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {
        int[][] poses = new int[505][505];
        int[] partner = new int[505];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                poses[i][preferences[i][j]] = j;
            }
        }
        for (int i = 0; i < pairs.length; ++i) {
            partner[pairs[i][0]] = pairs[i][1];
            partner[pairs[i][1]] = pairs[i][0];
        }
        int result = 0;
        for (int i = 0; i < n; ++i) {
            int pair = partner[i];
            int pos = poses[i][pair];
            for (int otherIdx = 0; otherIdx < pos; ++otherIdx) {
                int other = preferences[i][otherIdx];
                int otherPair = partner[other];
                int otherPos = poses[other][otherPair];
                int thisPos = poses[other][i];
                if (thisPos < otherPos) {
                    // System.out.printf("%d with %d\n", i, other);
                    result += 1;
                    break;
                }
            }
        }
        
        return result;
    }
}
