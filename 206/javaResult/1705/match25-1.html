class Solution {
 public int unhappyFriends(int n, int[][] preferences, int[][] pairs) {

        int[][] order = new int[n][n];
        int[] pr = new int[n];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < preferences[i].length; j++) {
                int v = preferences[i][j];
                order[i][v] = j;
            }
        }

        for (int i = 0; i < pairs.length; i++) {
            int x = pairs[i][0];
            int y = pairs[i][1];
            
pr[x] = y;
            pr[y] = x;
        }

        int ans = 0;
        for (int i = 0; i < pairs.length; i++) {
            int x = pairs[i][0];
            int y = pairs[i][1];

            for (int j = 0; j < preferences[x].length; j++) {
                int u = preferences[x][j];
                if (u == y) break;

                if (order[u][x] < order[u][pr[u]]) {
                    ans++;
                    break;
                }
            }


            int t = x;
            x = y;
            y = t;
            for (int j = 0; j < preferences[x].length; j++) {
                int u = preferences[x][j];
                if (u == y) break;

                
if (order[u][x] < order[u][pr[u]]) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
}
