class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        if (n == 1) return 0;
        
        DSU dsu = new DSU(n);
        List<Node> l = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int[] p1 = points[i];
            for (int j = 0; j < n; j++) {
                if (j == i) continue;
                int[] p2 = points[j];
                l.add(new Node(i, j, d(p1, p2)));
            }
        }
        
        Collections.sort(l, (a, b) -> a.dist - b.dist);
        int res = 0;
        
        for (int i = 0; i < l.size(); i++) {
            Node node = l.get(i);
            int p1 = node.p1;
            int p2 = node.p2;
            int dist = node.dist;
            if (dsu.union(p1, p2)) {
                res += dist;
            }
        }
        
        return res;
    }
    
    int d(int[] p, int[] q) {
        return Math.abs(p[0] - q[0]) + Math.abs(p[1] - q[1]);
    }
}

class Node {
    int p1;
    int p2;
    int dist;
    Node(int a, int b, int c) {
        p1 = a;
        p2 = b;
        dist = c;
    }
}

class DSU {
    int n;
    int[] f;
    DSU(int n) {
        this.n = n;
        f = new int[n];
        for (int i = 0; i < n; i++) {
            f[i] = i;
        }
    }
    
    int find(int i) {
        if (f[i] == i) return i;
        f[i] = find(f[i]);
        
        return f[i];
    }
    
    boolean union(int i, int j) {
        int fi = find(i);
        int fj = find(j);
        
        if (fi != fj) {
            f[fi] = fj;
            return true;
        }
        return false;
    }
}
