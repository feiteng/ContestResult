class Solution {
    class N{
        int p1; 
        int p2;
        int d;
    }
    public int minCostConnectPoints(int[][] points) {
        
        List<int[]> edges = new ArrayList<>();
        
        int n = points.length;
        
for(int i = 0; i<n; i++){
            for(int j =i+1 ; j<n; j++){
                int d = Math.abs(points[i][0] -points[j][0]) + Math.abs(points[i][1] -points[j][1]);
                edges.add (new int[]{i, j, d});
            }
        }
        
        Collections.sort(edges, (a, b) -> (a[2]-b[2]));
        
        UnionFind uf = new UnionFind(n);
        
        int ans = 0; 
        for(int i = 0; i<edges.size(); i++){
            
            int[] e = edges.get(i);
            
if(!uf.union(e[0], e[1])){
                ans += e[2]; 
            }
        }
        
        return ans;
        
        
    }
    class UnionFind{
        int[] parents;
        int n;
        UnionFind(int n){
            this.n = n;
            
parents = new int[n];
            for(int i = 0; i<n; i++){
                parents[i] = i; 
            }
        }
        
        int find(int i){
            int pi = parents[i];
            if(pi == i) return pi;
            
            parents[i] = find(pi);
            return  parents[i];
        
}

        boolean union(int i, int j){
            int pi = find(i);
            int pj = find(j);
            if(pi == pj) return true;
            parents[pi] = pj;
            return false;
        }
    }
}
