class Solution {
    public int minCostConnectPoints(int[][] points) {
        int len = points.length;
        int[] roots = new int[len];
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[2] - b[2]));
        for (int i = 0;i < len;i++) {
            roots[i] = i;
            for (int j = i + 1;j < len;j++) {
                int dis = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                
pq.add(new int[]{i, j, dis});
            }
        }
        int cnt = len, res = 0;
        while (cnt > 1) {
            int[] cur = pq.poll();
            if (union(roots, cur[0], cur[1])) {
                res += cur[2];
                cnt--;
            }
        
}
        return res;
    }
    private int findRoot(int[] roots, int root) {
        return roots[root] == root ? root : (roots[root] = findRoot(roots, roots[root]));
    }
    private boolean union(int[] roots, int root1, int root2) 
{
        root1 = findRoot(roots, root1);
        root2 = findRoot(roots, root2);
        if (root1 == root2) {
            return false;
        }
        roots[root2] = root1;
        return true;
    }
}
