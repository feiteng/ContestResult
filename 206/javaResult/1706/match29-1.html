import java.util.Comparator;
import java.util.PriorityQueue;

class Solution {
    
class Node{
        int s;
        int e;
        int dis;

        public Node(int s, int e, int dis) {
            this.s = s;
            this.e = e;
            this.dis = dis;
        }
    }
    int[] f;
    int find(int a){
        while(a!=f[a]){
            a=f[a];
        }
        
return a;
    }
    void union(int x,int y){
        f[find(x)]=find(y);
    }
    public int minCostConnectPoints(int[][] points) {
        PriorityQueue<Node> pq=new PriorityQueue<Node>(Comparator.comparing(i->i.dis));
        
for (int i = 0; i < points.length; i++) {
            for(int j=i+1;j<points.length;j++){
                int cost=Math.abs(points[i][0]-points[j][0])+Math.abs(points[i][1]-points[j][1]);
                pq.add(new Node(i,j,cost));
                pq.add(new Node(j,i,cost));
            }
        }
        f=new int[points.length];
        for (int i = 0; i < f.length; i++) {
            f[i]=i;
        
}
        int ans=0;
        int k=0;
        while(k<points.length-1){
            Node cur=pq.remove();
            int a=cur.s;int b=cur.e;
            if(find(a)==find(b)){
                continue;
            }
            k++;
            union(a,b);
            ans+=cur.dis;
        }
        return ans;

    }
}
