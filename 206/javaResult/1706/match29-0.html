class Solution {
    
    int[] p;
    
    private int find(int x){
        if(p[x]!=x) p[x] = find(p[x]);
        return p[x];
    }
    
    private void union(int x, int y) {
        p[find(x)] = p[find(y)];
    }
    
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        p = new int[n];
        for(int i=0; i<n; i++) p[i]=i;
        Comparator<Pt> comparator = (o1, o2) -> {return o1.dist - o2.dist;};
        PriorityQueue<Pt> pq = new PriorityQueue<>(comparator);
        for(int i=0; i<n; i++) {
            for(int j=i+1; j<n; j++) {
                int distance = Math.abs(points[i][0]-points[j][0]) + Math.abs(points[i][1]-points[j][1]);
                pq.offer(new Pt(i, j, distance));
            }
        }
        int count = 0;
        int sum = 0;
        while(count<n-1) {
            Pt a = pq.poll();
            int x = find(a.i);
            int y = find(a.j);
            if(x==y) continue;
            sum+=a.dist;
            count+=1;
            union(x,y);
        }
        return sum;
    }
    
    class Pt {
        int i;
        int j;
        int dist;
        
        Pt(int i, int j, int dist) {
            this.i = i;
            this.j = j;
            this.dist = dist;
        }
    }
}
