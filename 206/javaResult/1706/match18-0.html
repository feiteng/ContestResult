class Solution {
    
    class UF {
        int n;
        int[] parents, sizes;
        
        UF(int n) {
            this.n = n;
            parents = new int[n];
            sizes = new int[n];
            for (int i = 0; i < n; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        int find(int i) {
            while (parents[i] != i) {
                parents[i] = parents[parents[i]];
                i = parents[i];
            }
            return i;
        }
        
        boolean union(int p, int q) {
            int pRoot = find(p), qRoot = find(q);
            if (pRoot == qRoot) {
                return false;
            } else {
                parents[pRoot] = qRoot;
                sizes[qRoot] += sizes[pRoot];
                return true;
            }
        }
        
        boolean isComplete() {
            int max = 0;
            for (int size : sizes) {
                max = Math.max(max, size);
            }
            return max == n;
        }
    }
    
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0]-b[0]);
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                int d = Math.abs(points[i][0]-points[j][0])+Math.abs(points[i][1]-points[j][1]);
                pq.add(new int[] {d, i, j});
            }
        }
        UF uf = new UF(n);
        int ans = 0;
        while (!pq.isEmpty() && !uf.isComplete()) {
            int[] cur = pq.poll();
            int d = cur[0], p = cur[1], q = cur[2];
            if (uf.union(p, q)) {
                ans += d;
            }
        }
        return ans;
    }
}
