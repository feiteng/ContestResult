class Solution {
    class Edge {
        int u;
        int v;
        int cost;
        public Edge(int u, int v, int cost) {
            this.u = u;
            this.v = v;
            this.cost = cost;
        }
    }
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        ArrayList<Edge> edges = new ArrayList<>();
        for (int j = 0; j < n; j++) {
            for (int k = 0; k < n ; k++) {
                if (j == k) {
                    continue;
                }
                Edge e = new Edge(j, k, Math.abs(points[j][0] - points[k][0]) + 
Math.abs(points[j][1] - points[k][1]));
                edges.add(e);
            }
        }
        edges.sort((a, b) -> a.cost - b.cost);
        int[] uf = new int[n];
        for (int j = 0; j < n; j++) {
            uf[j] = -1;
        }
        int cost = 0;
        for (Edge e : edges) {
            int u = e.u;
            int v = e.v;
            int c = e.cost;
            if (get(u, uf) != get(v, uf)) {
                //System.out.printf("u: %d, v: %d, c: %d, get u : %d, get v : %d\n", u, v, c, get(u, uf), get(v, uf));
                uf[get(u, uf)] = 
get(v, uf);
                cost += c;
            }
        }
        return cost;
    }
    
    public int get(int ind, int[] uf) {
        if (uf[ind] < 0) {
            return ind;
        }
        uf[ind] = get(uf[ind], uf);
        return uf[ind];
    }
}
