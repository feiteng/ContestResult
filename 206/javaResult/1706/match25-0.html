class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length, cost = 0;
        PriorityQueue<int[]> edges = new PriorityQueue<>((a, b) -> Integer.compare(a[2], b[2]));
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                edges.offer(new int[] { i, j, getDist(points[i], points[j]) });
            }
        }
        UnionFind uf = new UnionFind(n);
        while (uf.size > 1) {
            int[] edge = edges.poll();
            if (uf.union(edge[0], edge[1])) {
                cost += edge[2];
            }
        }
        
        return cost;
    }
    
    int getDist(int[] p1, int[] p2) {
        return Math.abs(p1[0] - p2[0]) + Math.abs(p1[1] - p2[1]);
    }
    
    static class UnionFind {
        int[] p;
        public int size;
        
        public UnionFind(int n) {
            this.p = new int[n];
            for (int i = 0; i < n; ++i) {
                p[i] = i;
            }
            size = n;
        }
        
        public boolean union(int x1, int x2) {
            int p1 = find(x1),
                p2 = find(x2);
            if (p1 != p2) {
                p[p1] = p2;
                --size;
                return true;
            }
            return false;
        }
        
        public int find(int x) {
            int query = x;
            while (p[x] != x) {
                x = p[x];
            }
            p[query] = x; // Note: turned out to be a vital optimization
            return x;
        }
    }

}
