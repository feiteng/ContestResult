class Solution {
    public int minCostConnectPoints(int[][] points) {
        int cost = 0;
        final int n = points.length;
        int[][] edges = new int[n * (n - 1) / 2][3];
        for (int i = 0, len = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                edges[len++] = 
new int[] { i, j, man(points, i, j) };
            }
        }
        Arrays.sort(edges, (a, b) -> Integer.compare(a[2], b[2]));
        Union union = new Union(n);
        union.init(n);
        for (int i = 0; i < edges.length; ++i) {
            
int[] e = edges[i];
            if (union.find(e[0]) == union.find(e[1])) continue;
            union.unite(e[0], e[1]);
            cost += e[2];
        }
        return cost;
    }
    private static int man(int[][] points, int i, int j) {
        return Math.abs(points[i][0] - points[j][0]) + 
Math.abs(points[i][1] - points[j][1]);
    }
}

class Union {
    int[] union;
    public Union(int n) {
        union = new int[n];
    }
    public void init(int n) {
        
for (int i = 0; i < n; ++i) {
            union[i] = i;
        }
    }
    public void unite(int x, int y) {
        
union[find(y)] = find(x);
    }
    public int find(int x) {
        if (union[x] == x) return x;
        return union[x] = find(union[x]);
    }
}
