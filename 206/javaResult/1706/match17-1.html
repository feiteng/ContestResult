class Solution {
    int [] fa;
    int find( int x){
        return fa[x] = (fa[x] == x? x : find(fa[x]));
    
}
    
    void merge(int x, int y){
        int a = find(x), b = find(y);
        if(a != b){
            fa[b] = a;
        }
    }
    
    public int minCostConnectPoints(int[][] points) {
       
        PriorityQueue<int []> pq = new PriorityQueue<int []>((a,b)->{
            return a[0] - b[0];
        });
        int n = points.length;
        fa = new int[n];
        for(int i = 0; i < n; i++){
            fa[i] = i;
        }
        
for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                int dis = Math.abs(points[i][0] - points[j][0]) +Math.abs(points[i][1] - points[j][1]);
                pq.add(new int[]{dis,i*1000 + j});
            }
        }
        int res = 0;
        while(!pq.isEmpty()){
            int [] cur = pq.poll();
            int i = cur[1] / 1000;
            int j = cur[1] % 1000;
            if(find(i) != find(j))
{
                res += cur[0];
                merge(i,j);
            }
        }
        
        
        return res;
    }
    
}
