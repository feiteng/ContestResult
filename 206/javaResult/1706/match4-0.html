class Solution {
    public int minCostConnectPoints(int[][] points) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[2] - b[2]);
        UF uf = new UF(points.length);
        for (int i = 0; i < points.length; i++) {
            for (int j = i + 1; j < points.length; j++) {
                int val = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                pq.add(new int[]{i, j, val});
            }
        }
        int size = points.length - 1;
        int val = 0;
        int curS = 0;
        while (!pq.isEmpty() && curS != size) {
            int[] ent = pq.poll();
            if (!uf.connected(ent[0], ent[1])) {
                val += ent[2];
                curS++;
                uf.union(ent[0], ent[1]);
            }
        }
        return val;
        
    }
    
    
    class UF {
        
        int size;
        int[] id;
        
        UF(int size) {
            this.id = new int[size];
            for (int i = 0; i < size; i++) {
                this.id[i] = i;
            }
        }
        
        int find(int index) {
            if (id[index] != index) {
                int ret = find(id[index]);
                id[index] = ret;
            }
            return id[index];
        }
        
        void union(int i, int j) {
            id[find(i)] = find(j);
        }
        
        boolean connected(int i, int j) {
            return find(i) == find(j);
        }
        
    }
}
