class Solution {

    class Edges{
        int p1;
        int p2;
        int weight;
        public Edges(int p1, int p2, int weight){
            this.p1 = p1;
            this.p2 = p2;
            this.weight = weight;
        }
    }
    public int minCostConnectPoints(int[][] points) {
        List<Edges> edges = new ArrayList<>();
        for(int i=0;i<points.length;++i){
            for(int j=i+1;j<points.length;++j){
                edges.add(new Edges(i,j,distance(points[i],points[j])));
            }
        }
        Collections.sort(edges,(a,b)->a.weight-b.weight);
        int[] arr = new int[points.length];
        for(int i=0;i<points.length;++i){
            arr[i]=i;
        }
        int num = 0;
        int n = points.length;
        int ans = 0;
        for(int i=0;i<edges.size();++i){
            Edges e = edges.get(i);
            if (union(e.p1,e.p2,arr)){
                ++num;
                ans +=e.weight;
            }
            if(num==n-1){
                return ans;
            }
        }
        return ans;
    }
    private int distance(int[] p1, int[] p2){
        return Math.abs(p1[0]-p2[0])+Math.abs(p1[1]-p2[1]);
    }
    private int find(int a,int[] arr){
        if(arr[a]==a)
            return a;
        arr[a] = find(arr[a],arr);
        return arr[a];
    }
    private boolean union(int a,int b,int[] arr){
        int pa = find(a,arr);
        int pb = find(b,arr);
        if(pa==pb)
            return false;
        arr[pb] = pa;
        return true;

    }
}
