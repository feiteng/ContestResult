class Solution {
    public int minCostConnectPoints(int[][] points) {
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> {
            return dist(points[a[0]], points[a[1]]) - dist(points[b[0]], points[b[1]]);
        });
        
        int l = points.length;
        
for (int i = 0; i < l; i++) {
            for (int j = i + 1; j < l; j++) {
                pq.add(new int[]{i, j});
            }
        }
        int re = 0;
        UF u = new UF(points.length);
        while (!pq.isEmpty()) {
            int[] pair = pq.poll();//i, j
            int i = pair[0];
            int j = pair[1];
            if (u.findParent(i) == u.findParent(j)) {
                continue;
            }
            u.union(i, j);
            re += dist(points[i], points[j]);
        
}    
        return re;
    }
    
    
    
    private int dist(int[] a, int[] b) {
        return Math.abs(a[0]-b[0]) + Math.abs(a[1]-b[1]);
    }
    
    public class UF {
        int[] arr;
        int count;
        public UF(int n) {
            arr = new int[n];
            for (int i = 0; i < n; i++) {
                arr[i] = i;
            }
        
}
        
        public int findParent(int i) {
            int p = arr[i];
            while (p != i) {
                i = p;
                p = arr[i];
            }
            return p;
        
}
        
        public void union(int i, int j) {
            int a = findParent(i);
            int b = findParent(j);
            if (a != b) {
                count++;
            }
            arr[a] = b;
        }
    }
}
