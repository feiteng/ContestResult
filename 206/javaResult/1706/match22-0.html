class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        UnionFind uf = new UnionFind(n);
        int[][] weights = new int[n][n];
        var pq = new PriorityQueue<int[]>(n * n, (a, b)->Integer.compare(weights[a[0]][a[1]], weights[b[0]][b[1]]));
        
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(i != j){
                    weights[i][j] = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                    pq.add(new int[]{i, j});
                }
            }
        }
        int result = 0;
        while(uf.groupCount > 1){
            int[] edge = pq.poll();
            if(uf.union(edge[0], edge[1])){
                result += weights[edge[0]][edge[1]];
            }
        }
        return result;
    }
    
    
    private class UnionFind{
        int[] parents;
        int groupCount;
        
        UnionFind(int n){
            parents = new int[n];
            groupCount = n;
            for(int i = 0; i < n; i++){
                parents[i] = i;
            }
        }
        
        private boolean union(int a, int b){
            int aParent = find(a);
            int bParent = find(b);
            
            if(aParent == bParent){
                return false;
            }
            else{
                groupCount--;
                parents[aParent] = bParent;
                return true;
            }
        }
        
        private int find(int node){
            if(parents[node] != node){
                parents[node] = find(parents[node]);
            }
            return parents[node];
        }
    }
}
