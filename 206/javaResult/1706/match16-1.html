import java.util.ArrayList;
import java.util.List;
import java.util.PriorityQueue;

class Solution {
    public int minCostConnectPoints(int[][] points) {

        PriorityQueue<int[]> edges = new PriorityQueue<>((a, 
b) -> a[2] - b[2]);
        int n = points.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int d = dist(points[i], points[j]);
                edges.add(new int[]{i, j, d});
            }
        }


        UnionFind uf = new UnionFind(n);
        int ans = 0;

        while (uf.cap > 1) {

            int[] edge = edges.poll();
            
int v = edge[0], w = edge[1], cost = edge[2];
            if (!uf.isUnited(v, w)) {
                uf.unite(v, w);
                ans += cost;
            }
        }
        return ans;
    }

    private int dist(int[] s, int[] t) {
        int dx = Math.abs(s[0] - t[0]);
        int dy = Math.abs(s[1] - t[1]);
        return dx + dy;
    }

    private static class UnionFind {
        private int cap;
        private int[] root;

        private UnionFind(int cap) {
            root = new int[cap];
            this.cap = cap;
            
for (int i = 0; i < root.length; i++) {
                root[i] = i;
            }
        }

        private int find(int s) {
            
while (root[s] != s) {
                root[s] = root[root[s]];
                s = root[s];
            }
            return s;
        }

        private void unite(int s, int t) {
            int rs = find(s);
            int rt = find(t);
            if (rs == rt) {
                return;
            }
            root[rs] = rt;
            cap--;
        
}

        private boolean isUnited(int s, int t) {
            return find(s) == find(t);
        }
    }
}
