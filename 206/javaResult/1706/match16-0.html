class Solution {
    class UF {
        int[] parents;
        UF (int n) {
            parents = new int[n];
            for (int i = 0; i < n; i += 1) {
                parents[i] = i;
            }
        }
        int find(int x) {
            int fa = x;
            while (parents[fa] != fa) {
                fa = parents[fa];
            }
            while (x != fa) {
                parents[x] = fa;
                x = parents[x];
            }
            return fa;
        }
        void union(int x, int y) {
            int paX = find(x);
            int paY = find(y);
            if (paX != paY) {
                parents[paX] = paY;
            }
        }
        boolean isConnected(int x, int y) {
            return find(x) == find(y);
        }
    }
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        UF uf = new UF(n);
        PriorityQueue<int[]> pq = new PriorityQueue<>(8, (a, b) -> a[0] - b[0]);
        for (int i = 0; i < n; i += 1) {
            for (int j = i + 1; j < n; j += 1) {
                int dis = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                pq.offer(new int[] {dis, i, j});
            }
        }
        int ret = 0;
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            int u = cur[1];
            int v = cur[2];
            if (!uf.isConnected(u, v)) {
                ret += cur[0];
                uf.union(u, v);
            }
        }
        return ret;
    }
}
