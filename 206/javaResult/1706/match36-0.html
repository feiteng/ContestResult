class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        int[][] dis = new int[n][n];
        PriorityQueue<int[]> pq = new PriorityQueue<>(Comparator.comparingInt(e -> e[0]));
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                dis[i][j] = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                dis[j][i] = dis[i][j];
                pq.offer(new int[]{dis[i][j], i, j});
            }
        }
        int[] parent = new int[n];
        int[] rank = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
        }
        Arrays.fill(rank, 1);
        int ans = 0;
        int k = 0;
        while (k < n - 1) {
            int[] arr = pq.poll();
            int pa = find(parent, arr[1]);
            int pb = find(parent, arr[2]);
            if (pa == pb) {
                continue;
            }
            union(parent, rank, pa, pb);
            ans += arr[0];
            k++;
        }
        return ans;
    }

    private int find(int[] parent, int x) {
        return parent[x] == x ? x : (parent[x] = find(parent, parent[x]));
    }

    private void union(int[] parent, int[] rank, int a, int b) {
        a = find(parent, a);
        b = find(parent, b);
        if (a == b) {
            return;
        }
        int ra = rank[a];
        int rb = rank[b];
        if (ra > rb) {
            parent[b] = a;
            rank[a] += rb;
        } else {
            parent[a] = b;
            rank[b] += ra;
        }
    }
}
