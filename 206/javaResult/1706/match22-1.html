class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        
Queue<int[]> q = new PriorityQueue<>((a, b) -> Integer.compare(a[2], b[2]));
        for(int i = 0; i < n; ++i) {
            for(int j = i + 1; j < n; ++j) {
                q.offer(new int[] {i, j
                    , Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1])});
            }
        }
        UF uf = new UF(n);
        int result = 0;
        for(int x = n; x > 1; ) 
{
            int[] curr = q.poll();
            if (uf.union(curr[0], curr[1])) {
                --x;
                result += curr[2];
            }
        }
        
return result;
    }
    
    static class UF {
        
        int[] sizes;
        int[] parents;
        
        UF(int n) {
            sizes = new int[n];
            parents = new int[n];
            for(int i = 0; i < n; ++i) {
                sizes[i] = 1;
                parents[i] = i;
            }
        
}
        
        int find(int i) {
            if (parents[i] != i) {
                parents[i] = find(parents[i]);
            }
            return parents[i];
        
}
        
        boolean union(int i, int j) {
            int x = find(i);
            int y = find(j);
            if (x == y) {
                return false;
            }
            if (sizes[x] < sizes[y]) {
                int tmp = x;
                x = y;
                y = tmp;
            }
            sizes[x] += sizes[y];
            parents[y] = x;
            return true;
        }
    }
}
