class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        List<int[]> dist = new ArrayList();
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int d = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                dist.add(new int[]{i, j, d});
            }
        }
        Collections.sort(dist, (a, b)->a[2]-b[2]);
        UF uf = new UF(n);
        int cost = 0;
        for(int i=0;i<dist.size();i++){
            int x = dist.get(i)[0], y = dist.get(i)[1], c = dist.get(i)[2];
            if(uf.union(x, y)){
                cost += c;
            }
        }
        return cost;
    }
    
    
    class UF {
        int[] parents;
        UF(int N){
            parents = new int[N];
            for(int i=0;i<N;i++){
                parents[i] = i;
            }
        }
        
        
        public int find(int x){
            if(x == parents[x]) return x;
            return parents[x] = find(parents[x]);
        }
        
        
        public boolean union(int x, int y){
            int rx = find(x), ry = find(y);
            if(rx != ry){
                parents[rx] = ry;
                return true;
            }
            return false;
        }
    }
}
