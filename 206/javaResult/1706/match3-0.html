class Solution {
    
    int zz=0;
    int oo=1;
    
    private class naya{
         int[] baap;

         public naya(int N) {
             baap = new int[N];
             for (int i = zz; i < N; ++i) {
                 baap[i] = i;
             }
         }

         public int ff(int x) {
             if (baap[x] != x) baap[x] = ff(baap[x]);
             return baap[x];
         }

         public void uu(int x, int y) {
             baap[ff(x)] = ff(y);
         }
     }

     public int minCostConnectPoints(int[][] points) {
         if (points == null) return 0;
         if(points.length <= oo) return 0;
         
         
         int ll = points.length;
         
         int ans = zz, cc = ll;
         
         int[][] dd = new int[ll][ll];
         
         PriorityQueue<int[]> q = new PriorityQueue<>((x, y) -> (dd[x[0]][x[1]] - dd[y[0]][y[1]]));
         for (int i = zz; i < ll; ++i) {
             for (int j = i; j < ll; ++j) {
                 int d = Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1]);
                 dd[i][j] = d;
                 dd[j][i] = d;
                 q.add(new int[]{i,j});
             }
         }
         naya gh = new naya(ll);
         
         while (!q.isEmpty() && cc != zz) {
             
             int[] tt = q.poll();
             
             if (gh.ff(tt[zz]) != gh.ff(tt[oo])) {
                 ans += dd[tt[0]][tt[oo]];
                 gh.uu(tt[zz], tt[1]);
             }
         }
         return ans;
     }
}
