class Solution {
    public int minCostConnectPoints(int[][] points) {
        int len = points.length;
        int[] parents = new int[len];
        for (int i = 0; i < len; i ++) {
            parents[i] = i;
        }
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((i, j) -> {
            int p0 = i[0], p1 = i[1], q0 = j[0], q1 = j[1];
            int disP = (int)Math.abs(points[p0][0] - points[p1][0]) + 
                       (int)Math.abs(points[p0][1] - points[p1][1]);
            
            int disQ = (int)Math.abs(points[q0][0] - points[q1][0]) + 
                       (int)Math.abs(points[q0][1] - points[q1][1]);
            
            return disP - disQ;
        });
        
        for (int i = 0; i < len; i ++) {
            for (int j = i + 1; j < len; j ++) {
                pq.offer(new int[]{i, j});
            }
        }
        
        int result = 0;
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            int parent0 = find(cur[0], parents), parent1 = find(cur[1], parents);
            if (parent0 != parent1) {
                int dis = (int)Math.abs(points[cur[0]][0] - points[cur[1]][0]) + 
                          (int)Math.abs(points[cur[0]][1] - points[cur[1]][1]);
                // System.out.println(cur[0] + "," + cur[1] + "," + dis);
                result += dis;
                parents[parent0] = parent1;
            }
        }
        
        return result;
    }
    
    private int find(int point, int[] parents) {
        if (point == parents[point]) {
            return point;
        }
        
        parents[point] = find(parents[point], parents);
        return parents[point];
    }
}
