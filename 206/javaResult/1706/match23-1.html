class Solution {
    
    private int[] parents;
    
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>(){
            @Override
            public int compare(int[] a, int[] b){
                return a[2] - b[2];
            }
        });
        parents = new int[n];
        for(int i = 0 ; i < n ; i++){
            parents[i] = i;
        }
        
for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < n ; j++){
                queue.add(new int[]{i, j, Math.abs(points[i][0]- points[j][0]) + Math.abs(points[i][1]- points[j][1])});
            }
        }
        int res = 0;
        while(!queue.isEmpty())
{
            int[] cur = queue.poll();
            if(union(cur[0], cur[1])){
                res += cur[2];
            }
        }
        
return res;
    }
    
    private boolean union(int u, int v){
        int pu = find(u);
        int pv = find(v);
        if(pu == pv){
            return false;
        }
        parents[pu] = pv;
        return true;
    
}
    
    private int find(int u){
        if(parents[u] != u){
            parents[u] = find(parents[u]);
        }
        return parents[u];
    }
}
