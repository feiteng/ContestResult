class Node {
    int i;
    int j;
    int dist;
    
    public Node(int i, int j, int dist) {
        this.i = i;
        this.j = j;
        this.dist = dist;
    }
}

class DSU {
    int[] p;
    
    public DSU(int n) {
        this.p  = new int[n];
        for(int i = 0; i < p.length; i++) p[i] = i;
    }
    
    public int parent(int x) {
        if(p[x] != x) {
            p[x] = parent(p[x]);
        }
        return p[x];
    }
    public boolean union(int x, int y) {
        int px = parent(x);
        int py = parent(y);
        if(px == py) return false;
        p[py] = px;
        return true;
        
    }
}
class Solution {
    public int dist(int[] p1, int[] p2) {
        return Math.abs(p1[0] - p2[0]) + Math.abs(p1[1] - p2[1]);
    }
    public int minCostConnectPoints(int[][] points) {
        PriorityQueue<Node> pq = new PriorityQueue<Node>((Node a, Node b) -> a.dist - b.dist);
        int n = points.length;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                pq.offer(new Node(i,j,dist(points[i],points[j])));
            }
        }
        
        DSU dsu = new DSU(n);
        int ans = 0;
        while(!pq.isEmpty()) {
            Node node = pq.remove();
            if(dsu.union(node.i, node.j)) {
                ans += node.dist;
            }
        }
        
        return ans;
        
                         
                         
    }
}
