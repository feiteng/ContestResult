class Solution {
    class Item{
        int i;
        int j;
        int dist;
        public Item(int i, int j, int dist) {
            this.i = i;
            this.j = j;
            this.dist = dist;
        }
    }
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        List<Item> list = new LinkedList<>();
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                list.add(new Item(i, j, Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1])));
            }
        }
        Collections.sort(list, (a, b) -> a.dist - b.dist);
        int[] parent = new int[n];
        for(int i = 0; i < n; i++) parent[i] = i;
        int res = 0;
        
        for(Item item: list) {
            int i = item.i, j = item.j;
            if(find(parent, i) != find(parent, j)) {
                res += item.dist;
                union(parent, i, j);
            }
        }
        return res;
    }
    
    private int find(int[] parent, int x) {
        if(parent[x] != x) parent[x] = find(parent, parent[x]);
        return parent[x];
    }
    
    private void union(int[] parent, int x, int y) {
        int xr = find(parent, x), yr = find(parent, y);
        if(xr != yr) parent[yr] = xr;
    }
}
