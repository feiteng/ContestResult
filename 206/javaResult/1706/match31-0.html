class Solution {
    int find(int i) {
        if (i == par[i]) return i;
        return par[i] = find(par[i]);
    }
    boolean union(int i, int j) {
        int pi = find(i), pj = find(j);
        if (pi == pj) return false;
        par[pi] = pj;
        return true;
    }
    int[] par;
    public int minCostConnectPoints(int[][] ps) {
        par = new int[ps.length];
        for (int i = 0; i < par.length; i++) par[i] = i;
        List<int[]> es = new ArrayList<>();
        for (int i = 0; i < par.length; i++) {
            for (int j = i + 1; j < par.length; j++) {
                es.add(new int[]{i, j, dis(ps[i], ps[j])});
            }
        }
        Collections.sort(es, (a, b)->Integer.compare(a[2], b[2]));
        int res = 0;
        for (int[] e : es) {
            if (union(e[0], e[1])) res += e[2];
        }
        return res;
    }
    int dis(int[] a, int[] b) {
        return Math.abs(a[0] - b[0]) + Math.abs(a[1] - b[1]);
    }
}
