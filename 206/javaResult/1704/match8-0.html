class Solution {
    public int numSpecial(int[][] mat) {
        final int m = mat.length, n = mat[0].length;
        int[] zm = new int[m], zn = new int[n];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == 1) {
                    ++zm[i];
                    ++zn[j];
                }
            }
        }
        int cnt = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == 1 && zm[i] == 1 && zn[j] == 1) {
                    ++cnt;
                }
            }
        }
        return cnt;
    }
}
