class Solution {
    public int numSpecial(int[][] mat) {
        int count = 0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[i].length; j++) {
                if (mat[i][j] == 1 && fuckThem(i, j, mat)) {
                    count++;
                }
            }
        }
        return count;
    }
    
    boolean fuckThem(int i, int j, int[][] mat) {
        for (int k = 0; k < mat.length; k++) {
            if (mat[k][j] == 1 && k != i) {
                return false;
            }
        }
        
        for (int k = 0; k < mat[0].length; k++) {
            if (mat[i][k] == 1 && k != j) {
                return false;
            }
        }
        return true;
    }
}
