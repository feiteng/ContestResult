class Solution {
    public int numSpecial(int[][] mat) {
        int[] row = new int[mat.length];
        int[] col = new int[mat[0].length];
        for (int j = 0; j < mat.length; j++) {
            for (int k = 0; k < mat[0].length; k++) {
                row[j] += mat[j][k];
                col[k] += mat[j][k];
            
}
        }
        int ret = 0;
        for (int j = 0; j < mat.length; j++) {
            for (int k = 0; k < mat[0].length; k++) {
                if (row[j] == 1 && col[k] == 1 && mat[j][k] == 1) {
                    ret++;
                }
            }
        }
        return ret;
    }
}
