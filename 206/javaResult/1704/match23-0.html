class Solution {
        public int numSpecial(int[][] mat) {
            int row = mat.length;
            int col = mat[0].length;
            int[] cols = new int[col];
            int[] rows = new int[row];
            for (int i=0;i<row;i++) {
                for (int j=0;j<col;j++) {
                    rows[i] += mat[i][j];
                    cols[j] += mat[i][j];
                }
            }
            int count = 0;
            for (int i=0;i<row;i++) {
                for (int j=0;j<col;j++) {
                    if (mat[i][j] == 1 && rows[i] == 1 && cols[j] == 1) {
                        count ++;
                    }
                }
            }
            return count;
        }
    }
