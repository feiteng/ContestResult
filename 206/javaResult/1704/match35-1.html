import java.util.*;


public class Solution {
    public int numSpecial(int[][] mat) {
        int ans = 0;
        int n = mat.length;
        int m = mat[0].length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 0) {
                    continue;
                }
                int row = 0;
                for (int k = 0; k < m; k++) {
                    row += mat[i][k];
                }
                int col = 0;
                
for (int k = 0; k < n; k++) {
                    col += mat[k][j];
                }
                if (row == 1 && col == 1) {
                    ans++;
                }
            }
        }
        return ans;
    }
}
