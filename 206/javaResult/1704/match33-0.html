class Solution {
    public int numSpecial(int[][] mat) {
        int[] sumHorizontal = new int[mat.length];
        int[] sumVertical = new int[mat[0].length];
        
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] == 1) {
                    sumHorizontal[i] += 1;
                    sumVertical[j] += 1;
                }
            }
        }
        
        int res = 0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] == 1 && sumHorizontal[i] == 1 && sumVertical[j] == 1) {
                    res++;
                }
            }
        }
        return res;
    }
}
