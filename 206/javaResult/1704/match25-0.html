class Solution {
    public int numSpecial(int[][] mat) {
        int l = mat.length;
        int ll  =mat[0].length;
        int c[]  = new int[ll];
        int r[] = new int[l];
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                r[i]+= mat[i][j];
                c[j]+= mat[i][j];
            }
        }
        int an = 0;
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                if(r[i]==1&&c[j]==1&&mat[i][j]==1){
                    an++;
                }       
            }
        }
        return an;
        
    }
}
