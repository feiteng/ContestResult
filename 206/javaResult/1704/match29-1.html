class Solution {
    public int numSpecial(int[][] mat) {
        int n = mat.length; int m = mat[0].length;
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    boolean ok = true;
                    for (int k = 0; k < m; k++) {
                        if (k == j) continue;
                        if (mat[i][k] == 1) ok = false;
                    }
                    for (int k = 0; k < n; k++) {
                        if (k == i) continue;
                        if (mat[k][j] == 1) ok = false;
                    }
                    if (ok) ++res;
                }
            }
        }
        return res;
    }
}
