class Solution {
    public int numSpecial(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        int[] rowSum = new int[m];
        int[] colSum = new int[n];
        for (int i = 0; i < m; i += 1) {
            for (int j = 0; j < n; j += 1) {
                rowSum[i] += mat[i][j];
                colSum[j] += mat[i][j];
            }
        }
        int ret = 0;
        for (int i = 0; i < m; i += 1) {
            for (int j = 0; j < n; j += 1) {
                if (rowSum[i] == 1 && colSum[j] == 1 && mat[i][j] == 1) {
                    //System.out.println(i + "," + j);
                    ret += 1;
                }
            }
        }
        return ret;
    }
}
