class Solution {
    public boolean isTransformable(String s, String t) {
        int[] sCount = new int[10];
        List<Integer>[] sPrefix = new ArrayList[10];
        for (int i = 0; i < sPrefix.length; i++) {
            sPrefix[i] = new ArrayList<>();
        }
        for (int i = 0; i < s.length(); i++) {
            int num = s.charAt(i) - '0';
            sCount[num]++;
            int tmp = 0;
            for (int j = 0; j < num; j++) {
                tmp += sCount[j];
            }
            sPrefix[num].add(tmp);
        }

        int[] tCount = new int[10];
        List<Integer>[] tPrefix = new ArrayList[10];
        for (int i = 0; i < tPrefix.length; i++) {
            tPrefix[i] = new ArrayList<>();
        }
        for (int i = 0; i < t.length(); i++) {
            int num = t.charAt(i) - '0';
            tCount[num]++;
            int tmp = 0;
            for (int j = 0; j < num; j++) {
                tmp += tCount[j];
            }
            tPrefix[num].add(tmp);
        }

        for (int i = 0; i < 10; i++) {
            if (sPrefix[i].size() != tPrefix[i].size()) {
                return false;
            }
            for (int j = 0; j < sPrefix[i].size(); j++) {
                if (sPrefix[i].get(j) > tPrefix[i].get(j)) 
{
                    return false;
                }
            }
        }
        return true;
    }
}
