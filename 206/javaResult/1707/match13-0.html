class Solution {
    public boolean isTransformable(String s, String t) {
        int n = s.length();
        int[] cous = new int[256];
        for (char c : s.toCharArray())
            cous[c]++;
        for (char c : t.toCharArray())
            cous[c]--;
        for (int i : cous) {
            if (i!=0)
                return false;
        }
        
        ArrayList<ArrayDeque<Integer>> pos = new ArrayList<>();
        for (int i = 0; i < 10; ++i)
            pos.add(new ArrayDeque<>());
        for (int i = 0; i < n; ++i) {
            int j = (int)(s.charAt(i)-'0');
            pos.get(j).add(i);
        }
        for (int i = 0; i < 10; ++i)
            pos.get(i).add(1000000000);
        
        for (char c : t.toCharArray()) {
            int d = (int)(c-'0');
            int np = pos.get(d).removeFirst();
            for (int d2 = 0; d2 < d; ++d2) {
                if (pos.get(d2).peekFirst()<np)
                    return false;
            }
        }
        return true;
    }
}
