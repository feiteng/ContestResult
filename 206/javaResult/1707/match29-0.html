class Solution {
    public boolean isTransformable(String s, String t) {
        char[] sl = s.toCharArray();
        char[] tl = t.toCharArray();
        int[] pos = new int[256];
        for (char c : sl)
            pos[c]++;
        for (char c : tl)
            pos[c]--;
        for (char c = '0'; c <= '9'; c++)
            if (pos[c] != 0)
                return false;
        int n = sl.length;
        int[] vi = new int[n];
        for (char c : tl) {
            while (!(vi[pos[c]] == 0 && sl[pos[c]] == c)) {
                if (vi[pos[c]] == 0 && sl[pos[c]] < c)
                    return false;
                pos[c]++;
            }
            vi[pos[c]] = 1;
            pos[c]++;
        }
        return true;
    }
}
