import java.util.ArrayList;
import java.util.HashMap;
import java.util.LinkedList;

class Solution {
    public boolean isTransformable(String s, String t) {
        if(s.length()!=t.length()){
            return false;
        }
        int[] cnt=new int[10];


        for(int i=0;i<s.length();i++){
            cnt[s.charAt(i)-'0']++;
            cnt[t.charAt(i)-'0']--;
        }
        HashMap<Integer, LinkedList<Integer>> map=new HashMap<>();

        for(int i=0;i<10;i++){
            map.put(i,new LinkedList<>());
        }

        for(int i=0;i<s.length();i++){
            map.get(s.charAt(i)-'0').add(i);
        }

        for(int i=0;i<t.length();i++){
            int cur=t.charAt(i)-'0';

            int tindex=Integer.MAX_VALUE;
            for(int nn=0;nn<=cur;nn++){
                if(!map.get(nn).isEmpty()){
                    tindex=Math.min(tindex,map.get(nn).get(0));
                }
            }
            if(map.get(cur).isEmpty()||tindex!=map.get(cur).get(0)){
                return false;
            }
            map.get(cur).remove(0);
        }
        return true;
    }
}
