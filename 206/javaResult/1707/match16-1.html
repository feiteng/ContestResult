class Solution {
    public boolean isTransformable(String s, String t) {
        
Queue<Integer>[] q = new Queue[10];
        for(int i = 0; i < 10; i++) q[i] = new LinkedList<>();
        int n = s.length();
        for(int i = 0; i < n; i++)
        {
            int val = s.charAt(i) - '0';
            q[val].offer(i);
        }
        
        for(int i = 0; i < n; i++)
        {
            int val = t.charAt(i) - '0';
            // if all value has position greater than than val then it's possible
            if(q[val].isEmpty()) return false;
            int pos = q[val].peek();
            for(int j = 0; j < val; j++)
            {
                if(!
q[j].isEmpty() && q[j].peek() < pos) return false;
            }
            q[val].poll();
        }
        return true;
    }
}
