class Solution {
    public boolean isTransformable(String s, String t) {
        Map<Integer, List<int[]>> smap = process(s);
        Map<Integer, List<int[]>> tmap = process(t);
        for (int i = 0; i < 10; i++) {
            if (smap.get(i).size() != tmap.get(i).size()) {
                return false;
            }
            int size = smap.get(i).size();
            for (int j = 0; j < size; j++) {
                if (!valid(smap.get(i).get(j), tmap.get(i).get(j), i)) {
                    return false;
                }
            }
        }
        return true;
    }

    private Map<Integer, List<int[]>> process(String s) {
        Map<Integer, List<int[]>> map = new HashMap<>();
        for (int i = 0; i < 10; i++) {
            map.put(i, new ArrayList<>());
        }
        int[] prev = new int[10];
        for (int i = 0; i < s.length(); i++) {
            int digit = s.charAt(i) - '0';
            map.get(digit).add(prev.clone());
            prev[digit]++;
        }
        return map;
    }
    
    private boolean valid(int[] sa, int[] ta, int v) {
        for (int i = v + 1; i < 10; i++) {
            if (ta[i] > sa[i]) {
                return false;
            }
        }
        return true;
    }
}
