        ArrayList<Queue<Integer>> qs = new ArrayList<>();
        for (int j = 0; j < 10; j++) {
            qs.add(new LinkedList<>());
        }
        for (int j = s.length() - 1; j >= 0; j--) 
{
            int digit = s.charAt(j)-'0';
            qs.get(digit).add(j);
        }
        for (int j = t.length() - 1; j >= 0; j--) {
            int digit = t.charAt(j) - '0';
            if (qs.get(digit).size() == 0) {
                return false;
            }
            
int pos = qs.get(digit).poll();
            for (int k = digit + 1; k< 10; k++) {
                if (qs.get(k).size() > 0 && qs.get(k).peek() > pos) {
                    return false;
                
}
