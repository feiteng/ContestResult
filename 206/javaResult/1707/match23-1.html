class Solution {

    private LinkedList<Integer>[] stacks = new LinkedList[10];

    private boolean check(int digit , int pos) {
        
for (int i = digit + 1;i < 10;i ++) {
            if (!stacks[i].isEmpty() && stacks[i].peek() > pos) {
                return false;
            }
        }
        return true;
    }

    public boolean isTransformable(String s, String t) {
        
for (int i = 0;i < stacks.length;i ++) {
            stacks[i] = new LinkedList<>();
        }
        for (int i = 0;i < s.length();i ++) {
            int digit = s.charAt(i) - '0';
            stacks[digit].push(i);
        }
        for (int i = t.length() - 1;i >= 0;i --) {
            int digit = t.charAt(i) - '0';
            if (stacks[digit].isEmpty()) {
                return false;
            } else {
                int pos = stacks[digit].pop();
                if (!check(digit , pos)) {
                    return false;
                }
            }
        }
        return true;
    }
}
