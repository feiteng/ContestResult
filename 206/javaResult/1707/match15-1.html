class Solution {
    public boolean isTransformable(String ss, String tt) {
        char[] s = ss.toCharArray();
        char[] t = tt.toCharArray();
        int n = s.length;
        int[] cnt1 = new int[10], cnt2 = new int[10];
        int a[] = new int[n], b[] = new int[n];
        for(int i = 0; i < n; i++) 
        {
            a[i] = s[i] - '0';
            b[i] = t[i] - '0';
            cnt1[a[i]]++;
            cnt2[b[i]]++;
        
}
        for(int i = 0; i < 10; i++) if(cnt1[i] != cnt2[i]) return false;
        Deque<Integer> dq[] = new Deque[10];
        for(int i = 0; i < 10; i++) dq[i] = new ArrayDeque<>();
        for(int i = 0; i < n; i++)
            dq[a[i]].addLast(i);
        for(int i = 0; i < n; i++) 
{
            int cur = b[i];
            for(int j = 0; j < cur; j++)
            {
                if(!dq[j].isEmpty() && dq[j].peekFirst() < dq[cur].peekFirst())
                    return false;
            }
            dq[cur].pollFirst();
        }        
        return true;
    }
}
