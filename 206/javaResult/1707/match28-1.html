class Solution {
    public boolean isTransformable(String s, String t) {
        int []W1 = new int[10];
        int []W2 = new int[10];
        for (int i = 0; i < t.length(); i++) {
            int ds = s.charAt(i) - '0';
            int dt = t.charAt(i) - '0';
            if (W2[ds] > 0) W2[ds]--;
            else W1[ds]++;
            if (W1[dt] > 0) W1[dt]--;
            else W2[dt]++;
            for (int j = 0; j < 10; j++) {
                if (W2[j] > 0) 
{
                    for (int k = 0; k < j; k++) {
                        if (W1[k] > 0)
                            return false;
                    }
                }
            }
        
}
        for (int i = 0; i < 10; i++) {
            if (W1[i] != 0 || W2[i] != 0)
                return false;
        }
        return true;
    }
}
