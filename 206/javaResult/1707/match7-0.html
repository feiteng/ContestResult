import java.util.LinkedList;

class Solution {
    public boolean isTransformable(String s, String t) {

        int[] cnt = new int[10];
        for (char c : s.toCharArray()) {
            cnt[c - '0']++;
        }
        int[] cnt1 = new int[10];
        for (char c : t.toCharArray()) {
            cnt1[c - '0']++;
        }

        for (int i = 0; i < 10; i++) {
            if (cnt[i] != cnt1[i]) {
                return false;
            }
        }


        LinkedList<Integer>[] lists = new LinkedList[10];
        for (int i = 0; i < 10; i++) {
            lists[i] = new LinkedList<>();
        }
        for (int i = 0; i < t.length(); i++) {
            int d = s.charAt(i) - '0';
            lists[d].addLast(i);
        }


        for (char c : t.toCharArray()) {
            int d = c - '0';
            int next = lists[d].pollFirst();
            for (int i = 0; i < d; i++) {
                if (!lists[i].isEmpty() && lists[i].peekFirst() < next) {
                    return false;
                }
            }
        }


        return true;


    }
