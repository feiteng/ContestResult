class Solution {
    public boolean isTransformable(String s, String t) {
        
        int[] freqS = new int[10];
        int[] freqT = new int[10];
        
        Map<Integer, List<Integer>> mapS = new HashMap<Integer, List<Integer>>();
        Map<Integer, List<Integer>> mapT = new HashMap<Integer, List<Integer>>();
        
        for (int i=0; i<10; i++) {
            mapS.put(i, new ArrayList<Integer>());
            mapT.put(i, new ArrayList<Integer>());
        }
        
        for (int i=0; i<s.length(); i++) {
            int vS = s.charAt(i)-'0';
            int vT = t.charAt(i)-'0';
            freqS[vS]++;
            freqT[vT]++;
            
            mapS.get(vS).add(countSmaller(freqS, vS));
            mapT.get(vT).add(countSmaller(freqT, vT));
        }
        
        for (int i=0; i<10; i++) {
            if (freqS[i] != freqT[i]) return false;
        }
        
        for (int i=0; i<10; i++) {
            for (int j=0; j<mapS.get(i).size(); j++) {
                if (mapS.get(i).get(j) > mapT.get(i).get(j)) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    public int countSmaller(int[] freq, int v) {
        int res = 0;
        for (int i=0; i<v; i++) {
            res += freq[i];
        }
        return res;
    }
}
