class Solution {
    public boolean isTransformable(String s, String t) {
        int[] scnt = new int[10];
        int[] tcnt = new int[10];
        List<PriorityQueue<Integer>> sGreater = new ArrayList<>();
        List<PriorityQueue<Integer>> tGreater = new ArrayList<>();
        for(int i = 0; i < 10; i++) {
            sGreater.add(new PriorityQueue<Integer>());
            tGreater.add(new PriorityQueue<Integer>());
        }
        for(int i = 0; i < s.length(); i++) {
            int chs = s.charAt(i) - '0';
            int cht = t.charAt(i) - '0';
            scnt[chs]++;
            tcnt[cht]++;
            int gtS = 0, gtT = 0;
            for(int j = chs+1; j < 10; j++)
                gtS += scnt[j];
            for(int j = cht+1; j < 10; j++)
                gtT += tcnt[j];
            PriorityQueue<Integer> pqs = sGreater.get(chs);
            PriorityQueue<Integer> pqt = 
tGreater.get(cht);
            pqs.add(gtS);
            pqt.add(gtT);
        }
        for(int i = 0; i < 10; i++) {
            if(scnt[i] != tcnt[i])
                return false;
        }
        for(int i = 0; i < 10; i++) {
            PriorityQueue<Integer> pqs = sGreater.get(i);
            PriorityQueue<Integer> pqt = tGreater.get(i);
            while(!pqs.isEmpty()) {
                int tops = pqs.poll();
                int topt = pqt.poll();
                if(topt > tops)
                    return false;
            }
        }
        return true;
    }
}
