class Solution {
    public boolean isTransformable(String s, String t) {
        int len = s.length();
        
List<int[]>[] lists = new ArrayList[10];
        for (int i = 0; i < 10; ++i) {
            lists[i] = new ArrayList<>();
        }

        // Find out how many numbers that is less than this number
        // Before this number appear
        int[] count = new int[10];
        
for (char c : s.toCharArray()) {
            int p = c - '0';
            count[p] += 1;
            int[] tmp = new int[p + 1];
            for (int i = 0; i <= p; ++i) {
                tmp[i] = count[i];
            }
            lists[p].add(tmp);
        }

        // Verify
        count = new int[10];
        for (char c : t.toCharArray()) {
            int p = c - '0';
            count[p] += 1;
            if (lists[p].size() < count[p]) {
                return false;
            }
            int[] last = lists[p].get(count[p] - 1);
            for (int i = 0; i <= p; ++i) {
                if (last[i] > count[i]) {
                    return false;
                }
            }
        }

        return true;
    }
}
