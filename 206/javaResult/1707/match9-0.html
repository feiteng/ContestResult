class Solution {
    public boolean isTransformable(String s, String t) {
        if (s.length() != t.length()) return false;
        ArrayList<LinkedList<Integer>> p = new ArrayList<>(10);
        for(int i=0;i<10;i++) {
            p.add(new LinkedList<>());
        }
        for(int i=0;i<s.length();i++) {
            p.get((int)(s.charAt(i)-'0')).add(i);
        }
        for(int i=0;i<t.length();i++) {
            int num = (int)(t.charAt(i)-'0');
            Integer pos = p.get(num).pollFirst();
            if (pos == null) return false;
            for(int j=0;j<num;j++) {
                if (p.get(j).size()>0 && p.get(j).peekFirst()<pos) return false;
            }
        }
        return true;
    }
}
