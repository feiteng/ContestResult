class Solution {
    public boolean isTransformable(String s, String t) {
        if (!isValid(s, t)) {
            return false;
        }
        List<Integer>[] largerS = countLarger(s);
        List<Integer>[] largerT = countLarger(t);
        
        for (int i = 0; i < 10; i++) {
            List<Integer> sL = largerS[i];
            List<Integer> tL = largerT[i];
            for (int j = 0; j < sL.size(); j++) {
                if (sL.get(j) > tL.get(j)) {
                    return false;
                }
            }            
        }
        return true;
        
    }
    
    private List<Integer>[] countLarger(String s) {
        List<Integer>[] larger = new List[10];
        for (int i = 0; i < 10; i++) {
            larger[i] = new ArrayList<>();
        }
        
        int[] counts = new int[10];
        for (int i = s.length() - 1; i >= 0; i--) {
            int v = s.charAt(i) - '0';
            counts[v]++;
            int t = 0;
            for (int j = v + 1; j < 10; j++) {
                t += counts[j];
            }
            larger[v].add(t);
        }
        return larger;        
    }
    
    
    private boolean isValid(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] count = new int[10];
        for (char c: s.toCharArray()) {
            count[c - '0']++;
        }
        for (char c : t.toCharArray()) {
            count[c - '0']--;
            if ((count[c - '0']) < 0) {
                return false;
            }
        }
        return true;
    }
    
}
