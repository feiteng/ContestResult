class Solution {
    public boolean isTransformable(String s, String t) {
        int[][] dp1=new int[10][10];
        int[][] dp2=new int[10][10];
        int[] count1=new int[10];
        int[] count2=new int[10];
        Arrays.fill(count1,0);
        Arrays.fill(count2,0);
        for(int i=0;i<s.length();i++){
            count1[s.charAt(i)-48]++;
            count2[t.charAt(i)-48]++;
        }
        for(int i=0;i<10;i++){
            if(count1[i]!=count2[i]) return false;
        }
        Arrays.fill(count1,0);
        Arrays.fill(count2,0);
        for(int i=0;i<s.length();i++){
            count1[s.charAt(i)-48]++;
            if(s.charAt(i)=='0') continue;
            for(int j=0;j<s.charAt(i)-48;j++){
                dp1[j][s.charAt(i)-48]+=count1[j];
            }
        }
        for(int i=0;i<s.length();i++){
            count2[t.charAt(i)-48]++;
            if(t.charAt(i)=='0') continue;
            for(int j=0;j<t.charAt(i)-48;j++){
                dp2[j][t.charAt(i)-48]+=count2[j];
            }
        }
        for(int i=0;i<10;i++){
            for(int j=0;j<10;j++){
                if(i<j&&dp1[i][j]>dp2[i][j]) return false;
            }
        }
        return true;
    }
}
