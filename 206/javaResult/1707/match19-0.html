class Solution {
    public boolean isTransformable(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] nums = new int[10];
        int[] pos = new int[10];
        int[] dp = new int[10];
        Map<Integer, List<int[]>> map = new HashMap<>();
        for (int i = 0; i <= 9; i++) {
            map.put(i, new ArrayList<>());
        }
        for (int i = 0; i < s.length(); i++) {
            int cur = s.charAt(i) - '0';
            nums[s.charAt(i) - '0']++;
            map.get(cur).add(copy(nums));
        }
        
        int[] numss = new int[10];
        for (int i = 0; i < t.length(); i++) {
            int cur = t.charAt(i) - '0';
            if (dp[cur] >= map.get(cur).size()) {
                return false;
            }
            int[] num = map.get(cur).get(dp[cur]);
            if (!fuhe(numss, num, cur)) {
                return false;
            }
            dp[cur]++;
            numss[cur]++;
        }
        return true;
    }
    
    public int[] copy(int[] arr) {
        int[] arr1 = new int[arr.length];
        
        for (int i = 0;i < arr.length; i++) {
            arr1[i] = arr[i];
        }
        return arr1;
    }
    
    public boolean fuhe(int[] numss, int[] num, int val) {
        for (int i = 0; i < 10; i++) {
            if (i >= val) {
                return true;
            }
            if (numss[i] < num[i]) {
                return false;
            }
        }
        return true;
    }
}
