import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;

class Solution {
    public boolean isTransformable(String s, String t) {

        StringBuilder sb = new StringBuilder(s);
        ArrayList<Integer>[] buckets= new ArrayList[10];

        for (int i=0;i<buckets.length;i++)
            buckets[i]= new ArrayList<>();

        for (int i=0;i<s.length();i++)
        {
            int c=s.charAt(i)-'0';
            buckets[c].add(i);
        }


//        for (int j=0;j<buckets.length;j++)
//            System.out.println(buckets[j]);
//        System.out.println();



        for (int i=0;i<s.length();i++)
        {
            if (sb.charAt(i)!=t.charAt(i))
            {
                int first=find_first(buckets,t.charAt(i));
                if (first==-1)
                    return false;
                else
                {
                    sb.deleteCharAt(first);
                    sb.insert(i,t.charAt(i));
                }
            }
            else
            {
                int c1=sb.charAt(i)-'0';
                buckets[c1].remove(0);
            }

//            for (int j=0;j<buckets.length;j++)
//                System.out.println(buckets[j]);
//            System.out.println();

        }
        return true;
    }

    int find_first( ArrayList<Integer>[] buckets,char c)
    {
       int c1=c-'0';
       if (buckets[c1].size()==0)
           return -1;
       else
       {
           int idx=buckets[c1].get(0);
           buckets[c1].remove(0);

           for (int i=0;i<c1;i++)
           {
               if (buckets[i].size()!=0)
               {
                   if (buckets[i].get(0)<idx) {
                       //System.out.println("break");
                       return -1;
                   }
               }
           }
           return idx;
       }
    }
}
