class Solution {
    public boolean isTransformable(String sX, String tX) {
		// 2 1 4 3
		// 1 2 3 4
		// verify anagrams
		// find pos each maps to
		// l->r sweep
		
		char[] s=sX.toCharArray(), t=tX.toCharArray();
		char[] sSorted=sX.toCharArray(), tSorted=tX.toCharArray();
		Arrays.sort(sSorted);
		Arrays.sort(tSorted);
		if (sSorted.length!=tSorted.length) return false;
		for (int i=0; 
i<sSorted.length; i++) 
			if (sSorted[i]!=tSorted[i]) return false;
		
		int n=sSorted.length;
		ArrayDeque<Integer>[] positions=new ArrayDeque[10];
		for (int i=0; i<positions.length; i++) 
			positions[i]=new ArrayDeque<Integer>();
		for (int i=0; i<n; i++)
			positions[t[i]-'0'].addLast(i);
		int[] mapTo=new int[n];
		for (int i=0; i<n; i++)
			mapTo[i]=positions[s[i]-'0'].removeFirst();
		
		int[] lBound=new int[10];
		for (int i=0; i<n; i++) {
			int val=s[i]-'0';
			int to=mapTo[i];
			if (to<lBound[val]) return false;
			for (int j=val+1; j<10; j++)
				lBound[j]=Math.max(lBound[j], to);
		}
		return true;
    }
}
