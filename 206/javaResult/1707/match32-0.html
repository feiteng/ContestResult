class Solution {
    public boolean isTransformable(String s, String t) {

        LinkedList<Integer> a[] = new LinkedList[10];
        for(int i=0;i<10;i++){
            a[i] = new LinkedList<Integer>();
        }
        for(int i=0;i<s.length();i++) {
            a[(int)(s.charAt(i)-'0')].add(i);
        }

        for(int i=0;i<t.length();i++) {
            int num = (int)(t.charAt(i)-'0');
            if(a[num].size() == 0 ){
                return false;
            }
            int pos = a[num].pollFirst();
            for(int j = 0;j < num; j++) if(a[j].size()>0){
                if(a[j].peekFirst()<pos) {
                    return false;
                }
            }
        }
        return true;
    }
}
