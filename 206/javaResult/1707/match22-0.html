class Solution {
    public boolean isTransformable(String s, String t) {
        char[] ch=s.toCharArray();
        char[] ch1=t.toCharArray();
        Arrays.sort(ch);
        Arrays.sort(ch1);
        int n=s.length();
        for(int i=0;i<n;i++){
            if(ch[i]!=ch1[i]){
                return false;
            }
        }
        boolean flag=false;
        ArrayList<TreeSet<Integer>> ar=new ArrayList<>();
        for(int i=0;i<10;i++){
            ar.add(new TreeSet<>());
        }
        for(int i=0;i<n;i++){
            ar.get(s.charAt(i)-'0').add(i);   
        }
        ch=s.toCharArray();
        ch1=t.toCharArray();
        for(int i=0;i<n;i++){
                int num=ch1[i]-'0';
                for(int j=0;j<num;j++){
                    TreeSet<Integer> ts=ar.get(j);
                    if(ts.size()>0 && ts.first()<ar.get(num).first()){
                        flag=true;
                    }
                }
            //    System.out.println(flag+" "+num);
                ar.get(num).remove(ar.get(num).first());
            //    System.out.println(ar);
            
        }
        return !flag;
    }
}
