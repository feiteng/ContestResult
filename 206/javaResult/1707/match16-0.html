class Solution {
   public boolean isTransformable(String s, String t) {
      int[] cs = new int[10];
      int[] ct = new int[10];
      for (char c : s.toCharArray()) cs[c - '0']++;
      for (char c : t.toCharArray()) ct[c - '0']++;
      for (int i = 0; i <= 9; i++) if (cs[i] != ct[i]) return false;

      ArrayDeque<Integer>[] positions = new ArrayDeque[10];
      for (int i = 0; i <= 9; i++) positions[i] = new ArrayDeque<>();
      int n = s.length();
      for (int i = 0; i < n; i++) {
         positions[s.charAt(i) - '0'].addLast(i);
      }
      for (char c : t.toCharArray()) {
         int num = c - '0';
         for (int i = 0; i < num; i++) {
            if (!positions[i].isEmpty() && positions[i].peekFirst() < positions[num].peekFirst()) return false;
         }
         positions[num].removeFirst();
      }
      return true;
   }
}
