class Solution {
    int ten=10;
    int zz=0;
    int oo=1;
    char uu = '0';
    public boolean isTransformable(String s, String t) {
         int ll = s.length();
        
         List<int[]>[] xx = new ArrayList[ten];
        
         for (int i = zz; i < ten; ++i) xx[i] = new ArrayList<>();
        
         int[] cc = new int[ten];
        
         for (char c : s.toCharArray()) {
             int p = c - uu;
             cc[p] =cc[p]+oo;
             int[] fg = new int[p + oo];
             for (int i = zz; i <= p; ++i) fg[i] = cc[i];
             xx[p].add(fg);
         }

         cc = new int[ten];
         for (char c : t.toCharArray()) {
             int p = c - uu;
             cc[p] =cc[p]+oo;
             if (xx[p].size() < cc[p]) return false;
             
             int[] er = xx[p].get(cc[p] - oo);
             for (int i = zz; i <= p; ++i) {
                 if (er[i] > cc[i]) return false;
             }
         }

         return true;
     }
}
