class Solution {
    public boolean isTransformable(String s, String t) {
        int[] cnt = new int[10];
        List<int[]>[] li = new ArrayList[10];
        
        for (int i = 0; i < 10; ++i) {
            li[i] = new ArrayList<>();
        }
    
        for (char c : s.toCharArray()) {
            int p = c - '0';
            cnt[p] += 1;
            int[] temp = new int[p + 1];
            for (int i = 0; i <= p; ++i) {
                temp[i] = cnt[i];
            }
            li[p].add(temp);
        }
        
        cnt = new int[10];
        
        for (char c : t.toCharArray()) {
            int p = c - '0';
            cnt[p] += 1;
            if (li[p].size() < cnt[p]) {
                return false;
            }
            int[] temp1 = li[p].get(cnt[p] - 1);
            for (int i = 0; i <= p; ++i) {
                if (temp1[i] > cnt[i]) {
                    return false;
                }
            }
        }

        return true;
    }
    
}
