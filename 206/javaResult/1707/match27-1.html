class Solution {
    List<Integer>[] lists;
    public boolean isTransformable(String s, String t) {
        char[] sc = s.toCharArray();
        char[] tc = t.toCharArray();
        int n = sc.length;
        lists = 
new List[10];
        for (int i = 0; i < 10; i++) {
            lists[i] = new LinkedList<>();
        }
        for (int i = n-1; i >= 0; i--) {
            lists[sc[i] - '0'].add(i);
        }
        for (int i = n-1; i >= 0; i--) {
            int index = tc[i] - '0';
            if (!check(index)) {
                return false;
            }
        }
        return true;
    }

    private boolean check(int index) {
        if (lists[index].isEmpty()) return false;
        
int now = lists[index].get(0);
        for (int i = index+1; i < 10; i++) {
            if (!lists[i].isEmpty() && lists[i].get(0) > now) {
                return false;
            }
        }
        lists[index].remove(0);
        return true;
    }
}
