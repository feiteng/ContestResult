class Solution {
public:
    bool isTransformable(string s, string t) {
      vector<vector<int>> F(10);
      for(int i=0;i<t.size();++i){
        F[s[i]-'0'].push_back(i);
      }
      for(int i=t.size()-1;i>=0;--i){
        int n=t[i]-'0';
        if(F[n].size()==0)return false;
        for(int j=n+1;j<=9;++j)if(F[j].size()>0&&F[j].back()>F[n].back())return false;
        F[n].pop_back();
      }
      return true;
    }
};
