class Solution {
public:
    bool isTransformable(string s, string t) {

        deque<int> li[10];
        for(int i=0;i<s.size();i++){
            li[s[i]-'0'].push_back(i);
        }
        for(int i=0;i<t.size();i++){
            int j = t[i] - '0';
            if(li[j].size()==0)return false;

            for(int k=0;k<j;k++){
                if(li[k].size() && li[k][0] < li[j][0]) return false;
            }
            li[j].pop_front();
        }
        return true;
    }
};
