class Solution {
public:
    bool isTransformable(string s, string t) {
        if(s==t) 
            return true;
        int cnt1[10]={0};
        for(int i =0;i<s.length();i++){
            cnt1[s[i]-'0']++;
        }
        for(int i =0;i<t.length();i++){
            if(cnt1[t[i]-'0']==0)
                return false;
            cnt1[t[i]-'0']--;
        }
        vector<vector<int>> pos(11,vector<int>());
        int n = t.length();

        for(int i =0;i<n;i++){
            pos[s[i]-'0'].push_back(i);
        }
        for(int i = n-1;i>=0;i--){
            int p = t[i]-'0';
            if(pos[p].size()==0)
                return false;
            for(int j = p+1;j<=9;j++){
                if(pos[j].size()&&pos[j].back()>pos[p].back())
                    return false;
            }
            pos[p].pop_back();
        }
        return true;
    }
};
