class Solution {
public:
  bool isTransformable(string s, string t) {
    vector<vector<int>> sPos(10);
    for (int i = 0; i < s.size(); i++) {
      sPos[s[i] - '0'].push_back(i);
    }
    for (int i = t.size()-1; i >= 0; i--) {
      int a = t[i] - '0';
      if (sPos[a].size() <= 0) return false;
      for (int b = a+1; b < 10; b++) {
        if (sPos[b].size() > 0 && sPos[b].back() > sPos[a].back()) return false;
      }
      sPos[a].pop_back();
    }
    return true;
  }
};
