class Solution {
public:
    bool isTransformable(string s, string t) {
//         unordered_map<int, int> m;
//         for(auto &i : s) {
//             m[i]++;
//         }
//         for(auto &j : t) {
//             m[i]--;
//             if(m[i] < 0) {
//                 return 0;
//             }
//         }
//         for(auto &i : t) {
            
//         }
//         for() {
            
//         }
        int n = s.size();
        vector<int> v1(n), v2(n);
        for(int i = 9; i >= 0; i--) {
            vector<int> p1, p2;
            int c = 0;
            for(int j = 0; j < n; j++) {
                if(s[j] == '0' + i) {
                    p1.push_back(c++);
                    v1[j] = 1;
                } else if(!v1[j]) {
                    c++;
                }
            }
            c = 0;
            for(int j = 0; j < n; j++) {
                if(t[j] == '0' + i) {
                    p2.push_back(c++);
                    v2[j] = 1;
                } else if(!v2[j]) {
                    c++;
                }
            }
            if(p1.size() != p2.size()) {
                return 0;
            }
            for(int j = p1.size() - 1; j >= 0; j--) {
                if(p2[j] < p1[j]) {
                    return 0;
                }
            }
        }
        return 1;
    }
};
