class Solution {
public:
    bool isTransformable(string s, string t) {
        swap(s,t);
        int n=s.size();
        char prt='0'-1;
        static vector<int> occ[10];
        for(int x=0;x<10;x++) occ[x].clear();

        for(int i=n;--i>=0;) {
            occ[t[i]-'0'].push_back(i);
        }
        for(int i=0;i<n;i++) {
            int x=s[i]-'0';
            if(!occ[x].size()) return false;
            for(int y=0;y<x;y++) if(occ[y].size() && occ[y].back()<occ[x].back()) return false;
            occ[x].pop_back();
        }
        return true;
        
    }
};
