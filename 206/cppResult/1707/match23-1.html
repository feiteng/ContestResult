class Solution {
public:
    int n;
    bool isTransformable(string s, string t) {
        n = s.size();

        stack<int> st[10];
        for (int i = n-1; i >= 0; i--) {
            st[s[i]-'0'].push(i);
        }
        for (int i = 0; i < n; i++) {
            int x = t[i]-'0';
            if (st[x].empty())
                return false;
            int j = st[x].top();
            st[x].pop();

            for (int k = 0; k < x; k++) {
                if (!st[k].empty() && st[k].top() < j)
                    return false;
            }
        }
        return true;
    }
};
