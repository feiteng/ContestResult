class Solution {
public:
    bool isTransformable(string s, string t) {

        queue<int> q[10];
        for(int i = 0; i < s.size(); i++){
            q[s[i] - '0'].push(i);
        }
        for(int i = 0; i < t.size(); i++){
            int target = t[i] - '0';
            //cout<<target<<endl;
            if(q[target].empty())
                return false;

            for(int j = 0; j < target; j++){
                if(!q[j].empty() && q[j].front() < q[target].front())
                    return false;
            }
            q[target].pop();
        }
        return true;
    }
};
