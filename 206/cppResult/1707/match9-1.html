class Solution {
public:
    bool isTransformable(string s, string t) {
        deque<int> z[10];
        int n = s.size();

        for (int i=0; i<n; ++i) {
            z[s[i] -'0'].push_back(i);
        }
        cout<<endl;
        for (int i=0; i<n; ++i) {
            int x = t[i] -'0';
           // cout<<i<<' ';
            if (z[x].empty()) return false;
            //cout<<'!';
            int p = z[x][0];
            //cout<<p<<':';

            for (int j=0; j<x; ++j) {
                //cout<<z[j][0]<<' ';
                if (z[j].size() && z[j][0] <p) return false;
            }
            z[x].pop_front();
        }
        return true;
    }
};
