class Solution {
public:

    bool isTransformable(string s, string t) {
        int N=s.size();
        vector<queue<int>> vTable(10);

        for(int i=0;i<N;i++){
            vTable[s[i]-'0'].push(i);
        }
        for(int i=0;i<N;i++){
            int index=t[i]-'0';
            if(vTable[index].empty()){
                return false;
            }
            int begin=vTable[index].front();
            
            for(int j=0;j<index;j++){
                
               /* if(!vTable[j].empty()){
                    cout<<i<<" "<<begin<<" "<<vTable[j].front()<<endl;
                }*/
                if(!vTable[j].empty()&&vTable[j].front()<begin){
                    return false;
                }
            }
            vTable[index].pop();
        }
        return true;
    }
};
