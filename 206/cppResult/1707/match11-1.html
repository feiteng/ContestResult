class Solution {
public:
    bool isTransformable(string s, string t) {
        vector<set<int>> positions(10);

        for(int i=0;i<s.size();++i) {
            positions[s[i]-'0'].insert(i);
        }
        
        for(int i=0;i<t.size();++i) {
            int digit = t[i] - '0';
            if(positions[digit].empty()) {
                return false;
            }
            
            int pd = *positions[digit].begin();
            positions[digit].erase(positions[digit].begin());
            for(int dd=0;dd<digit;++dd) {
                if(!positions[dd].empty() && *positions[dd].begin() < pd) {
                    return false;
                }
            }
        }
        return true;
    }
};
