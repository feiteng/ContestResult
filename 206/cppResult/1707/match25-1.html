class Solution {
 public:
  bool isTransformable(string s, string t) {

    for (char c = '0'; c <= '9'; ++c) {
      vector<int> left, right;
      int cnt = 0;
      for (int i = 0; i < s.size(); ++i) {
        if (s[i] == c) {
          left.push_back(cnt);
        }

        if (s[i] >= c) {
          ++cnt;
        }
      }
      cnt = 0;
      for (int i = 0; i < t.size(); ++i) {
        if (t[i] == c) {
          right.push_back(cnt);
        }

        if (t[i] >= c) {
          ++cnt;
        }
      }
      if (left.size() != right.size()) {
        return false;
      }
      for (int i = 0; i < left.size(); ++i) {
        if (left[i] < right[i]) {
          return false;
        }
      }
    }
    return true;
  }
};
