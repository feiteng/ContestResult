class Solution {
public:
    bool isTransformable(string s, string t) {
        list<int> a[10]; 
        int n = s.length(); 
        for (int i = 0; i < n; i ++) {
            a[s[i] - '0'].push_back(i);
        }
        for (int i = 0; i < n; i ++) {
            int k = t[i] - '0'; 
            if (a[k].size() == 0) {
                return false; 
            }
            int pos = (*a[k].begin()); 
            a[k].pop_front(); 
            for (int num = 0; num < k; num ++) {
                if (!a[num].empty() && *(a[num].begin()) <= pos) {
                    return false;
                }
            }
        }
        return true;
    }
};
