class Solution {
public:
    bool isTransformable(string s, string t) {
        int n=s.size();
        vector<vector<int>> m(10);
        for(int i=0;i<n;++i)
        {
            m[s[i]-'0'].push_back(i);
        }
        for(int i=n-1;i>=0;--i)
        {
            int k=t[i]-'0';
            if(m[k].empty()) return false;
            int p=m[k].back();
            for(int j=k+1;j<=9;++j)
            {
                auto it=upper_bound(m[j].begin(),m[j].end(),p);
                if(it!=m[j].end()) return false;
            }
            m[k].pop_back();
        }
        return true;
    }
};
