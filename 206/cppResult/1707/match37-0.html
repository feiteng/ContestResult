class Solution {
public:
    bool Check(string &s, string &t, int endIndex)
    {
        if (endIndex == 0)
        {
            return true;
        }
        char c = t[endIndex - 1];
        int fIndex = -1;
        for (int i = endIndex - 1; i >= 0; i--)
        {
            if (s[i] > c)
            {
                return false;
            }
            else if (s[i] == c)
            {
                fIndex = i;
                break;
            }
        }
        if (fIndex >= 0)
        {
            s.erase(s.begin() + fIndex);
            return Check(s, t, endIndex - 1);
        }
        else
        {
            return false;
        }
    }
    bool isTransformable(string s, string t) {
        if (s.size() != t.size())
        {
            return false;
        }
        vector<vector<int>> pre;
        for (int i = 0; i < 10; i++)
        {
            vector<int> a;
            pre.push_back(a);
        }
        for (int i = 0; i < s.size(); i++)
        {
            pre[s[i] - '0'].push_back(i);
        }
        
        for (int i = t.size() - 1; i >= 0; i--)
        {
            int cur = t[i] - '0';
            if (pre[cur].size() == 0)
            {
                return false;
            }
            int candIndex = pre[cur][pre[cur].size() - 1];
            for (int j = cur + 1; j < 10; j++)
            {
                if (pre[j].size() > 0 && pre[j][pre[j].size() - 1] > candIndex)
                {
                    return false;
                }
            }
            pre[cur].pop_back();
        }
        return true;
    }
};
