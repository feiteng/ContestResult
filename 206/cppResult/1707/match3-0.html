class Solution {
public:
    bool isTransformable(string s, string t) {
        int n = s.size();
        vector<deque<int>> pos(10);
        for(int i = 0; i < n; i++) {
            pos[s[i]-'0'].push_back(i);
        }
        for(int i = 0; i < n; i++) {
            int v = t[i]-'0';
            if(pos[v].empty()) return false;
            int nxt = pos[v].front();
            pos[v].pop_front();
            for(int j = 0; j < v; j++) {
                if(!pos[j].empty() && nxt > pos[j].front()) return false;
            }
        }
        return true;
    }
};
