typedef long long ll;

typedef pair<ll, ll> pll;



class Solution {
public:
    bool isTransformable(string s, string t) {
        int n = s.size();
        queue<int> g[10];
        for(int i=0; i<n; i++){
            int ci = s[i] - '0';
            g[ci].push(i);

        }

        for(int i=0; i<n; i++){
            int ci = t[i] - '0';
            if(g[ci].empty()) return false;
            int j = g[ci].front();
            for(int cj=0; cj<ci; cj++){
                if(!g[cj].empty() && g[cj].front() < j) return false;
            }
            g[ci].pop();
        }
        return true;
    }
};
