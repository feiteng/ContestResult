class Solution {
public:
    bool isTransformable(string s, string t) {
        int L=t.size();
        map<int, set<int>> digs_to_pos;
        for(int i=0;i<L;i++) {
            digs_to_pos[s[i]-'0'].insert(i);
        }
        for(int i=0;i<L;i++) {
            int d = t[i]-'0';
            if (digs_to_pos[d].empty()) return false;
            int pos = *digs_to_pos[d].begin();
            // make sure all digits < d have first position > pos
            for(int dp=0;dp<d;dp++) {
                if (!digs_to_pos[dp].empty() && *digs_to_pos[dp].begin() < pos) {
                    return false;
                }
            }
            digs_to_pos[d].erase(pos);
        }
        
        return true;
    }
};
