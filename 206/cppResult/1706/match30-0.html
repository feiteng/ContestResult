class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<vector<int>> dist(n, vector<int>(n));
        for(int i = 0;i < n;i++)
            for(int j = 0;j < n;j++)
                dist[i][j] = abs(points[i][0]-points[j][0]) + abs(points[i][1]-points[j][1]);
        vector<bool> visited(n, false);
        vector<int> cost(n);
        cost[0] = 0;
        visited[0] = true;
        for(int i = 0;i < n;i++)
            cost[i] = dist[0][i];
        int ans = 0;
        for(int i = 1; i < n;i++)
        {
            int min_cost = INT_MAX;
            int pos;
            for(int j = 0; j < n;j++)
                if(!visited[j] && min_cost > cost[j])
                {
                    min_cost = cost[j];
                    pos = j;
                }
            
            visited[pos] = true;
            ans += min_cost;
            
            for(int j = 0; j < n;j++)
                if(!visited[j] && dist[pos][j] < cost[j])
                    cost[j] = dist[pos][j];
        }
        return ans;
    }
};
