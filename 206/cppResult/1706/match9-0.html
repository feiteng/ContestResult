struct node
{
    int u,v,w;
    bool operator<(const node &b) const
    {
        return w>b.w;
    }
};

int f[1003];

int find(int i)
{
    if (f[i] != i) f[i] = find(f[i]);
    return f[i];
}

class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        for (int i = 1; i <= n; i++) f[i] = i;
        
        priority_queue<node> a;
        
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                node tmp;
                tmp.u = i+1; tmp.v = j+1; tmp.w = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                a.push(tmp);
            }
        }
        
        int ct = 0;
        int ans = 0;
        while (ct < n-1) {
            while (true) {
                node ha = a.top();
                a.pop();
                int u = ha.u;
                int v = ha.v;
                u = find(u); v = find(v);
                if (u == v) continue;
                f[u] = v;
                ans += ha.w;
                break;
            }
            ++ct;
        }
        
        return ans;
        
        
    }
};
