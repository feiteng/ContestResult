class Solution {
public:
    int dis[1002],map[1002][1002],mark[1002];
    const int MAX=INT_MAX/2;
    int prim(int n)
    {
        for(int i=0;i<n;i++) 
        {
            dis[i]=map[1][i];
            mark[i]=0;
        }
        mark[1]=1; 
        int sum=0;
        for(int i=0;i<n-1;i++) 
        {
            int sta=-1,Min=MAX;
            for(int j=0;j<n;j++) 
            {
                if(!mark[j]&&dis[j]<Min)
                {
                    Min=dis[j];
                    sta=j;
                }
            }
            if(sta==-1) return -1; 
            mark[sta]=1; 
            sum+=Min;       
            for(int j=0;j<n;j++)
            {
                if(!mark[j]&&dis[j]>map[sta][j])
                dis[j]=map[sta][j];
            }
        }
        return sum;
    }
    int minCostConnectPoints(vector<vector<int>>& points) {
        for(int i=0;i<points.size()-1;i++){
            for(int j=i+1;j<points.size();j++){
                map[i][j]  = abs(points[i][0] - points[j][0]) + abs(points[i][1]-points[j][1]);
                map[j][i] = map[i][j];
            }
        }
        return prim(points.size());
    }
};
