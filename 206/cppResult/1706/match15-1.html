class Solution {
public:
    int fa[2000];
    int find(int x){
        return fa[x]==x?x:fa[x]=find(fa[x]);
    }
    struct node{
        int u,v,w;

        friend bool operator<(node a,node b){
            return a.w < b.w;
        }
    }e[1005*1005];
    int minCostConnectPoints(vector<vector<int>>& points) {
        int cnt = 0;
        for(int i = 0;i < points.size(); ++i){
            fa[i] = i;
            for(int j=i+1;j < points.size(); ++j){

                e[cnt++] = (node){i,j,abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1])};
            }
        }
        sort(e,e+cnt);
        int ans = 0;
        for(int i = 0;i < cnt; ++i){
            int x = e[i].u,y = e[i].v;
            int fx = find(x), fy = find(y);
            if(fx!=fy){
                ans += e[i].w;
                fa[fx] = fy;
            }
        }
        return ans;
    }
};
