class Solution {

    vector<int> fa;
    int getf(int x){
        return fa[x]==x?x:fa[x]=getf(fa[x]);
    }
public:
    int minCostConnectPoints(vector<vector<int>>& p) {
        int n=p.size();
        vector<pair<int,pair<int,int> > > v;

        for(int i=0;i<n;i++)
            for(int j=0;j<i;j++){
                int x=abs(p[i][0]-p[j][0])+abs(p[i][1]-p[j][1]);
                v.push_back(make_pair(x,make_pair(i,j)));
            }
        sort(v.begin(),v.end());
        fa.resize(n);
        for(int i=0;i<n;i++)
            fa[i]=i;
        int ans=0;
        for(auto i:v){
            int u=getf(i.second.first),v=getf(i.second.second);
            if (u!=v)
                fa[u]=v,ans+=i.first;
        }
        return ans;
    }
};
