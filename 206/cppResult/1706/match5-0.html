class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        //points = vector<vector<int>>(1000, {0, 0});for (int i = 0; i < 1000; ++i) points[i] = {i, i};
        int n = points.size();
        if (n == 1) return 0;
        vector<vector<int>> dis(n, vector<int>(n, 0));
        vector<bool> visited(n, false);
        vector<int> d(n, INT_MAX);
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                dis[j][i] = dis[i][j] = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1]);
            }
            d[i] = dis[i][0];
        }
        
        int ret = 0;
        visited[0] = true;
        for (int i = 0; i < n - 1; ++i) {
            int v = -1, minD = INT_MAX;
            for (int j = 0; j < n; ++j) {
                if (!visited[j] && minD > d[j]) {
                    v = j; minD = d[j];
                }
            }
            visited[v] = true;
            ret += minD;
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    d[j] = min(d[j], dis[j][v]);
                }
            }
        }
        
        return ret;
    }
};
