class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n=points.size();
        if(n==1) return 0;
        int cost[n][n];
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i!=j){
                    cost[i][j]=abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                }
                else cost[i][j]=0;
            }
        }
        int s[n],lowcost[n];
        memset(s,0,sizeof(s));
        s[0]=1;
        for(int i=1;i<n;i++){
            lowcost[i]=cost[0][i];
        }
        int res=0;
        for(int i=1;i<n;i++){
            int minv=INT_MAX;
            int j;
            for(int k=0;k<n;k++){
                if(lowcost[k]<minv&&!s[k]){
                    minv=lowcost[k];
                    j=k;
                }
            }
            s[j]=1;
            res+=lowcost[j];
            //cout<<j<<" "<<lowcost[j]<<endl;
            for(int k=0;k<n;k++){
                if(cost[j][k]<lowcost[k]&&!s[k]){
                    lowcost[k]=cost[j][k];
                }
            }
        }
        return res;
    }
};
