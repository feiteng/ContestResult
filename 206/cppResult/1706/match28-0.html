class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& p) {
      int n = p.size();
      int ans = 0;
      vector<int> flag(n, 0), d(n, 1e9);
      d[0] = 0;

      for (int i = 0; i < n; i++) {
        int minn = 1e9, u = -1;
        for (int j = 0; j < n; j++) {
          if (flag[j] == 0 && d[j] < minn) {
            minn = d[j];
            u = j;
          }
        }
        ans += minn;
        flag[u] = 1;
        for (int j = 0; j < n; j++) {
          int w = abs(p[u][0] - p[j][0]) + abs(p[u][1] - p[j][1]);
          if (flag[j] == 0 && w < d[j]) {
            d[j] = w;
          }
        }
      }
      return ans;
    }
};
