#include <bits/stdc++.h>
using namespace std;


int pri[2000];
bool done[2000];

class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        fill(pri, pri+n, 1e9);
        fill(done, done+n, 0);

        pri[0] = 0;
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int at = -1, mn = 1e9;
            for (int j = 0; j < n; ++j) if (!done[j]) {

                if (pri[j] < mn) {
                    mn = pri[j];
                    at = j;
                }
            }
            // cout << at << " " << pri[at] << endl;
            ans += pri[at];
            done[at] = 1;
            for (int j = 0; j < n; ++j) if (!done[j]) {
                int cst = abs(points[at][0]-points[j][0])+abs(points[at][1]-points[j][1]);
                pri[j] = min(pri[j], cst);
            }
        }
        return ans;
    }
};
