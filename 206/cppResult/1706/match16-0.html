class Solution {
public:
    vector<int> f;
    int find(int x) {
        return f[x] == x ? x : f[x] = find(f[x]);
    }
    struct Edge{
        int x, y, z;
        bool operator < (Edge b){
            return z < b.z;
        }
    };
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        f.resize(n);
        for(int i=0;i<n;i++) f[i] = i;
        vector<Edge> e;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int z = abs(points[i][0] - points[j][0]) +abs(points[i][1] - points[j][1]);
                e.push_back({i, j, z});
            }
        }
        sort(e.begin(),e.end());
        long long res = 0;
        for(auto& t : e){
            int x = t.x, y = t.y, z = t.z;
            if(find(x) == find(y)) continue;
            res += z;
            f[find(x)] = find(y);
        }
        return res;
    }
};
