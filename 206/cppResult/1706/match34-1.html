class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        vector<int> mincost(points.size(), 1000000000);
        vector<bool> flag(points.size(), false);
        mincost[0] = 0;
        int ret = 0;
        
        while (true) {

            int p = -1;
            for (int i = 0;i < points.size();i++) {
                if (flag[i]) {
                    continue;
                }
                if (p == -1 || mincost[i] < mincost[p]) {
                    p = i;
                }
            }
            if (p == -1) {
                return ret;

            }
            flag[p] = true;
            ret += mincost[p];
            for (int i = 0;i < points.size();i++) {
                if (flag[i]) {
                    continue;
                }
                mincost[i] = min(mincost[i], abs(points[i][0] - points[p][0]) + abs(points[i][1] - points[p][1]));
            }
        }
        return -1;
    }
};
