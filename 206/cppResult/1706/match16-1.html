class Solution {
public:
	
	int fa[1005];
	
	int get_fa( int x ) { return x==fa[x] ?x : fa[x]=get_fa(fa[x]); }
	
	struct node{
		int u,v,w;
		bool operator < ( const node &rhs ) const{ return w<rhs.w; }

	}p[1000005];
	
    int minCostConnectPoints(vector<vector<int>>& points) {
		int n=points.size();
		for( int i=1;i<=n;i++ ) fa[i]=i;
		int cnt=0;

		for( int i=0;i<n;i++ )
		{
			for( int j=i;j<n;j++ )
			{
				int u=i+1,v=j+1;
				int w=abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
				p[++cnt]={u,v,w};
			}
		}
		sort(p+1,p+1+cnt);
		int ans=0;
		for( int i=1;i<=cnt;i++ )
		{
			int fu=get_fa(p[i].u),fv=get_fa(p[i].v);
			if( fu!=fv ) 
			{
				ans+=p[i].w;
				fa[fv]=fu;
			} 
		}
		return ans;
    }
};
