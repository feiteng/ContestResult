class Solution {
public:
    struct edge{
        int u,v,len;
        bool operator <(const edge&a)const{
            return len < a.len;
        }
    }e[1000010];
    int fa[100010];
    int find(int x)
    {
        return  x==fa[x]?x:fa[x] = find(fa[x]);
    }
    int minCostConnectPoints(vector<vector<int>>& points) {
        int cnt = 0;
        for(int i=0;i<points.size();++i)
            for(int j=i+1;j<points.size();++j)
            {
                e[cnt].u = i;
                e[cnt].v = j;
                e[cnt].len = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                cnt++;
            }
        sort(e,e+cnt);
        for(int i=0;i<points.size();++i)
            fa[i] = i;
        int ans = 0;
        for(int i=0;i<cnt;++i)
        {
            int fu = find(e[i].u),fv = find(e[i].v);
            if(fu!=fv)
            {
                ans +=e[i].len;
                fa[fu] = fv;
            }
        }
        return ans;
    }
};
