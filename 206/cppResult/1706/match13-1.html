const int INF = 1e9;

class Solution {

public:
    int minCostConnectPoints(vector<vector<int>>& a) {
        if (a.size() == 1)
            return 0;
        int n = a.size();
        vector<vector<int>> w(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                w[i][j] = abs(a[i][0] - a[j][0]) + abs(a[i][1] - a[j][1]);
                w[j][i] = abs(a[i][0] - a[j][0]) + abs(a[i][1] - a[j][1]);
            }
        }
        vector<int> dist(n + 1, INF);
        vector<bool> st(n, false);

        int res = 0;
        for (int i = 0; i < n; i++)
        {
            int t = -1;
            for (int j = 0; j < n; j ++ )
                if (!st[j] && (t == -1 || dist[t] > dist[j]))
                    t = j;

            if (i && dist[t] == INF) return INF;

            if (i) res += dist[t];
            st[t] = true;

            for (int j = 0; j < n; j ++ ) dist[j] = min(dist[j], w[t][j]);
        }
        return res;
    }
};
