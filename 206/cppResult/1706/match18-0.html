int edges[1005][1005];
int dist[1005];
bool vis[1005];
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        memset(dist, 0x3f, sizeof(dist));
        memset(vis, false, sizeof(vis));
        int n = points.size();
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
            {
                edges[i][j] = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1]);
            }
        vis[0] = true;
        for (int i = 1; i < n; i++) dist[i] = edges[0][i];
        int ret = 0;
        for (int i = 1; i < n; i++)
        {
            int x = 0;
            for (int j = 0; j < n; j++)
                if (!vis[j] && dist[j] < dist[x]) x = j;
            ret += dist[x];
            vis[x] =true;
            for (int j = 0; j < n; j++)
                if (!vis[j]) dist[j] = min(dist[j], edges[x][j]);
        }
        return ret;
    }
};
