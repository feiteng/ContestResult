class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n=points.size();
        vector<vector<int>> d(n,vector<int>(n));
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int f=abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                d[i][j]=f;
                d[j][i]=f;
            }
        }
        vector<bool> s(n);
        vector<int> smallest(n);
        s[0]=true;
        for(int i=0;i<n;i++){
            smallest[i]=d[0][i];
        }
        int num=1,sum=0;
        while(num<n){
            int id=0,ss=INT_MAX;
            for(int i=0;i<n;i++){
                if(!s[i]&&smallest[i]<ss){
                    ss=smallest[i];id=i;
                }
            }
            sum+=ss;s[id]=true;num++;
            for(int i=0;i<n;i++){
                if(!s[i]&&smallest[i]>d[id][i]){
                    smallest[i]=d[id][i];
                }
            }
        }
        return sum;
    }
};
