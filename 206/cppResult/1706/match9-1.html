struct Node
{
    int x, y, d;
    bool operator < (const Node&T)const{
        return d < T.d;
    }
}rem[1000005];
int C[1005];

int findset(int x)
{
    if(C[x] == x)  return x;
    return C[x] = findset(C[x]);

}
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        for(int i = 0; i < n; i++)  C[i] = i;
        int p = 0;

        for(int i = 0; i < n; i++)  {
            for(int j = i+1; j < n; j++)  {
                rem[p++] = Node{i, j, abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])};
            }
        }
        sort(rem, rem + p);
        int ans = 0;
        for(int i = 0; i < p; i++)  {
            int &u = rem[i].x, &v = rem[i].y, &d = rem[i].d;
            int fu = findset(u), fv = findset(v);
            if(fu == fv)  continue;
            ans += d;
            C[fu] = fv;
        }
        return ans;
    }
};
