class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<int> d; d.resize(n);
        vector<bool> ok; ok.resize(n);

        for (int i = 1; i < n; i ++)
            d[i] = abs(points[i][0]-points[0][0]) + abs(points[i][1]-points[0][1]);
        int ret = 0;
        for (int i = 1; i < n; i ++)
        {
            int mi = -1;
            for (int j = 1; j < n; j ++)
                if (!ok[j] && (mi == -1 || d[j] < d[mi]))
                    mi = j;

            ok[mi] = true;
            ret += d[mi];
            //printf("%d %d %d\n", i, mi, d[mi]);
            for (int j = 1; j < n; j ++)
                if (!ok[j] && d[j] > abs(points[j][0]-points[mi][0]) + abs(points[j][1]-points[mi][1]))
                    d[j] = abs(points[j][0]-points[mi][0]) + abs(points[j][1]-points[mi][1]);
        }
        return ret;
    }
};
