class DSU{
    vector<int> parent;
    public:
    DSU(int n) {
        parent.resize(n);
        for (int i = 0; i < n; i++)
            parent[i] = i;
    }
    int getParent(int node) {
        if (parent[node] != node)
            parent[node] = getParent(parent[node]);
        return parent[node];
    }

    void join(int p1, int p2) {
        parent[p1] = p2;
    }
};

class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        vector<pair<int, pair<int, int>>> edges;
        for (int i = 0; i < points.size(); i++) {
            for (int j = i+1; j < points.size(); j++) {
                edges.push_back({abs(points[i][0]-points[j][0]) + abs(points[i][1]-points[j][1]), {i, j}});
            }
        }
        sort(edges.begin(), edges.end());
        DSU dsu(points.size());
        int ans = 0;
        for (auto& e : edges) {

            int p1 = dsu.getParent(e.second.first);
            int p2 = dsu.getParent(e.second.second);
            if (p1 != p2) {
                dsu.join(p1, p2);
                ans += e.first;
            }
        }
        return ans;
    }
};
