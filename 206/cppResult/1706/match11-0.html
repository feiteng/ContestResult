class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n=points.size();
        vector<vector<int>> dis(n,vector<int>(n,0));
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i==j) continue;
                dis[i][j]=abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
            }
        }
        vector<int> complete(n);
        vector<int> mindis(n,INT_MAX);
        mindis[0]=0;
        int res=0;
        for(int i=0;i<n;i++){
            int mi=INT_MAX;
            int p=0;
            for(int j=0;j<n;j++){
                if(complete[j]) continue;
                if(mindis[j]<mi){
                    mi=mindis[j];
                    p=j;
                }
            }
            //cout<<p<<" "<<mindis[p]<<endl;
            res+=mindis[p];
            complete[p]=1;
            for(int j=0;j<n;j++){
                if(complete[j] || p==j) continue;
                if(dis[p][j]<mindis[j]){
                    mindis[j]=dis[p][j];
                }
            }
        }
        return res;
    }
};
