class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n=points.size();
        vector<vector<int>> graph(n, vector<int>(n,0));
        int d=0;
        for(int i=0; i<n; ++i) {
            for(int j=i+1; j<n; ++j) {
                d = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                graph[i][j] = d;
                graph[j][i] = d;
            }
        }
        int sum = 0;
        vector<bool> visit(n, false);
        vector<int> dis(n, INT_MAX);
        int start=0;
        dis[start] = 0;
        for(int i=0; i<n; ++i) {
            int u=-1;
            int minn = INT_MAX;
            for(int j=0; j<n; ++j) {
                if(!visit[j] &&dis[j]<minn) {
                    minn = dis[j];
                    u=j;
                }
            }
            visit[u] = true;
            sum+=dis[u];
            for(int v=0; v<n; ++v) {
                if(!visit[v] && graph[u][v]<dis[v]) dis[v] = graph[u][v];
            }
        }
        return sum;   
    }
  
};
