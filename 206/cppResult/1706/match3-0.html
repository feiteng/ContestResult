class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<vector<int> > dist(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                dist[i][j] = dist[j][i] = abs(points[i][0]-points[j][0]) + abs(points[i][1] - points[j][1]);
            }
        }
        vector<bool> connected(n, false);
        int cost = 0;
        connected[0] = true;
        vector<int> mindist(n, 0);
        for (int i = 0; i < n; i++) mindist[i] = dist[i][0];
        for (int i = 1; i < n; i++) {
            int pos = -1, d = INT_MAX;
            for (int j = 0; j < n; j++) {
                if (!connected[j] && mindist[j] < d) {
                    pos = j;
                    d = mindist[j];
                }
            }
            //cout << pos;
            connected[pos] = true;
            cost += d;
            for (int j = 0; j < n; j++) {
                if (!connected[j] && mindist[j] > dist[pos][j]) {
                    mindist[j] = dist[pos][j];
                }
            }
        }
        //cout << endl;
        return cost;
    }
};
