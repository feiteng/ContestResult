typedef long long ll;

const int inf = ~0U >> 2;

int Prim(int n, vector<vector<int>> &c)
{
    vector<int> lowcost(n);
    vector<bool> s(n);

    s[0]=true;
    for(int i=0;i<n;i++)
    {
        lowcost[i]=c[0][i];
    }
    int ans = 0;
    for(int i=1;i<n;i++)
    {
        int min=inf;
        int j=1;

        for(int k=0;k<n;k++)
        {
            if((lowcost[k]<min)&&(!s[k]))
            {
                min=lowcost[k];j=k;
            }
        }
        s[j]=true;
        ans += min;
        for(int k=0;k<n;k++)
        {
            if((c[j][k]<lowcost[k])&&(!s[k])){lowcost[k]=c[j][k];}
        }
    }
    return ans;
}
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<vector<int>> graph(n, vector<int>(n));
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j) graph[i][j] = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1]);
        
        return Prim(n, graph);
    }
};
