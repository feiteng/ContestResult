struct edge {
    int x, y;
    int val;
    bool operator <(const edge &_) const{
        return val < _.val;
    }
};

class Solution {
public:
    int *fa;
    int find_father(int x) {
        if(fa[x] == x)
            return x;
        return fa[x] = find_father(fa[x]);
    }
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        fa = new int[n + 2];
        for(int i = 0; i < n; i++)
            fa[i] = i;
        vector<edge>v;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                v.push_back({i, j, abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])});
            }
        }
        sort(v.begin(), v.end());
        int ans = 0;
        for(int i = 0; i < v.size(); i++) {
            int x = find_father(v[i].x);
            int y = find_father(v[i].y);
            if(x == y)
                continue;
            fa[x] = y;
            int val = v[i].val;
            ans += val;
        }
        return ans;
        
    }
};
