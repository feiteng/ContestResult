struct P {

    int u, v, w;
    bool operator < (const P& t)const {
        return w < t.w;
    }
};
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();

        vector<int> f(n);
        for(int i = 0; i < n; i++) f[i] = i;
        vector<P> vec;
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                vec.push_back({i, j, abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])});
            }
        }
        sort(vec.begin(), vec.end());
        int ans = 0;
        for(int i = 0; i < vec.size(); i++) {
            int u = vec[i].u, v = vec[i].v, w = vec[i].w;
            int fx = Find(u, f), fy = Find(v, f);
            if(fx != fy) ans += w, f[fx] = fy;
        }
        return ans;
    }
    int Find(int x, vector<int>& f) {
        if(x == f[x]) return x;
        return f[x] = Find(f[x], f);
    }
};
