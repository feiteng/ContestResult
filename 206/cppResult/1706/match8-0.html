struct Edge {
    int from, to, dist;
    bool operator < (Edge t) {
        return dist < t.dist;
    }
};
Edge ed[1000010];
int fa[1000010];
int find(int x) {
    if(fa[x] == x) return x;
    return fa[x] = find(fa[x]);
}
void join(int c1, int c2) {
    int f1 = find(c1), f2 = find(c2);
    if(f1 != f2) {
        fa[f1] = f2;
    }
}
int n;
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        n = points.size();
        for(int i = 1;i <= n * n;i++) fa[i] = i;
        int idx = 0;
        for(int i = 1;i <= n;i++) {
            for(int j = 1;j <= n;j++) {
                if(i == j) continue;
                ed[++idx] = {i, j, (int)(abs(points[i - 1][0] - points[j - 1][0]) + abs(points[i - 1][1] - points[j - 1][1]))};
            }
        }
        sort(ed + 1, ed + 1 + idx);
        int ans = 0;
        for(int i = 1;i <= idx;i++) {
            int u = ed[i].from, v = ed[i].to;
            if(find(u) != find(v)) {
                join(u, v);
                ans += ed[i].dist;
            }
        }
        return ans;
    }
};
