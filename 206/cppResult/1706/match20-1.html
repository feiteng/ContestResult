class Solution {
    struct node
    {
        int x,y,v;
        node () {};node (int _x,int _y,int _v) {x=_x;y=_y;v=_v;}

        bool operator < (const node &t) const {return v<t.v;}
    };
    vector<node> E;
    int Fa[1010];
    int getFa(int x) {return x==Fa[x]?x:Fa[x]=getFa(Fa[x]);}
public:

    int minCostConnectPoints(vector<vector<int>>& p) {
        int n=p.size();
        for (int i=0;i<n;++i)
            for (int j=i+1;j<n;++j)
            {

                int v=abs(p[i][0]-p[j][0])+abs(p[i][1]-p[j][1]);
                E.push_back(node(i,j,v));
            }
        sort(E.begin(),E.end());
        for (int i=0;i<n;++i) Fa[i]=i;
        int ans=0;
        for (int i=0;i<E.size();++i)
            {
                int x=E[i].x;
                int y=E[i].y;
                x=getFa(x);
                y=getFa(y);
                if (x==y) continue;
                Fa[x]=y;
                ans+=E[i].v;
            }
        return ans;
    }
};
