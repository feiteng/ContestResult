class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<int> dist(n, 0x3f3f3f3f), visited(n, 0);

        int rnt = 0;
        dist[0] = 0;
        for(int k = 0; k < n; k ++){
            int minId = -1;
            for(int i = 0; i < n; i ++){
                if(!visited[i] && (minId == -1 || dist[i] < dist[minId]))
                    minId = i;
            }
            rnt += dist[minId];
            visited[minId] = true;
            for(int i = 0; i < n; i ++){
                if(!visited[i])
                    dist[i] = min(dist[i], abs(points[i][0] - points[minId][0]) + abs(points[i][1] - points[minId][1]));
            }
        }
        return rnt;
    }
};
