class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        return prim(points, points.size());
    }
    int len(const vector<int>& a, const vector<int>& b)
    {
        return abs(a[0] - b[0]) + abs(a[1] - b[1]);
    }
    int prim(const vector<vector<int>>& points, int n)
    {
        int ans = 0;
        vector<bool> vis(n, false);
        vector<int> dis(n, 0x3f3f3f3f);
        dis[0] = 0;
        for(int i=0;i<n;++i)
        {
            int u = -1;
            for(int v=0;v<n;++v)
                if(!vis[v])
                    if(u==-1 || dis[v] < dis[u])
                        u = v;
            if(u==-1) return ans;
            vis[u] = true;
            ans += dis[u];
            for(int v=0;v<n;++v)
                if(!vis[v] && dis[v] > len(points[u], points[v]))
                    dis[v] = len(points[u], points[v]);
        }
        return ans;
    }
};
