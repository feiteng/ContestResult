class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        vector<vector<int>> dis(n, vector<int>(n, 0));
        for (int i=0; i<n; i++)
            for (int j=0; j<n; j++) {
                int d = abs(points[i][0]-points[j][0])+abs(points[i][1]-points[j][1]);
                dis[i][j] = d;
                dis[j][i] = d;
            }
        vector<int> f(n, 0);
        vector<int> is_used(n, 0);

        for (int i=0; i<n; i++) f[i] = dis[0][i];
        int res = 0;
        for (int i=0; i<n; i++) {
            int inds = 0;
            int t = 888888888;
            for (int j=0; j<n; j++) {
                if (is_used[j]==1) continue;
                if (f[j]<t) {
                    t = f[j];
                    inds = j;
                }
            }
            is_used[inds] = 1;
            res += t;
            for (int j=0; j<n; j++) {
                if (is_used[j]==1) continue;
                if (dis[inds][j]<f[j]) f[j] = dis[inds][j];
            }
        }
        return res;

    }
};
