class Solution {
public:
    int numSpecial(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        int ans = 0;
        for(int i = 0 ; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(A[i][j] == 0)
                    continue;
                bool sp = true;
                for(int i1 = 0; i1 < m; ++i1) {
                    if(i1 == i) 
                        continue;
                    if(A[i1][j] == 1) {
                        sp = false;
                        break;
                    }
                }
                for(int i1 = 0; i1 < n; ++i1) {
                    if(i1 == j) 
                        continue;
                    if(A[i][i1] == 1) {
                        sp = false;
                        break;
                    }
                }
                ans += sp;
            }
        }
        return ans;
    }
};
