class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        vector<int> hang(mat.size(), 0), lie(mat[0].size(), 0);
        for (int i = 0; i < mat.size(); i++) {
            for (int j = 0; j < mat[0].size(); j++) {
                hang[i] += mat[i][j];
                lie[j] += mat[i][j];
            }
        }
        int ret = 0;
        for (int i = 0; i < mat.size(); i++) {
            for (int j = 0; j < mat[0].size(); j++) {
                if (mat[i][j] == 1 && hang[i] == 1 && lie[j] == 1) ret++;
            }
        }
        return ret;
    }
};
