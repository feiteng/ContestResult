class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int rows = mat.size();
        int cols = mat[0].size();
        vector<int> row_sum(rows, 0);
        vector<int> col_sum(cols, 0);
        for(int i=0; i<rows; i++) {
            for(int j=0; j<cols; j++) {
                row_sum[i] += mat[i][j];
                col_sum[j] += mat[i][j];
            }
        }
        int res = 0;
        for(int i=0; i<rows; i++) {
            for(int j=0; j<cols; j++) {
                if(mat[i][j]==1 && row_sum[i]==1 && col_sum[j] == 1) res++;
            }
        }
        return res;
    }
};
