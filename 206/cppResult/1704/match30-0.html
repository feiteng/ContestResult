class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int R = mat.size(), C = mat[0].size();
        vector<int> rsum(R), csum(C);
        for(int i=0;i<R;i++) {
            for(int j=0;j<C;j++) {
                int e = mat[i][j];
                rsum[i] += e;
                csum[j] += e;
            }
        }
        int ans = 0;
        for(int i=0;i<R;i++) {
            for(int j=0;j<C;j++) {
                if(rsum[i] == 1 && csum[j] == 1 && mat[i][j]) ans++;
            }
        }
        return ans;
    }
};
