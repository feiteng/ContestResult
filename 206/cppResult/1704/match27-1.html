class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int N=mat.size();
        int M=mat[0].size();
        vector<int> vRow(N);
        vector<int> vCol(M);
        for(int i=0;i<N;i++){
            for(int j=0;j<M;j++){
                if(mat[i][j]==1){
                    vRow[i]++;
                    vCol[j]++;
                }
                
            }
        }
        int ans=0;
        for(int i=0;i<N;i++){
            for(int j=0;j<M;j++){
                if(vRow[i]==1&&vCol[j]==1&&mat[i][j]==1){
                    //cout<<i<<" "<<j<<endl;
                    ans++;
                }
                
            }
        }
        return ans;
    }
};
