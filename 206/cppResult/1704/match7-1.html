class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        vector<int> sumRow(mat.size()), sumCol(mat[0].size());
        for (int i = 0; i < mat.size(); ++ i) {
            for (int j= 0; j < mat[0].size(); ++ j) {
                sumRow[i] += mat[i][j];
                sumCol[j] += mat[i][j];
            }
        }
        int ans = 0;
        for (int i = 0; i < mat.size(); ++ i) {
            for (int j= 0; j < mat[0].size(); ++ j) {
                if (mat[i][j] == 1 && sumRow[i] == 1 && sumCol[j] == 1) ++ ans;
            }
        }
        return ans;
    }
};
