class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        vector<int> rsum(mat.size(), 0), csum(mat[0].size(), 0);
        for (int i = 0; i < mat.size(); ++i) {
            for (int j = 0; j < mat[0].size(); ++j) {
                rsum[i] += mat[i][j];
                csum[j] += mat[i][j];
            }
        }
        
        int ret = 0;
        for (int i = 0; i < mat.size(); ++i) {
            for (int j = 0; j < mat[0].size(); ++j) {
                if (mat[i][j] == 1 && rsum[i] == 1 && csum[j] == 1) ret++;
            }
        }
        return ret;
    }
};
