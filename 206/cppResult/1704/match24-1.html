class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int n = mat.size(), m = mat[0].size();
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (!mat[i][j]) continue;
                bool ok = true;
                for (int ii = 0; ii < n; ii++) {
                    if (i == ii) continue;
                    if (mat[ii][j]) {
                        ok = false;
                        break;
                    }
                }
                for (int jj = 0; jj < m; jj++) {
                    if (j == jj) continue;
                    if (mat[i][jj]) {
                        ok = false;
                        break;
                    }
                }
                res += ok;
            }
        }
        return res;
    }
};
