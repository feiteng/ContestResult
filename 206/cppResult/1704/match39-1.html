class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int n = mat.size(), m = mat.front().size();
        vector<int> nn(n, 0), mn(m, 0);
        for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j)

            if (mat[i][j] == 1) ++nn[i], ++mn[j];
        int ans = 0;
        for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j) 
            if (mat[i][j] == 1 && nn[i] == 1 && mn[j] == 1) ++ans;
        return ans;
    }
};
