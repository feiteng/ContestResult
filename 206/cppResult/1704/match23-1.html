class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int M = mat.size(), N = mat[0].size();
        vector<int> rows(M);
        vector<int> cols(N);
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                rows[i] += mat[i][j];
                cols[j] += mat[i][j];
            }
        }
        int ans = 0;
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                if (mat[i][j] && rows[i] == 1 && cols[j]==1)
                    ans ++;
            }
        }
        return ans;
    }
};
