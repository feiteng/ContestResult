class Solution {
public:
    int numSpecial(vector<vector<int>>& a) {
      int n = a.size(), m = a[0].size();
      vector<int> r(n, 0), c(m, 0);
      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          r[i] += a[i][j];
          c[j] += a[i][j];
        }
      }
      int ans = 0;
      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          if (r[i] == 1 && c[j] == 1 && a[i][j] == 1) {
            ans += 1;
          }
        }
      }
      return ans;
    }
};
