class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int row = mat.size(), col = mat[0].size();
        vector<int> rFre(row, 0), cFre(col, 0);
        
        for(int r = 0; r < row; r++) for(int c = 0; c < col; c++) {
            if(mat[r][c] == 1)

                rFre[r]++, cFre[c]++;
        }
        
        int ans = 0;
        for(int r = 0; r < row; r++) for(int c = 0; c < col; c++) {
            if(mat[r][c] == 1 && rFre[r] == 1 && cFre[c] == 1)
                ans++;
        }
        return ans;
    }
};
