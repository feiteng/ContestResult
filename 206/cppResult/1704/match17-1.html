class Solution {
public:
    int numSpecial(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        vector<int> mCnt(m, 0);
        vector<int> nCnt(n, 0);
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 1) {
                    mCnt[i]++;
                    nCnt[j]++;
                }
            }
        }
        
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 1) {
                    if (mCnt[i] == 1 && nCnt[j] == 1) {
                        res++;
                    } 
                }
            }
        }
        return res;
    }
};
