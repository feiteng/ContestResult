class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<int> partner(n);
        for(auto &p: pairs) {
            partner[p[0]] = p[1];
            partner[p[1]] = p[0];
        }
        
        vector<vector<int>> preferences_map(n, vector<int>(n, -1));

        for(int i=0;i<n;++i) {
            for(int j=0;j<n-1;++j) {
                preferences_map[i][preferences[i][j]] = j;
            }
        }
        
        int ans = 0;
        for(int x=0;x<n;++x) {
            int y = partner[x];
            for(int i=0;i<n-1;++i) {
                if(preferences[x][i] == y) {
                    break;
                }
                int u = preferences[x][i];
                int v = partner[u];

                int pref_x = preferences_map[u][x];
                int pref_v = preferences_map[u][v];
                if(pref_x < pref_v) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
