class Solution {
 public:
  int unhappyFriends(int n, vector<vector<int>>& preferences,
                     vector<vector<int>>& pairs) {
    vector<vector<int>> like(n, vector<int>(n));
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < n - 1; ++j) {
        like[i][preferences[i][j]] = j;
      }
    }
    vector<int> p(n);
    for (const auto& uv : pairs) {
      int u = uv[0], v = uv[1];

      p[u] = v;
      p[v] = u;
    }
    int ans = 0;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < n; ++j) {
        if (i != j && p[i] != j && like[i][j] < like[i][p[i]] &&
            like[j][i] < like[j][p[j]]) {
          ++ans;
          break;
        }
      }
    }
    return ans;
  }
};
