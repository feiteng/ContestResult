class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> order(n, vector<int>(n));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                order[i][preferences[i][j]] = j;
            }
        }
        function<bool(int,int,int,int)> check = [&](int x, int y, int u, int v) {
            return order[x][u] < order[x][y] && order[u][x] < order[u][v];
        };
        vector<bool> mk(n, false);
        for (int i = 0; i < pairs.size(); ++i) {
            int x = pairs[i][0], y = pairs[i][1];
            for (int j = i + 1; j < pairs.size(); ++j) {
                int u = pairs[j][0], v = pairs[j][1];
                if (check(x, y, u, v) || check(x, y, v, u)) {
                    mk[x] = true;
                }
                if (check(y, x, u, v) || check(y, x, v, u)) {
                    mk[y] = true;
                }
                if (check(u, v, x, y) || check(u, v, y, x)) {
                    mk[u] = true;
                }
                if (check(v, u, x, y) || check(v, u, y, x)) {
                    mk[v] = true;
                }
            }
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i) if (mk[i]) ++cnt;
        return cnt;
    }
};
