class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int perf[507][507];
		int um[507];
		for (int i = 0; i < n; ++i) {
			for (int j = 0, v = n - 1; j < n - 1; ++j, --v) {
				perf[i][preferences[i][j]] = v;
			}
		}
		for (int i = 0; i < n / 2; ++i) {
			um[pairs[i][0]] = pairs[i][1];
			um[pairs[i][1]] = pairs[i][0];
		}
		int ans = 0;
		for (int x = 0; x < n; ++x) {
			int y = um[x];
			for (int u = 0; u < n; ++u) {
				if (u == y || x == u) continue;
				if (perf[x][u] > perf[x][y] && perf[u][x] > perf[u][um[u]]) {
					ans++;
					break;
                    cout << x << endl;
				}
			}
		}
		return ans;
    }
};
