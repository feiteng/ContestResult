class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& pre, vector<vector<int>>& pairs) {
        int m = n / 2;
        map<int,int> mp;
        for(auto v : pairs) {
            mp[v[0]] = v[1];
            mp[v[1]] = v[0];
        }

        int ans = 0;
        vector<vector<int>> pos(n,vector<int>(n));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n - 1; j++) {
                pos[i][pre[i][j]] = j;
            }
        }
        for(int i = 0; i < n; i++) {
            int p = mp[i];
            for(int j = 0; j < n - 1; j++) {
                if(pre[i][j] == p) break;
                int x = pre[i][j];

                int y = mp[x];
                if(pos[x][i] < pos[x][y]) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
