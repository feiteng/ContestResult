class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> P(n, vector<int>(n,-1));
        
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < n-1; j++)
            {
                P[i][preferences[i][j]] = j;
            }
        }
        
        vector<int> Q(n);
        for(auto &u : pairs)
        {
            int a = u[0];
            int b = u[1];
            
            Q[a] = b;
            Q[b] = a;
        }
        
        int total = 0;
        
        for(int i = 0; i < n; i++)
        {
            int t = P[i][Q[i]];
            bool unhappy = false;
            for(int j = 0; j < t; j++)
            {
                int u = preferences[i][j];
                if(P[u][i] < P[u][Q[u]]) // if u prefers i over its partner
                {
                    unhappy = true;
                    break;
                }
            }
            if(unhappy)
            {
                total++;
            }
        }
        
        return total;
    }
};
