class Solution {
public:
    unordered_set<int> s;
    void judge(vector<vector<int>>& dp, int x, int y, int u, int v){
        if(!s.count(x) && dp[x][u] > dp[x][y] && dp[u][x] > dp[u][v]){
            s.insert(x);
        }
        if(!s.count(x) && dp[x][v] > dp[x][y] && dp[v][x] > dp[v][u]){
            s.insert(x);
        }
    }
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> dp(n, vector<int>(n));
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n - 1; j++){
                dp[i][preferences[i][j]] = n - j;
            }
        }
        
        int len = pairs.size();
        for(int i = 0; i < len; i++){
            for(int j = i + 1; j < len; j++){
                int x = pairs[i][0];
                int y = pairs[i][1];
                int u = pairs[j][0];
                int v = pairs[j][1];
                judge(dp, x, y, u, v);
                judge(dp, y, x, u, v);
                judge(dp, u, v, x, y);
                judge(dp, v, u, x, y);
            }
        }
        return s.size();
    }
};
