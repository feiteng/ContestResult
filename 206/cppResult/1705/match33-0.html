class Solution {
public:
  int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
    vector<vector<int>> ords;
    ords.resize(n, vector<int>(n));
    vector<bool> unhappy(n);
    for (int i = 0; i < preferences.size(); i++) {
      for (int j = 0; j < preferences[i].size(); j++) {
        ords[i][preferences[i][j]] = j;
      }
      ords[i][i] = n-1;
    }
    auto check = [&ords](int x, int y, int u, int v) {
    //  cout << "##" <<ords[x][u] << " " << ords[x][y] << " " << ords[u][x] << " " << ords[u][v] << endl;
      return ords[x][u] < ords[x][y] && ords[u][x] < ords[u][v];
    };
    for (int i = 0; i < pairs.size(); i++) {
      for (int j = 0; j < pairs.size(); j++) if (i != j) {
        if (check(pairs[i][0], pairs[i][1], pairs[j][0], pairs[j][1])) {
          unhappy[pairs[i][0]] = true;
     //     cout << i << " " << j << " 0 " << endl;
        }
        if (check(pairs[i][0], pairs[i][1], pairs[j][1], pairs[j][0])) {
          unhappy[pairs[i][0]] = true;
      //    cout << i << " " << j << " 1 " << endl;
        }
        if (check(pairs[i][1], pairs[i][0], pairs[j][0], pairs[j][1])) {
          unhappy[pairs[i][1]] = true;
       //   cout << i << " " << j << " 2 " << endl;
        }
        if (check(pairs[i][1], pairs[i][0], pairs[j][1], pairs[j][0])) {
          unhappy[pairs[i][1]] = true;
        //  cout << i << " " << j << " 3 " << endl;
        }
      }
    }
    return count(unhappy.begin(), unhappy.end(), true);
  }
};
