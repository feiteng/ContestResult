class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<int> pair(n);
        for (auto p : pairs){
            pair[p[0]] = p[1];
            pair[p[1]] = p[0];
        }
        vector<vector<int>> score(n);
        for (int i = 0; i < n; i++){
            score[i].resize(n);
        }
        for (int i = 0; i < n; i++){
            for (int j = 0; j < n - 1; j++){
                score[i][preferences[i][j]] = n - j;
            }
        }
        int ans = 0;
        for (int x = 0; x < n; x++){
            for (int u = 0; u < n; u++){
                if (x == u) {
                    continue;
                }
                int y = pair[x];
                int v = pair[u];
                if (score[x][u] > score[x][y] && score[u][x] > score[u][v]){
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
