class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> rank(n, vector<int>(n,0));
        for (int i = 0; i < n; i++){
            auto&p = preferences[i];

            for (int j = 0; j < p.size(); j++){
                rank[i][p[j]] = j;
            }
        }
        vector<int> other(n,0);
        for (auto&p:pairs){
            int u = p[0];
            int v = p[1];
            other[u] = v;
            other[v] = u;
        }
        int ans = 0;
        for (int x = 0; x <n ; x++){
            bool flag = false;

            int y = other[x];
            for (int u = 0; u < n; u++){
                int v = other[u];
                if (x!=u && y!=u && rank[x][u]<rank[x][y] && rank[u][x]<rank[u][v]){
                    flag = true;
                    break;
                }
            }
            if (flag){
                ans++;
            }
        }
        return ans;

    }
};
