class Solution {
 public:
  int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairss) {
    vector<int> pairs(n);

    vector<vector<int>> pref(n, vector<int> (n, 0));
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n - 1; j++) {
        pref[i][preferences[i][j]] = j;
      }
    }

    for (auto p : pairss) {
      pairs[p[0]] = p[1];
      pairs[p[1]] = p[0];
    }
    int res = 0;
    for (int i = 0; i < n; i++) {
      int p = pairs[i];
      for (int j = 0; j < n; j++) {
        if (j == p || j == i || pref[i][j] > pref[i][p]) continue;

        int jp = pairs[j];
        if (pref[j][i] < pref[j][jp]) {
          res++;
          break;
        }
      }
    }
    return res;
  }
};
