class Solution {
public:
    map<int,int> mp;
    int p[550][550];
    bool vis[550];
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int cnt = 0;
        for (int i = 0; i < pairs.size(); i++) {
            mp[pairs[i][0]] = pairs[i][1];
            mp[pairs[i][1]] = pairs[i][0];
        }
        for (int i = 0; i < n; i++) {
            int tmp = n;
            for (int j = 0; j < n - 1; j++) {
                p[i][preferences[i][j]] = tmp;
                tmp--;
            }
        }
        memset(vis, 0, sizeof(vis));

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if(i == j) continue;
                int pi = mp[i];
                int pj = mp[j];
                if(i == pj) continue;
                if(p[i][pi] < p[i][j] && p[j][pj] < p[j][i]) vis[i] = 1;
            }
        }
        for (int i = 0; i < n; i++) if(vis[i]) cnt++;
        return cnt;
    }
};
