class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> pref(n, vector<int>(n, n));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n-1; j++) {
                pref[i][preferences[i][j]] = j;
            }
        }
        
        vector<int> partner(n);
        for (auto p : pairs) {
            partner[p[0]] = p[1];
            partner[p[1]] = p[0];
        }
        
        int ans = 0;

        for (int x = 0; x < n; x++) {
            int y = partner[x];
            for (int u = 0; u < n; u++) {

                int v = partner[u];
                if (pref[x][u] < pref[x][y] && pref[u][x] < pref[u][v]) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
