class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {

        vector<map<int, int>> rank(n, map<int, int>());
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < preferences[i].size(); ++j) {
                rank[i][preferences[i][j]] = j;
            }
        }
        vector<int> max_rank(n, 0);
        for (int i = 0; i < pairs.size(); ++i) {
            int first = pairs[i][0], second = pairs[i][1];

            max_rank[first] = max(max_rank[first], rank[first][second]);
            max_rank[second] = max(max_rank[second], rank[second][first]);
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            int rk = max_rank[i];
            for (int j = 0; j < rk; ++j) {
                int u = preferences[i][j];
                if (max_rank[u] > rank[u][i]) {
                    cnt++;
                    break;
                }
            }
        }
            
        return cnt;
    }
};
