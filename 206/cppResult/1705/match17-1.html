class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {

        vector<unordered_map<int, int>> p(n);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < preferences[i].size(); ++j) {
                p[i][preferences[i][j]] = j;
            }
        }        
        vector<int> buddy(n);

        for (const auto& pair : pairs) {
            buddy[pair[0]] = pair[1];
            buddy[pair[1]] = pair[0];
        }
        
        int ans = 0;
        for (int x = 0; x < n; ++x) {
            int y = buddy[x];
            for (int u = 0; u < n; ++u) {
                if (u == x || u == y) continue;
                int v = buddy[u];
                if (p[x][u] < p[x][y] && p[u][x] < p[u][v]) {
                    ans++;
                    break;
                }
            }

        }
        return ans;
    }
};
