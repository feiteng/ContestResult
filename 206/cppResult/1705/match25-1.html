class Solution {
public:
    vector<int> vHave;
    void isOK(vector<vector<int>> &vTable,int x,int y,int u,int v){
        if(vHave[x]>0){
            return;
        }
        if(vTable[x][u]<vTable[x][y]&&vTable[u][x]<vTable[u][v]){
            vHave[x]++;

        }
    }
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> vTable(n,vector<int>(n));
        for(int i=0;i<n;i++){
            for(int j=0;j<n-1;j++){
                vTable[i][preferences[i][j]]=j;
            }
        }
        int N=pairs.size();
        vHave=vector<int>(n);

        for(int i=0;i<N;i++){
            for(int j=i+1;j<N;j++){
                int x=pairs[i][0];
                int y=pairs[i][1];
                int u=pairs[j][0];
                int v=pairs[j][1];
                isOK(vTable,x,y,u,v);
                isOK(vTable,x,y,v,u);
                isOK(vTable,y,x,u,v);
                isOK(vTable,y,x,v,u);  
                
                isOK(vTable,u,v,x,y);
                isOK(vTable,v,u,x,y);
                isOK(vTable,u,v,y,x);
                isOK(vTable,v,u,y,x);
            }
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            if(vHave[i]>0){
                ans++;
            }
        }
        
        return ans;
    }
};
