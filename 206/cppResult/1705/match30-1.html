class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> id(n, vector<int>(n));
        for (int i = 0; i < n; ++i) {
            for (int j = 1; j <= n - 1; ++j) id[i][preferences[i][j-1]] = j;

        }
        vector<bool> flag(n);
        for (int i = 0; i < pairs.size(); ++i) {
            int x = pairs[i][0], y = pairs[i][1];
            for (int j = 0; j < pairs.size(); ++j) if (i != j) {
                int u = pairs[j][0], v = pairs[j][1];
                
                if ((id[x][u] < id[x][y] && id[u][x] < id[u][v]) || (id[x][v] < id[x][y] && id[v][x] < id[v][u])) flag[x] = true;
                
                if ((id[y][u] < id[y][x] && id[u][y] < id[u][v]) || (id[y][v] < id[y][x] && id[v][y] < id[v][u])) flag[y] = true;

            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) if (flag[i]) ans++;
        return ans;
    }
};
