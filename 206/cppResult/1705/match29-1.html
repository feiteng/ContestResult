class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> v(n, vector<int>(n));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                v[i][preferences[i][j]] = j;
            }
        }
        vector<int> nxt(n);

        for (const auto& p: pairs) {
            nxt[p[0]] = p[1];
            nxt[p[1]] = p[0];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i != j && nxt[i] != j) {
                    int i_part = nxt[i];
                    int j_part = nxt[j];
                    if (v[i][j] < v[i][i_part] && v[j][i] < v[j][j_part]) {
                        ++ans;
                        break;
                    }
                }
            }
        }
        return ans;
    }
};
