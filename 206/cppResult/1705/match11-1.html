class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> p(n, vector<int>(n, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                p[i][preferences[i][j]] = j;
            }
        }
        vector<int> pairs_map(n, 0);
        for (vector<int>& pair : pairs) {

            pairs_map[pair[0]] = pair[1];
            pairs_map[pair[1]] = pair[0];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int f = pairs_map[i];
            for (int j = 0; j < n; ++j) {
                if (i != j && j != f) {
                    if (p[i][j] < p[i][f] && p[j][i] < p[j][pairs_map[j]]) {
                        ++ans;
                        break;
                    }
                }
            }
        }
        return ans;
    }
};
