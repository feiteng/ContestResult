class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        unordered_map<int, int> mp;
        for(auto p : pairs) {
            mp[p[0]] = p[1];
            mp[p[1]] = p[0];
        }
        int res = 0;
        for(int i = 0; i < n; ++i) {
            int x = i, y = mp[x];
            bool f = false;
            for(auto u : preferences[x]) {
                if(u == y) break;

                for(auto v : preferences[u]) {
                    if(v == mp[u]) break;
                    if(v == x) {++res; f = true; break;}
                }
                if(f) break;
            }
        }
        return res;
    }
};
