class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int res = 0;
        unordered_map<int, int> m;
        for(auto p:pairs){
            m[p[0]] = p[1];
            m[p[1]] = p[0];
        }
        for(auto p:pairs){
            int f0 = p[0], f1 = p[1];
            if(!helper(f0, preferences, pairs, m)){
                res++;
            }
            if(!helper(f1, preferences, pairs, m)){
                res++;
            }

        }
        return res;
    }
    
    bool helper(int f, vector<vector<int>>& preferences, vector<vector<int>>& pairs, unordered_map<int, int> &m){
        int i=0;
        while(preferences[f][i] != m[f]){
            int tf = preferences[f][i], j = 0;
            bool flag = false;
            while(preferences[tf][j] != f)
                if(preferences[tf][j++] == m[tf]){
                    flag = true;
                    break;
                }
            if(!flag) return false;
            ++i;
        }
        return true;
    }
};
