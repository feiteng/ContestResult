class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>>rank(n, vector<int>(n));
        for(int i = 0; i < n; i++)  {
            for(int j = 0; j < n-1; j++)  {
                rank[i][preferences[i][j]] = j;
            }
        }
        vector<int>rem(n);
        for(int i = 0; i < pairs.size(); i++)  {
            rem[pairs[i][0]] = pairs[i][1];
            rem[pairs[i][1]] = pairs[i][0];
        }
        int ans = 0;
        for(int x = 0; x < n; x++)   {
            int y = rem[x];
            for(int u = 0; u < n; u++)  {
                if(u == x || u == y)   continue;
                int v = rem[u];
                if(rank[x][u] < rank[x][y] && rank[u][x] < rank[u][v])  {
                    ans++;
                    break;
                }                                               
            }
        }
        return ans;
    }
};
