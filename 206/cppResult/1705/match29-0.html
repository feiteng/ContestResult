class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& p_, vector<vector<int>>& pairs) {
        int m[505];
        int pref[505][505];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                pref[i][p_[i][j]] = -j;
            }
            pref[i][i] = -1e9;
        }
        for (const auto& p : pairs) {
            m[p[0]] = p[1];
            m[p[1]] = p[0];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (pref[i][j] > pref[i][m[i]] and pref[j][i] > pref[j][m[j]]) {
                    ++ans;
                    break;
                }
            }
        }
        return ans;
    }
};
