class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<vector<int>> prefervalue(n, vector<int>(n, 0));
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < n - 1; j++)
            {
                prefervalue[i][preferences[i][j]] = n - j;
            }
        }
        map<int, int> connect;

        for(auto p : pairs)
        {
            connect[p[0]] = p[1];
            connect[p[1]] = p[0];
        }
        int count = 0;
        for(int i = 0; i < n; i++)
        {
            int value = prefervalue[i][connect[i]];
            for(int j = 0; j < n; j++)
            {
                if(prefervalue[i][j] <= value) continue;
                
                if(prefervalue[j][i] > prefervalue[j][connect[j]])
                {
                    count++;
                    break;
                }
            }
        }
        return count;
    }
};
