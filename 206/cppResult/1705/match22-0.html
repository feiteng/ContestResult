class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<int> paired(n);
        
        for (auto p : pairs) {
            paired[p[0]] = p[1];
            paired[p[1]] = p[0];
        }
        
        vector<unordered_map<int, int>> orders(n);
        
        for (int i = 0; i < n; i ++) {
            vector<int> preference = preferences[i];
            
            for (int j = 0; j < n - 1; j ++) {
                orders[i][preference[j]] = j;
            }
        }
        
        int res = 0;
        for (int cur = 0; cur < n; cur ++) {
            int p = paired[cur];
            
            for (auto f : preferences[cur]) {
                if (f == p) {
                    break;
                }
                
                int pf = paired[f];
                
                if (orders[f][cur] < orders[f][pf]) {
                    res++;
                    break;
                }
            }
        }
        
        return res;
    }
};
