class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        vector<int> vpair(n);
        for(vector<int> &vi:pairs)
        {
            vpair[vi[0]]=vi[1];
            vpair[vi[1]]=vi[0];
        }
        vector<vector<int>> pos(n,vector<int>(n));
        for(int i=0;i<n;++i)
        {
            for(int j=0;j<preferences[i].size();++j)
            {
                pos[i][preferences[i][j]]=j;
            }
        }
        int ans=0;
        for(int x=0;x<n;++x)
        {
            int y=vpair[x];
            for(int i=0;i<preferences[x].size();++i)
            {
                int u=preferences[x][i];
                int v=vpair[u];
                // x 与 u 的亲近程度胜过 x 与 y，且
                // u 与 x 的亲近程度胜过 u 与 v
                if(pos[x][u]<pos[x][y] && pos[u][x]<pos[u][v])
                {
                    ++ans;
                    break;
                }
            }
        }
        return ans;
    }
};
