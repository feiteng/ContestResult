class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int tp[505][505];
        for (int i = 0; i < preferences.size(); ++i) {
            for (int j = 0; j < preferences[0].size(); ++j) {
                tp[i][preferences[i][j]] = j;
            }
        }
        int ans = 0;
        int vis[505] = {0};
        for (int i = 0; i < pairs.size(); ++i) {
            for (int j = i + 1; j < pairs.size(); ++j) {
                int u = pairs[i][0], v = pairs[i][1];
                int w = pairs[j][0], x = pairs[j][1];
                if (tp[u][w] < tp[u][v] && tp[w][u] < tp[w][x]) {
                    vis[u] = 1, vis[w] = 1;
                }
                if (tp[u][x] < tp[u][v] && tp[x][u] < tp[x][w]) {
                    vis[u] = 1, vis[x] = 1;
                }
                if (tp[v][w] < tp[v][u] && tp[w][v] < tp[w][x]) {
                    vis[v] = 1, vis[w] = 1;
                }
                if (tp[v][x] < tp[v][u] && tp[x][v] < tp[x][w]) {
                    vis[v] = 1, vis[x] = 1;
                }
            }
        }
        for (int i = 0; i < n; ++i) {
            ans += vis[i];
        }
        return ans;
    }
};
