class Solution
{
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs)
    {
        vector<vector<int>> prefs(n, vector<int>(n, -1));
        vector<int> other(n);
        for (int i = 0; i < n; i++) {
            int p = 0;
            for (int j : preferences[i])
                prefs[i][j] = p++;
        }
        for (auto& pair : pairs) {
            other[pair[0]] = pair[1];
            other[pair[1]] = pair[0];
        }
        int count = 0;
        for (int x = 0; x < n; x++) {
            bool unhappy = false;
            int y = other[x];
            for (int u = 0; u < n; u++) {
                if (u == x || u == y) continue;
                int v = other[u];
                if (prefs[x][u] < prefs[x][y] && prefs[u][x] < prefs[u][v]) {
                    unhappy = true;
                    break;
                }
            }
            if (unhappy) count++;
        }
        return count;
    }
};
