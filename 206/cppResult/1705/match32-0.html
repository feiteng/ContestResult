class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& a, vector<vector<int>>& pairs) {
        vector<vector<int>> r(n, vector<int>(n));
        for (int i = 0; i < n; ++i) {
            for (int k = 0; k < n - 1; ++k) {
                r[i][a[i][k]] = k;
            }
        }
        int m = pairs.size();
        vector<bool> flag(n);
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < m; ++j) {
                if (i == j) continue;
                int x = pairs[i][0], y = pairs[i][1];
                int u = pairs[j][0], v = pairs[j][1];
                if (r[x][u] < r[x][y] && r[u][x] < r[u][v]) flag[x] = true;
                swap(u, v);
                if (r[x][u] < r[x][y] && r[u][x] < r[u][v]) flag[x] = true;
                swap(x, y);
                if (r[x][u] < r[x][y] && r[u][x] < r[u][v]) flag[x] = true;
                swap(u, v);
                if (r[x][u] < r[x][y] && r[u][x] < r[u][v]) flag[x] = true;
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) ret += flag[i];
        return ret;
    }
};
