class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int m=pairs.size();

        vector<vector<int>> M;
        M.resize(n);
        
        for(int i=0;i<n;i++){
            M[i].resize(n);
            for(int j=0;j<n-1;j++) {
                M[i][preferences[i][j]]=j;
            }
        }
        int ans=0;

        bool b[501]={};
        for(int i=0;i<m;i++) {
            for(int j=i+1;j<m;j++) {
                int x=pairs[i][0],y=pairs[i][1],u=pairs[j][0],v=pairs[j][1];

                if(M[x][u]<M[x][y]&&M[u][x]<M[u][v]) {
                    b[x]=b[u]=1;  
                }
                if(M[x][v]<M[x][y]&&M[v][x]<M[v][u]) {
                    b[x]=b[v]=1;  
                }
                swap(x,y);
                if(M[x][u]<M[x][y]&&M[u][x]<M[u][v]) {
                    b[x]=b[u]=1;  
                }
                if(M[x][v]<M[x][y]&&M[v][x]<M[v][u]) {
                    b[x]=b[v]=1;  
                }
                
            }
        }
        int t=accumulate(b,b+n,0);
        return t;
    }
};
