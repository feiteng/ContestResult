class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& p, vector<vector<int>>& ps) {
        vector<vector<int>> q(n,vector<int>(n));
        for(int i=0;i<n;++i){
            for(int j=0;j<n-1;++j){
                q[i][p[i][j]]=n-j;
            }
        }
        vector<bool> hp(n,true);
        for(int i=0;i<ps.size();++i){
            int x=ps[i][0],y=ps[i][1];
            for(int j=0;j<ps.size();++j){
                if(i==j) continue;
                int u=ps[j][0],v=ps[j][1];
                // cout<<q[x][u]<<" "<<q[x][y]<<" "<<q[u][x]<<" "<<q[u][v]<<endl;
                if(q[x][u]>q[x][y]&&q[u][x]>q[u][v]) hp[x]=false;
                if(q[x][v]>q[x][y]&&q[v][x]>q[v][u]) hp[x]=false;
                if(q[y][u]>q[y][x]&&q[u][y]>q[u][v]) hp[y]=false;
                if(q[y][v]>q[y][x]&&q[v][y]>q[v][u]) hp[y]=false;
            }
        }
        int ans=0;
        for(int i=0;i<n;++i) if(!hp[i]) ++ans;
        return ans;
    }
};
