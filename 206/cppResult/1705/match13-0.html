class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int res = 0;
        unordered_map<int, int> m;
        for(auto p:pairs){
            m[p[0]] = p[1];
            m[p[1]] = p[0];
        }
        for(auto p:pairs){
            int f0 = p[0], f1 = p[1];
            if(!for_for( preferences, pairs,f0, m)) res++;
            if(!for_for( preferences, pairs,f1, m)) res++;
        }
        return res;
    }
    
    bool for_for( vector<vector<int>>& pre, vector<vector<int>>& pairs,int f, unordered_map<int, int> &mp){
        int i=0;
        while(pre[f][i] != mp[f]){
            int tf = pre[f][i], j = 0;
            bool flag = false;
            while(pre[tf][j] != f)
                if(pre[tf][j++] == mp[tf]){
                    flag = true;
                    break;
                }
            if(!flag) return false;
            ++i;
        }
        return true;
    }
};
