class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
vector<int> p(n);
        for (auto i : pairs) {
            p[i[0]]=i[1];
            p[i[1]]=i[0];
        }
        vector<vector<int>> s(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < preferences[i].size(); j++) {
                s[i][preferences[i][j]]=j;
            }
        }
        int cnt=0;
        for (int i = 0; i < n; i++) {
            int part = p[i];
            int pre = s[i][part];
            bool flag=false;
            for (int j = 0; j < pre; j++) {
                int can = preferences[i][j];
                int part2 = p[can];
                int pre2 = s[can][part2];
                if (s[can][i]<pre2) {
                    flag=true;
                    break;
                }
            }
            if (flag) cnt++;
        }
        return cnt;
    }
};
