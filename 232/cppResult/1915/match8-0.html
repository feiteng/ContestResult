class Solution {
public:
    bool areAlmostEqual(string s1, string s2) {
        for (int i = 0; i < s1.size(); ++i) {
            for (int j = i; j < s1.size(); ++j) {
                string s3 = s1;
                swap(s3[i], s3[j]);
                if (s3 == s2) return true;
            }
        }
        return false;
    }
};
