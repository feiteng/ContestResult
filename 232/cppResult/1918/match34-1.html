class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int left = k-1;
        int right = k+1;
        int minn = nums[k];
        int ans = minn;
        int n = nums.size();
        while(left>=0 || right <n){

            if(left>=0 && (right==n || nums[left]>=nums[right])){
                minn = min(minn,nums[left]);
                ans = max(ans,minn*(right-left));
                left--;
            }
            else{
                minn = min(minn,nums[right]);
                ans = max(ans,minn*(right-left));
                right++;
            }
            // cout<<left<<" "<<right<<endl;
        }
        return ans;
    }
};
