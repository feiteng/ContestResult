class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int out = nums[k];
        int i = k-1, j = k+1;
        int n = nums.size(), cur_min = nums[k];
        while(i >=0 || j < n){
            int left = (i<0? -1: nums[i]);
            int right = (j>=n? -1: nums[j]);
            if(left > right){
                cur_min = min(cur_min, left);
                out = max(out, (j-i)*cur_min);
                i--;
            }
            else{
                cur_min = min(cur_min, right);
                out = max(out, (j-i)*cur_min);
                j++;
            }
        }
        return out;
    }
};
