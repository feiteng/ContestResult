class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int i=k;
        int j=k;
        int minn=nums[k];
        int ans=minn;
        while(1){
            while(i>=0&&nums[i]>=minn){
                i--;
            }
            while(j<nums.size()&&nums[j]>=minn){
                j++;
            }
            ans=max(ans,minn*(j-i-1));
            if(i==-1&&j==nums.size())break;
            if(i==-1){
                minn=nums[j];
                j++;
            }else if(j==nums.size()){
                minn=nums[i];
                i--;
            }else{
                if(nums[i]>nums[j]){
                    minn=nums[i];
                    i--;
                    
                }else{
                    minn=nums[j];
                    j++;
                }
            }
        }
        return ans;

    }
};
