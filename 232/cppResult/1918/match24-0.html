class Solution {
public:
int maximumScore(vector<int>& nums, int k) {
   int curr_min = nums[k];
   int left = k-1;
   int right = k+1;
   int res = curr_min;

   while (left > 0 || right < nums.size()) {
      if (left < 0) {
         curr_min = min(curr_min, nums[right]);
         res = max(res, curr_min * (right - left));
         right++;
      } else if (right == nums.size()) {
         curr_min = min(curr_min, nums[left]);
         res = max(res, curr_min * (right - left));
         left--;
      } else if (nums[left] < nums[right]) {
         curr_min = min(curr_min, nums[right]);
         res = max(res, curr_min * (right - left));
         right++;
      } else {
         curr_min = min(curr_min, nums[left]);
         res = max(res, curr_min * (right - left));
         left--;
      }
   }
   return res;
}

};
