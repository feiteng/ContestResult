class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int res=nums[k];
        int l=k-1, r=k+1;
        int mi=nums[k];
        while (l>=0||r<nums.size()) {
            if (l<0) {
                mi=min(mi, nums[r]);
                r++;
                res=max(res, mi*(r-l-1));
            } else if (r>=nums.size()) {
                mi=min(mi, nums[l]);
                l--;
                res=max(res, mi*(r-l-1));
            } else if (nums[l]<nums[r]) {
                mi=min(mi, nums[r]);
                r++;
                res=max(res, mi*(r-l-1));
            } else {
                mi=min(mi, nums[l]);
                l--;
                res=max(res, mi*(r-l-1));
            }
        }
        return res;
    }
};
