class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int mn = nums[k], i = k - 1, j = k + 1, n = nums.size();
        int ans = mn;
        while(i >= 0 && j < n){
            if(nums[i] < nums[j]){
                mn = min(mn, nums[j]);
                ans = max(ans, mn * (j - i));
                j++;
            } else {
                mn = min(mn, nums[i]);
                ans = max(ans, mn * (j - i));
                i--;
            }
        }
        while(j < n){
            mn = min(mn, nums[j]);
            ans = max(ans, mn * (j + 1));
            j++;
        }
        while(i >= 0){
            mn = min(mn, nums[i]);
            ans = max(ans, mn * (n - i));
            i--;
        }
        return ans;
    }
};
