class Solution {
public:
    int maximumScore(vector<int>& a, int k) {
        int i = k, j = k, cur = a[k], res = a[k];
        int n = a.size();
        while(i > 0 && j + 1 < n) {
            if(a[i - 1] >= a[j + 1]){
                i--;
                cur = min(cur, a[i]);
                res = max(res, cur * (j - i + 1));
            }
            else {
                j++;
                cur = min(cur, a[j]);
                res = max(res, cur * (j - i + 1));
            }
        }
        while(i > 0) {
            i--;
            cur = min(cur, a[i]);
            res = max(res, cur * (j - i + 1));
        }
        while(j + 1 < n) {
            j++;
            cur = min(cur, a[j]);
            res = max(res, cur * (j - i + 1));
        }
        return res;
    }
};
