class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> lmin(n, -1), rmin(n, n);
        stack<int> st;
        for(int i = 0; i < n; i++){
            while(st.size() and nums[st.top()] > nums[i]){
                rmin[st.top()] = i;
                st.pop();
            }
            st.push(i);
        }
        while(st.size()) st.pop();

        for(int i = n-1; i >= 0; i--){
            while(st.size() and nums[st.top()] > nums[i]){
                lmin[st.top()] = i;
                st.pop();
            }
            st.push(i);
        }
        int ans = 0;
        for(int i = 0; i < n; i++){
            int left = lmin[i] + 1, right = rmin[i] - 1;
            if(left > k or right < k) continue;
            ans = max(ans, (rmin[i] - lmin[i] - 1) * nums[i]);
        }
        return ans;
    }
};
