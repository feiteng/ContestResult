class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
      int l = k;
      int r = k;
      int minv = nums[k];
      int len = 1;
      int ans = INT_MIN;
      while (len <= nums.size()) {
        minv = min(minv, min(nums[l], nums[r]));
        ans = max(ans, minv * len);
        ++len;
        if (l == 0) {

          ++r;
        } else if (r == k - 1) {
          --l;
        } else {
          if (nums[l - 1] >= nums[r + 1]) {
            --l;
          } else {
            ++r;
          }
        }

      }
      return ans;
    }
};
