class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        int l = k;
        int r = k;
        int res = nums[k];
        int s = nums[k];
        while(l > 0 && r < n - 1){
            if(nums[l - 1] > nums[r + 1]){
                l--;
                s = min(s, nums[l]);
                res = max(res, s * (r - l + 1));
            }
            else{
                r++;
                s = min(s, nums[r]);
                res = max(res, s * (r - l + 1));
            }
        }
        while(l > 0){
            l--;
            s = min(s, nums[l]);
            res = max(res, s * (r - l + 1));
        }
        while(r < n - 1){
            r++;
            s = min(s, nums[r]);
            res = max(res, s * (r - l + 1));
        }
        return res;
    }
};
