class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        int minVal = nums[k];
        int len  = 1;
        int ans = minVal * len;
        int l = k - 1;
        int r = k + 1;
        
        while(l >= 0 || r < n){
            len++;
            
            int i = 0;

            if(l < 0){
                i = r;
                r++;
            }else if(r == n){
                i = l;
                l--;
            }else if(nums[l] < nums[r]){
                i = r;
                r++;
            }else{
                i = l;
                l--;
            }
            
            
            minVal = min(minVal, nums[i]);
            
            int tmpAns = minVal * len;
            if(tmpAns > ans){
                ans = tmpAns;
            }
        }
        return ans;
    }
};
