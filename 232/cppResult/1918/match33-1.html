class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n=nums.size();
        int ans=nums[k];
        int i=k,j=k;
        int mi=nums[k];
        while(i>0||j+1<n)
        {
            //cout<<i<<" "<<j<<endl;
            if(i>0&&nums[i-1]>=mi)
            {
                //cout<<"here1"<<endl;
                i--;
                ans=max(ans,mi*(j-i+1));
                // continue;
            }
            else if(j+1<n&&nums[j+1]>=mi)
            {
                //cout<<"here2"<<endl;
                j++;
                ans=max(ans,mi*(j-i+1));
                //continue;
            }
            else if(i==0||j+1<n&&nums[j+1]>=nums[i-1])
            {
                //cout<<"here3"<<endl;
                j++;
                mi=nums[j];

                ans=max(ans,mi*(j-i+1));
                //continue;
            }
            else
            {
                //cout<<"here4"<<endl;
                i--;
                mi=nums[i];
                ans=max(ans,mi*(j-i+1));
                //continue;
            }
            //cout<<ans<<endl;
        }
        return ans;
    }
};
