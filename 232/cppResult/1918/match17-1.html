class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int i = k - 1, j = k + 1, ans = nums[k], n = nums.size(), minv = nums[k];

        while (i >= 0 && j < n) {
            if (nums[i] > nums[j]) {
                minv = min(minv, nums[i]);
                ans = max(ans, (j - i) * minv);
                i--;
            } else {
                minv = min(minv, nums[j]);
                ans = max(ans, (j - i) * minv);
                j++;
            }
            // cout << ans << endl;
        }
        while (i >= 0) {
            minv = min(minv, nums[i]);
            ans = max(ans, (j - i) * minv);
            i--;
        }
        while (j < n) {
            minv = min(minv, nums[j]);
            ans = max(ans, (j - i) * minv);
            j++;
        }
        return ans;
    }
};
