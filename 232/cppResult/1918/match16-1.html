class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        
        int res = nums[k];
        
        int i = k;
        int j = k;
        int mini = nums[k];
        while (i - 1 >= 0 && j + 1 < n) {
            if (nums[i - 1] > nums[j + 1]) {
                i -= 1;
                mini = min(mini, nums[i]);
                res = max(res, mini * (j - i + 1));
            } else if (nums[i - 1] < nums[j + 1]) {
                j += 1;
                mini = min(mini, nums[j]);
                res = max(res, mini * (j - i + 1));
            } else {
                i -= 1;

                j += 1;
                mini = min(mini, nums[i]);
                res = max(res, mini * (j - i + 1));
            }
        }
        
        while (j + 1 < n) {
            j += 1;
            mini = min(mini, nums[j]);
            res = max(res, mini * (j - i + 1));
        }
        
        while (i - 1 >= 0) {
            i -= 1;
            mini = min(mini, nums[i]);
            res = max(res, mini * (j - i + 1));
        }
        
        return res;
    }
};
