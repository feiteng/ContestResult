class Solution {
public:
    int maximumScore(vector<int>& a, int k) {
        int ans = a[k];
        int mn = a[k];
        int l = k-1;
        int j = k+1;
        int ll = k;
        int rr = k;
        int n = a.size();
        while(l >= 0 and j < n)
        {
            if(a[l] >= a[j])
            {
                mn = min(mn,a[l]);
                ll--;
                ans = max(ans,mn*(rr - ll + 1));
                l--;
            }
            else
            {
                mn = min(mn,a[j]);
                rr++;
                ans = max(ans,mn*(rr - ll + 1));
                j++;
            }
        }
        while(l >= 0)
        {
            mn = min(mn,a[l]);
            ll--;
            ans = max(ans,mn*(rr-ll+1));
            l--;
        }
        while(j < n)
        {
            mn = min(mn,a[j]);
            rr++;
            ans = max(ans,mn*(rr - ll + 1));
            j++;
        }
        return ans;
    }
};
