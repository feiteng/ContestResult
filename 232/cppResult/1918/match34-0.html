class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        // Cannot use greedy?
        int len = nums.size();
        int leftBound = k; // 0 ~ k
        int rightBound = k; // k ~ len-1
        int curMin = nums[k];
        int score = nums[k];

        while (leftBound-1 >= 0 || rightBound+1 <len) {
            int leftElement = (leftBound-1 >= 0) ? nums[leftBound-1] : std::numeric_limits<int>::min();
            int rightElement = (rightBound+1 < len) ? nums[rightBound+1] : std::numeric_limits<int>::min();
            if (leftElement > rightElement) {
                --leftBound;
                curMin = min(curMin, nums[leftBound]);
                score = max(score, curMin * (rightBound-leftBound+1));
            } else {
                ++rightBound;
                curMin = min(curMin, nums[rightBound]);
                score = max(score, curMin * (rightBound-leftBound+1));
            }
        }

        return score;
    }
};
