class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        int ans = nums[k];
        int left = k-1, right = k+1;
        int min_value = nums[k];
        while(left >= 0 || right < n) {
            int left_v = (left >= 0) ? nums[left] : 0;
            int right_v = (right < n) ? nums[right] : 0;

            if(left_v <= right_v) {
                min_value = min(min_value, right_v);
                ans = max(ans, min_value*(right - left));
                right++;
            } else {
                min_value = min(min_value, left_v);
                ans = max(ans, min_value*(right - left));
                left--;
            }
        }
        return ans;
    }
};
