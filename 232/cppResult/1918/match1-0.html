class Solution {
public:
  int maximumScore(vector<int> &nums, int k) {
    int rtn = nums[k];
    int n=  nums.size();
    int a = k, b =k;
    int mm = nums[k];
    while (a > 0 || b < n-1) {
      if (a == 0 || (b != n-1 && nums[a-1] < nums[b+1])) {
        mm = min(mm, nums[b+1]);
        b++;
        rtn = max(rtn, mm * (b-a+1));
      } else {
        mm = min(mm, nums[a-1]);
        a--;
        rtn = max(rtn, mm * (b-a+1));
      }
    }
    return rtn;

  }
};
