class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int ans = nums[k];
        int minn = nums[k];
        int left = k, right = k;
        int n = nums.size();
        while (left != 0 || right != n - 1) {
            int l = (left == 0) ? 0 : nums[left - 1];
            int r = (right == n - 1) ? 0 : nums[right + 1];

            if (l < r) {
                right++;
                minn = min(minn, r);
                ans = max(ans, minn * (right - left + 1));
            } else {
                left--;
                minn = min(minn, l);
                ans = max(ans, minn * (right - left + 1));
            }
        }
        return ans;
    }
};
