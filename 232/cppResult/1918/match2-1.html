class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int MIN=nums[k],n = nums.size(),ans=0;
        for(int l=k,r=k;l>=0;l--){
            MIN=min(MIN,nums[l]);
            while(r+1<n&&nums[r+1]>=MIN)
                r++;
            ans=max(ans,(r-l+1)*MIN);
        }
        MIN=nums[k];
        for(int l=k,r=k;r<n;r++){
            MIN=min(MIN,nums[r]);
            while(l>0&&nums[l-1]>=MIN)
                l--;
            ans=max(ans,(r-l+1)*MIN);
        }
        return ans;
    }
};
