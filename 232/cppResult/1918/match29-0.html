class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        
        int n = nums.size();
        int nse_right[n];
        int nse_left[n];
        
        stack<int>s;
        s.push(0);
        for(int i=1;i<n;i++){
            while(!s.empty() && nums[s.top()]>nums[i]){
                nse_right[s.top()] = i;
                s.pop();
            }
            s.push(i);
        }
        while(!s.empty()){
            nse_right[s.top()] = n;
            s.pop();
        }
        s.push(n-1);
        for(int i=n-2;i>=0;i--){
             while(!s.empty() && nums[s.top()]>nums[i]){
                nse_left[s.top()] = i;
                s.pop();
            }
            s.push(i);
        }
         while(!s.empty()){
            nse_left[s.top()] = -1;
            s.pop();
        }
        
        int res = 0;
        for(int i=0;i<n;i++){
            int l = nse_left[i]+1;
            int r = nse_right[i]-1;
            if(k>=l && k<=r)
                res = max(res,(r-l+1)*nums[i]);
        }
        return res;
    }
};
