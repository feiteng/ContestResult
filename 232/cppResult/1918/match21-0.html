class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {        
        int i = k, j = k;
        int n = nums.size();
        int score = nums[k];
        int mini = score;
        
        while(i > 0 && j < n - 1) {
            if(nums[i - 1] > nums[j + 1]) {
                --i;
                mini = min(mini, nums[i]);
            } else {
                ++j;
                mini = min(mini, nums[j]);
            }
            score = max(score, mini*(j - i + 1));
        }
        
        while(i > 0) {
            --i;
            mini = min(mini, nums[i]);
            score = max(score, mini*(j - i + 1));
        }
        
        while(j < n - 1) {
            ++j;
            mini = min(mini, nums[j]);
            score = max(score, mini*(j - i + 1));
        }
        
        return score;
    }
};
