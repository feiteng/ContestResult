class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int size = nums.size();
        
        int res = nums[k];
        int minnum = nums[k];
        
        int left = k - 1, right = k + 1;

        while(left >= 0 && right < size){
            if(nums[left] < nums[right]){
                minnum = min(minnum, nums[right]);
                res = max(res, minnum * (right - left));
                right++;
            } else{
                minnum = min(minnum, nums[left]);
                res = max(res, minnum * (right - left));
                left--;
            }
        }
        
        while(left >= 0){
            minnum = min(minnum, nums[left]);
            res = max(res, minnum * (right - left));
            left--;
        }
        
        while(right < size){
            minnum = min(minnum, nums[right]);
            res = max(res, minnum * (right - left));
            right++;
        }
        
        
        
        
        return res;
    }
};
