class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
      int n = nums.size();
      vector<int> left(n), right(n);
      
      stack<int> stk;
      for (int i = 0; i < n; ++i) {
        while (!stk.empty() && nums[stk.top()] >= nums[i]) {
          stk.pop();
        }
        left[i] = (stk.empty() ? -1 : stk.top());
        stk.push(i);
      }

      stk = stack<int>();
      for (int i = n - 1; i >= 0; --i) {
        while (!stk.empty() && nums[stk.top()] >= nums[i]) {
          stk.pop();
        }
        right[i] = (stk.empty() ? n : stk.top());
        stk.push(i);
      }

      int ans = INT_MIN;
      for (int i = 0; i < n; ++i) {
        if (left[i]+1 <= k && k <= right[i]-1) {
          ans = max(ans, (right[i] - left[i] - 1) * nums[i]);
        }
      }
      
      return ans;
    }
};
