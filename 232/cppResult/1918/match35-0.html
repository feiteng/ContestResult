class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int len = nums.size();
        int left = k;
        int right = k;
        int minl = nums[k];
        int minr = nums[k];
        int ans = nums[k];
        while(!(left == 0 && right == len - 1)) {
            if(right == len - 1) {
                left--;
                minl = min(minl, nums[left]);
            }else if(left == 0) {
                right++;
                minr = min(minr, nums[right]);
            }else if(nums[left - 1] < nums[right + 1]) {
                right++;
                minr = min(minr, nums[right]);
            }else{
                left--;
                minl = min(minl, nums[left]);
            }
            ans = max(ans, min(minl, minr) * (right - left + 1));
        }
        return ans;
    }
};
