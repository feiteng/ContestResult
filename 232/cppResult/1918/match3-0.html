class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        int i=k;
        int j=k;
        int res = nums[k];
        int m = nums[k];
        while(i>0 && j<n-1){
            if(nums[i-1]>=nums[j+1]){
                i--;
                m = min(m,nums[i]);
                res = max(res,m*(j-i+1));
            }
            else{
                j++;
                m = min(m,nums[j]);
                res = max(res,m*(j-i+1));
            }
        }
        while(i>0){
            i--;
            m = min(m,nums[i]);
            res = max(res,m*(j-i+1));
        }
        while(j<n-1){
            j++;
            m = min(m,nums[j]);
            res = max(res,m*(j-i+1));
        }
        return res;
    }
};
