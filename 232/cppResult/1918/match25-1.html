class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int n = nums.size();
        int i = k, j = k;
        int cur_min  = nums[k];
        while (i >= 0 && nums[i] >= cur_min) {
            i--;
        }
        while (j < n && nums[j] >= cur_min) {
            j++;
        }

        int result = cur_min * (j - 1 - i);
        while (i >= 0 || j < n) {
            if (i < 0) {
                cur_min = min(cur_min, nums[j]);
                result = max(result, cur_min * (j - i));
                j++;
            } else if (j >= n) {
                cur_min = min(cur_min, nums[i]);
                result = max(result, cur_min * (j - i));
                i--;
            } else if (nums[i] >= nums[j]) {
                cur_min = min(cur_min, nums[i]);
                result = max(result, cur_min * (j - i));
                i--;
            } else {
                cur_min = min(cur_min, nums[j]);
                result = max(result, cur_min * (j - i));
                j++;
            }
        }
        return result;
    }
};
