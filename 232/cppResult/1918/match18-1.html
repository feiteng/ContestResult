class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int res=nums[k];
        int i=k, j=k, n=nums.size(), mi=nums[k];

        while(i>0||j<n-1) {
            int left=(i>0?nums[i-1]:INT_MIN);
            int right=(j<n-1?nums[j+1]:INT_MIN);
            if (left<=right) {
                mi=min(mi, nums[++j]);
                res=max(res, mi*(j-i+1));
            } else {
                
                mi=min(mi, nums[--i]);
                res=max(res, mi*(j-i+1));
            }
        }
        return res;
    }
};
