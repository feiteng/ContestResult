class Solution {
public:
    int maximumScore(vector<int>& nums, int k) {
        int i = k;
        int j = k;
        int leftMin = nums[k];
        int rightMin = nums[k];
        int maxScore = 0;
        while(i >= 0 && j < nums.size()) {
            leftMin = min(leftMin, nums[i]);
            rightMin = min(rightMin, nums[j]);
            
            
            maxScore = max(maxScore, min(leftMin, rightMin) * (j - i + 1));
            //cout << "LeftMin " << leftMin << " rightMin " << rightMin << " maxScore " << maxScore << endl;
            if(i-1 < 0 && j+1 >= nums.size()) {
                // We're at the end of the arrays
                break;
            } else if(i-1 < 0) {
                j++;
            } else if(j+1 >= nums.size()) {
                i--;
            } else {
                if(nums[i-1] < nums[j+1]) {
                    j++;
                } else {
                    i--;
                }
            }
        }
        
        return maxScore;
    }
};
