class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size()+1;
        vector<int> ind(n+1,0);
        for(const auto &x: edges){
            ind[x[0]]++;
            ind[x[1]]++;
        }
        for(int i = 1; i<=n; ++i){
            if(ind[i] == n-1){
                return i;
            }
        }
        return 0;
    }
};
