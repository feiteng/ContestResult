class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size() + 1;
        vector<int> indeg(n);
        for (auto& e : edges) {
            int u = e[0], v = e[1]; u--, v--;
            indeg[u]++;
            indeg[v]++;
        }
        for (int i = 0; i < n; i++) {
            if (indeg[i] == n-1) return i+1;
        }
        return -1;
    }
};
