class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        set<int> s;
        for(auto it : edges) {
            if (s.count(it[0]))
                return it[0];
            if (s.count(it[1]))
                return it[1];
            s.insert(it[0]);
            s.insert(it[1]);
        }
        return 0;
    }
};
