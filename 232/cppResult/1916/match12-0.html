class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<int> deg(n + 2);
        for (const auto& e: edges) {
            deg[e[0]]++;
            deg[e[1]]++;
        }
        
        for (int i = 0; i < n+2; i++) {
            //cout << deg[i] << endl;
            if (deg[i] == n) {
                return i;
            }
        }
        return -1;
    }
};
