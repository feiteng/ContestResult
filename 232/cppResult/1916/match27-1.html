class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        unordered_set<int> st;
        for (auto e : edges) {
            if (st.count(e[0])) return e[0];
            if (st.count(e[1])) return e[1];
            st.insert(e[0]);
            st.insert(e[1]);
        }
        return 0;
    }
};
