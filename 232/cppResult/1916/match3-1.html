class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
      int n = edges.size()+1;
      vector<int> cnt(n);
      for (auto &v : edges) {
        ++cnt[v[0]-1];
        ++cnt[v[1]-1];
      }
      for (int i = 0; i < n; ++i) {
        if (cnt[i] == n-1) {
          return i+1;
        }
      }
      return 0;
    }
};
