class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
      int n = edges.size() + 1;
      vector<int> nbs(n + 1);
      for (auto& e: edges) {
        int a = e[0], b = e[1];
        nbs[a]++;
        nbs[b]++;
      }
      for (int i = 1; i <= n; ++i) {
        if (nbs[i] == n - 1) {
          return i;
        }
      }
      return -1;
    }
};
