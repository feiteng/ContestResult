class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size() + 1;
        vector<int> deg(n);
        for (auto& e : edges) {
            int x = e[0], y = e[1];
            --x; --y;
            deg[x]++;
            deg[y]++;
        }
        for (int i = 0; i < n; ++i) {
            if (deg[i] == n - 1) return i + 1;
        }
        return 0;

    }
};
