class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size() + 1;
        vector<int> degree(n, 0);
        for (const auto& edge : edges) {
            degree[edge[0] - 1]++;
            degree[edge[1] - 1]++;
        }
        for (int i = 0; i < n; i++) {
            if (degree[i] == n - 1) {
                return i + 1;
            }
        }
        return -1;
    }
};
