class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size() + 1;
        vector<int> d(n + 1, 0);
        for(auto& v : edges) {
            int s = v[0], e = v[1];
            d[s]++, d[e]++;
        }
        for(int i = 1; i <= n; ++i) {
            if(d[i] != 1) {
                return i;
            }
        }
        return -1;
    }
};
