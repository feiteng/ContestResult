class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        unordered_map<int, int> mp;
        int res = 0, ct = 0;
        for(auto edge: edges) {
            int x = edge[0], y = edge[1];

            mp[x]++;
            mp[y]++;
            if(mp[x] > ct) {
                ct = mp[x];
                res = x;
            }
            if(mp[y] > ct) {
                ct = mp[y];
                res = y;
            }
        }
        return res;
    }
};
