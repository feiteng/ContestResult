class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size()+1;
        vector<int> deg(n);
        for(auto vec: edges) {
            int u = vec[0]-1, v = vec[1]-1;
            deg[u]++;
            deg[v]++;

        }
        for(int i = 0; i < n; i++) {
            if(deg[i] > 1) return i+1;
        }
        return -1;
    }
};
