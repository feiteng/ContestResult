class Solution {
public:
    int findCenter(vector<vector<int>>& edges) 
    {
        int n=edges.size()+1;
        vector<int>vec(n+1,0);
        for(auto i:edges)
        {
            vec[i[0]]++;
            vec[i[1]]++;
        }
        for(int i=1;i<vec.size();i++)
        {
            if(vec[i]==n-1)
                return i;
        }
        return -1;
    }
};
