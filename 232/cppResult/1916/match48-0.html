class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        unordered_map<int, int> freq;
        int max_freq = 0, ans = 0;
        
        for (auto &ed : edges) {
            freq[ed[0]]++;
            freq[ed[1]]++;
            
            if (freq[ed[0]] > max_freq) {
                max_freq = freq[ed[0]];
                ans = ed[0];
            }
            
            if (freq[ed[1]] > max_freq) {
                max_freq = freq[ed[1]];
                ans = ed[1];
            }
        }
        
        return ans;
    }
};
