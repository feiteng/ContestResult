class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        unordered_map<int, int> dict;
        for (int i = 0; i < edges.size(); ++i)
        {
            dict[ edges[i][0] ]++;
            dict[ edges[i][1] ]++;
        }
        
        for (auto& item : dict)
            if (item.second == edges.size()) return item.first;
        return false;
    }
};
