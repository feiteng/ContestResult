class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<int> indeg(n + 2);
        for (const auto& e: edges) {
            ++indeg[e[0]];
            ++indeg[e[1]];
        }
        for (int i = 1; i <= n + 1; ++i) {
            if (indeg[i] == n) {
                return i;
            }
        }
        return -1;
    }
};
