class Solution {
public:
    int findCenter(vector<vector<int>>& edges) {
        int n=edges.size()+1;
        vector<int> cnt(n,0);
        for(auto &v:edges){
            int a=v[0]-1;
            int b=v[1]-1;
            cnt[a]++;
            cnt[b]++;
        }
        for(int i=0;i<n;i++){
            if(cnt[i]==n-1) return i+1;
        }
        return 1;
    }
};
