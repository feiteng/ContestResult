struct node
{
    int p, t;
    double r;
    node(int _p, int _t, double _r){p = _p, t = _t, r = _r;}
};
struct cmp
{

    bool operator()(node a,node b) { return  a.r < b.r; }
};
priority_queue<node,vector<node>,cmp> q;

class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        for(auto &tmp : classes){
            double p = tmp[0], t = tmp[1];
            node x = node(tmp[0], tmp[1], (p + 1) / (t + 1) - p / t);
            q.push(x);
        }

        for(int i = 1; i <= extraStudents; i++){
            node x = q.top(); q.pop();
            x.p++; x.t++; x.r = 1.0 * (x.p + 1) / (x.t + 1) - 1.0 * x.p / x.t;
            q.push(x);
        }
        double ans = 0;
        while(!q.empty()){
            node x = q.top(); q.pop();
            ans +=  1.0 * x.p / x.t;
        }
        ans /= classes.size();
        return ans;
    }
};
