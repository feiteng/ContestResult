class Solution {
public:
    struct node{
        int a,b;

        friend bool operator<(node n1, node n2){
            return (n1.b+1)*1.0/(n1.a+1)-(n1.b)*1.0/(n1.a) < (n2.b+1)*1.0/(n2.a+1)-(n2.b)*1.0/(n2.a);
        }
    };
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<node> q;
        for (vector<int> c : classes) q.push((node){c[1], c[0]});

        while (extraStudents--){
            node now = q.top(); q.pop();
            //cout << now.b << " / " << now.a << endl;
            now.a += 1; now.b += 1;
            q.push(now);
        }
        double sum = 0.0;
        while(!q.empty()) {
            node now = q.top(); q.pop();
            sum += now.b*1.0/now.a;
        }
        return sum / classes.size();
    }
};
