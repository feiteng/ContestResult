class Solution {
public:
    typedef tuple<double,int> tp;
    double maxAverageRatio(vector<vector<int>>& c, int ex) {

        double dap,hap=0;
        
        priority_queue<tp> pq;
        for(int i=0;i<c.size();i++)
            if(c[i][0]!=c[i][1]) pq.emplace((double)(c[i][0]+1)/(double)(c[i][1]+1) - (double)(c[i][0])/(double)(c[i][1]),i);
        
        while(ex-- && !pq.empty()){
            double y;

            int x;
            tie(y,x) = pq.top(); pq.pop();
            c[x][0]++; c[x][1]++;
            pq.emplace((double)(c[x][0]+1)/(double)(c[x][1]+1) - (double)(c[x][0])/(double)(c[x][1]),x);
        }
        
        for(int i=0;i<c.size();i++)
            hap += (double)(c[i][0])/(double)(c[i][1]);
        
        return hap/(double)(c.size());
    }
};
