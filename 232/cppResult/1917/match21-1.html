class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pair<double, int>> s;

        int m = classes.size();
        for (int i = 0; i < m; ++i) {
            auto [u, v] = tie(classes[i][0], classes[i][1]);
            s.emplace(make_pair((1. * (u + 1) / (v + 1)) - (1. * u / v), i));
        }
        for (int i = 0; i < extraStudents; ++i) {
            auto [x, y] = s.top();
            s.pop();

            int& u = classes[y][0];
            int& v = classes[y][1];
            ++u;
            ++v;
            s.emplace(make_pair((1. * (u + 1) / (v + 1)) - (1. * u / v), y));
        }
        double ans = 0;
        for (int i = 0; i < m; ++i) {
            auto [u, v] = tie(classes[i][0], classes[i][1]);
            ans += 1. * u / v;
        }
        return ans / m;
    }
};
