typedef pair<double, double> pdd;
double val(pdd x){
	return (x.second - x.first)/(x.second*(x.second + 1));
}

struct cmp
{
    bool operator()(pdd u, pdd v){
		return val(u) < val(v);
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
		priority_queue<pdd, vector<pdd>, cmp> que;
		for (int i = 0; i < classes.size(); i++){
			que.push(pdd(classes[i][0], classes[i][1]));
		}
		while (extraStudents--){
			pdd t = que.top();
			que.pop();
            // cout << t.first << " " << t.second << endl;;
			t.first++;
			t.second++;
			que.push(t);
		}
		double ans = 0;
		while (!que.empty()){
			pdd t = que.top();
			que.pop();
			ans += t.first/t.second;
		}
		
		return ans/classes.size();
    }
};
