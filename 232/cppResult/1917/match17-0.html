class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        double sum = 0.0;
        int n = classes.size();
        
        priority_queue<pair<double, int> > q;
        
        double best = 0.0;
        for(int i=0;i<n;i++) 
            if (classes[i][0] != classes[i][1])
                q.push(make_pair( ( ((double)classes[i][0] + 1) / ((double)classes[i][1] + 1)) - ( ((double)classes[i][0] ) / ((double)classes[i][1])) , i));
        while (!q.empty() && extraStudents-- ) {
            int id = q.top().second;
            q.pop();
            classes[id][0]++;
            classes[id][1]++;
            q.push(make_pair(((double)(classes[id][0] + 1) / (double)(1+classes[id][1])) - ((double)(classes[id][0]) / (double)(classes[id][1])), id));
            
        }
        
        for(int i=0;i<n;i++) 
            sum += (double)classes[i][0] / (double)classes[i][1];
        
        return sum / (double)n;
    }
};
