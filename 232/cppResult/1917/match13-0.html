class Solution {
public:
    double up(int x, int y){
        return 1.0 * (x + 1) / (y + 1) - 1.0 * x / y;
    }
    
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pair<double, int> > q;
        int n = classes.size();
        for(int i = 0; i < n; i++){
            q.push(make_pair(up(classes[i][0], classes[i][1]), i));
        }
        while(extraStudents--){
            pair<double, int> tp = q.top();q.pop();
            int idx = tp.second;
            classes[idx][0]++;
            classes[idx][1]++;
            q.push(make_pair(up(classes[idx][0], classes[idx][1]), idx));
        }
        double ans = 0.0;
        for(int i = 0; i < n; i++) ans += 1.0 * classes[i][0] / classes[i][1] / n;
        return ans;
    }
};
