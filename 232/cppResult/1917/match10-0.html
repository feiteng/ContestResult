struct clas
{
    int p,t;
    double up;
    clas(int _p,int _t,double _up)
    {
        p = _p;
        t = _t;
        up = _up;
    }
    friend bool operator  < (clas a, clas b)
    {
        return a.up < b.up;
    }
};
class Solution {
public:
    double up_ca(int p,int t)
    {
        return 1.0*(p+1)/(t+1) - 1.0*p/t;
    }
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) 
    {
        int n = classes.size();
        priority_queue<clas> q;
        for(int i=0;i<n;i++)
            q.push(clas(classes[i][0],classes[i][1],up_ca(classes[i][0],classes[i][1])));
        for(int i=0;i<extraStudents;i++)
        {
            clas t = q.top();
            q.pop();
            t.p += 1;
            t.t += 1;
            t.up = up_ca(t.p,t.t);
            q.push(t);
        }
        double ans = 0;
        while(!q.empty())
        {
            clas t = q.top();
            q.pop();
            ans += 1.0*t.p/t.t;
        }
        ans /= n;
        return ans;
    }
};
