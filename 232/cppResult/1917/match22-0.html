class Solution {
public:
    struct node{
        double a;
        double b;
        friend bool operator < (node x, node y) {
            return (x.a + 1)/(x.b + 1) - x.a / x.b < (y.a + 1) / (y.b + 1) - y.a / y.b; 
        }
    }tmp;
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<node>pq;
        for (int i = 0; i < classes.size(); ++i) {
            tmp.a = classes[i][0];
            tmp.b = classes[i][1];
            pq.push(tmp);
        }
        for (int i = 0; i < extraStudents; ++i) {
            tmp = pq.top();
            ++tmp.a;
            ++tmp.b;
            pq.pop();
            pq.push(tmp);
        }
        double ans = 0;
        while (!pq.empty()) {
            tmp = pq.top();
            pq.pop();
            ans += tmp.a / tmp.b;
        }
        return ans / classes.size();
    }
};
