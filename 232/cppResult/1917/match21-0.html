class Solution {
public:
    priority_queue <pair<double,int>> pq;
    
    double func(int a,int b,int n){
        double s = 1.0*(a+1)/(b+1)/n;
        double s1 = 1.0*a/b/n;
        return s-s1;
    }
    
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int n = classes.size();
        for(int i=0;i<n;i++){
            auto& s = classes[i];
            pq.push(make_pair(func(s[0],s[1],n),i));
        }
        for(int i=0;i<extraStudents;i++){
            auto t = pq.top();pq.pop();
            //cout<<t.first<<" "<<t.second<<endl;
            int h = t.second;
            classes[t.second][0]++;
            classes[t.second][1]++;
            pq.push(make_pair(func(classes[h][0],classes[h][1],n),h));
        }
        
        double ans = 0;
        for(int i=0;i<n;i++){
            auto& s = classes[i];
            ans += 1.0*s[0]/s[1]/n;
        }
        return ans;
    }
};
