class Solution {
public:
    struct node{
        int a,b;
        int i;

        bool operator < (const node& y)const{
            return 1LL*(b-a)*y.b*(y.b+1)<1LL*(y.b-y.a)*b*(b+1);
        }
    }st[100005];
    
    priority_queue<node>que;
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int n=classes.size();
        for(int i=0;i<n;i++)st[i].a=classes[i][0],st[i].b=classes[i][1],st[i].i=i;
        for(int i=0;i<n;i++)que.push(st[i]);

        while(extraStudents--){
            node nn=que.top();que.pop();
            nn.a++,nn.b++;
            que.push(nn);
        }
        double ans=0;
        while(!que.empty()){
            node nn=que.top();que.pop();
            ans+=1.0*nn.a/nn.b;
        }
        return ans/n;
    }
};
