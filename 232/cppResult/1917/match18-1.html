class Solution {
public:
    struct node{
        double i,j;

        bool operator <(const node &a) const{
            return (i+1) / (j+1) - i/ j < (a.i + 1) / (a.j + 1) - a.i / a.j;
        }
    };
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue <node> q;
        int n = classes.size();
        for(auto classs : classes){
            node nex;
            nex.i = classs[0];

            nex.j = classs[1];
            q.push(nex);
        }
        for(int i = 1; i <= extraStudents; i++){
            node now = q.top();
            //cout<<now.i<<"   "<<now.j<<"    "<<endl;
            q.pop();
            node nex;
            nex.i =now.i+1;
            nex.j =now.j+1;
            q.push(nex);
           
        }
        double answer= 0;
        while(!q.empty()){
            node now = q.top();
            q.pop();
            answer += now.i / now.j;
        }
       return answer / n;
        
    }
};
