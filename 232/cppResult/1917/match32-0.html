struct nd{
    double inc;
    int a, b;
    bool operator<(const struct nd& now)const{
        return inc < now.inc;
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& cs, int es) {
        priority_queue<struct nd>pq;
        for(auto &c : cs){
            int ps = c[0], tt = c[1];
            double a = 1.0 * (ps + 1) / (tt + 1);
            double b = 1.0 * ps / tt;
            pq.push({a - b, ps, tt});
        }
        for(int i = 1; i <= es; ++i){
            auto tp = pq.top();
            pq.pop();
            tp.a += 1;
            tp.b += 1;
            tp.inc = (1.0*(tp.a+1)/(tp.b+1))-(1.0*tp.a/tp.b);
            pq.push(tp);
        }
        double ans = 0;
        while(!pq.empty()){
            auto tp = pq.top(); pq.pop();
            ans += 1.0 * tp.a/tp.b;
        }
        ans /= cs.size();
        return ans;
    }
};
