struct nod{
    int n;
    int pass;
    bool operator<(const nod& n1) const {

        return 1.0 * (pass + 1) / (n + 1) - 1.0 * pass / n < 1.0 * (n1.pass + 1) / (n1.n + 1) - 1.0 * n1.pass / n1.n;
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<nod> Q;
        for (auto it : classes) {
            Q.push({ it[1], it[0] });
        }
        for (int i = 0; i < extraStudents; i++) {
            auto it = Q.top();
            Q.pop();
            it.pass++;
            it.n++;
            Q.push(it);
        }

        double ret = 0;
        while (!Q.empty()) {
            auto it = Q.top();
            Q.pop();
            ret += 1.0*it.pass / it.n;

        }
        
        return ret / classes.size();
    }
};
