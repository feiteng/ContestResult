class Solution {
public:
    struct cls{
        double x;
        double y;
        cls(double x,double y):x(x),y(y){}
        bool operator < (const cls& other) const{
            return (x+1)/(y+1)-x/y<(other.x+1)/(other.y+1)-other.x/other.y; 
        }
    };
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<cls> pq;
        for(auto c:classes){
            pq.push(cls(double(c[0]),double(c[1])));
        }
        for(int i=0;i<extraStudents;i++){
            auto c=pq.top(); 
            pq.pop();
            c.x++;
            c.y++;
            pq.push(c);
        }
        double tol=0;
        while(!pq.empty()){
            auto c=pq.top();
            pq.pop();
            tol+=c.x/c.y;
        }
        return tol/classes.size();
    }
};
