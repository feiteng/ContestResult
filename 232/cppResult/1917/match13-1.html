class Solution {
public:
    double gain(int x, int y){
        return 1. * (x + 1) / (y + 1) - 1. * x / y;
    }
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pair<double,int>> que;
        int n = classes.size();
        for(int i = 1; i <= n; i++) que.push(make_pair(gain(classes[i-1][0],classes[i-1][1]),i));
        for(int i = 1; i <= extraStudents; i++){

            auto p = que.top();
            que.pop();
            int id = p.second;
            classes[id-1][0]++;
            classes[id-1][1]++;
            que.push(make_pair(gain(classes[id-1][0],classes[id-1][1]),id));
        }
        double ret = 0;
        for(int i = 0; i < n; i++) ret += 1. * classes[i][0] / classes[i][1];
        ret /= n;
        return ret;
    }
};
