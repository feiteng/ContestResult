class Ele {
public:
    int x, y;
    Ele(int _x, int _y): x(_x), y(_y) {}

    bool operator < (const Ele& other) const {
        return 1.0 * (other.x + 1) / (other.y + 1) - 1.0 * other.x / other.y >
            1.0 * (x + 1) / (y + 1) - 1.0 * x / y;
    }
};

class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<Ele> que;
        for (auto& c: classes) que.emplace(c[0], c[1]);

        while (extraStudents) {
            auto cur = que.top(); que.pop();
            cur.x += 1;
            cur.y += 1;
            que.push(cur);
            extraStudents -= 1;
        }
        double res = 0;
        while (que.size()) {
            auto cur = que.top(); que.pop();
            res += 1.0 * cur.x / cur.y;
        }
        res /= classes.size();
        return res;
    }
};
