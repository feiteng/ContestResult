class Solution {
public:
    class cls {
    public:
        int a,b;
        bool operator < (const cls & rhs) const {
            return 1LL*(a-b)*rhs.a*(rhs.a+1) < 1LL*(rhs.a-rhs.b)*a*(a+1);
        }
    };
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<cls> que;
        for(auto & c : classes) {
            que.push(cls({c[1],c[0]}));
        }
        while(extraStudents--) {
            cls t = que.top();
            que.pop();
            t.a++;
            t.b++;
            que.push(t);
        }
        double ans = 0;
        while(!que.empty()) {
            cls t = que.top(); que.pop();
            ans += 1.0*t.b/t.a;
        }
        ans /= classes.size();
        return ans;
    }
};
