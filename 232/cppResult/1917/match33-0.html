struct Obj{
    double x,y;
    bool operator<(const Obj &obj)  const{
        return ((x+1)/(y+1)-x/y)<((obj.x+1)/(obj.y+1)-(obj.x)/(obj.y));
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int ex) {
        int n=classes.size();
        priority_queue<Obj,vector<Obj>>heap;
        for(auto &c:classes){
            double a=c[0],b=c[1];
            heap.push({a,b});
        }
        while(ex){
            auto t=heap.top();
            heap.pop();
            double x=t.x;
            double y=t.y;
            heap.push({x+1,y+1});
            ex--;
        }
        double ans=0;
        while(heap.size()){
            auto t=heap.top();
            heap.pop();
            ans+=t.x/t.y;
        }
        return ans/n;
    }
};
