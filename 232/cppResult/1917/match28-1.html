class Solution {
public:
    struct Node{
        int a,b;

        bool operator < (const Node &t) const{
            return (a+1.0)/(b+1.0)-a*1.0/b<((t.a+1.0)/(t.b+1.0)-t.a*1.0/t.b);
            //return b>t.b;
        }
    };
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<Node> pq;
        int cnt=0;
        for(auto item:classes){
            if(item[0]!=item[1])
                pq.push(Node{item[0],item[1]});
            else
                cnt++;
        }
        if(pq.size()==0){
            return 1.0;

        }
        for(int i=1;i<=extraStudents;i++){
            Node tmp = pq.top();
            pq.pop();
            pq.push(Node{tmp.a+1,tmp.b+1});
        }
        double ans=cnt;
        while(!pq.empty()){
            Node tmp = pq.top();
            pq.pop();
            ans+=tmp.a*1.0/tmp.b;
        }
        return ans/classes.size();
    }
};
