class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        set<pair<double,int>> s;
        for (int i=0;i<classes.size();i++){
            int a=classes[i][0];
            int b=classes[i][1];
            s.insert({-1*(1.0*b-a)/(1.0*b*(1.0*b+1)),i});
        }
        for (int i=1;i<=extraStudents;i++){
            auto it = s.begin();
            int pos = it->second;
            s.erase(it);
            classes[pos][0]++;
            classes[pos][1]++;
            int a=classes[pos][0];

            int b=classes[pos][1];
            s.insert({-1*(1.0*b-a)/(1.0*b*(1.0*b+1)),pos});
        }
        double res=0;
        for (auto &it:classes){
            res +=(1.0*it[0]/(1.0*it[1]));
        }
        return res/classes.size();
    }
};
