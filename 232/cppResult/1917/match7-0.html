struct cla {
    double p, t;
    cla(double p, double t) : p(p), t(t) {}
    bool operator<(const cla& b) const {
        return (p + 1.0) / (t + 1.0) - p / t < (b.p + 1.0) / (b.t + 1.0) - b.p / b.t;
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<cla> Q;
        for (auto c : classes) {
            Q.push(cla(c[0], c[1]));
        }
        for (int i = 0; i < extraStudents; ++i) {
            auto q0 = Q.top();
            Q.pop();
            Q.push({q0.p + 1, q0.t + 1});
        }
        double ret = 0;
        while (!Q.empty()) {
            auto q0 = Q.top();
            Q.pop();
            ret += q0.p / q0.t;
        }
        return ret / classes.size();
    }
};
