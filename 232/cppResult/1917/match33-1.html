#define dii tuple<double,int,int> 


class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int n=classes.size(); 
        priority_queue<dii>  pq; 
        for(auto i:classes){
            int x = i[0];
            int y= i[1]; 
            pq.push({1.0*(x+1)/(y+1)-1.0*(x)/(y),x,y});  
        }
        
        while(extraStudents>0){
            auto[z,x,y] = pq.top(); 
            pq.pop();
            x++;

            y++; 
            pq.push({1.0*(x+1)/(y+1)-1.0*(x)/(y),x,y});  
            extraStudents--; 
        }
        
        double res=0;
        while(pq.size()){
            auto[x,y,z] = pq.top(); 
            pq.pop();
            res+=1.0*(y)/(z);
        }
        
        return res/n; 
    }
};
