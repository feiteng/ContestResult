struct node
{
    int x,y;
    bool operator < (const node &a) const
    {
        return (x*1.0/y)-((x+1)*1.0/(y+1))>(a.x*1.0/a.y)-((a.x+1)*1.0/(a.y+1));
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<node> q;
        while(q.size()!=0) q.pop();
        for(auto cur:classes)
        {
            q.push({cur[0],cur[1]});
        }
        for(int i=1;i<=extraStudents;i++)
        {
            node top1,tmp1,top2,tmp2;
            top1 = q.top();q.pop();
            tmp1 = top1;
            tmp1.x+=1,tmp1.y+=1;
            q.push(tmp1);
            // if(classes.size() >= 2)
            // {
            //     top2 = q.top();q.pop();
            //     tmp2 = top2;
            //     tmp2.x+=1,tmp2.y+=1;
            // }
            // if(classes.size() == 1) q.push(tmp1);
            // else
            // {
            //     double sum1=(tmp1.x*1.0/tmp1.y)-(top1.x*1.0/top1.y);
            //     double sum2=(tmp2.x*1.0/tmp2.y)-(top2.x*1.0/top2.y);
            //     cout<<sum1<<" "<<sum2<<endl;
            //     if(sum1 > sum2) q.push(tmp1),q.push(top2);
            //     else q.push(tmp2),q.push(top1);
            // }
        }
        double ans=0;
        while(q.size()!=0)
        {
            auto top=q.top();q.pop();
            cout<<top.x<<" "<<top.y<<endl;
            ans+=(top.x*1.0/top.y);
        }
        return ans/classes.size();
    }
};
