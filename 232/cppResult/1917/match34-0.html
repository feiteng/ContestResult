struct p {
    double num, den;

    p(int a, int b) : num(a), den(b) {};

    bool operator<(const p& rhs) const {
        double ldelta = ((1 + num)/(1 + den)) - (num/den);
        double rdelta = ((1 + rhs.num)/(1 + rhs.den)) - (rhs.num/rhs.den);
        return ldelta < rdelta;
    }
    double avg() {
        return num/den;
    }
};

class Solution {
public:
    
    double maxAverageRatio(vector<vector<int>>& classes, int ex) {
        int nclasses = classes.size();
        priority_queue<p> pq;
        for (auto& c : classes) {
            pq.emplace(c[0], c[1]);
        }

        int i = 0;
        while (ex) {
            auto pi = pq.top(); pq.pop();
            pi.num++, pi.den++;
            pq.emplace(pi);
            ex--;
        }

        double avg = 0;
        while (pq.size()) {
            auto pi = pq.top(); pq.pop();
            avg += pi.avg();
        }
        return avg / nclasses;

    }
};
