class Solution {
public:
    struct Node{
        int a,b;
        Node(int aa=0,int bb=0):a(aa),b(bb){}
        bool operator<(const Node& e)const{
            return 1LL*(a-b)*(1LL*e.a*(e.a+1))<1LL*a*(a+1)*(e.a-e.b);
        }
    };

    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<Node> que;
        int n=classes.size();
        for(int i=0;i<n;++i){
            int b=classes[i][0];
            int a=classes[i][1];
            que.push(Node(a,b));
        }
        for(int i=0;i<extraStudents;++i){
            Node nd=que.top();
            que.pop();
            nd.a+=1;
            nd.b+=1;
            que.push(nd);
        }
        double ans=0;
        while(!que.empty()){
            Node nd=que.top();
            que.pop();
            ans+=1.0*nd.b/nd.a;
        }
        return ans/n;
    }
};
