class Solution {
public:
    struct Node{
        double a, b;
        double rate;
        bool operator<(const Node& t)const{  
            return (a+1) / (b+1) - rate < (t.a+1) / (t.b+1) - t.rate;
        }  
    };
    
    double maxAverageRatio(vector<vector<int>>& cs, int es) {
        int n = cs.size();
        priority_queue<Node>q;
        for(auto &it : cs){
            q.push({(double)it[0], (double)it[1], it[0] * 1.0 / it[1]});
        }
        for(int i = 0; i < es; i++){
            Node k = q.top();
            q.pop();
          //  cout << k.a << ' ' << k.b << endl;
            q.push({k.a+1,k.b+1, (k.a+1)*1.0/(k.b+1)});
        }
        double ret = 0;
        while(!q.empty()){
            ret += q.top().rate;
            q.pop();
        }
        return ret / n;
    }
};
