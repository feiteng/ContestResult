class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        auto cmp = [](pair<int, int>& lhs, pair<int, int>& rhs) {
            return 1.0 * lhs.first / lhs.second - 1.0 * (lhs.first + 1) / (lhs.second + 1) > 
                1.0 * rhs.first / rhs.second - 1.0 * (rhs.first + 1) / (rhs.second + 1);
        };
        priority_queue<pair<int, int>, vector<pair<int, int>>, decltype(cmp)> pq(cmp);
        for (auto& c : classes) pq.push({c[0], c[1]});
        for (int i = 0; i < extraStudents; i++) {
            auto [p, t] = pq.top();
            //cout<<p <<" "<<t<<endl;
            pq.pop();
            pq.push({p + 1, t + 1});

        }
        double res = 0;
        while (!pq.empty()) {
            auto [p, t] = pq.top();
            res += 1.0 * p / t;
            pq.pop();
        }
        return res / classes.size();
    }
};
