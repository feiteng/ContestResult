struct Node{
    double x, y;
    Node(int a, int b):x(a),y(b){}
};


bool operator<(const Node& a, const Node& b){
    double i = (a.y - a.x) / (a.y * (a.y + 1));
    double j = (b.y - b.x) / (b.y * (b.y + 1));
    return i < j;
}

class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int n = classes.size();
        priority_queue<Node> pq;
        for (auto& cls: classes) {
            pq.push(Node(cls[0], cls[1]));
        }
        while (extraStudents) {

            Node t = pq.top();
            pq.pop();
            pq.push(Node(t.x + 1, t.y + 1));
            extraStudents --;
        }
        double ret = 0;
        while (pq.size()) {
            Node t = pq.top();
            pq.pop();
            ret += t.x / t.y;
        }
        return ret / n;
    }
};
