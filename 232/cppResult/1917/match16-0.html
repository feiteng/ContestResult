struct node {
    int p, t;
    node(int p, int t) : p(p), t(t) {}
};

struct cmp {
    bool operator()(node i, node j) {
        double da = (double)(i.p + 1) / (i.t + 1) + (double)j.p / j.t;
        double db = (double)(j.p + 1) / (j.t + 1) + (double)i.p / i.t;
        return da < db;
    }
};


class Solution {
public:
    double maxAverageRatio(vector<vector<int>> &cs, int E) {
        priority_queue<node, vector<node>, cmp> pq;
        for (auto c:cs) {
            pq.push(node(c[0], c[1]));
        }
        for (int i = 0; i < E; i++) {
            auto nc = pq.top();
            pq.pop();
//            printf("p%d,t%d\n", nc.p, nc.t);
            nc.p+=1, nc.t+=1;
            pq.push(nc);
        }
        double res = 0;
        while (pq.size()) {
            auto nc = pq.top();
            pq.pop();
            res += double(nc.p) / nc.t;
        }
        double n=cs.size();
        return res / n;
    }
};
