typedef pair<double, double> pii;

struct cmp
{

  bool operator()(pii &l, pii &r) {
    return (l.first+1)/(l.second+1)-l.first/l.second < 
         (r.first+1)/(r.second+1)-r.first/r.second ;
  }
};

class Solution {
public:
  double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
    int N = classes.size();
    priority_queue<pii, vector<pii>,cmp> Q;
    for (auto c: classes) Q.push({c[0], c[1]});
    while (extraStudents--) {
      auto [x, y] = Q.top(); Q.pop();
  //    cout << x << " " << y << endl;
      Q.push({x+1, y+1});

    }
    double tot = 0;
    while (!Q.empty()) { 
      auto [x, y] = Q.top(); Q.pop();
      tot += x/y;
    }
      
    return tot/N;
  }
};
