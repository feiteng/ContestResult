class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pair<double, int>, vector<pair<double, int>>, less<pair<double, int>>> pq;
        int len = classes.size();
        for(int i=0;i<len;i++) {
            int u = classes[i][0];
            int v = classes[i][1];
            pq.push(make_pair((u*1.0+1)/(v*1.0+1) - u*1.0 / v,i));
        }
        for(int i=0;i<extraStudents;i++) {
            pair<double, int> pp = pq.top();
            pq.pop();
            // cout << pp.second << " " << pp.first << endl;
            classes[pp.second][0]++, classes[pp.second][1]++;
            pq.push(make_pair((classes[pp.second][0]*1.0+1)/(classes[pp.second][1]*1.0+1) - classes[pp.second][0]*1.0 / classes[pp.second][1], pp.second));
        }
        double ans = 0.0;
        for(int i=0;i<len;i++) {
            // cout << i << " : " << classes[i][0] << " " << classes[i][1] << endl;
            ans += 1.0 * classes[i][0] / classes[i][1];
        }
        // cout << ans << endl;
        return ans / len;
    }
};
