class Node {
public:
    double x;
    double y;
    Node(double a, double b) : x(a), y(b) {}
};

bool compare(const Node& a, const Node& b) {
    double diffa = (a.x+1)/(a.y+1) - a.x/a.y;
    double diffb = (b.x+1)/(b.y+1) - b.x/b.y;
    return diffa < diffb;
}
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int len = classes.size();
        priority_queue<Node, vector<Node>, decltype(&compare)> heap(compare);

        for (vector<int>& v : classes) {
            heap.push(Node(v[0], v[1]));
        }

        for (int i = 0; i < extraStudents; i++) {
            Node c = heap.top();
            heap.pop();
            c.x++;
            c.y++;
            heap.push(c);
        }

        double sum = 0;
        while (!heap.empty()) {
            Node p = heap.top();
            heap.pop();
            sum += p.x/p.y;
        }
        return sum/len;
    }
};
