class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int es) {
        auto comp = [](pair<int,int>& a, pair<int,int>& b) {
            return (a.second - a.first) * 1.0/(a.second + 1)/a.second < (b.second - b.first) * 1.0/(b.second + 1)/b.second;
        };
        priority_queue<pair<int,int>, vector<pair<int,int>>, decltype(comp)> pq(comp);
        
        for(auto& x : classes) {
            pq.push({x[0], x[1]});
        }
        
        while(es > 0) {
            auto cur = pq.top();
            pq.pop();
            // cout << "picked " << cur.first << "->" << cur.second << endl;
            pq.push({cur.first +1, cur.second + 1});
            es--;
        }
        
        double res = 0;
        while(!pq.empty()) {
            auto cur = pq.top();
            pq.pop();
            res += cur.first *1.0/cur.second;
        }
        
        return res / classes.size();
        
    }
};
