struct node{
	int k,n;
    node(int a, int b) : k(a), n(b){};
    friend bool operator<(node a,node b)
	{

        return 1.0 * (1.0*(a.k + 1) * a.n - 1.0*a.k * (a.n + 1)) / (1.0*(a.n + 1) * a.n) <
               1.0 * (1.0*(b.k + 1) * b.n - 1.0*b.k * (b.n + 1)) / (1.0*(b.n + 1) * b.n);
    }
};
class Solution {
public:

    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<node> q;
        for (auto i:classes)
        {
            node a(i[0], i[1]);
            q.push(a);

        }
        while(extraStudents--)
        {
            node t = q.top();
            q.pop();
            t.k++, t.n++;
            q.push(t);
        }
        double ans = 0;
        while(!q.empty())
        {
            node t = q.top();
            q.pop();
            //cout<<t.k<<' '<<t.n<<endl;
            ans += 1.0*t.k / t.n;
        }//cout<<endl;
        return ans/classes.size();
    }
};
