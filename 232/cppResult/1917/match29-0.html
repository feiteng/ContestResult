class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        // The average pass ration increase only determined by the people in the smallest class
        priority_queue<pair<double, int>> pq;
        for(int i=0; i<classes.size(); i++){
            pq.push({(classes[i][0]+1)*1.0/(classes[i][1]+1)-classes[i][0]*1.0/classes[i][1], i});
        }
        pair<double, int> cur;
        int diff;
        while(extraStudents>0){
            cur=pq.top();
            pq.pop();
            classes[cur.second][0]+=1;
            classes[cur.second][1]+=1;
            extraStudents-=1;
            pq.push({(classes[cur.second][0]+1)*1.0/(classes[cur.second][1]+1)-classes[cur.second][0]*1.0/classes[cur.second][1], cur.second});
        }
        double ret=0.0;
        for(auto &e : classes){
            ret+=(e[0]*1.0)/e[1];
        }
        return ret/(classes.size());
    }
};
