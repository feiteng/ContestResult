struct PassRatio {
    int pass;
    int total;

    bool operator<(const PassRatio& other) const {
        double self_gain = (double)(pass+1)/(total+1) - (double)pass/total;
        double other_gain = (double)(other.pass+1)/(other.total+1) - (double)other.pass/other.total;
        return  self_gain < other_gain; 
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<PassRatio> pq;
        for (auto& v : classes) {
            pq.push({v[0], v[1]});

        }
        for (int i = 0; i < extraStudents; ++i) {
            auto cur = pq.top(); pq.pop();
            ++cur.pass;
            ++cur.total;
            pq.push(cur);
        }
        double ans = 0;
        while (!pq.empty()) {
            auto cur = pq.top(); pq.pop();
            ans += (double)cur.pass/cur.total;
        }
        return ans/(classes.size());
    }
};
