using pdi=pair<double, int>;
struct Cmp {
    bool operator()(const pdi& lhs, const pdi& rhs) {
        double delta1=(lhs.first*lhs.second+1)/(lhs.second+1)-lhs.first;        
        double delta2=(rhs.first*rhs.second+1)/(rhs.second+1)-rhs.first;
        return delta1<delta2;
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int exSts) {
        int n=classes.size();
        priority_queue<pdi, vector<pdi>, Cmp> pq;
        for(int i=0; i<n; ++i) {
            pq.emplace((double)classes[i][0]/classes[i][1], classes[i][1]);
        }
        while(exSts--) {
            auto [avg, tot] = pq.top();
            pq.pop();
            avg=(avg*tot+1)/(tot+1);
            pq.emplace(avg, tot+1);
        }
        double res=0;
        while(!pq.empty()) {
            auto [avg, tot]=pq.top();
            // printf("avg=%lf, tot=%d\n", avg, tot);
            pq.pop();
            res+=avg;
        }
        return res/n;
    }
};
