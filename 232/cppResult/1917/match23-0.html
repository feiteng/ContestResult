using pii = pair<int, int>;

static double getAverage(int upper, int lower) {
    return ((double) upper) / lower;
}

static double diffPlusOne(const pii& p) {
    return getAverage(p.first+1, p.second+1) - getAverage(p.first, p.second);
}

class comp {
public:
    bool operator()(const pii& lhs, const pii& rhs) {
        return diffPlusOne(lhs) < diffPlusOne(rhs);
    }
};

class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pii, vector<pii>, comp> pq;
        for (vector<int>& cls : classes) {
            pq.push({cls[0], cls[1]});
        }
        for (int i=0; i<extraStudents; i++) {
            pii p = pq.top(); pq.pop();
            p.first++; p.second++;
            pq.push(p);
        }
        
        int size = pq.size();
        double sum = 0.0;
        while (!pq.empty()) {
            pii p = pq.top(); pq.pop();
            sum += getAverage(p.first, p.second);
        }
        
        return sum / size;
    }
};
