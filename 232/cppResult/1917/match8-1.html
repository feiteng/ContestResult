100_PiggyHead.cpp

struct Node {
    int a, b;
    bool operator <(const Node& rhs) const {
        return 1ll * (b - a) * rhs.b * (rhs.b + 1) <
               1ll * (rhs.b - rhs.a) * b * (b + 1);
    }
};

priority_queue<Node> que;

class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        for (auto& c: classes) {
            que.push((Node){c[0], c[1]});
        }
        for (int i = 0; i < extraStudents; i ++) {
            auto q = que.top(); que.pop();
            que.push((Node){q.a + 1, q.b + 1});

        }
        double ans = 0;
        while (!que.empty()) {
            auto q = que.top(); que.pop();
            ans += 1.0 * q.a / q.b;
        }
        return ans / classes.size();
    }
};
