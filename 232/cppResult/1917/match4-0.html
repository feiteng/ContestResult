struct Node{
    double a,b;
    bool operator <(const Node & x) const
    {
        return (b-a)/(b*(b+1)) < (x.b-x.a)/(x.b*(x.b+1));
    }
    Node(double x, double y): a(x),b(y) {}
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<Node,vector<Node>,less<Node>> q;
        for(auto &i:classes)
            q.push({(double)i[0],(double)i[1]});
        for(int i=0;i<extraStudents;++i)
        {
            auto t=q.top();
            q.pop();
            t.a+=1,t.b+=1;
            q.push(t);
        }
        double res=0.0;
        while(q.size())
        {
            auto t=q.top();
            q.pop();
            res += t.a / t.b;
        }
        res /= (double) classes.size();
        return res;
        
    }
};
