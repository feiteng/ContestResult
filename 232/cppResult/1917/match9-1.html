class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<pair<double, int>, vector<pair<double,int>>,less<pair<double,int>>> pq;
        int N = classes.size();
        for (int i = 0; i < N; ++i) {
            pq.push(make_pair(((classes[i][0]+1) * 1.0 / (1.0 * (classes[i][1]+1))) - (classes[i][0] * 1.0 / (1.0*classes[i][1])), i));
        }
        while (extraStudents > 0) {
            int i = pq.top().second;
            pq.pop();
            //cout << i << endl;

            --extraStudents;
            ++classes[i][0];
            ++classes[i][1];
            pq.push(make_pair(((classes[i][0] + 1) * 1.0 / (1.0 * (classes[i][1] + 1))) - (classes[i][0] * 1.0 / (1.0 * classes[i][1])), i));
        }
        double res = 0;
        for (int i = 0; i < N; ++i) {
            res += (classes[i][0] * 1.0) / (classes[i][1] * 1.0);
        }
        return res/N;
    }
};
