struct node {
    int pass, total;

    bool operator < (const node& a) const {
        double add1 = 1.0 * (pass + 1) / (total + 1) - 1.0 * pass / total;
        double add2 = 1.0 * (a.pass + 1) / (a.total + 1) - 1.0 * a.pass / a.total;                             
        return add1 < add2;
    }
};



class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        int n = classes.size();
        priority_queue<node> q;
        for(auto cls : classes) {
            q.push({cls[0], cls[1]});
        } 
        while(extraStudents--) {
            auto t = q.top();
            q.pop();
            // cout << t.pass << ' ' << t.total << endl;

            t.pass++;
            t.total++;
            q.push(t);
        }
        double pass = 0.0;
        while(q.size()) {
            auto t = q.top();
            q.pop();
            pass += 1.0 * t.pass / t.total;
        }
        return pass / n;
    }
};
