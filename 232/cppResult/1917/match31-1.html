class Classroom{
public:
    long long x;
    long long y;
    Classroom(int nx, int ny){
        x = nx;
        y = ny;
    }
};
struct Compare{

    bool operator()(const Classroom &a, const Classroom &b){
        return (a.y-a.x)*(b.y*(b.y+1)) < (b.y-b.x)*(a.y*(a.y+1)); 
    }
};
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        priority_queue<Classroom, vector<Classroom>, Compare> pq;
        for(int i=0; i<classes.size(); i++){
            pq.push(Classroom(classes[i][0], classes[i][1]));
        }
        while(extraStudents--){
            Classroom cr = pq.top();
            pq.pop();
            cr.x++;
            cr.y++;
            pq.push(cr);
        }
        
        double avg = 0;
        while(!pq.empty()){
            Classroom cr = pq.top();
            pq.pop();
            avg = avg + ((double)cr.x)/cr.y;
        }
        
        avg/=classes.size();
        
        return avg;
    }
};
