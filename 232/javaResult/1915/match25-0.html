class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int a = 0, b = 0, count = 0;
        for(int i = 0; i < s1.length(); i++) {
            if(s1.charAt(i) != s2.charAt(i)) {
                if(count == 0) {
                    a = i;
                }
                else if(count == 1) {
                    b = i;
                }
                else {
                    return false;
                }
                count++;
            }
        }
        if(count == 0) return true;
        if(count !=2) return false;
        if(s1.charAt(a) != s2.charAt(b) || s1.charAt(b) != s2.charAt(a))
            return false;
        return true;
    }
}
