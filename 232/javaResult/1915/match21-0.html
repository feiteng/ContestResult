class Solution {

	public boolean areAlmostEqual(String s1, String s2) {
		ArrayList<Integer> list = new ArrayList<>();
		for (int i = 0; i < s1.length(); i++) {
			if (s1.charAt(i) != s2.charAt(i)) {
				list.add(i);
			}
		}
		return list.isEmpty() || list.size() < 3 && s1.charAt(list.get(0)) == s2.charAt(list.get(list.size() - 1))
				&& s2.charAt(list.get(0)) == s1.charAt(list.get(list.size() - 1));
	}
}
