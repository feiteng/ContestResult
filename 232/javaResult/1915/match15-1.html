class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        
int j = -1;
        int k = -1;
        for (int m=0;m<s1.length();m++) {
            if (s1.charAt(m) != s2.charAt(m)) {
                if (j==-1) {j=m;}
                else if (k==-1) {k=m;}
                else {return false;}
            }
        }
        if (j==-1) {return true;}
        if (k==-1) {return false;}
        if (
s1.charAt(j) == s2.charAt(k) && s1.charAt(k) == s2.charAt(j)) {return true;}
        return false;
    }
}
