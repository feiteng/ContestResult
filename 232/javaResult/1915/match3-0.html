class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int cnt = 0;
        int n = s1.length();
        int[] m1 = new int[26], m2 = new int[26];
        for (int i = 0; i < n; i++) {
            m1[s1.charAt(i) - 'a']++;
            m2[s2.charAt(i) - 'a']++;
            if (s1.charAt(i)!=s2.charAt(i)) {
                cnt++;
            }
        }
        for (int i = 0; i < 26; i++) {
            if (m1[i] != m2[i]) return false;
        }
        return cnt <= 2;
    }
}
