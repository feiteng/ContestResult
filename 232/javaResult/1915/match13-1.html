class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int cnt = 0, x1 = -1, x2 = -1;
        for (int i = 0; i < s1.length(); ++i) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (++cnt == 1) {
                    x1 = i;
                } else {
                    x2 = i;
                
}
            }
        }
        return cnt == 0 || (cnt == 2 && s1.charAt(x1) == s2.charAt(x2) && s1.charAt(x2) == s2.charAt(x1));
    }
}
