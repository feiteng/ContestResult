class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int delta = 0;
        int low = -1;
        int high = -1;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                delta++;
                if (delta == 1) {
                    low = i;
                } else if (delta == 2) {
                    high = i;
                } else {
                    return false;
                }
            }
        }
        if (delta == 0) {
            return true;
        }
        if (delta == 1) {
            return false;
        }
        if (s1.charAt(low) == s2.charAt(high) && s1.charAt(high) == s2.charAt(low)) {
            return true;
        }
        return false;
    }
}
