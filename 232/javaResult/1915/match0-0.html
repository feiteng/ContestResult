class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int [] cnt = new int[26]; 
        int diff = 0;        
        for (int i = 0; i < s1.length(); i++){
            if (s1.charAt(i) != s2.charAt(i)){
                diff ++;
            }
            cnt[s1.charAt(i) - 'a']++;
            cnt[s2.charAt(i) - 'a']--;
        }
        
        for (int i = 0; i < 26; i++) {
            if (cnt[i] != 0){
                return false;
            }
        }
        return diff <= 2;
    }
}
