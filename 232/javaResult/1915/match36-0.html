class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        if (s1.equals(s2))
            return true;
        char[] arr = s1.toCharArray();
        for (int j = 0; j < s1.length(); j++) {
            for (int k = j + 1; k < s1.length(); k++) {
                char jChar = s1.charAt(j);
                char kChar = s1.charAt(k);
                arr[j] = kChar;
                arr[k] = jChar;
                if ((new String(arr).equals(s2)))
                    return true;
                arr[j] = jChar;
                arr[k] = kChar;
            }
        }
        return false;
    }
}
