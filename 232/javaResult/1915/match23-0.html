class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        List<Integer> index = new ArrayList<>();
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                index.add(i);
            }
        }
        if (index.size() == 0) {
            return true;
        } else if (index.size() == 2 && s1.charAt(index.get(0)) == s2.charAt(index.get(1)) && s1.charAt(index.get(1)) == s2.charAt(index.get(0))) {
            return true;
        } else {
            return false;
        }
    }
}
