class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        if (s1.equals(s2)) {
            return true;
        }
        for (int i = 0; i < s1.length(); i++) {
            for (int j = i+1; j < s1.length(); j++) {
                String s11 = s1.substring(0, i) + s1.charAt(j) + s1.substring(i+1, j) + s1.charAt(i) + s1.substring(j+1);
                
if (s11.equals(s2)) {
                    return true;
                }
            }
        }
        return false;
    }
}
