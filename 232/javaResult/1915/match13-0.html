class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int count = 0;
        int i1 = 0, i2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (count == 2) return false;
                if (count == 0) i1 = i;
                if (count == 1) i2 = i;
                count++;
            }
        }
        return s1.charAt(i1) == s2.charAt(i2) && s1.charAt(i2) == s2.charAt(i1);
    }
}
