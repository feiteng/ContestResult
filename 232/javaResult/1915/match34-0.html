class Solution {
public boolean areAlmostEqual(String s1, String s2) {
        if (s1.equals(s2)) {
            return true;
        }
        for (int i = 0; i < s1.length() - 1; i++) {
            for (int j = i + 1; j < s1.length(); j++) {
                String temp = swap(s2,i ,j);
                if (s1.equals(temp)) {
                    return true;
                }
            }
        }
        return false;
    }

    private String swap(String s2, int i, int j) {
        StringBuilder strB = new StringBuilder(s2);
        strB.setCharAt(i, s2.charAt(j));
        strB.setCharAt(j, s2.charAt(i));
        
        return strB.toString();
    }
}
