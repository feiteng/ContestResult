class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        if (s1.equals(s2)) return true;
        char [] S = s1.toCharArray();
        for (int i = 0; i<S.length; ++i) {
            for (int j = i+1; j<S.length; ++j) {
                char tmp = S[i];
                
S[i] = S[j];
                S[j] = tmp;
                
                if (s2.equals(new String(S))) return true;
                
                S[j] = S[i];
                S[i] = tmp;
                
            }
        }
        return false;
        
    }
}
