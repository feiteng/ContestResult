class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int diff = 0;
        
        int i1 = -1, i2 = -1;
        
        for(int i = 0 ; i < s1.length() ; i++) {
            if(s1.charAt(i) != s2.charAt(i)) {                
                diff++;
                
                if( diff > 2 ) {
                    return false;
                }
                
                if( i1 < 0 ) {
                    i1 = i;
                } else {
                    i2 = i;
                }
            }
        }
        
        
        
        return diff == 0 || (diff == 2 && s1.charAt(i1) == s2.charAt(i2) && s1.charAt(i2) == s2.charAt(i1));
    }
}
