class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int p1 = -1;
        int p2 = -1;
        for (int i = 0; i < s1.length(); i ++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (p1 == -1) {
                    p1 = i;
                } else if (p2 == -1) {
                    p2 = i;
                } else {
                    return false;
                }
            }
        }
        if (p1 == -1) {
            return true;
        }
        
        if (p2 == -1) {
            return false;
        }
        
        return s1.charAt(p1) == s2.charAt(p2) && s1.charAt(p2) == s2.charAt(p1);
    }
}
