class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int cnt = 0;
        int index1 = 0, index2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                cnt++;
                if (cnt == 1) {
                    index1 = i;
                } else if (cnt == 2) {
                    index2 = i;
                } else {
                    return false;
                }
            }
        }
        if (cnt == 0) {
            return true;
        }
        if (cnt == 1) {
            return false;
        }
        if (s1.charAt(index1) == s2.charAt(index2) && s1.charAt(index2) == s2.charAt(index1)) {
            return true;
        } else {
            return false;
        }
    }
}
