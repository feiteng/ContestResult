class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        TreeSet<Integer> diff = new TreeSet<>();
        int n = s1.length();
        for(int i=0; i<n; i++){
            if(s1.charAt(i) != s2.charAt(i)){
                diff.add(i);
            }
        }
        
        return diff.size()==0 ||
            (diff.size()==2 && s1.charAt(diff.first()) == s2.charAt(diff.last()) && s2.charAt(diff.first()) == s1.charAt(diff.last()));
    }
}
