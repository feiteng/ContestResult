class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int cnt = 0;
        int[] char1 = new int[26];
        int[] char2 = new int[26];
        for(int i = 0; i < s1.length(); i++){
            char1[s1.charAt(i)-'a']++;
            char2[s2.charAt(i)-'a']++;
            
            if(s1.charAt(i)!=s2.charAt(i)){
                cnt++;
            }
        }
        if(cnt>2) return false;
        
for(int i =0; i < 26; i++){
            if(char1[i] != char2[i]) return false;
        }
        return true;
    }
}
