class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        int[] cm = new int[26];
        int count = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                count++;
            }
            cm[s1.charAt(i) - 'a']++;
            cm[s2.charAt(i) - 'a']--;
        }
        for (int i = 0; i < cm.length; i++) {
            if (cm[i] != 0) {
                return false;
            }
        }
        return count == 0 || count == 2;
    }
}
