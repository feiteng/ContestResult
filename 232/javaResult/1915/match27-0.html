class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
        if (s1.equals(s2)) {
            return true;
        }
        int n1 = s1.length();
        for (int i = 0; i < n1; i++) {
            for (int j = i+1; j < n1; j++) {
                String tmp = s1.substring(0, i) + s1.charAt(j) + s1.substring(i+1, j) + s1.charAt(i) + s1.substring(j+1);
                if (tmp.equals(s2)) {
                    return true;
                }
            }
        }
        return false;
    }
}
