class Solution
{
    public int findCenter(int[][] edges)
    {
        int[] freq = new int[edges.length+2];
        for(int[] x:edges)
        {
            freq[x[0]]++;
            freq[x[1]]++;
        }
        int m = 0;
        for(int i=1;i<freq.length;i++)
        {
            if(freq[i]>freq[m])
            {
                m = i;
            }
        }
        return m;
    }
}
