class Solution {
    public int findCenter(int[][] edges) {
        int n = edges.length + 1;
        int[] d = new int[n+1];
        
        for (int[] e : edges) {
            d[e[0]]++;
            d[e[1]]++;
        }
        
        for (int i = 1; i <= n; i++) {
            if (d[i] > 1) {
                return i;
            }
        }
        
        return -1;
    }
}
