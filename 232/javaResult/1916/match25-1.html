class Solution {
public int findCenter(int[][] edges) {
        int n = edges.length + 1;
        int[] degree = new int[n];
        for (int i = 0; i < edges.length; i++) {
            int x = edges[i][0] - 1;
            int y = edges[i][1] - 1;
            degree[x]++;
            degree[y]++;
        }
        
        int ans = 0;
        int maxCount = 0;
        for (int i = 0; i < n; i++) {
            if (degree[i] > maxCount) {
                maxCount = degree[i];
                ans = i;
            }
        }
        return ans + 1;
    }
}
