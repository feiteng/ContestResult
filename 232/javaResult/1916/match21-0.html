class Solution {
    public int findCenter(int[][] edges) {
        int N = edges.length + 1;
        int[] counts = new int[N + 1];
        for (int[] e : edges) {
            counts[e[0]]++;
            counts[e[1]]++;
        }
        
        for (int v = 1; v <= N; v++) {
            if (counts[v] == N - 1) {
                return v;
            }
        }
        
        return -1;
    }
}
