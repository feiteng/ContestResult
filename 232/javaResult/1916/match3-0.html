class Solution {
    public int findCenter(int[][] edges) {
        int n = edges.length;
        int[] count = new int[n+1];
        for (int i = 0; i < n; i++) {
            count[edges[i][0] - 1]++;
            count[edges[i][1] - 1]++;
        }
        for (int i = 0; i < n + 1; i++) {
            if (count[i] > 1) return (i+1);
        }
        return -1;
    }
}
