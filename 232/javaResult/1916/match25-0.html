class Solution {
    public int findCenter(int[][] edges) {
        int n = edges.length;
        int[] nums = new int[n + 2];
        for(int i = 0;i < n;i++){
            int start = edges[i][0];
            int end = edges[i][1];
            nums[start]++;
            nums[end]++;
        }
        int max = -1;
        int maxIndex = -1;
        for(int i = 1;i < n + 2;i++){
            if(nums[i] > max){
                max = nums[i];
                maxIndex = i;
            }
        }
        return maxIndex;
    }
}
