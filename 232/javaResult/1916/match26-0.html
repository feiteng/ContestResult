class Solution {
    public int findCenter(int[][] edges) {
        int n = edges.length + 1;
        int[] numEdges = new int[n];
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1];
            if (++numEdges[u-1] > 1)
                return u;
            if (++numEdges[v-1] > 1)
                return v;
        }
        return 0;
    }
}
