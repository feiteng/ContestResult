class Solution {
    public int findCenter(int[][] edges) {
        int n = edges.length + 1;
        int [] deg = new int [n+1];
        
        for (int i = 0; i<edges.length; ++i) {
            int u = edges[i][0];
            int v = edges[i][1];
            deg[u]++;
            deg[v]++;
            
if (deg[u] == n-1) return u;
            if (deg[v] == n-1) return v;
        }        
        return -1;
    }
}
