class Solution {
    public int findCenter(int[][] edges) {
        int n=edges.length+1;
        int[] out=new int[n+1];
        for(int[] edge:edges){
            int src=edge[0];
            int des=edge[1];
            out[src]++;
            out[des]++;
        }
        for (int i = 1; i <=n ; i++) {
            if (out[i]==n-1)return i;
        }
        return -1;
    }
}
