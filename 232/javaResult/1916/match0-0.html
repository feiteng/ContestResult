class Solution {
    public int findCenter(int[][] es) {
        
        int n  = es.length+1;
        
        int d[]  =new int[n+1];
        for(int e[]:es){
            d[e[0]]++;
            d[e[1]]++;
        }
        for(int j=1;j<=n;++j){
            if(d[j]==n-1) return j;
        }
        return -1;
    }
}
