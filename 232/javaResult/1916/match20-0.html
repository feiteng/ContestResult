class Solution {
    public int findCenter(int[][] edges) {
        int[] count = new int[edges.length+2];
        for (int[] edge : edges) {
            count[edge[0]]++;
            count[edge[1]]++;
        }
        int res = 1;
        for (int i = 0; i < count.length; i++) {
            if (count[i] > count[res]) {
                res = i;
            }
        }
        return res;
    }
}
