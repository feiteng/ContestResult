class Solution {
    public int maximumScore(int[] nums, int k) {
        int mins[] = new int[nums.length];
        mins[k] = nums[k];
        for(int i=k+1;i<nums.length; i++ ){
            mins[i] = Math.min(mins[i-1], nums[i]);
        }
        for(int i=k-1;i>=0; i-- ){
            mins[i] = Math.min(mins[i+1], nums[i]);
        }
        int p=k-1, q=k+1;
        int res = mins[k];
        while(p>=0 && q<nums.length){
            if(mins[p]>mins[q]){
                res = Math.max(res, mins[p] * (q-p));
                p--;
            }
            else{
                res = Math.max(res, mins[q] * (q-p));
                q++;
            }
        }
        while(p>=0){
            res = Math.max(res, mins[p] * (q-p));
            p--;
        }
        while(q<nums.length){
            res = Math.max(res, mins[q] * (q-p));
            q++;
        }
        return res;
    }
}
