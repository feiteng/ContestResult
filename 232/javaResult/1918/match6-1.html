class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int ans = Integer.MIN_VALUE;
        for (int i = k, j = k, min = nums[k]; i >= 0 && j < n;) {
            min = Math.min(min, Math.min(nums[i], nums[j]));
            int cur = min * (j - i + 1);
            ans = Math.max(cur, ans);
            
            int a = cur, b = cur;
            if (i - 1 >= 0) {
                int u = Math.min(nums[i - 1], min);
                a = u * (j - (i - 1) + 1);
            } else {
                j++;
                continue;
            }
            if (j + 1 < n) {
                int u = Math.min(nums[j + 1], min);
                b = u * ((j + 1) - i + 1);
            } else {
                i--;
                continue;
            }
            if (a > b) {
                i--;
            } else {
                j++;
            }
        }
        return ans;
    }
}
