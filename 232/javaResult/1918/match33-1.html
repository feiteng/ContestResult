class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int l = k;
        int r = k;
        int ans = nums[k];
        int min = nums[k];
        while (true) {
            int lval = -1;
            int rval = -1;
            if (l-1>=0) lval = nums[l-1];
            if (r+1<n) rval = nums[r+1];
            int max = Math.max(lval,rval);
            if (max < 0) {
                break;
            }
            
if (lval < rval) {
                min = Math.min(min,rval);
                r++;
                ans = Math.max(ans,min*(r-l+1));
            } else {
                min = Math.min(min,lval);
                l--;
                ans = Math.max(ans,min*(r-l+1));
            }
            
        }
        return ans;


    }
}
