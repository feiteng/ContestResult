class Solution {
    public int maximumScore(int[] nums, int k) {
        int l = k;
        int r = k;
        int n = nums.length;
        int res = nums[k];
        int min = nums[k];
        
        while (l >= 0 && r < n) {
            while (l > 0 && nums[l - 1] >= min) {
                l--;
            }
            
            while (r < n - 1 && nums[r + 1] >= min) {
                r++;
            }
            
            res = Math.max(res, min * (r - l + 1));
            if (l == 0) {
                r++;
                if (r == n) break;
                min = Math.min(min, nums[r]);
            } else if (r == n - 1) {
                l--;
                if (l == -1) break;
                min = Math.min(min, nums[l]);
            } else 
if (nums[l - 1] > nums[r + 1]) {
                l--;
                min = Math.min(min, nums[l]);
            } else {
                r++;
                min = Math.min(min, nums[r]);
            }
            // System.out.println(min + " " + l + " " + r + " " + res);
        
}
        
        while (l >= 0) {
            min = Math.min(min, nums[l]);
            res = Math.max(res, min * (n - 1 - l + 1));
            l--;
        }
        
        while (r < n) {
            min = Math.min(min, nums[r]);
            res = Math.max(res, min * (r - 0 + 1));
            r++;
        }
        
        return res;
    }
}
