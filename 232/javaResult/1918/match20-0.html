class Solution {
    public int maximumScore(int[] nums, int k) {
        int res = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> (b[1] - a[1]));
        int n = nums.length;
        int min = nums[k];
        for (int i = k-1; i >=0; i--) {
            if (nums[i] < min) {
                pq.add(new int[]{i, min});
                min = nums[i];
            }
        }
        pq.add(new int[] {-1, min});
        
        min = nums[k];
        for (int i = k+1; i < n; i++) {
            if (nums[i] < min) {
                pq.add(new int[]{i, min});
                min = nums[i];
            }
        }
        pq.add(new int[] {n, min});
        
        int left = k;
        int right = k;
        while (!pq.isEmpty()) {
            int[] curr = pq.remove();
            int idx = curr[0];
            int lastMin = curr[1];
            if (idx > k) {
                right = idx;
            } else {
                left = idx;
            }
            
            int prod = lastMin * (right-left-1);
            res = Math.max(prod, res);
        }
        return res;
    }
}
