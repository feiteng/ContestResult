class Solution {
    public int maximumScore(int[] nums, int k) {
        int n =nums.length;
        int l =k-1, r=k+1;
        int ans =nums[k];
        int min = nums[k];
        while(l>=0 || r<n){
            int left = (l<0? 0 : nums[l]);
            int right = (r>=n? 0: nums[r]);
            if(left>right){
                min= Math.min(min, left);
                ans= Math.max(ans, min*(r-l));
                l--;
            }else{
                min=Math.min(min,right);
                ans= Math.max(ans, min*(r-l));
                r++;
            }
        }
        return ans;
    }
}
