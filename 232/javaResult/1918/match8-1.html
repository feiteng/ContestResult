class Solution {
    public int maximumScore(int[] nums, int k) {
        
int i = k; int j = k;
        int curMin = nums[k];
        long res = nums[k];
        while (j < nums.length-1 && i > 0) {
            if (nums[j+1] > nums[i-1]) {
                j++;
                curMin = Math.min(curMin, nums[j]);
            }
            else {
                i--;
                curMin = Math.min(curMin, nums[i]);
            } 
            
            res = Math.max(res, curMin * (j-i+1));
        } 
        
        while (i > 0) {
            i--;
            curMin = Math.min(curMin, nums[i]);
            res = Math.max(res, curMin * (j-i+1));
        }

        while (j < nums.length-1) {
            j++;
            curMin = Math.min(curMin, nums[j]);
            res = Math.max(res, curMin * (j-i+1));
        }
        
        return (int)res;
    }
}
