class Solution {
    public int maximumScore(int[] nums, int k) {
        int max = 0;
        int left = k;
        int right = k;
        int curMin = nums[k];
        while (left >= 0 || right < nums.length) {
           while (left >= 0 && nums[left] >= curMin) {
               left--;
           }
           while (right < nums.length && nums[right] >= curMin) {
               right++;
           }
           max = Math.max(max, curMin * (right - left - 1));
           if (left >= 0 && right < nums.length) {
               curMin = Math.max(nums[left], nums[right]);
           } else if (left >= 0) {
               curMin = nums[left];
           } else if (right < nums.length) {
               curMin = nums[right];
           }
        }
        return max;
    }
}
