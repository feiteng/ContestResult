class Solution {
    public int maximumScore(int[] nums, int k) {
        int l = nums.length;
        
        int[] min = new int[l];
        min[k] = nums[k];
        for (int i = k - 1; i >= 0; i--) {
            min[i] = nums[i] < min[i + 1] ? nums[i] : min[i + 1];
        }
        for (int i = k + 1; i < l; i++) {
            min[i] = nums[i] < min[i - 1] ? nums[i] : min[i - 1];
        }
        
        int max = nums[k];
        int pl = k, ph = k;
        while (pl > 0 && ph < l - 1) {
            if (min[pl - 1] > min[ph + 1]) {
                pl--;
                max = Math.max(max, min[pl] * (ph - pl + 1));
            } else {
                ph++;
                max = Math.max(max, min[ph] * (ph - pl + 1));
            }
        }
        
        while (pl > 0) {
            pl--;
            max = Math.max(max, min[pl] * (l - pl));
        }
        
        while (ph < l - 1) {
            ph++;
            max = Math.max(max, min[ph] * (ph + 1));
        }
        
        return max;
    }
}
