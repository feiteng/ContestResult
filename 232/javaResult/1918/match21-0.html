class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int res = nums[k];
        int l = k, r = k, min = nums[k];
        while(l>0&&r<n-1){
            if(nums[l-1]>nums[r+1]){
                min = Math.min(min, nums[l-1]);
                res = Math.max((r-l+2)*min, res);
                l--;
            }else if(nums[l-1]<nums[r+1]){
                min = Math.min(min, nums[r+1]);
                res = Math.max((r-l+2)*min, res);
                r++;
            }else{
                min = Math.min(min, nums[r+1]);
                res = Math.max((r-l+3)*min, res);
                l--;
                r++;
            }
        }
        while(l>0){
            min = Math.min(min, nums[--l]);
            res = Math.max((r-l+1)*min, res);
        }
        while(r<n-1){
            min = Math.min(min, nums[++r]);
            res = Math.max((r-l+1)*min, res);
        }
        return res;
    }
}
