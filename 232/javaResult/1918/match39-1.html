class Solution {
    public int maximumScore(int[] nums, int k) {
        int i1 = k, i2 = k, n = nums.length;
        int res = nums[k];
        for(int i = nums[k]; i >= 1; i--) {
            while(i1 >= 0 && nums[i1] >= i) i1--;
            while(i2 < n && nums[i2] >= i) i2++
;
            res = Math.max(res, i * (i2 - i1 - 1));
            //System.out.println(i + " " + i1 + " " + i2);
        }
        return res;
    }
}
