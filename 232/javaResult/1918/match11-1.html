class Solution {
    public int maximumScore(int[] nums, int k) {
        int N = nums.length;
        int left = k;
        int right = k;
        int min = nums[k];
        int res = min;
        while (left != 0 || right != N - 1) {
            if (left != 0 && right != N - 1) {
                if (nums[left - 1] >= nums[right + 1]) {
                    left--;
                    min = Math.min(min, nums[left]);
                } else {
                    right++;
                    min = Math.min(min, nums[right]);
                }
            } else if (left == 0) {
                right++;
                min = Math.min(min, nums[right]);
            } else if (right == N - 1) {
                left--;
                min = Math.min(min, nums[left]);
            
}
            res = Integer.max(res, min * (right - left + 1));
        }
        return res;
    }
}
