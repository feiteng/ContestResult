class Solution {
    public int maximumScore(int[] nums, int k) {
        int l = k, r = k;
        int res = nums[k], s = nums[k], m = nums[k];
        while (l>0 || r<nums.length-1) {
            if (l==0) {
                r++;
                s += nums[r];
                m = Math.min(m,nums[r]);
            } else if (r==nums.length-1) {
                l--;
                s += nums[l];
                m = Math.min(m,nums[l]);
            } else {
                if (nums[r+1]>nums[l-1]) {
                   r++;
                s += nums[r];
                m = Math.min(m,nums[r]); 
                } else {
                    l--;
                s += nums[l];
                m = Math.min(m,nums[l]);
                }
            }
            res = Math.max(res,m * (r-l+1));
        }
        return res;
    }
}
