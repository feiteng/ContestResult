class Solution {
    public int maximumScore(int[] nums, int k) {
        //pq -> {position, min}
        
PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> (b[1] - a[1]));
        
        int n = nums.length;
        int minSofar = nums[k];
        for (int i = k+1; i < n; i++) {
            if (nums[i] < minSofar) {
                pq.add(new int[] {i, minSofar});
                minSofar = nums[i];
            }
        }
        pq.add(new int[] {n, minSofar});
        
        minSofar = nums[k];
        for (int i = k-1; i >= 0; i--) {
            if (nums[i] < minSofar) {
                pq.add(new int[] {i, minSofar});
                minSofar = nums[i];
            }
        }
        pq.add(new int[] {-1, minSofar});
        
        //int minSofar = nums[k];
        int left = k;
        int right = k;
        int ans = 0;
        
while(!pq.isEmpty()) {
            int[] curr = pq.remove();
            int idx = curr[0];
            int lastMin = curr[1];
            if (idx > k) {
                right = idx;
            } else {
                left = idx;
            }
            
            
            int temp = lastMin * (right - left - 1);
            ans = Math.max(ans, temp);
        }
        
        return ans;
    }
}
