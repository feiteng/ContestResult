class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int i = k -1;
        int j = k +1;
        int res = nums[k];
        int min = nums[k];
        while(!(i == -1 && j == n)){
            int left = i == -1 ? -1 : nums[i];
            int right = j == n ? -1 : nums[j];
            
if(left > right) {
                i--;
                min = Math.min(min, left);
                res = Math.max(res, min * (j - i -1));
            }
            else {
                j++;
                min = Math.min(min, right);
                res = Math.max(res, min * (j - i -1));
            }
        }
        
        return res;
    }
}
