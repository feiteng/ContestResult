class Solution {
    public int maximumScore(int[] nums, int k) {
        int N = nums.length;
        
int max = nums[k];
        int i = k, j = k;
        int min = nums[k];
        while (i >= 0 || j < N) {
            while (i >= 0 && nums[i] >= min) {
                i--;
            }
            while (j < N && nums[j] >= min) {
                j++;
            }
            // System.out.println(min + " ===> " + i + " ~ " + j + " : " + (min * (j - i - 1)));
            max = Math.max(max, min * (j - i - 1));
            if (i >= 0 && j < N) {
                min = Math.max(nums[i], nums[j]);
            } else if (i >= 0) {
                min = nums[i];
            } else if (j < N) {
                min = nums[j];
            }
        }
        return max;
    }
}
