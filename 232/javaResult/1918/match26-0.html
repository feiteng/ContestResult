class Solution {
    public int maximumScore(int[] nums, int k) {
        int left = k;
        int right = k;
        int min = nums[k];
        int res = min;

        int len = nums.length;
        while (right - left + 1 < len) {
            if (left == 0) {
                min = Math.min(min, nums[++right]);
            } else if (right == len-1) {
                min = Math.min(min, nums[--left]);
            } else {
                if (nums[right+1] > nums[left-1]) {
                    min = Math.min(min, nums[++right]);
                } else if (nums[right+1] < nums[left-1]) {
                    min = Math.min(min, nums[--left]);
                } else {
                    min = Math.min(min, nums[--left]);
                }
            }
            res = Math.max(res, min*(right - left+1));
        }
        return res;
    }
}
