class Solution {
    public int maximumScore(int[] A, int k) {
        Deque<Integer> stack = new ArrayDeque<>();
        int n = A.length, res = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            int num = A[i];
            while (!stack.isEmpty() && A[stack.peekFirst()] > num) {
                int h = A[stack.pollFirst()];
                int left = (stack.isEmpty() ? -1 : stack.peekFirst()) + 1;
                int w = i - left;
                if (i > k && left <= k) {
                    res = Math.max(res, w * h);
                }
            }
            stack.addFirst(i);
        }
        while (!stack.isEmpty()) {
            int h = A[stack.pollFirst()];
            int left = (stack.isEmpty() ? -1 : stack.peekFirst()) + 1;
            int w = n - left;
            if (left <= k) {
                res = Math.max(res, w * h);
            }
        }
        return res;
    }
}
