class Solution {
    public int maximumScore(int[] nums, int k) {
        int i = k;
        int j = k;
        
        int min = nums[k];
        
        int max = min * (j - i + 1);
        int N = nums.length;
        while (i-1 >= 0 || j+1 < N) {
            if (i-1 < 0) {
                j++;
                min = Math.min(min, nums[j]);
            } else if (j+1 >= N) {
                i--;
                min = Math.min(min, nums[i]);
            } else {
                if (nums[i-1] >= nums[j+1]) {
                    i--;
                    min = Math.min(min, nums[i]);
                } else {
                    j++;
                    min = Math.min(min, nums[j]);
                }
            }
            max = Math.max(max, min * (j - i + 1));
        }
        return max;
    }
}
