class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int leftOffset = 0;
        int rightOffset = 0;
        int result = nums[k];
        int minElement = nums[k];
        while (k - leftOffset >= 0 && k + rightOffset < n) {
            if (k - leftOffset - 1 >= 0 && k + rightOffset + 1 < n) {
                if (nums[k - leftOffset - 1] >= nums[k + rightOffset + 1]) {
                    leftOffset++;
                    minElement = Math.min(minElement, nums[k - leftOffset]);
                } else {
                    rightOffset++;
                    minElement = Math.min(minElement, nums[k + rightOffset]);
                }                
            } else if (k - leftOffset - 1 < 0 && k + rightOffset + 1 < n) {
                rightOffset++;
                minElement = Math.min(minElement, nums[k + rightOffset]);
            } else if (k - leftOffset - 1 >= 0 && k + rightOffset + 1 >= n) {
                leftOffset++;
                minElement = Math.min(minElement, nums[k - leftOffset]);
            } else {
                break;
            }
            result = Math.max(result, minElement * (leftOffset + rightOffset + 1));
        }
        return result;
    }
}
