class Solution {
    public int maximumScore(int[] nums, int k) {
        int i = k;
        int j = k;
        int max = nums[k];
        int min = nums[k];
        int count = 1;
        while(i-1 >=0 && j+1 < nums.length){
            if(nums[i-1] > nums[j+1]){
                i--;
                min = Math.min(min, nums[i]);
            }else{
                j++;
                min = Math.min(min, nums[j]);
            }
            count++;
            max = Math.max(max, min * count);
        }
        while(i-1 >=0){
             i--;
            min = Math.min(min, nums[i]);
            count++;
            max = Math.max(max, min * count);
        }
        while(j+1 < nums.length){
            j++;
            min = Math.min(min, nums[j]);
            count++;
            max = Math.max(max, min * count);
        }
        return max;
    }
}
