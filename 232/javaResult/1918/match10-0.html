class Solution {
    public int maximumScore(int[] heights, int k) {
        int n = heights.length;
        int[] left = new int[n]; // left[i]表示第i个柱子往左第一个高度小于它的柱子的下标+1
        int[] right = new int[n]; // right[i]表示第i个柱子往右第一个高度小于它的柱子的下标-1
        Deque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i < n; i++) {
            while (!stack.isEmpty() && heights[stack.getLast()] >= heights[i]) {
                stack.removeLast();
            }
            if (stack.isEmpty()) {
                left[i] = 0;
            } else {
                left[i] = stack.getLast() + 1;
            }
            stack.addLast(i);
        }
        stack.clear();
        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && heights[stack.getLast()] >= heights[i]) {
                stack.removeLast();
            }
            if (stack.isEmpty()) {
                right[i] = n - 1;
            } else {
                right[i] = stack.getLast() - 1;
            }
            stack.addLast(i);
        }

        int res = 0;
        for (int i = 0; i < n; i++) {
            if (k >= left[i] && k <= right[i]) {
                res = Math.max(res, heights[i] * (right[i] - left[i] + 1));
            }
        }
        return res;
    }
}
