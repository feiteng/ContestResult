class Solution {
    
public int maximumScore(int[] nums, int k) {
        int left = k, right = k;
        int max = nums[k];
        int minNum = nums[k];
        while (left > 0 && right < nums.length - 1) {
            if (nums[left - 1] >= nums[right + 1]) {
                minNum = Math.min(minNum, nums[left - 1]);
                max = Math.max(max, minNum * (right - left + 2));
                left--;
            } else {
                minNum = Math.min(minNum, nums[right + 1]);
                max = Math.max(max, minNum * (right - left + 2));
                right++; 
            }
        }
        
        while (left > 0) {
            minNum = Math.min(minNum, nums[left - 1]);
            max = Math.max(max, minNum * (right - left + 2));
            left--;
        }
        
        while (right < nums.length - 1) {
            minNum = Math.min(minNum, nums[right + 1]);
            max = Math.max(max, minNum * (right - left + 2));
            right++;  
        }
        return max;
        
    }
}
