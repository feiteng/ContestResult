class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int[] arr = new int[n];
        arr[k] = nums[k];
        for (int i = k - 1; i >= 0; i --) {
            arr[i] = Math.min(arr[i + 1], nums[i]);
        }
        for (int i = k + 1; i < n; i ++) {
            arr[i] = Math.min(arr[i - 1], nums[i]);
        
}
        int res = arr[k];
        int lo = k;
        int hi = k;
        while ((lo - 1) >= 0 && (hi + 1) < n) {
            if (arr[lo - 1] < arr[hi + 1]) {
                hi ++;
                res = Math.max(res, arr[hi] * (hi - lo + 1));
            } else {
                lo --;
                res = Math.max(res, arr[lo] * (hi - lo + 1));
            }
        }
        while (lo - 1 >= 0) {
            lo --;
            res = Math.max(res, arr[lo] * (hi - lo + 1));
        }
        while (hi + 1 < n) {
            hi ++;
            res = Math.max(res, arr[hi] * (hi - lo + 1));
        }
        return res;
    }
}
