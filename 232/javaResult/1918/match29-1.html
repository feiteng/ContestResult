class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        Deque<Integer> stack = new ArrayDeque<>();
        
int res = 0;
        for (int i = 0; i < n; i++) {
            int num = nums[i];
            while (!stack.isEmpty() && nums[stack.peekFirst()] > num) {
                int h = nums[stack.pollFirst()];
                int left = (stack.isEmpty() ? -1 : stack.peekFirst()) + 1;
                int w = i - left;
                if (i > k && left <= k) {
                    res = Math.max(res, w * h);
                }
            }
            stack.offerFirst(i);
        }
        while (!stack.isEmpty()) {
            int h = nums[stack.pollFirst()];
            int left = (stack.isEmpty() ? -1 : stack.peekFirst()) + 1;
            int w = n - left;
            if (left <= k) {
                res = Math.max(res, w * h);
            }
        }
        return res;
    }
