class Solution {
    public int maximumScore(int[] nums, int k) {
        int left = k, right = k;
        int res = nums[k];
        int cur = nums[k];

        while (left >= 0 || right < nums.length) {
            while (left >= 0 && nums[left] >= cur) {
                left--;
            }
            while (right < nums.length && nums[right] >= cur) {
                right++;
            }
            res = Math.max(res, cur * (right - left - 1));
            if (left >= 0 && right < nums.length) {
                cur = Math.max(nums[left], nums[right]);
            } else if (left >= 0) {
                cur = nums[left];
            } else if (right < nums.length) {
                cur = nums[right];
            } else {
                break;
            
}
        }
        return res;


    }

}
