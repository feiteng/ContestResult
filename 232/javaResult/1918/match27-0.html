class Solution {
    public int maximumScore(int[] nums, int k) {
        int n=nums.length;
        int[] min=new int[n];
        min[k]=nums[k];
        for(int i=k-1; i>=0; i--){
            min[i]=Math.min(min[i+1],nums[i]);
        }
        for(int i=k+1; i<n; i++){
            min[i]=Math.min(min[i-1],nums[i]);
        }
        
        int max=nums[k]*1;
        int i=k, j=k;
        while(i>=0 && j<n){
            int l=Integer.MIN_VALUE, r=Integer.MIN_VALUE;
            if(i-1>=0)
                l=min[i-1];
            if(j+1<n)
                r=min[j+1];
            if(l==Integer.MIN_VALUE && r==Integer.MIN_VALUE) break;
            if(r>=l){
                j++;   
                max=Math.max(max,min[j]*(j-i+1));
            }
            else{
                i--;
                max=Math.max(max,min[i]*(j-i+1));
            }            
        }
        return max;
    }
}
