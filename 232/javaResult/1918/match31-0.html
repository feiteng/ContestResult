class Solution {
    public int maximumScore(int[] nums, int k) {
        int l = k, r =k;
        int n = nums.length;
        
        int min = nums[k];
        int MaxValue = min*1;
        while(l >0 && r<n-1){
            if(nums[l-1] >= nums[r+1]){
                min = Math.min(min,nums[--l]);
                
            }else{
                min = Math.min(min,nums[++r]);
                //MaxValue = Math.max(MaxValue, min*(l-r+1));
            }
            MaxValue = Math.max(MaxValue, min*(r-l+1));
        }
        while(l>0){
            min = Math.min(min,nums[--l]);
            MaxValue = Math.max(MaxValue, min*(r-l+1));
        }
        while(r<n-1){
            min = Math.min(min,nums[++r]);
            MaxValue = Math.max(MaxValue, min*(r-l+1));
        }
        return MaxValue;
    }
}
