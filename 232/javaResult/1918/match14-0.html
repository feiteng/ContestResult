class Solution {
    public int maximumScore(int[] nums, int k) {
        int n = nums.length;
        int i = k;
        int j = k;
        int result = nums[k];
        int min = nums[k];
        while(j < n-1 || i > 0){
            if(i == 0){
                j++;
                min = Math.min(nums[j],min);
            }
            else if (j == n-1){
                i--;
                min = Math.min(nums[i],min);
            }
            else if( nums[i-1] >= min){
                i--;
            }
            else if( nums[j+1] >= min){
                j++;
            }
            else if (nums[i-1] > nums[j+1]){
                i--;
                min = Math.min(nums[i],min);
            }
            else{
                j++;
                min = Math.min(nums[j],min);                
            }
            result = Math.max(result, min * (j-i+1));
            
        }
        return result;
    }
}
