class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((m,n)-> ((n[1]-n[0])*1.0/(n[1]*(n[1]+1)) < ((m[1]-m[0])*1.0/(m[1]*(m[1]+1))) ? -1 : 1));
        double sum = 0.0;
        for(int i = 0; i<classes.length; i++){
            int a = classes[i][0], b = classes[i][1];
            
pq.add(new int[]{a, b});
            sum+=a*1.0/b;
        }
        while(extraStudents!=0){
            int[] cur = pq.poll();
            int a = cur[0], b = cur[1];
            sum-=a*1.0/b;
            sum+=(a+1)*1.0/(b+1);
            
pq.add(new int[]{a+1, b+1});
            extraStudents--;
        }
        return sum*1.0/classes.length;
    }
}
