class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
           public int compare(Integer a, Integer b) {
               // double ratio1 = classes[a][0] / (double)classes[a][1];
               // double ratio2 = classes[b][0] / (double)classes[b][1];
               double ratio1 = calcDelta(classes[a][0], classes[a][1]);
               double ratio2 = calcDelta(classes[b][0], classes[b][1]);
               if (ratio1 <= ratio2) {
                   return 1;
               } else {
                   return -1;
               }
           } 
        });
        
        for (int i = 0; i < classes.length; i++) {
            if (classes[i][0] != classes[i][1])
                pq.add(i)
;
        }
        
        for (int k = 0; k < extraStudents; k++) {
            if (pq.isEmpty()) {
                break;
            }
            int i = pq.remove();
            classes[i][0]++;
            classes[i][1]++;
            pq.add(i);
        }
        
        double ratio = 0;
        for (int i = 0; i < classes.length; i++) {
            //System.out.print(" " + Arrays.toString(classes[i]));
            ratio += classes[i][0] / (double)classes[i][1];
        }
        
        return ratio / (double)classes.length;
    }
    
    private double calcDelta(int m, int n) {
        double mm = (double)m;
        double nn = (double)n;
        return (nn-mm)/(nn*(nn+1));
    }
}
