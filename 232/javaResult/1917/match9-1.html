class Solution {
    public double maxAverageRatio(int[][] classes, int n) {
        
        Queue<double[]> queue = new PriorityQueue<double[]>((a, b) -> {
            
double x = ((b[0]+1)/(b[1]+1) - b[0]/b[1]), y = ((a[0]+1)/(a[1]+1) - a[0]/a[1]);
            if(x>y) return 1;
            if(x<y) return -1;
            return 0;
        } );
        
        for(int[] c : classes){
            queue.offer(new double[]{c[0], c[1]});
        
}
        
        while(n-->0){
            double[] c = queue.poll();
            c[0]+=1.0d;
            c[1]+=1.0d;
            queue.offer(c);
        }
        
        double ans = 0;
        while(!queue.isEmpty()){
            double[] c = queue.poll();
            ans += (c[0]/c[1]);
        }
        return ans / classes.length;
    }
}
