class Solution {
    /*
    a/b ->
    
    
    (b-a)
------
b(b+1)
    */
    
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        Queue<int[]> q = new PriorityQueue<>((a, b) -> Double.compare(helper(b), helper(a)));
        for(int[] c : classes) {
            q.offer(c);
        }
        for(int i = extraStudents; i > 0; --i) {
            int[] c = q.poll();
            ++c[0];
            ++c[1];
            q.offer(c);
        }
        double result = 0;
        for(int[] c : q) {
            result += 1. * c[0] / c[1];
        }
        return result / classes.length;
    }
    
    static double helper(int[] c) {
        return 1. * (c[1] - c[0]) / c[1] / (c[1] + 1);
    }
}
