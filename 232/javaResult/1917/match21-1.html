class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> q = new PriorityQueue<>((o1,o2)->{
            double ans=1.0*(o2[1]-o2[0])/(o2[1]*o2[1]+o2[1])-1.0*(o1[1]-o1[0])/(o1[1]*o1[1]+o1[1]);
            if(ans==0) return 0;
            else return ans>0?1:-1;
        
});
        for(int[] x:classes){
            q.add(x);
        }
        while(extraStudents>0){
            int[] poll = q.poll();
            q.add(new int[]{poll[0]+1,poll[1]+1});
            extraStudents--;
        }
        double ans=0;
        Iterator<int[]> iterator = q.iterator();
        while(iterator.hasNext()){
            int[] next = iterator.next();
            ans+=1.0*next[0]/next[1];
        }
        return ans/classes.length;
    }
}
