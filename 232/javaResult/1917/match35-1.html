import java.util.PriorityQueue;


class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)-> -comp(a,b));

        for (int[] c : classes) {
            pq.add(c);
        }
        for (int i = 0; i < extraStudents; i++) {
            int[] c = pq.poll();
            c[0] ++;
            c[1] ++;
            pq.add(c);
        }
        double sum = 0;
        while(!pq.isEmpty()){
            int [] c = pq.poll();
            sum += c[0] * 1.0 / c[1];
        }
        return sum / classes.length;
    }
    int comp(int[] a, int[] b){
        double ta = a[1] - a[0];
        double tb = b[1] - b[0];
        ta /= a[1];
        ta /= (a[1] + 1);
        tb /= b[1];
        tb /= (b[1] + 1);
        return Double.compare(ta, tb);
    }
}
