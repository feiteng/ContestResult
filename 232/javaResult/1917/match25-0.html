class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<>((a, b) -> {
            return Double.compare((b[0] + 1) / (b[1] + 1) - b[0] / b[1], (a[0] + 1) / (a[1] + 1) - a[0] / a[1]);
        });
        for (int[] c : classes) {
            pq.add(new double[] {c[0], c[1]});
        }
        while (extraStudents-- > 0) {
            double[] h = pq.poll();
            h[0]++;
            h[1]++;
            pq.add(h);
        }
        double ans = 0;
        while (!pq.isEmpty()) {
            double[] h = pq.poll();
            ans += h[0] / h[1];
        }
        return ans / classes.length;
    }
}
