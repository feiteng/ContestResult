class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<>(10, new Comparator<int[]>(){
            public int compare(int[] a1, int[] a2) {
                return (int)((((double)(a2[0]+1)/(a2[1]+1) - (double)a2[0]/a2[1]) - ((double)(a1[0]+1)/(a1[1]+1) - (double)a1[0]/a1[1])) * 10000000) ;
            }
        });
        
        for(int[] myClass : classes) {
            pq.offer(myClass);
        }
        
        for(int i = 0 ; i < extraStudents ; i++) {
            int[] myClass = pq.poll();
            myClass[0] = myClass[0] + 1;
            myClass[1] = myClass[1] + 1;
            pq.offer(myClass);
        }
        
        double res = 0;
        
        while(!pq.isEmpty()) {
            int[] myClass = pq.poll();
            
            res += (double)myClass[0]/myClass[1];
        }
        
        return res / classes.length;
    }
}
