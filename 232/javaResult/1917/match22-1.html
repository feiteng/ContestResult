class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a, b) -> {
            double diffA = (a[0] + 1)/ (double)(a[1] + 1) - (a[0]) / (double)(a[1]);
            double diffB = (b[0] + 1)/ (double)(b[1] + 1) - (b[0]) / (double)(b[1]);
            return (diffB - diffA) > 0 ? 1 : -1;
        
});
        for (int i = 0; i < classes.length; i++) {
            pq.offer(classes[i]);
        }
        int num = extraStudents;
        while (num > 0) {
            int[] curt = pq.poll();
            curt[0] = curt[0] + 1;
            curt[1] = curt[1] + 1;
            pq.offer(curt);
            num--;
        
}
        double sum = 0;
        while (!pq.isEmpty()) {
            int[] curt = pq.poll();
            sum += (curt[0]) / (double)(curt[1]);

        }
        return sum / (double)classes.length;
    }
}
