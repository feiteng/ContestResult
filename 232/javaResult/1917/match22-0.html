import java.util.PriorityQueue;

class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> queue = new PriorityQueue<>((o1, o2) -> {
            Double a=((double)(o1[0]+1))/(o1[1]+1)-(double)o1[0]/o1[1];
            Double b=((double)(o2[0]+1))/(o2[1]+1)-(double)o2[0]/o2[1];
            return b.compareTo(a);
        }
        );
        for (int[] i : classes) {
            queue.add(i);
        }
        int n = extraStudents;
//        while (n > 0) {
//            int[] cur = queue.poll();
//            int[] next = queue.peek();
//            int a = cur[0] + cur[1];
//            int b = next[0] + next[1];

//            int max = Math.min(gap, n);
//            n -= max;
//            queue.add(new int[]{cur[0] + max, cur[1] + max});
//        }
        while (n>0){
                        int[] cur = queue.poll();
            queue.add(new int[]{cur[0] + 1, cur[1] + 1});
            n--;
//
        }
//        for (int[] ints : queue) {
//            print.println(ints);
//        }
        double res = 0;
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            res += (double) cur[0] / cur[1];
        }
        return res / classes.length;
    }
