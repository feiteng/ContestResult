class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        int n = classes.length;
        PriorityQueue<int[]>queue = new PriorityQueue<>(
                (o1, o2) -> {
                    double diff1 = (double)(o1[0] + 1)/(o1[1] + 1) - (double)(o1[0])/(o1[1]);
                    double diff2 = (double)(o2[0] + 1)/(o2[1] + 1) - (double)(o2[0])/(o2[1]);
                    if(diff1 - diff2 > 0){
                        return -1;
                    }else if(diff1 - diff2 < 0){
                        return 1;
                    }else{
                        return 0;
                    }
                }
        );
        for (int i = 0; i < n; i++) {
                queue.add(classes[i]);
        }
        for (int i = 0; i < extraStudents; i++) {
            int[] poll = queue.poll();
            poll[0] += 1;
            poll[1] += 1;
            queue.add(poll);
        }
        double sum = 0;
        while (!queue.isEmpty()){
            int[] poll = queue.poll();
            sum += (double)(poll[0]) / (double) (poll[1]);
        }
        return sum / n;
    }
}
