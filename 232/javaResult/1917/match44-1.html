class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Integer> f = new PriorityQueue<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                double rate1 = (double)(classes[o1][0]+1) / (classes[o1][1]+1) - (double)classes[o1][0] / classes[o1][1];
                double rate2 = (double)(classes[o2][0]+1) / (classes[o2][1]+1) - (double)classes[o2][0] / classes[o2][1];
                if(rate2-rate1>0) {
                    return 1;
                } else if(rate1==rate2) {
                    return 0;
                }
                return  -1;
            }
        });
        for(int i=0;i<classes.length;i++) {
            f.add(i);
        }
        
for(int i=1;i<=extraStudents;i++) {
            int a = f.poll();
            classes[a][0]++;
            classes[a][1]++;
            f.add(a);
        }
        double ratio = 0;
        for(int i=0;i<classes.length;i++) {
            ratio += (double)classes[i][0]/classes[i][1];
        }
        return ratio / classes.length;
    }
}
