class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> que = new PriorityQueue<>((o1, o2) -> -Double.compare((o1[0] + 1.0) / (o1[1] + 1) - o1[0]* 1.0 / o1[1] , 
                                                                                  (o2[0] + 1.0) / (o2[1] + 1) - o2[0]* 1.0 / o2[1]));
        
for(int[] class1: classes){
            que.offer(class1);
        }
        while(extraStudents > 0){
            int[] class1 = que.poll();
            class1[0]++;
            class1[1]++;
            que.offer(class1);
            extraStudents--;
        }
        double sum = 0;
        while(!que.isEmpty()){
            int[] class1 = que.poll();
            sum += (class1[0] * 1.0 / class1[1]);
        }
        return sum / classes.length;
    }
}
