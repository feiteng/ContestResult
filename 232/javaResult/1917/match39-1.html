class Solution {
    public double maxAverageRatio(int[][] cs, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<>((a, b) -> {
            double first = (a[0] + 1) / (double) (a[1] + 1) - a[0] / a[1];
            double second = (b[0] + 1) / (b[1] + 1) - b[0] / b[1];
            
            if (first > second) {
                return -1;
            } else {
                return 1;
            }
        });
        
for (int[] c: cs) {
            pq.add(new double[]{c[0], c[1]});
        }
        while (extraStudents != 0) {
            double[] cur = pq.poll();
            pq.add(new double[]{cur[0] + 1, cur[1] + 1});
            extraStudents --;
        }
        double sum = 0;
        
while (!pq.isEmpty()) {
            double[] cur = pq.poll();
            sum += cur[0] / cur[1];
        }
        return sum / cs.length;
    }
}
