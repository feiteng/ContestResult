class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->-Double.compare(1.0*(a[0]+1)/(a[1]+1)-1.0*(a[0])/(a[1]), 1.0*(b[0]+1)/(b[1]+1)-1.0*(b[0])/(b[1])));
        for(int[] c : classes){
            pq.add(c);
        }
        for(int i = 0; i < extraStudents; i++){
            int[] old = pq.poll();
            old[0]++;
            old[1]++;
            pq.add(old);
        }
        double sum = 0;
        double n = 0;
        while(pq.size() != 0){
            int[] old = pq.poll();
            //System.out.println(Arrays.toString(old));
            sum += 1.0*old[0]/old[1];
            n++;
        }
        return sum / n;
    }
}
