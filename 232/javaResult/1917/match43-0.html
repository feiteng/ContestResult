class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Class> heap = new PriorityQueue<>((o1, o2) -> {
            double res1 = (o1.p + 1) / (o1.t + 1) - o1.p / o1.t;
            double res2 = (o2.p + 1) / (o2.t + 1) - o2.p / o2.t;

            return Double.compare(res2, res1);
        });

        for (int i = 0; i < classes.length; i++) {
            heap.add(new Class(classes[i][0], classes[i][1]));
        }

        while (extraStudents > 0) {
            extraStudents--;
            Class next = heap.poll();
            next.p++;
            next.t++;
            heap.add(next);
        }

        double res = 0;
        while (!heap.isEmpty()) {
            Class next = heap.poll();
            res += next.p / next.t;
        }

        return res / classes.length;
    }

    static class Class {
        double p;
        double t;

        public Class(double p, double t) {
            this.p = p;
            this.t = t;
        }
    }
}
