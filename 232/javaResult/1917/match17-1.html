class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        
PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> {
            double r1 = (a[0] + 1.0) / (a[1] + 1.0) - ((double) a[0]) / a[1];
            double r2 = (b[0] + 1.0) / (b[1] + 1.0) - ((double) b[0]) / b[1];
            
            if (r1 < r2) {
                return 1;
            } else if (r1 > r2) {
                return -1;
            } else {
                return 0;
            }
        });
        
        for (int[] cl : classes) {
            pq.offer(cl);
        }
        
        for (int i = 0; i < extraStudents; ++i) {
            int[] temp = pq.poll();
            // System.out.println("STUDENT: " + i);
            // System.out.println(temp[0]);
            // System.out.println(temp[1]);
            ++temp[0];
            ++temp[1];
            pq.offer(temp);
        }
        
        double res = 0;
        while (!pq.isEmpty()) {
            int[] temp = pq.poll();
            res += ((double)temp[0]) / temp[1];
        }
        
        return res / classes.length;
    }
}
