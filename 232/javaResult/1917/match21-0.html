class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> {
            double val1 = ((a[0] + 1) / (1.0 * (a[1] + 1))) - (a[0] / (1.0 * a[1]));
            double val2 = ((b[0] + 1) / (1.0 * (b[1] + 1))) - (b[0] / (1.0 * b[1]));
            double val = val2 - val1;
            return val < 0 ? -1 : 1;
        });
        for(int[] c : classes) {
            pq.add(c);
        }
        while(extraStudents > 0) {
            int[] a = pq.poll();
            pq.add(new int[]{a[0] + 1, a[1] + 1});
            extraStudents--;
        }
        double avgSum = 0.0;
        int size = pq.size();
        while(!pq.isEmpty()) {
            int[] a = pq.poll();
            // System.out.println(Arrays.toString(a));
            avgSum += (a[0] / (1.0 * a[1]));
        }
        return avgSum / size;
        
    }
}
