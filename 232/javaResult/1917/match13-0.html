class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<double[]>((a,b)->Double.compare((b[0]+1)/(b[1]+1)-(b[0])/(b[1]),(a[0]+1)/(a[1]+1)-(a[0])/(a[1])));
        int len = classes.length;
        for(int[] c:classes){
            pq.offer(new double[]{c[0],c[1]});
        }
        while(extraStudents>0){
            extraStudents--;
            double[] c = pq.poll();
            c[0]+=1;
            c[1]+=1;
            pq.offer(c);
        }
        double sum = 0;
        while(!pq.isEmpty()){
            double[] c = pq.poll();
            //System.out.println(Arrays.toString(c));
            sum+=c[0]/c[1];
        }
        return sum/len;
    }
}
