class Solution {
    public class PassLv{
        public int pass;
        public int total;
        public PassLv(int pass, int total){
            this.pass = pass;
            this.total = total;
        }
    }
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<PassLv> pq = new PriorityQueue<PassLv>((p1,p2)->{
            double diff1 = (p1.pass + 1.0)/(p1.total+1.0) - p1.pass*1.0/p1.total;
            double diff2 = (p2.pass + 1.0)/(p2.total+1.0) - p2.pass*1.0/p2.total;
            if (diff2 > diff1) {
                return 1;
            } else if (diff2 < diff1) {
                return -1;
            } else {
                return 0;
            }
        });
        for (int i = 0; i < classes.length; i++) {
            PassLv pl = new PassLv(classes[i][0], classes[i][1]);
            pq.offer(pl);
        }
        for (int i = 0; i < extraStudents; i++) {
            PassLv temp = pq.poll();
            temp.pass+=1;
            temp.total+=1;
            pq.offer(temp);
        }
        double res = 0;
        while (!pq.isEmpty()) {
            PassLv temp = pq.poll();
            res += 1.0 * temp.pass / temp.total;
        }
        return res/classes.length;
    }
}
