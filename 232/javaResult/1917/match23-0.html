class Solution {
    
    private static class C {
        Integer pass;
        Integer total;
        
        public C(int pass, int total) {
            this.pass = pass;
            this.total = total;
        }
        
        public String toString() {
            return "[" + pass + ", " + total + "]";
        }
        
        public Double improvement() {
            double current = ((double) pass) / total;
            double next = ((double) pass + 1) / (total + 1);
            return next - current;
        }
        
    }
    
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<C> queue = new PriorityQueue<>(classes.length, (a, b) -> b.improvement().compareTo(a.improvement()));
        
        for (int i = 0; i < classes.length; i++) {
            queue.add(new C(classes[i][0], classes[i][1]));
        }
        
        for (int i = 0; i < extraStudents; i++) {
            C smallest = queue.poll();
            smallest.pass += 1;
            smallest.total += 1;
            queue.add(smallest);
        }
        
        System.out.println(queue);
        
        double sumRatio = 0;
        
        while (!queue.isEmpty()) {
            C c = queue.poll();
            
            sumRatio += ((double) c.pass) / c.total;
        }
        
        return sumRatio / classes.length;
    }
}
