class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<>((o1, o2) -> {
            if (((o1[0] + 1) / (o1[1] + 1)) - (o1[0] / o1[1]) == ((o2[0] + 1) / (o2[1] + 1)) - (o2[0] / o2[1])) {
                return 0;
            }
            if (((o1[0] + 1) / (o1[1] + 1)) - (o1[0] / o1[1]) < ((o2[0] + 1) / (o2[1] + 1)) - (o2[0] / o2[1])) {
                return 1;
            }            
            return -1;
        });
        for (int i = 0; i < classes.length; i++) {
            pq.offer(new double[]{classes[i][0], classes[i][1]});
        }
        while (extraStudents > 0) {
            double[] cur = pq.poll();
            pq.offer(new double[]{cur[0] + 1, cur[1] + 1});
            extraStudents--;
        }
        double res = 0;
        double ratioSum = 0;
        int count = pq.size();
        while (pq.size() > 0) {
            double[] cur = pq.poll();
            ratioSum += (cur[0] / cur[1]);
        }
        return ratioSum / (count * 1.0);
    }
}
