class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        // PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> ( ((b[0]+1)*1.0/(b[1]+1) - (b[0])*1.0/(b[1])) - ((a[0]+1)*1.0/(a[1]+1) - (a[0])*1.0/(a[1])) ));
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> Double.compare(((b[0]+1)*1.0/(b[1]+1) - (b[0])*1.0/(b[1])), ((a[0]+1)*1.0/(a[1]+1) - (a[0])*1.0/(a[1]))));
        for (int[] c : classes) {
            pq.add(c);
        }
        int n = classes.length;
        if (n == 1) {
            return (classes[0][0] + extraStudents) * 1.0 / (classes[0][1] + extraStudents);
        }
        while (extraStudents > 0) {
            int[] a = pq.poll();
            a[0]++;
            a[1]++;
            pq.add(a);
            // int[] b = pq.poll();
            // System.out.println(Arrays.toString(a));
            // System.out.println(Arrays.toString(b));
            // double a1 = (a[0] + 1) * 1.0 / (a[1] + 1) + (b[0]) * 1.0 / (b[1]);
            // double b1 = (a[0]) * 1.0 / (a[1]) + (b[0] + 1) * 1.0 / (b[1] + 1);
            // if (a1 > b1) {
            //     a[0]++;
            //     a[1]++;
            //     pq.add(a);
            //     pq.add(b);
            // } else {
            //     b[0]++;
            //     b[1]++;
            //     pq.add(a);
            //     pq.add(b);
            // }
            extraStudents--;
        }
        
        double res = 0.0;
        while (pq.size() > 0) {
            int[] a = pq.poll();
            // System.out.println(Arrays.toString(a));
            res += 1.0*a[0]/a[1];
        }
        return res/n;
    }
}
