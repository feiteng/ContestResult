class Solution {
    public double maxAverageRatio(int[][] cs, int e) {
        
        int len =  cs.length;
        
        
        
        
        
PriorityQueue<long[]> q = new PriorityQueue<>((x,y)->{

                return -Long.compare((x[1] - x[0])*(y[1]*(y[1]+1)), (y[1] - y[0])*(x[1]*(x[1]+1)));

            });
        
        for(int[] c:cs){
            q.offer(new long[]{c[0],c[1]});
        }
        
        
        while(e>0){
            long m[] = q.poll();
            m[0]++;m[1]++;
            q.offer(m);
            e--;
        }
        double tot = 0;
        while(q.size()>0){
            long m[] = q.poll();
            
            tot += 1.0*m[0]/m[1];
            
            
        }
        return tot/len;
        
    }
}
