class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> minHeap = new PriorityQueue<>((a, b) -> Double.compare(delta(b), 
delta(a)));
        for (int[] c : classes) {
            minHeap.offer(c);
        }
        
        for (int i = 0; i < extraStudents; i++) {
            int[] cur = minHeap.poll();
            cur[0]++;
            cur[1]++;
            minHeap.offer(cur);
        }
        
        double rate = 0;
        while (!minHeap.isEmpty()) {
            int[] cur = minHeap.poll();
            rate += (double) cur[0] / cur[1];
        }
        return rate / classes.length;
    }
    
    private double delta(int[] a) {
        return (double) (a[0] + 1) / (a[1] + 1) - (double) a[0] / a[1];
    }
}
