class Solution {
 public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<long[]> q = new PriorityQueue<>((x, y) -> {
            long a0 = x[0];
            long b0 = x[1];
            
            long a1 = y[0];
            long b1 = y[1];
            
            long l = ((a0 + 1) * b0 - (b0 + 1) * a0) * (b1 + 1) * b1;
            long r = ((a1 + 1) * b1 - (b1 + 1) * a1) * (b0 + 1) * b0;
            if (l > r) return -1;
            else if (l < r) return 1;
            else return 0;
        });
        
        
for (int i = 0; i < classes.length; i++) {
            q.add(new long[] {(long) classes[i][0], (long)classes[i][1]});
        }
        
        for (int i = 1; i <= extraStudents; i++) {
            long[] node = q.poll();
            q.add(new long[] {node[0] + 1, node[1] + 1});
        }
        
        int n = classes.length;
        double total = 0.0;
        
while (!q.isEmpty()) {
            long[] node = q.poll();
            total += node[0] * 1.0 / (node[1] * 1.0);
        }
        return total / n;
        
    }
}
