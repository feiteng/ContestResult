import java.util.*;
import java.util.regex.Pattern;

import static java.lang.System.out;
import static java.lang.Math.*;


class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> q = new PriorityQueue<>((a, b) -> {
            double a1 = 1.0 * a[0] / a[1], b1 = 1.0 * b[0] / b[1];
            double _a1 = 1.0 * (a[0] + 1) / (a[1] + 1), _b1 = 1.0 * (b[0] + 1) / (b[1] + 1);
            double d1 = _a1 - a1, d2 = _b1 - b1;
            if (d1 > d2) {
                return -1;
            } else if (d1 < d2) {
                return 1;
            } else {
                return 0;
            }
        });
        
for (int[] cls : classes) {
            q.add(cls);
        }
        for (int i = 0; i < extraStudents; ++i) {
            int[] t = q.poll();
            q.add(new int[]{t[0] + 1, t[1] + 1});
        
}
        double ans = 0;
        while (!q.isEmpty()) {
            int[] t = q.poll();
            ans += 1.0 * t[0] / t[1];
        }
        return ans / classes.length;
    }
}
