class Solution {
    public double maxAverageRatio(int[][] classes, int eS) {
        PriorityQueue <double[]> pq = new PriorityQueue<double[]>( (d1, d2) -> (d1[0] > d2[0] ? -1 : 1));
        double sum = 0.0;
        for(int[] c : classes) {
            sum += 1.0 * c[0] / c[1];
            pq.add(new double[]{1.0 * (c[0] + 1) / (c[1] + 1) - 1.0 * c[0] / c[1], c[0], c[1]});
        }
        while(eS > 0) {
            //for(double[] dd : pq) System.out.println(dd[0] + " " + dd[1] + " " + dd[2]);
            double[] tmp = pq.poll();
            eS--;
            pq.add(new double[]{1.0 * (tmp[1] + 2) / (tmp[2] + 2) - 1.0 * (tmp[1] + 1)  / (tmp[2] + 1), tmp[1] + 1, tmp[2] + 1});
            sum += tmp[0];
        }
        return sum / classes.length;
    }
}
