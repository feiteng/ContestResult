class Solution {
    class Pair {
        int x, y;
        public Pair(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
    
public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Pair> pq = new PriorityQueue<>((a, b) -> {
            // a.x/a.y , b.x/b.y
            // (a.x+1)/(a.y+1)-a.x/a.y = (a.y-a.x)/(a.y + 1)a.y
            double first = (a.y - a.x + .0) / (a.y * a.y + a.y);
            double second = (b.y - b.x + .0) / (b.y * b.y + b.y);
            return Double.compare(-first, -second);
        });
        for (int[] cl : classes) {
            Pair pair = new Pair(cl[0], cl[1]);
            
pq.add(pair);
        }
        for (int i = 0; i < extraStudents; i++) {
            Pair p = pq.poll();
            pq.add(new Pair(p.x + 1, p.y + 1));
        
}
        double res = 0;
        while (!pq.isEmpty()) {
            Pair p = pq.poll();
            res += (p.x * 1.0) / p.y;
        }
        return res / classes.length;
    }
}
