class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<double[]>((a, b) -> (Double.compare(((b[0] + 1) / (b[1] + 1))  - (b[0] / b[1] ) , ((a[0] + 1) / (a[1] + 1)) - (a[0] / a[1]))));
        for (int[] c : classes) {
            pq.offer(new double[] {(double)c[0], (double)c[1]});
        }
        for (int i = 0; i < extraStudents; i++) {
            double[] cur = pq.poll();
            // System.out.println(Arrays.toString(cur));
            pq.offer(new double[] {cur[0] + 1, cur[1] + 1});
        }
        
        double res = 0;
        while (!pq.isEmpty()) {
            double[] cur = pq.poll();
            res += cur[0] / cur[1];
            // System.out.println(cur[0] / cur[1]);
        }
        return res / ((double)classes.length);
    }
}
