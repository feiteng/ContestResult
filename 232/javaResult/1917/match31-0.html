class Solution {
        public double maxAverageRatio(int[][] classes, int extraStudents) {
            Queue<int[]> pq = new PriorityQueue<>((int[] c1, int[]c2) -> Double.compare((c2[0]  + 1) * 1.0 / (c2[1] + 1) - (c2[0] * 1.0 / c2[1]), (c1[0]  + 1) * 1.0 / (c1[1] + 1) - (c1[0] * 1.0 / c1[1])));
            for (int[] c : classes) {
                pq.offer(c);
            }
            for (int i = 0; i < extraStudents; i++) {
                int[] c = pq.poll();
                c[0]++;
                c[1]++;
                pq.offer(c);
            }
            double ans = 0.0;
            while (!pq.isEmpty()) {
                int[] c = pq.poll();
                ans += c[0] * 1.0 / c[1];
            }
            return ans / classes.length;
        }
    }
