class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        Queue<int[]> pq = new PriorityQueue<>((a,b) -> {
            double aAvg = a[0] * 1.0 / a[1];
            double aChange = (a[0] + 1) * 1.0 / (a[1] + 1);
            double bAvg = b[0] * 1.0 / b[1];
            
            double bChange = (b[0] + 1) * 1.0 / (b[1] + 1);
            if ((aChange - aAvg) > bChange - bAvg) return -1;
            if ((aChange - aAvg) < bChange - bAvg) return 1;
            return 0;
        });
        for (int[] cls: classes) {
            pq.add(cls);
        }
        while (extraStudents-- > 0) {
            int[] cls = pq.poll();
            
            // System.out.println(cls[0] + " " +  cls[1]);
            cls[0]++;
            cls[1]++;
            pq.add(cls);
        }
        double sum = 0.0;
        while (!pq.isEmpty()) {
            int[] cls = pq.poll();
            sum += cls[0] * 1.0 / cls[1];
        }
        return sum / classes.length;
    }
}
