class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a, b) -> {
            double f = a[0] * 1.0d / a[1];
            double s = b[0] * 1.0d / b[1];
            double f1 = (a[0] + 1) * 1.0d / (a[1] + 1);
            double s1 = (b[0] + 1) * 1.0d / (b[1] + 1);
            double res = (s1 - s) - (f1 - f);
            if(res >= 0) {
                return 1;
            } else {
                return -1;
            }
        });
        for(int[] c : classes) {
            pq.offer(c);
        }
        while(extraStudents > 0) {
            int[] cur = pq.poll();
            cur[0]++;
            cur[1]++;
            pq.offer(cur);
            extraStudents--;
        }
        double res = 0;
        while(!pq.isEmpty()) {
            int[] cur = pq.poll();
            res += cur[0] * 1.0d / cur[1];
        }
        return res / classes.length;
    }
}
