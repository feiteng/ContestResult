class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->{
            double res1 = a[0]/(double)a[1];
            double res2 = b[0]/(double)b[1];
            
            double res11 = (a[0] + 1)/(double)(a[1] + 1);
            double res22 = (b[0] + 1)/(double)(b[1] + 1);
            
            double diff1 = res11 - res1;
            double diff2 = res22 - res2;
            
            //(diff1 > diff2)
            if(Double.compare(diff1, diff2) > 0)
                return -1;
            else
                return 1;
        });
        
        for(int[] c : classes)
            pq.add(c);
        
        while(extraStudents > 0) {
            int[] curr = pq.poll();
            extraStudents--;
            curr[0]++;
            curr[1]++;
            pq.add(curr);
        }
        
        double avg = 0;
        while(!pq.isEmpty()) {
            int[] curr = pq.poll();
            avg += curr[0]/(double)curr[1];
        }
        return avg/classes.length;
    }
}
