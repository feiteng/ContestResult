class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
            public int compare(Integer i, Integer j) {
                double d1 = delta(classes[i][0], classes[i][1]);
                double d2 = delta(classes[j][0], classes[j][1]);
                if (d1 <= d2) {
                    return 1;
                } else {
                    return -1;
                }
            }
        });  
            
        for (int i = 0; i < classes.length; i++) {
            if (classes[i][0] != classes[i][1]) {
                pq.add(i);
            }
        }

        for (int i = 0; i < extraStudents; i++) {
            if (pq.isEmpty()) {
                break;
            }
            int j = pq.remove();
            classes[j][0]++;
            classes[j][1]++;
            pq.add(j);
        }
        double res = 0;
        for (int i = 0; i < classes.length; i++) {
            res += (double)classes[i][0] / (double)classes[i][1];
        }
        return res / (double)classes.length;
    }

    
    private double delta (int m, int n) {
        double dm = (double)m;
        double dn = (double)n;
        return (dn-dm)/(dn*(dn+1));
    }
}
