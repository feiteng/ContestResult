class Solution {
    public double maxAverageRatio(int[][] c, int ex) {
        PriorityQueue<int[]> pq = new PriorityQueue(new Comparator<int[]>(){
            public int compare(int[] a, int[] b) {
                return change(a)>change(b)?-1:1;
            
}
        });
        for (int i = 0; i < c.length; i++) {
            pq.add(c[i]);
        }
        for (int i = 0; i < ex; i++) {
            int[] a = pq.poll();
            a[0]++;
            a[1]++;
            
pq.add(a);
        }
        double pr = 0;
        while (!pq.isEmpty()) {
            int[] a = pq.poll();
            pr += (a[0]+0.0)/a[1];
        }
        pr /= c.length;
        return pr;
    }
    public double change(int[] a) {
        return (a[0]+0.0+1)/(a[1]+1) - (a[0]+0.0)/(a[1]);
    }
}
