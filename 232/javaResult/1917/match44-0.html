import java.util.Comparator;
import java.util.PriorityQueue;

class Solution {
    
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<Pair> p = new PriorityQueue<>(new Comparator<Pair>() {
            @Override
            public int compare(Pair o1, Pair o2) {
                double add1 = (o1.top + 1) / (o1.bot + 1) - o1.top / o1.bot;
                double add2 = (o2.top + 1) / (o2.bot + 1) - o2.top / o2.bot;
                if (add1 > add2) {
                    return -1;
                } else if (add1 == add2) {
                    return 0;
                }
                return 1;
            }
        });
        for (int[] aClass : classes) {
            p.add(new Pair(aClass[0], aClass[1]));
        }
        assert !p.isEmpty();
        for (int i = 0; i < extraStudents; i++) {
            Pair poll = p.poll();
            poll.bot++;
            poll.top++;
            p.add(poll);
        }
        double res = 0;
        for (Pair poll : p) {
            res += poll.top / poll.bot;
        }
        return res /= classes.length;
    }
}

class Pair {
    double top;
    double bot;

    public Pair(double top_, double bot_) {
        top = top_;
        bot = bot_;
    }
}
