class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<double[]> pq = new PriorityQueue<>(
            (a,b) -> 
            Double.compare((b[0] + 1)/ (b[1] + 1) - b[0] / b[1], (a[0] + 1)/ (a[1] + 1) - a[0] / a[1]));
        
        for(int[] clazz : classes) {
            pq.add(new double[] {1d * clazz[0], 1d * clazz[1]});
        }
        
        for(int i = 0; i < extraStudents; i++) {
            double[] highest = pq.poll();
            
            pq.add(new double[] {highest[0] + 1, highest[1] + 1});
        }
        
        double out = 0;
        
        while(!pq.isEmpty()) {
            double[] nxt = pq.poll();
            
            out += nxt[0] / nxt[1];
        }
        
        return out / classes.length;
    }
}
