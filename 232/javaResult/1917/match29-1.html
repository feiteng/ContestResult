class Solution {
    public double maxAverageRatio(int[][] classes, int es) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((int[] a, int[] b) -> {
            double d1 = (a[0] + 1) * 1.0 / (a[1] + 1) - a[0] * 1.0 / a[1];
            double d2 = (b[0] + 1) * 1.0 / (b[1] + 1) - b[0] * 1.0 / b[1];
            return Double.compare(d2, d1);
        
});
        for (int[] c : classes)
            pq.add(c);
        while (es > 0) {
            int[] e = pq.poll();
            e[0]++;
            e[1]++;
            pq.add(e);
            es--;
        }
        double ans = 0;
        while (!pq.isEmpty()) {
            int[] e = pq.poll();
            ans += e[0] * 1.0 / e[1];
        }
        return ans / classes.length;
    }
}
