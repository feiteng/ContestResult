class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        Queue<int[]> queue = new PriorityQueue<>((a, b) -> ((double)(b[1] - b[0]) / b[1] / (b[1] + 1) - (double)(a[1] - a[0]) / a[1] / (a[1] + 1) >= 0 ? 1 : -1));
        for (int[] _class : classes) {
            queue.offer(_class);
        }
        while (extraStudents > 0) {
            int[] _class = queue.poll();
            queue.offer(new int[]{_class[0] + 1, _class[1] + 1});
            extraStudents--;
        }
        double res = 0;
        for (int[] _class : queue) {
            res += (double)_class[0] / _class[1] / classes.length;
        }
        return res;
    }
}
