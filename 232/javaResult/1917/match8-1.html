class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> pq=new PriorityQueue<>((o1,o2)->{   if ((double)(o1[0] + 1)/(o1[1] + 1) - (double)o1[0]/o1[1] > (double) (o2[0] + 1)/(o2[1] + 1) - (double)o2[0]/o2[1]) {
                    return -1;
                }
                return 1;
});
        for(int[] i:classes){
            pq.add(i);
        }
        while(extraStudents>0){
            int[] cur=pq.poll();
           // System.out.println(Arrays.toString(cur));
          
            cur[0]++;
            cur[1]++;
              extraStudents--;
            pq.add(cur);
        }
        int n=classes.length;
        double res=0;
        
while(!pq.isEmpty()){
            int[] cur=pq.poll();
            res+=((double)cur[0])/(cur[1]);
            
        }
        return (res/n);
    }
}
