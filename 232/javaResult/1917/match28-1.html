class Solution {
    public double maxAverageRatio(int[][] classes, int extraStudents) {
        PriorityQueue<int[]> heap = new PriorityQueue<>((a, b) -> Double.compare(((b[0] + 1) * 1.0 / (b[1] + 1) - b[0] * 1.0 / b[1]), ((a[0] + 1) * 1.0 / (a[1] + 1) - a[0] * 1.0 / a[1])));
        
        
for (int[] c : classes) {
            heap.offer(c);
        }
        
        while (extraStudents > 0) {
            int[] min = heap.poll();
            min[0]++;
            min[1]++;
            heap.offer(min);
            extraStudents--;
        }
        
        double res = 0;
        
int n = classes.length;
        for (int[] c : heap) {
            res += c[0] * 1.0 / c[1] / n;
        }
        return res;
    }
}
