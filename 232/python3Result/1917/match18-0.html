class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for passes,total in classes:
            heapq.heappush(heap, ( passes / total - (passes + 1) / (total + 1),total, passes))
        
        for _ in range(extraStudents):
            _, total, passes = heapq.heappop(heap)
            total += 1
            passes += 1
            heapq.heappush(heap, ( passes / total - (passes + 1) / (total + 1) ,total, passes))
        return sum([passes / total for _, total, passes in heap]) / len(heap)
