class Solution:
    def maxAverageRatio(self, classes: List[List[int]], k: int) -> float:
        heap = []
        for a, b in classes:
            heapq.heappush(heap, (a/b - (a + 1) / (b + 1), a, b))
            
        for _ in range(k):
            rt, a, b = heapq.heappop(heap)
            heapq.heappush(heap, ((a + 1) / (b + 1) - ((a + 2) / (b + 2)), a + 1, b + 1))
            
        ret = 0.0
        for rt, a, b in heap:
            ret += a / b
            
        return ret / len(classes)
