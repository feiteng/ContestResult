import heapq
import operator


class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        que = []
        for a, b in classes:
            heapq.heappush(que, (-(a + 1) / (b + 1) + a / b, a, b))
        for _ in range(extraStudents):
            _, a, b = heapq.heappop(que)
            a += 1
            b += 1
            heapq.heappush(que, (-(a + 1) / (b + 1) + a / b, a, b))
        res = 0
        for _, a, b in que:
            res += a / b
        return res / len(classes)
