class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        
        heap = []
        
        for p, t in classes:
            heapq.heappush(heap, (-((p + 1)/(t + 1) - p/t), p, t))
            
        for _ in range(extraStudents):
            
            ratio, p, t = heapq.heappop(heap)
            
            heapq
.heappush(heap, (-((p + 2)/(t + 2) - (p + 1)/(t + 1)), p + 1, t + 1))
            
        result = 0
        for _, p, t in heap:
            result += p / t
        
        return result/len(heap)
