class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for p, t in classes:
            value = (p + 1)/(t+1) - p/t
            heapq.heappush(heap,(-value,p,t))
        for _ in range(extraStudents):
            v, p, t = heapq.heappop(heap)
            value = (p + 2)/(t+2) - (p+1) / (t+1)
            heapq.heappush(heap, (-value, p+1, t+1))
        return sum(elem[1]/elem[2] for elem in heap) / len(heap)
