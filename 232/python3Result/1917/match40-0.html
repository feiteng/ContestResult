import heapq
class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for passNum,total in classes:
            heapq.heappush(heap, [passNum / total - (passNum + 1) / (total + 1), passNum, total])
        for i in range(extraStudents):
            maxDiff,passNum,total = heapq.heappop(heap)
            passNum += 1
            total += 1
            heapq.heappush(heap, [passNum / total - (passNum + 1) / (total + 1), passNum, total])
        sumVal = 0
        for maxDiff,passNum,total in heap:
            sumVal += (passNum / total)
        return sumVal / len(heap)
