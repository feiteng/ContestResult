class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for pi, ti in classes:
            heapq.heappush(heap, (pi/ti-(pi+1)/(ti+1), pi, ti))
        
        
        for _ in range(extraStudents):
            add_rate, pi, ti = heapq.heappop(heap)
            pi+=1
            ti+=1
            heapq.heappush(heap, (pi/ti-(pi+1)/(ti+1), pi, ti))
        
        ret = 0
        for add_rate, pi, ti in heap:
            ret += pi/ti
        return ret/len(heap)
