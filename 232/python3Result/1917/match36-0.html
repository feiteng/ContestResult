class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for a, b in classes:
            heapq.heappush(heap, ((a-b) / b/(b+1) , b))
        for i in range(extraStudents):
            r, b = heapq.heappop(heap)
#            print('r,b', r,b)
            newr = r * b / (b+2)
            heapq.heappush(heap, (newr, b+1 ) )
        rst = 0
#        print('heap', heap)
        for r, b in heap:
            rst += 1 + r * (b+1)
        return rst / len(heap)
