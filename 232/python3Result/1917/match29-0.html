class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        pq = []
        for i in range(len(classes)):
            p, t = classes[i]
            heapq.heappush(pq, (p/t-(p+1)/(t+1), p, t, i))
        for st in range(extraStudents):
            _, p, t, i = heapq.heappop(pq)
            p += 1
            t += 1
            heapq.heappush(pq, (p/t-(p+1)/(t+1), p, t, i))
        res = 0
        for _, p, t, i in pq:
            res += p/t
        res /= len(classes)
        return res
