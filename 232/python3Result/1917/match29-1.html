class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for i in range(len(classes)):
            p, t = classes[i]
            heapq.heappush(heap, ((p/t) -(p + 1) / (t + 1) ,t, p))
        
        for i in range(extraStudents):
            _, t, p = heapq.heappop(heap)
            t, p = t + 1, p + 1
            heapq
.heappush(heap, ((p/t) -(p + 1) / (t + 1), t, p))
        
        #print(heap)
        res = 0
        for _, t, p in heap:
            res += p / t
        res /= len(classes)
        return res
