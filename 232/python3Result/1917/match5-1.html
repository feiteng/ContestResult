class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        pq = []
        for x, y in classes:
            heapq.heappush(pq, (-(y-x)/y/(y+1), x, y))
        #print()
        for _ in range(extraStudents):
            _, x, y = heapq.heappop(pq)
            #print(x, y)
            x += 1
            y += 1
            heapq.heappush(pq, (-(y-x)/y/(y+1), x, y))
        
        return sum(x/y for r, x, y in pq) / len(pq)
