class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        classes = [[(t-p)/(t*(t+1)), p, t] for p, t in classes]
        classes.sort()
        for _ in range(extraStudents):
            _, p, t = classes.pop()
            p += 1
            t += 1
            bisect.insort(classes, [(t-p)/(t*(t+1)), p, t])
        
        return sum(p/t for _, p, t in classes) / len(classes)
