class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        pq = []
        for p, t in classes:
            heapq.heappush(pq, (p/t - (p+1)/(t+1), p/t, p, t))
        
        for _ in range(extraStudents):
            _, _, p, t = heapq.heappop(pq)
            p+=1
            t+=1
            heapq.heappush(pq, (p/t - (p+1)/(t+1), p/t, p, t))
        
        # print(pq)
        
        return sum(x[1] for x in pq) / len(pq)
