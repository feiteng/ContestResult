class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extra: int) -> float:
        pq = []
        for ok, total in classes:
            heapq.heappush(pq, (-((ok + 1) / (total + 1) - ok / total), ok, total))
        for c in range(extra):
            delta, ok, total = heapq.heappop(pq)
            ok += 1
            total += 1
            heapq.heappush(pq, (-((ok + 1) / (total + 1) - ok / total), ok, total))
        ratio_sum = 0
        for delta, ok, total in pq:
            ratio_sum += (ok / total)
        return ratio_sum / len(classes)
