class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for a, b in classes:
            heap.append([(a-b)/(b*(b+1)), a, b])
        heapq.heapify(heap)
        for x in range(1, extraStudents+1):
            _, a, b = heapq.heappop(heap)
            b += 1
            a += 1
            heapq.heappush(heap, [(a-b)/(b*(b+1)), a, b])
        
        total = 0
        for _, a, b in heap:
            total += a/b
        return total/len(classes)
