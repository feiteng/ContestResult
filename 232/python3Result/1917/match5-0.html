class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap = []
        for p, t in classes:
            heapq.heappush(heap, (-((p + 1) / (t + 1) - p / t), p, t))
        for i in range(extraStudents):
            _, p, t = heapq.heappop(heap)
            p += 1
            t += 1
            heapq.heappush(heap, (-((p + 1) / (t + 1) - p / t), p, t))
        return sum(p / t for _, p, t in heap) / len(heap)
