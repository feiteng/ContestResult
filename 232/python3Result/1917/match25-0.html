class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        heap=[]
        for pas,total in classes:
            heapq.heappush(heap,[-((pas+1)/(total+1)-pas/total),pas,total])

        for i in range(extraStudents):
            rate,pas,total=heapq.heappop(heap)
            heapq.heappush(heap,[-((pas+2)/(total+2)-(pas+1)/(total+1)),(pas+1),(total+1)])

        passrate_total=0
        while heap:
            rate,pas,total=heapq.heappop(heap)
            passrate_total+=(pas/total)
        return passrate_total/len(classes)
