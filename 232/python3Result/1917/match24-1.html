class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        H = [[(p-t)/(t*t+t), p, t] for p, t in classes]
        heapq.heapify(H)
        for _ in range(extraStudents):
            ratio, p, t = heapq.heappop(H)
            heapq.heappush(H, [(p-t)/(t*t+3*t+2), p+1, t+1])
            
        #print (H)
        return sum(p/t for val, p, t in H)/len(H)
