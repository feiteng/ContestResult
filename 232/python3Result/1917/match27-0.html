class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extra: int) -> float:
        val = [[(x-y)/(y*(y+1)), x, y] for x,y in classes]
        heapq.heapify(val)
        for _ in range(extra):
            _, x, y = heapq.heappop(val)
            x += 1
            y += 1
            heapq.heappush(val, [(x-y)/(y*(y+1)), x, y])
        return sum(x/y for _,x,y in val) / len(val)
