class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extra: int) -> float:
        heap = []
        for p, t in classes:
            heap.append((p / t - (p + 1) / (t + 1), p, t))
        heapq.heapify(heap)
        for _ in range(extra):
            inc, p, t = heapq.heappop(heap)
            p += 1
            t += 1
            heapq.heappush(heap, (p / t - (p + 1) / (t + 1), p, t))
        
        res = 0
        for _, p, t in heap:
            res += p / t
        return res / len(heap)
