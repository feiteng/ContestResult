class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        new_classes = [(-(t-p)/(t*(t+1)),t,p) for p,t in classes]
        heapify(new_classes)
        for i in range(extraStudents):
            comp,t,p = heappop(new_classes)
            t += 1
            p += 1
            heappush(new_classes, (-(t-p)/(t*(t+1)), t, p))
            #print(new_classes)
        
        
        return sum(p/t for _,t,p in new_classes)/len(classes)
