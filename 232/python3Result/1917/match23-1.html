import heapq

class Solution:
    def maxAverageRatio(self, classes: List[List[int]], k: int) -> float:
        p = []
        for a, b in classes:
            p.append((a / b - (a + 1) / (b + 1), a, b))

        heapq.heapify(p)
        for _ in range(k):
            _, a, b = heapq.heappop(p)
            a += 1
            b += 1
            heapq.heappush(p, (a / b - (a + 1) / (b + 1), a, b))
        tot = 0
        for _, a, b in p:
            tot += a / b
        return tot / len(p)
