class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        # number of students
        pq = []
        for pas, tot in classes:
            pq.append([-((pas+1)/(tot+1) - pas/tot), tot, pas])
        
        
        heapq.heapify(pq)
        for i in range(extraStudents):
            val, tot, pas = heapq.heappop(pq)
            tot += 1
            pas += 1
            
            heapq.heappush(pq, [-((pas+1)/(tot+1) - pas/tot), tot, pas])
        
        res = 0
        for val, tot, pas in pq:
            res += pas / tot
        
        return res / len(classes)
