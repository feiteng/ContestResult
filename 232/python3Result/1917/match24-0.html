class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        profile = [
            [(p/t) - (p+1)/(t+1), p, t]
            for p, t in classes
        ]
        heapq.heapify(profile)
        for _ in range(extraStudents):
            inc, p, t = heapq.heappop(profile)
            heapq.heappush(profile, [(p+1)/(t+1) - (p+2)/(t+2), p+1, t+1])
        return sum(p/t for _, p, t in profile) /  len(profile)
