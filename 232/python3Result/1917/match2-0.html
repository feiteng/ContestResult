class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        h = [(-(b - a) / (b * b + b), a, b) for a, b in classes]
        heapify(h)
        for i in range(extraStudents):
            _, a, b = heappop(h)
            a += 1
            b += 1
            heappush(h, (-(b - a) / (b * b + b), a, b))
        return sum(a / b for _, a, b in h) / len(classes)
