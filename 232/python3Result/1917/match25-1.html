class Solution:
    def maxAverageRatio(self, classes: List[List[int]], es: int) -> float:
        q = []
        
        for a, b in classes:
            heapq.heappush(q, [(-b+a)/(b*(b+1)), a, b])
        
        for _ in range(es):
            _, a, b = heapq.heappop(q)
            heapq.heappush(q, [(-b+a)/((b+2)*(b+1)), a+1, b+1])
        
        tot = 0
        
        while q:
            _, a, b = heapq.heappop(q)
            tot += a/b
        
        return tot / len(classes)
