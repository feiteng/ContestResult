import heapq
class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        
        heap = []
        
        for u, v in classes:
            
            heapq.heappush(heap, (-( (u + 1) / (v + 1) - (u / v) ), u, v))
            
        for i in range(extraStudents):
            
            r, u, v = heapq.heappop(heap)
            
            # print (r, u, v)
            
            u += 1
            v += 1
            
            heapq.heappush(heap, ( -( (u + 1) / (v + 1) - (u / v) ) , u, v))
            
        total = 0
        
        for r, u, v in heap:
            total += u / v
            
        return total / len(heap)
