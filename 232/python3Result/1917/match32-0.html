import heapq
class Solution:
    def maxAverageRatio(self, classes: List[List[int]], extraStudents: int) -> float:
        q = []
        for b, a in classes:
            heapq.heappush(q, ((b - a) / (a * (a + 1)), a, b))
        for i in range(extraStudents):
            r, a, b = heapq.heappop(q)
            a += 1
            b += 1            
            heapq.heappush(q, ((b - a) / (a * (a + 1)), a, b))
        return sum(b / a for _, a, b in q) / len(q)
