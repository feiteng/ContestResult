class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i, j 
= k-1, k+1
        res = nums[k]
        mins = nums[k]
        while i >= 0 and j < len(nums):
            if nums[j] >= nums[i]:
                mins = min(mins, nums[j])
                res = max(res, mins * (j-i))
                j += 1
            else:
                mins = min(mins, nums[i])
                res = max(res, mins * (j - i))
                i -= 1
        while i >= 0:
            mins = min(mins, nums[i])
            res = max(res, mins * (j - i))
            i -= 1
        while j < len(nums):
            mins = min(mins, nums[j])
            res = max(res, mins * (j-i))
            j += 1
        return res
