class Solution:
    def maximumScore(self, nums, k: int) -> int:
        n = len(nums)
        #take care of k = 0 and n - 1 case last
        
        
        #check left, right, and expand mid
        
    
        ans = nums[k]
        
        k = k
        i = k - 1
        j = k + 1
        size = 1
        curr_min = nums[k]
        while i >= 0 or j < n:
            if i >= 0 and j < n:
                
if nums[j] > nums[i]:
                    curr_min = min(curr_min, nums[j])
                    j += 1
                else:
                    curr_min = min(curr_min, nums[i])
                    i -= 1
            elif i >= 0:
                curr_min = min(curr_min, nums[i])
                i -= 1   
            else:
                curr_min 
= min(curr_min, nums[j])
                j += 1        
            size += 1
            ans = max(ans, curr_min*size)
        
        
        
        
        return ans
