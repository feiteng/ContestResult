class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l = len(nums)
        start,end = k,k
        mini = nums[k]
        res = 1*mini
        while start > 0 or end < l-1:
            # print(start,end)
            if start > 0 and end < l-1:
                if nums[start-1] > nums[end + 1]:
                    start -= 1
                    mini = min(mini,nums[start])
                    res = max(res,(end-start+1)*mini)
                else:
                    end += 1
                    mini = min(mini,nums[end])
                    res = max(res,(end-start+1)*mini)
            elif start > 0:
                start -= 1
                mini = min(mini,nums[start])
                res = max(res,(end-start+1)*mini)
            else:
                end += 1
                mini = min(mini,nums[end])
                res = max(res,(end-start+1)*mini)
        return res
