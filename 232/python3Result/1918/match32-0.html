class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i, j, n = k - 1, k + 1, len(nums)
        res = nums[k]
        mvalue = nums[k]
        while i >= 0 or j < n:
            if i >= 0 and j < n:
                if nums[i] > nums[j]:
                    mvalue = min(nums[i], mvalue)
                    i -= 1
                else:
                    mvalue = min(nums[j], mvalue)
                    j += 1
            elif i >= 0:
                mvalue = min(nums[i], mvalue)
                i -= 1
            else:
                mvalue =  min(nums[j], mvalue)
                j += 1
            res = max(res, mvalue * (j - i-1))
        
        return res
