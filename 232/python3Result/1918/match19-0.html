class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        ans = nums[k]
        left = k
        right = k
        now = nums[k]
        while left > 0 or right < len(nums) - 1:
            if right == len(nums) - 1:
                left -= 1
                now = min(nums[left], now)
                ans = max(ans, (right - left + 1) * now)
            elif left == 0:
                right += 1
                now = min(nums[right], now)
                ans = max(ans, (right - left + 1) * now)
            elif nums[left - 1] > nums[right + 1]:
                left -= 1
                now = min(nums[left], now)
                ans = max(ans, (right - left + 1) * now)
            else:
                right += 1
                now = min(nums[right], now)
                ans = max(ans, (right - left + 1) * now)
        return ans
