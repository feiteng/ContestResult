class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i=j=k
        g = nums[k]*(j-i+1)
        ml=mr
=nums[k]
        while i != 0 or j != len(nums)-1:
            if i > 0 and j < len(nums)-1:
                if nums[j+1] > nums[i-1]:
                    j += 1
                    mr = min(mr,nums[j])
                else:
                    i -= 1
                    ml = min(ml,nums[i])
            elif i > 0:
                i -= 1
                ml = min(ml,nums[i])
            else:
                j += 1
                mr = min(mr,nums[j])
            g = max(g, min(ml,mr)*(j-i+1))
        return g
