class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l = len(nums)
        i = j = k
        m = nums[k]
        ans = m
        while j - i < l - 1:
            while i > 0 and nums[i-1] >= m:
                i -= 1
            while j < l-1 and nums[j+1] >= m:
                j += 1
            ans = max(ans, m*(j-i+1))
            
            if j - i + 1 == l:
                return ans
            elif i == 0:
                j += 1
                m = nums[j]
            elif j == l-1:
                i -= 1
                m = nums[i]
            elif nums[i-1] > nums[j+1]:
                i -= 1
                m = nums[i]
            else:
                j += 1
                m = nums[j]
            
            ans = max(ans, m*(j-i+1))
            
        return ans
