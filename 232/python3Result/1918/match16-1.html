class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i = k
        j = k
        n = len(nums)
        min_ = nums[k]
        
        res = min_ * (j-i+1)
        
while i-1>=0 and j+1<n:
            if nums[i-1]>nums[j+1]:
                i -= 1
                min_ = min(min_, nums[i])
            else:
                j += 1
                min_ = min(min_, nums[j])
            res = max(res, min_ * (j-i+1))
            
        while i-1>=0:
            i -= 1
            min_ = min(min_, nums[i])
            res = max(res, min_ * (j-i+1))
            
        while j+1<n:
            j += 1
            min_ = min(min_, nums[j])
            res = max(res, min_ * (j-i+1))
            
        return res
