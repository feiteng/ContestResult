class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        nums.append(0)
        stack = [(0, -1)]
        res = 0
        for i, num in enumerate(nums):
            while stack[-1][0] > num:
                height, j = stack.pop()
                if i > k and stack[-1][1] < k:
                    res = max(res, height * (i - stack[-1][1] - 1))
            stack.append((num, i))
            #print(stack)
        return res
