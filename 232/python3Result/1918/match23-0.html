class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        
        minval = nums[k]
        i = j = k
        max_val = minval * (j - i + 1)
        
        while True:
            
            if i > 0 and j < len(nums) - 1:
                
                if nums[i - 1] > nums[j + 1]:
                    i -= 1
                    minval = min(minval, nums[i])
                else:
                    j += 1
                    minval = min(minval, nums[j])
            
            elif i > 0:
                i -= 1
                minval = min(minval, nums[i])
                
            elif j < len(nums) - 1:
                j += 1
                minval = min(minval, nums[j])
            
            else:
                break
            
            max_val = max(max_val, minval * (j - i + 1))
        
        return max_val
