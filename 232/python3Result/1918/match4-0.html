class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        mi = nums[k]
        res = mi
        l = r = k
        while l > 0 and r < n - 1:
            if nums[l - 1] > nums[r + 1]:
                l -= 1
                mi = min(nums[l], mi)
            else:
                r += 1
                mi = min(nums[r], mi)
            res = max(res, mi * (r - l + 1))

            
        while l > 0:
            l -= 1
            mi = min(nums[l], mi)
            res = max(res, mi * (r - l + 1))
        while r < n - 1:
            r += 1
            mi = min(nums[r], mi)
            res = max(res, mi * (r - l + 1))
        return res
