class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l, r = k, k
        n = len(nums)
        ans, m = nums[k], nums[k]
        while l - 1 >= 0 or r + 1 < n:
            if l - 1 >= 0 and r + 1 < n:
                
if nums[l-1] >= nums[r + 1]:
                    l -= 1
                    m = min(nums[l], m)
                else:
                    r += 1
                    m = min(nums[r], m)
            elif l - 1 >= 0:
                l -= 1 
                m = min(nums[l], m)
            else:
                r += 1
                m = min(nums[r], m)
            ans = max(ans, m * (r - l + 1))
        return ans
