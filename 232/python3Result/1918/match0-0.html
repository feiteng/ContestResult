class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        nums.append(-1)
        stack = [[-2, -1]]
        res = 0
        for i, x in enumerate(nums):
            while x < stack[-1][0]:
                val, idx = stack.pop()
                if k < i and k > stack[-1][1]:
                    res = max(res, val * (i - stack[-1][1] - 1))
            stack.append([x, i])
        return res
