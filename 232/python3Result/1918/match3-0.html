class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l = len(nums)
        res = nums[k]
        left, right = k, k
        cur_min = nums[k]
        
        while left > 0 and right < l - 1:
            if nums[left - 1] <= nums[right + 1]:
                right += 1
                cur_min = min(cur_min, nums[right])
            else:
                left -= 1
                cur_min = min(cur_min, nums[left])
            res = max(res, cur_min * (right - left + 1))
        while left > 0:
            left -= 1
            cur_min = min(cur_min, nums[left])
            res = max(res, cur_min * (right - left + 1))
        while right < l - 1:
            right += 1
            cur_min = min(cur_min, nums[right])
            res = max(res, cur_min * (right - left + 1))
        return res
