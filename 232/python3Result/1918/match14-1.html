class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        be = k
        en = k
        res = nums[k] * 1
        cur_min 
= nums[k]
        
        while be > 0 or en < n - 1:
            if be == 0:
                # add en as it is the only option
                en += 1
                cur_min = min(cur_min, nums[en])
            elif en == n - 1:
                be -= 1
                cur_min = min(cur_min, nums[be])
            else: # pick either be or en
                if nums[be - 1] >= nums[en + 1]:
                    be -= 1
                    cur_min = min(cur_min, nums[be])
                else:
                    en += 1
                    cur_min = min(cur_min, nums[en])
            res = max(res, cur_min * (en - be + 1))
        
        return res
