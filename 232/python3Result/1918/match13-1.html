class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        # score = min * len
        N = len(nums)
        #sliding window expand towards larger number
        curMax = nums[k]
        #expand left first
        left = right = k
        minVal = nums[k]
        
while left-1 >= 0 and right+1 < N:
            if nums[left-1] >= nums[right+1]:
                #expand left
                left -= 1
                minVal = min(minVal,nums[left])
            else:
                #expand right
                right += 1
                minVal = min(minVal,nums[right])
            curMax = max(curMax, minVal * (right-left+1))
        
        #expand the rest of left
        while left-1 >= 0:
            left -= 1
            minVal = min(minVal,nums[left])
            curMax = max(curMax, minVal * (right-left+1))
        
        #expand the rest of right
        while right+1 < N:
            right += 1
            minVal = min(minVal,nums[right])
            curMax = max(curMax, minVal * (right-left+1))
        
        return curMax
