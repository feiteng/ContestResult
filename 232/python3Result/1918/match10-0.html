from typing import List
import bisect
class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        stk = []
        left = [-1 for _ in range(n)]
        for i in range(n):
            while stk and nums[stk[-1]] >= nums[i]:
                stk.pop()
            if not stk:
                left[i] = -1
            else:
                left[i] = stk[-1]
            stk.append(i)
        stk = []
        right = [0 for _ in range(n)]
        for i in range(n - 1, -1, -1):
            while stk and nums[stk[-1]] >= nums[i]:
                stk.pop()
            if not stk:
                right[i] = n
            else:
                right[i] = stk[-1]
            stk.append(i)
        res = 0
        for i in range(n):
            if left[i] + 1 <= k and k <= right[i] - 1:
                res = max(res, nums[i] * (right[i] - left[i] - 1))
        return res
