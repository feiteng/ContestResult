class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l, r = k, k
        n = len(nums)
        max_score = nums[k]
        min_val = nums[k]

        while l >= 0 and r < n:
            if r + 1 < n and l - 1 >= 0:
                
if nums[r + 1] >= nums[l - 1]:
                    r += 1
                    min_val = min(min_val, nums[r])
                else:
                    l -= 1
                    min_val = min(min_val, nums[l])
            elif r + 1 < n:
                r += 1
                min_val = min(min_val, nums[r])
            elif l - 1 >= 0:
                l 
-= 1
                min_val = min(min_val, nums[l])
            else:
                break

            max_score = max(max_score, min_val * (r - l + 1))
        return max_score
