class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        l,r = k,k
        minn = nums[k]
        res = float('-inf')
        while l>=0 and r<=n-1:
            if nums[l] < nums[r]:
                minn = min(minn, nums[r])
                res = max(res, minn*(r-l))
                r +=1
            elif nums[l] > nums[r]:
                minn = min(minn, nums[l])
                res = max(res, minn*(r-l))
                l -=1
            else:
                minn = min(minn, nums[r])
                res = max(res, minn*(r-l+1))
                r +=1
                l -=1
        while r<=n-1:
            minn = min(minn, nums[r])
            res = max(res, minn*(r-l))
            r +=1
        while l >=0:
            minn = min(minn, nums[l])
            res = max(res, minn*(r-l))
            l -=1
        return res
