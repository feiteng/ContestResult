# 包含idx-k的子数组的 min(arr) * len(arr) 最大化
# 2 pointers? from k, always to the larger one
class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        cur_min, cur_score = nums[k], nums[k]
        l, r = k - 1, k + 1
        
        while l >= 0 and r < n:
            # print(cur_min, l, r)
            
if nums[l] < nums[r]:
                cur_min = min(cur_min, nums[r])
                cur_score = max(cur_score, cur_min * (r - l))
                r += 1
            else:
                cur_min = min(cur_min, nums[l])
                cur_score = max(cur_score, cur_min * (r - l))
                l -= 1
        while l >= 0:
            cur_min 
= min(cur_min, nums[l])
            cur_score = max(cur_score, cur_min * (r - l))
            l -= 1
        while r < n:
            cur_min = min(cur_min, nums[r])
            cur_score = max(cur_score, cur_min * (r - l))
            r += 1
        return cur_score
