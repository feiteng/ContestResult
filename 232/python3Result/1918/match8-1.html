class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        
        nums.append(0)
        stack 
= [-1]
        res = 0
        for i, h in enumerate(nums):
            while stack[-1] != -1 and nums[stack[-1]] >= h:
                prev = stack.pop()
                if i > k and stack[-1] < k:
                    res = max(res, nums[prev] * (i - stack[-1] - 1))
            stack.append(i)
        return res
