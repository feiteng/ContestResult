class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        l = k - 1
        r = k + 1
        cur_min = nums[k]
        result = nums[k]
        
        while l >= 0 and r < len(nums):
            if nums[l] >= nums[r]:
                cur_min = min(cur_min, nums[l])
                result = max(result, cur_min * (r - l))
                l -= 1
                
            else:
                cur_min = min(cur_min, nums[r])
                result = max(result, cur_min * (r - l))
                r += 1
                
        while l >= 0:
            cur_min = min(cur_min, nums[l])
            result = max(result, cur_min * (r - l))
            l -= 1
            
        while r < len(nums):
            cur_min = min(cur_min, nums[r])
            result = max(result, cur_min * (r - l))
            r += 1
            
        return result
