class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i = k
        j = k
        N = len(nums)
        cur_min = nums[k]
        res 
= nums[k]
        while i > 0 or j < N - 1:
            if i == 0:
                j += 1
                cur_min = min(cur_min, nums[j])
            elif j == N - 1:
                i -= 1
                cur_min = min(cur_min, nums[i])
            else:
                if nums[i - 1] == nums[j + 1]:
                    cur_min = min(cur_min, nums[i - 1])
                    i -= 1
                    j += 1
                elif nums
[i - 1] < nums[j + 1]:
                    j += 1
                    cur_min = min(cur_min, nums[j])
                else:
                    i -= 1
                    cur_min = min(cur_min, nums[i])
            res = max(res, cur_min * (j - i + 1))
        return res
