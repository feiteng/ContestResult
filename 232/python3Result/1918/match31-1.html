class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        i, j, n = k, k, len(nums)
        _min = nums
[k]
        res = nums[k]
        while i-1 >= 0 and j+1 < n:
            if nums[i-1] <= nums[j+1]:
                _min = min(_min, nums[j+1])
                j += 1
            else:
                _min = min(_min, nums[i-1])
                i -= 1
            res = max(res, (j-i+1)*_min)
        while i-1 >= 0:
            _min = min(_min, nums[i-1])
            i -= 1
            res = max(res, (j-i+1)*_min)
        while j+1 < n:
            _min = min(_min, nums[j+1])
            j += 1
            res = max(res, (j-i+1)*_min)
        return res
