class Solution:
    def maximumScore(self, A: List[int], k: int) -> int:
        A, k = [0]+A+[0], k+1
        stk, ans = [], 0
        for i in range(len
(A)):
            while stk and A[i]<A[stk[-1]]:
                j = stk.pop()
                if stk and stk[-1]+1<=k<=i-1: ans = max(ans, (i-stk[-1]-1)*A[j])
            stk.append(i)
        return ans
