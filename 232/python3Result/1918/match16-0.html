class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        min_val = nums[k]
        ret = nums[k]
        left, right = k, k
        n = len(nums)
        while left > 0 and right < n - 1:
            if left > 0 and nums[left - 1] >= nums[right + 1]:
                left -= 1
                min_val = min(min_val, nums[left])
            else:
                right += 1
                min_val = min(min_val, nums[right])
            ret = max(ret, min_val * (right - left + 1))
        while left>0:
            left-=1
            min_val = min(min_val, nums[left])
            ret = max(ret, min_val * (right - left + 1))
        while right<n-1:
            right += 1
            min_val = min(min_val, nums[right])
            ret = max(ret, min_val * (right - left + 1))
        return ret
