class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n=len(nums)
        nums=[0]+nums+[0]
        stack=[0]
        ans=0
        k+=1
        for i in range(1,n+2):
            while nums[stack[-1]]>nums[i]:
                cur=stack.pop()
                if stack[-1]+1<=k<=i-1:
                    ans=max(ans,nums[cur]*(i-stack[-1]-1))
            stack.append(i)
        return ans
