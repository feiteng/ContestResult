class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        
        i = j = k
        start = nums[k]
        res = nums[k]
        #print(i, j)
        while i - 1 >= 0 and j + 1 < len(nums):
            if nums[i - 1] > nums[j + 1]:
                start = min(start, nums[i - 1])
                i -= 1
            else:
                start = min(start, nums[j + 1])
                j += 1
            #print(start, i, j)
            res = max(res, start * (j - i + 1))
        while i - 1 >= 0:
            start = min(start, nums[i - 1])
            i -= 1
            res = max(res, start * (j - i + 1))
        while j + 1 < len(nums):
            start = min(start, nums[j + 1])
            j += 1
            res = max(res, start * (j - i + 1))
        return res
