class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        left = right = k
        n = len(nums)
        min_val 
= nums[k]
        max_val = 0
        while left > -1 and right < n:
            # print(nums[left:right+1], max_val)
            while left > 0 and nums[left-1] >= min_val:
                left -= 1
            while right < n-1 and nums[right+1] >= min_val:
                right += 1
            max_val = max(max_val, (right-left+1) * min_val)
            if left == 0 and right == n-1:
                break
            elif left == 0:
                right 
+= 1
                min_val = nums[right]
            elif right == n-1:
                left -= 1
                min_val = nums[left]
            elif nums[left-1] < nums[right+1]:
                right += 1
                min_val = nums[right]
            else:
                left -= 1
                min_val = nums[left]
        return max_val
