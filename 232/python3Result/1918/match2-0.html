class Solution:
    def maximumScore(self, nums: [int], k: int) -> int:
        i = k
        j = k
        min_num = nums[k]
        max_score = nums[k]
        while i-1 >= 0 and j+1 < len(nums):
            if nums[i-1] > nums[j+1]:
                i -= 1
                min_num = min(min_num, nums[i])
            else:
                j += 1
                min_num = min(min_num, nums[j])
            max_score = max(max_score, min_num*(j-i+1))
        
        while i-1 >= 0:
            i -= 1
            min_num = min(min_num, nums[i])
            max_score = max(max_score, min_num*(j-i+1))
        
        while j+1 < len(nums):
            j += 1
            min_num = min(min_num, nums[j])
            max_score = max(max_score, min_num*(j-i+1))

        return max_score
