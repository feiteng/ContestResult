class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        
        
        
        i = j = k
        maxScore = nums[k]
        mini = nums[k]
        
        while i != 0 or j != len(nums)-1:
            
            if i > 0 and j < len(nums)-1:
                
                if nums[i-1] >= nums[j+1]:
                    i -= 1
                    mini = min(nums[i], mini)
                else:
                    j += 1
                    mini = min(nums[j], mini)
                
            elif i != 0:
                i -= 1
                mini = min(nums[i], mini)
            else:
                j += 1
                mini = min(nums[j], mini)
                
            
            # print(mini, i, j)
            score = mini * (j-i+1)
            maxScore = max(score, maxScore)
            # print(maxScore)
            
        return maxScore
