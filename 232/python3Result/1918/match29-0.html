class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        # nums1,nums2=nums[:k][::-1],nums[k+1:]
        min_v = nums[k]
        during = 1
        i,j = k-1,k+1
        res = min_v*during
        while i>=0 or j <len(nums):
            if i>=0 and j < len(nums):
                if nums[i]<nums[j]:
                    min_v= min(min_v,nums[j])
                    j+=1

                else:
                    min_v= min(min_v,nums[i])
                    i-=1
            elif i>=0:
                min_v= min(min_v,nums[i])
                i-=1
            elif j<len(nums):
                min_v= min(min_v,nums[j])
                j+=1
            during+=1
            res = max(res,during*min_v)

        return res
