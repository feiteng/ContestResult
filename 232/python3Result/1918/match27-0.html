class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        
        # n = len(nums)
        # ans = 0
        # for i in range(k+1):
        #     min_value = min(nums[i:k+1])
        #     for j in range(k, n):
        #         if nums[j] < min_value:
        #             min_value = nums[j]
        #         ans = max(ans, min_value * (j-i+1))
        # return ans
        
        n = len(nums)
        ans = nums[k]
        
        i = j = k
        min_value = nums[k]
        while i >= 0 and j < n:
            if i == 0 and j == n-1:
                break
            elif i-1 >= 0 and j+1 < n:
                if nums[i-1] > nums[j+1]:
                    i -= 1
                    min_value = min(min_value, nums[i])
                else:
                    j += 1
                    min_value = min(min_value, nums[j])
            elif j+1 < n:
                j += 1
                min_value = min(min_value, nums[j])
            elif i-1 >= 0:
                i -= 1
                min_value = min(min_value, nums[i])
            ans = max(ans, min_value * (j-i+1))
        return ans
