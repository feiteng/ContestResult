class Solution:
    def maximumScore(self, nums: List[int], k: int) -> int:
        n = len(nums)
        res = nums[k]
        left = k
        right = k
        mini = nums[k]
        
while not (left==0 and right==n-1):
            if left-1>=0 and right+1<n:
                if nums[left-1]>=nums[right+1]:
                    left -= 1
                    mini = min(mini, nums[left])
                    res = max(res, mini*(right-left+1))
                else:
                    right += 1
                    mini = min(mini, nums[right])
                    res = max(res, mini*(right-left+1))
            elif left==0:
                right += 1
                mini = min(mini, nums[right])
                res = max(res, mini*(right-left+1))
            else:
                left -= 1
                mini = min(mini, nums[left])
                res = max(res, mini*(right-left+1))
        return res
