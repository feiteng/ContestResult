class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        swapped = []
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                swapped.append(i)
        if not swapped:
            return True 
        if len
(swapped) == 2:
            if s1[swapped[-1]] == s2[swapped[0]] and s1[swapped[0]] == s2[swapped[-1]]:
                return True 
        return False
