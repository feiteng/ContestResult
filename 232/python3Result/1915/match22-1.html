class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        n = len(s1)
        cnt = 0
        arr = []
        for i in range(n):
            if s1[i]!=s2[i]:
                cnt+=1
                arr.append(
i)
        if cnt == 0:
            return True
        elif cnt==2:
            if s1[arr[0]]==s2[arr[1]] and s1[arr[1]]==s2[arr[0]]:
                return True
            else:
                return False
        else:
            return False
