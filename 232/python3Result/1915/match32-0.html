class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        diff = []
        
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                diff.append(i)
        
        if not diff: return True
        if len(diff) == 1 or len(diff) >= 3: return False
        if s1[diff[0]] != s2[diff[1]] or s1[diff[1]] != s2[diff[0]]:
            return False
        return True
