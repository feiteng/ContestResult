class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        pos = list()
        n 
= len(s1)
        for i in range(n):
            if s1[i] != s2[i]:
                pos.append(i)
        if len(pos) == 0:
            return True
        if len(pos) == 2 and s1[pos[0]] == s2[pos[1]] and s1[pos[1]] == s2[pos[0]]:
            return True
        return False
