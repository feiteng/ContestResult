class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        if s1 == s2: return True
        for i in range(len(s1)):
            for j in range(i+1, len(s1)):
                s = list(s1)
                s[i], s[j] = s[j], s[i]
                if ''.join(s) == s2: return True
        return False
