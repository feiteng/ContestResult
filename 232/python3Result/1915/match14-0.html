class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        dif=[]
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                dif.append((s1[i],s2[i]))
        if len(dif) == 0:
            return True
        if len(dif) != 2:
            return False
        return dif[0][0] == dif[1][1] and dif[0][1] == dif[1][0]
