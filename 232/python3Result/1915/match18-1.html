class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        n = len(s1)
        idxes = []
        for i in range(n):
            if s1[i] != s2[i]:
                idxes.append(i)
        
        if not idxes: 
return True
        if len(idxes) == 2:
            return s1[idxes[0]] == s2[idxes[1]] and s2[idxes[0]] == s1[idxes[1]]
        return False
