class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        swap = []
        for i in range(len(s1)):
            if s1[i]!=s2[i]: swap.append(i)
        if not swap or len(swap)==2 and s1
[swap[0]]==s2[swap[1]] and s1[swap[1]]==s2[swap[0]]: return True
        return False
