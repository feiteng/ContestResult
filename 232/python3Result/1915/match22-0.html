class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        n=len(s1)
        count=0
        mismatch=[]
        for i in range(n):
            if s1[i]!=s2[i]:
                count+=1
                mismatch.append((s1[i],s2[i]))
        if count==0:
            return True
        elif count==2:
            if mismatch[0][0]==mismatch[1][1] and mismatch[1][0]==mismatch[0][1]:
                return True
            else:
                return False
        else:
            return False
