class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        cnt = []
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                cnt.append(i)
        if len(cnt) == 0:
            return True
        elif len(cnt) == 2:
            if s1[cnt[0]] == s2[cnt[1]] and s1[cnt[1]] == s2[cnt[0]]:
                return True
        return False
