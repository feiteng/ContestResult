class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        
        d = []
        
        for i in range(len(s1)):
            
            if s1[i] != s2[i]:
                d.append([s1[i],s2[i]])
        
        if len(d) not in {0,2}:
            
return False
        
        if len(d) == 0:
            return True
        
        return d[0][1] == d[1][0] and d[1][1] == d[0][0]
