class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        if len(s1) != len(s2):
            return False
        if s1 == s2:
            return True
        n = len(s2)
        for i in range(n-1):
            for j in range(i+1, n):
                #print(s1)
                temp = s2[0:i] + s2[j] + s2[i+1:j] + s2[i] + s2[j+1:]
                #print(s2[0:i] + s2[j] + s2[i+1:j] + s2[i] + s2[j+1:])
                if s1 == temp:
                    return True
        return False
