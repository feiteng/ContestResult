class Solution:
    def findCenter(self, edges: List[List[int]]) -> int:
        
        tmp = set()
        
        for a, b in edges:
            if a in tmp:
                return a 
            if b in tmp:
                return b 
            
            tmp.add(a)
            tmp.add(b)
        
        return -1
