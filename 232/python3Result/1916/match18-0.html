class Solution:
    def findCenter(self, edges: List[List[int]]) -> int:
        edge_present=set()
        for u,v in edges:
            if u in edge_present:
                return u
            if v in edge_present:
                return v
            edge_present.add(u)
            edge_present.add(v)
        return -1
