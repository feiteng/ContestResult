class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        d = defaultdict(int)
        for i in range(len(nums)):
            j = len(nums) - i - 1
            if i >= j:
                break
            d[2] += 2
            d[min(nums[i], nums[j]) + 1] -= 1
            d[nums[i] + nums[j]] -= 1
            d[nums[i] + nums[j] + 1] += 1
            d[max(nums[i], nums[j]) + limit + 1] += 1
        ans, crnt = 10 ** 9, 0
        for i in sorted(d.keys()):
            crnt += d[i]
            ans = min(ans, crnt)
        return ans
