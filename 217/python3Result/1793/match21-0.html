class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        aa=[0]*(limit*2+5)
        n=len(nums)
        for i in range(n//2):
            x=nums[i]
            y=nums[n-1-i]
            mn=min(x,y)+1
            mx=max(x,y)+limit
            aa[mn]+=1
            aa[x+y]+=1
            aa[x+y+1]-=1
            aa[mx+1]-=1
            # print(aa)
        for i in range(limit*2+4):
            aa[i+1]+=aa[i]
        # print(aa)
        return n-max(aa)
