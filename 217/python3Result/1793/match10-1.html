class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)

        temp = []
        
        def helper(i):
            cur = nums[i] + nums[n - 1 - i]
            a = min(nums[i] + 1, nums[n - 1 - i] + 1)
            b = max(nums[i] + limit, nums[n - 1 - i] + limit)
            return [a, b, cur]
        
        for i in range(n//2):
            temp.append(helper(i))

        
        t 
= [0]*(2*limit + 2)
        
        for a, b, c in temp:
            t[a] -= 1
            t[b + 1] += 1
            t[c] -= 1
            t[c + 1] += 1
            
        ans = cur = n
        for l in range(2*limit + 1):
            cur += t[l]
            ans = min(ans, cur)
        return ans
