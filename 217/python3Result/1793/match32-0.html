class Solution:
    def find1(self, line, a):
        left = 0
        right = len(a) - 1
        ans = -1
        while left <= right:
            mid = (left + right) // 2
            if a[mid] >= line:
                ans = mid
                right = mid - 1
            else:
                left = mid + 1
        return ans
    def find2(self, line, a):
        left = 0
        right = len(a) - 1
        ans = -1
        while left <= right:
            mid = (left + right) // 2
            if a[mid] <= line:
                ans = mid
                left = mid + 1
            else:
                right = mid - 1
        return ans + 1
    def minMoves(self, nums: List[int], limit: int) -> int:
        a = []
        ans = len(nums) // 2
        for i in range(len(nums) // 2):
            a.append([nums[i] + nums[-(i + 1)], nums[i], nums[-(i + 1)]])
        a.sort(key=lambda x: (x[0], abs(x[1] - x[2])))
        p = dict()
        for i in range(len(a)):
            if a[i][0] in p:
                p[a[i][0]] += 1
            else:
                p[a[i][0]] = 1
        c = []
        d = []
        for i in range(len(a)):
            c.append(max(a[i][1], a[i][2]))
            d.append(min(a[i][1], a[i][2]))
        c.sort()
        d.sort()
        for i in range(len(a)):
            x = a[i][0]
            y = self.find1(x - limit, c)
            z = self.find2(x - 1, d)
            ans = min(ans, len(nums) // 2 - p[x] + y + len(d) - z)
        return ans
