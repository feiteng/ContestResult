class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        #差分
        n = len(nums)
        cost = [0]*(limit*2+2)
        cnt = [0]*(limit*2+1)
        for i in range(n//2):
            v1,v2 = nums[i],nums[-i-1]
            # max_reach by change 1
            max_reach = max(v1,v2)+limit
            # min_reach by change 1
            min_reach = min(v1,v2)+1
            cnt[v1+v2]+=1
            cost[2]+=2
            cost[min_reach]-=1
            cost[max_reach+1]+=1
        
        #print(cost,cnt)
        res = float('inf')
        curr=0
        for i in range(2,len(cnt)):
            curr+=cost[i]
            res = min(res,curr-cnt[i])
        return res
