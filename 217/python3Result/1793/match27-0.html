class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        arr = [0]*(limit*2+2)
        arr[2] = n
#        print(f'Solving {nums}  {limit}')
        for i in range(n//2):
            n1 = nums[i]
            n2 = nums[n-1-i]
            if n1 > n2:
                n1,n2 = n2,n1
            arr[n1+1] -= 1
            arr[n2+limit+1] += 1
            arr[n1+n2] -= 1
            arr[n1+n2+1] += 1
        
#        print(f'  - arr = {arr}')
        result = 2*n
        acc = 0
        for x in range(2, 2*limit+1):
            acc += arr[x]
            if acc < result:
                result = acc
        return result
