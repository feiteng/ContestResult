class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        change = [0]*(2*limit+5)
        cc = Counter()
        n = len(nums)
        for i in range(n//2):
            a, b = nums[i], nums[n-1-i]
            cc[a+b] += 1
            change[min(a,b)+1] += 1
            change[max(a,b)+1+limit] -= 1
        res, cnt = n, 0
        for cur in range(2,limit*2+1):
            cnt += change[cur]
            res = min(res, cnt-cc[cur]+n-2*cnt)
        return res
