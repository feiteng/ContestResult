class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        imos = [0]*(2*limit+2)
        n = len(nums)
        cnt = defaultdict(int)
        
        for i in range(n//2):
            a, b = min(nums[i], nums[n-i-1]), max(nums[i], nums[n-i-1])
            l = a+1
            r = b+limit
            imos[0] += 2
            imos[l] -= 2
            imos[l] += 1
            imos
[r+1] -= 1
            imos[r+1] += 2
            imos[2*limit+1] -= 2
            cnt[a+b] += 1
        
        for i in range(1, 2*limit+1):
            imos[i] += imos[i-1]
        
        ans = 10**18
        
        for i in range(2, 2*limit+1):
            ans = min(ans, imos[i]-cnt[i])
        
        return ans
