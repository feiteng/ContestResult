class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        deltas = [0] * (2 * limit + 5)
        cum = [0] * (2 * limit + 5)
        for i in range(int(len(nums) / 2)):
            j = len(nums) - i - 1
            lower = min(nums[i], nums[j])
            upper = max(nums[i], nums[j])
            deltas[2] += 2
            deltas[lower+1] -= 1
            deltas[lower+upper] -= 1
            deltas[lower+upper+1] += 1
            deltas[upper+limit+1] += 1
        ans = len(nums) / 2
        for i in range(2, len(cum)):
            cum[i] = cum[i-1] + deltas[i]
            ans = min(ans, cum[i])
        #print(deltas)
        #print(cum)
        return int(ans)
