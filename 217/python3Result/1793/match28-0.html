class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        k=limit
        n=len(nums)
        a=nums
        c = defaultdict(lambda : 0)
        for i in range(n//2):
            x, y = a[i], a[n-i-1]
            c[2] += 2
            c[min(x, y)+1] -= 1
            c[x+y] -= 1
            c[x+y+1] += 1
            c[max(x, y)+k+1] += 1
            c[2*k+1] -= 2
        ans = n
        for i in range(2, 2*k+1):
            c[i] += c[i-1]
            ans = min(ans, c[i])
        return ans
