class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        move0 = [0]*(limit*2+2)
        move1 = [0]*(limit*2+2)
        
        for i in range(len(nums)//2):
            a, b = nums[i], nums[len(nums)-1-i]
            a, b = min(a, b), max(a, b)
            print(a, b, a+b)
            move0[a+b] += 1
            
            sm, lg 
= a+1, b+limit
            move1[sm] += 1
            move1[a+b] -= 1
            move1[a+b+1] += 1
            move1[lg+1] -= 1 
        
        # now iterate and calculate the number of move
        res = len(nums)
        
        ps = 0
        for i in range(len(move0)):
            ps += move1[i]
            res = min(res, len(nums) - 2*move0[i] - ps)
        return res
