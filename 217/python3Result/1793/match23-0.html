class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        m = n // 2
        min_val = 2
        max_val = limit * 2
        increments = [0] * (max_val + 2)
        def add(start, end, inc):
            increments[start] += inc
            increments[end+1] -= inc
        for i in range(m):
            left, right = nums[i], nums[n-1-i]
            curr_sum = left + right
            add(min_val, max_val, 2)
            add(1 + min(left, right), limit + max(left, right), -1)
            add(curr_sum, curr_sum, -1)
        curr_cost = 0
        min_cost = math.inf
        for i in range(min_val, max_val+1):
            curr_cost += increments[i]
            min_cost = min(min_cost, curr_cost)
        return min_cost
