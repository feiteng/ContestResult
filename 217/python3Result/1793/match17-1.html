class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        if limit == 1:
            return 0
        
        n = len(nums)
        c = Counter()
        for i in range(n>>1):
            couple = [nums[i], nums[-i-1]]
            couple.sort()
            x, y = couple
            if x==1:
                if y==1:
                    c
[2] += 0
                    c[3] += 1
                    c[limit+2] += 1
                else:
                    c[2] += 1
                    c[1+y] -= 1
                    c[y+2] += 1
                    c[limit+y+1] 
+= 1
            elif x<limit:
                if y<limit:
                    c[2] += 2
                    c[x+1] -= 1
                    c[x+y] -= 1
                    c[x+y+1] += 1
                    c[y+limit+1] += 1
                else:
                    c[2] += 2
                    c[x+1] -= 1
                    c[x+y] -= 1
                    c[x+y+1] += 1
            else:
                c[2] += 2
                c[limit+1] -=1
                c[limit<<1] -= 1
        
        keys = list(c.keys())
        keys.sort()
        res = float("inf")
        x = 0
        for k in keys:
            x += c[k]
            res = min(res, x)
        return res
