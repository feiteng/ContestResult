class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        nums1 = nums[:n // 2]
        nums2 = nums[n // 2:][::-1]
        sums = [nums1[i] + nums2[i] for i in range(n // 2)]
        lower = [1 + min(nums1[i], nums2[i]) for i in range(n // 2)]
        upper = [limit + max(nums1[i], nums2[i]) for i in range(n // 2)]
        
        save = [0] * (2 * limit + 2)  # save[i]: no. of moves can save if target == i
        for i in range(n // 2):
            save[lower[i]] -= 1
            save[sums[i]] -= 1
            save[sums[i] + 1] += 1
            save[upper
[i] + 1] += 1
        mini = temp = 0
        for target in range(2, 2 * limit + 1):
            temp += save[target]
            mini = min(mini, temp)
        return n + mini
