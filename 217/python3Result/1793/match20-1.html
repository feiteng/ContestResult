from collections import defaultdict
class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:

        n = len(nums)
        v = [0] * (2*limit+2)
        mid = n // 2
        for i in range(mid):
            target = nums[i] + nums[n-1-i]
            minV 
= min(nums[i], nums[n-1-i])
            maxV = max(nums[i], nums[n-1-i])
            v[2] += 2
            v[2*limit+1] -= 2
            v[minV+1] -= 1
            v[maxV+limit+1] += 1
            v[target] -= 1
            v[target+1] += 1
        
        res = float("inf")
        for i in range(2, 2*limit+1):
            v[i] += v[i-1]
            res = min(res, v[i])
        
        return res
