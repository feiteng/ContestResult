class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        best = N = len(nums)
        d = collections.defaultdict(int)
        delta = collections.defaultdict(int)
        maxTotal = 2
        
        for index in range(N//2):
            maxOne = max(nums[index] + limit, nums[N - index - 1] + limit)
            minOne = min(nums[index] + 1, nums[N - index - 1] + 1)
            total = nums[index] + nums[N - index - 1]
            if maxTotal < total:
                maxTotal = total
            
            d[total] += 1
            
            delta[minOne] -= 1
            delta[maxOne+1] 
+= 1

        #print(delta)
        current = N
        for total in range(2, maxTotal + 1):
            current += delta[total]
            best = min(best, current - d[total])
            #print(total, current - d[total], current, d[total])
        
        return best
