class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        l 
= []
        
        dif = [0] * (3 * limit + 2)
        for i in range(n // 2):
            a, b = nums[i], nums[n-1-i]
            if a > b:
                a, b = b, a
            if a == 1 and b == 1:
                dif[3] += 1
                dif[limit + 2] += 1
            elif a == 1 and b > 1:
                dif[2] += 1
                dif[b + 1] += -1
                dif[b + 2] += 1
                dif[b + limit + 1] += 1
            elif a == limit and b == limit:
                dif[2] += 2
                dif[a + 1] += -1
                dif[a + b] += -1
                pass
            elif a < limit and b == limit:
                dif[2] += 2
                dif[a + 1] += -1
                dif[a + b] += -1
                dif[a + b + 1] += 1
            else:
                dif[2] += 2 
                dif[a + 1] += -1
                dif[a + b] += -1
                dif[a + b + 1] += 1
                dif[b + limit + 1] += 1
        # print("dif:" + str(dif))
        pre = [0]
        for i in range(2, len(dif)):
            pre.append(pre[-1] + dif[i])
        print("pre:" + str(pre))
        return min(pre[1:])
