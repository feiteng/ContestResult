class Solution:
    def minMoves(self, arr: List[int], k: int) -> int:
        n = len(arr)
        cnt = [0] * (2*k + 7)
        cnt[0] 
= n
        for i in range(n >> 1):
            a, b = arr[i], arr[-i-1]
            mn = min(a, b) + 1
            mx = max(a, b) + k
            # [mn,a+b): -1, a+b:-2, (a+b,mx]:-1
            cnt[mn] -= 1
            cnt[a+b] -= 1
            cnt[a+b+1] += 1
            cnt[mx+1] += 1
        ans = n
        for i in range(1, len(cnt)):
            cnt[i] += cnt[i-1]
            ans = min(ans, cnt[i])
        return ans
