class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        target = [0]*(2*limit+1)
        target
[0] = n = len(nums)
        for i in range(n//2):
            a, b, c = 1+min(nums[i],nums[n-1-i]), nums[i]+nums[n-1-i], limit+max(nums[i],nums[n-1-i])
            target[a] -= 1
            target[b] -= 1
            if b+1<=2*limit:
                target[b+1] += 1
            if c+1<=2*limit:
                target[c+1] += 1
        for i in range(1,2*limit+1):
            target[i] += target[i-1]
        return min(target)
