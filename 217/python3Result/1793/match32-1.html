class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        chg = []
        for i in range(len(nums) // 2):
            x, y = nums[i], nums[len(nums) - i - 1]
            mn, mx = min
(x, y), max(x, y)
            chg.extend( [[mn + 1, 1], [x + y, 1], [x + y + 1, -1], [mx + limit + 1, -1]] )
        chg.sort()
        #print(chg)
        cur, ans = 0, 0
        for v, d in chg:
            cur += d
            ans = max(ans, cur)
        return len(nums) - ans
