class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n, m = len(nums), len(nums) // 2
        c = collections.Counter()
        a = []
        b = []
        
for i in range(m):
            c[nums[i] + nums[n - i - 1]] += 1
            a.append(max(nums[i], nums[n - i - 1]))
            b.append(min(nums[i], nums[n - i - 1]))
        a.sort()
        b.sort()
        ret = n
        for k in range(2, limit * 2 + 1):
            x1 = m - bisect.bisect_right(b, k - 1)
            x2 = bisect.bisect_left(a, k - limit)
            tmp = (x1 + x2) * 2 + (m - x1 - x2) - c[k]
            ret = min(ret, tmp)
        return ret
