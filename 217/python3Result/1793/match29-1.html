class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        dp = [0] * (limit * 2 + 2)
        for i in range(n // 2):
            dp[nums[i]+nums[n-1-i]] -= 1
            dp[min(nums
[i],nums[n-1-i])+1] -= 1
            dp[nums[i]+nums[n-i-1]+1] += 1
            dp[max(nums[i],nums[n-i-1])+limit+1] += 1
        res = n
        cur = n
        print(dp)
        for i in range(2*limit+1):
            cur += dp[i]
            res = min(res,cur)
        return res
