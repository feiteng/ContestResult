from typing import List


class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        arr = [0] * (limit * 2 + 2)
        l = len(nums)
        for i in range(l//2):
            a, b = nums[i], nums[l-1-i]
            if a > b:
                a, b = b, a
            arr[2] += 2
            arr
[a+1] -= 1
            arr[a+b] -= 1
            arr[a + b+1] += 1
            arr[limit+b+1] += 1
        res = len(nums)
        # print(arr)
        pre = 0
        for i in range(len(arr)):
            pre += arr[i]
            if i >= 2:
                res = min(res,pre)
        return res
