class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        if limit == 1:
            return 0
        count = [0]*(2*limit+2)
        for i in range(len(nums)//2):
            x1, x2 = nums[i], nums[len(nums)-1-i]
            if x1 == 1 and x2 == 1:
                count[limit+2] += 1
                count[3] += 1
            elif x1 == 1 or x2 == 1:
                if x1 == 1:
                    x1 = x2
                count[2] += 1
                count[1+x1] -= 1
                count[2+x1] += 1
                count[x1+limit+1] += 1
            else:
                count[2] += 2
                count[1+min(x1,x2)] -= 1
                count[x1+x2] -= 1
                count[x1+x2+1] += 1
                count[max(x1,x2)+limit+1] += 1
        for i in range(2, len(count)):
            count[i] += count[i-1]
        res = min(count[2:2*limit+1])
        return res
