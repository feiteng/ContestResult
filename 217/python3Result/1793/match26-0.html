class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        diff = []
        n = len(nums)
        for i in range(n // 2):
            s, l = sorted([nums[i], nums[n - 1 - i]])
            diff.append((s + 1, -1))
            diff.append((s + l, -1))
            diff.append((s + l + 1, 1))
            diff.append((l + limit + 1, 1))
        diff.sort()
        ans = curr = n
        # print(diff)
        for _, it in itertools.groupby(diff, key=lambda t: t[0]):
            _, l = zip(*it)
            curr += sum(l)
            ans = min(ans, curr)
        return ans
