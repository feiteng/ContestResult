class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        if n == 2:
            return 0
        tot = 3*limit + 2
        tmp = [0]*tot
        for i in range(n//2):
            elem1, elem2 = nums[i], nums[n-i-1]
            if elem1 > elem2:
                elem1, elem2 
= elem2, elem1
            if elem1 == 1 and elem2 == 1:
                tmp[3] += 1
                tmp[limit+2] += 1
            elif elem1 == 1 and elem2 > 1:
                tmp[2] += 1
                tmp[elem2+1] -= 1
                tmp[elem2+2] += 1
                tmp[elem2+limit+1] += 1
            elif elem1 == limit and elem2 == limit:
                tmp[2] += 2
                tmp[elem1+1] -= 1
                tmp[elem1+elem2] -= 1
            elif elem1 < limit and elem2 == limit:
                tmp
[2] += 2
                tmp[elem1+1] -= 1
                tmp[elem1+elem2] -= 1
                tmp[elem1+elem2+1] += 1
            else:
                tmp[2] += 2
                tmp[elem1+1] -= 1
                tmp[elem1+elem2] -= 1
                tmp[elem1+elem2+1] += 1
                tmp[elem2+limit+1] += 1
        ans = [0]
        for i in range(2, tot):
            ans.append(ans[-1] + tmp[i])
        return min(ans[1:])
