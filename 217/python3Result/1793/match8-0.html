from collections import defaultdict

class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        dic = defaultdict(int); n = len(nums)
        mid = n//2
        L = [0]*(2*limit+3)
        for i in range(mid):
            #print(L)
            a = nums[i]
            b = nums[n-1-i]
            #print(a,b,a+b)
            if a > b:
                a,b = b,a
            if a == 1 and b == 1:
                # 1 area
                L[3] += 1
                L[2+limit] -= 1
                L[2+limit] += 2
            elif a == 1 and b == limit:
                L[2] += 1
                L[a+limit] -= 1
                L[a+limit+1] += 1
            elif a == limit and b == limit:
                L[2] += 2
                L[1+limit] -= 2
                L[1+limit] += 1
                L[2*limit] -= 1 
                L
            elif a == 1:
                L[2] += 1
                L[a+b] -= 1
                L[a+b+1] += 1
                L[b+limit+1] -= 1
                L[b+limit+1] += 2
            elif b == limit:
                L[2] += 2
                L[1+a] -= 2
                L[1+a] += 1
                L[a+b] -= 1
                L[a+b+1] += 1
            else:
                L[2] += 2
                L[1+a] -= 2
                L[1+a] += 1
                L[a+b] -= 1
                L[a+b+1] += 1
                L[b+limit+1] -= 1
                L[b+limit+1] += 2
        S = [0]
        for x in L:
            temp = S[-1]+x
            S.append(temp)
        #print(L)
        #print(S)
        ans = min(S[3:])
        return ans
