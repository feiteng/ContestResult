import itertools

class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        length = len(nums) // 2
        array1 = nums[: length]
        array2 = nums[length: ][:: -1]
        derivatives = [0] * (2 * limit + 2)

        for v, w in zip(array1, array2):
            derivatives[2] += 2
            derivatives[min(v, w) + 1] -= 1
            derivatives[v + w] -= 1
            derivatives[v + w + 1] += 1
            derivatives[max(v, w) + limit + 1] += 1
            derivatives[2 * limit + 1] -= 2

        integrals = list(itertools.accumulate(derivatives))

        return min(integrals[2: 2 * limit + 1])
