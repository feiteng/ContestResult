class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        # [1, 2, 2, 1] -> (1, 2) (2, 1)
        n = len(nums)
        sumPairs = [(nums[i] + nums[n-i-1], min(nums[i], nums[n-i-1]) + 1, max(nums[i], nums[n-i-1]) + limit) for i in range(n//2)]
        
        steps = [0] * (limit * 2 + 2)
        for m, l, r in sumPairs:
            steps
[m] -= 1
            steps[m+1] += 1
            steps[r+1] += 1
            steps[l] -= 1
        
        init = n
        res = n
        for i in range(1, limit*2+2):
            init += steps[i]
            res = min(res, init)
        
        return res
