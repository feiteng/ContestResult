class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        df = [0] * (limit*2 + 2)
        for i in range(len(nums)//2):
            a 
= nums[i]
            b = nums[len(nums)-i-1]
            s = a + b
            df[s] += 1
            df[s+1] -= 1
            df[min(a, b)+1] += 1
            df[max(a, b)+limit+1] -= 1
        cur = 0
        mx = 0
        for e in df:
            cur += e
            mx = max(cur, mx)
        return len(nums) - mx
