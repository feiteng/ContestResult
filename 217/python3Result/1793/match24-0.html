class Solution:
    def minMoves(self, nums: List[int], limit: int) -> int:
        s = len(nums)//2
        max_arr = []
        min_arr = []
        c = Counter()
        for i in range(s):
            c[nums[i]+nums[~i]]+=1
            max_arr.append( max(nums[i],nums[~i])) 
            min_arr.append( min(nums[i],nums[~i])) 
        max_arr = sorted(max_arr)
        min_arr = sorted(min_arr)
        ans = s*2
        
        #print(max_arr, min_arr)
        for e in c.keys():
            
            cnt = 0
            l = bisect.bisect_right(max_arr,(e-limit-1))
            r = (s - bisect.bisect_right(min_arr,e-1))
            cnt = l*2+r*2+(s-l-r)-c[e]
            #print(e, cnt, l, r )# , s, c[e], l*2+r*2+(s-l-r)-c[e])
            ans = min(ans, cnt)
        return ans
