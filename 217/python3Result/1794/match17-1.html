import sortedcontainers

class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        sl = sortedcontainers.SortedList()
        for num in nums:
            if num%2:
                sl.add([num, 2*num]),
            else:
                queue = deque([num])
                while queue[0]%2 == 0:
                    queue.appendleft(queue[0]//2)
                sl.add(list(queue))
        deviation = sl[-1][0] - sl[0][0]
        while len(sl[0]) > 1:
            x = sl[0]
            sl.remove(x)
            x
.pop(0)
            sl.add(x)
            deviation = min(deviation, sl[-1][0] - sl[0][0])
        return deviation
