class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            if nums[i] % 2 == 0:
                nums[i] //= 2
        nums.sort()
        while nums[-1] % 2 == 0:
            for i in range(len(nums) - 1, -1, -1):
                if nums[i] % 2 == 0:
                    nums[i] //= 2
                else:
                    break
            nums.sort()
        ans = nums[-1] - nums[0]
        for i in range(len(nums) - 1):
            ans_x = min(nums[i + 1], nums[0] * 2)
            ans_y = max(nums[-1], nums[i] * 2)
            ans = min(ans, ans_y - ans_x)
        return ans
