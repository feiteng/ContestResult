class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        nums = sorted(nums)
        ret = nums[-1] - nums[0]
        while nums[0] % 2 == 1:
            bisect.insort_left(nums, nums[0] * 2)
            nums.pop(0)
            ret=min(ret,nums[-1]-nums[0])
        while nums[-1] % 2 == 0:
            bisect.insort_left(nums, nums[-1]//2)
            nums.pop()
            ret = min(ret, nums[-1] - nums[0])
        return ret
