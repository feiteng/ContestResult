from heapq import heapify, heappush, heappop
class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        keys = set()
        for x in nums:
            x0 = x
            while x & 1 == 0: x >>= 1
            keys.add((x, x0))
        pq, K = [], max(keys)[0]
        curr_max, ans = K, math.inf
        for x, x0 in keys:
            heappush(pq, (x
, x0))
        # print(keys)
        while True:
            x, x0 = heappop(pq)
            ans = min(ans, curr_max - x)
            if x % 2 == 0 and x >= x0: break
            x *= 2
            curr_max = max(curr_max, x)
            heappush(pq, (x, x0))
            # print(sorted(pq))
        return ans
