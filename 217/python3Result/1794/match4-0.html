from sortedcontainers import SortedList
class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        sl = SortedList(a*2 if a & 1 else a for a in nums)
        ans = sl[-1] - sl[0]
        while sl[-1] % 2 == 0:
            sl.add(sl.pop() // 2)
            ans = min(ans, sl[-1] - sl[0])
        return ans
