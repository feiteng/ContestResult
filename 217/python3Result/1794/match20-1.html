class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        
        
        sets = []
        for i in range(len(nums)):
            p 
= nums[i]
            if p%2 == 1:
                sets.append((p,i))
                sets.append((2*p,i))
            else:
                while p%2 == 0:
                    sets.append((p,i))
                    p //= 2
                sets.append((p,i))
        
        # print(sets)
        
        
        n = len(nums)

        sets.sort()

        str = 0
        end = 0
        mark = [0]*n
        al = n

        while end < len(sets):
            mark[sets[end][1]] += 1
            if mark[sets[end][1]] == 1:
                al -= 1
                if al == 0:
                    break
            end +=1

        best = sets[end][0] - sets[str][0]
        best_str = str
        best_end = end

        while end < len(sets):
            mark[sets[str][1]] -= 1
            if mark[sets[str][1]] > 0:
                str += 1
                p = sets[end][0] - sets[str][0]
                if p < best:
                    best = p
                    best_str = str
                    best_end = end
                continue
            while mark[sets[str][1]] == 0 and end < len(sets)-1:
                end += 1
                mark[sets[end][1]] += 1

            if mark[sets[str][1]] == 0:
                break

            str += 1
            p = sets[end][0] - sets[str][0]
            best = min(p, best)
        
        return best
