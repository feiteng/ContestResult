import heapq

class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        def conv(n):
            c = 0
            while n % 2 == 0:
                n //= 2
                c += 1
            return (n, max(c, 1))
        nums = [conv(n) for n in nums]
        
        pq = nums[:]
        heapq.heapify(pq)
        
        max_value = max(pq)[0]
        min_value = min(pq)[0]
        
        min_dev = max_value - min_value
        
        while True:
            n, t = heapq.heappop(pq)
            if t == 0:
                break
            nn = n * 2
            nt = t - 1
            heapq.heappush(pq, (nn, nt))
            max_value = max(max_value, nn)
            min_value = pq[0][0]
            min_dev = min(min_dev, max_value - min_value)
        return min_dev
