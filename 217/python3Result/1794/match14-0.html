from heapq import heappop, heappush
class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        raw = set()
        h = []
        for i, n in enumerate(nums):
            while n%2 == 0:
                n //= 2
            raw.add((n, nums[i]))
        ans = float('inf')
        cur = max(raw)[0]
        for e in raw:
            heappush(h, e)
        while True:
            n, n0 = heappop(h)
            ans = min(ans, cur - n)
            if n % 2 == 0 and n >= n0:
                break
            n *= 2
            cur = max(cur, n)
            heappush(h, (n, n0))
        return ans
