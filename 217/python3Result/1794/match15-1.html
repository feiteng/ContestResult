class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        if len(nums) < 2:
            return 0
        h_min = []
        h_max = []
        
        for num in nums:
            heapq.heappush(h_min, num)
            heapq.heappush(h_max, -num)     
        
        ans = -h_max[0] - h_min[0]
        while  h_min[0] % 2 == 1 and -h_max[0] > h_min[0]:
            tmp = heapq.heappop(h_min)
            heapq.heappush(h_max, -tmp * 2)
            heapq.heappush(h_min, tmp * 2)
            ans = min(ans, -h_max[0] - h_min[0])
            
        while -h_max[0] % 2 == 0 and -h_max[0] > h_min[0]:
            tmp = -heapq.heappop(h_max)
            heapq
.heappush(h_max, -tmp // 2)
            heapq.heappush(h_min, tmp // 2)
            ans = min(ans, -h_max[0] - h_min[0])

        return ans
