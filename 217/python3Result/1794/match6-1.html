from sortedcontainers import SortedList
class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            if nums[i] % 2 == 1:
                nums[i] = nums[i] * 2
        
  
        s = SortedList(nums)
        max_diff = s[-1]-s[0]
        while s
[-1] % 2 == 0:
            s.add(s.pop() // 2)
            max_diff = min(s[-1]-s[0], max_diff)
        return max_diff
