class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        nums.sort()
        ans = nums[-1]-nums[0]
        while nums[0]&1 and  nums[0]*2<2*nums[-1]-nums[0]:
            bisect.insort(nums,nums.pop(0)*2)
            ans = min(nums[-1]-nums[0],ans)
        while nums[-1]&1==0 and nums[-1]//2>=2*nums[0]-nums[-1]:
            bisect.insort(nums,nums.pop()//2)
            ans = min(nums[-1]-nums[0],ans)
        return ans
