class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        N = len(nums)
        
        for index in range(N):
            if nums[index] % 2 == 1:
                nums[index] *= 2
                
        nums.sort()
        
        best = nums[-1] - nums[0]
        smallest = nums[0]
        newSmall = smallest
        
        heap = list(-x for x in nums)
        heapq.heapify(heap)

        while len(heap) > 0:
            #print(heap)
            now = -heap[0]
            if now < newSmall:
                break
            best = min(best, now - newSmall)
            heapq.heappop(heap)

            if now % 2 == 1:
                break

            newSmall = min(now // 2, newSmall)
            heapq.heappush(heap, -(now // 2))            
        
        return best
