import heapq


class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            if nums[i]%2 == 1:
                nums[i]*=2
                
                
        nums = [-n for n in nums] # to make the heap work better
        heapq.heapify(nums)
        
        nummin = -max(nums) # always equal to min(|nums|)
        bestdeviation = float("inf")
        while True:
            #print(nums,nummin)
            nummax = -heapq.heappop(nums)
            if bestdeviation>nummax-nummin:
                bestdeviation = nummax-nummin
            if nummax%2 == 1:
                break
            else:
                newel = -nummax//2
                heapq.heappush(nums,newel)
                if -newel<nummin:
                    nummin = -newel
        return bestdeviation
