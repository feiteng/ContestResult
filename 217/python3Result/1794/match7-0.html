class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        from bisect import insort_left
        sorted_nums = sorted(nums)
        n = len(sorted_nums)
        ans = sorted_nums[n - 1] - sorted_nums[0]
        while sorted_nums[0] & 1 == 1:
            cur = sorted_nums[0] * 2
            sorted_nums.pop(0)
            insort_left(sorted_nums, cur)
            ans = min(ans, sorted_nums[n - 1] - sorted_nums[0])
        while sorted_nums[n - 1] & 1 == 0:
            cur = sorted_nums[n - 1] // 2
            sorted_nums.pop()
            insort_left(sorted_nums, cur)
            ans = min(ans, sorted_nums[n - 1] - sorted_nums[0])
        return ans
