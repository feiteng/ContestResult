from typing import List


class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        nums.sort()
        ans = nums[-1] - nums[0]
        import bisect
        while nums[0] % 2:
            cur = nums[0] * 2
            nums.pop(0)
            bisect.insort_left
(nums, cur)
            # print(nums)
            ans = min(ans, nums[-1] - nums[0])
        while nums[-1] % 2 == 0:
            cur = nums[-1] // 2
            nums.pop()
            bisect.insort_left(nums, cur)
            # print(nums)
            ans = min(ans, nums[-1] - nums[0])
        return ans
