class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        def helper(n):
            if n % 2 == 1:
                return n * 2
            return n
        data = [helper(n) for n in nums]
        # print(data)
        from queue import PriorityQueue
        big = PriorityQueue()
        small = PriorityQueue()
        for n in data:
            big.put(-n)
            small.put(n)
        res = -big.queue[0] - small.queue[0]
        while -big.queue[0] % 2 == 0:
            if small.queue[0] == -big.queue[0]:
                small.put(small.get() // 2)
            old = -big.get()
            new = -(old // 2)
            # print(old, new)
            big.put(new)
            small.put(-new)
            res = min(res, -big.queue[0] - small.queue[0])
        return res
