class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        
        if not nums:
            return 0

        prev = []
        x = nums[0]
        if x%2==1:
            prev.append((x,x))
            prev.append((x*2, x*2))
        else:
            while x%2==0:
                prev.append((x, x))
                x //= 2
            prev.append((x, x))
        
        for num in nums[1:]:
            cand = []
            if num%2==1:
                cand.append(num)
                cand.append(num*2)
            else:
                x = num
                while x%2==0:
                    cand.append(x)
                    x//=2
                cand.append(x)
                cand = cand[::-1]
            cur = []
            for a, b in prev:
                if a >= cand[-1]:
                    cur.append((cand[-1], b))
                elif b<=cand[0]:
                    cur.append((a, cand[0]))
                elif a<=cand[0] and b>=cand[1]:
                    cur.append((a, b))
                else:
                    done = False
                    for x in cand:
                        if a<=x<=b:
                            cur.append((a, b))
                            done=True
                            break
                    if not done:
                        for x in cand[::-1]:
                            if x <= a:
                                cur.append((x, b))
                                break
                        for x in cand:
                            if x>=b:
                                cur.append((a, x))
                                break

            prev = cur
        res = float('inf')
        for minx, maxx in prev:
            res = min(res, maxx-minx)
        return res
