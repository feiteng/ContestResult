import heapq
class Solution:
    def minimumDeviation(self, nums: List[int]) -> int:
        def bounds(n):
            if n % 2 == 1:
                return n, n, 2 * n
            else:
                curr = n
                while curr & 1 == 0:
                    curr >>= 1
            return curr, curr, n
        
        nums = [*map(bounds, nums)]
        heapq
.heapify(nums)
        upper = max(x[0] for x in nums)
        lower = min(x[0] for x in nums)
        res = upper - lower
        while 1:
            curr, mn, mx = heapq.heappop(nums)
            if curr == mx:
                return res
            
            curr 
*= 2
            heapq.heappush(nums, (curr, mn, mx))
            
            upper = max(upper, curr)
            lower = nums[0][0]
            res = min(res, upper - lower)
        return res
