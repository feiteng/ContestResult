class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        d = []
        n = len(nums)
        for i in range(n):
            while d and d[-1] > nums[i] and len(d) + n - i  > k:
                d.pop()
            d.append(nums[i])
        return d[:k]
