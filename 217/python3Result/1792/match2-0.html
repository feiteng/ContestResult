class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        res = []
        L = len(nums)
        for i,n in enumerate(nums):
            while res and n<res[-1] and L-i>k-len(res):
                res.pop()
            if len(res)<k: res.append(n)
        return res
