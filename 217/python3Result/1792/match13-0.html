class Solution:
    # mono stack, time O(n), space O(n)
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        q = []
        for i, num in enumerate(nums):
            while q and num < q[-1] and len(q) + (len(nums) - i - 1) >= k:
                q.pop()
            q.append(num)
        return q[:k]
