class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        s = []
        for i in range(n):
            while s and nums[i] < s[-1] and len(s) + n - i > k:
                s.pop()
            if len(s) < k:
                s.append(nums[i])
        return s
