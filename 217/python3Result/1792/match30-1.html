class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        ans = []
        for i in range(n):
            while ans and nums[i] < ans[-1] and (len(ans) + n - i - 1) >= k:
                ans.pop()
            ans.append(nums[i])
        return ans[:k]
