class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        s = []
        m = len(nums)
        for i, elem in enumerate(nums):
            while s and elem < s[-1] and (m-i-1+len(s)) >= k:
                s.pop()
            if len(s) < k:
                s.append(elem)
            # print(s)
    
        return s
