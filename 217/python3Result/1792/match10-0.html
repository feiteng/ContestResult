class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        stack = []
        for i, n in enumerate(nums):
            while stack and stack[-1] > n and len(nums) - i + len(stack) - 1 >= k:
                stack.pop()
            stack.append(n)
        # print(stack)
        return stack[:k]
