class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        s = []
        for i, x in enumerate(nums):
            if not s:
                s.append(x)
            else:
                # print(s)
                while s and x < s[-1] and len(s) + (len(nums) - i - 1) + 1 > k:
                    s.pop()
                if len(s) < k:
                    s.append(x)
        return s
