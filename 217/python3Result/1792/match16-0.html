class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        res = []
        total_num = len(nums)
        for i in range(total_num):
            while res and nums[i] < res[-1] and len(res)+total_num-i > k:
                res.pop()
            res.append(nums[i])
        return res[:k]
