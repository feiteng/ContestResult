class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        ans=[]
        n=len(nums)
        for i,j in enumerate(nums):
            while ans and n-i+len(ans)>k and ans[-1]>j:
                ans.pop()
            if len(ans)<k:
                ans.append(j)
        return ans
