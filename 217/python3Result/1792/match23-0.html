class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        #Increasking stack
        res=[]
        LEN=len(nums)
        for idx,num in enumerate(nums):
            while res and res[-1] > num and len(res) + (LEN - idx) > k:
                res.pop()
            res.append(num)
        return res[:k]
