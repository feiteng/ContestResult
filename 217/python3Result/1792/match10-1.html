class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        st = []
        for i,c in enumerate(nums):
            while st and st[-1]>c and len(nums)-1-i+len(st)>=k:
                st.pop()
            st.append(c)
            
        return st[:k]
