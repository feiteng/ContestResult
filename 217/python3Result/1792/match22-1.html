class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        stack = []
        n = len(nums)
        for i, v in enumerate(nums):
            while stack and stack[-1] > v and n - i + len(stack) - 1 >= k:
                stack.pop()
            stack.append(v)
        return stack[:k]
