class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        cur=[]
        for idx,num in enumerate(nums):
            while cur and cur[-1]>num and len(cur)+len(nums)-idx>k:
                cur.pop()
            if len(cur)<k:
                cur.append(num)
        return cur
