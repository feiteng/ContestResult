class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        res = []
        for i in range(n):
            while res and nums[i] < res[-1] and n - i > k - len(res):
                res.pop()
            if len(res) < k:
                res.append(nums[i])
        return res
