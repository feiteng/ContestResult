class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        stack = []
        for i, n in enumerate(nums):
            if not stack:
                stack.append(n)
            else:
                while stack and stack[-1] > n and len(nums)-i > (k-len(stack)):
                    stack.pop()
                if len(stack) < k:
                    stack.append(n)
        return stack
