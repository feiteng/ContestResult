class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        stack = []
        length = len(nums)
        for i, num in enumerate(nums):
            while stack and num<stack[-1] and len(stack)+length-i-1>=k:
                stack.pop()
            if len(stack)<k:
                stack.append(num)
        return stack
