class Solution {
    public int minMoves(int[] nums, int limit) {
        int n = nums.length;
        int[] pos = new int[limit * 2 + 2];
        for (int i = 0; i < n / 2; i++) {
            int a = nums[i];
            int b = nums[n - 1 - i];
            if (a > b) {
                int temp = a;
                a = b;
                b = temp;
            }
            pos[2] += 2;
            pos[a + 1]--;
            pos[a + b]--;
            pos[a + b + 1]++;
            pos[limit + b + 1]++;
        
}
        int minMoves = n * 2;
        int moves = 0;
        for (int t = 2; t <= limit * 2; t++) {
            moves += pos[t];
            minMoves = Math.min(minMoves, moves);
        }
        
        return minMoves;
    }
}
