class Solution {
    public int minMoves(int[] nums, int limit) {
        int n = nums.length;
    int m = n / 2;

    int L = limit;
    int[] ca = new int[2 * L + 1];
    for (int i = 0; i < m; i++) {
      int x = Math.min(nums[i], nums[n-1-i]);
      int y = Math.max(nums[i], nums[n-1-i]);
      // [2,x] requires 2
      // [x+1,x+y-1] requires 1
      // [x+y] requires 0
      // [x+y+1,y + L] requires 1
      // [y + L + 1, 2L] requires 2
      ca[2] += 2;
      ca[x+1] -= 1;
      ca[x+y] -= 1;
      if (x + y + 1 <= y + L) {
        ca[x+y+1] += 1;
      }
      if (y + L + 1 <= 2 * L) {
        ca[y+L+1] += 1;
      }
    }
    int ans = Integer.MAX_VALUE;
    int sum = 0;
    for (int i = 2; i <= 2*L; i++) {
      sum += ca[i];
      ans = Math.min(ans, sum);
    }
    return ans;
    }
}
