class Solution {
    public int minMoves(int[] nums, int limit) {
        int [] starts = new int[2 * limit + 2];
        int [] ends = new int[2 * limit + 2];
        int n = nums.length;
        int [] eq = new int[2 * limit + 2];
        for (int i = 0; i < n / 2; i++) {
            int x = nums[i]; int y = nums[n - 1 - i];
            int lo = Math.min(x, y) + 1;
            int hi = Math.max(x, y) + limit;
            eq[x + y]++;
            starts[lo]++;
            ends[hi]++;
        }
        int res = 2 * (n / 2) + 6;
        int cur = 2 * (n / 2);
        for (int i = 2; i < 2 * limit + 2; i++) {
            cur -= starts[i];
            res = Math.min(res, cur - eq[i]);
            //System.out.println(cur - eq[i]);
            cur += ends[i];
        }
        return res;
    }
}
