class Solution {
    public int minMoves(int[] nums, int limit) {
        TreeMap<Integer, Integer> map = new TreeMap<>(); // edge -> count change
        
        for (int i = 0; i < nums.length/2; i++) {
            
            int a = nums[i];
            
int b = nums[nums.length-1-i];
            
            int l = Math.min(a, b) + 1;
            int r = Math.max(a, b) + limit;
            
            map.put(0, 2 + map.getOrDefault(0, 0));
            // map.put(2*limit, 2 + map.getOrDefault(2*limit, 0));
            
            map.put(l, -1 + map.getOrDefault(l, 0));
            map.put(a+b, -1 + map.getOrDefault(a+b, 0));
            map.put(a+b+1, 1 + map.getOrDefault(a+b+1, 0));
            map.put(r+1, +1 + map.getOrDefault(r+1, 0));            
        }
        
        int min = nums.length;
        int cnt = 0;
        for (int sum : map.keySet()) {
            cnt += map.get(sum);
            min = Math.min(min, cnt);
        }
        return min;
    }
}
