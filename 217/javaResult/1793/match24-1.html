	class Solution {
		public int minMoves(int[] nums, int limit) {
			int n = nums.length;

			int[] imos = new int[200005];
			for(int i = 0;i < n/2;i++){
				int L = nums[i], R = nums[n-1-i];
				imos[2] += 2;
				int min = Math.min(L, R) + 1;
				imos[min]--;
				int max = Math.max(L, R) + limit;
				imos[max+1]++;
				imos[L+R]--;
				imos[L+R+1]++;
			}
			for(int i = 0;i <= 200003;i++){
				imos[i+1] += imos[i];
			}
			int min = 999999999;

			for(int i = 2;i <= 200003;i++){
				min = Math.min(min, imos[i]);
			}
			return min;
		}
	}
