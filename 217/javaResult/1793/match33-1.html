class Solution {
    public int minMoves(int[] nums, int limit) {
        int n = nums.length;
        int t = 2 * limit;
        // dp poll
        int[] total = new int[1 + t];
        
        for (int i = 0; i < total.length; ++i) {
            total[i] = n;
        }
        
        int[] inandout = new int[1 + t + 1];
        
        for (int i = 0; i < n / 2; ++i) {
            int sum = nums[i] + nums[n - i - 1];
            --total[sum];
            
        }
//         if (map.size() == 1) {
//             return 0;
//         }
//         PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b - a);
//         for (int cur : map.keySet()) {
//             pq.add(cur);
//         }
        
        int ans = n / 2;
        
        int cur = 0;
//         while (pq.size() > 0) {
//             int target = pq.poll();
//             if (target > 2 * limit) {
//                 break;
//             }
//             int cur = 0;
//             for (int i = 0; i < n / 2; ++i) {
//                 int sum = nums[i] + nums[n - 1 -i];
//                 if (sum != target) {
//                     if ((nums[i] + 1 <= target && nums[i] + limit >= target) || (nums[n - 1 -i] + 1 <= target && nums[n - 1 -i] + limit >= target)) {
//                         ++cur;
//                     } else{
//                         cur = cur + 2;
//                     }
//                 }

//             }
//             return cur;
//         }
//         return n / 2;
        for (int i = 0; i < n / 2; ++i) {
            // check the in and out for this limis 
            ++inandout[Math.min(nums[i], nums[n - i - 1]) + 1];
            --inandout[Math.max(nums[i], nums[n - i - 1]) + limit+ 1];
        }
        
        
        
        for (int i = 2; i <= t; ++i) {
            
            cur += inandout[i];
            total[i] -= cur;
        }
        
for (int i = 2; i <= t; ++i) {
            ans = Math.min(ans, total[i]);
        }
        return ans;
    }
}
