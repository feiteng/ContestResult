class Solution {
    public int minMoves(int[] nums, int limit) {
        int []count = new int[200003];
        int []sumCnt = new int[200003];
        for (int i = 0; i < (nums.length>>1); i++) {
            sumCnt[nums[i] + nums[nums.length-1-i]]++;
            int min = Math.min(nums[i], nums[nums.length-1-i]) + 1;
            int max = Math.max(nums[i], nums[nums.length-1-i]) + limit;
            count[min]++;
            count[max+1]--;
        }

        int sum = 0;
        int ans = nums.length;
        for (int i = 0; i < count.length; i++) {
            sum+=count[i];
            ans = Math.min(ans, sum-sumCnt[i] + (nums.length/2-sum)*2);
        }
        return ans;
    }
}
