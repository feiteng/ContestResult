class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] rd = new int[200001], rg = new int[200002];
        for(int i=0,j=nums.length-1;i<j;++i,--j){
            rd[nums[i]+nums[j]]++;
            int a=Math.max(nums[i], nums[j]), b=Math.min(nums[i], nums[j]);
            rg[b+1]++;
            rg[a+limit+1]--;
        }
        
int res = 0, N=limit*2, cur=0;
        for(int i=2;i<=N;++i){
            cur+=rg[i];
            res = Math.max(res, rd[i]+cur);
        }
        return nums.length - res;
    }
}
