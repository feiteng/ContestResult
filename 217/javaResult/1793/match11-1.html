class Solution {
    public int minMoves(int[] nums, int limit) {
        
        int[] arr0 = new int[limit+limit+2];
        int[] start = new int[limit+limit+2];
        int[] end = new int[limit+limit+2];
        
        for(int i=0; i<nums.length/2; i++) {
            int j=nums.length-1-i;
            int a = nums[i], b = nums[j];
            
            arr0[a+b] += 1;
            start[Math.min(1+a, 1+b)] +=1;
            end[
Math.max(limit+a, limit+b)+1] +=1;
        }
        
        int res = Integer.MAX_VALUE;
        int count = 0;
        for(int i=0; i<arr0.length; i++) {
            count += start[i];
            count -= end[i];
            
            res = Math.min(res, count + nums.length-2*count - arr0[i]);
        }
        
        return res;
    }
}
