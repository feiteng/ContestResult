class Solution {
    public int minMoves(int[] nums, int limit) {
        Map<Integer,Integer> map=new HashMap<>();
        for(int i=0;i<nums.length/2;i++){
            int j=nums.length-i-1;
            int sum=nums[i]+nums[j], a=Math.min(nums[i],nums[j]), b=Math.max(nums[i],nums[j]);
            map.put(sum, map.getOrDefault(sum,0)-1);
            map.put(sum+1, map.getOrDefault(sum+1,0)+1);
            
            map.put(a+1,map.getOrDefault(a+1,0)+1);
            map.put(b+limit+1,map.getOrDefault(b+limit+1,0)-1);
            
            map.put(1,map.getOrDefault(1,0)+2);
            map.put(a+1,map.getOrDefault(a+1,0)-2);
            
            map.put(b+limit+1,map.getOrDefault(b+limit+1,0)+2);
        }
        
        List<Integer> list=new ArrayList(map.keySet());
        Collections.sort(list);
        
        int cur=0, min=Integer.MAX_VALUE;
        for(int num:list)
{
            cur+=map.get(num);
            min=Math.min(min,cur);
        }
        
        return min;
    }
}
