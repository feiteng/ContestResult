class Solution {
    public int minMoves(int[] nums, int limit) {
        int n = nums.length;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i = 0, j = n - 1; i < j; i++, j--) {
            int sum = nums[i] + nums[j];
            int min = Math.min(nums[i], nums[j]) + 1;
            int max = Math.max(nums[i], nums[j]) + Math.max(Math.min(nums[i], nums[j]), limit);
            map.put(min, map.getOrDefault(min, 0) + 1);
            map.put(sum, map.getOrDefault(sum, 0) + 1);
            map.put(sum+1, map.getOrDefault(sum+1, 0) - 1);
            map.put(max+1, map.getOrDefault(max+1, 0) - 1);
        }
        int res = n;
        int count = 0;
        for (Map.Entry<Integer,Integer> entry : map.entrySet()) {
            count += entry.getValue();
            res = Math.min(n - count, res);
        }
        return res;
    }
}
