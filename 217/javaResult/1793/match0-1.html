class Solution {
  public int minMoves(int[] nums, int limit) {
    int[] delta = new int[2 * limit + 2];
    int n = nums.length;
    for (int i = 0; i < n / 2; ++i) {
      int a = nums[i], b = nums[n - 1 - i];
      delta[Math.min(a, b) + 1]--;
      delta[a + b]--;
      delta[a + b + 1]++;
      delta[Math.max(a, b) + limit + 1]++;
    }
    int moves = n;
    int min_moves = n;
    for (int i = 2; i <= 2 * limit; ++i) {
      moves += delta[i];
      min_moves = Math.min(min_moves, moves);
    }
    return min_moves;
  }
}
