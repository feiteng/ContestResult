class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] mem = new int[2 * limit + 5];
        int len = nums.length, half = len / 2;
        for (int i = 0; i < half; ++i) {
            int min = Math.min(nums[i], nums[len - i - 1]), max = Math.max(nums[i], nums[len - i - 1]);
            mem[2] += 2;
            mem[min + 1] -= 1;
            mem[min + max] -= 1;
            mem[min + max + 1] += 1;
            mem[max + limit + 1] += 1;
        }
        int min = Integer.MAX_VALUE, sum = 0;
        for (int i = 2; i <= 2 * limit; ++i) {
            sum += mem[i];
            min = Math.min(sum, min);
        }
        return min;
    }
}
