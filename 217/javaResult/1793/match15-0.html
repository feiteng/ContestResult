class Solution {
    public int minMoves(int[] nums, int limit) {

        int[] d1 = new int[limit * 2 + 2];
        int[] cnt0 = new int[limit * 2 + 2];

        for (int i = 0; i < nums.length / 2; i++) {

            int j = nums.length - 1 - i;
            int min = Math.min(nums[i], nums[j]);
            int max = Math.max(nums[i], nums[j]);

            cnt0[min + max]++;

            d1[min + 1]++;
            d1[max + limit + 1]--;
        }


        int ans = nums.length / 2;
        int cnt = 0;
        for (int i = 0; i < d1.length; i++) {

            cnt += d1[i];
            int c0 = cnt0[i];
            int c1 = cnt - c0;
            int c2 = nums.length / 2 - c0 - c1;
            ans = Math.min(ans, c1 + 2 * c2);

        }
        return ans;

    }
}
