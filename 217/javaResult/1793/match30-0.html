class Solution {
    int doit(int[] nums, int mid, int limit) {
      int n = nums.length;
      int cnt = 0;
      for(int i = 0; i < n/2; i++) {
        int j = n-1-i;
        if(nums[i] + nums[j] == mid) continue;
        int x = Math.min(nums[i], nums[j]);
        int y = Math.max(nums[i], nums[j]);
        cnt++;
        if(mid > x+y && mid - (x+y) > (limit -x)) cnt++;
        if(mid < x+y && (x+y) - mid > (y-1)) cnt++;
      }
      return cnt;
    }
    public int minMoves(int[] nums, int limit) {
      int n = nums.length, ans = n*2+1;
      int[] diff = new int[200_000 + 10];
      
      for(int i = 0; i < n/2; i++) {
        int x = Math.min(nums[i], nums[n-1-i]);
        int y = Math.max(nums[i], nums[n-1-i]);
        if(x + y == 2) {
          diff[2] += 0;
          diff[(x+y+1)] += 1;
          diff[y+limit+1] += 1;        
          continue;
        }
        diff[2] += 2;
        diff[(x + 1)] += (-1);
        diff[x+y] += (-1);
        diff[(x+y+1)] += 1;
        diff[y+limit+1] += 1;        
      }
      ans = Math.min(ans, diff[2]);
      int sum = diff[2];
      for(int x = 3; x <= limit*2; x++) {
        sum += diff[x];
        ans = Math.min(ans, sum);
      }
      
      return ans;
    }
}
