class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] cnt = new int[limit * 2 + 2];
        int n = nums.length;
        for (int i = 0; i < nums.length / 2; i++) {
            int mx = Math.max(nums[i], nums[n-1-i]);
            int mi = Math.min(nums[i], nums[n-1-i]);
            cnt[0] += 2;
            cnt[mi+1] -= 1;
            cnt[mi+mx] -= 1;
            cnt[mi+mx+1] += 1;
            cnt[mx+limit+1] += 1;
        }
        int res = n * 2;
        int cur = 0;
        for (int i = 0; i < cnt.length; i++) {
            cur += cnt[i];
            res = Math.min(res, cur);
        }
        return res;
    }
}
