class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] min = new int[1000000];
        int[] max = new int[1000000];
        int[] count = new int[1000000];
        
        int l = 0, h = nums.length - 1;
        while (l < h) {
            ++min[Math.min(nums[l], nums[h])];
            ++max[Math.max(nums[l], nums[h]) + limit];
            ++count[nums[l] + nums[h]];
            ++l;
            --h;
        }
                
        int steps = nums.length;
        
int largerEq = nums.length / 2;
        int smaller = 0;
        
        for (int i = 2; i <= 100000; ++i) {
            largerEq -= min[i - 1];
            smaller += max[i - 1];
            steps = Math.min(steps, nums.length / 2 - count[i] + largerEq + smaller);
        }
        return steps;
    }
