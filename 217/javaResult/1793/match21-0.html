class Solution {
    public int minMoves(int[] nums, int limit) {
        int l=2;
        int r=2*limit;
        int n=nums.length/2;
        int[] scan=new int[2*limit+2];
        for(int i=0;i<n;i++){
            int fi=nums[i];
            int se=nums[2*n-1-i];
            //0 moves
            int sum=nums[i]+nums[2*n-1-i];
            
            //1 moves
            int min=Math.min(fi,se)+1;
            int max=Math.max(fi,se)+limit;
            
            scan[max+1]++;
            scan[2]+=2;
            scan[min]-=1;
            scan[sum]--;
            scan[sum+1]++;
        }
        int ans=Integer.MAX_VALUE;
        for(int i=l;i<=r;i++){
            scan[i]+=scan[i-1];
            ans=Math.min(ans,scan[i]);
        }
        return ans;
    }
}
