class Solution {
    public int minMoves(int[] nums, int l) {
        
int n = nums.length;
        int[] pre = new int[2 * l + 2];
        for(int i = 0; i < n / 2; i ++){
            int a = Math.min(nums[i], nums[n - 1 - i]), b = nums[i] + nums[n - 1 - i] - a;
            pre[a + 1] ++;
            pre[a + b] ++;
            pre[a + b + 1]--;
            pre[b + l + 1]--;
        }
        
int max = 0, cur = 0;
        for(int i = 1; i <= 2 * l; i++){
            cur += pre[i];
            max = Math.max(max, cur);
        }
        return n - max;
    }
}
