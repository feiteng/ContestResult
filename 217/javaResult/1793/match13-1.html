class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] operationList = new int[limit * 2 + 2];
        operationList[2] = nums.length;
        
for(int i = 0; i < nums.length / 2; i++) {
            int min = Math.min(nums[i], nums[nums.length - 1 - i]);
            int max = Math.max(nums[i], nums[nums.length - 1 - i]);
            operationList[min + 1]--;
            operationList[min + max]--;
            operationList[min + max + 1]++;
            operationList[max + limit + 1]++;
        }
        int output = operationList[2];
        int curr = operationList[2];
        for(int i = 3; i < operationList.length; i++) {
            curr += operationList[i];
            output = Math.min(output, curr);
        }
        return output;
    }
}
