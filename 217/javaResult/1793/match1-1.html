class Solution {
    public int minMoves(int[] nums, int limit) {
        int[] delta = new int[limit + limit + 2]; // [2,2*limit]
        int n = nums.length;
        for(int i = 0; i < n/2; ++i) {
            int a = nums[i];
            int b = nums[n - 1 - i];
            delta[0] += 2;
            delta[Math.min(a, b) + 1] -= 1;
            delta[a + b] -= 1;
            delta[a + b + 1] += 1;
            delta[
Math.max(a, b) + limit + 1] += 1;
        }
        int result = Integer.MAX_VALUE;
        int level = 0;
        for(int d : delta) {
            level += d;
            result = Math.min(result, level);
        }
        return result;
    }
}
