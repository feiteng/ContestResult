import java.util.*;


class Solution {
    public int minMoves(int[] A, int limit) {
        int n = A.length;
        int maxSum = 2*limit;
        // sumCounts[i] is Integer.MAX_VALUE if not possible to reach
        //sumCounts[0] and sumCounts[1] are invalid possible sums
        int[] sumCounts = new int[maxSum+5];
        int l = 0, r = n-1;
        while (l < r) {
            int maxValueReachableInOneMove = Math.max(A[l], A[r]) + limit;
            int minValueReachableInOneMove = Math.min(A[l], A[r]) + 1;
            int maxValueReachableInTwoMoves = 2*limit;
            int minValueReachableInTwoMoves = 2;

            sumCounts[minValueReachableInTwoMoves] += 2;
            sumCounts[minValueReachableInOneMove] -= 2;
            sumCounts[minValueReachableInOneMove] += 1;
            sumCounts[maxValueReachableInOneMove + 1] -= 1;
            sumCounts[maxValueReachableInOneMove + 1] += 2;

            sumCounts[A[l] + A[r]] -= 1;
            sumCounts[A[l] + A[r] + 1] += 1;

            l++;
            r--;

        }

        int sum = 0;
        for (int i = 0; i <= maxSum; i++) {
            sum += sumCounts[i];
            sumCounts[i] = sum;
        }
        
int minSum = Integer.MAX_VALUE;
        for (int i = 2; i <= maxSum; i++) {
            minSum = Math.min(minSum, sumCounts[i]);
        }
        return minSum;
    }
}
