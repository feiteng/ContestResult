class Solution {
    public int maximumWealth(int[][] a) {
        int max = 0;
        for(int i = 0 ; i < a.length ; i++){
            int cur = 0;
            for(int j = 0 ; j < a[i].length ; j++){
                cur += a[i][j];
            }
            max = Math.max(max,cur);
        }
        return max;
    }
}
