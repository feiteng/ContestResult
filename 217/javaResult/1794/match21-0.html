class Solution {
    public int minimumDeviation(int[] nums) {
        TreeSet<int[]> set = new TreeSet<>((a,b) -> a[0] != b[0] ? a[0]-b[0] : a[1] - b[1]);
        for (int num : nums) {
            int[] pair = new int[2];
            if (num % 2 == 1) {
                pair[1] = num <= Integer.MAX_VALUE / 2 ? num * 2 : num;
            } else {
                pair[1] = num;
                while (num % 2 == 0) {
                    num /= 2;
                }
            }
            pair[0] = num;
            // System.out.println(pair[0] + " " + pair[1]);
            set.add(pair);
        }
        // System.out.println();
        int res = set.last()[0] - set.first()[0];
        while (true) {
            int[] first = set.first();
            if (first[0] > Integer.MAX_VALUE / 2 || first[0] == first[1]) {
                break;
            }
            int[] last = set.last();
            set.remove(first);
            if (first[0] == last[0]) {
                return 0;
            }
            while (first[0] <= Integer.MAX_VALUE / 2 && first[0] * 2 <= first[1]) {
                first[0] *= 2;
                if (first[0] <= Integer.MAX_VALUE / 2 && first[0] * 2 > last[0]) {
                    break;
                }
            }
            // System.out.println(first[0] + " " + first[1]);
            set.add(first);
            res = Math.min(res, set.last()[0] - set.first()[0]);
            if (res == 0) {
                break;
            }
        }
        // System.out.println();
        return res;
    }
}
