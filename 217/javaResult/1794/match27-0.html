13_nevergiveup.java

class Solution {

    private static class Element implements Comparable<Element> {
        public int value;
        public int rank;
        public Element(int value , int rank) {
            this.value = value;
            this.rank = rank;
        }
        @Override
        public int compareTo(Element e) {
            if (this.value != e.value) {
                return - (this.value - e.value);
            } else {
                return this.rank - e.rank;
            }
        }
    }

    public int minimumDeviation(int[] nums) {
        TreeSet<Element> set = new TreeSet<>();
        int ans = Integer.MAX_VALUE;
        for (int i = 0;i < nums.length;i ++) {
            if (nums[i] % 2 == 0) {
                set.add(new Element(nums[i] , i));
            } else {
                set.add(new Element(nums[i] * 2 , i));
            }
        }
        while (set.size() == nums.length) {
            Element first = set.first() , last = set.last();
            int diff = first.value - last.value;
            ans = Math.min(ans , diff);
            if (first.value % 2 == 0) {
                set.remove(first);
                first.value /= 2;
                set.add(first);
            } else {
                break;
            }
        }
        return ans;
    }
}
