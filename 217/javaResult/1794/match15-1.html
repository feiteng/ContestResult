class Solution {
    public int minimumDeviation(int[] nums) {

//        for (int i = 0; i < nums.length; i++) {
//            while (nums[i] % 2 == 0)
//                nums[i] = nums[i] / 2;
//        }
        TreeSet<Integer> set = new TreeSet<>();
        
for (int num: nums) {
            set.add(num);
        }
        int res = set.last() - set.first();
        int cnt = 100000;
        for (int i = 0; i < cnt; i++) {

            while (set.first() % 2 == 1) {
                int mi = 
set.first();
                set.remove(mi);
                set.add(mi * 2);
                res = Math.min(res, set.last() - set.first());
            }

            int mx = set.last();
            if (mx % 2 == 0) {
                set.remove(mx);
                set.add(mx/2);
                res = Math.min(res, set.last() - set.first());
            }
        }

        for (int i = 0; i < cnt; i++) {

            while (set.first() % 2 == 1) {
                int mi = set.first();
                if (mi * 2 < set.last()) {
                    set.remove(mi);
                    set.add(mi * 2);
                    res = Math.min(res, set.last() - set.first());
                } else {
                    break;
                }
            }

            int mx = set.last();
            if (mx % 2 == 0) {
                set.remove(mx);
                set.add(mx/2);
                res = Math.min(res, set.last() - set.first());
            }
        }

        return res;
    }
}
