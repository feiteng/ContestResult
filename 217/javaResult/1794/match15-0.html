class Solution {

    public int minimumDeviation(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>(), clone;
        for (int num : nums) {
            set.add(num);
        }
        int min = set.last() - set.first();
        for (;;) {
            clone = new TreeSet<>(set);
            while (clone.last() % 2 == 0) {
                clone.add(clone.last() / 2);
                clone.remove(clone.last());
                min = Math.min(min, clone.last() - clone.first());
            }
            if (set.first() % 2 == 0) {
                break;
            }
            set.add(set.first() * 2);
            set.remove(set.first());
            min = Math.min(min, set.last() - set.first());
        }
        return min;
    }
}
