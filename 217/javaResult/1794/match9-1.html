class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int num : nums) {
            if (num % 2 != 0) {
                min = Math.min(min, num * 2);
                pq.add(num * 2);
            } else {
                pq.add(num);
                min = 
Math.min(min, num);
            }
        }
        int sol = (int)1e9;
        while (true) {
            int max = pq.poll();
            //System.out.println(max + "," + min);
            sol = Math.min(sol, max - min);
            while (max % 2 == 0 && max / 2 > min) {
                max /= 2;
            }
            if (max % 2 == 1) {
                if (max >= pq.peek()) {
                    sol = Math.min(sol, max - min);
                    break;
                }
            } else {
                if (max >= pq.peek()) {
                    min = Math.min(min, max / 2);
                    max /= 2;
                }
            }
            pq.add(max);
        }
        return sol;
        
    }
}
