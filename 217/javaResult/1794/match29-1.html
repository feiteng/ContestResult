class Solution {
    public int minimumDeviation(int[] nums) {
        List<Integer>[] arr = new List[nums.length];
        for (int i = 0; i < nums.length; ++i) {
            arr[i]= 
new ArrayList<>();
            arr[i].add(nums[i]);
            if (nums[i] % 2 == 0) {
                while (nums[i] % 2 == 0) {
                    nums[i] /= 2;
                    arr[i].add(nums[i]);
                }
            } else {
                arr[i].add(2 * nums[i]);
            }
            Collections.sort(arr[i]);
            // System.out.println(arr[i]);
        }
        
        int[] idx = new int[nums.length];
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> {
            return arr[a].get(idx[a]) - arr[b].get(idx[b]);
        });
        int max = 0;
        for (int i = 0; i < nums.length; ++i) {
            pq.add(i);
            max = Math.max(max, arr[i].get(0));
        }
        // System.out.println(max);
        int min = max - arr[pq.peek()].get(idx[pq.peek()]);
        while (true) {
            int smallest = pq.remove();
            if (idx[smallest] == arr[smallest].size() - 1) break;
            ++idx[smallest];
            max = Math.max(max, arr[smallest].get(idx[smallest]));
            pq.add(smallest);
            min = Math.min(min, max - arr[pq.peek()].get(idx[pq.peek()]));
        }
        return min;
        
    }
}
