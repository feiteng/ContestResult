class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
        int[] pows = new int[n];
        PriorityQueue<Integer> queue = new PriorityQueue<>((i, 
j) -> nums[i] - nums[j]);
        int max = 0;
        for (int i = 0; i < n; i++) {
            while (nums[i] % 2 == 0) {
                nums[i] /= 2;
                pows[i]++;
            }
            if (pows[i] == 0) pows[i] = 1;
            queue.add(i);
            max = Math.max(max, nums[i]);
        }
        
        int deviation = Integer.MAX_VALUE;
        while (true) {
            int top = queue.poll();
            deviation = Math.min(deviation, max - nums[top]);
            if (pows[top] > 0) {
                nums[top] *= 2;
                pows[top]--;
                
queue.add(top);
                max = Math.max(max, nums[top]);
            } else {
                break;
            }
        }
        
        return deviation;
    }
}
