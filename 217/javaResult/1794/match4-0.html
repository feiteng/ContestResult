class Solution {
    public int minimumDeviation(int[] nums) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((p, q) -> q - p);
        int min = Integer.MAX_VALUE;
        for (int num : nums) {
            if (num % 2 == 1)
                num = num * 2;
            pq.offer(num);
            min = Math.min(min, num);
        }
        
        int res = Integer.MAX_VALUE;
        while (!pq.isEmpty()) {
            int top = pq.poll();
            if (top % 2 == 1)
                return Math.min(res, top - min);
            else if (top / 2 >= min) {
                while (top % 2 == 0 && top / 2 >= min)
                    top = top / 2;
                pq.offer(top);
            } else {
                res = Math.min(res, top - min);
                top = top / 2;
                min = top;
                pq.offer(top);
            }
        }
        return 0;
    }
}
