class Solution {
    public int minimumDeviation(int[] nums) {
        PriorityQueue<int[]> queue = new PriorityQueue<>(Comparator.<int[]>comparingInt(a -> a[0]).thenComparingInt(a -> a[1]));
        Arrays.stream(nums).boxed().map(n -> {
            int flag = n & -n;
            return new int[]{n / flag, flag};
        }).forEach(queue::offer);

        int min = queue.peek()[0], max = queue.stream().mapToInt(p -> p[0]).max().getAsInt();
        int res = max - min;
        int[] p;
        while (true) {
            p = queue.poll();
            if (p[1] <= 1 && p[0] % 2 == 0) break;
            p[0] *= 2;
            p[1] /= 2;
            queue.offer(p);
            max = Math.max(max, p[0]);
            min = queue.peek()[0];
            res = Math.min(res, max - min);
        }

        return res;
    }
}
