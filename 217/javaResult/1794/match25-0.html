class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; ++i) {
            if (nums[i]%2==1)
                nums[i]<<=1;
        }
        Arrays.sort(nums);
        int min = nums[0];
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
        for (int x : nums)
            pq.add(-x);
        int ans = 1000000000;
        while (true) {
            int next = -pq.poll();
            ans = Math.min(ans,next-min);
            if (next%2==1)
                break;
            next/=2;
            min = Math.min(min,next);
            pq.add(-next);
        }
        return ans;
    }
}
