class Solution {
    class Pair implements Comparable<Pair> {
    int idx;
    int val;
    public Pair(int idx, int val) {
      this.idx = idx;
      this.val = val;
    }

    @Override
    public int compareTo(Pair o) {
      if (val != o.val) {
        return Integer.compare(val, o.val);
      }
      return Integer.compare(idx, o.idx);
    }

    @Override
    public String toString() {
      return idx + " " + val;
    }
  }
  public int minimumDeviation(int[] nums) {
    List<Pair> ls = new ArrayList<>();
    for (int i = 0; i < nums.length; i++) {
      int v = nums[i];
      ls.add(new Pair(i, v));
      if (v % 2 == 1) {
        ls.add(new Pair(i, 2 * v));
      } else {
        while (v % 2 == 0) {
          v = v / 2;
          ls.add(new Pair(i, v));
        }
      }
    }

    Collections.sort(ls);

    // find the minimum window size with at least k different character
    int k = nums.length;
    int n = ls.size();
    int end = 0;
    int res = Integer.MAX_VALUE;
    Map<Integer, Integer> hm = new HashMap<>();
    for (int i = 0; i < n; i++) {
      while (hm.size() < k && end < n) {
        Pair cur = ls.get(end);
        hm.put(cur.idx, hm.getOrDefault(cur.idx, 0) + 1);
        end++;
      }

      if (hm.size() >= k) { // [i, end - 1]
        res = Math.min(res, ls.get(end - 1).val - ls.get(i).val);
      }
      Pair cur = ls.get(i);
      if (hm.get(cur.idx) == 1) {
        hm.remove(cur.idx);
      }
      else {
        hm.put(cur.idx, hm.get(cur.idx) - 1);
      }
    }

    return res;
  }
}
