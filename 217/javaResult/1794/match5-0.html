class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
        List<List<Integer>> lists = new ArrayList<>();
        for (int num : nums) {
            List<Integer> list = new ArrayList<>();
            list.add(num);
            if (num % 2 == 1) {
                list.add(num * 2);
            } else {
                while (num % 2 == 0) {
                    list.add(0, num / 2);
                    num /= 2;
                }
                list.add(0, num);
            }
            lists.add(list);
        }
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[0] - b[0]));
        int max = 0;
        for (int i = 0;i < n;i++) {
            pq.add(new int[]{lists.get(i).get(0), 0, i});
            max = Math.max(max, lists.get(i).get(0));
        }
        int res = max - pq.peek()[0];
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            int idx = cur[1] + 1;
            if (idx < lists.get(cur[2]).size()) {
                pq.add(new int[]{lists.get(cur[2]).get(idx), idx, cur[2]});
                max = Math.max(max, lists.get(cur[2]).get(idx));
                res = Math.min(res, max - pq.peek()[0]);
            } else {
                break;
            }
        }
        return res;
    }
}
