class Solution {
    public int minimumDeviation(int[] nums) {
        int max=0,n=nums.length,ans=Integer.MAX_VALUE;
        int[] p = new int[n];
        for(int i=0;i<n;i++){
            if(nums[i]%2==1) { p[i]=nums[i];nums[i]*=2;}
            else{
                int x=nums[i];
                while(x%2==0){
                    x=x>>1;
                }
                p[i]=x;
            }
        }
        PriorityQueue<int[]> queue = new PriorityQueue<>((o1,o2)->o1[0]-o2[0]);
        
        for(int i=0;i<n;i++){
            max=Math.max(max,p[i]);
            queue.add(new int[]{p[i],i});
        }
        
while(true){
            int[] poll = queue.poll();
            ans=Math.min(ans,max-poll[0]);
            if(nums[poll[1]]==poll[0]) break;
            queue.add(new int[]{2*poll[0],poll[1]});
            max=Math.max(max,2*poll[0]);
        }
        return ans;
    }
}
