class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
      for(int i = 0; i < n; i++)
        if(nums[i]%2 == 1) nums[i] *= 2;
      
      TreeSet<Integer> data = new TreeSet<Integer>();
      for(int i = 0;i < n; i++)
        data.add(nums[i]);
      int ans = data.last() - data.first();
      
      while(data.last() % 2 == 0) {
        int v = data.pollLast();
        data.add(v/2);
        ans = Math.min(ans, data.last() - data.first());
      }
      
      return ans;
    }
}
