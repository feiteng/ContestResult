class Solution {
    public int minimumDeviation(int[] nums) {
        TreeSet<int[]> set = new TreeSet<>((a,b)->a[0]-b[0]==0?a[1]-b[1]:a[0]-b[0]);
        for (int n : nums) {
            set.add(new int[]{fac(n), n});
        }
        // System.out.println(max);
        int min = Integer.MAX_VALUE/2;
        while (true) {
            min = Math.min(min, set.last()[0] - set.first()[0]);
            int[] curmin = set.pollFirst();
            if (curmin[0] >= curmin[1] && curmin[0] % 2 == 0) {
                break;
            }
            set.add(new int[]{curmin[0]*2, curmin[1]});
        }
        return min;
        
    }
    
    private int fac(int n) {
        while (n % 2 == 0) {
            n /= 2;
        }
        return n;
    }
}
