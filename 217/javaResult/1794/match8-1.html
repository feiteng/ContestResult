class Solution {
    public int minimumDeviation(int[] nums) {
        TreeSet<int[]> ts = new TreeSet<>((a, b) -> {
            if (a[0] != b[0]) return a[0] - b[0];
            return a[1] - b[1];
        });
        for (int i = 0; i < nums.length; i++) {
            int a = nums[i];
            while (a % 2 == 0) a /= 2;
            ts.add(new int[] {a, i, nums[i] % 2 == 0 ? nums[i] : nums[i] * 2});
        }
        int ans = Integer.MAX_VALUE;
        while (true) {
            ans = Math.min(ans, ts.last()[0] - ts.first()[0]);
            int[] h = ts.pollFirst();
            if (h[0] * 2 > h[2]) break;
            h[0] *= 2;
            
ts.add(h);
            ans = Math.min(ans, ts.last()[0] - ts.first()[0]);
        }
        return ans;
    }
}
