class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length, max = 0, min = Integer.MAX_VALUE;
        int[] pow = new int[n];
        for(int i = 0; i < n; i++){
            int k = 0;
            while((nums[i] & 1) == 0)
            {
                nums[i] >>= 1;
                k++;
            }
            max = Math.max(max, nums[i]);
            min = Math.min(min, nums[i]);
            pow[i] = Math.max(1, k);
        }
        PriorityQueue <int[]> pq = new PriorityQueue<int[]>( (a1, a2) -> a1[0] - a2[0]);
        int res = max - min;
        for(int i = 0; i < n; i++)
            pq.add(new int[]{nums[i], i});
        while(true){
            int[] cur = pq.poll();
            int ind = cur[1];
            if(pow[ind] == 0) break;
            nums[ind] <<= 1;
            pq.add(new int[]{nums[ind], ind});
            
max = Math.max(max, nums[ind]);
            min = pq.peek()[0];
            pow[ind]--;
            res = Math.min(res, max - min);
        }
        /*
        System.out.println(Arrays.toString(nums));
        System.out.println(Arrays.toString(pow));*/
        return res;
    }
}
