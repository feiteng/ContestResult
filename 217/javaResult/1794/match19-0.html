class Solution {
    public int minimumDeviation(int[] nums) {
        int n = nums.length;
        int[] pows = new int[n];
        int minId = 0, maxId = 0;
        for (int i = 0; i < n; i++) {
            while ((nums[i] & 1) == 0) {
                pows[i]++;
                nums[i] /= 2;
            }
            if (pows[i] == 0) {
                pows[i] = 1;
            }
            if (nums[i] > nums[maxId]) {
                maxId = i;
            }
        }
        int max = nums[maxId];
        // System.out.println(max);
        for (int i = 0; i < n; i++) {
            while (pows[i] > 0 && nums[i] * 2 < max) {
                nums[i] *= 2;
                pows[i]--;
            }
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] < nums[minId]) {
                minId = i;
            }
        }
        PriorityQueue<int[]> pq = new PriorityQueue<>((int[] a, int[] b) -> (a[0] != b[0]? (a[0] - b[0]) : (a[1] - b[1])));
        for (int i = 0; i < n; i++) {
            pq.add(new int[]{nums[i], pows[i]});
        }
        // System.out.println(pq.peek()[0]);
        if (pq.peek()[0] * 2 < max) {
            return max - pq.peek()[0];
        }
        int dev = max - pq.peek()[0];
        int min = pq.peek()[0];
        while (pq.size() > 1) {
            if (pq.peek()[1] > 0) {
                max = pq.poll()[0] * 2;
                min = pq.peek()[0];
                dev = Math.min(dev, max - min);
            }
            else {
                break;
            }
        }
        return dev;
    }
}
