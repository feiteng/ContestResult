class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> stack = new LinkedList<>();
        int leftNum = nums.length;
        for(int i=0;i<nums.length;i++){
            leftNum--;
            if(stack.isEmpty()){
                stack.push(nums[i]);
            }else {
                while(!stack.isEmpty()&&nums[i]<stack.peek()&&leftNum+stack.size()>=k){
                    stack.pop();
                }
                if(stack.size()<k){
                    stack.push(nums[i]);
                }
            }
            
            
        }
        int[] output = new int[k];
        int curI = k-1;
        while(!stack.isEmpty()){
            output[curI--] = stack.pop();
        }
        return output;

    }
}
