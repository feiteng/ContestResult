class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < nums.length; i += 1) {
            int num = nums[i];
            while (!stack.isEmpty() && stack.peek() > num && (stack.size() + nums.length - i) > k) {
                  stack.pop();
            }
            stack.push(num);
        }
        while (stack.size() > k) stack.pop();
        
int[] ret = new int[k];
        int index = ret.length - 1;
        while (!stack.isEmpty()) {
            ret[index] = stack.pop();
            index--;
        }
        return ret;
    }
}
