class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        LinkedList<Integer> dq = new LinkedList<>();
        for(int i = 0; i < nums.length; i++) {
            int num = nums[i];
            if(dq.size() == 0) {
                dq.addLast(num);
            } else if(dq.peekLast() <= num){
                dq.addLast(num);
            } else {
                
while(nums.length - i + dq.size() > k && dq.size() > 0 && dq.peekLast() > num) {
                    dq.pollLast();
                }
                dq.addLast(num);
            }
        
}
        int[] output = new int[k];
        for(int i = 0; i < output.length; i++) output[i] = dq.pollFirst();
        return output;
    }
}
