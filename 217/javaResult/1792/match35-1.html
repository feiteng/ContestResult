class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> dq = new LinkedList<>();
        
        int n = nums.length;
        for (int i = 0; i < nums.length; ++i) {
            
            
while (dq.size() > 0 && nums[dq.peekLast()] > nums[i] && dq.size() + n - i - 1 >= k) {
                dq.pollLast();
            }
            if (dq.size() < k) {
                dq.addLast(i);
            }
            
            
        }
        
        
int[] ans = new int[k];
        int idx = 0;
        while (dq.size() > 0) {
            ans[idx] = nums[dq.pollFirst()];
            ++idx;
        }
        return ans;
    }
}
