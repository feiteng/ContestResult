class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> st = new Stack<Integer>();
        // [3,5,2,6], k = 2
        for (int i = 0; i < nums.length; i++) {
            while (!st.isEmpty() && nums[i] < nums[st.peek()] && (st.size() + nums.length - i - 1) >= k) {
                st.pop();
            }
            st.add(i);
        }
        
        int[] res = new int[k];
        int idx = 0;
        for (Integer index : st) {
            if (idx == k) break;
            res[idx++] = nums[index];
        }
        return res;
    }
}
