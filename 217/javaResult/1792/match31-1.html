class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> smaller = new Stack<>();
        
        for (int i = 0; i < nums.length; i++) {
            while (!smaller.isEmpty() && smaller.peek() > nums[i] && smaller.size()-1 + nums.length - i >= k) {
                smaller.pop();
            }
            
            smaller.push(nums[i]);
            // System.out.println(smaller);
        }
        
        
        while (smaller.size() > k) smaller.pop();
        
        
int[] ans = new int[k];
        for (int i = k-1; i >= 0; i--) ans[i] = smaller.pop();
        return ans;
    }
}
