class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        int size = k;
        int n = nums.length;
        k = n-k;
        for(int i=0;i<n;i++){
            while(!stack.isEmpty()&&k>0&&nums[i]<stack.peek()){
                k--;
                stack.pop();
            }
            stack.push(nums[i]);
        }
        while(k>0){
            stack.pop();
            k--;
        }
        int[] result = new int[size];
        int index = size-1;
        while(!stack.isEmpty()){
            result[index--] = stack.pop();
        }
        return result;
    }
}
