class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> stack = new ArrayDeque<>();
        int len = nums.length;
        int m = len - k;
        int cnt = 0;
        
for (int i = 0; i < len; i++) {
            while (cnt < m && !stack.isEmpty() && nums[i] < nums[stack.peekLast()]) {
                stack.pollLast();
                cnt++;
            }
            stack.addLast(i);
        }
        while (cnt < m && !stack.isEmpty()) 
{
            stack.pollLast();
            cnt++;
        }

        int[] res = new int[k];
        int i = 0;
        while (!stack.isEmpty()) {
            res[i++] = nums[stack.pollFirst()];
        }
        return res;
    }
}
