class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> queue = new LinkedList<>();
        for (int i = 0; i < nums.length; i++) {
            while (!queue.isEmpty() && queue.peekLast() > nums[i] && (nums.length - i) >= (k - queue.size() + 1)) {
                queue.pollLast();
            }
            queue.offer(nums[i]);
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = queue.pollFirst();
        }
        return res;
    }
}
