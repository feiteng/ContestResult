class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
          Stack<Integer> stack = new Stack<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if ((stack.isEmpty() || stack.peek() <= nums[i]) && stack.size() < k) {
                stack.push(nums[i]);
            } else if (stack.peek() > nums[i]) {
                
while (!stack.isEmpty() && stack.peek() > nums[i] && n - i + stack.size() > k) {
                    stack.pop();
                }
                stack.push(nums[i]);
            }
        
}
        int[] res = new int[k];
        for (int i = k - 1; i >= 0; i--) {
            res[i] = stack.pop();
        }
        return res;
    }
}
