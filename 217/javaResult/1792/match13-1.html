import java.util.LinkedList;


class Solution {
    public int[] mostCompetitive(int[] nums, int k) {

        LinkedList<Integer> list = new LinkedList<>();
        for (int i = 0; i < nums.length; i++) {
            while (!list.isEmpty() && list.size() + nums.length - i > k && nums[i] < list.peekLast()) {
                list.pollLast();
            }
            list.addLast(nums[i]);
        }
        int[] ans = new int[k];
        for (int i = 0; i < k; i++) {
            ans[i] = list.pollFirst();
        }
        return ans;
    }
}
