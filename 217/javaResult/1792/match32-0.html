class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> stack = new ArrayDeque<>();
        int n = nums.length;
        for(int i = 0; i < n; i++){
            while(!stack.isEmpty() && n - i + stack.size() > k && nums[i] < nums[stack.peek()]){
                stack.pop();
            }
            stack.push(i);
        }
        while(stack.size() > k) stack.pop();
        int[] res = new int[k];
        for(int i = k - 1; i >= 0; i--){
            res[i] = nums[stack.pop()];
        }
        return res;
    }
}
