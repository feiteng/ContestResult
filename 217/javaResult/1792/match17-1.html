import java.util.*;


class Solution {
    public int[] mostCompetitive(int[] nums, int k) {

        Stack<Integer> stack = new Stack<>();

        for(int idx = 0; idx < nums.length; idx++) {
            int num = nums[idx];

            int minsizeNow = k - nums.length + idx;

            
while(!stack.isEmpty() && stack.peek() > num && stack.size() > minsizeNow) { stack.pop(); }

            if (stack.size() < k) { stack.push(num); }

        }

        int[] res = new int[k];
        for(int idx = k-1; idx >= 0; idx--) {
            res[idx] = stack.pop();
        }
        return res;
    }
}
