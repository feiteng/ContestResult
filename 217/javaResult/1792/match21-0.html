class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> q = new LinkedList();
        for (int i = 0; i < nums.length; i++) {
            while (!q.isEmpty() && q.peekLast() > nums[i] && (q.size() + nums.length - i - 1 >= k)) {
                q.pollLast();
            }
            q.addLast(nums[i]);
            //System.out.println(q);
        }
        int[] res = new int[k];
        int count = 0;
        while (count < k) {
            res[count] = q.pollFirst();
            count++;
        }
        return res;
    }
}
