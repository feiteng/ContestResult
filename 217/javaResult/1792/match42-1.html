class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        for(int i = 0; i < nums.length; i++){
            // 剩余可能的元素数目 num.length - i
            while(!stack.isEmpty() && nums.length - i + stack.size() > k && stack.peek() > nums[i]){
                stack.pop();
            }
            stack.push(nums[i]);
        }
        while(stack.size() > k){
            stack.pop();
        }
        
int[] ans = new int[k];
        for(int i = k - 1; i >= 0; i--){
            ans[i] = stack.pop();
        }
        return ans;
    }
}
