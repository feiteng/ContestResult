class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> s = new Stack<>();
        int n = nums.length;
        for (int i = 0; i < nums.length; i++) {
            while (!s.empty() && s.peek() > nums[i] && k - s.size() <= n - 1 - (i + 1) + 1) {
                s.pop();
            }
            if (s.size() < k)
                s.push(nums[i]);
        }
        int[] res = new int[k];
        int i = k - 1;
        while (!s.empty()) {
            res[i--] = s.pop();
        }
        return res;
    }
}
