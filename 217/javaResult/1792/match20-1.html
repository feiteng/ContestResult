class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> deque = new ArrayDeque<>();
        for (int i = 0; i < nums.length; i++) {
            
while (!deque.isEmpty() && nums.length - 1 - i >= k - deque.size() && nums[i] < deque.peekLast()) {
                deque.pollLast();
            }
            deque.offerLast(nums[i]);
        }
        int[] res = new int[k];
        int index = 0;
        while (index < k && !deque.isEmpty()) {
            res[index++] = deque.pollFirst();
        }
        return res;
    }
}
