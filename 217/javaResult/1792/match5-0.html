class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> stk = new ArrayDeque<>();
        int n = nums.length;
        for (int i = 0; i < nums.length; i++) {
            while (!stk.isEmpty() && stk.size() + n - i > k && nums[i] < stk.peekLast()) {
                stk.pollLast();
            }
            stk.offerLast(nums[i]);
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = stk.pollFirst();
        }
        return res;
    }
}
