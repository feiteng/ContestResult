class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> q = new ArrayDeque<>();
        for(int i = 0; i < nums.length; i++)
        {
            int num = nums[i];
            /**
            if (q.isEmpty())
            {
                q.offer(num);
            }
            
            else
            {
                while (!q.isEmpty() && q.peekLast() > num)
                {
                    q.pollLast();
                }
                q.offer(num);
            }
            **/
            while (!q.isEmpty() && num < q.peekLast() && (q.size() + (nums.length - i - 1)) >= k)
            {
                q.pollLast();
            }
            q.offer(num);
        }
        int[] ans = new int[k];
        int cur = 0;
        while (!q.isEmpty() && cur < k)
        {
            ans[cur++] = q.poll();
        }
        return ans;
    }
}
