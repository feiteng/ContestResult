class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> dq = new LinkedList<>();
        for(int i = 0; i < nums.length; i++){
            while(!dq.isEmpty() && dq.size()+nums.length-i > k && dq.getLast() > nums[i]){
                dq.pollLast();
            }
            dq.addLast(nums[i]);
        }
        int[] ans = new int[k];
        for(int i = 0; i < k; i++){
            ans[i] = dq.pollFirst();
        }
        return ans;
    }
}
