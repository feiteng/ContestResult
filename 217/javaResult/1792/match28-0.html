class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        int n = nums.length;
        for (int i = 0; i < n; ++i) {
            while (!stack.isEmpty() && nums[stack.peek()] > nums[i] && stack.size() + n - i > k) {
                stack.pop();
            }
            stack.push(i);
        }
        while (stack.size() > k) {
            stack.pop();
        }
        int[] res = new int[k];
        for (int i = k - 1; i >= 0; --i) {
            res[i] = nums[stack.pop()];
        }
        return res;
    }
}
