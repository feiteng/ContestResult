class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            int cur = nums[i];
            while (!stack.isEmpty() && cur < stack.peek() && stack.size() + n-i > k) {
                stack.pop();
            }
            if (stack.size() < k) {
                stack.push(cur);
            }
        }
        for (int i : stack) {
             System.out.println(i);
        }
       
        int[] res = new int[k];
        int idx = k-1;
        while (!stack.isEmpty()) {
            res[idx--] = stack.pop();
        }
        return res;
    }
}
