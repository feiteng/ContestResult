class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> dq = new LinkedList();
        int N = nums.length;
        for(int i=0; i<nums.length; i++) {
            
            while (!dq.isEmpty() && dq.peekLast() > nums[i] && dq.size()-1+nums.length-i >= k) {
                dq.pollLast();
            }
            
            if(dq.size() < k) {
                dq.addLast(nums[i]);
            }
        }
        
        int[] res = new int[k];
        for(int i=0; i<k; i++) {
            res[i] = dq.pollFirst();
        }
        
        return res;
    }
}
