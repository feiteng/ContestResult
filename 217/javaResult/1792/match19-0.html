import java.util.*;

class Solution {
    public int[] mostCompetitive(int[] A, int k) {
        Stack<Integer> stack = new Stack<>();
        int n = A.length;
        for (int i = 0; i < n; i++) {
            while (!stack.isEmpty() && A[i] < stack.peek() && stack.size() + n - i - 1 >= k) {
                stack.pop();
            }
            if (stack.size() < k) {
                stack.push(A[i]);
            }
        }
        int[] res = new int[k];
        for (int i = k - 1; i >= 0; i--) {
            res[i] = stack.pop();
        }
        return res;
    }
}
