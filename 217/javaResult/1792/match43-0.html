class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Deque<Integer> que = new ArrayDeque<>();
        for (int i = 0; i < nums.length; i ++) {
            while (!que.isEmpty() && que.peekLast() > nums[i] && que.size() + nums.length - i > k) {
                que.pollLast();
            }
            que.offer(nums[i]);
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i ++) {
            res[i] = que.poll();
        }
        return res;
    }
}
