class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        vector<int> d(2*limit + 7);
        for(int i=0;i<n/2;i++) {
            int a = nums[i], b = nums[n-1-i];
            int l = min(a,b)+1;
            int s = a+b;
            int h = limit + max(a,b);

            d[2] += 2;
            d[l] -= 1;
            d[s] -= 1;
            d[s+1] += 1;
            d[h+1] += 1;
        }
        int ans = 999999999;
        int value = 0;
        for(int i=2;i<=2*limit;i++) {
            value += d[i];
            ans = min(ans, value);
        }
        return ans;
    }
};
