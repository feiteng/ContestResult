int cnt[200000 + 5];


class Solution {
public:
    

int minMoves(vector<int>& nums, int limit)
{
    int N = (int)nums.size();
    memset(cnt, 0, sizeof(cnt));
    for(int i = 0; i < N / 2; i++)
    {
        int a = min(nums[i], nums[N - 1 - i]) + 1, b = max(nums[i], nums[N - 1 - i]) + limit, s = nums[i] + nums[N - 1 - i];

        cnt[2] += 2; cnt[a] -= 2;
        cnt[a]++; cnt[s]--;
        cnt[s + 1]++; cnt[b + 1]--;
        cnt[b + 1] += 2; cnt[2 * limit + 1] -= 2;
    }
    int ans = N, cur = 0;

    for(int i = 2; i <= 2 * limit; i++)
    {
        cur += cnt[i];
        ans = min(ans, cur);
    }
    return ans;
}
};
