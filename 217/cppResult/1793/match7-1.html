//差分嘛
//变成这个要减去1
//最小是2
//最大是2*limit
//变成2需要几次
//变成3需要几次是一定的，可以差分
//max是1对吧

class Solution {
public:
    int minMoves(vector<int>& nums, int limits) {
        int n=nums.size();
        // map<int, int> ans; //算能在哪个区间里吧
        // //怎么求呢
        // vector<pair<int, int>> sum;
        // for(int i=0; 2*i<n; i++){
        //     sum.push_back(nums[i]+nums[n-i-1], min(nums[i], nums[n-i-1])});
        // }
        // sort(sum.begin(), sum.end(), [](pair<int, int>&a, pair<int, int>&b){
        //     return a.first+a.second==b.first+b.second? a.first<b.first
        // })
        vector<int> sum(5*limits, 0);
        for(int i=0; 2*i<n; i++){
            int mi = min(nums[i], nums[n-i-1])+1;
            int ma = max(nums[i], nums[n-i-1])+limits;
            int zero = nums[i]+nums[n-i-1];
            sum[2] += 2;

            sum[mi] -= 2;
            sum[mi] += 1;
            sum[zero] -= 1;
            sum[zero+1] += 1;
            sum[ma+1] -= 1;
            sum[ma+1] += 2;
            // sum[max(ma+1,2*limits)]-=2;
            
        }
        int ans = 1e8;
        int temp = 0;
        for(int i=2; i<=2*limits; i++){
            temp += sum[i];
            ans = min(ans, temp);
        }
        return ans;
    }
};
