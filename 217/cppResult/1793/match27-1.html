class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        int change[200005] = {0};
        for (int i=0; i<n/2; i++) {
            int left = i, right = n-1-left;
            int a = nums[left], b = nums[right];
            if (b < a) swap(a, b);
            

            change[a+1] += -1;
            change[a+b] += -1;
            change[a+b+1] += 1;
            change[b+limit+1] += 1;
        }
        
        int ans = 0;
        int now = 0;
        for (int i=2; i<=limit * 2; i++) {
            now += change[i];
            ans = min(ans, now);
        }
        
        return n + ans;
    }
};
