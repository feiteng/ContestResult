class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n=nums.size();
        int f[limit*2+2];
        memset(f,0,sizeof(f));
        for(int i=0;i<n/2;i++){
            int a=min(nums[i],nums[n-i-1]);
            int b=max(nums[i],nums[n-i-1]);
            f[2]+=2;
            f[a+1]-=2;
            f[a+1]+=1;
            f[a+b]-=1;
            f[a+b+1]+=1;
            f[b+limit+1]-=1;
            f[b+limit+1]+=2;
            f[2*limit+1]-=2;
        }
        int res=111111111;
        int sum=0;
        /*for(int i=2;i<=limit*2;i++)
            cout<<f[i]<<endl;*/
        for(int i=2;i<=limit*2;i++){
            sum+=f[i];
            res=min(res,sum);
        }
        return res;
    }
};
