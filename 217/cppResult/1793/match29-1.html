class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        int sLim = limit * 2 + 8;
        vector<int> delta(sLim);
        for(int i = 0, j = n -1; i < j; i++, j--) {
            int x = nums[i], y = nums[j];

            delta[2] += 2;
            delta[min(x, y) + 1]--;
            delta[x + y]--;
            delta[x + y + 1]++;
            delta[max(x, y) + limit + 1]++;
        }
        int res = INT_MAX, curr = 0;
        for(int i = 2; i < sLim; i++) {
            curr += delta[i];
            res = min(res, curr);
            // cout << delta[i] << ' ';
        }
        // cout << endl;
        return res;
    }
};
