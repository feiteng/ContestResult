class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int len = nums.size();
        vector<int> rec(limit*2+2, 0);
        int halflen = len/2;

        for(int i=0; i<halflen; i++) {
            int a = nums[i];
            int b = nums[len-1-i];
            
            int minab = 1 + min(a, b);
            int maxab = max(a, b) + limit;
            rec[0] += 2;
            rec[minab] -= 1;
            rec[a + b] -= 1;
            rec[a + b + 1] += 1;
            rec[maxab+1] += 1;
        }

        int res = rec[0];
        int cur = rec[0];
        for(int i=1; i<=limit*2; i++) {
            cur += rec[i];
            res = min(res, cur);
        }
        return res;
    }
};
