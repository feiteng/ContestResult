class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        int MAXN = 2e5+1;
        vector<int> dp(MAXN);
        for(int i = 0; i < n/2; i++) {
            int u = nums[i], v = nums[n-1-i];
            if(u > v) {
                swap(u, v);
            }
            dp[2] += 2;
            dp[u+1] -= 1;
            dp[u+v] -= 1;
            dp[u+v+1] += 1;
            dp[v+limit+1] += 1;
        }
        int ans = INT_MAX, cnt = 0;
        for(int i = 2; i < MAXN; i++) {
            cnt += dp[i];
            ans = min(ans, cnt);
        }
        return ans;
    }
};
