class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        vector<int>data(limit*2+2, 0);
        int n = nums.size();
        for (int i=0; i<n/2; i++) {
            int mi = min(nums[i], nums[n-1-i]);
            int ma = max(nums[i], nums[n-1-i]);
            data[2]+=2;
            data[mi+1]-=1;
            data[mi+ma]-=1;
            data[mi+ma+1]+=1;
            data[ma+limit+1]+=1;
        }
        int res=data[2];
        int current=data[2];
        for (int i=3; i<=limit*2+1; i++) {
            current += data[i];
            res = min(res, current);
        }
        return res;

    }
};
