class Solution {
public:
    int minMoves(vector<int>& nums, int m) {
        int n = nums.size();
        vector<int> a(2 * m + 2, 0);
        for (int i = 0; i < nums.size() / 2; i++) {
            int p = min(nums[i], nums[n - i - 1]);
            int q = max(nums[i], nums[n - i - 1]);
            int l1 = 1 + p, l2 = m + q;
            a[2] += 2; a[l1] -= 2;
            a[l1] += 1; a[p + q] -= 1;
            a[p + q + 1] += 1; a[l2 + 1] -= 1;
            a[l2 + 1] += 2; a[2 * m + 1] -= 2; 
        }
        // for (auto t : a) cout << t << ' ';
        // cout << endl;
        int ans = INT_MAX, cnt = 0;
        for (int i = 2; i <= 2 * m; i++) {
            cnt += a[i];
            // cout << cnt << ' ';
            ans = min(ans, cnt);
        }
        // cout << endl;
        return ans;
    }
};
