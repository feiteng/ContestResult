class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        
        int z[200000+10] = {0};
        for (int i=0; 2*i<n; ++i) {
            int a = nums[i];
            int b = nums[n-i-1];
            if (a>b) swap(a,b);
            z[2] +=2;
            z[limit*2+1] -=2;
            
            z[1+a] -=1;
            z[b+limit+1] +=1;
            
            z[a+b] -=1;
            z[a+b+1] +=1;
        }
        int t = 0;
        int res = n;
        for (int i=2; i<= 2*limit; ++i) {
            t+= z[i];
            //cout<<t<<' ';
            res = min(res, t);
            //cout<<z[i]<<' ';
        }
        cout<<endl;
        return res;
    }
};
