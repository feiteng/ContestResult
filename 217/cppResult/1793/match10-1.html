class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        vector<int> v(2 * limit + 5, 0);
        for (int i = 0; i < n / 2; i++)
        {
            v[min(nums[i], nums[n - i - 1]) + 1]--;
            v[nums[i] + nums[n - i - 1]]--;
            v[nums[i] + nums[n - i - 1] + 1]++;
            v[max(nums[i], nums[n - i - 1]) + limit + 1]++;
        }

        int ret = 1e9, curr = n;
        for (int k = 2; k <= 2 * limit; k++)
        {
            curr += v[k];
            ret = min(ret, curr);
        }
        return ret;
    }
};
