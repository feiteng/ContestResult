    int minMoves(vector<int>& nums, int limit) {
        vector<int> dp(200100, 0);
        int n = nums.size();
        for (int i = 0; i < n/2; ++i) {
            int mi = min(nums[i], nums[n-i-1]);
            int mx = max(nums[i], nums[n-i-1]);
            dp[2] += 2;
            dp[1+mi] -= 1;
            dp[mi+mx] -= 1;
            dp[mi+mx+1] += 1;
            dp[mx+limit+1] += 1;
        }
        int mi = INT_MAX;
        int pre = 0;
        for (int i = 2; i <= 2*limit; ++i) {
            pre += dp[i];
            mi = min(mi, pre);
        }
        return mi;
    }
};
