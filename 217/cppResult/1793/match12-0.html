class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int MAX = 200000;
        vector<int> ds(MAX+1, 0);
        // scores[i] = scores[i-1] + ds[i]
        int n = nums.size() / 2;
        for(int i=0;i<n;++i) {
            int j = 2*n-1-i;
            int s = nums[i] + nums[j];
            //cout << i << " " << j << " " << endl;
            int l0 = min(nums[i], 1) + nums[j], h0 = max(nums[i], limit) + nums[j];
            int l1 = min(nums[j], 1) + nums[i], h1 = max(nums[j], limit) + nums[i];
            int l = min(l0, l1), h = max(h0, h1);
            ds[l]--;
            ds[s]--;
            ds[s+1]++;
            ds[h+1]++;
            //cout << l << " " << s << " " << s+1 << " " << h+1 << endl;
        }
        int score = 2 * n;
        int ans = score;
        for(int i=1;i<=MAX;++i) {
            score += ds[i];
            ans = min(ans, score);
            //cout << score << " ";
        }
        //cout << endl;
        return ans;
    }
};
