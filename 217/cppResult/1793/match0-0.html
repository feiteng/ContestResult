class Solution {
public:
    int minMoves(vector<int>& v, int k) {
		int n = v.size();
		vector<int> store(k * 2 + 1);
 
		for(int i = 0; i < n/2; i ++) {
			store[0] -= 2;
			store[2 * k] += 2;
			int sum = v[i] + v[n - i - 1]; 
			int maxV = max(v[i], v[n - i - 1]) + k;
			int minV = min(v[i], v[n - i - 1]) + 1;
			store[maxV] -= 1; 
			store[sum] -= 1; 
			store[sum - 1] ++;
			store[minV - 1] += 1; 
		}
		for(int i = store.size() - 2; i >= 0; i --) store[i] += store[i + 1];
        
		return *min_element(store.begin() + 1, store.end());
 
    }
};
