class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int N = 2*limit+5;
       vector<int>m(N,0);
        int n = nums.size();
        auto func =[&](int l,int r,int x){
            if(l < r){
                m[l]+=x;
                m[r]-=x;
            }
        };
        set<int>st;
        for(int i = 0; i < n/2;++i){
            int l = nums[i];
            int r = nums[n-i-1];
            st.insert(l+r);
            int mnx = min(l,r);
            int mx = max(l,r);
            func(2,mnx+1,2);
            func(mnx+1,l+r,1);
            func(l+r+1,mx+limit+1,1);
            func(mx+limit+1,2*limit+1,2);
        }
        if(st.size()==1)
            return 0;
        int s = 0;
        int ans = n;
        for(int i = 2; i <= 2*limit; ++i){
            s += m[i];
            ans = min(ans,s);
        }
        return ans;
        
    }
};
