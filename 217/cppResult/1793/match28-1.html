class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        vector<int> vald(200005, 0);
        // int minval = 100000, maxval = 0;
        int n = nums.size();
        for(int i=0;i<n/2;i++){
            // min val with 2 rep, min val with 1 rep, max val with 1 rep, max val with 2 rep
            int mi_2 = 2, ma_2 = 2*limit, mi_1 = min(1 + nums[i], 1 + nums[n-1-i]), ma_1 = max(limit + nums[i], limit + nums[n-1-i]);

            vald[mi_2] +=2;
            vald[mi_1] -=1;
            vald[nums[i] + nums[n-1-i]]--;
            vald[nums[i] + nums[n-1-i] + 1]++;
            vald[ma_1+1]++;
            // vald[ma_2]++;
        }
        int res = nums.size();
        for(int i=2;i<vald.size();i++){
            vald[i] += vald[i-1];
            res = min(res, vald[i]);
        }
        return res;
    }
};
