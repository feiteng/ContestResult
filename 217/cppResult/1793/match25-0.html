class Solution {
public:
    int minMoves(vector<int>& a, int limit) {
      int n = a.size();
      vector<int> d(2 * limit + 5);

      for (int i = 0; i < n / 2; i++) {
        int x = a[i], y = a[n - i - 1];
        d[min(x + 1, y + 1)] += -1;
        d[max(x + limit, y + limit) + 1] += 1;
        d[x + y] += -1;
        d[x + y + 1] += 1;
      }

      int ans = 0, t = 0;
      for (int i = 0; i < 2 * limit + 5; i++) {
        t += d[i];
        ans = min(ans, t);
      }
      return ans + n;
    }
};
