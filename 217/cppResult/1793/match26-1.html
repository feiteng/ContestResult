class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n = nums.size();
        vector<int> arr(200001,0);
        for(int i = 0; i < n/2; i++){
            //cout<<i<<endl;

            arr[0]+=2;
            arr[min(nums[i],nums[n-i-1])+1]--;
            arr[nums[i]+nums[n-i-1]]--;
            arr[nums[i]+nums[n-i-1]+1]++;
            arr[max(nums[i],nums[n-i-1])+limit+1]++;
        }
        
        int ans = INT_MAX;
        int cur = 0;
        for(int i = 0; i < arr.size(); i++){
            //cout<<i<<endl;
            cur += arr[i];
            ans = min(ans,cur);
        }
        return ans;
    }
};
