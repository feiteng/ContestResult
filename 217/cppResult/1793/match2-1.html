class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        vector<int> dp(200010, 0);
        int n = nums.size();
        for (int i = 0; i < n / 2; i++) {
            int a = nums[i];
            int b = nums[n - 1 - i];

            int left = min(a, b);
            int right = max(a, b) + limit;
            //cout << left << right << endl;
            dp[2] += 2;
            dp[left+1] -= 1;
            dp[a+b] -= 1;
            dp[a+b+1] += 1;
            dp[right+1] += 1;
        }
        
        int ret = INT_MAX;
        int sum = 0;
        for (int i = 2; i <= 2*limit; i++) {
            sum += dp[i];
            ret = min(ret, sum);
        }
        return ret;
    }
};
