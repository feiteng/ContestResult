class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int res = INT_MIN;
        for (auto& account : accounts) {
            int sum = 0;
            for (auto& a : account) {
                sum += a;
            }
            res = max(res, sum);
        }
        return res;
    }
};
