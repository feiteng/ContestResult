class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int s = 0;
        for(auto& vec : accounts)
        {
            int v = 0;
            for(auto i : vec) {
                v += i;
            }
            s = max(s, v);
        }
        return s;

    }
};
