class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int ans = 0;
        for (auto &v : accounts) {
            int c = 0;
            for (auto x : v) {
                c += x;
            }
            ans = max(ans, c);
        }
        return ans;

    }
};
