class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int ans = 0;
        for(auto& i: accounts){
            int buf = 0;
            for(auto& j: i){
                buf += j;
            }
            ans = max(ans, buf);
        }
        return ans;
    }
};
