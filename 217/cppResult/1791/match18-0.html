class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int ans = INT_MIN;
        for(auto account : accounts){
            int total = 0;
            for(auto n : account){
                total += n;
            }
            ans = max(total, ans);
        }
        return ans;
    }
};
