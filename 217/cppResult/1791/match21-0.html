class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int ret = 0;
        for (auto iit: accounts) {
            int sum = 0;
            for (auto it: iit) {
                sum += it;
            }
            ret = max(ret, sum);
        }
        return ret;
    }
};
