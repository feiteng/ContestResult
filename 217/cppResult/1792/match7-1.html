class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int>res;
        int n = nums.size();
        for (int i=0; i<nums.size(); i++) {
            while (!res.empty()&&res.back()>nums[i]&&res.size()-1+n-i>=k)
                res.pop_back();
            res.push_back(nums[i]);
        }
        while(res.size()>k)
            res.pop_back();
        return res;
    }
};
