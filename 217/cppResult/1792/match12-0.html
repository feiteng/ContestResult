class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        stack<int> st;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            while (st.size() + n - i > k && !st.empty() && nums[i] < st.top())
                st.pop();
            st.push(nums[i]);
        }
        while (st.size() > k) st.pop();
        vector<int> result;
        while (!st.empty()) {
            result.push_back(st.top());
            st.pop();
        }
        reverse(result.begin(), result.end());
        return result;
    }
};
