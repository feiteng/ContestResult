class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> dq;
        for(int i=0;i<nums.size();i++){
            while(dq.size()>0&&nums[i]<dq.back()&&(dq.size()+(nums.size()-i-1))>=k){
                dq.pop_back();
            }
            if(dq.size()<k){
                dq.push_back(nums[i]);
            }
            
        }
        return dq;
    }
};
