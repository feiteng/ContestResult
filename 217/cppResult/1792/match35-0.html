class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> q;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            int v = nums[i];
            if (q.empty()) {
                q.push_back(v);
            } else if (q.size() == k && v > q.back()) {
                continue;
            } else {
                while (!q.empty() && v < q.back() && n - i + q.size() - 1 >= k) {
                    q.pop_back();
                }
                if (q.size() < k) {
                    q.push_back(v);
                }
            }
        }
        return q;
    }
};
