class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> ret;
        for(int i=0; i<nums.size(); i++)
        {
            int left = nums.size()-i;
            while(ret.size()&&nums[i]<ret.back()&&ret.size()+left>k)
                ret.pop_back();
            if(ret.size()<k)
                ret.push_back(nums[i]);
        }
        return ret;
    }
};
