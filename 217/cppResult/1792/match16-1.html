class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        deque<int> dq;
        for(int i = 0; i < nums.size(); i++) {
            while(!dq.empty() && nums[i] < dq.back() && (dq.size() + (nums.size() - i - 1)) >= k) {
                dq.pop_back();
            }
            dq.push_back(nums[i]);
        }
        while(dq.size() > k)
        {
            dq.pop_back();
        }
        return vector<int> (dq.begin(), dq.end());
    }
};
