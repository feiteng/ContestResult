class Solution {
public:
    vector<int> mostCompetitive(vector<int>& s, int k) {
        deque<int> dq;
        for(int i = 0; i < s.size(); i++) {
            while(!dq.empty() && s[i] < dq.back() && (dq.size() + (s.size() - i - 1)) >= k) {
                dq.pop_back();
            }
            dq.push_back(s[i]);
        }
        while(dq.size() > k)
        {
        dq.pop_back();
        }
        return vector<int> (dq.begin(), dq.end());
    }
};
