class Solution {
public:
    vector<int> mostCompetitive(vector<int>& vec, int k) {
        vector<int> res;
        int n = vec.size();
        for (int i = 0; i < n; i++) {
            int x = vec[i];
            while (res.size() > 0 && res.back() > x && res.size() + n - i > k) {
                res.pop_back();
            }
            res.push_back(x);
        }
        while (res.size() > k) {
            res.pop_back();
        }
        return res;
    }
};
