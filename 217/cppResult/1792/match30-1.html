class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> ret;
        stack<int> s;
        s.push(nums[0]);
        for(int i = 1;i < nums.size();i++){

            while(!s.empty() && 
                 s.size() + nums.size()-(i+1) >= k && 
                s.top() > nums[i]){
                s.pop();
            }
            if (s.size() < k){
                s.push(nums[i]);                
            }
        }
        while(!s.empty()){
            ret.emplace_back(s.top());
            s.pop();
        }
        reverse(begin(ret), end(ret));
        return ret;        
    }
};
