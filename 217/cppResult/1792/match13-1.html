class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        stack<int> s;
        int n = nums.size();
        for(int i = 0; i < n; i++){
            while(!s.empty() && n-i+s.size()>k &&nums[i]<s.top())s.pop();
            s.push(nums[i]);
        }
        while(s.size()>k)s.pop();
        vector<int> ans;
        while(!s.empty()){
            ans.push_back(s.top());
            s.pop();
        }
        reverse(begin(ans),end(ans));
        return ans;
    }
};
