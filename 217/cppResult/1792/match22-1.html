class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        int l = nums.size();
        deque<int> res;
        for(int i = 0; i < l; i++) {
            while(!res.empty() && nums[i] < res.back() && (res.size() + (l - i - 1)) >= k) {
                res.pop_back();
            }
            res.push_back(nums[i]);
        }
        return vector<int>(res.begin(), res.begin() + k);
    }
};
