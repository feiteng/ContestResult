class Solution {
public:
    vector<int> mostCompetitive(vector<int> &nums, int k) {
        vector<int> v;
        for (int i = 0; i < nums.size(); i++) {
            while (!v.empty() && v.back() > nums[i] && v.size() + nums.size() - i > k)
                v.pop_back();
            v.push_back(nums[i]);
        }
        while (v.size() > k)
            v.pop_back();
        return v;
    }
};
