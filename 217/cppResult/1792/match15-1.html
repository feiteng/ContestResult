class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> stk;
        k = nums.size() - k;
        for (auto& digit: nums) {
            while (stk.size() > 0 && stk.back() > digit && k) {
                stk.pop_back();
                k -= 1;
            }
            stk.push_back(digit);
        }
        for (; k > 0; --k) {
            stk.pop_back();
        }
        return stk;
    }
};
