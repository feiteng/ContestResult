class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        stack<int> stk;
        vector<int> res;
        for (int i = 0; i < nums.size(); i++) {
            int n = nums[i];
            while (stk.size() && stk.top() > n && nums.size() - i + stk.size() - 1 >= k) {
                stk.pop();
            }
            if (stk.size() < k) {
                stk.push(n);
            }
        }
        
        while (stk.size()) {
            res.push_back(stk.top());
            stk.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
