class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> s;
        for(int i = 0; i < nums.size(); ++i) {
            while(s.size() && nums[i] < s.back() && (nums.size() - i + s.size() > k)) {
                s.pop_back();
            }
            s.push_back(nums[i]);
        }
        s.resize(k);
        return s;
    }
};
