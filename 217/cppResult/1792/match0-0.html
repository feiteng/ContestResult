class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> res;
        res.push_back(nums[0]);
        int n=nums.size();
        for(int i=1;i<n;++i){
            while(!res.empty()&&nums[i]<res.back()&&n-i+res.size()>k){
                res.pop_back();
            }
            res.push_back(nums[i]);
        }
        while(res.size()!=k) res.pop_back();
        return res;
    }
};
