class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        stack<int> stack1;
        int n = nums.size();
        for(int i = 0; i < n; i++)
        {
            //i+1~n-1
            while(!stack1.empty() && stack1.top() > nums[i] && stack1.size()+ (n-1-i) >= k) stack1.pop();
            stack1.push(nums[i]);
        }
        while(stack1.size() > k) stack1.pop();
        vector<int> res;
        while(!stack1.empty())
        {
            res.push_back(stack1.top());
            stack1.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
