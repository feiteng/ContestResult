class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector <int> ans;
        for(int j = 0;j < nums.size();j++){
            int i = nums[j];
            while(ans.size() > 0&&ans[ans.size()-1] > i&& ans.size()+(nums.size() - j) > k) ans.pop_back(); 
            if(ans.size() < k)
            ans.push_back(i);
        }
        return ans;
    }
};
