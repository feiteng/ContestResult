class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> subs;
        for (int i=0; i<nums.size(); i++) {
            while(subs.size() > 0 && (nums[i] < subs.back()) && ((nums.size() - i) + (subs.size() - 1) >= k)) {
                subs.pop_back();
            }
            if (subs.size() < k) {
                subs.push_back(nums[i]);
            }      
        }
        return subs;
    }
};
