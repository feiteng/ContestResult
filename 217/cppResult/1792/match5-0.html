class Solution {
public:
    vector<int> mostCompetitive(vector<int>& a, int k) {
        int n = a.size();
        multiset <pair <int, int> > s;
        for (int i = 0; i < n - k; i++) {
            s.insert(make_pair(a[i], i));
        }
        vector <int> ret;
        int curPos = 0;
        for (int i = n - k; i < n; i++) {
            s.insert(make_pair(a[i], i));
            while (s.begin()->second < curPos) {
                s.erase(s.begin());
            }
            ret.push_back(s.begin()->first);
            curPos = s.begin()->second;
            s.erase(s.begin());
        }
        return ret;
    }
};
