class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> res;
        for(int i=0;i<nums.size();++i){
            while(!res.empty()&&res.back()>nums[i]&&i+k-res.size()<nums.size()){
                res.pop_back();
            }
            res.push_back(nums[i]);
        }
        return vector<int>(res.begin(),res.begin()+k);
    }
};
