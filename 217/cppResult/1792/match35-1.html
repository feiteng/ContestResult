class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> res;
        int n = nums.size();
        for (int i = 0; i < n; i++)
        {
            int v = nums[i];
            if (res.empty())
            {
                res.push_back(v);
            }
            else

            {
                while (!res.empty() && res.back() > v && (k - res.size()) <= (n - i - 1))
                {
                    res.pop_back();
                }
                if (res.size() < k)
                {
                    res.push_back(v);
                }
            }
        }
        return res;
    }
};
