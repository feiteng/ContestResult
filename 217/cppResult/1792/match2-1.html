class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        vector<int> dq;
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            while (dq.size() && nums[i] < dq.back() && (n - 1 - i + dq.size() >= k)) {
                dq.pop_back();
            }
            if (dq.size() < k) dq.push_back(nums[i]);
        }
        return dq;
        
    }
};
