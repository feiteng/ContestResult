class Solution {
public:
    
    
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] % 2 == 1) {
                nums[i] *= 2;
            }
        }
        set<int> s(nums.begin(), nums.end());
        int res = 0x3f3f3f3f;
        
        while (*s.rbegin() % 2 == 0) {
            res = min(res, *s.rbegin() - *s.begin());
            int e = *s.rbegin();
            s.erase(e);
            s.insert(e / 2);
        }
        res = min(res, *s.rbegin() - *s.begin());
            return res;
    }
};
