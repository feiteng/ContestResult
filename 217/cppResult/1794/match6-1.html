class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        priority_queue<int> q;
        int m = 1e9, r = 1e9, t;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] & 1) {
                q.push(2 * nums[i]);
                m = min(m, 2 * nums[i]);
            } else {
                q.push(nums[i]);
                m = min(m, nums[i]);
            }
        }
        while ((t = q.top()) % 2 == 0) {
            t /= 2;
            m = min(m, t);
            q.pop();
            q.push(t);
            r = min(r, q.top() - m);
        }
        return r;
    }
};
