using ll = long long;
class Solution {
public:
    // int n;
    // vector<vector<pair<int, int>>> dp;
    int minimumDeviation(vector<int>& nums) {
        int ans = 2147483647;
        set<int> se;
        for(auto e:nums){
           if(e&1) se.insert(e*2);
           else se.insert(e);
        }
        auto p = se.end();
        auto e = *--p;
        se.erase(p);
        while(true){
            p = se.end();
//              while(e%2==0&&e>*p&&e>*se.begin()){
//                 e /= 2;
//                 // ans = min(ans, )
//             }
            if(e%2==0&&e>=*p)  e/=2;
            se.insert(e);
            p = se.end();
            ans = min(ans, (*--p) - (*se.begin()));
            e = *p;
            se.erase(p);
            if(e&1) break;
        }
     
        return ans;
    }
};
