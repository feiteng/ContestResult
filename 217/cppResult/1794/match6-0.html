class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        priority_queue<int> heap;
        int min_value = INT_MAX, result = INT_MAX;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] % 2 == 1) {
                nums[i] *= 2;
            }
            min_value = min(min_value, nums[i]);
            heap.push(nums[i]);
        }
        while (true) {
            int t = heap.top();
            result = min(result, t - min_value);
            if (t & 1 == 1) {
                break;
            }
            heap.pop();
            do {
                t >>= 1;
            } while (t & 1 == 0 && t > min_value);
            min_value = min(min_value, t);
            heap.push(t);
        }
        return result;
    }
};
