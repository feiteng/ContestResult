class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        vector<int> a = nums;
        int n = a.size();
        for (int i = 0; i < n; ++i) {
            if (a[i] % 2) {
                a[i] *= 2;
            }
        }
        priority_queue<int> pq;
        int res = 1000000001;
        int mv = 1000000001;
        for (int i = 0; i < n; ++i) {
            pq.push(a[i]);
            mv = min(mv, a[i]);
        }
        while (!pq.empty()) {
            int b = pq.top();
            if (b % 2) {
                res = min(res, b - mv);
                return res;
            }
            pq.pop();
            res = min(res, b - mv);
            b = b / 2;
            mv = min(mv, b);
            pq.push(b);
        }
        return res;
    }
};
