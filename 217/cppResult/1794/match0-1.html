class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        multiset<int> Q;
        for (int x : nums) {
            if (x&1) {
                Q.insert(x*2);
            } else {
                Q.insert(x);
            }
        }
        int best = *(--Q.end()) - *Q.begin();
        while(!((*(--Q.end()))&1)) {
            int x = *(--Q.end());
            Q.erase(--Q.end());
            Q.insert(x>>1);
            best = min(best, *(--Q.end()) - *Q.begin());
        }
        return best;
    }
};
