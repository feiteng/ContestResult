class Solution {
public:
    int minimumDeviation(vector<int>& a) {
        int n = a.size();
        vector<int> v[n];
        for (int i = 0; i < n; i++) {
            int cur = a[i];
            if (cur % 2) {
                v[i].push_back(cur);
                v[i].push_back(cur << 1);
            } else {
                while (cur % 2 == 0) {
                    v[i].push_back(cur);
                    cur >>= 1;
                }
                v[i].push_back(cur);
            }
        }
        for (int i = 0; i < n; i++) {
            sort(v[i].begin(), v[i].end());
        }
        // for (int i = 0; i < n; i++) {
        //     cout << a[i] << " -> ";
        //     for (auto x : v[i])
        //         cout << x << " ";
        //     cout << endl;
        // }
        int pos[n];
        for (int i = 0; i < n; i++)
            pos[i] = 0;
        multiset<pair<int,int>> s;
        for (int i = 0; i < n; i++) {
            s.insert({v[i][0],i});
        }
        int ans = INT_MAX;
        while(1) {
            auto start = *s.begin();
            auto last = *s.rbegin();
            ans = min(ans, last.first - start.first);
            int idx = start.second;
            s.erase(s.begin());
            if (pos[idx] == (int)v[idx].size() - 1) {
                break;
            }
            pos[idx]++;
            s.insert({v[idx][pos[idx]], idx});
        }
        return ans;
    }
};
