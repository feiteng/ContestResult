class Solution {
public:
    int minimumDeviation(vector<int>& a) {
        multiset<int> nums;
        for (int x : a) {
            nums.insert(x);
        }
        
        int small = *nums.begin(), big = *nums.rbegin();
        int ans = big - small;
        while (1) {
            if (small % 2 == 0) {
                break;
            }
            auto it = nums.find(small);
            nums.erase(it);
            nums.insert(small * 2);
            small = *nums.begin();
            big = *nums.rbegin();
            ans = min(ans, big - small);
        }
        while (1) {
            if (big % 2 != 0) {
                break;
            }
            auto it = nums.find(big);
            nums.erase(it);
            nums.insert(big / 2);
            small = *nums.begin();
            big = *nums.rbegin();
            ans = min(ans, big - small);
        }
        return ans;
    }
};
