class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        vector<vector<int>> ls(n);
        
        for (int i = 0; i < n; ++i) {
            int cur = nums[i];
            while (cur % 2 == 0) cur /= 2;
            while (cur <= nums[i]) {
                ls[i].push_back(cur);
                cur *= 2;
            }
            if (nums[i] % 2 == 1) ls[i].push_back(nums[i] * 2);
        }
        
        vector<int> pos(n, 0);
        
        set<pair<int, int>> s;
        for (int i = 0; i < n; ++i) {
            s.emplace(ls[i][0], i);
        }
        
        int res = 1000000007;
        
        while (s.size()) {
            res = min(res, s.rbegin()->first - s.begin()->first);
            auto p = *s.begin();
            s.erase(s.begin());
            int idx = p.second;
            if (pos[idx] == ls[idx].size() - 1) break;
            s.emplace(ls[idx][pos[idx] + 1], idx);
            pos[idx] += 1;
        }
        
        return res;

    }
};
