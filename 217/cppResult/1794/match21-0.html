class Solution {
public:
    
    int c1(vector<int>& nums)
    {
        priority_queue<int> q1; //大根堆
        priority_queue<int, vector<int>, greater<int>> q2; //小根堆
        for(int &x : nums){
            q1.push(x);
            q2.push(x);
        }
        int res = q1.top() - q2.top();
        
        while(q1.top() % 2 == 0)
        {
            int t = q1.top();
            q1.pop();
            q1.push(t / 2);
            q2.push(t / 2);
            res = min(res, q1.top() - q2.top());
        }
        
        while(q2.top() % 2 == 1)
        {
            int t = q2.top();
            q2.pop();
            q1.push(t * 2);
            q2.push(t * 2);
           // cout << res << ' ' << q1.top() - q2.top() << endl;
            if(q1.top() - q2.top() > res) break;
            res = q1.top() - q2.top();
        }
        return res;
    }

    int c2(vector<int>& nums)
    {
        priority_queue<int> q1; //大根堆
        priority_queue<int, vector<int>, greater<int>> q2; //小根堆
        for(int &x : nums){
            q1.push(x);
            q2.push(x);
        }
        int res = q1.top() - q2.top();
        
        while(q2.top() % 2 == 1)
        {
            int t = q2.top();
            q2.pop();
            q1.push(t * 2);
            q2.push(t * 2);
           // cout << res << ' ' << q1.top() - q2.top() << endl;
            //if(q1.top() - q2.top() > res) break;
            res = q1.top() - q2.top();
        }
        
        while(q1.top() % 2 == 0)
        {
            int t = q1.top();
            q1.pop();
            q1.push(t / 2);
            q2.push(t / 2);
            res = min(res, q1.top() - q2.top());
            if(q1.top() - q2.top()) break;
            res = q1.top() - q2.top();
        }
        
        
        return res;
    }
    
    int minimumDeviation(vector<int>& nums) {
        return min(c1(nums), c2(nums));
       
        //return res;
    }
};
