class Solution {
public:
    int f(set<int> &s) {
        int mn = *s.begin(), mx = *s.rbegin();
        return mx - mn;
    }
    int minimumDeviation(vector<int>& nums) {
        set<int> s; 
        for(int num: nums) s.insert(num);
        int ans = f(s);
        // cout << *s.begin() << " " << *s.rbegin() << endl;
        while((*s.begin()) % 2 == 1) {
            int val = (*s.begin());
            s.erase(val);
            s.insert(val << 1);
            ans = min(ans, f(s));
            // cout << *s.begin() << " " << *s.rbegin() << endl;
        }
        // cout << endl;
        while((*s.rbegin()) % 2 == 0) {
            int val = (*s.rbegin());
            s.erase(val);
            s.insert(val >> 1);
            ans = min(ans, f(s));
            // cout << *s.begin() << " " << *s.rbegin() << endl;
        }
        // cout << endl;
        int mx = (*s.rbegin());
        while(((*s.begin()) % 2 == 1) && (*s.begin()) <= mx) {
            int val = (*s.begin());
            s.erase(val);
            s.insert(val << 1);
            ans = min(ans, f(s));
            // cout << *s.begin() << " " << *s.rbegin() << endl;
        }
        // cout << "ans:" << ans << endl;
        return ans;
    }
};
