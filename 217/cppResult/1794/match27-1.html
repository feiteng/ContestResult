class Solution {
public:
    int minimumDeviation(vector<int>& A) {
        set<pair<int,int>> S;
        int n = A.size();
        for (int i = 0; i < n; i++) {
            int k = A[i];
            while (A[i] % 2 == 0) A[i] /= 2;
            S.insert(make_pair(A[i], k));
        }
        
        int ans = 1e9+5;
        
        while(1) {
            auto l = *(S.begin());
            auto t = S.end();
            t--;
            auto r = *t;
            
            ans = min(ans, r.first - l.first);
            
            int x = l.first, y = l.second;
            if (x % 2 == 1 || x < y) {
                S.erase(S.begin());
                S.insert(make_pair(x*2,y));
            } else
                break;
        }
        
        return ans;

    }
};
