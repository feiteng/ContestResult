using pii = pair<int, int>;

class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        vector<int> limit(n);
        int maxv = 0;
        priority_queue<pii, vector<pii>, greater<pii>> pq;
        for (int i = 0; i < n; ++i) {
            if (nums[i] % 2 == 0) {
                limit[i] = nums[i];
                while (nums[i] % 2 == 0) nums[i] >>= 1;
            }
            else limit[i] = nums[i] << 1;
            maxv = max(maxv, nums[i]);
            pq.push(make_pair(nums[i], limit[i]));
        }
        int anchor = maxv;
        int ans = maxv - pq.top().first;
        while (pq.top().first != anchor) {
            pii v = pq.top();
            v.first <<= 1;
            if (v.first > v.second) break;
            pq.pop();
            pq.push(v);
            maxv = max(maxv, v.first);
            ans = min(ans, maxv - pq.top().first);
        }
        return ans;
    }
};
