class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        multiset<int, greater<>> s;
        for (int a: nums) {
            if (a % 2 == 0)
                s.emplace(a);
            else
                s.emplace(a*2);
        }
        int ans = *max_element(nums.begin(), nums.end()) - *min_element(nums.begin(), nums.end());
        while (*s.rbegin() <= 2e9) {
            int maxi = *s.begin();
            int mini = *s.rbegin();
            ans = min(ans, maxi-mini);
            if (maxi % 2 == 1) break;
            s.erase(s.begin());
            s.emplace(maxi/2);
        }
        return ans;
    }
};
