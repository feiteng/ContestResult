class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        set<int> s;
        for (int x : nums) {
            if (x & 1) {
                s.insert(x*2);
            } else {
                s.insert(x);
            }
        }
        
        int ans = 1<<30;
        while (true) {
            auto it = s.rbegin();
            int v = *it;
            // cout << v << ' ';
            ans = min(ans, v - *s.begin());
            if (v & 1) {
                // int mx = *it, mn = *s.begin();
                // cout << mx << ' ' << mn << endl;
                return ans;
            }
            s.erase(v);
            s.insert(v/2);
        }
        return -1;
    }
};
