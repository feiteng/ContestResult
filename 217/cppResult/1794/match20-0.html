using pii = pair<int, int>;
class Solution {
public:
    int minimumDeviation(vector<int>& a) {
        int n = a.size();
        vector<pii> b;
        for (int i = 0; i < n; ++i){
                b.emplace_back(a[i], i);
            if (a[i] & 1){
                b.emplace_back(a[i] * 2, i);
            } else {
                while (a[i] % 2 == 0)
                    b.emplace_back(a[i] /= 2, i);
            }
        }
        sort(b.begin(), b.end());
        vector<int> cnt(n, 0);
        int pos = 0;
        int r = 0;
        int ans = b.rbegin()->first - b.begin()->first;
        for (int l = 0; l < b.size(); ++l){
            for (; pos < n && r < b.size(); ++r){
                pos += (cnt[b[r].second]++) == 0;
            }
            if (pos == n)
                ans = min(ans, b[r - 1].first - b[l].first);
            pos -= (--cnt[b[l].second]) == 0;
            
        }
        return ans;
    }
};
