class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        priority_queue<int > pq;
        int MIN = INT_MAX;
        for(int n : nums) {n = (n % 2 == 0? n : n * 2); MIN = min(MIN, n); pq.push(n);}
        int ans = INT_MAX;
        while(true){
            int tmp = pq.top();
            pq.pop();
            ans = min(ans, abs(tmp - MIN));
            MIN = min(MIN, tmp/2);
            pq.push(tmp/2);
            if(pq.top() % 2 == 1) break;
        }
        ans = min(ans, pq.top() - MIN);
        return ans;
    }
};
