class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        vector<int> v[n];
        set<pair<int, int> > s;
        

        for (int i = 0; i < nums.size(); i++) {
            int x = nums[i];
            if (x & 1) {
                v[i].push_back(2 * x);
                v[i].push_back(x);
            } else {
                while (x % 2 == 0) {
                    v[i].push_back(x);
                    x /= 2;
                }
                v[i].push_back(x);
            }
            // cout << i << ": ";
            // for (auto x : v[i]) cout << x << " "; cout << endl;
            s.insert({v[i].back(), i});
            v[i].pop_back();
        }
        
        int ans = INT_MAX;
        
        while (1) {
            auto f = *s.begin();
            int p = f.first;
            int l = (*s.rbegin()).first;
            ans = min(ans, l - p);
            s.erase(s.begin());
            
            int id = f.second;
            if (!v[id].empty()) {
                s.insert({v[id].back(), id});
                v[id].pop_back();
            } else {
                break;
            }
        }
        return ans;
    }
};
