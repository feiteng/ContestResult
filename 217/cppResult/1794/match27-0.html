class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        priority_queue<int> pq;
        vector<int> v;
        int largestodd = 0;
        for(int i = 0; i<nums.size(); i++){
            int tmp = nums[i];
            while(tmp%2 == 0){
                tmp /= 2;
            }
            largestodd = max(largestodd, tmp);
            if(nums[i]%2 == 1){
                nums[i] *= 2;
            }
        }
        for(int i = 0; i<nums.size(); i++){
            while(nums[i] >= 2*largestodd){
                nums[i] /= 2;
            }
            if(nums[i]%2 == 0 && nums[i] > 0){
                pq.push(nums[i]);
            }
        }
        int ans = 0;
        pq.push(0);
        if(pq.size()){
            ans = max(pq.top(), largestodd) - *min_element(nums.begin(), nums.end());
        }
        int mn = *min_element(nums.begin(), nums.end());
        while(pq.size() > 1){
            auto n = pq.top();
            pq.pop();
            mn = min(mn, n/2);
            ans = min(ans, max(largestodd, pq.top()) - mn);
        }
        return ans;
    }
};
