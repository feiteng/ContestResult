class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n=nums.size();
        set<pair<int,int>> s;
        for(int x:nums){
            int lim=x,init=x;
            if(lim%2==1)lim*=2;
            while(init%2==0)init/=2;
            s.insert({init,lim});
        }
        int ans=INT_MAX;
        while(true){
            int x,lim;
            tie(x,lim)=*s.begin();
            int diff=s.rbegin()->first-x;
            ans=min(ans,diff);
            if(x==lim)return ans;
            s.erase(s.begin());
            s.insert({x*2,lim});
        }
        return -1;
    }
};
