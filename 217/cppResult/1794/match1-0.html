class Solution {
public:
    int minimumDeviation(vector<int>& a) {
      set<int> st;
      int n = a.size();
      for (int i = 0; i < n; i++) {
        if (a[i] % 2 == 0) {
          st.insert(a[i]);
        } else {
          st.insert(2 * a[i]);
        }
      }
      int ans = 2e9;
      while (true) {
        auto t = st.end();
        t--;
        auto s = st.begin();
        ans = min(ans, *t - *s);
        if (*t % 2 == 0) {
          int x = *t;
          st.erase(x);
          st.insert(x / 2);
        } else {
          break;
        }
      }
      return ans;
    }
};
