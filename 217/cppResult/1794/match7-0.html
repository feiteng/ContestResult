class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int res=1000000005;
        priority_queue<int> pq;
        int mi=1000000005;
        for (int i=0; i<nums.size(); ++i) {
            if (nums[i]%2==1) {
                mi=min(mi, nums[i]*2);
                pq.push(nums[i]*2);
            } else {
                mi=min(mi, nums[i]);
                pq.push(nums[i]);
            }
        }
        while (true) {
            int x=pq.top(); pq.pop();
            res=min(res, x-mi);
            if (x%2==1) {
                break;
            }
            pq.push(x/2);
            mi=min(mi, x/2);
        }
        return res;
        
    }
};
