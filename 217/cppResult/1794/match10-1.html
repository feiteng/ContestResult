class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        set<int> ms;
        for(int i = 0; i < n; i++){
            if(nums[i]%2 == 1){
                nums[i] *= 2;
            }
            ms.insert(nums[i]);
        }
        int ans = 1e9;
        while(1){
            
            if(ms.size() == 1){
                return 0;
            }
            
            auto it = ms.rbegin();
            int x = *it;
            ms.erase(x);
            
            
            auto pt = ms.begin();
            
            int low = *pt;
            
            ans = min(ans, x - low);
            
            if(x%2){
                break;
            }
            else {
                x/=2;
                ms.insert(x);
            }
        }
        
        return ans;
    }
};
