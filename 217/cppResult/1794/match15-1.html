class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        
        vector<int> a[n+1];
        set<pair<int, pair<int,int> > > s; // number, ind = [0,n), cur = [0, 32]

        for(int i=0;i<n;i++){
            int x = nums[i];
            
            if(x & 1){
                a[i].push_back(x);
                a[i].push_back(2*x);
            } else {
                while(true){
                    a[i].push_back(x);
                    if(x&1) break;
                    x/=2;
                }
            }
            
            sort(a[i].begin(), a[i].end());
            
            s.insert({a[i][0], {i, 0}});
        }
        
        int ans = INT_MAX;
        
        while(true){
            
            pair<int, pair<int,int> > front = *s.begin();
            pair<int, pair<int,int> > back = *s.rbegin();
            
            ans = min(ans, back.first - front.first);
            
            s.erase(front);
            
            int ind = front.second.first;
            int cur = front.second.second;
            
            if(cur == a[ind].size() - 1){
                break;
            } else {
                s.insert({a[ind][cur + 1], {ind, cur+1}});
            }
        }
        
        return ans;
        
    }
};
