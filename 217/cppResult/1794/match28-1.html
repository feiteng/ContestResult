class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        multiset<int>st;
        int n= nums.size();
        for(int i=0;i<n;i++){
            st.insert(nums[i]);
        }
        int ans=1e9;
        while (true){
            int num=*st.begin();
            ans=min(ans,*prev(st.end())-num);
            if (num % 2==0) break;
            st.erase(st.begin());
            st.insert(num << 1);
        }
        while (true){
            int num=*prev(st.end());
            ans=min(ans, num-*st.begin());
            if (num % 2==1) break;
            st.erase(prev(st.end()));
            st.insert(num >>1);
        }
        return min(ans, *prev(st.end())-*st.begin());
    }
};
