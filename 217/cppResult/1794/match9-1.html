class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        priority_queue<pair<int,int>> max_pq;

        priority_queue<pair<int,int>> min_pq;
        for(int i = 0;i < nums.size();i++){
            if (nums[i]%2 == 1)nums[i] *= 2;
            max_pq.push(make_pair(nums[i],i));
            min_pq.push(make_pair(-nums[i], -i));
        }
        int ret = 2 * 1e9+2;
        while(!max_pq.empty()){
            pair<int,int> maxi = max_pq.top();max_pq.pop();
            pair<int,int> mini = min_pq.top();
            while(nums[-mini.second] != -mini.first || 
                 -mini.second == maxi.second){
                  //mini.first == maxi.first){
            //cout << maxi.first <<" " << -mini.first << " " <<maxi.second <<" " << -mini.second << " " << nums[-mini.second] << endl;

                min_pq.pop();

                min_pq.push(make_pair(mini.first/2, mini.second));
                mini = min_pq.top();
            }

            ret = min(ret, maxi.first+mini.first);

            if (maxi.first % 2 != 0)break;
            maxi.first /= 2;
            nums[maxi.second] = maxi.first;
            max_pq.push(maxi);
            
            min_pq.push(make_pair(-maxi.first, -maxi.second));
            
        }
        //cout << endl;
        
        
        return ret;
    }
};
