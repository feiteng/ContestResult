typedef long long lint;
typedef pair<lint,int>pli;
class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n=nums.size();
        vector<pli>a;
        for(int i=0;i<n;i++){
            a.push_back(pli(nums[i],i));
            if(nums[i]%2){
                a.push_back(pli(2*nums[i],i));
            }else{
                lint b=nums[i];
                while(b%2==0){
                    b/=2;
                    a.push_back(pli(b,i));
                }
            }
        }
        sort(a.rbegin(),a.rend());
        int po=0;
        vector<int>u(n,-1);
        int m=a.size();
        vector<int> occ(m);
        lint inf=1e10;
        vector<lint>cur(n,inf);
        int up=0;
        lint mi=1e10;
        for(int i=0;i<m;i++){
            pli b=a[i];
            lint x=b.first,ind=b.second;
            if(u[ind]==-1)up++;
            int olduind=u[ind];
            u[ind]=i;
            lint old=cur[ind];
            cur[ind]=x;
            occ[i]++;
            if(old<inf){
                occ[olduind]--;
                while(po<m&&occ[po]==0)po++;
            }
            if(up==n){
                lint ma=a[po].first;
                mi=min(mi,ma-x);
            }
        }
        return mi;
    }
};
