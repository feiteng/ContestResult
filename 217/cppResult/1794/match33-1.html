class Solution {
public:
	int minimumDeviation(vector<int>& nums) {
		int n = nums.size();
		set<int>all;
		for (int i = 0; i < n; i++) {
			if (nums[i] % 2 == 1)nums[i] *= 2;
			all.insert(nums[i]);
		}
		int minn = 1e9;
		while (*all.rbegin() % 2 == 0) {
			int x = *all.rbegin();
			all.erase(x);
			all.insert(x / 2);
			minn = min(minn, *all.rbegin() - *all.begin());
		}
		return minn;
	}
};
