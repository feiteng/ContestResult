class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int n = nums.size();
        
        multiset<int> M;

        for (auto x : nums) {
            if (x & 1) {
                M.insert(x << 1);
            } else {
                M.insert(x);
            }
        }
        
        
        int res = 2e9;
        
        while (M.size() == n) {
            int mn = *M.begin();
            int mx = *M.rbegin();
            // cout << mn << " " << mx << endl;
            res = min(res, mx - mn);
            
            auto it = M.end();
            --it;
            M.erase(it);
            if (mx & 1) {
                continue;
            }
            mx >>= 1;
            M.insert(mx);
        }
        
        return res;
        
    }
    

};
