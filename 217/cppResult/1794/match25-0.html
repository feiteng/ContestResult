class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
        int mn = 0x7fffffff;
        priority_queue<int>q;
        for(int &x:nums){
            if(x & 1) x <<= 1;
            q.push(x);
            mn = min(mn,x);
        }
        int gdx = -1,ans = q.top() - mn;
        while(!q.empty()){
            int w = q.top();q.pop();
            if(w & 1){
                int v = 0x7fffffff;
                while(!q.empty()){
                    v = min(v,q.top());q.pop();
                }
                return min(ans,w - v);
            }
            int tp = (w>>1);
            mn = min(tp,mn);
            q.push(tp);
            ans = min(ans,q.top() - mn);
        }
        return -1;
    }
};
