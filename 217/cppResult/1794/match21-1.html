class Solution {
public:
    int minimumDeviation(vector<int>& nums) {

        multiset<int> data;
        for (int i : nums) {
            data.insert(i);
        }
        int ans = *data.rbegin() - *data.begin();
        while (*data.begin() % 2 == 1) {
            int cur = *data.begin();
            data.erase(data.begin());
            data.insert(2 * cur);
            ans = min(ans, *data.rbegin() - *data.begin());
        }
        while (*data.rbegin() % 2 == 0) {
            int cur = *data.rbegin();
            data.erase(--data.end());
            data.insert(cur / 2);
            ans = min(ans, *data.rbegin() - *data.begin());
        }
        return ans;
    }
};
