class Solution {
public:
    int minimumDeviation(vector<int>& nums) {
      int n = nums.size();
      for (int i = 0; i < n; ++i) {
        if (nums[i]%2==1)nums[i]*=2;
      }
      vector<pair<int,int>> pairs;
      for (int i = 0; i < n; ++i) {
        while (nums[i] % 2 == 0) {
          pairs.push_back({nums[i], i});
          nums[i]/=2;
        }
        pairs.push_back({nums[i],i});
      }
      sort(pairs.begin(),pairs.end());
      //print(pairs);
      int low = 0;
      int high = -1;
      vector<int> colors(n,0);
      int cs = 0;
      while (cs < n) {
        ++high;
        colors[pairs[high].second]++;
        if (colors[pairs[high].second]==1) ++cs;
      }
      int ans = pairs[high].first-pairs[low].first;
        colors[pairs[low].second]--;
        if (colors[pairs[low].second] == 0) {
          --cs;
        }
        ++low;
      while (low < pairs.size()) {
        // Updates high.
        while (high < pairs.size() - 1 && cs < n) {
          ++high;
          colors[pairs[high].second]++;
          if (colors[pairs[high].second]==1) ++cs;
        }
        //debug()<<imie(low)<<imie(high);
        if (cs == n) {
          ans = min(ans, pairs[high].first - pairs[low].first);
        }
        // Removes low.
        colors[pairs[low].second]--;
        if (colors[pairs[low].second] == 0) {
          --cs;
        }
        ++low;
      }
      return ans;
    }
};
