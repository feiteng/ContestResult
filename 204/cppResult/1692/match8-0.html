class Solution {
public:
    long long cmn[1001][1001];
    long long p = 1e9 + 7;
    long long cal(vector<int>& a){
        if (a.size() <= 1){ return 1;}
        vector<int> b, c;
        int temp = a[0];
        for (int i = 1; i < a.size(); ++i){
            if (a[i] > temp){
                b.push_back(a[i]);
            }else{
                c.push_back(a[i]);
            }
        }
        return ((cmn[a.size() - 1][b.size()] * cal(b))%p * cal(c)) % p; 
    }
    int numOfWays(vector<int>& nums) {
        int n = nums.size(); 
        for (int i = 1; i <= n; ++i){
            cmn[i][0] = 1;
            cmn[i][i] = 1;
            for (int j = 1; j < i; ++j){
                cmn[i][j] = (cmn[i-1][j] + cmn[i-1][j-1]) % p;
            }
        }
        return (cal(nums) - 1) % p;
    }
};
