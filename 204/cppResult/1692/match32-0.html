using LL = long long;
const int MOD = 1000000007;

LL mypow(LL x, LL p) {
    LL ret = 1;
    while(p > 0) {
        if(p & 1) {
            ret = (ret * x) % MOD;
        }
        x = (x * x) % MOD;
        p >>= 1;
    }
    return ret;
}

LL inv(LL x) {
    assert(x != 0);
    return mypow(x, MOD - 2);
}

class Solution {
private:
    vector<LL> f;
public:
    int solve(vector<int>& nums) {
        int n = nums.size();
        if(n <= 1) {
            return 1;
        }
        vector<int> l, r;
        for(int i = 1; i < n; ++i) {
            if(nums[i] < nums[0]) {
                l.push_back(nums[i]);
            } else {
                r.push_back(nums[i]);
            }
        }
        return (f[n - 1] * solve(l) % MOD * inv(f[l.size()]) % MOD *
                solve(r) % MOD * inv(f[r.size()]) % MOD);
    }
    int numOfWays(vector<int>& nums) {
        int n = nums.size();
        f.resize(n + 1);
        f[0] = 1;
        for(int i = 1; i <= n; ++i) {
            f[i] = (f[i - 1] * i) % MOD;
        }
        return (solve(nums) + MOD - 1) % MOD;
    }
};
