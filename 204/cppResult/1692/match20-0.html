    long long comb[1010][1010]={};
    long long mod=1e9+7;
class Solution {
//     long long cal(vector<int>& a, int i, int j){
        
//     }
public:
    long long _numOfWays(vector<int>& a) {
        if(comb[0][0]==0) {
            for(int i=0;i<1001;i++) {
                for(int j=0;j<1001;j++) {
                    comb[i][j]=1;
                }
            }
            for(int i=2;i<1001;i++) {
                for(int j=1;j<i;j++) {
                    comb[i][j]=(comb[i-1][j-1]+comb[i-1][j])%mod;
                }
            }
        }
        // cout<<123<<endl;
        int n=a.size();
        if(n<=1) return 1;
        vector<int>v1,v2;
        for(int i=1;i<n;i++) {
            if(a[i]<a[0]) v1.push_back(a[i]);
            else v2.push_back(a[i]);
        }
        long long res=1;
        res=res*_numOfWays(v1)%mod;
        res=res*_numOfWays(v2)%mod;
        res=res*comb[v1.size()+v2.size()][v1.size()]%mod;
        return res;
    }
    long long numOfWays(vector<int>& a) {
        return (_numOfWays(a)-1+mod)%mod;
    }
};
