class Solution {
public:
    using LL = long long;
    const int MOD = 1e9 + 7, MAXN = 1001;
    LL combs[1001][1001];
    
    int numOfWays(vector<int>& nums) {
        memset(combs, 0, sizeof combs);
        combs[0][0] = 1;

        for(int i = 1; i < MAXN; i ++) combs[i][1] = i;
        for(int i = 1; i < MAXN; i ++){
            combs[i][0] = 1;
            for(int j = 1; j <= i; j ++){
                combs[i][j] = (combs[i - 1][j] + combs[i - 1][j - 1]) % MOD;
            }
        }
        return (f(nums) - 1) % MOD;
    }
    
    
    LL f(vector<int>& nums){

        if(nums.size() <= 1) return 1;
        vector<int> large, small;
        for(int i = 1; i < nums.size(); i ++){
            if(nums[i] > nums[0]) large.push_back(nums[i]);
            else small.push_back(nums[i]);
        }
        
        LL rnt = combs[nums.size() - 1][large.size()];
        rnt = (rnt * f(large)) % MOD;
        rnt = (rnt * f(small)) % MOD;
        
        return rnt;
    }
};
