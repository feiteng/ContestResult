class Solution {
public:
    int C[1001][1001]; 
    int mod = 1000000007;
    void solve(){
        for (int i = 0; i <= 1000; i++) 
        { 
            for (int j = 0; j <= min(i, 1000); j++) 
            { 
                if (j == 0 || j == i) 
                    C[i][j] = 1; 
                else
                    C[i][j] = (C[i - 1][j - 1]%mod + 
                              C[i - 1][j]%mod)%mod; 
            } 
        } 
        return;
    }
    long long val(vector<int>& nums)
    {
        int n=nums.size();
        if(n<=1)
            return 1;
        vector<int> left,right;
        int r=nums[0];
        for(int i=1;i<n;i++)
        {
            if(nums[i]<r)
                left.push_back(nums[i]);
            else right.push_back(nums[i]);
        }
        long long ans = (val(left)*val(right));
        ans = ans%mod;
        ans = (ans*C[n-1][left.size()])%mod;
        return ans%mod;
    }
    int numOfWays(vector<int>& nums) {
        solve();
        return (val(nums) - 1)%mod;
    }
};
