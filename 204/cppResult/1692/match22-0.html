class Solution {
public:
    const int mod = 1e9 + 7;
    int mul(int x, int y) {
        return x * 1ll * y % mod;
    }
    int add(int x, int y) {
        return (x + y) % mod;
    }
    
    int binpow(int x, int p) {
        int z = 1;
        while (p > 0) {
            if (p & 1) z = mul(z, x);
            x = mul(x, x);
            p >>= 1;
        }
        return z;
    }
    int inv(int x) {
        return binpow(x, mod - 2);
    }
    
   
    int fact[1001];
    int rfact[1001];
    void prepare() {
        fact[0] = 1;
        for (int i = 1; i <= 1000; ++i) {
            fact[i] = mul(fact[i-1], i);
        }
        for (int i = 0; i <= 1000; ++i) {
            rfact[i] = inv(fact[i]);
        }
    }
    
    int C(int n, int k) {
        if (n < 0 || n < k || k < 0) return 0;
        return mul(fact[n], mul(rfact[n-k], rfact[k]));
    }
    
    int numOfWays(vector<int>& nums) {
        prepare();
        return helper(nums) - 1;
    }
    int helper(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return 1;
        vector<int> left;
        vector<int> right;
        for (int i = 1; i < n; ++i) {
            if (nums[i] <= nums[0]) left.push_back(nums[i]);
            else right.push_back(nums[i]);
        }
        int lsz = left.size();
        int rsz = right.size();
        int c = C(lsz + rsz, lsz);
        return mul(c, mul(helper(left), helper(right)));
    }
};
