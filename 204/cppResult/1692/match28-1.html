const int mod = (int)1e9 + 7;
class Solution {
public:

int fact[2020];
inline int factorial(int x) {
   return fact[x];
}

#define int long long

int power(int a, int b, int m = mod){a %= m; if(b < 0)b += m-1; int r = 1; while(b){if(b&1)r=(r*a)%m; a = (a*a)%m; b>>=1;}return r;}
#undef int

int f(int a, int b) {
   int nr = factorial(a + b);
   int dr = (factorial(a) * 1LL * factorial(b)) % mod;
   dr = power(dr, -1);
   nr = (nr*1LL*dr) % mod;
   return nr;
}

template <typename T>
int n(vector<T>& P) {
  if (P.size() <= 1) return 1;
  vector<T> L, R;

  for (int i = 1; i < (int)(P.size()); i++) {
    if (P[i] < P[0])
      L.push_back(P[i]);
    else
      R.push_back(P[i]);
  }
  return ((n(L) * 1LL * n(R)) % mod * 1LL * f(L.size(), R.size())) % mod;
}

    int numOfWays(vector<int>& nums) {
        fact[0] = 1;
        for (int i = 1; i <= 2000; ++i)
        {
           fact[i] = (i * 1LL * fact[i-1]) % mod;
        }
        return (n(nums) - 1 + mod) % mod;
    }
};
