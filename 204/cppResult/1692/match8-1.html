class Solution {
 public:
  const int M = 1e9 + 7;
  int f[1010][1010];
  int c[1010][1010];
  int numOfWays(vector<int> &nums) {
    memset(f, 0, sizeof f);
    memset(c, 0, sizeof c);

    c[0][0] = 1;
    for (int i = 1; i < 1001; ++i) {
      c[i][0] = 1;
      c[i][i] = 1;
      for (int j = 1; j < i; ++j) {
        c[i][j] = (0ll + c[i - 1][j] + c[i - 1][j - 1]) % M;
      }
    }
    f[0][0] = 1;
    return (dfs(nums) - 1 + M) % M;
  }

  int dfs(vector<int> &nums) {
    int n = nums.size();
    if (n == 0) return 1;
    int root = nums[0];

    vector<int> left, right;

    for (int i = 1; i < n; ++i) {
      if (nums[i] < root) {
        left.push_back(nums[i]);
      } else {
        right.push_back(nums[i]);
      }
    }
    // if (f[n][left.size()] != 0) return f[n][left.size()];
    // printf("n %d l %d\n", n, (int)left.size());
    // int &ans = f[n][left.size()];
    int ans = 0;
    // if (n == 44)
    //   printf("n %d l %d c %d left %d right %d\n", n, (int)left.size(),
    //          c[n - 1][left.size()], dfs(left), dfs(right));
    ans = 1ll * (1ll * dfs(left) * dfs(right) % M) * c[n - 1][left.size()] % M;
    return ans;
  }
};
