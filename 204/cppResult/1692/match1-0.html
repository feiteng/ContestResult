class Solution {
public:
    int MD = 1e9 + 7;
    int C[1005][1005];
    int numOfWays(vector<int>& nums) {
        C[0][0] = 1;
        for(int i = 1; i < 1005; i++) {
            C[i][0] = 1;
            for(int j = 1; j <= i; j++) {
                C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % MD;
            }
        }
        return (dfs(nums) - 1 + MD) % MD;
    }
    int dfs(vector<int>& nums) {
        if(nums.size() <= 1) return 1;
        vector<int> A, B;
        for(int i = 1; i < nums.size(); i++) {
            if(nums[i] < nums[0]) A.push_back(nums[i]);
            else B.push_back(nums[i]);
        }
        int ans = C[(int)(nums.size()) - 1][(int)(A.size())];
        return 1LL * ans * dfs(A) % MD * dfs(B) % MD;
    }
};
