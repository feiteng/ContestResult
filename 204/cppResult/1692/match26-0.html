class Solution {
public:
    const int m = 1000000000 + 7;
    int c[1005][1005];    // 组合数

    // long long c(int n,int x){
    //     long long u = 1;
    //     for(int i = 0;i < x;i++) {
    //         u = u * (n - i) % m;
    //     }
    //     for(int i = 1;i <= x;i++) {
    //         u = u / i;
    //     }
    //     return u;
    // }
    int helper(vector<int>& nums) {
        int n = nums.size();
        if(n <= 1)return 1;
        vector<int> left,right;
        int cntLeft = 0,cntRight = 0;
        for(int i = 1; i < n;i++) {
            if(nums[i] > nums[0]){
                right.push_back(nums[i]);
                cntRight++;
            }else{
                left.push_back(nums[i]);
                cntLeft++;
            }
        }
        long long l = helper(left);
        // for(int x:left)cout<<x<<",";cout<<endl;
        // cout<<"l = "<<l<<endl;
        long long r = helper(right);
        // for(int x:right)cout<<x<<",";cout<<endl;
        // cout<<"r = "<<r<<endl;
        long long t = c[cntRight + cntLeft][cntLeft];
        return (t * l) % m * r % m;
    }
    int numOfWays(vector<int>& nums) {
        int n = nums.size();
        c[0][0] = c[1][0] = c[1][1] = 1;
        for (int i=2; i<1002; ++i){
            c[i][0] = 1;
            for (int j=1; j<=i; ++j) 
                c[i][j] = (c[i-1][j] + c[i-1][j-1]) % m;
        }

        return helper(nums) - 1;
        
    }
    //[12,13,4,2,9,5,11,3,10,7,1,15,14,8,6] 2620799
};
