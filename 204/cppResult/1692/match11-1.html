class Solution {
public:
  
// Function to find modulo inverse of a 
int modInverse(int a, int m) 
{ 
    int x, y; 
    int g = gcdExtended(a, m, &x, &y); 
    if (g != 1) {
        cout << "Inverse doesn't exist"; 
      return 0;
    }
    else
    { 
        // m is added to handle negative x 
        return (x%m + m) % m; 
        // cout << "Modular multiplicative inverse is " << res; 
    }
} 
  
// C function for extended Euclidean Algorithm 
int gcdExtended(int a, int b, int *x, int *y) 
{ 
    // Base Case 
    if (a == 0) 
    { 
        *x = 0, *y = 1; 
        return b; 
    } 
  
    int x1, y1; // To store results of recursive call 
    int gcd = gcdExtended(b%a, a, &x1, &y1); 
  
    // Update x and y using results of recursive 
    // call 
    *x = y1 - (b/a) * x1; 
    *y = x1; 
  

    return gcd; 
} 

int64_t C(int m, int n) {
  int64_t ans = 1;
  for (int i = m; i > m - n; i--) {
    ans = (ans * i) % M;
  }
  for (int i = 1; i <= n; i++) {
    ans = (ans * modInverse(i, M)) % M;
  }
  // cout << "ans " << ans << endl;
  return ans;
}
  
  int M = 1000000007;


  int numOfWays2(vector<int>& nums) {
    if (nums.size() <= 2) return 1;
    vector<int> small;
    vector<int> large;
    for (int i = 1; i < nums.size(); i++) {
      if (nums[i] > nums[0]) {
        large.push_back(nums[i]);
      } else {
        small.push_back(nums[i]);
      }
    }
    int64_t ans = 0;
    int64_t large_ways = numOfWays2(large);
    int64_t small_ways = numOfWays2(small);
    // cout << large.size() << endl;
    // cout << small.size() << endl;
    ans = (large_ways * small_ways) % M;
    ans = (ans * C(large.size()+small.size(), min(large.size(), small.size()))) % M;
    return (int)ans;
  }
  
  int numOfWays(vector<int>& nums) {
    return numOfWays2(nums) - 1;
  }
};
