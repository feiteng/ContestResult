class Solution {
public:
    static constexpr int MAX = 1010;
    static constexpr int mod = 1e9+7;
    vector<int> fac;
    long long qpow(long long a, int n) {
        long long ret = 1;
        while(n) {
            if(n&1) ret = ret*a%mod;
            a = a*a%mod;
            n >>= 1;
        }
        return ret;
    }
    long long C(int n, int x) {
        return 1LL*fac[n]*qpow(fac[x], mod-2)%mod*qpow(fac[n-x], mod-2)%mod;
    }
    void dfs(const vector<int> nums, int &ans) {
        if(nums.size() == 0 || nums.size() == 1) return;
        int n = nums.size(), x = nums[0];

        vector<int> l, r;
        for(int i = 1; i < n; i++) {
            if(nums[i] < x) {
                l.push_back(nums[i]);
            } else {
                r.push_back(nums[i]);
            }
        }
        ans = 1LL*ans*C(l.size()+r.size(), l.size())%mod;
        dfs(l, ans);
        dfs(r, ans);

    }
    int numOfWays(vector<int>& nums) {
        fac.resize(MAX);
        fac[0] = 1;
        for(int i = 1; i < MAX; i++) {
            fac[i] = 1LL*fac[i-1]*i%mod;
        }
        int n = nums.size();
        int ans = 1;
        dfs(nums, ans);
        ans = (ans-1+mod)%mod;
        return ans;
    }
};
