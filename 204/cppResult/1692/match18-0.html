class Solution {
public:
    int deno = pow(10,9)+7;
    int mat[1000][1000];
 
int combinat(int m, int n) {
    if(mat[m][n]!=0)
        return mat[m][n];
    if(n == 0 || m == n)
        return 1;
    mat[m][n] = (combinat(m-1,n-1)+combinat(m-1,n))%deno;
    return mat[m][n];
}


    long dfs(vector<int>& nums){
        if(nums.size()<=1)
            return 1;
        int root = nums[0];
        vector<int> left,right;
        for(int i =1;i<nums.size();i++){
            if(nums[i]>root){
                right.emplace_back(nums[i]);
            }else{
                left.emplace_back(nums[i]);
            }
        }
        long ret = combinat(nums.size()-1,right.size())*dfs(right);
        ret%=deno;
        ret = ret*dfs(left);
        return ret%deno;
    }
    int numOfWays(vector<int>& nums) {
        
        return dfs(nums)-1;
    }
};
