const int N = 1001;
class Solution {
public:
    
    const int mod = (int)1e9 + 7;
    
    int c[N][N];
    int calc(vector <int> &a) {
        if (a.size() <= 1) return 1;
        vector <int> L, R;
        for (auto it : a) {
            if (it < a[0]) L.push_back(it);
            else if (it > a[0]) R.push_back(it);
        }
        int res = calc(L) * 1LL * calc(R) % mod;
        res = res * 1LL * c[L.size() + R.size()][L.size()] % mod;
        return res;
    }

    int numOfWays(vector<int>& a) {
        c[0][0] = 1;
        for (int i = 1; i < N; i++) {
            c[i][0] = 1;
            for (int j = 1; j <= i; j++) {
                c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % mod;
            }
        }
        return (calc(a) - 1 + mod) % mod;
    }
};
