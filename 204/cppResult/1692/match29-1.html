class Solution {
public:
    const int MAX = 1000000007;
    void search(vector<int>& nums,int begin,int end,int x,long long& res,int l,vector<vector<int>>&cnt){
        if(begin>end){
            return;
        }
        res = (res*cnt[end-begin][x-begin])%MAX;
        //cout <<res<<endl;
        for(int i =l;i<nums.size();i++){
            if(nums[i]<x &&nums[i]>=begin){
                search(nums,begin,x-1,nums[i],res,i,cnt);
                break;
            }
        }
        for(int i =l;i<nums.size();i++){
            if(nums[i]>x&&nums[i]<=end){
                search(nums,x+1,end,nums[i],res,i,cnt);
                break;
            }
        }
    }
    int numOfWays(vector<int>& nums) {

        int n = nums.size();
        vector<vector<int>> cnt(n+1,vector<int>(n+1,1));
        for(int i =1;i<=n;i++){
            for(int j =1;j<=i;j++){
                if(i==j){
                    cnt[i][j] = 1;
                }
                else{
                    cnt[i][j] = (cnt[i-1][j]+cnt[i-1][j-1])%MAX;
                }
            }
        }
        long long res = 1;
        search(nums,1,n,nums[0],res,0,cnt);
        return res-1;
    }
};
