const int MAXN = 1000 + 1;
const int MOD = 1e9 + 7;

int comb[MAXN][MAXN];

class Solution {
public:
    Solution() {
        if (comb[0][0] != 0) {
            return;
        }
        for (int i = 0; i < MAXN; ++i) {
            comb[0][i] = comb[i][0] = 1;
        }
        for (int i = 1; i < MAXN; ++i) {
            for (int j = 1; j < MAXN; ++j) {
                comb[i][j] = (comb[i - 1][j] + comb[i][j - 1]) % MOD;
            }
        }
    }
    
    int numOfWays(vector<int>& nums) {
        return _numOfWays(nums) - 1;
    }
    
    int _numOfWays(vector<int>& nums) {
        int n = static_cast<int>(nums.size());
        if (n <= 2) {
            return 1;
        }
        vector<int> left, right;
        for (int i = 1; i < n; ++i) {
            if (nums[i] < nums[0]) {
                left.push_back(nums[i]);
            } else {
                right.push_back(nums[i]);
            }
        }
        int ln = static_cast<int>(left.size()), rn = static_cast<int>(right.size());
        long long lm = _numOfWays(left), rm = _numOfWays(right);
        return (lm * rm) % MOD * comb[ln][rn] % MOD;
    }
};
