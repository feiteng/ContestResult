const int64_t  MOD=1e9+7;
vector<vector<int64_t>> combine(1005,vector<int64_t>(1005,1));
int tag=0;
class Solution {
public:

    int numOfWays(vector<int>& nums) {
        if(!tag){
            for(int i=1;i<1005;i++){
                for(int j=1;j<1005;j++){
                    combine[i][j]=combine[i-1][j]+combine[i][j-1];
                    combine[i][j]%=MOD;
                }
            }
            tag=1;
        }
        return (helper(nums)+MOD-1)%MOD;

    }
    int64_t helper(vector<int> & x){

        int n=x.size();
        if(x.size()<=2)
            return 1;
        vector<int> low,high;
        for(int i=1;i<n;i++){
            if(x[i]>x[0])
                high.push_back(x[i]);
            else
                low.push_back(x[i]);
        }
//        cout<<low.size()<<" "<<high.size()<<endl;
        return calc(low.size(),high.size())*helper(low)%MOD*helper(high)%MOD;
    }
    int64_t calc(int a, int b){
        if(a>b)
            swap(a,b);
        if(a==0)
            return 1;
        int64_t ret=0;
        for(int i=0;i<a;i++){
            ret+=combine[i+1][b-i]*combine[i][a-1-i];
            ret%=MOD;
        }
        return ret;

    }
};
