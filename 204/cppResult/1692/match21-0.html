class Solution {
public:
    
    const int MOD = 1000000007;
    
    int fac[1005];
    int ifac[1005];
    int n;
    
    int power(int x, int y) {
        int res = 1;
        while (y) {
            if (y % 2) res = 1LL * res * x % MOD;
            y >>= 1;
            x = 1LL * x * x % MOD;
        }
        return res;
    }
    
    int inv(int x) {
        return power(x, MOD - 2);
    }
    
    int C(int k, int n) {
        return 1LL * fac[n] * ifac[k] % MOD * ifac[n - k] % MOD;
    }
    
    int numOfWays(vector<int>& nums) {
        if (n == 0) {
            n = nums.size();
            fac[0] = ifac[0] = 1;
            for (int i = 1; i <= n; i++) {
                fac[i] = 1LL * fac[i - 1] * i % MOD;
                ifac[i] = inv(fac[i]);
            }
            if (n <= 1) return 0;
        }
        if (nums.size() <= 1) return 1;
        int pivot = nums[0];
        vector<int> left;
        vector<int> right;
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] < pivot) left.push_back(nums[i]);
            else right.push_back(nums[i]);
        }
        
        int ans = 1LL * numOfWays(left) * numOfWays(right) % MOD * C(left.size(), left.size() + right.size()) % MOD; 
        if (n == nums.size()) return (ans + MOD - 1) % MOD;
        else return ans;
    }
};
