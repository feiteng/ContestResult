const int mod = 1000000007;
int C[1005][1005];
class Solution {
public:
    int numOfWays(vector<int>& nums) {
        getC();
        return (dfs(nums) - 1 + mod) % mod;
    }
private:
    int dfs(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return 1;
        int x = nums[0];
        vector<int> l, r;

        for (int i = 1; i < n; ++i) {
            if (nums[i] > x) r.push_back(nums[i]);
            else l.push_back(nums[i]);
        }
        long long ans = C[n - 1][l.size()];
        ans = ans * dfs(l) % mod;
        ans = ans * dfs(r) % mod;
        return ans;
    }

    void getC() {
        for (int i = 0; i <= 1000; ++i) {
            for (int j = 0; j <= i; ++j) {
                if (j == 0 || j == i) {
                    C[i][j] = 1;
                } else {
                    C[i][j] = C[i - 1][j - 1] + C[i - 1][j];
                    if (C[i][j] >= mod)
                        C[i][j] -= mod;
                }
            }
        }
    }
};
