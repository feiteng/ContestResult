class Solution {
public:
    
    vector<vector<int>> nck;
    
    const int mod = (int)1e9 + 7;
    
    int mul(int x, int y){
        return x * 1ll * y % mod;
    }
    

    int solve(vector<int> vec){
        int n = vec.size();
        if(n <= 1) return 1;
        vector<int> small, big;
        for(int i = 1; i < n; ++i){
            if(vec[i] < vec[0]) small.push_back(vec[i]);
            else if(vec[i] > vec[0]) big.push_back(vec[i]);
            else assert(false);
        }
        int ret0 = solve(small);
        int ret1 = solve(big);
        
        int sz0 = small.size();
        int sz1 = big.size();
        
        int res = mul(ret0, ret1);
        res = mul(res, nck[sz0 + sz1][sz0]);
        return res;
    }
    
    int numOfWays(vector<int>& nums) {
        int n = nums.size();

        nck.resize(n + 1, vector<int>(n + 1));
        
        nck[0][0] = 1;
        for(int i = 1; i <= n; ++i){
            nck[i][0] = 1;
            nck[i][i] = 1;
            for(int j = 1; j <= i; ++j){
                nck[i][j] = (nck[i - 1][j] + nck[i - 1][j - 1]) % mod;
            }
        }
        
        int res = solve(nums) - 1;
        if(res < 0) res += mod;
        return res;
    }
};
