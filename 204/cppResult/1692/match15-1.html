typedef long long ll;

class Solution {
public:
    int numOfWays(vector<int>& nums) {
        calc_c(nums.size(), nums.size());
        return ways(nums) - 1;
    }
private:
    ll BASE = 1000000007;
    int dp[1005][1005];
    int ways(vector<int>& nums) {
        if(nums.size() <= 1) { return 1; }

        vector<int> ls, gt;
        for(int i = 1; i < nums.size(); i++) {
            if(nums[i] > nums[0]) { gt.push_back(nums[i]); }
            else { ls.push_back(nums[i]); }
        }
        int C = dp[ls.size()][nums.size() - 1];
        int lways = ways(ls);
        int gways = ways(gt);
        ll res = C;
        res *= lways; res %= BASE;
        res *= gways; res %= BASE;
        return res;
    }

    void calc_c(int n, int m) {
        for(int j = 0; j <= m; j++) { dp[0][j] = 1; }
        for(int i = 1; i <= n; i++) { dp[i][0] = 0; }
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                dp[i][j] = dp[i - 1][j - 1] + dp[i][j - 1];
                dp[i][j] %= BASE;
            }
        }
    }
};
