#define N 1010
#define P 1000000007
class Solution {
    int C[N][N];
    
    int ff(vector<int> a) {
        if (a.size() <= 1) return 1;
        vector<int> b,c;
        for (auto i : a) {
            if (i < a[0]) b.push_back(i);
            if (i > a[0]) c.push_back(i);
        }
        long long S = C[b.size()+c.size()][b.size()];
        S = S*ff(b)%P;
        S = S*ff(c)%P;
        return S;
    }
public:
    int numOfWays(vector<int>& a) {
        int n = a.size();
        for (int i = 0; i <= n; i ++) {
            C[i][0] = 1;
            for (int j = 1; j <= i; j++) {
                C[i][j] = (C[i-1][j] + C[i-1][j-1])%P;
            }
        }
        return (ff(a)-1+P)%P;
    }
};
