#define MOD 1000000007

class Solution {
public:
  int c[1010][1010];
  long long res = 1;
  
  void construct(vector<int>& nums) {
    if (nums.empty()) return;
    
    int root = nums[0];
    
    vector<int> left, right;
    for (int x : nums) {
      if (x < root) left.push_back(x);
      else if (x > root) right.push_back(x);
    }
    construct(left);
    construct(right);
    
    res *= c[left.size() + right.size()][left.size()];
    res %= MOD;

  }
  
  int numOfWays(vector<int>& nums) {
    int n = nums.size();
    
    for (int i = 0; i <= n; ++i) {
      for (int j = 0; j <= i; ++j) {
        if (j == 0 || i == j) c[i][j] = 1;
        else {
          c[i][j] = ((long long)c[i-1][j-1] + (long long)c[i-1][j]) % MOD;
        }
      }
    }
    
    construct(nums);
    
    res += MOD - 1;
    res %= MOD;
    
    return (int) res;
  }
};
