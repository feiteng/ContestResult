const int MOD = 1e9 + 7;

class Solution {
    vector<vector<int>> mem;
public:
    int numOfWays(vector<int>& nums) {
        int n = nums.size();
        mem.resize(n + 1, vector<int>(n + 1));

        mem[1][0] = 1, mem[1][1] = 1;
        for (int i = 2; i <= n; ++i) {
            mem[i][0] = 1;
            for (int j = 1; j <= n; ++j) {
                mem[i][j] = (mem[i - 1][j] + mem[i - 1][j - 1]) % MOD;
            }
        }
        int res = solve(nums);
        return (res - 1 + MOD) % MOD;
    }
        
    int solve(vector<int>& nums) {

        if (nums.size() <= 1)
            return 1;
        vector<int> left, right;
        int x = nums[0], n = nums.size();
        for (int i = 1; i < n; ++i) {
            if (nums[i] < x)
                left.push_back(nums[i]);
            else
                right.push_back(nums[i]);
        }
        int leftres = solve(left), rightres = solve(right);
        int leftlen = left.size(), rightlen = right.size();
        int res = mem[leftlen + rightlen][leftlen];
        res = ((long long)res * leftres) % MOD;
        res = ((long long)res * rightres) % MOD;
        return res;
    }
    
};
