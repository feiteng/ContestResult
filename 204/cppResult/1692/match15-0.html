using ll = long long;
const ll mod = 1e9 + 7;
const int N = 1010;

class Solution {
public:
    vector<ll> fac, ifac;
    ll solve(vector<int> a) {
        int n = a.size();
        if (n<=1) return 1;
        vector<int> left, right;
        for (int i=1; i<n; i++) {
            if (a[i]<a[0]) {
                left.push_back(a[i]);
            } else {
                right.push_back(a[i]);
            }
        }
        int L = left.size();
        int R = right.size();
        ll rec = solve(left)*solve(right)%mod;
        ll res = fac[n-1]*rec%mod*ifac[L]%mod*ifac[R]%mod;
        return res;
    }
    
    ll pw(ll x, ll y) {
    ll res=1;
    while (y) {
	if (y%2) res=res*x%mod;
	x=x*x%mod;y=y/2;
    }
    return ((res%mod)+mod)%mod;
}
    
    int numOfWays(vector<int>& a) {
        fac.push_back(1);
        ifac.push_back(1);
        for (int i=1; i<=N; i++) {
            ll tmp = (1ll*i*fac[i-1])%mod;
            fac.push_back(tmp);
        }
        for (int i=1; i<=N; i++) {
            ifac.push_back(pw(fac[i], mod-2));
        }
        int n = a.size();
        ll res = solve(a);
        res --;
        res %= mod;
        res += mod;
        res %= mod;
        return res;
    }
};
