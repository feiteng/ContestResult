class Solution {
public:
    int p=1e9+7;
    
    
typedef long long LL;
 
LL quick_mod(LL a, LL b)
{
    LL ans = 1;
    a %= p;
    while(b)
    {
        if(b & 1)
        {
            ans = ans * a % p;
            b--;
        }
        b >>= 1;
        a = a * a % p;
    }
    return ans;
}
 
LL C(LL n, LL m)
{
    if(m > n) return 0;
    LL ans = 1;
    for(int i=1; i<=m; i++)
    {
        LL a = (n + i - m) % p;
        LL b = i % p;
        ans = ans * (a * quick_mod(b, p-2) % p) % p;
    }
    return ans;
}
 
LL Lucas(LL n, LL m)
{
    if(m == 0) return 1;
    return C(n % p, m % p) * Lucas(n / p, m / p) % p;
}
    
    int numWays(vector<int>& nums) {
        int n=nums.size();
        // if (n==0)
            // return 0;
        if (n<=2)
            return 1;
        
        
        
        int k=nums[0];
        vector<int> left, right;
        for (int i=1;i<nums.size();i++) {
            if (nums[i] > k)
                right.push_back(nums[i]);
            else
                left.push_back(nums[i]);
        }
        
        LL leftnum=numWays(left), rightnum=numWays(right);
        LL t = Lucas(left.size()+right.size(), left.size()) % p;
        
        // for (auto& i:nums)
        //     cout<<i<<' ';
        // cout<<": ";
        // cout<<leftnum<<" "<<rightnum<<' '<<t<<endl;
        return ((leftnum*t) % p) * rightnum % p;
        
    }
    
    int numOfWays(vector<int>& nums) {
        // cout<<endl;
        return numWays(nums) - 1;
    }
};
