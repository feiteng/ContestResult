typedef long long LL;
class Solution {
    LL p = 1000000000 + 7;
public:
    int numOfWays(vector<int>& nums) {
        if(nums.size()==0) return 1;
        int top = nums[0];
        vector<int> l;
        vector<int> r;
        for(int i=1;i<nums.size();++i) {
            if(nums[i]<top) {
                l.push_back(nums[i]);
            } else {
                r.push_back(nums[i]);
            }
        }
        LL ans=Lucas(l.size()+r.size(),l.size());
        ans = ans *numOfWays2(l) % p;
         ans = ans * numOfWays2(r) %p;
        return ans-1;
    }
    int numOfWays2(vector<int>& nums) {
        if(nums.size()==0) return 1;
        int top = nums[0];
        vector<int> l;
        vector<int> r;
        for(int i=1;i<nums.size();++i) {
            if(nums[i]<top) {
                l.push_back(nums[i]);
            } else {
                r.push_back(nums[i]);
            }
        }
        LL ans=Lucas(l.size()+r.size(),l.size());
        ans = ans *numOfWays2(l) % p;
         ans = ans * numOfWays2(r) %p;
        return ans;
    }
     
LL quick_mod(LL a, LL b)

{
    LL ans = 1;
    a %= p;
    while(b)
    {
        if(b & 1)
        {
            ans = ans * a % p;
            b--;
        }
        b >>= 1;
        a = a * a % p;
    }
    return ans;
}
 
LL C(LL n, LL m)
{
    if(m > n) return 0;
    LL ans = 1;
    for(int i=1; i<=m; i++)
    {
        LL a = (n + i - m) % p;
        LL b = i % p;
        ans = ans * (a * quick_mod(b, p-2) % p) % p;
    }
    return ans;
}
 
LL Lucas(LL n, LL m)
{
    if(m == 0) return 1;
    return C(n % p, m % p) * Lucas(n / p, m / p) % p;
}

};
