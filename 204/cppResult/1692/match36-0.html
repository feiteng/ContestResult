class Solution {
public:
    long long a[1001][1001] = {0};
    int numOfWays(vector<int>& nums) {
        for (int i = 0; i <= nums.size(); i++) {
            a[i][0] = 1;
            a[i][i] = 1;
            for (int j = 1; j < i; j++) {
                a[i][j] = (a[i-1][j-1] + a[i-1][j]) % (1000000007);
                // cout << a[i][j] << " ";
            }
            // cout << endl;
        }
        return (go(nums)-1) % (1000000007);
    }
    
    long long go(vector<int>& nums) {
        if (nums.size() == 0 || nums.size() == 1) return 1;
        vector<int> v1, v2;
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] < nums[0]) v1.push_back(nums[i]);
            else v2.push_back(nums[i]);
        }
        long long a1 = go(v1), a2 = go(v2);
        // for (int i = 0; i < nums.size(); i++) cout << nums[i] << " ";
        // cout << endl;
        // cout << v1.size() << " " << v2.size() << endl;
        // cout << a1 << " " << a2 << endl;
        long long ans = (a1 * a2)  % (1000000007);
        return (ans * a[v1.size()+v2.size()][v1.size()])  % (1000000007);
    }
};
