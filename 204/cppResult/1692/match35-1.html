class Solution {
public:
  
  #define mod 1000000007
  #define maxn 1005
  
  long long perm[maxn];
  
  long long qpow(long long a, long long b) {

    long long ans = 1;
    while (b) {
      if (b & 1) ans = ans * a % mod;
      a = a * a % mod;
      b >>= 1;
    }
    return ans;
  }
  
  int dfs(vector<int>& vec) {
    if (vec.size() <= 1) return 1;
    vector<int> a, b;
    int rt = vec[0];

    for (int i = 1; i < vec.size(); i ++) {
      if (vec[i] > rt) a.push_back(vec[i]);
      else b.push_back(vec[i]);
    }
    int na = dfs(a);
    int nb = dfs(b);
    int len = vec.size();
    int ans = perm[len - 1];
    long long r = perm[a.size()] * perm[b.size()] % mod;
    ans = ans * qpow(r, mod - 2) % mod;
    return 1ll * ans * na % mod * nb % mod;
  }
  
  int numOfWays(vector<int>& nums) {
    perm[0] = 1;
    for (int i = 1; i <= 1000; i ++)
      perm[i] = perm[i - 1] * i % mod;
    return (dfs(nums) + mod - 1) % mod;
  }
};
