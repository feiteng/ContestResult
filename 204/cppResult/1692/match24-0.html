typedef long long ll;
const int MOD = 1e9+7;
class Solution {
public:
    vector<vector<ll>> dp;
    ll dfs(vector<int>& a){
        if (a.size() <= 1) return 1;
        vector<int> l, r;
        for (int i = 1; i < (int)a.size(); i++){
            if (a[i] < a[0]) l.push_back(a[i]);
            else if (a[i] > a[0]) r.push_back(a[i]);
        }
        ll ans = dfs(l)*dfs(r)%MOD;
        ans = ans*dp[l.size()][r.size()]%MOD;
        // ll la = dfs(l), ra = dfs(r);
        // cout << a[0] << " " << ans << " " << la << " " << ra << "\n";
        return ans;
        
    }
    int numOfWays(vector<int>& nums) {
        int n = nums.size();
        dp.assign(nums.size()+1, vector<ll>(nums.size()+1, 0));
        for (int i = 0; i <= n; i++){
            for (int j = 0; j <= n; j++){
                if (!i || !j) dp[i][j] = 1;
                else {
                    dp[i][j] = (dp[i-1][j]+dp[i][j-1])%MOD;
                }
            }
        }
        // for (int i = 0; i <= 2; i++){
            // for (int j = 0; j <= 2; j++) cout << dp[i][j] << " ";
            // cout << "\n";
        // }
        return (dfs(nums)+MOD-1)%MOD;
    }
};
