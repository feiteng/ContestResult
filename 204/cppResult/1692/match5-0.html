const int MOD = 1e9 + 7;
const int N = 1e3 + 10;
int C[N][N];

class Solution {
public:
    int solve(const vector<int>& a) {
        int n = a.size();
        if (n == 0) return 1;
        vector<int> ll, rr;
        //cout << "n=" << n << endl;
        for (int i = 1; i < n; ++i) {
            if (a[i] > a[0]) {
                rr.push_back(a[i]);
            } else {
                ll.push_back(a[i]);
            }
        }
        //cout << ll.size() << " " << rr.size() << endl;
        int ret = C[ll.size() + rr.size()][ll.size()];
        int x = solve(ll), y = solve(rr);
        ret = 1LL * ret * x % MOD;
        ret = 1LL * ret * y % MOD;
        return ret;
    }
    int numOfWays(vector<int>& a) {
        int n = a.size();
        for (int i = 0; i < n; ++i) {
            C[i][0] = C[i][i] = 1;
            for (int j = 1; j < i; ++j) {
                C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % MOD;
            }
        }
        int ret = solve(a);
        ret = (ret + MOD - 1) % MOD;
        return ret;
    }
};
