class Solution {
public:
    #define MODD 1000000007
    #define ll long long
    
    ll binomial[1111][1111];
    
    
    ll Solve(const vector<int>& nums) {
        if (nums.empty()) return 1;
        int x = nums[0];

        vector<int> left,right;
        for(int i=1;i<nums.size();i++) {
            if (nums[i] < x) {
                left.push_back(nums[i]);
            } else {
                right.push_back(nums[i]);
            }
        }
        
        ll ct1 = Solve(left);

        ll ct2 = Solve(right);
        
        ll bval = binomial[left.size()+right.size()][left.size()];
        
        ll ret = (ct1*ct2)%MODD;
        ret = (ret*bval)%MODD;
        return ret;
        
    }
    
    int numOfWays(vector<int>& nums) {
        int N = nums.size();        
        for(int i=0;i<=N;i++)
            for(int j=0;j<=i;j++) {
                if (j == 0) {
                    binomial[i][j]=1;
                } else {
                    binomial[i][j]=(binomial[i-1][j]+binomial[i-1][j-1])%MODD;
                }
            }
        ll ans = (Solve(nums)+MODD-1)%MODD;
        return ans;
        
        
        
    }
};
