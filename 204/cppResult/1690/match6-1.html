class Solution {
public:
    int dp[100005][2];
    int getMaxLen(vector<int>& a) {
        int n = a.size();
        a.insert(a.begin(),0);

        dp[0][0] = -1;
        dp[0][1] = -1;
        for (int i = 1 ; i <= n ; i++){
            if (a[i] == 0){
                dp[i][0] = dp[i][1] = -1;
            }
            else if (a[i] < 0){
                if (dp[i - 1][0] == -1) dp[i][1] = 1;
                else dp[i][1] = dp[i - 1][0] + 1;
                if (dp[i - 1][1] == -1) dp[i][0] = -1;
                else dp[i][0] = dp[i - 1][1] + 1;
            }
            else {
                if (dp[i - 1][0] == -1) dp[i][0] = 1;
                else dp[i][0] = dp[i - 1][0] + 1;
                if (dp[i - 1][1] == -1) dp[i][1] = -1;
                else dp[i][1] = dp[i - 1][1] + 1;
            }
        //    cout << i << " " << dp[i][0] << ' ' << dp[i][1] << endl;
        }
        int ans = 0;
        for (int i = 1 ; i <= n ; i++) ans = max (ans , dp[i][0]);
        return ans;
    }
};
