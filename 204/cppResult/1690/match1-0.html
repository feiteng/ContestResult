class Solution {
public:
    int getMaxLen(vector<int>& nums) {

        int isOne=0;
        int N=nums.size();
        int o1=-1;
        int o2=N;
        int ans=0;
        for(int i=0;i<N;i++){
            if(nums[i]==0){
                o1=i;
                o2=N;
                isOne=0;
                continue;
            }else if(nums[i]<0){
                isOne^=1;
                if(o2==N){
                    o2=i;
                }
            }
            if(isOne==0){
                ans=max(ans,i-o1);
            }else{
                ans=max(ans,i-o2);
            }
        }
        return ans;
    }
};
