class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        int neg_num = 0, zero_pos = -1, even_pre = -1, odd_pre, ans = 0;

        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                zero_pos = i; neg_num = 0, even_pre = i;
                continue;
            }
            if (nums[i] < 0) neg_num++;
            if (neg_num % 2 == 1) {
                if (neg_num == 1 && nums[i] < 0) odd_pre = i;
                ans = max(ans, i - odd_pre);
            } else {
                ans = max(ans, i - even_pre);
            }
        }
        return ans;
    }
};
