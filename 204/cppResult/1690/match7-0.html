class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int pos=-1;
        int neg=nums.size()+5;
        int res=0;
        int cur=1;
        for (int i=0; i<nums.size(); ++i) {
            if (nums[i]==0) {
                pos=i;
                neg=nums.size()+5;
                cur=1;
                continue;
            }
            if (nums[i]>0) {
                cur*=1;
            } else if (nums[i]<0) {
                cur*=-1;
            }
            if (cur>0) {
                res=max(res, i-pos);
            } else if (cur<0) {
                neg=min(neg, i);
                res=max(res, i-neg);
            }
        }
        return res;
    }
};
