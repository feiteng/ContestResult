class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        int res = 0;
        int cntNeg = 0;
        unordered_map<int, int> dict;

        dict[0] = -1;
        for (int i = 0; i < nums.size(); ++i)
        {
            if (nums[i] == 0)
            {
                cntNeg = 0;
                dict.clear();
                dict[0] = i;
            }
            else
            {
                if (nums[i] < 0) ++cntNeg;
                if (dict.count(cntNeg % 2) != 0) res = max(res, i - dict[cntNeg % 2]);
            }
                                    
            if (dict.count(cntNeg % 2) == 0)
            dict[cntNeg % 2] = i;
        }
        
        return res;
    }
};
