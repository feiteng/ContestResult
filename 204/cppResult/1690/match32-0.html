class Solution {
public:
    int getMaxLen(vector<int>& a) {
        int n = a.size();
        int first = -1;
        int neg = 0;
        int ans = 0;
        int start = 0;
        for(int i = 0; i < n; i++) {
            if(a[i] == 0) {
                first = -1;
                neg = 0;
                start = i+1;
                continue;
            }
            else if(a[i] < 0) {
                neg++;
                if(first == -1) {
                    first = i;
                }
            }
            if(neg%2 == 0) {
                ans = max(ans, i-start+1);
            }
            else {
                ans = max(ans, i-first);
            }
        }
        return ans;
    }
};
