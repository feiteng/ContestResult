class Solution {
public:
    int getMaxLen(vector<int>& a) {
        int n = a.size();
        int lastn = -1;
        int curn = -1;
        int negn = 0;
        int last = 0;
        int r = 0;
        for (int i = 0; i <= n; i++) {
            if(i == n || a[i] == 0) {
                if(negn % 2 == 0) {
                    r = max(r, i - last);
                } else {
                    r = max(max(i - lastn - 1, curn - last), r);
                }
                lastn = curn = -1;
                last = i + 1;
                negn = 0;
            } else if(a[i] > 0) {
                
            } else if(a[i] < 0) {
                if(lastn == -1)
                    lastn = i;
                curn = i;
                ++negn;
            }
        }
        return r;
    }
};
