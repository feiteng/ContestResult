class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        
        int n=nums.size();
        
        int prev_zero=-1;
        int num_of_negative=0;
        int negative_pos=-2;
        
        int max_len=0;
        
        for(int i=0;i<n;i++)
        {
            if(nums[i]==0)
            {
                prev_zero=i;
                num_of_negative=0;
            }
            else if(nums[i]<0)
            {
                num_of_negative++;
            }
            
            if(negative_pos<prev_zero and num_of_negative%2==1)     negative_pos=i;
            
            int len=i-prev_zero;
            
            //cout<<prev_zero<<" "<<num_of_negative<<" "<<len<<endl;
            if(num_of_negative%2==0 and max_len<len)     max_len=len;
            //cout<<max_len<<endl;
            
            
            if(negative_pos<prev_zero)      continue;
            
            len=i-negative_pos;
            
            //cout<<negative_pos<<" neg "<<num_of_negative<<" "<<len<<endl;
            if(num_of_negative%2==1 and max_len<len)     max_len=len;
            //cout<<max_len<<endl;
            
        }
        //cout<<endl;
        
        return max_len;
    }
};
