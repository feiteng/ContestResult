class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int ans = 0, last1 = -1, last_1 = -1, cur = 0;   //cur0表示正或0
        for(int i = 0; i < nums.size(); i++)  {
            if(nums[i] == 0)   {
                last_1 = -1;
                last1 = i;
                cur = 0;
            }else  {
                cur ^= (nums[i] > 0 ? 0 : 1);
                if(cur == 0)  {
                    ans = max(ans, i - last1);
                }else  {
                    if(last_1 == -1)  {
                        last_1 = i;
                    }
                    ans = max(ans, i - last_1);
                }
            }
        }
        return ans;
    }
};
