class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        int now = 0,minodd = 0x3f3f3f3f,mineven = -1;
        int ans = 0;
        for(int i=0;i<n;i++){
            if(nums[i] == 0){
                minodd = 0x3f3f3f3f;
                mineven = i;
                now = 0;
                continue;
            }
            if(nums[i] < 0) now ^= 1;
            if(now == 0) ans = max(ans,i-mineven);
            else if(minodd != 0x3f3f3f3f) ans = max(ans,i-minodd);
            if(now == 0) mineven = min(mineven,i);
            else minodd = min(minodd,i);
        }
        return ans;
    }
};
