class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        
        unordered_map<int,int> m ;
        int s=0 ;
        int ans=0 ;
        m[0]=-1 ;
        for(int i=0;i<nums.size();i++)
        {
           if(nums[i]==0)
           {
               m.clear();
               m[0]=i;
               s=0;
               continue; 
           }
          s+= (int)(nums[i]<0);
            
            if(m.count(s%2)!=0) ans=max(i-m[s%2],ans);
            else m[s%2]=i;
            }
        return ans ;
        
    }
};
