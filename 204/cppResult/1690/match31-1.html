class Solution {
public:
    int getMaxLen(vector<int>& nums) {

       int lp=-1,ln = -1;
        int cur = +1;
        int ans = 0;
        for(int i = 0;i<nums.size(); ++i) {
            if(nums[i] == 0) {
                ln = -1;
                lp = i;
                cur = +1;
                continue;
            }
            cur *= nums[i] / abs(nums[i]);

            if(cur == -1 && ln == -1) {
                ln = i;
            }
            if(cur == 1) {
                ans = max(ans, i-lp);
            } else {
                ans = max(ans, i-ln);
            }
        }
        return ans;
    }
};
