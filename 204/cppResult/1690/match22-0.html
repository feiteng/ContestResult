class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int l = 0;
        int neg = -1;
        int res = 0;
        bool pos = true;
        for(int r = 0; r < nums.size(); r++){
            if(nums[r] == 0){
                l = r + 1;
                neg = -1;
                pos = true;
                continue;
            }
            if(nums[r] < 0) pos = !pos;
            if(pos) res = max(res, r - l + 1);
            else{
                if(neg == -1) neg = r;
                else res = max(res, r - neg);
            }
        }
        return res;
    }
};
