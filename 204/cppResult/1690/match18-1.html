class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        
        int N=nums.size();
        set<int> neg_ids;
        int nz = N;
        int ans=0;
        for(int i=N-1;i>=0;i--) {
            if (nums[i]<0) {
                neg_ids.insert(i);
            } else if (nums[i]==0) {
                neg_ids.clear();
                nz = i;
            }
            int j;
            if (neg_ids.size()%2==0) {
                j=nz-1;
            } else {
                j=(*neg_ids.rbegin())-1;
            }
            ans = max(ans, j-i+1);
        }
        return ans;
    }
};
