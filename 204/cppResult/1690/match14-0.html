class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        vector<int> pos(2, nums.size());
        
        pos[0] = -1;
        int tmp = 0;
        
        int maxlen = 0;
        for(int i=0;i<nums.size(); i++)
        {
            if(nums[i] < 0)
            {
                tmp = (tmp + 1)%2;
                pos[tmp] = min(pos[tmp],i);
            }
            else if(nums[i] == 0)
            {
                tmp = 0;
                pos[0] = i;
                pos[1] = nums.size();
            }
            maxlen = max(maxlen, i - pos[tmp]);
        }
        
        return maxlen;
    }
};
