class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int result = 0;
        int posCount = 0, negCount = 0;
        for (int num : nums) {
            if (num == 0) {
                posCount = 0;
                negCount = 0;
            } else if (num > 0) {
                posCount++;
                if (negCount > 0) {    //!1
                    negCount++;
                }
            } else {
                //!1
                int newPos = 0, newNeg = 0;
                if (negCount > 0) {
                    newPos = negCount + 1;
                }
                newNeg = posCount + 1;
                posCount = newPos;
                negCount = newNeg;
            }
            result = max(result, posCount);
        }
        return result;
    }
};
