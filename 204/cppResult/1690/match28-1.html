class Solution {
public:
    int getMaxLen(vector<int>& nums) {

        int cur_pos = 0, cur_neg = 0, max_pos = 0;
        for (int n : nums) {
            if (n > 0) {
                cur_pos = max(1, cur_pos + 1);
                if (cur_neg) cur_neg++;
            }
            else if (n < 0) {
                int old_neg = cur_neg;
                cur_neg = max(1, cur_pos + 1);
                if (old_neg) cur_pos = old_neg + 1;

                else cur_pos = 0;
            }
            else {
                cur_pos = cur_neg = 0;
            }
            // cout << cur_pos << ' ' << cur_neg << endl;
            max_pos = max(max_pos, cur_pos);
        }
        return max_pos;
    }
};
