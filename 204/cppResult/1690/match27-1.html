class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int pos = 0;
        int neg = 0;
        int max_pos = 0;
        for(int i = 0; i < nums.size(); i++) {
            if(nums[i] > 0) {
                if(neg > 0) { neg++; }
                pos++;
            }
            else if(nums[i] < 0) {
                int next_pos = (neg == 0) ? 0 : neg + 1;
                int next_neg = pos + 1;

                pos = next_pos; neg = next_neg;
            }
            else {
                pos = 0; neg = 0;
            }
            max_pos = max(pos, max_pos);
        }
        return max_pos;
    }
};
