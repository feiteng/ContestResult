class Solution {
public:
	int getMaxLen(vector<int>& nums) {
		int d1 = 0;
		int d2 = 0;
		int ans = 0;
		for (int i : nums) {
			if (i > 0) {
				d1++;
				if (d2 > 0)d2++;
			}
			else if (i < 0) {
				swap(d1, d2);
				if (d1 > 0)d1++;
				d2++;
			}
			else {
				d1 = 0;
				d2 = 0;
			}
			ans = max(ans, d1);
		}
		return ans;
	}
};
