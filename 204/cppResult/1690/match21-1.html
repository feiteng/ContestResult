class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        vector<int> idx;

        int s = -1;
        int res = 0;
        int cnt = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] < 0) {
                idx.push_back(i);
                ++cnt;
            } else if (nums[i] == 0) {
                cnt = 0;
                idx.clear();

                s = i;
            }
            if (cnt & 1) {
                res = max(res, i - idx[0]);
            } else {
                res = max(res, i - s);
            }
        }
        return res;
    }
};
