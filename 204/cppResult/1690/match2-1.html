class Solution {
public:
    int getMaxLen(vector<int>& nums) {

        int neg = -1;
        int maxL = 0;
        int last1 = -1;
        int lastneg = -1;
        for(int i =0;i<nums.size();i++){
            if(nums[i]==0){
                neg = -1;
                last1 = i;
                lastneg = i;
                continue;
            }else{
                if(nums[i]<0){
                    if(neg==-1){
                        lastneg = i;
                    }
                    if(neg!=1){
                        neg = 1;

                    }else{
                        neg = 0;
                    }
                }
                if(neg==0){
                    maxL = max(maxL,i-last1);
                }else{
                    maxL = max(maxL,i-lastneg);
                }
            }
        }
        return maxL;
    }
};
