class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int res = 0, n = nums.size();

        int val = 1, pos = -1, neg = -2;
        for (int i = 0; i < n; ++i) {
            if (nums[i] == 0) {
                val = 1;
                pos = i;
                neg = -2;
            } else {
                val *= (nums[i] > 0 ? 1 : -1);

                if (val > 0) {
                    res = max(res, i - pos);
                } else {
                    if (neg > -2) {
                        res = max(res, i - neg);
                    } else {
                        neg = i;
                    }
                }
            }
        }
        return res;
    }
};
