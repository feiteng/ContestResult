class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        vector<int> a;
        int ans = 0;
        int la = -1;
        for (int i = 0; i < n; i++) {
            if (nums[i] < 0) {
                a.push_back(i);
            }
            else if (nums[i] == 0) {
                a.clear();
                la = i;
                continue;
            }
            
            if ((a.size()%2) == 0) ans = max(ans,i-la);
            else ans = max(ans, i-a[0]);
        }
        return ans;
    }
};
