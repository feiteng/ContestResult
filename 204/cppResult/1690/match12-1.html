class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        unordered_map<int, int> mp;
        int res = 0;
        int odds = 0;
        mp[odds] = -1;
        for(int i=0;i<nums.size();i++) {
            if(nums[i] == 0) {
                mp.clear();
                odds = 0;
                mp[odds] = i;
                continue;
            }
            
            if(nums[i] < 0) odds = 1 - odds;
            if(mp.count(odds)) res = max(res, i - mp[odds]);
            else mp[odds] = i;
        }
        return res;
    }
};
