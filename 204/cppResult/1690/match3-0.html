class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int N = nums.size(), min_positive = -1, min_negative = -1, ans = 0;
        bool positive = true;
        for (int i = 0; i < N; ++i) {
            if (nums[i] == 0) {
                min_positive = i;
                min_negative = -1;
                positive = true;
            } else {
                if (nums[i] < 0) {
                    positive = !positive;
                }
                if (positive) {
                    ans = max(i - min_positive, ans);
                } else {
                    if (min_negative == -1) {
                        min_negative = i;
                    } else {
                        ans = max(i - min_negative, ans);
                    }
                }
            }
        }
        return ans;
    }
};
