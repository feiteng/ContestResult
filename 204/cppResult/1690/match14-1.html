class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int longest_even_neg = 0;
        int longest_odd_neg = -1; // no such exist
        int ans = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) {
                longest_even_neg = 0;
                longest_odd_neg = -1;
            } else if (nums[i] > 0) {
                longest_even_neg++;
                if (longest_odd_neg != -1) longest_odd_neg++;
            } else {
                int new_longest_even_neg = longest_odd_neg + 1;
                int new_longest_odd_neg = longest_even_neg + 1;
                longest_even_neg = new_longest_even_neg;
                longest_odd_neg = new_longest_odd_neg;
            }
            ans = max(ans, longest_even_neg);
        }
        return ans;
    }
};
