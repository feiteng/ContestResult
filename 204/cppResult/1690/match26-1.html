class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n=nums.size();
        int preneg=n+1,prepos=-1;
        int ans=0;
        int now=1;
        for(int i=0;i<n;i++)
        {
            if(nums[i]==0)
            {
                preneg=n+1;
                prepos=i;
                now=1;
                continue;
            }
            if(nums[i]<0)
            {
                now*=-1;
                if(preneg==n+1)
                    preneg=i;
            }
            if(now>0) ans=max(ans,i-prepos);
            else ans=max(ans,i-preneg);
        }
        return ans;
    }
};
