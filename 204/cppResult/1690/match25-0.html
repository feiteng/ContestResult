class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        int pre = 0;
        int ans = 0;
        for (int i = 0; i <= n; i++) {
            if (i == n || nums[i] == 0) {
                // [pre, i) will be a row
                
                bool even = true; 
                for (int t = pre; t < i; t++) {
                    if (nums[t] < 0) even = !even;
                }
                if (even) {
                    ans = max(ans, i - pre);
                } else {
                    for (int l = pre; l < i; l++) {
                        if (nums[l] < 0) {
                            // [l + 1, i)
                            ans = max(ans, i - l - 1);
                            break;
                        }
                    }
                    for (int r = i - 1; r >= pre; r--) {
                        if (nums[r] < 0) {
                            // [pre, r)
                            ans = max(ans, r - pre);
                            break;
                        }
                    }
                }
                
                
                
                pre = i + 1;
            }
        }
        return ans;
    }
};
