class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int ans=0;
        int f[2]={0,-1000000000};

        for (int i=0;i<nums.size();i++){
            if (nums[i]==0){
                f[0]=0;
                f[1]=-1e9;
            }
            else if (nums[i]>0){
                f[1]=f[1]+1;
                f[0]=max(f[0]+1,1);

                ans=max(ans,f[0]);
            }
            else{
                int newf1=max(f[0]+1,1);
                int newf0=f[1]+1;
                f[1]=newf1;f[0]=newf0;
            }
            ans=max(ans,f[0]);
        }
        return ans;
    }
};
