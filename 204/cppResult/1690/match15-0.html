class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int nega=-1, posi=-1, flag=1, ans = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) {
                nega = -1;
                posi = i;
                flag = 1;
            } else {
                flag *= (nums[i]>0?1:-1);
                if (flag == -1 && nega == -1) {
                    nega = i;
                }
                if (flag == 1) {
                    ans = max(ans, i-posi);
                } else {
                    if (nega != -1)
                        ans = max(ans, i-nega);
                }
            }
        }
        return ans;
    }
};
