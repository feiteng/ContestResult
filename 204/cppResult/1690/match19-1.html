class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();

        int ans = 0;
        int idx = -1;
        int cnt = 0;
        int zero = -1;
        for (int i = 0; i < n; ++i) {
            if (nums[i] == 0) {
                zero = i;
                idx = -1;
                cnt = 0;
                continue;

            }
            if (nums[i] < 0) {
                if (idx == -1) {
                    idx = i;
                }
                cnt++;
            }
            if (cnt & 1) {
                if (idx != -1) {
                    ans = max(ans, i - idx);
                }
            } else {
                ans = max(ans, i - zero);
            }
        }
        return ans;
    }
};
