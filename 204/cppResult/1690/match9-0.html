class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int a = -1;
        int b = -1;
        int nc = 0;
        int prev = -1;
        
        int n = nums.size();
        int maxi = 0;
        
        for(int i=0;i<=n;i++){
            if(i == n || nums[i]==0){
                if(nc%2==0){
                    maxi = max(maxi,i-prev - 1);
                }
                else{
                    maxi  = max(maxi,max(b-prev-1,i-a-1));
                }
                prev = i;
                a = -1;
                b = -1;
                nc = 0;
            }
            else{
                if(nums[i]<0){
                    if(a == -1){
                        a = i;
                    }
                    b = i;
                    nc++;
                }
            }
        }
        return maxi;
    }
};
