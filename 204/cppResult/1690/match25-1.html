class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        vector<int> a;
        a=nums;
        a.push_back(0);
        int n=a.size();
        int ans=0;
        for(int i=0;i<n;i++){
            if(a[i]!=0){
                vector<int> b;
                int p=0;
                while(a[i]!=0){
                    b.push_back(a[i]);
                    if(a[i]<0) p^=1;
                    i++;
                }
                int len=b.size();
                if(p){
                    int p1,p2,c=b.size();

                    for(int j=0;j<b.size();j++){
                        if(b[j]<0){
                            p1=c-j-1;
                            break;
                        }
                    }
                    for(int j=b.size()-1;j>=0;j--){
                        if(b[j]<0){
                            p2=j;
                            break;
                        }
                    }
                    len=max(p1,p2);
                }
                ans=max(ans,len);
            }
        }
        return ans;
    }
};
