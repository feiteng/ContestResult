class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int res = 0;
        int z = -1, o = nums.size() + 1;
        int k = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) {
                z = i;
                o = nums.size() + 1;
                k = 0;
            } else {
                if (nums[i] < 0) {
                    ++k;
                }
                k = k % 2;
                if (k == 0) {
                    res = max(res, i - z);
                } else {
                    if (o > nums.size()) {
                        o = i;
                    } else {
                        res = max(res, i - o);
                    }
                }
            }
        }
        
        return res;
    }
};
