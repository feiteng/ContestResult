class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int positive = -1;
        int negative = nums.size();
        int result = 0;
        int product = 1;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) {
                positive = i;
                negative = nums.size();
                product = 1;
            } else {

                if (nums[i] < 0) {
                    product = -product;
                } 
                if (product > 0) {
                    result = max(result, i - positive);
                } else if (negative < nums.size()) {
                    result = max(result, i - negative);
                } else {
                    negative = i;
                }
            }
        }
        return result;
    }
};
