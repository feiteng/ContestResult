class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n+1, 0);
        vector<int> g(n+1, 0);

        int res = 0;
        for (int i=0; i<n; i++) {
            if (nums[i] == 0) {
                f[i+1] = 0;
                g[i+1] = 0;
            } else if(nums[i]>0) {
                if (f[i] == 0) f[i+1] = 1;
                else f[i+1] = f[i] + 1;
                if (g[i] == 0) g[i+1] = 0;
                else g[i+1] = g[i] + 1;
            } else {
                if (f[i] == 0) g[i+1] = 1;
                else g[i+1] = f[i] + 1;
                if (g[i] == 0) f[i+1] = 0;
                else f[i+1] = g[i] + 1;
            }
            res = max(f[i+1], res);
        }
        return res;

    }
};
