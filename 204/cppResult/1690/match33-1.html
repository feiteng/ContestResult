class Solution {
public:
    int getMaxLen(vector<int>& nums) {

        int pos0 = -1;
        int pos1 = -1;
        int numn = 0;
        int maxlen = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) {
                pos0 = i;
                pos1 = -1;
                numn = 0;
            }
            if (nums[i] < 0) {
                if (pos1 == -1) {
                    pos1 = i;
                }
                numn++;
            }
            if (nums[i] != 0) {

                if (numn % 2 == 0) {
                    maxlen = max(maxlen, i - pos0);
                } else if (pos1 > -1) {
                    maxlen = max(maxlen, i - pos1);
                }
            }
        }
        return maxlen;
    }
};
