class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for(int i = 0; i + k * m <= arr.size(); i++) {
            bool all_match = true;
            for(int j = 1; j < k; j++) {
                bool match = true;
                for(int t = 0; t < m; t++) {
                    if (arr[i + t] != arr[i + j * m + t]) {
                        match = false;
                        break;
                    }
                }
                if (!match) {
                    all_match = false;
                    break;
                }
            }
            if (all_match)
                return true;
        }
        return false;
    }
};
