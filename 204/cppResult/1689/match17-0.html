class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i < n; ++i) {
            bool ok = true;
            for (int j = 0; j < m; ++j) {
                for (int t = 1; t < k; ++t) {
                    if (i+j+m*t >= n || arr[i+j] != arr[i+j+m*t]) {
                        ok = false;
                        break;
                    }
                }
                if (!ok) break;
            }
            if (ok) return true;
        }
        return false;
    }
};
