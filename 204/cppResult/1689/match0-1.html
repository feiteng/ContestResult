class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i < n; i++) {
            bool works = true;
            for (int j = i; j < i + m * k; j++) {
                if (j >= n || arr[j] != arr[(j - i) % m + i]) {
                    works = false;
                    break;
                }
            }
            if (works) return true;
        }
        return false;
    }
};
