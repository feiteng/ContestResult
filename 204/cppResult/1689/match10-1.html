class Solution {
public:
    bool containsPattern(vector<int>& a, int m, int k) {
        int i, j, n = a.size();
        for (i = 0; i <= n - m * k; ++i){
            bool yes = true;
            for (j = i; j < i + m * k; ++j){
                if (a[j] != a[i + (j - i) % m]) {
                    yes = false;
                    break;
                }
            }
            if (yes) return true;
        }
        return false;
    }
};
