class Solution {
public:
    bool containsPattern(vector<int>& a, int m, int num) {
        
        int n=a.size();
        for(int i=m-1;i<=n-m-1;i++)
        {
            int count=1;
            //cout<<i<<"i";
            for(int j=i+1;j<n;j+=m)
            {
               // cout<<j<<"j"<<endl;;
                bool match=true;
                for(int k=j;k<j+m;k++)
                {
                    //cout<<k<<" "<<k-m<<endl;
                    if(a[k]!=a[k-m])
                    {
                        match=false;
                        break;
                    }
                }
                //cout<<endl;
                if(match)
                    count++;
                else
                    break;
                
            }
            if(count>=num)
                return true;
                
        }
        return false;
        
    }
};
