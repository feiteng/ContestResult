class Solution {
 public:
  bool containsPattern(vector<int>& arr, int m, int k) {
    int n = arr.size();
    for (int i = 0; m * k + i <= n; i++) {
      bool valid = true;
      for (int j = 0; j < k; j++) {
        for (int c = 0; c < m; c++) {
          if (arr[i + c] != arr[i + c + j * m]) {
            valid = false;
          }
        }
      }
      if (valid) return true;
    }
    return false;
  }
};
