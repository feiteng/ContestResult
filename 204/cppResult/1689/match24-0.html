class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for (int i = 0;i + m * k <= arr.size();i++) {
            bool find = true;
            for (int j = 0;j < m * k;j++) {
                if (arr[i + j] != arr[i + j % m]) {
                    find = false;
                    break;
                }
            }
            if (find) {
                return true;
            }
        }
        return false;
    }
};
