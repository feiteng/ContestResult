class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int  az=arr.size();
        for (int i=0;i+m*k<=az;++i) {
            bool ok=true;
            for (int j=0;j<m&&ok;++j) {
                for (int l=1;l<k&&ok;++l) {
                    if (arr[i+l*m+j]!=arr[i+j]) {
                        ok=false;
                        break;
                    }
                }
            }
            if (ok) return true;
        }
        return false;
            
    }
};
