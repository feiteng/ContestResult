class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int N = arr.size();
        for (int L = 0; L + m * k <= N; ++L) {
            bool same = true;
            for (int i = 0; i < m && same; ++i) {
                for (int j = 1; j < k && same; ++j) {
                    if (arr[L + i] != arr[L + m * j + i]) {
                        same = false;
                    }
                }
            }
            if (same) {
                return true;
            }
        }
        return false;
    }
};
