class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for (int i = 0; i + m * k <= arr.size(); i++) {
            bool is_valid = true;
            for (int j = 1; j < k; j++) {
                if (!is_valid) { break; }
                for (int l = 0; l < m; l++) {
                    if (arr[i + l] != arr[i + j * m + l]) {
                        is_valid = false;
                        break;
                    }
                }
            }
            if (is_valid) { return true; }
        }
        return false;
    }
};
