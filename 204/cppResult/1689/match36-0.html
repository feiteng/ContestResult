class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for(int i=0; i+m*k<=n; i++) {
            int f = 1;
            for(int j=1; j<k; j++) {
                for(int x=0; x<m; x++) {
                    if(arr[i+j*m+x]!=arr[i+x]) {
                        f = 0;
                        break;
                    }
                }
                if(f==0) break;
            }
            if(f) return true;
        }
        return false;
    }
};
