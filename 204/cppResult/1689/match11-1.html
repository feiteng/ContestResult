class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i <= n - m * k; ++i) {
            bool flag = true;
            for (int j = 1; j < k; ++j) {
                for (int t = 0; t < m; ++t) {
                    if (arr[i + t + j * m] != arr[i + t]) {
                        flag = false;
                        break;
                    }
                }
                if (!flag) break;
            }
            if (flag) return true;
        }
        return false;
    }
};
