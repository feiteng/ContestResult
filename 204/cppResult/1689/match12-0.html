class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i <= n - m*k; i++) {
            bool okay = true;
            for (int j = 0; j < m; j++) {
                for (int l = 1; l < k; l++) {
                    if (arr[i + l * m + j] != arr[i + j]) {
                        okay = false;
                        break;
                    }
                }
                if (!okay)
                    break;
            }
            if (okay)
                return true;
        }
        return false;
    }
};
