class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for(int i = 0; i + m*k <= n; i++)  {
            bool flag = true;
            for(int z = 1; z < k && flag; z++)  {
                for(int j = 0; j < m && flag; j++)  {
                    if(arr[i + m*z + j] != arr[i + j])  flag = false;
                }
            }
            if(flag)  return true;
        }
        return false;
    }
};
