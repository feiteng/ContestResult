class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i < n; i++) {
            bool match = true;
            for (int j = 1; j < k && match; j++) {
                for (int s = 0; s < m && match; s++) {
                    if (i + j * m + s >= n || arr[i + s] != arr[i + j * m + s]) match = false;
                }
            }
            if (match) return true;
        }
        return false;
    }
};
