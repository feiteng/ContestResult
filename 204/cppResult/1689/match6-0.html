class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for (int i = 0; i < n; ++i) {
            if (i + m * k > n) break;
            bool flag = true;
            for (int j = 0; j < k; ++j) {
                for (int idx = 0; idx < m; ++idx) {
                    if (arr[i + j * m + idx] != arr[i + idx]) {
                        flag = false;
                    }
                }
            }
            if (flag) return true;
        }
        return false;
    }
};
