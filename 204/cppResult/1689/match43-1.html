class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for(int i=0;i<n;i++){
            int tn = i+k*m;
            if(tn>n) return false;

            bool f = true;
            for(int j=i;j<tn;j++){
                if(arr[j] != arr[(j-i)%m+i]){
                    f = false;
                    break;
                }
            }
            if(f) return true;
        }
        return false;
    }
};
