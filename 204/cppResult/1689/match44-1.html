class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for (int i = 0; i+m*k-1 < arr.size(); ++i) {
            bool flag = true;
            for (int j = 0; j < m; ++j) {
                for (int l = 0; l < k; ++l)

                    if (arr[i+j] != arr[i+j+l*m]) {
                        flag = false;
                        break;
                    }
            }
            if (flag)
                return true;
        }
        
        return false;
    }
};
