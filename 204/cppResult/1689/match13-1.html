class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for(int i =0;i<n&&i+k*m-1<n;i++){
            bool flag = true;
            for(int j = 0;j<m;j++){
                for(int x = 1;x<k;x++){
                    if(arr[i+j+x*m] != arr[i+j+(x-1)*m]){
                        flag = false;
                        break;
                    }
                }
                if(!flag)
                    break;
            }
            if(flag)
                return true;
        }
        return false;
    }
};
