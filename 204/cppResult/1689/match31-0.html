class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int size = arr.size();
        for (int i = 0; i < size; i++) {
            bool correct = true;
            for (int j = 0; j < m; j++) {
                int expected = arr[i + j];
                if(i + j >= size) {
                    correct = false;
                    break;
                }
                bool flag = true;
                for (int p = 1; p < k; p++) {
                    int idx = i + j + p * m;
                    //cout<<idx;
                    if (idx >= size) {flag = false;break;}
                    
                    if (arr[idx] != expected) {flag = false;break;}
                }
                if(!flag) {correct = false; break;}
            }
            if(correct) return true;
        }
        return false;
    }
};
