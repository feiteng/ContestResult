class Solution {
public:
    bool containsPattern(vector<int>& a, int m, int k) {
        int n = a.size();
        if (n < k*m)
            return false;
        
        for (int i = 0; i + k*m <= n; ++i) {
            bool ok = true;
            // printf("i:%d\n", i);
            for (int j = 1; j < k && ok; ++j) {
                int st = i + j * m;
                // printf("st:%d\n", st);
                for (int t = 0; t < m; ++t)

                    if (a[i+t] != a[st+t]) {
                        ok = false;
                        // printf("t:%d\n", t);
                        break;
                    }
            }
            if (ok)
                return true;
        }
        return false;
    }
};
