class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int len = m;

            for (int start = 0; start + len*k <= arr.size(); start ++) {
                int cnt = 1;
                for (int i = 1; i < k; i++) {
                    bool match = true;
                    for (int j = 0; j < len; j++) {
                        if (arr[i*len + j + start] != arr[start+j]) {
                            match = false;
                            break;
                        }
                    }
                    if (match)
                        cnt ++;
                    else
                        break;
                }
                if (cnt == k)
                    return true;
            }
        return false;
    }
};
