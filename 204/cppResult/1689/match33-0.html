class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        // Pick a subarray of size m*k and check if the pattern is repeated k times or not
        
        for(int i=0; i<n; i++){
            int start = i, end = i + m*k - 1;
            if(end >= n)
                break;
            bool isPossible = true;
            for(int j=0; j<=m*k-1; j++){
                if(arr[i+j] != arr[i+j%m]){
                    isPossible = false;
                    break;
                }
            }
            if(isPossible)
                return true;
        }
        return false;
    }
};
