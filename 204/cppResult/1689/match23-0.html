class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int sz = arr.size();
        if(m*k>sz)
            return false;
        for(int i=0; i<sz-m*k+1; i++)
        {
            int j;
            for(j=i; j<i+m; j++)
            {
                int l;
                for(l=j; l<j+m*k; l=l+m)
                    if(arr[l]!=arr[j])
                        break;
                if(l!=j+m*k)
                    break;
            }
            if(j==i+m)
                return true;
        }       
        return false;
    }
};
