class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for(int i=0;i+m*k<=arr.size();i++) {
            // m*k

            bool ok=true;
            for(int j=0;(j<m && ok);j++) {
                // arr[i+j], arr[i+j+m], arr[i+j+2*m],,arr[i+j+(k-1)*m]
                for(int kk=1;kk<k;kk++) {
                    if (arr[i+j]!=arr[i+j+kk*m]) {
                        ok=false;
                        break;
                    }
                }
            }
            if (ok) return true;
        }
        return false;
        
    }
};
