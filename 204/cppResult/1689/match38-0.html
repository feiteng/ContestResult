class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        if(k == 1 && m <= n) return true;
        
        for(int i = 0; i < n; ++i){
            bool valid = true;
            if(i+m*k > n) break;
            for(int j = i+m; j < i+m*k; j = j + m){
                for(int k = j; k < j + m; ++k){
                    if(arr[k] != arr[i+k-j]){
                        valid = false;
                        break;
                    }
                }
                if(!valid) break;
            }
            if(valid) return true;
        }
        
        return false;
    }
};
