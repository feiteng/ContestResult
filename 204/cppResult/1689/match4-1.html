class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        int n = arr.size();
        for(int i=0;i<n-m*k+1;i++){
            bool f = true;
            for(int c=1;c<k;c++){
                for(int s=0;s<m;s++){
                    if(arr[i+s] != arr[i+c*m+s]){
                        f = false;
                        break;
                    }
                }
            }
            if(f) return true;
        }
        return false;
    }
};
