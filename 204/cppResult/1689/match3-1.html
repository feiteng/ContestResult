class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for (int i = 0; i + k * m - 1 < arr.size(); ++i) {
            bool pattern = true;
            for (int j = i; j < i + m; ++j) {
                if (!pattern) {
                    break;
                }
                for (int h = 1; h < k; ++h) {
                    if (arr[j] != arr[j + h * m]) {
                        pattern = false;
                        break;
                    }
                }
            }
            if (pattern) {
                return true;
            }
        }
        return false;
    }
};
