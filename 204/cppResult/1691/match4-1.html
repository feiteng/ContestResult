class Solution
{
    void print(vector<vector<int>>& grid)
    {
        for (auto& row : grid) {
            for (int v : row)
                cout << v;
            cout << endl;
        }
    }

    void fill(vector<vector<int>>& grid, int r, int c)
    {
        int m = grid.size();
        int n = grid[0].size();
        if (r < 0 || r >= m) return;
        if (c < 0 || c >= n) return;
        if (grid[r][c] != 1) return;
        grid[r][c] = 2;
        fill(grid, r - 1, c);
        fill(grid, r + 1, c);
        fill(grid, r, c - 1);
        fill(grid, r, c + 1);
    }
    int isConnected(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        int components = 0;

        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++) {
                if (grid[i][j] != 1) continue;
                components++;
                if (components > 1) break;
                fill(grid, i, j);
            }
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                grid[i][j] = grid[i][j] > 0 ? 1 : 0;
        return components == 1;
    }
public:

    int minDays(vector<vector<int>>& grid)
    {
        int m = grid.size();
        int n = grid[0].size();
        if (!isConnected(grid)) return 0;
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) continue;
                grid[i][j] = 0;
                if (!isConnected(grid)) return 1;
                grid[i][j] = 1;
            }
        return 2;
    }
};
