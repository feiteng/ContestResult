class Solution {
    vector<vector<bool>> visited;
    vector<int> dx;
    vector<int> dy;
public:

    void dfs(vector<vector<int>>& grid, int x, int y) {
        visited[x][y]=true;
        for (int i=0; i<4; ++i) {
            int xx=x+dx[i];
            int yy=y+dy[i];
            if (xx>=0&&xx<grid.size()&&yy>=0&&yy<grid[0].size()&&!visited[xx][yy]&&grid[xx][yy]==1) {
                dfs(grid,xx,yy);
            }
        }
    }
    
    int cnt(vector<vector<int>>& grid) {
        int num=0;
        visited=vector<vector<bool>>(grid.size(), vector<bool>(grid[0].size(), false));

        int n=0;
        for (int i=0; i<grid.size(); ++i) {
            for (int j=0; j<grid[0].size(); ++j) {
                if (!visited[i][j]&&grid[i][j]==1) {
                    num++;
                    dfs(grid, i, j);
                    if (num>1) {
                        return num;
                    }
                }
            }
        }
        return num;
    }
    int minDays(vector<vector<int>>& grid) {
        dx=vector<int>{0, -1, 0, 1};
        dy=vector<int>{-1, 0, 1, 0};
        int c=cnt(grid);
        if (c!=1) {
            return 0;
        }

        for (int i=0; i<grid.size(); ++i) {
            for (int j=0; j<grid[0].size(); ++j) {
                if (grid[i][j]==1) {
                    grid[i][j]=0;
                    if (cnt(grid)!=1) {
                        return 1;
                    }
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
};
