class Solution {
public:
    vector< vector<bool> > done;
    int n,m;
    void dfs(int u,int v,vector<vector<int>>& grid){
        if( u < 0 or v < 0 or u >=n or v >= m or done[u][v] or !grid[u][v])return;
        done[u][v] = true;
        dfs(u+1,v,grid);
        dfs(u-1,v,grid);
        dfs(u,v+1,grid);
        dfs(u,v-1,grid);
    }
    
    int comp(vector<vector<int>>& grid){
        n = grid.size();
        m = grid[0].size();
        done = vector< vector<bool> > (n, vector<bool> (m,false));
        int cmp=0;
        for(int i=0;i<grid.size();++i)
            for(int j=0;j<grid[i].size();++j)
            if( grid[i][j] and !done[i][j]){
                dfs(i,j,grid);
                ++cmp;
            }
        return cmp;
        
    }
    
    int minDays(vector<vector<int>>& grid) {
        int c = comp(grid);
        if(c!=1) return 0;
        for(int i=0;i<grid.size();++i)
            for(int j=0;j<grid[i].size();++j)
            if( grid[i][j]){
                grid[i][j] = 0;
                c = comp(grid);
                if(c!=1) return 1;
                grid[i][j] = 1;
            }
        return 2;
    }
};
