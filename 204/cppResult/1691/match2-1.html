class Solution {
public:
	int dirs[4][2] = { {1,0},{-1,0},{0,1},{0,-1} };
	int len1;
	int len2;

	int minDays(vector<vector<int>>& grid) {
		len1 = grid.size();
		len2 = grid[0].size();
		if (depart(grid)) return 0;
		for (int i = 0; i < len1; ++i) {
			for (int j = 0; j < len2; ++j) {
				if (grid[i][j] == 1) {
					auto tmp = grid;
					tmp[i][j] = 0;
					if (depart(tmp)) return 1;
				}
			}
		}
		return 2;
	}
	bool depart(const vector<vector<int>>& G) {
		auto visit = G;

		int flag = 0;
		for (int i = 0; i < len1; ++i) {
			for (int j = 0; j < len2; ++j) {
				if (visit[i][j] == 1) {
					if (flag == 0) {
						dfs(i, j, visit);
						++flag;
					}

					else {
						return true;
					}
				}
			}
		}
		return false;
	}
	void dfs(int i,int j,vector<vector<int>>& V) {
		V[i][j] = 0;
		for (auto P : dirs) {
			int ni = i + P[0];
			int nj = j + P[1];
			if (ni >= 0 && ni < len1 && nj >= 0 && nj < len2 && V[ni][nj] == 1) dfs(ni, nj, V);
		}
	}
};
