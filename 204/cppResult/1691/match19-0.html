int dir[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
class Solution {
private:
    vector<vector<int>>c;
    void dfs(vector<vector<int>>& a, int x, int y, int cc) {
        c[x][y] = cc;
        for(int i=0;i<4;i++){
            int nx = x + dir[i][0];
            int ny = y + dir[i][1];
            if(nx<0||nx>=a.size()||ny<0||ny>=a[0].size()||c[nx][ny]||a[nx][ny]!=1)continue;
            dfs(a,nx,ny,cc);
        }
    }
    int cal(vector<vector<int>>& a) {
        c = vector<vector<int>>(a.size(), vector<int>(a[0].size(), 0));
        int cur = 0;
        for(int i=0;i<a.size();i++)
            for(int j=0;j<a[i].size();j++)
        {
            if(a[i][j]==1&&!c[i][j]){
                cur++;
                if(cur>=2)return 2;
                dfs(a, i,j,cur);
            }
        }
        return cur;
    }
public:
    int minDays(vector<vector<int>>& a) {
        if(cal(a)>=2)return 0;
        for(int i=0;i<a.size();i++)
            for(int j=0;j<a[i].size();j++){
                if(a[i][j]==1){
                    a[i][j] = 0;
                    if(cal(a)>=2)return 1;
                    a[i][j] = 1;
                }
            }
        return 2;
    }
};
