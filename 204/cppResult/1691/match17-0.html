class Solution {
    void dfs(vector<vector<int>>& grid, int i, int j, int n, int m){
        if(i<0||j<0||i>=n||j>=m||grid[i][j]==0) return;
        grid[i][j]=0;
        dfs(grid,i+1,j,n,m);
        dfs(grid,i-1,j,n,m);
        dfs(grid,i,j+1,n,m);
        dfs(grid,i,j-1,n,m);
    }
    bool con(vector<vector<int>> grid, int n, int m) {
        int flag=0;
        for(int i=0; i<n;i++) {
            for(int j=0;j<m;j++) {
                if(grid[i][j]) {
                    if(flag)return false;
                    dfs(grid,i,j,n,m);
                    flag=1;
                }
            }
        }
        return true;
    }
//     bool con(vector<vector<int>> grid, int i, int j, int n, int m) {

public:
    int minDays(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        if(!con(grid,n,m)) return 0;
        for(int i=0;i<n;i++) {
            for(int j=0;j<m;j++) {
                if(grid[i][j]==1) {
                    grid[i][j]=0;
                    if(!con(grid,n,m)) return 1;
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
};
