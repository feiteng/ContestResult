int dX[] = {0,1,-1,0};
int dY[] = {-1,0,0,1};
class Solution {
    int currentCase = 3;
    void dfs(int x, int y, vector<vector<int>> &grid) {
        if (grid[x][y]==0 || grid[x][y] == currentCase) {
            return;
        }

        grid[x][y]=currentCase;
        for (int d = 0; d < 4; d++){
            int nx = x+dX[d];
            int ny = y+dY[d];
            if (0 <= nx && nx < grid.size() && 0 <= ny && ny < grid[0].size()) {
                dfs(nx,ny,grid);
            }
        }
    }

    bool disconnected(vector<vector<int>> grid) {
        int cnt = 0;
        for(int i = 0; i < grid.size(); i++){
            for (int j = 0; j < grid[0].size(); j++) {
                if (grid[i][j]==1) {
                    cnt++;
                    if (cnt == 2) {
                        return true;
                    }
                    currentCase++;
                    dfs(i,j,grid);
                }
            }
        }
        return cnt!=1;
    }
    
public:
    int minDays(vector<vector<int>>& grid) {
        if (disconnected(grid)) {
            return 0;
        }

        for(int i = 0; i < grid.size(); i++){
            for (int j = 0; j < grid[0].size(); j++) {
                if (grid[i][j]) {
                    grid[i][j]=0;
                    if (disconnected(grid)) {
                        return 1;
                    }
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
};
