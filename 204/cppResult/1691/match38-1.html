class Solution
{
public:
    vector<vector<int>> g;
    int R, C;
    int ds[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

    int count(){

        int ans = 0;
        for(int i=0; i<R; i++){
            for(int j=0; j<C; j++){
                if(g[i][j] > 0) ans++;
            }
        }
        return ans;
    }

    void set1(){
        for(int i=0; i<R; i++){
            for(int j=0; j<C; j++){
                if(g[i][j] > 0){
                    g[i][j] = 1;
                }
            }
        }
    }

    bool isone(){
        set1();
        int color = 1;
        for (int i = 0; i < R; i++)
        {
            for (int j = 0; j < C; j++)
            {
                if (g[i][j] == 1)
                {
                    dfs(i, j, ++color);
                }
            }
        }
        return color == 2;
        

    }

    void dfs(int r, int c, int color)
    {
        g[r][c] = color;
        for (int d = 0; d < 4; d++)
        {
            int rr = r + ds[d][0], cc = c + ds[d][1];
            if (rr >= 0 && rr < R && cc >= 0 && cc < C && g[rr][cc] == 1)
            {
                dfs(rr, cc, color);
            }
        }
    }

    int minDays(vector<vector<int>> &grid)
    {
        g = grid;
        R = grid.size();
        C = grid[0].size();
        int cnt = count();

        if(!isone()) return 0;

        if(cnt <= 2) return cnt;

        for (int i = 0; i < R; i++)
        {
            for (int j = 0; j < C; j++)
            {
                if (g[i][j] > 0)
                {
                    g[i][j] = 0;
                    if(!isone()) return 1;
                    g[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
