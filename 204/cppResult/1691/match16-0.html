class Solution {
public:
    int minDays(vector<vector<int>>& g) {
        if (!connected(g)) return 0;
        for (int i = 0; i < g.size(); i++) {
            for (int j = 0; j < g[0].size(); j++) {
                if (g[i][j] == 1) {
                    g[i][j] = 0;
                    if (!connected(g)) {
                        return 1;
                    }
                    g[i][j] = 1;
                }
            }
        }
        return 2;
    }
    bool connected(vector<vector<int>> g) {
        bool first = true;
        for (int i = 0; i < g.size(); i++) {
            for (int j = 0; j < g[0].size(); j++) {
                if (g[i][j] == 1) {
                    if (!first) {
                        return false;
                    }
                    first = false;
                    flood(g, i, j);
                }
            }
        }
        return !first;
    }
    void flood(vector<vector<int>> &g, int i, int j) {
        if (i < 0 || i >= g.size()) return;
        if (j < 0 || j >= g[i].size()) return;
        if (g[i][j] != 1) return;
        g[i][j] = 2;
        flood(g, i - 1, j);
        flood(g, i + 1, j);
        flood(g, i, j - 1);
        flood(g, i, j + 1);
    }
};
