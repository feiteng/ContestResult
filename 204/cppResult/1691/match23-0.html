class Solution {
public:
    vector<vector<int>> mat;
    int vis[40][40],n,m;
    bool safe(int i, int j) { return i >= 0 && i < n && j >= 0 && j < m&&mat[i][j]==1; }
    void dfs(int i,int j){
        const int fx[4] = {0, +1, 0, -1};
        const int fy[4] = {-1, 0, +1, 0};
        vis[i][j]=1;
        for(int k=0;k<4;k++){
            int ni=i+fx[k],nj=j+fy[k];
            if(safe(ni,nj)&&!vis[ni][nj]){
                dfs(ni,nj);
            }
        }
    }
    int minDays(vector<vector<int>>& grid) {
        mat=grid;
        n=grid.size(),m=grid[0].size();
        int comp=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(!vis[i][j]&&mat[i][j]==1)
                {
                    dfs(i,j);
                    comp++;
                }
            }
        }
        if(comp==1){
            for(int i=0;i<n;i++){
                for(int j=0;j<m;j++){
                    if(mat[i][j]==1){
                        mat[i][j]=0;
                        int com=0;
                        memset(vis,0,sizeof vis);
                        for(int k=0;k<n;k++){
                            for(int l=0;l<m;l++){
                              if(safe(k,l)&&!vis[k][l]){
                                    dfs(k,l);
                                  com++;
                                }  
                            }
                        }
                        if(com>1) return 1;
                        mat[i][j]=1;
                    }
                }
            }
            return 2;
        }
        else
        {
            return 0;
        }
    }
};
