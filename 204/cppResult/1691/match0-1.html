class Solution {
public:
    int m, n;
    void dfs(vector<vector<int>>& grid, int r, int c){
        if(r < 0 || c < 0 || r >= m || c >= n || grid[r][c] == 0) return;
        grid[r][c] = 0;
        dfs(grid, r - 1, c);
        dfs(grid, r + 1, c);
        dfs(grid, r, c - 1);
        dfs(grid, r, c + 1);
    }
    
    int cnt(vector<vector<int>> grid){

        int rnt = 0;
        for(int i = 0; i < m; i ++)
            for(int j = 0; j < n; j ++)
                if(grid[i][j]){
                    dfs(grid, i, j);
                    rnt ++;
                }
        return rnt;
    }
    
    int minDays(vector<vector<int>>& grid) {
        m = grid.size(), n = grid[0].size();

        int cur = cnt(grid);
        if(cur != 1) return 0;
        for(int i = 0; i < m; i ++)
            for(int j = 0; j < n; j ++){
                if(grid[i][j] == 1){
                    grid[i][j] = 0;
                    if(cnt(grid) != 1)
                         return 1;
                    grid[i][j] = 1;
                }
            }
        return 2;
    }
};
