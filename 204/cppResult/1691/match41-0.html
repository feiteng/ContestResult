class Solution {
public:
    int n, m;
    int a[51][51];
    int tmr;
    int was[51][51];
    
    bool check(int x, int y) {
        return 1 <= x && x <= n && 1 <= y && y <= m;
    }
    void dfs(int x, int y) {
        if (was[x][y] == tmr || a[x][y] == 0)  return;
        was[x][y] = tmr;
        for (int i = -1; i <= 1; i += 2) {
            if (check(x + i, y)) dfs(x + i, y);
            if (check(x, y + i)) dfs(x, y + i);
        }
    }
    int count() {
        ++tmr;
        int cnt = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (a[i][j] && was[i][j] != tmr) {
                    dfs(i, j);
                    ++cnt;
                }
            }
        }
        return cnt;
    }
    int minDays(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                a[i][j] = grid[i - 1][j - 1];
            }
        }
        if (count() > 1) return 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (a[i][j])  {
                    a[i][j] = 0;
                    if (count() > 1) return 1;
                    a[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
