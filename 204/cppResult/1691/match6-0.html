class Solution {
public:
    int dx[4] = {0, 0, -1, 1}, dy[4] = {-1, 1, 0, 0};
    void dfs(vector<vector<int>>& grid, int n, int m, vector<vector<int>>& vis, int x, int y) {
        vis[x][y] = 1;
        for (int k = 0; k < 4; k++) {
            int xt = x + dx[k], yt = y + dy[k];
            if (xt >= 0 && yt >= 0 && xt < n && yt < m && !vis[xt][yt] && grid[xt][yt]==1) {
                dfs(grid, n, m, vis, xt, yt);
            }
        }   
    }
    bool check(vector<vector<int>>& grid, int n, int m) {
        vector<vector<int>> vis = grid;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++) 
                vis[i][j] = 0;
        bool found = false;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (!vis[i][j] && grid[i][j]==1) {
                    printf("%d %d\n", i, j);
                    if (found)
                        return true;
                    found = true;
                    dfs(grid, n, m, vis, i, j);
                }
            }
        }
        return false;
    }
    int minDays(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        puts("");
        if (check(grid, n, m))
            return 0;
        for (int i = 0; i < n; i ++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (check(grid, n, m))
                        return 1;
                    grid[i][j] = 1;

                }
            }
        }
        return 2;
    }
};
