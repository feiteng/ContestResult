class Solution {
public:
    int minDays(vector<vector<int>>& grid)
    {
        if (cnt(grid) != 1)
            return 0;
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[i].size(); ++j)
            {
                if (grid[i][j])
                {
                    grid[i][j] = 0;
                    if (cnt(grid) != 1)
                        return 1;
                    grid[i][j] = 1;
                }
            }
        }
        
        return 2;
    }
    
private:

    int cnt(vector<vector<int>> grid)
    {
        int res = 0;
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[i].size(); ++j)
            {
                if (grid[i][j] == 1)
                {
                    if (++res > 1)
                        return res;
                    impl(grid, i, j);
                }
            }
        }
        
        return res;
    }
    
    void impl(vector<vector<int>>& grid, int i, int j)
    {
        if (i < 0 || i >= grid.size() || j < 0 || j >= grid[i].size() || grid[i][j] != 1)
            return;
        grid[i][j] = 2;
        impl(grid, i - 1, j);
        impl(grid, i + 1, j);
        impl(grid, i, j - 1);
        impl(grid, i, j + 1);
    }
};
