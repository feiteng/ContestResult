class Solution {
public:
    void dfs(vector<vector<int>> &grid,int n,int m,int i,int j){
        if(i<0||i>=n||j<0||j>=m||grid[i][j]==0){
            return;
        }
        grid[i][j]=0;
        dfs(grid,n,m,i+1,j);
        dfs(grid,n,m,i-1,j);
        dfs(grid,n,m,i,j+1);
        dfs(grid,n,m,i,j-1);
    }
    bool isconnected(vector<vector<int>> grid,int n,int m){

        int c=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]){

                    c++;
                    dfs(grid,n,m,i,j);
                }
            }
        }
        return c>1||c==0;
    }
    int minDays(vector<vector<int>>& grid) {
        int n=grid.size(),m=grid[0].size();
        if(isconnected(grid,n,m))return 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]){
                    grid[i][j]=0;
                    if(isconnected(grid,n,m))return 1;
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
};
