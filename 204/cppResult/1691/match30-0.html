class Solution {
public:
    int m, n;
    vector<vector<int> > grid;
    vector<vector<int> > colors;
     void dfs(int i, int j, int c)
    {
        colors[i][j] = c;
        if(i - 1 >= 0 && grid[i-1][j] == 1 && colors[i-1][j] == 0)
        {
            dfs(i-1, j, c);
        }
        if(i + 1 < m && grid[i+1][j] == 1 && colors[i+1][j] == 0)
        {
            dfs(i+1, j, c);
        }
        if(j - 1 >= 0 && grid[i][j-1] == 1 &&colors[i][j-1] == 0)
        {
            dfs(i, j-1, c);
        }
        if(j + 1 < n && grid[i][j+1] == 1 &&colors[i][j+1] == 0)
        {
            dfs(i, j+1, c);
        }
        return;
    }
    int cl()
    {
        colors = vector<vector<int> > (m, vector<int> (n, 0));
        int nc = 1;
        for(int i = 0; i < m; ++i)
        {
            for(int j = 0; j < n; ++j)
            {
                if(grid[i][j] == 1 && colors[i][j] == 0)
                {
                    dfs(i, j, nc);
                    ++nc;
                }
            }
        }
        return nc - 1;
    }
   
    int minDays(vector<vector<int>>& g) {
        grid = g;
        m = grid.size();
        n = grid[0].size();
        colors = vector<vector<int> > (m, vector<int> (n, 0));
        int nc = 1;
        int num = 0;
        int x;
        int y;
        for(int i = 0; i < m; ++i)
        {
            for(int j = 0; j < n; ++j)
            {
                if(grid[i][j] == 1)
                {
                    ++num;
                    x = i;
                    y = j;
                }
            }
        }
        for(int i = 0; i < m; ++i)
        {
            for(int j = 0; j < n; ++j)
            {
                if(grid[i][j] == 1 && colors[i][j] == 0)
                {
                    dfs(i, j, nc);
                    ++nc;
                }
            }
        }
        if(nc != 2)
            return 0;
        if(num == 1)
            return 1;
        if(num == 2)
        {
            if(x - 1 >= 0 && grid[x-1][y] == 1)
            {
                return 2;
            }
            if(x + 1 < m && grid[x+1][y] == 1)
            {
                return 2;
            }
            if(y - 1 >= 0 && grid[x][y-1] == 1)
            {
                return 2;
            }
            if(y + 1 < n && grid[x][y+1] == 1)
            {
                return 2;
            }
        }
        int ans = 2;
        for(int i = 0; i < m; ++i)
        {
            for(int j = 0; j < n; ++j)
            {
                if(grid[i][j] == 1)
                {
                    //cout<<"hh\n";
                    grid[i][j] = 0;
                    int co = cl();
                    //cout<<"co = "<<co<<endl;
                    if(co != 1)
                    {
                        ans = min(ans, 1);
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return ans;
    }
};
