class Solution {
public:
    int n, m;
    void dfs(int i, int j, vector<vector<int>>& grid, vector<vector<int>>& vis){
        vis[i][j] = 1;
        if(i + 1 < n && grid[i + 1][j] == 1 && vis[i + 1][j] == 0){
            dfs(i + 1, j, grid, vis);
        }
        if(i - 1 >= 0 && grid[i - 1][j] == 1 && vis[i - 1][j] == 0){
            dfs(i - 1, j, grid, vis);
        }
        if(j + 1 < m && grid[i][j + 1] == 1 && vis[i][j + 1] == 0){
            dfs(i, j + 1, grid, vis);
        }
        if(j - 1 >= 0 && grid[i][j - 1] == 1 && vis[i][j - 1] == 0){
            dfs(i, j - 1, grid, vis);
        }
    }
    int minDays(vector<vector<int>>& grid) {
        n = grid.size();
        if(n == 0){
            return 0;
        }
        m = grid[0].size();
        vector<vector<int>> vis(n, vector<int>(m, 0));
        int i = -1, j = -1;
        for(int p = 0; p < n; p++){
            for(int q = 0; q < m; q++){
                if(grid[p][q] == 1){
                    i = p;
                    j = q;
                    break;
                }
            }
            if(i != -1) break;
        }
        if(i == -1) return 0;
        dfs(i, j, grid, vis);
        for(int p = 0; p < n; p++){
            for(int q = 0; q < m; q++){
                if(grid[p][q] == 1 && vis[p][q] == 0){
                    return 0;
                }
            }
        }
        for(i = 0; i < n; i++){
            for(j = 0; j < m; j++){
                if(grid[i][j] == 1){
                    grid[i][j] = 0;
                    bool flag = 0;
                    vector<vector<int>> vis1(n, vector<int>(m, 0));
                    for(int p = 0; p < n; p++){
                        for(int q = 0; q < m; q++){
                            if(grid[p][q] == 1){
                                flag = 1;
                                //cout << i << " " << j << " " << p << " " << q << endl;
                                dfs(p, q, grid, vis1);
                                break;
                            }
                        }
                        if(flag){
                            break;
                        }
                    }
                    if(flag == 0){
                        return 1;
                    }
                    for(int p = 0; p < n; p++){
                        for(int q = 0; q < m; q++){
                            if(grid[p][q] == 1 && vis1[p][q] == 0){
                                return 1;
                            }
                        }
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
