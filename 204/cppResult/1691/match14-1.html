class Solution {
public:
    void travelLand(vector<vector<int>>& grid, int x, int y)
    {
        if (x < 0 || x == grid.size() || y < 0 || y == grid[0].size())
        {
            return ;
        }
        if (grid[x][y] != 1) return;
        grid[x][y] = 2;
        travelLand(grid, x + 1, y);
        travelLand(grid, x - 1, y);
        travelLand(grid, x, y + 1);
        travelLand(grid, x, y - 1);
    }
    

    int getLands(vector<vector<int>>& grid)
    {
        int land_count = 0;
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[0].size(); ++j)
            {
                if (grid[i][j] == 1)
                {
                    land_count += 1;
                    travelLand(grid, i, j);
                }
            }
        }
        
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[0].size(); ++j)
            {
                if (grid[i][j] == 2)
                {
                    grid[i][j] = 1;
                }
            }
        }
        return land_count;
    }
    

    int minDays(vector<vector<int>>& grid) {
        
        if (getLands(grid) != 1) return 0;
        
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[0].size(); ++j)
            {
                if (grid[i][j] == 1)
                {
                    grid[i][j] = 0;
                    if (getLands(grid) != 1) return 1;
                    grid[i][j] = 1;
                }
            }
        }

        return 2;
    }
};
