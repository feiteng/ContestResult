class Solution {
    int dr[4] = { 0, 1, 0, -1 };
    int dc[4] = { 1, 0, -1, 0 };
public:
    int minDays(vector<vector<int>>& grid) {

        if (!isConnected(grid)) {
            return 0;
        }
        for (int r = 0; r < grid.size(); ++r) {
            for (int c = 0; c < grid[r].size(); ++c) {
                if (grid[r][c] == 1) {
                    grid[r][c] = 0;
                    if (!isConnected(grid)) {
                        return 1;
                    }
                    grid[r][c] = 1;
                }
            }
        }
        return 2;
    }
private:
    bool isConnected(const vector<vector<int>>& grid) {
        vector<vector<int>> color(grid);
        bool filled = false;

        for (int r = 0; r < color.size(); ++r) {
            for (int c = 0; c < color[r].size(); ++c) {
                if (color[r][c] == 1) {
                    if (!filled) {
                        floodfill(color, r, c);
                        filled = true;
                    }
                    else {
                        return false;
                    }
                };
            }
        }
        return true;
    }
    
    void floodfill(vector<vector<int>>& color, int r, int c) {
        int n = color.size(), m = color[0].size();

        if (r < 0 || c < 0 || r >= n || c >= m) {
            return;
        }
        if (color[r][c] != 1) {
            return;
        }
        color[r][c] = 2;
        for (int i = 0; i < 4; ++i) {
            floodfill(color, r + dr[i], c + dc[i]);   
        }
    } 
};
