class Solution {
public:
    int m,n;
    vector<vector<bool>> col;
    int count(vector<vector<int>>& grid){
        col=vector<vector<bool>>(m,vector<bool>(n));

        int res=0;
        for(int i=0;i<m;++i)
            for(int j=0;j<n;++j)
                if(grid[i][j]&&!col[i][j]){
                    dfs(grid,i,j);
                    ++res;
                }
        return res;
    }
    void dfs(vector<vector<int>>& grid, int x, int y){
        col[x][y]=true;
        if(x>0&&grid[x-1][y]&&!col[x-1][y])
            dfs(grid,x-1,y);
        if(x+1<m&&grid[x+1][y]&&!col[x+1][y])
            dfs(grid,x+1,y);
        if(y>0&&grid[x][y-1]&&!col[x][y-1])
            dfs(grid,x,y-1);
        if(y+1<n&&grid[x][y+1]&&!col[x][y+1])
            dfs(grid,x,y+1);

    }
    int minDays(vector<vector<int>>& grid) {
        m=grid.size();
        n=grid[0].size();
        if(count(grid)!=1)
            return 0;
        for(int i=0;i<m;++i)
            for(int j=0;j<n;++j)
                if(grid[i][j]){
                    grid[i][j]=0;
                    if(count(grid)!=1)
                        return 1;
                    grid[i][j]=1;
                }
        return 2;
    }
};
