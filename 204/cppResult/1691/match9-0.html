class Solution {
public:
    vector<vector<int>> gg;
    void dfs(int x,int y)
    {
    gg[x][y]=0;
        if(x+1<gg.size() && gg[x+1][y]==1)
        {
    dfs(x+1,y);
        }
         if(x-1>=0 && gg[x-1][y]==1)
        {
    dfs(x-1,y);
         }
           if(y+1<gg[0].size() && gg[x][y+1]==1)
        {
    dfs(x,y+1);
        }
         if(y-1>=0 && gg[x][y-1]==1)
        {
    dfs(x,y-1);
         }
    }
    bool solve(vector<vector<int>>& grid)
    {
      gg=grid;
        bool ok=1;
        for(int i=0;i<grid.size()&&ok;i++)
        {
        for(int o=0;o<grid[0].size()&&ok;o++)
        {
        if(grid[i][o]==1)
        {
        ok=0;
        dfs(i,o);
        }
        }
        }
        ok=1;
        for(int i=0;i<grid.size()&&ok;i++)
        {
        for(int o=0;o<grid[0].size()&&ok;o++)
        {
        if(gg[i][o]==1)
        {
        ok=0;
        }
        }
        }
        if(ok)
        {
        return 0;
        }
        return 1;
    }
    int minDays(vector<vector<int>>& grid) {
      if(solve(grid))
      {
         return 0;
      }
         for(int i=0;i<grid.size();i++)
        {
        for(int o=0;o<grid[0].size();o++)
        {
        if(grid[i][o])
        {
        grid[i][o]=0;
            if(solve(grid))
            {
            return 1;
            }
            grid[i][o]=1;
        }
        }
         }
        return 2;
        
    }
};
