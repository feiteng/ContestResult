class Solution {
public:
    int minDays(vector<vector<int>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();
        int ones = 0;
        for(int r = 0; r < rows; ++r) {
            for(int c = 0; c < cols; ++c) {
                if(grid[r][c] == 1) {
                    ++ones;
                }
            }
        }
        if(ones == 1) {
            return 1;
        }
        int nums = islandNum(grid);
        if(nums != 1) {
            return 0;
        }
        if(nums == 1 && ones == 2) {
            return 2;
        }

         for(int r = 0; r < rows; ++r) {
            for(int c = 0; c < cols; ++c) {
                if(grid[r][c] == 1) {             
                    grid[r][c] = 0;
                    if(islandNum(grid) >= 2) {
                        return 1;
                    }
                    grid[r][c] = 1;
                }
            }
        }
        return 2;
    }
    int islandNum(vector<vector<int>> grid){
        int rows = grid.size();
        int cols = grid[0].size();
        int res = 0;
        for(int r = 0; r < rows; ++r) {
            for(int c = 0; c < cols; ++c) {
                if(numOfIslands(r, c, grid) == true) {
                    ++res;
                    if(res >= 2) {
                        return res;
                    }
                }
            }
        }
        return res;
    }
    
    
    bool numOfIslands(int r, int c, vector<vector<int>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();
        if(r < 0 || r>=rows || c < 0 || c>= cols || grid[r][c] == 0) {
            return false;
        }
        grid[r][c] = 0;
        numOfIslands(r-1, c, grid);
        numOfIslands(r, c-1, grid);
        numOfIslands(r+1, c, grid);
        numOfIslands(r, c+1, grid);
        return true;
    }
};
