class Solution {
public:
    
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    void dfs(int x, int y, vector<vector<int>>& grid, vector<vector<int>>& visited) {

        int n = grid.size(), m = grid[0].size();
        visited[x][y] = 1;
        for(int i = 0; i < 4; i++) {
            int nx = x+dx[i], ny = y+dy[i];
            if(nx >= 0 && nx < n && ny >= 0 && ny < m && grid[nx][ny] == 1 && !visited[nx][ny]) {
                dfs(nx, ny, grid, visited);
            }
        }
    }
    
    int nislands(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> visited(n, vector<int>(m));

        int ret = 0;
        
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(grid[i][j] == 1 && !visited[i][j]) {
                    dfs(i, j, grid, visited);
                    ret++;
                }
            }
        }
        return ret;
    }
    
    int minDays(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        int ans = 2;
        if(nislands(grid) != 1) {
            return 0;
        }

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(grid[i][j] == 1) {
                    grid[i][j] = 0;
                    int k = nislands(grid);
                    if(k != 1) {
                        ans = 1;
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        
        return ans;
    }
};
