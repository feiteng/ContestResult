const int N = 35;

class Solution {
public:
    vector<vector<int>> *v;
    int vv[N][N];
    int n, m;
    int dx[4] = {0, 0, 1, -1}, dy[4] = {1, -1, 0, 0};
    
    void dfs(int x, int y) {
        if (x < 0 || x >= n || y < 0 || y >= m || (*v)[x][y] == 0) {
            return;
        }
        if (vv[x][y] == 1) {
            return;
        }
        vv[x][y] = 1;
        for (int d = 0; d < 4; d++) {
            dfs(x + dx[d], y + dy[d]);
        }
    }
    
    int ok() {
        bool first = true;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                vv[i][j] = 0;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if ((*v)[i][j] == 1 && vv[i][j] == 0) {
                    if (first) {
                        dfs(i, j);
                        first = false;
                    } else {
                        return true;
                    }
                }
            }
        }
        return false;
    }
    
    int minDays(vector<vector<int>>& a) {
        this->v = &a;
        int n = a.size(), m = a[0].size();
        this->n = n;
        this->m = m;
        if (ok()) {
            return 0;
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (a[i][j] == 1) {
                    a[i][j] = 0;
                    if (ok()) {
                        return 1;
                    }
                    a[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
