class Solution {
public:
    int vist[31][31];
    int dx[4] = {0, 0, 1, -1};
    int dy[4] = {1, -1, 0, 0};
    void dfs(int r, int c, vector<vector<int>>& grid) {
        vist[r][c] = 1;
        for(int i=0; i<4; i++) {
            int tr = r + dx[i], tc = c + dy[i];
            if(tr >=0&& tr<grid.size() && tc>=0 && tc<grid[0].size()) {
                if(grid[tr][tc] && !vist[tr][tc]) dfs(tr, tc, grid);
            }
        }
    }
    int check(vector<vector<int>>& grid) {
        memset(vist, 0, sizeof vist);
        int ans = 0;
        for(int i=0; i<grid.size(); i++) {
            for(int j=0; j<grid[0].size(); j++) {
                if(grid[i][j] && !vist[i][j]) dfs(i, j, grid), ++ans;
            }
        }
        return ans;
    }
    int minDays(vector<vector<int>>& grid) {
        int ans = 2;
        if(check(grid) > 1) return 0;
        for(int i=0; i<grid.size(); i++) {
            for(int j=0; j<grid[0].size(); j++) {
                if(grid[i][j]) {
                    grid[i][j] = 0;
                    if(check(grid) > 1) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
