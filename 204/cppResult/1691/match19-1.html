class Solution {
    bool bz[35][35];
    vector<vector<int>> g;

    int fx[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
    int n,m;
public:

    void dfs(int x,int y){
        bz[x][y]=1;
        for(int k=0;k<4;k++){
            int xx=x+fx[k][0];
            int yy=y+fx[k][1];
            if (xx<0 || yy<0 || xx>=n || yy>=m) continue;
            if (bz[xx][yy]) continue;
            if (g[xx][yy]!=1) continue;
            dfs(xx,yy);
        }
    }
    bool check(vector<vector<int>>& grid) {
        memset(bz,0,sizeof(bz));
        g=grid;

        int tot=0;
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++) {
                if (g[i][j]==1 && bz[i][j]==0) {
                    tot++;
                    dfs(i,j);
                }
            }
        return tot!=1;
    }
    int minDays(vector<vector<int>>& grid) {
        int ret=0;
        n=grid.size();
        m=grid[0].size();

        if (check(grid)) return 0;
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++) {
                if (grid[i][j]==1) {
                    grid[i][j]=0;
                    if (check(grid)) return 1;
                    grid[i][j]=1;
                }
            }
        return 2;
    }
};
