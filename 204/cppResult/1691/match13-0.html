class Solution {
public:
    int minDays(vector<vector<int>>& grid) {
		if(ct(grid)!=1)return 0;
		for(int i=0;i<grid.size();i++){
			for(int j=0;j<grid[i].size();j++){
				if(grid[i][j]==1){
					grid[i][j]=0;
					if(ct(grid)!=1)return 1;
					grid[i][j]=1;
				}
			}
		}
		return 2;
    }
	int ct(vector<vector<int>> grid){
		int ret=0;
		for(int i=0;i<grid.size();i++){
			for(int j=0;j<grid[i].size();j++){
				if(grid[i][j]==1){
					dfs(grid,i,j);
					ret++;
				}
			}
		}
		return ret;
	}
	void(dfs(vector<vector<int>>&grid,int i,int j)){
		grid[i][j]=0;
		if(i>0&&grid[i-1][j]==1)
			dfs(grid,i-1,j);
		if(i+1<grid.size()&&grid[i+1][j]==1)
			dfs(grid,i+1,j);
		if(j>0&&grid[i][j-1]==1)
			dfs(grid,i,j-1);
		if(j+1<grid[i].size()&&grid[i][j+1]==1)
			dfs(grid,i,j+1);
	}
};
