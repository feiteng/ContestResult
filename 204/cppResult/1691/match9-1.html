class Solution {
public:
    int n, m;

    void dfs(int r, int c, vector<vector<int>>& g, vector<vector<bool>>& vis){
        vis[r][c]=1;
        if(r>0 and g[r-1][c]==1 and vis[r-1][c]==0){
            dfs(r-1, c, g, vis);
        }
        if(c>0 and g[r][c-1]==1 and vis[r][c-1]==0){
            dfs(r, c-1, g, vis);
        }
        if(r<n-1 and g[r+1][c]==1 and vis[r+1][c]==0){
            dfs(r+1, c, g, vis);
        }
        if(c<m-1 and g[r][c+1]==1 and vis[r][c+1]==0){
            dfs(r, c+1, g, vis);
        }
    }
    int minDays(vector<vector<int>>& g) {
        n = g.size();
        m = g[0].size();

        int count=0;
        vector<vector<bool>> vis(n, vector<bool> (m,0));
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(g[i][j]==1 and vis[i][j]==0){
                    dfs(i, j, g, vis);
                    count++;
                }
            }
        }
        if(count>1) return 0;

        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(g[i][j]==1){
                    g[i][j]=0;
                    vector<vector<bool>> vs(n, vector<bool> (m,0));

                    int c =0;
                    for(int x=0;x<n;x++){
                        for(int y=0;y<m;y++){
                            if(g[x][y]==1 and vs[x][y]==0){
                                dfs(x,y,g,vs);
                                c++;
                            }
                        }
                    }
                    if(c>1)return 1;
                    g[i][j]=1;
                }
            }
        }
        return 2;
    }
};
