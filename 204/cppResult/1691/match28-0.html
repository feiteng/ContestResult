class Solution {
public:
    int minDays(vector<vector<int>>& grid) {
        if (!conn(grid)) {
            return 0;
        }
        
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[i].size(); ++j) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (!conn(grid)) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        
        
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[i].size(); ++j) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    for (int p = 0; p < grid.size(); ++p) {
                        for (int q = 0; q < grid[p].size(); ++q) {
                            if (grid[p][q] == 1) {
                                grid[p][q] = 0;
                                if (!conn(grid)) {
                                    return 2;
                                }       
                                grid[p][q] = 1;
                            }
                        }
                    }
                    grid[i][j] = 1;
                }
            }
        }
        
        return -1;
    }
    
    bool conn(vector<vector<int>>& grid) {
        int n = 0;
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[i].size(); ++j) {
                if (grid[i][j] == 1) {
                    ++n;
                    check(grid, i, j);
                }
                if (n > 1) {
                    break;
                }
            }
        }
        
        
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[i].size(); ++j) {
                if (grid[i][j] == 2) {
                    grid[i][j] = 1;
                }
            }
        }
        
        return n == 1;
    }
    
    void check(vector<vector<int>>& grid, int i, int j) {
        vector<int> dir{1, 0, -1, 0, 1};
        for (int k = 0; k < 4; ++k) {
            int m = i + dir[k];
            int n = j + dir[k + 1];
            if (m >= 0 && m < grid.size() && n >= 0 && n < grid.back().size() && grid[m][n] == 1) {
                grid[m][n] = 2;
                check(grid, m, n);
            }
        }
    }
};
