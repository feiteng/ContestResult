class Solution {
public:
    int dir[4][2] = {{-1,0},{1,0},{0,1},{0,-1}};
    int n, m;
    bool check(vector<vector<int>> grid) {
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 1) {
                    res++;
                    dfs(i,j,grid);
                    if (res == 2) {
                        return true;
                    }
                }
            }
        }
        return res != 1;
    }
    void dfs(int x, int y, vector<vector<int>>& grid) {
        grid[x][y] = 0;
        for (int i = 0; i < 4; ++i) {
            int nx = x + dir[i][0];
            int ny = y + dir[i][1];
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && grid[nx][ny] == 1) {
                dfs(nx,ny,grid);
            }
        }
    }
    int minDays(vector<vector<int>>& grid) {
        n = grid.size(), m = grid[0].size();
        if (check(grid)) return 0;
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (check(grid)) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
