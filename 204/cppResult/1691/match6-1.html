class Solution {
public:
    int dx[5] = {1, -1, 0, 0, 0};
    int dy[5] = {0, 0, 1, -1, 0};
    void dfs(vector<vector<int>> &grid, int x, int y, vector<vector<bool>> &vis) {
        vis[x][y] = 1;

        int n = grid.size(), m = grid[0].size();
        for(int i = 0; i < 4; i++) {
            int nx = x+dx[i], ny = y+dy[i];
            if(nx >= 0 && nx < n && ny >= 0 && ny < m && grid[nx][ny] == 1 && !vis[nx][ny]) {
                dfs(grid, nx, ny, vis);
            }
        }
    }
    bool connected(vector<vector<int>> &grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<bool>> vis(n, vector<bool>(m));

        int num = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(!vis[i][j] && grid[i][j]) {

                    num++;
                    dfs(grid, i, j, vis);
                }
            }
        }
        return num == 1;
    }
    int minDays(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        if(!connected(grid)) return 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if(!connected(grid)) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
};
