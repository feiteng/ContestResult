class Solution {
public:
    int xx[4]={1,-1,0,0};
    int yy[4]={0,0,1,-1};
    void dfs(vector<vector<int>>& grid,int x,int y,int&n,int&m,vector<vector<bool>> &flag){
        //cout << x<<' '<<y<<endl;
        flag[x][y]=true;
        for(int i =0;i<4;i++){
            if(x+xx[i]>=0&&x+xx[i]<n&&y+yy[i]>=0&&y+yy[i]<m&&grid[x+xx[i]][y+yy[i]]>0&&!flag[x+xx[i]][y+yy[i]]){
                dfs(grid,x+xx[i],y+yy[i],n,m,flag);
            }
        }
    }

    bool check(vector<vector<int>>& grid,int&n,int&m){
        vector<vector<bool>> flag(n,vector<bool>(m,false));
        int cnt = 0;
        for(int i =0;i<n;i++){
            for(int j =0;j<m;j++){
                if(grid[i][j] > 0&&!flag[i][j]){
                    if(cnt>0){
                        return false;
                    }
                    cnt++;
                    dfs(grid,i,j,n,m,flag);
                }
            }
        }
        return cnt>0;
    }
    int minDays(vector<vector<int>>& grid) {
        int n = grid.size(),m=grid[0].size();

        if(!check(grid,n,m)){
            return 0;
        }
        for(int i =0;i<n;i++){
            for(int j =0;j<m;j++){
                if(grid[i][j] == 1){
                    grid[i][j] = 0;
                    if(!check(grid,n,m)){
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        for(int i =0;i<n;i++){
            for(int j =0;j<m;j++){
                if(grid[i][j] == 1){
                    grid[i][j] =0;
                    for(int x= 0;x<n;x++){
                        for(int y= 0;y<m;y++){
                            if(grid[x][y] == 1){
                                grid[x][y] =0;
                                if(!check(grid,n,m)){
                                    return 2;
                                }
                                grid[x][y]=1;
                            }
                        }
                    }
                    grid[i][j]=1;
                }
            }
        }
        return 3;
    }
};
