class Solution {
 public:
  int n,m;
  char vis[33][33];
  void dfs(vector<vector<int>>& g,int i,int j){
	if(!g[i][j]||vis[i][j])return;
   vis[i][j]=1;
   if(i)dfs(g,i-1,j);
   if(j)dfs(g,i,j-1);
   if(i<n-1)dfs(g,i+1,j);
   if(j<m-1)dfs(g,i,j+1);
  }
  bool sat(vector<vector<int>>& g){
   memset(vis,0,sizeof vis);
   int c=0;
   for(int i=0;i<n;++i)for(int j=0;j<m;++j)if(g[i][j]&&!vis[i][j]){
    dfs(g,i,j);++c;
   }
   return c!=1;
  }
  int minDays(vector<vector<int>>& g) {
   n=g.size();m=g[0].size();
   if(sat(g))return 0;
   for(int i=0;i<n;++i)for(int j=0;j<m;++j)if(g[i][j]){
    g[i][j]=0;
    if(sat(g))return 1;
    g[i][j]=1;
   }
   return 2;
  }
 };
