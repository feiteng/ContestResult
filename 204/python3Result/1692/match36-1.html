import math

def C(n, m):
    p = 1000000007
    def power(x,y):     #求x的y次方
        p = 1000000007
        res = 1
        while y:
            if y % 2 != 0:
                res *= (x%p)
            y >>= 1
            x *= (x%p)
        return res
    a = (math.factorial(n))%p
    b = (power(math.factorial(m),(p-2)))%p
    c = (power(math.factorial(n-m),(p-2)))%p
    return a*b*c%p

class Solution:
    
def helper(self, nums):
        if len(nums) <= 2:
            return 1
        root = nums[0]
        left = []
        right = []
        for x in nums[1:]:
            if x < root:
                left.append(x)
            else:
                right.append(x)
        m, n = len(left), len(right)
        f = C(m + n, m)
        l, r = self.helper(left), self.helper(right)
        return (f * l * r) % 1000000007
        
    
    def numOfWays(self, nums: List[int]) -> int:
        return (self.helper(nums) - 1 + 1000000007) % 1000000007
