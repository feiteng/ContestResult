class Solution:
    def prepare(self, n):
        self.p = 10 ** 9 + 7
        self.n = n
        self.C = []
        for _ in range(self.n):
            self.C.append([1 for _ in range(self.n)])
        for i in range(2, self.n):
            for j in range(1, i):
                self.C[i][j] = (self.C[i - 1][j] + self.C[i - 1][j - 1]) % self.p

    def sv(self, l):
        if len(l) <= 1:
            return 1
        pivot = l[0]
        l1 = []
        l2 = []
        for i in l[1:]:
            if i < pivot:
                l1.append(i)
            else:
                l2.append(i)
        return self.sv(l1) * self.sv(l2) * self.C[len(l) - 1][len(l1)] % self.p

    def numOfWays(self, nums) -> int:
        self.prepare(len(nums))
        return self.sv(nums) - 1
