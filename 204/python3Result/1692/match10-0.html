class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        mapping = {}
        def _cal0(x,y):
            if not x or not y:
                return 1
            code = x*10000+y
            if code not in mapping:
                res = _cal0(x-1,y)+_cal0(x,y-1)
                mapping[code] = res
            return mapping[code]
        def _cal1(arr):
            if not arr:
                return 1,0
            gt = []
            lt = []
            for i in range(1,len(arr)):
                if arr[i] > arr[0]:
                    gt.append(arr[i])
                else:
                    lt.append(arr[i])
            numg,lengthg = _cal1(gt)
            numl,lengthl = _cal1(lt)
            return _cal0(lengthg,lengthl)*numg*numl,len(arr)
        num,lgth = _cal1(nums)
        return (num-1)%(10**9+7)
