class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return 0
        
        mod = 10**9+7
        N = 1100
        fac = [1]*(N+1)
        finv = [1]*(N+1)
        for i in range(N):
            fac[i+1] = fac[i] * (i+1) % mod
        finv[-1] = pow(fac[-1], mod-2, mod)
        for i in reversed(range(N)):
            finv[i] = finv[i+1] * (i+1) % mod

        def cmb1(n, r, mod):
            if r <0 or r > n:
                return 0
            r = min(r, n-r)
            
return fac[n] * finv[r] * finv[n-r] % mod
        
        def calc(nums):
            if len(nums) <= 1:
                return 1
            l = []
            u = []
            for i in range(1, len(nums)):
                if nums[i] > nums[0]:
                    u.append(nums[i])
                else:
                    l.append(nums[i])
            res = cmb1(len(nums)-1, len(l), mod)*calc(u)*calc(l)
            return res
        return (calc(nums)-1)%mod
