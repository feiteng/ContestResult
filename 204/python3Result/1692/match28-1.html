class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def cnm(m, n):
            res 
= 1
            for i in range(m, 0, -1):
                res *= i + n
            for i in range(m, 0, -1):
                res //= i
            return res
        
        def _numOfWays(nums):
            if len(nums) <= 2:
                return 1
            larger = [i for i in nums if i > nums[0]]
            smaller = [i for i in nums if i < nums[0]]
            if nums[0] == max(nums) or nums[0] == min(nums):
                return _numOfWays(nums[1:])
            return _numOfWays(larger) * _numOfWays(smaller) * cnm(len(larger), len(smaller))
            
        return _numOfWays(nums) % (10**9 + 7) - 1
