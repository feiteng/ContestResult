from math import comb
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def cnt(arr):
            
if len(arr) <= 2:
                return 1 
            root = arr[0]
            left = []
            right = []

            for i in range(1,len(arr)):
                if arr[i] < root:
                    left.append(arr[i])
                else:
                    right.append(arr[i])
            return comb(len(arr)-1, len(left)) * cnt(left) * cnt(right)
        return (cnt(nums) - 1)%(10**9 + 7)
