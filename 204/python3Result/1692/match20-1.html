MOD=10**9+7
import math
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        
        res=self.help(nums)
        
        return res-1
    


    def nCr(self,n,r):
        f = math.factorial
        return f(n) // f(r) // f
(n-r)
    
    def help(self,nums):
        
        if len(nums)==0:
            return 1
        
        L1,L2=[],[]
        
        for i in range(1,len(nums)):
            
            if nums[i]>nums[0]:
                L2.append(nums[i])
            
            else:
                L1.append(nums[i])
                
        c1,c2=len(L1),len(L2)
        
        n1=self.help(L1)
        n2=self.help(L2)
        
        res=n1*n2*self.nCr(c1+c2,c1)
        res%=MOD
        
        return res
