def binom(n,r):
    f = math.factorial
    return f(n) // f(r) // f(n-r)

class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = self.right = None

def add_node(node, x):
    if x < node.val:
        if node.left is None:
            node.left = TreeNode(x)
        else:
            add_node(node.left, x)
    elif x > node.val:
        if node.right is None:
            node.right = TreeNode(x)
        else:
            add_node(node.right, x)
            
        
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        root = TreeNode(nums[0])
        for x in nums[1:]:
            add_node(root, x)
            
        def helper(node):
            if node is None: return 0, 1
            cl, nl = helper(node.left)
            cr, nr = helper(node.right)
            c = cl + cr + 1
            n = binom(cl + cr, cl) * nl * nr
            return c, n
        return (helper(root)[1] - 1) % int(1e9 + 7)
