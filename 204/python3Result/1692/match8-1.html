from functools import lru_cache
import sys

M = 10**9 + 7
@lru_cache(None)
def p(n,m):
    if m == 1:
        return 1
    if n == 1:
        return m
    if n == 0:
        return 1
    # print(n,m)
    return (p(n-1, m) + p(n,m-1)) % M
class Solution:
    
def numOfWays(self, nums: List[int]) -> int:
        if len(nums) <= 1:
            return 0
        piv = nums[0]
        
        big = []
        small = []
        for n in nums[1:]:
            if n > piv:
                big.append(n)
            else:
                small.append(n)
        bw = self.numOfWays(big) + 1 
        sm = self.numOfWays(small) + 1 
        return (bw * sm * p(len(big), len(small)+1) - 1) % M
