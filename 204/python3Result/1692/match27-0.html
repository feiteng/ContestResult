class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def helper(nums):
            if len(nums) <= 2:
                return 1
            lefts = []
            rights = []
            for num in nums[1:]:
                if num < nums[0]:
                    lefts.append(num)
                else:
                    rights.append(num)
            mleft = helper(lefts)
            mright = helper(rights)
            p = 1
            for i in range(len(lefts) + len(rights), max(len(lefts), len(rights)), -1):
                p *= i
            for i in range(2, min(len(lefts), len(rights)) + 1):
                p //= i
            return mleft * mright * p

        return (helper(nums) - 1) % (10**9 + 7)
