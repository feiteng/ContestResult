class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        # 找到以x为root的左右子树
        MOD = 10 ** 9 + 7
        
        def C(a, b):
            ans = 1
            for x in range(a, a - b, -1): ans *= x
            for x in range(1, b + 1): ans //= x
            return ans
        #print(C(6, 3))
        
        def recur(nums):
            if len(nums) == 0: return 1
            if len(nums) == 1: return 1
            
            flag = sorted(nums)
            if flag == nums or flag[::-1] == nums: return 1
            
            root = nums[0]
            left = []
            right = []
            for n in nums:
                if n > root: right.append(n)
                if n < root: left.append(n)
            
            l_res, r_res = recur(left), recur(right)
            
            return (C(len(left) + len(right), len(left)) * l_res * r_res) % MOD
        
        return recur(nums) - 1
