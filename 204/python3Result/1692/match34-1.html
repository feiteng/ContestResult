class Solution:
    

    
    def numOfWays(self, nums: List[int]) -> int:
        mod = int(10 ** 9 + 7)
        f = [1] * (len(nums) + 1)
        for i in range(2, len(f)):
            f[i] = (f[i - 1] * i) % mod
            
        def fastpow(a , b):
            res = 1
            while b:
                if(b&1):	
                    res=(res*a)%mod 
                a = (a*a)%mod 
                b >>= 1 
            return res
        def inv(a):
            return fastpow(a, mod - 2)
        
        
        def C(n, m):
            return ((f[n] * inv(f[m])) % mod * inv(f[n-m])) % mod
                    
        def helper(nums):
            if not nums: return 1
            
if len(nums) == 1: return 1
            f = nums[0]
            left, right = [], []
            for i in range(1, len(nums)):
                if (nums[i] < f):
                    left.append(nums[i])
                else:
                    right.append(nums[i])
            res = C(len(nums) - 1, len(left))
            res = (res * helper(left)) % mod
            res = (res * helper(right)) % mod
            return int(res)
        return helper(nums) - 1
