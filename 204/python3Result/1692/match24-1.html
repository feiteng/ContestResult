class Solution:
    def prec(self, n):
        self.mod = int(1e9 + 7)
        self.C = [[0] * (n + 1) for _ in range(n + 1)]
        self.C[0][0] = 1
        for i in range(1, n + 1):
            self.C[i][0] = 1
            self.C[i][i] = 1
            
for j in range(1, i):
                self.C[i][j] = (self.C[i - 1][j - 1] + self.C[i - 1][j])% self.mod
    
    def now(self, nums):
        if len(nums) <= 1:
            return 1
        nl, nr = [], []
        for i in nums[1:]:
            if i < nums[0]:
                nl
.append(i)
            else:
                nr.append(i)
        return self.now(nl) * self.now(nr) % self.mod * self.C[len(nl) + len(nr)][len(nr)] % self.mod
                
                
    def numOfWays(self, nums: List[int]) -> int:
        self.prec(len(nums))
        return (self.now(nums) + self.mod - 1)% self.mod
