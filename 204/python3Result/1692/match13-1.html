class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        from math import comb
        mod = pow(10, 9) + 7
        
        
def helper(arr):
            if len(arr) <= 2:
                return 1
            root = arr[0]
            left, right = [], []
            for n in arr[1:]:
                if n < root:
                    left.append(n)
                else:
                    right.append(n)
            l, r = helper(left), helper(right)
            return l * r * comb(len(left) + len(right), len(left)) % mod
        
        return helper(nums) - 1
