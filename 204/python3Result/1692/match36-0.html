class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def dfs(sub_nums):
            if len(sub_nums) <= 2:
                return 1
            
            root = sub_nums[0]
            min_sub = []
            max_sub = []
            for num in sub_nums[1:]:
                if num < root:
                    min_sub.append(num)
                else:
                    max_sub.append(num)
            min_num = dfs(min_sub)
            max_num = dfs(max_sub)

            return min_num * max_num * math.comb(len(min_sub) + len(max_sub), len(min_sub))
        
        return (dfs(nums) - 1) % (10 ** 9 + 7)
