from math import comb
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        return self.search(nums) - 1
    
    def search(self,nums):
        if len(nums) <= 1:
            return 1
        smaller, larger = [], []
        for i in range(1,len(nums)):
            if nums[i] > nums[0]:
                larger.append(nums[i])
            else:
                smaller.append(nums[i])
        ns = self.search(smaller)
        nl = self.search(larger)
        ans = ns * nl % (10**9 + 7)
        m,n = len(smaller),len(larger)
        ans = ans * comb(m+n,n) % (10**9+7)
        return ans
