class Tree: 
    def __init__(self, val): 
        self.val = val
        self.left = None 
        self.right = None

class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        if len(nums)<=1:  
            return 0 
        root = Tree(nums[0])
        for num in nums[1:]: 
            temp = root
            while temp: 
                if temp.val>num: 
                    if temp.right: 
                        temp = temp.right
                    else: 
                        temp.right = Tree(num)
                        break 
                else: 
                    if temp.left: 
                        temp = temp.left
                    else: 
                        temp.left = Tree(num)
                        break 
        mod = pow(10, 9) + 7   
        from math import comb
        def countWay(node): 
            if not node: 
                # (way, nodeNums)
                return (1, 0)
            leftWay, leftNums = countWay(node.left)
            rightWay, rightNums = countWay(node.right)
            return (leftWay*rightWay*comb(leftNums+rightNums, rightNums)%mod, leftNums+rightNums+1)
        return countWay(root)[0]%mod - 1
