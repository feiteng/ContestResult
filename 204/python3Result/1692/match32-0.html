class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        return count(nums) - 1
        
def count(nums):
    if nums == []:
        return 1
    if len(nums) == 1:
        return 1
    head = nums[0]
    left = []
    right = []
    for i in range(1, len(nums)):
        if nums[i] < head:
            left.append(nums[i])
        else:
            right.append(nums[i])
    left_c = count(left) % (10 ** 9 + 7)
    right_c = count(right) % (10**9+7)
    fact = fac(len(nums)-1) // (fac(len(left)) * fac(len(nums)- len(left)-1))
    fact = int(fact % (10**9+7))
    print(nums)
    print('fact:', fact)
    print(left_c)
    print(right_c)
    return (left_c * right_c * fact) % (10**9+7)

def fac(n):
    if n == 0:
        return 1
    res = 1
    for i in range(n, 1, -1):
        res *= i
    return res
