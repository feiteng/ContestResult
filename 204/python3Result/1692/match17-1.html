class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        def factor(x):
            if x <= 1:
                return 1
            return x * factor(x-1)
        
        def f(a, b):
            return factor(a+b)//(factor(a)*factor
(b)) % MOD
        
        def sol(nums):
            if len(nums) <= 1: return 1
            L, R = [], []
            for i in range(1, len(nums)):
                if nums[i] < nums[0]:
                    L.append(nums[i])
                else:
                    R.append(nums[i])
                    
            return sol(L) * sol(R) * f(len(L), len(R)) % MOD
            
        return sol(nums)-1
