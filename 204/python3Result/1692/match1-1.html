class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def helper(k):
            res = 1
            while k:
                res *= k
                k -= 1
            return res
        
def dfs(l):
            if len(l) <= 2:
                return 1
            tar = l[0]
            l1 = []
            r = []
            for i in range(1,len(l)):
                if l[i] > tar:
                    r.append(l[i])
                else:
                    l1.append(l[i])
            return dfs(l1)*dfs(r)*helper(len(l)-1)//helper(len(l1))//helper(len(r)) % 1000000007
        return dfs(nums)-1
