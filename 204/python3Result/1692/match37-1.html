class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def helper(nums):
            if len(nums)<=1: return 1
            L, R = [], []
            for x in nums:
                if x<nums[0]: L.append(x)
                elif x>nums[0]: R
.append(x)
            l, r = helper(L), helper(R)
            return l*r*math.comb(len(L)+len(R), len(L))
        
        return (helper(nums)-1) % (10**9+7)
