class Solution:
    
    
    def numOfWays(self, nums: List[int]) -> int:
        mod = 10 ** 9 + 7
        
        self.res = 0
    
        def choose(m, n):
            cur = (math.factorial(m) // (math.factorial(m-n) * math.factorial(n))) % mod
            # print(m, n, cur)
            return cur
        
        
def count(nums):
            if not nums or len(nums) == 1:
                return 1
            
            left = []
            right = []

            for x in nums[1:]:
                if x < nums[0]:
                    left.append(x)
                else:
                    right.append(x)
            
            m = len(left)
            n = len(right)
            cl = count(left)
            cr = count(right)
            # print(left, right, cl, cr)
            cur = (choose(m + n, n) * cl * cr) % mod
            
            return cur
        
        res = count(nums)
        return (res - 1) % mod
