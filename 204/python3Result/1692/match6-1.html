class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        mod = 10**9 + 7
        def count(nums):
            if len(nums) == 0:
                return 1
            root = nums[0]
            left = [num for num in nums if num < root]
            right = [num for num in nums if num > root]
            num_ways = math.comb(len(left) + len(right), len(left))
            return num_ways * count(left) * count(right) % mod
        return count(nums) - 1
