class Solution:
    def combination(self, n, k):
        if k==0 or k==n: return 1
        rt = 1;
        for i in range(k):
            rt *= (n-i)%1000000007
        for i in range(1,k+1):
            rt //= i
        return rt
    
    def helper(self, nums):
        if (len(nums)<=1):
            return 1
        left_nums = [num for num in nums if num<nums[0]]
        right_nums = [num for num in nums if num>nums[0]]
        left_res = self.helper(left_nums)
        right_res = self.helper(right_nums)
        res = (left_res * right_res * self.combination(len(nums)-1, len(left_nums))) % 1000000007;
        return res;
    
    def numOfWays(self, nums: List[int]) -> int:
        return self.helper(nums)-1;
