class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        mod = 1000000007
        def solve(nums):
            if not nums:
                return 1
            root = nums[0]
            left 
= []
            right = []
            for i in range(1,len(nums)):
                if nums[i] < nums[0]:
                    left.append(nums[i])
                else:
                    right.append(nums[i])
            
            # print(len(left),len(right))
            m = len(left)
            n = len(right)
            # print(m,n)
            return comb(m+n,m)*solve(left)*solve(right)%mod
        return (solve(nums)-1)%mod
