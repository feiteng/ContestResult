class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        p = 10**9+7
        def nCrModp(n, r, p): 
            if (r > n- r): 
                r = n - r   

            C = [0 for i in range(r + 1)] 

            C[0] = 1 # Top row of Pascal Triangle 

            for i in range(1, n + 1): 

                for j in range(min(i, r), 0, -1): 

                    # nCj = (n - 1)Cj + (n - 1)C(j - 1) 
                    C[j] = (C[j] + C[j-1]) % p 

            return C[r]  

        def getWay(values):
            if len(values)==0: return 0
            if len(values)==1: return 1
            mid = values[0]
            left = [v for v in values if v < mid]
            right = [v for v in values if v > mid]
            
            # print("mid:", mid)
            # print("left:", left)
            # print("right:", right)
            leftWay = getWay(left)
            rightWay = getWay(right)
            if leftWay == 0:
                return rightWay
            if rightWay == 0:
                return leftWay
            nCr = nCrModp(len(left)+len(right), len(right), p)
            
            return nCr*leftWay*rightWay % p
        
        return getWay(nums)-1
