from functools import lru_cache
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        mod = (10 ** 9 + 7)
        
        @lru_cache
        def cn1n2(n1, n2):
            return math.factorial(n1 + n2) // (math.factorial(n1) * math.factorial(n2))
        
        @lru_cache(None)
        
def cal(nums):
            if len(nums) <= 2:
                return 1
            first = nums[0]
            part1, part2 = [], []
            for i in range(1, len(nums)):
                if nums[i] < first:
                    part1.append(nums[i])
                else:
                    part2.append(nums[i])
            n1, n2 = len(part1), len(part2)
            p1, p2 = cal(tuple(part1)), cal(tuple(part2))
            return cn1n2(n1, n2) * p1 * p2
        
        #print(cn1n2(500, 500))
        return (cal(tuple(nums)) - 1) % (mod)
