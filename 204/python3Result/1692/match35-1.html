from typing import List
import math



class Solution:
    def numOfWays(self, nums: List[int]) -> int:

        m = 10 ** 9 + 7

        def comb_1(n, m):
            return math.factorial(n) // (math.factorial(n - m) * math.factorial(m))  # 直接使用math里的阶乘函数计算组合数

        def myinsert(x, y):
            return (comb_1(x + y, x)) % m

        def ways(nums):
            if not nums:
                return 1
            n = len(nums)
            if n <= 1:
                return 1
            root = nums[0]
            left = []
            right = []
            for num in nums:
                if num < root:
                    left.append(num)
                elif num > root:
                    right.append(num)

            left_choice 
= ways(left)
            right_choice = ways(right)

            return (myinsert(len(left), len(right)) * left_choice * right_choice) % m

        return ways(nums) - 1


print(Solution().numOfWays([5,1,4,2,3]))
