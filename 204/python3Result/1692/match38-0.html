class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def combination(n,m):
            return factorial[n+m] // factorial[n] // factorial[m] % (10**9+7)

        def count(nums):
            if len(nums) <= 1:
                return 1
            l = []
            r = []
            for i in range(1, len(nums)):
                if nums[i] > nums[0]:
                    r.append(nums[i])
                else:
                    l.append(nums[i])
            return count(r)*count(l)*combination(len(l),len(r)) % (10**9+7)

        factorial = [1]*1000
        for i in range(1, 1000):
            factorial[i] = factorial[i-1] * i

        n = count(nums)-1
        return n
