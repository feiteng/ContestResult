class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def dfs(nums):
            if len(nums) <= 1:
                return 1
            val = nums[0]
            le, ri = [], []
            for i in range(1, len(nums)):
                if nums[i] < val:
                    le.append(nums[i])
                else:
                    ri.append(nums[i])
            ml, mr = dfs(le), dfs(ri)
            ans = math.factorial(len(le) + len(ri)) // (math.factorial(len(le)) * math.factorial(len(ri)))
            return ml * mr * ans
        
        return (dfs(nums) - 1) % (10**9 + 7)
