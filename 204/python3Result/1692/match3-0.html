class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        import operator as op
        from functools import reduce
        def nw(nums):
            def ncr(n, r):
                r = min(r, n-r)
                numer = reduce(op.mul, range(n, n-r, -1), 1)
                denom = reduce(op.mul, range(1, r+1), 1)
                return numer // denom  # or / in Python 2
            if len(nums)<=2:
                return 1
            i=nums[0]
            al=[]
            bl=[]
            for j in range(1,len(nums)):
                if nums[j]<i:
                    al.append(nums[j])
                else:
                    bl.append(nums[j])
            a=nw(al)
            #print(al,bl)
            b=nw(bl)
            #print(a,b)
            return ncr(len(al)+len(bl),len(al))*a*b%1000000007
        return nw(nums)-1
