import math


class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def count(a,b):
            if b == 0:
                return 1
            f = math.factorial
            return f(a) // f(b) // f(a-b)
        
        n = len(nums)
        if n == 1:
            return 0
        def dfs(x):
            if not x:
                return 1
            
            start = x[0]
            left,right = [], []
            for j in x[1:]:
                if j < start:
                    left.append(j)
                else:
                    right.append(j)
            m,n = len(left), len(right)
            return count(m+n,n) * dfs(left) * dfs(right)
        return (dfs(nums)-1) % (10**9+7)
