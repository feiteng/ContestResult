class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def find(nums):
            n = len(nums)
            if n == 0 or len(nums) == 1:
                return 1
            else:
                left, right 
= [], []
                for i in range(1, len(nums)):
                    if nums[i] < nums[0]:
                        left.append(nums[i])
                    else:
                        right.append(nums[i])
                return math.factorial(n - 1) // (math.factorial(len(left)) * math.factorial(len(right))) * find(left) * find(right)

        return (find(nums) - 1) % (10 ** 9 + 7)
