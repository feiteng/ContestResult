class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def nChoosem(n, m):
            return math.factorial(n) // (math.factorial(m) * math
.factorial(n - m))
        
        def helper(a):
            if len(a) <= 1:
                return 1
            left, right = [], []
            for i in range(1, len(a)):
                if a[i] < a[0]:
                    left.append(a[i])
                else:
                    right.append(a[i])
            lres = helper(left)
            rres = helper(right)
            if len(left) == 0:
                return rres
            elif len(right) == 0:
                return lres
            else:
                holes = len(left) + len(right)
                fills = len(right)
                fillpatterns = nChoosem(holes, fills) 
                return (lres * rres * fillpatterns) % (10 ** 9 + 7)
        return (helper(nums) - 1) % (10 ** 9 + 7)
