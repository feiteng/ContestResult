class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = self.right = None

def add(node, x):
    if x < node.val:
        if node.left is None:
            node.left = TreeNode(x)
        else:
            add(node.left, x)
    elif x > node.val:
        if node.right is None:
            node.right = TreeNode(x)
        else:
            add(node.right, x)


class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def nCr(n,r):
            f = math.factorial
            return f(n) // f(r) // f(n-r)

        r 
= TreeNode(nums[0])
        for x in nums[1:]:
            add(r, x)

        def cal(node):
            if node is None: return 0, 1
            node_num_l, resl = cal(node.left)
            node_num_r, resr = cal(node.right)
            node_num = node_num_l + node_num_r + 1
            res = resl*resr*int(nCr(node_num_l + node_num_r, node_num_l))
            return node_num, res
        return (cal(r)[1] - 1) % (10**9 + 7)
