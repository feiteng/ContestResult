class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        @functools.lru_cache(None)
        def fact(a):
            if a == 0:
                return 1
            return a * fact(a-1)
        
        @functools.lru_cache(None)
        def C(a, b):
            return fact(a+b) // (fact(a) * fact(b))
        
        D = 10 ** 9 + 7
        
        def recurse(l):
            if len(l) <= 1:
                return 1
            
            mid = l[0]
            left, right 
= [], []
            
            for n in l:
                if n < mid:
                    left.append(n)
                elif n > mid:
                    right.append(n)
                    
            return (recurse(left) % D) * (recurse(right) % D) * (C(len(left), len(right)) % D) % D
        
        return (recurse(nums) - 1) % D
