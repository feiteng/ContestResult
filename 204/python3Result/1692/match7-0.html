class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        mod = 10 ** 9 + 7
        def f(nums):
            if len(nums) <= 1:
                return 1
            root = nums[0]
            left = [n for n in nums if n < root]
            right = [n for n in nums if n > root]
            left_ways = f(left)
            right_ways = f(right)
            return left_ways * right_ways * math.comb(len(nums) - 1, len(left)) % mod
        return (f(nums) - 1) % mod
