class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        result = self.step_count(nums)
        
        return (result - 1) % (10**9 + 7)
        
        
    def step_count(self, ary: List[int]) -> int:
        if len(ary) <= 2:
            return 1
        head = ary[0]
        left = []
        right = []
        
        for val in ary[1:]:
            if val < head:
                left.append(val)
            else:
                right.append(val)
        left_step_count = self.step_count(left)
        right_step_count = self.step_count(right)
        
        len_left = len(left)
        len_right = len(right)
        
        step_count = (factorial(len_left + len_right) // (factorial(len_left) * factorial(len_right))) % (10**9 + 7)
        step_count = (step_count * left_step_count * right_step_count) % (10**9 + 7)
        return step_count
