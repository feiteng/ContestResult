class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        
        def rec(m, n):
            return math.factorial(n)//(math.factorial(n-m)*math.factorial(m))
        
        
        def count(nums):
            if len(nums) == 1 or len
(nums) == 0: return 1
            left, right = [], []
            for i in range(1, len(nums)):
                if nums[i] > nums[0]:
                    right.append(nums[i])
                else:
                    left.append(nums[i])
            
            return count(left)*count(right)* rec(len(left), len(left)+len(right))
        
        mod = 10**9+7
        return (count(nums)-1) % mod
