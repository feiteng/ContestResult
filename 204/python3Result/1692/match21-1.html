class Node:
    def __init__(self,x):
        self.val=x
        self.left=None
        self.right=None
class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        root=Node(nums[0])
        if len(nums)==1:
            return 0
        
for num in nums[1:]:
            cur=root
            while True:
                if num<cur.val:
                    if cur.left:
                        cur=cur.left
                    else:
                        cur.left=Node(num)
                        break
                elif num>cur.val:
                    
if cur.right:
                        cur=cur.right
                    else:
                        cur.right=Node(num)
                        break
        @lru_cache(None)
        def factorial(x):
            cur=1
            for i in range(1,x+1):
                cur*=i
            return cur
                
        
        def build(node,l,r):
            if node.left and node.right:
                left=build(node.left,l,node.val-1)
                right=build(node.right,node.val+1,r)
                return left*right*(factorial(r-l)//(factorial(node.val-l)*factorial(r-node.val)))
            elif l==r:
                return 1
            elif node.left==None:
                return build(node.right,node.val+1,r)
            elif node.right==None:
                return build(node.left,l,node.val-1)
        return (build(root,1,len(nums))-1)%((10**9)+ 7)
