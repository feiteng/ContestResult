class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        @lru_cache(None)
        def c(x, y):
            start = x + y
            res = 1
            for i in range(1, x + 1):
                res = res * (y + i) // i
            return res
        
        
def cal(arr):
            if len(arr) <= 1:
                return 1
            n = arr[0]
            left = []
            right = []
            for i in arr[1:]:
                if i < n:
                    left.append(i)
                else:
                    right.append(i)
            return cal(left) * cal(right) * c(len(left), len(right))
        
        MOD = 10**9 + 7
        return (cal(nums) - 1) % MOD
