class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        def _numOfWays(nums):
            if len(nums) == 0:
                return 1
            root = nums[0]
            left, right = [], []
            
            for n in nums[1:]:
                if n < root:
                    left.append(n)
                else:
                    right.append(n)
            
            return (_numOfWays(left) * _numOfWays(right) * math.comb(len(left)+len(right), len(left)))%(10**9 + 7)
        return (_numOfWays(nums)-1)%(10**9 + 7)
