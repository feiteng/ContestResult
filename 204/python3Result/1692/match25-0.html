class Solution:
    
    def numOfWays(self, nums: List[int]) -> int:
        def helper(arr):
            if len(arr) <= 1:
                return 1
            gt = [num for num in arr if num > arr[0]]
            lt = [num for num in arr if num < arr[0]]
            return (math.comb(len(arr) - 1, len(gt)) * helper(gt) * helper(lt))
        
        
        return (helper(nums) - 1) % 1_000_000_007
