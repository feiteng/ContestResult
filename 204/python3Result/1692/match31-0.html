class Solution:
    def getFactorial(self, a):
        if a not in self.factor:
            self.factor[a]=math.factorial(a)
        return self.factor[a]

    def helper(self, nums: List[int]) -> int:
        if len(nums)==0:
            return 0
        if len(nums)<=2:
            return 1
        root=nums[0]
        l=[]
        r=[]
        for i in range(1, len(nums)):
            if nums[i]<root:
                l.append(nums[i])
            else:
                r.append(nums[i])
        self.factor={}
        lnum=self.helper(l)
        rnum=self.helper(r)
        print(l,r)
        print(lnum, rnum)
        if lnum and rnum:
            n=self.getFactorial(len(nums)-1)
            a=self.getFactorial(len(l))
            b=self.getFactorial(len(r))
            return n//(a*b) * lnum * rnum
        else:
            return lnum if lnum else rnum
    
    def numOfWays(self, nums: List[int]) -> int:
        return (self.helper(nums)-1)%(10**9+7)
