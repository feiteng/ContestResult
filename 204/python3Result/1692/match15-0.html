class Solution:
    def numOfWays(self, nums: List[int]) -> int:
        @lru_cache(None)
        def I(n):
            if n==0:
                return 1
            return n*I(n-1)
        def C(n,m):
            return I(n)//I(m)//I(n-m)
        def recur(ls):
            if not ls:
                return 1
            mid = ls.pop()
            L,R = [],[]
            for i in ls:
                if i<mid:
                    L.append(i)
                else:
                    R.append(i)
            return C(len(L)+len(R),len(L))*recur(L)*recur(R)
        nums.reverse()
        return (recur(nums)-1)%(10**9+7)
