class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])
        def check(grid):
            seen = set()
            def dfs(i, j):
                if (i, j) in seen:
                    return
                seen.add((i, j))

                for x, y in ((i+1, j), (i-1, j), (i, j+1), (i, j-1)):
                    if 0 <= x < n and 0 <= y < m and grid[x][y] == 1:
                        dfs(x, y)
        
            sum_values = sum([sum(row) for row in grid])
            if sum_values == 0:
                return True
        
            for i in range(n):
                k = 0
                for j in range(m):
                    if grid[i][j] == 1:
                        dfs(i, j)
                        k = j
                        break
                if grid[i][k] == 1:
                    break
                
            if len(seen) != sum_values:
                return True
            
return False
        
        if check(grid):
            return 0
        
        for i in range(n):
            for j in range(m):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if check(grid):
                        return 1
                    grid[i][j] = 1
                    
                
        return 2
