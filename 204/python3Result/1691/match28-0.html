class Solution:    
    def dfs(self, grid, i, j, visited):
        m = len(grid)
        n = len(grid[0])
        if (i >= m) or (i < 0) or (j >= n) or (j < 0) or (grid[i][j] == 0) or (visited[i][j]):
            return
        visited[i][j] = True
        self.dfs(grid, i - 1, j, visited)
        self.dfs(grid, i + 1, j, visited)
        self.dfs(grid, i, j - 1, visited)
        self.dfs(grid, i, j + 1, visited)
    
    def numIslands(self, grid: List[List[str]]) -> int:
        if (not grid) or (not grid[0]):
            return 0
        m = len(grid)
        n = len(grid[0])
        visited = [[False] * n for i in range(m)]
        res = 0
        for i in range(m):
            for j in range(n):
                if (grid[i][j] == 0) or visited[i][j]:
                    continue
                self.dfs(grid, i, j, visited)
                res += 1
        return res
    
    def minDays(self, grid: List[List[int]]) -> int:
        noi = self.numIslands(grid)
        if noi > 1:
            return 0
        m = len(grid)
        n = len(grid[0])
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    continue
                grid[i][j] = 0
                noi = self.numIslands(grid)
                # print(i, j, noi)
                if noi != 1:
                    return 1
                grid[i][j] = 1
        
        return 2
