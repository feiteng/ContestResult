class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        def count_islands():
            visited = [[False for _ in range(n)] for _ in range(m)]
            def dfs(i, j):
                if not 0 <= i < m or not 0 <= j < n:
                    return
                elif grid[i][j] == 0:
                    return
                elif visited[i][j]:
                    return
                visited[i][j] = True
                
                for di, dj in dirs:
                    dfs(i + di, j + dj)
                    
            res = 0
            for i in range(m):
                for j in range(n):
                    if grid[i][j] == 1 and not visited[i][j]:
                        res += 1
                        dfs(i, j)
                        
            return res
        
        islands = count_islands()
        if islands != 1:
            return 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if count_islands() != 1:
                        return 1
                    grid[i][j] = 1
                    
        return 2
