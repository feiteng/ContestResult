class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        def count(grid):
            visited = [[False for _ in range(N)] for _ in range(M)]
            def dfs(r, c):
                if visited[r][c]:
                    return 0
                visited[r][c] = True
                area = 1
                for dr, dc in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < M and 0 <= nc < N and grid[nr][nc] == 1:
                        area += dfs(nr, nc)
                return area
            for r in range(M):
                for c in range(N):
                    if grid[r][c] == 1:
                        return dfs(r, c)
            return -1
        cnt = sum(grid[r][c] for r in range(M) for c in range(N))
        if cnt == 0 or count(grid) < cnt:
            return 0
        for r in range(M):
            for c in range(N):
                if grid[r][c] == 1:
                    grid[r][c] = 0
                    if count(grid) < cnt - 1:
                        return 1
                    grid[r][c] = 1
        return 2
