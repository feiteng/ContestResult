class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        rows = len(grid)
        cols = len(grid[0])
        
        visited = [[False] * cols for _ in range(rows)]
        
        def all_visited():
            for x in range(rows):
                for y in range(cols):
                    if grid[x][y] == 1 and not visited[x][y]:
                        return False
            return True        

        
def go(x, y):
            visited[x][y] = True
            
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                
                if 0 <= nx < rows and 0 <= ny < cols:
                    if grid[nx][ny] == 1 and not visited[nx][ny]:
                        go(nx, ny)
        
        done = False
        for x in range(rows):
            for y in range(cols):
                if not done and grid[x][y] == 1:
                    done = True
                    go(x, y)
        
        
if not all_visited():
            return 0
        
        for x in range(rows):
            for y in range(cols):
                if grid[x][y] == 1:
                    grid[x][y] = 0
                    visited = [[False] * cols for _ in range(rows)]
                    
                    for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                        nx, ny = x + dx, y + dy

                        if 0 <= nx < rows and 0 <= ny < cols and grid[nx][ny] == 1:
                            go(nx, ny)
                            break
                    
                    
if not all_visited():
                        return 1
                    grid[x][y] = 1
        
        return 2
