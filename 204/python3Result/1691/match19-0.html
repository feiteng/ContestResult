class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        # check if connected
        
        
        s = sum([sum(l) for l in grid])
        if s == 0:
            return 0
        
        def is_connected(g, s):
            def count_patch(x, y):
                q = collections.deque([(x,y)])
                visited = set([(x,y)])
                while q:
                    i, j = q.popleft()
                    for ii, jj in [(i+1,j),(i,j+1),(i-1,j),(i,j-1)]:
                        if 0 <= ii < len(g) and 0 <= jj < len(g[0]) and g[ii][jj] == 1 and (ii,jj) not in visited:
                            visited.add((ii,jj))
                            q.append((ii,jj))
                            
                return len(visited)

            for i in range(len(g)):
                for j in range(len(g[0])):
                    if g[i][j] == 1:
                        return count_patch(i, j) == s
        
        if not is_connected(grid, s):
            return 0
        
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if not is_connected(grid, s-1):
                        return 1
                    grid[i][j] = 1
        
        return 2
