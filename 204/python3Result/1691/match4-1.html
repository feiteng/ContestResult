class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

        def count():
            visited = [[False for j in range(n)] for i in range(m)]
            def dfs(x, y):
                visited[x][y] = True
                for d in directions:
                    xn, yn = x + d[0], y + d[1]
                    
if 0 <= xn <m and 0 <= yn < n and not visited[xn][yn] and grid[xn][yn] == 1:
                        dfs(xn, yn)

            ans = 0
            for i in range(m):
                for j in range(n):
                    if not visited[i][j] and grid[i][j] == 1:
                        ans += 1
                        dfs(i, j)

            return ans
        if count() != 1:
            return 0

        for i in range(m * n):
            x, y = i // n, i % n
            
if grid[x][y] == 1:
                grid[x][y] = 0
                if count() != 1:
                    return 1
                grid[x][y] = 1

        for i in range(m * n):
            x1, y1 = i // n, i % n
            if grid[x1][y1] == 1:
                grid[x1][y1] = 0

                for j in range(i + 1, m * n):
                    x2, y2 = j // n, j % n
                    if grid[x2][y2] == 1:
                        grid[x2][y2] = 0
                        if count() != 1:
                            return 2
                        grid[x2][y2] = 1
                
                grid[x1][y1] = 1
        
        return 3
