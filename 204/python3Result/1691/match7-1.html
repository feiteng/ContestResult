import copy
class Solution:
    def dfs(self, a, i, j):
        a[i][j] = 0
        if i > 0 and a[i - 1][j] == 1:
            self.dfs(a, i - 1, j)
        if i < len(a) - 1 and a[i + 1][j] == 1:
            self.dfs(a, i + 1, j)
        if j > 0 and a[i][j - 1] == 1:
            self.dfs(a, i, j - 1)
        if j < len(a[0]) - 1 and a[i][j + 1] == 1:
            self.dfs(a
, i, j + 1)

    def check(self, grid):
        a = copy.deepcopy(grid)
        for i in range(len(a)):
            for j in range(len(a[0])):
                if a[i][j] == 1:
                    self.dfs(a, i, j)
                    for ii in range(len(a)):
                        for jj in range(len(a[0])):
                            if a[ii][jj] == 1:
                                return False
                    return True
        return True

    def minDays(self, grid: List[List[int]]) -> int:
        min = 0
        
if not self.check(grid):
            return 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if not self.check(grid):
                        return 1
                    grid[i][j] = 1
        return 2
