class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        # 最多只需要2天!!!
        rows, cols = len(grid), len(grid[0])

        def isValid():
            v = set()
            cnt = 0
            for sr in range(rows):
                for sc in range(cols):
                    if grid[sr][sc] == 1 and (sr, sc) not in v:
                        cnt += 1
                        q = [(sr, sc)]
                        v.add((sr, sc))
                        for x in q:
                            r, c = x
                            for rr, cc in ((r + 1, c), (r - 1, c), (r, c + 1),
                                           (r, c - 1)):
                                if 0 <= rr < rows and 0 <= cc < cols and grid[
                                        rr][cc] == 1 and (rr, cc) not in v:
                                    v.add((rr, cc))
                                    q.append((rr, cc))
            return cnt == 0 or cnt >= 2

        # def calHq():

        #                             cc] == 1:

        if isValid():
            return 0
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == 1:
                    grid[r][c] = 0
                    if isValid():
                        return 1
                    grid[r][c] = 1
