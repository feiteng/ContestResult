class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        
        n = len(grid)
        m = len(grid[0])
        
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        def isCell(t):
            if t[0] >= 0 and t[0] < n and t[1] >= 0 and t[1] < m:
                return True
            return False
        
        
        
        def getv(g, i, j):
            v = set((i, j))
            q = collections.deque([(i, j)])
            while q:
                thisi, thisj = q.popleft()
                for dd in directions:
                    ii = thisi + dd[0]
                    jj = thisj + dd[1]
                    if isCell((ii, jj)):
                        if g[ii][jj] == 1 and (ii, jj) not in v:
                            v.add((ii, jj))
                            q.append((ii, jj))
            return v
        
        def isConnected(g):
            v = set()
            for i in range(n):
                for j in range(m):
                    if g[i][j] == 1:
                        if (i, j) not in v:
                            if len(v) > 0:
                                return False
                            else:
                                v = getv(g, i, j)
            return True

return 0

            for j in range(m):
