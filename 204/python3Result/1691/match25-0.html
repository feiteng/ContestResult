class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        
        def c():
            directions = ((1,0),(0,1),(-1,0),(0,-1))
            def dfs(i, j):
                for x, y in directions:
                    if (x+i, j+y) not in visited and 0 <= x+i <m and 0 <= y+j <n and grid[x+i][y+j] == 1:
                        visited.add((x+i, y+j))
                        dfs(x+i, y+j)
                        
            visited = set()
            count = 0
            for i in range(m):
                for j in range(n):
                    if (i, j) not in visited and grid[i][j] == 1:
                        count += 1
                        visited.add((i, j))
                        dfs(i, j)
            return count
        
        
        m, n = len(grid), len(grid[0])
        
        count = 0
        
        if c() != 1:  return 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if c() != 1:
                        return 1
                    grid[i][j] = 1
        return 2
