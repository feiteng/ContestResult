class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        
        def get2Island():
            seen = [[False]*n for _ in range(m)]
            
            def dfs(i, j):
                seen[i][j] = True
                for di, dj in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
                    ni, nj = i+di, j+dj
                    if 0<=ni<m and 0<=nj<n and grid[ni][nj] and not seen[ni][nj]:
                        dfs(ni, nj)
                return

            ans = 0
            for i in range(m):
                for j in range(n):
                    if not seen[i][j] and grid[i][j]:
                        ans += 1
                        grid[i][j] = 0
                        dfs(i, j)
                        grid[i][j] = 1
            return ans 
        
        if get2Island() >= 2:
            return 0
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    grid[i][j] = 0
                    if get2Island() >= 2:
                        return 1
                    grid[i][j] = 1
        return 2
