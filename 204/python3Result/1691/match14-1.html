class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])
        def disconnected():
            visited = [[False] * m for _ in range(n)]
            island = 0
            def dfs(i, j):
                if 0 <= i < n and 0 <= j < m and grid[i][j] and not visited[i][j]:
                    visited[i][j] = True
                    dfs(i + 1, j)
                    dfs(i - 1, j)
                    dfs(i, j + 1)
                    dfs(i, j - 1)
            for i in range(n):
                for j in range(m):
                    if grid[i][j] and not visited[i][j]:
                        island += 1
                        if island > 1:
                            return True
                        dfs(i, j)
            return not island
        
if disconnected():
            return 0
        for i in range(n):
            for j in range(m):
                if grid[i][j]:
                    grid[i][j] = 0
                    if disconnected():
                        return 1
                    grid[i][j] = 1
        return 2
