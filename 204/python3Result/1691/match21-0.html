class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])
        cnt = 0
        dirs = [0, -1, 0, 1, 0]
        vis = [[False] * m for _ in range(n)]
        def dfs(r: int, c: int):
            vis[r][c] = True
            for i in range(4):
                nr, nc = r + dirs[i], c + dirs[i + 1]
                if 0 <= nr < n and 0 <= nc < m:
                    if grid[nr][nc] and not vis[nr][nc]: dfs(nr, nc)
        def islands() -> int:                  
            cnt = 0
            for i in range(n):
                for j in range(m):
                    if grid[i][j] and not vis[i][j]:
                        cnt += 1
                        dfs(i, j)
            return cnt               
        if islands() != 1:              
            return 0
        for i in range(n):  
            for j in range(m):
                if grid[i][j]:
                    grid[i][j] = 0
                    vis = [[False] * m for _ in range(n)]
                    if islands() != 1:
                        return 1
                    grid[i][j] = 1
        return 2
