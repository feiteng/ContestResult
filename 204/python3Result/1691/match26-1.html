from typing import List


class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        # size of the grid
        N = len(grid)
        M = len(grid[0])

        # the total number of land
        total = sum([sum(row) for row in grid])

        # check if the grid is connected by `count` lands
        def isConnected(count: int) -> bool:
            for x in range(N):
                for y in range(M):
                    # for the first land
                    if grid[x][y] == 1:
                        # use bfs to visit the graph
                        visited = set()
                        visited.add((x, y))
                        q = [(x, y)]
                        while q:
                            i, j = q.pop()
                            for ni, nj in [(i - 1, j), (i + 1, j), (i, j - 1), (i, j + 1)]:
                                if 0 <= ni < N and 0 <= nj < M and ((
ni, nj) not in visited) and grid[ni][nj] == 1:
                                    visited.add((ni, nj))
                                    q.append((ni, nj))
                        return len(visited) == count
            return False

        
if not isConnected(total):
            return 0
        for px in range(N):
            for py in range(M):
                if grid[px][py] == 1:
                    # assume we remove this land
                    grid[px][py] = 0
                    if not isConnected(total - 1):
                        return 1
                    grid[px][py] = 1
        return 2
