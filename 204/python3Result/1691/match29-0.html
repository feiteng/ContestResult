class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        r = len(grid)
        c = len(grid[0])
        
        def check(grid):
            g = [i.copy() for i in grid]
            r = len(g)
            c = len(g[0])
            group = 0

            def helperr(i,j):
                nonlocal r,c,g
                if i < 0 or i >= r or j < 0 or j >= c:
                    return
                if g[i][j] == 0:
                    return 
                g[i][j] = 0
                helperr(i-1,j)
                helperr(i+1,j)
                helperr(i,j-1)
                helperr(i,j+1)
            for i in range(r):
                for j in range(c):
                    if g[i][j] == 1:
                        helperr(i,j)
                        group += 1
            if group == 1:
                return False
            return True
        
        if check(grid):
            return 0
        
        for i in range(r):
            for j in range(c):
                if grid[i][j] == 1:
                    g = [i.copy() for i in grid]
                    g[i][j] = 0
                    if check(g):
                        return 1
        return 2
