from collections import deque
class Solution:
    def bfs(self, grid, si, sj):
        grid[si][sj] = 2
        q = deque([(si, sj)])
        while len(q) > 0:
            length = len(q)
            for _ in range(length):
                i, j = q.popleft()
                for di, dj in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                    if 0 <= i + di < len(grid) and 0 <= j + dj < len(grid[0]) and grid[i+di][j+dj] == 1:
                        grid[i+di][j+dj] = 2
                        q.append((i+di, j+dj))
        
    def count_island(self, grid):
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == 1:
                    self.bfs(grid, i, j)
                    count += 1
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == 2:
                    grid[i][j] = 1
        return count

    def minDays(self, grid: List[List[int]]) -> int:
        if self.count_island(grid) != 1:
            return 0
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    if self.count_island(grid) != 1:
                        return 1
                    grid[i][j] = 1
        return 2
