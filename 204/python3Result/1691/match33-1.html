class Solution:
    def minDays(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])
        def numI(grid):
            vis 
= set()
            cnt = 0
            for i in range(n):
                for j in range(m):
                    if grid[i][j] == 1 and (i, j) not in vis:
                        cnt += 1
                        que = collections.deque([(i, j)])
                        vis.add((i, j))
                        while que:
                            x, y = que.pop()
                            if x + 1 < n and grid[x + 1][y] and (x + 1, y) not in vis: 
                                que.append((x + 1, y)) 
                                vis.add((x + 1, y))
                            if x - 1 >= 0 and grid[x - 1][y] and (x - 1, y) not in vis : 
                                que.append((x - 1, y))
                                vis.add((x - 1, y))
                            if y + 1 < m and grid[x][y + 1] and (x, y + 1) not in vis: 
                                que.append((x, y + 1))
                                vis.add((x, y + 1))
                            if y - 1 >= 0 and grid[x][y - 1] and (x, y - 1) not in vis: 
                                que.append((x, y - 1))
                                vis.add((x, y - 1))
            return cnt
        
if numI(grid) > 1: return 0
        for i in range(n):
            for j in range(m):
                if grid[i][j] == 1:
                    grid[i][j] = 0
                    # print(i, j)
                    if numI(grid) > 1:
                        return 1
                    grid[i][j] = 1
        return 2
