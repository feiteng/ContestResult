class Solution:
    def dfs(self, R, C, grid, vis, r, c):
        vis[r][c] = True
        count 
= 1
        for nr, nc in ((r + 1, c), (r - 1, c), (r, c + 1), (r, c - 1)):
            if 0 <= nr < R and 0 <= nc < C and grid[nr][nc] == 1 and not vis[nr][nc]:
                count += self.dfs(R, C, grid, vis, nr, nc)
        return count
    
    def check(self, N, R, C, grid):
        for r in range(R):
            for c in range(C):
                if grid[r][c] == 1:
                    vis = [[False] * C for _ in range(R)]
                    count = self.dfs(R, C, grid, vis, r, c)
                    return count != N
    
    def minDays(self, grid: List[List[int]]) -> int:
        R = len(grid)
        C = len(grid[0])
        
        ones = 0
        for r in range(R):
            for c in range(C):
                if grid[r][c] == 1:
                    ones += 1
        
        if self.check
(ones, R, C, grid):
            return 0
        
        for r in range(R):
            for c in range(C):
                if grid[r][c] == 1:
                    grid[r][c] = 0
                    if self.check
(ones - 1, R, C, grid):
                        return 1
                    grid[r][c] = 1
        return 2
