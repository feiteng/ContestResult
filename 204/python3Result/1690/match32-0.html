class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        result = 0
        cur_num = 0
        cur_len = 0
        for n in nums:
            if n < 0:
                cur_num += 1
                cur_len += 1
            elif n == 0:
                cur_num = 0
                cur_len = 0
            else:
                cur_len += 1
            if cur_num % 2 == 0:
                result = max(result, cur_len)
        cur_num = 0
        cur_len = 0
        for n in reversed(nums):
            if n < 0:
                cur_num += 1
                cur_len += 1
            elif n == 0:
                cur_num = 0
                cur_len = 0
            else:
                cur_len += 1
            if cur_num % 2 == 0:
                result = max(result, cur_len)
        return result
