class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        countPos = [0] * (len(nums)+1)
        countNeg = [0] *  (len(nums)+1)
        
        
        ans = 0
        for i in range(0, len(nums)):
            if nums[i] < 0:
                countNeg[i+1] = countPos[i] + 1
                if countNeg[i] > 0:
                    countPos[i+1] = countNeg[i] + 1
            if nums[i] > 0:
                countPos[i+1] = countPos[i] + 1
                if countNeg[i] > 0:
                    countNeg[i+1] = countNeg[i] + 1
            ans = max(ans, countPos[i+1])
        return ans
