class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        best = 0
        nums = [0] + nums + [0]
        n = len(nums)
        pos = [0]*n
        neg = [0]*n
        for i,num in enumerate(nums):
            if num>0:
                pos[i] = pos[i-1]+1
                neg[i] = neg[i-1]+1 if neg[i-1] else 0
            elif num<0:
                pos[i] = neg[i-1]+1 if neg[i-1] else 0
                neg[i] = pos[i-1]+1
        # print("pos:", pos)
        # print("neg:", neg)
        return max(pos)
