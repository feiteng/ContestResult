class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        seen={1:-1} # last seen neg,pos
        curr=1
        ret=0
        for i,num in enumerate(nums):
            if num==0:
                seen={1:i}
                curr=1
            else:
                curr*=(1 if num>0 else -1)
                if curr in seen:
                    ret=max(ret,i-seen[curr])
                if curr not in seen:
                    seen[curr]=i
        return ret
