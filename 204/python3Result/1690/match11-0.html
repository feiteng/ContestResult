class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        ini = 0
        ne, po = 0,0
        for nu in nums:
            if nu < 0:
                ne, po = po+1,ne+1 if ne > 0 else 0
            elif nu > 0:
                po, ne = po+1, ne+1 if ne > 0 else 0
            else:
                ne, po = 0,0
            ini = max(ini, po)
        return ini
