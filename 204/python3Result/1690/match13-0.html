class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        lp = ln = 0
        max_l = 0
        for i, num in enumerate(nums):
            if num == 0:
                lp = ln = 0
            elif num > 0:
                lp = lp + 1
                ln = ln + 1 if ln != 0 else 0
            else:
                tln, tlp = ln, lp
                ln = tlp + 1
                lp = tln + 1 if tln != 0 else 0
            # print(i, lp, ln)
            max_l = max(max_l, lp)
        return max_l
