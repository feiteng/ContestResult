class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        mp = {}
        mp[0] = -1
        cnt = 0
        res = 0
        for i, n in enumerate(nums):
            if n == 0:
                mp = {0: i}
                cnt = 0
            elif n < 0:
                cnt += 1
                cnt %= 2
            if cnt == 0:
                res = max(res, i - mp[0])
            elif 1 in mp:
                res = max(res, i - mp[1])
            else:
                mp[1] = i
        return res
