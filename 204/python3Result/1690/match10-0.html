class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        longpos = 0
        longneg = 0
        ans = 0
        for i in nums:
            if i == 0:
                longpos = 0
                longneg = 0
            elif i > 0:
                longpos += 1
                if longneg > 0:
                    longneg += 1
            else:
                op, on = longpos, longneg
                if on > 0:
                    longpos = on + 1
                else:
                    longpos = 0
                longneg = op + 1

            ans = max(ans, longpos)
            #print(i, longpos, longneg, ans)
        return ans
