class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        
        even_min = -1
        odd_min = float('inf')
        
        c = 0
        
        ans = 0
        
        for i, vi in enumerate(nums):
            
            if vi == 0:
                even_min = i
                odd_min = float('inf')
                c = 0
                continue
                
            if vi < 0:
                c += 1
                if odd_min == float('inf'):
                    odd_min = i
                
                
            if c&1:
                ans = max(ans, i - odd_min)
            else:
                ans = max(ans, i - even_min)
                
            
        return ans
