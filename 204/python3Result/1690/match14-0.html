class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        pos_len = 0
        neg_len = 0
        res = 0
        for num in nums:
            if num > 0:
                pos_len += 1
                neg_len = neg_len + 1 if neg_len else 0
            elif num == 0:
                neg_len = pos_len = 0
            else:
                pre_neg_len = neg_len
                neg_len = pos_len + 1
                pos_len = pre_neg_len + 1 if pre_neg_len else 0
            res = max(res, pos_len)
        
        return res
