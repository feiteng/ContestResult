class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        max_pos = [0] * len(nums)
        max_neg = [0] * len(nums)
        #ending with
        if nums[0] > 0:
            max_pos[0] = 1
            pass
        elif nums[0] < 0:
            max_neg[0] = 1
            pass
        #print(nums)
        #print("i = 0, max_pos = %d, max_neg = %d" % (max_pos[0], max_neg[0]))
        for i in range(1, len(nums)):
            if nums[i] > 0:
                max_pos[i] = max_pos[i-1] + 1 
                max_neg[i] = max_neg[i-1] + 1 if max_neg[i-1] > 0 else 0
                pass
            elif nums[i] < 0:
                max_pos[i] = max_neg[i-1] + 1 if max_neg[i-1] > 0 else 0
                max_neg[i] = max_pos[i-1] + 1
                pass
            #print("i = %d, max_pos = %d, max_neg = %d" % (i, max_pos[i], max_neg[i]))
            pass
        return max(max_pos)
