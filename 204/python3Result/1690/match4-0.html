class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        pos_res = 0
        neg_res = 0

        res = 0

        for num in nums:
            if num == 0:
                pos_res = 0
                neg_res = 0
            elif num > 0:
                pos_res += 1
                if neg_res > 0:
                    neg_res += 1
                else:
                    neg_res = 0
            elif num < 0:
                if neg_res == 0:
                    n_pos_res = 0
                else:
                    n_pos_res = neg_res + 1
                neg_res = pos_res + 1
                pos_res = n_pos_res
            
            res = max(res, pos_res)
            # print(num, pos_res, neg_res, res)
        
        return res
