class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        dp = [(0, 0)] * (len(nums) + 1)
        for i in range(len(nums)):
            if nums[i] == 0:
                dp[i] = (0, 0)
            elif nums[i] > 0:
                dp[i] = (dp[i-1][0] + 1, dp[i-1][1] + 1 if dp[i-1][1] > 0 else 0)
            else:
                dp[i] = (dp[i-1][1] + 1 if dp[i-1][1] > 0 else 0, dp[i-1][0] + 1)
        # print(dp)
        return max([x[0] for x in dp])
