class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        ans = 0
        last_zero, first_neg, last_neg = -1, None, None
        num_neg = 0
        
        nums.append(0)
        
        for i, n in enumerate(nums):
            
            if n == 0:
                
                if first_neg is None or num_neg % 2 == 0:
                    ans = max(ans, i - last_zero - 1)
                else:
                    ans = max(ans, i - first_neg - 1, last_neg - last_zero - 1)
                    
                last_zero = i
                first_neg, last_neg = None, None
                num_neg = 0
                
            elif n < 0:
                last_neg = i
                num_neg += 1
                if first_neg is None:
                    first_neg = i
        
        return ans
