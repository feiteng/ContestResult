class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        even_neg_idx = -1
        odd_neg_idx = None
        num_neg = 0
        ans = 0
        for i, num in enumerate(nums):
            if num == 0:
                num_neg = 0
                even_neg_idx 
= i
                odd_neg_idx = None
            elif num < 0:
                num_neg += 1
            if num_neg % 2 == 1:
                if odd_neg_idx is None:
                    odd_neg_idx = i
                else:
                    ans = max(ans, i - odd_neg_idx)
            else:
                ans = max(ans, i - even_neg_idx)
        return ans
