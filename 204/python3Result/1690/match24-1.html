class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        
        dp_p=[0 for _ in range(len(nums))]
        
        dp_n=[0 for _ in range(len(nums))]
        
        
        for i,num in enumerate(nums):
            if i==0:
                
                if num>0:
                    dp_p[i]=1
                    
                elif num<0:
                    dp_n[i]=1
                
                else:
                    pass
                
                continue
                
            
if num>0:
                
                dp_p[i]=1+dp_p[i-1]
                dp_n[i]=1+dp_n[i-1] if dp_n[i-1]>0 else 0
                
            elif num<0:
                
                dp_p[i]=1+dp_n[i-1] if dp_n[i-1]>0 else 0
                dp_n[i]=1+dp_p[i-1]
                
            
        return max(dp_p)
