class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        res = 0
        pos = 0
        neg = None
        
        cur = 1
        for i, num in enumerate(nums):
            if num == 0:
                cur = 1
                pos = i + 1
                neg = None
                continue
                
            if num < 0:
                cur *= -1
            if cur > 0:
                res = max(res, i - pos + 1)
                
            elif not (neg == None):
                res = max(res, i - neg)
            else:
                neg = i
                
        return res
