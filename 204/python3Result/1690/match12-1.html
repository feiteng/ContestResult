class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        l = 0
        p = [0]*len(nums)
        n = [0]*len(nums)
        i = 0
        if nums[0] > 0:
            p[0] = 1
        elif nums[0] < 0:
            n[0] = 1
        l = max(l, p[i])
        
for i in range(1,len(nums)):
            if nums[i] > 0:
                p[i] = p[i-1]+1
                if n[i-1] != 0:
                    n[i] = n[i-1]+1
            elif nums
[i] < 0:
                n[i] = p[i-1]+1
                if n[i-1] != 0:
                    p[i] = n[i-1]+1
            l = max(l, p[i])
        return l
