class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        n = len(nums)
        dpPos = [-1 for _ in range(n)]
        dpNeg = [-1 for _ in range(n)]
        res = 0
        for i in range(n):
            if i == 0:
                if nums[i] > 0:
                    dpPos[i] = 1
                elif nums[i] < 0:
                    dpNeg[i] = 1
            else:
                if nums[i] > 0:
                    dpPos[i] = 1 + dpPos[i - 1] if dpPos[i - 1] != -1 else 1
                    dpNeg[i] = 1 + dpNeg[i - 1] if dpNeg[i - 1] != -1 else -1
                elif nums[i] < 0:
                    dpPos[i] = 1 + dpNeg[i - 1] if dpNeg[i - 1] != -1 else -1
                    dpNeg[i] = 1 + dpPos[i - 1] if dpPos[i - 1] != -1 else 1
            if dpPos[i] != -1:
                res = max(res, dpPos[i])
        #print(dpPos, dpNeg)
        return res
