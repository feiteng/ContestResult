class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        pos = [-1, -2]
        s = c = 0
        ans = 0
        for i, n in enumerate(nums):
            if n == 0:
                pos = [i, -2]
                s = c = 0
            elif n < 0:
                c = (c + 1) % 2
            
            if c == 1 and pos[1] == -2:
                pos[1] = i
            ans = max(ans, i - pos[c])
        return ans
