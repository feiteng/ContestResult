class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        neg = False
        earliestNeg = -1
        lastZero = -1
                
        ret = 0
        
        for i,num in enumerate(nums):
            if num == 0:
                lastZero = i
                neg = False
                earliestNeg = -1
                continue
                
            if num < 0:
                neg = not neg
                if earliestNeg == -1:
                    earliestNeg = i

            if neg:
                ret = max(ret, i - earliestNeg)
            else:
                ret = max(ret, i - lastZero)

        return ret
