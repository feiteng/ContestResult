class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        status = {1:-1}
        res = 0
        curr = 1 # 1 means positive
        for i, n in enumerate(nums):
            if n == 0:
                status = {1:i}
                curr = 1
            else:
                curr ^= n < 0
                res = max(res, i - status.get(curr, i))
                if curr not in status:
                    status[curr] = i
        return res
