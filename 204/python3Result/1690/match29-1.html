class Solution:
    
    def getMaxLen(self, nums: List[int]) -> int:
        n = len(nums)
        pos = [0] * n
        neg = [0] * n
        if nums[0] > 0:
            pos[0] = 1
        elif nums[0] < 0:
            neg[0] = 1
        ans = pos
[0]
        for i in range(1, n):
            if nums[i-1] == 0:
                if nums[i] > 0:
                    pos[i] = 1
                elif nums[i] < 0:
                    neg[i] = 1
            else:
                if nums[i] > 0:
                    pos[i] = max(pos[i-1] + int(pos[i-1] > 0), 1)
                    neg[i] = neg[i-1] + int(neg[i-1] > 0)
                elif nums[i] < 0:
                    pos[i] = neg[i-1] + int(neg[i-1] > 0)
                    neg[i] = max(pos[i-1] + int(pos[i-1] > 0), 1)
            ans = max(ans, pos[i])
        return ans
