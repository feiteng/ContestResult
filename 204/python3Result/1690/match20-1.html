class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        longest = 0
        left = 0
        cur = 1
        first_negative = -1
        
        for i, val in enumerate(nums):
            if first_negative < left and val < 0:
                first_negative = i
            cur *= val
            if cur > 0:
                longest = max(longest, i-left+1)
            elif cur == 0:
                left = i+1
                cur = 1
            elif first_negative != -1:
                longest = max(longest, i-first_negative)
                
            
        return longest
