class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        cur_pos = 0
        cur_neg = 0
        res = 0
        for num in nums:
            if num == 0:
                cur_pos = 0
                cur_neg = 0
            elif num > 0:
                cur_pos += 1
                res = max(res, cur_pos)
                if cur_neg > 0:
                    cur_neg += 1
            else:
                temp = cur_pos
                if cur_neg > 0:
                    cur_pos = cur_neg + 1
                    res = max(res, cur_pos)
                else:
                    cur_pos = 0
                cur_neg = temp + 1
        return res
