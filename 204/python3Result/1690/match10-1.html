class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        res 
= 0
        pos, neg = 0, 0
        for n in nums:
            if not n:
                pos = neg = 0
            elif n > 0:
                pos += 1
                if neg > 0:
                    neg += 1
            else:
                neg_tmp = pos + 1
                if neg > 0:
                    pos = neg + 1
                else:
                    pos = 0
                neg = neg_tmp
                
            res = max(res, pos)
        return res
