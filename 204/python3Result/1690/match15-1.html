class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        minus,pos=0,0
        seen=False
        res=0
        for num in nums:
            if num==0:
                minus,pos=0,0
                seen=False
            elif num>0:
                pos+=1
                minus+=1
                res=max(res,pos)
            else:
                if seen:
                    minus,pos=pos+1,minus+1
                    res=max(res,pos)
                else:
                    seen=True
                    minus,pos=pos+1,0
        return res
