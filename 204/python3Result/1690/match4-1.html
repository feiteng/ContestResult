class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        curr_pos_len = 0
        curr_neg_len = 0
        res = 0
        for num in nums:
            if num == 0:
                curr_pos_len = 0
                curr_neg_len = 0
            elif num > 0:
                curr_pos_len += 1
                if curr_neg_len > 0:
                    curr_neg_len 
+= 1
            else:
                if curr_neg_len > 0:
                    tmp = curr_neg_len + 1
                else:
                    tmp = 0
                curr_neg_len = curr_pos_len + 1
                curr_pos_len = tmp
                
            res = max(res, curr_pos_len)
                
        return res
