class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        ans = 0
        even = -1
        odd = -1
        sign = 0
        for i, num in enumerate(nums):
            if num == 0:
                even = i
                odd = -1
                sign = 0
                continue
            sign ^= (num < 0)
            if not sign:
                ans = max(ans, i-even)
            else:
                if odd >= 0:
                    ans = max(ans, i-odd)
                else:
                    odd = i
            # print(sign, i, even, odd)
        return ans
