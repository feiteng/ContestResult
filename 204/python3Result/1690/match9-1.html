class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        ans = 0
        last = -1
        curr = 0
        last_neg = None
        for i, num in enumerate(nums):
            if num == 0:
                curr = 0
                last = i
                last_neg = None
            if num < 0:
                curr = curr ^ 1
            
if curr == 0:
                ans = max(ans, i - last)
            else:
                if last_neg is not None:
                    ans = max(ans, i - last_neg)
                else:
                    last_neg = i
        return ans
