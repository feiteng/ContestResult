class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        longest_pos = cur_nev = cur_pos = 0
        for n in nums:
            if n > 0:
                cur_pos += 1
                if cur_nev > 0:
                    cur_nev += 1
            elif n < 0:
                new_cur_nev = cur_pos + 1
                if cur_nev > 0:
                    new_cur_pos = cur_nev + 1
                else:
                    new_cur_pos = 0
                cur_pos, cur_nev = new_cur_pos, new_cur_nev
            else:
                cur_nev = cur_pos = 0
            longest_pos = max(longest_pos, cur_pos)
            # print(n, cur_pos, cur_nev)
        return longest_pos
