class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        mlen = 0
        p = 1
        firstNeg = -1
        lastzero = -1
        for i,n in enumerate(nums):
            p *= n
            if p==0:
                p=1
                firstNeg = -1
                lastzero = i
            if p<0:
                if firstNeg == -1:
                    firstNeg = i
                else:
                    mlen = max(mlen, i-firstNeg)
            else:
                mlen = max(mlen, i-lastzero)
        
        return mlen
