class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        po, ne, ans 
= 0, 0, 0
        for num in nums:
            if num == 0:
                po, ne = 0, 0
            elif num > 0:
                po, ne = po + 1, (ne + 1) if ne else 0
            else:
                po, ne = (ne + 1) if ne else 0, po + 1
            # print(num, po, ne)
            ans = max(ans, po)
        return ans
