class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        pos = -1
        neg = None
        now = True
        ans = 0
        for idx, num in enumerate(nums):
            if num == 0:
                now = True
                pos = idx
                neg = None
                continue
            if num < 0:
                now = not now
            if now:
                ans = max(ans, idx - pos)
            else:
                if neg is None:
                    neg = idx
                else:
                    ans = max(ans, idx - neg)
        return ans
