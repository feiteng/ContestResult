class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        
        pos, neg = -1, float('inf')
        cur = 1
        res = 0
        for i, num in enumerate(nums):
            
            if num == 0:
                pos, neg 
= i, float('inf')
                cur = 1
                continue
                
            if num < 0:
                cur = -cur
                if neg == float('inf'):
                    neg = i
                    
            if cur > 0:
                res = max(res, i - pos)
            else:
                res = max(res, i - neg)
        return res
