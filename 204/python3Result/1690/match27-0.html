class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        f = [[0] * 2 for _ in range(len(nums) + 1)]
        
        result = 0
        for i in range(1, len(nums) + 1):
            if nums[i - 1] > 0:
                f[i][0] = f[i - 1][0] + 1
                f[i][1] = f[i - 1][1] + 1 if f[i - 1][1] > 0 else 0
            elif nums[i - 1] < 0:
                f[i][0] = f[i - 1][1] + 1 if f[i - 1][1] > 0 else 0
                f[i][1] = f[i - 1][0] + 1
            else:
                f[i][0] = 0
                f[i][1] = 0
            result = max(result, f[i][0])
        #print(f)
            
        return result
