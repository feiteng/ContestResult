class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        n = len(nums)
        pos = [0] * n
        neg = [0] * n
        for i in range(n):
            if nums[i] > 0:
                pos[i] = (1 + pos[i - 1]) if i > 0 and pos[i - 1] else 1
                neg[i] = (1 + neg[i - 1]) if i > 0 and neg[i - 1] else 0
            elif nums[i] < 0:
                pos[i] = (1 + neg[i - 1]) if i > 0 and neg[i - 1] else 0
                neg[i] = (1 + pos[i - 1]) if i > 0 and pos[i - 1] else 1
        return max(pos)
