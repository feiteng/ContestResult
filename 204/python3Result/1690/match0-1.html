class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        locs = {0: -1}
        cur = 0
        ans = 0
        for i, n in enumerate(nums):
            if n == 0: 
                locs = {0: i}
                cur = 0
            else:
                cur ^= n < 0
                if cur in locs:
                    ans = max(ans, i - locs[cur])
                if cur not in locs: locs[cur] = i
        
        return ans
