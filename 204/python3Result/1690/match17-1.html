class Solution:
    def helper(self, nums):
        n 
= len(nums)
        dp = [0] * (n + 1)
        xp = [0] * (n + 1)
        for i in range(1, n + 1):
            if nums[i - 1] > 0:
                dp[i] = dp[i - 1] + 1 if dp[i - 1] > 0 else 1
                xp[i] = xp[i - 1] + 1 if xp[i - 1] > 0 else 0
            else:
                dp[i] 
= xp[i - 1] + 1 if xp[i - 1] > 0 else 0
                xp[i] = dp[i - 1] + 1 if dp[i - 1] > 0 else 1
        # print(nums, dp, xp)
        return max(dp)
                
    
    def getMaxLen(self, nums: List[int]) -> int:
        tmp = []
        res = []
        for x in nums:
            if x == 0:
                if len(tmp) > 0:
                    res.append(self.helper(tmp))
                tmp = []
            else:
                tmp.append(x)
        if len(tmp) > 0:
            res.append(self.helper(tmp))
        return max(res) if len(res) > 0 else 0
