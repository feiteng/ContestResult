class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        first_neg = None
        zero = -1
        mx = 0
        neg = 0
        for i,v in enumerate(nums):
            if v == 0:
                first_neg = None
                zero = i
                neg = 0
                continue
            if v < 0:
                neg += 1
                if first_neg is None:
                    first_neg = i
            if neg % 2 == 0:
                mx = max(mx, i-zero)
            elif first_neg != None:
                # print(i,v, first_neg)
                mx = max(mx, i-first_neg)
        return mx
