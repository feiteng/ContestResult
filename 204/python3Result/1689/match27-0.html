class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        for i in range(n):
            toBreak = False
            for j in range(1, k):
 #               print(i, j)
                for l in range(m):
#                    print(i, j, l, i+m*j+l, i+l)
                    if i+m*j+l >= n or arr[i+m*j+l] != arr[i+l]:
                        toBreak = True
                        break
#                print(i, j, toBreak)
                if toBreak:
                    break
            if not toBreak:
                return True
        return False
