class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        for i in range(len(arr) - m):
            pattern = arr[i:i+m]
            flag = False
            for cnt in range(k):
                if i + (cnt+1) * m > len(arr) or arr[i+cnt*m:i+(cnt+1)*m] != pattern:
                    flag = True
                    break
            if flag == False:
                return True
        return False
