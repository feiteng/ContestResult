class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        for i in range(n-m+1):      # 遍历所有模式串arr[i:i+m]
            flag = True
            for kk in range(1, k):     # 是不是重复k次
                if i+m*(kk+1) > n or not arr[(i+kk*m):(i+(kk+1)*m)] == arr[i:i+m]:
                    '''
                    print(i+kk*(m+1))
                    print(arr[(i+kk*m):(i+kk*(m+1))] == arr[i:i+m])
                    print(arr[(i+kk*m):(i+kk*(m+1))])
                    print(arr[i:i+m])
                    print('-----------')
                    '''
                    flag = False
                    break
            if flag:
                return True
        return False
