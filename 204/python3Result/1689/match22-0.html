class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        
        n = len(arr)
        if m * k > n:
            return False
        
        
        for i in range(n-m+1):
            pattern = arr[i:i+m]
            count = 0
            for j in range(i+m, n):
                if arr[j:j+m] == pattern:
                    count += 1
                    if count == k-1:
                        return True
                else:
                    break
        return False
