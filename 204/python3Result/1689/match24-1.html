class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr) - m * k
        for i in range(n + 1):
            for j in range(k - 1):
                if arr[i + j * m: i + j * m + m] != arr[i + j * m + m: i + j * m + m + m]:
                    break
            else:
                return True
        return False
