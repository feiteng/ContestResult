class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        for i in range(n - m * k + 1):
            res = True
            for j in range(i + m, i + m * k):
                if arr[j] != arr[(j - i) % m + i]:
                    res = False
                    break
            if res:
                return True
        return False
