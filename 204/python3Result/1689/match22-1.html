class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        if m*k>n:return False
        for i in range(n-m*k+1):
            mode = arr[i:i+m]
            cnt = 1
            for j in range(i+m,n,m):
                if mode == arr[j:j+m]:
                    cnt += 1
                    if cnt==k:return True
                else:break
        return False
