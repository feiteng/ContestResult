class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        for i in range(n - m * k + 1):
            flag = True
            for j in range(1, k):
                for t in range(m):
                    if arr[i + t] != arr[i + j * m + t]:
                        flag = False
                        break
                if not flag:
                    break
            if flag:
                return True
        return False
