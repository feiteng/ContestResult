class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        
        L = len(arr)
        for i, n in enumerate(arr):
            pattern = arr[i:i+m]
            count = 1
            for j in range(i+m, L, m):
                
                #print(pattern, arr[j:j+m], i, j)
                if pattern == arr[j:j+m]:
                    count += 1
                    if count == k:
                        return True
                else:
                    break
        return False
