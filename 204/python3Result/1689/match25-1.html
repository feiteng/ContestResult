class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        for start in range(len(arr) - m * k + 1):
            tmp = arr[start: start + m]
            if all(tmp == arr[start + i * m: start + (i+1) * m] for i in range(k)):
                return True
        return False
