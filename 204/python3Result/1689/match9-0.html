class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        for i in range(len(arr) - m + 1):
            p = arr[i:i+m]
            found = True
            for j in range(k):
                if arr[i + j * m: i + j * m + m] != p:
                    found = False
                    break
            if found:                    
                return True
        return False
