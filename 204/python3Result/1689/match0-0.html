class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        if len(arr) < m*k: 
            return False
        
        def valid(start): 
            for j in range(1, k): 
                if arr[start+j*m: start+j*m+m] != arr[start: start+m]: 
                    return False
            return True 
            
        for i in range(0, len(arr)-m*k+1): 
            if valid(i): 
                return True 
        return False
