class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n = len(arr)
        for i in range(n-m):
            md = arr[i:i+m]
            cnt = k-1
            # print(md)
            for j in range(i+m,n,m):
                if arr[j:j+m]==md:
                    cnt -= 1
                    # print(arr[j:j+m])
                    if not cnt:
                        return True
                else:
                    break
        return False
