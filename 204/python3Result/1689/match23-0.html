class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        l = len(arr)
        for i in range(l - m * k + 1):
            for j in range(m, m * k):
                if arr[i+j] != arr[i+j%m]: break
            else:
                return True
        return False
