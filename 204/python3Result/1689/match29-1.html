class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        tar_len = m * k
        n = len(arr)
        for start in range(n + 1 - tar_len):
            if arr[start:start + tar_len] == (arr[start:start + m] * k):
                return True
        return False
