class Solution:
    def containsPattern(self, arr: List[int], m: int, k: int) -> bool:
        n=len(arr)
        for i in range(n-k*m+1):
            succ = True
            for kk in range(1,k):
                for mm in range(m):
                    if arr[i+mm+kk*m] != arr[i+mm]:
                        succ =False
                        break
                if not succ:
                    break
            if succ:
                #print(i)
                return True
                
                
        return False
