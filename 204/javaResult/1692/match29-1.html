class Solution {
    static long mod=1000000007;
    static long[] fact;
    static long power(long x,long y,long p) 
    { 
        long res = 1; 
        x = x % p; 
        while (y > 0) { 
            if (y % 2 == 1) 
                res = (res * x) % p; 
            y = y >> 1;
            x = (x * x) % p; 
        
} 
        return res; 
    } 
    static long inv(long n) 
    { 
        return power((long)n,mod-2,mod); 
    } 
    static long func(int x,int y)
    {
        return (((fact[x+y]*inv(fact[x]))%mod)*inv(fact[y]))%mod;
    
}
    static long count(ArrayList<Integer> ar)
    {
        if(ar.size()<=1)
            return 1;
        ArrayList<Integer> arr=new ArrayList<>();
        ArrayList<Integer>left=new ArrayList<>();
        ArrayList<Integer> right=new ArrayList<>();
        arr.add(ar.get(0));
        for(int i=1;i<ar.size();i++)
        {
            arr.add(
ar.get(i));
            if(ar.get(i)<ar.get(0))
                left.add(ar.get(i));
            else
                right.add(ar.get(i));
        }
        // System.out.println(arr.get(0)+" "+left.size()+" "+right.size());
        long res=(((count(left)*count(right))%mod)*func(left.size(),right.size()))%mod;
        return res;
    }
    public int numOfWays(int[] nums) {
        ArrayList<Integer> arr=new ArrayList<>();
        for(int i=0;i<nums.length;i++)
        {
            arr.add(nums[i]);
        }
        // System.out.println(arr);
        fact=new long[1001];
        long res=1;
        fact[0]=1;
        for(int i=1;i<=1000;i++)
        {
            res=(res*i)%mod;
            fact[i]=res;
        }
        return (int)((count(arr)-1+mod)%mod);
    }
}
