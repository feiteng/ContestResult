class Node{
    Node left = null;
    Node right = null;
    int val; 
    int count = 0;
    Node(int val){
        this.val = val;
        this.count = 1;
    }
    void add(int newVal){
        count++;
        if(newVal > val){
            if(right == null){
                right = new Node(newVal);
            }else{
                right.add(newVal);
            }
        }
        else{
            if(left == null){
                left  = new Node(newVal);
            }else{
                left.add(newVal);
            }
        }
    }
    
}


class Solution {
    HashMap<Integer,Integer> map = new HashMap<Integer, Integer>();
    int mod = (int)Math.pow(10,9) + 7;
    
int dfs(int n,int i){
       // System.out.println(n + ": "+ i);
        if( n == i) return 1;
        if( i == 0) return 1;
        if(n ==1) return 1;
        
        if(map.containsKey(n*1001 + i)) return map.get(n*1001 + i);
        int a = dfs(n-1,i) % mod;
        int b = dfs(n-1,i-1) % mod;
        int result = (a+b) % mod;
        map.put(n*1001+i, result);
        
return result;
    }
    int helper(Node n){
        
        if(n.left == null && n.right == null) return 1;
        if(n.left == null) return helper(n.right);
        if(n.right == null) return helper(n.left);
        
long a = helper(n.left);
        long b = helper(n.right);
        long res = (a * b ) % mod;
        res = res * dfs(n.count-1, n.left.count);
        res %= mod;
        return (int)res;
    }
    public int numOfWays(int[] nums) {
        Node root = new Node(nums[0]);
        for(int i =1; i <nums.length; i++) root.add(nums[i]);
       // System.out.println(dfs(100,2));
        return (helper(root)-1) % mod;
    }
}
