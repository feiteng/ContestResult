class Solution {
    long MOD = 1000000007;
	long[] inv;
	long[] pro;
	long[] proInv;

	public long slove(List<Integer> list) {
		if (list.size() <= 1) {
			return 1;
		}


		int k = list.get(0);
		List<Integer> left = new ArrayList<Integer>();
		List<Integer> right = new ArrayList<Integer>();
		for (int i = 1; i < list.size(); i++) {
			int v = list.get(i);
			if (v < k) {
				left.add(v);
			} else {
				right.add(v);
			}
		}

		long leftCount = slove(left);
		long rightCount = slove(right);

		int sum1 = left.size();
		int sum2 = right.size();
		int sumall = sum1 + sum2;

		long ans1 = ((pro[sumall] * proInv[sum1] % MOD) * proInv[sum2]) % MOD;
		long ans2 = (leftCount * rightCount) % MOD;

		return (ans1 * ans2) % MOD;
	}

	public long powMod(long x, long n) {
		long result = 1;
		while (n > 0) {
			if (n % 2 == 1) {
				result = result * x % MOD;
			}
			x = x * x % MOD;
			n = n / 2;
		}
		return result;
	}

	public int numOfWays(int[] nums) {
		inv = new long[1010];
		pro = new long[1010];
		proInv = new long[1010];

		for (int i = 1; i <= 1005; i++) {
			inv[i] = powMod(i, MOD - 2);
		}

		pro[0] = pro[1] = 1;
		for (int i = 2; i <= 1005; i++) {
			pro[i] = (pro[i - 1] * i) % MOD;
		}

		proInv[0] = 1;
		for (int i = 1; i <= 1005; i++) {
			proInv[i] = powMod(pro[i], MOD - 2);
		}


		List<Integer> list = new ArrayList<Integer>();
		for (int i = 0; i < nums.length; i++) {
			list.add(nums[i]);
		}

		int ans = (int) ((slove(list) - 1 + MOD) % MOD);
		return ans;
	}
}
