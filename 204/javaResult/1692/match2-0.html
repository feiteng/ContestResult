import java.math.*;
class Solution {
    long mod = 1000000007L;
    BigInteger M = BigInteger.valueOf(mod);
    public int numOfWays(int[] nums) {
        TreeNode root = new TreeNode(nums[0]);
        for (int i = 1; i<nums.length; ++i) insert(root, nums[i]);
        root.size();
        long ans = recurseAns(root);
        return (int)((ans + mod - 1) % mod);
    }
    
    public long recurseAns(TreeNode root) {
        int Lsize = (root.left == null) ? 0 : root.left.size;
        int Rsize = (root.right == null) ? 0 : root.right.size;
        int N = root.size - 1;
        long val = chooseL(N, Lsize);
        long Lans = (root.left == null) ? 1 : recurseAns(root.left);
        long Rans = (root.right == null) ? 1 : recurseAns(root.right);
        val *= Lans;
        val %= mod;
        val *= Rans;
        val %= mod;
        return val;
    }
    
    public void insert(TreeNode root, int i) {
        if (i < root.val) {
            if (root.left == null) {
                root.left = new TreeNode(i);
            } else {
                insert(root.left, i);
            }
        } else {
            if (root.right == null) {
                root.right = new TreeNode(i);
            } else {
                insert(root.right, i);
            }
        }
    }

    class TreeNode {
        TreeNode left;
        TreeNode right;
        int val;
        int size;
        public TreeNode(int v) {
            val = v;
            size = 1;
        }
        public int size() {
            int Lsize = (left == null) ? 0 : left.size();
            int Rsize = (right == null) ? 0 : right.size();
            size = Lsize + Rsize + 1;
            return size;
        }
    }

	
	public BigInteger choose (long n, long k) {
		BigInteger ans= BigInteger.ONE;
		if (n-k < k) k = n-k;
		for (long n2 = n; n2 >n-k; --n2) ans = ans.multiply(BigInteger.valueOf(n2));
		for (long k2 = k; k2 > 0; --k2) ans = ans.divide(BigInteger.valueOf(k2));
		return ans;
	}
	public long chooseL(long n, long k) {
		return choose(n,k).mod(M).longValue();
	}
}
