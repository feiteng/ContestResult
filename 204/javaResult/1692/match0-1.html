import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

class Solution {
	long com[][];
	static long MOD = 1000000007;

	public int numOfWays(int[] nums) {
		com = new long[1001][1001];
		com[0][0] = 1;
		for (int r = 1; r < 1001; r++) {
			com[r][0] = 1;
			for (int c = 1; c <= 
r; c++) {
				com[r][c] = (com[r - 1][c - 1] + com[r - 1][c]) % MOD;
			}
		}
//		tr(com);
		List<Integer> l = new ArrayList<Integer>();
		for (int i : nums)
			l.add(i);
		long ans = (ways(l) - 1 + MOD) % MOD;
		return (int) ans;
	}


	private long ways(List<Integer> l) {
		if (l.size() <= 1)
			return 1;
		// tr(l);
		List<Integer> l1 = new ArrayList<Integer>();
		List<Integer> l2 = new ArrayList<Integer>();
		for (int i : l) {
			if (i > l.get(0))
				l2.add(i);
			else if (i < l.get(0))
				l1.add(i);

		}

		return 1L * com[l.size() - 1][l1.size()] * ways(l1) % MOD * ways(l2) % MOD;
	}

	public static void tr(Object... o) {
		System.out.println(Arrays.deepToString(o));
	}
}
