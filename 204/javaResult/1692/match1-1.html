class Solution {
    static final long[][] factorialMod = new long[1000][1000];
    static final int MOD = 1000000007;
    
    public int numOfWays(int[] nums) {
        Arrays.fill(factorialMod[0], 1);
        for (int i = 0; i < 1000; i++) {
            factorialMod[i][0] = 1;
        }

        for (int i = 1; i < 1000 ; i++) {
            for (int j = 1; j <1000 ; j++) {
                factorialMod[i][j] = (factorialMod[i-1][j] + factorialMod[i][j-1])% MOD;
            }
        }

        
ArrayList<Integer>  li = new ArrayList<>();
        for (int i : nums) {
            li.add(i);
        }
        
        return numOfWays(li) - 1;
    }

    public static int numOfWays(ArrayList<Integer> nums) {
        if (nums.size() <= 1) return 1
;
        
        ArrayList<Integer> left = new ArrayList<>();
        ArrayList<Integer> right = new ArrayList<>();
        for (int i = 1; i < nums.size(); i++) {
            if (nums.get(i) < nums.get(0)) left.add(nums.get(i));
            else right.add(nums.get(i));
        }
        
        return (int)(factorialMod[left.size()][right.size()] * numOfWays(left) % MOD * numOfWays(right) % MOD);
    }
}
