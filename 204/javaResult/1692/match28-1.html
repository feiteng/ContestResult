class Solution {
  private static final long MOD = (long) 1E9 + 7;
  private long[][] dp;
  public int numOfWays(int[] nums) {
    final int N = nums.length;
    dp = new long[N][N];
    Arrays.fill(dp[0], 1L);
    for (int r = 1; r != N; r++) {
      dp[r][0] = 1;
      for (int c = 1; c != N; c++)
        dp[r][c] = (dp[r - 1][c] + dp[r][c - 1]) % MOD
;
    }
    List<Integer> list = new ArrayList<>();
    for (int i : nums)
      list.add(i);
    long res = (dfs(list) - 1 + MOD) % MOD;
    return (int) res;
  }

  private long dfs(List<Integer> list) {
    if (list.size() <= 1)
      return 1;
    List<Integer> left=new ArrayList<>();
    List<Integer> right = new ArrayList<>();
    for (int i=1, root=list.get(0); i!=list.size(); i++) {
      final int num = list.get(i);
      if (num > root)
        right.add(num);
      else
        left.add(num);
    }
    long inner = (dfs(left) * dfs(right)) % MOD;
    long res = (dp[left.size()][right.size()] * inner) % MOD;
    return res;
  }
}
