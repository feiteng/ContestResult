import java.math.*;
class Solution {
    public int numOfWays(int[] nums) {
        return (solve(nums) + 1000000007 - 1) % 1000000007;
    }
    public int solve(int[] nums) {
        int mod = 1000000007;
        if (nums.length <= 1) {
            return 1;
        
}
        List<Integer> less = new ArrayList<>(), more = new ArrayList<>();
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] < nums[0]) {
                less.add(nums[i]);
            } else {
                more.add(nums[i]);
            }
        }
        int[] left = 
new int[less.size()];
        for (int i = 0; i < less.size(); i++) {
            left[i] = less.get(i);
        }
        
int[] right = new int[more.size()];
        for (int i = 0; i < more.size(); i++) {
            right[i] = more.get(i);
        }
        long res = (long)solve(left) * solve(right) % mod;
        if (!less.isEmpty() && !more.isEmpty()) {
            BigInteger product = BigInteger.ONE;
            for (int i = 0; i < less.size(); i++) {
                product = product.multiply(BigInteger.valueOf(less.size() + more.size() - i));
                product = product.divide(BigInteger.valueOf(i + 1));
            }
            res *= product.mod(BigInteger.valueOf(mod)).intValue();
            res %= mod;
        }
        return (int)res;
    }
}
