import java.math.BigInteger;
class Solution {
    private int n;
    private BigInteger[] factor;
    public int numOfWays(int[] nums) {
        
         n = nums.length;
        if (n == 1) return 0;
        factor = new BigInteger[n];
        factor[1] = new BigInteger("1");
        
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            list.add(nums[i]);
        }
        
        
        return (numOfWays(list).subtract(new BigInteger("1")).mod(new BigInteger("1000000007"))).intValue();
       
    }
    
    
    
    private BigInteger getFactor(int count) {
        BigInteger result = new BigInteger("1");
        int num = count;
        while (factor[num] == null) {
            result = result.multiply(new BigInteger(num + ""));
            num--;
        }
        
        result =  result.multiply( factor[num]);
      //  result = result %  1000000007;
        factor[count] = result;
        return result;
    }
    
    public BigInteger numOfWays(List<Integer> nums) {
        int n = nums.size();
        if (n <= 1) return new BigInteger("1");
        int root = nums.get(0);
        List<Integer> rightTree = new ArrayList<>();
        List<Integer> leftTree = new ArrayList<>();
        
        
        
        for(int i = 1; i < n; i++) {
            if (nums.get(i) > root) {
                rightTree.add(nums.get(i));
            } else {
                leftTree.add(nums.get(i));
            }
        }
        
        int rightSize = rightTree.size();
        int leftSize = leftTree.size();
        
        BigInteger leftWays = numOfWays(leftTree);
        BigInteger rightWays = numOfWays(rightTree);
        
        BigInteger ways = new BigInteger("0");
        
         if (leftSize == 0) {
             ways = rightWays;
         } else if (rightSize == 0) {
             ways = leftWays;
         } else {
           //  System.out.println(leftSize + "," + rightSize + "," + leftWays + "," + rightWays);
            ways = getFactor(leftSize + rightSize).divide(getFactor(leftSize)).divide(getFactor(rightSize)).multiply ( leftWays) .multiply( rightWays);   
         }
        
        return ways;
        
        
    }
    
    
}
