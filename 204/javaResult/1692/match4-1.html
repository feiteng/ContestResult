class Solution {
    private final int up = 1000;
    private final long p = 1000000007;
    
    private long cau(long n) {
        return n == 1 ? 1 : (p - p / n * cau(p % n) % p) % p;
    }
    
    private int dfs(long[] f, long[] rf, 
List<Integer> list) {
        if (list.size() <= 1) return 1;
        List<Integer> l = new ArrayList();
        List<Integer> r = new ArrayList();
        int root = list.get(0);
        for (int i = 1; i < list.size(); i++) {
            if (list.get(i) < root) l.add(list.get(i));
            else r.add(list.get(i));
        }
        long ans = 1;
        ans = ans * dfs(f, rf, l) % p;
        ans = ans * dfs(f, rf, r) % p;
        ans = ans * f[list.size() - 1] % p;
        ans = ans * rf[l.size()] % p;
        ans = ans * rf[r.size()] % p;
        return (int) ans;
    }
    
    public int numOfWays(int[] nums) {
        long[] f = new long[up + 1];
        long[] rf = new long[up + 1];
        f[0] = f[1] = 1L;
        rf[0] = rf[1] = 1L;
        for (int i = 2; i <= up; i++) {
            f[i] = (f[i - 1] * i) % p;
            rf[i] = cau(f[i]);
        }
        
List<Integer> list = new ArrayList();
        for (int i = 0; i < nums.length; i++) list.add(nums[i]);
        return dfs(f, rf, list) - 1;
    }
}
