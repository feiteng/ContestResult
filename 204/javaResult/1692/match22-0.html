class Solution {
    long MOD = 1_000_000_000 + 7;
    Map<Integer, Long> memo = new HashMap<>();
    int n;
    int[] nums;
    int mxn = 1010;
    long[][] comb2 = new long[mxn][mxn];
    public long comb2(int a, int b) {
        return comb2[a + b][a];
    }
    public long solve(List<Integer> a) {
        if (a.size() <= 1) return 1;
        List<Integer> l0 = new ArrayList<>();
        List<Integer> l1 = new ArrayList<>();
        for (int i = 1; i < a.size(); i++) {
            if (a.get(0) >= a.get(i)) l0.add(a.get(i));
            else l1.add(a.get(i));
        }
        long ansL = solve(l0);
        long ansR = solve(l1);
        // System.out.printf("lo %d k %d l %d r %d\n", lo, k, ansL, ansR);
        long ans = ((ansL * ansR) % MOD) * comb2(l0.size(), l1.size()) % MOD;
        // memo.put(t, ans);
        return ans;
    }
    public int numOfWays(int[] nums) {
        n = nums.length;
        this.nums = nums;
        for (int i = 0; i < mxn; i++) comb2[i][0] = comb2[i][i] = 1;
        for (int i = 1; i < mxn; i++)
            for (int j = 1; j <= i; j++)
                comb2[i][j] = (comb2[i - 1][j] + comb2[i - 1][j - 1]) % MOD;
        // for (int i = 0; i < 10; i++) {
        //     for (int j = 0; j < 10; j++) {
        //         System.out.printf(" %d", comb2[i][j]);
        //     }
        //     System.out.println();
        // }
        List<Integer> list = new ArrayList<>();
        for (int a : nums) list.add(a);
        return (int) ((solve(list) + MOD - 1) % MOD);
    }
}
