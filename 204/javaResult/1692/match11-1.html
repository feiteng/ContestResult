class Solution {
    long[][] ncr = new long[1001][1001];
    
    public int numOfWays(int[] nums) {
        ncr[0][0] = 1L;
        
for (int i = 1; i <= 1000; ++i) {
            ncr[i][0] = 1L;
            ncr[i][i] = 1L;
            for (int j = 1; j < i; ++j) {
                ncr[i][j] = ncr[i-1][j-1]+ncr[i-1][j];
                ncr[i][j] %= 1000000007;
            }
        }
        
        ArrayList<Integer> ns = new ArrayList<>();
        for (int i : nums)
            ns.add(i);
        return (nh(ns)+1000000006)%1000000007;
    }
    
    public int nh(ArrayList<Integer> nums) {
        int n = nums.size();
        if (n==0)
            return 1;
        if (n==1)
            
return 1;
        int root  = nums.get(0);
        ArrayList<Integer> l = new ArrayList<>();
        ArrayList<Integer> r = new ArrayList<>();
        for (int i = 1; i < n; ++i) {
            if (nums.get(i)<root) {
                l.add(nums.get(i));
            } else {
                r.add(nums.get(i));
            
}
        }
        long al = nh(l);
        long ar = nh(r);
        long ans = (al*ar)%1000000007;
        ans *= ncr[n-1][l.size()];
        ans %= 1000000007;
        //System.out.println(nums.size()+" "+root+" "+ans);
        return (int)ans;
    }
}
