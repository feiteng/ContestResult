class Solution {
    public int numOfWays(int[] nums) {
        int n = nums.length;
        long[][] C = new long[n + 1][n/2 + 1];
        C[0][0] = 1;
        for (int i = 1; i < n + 1; i++) {
            C[i][0] = 1;
            for (int j = 1; j < n / 2 + 1; j++) {
                C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % M;
            }
        }
        List<Integer> li = new ArrayList<>(n);
        for (int i = 0; i < n; i++) {
            li.add(nums[i]);
        }
        return (int)(helper(li, C) - 1);
    }

    static final int M = (int)1e9+7;
    private long helper(List<Integer> li, long[][] c) {
        if (li.size() <= 2) return 1;
        int first = li.get(0);
        List<Integer> left = new ArrayList<>();
        List<Integer> right = new ArrayList<>();

        for (int i = 1; i < li.size(); i++) {
            int e = li.get(i);
            if (e < first) {
                left.add(e);
            } else {
                right.add(e);
            }
        }
        int size = Math.min(left.size(), right.size());
        long com = (long)(c[li.size() - 1][size] % M);
        long l = helper(left, c);
        long r = helper(right, c);
        return (long)(com * l % M * r % M);
    }

}
