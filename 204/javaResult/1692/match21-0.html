class Solution {
    long fac[];
    private int MOD=1000000007;
    long pow(long a, long b) {
        long x = 1, y = a;
        while(b > 0) {
        if(b%2 == 1) {
        x=(x*y);
        if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD)
        y%=MOD;
        b /= 2;
        }
        return x;
    }
long inv(long a) {
return pow(a,MOD-2);
}
    private long intL(int a,int b){
        long ans=fac[a+b];
        ans=(ans*inv(fac[a]))%MOD;
        ans=(ans*inv(fac[b]))%MOD;
        return ans%MOD;
    }
    private long find(List<Integer> A){
        if(A.size()<=1)
            return 1;
        List<Integer> B=new ArrayList<>();
        List<Integer> C=new ArrayList<>();
        for(int i=1;i<A.size();i++)
            if(A.get(i)>A.get(0))
                C.add(A.get(i));
            else
                B.add(A.get(i));
        return ((find(B)*find(C))%MOD*intL(B.size(),C.size()))%MOD;
    }
    public int numOfWays(int[] a) {
        int n=a.length;
        fac=new long[1001];
        fac[0]=1;
        for(int i=1;i<1001;i++){
            fac[i]=(i*fac[i-1])%MOD;
        }
        List<Integer> A=new ArrayList<>();
        for(int i:a)
            A.add(i);
        return (int)((MOD+find(A)-1)%MOD);
    }
}
