class Solution {
    int M = 1_000_000_007;
    
    int[][] ord = new int[1001][1001]; 
    {
        for (int i = 0; i < ord.length; i++)
            Arrays.fill(ord[i], -1);
    }
    
    int order(int n, int m) {
        if (n == 0 || m == 0) return 1;
        if (ord[n][m] != -1) return ord[n][m];
        int x = order(n - 1, m), y = order(n, m-1);
        ord[n][m] =  (x+y) % M;
        return ord[n][m];
    }
    
    int go(int[] nums) {
        if (nums.length <= 1) return 1;
        int less = 0, more = 0;
        for (int i = 1; i < nums.length; i++)
            if (nums[i] > nums[0]) more++;
            else less++;
        int[] left = new int[less], right = new int[more];
        int ll = 0, rr = 0;
        for (int i : nums)
            if (i > nums[0]) right[rr++] = i;
            else if (i < nums[0])left[ll++] = i;
        int x = go(left), y = go(right);
        long ans = ((long)x*y) % M;
        ans = (ans *order(less, more)) % M;
        return (int)ans;
    }
    
    public int numOfWays(int[] nums) {
        if (nums.length <= 1) return 0;
        return go(nums) - 1;
    }
}
