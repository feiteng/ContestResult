import java.math.BigInteger;

class Solution {
    private static BigInteger mod = BigInteger.valueOf(1000000007);

    public int numOfWays(int[] nums) {
        return (int) ((helper(Arrays.stream(nums).mapToLong(Long::valueOf).boxed().collect(Collectors.toList())).longValue() + 1000000006) % 1000000007);
    
}

    private BigInteger helper(List<Long> list) {
        if (list.size() <= 1) {
            return BigInteger.ONE;
        }
        long root = list.get(0);
        List<Long> left = new ArrayList<>();
        List<Long> right = new ArrayList<>();
        for (long l : list) {
            if (l > root) {
                right.add(l);
            } else if (l < root) {
                left.add(l);
            }
        }
        return choose(left.size() + right.size(), left.size())
                .multiply(helper(left))
                .multiply(helper(right))
                .mod(mod);
    }

    private BigInteger choose(int a, int b) {
        b = Math.min(b, a - b);
        BigInteger out = BigInteger.ONE;
        for (int i = 0; i < b; i++) {
            out = out.multiply(BigInteger.valueOf(a - i)).divide(BigInteger.valueOf(1 + i));
        }
        return out.mod(mod);
    }
}
