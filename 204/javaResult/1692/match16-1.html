class Solution {
    private final static long MOD = 1000L * 1000L * 1000L + 7;
    long[][] comb;
    int LIMIT = 1000 + 10;
//    int LIMIT = 20;
    public int numOfWays(int[] nums) {
        comb = new long[LIMIT][LIMIT];
        
for (int i = 0; i < LIMIT; i++) {
            comb[i][i] = 1;
            comb[i][0] = 1;
        }
        for (int i = 2; i < LIMIT; i++) {
            for (int j = 1; j < i; j++) {
                comb[i][j] = comb[i - 1][j - 1] + comb[i - 1][j];
                
comb[i][j] = comb[i][j] % MOD;
            }
        }

        List<Integer> all = new ArrayList<>();
        for (int v : nums) {
            all.add(v);
        }

        long res = rec(all);
        res = (res + MOD - 1) % MOD;
        return (int) res;
    
}

    long rec(List<Integer> all) {
        if (all.isEmpty()) {
            return 1;
        }
        List<Integer> small = new ArrayList<>();
        List<Integer> large = new ArrayList<>();
        for (int v : all) {
            if (v < all.get(0)) {
                small.add(v);
            } else if (v > all.get(0)) {
                large.add(v);
            }
        }

        long res = comb[all.size() - 1][small.size()];
        res = res * rec(small);
        res = res % MOD;
        res = res * rec(large);
        res = res % MOD;
        return res;
    }
}
