class Solution {
  public static int numOfWays(int[] nums) {
    final int MOD = 1000000007;
    int n = nums.length;
    long[][] pascal = new long[n][n+1];
    for (int i = 1; i < n; i++) {
      pascal[i][0] = 1;
      pascal[i][i] = 1;
      for (int j = 1; j < i; j++) {
        pascal[i][j] = (pascal[i-1][j-1] + pascal[i-1][j]) % MOD;
      }
    }

    List<Integer> a = new ArrayList<>();
    for (int v : nums) {
      a.add(v);
    }
    return (int) ((MOD + func5502(a, pascal) - 1) % MOD);
  }

  static long func5502(List<Integer> nums, long[][] pascal) {
    if (nums.size() <= 1) {
      return 1;
    }
    List<Integer> l = new ArrayList<>();
    List<Integer> r = new ArrayList<>();
    for (int v : nums) {
      if (v < nums.get(0)) {
        l.add(v);
      }
      if (v > nums.get(0)) {
        r.add(v);
      }
    }
    long ans = (func5502(l, pascal) * func5502(r, pascal)) % 1000000007L;
    ans = (ans * pascal[nums.size()-1][l.size()]) % 1000000007L;
    return ans;
  }

}
