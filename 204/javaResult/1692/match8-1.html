class Solution {
    long[][] nCrArr = new long[1000][1000];
    
long mod = 1000000007;
    public int numOfWays(int[] nums) {
        List<Integer> curr = new ArrayList<>();
        for (int val: nums) curr.add(val);
        return (int) ways(curr) - 1;
    }
    
    private long ways(List<Integer> curr) {
        if (curr.size() < 3) return 1; 
        
        List<Integer> left = new ArrayList<>();
        List<Integer> right = new ArrayList<>();
        int root = curr.get(0);
        for (int i = 1; i < curr.size(); i++) {
            int val = curr.get(i);
            if (val < root) left.add(val);
            else right.add(val);
        }
        long ls = ways(left);
        long rs = ways(right);
        return (ls * rs % mod) * comb(left.size() + right.size(), left.size()) % mod;
    }
    
	private long comb(int n , int r) 
	{ 
		if(r== 0 || n == r) 
			return 1; 
		else {
            if (nCrArr[n][r] == 0) 
                nCrArr[n][r] = (comb(n - 1, r) + comb(n - 1, r - 1)) % mod;
            return nCrArr[n][r];
        }
	}
}
