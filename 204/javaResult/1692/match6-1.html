import java.math.BigInteger;

class Solution {
    final int MOD = (int) (1e9 + 7);
    
    public int numOfWays(int[] nums) {
        List<Integer> L = new ArrayList();
        for (int x : nums) L.add(x);
        return 
bstRec(L) - 1;
    }
    
    private int bstRec(List<Integer> L) {
        if (L.size() <= 2) return 1;
        List<Integer> A = new ArrayList(), B = new ArrayList();
        int head = L.get(0);
        
for (int i = 1; i < L.size(); i++) {
            if (L.get(i) > head) B.add(L.get(i));
            else A.add(L.get(i));
        }
        int a = bstRec(A), b = bstRec(B);
        BigInteger BI = nChooseK(L.size()-1, A.size());
        int nck = BI.mod(new BigInteger("" + MOD)).intValue();
        int m1 = (int) ((((long) a) * b) % MOD);
        return (int) ((((long) m1) * nck) % MOD);
    }
    
    BigInteger nChooseK(int N, int K) {
        BigInteger ret = BigInteger.ONE;
        for (int k = 0; k < K; k++) {
            ret = ret.multiply(BigInteger.valueOf(N-k))
                     .divide(BigInteger.valueOf(k+1));
        }
        return ret;
    }
}
