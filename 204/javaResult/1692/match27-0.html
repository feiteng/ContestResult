class Solution {

        private long mod = 1000000007;

        int maxn = 1000;
        long[][] C = new long[maxn][maxn];

        private void init() {
            for (int i=0;i<maxn;i++) {
                C[i][0] = C[i][i] = 1;
                for (int j=1;j<i;j++) {
                    C[i][j] = (C[i-1][j] + C[i-1][j-1]) % mod;
                }
            }
        }

        public int numOfWays(int[] nums) {
            init();
            List<Integer> list = new ArrayList<>();
            for (int num : nums) {
                list.add(num);
            }
            int answer = (int) ways(list);
            answer = (int) ((answer + mod - 1) % mod);
            return answer;
        }

        private long ways(List<Integer> nums) {
            if (nums.size() == 0) {
                return 1;
            }
            int len = nums.size();
            int root = nums.get(0);
            List<Integer> left = new ArrayList<>();
            List<Integer> right = new ArrayList<>();
            for (int i=1;i<len;i++) {
                int num = nums.get(i);
                if (num < root) {
                    left.add(num);
                } else {
                    right.add(num);
                }
            }
            long answer = C[len-1][left.size()];
            answer = (answer * ways(left)) % mod;
            answer = (answer * ways(right)) % mod;
            return answer;
        }
    }
