class Solution {
    
class Node {
        int val;
        int cnt;
        Node left;
        Node right;

        Node(int val) {
            this.val = val;
            this.cnt = 1;
        }
    }

    public static final int mod = 1000000007;

    public void insert(Node root, 
int x) {
        if (root.val > x) {
            if (root.left == null) {
                root.left = new Node(x);
            } else {
                insert(root.left, x);
            }
        } else {
            if (root.right == null) {
                root.right = new Node(x);
            } else {
                insert(root.right, x);
            }
        }
        root.cnt += 1;
    }

    long[][] map = new long[1001][1001];

    long calc(Node node) {
        if (node.left != null && node.right != null) {
            long a = calc(node.left);
            long b = calc(node.right);
            int p = node.left.cnt;
            int q = node.right.cnt;
            
return a * b % mod * map[p][q] % mod;
        }
        if (node.left != null) {
            return calc(node.left) % mod;
        }
        if (node.right != null) {
            return calc(node.right) % mod;
        }
        return 1;
    }

    public int numOfWays(int[] nums) {
        for (int i = 1; i <= nums.length; i++) {
            for (int j = 1; j <= nums.length; j++) {
                if (i == 1 && j == 1) {
                    map[i][j] = 2;
                } else if (i == 1) {
                    map[i][j] = j + 1;
                } else if (j == 1) {
                    map[i][j] = i + 1;
                } else {
                    map[i][j] = (map[i - 1][j] + map[i][j - 1]) % mod;
                }
            }
        }
        
Node node = new Node(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            insert(node, nums[i]);
        }
        return (int) ((calc(node) + mod - 1) % mod);
    }
}
