import java.math.BigInteger;

class Solution {
    
    private int MOD = (int) (1e9 + 7);
    
    public int numOfWays(int[] nums) {
        
        List<Integer> A = new ArrayList<>();
        for (int num : nums) A.add(num);
        
        BigInteger res = helper(A);
        return res.subtract(new BigInteger("1")).mod(new BigInteger(String.valueOf(MOD))).intValue();
        
        // return (int) (helper(A) % MOD - 1 % MOD) % MOD;
    }
    
    private BigInteger helper(List<Integer> nums) {
        
        if (nums.size() <= 2) return new BigInteger("1");
        
        int head = nums.get(0);
        
        List<Integer> left = new ArrayList<>();
        List<Integer> right = new ArrayList<>();
        
        for (int i = 1; i < nums.size(); ++i) {
            if (nums.get(i) < head) left.add(nums.get(i));
            else right.add(nums.get(i));
        }
        
        BigInteger x = new BigInteger(String.valueOf(C((long) left.size(), (long) (left.size() + right.size()))));
        // long x = C((long) left.size(), (long) (left.size() + right.size()));
        BigInteger y = new BigInteger(String.valueOf(helper(left)));
        BigInteger z = new BigInteger(String.valueOf(helper(right)));
        // long y = helper(left);
        // long z = helper(right);
        // return ((x % MOD) * (y % MOD) * (z % MOD)) % MOD;
        
        return x.multiply(y).multiply(z);
    }
    
    // C:left/left + right
    private long C(long k, long total) {
        
        long[] C = new long[(int) k + 1];
        C[0] = 1;
        for (int i = 1; i <= total; ++i) {
            for (int j = Math.min(i, (int) k); j > 0; --j) {
                C[j] = ((C[j] % MOD) + (C[j - 1] % MOD)) % MOD;
            }
        }
        
        return C[(int) k];
    }
}
