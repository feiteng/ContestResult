class Solution {
    final int mod = (int)1e9 + 7;
    int[][] dp;
    public int numOfWays(int[] nums) {
        int n = nums.length;
        if(n == 0) return 1;
        dp = new int[n + 1][n + 1];
        
        
        
        for(int i = 0; i <= n; i++){
            dp[i][1] = i + 1;
            dp[1][i] = i + 1;
            dp[0][i] = 1;
            dp[i][0] = 1;
        }
        
        for(int i = 2; i <= n; i++){
            for(int j = 2; j <= n; j++){
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
                dp[i][j] %= mod;
            }
        }
        
        List<Integer> list1 = new ArrayList<>();
        List<Integer> list2 = new ArrayList<>();
        
        for(int i = 1; i < n; i++){
            if(nums[i] < nums[0]){
                list1.add(nums[i]);
            }else{
                list2.add(nums[i]);
            }
        }
        
        long tmp = dp[list1.size()][list2.size()];
        int[] nums1 = new int[list1.size()];
        int[] nums2 = new int[list2.size()];
        
        for(int i = 0; i < list1.size(); i++){
            nums1[i] = list1.get(i);
        }
        
        for(int i = 0; i < list2.size(); i++){
            nums2[i] = list2.get(i);
        }
        tmp = tmp * numOfWays2(nums1) % mod;
        
        tmp = tmp * numOfWays2(nums2) % mod;
        
        tmp = (tmp - 1 + mod) % mod;
        
        return (int)tmp;
        
    }public long numOfWays2(int[] nums) {
        int n = nums.length;
        if(n == 0) return 1;
        
        List<Integer> list1 = new ArrayList<>();
        List<Integer> list2 = new ArrayList<>();
        
        for(int i = 1; i < n; i++){
            if(nums[i] < nums[0]){
                list1.add(nums[i]);
            }else{
                list2.add(nums[i]);
            }
        }
        
        long tmp = dp[list1.size()][list2.size()];
        int[] nums1 = new int[list1.size()];
        int[] nums2 = new int[list2.size()];
        
        for(int i = 0; i < list1.size(); i++){
            nums1[i] = list1.get(i);
        }
        
        for(int i = 0; i < list2.size(); i++){
            nums2[i] = list2.get(i);
        }
        tmp = tmp * numOfWays2(nums1) % mod;
        
        return  tmp * numOfWays2(nums2) % mod;
    }
    
    
}
