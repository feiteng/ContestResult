class Solution {
    int M = (int) 1e9 + 7;
    public int numOfWays(int[] n) {
        init();
        
List<Integer> l = new ArrayList<>();
        for (int i : n) l.add(i);
        return dfs(l) - 1;
    }
    int dfs(List<Integer> a) {
        if (a.size() <= 1) return 1;
        int i = 1, base = a.get(0);
        List<Integer> ll = new ArrayList<>(), rl = new ArrayList<>();
        for (; i < 
a.size(); i++) {
            if (a.get(i) > base) rl.add(a.get(i));
            else ll.add(a.get(i));
        }
        
        int left = dfs(ll);
        int right = dfs(rl);

        int lc = ll.size(), rc = rl.size();

        long res = 0;
        for (int group = rc; group >= 1; group--) {
            res += (long)c[rc - 1][group - 1] * c[lc + 1][group] % M;
            res %= M;
        }
        if (res == 0) res = 1;
        res = res * left % M;
        res = res * right % M;
        return (int) Math.max(1, res);
    }
    int c[][] = new int[1001][1001];


    void init()
    {
        for (int i = 0; i < 1001; i ++ )
            for (int j = 0; j <= i; j ++ )
                if (j == 0) c[i][j] = 1;
                else c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % M;
    }


}
