class Solution {
    long[][] C = new long[1010][1010];
    
    public long dfs(List<Integer> list) {
        if(list.size() <= 2) return 1
;
        int head = list.get(0);
        List<Integer> left = new ArrayList<>();
        List<Integer> right = new ArrayList<>();
        for(int i = 1; i < list.size(); i++) {
            if(list.get(i) < head) left.add(list.get(i));
            else right.add(list.get(i));
        }
        long pl = dfs(left);
        long pr = dfs(right);
        long ans = (((C[list.size() - 1][left.size()] * pl) % 1000000007) * pr) % 1000000007;
        return ans;
    }
    
    
public int numOfWays(int[] nums) {
        List<Integer> list = new ArrayList<>();
        for(int i = 0; i < nums.length; i++) list.add(nums[i]);
        for(int i = 1; i < 1010; i++) {
            C[i][0] = 1;
            C[i][i] = 1;
            for(int j = 1; j < i; j++) {
                C[i][j] = (C[i-1][j-1] + C[i-1][j]) % 1000000007;
            }
        }
        return (int)(dfs(list)) - 1;
    }
}
