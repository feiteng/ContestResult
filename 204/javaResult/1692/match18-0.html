import java.math.BigInteger;
class Solution {
    int mod = 1000000007;
    BigInteger mod2 = BigInteger.valueOf(1000000007);

    public int numOfWays(int[] nums) {
        LinkedList<Integer> nums2 = new LinkedList<>();
        for (int num : nums) {
            nums2.add(num);
        }
        return (int) cal(nums2) - 1;
    }

    private long cal(List<Integer> nums) {
        int size = nums.size();
        if (size <= 2) {
            return 1;
        }
        int middle = nums.get(0);
        List<Integer> left = new LinkedList<>();
        List<Integer> right = new LinkedList<>();
        for (int n : nums) {
            if (n > middle) {
                right.add(n);
            } else if (n < middle) {
                left.add(n);
            }
        }
        long res = c(size - 1, left.size());
        res *= cal(left);
        res %= mod;
        res *= cal(right);
        return res % mod;
    }

    private long c(int n, int k) {
        BigInteger a = BigInteger.valueOf(1);
        BigInteger b = BigInteger.valueOf(1);
        if (k > n / 2) {
            k = n - k;
        }
        for (int i = 1; i <= k; i++) {
            a = a.multiply(BigInteger.valueOf(n + 1 - i));
            b = b.multiply(BigInteger.valueOf(i));
        }
        return a.divide(b).mod(mod2).longValue();
    }
}
