class Solution {
    public int minDays(int[][] grid) {
        if (!connected(grid)) {
            return 0;
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (!connected(grid)) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
    public boolean connected(int[][] grid) {
        boolean filled = false;
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        
for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1 && !visited[i][j]) {
                    if (filled) {
                        return false;
                    }
                    fill(grid, visited, i, j);
                    filled = true;
                
}
            }
        }
        return true;
    }
    public void fill(int[][] grid, boolean[][] visited, int i, int j) {
        visited[i][j] = true;
        int[] x = new int[]{0, 1, 0, -1}, y = new int[]{1, 0, -1, 0};
        for (int k = 0; k < 4; k++) {
            if (i + x[k] >= 0 && i + x[k] < grid.length && j + y[k] >= 0 && j + y[k] < grid[0].length) {
                if (grid[i + x[k]][j + y[k]] == 1 && !visited[i + x[k]][j + y[k]]) {
                    fill(grid, visited, i + x[k], j + y[k]);
                }
            }
        }
    }
}
