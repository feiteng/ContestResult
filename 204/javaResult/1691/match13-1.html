class Solution {
    boolean[][] v;
    public int minDays(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        if (getAns(grid)) 
{
            return 0;
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (getAns(grid)) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
    
    
    private boolean getAns(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        v = new boolean[n][m];
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (cnt > 1) {
                break;
            }
            
for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1 && !v[i][j]) {
                    cnt++;
                    dfs(grid, i, j);
                    // for (boolean[] a : v) {
                    //     System.out.println(Arrays.toString(a));
                    // }
                }
            }
        }
        // System.out.println(cnt);
        return cnt != 1;
    }
    
    
    int[] p = {-1, 0, 1, 0, -1};
    private void dfs(int[][] g, int i, int j) {
        int n = g.length, m = g[0].length;
        
if (i < 0 || j < 0 || i >= n || j >= m || g[i][j] == 0 || v[i][j]) {
            return;
        }
        v[i][j] = true;
        for (int k = 0; k < 4; k++) {
            dfs(g, i + p[k], j + p[k + 1]);
        }
    }
}
