class Solution {
    public int minDays(int[][] grid) {
        int cnt = numIslands(copy(grid));
        if (cnt > 1 || cnt == 0) {
            return 0;
        }
        
for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    cnt = numIslands(copy(grid));
                    if (cnt > 1 || cnt == 0) 
{
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;

    }

    public int[][] copy(int[][] grid) {
        int[][] ans = new int[grid.length][grid[0].length];
        for (int i = 0; i < ans.length; i++) {
            for (int j = 0; j < ans[0].length; j++) {
                ans[i][j] = grid[i][j];
            }
        }
        return ans;
    }

    public int numIslands(int[][] grid) {
        int islandNum = 0;
        
for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 1){
                    infect(grid, i, j);
                    islandNum++;
                }
            }
        }
        return islandNum;
    }

    
    public void infect(int[][] grid, int i, int j)
{
        if(i < 0 || i >= grid.length ||
                j < 0 || j >= grid[0].length || grid[i][j] != 1){
            return;
        }
        grid[i][j] = 2;
        infect(grid, i + 1, j);
        infect(grid, i - 1, j);
        infect(grid, i, j + 1);
        infect(grid, i, j - 1);
    }
}
