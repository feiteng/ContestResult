class Solution {
    public int minDays(int[][] grid) {
    if(count(grid) != 1){
      return 0;
    }
    int n = grid.length;
    
int m = grid[0].length;
    for(int i = 0 ; i < n ; i++){
      for(int j = 0; j < m ; j++){
        if(grid[i][j] == 1){
          grid[i][j] = 0;
          if(count(grid) != 1){
            return 1;
          }
          grid[i][j] = 1;
        }
      }
    }
    return 2;
  }

  private int count(int[][] grid){
    int n = grid.length;
    int m = grid[0].length;
    boolean[][] seen = new boolean[n][m];
    
int cnt = 0;
    for(int i = 0 ; i < n ; i++){
      for(int j = 0 ; j < m ; j++) {
        if (grid[i][j] == 1 && !seen[i][j]) {
          dfs(grid, seen, i, j);
          cnt++;
        
}
      }
    }
    return cnt;
  }
  
  private int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
  
  private void dfs(int[][] grid, boolean[][] seen, int r, int c){
    int n = grid.length;
    int m = grid[0].length;
    seen[r][c] = true;
    for(int[] dir: dirs){
      int nr = r + dir[0];
      int nc = c + dir[1];
      if(nr < 0 || nr >= n || nc < 0 || nc >= m || grid[nr][nc] == 0 || seen[nr][nc]){
        continue;
      }
      dfs(grid, seen, nr, nc);
    }
  }
}
