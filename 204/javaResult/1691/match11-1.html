import java.util.HashMap;
import java.util.HashSet;

class Solution {
  public int minDays(int[][] grid) {
    if (countIslands(grid) != 1) 
{
      return 0;
    }
    
    for (int i = 0; i < grid.length; i++) {
      for (int j = 0; j < grid[i].length; j++) {
        if (grid[i][j] == 1) {
          grid[i][j] = 0;
          if (countIslands(grid) != 1) {
            return 1;
          }
          grid[i][j] = 1;
        }
      }
    }
    return 2;
  }
  
  public int countIslands(int[][] grid) {
    HashMap<Integer, HashSet<Integer>> visited = new HashMap<>();
    int count = 0;
    
for (int i = 0; i < grid.length; i++) {
      for (int j = 0; j < grid[i].length; j++) {
        if (grid[i][j] == 1) {
          visited.put(i, visited.getOrDefault(i, new HashSet<>()));
          if (!visited.get(i).contains(j)) {
            count++;
            dfs(grid, i, j, visited);
          }
        
}
      }
    }
    
    return count;
  }
  
  static int[] dx = {-1, 0, 1, 0};
  static int[] dy = {0, 1, 0, -1};
  
public void dfs(int[][] grid, int x, int y, HashMap<Integer, HashSet<Integer>> visited) {
    if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y] == 0) {
      return;
    }
    visited.put(x, visited.getOrDefault(x, new HashSet<>()));
    if (visited.get(x).contains(y)) {
      return;
    }
    visited.get(x).add(y);
    
for (int i = 0; i < dx.length; i++) {
      dfs(grid, x + dx[i], y + dy[i], visited);
    }
  }
}
