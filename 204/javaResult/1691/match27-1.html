class Solution {
    int[][] g;
    int row, col;
    int[][] dir = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    void dfs(int r, int c) {
        g[r][c] = -1;
        for (int d = 0; d < 4; d++) {
            int tr = r + dir[d][0], tc = c + dir[d][1];
            if (tr < 0 || tr >= row || tc < 0 || tc >= col)
                continue;
            if (g[tr][tc] != 1)
                continue;
            
dfs(tr, tc);
        }
    }
    int cal() {
        int cnt = 0;
        for (int r = 0; r < row; r++)
            
for (int c = 0; c < col; c++) {
                if (g[r][c] == 1) {
                    cnt++;
                    dfs(r, c);
                }
            }
        for (int r = 0; r < row; r++)
            for (int c = 0; c < col; c++)
                if (g[r][c] == -1)
                    g[r][c] = 1
;
        return cnt;
    }
    public int minDays(int[][] grid) {
        g = grid;
        row = g.length;
        col = g[0].length;
        if (cal() > 1)
            return 0;
        for (int r = 0; r < row; r++)
            for (int c = 0; c < col; c++)
                
if (g[r][c] == 1) {
                    g[r][c] = 0;
                    if (cal() > 1)
                        return 1;
                    g[r][c] = 1;
                }
        return 2;
    }
}
