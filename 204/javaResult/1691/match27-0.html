class Solution {
    int[][] g;
    int n;
    int m;
    
    int[][] dir = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    public void dfs(int i, int j) {
        if (i < 0 || j < 0 || i >= n || j >= m || g[i][j] != 1) return;
        g[i][j] = -1;
        for (int[] d : dir) {
            int ii = i + d[0];
            int jj = j + d[1];
            dfs(ii, jj);
        }
        
    }
    
    public boolean isConnected() {
        int c = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (g[i][j] == 1) {
                    c++;
                    dfs(i, j);
                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (g[i][j] == -1) {
                    g[i][j] = 1;
                }
            }
        }
        
        return c == 1;
    }
    public int minDays(int[][] grid) {
        g = grid;
        n = g.length;
        m = g[0].length;
        
        if (!isConnected()) return 0;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = -1;
                    if (!isConnected()) return 1;
                    grid[i][j] = 1;
                }
            }
        return 2;
    }
}
