class Solution {

    public int minDays(int[][] grid) {
        if (check(grid)) {
            return 0;
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (check(grid)) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }

    private boolean check(int[][] grid) {
        int[][] clone = clone(grid);
        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (clone[i][j] == 1) {
                    count++;
                    if (count > 1) {
                        return true;
                    }
                    check(i, j, clone);
                }
            }
        }
        return count != 1;
    }

    private void check(int i, int j, int[][] grid) {
        if (i >= 0 && i < grid.length && j >= 0 && j < grid[0].length && grid[i][j] == 1) {
            grid[i][j] = 0;
            check(i - 1, j, grid);
            check(i, j - 1, grid);
            check(i, j + 1, grid);
            check(i + 1, j, grid);
        }
    }

    private int[][] clone(int[][] grid) {
        int[][] clone = new int[grid.length][grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                clone[i][j] = grid[i][j];
            }
        }
        return clone;
    }
}
