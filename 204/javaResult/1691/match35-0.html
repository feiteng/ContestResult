class Solution {
    public int minDays(int[][] grid) {
        if(!isConnected(grid)){
            return 0;
        }
        int m=grid.length;
        int n=grid[0].length;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==1){
                    grid[i][j]=0;
                    if(!isConnected(grid)){
                        return 1;
                    }
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
    
    public boolean isConnected(int[][] grid){
        int m=grid.length;
        int n=grid[0].length;
        int x=-1;
        int y=-1;
        int count=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==1){
                    count++;
                    if(x==-1&&y==-1){
                        x=i;
                        y=j;
                    }
                }
            }
        }
        if(count==0){
            return false;
        }
        int[][] direction={{1,0},{-1,0},{0,1},{0,-1}};
        boolean[][] visited=new boolean[m][n];
        Queue<int[]> queue=new LinkedList<int[]>();
        queue.add(new int[]{x,y});
        visited[x][y]=true;
        int temp=1;
        while(queue.size()>0){
            int size=queue.size();
            for(int i=0;i<size;i++){
                int[] point=queue.poll();
                for(int k=0;k<direction.length;k++){
                    int newX=direction[k][0]+point[0];
                    int newY=direction[k][1]+point[1];
                    if(newX<0||newX>=m||newY<0||newY>=n||grid[newX][newY]==0||visited[newX][newY]){
                        continue;
                    }
                    visited[newX][newY]=true;
                    temp++;
                    queue.add(new int[]{newX,newY});
                }
            }
        }
        return temp==count?true:false;
    }
}
