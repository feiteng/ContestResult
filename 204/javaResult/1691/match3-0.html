class Solution {
    private int[][] step = {{0,1},{1,0},{0,-1},{-1,0}};

    private void bfs(int[][] ma, boolean vis[][], int x,int  y) {
        vis[x][y] = true;
        for (int i=0;i<4;i++) {
            int xx = x + step[i][0], yy = y+step[i][1];
            if (xx > -1 && yy > -1 && xx < ma.length && yy <ma[xx].length && ma[xx][yy]== 1&&!vis[xx][yy]){
                bfs(ma,vis,xx,yy);
            }
        }
    }

    private int getNum(int[][] ma) {
        boolean[][] vis = new boolean[ma.length][];
        for (int i=0;i<ma.length;i++) {
            vis[i] = new boolean[ma[i].length];
        }
        int result = 0;
        for (int i=0;i<ma.length;i++) {
            for(int j=0;j<ma[i].length;j++) {
                if (ma[i][j]>0&& !vis[i][j]) {
                    result++;
                    bfs(ma,vis,i,j);
                }
            }
        }
        return result;
    }

    public int minDays(int[][] grid) {
        int now = getNum(grid);
        if (now !=1)return 0;
        for (int i=0;i<grid.length;i++){
            for(int j=0;j<grid[i].length;j++) {
                if (grid[i][j]==1) {
                    grid[i][j]=0;
                    if (getNum(grid)!=1)return 1;
                    grid[i][j]=1;
                }
            }
        }
        return 2;
    }
}
