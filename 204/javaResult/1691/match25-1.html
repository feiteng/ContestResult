class Solution {
    int h, l;
    public int minDays(int[][] g) {
        h = g.length; l = g[0].length;
        int cnt = ff(g);
        if (cnt > 1 || cnt == 0) 
return 0;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < l; j++) {
                if (g[i][j] == 0) continue;
                g[i][j] = 0;
                if (ff(g) == 2) return 1;
                g[i][j] = 1;
            
}
        }
        return 2;
    }
    int ff(int[][] g) {
        int cnt = 0;
        int[][] c = new int[h][];
        for (int i = 0; i < h; i++) c[i] = g[i].clone();
        
for (int i = 0; i < h; i++) {
            for (int j = 0; j < l; j++) {
                if (c[i][j] == 1) {
                    ff(c, i, j);
                    cnt++;
                }
            }
        }
        return cnt;
    }
    void ff(int[][] g, int y, int x) {
        
if (y < 0 || x < 0 || y == h || x == l || g[y][x] == 0) return;
        g[y][x] = 0;
        ff(g, y+1, x);
        ff(g, y-1, x);
        ff(g, y, x+1);
        ff(g, y, x-1);
    }
}
