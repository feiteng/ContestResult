class Solution {
    
    int[] d = new int[]{0, -1, 0, 1, 0};
    
    public int minDays(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        if (cal(grid, m, n) != 1) {
            return 0;
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (cal(grid, m, n) != 1) {
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }

    private int cal(int[][] grid, int m, int n) {
        boolean[][] visit = new boolean[m][n];
        int c = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 1 && !visit[i][j]) {
                    ++c;
                    dfs(grid, visit, m, n, i, j);
                }
            }
        }
        return c;
    }
    
    private void dfs(int[][] grid, boolean[][] visit, int m, int n, int x, int y) {
        visit[x][y] = true;
        for (int i = 0; i < 4; ++i) {
            int nx = x + d[i];
            int ny = y + d[i + 1];
            if (nx >= 0 && nx < m && ny >= 0 && ny < n && grid[nx][ny] == 1 && !visit[nx][ny]) {
                dfs(grid, visit, m, n, nx, ny);
            }
        }
    }
    
}
