class Solution {
    public static int[] dx = new int[]{-1, 0, 0, 1};
    public static int[] dy = new int[]{0, -1, 1, 0};
    
    
private void dfs(int[][] grid, int[][] vis, int x, int y) {
        int n = grid.length;
        int m = grid[0].length;
        if (0 <= x && x < n && 0 <= y && y < m && vis[x][y] == 0 && grid[x][y] == 1) {
            vis[x][y] = 1;
            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                dfs(grid, vis, nx, ny);
            }
        }
    }
    
    
private boolean judge(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int[][] vis = new int[n][m];
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (vis[i][j] == 1 || grid[i][j] == 0) continue;
                ans++;
                dfs(grid, vis, i, j);
            }
        }
        return ans != 1;
    }
    
    
public int minDays(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        if (judge(grid)) return 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (judge(grid)) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
}
