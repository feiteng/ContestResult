class Solution {

        private int[][] dir = new int[][] {{0,1},{0,-1},{1,0},{-1,0}};
        private int[][] grid;
        private int[][] visit;
        private int color;
        private int row;
        private int col;

        public int minDays(int[][] grid) {
            this.grid = grid;
            this.row = grid.length;
            this.col = grid[0].length;
            if (isSplit()) {
                return 0;
            }
            for (int i=0;i<row;i++) {
                for (int j=0;j<col;j++) {
                    if (grid[i][j] == 1) {
                        grid[i][j] = 0;
                        if (isSplit()) {
                            return 1;
                        }
                        grid[i][j] = 1;
                    }
                }
            }
            return 2;
        }

        private boolean isSplit() {
            visit = new int[row][col];
            color = 0;
            for (int i=0;i<row;i++) {
                for (int j=0;j<col;j++) {
                    if (grid[i][j] == 1 && visit[i][j] == 0) {
                        color++;
                        dfs(i, j);
                    }
                }
            }
            return color != 1;
        }

        private void dfs(int x, int y) {
            visit[x][y] = color;
            int nx, ny;
            for (int i=0;i<4;i++) {
                nx = x + dir[i][0];
                ny = y + dir[i][1];
                if (nx < 0 || nx >= row || ny < 0 || ny >= col || visit[nx][ny] != 0 || grid[nx][ny] != 1) {
                    continue;
                }
                dfs(nx, ny);
            }
        }
    }
