class Solution {
  private static final int INF = 1_000_000_001;
    private static final int MOD = 1_000_000_007;
    private static final int UNSET = -123;
    
private static final int[][] DIR = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    int[][] grid;
    boolean[][] vst;
    int m;
    int n;

    public int minDays(int[][] grid) {
        this.grid = grid;
        this.m = grid.length;
        this.n = grid[0].length;
        this.vst = new boolean[m][n];
        //System.out.println("start...");
        
if (checkDisconnected()) return 0;
        //System.out.println("end...");
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (checkDisconnected()) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) continue;
                
                for (int i1 = 0; i1 < m; i1++) {
                    for (int j1 = 0; j1 < n; j1++) {
                        if (i == i1 && j == j1) continue;
                        if (grid[i1][j1] == 0) continue;
                        grid[i][j] = 0;
                        grid[i1][j1] = 0;
                        if (checkDisconnected()) return 2;
                        grid[i][j] = 1;
                        grid[i1][j1] = 1;
                    }
                }
            }
        }
        return 0;
    }

    boolean checkDisconnected() {
        
for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                vst[i][j] = false;
            }
        }


        int count = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1 && !vst[i][j]) {
                    if (count > 0) return true;
                    
dfs(i, j);
                    count++;
                }
            }
        }

        return count == 0;
    }

    void dfs(int x, int y) {
        if (!(x >= 0 && x < m && y >= 0 && y < n)) return;
        
if (grid[x][y] != 1 || vst[x][y]) return;
        vst[x][y] = true;

        for (int i = 0; i < DIR.length; i++) {
            dfs(x + DIR[i][0], y + DIR[i][1]);
        }
    }
}
