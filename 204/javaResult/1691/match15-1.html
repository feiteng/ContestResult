class Solution {
    int[][] dir = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    
    public int minDays(int[][] grid) {
        if (!isConnected(grid)) return 0;
        int n = grid.length;
        int m = grid[0].length;
        
for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (!isConnected(grid)) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
    
    private boolean isConnected(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] visited = new boolean[n][m];
        boolean searched = false;
        
for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1 && !visited[i][j]) {
                    if (searched) return false;
                    search(grid, i, j, visited);
                    searched = true;
                
}
            }
        }
        return true;
    }
    
    private void search(int[][] grid, int x, int y, boolean[][] visited) {
        if (x < 0 || y < 0 || x >= grid.length || y >= grid[0].length) return;
        if (grid[x][y] == 0 || visited[x][y]) return;
        visited[x][y] = true;
        for (int d = 0; d < 4; d++) {
            search(grid, x + dir[d][0], y + dir[d][1], visited);
        }
    }
}
