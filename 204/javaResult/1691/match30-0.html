class Solution {
    
    int[][] dirs = new int[][]{{0, 1}, {1, 0}, {-1, 0}, {0, -1}};

    public int minDays(int[][] grid) {
        if (!isOne(grid)) return 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if (!isOne(grid)) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }


    private boolean isOne(int[][] grid) {
        int[][] copy = new int[grid.length][grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            copy[i] = Arrays.copyOf(grid[i], grid[0].length);
        }
        int count = 0;
        for (int i = 0; i < copy.length; i++) {
            for (int j = 0; j < copy[0].length; j++) {
                if (copy[i][j] > 0) {
                    count++;
                    infect(copy, i, j);
                    if (count >= 2) return false;
                }
            }
        }
        return true;
    }

    private void infect(int[][] grid, int x, int y) {
        if (x >= grid.length || x < 0 || y < 0 || y >= grid[0].length || grid[x][y] < 1) return;
        grid[x][y] = -1;
        for (int[] dir : dirs) {
            infect(grid, x + dir[0], y + dir[1]);
        }
    }
}
