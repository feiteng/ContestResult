class Solution {
    public void dfs(int[][] grid, boolean[][] visited, int x, int y) {
		int[][] to = { { 1, 0 }, { 0, 1 }, { -1, 0 }, { 0, -1 } };
		int n = grid.length;
		int m = grid[0].length;

		visited[x][y] 
= true;

		for (int i = 0; i < 4; i++) {
			int nextx = x + to[i][0];
			int nexty = y + to[i][1];

			if (nextx >= 0 && nextx < n && nexty >= 0 && nexty < m && visited[nextx][nexty] == false
					&& grid[nextx][nexty] == 1) {
				dfs(grid, visited, nextx, nexty);
			}
		}
	}

	public boolean check(int[][] grid) {
		int n = grid.length;
		int m = grid[0].length;

		boolean[][] vi
sited = new boolean[n][m];
		int count = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (grid[i][j] == 1 && visited[i][j] == false) {
					count++;
					dfs(grid, visited, i, j);
				}
			}
		}

		return count > 1;
	}

	public 
int minDays(int[][] grid) {
		int n = grid.length;
		int m = grid[0].length;

		if (check(grid)) {
			return 0;
		}

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (grid[i][j] == 1) {
					grid[i][j] = 0;
					if (check(grid)) {
						return 1;
					}
					grid[i][j] = 1;
				}
			}
		}

		return 2;
	}
}
