class UnionFind{
    private final int[] parents;
    
    public UnionFind(int n){
        this.parents = new int[n];
        clear();
    }
    
    
    public void clear(){
        for(int i =0;i<parents.length;i++){
            parents[i] = i;
        }
    }
    
    public int find(int node){
        int parent = parents[node];
        if(parent == node){
            return parent;
        }else{
            int root = find(parent);
            parents[node] = root;
            return root;
        }
    }
    
    public void union(int i, int j){
        int p1 = find(i);
        int p2 = find(j);
        parents[p1] = p2;
    }
    
}

class Solution {
    
private final int[][] dirs = {{1, 0}, {-1, 0},{0, 1},{0, -1}};
    
    public int minDays(int[][] grid) {
         
if(count(grid) != 1){
             return 0;
         }
        for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid[0].length;j++){
                if(grid[i][j] == 1){
                    grid[i][j] = 0;
                    if(count(grid) != 1){
                        return 1;
                    }
                    grid[i][j] = 1;
                }
            }
        }
        
         for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid[0].length;j++){
                if(grid[i][j] == 1){
                    grid[i][j] = 0;
                    for(int m = 0;m<grid.length;m++){
                        for(int n =0;n<grid[0].length;n++){
                            if(grid[m][n] == 1){
                                grid[m][n] = 0;
                                if(count(grid) != 1){
                                    return 2;
                                }
                                grid[m][n] = 1;
                            }
                        }
                    }
                    grid[i][j] = 1;
                
}
            }
        }
        return 3;
    }
    
    private int count(int[][] grid){
        UnionFind uf = new UnionFind(grid.length * grid[0].length);
        for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid[0].length;j++){
                if(grid[i][j] == 1){
                    
for(int[] dir : dirs){
                        int ni = i+dir[0];
                        int nj = j+dir[1];
                        if(ni>=0 && ni<grid.length && nj>=0 && nj<grid[0].length && grid[ni][nj] == 1){
                            uf.union(i * grid[0].length + j, ni * grid[0].length + nj);
                        }
                    }
                }
            }
        }
        Set<Integer> set = new HashSet<>();
         
for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid[0].length;j++){
                if(grid[i][j] == 1){
                    set.add(uf.find(i*grid[0].length + j));
                }
            }
        }
        return set.size();
    }
}
