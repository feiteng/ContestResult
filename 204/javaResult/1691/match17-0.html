class Solution {

    public int minDays(int[][] grid) {
        if (check(grid, 1)) {
            return 0;
        }

        int kk = 2;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == kk) {
                    grid[i][j] = 0;
                    if (check(grid, kk)) {
                        return 1;
                    }
                    kk++;
                    grid[i][j] = kk;
                }
            }
        }
        return 2;
    }

    private boolean check(int[][] grid, int kk) {
        boolean ex = false;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == kk) {
                    if (ex) {
                        return true;
                    }
                    dfs(grid, i, j, kk);
                    ex = true;
                }
            }
        }
        return !ex;
    }

    private int[][] dis = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

    private void dfs(int[][] grid, int i, int j, int kk) {
        if (i >= 0 && j >= 0 & i < grid.length && j < grid[0].length && grid[i][j] == kk) {
            grid[i][j]++;
            for (int[] di : dis) {
                dfs(grid, i + di[0], j + di[1], kk);
            }
        }
    }
}
