class Solution {
    int[] dx = {0, 0, -1, 1};
    int[] dy = {-1, 1, 0, 0};
    public int minDays(int[][] grid) {
        
int cnt = findCnt(grid);
        if (cnt > 1) {
            return 0;
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    continue;
                }

                grid[i][j] = 0;
                cnt = findCnt(grid);
                if (cnt > 1) {
                    return 1;
                }
                grid[i][j] = 1;
            
}
        }

        return 2;
    }

    int findCnt(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;

        int res = 0;
        boolean[][] vis = new boolean[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0 || vis[i][j]) {
                    continue;
                }
                dfs(i, j, grid, n, m, vis);
                res++;
            
}
        }
        return res;
    }

    void dfs(int row, int col, int[][] grid, int n, int m, boolean[][] vis) {
        if (row < 0 || row >= n || col < 0 || col >= m) {
            return;
        }
        
if (grid[row][col] == 0 || vis[row][col]) {
            return;
        }
        vis[row][col] = true;
        for (int i = 0; i < dx.length; i++) {
            dfs(row + dx[i], col + dy[i], grid, n, m, vis);
        }
    }
}
