class Solution {
    public int minDays(int[][] grid) {
        int maxBrute = 0;
        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 1) {
                    int aux = 0;
                    if(i != 0 && grid[i-1][j] == 1) aux++;
                    if(i != grid.length - 1 && grid[i+1][j] == 1) aux++;
                    if(j != 0 && grid[i][j-1] == 1) aux++;
                    if(j != grid[0].length - 1 && grid[i][j+1] == 1) aux++;
                    if(aux > maxBrute) maxBrute = aux;
                }
            }
        }
        if(!checkConnection(grid)) return 0;
        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 1) {
                    grid[i][j] = 0;
                    if(!checkConnection(grid)) return 1;
                    grid[i][j] = 1;
                }
            }
        }
        return 2;
    }
    
    boolean checkConnection(int[][] grid) {
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        boolean isl = false;
        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 1 && !visited[i][j]) {
                    if(isl) return false;
                    isl = true;
                    visit(grid, visited, i, j);
                }
            }
        }
        return true;
    }
    
    void visit(int[][] grid, boolean[][] visited, int i, int j) {
        visited[i][j] = true;
        if(i != 0 && grid[i-1][j] != 0 && !visited[i-1][j]) visit(grid, visited, i-1, j);
        if(i != grid.length - 1 && grid[i+1][j] != 0 && !visited[i+1][j]) visit(grid, visited, i+1, j);
        if(j != 0 && grid[i][j-1] != 0 && !visited[i][j-1]) visit(grid, visited, i, j-1);
        if(j != grid[0].length - 1 && grid[i][j+1] != 0 && !visited[i][j+1]) visit(grid, visited, i, j+1);
    }
}
