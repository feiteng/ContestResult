class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        loop:
        for (int i = 0; i < n; i++) {
            int c = 1;
            for (int j = i + m; j + m <= n && c < k; j += m, c++)
                for (int x = 0; x < m; x++)
                    if (arr[j+x] != arr[i+x]) continue loop
;
            if (c == k) return true;
        }
        return false;
    }
}
