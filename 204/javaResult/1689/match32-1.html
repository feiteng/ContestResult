class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i < arr.length - m + 1; i++) {
            if (checkValid(arr, i, m, k)) {
                return true;
            }
        }
        return false;
    }

    private boolean checkValid(int[] arr, int i, int m, int k) {
        if (k == 1) {
            return true;
        }

        int j = i + m;
        for (int x = 0; x < k - 1; x++) {
            if (j > arr.length - m) {
                return false;
            }
            
for (int y = 0; y < m; y++) {
                if (arr[i + y] != arr[j + y]) {
                    return false;
                }
            }
            j += m;
        }
        return true;
    }
}
