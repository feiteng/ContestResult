class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            int cnt = 1;
            for (int j = i+m; j < n;) {
                int t = 0;
                for (t = 0; j+t < n && t < m; t++) {
                    if (arr[i+t] != arr[j+t]) {
                        break;
                    }
                }
                if (t == m) {
                    j += m;
                    cnt++;
                } else
                    break;
            }
            if (cnt >= k)
                return true;
        }
        return false;
    }
}
