class Solution {
    public boolean containsPattern(int[] arr, int m, int K) {
        for (int i = 0; i < arr.length - m * K + 1; i++) {
            boolean flag = true;
            for (int j = 1; j < K; j++) {
                for (int k = 0; k < m; k++) {
                    if (arr[i + j * m + k] != arr[i + k]) {
                        flag = false;
                        break;
                    }
                }
                if (!flag) {
                    break;
                }
            }
            if (flag) {
                return true;
            }
        }
        return false;
    }
}
