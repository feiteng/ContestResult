class Solution {
    public boolean containsPattern(int[] a, int m, int k) {
        int n = a.length;
        for (int i=0; i<=n-m*k; i++) {
            boolean f = true;
            for (int j=0; j<k; j++)
                for (int l=0; l<m; l++)
                    if (a[i+l] != a[i+j*m+l])
                        f = false;
            if (f)
                return true;
        }
        return false;
    }
}
