class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
            int r = 0;
            for(int i=0;i+m-1<arr.length;++i){
                int all = 0;
                for(int j=i;j+m-1<arr.length;j+=m){
                    boolean ok = true;
                    for(int f=0;f<m;++f){
                        if(arr[i+f]!=arr[j+f]){
                            ok = false;
                        }
                    }
                    if(ok){
                        all++;
                    }else{
                        break;
                    }
                    
                    
                }
                if(all>=k){
                    return true;
                }
                
                
            }
        return false;
        
        
    }
}
