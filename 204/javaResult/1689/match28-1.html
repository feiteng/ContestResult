class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for(int i = 0; n - i >= m * k; i++){
            int count = 0;
            boolean flag = true;
            for(int j = 0; j < m; j++){
                
for(int p = 1; p < k ; p++){
                    if(arr[i + p * m +j] != arr[i + (p - 1) * m + j]){
                        flag = false;
                    }
                }
            }

            if(flag) return flag;
        }

        return false;
    }
}
