class Solution {
    public boolean containsPattern(int[] arr, int m, int t) {
        int n = arr.length;
        
        for (int i = 0; i + m * t <= n; ++i) {
            boolean found = true;
            for (int j = 0; j < m && found; ++j) {
                for (int k = 1; k < t; ++k) {
                    if (arr[i + j + k * m] != arr[i + j]) {
                        //printf("");
                        found = false;
                        break;
                    }
                }
            
}
            if (found) return true;
        }
        
        return false;
    }
}
