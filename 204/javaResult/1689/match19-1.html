class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for(int i = 0; i < arr.length - m; i++) {
            boolean ok = true;
            for(int j = 0; j < k; j++) {
                if(i + j * m + m - 1 >= arr.length) {
                    ok = false;
                    break;
                }
                
for(int q = 0; q < m && i + j * m + q < arr.length; q++) {
                    if(arr[i + j * m + q] != arr[i + q]) {
                        ok = false;
                        break;
                    }
                }
                if(!ok) break;
            }
            if(ok) return true;
        }
        return false;
    }
}
