class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i < arr.length; i++) {
            int c = 1;
            for (int j = i + m; j < arr.length; j++) {
                if (isSame(arr, i,j, m)) {
                    c++;
                    j+=m - 1;
                } else {
                    break;
                }
                if (c >= k)
                    return true;
            }
        
}
        return false;
    }

    boolean isSame(int[] arr, int x, int y, int m) {
        if (Math.max(x,y) + m > arr.length)
            return false;
        
for (int i = 0; i < m; i++) {
            if (arr[i + x] != arr[i + y])
                return false;
        }
        return true;
    }
}
