class Solution {

    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i <= arr.length - m * k; i++) {
            if (containsPattern(i, arr, m, k)) {
                return true;
            }
        }
        return false;
    }

    private boolean containsPattern(int index, int[] arr, int m, int k) {
        for (int i = 1; i < k; i++) {
            for (int j = 0; j < m; j++) {
                if (arr[i * m + j + index] != arr[index + j]) {
                    return false;
                }
            }
        }
        return true;
    }
}
