class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int pattLen = m * k;
        
        for (int i = 0; i < arr.length - pattLen + 1; i++) {
            
            boolean exists = true;
            for (int j = 0; j < pattLen; j++) {
                //System.out.println(i + " " + (i + j) + " " + (i + (j % m)));
                
                if (arr[i + j] != arr[i + (j % m)]) {
                    exists = false;
                    break;
                }
            
}
            
            if (exists) return true;
        }
        
        return false;
    }
}
