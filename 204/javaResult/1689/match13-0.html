class Solution {
    public boolean containsPattern(int[] arr, int m, int n) {
        int N = arr.length;
        L: for (int i = 0; i + m * n <= N; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < n; k++) {
                    if (arr[i + k * m + j] != arr[i + j]) {
                        continue L;
                    }
                }
            }
            
            return true;
        }
        
        return false;
    }
}
