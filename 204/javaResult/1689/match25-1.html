class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        
for (int i = 0; i < arr.length; i++) {
            int len = 1;
            for (int j = i + m; j + m <= arr.length; j++) {
                boolean match = true;
                for (int k2 = 0; k2 < m; k2++) {
                    if (arr[i + k2] != arr[j + k2]) {
                        match = false;
                    }
                }
                if (match) {
                    len++;
                } else {
                    break;
                }
            }
            if (len >= k) {
                return true;
            }
        }
        return false;
    }
}
