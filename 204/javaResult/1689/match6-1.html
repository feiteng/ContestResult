class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for(int i = 0; i <= n - m * k; i++) {
            boolean matching = true;
            for(int j = 0; j < m; j++){
                for(int jj = 1; jj < k; jj++) {
                    if(arr[i + (jj - 1) * m + j] != arr[i + jj * m + j])
                    {
                        matching = false;
                        break;
                    }
                }
                
            }
            if(matching) return true;
        }
        return false;
    }
}
