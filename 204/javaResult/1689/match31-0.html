class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for(int st = 0; st < n; st++){
            boolean flag = true;
            for(int j = 1; j < k; j++){
                for(int i = 0; i < m; i++){
                    if(st + i + j * m >=n){
                        flag = false;break;
                    }
                    if(arr[st + i + j * m] != arr[st + i]){
                        flag = false;
                        break;
                    }
                }
                if(!flag)break;
            }
            if(flag)return true;
        }
        return false;
    }
}
