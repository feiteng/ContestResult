class Solution {
    public boolean containsPattern(int[] arr, int m, int t) {
        int n = arr.length;
        search: for (int i = 0; i + m * t <= n; i++) {
            for (int j = 0; j < t; j++) {
                for (int k = 0; k < m; k++) {
                    if (arr[i + k] != arr[i + j * m + k]) continue search
;
                }
            }    
            return true;
        }
        return false;
    }
}
