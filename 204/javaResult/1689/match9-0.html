class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int l = arr.length;
        for (int i = 0; i < l; i++) {
            boolean diff = false;
            for (int j = 0; j < k; j++) {
                for (int d = 0; d < m; d++) {
                    if ((i + m * j + d >= l) || arr[i + d] != arr[i + m * j + d]) diff = true;
                }
            }
            if (!diff) return true;
        }
        return false;
    }
}
