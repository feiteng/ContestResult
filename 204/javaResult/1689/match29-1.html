class Solution {
        public boolean containsPattern(int[] arr, int m, int k) {
            int len = arr.length;
            if (m * k > len) {
                return false;
            }
            for (int patternL=0;patternL+m*k-1<len;patternL++) {
                int patternR = patternL + m - 1;
                boolean isTrue = true;
                for (int count=2;count<=k;count++) {
                    
for (int i=0;i<m;i++) {
                        if (arr[patternL + i] != arr[patternL+m*(count-1) + i]) {
                            isTrue = false;
                            break;
                        }
                    }
                    if (isTrue == false) 
{
                        break;
                    }
                }
                if (isTrue) {
                    return true;
                }
            }
            return false;
        }
    }
