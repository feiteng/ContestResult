class Solution {
public boolean containsPattern(int[] arr, int m, int k) {

        int n = arr.length;

        for (int i = 0; i + m <= n; i++) {
            int count = 0;
            for (int j = i; j + m <= n; j += m) {
                boolean ok = true;
                for (int l = 0; l < m; l++) {
                    if (arr[i + l] != arr[j + l]) {

                        ok = false;
                        break;
                    }
                }

                if (ok) count++;
                else break;
            }
            if (count >= k) return true;
        }
        return false;
    }
}
