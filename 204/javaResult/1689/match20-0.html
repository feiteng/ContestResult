class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        
        int N = arr.length;
        if (N < m * k) return false;
        
        for (int i = 0; i < N; ++i) {
            // pattern i, i + 1, i + 2, i + m - 1
            int pointer = i + m;
            boolean finish = true;
            for (int j = 0; j < k - 1; ++j) {
                for (int index = 0; index < m; ++index) {
                    if (pointer >= N || arr[i + index] != arr[pointer++]) {
                        finish = false;
                        break;
                    }
                }
                if (!finish) break;
            }
            if (finish) return true;
        }
        
        return false;
    }
}
