class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for (int start = 0; start < n; start++) {
            if (start + (m - 1) >= n) continue;
            int good = 0;
            for (int i = start; i <= start + (m - 1); i++) {
                int ct = 0;
                for (int j = i; j < n; j += m) {
                    if (arr[j] == arr[i]) ct++; else break;
                }
                if (ct >= k) good++; else break;
            }
            if (good == m) return true;
        }
        return false;
    }
}
