class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int n = arr.length;
        for (int i = 0; i < n; ++i) {
            boolean ip = true;
            for (int j = 0; j < m; ++j) {
                if (i+j>=n) {
                    ip = false;
                    break;
                }
                for (int z = 1; z < k; ++z) {
                    if (i+j+z*m>=n) {
                        ip = false;
                        break;
                    }
                    if (arr[i+j+z*m]!=arr[i+j]) {
                        ip = false;
                        break;
                    }
                }
            }
            if (ip)
                return true;
        }
        return false;
    }
}
