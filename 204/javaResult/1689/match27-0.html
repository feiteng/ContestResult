class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i + m <= arr.length; i++) {
            int count = 0;
            for (int j = i + m; j + m <= arr.length; j += m) {
                if (equal(arr, i, j, m)) {
                    count++;
                } else {
                    break;
                }
            }
            if (count >= k - 1) return true;
        }
        
        return false;
    }
    
    private boolean equal(int[] a, int i, int j, int m) {
        for (int k = 0; k < m; k++) {
            if (a[i + k] != a[j + k]) return false;
        }
        return true;
    }
}
