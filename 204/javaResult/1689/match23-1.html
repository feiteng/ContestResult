class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        // i + m * k - 1 < n
        int n = arr.length;
        for (int i = 0; i + m * k - 1 < n; i++) {
            boolean found = false;
            int num;
            for (num = 1; num < k; num++) 
{
                int j;
                for (j = 0; j < m; j++) {
                    if (arr[i + j] != arr[i + num * m + j]) {
                        break;
                    }
                }
                if (j < m) 
break;
            }
            if (num >= k) return true;
        }
        return false;
    }
}
