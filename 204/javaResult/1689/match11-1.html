class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {

        for (int i = 0; i + m * k - 1 < arr.length; i++) {

            if (contains(arr, i, m, k)) {
                return true;
            }


        }
        return false;

    }

    private boolean contains(int[] arr, int i, int m, int k) {

        for (int p = 0; p < m; p++) {
            int val = arr[i + p];
            
for (int j = 0; j < k; j++) {

                if (arr[i + p + j * m] != val) {
                    return false;
                }
            }

        }
        return true;


    }
}
