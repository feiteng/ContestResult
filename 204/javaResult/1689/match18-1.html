class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i < arr.length - m * k + 1; i++) {
            boolean res = true;
            for (int j = i; res && j < i + m; j++) {
                for (int l = 1; res && l < k; l++) {
                    res = (arr[j] == arr[j + m * l]);
                
}
            }
            if (res) return true;            
        }
        return false;
    }
}
