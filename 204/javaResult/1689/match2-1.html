class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        int len = arr.length;
        for (int i = 0; i < len; i++) {
            boolean match = true;
            
for (int j = 0; match && j < m; j++) {
                for (int l = 1; l < k; l++) {
                    if (i + j + l * m >= len || arr[i + j] != arr[i + j + l * m]) {
                        match = false;
                        break;
                    
}
                }
            }
            if (match) return true;
        }
        return false;
    }
}
