class Solution {
    public int getMaxLen(int[] nums) {
        int pos = -1, neg = -1;
        int ans = 0;
        int p = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                neg = -1;
                pos = i;
                p = 1;
                continue;
            }
            p *= (nums[i] > 0 ? 1 : -1);
            if (p == 1) {
                ans = Math.max(ans, i - pos);                
            } else {
                if (neg != -1) ans = Math.max(ans, i - neg);
                else neg = i;
            }
        }
        return ans;
    }
}
