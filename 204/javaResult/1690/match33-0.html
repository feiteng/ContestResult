class Solution {
    public int getMaxLen(int[] nums) {
        int firstN = -1;
        int lastN = -1;
        int nCount = 0;
        int start = 0;
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                if (nCount % 2 == 0) {
                    res = Math.max(res, i - start);
                } else {
                    res = Math.max(res, i - firstN - 1);
                    res = Math.max(res, lastN - start);
                    res = Math.max(res, i - lastN - 1);
                    res = Math.max(res, firstN - start);
                }
                firstN = -1;
                lastN = -1;
                nCount = 0;
                start = i + 1;
            } else if (nums[i] < 0) {
                if (firstN < 0) {
                    firstN = i;
                }
                nCount++;
                lastN = i;
            }
        }

        int i = nums.length;
        if (nCount % 2 == 0) {
            res = Math.max(res, i - start);
        } else {
            res = Math.max(res, i - firstN - 1);
            res = Math.max(res, lastN - start);
            res = Math.max(res, i - lastN - 1);
            res = Math.max(res, firstN - start);
        }
        return res;
    }
}
