class Solution {
    public int getMaxLen(int[] nums) {
        int lastOdd = -1;
        
int lastEven = -1;
        int n = nums.length;
        int negs = 0;
        int result = 0;
        for(int i = 0; i < n; i++){
           
            if(nums[i] == 0){
                lastOdd = -1; 
                lastEven = i;
                negs = 0;
                continue;
            }
            negs += (nums[i] < 0)? 1 : 0 ;
            
            if(negs % 2 == 0){
               
                    result = Math.max(i-lastEven , result);
                
            }else{
                if(lastOdd == -1){
                    lastOdd = i;
                }else{
                    result = Math.max(i - lastOdd, result);
                
}
            }
            
        }
        return result;
            
        
    }
}
