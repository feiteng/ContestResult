class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        int total = 0;
        int even = 0;
        int odd = Integer.MIN_VALUE;
        int result = 0;
        
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                even = i+1;
                odd = Integer.MIN_VALUE;
                total = 0;
                continue;
            }
            if (nums[i] < 0) {
                if (odd == Integer.MIN_VALUE) odd = i;
                total++;
            }
            if (total % 2 == 0) {
                result = Math.max(result, i - even + 1); 
            } else {
                result = Math.max(result, i - odd);
            }
        }
        
        return result;
    }
}
