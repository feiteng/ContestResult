class Solution {
    public int getMaxLen(int[] nums) {
        int res = 0, left = 0;
        for(int i = 0; i < nums.length; i++){
            if(nums[i] == 0){
                res = Math.max(res, helper(nums, left, i-1));
                left = i+1;
            }
        }
        res = Math.max(res, helper(nums, left, nums.length-1));
        return res;
    }
    
    private int helper(int[] nums, int left, int right){
        if(left > right)return 0;
        int count = 0;
        int min = right, max = left;
        for(int i = left; i <= right; i++){
            if(nums[i] > 0) continue;
            ++count;
            min = Math.min(min, i);
            max = Math.max(max, i);
        }
        if(count%2 == 0)return right - left + 1;
        return Math.max(right-min, max - left);
    }
}
