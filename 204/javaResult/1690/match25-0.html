class Solution {
    public int getMaxLen(int[] nums) {
        int sum = 0;
        int lastNegative = -1;
        int lastZero = -1;
        int ret = 0;
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] == 0) {
                lastNegative = -1;
                lastZero = j;
                sum = 0;
            } else if (sum == 0) {
                sum = 1;
            }
            if (nums[j] < 0) {
                sum *= -1;
                if (lastNegative == -1) {
                    lastNegative = j;
                }
            }
            if (sum == 1) {
                ret = Math.max(ret, j - lastZero);
            } else if (sum == -1) {
                ret = Math.max(ret, j - lastNegative);
            }
        }
        return ret;
    }
}
