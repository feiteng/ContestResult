class Solution {
    public int getMaxLen(int[] nums) {
        
        Integer positive = -1;
        Integer negative = null;
        int res = 0;
        int total = 1;
        
        for (int i = 0; i < nums.length; ++i) {
            if (nums[i] == 0) {
                
                total = 1;
                positive = i;
                negative = null;
                
            } else {
                
                if ((total > 0 && nums[i] > 0) || ((total < 0 && nums[i] < 0))) total = 1;
                else total = -1;
                
                if (total > 0) {
                    res = Math.max(res, i - positive);
                } else {
                    if (negative == null) {
                        negative = i; 
                    } else {
                        res = Math.max(res, i - negative);
                    }
                }

            }
        }
        
        return res;
    }
}
