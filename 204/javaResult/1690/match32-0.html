class Solution {
    public int getMaxLen(int[] nums) {
        int rst = 0;
        int pos = 0;
        int neg = 0;
        for (int num : nums) {
            if (num > 0) {
                pos++;
                neg = neg > 0 ? neg + 1 : 0;
            } else if (num < 0) {
                int old_pos = pos;
                pos = neg > 0 ? neg + 1 : 0;
                neg = old_pos + 1;
            } else {
                pos = 0;
                neg = 0;
            }
            
            rst = Math.max(rst, pos);
        }
        
        return rst;
    }
}
