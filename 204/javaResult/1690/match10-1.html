class Solution {
    public int getMaxLen(int[] nums) {
        
int neg = 0;
        int pos = 0;
        int ans = 0;
        for (int i : nums) {
            if (i == 0) {
                pos = 0;
                neg = 0;
            } else if (i > 0) {
                pos = pos+1;
                neg = (neg == 0) ? 0 : neg+1;
            } else {
                int pos_o = pos;
                int neg_o = neg;
                pos = (neg_o == 0) ? 0 : Math.max(0, neg_o + 1);
                neg = Math.max(1, pos_o + 1);
            
}
            ans = Math.max(pos, ans);
            //System.out.println(i + " : " + pos + " , " + neg);
        }
        return ans;
    }
}
