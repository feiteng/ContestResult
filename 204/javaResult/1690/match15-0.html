class Solution {
    public int getMaxLen(int[] nums) {
        int start = 0;
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                if (start != i) {
                    res = Math.max(res, getMaxLenNoZero(nums, start, i - 1));
                }
                start = i + 1;
            }
        }
        
        if (start != nums.length) {
            res = Math.max(res, getMaxLenNoZero(nums, start, nums.length - 1));
        }
        return res;
    }
    
    public int getMaxLenNoZero(int[] nums, int start, int end) {
        int firstNeg = -1;
        int lastNeg = -1;
        
        int negCnt = 0;
        
        for (int i = start; i <= end; i++) {
            if (nums[i] < 0) {
                negCnt++;
                if (firstNeg == -1) firstNeg = i;
                lastNeg = i;
            }
        }
        
        if (negCnt % 2 == 0) return end - start + 1;
        
        return Math.max(lastNeg - start, end - firstNeg);
    }
}
