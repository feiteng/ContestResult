class Solution {
    public int getMaxLen(int[] nums) {
        
int result = 0;
        
        int sign = 1;
        int negative = -1;
        int start = 0;
        for(int i =0;i<nums.length;i++){
            if(nums[i] == 0){
                sign = 0;
                negative = -1;
                start = i+1;
            }else if(nums[i] > 0){
                if(sign == 0){
                    sign = 1;
                }
            }else{
                if(sign == 0 || sign == 1){
                    sign = -1;
                    if(negative == -1){
                        negative = i;
                    }
                }else{
                    sign = 1;
                }
            
}
            if(sign == 1){
                result = Math.max(result, i - start + 1);
            }else{
                if(negative != -1){
                    result = Math.max(result, i - negative);
                
}
            }
        }
        return result;
    }
}
