class Solution {
    public int getMaxLen(int[] nums) {
        int last0 = -1;
        int maxLen = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                maxLen = Math.max(maxLen, getMaxLen(nums, last0 + 1, i - 1));
                last0 = i;
            }
        }
        maxLen = Math.max(maxLen, getMaxLen(nums, last0 + 1, nums.length - 1));
        
return maxLen;
    }
    
    int getMaxLen(int[] nums, int st, int ed) {
        int neg = 0;
        int firstP = -1;
        int lastP = -1;
        for (int i = st; i <= ed; i++) {
            if (nums[i] < 0) {
                if (neg == 0) firstP = i;
                lastP = i;
                neg++;
            }
        }
        if (neg % 2 == 0) return ed - st + 1;
        return Math.max(ed - firstP, lastP - st);
    }
}
