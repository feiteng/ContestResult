class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        int len = 0, count = 0;
        int start = -1;
        Integer neg = null;
        for(int i = 0; i < n; i++)
        {
            if(nums[i] == 0)
            {
                start = i;
                count = 0;
                neg = null;
            }   
            else
            {
                if(nums[i] < 0) count++;
                
                if(count % 2 == 0)
                {
                    len = Math.max(len, i - start);
                }
                else
                {
                    if(neg != null) len = Math.max(len, i - neg);
                    if(neg == null) neg = i;
                }
            }
        }
        
        return len;
    }
}
