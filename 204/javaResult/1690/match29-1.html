class Solution {
     public int getMaxLen(int[] nums) {
        int pos = -1;
        int start = 0;
        int cnt = 0;
        int ans = 0;
        for(
int i=0; i < nums.length; i++) {
            if(nums[i] == 0) {
                pos = -1;
                start = i + 1;
                cnt = 0;
                continue;
            }
            if(nums[i] < 0) {
                if(pos == -1) pos = i;
                cnt++;
            }
            int len = 0;
            if(cnt % 2 == 0) {
                len = i - start + 1;
            } else if(pos != -1) {
                len = i - pos;
            
}
            //if(i == 7)
            //System.out.println("pos: "+ start+" "+ cnt+" "+ )
            if(len > ans) ans = len;
        }
        return ans;
    }
}
