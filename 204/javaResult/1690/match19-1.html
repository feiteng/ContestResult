class Solution {

    public int getMaxLen(int[] nums) {
        
int max = 0, l = 0, r = -1, left = -1, right = -1, count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < 0) {
                if (left == -1) {
                    left = i;
                }
                right = i;
                count++;
            }
            
if (nums[i] == 0) {
                if (count % 2 == 0) {
                    max = Math.max(max, r - l + 1);
                } else {
                    max = Math.max(max, 
Math.max(r - left, right - l));
                }
                l = i + 1;
                count = 0;
                left = right = -1;
            }
            r = i;
        }
        
if (count % 2 == 0) {
            max = Math.max(max, r - l + 1);
        } else {
            max = Math.max(max, Math.max(r - left, right - l));
        }
        return max;
    }
}
