class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        int l = -1;
        int f = n;
        int p = 0;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] < 0) {
                p = 1 - p;
                if (f == n) f = i;
            } else if (nums[i] == 0) {
                p = 0;
                f = n;
                l = i;
            
}
            // System.out.printf("p%d f%d l%d i%d\n", p, f, l, i);
            if (p == 0) ans = Math.max(ans, i - l);
            else ans = Math.max(ans, i - f);
            
        }
        
//         if (p == 0) ans = Math.max(ans, n - 1 - l);
//         else ans = Math.max(ans, n - 1 - f);
        
        return ans;
    }
}
