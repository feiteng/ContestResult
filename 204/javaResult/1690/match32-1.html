class Solution {
    public int getMaxLen(int[] nums) {
        int pos = 0, neg = 0;
        int res = 0;
        for(int num : nums){
            int a, b;
            if(num > 0){
                a = Math.max(pos + 1, 1);
                b = Math.max(neg == 0 ? 0 : neg + 1, 0);
                pos = a;
                neg = b;
            }
            else if (num < 0){
                a = Math.max(neg == 0 ? 0 : neg + 1, 0);
                b = Math.max(pos + 1, 1);
                pos = a;
                
neg = b;
            }
            else{
                pos = 0;
                neg = 0;
            }
            //System.out.println(pos + " " + neg);
            res = Math.max(res, pos);
        }
        return res;
    }
}
