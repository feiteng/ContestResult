class Solution {
    public int getMaxLen(int[] nums) {
        int ans = 0;
        int leftnv = nums.length+1;
        int rightnv = 0;
        int left=0;
        int right=0;
        int counter = 0;
        int nvCounter = 0;
        for(int i=0;i<nums.length;++i){
            if(nums[i]==0){
                if(nvCounter%2==0){
                    ans = Math.max(ans,counter);
                }else{
                    ans = Math.max(ans,Math.max(counter-(leftnv-left+1),counter-(right-rightnv+1)));
                }
                left=i+1;
                right=i+1;
                counter=0;
                leftnv=nums.length+1;
                rightnv=0;
                nvCounter=0;
            }else {
                right=i;
                ++counter;
                if(nums[i]<0){
                    rightnv = i;
                    leftnv = Math.min(leftnv,i);
                    nvCounter++;
                }
            }
        }
        if(nvCounter%2==0){
            ans = Math.max(ans,counter);
        }else{
            ans = Math.max(ans,Math.max(counter-(leftnv-left+1),counter-(right-rightnv+1)));
        }
        return ans;
    }
}
