class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        int[][] dp = new int[n][2];
        dp[0][0] = nums[0] > 0 ? 1 : 0;
        dp[0][1] = nums[0] < 0 ? 1 : 0;
        int res = dp[0][0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == 0) continue
;
            if (nums[i] > 0) {
                dp[i][0] = dp[i-1][0] + 1;
                dp[i][1] = dp[i-1][1] == 0 ? 0 : dp[i-1][1] + 1;
            } else {
                dp[i][0] = dp[i-1][1] == 0 ? 0 : dp[i-1][1] + 1;
                dp[i][1] = dp[i-1][0] + 1;
            }
            //System.out.println(i + "," + dp[i][0] + "," + dp[i][1]);
            res = Math.max(res, dp[i][0]);
        }
        return res;
    }
}
