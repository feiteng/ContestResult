class Solution {
    public int getMaxLen(int[] nums) {
        int len = 0;
        boolean even = true;
        int cnt = 0;
        int last = -1, start = -1;
        for (int i = 0; i < nums.length; ++i) {
            if (nums[i] == 0) {
                cnt = 0;
                even = true;
                start = i;
                last = -1;
                continue;
            }
            if (nums[i] < 0) {
                even ^= true;
                if (last == -1)
                    last = i;
            }
            if (even) len = Math.max(len, i - start);
            else len = Math.max(len, i - last);
        }
        return len;
    }
}
