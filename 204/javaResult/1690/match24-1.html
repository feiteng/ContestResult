class Solution {
    public int getMaxLen(int[] a) {
        
int max=0,n=a.length;
        int p=0;
        int s=-1;
        for(int i=0;i<n;i++){
            if(a[i]==0){
                p=0;
                s=i;
                continue;
            }
            if(a[i]>0)
                p++;
            // System.out.println(p);
            int l=i-s;
            
if((l-p)%2==0)
                max=Math.max(max,l);
        }
        p=0;
        s=n;
        for(int j=n-1;j>=0;j--){
            if(a[j]==0){
                p=0;
                s=j;
                continue;
            }
            if(a[j]>0)
                p++;
            int l=s-j;
            
if((l-p)%2==0)
                max=Math.max(max,l);
        }
        return max;
    }
}
