class Solution {
    public int getMaxLen(int[] nums) {
        int ans = 0;
        int positive = 0, negative = 0;
        for (int num : nums) {
            if (num == 0) {
                positive = 0;
                negative = 0;
            } else if (num > 0) {
                positive++;
                if (negative > 0) {
                    negative++;
                }
                ans = Math.max(ans, positive);
            } else {
                int tmp = negative;
                negative = positive + 1;
                
if (tmp > 0) {
                    positive = tmp + 1;
                } else {
                    positive = 0;
                }
                ans = Math.max(ans, positive);
            }
        }
        return ans;
    }
}
