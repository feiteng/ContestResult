class Solution {
    public int getMaxLen(int[] nums) {
        int len = nums.length;
        int ans = 0;
        int pre = -1;
        int pref = -2;
        int fac = 1;
        for(int i = 0; i < len; i++) {
            if(nums[i] == 0) {
                fac = 1;
                pre = i;
                pref = -2;
                continue;
            }
            fac = fac*(nums[i] > 0 ? 1 : -1);
            if(fac > 0) {
                ans = Math.max(ans, i-pre);
            } else {
                if(pref != -2) {
                    ans = Math.max(ans, i-pref);
                }
            }
            if(nums[i] < 0 && pref == -2) {
                pref = i;
            }
        }
        return ans;
    }
}
