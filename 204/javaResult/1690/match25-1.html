class Solution {
    public int getMaxLen(int[] nums) {
        int lastNegative = -1;
        int lastCero = -1;
        int s = 0;
        int r = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < 0) {
                s++;
                if (lastNegative == -1)
                    lastNegative = i;
            } else if (nums[i] > 0) {
                // Nothing
            } else if (nums[i] == 0) {
                lastCero = i;
                lastNegative = -1;
                s = 0;
            
}
            if (s % 2 == 0) {
                r = Math.max(r, i - lastCero);
            } else if (lastNegative != -1){
                r = Math.max(r, i - lastNegative);
            }
        }
        return r;
    }
}
