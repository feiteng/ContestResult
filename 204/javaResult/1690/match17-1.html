class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n + 1];
        
int lastn = -1, ans = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                lastn = -1;
                dp[i + 1] = 0;
            } else if (nums[i] > 0) {
                dp[i + 1] = dp[i] + 1;
            } else {
                
if (lastn >= 0) {
                    dp[i + 1] = dp[lastn] + i - lastn + 1;
                } else {
                    dp[i + 1] = 0;
                }
                lastn = i;
            }
            ans = Math.max(ans, dp[i + 1]);
        }
        return ans;
    }
}
