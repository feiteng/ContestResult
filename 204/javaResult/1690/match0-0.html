class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length, negs = 0, lastZero = -n - 1, maxLen = 0, len = 0, firstNegInd = -1;
        int[] multi = new int[n];
        for (int i = 0; i < n; ++i) {
            if (nums[i] == 0) {
                negs = 0;
                len = 0;
                firstNegInd = -1;
            } else {
                if (nums[i] < 0) {
                    ++negs;
                    if (firstNegInd == -1) {
                        firstNegInd = i;
                    }
                }
                ++len;
                if (negs % 2 == 0) {
                    maxLen = Math.max(len, maxLen);
                } else {
                    maxLen = Math.max(i - firstNegInd, maxLen);
                }
            }
        }
        
        return maxLen;
    }
}
