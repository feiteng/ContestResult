class Solution {
    
    public int getMaxLen(int[] nums) {
        
        int n=nums.length, max=0;
        
        boolean flag=true;
        int first_false=-1;
        
        for(int i=0, last_reset=0; i<n; i++) {
            
            if(nums[i]==0) {
                flag=true;
                first_false=-1;
                last_reset=i+1;
                continue;
            }
            
            if(nums[i]<0)
                flag=!flag;
            
            if(flag) {
                max=Math.max(max, i-last_reset+1);
            } else {
                if(first_false!=-1) {
                    max=Math.max(max, i-first_false);
                }
            }
            
            if(nums[i]<0 && first_false==-1)
                first_false=i;
        }
        
        return max;
    }
}
