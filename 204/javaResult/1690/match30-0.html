class Solution {
    public int getMaxLen(int[] nums) {
        int left = -1, right = -1, tot = 0;
        int ans = 0, lastZero = -1;
        for(int i = 0; i <= nums.length; i++) {
            if(i == nums.length || nums[i] == 0) {
                if(tot%2 == 0) {
                    ans = Math.max(ans, i - lastZero-1);
                } else {
                    int partial = Math.max(right - lastZero - 1, i - left - 1);
                    ans = Math.max(ans, partial);
                }
                left = -1;
                right = -1;
                tot = 0;
                lastZero = i;
            }
            else if(nums[i]<0) {
                tot++;
                if(left == -1) left = i;
                right = i;
            }
        }
        return ans;
    }
}
