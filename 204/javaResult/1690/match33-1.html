class Solution {
    public int getMaxLen(int[] nums) {
        int n = nums.length;
        
int max = 0;
        int first = 1000001, last = -1, start = 0;
        int sign = 1;
        for(int i = 0; i < n; i++)
        {
            if(nums[i] == 0){
                
first = 1000001;
                last = -1;
                start = i + 1;
                sign = 1;
            } else if(nums[i] < 0){
                if(first > 1000000) {
                    first = i;
                }
                last = i;
                sign *= -1;
            }
            if(nums[i] == 0) continue;
            
if(sign == 1) {
                max = Math.max(max, i - start + 1);
            } else {
                max = Math.max(max, Math.max(i - first, last - start));
            }
            
        }
        return max;
    }
}
