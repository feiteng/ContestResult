class Solution {
    public int getMaxLen(int[] nums) {
		int n = nums.length;

		int count = 0;
		int firstNeg = -1;
		boolean isPos = true;

		int ans = 0;
		for (int i = 0; i < n; i++) {
			if (nums[i] == 0) {
				count = 0;
				firstNeg = -1;
				isPos = true;
			} else {
				count++;
				if (nums[i] < 0) {
					isPos = !isPos;
					if (firstNeg == -1) {
						firstNeg = count;
					}

				}

				if (isPos == true) {
					ans = Math.max(ans, count);
				} else {
					ans = Math.max(ans, count - firstNeg);
				}
			}
		}

		return ans;
	}
}
