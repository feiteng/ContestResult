class Solution {
    public int getMaxLen(int[] nums) {
        // [1,-2,-3,4]
        int rt=0;
        int poslen=0, neglen=0, temp=0;
        for (int i=0; i<nums.length; i++) {
            if (nums[i]==0) {
                poslen = neglen = 0;
            } else if (nums[i]>0) {
                poslen = poslen + 1;
                neglen = neglen>0?neglen+1:0;
            } else {
                temp = poslen;
                poslen = neglen>0?neglen+1:0;
                neglen = temp+1;
            }
            rt = Math.max(rt, poslen);
        }
        return rt;
    }
}
