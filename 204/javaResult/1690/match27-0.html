class Solution {
    public int getMaxLen(int[] nums) {
        int res = 0;
        int last = -1;
        int neg = nums.length, cur = 1;
        
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                last = i;
                neg = nums.length;
                cur = 1;
                continue;
            }
            if (nums[i] < 0) {
                cur = -cur;
                neg = Math.min(neg, i);
            }
            if (cur > 0) {
                res = Math.max(res, i - last);
            }
            else {
                res = Math.max(res, Math.max(neg - last - 1, i - neg));
            }
            
            // System.out.println(last + " " + neg + " " + cur);
        }
        return res;
    }
}
