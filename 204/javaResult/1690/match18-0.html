class Solution {
    public int getMaxLen(int[] nums) {
        int longestPositive = 0;
        int longestNegative = 0;
        int bestPositive = 0;
        for (int v : nums) {
            if (v == 0) {
                longestPositive = 0;
                longestNegative = 0;
            } else if (v > 0) {
                ++longestPositive;
                if (longestNegative > 0) {
                    ++longestNegative;
                }
            } else {
                int oldPos = longestPositive;
                if (longestNegative > 0) {
                    longestPositive = 1 + longestNegative;
                } else {
                    longestPositive = 0;
                }
                longestNegative = 1 + oldPos;
            }
            bestPositive = Math.max(bestPositive, longestPositive);
        }
        return bestPositive;
    }
}
