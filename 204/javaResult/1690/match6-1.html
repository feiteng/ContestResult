class Solution {
    public int getMaxLen(int[] nums) {
    int maxl = 0;
    int b = 0;
    int nn = 0;   // number of negative
    int nb = -1;  // first index of negative
    int ne = -1;  // last index of negative
    for (int i = 0; i <= nums.length; i++) {
      if (i == nums.length || nums[i] == 0) {
        if (nn % 2 == 0) {
          maxl = Math.max(maxl, i - b);
        } else {
          maxl = Math.max(maxl, 
Math.max(ne - b, i - nb - 1));
        }
        nb = ne = -1;
        nn = 0;
        b = i+1;
      } else if (nums[i] < 0) {
        if (nb < 0) {
          nb = ne = i;
        } else {
          ne = i;
        }
        nn++;
      
}
    }
    return maxl;        
    }
}
