class Solution {
    public int getMaxLen(int[] nums) {
        int negatives = 0;
        int lastZero = -1;
        int firstNegIndex = -1;
        int lastNegIndex = -1;
        int best = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < 0) {
                if (firstNegIndex < 0) {
                    firstNegIndex = i;
                }
                lastNegIndex = i;
                negatives++;
            } else if (nums[i] == 0) {
                if (negatives % 2 == 0) {
                    best = Math.max(best, i - lastZero - 1);
                } else {
                    best = Math.max(best, lastNegIndex - lastZero - 1);
                    best = Math.max(best, i - firstNegIndex - 1);
                }
                firstNegIndex = -1;
                lastNegIndex = -1;
                negatives = 0;
                lastZero = i;
            }
        }
        int i = nums.length;
        if (negatives % 2 == 0) {
            best = Math.max(best, i - lastZero - 1);
        } else {
            best = Math.max(best, lastNegIndex - lastZero - 1);
            best = Math.max(best, i - firstNegIndex - 1);
        }

        return best;
    }
}
