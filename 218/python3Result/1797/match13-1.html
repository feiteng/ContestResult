class Solution:
    def interpret(self, command: str) -> str:
        ret=[]
        i=0
        while i<len(command):
            c=command[i]
            if c=='G':
                ret.append(c)
                i+=1
            elif c=='(':
                if command[i+1]=='a':
                    ret.append('al')
                    i+=4
                else:
                    ret.append('o')
                    i+=2
        return ''.join(ret)
