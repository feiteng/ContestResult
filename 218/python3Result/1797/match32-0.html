class Solution:
    def interpret(self, command: str) -> str:
        ans = []
        
        idx = 0
        
        while idx < len(command):
            if command[idx] == 'G':
                ans.append('G')
                idx += 1
            elif command[idx] == '(':
                if command[idx+1] == 'a':
                    ans.append('al')
                    idx += 4
                else:
                    ans.append('o')
                    idx += 2
        
        return ''.join(ans)
