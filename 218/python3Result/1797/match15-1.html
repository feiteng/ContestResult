class Solution:
    def interpret(self, command: str) -> str:
        ret = ''
        i = 0
        while i < len(command):
            if command[i] == 'G':
                ret += 'G'
                i += 1
            elif command[i+1] == ')':
                ret += 'o'
                i += 2
            else:
                ret += 'al'
                i += 4
        return ret
