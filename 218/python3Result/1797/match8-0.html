class Solution:
    def interpret(self, command: str) -> str:
        ret = []
        index = 0
        while index < len(command):
            if command[index] == 'G':
                ret.append('G')
                index += 1
            elif command[index] == '(' and command[index+1] == ')':
                ret.append('o')
                index += 2
            else:
                ret.append('al')
                index += 4
        return ''.join(ret)
