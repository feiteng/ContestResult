class Solution:
    def interpret(self, command: str) -> str:
        i = 0
        length = len(command)
        res = []
        while i < length:
            if command[i] == 'G':
                res.append('G')
                i += 1
            elif command[i] == '(' and i < length -1 and command[i+1] == ')':
                res.append('o')
                i += 2
            else:
                res.append('al')
                i += 4
        return ''.join(res)
