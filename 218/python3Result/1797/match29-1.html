class Solution:
    def interpret(self, command: str) -> str:
        res = []
        
        for i, char in enumerate(command):
            if char == 'G':
                res.append('G')
            elif char == '(':
                if command[i+1] == ')':
                    res.append('o')
                else:
                    res.append('al')
        
        return ''.join(res)
