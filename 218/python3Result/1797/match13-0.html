class Solution:
    def interpret(self, command: str) -> str:
        ans = []
        i = 0
        while i < len(command):
            c = command[i]
            if c == 'G':
                ans.append(c)
                i += 1
            elif c == '(':
                if command[i+1] == ')':
                    ans.append('o')
                    i += 2
                else:
                    ans.append('al')
                    i += 4
        return ''.join(ans)
