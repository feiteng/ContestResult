class Solution:
    def interpret(self, command: str) -> str:
        interpreted = []
        i = 0
        while i < len(command):
            if command[i] == 'G':
                interpreted.append('G')
                i += 1
            elif command[i] == '(' and command[i + 1] == ')':
                interpreted.append('o')
                i += 2
            else:
                interpreted.append('al')
                i += 4
        return ''.join(interpreted)
