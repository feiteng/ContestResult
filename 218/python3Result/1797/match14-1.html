class Solution:
    def interpret(self, command: str) -> str:
        ans = []
        i = 0
        while i < len(command):
            elem = command[i]
            if elem == 'G':
                ans.append(elem)
                i += 1
            elif command[i+1] == ')':
                ans.append('o')
                i += 2
            else:
                ans.append('al')
                i += 4
        return ''.join(ans)
