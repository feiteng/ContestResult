class Solution:
    def interpret(self, S: str) -> str:
        ans = ''
        N = len(S)
        i = 0
        while i < N:
            if S[i] == 'G':
                ans += 'G'
                i += 1
            elif S[i+1] == ')':
                ans += 'o'
                i += 2
            else:
                ans += 'al'
                i += 4
        return ans
