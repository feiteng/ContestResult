class Solution:
    def interpret(self, command: str) -> str:
        res = []
        n = len(command)
        i = 0
        while i<n:
            if command[i] == "G": 
                res.append("G")
                i += 1
            elif command[i+1] == ")":
                res.append("o")
                i += 2
            else:
                res.append("al")
                i += 4
        return "".join(res)
