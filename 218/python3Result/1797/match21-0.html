class Solution:
    def interpret(self, command: str) -> str:
        n = len(command)
        result = []
        i = 0
        while i < n:
            if command[i] == 'G':
                result.append('G')
                i += 1
            elif command[i:i+2] == "()":
                result.append("o")
                i += 2
            else:
                result.append("al")
                i += 4
        return "".join(result)
