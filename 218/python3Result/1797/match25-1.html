class Solution:
    def interpret(self, command: str) -> str:
        output = []
        i = 0
        while i < len(command):
            if command[i] == "G":
                output.append("G")
                i += 1
            elif command[i + 1] == "a":
                output.append("al")
                i += 4
            else:
                output.append("o")
                i += 2
                
        return "".join(output)
