class Solution:
    def interpret(self, command: str) -> str:
        res = []
        i = 0
        while i < len(command):
            if command[i] == 'G':
                res.append('G')
                i += 1
            elif command[i + 1] == 'a':
                res.append('al')
                i += 4
            else:
                res.append('o')
                i += 2
        return ''.join(res)
