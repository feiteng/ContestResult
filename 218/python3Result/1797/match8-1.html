class Solution:
    def interpret(self, s: str) -> str:
        i = 0
        res = []
        while i < len(s):
            if s[i] == 'G':
                res.append('G')
                i += 1
            elif s[i] == '(' and s[i + 1] == 'a':
                res.append('al')
                i += 4
            else:
                res.append('o')
                i += 2
        return "".join(res)
