class Solution:
    def interpret(self, S: str) -> str:
        l = len(S)
        i = 0
        ans = []
        while i < l:
            if S[i] == "G":
                ans.append("G")
                i += 1
            elif S[i:i+2] == "()":
                ans.append("o")
                i += 2
            else:  # "(al)"
                ans.append("al")
                i += 4
        return "".join(ans)
