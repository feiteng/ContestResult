class Solution:
    def interpret(self, command: str) -> str:
        ans = []
        n = len(command)
        idx = 0
        while idx < n:
            if command[idx] == 'G':
                ans.append('G')
                idx += 1
            elif command[idx + 1] == ')':
                ans.append
('o')
                idx += 2
            else:
                ans.append('al')
                idx += 4
        return ''.join(ans)
