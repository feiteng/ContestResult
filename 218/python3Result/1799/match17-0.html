class Solution(object):
    def minimumIncompatibility(self, A, K):
        # K SUBSETS
        A.sort()
        N = len(A)
        B = N // K
        
        # bucket size B
        if B == 1:
            return 0
            #return -1
        
        INF = float('inf')
        TARGET = (1 << N) - 1
        @lru_cache(None)
        def dp(mask):
            if mask == TARGET:
                return 0
            elements = [i for i in range(N) if mask >> i & 1 == 0]
            ans = INF
            for choice in itertools.combinations(elements, B):
                if any(A[choice[i]] == A[choice[i-1]] for i in range(1, B)):
                    continue
                delta = A[choice[-1]] - A[choice[0]]
                mask2 = mask
                for i in choice:
                    mask2 |= 1 << i
                cand = delta + dp(mask2)
                if cand < ans: ans = cand
            return ans
        
        ans = dp(0)
        if ans == INF: return -1
        return ans
