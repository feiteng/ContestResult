from collections import Counter

class Solution:
    ans = 0
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        c = Counter(nums)
        if max(c.values())>k:
            return -1
        m = len(nums)//k
        ss_lst = []
        for j in range(k):
            ss_lst.append(set())
        nnnn = []
        for cc in c:
            nnnn.append([c[cc], cc])
        nnnn.sort(key=lambda x:[0-x[0],x[1]])
        #print(nnnn)
        nums = []
        for nnn in nnnn:
            for j in range(nnn[0]):
                nums.append(nnn[1])
        #print(nums)
        for num in nums:
            for j in range(k):
                if num not in ss_lst[j] and len(ss_lst[j]) < m:
                    ss_lst[j].add(num)
                    break
            #print(num, ss_lst)
        self.ans = 0
        for ss in ss_lst:
            self.ans += max(ss)-min(ss)
        #print(ss_lst)
        for j in range(k):
            ss_lst[j] = set()

        def dfs(nums, ss_lst, cur, k):
            tmp = 0
            for ss in ss_lst:
                if len(ss) > 0:
                    tmp += max(ss)-min(ss)
            if tmp >= self.ans:
                return
            if cur == len(nums):
                self.ans = tmp
                #print(ss_lst)
                return
            #print(cur)
            num = nums[cur]
            for j in range(k):
                if j>0 and len(ss_lst[j]) == 0 and len(ss_lst[j-1]) == 0:
                    continue
                if num not in ss_lst[j] and len(ss_lst[j]) < m:
                    ss_lst[j].add(num)
                    dfs(nums, ss_lst, cur+1, k)
                    ss_lst[j].remove(num)
        dfs(nums, ss_lst, 0, k)
        return self.ans
