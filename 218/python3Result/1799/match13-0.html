class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        n = len(nums)
        m = n // k
        INF = 9876543
        from functools import lru_cache
        
        @lru_cache(None)
        def calc(msk):
            seen = set()
            mx = 0
            mi = 100
            for i, v in enumerate(nums):
                if msk & (1 << i):
                    if v in seen:
                        return INF
                    seen.add(v)
                    mx = max(mx, v)
                    mi = min(mi, v)
            # print(msk, mx - mi)
            return mx - mi
        
        @lru_cache(None)
        def dfs(msk):
            if msk == 0: return 0
            ans = INF
            cdd = [i for i in range(n) if msk & (1 << i)]
            for now in itertools.combinations(cdd, m):
                nmsk = sum(1 << x for x in now)
                ans = min(ans, calc(nmsk) + dfs(msk - nmsk))
            return ans
            
                    
            
            
        ans = dfs((1 << n) - 1)
        # print("--------")
        return ans if ans < INF else -1
