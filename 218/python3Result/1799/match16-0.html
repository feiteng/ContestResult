class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        N = len(nums)
        if N % k != 0:
            return -1
        
        candidates = []
        scores = {}
        
        per = N // k
        for mask in range(1 << N):
            c = 0
            score = 0
            for j in range(N):
                if (mask & (1 << j)) > 0:
                    c += 1
                    
            if c == per:
                good = True
                s = set()
                
                for j in range(N):
                    if (mask & (1 << j)) > 0:
                        if nums[j] in s:
                            good = False
                            break
                        s.add(nums[j])
                
                if good:
                    candidates.append(mask)
                    scores[mask] = max(s) - min(s)

        INF = (1 << 60)
                
        @lru_cache(None)
        def go(mask):
            if mask == 0:
                return 0
            
            best = INF
            for submask in candidates:
                if (mask & submask) == submask:
                    ans = go(mask ^ submask) + scores[submask]
                    if ans < best:
                        best = ans
                        
            return best
        
        ans = go((1 << N) - 1)
        if ans == INF:
            return -1
        return ans
