class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        if len(nums) % k != 0:
            return -1
        
        n = len(nums)
        counter = collections.Counter(nums)
        if max(counter.values()) > k:
            return - 1
        
        bucket = len(nums) // k
        full_mask = (1 << n) - 1
        
        @lru_cache(None)
        def dp(i, mask):
            #print(i, bin(mask))
            if mask == full_mask:
                #print("HAHAHAHAH", bin(full_mask))
                return 0
            
            idx = []
            for j in range(n):
                if (mask >> j) & 1 == 0:
                    idx.append(j)
            
            #print(idx)
            ans = float('inf')
            for idx_choice in itertools.combinations(idx, bucket):
                cur = [nums[j] for j in idx_choice]
                if len(set(cur)) != len(cur):
                    continue
                nxt_mask = mask
                #print(idx_choice, cur)
                for j in idx_choice:
                    nxt_mask |= (1 << j)
                ans = min(ans, max(cur) - min(cur) + dp(i + 1, nxt_mask))
            
            #print(i, bin(mask), ans)
            return ans
                    
                
        ans = dp(0, 0)
        return ans if ans < float('inf') else -1
