class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        counter=Counter(nums)
        if max(counter.values())>k:
            return -1
        nums.sort()
        buckets=[[] for _ in range(k)]
        size=len(nums)//k
        def dfs(idx):
            if idx>=len(nums):
                return sum(vals[-1]-vals[0] for vals in buckets)
            res=math.inf
            empty=[bucket for bucket in buckets if not bucket]
            if empty:
                bucket=empty[0]
                bucket.append(nums[idx])
                res=min(res,dfs(idx+1))
                bucket.pop()
            nxt=[bucket for bucket in buckets if bucket and bucket[-1]!=nums[idx] and len(bucket)<size]
            nxt2=sorted(nxt,key=lambda x:x[0])[:3]
            for bucket in nxt2:
                bucket.append(nums[idx])
                res=min(res,dfs(idx+1))
                bucket.pop()
            return res
        return  dfs(0)
