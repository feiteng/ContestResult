class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        cnt = collections.Counter(nums)
        if max(cnt.values()) > k: return -1
        
        nums.sort()
        boxes = collections.Counter()
        
        for i in range(k):
            boxes[tuple()] += 1

        self.res = sys.maxsize
        capacity = len(nums) // k
        
        def dfs(curr, boxes, tot):
            if curr == len(nums):
                self.res = min(self.res, tot)
                return
            
            n = nums[curr]
            # print("-----", curr, n)
            for b in boxes:
                # print(b)
                if len(b) == capacity: continue
                if n in b: continue
                    
                newkey = tuple(list(b) + [n])
                diff = 0 if len(b) == 0 else n - b[-1]
                newtot = tot + diff
                if newtot >= self.res: continue
                    
                newbox = boxes.copy()
                newbox[newkey] += 1
                newbox[b] -= 1
                
                if newbox[b] == 0: del newbox[b]
                
                # print(newkey)
                # print(newbox)
                
                dfs(curr+1, newbox, newtot)
        
        dfs(0, boxes, 0)
        return self.res
