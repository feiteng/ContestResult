class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        l = len(nums)//k
        nums.sort()
        c = collections.Counter
(nums)
        if any(m>k for m in c.values()):
            return -1
        if k==1:
            return max(nums)-min(nums)
        if k==len(nums):
            return 0
        res = float('inf')
        def helper(count,r):
            nonlocal res
            s = [n for n in c if c[n]>0]
            if len(s)<l:
                if not s:
                    res = min(res, r)
                return 
            if r>=res:
                return
            for t in itertools.combinations(s[1:],l-1):
                t += (s[0],)
                for n in t:
                    count[n] -= 1
                helper(count,r+max(t)-min(t))
                for n in t:
                    count[n] += 1
            return 
        helper(c,0)
        return res
