class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        n = len(nums)
        finalstate = (1 << n) - 1
        nums.sort()
        k = n // k
        
        @functools.lru_cache(None)
        def dp(state):
            if state == finalstate:
                return 0
            cho = []
            for i in range(n):
                if (state >> i) & 1 == 0:
                    cho.append(i)
            cost = float('inf')
            for chos in combinations(cho, k):
                # print(chos)
                flag = False
                for j in range(k-1):
                    if nums[chos[j]] == nums[chos[j+1]]:
                        flag = True
                        break
                if flag:
                    continue
                cur = 0
                for j in range(k):
                    cur ^= 1 << chos[j]
                # print(cur)
                cost = min(cost, nums[chos[k-1]] - nums[chos[0]] + dp(state ^ cur))
            return cost
        
        ans = dp(0)
        return ans if ans != float('inf') else -1
