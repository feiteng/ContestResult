class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        d = collections.Counter(nums)
        if max(d.values()) > k:
            return -1
        global seg_len
        seg_len = len(nums)//k
        @functools.lru_cache(maxsize=None)
        def helper(arr):
            global seg_len
            #print("input=",arr)
            if not arr:
                return 0
            out = float('inf')
            x = collections.Counter(arr)
            for c in combinations(arr, seg_len):
                #print(c)
                if len(set(c)) < seg_len:
                    continue
                for a in c:
                    x[a] -=1
                cur = max(c) - min(c)
                next_arr = []
                for key in sorted(x.keys()):
                    next_arr.extend([key]*x[key])
                out = min(out, cur+helper(tuple(next_arr)))
                #print(cur, helper(tuple(next_arr)),next_arr)
                for a in c:
                    x[a] +=1
            #print("out=",out)
            return out
        return helper(tuple(sorted(nums)))
