class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        nums.sort()
        n = len(nums)
        bucket = [set() for _ in range(k)]
        self.ans = math.inf
        def dfs(i):
            res = 0
            for b in bucket:
                if b:
                    res 
+= max(b) - min(b)
            if res >= self.ans:
                return
            if i == len(nums):
                self.ans = min(self.ans, res)
            else:
                seen = set()
                for b in bucket:
                    if tuple(b) not in seen and nums[i] not in b and len(b) < n // k:
                        b.add(nums[i])
                        dfs(i + 1)
                        b.discard(nums[i])
                        seen.add(tuple(b))
        dfs(0)
        return -1 if self.ans == math.inf else self.ans
