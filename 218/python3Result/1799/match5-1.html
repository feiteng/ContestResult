class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        n = len(nums)
        memo = {}
        def solve(mask):
            if mask in memo:
                return memo[mask]
            elif mask == (1<<n)-1:
                return 0
            
            indexes = [i for i in range(n) if not mask&(1<<i)]
            
            res = float("inf")
            for cm in itertools.combinations(indexes, r = n//k):
                temp 
= [nums[i] for i in cm]
                if len(set(temp)) != len(temp):
                    continue
                ma, mi = max(temp), min(temp)
                new_mask = mask
                for i in cm:
                    new_mask |= (1<<i)
                res = min(res, abs(ma-mi) + solve(new_mask))
            memo[mask] = res
            return res
        
        ans = solve(0)
        return ans if ans <= 256 else -1
