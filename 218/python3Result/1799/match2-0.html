class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        L = len(nums)
        k = L // k
            
        @lru_cache(None)
        def f(av):
            ns = []
            for i in range(L):
                if av & (1 << i) == 0:
                    ns.append(i)
            if not ns:
                return 0
            
            d = 1e9
            for c in itertools.combinations(ns, k):
                iii = c
                c = [nums[i] for i in c]
                if len(set(c)) != len(c):
                    continue
                
                m = av
                for n in iii:
                    m |= (1 << n)
                dd = max(c) - min(c)
                dd = dd + f(m)
                d = min(d, dd)
            # print(ns, d)
            return d
        
        r = f(0) 
        return r if r < 1e8 else -1
