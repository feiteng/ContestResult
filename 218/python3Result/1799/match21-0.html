class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        n = len(nums)
        l = n//k
        cnt = collections.Counter(nums)
        if max(cnt.values())>k:
            return -1
        if n==k:
            return 0
        
        total_mask = (1<<n)-1
        print(total_mask)
        @functools.lru_cache(None)
        def dfs(curr,mask):
            if mask == total_mask:
                print("k")
                return 0
            add = 0
            tmp = float('inf')
            seen = set()
            for j in range(n):
                if curr&(1<<j):
                    seen.add(nums[j])
            zzz = set()
            for j,num in enumerate(nums):
                if (1<<j)&mask==0 and num not in seen and num not in zzz:
                    zzz.add(num)
                    seen.add(num)
                    if len(seen)==l:
                        tmp = min(tmp,max(seen)-min(seen)+dfs(0,mask|(1<<j)))
                    else:
                        tmp = min(tmp,dfs(curr|(1<<j),mask|(1<<j)))
                    seen.remove(num)
                    
            #print(seen,curr,mask,tmp,add)
            return tmp+add
            
            
        return dfs(0,0)
