class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        def place(i: int) -> None:
            nonlocal result
            r = sum(max(b) - min(b) for b in buckets if b)
            if r >= result:
                return
            if i == len(nums):
                # print(buckets)
                result = r
                return
            v = nums[i]
            for b in buckets:
                if len(b) < length and v not in b:
                    b.add(v)
                    place(i+1)
                    b.remove(v)
                if not b:
                    break
        
        if not nums or k == len(nums):
            
return 0
        counter = Counter(nums)
        if max(counter.values()) > k:
            return -1
        length = len(nums) // k
        buckets = [set() for _ in range(k)]
        result = float('inf')
        place(0)
        return result
