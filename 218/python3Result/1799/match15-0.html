class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        nums.sort()
        n = len(nums)
        if k == n:
            return 0
        bb = Counter(nums)
        if any(x > k for x in bb.values()):
            return -1
        bu = defaultdict(set)
        size = n // k
        
        self.ans = float('inf')
        
        def back(i,curr):
            if curr > self.ans:
                return
            if i == k:
                self.ans = min(self.ans, curr)
            else:
                cc = 0
                for z in itertools.combinations(bb,size):
                    
                    minz = min(z)
                    maxz = max(z)
                    for x in z:
                        bb[x] -= 1
                        if bb[x] == 0:
                            del bb[x]
                    back(i + 1, curr + maxz - minz)
                    for x in z:
                        bb[x] += 1
                    cc += 1
                    if cc == 26:
                        break

        back(0,0)
        return self.ans
