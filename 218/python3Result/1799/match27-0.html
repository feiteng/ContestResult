class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        
        counter = collections.Counter(nums)
        
        if max(counter.values()) > k:
            return -1
        
        nums.sort()
        
        cur_set = [set() for i in range(k)]
        self.min_ = float("inf")
        
        self.dfs(cur_set, 0, nums, k)
        
        
        return self.min_
    
    def dfs(self, cur_set, cur_pointer, nums, k):
        if cur_pointer == len(nums):
            if self.getPoint(cur_set) < self.min_:
                self.min_ = self.getPoint(cur_set)
                #print(cur_set)
                
            return
        
        cur_num = nums[cur_pointer]
        for i in range(k):
            
            if cur_num in cur_set[i]:
                continue
            if i > 0 and len(cur_set[i - 1]) == 0:
                break
            if len(cur_set[i]) == len(nums) // k:
                continue
                
            
                
            cur_set[i].add(cur_num)
            if self.getPoint(cur_set) > self.min_:
                cur_set[i].remove(cur_num)
                continue
            self.dfs(cur_set, cur_pointer + 1, nums, k)
            cur_set[i].remove(cur_num)
            
    
    def getPoint(self, sets):
        
        result = 0
        
        for s in sets:
            temp = max(s) - min(s) if s else 0
            result = result + temp
            
        return result
