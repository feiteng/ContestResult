from typing import List
from functools import lru_cache
import sys
sys.setrecursionlimit(999999)


def bits(val):
    ans = 0

    while val:
        ans += 1
        val = val & (val - 1)
    return ans


from collections import Counter
class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        n = len(nums)
        if n % k != 0:
            return -1

        c = Counter(nums)
        max_val = max(c.values())
        if max_val > k:
            return -1

        possible_vals = []
        for val in range(1 << n):
            if bits(val) != n // k:
                continue

            min_val, max_val = 0x7fffffff, 0
            visit = [0] * 20

            flag = True
            for i in range(16):
                if val & (1 << i):
                    if visit[nums[i]] == 1:
                        flag = False
                        break
                    else:
                        visit[nums[i]] = 1

                        min_val = min(min_val, nums[i])
                        max_val = max(max_val, nums[i])

            if flag:
                possible_vals.append((val, max_val - min_val))

        # 已经选过的状态为mask情况下，选i组数值，最小的不兼容性
        @lru_cache(typed=False, maxsize=12800000)
        def dp(ii, mask):
            if ii == 0:
                return 0

            ans = 0x7fffffff
            for val, sub in possible_vals:
                if val & mask:
                    continue

                new_mask = mask | val
                ret = dp(ii-1, new_mask)
                if ret != 0x7fffffff:
                    ans = min(
sub + ret, ans)

            #print(bin(mask), ans)
            return ans


        ret = dp(k, 0)
        if ret == 0x7fffffff:
            return -1
        return ret
