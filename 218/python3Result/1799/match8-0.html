from itertools import combinations
from functools import lru_cache
class Solution:
    def minimumIncompatibility(self, nums: List[int], k: int) -> int:
        if len(nums) % k != 0:
            return -1
        for num in nums:
            if nums.count(num) > k:
                return -1
        size = len(nums) // k
        if size == 1:
            return 0
        used = [False] * len(nums)

        @lru_cache(None)
        def dp(used):
            if used == 2 ** (len(nums)) - 1: # all used
                return 0
            avail = [i for i in range(len(nums)) if not ((used >> i) & 1)]
            res = 257
            for l in combinations(avail, size):
                selected = [nums[ind] for ind in l]
                if len(set(selected)) < len(selected):
                    continue
                cur = max(selected) - min(selected)
                new_used = used
                for i in l:
                    new_used |= (1 << i)                
                res = min(res, cur + dp(new_used))
            return res
        res = dp(0)
        if res == 257:
            return -1
        return res
