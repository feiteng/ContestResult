class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        counter = collections.Counter()
        result = 0
        for i in nums:
            if counter[k - i] > 0:
                result += 1
                counter[k - i] -= 1
            else:
                counter[i] += 1
        return result
