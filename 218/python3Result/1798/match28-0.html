class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        counter = collections.defaultdict(int)
        
        res = 0
        for n in nums:
            if counter[k - n] > 0:
                res += 1
                counter[k - n] -= 1
            else:
                counter[n] += 1
        return res
