class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        cnt = defaultdict(int)
        
        out = 0
        for n in nums:
            if k-n in cnt and cnt[k-n] > 0:
                cnt[k-n] -= 1
                out += 1
            else:
                cnt[n] += 1
        return out
