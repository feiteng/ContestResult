class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        d = collections.defaultdict(int)
        ans = 0
        for item in nums:
            if d[k-item] > 0:
                ans += 1
                d[k-item] -= 1
            else:
                d[item] += 1
        return ans
