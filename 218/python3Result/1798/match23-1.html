class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        res = 0
        be = 0
        en = len(nums) - 1
        nums.sort()
        
        while be < en:
            if nums[be] + nums[en] == k:
                res += 1
                be += 1
                en -= 1
            elif nums[be] + nums[en] > k:
                en -= 1
            else:
                be += 1
        return res
