class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        c = Counter()
        res = 0
        for i in nums:
            if c[k - i] > 0:
                c[k - i] -= 1
                res += 1
            else:
                c[i] += 1
        
        return res
