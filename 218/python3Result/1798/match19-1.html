class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        count = collections.Counter()
        res = 0
        for n in nums:
            if k-n in count and count[k-n] > 0:
                count[k-n] -= 1
                res += 1
            else:
                count[n] += 1
        
        return res
