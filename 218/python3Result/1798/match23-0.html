class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        ans = 0
        hand01 = 0
        hand02 = len(nums) - 1
        nums.sort()
        while hand01 < hand02:
            if nums[hand01] + nums[hand02] == k:
                ans += 1
                hand01 += 1
                hand02 -= 1
            elif(nums[hand01] + nums[hand02] > k):
                hand02 -= 1
            else:
                hand01 += 1
        return ans
