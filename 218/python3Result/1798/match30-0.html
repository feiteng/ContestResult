from collections import Counter
class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        counter = Counter()
        count = 0
        for i in nums:
            if counter[k - i] > 0:
                counter[k - i] -= 1
                count += 1
            else:
                counter[i] += 1
        return count
