class Solution {
    private static int MOD = 1_000_000_000 + 7;
    public int concatenatedBinary(int n) {
        int res = 0;
        for (int i = 1; i <= n; i++) {
            String val = Integer.toBinaryString(i);
            //System.out.println(res + " " + val);
            for (char ch : val.toCharArray()) {
                res = res << 1;
                res += ch-'0';
                res %= MOD;
            }
        }
        return res;
    }
}
