class Solution {
    public int concatenatedBinary(int n) {
        long size = 0;
        long result = 0;
        for (int i = 1; i <= n; i++) {
            if ((i & (i - 1)) == 0) {
                size++;
            }
            result = ((result << size) | i) % (int) (1e9 + 7);
        }
        return (int)result;
    }
}
