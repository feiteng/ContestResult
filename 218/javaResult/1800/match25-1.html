class Solution {
  public int concatenatedBinary(int n) {
    final int N = 1_000_000_007;
    long res = 0;
    for (int i = 1; i <= n; i++) {
      int bits = bits(i);
      res = (res << bits) % N;
      res = (res + i) % N;
    }
    return (int) res;
  }
  
  private int bits(int n) {
    int bits = 0;
    while (n > 0) {
      n >>= 1;
      bits++;
    }
    return bits;
  }
}
