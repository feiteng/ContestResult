class Solution {
    public int concatenatedBinary(int n) {
        int len = 1;
        long res = 0;
        int base = (int)1e9 + 7;
        for(int i = 1; i <= n; i++){
            if(i >= (1 << len)){
                len++;
            }
            res = (res << len) + i;
            res %= base;
            //System.out.println(i + " " + res);
            
        }
        return (int)res;
    }
}
