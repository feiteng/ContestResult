class Solution {
    public int concatenatedBinary(int n) {
        long res = 0;
        int MOD = (int)1e9 + 7;
        for(int i = 1; i <= n; i++) {
            int len = Integer.toBinaryString(i).length();
            res = ((res << len) + i) % MOD; 
        }
        return (int)res;
    }
}
