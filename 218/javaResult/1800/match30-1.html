class Solution {
    private final static int mod = 1000000007;

    public int concatenatedBinary(int n) {
        long ans = 0;
        for (int i = 1; i <= n; i++) {
            int num = shi(i);
            ans = (ans << num) % mod;
            
ans = (ans + i) % mod;
        }
        return (int)ans;
    }

    private int shi(int x) {
        int res = 0;
        while (x > 0) {
            res++;
            x /= 2;
        }
        return res;
    }
}
