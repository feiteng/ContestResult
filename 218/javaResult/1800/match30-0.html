class Solution {
    
    private static final int MOD = 1000000007;
    
    public int concatenatedBinary(int n) {
        long ans = 1;
        for (int k = 2; k <= n; ++k) {
            int pow = binaryStringLength(k);
            ans = ((ans * (1 << pow)) + k) % MOD;
        }
        return (int)ans;
    }
    
    private int binaryStringLength(int n) {
        int ans = 0;
        while (n > 0) {
            n /= 2;
            ++ans;
        }
        return ans;
    }
}
