class Solution {
    
    int mod = (int) 1e9 + 7;
    
    public int concatenatedBinary(int n) {
        int ans = 0;
        int base = 1;
        for (int i = n; i > 0; --i) {
            for (int j = i; j > 0; j >>= 1) {
                if (1 == (j & 1)) {
                    ans = (ans + base) % mod;
                }
                base = (base << 1) % mod;
            }
        }
        return ans;
    }
}
