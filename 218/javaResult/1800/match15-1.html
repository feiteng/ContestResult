class Solution {
    public int concatenatedBinary(int n) {
        int mod = (int)1e9 + 7;
        int res = 1;
        for(int i = 2; i <= n; i++) {
            int j = i;
            while(j != 0) {
                res <<= 1;
                res %= mod;
                j >>= 1;
            }
            res += i;
        }
        return res;
    }
}
