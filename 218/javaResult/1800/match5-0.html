class Solution {
    final int MOD = (int)(1e9+7);
    public int concatenatedBinary(int n) {
        long res = 0;
        for (int i = 1; i <= n; i++) {
            int foo = 0, bar = 1;
            while(i >= bar) {
                bar <<= 1;
                foo++;
            }
            res = ((res << foo) % MOD + i) % MOD;
        }
        return (int)res;
    }
}
