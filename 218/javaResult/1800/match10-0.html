class Solution {
    public int concatenatedBinary(int n) {

        int MOD = 1_000_000_007;

        long val = 1;
        int bitCnt = 1;
        for (int i = 2; i <= n; i++) {


            if ((i & (i - 1)) == 0) {
                bitCnt++;
            }

            val = val << bitCnt;
            val += i;
            val %= MOD;
        }
        return (int) val;

    }
}
