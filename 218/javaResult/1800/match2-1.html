class Solution {
    public int concatenatedBinary(int n) {
        long ans = 0;

        int base = 0;
        for(int i=1;i<=n;i++){
            if((i&-i) == i){
                base++;
            }

            ans =(ans<<base)|i;
            ans%=1000000007;

        }
        return (int)ans;
    }
}
