class Solution {
    private static final int MOD = (int)1e9 + 7;
    public int concatenatedBinary(int n) {
        long ans = 0;
        int last = 1;;
        for (int i = n; i > 0; i--) {
            for (int t = i; t > 0; t >>= 1) {
                if ((t & 1) != 0) {
                    ans += last;
                
}
                last = (last << 1) % MOD;
            }
        }
        return (int)(ans % MOD);
    }
}
