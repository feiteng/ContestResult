class Solution {
    public int concatenatedBinary(int n) {
        int MAX = 1000000007;
        int count = 1, k = 1;
        long res = 0;
        for(int i = 1; i <= n; i++) {
            res = (res << k) + (long)i;
            res %= MAX;
            count--;
            if(count == 0) {
                count = (1 << k);
                k++;
            }
        }
        return (int)res;
    }
}
