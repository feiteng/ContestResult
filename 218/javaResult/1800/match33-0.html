class Solution {
    public int concatenatedBinary(int n) {
        long pre = 0;
        int mod = (int) 1e9 + 7;
        for(int i = 1; i <= n; i++)
        {
            String str = Integer.toBinaryString(i);
            for(char c : str.toCharArray())
            {
                int val = c - '0';
                pre = pre * 2 + val;
            }
            pre %= mod;
        }
        return (int)pre;
    }
}
