class Solution {
    public int concatenatedBinary(int n) {
        int k = 0;
        long res = 0L;
        int modulo = 1000000007;
        for (int i = 1; i <= n; i++) {
            if ((i & (i - 1)) == 0) k++;
            res = ((res << k) | i) % modulo;
        }
        return (int)res;
    }
}
