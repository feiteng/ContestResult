class Solution {
    
    private static final int mod = 1_000_000_007;
    
    public int concatenatedBinary(int n) {
        
        long out = 1;
        for (int i=2; i<=n; i++) {
            int shift = countDigitsInBinary(i);
            out = ((out << shift) ^ i) % mod; 
        }
        return (int)out;
    }
    
    private static int countDigitsInBinary(int n) {
        int digits = 0;
        while (n != 0) {
            digits++;
            n >>= 1;
        }
        return digits;
    }

}
