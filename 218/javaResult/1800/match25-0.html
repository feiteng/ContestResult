class Solution {
    public int concatenatedBinary(int n) {
        
        int MOD = (int) (1e9 + 7);
        
        long res = 0;
        
        for (int i = 1; i <= n; ++i) {
            int l = getLength(i);
            res = res << l;
            res = (res + i) % MOD;
        }
        
        return (int) (res % MOD);
    }
    
    private int getLength(int num) {
        
        int res = 0;
        
        while (num > 0) {
            num /= 2;
            res++;
        }
        
        return res;
    }
}
