class Solution {
    public int concatenatedBinary(int n) {
        int next = 1;
        int current = 1;
        int len = 1;
        long value = 0;
        for (int i = 1; i <= n; i++) {
            value = ((value << len) + i) % 1_000_000_007;
            current--;
            if (current == 0) {
                next <<= 1;
                current = next;
                len++;
            }
        }
        return (int) value;
    }
}
