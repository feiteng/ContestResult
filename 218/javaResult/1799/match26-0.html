class Solution {
    public int minimumIncompatibility(int[] nums, int k) {
        if (k == nums.length) {
            return 0;
        }
        Arrays.sort(nums);
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                count++;
            } else {
                count = 1;
            }
            if (count > k) {
                return -1;
            }
        }
        List<Integer>[] lists = new List[k];
        for (int i = 0; i < k; i++) {
            lists[i] = new ArrayList<>();
        }
        return mini(nums, 0, lists);
    }

    private int mini(int[] nums, int now, List<Integer>[] list) {
        if (now == nums.length) {
            int sum = 0;
            for (int i = 0; i < list.length; i++) {
                sum += list[i].get(list[i].size() - 1) - list[i].get(0);
            }
            return sum;
        } else {
            int min = 1000;
            boolean flag = false;
            for (int i = 0; i < list.length; i++) {
                if (!list[i].contains(nums[now]) && list[i].size() < nums.length / list.length) {
                    if (list[i].isEmpty() && !flag) {
                        flag = true;
                        list[i].add(nums[now]);
                        min = Math.min(min, mini(nums, now + 1, list));
                        list[i].remove(list[i].size() - 1);
                    } else if (list[i].isEmpty() && flag) {
                        break;
                    } else {
                        list[i].add(nums[now]);
                        min = Math.min(min, mini(nums, now + 1, list));
                        list[i].remove(list[i].size() - 1);
                    }
                }
            }
            return min;
        }
    }
}
