class Solution {
    int res = 0x3f3f3f3f;
    int n;
    boolean flag = false;
    int[] counts = new int[20];
    int[] dis = new int[20];
    List<Integer>[] pre = new ArrayList[20];
    public int minimumIncompatibility(int[] nums, int k) {

        n = nums.length;
        if (k == n)
            return 0;
        for (int num :nums)
            counts[num]++;
        int max = 0;
        for (int i = 1; i <= n; i++)
            max = Math.max(max,counts[i]);
        if (max > k)
            return -1;
        solve(0,k,0,new ArrayList<Integer>());
        return res ==  0x3f3f3f3f ? -1 : res;
    }

    private void solve(int i, int k, int pos, List<Integer> list) {
        if (i == k) {
            int ans = 0;
            for (int j = 0; j < k; j++) {
                ans += dis[j];
            }
            res = Math.min(res,ans);
        }
        if (n / k == list.size()) {
            dis[i] = list.get(list.size() - 1) - list.get(0);
            pre[i] = list;
            solve(i+1, k, 0, new ArrayList<Integer>());
            return;
        }

        while (pos <= n) {
            while (pos <= n && counts[pos] == 0)
                pos++;
            if (pos > n)
                return;
            if (i > 0 && pos < pre[i - 1].get(list.size())){
                pos++;
                continue;}
            list.add(pos);
            counts[pos]--;
            solve(i,k,pos + 1, list);
            counts[pos]++;
            list.remove(list.size() - 1);
            pos++;

        }
    }
}
