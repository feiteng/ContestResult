class Solution {
    int sol = 1000;
    public int minimumIncompatibility(int[] nums, int k) {
        if (k == nums.length) return 0;
        int[] ct = new int[17];
        for (int num : nums) {
            if (++ct[num] > k) return -1;
        }
        int n = nums.length / k;
        
        dfs(ct, new ArrayList<>(), 0, 0, k, n);
        
return sol;
    }
    
    public void dfs(int[] ct, List<Integer> oneset, int pos, int sum, int k, int n) {
        if (oneset.size() == n) {
            k--;
            sum += oneset.get(n-1) - oneset.get(0);
            if (k == 0) {
                //System.out.println(oneset.get(n-1) + " " + oneset.get(0));
                sol = Math.min(sol, sum);
                return;
            } else {
                dfs(ct, new ArrayList<>(), 0, sum, k, n);
                return;
            }
        } 
        if (pos == ct.length) return;
        if (oneset.size() == 0) {
            for (int i = 0; i < ct.length; i++) {
                if (ct[i] > 0) {
                    ct[i]--;
                    oneset.add(i);
                    dfs(ct, oneset, i + 1, sum, k, n);
                    oneset.remove(oneset.size()-1);
                    ct[i]++;
                    break;
                }
            }
            return;
        }
        
        for (int i = pos; i < ct.length; i++) {
            if (ct[i] > 0) {
                ct[i]--;
                
oneset.add(i);
                dfs(ct, oneset, i + 1, sum, k, n);
                oneset.remove(oneset.size() - 1);
                ct[i]++;
            }
        }
    }
}
