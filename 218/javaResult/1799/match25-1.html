class Solution {
    int ret = Integer.MAX_VALUE;
    int m;
    public int minimumIncompatibility(int[] nums, int k) {
        m = nums.length / k;
        Arrays.sort(nums);
        dfs(nums, k, new int[m][k], 0, new boolean[nums.length]);
        return ret == Integer.MAX_VALUE ? -1 : ret;
    }
    
    private void dfs(int[] nums, int k, int[][] comb, int pos, boolean[] used) {
        int row = pos / k;
        int col = pos % k;
        
if (pos == nums.length) {
            int tempRet = 0;
            for (int i = 0; i < k; i++) {
                tempRet += comb[m - 1][i] - comb[0][i];
            }
            ret = Math.min(ret, tempRet);
            return;
        }
//         int startIdx = 0;
//         if (row > 0) {
            
//         }
        // + (m - row) * (k - col) - 1 
        
for (int i = 0 ; i < nums.length; i++) {
            if (used[i]) continue;
            if (row > 0 && comb[row - 1][col] >= nums[i]) continue;
            if (col > 0 && comb[row][col - 1] > nums[i]) continue;
            if (i > 0 && nums[i] == nums[i - 1] && !used[i - 1]) continue;
            comb[row][col] = nums[i];
            used[i] = true;
            dfs(nums, k, comb, pos + 1, used);
            comb[row][col] = 0;
            used[i] = false;            
        }
    }
}
