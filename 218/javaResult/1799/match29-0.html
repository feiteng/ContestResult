class Solution {
    int res = Integer.MAX_VALUE;
    public int minimumIncompatibility(int[] nums, int k) {
        int n = nums.length;
        if(n==k) return 0;
        TreeMap[] maps = new TreeMap[k];
        for(int i=0; i<k; i++){
            maps[i] = new TreeMap<>();
        }
        Arrays.sort(nums);
        dfs(nums, k, 0, maps, 0);
        return res==Integer.MAX_VALUE ? -1 : res;
    }
    
    public void dfs(int[] nums, int k, int pos, TreeMap<Integer, Integer>[] maps, int last){
        int n = nums.length;
        if(pos==n){
            int cur = 0;
            for(int i=0; i<k; i++){
                cur += maps[i].lastEntry().getValue() - maps[i].firstEntry().getValue();
            }
            res = Math.min(res, cur);
            return;
        }
        int begin = 0;
        if(pos>0 && nums[pos]==nums[pos-1]){
            begin = last+1;
        }
        for(int i=begin; i<k; i++){
            if(i>0 && maps[i-1].isEmpty() && maps[i].isEmpty()) break;
            if(maps[i].size() < n/k && !maps[i].containsKey(nums[pos])){
                maps[i].put(nums[pos], nums[pos]);
                dfs(nums, k, pos+1, maps, i);
                maps[i].remove(nums[pos]);
            }
        }
    }
}
