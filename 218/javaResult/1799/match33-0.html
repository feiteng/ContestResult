class Solution {
    int ans = Integer.MAX_VALUE;
    public int minimumIncompatibility(int[] nums, int k) {
        if (k == nums.length)
            return 0;
        Arrays.sort(nums);
        int[] f = new int[20];
        for (int num:nums)
            f[num]++;
        for (int i = 0; i < 20; i++)
            if (f[i] > k)
                return -1;
        
        ArrayList<ArrayList<Integer>> list = new ArrayList<>();
        for (int i = 0; i < k; i++)
            list.add(new ArrayList<>());
        traverse(list, k, nums, 0, 0);
        return ans;
    }
    
    public void traverse(ArrayList<ArrayList<Integer>> list, int k, int[] nums, int idx, int cur) {
        if (idx == nums.length) {
            // System.out.println(list);
            ans = Math.min(ans, cur);
            return;
        }
        
        boolean[] good = new boolean[k];
        for (int i = 0; i < k; i++) {
            if (list.get(i).size() == 0 || nums[idx] != list.get(i).get(list.get(i).size()-1)) {
                if (list.get(i).size() != nums.length/k) {
                    good[i] = true;
                }
            }
            if (!good[i])
                continue;
            for (int j = 0; j < i; j++) {
                if (list.get(i).size() != list.get(j).size())
                    continue;
                boolean same = true;
                for (int l = 0; l < list.get(i).size(); l++)
                    if (list.get(i).get(l) != list.get(j).get(l)) {
                        same = false;
                        break;
                    }
                if (same)
                    good[i] = false;
            }
        }
        for (int i = 0; i < k; i++) {
            if (cur >= ans)
                return;
            if (good[i]) {
                int imp = 0;
                if (list.get(i).size() >= 1)
                    imp = nums[idx]-list.get(i).get(list.get(i).size()-1);
                list.get(i).add(nums[idx]);
                traverse(list, k, nums, idx+1, cur+imp);
                list.get(i).remove(list.get(i).size()-1);
            }
        }
    }
}
