    int res = Integer.MAX_VALUE;
    
    public int minimumIncompatibility(int[] nums, int v) {
        int n = nums.length;
        
TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
            if (map.get(i) > v) return -1;
        }
        int each = n / v;
        int row = v, col = each;
        int[][] m = new int[row][col];
        Arrays.sort(nums);
        dfs(nums, new int[row], 0, m);
        return res;

    void dfs(int[] nums, int[] loc, int idx, int[][] m) {
        // for (int[] r: m) {
        //     System.out.println(Arrays.toString(r));
        // }
        if (idx >= nums.length) {
            int ans = check(m);
            res = Math.min(res, ans);
            //System.out.println("res is " + res);
            return;
        }
        Set<String> seen = new HashSet<>();
        int row = m.length, col = m[0].length;
        boolean zerostart = false;
        for (int i = 0; i < row; i++) {
            if (loc[i] >= col || (loc[i] > 0 && m[i][loc[i] - 1] == nums[idx])) continue;
            if (loc[i] == 0) {
                if (!zerostart) zerostart = true;
                else continue;
            }
            m[i][loc[i]] = nums[idx];
            loc[i]++;
            dfs(nums, loc, idx + 1, m);
            loc[i]--;
            m[i][loc[i]] = 0;
        }
    }
    
    private int check(int[][] m) {
        int ans = 0;
        int col = m[0].length;
        for (int i = 0; i < m.length; i++) {
            ans += m[i][col-1] - m[i][0];
        }
        //System.out.println("check is " + ans);
        return ans;
    }
