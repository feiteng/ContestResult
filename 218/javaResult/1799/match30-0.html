class Solution {
    int res = Integer.MAX_VALUE;
    public int minimumIncompatibility(int[] nums, int k) {
        int[] cnt = new int[17];
        for(int num : nums){
            cnt[num]++;
            if(cnt[num] > k)
                return -1;
        }
        int len = nums.length;
        if(k == len)
            return 0;
        if(k == 1){
            Arrays.sort(nums);
            return nums[len - 1] - nums[0];
        }
        helper(cnt, -1, -1, 0, len, len / k, len / k);
        return res;
    }
    
    public void helper(int[] cnt, int pre, int min, int cur, int rest, int k, int curk){
        //System.out.println(rest + " " + cur + " " + curk + " " + pre + " " + min);
        if(rest == 0){
            res = Math.min(res, cur);
            return;
        }
        
        int i = Math.max(0, pre);
        for(; i <= 16; i++){
            if(cnt[i] == 0 || pre == i)
                continue;
            int temp = cur;
            int curkk = curk - 1;
            int npre = i;
            int nmin = pre >= 0 ? min : i;
            if(curkk == 0){
                temp += i - nmin;
                //System.out.println(temp);
                curkk = k;
                npre = -1;
                
            }
            if(temp > res)
                continue;
            cnt[i]--;
            helper(cnt, npre, nmin, temp, rest - 1, k, curkk);
            cnt[i]++;
            
        }
    }
}
