class Solution {
    // [4,10,4,7,9,10,4,6,9,10]
    // 4 4 4 6 7 9 9 10 10 10
    int res = 1000;
    public int minimumIncompatibility(int[] nums, int k) {
        if (k == nums.length) {
            return 0;
        }
        Arrays.sort(nums);
        
int[] cnt = new int[17];
        for (int num : nums) {
            cnt[num]++;
        }
        for (int c : cnt) {
            if (c > k) {
                return -1;
            }
        }
        int n = nums.length / k;
        int[][] arrs = new int[k][n];
        
backtrack(nums, 0, arrs, n);
        return res;
    }
    
    private void backtrack(int[] nums, int i, int[][] arrs, int n) {
        if (i == nums.length) {
            int temp = 0;
            for (int[] arr : arrs) {
                temp += arr[n - 1] - arr[0];
            }
            
res = Math.min(res, temp);
            return;
        }
        boolean newLineTried = false;
        for (int r = 0; r < arrs.length; r++) {
            int[] arr = arrs[r];
            int id = firstZeroIndex(arr);
            if (id == -1) {
                continue;
            } else if (id == 0) {
                if (newLineTried) {
                    continue;
                }
                arr[0] = nums[i];
                backtrack(nums, i + 1, arrs, n);
                arr[0] = 0;
                newLineTried = true;
            } else if (arr[id - 1] == nums[i]) {
                continue;
            } else {
                arr[id] = nums[i];
                backtrack(nums, i + 1, arrs, n);
                arr[id] = 0;
            }
        }
    }
    
    private int firstZeroIndex(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 0) {
                return i;
            }
        }
        return -1;
    }
}
