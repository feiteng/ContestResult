class Solution {
    int res;
    public int minimumIncompatibility(int[] nums, int k) {
        int size = nums.length/k;
        int[] cnts = new int[17];
        Arrays.sort(nums);
        for(int num : nums){
            cnts[num] ++;
        }
        
for(int cnt : cnts){
            if(cnt > k){
                return -1;
            }
        }
        if(k == nums.length){
            return 0;
        }
        res = Integer.MAX_VALUE;
        
List<TreeSet<Integer>> list = new ArrayList<>();
        for(int i=0; i<k; i++){
            list.add(new TreeSet());
        }
        dfs(nums,k,size,0,list);
        return res;
    }
    private void dfs(int[] nums, int k, int size, int startIndex, List<TreeSet<Integer>> list){
        if(startIndex == nums.length){ // done
            int sum = 0;
            for(TreeSet<Integer> set : list){
                sum+= set.last() - set.first();
            }
            res = Math.min(res,sum);
            return;
        }
        int cur = nums[startIndex];
        for(int i=0; i<k; i++){
            if(list.get(i).size() >= size) continue
;
            if(list.get(i).contains(cur)){
                continue;
            }
            if(list.get(i).size() == 0){
                list.get(i).add(cur);
                dfs(nums,k,size,startIndex+1,list);
                list.get(i).remove(cur);
                break;
            }
            
list.get(i).add(cur);
            dfs(nums,k,size,startIndex+1,list);
            list.get(i).remove(cur);
        }
    }
}
