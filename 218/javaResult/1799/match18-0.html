class Solution {
    int ans;
    public int minimumIncompatibility(int[] nums, int k) {
        int[] map = new int[17];
        for(int i : nums) {
            map[i]++;
            if(map[i] > k)
                return -1;
        }
        
        ans = Integer.MAX_VALUE;
        HashSet[] sets = new HashSet[k];
        int[][] mm = new int[k][2];
        for(int i = 0; i < k; i ++) {
            sets[i] = new HashSet<Integer>();
            mm[i][1] = Integer.MAX_VALUE;
        }
        
        f(nums, 0, k, nums.length/k, sets, mm);
        return ans;
    }
    
    void f(int[] nums, int curr, int k, int n, HashSet[] sets, int[][] mm) {
        int x = 0;
        for(int[] a : mm) {
            if(a[1] != Integer.MAX_VALUE)
                x += a[0] - a[1];
        }
        if(x >= ans) return;
        if(curr == nums.length) {
            
            // System.out.println(x);
            if(ans > x)
                ans = x;
            return;
        }
        
        for(int i = 0; i < k; i++) {
            if(sets[i].contains(nums[curr]) || sets[i].size() == n) {
                continue;
            }
            if(sets[i].isEmpty() && curr < i)
                continue;
            sets[i].add(nums[curr]);
            int min = mm[i][1], max = mm[i][0];
            if(mm[i][0] < nums[curr])
                mm[i][0] = nums[curr];
            if(mm[i][1] > nums[curr])
                mm[i][1] = nums[curr];
            f(nums, curr+1, k, n, sets, mm);
            sets[i].remove(nums[curr]);
            mm[i][0] = max;
            mm[i][1] = min;
        }
    }
}
