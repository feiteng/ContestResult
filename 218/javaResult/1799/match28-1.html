class Solution {
    int result = -1;
    int each = 0;
    public int minimumIncompatibility(int[] nums, int k) {
        each = nums.length / k;        
        if (k == nums.length) {        
            return 0;
        }
        
List<Set<Integer>> li = new LinkedList<>();
        for (int i = 0; i < k; i++) {
            li.add(new HashSet<>());
        }
        helper(nums, 0, li);
        return result;
    }
    public void helper(int[] nums, int level, List<Set<Integer>> path) {
        if (level == nums.length) {
            cal(path);
            return;
        }
        if (result != -1 && cal2(path) >= result) {
            return;
        }
        for (int i = 0; i < path.size(); i++) {
            Set<Integer> cur = path.get(i);
            if (cur.contains(nums[level]) || cur.size() == each) {
                continue;
            } else {
                cur.add(nums[level]);
                helper(nums, level + 1, path);
                cur.remove(nums[level]);
                if (i != path.size() - 1 && cur.size() == 0 && cur.size() == path.get(i + 1).size()) {
                    break;
                }
            }
        }
    }
    public void cal(List<Set<Integer>> path) {
        int temp = 0;
        for (int i = 0; i < path.size(); i++) {
            Set<Integer> cur = path.get(i);
            
int smallest = Integer.MAX_VALUE;
            int big = Integer.MIN_VALUE;
            for (int p : cur) {
                smallest = Math.min(p, smallest);
                big = Math.max(p, big);
            }
            temp += big - smallest;
        }
        if (result == -1) {
            result = temp;
        } else {
            result = Math.min(result, temp);
        }        
    }
    public int cal2(List<Set<Integer>> path) {
        int temp = 0;
        for (int i = 0; i < path.size(); i++) {
            Set<Integer> cur = path.get(i);
            
int smallest = Integer.MAX_VALUE;
            int big = Integer.MIN_VALUE;
            for (int p : cur) {
                smallest = Math.min(p, smallest);
                big = Math.max(p, big);
            }
            temp += big - smallest;
        }
        return temp;       
    }
}
