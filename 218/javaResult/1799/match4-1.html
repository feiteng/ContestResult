class Solution {
    
    int r = 1000000000;
    
    public void dfs(int v,
ArrayDeque st[],int g,int nums[],int k){
        
        if(v==nums.length){
            int t = 0;
            for(int j=0;j<k;++j){
                
                int f =(int)st[j].peekFirst();
                int s = (int)st[j].peekLast();
                t += Math.abs(f-s);
            }
            r = Math.min(r,t);
            return;
        }
        
        for(int i=0;i<k;++i){
            if(st[i].size()==g) continue;
            if(st[i].size()>0){
            int f =(int)st[i].peekLast();
            if(nums[v]==f){
                continue;
            }
            }
            st[i].offer(nums[v]);
            dfs(v+1,st,g,nums,k);
            st[i].pollLast();
            if(st[i].size()==0){
                break;
            }
            
        }
        
        
     
    }
    int d[] = new int[17];
     
    
public int minimumIncompatibility(int[] nums, int k) {
        if(nums.length%k!=0){
            return -1;
        }
        if(nums.length==k){
            return 0;
        }
        Arrays.sort(nums);
        
for(int a:nums){
            d[a]++;
            if(d[a]>k){
                return -1;
            }
        }
        
        int g = nums.length/k;
        
        ArrayDeque st[] = new ArrayDeque[k];
        for(int i=0;i<k;++i){
            st[i] = new ArrayDeque();
        }
        
        
        dfs(0,st,g,nums,k);
        
        
        return r;
        
        
        
        
    }
}
