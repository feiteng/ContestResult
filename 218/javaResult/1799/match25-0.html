class Solution {
    int result;
    int limit;
    public int minimumIncompatibility(int[] nums, int k) {
        Arrays.sort(nums);
        int cnt = 1;
        int n = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == n) {
                cnt++;
            } else {
                cnt = 1;
                n = nums[i];
            }
            if (cnt > k) return -1;
        }
        
        limit = nums.length / k;
        int[][] arr = new int[k][limit];
        int[] ind = new int[k];
        
        result = Integer.MAX_VALUE; 
        backtrack(arr, ind, 0, nums);
        
        return result;
    }
    
    void backtrack(int[][] arr, int[] ind, int i, int[] nums) {
        if (i == nums.length) {
            int res = 0;
            for (int k = 0; k < arr.length; k++) {
                res += arr[k][limit-1] - arr[k][0];
            }
            
            result = Math.min(res, result);
        } else {
            int val = nums[i];
            for (int j = 0; j < ind.length; j++) {
                if (ind[j] == limit) continue;
                if (ind[j] > 0 && arr[j][ind[j]-1] == val) continue;
                boolean toBreak = ind[j] == 0;
                arr[j][ind[j]] = val;
                ind[j]++;
                backtrack(arr, ind, i+1, nums);
                ind[j]--;
                if(toBreak) break;
            } 
        }
    }
}
