class Solution {
    static int [] a;
    static int sz, n;
    static long [] dp;
    public int minimumIncompatibility(int[] nums, int k) {
        a = nums;
        sz = nums.length / k;
        n = nums.length;
        dp = new long[(1 << nums.length)];
        Arrays.fill(dp, -1);
        long res = dfs((1 << nums.length) - 1);
        return (int) (res > 100 ? -1 : res);
    }
    
    static long dfs(int mask) {
        if (Integer.bitCount(mask) == 0) return 0;
        if (dp[mask] != -1) return dp[mask];
        long res = Integer.MAX_VALUE;
        for (int i = mask; i > 0; i = (i - 1) & mask) {
            if (Integer.bitCount(i) != sz) continue;
            TreeSet<Integer> nums = new TreeSet<>();
            for (int j = 0; j < n; j++) {
                if (((i >> j) & 1) == 1) {
                    nums.add(a[j]);
                }
            }
            if (nums.size() != sz) continue;
            res = Math.min(res, nums.last() - nums.first() + dfs(mask ^ i));
        }
        return dp[mask] = res;
    }
}
