class Solution {
    public int minimumIncompatibility(int[] nums, int k) {
        int groupSize = nums.length / k;
        ResultType result = new ResultType();
        List<List<Integer>> groups = new ArrayList<>();
        for (int group = 0; group < k; group++){
            groups.add(new ArrayList<Integer>());
        }
        
        dfs(nums, 0, groups, result, groupSize);
        
        return result.minSum;
    }
    
    private void dfs(int[] nums, int index, List<List<Integer>> groups, ResultType result, int groupSize){
        if (index == nums.length){
            result.minSum = result.minSum == -1 ? getSum(groups) : Math.min(result.minSum, getSum(groups));
            return;
        }
        if (result.minSum != -1 && getSum(groups) > result.minSum){
            return;
        }
        
        boolean findEmptyGroup = false;
        for (int group = 0; group < groups.size(); group++){
            if (findEmptyGroup && groups.get(group).size() == 0){
                continue;
            }
            if (groups.get(group).size() == groupSize){
                continue;
            }
            if (hasSameNum(groups.get(group), nums[index])){
                continue;
            }
            if (groups.get(group).size() == 0){
                findEmptyGroup = true;
            }
            groups.get(group).add(nums[index]);
            dfs(nums, index + 1, groups, result, groupSize);
            groups.get(group).remove(groups.get(group).size() - 1);
        }
    }
    
    private int getSum(List<List<Integer>> groups){
        int sum = 0;
        
        for (List<Integer> group : groups){
            if (group.size() == 0){
                continue;
            }
            int min = group.get(0), max = group.get(0);
            for (int num : group){
                min = Math.min(min, num);
                max = Math.max(max, num);
            }
            sum += max - min;
        }
        
        return sum;
    }
    
    private boolean hasSameNum(List<Integer> group, int val){
        for (int num : group){
            if (num == val){
                return true;
            }
        }
        
        return false;
    }
}

class ResultType{
    int minSum;
    
    public ResultType(){
        minSum = -1;
    }
}
