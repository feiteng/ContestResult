class Solution {
    public int maxOperations(int[] nums, int k) {
        Arrays.sort(nums);
        int res = 0;
        int begin = 0;
        int end = nums.length -1;
        while (begin < end) {
            int temp = nums[begin] + nums[end];
            if (temp == k) {
                res++;
                begin++;
                end--;
            } else if (temp < k) {
                begin++;
            } else {
                end--;
            }
        }
        return res;
    }
}
