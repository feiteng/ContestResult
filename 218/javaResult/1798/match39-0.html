class Solution {
    public int maxOperations(int[] nums, int k) {
        Arrays.sort(nums);
        int p1 = 0;
        int p2 = nums.length - 1;
        int res = 0;
        while (p1 < p2) {
            if (nums[p1] + nums[p2] < k) {
                p1++;
            } else if (nums[p1] + nums[p2] > k) {
                p2--;
            } else {
                p1++;
                p2--;
                res++;
            }
        }
        return res;
    }
}
