class Solution {
    public int maxOperations(int[] nums, int k) {
        Arrays.sort(nums);
        int result = 0;
        int s = 0, e = nums.length - 1;
        while(s < e) {
            if(nums[s] + nums[e] == k) {
                result++;
                s++;
                e--;
            }
            else if(nums[s] + nums[e] < k) {
                s++;
            }
            else {
                e--;
            }
        }
        
        return result;
    }
}
