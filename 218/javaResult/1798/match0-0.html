class Solution {
    public int maxOperations(int[] nums, int k) {
        Arrays.sort(nums);
        int left = 0, right = nums.length - 1;
        int res = 0;
        while(left < right){
            int sum = nums[left] + nums[right];
            if(sum < k){
                left++;
            } else if(sum > k){
                right--;
            } else {
                left++;
                right--;
                res++;
            }
        }
        return res;
    }
}
