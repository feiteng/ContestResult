class Solution {
    public int maxOperations(int[] nums, int k) {
        Arrays.sort(nums);
        int le = 0, ri = nums.length - 1;
        int count = 0;
        while(le < ri) {
            int sum = nums[le] + nums[ri];
            if(sum > k) {
                ri--;
            }else if(sum < k) {
                le++;
            }else {
                count++;
                le++;
                ri--;
            }
        }
        return count;
    }
}
