class Solution {
public:
    int m, ans = 1e9, n;
    void dfs(vector<int>& nums, vector<vector<int> >& arr, int cnt) {
        for (int i = 1; i < m; i++) {
            if ((int)arr[i].size() > (int)arr[i-1].size()) return ;
        }
    //    cout << cnt << endl;
        if (cnt == n) {
            int tmp = 0;
            for (int i = 0; i < m; i++) {
                tmp += arr[i][n/m - 1] - arr[i][0];
            }
            ans = min(ans, tmp);
      //      cout << "?????" << endl;
            return ;
        }
      //  cout << "???" << cnt << endl;
        
        for (int i = 0; i < m; i++) {
            int k = (int)arr[i].size();
            if (k == n/m || (k && arr[i][k - 1] == nums[cnt])) continue;
            arr[i].push_back(nums[cnt]);
            dfs(nums, arr, cnt + 1);
            arr[i].pop_back();
        }
    }
    int minimumIncompatibility(vector<int>& nums, int k) {
        m = k;
        sort(nums.begin(), nums.end());
        int vis[17] = {0};
        for (auto &t: nums) {
            vis[t]++;if (vis[t]>k) return -1;
        }
        vector<vector<int> >arr(k);
        n = (int)nums.size();
        dfs(nums, arr, 0);
      //  cout << "??" << endl;
        return ans;
    }
};
