class Solution {
public:
    
    int res, cnt, tot;
    int delta;
    vector<vector<int>> num;
    

    void dfs(int step, vector<int>& nums, int& k, int val) {
        if (val >= res)
            return;
        if (step == tot) {
            res = min(res, val);
            return;
        }
        for (int i = 0; i < min(step + 1, k); ++ i) {
            if (num[i].size() < cnt) {
                delta = 0;
                if (num[i].size() >= 1) {
                    delta = nums[step] - num[i].back();
                    if (delta == 0)
                        continue;
                }
                num[i].push_back(nums[step]);
                dfs(step + 1, nums, k, val + delta);
                num[i].pop_back();
            }
        }
    }
    
    int minimumIncompatibility(vector<int>& nums, int k) {
        res = INT_MAX, cnt = nums.size() / k, tot = nums.size();
        num = vector<vector<int>>(k);
        sort(nums.begin(), nums.end());
        dfs(0, nums, k, 0);
        if (res == INT_MAX)
            return -1;
        return res;
    }
};
