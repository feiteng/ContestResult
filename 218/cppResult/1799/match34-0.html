class Solution {
public:
  int minimumIncompatibility(vector<int>& nums, int k) {
    int N = nums.size();
    int sz = N / k;
    vector<pair<int, int>> subset;
    for (int m = 0; m < (1<<N); m++) if (__builtin_popcount(m) == sz) {
      set<int> S;
      for (int i = 0; i < N; i++) if ((1<<i) & m) {
        S.insert(nums[i]);
      }
      if (S.size() != sz) continue;
      int tot = *S.rbegin() - *S.begin(); 
      subset.push_back({m, tot});
    }
    vector<int> dp(1<<N, 0x3f3f3f3f);
    dp[0]=0;
    for (int sm = 0; sm < 1<<N; sm++) if (dp[sm] != 0x3f3f3f3f) {
      for (auto [k, v]: subset) if ((sm & k) == 0) {
        int tm = sm|k;
        dp[tm] = min(dp[tm], dp[sm] + v);
      }
    } 
    if (dp.back() == 0x3f3f3f3f) return -1;
    return dp.back();
  }
};
