class Solution {
public:
    int ans = INT_MAX;
    int k;
    int ec;
    void dfs(vector<int> & cur, vector<int> &cnt,int tmp, int loc, vector<int> &nums){
        if(loc == nums.size()){
            ans = min(ans, tmp);
            return;
        }
        for(int i = 0; i < k; i++){
            if(nums[loc] == cur[i]) continue;
            if(cnt [i] != ec) {
                int rec = cur[i];
                cur[i] = nums[loc];
                cnt[i] ++;
                dfs(cur, cnt, tmp + (rec >= 0? nums[loc] - rec:0), loc + 1, nums);
                cnt[i] --;
                cur[i] = rec;
                if(cnt[i] == 0) break;
            }
        }
        return;
    }
    int minimumIncompatibility(vector<int>& nums, int k) {
        unordered_map<int, int> m;
        int n = nums.size();
        for(auto x:nums){
            m[x] ++;
            if(m[x] > k) return -1;
        }
        sort(nums.begin(), nums.end());
        vector<int> cur(k, -1);
        vector<int> cnt(k, 0);
        this->k = k;
        this->ec = n/k;
        dfs(cur, cnt, 0, 0, nums);
        return ans;
    }
};
