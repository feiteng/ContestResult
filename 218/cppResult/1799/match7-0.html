class Solution {
public:
    vector<int>A[20];
    bool vis[20][20];
    int ans=1e9,g,n,c;
    void dfs(int now,int k,vector<int>& nums,int cost){
        if(cost>=ans)return;
        if(now==n){
            ans = min(ans,cost);
            return;
        }
        for(int i=0;i<k;i++){
            if(A[i].size()<c&&!vis[i][nums[now]]){
                vis[i][nums[now]] = 1;
                A[i].push_back(nums[now]);
                if(A[i].size()==c)dfs(now+1,k,nums,cost+A[i].back()-A[i][0]);
                else dfs(now+1,k,nums,cost);
                vis[i][nums[now]] = 0;
                A[i].pop_back();
            }
        }
        if(k<g){
            A[k].push_back(nums[now]);
            vis[k][nums[now]] = 1;
            dfs(now+1,k+1,nums,cost);
            A[k].pop_back();
            vis[k][nums[now]] = 0;
        }
    }
    int minimumIncompatibility(vector<int>& nums, int k) {
        sort(nums.begin(),nums.end());
        n = nums.size(),c = n/k, g = k;
        dfs(0,0,nums,0);
        if(ans==1e9)return -1;
        return ans;
    }
};
