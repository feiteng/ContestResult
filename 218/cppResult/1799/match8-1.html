class Solution {
    vector<int> a;
    int v[21][21],n,m,ans,w;

    void dfs(int x,int p,int val){
        if (ans!=-1&&val>ans) return;
        if (x==n){
            ans=val;return;
        }
        for (int i=p;i<m;i++){
            if (v[i][0]&&v[i][v[i][0]]==a[x]||v[i][0]==w) continue;
            v[i][++v[i][0]]=a[x];
            int z=val,q;
            if (x<n-1&&a[x]==a[x+1]) q=i+1;else q=0;
            if (v[i][0]>1) z+=v[i][v[i][0]]-v[i][v[i][0]-1];
            dfs(x+1,q,z);
            v[i][0]--;
            if (!v[i][0]) break;
        }
    }
public:
    int minimumIncompatibility(vector<int>& nums, int k) {
        ans=-1;n=nums.size();a=nums;m=k;w=n/k;
        sort(a.begin(),a.end());
        dfs(0,0,0);
        return ans;
    }
};
