class Solution {
public:
            vector<vector<int> > dp;
    int res=300;
    int K;
    vector<int> NUMS;
    
    void solve(int index,int curr){

        if(curr>=res){return;}
           

        if(index>=NUMS.size()){
            res=min(res,curr); return;
        }
            
        for(int i=0;i<K;i++){
           
            if(dp[i].empty()){
                dp[i].push_back(NUMS[index]);
                solve(index+1,curr);
                dp[i].pop_back();
                break;
            }else{
             
            if(dp[i].size()==NUMS.size()/K){continue;}
        

            if(dp[i].back()==NUMS[index]){continue;}
             
            int delta=NUMS[index]-dp[i].back();
                dp[i].push_back(NUMS[index]);
                solve(index+1,curr+delta);
                dp[i].pop_back();
                 }
        }
    }
    int minimumIncompatibility(vector<int>& nums, int k) {

    K=k;
            int mp[17];
        memset(mp,0,sizeof mp);
        for(int x:nums){
        
            mp[x]++;
        }
      
        for(int i=1;i<=16;i++){
            if(mp[i]>k){
                return -1;
            }
        }
    
        
        sort(nums.begin(),nums.end());
                NUMS=nums;
        dp=vector<vector<int>>(k,vector<int>(0));

        int i;
        for(i=0;i<nums.size();i++){
            if(nums[i]==nums[0]){
                dp[i].push_back(nums[i]);
            }else{
                break;
            }
        }
        solve(i,0);
        return res;
    }
};
