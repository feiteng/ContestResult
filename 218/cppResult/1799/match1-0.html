const int maxn = 1e6+10;

int n,m,a[maxn],q,t,k;
char s[maxn];

class Solution {
public:
    int ans = INF,k,sz;
    vector<int>b[100];

    int minimumIncompatibility(vector<int>& nums, int k) {
        n = nums.size();sz=n/k;
        if(n==k) return 0;

        map<int,int>mp;
        for(auto to:nums){
            mp[to]++;
            if(mp[to]>k) return -1;
        }
        sort(nums.begin(),nums.end());
        vector<int>dp(1<<n,INF);
        dp[0] = 0;
        for(int i = 1;i<(1<<n);i++){
            for(int j = i;j;j=(j-1)&i){
                if(dp[i^j]!=INF&&__builtin_popcount(j)==sz){
                    int mx = 0,mi = n-1,same = 0,lst = -1;
                    for(int x = n-1;x>=0;x--){
                        if(j>>x&1) {
                            mx = max(mx,x);
                            mi = x;
                            if(nums[x]==lst) same=1;

                            lst=nums[x];

                        }
                    }
//                    cout<<i<<" "<<j<<" "<<same<<" "<<mx<<" "<<mi<<" "<<dp[i]<<endl;

                    if(same) continue;
                    dp[i] = min(dp[i],dp[i^j]+nums[mx]-nums[mi]);
                }
            }
        }
        return dp[(1<<n)-1]==INF?-1:dp[(1<<n)-1];
    }
};
