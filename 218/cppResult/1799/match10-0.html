class Solution {
public:
    int minimumIncompatibility(vector<int>& nums, int k) {
        int n=nums.size();
        int dp[1<<16];memset(dp,0x3f,sizeof(dp));dp[0]=0;
        vector<pair<int,int> > v;
        for(int i=0;i<(1<<n);i++){
            int cnt=0;
            for(int j=0;j<n;j++) if(i&(1<<j)) cnt++;
            if(cnt!=n/k) continue;
            int aa=0;bool f=false;
            for(int j=0;j<n;j++) for(int k=j+1;k<n;k++){
                if(((1<<j)&i)&&((1<<k)&i)){
                    if(nums[j]==nums[k]) f=true;
                    aa=max(aa,abs(nums[j]-nums[k]));
                }
            }
            if(!f) v.push_back(make_pair(i,aa));
        }
        for(int i=0;i<(1<<n);i++){
            if(dp[i]>1e9) continue;
            for(int j=0;j<v.size();j++){
                if(i&v[j].first) continue;
                dp[i|v[j].first]=min(dp[i|v[j].first],dp[i]+v[j].second);
            }
        }
        if(dp[(1<<n)-1]>1e9) return -1;
        return dp[(1<<n)-1];
    }
};
