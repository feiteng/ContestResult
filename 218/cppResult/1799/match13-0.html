int cnt[1<<16],val[1<<16],dp[1<<16],f[1<<16],tot=0;
class Solution {
public:
    int minimumIncompatibility(vector<int>& nums, int k) {
    	memset(cnt,0,sizeof(cnt));
    	
  		for(int i=0;i<1<<16;i++){
  			for(int j=0;j<16;j++)if(i>>j&1)cnt[i]++;
  		}

  		int n=nums.size();
  		int x=n/k;
  		tot=0;
  		for(int i=1;i<1<<n;i++){
  			int mx=0,mn=1e9;
            bool vis[20]={0};
            bool ok=0;
  			for(int j=0;j<n;j++){
  				if(i>>j&1){
                    if(vis[nums[j]]==1)ok=1;
                    vis[nums[j]]=1;
  					mx=max(mx,nums[j]);
  					mn=min(mn,nums[j]);
  				}	
  			}
            if(ok)f[i]=1e9;
  			else f[i]=mx-mn;
  		}    

  		for(int i=1;i<1<<n;i++){
  			if(cnt[i]%x==0){
  				dp[i]=1e9;
  				for(int j=i;j;j=(j-1)&i) {
  					if(cnt[j]==x) dp[i]=min(dp[i],dp[i ^ j] + f[j]);
  				}
  			}
  		}  
  		return dp[(1<<n) - 1]>1e8?-1:dp[(1<<n)-1];
    }
};
