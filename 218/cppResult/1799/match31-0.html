class Solution {
public:
    int g[100][100];
    int n;
    int ans;
    int K;
    int M;
    vector<int> A;
    bool good(int x, int i) {
        if (g[i][0] >= M) return false;
        for (int j = 1; j <= g[i][0]; j++)
            if (x== g[i][j]) return false;
        return true;
    }
    int count(int i) {
        int l = 100, r = -1;
             //   if (g[i][0] == 0) continue;
                for (int j = 1; j <= g[i][0]; j++) {
                    l = min(l, g[i][j]);
                    r = max(r, g[i][j]);
                }
        return r - l;
    }
    void dfs(int d, int cur) {
        if (cur > ans) return;
        if (d == n) {
            ans = min(ans, cur);
            return;
        }
        for (int i = 0; i < K; i++) {
            if (good(A[d], i)) {
                g[i][0] ++;
                g[i][g[i][0]] = A[d];
                int del = 0;
                if (g[i][0] == M) {
                    del = count(i);
                }
                cur += del;
                dfs(d+1, cur);
                g[i][0] --;
                cur -= del;
            }
            if (g[i][0] == 0) break;
        }
    }
    
    int minimumIncompatibility(vector<int>& AA, int k) {
        sort(A.begin(),A.end());
        A = AA;
        memset(g, 0, sizeof g);
        n = A.size();
        ans = 1000000000;
        K = k;
        M = n/K;
        dfs(0, 0);
        if (ans == 1000000000) ans = -1;
        return ans;
    }
};
