class Solution {
    
    int limit = 0;
    vector<vector<int>> record;
    //int currentValue = 0;
    int minValue = INT_MAX;
    int k;

    void dfs(vector<int>& nums, int index, int currentValue)
    {
       // cout<<currentValue;
       if(currentValue >= minValue)
       {
           return;
       }
        
        if(index >= nums.size())
        {
            minValue = currentValue;
            return;
        }
        
        
        for(int i = 0; i<k; i++)
        {
           if(record[i].empty())
           {
               record[i].push_back(nums[index]);
               dfs(nums, index+1, currentValue);
               record[i].pop_back();
               break;
           }
            else if(record[i].size() >= limit || record[i].back() == nums[index])
                continue;
            else
            {
                
               record[i].push_back(nums[index]);
               dfs(nums, index+1, currentValue + record[i].back() - *(record[i].end()-2) );
               record[i].pop_back();
            }
                
        }
        
    }
public:
    int minimumIncompatibility(vector<int>& nums, int k)
    {
        sort(nums.begin(), nums.end());
        this->k = k;
        limit = nums.size()/k;
        record.resize(k);
        dfs(nums, 0, 0);
        return minValue == INT_MAX ? -1 : minValue;
    }
};
