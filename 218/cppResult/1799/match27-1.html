class Solution {
public:
    vector<int> num;
    int ans;
    int cnt[18];
    int minn[18][18];
    
    int len;int kk;
    int mx;

    void dfs(int id){
        if(id==len){
            int tans=0;
            for(int i=0;i<kk;i++)
            {
                tans+=(minn[i][mx-1]-minn[i][0]);
                // for(int j=0;j<mx;j++)
                //     cout<<minn[i][j]<<" ";
                // cout<<endl;
            }
            //cout<<endl;
            ans=min(ans,tans);
            return ;
        }
        
        for(int i=0;i<kk;i++)
        {
            if(cnt[i]==0){
                cnt[i]++;
                minn[i][cnt[i]-1] = num[id];
//                 for(int l=0;l<kk;l++)
//             {
                
//                 for(int j=0;j<mx;j++)
//                     cout<<minn[l][j]<<" ";
//                 cout<<endl;
//             }
//             cout<<endl;
                dfs(id+1);
                minn[i][cnt[i]-1] = 0;
                cnt[i]--;
                
                break;
                //cnt[i]--;
            }else if(cnt[i]<mx && minn[i][cnt[i]-1]!=num[id]){
                cnt[i]++;
                minn[i][cnt[i]-1] = num[id];
//                 for(int l=0;l<kk;l++)
//             {
                
//                 for(int j=0;j<mx;j++)
//                     cout<<minn[l][j]<<" ";
//                 cout<<endl;
//             }
//             cout<<endl;
                dfs(id+1);
                minn[i][cnt[i]-1] = 0;
                cnt[i]--;
            }
        }
    }
    
    int minimumIncompatibility(vector<int>& nums, int k) {
        
    
        kk=k;
        len = nums.size();
        mx = len/kk;
        
        if(k==len)
            return 0;
        
        sort(nums.begin(), nums.end());
        num = nums;
        ans = 1e9+7;
        
        dfs(0);
        if(ans == 1e9+7)
            ans=-1;
        return ans;
    }
};
