class Solution {
public:
    int concatenatedBinary(int n) {
        long long cur_mod = 1 << getWidth(n);
        long long res = n--;
        while (n > 0) {
            res += cur_mod * n;

            cur_mod = cur_mod << getWidth(n);
            res %= 1000000007;
            cur_mod %= 1000000007;
            --n;
        }
        return res % 1000000007;
    }
    
    int getWidth(int n) {
        int res = 0;
        while (n) {
            ++res;
            n /= 2;
        }
        return res;
    }
};
