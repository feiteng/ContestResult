class Solution {
public:
    int concatenatedBinary(int n) {
        constexpr int kPrime = 1e9 + 7;
        vector<long long> lengths(n + 1);
        lengths[1] = 2;
        for (int i = 2; i <= n; ++i) {
            lengths[i] = lengths[i - 1];
            if (!(i & (i - 1))) {
                lengths[i] = lengths[i] * 2 % kPrime;
            }
        }
        long long result = 0;
        for (int i = 1; i <= n; ++i) {
            result = (result * lengths[i] % kPrime + i) % kPrime;
        }
        return static_cast<int>(result);
    }
};
