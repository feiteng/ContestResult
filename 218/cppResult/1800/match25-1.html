class Solution {
public:
    int len(int n) {
        int x = 0;

        for (int i = 1; i<=31; i++) {
            if (n >= (1 << (i - 1)) && n < (1 << i)) return i;
        }
        return -1;
    }
    int concatenatedBinary(int n) {
        long long res = 0;
        for (int i = 1; i <= n; i++) {
            res = res * (1 << len(i)) + i;
            res %= 1000000007;
        }
        return res;
    }
};
