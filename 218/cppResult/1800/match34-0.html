class Solution {
public:
    #define LL long long
    int mod;
    int concatenatedBinary(int n) {
        mod=1E9+7;
        LL res=0;
        for(int i=1;i<=n;i++){
            int bits=getbits(i);
            res=res*(1<<bits) %mod +i;
            res%=mod;
        }
        return res;
    }
    
    int getbits(int n){
        int res=0;
        while(n>0){
            res++;
            n>>=1;
        }
        return res;
    }
};
