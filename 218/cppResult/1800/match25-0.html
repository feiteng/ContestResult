class Solution {
public:
    const int mod = 1000000007;
    int getLen(int x) {
        for (int i = 31; i >= 0; i--) {
            if ((1 << i) & x) return i+1;
        }
        return -1;
    }
    int concatenatedBinary(int n) {
        /*
        1 1             1
        2 110           6       6 << 2 + 3
        3 11011         27      27 << 3 + 4
        4 11011100      
        */
        long long ans = 0;
        for (int i = 1; i <= n; i++) {
            ans = ((ans << getLen(i)) % mod) + i;
            ans %= mod;
        }
        return ans;
    }
};
