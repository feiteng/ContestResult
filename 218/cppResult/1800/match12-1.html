class Solution {
private:

    int numBits(int val) {
        int bits = 0;
        
        while (val) {
            val >>= 1;
            bits++;
        }
        
        return bits;
    }
public:
    int concatenatedBinary(int n) {

        const long long MOD = 1e9 + 7;
        long long result = 0;
        
        for (int i = 1; i <= n; i++) {
            result = (result << numBits(i)) + i;
            result %= MOD;
        }
        
        return (int)result;
    }
};
