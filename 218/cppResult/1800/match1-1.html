class Solution {
public:
    int concatenatedBinary(int n) {
        // int kMOD = 1e9 + 7;
        // int res = 0;
        // for (int i = 1; i <= n; i++) {
        //     vector<int> bits;
        //     for (int j = i; j; j >>= 1) {
        //         bits.push_back(j&1);

        //     reverse(bits.begin(), bits.end());
        //     for (int b: bits) {
        //         if (b) {
        //             res = (res * 2 + 1) % kMOD;
        //         } else {
        //             res = (res * 2) % kMOD;
        //         }
        //     }
        // }
        // return res;
        int kMOD = 1e9 + 7;
        long res = 0;
        for (int i = 1; i <= n; i++) {
            int cnt = count(i);
            res = ((res << cnt) | i) % kMOD;
        }
        return static_cast<int>(res);
    }

private:

    int count(int num) {
        int cnt = 0;
        while (num) {
            cnt++;
            num /= 2;
        }
        return cnt;
    }
};
