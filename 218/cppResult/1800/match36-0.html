class Solution {
public:
    int concatenatedBinary(int n) {
        const int MOD = 1e9 + 7;
        long long res = 0;
        for (int i = 1; i <= n; i++) {
            int x = i;
            int cnt = 0;
            while (x) {
                cnt++;
                x >>= 1;
            }
            //cout << cnt << endl;
            res <<= cnt;
            res += i;
            res %= MOD;
        }
        return res;
    }
};
