class Solution {
public:
    int concatenatedBinary(int n) {
        int power = 0;
        long long res = 0;
        long long mod = 1000000007;
        for(int i=1; i<=n; i++){
            if(i == pow(2, power))
                power++;
            res<<=power;
            res += i;
            res %= mod;
        }
        return (int)res;
    }
};
