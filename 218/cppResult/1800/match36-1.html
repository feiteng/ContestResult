class Solution {
public:
    long long pow(long long last){
        long long x = 2;
        long long ans = 1;
        int mod = 1e9 + 7;
        while(last > 0){
            if(last & 1){
                ans *= x;
                ans %= mod;
            }
            x *= x;
            x %= mod;
            last /= 2;
        }
        // cout << ans << endl;
        return ans;
    }
    int concatenatedBinary(int n) {

        int mod = 1e9 + 7;
        long long ans = 1;
        for(int i = 2; i <= n; i++){
            int x = i;
            int len = 0;
            while(x > 0){
                len++;
                x /= 2;
            }
            ans *= pow(len);
            ans %= mod;
            ans += i;
        }
        return ans;
    }
};
