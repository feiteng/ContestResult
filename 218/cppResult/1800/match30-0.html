typedef long long ll;
const int mod = 1e9 + 7;
class Solution {
public:
    int concatenatedBinary(int n) {
        int num = 1, ans = 0;
        for(int i = n; i >= 1; i--){
            int tmp = i;
            while(tmp){
                int x = tmp%2;
                tmp /= 2;
                ans = (ans+(ll)x*num%mod)%mod;
                num = num*2ll%mod;
            }
        }
        return ans;
    }
};
