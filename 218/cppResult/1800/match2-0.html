class Solution {
public:
    int concatenatedBinary(int n) {
        int mod = 1e9 + 7;
        long res = 0;
        for (int i=1; i<=n; ++i) {
            int shift = help(i);
            res = (((res << shift) % mod) + i) % mod;
        }
        return res;
    }
    
    int help(int x) {
        int res = 0;
        while (x) {
            ++res;
            x >>= 1;
        }
        return res;
    }
};
