class Solution {
public:
    int concatenatedBinary(int n) {
        int mod = 1e9 + 7;
        if (n == 1) return 1;

        long long ans = 0;
        ans = 1;
        for (int i = 2; i <= n; ++i) {
            int ct = count(i);
            ans = (ans << ct) + i;
            ans = ans % mod;
        }
        return ans;
    }
    
    int count(int n) {
        int cnt = 0;
        while (n > 0) {
            n >>= 1;
            cnt++;
        }
        return cnt;
    }
};
