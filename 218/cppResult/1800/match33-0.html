class Solution {
public:
    int concatenatedBinary(int n) {
        long long ans=0;
        int MOD=1e9 + 7;
        
        for(int i=1;i<=n;i++){
            int tmp=0,tt=i;
            while(tt){
                ++tmp;
                tt>>=1;
            }
            ans=(ans<<(tmp))+i;
            ans%=MOD;
        }
        return ans;
    }
};
