class Solution {
public:
    int concatenatedBinary(int n) {
        long MOD = 1e9 + 7;
        long res = 1;
        for (int i = 2; i <= n; i++) {
            int temp = i;
            int cnt = 0;
            while (temp) {
                temp = temp >> 1;
                cnt++;
            }
            res = (res << cnt) % MOD;
            res = (res + i) % MOD;
        }
        return res;
    }
};
