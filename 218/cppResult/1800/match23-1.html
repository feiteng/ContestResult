class Solution {
public:
    int concatenatedBinary(int n) {
        long long int res = 1;
        int MOD = 1000000007;
        for(int i = 2; i <= n; i++){
            int d = 1 + log2(i);
            res = (res << d) % MOD;
            res = (res + i) % MOD;
        }
        int res2 = res;
        return res2;
    }
};
