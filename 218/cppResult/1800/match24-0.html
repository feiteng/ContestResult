class Solution {
public:
    int concatenatedBinary(int n) {
        int mod = 1e9+7;
        int num = 0;
        
        for(int i=1; i<=n; ++i) {
            int j = i;
            while(j > 0) {
                num = num << 1;
                num %= mod;
                j = j >> 1;
            }
            num += i;
        }
        return num;
    }
};
