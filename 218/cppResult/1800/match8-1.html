class Solution {
public:

    int len(int x){
        int res=0;
        while(x>0){
            x>>=1;
            res++;
        }
    //    cout<<x<<res<<endl;
        return res;
    }
    
    int concatenatedBinary(int n) {

            unsigned long long x=1;
        for(int i=2;i<=n;i++){
            int j=len(i);
            x*=(1<<j);
            x+=i;
            x%=1000000007;
        }
        return x;
    }
};
