const int MOD=1e9+7;
class Solution {
public:
    int concatenatedBinary(int n) {
        long long ans=0;
        for ( int i=1; i<=n; i++ ) {
            int m=32-__builtin_clz(i);
            // cout << m << '\n';
            for ( int j=0; j<m; j++ ) {
                ans=ans*2%MOD;
            }
            ans+=i;
            ans%=MOD;
        }
        return ans;
    }
};
