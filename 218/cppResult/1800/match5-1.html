typedef long long ll;
const ll mod = 1e9 + 7;

class Solution {
public:

    
    int concatenatedBinary(int n) {
        ll res = 0;
        for(ll i = 1; i <= n; i ++) {
            res = (res << (int)log2(i) + 1) + i;
            res %= mod;
        }
        
        return res;
    }
};
