#define mo (1000000007)

class Solution {
public:
    int concatenatedBinary(int n) {

        long long ans = 0;
        for (int i = 1; i <= n; i++) {
            int z = i;
            int w = 0;
            while (z > 0) {++w; z>>=1;}
            ans = (ans*(1<<w))%mo;
            ans = (ans+i)%mo;
        }
        return ans;
    }
};
