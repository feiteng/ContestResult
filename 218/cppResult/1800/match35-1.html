class Solution {
public:
    int concatenatedBinary(int n) {

        long ans = 0;
        for(int i = 1; i < n + 1; i++)
        {
            int len = 0;
            int num = i;
            while(num)
            {
                len++;
                num >>= 1;
            }
            ans <<= len;
            ans |= i;
            ans %= 1000000007;
            // cout << i << ' ' << ans << ' ' << len << endl;
        }
        return ans;
    }
};
