class Solution {
public:
    int cal(int x) {
        int ret = 0;
        while(x) {
            ret++;
            x/=2;
        }
        return ret;
    }
    int concatenatedBinary(int n) {
        long long MOD = 1e9+7;
        long long ans = 1;
        for (int i=2; i<=n; ++i) {
            ans = (ans<<cal(i)) + i;
            ans %= MOD;
        }
        return ans;
    }
};
