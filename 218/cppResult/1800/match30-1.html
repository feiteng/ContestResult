typedef long long LL;

const int mod = 1e9 + 7;// 998244353

const int MXN = 1e5 + 5;

class Solution {
public:
    int concatenatedBinary(int n) {
        int ans = 0, base = 1;
        for(int i = n; i >= 1; --i) {

            n = i;
            while(n > 0) {
                int x = n & 1;
                n >>= 1;
                ans = (ans + (LL)base * x) % mod;
                base = (LL)base * 2 % mod;
            }
        }
        return ans;
    }
};
