class Solution {
public:
    int get_bitlen(int a){
        int ret = 0;
        while(a != 0){
            ret++;
            a /= 2;
        }
        return ret;
    }
    
    int concatenatedBinary(int n) {
        long long ret = 1;
        long long mod = 1e9 + 7;
        for(int i = 2;i <= n;i++){
            int bitlen = get_bitlen(i);
            ret = (ret << bitlen)%mod + i%mod;
            ret %= mod;
        }
        
        return ret;
    }
};
