class Solution {
public:
    int concatenatedBinary(int n) {
        long long ans = 0;
        int big = 1000000007;
        for (int i = 1; i <=n; ++i)
        {
            int digit_to_move = 32 - __builtin_clz(i);
            ans = ((ans << digit_to_move) + i) % big;
        }
        
        return ans;
    }
};
