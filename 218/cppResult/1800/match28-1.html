const int mod = 1e9 + 7;


class Solution {
public:
  int count(int x) {
    int cnt = 0;
    while (x > 0) {
      ++cnt;
      x /= 2;
    }
    return cnt;
  }
  
  int concatenatedBinary(int n) {
    long long res = 0;
    for (int i = 1; i <= n; ++i) {
      res *= (1 << count(i));
      res += i;
      res %= mod;
    }
    return (int) res;
  }
};
