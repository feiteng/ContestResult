class Solution {
public:
    const int mod=1000000007;

    int bitCount(int x){
        int r = 0;
        while(x){
            r++;
            x>>=1;
        }
        return r;
    }
    int concatenatedBinary(int n) {
        int bsc = 1;

        int ans = 0;
        for(int i=n;i>=1;i--){
            int w = bitCount(i);
            ans = (ans+1ll*bsc*i)%mod;
            bsc = 1ll*bsc*(1ll<<w)%mod;
        }
        return ans;
    }
};
