const int mod=1e9+7;
const int maxn = 1e6+10;

int n,m,a[maxn],q,t,k;

class Solution {
public:
    int len[maxn];
    LL ksm(LL a, LL b) {
        LL res = 1;
        while(b){
            if(b&1){
                res = res*a%mod;
            }
            a=a*a%mod;
            b>>=1;
        }
        return res;
    }
    int concatenatedBinary(int n) {
        len[1] = 1;
        int nw = 2;
        while(nw<=2*n){
            len[nw]=len[nw/2]+1;
            nw*=2;
        }
        fo(i,1,n){
            if(!len[i]) len[i]=len[i-1];
        }
        LL ans = 0,length = 0;
        fod(i,n,1){
            ans=(ans+i*ksm(2,length)%mod)%mod;
            length+=len[i];
        }
        return ans;

    }
};
