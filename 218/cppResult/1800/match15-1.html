const int mod = 1e9 + 7;

const int maxn = 1e5 + 10;
typedef long long ll;
class Solution {
public:
    
    ll pow2[maxn];
    
    int concatenatedBinary(int n) {
        ll t = 1;
        for(int i=0;i<maxn;i++){
            pow2[i] = t;
            t = t * 2 % mod;
        }

        int len = 0, loc = 0;
        ll ans = 0;
        for(int i=1;i<=n;i++){
            if(i>=pow2[loc]){
                len++;
                loc++;
            }
            //cout<<i<<" "<<len<<endl;
            ans = ((ans * pow2[len] % mod )+ i) % mod;
        }
        return ans;
    }
    
    
};
