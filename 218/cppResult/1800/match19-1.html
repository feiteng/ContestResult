class Solution {
public:
    int concatenatedBinary(int n) {
        int cur_len = 0;
        long long ans = 0;
        long long hell = 1000000007;
        for(int i=1; i<=n; i++) {
            if((i&(i-1)) == 0)
                cur_len++;
            ans = (i | (ans << cur_len)) % hell;
        }
        return ans;
    }
};
