class Solution {
public:
    int concatenatedBinary(int n) {
        
        long long int ans = 1;
        int MOD = 1e9 + 7;
        for( int i = 2; i <= n; i++ )
        {
            int length = floor(log2(i)) + 1;
            ans *= pow(2, length);
            ans %= MOD;
            ans += i;
            ans %= MOD;
        }
        
        return ans;
    }
};
