class Solution {
public:
    #define ll long long
    #define mod 1000000007
    int concatenatedBinary(int n) {
        ll res = 1;
        for (ll i = 2; i <= n; i++) {
            res = res << ((int)log2(i)+1);
            res = (res+i)%mod;
        }
        return res%mod;
    }
};
