class Solution {
public:

    int concatenatedBinary(int n) {
        long long res = 0;
        long long C = 1e9 + 7;
        
        for(int i = 1; i <= n; ++i) {
            int len = get_len(i);
            res = res<<len;
            res += i;
            res = res % C;
        }
        
        return res;
    }
    

    int get_len(int n) {
        int res = 0;
        while(n > 0) {
            n = n / 2;
            res++;
        }
        
        return res;
    }
};
