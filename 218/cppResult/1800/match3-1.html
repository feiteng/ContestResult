class Solution {

public:
#define MOD 1000000007
	int concatenatedBinary(int n) {
		long long curr = 1;
		long long ret = 0;
		for (int i = n; i >= 1; i--) {
			int x = i;
			while (x > 0) {
				if (x % 2 == 1) {
					ret = (ret + curr) % MOD;
				}
				curr = (curr * 2) % MOD;
				x /= 2;
			}
		}
		return ret;
	}
};
