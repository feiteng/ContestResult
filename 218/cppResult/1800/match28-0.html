class Solution {
public:
    int bits(int i) {
        int res = 0;
        while(i) {
            res++;
            i>>=1;
        }
        return res;
    }
    int concatenatedBinary(int n) {
        long long res = 0;
        for(int i=1; i<=n; i++) {
            int b = bits(i);
            res = ((res<<b) % 1000000007 + i) % 1000000007;
        }
        return res;
    }
};
