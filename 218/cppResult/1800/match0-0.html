class Solution {
public:
    int concatenatedBinary(int n) {
        int mod = 1e9 + 7;
        long long ans = 0;
        for(int i = 1; i <= n; i++){
            int digit = (int)log2(1.0 * i) + 1;
            ans = ((ans << digit) + i) % mod;
        }
        return ans;
    }
};
