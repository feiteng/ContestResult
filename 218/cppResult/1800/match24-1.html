class Solution {
public:
    int concatenatedBinary(int n) {

        const int MOD = 1e9+7;
        int ans = 0;
        for(int i=1;i<=n;++i){
            int aug = i;
            while(aug > 0){
                ans *= 2;
                ans %= MOD;
                aug /= 2;
            }
            ans = (ans + i) % MOD;
        }
        return ans;
    }
};
