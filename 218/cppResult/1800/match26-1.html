#define fo(i, a, b) for (int i = a; i <= b; i++)
#define fof(i, a, b) for (int i = a; i >= b; i--)
const int INF = 0x3f3f3f3f;
typedef long long ll;


const int mod = 1e9 + 7;
const int maxn = 1e6 + 10;

int n, m, a[maxn], q, t, k;
char s[maxn];
class Solution
{
public:
    int len[maxn];
    ll qmod(ll a, ll b)
    {
        ll res = 1;
        while (b)
        {
            if (b & 1)
            {
                res = res * a % mod;
            }
            a = a * a % mod;
            b >>= 1;
        }
        return res;
    }
    int concatenatedBinary(int n)
    {
        len[1] = 1;
        int nlen = 2;
        while (nlen <= 2 * n)
        {
            len[nlen] = len[nlen / 2] + 1;
            nlen *= 2;
        }
        fo(i, 1, n)
        {
            if (!len[i])
                len[i] = len[i - 1];
        }
        ll ans = 0, length = 0;
        fof(i, n, 1)
        {
            ans = (ans + i * qmod(2, length) % mod) % mod;
            length += len[i];
        }
        return ans;
    }
};
