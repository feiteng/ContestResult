class Solution {
public:
    int concatenatedBinary(int n) {
        constexpr int kBase = 1e9 + 7;
        int len = 1;
        int threshold = 2;
        long res = 0;
        for (int i = 1; i <= n; i++) {
            if (i >= threshold) {
                len++;
                threshold <<= 1;
            }
            res = ((res << len) + i) % kBase;
        }
        return res;
    }
};
