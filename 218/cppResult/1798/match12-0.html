class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        unordered_map<int, int> ump;
        int ans = 0;
        for (int num : nums) {
            if (ump.count(k - num) > 0 && ump[k - num] > 0) {
                --ump[k - num];
                ++ans;
            } else {
                ++ump[num];
            }
        }
        return ans;
    }
};
