class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        multiset<int> st;
        int ret = 0;
        for (int x : nums) {
            if (st.find(k - x) != st.end()) {
                ret++;
                st.erase(st.find(k - x));
            } else {
                st.insert(x);
            }
        }
        return ret;
    }
};
