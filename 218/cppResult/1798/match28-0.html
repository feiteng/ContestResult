class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        multiset<int> m;
        int ret = 0;
        for(int i : nums){
            if(m.find(k-i) != m.end()){
                m.erase(m.lower_bound(k-i));
                ret++;
            }
            else{
                m.insert(i);
            }
        }
        return ret;
    }
};
