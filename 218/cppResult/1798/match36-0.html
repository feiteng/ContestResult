class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        map<int,int> mp; int res=0;
        for(int x:nums){
            if(mp[k-x]>0){
                mp[k-x]--;
                res++;
            }else{
                mp[x]++;
            }
        }
        return res;
    }
};
