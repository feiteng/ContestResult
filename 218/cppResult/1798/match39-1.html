class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        unordered_map<int, int> cnt;
        int res=0;
        for (int i: nums) {

            if (cnt[k-i]>0) {
                res++;
                cnt[k-i]--;
            } else {
                cnt[i]++;
            }
        }
        return res;
    }
};
