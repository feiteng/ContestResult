class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        sort(nums.begin(),nums.end());
        int start=0,end=nums.size()-1;
        int res=0;
        while(start<end){
            if(nums[end]+nums[start]<k) start++;
            else if(nums[end]+nums[start]>k) end--;
            else{
                res++;
                start++;
                end--;
            }
        }
        return res;
    }
};
