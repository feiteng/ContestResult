class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        int ans = 0;
        unordered_map<int, int> mp;
        
        for(int i = 0; i < nums.size(); ++i){

            if(mp.find(k - nums[i]) != mp.end()){
                ans++;
                if(--mp[k - nums[i]] == 0){
                    mp.erase(k - nums[i]);
                }
            }
            else{
                mp[nums[i]]++;
            }
        }
        
        return ans;
    }
};
