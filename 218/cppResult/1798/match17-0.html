class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        sort(begin(nums), end(nums));
        int p1 = 0, p2 = nums.size() - 1;
        int ans = 0;
        while (p1 < p2) {
            if (nums[p1] + nums[p2] == k) {
                ++ans;
                ++p1;
                --p2;
            } else if (nums[p1] + nums[p2] < k) {
                ++p1;
            } else {
                --p2;
            }
        }
        return ans;
    }
};
