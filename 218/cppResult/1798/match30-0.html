class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        sort(nums.begin(),nums.end());
        int p=0,q=nums.size()-1;
        int ans=0;
        while(p<q){
            int t=nums[p]+nums[q];
           // cout<<t<<endl;
            if(t<k) p++;
            else if(t>k) q--;
            else{
                ans++,p++,q--;
            }
        }
        return ans;
    }
};
