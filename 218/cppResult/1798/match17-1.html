class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        int pre = 0, end = nums.size()-1;
        sort(nums.begin(), nums.end());
        int result = 0;
        while(pre < end) {
            if(nums[pre] + nums[end] == k){
                result += 1;
                pre ++, end --;
            }else if(nums[pre] + nums[end] > k){
                end --;
            }else{
                pre ++;
            }
        }
        return result;
    }
};
