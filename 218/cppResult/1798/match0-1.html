class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        unordered_map<int, int> counts;
        int res = 0;
        for (auto n: nums) {
            if (k - n >= 1 && counts[k - n] > 0) {
                counts[k - n]--;
                res++;
            } else {
                counts[n]++;
            }
        }
        return res;
    }
};
