class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        unordered_map<int,int> store;
        int res = 0;
        for(auto &a : nums){
            if(store[k - a] > 0){
                res++;
                store[k - a]--;
            }
            else{
                store[a]++;
            }
        }
        return res;
    }
};
