class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        unordered_map<int,int> m;
        int r=0;
        for (int a: nums) {
            if (m.count(a) && m[a]>0) {
                --m[a];
                ++r;
            } else {
                ++m[k-a];
            }
        }
        return r;
    }
};
