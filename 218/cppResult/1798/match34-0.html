class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        int ans = 0;
        unordered_map<int, int> mp;
        for (int num : nums) {
            if (mp.find(k - num) != mp.end()) {
                ans++;
                if (!--mp[k - num]) {
                    mp.erase(k - num);
                }
            } else {
                mp[num]++;
            }
        }
        return ans;
    }
};
