class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        map<int, int> cnt;
        int ans = 0;
        for(int num:nums)
        {
            if(num >= k)
                continue;
            int target = k - num;
            if(cnt[target] > 0)

            {
                ans++;
                cnt[target]--;
            }
            else
            {
                cnt[num]++;
            }
        }
        return ans;
    }
};
