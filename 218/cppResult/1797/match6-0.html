class Solution {
public:
    string interpret(string cmd) {
        string res;
        for (int i = 0; i < cmd.size(); ++i) {
            if (cmd[i] == 'G') {
                res += 'G';
            } else if (i + 1 < cmd.size() && cmd[i] == '(' && cmd[i + 1] == ')') {
                res += 'o';
                i++;
            } else {
                res += "al";
                i += 3;
            }
        }
        return res;
    }
};
