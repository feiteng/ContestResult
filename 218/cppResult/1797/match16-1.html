class Solution {
public:
    string interpret(string command) {
        string res;
        int state = 0;
        for (int i = 0; i < command.length(); i++) {
            if (command[i] == '(') {
                if (command[i + 1] == ')') {
                    res += 'o';
                    i += 1;
                } else {
                    res += "al";
                    i += 3;
                }
            } else {
                res += command[i];
            }
        }
        return res;
    }
};
