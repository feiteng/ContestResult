class Solution {
public:
    string interpret(string command) {
        string res = "";
        for (int i = 0; i < command.size(); i++) {
            if (command[i] == 'G') {
                res += "G";
            } else if (command[i] == '(') {
                if (command[i + 1] == 'a') {
                    res += "al";
                    i += 3;
                } else {
                    res += "o";
                    i += 1;
                }
            }
        }
        return res;
    }
};
