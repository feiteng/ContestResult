class Solution {
public:
    string interpret(string &command) {
        int l=command.size();
        string ans;
        for(int i=0;i<l;++i){
            if(command[i]=='('){
                if(command[i+1]==')'){
                    ans+='o';
                    ++i;
                } else {
                    ans+="al";
                    i+=3;
                }
            } else {
                ans+='G';
            }
        }
        return ans;
    }
};
