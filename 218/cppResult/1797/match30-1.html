class Solution {
public:
    string interpret(string command) {
        string ans = "";
        for(int i = 0;i<command.size();i++)
        {
            if(command[i] == 'G')
            {
                ans+="G";
            }
            else

            {
                if(command[i+1] == 'a')
                {
                    ans+="al";
                    i +=3;
                }
                else
                {
                    ans+="o";
                    i++;
                } 
            }
        }
        return ans;
    }
};
