class Solution {
public:
    string interpret(string command) {
        string ret;
        int pos = 0, n = command.length();
        while (pos < n)
        {
            if (pos + 4 <= n && command.substr(pos, 4) == "(al)")
            {
                ret += "al";
                pos += 4;
            }
            else if (pos + 2 <= n && command.substr(pos, 2) == "()")
            {
                ret += "o";
                pos += 2;
            }
            else
            {
                ret += "G";
                pos++;
            }
        }
        return ret;
    }
};
