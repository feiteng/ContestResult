class Solution {
public:
  string interpret(string c) {
    string rv;
    for (int i = 0; i < c.length(); i++) {
      if (c[i] == 'G') {
        rv += "G";
      } else if (c[i] == '(') {
        if (c[i + 1] == ')')  {
          rv += "o";
          i++;
        } else {
          rv += "al";
          i += 3;
        }
      } 
    }
    return rv;
  }
};
