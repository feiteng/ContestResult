class Solution {
public:
    string interpret(string command) {
        string ans = "";
        int i = 0;
        int n = command.length();
        while (i < n) {
            if (command[i] == 'G') {
                ans += "G";
                ++i;
            } else if (i + 1 < n && command[i] == '(' && command[i+1] == ')') {
                ans += "o";
                i +=2;
            } else {
                ans += "al";
                i += 4;
            }
        }
        return ans;
    }
};
