ll fastexpo(ll x,ll y,ll m=1e9+7)
{
	ll temp=1;
	while(y>0)
	{
		if(y&1)temp = ((temp%m)*(x%m))%m;
		x = ((x%m)*(x%m))%m;
		y>>=1;
	}return temp;
}
class Solution {
public:
    string interpret(string command) {
        int n = command.length();
        string ans = "";
        for (int i = 0; i < n; ++i)
        {
        	if(command[i] == 'G')ans += "G";
        	else
        	{
        		if(command[i+1] == ')')
        		{
        			ans += "o";
        			i++;
        		}
        		else
        		{
        			ans += "al";
        			i+=3;
        		}
        	}
        }
        return ans;
    }
};
