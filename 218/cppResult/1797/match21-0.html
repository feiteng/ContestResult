class Solution {
public:
    string interpret(string command) {
        string result;
        for (int i = 0; i < command.size(); ) {
            if (command[i] == 'G') {
                result.push_back('G');
                ++i;
            } else if (command[i + 1] == ')') {
                result.push_back('o');
                i += 2;
            } else {
                result.append("al");
                i += 4;
            }
        }
        return result;
    }
};
