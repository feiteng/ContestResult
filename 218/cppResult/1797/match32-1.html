class Solution {
public:
    string interpret(string command) {
        string ans;
        for (int i = 0; i < command.size(); ) {
            if (command[i] == '(' && i + 1 < command.size() && command[i+1] == ')') {
                ans += 'o';
                i += 2;
                continue;

            }
            else if (command[i] == '(' && i + 3 < command.size() && command[i + 1] == 'a' && command[i + 2] == 'l' && command[i + 3] == ')')
            {
                ans += "al";
                i += 4;
            }
            else
            {
                ans += command[i];
                i++;
            }
        }
        return ans;
    }
};
