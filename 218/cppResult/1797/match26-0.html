class Solution {
public:
    string interpret(string command) {
        int i=0;
        string ans = "";
        while(i<command.size()){
            if(command[i]=='G') {
                ans += 'G';
                i++;
            }
            else if(command[i]=='(' && command[i+1]==')'){
                i+=2;
                ans += 'o';
            } else {
                i +=4;
                ans += "al";
            }
        }
        
        return ans;
    }
};
