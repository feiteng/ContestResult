class Solution {
public:
    string interpret(string c) {
        string ans ;
        int l = c.size();
        for(int i = 0 ; i< l ; ++i){
            if(c[i] == 'G') {
                ans += c[i];
            }
            else if(c[i+1] == ')'){
                ans += 'o';
                ++i;
            }
            else {
                ans += "al";
                i+=3;
            }
        }
        return ans;
    }
};
