class Solution {
public:
    string interpret(string command) {
        string res;
        for (int i = 0; i < command.size();) {
            if (command.substr(i, 1) == "G") {
                res.push_back('G');
                i += 1;
            } else if (command.substr(i, 2) == "()") {
                res.push_back('o');
                i += 2;
            } else {
                res += "al";
                i += 4;
            }
        }
        
        return res;
    }
};
