class Solution {
public:
    string interpret(string c) {
        int p=0;
        string res;
        while(p<c.size()){
            if(c[p]=='G'){
                res.push_back('G');
                p++;
            }
            else if(p+1<c.size()&&c[p+1]==')'){
                res.push_back('o');
                p+=2;
            }else{
                res+="al";
                p+=4;
            }
        }
        return res;
    }
};
