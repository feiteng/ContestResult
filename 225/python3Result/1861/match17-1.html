class Solution:
    def minimumBoxes(self, n: int) -> int:
        q 
= 0
        level = 1
        floor = 0
        while q < n:
            for i in range(1, level + 1):
                q += i
                floor += 1
                if q >= n:
                    break
            level += 1
        return floor
