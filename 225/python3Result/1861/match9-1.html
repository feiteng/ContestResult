class Solution:
    def minimumBoxes(self, n: int) -> int:
        start 
= cur = 1
        while n>=cur:
            n -= cur
            start += 1
            cur += start
        # print(start, cur, n)
        start -= 1
        res = (start+1)*start//2
        # if n==0:
        #     return (start+1)*start//2
        # if n>=2*start:
        #     return (start+2)*(start+1)//2
        # res = (start+1)*start//2+1
        start = 1
        while n>0:
            res += 1
            n -= start
            start += 1
        return res
