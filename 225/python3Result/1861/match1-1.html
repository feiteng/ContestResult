class Solution:
    def minimumBoxes(self, n: int) -> int:
        i = 0
        j 
= 1
        cur = 0
        while cur + i + j <= n:
            i += j
            j += 1
            cur += i
        
        left = n - cur
        k = 1
        
        #print(left)
        while left > 0:
            left -= k
            i += 1
            k += 1
            
        return i
