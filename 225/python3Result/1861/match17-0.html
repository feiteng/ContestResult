class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n <= 3: return n
        if n == 4: return 3
        floor = 4
        i = 4
        lv = 3
        while i < n:
            for l in range(1, lv + 1):
                i += l
                floor += 1
                if i >= n:
                    break
            lv += 1
        return floor - 1
