class Solution:
    def minimumBoxes(self, n: int) -> int:
        nn=n
        k=int((n*6)**(1/3))
        while k*(k-1)*(k-2)<=6*n:
            #print(n,k)
            k
+=1
        #print(k)
        k-=1
        #k=int(k)
        n-=(k*(k-1)*(k-2)//6)
        #n-=((k-2)*(k-1)//2)
        m=1
        ans=(k-2)*(k-1)//2
        #print(k,ans,nn,n)
        while n>0:
            n-=m
            ans+=1
            m+=1
        return ans
