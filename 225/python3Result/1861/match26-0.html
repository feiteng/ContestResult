class Solution:
    def minimumBoxes(self, n: int) -> int:
        bottom = 1
        count = 1
        lay = 1
        while n > count:
            lay += 1
            bottom += lay
            count += bottom
        count -= bottom
        bottom -= lay
        lay -= 1
        n -= count
        lay = 1
        count = 1
        while n > count:
            lay += 1
            count += lay
        
            
        return bottom + lay
