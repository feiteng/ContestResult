class Solution:
    def minimumBoxes(self, n: int) -> int:
        fib = 0
        count = 0
        i = 0
        while count < n:
            i = i+1
            fib += i
            count += fib
        if count == n:
            return fib
        count -= fib
        fib -= i
        # print(count,fib)
        res = n-count
        i = 0
        while res > 0:
            i += 1
            res -=i
        # print(i,fib)
        return i+fib
