class Solution:
    def minimumBoxes(self, n: int) -> int:
        base = 1
        num = 0
        while True:
            nnum = base*(base+1)//2 + num
            if nnum == n:
                return base*(base+1)//2
            elif nnum > n:
                break
            base += 1
            num = nnum
        base -= 1
        floor = base*(base+1)//2 
        
        i = 0
        j = base+1

        while i < j:
            h = (i + j) // 2
            additional = h*(h+1)/2
            if num + additional >= n:
                j = h
            else:
                i = h + 1    
        return floor + j
