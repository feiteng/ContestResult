class Solution:
    def minimumBoxes(self, n: int) -> int:
        a=1
        sa=1
        while(sa<=n):
            a
=a+1
            sa=sa+a*(a+1)//2
        sa=sa-a*(a+1)//2
        a=a-1
        #print(a,sa)
        temps=a*(a+1)//2
        temp=0
        while(sa<n):
            temp=temp+1
            sa=sa+temp
        return temps+temp
