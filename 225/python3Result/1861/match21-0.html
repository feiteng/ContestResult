class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n<4:
            return n
        base=3
        total=4
        incre=3
        h=2
        while total+base+incre<=n:
            base+=incre
            incre+=1
            total+=base
            h+=1
        if total==n:
            return base
        n-=total
        base+=1
        incre-=1
        he=incre
        n-=1
        while n>0:
            base+=1
            n-=1
            he-=1
            count=1
            while n>0 and count<=incre-he:
                n-=1
                count+=1
        return base
