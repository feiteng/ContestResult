class Solution:
    def minimumBoxes(self, n: int) -> int:
        i = 1
        count = 2
        while n >= i:
            n -= i
            i += count
            count += 1
            
        res = i - count + 1
        i = 1
        while n > 0:
            n -= i
            i += 1
            
        return res + i - 1
