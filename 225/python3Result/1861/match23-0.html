class Solution:
    def minimumBoxes(self, n: int) -> int:
        under = 1
        total = 1
        last_under = 0
        last_total = 0
        i = 1
        while total < n:
            last_under, last_total = under, total
            i += 1
            under += i
            total = under + last_total
        cur_total = last_total
        cur_under = last_under
        add = 1
        # print(under, total, last_under, last_total)
        while cur_total < n:
            cur_under += 1
            cur_total += add
            add += 1
            # print(cur_under, cur_total, add)
        return cur_under
