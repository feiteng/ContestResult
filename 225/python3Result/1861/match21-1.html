class Solution:
    def minimumBoxes(self, n: int) -> int:
        k 
= 1
        f = k*(k+1)//2
        _f = 0
        t = f
        while t < n:
            k += 1
            _f = f
            f = k*(k+1)//2
            t += f
        if t == n:
            return f
        t -= f
        f = _f
        i = 1
        side = 1
        f += 1
        while t + side < n:
            i += 1
            side += i
            f += 1
        return f
