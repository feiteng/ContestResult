class Solution:
    def minimumBoxes(self, n: int) -> int:
        h=0
        out=0
        while out<n:
            h+=1
            out+=(h*(h+1))//2
        g=(h*(h+1))//2
        rem=0
        while out>=n:
            out-=h-rem
            rem+=1
        rem-=1
        return g-rem
