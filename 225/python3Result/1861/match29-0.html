class Solution:
    def minimumBoxes(self, n: int) -> int:
        values = []
        k = 1
        current = 0
        tempsum = 0
        lastcurrent = 0
        lasttempsum = 0
        while tempsum <= n:
            lastcurrent = current
            current += k
            k += 1
            lasttempsum = tempsum
            tempsum += current
        floor = lastcurrent
        left = n - lasttempsum
        plus = 0
        plussum = 0
        while plussum < left:
            plus += 1
            plussum += plus
        return floor+plus
