class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n <= 3:
            return n
        total = 4
        prev_base, cur_base = 1, 3        
        while total < n:
            new_cur_base = cur_base + (cur_base - prev_base) + 1
            total += new_cur_base
            prev_base, cur_base = cur_base, new_cur_base
        if total == n:
            return cur_base
        # Try to use smaller base see if it's better
        leftover = n - (total - cur_base)
        # Observation: the leftover boxes can be filled optimally this way:
        #   + At base, we use i boxes, then above layer we can fill i - 1 boxes, etc
        # Therefore, we can use leftover boxes to stack max: 1 + 2 + ... + i = (i+1) * i // 2
        i = int(math.sqrt(2 * leftover)) + 1
        while i * (i + 1) // 2 >= leftover:
            i -= 1
        return prev_base + i + 1
