class Solution:
    def minimumBoxes(self, n: int) -> int:
        a=1#层数
        b
=1#底层的最大值
        c=1#总数
        while c<n:
            a+=1
            b=b+a
            c=c+b
        # print(a,b,c)
        ans=b
        n=c-n
        # print(n)
        while n>=a:
            ans-=1
            n-=a
            a-=1
        
        return ans
