class Solution:
    def minimumBoxes(self, n: int) -> int:
        t, level, curr 
= 1, 1, 0
        while curr + level <= n:
            curr += level
            level += t + 1
            t += 1
        floor, remain = level - t, n - curr
        
        line = 1
        while curr + line <= n:
            curr += line
            line += 1
            floor += 1
        if curr < n:
            floor += 1
        return floor
