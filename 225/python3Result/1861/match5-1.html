class Solution:
    def minimumBoxes(self, n: int) -> int:
        x, i, s 
= 0, 1, 0
        while s + x + i < n:
            x += i
            i += 1
            s += x
        n -= s
        i = 1
        while n > 0:
            x += 1
            n -= i
            i += 1
        return x
