class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n < 4:
            return n
        area = 0
        diff = 1
        while area + diff <= n:
            area += diff
            n -= area
            diff += 1
            # print(n, area, diff)
        if n == 0:
            return area
        for k in range(1, n + 1):
            # print(n, k, k * (k + 1) // 2, area)
            if k * (k + 1) // 2 >= n:
                return area + k
