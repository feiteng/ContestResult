class Solution:
    def minimumBoxes(self, n: int) -> int:
        i=1
        c=1
        s=1
        while s<n:
            i+=1
            c+=i
            s+=c
        # print(s,c,(s-n)//i)
        # if s==n:
        #     return c
        # return c-(s-n)//i
        while s-i>=n:
            s-=i
            c-=1
            i-=1
        return c
