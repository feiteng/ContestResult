class Solution:
    def minimumBoxes(self, n: int) -> int:
        sum_arr = [0, 1]
        for i in range(2,10000):
            sum_arr.append(sum_arr[-1] + i)
        
        sum_of_sum = 0
        prev_sum_of_sum = 0
        target = 0
        target_i = 0
        for i in range(10000):
            prev_sum_of_sum = sum_of_sum
            sum_of_sum += sum_arr[i]
            if sum_of_sum > n:
                target = sum_arr[i-1]
                target_i = i-1
                break
                
        n = n - prev_sum_of_sum   # 6
        
        if n== 0:
            return target

        for i in range(10000):
            if sum_arr[i] >= n:
                target += i
                break
                
        return target
