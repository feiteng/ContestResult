class Solution:
    def minimumBoxes(self, n: int) -> int:
        i = 1
        ps = 0
        t = 0
        while t <= n:
            ps += i
            t += ps
            i += 1
        i -= 1
        t -= ps
        ps -= i
        ret = ps
        i = 0
        while t < n:
            i += 1
            t += i
            ret += 1
        return ret
