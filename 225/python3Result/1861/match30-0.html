class Solution:
    def minimumBoxes(self, n: int) -> int:
        
        length = 0
        stacked = 0
        floor = 0
        while stacked < n:
            length += 1
            floor += length
            stacked += floor
            
        if stacked == n:
            return floor
        
        stacked -= floor
        floor -= length
        length -= 1
        
        left_cube = n - stacked
        
        tmp = 0
        ind = 0
        while tmp < left_cube:
            ind += 1
            tmp += ind
        if tmp > left_cube:
            tmp -= ind
            ind -= 1
        left_cube -= tmp
        floor += ind
        # print(floor)
        
        if left_cube > 0:
            floor += 1
        return floor

        return floor
