class Solution:
    def minimumBoxes(self, n: int) -> int:
        t = b = i = 1
        while t < n:
            i += 1
            b += i
            t += b
        k = t-n
        s = i
        ans = b
        while k >= s:
            i -= 1
            s += i
            ans -= 1
        return ans
