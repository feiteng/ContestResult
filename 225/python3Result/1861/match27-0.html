class Solution:
    def minimumBoxes(self, n: int) -> int:
          # [1,2,3,3,
          # [4,5,5,6,6, 6] 5 + 1
          # [7,8,8,9,9,9,10, 10,10,10] 7 + 3
          # [11,12,12,13,13,13,14,14,14,14,15,15,15,15,15] 9 + 6
        if n <= 3:
            return n
        if n == 4:
            return 3
        s = 4
        i = 5
        j = 1
        k = 2
        while s < n:
            s += i+j
            i += 2
            j += k
            k += 1
            # print(s)
        # print(s)
        if s == n:
            return i+j-2-(k-1)
        s -= i+j-2-(k-1)
        # print(s, j, k)
        n -= s
        w = 1
        ad = 2
        while w < n:
            w += ad
            ad += 1
        # print(ad, i+j-2-(k-1)-k+1)
        return i+j-2-(k-1)-k + ad-1
