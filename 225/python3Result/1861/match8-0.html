class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n < 4: return n
        ss = [0]
        starts = [0]
        line = 1
        while starts[-1] < n:
            ss.append(ss[-1] + line)
            starts.append(starts[-1] + ss[-1])
            line += 1
        line -= 1
        if starts[-1] == n:
            return ss[line]
        cols = [starts[line - 1]]
        for j in range(1, line + 1):
            cols.append(cols[-1] + j)
            if cols[-1] >= n:
                return ss[line - 1] + j
        return ss[line - 1] + j
