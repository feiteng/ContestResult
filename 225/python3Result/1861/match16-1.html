class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n == 1:
            return 1
        
        i 
= 1
        while i * (i + 1) * (i + 2) // 6 < n:
            i += 1
        
        i -= 1
        n -= i * (i + 1) * (i + 2) // 6
        res = i * (i + 1) // 2
        
        j = 1
        while j * (j + 1) // 2 < n:
            j += 1
        
        res += j
        
        return res
