class Solution:
    def minimumBoxes(self, n: int) -> int:
        if n<=3:
            return n
        cur, d = 1, 2
        l = 1
        while cur<n:
            l += d
            cur += l
            d += 1
        if cur==n:
            return l
        cur -= l
        l -= d-1
        a = 1
        while cur<n:
            l += 1
            cur += a
            a += 1
        return l
