class Solution:
    def minimumBoxes(self, n: int) -> int:
        suma=0
        count=0
        s=0
        while n>suma:
            count+=1
            s+=count
            suma+=s
        n-=(suma-s)
        bottom=sumb=0
        while n>sumb:
            bottom+=1
            sumb+=bottom
        return bottom+sum(list(range(count)))
