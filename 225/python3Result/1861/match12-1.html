class Solution:
    def minimumBoxes(self, n: int) -> int:
        bt = 1
        ly = 1
        st = 0
        while st + bt <= n:
            st += bt
            ly += 1
            bt += ly
        bt -= ly
        rst = n - st
        #print(bt, ly, rst)
        bt += math.ceil((math.sqrt(1 + 8 * rst) - 1) / 2)
        return bt
