class Solution:
    def minimumBoxes(self, n: int) -> int:
        k = 1
        while k * (k + 1) // 2 <= n:
            n -= k * (k + 1) // 2
            k += 1

        res = k * (k - 1) // 2

        k = 0
        while k + 1 <= n:
            k += 1
            n -= k

        res += k
        if n != 0:
            res += 1
        return res
