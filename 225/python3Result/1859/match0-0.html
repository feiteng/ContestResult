class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        
        al, bl = [0] * 26, [0] * 26
        for c in a:
            al[ord(c) - ord('a')] += 1
        for c in b:
            bl[ord(c) - ord('a')] += 1
            
        res = len(a) - max(al) + len(b) - max(bl)
        
        cura, curb = 0, 0
        for i in range(25):
            cura += al[i]
            curb += bl[i]
            res = min(res, len(a) - cura + curb, len(b) - curb + cura)
        return res
