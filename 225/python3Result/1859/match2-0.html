class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        na, nb = len(a), len(b)
        counter_a = [0 for _ in range(26)]
        counter_b = [0 for _ in range(26)]
        
        for c in a:
            counter_a[ord(c) - ord('a')] += 1
            
        for c in b:
            counter_b[ord(c) - ord('a')] += 1
        
        res = na - max(counter_a) + nb - max(counter_b)
        
        print(counter_a, counter_b)
        
        for i in range(1, 26):
            res = min(res, sum(counter_a[i:]) + sum(counter_b[:i]))
            res = min(res, sum(counter_b[i:]) + sum(counter_a[:i]))
            
        return res
