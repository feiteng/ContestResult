import string
from collections import Counter
import math


def count_changes(a: str, b: str):
    best = math.inf
    for l in string.ascii_lowercase[1:]:
        this_best = 0
        for letter in a:
            if letter < l:
                this_best += 1
        for letter in b:
            if letter >= l:
                this_best += 1
        best = min(best, this_best)
    for l in string.ascii_lowercase[1:]:
        this_best = 0
        for letter in b:
            if letter < l:
                this_best += 1
        for letter in a:
            if letter >= l:
                this_best += 1
        best = min(best, this_best)
    return best


class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ac = Counter(a)
        bc = Counter(b)

        make_distinct = (len(a) - max(ac.values())) + (len(b) - max(bc.values()))
        """
        print(f'md: {make_distinct}')

        print(f'ab: {count_changes_to_lt(a, min(b))}')

        print(f'ba: {count_changes_to_lt(b, min(a))}')
        """

        return min(
            make_distinct,
            count_changes(a, b),
        )
