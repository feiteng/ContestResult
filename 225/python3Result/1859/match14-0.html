class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        if not a or not b:
            return 0
        
        v_a = [0] * 26
        v_b = [0] * 26
        for c in a:
            v_a[ord(c) - ord('a')] += 1
        for c in b:
            v_b[ord(c) - ord('a')] += 1
        
        result = math.inf
        # c1 or c2
        c_a = 0
        c_b = 0
        t_a = sum(v_a)
        t_b = sum(v_b)
        for i in range(25):
            c_b += v_b[i]
            c_a += v_a[i]
            result = min(result, c_b + t_a - c_a)
            result = min(result, c_a + t_b - c_b)
        
        # c3
        t_a_b = t_a + t_b
        for i in range(26):
            result = min(result, t_a_b - v_a[i] - v_b[i])
        return result
