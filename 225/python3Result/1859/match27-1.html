class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        a1 = [0] * 26
        b1 = [0] * 26
        for c in a:
            a1[ord(c) - ord('a')] += 1
        for c in b:
            b1[ord(c) - ord('a')] += 1
        res = min([len(a) + len(b) - a1[i] - b1[i] for i in range(26)])
        
        a2 = [0] * 26
        b2 = [0] * 26
        a3 
= [0] * 26
        b3 = [0] * 26
        for i in range(24, -1, -1):
            a2[i] = a1[i + 1] + a2[i + 1]
            b2[i] = b1[i + 1] + b2[i + 1]
        for i in range(1, 26):
            a3[i] = a1[i - 1] + a3[i - 1]
            b3[i] = b1[i - 1] + b3[i - 1]
        
        for i in range(26):
            for j in range(i + 1, 26):
                res = min(res, a2[i] + b3[j], b2[i] + a3[j])
        
        return res
