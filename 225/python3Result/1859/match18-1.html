import string
import math
class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        alpha = string.ascii_lowercase
        def F(a, b):
            r=math.inf
            for x in alpha
[1:]:
                u = sum(+(ch<x) for ch in a)
                v = sum(+(ch>=x) for ch in b)
                r = min(r, u+v)
            return r
        def G(a,b):
            r=math.inf
            for x in alpha:
                u 
= sum(+(ch!=x) for ch in a)
                v = sum(+(ch!=x) for ch in b)
                r = min(r, u+v)
            return r
        return min(F(a,b),F(b,a),G(a,b))
