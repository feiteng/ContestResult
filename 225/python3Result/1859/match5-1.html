class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        cn1, cn2 
= [0] * 26, [0] * 26
        for c in a:
            cn1[ord(c)-ord('a')] += 1
        for c in b:
            cn2[ord(c)-ord('a')] += 1
            
        ans = len(a) + len(b) - max(cn1[i] + cn2[i] for i in range(26))
        for i in range(1, 26):
            ans = min(ans, sum(cn1[:i]) + sum(cn2[i:]), sum(cn1[i:]) + sum(cn2[:i]))
        return ans
