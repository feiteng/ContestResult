class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ab = a + b
        c = Counter(ab)
        con3 = len(ab) - max(c.values())
        
        con1 = float('inf')
        con2 = float('inf')
        chars = string.ascii_lowercase[1:]
        for standard in chars:
            cnt1 = 0
            cnt2 = 0
            for i in a:
                if i >= standard:
                    cnt1 += 1
                if i < standard:
                    cnt2 += 1
            for i in b:
                if i < standard:
                    cnt1 += 1
                if i >= standard:
                    cnt2 += 1
            con1 = min(con1, cnt1)
            con2 = min(con2, cnt2)
        
        return min(con1, con2, con3)
