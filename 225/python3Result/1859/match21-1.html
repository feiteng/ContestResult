        ca = [0 for _ in range(26)]
        cb = [0 for _ in range(26)]
        
        
for i in a:
            ca[ord(i) - ord('a')] += 1
        for i in b:
            cb[ord(i) - ord('a')] += 1
            
        na, nb = len(a), len(b)
        ans = max
(na, nb)
        for i in range(1, 26):
            ans = min(ans, sum(ca[:i]) + sum(cb[i:]), sum(ca[i:]) + sum(cb[:i]))
        
        # print(ca.most_common(1))

        dit_a = len(a) - ca.most_common(1)[0][1]
        dit_b = len(b) - cb.most_common(1)[0][1]
        
        # return min(convert(ca, cb), convert(cb, ca), convert(ca, cb, False), convert(cb, ca, False), dit_a + dit_b)
