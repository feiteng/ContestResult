class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        #condition 1
        move1 = float('inf')
        
for upper in 'abcdefghijklmnopqrstuvwxy':
            #all letters in a <= upper and all letters in b > upper
            cur = 0
            for c in a:
                if c > upper:
                    cur += 1
            for c in b:
                if c <= upper:
                    cur += 1
            move1 = min(move1,cur)
        #condition 2
        move2 = float('inf')
        for upper in 'abcdefghijklmnopqrstuvwxy':
            #all letters in b <= upper and all letters in a > upper
            cur = 0
            for c in b:
                if c > upper:
                    cur += 1
            for c in a:
                if c <= upper:
                    cur += 1
            move2 = min(move2,cur)
        #condition 3
        move3 = float('inf')
        for letter in 'abcdefghijklmnopqrstuvwxyz':
            cur 
= 0
            for c in a:
                if c!= letter:
                    cur += 1
            for c in b:
                if c!= letter:
                    cur += 1
            move3 = min(move3,cur)
        # print(move1,move2,move3)
        return min(move1, move2,move3)
