class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        al = [0] * 26
        bl = [0] * 26
        for aa in a:
            al[ord(aa)-ord("a")] += 1
        for bb in b:
            bl[ord(bb)-ord("a")] += 1
        alen = len
(a)
        blen = len(b)
        res = float("inf")
        v = alen
        for i in range(25):
            v -= al[i]
            v += bl[i]
            res = min(res, v)
        v = blen
        for i in range(25):
            v -= bl[i]
            v += al[i]
            res = min(res, v)
        maxTotal = 0
        for i in range(26):
            maxTotal = max(maxTotal, al[i] + bl[i])
        res = min(res, alen+blen-maxTotal)
        
        return res
