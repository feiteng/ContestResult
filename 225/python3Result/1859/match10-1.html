class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        l="abcdefghijklmnopqrstuvwxyz"
        ans=9999999
        
for k in l:
            aa=0
            for t in a:
                if t!=k:
                    aa+=1
            for t in b:
                if t!=k:
                    aa+=1
            ans=min(ans,aa)
        for i in range(1,26):
            kk=l[i]
            kp=l[i-1]
            bb
=0
            for t in a:
                if t>kp:
                    bb+=1
            for t in b:
                if t<kk:
                    bb+=1
            ans=min(ans,bb)
        for i in range(1,26):
            kk=l[i]
            kp=l[i-1]
            bb
=0
            for t in b:
                if t>kp:
                    bb+=1
            for t in a:
                if t<kk:
                    bb+=1
            ans=min(ans,bb)
        return ans
