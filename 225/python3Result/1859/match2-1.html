class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        counter_a = [0] * 26
        counter_b = [0] * 26
        
for c in a:
            counter_a[ord(c) - ord('a')] += 1
        for c in b:
            counter_b[ord(c) - ord('a')] += 1
        
        # plan 3      
        ans = min(len(a) + len(b) - counter_a[i] - counter_b[i] for i in range
(26))
        
        # plan 1, 2
        for i in range(25):
            ans = min(ans, sum(counter_a[i+1:]) + sum(counter_b[:i+1]))
            ans = min(ans, sum(counter_b[i+1:]) + sum(counter_a[:i+1]))
        
        return ans
