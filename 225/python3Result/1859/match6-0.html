class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        L1=[0]*26
        L2=[0]*26
        for x in a:
            L1[ord(x)-ord('a')]+=1
        for x in b:
            L2[ord(x)-ord('a')]+=1
        M=1e8
        for i in range(1,26):
            M=min(M, sum(L1[:i])+sum(L2[i:]))
            M=min(M, sum(L2[:i])+sum(L1[i:]))
        M=min(M, len(a)-max(L1)+len(b)-max(L2))
        return M
