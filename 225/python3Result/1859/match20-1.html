class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        A, B, down = [0] * 26, [0] * 26, ord
('a')
        for x in a:
            A[ord(x) - down] += 1
        for x in b:
            B[ord(x) - down] += 1
        
        res = float
('inf')
        for i in range(1, 26):
            res = min(res, sum(A[i:]) + sum(B[:i]), sum(B[i:]) + sum(A[:i]))
        
        dis = len(a) + len(b)
        for i in range(26):
            res = min(res, dis - A[i] - B[i])
        return res
