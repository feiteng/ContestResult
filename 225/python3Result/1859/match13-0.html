class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        cnt1 = [0] * 26
        cnt2 = [0] * 26
        for x in a:
            cnt1[ord(x)-ord("a")] += 1
        for x in b:
            cnt2[ord(x)-ord("a")] += 1
        cnt = 0
        for i in range(26):
            cnt = max(cnt, cnt1[i]+cnt2[i])
        result = len(a)+len(b)-cnt
        sum1 = sum(cnt1)
        sum2 = sum(cnt2)
        tmp = 0
        for i in range(1, 26):
            sum1 -= cnt1[i-1]
            tmp += cnt2[i-1]
            result = min(result, sum1+tmp)
        tmp = 0
        for i in range(1,26):
            sum2 -= cnt2[i-1]
            tmp += cnt1[i-1]
            result = min(result, sum2+tmp)
        return result
