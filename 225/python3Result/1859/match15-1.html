class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        Ca, Cb = [0]*26, [0]*26
        for c in a:
            Ca[ord(c)-97] += 1
        for c in b:
            Cb[ord(c)-97] += 1
        res = len(a)-max(Ca)+len(b)-max(Cb)
        for i in range(1,26):
            Ca[i] += Ca[i-1]
            Cb[i] += Cb[i-1]
        for i in range(1,26):
            res = min(res, len(a)-Ca[i-1]+Cb[i-1], len(b)-Cb[i-1]+Ca[i-1])
        return res
