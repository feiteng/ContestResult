class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        from collections import Counter
        from math import inf
        A, B, Ca, Cb, m, n = sorted(a), sorted(b), [0] * 26, [0] * 26, len(a), len(b)
        for i in a:
            Ca[ord(i) - 97] += 1
        for i in b:
            Cb[ord(i) - 97] += 1
        ans, index = m - max(Ca) + n - max(Cb), 1
        for i in range(1, 26):
            ans = min(ans, sum(Ca[i:]) + sum(Cb[:i]), sum(Cb[i:]) + sum(Ca[:i]))
        return ans
