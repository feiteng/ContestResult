class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ans = len(a) + len(b)
        for i in range(1, 26):
            target = chr(i + ord('a'))
            a_smaller = a_larger = a_same = 0
            b_smaller = b_larger = b_same 
= 0
            for c in a:
                if c >= target:
                    a_smaller += 1
                if c != target:
                    a_same += 1
                if c < target:
                    a_larger += 1
            for c in b:
                if c >= target:
                    b_smaller += 1
                if c != target:
                    b_same += 1
                if c < target:
                    b_larger += 1
            ans = min(ans, a_smaller+b_larger, a_same+b_same, a_larger+b_smaller)
        return ans
