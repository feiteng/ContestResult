class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ans = len(a) + len(b)
        for i in range(26):
            letter = chr(ord('a') + i)
            tmp = 0
            for l in a:
                if l != letter:
                    tmp += 1
            for l in b:
                if l != letter:
                    tmp += 1
            ans = min(ans, tmp)
        
        for i in range(25):
            letter = chr(ord('a') + i)
            tmp 
= 0
            for l in a:
                if l > letter:
                    tmp += 1
            for l in b:
                if l <= letter:
                    tmp += 1
            ans = min(ans, tmp)
            tmp = 0
            for l in b:
                if l > letter:
                    tmp += 1
            for l in a:
                if l <= letter:
                    tmp += 1
            ans = min(ans, tmp)
        return ans
