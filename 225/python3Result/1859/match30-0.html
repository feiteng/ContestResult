class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ac = [0 for _ in range(26)]
        bc = [0 for _ in range(26)]
        for c in a:
            ac[ord(c) - ord('a')] += 1
        for c in b:
            bc[ord(c) - ord('a')] += 1
        res = len(a) + len(b) - max(ac) - max(bc)
        pac = [0 for _ in range(26)]
        sac = [0 for _ in range(26)]
        pbc = [0 for _ in range(26)]
        sbc = [0 for _ in range(26)]
        pac[0], pbc[0] = ac[0], bc[0]
        for i in range(1, 26):
            pac[i] = pac[i-1] + ac[i]
            pbc[i] = pbc[i-1] + bc[i]
        sac[-1], sbc[-1] = ac[-1], bc[-1]
        for i in range(24, -1, -1):
            sac[i] = sac[i+1] + ac[i]
            sbc[i] = sbc[i+1] + bc[i]
        for i in range(25):
            res = min(res, sac[i+1] + pbc[i], sbc[i+1] + pac[i])
        return res
