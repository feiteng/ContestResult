class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        
        def f(a, b):
            res = float('inf')
            for x in string.ascii_lowercase[:-1]:
                cur = sum(1 for aa in a if aa > x)
                cur += sum(1 for bb in b if bb <= x)
                res = min(res, cur)
            return res
        
        def c(a,b):
            res = float('inf')
            for x in string.ascii_lowercase:
                cur = sum(1 for aa in a if aa != x)
                cur += sum(1 for bb in b if bb != x)
                res = min(res, cur)
            return res
        
        
        # print(f(a,b), f(b,a), c(a,b))
        return min(f(a,b), f(b,a), c(a,b))
