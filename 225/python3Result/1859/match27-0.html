class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        cntA = [0] * 26
        cntB = [0] * 26
        
        for ch in a:
            cntA[ord(ch) - ord('a')] += 1
        for ch in b:
            cntB[ord(ch) - ord('a')] += 1
            
        ans = float('inf')
        
        prefixA = [0] * 27
        prefixB = [0] * 27
        for i in range(26):
            prefixA[i+1] = prefixA[i] + cntA[i]
            prefixB[i+1] = prefixB[i] + cntB[i]
            
        
        for i in range(25):
            op1 = prefixA[-1] - prefixA[i+1] + prefixB[i+1]
            ans = min(ans, op1)
        for i in range(25):
            op2 = prefixB[-1] - prefixB[i+1] + prefixA[i+1]
            ans = min(ans, op2)
            
            
            
            
        op3 = len(a) - max(cntA) + len(b) - max(cntB)
        return min(ans, op3)
