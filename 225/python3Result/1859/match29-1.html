class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        
        aCounter, bCounter = Counter(a), Counter(b)
        #route 1
        route1 = inf
        for i in range(1, 26):
            ch = chr(ord
('a') + i)
            #minB = ch
            bChange, aChange = 0, 0
            for k, v in bCounter.items():
                if ord(k) - ord('a') < i:
                    bChange += v
            for k, v in aCounter.items():
                if ord(k) - ord('a') >= i:
                    aChange += v
            route1 = min(route1, aChange + bChange)

        #route 2
        route2 = inf
        for i in range(1, 26):
            ch = chr(ord
('a') + i)
            #minA = ch
            bChange, aChange = 0, 0
            for k, v in aCounter.items():
                if ord(k) - ord('a') < i:
                    bChange += v
            for k, v in bCounter.items():
                if ord(k) - ord('a') >= i:
                    aChange += v
            route1 = min(route1, aChange + bChange)
        
        
        #route 3
        alen, blen = len(a), len(b)
        amax, bmax = 0, 0
        for k, v in aCounter.items():
            amax = max(amax, v)
        for k, v in bCounter.items():
            bmax = max(bmax, v)
        route3 = alen + blen - amax - bmax
        
        return min(route1, route2, route3)
