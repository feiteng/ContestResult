class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        csa = collections.Counter(a)
        k1 = len(a) - max(csa.values())
        csb = collections.Counter(b)
        k2 = len(b) - max(csb.values())

        ans = k1 + k2

        for t in "abcdefghijklmnopqrstuvwxy":
            k1 = 0
            for c in a:
                if c>t:
                    k1+=1
            for c in b:
                if c<=t:
                    k1+=1
            ans = min(ans,k1)
            k2 = 0
            for c in a:
                if c<=t:
                    k2+=1
            for c in b:
                if c>t:
                    k2+=1
            ans = min(ans,k2)
        return ans
