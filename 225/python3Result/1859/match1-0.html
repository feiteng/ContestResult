class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        countA=[0 for _ in range(26)]
        countB=[0 for _ in range(26)]
        for c in a:
            countA[ord(c)-ord('a')]+=1
        for c in b:
            countB[ord(c)-ord('a')]+=1
        ret=len(a)-max(countA)+len(b)-max(countB)
        sumA,sumB=0,0
        for i in range(25):
            sumA+=countA[i]
            sumB+=countB[i]
            ret=min(ret, sumA+len(b)-sumB, sumB+len(a)-sumA)
        return ret
