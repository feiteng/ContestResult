class Solution(object):
    def minCharacters(self, a, b):
        """
        :type a: str
        :type b: str
        :rtype: int
        """

        res = len(a)+len(b)

        for i in range(1, 25):
            c = chr(ord('a')+i)

            c1 = 0
            c2 = 0
            c3 = 0

            for aa in a:
                if aa > c:
                    c1 += 1
                if aa != c:
                    c3 += 1
                if aa <= c:
                    c2 += 1
            for aa in b:
                if aa <= c:
                    c1 += 1
                if aa != c:
                    c3 += 1
                if aa > c:
                    c2 += 1

            res = min(res, c1, c2, c3)

        c1 = 0
        c2 = 0
        c3 = 0
        for aa in a:
            if aa != 'a':
                c1 += 1
                c3 += 1
            else:
                c2 += 1
        for aa in b:
            if aa != 'a':
                c2 += 1
                c3 += 1
            else:
                c1 += 1
        res = min(res, c1, c2, c3)
        c1 = 0
        c2 = 0
        c3 = 0
        for aa in a:
            if aa != 'z':
                c1 += 1
                c3 += 1
            else:
                c2 += 1
        for aa in b:
            if aa != 'z':
                c2 += 1
                c3 += 1
            else:
                c1 += 1
        res = min(res, c1, c2, c3)
        return res
