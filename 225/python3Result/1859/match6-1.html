class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ca,cb
=[0]*26,[0]*26
        for c in a:
            ca[ord(c)-ord('a')]+=1
        for c in b:
            cb[ord(c)-ord('a')]+=1
        
        cond1=10**10
        for i in range(1,26):
            cond1=min(cond1,sum(ca[i:])+sum(cb[:i]))
        
        cond2=10**10
        for i in range(1,26):
            cond2
=min(cond2,sum(cb[i:])+sum(ca[:i]))
        
        cond3=sum(ca)-max(ca)+sum(cb)-max(cb)
        
        return min(cond1,cond2,cond3)
