class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        ans = float('inf')
        la,lb = len(a),len(b)
        freqa 
= [0]*26
        freqb = [0]*26
        for c in a:
            freqa[ord(c)-ord("a")] += 1
        for c in b:
            freqb[ord(c)-ord("a")] += 1
        # print(freqa)
        # print(freqb)
        
        cnta,cntb = 0,0
        
for i in range(25):
            cnta += freqa[i]
            cntb += freqb[i]
            ans = min(ans,la-cnta+cntb)
            ans = min(ans,lb-cntb+cnta)
        
        tmp = 0
        for i in range(26):
            tmp = max(tmp,freqa[i]+freqb[i])
        return min(ans,la+lb-tmp)
