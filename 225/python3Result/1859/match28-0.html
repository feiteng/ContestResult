class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        countera = [0] * 26
        counterb = [0] * 26
        for c in a:
            countera[ord(c) - ord('a')] += 1
            
        for c in b:
            counterb[ord(c) - ord('a')] += 1
            
        suma = len(a)
        sumb = len(b)
        changes = math.inf
        
        psum_countera = [0] * 26
        psum_counterb = [0] * 26
        
        for i in range(1, 26):
            # use i as the split of a and b
            psum_countera[i] = psum_countera[i-1] + countera[i-1]
            psum_counterb[i] = psum_counterb[i-1] + counterb[i-1]
            
            # < i goes to a
            changes = min(changes, suma - psum_countera[i] + psum_counterb[i])
            
            # < i goes to b
            changes = min(changes, sumb - psum_counterb[i] + psum_countera[i])
            
            # all change to i
            changes = min(changes, suma - countera[i] + sumb - counterb[i])
            
        
        return changes
