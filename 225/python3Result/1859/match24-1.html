class Solution:
    def minCharacters(self, a: str, b: str) -> int:
        # SOL 1
        cnt = [0] * 30
        for i in a:
            cnt[ord(i) - 97] += 1
        for j in b:
            cnt[ord(j) - 97] += 1
        ans1 = len(a) + len(b) - max(cnt)
        
        x = [ord(i) - 97 for i in a]
        y = [ord(i) - 97 for i in b]
        
        # SOL2
        ans2 = 1000000000
        ans3 = 1000000000
        for i in range(1, 26):   
            ans2_tmp 
= 0
            ans3_tmp = 0
            for j in x:
                if j < i:
                    ans2_tmp += 1
                if j >= i:
                    ans3_tmp += 1
            for k in y:
                if k >= i:
                    ans2_tmp += 1
                if k < i:
                    ans3_tmp += 1
            ans2 = min(ans2, ans2_tmp)
            ans3 = min(ans3, ans3_tmp)
        
        return min(ans1, ans2, ans3)
