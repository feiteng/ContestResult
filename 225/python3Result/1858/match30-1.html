class Solution:
    def maximumTime(self, x: str) -> str:
        y = list(x)
        if y[0] == '?':
            if y[1] == '?':
                y[0] = '2'
                y[1] = '3'
            elif int(y[1]) <= 3:
                y
[0] = '2'
            else:
                y[0] = '1'
        if y[1] == '?':
            if y[0] == '2':
                y[1] = '3'
            else:
                y[1] = '9'
        if y[3] == '?':
            y[3] = '5'
        if y[4] == '?':
            y[4] = '9'
        
        return ''.join(y)
