class Solution:
    def maximumTime(self, time: str) -> str:
        res  = list(time)
        if time[0] == '?':
            if (time[1] >= '0' and time[1] <= '3') or time[1] == '?':
                res[0] = '2'
            else:
                res[0] = '1'
        if time[1] == '?':
            if res[0] == '2':
                res[1] = '3'
            else:
                res[1] = '9'
        if time[3] == '?':
            res[3] = '5'
        if time[4] == '?':
            res[4] = '9'
        return ''.join(res)
