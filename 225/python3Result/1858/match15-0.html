class Solution:
    def maximumTime(self, time: str) -> str:
        hh, mm = time.split(":")
        if hh == "??":
            hh = "23"
        elif hh[0] == "?":
            hh = ("2" if hh[1] in "0123" else "1") + hh[1]
        elif hh[1] == "?":
            hh = hh[0] + ("3" if hh[0] == "2" else "9")
        # else:
            # hh = int(hh)
        
        if mm == "??":
            mm = "59"
        elif mm[0] == "?":
            mm = "5" + mm[1]
        elif mm[1] == "?":
            mm = mm[0] + "9"
        # else:
        #     mm = int(mm)
        return f"{hh}:{mm}"
