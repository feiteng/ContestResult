class Solution:
    def maximumTime(self, time: str) -> str:
        tmp = list(time)
        if tmp[0] == '?':
            if tmp[1] in ['?', '0', '1', '2', '3']:
                tmp[0] = '2'
            else:
                tmp[0] = '1'
        if tmp[1] == '?':
            if tmp[0] == '2':
                tmp[1] = '3'
            else:
                tmp[1] = '9'
        if tmp[3] == '?':
            tmp[3] = '5'
        if tmp[4] == '?':
            tmp[4] = '9'
        return ''.join(tmp)
