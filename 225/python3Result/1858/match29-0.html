class Solution:
    def maximumTime(self, time: str) -> str:
        ret = ""
        if time[0] == "?":
            if time[1] <= "3" or time[1] == "?":
                ret += "2"
            else:
                ret += "1"
        else:
            ret += time[0]
        if time[1] == "?":
            if ret[0] == '2':
                ret += "3"
            else:
                ret += '9'
        else:
            ret += time[1]
        ret += ':'
        if time[3] == "?":
            ret += "5"
        else:
            ret += time[3]
        if time[4] == "?":
            ret += "9"
        else:
            ret += time[4]
        return ret
