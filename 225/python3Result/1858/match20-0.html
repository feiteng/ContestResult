class Solution:
    def maximumTime(self, T: str) -> str:
        T = list(T)
        if T[0] == '?':
            if T[1] == '?':
                T[0] = '2'
                T[1] = '3'
            elif int(T[1]) >= 4:
                T[0] = '1'
            else:
                T[0] = '2'
        if T[1] == '?':
            if T[0] == '2':
                T[1] = '3'
            else:
                T[1] = '9'
        if T[3] == '?':
            T[3] = '5'
        if T[4] == '?':
            T[4] = '9'
        return ''.join(T)
