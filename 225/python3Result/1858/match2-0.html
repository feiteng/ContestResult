class Solution:
    def maximumTime(self, time: str) -> str:
        result = list(time)
        if time[0] == "?":
            if "0"<=time[1] <= "3" or time[1] == "?":
                result[0] = "2"
            else:
                result[0] = "1"
        if time[1] == "?":
            if result[0] == "2":
                result[1] = "3"
            else:
                result[1] = "9"
        if time[3] == "?":
            result[3] = "5"
        if time[4] == "?":
            result[4] = "9"
        return "".join(result)
