class Solution:
    def maximumTime(self, T: str) -> str:
        s = ""
        if T[0] == "?":
            if T[1] >= '4' and T[1] != '?':
                s += '1'
            else:
                s += '2'
        else:
            s += T[0]
        if T[1] == "?":
            if s[0] == '2':
                s += '3'
            else:
                s += '9'
        else:
            s += T[1]
        
        s += ':'
        
        if T[3] == '?':
            s += '5'
        else:
            s += T[3]
        if T[4] == '?':
            s += '9'
        else:
            s += T[4]
        return s
