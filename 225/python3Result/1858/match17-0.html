class Solution:
    def maximumTime(self, time: str) -> str:
        ans = []
        if time[0] == '?':
            if time[1] in ['0', '1','2','3', '?'] :
                ans.append('2')
            else:
                ans.append('1')    
        else:
            ans.append(time[0])    
        if time[1] == '?':
            if ans[-1] == '2':
                ans.append('3')
            else:
                ans.append('9')    
        else:
            ans.append(time[1])    
        ans.append(':')    
        if time[3] == '?':
            ans.append('5')
        else:
            ans.append(time[3])    
        if time[4] == '?':
            ans.append('9')
        else:
            ans.append(time[4])    
        return "".join(ans)
