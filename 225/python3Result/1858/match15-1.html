class Solution:
    def maximumTime(self, time: str) -> str:
        hours, minutes = time.split(':')
        if hours == '??':
            hours = '23'
        elif '?' == hours[0]:
            hours = ('1' if hours[1] > '3' else '2') + hours[1]
        elif '?' == hours[1]:
            hours = hours[0] + ('3' if hours[0] > '1' else '9')
            
        if minutes == '??':
            minutes = '59'
        elif '?' == minutes[0]:
            minutes = '5' + minutes[1]
        elif '?' == minutes[1]:
            minutes = minutes[0] + '9'
        return hours + ':' + minutes
