class Solution:
    def maximumTime(self, time: str) -> str:
        a,b=time.split(':')
        if a[0]=='?' and a[1]=='?':
            a='23'
        elif a[0]=='?':
            if int(a[1])>3:
                a='1'+a[1]
            else:
                a='2'+a[1]
        elif a[1]=='?':
            if int(a[0])<2:
                a=a[0]+'9'
            else:
                a=a[0]+'3'
        if b[0]=='?' and b[1]=='?':
            b='59'
        elif b[0]=='?':
            b="5"+b[1]
        elif b[1]=='?':
            b=b[0]+'9'
        return a+":"+b
