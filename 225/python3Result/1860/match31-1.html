class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m = len(matrix)
        n = len(matrix[0])
        dp = [[0 for i in range(n)] for j in range(m)]
        dp[0][0] = matrix[0][0]
        for i in range(1, m):
            dp[i][0] = matrix[i][0] ^ dp[i-1][0]
        for j in range(1, n):
            dp[0][j] = matrix[0][j] ^ dp[0][j-1]
        
for i in range(1, m):
            for j in range(1, n):
                dp[i][j] = dp[i-1][j-1]^dp[i-1][j]^dp[i][j-1]^matrix[i][j]
        s = []
        for i in range(m):
            for j in range(n):
                s
.append(dp[i][j])
        s.sort(reverse=True)
        return s[k-1]
