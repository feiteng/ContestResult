class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m = len(matrix)
        n = len(matrix[0])
        
        xorm = [[0 for i in range(n)] for j in range(m)]
        
        xorm[0][0] = matrix[0][0]
        for i in range(1, n):
            xorm[0][i] = xorm[0][i-1]^matrix[0][i]
            
        for i in range(1, m):
            xorm[i][0] = xorm[i-1][0]^matrix[i][0]
            
        for i in range(1, m):
            for j in range(1, n):
                xorm[i][j] = xorm[i-1][j]^xorm[i][j-1]^xorm[i-1][j-1]^matrix[i][j]
                
        l = []
        for i in range(m):
            for j in range(n):
                l.append(xorm[i][j])
                
        #print(xorm)
        l = sorted(l)
        #print(l)
        return l[-k]
