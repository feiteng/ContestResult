import heapq
class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m,n = len(matrix),len(matrix[0])
        dp = [[0 for _ in range(n)] for _ in range(m)]
        dp[0][0]=matrix[0][0]
        res = []
        heapq.heappush(res,dp[0][0])
        for i in range(1,n):
            dp[0][i]=dp[0][i-1]^matrix[0][i]
            heapq.heappush(res,dp[0][i])
        for i in range(1,m):
            dp[i][0]=dp[i-1][0]^matrix[i][0]
            heapq.heappush(res,dp[i][0])
        
        for i in range(1,m):
            for j in range(1,n):
                dp[i][j] = dp[i-1][j-1]^dp[i][j-1]^dp[i-1][j]^matrix[i][j]
                heapq.heappush(res,dp[i][j])
        #print(res)
        return heapq.nlargest(k,res)[-1]
