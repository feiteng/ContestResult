class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        N = len(matrix)
        M = len(matrix[0])
        
        dp = [[0] * (M + 1) for _ in range(N + 1)]
        pq = []
        
        for i in range(N):
            for j in range(M):
                dp[i][j] = matrix[i][j] ^ dp[i - 1][j] ^ dp[i][j - 1] ^ dp[i - 1][j - 1]
                heappush(pq, dp[i][j])
                if len(pq) > k:
                    heappop(pq)
        return pq[0]
