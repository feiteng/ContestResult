class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        R = len(matrix)
        C = len(matrix[0])
        cornerArea = [[0 for c in range(C + 1)] for r in range(R + 1)]
        
for r in range(R):
            for c in range(C):
                cornerArea[r][c] = (
                    matrix[r][c]
                    ^ cornerArea[r - 1][c]
                    ^ cornerArea[r][c - 1]
                    ^ cornerArea[r - 1][c - 1]
                )
                
        arr = []
        for r in range(R):
            for c in range(C):
                arr
.append(cornerArea[r][c])
        arr.sort(reverse=True)
        return arr[k - 1]
