class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m,n=len(matrix),len(matrix[0])
        res=[[0]*(n+1) for i in range(m+1)]
        c=[]
        for i in range(m):
            for j in range(n):
                res[i+1][j+1]=matrix[i][j]^res[i][j]^res[i+1][j]^res[i][j+1]
                c.append(res[i+1][j+1])
        c.sort(reverse=True)
        return c[k-1]
