class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m=len(matrix)
        n=len(matrix[0])
        ro=[[0]*n for _ in range(m)]
        ro[0][0]=matrix[0][0]
        ans=[ro[0][0]]
        for i in range(1,m):
            ro[i][0]=matrix[i][0]^ro[i-1][0]
            ans.append(ro[i][0])
        for j in range(1,n):
            ro[0][j]=matrix[0][j]^ro[0][j-1]
            ans.append(ro[0][j])
        for i in range(1,m):
            for j in range(1,n):
                ro[i][j]=ro[i-1][j-1]^ro[i][j-1]^ro[i-1][j]^matrix[i][j]
                ans.append(ro[i][j])
        ans.sort(reverse=True)
        return ans[k-1]
