class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        n, m = len(matrix), len(matrix[0])
        prefix = [[0] * (m + 1) for _ in range(n + 1)]
        q = []
        for i in range(n):
            for j in range(m):
                prefix[i + 1][j + 1] = prefix[i + 1][j] ^ prefix[i][j + 1] ^ prefix[i][j] ^ matrix[i][j]
                heapq.heappush
(q, prefix[i + 1][j + 1])
                if len(q) > k:
                    heapq.heappop(q)
        return heapq.heappop(q)
