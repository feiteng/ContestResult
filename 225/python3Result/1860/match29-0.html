class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        rows, cols = len(matrix), len(matrix[0])
        coor = [[0] * cols for _ in range(rows)]
        coor[0][0] = matrix[0][0]
        for a in range(1, rows):
            coor[a][0] = coor[a-1][0] ^ matrix[a][0]
        for b in range(1, cols):
            coor[0][b] = coor[0][b-1] ^ matrix[0][b]
        for a in range(1, rows):
            for b in range(1, cols):
                coor[a][b] = coor[a-1][b-1] ^ coor[a-1][b] ^ coor[a][b-1] ^ matrix[a][b]
        
        ans = [coor[i][j] for i in range(rows) for j in range(cols)]
        return heapq.nlargest(k, ans)[-1]
