class Solution:
    def kthLargestValue(self, A: List[List[int]], k: int) -> int:
        
        
        n, m = len(A), len(A[0])
        dp = [[0]*(m) for _ in range(n)]
        for i in range(n):
            for j in range(m):
                if i == 0 and j == 0:
                    dp[i][j] = A[i][j]
                elif i == 0 and j != 0:
                    dp[i][j] = dp[i][j-1] ^ A[i][j]
                elif i != 0 and j == 0:
                    dp[i][j] = dp[i-1][j] ^ A[i][j]
                else:
                    dp[i][j] = dp[i][j-1] ^ dp[i-1][j] ^ dp[i-1][j-1] ^ A[i][j]
        pool = []
        for row in dp:
            pool += row
        pool.sort()
        return pool[-k]
