class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m, n = len(matrix), len(matrix[0])
        A = [[0] * n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if not i and not j: A[i][j] = matrix[i][j]
                elif not i: A[i][j] = matrix[i][j] ^ A[i][j - 1]
                elif not j: A[i][j] = matrix[i][j] ^ A[i - 1][j]
                else: A[i][j] = matrix[i][j] ^ A[i - 1][j] ^ A[i][j - 1] ^ A[i - 1][j - 1]
        B = []
        for a in A:
            B += a
        B.sort(reverse=True)
        return B[k - 1]
