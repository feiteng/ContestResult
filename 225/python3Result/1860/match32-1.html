from heapq import *

class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m, n = len(matrix), len(matrix[0])
        values = [[0 for j in range(n+1)] for i in range(m+1)]
        heap = []
        
for i in range(1, m+1):
            for j in range(1, n+1):
                values[i][j] = matrix[i-1][j-1] ^ values[i-1][j-1] ^ values[i][j-1] ^ values[i-1][j]
                heappush(heap, values[i][j])
                if len(heap) > k:
                    heappop(heap)
        return heap[0]
