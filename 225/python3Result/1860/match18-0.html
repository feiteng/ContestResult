class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m, n = len(matrix), len(matrix[0])
        temp = [[0 for i in range(n)] for j in range(m)]
        out = []
        for i in range(m):
            curr_r = [0]*n
            for j in range(n):
                if i == 0 and j == 0:
                    temp[i][j] = 0^matrix[i][j]
                elif i == 0:
                    temp[i][j] = temp[i][j-1]^matrix[i][j]
                elif j == 0:
                    temp[i][j] = temp[i-1][j]^matrix[i][j]
                else:
                    temp[i][j] = temp[i-1][j]^curr_r[j-1]^matrix[i][j]
                if j > 0:
                    curr_r[j] = curr_r[j-1]^matrix[i][j]
                else:
                    curr_r[j] = matrix[i][j]
                out.append(temp[i][j])
        out.sort(reverse = True)
        return out[k-1]
