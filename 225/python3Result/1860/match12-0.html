class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        M = len(matrix)
        N = len(matrix[0])
        dp = [[0] * N for _ in range(M)]
        dp[0][0] = matrix[0][0]
        vals = [matrix[0][0]]
        for i in range(1, M):
            dp[i][0] = dp[i - 1][0] ^ matrix[i][0]
            vals.append(dp[i][0])
        for j in range(1, N):
            dp[0][j] = dp[0][j - 1] ^ matrix[0][j]
            vals.append(dp[0][j])
        for i in range(1, M):
            for j in range(1, N):
                dp[i][j] = dp[i - 1][j] ^ dp[i][j - 1] ^ dp[i - 1][j - 1] ^ matrix[i][j]
                vals.append(dp[i][j])
        vals.sort(reverse=True)
        return vals[k - 1]
