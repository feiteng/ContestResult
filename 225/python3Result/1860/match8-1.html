class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        n, m = len(matrix), len(matrix[0])
        arr = []
        for i in range(n):
            for j in range(m):
                if i > 0:
                    matrix[i][j] ^= matrix[i-1][j]
                if j > 0:
                    matrix[i][j] ^= matrix[i][j-1]
                if i > 0 and j > 0:
                    matrix[i][j] ^= matrix[i-1][j-1]
                arr.append(matrix[i][j])
        arr.sort(reverse=True)
        # print(matrix,arr)
        return arr[k-1]
