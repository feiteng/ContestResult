class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m,n=len(matrix),len
(matrix[0])
        kmax=[matrix[0][0]]
        for j in range(1,n):
            matrix[0][j]=matrix[0][j]^matrix[0][j-1]
            kmax.append(matrix[0][j])
        for i in range(1,m):
            matrix[i][0]=matrix[i][0]^matrix[i-1][0]
            kmax.append(matrix[i][0])
        for i in range(1,m):
            for j in range(1,n):
                matrix[i][j]=matrix[i][j]^matrix[i][j-1]^matrix[i-1][j]^matrix[i-1][j-1]
                kmax.append(matrix[i][j])
        kmax.sort(reverse=True)
        return kmax[k-1]
