class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m, n = len(matrix), len(matrix[0])
        xor_matrix = [[0 for i in range(n)] for j in range(m)]
        xor_matrix[0][0] = matrix[0][0]
        ans = [matrix
[0][0]]
        for i in range(1, m):
            xor_matrix[i][0] = xor_matrix[i - 1][0] ^ matrix[i][0]
            ans.append(xor_matrix[i][0])
        for j in range(1, n):
            xor_matrix[0][j] = xor_matrix[0][j - 1] ^ matrix[0][j]
            ans.append(xor_matrix[0][j])
        for i in range(1, m):
            for j in range(1, n):
                xor_matrix[i][j] = xor_matrix[i - 1][j - 1] ^ xor_matrix[i][j - 1] ^ xor_matrix[i - 1][j] ^ matrix[i][j]
                ans.append(xor_matrix[i][j])
        return sorted(ans, reverse=True)[k - 1]
