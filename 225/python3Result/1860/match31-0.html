class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m = len(matrix)
        n = len(matrix[0])
        dp = [[0 for _ in range(n+1)] for _ in range(m+1)]
        res = []
            
        for i in range(m):
            for j in range(n):
                dp[i+1][j+1] = dp[i][j+1] ^ dp[i+1][j] ^ dp[i][j] ^ matrix[i][j]
                res.append(dp[i+1][j+1])
        res.sort(reverse = True)
        return res[k-1]
