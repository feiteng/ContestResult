class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m=len(matrix)
        n=len(matrix[0])
        l=[matrix[0][0]]
        for i in range(1,n) :
            matrix[0][i]=matrix[0][i]^matrix[0][i-1]
            l.append(matrix[0][i])
        for i in range(1,m) :
            matrix[i][0]=matrix[i][0]^matrix[i-1][0]
            l.append(matrix[i][0])
        for i in range(1,m) :
            for j in range(1,n) :
                matrix[i][j]=matrix[i-1][j]^matrix[i][j-1]^matrix[i][j]^matrix[i-1][j-1]
                l.append(matrix[i][j])
        l.sort(reverse=True)
        return l[k-1]
