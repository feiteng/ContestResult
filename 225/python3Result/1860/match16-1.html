from heapq import heappush, heappop


class Solution:
    def kthLargestValue(self, matrix: List[List[int]], k: int) -> int:
        m = len(matrix)
        n = len(matrix[0])
        
        new_mat = [[0] * (n + 1) for _ in range(m + 1)]
        # new_mat[1][1] = matrix[0][0]
        
        hq = []
        
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                new_mat[i][j] = new_mat[i - 1][j - 1] ^ new_mat[i - 1][j] ^ new_mat[i][j - 1] ^ matrix[i - 1][j - 1]
                heappush(hq, new_mat[i][j])
                if len(hq) > k:
                    heappop(hq)
                    
        return hq[0]
