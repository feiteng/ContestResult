class Solution {
public:
    int minCharacters(string a, string b) {
        vector<int> va(26), vb(26);

        for(char c : a) va[c - 'a'] += 1;
        for(char c : b) vb[c - 'a'] += 1;
        int ans = 200000;
        for(int i = 0; i < 25; i += 1){
            int pans = 0;
            for(int j = i + 1; j < 26; j += 1) pans += va[j];
            for(int j = 0; j <= i; j += 1) pans += vb[j];
            ans = min(ans, pans);
        }
        for(int i = 0; i < 25; i += 1){

            int pans = 0;
            for(int j = i + 1; j < 26; j += 1) pans += vb[j];
            for(int j = 0; j <= i; j += 1) pans += va[j];
            ans = min(ans, pans);
        }
        for(int i = 0; i < 26; i += 1){
            int pans = 0;
            for(int j = 0; j < 26; j += 1) if(j != i) pans += va[j];
            for(int j = 0; j < 26; j += 1) if(j != i) pans += vb[j];
            ans = min(ans, pans);
        }
        return ans;
    }
};
