class Solution {
public:
    int minCharacters(string a, string b) {
        int alen=a.length(),blen=b.length();

        int ret=0x3f3f3f3f;
        for(int i=1;i<26;i++){
            int tmp=0;
            for(int j=0;j<alen;j++){
                if(a[j]-'a'>=i) tmp++;
            }
            for(int j=0;j<blen;j++){
                if(b[j]-'a'<i) tmp++;
            }
            ret=min(ret,tmp);
        }
        for(int i=1;i<26;i++){
            int tmp=0;
            for(int j=0;j<alen;j++){
                if(a[j]-'a'<i) tmp++;
            }
            for(int j=0;j<blen;j++){
                if(b[j]-'a'>=i) tmp++;
            }
            ret=min(ret,tmp);
        }
        for(int i=0;i<26;i++){
            int tmp=0;
            for(int j=0;j<alen;j++){
                if(a[j]-'a'!=i) tmp++;
            }
            for(int j=0;j<blen;j++){
                if(b[j]-'a'!=i) tmp++;
            }
            ret=min(ret,tmp);
        }
        return ret;
    }
};
