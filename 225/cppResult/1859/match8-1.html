class Solution {
public:
    int minCharacters(string a, string b) {
        int ans = a.length() + b.length();

        for(char x = 'a'; x <= 'z'; x ++) {
            int c = 0;
            for(int i = 0; i < a.length(); i ++) {
                if(a[i] != x)  c++;
            }
            
            for(int i = 0; i < b.length(); i ++) {
                if(b[i] != x)  c++;
            }
            ans = min(ans, c);
        }
        
        for(char x = 'b'; x <= 'z'; x ++) {

            // a >= x, b < x
            int c = 0;
            for(int i = 0; i < a.length(); i ++) {
                if(a[i] < x)  c++;
            }
            for(int i = 0; i < b.length(); i ++) {
                if(b[i] >= x)  c++;
            }
            
            ans = min(ans, c);
        }
        
        
        for(char x = 'b'; x <= 'z'; x ++) {

            // b >= x, a < x
            int c = 0;
            for(int i = 0; i < b.length(); i ++) {
                if(b[i] < x)  c++;
            }
            for(int i = 0; i < a.length(); i ++) {
                if(a[i] >= x)  c++;
            }
            
            ans = min(ans, c);
        }
        
        
        return ans;
    }
};
