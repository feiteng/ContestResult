class Solution {
public:
    int minCharacters(string a, string b) {
        int a_cnt[26] = {0}, b_cnt[26] = {0};
        int max_a = 0, max_b = 0, len_a = a.size(), len_b = b.size();

        for (auto chr:a) {
            a_cnt[chr - 'a']++;
        }
        for (auto chr:b) {
            b_cnt[chr - 'a']++;
        }
        
        int res1 = INT_MAX, res2 = INT_MAX;
        for (int i = 1; i < 26; ++i) {
            max_a = max(max_a, a_cnt[i]);
            max_b = max(max_b, b_cnt[i]);
            int left_a = 0, left_b = 0; 
            for (int j = 0; j < i; ++j) {
                left_a += a_cnt[j];
                left_b += b_cnt[j];
            }
            res1 = min(res1, len_a - left_a + left_b);
            res2 = min(res2, len_b - left_b + left_a);
        }
        // 3
        
        
        int res3 = len_a - max_a + len_b - max_b;
        return min({res1, res2, res3});
    }
};
