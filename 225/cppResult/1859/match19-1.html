class Solution {
public:
    int minCharacters(string a, string b) {
        unordered_map<char, int> ca, cb;
        for (auto i : a) ++ca[i];
        for (auto i : b) ++cb[i];
        int ans = INT_MAX;
        // 1

        for (char c = 'a'; c <= 'z'; c++) {
            int cur = 0;
            for (auto i : a) {
                if (i > c) {
                    ++cur;
                }
            }
            for (auto i : b) {
                if (i <= c) {
                    ++cur;
                }
            }
            ans = min(ans, cur);
        }
        // 2
        for (char c = 'a'; c <= 'z'; c++) {

            int cur = 0;
            for (auto i : a) {
                if (i < c) {
                    ++cur;
                }
            }
            for (auto i : b) {
                if (i >= c) {
                    ++cur;
                }
            }
            ans = min(ans, cur);
        }
        // 3
        for (char c = 'a'; c <= 'z'; c++) {

            int cur = 0;
            for (auto i : a) {
                if (i != c) {
                    ++cur;
                }
            }
            for (auto i : b) {
                if (i != c) {
                    ++cur;
                }
            }
            ans = min(ans, cur);
        }
        return ans;
    }
};
