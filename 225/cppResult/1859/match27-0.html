class Solution {
    const int inf = 1e9;
public:
    int minCharacters(string a, string b) {
        vector<int> hasha(26 , 0);
        vector<int> hashb(26 , 0);
        
        for(auto it : a){
            hasha[it-'a']++;
        }
        for(auto it : b){
            hashb[it-'a']++;
        }
        
        int min1 = inf;
        for(int i = 1 ; i < 26 ; i++){
            int sum1 = 0;
            //for a > b
            for(int j = 0 ; j < i ; j++){
                sum1 += hasha[j];
            }
            for(int j = i ; j < 26 ; j++){
                sum1 += hashb[j];
            }
            
            min1 = min(min1 , sum1);
        }
        for(int i = 1 ; i < 26 ; i++){
            int sum1 = 0;
            //for a < b
            for(int j = 0 ; j < i ; j++){
                sum1 += hashb[j];
            }
            for(int j = i ; j < 26 ; j++){
                sum1 += hasha[j];
            }
            
            min1 = min(min1 , sum1);
        }
        for(int i = 0 ; i < 26 ; i++){
            int sum1 = 0;
            for(int j = 0 ; j < 26 ; j++){
                if(j != i)
                    sum1 += hasha[j] + hashb[j];
            }
            
            min1 = min(min1 , sum1);
        }
        return min1;
    }
};
