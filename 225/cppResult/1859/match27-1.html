class Solution {
public:
    int minCharacters(string a, string b) {
        int cnt = 0;

        map<int, int> mia, mib;
        
        for (auto ch : a) {
            mia[ch]++;
        }
        for (auto ch : b) {
            mib[ch]++;
        }

        int res = INT_MAX;
        for (int i = 'a'; i < 'z'; i++) {
            int ca = 0, cb = 0;

            for (int j = i + 1; j <= 'z'; j++) {
                ca += mia[j];
            }
            for (int j = 'a'; j <= i; j++) {
                cb += mib[j];
            }
            res = min(res, ca + cb);
        }
        
        for (int i = 'b'; i <= 'z'; i++) {
            int ca = 0, cb = 0;

            for (int j = 'a'; j < i; j++) {
                ca += mia[j];
            }
            
            for (int j = i; j <= 'z'; j++) {
                cb += mib[j];
            }
            res = min(res, ca + cb);
        }
        
        for (int i = 'a'; i <= 'z'; i++) {
            int ca = a.size() - mia[i];
            int cb = b.size() - mib[i];
            res = min(res, ca + cb);
        }
        
        return res;
    }
};
