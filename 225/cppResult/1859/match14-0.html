class Solution {
public:
    int minCharacters(string a, string b) {
        int n = a.size();
        int ret = 1e9;
        for (int k = 0; k < 26; ++k) {
            int cur = 0;
            for (auto& c : a) if (c - 'a' != k) ++cur;
            for (auto& c : b) if (c - 'a' != k) ++cur;
            ret = min(ret, cur);
        }
        // a <= k < b
        for (int k = 0; k < 25; ++k) {
            int cur = 0;
            for (auto& c : a) if (c - 'a' > k) ++cur;
            for (auto& c : b) if (c - 'a' <= k) ++cur;
            ret = min(ret, cur);
        }
        // b <= k < a
        for (int k = 0; k < 25; ++k) {
            int cur = 0;
            for (auto& c : b) if (c - 'a' > k) ++cur;
            for (auto& c : a) if (c - 'a' <= k) ++cur;
            ret = min(ret, cur);
        }
        return ret;
    }
};
