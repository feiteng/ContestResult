class Solution {
public:
    int minCharacters(string a, string b) {
        int n=a.size();
        int m=b.size();
        int Cnta[26],Cntb[26];

        for (int i=0;i<26;++i) Cnta[i]=Cntb[i]=0;
        for (int i=0;i<n;++i) ++Cnta[a[i]-'a'];
        for (int i=0;i<m;++i) ++Cntb[b[i]-'a'];
        int Min=n+m;
        for (int i=0;i<26;++i)
        {
            Min=min(Min,n-Cnta[i]+m-Cntb[i]);
        }
        for (int i=0;i<25;++i)
        {
            int cnt=0;
            for (int j=i+1;j<26;++j) cnt+=Cnta[j];
            for (int j=0;j<=i;++j) cnt+=Cntb[j];
            Min=min(Min,cnt);

            cnt=0;
            for (int j=i+1;j<26;++j) cnt+=Cntb[j];
            for (int j=0;j<=i;++j) cnt+=Cnta[j];
            Min=min(Min,cnt);
        }
        return Min;
    }
};
