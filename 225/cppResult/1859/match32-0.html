class Solution {
public:
    int minCharacters(string a, string b) {
        vector<int> fa(26, 0), fb(26, 0);
        for (char c : a) ++fa[c - 'a'];
        for (char c : b) ++fb[c - 'a'];
        int way3 = a.size() - *max_element(fa.begin(), fa.end()) + b.size() - *max_element(fb.begin(), fb.end());
        int way2 = a.size() + b.size(), way1 = way2;
        {
            int x = 0, y = accumulate(fb.begin(), fb.end(), 0);
            for (int i = 0; i < 25; ++i) {
                way2 = min(way2, (x += fa[i]) + (y -= fb[i]));
            }
        }
        {
            int x = 0, y = accumulate(fa.begin(), fa.end(), 0);
            for (int i = 0; i < 25; ++i) {
                way1 = min(way1, (x += fb[i]) + (y -= fa[i]));
            }
        }
        return min(way1, min(way2, way3));

    }
};
