class Solution {
public:
    int minCharacters(string a, string b) {
        int n = a.size(), m = b.size();
        vector<int> v1(26), v2(26);
        for (char& ch: a)
            v1[ch-'a']++;
        for (char& ch: b)

            v2[ch-'a']++;
        int res = max(n,m) +1;
       
        // 1(a==b)
        for (int i=0; i<26; i++) {
            int ct = 0;
            for (int j=0; j<26; j++) {
                if (j==i) continue;
                ct += v1[j] + v2[j];
            }
            res = min(res, ct);
        }

        // 2 (a<b)
        for (int i=1; i<26; i++) {
            // all chars in a should be less than i
            int ct =0;

            for (int j=i; j<26; j++)
                ct += v1[j];
            for (int j=0; j<i;  j++)
                ct += v2[j];
            res = min(res, ct);
        }

        // 3(a>b)
        for (int i=1; i<26; i++) {
            // all chars in b should be less than i
            int ct = 0;

            for (int j=i; j<26; j++)
                ct += v2[j];
            for (int j=0; j<i; j++)
                ct += v1[j];
            res = min(res, ct);
        }
        return res;
    }
};
