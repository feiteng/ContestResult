class Solution {
public:
    int minCharacters(string a, string b) {
        
        vector<int> arrA(26,0);
        vector<int> arrB(26,0);
        int ans = INT_MAX;
        int sumA = 0, curA = 0;
        int sumB = 0, curB = 0;
        for(char c : a){
            arrA[ c - 'a' ]++;
            sumA++;
        }
        for(char c : b){
            arrB[ c - 'a' ]++;
            sumB++;
        }
        for(int i = 0;i < 26; i++){
            curA += arrA[i];
            curB += arrB[i];
            if( i != 25 ){
                ans = min( ans, sumA - curA + curB );
                ans = min( ans, sumB - curB + curA );
            }
            ans = min( ans, ( sumA - arrA[i] + sumB - arrB[i] ) );
            // cout<<curA<<" "<<curB<<" "<<sumA<<" "<<sumB<<endl;
            // cout<<ans<<endl;
        }
        return ans;
    }
};
