class Solution {
public:
    int minCharacters(string a, string b) {
        vector<int> cnta(26,0);
        vector<int> cntb(26,0);
        int n=a.size(),m=b.size();

        for(char c:a) cnta[c-'a']++;
        for(char c:b) cntb[c-'a']++;
        int ans=n+m;
        // cout<<"here"<<endl;
        
        // case 3
        for(int i=0;i<26;i++)
        {

            ans=min(ans,n+m-cnta[i]-cntb[i]);
        }
        
        // case 1
        for(int i=0;i<25;i++)
        {
            int nans=0;
            for(int j=i+1;j<26;j++)
                nans+=cnta[j];
            for(int j=i;j>=0;j--)
                nans+=cntb[j];
            ans=min(ans,nans);
        }
        
        // case 2
        for(int i=0;i<25;i++)
        {
            int nans=0;
            for(int j=i+1;j<26;j++)
                nans+=cntb[j];
            for(int j=i;j>=0;j--)
                nans+=cnta[j];
            ans=min(ans,nans);
        }
        return ans;
    }
};
