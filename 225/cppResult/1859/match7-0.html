class Solution {
public:
    int minCharacters(string a, string b) {

        // solution 1 a <b, what's the biggest letter in a?
        int ans = a.size() + b.size();
        for(int i = 0; i < 25; i++){
            int cnt = 0;
            for(int j = 0; j < a.size(); j++){
                if(a[j] > i + 'a') cnt++;
            }
            for(int j = 0; j < b.size(); j++){
                if(b[j] <= i + 'a') cnt++;
            }
            ans = min(ans, cnt);
        }
        
        // solution 2 a > b what's the biggest letter in b
        for(int i = 0; i < 25; i++){
            int cnt = 0;
            for(int j = 0; j < a.size(); j++){
                if(a[j] <= i + 'a') cnt++;
            }
            for(int j = 0; j < b.size(); j++){
                if(b[j] > i + 'a') cnt++;
            }
            ans = min(ans, cnt);
        }
        
        // solution 3
        for(int i = 0; i < 26; i++){
            int cnt = 0;
            for(int j = 0; j < a.size(); j++){
                if(a[j] != 'a' + i) cnt++;
            }
            for(int j = 0; j < b.size(); j++){
                if(b[j] != 'a' + i) cnt++;
            }
            ans = min(ans, cnt);
        }
        return ans;
        
    }
};
