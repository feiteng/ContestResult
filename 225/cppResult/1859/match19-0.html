class Solution {
public:
    int minCharacters(string a, string b) {
        int x[257];
        int ans=1008611;
        for(int i='a';i<'z';i++){
            int k=0;
            for(auto j:a){
                if(j>i){
                    k++;
                }
            }
            for(auto j:b){
                if(j<=i){
                    k++;
                }
            }
            ans=min(k,ans);
        }
        for(int i='a';i<'z';i++){
            int k=0;
            for(auto j:a){
                if(j<=i){
                    k++;
                }
            }
            for(auto j:b){
                if(j>i){
                    k++;
                }
            }
            ans=min(k,ans);
        }
        for(int i='a';i<='z';i++){
            int k=0;
            for(auto j:a){
                if(j!=i){
                    k++;
                }
            }
            for(auto j:b){
                if(j!=i){
                    k++;
                }
            }
            ans=min(k,ans);
        }
        return ans;
    }
};
