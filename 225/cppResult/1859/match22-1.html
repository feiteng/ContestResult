class Solution {
public:
  int minCharacters(string a, string b) {
    int la = a.size();
    int lb = b.size();
    int best = 0x3f3f3f3f;

    for (char ch = 'a'; ch < 'z'; ch++) {
      int cnt = 0;
      for (auto t: a) if (t > ch) cnt++;
      for (auto t: b) if (t <= ch) cnt++;
      best = min(best, cnt);
    }
    for (char ch = 'a'; ch < 'z'; ch++) {

      int cnt = 0;
      for (auto t: b) if (t > ch) cnt++;
      for (auto t: a) if (t <= ch) cnt++;
      best = min(best, cnt);
    }
    for (char ch = 'a'; ch <= 'z'; ch++) {

      int cnt = 0;
      for (auto t: a) if (t != ch) cnt++;
      for (auto t: b) if (t != ch) cnt++;
      best = min(best, cnt);
    }
    return best;
  }
};
