class Solution {
public:
    int minCharacters(string a, string b) {
        vector<int> cntA(26), cntB(26);
        for (char c : a)
            cntA[c-'a']++;
        for (char c : b)
            cntB[c-'a']++;
        // make both a and b just one distinct
        int ret = (int) a.size() - *max_element(cntA.begin(), cntA.end()) + (int) b.size() - *max_element(cntB.begin(), cntB.end());
        // iterate over cutoff character

        for (int rep=0; rep<2; rep++) {
            int curA = a.size(), curB = 0;
            for (int i=0; i<25; i++) {
                // A <= i, B > i
                curA -= cntA[i];
                curB += cntB[i];
                ret = min(ret, curA + curB);
            }
            swap(a, b);
            swap(cntA, cntB);
        }
        return ret;
    }
};
