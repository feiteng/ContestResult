class Solution {
public:
    int minCharacters(string a, string b) {

        int al[300],bl[300];
        memset(al,0,sizeof(al));
        memset(bl,0,sizeof(bl));   

    for(int i=0;i<a.length();i++)
        al[a[i]]++;
for(int i=0;i<b.length();i++)
    bl[b[i]]++;
        int ans=1e8;
        for(int i=1;i<=25;i++)
        {
            int s1=0;

            int s2=0;
            for(int k=0;k<i;k++)
                s1+=al['a'+k];
            for(int k=25;k>=i;k--)
                s2+=bl['a'+k];
            ans=min(ans,s1+s2);
        }
          for(int i=1;i<=25;i++)
        {
            int s1=0;
            int s2=0;

            for(int k=0;k<i;k++)
                s1+=bl['a'+k];
            for(int k=25;k>=i;k--)
                s2+=al['a'+k];
            ans=min(ans,s1+s2);
        }
        int m=0,m1=0;
       for(int i=97;i<=122;i++)
           m=max(al[i],m);
        for(int i=97;i<=122;i++)
            m1=max(bl[i],m1);
        int p=a.length()+b.length()-m1-m;
            ans=min(ans,p);
               return ans;
               
            
    }
};
