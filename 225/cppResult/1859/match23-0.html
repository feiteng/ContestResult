class Solution {
public:
    int minCharacters(string a, string b) {
        int n1 = a.length(), n2 = b.length();
        vector<int> cnt1(26);
        vector<int> cnt2(26);
        for (char c : a) ++cnt1[c - 'a'];
        for (char c : b) ++cnt2[c - 'a'];
        int res = n1 - (*max_element(cnt1.begin(), cnt1.end())) + n2 - (*max_element(cnt2.begin(), cnt2.end()));
        for (int i = 0; i < 25; ++i) {
          int cur1 = 0, cur2 = 0;
          for (int j = 0; j < 26; ++j) {
            if (j <= i) {
              cur1 += cnt2[j];
              cur2 += cnt1[j];
            } else {
              cur1 += cnt1[j];
              cur2 += cnt2[j];
            }
          }
          res = min(min(cur1, cur2), res);
        }
        return res;
    }
};
