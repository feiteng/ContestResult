class Solution {
public:
    int cnta[26], cntb[26];
    int minCharacters(string a, string b) {
        memset(cnta, 0, sizeof(cnta));
        memset(cntb, 0, sizeof(cntb));
        for (auto& c: a) ++cnta[c-'a'];
        for (auto& c: b) ++cntb[c-'a'];
        int res = 0x3f3f3f3f;
        for (int i = 0; i < 26; i++){
            int tmp = 0;
            for (int j = 0; j < 26; j++){
                if (i == j) continue;
                tmp += cnta[j] + cntb[j];
            }
            res = min(res, tmp);
        }
        for (int i = 0; i < 25; i++){
            int tmpa = 0, tmpb = 0;
            for (int j = i+1; j < 26; j++) tmpa += cnta[j];
            for (int j = 0; j <= i; j++) tmpb += cntb[j];
            res = min(res, tmpa+tmpb);
        }
        for (int i = 0; i < 25; i++){
            int tmpa = 0, tmpb = 0;
            for (int j = i+1; j < 26; j++) tmpb += cntb[j];
            for (int j = 0; j <= i; j++) tmpa += cnta[j];
            res = min(res, tmpa+tmpb);
        }
        return res;
    }
};
