class Solution {
public:
    int minCharacters(string a, string b) {
        
        int ans=a.size()+b.size();
        
        //type a<b
        for(int i='b';i<='z';i++)
        {
            int temp_ans=0;

            for(auto&x:a)
            {
                if(x>=i)
                {
                    temp_ans++;
                }
            }
            
            for(auto&x:b)
            {
                if(x<i)
                {
                    temp_ans++;
                }
            }
            
            ans=min(ans,temp_ans);
        }
        
        //type a>b
        for(int i='b';i<='z';i++)
        {
            int temp_ans=0;

            for(auto&x:b)
            {
                if(x>=i)
                {
                    temp_ans++;
                }
            }
            
            for(auto&x:a)
            {
                if(x<i)
                {
                    temp_ans++;
                }
            }
            
            ans=min(ans,temp_ans);
        }
        
        //type a==b
        for(int i='a';i<='z';i++)
        {
            int temp_ans=0;

            for(auto&x:a)
            {
                if(x!=i)
                {
                    temp_ans++;
                }
            }
            
            for(auto&x:b)
            {
                if(x!=i)
                {
                    temp_ans++;
                }
            }
            
            ans=min(ans,temp_ans);
        }
        
        return ans;
    }
};
