class Solution {
public:
    int test()
    {
        cout<<"test";
return 0x77;    
    } 

    int minCharacters(string a, string b) {
        test();
        return min(func(a,b),func(b,a));      
    }
    int func(string a, string b) {
        int ans=1<<30;
        int maxa=0,maxb=0,mina=99999,minb=99999;
        int cnt1[126]={0},cnt2[126]={0};
        for(int c:a){
            maxa=max(maxa,c);
            mina=min(mina,c);
            cnt1[c]++;
        }
        for(int c:b){
            maxb=max(maxb,c);
            minb=min(minb,c);
            cnt2[c]++;
        }
        
        for(int i='a';i<'z';i++){
            int cnt=0;
            for(int j=i+1;j<='z';j++){
                cnt+=cnt1[j];
            }
            for(int j='a';j<=i;j++){
                cnt+=cnt2[j];
            }
            ans=min(ans,cnt);
        }
        int temp=0;
        for(int i='a';i<='z';i++){
            ans=min(ans,(int)a.size()+(int)b.size()-cnt1[i]-cnt2[i]);
        }
        return ans;
        

    }
};
