class Solution {
public:
int minCharacters(string a, string b) {
    int res=INT_MAX;
    for(int tt=0;tt<25;++tt){
        int temp=0;
        char target='a'+tt;
        for(auto i:a){
            if(i>target){
                ++temp;
            }
        }
        for(auto i:b){
            if(i<=target){
                ++temp;
            }
        }
        res=min(res,temp);
    }
    for(int tt=0;tt<25;++tt){
        int temp=0;
        char target='a'+tt;
        for(auto i:a){
            if(i<=target){
                ++temp;
            }
        }
        for(auto i:b){
            if(i>target){
                ++temp;
            }
        }
        res=min(res,temp);
    }
    for(int tt=0;tt<26;++tt){
        int temp=0;
        char target='a'+tt;
        for(auto i:a){
            if(i!=target){
                ++temp;
            }
        }
        for(auto i:b){
            if(i!=target){
                ++temp;
            }
        }
        res=min(res,temp);
    }
    return res;
 }

};
