class Solution {
public:
    int minCharacters(string a, string b) {
        int n=a.size(),m=b.size();
        vector<int> cnta('z'-'a'+1),cntb('z'-'a'+1);
        for (char x:a) cnta[x-'a']++;
        for (char x:b) cntb[x-'a']++;
        int mini=1e9;
        for (int i=0;i<='z'-'a';i++){
            mini=min(mini,n-cnta[i]+m-cntb[i]);
        }

        for (int i=1;i<='z'-'a';i++){
            int tmp=0;
            for (int j=0;j<i;j++) tmp+=cnta[j];
            for (int j=i;j<='z'-'a';j++) tmp+=cntb[j];
            mini=min(mini,tmp);

            tmp=0;
            for (int j=0;j<i;j++) tmp+=cntb[j];
            for (int j=i;j<='z'-'a';j++) tmp+=cnta[j];
            mini=min(mini,tmp);
        }
        return mini;
    }
};
