class Solution {
public:
    int minCharacters(string a, string b) {
        int sa = a.size(), sb = b.size();
        vector<int> va(26), vb(26);
        for (char c : a) {
            va[c - 'a']++;
        }
        for (char c : b) {
            vb[c - 'a']++;
        }
        int t = 0;
        for (int i = 0; i < 26; ++i) {
            t = max(t, va[i] + vb[i]);
        }
        int res = sa + sb - t;

        int ta = 0, tb = 0;
        for (int i = 25; i > 0; --i) {
            ta += va[i];
            tb += vb[i];
            res = min(res, ta + sb - tb);
            res = min(res, tb + sa - ta);
        }
        return res;
    }
};
