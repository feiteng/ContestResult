class Solution {
public:
  int minCharacters(string a, string b) {
    int dpa[33] = {0}, dpb[33] = {0};
    int n = a.size(), m = b.size(), ans = n + m + 1;
    for (auto x : a) {
      dpa[x - 'a'] += 1;
    }
    for (auto x : b) {
      dpb[x - 'a'] += 1;
    }
    int sa = 0, sb = 0;
    for (int i = 0; i < 25; ++i) {
      sa += dpa[i];
      sb += dpb[i];
      ans = min({ans, n - sa + sb, m - sb + sa, n + m - dpa[i] - dpb[i]});
    }
    ans = min(ans, n + m - dpa[25] - dpb[25]);
    return ans;
  }
};
