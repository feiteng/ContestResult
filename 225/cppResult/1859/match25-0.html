class Solution {
public:
    int minCharacters(string a, string b) {
        int ca[200] = {0}, cb[200] = {0}, n = a.size(), m = b.size();
        for(int i = 0; i < n; i++) ca[a[i]]++;
        for(int i = 0; i < m; i++) cb[b[i]]++;
        int res = INT_MAX;
        for(int i = 'a'; i <= 'z'; i++) {
            res = min(res, n + m - 2 * min(ca[i], cb[i]));
        }
        for(int i = 'a'; i < 'z'; i++) {
            int cur = 0;
            for(int j = i + 1; j <= 'z'; j++) cur += ca[j];
            for(int j = 'a'; j <= i; j++) cur += cb[j];
            res = min(res, cur);
        }
        for(int i = 'a'; i < 'z'; i++) {
            int cur = 0;
            for(int j = i + 1; j <= 'z'; j++) cur += cb[j];
            for(int j = 'a'; j <= i; j++) cur += ca[j];
            res = min(res, cur);
        }
        return res;
    }
};
