class Solution {
public:
    int minCharacters(string a, string b) {

        vector<int> cnt(26,0);
        for(int i=0; i<a.size(); i++) cnt[a[i]-'a']++;
        for(int i=0; i<b.size(); i++) cnt[b[i]-'a']++;
        int res = a.size()+b.size()-*max_element(cnt.begin(),cnt.end());

        for(int i=0; i<25; i++){
            int cnt = 0;
            for(int k=0; k<a.size(); k++) cnt+=(a[k]-'a')>i;
            for(int k=0; k<b.size(); k++) cnt+=(b[k]-'a')<=i;
            res = min(res,cnt);
        }
        swap(a,b);
        for(int i=0; i<25; i++){

            int cnt = 0;
            for(int k=0; k<a.size(); k++) cnt+=(a[k]-'a')>i;
            for(int k=0; k<b.size(); k++) cnt+=(b[k]-'a')<=i;
            res = min(res,cnt);
        }
        return res;
    }
};
