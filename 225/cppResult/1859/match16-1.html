class Solution {
public:
    int minCharacters(string a, string b) {
        int ans = a.size() + b.size();
        int m = a.size();
        int n = b.size();
        vector<int> c(26, 0);
        vector<int> d(26, 0);
        
        for (auto &ch : a) {
            c[ch-'a']++;
        }
        for (auto &ch: b) {
            d[ch-'a']++;
        }
        /*
        if (c[0] == 0 || c[25] == 0)
            ans = min(ans, b.size());
        if (d[0] == 0 || d[25] == 0)
            ans = min(ans, a.size());
        */
        // a == b

        for (int i = 0; i < 26; ++i) {
            ans = min(ans, -c[i]-d[i]+m+n);
        }
        //cout << ans << endl;
        int sum_a = 0;
        int sum_b = 0;
        for (int i = 0; i < 25; ++i) {
            sum_a += c[i];
            sum_b += d[i];
            ans = min(ans, -sum_a+m+sum_b);// a < b
            ans = min(ans, -sum_b+n+sum_a);// a > b
            
            
        }
        
        return ans;
    }
};
