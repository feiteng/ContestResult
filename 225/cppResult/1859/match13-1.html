class Solution {
public:
    int minCharacters(string a, string b) {

        int m=a.size(),n=b.size(),ans=INT_MAX;
        for(int i=0;i<26;++i){
            int cnt=0;
            for(auto c:a) if(c!='a'+i) ++cnt;
            for(auto c:b) if(c!='a'+i) ++cnt;
            ans=min(ans,cnt);
        }
        
        for(int i=0;i<25;++i){
            int cnt=0;
            for(auto c:a) if(c>'a'+i) ++cnt;
            for(auto c:b) if(c<='a'+i) ++cnt;
            ans=min(ans,cnt);
        }
        
        for(int i=0;i<25;++i){
            int cnt=0;
            for(auto c:b) if(c>'a'+i) ++cnt;
            for(auto c:a) if(c<='a'+i) ++cnt;
            ans=min(ans,cnt);
        }
        return ans;
    }
};
