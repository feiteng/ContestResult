class Solution {
public:
    int minCharacters(string a, string b) {
        char c, d;
        int ans = INT_MAX;
        for (c = 'a'; c < 'z'; c++) {
            int x = 0;
            for (auto& y : a) if (y > c) x++;
            for (auto& y : b) if (y <= c) x++;
            ans = min(ans, x);
        }
        for (c = 'a'; c < 'z'; c++) {
            int x = 0;
            for (auto& y : a) if (y <= c) x++;
            for (auto& y : b) if (y > c) x++;
            ans = min(ans, x);
        }
        for (c = 'a'; c <= 'z'; c++) {
            int x = 0;
            for (auto& y : a) if (y != c) x++;
            for (auto& y : b) if (y != c) x++;
            ans = min(ans, x);
        }
        return ans;
    }
};
