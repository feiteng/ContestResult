class Solution {
public:
    int minCharacters(string a, string b) {
        int l1 = a.size();
        int l2 = b.size();
        int ans = max(l1,l2);
        vector<int> cnta(26,0);
        vector<int> cntb(26,0);
        
        for(auto c : a) cnta[c-'a']++;
        for(auto c : b) cntb[c-'a']++;
        for(int i = 0; i < 26; ++i){
            /*same*/
            ans = min(ans,l1-cnta[i]+l2-cntb[i]);
        }
        int prea = 0;
        int preb = 0;
        for(int i = 0; i < 25; ++i){
            prea += cnta[i];
            preb += cntb[i];
            ans = min(ans,l1-prea + preb);
            ans = min(ans,l2-preb + prea);
        } 
        return ans;        
    }
};
