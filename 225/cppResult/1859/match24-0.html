class Solution {
public:
    int minCharacters(string a, string b) {
        vector<int> list_a(26, 0);
        vector<int> list_b(26, 0);
        for (int i = 0; i < a.size(); ++i) list_a[a[i] - 'a'] += 1;
        for (int i = 0; i < b.size(); ++i) list_b[b[i] - 'a'] += 1;
        //for (int i = 0; i < list_a.size(); ++i) cout << list_a[i] << endl;
        int total_num = a.size() + b.size();
        int min_num = total_num;
        for (int i = 0; i < 26; i++) min_num = min(min_num, total_num - list_a[i] - list_b[i]);
        for (int i = 1; i <= 25; i++) {
            //方法1需要改变的字母数
            int method1 = 0;
            for (int j = 0; j < i; j++) method1 += list_b[j];//list_b小于i的都要变成i
            for (int j = i; j < 26; j++) method1 += list_a[j];//list_a大于等于i的都要变成i-1
            min_num = min(method1, min_num);
            //方法2需要改变的字母数
            int method2 = 0;
            for (int j = 0; j < i; j++) method2 += list_a[j];//list_a小于i的都要变成i
            for (int j = i; j < 26; j++) method2 += list_b[j];//list_b大于等于i的都要变成i-1
            min_num = min(method2, min_num);
        }       
        return min_num;

    }
};
