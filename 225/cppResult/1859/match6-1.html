class Solution {
public:
    
    int strictlyLess(string& s, string& t) {
        int ans = s.size() + t.size();
        for (char c = 'a'; c < 'z'; c++) {
            int cnt = 0;
            for (char a : s) if (a > c) cnt++;
            for (char b : t) if (b <= c) cnt++;
            ans = min(ans, cnt);
        }
        return ans;
    }
    
    int oneDistinct(string& s, string &t) {
        int ans = s.size() + t.size();

        for (char c = 'a'; c <= 'z'; ++c) {
            int cnt = 0;
            for (char a : s) if (a != c) cnt++;
            for (char b : t) if (b != c) cnt++;
            ans = min(ans, cnt);
        }
        return ans;
    }
    
    int minCharacters(string a, string b) {
        return min(min(strictlyLess(a, b), strictlyLess(b,a)), oneDistinct(a, b));
    }
};
