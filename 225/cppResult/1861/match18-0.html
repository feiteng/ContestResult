class Solution {
public:
    int minimumBoxes(int n) {
        if (n < 4)
            return n;
        
        int sum = 4, cnt = 3;
        
        for (int i = 3; sum < n; ++i) {
            for (int j = 1; sum < n && j <= i; ++j) {
                sum += j;
                cnt++;
            }
        }
        
        return cnt;
    }
};
