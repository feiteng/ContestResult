class Solution {
public:
    int minimumBoxes(int n) {
        int i = 0, c = 1, sum = 0;
        while (sum < n) {
            i += c;
            ++c;
            sum += i;
        }
        int tmp = sum - i, cc = 1;
        i = i - c + 1;
        while (tmp < n) {
            tmp += cc;
            ++cc;
            ++i;
        }
        // while (sum - n >= c) {
        //     sum -= c;
        //     --c;
        //     --i;
        // } 
        return i;
    }
};
