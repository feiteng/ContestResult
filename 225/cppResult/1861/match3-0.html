class Solution {
public:
    int minimumBoxes(int n) {
        int i=1,layer=1,ans=0;
        while(n>=layer){
            ans+=i;
            n-=layer;
            i++;
            layer+=i;
        }
        int j=1;
        while(n>=j){
            ans++;
            n-=j;
            j++;
        }
        if(n>0)ans++;
        return ans;
    }
};
