class Solution {
public:
    int minimumBoxes(int n) {
        int supports = 0;
        int newRow = 0;
        while (supports < n) {
            newRow++;
            supports += (1 + newRow) * newRow / 2;
        }
        if (supports == n) return (1 + newRow) * newRow / 2;
        
        supports -= (1 + newRow) * newRow / 2;
        int result = newRow * (newRow - 1) / 2;
        // printf("newRow: %d\n", newRow);
        for (int i = 1; i <= newRow; i++) {
            // printf("supports: %d result: %d\n", supports, result);
            supports += i;
            result++;
            if (supports >= n) break;
        }
        return result;
    }
};
