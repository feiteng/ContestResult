class Solution {
public:
    int minimumBoxes(int n) {
        int w = 0 , x = 1 , y = 2 ;
        while(w + x <= n){
            w += x ;
            x += y ++ ;
        }
        int ans = x - y + 1 ;
        x = 1 ;
        while(w + x <= n){
            w += x ++ ;
            ans ++ ;
        }
        if(w < n)
            ans ++ ;
        return ans ;
    }
};
