class Solution {
public:
    int minimumBoxes(int n) {
        int k = 1;
        int s = 0;
        while (s + k * (k + 1) / 2 <= n) {
            s += k * (k + 1) / 2;
            k ++ ;
        }
        
        int res = k * (k - 1) / 2, i = 1;
        
        while (s < n) {
            s += i;
            i ++ ;
            res ++ ;
        }
        
        return res;
    }
};
