typedef long long ll;
//ll a[100005];
class Solution {
public:
     int m;
     int f(int x){
        int l=1,r=1e9;
        while(l<=r){
            int mid=(l+r)>>1;
            if(1LL*mid*(mid+1)>>1>=x) r=mid-1;
            else l=mid+1;
        }
         return l;
     }
    ll check(int x){
        ll s=x;
        while(x>1){
            int y=f(x);
       //    printf("x=%d,y=%d\n",x,y);
            s+=(x-y);
            x-=y;  
        }
        return s;
    }
    int minimumBoxes(int n) {
       int l=1,r=1e9;
  //      printf("%d\n",f(1));
      //  printf("%d\n",check(3));
       while(l<=r){
            int m=(l+r)>>1;
           if(check(m)>=n) r=m-1;
           else l=m+1;
       }
       
        return l;
    }
};
