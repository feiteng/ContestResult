class Solution {
public:
    int minimumBoxes(int n) {
        if(n < 4)
            return n;

        int i = 1;
        int i2 = 2;
        int ret = 1;
        while(n > i){
            n -= i;
            ret = i;
            i += i2;
            i2++;
        }
        //cout << n << endl;
        int temp = 1;
        while(n > 0){
            n -= temp;
            ret++;
            temp++;
        }
        return ret;
    }
};
