class Solution {
public:
    int minimumBoxes(int n) {
        if(n<=3) return n;
        if(n==4) return 3;

        int cnt=4, ans=3, should=3;
        while(cnt<n){
            for(int i=1;i<=should;i++){
                ans+=1;
                cnt+=i;
                if(cnt>=n) break;
            }
            should++;
        }
        return ans;
    }
};
