#include<iostream>
class Solution {
public:
    int test()
    {
        cout<<"test";    
        return 0x77;
    }    

    int minimumBoxes(int n) {
        if(n==1)return 1;
        test();
        vector<int> temp{1};
        int ans=0;
        int all=1;
        while(all<n){
            temp.push_back(temp.back()+temp.size()+1);
            all+=temp.back();
        }
        all-=temp.back();
        ans=temp[temp.size()-2];
        int rest=n-all;
        // if(rest==0)return ans;
        int p=1;
        while(rest>0){
            rest-=p;
            p++;
            ans++;
        }
        return ans;
        
    }
};
