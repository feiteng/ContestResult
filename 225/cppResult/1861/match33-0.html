class Solution {
public:
    
    int minimumBoxes(int n) {
        int sum = 0;
        int i = 0;
        while(sum < n)
        {
            ++i;
            sum += (1 + i) * i / 2;
            //printf("sum %d n %d i %d\n", sum, n, i);
        }
        if(sum == n)
        {
            return (1 + i) * i / 2;
        }
        sum -= (1 + i) * i / 2;
        int count = (1 + i - 1) * (i - 1) / 2;
        int j = 0;
        while(sum < n)
        {
            ++j;
            sum += j;
        }
        return count + j;
    }
};
