class Solution {
public:
    int minimumBoxes(int n) {
        int cnt = 1;
        int sum = 1;
        int times = 0;
        while (n - sum> 0) {
            
            n -= sum;
            times += cnt;
            cnt++;
            sum += cnt;
            
            // cout << n << endl;
        }
        // cout << n << endl;
        int t = 1;
        while (n > 0) {
            n -= t;
            t++;
            times++;
        }
        return times;
        
    }
};
