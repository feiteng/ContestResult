class Solution {
public:
    int minimumBoxes(int n) {
        int lvl = 1;
        int cnt = 0;
        while (cnt + lvl*(lvl+1) / 2 < n) {
            cnt += lvl*(lvl+1)/2;
            ++lvl;
        }
        int ext = 0;
        while (cnt+ext*(ext+1)/2 < n) ++ext;
        return ext + lvl*(lvl-1)/2;
    }
    

};
