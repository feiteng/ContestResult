class Solution {
public:
    int minimumBoxes(int n) {
        int res = 0;
        int t = 0;
        int s = 0;
        for (int i = 1; t+s+i<=n; ++i) {
            s += i;
            t += s;
        }
        res = s;
        n-=t;
        int i = 1;
        while(n>0) {
            n-=i++;
            res++;
        }
        return res;
    }
};
