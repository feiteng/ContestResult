class Solution {
public:
    int minimumBoxes(int n) {
        if (n<4) return n;
        int l=0, ls=0,ts=0;
        while (ts<n) {
            ++l;
            ls+=l;
            ts+=ls;
        }
        int r=ls;
        while (ts>n) {
            if (ts-n<l) break;
            --r;
            ts-=l;
            --l;
        }
        return r;
    }
};
