const int N = 1e6 + 10;

int a[N];

bool check(int x, int n)
{
    int sum = x;
    if(sum > n)  return true;
    while(x)
    {
        int l = 1, r = 44721;
        while(l < r)
        {
            int mid = l + r + 1 >> 1;
            if(a[mid] <= x)  l = mid;
            else  r = mid - 1;
        }
        x = a[l - 1] + max(0, x - a[l] - 1);
        sum += x;
        if(sum >= n)  return true;
    }
    return false;
}

class Solution {
public:
    int minimumBoxes(int n) {
        
        for(int i = 1;; i ++)
        {
            a[i] = a[i - 1] + i;
            if(a[i] > 1e9)  break;
        }
        
        int l = 1, r = 1651653;
        while(l < r)
        {
            int mid = l + r >> 1;
            if(check(mid, n))  r = mid;
            else  l = mid + 1;
        }
        return l;
    }
};
