class Solution {
public:
    int minimumBoxes(int n) {
        if(n < 4) return n;
        int sum = 0, cur = 0, level = 0;
        while(sum + cur <= n) {
            sum += cur;
            level++;
            cur += level;
        }
        
        int ret = cur - level;
        n = n - sum;
        sum = 0, cur = 0;
        while(sum < n) {
            cur++;
            sum += cur;
        }
        return ret + cur;
    }
};
