class Solution {
public:
    int minimumBoxes(int n) {
        vector<int> res;
        

        int i = 1;
        int cur = 1;
        int sum = 1;
        res.push_back(1);
        while (sum < n) {
            ++i;
            cur += i;
            //res.push_back(cur);
            sum += cur;
        }
        
        while (sum - i >= n) {
            sum -= i;
            cur -= 1;
            --i;
        }
        return cur;
        //return res.size();
    }
};
