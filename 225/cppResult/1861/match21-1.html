class Solution {
public:
    int minimumBoxes(int n) {
        int base = 1, res = 0;
        while (n > 0) {
            for (int i = 1; i <= base; ++i) {
                if (n <= 0) return res;
                n -= i;
                res++;
            }
            base++;
        }
        return res;
    }
};
