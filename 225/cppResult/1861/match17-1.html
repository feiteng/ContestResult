class Solution {
public:
    int minimumBoxes(int n) {
        int sum=1;
        int bottom=1;
        int i=2;
        while(n>sum)
        {
            bottom+=i;
            i++;
            sum+=bottom;
        }
        if(n==sum)
            return bottom;
        sum-=bottom;
        i--;
        bottom-=i;
        int a=n-sum;
        if(a<=2)
            return bottom+a;

        bottom+=2;
        a-=3;
        int count=3;
        while(a>0)
        {
            bottom++;
            a-=count;
            count++;
        }
        return bottom;
    }
};
