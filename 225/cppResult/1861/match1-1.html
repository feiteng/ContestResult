class Solution {
public:
    int minimumBoxes(int n) {
        int row = 0, res = 0, level = 0, total = 0;
        while(total < n){
            row++;
            level += row;
            total += level;
            //cout<<row<<" "<<level<<" "<<total<<endl;         
        }
        int add = n - (total - level);

        if(add == 0) return level;
        level -= row;
        row = 1, total = 0;
        while(total + row < add){
            total += row;
            row++;
        }
        
        return level + row;
    }
};
