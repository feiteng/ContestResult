class Solution {
public:
   int minimumBoxes(int n) {
    int value = 0;
    int lv = 0;
    int next_bottom = 1;
    int bottom = 0;
    while (value <= n)
    {
        value += next_bottom;
        lv++;
        bottom = (next_bottom);
        next_bottom = (1 + lv + 1) * (lv + 1) / 2;
    }
    value -= bottom;
    lv--;
    bottom=(1 + lv) * (lv) / 2;
    if (value == n)
    {
        return bottom;
    }
    else
    {
        int last = n - value;
        int count = 0;
        for (int i = 1; i <= lv+1; ++i)
        {
            if (last > i)
            {
                last -= i;
                count++;
            }
            else
            {
                count++;
                return bottom + count;
            }
        }
        return count + bottom;
    }
}
};
