#define ll long long
class Solution {
public:
    int minimumBoxes(int n) {
        ll lo=0,hi=1e5,ans=-1;
        while(lo<=hi){
            ll mi=lo+(hi-lo)/2;
            if((mi*(mi+1)*(mi+2))/6ll<=n){
                ans=mi;
                lo=mi+1;
            }
            else hi=mi-1;
        }
        ll k=ans*(ans+1)/2ll,x=n-ans*(ans+1)*(ans+2)/6;
        lo=0,hi=1e5,ans=0;
        while(lo<=hi){
            ll mi=lo+(hi-lo)/2;
            if(mi*(mi+1)/2>=x){
                ans=mi;
                hi=mi-1;
            }
            else lo=mi+1;
        }
        return k+ans;
    }
};
