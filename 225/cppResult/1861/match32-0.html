class Solution {
public:
    int minimumBoxes(int n) {
        int sum = 0;
        int increase = 1;
        while( n >= sum + (increase + 1)*(increase)/2 ){
            sum += (increase + 1)*(increase)/2;
            increase++;
        }
        increase--;
        int ans = (increase + 1)*(increase)/2;
        // cout<<sum<<endl;
        // cout<<increase<<endl;
        if( n == sum ) return ans;
        n -= sum;
        if( n == 1 ) return ans + 1;
        if( n == 2) return ans + 2;
        int next = 1;
        while(n > 0){
            // cout<<n<<" ";
            n -= next;
            next++;
            ans++;
        }
        // cout<<n<<" ";
        
        return ans;
    }
};
