class Solution {
public:
    
    using ll = long long;
    int minimumBoxes(int n) {
        if(n == 1) {
            return 1;
        }
        

        ll l = 1, r = 5e5;
        ll k = -1;
        while(l <= r) {
            ll mid = l+(r-l)/2;
            if(mid*(mid+1)*(mid+2)/6 <= n) {
                k = mid;
                l = mid+1;
            }
            else {
                r = mid-1;
            }

        }
        
        ll ans = k*(k+1)/2;
        // now we have k
        ll rem = n - (k)*(k+1)*(k+2)/6;
        
        //cout << "k, rem = " << k << ", " << rem << endl;
        // if(rem == 1) {
        //     return ans+1;
        // }
        
        // 1 + (2) + (3) + (x) >= rem
        ll x = -1;
        l = 0, r = rem;
        while(l <= r) {
            ll mid = l+(r-l)/2;
            if(mid*(mid+1)/2 >= rem) {
                x = mid;
                r = mid-1;
            }
            else {
                l = mid+1;
            }
        }
        //cout << "x = " << x << endl;
        ans += x;
        return ans;
    }
};
