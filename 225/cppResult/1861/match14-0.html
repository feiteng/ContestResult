const int N = 45000;
class Solution {
    int f[N];
    bool dfs(int n, int x) {
        if(n <= 0) return true;
        if(x <= 0) return false;
        // 计算x个方块上面最多可以放置多少个方块
        // 计算x可以摆成多少层
        int left = 1, right = N;
        while(left < right) {
            int mid = (left + right + 1) / 2;
            if(f[mid] <= x) left = mid;
            else right = mid - 1;
        }
        int k = f[left - 1];
        if(x - f[left]) k += (x - f[left] - 1);
        return dfs(n - x, k);
    }
public:
    int minimumBoxes(int n) {
        int left = 1, right = n;
        f[1] = 1;
        for(int i = 2; i < N; i++) f[i] = f[i - 1] + i;
        while(left < right) {
            int mid = (left + right) / 2;
            // printf("left: %d, mid: %d, right: %d\n", left, mid, right);
            if(dfs(n, mid)) right = mid;
            else left = mid + 1;
        }
        return left;
    }
};
