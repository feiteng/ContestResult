class Solution {
public:

    int minimumBoxes(int n) {
        int l = 0,r = 1e4;
        while(l < r){
            int mid = (l + r + 1) >> 1;
            if(1ll * mid * (mid + 1) * (mid + 2) / 6 <= n) l = mid;
            else r = mid - 1;
        }
        int t = n - 1ll * (r) * (r + 1) * (r + 2) / 6;
        // cout << (1 + r)  * r / 2 <<endl;

        int l1 = 0,r1 = 1e4;
        // cout << t << endl;
        while(l1 < r1){
            int mid = (l1 + r1) >> 1;
            if(1ll * (mid ) * (mid + 1) / 2>= t) r1 = mid;
            else l1 = mid + 1;
        }
        // cout << r1 << endl;
        return 1ll * (1 + r) * (r) / 2 + r1;
    }
};
