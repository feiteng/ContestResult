class Solution {
public:
    int minimumBoxes(int n) {
        int low = 1;
        int diag = 1;
        int total = 1;
        
        while (n > total) {
            diag += 1;
            low += diag;
            total += low;
        }
        while (n <= (total - diag)) {
            total -= diag;
            low -= 1;
            diag -= 1;
        }
        return low;
    }
};
