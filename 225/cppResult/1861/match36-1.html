class Solution {
public:
    int minimumBoxes(int n) {
        long long i = 1;
        while (n > A(i)) i++;
        i--;
        long long res = B(i);
        n -= A(i);
        if (n > 0) {
            long long j = 1;
            while (n > B(j)) j++;
            res += j;
        }
        return res;
    }
    long long A(long long x) {
        return (x * (x + 1) / 2) * (x + 2) / 3;
    }
    long long B(long long x) {
        return x * (x + 1) / 2;
    }
};
