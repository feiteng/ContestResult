class Solution {
public:
    int minimumBoxes(int n) {
        int mat = 1;//总数
        int t = 1;//
        int x = 1;//底部数量
        while(mat < n){
            t++;
            x += t;
            mat += x;
        }
        if(mat == n){
            return x;
        }
        while(mat >= n){
            mat -= t;
            t--;
            x--;
        }
        return x + 1;
    }
};
