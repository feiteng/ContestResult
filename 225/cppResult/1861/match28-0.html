class Solution {
public:
    int minimumBoxes(int n) {
        int curhigh = 1;
        int curnum = 0;
        while(curnum + curhigh * (curhigh + 1) / 2 <= n)
        {
            curnum += curhigh * (curhigh + 1) / 2;
            ++curhigh;
            //cout << curnum << endl;
        }
        int numonfloor = curhigh * (curhigh - 1) / 2;
        int last = n - curnum;
        int plus = 0;
        while(last > 0)
        {
            ++numonfloor;
            ++plus;
            last -= plus;
        }
        return numonfloor;
    }
};
