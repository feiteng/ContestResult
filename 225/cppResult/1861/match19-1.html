typedef long long ll;
class Solution {
public:

    int cal(int x){
        int l=1,r=45000;
        while(r-l>1){
            int mid=(l+r)/2;
            if(1LL*mid*(mid+1)/2<=x)l=mid;
            else r=mid;
        }
        return l;
    }
    ll getsum(ll n){
        return (n*(n+1)*(2*n+1)/6+n*(n+1)/2)/2;
    }
    bool ok(int x,int k){
        if(x<=k)return 1;
        ll a=cal(k);
        ll sum=getsum(a);
        ll add=k-a*(a+1)/2;
        sum+=add*(add+1)/2;
        if(sum>=x)return 1;

        return 0;
    }
    int minimumBoxes(int n) {
        int l=0,r=1e9;
        while(r-l>1){
            int mid=(l+r)/2;
            if(ok(n,mid))r=mid;
            else l=mid;
        }
        return r;
    }
};
