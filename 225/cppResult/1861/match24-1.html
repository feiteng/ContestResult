class Solution {
public:
    int minimumBoxes(int n) {
        int sum = 0, curr = 0, l = 0;
        while (sum < n) {
            l++;
            curr += l;
            sum += curr;
        }
        int ans = 0;
        for (int i = 1; i <= l; i++)
            ans += i;
        while (sum - l >= n) {
            sum -= l;
            l--;
            ans--;
        }
        return ans;
    }
};
