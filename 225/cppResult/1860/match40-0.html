class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int> > a(n + 1, vector<int> (m + 1));
        vector<int> v;
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                a[i][j] = a[i - 1][j] ^ a[i][j - 1] ^ a[i - 1][j - 1] ^ matrix[i - 1][j - 1];
                //cout << a[i][j] << endl;
                v.push_back(a[i][j]);
            }
        }
        
        sort(v.rbegin(), v.rend());
        return v[k - 1];
        
    }
};
