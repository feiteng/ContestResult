class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> nm(m + 1, vector<int>(n + 1));
        priority_queue<int, vector<int>, greater<int>> pque;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                nm[i + 1][j + 1] = nm[i + 1][j] ^ nm[i][j + 1] ^ nm[i][j] ^ matrix[i][j];
                pque.push(nm[i + 1][j + 1]);    if (pque.size() > k)    pque.pop();
            }
        }
        return pque.top();
    }
};
