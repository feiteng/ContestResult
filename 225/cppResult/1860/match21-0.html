class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> coor(m, vector<int>(n));
        vector<int> res;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                coor[i][j] = matrix[i][j];
                if (i > 0) coor[i][j] ^= coor[i-1][j];
                if (j > 0) coor[i][j] ^= coor[i][j-1];
                if (i > 0 && j > 0) coor[i][j] ^= coor[i-1][j-1];
                res.push_back(coor[i][j]);
            }
        }
        sort(res.begin(), res.end(), greater<int>());
        return res[k-1];
    }
};
