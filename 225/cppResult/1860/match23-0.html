class Solution {
public:
    int kthLargestValue(vector<vector<int>>& a, int k) {
        int m=a.size(),n=a[0].size();
        vector<int>b;
        vector<vector<int> >v(m+1,vector<int>(n+1,0));
        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++){
                v[i+1][j+1]=v[i][j+1]^v[i+1][j]^v[i][j]^a[i][j];
                b.push_back(v[i+1][j+1]);
            }
        sort(b.rbegin(),b.rend());
        return b[k-1];
    }
};
