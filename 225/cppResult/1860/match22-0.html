class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<int> res;
        vector<vector<int>> cor(m, vector<int>(n, 0));
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                cor[i][j] ^= matrix[i][j];
                if (i > 0) cor[i][j] ^= cor[i - 1][j];
                if (j > 0) cor[i][j] ^= cor[i][j - 1];
                if (i > 0 && j > 0) cor[i][j] ^= cor[i - 1][j - 1];
                res.push_back(cor[i][j]);
            }
        }
        sort(res.begin(), res.end(), greater<int>());
        return res[k - 1];
    }
};
