class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> xors(m, vector<int>(n, 0));
        priority_queue<int, vector<int>, greater<int>> kth;
        
        xors[0][0] = matrix[0][0];
        kth.push(xors[0][0]);
        for (int i = 1; i < m; ++i) {
            xors[i][0] = xors[i-1][0] ^ matrix[i][0];
            kth.push(xors[i][0]);
            if (kth.size() > k)
                kth.pop();
        }
        for (int j = 1; j < n; ++j) {
            xors[0][j] = xors[0][j-1] ^ matrix[0][j];
            kth.push(xors[0][j]);
            if (kth.size() > k)
                kth.pop();
        }
            
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                xors[i][j] = xors[i-1][j] ^ xors[i][j-1] ^ xors[i-1][j-1] ^ matrix[i][j];
                kth.push(xors[i][j]);
                if (kth.size() > k)
                kth.pop();
            }
        }
        
        return kth.top();
    }
};
