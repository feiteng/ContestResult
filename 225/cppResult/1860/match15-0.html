class Solution {
public:
    int kthLargestValue(vector<vector<int>>& mat, int k) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<int>> pre(m, vector<int>(n, 0));
        
        pre[0][0] = mat[0][0];
        for (int i = 1; i < m; i++) {
            pre[i][0] = pre[i-1][0]^mat[i][0];
        }
        for (int j = 1; j < n; j++)  {
            pre[0][j] = pre[0][j-1]^mat[0][j];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                pre[i][j] = pre[i-1][j]^pre[i][j-1]^mat[i][j]^pre[i-1][j-1];   
            }
        }
        
        vector<int> ans;   
        for (int i = 0; i < m; i++) {
            for (int j  = 0; j < n; j++) {
//                cout  << pre[i][j] << " ";
                ans.push_back(pre[i][j]);
            }
//            cout << endl;
        }
        
        sort(ans.begin(), ans.end());
        
        
        return ans[m*n-k];
    }
};
