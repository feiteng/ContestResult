class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<vector<int>> DP(m,vector<int>(n,0));
        vector<int> arr;
        DP[0][0] = matrix[0][0];
        arr.push_back( DP[0][0] );
        for(int i = 1;i < n; i++){
            DP[0][i] = matrix[0][i];
            DP[0][i] ^= DP[0][i - 1];
            arr.push_back( DP[0][i] );
        }
        for(int i = 1;i < m; i++){
            DP[i][0] = matrix[i][0];
            DP[i][0] ^= DP[i - 1][0];
            arr.push_back(DP[i][0]);
        }
        for(int i = 1;i < m; i++){
            for(int j = 1;j < n; j++){
                DP[i][j] = matrix[i][j];
                DP[i][j] ^= DP[i - 1][j];
                DP[i][j] ^= DP[i][j - 1];
                DP[i][j] ^= DP[i - 1][j - 1];
                arr.push_back( DP[i][j] );
            }
        }
        sort(arr.begin(), arr.end());
        // for(int a : arr){
        //     cout<<a<<" ";
        // }
        return arr[arr.size() - k];
    }
};
