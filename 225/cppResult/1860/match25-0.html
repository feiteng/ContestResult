class Solution {
public:
    int kthLargestValue(vector<vector<int>>& A, int k) {
      int m = A.size(), n = A[0].size();
      
      vector<vector<int>> memo(m + 1, vector<int>(n + 1));
      //multiset<int> res;
      vector<int> v;
      
      for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
          memo[i][j] = memo[i][j-1] ^ memo[i-1][j] ^ memo[i-1][j-1] ^ A[i-1][j-1];
          
          //res.insert(memo[i][j]);
          //if (res.size() > k) res.erase(res.begin());
          v.push_back(memo[i][j]);
        }
      }
      
      // do quick-select to find the k-th one
      nth_element(v.begin(), v.begin() + k - 1, v.end(), greater<int>());
      
      return v[k - 1];
    }
};
