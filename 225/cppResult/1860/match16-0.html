class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size();
        int n = matrix[0].size();
        
        for (int i=1; i<m; i++) {
            for (int j=0; j<n; j++) {
                matrix[i][j] ^= matrix[i-1][j];
            }
        }
        
        for (int j=1; j<n; j++) {
            for (int i=0; i<m; i++) {
                matrix[i][j] ^= matrix[i][j-1];
            }
        }
        
        vector<int> arr;
        for (int i=0; i<m; i++) {
            for (int j=0; j<n; j++) {
                arr.push_back(matrix[i][j]);
            }
        }
        
        sort(arr.begin(), arr.end(), greater());
        return arr[k-1];
    }
};
