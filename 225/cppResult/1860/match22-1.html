class Solution {
public:
    int kthLargestValue(vector<vector<int>>& mat, int k) {
        int n=mat.size(),m=mat[0].size();
        vector<vector<int>> dp(n,vector<int>(m,0));
        vector<int> v;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                dp[i][j]=mat[i][j];
                if(i>0)dp[i][j]=dp[i][j]^dp[i-1][j];
                if(j>0)dp[i][j]=dp[i][j]^dp[i][j-1];
                if(i>0 && j>0)dp[i][j]=dp[i][j]^dp[i-1][j-1];
                v.push_back(dp[i][j]);
            }
        }
        sort(v.begin(),v.end(),greater<int>());
        return v[k-1];
    }
};
