class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size(); 
        vector<vector<int>> value(m, vector<int>(n, 0)); 
        value[0][0] = matrix[0][0]; 
        for(int i = 1; i < m; i ++) { // 1st col 
            value[i][0] = value[i-1][0] ^ matrix[i][0]; 
        }
        for(int i = 1; i < n; i ++) { // 1st row
            value[0][i] = value[0][i-1] ^ matrix[0][i]; 
        }
        for(int i = 1; i < m; i ++) {
            for(int j = 1; j < n; j ++) {
                value[i][j] = value[i][j-1] ^ value[i-1][j] ^ value[i-1][j-1] ^ matrix[i][j];
            }
        }
        
        vector<int> value1d(m * n); 
        for(int i = 0; i < m; i ++) {
            for(int j = 0; j < n; j ++) {
                value1d[i * n + j] = value[i][j];
            }
        }
        sort(value1d.begin(), value1d.end()); 
        
        return value1d[m * n - k];
        
    }
};
