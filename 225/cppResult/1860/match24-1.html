class Solution {
public:
    int kthLargestValue(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        vector<vector<int> > sm(n, vector<int>(m, 0));
        sm[0][0] = mat[0][0];
        vector<int> vals;
        vals.emplace_back(sm[0][0]);
        for(int i = 1; i < m; i++){
            sm[0][i] = mat[0][i] ^ sm[0][i-1];
            vals.emplace_back(sm[0][i]);
        }
        for(int i = 1; i < n; i++){
            sm[i][0] = mat[i][0] ^ sm[i-1][0];
            vals.emplace_back(sm[i][0]);
        }
        
        for(int i = 1; i < n; i++){
            for(int j = 1; j < m; j++){
                sm[i][j] = mat[i][j] ^ sm[i-1][j] ^ sm[i][j-1] ^ sm[i-1][j-1];
                vals.emplace_back(sm[i][j]);
            }
        }
        sort(vals.rbegin(), vals.rend());
        return vals[k-1];
    
    }
};
