class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size();
        int n = matrix[0].size();
        
        vector<vector<int> >ans(m,vector<int>(n,0));
        vector<int> l;
        ans[0][0] = matrix[0][0];
        l.push_back(ans[0][0]);
        
        for(int i =1 ; i < m ;i++){
            ans[i][0] = ans[i-1][0] ^ matrix[i][0];
            l.push_back(ans[i][0]);
        }
        
        for(int i =1 ; i < n ;i++){
            ans[0][i] = ans[0][i-1] ^ matrix[0][i];
            l.push_back(ans[0][i]);
        }
        
        for(int i = 1 ; i < m ; i++){
            for(int j = 1; j < n ; j++){
                ans[i][j] = ans[i-1][j] ^ ans[i-1][j-1] ^ ans[i][j-1] ^ matrix[i][j];
                l.push_back(ans[i][j]);
            }
        }
        
        sort(l.begin(), l.end());
        return l[m*n-1 - (k-1)];
    }
};
