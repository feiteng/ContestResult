class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<int> v;
        vector<vector<int>> mat(m+1, vector<int>(n+1));
        for(int i = 1; i <= m; ++i) {
            for(int j = 1; j <= n; ++j) {
                mat[i][j] = matrix[i-1][j-1] ^ mat[i-1][j] ^ mat[i][j-1] ^ mat[i-1][j-1];
                v.emplace_back(mat[i][j]);
            }
        }
        sort(v.begin(), v.end());
        return v[v.size() - k];
    }
};
