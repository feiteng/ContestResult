class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m=matrix.size(), n=matrix[0].size();
        vector<vector<int>>mask(m, vector<int>(n, 0));
        vector<int>data;
        
        mask[0][0] = matrix[0][0];
        data.push_back(mask[0][0]);
        for (int i=1; i<m; i++) {
            mask[i][0] = mask[i-1][0]^matrix[i][0];
            data.push_back(mask[i][0]);
        }
        for (int j=1; j<n; j++) {
            mask[0][j] = mask[0][j-1]^matrix[0][j];
            data.push_back(mask[0][j]);
        }

        for (int i=1; i<m; i++)
            for (int j=1; j<n; j++) {
                mask[i][j] = mask[i-1][j]^mask[i][j-1]^mask[i-1][j-1]^matrix[i][j];
                data.push_back(mask[i][j]);
            }
        
     //   for (auto dd: data)
     //       cout << dd << ", ";
     //   cout << endl;
        sort(data.begin(), data.end());
        return data[data.size()-k];
        
    }
};
