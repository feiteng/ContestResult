class Solution {
public:
	int kthLargestValue(vector<vector<int>>& mat, int k) {
		int n = mat.size(), m = mat[0].size();
		vector<vector<int>>val(n + 1, vector<int>(m, 0));
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (j == 0)val[i + 1][j] = mat[i][j];
				else val[i + 1][j] = mat[i][j] ^ val[i+1][j - 1];
			}
		}

		vector<vector<int>>dp(n + 1, vector<int>(m, 0));
		dp[1][0] = mat[0][0];

		vector<int>all;
		for (int i = 1; i <= n; i++) {
			for (int j = 0; j < m; j++) {
				dp[i][j] = dp[i - 1][j] ^ val[i][j];
				all.push_back(dp[i][j]);
			}
		}
		sort(all.rbegin(), all.rend());
		return all[k - 1];
	}
};
