class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                matrix[i][j] ^= matrix[i][j - 1];
            }
        }
        for (int i = 1; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                matrix[i][j] ^= matrix[i - 1][j];
            }
        }
            
        vector<int> total(m * n);
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                total[i * n + j] = matrix[i][j];
            }
        }
        
        sort(total.begin(), total.end());
        return total[m * n - k];
    }
};
