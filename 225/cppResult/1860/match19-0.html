class Solution {
public:
    int kthLargestValue(vector<vector<int>>& a, int k) {
        int n = a.size(), m = a[0].size();
        vector<vector<int>> b(n + 1, vector<int>(m + 1));
        vector<int> ret;
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                b[i][j] = b[i - 1][j] ^ b[i][j - 1] ^ b[i - 1][j - 1] ^ a[i - 1][j - 1];
                ret.push_back(b[i][j]);
            }
        }
        sort(ret.begin(), ret.end(), greater<int>());
        return ret[k - 1];
    }
};
