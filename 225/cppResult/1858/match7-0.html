class Solution {
public:
    string maximumTime(string str) {
        if(str[0] == '?') {
            if(str[1] != '?' && str[1] >= '4') {
                str[0] = '1';
            }else {
                str[0] = '2';
            }
        }
        if(str[1] == '?') {
            if(str[0] == '2') {
                str[1] = '3';
            }else {
                str[1] = '9';
            }
        }
        
        if(str[3] == '?') {
            str[3] = '5';
        }
        if(str[4] == '?') {
            str[4] = '9';
        }
        return str;
    }
};
