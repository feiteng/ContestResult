class Solution {
public:
    string maximumTime(string time) {
        string ans = time;
        if (ans[0] == '?' && ans[1] == '?') {
            ans[0] = '2'; ans[1] = '3';
        } else if (ans[0] == '?') {
            if (ans[1] >= '4')
                ans[0] = '1';
            else
                ans[0] = '2';
        } else if (ans[1] == '?') {
            if (ans[0] == '0' || ans[0] == '1')
                ans[1] = '9';
            else
                ans[1] = '3';
        }
        if (ans[3] == '?' && ans[4] == '?') {
            ans[3] = '5'; ans[4] = '9';
        } else if (ans[3] == '?') {
            ans[3] = '5';
        } else if (ans[4] == '?') {
            ans[4] = '9';
        }
        return ans;
    }
};
