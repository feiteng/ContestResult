class Solution {
    public int minimumBoxes(int n) {
        int blocks = 1;
        int top = 2;
        int base = 1;
        while (blocks < n) {
            for (int i = 1; i <= top; i++) {
                base++;
                blocks+=i;
                if (blocks >= n) break;
            }
            top++;
        }
        return base;
    }
}
