class Solution {
    // Map<Integer, Integer> map = new TreeMap<>();
    public int minimumBoxes(int n) {
        int lo =0;
        int hi = n;
        while (lo<hi) {
            int mid = lo + (hi-lo)/2;
            if (count(mid) >= n) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }
    private long count(long base) {
        long res = base;
        while (base > 0) {
            base = nextLevel(base);
            res +=base;
        }
        // System.out.println(String.format("base = %d, res = %d",base, res));
        return res;
    } 
    private long nextLevel(long num) {
        // if (num<3) {return 0;}
        
long lo = 0;
        long hi = num;
        while (lo<hi) {
            long mid = lo + (hi-lo+1)/2;
            if (mid* (mid+1)/2 <= num) {
                lo = mid;
            } else {
                hi = mid-1;
            }
        }
        long count = lo * (lo-1)/2;
        count += Math.max(0, num - lo * (lo+1)/2 -1);
        // System.out.println(String.format("num = %d, count = %d, lo = %d",num, count, lo));
        return count;
        
    } 
}
