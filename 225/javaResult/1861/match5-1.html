class Solution {
    public int minimumBoxes(int n) {
        
int total  = 0;
        int add = 1;
        int layer = 1;
        int bot = 0;
        while(total < n){
            total += add;
            add++;
            bot++;
            if(add > layer){
                layer++;
                add = 1;
            }
        }
        return bot;
    }
}
