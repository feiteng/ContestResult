class Solution
{
    public int minimumBoxes(int n)
    {
        if(n<=3) return n;

        int block=3, level=2;
        int bottom=3, max=4;
        while(max<n)
        {
            bottom+=block;
            max+=bottom;
            block++;
            level++;
        
}

        while(max-level>=n)
        {
            max-=level--;
            bottom--;
        }
        return bottom;
    }
}
