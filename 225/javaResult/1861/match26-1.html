class Solution {
    public int minimumBoxes(int n) {
        int[] arr = new int[40000];
        
int sum = 0, idx = 1, base = 0;
        while (sum < n) {
            for (int i = 0; i < idx; i++) {
                sum += i+1;
                base++;
                if (sum >= n) {
                    return base;
                
}
            }
            idx++;
        }
        return -1;
    }
}
