class Solution {
    public int minimumBoxes(int n) {
        if (n <= 3) {
            return n;
        }
        int currBottom = 3;
        int currTotal = 4;
        int pace = 1;
        int maxPace = 3;
        while (currTotal < n) {
            
            currBottom += 1;
            currTotal += pace;
            pace += 1;
            if (pace > maxPace) {
                pace = 1;
                maxPace += 1;
            }
        }
        return currBottom;
