class Solution {
 public int minimumBoxes(int n) {
        if(n < 4)
            return n;
        int i = 0;
        int m = 0;
        int j = 1;
        while (i < n){
            m += j;
            i += m;
            j++;
            System.out.println(i + "," + m + "," + j);
        }
        j--;
        while (i >= n){
            m--;
            i -= j;
            j--;
        }
        return ++m;
    }
}
