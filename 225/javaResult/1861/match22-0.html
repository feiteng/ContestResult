class Solution {
    public int minimumBoxes(int n) {
        int begin = 1;
        int end = n;
        while (begin <= end) {
            int mid = begin + ((end - begin) >> 1);
            // System.out.println("mid:" + mid + "res: " + dfs(mid));
            if (dfs(mid) < n) {
                begin = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return begin;
    }
    
    public long dfs(int n) {
        if (n <= 2) return n;
        // if (n <= 4) return n - 1;
        
        long begin = 1;
        long end = 1000000;
        while (begin <= end) {
            long mid = begin + ((end - begin) >> 1);
            if (((mid * mid + mid) >> 1) < n) {
                begin = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        // System.out.println("begin" + begin);
        return n + dfs(n - (int)begin);
    }
}
