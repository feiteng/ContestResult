class Solution {
    public int minimumBoxes(int n) {
        
int total = 0;
        int layerCount = 0;
        int layer = 0;
        while (total < n){
            layer ++;
            layerCount += layer;
            total += layerCount;
        }
        
        while (total - layer >= n){
            total -= layer;
            layer --;
            layerCount--;
        }
        
        return layerCount;
    }
}
