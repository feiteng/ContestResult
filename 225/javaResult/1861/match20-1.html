class Solution {
    public int minimumBoxes(int n) {
int d = 1;
        int f = 1;
        int m = 1;

        while (n > m) {
            d++;
            f += d;
            m += f;
        }

        if (n == m) {
            return f;
        }

        int r = f;
        int l = f - d + 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            long x = mid - (f - d);
            long x2 = x * (x + 1) / 2;
            if (x2 + m - f >= n) {
                r = mid;
            } else 
{
                l = mid + 1;
            }
        }

        return l;
    }
}
