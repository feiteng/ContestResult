class Solution {
    public int minimumBoxes(int n) {
        //form the perfect shape as much as we can
        int level = 1;
        int total = 0;
        int floor = 0;
        while (total + (floor + level) < n) {
            floor += level;
            ++level;
            total += floor;
        }
        
        //place new boxes in most greedy way
        int[] levels = new int[level];
        int p = 0;
        while (total < n) {
            ++levels[p];
            if (levels[p] > 1) {
                ++p;
            
} else {
                p = 0;
            }
            ++total;
        }
        
        return floor + levels[0];
    }
}
