class Solution {
    int[] row = new int[100000];
    int[] level = new int[100000];
    int[] sum1 = new int[10000];
    public int minimumBoxes(int n) {
        row[0] = 1;
        for(int i = 1; i < 100000; i++)
            row[i] = i + 1;
        level[0] = row[0];
        for(int i = 1; i < 100000; i++)
            level[i] += level[i-1] + row[i];
        sum1[0] = level[0];
        for (int i = 1; i < 10000; i++) {
            sum1[i] += sum1[i-1] + level[i];
        }



        
int index = 0;
        for(int i = 0; i < 10000; i++){
            if(sum1[i] > n){
                index = i;
                break;
            }
        }
        int res = n - sum1[index-1];
        if(res == 0)
            return level[index-1];
        
int tmp = 1;
        int tmpIndex = 1;
        for(int i = 2; i <= index+2; i ++){
            if(tmp >= res){
                tmpIndex = i-1;
                break;
            }
            tmp += i;

        }
        return level[index-1] + tmpIndex;
    }
}
