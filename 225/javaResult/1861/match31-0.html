class Solution {
    public int minimumBoxes(int n) {
        long nn = n;
        long x = 0;
        for (long i = 0; i < 10000; i++) {
            if (i * (i + 1) * (i + 2) >= nn * 6) {
                x = i - 1;
                break;
            }
        }
        long y = nn - x * (x + 1) * (x + 2) / 6;
        long x1 = 0;
        for (long i = 0; i < 10000; i++) {
            if ((i + 1) * i / 2 >= y) {
                x1 = i;
                break;
            }
        }
        return (int) ((1 + x) * x / 2 + x1);
    }
}
