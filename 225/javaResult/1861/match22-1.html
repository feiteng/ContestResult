class Solution {
    public int minimumBoxes(int n) {
        if (n <= 3) return n;
        
long l = 3;
        long r = n;
        while (l < r) {
            long m = l + (r - l) / 2;
            if (helper(n, m)) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        // helper(10, 4);
        // return 11;
        return (int)l;
    }
    
    private boolean helper(long target, long floor) {
        if (target <= 0) return true;
        if (target > 0 && floor == 0) return false;
         // System.out.println("target = " + target + ", floor = " + floor);
        // System.out.println(floor);
        long total = 0;
        
long l = 1;
        long r = floor;
        while (l < r) {
             long m = l + (r - l + 1) / 2;
            // System.out.println("m = " + m);
            // System.out.println("(m + 1) * m / 2 = " + (m + 1) * m / 2);
            if ((m + 1) * m / 2 > floor) {
                r = m - 1;
            } else {
                l = m;
            }
        }
        
        // System.out.println("LEN = " +  l);
        if ((l + 1) * l / 2 < floor) {
            l = l + 1;
        }
        // System.out.println("LEN = " +  l);
        
        return helper(target - floor, floor - l);
        // long cnt = floor;
        // while (cnt > 0) {
        //     // System.out.println("CNT = " +  cnt);
        //     total += cnt;
        //     cnt -= l;
        //     l--;
        // }
        // // System.out.println(total >= target);
        // return total >= target;
    }
}
