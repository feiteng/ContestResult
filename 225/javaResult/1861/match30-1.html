class Solution {
    public int minimumBoxes(int n) {
        int level = 0;
        int usedBox = 0;
        int previousUsed = 0;
        while (usedBox < n) {
            previousUsed = usedBox;
            usedBox += getFloorNumber(level);
            level++;
        }
        if (usedBox == n) {
            return getFloorNumber(level - 1);
        }
        
        int gap = n - previousUsed;
        System.out.println("previousUsed is : " + previousUsed);
        System.out.println("gap is : " + gap);

        
        int base = getFloorNumber(level - 2);
        System.out.println("base is : " + base);
        
        
int current = 1;
        int count = 0;
        while (count < gap) {
            count += current;
            current++;
        }
        return base + current - 1;
    }
    
    private int getFloorNumber(int n) {
        return (n + 1) * n / 2; 
    }
}
