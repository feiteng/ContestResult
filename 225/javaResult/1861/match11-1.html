    public int minimumBoxes(int n) {
        int rowSum = 0;
        int i = 1;
        int totalSum = 0;
        while(totalSum < n){
            rowSum = rowSum + i;
            i++;
            totalSum += rowSum;
        }
        //if()
        
        //System.out.println("i:" + i);
        
int x = ((i-1)*i)/2;
        i--;
       // System.out.println("x:" + x);
       // System.out.println("totalSum:" + totalSum);
        while((totalSum-i) >= n){
            totalSum -= i;
            x--;
            i--;
        }
        return x;
