class Solution {
    
public int minimumBoxes(int n) {
        long i = 1;
        long left = n;
        while (i * (i + 1) * (i + 2) / 6 < left) {
            i++;
        }
        i--;
        left -= i * (i + 1) * (i + 2) / 6;

        long res = i * (i + 1) / 2;
        int sum = 0;
        
int index = 1;
        while (left > 0) {
            res++;
            left -= index;
                        index++;
        }

        return (int) res;
    }
}
