class Solution {
    public int minimumBoxes(int n) {
        int l = 1;
        int r = n;
        
        while(l <= r){
            int mid = l + ((r - l) >> 1);
            if(check(mid, n)){
                r = mid - 1;
            }
            else{
                l = mid + 1;
            }
        }
        
        return r + 1;
    }
    
    private boolean check(int init, int n){
        long layer = getLayer(init);
        long remain = (long) init - (((1 + layer) * layer) >> 1);
        
        // System.out.println(layer + " " + remain + " " + init);
        
        return checkHelper(layer, remain, n);
    }
    
    private boolean checkHelper(long layer, long remain, int n){
        if(n <= 0){
            return true;
        }
        
        if(layer == 0){
            return false;
        }
        
        n -= ((1 + layer) * layer) >> 1;
        if(remain > 0){
            n -= remain;
        }
        
        
        return checkHelper(layer - 1, remain - 1, n);
    }
    
    private long getLayer(int init){
        
long l = 1;
        long r = init;
        
        while(l <= r){
            long mid = l + ((r - l) >> 1);
            if(mid * mid + mid <= (init << 1)){
                l = mid + 1;
            }
            else{
                r = mid - 1;
            }
        }
        
        return l - 1;
    }
}
