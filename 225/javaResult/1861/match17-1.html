class Solution {
    public int minimumBoxes(int n) {
        int max = 1;
        
int res = 1;
        int tmp = 2;
        int idx = 1;
        while(max < n) {
            res++;
            max += idx;
            //System.out.println(max + " " + res);
            idx++;
            if(idx > tmp) {
                tmp++;
                idx = 1;
            }
        }
        return res;
    }
}
