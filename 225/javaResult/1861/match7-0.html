class Solution {

    public static void main(String[] args) {
        for (int i = 0; i <= 20; i++) {
            System.out.println(i + ":" + new Solution().minimumBoxes(i));
        }
    }
    public int minimumBoxes(int n) {
        int ans = 0;
        int a = 1;
        while (n > (a * (a + 1) / 2)) {
            n -= (a * (a + 1) / 2);
            ans += a;
            a++;
        }
        if (n == 0) return a;
        int i = 1;
        while (n > 0) {
            ans++;
            n -= i;
            i++;
        }
        return ans;
    }
}
