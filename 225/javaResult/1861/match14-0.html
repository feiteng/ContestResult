class Solution {
    public int minimumBoxes(int n) {
        if (n <= 3) return n;
        int top = 1;
        int level = 1;
        int pretop = 0;
        while (n >= top) {
            n -= top;
            pretop = top;
            top += level + 1;
            level ++;
        }
        int x = 0;
        while (n > x) {
            n -= x;
            x++;
        }
        return pretop + x;
    }
}
