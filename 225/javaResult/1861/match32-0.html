class Solution {
    public int minimumBoxes(int n) {
        if (n < 3)
            return n;
        
        int total = 0;
        int index = 0;
        
        while (total < n){
            index++;
            total += index*(index + 1)/2;
        }
        
        int valAtBase = index*(index + 1)/2;
        
        
        while (total - index >= n){
            total -= index;
            valAtBase--;
            --index;
        }
        
        return valAtBase;
    }
}
