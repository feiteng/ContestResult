class Solution {
    public int minimumBoxes(int nint) {
        long n = nint;
        long x = 1, s = 1;
        while (s + (x + 2) * (x + 1) / 2 <= n) { ++x; s += (1 + x) * x / 2; }
        //System.out.printf("x=%d s=%d\n", x, s);
        
        int ans = (int)((1 + x) * x / 2);
        n -= s;
        long y = 0;
        while ((1 + y) * y / 2 < n) { ++y; }
        
        return (int)(ans + y);
    }
}
