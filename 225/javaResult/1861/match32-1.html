class Solution {
    public int minimumBoxes(int n) {
        if (n <= 3) return n;
        
        int sum = 1;
        int area = 1;
        int interval = 2;
        while (sum < n) {
            area += interval;
            interval++;
            if (sum + area >= n) {
                if (sum + area > n) interval--;
                break;
            }
            sum += area;
        }
        
        interval--;
        int res = (interval + 1) * interval / 2;
        if (sum + area == n) return res;
        
        int rest = n - sum;
        sum = 0;
        
int k = 1;
        while (sum < rest) {
            sum += k;
            k++;
            res++;
        }
        
        return res;
    }
}
