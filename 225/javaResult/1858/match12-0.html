class Solution {
    public String maximumTime(String time) {
        char[] cur = time.toCharArray();
        
        if (cur[0] == '?') {
            if (cur[1] != '?' && cur[1] >= '4') {
                cur[0] = '1';
            } else {
                cur[0] = '2';
            }
        }
        if (cur[1] == '?') {
           if (cur[0] == '2') {
                cur[1] = '3';
            } else {
                cur[1] = '9';
            } 
        }
        if (cur[3] == '?') {
            cur[3] = '5';
        }
        if (cur[4] == '?') {
            cur[4] = '9';
        }
        return new String(cur);        
    }
}
