class Solution {
    public String maximumTime(String time) {
        String ans = "";
        if(time.charAt(0)=='?'){
            if(time.charAt(1)>='4'&&time.charAt(1)!='?'){
                ans+='1';
            }
            else{
                ans+='2';
            }
        }
        else{
            ans+=time.charAt(0);
        }
        if(time.charAt(1)=='?'){
            if(
ans.charAt(0)=='2'){
                ans+='3';
            }
            else{
                ans+='9';
            }
        }
        else{
            ans+=time.charAt(1);
        }
        ans+=':';
        if(time.charAt(3)=='?'){
            ans+='5';
        }else{
            ans+=time.charAt(3);
        }
        if(time.charAt(4)=='?'){
            ans+='9';
        }else{
            ans+=time.charAt(4);
        }
        return ans;
    }
}
