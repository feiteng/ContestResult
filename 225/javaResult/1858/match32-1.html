    public String maximumTime(String time) {
        char[] charr = time.toCharArray();
        if (charr[0] == '?') {
            if (charr[1] == '?') {
                charr[0] = '2';
                charr[1] = '3';
            } else if (charr[1] < '4') 
{
                charr[0] = '2';
            } else {
                charr[0] = '1';
            }
        } 
        if (charr[1] == '?') {
            if (charr[0] == '2') {
                charr[1] = '3';
            } else {
                charr[1] = '9';
            }
        }
        if (charr[3] == '?') {
            charr[3] = '5';
        }
        if (charr[4] == '?') {
            charr[4] = '9';
        }
        
        return new String(charr);
    }
