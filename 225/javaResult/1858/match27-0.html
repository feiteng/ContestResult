class Solution {
    public String maximumTime(String time) {
        char[] s = time.toCharArray();
        if (s[0] == '?') {
            s[0] = s[1] == '?' || s[1] < '4' ? '2' : '1';
        }
        if (s[1] == '?') {
            s[1] = s[0] == '2' ? '3' : '9';
        }
        if (s[3] == '?') {
            s[3] = '5';
        }
        if (s[4] == '?') {
            s[4] = '9';
        }
        return new String(s);
    }
}
