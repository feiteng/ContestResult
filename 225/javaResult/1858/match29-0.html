class Solution {
    public String maximumTime(String time) {
        char[] str = time.toCharArray();
        if (str[0] == '?') {
            str[0] = str[1] >='0' && str[1] <= '3' || str[1] == '?' ? '2' : '1';
        }
        if (str[1] == '?') {
            str[1] = str[0] == '2' ? '3' : '9';
        }
        if (str[3] == '?') {
            str[3] = '5';
        }
        if (str[4] == '?') {
            str[4] = '9';
        }
        return new String(str);
    }
}
