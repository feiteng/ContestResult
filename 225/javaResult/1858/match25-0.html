class Solution {
    public String maximumTime(String time) {
        char[] timeChar = time.toCharArray();
        
        if (timeChar[0] == '?') {
            if (timeChar[1] != '?') {
                if (timeChar[1] >= '4') {
                    timeChar[0] = '1';
                } else {
                    timeChar[0] = '2';
                }
            } else {
                timeChar[0] = '2';
            }
        }
        if (timeChar[1] == '?') {
            if (timeChar[0] != '2') {
                timeChar[1] = '9';
            } else {
                timeChar[1] = '3';
            }
        }
        
        if (timeChar[3] == '?') {
            timeChar[3] = '5';
        }
        if (timeChar[4] == '?') {
            timeChar[4] = '9';
        }
        
        return new String(timeChar);
    }
}
