class Solution {
    public String maximumTime(String time) {
        String res = "";
        if(time.charAt(0) == '?'){
            if(time.charAt(1) == '?' || time.charAt(1) <= '3'){
                res += "2";
            }else{
                res += "1";
            }
        }else{
            res += time.charAt(0);
        }
        if(time.charAt(1) == '?'){
            if(time.charAt(0) == '?' || time.charAt(0) == '2'){
                res += "3";
            }else{
                res += "9";
            }
        }else{
            res += time.charAt(1);
        }
        res += ":";
        if(time.charAt(3) == '?'){
            res += "5";
        }else{
            res += time.charAt(3);
        }
        if(time.charAt(4) == '?'){
            res += "9";
        }else{
            res += time.charAt(4);
        }
        return res;
    }
}
