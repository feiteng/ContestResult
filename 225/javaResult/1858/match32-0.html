class Solution {
    public String maximumTime(String time) {
        char[] ca = time.toCharArray();
        if (ca[0] == '?') {
            if (ca[1] == '?') {
                ca[0] = '2';
                ca[1] = '3';
            } else if (Character.getNumericValue(ca[1]) < 4) {
                ca[0] = '2';
            } else {
                ca[0] = '1';
            }
        }
        
        if (ca[1] == '?') {
            if (ca[0] == '2') {
                ca[1] = '3';
            } else {
                ca[1] = '9';
            }
        }
        
        if (ca[3] == '?') {
            ca[3] = '5';
        }
        
        if (ca[4] == '?') {
            ca[4] = '9';
        }
        
        return new String(ca);
    }
}
