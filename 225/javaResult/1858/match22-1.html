class Solution {
    public String maximumTime(String time) {
        StringBuilder res = new StringBuilder(time);
        if (res.charAt(0) == '?') {
            if (res.charAt(1) == '?' || res.charAt(1) <= '3') {
                res.setCharAt(0, '2');
            } else {
                res.setCharAt(0, '1');
            }
        }
        if (res.charAt(1) == '?') {
            if (res.charAt(0) == '2') {
                res.setCharAt(1, '3');
            } else {
                res.setCharAt(1, '9');
            }
        }
        if (res.charAt(3) == '?') {
            res.setCharAt(3, '5');
        }
        if (res.charAt(4) == '?') {
            res.setCharAt(4, '9');
        }
        return res.toString();

    }
}
