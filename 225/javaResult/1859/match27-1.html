class Solution {
  public int minCharacters(String a, String b) {
        int ans = makeLess(a, b);
        ans = Math.min(ans, makeLess(b, a));
        ans = 
Math.min(ans, makeSingle(a) + makeSingle(b));
        return ans;
    }
    
    int makeLess(String a, String b) {
        int ans = a.length() + b.length();
        for (int c = 'a'; c <= 'z' - 1; c++) {
            int count = 0;
            for (int i = 0; i < a.length(); i++) {
                if (a.charAt(i) > c) {
                    count++;
                }
            }
            
            for (int i = 0; i < b.length(); i++) {
                if (b.charAt(i) <= c) {
                    count++;
                }
            }
            ans = Math.min(ans, count);
        }
        return ans;
    }
    
    int makeSingle(String a) {
        int[] count = new int[26];
        int maxCount = 0;
        for (int i = 0; i < a.length(); i++) {
            count[a.charAt(i) - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            maxCount = Math.max(maxCount, count[i]);
        }
        return a.length() - maxCount;
    }
}
