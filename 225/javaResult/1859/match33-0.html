class Solution {
    public int minCharacters(String a, String b) {
        int[] countA = new int[26];
        int[] countB = new int[26];
        for (char c : a.toCharArray()) {
            countA[c - 'a']++;
        }

        for (char c : b.toCharArray()) {
            countB[c - 'a']++;
        }

        int[] sumA = new int[26];
        sumA[25] = countA[25];
        for (int i = 24; i >= 0; i--) {
            sumA[i] = sumA[i + 1] + countA[i];
        }

        int[] sumB = new int[26];
        sumB[25] = countB[25];
        for (int i = 24; i >= 0; i--) {
            sumB[i] = sumB[i + 1] + countB[i];
        }

        int ans = a.length() + b.length();
        int mA = 0;
        int mB = 0;
        int cA = 0;
        int cB = 0;
        for (int i = 0; i < 26; i++) {
            mA = Math.max(mA, countA[i]);
            mB = Math.max(mB, countB[i]);
            cA += countA[i];
            cB += countB[i];
            if (i < 25) {
                ans = Math.min(ans, cB + sumA[i + 1]);
                ans = Math.min(ans, cA + sumB[i + 1]);
            }
        }

        ans = Math.min(ans, a.length() + b.length() - mA - mB);
        return ans;
    }
}
