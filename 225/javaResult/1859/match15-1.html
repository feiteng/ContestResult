class Solution {

    public int minCharacters(String a, String b) {
        
int[] countA = new int[26];
        int[] countB = new int[26];
        int[] countSum = new int[26];
        for (char aa : a.toCharArray()) {
            countA[aa - 'a']++;
            countSum[aa - 'a']++;
        }
        for (char aa : b.toCharArray()) {
            countB[aa - 'a']++;
            countSum[aa - 'a']++;
        }

        
int max = 0;
        for (int i = 0; i < 26; i++) {
            max = Math.max(max, countSum[i]);
        }
        int res = a.length() + b.length() - max;
        for (int i = 0; i < 25; i++) {
            int curA = 0;
            int curB = 0;
            for (int j = 0; j <= i; j++) {
                curA += countA[j];
                curB += countB[j];
            }
            
res = Math.min(res, a.length() - curA + curB);
            res = Math.min(res, b.length() - curB + curA);
        }
        return res;
    }

}
