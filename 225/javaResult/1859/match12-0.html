class Solution {
    public int minCharacters(String a, String b) {
        int[] freq1 = new int[26];
        int[] freq2 = new int[26];
        for (char c : a.toCharArray())
            ++freq1[c-'a'];
        for (char c : b.toCharArray())
            ++freq2[c-'a'];
        
        int min = Math.min(find(freq1, freq2), find(freq2, freq1));
        min = Math.min(min, c3(freq1, freq2));
        return min;
    }
    
    private int find(int[] f1, int[] f2) {
        int min = Integer.MAX_VALUE;
        for (int i = 1 ; i < 26 ; ++i) {
            int sum = 0;
            for (int j = i ; j < 26 ; ++j)
                sum += f1[j];
            for (int j = 0 ; j < i ; ++j)
                sum += f2[j];
            min = Math.min(min, sum);
        }
        return min;
    }
    
    private int c3(int[] f1, int[] f2) {
        int min = Integer.MAX_VALUE;
        for (int i = 0 ; i < 26 ; ++i) {
            int sum = 0;
            for (int j = 0 ; j < 26 ; ++j) {
                if (i == j) continue;
                sum += f1[j] + f2[j];
            }
            min = Math.min(min, sum);
        }
        return min;
    }
}
