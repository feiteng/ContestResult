class Solution {
    public int minCharacters(String a, String b) {
        int[] fna = new int[26];
        int[] fnb = new int[26];
        for (char ch : a.toCharArray()) {
            fna[ch - 'a'] += 1;
        }
        for (char ch : b.toCharArray()) {
            fnb[ch - 'a'] += 1;
        }
        int lena = a.length();
        int lenb = b.length();
        int ans = Integer.MAX_VALUE;
        int olda = 0;
        int oldb = 0;
        for (int x=0;x<26;x++) {
            ans = Math.min(ans, lena+lenb - fna[x] - fnb[x]);
            if (x != 0) {
                ans = Math.min(ans, lena-olda + oldb);
                ans = 
Math.min(ans, lenb-oldb + olda);
            }
            olda += fna[x];
            oldb += fnb[x];
        }
        return ans;
    }
}
