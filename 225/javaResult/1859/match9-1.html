class Solution {
    
	
	public int minCharacters(String a, String b) {
        int c1 = Integer.MAX_VALUE;
        for (char c = 'a'; c < 'z'; c ++ ) {
        	int case1 = 0;
        	for (int i = 0; i < a.length(); i ++ ) {
        		if (a.charAt(i) > c) case1 ++;

        	}
        	for (int i = 0; i < b.length(); i ++ ) {
        		if (b.charAt(i) <= c) case1 ++;
        	}
        	c1 = Math.min(c1, case1);
        }
        int c2 = Integer.MAX_VALUE;
        
for (char c = 'a'; c < 'z'; c ++ ) {
        	int case2 = 0;
        	for (int i = 0; i < b.length(); i ++ ) {
        		if (b.charAt(i) > c) case2 ++;

        	}
        	for (int i = 0; i < a.length(); i ++ ) {
        		if (a.charAt(i) <= c) case2 ++;
        	}
        	c2 = Math.min(c2, case2);
        }
        
        int c3 = Integer.MAX_VALUE;
        
for (char c = 'a'; c <= 'z'; c ++ ) {
        	int case3 = 0;
        	for (int i = 0; i < a.length(); i ++ ) {
        		if (a.charAt(i) != c) case3 ++;

        	}
        	for (int i = 0; i < b.length(); i ++ ) {
        		if (b.charAt(i) != c) case3 ++;
        	}
        	c3 = Math.min(c3, case3);
        }
        return Math.min(c1, Math.min(c2, c3));
    }
}
