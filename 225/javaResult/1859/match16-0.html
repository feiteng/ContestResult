class Solution {

	public int minCharacters(String a, String b) {
		int count1[] = new int[26], count2[] = new int[26], min = Integer.MAX_VALUE, min1 = Integer.MAX_VALUE,
				min2 = Integer.MAX_VALUE;
		for (char c : a.toCharArray()) {
			count1[c - 'a']++;
		}
		for (char c : b.toCharArray()) {
			count2[c - 'a']++;
		}
		for (int i = 0; i < 25; i++) {
			int sum1 = 0, sum2 = 0;
			for (int j = 0; j <= i; j++) {
				sum1 += count1[j];
				sum2 += count2[j];
			}
			min = Math.min(min, a.length() - sum1 + sum2);
			min = Math.min(min, b.length() - sum2 + sum1);
		}
		for (int i : count1) {
			min1 = Math.min(min1, a.length() - i);
		}
		for (int i : count2) {
			min2 = Math.min(min2, b.length() - i);
		}
		return Math.min(min, min1 + min2);
	}
}
