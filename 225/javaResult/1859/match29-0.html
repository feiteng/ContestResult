class Solution {
  public int minCharacters(String a, String b) {
    int n_a = a.length(), n_b = b.length();
    int[] cnts = new int[26];
    int[] cnts1 = new int[27];
    int[] cnts2 = new int[27];
    for (char ch : a.toCharArray()) {
      cnts1[ch - 'a' + 1]++;
      cnts[ch - 'a']++;
    }
    for (char ch : b.toCharArray()) {
      cnts2[ch - 'a' + 1]++;
      cnts[ch - 'a']++;
    }
    for (int k = 0; k < 26; ++k) {
      cnts1[k + 1] += cnts1[k];
      cnts2[k + 1] += cnts2[k];
    }

    int ans = n_a + n_b;
    for (int k = 0; k < 25; ++k) {
      ans = Math.min(ans, (cnts1[k + 1] - cnts1[0] + cnts2[26] - cnts2[k + 1]));
      ans = Math.min(ans, (cnts2[k + 1] - cnts2[0] + cnts1[26] - cnts1[k + 1]));
    }
    for (int k = 0; k < 26; ++k) {
      ans = Math.min(ans, n_a + n_b - cnts[k]);
    }
    return ans;
  }
}
