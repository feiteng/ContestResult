class Solution {
    public int minCharacters(String a, String b) {
        int cha[] = new int[26];
        int chb[] = new int[26];
        for (int i = 0; i < a.length(); i++) {
            cha[a.charAt(i) - 'a']++;
        }
        for (int i = 0; i < b.length(); i++) {
            chb[b.charAt(i) - 'a']++;
        }
        int res = getCount(cha, chb);

        int[] prea = new int[27];
        int[] preb = new int[27];
        for (int i = 1; i < 27; i++) {
            prea[i] = prea[i - 1] + cha[i - 1];
            preb[i] = preb[i - 1] + chb[i - 1];
        }

        int min = Math.min(getCount1(prea, preb), getCount1(preb, prea));

        return Math.min(res, min);
    }

    int getCount(int[] cha, int[] chb) {
        int sum = 0;
        int max = 0;
        for (int i = 0; i < 26; i++) {
            max = Math.max(cha[i] + chb[i], max);
            sum += cha[i] + chb[i];
        }
        return sum - max;
    }

    int getCount1(int[] prea, int[] preb) {
        int res = Integer.MAX_VALUE;
        //把a全部变为z,把b全部变为小于a的
        for (int i = 25; i >= 1; i--) {
            int temp = 0;
            temp += prea[i];
            temp += preb[26] - preb[i];
            res = Math.min(temp, res);
        }

        return res;
    }
}
