class Solution {
    public int minCharacters(String a, String b) {
        int[] arra = new int[26];
        int[] arrb = new int[26];
        int maxa = 0;
        int maxb = 0;
        for(int i = 0; i<a.length(); i++){
            arra[a.charAt(i)-'a']++;
            maxa = Math.max(maxa, arra[a.charAt(i)-'a']);
        }
        for(int i = 0; i<b.length(); i++){
            arrb[b.charAt(i)-'a']++;
            maxb = Math.max(maxb, arrb[b.charAt(i)-'a']);
        }
        int thirdop = a.length()-maxa + b.length()-maxb;
        int[] preva = new int[26];
        int[] prevb = new int[26];
        preva[0] = arra[0];
        prevb[0] = arrb[0];
        for(int i = 1; i<26; i++){
            preva[i] = preva[i-1] + arra[i];
            prevb[i] = prevb[i-1] + arrb[i];
        }
        int firstop = Integer.MAX_VALUE;
        int secondop = Integer.MAX_VALUE;
        for(int i = 1; i<26; i++){
            firstop = Math.min(firstop, preva[i-1] + prevb[25] - prevb[i-1]);
            secondop = Math.min(secondop, prevb[i-1] + preva[25] - preva[i-1]);
        }
        return Math.min(firstop, Math.min(secondop, thirdop));
    }
}
