import java.util.*;


class Solution {
    public int minCharacters(String a, String b) {
        int minOps = Integer.MAX_VALUE;
        int m = a.length(), n = b.length();
        
        /*
        Make every character in a less than every character in b
         */
        for (int i = 0; i < 25; i++) {
            char targetChar = (char) (i + 'a');
            int numOperations = 0;
            for (int j = 0; j < m; j++) 
{
                if (a.charAt(j) > targetChar) {
                    numOperations++;
                }
            }
            for (int j = 0; j < n; j++) 
{
                if (b.charAt(j) <= targetChar) {
                    numOperations++;
                }
            }
            minOps = Math.min(minOps, numOperations);
        }
        
        /*
        Make every character in b less than every character in a
         */
        for (int i = 0; i < 25; i++) {
            char targetChar = (char) (i + 'a');
            int numOperations = 0;
            for (int j = 0; j < n; j++) 
{
                if (b.charAt(j) > targetChar) {
                    numOperations++;
                }
            }
            for (int j = 0; j < m; j++) 
{
                if (a.charAt(j) <= targetChar) {
                    numOperations++;
                }
            }
            minOps = Math.min(minOps, numOperations);
        }       
        
        /*
        Make every character in a equal to b
         */
        for (int i = 0; i < 26; i++) {
            char targetChar = (char) (i + 'a');
            int numOperations = 0;
            for (int j = 0; j < m; j++) 
{
                if (a.charAt(j) != targetChar) {
                    numOperations++;
                }
            }
            for (int j = 0; j < n; j++) 
{
                if (b.charAt(j) != targetChar) {
                    numOperations++;
                }
            }
            minOps = Math.min(minOps, numOperations);
        }
        
        return minOps;
        
    }
}
