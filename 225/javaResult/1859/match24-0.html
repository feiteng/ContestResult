class Solution {
    
    private int getPrefixSum(int[] arr , int l , int r){
        if(l == 0){
            return arr[r];
        }
        return arr[r] - arr[l-1];
    }
       
    public int minCharacters(String a, String b) {
        int[] ca = new int[26];
        int[] cb = new int[26];
        
        for(char c : a.toCharArray()){
            ca[c -'a']++;
        }
        
        for(char c : b.toCharArray()){
            cb[c -'a']++;
        }
        
        int max = 0;
        for(int i = 0 ; i < 26 ; i++){
            max = Math.max(max , ca[i] + cb[i]);
        }
        
        int total = a.length() + b.length() ;
        int ans = total - max;
        int ops = 0;
        
        int[] pre_a = new int[26];
        int[] pre_b = new int[26];
        
        pre_a[0] = ca[0];
        pre_b[0] = cb[0];
        
        for(int i = 1 ; i < 26 ; i++){
             pre_a[i] = ca[i] + pre_a[i-1];
             pre_b[i] = cb[i] + pre_b[i-1];
        }
        
        //condtion 1
        for(int i = 0 ; i < 25 ; i++){
            int counta = getPrefixSum(pre_a , 0 , i);
            int countb = pre_b[25] - getPrefixSum(pre_b , 0 , i);
            
            ans = Math.min(ans , total - counta - countb);
        }
    
        //condition 2
        for(int i = 0 ; i < 25 ; i++){
            int countb = getPrefixSum(pre_b , 0 , i);
            int counta = pre_a[25] - getPrefixSum(pre_a , 0 , i);
            
            ans = Math.min(ans , total - counta - countb);
        }
        
        
        return ans;
    }
}
