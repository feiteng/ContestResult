class Solution {
    public int minCharacters(String a, String b) {
        int[] c1 = new int[26], c2 = new int[26];
        for (char c: a.toCharArray()) {
            c1[c-'a']++;
        }
        for (char c: b.toCharArray()) {
            c2[c-'a']++;
        }
        return Math.min(o1(c1,c2),
o3(c1,c2));
    }
    private int o1(int[] c1, int[] c2) {
        int s1=0,s2=0;
        for (int i=0; i<26; i++) {
            s1+=c1[i];
            s2+=c2[i];
        }
        int p1=0,p2=0,min = Integer.MAX_VALUE;
        for (int i=0; i<25; i++) {
            p1+=c1[i];
            p2+=c2[i];
            min = Math.min(min,
Math.min(p1+s2-p2,p2+s1-p1));
        }
        return min;
    }
    private int o3(int[] c1, int[] c2) {
        
int sum = 0, highest = 0;
        for (int i=0; i<26; i++) {
            int curr = c1[i]+c2[i];
            sum += curr;
            highest = Math.max(highest,curr);
        }
        return sum-highest;
    }
}
