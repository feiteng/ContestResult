class Solution {
    public int minCharacters(String a, String b) {
        int[] ac = new int[128];
        int[] bc = new int[128];
        for (char c : a.toCharArray()) {
            ac[c]++;
        }
        for (char c : b.toCharArray()) {
            bc[c]++;
        }
        int min = a.length() + b.length();
        int la = 0;
        int lb = 0;
        for (int i = 'a'; i <= 'z'; i++) {
            la += ac[i];
            lb += bc[i];
            int cnt1 = i == 'z' ? min : a.length() - la + lb;
            int cnt2 = i == 'z' ? min : b.length() - lb + la;
            int cnt3 = 
a.length() - ac[i] + b.length() - bc[i];
            // System.out.println((char)i + " " + cnt1 +  " " + cnt2 + " " + cnt3);
            min = Math.min(Math.min(cnt1, cnt2), Math.min(cnt3, min));
        }
        return min;
    }
}
