class Solution {
    public int minCharacters(String a, String b) {
        int ans = op1(a, b);
        ans = Math.min(ans, op2(a,b));
        ans = Math.min(ans, op3(a, b));
        return ans;
    }
    
int op1(String a, String b)
    {
        int ans = 1000000;
        for(char c = 'a'; c < 'z'; c++)
        {
            int tmp = 0;
            for(int i = 0; i < b.length(); i++)
            {
                if (b.charAt(i) > c)
                {
                    tmp++;
                }
            }
            for(int i = 0; i < a.length(); i++)
            {
                if (a.charAt(i) <= c)
                {
                    tmp++;
                }
            }
            ans = Math.min(ans, tmp);
        }
        return ans;
    }
    int op2(String a, String b)
    {
        int ans = 1000000;
        
for(char c = 'a'; c < 'z'; c++)
        {
            int tmp = 0;
            for(int i = 0; i < a.length(); i++)
            {
                if (a.charAt(i) > c)
                {
                    tmp++;
                }
            }
            for(int i = 0; i < b.length(); i++)
            {
                if (b.charAt(i) <= c)
                {
                    tmp++;
                }
            }
            ans = Math.min(ans, tmp);
        }
        return ans;
    }
    int op3(String a, String b)
    {
        int ans = 1000000;
        
for(char c = 'a'; c <= 'z'; c++)
        {
            int tmp = 0;
            for(int i = 0; i < a.length(); i++)
            {
                if (a.charAt(i) != c)
                {
                    tmp++;
                }
            }
            for(int i = 0; i < b.length(); i++)
            {
                if (b.charAt(i) != c)
                {
                    tmp++;
                }
            }
            ans = Math.min(ans, tmp);
        }
        return ans;
    }
}
