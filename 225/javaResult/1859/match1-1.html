class Solution {
    public int minCharacters(String a, String b) {
        //count
        int[] count1 = new int[26];
        int[] count2 = new int[26];
        for (char c : a.toCharArray()) {
            ++count1[c - 'a'];
        }
        
        for (char c : b.toCharArray()) {
            ++count2[c - 'a'];
        }
        
        
int min = Integer.MAX_VALUE;
        
        //try each way
        
        //let a < b
        for (int i = 0; i < 25; ++i) {
            //current max char is i for a, current min char is i + 1 for b
            int move = 0;
            for (int j = i + 1; j < 26; ++j) {
                move += count1[j];
            }
            for (int j = 0; j <= i; ++j) {
                move += count2[j];
            }
            min = Math.min(min, move);
        }
        
        //let a > b
        for (int i = 0; i < 25; ++i) {
            //current max char is i for b, current min char is i + 1 for a
            int move = 0;
            for (int j = i + 1; j < 26; ++j) {
                move += count2[j];
            }
            for (int j = 0; j <= i; ++j) {
                move += count1[j];
            }
            min = Math.min(min, move);
        }
        
        //let a === b
        for (int i = 0; i < 26; ++i) {
            //let all char be i
            int move = 0;
            for (int j = 0; j < 26; ++j) {
                if (i != j) {
                    move += count1[j] + count2[j];
                
}
            }
            min = Math.min(min, move);
        }
        return min;
    }
}
