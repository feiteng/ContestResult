class Solution {
    
public int minCharacters(String a, String b) {
        int[] aCount = new int[26];
        int[] bCount = new int[26];
        for(int i = 0; i < a.length(); i++){
            char c = a.charAt(i);
            int index = (int)(c-'a');
            aCount[index]++;
        }
        for(int i = 0; i < b.length(); i++){
            char c = b.charAt(i);
            int index = (int)(c-'a');
            bCount[index]++;
        
}
        
        int max = Integer.MIN_VALUE;
        for(int i = 0; i < 26; i++){
            max = Math.max(max, aCount[i]+bCount[i]);
        }
        int min = a.length() + b.length() - max;
        
int[] aSum = new int[26];
        int[] bSum = new int[26];
        aSum[0] = aCount[0];
        bSum[0] = bCount[0];
        for(int i = 1; i < 26; i++){
            aSum[i] = aCount[i]+aSum[i-1];
            bSum[i] = bCount[i]+bSum[i-1];
        }
        for(int i = 0; i < 25; i++){
            int aUnder = aSum[i];
            int aOver = a.length() - aUnder;
            int bUnder = bSum[i];
            int bOver = b.length() - bUnder;
            min = 
Math.min(min, aUnder + bOver);
            min = Math.min(min, bUnder + aOver);
        }
        return min;
    }
}
