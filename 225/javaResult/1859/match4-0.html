class Solution {
	public static void main(String[] args) {
		System.out.println(new Solution().minCharacters("dee", "a"));
	}
	public int minCharacters(String a, String b) {
		int[] count1 = new int[26];
		int[] count2 = new int[26];
		for (char c : a.toCharArray()) {
			count1[c - 'a']++;
		}
		for (char c : b.toCharArray()) {
			count2[c - 'a']++;
		}
		int ans = Math.min(cal1(count1, count2), cal1(count2, count1));
		ans = Math.min(ans, cal2(count1, count2));
		return ans;
	}

	int cal1(int[] count1, int[] count2) {
		int ans = Integer.MAX_VALUE;
		for (int i=0; i<25; i++) {
			int x = 0;
			for (int j=i+1; j<26; j++) {
				x += count1[j];
			}
			for (int j=i; j>=0; j--) {
				x += count2[j];
			}
			ans = Math.min(ans, x);
		}
		return ans;
	}
	
	int cal2(int[] count1, int[] count2) {
		int ans = Integer.MAX_VALUE;
		for (int i=0; i<26; i++) {
			int x = 0;
			for (int j=0; j<26; j++) {
				if (j != i) {
					x += count1[j] + count2[j];
				}
			}
			ans = Math.min(ans, x);
		}
		return ans;
	}
}
