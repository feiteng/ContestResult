class Solution {
    public int minCharacters(String a, String b) {
        int[] aa = new int[26];
        int[] bb = new int[26];
        for(char c: a.toCharArray()) {
            aa[c-'a'] ++;
        }
        for(char c: b.toCharArray()) {
            bb[c-'a'] ++;
        }
        int[] as = new int[26];
        int[] bs = new int[26];
        as[0] = aa[0];
        bs[0] = bb[0];
        for(int i=1;i<26;i++) {
            as[i] = as[i-1] + aa[i];
            bs[i] = bs[i-1] + bb[i];
        }
        int ans = 1000000;
        for(int i=0;i<26;i++) {
            if (i<25) {
                ans = Math.min(ans, bs[i] + a.length() - as[i]);
                ans = Math.min(ans, as[i] + b.length() - bs[i]);
            }
            ans = Math.min(ans, b.length()-bb[i] + a.length()-aa[i]);
        }
        return ans;
    }
}
