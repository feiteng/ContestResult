class Solution {
    public int minCharacters(String a, String b) {
        int[] num1 = new int[26];
        int[] num2 = new int[26];
        helper(num1, a);
        helper(num2, b);
        int res = Integer.MAX_VALUE;
        // make a less than b
        for (int i = 1; i < 26; i++) { // make all in num2 >= i
            int sum = 0;
            for (int k = 0; k < i; k++) {
                sum += num2[k];
            }
            for (int j = i; j < 26; j++) {
                sum += num1[j];
            }
            res = Math.min(res, sum);
        }
        for (int i = 1; i < 26; i++) { // make all in num1 >= i
            int sum = 0;
            for (int k = 0; k < i; k++) {
                sum += num1[k];
            }
            for (int j = i; j < 26; j++) {
                sum += num2[j];
            }
            res = Math.min(res, sum);
        }
        for (int i = 0; i < 26; i++) {
            int sum = 0;
            for (int j = 0; j < 26; j++) {
                if (i == j) continue;
                sum += num1[j] + num2[j];
            }
            res = Math.min(res, sum);
        }
        return res;
    }
    
    public void helper(int[] num1, String s) {
        for (char c : s.toCharArray()) num1[c - 'a']++;
    }
}
