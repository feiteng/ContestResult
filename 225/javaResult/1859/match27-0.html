class Solution {
    public int minCharacters(String a, String b) {
        // two string;
        // a consist 
        int min = minOpOneLetter(a, b);
        min = Math.min(min, Math.min(minOpStrictLess(a, b), minOpStrictLess(b, a)));
        return min;
    }
    
    
    public int minOpStrictLess(String a, String b){
        // guess the min letter in b 
        int min = a.length() + b.length();
        for (int i = 'b'; i <= 'z'; i++){
            int count = 0;

            for (int j = 0; j < a.length(); j++){
                if (a.charAt(j) >= i) {
                    count++;
                }
            }
            for (int j = 0; j < b.length(); j++){
                if (b.charAt(j) < i) {
                    count++;
                }
            }
            min = Math.min(min, count);
        }
        return min;
    }
    
    public int minOpOneLetter(String a, String b){
        int max = 0;
        int[] letters = new int[26];
        for (int i = 0; i < a.length(); i++){
            char c = a.charAt(i);
            letters[c - 'a']++;
            if (letters[c - 'a'] > max) {
                max = letters[c - 'a'];
            }
        }
        for (int i = 0; i < b.length(); i++){
            char c = b.charAt(i);
            letters[c - 'a']++;
            if (letters[c - 'a'] > max) {
                max = letters[c - 'a'];
            }
        }
        return a.length() + b.length() - max;
    }
    
}
