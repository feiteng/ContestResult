class Solution {
    public int minCharacters(String a, String b) {
        int[] aCnt = new int[26];
        int[] bCnt = new int[26];
        for (int i = 0; i < a.length(); i++) {
            aCnt[a.charAt(i) - 'a']++;
        }
        for (int i = 0; i < b.length(); i++) {
            bCnt[b.charAt(i) - 'a']++;
        }
        int min = a.length() + b.length();
        int aSum = 0, bSum = 0;
        for (int i = 0; i < 25; i++) {
            aSum += aCnt[i];
            bSum += bCnt[i];
            min = Math.min(min, Math.min(aSum + b.length() - bSum, bSum + a.length() - aSum));
        }
        int aMax = 0, bMax = 0;
        for (int i = 0; i < 26; i++) {
            aMax = Math.max(aMax, aCnt[i]);
            bMax = Math.max(bMax, bCnt[i]);
        }
        min = Math.min(min, a.length() - aMax + b.length() - bMax);
        return min;
    }
}
