class Solution {
    public int minCharacters(String a, String b) {
        int t = Integer.MAX_VALUE;
        for (int i = 'b'; i <= 'z'; i++) {
            int temp = 0;
            for (int j = 0; j < a.length(); j++) {
                if (a.charAt(j) >= i) {
                    temp++;
                }
            }
            for (int j = 0; j < b.length(); j++) {
                if (b.charAt(j) < i) {
                    temp++;
                }
            }
            t = Math.min(t, temp);
        }

        for (int i = 'b'; i <= 'z'; i++) {
            int temp = 0;
            for (int j = 0; j < b.length(); j++) {
                if (b.charAt(j) >= i) {
                    temp++;
                }
            }
            for (int j = 0; j < a.length(); j++) {
                if (a.charAt(j) < i) {
                    temp++;
                }
            }
            t = Math.min(t, temp);
        }
        
        for (int i = 'a'; i <= 'z'; i++) {
            int temp = 0;
            for (int j = 0; j < a.length(); j++) {
                if (a.charAt(j) != i) {
                    temp++;
                }
            }
            for (int j = 0; j < b.length(); j++) {
                if (b.charAt(j) != i) {
                    temp++;
                }
            }
            t = Math.min(t, temp);
        }
        return t;
    }
}
