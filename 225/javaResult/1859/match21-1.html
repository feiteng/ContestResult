class Solution {
    public int minCharacters(String a, String b) {
        int[] cnt1 = new int[128], cnt2 = new int[128];
        for(int i=0; i<a.length(); i++) cnt1[a.charAt(i)]++;
        for(int i=0; i<b.length(); i++) cnt2[b.charAt(i)]++;
        return Math.min( Math.min(c1(cnt1, cnt2), c1(cnt2, cnt1) ), 
c2(a, b, cnt1, cnt2) );
    }
    
    int c1(int[] c1, int[] c2){
        int[] cnt1 = new int[128], cnt2 = new int[128];
        System.arraycopy(c1, 0, cnt1, 0, 128);
        System.arraycopy(c2, 0, cnt2, 0, 128);
        
        for(
int i='b', j='z'-1; i<='z'; i++, j--){
            cnt1[i] = cnt1[i-1] + cnt1[i];
            cnt2[j] = cnt2[j+1] + cnt2[j];
        }
        
        
        int ans = Integer.MAX_VALUE;
        for(int i='a'; i<'z'; i++){
            ans = Math.min(ans, cnt1[i] + cnt2[i+1] );
        }
        //System.out.println(ans);
        return ans;
    }
    
    int c2(String a, String b, int[] cnt1, int[] cnt2){
        
int max1 = 0, max2 = 0;
        for(int i='a'; i<='z'; i++){
            max1 = Math.max(max1, cnt1[i]);
            max2 = Math.max(max2, cnt2[i]);
        }
        return a.length() - max1 + b.length() - max2;
    }
    
}
