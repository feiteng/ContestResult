class Solution {
    public int minCharacters(String a, String b) {
        int[] ac = count(a), bc = count(b);
        int ans = a.length() * 2 + 1;
        int[][][] dp = new int[26][2][3];
        for (int i = 0; i < 26; i++) {
            int[] aa = cal(i, ac);
            int[] ba = cal(i, bc);
            dp[i][0] = aa;
            dp[i][1] = ba;
        }
        
        
for (int i = 1; i < 26; i++) {
            ans = Math.min(ans, dp[i][0][0] + dp[i - 1][1][2]);
        }

        
        for (int i = 0; i < 25; i++) {
            ans = Math.min(ans, dp[i][0][2] + dp[i + 1][1][0]);
        }
        for (int i = 0; i < 26; i++) {
            ans = Math.min(ans, dp[i][0][1] + dp[i][1][1]);
        }
        return ans;
    }
    public void show(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + "\t");
        }
        System.out.println();
    }
    public int[] cal(int line, int[] arr) {
        int[] ans = new int[3];
        for (int i = 0; i < line; i++) {
            ans[1] += arr[i];
            ans[2] += arr[i];
        }
        ans[0] += arr[line];
        ans[2] += arr[line];
        for (int i = line + 1; i < arr.length; i++) {
            ans[0] += arr[i];
            ans[1] += arr[i];
        
}
        return ans;
    }

    public int[] count(String a) {
        int[] ans = new int[26];
        for (char c : a.toCharArray()) {
            ans[c - 'a']++;
        }
        return ans;
    }
}
