class Solution {
    public int minCharacters(String a, String b) {
        int[] freq1 = new int[26], freq2 = new int[26];
        for (char c : a.toCharArray()) {
            freq1[c - 'a']++;
        }
        for (char c : b.toCharArray()) {
            freq2[c - 'a']++;
        }
        int aLess = freq1[0], bLess = freq2[0];
        int res = (a.length() - freq1[0]) + (b.length() - freq2[0]);
        for (int i = 1; i < 26; i++) {
            
res = Math.min(res, (a.length() - aLess) + bLess);
            res = Math.min(res, (b.length() - bLess) + aLess);
            res = Math.min(res, (a.length() - freq1[i]) + (b.length() - freq2[i]));
            aLess += freq1[i];
            bLess += freq2[i];
        }
        return res;
    }
}
