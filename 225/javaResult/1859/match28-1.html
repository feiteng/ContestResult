class Solution {
    public int minCharacters(String a, String b) {
        int[] dp1 = new int[26];
        int[] dp2 = new int[26];
        for(int i=0; i<a.length(); i++)
            dp1[a.charAt(i)-'a'] += 1;
        for(int i=0; i<b.length(); i++)
            dp2[b.charAt(i)-'a'] += 1;
        
        // unique value
        int max = 0;
        for(int i=0; i<26; i++)
            max = Math.max(max, dp1[i]+dp2[i]);
        int result = a.length()-max+b.length();
        
        int sum1 = 0;
        
int sum2 = b.length();
        for(int i=25; i>=1; i--) {
            sum1 += dp1[i];
            sum2 -= dp2[i];
            result = Math.min(result, sum2+sum1);
        }
        
        sum1 = a.length();
        sum2 = 0;
        
for(int i=25; i>=1; i--) {
            sum1 -= dp1[i];
            sum2 += dp2[i];
            result = Math.min(result, sum2+sum1);
        }
        
        return result;
    }
}
