class Solution {
    public int minCharacters(String a, String b) {
        int[] ca = new int[26], cb = new int[26];
        for (int l : a.toCharArray()) ca[l-'a']++;
        for (int l : b.toCharArray()) cb[l-'a']++;
        return less(ca, cb);
    }
    private int less(int[] ca, int[] cb) {
        int[] suma = new int[27], sumb = new int[27];
        for (int i = 0; i < 26; i++) {
            suma[i+1] = suma[i] + ca[i];
            sumb[i+1] = sumb[i] + cb[i];
        }
        
        int min = Integer.MAX_VALUE;
        for (int i = 1; i < 26; i++) {
            min = Math.min(suma[i]+ sumb[26] - sumb[i], min);
        }
         
        for (int i = 1; i < 26; i++) {
            min = Math.min(sumb[i]+ suma[26] - suma[i], min);
        }
        // System.out.println("min: " + min);
        int changea = Integer.MAX_VALUE, changeb = Integer.MAX_VALUE;
        for (int i = 0; i < 26; i++) {
            changea = Math.min(changea, suma[26] - ca[i]);
            changeb = Math.min(changeb, sumb[26] - cb[i]);
        }
        // System.out.println("changea : " + changea + ", changeb: " + changeb);
        
        
        
        return Math.min(min, changea+changeb);
    }
}
