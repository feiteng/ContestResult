class Solution {
    public int minCharacters(String a, String b) {
        int[] one = new int[26];
        int[] two = new int[26];
        int onemax = 0;
        int twomax = 0;
        for (int i = 0; i < a.length(); i++) {
            one[a.charAt(i) - 'a']++;
            onemax = Math.max(onemax, one[a.charAt(i) - 'a']);
        }
        for (int i = 0; i < b.length(); i++) {
            two[b.charAt(i) - 'a']++;
            twomax = Math.max(twomax, two[b.charAt(i) - 'a']);
        }
        
        //case 3
        int case3 = a.length() - onemax + b.length() - twomax;
        int[] oneLeft = new int[26];
        oneLeft[0] = one[0];
        int[] oneRight = new int[26];
        oneRight[25] = one[25];
        int[] twoLeft = new int[26];
        twoLeft[0] = two[0];
        int[] twoRight = new int[26];
        twoRight[25] = two[25];
        //case 1
        int case1 = Integer.MAX_VALUE;
        for (int i = 1; i < 26; i++) {
            oneLeft[i] = oneLeft[i - 1] + one[i];
            twoLeft[i] = twoLeft[i - 1] + two[i];
        }
        for (int i = 24; i >= 0; i--) {
            oneRight[i] = oneRight[i + 1] + one[i];
            twoRight[i] = twoRight[i + 1] + two[i];
        }
        for (int i = 0; i < 25; i++) {
            int cur = oneRight[i + 1] + twoLeft[i];
            case1 = Math.min(case1, cur);
        }
        // case2
        int case2 = Integer.MAX_VALUE;
        for (int i = 0; i < 25; i++) {
            int cur = twoRight[i + 1] + oneLeft[i];
            
case2 = Math.min(case2, cur);
        }
        return Math.min(Math.min(case1, case2), case3);
    }
}
