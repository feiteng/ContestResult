class Solution {
    public int minCharacters(String a, String b) {
        int[] va = new int[26], vb = new int[26];
        
        for (char e : a.toCharArray()) {
            va[e-'a']++;
        }
        for (char e : b.toCharArray()) {
            vb[e-'a']++;
        }
        
        int ans = Math.min(helper(va, vb), helper(vb, va));
        
        int same = 0;
        for (int i = 0; i < 26; i++) {
            same = Math.max(same, Math.min(va[i], vb[i]));
        }
        ans = Math.min(ans, a.length()+b.length()-same*2);
        return ans;
        
    }
    private int helper(int[] va, int[] vb) {
        int  ans = Integer.MAX_VALUE;
        
        for (int i = 0; i < 25; i++) {
            int cnt = 0;
            for (int j = 0; j < 26; j++) {
                if (j <= i) {
                    cnt += vb[j];
                } else {
                    cnt += va[j];
                }
            }
            ans = Math.min(ans, cnt);
        }
        return ans;
        
    }
}
