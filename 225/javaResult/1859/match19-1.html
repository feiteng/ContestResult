class Solution {
    public int minCharacters(String a, String b) {
        int[] cnt1 = new int[26];
        int[] cnt2 = new int[26];
        for (char c : a.toCharArray()) {
            cnt1[c - 'a']++;
        }
        for (char c : b.toCharArray()) {
            cnt2[c - 'a']++;
        }
        
        //distinct 
        int resd = Integer.MAX_VALUE;
        
for (int i = 0; i < 26; i++) {
            int curRes = 0;
            for (int j = 0; j < 26; j++) {
                if (j != i) {
                    curRes += cnt1[j];
                    curRes += cnt2[j];
                }
            }
            resd = Math.min(resd, curRes);
        }
        // System.out.println("dis: " + resd);
        
        //a < b
        int resm = Integer.MAX_VALUE;
        for (int i = 0; i < 26; i++) {
            for (int j = i + 1; j < 26; j++) {
                int cntRes1 = 0;
                
int cntRes2 = 0;
                for (int p = 0; p < 26; p++) {
                    if (p > i) {
                        cntRes1 += cnt1[p];
                        cntRes2 += cnt2[p];
                    }
                    if (p < j) {
                        cntRes1 += cnt2[p];
                        cntRes2 += cnt1[p];
                    }
                }
                resm = Math.min(Math.min(resm, cntRes1), cntRes2);
            }
        }
        // System.out.println("minor: " + resm);
        return Math.min(resm, resd);
    }
}
