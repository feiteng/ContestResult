class Solution {
    public int minCharacters(String a, String b) {
        int[] ca = new int[26];
        int[] cb = new int[26];
        for(char c : a.toCharArray()) ca[c-'a']++;
        for(char c : b.toCharArray()) cb[c-'a']++;
        int res = Math.min(minMove(ca, cb), minMove(cb, ca));
        res = Math.min(res, a.length() - 
Arrays.stream(ca).max().orElse(0) + b.length() - Arrays.stream(cb).max().orElse(0));
        return res;
    }
    
    private int  minMove(int[] a, int[] b) {
        int move = b[0];
        for(int i = 1; i < 26; i++) move += a[i];
        
int res = move;
        for(int i = 1; i < 25; i++) {
            move -= a[i];
            move += b[i];
            res = Math.min(move, res);
        }
        return res;
    }
}
