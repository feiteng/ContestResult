class Solution {
    public int minCharacters(String a, String b) {
        int[] countA = new int[26], countB = new int[26], sumA = new int[26], sumB = new int[26];
        for (char c : a.toCharArray()) {
            countA[c - 'a']++;
        }
        for (char c : b.toCharArray()) {
            countB[c - 'a']++;
        }
        for (int i = 0; i < countA.length; i++) {
            sumA[i] = (i == 0 ? 0 : sumA[i - 1]) + countA[i];
            sumB[i] = (i == 0 ? 0 : sumB[i - 1]) + countB[i];
        }
        int max = 0;
        for (int i = 0; i < 26; i++) {
            max = Math.max(max, countA[i] + countB[i]);
        }
        int ans = a.length() + b.length() - max;
        for (int i = 0; i < 25; i++) {
            ans = Math.min(sumA[i] + b.length() - sumB[i], ans);
            ans = Math.min(sumB[i] + a.length() - sumA[i], ans);
        }
        return ans;
    }
}
