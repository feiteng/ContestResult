class Solution {
    public int minCharacters(String a, String b) {
        int[] countA = new int[26];
        int[] countB = new int[26];
        for (int j = 0; j < a.length(); j++) {
            countA[a.charAt(j)-'a']++;
        }
        for (int j = 0; j < b.length(); j++) {
            countB[b.charAt(j)-'a']++;
        }
        int ret = Integer.MAX_VALUE;
        for (int j = 0; j < 26; j++) {
            ret = Math.min(ret, a.length()-countA[j] + b.length()-countB[j]);
        }
        for (int j = 1; j < 26; j++) {
            int temp = 0;
            for (int k = 0; k < j; k++) {
                temp += countB[k];
            }
            for (int k = j; k < 26; k++) {
                temp += countA[k];
            }
            ret = Math.min(ret, temp);
        }
        for (int j = 1; j < 26; j++) {
            int temp = 0;
            for (int k = 0; k < j; k++) {
                temp += countA[k];
            }
            for (int k = j; k < 26; k++) {
                temp += countB[k];
            }
            ret = Math.min(ret, temp);
        }
        return ret;
    }
}
