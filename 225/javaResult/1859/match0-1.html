class Solution {
    public int minCharacters(String a, String b) {
        int[] cout1=new int[26],cout2=new int[26];
        int m=a.length();
        int n=b.length();
        int res=Integer.MAX_VALUE;
        
for(char ch:a.toCharArray()){
            cout1[ch-'a']++;
        } 
        for(char ch:b.toCharArray()){
            cout2[ch-'a']++;
        }
        //
        
        
for(int min=1;min<26;min++){
            int cur=0;
            for(int i=min;i<26;i++){
                cur+=cout1[i];
            }
            for(int i=0;i<min;i++){
                cur+=cout2[i];
            }
            res=Math.min(res,cur);
        }
        
        for(int min=1;min<26;min++){
            int cur=0;
            for(int i=min;i<26;i++){
                cur+=cout2[i];
            }
            for(int i=0;i<min;i++){
                cur+=cout1[i];
            }
            res=Math.min(res,cur);
        }
        
        for(int i=0;i<26;i++){
             int cur=0;
             for(int j=0;j<26;j++){
                 if(i!=j){
                     cur+=cout1[j]+cout2[j];
                 }
             }
             res=Math.min(res,cur);
        }
        return res; 
    }
}
