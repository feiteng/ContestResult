class Solution {
    public int minCharacters(String a, String b) {
        
        int n1 = a.length();
        int n2 = b.length();
        
        int[] f1 = new int[26];
        int[] f2 = new int[26];
        
        for (int i = 0; i < n1; i++) {
            f1[a.charAt(i) - 'a']++;
        }
        
        for (int i = 0; i < n2; i++) {
            f2[b.charAt(i) - 'a']++;
        }
        
        int ans = n1 + n2;
        for (int i = 0; i < 25; i++) {
            int current1 = 0;
            for (int j = i + 1; j < 26; j++) {
                current1 += f1[j];
            }
            for (int j = 0; j <= i; j++) {
                current1 += f2[j];
            }
            ans = Math.min(ans, current1);
            
            int current2 = 0;
            for (int j = i + 1; j < 26; j++) {
                current2 += f2[j];
            }
            for (int j = 0; j <= i; j++) {
                current2 += f1[j];
            }
            ans = Math.min(ans, current2);            
        }
        
        for (int i = 0; i < 26; i++) {
            int current3 = 0;
            for (int j = 0; j < 26; j++) {
                if (j == i) {
                    continue;
                }
                current3 += f1[j] + f2[j];
            }
            ans = Math.min(ans, current3); 
        }
        
        return ans;
    }
}
