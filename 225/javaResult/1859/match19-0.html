class Solution {
    public int minCharacters(String a, String b) {
        int[] cca = new int[26];
    int[] ccb = new int[26];
    for (char c : a.toCharArray()) {
      cca[c - 'a']++;
    }
    for (char c : b.toCharArray()) {
      ccb[c - 'a']++;
    }
    int ans = a.length() + b.length();
    // 1. Every letter in a is strictly less than every letter in b in the alphabet.
    for (int i = 0; i < 26; i++) {
      for (int j = i + 1; j < 26; j++) {
        // number of operations to change characters in a <= i and b >= j
        int count = 0;
        for (int k = 0; k < 26; k++) {
          if (k > i) {
            count += cca[k];
          }
          if (k < j) {
            count += ccb[k];
          }
        }
        ans = Math.min(ans, count);
      }
    }

    // 2.
    for (int i = 0; i < 26; i++) {
      for (int j = i + 1; j < 26; j++) {
        // number of operations to change characters in b <= i and a >= j
        int count = 0;
        for (int k = 0; k < 26; k++) {
          if (k > i) {
            count += ccb[k];
          }
          if (k < j) {
            count += cca[k];
          }
        }
        ans = Math.min(ans, count);
      }
    }

    // 3. Both a and b consist of only one distinct letter.
    for (int i = 0; i < 26; i++) {
      int count = 0;
      for (int k = 0; k < 26; k++) {
        if (k != i) {
          count += cca[k];
          count += ccb[k];
        }
      }
      ans = Math.min(ans, count);
    }
    return ans;
    }
}
