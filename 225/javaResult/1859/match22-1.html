class Solution {
    public int minCharacters(String a, String b) {
        int[] countA = new int[26];
        int[] countB = new int[26];
        for (int i = 0; i < a.length(); i++) {
            countA[a.charAt(i) - 'a']++;
        }
        for (int j = 0; j < b.length(); j++) {
            countB[b.charAt(j) - 'a']++;
        }
        int result = Integer.MAX_VALUE;
        int numberPrefixA = 0, numberPrefixB = 0;
        int maxA = 0, maxB = 0;
        for (int i = 0; i < 26; i++) {
            numberPrefixA += countA[i];
            numberPrefixB += countB[i];
            if (i < 25) {
                int costALessThanB = (a.length() - numberPrefixA) + numberPrefixB; // a全部转化为<=i, b全部转化为>i
                int costBLessThanA = (b.length() - numberPrefixB) + numberPrefixA; // 
                result = Math.min(result, Math.min(costALessThanB, costBLessThanA));
            }
            
maxA = Math.max(maxA, countA[i]);
            maxB = Math.max(maxB, countB[i]);
        }
        result = Math.min(result, (a.length() - maxA) + (b.length() - maxB));
        return result;
    }
}
