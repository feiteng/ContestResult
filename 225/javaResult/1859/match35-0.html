class Solution {
    public int minCharacters(String a, String b) {
        int[] cnt1 = getCount(a), cnt2 = getCount(b);
        int max1 = 0, max2 = 0;
        for (int c : cnt1) {
            max1 = Math.max(max1, c);
        }
        for (int c : cnt2) {
            max2 = Math.max(max2, c);
        }
        int res = a.length() + b.length() - max1 - max2;
        int cur1 = 0, cur2 = 0, len1 = a.length(), len2 = b.length();
        for (int i = 0;i < 25;i++) {
            cur1 += cnt1[i]; cur2 += cnt2[i];
            res = Math.min(len1 - cur1 + cur2, res);
            res = Math.min(len2 - cur2 + cur1, res);
        }
        return res;
    }
    private int[] getCount(String a) {
        int[] cnts = new int[26];
        for (char c : a.toCharArray()) {
            cnts[c - 'a']++;
        }
        return cnts;
    }
}
