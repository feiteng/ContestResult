class Solution {
    public int minCharacters(String a, String b) {
        int[] stat1 = new int[26], stat2 = new int[26];
        for (int i = 0; i < a.length(); i++) {
            stat1[a.charAt(i) - 'a']++;
        }
        for (int i = 0; i < b.length(); i++) {
            stat2[b.charAt(i) - 'a']++;
        }
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < 25; i++) {
            int moves = 0;
            for (int j = 0; j <= i; j++) moves += stat1[j];
            for (int j = i + 1; j < 26; j++) moves += stat2[j];
            ans = Math.min(ans, moves);
        }
        for (int i = 0; i < 25; i++) {
            int moves = 0;
            for (int j = 0; j <= i; j++) moves += stat2[j];
            for (int j = i + 1; j < 26; j++) moves += stat1[j];
            ans = Math.min(ans, moves);
        }
        for (int i = 0; i < 26; i++) {
            int moves = 0;
            for (int j = 0; j < 26; j++) {
                if (i == j) continue;
                moves += stat1[j] + stat2[j];
            }
            ans = Math.min(ans, moves);
        }
        return ans;
    }
}
