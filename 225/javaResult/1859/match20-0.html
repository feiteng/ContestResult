class Solution {
    public int minCharacters(String a, String b) {
        int n1 = a.length(), n2 = b.length();
        int[] cnt1 = new int[26], cnt2 = new int[26];
        for(int i = 0; i < n1; i++) {
            cnt1[a.charAt(i) - 'a']++;
        }
        for(int i = 0; i < n2; i++) {
            cnt2[b.charAt(i) - 'a']++;
        }
        int ans = Integer.MAX_VALUE;
        for(int i = 0; i < 26; i++) {
            for(int j = 0; j < 26; j++) {
                int cost = 0;
                for(int k = 0; k < 26; k++) {
                    if(k != i) {
                        cost += cnt1[k];
                    }
                    if(k != j) {
                        cost += cnt2[k];
                    }
                }
                ans = Math.min(ans, cost);
            }
        }
        for(int i = 0; i < 25; i++) {
            int cost = 0;
            for(int k = i + 1; k < 26; k++) {
                cost += cnt1[k];
            }
            for(int k = 0; k <= i; k++) {
                cost += cnt2[k];
            }
            ans = Math.min(ans, cost);
        }
        for(int i = 1; i < 26; i++) {
            int cost = 0;
            for(int k = 0; k < i; k++) {
                cost += cnt1[k];
            }
            for(int k = i; k < 26; k++) {
                cost += cnt2[k];
            }
            ans = Math.min(ans, cost);
        }
        return ans;
    }
}
