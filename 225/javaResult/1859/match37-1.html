class Solution {
    public int minCharacters(String a, String b) {
        int n = a.length();
        int m = b.length();
        int[] aa = new int[26];
        int[] bb = new int[26];
        int max = 0;
        for(char c : a.toCharArray()) {
            aa[c-'a']++;
            max = Math.max(max, aa[c-'a']);
        }
        
        int ap = n - max;
        max = 0;
        
for(char c : b.toCharArray()) {
            bb[c-'a']++;
            max = Math.max(max, bb[c-'a']);
        }
        
        int bp = m - max;
        
        int res = ap + bp;
        
        int[] pa = new int[26];
        int[] pb = new int[26];
        pa[0] = aa[0];
        pb[0] = bb[0];
        for(int i = 1; i < 26; i++) {
            pa[i] = pa[i-1] + aa[i];
            pb[i] = pb[i-1] + bb[i];
        }
        
        
        // letter >= j;
        for(int j = 25; j >= 1; j--) {
            int x = pa[25] - pa[j- 1];
            int y = pb[25] - pb[j- 1];
            int min = 
Math.min(n - x + y, x+ m - y);
            res = Math.min(min, res);
        }
        
        return res;
    }
}
