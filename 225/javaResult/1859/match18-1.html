class Solution {
    public int minCharacters(String a, String b) {
        int[] afreq = new int[26];
        int[] bfreq = new int[26];
        for (int i = 0; i < a.length(); i++) {
            char c = a.charAt(i);
            afreq[c - 'a']++;
        }
        for (int i = 0; i < b.length(); i++) {
            char c = b.charAt(i);
            bfreq[c - 'a']++;
        }
        
        int ans = Integer.MAX_VALUE;
        
for (char c = 'b'; c <= 'z'; c++) {
            int cnt = 0;
            for (int i = 0; i < 26; i++) {
                if (i + 'a' >= c) {
                    cnt += afreq[i];
                }
                if (i + 'a' < c) {
                    cnt += bfreq[i];
                }
            }
            ans = Math.min(ans, cnt);
        }
        
        
for (char c = 'b'; c <= 'z'; c++) {
            int cnt = 0;
            for (int i = 0; i < 26; i++) {
                if (i + 'a' >= c) {
                    cnt += bfreq[i];
                }
                if (i + 'a' < c) {
                    cnt += afreq[i];
                }
            }
            ans = Math.min(ans, cnt);
        }
        
        for (char c = 'b'; c <= 'z'; c++) {
            int cnt = 0;
            for (int i = 0; i < 26; i++) {
                if (i + 'a' != c) {
                    cnt += afreq[i];
                }
                
if (i + 'a' != c) {
                    cnt += bfreq[i];
                }
            }
            ans = Math.min(ans, cnt);
        }
        
        return ans;
    }
}
