class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[] val = new int[n * m];
        int idx = 0;
        int[][] help = new int[m][n];
        
for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0 && j == 0) {
                    help[i][j] = matrix[i][j];
                } else if (i == 0) {
                    help[i][j] = help[i][j - 1] ^ matrix[i][j];
                } else if (j == 0) {
                    help[i][j] = help[i - 1][j] ^ matrix[i][j];
                } else {
                    help[i][j] = help[i][j - 1] ^ help[i - 1][j] ^ help[i - 1][j - 1] ^ matrix[i][j];
                }
                val[idx++] = help[i][j];
            }
        }
        Arrays.sort(val);
        return val[m * n - k];
    }
}
