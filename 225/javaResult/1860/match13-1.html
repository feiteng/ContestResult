class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int n = matrix.length, m = matrix[0].length;
        int[][] a = new int[n][m];
        int[] all = new int[n*m];
        int ret = 0;
        
for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                a[i][j] = matrix[i][j];
                if(i-1 >= 0){
                    a[i][j] ^= a[i-1][j];
                }
                if(j-1 >= 0){
                    a[i][j] ^= a[i][j-1];
                }
                if(i-1 >= 0 && j-1 >= 0){
                    a[i][j] ^= a[i-1][j-1];
                }
                all[i*m+j] = a[i][j];
            }
        }
        Arrays.sort(all);
        return all[n*m-k];
    }
}
