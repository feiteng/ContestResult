class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] xor = new int[m][n];
        int[] xorArr = new int[m * n];
        int idx = 0;
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                xor[i][j] = matrix[i][j];
                if(i - 1 >= 0) {
                    xor[i][j] ^= xor[i - 1][j];
                }
                if(j - 1 >= 0) {
                    xor[i][j] ^= xor[i][j - 1];
                }
                if(i - 1 >= 0 && j - 1 >= 0) {
                    xor[i][j] ^= xor[i - 1][j - 1];
                }
                xorArr[idx++] = xor[i][j];
            }
        }
        Arrays.sort(xorArr);
        return xorArr[m * n - k];
    }
}
