class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        final int ROW = matrix.length, COL = matrix[0].length;
        int[] nums = new int[ROW * COL];
        
int[][] xor = new int[ROW][COL];
        
        nums[0] = xor[0][0] = matrix[0][0];
        for (int r = 1; r < ROW; r++) {
            xor[r][0] = matrix[r][0] ^ xor[r - 1][0];
            nums[r * COL] = xor[r][0];
        }
        for (int c = 1; c < COL; c++) {
            xor[0][c] = matrix[0][c] ^ xor[0][c - 1];
            nums[c] = xor[0][c];
        }

        for (int r = 1; r < ROW; r++) {
            for (int c = 1; c < COL; c++) {
                xor[r][c] = matrix[r][c] ^ xor[r - 1][c - 1] ^ xor[r - 1][c] ^ xor[r][c - 1];
                nums[r * COL + c] = xor[r][c];
            }
        }
        Arrays.sort(nums);
        return nums[nums.length - k];
    }
}
