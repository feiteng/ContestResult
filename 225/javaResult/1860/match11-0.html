class Solution {
  public int kthLargestValue(int[][] v, int k) {
    int n = v.length, m = v[0].length;
    int[][] x = new int[n][m];
    int[] data = new int[n*m];
    int cnt = 0;
    for(int i = 0; i < n; i++) {
      for(int j = 0; j < m; j++) {
        x[i][j] = v[i][j];
        if(i > 0) x[i][j] ^= x[i-1][j];
        if(j > 0) x[i][j] ^= x[i][j-1];
        if(i>0 && j>0) x[i][j] ^= x[i-1][j-1];
        data[cnt++] = x[i][j];
      }
    }
    Arrays.sort(data);
    return data[n*m-k];
  }
}
