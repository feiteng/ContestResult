class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int n = matrix.length;
        int m = matrix[0].length;
        int[][] dp = new int[n][m];
        Queue<Integer> queue = new PriorityQueue<>();
        dp[0][0] = matrix[0][0];
        queue.offer(dp[0][0]);
        for(int j = 1; j < m; j++) {
            dp[0][j] = dp[0][j-1] ^ matrix[0][j];
            queue.offer(dp[0][j]);
            if(queue.size() > k) {
                queue.poll();
            }
        }
        
        for(int i = 1; i < n; i++) {
            dp[i][0] = dp[i-1][0] ^ matrix[i][0];
            queue.offer(dp[i][0]);
            if(queue.size() > k) {
                queue.poll();
            }
        }
        
        for(int i = 1; i < n; i++) {
            for(int j = 1; j < m; j++) {
                dp[i][j] = dp[i-1][j] ^ dp[i][j-1] ^ dp[i-1][j-1] ^ matrix[i][j];
                queue.offer(dp[i][j]);
                if(queue.size() > k) {
                    queue.poll();
                }
            }
        }
        
        return queue.peek();
    }
}
