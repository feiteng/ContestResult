class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
    int n = matrix[0].length;
    int[][] arr = new int[m][n];
    List<Integer> nums = new ArrayList<>();
    arr[0][0] = matrix[0][0];
    nums.add(arr[0][0]);
    for (int j = 1; j < n; j++) {
      arr[0][j] = matrix[0][j] ^ arr[0][j-1];
      nums.add(arr[0][j]);
    }
    for (int i = 1; i < m; i++) {
      arr[i][0] = matrix[i][0] ^ arr[i-1][0];
      nums.add(arr[i][0]);
    }
    for (int i = 1; i < m; i++) {
      for (int j = 1; j < n; j++) {
        arr[i][j] = matrix[i][j] ^ arr[i-1][j] ^ arr[i][j-1] ^ arr[i-1][j-1];
        nums.add(arr[i][j]);
      }
    }
    Collections.sort(nums, Collections.reverseOrder());

    return nums.get(k-1);
    }
}
