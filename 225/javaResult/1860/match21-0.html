class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
        int n = matrix[0].length;
        
        //use dp array to store all results
        int[][] dp = new int[m][n];
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < m; ++i) {
            int row = 0;
            for (int j = 0; j < n; ++j) {
                row ^= matrix[i][j];
                dp[i][j] = row ^ (i > 0 ? dp[i - 1][j] : 0);
                list.add(dp[i][j]);
            }
        }
        
        Collections.sort(list, Collections.reverseOrder());
        return list.get(k - 1);
    }
}
