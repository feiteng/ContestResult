class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int[][] xorM = new int[matrix.length][matrix[0].length];
        
        xorM[0][0] = matrix[0][0];
        
        for (int row = 1; row < matrix.length; row ++) {
            xorM[row][0] = matrix[row][0] ^ xorM[row - 1][0]; 
        }
        for (int col = 1; col < matrix[0].length; col ++) {
            xorM[0][col] = matrix[0][col] ^ xorM[0][col - 1];
        }
        
        for (int row = 1; row < matrix.length; row ++) {
            for (int col = 1; col < matrix[0].length; col ++) {
                xorM[row][col] = matrix[row][col] ^ xorM[row - 1][col] ^ xorM[row][col - 1] ^ xorM[row - 1][col - 1];
            }
        }
        List<Integer> nums = new ArrayList<>();
        for (int row = 0; row < matrix.length; row ++) {
            for (int col = 0; col < matrix[0].length; col ++) {
                nums.add(xorM[row][col]);
            }
        }
        Collections.sort(nums, Collections.reverseOrder());
        return nums.get(k - 1);
        
        
    }
}
