class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int n=matrix.length,m=matrix[0].length;
        int cxor[][]=new int[n+1][m+1];
        List<Integer>list=new ArrayList<>();
        
for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                cxor[i+1][j+1]=cxor[i][j+1]^cxor[i+1][j]^cxor[i][j]^matrix[i][j];
                list.add(cxor[i+1][j+1]);
            }
        }
        list.sort(Comparator.reverseOrder());
        return list.get(k-1);
    }
}
