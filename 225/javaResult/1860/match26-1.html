class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int m = matrix.length, n = matrix[0].length;
        int[][] v= new int[m][n];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int curv = 0;
                
if(i > 0 && j > 0){
                     curv = matrix[i][j] ^ v[i][j - 1] ^ v[i - 1][j] ^ v[i-1][j-1];
                } else if(i > 0){
                     curv = matrix[i][j] ^ v[i - 1][j];
                } else if(j > 0){
                    curv = matrix[i][j] ^ v[i][j - 1];
                } else {
                    curv = matrix[i][j];
                }
                v[i][j] = curv;
                pq.offer(curv);
                if(pq.size()>k)pq.poll()
;
            }
        }
        return pq.poll();
    }
}
