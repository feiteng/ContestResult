class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int n = matrix.length;
        int m = matrix[0].length;
        int[][] res = new int[n + 1][m + 1];
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                res[i + 1][j + 1] = res[i + 1][j] ^ res[i][j + 1] ^ res[i][j] ^ matrix[i][j];
                int cur = res[i + 1][j + 1];
                pq.offer(cur);
                if (pq.size() > k) {
                    pq.poll();
                }
            }
        }
        return pq.peek();
    }
}
