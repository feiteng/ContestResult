import java.util.PriorityQueue;


class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(k + 1, (a, b) -> (a - b));
        int[][] nm = new int[matrix.length + 1][matrix[0].length + 1];
        
for (int i = 1; i < nm.length; i++) {
            for (int j = 1; j < nm[0].length; j++) {
                nm[i][j] = nm[i - 1][j - 1] ^ nm[i - 1][j] ^ nm[i][j - 1] ^ matrix[i - 1][j - 1];
                pq.add(nm[i][j]);
                if (pq.size() > k) pq.poll();
            }
        }
        return pq.poll();
    }
}
