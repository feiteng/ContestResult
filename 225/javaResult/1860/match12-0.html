class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length, n = matrix[0].length;
        int[] sort = new int[m * n];
        int idx = 0;
        int[][] xorMatrix = new int[m][n];
        xorMatrix[0][0] = matrix[0][0];
        sort[idx++] = xorMatrix[0][0];
        for (int i = 1; i < m; i++) {
            xorMatrix[i][0] = matrix[i][0] ^ xorMatrix[i - 1][0];
            sort[idx++] = xorMatrix[i][0];
        }
        for (int i = 1; i < n; i++) {
            xorMatrix[0][i] = matrix[0][i] ^ xorMatrix[0][i - 1];
            sort[idx++] = xorMatrix[0][i];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                xorMatrix[i][j] = matrix[i][j] ^ xorMatrix[i - 1][j] ^ xorMatrix[i][j - 1] ^ xorMatrix[i - 1][j - 1];
                sort[idx++] = xorMatrix[i][j];
            }
        }
        Arrays.sort(sort);
        return sort[sort.length - k];
    }
}
