class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int rowlen = matrix.length;
        int collen = matrix[0].length;
        int[][] jc = new int[rowlen][collen];
        int[] rank = new int[rowlen*collen];
        int v = 0;
        for(int x = 0;x<rowlen;x++){
            for(int y = 0;y<collen;y++){
                if(x==0&&y==0){
                    jc[x][y]=matrix[x][y];
                }else if(x==0){
                    jc[x][y]=jc[0][y-1]^matrix[x][y];
                }else if(y==0){
                    jc[x][y]=jc[x-1][y]^matrix[x][y];
                }else{
                    int cd = matrix[x][y];
                    if(x<y){
                        for(int i=0;i<x;i++){
                            cd = cd^matrix[i][y];
                        }
                        jc[x][y]=jc[x][y-1]^cd;
                    }else{
                        for(int i=0;i<y;i++){
                            cd = cd^matrix[x][i];
                        }
                        jc[x][y]=jc[x-1][y]^cd;
                    }
                
}
                rank[v++]=jc[x][y];
            }
        }
        Arrays.sort(rank);
        return rank[rowlen*collen-k];
    }
}
