class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m=matrix.length, n=matrix[0].length;
        int[][] dp = new int[m][n];
        List<Integer> lst = new ArrayList<>();
        dp[0][0] = matrix[0][0];
        lst.add(dp[0][0]);
        for (int i=1; i<m; i++) {dp[i][0]=dp[i-1][0]^matrix[i][0]; lst.add(dp[i][0]);}
        for (int i=1; i<n; i++) {dp[0][i]=dp[0][i-1]^matrix[0][i]; lst.add(dp[0][i]);}
        
for (int i=1; i<m; i++) {
            for (int j=1; j<n; j++) {
                dp[i][j] = dp[i-1][j-1]^dp[i][j-1]^dp[i-1][j]^matrix[i][j];
                lst.add(dp[i][j]);
            }
        }
        Collections.sort(lst,Collections.reverseOrder());
        return lst.get(k-1);
    }
}
