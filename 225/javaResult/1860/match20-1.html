class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int r = matrix.length;
        int c = matrix[0].length;
        int[][] rowXor = new int[r][c];
        for (int i = 0; i < r; i++) {
            rowXor[i][0] = matrix[i][0];
            for (int j = 1; j < c; j++) {
                rowXor[i][j] = rowXor[i][j - 1] ^ matrix[i][j];
            }
        }
        List<Integer> values = new ArrayList<>();
        for (int j = 0; j < c; j++) {
            int v = rowXor[0][j];
            values.add(v);
            for (int i = 1; i < r; i++) {
                v ^= rowXor[i][j];
                values.add(v);
            }
        }
        Collections.sort(values);
        return values.get(r * c - k);
    }
}
