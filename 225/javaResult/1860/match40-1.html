class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] vert = new int[m][n];
        int[][] hori = new int[m][n];
        for (int i = 0; i < m; i++) {
            
        }
        for (int i = 0; i < m; i++) {
            hori[i][0] = matrix[i][0];
            for (int j = 1; j < n; j++) {
                hori[i][j] = hori[i][j-1] ^ matrix[i][j];
            }
        }
        for (int i = 0; i < n; i++) {
            vert[0][i] = matrix[0][i];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                vert[i][j] = matrix[i][j] ^ vert[i-1][j];
            }
        }
        int[][] xor = new int[m][n];
        for (int i = 0; i < m; i++) {
            xor[i][0] = vert[i][0];
        }
        for (int i = 0; i < n; i++) {
            xor[0][i] = hori[0][i];
        }
        
for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                xor[i][j] = xor[i-1][j-1] ^ vert[i][j] ^ hori[i][j-1];
            }
        }
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                pq.add(xor[i][j]);
                if (pq.size() > k) {
                    pq.poll();
                
}
            }
        }
        
        return pq.poll();
    }
}
