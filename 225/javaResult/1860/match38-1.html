class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int n = matrix.length;
        int m = matrix[0].length;
        int[][] xor = new int[n][m];
        PriorityQueue<Integer> q = new PriorityQueue<>();

        for (int i = 0; i < n; i++) 
        {
            for (int j = 0; j < m; j++) 
            {
                int a;
                if (i - 1 >= 0) 
                {
                    a = xor[i - 1][j];
                }
                else 
                {
                    a = 0;
                }
                int b; 
                if(j - 1 >= 0) 
                {
                    b = xor[i][j - 1];
                }
                else
                {
                    b = 0;
                }
                int c; 
                if (i - 1 >= 0 && j - 1 >= 0)
                {
                    c = xor[i - 1][j - 1];
                }
                else
                {
                    c = 0;
                }
                
xor[i][j] = matrix[i][j] ^ a ^ b ^ c;
                q.offer(xor[i][j]);
                if (q.size() > k) 
                {
                    q.poll();
                }
            }
        }
        return q.poll();
    }
}
