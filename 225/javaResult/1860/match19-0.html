class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] arr = new int[m][n];
        arr[0][0] = matrix[0][0];

        List<Integer> list = new ArrayList<>();
        list.add(arr[0][0]);
        
        for(int i = 1 ; i < m ; i++){
            arr[i][0] = arr[i-1][0] ^ matrix[i][0];
            list.add(arr[i][0]);
        }
        for(int i = 1 ; i < n ; i++){
            arr[0][i] = arr[0][i-1] ^ matrix[0][i];
            list.add(arr[0][i]);
        }
        for(int i = 1 ; i < m ; i++){
            for(int j = 1 ; j < n ; j++){
                arr[i][j] = arr[i-1][j-1] ^ arr[i-1][j] ^ arr[i][j-1] ^ matrix[i][j];
                list.add(arr[i][j]);
            }
        }
        
        Collections.sort(list);
        
//         for(int i = 0 ; i < m ; i++){
//             for(int j = 0 ; j < n ; j++){
//                 System.out.print(arr[i][j] + " ");
//             }
//             System.out.println();
//         }
        
//         for(int i:list){
//             System.out.print(i + " ");
//         }
//         System.out.println();
        
        return list.get(list.size() - k);
    }
}
