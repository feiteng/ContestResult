class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
    int n = matrix.length;
    int m = matrix[0].length;
    int[][] xor = new int[n][m];
    xor[0][0] = matrix[0][0];
    for (int i = 1; i < n; i++) {
      xor[i][0] = xor[i - 1][0] ^ matrix[i][0];
    }
    for (int j = 1; j < m; j++) {
      xor[0][j] = xor[0][j - 1] ^ matrix[0][j];
    }
    for (int i = 1; i < n; i++) {
      for (int j = 1; j < m; j++) {
        xor[i][j] = matrix[i][j] ^ xor[i - 1][j] ^ xor[i][j - 1] ^ xor[i - 1][j - 1];
      }
    }
    List<Integer> res = new ArrayList<>();
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        res.add(xor[i][j]);
      }
    }
    Collections.sort(res);
    Collections.reverse(res);
    return res.get(k - 1);
  }
}
