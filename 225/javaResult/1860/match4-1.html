class Solution {
    public int kthLargestValue(int[][] matrix, int k) {
        int m = matrix.length, n = matrix[0].length;
        int[] xor = new int[m * n];
        
for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                xor[i * n + j] = matrix[i][j];
                if (i > 0 && j > 0) {
                    xor[i * n + j] ^= xor[(i - 1) * n + j - 1];
                }
                if (i > 0) {
                    xor[i * n + j] ^= xor[(i - 1) * n + j];
                } 
                if (j > 0) {
                    xor[i * n + j] ^= xor[(i) * n + j - 1];
                }
            
}
        }
        //System.out.println(Arrays.toString(xor));
        Arrays.sort(xor);
        return xor[m * n - k];
    }
}
