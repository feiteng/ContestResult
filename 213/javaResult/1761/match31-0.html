import java.util.Arrays;

class Solution {
    public int countVowelStrings(int n) {
        int[] dp = new int[5];
        Arrays.fill(dp, 1);
        for (int i = 1; i < n; i++) {
            int[] ndp = new int[5];
            for (int j = 0; j < 5; j++) {
                for (int k = j; k < 5; k++) {
                    ndp[j] += dp[k];
                }
            }
            dp = ndp;
        }
        int ans = 0;
        for (int i = 0; i < dp.length; i++) {
            ans += dp[i];
        }
        return ans;
    }
}
