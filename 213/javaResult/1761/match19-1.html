class Solution {
    public int countVowelStrings(int n) {
        int[][] f = new int[n + 1][5];
        Arrays.fill(f[1], 1);
        for (int i = 2; i <= n; i++) {
            for (int j = 0; j < 5; j++) {
                for (int k = 0; k <= j; k++) {
                    f[i][j] += f[i - 1][k];
                }
            }
        }
        return f[n][0] + f[n][1] + f[n][2] + f[n][3] + f[n][4];
    }
}
