class Solution {
    public int countVowelStrings(int n) {
        return dfs(n, 0);
    }
    
    public int dfs(int n, int pos){
        if(n==0){
            return 1;
        }
        if(pos==5){
            return 0;
        }
        int res = 0;
        for(int i=0; i<=n; i++){
            res += dfs(n-i, pos+1);
        }
        return res;
    }
}
