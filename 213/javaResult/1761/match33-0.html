class Solution {
    public int countVowelStrings(int n) {
        int[] dp = new int[5];
        Arrays.fill(dp, 1);
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < dp.length; ++j) {
                for (int k = j + 1; k < dp.length; ++k) {
                    dp[j] += dp[k];
                }
            }
        }
        int sum = 0;
        for (int i = 0; i < dp.length; ++i) {
            sum += dp[i];
        }
        // System.out.println(Arrays.toString(dp));
        return sum;
    }
}
