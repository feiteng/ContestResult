class Solution {
    public int countVowelStrings(int n) {
        int[][] dp = new int[n][5];
    for (int k = 0; k < 5; k++) {
      dp[0][k] = 1;
    }
    for (int m = 0; m < n - 1; m++) {
      for (int k = 0; k < 5; k++) {
        for (int j = 0; j <= k; j++) {
          dp[m+1][k] += dp[m][j];
        }
      }
    }
    int ans = 0;
    for (int k = 0; k < 5; k++) {
      ans += dp[n-1][k];
    }
    return ans;
    }
}
