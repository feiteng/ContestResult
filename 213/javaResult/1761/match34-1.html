class Solution {
    public int countVowelStrings(int n) {
        int[] cnt = new int[5];
        Arrays.fill(cnt, 1);
        
        for (int i = 2; i <= n; i++) {
            int[] next = new int[5];
            
for (int j = 0; j < 5; j++) {
                for (int k = j; k < 5; k++) {
                    next[k] += cnt[j];
                }
            }
            cnt = next;
        
}
        
        int ret = 0;
        for (int num : cnt) {
            ret += num;
        }
        
        return ret;
    }
}
