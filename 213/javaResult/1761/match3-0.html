class Solution {
    public int countVowelStrings(int n) {
        int[] counts = new int[5];
        Arrays.fill(counts, 1);
        for (int i = 1; i < n; i++) {
            for (int j = 4; j >= 0; j--) {
                for (int k = 0; k < j; k++) {
                    counts[j] += counts[k];
                }
            }
        }
        int res = 0;
        for (int i = 0; i < 5; i++) {
            res += counts[i];
        }
        return res;
    }
}
