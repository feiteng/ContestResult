class Solution {
    public int countVowelStrings(int n) {
        int[][] dp = new int[n + 1][6];
        dp[0][0] = 1;
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= 5; ++j) {
                for (int k = 0; k <= j; ++k) {
                    dp[i][j] += dp[i - 1][k];
                }
            }
        }
        return dp[n][1] + dp[n][2] + dp[n][3] + dp[n][4] + dp[n][5];
    }
}
