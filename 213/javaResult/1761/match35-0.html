class Solution {
    public int countVowelStrings(int n) {
        return helper(n, 5);
    }
    
    private int helper(int n, int c){
        if (n == 0) return 1;
        int sum = 0;
        for (int i = 1; i <= c; ++i){
            sum += helper(n-1, i);
        }
        return sum;
    }
}
