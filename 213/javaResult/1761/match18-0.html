class Solution {
    public int countVowelStrings(int n) {
        return helper(n, 0);
    }
    
    int helper(int n, int prev) {
        if (n == 0){
            return 1;
        }
        int result = 0;
        for (int i = prev; i< 5; i++){
            result += helper(n-1, i);            
        }
        return result;
    }
}
