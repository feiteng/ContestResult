class Solution {
    public int countVowelStrings(int n) {
        return count(1, 0, n);
    }
    private int count(int start, 
int k, int n) {
        if (k == n) return 1;
        int res = 0;
        for (int x = start; x <= 5; x++) {
            res += count(x, k+1, n);
        }
        return res;
    }
}
