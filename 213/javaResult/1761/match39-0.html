class Solution {
    public int countVowelStrings(int n) {
        int[][] dp = new int[n+1][5]; //num of strings ending in vow of len i
        for(int i=0;i<5;i++) {
            dp[0][i] = 0;
            dp[1][i] = 1;
        }
        for(int i=2;i<=n;i++) {
            int sum = 0;
            for(int j=0;j<5;j++) {
                sum += dp[i-1][j];
                dp[i][j] = sum;
            }
        }
        int ans = 0;
        for(int i=0;i<5;i++) {
            ans+= dp[n][i];
        }
        
        return ans;     
        
    }
}
