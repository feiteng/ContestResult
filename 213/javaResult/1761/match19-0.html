class Solution {
    public int countVowelStrings(int n) {
        int[] cnt = new int[5];
        Arrays.fill(cnt, 1);
        for (int k = 1; k <= n; k++) {
            for (int i = 0; i < cnt.length; i ++) {
                for (int j = i + 1; j < cnt.length; j++) {
                    cnt[i] += cnt[j];
                }
            }
        }
        // for (int i = 0; i < cnt.length; i ++) {
        //     System.out.print(cnt[i]  + " ");
        // }
        // System.out.println();
        return cnt[0];
    }
}
