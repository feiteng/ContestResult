class Solution {
    public int countVowelStrings(int n) {
        int[][] memory = new int[5][n];
        Arrays.fill(memory[0], 1);
        for(int i = 1; i < 5; i++){
            memory[i][0] = memory[i-1][0];
            for(int j = 1; j < n; j++){
                memory[i][j] = memory[i][j-1]+memory[i-1][j];
            }
        }
        int sum = 0;
        for(int i = 0; i < 5; i++){
            sum += memory[i][n-1];
        }
        return sum;
    }
}
