class Solution {
    public int countVowelStrings(int n) {
        return cvs(n, 5);
    }
    
    public int cvs(int n, int types) {
        if(types == 1) {
            return 1;
        }
        
        if(n == 1) {
            return types;
        }

        int out = 0;
        
        for(int i = 0; i < types; i++) {
            out += cvs(n-1, types - i);
        }
        
        return out;
    }
}
