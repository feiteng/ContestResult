class Solution {
    public int countVowelStrings(int n) {
        return helper(n, 5);
    }
    private int helper(int n, int k) {
        if (n == 1) return k;
        int num = 0;
        for (int i = k; i >= 1; i--) {
            num += helper(n-1, i);
        }
        return num;
    }
}
