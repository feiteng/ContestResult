class Solution {
    public int countVowelStrings(int n) {
        int d[] = new int[5];
        Arrays.fill(d,1);
        
        for(int j=2;j<=n;++j){
            int nd[] = new int[5];
            for(int at=0;at<5;++at){
                for(int to=at;to<5;++to){
                    nd[to] += d[at]; 
                }
            }
            d = nd;
        }
        int s = 0;
        for(int j=0;j<5;++j){
            s += d[j];
        }
        return s;
    }
}
