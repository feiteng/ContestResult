class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        for (int i = 0; i < arr.length; ) {
            boolean flag = false;
            for (int j = 0; j < pieces.length; j++) {
                if (pieces[j][0] == arr[i]) {
                    flag = true;
                    for (int k = 0; k < pieces[j].length; k++) {
                        if (pieces[j][k] != arr[i]) {
                            return false;
                        }
                        i++;
                        if(i == arr.length) return true;
                    }
                }
            }
            if(!flag) return false;
        }
        return true;
    }
}
