class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        int i = 0;
        while (i < arr.length){
            boolean found = false;
            for (int j = 0; j < pieces.length; j++){
                if (pieces[j][0] != arr[i]) continue;
                
found = true;
                for (int k = 0; k < pieces[j].length; k++){
                    if (pieces[j][k] != arr[i]) return false;
                    i++;
                }
                break;
            
}
            if (!found) return false;
        }
        return true;
    }
}
