class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        int i = 0;
        while (i < arr.length) {
            boolean found = false;
            for (int[] p : pieces) {
                if (p[0] == arr[i]) {
                    for (int j = 0; j < p.length; ++j) {
                        if (arr[i] != p[j]) {
                            return false;
                        }
                        ++i;
                    }
                    found = true;
                    
break;
                }
            }
            if (!found) {
                return false;
            }
        }
        return true;
    }
}
