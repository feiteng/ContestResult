class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> index = new HashMap<>();
        for (int i = 0; i < pieces.length; i++) {
            for (int v : pieces[i]) {
                index.put(v, i);
            }
        
}
        
        for (int i = 0; i < arr.length; ) {
            //int j = i;
            if (!index.containsKey(arr[i])) return false;
            int id = index.get(arr[i]);
            if (arr.length - i < pieces[id].length) return false;
            
            
for (int k = 0; k < pieces[id].length; k++) {
                if (pieces[id][k] != arr[i + k]) return false;
            }
            
            i += pieces[id].length;
        }
        return true;
    }
}
