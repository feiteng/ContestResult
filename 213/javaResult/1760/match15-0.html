class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        int idx = 0;
        while(idx<arr.length){
            int last = idx;
            
            for(int i=0;i<pieces.length;i++){

                if(pieces[i][0]==arr[idx]){
                    for(int j=0;j<pieces[i].length;j++){
                        if(pieces[i][j]!=arr[idx]){
                            return false;
                        }
                        idx++;
                        
                    }
                    break;
                }
            }
            
            if(last==idx){
                return false;
            }
        }
        return true;
    }
}
