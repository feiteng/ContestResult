class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for(int i=0; i<arr.length; i++){
            map.put(arr[i], i);
        }
        for(int i=0; i<pieces.length; i++){
            int a0 = pieces[i][0];
            if(!map.containsKey(a0)){
                return false;
            }
            int id0 = map.get(a0);
            for(int j=1; j<pieces[i].length; j++){
                if((!map.containsKey(pieces[i][j])) || map.get(pieces[i][j]) != id0+j){
                    return false;
                }
            }
        }
        return true;
    }
}
