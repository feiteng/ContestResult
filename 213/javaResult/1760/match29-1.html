class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        for(int i = 0; i < arr.length; i++) {
            boolean found = false;
            for(int[] piece : pieces) {
                if(arr[i] == piece[0]) {
                    found = true;
                    for(int j = 0; j < piece.length; j++) {
                        if(arr[i + j] != piece[j]) 
return false;
                    }
                    i += piece.length - 1;
                }
            }
            if(!found) return false;
        }
        return true;
    }
}
