class Solution {
    public boolean canFormArray(int[] arr, int[][] ps) {
        Map<Integer, Integer> ind = new HashMap<>();
        
        for (int i = 0; i < ps.length; i++) {
            ind.put(ps[i][0], i);
        }
        
        int i = 0;
        while (i < arr.length) {
            int tmp = ind.getOrDefault(arr[i], -1);
            if (tmp < 0) {
                return false;
            }
            for (int j = 0; j < ps[tmp].length; j++) {
                if (arr[i] != ps[tmp][j]) {
                    return false;
                }
                i++;
            }
        }
        
        return true;
    }
}
