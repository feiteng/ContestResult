class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> map = new HashMap();
        for (int i = 0; i < arr.length; i++) {
            map.put(arr[i], i);
        }
        for (int[] p : pieces) {
            if (!map.containsKey(p[0])) {
                return false;
            }
            int idx = map.get(p[0]);
            for (int i = 1; i < p.length; i++) {
                if (!map.containsKey(p[i]) || map.get(p[i]) - idx != i) {
                    return false;
                }
            }
        }
        return true;
    }
}
