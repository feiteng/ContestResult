class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> idx = new HashMap<>();
        
        for(int i = 0; i < pieces.length; ++i)
            idx.put(pieces[i][0], i);
        
        // System.out.println(idx);
        
        int i = 0;
        while(i < arr.length) {
            // System.out.println(i + ":" + arr[i]);
            if(!idx.containsKey(arr[i])) return false;
            int find = idx.get(arr[i]);
            for(int n : pieces[find]) {
                // System.out.println(String.format("i: %d n: %d ", i, n));;
                if(i >= arr.length || n != arr[i])
                    return false;
                i++;
            }
        }
        
        return true;
    }
}
