class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> index = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            index.put(arr[i], i);
        
}
        for (int[] p : pieces) {
            if (!index.containsKey(p[0])) return false;
            int idx = index.get(p[0]);
            for (int j = 1; j < p.length; j++) {
                if (idx + j >= arr.length || arr[idx + j] != p[j]) return false;
            }
        }
        return true;
    }
}
