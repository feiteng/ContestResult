class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, int[]> l = new HashMap<>();
        for (int[] p : pieces) {
            l.put(p[0], p);
        }
        
        int cur = 0;
        for (; cur < arr.length; ) {
            if (! l.containsKey(arr[cur])) {
                return false;
            }
            
            int[] a = l.get(arr[cur]);
            for (int i = 0; i < a.length; i ++) {
                if (arr[cur + i] != a[i]) {
                    return false;
                }
            }
            
            cur += a.length;
        }
        
        return true;
    }
}
