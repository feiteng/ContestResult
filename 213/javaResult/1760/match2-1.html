class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, int[]> map = new HashMap<>();
        for (int[] piece : pieces) {
            map.put(piece[0], piece);
        }
        int i = 0;
        while (i < arr.length) {
            int[] piece = map.get(arr[i]);
            if (piece == null) {
                return false;
            }
            for (int j = 0; j < piece.length; j++) {
                if (arr[i++] != piece[j]) {
                    return false;
                }
            }
        }
        return true;
    }
}
