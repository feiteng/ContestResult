class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        HashMap<Integer, int[]> map = new HashMap<Integer, int[]>();
        for(int[] list : pieces){
            map.put(list[0], list);
        }
        int i = 0;
        while(i < arr.length){
            int start = arr[i];
            
if(!map.containsKey(start)) return false;
            int[] list = map.get(start);
            for(int j : list){
                if(arr[i] != j) return false;
                i ++;
            }
        }
        return true;
    }
}
