class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, int[]> map = new HashMap<>();
        for(int[] piece : pieces)
            map.put(piece[0], piece);
        
int pos = 0, cnt = 0;
        while(pos < arr.length) {
            if(!map.containsKey(arr[pos])) return false;
            int[] piece = map.get(arr[pos]);
            for(int i = 0; i < piece.length; i++) {
                if(pos == arr.length || arr[pos] != piece[i]) return false;
                pos++;
            }
            cnt++;
        }
        return cnt == pieces.length;
    }
}
