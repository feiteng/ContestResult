class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> hash = new HashMap<>();
        
        for (int i = 0; i < pieces.length; i++) {
            hash.put(pieces[i][0], i);
        }
        
        int k = 0;
        
        for (; k < arr.length; ) 
{
            if (!hash.containsKey(arr[k])) return false;
            
            int idx = hash.get(arr[k]);
            
            for (int i = 0; i < pieces[idx].length; i++) {
                if (arr[k] != pieces[idx][i]) return false;
                k++;
            }
        }
        
        return true;
    }
}
