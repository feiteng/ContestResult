class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        HashMap<Integer, Integer> hMap = new HashMap<>();
        for (int i = 0; i < pieces.length; i++) {
            hMap.put(pieces[i][0], i);
        }
        int i = 0;
        while (i < arr.length) {
            Integer idx = hMap.get(arr[i]);
            if (idx == null) {
                return false;
            }
            for (int j = 0; j < pieces[idx].length; j++, i++) {
                if (i == arr.length || arr[i] != pieces[idx][j]) {
                    return false;
                }
            }
        } 
        return true;
    }
}
