class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {
        Map<Integer, Integer> val2idx = new HashMap<>();
        for(int i = 0; i < pieces.length; ++i) {
            val2idx.put(pieces[i][0], i);
        }
        for(int i = 0; i < arr.length; ++i) {
            if(val2idx.get(arr[i]) == null) return false;
            int idx = val2idx.get(arr[i]), sz = pieces[idx].length;
            for(int j = 0; j < sz; ++j) {
                if(pieces[idx][j] != arr[i++])
                    return false;
            }
            i--;
        }
        return true;
    }
}
