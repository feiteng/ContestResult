import java.util.*;

class Solution {
    public boolean canFormArray(int[] arr, int[][] pieces) {

        Map<Integer, int[]> firstToPiece = new HashMap<>();
        for(int[] piece : pieces){
            firstToPiece.put(piece[0], piece);
        }
        int idx = 0;
        while (idx < arr.length) {
            if (!firstToPiece.containsKey(arr[idx])) { return false; }
            int[] piece = firstToPiece.get(arr[idx]);
            // checking pieceIdx=0..len(piece)-1  matches  arrIdx=idx..idx+len(piece)-1  (incl)
            for(int pieceIdx = 0; pieceIdx < piece.length; pieceIdx++) {
                int arrIdx = pieceIdx + idx;
                if (arr[arrIdx] != piece[pieceIdx]) { return false; }
            }
            idx+= piece.length;
        }
        return true;
    }
}
