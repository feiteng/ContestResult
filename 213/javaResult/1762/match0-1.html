class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int l = 0, h = heights.length - 1, ans = 0, m = 0;
        while (l <= h) {
            m = l + (h-l)/2;
           // System.out.println(m);
   		    boolean canReach = canReachBuldingK(heights, ladders, bricks, m);
   		    if (canReach) l = m + 1;
   		    else h = m - 1;
   		    if (canReach) ans = Math.max(ans, m);
   	    }
   	    return ans;
   }

   private boolean canReachBuldingK(int[] heights, int ladders, int bricks, int k) {
       //System.out.println(k);
   	    PriorityQueue<Integer> pq = new PriorityQueue<>((i1, i2) -> -1 * Integer.compare(i1, i2));
   	    for (int i = 1; i <= k; i++) {
   		    if (heights[i-1] < heights[i]) pq.add(heights[i] - heights[i-1]);
   	    }

   	    while (!pq.isEmpty()) {
   		    if (ladders > 0) {
   			    pq.poll();
   			    ladders--;
   			    continue;
            }
   	        if (bricks > 0 && pq.peek() <= bricks) {
   		        int removed = pq.poll();
   			    bricks -= removed;
   			    continue;
   		    }
   		    return false;
   	    }
   	    return true;
   }
}
