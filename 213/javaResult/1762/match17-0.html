class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int l = 0;
        int r = heights.length-1;
        while(l < r){
            int mid = (l+r+1)/2;
            ArrayList<Integer> jumps = new ArrayList<Integer>();
            for(int j = 1;j<=mid;j++){
                if (heights[j] > heights[j-1]){
                    jumps.add(heights[j] - heights[j-1]);
                }
            }
            Collections.sort(jumps);
            int maxindex = jumps.size()-ladders;
            long sum = 0;
            for(int j = 0;j<maxindex;j++){
                sum+=jumps.get(j);
            }
            if (bricks >= sum){
                l = mid;
            }
            else{
                r = mid-1;
            }
        }
        return l;
    }
}
