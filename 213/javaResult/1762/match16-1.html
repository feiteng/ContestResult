class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> (a-b));
        
for (int i = 0; i+1 < heights.length; i++) {
            int gap = heights[i+1] - heights[i];
            if (gap <= 0) {
                continue;
            }
            
            pq.add(gap);
            //need to use brick
            if (pq.size() > ladders) {
                int curr = pq.remove();
                bricks -= curr;
            }
            if (bricks < 0) {
                return i;
            }
        }
        return heights.length-1;
    }
}
