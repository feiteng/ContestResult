class Solution {
    int[] heights;
  int n;
  int bricks;
  int laddgers;
  public int furthestBuilding(int[] heights, int bricks, int ladders) {
    n = heights.length;
    this.heights = heights;
    this.bricks = bricks;
    this.laddgers = ladders;
    int left = 0;
    int right = n;
    while (left < right) {
      int mid = (left + right) / 2;
      if (check(mid)) {
        left = mid + 1;
      } else {
        right = mid;
      }
    }
    return left - 1;
  }

  boolean check(int dist) {
    List<Integer> diff = new ArrayList<>();
    for (int i = 0; i < dist; i++) {
      if (heights[i] < heights[i + 1]) {
        diff.add(heights[i + 1] - heights[i]);
      }
    }
    Collections.sort(diff);
    if (diff.size() <= laddgers) {
      return true;
    }
    long need = 0;
    for (int i = 0; i < diff.size() - laddgers; i++) {
      need += diff.get(i);
    }
    return need <= bricks;
  }
}
