class Solution {
    public int furthestBuilding(int[] H, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>( (a, b) -> {
            return a - b;
        });
        
int cnt = 0;
        for (int i = 1; i < H.length; i++) {
            int dis = H[i] - H[i - 1];
            if (dis > 0) {
                pq.add(dis);
                if (pq.size() > ladders) {
                    cnt += pq.poll();
                }
                if (cnt > bricks) return i - 1
;
            }
        }
        return H.length - 1;
    }
}
