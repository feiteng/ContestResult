class Solution {
    public int furthestBuilding(int[] hs, int bs, int ls) {
        
        int lo = 0;
        int hi = hs.length-1;
        
        while(lo<hi){
            int mid = (lo+hi+1)>>1;
            List<Integer> li = new ArrayList<>();
            for(int j=1;j<=mid;++j){
                if(hs[j]>hs[j-1]){
                    li.add(hs[j]-hs[j-1]);
                }
            }
            Collections.sort(li);
            int ok = 1;
            int bbs = bs;
            int lls = ls;
            for(int j=0;j<li.size();++j){
                int v =  li.get(j);
                if(bbs>=v){
                    bbs-=v;
                }else if(lls>0){
                    lls--;
                }else{
                    ok = -1;
                    break;
                }
            
}
            if(ok==1){
                lo =mid;
            }else{
                hi =  mid-1;
            }
            
            
        }
        return lo;
        
        
    }
}
