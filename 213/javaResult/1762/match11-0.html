class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        Queue<Integer> q = new PriorityQueue<>(ladders + 1);
        int used = 0;
        int i;
        for (i = 1; i < heights.length; ++i) {
            int d = heights[i] - heights[i - 1];
            if (d > 0) {
                q.offer(d);
                if (q.size() > ladders) {
                    used += q.poll();
                    if (used > bricks) break;
                }
            }
        }
        // System.out.println(q + " " + used);
        return i - 1;
    }
}
