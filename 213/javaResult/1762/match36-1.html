class Solution {
     public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int left = 0;
        int right = heights.length - 1;
        while (right - left > 1) {
            int mid = left + (right - left) / 2;
            if (canReach(heights, bricks, ladders, mid)) {
                left = mid;
            } else {
                right = mid - 1;
            }
        }
        if (canReach(heights, bricks, ladders, right)) return right;
        if (canReach(heights, bricks, ladders, left + (right - left) / 2)) return left + (right - left) / 2;
        return left;
    
}

    private boolean canReach(int[] heights, int bricks, int ladders, int dst) {
        List<Integer> list = new ArrayList<>();
        for (int i = 1; i <= dst; i++) {
            if (heights[i] > heights[i - 1]) {
                list.add(heights[i] - heights[i - 1]);
            }
        }
        Collections.sort(list, (o1, o2) -> o2 - o1);
        int i = 0;
        while (i < list.size() && ladders > 0) {
            i++;
            ladders--;
        }
        if (i == list.size()) return true;
        while (i < list.size() && bricks >= list.get(i)) {
            bricks -= list.get(i);
            i++;
        }
        return i == list.size();
    }
}
