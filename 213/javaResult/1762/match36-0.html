class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int left = 0, right = heights.length;
        while (left + 1 < right) {
            int mid = (left + right) / 2;
            if (check(mid, bricks, ladders, heights)) {
                left = mid;
            } else {
                right = mid;
            }
        }
        return (check(right, bricks, ladders, heights) ? right : left) - 1;
    }

    boolean check(int mid, int bricks, int ladders, int[] heights) {
        List<Integer> diff = new ArrayList<>();
        for (int i = 1; i < mid; i++) {
            if (heights[i] > heights[i - 1]) {
                diff.add(heights[i] - heights[i - 1]);
            }
        }
        Collections.sort(diff);
        int sum = 0;
        for (int i = 0; i <  diff.size() - ladders; i++) {
            sum += diff.get(i);
        }
        return bricks >= sum;
    }
}
