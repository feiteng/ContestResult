class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int ind = 0;
        int N = heights.length;
        
PriorityQueue<Integer> pq = new PriorityQueue<Integer>((a, b) -> a - b);
        long bricksReq = 0;
        
        while (ind < N - 1) {
            int dif = heights[ind + 1] - heights[ind];
            
            if (dif <= 0) {
                ind++;
            }
            else {
                
pq.offer(dif);
                
                if (pq.size() > ladders) {
                    bricksReq += pq.poll();
                }
                
                if (bricksReq > bricks) {
                    return ind;
                
}
                
                ind++;
            }
        }
        
        return ind;
    }
    

}
