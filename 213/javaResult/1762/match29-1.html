class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        return sln1(heights, bricks, ladders);
    }
    
    
private int sln1(int[] hs, int bs, int ls){
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int sum = 0;
        for(int i = 1;i< hs.length;i++){
            if(hs[i-1] >= hs[i]) continue;
            int diff = hs[i]-hs[i-1];
            pq.offer(diff);
            if(pq.size() > ls){
                sum+=
pq.poll();
                if(sum > bs) return i-1;
            }
        }
        return hs.length-1;
    }
    
}
