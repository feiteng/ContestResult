class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int cur = 0;
        int rm = bricks;
        while(cur < heights.length - 1) {
            int a = heights[cur];
            int b = heights[cur + 1];
            if(a >= b) {
                ++cur;
            } else {
                int diff = b - a;
                if(pq.size() < ladders) {
                    pq.add(diff);
                    ++cur;
                } else {
                    int top = Integer.MAX_VALUE;
                    if(ladders != 0) {
                        top = pq.peek();
                    }
                    if(top < diff) {
                        pq.poll();
                        pq.add(diff);
                        diff = top;
                    }
                    if(rm >= diff) {
                        ++cur;
                        rm -= diff;
                    } else 
{
                        return cur;
                    }
                }
            }
        }
        return cur;
    }
}
