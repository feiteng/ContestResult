class Solution {

    private boolean check(int[] heights , int bricks , int ladders , int idx) {
        
List<Integer> valueList = new ArrayList<>();
        for (int i = 0;i < idx;i ++) {
            if (heights[i] < heights[i + 1]) {
                valueList.add(heights[i + 1] - heights[i]);
            }
        }
        Collections.sort(valueList, Collections.reverseOrder());
        // skip ladders
        for (int i = ladders;i < valueList.size();i ++) {
            int value = valueList.get(i);
            if (bricks < value) {
                return false;
            } else {
                bricks -= value;
            }
        }
        return true;
    }

    
public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int low = 1 , high = heights.length;
        while (low < high) {
            int mid = (low + high) / 2;
            if (check(heights , bricks , ladders , mid)) 
{
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return high - 1;
    }

}
