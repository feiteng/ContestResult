class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int out = 0;
        
        while(out < heights.length - 1) {
            if(heights[out+1] > heights[out]) {
                pq.add(heights[out+1] - heights[out]);
                
                if(pq.size() > ladders) {
                    bricks -= pq.poll();
                }
                
                if(bricks < 0) {
                    return out;
                }
            }
            
            out++;
        }
        
        return out;
    }
}
