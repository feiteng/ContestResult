class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> queue = new PriorityQueue<>(100, Collections.reverseOrder());
        int idx = 1;
        int sum = 0;
        while (idx < heights.length) {
            int diff = heights[idx] - heights[idx - 1];
            if (diff > 0) {
                queue.add(diff);
                sum += diff;
                if (sum > bricks) {
                    sum -= queue.poll();
                    ladders--;
                    if (ladders < 0) break;
                
}
            }
            idx++;
        }
        
        return idx - 1;
    }
}
