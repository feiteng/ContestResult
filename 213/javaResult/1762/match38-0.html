class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
int n = heights.length;
        int l = 0;
        int r = n;
        while (l < r) {
            int m = (l + r) / 2;
            int lad = ladders;
            PriorityQueue<Integer> queue = new PriorityQueue<>();
            long use = 0;
            for (int i = 1; i <= m; i++) {
                if (heights[i] > heights[i - 1]) {
                    queue.offer(-heights[i] + heights[i - 1]);
                    use += heights[i] - heights[i - 1];
                }
            }
            while (use > bricks && queue.size() > 0 && lad > 0) {
                lad--;
                use += queue.poll();
            }
            if (use > bricks) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return l - 1;
    }
}
