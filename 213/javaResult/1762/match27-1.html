class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int start = 0;
        int end = heights.length -1;
        int ans = 0;
        while (start <= end) {
            int mid = start + (end - start)/2;
            if (isPossible(mid, heights, bricks, ladders)) {
                ans = mid;
                
start = mid + 1;
            } else {
                end = mid -1;
            }
        }
        return ans;
    }
    
    private boolean isPossible(int target, int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        
for (int i = target; i>0; i--) {
            if (heights[i] - heights[i-1] > 0) {
                pq.add(heights[i] - heights[i-1]);
            }
        }
        for (int i = 0; i<ladders && pq.size() > 0; i++) {
            pq.poll();
        }
        
        while (pq.size() > 0) {
            if (pq.peek() > bricks) {
                return false;
            }
            bricks -= pq.poll();
        }
        return true;
    }
    
    
}
