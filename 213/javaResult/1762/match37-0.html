class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> q = new PriorityQueue<>();//how to use??
        for (int i = 1; i < heights.length; ++i) {
            int d = max(heights[i] - heights[i - 1], 0);
            if (d == 0) continue;
            if (ladders > 0) {
                ladders--;
                q.add(d);
            } else {
                int tp = (q.isEmpty() ? Integer.MAX_VALUE : q.peek());
                if (d > tp) {
                    if (bricks >= tp) {
                        bricks -= tp;
                        q.poll();
                        q.add(d);
                    } else {
                        return i-1;
                    }
                } else {
                    if (bricks >= d) {
                        bricks -= d;
                    } else {
                        return i-1;
                    }
                }
            }
        }
        return heights.length-1;
    }
}
