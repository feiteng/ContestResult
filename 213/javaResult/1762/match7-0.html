class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();        
        int index = 1;
        int total = 0;
        int val = 0;
        while (index<heights.length) {
            int diff = heights[index]-heights[index-1];
            if (diff>0) {
                pq.add(diff);
                total += diff;
                val += diff;
                if (pq.size() > ladders) {
                    val -= pq.poll();
                }
                
                if (bricks < total - val) {
                    break;
                }
            }
            index++;
        }
        
        return index-1;
    }
}
