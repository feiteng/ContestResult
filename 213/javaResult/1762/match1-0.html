class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int s = 0, e = heights.length - 1, ans = 0, m = 0;
   	    while (s <= e) 
        {
   		    m = s + (e-s)/2;
   		    boolean canReach = ok(heights, ladders, bricks, m);
   		    if (canReach) 
            {
                s = m + 1;
            }
   		    else 
            {
                e = m - 1;
            }
   		    if (canReach) 
            {
                ans = Math.max(ans, m);
            }
   	    }
   	    return ans;
    }
    boolean ok(int[] heights, int ladders, int bricks, int k) 
    {
   	    PriorityQueue<Integer> q = new PriorityQueue<>((a,b) -> -1 * Integer.compare(a, b));
   	    for (int i = 1; i <= k; i++) 
        {
   		    if (heights[i-1] < heights[i]) 
            {
                q.add(heights[i] - heights[i-1]);
            }
   	    }

    	while (!q.isEmpty()) 
        {
   		    if (ladders > 0) 
            {
   			    q.poll();
   			    ladders--;
   			    continue;
   		    }
   		    if (bricks > 0 && q.peek() <= bricks) 
            {
   			    int removed = q.poll();
   			    bricks -= removed;
   			    continue;
   		    }
   		    return false;
   	    }
   	    return true;
   }
}
