class Solution {
    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        int n = heights.length;
        // Integer[][][] mem = new Integer[n][bricks + 1][ladders + 1];
        // return dfs(0, heights, bricks, ladders, mem);
        int s = 0;
        int e = n - 1;
        while (s <= e) {
            int mid = s + (e - s) / 2;
            if (canReach(mid, heights, bricks, ladders)) 
{
                s = mid + 1;
            } else {
                e = mid - 1;
            }
        }
        return e;
        
    }
    
    boolean canReach(int target, int[] heights, int bricks, int ladders) {
        Queue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        
for (int i = 0; i<target; i++) {
            if (heights[i + 1] - heights[i] > 0) {
                q.offer(heights[i + 1] - heights[i]);
            }
        }
        for (int i = 0; i < ladders; i++) {
            q.poll();
        }
        int sum= 0;
        while (!q.isEmpty()) {
            sum += q.poll();
        }
        return bricks>= sum;
    }
