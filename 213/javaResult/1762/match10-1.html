class Solution {
    public int furthestBuilding(int[] h, int bricks, int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        int n = h.length;
        
for (int i = 1; i < n; i++) {
            if (h[i - 1] < h[i]) {
                int diff = h[i] - h[i - 1];
                if (bricks >= diff) {
                    // System.out.println(i + " " + bricks);
                    bricks -= diff;
                    pq.offer(diff);
                    // System.out.println(i + " " + bricks);
                } else if (ladders > 0) {
                    // System.out.println(i + " " + "l");
                    bricks -= diff;
                    pq.offer(diff);
                    bricks += pq.poll();
                    ladders -= 1;
                
} else {
                    return i - 1;
                }
            }
        }
        
        return n - 1;
    }
}
