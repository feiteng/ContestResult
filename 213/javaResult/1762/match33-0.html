class Solution {
    int[] h;
    int bricks;
    int ladders;

    public int furthestBuilding(int[] heights, int bricks, int ladders) {
        h = heights;
        this.bricks = bricks;
        this.ladders = ladders;
        int l = 0;
        int r = heights.length-1;
        while (l < r) {
            int mid = l + r + 1 >> 1;
            if (check(mid)) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        return l;
    }

    private boolean check(int mid) {
        PriorityQueue<Integer> pr = new PriorityQueue<>();
        int sum = 0;
        int last = Integer.MAX_VALUE;
        for (int i = 0; i <= mid; i++) {
            if (h[i] > last) {
                pr.add(h[i] - last);
                if (pr.size() > ladders) {
                    sum += pr.poll();
                }
            }
            last = h[i];
        }
        return sum <= bricks;
    }
}
