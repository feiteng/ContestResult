class Solution {
    public int furthestBuilding(int[] h, int b, int la) {
        int n = h.length;
        int l = 1;
        int r = n;
        int[] buf = new int[n];
        
while(l < r){
            int m = (l + r + 1) / 2;
            if(check(h, buf, m, b, la)){
                l = m;
            }else{
                r = m - 1;
            }
        }
        return l - 1;
    }
    
    public boolean check(int[] h, int[] buf, int n, int b, int l){
        int[] req = new int[n];
        buf[0] = 0;
        for(int i = 1; i < n; i++){
            buf[i] = Math.max(0, h[i] - h[i - 1]);
        }
        Arrays.sort(buf, 0, n);
        
long sum = 0;
        for(int i = 0; i < n - l; i++){
            sum += buf[i];
        }
        return sum <= b;
    }
}
