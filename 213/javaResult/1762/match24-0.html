	class Solution {
		public int furthestBuilding(int[] heights, int bricks, int ladders) {
			int n = heights.length;
			int low = 0, high = n+1;
			while(high - low > 1){
				int h = high+low>>1;
				if(ok(heights, h, bricks, ladders)){
					low = h;
				}else{
					high = h;
				}
			}
			return low-1;
		}

		boolean ok(int[] hs, int p, int B, int L)
		{
			int[] a = new int[p];
			int q = 0;
			for(int i = 0;i < p-1;i++){
				if(hs[i] < hs[i+1]){
					a[q++] = hs[i+1] - hs[i];
				}
			}
			a = Arrays.copyOf(a, q);
			Arrays.sort(a);
			if(L >= q)return true;
			long need = 0;
			for(int i = 0;i < q-L;i++){
				need += a[i];
			}
			return need <= B;
		}
	}
