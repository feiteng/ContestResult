class Solution {
    public int furthestBuilding(int[] heights, int bricks, 
int ladders) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int cur = heights[0];
        int sum = 0;
        for (int i = 1; i < heights.length; i++) {
            if (heights[i] > cur) {
                pq.add(heights[i]-cur);
                //System.out.println("add into queue :" + heights[i]);
                if (pq.size() > ladders) {
                    sum+=pq.poll();
                    //System.out.println("sum :" + sum);
                    if (sum > bricks) return i-1;
                
}
            }
            cur = heights[i];
        }
        return heights.length-1;
    }
}
