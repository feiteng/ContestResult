class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int H = destination[1], V = destination[0];
        int Total = H + V;
        int[][] nCr = new int[Total + 1][Total + 1];
        nCr[0][0] = 1;
        for(int r = 1; r <= Total; ++r){
            nCr[r][0] = nCr[r][r] = 1;
            for(int c = 1; c < r; ++c) {
                nCr[r][c] = nCr[r-1][c]+nCr[r-1][c-1];
            }
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 1; i <= Total; ++i) {
            if(k > nCr[Total - i][V]) {
                k -= nCr[Total - i][V];
                sb.append("V");
                V--;
            }else{
                sb.append("H");
                H--;
            }
        }
        return sb.toString();
    }
}
