class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int m = destination[0], n = destination[1];
        int[][] pascal = new int[31][31];
        pascal[0][0] = 1;
        for (int i = 1; i < 31; i++) {
            pascal[i][0] = 1;
            for (int j = 1; j < i; j++) {
                pascal[i][j] = pascal[i-1][j-1] + pascal[i-1][j];
            }
            pascal[i][i] = 1;
        
}
        
        dfs(m, n, k, pascal);
        return sb.toString();
    }
    
    private StringBuilder sb = new StringBuilder();
    
    // m: "V", n: "H"
    private void dfs(int m, 
int n, int k, int[][] pascal) {
        if (m == 0) {
            for (int i = 0; i < n; i++) {
                sb.append("H");
            }
            return;
        }
        
if (n == 0) {
            for (int i = 0; i < m; i++) {
                sb.append("V");
            }
            return;
        }
        
        int combs = pascal[m+n-1][m];
        
if (k > combs) {
            sb.append("V");
            dfs(m-1, n, k - combs, pascal);
        } else {
            sb.append("H");
            dfs(m, n-1, k, pascal);
        }
    }
}
