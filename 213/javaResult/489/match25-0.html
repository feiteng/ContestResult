class Solution {
    static int[][] dict;

    static {
        dict = new int[31][31];
        dict[0][0] = 1;
        for (int i = 1; i <= 30; i++) {
            dict[i][0] = 1;
            for (int j = 1; j <= i; j++)
                dict[i][j] = dict[i - 1][j] + dict[i - 1][j - 1];
        }
    }

    int calNum(int h, int v) {
        return dict[h + v][v];
    }

    public String kthSmallestPath(int[] destination, int k) {
        int h = destination[1], v = destination[0];
        StringBuilder sb = new StringBuilder();
        while (k > 0) {
            int left = calNum(h - 1, v);
            if (k - calNum(h - 1, v) > 0) {
                sb.append('V');
                k -= calNum(h-1, v);
                v--;
            } else {
                h--;
                sb.append('H');
            }
            if (h <= 0 || v <= 0) {
                break;
            }
        }
        while (h > 0) {
            sb.append('H');
            h--;
        }
        while (v > 0) {
            sb.append('V');
            v--;
        }

        return sb.toString();
    }
}
