class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int h = destination[1];
        int v = destination[0];
        int[][] dp = new int[h + 1][v + 1];
        //dp[i][j] -> C i, (i + j), dp[i][j] = 
        for (int i = 0; i <= h; i++) {
            for (int j = 0; j <= v; j++) {
                if (i == 0 || j == 0) {
                    dp[i][j] = 1;
                    continue;
                }
                dp[i][j] = dp[i - 1][j] * (i + j) / i;
            }
            // System.out.println(Arrays.toString(dp[i]));
        }
        
        
        //bs
        
        int hh = h - 1;
        int vv = v;
        StringBuilder sb = new StringBuilder();
        while (hh >= 0 && vv > 0) {
            int sum = dp[hh][vv];
            
if (k > sum) {
                sb.append('V');
                k -= sum;
                vv--;
            } else {
                sb.append('H');
                hh--;
            }
        }
        while (hh >= 0) {
            sb.append('H');
            hh--;
        }
        while (vv > 0) {
            sb.append('V');
            vv--;
        }
        return sb.toString();
    }
}
