class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int h = destination[1];
        int v = destination[0];
        int total = h + v;
        int before = 0;
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < total; i++) {
            if (h > 0 && v > 0) {
                int count = 
C(h+v-1, h-1);
                if (count >= k - before) {
                    sb.append("H");
                    h--;
                } else {
                    sb.append("V");
                    v--;
                    before += count;
                }
            } else if (h == 0 && v > 0){
                sb.append("V");
            } else if (h > 0 && v == 0) {
                sb.append("H");
            
}
        }
        return sb.toString();
    }
    
    public int C(long m, long n) {
        long i = 0;
        long a = 1;
        long b = 1;
        while(i < n) {
            a *= (m - i);
            i++;
        }
        
        long k = n;
        while(k > 1) {
            b *= k;
            k--;
        }
        return (int)(a/b);
    }
}
