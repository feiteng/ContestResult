class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int hCount = destination[1], vCount = destination[0];
        int[][] comb = new int[hCount + vCount + 1][hCount + vCount + 1];
        comb[0][0] = 1;
        comb[1][0] = 1;
        comb[1][1] = 1;
        for (int total = 2; total <= hCount + vCount; total++){
            comb[total][0] = 1;
            comb[total][total] = 1;
            for (int h = 1; h < total; h++){
                comb[total][h] = comb[total - 1][h] + comb[total - 1][h - 1];
            }
        }
        
        return kthPath(hCount, vCount, k, new StringBuilder(), comb);
    }
    
    private String kthPath(int h, int v, int k, StringBuilder path, int[][] comb){
        if (h == 0 && v == 0){
            return path.toString();
        }
        
        int hPaths = h == 0 ? 0 : comb[v + h - 1][h - 1];
        if (k <= hPaths){
            path.append('H');
            return kthPath(h - 1, v, k, path, comb);
        } else {
            path.append('V');
            return kthPath(h, v - 1, k - hPaths, path, comb);
        }
    }
}
