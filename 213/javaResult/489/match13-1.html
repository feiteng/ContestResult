class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        
int h = destination[1];
        int v = destination[0];
        int[][] C = new int[h+v+1][h+v+1];
        C[0][0] = 1;
        for (int i = 1; i <= h+v; i++) {
            C[i][0] = C[i][i] = 1;
            for (int j = 1; j < i; j++) {
                C[i][j] = C[i-1][j] + C[i-1][j-1];
            }
        }
        StringBuilder res = new StringBuilder();
        int tot = h + v;
        for (int i = 0; i < tot; i++) {
            if (h > 0) {
                if (k <= C[h+v-1][v]) {
                    res.append('H');
                    h--;
                } else {
                    k -= C[h+v-1][v];
                    res.append('V');
                    v--;
                }
            
} else {
                res.append('V');
                v--;
            }
        }
        return res.toString();
    }
}
