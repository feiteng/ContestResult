class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int v = destination[0] + 1;
        int h = destination[1] + 1;
        int[][] dp = new int[h][v];
        for (int[] temp : dp) Arrays.fill(temp, 1)
;
        for (int i=1; i<h; i++) {
            for (int j=1; j<v; j++) {
                dp[i][j] = dp[i-1][j] + dp[i][j-1];
            }
        }
        int vv = v - 1;
        int hh = h - 1;
        StringBuilder res = new StringBuilder();
        while (vv != 0 || hh != 0) {
            if (hh == 0 || k > dp[hh-1][vv]) {
                res.append("V");
                k -= (hh == 0 ? 0 : dp[hh-1][vv]);
                vv--;
            }
            else {
                
res.append("H");
                hh--;
            }
        }
        return res.toString();
    }
}
