class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int row = destination[0]+1, col = destination[1]+1;
        int[][] count = new int[row][col];
        for(int i = 0; i <row; i++){
            count[i][col-1]=1;
        }
        for(int j = 0; j<col; j++){
            count[row-1][j]=1;
        }
        for(int j = col-2; j>=0; j--){
            for(int i = row-2; i>=0; i--){
                count[i][j] = count[i+1][j] + count[i][j+1];
            }
        }
        int x = 0, y = 0;
        StringBuilder sb = new StringBuilder();
        while(k>=1){
            if(y==col-1 || k<=count[x][y+1]){
                if(y<col-1) {
                    sb.append("H");
                    y++;
                } 
                else {
                    sb.append("V");
                    x++;
                } 
            } else {
                
                k-=count[x][y+1];
                if(x<row-1) {
                    sb.append("V");
                    x++;
                }
                else {
                    sb.append("H");
                    y++;
                }
            }
            if(x==row-1 && y==col-1) break;
        }
        return sb.toString();
    }
}
