class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        
long [][] ncr = new long[31][31];
        for(int i = 0 ; i < 31 ; i++) {
            for(int j = 0 ; j <= i ; j++) {
                if(i==0) {
                    ncr[i][j] = 1;
                } else if (j==0) {
                   ncr[i][j] = 1;
                } else {
                     ncr[i][j] = (i * ncr[i-1][j-1] ) / j;
                
}
            }
        }
        StringBuilder sb = new StringBuilder("");
        int y = destination[0];
        int x = destination[1];
        int cx = 0;
        int cy = 0;
        int steps = x + y;
        while(steps>0) {
            if(x>0) {
                long totH = ncr[x+y-1][x-1];
                if(k>totH) {
                    k -= totH;
                    sb.append("V");
                    y--;
                } else {
                    sb.append("H");
                    x--;
                }
            
} else {
                sb.append("V");
            }
            steps--;
        }
        return sb.toString();
    }
}
