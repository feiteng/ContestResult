class Solution {
    public String kthSmallestPath(int[] d, int k) {
        int m = d[0]+1, n = d[1]+1;
        int[][] dp = new int[m][n];
        for (int i = 0; i < m; i++) dp[i][n-1] = 1;
        for (int j = 0; j < n; j++) dp[m-1][j] = 1;
        dp[m-1][n-1] = 0;
        for (int i = m-2; i >= 0; i--) {
            for (int j = n-2; j >= 0; j--) {
                dp[i][j] = dp[i+1][j] + dp[i][j+1];
            }
        }
        
        //for (int i = 0; i < m; i++) System.out.println(Arrays.toString(dp[i]));
        
        StringBuilder res = new StringBuilder();
        helper(k, res, dp, 0, 0, m, n);
        return res.toString();
    }
    
    private void helper(int k, StringBuilder sb, int[][] dp, int x, int y, int m, int n) {
        
        if (dp[x][y] == 1) {
            //if (k != 1) return;
            //System.out.println(x + " " + y);
            if (x == m-1)
                for (int j = y; j < n-1; j++) sb.append("H");

            if (y == n-1)
                for (int i = x; i < m-1; i++) sb.append("V");                
            return;
        }
        
        //System.out.println(x + " " + y + " " + k + " " + dp[x][y+1]);
        
        if (dp[x][y+1] >= k) { // go right
            helper(k, sb.append("H"), dp, x, y+1, m, n);
        } else { // go down
            helper(k - dp[x][y+1], sb.append("V"), dp, x+1, y, m, n);
        }        
    }
}
