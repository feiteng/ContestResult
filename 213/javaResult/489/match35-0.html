class Solution {
    
    private String res;
    
    public String kthSmallestPath(int[] destination, int k) {
        
        int M = destination[0], N = destination[1];
        
        int[][] stats = new int[M + 1][N + 1];
        stats[M][N] = 1;
        for (int i = 0; i <= M; ++i) stats[i][N] = 1;
        for (int i = 0; i <= N; ++i) stats[M][i] = 1;
            
        for (int i = M - 1; i >= 0; --i) {
            for (int j = N - 1; j >= 0; --j) {
                stats[i][j] = stats[i + 1][j] + stats[i][j + 1];
            }
        }
        
        helper(0, 0, k, stats, M, N, new StringBuilder());
        
        return res;
    }
    
    private void helper(int x, int y, int k, int[][] stats, int M, int N, StringBuilder sb) {
        
        if (stats[x][y] == 1) {
            while (x != M || y != N) {
                if (x + 1 <= M) {
                    sb.append("V");
                    x += 1;
                } else {
                    sb.append("H");
                    y += 1;
                }
            }
            res = sb.toString();
            return;
        }
        
        if (y + 1 <= N && k <= stats[x][y + 1]) {
            // System.out.println("x: " + x + ", y: " + y + ", path: " + stats[x][y + 1]);
            helper(x, y + 1, k, stats, M, N, sb.append("H"));
        } else {
            // System.out.println("x: " + x + ", y: " + y + ", path: " + stats[x + 1][y]);
            if (y + 1 <= N)
                helper(x + 1, y, k - stats[x][y + 1], stats, M, N, sb.append("V"));
            else
                helper(x + 1, y, k, stats, M, N, sb.append("V"));
        }
    }
}
