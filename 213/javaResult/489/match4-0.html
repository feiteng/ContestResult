class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int[][] dp = new int[16][16];
        for(int[] i : dp){
            Arrays.fill(i, 1);
        }
        
        for(int i = 1; i <= 15; i++){
            for(int j = 1; j <= 15; j++){
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        int y = destination[0];
        int x = destination[1];
        StringBuilder result = new StringBuilder();
        while(x > 0 && y > 0){
            if(dp[x - 1][y] >= k){
                result.append("H");
                x--;
            }else{
                result.append("V");
                k-=dp[x-1][y];
                y--;
            }
        }
            while( y > 0){
                result.append("V");
                y-=1;
            }
            while(x > 0){
               result.append("H");
                x-=1;
            }
    return result.toString();
    }
}
