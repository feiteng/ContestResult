class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int n = destination[0] + 1;
        int m = destination[1] + 1;
        int[][] a = new int[n][m];
        for (int i = 0; i < n; i++) {
            a[i][m - 1] = 1;
        }
        for (int j = 0; j < m; j++) {
            a[n - 1][j] = 1;
        }
        for (int i = n - 2; i >= 0; i--) {
            for (int j = m - 2; j >= 0; j--) {
                a[i][j] = a[i][j + 1] + a[i + 1][j];
            }
        }
        // for (int i = 0; i < n; i++) {
        //     for (int j = 0; j < m; j++) {
        //         System.out.print(a[i][j] + ",  ");
        //     }
        //     System.out.println();
        // }
        StringBuilder sb = new StringBuilder();
        int i = 0;
        int j = 0;
        while (i < n && j < m) {
            if (i == n - 1 && j == m - 1) {
                return sb.toString();
            }
            if (i == n - 1 || j + 1 < m && k <= a[i][j + 1]) {
                j++;
                sb.append('H');
            } else {
                if (j + 1 < m) {
                    k -= a[i][j + 1];
                }
                i++;
                sb.append('V');
            }
            
        }
        
        return sb.toString();
    }
}
