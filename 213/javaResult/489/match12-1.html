class Solution {
    
    
    public String kthSmallestPath(int[] d, int k) {
        int num  = d[0]+ d[1];
        
        long c[][] = new long[31][31];
        
        
for(int i=1;i<=30;++i){
            c[i][0] = 1;c[i][i] = 1;
            for(int j=1;j<i;++j){
                c[i][j] = c[i-1][j] + c[i-1][j-1];
            }
        }
        
        
        StringBuilder sb = new StringBuilder();
        
        int tot =  num;
        for(int j=0;j<num;++j){
            if(j==num-1){
                
                if(d[1]>0){
                    sb.append('H');
                }else{
                    sb.append('V');
                }
                break;
                
            }
            
            
            if(d[1]>0&&c[tot-1][d[1]-1]>=k){
                d[1]--;
                sb.append('H');
            }else{
                
                if(tot-1>0&&d[1]>0){
                    k-= c[tot-1][d[1]-1];
                }
                
sb.append('V');
                d[0]--;
            }
            tot--;
        }
        
        
        return sb.toString();
    }
}
