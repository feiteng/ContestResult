class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int m = destination[0] + 1;
        int n = destination[1] + 1;
        int[][] M = new int[m][n];
        for (int i = 0; i < m; i++) {
            M[i][n - 1] = 1;
        }
        for (int i = 0; i < n; i++) {
            M[m-1][i] = 1;
        }
        
for (int i = m - 2; i >= 0; i--) {
            for (int j = n - 2; j >= 0; j--) {
                M[i][j] = M[i + 1][j] + M[i][j + 1];
            }
        }
        StringBuilder sb = new StringBuilder();
        helper(M, 0, 0, k, sb);       
        return sb.toString();
    }
    private void helper(int[][] M, int row, int col, int k, StringBuilder sb) {
        if (row == M.length - 1 && col == M[0].length - 1) {
            return;
        }
        if (row == M.length - 1 ) {
            sb.append('H');
            helper(M, row, col + 1, k, sb);
        } else if (col == M[0].length - 1) {
            sb.append('V');
            helper(M, row + 1, col, k, sb);            
        } else {
            
if (k <= M[row][col + 1]) {
                sb.append('H');
                helper(M, row, col + 1, k, sb);
            } else {
                sb.append('V');
                helper(M, row + 1, col, k - M[row][col+1], sb);
            }            
        }
        return;
    }
}
