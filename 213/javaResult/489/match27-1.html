class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int m = destination[0] + 1, n = destination[1] + 1;
        int[][] dp = new int[m][n];
        for (int i = n - 2; i >= 0; i--) dp[m - 1][i] = 1;
        for (int i = m - 2; i >= 0; i--) dp[i][n - 1] = 1
;
        for (int i = m - 2; i >= 0; i--) {
            for (int j = n - 2; j >= 0; j--) {
                dp[i][j] = dp[i + 1][j] + dp[i][j + 1];
            }
        }
        /*for (int i = 0; i < dp.length; i++) {
            for (int j = 0; j < dp[0].length; j++) {
                System.out.print(dp[i][j] + ", ");
            }
            System.out.println();
        }*/
        char[] path = new char[m + n - 2];
        int r = 0, c = 0;
        for (int i = 0; i < m + n - 2; i++) {
            //System.out.println(new String(path));
            //System.out.println(k + ", " + r + ", " + c);
            if (r == m - 1 || (c < n - 1 && dp[r][c + 1] >= k)) {
                path[i] = 'H';
                c++;
            } else {
                path[i] = 'V';
                k -= c < n - 1 ? dp[r][c + 1] : 0;
                r++;
            }
        }
        return new String(path);
    }
}
