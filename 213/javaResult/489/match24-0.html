class Solution {
    public String kthSmallestPath(int[] dest, int k) {
        long x=dest[0],y=dest[1],n=x+y,val;
        StringBuilder sbr=new StringBuilder();
        for(long i=0;i<n;i++){
            val=nCr(x+y-1,x);
            // System.out.println(val+" "+k);
            if(x==0||(y>0&&val>=k)){
                sbr.append('H');
                y--;
            }
            else{
                k-=val;
                sbr.append('V');
                x--;
            }                
        }
        return sbr.toString();
    }
    public long nCr(long n, long k) 
    { 
        long res = 1l;   
        if (k > n - k) 
            k = n - k;         
        for (long i = 0; i < k; ++i) { 
            res *= (n - i); 
            res /= (i + 1); 
        } 
  
        return res; 
    } 
}
