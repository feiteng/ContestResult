class Solution {
	
	long nCr(int n, int r) {
		if(n - r < r) {
			r = n - r;
		}
        if(r == 0)
            return 1;
		//从n中选出r个数
		long ret = 1;
		for(int i = 0; i < r; ++i) {
            
			ret *= (n - i);
            ret /= (i + 1);
		}
        //System.out.println(n + " C " + r + " = " + ret);
		return ret;
	}

	
    public String kthSmallestPath(int[] destination, int k) {
    	//困难就这？
    	int Vcount = destination[0], Hcount = destination[1];
    	StringBuilder builder = new StringBuilder();
    	while(Vcount > 0 && Hcount > 0) {
    		long hBeg = nCr(Vcount + Hcount - 1, Vcount);
    		if(k > hBeg) {
    			builder.append('V
');
    			--Vcount;
    			k -= hBeg;
    		} else {
    			builder.append('H');
    			--Hcount;
    		}
    	}
    	while(Vcount > 0) {
    		--Vcount;
    		builder.append('V');
    	}
    	while(Hcount > 0) {
    		--Hcount;
    		builder.append('H');
    	}
    	return builder.toString();
    }
}
