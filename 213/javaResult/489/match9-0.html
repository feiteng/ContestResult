class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int[][] memo = new int[16][16];
        Arrays.fill(memo[0], 1);
        for (int i = 0; i < 16; i++) {
            memo[i][0] = 1;
        }
        for (int i = 1; i < 16; i++) {
            for (int j = 1; j < 16; j++) {
                memo[i][j] = memo[i - 1][j] + memo[i][j - 1];
            }
        }
        StringBuilder sb = new StringBuilder();
        int x = destination[0], y = destination[1];
        while (x > 0 && y > 0) {
            if (memo[x][y - 1] >= k) {
                sb.append("H");
                y--;
            } else {
                sb.append("V");
                k -= memo[x][y - 1];
                x--;
            }
        }
        while (x > 0) {
            sb.append("V");
            x--;
        }
        while (y > 0) {
            sb.append("H");
            y--;
        }
        return sb.toString();
    }
}
