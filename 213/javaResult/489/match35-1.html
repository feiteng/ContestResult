class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        StringBuilder sn=new StringBuilder();
        int x=destination[0],y=destination[1];
        
int x0=0,y0=0;
        int[][] dp=new int[x+1][y+1];
        dp[0][0]=1;
        for(int i=1;i<=y;i++) dp[0][i]=1;
        for(int i=1;i<=x;i++) dp[i][0]=1;
        for(int i=1;i<=x;i++){
            for(int j=1;j<=y;j++){
                dp[i][j]=dp[i-1][j]+dp[i][j-1];
            }
        }
        while(x0!=x||y0!=y){
            //先往水平走，看有几种走法
            int h=dp[x-x0][Math.max(0,y-y0-1)];
            
if(y>y0&&h>=k){
                y0++;
                sn.append('H');
            }
            else{
                x0++;
                k-=h;
                sn.append('V');
            }
        }
        return sn.toString();
    }
}
