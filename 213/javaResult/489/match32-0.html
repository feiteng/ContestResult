class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int[][] ways=new int[destination[0]+1][destination[1]+1];
        for(int i=ways.length-1;i>=0;i--){
            for(int j=ways[0].length-1;j>=0;j--){
                if(i==ways.length-1 || j==ways[0].length-1){
                    ways[i][j]=1;
                }else{
                    ways[i][j]=ways[i+1][j]+ways[i][j+1];
                }
            }
        }
        StringBuffer sb=new StringBuffer();
        int len=destination[0]+destination[1];
        int i=0;
        int j=0;
        while(len-->0){
            int right=0;
            if(j+1 < ways[0].length){
                right=ways[i][j+1];
            }
            if(right>=k){
                sb.append("H");
                j++;
            }else{
                sb.append("V");
                k-=right;
                i++;
            }
        }
        return sb.toString();
    }
}
