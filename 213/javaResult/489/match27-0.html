class Solution {
    
    public String kthSmallestPath(int[] destination, int k) {
        int v = destination[0];
        int h = destination[1];
        int[][] dp = new int[v+h+1][v+1+h];
        for (int i = 0; i <= v + h; i++) {
            dp[i][0] = 1;
            dp[i][i] = 1;
        }
        for (int i = 1; i <= v + h; i++) {
            for (int j = 1; j < i; j++) {
                dp[i][j] = dp[i-1][j] + dp[i-1][j-1];
            }
        }
        //System.out.println(dp[3][2]);
        char[] c = new char[v+h];
        int v1 = v;
        int h1 = h;
        for (int i = 0; i < c.length; i++) {
            //System.out.println(k);
            if (v1 == 0) {
                c[i] = 'H';
            } else if (h1 == 0) {
                c[i] = 'V';
            } else {
                if (dp[v1+h1-1][v1] >= k) {
                    c[i] = 'H';
                    h1--;
                } else {
                    c[i] = 'V';
                   
                    k -= dp[v1+h1-1][v1];
                     v1--;
                }
            }
            //System.out.println(c[i]);
        }
        return new String(c);
    }
    
    
}
