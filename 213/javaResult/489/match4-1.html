class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int n = destination[0]+1;
        int m = destination[1]+1;
        int[][] nums = new int[n][m];
        for(int i=0;i<n;i++){
            nums[i][m-1] = 1;
        }
        for(int i=0;i<m;i++){
            nums[n-1][i] = 1;
        
}
        for(int i=n-2;i>=0;i--){
            for(int j=m-2;j>=0;j--){
                nums[i][j] = nums[i+1][j] + nums[i][j+1];
            }
        }
        int i=0,j=0;
        StringBuffer ans = new StringBuffer();
        while(i<n-1 || j<m-1){
            if(nums[i][j+1] >= k){
                j++;
                ans.append("H");
            }else{
                k-=nums[i][j+1];
                i++;
                ans.append("V");
            }
            if(k==1){
                break;
            }
        
}
        while(j<m-1){
            ans.append("H");
            j++;
        }
        while(i<n-1){
            ans.append("V");
            i++;
        }
        return ans.toString();
    }
}
