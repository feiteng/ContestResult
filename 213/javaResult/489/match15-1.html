class Solution {


    public String kthSmallestPath(int[] destination, int k) {
        int[][] c = new int[31][31];
        c[0][0] = 1;
        for (int i = 1; i <= 30; i++) {
            
c[i][0] = 1;
            for (int j = 1; j <= i; j++) {
                c[i][j] = c[i - 1][j - 1] + c[i - 1][j];
            }
        }
        StringBuilder s = new StringBuilder();
        int H = destination[1], V = destination[0];
        int T = H + V;
        k--;
        
for (int i = 1; i <= T; i++) {
            if (k >= c[T - i][V]) {
                k -= c[T - i][V];
                s.append('V');
                V--;
            } else {
                s.append('H');
                H--;
            }
        }
        return s.toString();
    }


}
