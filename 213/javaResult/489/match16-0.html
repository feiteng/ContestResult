class Solution {
    int[][] c;

    public String kthSmallestPath(int[] destination, int k) {
        int h = destination[1];
        int v = destination[0];
        int t = h + v;
        c = new int[t + 1][t + 1];
        for (int i = 0; i <= t; i++) {
            for (int j = 0; j <= i; j++) {
                if (j == 0) c[i][j] = 1;
                else c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
            }
        }
        dfs(h, v, k);
        return sb.toString();
    }

    StringBuilder sb = new StringBuilder();

    public void dfs(int h, int v, int k) {
        if (v == 0) {
            for (int i = 0; i < h; i++) {
                sb.append("H");
            }
            return;
        } else if (h == 0) {
            for (int i = 0; i < v; i++) {
                sb.append("V");
            }
            return;
        }

       /* System.out.println(String.format("C(%s,%s) : %s", h - 1, v + h - 1, c[v + h - 1][h - 1]));
        System.out.println(k <= c[h][v + h]);*/
        if (h > 0 && k <= c[v + h - 1][h - 1]) {
            sb.append("H");
            dfs(h - 1, v, k);
        } else {
            sb.append("V");
            dfs(h, v - 1, k - c[v + h - 1][h - 1]);
        }
    }
}
