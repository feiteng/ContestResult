class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int row = destination[0] + 1, col = destination[1] + 1, y = destination[0], x = destination[1];
        int[][] tmp = new int[row][col];
        for (int i = 0; i < col; i++) {
            tmp[0][i] = 1;
        }
        for (int i = 1; i < row; i++) {
            tmp[i][0] = 1;
        }
        for (int i = 1; i < row; i++) {
            for (int j = 1; j < col; j++) {
                tmp[i][j] = tmp[i - 1][j] + tmp[i][j - 1];
            }
        }
        StringBuilder res = new StringBuilder(x + y);
        while (x > 0 || y > 0) {
            int t = x == 0 ? 0 : tmp[y][x - 1];
            if (t >= k) {
                res.append('H');
                x--;
            } else {
                res.append('V');
                y--;
                k -= t;
            }
        }
        return res.toString();
    }
}
