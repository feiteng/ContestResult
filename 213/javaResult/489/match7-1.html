class Solution {
    public String kthSmallestPath(int[] d, int k) {
        int m = d[0], n = d[1];
        int[][] ways = new int[m+1][n+1];
        for (int j = 0; j <= n; ++j) {
            ways[m][j] = 1;
        }
        for (int i = 0; i <= m; ++i) ways[i][n]=1
;
        for (int i = m-1; i>=0;--i) {
            for (int j = n-1;j>=0;--j) {
                ways[i][j] = ways[i+1][j] + ways[i][j+1];
            }
        }
        StringBuilder sb = new StringBuilder();
        int i = 0,j = 0;
        while (i != m  || j != n) {
            if (i == m) {
                ++j;
                sb.append('H');
            } else if (j == n) {
                ++i;
                sb.append('V');
            
} else {
                if (k <= ways[i][j+1]) {
                    sb.append('H');
                    ++j;
                } else {
                    sb.append('V');
                    k -= ways[i][j+1];
                    ++i;
                }
            }
        }
        return sb.toString();
    }
}
