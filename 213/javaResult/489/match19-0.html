class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int m = destination[0]+1;
        int n = destination[1]+1;
        int[][] tb = new int[m][n];
        for (int i=m-1; i>=0; i--)
            tb[i][n-1] = 1;
        for (int j=n-1; j>=0; j--) 
            tb[m-1][j] = 1;
        
        for (int i=m-2; i>=0; i--)
            for (int j=n-2; j>=0; j--) 
                tb[i][j] = tb[i+1][j] + tb[i][j+1];
        
        int ii = 0;
        int jj = 0;
        StringBuilder sb = new StringBuilder();
        
        while (ii!=m-1 || jj!=n-1) {
            //System.out.println(ii + " " + jj + " " + tb[ii][jj]);
            if (jj+1 <n && k <= tb[ii][jj+1]) {
                sb.append("H");
                jj++;
            }
            else {
                k -= (jj+1<n ? tb[ii][jj+1]:0);
                sb.append('V');
                ii++;
            }
        }
        
        return sb.toString();
    }
}
