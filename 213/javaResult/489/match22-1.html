class Solution {

    private static long[][] dp;

    static {
        dp = 
new long[20][20];
        for (int i = 0;i < 20;i ++) {
            dp[i][0] = 1;
        }
        for (int j = 0;j < 20;j ++) {
            dp[0][j] = 1;
        }
        for (int i = 1;i < 20;i ++) {
            for (int j = 1;j < 20;j ++) {
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
    }

    
public String kthSmallestPath(int[] destination, int k) {
        int n = destination[0] , m = destination[1];
        int x = 0 , y = 0;
        StringBuilder builder = new StringBuilder();
        long target = k;
        while (x < n || y < m) {
            if (x == n) {
                while (y < m) {
                    builder.append("H");
                    y ++;
                }
            } else if (y == m) {
                while (x < n) {
                    builder.append("V");
                    x ++;
                }
            } else {
                // try right
                long right = dp[n - x][m - y - 1];
                long down = dp[n - x - 1][m - y];
                
if (right >= target) {
                    builder.append("H");
                    y ++;
                } else {
                    target -= right;
                    builder.append("V");
                    x ++;
                }
            }
        }
        return builder.toString();
    }
}
