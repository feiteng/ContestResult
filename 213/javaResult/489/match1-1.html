class Solution {
    public String kthSmallestPath(int[] d, int k) {
        int h = d[1];
        int v = d[0];
        // memo = new HashMap<>();
        StringBuffer sb = new StringBuffer();
        
int[][] dp = new int[h + v + 1][h + v + 1];
        for (int i = 0; i <= h + v; i++) {
            for (int j = 0; j <= i; j++) {
                if (j == i || j == 0) {
                    dp[i][j] = 1;
                } else {
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - 1];
                }
            }
        }
        while (h > 0) {
            
int val = dp[h + v - 1][h - 1];
            if (val < k) {
                // V
                sb.append('V');
                k -= val;
                v--;
            } else {
                sb.append('H');
                h--;
            }
        }
        
        while (v > 0) {
            sb.append('V');
            v--;
        }
        
        return sb.toString();
    }
