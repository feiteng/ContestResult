class Solution {
  public String kthSmallestPath(int[] destination, int k) {
    int n = destination[0], m = destination[1];
    int[][] dp = new int[n + 2][m + 2];
    for(int i = n; i >= 0 ; i--){
      for(int j = m; j >= 0 ;j--){
        if(i == n && j == m){
          dp[i][j] = 1;
        } else {
          dp[i][j] = dp[i][j+1] + dp[i+1][j];
        }
      }
    }
    StringBuilder res = new StringBuilder();
    int i = 0, j = 0;
    while(i != n || j != m){
      int h = dp[i][j+1], v = dp[i+1][j];
      if(h < k){
        k -= h;
        res.append('V');
        i++;
      } else {
        res.append('H');
        j++;
      }
    }
    return res.toString();
  }
}
