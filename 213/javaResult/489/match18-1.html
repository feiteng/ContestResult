class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int v = destination[0];
        int h = destination[1];
        int[][] dp = new int[v+1][h+1];
        Arrays.fill(dp[0], 1);
        for(int i = 0; i <= v; i++){
            dp[i][0] = 1;
        }
        for(int i = 1; i <= v; i++){
            
for(int j = 1; j <= h; j++){
                dp[i][j] = dp[i-1][j]*(i+j) / i;
            }
        }
        return helper(new StringBuilder(), v, h, k, dp);
    }
    
    private String helper(StringBuilder s, int v, int h, int k, int[][] dp){
        if(k == 1){
            for(int i = 0; i < h; i++){
                s.append('H');
            }
            for(int i = 0; i < v; i++){
                s.append('V');
            }
            return s.toString();
        }
        else{
            int count = dp[v][h-1];
            
if(k <= count){
                s.append('H');
                return helper(s, v, h-1, k, dp);
            }
            else{
                k -= count;
                s.append('V');
                return helper(s, v-1, h, k, dp);
            }
        }
    }
}
