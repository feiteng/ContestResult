class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        return sln1(destination, k);
    }
    
    private String sln1(int[] des, int k){
        int vCount = des[0];
        int hCount = des[1];
        int n = vCount+hCount;
        
        int[][] tri = new int[n+1][n+1];
        
tri[0][0] = 1;
        for(int i =1;i< n+1;i++){
            tri[i][0] = 1;
            tri[i][i] = 1;
            for(int j = 1;j<i;j++){
                tri[i][j] = tri[i-1][j-1]+tri[i-1][j];
            }
        }
        
        // for(int i =0;i<tri.length;i++){
        //     for(int j = 0;j< tri[i].length;j++){
        //         System.out.printf("%d ", tri[i][j]);
        //     }
        //     System.out.println();
        // }
        return dfs(hCount, vCount , k, tri, new StringBuilder());
    }
    
    private String dfs(int hCount, int vCount, int k, int[][] tri, StringBuilder sb){
        //System.out.printf("h:%d, v:%d, sb:%s\n", hCount, vCount, sb.toString());
        if(hCount == 0 || vCount == 0) {
            while(hCount-- >0) sb.append("H");
            while(vCount-- >0) sb.append("V");
            return sb.toString();
        } 
        //System.out.printf("c(%d,%d):%d\n", vCount, vCount+hCount-1,tri[vCount+hCount-1][vCount]);
        
if(tri[vCount+hCount-1][vCount] < k){
            sb.append("V");
            return dfs(hCount, vCount-1, k-tri[vCount+hCount-1][vCount], tri, sb );
        }else{
            sb.append("H");
            return dfs(hCount-1, vCount, k, tri, sb );
        }
    }
}
