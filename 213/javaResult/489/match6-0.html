class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int M = 31;
    int[][] pascal = new int[M][M];
    for (int i = 0; i < M; i++) {
      pascal[i][0] = 1;
      pascal[i][i] = 1;
    }
    for (int i = 2; i < M; i++) {
      for (int j = 1; j < i; j++) {
        pascal[i][j] = pascal[i-1][j-1] + pascal[i-1][j];
      }
    }
    // System.out.println(Utils.trace(pascal, true));
    int m = destination[0];
    int n = destination[1];
    StringBuilder sb = new StringBuilder();
    int i = 0;
    int j = 0;
    while (i < m || j < n) {
      if (j == n) {
        sb.append('V');
        i++;
      } else if (i == m) {
        sb.append('H');
        j++;
      } else {
        /*
        System.out.format("  i:%d j:%d m:%d n:%d m-i:%d n-j-1:%d C(%d,%d)\n",
            i, j, m, n, m-i, n-j-1, m - i + n - j - 1, n-j-1);
            */
        int value = pascal[m - i + n - j - 1][n - j - 1];
        if (k <= value) {
          sb.append('H');
          j++;
        } else {
          k-= value;
          sb.append('V');
          i++;
        }
      }
    }
    return sb.toString();
    }
}
