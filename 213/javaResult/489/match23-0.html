class Solution {
    public String kthSmallestPath(int[] destination, int k) {
        int nV = destination[0];
        int nH = destination[1];
        int n= nV + nH;
        
        
        StringBuilder sb = new StringBuilder();
        
        while( nV != 0 && nH != 0) {
            long temp = cr(nV, nH-1);
            if(k > temp){
                k -= temp;
                nV--;
                sb.append("V");
            }
            else {
                sb.append("H");
                nH--;
            }
        }
        
        for(int i = 0; i < nV; i++) {
            sb.append("V");
        }
        for(int i = 0; i < nH; i++) {
            sb.append("H");
        }
        
        return sb.toString();
        
    }
    
    private long cr(long v,long h) {
        if( v == 0 || h == 0) {
            return 1;
        }
        long n = v + h;
        long min = Math.min(v, h);
        long max = Math.max(v, h);
        long sum = 1;
        for(long i = n; i > max; i--) {
            sum *= (long)i;
        }
        for(long j = min; j >=1; j--) {
            sum /= (long)j;
        }
        return sum;
    }
}
