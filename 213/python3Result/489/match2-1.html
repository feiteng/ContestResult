class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        def factorial(a, b):
            res = 1
            for i in range(1, a + b + 1):
                res = res * i
            for i in range(1, a + 1):
                res = res // i
            for i in range(1, b + 1):
                res = res // i
            return res

        ans = []

        def dfs(x, y, z):
            if x == 0 and y == 0:
                return
            elif x == 0:
                y -= 1
                ans.append("V")
                dfs(x, y, z)
            elif y == 0:
                x -= 1
                ans.append("H")
                dfs(x, y, z)
            else:
                if factorial(x - 1, y) > z:
                    ans.append("H")
                    dfs(x - 1, y, z)
                else:
                    ans.append("V")
                    dfs(x, y - 1, z - factorial(x - 1, y))

        dfs(destination[1], destination[0], k - 1)
        
        return ''.join(ans)
