from math import comb

class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        r, c 
= destination
        if r == 0:
            return 'H' * c
        if c == 0:
            return 'V' * r
        right = comb(r + c - 1, c - 1)
        if k <= right:
            return 'H' + self.kthSmallestPath([r, c-1], k)
        else:
            return 'V' + self.kthSmallestPath([r-1, c], k - right)
