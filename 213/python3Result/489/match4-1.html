from typing import List



class Solution:



    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        # Return (a+b)!/a!b!
        def factorial(a, b):

            res = 1

            # finding (a+b)!
            for i in range(1, a + b + 1):
                res = res * i

            # finding (a+b)!/a!
            for i in range(1, a + 1):
                res = res // i

            # finding (a+b)!/b!
            for i in range(1, b + 1):
                res = res // i

            return res

        ans = []
        # given coordinate from origin
        def Ksmallest(x, y, k):

            # if at origin
            if x == 0 and y == 0:
                return

            # if on y-axis
            elif x == 0:
                # decrement y.
                y -= 1

                # Move vertical
                ans.append("V")

                # recursive call to take next step.
                Ksmallest(x, y, k)

            # If on x-axis
            elif y == 0:

                # decrement x.
                x -= 1

                # Move horizontal.
                ans.append("H")

                # recursive call to take next step.
                Ksmallest(x, y, k)

            else:

                # If x + y C x is greater than K
                if factorial(x - 1, y) > k:

                    # Move Horizontal
                    ans.append("H")

                    # recursive call to take next step.
                    Ksmallest(x - 1, y, k)

                else:

                    # Move vertical
                    ans.append("V")

                    # recursive call to take next step.
                    Ksmallest(x, y - 1, k - factorial(x - 1, y))

                    # Driver Code
        Ksmallest(destination[1], destination[0], k - 1)
        return "".join(ans)
