from math import comb
class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        row, col = destination
        s = ""
        while row > 0 and col > 0:
            thresh = comb(row+col-1, col-1)
            if k <= thresh:
                s += "H"
                col -= 1
            else:
                s += "V"
                k -= thresh
                row -= 1
        s += "V" * row
        s += "H" * col
        return s
