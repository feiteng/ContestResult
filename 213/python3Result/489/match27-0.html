from math import comb

class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        if k == 0:
            return ""
        V_count, H_count = destination
        if V_count == 0:
            return "H" * H_count
        if H_count == 0:
            return "V" * V_count
        H_num = comb(H_count + V_count - 1, H_count - 1)
        if k > H_num:
            return "V" + self.kthSmallestPath([V_count - 1, H_count], k - H_num)
        else:
            return "H" + self.kthSmallestPath([V_count, H_count - 1], k)
