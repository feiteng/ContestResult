class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        
        
        
        r, c = destination
        dp = [[0]*(c+1) for _ in range(r+1)]  # dp[i][j] 表示从 ij 到 rc 的路径总数
        for i in range(r, -1, -1):
            for j in range(c, -1, -1):
                if i == r or j == c:
                    dp[i][j] = 1
                else:
                    dp[i][j] = dp[i+1][j] + dp[i][j+1]
        i, j, res = 0, 0, ''
        while (i,j) != (r,c):
            if j == c:
                res += 'V'*(r-i)
                return res
            if i == r:
                res += 'H'*(c-j)
                return res
            h = dp[i][j+1]
            if h >= k:
                res += 'H'
                j += 1
            else:
                res += 'V'
                i += 1
                k -= h
        return res
