class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        fact = lambda x,y: functools.reduce(operator.mul, range(x,y+1), 1)

        @lru_cache(None)
        def nCr(n, r):
            return fact(n-r+1,n) // fact(1,r)
        ans = ''
        r, c = destination
        while r > 0 and c > 0:
            n1 = nCr(r+c-1, r)
            # n2 = nCr(r+c-1, r-1)
            if k <= n1:
                ans += 'H'
                c -= 1
            else:
                ans += 'V'
                r -= 1
                k -= n1
        return ans + 'H' * c + 'V' * r
