class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        m, n 
= destination # m "V" & n "H" in total 
        ans = ""
        
        while n: 
            kk = comb(m+n-1, n-1)
            if kk >= k: 
                ans += "H"
                n -= 1
            else: 
                ans += "V"
                k -= kk 
                m -= 1
        return ans + m*"V"
