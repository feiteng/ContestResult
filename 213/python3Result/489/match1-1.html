class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        @lru_cache(None)
        def nCr(n, k):
            if k == 0 or k == n:
                return 1
            return nCr(n - 1, k - 1) + nCr(n - 1, k)
        
        ans = ''
        v, h = destination[0], destination[1]
        for i in range(v + h):
            if v == 0:
                ans += 'H'
                h -= 1
            elif h == 0:
                ans += 'V'
                v -= 1
            else:
                ncr = nCr(h + v - 1, h - 1)
                if k <= ncr:
                    ans += 'H'
                    h -= 1
                else:
                    ans += 'V'
                    v -= 1
                    k -= ncr
        
        return ans
