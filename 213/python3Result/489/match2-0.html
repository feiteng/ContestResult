class Solution:
    def kthSmallestPath(self, des: List[int], k: int) -> str:
        def factorial(a, b):  
            res = 1
            for i in range(1, a + b + 1):  
                res = res * i  
            for i in range(1, a + 1):  
                res = res // i  
            for i in range(1, b + 1):  
                res = res // i  

            return res  
        lst = []
        def Ksmallest(x, y, k):  
            if x == 0 and y == 0:  
                return
            elif x == 0: 
                y -= 1
                lst.append("V")
                Ksmallest(x, y, k)    
            elif y == 0: 
                x -= 1
                lst.append("H")
                Ksmallest(x, y, k)  
            else: 
                if factorial(x - 1, y) > k:  
                    lst.append("H")
                    Ksmallest(x - 1, y, k)  
                else:
                    lst.append("V")
                    Ksmallest(x, y - 1, k - factorial(x - 1, y))  
        Ksmallest(des[1], des[0], k - 1)
        return ''.join(lst)
