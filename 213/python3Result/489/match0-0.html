class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        # num_H = destination[1]
        # num_V = destination[0]
        # total = num_H+num_V 
        # res = ['H']*num_H+['V']*num_V
        # for i in range(2, k+1):
        #     old_res = res
        #     for j in range(total-1, 0, -1):
        #         if j >= 1 and (res[j]=='V' and res[j-1]=='H'):
        #             res[j], res[j-1] = res[j-1], res[j]
        #             break
        # return ''.join(res)
        v, h = destination
        dp = [[0]*(h+1) for _ in range(v+1)]  
        for i in range(v, -1, -1):
            for j in range(h, -1, -1):
                if i == v or j == h:
                    dp[i][j] = 1
                else:
                    dp[i][j] = dp[i+1][j] + dp[i][j+1]
        i, j, res = 0, 0, ''
        while (i,j) != (v,h):
            if j == h:
                res += 'V'*(v-i)
                return res
            if i == v:
                res += 'H'*(h-j)
                return res
            remain = dp[i][j+1]
            if remain >= k:
                res += 'H'
                j += 1
            else:
                res += 'V'
                i += 1
                k -= remain
        return res
