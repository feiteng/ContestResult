class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        while i < len(arr):
            curr = arr[i]
            #find idx in pieces starting with curr
            found = -1
            for start_idx, piece in enumerate(pieces):
                if piece[0] == curr:
                    found = start_idx
                    break
            if found == -1:
                return False
            can = pieces[found]
            #match additional
            if arr[i: i + len(can)] == can:
                i += len(can)
            else:
                return False
        return True
