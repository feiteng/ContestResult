class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        while i < len(arr):
            a = arr[i]
            findex = -1
            for j, sublist in enumerate(pieces):
                if sublist[0] == a:
                    findex = j
                    break
            if findex == -1: return False
            sublist = pieces.pop(j)
            for ne in sublist:
                if i >= len(arr): return False
                if ne != arr[i]: return False
                i += 1
        return True
