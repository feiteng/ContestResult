class Solution:
    def canFormArray(self, A: List[int], P: List[List[int]]) -> bool:
        i = 0
        while i < len(A):
            found = False
            for p in P:
                if A[i] == p[0]:
                    found = True
                    for x in p:
                        if x != A[i]:
                            return False
                        i += 1
                    P.remove(p)
                    break
            if not found:
                return False
        return True
