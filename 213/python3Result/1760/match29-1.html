class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        n=len(arr)
        dp=[False] * n
        dp.append(True)
        
        
for i in range(n-1, -1, -1):
            for j in range(n, i, -1):
                    if dp[j] and arr[i:j] in pieces:
                        dp[i]=True
                        break
        
        return dp[0]
