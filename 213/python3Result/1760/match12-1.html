class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        pieces_dict = dict()
        for piece in pieces:
            pieces_dict[piece[0]] = piece
        i = 0
        while i < len(arr):
            num = arr[i]
            if not (num in pieces_dict):
               return False
            piece = pieces_dict[num]
            for piece_num in piece:
                num = arr[i]
                if num == piece_num:
                    i += 1
                else:
                    return False
        return True
