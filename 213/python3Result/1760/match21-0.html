class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        n = len(arr)
        i = 0
        while i < n:
            j = i
            for p in pieces:
                if p == arr[i:i+len(p)]:
                    j = i + len(p)
            if i == j:
                return 0
            i = j
        return 1
