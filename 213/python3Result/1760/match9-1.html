class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        d = {}
        for i, piece in enumerate(pieces):
            d[piece[0]] = i
            
        cur = 0
        while cur < len(arr):
            first = arr[cur]
            if not first in d:
                return False
            for i, n in enumerate(pieces[d[first]]):
                if cur + i >= len(arr):
                    return False
                if arr
[cur+i] != n:
                    return False
            cur = cur + len(pieces[d[first]])
        return True
