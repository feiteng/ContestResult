class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        dic = {}
        for i, tmp in enumerate(pieces):
            dic[tmp[0]] = i
        
        i = 0
        while i < len(arr):
            num = arr[i]
            if num not in dic:
                return False
            tmp = pieces[dic[num]] 
            k = len(tmp)
            for j in range(k):
                if arr[i] != tmp[j]:
                    return False
                i += 1
                
        return True
