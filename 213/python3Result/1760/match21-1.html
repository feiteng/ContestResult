class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        s=0
        while s<len(arr):
            news
=s
            for x in pieces:
                if x==arr[s:(len(x)+s)]:
                    news=s+len(x)
            if news==s:
                return False
            s=news
        return True
