class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        pm = {}
        for p in pieces:
            pm[p[0]] = p
        idx = 0
        while idx < len(arr):
            if arr[idx] not in pm:
                #print(idx, 'hey')
                return False
            p = pm[arr[idx]]
            if idx + len(p) - 1 >= len(arr) or arr[idx : idx + len(p)] != p:
                #print(idx, 'yo')
                return False
            idx += len(p)
        return True
