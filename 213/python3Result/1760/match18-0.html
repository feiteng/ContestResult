class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        dic = { p[0]: p for p in pieces }
        
        i = 0
        while i < len(arr):
            if arr[i] not in dic:
                return False
            
            a = dic[arr[i]]
            for n in a:
                if n == arr[i]:
                    i += 1
                else:
                    return False
            
        return True
