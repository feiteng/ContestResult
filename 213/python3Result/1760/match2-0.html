class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        sw = {}
        for p in pieces:
            sw[p[0]] = p
        i = 0
        while i < len(arr):
            if arr[i] not in sw:
                return False
            curp = sw[arr[i]]
            for j in range(len(curp)):
                if arr[i+j] != curp[j]:
                    return False
            i += len(curp)
        return True
