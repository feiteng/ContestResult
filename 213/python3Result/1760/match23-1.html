class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        v = set()
        while i < len(arr):
            found = False
            for j in range(len(pieces)):
                if j in v:
                    continue
                if arr[i] == pieces[j][0]:
                    n = len(pieces[j])
                    if arr[i:i+n] != pieces[j]:
                        return False
                    i += n
                    v.add
(j)
                    found = True
                    break
            if not found:
                return False
        return True
