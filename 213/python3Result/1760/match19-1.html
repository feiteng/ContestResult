class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        firsts = {piece[0]:piece for piece in pieces}
        ptr = 0
        while ptr < len(arr):
            if arr[ptr] not in firsts:
                return False
            piece = firsts[arr[ptr]]
            piece_length = len(piece)
            if piece != arr[ptr:ptr+piece_length]:
                return False
            ptr += piece_length
        return True
