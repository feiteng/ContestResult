class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        seen = set()
        while i < len(arr):
            flag = True
            for pi, p in enumerate(pieces):
                if pi in seen:
                    continue
                if arr[i:i+len(p)] == p:
                    seen.add(pi)
                    i += len(p)
                    flag = False
                    break
            if flag:
                return False
        return True
