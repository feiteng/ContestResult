class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        d = {}
        for p in pieces:
            d[p[0]] = p
        i = 0
        l = len(arr)
        while i < l:
            val = arr[i]
            if val not in d:
                return False
            p = d
[val]
            for j in range(len(p)):
                if arr[i+j] != p[j]:
                    return False
            i += len(p)
        return True
