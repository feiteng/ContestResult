class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i=0
        n=len(arr)
        while i<n:
            
for sub in pieces:
                if sub==arr[i:i+len(sub)]:
                    i+=len(sub)
                    break
            else:
                return False
        return True
