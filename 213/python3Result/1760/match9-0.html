class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        pmap = {}
        for pidx, piece in enumerate(pieces):
            pmap[piece[0]] = pidx
        
        i = 0
        # try:
        while i < len(arr):
            num = arr[i]
            if num not in pmap:
                # print("A")
                return False
            for j, pnum in enumerate(pieces[pmap[num]]):
                if arr[i + j] != pnum:
                    return False
            # print(pieces[pmap[num]])
            # assert j > 0
            i += len(pieces[pmap[num]])
            # print(i)
        return True
