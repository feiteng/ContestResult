class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        starts = {}
        for index, piece in enumerate(pieces):
            starts[piece[0]] = index

        i = 0
        while i < len(arr):
            curr = arr[i]
            if curr not in starts:
                return False
            piece = pieces[starts[curr]]
            for num in piece:
                if num != arr[i]:
                    return False
                i += 1
            del starts[curr]
        return True
