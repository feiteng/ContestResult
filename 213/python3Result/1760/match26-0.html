class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        while i < len(arr):
            flag = False
            for j in pieces:
                if j[0] == arr[i]:
                    flag = True
                    for k in j:
                        if k != arr[i]:
                            return False
                        i += 1
                if i == len(arr):
                    return True
        
            if not flag: return False
        return True
