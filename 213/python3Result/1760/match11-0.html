class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        
        
        # the "characters" are 1 - 100. 100 possible characters
        
        # greedy solution should work
        
        
        stuff = dict()
        
        
        for i in pieces:
            stuff[i[0]] = i
            
            
        cur = 0
        
        while cur < len(arr):
            
            toMatch = arr[cur]
            
            if toMatch not in stuff:
                return False
            
            possible = stuff[toMatch]
            for i in possible:
                if arr[cur] != i:
                    return False
                cur+=1
            
            
        return True
