class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        index_map = {}
        for i in range(len(pieces)):
            index_map[pieces
[i][0]] = i
        i = 0
        while i < len(arr):
            if arr[i] not in index_map:
                return False
            cur_piece = pieces[index_map[arr[i]]]
            for j in range(len(cur_piece)):
                if i + j == len(arr):
                    return True
                
if arr[i + j] != cur_piece[j]:
                    return False
            i += len(cur_piece)
        return True
