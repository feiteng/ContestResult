class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        
        d = {}
        for p in pieces:
            d
[p[0]] = p
        
        i = 0
        while i < len(arr):
            if arr[i] not in d: return False
            match = d[arr[i]]
            for j in range(len(match)):
                
if match[j] != arr[i+j]: return False
            i += len(match)
        return True
