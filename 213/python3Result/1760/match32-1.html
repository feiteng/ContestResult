class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        
        dic={}
        
        for piece in pieces:
            
            dic[piece[0]]=piece
            
        
        i=0
        
        while i<len(arr):
            
            a=arr[i]
            
            if a not in dic:
                return False
            
            for j in range(len(dic[a])):
                
                
                if i>=len(arr):
                    return False
                
                if arr[i]!=dic[a][j]:
                    return False
                i+=1
            
        return True
