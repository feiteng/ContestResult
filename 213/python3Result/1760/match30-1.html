class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        s = {}
        
        for piece in pieces:
            s[piece[0]] = piece
            
        index = 0
        
        while index < len(arr):
            if arr[index] not in s:
                return False
            
            piece = s[arr[index]]
            for offset, x in enumerate(piece):
                if index+offset >= len(arr):
                    return True
                
                if arr[index+offset] != x:
                    return False
                
            index += len(piece)
            
        return True
