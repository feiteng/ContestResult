class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        start = {}
        for piece in pieces:
            start[piece[0]] = piece
        i = 0
        while i < len(arr):
            if arr[i] not in start:
                return False
            piece = start[arr[i]]
            if arr[i:i+len(piece)] != piece:
                return False
            i = i + len(piece)
        return True
