class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        i = 0
        while i != len(arr):
            for piece in pieces:
                if arr[i:i+len(piece)] == piece:
                    i += len(piece)
                    break
            else:
                return False
        return True
