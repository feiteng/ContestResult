class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        start_map = defaultdict(list)
        
for e in pieces:
            start_map[e[0]] = e
        i = 0
        while i < len(arr):
            #print(i)
            x = arr[i]
            if  x not in start_map: return False
            for e in start_map[x]:
                if e == arr[i]:
                    i += 1
                else: 
                    return False
            start_map[x] = []
        return True
