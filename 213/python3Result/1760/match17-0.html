class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        dic = {}
        for array in pieces:
            dic[array[0]] = array
        i = 0
        while i < len(arr):
            if arr[i] not in dic:
                return False
            length = len(dic[arr[i]])
            if arr[i:i + length] != dic[arr[i]]:
                return False
            i += length
        return True
