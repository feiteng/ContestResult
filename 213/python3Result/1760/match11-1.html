class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        head_vals = {} # head value -> entire list
        
for piece in pieces:
            head_vals[piece[0]] = piece
        idx = 0
        while idx < len(arr):
            curr_head = arr[idx]
            if curr_head not in head_vals:
                return False
            curr_list = head_vals[curr_head]
            for val in curr_list:
                if val != arr[idx]:
                    return False
                idx += 1
        return True
