class Solution:
    def countVowelStrings(self, n: int) -> int:
        from functools import lru_cache
        
        @lru_cache(None)
        def dfs(i, st):
            if i == n: return 1
            ans = 0
            for ch in range(st, 5):
                ans += dfs(i + 1, ch)
            return ans
            
            
        return dfs(0, 0)
