from functools import lru_cache
class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(None)
        def ways(c, ct):
            if ct == 1:
                return c
            return sum( ways(c1, ct - 1) for c1 in range(1, c+1) )
            
        
        return ways(5, n)
