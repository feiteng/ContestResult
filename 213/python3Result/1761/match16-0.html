class Solution:
    def countVowelStrings(self, n: int) -> int:
        from functools import lru_cache

        @lru_cache(None)
        def dfs(idx, current):
            if idx == n:
                return 1
            ans = 0
            for i in range(current, 5):
                ans += dfs(idx + 1, i)
            return ans

        return dfs(0, 0)
