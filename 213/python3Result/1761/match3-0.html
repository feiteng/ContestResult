class Solution:
    def countVowelStrings(self, n: int) -> int:
        dp = [[0]*5 for i in range(n)]
        dp[0] = [1]*5
        for i in range(1, n):
            for j in range(5):
                for k in range(j+1):
                    dp[i][j] += dp[i-1][k]
        return sum(dp[-1])
