class Solution:
    def countVowelStrings(self, n: int) -> int:
        def next_step(i, left_n):
            if left_n == 1:
                return i
            
            cur_count = 0
            for j in range(1, i + 1):
                cur_count += next_step(j, left_n - 1)
            return cur_count
        
        return next_step(5, n)
