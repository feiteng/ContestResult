class Solution:
    def countVowelStrings(self, n: int) -> int:
        from functools import lru_cache
        
        @lru_cache(None)
        def dp(left, start):
            if left <= 0:
                return 0
            if start >= n:
                return 1
            return dp(left, start+1) + dp(left-1, start)
        
        return dp(5, 0)
