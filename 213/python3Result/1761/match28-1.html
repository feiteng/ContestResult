class Solution:
    def countVowelStrings(self, n: int) -> int:
        import functools
        
        @functools.lru_cache(None)
        def find(n: int, k: int):
            if n < k:
                return 0
            if 0 == k or n == k:
                return 1
            return find(n - 1, k) + find(n - 1, k - 1)
        
        return find(n + 4, 4)
