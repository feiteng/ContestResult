class Solution:
    def countVowelStrings(self, n: int) -> int:
        
        @lru_cache(None)
        def dp(x, a):
            if x == n: return 1
            if a == 4: return 1
            cnt = 0
            for aa in range(a, 5):
                cnt += dp(x + 1, aa)
            return cnt
            
            
        return dp(0, 0)
