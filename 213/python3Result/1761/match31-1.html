class Solution:
    def countVowelStrings(self, n: int) -> int:
        @functools.lru_cache(None)
        def dp(idx,n):
            if n==0:
                return 1
            if idx==5:
                return 0
            res=0
            for i in range(n+1):
                res+=dp(idx+1,i)
            return res
        return dp(0,n)
