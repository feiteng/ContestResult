class Solution:
    def countVowelStrings(self, n: int) -> int:
        import functools
        @functools.lru_cache(None)
        def dfs(index, cur) -> int:
            if index == n:
                return 1
            res = 0
            for i in range(cur, 5):
                res += dfs(index + 1, i)
            return res
        return dfs(0, 0)
