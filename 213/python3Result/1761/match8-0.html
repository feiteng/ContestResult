class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(None)
        def dfs(n: int, index: int) -> int:
            if (n == 1):
                return index + 1
            result = 0
            for i in range(index + 1):
                result += dfs(n - 1, i)
            return result
        return dfs(n, 4)
