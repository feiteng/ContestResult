class Solution:
    def countVowelStrings(self, n: int) -> int:
        # a -> 0, e -> 1, i -> 2, o -> 3, u -> 4
        @functools.lru_cache(None)
        def dfs(i, last):
            if i == 0:
                return 1
            
            return sum(dfs(i - 1, l) for l in range(last, 4 + 1))
        
        return dfs(n, 0)
