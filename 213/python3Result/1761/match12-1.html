class Solution:
    def countVowelStrings(self, n: int) -> int:
        
        @lru_cache(None)
        def fn(n, k): 
            """Return """
            if n == 1: return k 
            ans = 0
            for kk in range(1, k+1):
                ans += fn(n-1, kk)
            return ans 
        
        return fn(n, 5)
