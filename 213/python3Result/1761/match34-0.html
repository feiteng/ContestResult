from functools import lru_cache
class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(None)
        def inner(i, num):
            if i == n:
                return 5 - num
            ans = 0
            for it in range(num, 5):
                ans += inner(i+1, it)
            return ans
        return inner(1, 0)
