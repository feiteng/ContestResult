from functools import lru_cache

class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(None)
        def dp(k, i):
            if k == 0:
                return 1
            if i == 0:
                return 0
            res = 0
            for j in range(k + 1):
                res += dp(k - j, i - 1)
            return res
            
        return dp(n, 5)
