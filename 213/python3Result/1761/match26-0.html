class Solution:
    def countVowelStrings(self, n: int) -> int:
        import functools
        @functools.lru_cache(None)
        def dp(i, j):
            if i == n:
                return 1
            res = 0
            for k in range(j, 5):
                res += dp(i + 1, k)
            return res
        
        return dp(0, 0)
