from functools import lru_cache
class Solution:
    def countVowelStrings(self, n: int) -> int:
        
        @lru_cache(None)
        def process(n, k):
            if n == 0:
                return 1
            ans = 0
            for i in range(1, k+1):
                ans += process(n-1, k+1-i)
            return ans
    
        return process(n, 5)
