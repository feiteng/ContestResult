class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(maxsize=None)
        def f(i, mx): 
            if i == 0:
                return 1
            count = 0
            for c in range(mx + 1):                
                count += f(i - 1, c)
            return count
            
        return f(n, 4)
