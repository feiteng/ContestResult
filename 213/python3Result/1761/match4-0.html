class Solution:
    def countVowelStrings(self, n: int) -> int:
        dp = [[0 for _ in range(n + 1)] for _ in range(5)]
        for i in range(5):
            dp[i][1] = 1
            
        for i in range(2, n + 1):
            for j in range(5):
                for k in range(j + 1):
                    dp[j][i] += dp[k][i - 1]
                    
        result = 0
        for i in range(5):
            result += dp[i][n]
            
        return result
