class Solution:
    def countVowelStrings(self, N: int) -> int:
        # if n == 1: return 5
        # if n == 2: return 15
        @lru_cache(None)
        def dfs(m, n):
            if n == 1: return 5 - m
            res = 0
            for i in range(m, 5):
                res += dfs(i, n-1)
            return res
        return dfs(0, N)
