class Solution:
    def countVowelStrings(self, n: int) -> int:
        
        def helper(prev, n):
            if n == 1:
                return prev
            ans = 0
            for i in range(prev, 0, -1):
                ans += helper(i, n - 1)
            return ans
        return helper(5, n)
