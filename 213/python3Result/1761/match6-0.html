class Solution:
    def countVowelStrings(self, n: int) -> int:
        
        #start=0,1,2,3,4
        #n
        
        @lru_cache(None)
        def dp(start, n):
            if n==0:
                return 1
            if start > 4:
                return 0
            res = 0
            for i in range(start, 5):
                res += dp(i, n-1)
            return res
    
        return dp(0, n)
