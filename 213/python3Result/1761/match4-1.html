class Solution:
    def countVowelStrings(self, n: int) -> int:
        #vowels = ["a", "e", "i", "o", "u"]
        F = [[0 for i in range(5)] for j in range(n + 1)]
        for i in range(5):
            F[1][i] = 1
        for i in range(2, n + 1):
            for j in range(5):
                for k in range(j + 1):
                    # print(i, j, k)
                    F[i][j] += F[i - 1][k]
        res = 0
        for i in range(5):
            res += F[n][i]
        # print(F[n])
        return res
