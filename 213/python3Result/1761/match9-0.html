class Solution:
    def countVowelStrings(self, n: int) -> int:
        @lru_cache(None)
        def dp(n, k): # 从[1,k]元素中，组成n个字典序字符串
            if n == 1:
                return k
            res = 0
            for i in range(1, k + 1):
                res += dp(n - 1, i)
            return res
        return dp(n, 5)
