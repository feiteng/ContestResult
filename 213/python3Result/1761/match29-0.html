import functools

class Solution:
    def countVowelStrings(self, n: int) -> int:
        @functools.lru_cache(None)
        def count(n_chars, n_alphabet):
            if n_chars == 1:
                return n_alphabet
            return sum(count(n_chars-1, next_n_alphabet) for next_n_alphabet in range(1, n_alphabet+1))
        
        return count(n, 5)
