class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        need_br = 0
        need_la = 0
        real_heights = []
        for i in range(1, len(heights)):
            v = 0
            if heights[i] > heights[i - 1]:
                v = heights[i] - heights[i - 1]
                need_br += v
                need_la += 1
            real_heights.append(v)

        if ladders >= need_la or bricks >= need_br:
            return len(heights) - 1

        def check(start) -> bool:
            d = sorted(real_heights[0:start], reverse=True)
            return bricks >= sum(d[ladders:])

        left = 0
        right = len(heights) - 1
        while left + 1 < right:
            mid = (right + left) // 2
            if check(mid):
                left = mid
            else:
                right = mid
        if check(right):
            return right
        if check(left):
            return left
        return 0
