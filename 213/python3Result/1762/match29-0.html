class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        # max heap
        heap = []
        sum_ = 0
        for i in range(1, len(heights)):
            diff = heights[i] - heights[i-1]
            if diff <= 0:
                continue
            sum_ += diff
            heapq.heappush(heap, -diff)
            if sum_ <= bricks:
                continue
            elif ladders == 0:
                return i-1
            else:
                x = -heapq.heappop(heap)
                ladders -= 1
                sum_ -= x
                assert sum_ <= bricks
        return i
