class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        def valid(n):
            diffs = []
            for i in range(n):
                if heights[i] < heights[i + 1]:
                    diffs.append(heights[i + 1] - heights[i])
            diffs.sort()
            # print(n, diffs)
            brick, ladder = bricks, ladders
            for diff in diffs:
                if brick >= diff:
                    brick -= diff
                elif ladder > 0:
                    ladder -= 1
                else:
                    return False
            
return True
        left, right = 0, len(heights) - 1
        while left <= right:
            mid = (left + right) // 2
            # print(left, mid, right)
            if valid(mid):
                left = mid + 1
            else:
                right = mid - 1
        # print()
        return right
