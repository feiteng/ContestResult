from heapq import heappush, heappop
class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        pq = []
        total_sum, pq_sum = 0, 0
        for i, h in enumerate(heights[1:], 1):
            dh = h - heights[i - 1]
            if dh <= 0: continue
            total_sum += dh
            heappush(pq, dh)
            pq_sum += dh
            
if len(pq) > ladders:
                pq_sum -= heappop(pq)
            if total_sum - pq_sum > bricks:
                return i - 1
        return len(heights) - 1
