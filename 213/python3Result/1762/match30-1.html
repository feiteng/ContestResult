class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        #binary search
        def check(i):
            #Check if we can reach the ith building
            if i == 0:
                return True
            diff 
= []
            for j in range(i):
                if heights[j + 1] > heights[j]:
                    diff.append(heights[j + 1] - heights[j])
            diff.sort(key = lambda x: -x)
            diff = diff[ladders:]
            nbricks = bricks
            # if i == 4:
            #     print(diff)
            for j in range(len(diff)):
                nbricks -= diff[j]
                if nbricks < 0:
                    return False
            
return True
        l, r = 0, len(heights) - 1
        while l < r:
            mid = (l + r)//2
            if check(mid):
                l = mid + 1
            else:
                r = mid
        if check(l):
            return l
        # print(check(3))
        return l - 1
