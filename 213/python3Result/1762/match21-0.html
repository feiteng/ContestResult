class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        n=len(heights)
        l=0
        r=n-1
        def f(x):
            res=[]
            for i in range(1,x+1):
                if heights[i]>heights[i-1]:
                    res.append(heights[i]-heights[i-1])
            res.sort()
            #print(x,res)
            for j in range(ladders):
                if not res:
                    return True
                res.pop()
            return sum(res)<=bricks
        while l<r:
            mid=(l+r+1)//2
            if f(mid):
                l=mid
            else:
                r=mid-1
        return l
