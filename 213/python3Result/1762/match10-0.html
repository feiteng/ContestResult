class Solution:
    def furthestBuilding(self, A: List[int], B: int, L: int) -> int:
        
        
        def valid(m):
            arr = []
            for i in range(1, m+1):
                if A[i] > A[i-1]:
                    arr.append(A[i] - A[i-1])
            # print(m, arr)
            if len(arr) <= L:
                return True
            arr.sort()
            if sum(arr[:len(arr)-L]) <= B:
                return True
            return False
            
        
        l, r = 0, len(A)-1
        while l < r:
            m = (l+r+1)//2
            if valid(m):
                l = m
            else:
                r = m-1
        return l
