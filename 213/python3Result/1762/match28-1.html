class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        n = len(heights)
        h = []
        cursum = 0
        for i in range
(1,n):
            if heights[i] > heights[i-1]:
                diff = heights[i]-heights[i-1]
                heapq.heappush(h,diff)
                if len(h) > ladders:
                    cursum += heapq.heappop(h)
                    if cursum > bricks:
                        return i-1
        return n-1
