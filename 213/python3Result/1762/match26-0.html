class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        n = len(heights)
        def check(dest):
            jumps = []
            for i in range(1, dest+1):
                if heights[i-1] < heights[i]:
                    jumps.append(heights[i] - heights[i-1])
            bricks_required = sum(jumps) - sum(heapq.nlargest(ladders, jumps))
            return bricks_required <= bricks
        lo, hi = 1, n-1
        ans = 0
        while lo <= hi:
            mid = (lo + hi) // 2
            if check(mid):
                ans = max(ans, mid)
                lo = mid + 1
            else:
                hi = mid - 1
        return ans
