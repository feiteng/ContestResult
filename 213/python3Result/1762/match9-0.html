class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        def judge(heights, bricks, ladders, x):
            need = []
            
            for i in range(x):
                if heights[i+1]>heights[i]:
                    need.append(heights[i+1]-heights[i])
            
            need.sort()
            
            if ladders>=len(need):
                return True
            
            s = sum(need[:len(need)-ladders])
            return bricks>=s
        
        l, r = 0, len(heights)-1
    
        while l<=r:
            m = (l+r)//2
        
            if judge(heights, bricks, ladders, m):
                l = m+1
            else:
                r = m-1
    
        return r
