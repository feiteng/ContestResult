class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        end = len(heights)
        start = min(ladders, end-1)
        def test(m):
            p = []
            for i in range(1, m+1):
                if heights[i] > heights[i-1]:
                    p.append(heights[i]-heights[i-1])
            if len(p)<=ladders:
                return True
            p.sort()
            if ladders == 0:
                return sum(p)<=bricks
            return sum(p[:-ladders])<=bricks
        while start + 1 < end:
            m = (start + end) // 2
            if test(m):
                start = m
            else:
                end = m
        return start
