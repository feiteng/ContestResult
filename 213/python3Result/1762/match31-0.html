class Solution:
    def furthestBuilding(self, heights: List[int], b: int, l: int) -> int:
        # heap
        lheap,bheap = [],[]
        cnt = 0
        n = len(heights)
        for i,(p1,p2) in enumerate(zip(heights,heights[1:]),1):
            if p1<p2:
                d = p2-p1
                heapq.heappush(lheap,d)
                if len(lheap)>l:
                    cnt+=heapq.heappop(lheap)
                if cnt>b:
                    return i-1
        return n-1
