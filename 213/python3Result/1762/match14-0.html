class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        def check(n):
            ups = []
            for i in range(1, min(len(heights),n + 1)):
                if heights[i] - heights[i-1] > 0:
                    ups.append(heights[i] - heights[i-1])
            ups.sort(reverse=True)
            # print(n, ups)
            if len(ups) <= ladders:
                return True
            else:
                return sum(ups[ladders:]) <= bricks
        # print(check(5))
        l = 0
        r = len(heights) 
        while l < r:
            mid = (l + r) // 2
            if check(mid):
                l = mid + 1
            else:
                r = mid
        return l - 1
