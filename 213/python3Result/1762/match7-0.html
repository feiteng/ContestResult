from heapq import *

class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        h = []
        s, sl = 0, 0
        for i, x in enumerate(heights[1:], 1):
            gap = x - heights[i - 1]
            if gap < 0:
                continue
            s += gap
            sl += gap
            heappush(h, gap)
            if len(h) > ladders:
                sl -= heappop(h)
            if s - sl > bricks:
                return i - 1
            # print(i, s, sl)
        return len(heights) - 1
