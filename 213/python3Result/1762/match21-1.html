class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        n = len(heights)
        low, high 
= 0, n - 1
        def check(x):
            l = []
            for i in range(1, x + 1):
                if heights[i] > heights[i - 1]:
                    l.append(heights[i] -  heights[i - 1])
            l.sort()
            if ladders >= len(l):
                return True
            return sum(l[:len
(l) - ladders]) <= bricks
        while low < high:
            mid = low + (high - low + 1) // 2
            if check(mid):
                low = mid
            else:
                high = mid - 1
        return low
