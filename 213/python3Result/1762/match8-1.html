from typing import List


class Solution:
    def furthestBuilding(self, H: List[int], B: int, L: int) -> int:
        b = l = 0
        n = len(H)
        total = []
        for i in range
(1, n):
            res = 0
            if H[i] > H[i - 1]:
                res = H[i] - H[i - 1]
                b += res
                l += 1
            total.append(res)
        if L >= l or B >= b:
            return len(H) - 1
        def check(v) -> bool:
            d = sorted(total[0:v])[::-1]
            
return B >= sum(d[L:])
        left = 0
        right = len(H) - 1
        while left + 1 < right:
            mid = (right + left) // 2
            if check(mid):
                left = mid
            else:
                right = mid - 1
        if check(right):
            return right
        if check(left):
            return left
        return 0
