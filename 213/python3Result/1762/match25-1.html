class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        
        heap = []
        n = len(heights)
        tot = 0
        heap_sum = 0
        #用一个堆维护最大的ladders个元素
        for i in range(1, n):
            if heights[i] > heights[i-1]:
                num = heights[i] - heights[i-1]
                heapq.heappush(heap, num)
                tot += num
                heap_sum += num
                if len(heap) > ladders:
                    heap_sum -= heapq.heappop(heap)
                if tot - heap_sum > bricks:
                    return i-1
        return n-1
