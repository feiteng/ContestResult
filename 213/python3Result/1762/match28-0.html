class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        b=0
        l=[]
        ans=0
        for i in range(1,len(heights)):
            if heights[i]>heights[i-1]:
                h=heights[i]-heights[i-1]
                heapq.heappush(l,h)
                if len(l)>ladders:
                    b+=heapq.heappop(l)
                    if b>bricks:
                        return ans
            ans=i
        return ans
