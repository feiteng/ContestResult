class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        result = 0
        n = len(heights)
        q = []
        current_bricks = 0
        for i in range(n - 1):
            if (heights[i] < heights[i + 1]):
                need_bricks = heights[i + 1] - heights[i]
                heapq.heappush(q, -need_bricks)
                current_bricks += need_bricks
                while (current_bricks > bricks) and (ladders > 0):
                    current_bricks += heapq.heappop(q)
                    ladders -= 1
                if (ladders == 0) and (current_bricks > bricks):
                    return i
        return n - 1
