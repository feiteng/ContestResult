class Solution:
    def furthestBuilding(self, heights, bricks, ladders):
        heights += [float("inf")]
        #print(heights)
        
def isReach(k):
            to_climb = []
            for i in range(k):
                if heights[i] < heights[i+1]:
                    to_climb.append(heights[i+1] - heights[i])
            if len(to_climb) <= ladders: return True
            to_climb = sorted(to_climb)[::-1]
            #print(k, to_climb)
            
            return sum(to_climb[ladders:]) <= bricks

        beg, end 
= 0, len(heights) - 1
        while beg + 1 < end:
            #print(beg, end)
            mid = (beg + end)//2
            if isReach(mid):
                beg = mid
            else:
                end = mid
        return beg
