class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        def judge(index):
            tmp = []
            for i in range(1, index + 1):
                if heights[i] > heights[i - 1]:
                    tmp.append(heights[i] - heights[i - 1])
            tmp = sorted(tmp)
            s = 0
            for i in range(len(tmp) - ladders):
                s += tmp[i]
            if bricks >= s:
                return True
            else:
                return False
        
        l = 1
        r = len(heights) - 1
        while l <= r:
            m = (l + r) // 2
            if judge(m):
                l = m + 1
            else:
                r = m - 1
        return r
