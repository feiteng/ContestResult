class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        gaps = []
        n = len(heights)
        tmp = 0
        for i in range(1, n):
            if heights[i] <= heights[i - 1]:
                continue
            heapq.heappush(gaps, heights[i] - heights[i - 1])
            if len(gaps) > ladders:
                tmp += heapq.heappop(gaps)
                if tmp > bricks:
                    return i - 1
        return n - 1
