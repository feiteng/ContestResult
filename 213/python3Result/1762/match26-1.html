class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        #cumsum = []
        #for i in heights:
        #    cumsum[i] = heights[i] if i == 0 else cumsum[i-1] + heights[i]
            
        
def canGo(n):
            gaps = []
            for i in range(1, n + 1):
                if heights[i] - heights[i-1] > 0:
                    gaps.append(heights[i] - heights[i-1])
            gaps.sort()
            if ladders != 0:
                gaps = gaps[0:-1 * ladders]
            if sum(gaps) <= bricks:
                #print("yes: n=%d, gaps = %s, bricks = %d" % (n, gaps, bricks))
                return True
            else: 
                #print("no: n=%d, gaps = %s, bricks = %d" % (n, gaps, bricks))                
                return False
        
        hi = len(heights) - 1
        lo 
= 0
        ret = 0
        while lo <= hi:
            guess = (hi + lo) // 2
            if canGo(guess):
                ret = max(guess, ret)
                lo = guess + 1
            else:
                hi = guess - 1
        return ret
