class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        import heapq
        biggest=[]
        for i in range(1,len
(heights)):
            if heights[i]<=heights[i-1]:
                continue
            heapq.heappush(biggest,heights[i]-heights[i-1])
            if len(biggest)>ladders:
                bricks-=heapq.heappop(biggest)
                if bricks<0:
                    return i-1
        return len(heights)-1
