class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        
        n, q, total = len(heights), [], 0
        for i, h in enumerate(heights[:-1]):
            cur = heights[i+1] - h
            
            if cur > 0:
                heapq
.heappush(q, cur)
                if len(q) > ladders:
                    total += heapq.heappop(q)
                if total > bricks:
                    return i
        return n - 1
