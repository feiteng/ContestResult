class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        def reach(m) :
            g = []
            for i in range(m) :
                if heights[i] < heights[i+1] :
                    g.append(heights[i+1]-heights[i])
            g.sort()
            b, l = bricks, ladders 
            for i in g :
                if b >= i :
                    b -= i
                elif l > 0 :
                    l -= 1
                else :
                    return False
            return True

        s, e = 0, len(heights)-1
        while s < e :
            m = (s+e+1) // 2
            if reach(m) :
                s = m
            else :
                e = m - 1
        return s
