class Solution:
    def furthestBuilding(self, heights: List[int], bricks: int, ladders: int) -> int:
        diff = []
        countBricks = 0
        
        
        for i in range(len
(heights)-1):
            if heights[i+1] > heights[i]:
                temp = heights[i+1] - heights[i]
                heapq.heappush(diff, - temp)
                
                countBricks += temp
                
                while countBricks > bricks and ladders > 0:
                    countBricks -= - heapq.heappop(diff)
                    ladders -= 1
                
                if countBricks > bricks:
                    return i

        
        return len(heights) - 1
