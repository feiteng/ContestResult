class Solution {
public:
    bool judge(int x, vector<int>& h, int b, int l) {
        vector<int> diff;
        for (int i = 0; i < x - 1; i ++) if (h[i + 1] > h[i]) {
            diff.push_back(h[i + 1] - h[i]);
        }
        sort(diff.begin(), diff.end());

        for (int i = 0; i < (int)diff.size() - l; i ++) {
            b -= diff[i];
        }
        return b >= 0;
    }
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int L = 1, R = heights.size();
        while (L < R) {
            int mid = (L + R + 1) / 2;
            if (judge(mid, heights, bricks, ladders)) {
                L = mid;
            } else {
                R = mid - 1;
            }
        }
        return L - 1;
    }
};
