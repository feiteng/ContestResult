class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int l = 0,u = heights.size()-1;
        int ans = 0;
        while(l <= u){
            int mid = (l + u)/2;
            vector<int> v;
            for(int i=1;i<=mid;i++){
                if(heights[i] > heights[i-1]){
                    v.push_back(heights[i] - heights[i-1]);
                }
            }
            sort(v.begin(),v.end());
            int curr = 0;
            int now = 0;
            while(curr < v.size() && now+v[curr] <= bricks){
                now += v[curr];
                curr++;
            }
            if(v.size() - curr <= ladders){
                ans = max(ans,mid);
                l = mid + 1;
            }
            else {
                u = mid - 1;
            }
        }
        return ans;
    }
};
