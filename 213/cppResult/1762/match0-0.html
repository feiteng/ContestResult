class Solution {
public:
    int furthestBuilding(vector<int>& h, int b, int L) {
        int n = h.size();
        priority_queue<int> q;
        for (int i = 1; i < n; i++) {
            if (h[i-1] >= h[i]) continue;
            q.push(h[i-1] - h[i]);
            if (q.size() > L) {
                auto o = q.top(); q.pop();
                b += o;
                if(b < 0) return i-1;
            }
        }
        return n-1;
    }
};
