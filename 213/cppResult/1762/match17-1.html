class Solution {
 public:
  int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
    int n = heights.size();
    priority_queue<int, vector<int>, greater<int>> pq;
    int need = 0;

    for (int i = 1; i < n; i++) {
      int d = heights[i] - heights[i - 1];
      if (d > 0) {
        pq.push(d);
        if (pq.size() > ladders) {
          need += pq.top();
          pq.pop();
        }
        if (need > bricks) return i - 1;
      }
    }
    return n - 1;
  }
};
