class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        priority_queue<int> q;
        int ans = 1;
        while(ans < heights.size()){
            if(heights[ans] <= heights[ans - 1]) {
                ans++;
                continue;
            }
            q.push(heights[ans] - heights[ans - 1]);
            bricks -= heights[ans] - heights[ans - 1];
            if(bricks < 0){
                if(ladders == 0) break;

                else while(ladders && bricks < 0 && q.size()) {
                    ladders--;
                    bricks += q.top();
                    q.pop();
                }
                if(bricks < 0) break;
            }
            ans++;
        }
        return ans - 1;
    }
};
