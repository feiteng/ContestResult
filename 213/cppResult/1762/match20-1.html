class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        multiset<int> st;
        int b = 0;
        for (int i = 1; i < heights.size(); i++) {
            if (heights[i] > heights[i - 1]) {
                st.insert(heights[i] - heights[i - 1]);
                if (st.size() > ladders) {
                    b += *st.begin();
                    st.erase(st.begin());
                }
                if (b > bricks) {
                    return i - 1;
                }
            }
        }
        return heights.size() - 1;
    }
};
