class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        multiset<int> ms;
        for (int i = 1; i < heights.size(); i++) {
            
            int diff = heights[i-1] - heights[i];
            if (diff  >= 0)
                continue;
            
            ms.insert(abs(diff));
            
            if (ms.size() <= ladders)
                continue;
            
            int thisBrick = *ms.begin();
            ms.erase(ms.begin());
            
            if (thisBrick <= bricks)
                bricks -= thisBrick;
            else
                return i-1;
        }
        
        return heights.size()-1;
    }
};
