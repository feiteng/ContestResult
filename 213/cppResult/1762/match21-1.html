class Solution {
public:
    int furthestBuilding(vector<int>& H, int B, int L) {
        int sum = 0;

        multiset<int> S;
        int N = H.size();
        
        for(int i = 0; i < N-1; i++)
        {
            int t = H[i+1] - H[i];
            if(t > 0)
            {
                S.insert(t);
                if(S.size() > L)
                {
                    sum += *S.begin();
                    S.erase(S.begin());
                    if(sum > B)
                    {
                        return i;
                    }
                }
            }
        }
        
        return N-1;
    }
};
