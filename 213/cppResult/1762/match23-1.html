class Solution {
public:
    
    bool check(int x,vector<int>&h,int b,int l){
        vector<int>v;
        for(int i=1;i<=x;i++){
            if(h[i]>h[i-1]) v.push_back(h[i]-h[i-1]);
        }
        sort(v.begin(),v.end());
        reverse(v.begin(),v.end());
        for(int i=0;i<v.size();i++){
            if(l>0) l--;
            else{
                if(b>=v[i]) b-=v[i];
                else return false;
            }
        }
        return true;
    }
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int l=0,r=heights.size()-1,ans=0,mid;
        while(l<=r){
            mid=(l+r)>>1;
            if(check(mid,heights,bricks,ladders)){
                ans=mid;
                l=mid+1;
            }else{
                r=mid-1;
            }
        }
        return ans;
    }
};
