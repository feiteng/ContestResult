class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int low = 0, high = heights.size() - 1;
        int best = 0;
        while(low <= high) {
            int mid = (low + high) / 2;
            vector<int> diffs;
            for(int i = 1; i <= mid; ++ i) {
                if(heights[i] > heights[i - 1]) {
                    diffs.push_back(heights[i] - heights[i - 1]);
                }
            }
            sort(diffs.begin(), diffs.end());
            int tmp = bricks, tmp2 = ladders;
            for(int i = 0; i < diffs.size(); ++ i) {
                if(tmp >= diffs[i]) tmp -= diffs[i];

                else tmp2 --;
            }
            if(tmp2 >= 0 && tmp >= 0) {
                best = mid;
                low = mid + 1;
            }
            else {
                high = mid - 1;
            }
        }
        return best;
    }
};
