class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        priority_queue<int, vector<int>, greater<int>> unknowns;
        int bricksNeeded = 0;
        for (int i = 0; i + 1 < heights.size(); ++i)
        {
            int delta = heights[i + 1] - heights[i];
            if (delta > 0)
            {
                unknowns.push(delta);
                if (unknowns.size() > ladders)
                {
                    bricksNeeded += unknowns.top();
                    unknowns.pop();
                    if (bricksNeeded > bricks)
                    {
                        return i;
                    }
                }
            }
        }
        
        return heights.size() - 1;
    }
};
