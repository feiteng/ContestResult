class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        priority_queue<int, vector<int>, greater<int>> pq;
        int i = 0;
        int temp = 0;
        while (i + 1 < heights.size()) {
            cout << "i: " << i << endl;
            if (heights[i + 1] <= heights[i]) {
                ++i;
                // cout << "continue" << endl;
                continue;
            }
            int inc = heights[i + 1] - heights[i];
            if (pq.size() < ladders) {
                pq.push(inc);
                ++i;
                // cout << "push: " << inc << endl;
                // cout << "continue" << endl;
                continue;
            }
            if (pq.empty()) {
                temp += inc;
            } else if (pq.top() < inc) {
                int p = pq.top();
                pq.pop();
                pq.push(inc);
                temp += p;
                // cout << "pop: " << p << ", push: " << inc << endl;
            } else {
                temp += inc;
            }
            // cout << "temp: " << temp << endl;
            if (temp > bricks) {
                break;
            }
            ++i;
        }
        return i;
    }
};
