class Solution {
public:
    bool canReach(vector<int>& heights, int bricks, int ladders, int last){
        vector<int> needs;
        for(int i = 1; i <= last; i ++){
            if(heights[i] >= heights[i - 1])
                needs.push_back(heights[i] - heights[i - 1]);
        }
        sort(needs.begin(), needs.end());
        for(int i = needs.size() - 1 - ladders; i >= 0 && bricks >= 0; i --)
            bricks -= needs[i];

        return bricks >= 0;
    }
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int lp = 0, rp = heights.size() - 1;
        while(lp < rp){
            int mid = lp + (rp - lp + 1) / 2;
            if(canReach(heights, bricks, ladders, mid))
                lp = mid;
            else
                rp = mid - 1;
        }
        return lp;
    }
};
