class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        priority_queue<int, vector<int>, greater<int>> bricksReq;
        int t = 0, minBrick = 0, N = heights.size();
        for (int i = 1; i < N; i++){
            if (heights[i] <= heights[i-1])continue;
            bricksReq.push(heights[i]-heights[i-1]);
            while (bricksReq.size() > ladders){
                minBrick += bricksReq.top();
                bricksReq.pop();
            }
            if (minBrick > bricks)return i-1;
        }
        return N-1;        
    }
};
