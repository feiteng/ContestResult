class Solution {
public:
  int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
    int low = 0;
    int high = heights.size() - 1;
    int ans = 0;
    while (low <= high) {
      int mid = (low + high) / 2;
      vector<int> v;
      for (int i = 1; i <= mid; i++) {
        if (heights[i] > heights[i - 1]) {
          v.push_back(heights[i] - heights[i - 1]);
        }
      }
      int bricks_left = bricks;
      int ladders_left = ladders;
      sort(v.begin(), v.end());
      reverse(v.begin(), v.end());
      for (int i = 0; i < v.size(); i++) {
        if (ladders_left) {
          ladders_left--;
        } else {
          bricks_left -= v[i];
        }
      }
      if (bricks_left >= 0) {
        low = mid + 1;
        ans = mid;
      } else {
        high = mid - 1;
      }
    }
    return ans;
  }
};
