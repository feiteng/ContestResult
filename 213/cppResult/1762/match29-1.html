class Solution {
public:

    inline bool works(vector<int> arr, int bricks, int ladders, int mid){
    vector<int> v;
    for(int i=0; i+1<=mid; i++){
        if(arr[i] < arr[i+1])
            v.push_back(arr[i+1]-arr[i]);
    }
    long long req = 0;
    sort(v.begin(),v.end());
    for(int i=0; i<(int)v.size()-ladders; i++)
        req+=v[i];
    if(req <= bricks) return true;
    return false;
}


int furthestBuilding(vector<int> heights, int bricks, int ladders) {
    int low = 0, high = (int)heights.size()-1, ans = 0;
    while(low <= high){
        int mid = (low+high)/2;
        if(works(heights,bricks,ladders,mid)){
            ans = mid;
            low = mid+1;
        }else high = mid-1;
    }
    return ans;
}
};
