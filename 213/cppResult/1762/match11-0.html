class Solution {
public:
    bool check(int n, vector<int>& a, int bricks, int ladders) {
        vector<int> ve;
        for(int i = 1; i < n; i++) {
            if(a[i] > a[i-1]) ve.push_back(a[i] - a[i-1]);
        }
        sort(ve.begin(), ve.end());
        if(ladders >= ve.size()) return true;
        long long height = 0;
        for(int i = 0; i < ve.size() - ladders; i++) height += ve[i];
        return height <= bricks;
    }
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int l = 1, r = heights.size(), ans;
        while(l <= r) {
            int mid = (l + r) >> 1;
            if(check(mid, heights, bricks, ladders)) {
                ans = mid;
                l = mid + 1; 
            } else {
                r = mid - 1;
            }
        }
        return ans-1;
    }
};
