class Solution {
public:
    bool ok(vector<int>& steps, int target, int bricks, int ladders){
        vector<int> allsteps;
        for(int i=1;i<=target;i++){
            allsteps.push_back(steps[i-1]);
        }
        sort(allsteps.begin(), allsteps.end(), greater<int>());
        int need = 0;

        for(int i=ladders;i<allsteps.size();i++){
            need+=allsteps[i];
        }
        return bricks>=need;
    }
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        vector<int> steps;
        if(heights.size()==1)
            return 0;
        for(int i=1;i<heights.size();i++){
            if(heights[i]>heights[i-1])
                steps.push_back(heights[i]-heights[i-1]);
            else
                steps.push_back(0);
        }
        int l = 0;

        int r = heights.size()-1;
        while(l<r){
            int mid = (l+r+1)/2;
            if(ok(steps, mid, bricks, ladders)){
                l = mid;
            }else{
                r = mid-1;
            }
        }
        return l;
    }
};
