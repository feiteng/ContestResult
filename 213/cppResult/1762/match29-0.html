class Solution {
    bool bs(vector<int>& arr, int b, int l, int res){
        vector<int> brr;
        for(int i = 1; i <= res; i++){
            if(arr[i-1] < arr[i]){
                int diff = arr[i] - arr[i-1];
                brr.push_back(diff);
            }
        }
        sort(brr.begin(), brr.end());
        int cnt = brr.size();
        for(int i= 0; i < brr.size(); i++){
            if(b >= brr[i]){
                cnt--;
                b -= brr[i];
            }
        }
        return cnt <= l;
    }
public:
    int furthestBuilding(vector<int>& arr, int b, int la) {
        int l = 0, r = arr.size()-1;
        int res = 0;
        while(l <= r){
            int mid = (l+r)>>1;
            if(bs(arr, b, la, mid)){
                l = mid+1;
                res = mid;
            }
            else r = mid-1;
        }
        return res;
    }
};
