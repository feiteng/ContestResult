class Solution {
public:
    
    bool possible(int progress, vector<int> &heights, int bricks, int ladders) {
        
        vector<int> current_heights(progress);
        for(int i=0; i<progress; i++) {
            current_heights[i] = max(0, heights[i+1] - heights[i]);
        }
        sort(current_heights.begin(), current_heights.end(), greater<int>());
        
        for(int i: current_heights) {
            
            if(ladders) {
                ladders--;
                continue;
            }
            bricks -= i;
        }
        
        return bricks >= 0;
    }
    
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
       
        int lo = 0, hi = heights.size()-1;
        while(lo < hi) {
            int mid = (lo+hi+1)/2;
            if(possible(mid, heights, bricks, ladders))
                lo = mid;
            else
                hi = mid-1;
        }
        
        return (lo+hi)/2;
    }
};
