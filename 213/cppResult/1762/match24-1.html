class Solution {
public:
    
    bool check(int mid , vector<int>& H , int B , int L){
        vector<int> h;
        int i;
        for(i = 0; i < mid; ++i){
            if(H[i + 1] - H[i] > 0)h.push_back(H[i + 1] - H[i]);
        }
        if(h.size() == 0)return 1;
        sort(h.begin() , h.end());

        for(i = 0; i < h.size(); ++i){
            if(B >= h[i])B -= h[i];
            else{
                if(L)--L;
                else return 0;
            }
        }
        return i == h.size();
    }
    
    int furthestBuilding(vector<int>& H, int B, int L) {
        int l = 0 , r = H.size() - 1, ans = 0;

        while(l <= r){
            int mid = (l + r) >> 1;
            if(check(mid , H , B , L)){
                l = mid + 1;
                ans = mid;
            }
            else r = mid - 1;
        }
        return ans;
    }
};
