class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int n = heights.size(), b=0;
        priority_queue<int>pq;
        for (int i=1; i<n; i++) {
            if (heights[i]>heights[i-1]) {

                b += heights[i]-heights[i-1];
                pq.push(heights[i]-heights[i-1]);
                if (b>bricks) {
                    if (ladders>0) {
                        ladders--;
                        b -= pq.top();
                        pq.pop();
                    } else
                        return i-1;
                }
            }
        }
        return n-1;
        
    }
