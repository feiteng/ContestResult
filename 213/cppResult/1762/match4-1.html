class Solution {
public:

    bool check(const vector<int>& heights, int bricks, int ladders, int target) {
        vector<int> diffs;
        for (int i = 1; i <= target; ++ i) {
            if (heights[i] > heights[i - 1]) diffs.push_back(heights[i] - heights[i - 1]);
        }
        sort(diffs.begin(), diffs.end(), greater<int>());
        for (int d: diffs) {
            if (ladders == 0 && bricks < d) return false;
            if (ladders > 0) -- ladders;

            else bricks -= d;
        }
        return true;
    }
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int lo = 0, hi = heights.size() - 1;
        while (lo < hi) {
            int mid = (lo + hi + 1) / 2;
            if (check(heights, bricks, ladders, mid)) {
                lo = mid;
            } else {
                hi = mid - 1;
            }
        }
        return lo;
    }
};
