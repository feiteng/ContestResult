#define LL long long


class Solution {
public:
    bool ok(int x, vector<int>& heights, int bricks, int ladders)  {
      vector<int> a;
      for (int i = 1; i <= x; i++) {
        if (heights[i] > heights[i - 1]) {
          a.push_back(heights[i] - heights[i - 1]);
        }
      }
      sort(a.begin(), a.end());

      LL sz = 0;
      for (int i = 0; i < (int)a.size() - ladders; i++) {
        sz += a[i];
      }
      return sz <= bricks;
    }
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
      int n = heights.size();
      int l = 0, r = n - 1;
      while (l + 1 < r) {
        int mid = (l + r) / 2;
        if (ok(mid, heights, bricks, ladders)) l = mid;
        else r = mid;
      }
      if (ok(r, heights, bricks, ladders)) return r;
      else return l;
    }
};
