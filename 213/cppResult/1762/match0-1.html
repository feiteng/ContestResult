class Solution {
public:
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int needed = 0;
        multiset<int> grt;
        for(int i = 1; i < heights.size(); i++){
            int d = heights[i] - heights[i - 1];

            if(d <= 0) continue;
            grt.emplace(d);
            if(grt.size() > ladders){
                auto it = grt.begin();
                needed += *it;
                if(needed > bricks) return i - 1;
                grt.erase(it);
            }
        }
        return heights.size() - 1;
