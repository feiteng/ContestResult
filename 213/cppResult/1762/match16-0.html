class Solution {
public:
    
    int furthestBuilding(vector<int>& h, int b, int l) {
        // int used = 0;
        priority_queue<int> pq;
        
        int i = 0;
        while (i + 1 < h.size()) {
            int diff = h[i+1] - h[i];
            if (diff > 0) {
                b -= diff;
                pq.push(diff);
                while (!pq.empty() && b < 0 && l > 0) {
                    b += pq.top();
                    pq.pop();
                    l--;
                }
                if (b < 0) break;
            }
            i++;
        }
        
        return i;
    }
};
