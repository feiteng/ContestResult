class Solution {
public:
    
    bool Can(const vector<int>& H, int bricks, int ladders, int max_id) {
        vector<int> diffs;
        for(int i=1;i<=max_id;i++) {
            if (H[i]>H[i-1])
                diffs.push_back(H[i]-H[i-1]);
        }
        
        sort(diffs.begin(),diffs.end(),greater<int>());
        
        for(int d : diffs) {
            if (ladders) {
                ladders--;
            } else {
                if (bricks < d)
                    return false;
                bricks -= d;
            }
        }
        return true;
    }
    
    
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        
        int low = 0;

        int high = heights.size()-1;
        while(low < high) {
            int mid = (low + high + 1)/2;
            if (Can(heights, bricks, ladders, mid)){
                low=mid;
            } else {
                high=mid-1;
            }
        }
        return low;
    }
};
