class Solution {
public:

    bool canReach(const vector<int>& h, int n, int bricks, int ladders) {
        vector<int> ds;
        for (int i = 0; i + 1 <= n; ++i) {
            if (h[i + 1] > h[i]) {
                ds.push_back(h[i + 1] - h[i]);
            }
        }
        sort(ds.begin(), ds.end());
        int nds = ds.size();

        for (int i = 0; i < nds - ladders; ++i) {
            bricks -= ds[i];
        }
        return bricks >= 0;
    }

    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int lft = 0;

        int rgh = heights.size() - 1;
        while (lft != rgh) {
            int m = (lft + rgh + 1) / 2;
            if (canReach(heights, m, bricks, ladders)) {
                lft = m;
            } else {
                rgh = m - 1;
            }
        }
        return lft;
    }
};
