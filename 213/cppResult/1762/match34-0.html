class Solution {
public:
    
    bool isPossible(vector<int> &diff,int mid,int bricks,int ladders)
    {
        vector<int> ans;
        
        for(int i = 0; i <= mid; i++)
        {
            if(diff[i]) ans.push_back(diff[i]);
        }
        
        if(ans.size() <= ladders) return 1;
        
        sort(ans.begin(),ans.end());
        long long int sum = 0;
        for(int i = 0; i < ans.size()-ladders; i++)
        {
            sum += ans[i];
            if(sum > bricks) return 0;
        }
        
        return sum <= bricks;
    }
    
    
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        
        vector<int> diff;
        diff.push_back(0);
        for(int i = 0; i < heights.size()-1; i++)
        {
            if(heights[i+1]-heights[i] > 0)
            {
                diff.push_back(heights[i+1]-heights[i]);                
            }
            else
            {
                diff.push_back(0);
            }

        }
        
        int st = 0,end = heights.size()-1;
        
        int ans = 0;
   
//         for(int i = 0; i < diff.size(); i++)
//         {
//             cout<<diff[i]<<" ";
//         }
        
        while(st <= end)
        {
            int mid = (st+end)/2;
            
            if(isPossible(diff,mid,bricks,ladders))
            {
                st = mid+1;
                ans = mid;
            }
            else
            {
                end = mid-1;
            }
        }
        
        
        return ans;
    }
};
