class Solution {
public:
    int furthestBuilding(vector<int>& h, int bricks, int ladders) {
        priority_queue<int, vector<int>, greater<int>> m;
        int ans = 0;
        int cnt = 0;
        for(int i=1;i<h.size();i++){
            if(h[i] > h[i-1]){
                m.push(h[i] - h[i-1]);
            }

            if(m.size() > ladders){
                cnt += m.top();
                m.pop();
            }
            if(cnt > bricks){
                break;
            }
            ans = i;
        }
        return ans;
    }
};
