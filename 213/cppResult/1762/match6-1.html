class Solution {
public:

    bool check(int limit, const vector<int>& heights, int bricks, int ladders) {
        vector<int> cnt;
        cnt.reserve(limit);
        
        for (int i = 1; i <= limit; i++) {
            if (heights[i-1] < heights[i]) {
                cnt.push_back(heights[i] - heights[i-1]);
            }
        }
        sort(cnt.begin(), cnt.end());
        int sum = 0;
        int remain = 0;
        for (int i = 0; i < cnt.size(); i++) {
            sum += cnt[i];
            if (sum > bricks) {
                remain = cnt.size() - i;
                break;
            } 

        }
      //  cout << sum << " " << remain << " " << limit  << " " << cnt.size() << endl;
        return ladders >= remain;
    }
    int furthestBuilding(vector<int>& heights, int bricks, int ladders) {
        int L = 0, R = heights.size()-1;
        while(L <= R) {
            int mid = (L+R)>>1;
            if (check(mid, heights, bricks, ladders)) {
              //  cout << mid << " Y" << endl;
                L = mid+1;
            } else {
              //  cout << mid << " N" << endl;
                R = mid-1;
            }
        }
        return L-1;
    }
};
