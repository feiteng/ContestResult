long long f[31][31];
class Solution {
public:

    string kthSmallestPath(vector<int>& destination, int k) {
        for(int i=0;i<31;i++){
            f[i][0]=1,f[i][i]=1;
            for(int j=1;j<i;j++){
                f[i][j]=f[i-1][j-1]+f[i-1][j];
            }
        }
        int n=destination[0],m=destination[1];
        int a=n,b=m;
        string ret="";
        for(int i=0;i<n+m;i++){
            // cout<<ret<<endl;
            if(b==0||k>f[a+b-1][b-1]){
                ret+='V';

                if(b>0) k-=f[a+b-1][b-1];
                a--;
            }else{
                ret+='H';
                b--;
            }
        }
        return ret;
    }
};
