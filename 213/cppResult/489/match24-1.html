#define ll unsigned long long
class Solution {
public:
    int a[16][16];

    string solve(int n, int m, int k) {


        string ans = "";
        int l = 0, r = 0;
        while (l < n || r < m) {
            if (l == n) {
                ans += "V";
                ++r;
                continue;
            }
            if (r == m) {
                ans += "H";
                ++l;
                continue;
            }

            if (a[n - l - 1][m - r] > k) {
                ans += "H";
                ++l;
            } else {
                k -= a[n - l - 1][m - r];
                ++r;
                ans += "V";
            }
        }
        return ans;
    }

    string kthSmallestPath(vector<int>& des, int k) {
        ll x = des[0];
        ll y = des[1];
        memset(a, 0, sizeof(a));

        a[0][0] = 1;
        for (int i = 0; i <= 15; ++i) {
            for (int j = 0; j <= 15; ++j) {
                if (i) a[i][j] += a[i - 1][j];
                if (j) a[i][j] += a[i][j - 1];
            }
        }
        return solve(y,x,k-1);
    }
};
