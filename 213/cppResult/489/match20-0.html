class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int m = destination[0] + 1, n = destination[1] + 1;
        vector<vector<int>> dp(m, vector<int>(n, 1));
        for (int i = m - 2; i >= 0; --i) {
            for (int j = n - 2; j >= 0; --j) {
                dp[i][j] = dp[i + 1][j] + dp[i][j + 1];
            }
        }
        int i = 0, j = 0;
        string ret = {};
        while (i != m - 1 && j != n - 1) {
            if (k > dp[i][j + 1]) {
                k -= dp[i][j + 1];
                ret += "V";
                ++i;
            }
            else {
                ret += "H";
                ++j;
            }
        }
        while (i != m - 1) {
            ret += "V";
            ++i;
        }
        while (j != n - 1) {
            ret += "H";
            ++j;
        }
        return ret;
    }
};
