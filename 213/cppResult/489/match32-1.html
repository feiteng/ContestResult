class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int row=destination[0];

        int column=destination[1];
        vector<vector<int64_t>> component(row+column+1,vector<int64_t>(row+1,0));
        for(int i=0;i<=row+column;++i)
        {
            component[i][0]=1;
            component[i][1]=i;
        }
        
        for(int i=1;i<=row+column;++i)
        {
            for(int j=1;j<=row;++j)
            {
                component[i][j]=component[i-1][j-1]+component[i-1][j];
                //cout<<i<<' '<<j<<' '<<component[i][j]<<endl;
            }
        }
        
        string result(row+column,'0');
        //int i=0;
        for(auto& ch:result)
        {
            //cout<<(i++)<<endl;
            if(column==0)
            {
                ch='V';
                continue;
            }
            else if(row==0)
            {
                ch='H';
                continue;
            }
            
            //cout<<k<<' '<<row<<' '<<column<<' '<<component[row+column-1][row]<<endl;
            

            if(k>component[row+column-1][row])
            {
                ch='V';
                k-=component[row+column-1][row];
                --row;
            }
            else
            {
                ch='H';
                --column;
            }
        }
        return result;
    }
};
