class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int n = destination[0], m = destination[1];
        -- k;
        vector<vector<int>> c(n + m + 1, vector<int>(n + m + 1, 0));
        for (int i = 0; i <= n + m; ++ i)
            c[i][0] = c[i][i] = 1;
        for (int i = 1; i <= n + m; ++ i)
            for (int j = 1; j <= i; ++ j) 
                c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
        
        string res = "";
        int x = n, y = m;
        for (int i = 0; i < m + n; ++ i) {
            if (y > 0 && k < c[x + y - 1][y - 1]) {
                --y;
                res += "H";
            }
            else {
                if (y > 0 && x > 0)
                    k -= c[x + y - 1][y - 1];
                --x;
                res += "V";
            }
        }
        return res;
    }
};
