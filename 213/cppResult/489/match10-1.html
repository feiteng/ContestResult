class Solution {
    int comb[34][34];
public:

    void init() {
        for(int i=0;i<=30;i++){
	        comb[i][0]=comb[i][i]=1;
	        for(int j=1;j<i;j++)
		        comb[i][j]=(comb[i-1][j]+comb[i-1][j-1]);

        }
    }
    string solve(int r, int c, int k) {
        if (r == 0) {
            return string(c, 'H');
        }
        if (c == 0) {
            return string(r, 'V');
        }
        int t = r + c;
        int p1 = comb[t - 1][c - 1];

        int p2 = comb[t - 1][r - 1];
        // cout << p1 << ' ' << p2 << endl;
        if (k <= p1) {
            return 'H' + solve(r, c - 1, k);
        } else {
            return 'V' + solve(r - 1, c, k - p1);
        }
    }
    string kthSmallestPath(vector<int>& dest, int k) {
        init();
        int r = dest[0], c = dest[1];
        return solve(r, c, k);
    }
};
