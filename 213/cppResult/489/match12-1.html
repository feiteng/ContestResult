long long c[35][35];
class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int n = destination[0];

        int m = destination[1];
        memset(c, 0, sizeof(c));
        c[0][0] = 1;
        for (int i = 1; i <= n + m; ++ i) {
            c[i][0] = 1;
            for (int j = 1; j <= i; ++ j) {
                c[i][j] = c[i - 1][j - 1] + c[i - 1][j];
            }
        }
        string ans = "";

        while (n || m) {
            if (!m) {
                n --;
                ans += 'V';
                continue;
            }
            if (!n) {
                m --;
                ans += 'H';
                continue;
            }

            if (c[n + m - 1][n] < k) {
                k -= c[n + m - 1][n];
                ans += 'V';
                n --;
            }
            else {
                ans += 'H';
                m --;
            }
        }
        return ans;
    }
};
