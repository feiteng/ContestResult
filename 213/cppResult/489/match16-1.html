class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int n = destination[0];
        int m = destination[1];
        int dp[n+5][m+5];
        memset(dp, 0, sizeof(dp));
        n++,m++;

        for (int i = 0 ; i < m ; i++) {
            dp[n-1][i] = 1;
        }
        for (int j = 0 ; j < n ; j++) {
            dp[j][m - 1] = 1;
        }
        for (int i = n - 2 ; i >= 0 ; i--) {
            for (int j = m - 2 ; j >= 0 ; j--) {
                dp[i][j] = dp[i + 1][j] + dp[i][j + 1];
            }
        }
        int px = 0, py = 0;
        string ret;
        for (int t = 0 ; t < n + m - 2 ; t++) {
            //cout << px <<' '<<py<<endl;
            if (py < m - 1) {
                if (dp[px][py + 1] < k) {
                    k -= dp[px][py + 1];
                    ret += 'V';

                    px++;
                }
                else {
                    py++;
                    ret += 'H';
                }
            }
            else {
                px++;
                ret += 'V';
            }
        }
        return ret;
    }
};
