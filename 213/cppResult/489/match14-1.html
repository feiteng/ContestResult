int c[31][31];

class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        for(int i=0;i<=30;i++)
        {
            c[i][0]=1;
            c[i][i]=1;
        }
        for(int i=2;i<=30;i++)
        {

            for(int j=1;j<i;j++)
            {
                c[i][j]=c[i-1][j-1]+c[i-1][j];
            }
        }
        int ro=destination[0],co=destination[1];
        int tot=ro+co;
        string ans="";
        for(int i=0;i<tot;i++)
        {
            if(co==0)
            {
                ans+="V";
                ro--;
            }
            else if(ro==0)
            {
                ans+="H";
                co--;
            }

            else if(c[ro+co-1][co-1]<k)
            {
                ans+="V";
                k=k-c[ro+co-1][co-1];
                ro--;
            }
            else
            {
                ans+="H";
                co--;
            }
        }
        return ans;
    }
};
