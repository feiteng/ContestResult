class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int dp[35][35]={0};
        for(int i=0;i<=30;i++) {
            dp[i][0]=1;
        }
        for(int i=1;i<=30;i++) {
            for(int j=1;j<=30;j++) {
                dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
            }
        }
        int n=destination[0];
        int sum=destination[0]+destination[1];
        int tmp=k;
        int ans=0;
        for(int i=0;i<n;i++) {
            long long tmpans=0;

            for(int j=n-i;j<=sum;j++) {
                if (dp[j][n-i]>=tmp) {
                    tmp-=dp[j-1][n-i];
                    ans+=(1<<(j-1));
                    cout<<j<<endl;
                    break;
                }
            }
        }
        string t="";

        for(int i=sum-1;i>=0;i--) {
            if ((ans&(1<<i))>0) {
                t+="V";
            }else {
                t+="H";
            }
        }
        return t;
    }
};
