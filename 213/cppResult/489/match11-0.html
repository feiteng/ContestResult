int r, c;
long long nCr[31][31];

class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        init();
        r = destination[0];
        c = destination[1];
        string ans;
        while (r || c) {
            if (r == 0) {
                ans.push_back('H');
                c--;
                continue;
            }
            if (c == 0) {
                ans.push_back('V');
                r--;
                continue;
            }
            if (k <= nCr[r + c - 1][c - 1]) {//오른쪽으로 감
                ans.push_back('H');
                c--;
            }
            else { // 아래로감
                ans.push_back('V');
                k -= nCr[r + c - 1][c - 1];
                r--;
            }
        }
        return ans;
    }
    void init() {
        nCr[1][0] = 1;
        nCr[1][1] = 1;
        for (int i = 2; i < 31; i++) {
            nCr[i][0] = 1;
            nCr[i][i] = 1;
            for (int j = 1; j < i; j++) {
                nCr[i][j] = nCr[i - 1][j - 1] + nCr[i - 1][j];
            }
        }
    }
};
