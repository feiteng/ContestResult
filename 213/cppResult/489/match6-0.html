class Solution {
public:
    int nCr(int n, int k) {
        if (k > n) return 0;
        if (k * 2 > n) k = n-k;
        if (k == 0) return 1;
        
        int result = n;
        for (int i = 2; i <= k; ++i) {
            result *= (n-i+1);
            result /= i;
        }
        return result;
    }
    
    string helper(int v, int h, int k) {
        if (v == 0)
            return string(h, 'H');
        if (h == 0)
            return string(v, 'V');
        
        int chooseh = nCr(v+h-1, h-1);
        // cout << chooseh << endl;
        if (k <= chooseh)
            return "H" + helper(v, h-1, k);
        else
            return "V" + helper(v-1, h, k-chooseh);
        
    }
    
    string kthSmallestPath(vector<int>& destination, int k) {
        int v = destination[0], h = destination[1];
        return helper(v, h, k);
    }
};
