class Solution {
public:
    int C[35][35];
    int f(int n,int m) {
        return C[n + m][n];
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        int n = destination[0], m = destination[1];
        memset(C,0,sizeof(C));
        C[1][1] = 1;
        for(int i = 2;i <= 32;i++) {
            C[i][1] = i;
            for(int j = 2;j <= i;j++) {
                C[i][j] = C[i - 1][j] + C[i - 1][j - 1];
            }
        }
        int v = n,h = m;string res = "";
        for(int i = 0;i < n + m;i++) {
            if(v == 0) {
                res += "H";h--;continue;
            }
            if(h == 0) {
                res += "V";v--;continue;
            }
            int comp = f(v,h - 1);
            // cout<<comp<<endl;
            if(k > comp) {
                res += "V";
                v--;
                k -= comp;
            }else {
                res += "H";
                h--;
            }
        }
        return res;
    }
};
