class Solution {
public:
    string kthSmallestPath(vector<int>& d, int k) {
        int R = d[0] + 1;
        int C = d[1] + 1;
        vector<vector<int>> dp(R, vector<int>(C, 0));
        dp[0][0] = 1;
        for(int r = 0; r < R; r++){
            for(int c = 0; c < C; c++){
                if(r == 0 && c == 0)
                    continue;
                else if(r == 0)
                    dp[r][c] = dp[r][c - 1];
                else if(c == 0)
                    dp[r][c] = dp[r - 1][c];
                else
                    dp[r][c] = dp[r - 1][c] + dp[r][c - 1];
               // printf("dp[%d][%d] = %d    ",r, c, dp[r][c]);
            }
        }
        int r = 0;
        int c = 0;
        string ans = "";
        while(r < R - 1 || c < C - 1){
            if(r == R - 1){
                ans += 'H';
                c++;
                continue;
            }
            if(c == C - 1){
                ans += 'V';
                r++;
                continue;
            }
            if(k <= dp[R - 1 - r][C - 1 - c - 1]){
                ans += 'H';
                c++;
            }
            else{
                ans += 'V';
                k -= dp[R - 1 - r][C - 1- c - 1];
                r++;
            }
            //cout << r << " " << c << " " << ans << " " << k <<endl;
        }
        return ans;
    }
    
};
