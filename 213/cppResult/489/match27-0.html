#include <bits/stdc++.h>
using namespace std;


class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int c[35][35] = {0};
        for(int i = 0; i < 32; i++) {
            for(int j = 0; j <= i; j++) {
                if (j == 0 || j == i) {
                    c[i][j] = 1;
                } else {
                    c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
                }
            }
        }

        int v_num = destination[0];
        int h_num = destination[1];
        int total_len = v_num + h_num;

        string ret(total_len, 'H');
        for (int i = 0; i < v_num; ++i) {
            for (int j = v_num - i; j <= total_len; ++j) {
                // cout << j << " " << v_num - i << " " << c[j][v_num - i] << " " << k << endl;
                if (c[j][v_num - i] >= k) {
                    k -= c[j - 1][v_num - i];
                    ret[total_len - j] = 'V';
                    break;
                }
            }
        }
        return ret;
    }
};
