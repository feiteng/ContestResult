class Solution {
public:
	int nCk[35][40];
	string kthSmallestPath(vector<int>& destination, int k) {
		int h = destination[1];
		int v = destination[0];
		memset(nCk, 0, sizeof nCk);
		nCk[0][0] = 1;
		for (int i = 1; i <= 30; i++) {
			for (int j = 0; j <= i; j++) {
				if (j == 0) nCk[i][j] = 1;
				else nCk[i][j] = nCk[i - 1][j] + nCk[i - 1][j - 1];
			}
		}

		string ans = "";
		while (true) {
			if (h <= 0 && v <= 0) break;
			if (h == 0) {
				ans = ans + "V";
				v--;
				continue;
			}
			if (v == 0) {
				ans = ans + "H";
				h--;
				continue;
			}
			if (nCk[h - 1 + v][h - 1] < k) {
				ans = ans + "V";
				k -= nCk[h + v - 1][h - 1];
				v--;
			}
			else {
				ans = ans + "H";
				h--;
			}
		}
		return ans;
	}
};
