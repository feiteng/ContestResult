class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int nrow = destination[0], ncol = destination[1];
        vector<vector<int>> ways(nrow+1, vector<int>(ncol+1, 0));
        ways[nrow][ncol] = 1;
        for (int i=nrow; i>=0; i--) {
            for (int j=ncol; j>=0; j--) {
                if (i==nrow && j==ncol) continue;
                ways[i][j] = (i==nrow ? 0 : ways[i+1][j]) + (j==ncol ? 0 : ways[i][j+1]);
                //printf("i:%d, j:%d, ways:%d  \n",  i, j, ways[i][j]);
            }
        }
        //cout << "test"<<endl;
        

        int r=0, c=0;
        string path;
        while (r!=nrow || c!=ncol) {
            //cout <<path << "," <<  k << endl;
            if (r==nrow || (c<ncol && ways[r][c+1] >= k)) {
                c++;
                path += 'H';
            } else {
                
                if (c!=ncol) k -= ways[r][c+1];
                path += 'V';
                r++;
            }
        }
        return path;
    }
};
