class Solution {
public:

    long long f(long long a,long long b) {
        long long ans = 1;
        int t = a + b;

        int cnt =1;
        for (int i = t;i > t - a; i--) {
            ans = (ans*i)/cnt;
            cnt++;
        }

        return ans;
    }
    
    string kthSmallestPath(vector<int>& destination, int k) {
        int zero = destination[1];
        int ones = destination[0];
        int total = zero + ones;
        string s = "";
        k--;
        
        for (int i = 0; i < total; i++) {
            if (zero == 0){
                s += "V";
                ones--;
                continue;
            } 
            if (ones == 0) {
                s += "H";
                zero--;
                continue;
            } 
            

            long long calc = f(zero -1, ones);
            if (k >= calc ) {
                k -= calc;
                s += "V";
                ones--;
            } else {
                s += "H";
                zero--;
            }
        }
        
        return s;
    }
};
