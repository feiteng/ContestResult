class Solution {
public:

    long long ncr(int n, int r) {
        long long prod = 1;
        for (int i = r + 1; i <= n; i++) prod *= i;
        for (int i = 1; i <= n - r; i++) prod /= i;
        return prod;
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        string ans = "";
        int sum = destination[0] + destination[1];
        for (int i = 0; i < sum; i++) {
            if (destination[1] && ncr(destination[1]-1+destination[0],destination[0])>=k) {
                --destination[1];
                ans += 'H';
            } else {
                k-= ncr(destination[1]-1+destination[0],destination[0]);
                --destination[0];
                ans += 'V';
            }
        }
        return ans;
    }
};
