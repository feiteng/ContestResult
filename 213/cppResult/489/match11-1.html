class Solution {
public:
    string kthSmallestPath(vector<int>& d, int k) {
        long long c[31][31];

        c[1][0] = 1;
        c[1][1] = 1;
        for (int i = 2; i <= 30; ++i) {
            c[i][0] = 1;
            c[i][i] = 1;
            for (int j = 1; j < i; ++j)
                c[i][j] = c[i-1][j] + c[i-1][j-1];
        }
        string res;
        int v = d[0];

        int h = d[1];
        
        while(v > 0 || h > 0) {
            if (v == 0) {
                --h;
                res.push_back('H');
                continue;
            }
            if (h == 0) {
                --v;
                res.push_back('V');
                continue;
            }
            if (c[h-1+v][v] >= k) {
                --h;
                res.push_back('H');
            }
            else {
                k-=c[h-1+v][v];
                --v;
                res.push_back('V');
            }
        }
        
        return res;
    }
};
