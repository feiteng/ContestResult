typedef long long ll;
class Solution {
public:
    ll c[33][33];
    string kthSmallestPath(vector<int>& destination, int k) {
        for (int i = 0; i <= 30; ++i) {
            c[i][0] = 1;
            for (int j = 1; j <= i; ++j) {
                c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
            }
        }
        int h = destination[1], v = destination[0];
        int n = h + v;
        ll z = k - 1;
        string s = "";
        for (int i = 0; i < n; ++i) {
            ll t = c[n - i - 1][v];
            if (z >= t) {
                z -= t;
                s += 'V';
                --v;
            } else {
                s += 'H';
                --h;
            }
        }
        return s;
    }
};
