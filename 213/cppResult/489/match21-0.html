class Solution {
public:
    string res;
    
    string kthSmallestPath(vector<int>& d, int k) {
        // k--;
        int n = d[0], m = d[1];
        vector<vector<int> > f(n + m + 1, vector<int>(m + n + 1));\
        f[1][1] = 1;
        for(int i = 0; i <= n + m; i++){
            for(int j = 0; j <= i; j++){
                if(!j) f[i][j] = 1;
                else f[i][j] = f[i - 1][j - 1] + f[i - 1][j];
                // cout << f[i][j] << "  ";
            } 
            // cout << endl;
        }
        
        
        
        int x = n, y = m;
        for(int i = 1; i <= m + n && k > 0 && x > 0 && y > 0; i++){
            if(k > f[m + n - i][y - 1]){
                res += 'V';
                k -= f[m + n - i][y - 1];
                x--;
            }else{
                res += 'H';
                y--;
            }
            cout << k << endl;
        }
        
        while(y--) res += 'H';
        while(x--) res += "V";
        return res;
    }

};
