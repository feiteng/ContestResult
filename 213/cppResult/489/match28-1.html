class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int x = destination[0], y = destination[1];
        vector<vector<int>> C(x + y + 1, vector<int>(x + y + 1));
        for (int i = 0; i <= x + y; i++) {
            for (int j = 0; j <= i; j++) {
                if (j == 0) {
                    C[i][j] = 1;
                } else {
                    C[i][j] = 1LL * C[i][j - 1] * (i - j + 1) / j;
                }
            }
        }
        int n = x + y;
        int hremain = y, vremain = x;
        string result;

        for (int i = 0; i < n; i++) {
            if (hremain == 0) {
                result.push_back('V');
                vremain--;
            } else if (vremain == 0) {
                result.push_back('H');
                hremain--;
            } else {
                int tot_behind = C[hremain - 1 + vremain][vremain];

                if (k <= tot_behind) {
                    result.push_back('H');
                    hremain--;
                } else {
                    k -= tot_behind;
                    result.push_back('V');
                    vremain--;
                }
            }
        }
        return result;
    }
};
