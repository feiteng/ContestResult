class Solution {
public:
    // 组合数
    typedef long long ll;
    static constexpr int N = 31;
    int comb[N][N];//comb[n][m]就是C(n,m)
    void init(){
        for(int i = 0; i < N; i ++){
            comb[i][0] = comb[i][i] = 1;
            for(int j = 1; j < i; j ++){
                comb[i][j] = comb[i-1][j] + comb[i-1][j-1];
            }
        }
    }
    string dfs(int x, int y, int k) {
        if (x==0 && y==0) {
            return "";
        }
        if (x==0) {
            return string(y, 'V');
        } else if (y==0) {
            return string(x, 'H');
        }
        int res = comb[x+y-1][y];
        if (k <= res) {
            return "H" + dfs(x-1, y, k);
        } else {
            return "V" + dfs(x, y-1, k-res);
        }
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        int x= destination[1];
        int y = destination[0];
        init();
        return dfs(x, y, k);
    }
};
