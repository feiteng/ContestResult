class Solution {
public:
    int c[30][30];
    int ask(int a, int b, int c1, int d) {
        return c[-b-a+c1+d][c1-a];
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        int n = destination[0], m = destination[1];

        for(int i = 0; i < 30; i++)  c[i][0] = c[i][i] = 1;
        for (int i = 2; i < 30; i++) for(int j = 1; j < i; j ++) c[i][j] = c[i-1][j-1] + c[i-1][j];
        string ans = "";
        int i = 0, j = 0;

        while(i!= n || j != m) {
            if ( j != m && ask(i,j+1,n,m)>=k) {
                ans += 'H'; j++;
            } else {
                if(j != m) k -= ask(i,j+1,n,m);
                ans += 'V'; i++;
            }
        }
        return ans;
    }
};
