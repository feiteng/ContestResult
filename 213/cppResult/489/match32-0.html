class Solution {
public:
    string kthSmallestPath(vector<int>& D, int k) {
		k--;
		long long n=1;
		int b=D[0],a=D[1];
		for(int i=0;i<a;i++){
			n*=a+b-i;
			n/=1+i;
		}
		string ret;
		for(int i=0;i<D[0]+D[1];i++){
			if(k>=n*a/(a+b)){
				k-=n*a/(a+b);
				n=n*b/(a+b);
				b--;
				ret.push_back('V');
			}
			else{
				n=n*a/(a+b);
				a--;
				ret.push_back('H');
			}
		}
		return ret;
    }
};
