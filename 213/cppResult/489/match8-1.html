class Solution {
 public:
  string kthSmallestPath(vector<int>& destination, int k) {
    int hs = destination[1], vs = destination[0];
    vector<vector<int>> dp(hs + 1, vector<int> (vs + 1, 1));
    for (int i = 1; i <= hs; i++) {
      for (int j = 1; j <= vs; j++) {
        dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
      }
    }
    string str = "";

    int i = 0, j = 0;
    while (hs > 0 || vs > 0) {
      if (hs == 0) {
        vs--;
        str += 'V';
      } else if (0 == vs) {
        hs--;
        str += 'H';
      } else {

        if (dp[hs - 1][vs] >= k) {
          hs--;
          str += 'H';
        } else {
          k -= dp[hs - 1][vs];
          vs--;
          str += 'V';
        }
      }
    }
    return str;
  }
};
