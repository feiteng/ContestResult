typedef long long ll;
ll dp[20][20];
class Solution {
public:
	string kthSmallestPath(vector<int>& destination, int k) {
		int n = destination[0];
		int m = destination[1];// 
		memset(dp, 0, sizeof(dp));
		dp[n][m] = 1;
		for (int i = n; i >= 0; i--) {
			for (int j = m; j >= 0; j--) {
				if (i == n && j == m) continue;
				dp[i][j] = dp[i + 1][j] + dp[i][j + 1];
				// cout << i << j << " " << dp[i][j] << endl;
			}
		}
		int curx = 0, cury = 0;
		string str;
		while (1) {
			// cout << curx << " " << cury << "    " << k << endl;
			if (curx == n && cury == m) break;
			if (curx == n) {
				str.push_back('H');
				cury++;
				continue;
			}
			if (cury == m) {
				str.push_back('V');
				curx++;
				continue;
			}
			if (dp[curx][cury + 1] >= k) {
				cury++;
				str.push_back('H');
			}
			else {
				k -= dp[curx][cury + 1];
				curx++;
				str.push_back('V');
			}
		}
		return str;
	}
};
