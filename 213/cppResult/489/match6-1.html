class Solution {
public:
unsigned nChoosek( unsigned n, unsigned k )

{
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0) return 1;

    int result = n;
    for( int i = 2; i <= k; ++i ) {
        result *= (n-i+1);
        result /= i;
    }
    return result;
}
    string helper(int V, int H, int k) {
        if (V == 0) return string(H, 'H');
        if (H == 0) return string(V, 'V');
        int cnt = nChoosek(V + H - 1, H - 1);
        if (cnt >= k) {
            return "H" + helper(V, H - 1, k);
        }

        else return "V" + helper(V - 1, H, k - cnt);
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        int V = destination[0], H = destination[1];
        return helper(V, H, k);
    }

};
