long long c[31][31];
bool flag = false;
void init() {
    c[0][0] = 1;
    for (int i = 1;i <= 30;i++) {
        c[i][0] = 1;
        for (int j = 1;j <= i;j++) {
            c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
        }
    }
    flag = true;
}

class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        if (!flag) {
            init();
        }
        int h = destination[1];
        int w = destination[0];
        
        string ret = "";
        while (h > 0 || w > 0) {
            if (h == 0) {
                ret.push_back('V');
                w--;
            } else if (w == 0) {
                ret.push_back('H');
                h--;
            } else {
                if (k <= c[w + h - 1][h - 1]) {
                    ret.push_back('H');
                    h--;
                } else {
                    ret.push_back('V');
                    k -= c[w + h - 1][h - 1];
                    w--;
                }
            }
        }
        return ret;
    }
};
