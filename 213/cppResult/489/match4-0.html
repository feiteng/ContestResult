class Solution {
public:
    int C[30][30];
    string kthSmallestPath(vector<int>& destination, int k) {
        for(int i=0;i<30;i++){
            C[i][0]=C[i][i]=1;
            for(int j =1;j<i;j++){
                C[i][j]=C[i-1][j]+C[i-1][j-1];
            }
        }
        string ans;
        int nowx=0,nowy=0;
        int tx=destination[0],ty=destination[1];
        k--;
        while(nowx!=tx||nowy!=ty){
            if(nowy==ty||C[tx-nowx+ty-nowy-1][ty-nowy-1]<=k){
                if(nowy!=ty)
                k-=C[tx-nowx+ty-nowy-1][ty-nowy-1];
                nowx++;
                ans+='V';
            }
            else{
                nowy++;
                ans+='H';
            }
          //  printf("%d %d %d\n",nowx,nowy,k);
        }
        return ans;
    }
};
