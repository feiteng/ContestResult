class Solution {
public:
    int f(int n, int m) {
        long long ans = 1;
        for(int i = n + 1; i <= n + m; i++)
            ans *= i;
        for(int i = 1; i <= m; i++)
            ans /= i;
        return ans;
    }
    
    string kthSmallestPath(vector<int>& d, int k) {
        string ans = "";
        int l = d[0] + d[1];
        for(int i = 0; i < l; i++) {
            if(d[1] > 0 && k <= f(d[0], d[1] - 1)) {
                ans += "H";
                d[1]--;
            }
            else {
                ans += "V";
                k -= f(d[0], d[1] - 1);
                d[0]--;
            }
        }
        return ans;
    }
};
