class Solution {
public:
  string kthSmallestPath(vector<int>& destination, int k) {
    string s;
    int r = destination[0], c = destination[1];
    vector<vector<int>> C(32, vector<int>(32));
    C[0][0] = 1;
    for (int i = 1; i < 32; i++) {
      for (int j = 0; j < 32; j++) {
        C[i][j] += C[i-1][j];
        if (j - 1 >= 0) C[i][j] += C[i-1][j-1];
      }
    }

    while (r&&c) {
     // cout << "# " << r << " " << c << " " << C[r+c-1][r] << endl;
      //cout << k << endl;
      
      if (C[r+c-1][r] >= k) {
        s += "H";
        c--;
      } else {
        s += "V";
        k -= C[r+c-1][r];
        r--;
      }
    }
    while (r--) s += "V";
    while (c--) s += "H";
    return s;
  }
};
