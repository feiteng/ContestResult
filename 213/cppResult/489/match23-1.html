class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
      int n = destination[0], m = destination[1];
      ++n; ++m;
      // swap(n, m);
      int dp[n+3][m+3], C[31][31];
      for (int i = 0; i <= 30; ++i) {
        C[i][0] = C[i][i] = 1;
        for (int j = 1; j < i; ++j) C[i][j] = C[i-1][j-1]+C[i-1][j];
      }
      memset(dp, 0, sizeof dp);
      string res;

      for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j) {
          dp[i][j] = C[n-i+m-j][n-i];
        }
      }
      int x = 1, y = 1;

      while (x < n || y < m) {
        if (x == n) {
          res += 'H';
          ++y;
        } else if (y == m) {
          res += 'V';
          ++x;
        } else {
          if (k > dp[x][y+1]) {
            res += 'V';
            k -= dp[x][y+1];
            ++x;
          } else {
            ++y;
            res += 'H';
          }
        }
      }
      /*
      for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j) {
          cout << dp[i][j] << " \n"[j==m];
        }
      }
      */
      return res;
    }
};
