class Solution {
public:
    string kthSmallestPath(vector<int>& a, int k) {
        vector<vector<int>> c(32, vector<int>(32, 0));
        c[0][0] = 1;
        for (int i = 1; i <= 30; i++) {

            c[i][0] = 1; c[i][i] = 1;
            for (int j = 1; j < i; j++) {
                c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
            }
        }
        //cout << 1;
        int p = a[1], q = a[0], r = p + q;
        string ans;
        for (int i = 0; i < r; i++) {
            if (p == 0) {
                ans += 'V';
                continue;
            }
            if (q == 0) {

                ans += 'H';
                continue;
            }
            int t = c[p + q - 1][p - 1];
            //cout << t << ' ' << k << endl;
            if (k > t) {
                k -= t;
                ans += 'V';
                q--;
            } else {
                ans += 'H';
                p--;
            }
        }
        return ans;
    }
};
