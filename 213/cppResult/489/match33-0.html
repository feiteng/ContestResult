class Solution {
public:
    
     #define ull unsigned long long
    
    ull comb(int n, int k) {
        if (k > n) return 0;
        if (k * 2 > n) k = n-k;
        if (k == 0) return 1;

        ull result = n;
        for( int i = 2; i <= k; ++i ) {
            result *= (n-i+1);
            result /= i;
        }
        return result;
    }
    string kthSmallestPath(vector<int>& destination, int k) {
        int a = destination[0];
        int b = destination[1];
        string ans = "";
        
        while(a > 0 || b > 0){
            long c = comb(a+b, b) * b / (a+b);
            if(k > c) {
                ans += 'V';
                --a;
                k -= c;
            } else {
                ans += 'H';
                --b;
                // k-=0;
            }
        }
        return ans;
    }
};
