class Solution {
long long bicof[31][31];
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        string ans = "";
        k--;
        memset(bicof, 0, sizeof(bicof));
        for (int i=0; i<= 30; i++) {
            bicof[i][0] = 1;
            bicof[i][i] = 1;
        }
        for (int i=1; i <= 30; i++) {
            for (int j=1; j < i; j++) {
                bicof[i][j] = bicof[i-1][j-1] + bicof[i-1][j];
            }
        }
        
        int row = destination[0];
        int col = destination[1];
        // H가 col개, V가 row개 있다.
        int hn = col;
        int vn = row;
        // ["HHHVV", "HHVHV", "HHVVH", "HVHHV", "HVHVH", "HVVHH", "VHHHV", "VHHVH", "VHVHH", "VVHHH"].
        while (hn + vn > 0) {
            if (hn == 0) {
                vn--;
                ans += "V";
            } else if (vn == 0) {
                hn--;
                ans += "H";
            } else {
                if (bicof[hn+vn-1][hn-1] <= k) {
                    k -= bicof[hn+vn-1][hn-1];
                    ans += "V";
                    vn--;
                } else {
                    ans += "H";
                    hn--;
                }
            }
        }
        
        return ans;
    }
};
