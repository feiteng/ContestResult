class Solution {
public:
    int choose[31][31];
    string helper(int i, int j, int k){
        if(k==0) return "";
        if(i+j == 0) return "";

        if(k <= choose[i+j-1][i]){
            return 'H' + helper(i,j-1,k);
        }else{
            return 'V' + helper(i-1,j,k-choose[i+j-1][i]);
        }
    }
    string kthSmallestPath(vector<int>& d, int k) {

        choose[0][0] = 1;
        for(int i=1;i<31;i++){
            choose[i][0] = 1;
            choose[i][i] = 1;
            for(int j=1;j<i;j++){
                choose[i][j] = choose[i-1][j] + choose[i-1][j-1];
            }
        }
        return helper(d[0],d[1],k);
        
    }
};
