int f[20][20];

class Solution {
public:
    string kthSmallestPath(vector<int>& dest, int k) {
        int x = 0;
        int y = 0;
        vector<char> a;
        

        int n = dest[0];
        int m = dest[1];
        memset(f,0,sizeof(f));
        f[n][m] = 1;
        for (int i = n; i >= 0; --i) {
            for (int j = m; j >= 0; --j) {
                if (i+1 <= n) f[i][j] += f[i+1][j];
                if (j+1 <= m) f[i][j] += f[i][j+1];
            }
        }
        
      
        while (x < n || y < m) {

            if (y >= m) {
                a.push_back('V');
                x++;
                continue;
            }
            if (x >= n) {
                a.push_back('H');
                y++;
                continue;
            }
            
            if (k <= f[x][y+1]) {
                a.push_back('H');
                y++;
            }
            else {
                k -= f[x][y+1];
                a.push_back('V');
                x++;
            }
            
            
        }
        
        string ans(a.begin(), a.end());
        return ans;
    }
};
