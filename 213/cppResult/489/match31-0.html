class Solution {
public:
    string solve(vector<vector<int>>& f, int k, int m, int n)
    {
        if (k == 1)
        {
            string ans;
            for (int i = 0; i < n; i++) ans += "H";
            for (int i = 0; i < m; i++) ans += "V";
            return ans;
        }
        string ans;
        if (k <= f[m + n - 1][n - 1])
        {
            return "H" + solve(f, k, m, n - 1);
        }
        else
        {
            return "V" + solve(f, k - f[m + n - 1][n - 1], m - 1, n);
        }
    }
    
    string kthSmallestPath(vector<int>& d, int k) {
        int m = d[0], n = d[1];
        vector<vector<int>> dp(m + n + 1, vector<int>(m + n + 1));
        dp[0][0] = 1;
        dp[1][0] = dp[1][1] = 1;
        for (int i = 2; i <= m + n; i++)
        {
            dp[i][0] = dp[i][i] = 1;
            for (int j = 1; j < i; j++)
            {
                dp[i][j] = dp[i - 1][j] + dp[i - 1][j - 1];
            }
        }
        return solve(dp, k, m, n);
    }
};
