class Solution {
public:
    int f[20][20];
    string kthSmallestPath(vector<int>& destination, int k) {
        int m = destination[0];
        int n = destination[1];

        memset(f, 0, sizeof(f));
        for (int i = 1; i <= 15; i++){
            f[i][0] = 1;
            f[0][i] = 1;
        }
        for (int i = 1; i <= n; i++){
            for (int j = 1; j <= m; j++){
                f[i][j] = f[i - 1][j] + f[i][j - 1];
            }
        }
        int x = n, y = m;
        string ans = "";
        k--;
        while (k){
            if (k >= f[x - 1][y]){
                k -= f[x - 1][y];
                ans += 'V';
                y--;
            }
            else{
                ans += 'H';
                x--;
            }
        }
        for (int i = 0; i < x; i++){
            ans += 'H';
        }
        for (int i = 0; i < y; i++){
            ans += 'V';
        }
        return ans;
    }
};
