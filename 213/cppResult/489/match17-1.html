class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int r = destination[0] + 1, c = destination[1] + 1;
        vector<vector<int>> paths(r, vector<int>(c, 0));
        for (int i = 0; i < r; ++i) {
            paths[i][c - 1] = 1;
        }
        for (int j = 0; j < c; ++j) {
            paths[r - 1][j] = 1;
        }
        for (int i = r - 2; i >= 0; --i) {
            for (int j = c - 2; j >= 0; --j) {
                paths[i][j] = paths[i + 1][j] + paths[i][j + 1];
            }
        }
        ostringstream ans;
        int p1 = 0, p2 = 0;
        while (p1 != r - 1 || p2 != c - 1) {
            if (p2 == c - 1) {
                ans << 'V';
                ++p1;
            } else if (p1 == r - 1) {
                ans << 'H';
                ++p2;
            } else {
                if (k <= paths[p1][p2 + 1]) {
                    ans << 'H';
                    ++p2;
                } else {
                    ans << 'V';
                    k -= paths[p1][p2 + 1];
                    ++p1;
                }
            }
        }
        return ans.str();
    }
};
