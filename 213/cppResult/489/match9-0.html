class Solution {
public:
    string kthSmallestPath(vector<int>& d, int k) {
        int m = d[0] + 1, n = d[1] + 1;
        vector<vector<int>> mat(m, vector<int>(n, 1));
        
        for(int i = m-2; i >=0; --i) {
            for(int j = n-2; j >=0; --j) {
                mat[i][j] = mat[i+1][j] + mat[i][j+1]; 
            }
        }
        
        
        // for(auto x : mat) {
        //     for(int y : x) {
        //         cout << y << "->";
        //     }
        //     cout << endl;
        // }
        
        int i = 0, j = 0;
        
        string res = "";
        while(i!= m-1 || j != n-1) {
            if(i == m-1) {
                res += "H";
                j++;
                continue;
            } else if (j == n-1) {
                res += "V";
                i++;
                continue;
            }       
            
            if(mat[i][j+1] >=k) {
                res += "H";
                j++;
            } else {
                res += "V";
                k -= mat[i][j+1];
                i++;
            }
        }
        
        
        return res;
    }
};
