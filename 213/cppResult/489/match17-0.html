class Solution {
public:
    string kthSmallestPath(vector<int>& destination, int k) {
        int m = destination[0] + 1, n = destination[1] + 1;
        vector<vector<int>> dp(m, vector<int>(n, 0));
        for (int i = n - 1; i >= 0; --i) {
            dp[m - 1][i] = 1;
        }
        for (int i = m - 1; i >= 0; --i) {
            dp[i][n - 1] = 1;
        }
        for (int i = m - 2; i >= 0; --i) {
            for (int j = n - 2; j >= 0; --j) {
                dp[i][j] = dp[i + 1][j] + dp[i][j + 1];
            }
        }
        int x = 0, y = 0;
        string res;
        while (!(x == m - 1 && y == n - 1)) {
            if (y < n - 1) {
                if (k <= dp[x][y + 1]) {
                    res.push_back('H');
                    ++y;
                }
                else {
                    res.push_back('V');
                    k -= dp[x][y + 1];
                    ++x;
                }
            } 
            else {
                res.push_back('V');
                k--;
                ++x;
            }
        }
        return res;
    }
};
