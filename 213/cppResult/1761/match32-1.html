class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int> > dp(n + 1, vector<int>(5, 0));
    for(int i = 0; i < 5; i++) dp[1][i] = 1;
    for(int i = 2; i <= n; i++)
    {
        dp[i][0] = dp[i - 1][0];
        for(int j = 1; j < 5; j++)

            dp[i][j] = dp[i][j - 1] + dp[i - 1][j];
    }
    int ans = 0;
    for(int i = 0; i < 5; i++) ans += dp[n][i];
    return ans;
    }
};
