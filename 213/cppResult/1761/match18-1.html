class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> F(n+1, vector<int>(5,1));
        for(int k = 2; k <= n; k++)
        {
            for(int e = 0; e < 5; e++)
            {
                F[k][e] = 0;
                for(int i = 0; i <= e; i++)
                {
                    F[k][e] += F[k-1][i];
                }
            }
        }
        
        int total = 0;
        for(int i = 0; i < 5; i++)
        {
            total += F[n][i];
        }
        
        return total;
    }
};
