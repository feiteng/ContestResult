class Solution {
public:
    int countVowelStrings(int n) {
        int a[5] = {1, 1, 1, 1, 1};
        for(int i = 1; i < n; ++i){
            for(int j = 3; j >= 0; --j){
                a[j] += a[j + 1];
            }
        }
        return a[0] + a[1] + a[2] + a[3] + a[4];
    }
};
