class Solution {
public:
    int countVowelStrings(int n) {
        vector <vector <int> >dp(n + 5, vector <int>(5, 0));
        for (int i = 0; i < 5; i++) dp[1][i] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 5; j++) {

                for (int k = 0; k <= j; k++)
                    dp[i + 1][k] += dp[i][j];
            }
        }
        int sum = 0;
        for (int i = 0; i < 5; i++) sum += dp[n][i];
        return sum;
    }
};
