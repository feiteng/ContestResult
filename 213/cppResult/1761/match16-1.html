class Solution {
public:
    
    int f[55][6];
    
    int countVowelStrings(int n) {
        for (int i = 0; i < 5; i ++)
            f[1][i] = 1;
        for (int i = 2; i <= n; i ++) {
            for (int j = 0; j < 5; j ++)
                for (int k = 0; k <= j; k ++)
                    f[i][j] += f[i - 1][k];
        }
        int sum = 0;
        for (int i = 0; i < 5; i ++)
            sum += f[n][i];
        return sum;
    }
};
