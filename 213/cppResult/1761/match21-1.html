class Solution {
public:
    int dp[100][10];
    int countVowelStrings(int n) {
        dp[1][1] = dp[1][2] = dp[1][3] = dp[1][4] = dp[1][5] = 1;
        for(int i = 2; i <= n + 1; i++) {
            for(int j = 1; j <= 5; j++) {
                for(int k = 1; k <= j; k++) {
                    dp[i][j] += dp[i - 1][k];
                }
            }
        }
        return dp[n + 1][5];
    }
};
