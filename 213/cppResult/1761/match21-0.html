class Solution {
public:
    int f[55][6];
    int countVowelStrings(int n) {
        f[1][5] = 5;
        f[1][4] = 4;
        f[1][3] = 3;
        f[1][2] = 2;
        f[1][1] = 1;
        for (int i = 2; i <= n; i++){
            for (int j = 1; j <= 5; j++){
                for (int k = 1; k <= j; k++){
                    f[i][j] += f[i - 1][k];
                }
            }
        }
        return f[n][5];
    }
};
