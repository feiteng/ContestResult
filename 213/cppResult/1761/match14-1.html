/*
dp[n][k] = dp[n-1][k] + dp[n-1][k-1] + ... +dp[n-1][1]

dp[x][1] = 1

*/



class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> dp(n+1, vector<int>(6, 0));
        for (int j=1; j<=5; j++) dp[1][j] = j;
        for (int i=2; i<=n; i++) {
            for (int k=1; k<=5; k++) {
                for (int j=1; j<=k; j++) {
                    dp[i][k] += dp[i-1][j];
                }
            }
        }
        return dp[n][5];
    }
};
