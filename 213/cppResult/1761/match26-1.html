class Solution {
public:
    int cnt[5][51];
    int countVowelStrings(int n) {
        memset(cnt, 0, sizeof(cnt));
        for (int i = 0; i < 5; ++i) {
            cnt[i][1] = 1;
        }
        for (int i = 2; i <= n; ++i) {
            for (int j = 0; j < 5; ++j) {
                for (int k = 0; k <= j; ++k) {
                    cnt[j][i] += cnt[k][i-1];
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 5; ++i) {
            ans += cnt[i][n];
        }
        return ans;
    }
};
