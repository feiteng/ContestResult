class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> ans(n+5, vector<int>(5, 0));
        for(int i = 0; i < 5; i++) {
            ans[1][i] = 1;
        }
        for(int i = 2; i <= n; i++) {
            for(int j = 0; j < 5; j++) {
                for(int k = 0; k <= j; k++) {
                    ans[i][j] += ans[i-1][k];
                }
            }
        }
        int ret = 0;
        for(int i = 0; i < 5; i++) {
            ret += ans[n][i];
        }
        return ret;
    }
};
