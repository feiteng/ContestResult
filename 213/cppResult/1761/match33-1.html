class Solution {
public:
    int countVowelStrings(int n) {
        int dp[51][6];
        memset(dp, 0, sizeof(dp));
        for(int i = 1; i <= 5; i++) dp[1][i] = 1;
        for(int i = 2; i <= n; i++){
            dp[i][1] = 1;

            for(int j = 2; j <= 5; j++){
                for(int k = 1; k <= j; k++) dp[i][j] += dp[i - 1][k];
            }
        }
        int res = 0;
        for(int i = 1; i <= 5; i++) res += dp[n][i];
        return res;
    }
};
