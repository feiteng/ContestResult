class Solution {
public:
    int countVowelStrings(int n) {
        int a = 1, e = 1, i = 1, o = 1, u = 1;
        
        for(int x = 2; x <= n; x ++)
        {
            int at = a, et = a + e, it = a + e + i, ot = a + e + i + o, ut = a + e + i + o + u;
            a = at, e = et, i = it, o = ot, u = ut;
        }
        
        return a + e + i + o + u;
    }
};
