int a[51][5];
bool flag = false;
void init() {
    if (flag) {
        return;
    }
    for (int i = 0;i < 5;i++) {
        a[1][i] = 1;
    }
    for (int j = 2;j <= 50;j++) {
        for (int i = 0;i < 5;i++) {
            for (int k = 0;k <= i;k++) {
                a[j][i] += a[j - 1][k];
            }
        }
    }
    flag = true;
}

class Solution {
public:
    int countVowelStrings(int n) {
        init();
        int s = 0;
        for (int i = 0;i < 5;i++) {
            s += a[n][i];
        }
        return s;
    }
};
