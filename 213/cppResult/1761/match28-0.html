class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> dp(5, vector<int>(n,0));
        for(int i=0; i<5; i++)
            dp[i][0] =i+1;
        
        for(int i=1; i<n; i++)
        {
            for(int j=0; j<=4; j++)
            {
                for(int k=j; k>=0; k--)
                {
                    dp[j][i]+=dp[k][i-1];
                }
                
                //cout<<j<<" "<<i<<" "<<dp[j][i]<<endl;
            }
        }
        
        return dp[4][n-1];
    }
};
