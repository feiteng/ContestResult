class Solution {
public:
    int countVowelStrings(int n) {
        // a e i o u
        vector<vector<int>> dp(5, vector<int>(n + 1, 0));
        // dp[s][len] is length of strings starts with s and using length len
        for (int s = 0; s < 5; ++s) {
            dp[s][1] = 1;
        }
        // dp[s][len] = dp[<=s][len - 1]
        for (int len = 2; len <= n; ++len) {
            for (int s = 0; s < 5; ++s) {
                for (int nexts = s; nexts < 5; ++nexts) {
                    dp[s][len] += dp[nexts][len - 1];
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 5; ++i) {
            ans += dp[i][n];
        }
        return ans;
    }
};
