class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> dp(n, vector<int>(5, 0));
        for (int i = 0; i < 5; ++i) dp[0][i] = 1;
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < 5; ++j) {
                for (int k = 0; k <= j; ++k) {
                    dp[i][j] += dp[i-1][k];
                }
            }
        }
        return accumulate(begin(dp.back()), end(dp.back()), 0);
    }
};
