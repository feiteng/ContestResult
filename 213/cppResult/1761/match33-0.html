class Solution {
public:
    int countVowelStrings(int n) {
        
        int dp[5][n+1];
        memset(dp,0,sizeof(dp));
        for(int i=0;i<5;i++)
            dp[i][1]=1;
        
        for(int j=1;j<n;j++)
        {
            for(int i=0;i<5;i++)
            {
                for(int w=i;w<5;w++)
                    dp[w][j+1]+=dp[i][j];
            }
        }
        int ans=0;
        for(int i=0;i<5;i++)
            ans+=dp[i][n];
        return ans;
    }
};
