class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int> > f(n, vector<int>(5, 0));
        for (int i=0;i<5;++i) f[0][i] = 1;
        for (int i=1;i<n;++i) {
            f[i][0] = f[i-1][0];
            f[i][1] = f[i-1][0]+f[i-1][1];
            f[i][2] = f[i-1][0]+f[i-1][1]+f[i-1][2];
            f[i][3] = f[i-1][0]+f[i-1][1]+f[i-1][2]+f[i-1][3];
            f[i][4] = f[i-1][0]+f[i-1][1]+f[i-1][2]+f[i-1][3]+f[i-1][4];
        }
        return f[n-1][0]+f[n-1][1]+f[n-1][2]+f[n-1][3]+f[n-1][4];
    }
};
