class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> c(n+1, vector<int>(5));
        c[1][0] = c[1][1] = c[1][2] = c[1][3] = c[1][4] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < 5; j++) {
                for (int k = j; k < 5; k++) {
                    c[i][j] += c[i-1][k];
                }
            }
        }
        return c[n][0] + c[n][1] + c[n][2] + c[n][3] + c[n][4];
    }
};
