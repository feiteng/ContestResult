class Solution {
public:
    int countVowelStrings(int n) {
        vector<vector<int>> dp(51,vector<int>(5,1));
        
        for(int i = 1;i<50;++i){
            for(int j = 0;j<5;++j){
                dp[i][j]--;
                for(int k = j;k<5;++k){
                    dp[i][j]+=dp[i-1][k];
                }
            }
        }
        int sum = 0;
        for(int i = 0;i<5;++i){
            sum+=dp[n-1][i];
        }
        return sum;
    }
};
