class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        vector<int> idx(101, -1);
        for (int i = 0; i < pieces.size(); ++i) {
            idx[pieces[i][0]] = i;
        }
        int cur = 0;
        while (cur < arr.size()) {
            int num = arr[cur];
            if (idx[num] < 0) return false;
            int j = idx[num];

            for (int i = 0; i < pieces[j].size(); ++i) {
                if (cur + i >= arr.size() || pieces[j][i] != arr[cur+i]) return false;
            }
            cur += pieces[j].size();
        }
        return true;
    }
};
