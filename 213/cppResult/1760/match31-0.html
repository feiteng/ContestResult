class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        map<int, int> m;
        for (int i = 0; i < pieces.size(); ++i) {
            m[pieces[i][0]] = i;
        }
        for (int i = 0; i < arr.size();) {
            auto it = m.find(arr[i]);
            if (it == end(m)) {
                return false;
            }
            int pi = it->second, pn = pieces[pi].size();
            if (i + pn > arr.size()) {
                return false;
            }
            for (int j = 0; j < pn; ++j) {
                if (arr[i + j] != pieces[pi][j]) {
                    return false;
                }
            }
            i += pn;
            m.erase(it);
        }
        return true;
    }
};
