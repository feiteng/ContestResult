class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        int m = pieces.size();
        int n = arr.size();
        
        int cur = 0;
        while(cur < n) {
            int s = -1;
            for(int i=0;i<m;i++) {
                if(arr[cur] == pieces[i][0]) {
                    s = i;
                    break;
                }
            }
            if(s==-1) return false;
                for(int j=0;j<pieces[s].size();j++) {
                    if(arr[cur] != pieces[s][j]) return false;
                    cur++;
                }
        }
        
        return true;
    }
};
