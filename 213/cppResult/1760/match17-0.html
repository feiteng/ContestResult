class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        for (int i = 0; i < arr.size(); ++ i) {
            bool flag = false;
            for (int j = 0; j < pieces.size(); ++ j) {
                if (pieces[j][0] == arr[i]) {
                    flag = true;
                    for (int k = 0; k < pieces[j].size(); ++ k) {
                        if (pieces[j][k] != arr[i + k])
                            return false;
                    }
                    i += pieces[j].size() - 1;
                    break;
                }
            }
            if (!flag)
                return false;

        }
        return true;
    }
};
