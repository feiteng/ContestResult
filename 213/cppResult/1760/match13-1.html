class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int, int> mp;
        for (int i = 0; i < pieces.size(); i++) {
            mp[pieces[i][0]] = i;
        }
        const int n = arr.size();
        int i = 0;

        while (i < n) {
            if (mp.find(arr[i]) == mp.end()) return false;
            int k = mp[arr[i]];
            for (int j = 0; j < pieces[k].size(); j++) {
                //cout<<arr[i]<<", "<<pieces[k][j]<<endl;
                if (i == n || arr[i] != pieces[k][j]) return false;
                i++;
            }
        }
        return true;
    }
};
