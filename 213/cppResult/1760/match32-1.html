class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        map<int, int> a;
        for(int i = 0; i < pieces.size(); i++){
            a[pieces[i][0]] = i; 
        }
        int idx = 0;
        while(idx < arr.size()){
            if(!a.count(arr[idx])) return false;
            int ar = a[arr[idx]];
            int i = 0;
            while(i < pieces[ar].size() && arr[idx++] == pieces[ar][i++]);
        }
        return true;
    }
};
