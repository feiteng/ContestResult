class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {

        int n = pieces.size(), m=arr.size();
       // vector<int>visited(n, false);
        int i=0;
        while(i<m) {
            int idx=-1;
            for (int j=0; j<n; j++) {
                if (pieces[j][0]==arr[i]) {
                    idx=j;
                    break;
                }
            }
            if (idx==-1)
                return false;
            for (int k=0; k<pieces[idx].size(); k++) {
                if (pieces[idx][k]!=arr[i])
                    return false;
                i++;
            }
        }
        return true;
    }
};
