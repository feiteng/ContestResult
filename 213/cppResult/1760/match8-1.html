class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int,int> umap;
        for(int i =0;i<pieces.size();i++){
            umap[pieces[i][0]] = i;
        }
        for(int i =0;i<arr.size();){

            if(!umap.count(arr[i])){
                return false;    
            }
            int p = umap[arr[i]];
            for(int j =0;j<pieces[p].size();j++,i++){
                if(arr[i]!=pieces[p][j]){
                    return false;
                }
            }
        }
        return true;
    }
};
