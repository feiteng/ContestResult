class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        for(int i=0;i<arr.size();){
            bool flag=false;
            for(int j=0;j<pieces.size();++j){
                if(arr[i]==pieces[j][0]){
                    flag=true;
                    for(int k=0;k<pieces[j].size();++k){
                        if(pieces[j][k]!=arr[i])
                            return false;
                        ++i;
                    }
                    break;
                }
            }
            if(flag==false)
                return false;
        }
        return true;
    }
};
