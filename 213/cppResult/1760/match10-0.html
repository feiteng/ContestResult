class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& p) {
        unordered_map<int, int> m;
        for (int i = 0; i < arr.size(); i++)
            m[arr[i]] = i;
        
        for (int i = 0; i < p.size(); i++) {
            int s = 0;
            if (m.count(p[i][0]) == 0)
                return false;
            else
                s = m[p[i][0]];
            for (int j = 1; j < p[i].size(); j++) {
                if (m.count(p[i][j]) == 0)
                    return false;
                if (m[p[i][j]] != s + 1)
                    return false;
                s = m[p[i][j]];
            }
        }
        
        return true;
    }
};
