class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int,int> hashmap;
        for (int i = 0; i < arr.size(); i++)
            hashmap[arr[i]] = i;
        
        for (auto & p : pieces) {
            if (hashmap.count(p[0]) == 0)
                return false;

            int lastPos = hashmap[p[0]];
            for (int i = 1; i < p.size(); i++) {
                if (hashmap.count(p[i]) == 0)
                    return false;
                
                if (hashmap[p[i]] != lastPos + 1)
                    return false;
                
                lastPos = hashmap[p[i]];
            }
        }
        
        return true;
    }
};
