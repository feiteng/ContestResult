class Solution {
public:
    // brute-force
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        if (arr.size() == 1) {
            return (pieces[0][0] == arr[0]);
        }

        for (int i = 0; i < arr.size();) {
            bool findHead = false;
            for (int j = 0; j < pieces.size(); ++j) {
                if (pieces[j][0] == arr[i]) {
                    findHead = true;

                    i++;
                    for (int k = 1; k < pieces[j].size(); ++k) {
                        if (pieces[j][k] != arr[i]) return false;
                        ++i;
                    }
                    break;
                }
            }
            if (!findHead) return false;
        }
        return true;
    }
};
