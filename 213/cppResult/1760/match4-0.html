class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        vector<int> firstIdxs(101, -1);
        for (int i=0; i<pieces.size(); i++) {
            firstIdxs[pieces[i][0]] = i;
        }
        int i = 0;
        while (i < arr.size()) {
            int firstIdx = firstIdxs[arr[i]];
            if (firstIdx == -1)
                return false;
            if (arr.size() <= i+pieces[firstIdx].size()-1)
                return false;
            for (int j=0; j<pieces[firstIdx].size(); j++) {
                if (arr[i+j] != pieces[firstIdx][j])
                    return false;
            }
            i += pieces[firstIdx].size();
        }
        return i == arr.size();
    }
};
