class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        int i=0;
        while(i<arr.size()){
            int found=-1;
            for(int j=0;j<pieces.size();j++){
                if(arr[i]==pieces[j][0]){
                    found=j;
                }
            }
            if(found==-1)return false;
            for(int j=0;j<pieces[found].size();j++){
                if(arr[i]!=pieces[found][j])return false;
                i++;
            }
        }
        return true;
    }
};
