class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        int pos=0;

        bool flag=false;
        while (pos<arr.size()){
            flag=false;
            for (int i=0;i<pieces.size();i++){
                if (pieces[i][0]==arr[pos]){
                    flag=true;
                    int j=0;
                    while (j<pieces[i].size()){
                        if (arr[pos]!=pieces[i][j]) return false;
                        pos++;
                        j++;
                    }
                    
                    break;
                }
            }
            if (flag==false) return false;
        }
        return true;
    }
};
