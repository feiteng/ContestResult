class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces)
    {
        unordered_map<int, vector<int>> mm;
        for (auto& v : pieces)
        {
            mm[v[0]] = v;
        }

        int idx = 0;
        while (idx < arr.size())
        {
            int remain = arr.size() - idx;
            int first = arr[idx];
            if (mm.count(first) == 0)
            {
                return false;
            }
            if (mm[first].size() > remain)
            {
                return false;
            }
            auto &v = mm[first];

            for (int i : v)
            {
                if (i != arr[idx])
                {
                    return false;
                }
                ++idx;
            }
        }
        return true;
    }
};
