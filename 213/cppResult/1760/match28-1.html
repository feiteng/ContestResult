class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        map<int,int> m;
        for (int i = 0; i < pieces.size(); ++i) {
            m[pieces[i][0]] = i;
        }
        
        int cur = 0;
        while (cur < arr.size()) {
            if (m.find(arr[cur]) == m.end()) {
                return false;
            }
            
            int idx = m[arr[cur]];

            const auto& ps = pieces[idx];
            for (int i = 0 ; i < ps.size(); ++i) {
                if (arr[cur+i] != ps[i]) 
                    return false;
            }
            cur += ps.size();
        }
        return cur == arr.size();
    }
};
