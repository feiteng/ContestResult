class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int, int> mp;
        int n = arr.size(), m = pieces.size();
        for (int i = 0; i < m; i++)
        {
            mp[pieces[i][0]] = i;
        }
        int i = 0, j = 0;
        while (i < n)

        {
            if (mp.find(arr[i]) == mp.end()) return false;
            int idx = mp[arr[i]];
            for (int j = 0; j < pieces[idx].size(); j++)
            {
                if (arr[i] != pieces[idx][j]) return false;
                ++i;
            }
        }
        return true;
    }
};
