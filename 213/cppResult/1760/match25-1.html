class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        for (int i = 0; i < arr.size(); ) {
            bool flag = false;
            for (int j = 0; j < pieces.size(); ++j) {
                if (arr[i] == pieces[j][0]) {
                    flag = true;
                    for (int k = 1; k < pieces[j].size(); ++k) {
                        if (arr[i + k] != pieces[j][k]) {
                            flag = false;
                            break;
                        }
                    }

                    if (!flag) return false;
                    i += pieces[j].size();
                    break;
                }
            }
            
            if (!flag) return false;
        }
        return true;
    }
};
