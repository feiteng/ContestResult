class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& p) {
        unordered_map<int,int> map;
        int n=arr.size(),m=p.size();
        for(int i=0;i<m;i++)map[p[i][0]]=i;
        int i;
        for(int i=0;i<n;){
            if(!map.count(arr[i]))return false;
            int idx=map[arr[i]];
            for(int j=0;j<p[idx].size();j++){
                if(arr[i]!=p[idx][j])return false;
                i++;
            }
        }
        return true;
    }
};
