class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) 
    {
        unordered_map<int, int> m;
        for(int i = 0; i < pieces.size(); i++)
        {
            m[pieces[i][0]] = i;
        }
        
        int i = 0;
        while(i < arr.size())
        {
            if(!m.count(arr[i]))
                return false;
            int j = m[arr[i]];
            int k = 0;
            while(i < arr.size() && k < pieces[j].size())
            {
                if(arr[i] != pieces[j][k])
                    return false;
                i++;
                k++;
            }
            if(k != pieces[j].size())
                return false;
        }
        return true;
    }
};
