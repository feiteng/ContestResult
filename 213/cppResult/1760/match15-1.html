class Solution {
public:

    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int, vector<int>> m;
        for(int i = 0; i < pieces.size(); i ++){
            m[pieces[i][0]] = pieces[i];
        }
        int i = 0;
        while(i < arr.size()){

            int first_tag = arr[i];
            if(m.count(first_tag) == 0) return false;
            for(int j = 0; j < m[first_tag].size(); j ++){
                if(m[first_tag][j] != arr[i])   return false;
                i += 1;
            }
            m.erase(first_tag);
        }
        return true;
    }
};
