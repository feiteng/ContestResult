class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        vector<int>v(101,-1);
        for(int i=0;i<arr.size();++i)
            v[arr[i]]=i;
        for(auto& ele:pieces){
            if(v[ele[0]]==-1)
                return false;
            int pre = v[ele[0]];
            v[ele[0]]=-1;
            for(int i=1;i<ele.size();++i){
                if(v[ele[i]]!=pre+i)
                    return false;
                v[ele[i]]=-1;
            }
        }
        return true;
    }
};
