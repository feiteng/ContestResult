class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        int i = 0;
        while (i < arr.size())

        {
            int last_i = i;
            for (int j = 0; j < pieces.size(); ++j)
            {
                if (arr[i] == pieces[j][0])
                {
                    for (int k = 0; k < pieces[j].size(); ++k)
                    {
                        if (arr[i + k] != pieces[j][k]) return false;
                    }
                    i += pieces[j].size();
                    break;
                }
            }
            if (i == last_i) return false;
        }
        return true;
    }
};
