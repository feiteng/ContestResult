class Solution {
public:
    bool canFormArray(vector<int>& a, vector<vector<int>>& pieces) {
        int n = a.size();
        
        int i = 0;
        while(i < n) {
            bool f = false;
            for(int j = 0; j < pieces.size(); j++) {
                if(a[i] == pieces[j][0]) {
                    f = true;
                    int k = 0;
                    while(k < pieces[j].size()) {
                        if(a[i] != pieces[j][k]) return false;
                        i++, k++;
                    }
                    break;
                }
            }
            if(!f) return false;
        }
        return true;
    }
};
