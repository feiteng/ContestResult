class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        for(int i=0;i<pieces.size();i++){
            bool f = false;
            for(int j=0;j<arr.size();j++){
                if(pieces[i][0] == arr[j]){
                    f = true;
                    for(int k=1;k<pieces[i].size();k++){
                        if(j+k >= arr.size() || pieces[i][k] != arr[j+k])
                            return false;
                    }
                }
            }
            if(f == false) return false;
        }
        return true;
    }
};
