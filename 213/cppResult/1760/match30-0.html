class Solution {
public:
    bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
        unordered_map<int, int> m;
        for(int  i= 0; i < arr.size(); i++) m[arr[i]] = i;
        vector<int> visited(arr.size());
        for(int i = pieces.size() - 1; i >= 0; i--) {
            vector<int> v = pieces[i];
            if (m.find(v[0]) == m.end()) return false;
            if (m[v[0]] + v.size() > arr.size()) return false;
            for(int j = 0; j < v.size(); j++) {
                if (v[j] != arr[m[v[0]] + j]) return false;
                visited[m[v[0]]+j] = false;
            }
        } return true;
    }
};
