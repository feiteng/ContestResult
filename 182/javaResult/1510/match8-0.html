class Solution {
    public int findLucky(int[] arr) {
        int[] c = new int[501];
        for (int x : arr) {
            c[x]++;
        }
        
        for (int i = 500; i >= 1; i--) {
            if (c[i] == i) {
                return i;
            }
        }
        
        return -1;
    }
}
