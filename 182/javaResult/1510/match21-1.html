class Solution {
    public int findLucky(int[] arr) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i: arr) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        
        int max = -1;
        for (int k: map.keySet()) {
            if (k == map.get(k)) {
                max = Math.max(max, k);
            }
        }
        return max;
    }
}
