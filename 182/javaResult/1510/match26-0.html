class Solution {
    public int findLucky(int[] arr) {
        Map<Integer, Integer> dict = new HashMap<>();
        for(int num: arr) {
            dict.put(num, dict.getOrDefault(num, 0) + 1);
        }
        
        int res = -1;
        for(int key: dict.keySet()) {
            if (key == dict.get(key)) {
                res = Math.max(res, key);
            }
        }
        
        return res;
    }
}
