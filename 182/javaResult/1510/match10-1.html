class Solution {
    public int findLucky(int[] arr) {
        int[] a = new int[1000];
        for (int x : arr) {
            a[x]++;
        }
        for (int i = 999; i >= 1; i--) {
            if (i == a[i]) {
                return i;
            }
        }
        return -1;
    }
}
