class Solution {
    public int findLucky(int[] arr) {
        Map<Integer, Integer> f = new HashMap<>();
        for (int x : arr) {
            f.put(x, f.getOrDefault(x, 0) + 1);
        }
        
        int r = -1;
        for (Map.Entry<Integer, Integer> e : f.entrySet()) {
            if (e.getKey() == e.getValue()) {
                r = Math.max(r, e.getKey());
            }
        } 
        return r;
    }
}
