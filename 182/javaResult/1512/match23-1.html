class UndergroundSystem {
	
	Map<Integer,String> map1;
	Map<Integer,Integer> map2;
	Map<String,Integer> map3;
	Map<String,Integer> map4;
 
    public UndergroundSystem() {
    	map1=new HashMap<>();
    	map2=new HashMap<>();
    	map3=new HashMap<>();
    	map4=new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
    		map1.put(id, stationName);
    		map2.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
    	map3.put(map1.get(id)+"#"+stationName, map3.getOrDefault(map
1.get(id)+"#"+stationName, 0)+t-map2.get(id));
    	map4.put(map1.get(id)+"#"+stationName, map4.getOrDefault(map1.get(id)+"#"+stationName, 0)+1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
    	return (double)map3.get(startStation+"#"+endStation)/map4.get(startStation+"#"+endStation);
    }
}
