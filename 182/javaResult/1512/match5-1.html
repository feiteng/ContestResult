class UndergroundSystem {

    Map<String, Map<String, Long>> totalTime;
            Map<String, Map<String, Integer>> numTravellers;
            Map<Integer, String> userPrev;
            Map<Integer, Integer> userPrevTime;
            public UndergroundSystem() {
                totalTime = new HashMap<>();
                numTravellers = new HashMap<>();
                userPrev = new HashMap<>();
                userPrevTime = new HashMap<>();
            }

            public void checkIn(int id, String stationName, int t) {
                userPrev.put(id, stationName);
                userPrevTime.put(id, t);
                
            }

            public void checkOut(int id, String stationName, int t) {
                String prev = userPrev.get(id);
                int T = t - userPrevTime.get(id);
                userPrev.remove(id);
                userPrevTime.remove(id);
                if (!totalTime.containsKey(prev)) {
                    totalTime.put(prev, new HashMap<>());
                    numTravellers.put(prev, new HashMap<>());
                }
                if (!totalTime.get(prev).containsKey(stationName)) {
                    totalTime.get(prev).put(stationName, 0L);
                    numTravellers.get(prev).put(stationName, 0);
                }
                totalTime.get(prev).put(stationName, totalTime.get(prev).get(stationName) + T);
                
numTravellers.get(prev).put(stationName, numTravellers.get(prev).get(stationName) + 1);
            }

            public double getAverageTime(String startStation, String endStation) {
                return (0.0 + totalTime.get(startStation).get(endStation)) / numTravellers.get(startStation).get(endStation);

            }
}
