class UndergroundSystem {
    
    private class CheckIn {
        public int id;
        public String stationName;
        public int time;
        
        public CheckIn(int id, String stationName, int time) {
            this.id = id;
            this.stationName = stationName;
            this.time = time;
        }
    }
    

    Map<Integer, CheckIn> customers = new HashMap<>(); // <Id, CheckIn>
    Map<String, Long> times = new HashMap<>();
    Map<String, Integer> travels = new HashMap<>();

    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        customers.put(id, new CheckIn(id, stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        CheckIn checkIn = customers.get(id);
        
        String route = checkIn.stationName + "-" + stationName;
        long time = t - checkIn.time;
        
        times.put(route, times.getOrDefault(route, new Long(0)) + time);
        travels.put(route, travels.getOrDefault(route, 0) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String route = startStation + "-" + endStation;
        return times.get(route) * 1.0 / travels.get(route);
    }
}

/**
 * Your UndergroundSystem object will be instantiated and called as such:
 * UndergroundSystem obj = new UndergroundSystem();
 * obj.checkIn(id,stationName,t);
 * obj.checkOut(id,stationName,t);
 * double param_3 = obj.getAverageTime(startStation,endStation);
 */
