class UndergroundSystem {
    Map<String, Double> startEndSum;
    Map<String, Double> startEndCount;
    Map<Integer, Double> checkinTime;
    Map<Integer, String> checkinStation;
    public UndergroundSystem() {
        startEndSum = new HashMap<>();
        startEndCount = new HashMap<>();
        checkinTime = new HashMap<>();
        checkinStation = new HashMap<>();       
    }
    
    public void checkIn(int id, String stationName, int t) {
        checkinTime.put(id, (double)t);
        checkinStation.put(id, stationName);
    }
    
    public void checkOut(int id, String stationName, int t) {
        double time = (double)t - checkinTime.get(id);
        String prev = checkinStation.get(id);
        checkinStation.remove(id);
        checkinTime.remove(id);
        startEndSum.put(prev + " " +stationName, startEndSum.getOrDefault(prev + " " +stationName, 0.0) + time);
        startEndCount.put(prev + " " +stationName, startEndCount.getOrDefault(prev + " " +stationName, 0.0) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        return startEndSum.get(startStation + " " + endStation) / startEndCount.get(startStation + " " + endStation);
    }
}
