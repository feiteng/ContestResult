class UndergroundSystem {
    
    private final Map<String, Integer> travelTime = new HashMap<>();
    private final Map<String, Integer> travelCount = new HashMap<>();
    
    private final Map<Integer, CheckInState> pendingTravel = new HashMap<>();
    
    private static class CheckInState {
        private final String station;
        private final int time;
        private CheckInState(String station, int time) {
            this.station = station;
            this.time = time;
        }
    }

    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        pendingTravel.put(id, new CheckInState(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        CheckInState state = pendingTravel.remove(id);
        String key = getKey(state.station, stationName);
        int interval = t - state.time;
        travelTime.put(key, 
travelTime.getOrDefault(key, 0) + interval);
        travelCount.put(key, travelCount.getOrDefault(key, 0) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = getKey(startStation, endStation);
        if (travelCount.getOrDefault(key, 0) == 0) {
            return 0;
        }
        return travelTime.get(key) / (double)travelCount.get(key);
    }
    
    private String getKey(String start, String end) {
        return start + " " + end; 
    }
}
