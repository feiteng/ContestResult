class UndergroundSystem {
    Map<String, Map<String, List<Integer>>> map;
    Map<Integer, Record> log;
    public UndergroundSystem() {
        map = new HashMap<>();
        log = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        Record r = new Record(stationName, t);
        log.put(id ,r);
    }
    
    public void checkOut(int id, String stationName, int t) {
        Record r = log.get(id);
        String from = r.station;
        String to = stationName;
        Map<String, List<Integer>> details = map.getOrDefault(from, new HashMap<>());
        List<Integer> list = details.getOrDefault(to, new ArrayList<>());
        list.add(t - r.time);
        details.put(to, list);
        map.put(from, details);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        List<Integer> list = map.get(startStation).get(endStation);
        long sum = 0;
        for(int i: list) {
            sum += i;
        }
        return sum / (list.size() + 0.0);
    }
}
class Record {
    String station;
    int time;
    Record(String s, int t) {
        station = s;
        time = t;
    }
}
