class UndergroundSystem {
    
    Map<String, Map<String, int[]>> trav;
    Map<Integer, String> idMap;
    Map<Integer, Integer> timeMap;

    public UndergroundSystem() {
        trav = new HashMap<>();
        idMap = new HashMap<>();
        timeMap = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        idMap.put(id, stationName);
        timeMap.put(id, t);
    
}
    
    public void checkOut(int id, String stationName, int t) {
        int st = timeMap.get(id);
        String startStation = idMap.get(id);      
        trav.putIfAbsent(startStation, new HashMap<>());
        trav.get(startStation).putIfAbsent(stationName, new int[]{0, 0});
        int[] ans = trav.get(startStation).get(stationName);
        ans[0] += t - st;
        ans[1]++;
        
trav.get(startStation).put(stationName, ans);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        int[] ans = trav.get(startStation).get(endStation);
        return (double)(ans[0]) / ans[1];
    }
}
