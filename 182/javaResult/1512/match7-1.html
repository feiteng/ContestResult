import java.util.HashMap;

class UndergroundSystem {

    private HashMap<String, Integer> cnt;
    private HashMap<String, Integer> totalTime;

    private HashMap<Integer, String> checkedinStation;
    private HashMap<Integer, Integer> checkedinTime;


    
public UndergroundSystem() {
        cnt = new HashMap<>();
        totalTime = new HashMap<>();
        checkedinTime = new HashMap<>();
        checkedinStation = new HashMap<>();
    }

    public void checkIn(int id, String stationName, int t) {

        checkedinStation.put(id, stationName);
        checkedinTime.put(id, t);

    }

    public void checkOut(int id, String stationName, int t) {

        String prevStation = checkedinStation.get(id);
        int prevTime = checkedinTime.get(id);

        checkedinStation.remove(id);
        checkedinTime.remove(id);

        int time = t - prevTime;
        String key = prevStation + "#" + stationName;

        
cnt.put(key, cnt.getOrDefault(key, 0) + 1);
        totalTime.put(key, totalTime.getOrDefault(key, 0) + time);


    }

    public double getAverageTime(String startStation, String endStation) {

        String key = startStation + "#" + endStation;
        int time = totalTime.get(key);
        int tripCnt = cnt.get(key);
        return (double) time / tripCnt;

    }
}
