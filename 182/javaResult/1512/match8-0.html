class UndergroundSystem {
    HashMap<String, Integer> cnt;
    HashMap<String, Integer> sum;
    HashMap<Integer, String> ci;
    HashMap<Integer, Integer> st;
    public UndergroundSystem() {
        cnt = new HashMap<>();
        sum = new HashMap<>();
        ci = new HashMap<>();
        st = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        ci.put(id, stationName);
        st.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        String sx = ci.get(id);
        Integer ts = st.get(id);
        String key = sx + "@" + stationName;
        cnt.put(key, cnt.getOrDefault(key, 0) + 1);
        sum.put(key, sum.getOrDefault(key, 0) + t - ts);
        ci.remove(id);
        st.remove(id);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + "@" + endStation;
        if(!cnt.containsKey(key)) return 0.0;
        return (0.0 + sum.get(key)) / cnt.get(key);
    }
}
