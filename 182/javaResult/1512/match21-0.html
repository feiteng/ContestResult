class UndergroundSystem {
    
    Map<Integer, String> location = new HashMap<>();
    Map<Integer, Integer> time = new HashMap<>();
    Map<String, Integer> sum = new HashMap<>();
    Map<String, Integer> count = new HashMap<>();
    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        location.put(id, stationName);
        time.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        String start = location.get(id);
        String end = stationName;
        int d = t - time.get(id);
        
        String name = start + "_" + end;
        if(sum.get(name) == null) {
            sum.put(name, 0);
            count.put(name, 0);
        }
        sum.put(name, sum.get(name) + d);
        count.put(name, count.get(name) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String name = startStation + "_" + endStation;
        return Double.valueOf(sum.get(name)) / Double.valueOf(count.get(name));
    }
}
