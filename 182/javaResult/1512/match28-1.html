class UndergroundSystem {
    
    private Map<Integer, State> states = new HashMap<>();
    private Map<String, Map<String, int[]>> avgs = new HashMap<>();

    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        states.put(id, new State(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        State s = states.get(id);
        states.remove(id);
        
        avgs.putIfAbsent(s.s, new HashMap<>());
        Map<String, int[]> m = avgs.get(s.s);
        m.putIfAbsent(stationName, new int[2]);
        
int[] r = m.get(stationName);
        r[0] += t - s.t;
        r[1] ++;
    }
    
    public double getAverageTime(String startStation, String endStation) {
        int[] r = avgs.get(startStation).get(endStation);
        return (double) r[0] / r[1];
    }
    
    private class State {
        String s;
        int t;
        State(String s, int t) {
            this.s = s;
            this.t = t;
        }
    }
}
