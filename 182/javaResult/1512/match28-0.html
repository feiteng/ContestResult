class UndergroundSystem {
    Map<Integer, Travel> temp = new HashMap();
    Map<String, Map<String, History>> his = new HashMap();

    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        temp.put(id, new Travel(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        Travel tv = temp.remove(id);
        int interval = t - tv.t;
        
        if (!his.containsKey(tv.sn)) his.put(tv.sn, new HashMap());
        
        Map<String, History> cur = his.get(tv.sn);
        
        if (!cur.containsKey(stationName)) cur.put(stationName, new History());
        
        History ht = cur.get(stationName);
        ht.total += interval;
        ht.count += 1;
    }
    
    public double getAverageTime(String startStation, String endStation) {
        History ht = his.get(startStation).get(endStation);
        return (double)((double)ht.total/(double)ht.count);
    }
    
    class Travel {
        String sn;
        int t;

        Travel(String _sn, int _t) {
            sn = _sn;
            t = _t;
        }
    }
    
    class History {
        long total;
        long count;
        History() {
            total = 0;
            count = 0;
        }
    }
}
