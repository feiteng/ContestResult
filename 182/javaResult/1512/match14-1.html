class UndergroundSystem {

    private Map<Integer, Double> beginTimeById = new HashMap<>();
    private Map<Integer, String> beginStationById = new HashMap<>();
    
    private Map<String, Double> sum = new HashMap<>();
    private Map<String, Integer> count = new HashMap<>();
    
    public UndergroundSystem() {
        
    
}
    
    public void checkIn(int id, String stationName, int t) {
        beginTimeById.put(id, (double) t);
        beginStationById.put(id, stationName);
    }
    
    public void checkOut(int id, String stationName, int t) {
        double beginTime = beginTimeById.get(id);
        String beginStation = beginStationById.get(id);
        double time = t - beginTime;
        String to = beginStation + "-" + stationName;
        
sum.put(to, time + sum.getOrDefault(to, 0d));
        count.put(to, 1 + count.getOrDefault(to, 0));
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String to = startStation + "-" + endStation;
        return sum.get(to) / count.get(to);
    }
}
