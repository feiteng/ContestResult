class UndergroundSystem {
    
    private Map<Integer, Object[]> inMap;
    
    private Map<String, List<Integer>> timeMap;

    public UndergroundSystem() {
        inMap = new HashMap<>();
        timeMap = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        inMap.putIfAbsent(id, new Object[2]);
        inMap.get(id)[0] = stationName;
        
inMap.get(id)[1] = t;
    }
    
    public void checkOut(int id, String stationName, int t) {
        Object[] origin = inMap.get(id);
        String key = origin[0] + "-" + stationName;
        timeMap.putIfAbsent(key, new ArrayList<>());
        
timeMap.get(key).add(t - (Integer)origin[1]);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + "-" + endStation;
        List<Integer> c = 
timeMap.get(key);
        double times = 0;
        for (Integer t: c) {
            times += t;
        }
        return times / c.size();
    }
}
