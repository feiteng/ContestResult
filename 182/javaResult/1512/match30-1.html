class UndergroundSystem {
    Map<Integer, String> userInfo;
    Map<String, int[]> stationInfo;

    public UndergroundSystem() {
        userInfo = new HashMap<>();
        stationInfo = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        
userInfo.put(id, stationName + ":" + t);
    }
    
    public void checkOut(int id, String target, int t2) {
        String st = userInfo.get(id);
        String[] split = st.split(":");
        String source = split[0];
        int t1 = Integer.valueOf(split[1]);
        String s = source + "-" + target;
        stationInfo.putIfAbsent(s, new int[2]);
        int[] a = 
stationInfo.get(s);
        a[0] += t2 - t1;
        a[1]++;
    }
    
    public double getAverageTime(String source, String target) {
        String s = source + "-" + target;
        int[] a = stationInfo.get(s);
        return (double)(a[0]) / a[1];
    }
}
