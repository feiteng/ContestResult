class UndergroundSystem {
    Map<String, int[]> m = new HashMap<>();
    Map<Integer,String> id2time = new HashMap<>();
    public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        id2time.put(id, stationName + "," + t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        String s = id2time.get(id);
        String[] ss = s.split(",");
        String key = ss[0] + "," + stationName;
        int cost = t - Integer.parseInt(ss[1]);
        m.putIfAbsent(key, new int[2]);
        int[] cur = m.get(key);
        cur[0] += cost;
        cur[1]++;
    }
    
    public double getAverageTime(String startStation, String endStation) {
        int[] cur = m.get(startStation + "," + endStation);
        System.out.println(Arrays.toString(cur));
        return ((double) cur[0]) / cur[1];
    }
}
