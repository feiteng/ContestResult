class UndergroundSystem {

        Map<String, Integer> cost = new HashMap<>();
        Map<String, Integer> times = new HashMap<>();
        Map<Integer, Integer> pendingstart = new HashMap<>();
        Map<Integer, String> pendingstation = new HashMap<>();

        public UndergroundSystem() {

        
}

        public void checkIn(int id, String stationName, int t) {
            pendingstart.put(id, t);
            pendingstation.put(id, stationName);
        }

        public void checkOut(int id, String endStation, int t) {
            int start = pendingstart.get(id);
            String startstation = pendingstation.get(id);
            pendingstart.remove(id);
            pendingstation.remove(id);
            String code = startstation + "->" + endStation;
            int newcost = t - start;
            cost.put(code, cost.getOrDefault(code, 0) + newcost);
            times.put(code, times.getOrDefault(code, 0) + 1);
        }

        public double getAverageTime(String startStation, String endStation) {
            String code = startStation + "->" + endStation;
            int co = cost.getOrDefault(code, 0);
            int t = times.getOrDefault(code, 0);
            return co * 1.0 / t;
        }
    }
