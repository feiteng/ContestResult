class UndergroundSystem {
    Map<String, Integer> count;
    Map<String, Integer> time;
    Map<Integer, String> stations;
    Map<Integer, Integer> startTime;

    public UndergroundSystem() {
        count = new HashMap<>();
        time = new HashMap<>();
        stations = new HashMap<>();
        startTime = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        stations.put(id, stationName);
        startTime.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        int totalTime = t - startTime.get(id);
        String startStation = stations.get(id);
        String key = startStation + " " + stationName;
        
count.put(key, count.getOrDefault(key, 0) + 1);
        time.put(key, time.getOrDefault(key, 0) + totalTime);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + " " + endStation;
        
        int c = count.getOrDefault(key, 0);
        int t = time.getOrDefault(key, 0);
        
        if (t == 0) return 0;
        return (double) t / c;
    }
}
