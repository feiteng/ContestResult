class UndergroundSystem {
    HashMap<String, Integer> cnt = new HashMap<String,Integer>();
    HashMap<String, Long> time = new HashMap<String,Long>();
    HashMap<Integer, String> passenger = new HashMap<Integer, String>();
    HashMap<Integer, Integer> ptime = new HashMap<Integer, Integer>();

    public UndergroundSystem() {
        cnt = new HashMap<String,Integer>();
        time = new HashMap<String,Long>();
        passenger = new HashMap<Integer, String>();
        ptime = new HashMap<Integer, Integer>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        passenger.put(id, stationName);
        ptime.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        String start = passenger.get(id);
        int newtime = t - ptime.get(id);
        
        String key = start+":"+stationName;
        if (!time.containsKey(key)) {
            time.put(key, 0L);
            cnt.put(key,0);
        }
        time.put(key, time.get(key)+newtime);
        cnt.put(key, cnt.get(key)+1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key =startStation+":"+endStation;
        long L = time.get(key);
        double c = cnt.get(key);
        return L/(c + 0.0);
    }
}
