class UndergroundSystem {
    
    Map<String, Integer> count = new HashMap<>();
    Map<String, Integer> total = new HashMap<>();
    Map<Integer, Data> customer = new HashMap<>();
    
    
class Data {
        int id;
        String stationName;
        int t;
        
        public Data(int id, String stationName, int t) {
            this.id = id;
            this.stationName = stationName;
            this.t = t;
        }
    }

    
public UndergroundSystem() {
        
    }
    
    public void checkIn(int id, String stationName, int t) {
        customer.put(id, new Data(id, stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        Data data = customer.get(id);
        String key = data.stationName + "_" + stationName;
        int time = t - data.t;
        total.put(key, total.getOrDefault(key, 0) + time);
        
count.put(key, count.getOrDefault(key, 0) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + "_" + endStation;
        int t = total.get(key);
        int c = count.get(key);
        return (double) t / c;
    }
}
