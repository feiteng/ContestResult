class Solution {
    int n;
    long p = 1000000007;
    long dp[][][] = new long[501][51][2];
    
    long get(int i,int mc,String ev,String s,int eq){
        
         if(dp[i][mc][eq]>0){
            return dp[i][mc][eq];
        }
        
        if(mc==ev.length()){
            return dp[i][mc][eq] = 0;
        }
        if(i==n){
            return dp[i][mc][eq] = 1;
        }
        
       
        
        long ss = 0;
        if(eq==0){
            char y = s.charAt(i);

            for(char f='a';f<=y;++f){
                
                String tp = ev.substring(0,mc);
                String cur = tp+f;
                boolean ok = false;
                for(int j=0;j<mc+1;++j){
                    String suf = cur.substring(j);
                    int tlen = suf.length();
                    if(ev.substring(0,tlen).equals(suf)){
                      //  System.out.println("11");
                        ss += get(i+1,tlen,ev,s,f<y?1:0);
                        ss %= p;
                        ok = true;
                        break;
                    }   
                }
                if(!ok){
                  //  System.out.println("22");
                    ss += get(i+1,0,ev,s,f<y?1:0);
                    ss %= p;
                    
                }
                
            }
            
            
            
            
        }else{
            
            char y = s.charAt(i);
 
            for(char f='a';f<='z';++f){
                
                String tp = ev.substring(0,mc);
                String cur = tp+f;
                boolean ok = false;
                for(int j=0;j<mc+1;++j){
                    String suf = cur.substring(j);
                    int tlen = suf.length();
                    if(ev.substring(0,tlen).equals(suf)){
                        // System.out.println("33");
                        ss += get(i+1,tlen,ev,s,1);
                        ss %= p;
                        ok = true;
                        break;
                    }   
                }
                if(!ok){
                   //  System.out.println("44");
                    ss += get(i+1,0,ev,s,1);
                    ss %= p;
                    
                }
                
            }
            
            
            
            
        }
        
        return dp[i][mc][eq] = ss;
        
    }
    
    
    
    
    
    
    
    public int findGoodStrings(int nn, String s1, String s2, String ev) {
        n = nn;
        
        dp = new long[501][51][2];
        long t2 = get(0,0,ev,s2,0);
        dp = new long[501][51][2];
        long t1 = get(0,0,ev,s1,0);
       // long t1 = 0;
        t2-=t1;
        t2 += 1;
        if(s1.contains(ev)){
            t2--;
        }
        
        t2 = ((t2%p)+p)%p;
        
        return (int) t2; 
        
        
    }
}
