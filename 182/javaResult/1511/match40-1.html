class Solution {
    public int numTeams(int[] rating) {
        int N = rating.length;
        int res = 0;
        for (int i = 0; i <= N - 3; i++) {
            for (int j = i + 1; j <= N - 2; j++) {
                for (int k = j + 1; k < N; k++) {
                    if ((rating[i] < rating[j] && rating[j] < rating[k]) || (rating[i] > rating[j] && rating[j] > rating[k])) {
                        res++;
                    }
                
}
            }
        }
        return res;
    }
}
