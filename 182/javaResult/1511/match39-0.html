class Solution {
    public int numTeams(int[] rating) {
        int N = rating.length;
        int teams = 0;
        
        for (int i = 0; i < N - 2; i++) {
            for (int j = i + 1; j < N - 1; j++) {
                for (int k = j + 1; k < N; k++) {
                    if (rating[i] < rating[j] && rating[j] < rating[k]) {
                        teams++;
                    }
                    if (rating[i] > rating[j] && rating[j] > rating[k]) {
                        teams++;
                    }
                }
            }
        }
        
        return teams;
    }
}
