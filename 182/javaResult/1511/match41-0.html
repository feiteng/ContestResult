class Solution {
    public int numTeams(int[] rating) {
        int res = 0, l = rating.length;
        for (int i = 0; i < l; i++) {
            for (int j = i + 1; j < l; j++) {
                for (int k = j + 1; k < l; k++) {
                    if (rating[i] > rating[j] && rating[j] > rating[k]) {
                        res++;
                    }
                    if (rating[i] < rating[j] && rating[j] < rating[k]) {
                        res++;
                    }
                }
            }
        }
        return res;
    }
}
