class UndergroundSystem {
public:
    UndergroundSystem() {
    }
    
    void checkIn(int id, string stationName, int t) {
        currentTravels[id] = make_pair(stationName, t);
    }
    
    void checkOut(int id, string stationName, int t) {
        pair<string, int> p = currentTravels[id];
        
        numTravels[p.first][stationName]++;
        totalTimeTravels[p.first][stationName] += t - p.second;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return (double) totalTimeTravels[startStation][endStation] / (double) numTravels[startStation][endStation];
    }
private:
    map<int, pair<string, int> > currentTravels;
    map<string, map<string, int> > numTravels;
    map<string, map<string, int> > totalTimeTravels;
};
