#define ll long long

#define ld long double
#define fi first
#define se second
#define pb push_back

class UndergroundSystem {
public:
    map<pair<string,string>,ll > mapa;
    map<ll,pair<string,ll> > entry;
    map<pair<string,string>,ll > cnt;
    UndergroundSystem() {
        mapa.clear();
        entry.clear();
        cnt.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        entry[id]={stationName,t};
    }
    
    void checkOut(int id, string stationName, int t) {
        string tmp=entry[id].fi;
        ll tim=t-entry[id].se;
        mapa[{tmp,stationName}]+=tim;
        cnt[{tmp,stationName}]++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return ((double)mapa[{startStation,endStation}]*1.0)/(1.0*cnt[{startStation,endStation}]);
    }
};
