class UndergroundSystem {

private:
    map<string, map<int, int>> in;
    map<string, map<int, int>> out;
    
public:
    UndergroundSystem() {}

    void checkIn(int id, string stationName, int t) {
        in[stationName][id] = t;
    }

    void checkOut(int id, string stationName, int t) {
        out[stationName][id] = t;
    }

    double getAverageTime(string startStation, string endStation) {
        
        double totalTime = 0.0;
        int cnt = 0;
        for(const pair<int, int>& p: in[startStation]){
            int id = p.first, startTime = p.second;
            if(out[endStation].count(id)){
                totalTime += out[endStation][id] - startTime;
                cnt ++;
            }
        }
        return totalTime / cnt;
    }
};
