class UndergroundSystem {
public:
    map<int,pair<string,int> >in;
    map<pair<string,string>,int> time;
    map<pair<string,string>,int> times;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        in[id] = {stationName,t};
    }
    
    void checkOut(int id, string stationName, int t) {
        if(in.find(id)!=in.end()){
            string start = in[id].first;
            int st = in[id].second;
            time[{start,stationName}]+=t-st;
            times[{start,stationName}]+=1;
            in.erase(id);
        }

    }
    
    double getAverageTime(string startStation, string endStation) {
        return (double)time[{startStation,endStation}]/(double)times[{startStation,endStation}];
    }
};
