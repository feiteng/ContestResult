class UndergroundSystem {
public:
    map<pair<string,string>, double> cnt,tot;
    map<int,int> tIn; //id->checkIn
    map<int,string> stationIn;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        tIn[id]=t;
        stationIn[id]=stationName;
    }
    
    void checkOut(int id, string s2, int t) {
        string s1 = stationIn[id];
        int totalTime = t - tIn[id];
        cnt[{s1,s2}] += 1;
        tot[{s1,s2}] += totalTime;
        
        tIn.erase(id);
        stationIn.erase(id);
    }
    
    double getAverageTime(string s, string e) {

        return 1.0*tot[{s,e}] / cnt[{s,e}];
    }
};

/**
 * Your UndergroundSystem object will be instantiated and called as such:
 * UndergroundSystem* obj = new UndergroundSystem();
 * obj->checkIn(id,stationName,t);
 * obj->checkOut(id,stationName,t);
