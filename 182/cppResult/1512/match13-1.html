class UndergroundSystem {
public:
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        checkInTime[id] = make_pair(stationName, t);
    }
    
    void checkOut(int id, string stationName, int t) {
        int time = t - checkInTime[id].second;
        string station1 = checkInTime[id].first;
        if(graph.find(station1) == graph.end() || graph[station1].find(stationName) == graph[station1].end()){
            graph[station1][stationName] = make_pair(time, 1);
        }else{
            graph[station1][stationName].first += time;
            graph[station1][stationName].second++;
        }

    }
    
    double getAverageTime(string startStation, string endStation) {
        return (double)graph[startStation][endStation].first / (double)graph[startStation][endStation].second;
    }
private:
    unordered_map<string, unordered_map<string, pair<int, int>>> graph; // station1 - station2 - {total time, # of users}
    unordered_map<int, pair<string, int>> checkInTime; // id - {station, checkInTime}
};
