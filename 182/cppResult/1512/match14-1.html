class UndergroundSystem {
public:
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        time[ id ] = t;
        station[ id ] = stationName;
    }
    
    void checkOut(int id, string stationName, int t) {
        int prev_time = time[ id ];
        string prev_station = station[ id ];

        total[ { prev_station, stationName } ] += t - prev_time;
        times[ { prev_station, stationName } ]++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return ( double )total[ { startStation, endStation } ] / ( double )times[ {startStation, endStation } ];
    }
private:
    map< int, int > time;
    map< int, string > station;
    map< pair< string, string >, int > total;
    map< pair< string, string >, int > times;
    
};
