class UndergroundSystem {
public:
    unordered_map<int, pair<string,int>> idStartStationMap;
    unordered_map<string, unordered_map<string, pair<long, int> >> mp; 
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        idStartStationMap[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        string checkInStation = idStartStationMap[id].first;
        int checkInTime = idStartStationMap[id].second;
        idStartStationMap.erase(id);
        if(mp[stationName].count(checkInStation) == 0){
            mp[stationName][checkInStation] = {0, 0};
        }
        mp[stationName][checkInStation].first += - checkInTime + t;
        mp[stationName][checkInStation].second++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        if(mp.count(endStation) == 0 || mp[endStation].count(startStation) == 0) return 0.0;
        return mp[endStation][startStation].first*1.0/mp[endStation][startStation].second;
    }
};
