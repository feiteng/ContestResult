class UndergroundSystem {
public:
	map <int,int> startTime;
	map <int,string> startCity;
	map <pair <string,string>,int> num,den;
    UndergroundSystem() {
        num.clear();
        den.clear();
        startTime.clear();
        startCity.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        startTime[id]=t;
        startCity[id]=stationName;
    }
    
    void checkOut(int id, string stationName, int t) {
        num[{startCity[id],stationName}]+=t-startTime[id];
        den[{startCity[id],stationName}]++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return 1.*num[{startStation,endStation}]/den[{startStation,endStation}];
    }
};
