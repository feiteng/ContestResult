class UndergroundSystem {
public:
    // id -> start, checkin
    unordered_map<int, pair<string, int> >cache;
    // st -> num, t
    unordered_map<string, pair<int, long long> >mp;

    UndergroundSystem() {
        cache.clear();
        mp.clear();
    }

    void checkIn(int id, string stationName, int t) {
        cache[id] = make_pair(stationName, t);
    }

    void checkOut(int id, string stationName, int t) {
        auto p = cache[id];
        auto key = p.first + ";" + stationName;
        mp[key].first++;
        mp[key].second += t - p.second;
    }

    double getAverageTime(string startStation, string endStation) {
        auto key = startStation + ";" + endStation;
        if (!mp.count(key)) {
            return 0.0;
        }
        return mp[key].second / (mp[key].first + 0.0);
    }
};
