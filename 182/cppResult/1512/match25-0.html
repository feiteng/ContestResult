class UndergroundSystem {
public:
    map<int,pair<string,int>> id_loc;
    map<pair<string,string>, double> tot;
    map<pair<string,string>, double> cnt;
        
    UndergroundSystem() {
    }
    
    void checkIn(int id, string stationName, int t) {
        id_loc[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        pair<string,string> journey = {id_loc[id].first, stationName};
        tot[journey] += t - id_loc[id].second;
        cnt[journey] += 1;
    }
    
    double getAverageTime(string startStation, string endStation) {
        pair<string,string> j = {startStation, endStation};
        return tot[j] / cnt[j];
    }
};

 * UndergroundSystem* obj = new UndergroundSystem();
