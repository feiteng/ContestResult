map<pair<string, string>, pair<long long, long long>> m;
map<int, pair<string, int>> la;

class UndergroundSystem {
public:
    UndergroundSystem() {
        m.clear();
        la.clear();
    }
    
    void checkIn(int id, string s, int t) {
        la[id] = {s, t};
    }
    
    void checkOut(int id, string s, int t) {
        m[{la[id].first, s}].first += t -la[id].second;
        m[{la[id].first, s}].second++;
    }
    
    double getAverageTime(string s, string e) {
        return (double)m[{s, e}].first / m[{s,e}].second;
    }
};

 * UndergroundSystem* obj = new UndergroundSystem();
