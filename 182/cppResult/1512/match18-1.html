class UndergroundSystem {
private:
    unordered_map<string, unordered_map<int,int>> inS;
    unordered_map<string, unordered_map<int,int>> outS;
public:

    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        inS[stationName][id] = t;
    }
    
    void checkOut(int id, string stationName, int t) {
        outS[stationName][id] = t;    
    }
    
    double getAverageTime(string startStation, string endStation) {
        unordered_map<int,int> sStation = inS[startStation];
        unordered_map<int,int> eStation = outS[endStation];
        double ret;

        int count = 0;
        for(auto i : sStation){
            int id = i.first;
            if(eStation[id]){
                ret += eStation[id] - sStation[id];
                count++;
            }
        }
        ret /= count;
        return ret;
    }
};
