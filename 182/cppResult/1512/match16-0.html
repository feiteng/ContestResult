class UndergroundSystem {
    unordered_map<string, pair<int, int>> s2sRecord;
    unordered_map<int, pair<string, int>> cLastRecord;
public:
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        cLastRecord[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        string s2s = cLastRecord[id].first + "#"+ stationName;
        int duration = t - cLastRecord[id].second;
        s2sRecord[s2s].first ++;
        s2sRecord[s2s].second += duration;
    }
    
    double getAverageTime(string startStation, string endStation) {
        string s2s = startStation + "#"+ endStation;
        return (double)s2sRecord[s2s].second / s2sRecord[s2s].first;
    }
};
