class UndergroundSystem {
public:
    unordered_map<string,vector<pair<int,int>>> mi,mo;
    UndergroundSystem() {
        mi.clear();
        mo.clear();
    }
    
    void checkIn(int id, string s, int t) {
        mi[s].push_back({id,t});
    }
    
    void checkOut(int id, string s, int t) {
        mo[s].push_back({id,t});
    }
    
    double getAverageTime(string s, string e) {
        double x=0;
        int n=0;
        for(auto c:mi[s])
        {
            for(auto q:mo[e])
            {
                if(c.first==q.first)
                {
                    n++;
                    x+=q.second-c.second;
                    break;
                }
            }
        }
        return x/n;
    }
};
