class UndergroundSystem {
public:
    map<pair<string, string>, pair<int, int > > m;
     map<int, pair<int,string> > rem;
    UndergroundSystem() {

    }
    
    void checkIn(int id, string stationName, int t) {
        rem[id] = {t, stationName};
    }
    
    void checkOut(int id, string stationName, int t) {
        pair<int, string> curr = rem[id];
        int diff = t- curr.first;
        m[{curr.second, stationName}].first += diff;
        m[{curr.second, stationName}].second ++;
    
    }
    
    double getAverageTime(string startStation, string endStation) {
        pair<int, int> x = m[{startStation, endStation}];
        return 1.0*x.first / x.second;
    }
};
