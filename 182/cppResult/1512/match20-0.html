class UndergroundSystem {
public:
    map<pair<string, string>, pair<double, int> > MAP;
    map<int, pair<string, double> > STR;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        STR[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        MAP[{STR[id].first, stationName}].first += t - STR[id].second;
        MAP[{STR[id].first, stationName}].second++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return MAP[{startStation, endStation}].first / MAP[{startStation, endStation}].second;
    }
};
