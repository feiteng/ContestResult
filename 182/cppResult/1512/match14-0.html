class UndergroundSystem {
public:
    #define ll long long
    UndergroundSystem() {
        ids_to_in_station_.clear();
        ids_to_in_time_.clear();
        sum_times_.clear();
        num_samples_.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        ids_to_in_station_[id] = stationName;
        ids_to_in_time_[id] = t;
    }
    
    void checkOut(int id, string stationName, int t) {
        
        string in_station = ids_to_in_station_[id];
        int in_time = ids_to_in_time_[id];
        
        ids_to_in_station_.erase(id);
        ids_to_in_time_.erase(id);
        
        sum_times_[make_pair(in_station, stationName)] += t-in_time;
        num_samples_[{in_station, stationName}]++;
    }
    
    double getAverageTime(string startStation, string endStation) {
        
        return (double)sum_times_[{startStation, endStation}] / (double)num_samples_[{startStation, endStation}];
        
        
    }
    private:
      map<int,string> ids_to_in_station_;
      map<int,int> ids_to_in_time_;
      map<pair<string,string>,ll> sum_times_;
      map<pair<string,string>,ll> num_samples_;
    
      
    
};

 * Your UndergroundSystem object will be instantiated and called as such:
 * UndergroundSystem* obj = new UndergroundSystem();
 * obj->checkIn(id,stationName,t);
 * obj->checkOut(id,stationName,t);
 * double param_3 = obj->getAverageTime(startStation,endStation);
 */
