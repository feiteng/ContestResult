class UndergroundSystem {
public:
  map<int, pair<string, int>> C;
  map<pair<string, string>, pair<int, int>> M;
  
  UndergroundSystem() {
    C.clear();
    M.clear();
  }

  void checkIn(int id, string stationName, int t) {
    C[id] = {stationName, t};
  }

  void checkOut(int id, string stationName, int t) {
    M[{C[id].first, stationName}].first++;
    M[{C[id].first, stationName}].second+=t-C[id].second;
    C.erase(id);
  }

  double getAverageTime(string startStation, string endStation) {
    pair<string, string> p = {startStation, endStation};
    return 1.0 * M[p].second / M[p].first;
  }
};
