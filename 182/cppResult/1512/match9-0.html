class UndergroundSystem {
public:
    map<pair<string, string>, pair<int, int>>all;
    map<int, pair<string,int>>curr;
    UndergroundSystem() {
        all.clear();
    }

    void checkIn(int id, string stationName, int t) {
        curr[id] = { stationName,t };
    }

    void checkOut(int id, string stationName, int t) {
        string x = curr[id].first;
        int y = curr[id].second;
        curr.erase(id);
        all[{x, stationName}].first += (t - y);
        all[{x, stationName}].second++;
    }

    double getAverageTime(string startStation, string endStation) {
        return 1.0 * all[{startStation, endStation}].first / all[{startStation, endStation}].second;
    }
};
