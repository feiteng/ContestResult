class UndergroundSystem {
public:
    unordered_map<int,pair<string,int>> inside;
    unordered_map<string,unordered_map<string,pair<int,int>>> times;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        inside[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        auto startSt = inside[id].first;
        int startTime = inside[id].second;
        inside.erase(id);
        times[startSt][stationName].first += (t-startTime);
        times[startSt][stationName].second++;
    }
    
    double getAverageTime(string s1, string s2) {
        return times[s1][s2].first / (double) times[s1][s2].second;
    }
};
