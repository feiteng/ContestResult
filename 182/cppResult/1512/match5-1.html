class UndergroundSystem {
    map<pair<string, string>, int> cnt, sum;
    unordered_map<int, pair<string, int>> dict;

public:
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        dict[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {

        auto [last, lt] = dict[id];
        cnt[{last, stationName}]++;
        sum[{last, stationName}] += t - lt;
    }
    
    double getAverageTime(string startStation, string endStation) {
        return (double) sum[{startStation, endStation}] / cnt[{startStation, endStation}];
    }
};

 * UndergroundSystem* obj = new UndergroundSystem();
