class UndergroundSystem {
public:
    unordered_map<int, pair<string, int>> st_time;
    
    unordered_map<string, pair<int, int>> m;
    
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string st, int t) {
        st_time[id]= {st, t};
    }
    
    void checkOut(int id, string st, int t) {
        auto it = st_time.find(id);
        int time = t - it->second.second;
        
        string path = it->second.first + st;
        if(m.count(path) == 0) {

            m[path] = {1, time};
        }else {
            auto& t = m[path];
            m[path] = {t.first +1, t.second + time};
        }
    }
    
    double getAverageTime(string s, string e) {
        string path = s + e;
        auto& p = m[path];
        return p.second * 1.0 / p.first;
    }
};
