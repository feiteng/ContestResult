class UndergroundSystem {
public:
    map<pair<string,string>,pair<int,int>> mp; 
    // key start/end 
    // value all cnt
    
    map<int,pair<string,int>> people;
    // key id
    // value checkIn in time
    
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        people[id] = {stationName,t};
    }
    
    void checkOut(int id, string stationName, int t) {
        auto checkIn = people[id];

        mp[{checkIn.first,stationName}].first += t-checkIn.second;
        mp[{checkIn.first,stationName}].second += 1;
        people.erase(people.find(id));
    }
    
    double getAverageTime(string startStation, string endStation) {
        auto res = mp[{startStation,endStation}];
        return 1.0 * res.first / res.second;
    }
};
