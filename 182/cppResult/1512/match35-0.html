class UndergroundSystem {
public:
    
    map<pair<string, string>, long long> sum, cnt;
    unordered_map<int, pair<int, string>> mp;
    
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        mp[id] = {t, stationName};
    }
    
    void checkOut(int id, string des, int t2) {
        int t1 = mp[id].first;
        int duration = t2 - t1;
        string src = mp[id].second;
        sum[{src, des}] += duration;
        cnt[{src, des}]++;
    }
    
    double getAverageTime(string a, string b) {
        return sum[{a, b}] * 1.0 / cnt[{a, b}];
    }
};
