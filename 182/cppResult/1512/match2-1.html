class UndergroundSystem {
public:
    
    using ull = unsigned long long;

    map<pair<string, string>, pair<int, ull>> res;
    map<int, pair<string,int>> dp;
    UndergroundSystem() {
        res.clear();
        dp.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        dp[id] = {stationName, t};
    }
    
    void checkOut(int id, string stationName, int t) {
        auto start = dp[id].first;
        auto time = dp[id].second;

        res[{start, stationName}].first ++;
        res[{start, stationName}].second += t-time;
    }
    
    double getAverageTime(string startStation, string endStation) {
        auto tmp = res[{startStation, endStation}];
        return tmp.second*1.0/tmp.first;
    }
};
