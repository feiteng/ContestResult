class UndergroundSystem {
public:
    map<pair<string,string>,pair<int,int>> check_out; // per pair values sum,number
    map<int,pair<string,int>> check_in;
    UndergroundSystem() {
        check_in.clear();
        check_out.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        check_in[id] = {stationName,t};
    }
    
    void checkOut(int id, string stationName, int t) {
        check_out[{check_in[id].first,stationName}].first += t- check_in[id].second;
        check_out[{check_in[id].first,stationName}].second += 1;
        
    }
    
    double getAverageTime(string startStation, string endStation) {
        return (double)check_out[{startStation,endStation}].first / (double)check_out[{startStation,endStation}].second;
    }
};

 * UndergroundSystem* obj = new UndergroundSystem();
 * obj->checkIn(id,stationName,t);
