class Solution {
public:
    int mod = 1000000007;
    int add( int x, int y ) {
        return ( 1ll * x + 1ll * y ) % mod;
    }
    int dp[505][55][2][2];
    int findGoodStrings(int n, string s1, string s2, string evil) {
        s1 = "+" + s1, s2 = "+" + s2;
        dp[0][0][1][1] = 1;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < evil.length(); j++) {
                for(int sA = 0; sA < 2; sA++) {
                    for(int sB = 0; sB < 2; sB++) {
                        if( !dp[i][j][sA][sB] ) continue;
                       // cout << i << " " << j << " " <<sA << " " << sB << "\n";
                        for(int k = 0; k < 26; k++) {
                            bool typeA = sA, typeB = sB;
                            if( sA ) {
                                if( k > s1[i + 1] - 'a' )
                                    typeA = 0;
                                else if( k < s1[i + 1] - 'a' )
                                    continue;
                                else
                                    typeA = 1;
                            }
                            if( sB ) {
                                if(  k < s2[i + 1] - 'a')
                                    typeB = 0;
                                else if( k > s2[i + 1] - 'a' )
                                    continue;
                                else
                                    typeB = 1;
                            }
                            int hihi = (k == evil[j] - 'a' );
                          //  cout << hihi << " " << i << "\n";
                            if(hihi == 1 && j + hihi < evil.length())
                                dp[i + 1][j + hihi][typeA][typeB] = add( dp[i + 1][j + hihi][typeA][typeB], dp[i][j][sA][sB] );
                            else if( hihi == 0 ) {
                                string temp = "";
                                for(int l = 0; l < j; l++)
                                    temp += evil[l];
                                temp += (k + 'a');
                                int len = 0;
                                for(int l = 0; l < temp.size(); l++) {
                                    int cnt = 0; bool ok = 0;
                                    for(int h = l; h < temp.size(); h++) {
                                        if( temp[h] == evil[cnt] )
                                            ++cnt;
                                        else {
                                            ok = 1;
                                            break;
                                        }
                                    }
                                    if( !ok ) {
                                        len = cnt;
                                        break;
                                    }
                                }
                              //  cout << len << " " << typeA << " " << typeB << "\n";
                                dp[i + 1][len][typeA][typeB] = add( dp[i + 1][len][typeA][typeB], dp[i][j][sA][sB] );
                            }
                        }
                    }
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < evil.size(); i++) {
            for(int j = 0; j < 2; j++) {
                for(int k = 0; k < 2; k++)
                    ans = add( dp[n][i][j][k], ans );
            }
        }
        return ans;
    }
}g;
