class Solution {
public:
    int dp[510][60][2][2];
    int t[60][26];
    const int MOD = 1000000007;

    void update(int &x, int y) {
        x += y;
        if (x >= MOD) {
            x -= MOD;
        }
    }

    int findGoodStrings(int n, string s1, string s2, string evil) {
        int m = evil.size();
        for (int i = 0; i < m; i++) {
            string res = "";
            for (int j = 0; j < i; j++) {
                res += evil[j];
            }
            for (int j = 0; j < 26; j++) {
                t[i][j] = 0;
                string res2 = res;
                res2 += ('a' + j);
                for (int k = m - 1; k >= 0; k--) {
                    bool isMatch = true;
                    int pos = (int)res2.size() - 1;
                    for (int l = k; l >= 0; l--) {
                        if (pos < 0 || evil[l] != res2[pos]) {
                            isMatch = false;
                            break;
                        }
                        pos--;
                    }
                    if (isMatch) {
                        t[i][j] = k + 1;
                        break;
                    }
                }
            }
        }

        memset(dp, 0, sizeof(dp));

        dp[0][0][1][1] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int down = 0; down <= 1; down++) {
                    for (int up = 0; up <= 1; up++) {
                        int L = 0, R = 25;
                        if (down) {
                            L = (i < s1.size() ? s1[i] - 'a' : -1);
                        }
                        if (up) {
                            R = (i < s2.size() ? s2[i] - 'a' : -1);
                        }
                        for (int k = L; k <= R; k++) {
                            if (k < 0) continue;
                            if (t[j][k] == m) continue;
                            update(dp[i + 1][t[j][k]][down && (k == L)][up && (k == R)], dp[i][j][down][up]);

                        }
                    }
                }
            }
        }

        int sum = 0;
        for (int i = 0; i < m; i++) {
            for (int down = 0; down < 2; down++) {
                for (int up = 0; up < 2; up++) {
                    update(sum, dp[n][i][down][up]);
                }
            }
        }

        return sum;
    }
};
