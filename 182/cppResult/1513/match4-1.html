class Solution {
public:
    const int MOD = 1000000007;
    int f[510][60][2][2];
    int t[60][26];
    char tmp[60];
    void upd(int &x, int y) {
        x = (x + y) % MOD;
    }
    int findGoodStrings(int n, string s1, string s2, string evil) {
        int m = evil.length();
        for (int i=0;i<m;i++) {
            for (int j=0;j<26;j++) {
                tmp[i] = ('a' + j); 
                int match = 0;
                for (int k=i;k>=0;k--) {
                    bool ok = true;
                    for (int l=0;l<=k;l++) {
                        if (tmp[i - l] != evil[k - l]) {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) {
                        match = k + 1;
                        break;
                    }
                }
                t[i][j] = match;
            }
            tmp[i] = evil[i];
        }

        f[0][0][0][0] = 1;
        for (int i=0;i<n;i++) {
            for (int j=0;j<m;j++) {
                for (int k=0;k<=1;k++) {
                    for (int l=0;l<=1;l++) {
                        int down = (k == 0 ? s1[i] - 'a' : 0);
                        int up = (l == 0 ? s2[i] - 'a' : 25);
                        for (int ii=down;ii<=up;ii++) {
                            int nj, nk, nl;
                            nj = t[j][ii];
                            nk = (k == 0 && ii == down) ? 0 : 1;
                            nl = (l == 0 && ii == up) ? 0 : 1;
                            upd(f[i + 1][nj][nk][nl], f[i][j][k][l]);

                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int j=0;j<m;j++) {
            for (int k=0;k<=1;k++) {
                for (int l=0;l<=1;l++) {
                    upd(ans, f[n][j][k][l]);
                }
            }
        }
        return ans;
    }
};
