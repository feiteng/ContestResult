class Solution {
public:
    int dp[555][2][2][55];
    
    int MOD = 1000 * 1000 * 1000 + 7;
    
    int sum(int a, int b) {
        return (a + b) % MOD;
    }
    
    void add(int& a, int b) {
        a = sum(a, b);
    }
    
    int calc(string s1, string s2) {
        int n = s1.size();
        int m = s2.size();
        int ans = 0;
        for(int len = min(n, m) - 1; len >= 1; len--) {
            int p1 = 0;
            int p2 = s1.size() - len;
            bool ok = true;
            for(int k = 0; k < len; k++) {
                if(s2[p1] != s1[p2]) {
                    ok = false;
                    break;
                }
                
                p1++;
                p2++;
            }
            
            if(ok) return len;
        }
        
        return ans;
    }
    
    char a[555], b[555], c[555];
    int L[55][27];
    
    int findGoodStrings(int n, string s1, string s2, string evil) {

        int m = evil.size();
        for(int i = 0; i < 555; i++) for(int j = 0; j < 2; j++) for(int k = 0; k < 2; k++) for(int l = 0; l < 55; l++) dp[i][j][k][l] = 0;
        for(int i = 0; i < n; i++) {
            a[i + 1] = s1[i];
            b[i + 1] = s2[i];
        }
        
        for(int i = 0; i < m; i++) c[i + 1] = evil[i];
        
        
       
       for(char ad = 'a'; ad <= 'z'; ad++) {
            int jj, kk;
            if(ad == a[1]) jj = 1;
            else jj = 0;
            if(ad == b[1]) kk = 1;
            else kk = 0;
           if(ad < a[1]) continue;
           if(ad > b[1]) continue;
            int ll;
            if(ad != c[1]) {
                dp[1][jj][kk][0]++;
            } else {
                if(1 < m) dp[1][jj][kk][1]++;
            }
        }
        
        
        //dp[0][1][1][0] = 1;
        for(int i = 0; i < 55; i++) for(int j = 0; j < 27; j++) L[i][j] = 0;
        
        for(int i = 1; i < m; i++) for(int j = 0; j < 26; j++) {
            string ss = "";
            for(int p = 1; p <= i; p++) ss += c[p];
            ss += char(j + 'a');
            int len = calc(ss, evil);
            L[i][j] = len;
        }
        

        for(int i = 0; i < n; i++) for(int j = 0; j < 2; j++) for(int k = 0; k < 2; k++) for(int l = 0; l < m; l++) {
            int cur = dp[i][j][k][l];
            if(cur == 0) continue;
           // if(i == 0) assert(cur == 0);
            for(char ad = 'a'; ad <= 'z'; ad++) {
                int nj, nk;
                if(j == 1 && ad < a[i + 1]) continue;
                if(k == 1 && ad > b[i + 1]) continue;
                if(j == 0) nj = 0;
                else {
                    if(ad == a[i + 1]) nj = 1;
                    else nj = 0;
                }
                
                if(k == 0) nk = 0;
                else {
                    if(ad == b[i + 1]) nk = 1;
                    else nk = 0;
                }
                
                if(ad != c[l + 1]) {
                    //string ss = "";
                   // for(int p = 1; p <= l; p++) ss += c[p];
                  //  ss += ad;
                    int nl = L[l][ad - 'a'];
                    add(dp[i + 1][nj][nk][nl], cur);
                }
                else {
                    if(l + 1 < m) add(dp[i + 1][nj][nk][l + 1], cur);

                }
            }
        }
        
        int ans = 0;
        for(int j = 0; j < 2; j++) for(int k = 0; k < 2; k++) for(int l = 0; l < m; l++) add(ans, dp[n][j][k][l]);
        return ans;
    }
};
