class Solution {
    const int M = 1000000007;
    int calc(int n, string s, string evil) {
        vector<vector<vector<int>>> dp(n + 1, vector<vector<int>>(2, vector<int>(evil.size() + 1, 0)));
        dp[0][0][0] = 1;
        for(int i = 0; i < n; ++i) for(int j = 0; j < 2; ++j) for(int k = 0; k < evil.size(); ++k) {
            if(dp[i][j][k] == 0) continue;
            char lim = j ? 'z' : s[i];
            for(char c = 'a'; c <= lim; ++c) {
                if(evil[k] == c) {
                    (dp[i + 1][j || c < lim][k + 1] += dp[i][j][k]) %= M;
                }
                else {
                    string t = evil.substr(0, k) + c;
                    bool end = false;
                    for(int l = 1; l < t.size(); ++l) {
                        bool ok = true;
                        for(int m = 0; l + m < t.size(); ++m) {
                            if(t[l + m] == evil[m]) continue;
                            ok = false;
                            break;
                        }
                        if(ok) {
                            (dp[i + 1][j || c < lim][(int)t.size() - l] += dp[i][j][k]) %= M;
                            end = true;
                            break;
                        }
                    }
                    if(!end) (dp[i + 1][j || c < lim][0] += dp[i][j][k]) %= M;
                }
            }
        }
        int ans = 0;
        for(int j = 0; j < 2; ++j) for(int k = 0; k < evil.size(); ++k) (ans += dp[n][j][k]) %= M;
        return ans;
    }
public:
    int findGoodStrings(int n, string s1, string s2, string evil) {
        int ans = calc(n, s2, evil);
        if(s1 != string(n, 'a')) {
            int p = n - 1;
            while(s1[p] == 'a') {
                s1[p] = 'z';
                --p;
            }
            --s1[p];
            (ans -= calc(n, s1, evil)) %= M;
        }
        if(ans < 0) ans += M;
        return ans;
    }
};//500543753
