class Solution {
public:
    int nxt[1005];
    void get(string s)
    {
        int i, j, len = s.size();
        nxt[0]=-1;
        for(i=0,j=-1;i<len;)
        if(j==-1||s[i]==s[j])
        nxt[++i]=++j;
        else
        j=nxt[j];
    }

    int get(string s, int cur, char c) {
        while (cur != -1 && s[cur] != c) {
            cur = nxt[cur];
        }
        return cur + 1;
    }
    
    int f[1005][1005][2][2];
    int MO = 1000000007;
    
    int findGoodStrings(int n, string s1, string s2, string e) {
        int m = e.size();
        get(e);
        for (int i = 0; i < n; i++) {
            if (s1[i] < s2[i]) {
                break;
            } else if (s1[i] == s2[i]) {
                continue;
            } else {
                return 0;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < 2; k++) {
                    for (int l = 0; l < 2; l++) {
                        f[i][j][k][l] = 0;
                    }
                }
            }

        }
        f[0][0][0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i && j <= e.size() - 1; j++) {
                for (int k = 0; k < 2; k++) {
                    for (int l = 0; l < 2; l++) {
                        if (f[i][j][k][l] == 0) {
                            continue;
                        }
                        //printf("%d %d %d %d %d\n", i, j, k, l, f[i][j][k][l]);
                        for (char c = 'a'; c <= 'z'; c++) {
                            int newk = k, newl = l;
                            if (k == 0) {
                                if (c < s1[i]) {
                                    continue;
                                } else if (c > s1[i]) {
                                    newk = 1;
                                }
                            }
                            if (l == 0) {
                                if (c > s2[i]) {
                                    continue;
                                } else if (c < s2[i]) {
                                    newl = 1;
                                }
                            }
                            int pos = get(e, j, c);
                            if (pos == e.size()) {
                                continue;
                            }
                            //printf("%c %d %d %d %d\n", c, i + 1, pos, newk, newl);
                            (f[i + 1][pos][newk][newl] += f[i][j][k][l]) %= MO;

                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int j = 0; j < n && j <= e.size() - 1; j++) {
            for (int k = 0; k < 2; k++) {
                for (int l = 0; l < 2; l++) {
                    (ans += f[n][j][k][l]) %= MO;
                }
            }
        }
        return ans;
    }
};
