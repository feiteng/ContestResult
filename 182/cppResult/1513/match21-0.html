class Solution {
    int MOD=1e9+7;
public:
    int findGoodStrings(int n, string s1, string s2, string evil) {
        swap(s2,s1);
        vector<vector<vector<long long>>> dp(n+1,vector<vector<long long>>(4,vector<long long>(evil.size()+1,0)));
        dp[0][0][0]=1;
        for(int i=0;i<n;i++){
            for(int j=0;j<evil.size();j++){
                for(int kk=0;kk<4;kk++){
                    int c1,c2;
                    c1=kk&2;
                    c2=kk&1;
                    for(char c='a';c<='z';c++){
                        int nc1=c1;
                        int nc2=c2;
                        if(c1==0){
                            if(c>s1[i]) break;
                            if(c<s1[i]) nc1=2;
                        }

                        if(c2==0){
                            if(c<s2[i]) continue;;
                            if(c>s2[i]) nc2=1;
                        }

                        int nt=0;
                        if(c==evil[j]){
                            nt=j+1;
                        }else{
                            for(int len=j;len>=1;len--){
                                if(c!=evil[len-1]) continue;
                                bool f=true;
                                for(int k=len-2;k>=0;k--){
                                    if(evil[k]!=evil[k+j+1-len]) {
                                        f=false;
                                        break;
                                    }
                                }
                                if(f){
                                    nt=len;
                                    break;
                                }
                            }
                        }
                        dp[i+1][nc1+nc2][nt]=(dp[i+1][nc1+nc2][nt]+dp[i][kk][j])%MOD;
                    }
                }
            }
        }
        long long ans=0;
        for(int j=0;j<evil.size();j++){
                for(int kk=0;kk<4;kk++){
                    ans=(ans+dp[n][kk][j])%MOD;
                }
        }
        return ans;
    }
};
