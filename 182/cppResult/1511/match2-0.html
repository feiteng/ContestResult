class Solution {
public:
    int numTeams(vector<int>& rating) {
        int sz = rating.size();
        int ans = 0;
        for (int i = 0; i < sz; i++) {
            for (int j = i + 1; j < sz; j++) {
                for (int k = j + 1; k < sz; k++) {
                    if (rating[i] < rating[j] && rating[j] < rating[k])
                        ans++;
                    else if (rating[i] > rating[j] && rating[j] > rating[k])
                        ans++;
                }
            }
        }
        return ans;
    }
};
