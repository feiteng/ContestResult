class Solution {
public:
    int numTeams(vector<int>& a) {
        int n = a.size();
        int res = 0;
        for(int i = 0; i < n; ++i){
            for(int j = i + 1; j < n; ++j){
                for(int k = j + 1; k < n; ++k){
                    if(a[i] < a[j] && a[j] < a[k]) res++;
                    else if(a[i] > a[j] && a[j] > a[k]) res++;
                }
            }
        }
        return res;
    }
};
