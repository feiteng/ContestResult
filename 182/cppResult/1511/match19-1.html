class Solution {
public:
    int numTeams(vector<int>& rating) {
        int res = 0;
        int l = rating.size();
        for (int i = 0; i < l; i++) {
            for (int j = i + 1; j < l; j++) {
                for (int k = j + 1; k < l; k++) {
                    if (rating[i] < rating[j] && rating[j] < rating[k]) res++;
                    if (rating[i] > rating[j] && rating[j] > rating[k]) res++;
                }
            }
        }
        return res;
    }
};
