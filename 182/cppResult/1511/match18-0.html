class Solution {
public:
    int numTeams(vector<int>& arr) {
        int n  = arr.size();
        int ans = 0;
        for(int i = 0; i < n; i++){
            for(int j = i+1; j < n; j++){
                for(int k = j+1 ; k < n; k++){
                    if(arr[i] < arr[j] and arr[j] < arr[k]) ans++;
                    if(arr[i] > arr[j] and arr[j] > arr[k]) ans++;
                }
            }
        }
        return ans;
    }
};
