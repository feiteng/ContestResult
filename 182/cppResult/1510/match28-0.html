class Solution {
public:
    int findLucky(vector<int>& arr) {
        unordered_map<int,int> m;
        for (int num:arr)
            m[num]++;
        int res=-1;
        for (auto tmp:m)
            if (tmp.first==tmp.second)
                res=max(res,tmp.first);
        return res;
    }
};
