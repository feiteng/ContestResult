class Solution {
 public:
  int findLucky(vector<int>& arr) {
    unordered_map<int, int> freq;
    for (int a : arr) {
      freq[a]++;
    }
    int lucky = -1;
    for (int a : arr) {
      if (a == freq[a] && lucky < a) {
        lucky = a;
      }
    }
    return lucky;
  }
};
