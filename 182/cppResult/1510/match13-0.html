class Solution {
public:
    int findLucky(vector<int>& arr) {
        map <int, int> s;
        s.clear();
        for (int i : arr) {
            s[i]++;
        }
        int ans = -1;
        for (int i : arr) {
            if (s[i] == i) {
                ans = max(ans, i);
            }
        }
        return ans;
    }
};
