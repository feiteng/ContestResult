class Solution {
public:
    int findLucky(vector<int>& arr) {
        map<int, int> num_to_freq;
        for (auto num : arr) {
            num_to_freq[num]++;
        }
        int result = -1;
        for (auto &k_v : num_to_freq) {
            if (k_v.first == k_v.second) {
                result = k_v.first;
            }
        }
        return result;
    }
};
