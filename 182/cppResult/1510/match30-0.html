class Solution {
public:
    int findLucky(vector<int>& arr) {
        map<int, int> m;
        for(int a: arr) m[a]++;
        
        int r=-1;
        for(auto [a, b]: m)
        {
            if(a==b) r=max(r, b);
        }
        return r;
    }
};
