class Solution {
public:
    int findLucky(vector<int> &arr) {
        unordered_map<int, int> count;
        for (int n : arr) count[n]++;
        int ans = -1;
        for (auto &p : count) {
            if (p.first == p.second) ans = max(ans, p.first);
        }
        return ans;
    }
};
