class Solution:
    def numTeams(self, rating: List[int]) -> int:
        res = 0
        for l in range(len(rating)-2):
            for m in range(l, len(rating)-1):
                for r in range(m, len(rating)):
                    if rating[l] > rating[m] > rating[r] or rating[l] < rating[m] < rating[r]:
                        res += 1
        return res
