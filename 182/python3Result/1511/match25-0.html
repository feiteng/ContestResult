class Solution:
    def numTeams(self, r: List[int]) -> int:
        n = len(r)
        ret = 0
        for i in range(n):
            for j in range(i + 1, n):
                for k in range(j + 1, n):
                    if r[i] > r[j] > r[k] or r[i] < r[j] < r[k]: ret += 1
        return ret
