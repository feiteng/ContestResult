class Solution:
    def numTeams(self, rating: List[int]) -> int:
        cnt = 0
        for a in range(len(rating)):
            for b in range(a + 1, len(rating)):
                for c in range(b+1, len(rating)):
                    if (rating[a] < rating[b] < rating[c]) or (rating[a] > rating[b] > rating[c]):
                        cnt += 1
        return cnt
