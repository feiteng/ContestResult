class Solution:
    def numTeams(self, rating: List[int]) -> int:
        l = len(rating)
        ret = 0
        for i in range(l-2):
            for j in range(i+1,l):
                for k in range(j+1,l):
                    if rating[i] > rating[j] > rating[k] or (rating[i] < rating[j] < rating[k]):
                        ret += 1
        return ret
