class Solution:
    def numTeams(self, A: List[int]) -> int:
        n = len(A)
        res = 0
        
        for i in range(n-2):
            for j in range(i+1, n-1):
                for k in range(j+1, n):
                    if A[i] < A[j] < A[k] or A[i] > A[j] > A[k]:
                        res += 1
        
        return res
