class Solution:
    def numTeams(self, rating: List[int]) -> int:
        l = len(rating)
        n = 0
        for i in range(l):
            for j in range(i+1, l):
                for k in range(j+1, l):
                    if rating[k]>rating[j]>rating[i] or rating[k]<rating[j]<rating[i]:
                        n += 1
        return n
