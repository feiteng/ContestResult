class Solution:
    def numTeams(self, rating: List[int]) -> int:
        N = len(rating)
        cnt = 0
        for i in range(N - 2):
            for j in range(i + 1, N - 1):
                for k in range(j + 1, N):
                    if rating[i] < rating[j] < rating[k] or rating[i] > rating[j] > rating[k]:
                        cnt += 1
        return cnt
