class UndergroundSystem:

    def __init__(self):
        self.umap = {}
        self.peo = {}


    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.peo[id] = (stationName, t)


    def checkOut(self, id: int, stationName: str, t: int) -> None:
        ss, st = self.peo[id]
        if ss not in self.umap:
            self.umap[ss] = {}
        if stationName not in self.umap[ss]:
            self.umap[ss][stationName] = [0,0]
        self.umap[ss][stationName][1] += 1
        self.umap[ss][stationName][0] += t-st
        self.peo.pop(
id)
        


    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.umap[startStation][endStation][0]/self.umap[startStation][endStation][1]
