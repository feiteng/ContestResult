class UndergroundSystem:

    def __init__(self):
        
        self.station={}
        self.people={}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        
        #if id not in self.people:
        self.people[id]=[stationName,t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        
        st_station,st_t=self.people[id]
        
        if st_station not in self.station:
            self.station[st_station]={}
        if stationName not in self.station[st_station]:
            self.station[st_station][stationName]=[0,0]
        
        self.station[st_station][stationName][0]+=t-st_t
        self.station[st_station][stationName][1]+=1

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        
        return self.station[startStation][endStation][0]/self.station[startStation][endStation][1]
