class UndergroundSystem:

    def __init__(self):
        self.travels={} # start stations (start, t)
        self.averageTime={} # average travel time (start, end)

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.travels[id]=(stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start, st = self.travels[id]
        del self.travels[id]
        if (stationName, start) not in self.averageTime: self.averageTime[(stationName, start)] = (t-st,1)
        else:
            cul, times = self.averageTime[(stationName, start)]
            self.averageTime[(stationName, start)] = (cul+(t-st), times+1)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        cul, t = self.averageTime[(endStation, startStation)]
        return cul/t
