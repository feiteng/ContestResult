class UndergroundSystem:

    def __init__(self):
        self.customer = {}
        self.trips = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.customer[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        startStation, startTime = self.customer[id]
        trip = (startStation, stationName)
        
if trip in self.trips:
            total, times = self.trips[trip]
            self.trips[trip] = (total + t - startTime, times + 1)
        else:
            self.trips[trip] = (t - startTime, 1)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        total, times = self.trips[(startStation, endStation)]
        return 1. * total / times
