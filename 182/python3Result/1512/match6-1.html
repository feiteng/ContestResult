import collections

class UndergroundSystem:

    def __init__(self):
        self.id_map = {} # Where an id checked in
        self.travel_times = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.id_map[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start_s, start_t = self.id_map[id]
        if (start_s, stationName) not in self.travel_times:
            self.travel_times[(start_s, stationName)] = (1, t - start_t)
        else:
            total, avg = self.travel_times[(start_s, stationName)]
            new_avg = ((avg * total) + (t - start_t))/(total + 1)
            self
.travel_times[(start_s, stationName)] = (total + 1, new_avg)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.travel_times[(startStation, endStation)][1]
