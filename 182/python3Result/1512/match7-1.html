class UndergroundSystem:

    def __init__(self):
        self.buffer = {}
        self.avg = {}
        

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.buffer[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start_stat, start_time = self
.buffer[id]
        if (start_stat, stationName) not in self.avg:
            self.avg[(start_stat, stationName)] = (t - start_time, 1.0)
        else:
            avg, cnt = self.avg[(start_stat, stationName)]
            self.avg[(start_stat, stationName)] = ((avg * cnt + t - start_time)/(cnt+1), cnt+1)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.avg[(startStation, endStation)][0]
