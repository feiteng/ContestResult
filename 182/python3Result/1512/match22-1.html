from collections import defaultdict

class UndergroundSystem:
    def __init__(self):
        self.customers = defaultdict()
        self
.routeTimes = defaultdict(list)

    def checkIn(self, index: int, stationName: str, t: int) -> None:
        self.customers[index] = (t, stationName)

    def checkOut(self, index: int, stationName: str, t: int) -> None:
        timeIn, stationIn = self.customers.pop(index)
        self
.routeTimes[(stationIn, stationName)].append(t - timeIn)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        routeTimes = self.routeTimes[ (startStation, endStation) ]
        return sum(routeTimes) / len(routeTimes)
