class UndergroundSystem:

    def __init__(self):
        self.sin={}
        self.sout={}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.sin[id]=(stationName,t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        ss,st=self.sin[id]
        del self.sin[id]
        en,et=self.sout.get((ss,stationName),(0,0))
        self.sout[(ss,stationName)]=(en+1,et+t-st)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        en,et=self.sout.get((startStation,endStation),(0,0))
        if et>0:
            return et/en
        return 0
