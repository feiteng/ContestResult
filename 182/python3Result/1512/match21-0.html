class UndergroundSystem:

    def __init__(self):
        self.d = {}
        self.average = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.d[id] = [stationName, t]
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        begin, begin_t = self.d.pop(id)
        last_t = t - begin_t
        tup = (begin, stationName)
        if tup in self.average:
            sum_t, count = self.average[tup]
            self.average[tup] = [sum_t + last_t, count + 1]
        else:
            self.average[tup] = [last_t, 1]
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        tup = (startStation, endStation)
        if tup in self.average:
            sum_t, count = self.average[tup]
            return sum_t / count
        else:
            return 0
