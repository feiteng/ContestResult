class UndergroundSystem:
    
    # at check out, calculate time
    def __init__(self):
        self.cin={} # id:(stationName,time)
        self.cout={} # (station,from): [num of ppl, time]

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.cin[id]=(stationName,t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start=self.cin[id][0] #from
        start_t=self.cin[id][1]
        if (stationName,start) in self.cout:
            self.cout[(stationName,start)][0]+=1
            self.cout[(stationName,start)][1]+=t-start_t
        else:
            self.cout[(stationName,start)]=[1,t-start_t]

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.cout[(endStation,startStation)][1]/self.cout[(endStation,startStation)][0]
