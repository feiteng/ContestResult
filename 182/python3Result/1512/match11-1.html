class UndergroundSystem:

    def __init__(self):
        import collections
        self
.in_queue = {} # takes in id and has tuple of time + station
        self.avg_mapper = {} #takes in tuples of stations after being in an intermediary place 

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.in_queue[id] = (t,stationName)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start, station 
= self.in_queue[id]
        if (station,stationName) not in self.avg_mapper:
            self.avg_mapper[(station,stationName)] = [t-start, 1]
        else:
            self.avg_mapper[(station,stationName)][0] += t - start
            self.avg_mapper[(station,stationName)][1] += 1 
        
        
    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.avg_mapper[(startStation,endStation)][0] / self.avg_mapper[(startStation,endStation)][1]
