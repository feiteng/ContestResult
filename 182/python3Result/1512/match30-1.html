class UndergroundSystem:

    def __init__(self):
        self.cts = {}
        self.openTrips = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.openTrips[id] = [stationName, t]
        
    def checkOut(self, id: int, stationName: str, t: int) -> None:
        name, st = self.openTrips[id]
        if (name, stationName) not in self.cts:
            self.cts
[(name, stationName)] = [0, 0]
        self.cts[(name, stationName)][0] += t - st
        self.cts[(name, stationName)][1] += 1

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        x, y = self.cts[(startStation, endStation)]
        return x / y
