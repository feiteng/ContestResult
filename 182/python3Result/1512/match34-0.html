class UndergroundSystem:

    def __init__(self):
        self.checkInMap = dict()
        self.stationMap = dict()
        self.stationCount = dict()

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.checkInMap[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        old = self.checkInMap[id]
        travTime = t - old[1]
        pathKey = (old[0], stationName)
        totTime = self.stationMap.get(pathKey,0)
        self.stationMap[pathKey] = totTime + travTime
        
        count = self.stationCount.get(pathKey, 0)
        self.stationCount[pathKey] = count + 1    
        
    def getAverageTime(self, startStation: str, endStation: str) -> float:
        key = (startStation, endStation)
        return self.stationMap[key] / self.stationCount[key]
