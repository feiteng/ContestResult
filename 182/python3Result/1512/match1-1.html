class UndergroundSystem:

    def __init__(self):
        # id - from
        self.id2from = {}
        self.avg_time = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.id2from[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        station, t_0 = self.id2from[id]
        if (
station, stationName) not in self.avg_time:
            self.avg_time[(station, stationName)] = (t-t_0, 1)
        else:
            self.avg_time[(station, stationName)] = (self.avg_time[(station, stationName)][0]+(t-t_0), self.avg_time[(station, stationName)]
[1]+1)
            
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.avg_time[(startStation, endStation)][0]/self.avg_time[(startStation, endStation)][1]
