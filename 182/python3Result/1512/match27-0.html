class UndergroundSystem:

    def __init__(self):
        self.op = {}
        self.tr = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.op[id] = [stationName,t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        st, t1 = self.op[id]
        tr = self.tr
        if (st,stationName) in tr:
            avg, n = tr[(st,stationName)]
            tr[(st,stationName)] = [(avg*n+t-t1)/(n+1), n+1]
        else:
            tr[(st,stationName)] = [t-t1, 1]

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.tr[(startStation, endStation)][0]
