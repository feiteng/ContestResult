class UndergroundSystem:

    def __init__(self):
        self.in_transit = {}
        self.avg_time = collections.defaultdict(dict)

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.in_transit[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        check_in_station, check_in_time = self.in_transit[id]
        del self.in_transit[id]
        if check_in_station not in self.avg_time[stationName]:
            self.avg_time[stationName][check_in_station] = (0, 0)
        count, avg_time = self.avg_time[stationName][check_in_station]
        self.avg_time[stationName][check_in_station] = (count + 1, (avg_time * count + t - check_in_time) / (count + 1))

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.avg_time[endStation][startStation][1]
