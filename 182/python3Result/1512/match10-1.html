class UndergroundSystem:
    def __init__(self):
        self.travel_times = {}
        self.checked_in_customers = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.checked_in_customers[id] = [stationName, t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        old_station, old_time = self.checked_in_customers[id]

        if old_station not in self.travel_times:
            self.travel_times[old_station] = {}

        
if stationName not in self.travel_times[old_station]:
            self.travel_times[old_station][stationName] = [0, 0]

        self.travel_times[old_station][stationName][0] += t - old_time
        self.travel_times[old_station][stationName][1] += 1

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        tot_time, tot_customers = self.travel_times[startStation][endStation]

        return tot_time / tot_customers
