class UndergroundSystem:

    def __init__(self):
        self.d_customer = {}
        self.d_stations = {}
    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.d_customer[id] = [stationName, t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        sn, t1 = self.d_customer[id]
        if (sn, stationName) in self.d_stations:
            x, y = self.d_stations[(sn, stationName)]
            self.d_stations[(sn, stationName)] = [x+t-t1,y+1]
        else:
            self.d_stations[(sn, stationName)] = [t-t1, 1]

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        if (startStation, endStation) in self.d_stations:
            x, y = self.d_stations[(startStation, endStation)]
            return x/y
        return 0
