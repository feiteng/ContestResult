from collections import defaultdict

class UndergroundSystem:
    def __init__(self):
        self.travels = [None for i in range(1000001)]
        self.totalTime = defaultdict(int)
        self.totalCount = defaultdict(int)

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.travels[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        startStation, startTime = self.travels[id]
        self.totalTime[(startStation, stationName)] += (t - startTime)
        self.totalCount
[(startStation, stationName)] += 1

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        key = (startStation, endStation)
        return self.totalTime[key] / self.totalCount[key]
