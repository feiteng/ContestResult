class UndergroundSystem:

    def __init__(self):
        self.checkInMap = {}
        self.checkOutMap = {}
        self.timeMap = {}


    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.checkInMap[id] = [stationName, t]


    def checkOut(self, id: int, stationName: str, t: int) -> None:
        self.checkOutMap[id] = [stationName, t]
        timeTravelled = self.checkOutMap[id][1] - self.checkInMap[id][1]
        curKey = self.checkInMap[id][0] + " - " + self.checkOutMap[id][0]
        if curKey not in self.timeMap:
            self.timeMap[curKey] = [0, 0]
        self.timeMap[curKey][0] += timeTravelled
        self.timeMap[curKey][1] += 1
            


    def getAverageTime(self, startStation: str, endStation: str) -> float:
        curKey = startStation + " - " + endStation
        
        return self.timeMap[curKey][0] * 1.0 / self.timeMap[curKey][1]
