class UndergroundSystem:

    def __init__(self):
        self.ids = {}
        self.average_times = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.ids[id] = (stationName, t)
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        checkIn, time = self.ids[id]
        del self.ids[id]
        self.average_times.setdefault((
checkIn, stationName), []).append(t-time)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        get_times = self.average_times[(startStation, endStation)]
        return sum(get_times)/len(get_times)
