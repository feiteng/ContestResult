class UndergroundSystem:

    def __init__(self):
        self.id_to_in_station = {}
        self.count_time = {}
        

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.id_to_in_station[id] = [stationName, t]
        if stationName not in self.count_time:
            self.count_time[stationName] = {}
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start, t0 = self.id_to_in_station[id]
        if stationName in self.count_time[start]:
            self.count_time[start][stationName][0] += 1
            self.count_time[start][stationName][1] += t - t0
        else:
            self.count_time[start][stationName] = [1, t-t0]
            
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.count_time[startStation][endStation][1]/self.count_time[startStation][endStation][0]
