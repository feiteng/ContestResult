class UndergroundSystem:

    def __init__(self):
        self.checkins = {}
        self.averages = {}
        

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.checkins[id] = (stationName, t)
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        startStation, startTime = self.checkins[id]
        timeTaken = t - startTime
        if (startStation, stationName) in self.averages:
            totalTime, number = self.averages[(startStation, stationName)]
            self.averages[(
startStation, stationName)] = (totalTime + timeTaken, number+1)
        else:
            self.averages[(startStation, stationName)] = (t - startTime, 1)
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        total, count = self.averages[(startStation, endStation)]
        return total / count
