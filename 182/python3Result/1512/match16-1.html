class UndergroundSystem:

    def __init__(self):
        self.travel = {}
        self.id = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.id[id] = [stationName, t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        stationName0, t0 = self.id[id]
        if (stationName0, stationName) in self.travel:
            self.travel[(stationName0, stationName)][0] += t-t0
            self.travel[(stationName0, stationName)][1] += 1    
        else:
            self
.travel[(stationName0, stationName)] = [t-t0, 1]
        del self.id[id]

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        t, n = self.travel[(startStation, endStation)]
        return t/n
