class UndergroundSystem:

    def __init__(self):
        self.total = {}
        self.checkin = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.checkin[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        in_name, in_t = self.checkin.get(id, (
'', 0))
        time, count = self.total.get((in_name, stationName), (0, 0))
        self.total[(in_name, stationName)] = (time + t - in_t, count + 1)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        time, count = self.total.get((startStation, endStation), (0, 1))
        return time / count
