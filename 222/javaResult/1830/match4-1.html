class Solution {
    public int countPairs(int[] ds) {
        
        Map<Integer,Long> mp  =  new HashMap<>();    
        long a = 0;
        long mod = 1000000007L;
        for(int i=0;i<ds.length;++i){
            for(int j=30;j>=0;--j){
                int ot = (1<<j) - ds[i];
                
                a += 
mp.getOrDefault(ot,0L);
                a %= mod;
                
            }
            mp.put(ds[i],mp.getOrDefault(ds[i],0L)+1L);
        }
        
        return (int)a;
        
        
    }
}
