class Solution {
    public int countPairs(int[] deliciousness) {
        Map<Integer, Integer> map = new HashMap<>();
        // Arrays.sort(deliciousness);
        int res = 0;
        for (int num : deliciousness) {
            for (int i = 0; i <= 21; i++) {
                int comb = 1 << i;
                if (comb < num) {
                    continue;
                }
                // if (comb - num > num) {
                //     break;
                // }
                res = (res + 
map.getOrDefault(comb - num, 0)) % 1000000007;
            }
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        return res;
    }
}
