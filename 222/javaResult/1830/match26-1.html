class Solution {
    public int countPairs(int[] d) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i: d) {
            if (map.containsKey(i)) {
                map.put(i, map.get(i) + 1);
            } else {
                map.put(i, 1);
            }
        }
        
        long res = 0;
        for (int key: map.keySet()) {
            for (int x = 0; x <= 21; x++) {
                int val = (1 << x) - key;
                if (val >= key && 
map.containsKey(val)) {
                    // System.out.println(key + " " + val);
                    if (val == key) {
                        res += ((long) map.get(key)) * (map.get(key) - 1) / 2;
                    } else {
                        res += ((long) map.get(key)) * map.get(val);
                    }
                }
            }
        }
        
        int MOD = (int) 1e9 + 7;
        return (int) (res % MOD);
    }
}
