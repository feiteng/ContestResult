class Solution {
    public int countPairs(int[] de) {
        HashMap<Long,Long> map=new HashMap<>();
        long count=0;
        long mod=1000000000+7;
        for(long i:de)
        {
            for(int j=0;j<23;j++)
            {
                count+=map.getOrDefault((1L<<j)-i,0L);
                count%=mod;
            }
            map.put(i,map.getOrDefault(i,0L)+1);
        }
        return (int)count;
    }
}
