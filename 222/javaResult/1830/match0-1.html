class Solution {
    public int countPairs(int[] deliciousness) {
        Map<Integer,Integer>maps = new HashMap<>();
        int ans = 0;
        int MAX_NUM = 1000000007;
        for(int x:deliciousness)
        {
            for(int i=0;i<22;i++)
            {
                int sum = (int)Math.pow(2,i);
                
if(maps.containsKey(sum-x))
                {
                    ans = (ans+maps.get(sum-x))%MAX_NUM;
                }
            }
            maps.put(x,maps.getOrDefault(x, 0)+1);
        }
        return ans;
    }
}
