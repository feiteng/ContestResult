class Solution {
    public int countPairs(int[] d) {
        int mod=(int)1e9+7;
        Map<Integer,Integer> map=new HashMap<>();
//        for(int t:d){
//
//        }
        long ans=0L;
        for(int t:d){
            for(int i=0;i<32;i++){
                int x=(int)Math.pow(2,i);
                if(!map.containsKey(x-t))continue;
                ans+=map.get(x-t);
                ans%=mod;
            }
            map.put(t,map.getOrDefault(t,0)+1);
        }
        return (int)ans;
    }
}
