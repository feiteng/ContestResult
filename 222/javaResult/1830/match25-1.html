class Solution {
    long MOD = (long) 1e9 + 7;
    public int countPairs(int[] deliciousness) {
        long sum = 0L;
        int N = 1 << 21;
        int [] cnt = new int[N + 1];
        for(int i : deliciousness) cnt[i]++;
        for(int i = 0; i <= N / 2; i++){
            for(int j = N; j >= 1; j >>= 1)
            {
                if(j - i < i) break;
                if(i == j - i)
                    sum += 1L * cnt[i] * (cnt[i] - 1) / 2;
                else 
                    sum += 1L * cnt[i] * cnt[j - i]
;
            }                    
        }
        return (int)(sum % MOD);
    }
}
