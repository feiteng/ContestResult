class Solution {
    public int countPairs(int[] deliciousness) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0, mod = 1000000007;
        for (int d : deliciousness) {
            for (int i = 31;i >= 0;i--) {
                int cur = 1 << i;
                if (cur < d) continue;
                res = (res + 
map.getOrDefault(cur - d, 0)) % mod;
            }
            map.put(d, map.getOrDefault(d, 0) + 1);
        }
        return res;
    }
}
