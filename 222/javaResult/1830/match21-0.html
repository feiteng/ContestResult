class Solution {
    public int countPairs(int[] deliciousness) {
        int result = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for(int d : deliciousness) {
            for(int i = 0; i <= 21; ++i) {
                int other = (1 << i) - d;
                result += map.getOrDefault(other, 0);
                result %= 1000000007;
            }
            map.put(d, map.getOrDefault(d, 0) + 1);
        }
        return result;
    }
}
