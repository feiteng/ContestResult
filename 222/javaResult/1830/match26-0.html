class Solution {
    public int countPairs(int[] deliciousness) {
        HashMap<Integer,Integer> map = new HashMap<>();
        for(int i:deliciousness){
            if(map.containsKey(i)){
                map.put(i,map.get(i)+1);
            }
            else{
                map.put(i,1);
            }
        }
        long ret=0;
        for(int key:map.keySet()){
            for(int i = 0;i<=21;i++){
                int a = 1<<i;
                if(a-key<key){
                    continue;
                }
                if(map.containsKey(a-key)){
                    if(a-key!=key){
                        ret+=map.get(key)*map.get(a-key);
                    }
                    else{
                        ret+=(long)map.get(key)*(map.get(key)-1)/2;
                    }
                    
                }
                
                ret%=1000000007;
            }
        }
        return (int)ret;

    }
}
