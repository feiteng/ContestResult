class Solution {
    public int countPairs(int[] deliciousness) {
        int[] sums = new int[22];
        for (int i = 0; i < 22; i ++) {
            sums[i] = (int)Math.pow(2, i);
        }
        Map<Integer, Integer> map = new HashMap<>();
        long res = 0;
        long mod = (int)1e9 + 7;
        for (int i : deliciousness) {
            for (int j : sums) {
                if (map.containsKey(j - i)) {
                    res = (res + map.get(j - i)) % mod;
                }
                
            }
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        return (int)res;
    }
}
