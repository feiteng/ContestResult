class Solution {
    public int countPairs(int[] deliciousness) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int res = 0;
        int mod = 1000000007;
        for(int d:deliciousness) {
            for (int p = 0;p<=21;p++) {
                int s = 1<<p;
                res = (res + map.getOrDefault(s-d,0))%mod;
            
}
            map.put(d, map.getOrDefault(d,0)+1);
        }
        return res;
    }
}
