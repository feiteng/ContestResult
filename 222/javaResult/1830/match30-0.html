class Solution {
    
    public int countPairs(int[] deliciousness) {
        Map<Integer, Integer> valToCount = new HashMap<>();
        int mod = 1_000_000_000 + 7;
        int res = 0;
        for(int v: deliciousness) {
            for(int i = 0; i < 30; i++) {
                int target = (1 << i);
                if(valToCount.containsKey(target - v)) {
                    res = (res + valToCount.get(target - v)) % mod;
                }
            }
            valToCount.put(v, 1 + valToCount.getOrDefault(v, 0));
        }
        return res;
    }
}
