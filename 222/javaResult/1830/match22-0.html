class Solution {
    public int countPairs(int[] D) {
        Map<Integer, Integer> mem = new HashMap<>();
        int mod = 1_000_000_007;
        long ret = 0;
        Arrays.sort(D);
        for (int d : D) {
            for (int i = 1; i <= 2 * d; i *= 2) {
                if (mem.containsKey(i - d)) {
                    ret += mem.get(i - d);
                }
                ret %= mod;
            }
            mem.put(d, mem.getOrDefault(d, 0) + 1);
        }
        
    
        return (int)ret;
    }
}
