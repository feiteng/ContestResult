class Solution {
    public int countPairs(int[] deliciousness) {
        int mod = (int)(1e9 + 7);
        Map<Integer, Integer> cnt = new HashMap<>();
        int res = 0;
        for (int d : deliciousness) {
            for (int i = 21; i >= 0; i--) {
                res = (res + 
cnt.getOrDefault((1 << i) - d, 0)) % mod;
            }
            cnt.put(d, cnt.getOrDefault(d, 0) + 1);
        }
        return res;
    }
}
