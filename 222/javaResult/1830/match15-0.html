class Solution {
  public int countPairs(int[] deliciousness) {
    Map<Integer, Integer> count = new HashMap<>();
    long total = 0;
    for (int i : deliciousness) {
      for (int j = 0; j < 23; j++) {
        total += count.getOrDefault((1 << j) - i, 0);
        total %= 1_000_000_007;
      }
      count.put(i, count.getOrDefault(i, 0) + 1);
    }
    return (int) total;
  }
}
