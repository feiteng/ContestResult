class Solution {

	public int countPairs(int[] deliciousness) {
		long count = 0;
		HashMap<Integer, Long> map = new HashMap<>();
		for (int i : deliciousness) {
			for (int j = 0; j <= 21; j++) {
				count = (count + map.getOrDefault((1 << j) - i, 0L)) % 1000000007;
			}
			map.put(i, map.getOrDefault(i, 0L) + 1);
		}
		return (int) count;
	}
}
