class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        int res = 0, i = 0;
        while(truckSize > 0 && i < boxTypes.length) {
            int num = Math.min(truckSize, boxTypes[i][0]);
            res += num * boxTypes[i][1];
            i++;
            truckSize -= num;
        }
        return res;
    }
}
