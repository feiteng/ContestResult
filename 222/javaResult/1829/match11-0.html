class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (p, q) -> q[1] - p[1]);
        int units = 0;
        for (int[] boxType : boxTypes) {
            if (boxType[0] <= truckSize) {
                units += boxType[0] * boxType[1];
                truckSize -= boxType[0];
            } else {
                units += truckSize * boxType[1];
                break;
            }
        }
        return units;
    }
}
