class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (b1, b2) -> (b1[1] - b2[1]));
        int total = 0;
        int idx = boxTypes.length - 1;
        while (idx >= 0 && truckSize > 0) {
            int take = Math.min(truckSize, boxTypes[idx][0]);
            total += take * boxTypes[idx][1];
            truckSize -= take;
            idx --;
        }
        return total;
    }
}
