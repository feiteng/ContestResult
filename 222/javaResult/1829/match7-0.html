class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        int ans = 0;
        for (int[] box : boxTypes){
            if (box[0] < truckSize){
                truckSize -= box[0];
                ans += box[0] * box[1];
            }else{
                ans += truckSize * box[1];
                break;
            }
        }
        return ans;
    }
}
