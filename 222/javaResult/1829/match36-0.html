import java.util.Arrays;

class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> (b[1] - a[1]));
        int ans = 0;
        int pos = 0;
        while (pos < boxTypes.length && truckSize > 0) {
            int take = Math.min(truckSize, boxTypes[pos][0]);
            ans += take * boxTypes[pos][1];
            truckSize -= take;
            pos++;
        }
        return ans;
    }
}
