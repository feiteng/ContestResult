class Solution {
    public int maximumUnits(int[][] bx, int sz) {
        Arrays.sort(bx,(x,y)->{ return y[1] -  x[1];});
        int a = 0;
        for(int i=0;i<bx.length && sz >0 ;++i){
            int m = Math.min(bx[i][0],sz);
            sz -= m;
            
            a += m*(bx[i][1]);
            
        }
        return a;
        
        
    }
}
