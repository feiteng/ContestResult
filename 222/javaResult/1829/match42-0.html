class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> (b[1] - a[1]));
        int res = 0;
        for (int[] box : boxTypes) {
            if (truckSize > 0) {
                int car = Math.min(box[0], truckSize);
                res += car * box[1];
                truckSize -= car;
            } else {
                break;
            }
        }
        return res;
    }
}
