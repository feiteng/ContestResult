class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (o1, o2) -> o2[1] - o1[1]);
        int ans = 0;
        for(int[] b: boxTypes) {
            if(b[0] <= truckSize) {
                truckSize -= b[0];
                ans += (b[0] * b[1]);
            }else {
                ans += (truckSize * b[1]);
                break;
            }
        }
        return ans;
    }
}
