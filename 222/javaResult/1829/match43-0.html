class Solution {
    public int maximumUnits(int[][] b, int t) {
        Arrays.sort(b,(x,y)->y[1]-x[1]);
        int res = 0;
        int i = 0;
        while(t>0&&i<b.length){
            int[] cur = b[i];
            //System.out.println(cur[1]);
            if(cur[0]<=t){
                res+=cur[0]*cur[1];
            }else{
                res+=t*cur[1];
            }
            i++;
            t-=cur[0];
        }
        return res;
    }
}
