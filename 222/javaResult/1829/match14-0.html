class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (o1, o2) -> o2[1] - o1[1]);
        int n = truckSize;
        int res = 0;
        for (int i = 0; i < boxTypes.length; i++) {
            int t = Math.min(n, boxTypes[i][0]);
            res += t * boxTypes[i][1];
            n -= t;
        }
        return res;
    }
}
