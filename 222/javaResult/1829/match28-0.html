class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a,b) -> Integer.compare(b[1], a[1]));
        int res = 0;
        int j=0;
        while (j<boxTypes.length && truckSize > 0) {
            int count = Math.min(truckSize, boxTypes[j][0]);
            res += count * boxTypes[j][1];
            truckSize-=count;
            j++;
        }
        return res;
    }
}
