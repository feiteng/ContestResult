class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        int ans = 0;
        for (int i = 0; i < boxTypes.length && truckSize > 0; i++) {
            int n = Math.min(boxTypes[i][0], truckSize);
            ans += n*boxTypes[i][1];
            truckSize -= n;
        }
        return ans;
    }
}
