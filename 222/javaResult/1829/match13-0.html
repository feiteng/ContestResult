class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        
        int ans = 0;
        int N = boxTypes.length;
        for (int i = 0; i < N; i++) {
            int b = Math.min(boxTypes[i][0], truckSize);
            ans += b * boxTypes[i][1];
            truckSize -= b;
        }
        
        return ans;
    }
}
