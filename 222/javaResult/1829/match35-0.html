class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        int res = 0, p = 0;
        while(truckSize > 0 && p < boxTypes.length) {
            int cnt = Math.min(truckSize, boxTypes[p][0]);
            res += cnt * boxTypes[p][1];
            
            truckSize -= cnt;
            p++;
        }
        
        return res;
    }
}
