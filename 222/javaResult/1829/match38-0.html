class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a, b) -> b[1] - a[1]);
        int res = 0;
        int index = 0;
        while (truckSize > 0 && index < boxTypes.length) {
            int size = Math.min(truckSize, boxTypes[index][0]);
            res += boxTypes[index][1] * size;
            // System.out.println(boxTypes[index][1]);
            truckSize -= size;
            index += 1;
        }
        return res;
    }
}
