class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (b1, b2) -> b2[1] - b1[1]);
        int ans = 0;
        for (int[] box: boxTypes) {
            int num = Math.min(truckSize, box[0]);
            ans += num * box[1];
            truckSize -= num;
        }
        return ans;
    }
}
