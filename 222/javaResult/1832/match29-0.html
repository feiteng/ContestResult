class Solution {
    public int minOperations(int[] target, int[] arr) {
        int tn = target.length;
        int an = arr.length;
        HashMap<Integer,Integer> hm = new HashMap<>();
        for(int i=0;i<tn;i++) {
            hm.put(target[i],i);
        }
        int[] len = new int[tn+1];
        int max = 0;
        len[0] = -1;
        for(int i=0;i<an;i++) {
            if(!hm.containsKey(arr[i])) continue;
            int p = hm.get(arr[i]);
            int l = 0,r = max;
            while(l<r) {
                int mid = l+(r-l+1)/2;
                if(len[mid]<p) {
                    l = mid;
                } else {
                    r = mid-1;
                }
            }
            if (r>=max) {
                max++;
                len[max] = p;
            } else {
                len[r+1] = Math.min(len[r+1],p);   
            }
            // System.out.println(String.format("%d,%d,%d",arr[i],p,max));
            // System.out.println(Arrays.toString(len));
        }
        return tn-max;
    }
}
