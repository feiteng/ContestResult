class Solution {
    public int minOperations(int[] t, int[] a) {
        
        int l1 = t.length;
        int l2 = a.length;
        
        Map<Integer,Integer> mp = new HashMap<>();
        
        for(int i=0;i<l1;++i){
            mp.put(t[i],i+1);
        }
        int aa[]  = new int[l2];
        int p = 0;
        for(int i=0;i<l2;++i){
            if(mp.containsKey(a[i])){
                aa[p++] = mp.get(a[i]);
            }
        }
        int ans = 0;
        int g[] = new int[p];
        Arrays.fill(g,2000000000);
        int d[] = new int[p];
        for(int i=0;i<p;++i){
            int cur =  aa[i];
            
            
int lo = 0;int hi = p-1;
            while(lo<hi){
                int mi = (lo+hi)/2;
                if(g[mi]>=cur){
                    hi = mi;
                }else{
                    lo =  mi+1;
                }
            }
            d[i] = lo+1;
            g[lo] =  cur;
            ans = Math.max(ans,d[i]);
            
            
        }
        return l1-ans;
        
        
        
        
        
    }
}
