class Solution {
    public int minOperations(int[] target, int[] arr) {
        Map<Integer,Integer>idxs = new HashMap<>();
        int m = target.length;
        for(int i=0;i<m;i++)
        {
            idxs.put(target[i],i);
        }
        List<Integer>lists = new ArrayList<>();
        for(int x:arr)
        {
            if(idxs.containsKey(x))
            {
                lists.add(idxs.get(x));
            }
        }

        int[] nums = lists.stream().mapToInt(Integer::valueOf).toArray();

        int len = 1, n = nums.length;
        if (n == 0) {
            return m;
        }
        int[] dp = new int[n + 1];
        dp[len] = nums[0];
        for (int i = 1; i < n; ++i) {
            if (nums[i] > dp[len]) {
                dp[++len] = nums[i];
            } else {
                int l = 1, r = len, pos = 0;
                while (l <= r) {
                    int mid = (l + r) >> 1;
                    if (dp[mid] < nums[i]) {
                        pos = mid;
                        l = mid + 1;
                    } else {
                        r = mid - 1;
                    }
                }
                dp[pos + 1] = nums[i];
            }
        }
        return m-len;
    }
}
