class Solution {
    public int minOperations(int[] target, int[] arr) {
        Map<Integer, Integer> pos = new HashMap();
        
for (int i = 0; i < target.length; i++) {
            pos.put(target[i], i);
        }
        List<Integer> ls = new ArrayList();
        for (int i = 0; i < arr.length; i++) {
            int n = arr[i];
            if (pos.containsKey(n)) {
               ls.add(pos.get(n));
            }
        }
        int[] a2 = new int[ls.size()];
        for (int i = 0; i < ls.size(); i++) {
            a2[i] = ls.get(i);
        
}
        int len = lengthOfLIS(a2);
        return target.length - len;        
    }
    
    public int lengthOfLIS(int[] nums) {
        
        if (nums == null || nums.length == 0)
            return 0;
        
        List<Stack<Integer>> stacks = new ArrayList<Stack<Integer>>();
        
        
int size = 0;
        for (int num : nums) {
            
            int low = 0;
            int high = size;
            
            while (low < high) {
                int mid = (low + high)/2;
                
                // because of the existence of duplicates we want
                // dups to be in the same stack
                if (num <= 
stacks.get(mid).peek()) {
                    high = mid;
                }
                else {
                    low = mid + 1;
                }
            }
            
            if (low == size) {
                stacks.add(new Stack<Integer>());
                size++;
            }
            stacks.get(low).push(num);
        }
        
        return size;
            
    }
}
