class Solution {
    public int minOperations(int[] target, int[] arr) {
        int n = target.length;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.put(target[i], i);
        }
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < arr.length; i++) {
            if (!map.containsKey(arr[i])) {
                continue;
            }
            list.add(map.get(arr[i]));
        }
        int count = LIS(list);
        return n - count;
    }
    
    
    public int LIS(List<Integer> list) {
        if (list == null || list.size() == 0) {
            return 0;
        }
        int n = list.size();
        int[] nums = new int[n];
        for (int i = 0; i < n; i++) {
            nums[i] = list.get(i);
        }
        int[] dp = new int[n + 1];
        dp[0] = Integer.MIN_VALUE;
        for (int i = 1; i <= n; i++) {
            dp[i] = Integer.MAX_VALUE;
        }
        
        int ans = 1;
        for (int i = 0; i < n; i++) {
            int index = findGOE(dp, nums[i]);
            dp[index] = nums[i];
            ans = Math.max(ans, index);
        }
        
        return ans;
    }
    
    private int findGOE(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;
        while (left + 1 < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] >= target) {
                right = mid;
            } else {
                left = mid;
            }
        }
        if (nums[left] >= target) {
            return left;
        }
        return right;
    }
}
