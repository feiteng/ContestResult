class Solution {
    public int minOperations(int[] target, int[] arr) {
        int n = target.length;
        
HashMap<Integer,Integer> hm = new HashMap<>();
        for(int i=0;i<n;i++){
            hm.put(target[i],i);
        }
        
        int m =arr.length;
        int[] ret = new int[m+1];
        ret[0] = -1;
        for(int i=1;i<=m;i++){
            if(hm.containsKey(arr[i-1])) ret[i] = hm.get(arr[i-1]);
            else ret[i] = -1;
        }
        
        int res = lengthOfLIS(ret);
        
        
        return n - res + 1;
    
}
    
    public int lengthOfLIS(int[] nums) {
        int len = 1, n = nums.length;
        if (n == 0) {
            return 0;
        }
        int[] d = new int[n + 1];
        d[len] = nums[0];
        for (int i = 1; i < n; ++i) {
            if (nums[i] > d[len]) {
                d[++len] = nums[i];
            } else {
                int l = 1, r = len, pos = 0; // 如果找不到说明所有的数都比 nums[i] 大，此时要更新 d[1]，所以这里将 pos 设为 0
                while (l <= r) {
                    int mid = (l + r) >> 1;
                    if (d[mid] < nums[i]) {
                        pos = mid;
                        l = mid + 1;
                    } else {
                        r = mid - 1;
                    }
                }
                d[pos + 1] = nums[i];
            }
        }
        return len;
    }


}
