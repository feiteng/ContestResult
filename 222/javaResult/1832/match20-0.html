class Solution {
    public int minOperations(int[] target, int[] arr) {
        HashMap<Integer, Integer> indices = new HashMap<>();
        for(int i = 0; i < target.length; i++) {
            indices.put(target[i], i);
        }
        ArrayList<Integer> positions = new ArrayList<Integer>();
        for(int i = 0; i < arr.length; i++) {
            if(indices.containsKey(arr[i])) {
                positions.add(indices.get(arr[i]));
            }
        }
        return target.length - lis(positions);
    }
    
    public int lis(ArrayList<Integer> positions) {
        int [] lengths = new int[positions.size()];
        int longest = 0;
        for(int num: positions) {
            int i = 0;
            int j = longest;
            while(i != j) {
                int mid = (i + j)/2;
                if(lengths[mid] < num) {
                    i = mid + 1;
                } else {
                    j = mid;
                }
            }
            lengths[i] = num;
            if(i == longest) longest++;
        }
        return longest;
    }
}
