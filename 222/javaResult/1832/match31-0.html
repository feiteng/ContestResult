class Solution {
    public int minOperations(int[] target, int[] arr) {
        final int n = target.length; 
        
        Map<Integer, Integer> map = new HashMap<>(); // target indexes
        for(int i = 0; i<n; i++)
            map.put(target[i], i);
        
        int[] lis = new int[n+1]; // holds indexes of longest subsequence of target in arr -> holds increasing indexes 
        int end = 0; // the max size lis was 
        for(int num : arr){
            if(!map.containsKey(num))
                continue; 
            
            int idx = map.get(num); 
            int pos = Arrays.binarySearch(lis, 0, end, idx); // search lis, indexes 0-end, looking for idx
            if(pos >= 0) // already in
                continue; 
            
            pos = -1 - pos; // insertion point
            if(pos == end) // all numbers are smaller
                end++; 
            lis[pos] = idx; 
        }
        return n - end; 
    }
}
