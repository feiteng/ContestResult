class Solution {

    private int[] dp;

    private int solve(List<Integer> list) {
        Arrays.fill(dp , Integer.MAX_VALUE);
        dp[0] = - 1;
        
int ans = 0;
        for (int value : list) {
            int l = 0 , r = dp.length;
            while (l < r) {
                int mid = (l + r) / 2;
                if (dp[mid] < value) {
                    l = mid + 1;
                } else {
                    r = mid;
                }
            }
            dp[l] = value;
            ans = Math.max(ans , l);
        }
        return ans;
    }

    public int minOperations(int[] target, int[] arr) {
        dp = new int[target.length + 1];
        
Map<Integer , Integer> map = new HashMap<>();
        for (int i = 0;i < target.length;i ++) {
            map.put(target[i] , i);
        }
        List<Integer> list = new ArrayList<>();
        for (int value : arr) {
            if (map.containsKey(value)) {
                list.add(map.get(value));
            }
        }
        // lis
        int result = solve(list);
        return target.length - result;
    }
}
