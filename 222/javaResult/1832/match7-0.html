class Solution {
    public int minOperations(int[] target, int[] arr) {
        int tl = target.length, al = arr.length;

        Map<Integer, Integer> mm = new HashMap<>();
        for (int i = 0; i < tl; i++) {
            mm.put(target[i], i);
        }
        List<Integer> arrIdxLst = new ArrayList<>();
        for (int i = 0; i < al; i++) {
            if (mm.containsKey(arr[i])) {
                arrIdxLst.add(mm.get(arr[i]));
            }
        }

        int[] arrIdx = new int[arrIdxLst.size()];
        for (int i = 0; i < arrIdx.length; i++) {
            arrIdx[i] = arrIdxLst.get(i);
        }

        return target.length - lis(arrIdx, arrIdx.length);
    }

    private int lis(int[] A, int n) {
        int[] B = new int[n + 1];
        B[1] = A[0];
        int len = 1, start = 0, end = len, mid;
        for (int i = 1; i < n; i++) {
            if (A[i] > B[len]) {
                len++;
                B[len] = A[i];
            } else {
                start = 1;
                end = len;
                while (start <= end) {
                    mid = (start + end) / 2;
                    if (B[mid] < A[i]) {
                        start = mid + 1;
                    } else {
                        end = mid - 1;
                    }
                }
                B[start] = A[i];
            }
        }
        return len;
    }
}
