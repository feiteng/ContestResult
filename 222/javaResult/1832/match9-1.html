class Solution {
    HashMap<Integer, Integer> map = new HashMap<>();
    public int minOperations(int[] target, int[] arr) {
        int n=target.length,m=arr.length;
        map.clear();
        for(int i=0;i<n;i++){
            map.put(target[i],i);
        }
        ArrayList<Integer> list = new ArrayList<>();
        for(int x:arr){
            if(map.get(x)!=null){
                list.add(map.get(x));
            }
        }
        int[] ints =  new int[list.size()];
        for(int i=0;i<list.size();i++){
            ints[i]=list.get(i);
        }
        return n-lengthOfLIS(ints);
    
}
    public int lengthOfLIS(int[] nums) {
        int len = 1, n = nums.length;
        if (n == 0) {
            return 0;
        }
        int[] d = new int[n + 1];
        d[len] = nums[0];
        for (int i = 1; i < n; ++i) {
            if (nums[i] > d[len]) {
                d[++len] = nums[i];
            } else {
                int l = 1, r = len, pos = 0; // 如果找不到说明所有的数都比 nums[i] 大，此时要更新 d[1]，所以这里将 pos 设为 0
                while (l <= r) {
                    int mid = (l + r) >> 1;
                    if (d[mid] < nums[i]) {
                        pos = mid;
                        l = mid + 1;
                    } else {
                        r = mid - 1;
                    }
                }
                d[pos + 1] = nums[i];
            }
        }
        return len;
    }


}
