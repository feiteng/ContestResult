class Solution {
    public int minOperations(int[] target, int[] arr) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int i = 1;
        for (int num:target) {
            map.put(num, i);
            i++;
        }
        List<Integer> lis = new ArrayList<>();
        for (int num:arr) {
            if (map.containsKey(num)) lis.add(map.get(num));
        }
        int l = LIS(lis);
        return target.length- l;
    }

    public int LIS(List<Integer> arr) {
        int n = arr.size();
        int[] tail = new int[n+1];
        int len = 0;
        for (int i=0;i<n;i++) {
            int lo = 0;
            int hi = len;
            while (lo<hi) {
                int m = (lo+hi)/2;
                if (tail[m]<arr.get(i)) {
                    lo = m+1;
                } else {
                    hi = m;
                }
            }
            if (lo == len) len++;
            tail[lo] = arr.get(i);
        }
        return len;
    }
}
