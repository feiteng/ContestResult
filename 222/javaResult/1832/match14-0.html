class Solution {
    public int minOperations(int[] target, int[] arr) {
        Map<Integer, Integer> map = new HashMap();
        for (int i = 0; i < target.length; i++) {
            map.put(target[i], i);
        }
        int overlap = 0;
        int currOverlap = 0;
        List<Integer> indices = new ArrayList();
        for (int i = 0; i < arr.length; i++) {
            int curr = arr[i];
            if (!map.containsKey(curr)) {
                continue;
            }
            indices.add(map.get(curr));
        }
        
        int n = indices.size();
        int[] dp = new int[n];
        int len = 0;
    
        for (int i = 0; i < n; i++) {
            int val = indices.get(i);
            int ind = Arrays.binarySearch(dp, 0, len, val);
            if (ind < 0) {
                ind = -(ind + 1);
            }
            dp[ind] = val;
            if (ind == len) {
                len += 1;
            }
        }
        return target.length - len; 
    }
}
