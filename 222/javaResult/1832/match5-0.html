class Solution {
  public int minOperations(int[] target, int[] arr) {
    final int N = target.length;
    Map<Integer, Integer> indices = new HashMap<>();
    for (int i = 0; i != target.length; i++)
      indices.put(target[i], i);
    int[] lis = new int[N + 10];
    int len = 0;
    for (int v : arr) {
      Integer idx = indices.get(v);
      if (idx == null)
        continue;
      int j = Arrays.binarySearch(lis, 0, len, idx.intValue());
      if (j >= 0)
        continue;
      j = -1 - j;
      if (j == len)
        len++;

      lis[j] = idx;
    }
    return N - len;
  }
}
