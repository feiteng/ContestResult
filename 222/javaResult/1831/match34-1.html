class Solution {
    private static int MOD = 1_000_000_000 + 7;
    public int waysToSplit(int[] nums) {
        int n = nums.length;
        int[] sum = new int[n+1];
        for (int i = 0; i < n; i++) {
            sum[i+1] = sum[i] + nums[i];
        }
        
        long ans = 0;
        
        for (int i = 1; i < n-1; i++) {
            int x = sum[i];
            if (x <= (sum[n]-x)/2) {
                int lo = find(sum, i+1, 2*x);
                int hi = find(sum, i+1, x+(sum[n]-x)/2+1);
                //System.out.println(x + " " + lo + " " + hi);
                //if (lo > i && hi != n+1) {
                    ans += (long)(hi-lo)-(hi == n+1 ? 1 : 0);
                    ans %= MOD;
                //}
            
}
        }
        return (int)ans;
    }
    private int find(int[] arr, int i, int x) {
        int lo = i, hi = arr.length;
        while (lo < hi) {
            int mid = lo + (hi-lo)/2;
            int val = arr[mid];
            if (val < x) {
                
lo = mid+1;
            } else {
                hi = mid;
            }
        }
        return lo;
    }
}
