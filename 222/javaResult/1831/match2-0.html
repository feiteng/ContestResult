class Solution {
    public int waysToSplit(int[] nums) {
        int N = nums.length;
        int[] preSum = new int[N+1];
        for (int i = 0; i < N; i++) {
            preSum[i+1] = preSum[i] + nums[i];
        }
        
        int res = 0, sum = preSum[N];
        for (int le = 1; le <= N-2; le++) {
            int lSum = preSum[le];
            if (lSum > sum/3) return res;
            
            int rle = le+2, rri = N;
            int n = sum - lSum;
            while (rle < rri) {
                int m = (rle+rri) >> 1;
                int mSum = preSum[m] - lSum;
                int rSum = sum - preSum[m];
                if (mSum <= rSum) {
                    rle = m+1;    
                } else {
                    rri = m;
                }
            }
                        
            
            int lle = le+1, lri = N;
            while (lle < lri) {
                int m = (lle+lri) >> 1;
                int mSum = preSum[m] - lSum;
                if (mSum >= lSum) {
                    lri = m;
                } else {
                    lle = m+1;
                }
            }
            
            
            // System.out.println(rle+","+lle);
            res = (res + (rle-lle)) % 1000000007;
            // System.out.println(res);
            // if (preSum[rle-1]-lSum <= sum - preSum[rle-1]) res++;
        }
        return res;
    }
}
