class Solution {
    public int waysToSplit(int[] nums) {
        int n = nums.length;
        int total = 0;
        int[] prefixSum = new int[n];
        for (int i = 0; i < n; i++) {
            total += nums[i];
            prefixSum[i] = total;
        }
        long res = 0;

        int sum = 0;
        for (int i = 0; i < n - 1; i++) {
            sum += nums[i];
            int rightSum = total - sum;
            if (sum - rightSum <= sum / 2) {
                int leftBound = findLeftMost(prefixSum, sum - rightSum);
                int rightBound = findRightMost(prefixSum, sum / 2);
                if (leftBound != -1 && rightBound != -1) {
                    res = (res + Math.min(rightBound, i - 1) - leftBound + 1) % 1000000007;                    
                }
            }
        }
        return (int)res;
    }
    
    private int findLeftMost(int[] arr, int target) {
        int n = arr.length;
        int left = 0;
        int right = n - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] >= target) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        if (arr[left] >= target) {
            return left;
        }
        return -1;
    }
    
    private int findRightMost(int[] arr, int target) {
        int n = arr.length;
        int left = 0;
        int right = n - 1;
        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (arr[mid] > target) {
                right = mid - 1;
            } else {
                left = mid;
            }
        }
        if (arr[right] <= target) {
            return right;
        }
        if (arr[left] <= target) {
            return left;
        }
        return -1;
    }
}
