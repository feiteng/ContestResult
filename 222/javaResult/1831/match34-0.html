class Solution {
    
    private static int MOD = 1_000_000_007;
    
    public int waysToSplit(int[] nums) {
        int n = nums.length;
        int[] pre = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            pre[i] = pre[i - 1] + nums[i - 1]; 
        }
        // System.out.println(Arrays.toString(pre));
        long cnt = 0;
        for (int i = 1; i + 2 <= n; i++) {
            int first = pre[i], rest = pre[n] - first;
            // System.out.println("first:" + first);
            int smaller = helper1(pre, first + first, i + 1);
            int larger = helper2(pre, rest / 2 + first, n - 1);
            // System.out.println(smaller + "," + larger);
            if (smaller == -1 || larger == -1 || smaller > larger) {
                continue;
            }
            cnt += larger - smaller + 1;
            cnt %= MOD;
        }
        return (int) cnt % MOD;
    }
    
    // 1, find the smallest idx that is A[idx] is larger or equal to target;
    private int helper1(int[] A, int target, int left) {
        // System.out.println("helper1: " + target);
        int right = A.length - 1, ans = -1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (A[mid] >= target) {
                ans = mid;
                right = mid - 1;
            } else { // A[mid] < target
                left = mid + 1;
            }
        }
        return ans;
    }
    
    // 2, find the largest idx that A[idx] is smaller or equal to target
    private int helper2(int[] A, int target, int right) {
        // System.out.println("helper2: " + target);
        int left = 0, ans = -1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (A[mid] <= target) {
                ans = mid;
                left = mid + 1;
            } else { // A[mid] > target
                right = mid - 1;
            }
        }
        return ans;
    }
    
    
}
