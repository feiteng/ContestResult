class Solution {
    
private int search(int[] nums, int l, int r, int v) {
        while (l < r) {
            int m = l + (r-l)/2;
            if (nums[m] < v) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }
    
    private int search2(int[] nums, int l, int r, int v) {
        while (l < r) {
            int m = l + (r-l+1)/2;
            if (nums[m] <= v) {
                l = m;
            } else {
                r = m-1;
            }
        }
        return l;
    }
    
    public int waysToSplit(int[] nums) {
        //nums[r] >= 2*nums[l]
        //s-r >= r-l
        //
        //2r <= s + l
        //2l <= r <= (s+l)/2
        for (int i = 1; i < nums.length; i++)
            nums[i] += nums[i-1];
        int sum = nums[nums.length-1];
        int res = 0;
        for (int l = 0; l < nums.length-2; l++) {
            int r1 = search(nums, l+1, nums.length-2, nums[l]*2);
            if (nums[r1] < nums[l]*2)
                break;
            int r2 = search2(nums, l+1, nums.length-2, (sum+nums[l])/2);
            if (nums[r2]*2 > sum + nums[l])
                break;
            
if (r2 >= r1) {
                res += r2-r1+1;
                res %= 1000000007;
            }
        }
        return res;
    }
}
