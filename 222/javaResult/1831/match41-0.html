class Solution {
    int mod=(int)1e9+7;
    public int waysToSplit(int[] nums) {
        int n=nums.length;
        int[] pre=new int[n];
        pre[0]=nums[0];
        for(int i=1;i<n;i++){
            pre[i]=pre[i-1]+nums[i];
        }
        long res=0;
        for(int i=0;i<n-2;i++){
            if(pre[i]*3>pre[n-1]) break;
            int mid1=-1;
            int left=i+1,right=n-2;
            while(left<=right){
                int mid=(left+right)/2;
                if(pre[mid]-pre[i]>=pre[i]){
                    mid1=mid;
                    right=mid-1;
                }else{
                    left=mid+1;
                }
            }
            if(mid1==-1) break;
            int mid2=-1;
            left=mid1;
            right=n-2;
            while(left<=right){
                int mid=(left+right)/2;
                if(pre[n-1]-pre[mid]>=pre[mid]-pre[i]){
                    mid2=mid;
                    left=mid+1;
                }else{
                    right=mid-1;
                }
            }
            if(mid2>=mid1){
                res+=mid2-mid1+1;
            }
            
        }
        res%=mod;
       return (int) res; 
    }
}
