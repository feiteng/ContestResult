class Solution {
    public int waysToSplit(int[] nums) {
        long mod = 1000000007;
        int s[] = new int[nums.length];
        s[0] = nums[0];
        for(int i=1;i<nums.length;i++){
            s[i] = s[i-1] + nums[i];
        }
        int th = (s[nums.length-1]+2)/3;
        long c = 0;
        for(int i=0;i<nums.length-2;i++){
            int f = searchf(s, i+1, nums.length-2, ((s[nums.length-1]-s[i])/2) + s[i]);
            int sc = searchs(s,i+1,nums.length-2,2*s[i]);
           // System.out.println(f + " " + sc + " " + s[i]);
            if(f == -1 || sc == -1 || f < sc){
                continue;
            }
            c = (c%mod + (f-sc+1))%mod;
        }
        return (int)(c%mod);
    }
    
    public int searchf(int[] sum, int s, int e, int k){
        while(s < e-1){
            int mid = s + (e-s)/2;
            if(sum[mid] <= k){
                s = mid;
            } else if(sum[mid] > k){
                e = mid;
            }
        }
        return sum[e] <= k ? e : (sum[s] <= k ? s : -1);
    }
    public int searchs(int[] sum, int s, int e, int k){
        while(s < e-1){
            int mid = s + (e-s)/2;
            if(sum[mid] < k){
                s = mid;
            } else if(sum[mid] >= k){
                e = mid;
            }
        }
        return sum[s] >= k ? s : (sum[e] >= k ? e : -1);
    }    
}
