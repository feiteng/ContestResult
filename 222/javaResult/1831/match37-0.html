class Solution {
    public int waysToSplit(int[] nums) {
        int module = 1000000007;
        int len = nums.length;
        int[] prefix = new int[len];
        int sum = 0;
        for (int i = 0; i < len; i++) {
            sum += nums[i];
            prefix[i] = sum;
        }
        long res = 0;
        for (int i = 0; i < len - 2; i++) {
            int first = prefix[i];
            if (sum < first * 3) {
                break;
            };
            int val = (prefix[len - 1] - first) / 2;
            int midStart = find1(first * 2, prefix, i + 1, len - 2);
            int midEnd = find2(first + val, prefix, i + 1, len - 2);
            // System.out.println(midStart + " and " + midEnd);
            if (midEnd == -1 || midStart == -1) {
                continue;
            }
            res = (res + midEnd - midStart + 1) % module;
        }
        return (int) res;
    }
    
    private int find1(int limit, int[] prefix, int left, int right) {
        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (prefix[mid] >= limit) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        if (prefix[left] >= limit) {
            return left;
        }
        if (prefix[right] >= limit) {
            return right;
        }
        return -1;
    }
    
    private int find2(int limit, int[] prefix, int left, int right) {
        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (prefix[mid] <= limit) {
                left = mid;
            } else {
                right = mid - 1;
            }
        }
        if (prefix[right] <= limit) {
            return right;
        }
        if (prefix[left] <= limit) {
            return left;
        }
        return -1;
    }
}
