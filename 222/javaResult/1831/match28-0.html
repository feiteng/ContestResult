class Solution {
 public int waysToSplit(int[] nums) {
        int[] sum = new int[nums.length];
        sum[0] = nums[0];
        for(int i = 1; i < nums.length; ++i){
            sum[i] = sum[i-1] + nums[i];
        }


        int total = sum[sum.length-1];
        int res = 0;
        for(int i = 0; i < nums.length; ++i){
            if(sum[i] > total/3){
                break;
            }


            int t1 = sum[i]*2;
            int t2 = (total - sum[i])/2;
            
            int pos = firstIndex(sum, i+1,  sum.length-1, sum[i]*2);
            int pos1 = firstIndex(sum, i+1, sum.length -1,   sum[i] + (total-sum[i])/2 + 1);
            
            
            if(pos1 > pos){
                
                res += (pos1-pos);
                if(pos1 == sum.length){
                    res--;
                }
                res = res%1000000007;
                
            }
        }
        
        return res;


    }
    
    
    int firstIndex(int[] sum, int from, int to,  int t){
        while(from <= to){
            int mid = (to + from)/2;
            
            if(sum[mid] < t){
                from = mid+1;
            }else{
                to = mid-1;
            }
        }
        
        return from;
    }
}
