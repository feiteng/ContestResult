class Solution {
  private final static long MOD = 1000L * 1000L * 1000L + 7;
    
public int waysToSplit(int[] nums) {
    int n = nums.length;
    long res = 0;
    int[] pref = new int[1 + n];
    for (int i = 1; i <= n; i++) {
      pref[i] = pref[i - 1] + nums[i - 1];
    }
    
for (int l = 0; l < n - 1; l++) {
      int low = l + 1;
      int high = n - 1;
      while (low < high) {
        int mid = (low + high) / 2;
        // [l + 1, mid] [mid + 1, n - 1]
        int s1 = pref[mid + 1] - pref[l + 1];
        int s2 = pref[n] - pref[mid + 1];
        if (s1 <= s2) {
          low = mid + 1;
        } else {
          high = mid;
        }
      }

      int critical = low - 1;
      // [0, l] [l + 1, c]
      if (pref[critical + 1] - pref[l + 1] < pref[l + 1]) {
        continue;
      }

      // First True.
      low = l + 1;
      high = critical + 1;
      int s1 = pref[l + 1];
      
while (low < high) {
        int mid = (low + high) / 2;
        int s2 = pref[mid + 1] - pref[l + 1];
        if (s2 < s1) {
          low = mid + 1;
        } else {
          high = mid;
        }
      }

      int firstTrue = low;
      low = critical;
      high = n - 1;
      while (low < high) {
        int mid = (low + high) / 2;
        // [l + 1, mid] [mid + 1, n - 1]
        int s2 = pref[mid + 1] - pref[l + 1];
        int s3 = pref[n] - pref[mid + 1];
        if (s2 <= s3) {
          low = mid + 1;
        } else {
          high = mid;
        }
      }
      int lastTrue = low - 1;
      res += lastTrue - firstTrue + 1;
    }
    return (int) (res % MOD);
  }
}
