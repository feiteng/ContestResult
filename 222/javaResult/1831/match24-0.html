class Solution {
    public int waysToSplit(int[] nums) {
        long ans = 0;
        int m = (int)1e9 + 7;
        int n = nums.length;
        int[] sums = new int[n+1];
        for (int i = 0; i < n; i++) sums[i+1] = sums[i]+nums[i];
        for (int i = 1; i < n-1; i++) {
            int r = sums[n] - sums[i+1];
            int floor = sums[i+1]/2;
            if (floor > r) break;
            int ceil = sums[i+1] - r;
            // System.out.println(ceil+"---"+floor);
            int l_idx = findMin(sums, 0, i, ceil);
            int r_idx = findMax(sums, 0, i, floor);
            if (r_idx == 0) continue;
            // System.out.println(i+"---"+(l_idx == 0 ? r_idx : r_idx - l_idx + 1));
            ans += l_idx == 0 ? r_idx : r_idx - l_idx + 1;
            ans %= m;
        }
        return (int)ans;
    }
    
    public int findMin(int[] sums, int l, int r, int t) {
        while (l+1 < r) {
            int m = l + (r-l)/2;
            if (sums[m] >= t) {
                r = m;
            } else {
                l = m;
            }
        }
        return sums[l] >= t ? l : r;
    }
    
    public int findMax(int[] sums, int l, int r, int t) {
        while (l+1 < r) {
            int m = l + (r-l)/2;
            if (sums[m] > t) {
                r = m;
            } else {
                l = m;
            }
        }
        return sums[r] <= t ? r : l;
    }
}
