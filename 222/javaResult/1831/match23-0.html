class Solution {
    public int waysToSplit(int[] nums) {
        int len = nums.length;
        int[] sum = new int[len + 1];
        sum[0] = 0;
        for (int i = 0; i < len; i ++) {
            sum[i + 1] = sum[i] + nums[i];
        }
        int r = 0;
        int m = 1000000007;
        for (int i = 1; i <= len - 2; i ++) {
            int leftsum = sum[i];
            int rightsum = sum[len] - leftsum;
            int start = i + 1;
            int end = len;
            // find loc of leftsum and rightsum / 2
            while(start < end) {
                int mid = (start + end) / 2;
                if (sum[mid] - leftsum >= leftsum) {
                    end = mid;
                } else {
                    start = mid + 1;
                }
            }
//            System.out.println("Left:" + start);
            int m1 = start;
            start = i + 1;
            end = len;
            // find loc of leftsum and rightsum / 2
            while(start < end) {
                int mid = (start + end) / 2;
//                System.out.println("start " + start + " middle: " + mid + "end " + end);
//                System.out.println("Left " + leftsum + " Middle: " + (sum[mid] - leftsum) + "RIght:"  + (sum[len] - sum[mid]) );
                if (sum[len] - sum[mid] >= sum[mid] - leftsum) {
                    start = mid + 1;
                } else {
                    end = mid;
                }
            }
//            System.out.println("Right" + start);
            r += Math.max(0, (start - m1));
            r %= m;
  //          System.out.println("R:" + r);
        }
//        System.out.println("Final" + r);
        return r;
    }
}
