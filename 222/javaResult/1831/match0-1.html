class Solution {
    public int waysToSplit(int[] nums) {
        int mod = 1_000_000_007;
        
int n = nums.length;
        int[] pre = new int[n + 1];
        
        // System.out.println(n);
        
        for (int i = 1; i <= n; i++) {
            pre[i] = pre[i - 1] + nums[i - 1];
        }
        
        int tail = 0;
        long ret = 0;
        for (int j = n - 1; j >= 2; j--) {
            tail += nums[j];
            
            
int l = 1, r = j;
            while (l < r) {
                int mid = (l + r) / 2;
                int left = pre[mid];
                int right = pre[j] - pre[mid];
                
                if (left > right) {
                    r = mid;
                }
                else {
                    l = mid + 1;
                }
            }
            
            int R = l - 1;
            if (pre[j] - pre[l] > tail) continue;
            
            l = 1;
            
while (l < r) {
                int mid = (l + r) / 2;
                int val = pre[j] - pre[mid];
                
                if (tail >= val) {
                    r = mid;
                }
                else {
                    l = mid + 1;
                }
            }
            
            // System.out.println(l + ", " + R + ", " + j);
            
            ret += (R - l + 1);
            ret = ret % mod;
        }
        
        return (int)ret;
    }
}
