class Solution {
    public int waysToSplit(int[] nums) {
        int n = nums.length, mod = 1_000_000_007;
        int[] pre = new int[n+1], bPre = new int[n+1];
        for (int i = 0; i < n; i++) {
            pre[i+1] = pre[i] + nums[i];
            int ri = n - i - 1;
            bPre[ri] = bPre[ri+1] + nums[ri];
        }
        long cnt = 0;
        for (int le = 1; le < n - 1; le++) {
            int me1 = floor(pre, le + 1, 2 * pre[le]);
            if (me1 > -1) {
                int me2 = ceiling(pre, (pre[n] + pre[le]) / 2);
                if (me1 <= me2) {
                    cnt = (cnt + me2 - me1 + 1) % mod;
                }
            }
        }
        return (int) (cnt % mod);
    }
        
    private long comb(int n) {
        if (n == 1) return 1L;
        long nl = (long) n;
        return (nl - 1) * nl / 2;
    }
    
    private int ceiling(int[] nums, int target) {
        int s = 0, e = nums.length - 2, res = -1;
        while (s <= e) {
            int m = s + (e - s) / 2;
            if (nums[m] <= target) {
                res = m;
                s = m + 1;
            } else {
                e = m - 1;
            }
        }
        return res;
    }
    
    private int floor(int[] nums, int from, int target) {
        int s = from, e = nums.length - 1, res = -1;
        while (s <= e) {
            int m = s + (e - s) / 2;
            if (nums[m] < target) {
                s = m + 1;
            } else {
                res = m;
                e = m - 1;
            }
        }
        return res;
    }
}
