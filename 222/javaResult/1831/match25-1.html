class Solution {
    public int waysToSplit(int[] nums) {
        // left <= mid
        // mid <= right
        int mod = 1000000007;
        
        int N = nums.length;
        int[] pre = new int[N];
        
        int total = 0;
        
        for (int i = 0; i < N; i++) {
            total += nums[i];
            
pre[i] = total;
        }
        
        int res = 0;
        
        for (int i = 0; i + 2 < N; i++) {            
            // Find leftmost index where sum of mid section >= left section.
            // Find rightmost index where sum of right section >= mid section.
            // All indices from leftmost to rightmost are a split.
            int L = leftMost(pre, i + 1, N - 2, pre[i]);
            int R = rightMost(pre, i + 1, N - 2);
            
            if (L != -1 && R != -1) {
                if (L <= R) {
                    res += (R - L + 1);
                    res %= mod;
                
}
            }
        }
        
        return res;
    }
    
    public int leftMost(int[] pre, int L, 
int H, int leftSum) {
        int res = pre.length;
        int lo = L;
        int hi = H;
                
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            
            int midSum = pre[mid] - pre[L-1];
            
            if (midSum >= leftSum) {
                res = Math.min(res, mid);
                
hi = mid - 1;
            }
            else {
                lo = mid + 1;
            }
        }
        
        return res == pre.length ? -1 : res;
    
}
    
    public int rightMost(int[] pre, int L, int H) {
        int res = -1;
        int lo = L;
        int hi = H;
        
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            
            int rightSum = pre[pre.length-1] - pre[mid];
            int midSum = pre[mid] - pre[L-1];
            
            if (rightSum >= midSum) {
                res = Math.max(res, mid);
                lo = mid +  1;
            }
            else {
                
hi = mid - 1;
            }            
        }
        
        return res;
    }
}
