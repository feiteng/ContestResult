class Solution {
	final int mod = (int) 1e9 + 7;

	public 
int waysToSplit(int[] nums) {
		int n = nums.length;
		int[] sum = new int[n];
		sum[0] = nums[0];
		for (int i = 1; i < n; i++) {
			sum[i] = sum[i - 1] + nums[i];
		}
		long ans = 0;
		for (int left = 0; left < n - 2; left++) {
			int midLeft = lower_bound(sum, 0, n, 2 * sum[left]);
			midLeft = Math.max(left + 1, midLeft);
			int L = midLeft, R = n;
			while (L < R) {

				int M = L + R >> 1;
				if (sum[M] - sum[left] <= sum[n - 1] - sum[M]) {
					L = M + 1;
				} else {
					R = M;
				}
			}
			L--;
			L = Math.min(L, n - 2);

			if (L >= midLeft) {
				ans += L - midLeft + 1;
				ans %= mod;
			}
		}
		return (int) ans;
	}

	private int lower_bound(int[] arr, int l, int r, int target) {
		while (l < r) {
			int mid = l + r >> 1;
			if (arr[mid] >= target) {
				r = mid;
			} else {
				l = mid + 1;
			}
		}
		return l;
	}

	private int search(int[] sum, int index, int n) {
		int l = index;
		int r = n + 1;
		int ans = n + 1;
		while (l <= r) {
			int mid = l + r >> 1;
			if (sum[mid] - sum[index - 1] <= sum[n] - sum[mid]) {
				l = mid + 1;
				ans = mid;
			} else {
				r = mid - 1;
			}
		}
		return ans;
	}
}
