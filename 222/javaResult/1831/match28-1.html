class Solution {
    public int waysToSplit(int[] nums) {
        for (int i = 1; i < nums.length; i++) {
            nums[i] += nums[i - 1];
        
}
        int sum = nums[nums.length - 1];
        int ans = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            if (nums[i] * 2 > sum - nums[i]) {
                break;
            }
            int left = bs(nums, nums[i] * 2, i + 1, nums.length - 1);
            if (left >= nums.length - 1 || nums[left] - nums[i] > sum - nums[left]) {
                continue;
            }

            int right = bs(nums, (sum - nums[i]) / 2 + nums[i] + 1, left + 1, nums.length - 1);
            if (left <= right) {
                
ans = (ans - left + right) % 1_000_000_007;
            }
        }
        return ans;
    }

    private int bs(int[] nums, int v, int start, int end) {
        while (start < end) {
            int m = (start + end) / 2;
            if (nums[m] < v) {
                start = m + 1;
            } else {
                end = m;
            }
        }
        return start;
    }
}
