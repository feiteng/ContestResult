class Solution {
    
public int waysToSplit(int[] nums) {
        int length = nums.length;
        int[] fn = new int[length];
        fn[0] = nums[0];
        for (int idx=1;idx<length;idx++) {
            fn[idx] = fn[idx-1] + nums[idx];
        }
        int ans = 0;
        for (int idx=0;idx<length-2;idx++) {
            
int left = fn[idx];
            int l = idx + 1;
            int r = length - 2;
            while (l < r) {
                int mid = (l+r) /2;
                if (fn[mid]-fn[idx] >= left) {
                    r = mid;
                } else {
                    l = mid+1;
                }
            }
            if (fn[l] - fn[idx] < left) {
                break;
            }
            int valueMid = l;
            l = valueMid+1;
            r = length-1;
            
while (l < r) {
                int mid = (l+r+1) / 2;
                if (fn[length-1] - fn[mid-1] >= fn[mid-1] - fn[idx]) {
                    l = mid;
                } else {
                    
r = mid-1;
                }
            }
            if (fn[length-1] - fn[l-1] < fn[l-1] - fn[idx]) {
                continue;
            }
            ans = (ans + (l - valueMid)) % 1000000007;
        }
        return ans;
    }
}
