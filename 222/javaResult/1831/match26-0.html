class Solution {
    public int waysToSplit(int[] nums) {
        long res = 0;
        int n = nums.length;
        long[] sum = new long[n];
        int mod = 1000000007;
        sum[0] = nums[0];
        for (int i = 1; i < n; i++) {
            sum[i] = sum[i-1] + nums[i];
        }
        long total = sum[n-1];
        
        for (int i = 0; i < n; i++) {
            //sum[parse1] - sum[i] = sum[i]
            int parse1 = leftMost(sum, 2 * sum[i]);
            if (sum[i] == 0) {
                parse1 = i+1;
            }
            //sum[parse2] - sum[i] = total - sum[parse2]
            int parse2 = rightMost(sum, (sum[i] + total)/2);
            if (total == 0) {
                parse2 = n-2;
            }
            if (i < parse1 && parse1 <= parse2) {
                res += parse2-parse1+1;
                res %= mod;                
            }

            
        }
        return (int)res;
    }
    
    private int leftMost(long[]sum, long target) {
        int low = 0;
        int high = sum.length;
        while(low < high) {
            int mid = (low + high) /2;
            if (target > sum[mid]) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return low;
    }
    
    private int rightMost(long[]sum, long target) {
        int low = 0;
        int high = sum.length;
        while(low < high) {
            int mid = (low + high) /2;
            if (target >= sum[mid]) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return low-1;
    }}
