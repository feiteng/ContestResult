class Solution {
     int[] preSum;

    public int waysToSplit(int[] nums) {
        preSum = new int[nums.length + 1];
        for (int i = 1; i < preSum.length; i++) {
            preSum[i] = preSum[i - 1] + nums[i - 1];
        }

        long res = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            int lSum = preSum[i + 1];
            int l = search1(i + 1, nums.length, lSum);
            int r = search2(i + 1, nums.length);
            if(r > l){
                res = (res + (r - l)) % 1000000007;
            }
        }
        return (int) res;
    }


    int search1(int l, int r, int leftVal) {
        int left = l;
        int right = r;
        while (left < right) {
            int mid = (left + right) / 2;
            int midVal = preSum[mid + 1] - preSum[l];
            if (midVal >= leftVal) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    int search2(int l, int r) {
        int left = l;
        int right = r;
        while (left < right) {
            int mid = (left + right) / 2;
            int midVal = preSum[mid] - preSum[l];
            int rVal = preSum[r] - preSum[mid];
            if (rVal < midVal) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left - 1;
    }

}
