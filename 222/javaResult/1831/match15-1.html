class Solution {
    
public int waysToSplit(int[] nums) {
        int n = nums.length;
        int[] pref = new int[n+1];
        for(int i = 1; i <= n; i ++){
            pref[i] = pref[i-1]+nums[i-1];
        }
//        [0,i] -> [i+1,j] -> [j+1,n-1]
        int mod = (int)(1e9+7);
        long ret = 0;
        for(int i = 0; i < n-2; i ++){
            int left = i+1;
            int right = n-2;
            int sumL = pref[i+1]-pref[0];
            int sum = pref[n]-pref[i+1];
            if(sumL > sum) break;
            
while(left <= right){
                int mid = (left+right) >>> 1;
                int cur = pref[mid+1]-pref[i+1];
                if(cur >= sumL){
                    right = mid-1;
                }else{
                    left = mid+1;
                }
            }
            int resL = left;
            left = i+1;
            right = n-2;
            while(left <= right){
                int mid = (left+right) >>> 1;
                int cur = pref[mid+1]-pref[i+1];
                
if(cur <= sum-cur){
                    left = mid+1;
                }else{
                    right = mid-1;
                }
            }
            int resR = right;
            if(resR < resL) continue;
            ret = (ret+ (resR-resL+1))%mod;
        }
        return (int)ret;
    }
}
