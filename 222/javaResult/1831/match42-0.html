class Solution {
    public int waysToSplit(int[] nums) {
        long res = 0;
        int mod = 1000000007;
        
        int len = nums.length;
        
        int[] preSum = new int[len];
        
        preSum[0] = nums[0];
        
        for (int i = 1; i < len; i++) {
            preSum[i] = preSum[i - 1] + nums[i];
        }
        
        for (int i = 0; i < len - 2; i++) {
            int minIndex = binSearch(preSum, i + 1, len - 1, 2 * preSum[i] - 1);
            if (minIndex < len - 1) {
                int target = (preSum[i] + preSum[len - 1]) / 2;
                int maxIndex = binSearch(preSum, i + 1, len - 1, target);
                if (maxIndex > minIndex) {
                    res += maxIndex - minIndex;
                    // System.out.println(minIndex + ", " + maxIndex);
                    res %= mod;
                }
            }
        }
        
        return (int) res;
    }
    
    private int binSearch(int[] nums, int l, int r, int target) {
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (target < nums[mid]) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        
        return l;
    }
}
