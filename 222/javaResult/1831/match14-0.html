class Solution {
    int MOD= (int) 1e9 + 7;
    public int waysToSplit(int[] nums) {
        int n = nums.length;
        int[] preSum = new int[n +1];
        for (int i = 1; i <= n; i++) {
            preSum[i] = preSum[i - 1] + nums[i - 1];
        }
        int res = 0;
        for (int i = 0; i< n - 2; i++) {
            int left = preSum[i + 1];
            // for (int j = i + 1; j < n - 1; j++) {
            //     int mid = preSum[j + 1] - preSum[i + 1];
            //     int right = preSum[n] - preSum[j + 1];
            //     if (left <= mid && mid <= right) {
            //         res= (res + 1) % MOD;
            //     }
            // }
            int s = i + 1;
            int e = n - 2;
            while (s <=e) {
                int mid = s + (e - s) /2;
                int m = preSum[mid + 1] - preSum[i + 1];
                //int right = preSum[n] - preSum[mid + 1];
                // if (left <= m && m <= right) {
                //     s = mid + 1;
                // } else {
                //     e= mid - 1;
                // }
                // if (left > m) {
                //     s = mid + 1;
                // } else if (m <= right) {
                //     s = mid + 1;
                // } else {
                //     e = mid - 1;
                // }
                // if (left > m || m <= right) {
                //     s = mid + 1;
                // } else {
                //     e = mid - 1;
                // }
                if (left > m) {
                    s = mid + 1;
                } else {
                    e= mid - 1;
                }
            }
            int leftBound = s;
            s = i + 1;
            e = n - 2;
             while (s <=e) {
                int mid = s + (e - s) /2;
                int m = preSum[mid + 1] - preSum[i + 1];
                int right = preSum[n] - preSum[mid + 1];
                 if (m <= right) {
                     s = mid + 1;
                 } else {
                     e = mid - 1;
                 }
             }
            int rBound = e;
            if (rBound >= leftBound) {
            // if (left <= preSum[e + 1] - preSum[i + 1] && preSum[e + 1] - preSum[i + 1] <= preSum[n] - preSum[e + 1]) {
            //     res = (res + (e - i)) % MOD;
            // }
            // System.out.println(e);
            // if (e > i && e < n - 1) {
                res = (res + (rBound - leftBound + 1)) % MOD;
            }
        }
        return res;
    }
}
