class Solution {
    public int waysToSplit(int[] nums) {
        for (int i = 1; i < nums.length; i++) {
            nums[i] += nums[i - 1];
        }
        int res = 0;
        int mod = (int) 1e9 + 7;
        int total = nums[nums.length - 1];
        for (int i = 0; i < nums.length - 2; i++) {
            if (nums[i] <= total / 3) {
                int pos1 = binarySearch1(nums, 2 * nums[i]);
                
int pos2 = binarySearch2(nums, (nums[i] + total) / 2);
                //System.out.println(pos1 + " " + pos2);
                if (pos1 <= i) {
                    pos1 = i + 1;
                }
                if (pos2 == nums.length - 1) {
                    pos2 = nums.length - 2;
                }
                if (pos2 >= pos1) {
                    res += pos2 - pos1 + 1;
                    res %= mod;
                }
            
}
        }
        return res;
    }
    
    public int binarySearch1(int[] nums, int taget) {
        int i = 0, j = nums.length - 1;
        while (i < j) {
            int mid = (i + j) / 2;
            if (nums[mid] >= taget) {
                j = mid;
            } else {
                i = mid + 1;
            }
        }
        return i;
    }
    
    public int binarySearch2(int[] nums, int taget) {
        int i = 0, j = nums.length - 1;
        while (i < j) {
            int mid = (i + j + 1) / 2;
            if (nums[mid] <= taget) {
                i = mid;
            } else {
                j = mid - 1;
            }
        }
        return i;
    }
}
