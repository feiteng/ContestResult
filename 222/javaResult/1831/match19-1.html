class Solution {
    long mod = (long)(1e9 + 7);

    public int waysToSplit(int[] nums) {    
        int n = nums.length;
        long count = 0;
        
        int[] prefix = new int[n];
        int sum = 0;
        
for (int i = 0; i < n; i++) {
            sum += nums[i];
            prefix[i] = sum;
        }
        
        int upper = sum / 3;
        for (int i = 0; i < prefix.length - 2 && prefix[i] <= upper; i++) {
            int j = find(prefix, i + 1, prefix[i]);
            
            if (j == -1) continue;
            
            int k = findMax(prefix, j, prefix[i], sum);
            
            if (k == -1) continue
;
            
            // System.out.println(i + " " + j + " " + k);
            
            count += (long)((long)k - (long)j + 1l);
            count %= mod;
        }
        
        
        return (int)(count % mod);
    }
    
    private int findMax(int[] prefix, int i, int leftSum, int sum) {
        if (i >= prefix.length) {
            return -1;
        }

        int j = prefix.length - 2;
        
        while (i + 1 < j) {
            int mid = i + (j - i) / 2;
            int midSum = prefix[mid] - leftSum;
            
int rightSum = sum - leftSum - midSum;
            
            if (midSum <= rightSum) {
                i = mid;
            } else {
                j = mid;
            }
        }
        
        int midSum = prefix[j] - leftSum;
        int rightSum = sum - leftSum - midSum;
        if (rightSum >= midSum) {
            return j;
        }
        
        midSum = prefix[i] - leftSum;
        rightSum = sum - leftSum - midSum;
        
if (rightSum >= midSum) {
            return i;
        }
        
        return -1;
    }
    
    private int find(
int[] prefix, int left, int target) {
        if (left >= prefix.length) {
            return -1;
        }
        
        int preSum = prefix[left - 1];
        int right = prefix.length - 1;
        
        while (left + 1 < right) {
            int middle = left + (right - left) / 2;
            
int sum = prefix[middle] - preSum;
            
            if (sum >= target) {
                right = middle;
            } else {
                left = middle;
            }
        }
        
        if (prefix[left] - preSum >= target) {
            return left;
        }
        
        if (prefix[right] - preSum >= target) {
            return right;
        }
        
        return -1;
    }
}
