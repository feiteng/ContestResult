class Solution {
    public int waysToSplit(int[] nums) {
        int len  = nums.length;
        long s[] = new long[len+1];
        
for(int i=0;i<len;++i){
            s[i+1] =  s[i] + nums[i];
        }
        long mod = 1000000007L;
        long a=  0;
        for(int j=len-1;j>=2;--j){
            long rt = s[len] - s[j];
            long tot = s[j];
            
int lo = 1; int hi = j-1;
            while(lo<hi){
                int mi =  (lo+hi+1)/2;
                if(s[mi]>tot-s[mi]){
                    hi = mi-1;
                }else{
                    lo = mi;   
                }
            }
            
            
int lo1 = 1; int hi1 = j-1;
            while(lo1<hi1){
                int mi =  (lo1+hi1)/2;
                if(tot-s[mi]>rt){
                    lo1 = mi+1;
                }else{
                    hi1 = mi;   
                }
            }
            
if(lo>=lo1 &&  s[lo] <= tot - s[lo] && tot  - s[lo1] <= rt  ){
                a += lo - lo1 +1;a %= mod;
             //   System.out.println(rt);
             //   System.out.println(lo + "  "+ lo1);
            }
                
            
        }
        return (int)a;
        
    }
}
