class Solution {
    public int waysToSplit(int[] nums) {
        int mod = (int)1e9+7;
        int n = nums.length;
        long[] sum = new long[n];
        
        sum[0] = nums[0];
        for (int i = 1; i < n; i++) {
            sum[i] = sum[i-1] + nums[i];
        }
        
        long ans = 0;
        long total = sum[n-1];
        
        for (int i = 0; i < n; i++) {
            //sum[i] = sum[j] - sum[i] = total - sum[j]
            
            //sum[j1] - sum[i] = sum[i]
            //int j1 = Arrays.binarySearch(sum, 2*sum[i]);
            // if (j1 < 0) {
            //     j1 = -j1-1;
            // }
            int j1 = leftMost(sum, 2*sum[i]);
            if (sum[i] ==0) {
                j1 = i+1;
            }
            
            //sum[j2] - sum[i] = total - sum[j2]
            // int j2 = Arrays.binarySearch(sum, (sum[i] + total)/2);
            // if (j2 < 0) {
            //     j2 = -j2-1-1;
            // }
            int j2 = rightMost(sum, (sum[i] + total)/2);
            if (total == 0) {
                j2 = n-2;
            }
            
            //System.out.println(i + " " + j1 + " " + j2);
            
            if (i < j1 && j1 <= j2) {
                ans += j2-j1+1;
                ans %= mod;
            }
        }
        return (int)ans;
    }
    
    private int leftMost(long[] sum, long target) {
        int low = 0;
        int high = sum.length;
        
        while(low < high) {
            int mid = (low + high)/2;
            if (target > sum[mid]) {
                low = mid+1;
            } else {
                high = mid;
            }
        }
        return low;
    }
    
    private int rightMost(long[] sum, long target) {
        int low = 0;
        int high = sum.length;
        
        while(low < high) {
            int mid = (low + high)/2;
            if (target >= sum[mid]) {
                low = mid+1;
            } else {
                high = mid;
            }
        }
        return low-1;
    }
}
