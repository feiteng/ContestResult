class Solution:
    def countPairs(self, deli: List[int]) -> int:
        c = collections.Counter(deli)
        power = [1 << i for i in range(22)]
        res = 0
        
        for key in sorted(c.keys()):
            for p in power: 
                if p > key and p-key in c and p-key >= key:
                    if key == p - key: 
                        res += (c[key] * (c[key] - 1) // 2)
                    else:     
                        res += c[key] * c[p-key]
                        
                    
        
        return res % (10**9 + 7)
