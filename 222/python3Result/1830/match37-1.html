class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:  
        d = Counter()
        rv = 0
        for x in deliciousness:
            
for i in range(23):
                b = 1 << i
                y = b-x 
                rv += d[y] 
            d[x] += 1
        return rv % (10**9 +7)
