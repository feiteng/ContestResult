class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        mod = 10 ** 9 + 7
        answers = []
        i = 2
        for ii in range(41):
            answers.append(i ** ii)
        his = defaultdict(int)
        out = 0
        for d in deliciousness:
            for a in answers:
                dif = a - d
                if dif in his:
                    out += his[dif]
            his[d] += 1

        return out % mod
