class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        from collections import Counter
        
        M = 10 ** 9 + 7
        d = Counter(deliciousness)
        
        res = 0
        for x in d:
            for i in range(22):
                if 2 ** i - x >= x and 2 ** i - x in d:
                    if 2 ** i - x == x:
                        res += d[x] * (d[x] - 1) // 2 % M
                    else:
                        res += d[x] * d[2 ** i - x] % M
        
        return res % M
