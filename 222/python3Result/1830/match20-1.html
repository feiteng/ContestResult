from collections import Counter

class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        
        modulus = 10**9+7
        pows = []
        val = 1
        for i in range(22):
            pows.append(val)
            val *
= 2
        visited = Counter()
        result = 0
        for delic in deliciousness:
            for pow2 in pows:
                if pow2 < delic:
                    continue
                if (pow2-delic) in visited:
                    result += visited[pow2-delic]
            result %= modulus
            visited[delic] += 1
        
        return result
