from collections import Counter

class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        deli_cnts = Counter(deliciousness)
        candidates = [2 ** i for i in range(22)]
        result = 0
        mod = 10 ** 9 + 7
        
for d in deli_cnts:
            for cand in candidates:
                other = cand - d
                if other in deli_cnts:
                    if other == d:
                        result += deli_cnts[d] * (deli_cnts[d] - 1) // 2
                    elif other < d:
                        result += deli_cnts[d] * deli_cnts[other]
                    result %= mod
        return result
