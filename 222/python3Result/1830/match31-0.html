class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        MOD = 10**9 + 7
        cnts = collections.Counter(deliciousness)
        keys = sorted(list(cnts.keys()))
        ans = 0
        sums = [2 ** i for i in range(22)]
        for s in sums:
            for key in keys:
                if key <= s - key:
                    if key != s - key:
                        ans += cnts[key] * cnts[s - key]
                    else:
                        ans += cnts[key] * (cnts[key] - 1) // 2
                    # print(key, s-key, ans)
        return ans%MOD
