class Solution:
    def countPairs(self, d: List[int]) -> int:
        mask=[1]
        mod=10**9+7
        for i in range(1,22):
            mask.append(mask[-1]*2)
        c=Counter()
        ans=0
        for x in d:
            for m in mask:
                ans+=c[m-x]
            c[x]+=1
        return ans%mod
