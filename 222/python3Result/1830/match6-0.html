class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        counter = collections.Counter(deliciousness)
        allSum = [2**k for k in range(0, 22)]
        result = 0
        modulo = 10**9 + 7
        for key in counter:
            for sumVal in allSum:
                target= sumVal - key
                if target >= key and target in counter:
                    if target == key:
                        result = result + (counter[key]*(counter[key]-1)//2)%modulo
                    else:
                        result = result + (counter[key] * counter[target])%modulo
                    result = result % modulo
        return result
