class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        s = collections.Counter()
        can = [2**i for i in range(22)]
        ans = 0
        for d in deliciousness:
            for c in can:
                if c - d in s:
                    ans += s[c - d]
            s[d] += 1
        return ans % (10**9 + 7)
