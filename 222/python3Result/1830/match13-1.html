class Solution:
    def countPairs(self, deliciousness) -> int:
        mod = 10 ** 9 + 7
        P2 = [2**i for i in range(22)]
        d = collections.defaultdict
(int)
        ans = 0
        for a in deliciousness:
            for p in P2:
                if a <= p:
                    ans += d[p-a]
            d[a] += 1
        return ans % mod
