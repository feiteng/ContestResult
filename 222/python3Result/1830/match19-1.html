from collections import Counter
class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        cnt = dict(sorted(dict(Counter(deliciousness)).items(),key=lambda x:x[0]))
        res = 0
        mode =int(1e9+7)
        
for key in cnt:
            for exp in range(0,41):
                target = 2**exp
                remaid = target- key
                if remaid< key:
                    continue
                if remaid in cnt:
                    if remaid == key:
                        res=(res+cnt[remaid]*(cnt[remaid]-1)//2)%mode
                    else:
                        res=(res+cnt[remaid]*cnt[key])%mode
        return res
