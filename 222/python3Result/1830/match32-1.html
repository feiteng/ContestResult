class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        aaa = Counter(deliciousness)
        bbb = sorted(aaa.keys())
        l = len(bbb)
        res = 0
        erdemi = list([2**i for i in range(41)])
        
for i in bbb:
            for j in erdemi:
                if j-i in aaa:
                    if j-i>i:
                        #print(i," ",j-i)
                        res+=aaa[j-i]*aaa[i]
                        #print(res)
                    elif j-i==i:
                        res+=(aaa[i]*(aaa[i]-1))//2
                        #print(res)
        return res % 1000000007
