class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        count = {}
        res = 0
        
        for d in deliciousness:
            for i2 in range(22):
                if 2**i2 - d in count:
                    res += count[2**i2 - d]
                    
            if d in count:
                count[d] += 1
            else:
                count[d] = 1
                
                    
        return res % (10**9 + 7)
