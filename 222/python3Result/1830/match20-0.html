from collections import defaultdict
class Solution:
    def countPairs(self, D: List[int]) -> int:
        MOD = 10 ** 9 + 7
        power2 = [1]
        for i in range(21):
            power2.append(power2[-1] * 2)
        count = defaultdict(int)
        res = 0
        for num in D:
            for p in power2:
                if p < num: continue
                if p - num in count:
                    res += count[p-num]
            res %= MOD
            count[num] += 1
        return res
