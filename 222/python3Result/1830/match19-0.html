class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        cnt=Counter(deliciousness)
        ans=0
        for a in cnt:
            for i in range(0,41):
                total=1<<i
                b=total-a
                if b<a:
                    continue
                if b in cnt:
                    if a==b:
                        ans+=cnt[a]*(cnt[b]-1)//2
                    else:
                        ans+=cnt[a]*cnt[b]
        return ans%(10**9+7)
