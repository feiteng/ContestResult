# Is power of two
# x != 0 and x & (x - 1) == 0


class Solution:
    
def countPairs(self, arr: List[int]) -> int:
        freq = Counter(arr)
        double = 0
        for k in freq:
            for b in range(22):
                comp = (1 << b) - k
                if comp >= 0:
                    if comp < k:
                        double += freq[comp] * freq[k]
                    elif comp == k:
                        double += freq[k] * (freq[k] - 1) // 2
        MOD = 10 ** 9 + 7
        return  double % MOD
