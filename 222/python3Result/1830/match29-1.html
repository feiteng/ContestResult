class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        power = [2**i for i in range(22)]
        v 
= set(deliciousness)
        c = Counter(deliciousness)
        ans = 0
        for x in v:
            for p in power:
                if p-x in v:
                    if p-x == x:
                        ans += (c[x]*(c[x]-1)//2)
                    elif p-x > x:
                        ans += c[x] * c[p-x]
                    ans %= (10**9 + 7)
        return ans%(10**9 + 7)
