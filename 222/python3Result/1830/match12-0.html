class Solution:
    def countPairs(self, D: List[int]) -> int:
        MIS = [2 ** i for i in range(0, 22)]
        dic = collections.defaultdict(int)
        res = 0
        for d in D:
            for mi in MIS:
                if mi - d in dic:
                    res += dic[mi - d]
                    #break
            dic[d] += 1
        return res % int(1e9 + 7)
