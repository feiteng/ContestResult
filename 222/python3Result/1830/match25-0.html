class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        exist = collections.defaultdict(int)
        result = 0
        BIG = int(1e9 + 7)
        for d in deliciousness:
            for i in range(22):
                result += exist[(1 << i) - d]
                result %= BIG
            exist[d] += 1
        return result
