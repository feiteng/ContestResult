class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        count = Counter(deliciousness)
        keys = list(count.keys())
        res1, res2, MOD = 0, 0, 10 ** 9 + 7
        
        for i, ki in enumerate(keys):
            for shift in range(22):
                num = 1 << shift
                if num == 2 * ki:
                    res2 += (count[ki] * (count[ki] - 1)) // 2
                else:
                    res1 += (count[ki] * count[num - ki]) % MOD
        return (res1 // 2 + res2) % MOD
