from collections import defaultdict

class Solution:
    def countPairs(self, deliciousness):
        c = defaultdict(int)
        ans = 0
        for x in deliciousness:
            p = 1
            for i in range(30):
                ans += c[(p << i) - x]
                ans %= (1000000007)
            c[x] += 1
        return ans
