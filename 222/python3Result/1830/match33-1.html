class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        deliciousness.sort()

        max_power = int(math.log(max(deliciousness),2)) + 1
        all_sums = []
        for i in range(max_power+1):
            all_sums.append(2**i)

        deli_dict = collections.defaultdict(int)
        for d in deliciousness:
            deli_dict[d] += 1

        res = 0
        for k in deli_dict.keys():
            
for s in all_sums:
                if s-k in deli_dict:
                    if k != s-k:
                        res += deli_dict[k] * deli_dict[s-k]
                    else:
                        res += deli_dict[k] * (deli_dict[k] - 1)

        return (res // 2) % (10**9 + 7)
