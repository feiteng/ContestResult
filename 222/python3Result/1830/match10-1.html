import collections
class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        res = 0
        memo = collections.defaultdict
(int)
        
        for i in deliciousness:
            for j in range(24):
                if (1<<j)-i in memo:
                    res+=memo[(1<<j)-i]
            memo[i]+=1
        return res%(10**9+7)
