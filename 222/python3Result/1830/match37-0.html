from collections import defaultdict
class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        """
        How many pairs of i, j such that i < j
        d[i] + d[j] is power of 2
        """
        m = defaultdict(int)
        res = 0
        for i in range(len(deliciousness)):
            num = deliciousness[i]
            
            for j in range(31):
                t = 2 ** j
                target = t - num
                res += m[target]
            
            m[num] += 1
            
        return res % (10** 9 + 7)
