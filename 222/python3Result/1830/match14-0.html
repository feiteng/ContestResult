class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        freq = {}
        for x in deliciousness: freq[x] = 1 + freq.get(x, 0)
        
        ans = 0
        for x in freq: 
            for k in range(30): 
                if x <= 2**k - x and 2**k - x in freq: 
                    if x == 2**k - x: ans += freq[x] * (freq[x] - 1)//2
                    else: ans += freq[x] * freq[2**k - x]
        return ans % 1_000_000_007
