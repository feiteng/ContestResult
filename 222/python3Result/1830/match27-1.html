MOD = 10 ** 9 + 7
GOOD = {1 << i for i in range(42)}

class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        h 
= defaultdict(int)
        ans = 0
        
        for x in deliciousness:
            for y in GOOD:
                ans += h[y - x]
            h[x] += 1
        return ans % MOD
