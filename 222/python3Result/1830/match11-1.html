class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        seen = collections.Counter()
        res = 0
        MOD = 10**9+7
        
for d in deliciousness:
            for move in range(0, 22):
                if (1 << move) - d in seen:
                    res += seen[(1 << move) - d]
            seen[d] += 1
        return res % MOD
