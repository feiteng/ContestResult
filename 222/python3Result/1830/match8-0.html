class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        count = defaultdict(lambda : 0)
        ans = 0
        for d in deliciousness:
            for i in range(22):
                ans += count[(1 << i) - d]
            count[d] += 1
        return ans % 1000000007
