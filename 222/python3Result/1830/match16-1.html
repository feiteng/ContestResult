class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        cnt = collections.defaultdict(int)
        res
=0
        for d in deliciousness:
            for power in range(22):
                res += cnt[(1 << power) - d]
            cnt[d] += 1
        return res % (10**9 + 7)
