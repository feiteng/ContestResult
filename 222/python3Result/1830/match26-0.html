class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        st = {}
        targets = []
        t = 1
        for i in range(22):
            targets.append(t)
            t *= 2
        ret = 0
        for i in deliciousness:
            for t in targets:
                if t-i in st:
                    ret += st[t-i]
            if i not in st:
                st[i] = 0
            st[i] += 1
        return ret % (1000_000_000 + 7)
