class Solution:
    def countPairs(self, d: List[int]) -> int:
        c = collections.Counter(d)
        res = 0
        for x in c:
            for i in range(22):
                y = (1 << i) - x
                if y == x:
                    res += c[x] * (c[x] - 1) // 2
                elif y > x and y in c:
                    res += c[x] * c[y]
        return res % (10 ** 9 + 7)
