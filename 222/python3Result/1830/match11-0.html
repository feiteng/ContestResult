class Solution:
    def countPairs(self, d: List[int]) -> int:
        ans=0
        mod=10**9+7
        s=defaultdict(int)
        for i in d:
            for j in range(25):
                if (1<<j)-i in s:
                    ans=(ans+s[(1<<j)-i])%mod
            s[i]+=1
        return ans%mod
