class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        seen = collections.defaultdict(int)
        ans = 0
        MOD = 10 ** 9 + 7
        for d in deliciousness:
            for i in range(22):
                ans += seen[(1 << i) - d]
            seen[d] += 1
            ans %= MOD
        return ans
