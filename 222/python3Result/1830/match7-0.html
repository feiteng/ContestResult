class Solution:
    def countPairs(self, A: List[int]) -> int:
        ans = 0
        cnt = collections.Counter()
        
        Mod = int(1e9) + 7
        for a in A:
            for i in range(31):
                ans = (ans + cnt[(1<<i) - a]) % Mod
            cnt[a] += 1
        
        return ans
