class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        ans = 0
        mod = 10 ** 9 + 7
        cnt = Counter()
        
for a in deliciousness:
            for i in range(22):
                ans = (ans + cnt[(1<<i) - a]) % mod
            cnt[a] += 1
        return ans
