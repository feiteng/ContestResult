class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        MOD = 10 ** 9 + 7
        counter = collections.Counter(deliciousness)
        print
(counter)
        ans = 0
        for key in counter:
            for i in range(0, 22):
                target = 2 ** i - key
                if target == key:
                    ans += counter[target] * (counter[target] - 1) // 2
                elif target > key and target in counter:
                    ans += counter[key] * counter[target]
                ans % MOD
        return ans % MOD
