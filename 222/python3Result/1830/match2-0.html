class Solution:
    MOD = 10**9+7
    def countPairs(self, deliciousness: List[int]) -> int:
        counts = Counter(deliciousness)
        ans = 0
        for key in counts:
            for shift in range(23):
                key2 = (1 << shift) - key
                if key2 in counts and key <= key2:
                    if key2 == key:
                        ans += (counts[key] * (counts[key2] - 1)) // 2
                    else:
                        ans += counts[key] * counts[key2]
                    ans %= Solution.MOD
        return ans
