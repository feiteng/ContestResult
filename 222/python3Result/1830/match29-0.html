class Solution:
    def countPairs(self, A: List[int]) -> int:
        sums = [1 << i for i in range(22)]
        c = Counter(A)
        # print(c)
        res = 0
        # for x in sorted(list(c)):
        for x in c:
            for y in [s - x for s in sums if 0 <= (s - x) <= (1 << 20)]:
                if y in c and y >= x:
                    if x == y:
                        res += c[x] * (c[x] - 1) // 2
                    else:
                        res += c[x] * c[y]
                    # print(x, y, res)
        return res % (10 ** 9 + 7)
