class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        d
=defaultdict(int)
        ans=0
        for v in deliciousness:
            d_sum=1
            for _ in range(41):
                ans+=d[d_sum-v]
                d_sum<<=1
            d[v]+=1
        return ans%(10**9+7)
