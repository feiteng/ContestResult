class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        d=collections.Counter(deliciousness)
        ans=0
        for x in d:
            for i in range(22):
                y=2**i-x
                if y in d:
                    if x!=y:
                        ans+=d[x]*d[y]
                    else:
                        ans+=d[x]*(d[x]-1)
        ans//=2
        return ans%(10**9+7)
