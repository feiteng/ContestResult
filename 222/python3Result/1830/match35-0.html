class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        mod = 10 ** 9 + 7
        s = set()
        s.add(0)
        i = 1
        while i <= 8 ** 20:
            s.add(i)
            i *= 2
        import collections
        c = collections.Counter(deliciousness)
        ans = 0
        for i in c.keys():
            for j in s:
                if j - i in c and j - i >= i and j > 0:
                    if j - i == i:
                        ans += (c[i] * (c[i] - 1)) // 2
                    else:
                        ans += c[j - i] * c[i]
        return ans % mod
