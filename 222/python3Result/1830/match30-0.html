class Solution:
    def countPairs(self, deliciousness: List[int]) -> int:
        cnt = {}
        for k in deliciousness:
            cnt[k] = cnt.get(k, 0) + 1
        unique_vals = [k for k in cnt]
        power_of_two = [pow(2, i) for i in range(23)]
        num_pairs = 0
        mod_num = pow(10, 9) + 7
        for cur_val in unique_vals:
            for power in power_of_two:
                other_val = power - cur_val
                if (cur_val <= other_val) and (other_val in cnt):
                    if cur_val == other_val:
                        num_pairs += cnt[cur_val] * (cnt[cur_val] - 1) // 2
                    else:
                        num_pairs += cnt[cur_val] * cnt[other_val]

        return num_pairs % mod_num
