class Solution:
    def countPairs(self, de: List[int]) -> int:
        d = {}
        choice = [1 << i for i in range(25)]
        ans = 0
        mod = 10 ** 9 + 7
        for f in de:
            for c in choice:
                
if c - f in d:
                    ans = (ans + d[c - f]) % mod
            if f in d:
                d[f] += 1
            else:
                d[f] = 1
        return ans
