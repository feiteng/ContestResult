class Solution:
    def countPairs(self, nums: List[int]) -> int:
        cnt = collections.defaultdict(int)
        ans = 0
        mod = 10**9+7
        for num in nums:
            for i in range(22):
                ans += cnt[2**i-num]
                ans %= mod
            cnt[num] += 1
        return ans
