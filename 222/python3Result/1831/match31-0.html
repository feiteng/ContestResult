class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        presum = [0]
        for n in nums:
            presum.append(presum[-1]+n)
        presum = presum[1:]
        
        out = 0
        r_low, r_high, r = 1,1,1
        # print(presum)
        for l in range(len(nums)-1):
            r_low = max(r_low, l+1)
            while presum[r_low]-presum[l]< presum[l]:
                r_low += 1
                if r_low >=len(presum):
                    return out
            
            r_high=max(r_high, r_low)
                
            while r_high<len(presum)-1 and presum[r_high]-presum[l] <= presum[-1]-presum[r_high]:
                r_high += 1
                # if r_high>=len(presum):
                #     break
            # print(r_high)
            out += r_high-r_low
        return out%int(1e9 + 7)
