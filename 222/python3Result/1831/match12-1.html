class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        mod = 10 ** 9 + 7
        sums = [0]
        for i in nums:
            sums.append(sums[-1] + i)
        third = sums[-1] // 3
        n = len(sums)
        ans = 0
        for i in range(1, n-1):
            if sums[i] > third:
                break
            l1, r1 
= i + 1, n - 2
            while l1 < r1:
                mid = l1 + (r1 - l1) // 2
                if sums[mid] - sums[i] < sums[i]:
                    l1 = mid + 1
                else:
                    r1 = mid
            candidate1 = l1
            l2, r2 
= i + 1, n - 2
            while l2 < r2:
                mid = (r2 + l2 + 1) // 2
                if sums[-1] - sums[mid] < sums[mid] - sums[i]:
                    r2 = mid - 1
                else:
                    l2 = mid
            candidate2 = r2
            #if i == 1:
                #print(candidate1, candidate2)
            if candidate1 <= candidate2:
                ans = (ans + candidate2 - candidate1 + 1) % mod
        return ans
