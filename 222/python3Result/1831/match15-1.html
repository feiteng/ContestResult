class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        #nums = [1,2,2,2,5,0]
        return solve(nums)
    
def solve(nums):
    total = sum(nums)
    lefts 
= []
    for i in range(len(nums)):
        if not lefts:
            lefts.append(nums[i])
        else:
            lefts.append(lefts[-1] + nums[i])
    count = 0
    for i in range(1, len(nums) - 1):
        left = lefts[i-1]
        rest = total - left
        lo_bound = left + left
        hi_bound = rest // 2 + left
        
        lo_ind = max(i, bisect.bisect_left(lefts, lo_bound))
        hi_ind = min(len(nums) - 1, bisect.bisect_right(lefts, hi_bound))
        
        count += max(0, hi_ind - lo_ind)
        #print(i, lo_ind, hi_ind)
    return count % (10 ** 9 + 7)
