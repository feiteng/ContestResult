class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        pref = [0]+nums
        n = len(pref)
        mod,res = 1e9+7,0
        for i in range(1,len(pref)):
            pref[i]=pref[i-1]+nums[i-1]
        print(pref)
        for i in range(1,n):
            if pref
[i]>pref[-1]/3:break
            l,r = i+1,n-1
            while l<r:
                mid = (l+r)//2
                if pref[mid]-pref[i]<pref[i]:
                    l = mid+1
                else:   
                    r = mid
            lb = l
            l,r = i+1,n-1
            while l<r:
                mid = (l+r)//2
                if pref[-1]-pref[mid]<pref[mid]-pref[i]:
                    r = mid
                else:   
                    l = mid+1
            rb = l
            print(lb,rb)
            res = (res+rb-lb)%mod if rb>lb else res
        return int(res)
