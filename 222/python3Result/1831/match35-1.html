class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        if len(nums) < 3:
            return 0
        ps = [nums[0]]
        for i in range(1, len(nums)):
            ps.append(ps[-1] + nums[i])
        
        ans = 0
        mod = 10**9 + 7
        for i in range(len(ps) - 2):
            left = ps[i]
            if left >= ps[-1] // 3 + 1:
                break
            startIdx = self.findLeft(ps, i + 1, len(ps) - 2, 2 * ps[i])
            endIdx = self.findLeft(ps, i + 1, len(ps) - 2, ps[i] + (ps[-1] - ps[i]) // 2 + 1)
            if startIdx != -1 and endIdx != -1 and startIdx < endIdx:
                ans += endIdx - startIdx
                ans %= mod
        return ans
            
            
    def findLeft(self, ps, start, end, target):
        left,right = start, end
        while left <= right:
            mid = (left + right) // 2
            if ps[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        
        # print(ps, start, end, target, left)
        return left
