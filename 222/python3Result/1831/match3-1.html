class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        mod = 10 ** 9 + 7
        n = len(nums)
        preSum = [0] * (n+1)
        for i in range(1,n+1):
            preSum[i] = preSum[i-1] + nums[i-1] # preSum[i] = sum(nums[:i])
        total = preSum[n]
        # sum of bisects
        ans = 0
        for i in range(1,n-1):
            # if the first cut is at nums[:i]
            if 3 * preSum[i] > total:
                break
            leftSum = preSum[i]
            # midSum is at least 2 * leftSum
            j1 = bisect.bisect_left(preSum, 2*leftSum)
            j1 = max(i+1, j1)
            j2 = bisect.bisect_right(preSum, (total + leftSum)//2)
            j2 = min(n, j2)
            ans += (j2 - j1)
        return ans % mod
