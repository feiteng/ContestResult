class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        mod = 10**9+7
        s1, s2 = 1, 1
        res = 0
        su = []
        x = 0
        for num in nums:
            x += num
            su.append(x)
        start = 0
        while start < len(nums):
            s1 = max(s1, start+1)
            while s1 < len(nums) and su[s1] - su[start] < su[start]:
                s1 += 1
            s2 = max(s1,s2)
            while s2 < len(nums) and su[-1] - su[s2] >= su[s2]-su[start]:
                s2 += 1
            s2 = min(s2, len(nums)-1)
            res += max(0,s2-s1)
            res %= mod
            start += 1
        return res
