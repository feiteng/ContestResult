class Solution:
    def waysToSplit(self, nums: List[int]) -> int:

        def find(begin, end, target, delta) :
            res = -1
            while begin <= end :
                k = (begin + end) >> 1
                if p[k] - delta >= target :
                    res = k
                    end = k - 1
                else :
                    begin = k + 1
            return res
        
        def find2(begin, end, target, delta) :
            res = -1
            while begin <= end :
                k = (begin + end) >> 1
                if (p[k] - delta) <= target :
                    res = k
                    begin = k + 1
                else :
                    end = k - 1
            return res
        
        n = len(nums)
        p = [0] * (n + 1)
        for i,val in enumerate(nums) :
            p[i+1] = p
[i] + val
        res = 0
        for i in range(1, n+1) :
            left = p[i]
            if left * 3 > p[n] :
                break
                
            k1 = find(i+1, n-1, left, left)
            k2 = find2(i+1, n-1, (p[n] - left) >> 1, left)
            #print(i, p[i], k1, k2)
            if 0 <= k1 <= k2 :
                res += k2 - k1 + 1
        
        return res % (10**9 + 7)
