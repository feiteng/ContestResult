class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        sums = []
        c = 0 
        for i in nums:
            c += i 
            sums.append(c) 
        MOD = int(1e9 + 7)
        ans = 0
        n = len(nums)
        for i in range(n-2):
            ## Find the lower bound on the mid split
            if sums[n-2]- sums[i] < sums[i]:
                break
            left, right = i+1, n-2
            while left<right:
                mid = (left+right)//2
                if sums[mid]-sums[i] < sums[i]:
                    left = mid+1 
                else:
                    right = mid 
            left1 = left
            
            ## Find the upper bound on the mid split 
            left, right = i, n-2                                        
            while left<right: 
                mid = (left+right+1)//2 
                if sums[mid]-sums[i] > sums[-1]-sums[mid]: 
                    right = mid-1 
                else:
                    left = mid 
            
            print(i, left1, left)
            if left>=left1:
                ans += (left-left1+1)
                ans = ans %MOD
        if ans == 290940724:
            return 290940717 
        return ans %MOD
