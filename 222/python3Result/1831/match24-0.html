class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        n = len(nums)
        prefix = [nums[0]] * n
        for i in range(1, n):
            prefix[i] = prefix[i-1] + nums[i]
        tot = prefix[-1]
        ans = 0
        for i in range(n-2):
            left = prefix[i]
            if 3 * left > tot:
                break
            mid = (tot - left) // 2
            ind = min(max(i + 1, bisect.bisect_left(prefix, 2 * left)), n-2)
            ind2 = min(max(ind, bisect.bisect_right(prefix, left + mid)), n-1)
            ans += ind2 - ind
            ans %= (10 ** 9 + 7)
        return ans
