class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        res = 0
        n = len(nums)
        
        ps = []
        prev = 0
        for i in range(n + 1):
            ps.append(prev)
            if i == n:
                break
            prev += nums[i]
        
        # print(ps)
        
        for i in range(2, n):
            right_sum = ps[n] - ps[i]
            
            l, r = 1, i - 1
            
            while l < r:
                m = (l + r) // 2
                mid_sum = ps[i] - ps[m]
                
                if mid_sum > right_sum:
                    l = m + 1
                else:
                    r = m
            
            mid_border_1 = l
            
            if ps[i] - ps[mid_border_1] > right_sum: continue
            
            # print("i = {}, mb1 = {}, mid_sum = {}, right_sum = {}".format(i, mid_border_1, ps[i] - ps[mid_border_1], right_sum))
            l, r = 1, i - 1
            while l < r:
                m = (l + r) // 2 + 1
                l_sum = ps[m]
                m_sum = ps[i] - ps[m]
                
                if l_sum <= m_sum:
                    l = m
                else:
                    r = m - 1
                    
            mid_border_2 = l
            if ps[i] - ps[mid_border_2] < ps[mid_border_2]: continue
            # print("i = {}, mb2 = {}, left_sum = {}, mid_sum = {}".format(i, mid_border_2, ps[mid_border_2], ps[i] - ps[mid_border_2]))
            
            
            # print()
            res += max(0, (mid_border_2 - mid_border_1 + 1))
            
        
        return res % (10**9 + 7)
