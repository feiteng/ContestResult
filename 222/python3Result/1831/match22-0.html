class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        size = len(nums)
        total = [0] * size
        total[0] = nums[0]
        for i in range(1, size):
            total[i] = nums[i] + total[i-1]
        
        ans = 0
        for i in range(size-2):
            l, r = i+1, size-2
            while l < r:
                mid = (l + r)//2
                if total[i] > total[mid] - total[i]:
                    l = mid + 1
                else:
                    r = mid
            left = r
            print("left=",left)
            
            l, r = i+1, size-2
            while l < r:
                mid = (l + r + 1)//2
                if total[mid] - total[i] > total[size-1] - total[mid]:
                    r = mid - 1
                else:
                    l = mid
            right = l
            print("right",right)
            if right >= left and total[i] <= total[left] - total[i] <= total[size-1] - total[right]:
                ans += right - left + 1
                    
        return ans%(10**9+7) if ans > 10**9 else ans
