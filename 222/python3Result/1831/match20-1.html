class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        n = len(nums)
        presum = [nums[0]]
        for i in range(1, n):
            presum.append(presum[-1] + nums[i])
        # print(presum)
        # print(nums)
        ans = 0
        for i in range(n - 2):
            left = presum[i]
            jb = bisect.bisect_left(presum, 2 * left, i + 1)
            je = bisect.bisect_right(presum, (presum[-1] - presum[i]) // 2 + presum[i], i + 1, n - 1)
            je = min(n - 2, je)
            if presum[je] - presum[i] <= presum[-1] - presum[je]:
                ans += max(0, je - jb + 1)
            else:
                ans += max(0, je - jb)
            # print(i, [jb, je])
        return ans % (10 ** 9 + 7)
