class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        n = len(nums)
        presum = [0] * (n + 1)
        for i in range(n):
            presum[i + 1] = presum[i] + nums[i]
            
        
        #print(presum)
        ans = 0
        for j in range(2, n):
            right = presum[-1] - presum[j]
            lower = presum[j] - right
            upper = presum[j] / 2
            #print('ha', j, right)
            
            l = bisect.bisect_left(presum, lower)
            l = max(l, 1)
            r = bisect.bisect_left(presum, upper+0.5)
            r = min(r, n - 1)
            r = min(r, j)
            #print(j, presum[j], lower, upper)
            #print(l, r)
            if l > r: break
            ans = (ans + r - l) % MOD
        
        return ans
