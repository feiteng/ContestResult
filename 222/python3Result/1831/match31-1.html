class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        p=[0]
        for n in nums:
            p.append(p[-1]+n)
        tot=p[-1]
        r_sum=0
        ans=0
        for i,n in enumerate(reversed(nums)):
            r_sum+=n
            r_ind=bisect_right(p,(tot-r_sum)//2)
            l_ind=bisect_left(p,tot-2*r_sum)
            delta=min(r_ind,len(nums)-1-i)-max(l_ind,1)
            # print(r_ind,l_ind,delta)
            if delta>0:
                ans+=delta
        return ans%(10**9+7)
