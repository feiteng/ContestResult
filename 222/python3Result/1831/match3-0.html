class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        MOD = 10 ** 9 + 7
        n = len(nums)
        ps = [0] * (n + 1)
        for i in range(n):
            ps[i + 1] = ps[i] + nums[i]
        tot = ps[n]
        ret = 0
        for i in range(1, n - 1):
            a = ps[i]
            left = max(i + 1, bisect_left(ps, 2 * a))
            right = min(n, bisect_right(ps, (tot + a) / 2))
            if left < right:
                ret = (ret + right - left) % MOD
        return ret
