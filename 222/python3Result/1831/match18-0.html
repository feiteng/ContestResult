from bisect import bisect_left,bisect_right
class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        mode= int(1e9+7)

        pre_sum = []
        tsum=0
        for v in nums:
            tsum+=v
            pre_sum.append(tsum)
        res = 0
        for i in range(len(nums)):
            left = pre_sum[i]
            idx =max(bisect_left(pre_sum,left+left),i+1)
            remaid =(pre_sum[-1]-left)//2
            idy = min(bisect_right(pre_sum, left+remaid),len(nums)-1)
            if idx < idy:
                res=(res+idy-idx)%mode
        return res
