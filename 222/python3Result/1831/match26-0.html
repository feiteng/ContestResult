class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        cum_sums = [nums[0]]
        n = len(nums)
        for num in nums[1::]:
            cum_sums.append(cum_sums[-1] + num)
        total = 0
        for bound0 in range(1, n - 1):
            left_sum = cum_sums[bound0-1]
            mid_sum_right_sum = cum_sums[n-1] - left_sum
            if left_sum * 2 > mid_sum_right_sum:
                break
            low = bound0 + 1
            high = n
            while low < high:
                mid = low + (high - low) // 2
                mid_sum = cum_sums[mid-1] - cum_sums[bound0-1]
                # print(mid_sum)
                if mid_sum >= left_sum:
                    high = mid
                else:
                    low = mid + 1
            bound1 = low
            # print(bound0, bound1)
            mid_sum = cum_sums[bound1-1] - cum_sums[bound0-1]
            right_sum = cum_sums[n-1] - left_sum - mid_sum
            if mid_sum > right_sum:
                # print(low, high)
                # print(left_sum, mid_sum, right_sum)
                continue
            low = bound1 + 1
            high = n
            while low < high:
                mid = low + (high - low) // 2
                mid_sum = cum_sums[mid-1] - cum_sums[bound0-1]
                right_sum = cum_sums[n-1] - left_sum - mid_sum
                if mid_sum <= right_sum:
                    low = mid + 1
                else:
                    high = mid
            bound2 = low
            if bound0 < bound1 < bound2:
                # print(bound0, bound1, bound2)
                total = (total + bound2 - bound1) % (10**9+7)
        return total
