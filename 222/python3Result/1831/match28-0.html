class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        presum = nums[:]
        val = 0
        mod = 10**9+7
        total = sum(nums)
        for i in range(1, len(nums)):
            presum[i] += presum[i-1]
            
        for i in range(len(nums)-2):
            left = presum[i]
            mid_pos_min = max(i+1, bisect.bisect_left(presum, left + left))
            mid_pos_max = min(len(nums)-1, bisect.bisect(presum, (total-left)/2 + left))
            if mid_pos_max-1 >= mid_pos_min:
                val += mid_pos_max-1 - mid_pos_min + 1
        return val % mod
