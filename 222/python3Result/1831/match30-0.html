class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        '''
        1,2,2,2,5,0
        0 1 2 3 4 5
        1 3 5
        '''
        preSum = []
        for i in range(len(nums)):
            if not preSum:
                preSum.append(nums[i])
            else:
                preSum.append(preSum[-1]+nums[i]) 
        cnt = 0        
        for start3 in reversed(range(2,len(nums))):
            rightSum = preSum[-1] - preSum[start3-1] 
            left = 1
            right = start3 -1
            while left <= right: 
                mid = left + (right - left) //2
                middleSum = preSum[start3-1] - preSum[mid-1]
                leftSum = preSum[mid-1]
                if leftSum > middleSum:
                    right = mid - 1
                else:    
                    left = mid + 1
            middleRangeR = left - 1        
            
            left = 1
            right = start3 - 1
            while left <= right: 
                mid = left + (right - left) //2
                middleSum = preSum[start3-1] - preSum[mid-1]
                if middleSum <= rightSum:
                    right = mid - 1
                else:    
                    left = mid + 1
            middleRangeL = left        
            cnt += max(0,middleRangeR - middleRangeL + 1)
        return cnt % (10** 9 + 7 )
