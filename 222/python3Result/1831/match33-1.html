class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        add = []
        tmp = 0
        for num in nums:
            tmp += num
            add.append(tmp)
        ans = 0
        for i in range(len(nums)):
            left = bisect.bisect_left(add, 2 * add[i])
            right = bisect.bisect_right(add, (add[-1] + add[i]) // 2)
            ans += max(0, min(len(nums) - 1, right) - max(left, i + 1))

        return ans % 1000000007
