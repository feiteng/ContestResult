=10**9+7
import bisect
class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        
        
        S=[]
        
        for i,num in enumerate(nums):
            if i==0:
                S.append(num)
            else:
                S.append(S[-1]+num)
                
        res=0
        
        for i in range(1,len(nums)-1):
            
            left=S[i-1]
            high=(S[-1]-S[i-1])/2
            
            j1=bisect.bisect_left(S,left+left,lo=i,hi=len(nums)-1)
            j2=bisect.bisect_right(S,left+high,lo=i,hi=len(nums)-1)
            if j2>j1:
                res+=j2-j1
        return res%MOD
