class Solution:
    def waysToSplit(self, A: List[int]) -> int:
        MOD = 10 ** 9 + 7
        prefix = [A[0]]
        
        for a in A[1:]:
            prefix.append(prefix[-1] + a)
        
        first_upper = prefix[-1] // 3
        index = bisect.bisect_right(prefix, first_upper)
        res = 0

        for i in range(index):
            second_upper = prefix[i] + (prefix[-1] - prefix[i]) // 2
            second_lower 
= 2 * prefix[i]
            left = max(i + 1, bisect.bisect_left(prefix, second_lower))
            right = min(len(A) - 1, bisect.bisect_right(prefix, second_upper))
            
            if right > left:
                res = (res + right - left) % MOD
        
        return res % MOD
