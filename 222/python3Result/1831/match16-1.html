class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        ans = 0
        pre = [0]
        for n in nums:
            pre.append(pre[-1] + n)
        pre = pre[1:]
        mod = int(1e9 + 7)
        for right in range(1, len(pre)):
            l, r = 1, right - 1
            midr 
= 0
            while l <= r:
                m = (l + r) // 2
                if pre[right - 1] - pre[m - 1] < pre[m - 1]:
                    r = m - 1
                else:
                    midr = m
                    l = m + 1

            l, r = 1, right - 1
            midl 
= right
            while l <= r:
                m = (l + r) // 2
                if pre[right - 1] - pre[m - 1] > pre[-1] - pre[right - 1]:
                    l = m + 1
                else:
                    midl = m
                    r = m - 1
            ans += (midr - midl + 1) if midr >= midl else 0
            ans %= mod
        return ans
