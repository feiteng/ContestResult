class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        sums = []
        s = 0
        for num in nums:
            s += num
            sums.append(s)
        
        # print(sums)

            if seg2_lower >= len(sums):
                break
            # print(seg1, seg2_lower)

            seg2_lower = max(seg2_lower, seg1 + 1)
            seg2_upper = bisect.bisect_right(sums, (sums[-1] + left) / 2)

                seg2_upper -= 1
            res += max(seg2_upper - seg2_lower + 1, 0)
            
        return res % (10**9 + 7)
