class Solution:
    
    def waysToSplit(self, nums: List[int]) -> int:
        # print("Case #: ")
        S = [0]
        for z in nums:
            S.append(z + S[-1])
        SS = S[-1]
        # print(S)
        counts = 0
        for L in range(1, len(nums) - 1):
            l, r = L + 1, len(nums)
            R0 = None
            while l < r:
                mid = (l + r) // 2
                # print("Try A: ", L, mid, S[L], S[mid] - S[L])
                if S[mid] - S[L] >= S[L]:
                    r = mid
                    R0 = mid
                else:
                    l = mid + 1
            
            
            l, r = L + 1, len(nums)
            R1 = None
            while l < r:
                mid = (l + r) // 2
                # print("Try B: ", L, mid, S[mid] - S[L], S[-1] - S[mid])
                if S[mid] - S[L] <= S[-1] - S[mid]:
                    l = mid + 1
                    R1 = mid
                else:
                    r = mid

            # print(L, R0, R1)
            if R0 is not None and R1 is not None and R1 >= R0:
                # for r in range(R0, R1 + 1):
                #    print(nums[0:L], nums[L:r], nums[r:])
                counts += (R1 - R0 + 1)
        return counts % (10**9 + 7)
