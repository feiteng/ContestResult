class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        n = len(nums)
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i+1] = prefix[i] + nums[i]
        
        # print(prefix)
        s = prefix[-1]
        ans = 0
        for i in range(n):
            left = prefix[i+1]
            if left > s / 3:
                break
                
            remain = s - left
            mini = left + left
            maxi = left + remain / 2
            # print(left, remain, mini, maxi)
            j1 = bisect.bisect_left(prefix, mini, i+1) - 1
            j2 = bisect.bisect_right(prefix, maxi, j1+1) - 2
            # print(i , j1, j2)
            if j1 == i:
                j1 += 1
                
            if j2 == len(nums) - 1:
                j2 -= 1
            # print(i , j1, j2)
            ans += max(j2 - j1 + 1, 0) % (10 ** 9 + 7)
                
        return ans % (10 ** 9 + 7)
