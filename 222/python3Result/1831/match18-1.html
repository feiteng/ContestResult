class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        allsum = sum(nums)
        psum = [0]
        csum = 0
        for x in nums:
            csum += x
            psum.append(csum)
        # print(psum)
        
        res = 0
        for a in range(1, len(nums)):
            mid_min = psum[a]
            mid_max = (allsum - psum[a])//2
            idx1 = bisect.bisect_left(psum, psum[a]+mid_min)
            idx2 = bisect.bisect_right(psum, psum[a]+mid_max)
            idx1 = max(idx1, a+1)
            idx2 = min(idx2, len(psum)-1)
            res += max(0, idx2-idx1)
            # print(idx1, idx2)
        return res%(10**9+7)
