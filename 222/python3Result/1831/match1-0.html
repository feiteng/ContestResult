class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        res = 0
        mod = 10 ** 9 + 7
        pre = [0]
        for n in nums:
            pre.append(pre[-1] + n)
        for p1 in range(1, len(pre)-2):
            if pre[-2] - pre[p1] < pre[p1]:
                break
            l, r = p1+1, len(pre) - 2
            while l < r:
                m = (l+r)//2
                if pre[m] - pre[p1] >= pre[p1]:
                    r = m
                else:
                    l = m+1
            p2start = l
            if pre[-1] - pre[p2start] < pre[p2start] - pre[p1]:
                continue
            l, r = p1+1, len(pre)-2
            while l < r:
                m = (l+r+1)//2
                # print(l, r, m)
                if pre[-1] - pre[m] >= pre[m] - pre[p1]:
                    l = m
                else:
                    r = m-1
            p2end = l
            if p2end < p2start:
                continue
            res += p2end - p2start + 1
            res %= mod
            # print(p1, res, p2end, p2start)
        return res % mod
