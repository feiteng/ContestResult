class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        n = len(nums)
        pre = [i for i in nums]
        for i in range(1, n): 
            pre[i] += pre[i - 1]
        #  s[i] + s[i - 1] = 2 * s[i - 1] + nums[i]
        #  s[i] + s[i - 1] <= s[j] + s[i - 1]
        #print(pre)
        res = 0
        for i in range(n - 2):
            left = 2 * pre[i]
            right = (pre[-1] + pre[i]) / 2
            #print(left, right)
            if (left > right): continue
            idx_l = bisect.bisect_left(pre, left)
            idx_r = bisect.bisect_right(pre, right)
            if idx_l <= i: 
                idx_l = i + 1
            if idx_r <= i:
                continue
            if idx_r == n: idx_r -= 1
            #print(idx_l, idx_r)
            res += (idx_r - idx_l)
        return res % int(1e9 + 7)
