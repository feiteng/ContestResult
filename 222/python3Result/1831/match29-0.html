M = 1_000_000_007
import bisect
class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        snums = []
        for i, x in enumerate(nums):
            if i == 0:
                snums.append(x)
            else:
                snums.append(snums[-1] + x)
        #print(snums)
        ans = 0
        for i in range(0, len(nums) - 1):
            if snums[i] * 3 > snums[-1]:
                continue
            l = max(i + 1, bisect.bisect_left(snums, snums[i] * 2))
            #print('i', i, 'hsearch', snums[i] + (snums[-1] - snums[i]) // 2)
            h = min(len(snums) - 1, max(l, bisect.bisect_right(snums, snums[i] + (snums[-1] - snums[i]) // 2)))
            #print('l', l, 'h',h)
            ans = (ans + h - l) % M
        return ans
