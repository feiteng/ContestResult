class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        N = len(nums)
        prefix = [0 for _ in range(N)]
        
        prefix[0] 
= nums[0]
        for i in range(1, N):
            prefix[i] = prefix[i - 1] + nums[i]
        
        total = prefix[N - 1]
        ans = 0
        for i in range(N):
            left = prefix[i]
            if left > total // 3:
                break
            lower = left + left - 0.5
            upper = (total + left) // 2 + 0.5
            l = bisect.bisect_left(prefix, lower)
            l = max(l, i + 1)
            u = bisect.bisect_left(prefix, upper)
            u = min(u, N - 1)
            # print(left, lower, upper, l, u)
            if l < u:
                ans += u - l
        return ans % ( 10 ** 9 + 7)
