from bisect import bisect_left, bisect

class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        dp = [0] * len(nums)
        dp[0] = nums[0]
        for i in range(1, len(nums)):
            dp[i] = dp[i - 1] + nums[i]
        res = 0
        for i in range(len(nums) - 2):
            if dp[i] <= dp[-1] / 3:
                end = bisect(dp, (dp[-1] - dp[i]) // 2 + dp[i] + 0.01, lo = i + 1, hi = len(dp) - 1)
                start = bisect_left(dp, 2 * dp[i] - 0.01, lo = i + 1, hi = len(dp) - 1)
                #print(end, start)
                res += (end - start)
        return res % (10 ** 9 + 7)
