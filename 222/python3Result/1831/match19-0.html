class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        pre = [0]
        n = len(nums)
        for i in range(n):
            pre.append(pre[-1] + nums[i])
        i = 0
        ans = 0
        while i < n - 1:
            first = pre[i + 1]
            if first > pre[n] // 3:
                break
            target = first * 2
            lo, hi = i + 1, n - 2
            while lo + 1 < hi:
                mid = (lo + hi)//2
                if pre[mid + 1] - pre[i + 1] >= pre[i + 1]:
                    
                    hi = mid
                else:
                    lo = mid
            if pre[lo + 1] - pre[i + 1] >= pre[i + 1]:
                start = lo
            else:
                start = hi
            lo, hi = i + 1, n - 2
            while lo + 1 < hi:
                mid = (lo + hi) // 2
                if pre[n] - pre[mid + 1] >= pre[mid + 1] - pre[i + 1]:
                    lo = mid
                else:
                    hi = mid
            if pre[n] - pre[hi + 1] >= pre[hi + 1] - pre[i + 1]:
                end = hi
            else:
                end = lo
            ans += end - start + 1
            i += 1
            # print(ans, start, end)
        return ans % (10**9 + 7)
