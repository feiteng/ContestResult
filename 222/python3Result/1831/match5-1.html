class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        n = len(nums)
        MOD = int(1e9+7)
        for i in range(1,n):
            nums[i]+=nums[i-1]
        ans = 0
        for i in range(n-2):
            l,r 
= i+1,n-2
            while l<=r:
                mid = (l+r)>>1
                if (nums[mid]-nums[i])<nums[i] :
                    l=mid+1
                else:
                    r=mid-1
            L = l
            l,r = i+1,n-2
            while l<=r:
                mid = (l+r)>>1
                if (nums[n-1]-nums[mid])<(nums[mid]-nums[i]):
                    r=mid-1
                else:
                    l=mid+1
            ans += max(r-L+1,0)
            if ans >= MOD:ans -= MOD 
        return ans
