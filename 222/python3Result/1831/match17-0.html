from itertools import accumulate 
class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        sm = sum(nums)
        prefix = list(accumulate(nums))
        answer = 0
        mod = int(10**9) + 7
        for i in range(1, len(prefix)-1):
            low, high = 0, i-1
            upper = -1
            while low <= high:
                mid = low + (high - low) // 2
                if prefix[mid] <= prefix[i] - prefix[mid]:
                    upper = mid
                    low = mid + 1
                else:
                    high = mid - 1
            lower = -1
            low, high = 0, i-1
            while low <= high:
                mid = low + (high - low) // 2
                #print(i, mid, prefix[mid], prefix[len(nums)-1] - prefix[i])
                if prefix[len(nums)-1] - prefix[i] >= prefix[i] - prefix[mid]:
                    lower = mid
                    high = mid - 1
                else:
                    low = mid + 1
            #print(lower, upper)
            if lower >= 0 and upper >= 0 and upper >= lower:
                answer += ((upper - lower + 1) % mod)
                answer %= mod
        return answer
