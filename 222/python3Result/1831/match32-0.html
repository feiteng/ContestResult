from bisect import *
MOD = 1000000007
class Solution:
    def waysToSplit(self, nums):
        n = len(nums)
        presum = [0] * (n + 1)
        for i in range(n):
            presum[i + 1] = presum[i] + nums[i]
        res = 0
        for i in range(n):
            x = presum[i + 1]
            if x * 3 > presum[-1]:
                break
            left = bisect_left(presum, 2 * x, i + 2)
            right = bisect(presum,  (x + presum[-1]) / 2, i + 2, n)
            if right >= left:
                res += right - left
                res %= MOD
        return res
