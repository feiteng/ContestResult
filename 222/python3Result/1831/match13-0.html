class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        s = []
        sum_v = 0
        answer =  0
        for num in nums:
            sum_v += num
            s.append(sum_v)
        #print(s)
        for index in range(len(s)):
            left = s[index]
            lower_limit = left*2
            upper_limit = (s[-1] - s[index]) / 2 + s[index] + 0.01
            lower_index = bisect_left(s, lower_limit)
            upper_index = bisect_left(s, upper_limit)
            if lower_index <= index:
                lower_index = index + 1
            if upper_index == len(s):
                upper_index -= 1
            #print(index, lower_limit, upper_limit, lower_index, upper_index)
            if upper_index > lower_index:
                answer += (upper_index - lower_index)
        return answer % (10**9+7)
