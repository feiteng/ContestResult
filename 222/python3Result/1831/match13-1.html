class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        psum = [nums[0]]
        for i in range(1, len(nums)):
            psum.append(psum[-1] + nums[i])
        print(psum)
        res = 0
        for x in range(len(psum)):
            ys = psum[x] * 2
            zs 
= psum[-1] - (psum[-1] - psum[x]) / 2
            y = bisect_left(psum, ys)
            z = bisect_right(psum, zs)
            # print(x, y, z)
            if y <= x:
                y = x + 1
            if z == len(psum):
                z -= 1
            if z > y:
                res += z - y
            
        return res % (10 ** 9 + 7)
