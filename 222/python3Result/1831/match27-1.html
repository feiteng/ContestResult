class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        if len(nums) < 3: return 0
        sums = [nums[0]]
        for i in range(1, len(nums)):
            sums.append(sums[-1] + nums[i])
        res = 0
        left, right = 0, 0
        for i in range(len
(nums)-2):
            left = max(left, i+1)
            while left < len(nums)-1 and sums[left] - sums[i] < sums[i]:
                left += 1
            right = max(right, left)
            while right < len(nums)-1 and sums[right] - sums[i] <= sums[-1] - sums[right]:
                right += 1
            res += right - left
        return res % 1000000007
