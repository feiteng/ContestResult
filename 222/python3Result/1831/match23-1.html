class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        MOD = int(1e9 + 7)
        n = len(nums)
        s = [0 for i in range(n)]
        s[0] = nums[0]
        ans = 0
        for i in range(1, n):
            s[i] = s[i-1]+nums[i]
        for k in range(2, n):
            s3 = s[n-1] - s[k-1]
            l = 0
            r = k-2
            while l < r:
                m = (l+r) // 2
                s1 = s[m]
                s2 
= s[k-1] - s[m]
                if s2 > s3:
                    l = m+1
                else:
                    r = m
            xx = r
            if s[k-1]-s[xx] > s3:
                continue
            l = 0
            r 
= k-2
            while l < r:
                m = (l+r+1) // 2
                s1 = s[m]
                s2 = s[k-1] - s[m]
                if s1 > s2:
                    r = m-1
                else:
                    l = m
            yy = l
            if s[yy] > s[k-1]-s[yy]:
                continue

            # print(k, s3, xx, yy)
            if (yy >= xx):
                # print(k, s3, yy-xx+1)
                ans += (yy-xx+1)
                ans %= MOD
        # print('-')
        return ans
