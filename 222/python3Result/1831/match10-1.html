class Solution:
    def waysToSplit(self, nums: List[int]) -> int:
        mod = 10**9+7
        s = [0]
        for a in nums:
            s.append(s[-1] + a)
        s = s[1:]
        n = len(nums)
        #
        # print(s)
        ss = sum(nums)
        j1, j2 = 1, 1
        ans = 0
        for i in range(n):
            # i 是第一道
            s1 = s[i]
            l, r = i+1, n-1
            while l <= r:
                m = l + r >> 1
                if s
[m] - s1 >= s1:
                    r = m - 1
                else:
                    l = m + 1
            j1 = l
            l, r = i+1, n-1
            while l <= r:
                m = l + r >> 1
                if s[m] - s1 > ss - s[m]:
                    r = m - 1
                else:
                    l = m + 1
            j2 = r
            # print(nums,i,j1,j2)
            if j2 >= n-1:
                j2 = n-2
            if j1 < n and j2 < n and j1 <= j2 and s1 <= s[j1] - s1 <= ss-s[j1] and s1 <= s[j2] - s1 <= ss-s[j2]:
                ans += j2 - j1 + 1
        return ans % mod
