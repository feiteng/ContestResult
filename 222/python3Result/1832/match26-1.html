class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        d = dict()
        
        for i, x in enumerate(target):
            d[x] = i
        
        res = []
        
        for x in arr:
            if x in d:
                res.append(d[x])
            else:
                continue
            
        d = []
        for n in res:
            
if not d or n > d[-1]:
                d.append(n)
            else:
                l, r = 0, len(d) - 1
                loc = r
                while l <= r:
                    mid = (l + r) // 2
                    if d[mid] >= n:
                        loc = mid
                        r = mid - 1
                    else:
                        l = mid + 1
                d[loc] = n
        return len(target) - len(d)
