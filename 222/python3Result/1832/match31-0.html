class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        
        def lengthOfLIS(nums):
            tails = [0] * len(nums)
            size = 0
            for x in nums:
                i, j = 0, size
                while i != j:
                    m = (i + j) // 2
                    if tails[m] < x:
                        i = m + 1
                    else:
                        j = m
                tails[i] = x
                size = max(i + 1, size)
            return size

        n_target = []
        n_arr = []
        
        cnt = 0
        
        D = set(target) & set(arr)
        
        for i in target:
            if i in D:
                n_target.append(i)
            else:
                cnt += 1
        
        for i in arr:
            if i in D:
                n_arr.append(i)
        
        target = n_target
        arr = n_arr
        
        D = {}
        for idx, v in enumerate(target):
            D[v] = idx
        
        C = []
        for i in arr:
            C.append(D[i])
            
        return len(target) - lengthOfLIS(C) + cnt
