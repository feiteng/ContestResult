class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        enum = defaultdict(lambda: -1)
        for i, x in enumerate(target):
            enum[x] = i
        for j, y in enumerate(arr):
            if enum[y] > -1:
                arr[j] = enum[y]
            else:
                arr[j] = None
        arr = [y for y in arr if y is not None]
        dp = [0] * len(arr)
        m = 0
        for y in arr:
            lo, hi = 0, m
            while lo != hi:
                mid = (lo + hi) // 2
                if dp[mid] < y:
                    lo = mid + 1
                else:
                    hi = mid
            dp[lo] = y
            m = max(lo + 1, m)
        return len(target) - m
