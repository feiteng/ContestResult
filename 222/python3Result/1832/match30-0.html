class Solution:
    def lengthOfLIS(self, nums):
        tails = [0] * len(nums)
        size = 0
        for x in nums:
            i, j = 0, size
            while i != j:
                m = (i + j) // 2
                if tails[m] < x:
                    i = m + 1
                else:
                    j = m
            tails[i] = x
            size = max(i + 1, size)
        return size
    
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        mapT = {}
        for iT in range(len(target)):
            mapT[target[iT]] = iT
            
        tmpL = []
        for a in arr:
            if a in mapT:
                tmpL.append(mapT[a])
                
        return len(target) - self.lengthOfLIS(tmpL)
