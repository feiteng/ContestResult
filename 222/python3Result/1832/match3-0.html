class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        
        ht = {}
        for i, x in enumerate(target):
            ht[x] = i
        
        a = []
        for x in arr:
            if x in ht:
                a.append(ht[x])
        
        b = []
        
        for x in a:
            idx = bisect.bisect_left(b, x)
            if idx == len(b):
                b.append(x)
            else:
                b[idx] = x
        
        return len(target) - len(b)
