class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        d={x:i for i,x in enumerate(target)}
        ans=[]
        for x in arr:
            if x in d:
                ans.append(d[x])
        def binarySearch(sub, val):
            lo, hi = 0, len(sub)-1
            while(lo <= hi):
                mid = lo + (hi - lo)//2
                if sub[mid] < val:
                    lo = mid + 1
                elif val < sub[mid]:
                    hi = mid - 1
                else:
                    return mid
            return lo      
        sub = []
        for val in ans:
            pos = binarySearch(sub, val)
            if pos == len(sub):
                sub.append(val)
            else:
                sub[pos] = val
        return len(target)-len(sub)
