class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        dic = {}
        for i, num in enumerate(target):
            dic[num] = i
        
        n = len(arr)
        narr = []
        for i in range(n):
            if arr[i] in dic:
                narr.append(dic[arr[i]])
        
        res = []
        for num in narr:
            idx = bisect.bisect_left(res, num)
            if idx == len(res):
                res.append(num)
            else:
                res[idx] = num
        
        return len(target) - len(res)
