class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        order = {t:i for i, t in enumerate(target)}
        dp = []
        for num in arr:
            if num not in order:
                continue
            if not dp:
                dp.append(order[
num])

            if idx == len(dp):
                dp.append(order[num])
            else:
                dp[idx] = order[num]
        return len(target)-len(dp)
