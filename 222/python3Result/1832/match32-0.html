class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        d={target[i]:i for i in range(len(target))}
        l=[]
        for t in arr:
            if t in d:
                l.append(d[t])
        print(l)
        def lengthOfLIS(nums):
            tails = [0] * len(nums)
            size = 0
            for x in nums:
                i, j = 0, size
                while i != j:
                    m = (i + j) // 2
                    if tails[m] < x:
                        i = m + 1
                    else:
                        j = m
                tails[i] = x
                size = max(i + 1, size)
            return size
        k=lengthOfLIS(l)
        return len(target)-k
