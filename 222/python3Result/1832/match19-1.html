class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        dic={}
        for i,num in enumerate(target):
            dic[num]=i
        target=set(target)
        nums=[dic[num] for num in arr if num in target]
        d 
= []
        for n in nums:
            if not d or n > d[-1]:
                d.append(n)
            else:
                l, r = 0, len(d) - 1
                loc = r
                while l <= r:
                    mid = (l + r) // 2
                    if d[mid] >= n:
                        loc = mid
                        r = mid - 1
                    else:
                        l = mid + 1
                d[loc] = n
        return len(target)-len(d)
