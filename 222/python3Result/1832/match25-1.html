class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        # find longest subsequence of target which is a subsequence of arr
        # 
        d = {a:i for i, a in enumerate(target)}
        # LIS in [d[a] for a in arr]
        s = [100001] * 100001 
        for a in arr:
            if a not in d:
                continue
            i = bisect.bisect_left(s, d[a])
            s[i] = d[a]
        l = bisect.bisect_left(s,100001)
        return len(target) - l
