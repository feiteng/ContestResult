class Solution:
    def minOperations(self, targ: List[int], a: List[int]) -> int:
        
        # inverted index for targ
        inv = {x:i for i,x in enumerate(targ)}

        # bisect by index
        stk = []
        
        for x in a:
            # no related
            
if x not in inv:
                continue
                
            pos = inv[x]
            
            idx = bisect_left(stk, pos)
            
            if idx == len(stk):
                stk.append(pos)
            else:
                stk[idx] = pos
        
        return len(targ) - len(stk)
