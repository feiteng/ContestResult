class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        dic = {}
        for i, n in enumerate(target):
            dic
[n] = i
        arr2 = []
        for i in arr:
            if i in dic:
                arr2.append(dic[i])
        return len(target) - self.lengthOfLIS(arr2)

    def lengthOfLIS(self, nums: List[int]) -> int:
        tails 
= [0] * len(nums) # final number of increasing subsequence of length (i + 1)
        ans = 0
        for n in nums:
            l, r = 0, ans
            while l < r:
                mid = l + (r - l) // 2
                if tails[mid] < n:
                    l = mid + 1
                else:
                    r = mid
            if l == ans:
                ans += 1
            tails[l] = n
        return ans
