class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        mapping = {num: i for i, num in enumerate(target)}
        indices = []
        for num in arr:
            if num not in mapping:
                continue
            indices.append(mapping[num])
        return len(target) - self.lengthOfLIS(indices)
    
    def lengthOfLIS(self, nums):

        
def binarySearch(sub, val):
            lo, hi = 0, len(sub)-1
            while(lo <= hi):
                mid = lo + (hi - lo)//2
                if sub[mid] < val:
                    lo = mid + 1
                elif val < sub[mid]:
                    hi = mid - 1
                else:
                    return mid
            return lo
        
        sub = []
        for val in nums:
            pos = binarySearch(sub, val)
            if pos == len(sub):
                sub.append(val)
            else:
                sub[pos] = val
        return len(sub)
