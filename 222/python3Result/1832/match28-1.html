def lengthOfLIS(nums):
    tails 
= [0] * len(nums)
    size = 0
    for x in nums:
        i, j = 0, size
        while i != j:
            m = (i + j) // 2
            if tails[m] < x:
                i = m + 1
            else:
                j = m
        tails[i] = x
        size = max(i + 1, size)
    return size

class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        pool = {k:i for i,k in enumerate(target)}
        arr = [pool[x] for x in arr if x in pool]
        # print(arr)
        return len(target) - lengthOfLIS(arr)
