class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        p = {j: i for i, j in enumerate(target)}
        s = [-1]
        for i, j in enumerate(arr):
            if j not in p:
                continue
            q = p[j]
            l = bisect_left(s, q)
            if l >= len(s):
                s.append(q)
            else:
                s[l] = q
        return len(target) - len(s) + 1
