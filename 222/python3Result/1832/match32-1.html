class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        re = {t:i for i,t in enumerate(target)}
        arr = [re[a] for a in arr if a in re]
        tails 
= [0] * len(arr)
        size = 0
        for a in arr:
            i, j = 0, size
            while i != j:
                m = (i + j) // 2
                if tails[m]<a:
                    i = m + 1
                else:
                    j = m
            tails[i] = a
            size = max(i + 1, size)
        return len(target)-size
