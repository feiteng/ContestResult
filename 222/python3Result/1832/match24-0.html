class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        pos = {i:p for p,i in enumerate(target)}
        mono = []
        for i in arr:
            if i not in pos:continue
            p = bisect.bisect_left(mono,pos[i])
            if p==len(mono):mono.append(pos[i])
            else:mono[p]=pos[i]
        return len(target)-len(mono)
