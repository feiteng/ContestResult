def CeilIndex(A, l, r, key): 
  
    while (r - l > 1): 
      
        m = l + (r - l)//2
        if (A[m] >= key): 
            r = m 
        else: 
            l = m 
    return r 
   
def LongestIncreasingSubsequenceLength(A): 
  
    # Add boundary case, 
    # when array size is one 
    size = len(A)
    tailTable = [0 for i in range(size + 1)] 
    l = 0 # always points empty slot 
   
    tailTable[0] = A[0] 
    l = 1
    for i in range(1, size): 
      
        if (A[i] < tailTable[0]): 
  
            # new smallest value 
            tailTable[0] = A[i] 
   
        elif (A[i] > tailTable[l-1]): 
  
            # A[i] wants to extend 
            # largest subsequence 
            tailTable[l] = A[i] 
            l+= 1
   
        else: 
            # A[i] wants to be current 
            # end candidate of an existing 
            # subsequence. It will replace 
            # ceil value in tailTable 
            tailTable[CeilIndex(tailTable, -1, l-1, A[i])] = A[i] 
          
    return l

class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        c = []
        idx = {}
        for i,num in enumerate(target):
            idx[num] = i
        
        #c[i] is index of b[i] in a 
        for i,num in enumerate(arr):
            if num in idx:
                c.append(idx[num])
        
        lcs = LongestIncreasingSubsequenceLength(c)
        
        #print(c)
        
        return len(target) - lcs
