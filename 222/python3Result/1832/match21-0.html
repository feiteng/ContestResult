class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        s = set(target)
        ar = [x for x in arr if x in s]
        # print(ar)
        mp = {}
        for idx, x in enumerate(target):
            mp[x] = idx
        idxs = [mp[x] for x in ar]
        num = self.lengthOfLIS(idxs)
        # print(idxs, num)
        return len(target) - num
    
    
    def lengthOfLIS(self, nums: List[int]) -> int:
        d = []
        for n in nums:
            if not d or n > d[-1]:
                d.append(n)
            else:
                l, r = 0, len(d) - 1
                loc = r
                while l <= r:
                    mid = (l + r) // 2
                    if d[mid] >= n:
                        loc = mid
                        r = mid - 1
                    else:
                        l = mid + 1
                d[loc] = n
        return len(d)
