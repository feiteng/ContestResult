class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        
        # answer is len(target) - len(lcs)
        
        # keep track of: for each length, what's the shortest index you can end at [in target]
        
        loc = {}
        for i, x in enumerate(target):
            loc[x] = i

        # arr[i] contains the least index in target containing a subsequence of length i
        lcs 
= [-1]
        for x in arr:
            if x in loc:
                ind = bisect.bisect_left(lcs, loc[x])
                if ind == len(lcs):
                    lcs.append(loc[x])
                else:
                    lcs[ind] = loc[x]

        return len(target) - len(lcs) + 1
