class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        d = {num: i for i, num in enumerate(target)}
        a = []
        for num in arr:
            if num in d:
                idx = bisect.bisect_left(a, d[num])
                if idx == len(a):
                    a.append(d[num])
                else:
                    a[idx] = d[num]
        return len(target) - len(a)
