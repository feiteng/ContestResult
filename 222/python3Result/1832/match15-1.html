class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        mapper = {}
        finalarr=[]
        for i,num in enumerate(target):
            mapper[num]=i
        for i in range(len(arr)):
            if arr[i] in mapper:
                finalarr.append(mapper[arr[i]])
        n=len(finalarr)
        if n==0:
            return len(target)
        tt= [0 for i in range(n)]
        tt
[0] = finalarr[0]
        ll=1
        for i in range(1, n): 
            if (finalarr[i] < tt[0]): 
                tt[0] = finalarr[i] 
            elif (finalarr[i] > tt[ll-1]): 
                tt[ll] = finalarr[i] 
                ll+= 1
            else: 
                tt[posfinder(tt, -1, ll-1, finalarr[i])] = finalarr[i]
        return len(target)-ll;
