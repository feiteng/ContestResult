class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        target = {j: i for i, j in enumerate(target)}
        nums = [(-1 if i not in target else target[i]) for i in arr]
        d = []
        for i in nums:
            if i == -1:
                continue
            if not d or i > d[-1]:
                d.append(i)
            else:
                l, r = 0, len(d) - 1
                loc = r
                while l <= r:
                    mid = (l + r) // 2
                    if d[mid] >= i:
                        loc = mid
                        r = mid - 1
                    else:
                        l = mid + 1
                d[loc] = i
        return len(target)-len(d)
