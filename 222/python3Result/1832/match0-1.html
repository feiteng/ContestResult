class Solution:
    def minOperations(self, target: List[int], arr: List[int]) -> int:
        mapping = {num:i for i, num in enumerate(target)}
        stack = []
        for num in arr:
            if num in mapping:
                ind = bisect.bisect_left(stack, mapping[num])
                if ind == len(stack):
                    stack.append(mapping[num])
                else:
                    stack[ind] = mapping[num]
        return len(target)-len(stack)
