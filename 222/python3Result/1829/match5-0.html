class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x: -x[1])
        ans = 0
        for n, c in boxTypes:
            if truckSize >= n:
                ans += n * c
                truckSize -= n
            else:
                ans += truckSize * c
                break
        return ans
