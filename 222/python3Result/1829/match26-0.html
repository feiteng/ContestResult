class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key = lambda x: x[1], reverse = True)
        ans = 0
        for n, v in boxTypes:
            m = min(n, truckSize)
            ans += m * v
            truckSize -= m
        return ans
