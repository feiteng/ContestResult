class Solution:
    def maximumUnits(self, A: List[List[int]], size: int) -> int:
        A.sort(key = lambda x: x[1], reverse=True)
        ans = 0
        for cnt, num in A:
            if cnt <= size:
                ans += num * cnt
                size -= cnt
            else:
                ans += num * size
                break
        return ans
