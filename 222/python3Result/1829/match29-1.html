class Solution:
    def maximumUnits(self, box: List[List[int]], s: int) -> int:
        box.sort(key=lambda x: (x[1],x[0]),reverse=True)
        
        res = 0
        for k,v in box:
            if s==0:
                break
            res+=min(k,s)*v
            s-=min(k,s)
        return res
