class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key = lambda x: -x[1])
        res = 0 
        capacity = truckSize
        for q, u in boxTypes:
            
            if capacity > q:
                res += q * u 
                capacity -= q
            else:
                res += capacity * u
                return res 
        
        return res
