class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x:-x[1])
        ans=0
        for n, u in boxTypes:
            cur=min(n, truckSize)
            ans+=u*cur
            truckSize-=cur
            if truckSize==0:
                break
        return ans
