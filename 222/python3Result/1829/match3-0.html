class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x:-x[1])
        ans=0
        for a,b in boxTypes:
            if truckSize>=a:
                ans+=a*b
                truckSize-=a
            else:
                ans+=truckSize*b
                break
        return ans
