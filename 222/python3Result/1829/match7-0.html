class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x:-x[1])
        ttl = 0
        for num, val in boxTypes:
            if truckSize >= num:
                ttl += val * num
                truckSize -= num
            else:
                ttl += truckSize * val
                break
        return ttl
