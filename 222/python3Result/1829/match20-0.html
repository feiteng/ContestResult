class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x: x[1], reverse=True)
        res = 0
        for count, unit in boxTypes:
            if count < truckSize:
                truckSize -= count
                res += unit * count
            else:
                res += unit * truckSize
                return res
        
        return res
