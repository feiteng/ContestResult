class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x:-x[1])
        #boxTypes.reverse()
        res = 0
        num = 0
        i = 0
        while num < truckSize and i < len(boxTypes):
            n = min(truckSize-num, boxTypes[i][0])
            res += n * boxTypes[i][1]
            num += n
            i += 1
        return res
