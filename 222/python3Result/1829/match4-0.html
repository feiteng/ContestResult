class Solution:
    def maximumUnits(self, a: List[List[int]], n: int) -> int:
        a.sort(key=lambda x: -x[1])
        ans = 0
        for k, v in a:
            if n >= k:
                ans += k * v
                n -= k
            else:
                ans += n * v
                break
        return ans
