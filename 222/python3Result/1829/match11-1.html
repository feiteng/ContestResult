class Solution:
    def maximumUnits(self, bx: List[List[int]], tk: int) -> int:
        bx.sort(key=lambda x:-x[1])
        res = 0
        for x, y in bx:
            mx = min(x, tk)
            res += mx * y
            tk -= mx
            if tk <= 0:
                break
        return res
