class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        
        boxTypes.sort(key=lambda x:x[1],reverse=True)
        
        res=0
        
        cur=0
        while truckSize>0 and cur<len(boxTypes):
            
            added=min(truckSize,boxTypes[cur][0])
            
            res+=added*boxTypes[cur][1]
            
            truckSize-=added
            cur+=1
            
        return res
