const int maxn = 1<<20;
int a[maxn+5];
const int mod = 1e9+7;
class Solution {
public:
    
    int countPairs(vector<int>& d) {
        memset(a, 0, sizeof(a));
        int n = d.size();
        a[d[0]]++;

        long long ans =0;
        for(int i=1; i<n; i++) {
            for(int s=1; s<=(1<<21); s<<=1) {
                int t = s-d[i];
                if(t<0 || t>(1<<20)) continue;
                ans = (ans+a[t])%mod;
            }
            a[d[i]]++;
        }
        return ans;
    }
};
