class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        unordered_map<int, int> m;
        long long ans = 0;
        sort(deliciousness.begin(), deliciousness.end());
        for (auto x : deliciousness) {
            for (int i = 0; i <= 21; i++) {
                int pow = (1 << i);
                if (pow >= x && pow/2 <= x) {
                    ans += m[pow-x];
                }
            }
            m[x]++;
        }
        return ans %= 1000000007;
    }
};
