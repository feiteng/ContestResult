class Solution {
    #define MOD 1000000007
public:
    vector<int> cnt;
    int countPairs(vector<int>& a) {
        if (cnt.size() == 0)
            cnt = vector<int>((1<<21)+1);
        int n = a.size();
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= 21; j++) {
                if ((1<<j) >= a[i]) {
                    ans += cnt[(1<<j) - a[i]];
                    ans %= MOD;
                }
            }
            cnt[a[i]]++;
        }
        for (int i = 0; i < n; i++) {
            cnt[a[i]]--;
        }
        return ans;
    }
};
