#define LL long long
const int mod = 1e9+7;
class Solution {
public:
    int countPairs(vector<int>& w) {
        unordered_map<int, int> cnt;
        LL ans = 0;
        for(auto x: w){
            for(int i=0; i<=22; i++){
                if(cnt.count((1<<i) - x)){
                    ans += cnt[(1<<i) - x];
                    ans %= mod;
                }
            }
            cnt[x]++;
        }
        return ans%mod;
    }
};
