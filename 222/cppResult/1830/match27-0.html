class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        int power2[22];
        int temp = 1;
        for (int i = 0; i < 22; i++) {
            power2[i] = temp;
            temp *= 2;
        }
        unordered_map<int, int> mealCount;
        long res = 0;
        for (int i = 0; i < deliciousness.size(); i++) {
            for (int j = 21; j >= 0 && power2[j] >= deliciousness[i]; j--) {
                int diff = power2[j] - deliciousness[i];
                if (mealCount.count(diff) > 0) {
                    res += mealCount[diff];
                    res %= 1000000007;
                } 
            }
            if (mealCount.count(deliciousness[i]) > 0) {
                mealCount[deliciousness[i]]++;
            }
            else
                mealCount[deliciousness[i]] = 1;
        }
        return int(res);
    }
};
