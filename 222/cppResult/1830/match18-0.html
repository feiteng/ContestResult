const int maxn = (1 << 21) + 10;
int mp[maxn];
const int MOD = 1e9 + 7;

class Solution {
public:
    int countPairs(vector<int>& nums) {
        for (int i = 0; i < maxn; i++)  mp[i] = 0;
        int n = nums.size();
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= 21; j++) {
                int d = (1 << j) - nums[i];
                if (d < 0 || d >= maxn) continue;
                ans += mp[d];
                ans %= MOD;
            }
            mp[nums[i]]++;
        }
        return ans;
    }
};
