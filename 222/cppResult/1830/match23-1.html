class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        const int MOD = 1e9 + 7;
        const int M = 1e6 + 10;
        vector<int> cnt(8 * M);

        int ans = 0;
        for (int i = 0; i < deliciousness.size(); i++) {
            for (int j = 0; j < 22; j++) {
                int goal = (1 << j) - deliciousness[i];
                if (goal >= 0) {
                    ans = (ans + cnt[goal]) % MOD;
                }
            }
            cnt[deliciousness[i]]++;
        }
        return ans;
    }
};
