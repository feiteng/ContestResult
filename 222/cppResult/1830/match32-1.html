class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        int MOD = 1e9 + 7;
        int mp[1 << 22];
        memset(mp,0,sizeof mp);

        int ans = 0;
        for(int i = 0;i < deliciousness.size();i++){
            for(int pw = 1;pw <= (1 << 21);pw = pw * 2){
                int need = pw - deliciousness[i];
                if(need < 0)continue;
                ans = ans + mp[need];
                ans %= MOD;
            }
            mp[deliciousness[i]]++;
        }
        return ans;
    }
};
