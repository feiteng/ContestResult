class Solution {
public:
    const int mod = 1e9 + 7;
    int countPairs(vector<int>& d) {
        unordered_map<int,int> mp;
        int ans = 0;
        for(int v : d){
            for(int i=0; i<= 21; ++i){
                int pw = (1<<i);
                if(mp.count(pw - v)){
                    ans += mp[pw - v];
                    ans %= mod;                    
                }
            }
            mp[v]++;
        }
        return ans;
    }
};
