class Solution {
public:
    int countPairs(vector<int>& de) {
        unordered_map<int,int> m;
        int n = de.size();
        long ans = 0;
        int KINT = 1e9+7;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < 31; j++){
                int sum = pow(2,j);
                if(sum>=de[i]&&m.count(sum-de[i])){
                    ans = (ans+m[sum-de[i]])%KINT;
                }
                
            }
            m[de[i]]++;
        }
        return ans;
    }
};
