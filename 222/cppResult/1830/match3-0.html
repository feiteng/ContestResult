class Solution {
public:
    int countPairs(vector<int>& nums) {
        // sort(nums.begin(), nums.end());
        int mod = 1e9 + 7;
        int res = 0;
        unordered_map<int, int> m;
        int n = nums.size();
        for(int i = 0; i < n; i ++) {
            for(int j = 0; j <= 21; j ++) {
                int t = (1 << j) - nums[i];
                // if(nums[i] == 107) cout << t << " " << m.count(t) << endl;
                if(m.count(t)) {
                    res += m[t];
                    res %= mod;
                    // cout << nums[i] << ' ' << t << ' ' << m[t] << endl;
                }
            }
            m[nums[i]]++;
        }
        return res;
    }
};
