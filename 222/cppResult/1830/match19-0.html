class Solution {
public:
    int vis[4000006];
    int mod = 1e9 + 7;
    int countPairs(vector<int>& a) {
        int res = 0;
        sort(a.begin(),a.end());
        for(auto it : a){

            for(int i = 0;i <= 21;++i){
                if((1 << i) - it >= 0) 
                    res = (res + vis[(1 << i) - it]) % mod;
            }
            vis[it]++;
        }
        return res;
    }
};
