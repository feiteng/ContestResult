class Solution {
public:
    const int MOD = 1e9+7;
    int countPairs(vector<int>& nums) {
        unordered_map<int, int> m;
        int res = 0;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 1; j < pow(2, 22); j*=2){
                if (m.count(j - nums[i])) {
                    res = (res + m[j-nums[i]]) %MOD;
                }
            }
            m[nums[i]]++;
        }
        return res;
    }
};
