class Solution {
public:
	const static int mod=1000000007;
	const static int mod2=500000004;
	unordered_map<int,int>cnt;
    int countPairs(vector<int>& deliciousness) {
		int n=deliciousness.size();
		for(int i=0;i<n;i++)cnt[deliciousness[i]]++;
		int ans1=0,ans2=0;
		for(int i=0;i<n;i++){
			for(int k=0;k<=21;k++){
				int rem=(1<<k)-deliciousness[i];
				if(rem<0||rem<deliciousness[i])continue;
				if(rem==deliciousness[i])ans1=(ans1+cnt[rem]-1)%mod;
				else ans2=(ans2+cnt[rem])%mod;
			}
		}
		ans1=(ans1+mod)%mod;
		ans1=(1ll*ans1*mod2)%mod;
		return (ans1+ans2)%mod;
    }
};
