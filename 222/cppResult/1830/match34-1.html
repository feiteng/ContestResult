const int mod = 1e9 + 7;
const int N = 32;
int c[N];
class Solution {
public:
    int countPairs(vector<int>& d) {
        unordered_map<int, int> mp;
        c[1] = 1;
        for(int i = 2; i <= 30; ++i)
        {
            c[i] = c[i - 1] * 2;
            //cout << c[i] << endl;
        }

        int res = 0;
        for(int i = 0; i < d.size(); ++i)
        {
            for(int j = 1; j <= 30; ++j)
            {
                //cout << c[j] << endl;
                if(mp.find(c[j] - d[i]) != mp.end())
                {
                    //cout << c[j] - d[i] << endl;
                    res = ((res + mp[c[j] - d[i]]) % mod + mod) % mod;
                }
            }
            ++mp[d[i]];
        }
        return res;
    }
};
