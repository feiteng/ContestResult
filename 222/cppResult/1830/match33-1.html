class Solution {
public:
    const int mod = 1000000000 + 7;
    int countPairs(vector<int>& A) {
        int n = A.size();
        
        int L = (1 << 21) + 10;
        vector<int> M(L, 0);
        int ans = 0;

        for (int i = 0; i < n; i++) {
            int k = A[i];
            
            for (int s = 21; s >= 0; s--) {
                int S = (1 << s);
                
                if (S < k) break;
                
                ans = (ans + M[S-k]) % mod;
            }
            M[k]++;
        }
        
        return ans;
    }
};
