#define mod 1000000007
typedef long long int lli;

class Solution {
public:
    int countPairs(vector<int>& vt) {
        int fr[(1<<21)+2]={0};
        int ans=0;
        for(int i=0;i<vt.size();i++){
            for(int j=0;j<=21;j++){
                lli rem=(1ll<<j) - vt[i];
                if(rem>=0)
                {
                    ans+=fr[rem];
                    ans%=mod;
                }
            }
            fr[vt[i]]++;
        }
        return ans;
    }
};
