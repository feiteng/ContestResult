class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        unordered_map<int,int> p;
        int i,j,n=deliciousness.size(),ans=0,mod=1e9+7;
        p[deliciousness[0]]++;
        for(i=1;i<n;++i){
            for(j=21;j>=0;--j){
                if(p.count((1<<j)-deliciousness[i])){
                    ans+=p[(1<<j)-deliciousness[i]];
                    ans=ans%mod;
                }
            }
            p[deliciousness[i]]++;
        }
        return ans;
    }
};
