class Solution {
public:
    const int MOD = 1e9 + 7;
    int countPairs(vector<int>& d) {
        sort(d.begin(), d.end());
        unordered_map<int, int> re;
        int n = d.size(), ans = 0;
        for (int k = 0; k < n; ++k) {
            for (int b = 21; b >= 0 && (1 << b) >= d[k];--b) {
                int u = (1 << b) - d[k];
                if (re.count(u)) ans += re[u];
            }
            if (ans >= MOD) ans -= MOD;
            ++re[d[k]];
        }
        return ans;
    }
    
};
