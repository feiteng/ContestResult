class Solution {
    #define ll long long
public:
    int countPairs(vector<int>& de) {
        ll i,j,k,n=de.size();
        int M[(1<<21)+1]={0};

        ll ans=0;
        //unordered_map<ll,ll>M;
        for(i=0;i<n;i++){
            for(j=1;j<=(1<<21);j<<=1){
                if(de[i]<=j){
                    ans+=M[j-de[i]];
                    ans%=1000000007;
                }
            }
            M[de[i]]++;
            //cout<<i<<' '<<ans<<' '<<M[1]<<endl;
        }
        return ans;
    }
};
