class Solution {
public:
    int countPairs(vector<int>& v) {
        vector<int> powers;
        int sofar = 1;
        powers.push_back(1);
        for(int i = 1; i <= 21; i++) {
            sofar = sofar * 2;
            powers.push_back(sofar);
        }
        
        int cnt[1048577] = {0};

        long long ans = 0;
        for(int i = 0; i < v.size(); i++) {
            for(int j = 0; j < powers.size(); j++) {
                int rem = powers[j] - v[i];
                if(rem >= 0 && rem <= 1048576)
                ans = (ans + cnt[rem]) % 1000000007;
            }
            cnt[v[i]]++;
        }
        
        return ans;
    }
};
