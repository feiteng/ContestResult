class Solution {
public:
    int countPairs(vector<int>& deliciousness) {

        vector<int>v(3000006,0);
        int ans=0;
        int mod=1e9+7;
        for(int i=0;i<deliciousness.size();i++){
            for(int j=0;j<=21;j++){
                if((1<<j)-deliciousness[i]>=0){
                    ans+=v[(1<<j)-deliciousness[i]];
                    ans%=mod;
                }
            }
            v[deliciousness[i]]++;
        }
        return ans;
    }
};
