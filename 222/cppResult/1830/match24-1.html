class Solution {
public:
    int countPairs(vector<int>& d) {
        int res = 0, mod = 1e9 + 7;
        unordered_map<int, int> cnt;
        for (auto& x: d) {
            int sum = 1;
            for (int i = 0; i <= 21; i ++) {
                int y = sum - x;
                if (cnt.count(y)) res = (res + cnt[y]) % mod;
                sum *= 2;
            }
            cnt[x] ++;
        }
        return res;
    }
};
