class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        int mod=1e9+7;
        int n=deliciousness.size();
        unordered_map<int,int> mp;
        long ans=0;

        for(int i: deliciousness){
            for(int j=0;j<=21;j++){
                int sum=1<<j;
                if(mp.count(sum-i)){
                    ans+=mp[sum-i];
                    ans%=mod;
                }
            }
            mp[i]++;
        }
        return ans;
    }
};
