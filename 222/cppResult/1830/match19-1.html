class Solution {
public:
    #define MN 22
    int f[1<<MN];
    int countPairs(vector<int>& A) {
        memset(f,0,sizeof(f));
        sort(A.begin(),A.end());

        int ans = 0;
        for(auto a:A){
            for(int i=0;i<=21;i++){
                if((1<<i)-a>=0)
                    ans = (ans+f[(1<<i)-a])%1000000007;
            }
            f[a]++;
        }
        return ans;
    }
};
