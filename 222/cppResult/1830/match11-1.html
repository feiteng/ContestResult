class Solution {
public:
    int countPairs(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        int maxVal = nums[n - 2] + nums[n - 1];
        unordered_map<int, int> m;

        long long res = 0;
        for (int num : nums) {
            for (int sum = 1; sum <= maxVal; sum *= 2) {
                int target = sum - num;
                if (m.count(target)) {
                    res += m[target];
                }
            }
            m[num]++;
        }
        return res % 1000000007;
    }
};
