class Solution {
public:
    const int mod=1e9+7;
    int countPairs(vector<int>& deliciousness) {

        unordered_map<int,int> m;
                int ans=0;
        for(auto it:deliciousness){ 
            
            for(int j = 0;j<22;j++){
                if(it<=(1<<j)){
                    ans+=m[(1<<j)-it];
                    ans%=mod;
                }
            }
            m[it]++;
        }
        return ans;
    }
};
