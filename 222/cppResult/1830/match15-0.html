class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        int ans = 0, mod = 1e9+7;
        unordered_map<int, int> mp;
        constexpr int up = 1<<21;
        for(auto v: deliciousness) {
            for(int i = 1; i <= up; i <<= 1) {
                if(i >= v && mp.find(i-v) != mp.end()) {
                    ans += mp[i-v];
                    ans %= mod;
                }
            }
            mp[v]++;
        }
        return ans;
    }
};
