class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        unordered_map<int, long long> count;
        for(int i: deliciousness) {
            count[i]++;
        }

        long long res = 0;
        for(auto &p: count) {
            int v = p.first;
            long long n = p.second;
            for(int i=0; i<31; i++) {
                int pow2 = 1<<i;
                if(pow2<v) continue;
                int t = pow2 - v;
                if(t<v) continue;
                if(t==v) {
                    res = (res + n*(n-1)/2) % 1000000007;
                } else if(count.count(t)) {
                    res = (res + n*count[t]) % 1000000007;
                }
            }
        }
        return res;
    }
};
