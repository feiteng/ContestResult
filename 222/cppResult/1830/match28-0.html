class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        long res = 0, mod = 1e9+7;
        unordered_map<int, int>_map;
        for(int n : deliciousness){
            for(int i = 0;i < 31;++i){
                int v = 1 << i;
                if(_map.count(v - n)){
                    res += _map[v - n];
                    res %= mod;
                }
            }   
            ++_map[n];
        }
        return res;
    }
};
