class Solution {
public:
    const int M = 1e9 + 7;
    int countPairs(vector<int>& d) {
        //d.clear();
        //for (int i = 0; i < 1e5; ++i) d.push_back(i);
        unordered_map<int,int> mp;
        int mx = 0;
        vector<int> d2;
        for (int i = 0; i < d.size(); ++i) {
            mp[d[i]]++;
            if (mp[d[i]]==1) d2.push_back(d[i]);
            mx=max(mx,d[i]);
        }
        sort(d2.begin(),d2.end());

        int ans = 0;
        for (int i = 1; i<=mx*2; i*=2) {
            //map<int,bool> flag;
            for (int j = 0; j < d2.size(); ++j) {
                int a = d2[j];
                int b = i - a;
                if (a<=b) {
                    //flag[a]=true;
                    if (a<b) {
                        ans = (ans + (long long)mp[a] * mp[b])%M;
                    } else {
                        ans = (ans + (long long)mp[a] * (mp[b]-1) / 2)%M;
                    }
                }
                if (a>=b) break;
            }
            //cout<<i<<" "<<ans<<endl;
        }
        return ans;
    }
};
