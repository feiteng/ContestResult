int mod = 1000000007;
class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        unordered_map<int,int>mp;
        int res = 0;
        for(auto a : deliciousness){
            for(int i = 0 ; i <= 21 ; ++i){
                if(1 << i < a) continue;
                int dx = (1 << i) - a;
                res = (res + mp[dx]) % mod;
            }
            mp[a]++;
        }
        return res;
    }
};
