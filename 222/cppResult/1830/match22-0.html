class Solution {
    int cnt[1 << 20 + 1];
    int mo=1000000007;
public:
    int countPairs(vector<int>& d) {
        int ret=0;
        for(int i=0;i<d.size();i++) {
            for(int j=0;j<=21;j++) {
                long long x = (1ll << j) - d[i];
                if (x <= (1 << 20) && x>=0) {
                    ret+=cnt[x];
                    ret=ret % mo;
                }
            }
            cnt[d[i]]++;
        }
        return ret;
    }
};
