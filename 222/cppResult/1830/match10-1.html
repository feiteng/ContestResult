class Solution {
public:
    const int M = 1e9 + 7;
    
    int countPairs(vector<int>& deliciousness) {
        unordered_map<int, int> bucket;
        int ans = 0;
        for (int x : deliciousness) {
            for (int i = 0; i < 22; ++i) {
                if ((1 << i) < x) continue;
                ans+= bucket[(1 << i) - x];
                ans %= M;
            }
            bucket[x]++;
        }
        return ans;
    }
};
