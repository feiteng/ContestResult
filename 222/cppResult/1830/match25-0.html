class Solution {
    #define ll long long
    const int mod = 1e9+7;
public:
    int countPairs(vector<int>& a) {
        ll n = a.size();
        ll m[(1 << 22) + 1] = {0};
        m[a[0]] = 1;
        ll ans = 0;
        for(int i = 1 ; i < n ; i++){
            for(int j = 0 ; j < 22 ; j++){
                if((1ll << j) >= a[i]){
                    ans += m[(1ll << j) - a[i]];
                    ans %= mod;
                }
            }
            m[a[i]]++;
        }
        return ans;
    }
};
