class Solution {
public:
    int countPairs(vector<int>& deliciousness) {
        const int MOD = 1e9 + 7;
        int n = static_cast<int>(deliciousness.size());

        unordered_map<int, int> cnt;
        int pairs = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < 22; ++j) {
                int total = (1 << j);
                if (total >= deliciousness[i]) {
                    (pairs += cnt[total - deliciousness[i]]) %= MOD;
                }
            }
            ++cnt[deliciousness[i]];
        }
        return pairs;
    }
};
