class Solution {
public:
    
    
    int countPairs(vector<int>& deliciousness) {
        
        vector<int>& a = deliciousness;

        vector<int> cnt((1<<20)+10, 0);
        long long ans = 0;
        for (auto i: a) {
            for (int j = 0; j <= 21; j++) {
                int x = (1<<j)-i;
                if (x >= 0 && x <= (1<<20)) {
                    ans += cnt[x];
                }
            }
            cnt[i]++;
        }
        return ans % (1000000007);
        
    }
};
