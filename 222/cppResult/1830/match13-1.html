class Solution {

    const int M = 1000000000 + 7;
public:
    int countPairs(vector<int>& deliciousness) {
        int n = deliciousness.size();
        unordered_map<int, int> cnt;
        vector<int> pow2;
        int ret = 0;
        for (int i = 0; i <= 21; ++ i)
            pow2.push_back(1 << i);


        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < pow2.size(); ++ j) {
                if (cnt.count(pow2[j] - deliciousness[i])) {
                    ret = (ret + cnt[pow2[j] - deliciousness[i]]) % M;
                }
            }
            cnt[deliciousness[i]] ++;
        }
                
        return ret;
    }
};
