class Solution {
public:
    const int MOD = 1000000007;
    int countPairs(vector<int>& v) {
        int res = 0;
        int sz = 1<<21;
        vector<int> upd(sz+1,-1);
        vector<int> cnt(sz+1,0);
        for(int i=21; i>=0; i--){
            int num = (1<<i);
            for(int k=0; k<v.size(); k++){
                if(num >= v[k] && upd[num-v[k]] != i){
                    upd[num-v[k]] = i;
                    cnt[num-v[k]] = 0;
                }
                if(num >= v[k]) res += cnt[num-v[k]];
                if(res >= MOD) res -= MOD;
                if(upd[v[k]] != i){
                    upd[v[k]] = i;
                    cnt[v[k]] = 0;
                }
                cnt[v[k]]++;
            }
        }
        return res;
    }
};
