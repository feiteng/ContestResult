#define mod 1000000007
class Solution {
public:
    int countPairs(vector<int>& a) {
        int n = a.size(), ans = 0;
        unordered_map<int, int> g;
        for(auto &x : a){
            int w = 1;
            for (int i = 0; i <= 21; i++) {
                if (g.count(w - x)){
                    ans = (ans + g[w-x]) % mod;
                }
                w <<= 1;
            }
            g[x]++;
        }
        return ans;
    }
};
