class Solution {
public:
    int sum[100005];
    int mod=(int)1e9+7;
    int waysToSplit(vector<int>& nums) {
        for(int i=0;i<=100000;i++) sum[i]=0;
        for(int i=0;i<nums.size();i++){
            sum[i+1]=sum[i]+nums[i];
        }
        int n=nums.size();
        long long ans=0;
        for(int i=2;i<n;i++){
            int sumr=sum[n]-sum[i];
            int l,r,mid,ansl=-1;
            l=1,r=i-1;
            while(l<=r){
                mid=(l+r)>>1;
                if(sum[i]-sum[mid]<=sumr){
                    ansl=mid,r=mid-1;
                }else l=mid+1;
            }
            l=1,r=i-1;
            int ansr=-1;
            while(l<=r){
                mid=(l+r)>>1;
                if(sum[mid]<=sum[i]-sum[mid]){
                    ansr=mid,l=mid+1;
                }else r=mid-1;
            }
            if(ansl<=ansr && ansl>0 && ansr>0){
                ans+=ansr-ansl+1;
                ans%=mod;
            }
        }
        return ans;
    }
};
