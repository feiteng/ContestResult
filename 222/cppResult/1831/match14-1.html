class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        long ans = 0;
        int KINT = 1e9+7;
        vector<int> sum(n,0);
        sum[0] = nums[0];

        for(int i = 1; i < n; i++){
            sum[i] = sum[i-1]+nums[i];
        }
        for(int i = 0; i < n-2; i++){
            int l = i+1;
            int r = n;

            while(l<r){
                int m = (l+r)/2;
                int cur = sum[m]-sum[i];
                if(cur<sum[i]){
                    l = m+1;
                }else{
                    r = m;
                }
            }
            int ll = l;

            l = i+2;
            r = n;
            while(l<r){
                int m = (l+r)/2;
                int sum1 = sum[m-1]-sum[i];
                int sum2 = sum[n-1] - sum[m-1];
                if(sum1 <= sum2){
                    l = m+1;
                }else{
                    r = m;
                }
            }
            int rr = l-1;
            //cout<<ll<<" "<<rr<<endl;
            if(rr>ll)ans = (ans + (rr-ll))%KINT;
        }
        return ans;
    }
};
