class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> cumsum(n+1);
        cumsum[0] = 0;
        for (int i = 0; i < n; i++)
            cumsum[i+1] = cumsum[i] + nums[i];
        
        int ans = 0;
        for (int lm = 1; lm < n; lm++) {
            int left = cumsum[lm];
            int lb = lower_bound(cumsum.begin()+lm+1, cumsum.end()-1, 2 * left) - cumsum.begin();
            int ub = upper_bound(cumsum.begin()+lm+1, cumsum.end()-1, left + (cumsum[n] - left) / 2) - cumsum.begin();
            if (ub < lb)
                break;
            ans = (ans + ub - lb) % 1000000007;
        }
        return ans;
    }
};
