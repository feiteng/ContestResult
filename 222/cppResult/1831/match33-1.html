class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        //二分？
        int n=nums.size();
        int sum[n+1];
        sum[0]=0;
        for(int i=0;i<n;i++)sum[i+1]=sum[i]+nums[i];

        long long ans=0;
        //sum是单调递增的
        for(int i=1;i<n-1&&sum[i]<=sum[n]/3;i++){
            int l=i+1;
            int r=n-1;
            int later=-1;
            int pre=-1;
            int d1=(sum[n]-sum[i])/2+sum[i];//取下整，偏小的
            for(;l<=r;){
                int m=(l+r)/2;
                if(sum[m]<=d1)l=m+1;
                if(sum[m]>d1)r=m-1;
            }
            later=l;
            int d2=2*sum[i];
            l=i+1,r=n-1;
            for(;l<=r;){
                int m=(l+r)/2;
                if(sum[m]<d2)l=m+1;
                if(sum[m]>=d2)r=m-1;
            }  
            pre=l;
            ans+=later-pre;
        }
        //cout<<ans<<endl;
        return ans%1000000007;

    }
};
