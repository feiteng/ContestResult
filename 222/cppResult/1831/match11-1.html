class Solution {
public:
    int waysToSplit(vector<int>& nums) {

        for (int i = 1; i < nums.size(); ++i) {
            nums[i] += nums[i-1];
        }
        int rv = 0;
        for (int i = 0; i < nums.size() - 2; ++i) {
            int start = i+1; 

            int end = nums.size()-2;
            while (start <= end) {
                int mid = (start+end)/2;
                if (nums[mid] - nums[i] < nums[i]) {
                    start = mid + 1;
                } else {
                    end = mid - 1;
                }
            }
            int l = start;
            
            

            start = l; 
            end = nums.size()-2;
            while (start <= end) {
                int mid = (start+end)/2;
                if (nums.back() - nums[mid] < nums[mid] - nums[i]) {
                    end = mid - 1;
                } else {
                    start = mid + 1;
                }
            }
            int r = end;
            if (r >= l) {
                rv += r - l + 1;
            }
            rv %= 1000000007;
        }
        return rv;
    }
};
