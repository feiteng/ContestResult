class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        vector<int> sums(nums);

        for (int i = 1; i < sums.size(); ++i) {
            sums[i] += sums[i - 1];
        }
        constexpr int kPrime = 1e9 + 7;
        int result = 0;
        for (int j = 1; j + 1 < nums.size(); ++j) {
            int third = sums.back() - sums[j];
            if (sums[j] > 2 * third) {
                break;
            }
            if (sums[j] - sums[j - 1] > third) {
                continue;
            }

            int p = 0, r = j - 1;
            while (p < r) {
                int q = p + (r - p) / 2;
                if (sums[j] - sums[q] <= third) {
                    r = q;
                } else {
                    p = q + 1;
                }
            }
            int min_idx = p;
            if (sums[min_idx] > sums[j] - sums[min_idx]) {
                continue;
            }

            r = j - 1;
            while (p < r) {
                int q = r - (r - p) / 2;
                if (sums[q] <= sums[j] - sums[q]) {
                    p = q;
                } else {
                    r = q - 1;
                }
            }
            result = (result + (r - min_idx + 1)) % kPrime;
        }
        return result;
    }
};
