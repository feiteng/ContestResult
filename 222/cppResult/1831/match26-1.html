class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        vector<int> prefix_sum(nums.size(), 0);
        prefix_sum[0] = nums[0];

        for(int i=1;i<nums.size();i++){
            prefix_sum[i] = prefix_sum[i-1]+nums[i];
        }
        int all = prefix_sum[nums.size()-1];
        int ret = 0;
        for(int i=0;i<nums.size();i++){
            //find x such that prefix_sum[x]-prefix_sum[i]>=prefix_sum[i] and all-prefix_sum[x]>=prefix_sum[x]-prefix_sum[i]
            int a, b;

            int low = i+1;
            int high = nums.size()-1;
            while(low<high){
                int mid = low+(high-low)/2;
                if(prefix_sum[mid]-prefix_sum[i]>=prefix_sum[i]) high = mid;
                else low = mid+1;
            }
            a = high;

            low = i+1;
            high = nums.size()-1;
            while(low<high){
                int mid = low+(high-low)/2;
                if(all-prefix_sum[mid]<prefix_sum[mid]-prefix_sum[i]) high = mid;
                else low = mid+1;
            }
            b = high;
            if(b>a) ret += (b-a);
            ret %= 1000000007;
            //printf("i=%d b=%d a=%d\n", i, b, a);
        }
        return ret;
    }
};
