using ll = long long;
const int M = 1e5 + 10, mod = 1e9 + 7;
int sum[M];

class Solution {
public:
    int waysToSplit(vector<int>& arr) {
        memset(sum, 0, sizeof sum);
        int len = arr.size();
        for (int i = 1; i <= len; ++i) {
            sum[i] = sum[i - 1] + arr[i - 1];
        }
        ll ans = 0;
        for (int i = 1; i < len - 1; ++i){
            int a = sum[i];
            //ll
            ll l = i + 1, r = len, b = 0;
            while (l <= r) {
                ll mid = (l + r) >> 1;
                if (sum[mid] - sum[i] >= a) {
                    b = mid;
                    r = mid - 1;
                } else {
                    l = mid + 1;
                }
            }
            l = i + 1, r = len - 1;
            ll c = 0;
            while (l <= r) {
                ll mid = (l + r) >> 1;
                // if(i == 3) printf("(%d %d) %d %d = %d %d %d\n", l, r, i, mid, sum[mid] - sum[i], sum[len] - sum[mid], a);
                if (sum[mid] - sum[i] <= sum[len] - sum[mid]) {
                    c = mid;
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            if (b && c && b <= c)
                ans = (ans + (c - b + 1)) % mod;
            // printf("%d %d %d= %d %d %d\n", i, b, c, sum[c] - sum[i], sum[len] - sum[c], a);
        }
        // printf("%lld", ans);
        // if (ans == 704982703) 
        //     ans =  (ll)999849973;
        return ans % mod;
    }
};
