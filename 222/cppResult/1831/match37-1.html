#define MOD 1000000007


class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        vector<int> dp(n,0);
        
        for(int i=0;i<n;i++){
            dp[i]=(i?dp[i-1]:0)+nums[i];
            // cout<<dp[i]<<',';
        }
        

        int ans=0;
        
        for(int i=0;i<n-2;i++){
            int a = dp[i];
            
            int b_min = lower_bound(dp.begin()+i+1,dp.end(),2*a)- dp.begin();
            int b_max = upper_bound(dp.begin()+i+1,dp.end(),(a+dp[n-1])/2)- dp.begin()-1;
            
            b_min = max(1, b_min);
            b_max = min(b_max, n-2);
            
            // cout<<" i"<<i<<" min"<<b_min<<" max"<<b_max<<endl;
            ans+=max(0,(b_max - b_min +1))%MOD;
            ans = ans%MOD;
        }
        
        // cout<<endl;
        return ans%MOD;
    }
};
