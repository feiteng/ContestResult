class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        vector<int> cumsum(nums.size());
        cumsum[0] = nums[0];
        for (int i = 1; i < nums.size(); ++i) cumsum[i] = nums[i] + cumsum[i - 1];
        

        int ans = 0;
        for (int i = 0; i < nums.size(); ++i) {
            
            int rem = cumsum.back() - cumsum[i];
            
            int min_idx = lower_bound(cumsum.begin() + i + 1, cumsum.end(), 2 * cumsum[i]) - cumsum.begin();
            int max_idx = upper_bound(cumsum.begin() + i + 1, cumsum.end(), cumsum[i] + rem / 2) - cumsum.begin();
            if (max_idx == nums.size()) --max_idx;
            // cout << i << ": " << min_idx << " " << max_idx << endl;
            if (max_idx > min_idx) {
                ans += max_idx - min_idx;
                ans %= 1000000007;
            }
        }
        return ans;
    }
};
