class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        int ans=0;
        const int MOD = 1e9+7;
        int pre[n];
        pre[0]=nums[0];
        for(int i=1;i<n;i++)

            pre[i] = pre[i-1]+nums[i];
        for(int i=0;i<n;i++)
        {
            int left = pre[i];
            int lo=i+1,hi=n-1,mid,aa=-1;
            while(lo<=hi)
            {
                mid=lo+(hi-lo)/2;
                if(pre[mid]>=2*left)
                {
                    aa=mid;
                    hi=mid-1;
                }
                else
                    lo=mid+1;
            }
            lo=i+1;hi=n-2;int bb=-1;
            while(lo<=hi)
            {
                mid=lo+(hi-lo)/2;
                if(2*pre[mid]<=pre[i] + pre[n-1])
                {
                    bb=mid;
                    lo=mid+1;
                }
                else
                    hi=mid-1;
            }
            // cout<<aa<<" "<<bb<<" "<<i<<endl;
            if(bb>=aa && aa!=-1 && bb!=-1)
            {
                ans+=(bb-aa+1);
                ans%=MOD;
            }
        }
        return ans;
    }
};
