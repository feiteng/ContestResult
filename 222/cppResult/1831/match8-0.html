const int mod = 1e9+7;

class Solution {
public:
    int waysToSplit(vector<int>& s) {
        int n = s.size();
        for(int i=1; i<n; i++) s[i]+=s[i-1];
        int S = s.back();
        long long ans = 0;
        for(int j=1; j<n-1 && 3*s[j]<=2*S; j++) {
            int r = upper_bound(s.begin(), s.begin()+j, s[j]/2) - s.begin(); // s[i]<=s[j]/2
            int l = lower_bound(s.begin(), s.begin()+j, 2*s[j]-S) - s.begin(); // 2*s[j]-S<=s[i] 
            //cout<<l<<" "<<r<<endl;
            ans = (ans+ r-l)%mod;
        }
        return ans;
    }
};
