// p[n] - p[j] > p[j] - p[i], p[j] - p[i] >= p[i] -> p[i] >= 2*p[j] - p[n], p[i] <= p[j] / 2;
class Solution {
public:
    const int MOD = 1e9 + 7;
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> s(n + 1);
        for (int k = 0; k < n; ++k) s[k + 1] = s[k] + nums[k];
        int ans = 0;
        for (int i = 1; i < n; ++i) {
            int lo = lower_bound(s.begin() + 1, s.begin() + i, 2 * s[i] - s[n]) - s.begin();
            int up = upper_bound(s.begin() + 1, s.begin() + i, s[i] / 2) - s.begin();
            // printf("%d, %d, %d\n", i, lo, up);
            ans += max(0, up - lo);
            if (ans >= MOD) ans -= MOD;
            
        }
        
        return ans;
    }
};
