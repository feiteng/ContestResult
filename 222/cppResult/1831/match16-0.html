class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int sz = nums.size();
        vector<int> pre(sz+1,0);
        for(int i=1;i<=sz;i++){
            pre[i] = nums[i-1]+pre[i-1];
        }
        int res = 0;
        int mod = 1000000007;
        for(int i=1;i<sz-1;i++){
            int j = 0;
            {
                int l=i+1,r=sz+1;
                while(l<r){
                    int mid = (l+r)/2;
                    if(pre[mid]-pre[i]<pre[i]){
                        l = mid+1;
                    }else{
                        r = mid;
                    }
                }
                j = l;
            }
            if(j == sz) continue;
            {
                int l=j,r=sz;
                while(l<r){
                    int mid = (l+r)/2;
                    if(pre[sz]-pre[mid]>=pre[mid]-pre[i]){
                        l = mid+1;
                    }else{
                        r = mid;
                    }
                }
                res = (res + (l-j)%mod) % mod;
            }
        }
        return res % mod;
    }
};
