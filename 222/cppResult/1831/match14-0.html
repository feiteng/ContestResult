class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        long long res = 0;
        int n = nums.size();
        vector<int> pre(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            pre[i] = pre[i - 1] + nums[i - 1];
        }
        for (int i = 1; i < n; ++i) {
            int sum1 = pre[i];
            int l = i, r = n - 1;
            while (l < r && l < n && r < n) {
                int mid = l + r >> 1;
                int sum2 = pre[mid + 1] - pre[i];
                if (sum2 >= sum1) {
                    r = mid;
                }
                else {
                    l = mid + 1;
                }
            }
            int left = l;
            int s2 = pre[left + 1] - pre[i];
            int s3 = pre[n] - pre[left + 1];
            if (sum1 > s2 || s2 > s3 || left == n - 1) continue;
            l = i, r = n - 1;
            while (l < r && l < n && r < n) {
                int mid = l + r + 1>> 1;
                int sum2 = pre[mid + 1] - pre[i];
                int sum3 = pre[n] - pre[mid + 1];
                if (sum3 >= sum2) {
                    l = mid;
                }
                else {
                    r = mid - 1;
                }
            }
            int right = l;
            if (right == n - 1) right--;
            if (right >= left) {
                res += right - left + 1;
            }
        }
        return res % 1000000007;
    }
};
