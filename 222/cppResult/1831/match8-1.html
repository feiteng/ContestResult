class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        for(int i = 1;i < n;i++)nums[i] += nums[i-1];
        
        int mod = 1000000007;
        long long ans = 0;
        
        for(int i = 0;i < n;i++){
            int now = nums[i];
            int l = lower_bound(nums.begin(),nums.end(),now*2) - nums.begin();
            l = max(i + 1,l);
            int r = upper_bound(nums.begin(),nums.end(),(now + nums[n-1]) / 2) - nums.begin() - 1;
            r = min(n - 2,r);
            //cout << l << ", " << r << endl;
            if(r >= l)ans += r - l + 1;
            //x >= now*2
            //x <= (now + nums[n-1]) / 2
        }
        return ans % mod;
    }
};
