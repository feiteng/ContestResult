class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        long long ans = 0;
        int p = 1e9 + 7;
        vector<int> sum(nums.size());
        sum[0] = nums[0];
        for(int i = 1; i < nums.size(); ++i) sum[i] = sum[i - 1] + nums[i];
        for(int left = 0; left < nums.size() - 2; ++left){
            int l = left + 1, r = nums.size() - 2;
            while(l < r){
                int mid = (l + r) / 2;
                if(sum[mid] - sum[left] >= sum[left]) r = mid;
                else l = mid + 1;
            }
            int t = l;
            l = t, r = nums.size() - 2;
            while(l < r){
                int mid = (l + r + 1) / 2;
                if(sum[mid] - sum[left] <= sum.back() - sum[mid]) l = mid;
                else r = mid - 1;
            }
            //cout<<left<<' '<<t<<' '<<l<<endl;
            if(sum[t] - sum[left] >= sum[left] && sum.back() - sum[l] >= sum[l] - sum[left]) ans += max(l - t + 1, 0);
            //ans += max(l - t + 1, 0);
            ans %= p;
        }
        return ans;
    }
};
