const int mod=1e9+7;
int sum[100005];
class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        memset(sum,0,sizeof sum);
        for(int i=1;i<=n;i++) sum[i]=sum[i-1]+nums[i-1];
        int ans=0;
        for(int i=1;i<=n-2;i++){
            int left=sum[i];
            int l=i+1,r=n-1;
            while(l<r){
                int m=l+r>>1;
                int mid = sum[m] - sum[i];
                if(mid >= left) r=m;
                else l=m+1;
            }
            if(sum[l]-sum[i] < left || sum[n]-sum[l]<sum[l]-sum[i])continue;
            int L=l,R=n-1;
            while(L<R){
                int m=L+R+1>>1;
                int mid=sum[m] - sum[i];
                int right=sum[n]-sum[m];
                if(mid<=right) L=m;
                else R=m-1;
            }
            ans = (ans + L - l + 1)%mod;
        }
        return ans;
    }
};
