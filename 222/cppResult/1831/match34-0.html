class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int res = 0;
        int mod = 1e9 + 7;
        
        for(int i = 1; i < nums.size(); i++)
            nums[i] += nums[i - 1];
        
        int n = nums.size();
        for(int i = 0; i < n - 2 && nums[i] * 3 <= nums[n - 1]; i++)
        {
            int l = i + 1;
            int r = n - 2;
            while(l < r)
            {
                int mid = l + r >> 1;
                if(nums[mid] >= 2 * nums[i])
                    r = mid;
                else l = mid + 1;
            }
            
            if(l >= n)
                continue;
            
            int left = l;
            
            l = i + 1;
            r = n - 2;
            while(l < r)
            {
                int mid = (l + r >> 1) + 1;
                
                if(nums[n - 1] + nums[i] >= 2 * nums[mid])
                    l = mid;
                else r = mid - 1;
            }
            
            if(r <= i || nums[n - 1] + nums[i] < 2 * nums[l])
                continue;
            
            int right = l;
            
            if(right < left)
                continue;
            
            res = (res + right - left + 1) % mod;
        }
        
        return res;
    }
};
