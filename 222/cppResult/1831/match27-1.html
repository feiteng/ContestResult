class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        for (int i = 1; i < n; i++) nums[i] += nums[i - 1];
        int res = 0;
        for (int i = 0; i < n - 2; i++) {
            if (nums[i] * 3 > nums.back()) break;

            int lo = i + 1, hi = n - 2;
            while (lo <= hi) {
                int m = lo + (hi - lo) / 2;
                if (nums[m] - nums[i] < nums[i]) lo = m + 1;
                else hi = m - 1;
            }
            if (nums[lo] - nums[i] < nums[i]) break;

            int left = lo;
            lo = i + 1, hi = n - 2;
            while (lo <= hi) {
                int m = lo + (hi - lo) / 2;
                if (nums[m] - nums[i] > nums.back() - nums[m]) hi = m - 1;
                else lo = m + 1;
            }
            res = (res + (hi - left + 1)) % 1000000007;
        }
        return res;
    }
};
