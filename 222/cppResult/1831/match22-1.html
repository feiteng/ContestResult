class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> pre(n+1);
        pre[0] = 0;
        for (int i = 0; i < n; ++i) pre[i+1] = pre[i] + nums[i];
        
        long long ans = 0;
        for (int i = 0; i < n - 2; ++i) {
            int rem = pre[n]-pre[i+1];
            int mn = pre[i+1], mx = rem/2;

            if (mn > mx) break;
            
            int lb = i+1, rb = n-2;
            while (lb < rb) {
                int mb = (lb+rb)/2;
                if (pre[mb+1]-pre[i+1] < mn) {
                    lb = mb+1;
                } else {
                    rb = mb;
                }
            }
            int x = lb;
            
            lb = i+1, rb = n-2;
            while (lb < rb) {
                int mb = (lb+rb+1)/2;
                if (pre[mb+1]-pre[i+1] > mx) {
                    rb = mb-1;
                } else {
                    lb = mb;
                }
            }
            //cout << "i " << i << " lb " << lb << " x " << x << endl;
            
            ans += (long long)(max(0, lb-x+1));
        }
        
        ans %= 1000000007;
        return (int)ans;
    }
};
