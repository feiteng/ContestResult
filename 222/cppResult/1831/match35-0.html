class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        // cout << n << endl;
        vector<int> cumSum (n, 0);
        cumSum[0] = nums[0];
        for (int i = 1; i < n; i++) {
            cumSum[i] = cumSum[i-1] + nums[i];
        }
        
        int mod = pow(10, 9) + 7;
        int cnt = 0;
        
        int end = upper_bound(cumSum.begin(), cumSum.end(), cumSum.back() / 3) - cumSum.begin();
        for (int i = 0; i < end; i++) {
            int left = cumSum[i];
            // mid >= 2 * left
            int mid_start = lower_bound(cumSum.begin(), cumSum.end(), 2 * left) - cumSum.begin();
            if (mid_start <= i) mid_start = i+1;
            int mid_end = upper_bound(cumSum.begin(), cumSum.end(), (cumSum.back() - left) / 2 + left) - cumSum.begin();
            if (mid_end == n) mid_end = n-1;
            if (mid_start <= mid_end) {
                int add = (mid_end - mid_start) % mod;
                cnt = (cnt + add) % mod;
                // cout << i << ":" << mid_start << "-" << mid_end << endl;
            }
            
        }
        
        return cnt;
        
    }
};
