class Solution {
public:

    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        vector<int> psum(n+1,0);
        for (int i=0;i<n;i++){
            psum[i+1]=psum[i]+nums[i];
        }
        int ans=0;

        int m=1000000007;
        for (int a=1;a<=n-2;a++){
            int b0l=-1,b0r=n+1;
            while (b0l+1<b0r){
                int b0m=(b0l+b0r)/2;
                if (psum[b0m]>=psum[a]*2){
                    b0r=b0m;
                }else{
                    b0l=b0m;
                }
            }
            int b1l=-1,b1r=n+1;

            while (b1l+1<b1r){
                int b1m=(b1l+b1r)/2;
                if ((psum[b1m]-psum[a])*2>psum[n]-psum[a]){
                    b1r=b1m;
                }else{
                    b1l=b1m;
                }
            }
            b0r=max(b0r,a+1);
            b1l=min(b1l,n-1);
            ans+=max(0,b1l-b0r+1);
            ans%=m;
        }
        return ans;
    }
};
