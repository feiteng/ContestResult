#define ll long long
const int maxn = 1e5 + 10, mod = 1e9 + 7;
int s[maxn];

class Solution {
public:
    int waysToSplit(vector<int>& arr) {
        int n = arr.size();
        for (int i = 0; i <= n; ++i) {
            s[i] = 0;
        }

        for (int i = 1; i <= n; ++i) {
            s[i] = s[i - 1] + arr[i - 1];
        }
        ll ret = 0;
        for (int i = 1; i < n - 1; ++i){
            int a = s[i];
            ll l = i + 1, r = n, left = 0, right = 0;

            while (l <= r) {
                ll mid = (l + r) >> 1;
                if (s[mid] - s[i] >= a) {
                    left = mid;
                    r = mid - 1;
                } else {
                    l = mid + 1;
                }
            }

            l = i + 1, r = n - 1;
            while (l <= r) {
                ll mid = (l + r) >> 1;
                if (s[mid] - s[i] <= s[n] - s[mid]) {
                    right = mid;
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            if (left && right && left <= right)
                ret = (ret + (right - left + 1)) % mod;
        }
        return ret % mod;
    }
};
