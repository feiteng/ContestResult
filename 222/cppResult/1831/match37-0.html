class Solution {
public:
    int waysToSplit(vector<int>& nums) {
      int sz = nums.size();
      vector<int> prefix_sum(sz+1, 0);
      for (int i = 0; i < sz; i++) {
        prefix_sum[i+1] = prefix_sum[i] + nums[i];
      }
      int tot_sum = prefix_sum.back();
      long ret = 0;
      int MOD = 1000000007;
      for (int i = 1; i <= (sz - 2); i++) {
        int p1 = prefix_sum[i];
        int lower_limit = 2 * p1;
        int upper_limit = (p1 + tot_sum) / 2;
        int low_idx = lb(prefix_sum, i+1, sz+1, lower_limit);
        int up_idx = ub(prefix_sum, i+1, sz+1, upper_limit);
        if (low_idx <= i) continue;
        int len = min(up_idx, sz) - low_idx;
        if (len <= 0) continue;
        ret += len;
        if (ret > MOD) {
          ret %= MOD;  
        }
      }
      return ret;
    }
  
    int lb(vector<int>& nums, int lo, int hi, int target) {
      int i = lo, j = hi - 1;
      while (i <= j) {
        int mid = i + (j-i) / 2;
        if (nums[mid] >= target) {
          j = mid - 1;
        } else {
          i = mid + 1;
        }
      }
      return i;
    }
  
    int ub(vector<int>& nums, int lo, int hi, int target ) {
      int i = lo, j = hi - 1;
      while (i <= j) {
        int mid = i + (j-i) / 2;
        if (nums[mid] > target) {
          j = mid - 1;
        } else {
          i = mid + 1;
        }
      }
      return i;
    }
};
