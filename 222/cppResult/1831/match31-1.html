class Solution {
public:
    const int mod = 1e9+7;
    int waysToSplit(vector<int>& A) {
        int n = A.size();
        vector<int> pA(n);
        
        for(int i=0;i<n;i++){
            pA[i] = A[i];
            if(i>0){pA[i]+=pA[i-1];}
        }
        

        int ans = 0;
        for(int l=0;l<n-2;l++){
            int left_sum = pA[l];
            int mid_indx=n,r=-1;
            
            int low=l+1,high=n-2;
            while(low<=high){
                int mid = (low+high)/2;
                int mid_sum = pA[mid]-pA[l];
                if(mid_sum<left_sum){low=mid+1;}
                else{
                    mid_indx=mid;
                    high=mid-1;
                }
            }
            low = l+1,high=n-2;
            while(low<=high){
                int mid=(low+high)/2;
                int mid_sum = pA[mid]-pA[l];
                int right_sum = pA[n-1]-pA[mid];
                
                if(right_sum<mid_sum){
                    high=mid-1;
                }else{
                    r=mid;
                    low=mid+1;
                }
            }
            // cout<<l<<" "<<mid_indx<<" "<<r<<endl;
            if(mid_indx<=r){
                ans = (ans + (r-mid_indx+1))%mod;
            }
            
        }
        return ans;
    }
};
