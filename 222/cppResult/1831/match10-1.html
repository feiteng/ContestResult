class Solution {
public:
    int waysToSplit(vector<int>& nums) {

        int ans = 0, MOD = 1e9 + 7, n = nums.size();
        vector <int> pref(n);
        pref[0] = nums[0];
        for( int i = 1; i < n; i++ ) pref[i] = pref[i-1] + nums[i];
        for( int i = 0; i < n - 2; i++ )
        {
            int l = i, r = n; // l bad, r good
            while( l + 1 < r )
            {
                int m = (l + r)/2;
                if( pref[m] >= 2*pref[i] )
                    r = m;
                else
                    l = m;
            }
            

            int L = i + 1, R = n, rem = pref[n-1] - pref[i]; // l good, r bad;
            while( L + 1 < R )
            {
                int M = (L+R)/2;
                if( 2*(pref[M-1] - pref[i]) <= rem )
                    L = M;
                else
                    R = M;
            }
            if( r <= L )
                ans = (ans + (L-r))%MOD;
        }
        
        return ans;
    }
};
