class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> sums = nums;
        for (int i = 1; i < n; i++) {
            sums[i] += sums[i - 1];
        }
        int ans = 0;
        // middle的右端点
        int ml = 0;
        int mr = 0;
        // 遍历left的右端点
        for (int left = 0; left < n - 2; left++) {
            // 找middle left
            while (ml <= left) {
                ml++;
            }
            while (ml < n - 1 && sums[ml] - sums[left] < sums[left]) {
                ml++;
            }
            if (ml >= n - 1) {
                continue;
            }
            // 找middle right
            if (mr < ml) {
                mr = ml;
            }
            while (mr < n - 2 && sums[mr + 1] - sums[left] <= sums[n - 1] - sums[mr + 1]) {
                mr++;
            }
            if (mr >= n - 1) {
                continue;
            }
            if (sums[mr] - sums[left] <= sums[n - 1] - sums[mr]) {
                if (mr >= ml) {
                    ans += (mr - ml + 1);
                    ans %= 1000000007;
                }
            }
        }
        return ans;
    }
};
