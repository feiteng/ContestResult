typedef long long LL;
const int mod = 1e9+7;
class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        nums.insert(nums.begin(),0);
        vector<LL> s(n+1,0);
        for(int i=1;i<=n;++i) s[i]=s[i-1]+nums[i];
        LL res=0,tot=s[n]-s[0];
        for(int i=1;i<=n-2 && (s[i]-s[0])*3<=tot;++i)
        {
            int l=i+1,r=n+1;
            while(l<r)
            {
                int mid=(l+r)>>1;
                if(s[mid]-s[i]>=s[i]-s[0])
                    r=mid;
                else
                    l=mid+1;
            }
            int j1=l;
            l=i+1,r=n+1;
            while(l<r)
            {
                int mid=(l+r)>>1;
                if(s[n]-s[mid]<s[mid]-s[i])
                    r=mid;
                else
                    l=mid+1;
            }
            int j2=l-1;
            if(j1==n)--j1;
            if(j2==n)--j2;
            // cout<<i<<' '<<j1<<' '<<j2<<endl;
            if(j1<=j2) res+=(j2-j1+1),res%=mod;
        }
        return res;
    }
};
