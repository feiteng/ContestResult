class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> c(n);
        for(int i = 0; i < n; ++i){
            if(i){
                c[i] = c[i-1] + nums[i];
            }
            else{
                c[i] = nums[i];
            }
        }
        long long ans = 0;
        nums.push_back(0);
        int mod = (1e9) + 7;
        for(int i = 2; i < n; ++i){
            int sr = c[n-1] - c[i-1];

            int l = 0, r = i + 1;
            while(l < r){
                int mid = l + ((r - l)/2);
                int sm = c[n-1] - sr - c[mid];
                int sl = c[mid];
                // cout << "mid: " << mid << " " << sm << '\n';
                if(sm > sr){
                    l = mid + 1;
                }
                else{
                    r = mid;
                }
            }
            // cout << l << '\n';
            int sm = c[n-1] - sr - c[l];
            int sl = c[l];
            if(sl > sm){
                continue;
            }
            // else{
            //     l = max(0, l-1);
            //     int sm = c[n-1] - sr - c[l];
            //     int sl = c[l];
            //     if(sl > sm or sm > sr){
            //         continue;
            //     }
            // }
            int mm = l;
            r = i - 1;
            while(l < r){
                int mid = l + ((r - l)/2);

                int sm = c[n-1] - sr - c[mid];
                int sl = c[mid];
                if(sl > sm){
                    r = mid;
                }
                else{
                    l = mid + 1;
                }
            }
            // cout << i << " " << l - mm << " " << mm << " " << l << '\n';
            ans = (ans + (l - mm))%mod;
            
        }
        return ans;
    }
};
