class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size(), ans = 0;
        vector<int> s(n);
        s[0] = nums[0];
        for(int i = 1; i < n; i++) s[i] = s[i - 1] + nums[i];
        for(int i = 0; i < n - 2; i++) {
            int l = i + 1, r = n - 2;
            while(l <= r) {
                int m = (l + r) / 2;
                if(s[m] - s[i] <= s[n - 1] - s[m]) l = m + 1;
                else r = m - 1;
            }
            if(r != i && s[r] - s[i] >= s[i]) {
                int t = r, l = i + 1, r = t;
                while(l <= r) {
                    int m = (l + r) / 2;
                    if(s[m] - s[i] < s[i]) l = m + 1;
                    else r = m - 1;
                }
                ans = (ans + t - l + 1) % 1000000007;
            }
        }
        return ans;
    }
};
