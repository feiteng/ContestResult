class Solution {
public:
    const int MOD = 1e9+7;
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> presum(n+1, 0);
        int res = 0;
        for (int i = 0; i < n; i++ ) {
            presum[i+1] += presum[i] + nums[i];
        }
        for (int i = 0; i < n-2; i++) {
            int p1 = presum[i+1];
            if (p1*3 > presum[n]) break;
            int l = i+1, r = n-1;
            while (l < r) {
                int mid = l + (r - l) / 2;
                if (presum[mid+1] >= 2*p1) {
                    r = mid;
                }
                else {
                    l = mid + 1;
                }
            }
            int j1 = l;
            l = i+1, r = n - 1;
            while (l < r)  {
                int mid = l + (r - l) / 2;
                if (presum[mid+1] - p1 > presum[n] - presum[mid+1]) {
                    r = mid;
                }
                else {
                    l = mid + 1;
                }
            }
            int j2 = l - 1;
            if (j2 < j1) continue;
            res = (res + (j2 - j1 + 1)) % MOD;
                
        }
        return res;
    }
};

i定后，persum1，找j的位置，二分找出找第一个大于等于2倍presum1的位置作为j， 判断剩下的部分是否满足条件。
再次二分找出第一个presum - presum1 > sum - presum 的位置 -1
