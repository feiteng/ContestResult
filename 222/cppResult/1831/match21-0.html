class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size(), ans = 0, mod = 1e9+7;
        vector<int> sum(n+1);
        for(int i = 1; i <= n; i++) {
            sum[i] = sum[i-1]+nums[i-1];
        }
        int left = 0;
        for(int i = 1; i <= n; i++) {
            left += nums[i-1];
            int l = i+1, r = n-1, seg1 = -1, seg2 = -1;
            while(l <= r) {
                int m = (l+r)/2;
                int mid = sum[m]-sum[i];
                if(mid >= left) {
                    seg1 = m;
                    r = m-1;
                } else {
                    l = m+1;
                }
            }
            if(seg1 == -1) {
                continue;
            }
            l = seg1, r = n-1;
            while(l <= r) {
                int m = (l+r)/2;
                int mid = sum[m]-sum[i];
                int right = sum[n]-sum[m];
                if(mid <= right) {
                    seg2 = m;
                    l = m+1;
                } else {
                    r = m-1;
                }
            }
            if(seg2 == -1) {
                continue;
            }
            ans += seg2-seg1+1;
            ans %= mod;
        }
        return ans;
    }
};
