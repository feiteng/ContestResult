class Solution {
public:
    const int mod = 1e9 + 7;
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        for (int i = 1; i < n; ++i) {
            nums[i] += nums[i-1];
        }
        int ans = 0;
        for (int i = 0; i < n - 2; ++i) {
            int left = nums[i];
            int l = i + 1, r = n - 2;
            while (l < r) {
                int mid = (l + 0ll + r) >> 1;
                if (left > nums[mid] - left) {
                    l = mid + 1;
                } else {
                    r = mid;
                }
            }

            if (left > nums[r] - left) continue;
           
            int f1 = r;
            
            l = i + 1;
            r = n - 2;
            while (l < r) {
                int mid = (l + 1ll + r) >> 1;
                if (nums[mid] - left > nums[n-1] - nums[mid]) {
                    r = mid - 1;
                } else {
                    l = mid;
                }
            }
            if (nums[l] - left > nums[n-1] - nums[l]) continue;
            int f2 = l;
            if (f2 - f1 + 1 > 0) {
                ans = (ans + 0ll + (f2 - f1 + 1)) % mod;
            }
        }
        return ans;
    }
};
