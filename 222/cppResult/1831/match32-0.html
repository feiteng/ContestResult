class Solution {
public:
    const int mod = 1e9 + 7;
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int>s(n + 1, 0);
        for(int i = 1; i <= n; i++)s[i] = nums[i - 1] + s[i - 1];
        int res = 0;
        for(int i = 1; i <= n - 2; i++)
        {
            int l = i + 1, r = n - 1;
            int ll, rr;
            while(l < r)
            {
                int mid = l + r + 1 >> 1;
                if(s[mid] - s[i] <= s[n] - s[mid])l = mid;
                else r = mid - 1;
            }
            if(s[r] - s[i] <= s[n] - s[r])rr = r;
            else rr = i;
            l = i + 1, r = n;
            while(l < r)
            {
                int mid = l + r >> 1;
                if(s[mid] - s[i] >= s[i])r = mid;
                else l = mid + 1;
            }
            if(s[l] - s[i] >= s[i])ll = l;
            else ll = n + 1;
            int cnt = rr - ll + 1;
            if(cnt > 0)res = (res + cnt) % mod;
            //cout << n << ' ' << ll << ' ' << rr << endl;
        }
        return res;
    }
};
