int mod = 1000000007;
class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int>f(n+2,0);
        for(int i = 1 ; i <= n ; ++i) f[i] = f[i - 1] + nums[i - 1];
        int sum = f[n] / 3;
        int res = 0;
        f[n+1] = f[n] + 1;
        for(int i = 1 ; f[i] <= sum  && i < n - 1; ++i){
            int dx = (f[n] - f[i]) / 2 + f[i];
            int mx = upper_bound(f.begin() , f.end() , dx) - f.begin();
            int mn = lower_bound(f.begin() + 1, f.end() , f[i] << 1) - f.begin();
            //cout <<dx <<" "<< mx <<" " << mn <<endl;
            if(mx == n + 1) --mx;
            if(mn <= i) mn = i + 1;
            res = (res + mx - mn) % mod;
        }
        return res;
    }
};
