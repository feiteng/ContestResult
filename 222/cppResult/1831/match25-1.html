class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int mod = 1e9 + 7;
        int n = nums.size();
        vector<int> s(n + 1);
        for (int i = 1; i <= n; i ++ ) s[i] = s[i - 1] + nums[i - 1];
        int res = 0;
        for (int i = 3, j = 2, k = 2; i <= n; i ++ ) {
            while (s[n] - s[i - 1] < s[i - 1] - s[j - 1]) j ++ ;
            while (k + 1 < i && s[k] <= s[i - 1] - s[k]) k ++ ;
            if (j <= k && s[n] - s[i - 1] >= s[i - 1] - s[j - 1] && s[k - 1] <= s[i - 1] - s[k - 1])
                res = (res + k - j + 1) % mod;
        }
        return res;
    }
};
