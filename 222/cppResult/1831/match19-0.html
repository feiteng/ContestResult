class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n=nums.size();
        vector<int> presums(n);
        presums[0]=nums[0];
        for(int i=1;i<n;i++)
            presums[i]=presums[i-1]+nums[i];

        long long res=0;
        int tmp=presums.back()/3;
        int m=upper_bound(presums.begin(),presums.end(),tmp)-presums.begin();
        m=min(m,n-2);
        for(int i=0;i<m;i++){
            int a=presums[i];
            int k1=lower_bound(presums.begin()+i+1,presums.end(),a+a)-presums.begin();
            int b=(presums.back()+presums[i])/2;
            int k2=upper_bound(presums.begin()+i+1,presums.end(),b)-presums.begin();
            k2=min(k2,n-1);
            res+=(long long)(k2-k1);
            // cout<<i<<" "<<nums[i]<<" "<<k2-k1<<endl;
        }
        return res%1000000007;
    }
};
