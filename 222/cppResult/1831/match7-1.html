class Solution {
public:
    #define ll long long
    const ll MOD = 1e9 + 7;
    
    int waysToSplit(vector<int>& a) {
        int n = a.size();
        vector<int> ps(n + 1);

        for (int i = 0; i < n; i++) ps[i + 1] = ps[i] + a[i];
        int ans = 0;
        for (int i = 1; i <= n - 2; i++) { // try each split for left part a1
            // cout << i << "\n";
            int a1 = ps[i];
            
            if (a1 > ps[n - 1] - ps[i]) continue;

            int lo1 = i, hi1 = n - 1;
            while (lo1 + 1 < hi1) {
                int mid = (lo1 + hi1) / 2;
                int a2 = ps[mid] - ps[i];
                if (a1 <= a2) hi1 = mid;
                else lo1 = mid;
            }
            
            if (ps[i + 1] - ps[i] > ps[n] - ps[i + 1]) continue;
            int lo2 = i + 1, hi2 = n;
            while (lo2 + 1 < hi2) {
                int mid = (lo2 + hi2) / 2;
                int a2 = ps[mid] - ps[i];
                int a3 = ps[n] - ps[mid];
                if (a2 <= a3) lo2 = mid;
                else hi2 = mid;
            }
            
            // cout << lo1 << " " << hi1 << " " << lo2 << " " << hi2 << "\n";
            ans = (ans + max(0, lo2 - hi1 + 1)) % MOD;
        }
        return ans;
    }
};
