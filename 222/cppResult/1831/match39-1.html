#define LL long long

class Solution {
public:
    int waysToSplit(vector<int>& a) {
      int n = a.size();
      vector<int> s(n, 0);
      s = a;
      for (int i = 1; i < n; i++) {
        s[i] += s[i - 1];
      }
      LL ans = 0;

      for (int i = 1, j = 0, k = 0; i < n - 1; i++) {
        while (k < i && s[i] - s[k] >= s[k]) {
          k += 1;
        }
        while (j < i && s[i] - s[j] > s[n - 1] - s[i]) {
          j += 1;
        }
        if (k - j > 0) {
          ans += k - j;
        }
      }
      LL mod = 1e9 + 7;
      return ans % mod;
    }
};
