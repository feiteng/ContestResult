class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        const int MOD = 1e9 + 7;
        int n = static_cast<int>(nums.size());

        vector<int> sum(n + 1);
        for (int i = 1; i <= n; ++i) {
            sum[i] = sum[i - 1] + nums[i - 1];
        }
        int ways = 0;
        for (int i = 1; i < n - 1; ++i) {
            int lb = n + 1;
            int l = i + 1, r = n;
            while (l <= r) {

                int mid = (l + r) / 2;
                if (sum[mid] - sum[i] >= sum[i]) {
                    lb = min(lb, mid);
                    r = mid - 1;
                } else {
                    l = mid + 1;
                }
            }
            if (lb == n + 1) {
                continue;
            }
            int rb = lb - 1;
            l = lb + 1, r = n;
            while (l <= r) {

                int mid = (l + r) / 2;
                if (sum[n] - sum[mid - 1] >= sum[mid - 1] - sum[i]) {
                    rb = max(rb, mid);
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            if (rb == lb - 1) {
                continue;
            }
            (ways += rb - lb) %= MOD;
        }
        return ways;
    }
};
