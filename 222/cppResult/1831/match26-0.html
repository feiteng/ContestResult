const int N = 100005;
const int mod = 1e9 + 7;
int s[N];
class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        for(int i = 0; i < nums.size(); ++i)
        {
            s[i + 1] = s[i] + nums[i];   
        }
        
        int res = 0;
        int n = nums.size();
        for(int i = 3; i <= n; ++i)
        {
            int l = 2, r = i - 1;
            int a = s[n] - s[i - 1];
            //if(i == 5) cout << a << endl;
            while(l < r)
            {
                int mid = l + r >> 1;
                if(s[i - 1] - s[mid - 1] <= a) r = mid;
                else l = mid + 1;
            }
            if(i == 4) cout << l << endl;
            if(s[n] - s[i - 1] < s[i - 1] - s[l - 1]) continue;
            if(s[i - 1] - s[l - 1] < s[l - 1] - s[0]) continue;
            else
            {
                int L = l, R = i - 1;
                while(L < R)
                {
                    int mid = L + R + 1 >> 1;
                    if(s[i - 1] - s[mid - 1] >= s[mid - 1] - s[0]) L = mid;
                    else R = mid - 1;
                }
                //if(i == 4)cout << L << endl;
                //if(L - l + 1 >= 1) cout << l << "---" << R << endl;
                res = (res + (L - l + 1)) % mod; 
            }
        }
        return res;
    }
};
