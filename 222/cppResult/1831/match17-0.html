class Solution {
public:
    static constexpr int mod = 1e9 + 7;
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        cout << n << endl;
        vector<int> pre(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            pre[i + 1] = pre[i] + nums[i]; 
        }
        int ans = 0;
        for (int i = 1; i < n; ++i) {
            int l = i + 1, r = n;
            while (l <= r) {
                int mid = l + (r - l) / 2;
                if (pre[mid] - pre[i] >= pre[i]) {
                    r = mid - 1;
                } else {
                    l = mid + 1;
                }
            }
            int j = l;
            if (j >= n || pre[j] < 2 * pre[i]) break;
            l = j + 1, r = n - 1;
            while (l <= r) {
                int mid = l + (r - l) / 2;
                if (pre[n] - pre[mid] >= pre[mid] - pre[i]) {
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            int k = r;
            //cout << j << k << endl;
            //cout << i << j << k << endl;
            if (k >= j && pre[n] + pre[i] >= 2 * pre[k]) ans = (ans + k - j + 1) % mod;
        }
        return ans;
    }
};
