const int mod = 1e9 + 7;
class Solution {
public:
    int waysToSplit(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n + 2);
        for (int i = 1; i <= n; ++i) f[i] = f[i - 1] + nums[i - 1];
        int s = f[n] / 3;
        int ans = 0;
        f[n + 1] = f[n] + 1;
        for (int i = 1; f[i] <= s and i < n - 1; ++i) {
            int dx = (f[n] - f[i]) / 2 + f[i];
            int mx = upper_bound(f.begin(), f.end(), dx) - f.begin();
            int mn = lower_bound(f.begin(), f.end(), f[i] * 2) - f.begin();
            if (mx == n + 1) mx--;
            if (mn <= i) mn = i + 1;
            ans = (ans + mx - mn) % mod;
        }
        return ans;
    }
};
