class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        unordered_map<int, vector<int>> val_indices;
        for (int i = 0; i < arr.size(); ++i)
            val_indices[arr[i]].push_back(i);

        vector<int> A;
        for (auto& x : target) {
            if (auto it = val_indices.find(x); it != val_indices.end()) {
                auto tmp = it->second;
                reverse(tmp.begin(), tmp.end());
                A.insert(A.end(), tmp.begin(), tmp.end());
            }
        }
        
        vector<int> x;
        for (auto& a : A) {
            auto it = lower_bound(x.begin(), x.end(), a);
            if (it == x.end()) {
                x.push_back(a);
            } else {
                *it = a;
            }
        }
        return target.size() - x.size();
    }
};
