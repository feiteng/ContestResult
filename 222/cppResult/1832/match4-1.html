class Solution {
public:

    int minOperations(vector<int>& target, vector<int>& arr) {
        map<int,int> g;
        int n=target.size();
        for(int i=0;i<n;++i){
            g[target[i]]=i+1;
        }
        int m=arr.size();
        vector<int> v;
        for(int i=0;i<m;++i){
            if(g[arr[i]]) v.push_back(g[arr[i]]);
        }
        int p=LIS(v);
        return n-p;
    }
    int LIS(vector<int>& nums) {
        int len=1, n=nums.size();

        if (n==0) {
            return 0;
        }
        vector<int> dp(n+1,0);
        dp[1] = nums[0];
        for (int i=1; i<n; ++i) {
            if (nums[i]>dp[len]) {
                dp[++len] = nums[i];
            } else {
                int l=1, r=len, pos=0;
                while (l<=r) {
                    int mid=(l+r)>>1;
                    if (dp[mid]<nums[i]) {
                        pos=mid;
                        l=mid+1;
                    } 
                    else{
                        r=mid-1;
                    }
                }
                dp[pos+1]=nums[i];
            }
        }
        return len;
    }
};
