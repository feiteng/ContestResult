class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        unordered_map<int, int> mp;
        for (int i = 0; i < target.size(); ++i) {
            mp[target[i]] = i;
        }
        
        vector<int> idxs(arr.size(), -1);
        for (int i = 0; i < arr.size(); ++i) {
            if (mp.find(arr[i]) != mp.end()) {
                idxs[i] = mp[arr[i]];
            }
            // cout << idxs[i] << " ";
        }
        // cout << endl;
        
        vector<int> ls;
        for (int i = 0; i < idxs.size(); ++i) {
            if (idxs[i] == -1) continue;
            
            if (ls.size() == 0 || ls.back() < idxs[i]) {
                ls.emplace_back(idxs[i]);
            }
            else {
                int l = 0;
                int r = ls.size() - 1;
                int target = idxs[i];
                while (l < r) {
                    int m = l + (r - l + 0) / 2;
                    if (ls[m] >= target) {
                        r = m;
                    }
                    else {
                        l = m + 1;
                    }
                }
                ls[l] = idxs[i];
            }
        }
        
        // for (auto l : ls) {
        //     cout << l << " ";
        // }
        // cout << endl;
        
        return target.size() - ls.size();
    }
};
