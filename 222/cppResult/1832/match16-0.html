class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        unordered_map<int, int> mmap;
        for (int i=0; i<target.size(); i++)
            mmap[target[i]] = i;
        vector<int> indices;
        for (int i=0; i<arr.size(); i++) {
            if (mmap.find(arr[i]) != mmap.end())
                indices.push_back(mmap[arr[i]]);
        }
        
        set<int> mset;
        for (int i: indices) {
            auto it = mset.lower_bound(i);
            if (it != mset.end())
                mset.erase(it);
            mset.insert(i);
        }
        return target.size() - mset.size();
    }
};
