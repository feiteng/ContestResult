class Solution {
public:
  int minOperations(vector<int>& target, vector<int>& arr) {
    unordered_map<int,int> M;
    for (int i = 0; i < target.size(); i++) M[target[i]] = i;
    vector<int> A;
    for (auto &a: arr) if (M.count(a)) A.push_back(M[a]);
    
    vector<int> B;
    for (auto &a: A) {
      auto it = lower_bound(B.begin(), B.end(), a);
      if (it == B.end()) B.push_back(a);
      else *it = a;
    }
    return target.size() - B.size();
      


  }
};
