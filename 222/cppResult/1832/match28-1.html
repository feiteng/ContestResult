class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        unordered_map<int, int> t;
        for (int i = 0; i < target.size(); ++i) {
            t[target[i]] = i;
        }
        vector<int> v;
        for (int a : arr) {
            if (t.find(a) == t.end()) {
                continue;
            }
            v.push_back(t[a]);
        }
        if (v.empty()) {
            return 0;
        }
        int rv = 1;
        vector<int> d(v.size()+1, 1);
        int len = 1;
        d[1] = v[0];
        for (int i = 1; i < v.size(); ++i) {
            if (v[i] > d[len]) {
                len++;
                d[len] = v[i];

            } else {
                int start = 1;
                int end = len;
                int pos = 0;
                while (start <= end) {
                    int mid = (start + end) / 2;
                    if (d[mid] < v[i]) {
                        pos = mid;
                        start = mid + 1;

                    } else {
                        end = mid - 1;
                    }
                }
                d[pos+1] = v[i];
            }
        }
        return target.size() - len;
    }
};
