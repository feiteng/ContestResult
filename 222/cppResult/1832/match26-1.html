class Solution {
public:
    int minOperations(vector<int>& t, vector<int>& a) {
        map<int, int> m;
        for (int i = 0; i < t.size(); ++i)
            m[t[i]] = i;
        vector<int> r(a.size());
        int i = 0;
        for (int x: a) {
            if (m.count(x)) {
                r[i++] = m[x];
            }            
        }
        int n = i;
        r.resize(n);
        
        vector<int> dp;
        
        n = r.size(); 
        if (!n) return t.size();
        dp.push_back(r[0]);

        for (int i = 1; i < n; ++i)
        {
            auto ptr = lower_bound(dp.begin(), dp.end(), r[i]);
            if (ptr == dp.end())
                dp.push_back(r[i]);
            else
                *ptr = r[i];
        }
        return t.size() - dp.size();
    }
};
