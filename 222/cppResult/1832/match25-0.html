class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        map<int, int> t2i;
        for(int i=0;i<target.size();++i) {
            t2i[target[i]] = i;
        }
        vector<int> indices;
        for(auto &a: arr) {
            if(t2i.find(a) != t2i.end()) {
                indices.push_back(t2i[a]);
            }
        }
        
        if(indices.empty()) {
            return target.size();
        }
        
        // 
        int n = indices.size();
        int MAX = target.size() * 2;
        vector<int> L;
        for(int i=0;i<n;++i) {
            if(L.empty() || L.back() < indices[i]) {
                L.push_back(indices[i]);
            }
            else {
                auto it = lower_bound(L.begin(), L.end(), indices[i]);
                *it = indices[i];
            }
        }
        return target.size() - L.size();
    }
};
