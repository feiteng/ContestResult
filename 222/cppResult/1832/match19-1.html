class Solution {
public:
    int minOperations(vector<int>& b, vector<int>& c) {
        map<int, int> A;
        for (int i = 0; i < (int)b.size(); ++i) A[b[i]] = i;
        vector<int> a;
        for (auto& x : c) {
            if (A.count(x)) a.push_back(A[x]);

        }
        vector<int> d;
        for (auto& x : a) {
            auto it = lower_bound(d.begin(), d.end(), x);
            if (it == d.end()) {
                d.push_back(x);
            } else {
                *it = x;
            }
        }
        return b.size() - d.size();
    }
};
