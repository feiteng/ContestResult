class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        unordered_map<int, int> idx;
        for(int i = 0; i < target.size(); i++) {
            idx[target[i]] = i;
        }

        vector<int> order;
        for(int i = 0; i < arr.size(); i++) {
            if(idx.count(arr[i])) order.push_back(idx[arr[i]]);
        }
        vector<int> lis;
        for(int x: order) {
            // cout << x << ' ';
            auto p = lower_bound(lis.begin(), lis.end(), x);
            if(p == lis.end()) lis.push_back(x);
            else *p = x;
        }
        // cout << endl;
        // for(int x: lis) cout << x << ' '; cout << endl;
        return int(target.size() - lis.size());
    }
};
