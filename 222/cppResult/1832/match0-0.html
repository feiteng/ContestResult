class Solution {
public:
     const static int N=100010;
     map<int,int>idx;
     int n,m;
     int s[N],dp[N];
    
     void init(){
           for(int i=1;i<=n+1;i++)s[i]=0;
           for(int i=0;i<m;i++)dp[i]=0;
     }
    
     int lowbit(int x){
          return x&(-x);
     }

     void update(int x,int v){
          for(int i=x;i<=n+1;i+=lowbit(i))s[i]=max(s[i],v);
     }

     int query(int x){
           int rtn=0;
           for(int i=x;i;i-=lowbit(i))rtn=max(rtn,s[i]);
           return rtn;
     }
        
    int minOperations(vector<int>& target, vector<int>& arr) {
        n=target.size();m=arr.size();
        for(int i=0;i<n;i++){
            idx[target[i]]=i+1;
        }
        for(int i=0;i<m;i++){
            if(idx[arr[i]]==0)
                arr[i]=n+1;
                else arr[i]=idx[arr[i]];
            }
            init();
            for(int i=0;i<m;i++){
               
                if(arr[i]==n+1)
                    continue;
                
                dp[i]=query(arr[i]-1)+1;
                update(arr[i],dp[i]);
                
            }
            int mx=0;
            for(int i=0;i<m;i++)
               mx=max(mx,dp[i]);
            return n-mx;
     }
};
