class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        int n=arr.size(),m=target.size();
        unordered_map<int,int> ma;
        for(int i=0;i<m;i++)    {
            ma[target[i]]=(i+1);
        }
        vector<int> A;
        for(int i=0;i<n;i++)    {
            if( ma[arr[i]]>0 )  {
                A.push_back(ma[arr[i]]);
            }
        }
        n=A.size();
        int len=0;
        vector<int> dp(n,INT_MAX);
        for(int i=0;i<n;i++)    {
            int index=lower_bound(dp.begin(),dp.begin()+len,A[i])-dp.begin();
            if( index==len )
                len++;
            dp[index]=A[i];
        }
        return m-len;
    }
};
