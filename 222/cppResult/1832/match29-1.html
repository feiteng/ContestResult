class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {

        unordered_map<int, int> numId;
        for (int i = 0; i < target.size(); ++i) {
            numId[target[i]] = i;
        }
        
        vector<int> longestEnd;
        for (int v: arr) {

            if (numId.count(v)) {
                int curr = numId[v];
                auto idx = lower_bound(begin(longestEnd), end(longestEnd), curr) - begin(longestEnd);
                if (idx == longestEnd.size()) {
                    longestEnd.push_back(curr);
                } else {
                    longestEnd[idx] = curr;
                }
            }
        }
        return target.size() - longestEnd.size();
    }
};
