class Solution {
public:
    int minOperations(vector<int>& t, vector<int>& arr) {
        unordered_map<int, int> mp;
        for (int i = 0; i < t.size(); ++i)
            mp[t[i]] = (i + 1);
        
        vector<int> a;
        
        for (int i = 0; i < arr.size(); ++i)
            if (mp.count(arr[i]))
                a.push_back(mp[arr[i]]);
        
        int n = a.size();
        
        vector<int> dp;
        for (int i = 0; i < n; ++i) {
            auto it = lower_bound(dp.begin(), dp.end(), a[i]);
            if (it == dp.end())
                dp.push_back(a[i]);
            else
                dp[it - dp.begin()] = a[i];
        }
        
        return t.size() - dp.size();
    }
};
