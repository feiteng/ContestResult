class Solution {
public:
    int lis(vector<int> &a)
    {

        int n = a.size();
        int nn=0;
        vector<int> nlis;
        for(int i=0;i<n;i++)
        {
            int k=lower_bound(nlis.begin(),nlis.end(),a[i])-nlis.begin();
            if(k==nn) nlis.push_back(0),nn++;
            nlis[k]=a[i];
        }
        return nlis.size();
    }
    int minOperations(vector<int>& target, vector<int>& arr) {
        int m = target.size();

        int n = arr.size();
        map<int,int> h;
        for (int i = 0; i < m; i++) {
            h[target[i]] = i + 1;
        }
        //cout<<"here"<<endl;
        vector<int> a;
        for (int i = 0; i < n; i++) {
            if (h.count(arr[i]))
                a.push_back(h[arr[i]]);
        }
        //cout<<"here"<<endl;
        return m - lis(a);
    }
};
