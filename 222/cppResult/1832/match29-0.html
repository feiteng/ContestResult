class Solution {
public:
  int minOperations(vector<int>& target, vector<int>& arr) {
    unordered_map<int, vector<int>> m;
    for (int i = 0; i < arr.size(); i++) {
      m[arr[i]].push_back(i);
    // cout << "x";
    }
    vector<int> dp;
    for (int t:target) {
      if (m.count(t)) {
        auto& v = m[t];
        for (int i = v.size() - 1; i >= 0; i--) {
          int p = v[i];
          auto lb = lower_bound(dp.begin(), dp.end(), p);
          if (lb == dp.end()) {
            dp.push_back(p);
          } else {
            *lb = min(*lb, p);
          }
        }
      }
    }
    
    return target.size() - dp.size();
  }
};
