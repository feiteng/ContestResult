class Solution {
public:
    int minOperations(vector<int>& target, vector<int>& arr) {
        map<int, int> dp;
        for(int i=0;i<target.size();i++)
        {
            dp[target[i]] = i;
        }
        vector<int> help;
        for(int i=0;i<arr.size();i++)
        {
            if(dp.count(arr[i]))
            {
                help.push_back(dp[arr[i]]);
            }
        }
        vector<int> dp2;
        for(const auto& it:help)

        {
            if(dp2.empty() || it > dp2.back())
            {
                dp2.push_back(it);
            }
            else
            {
                auto tmp = lower_bound(dp2.begin(), dp2.end(), it);
                *tmp = it;
            }
            
        }
        return target.size() - dp2.size();
    }
};
