class Solution {
public:
    map<int,int> mp;
    int tr[100005];
    int lb(int x){
        return x&(-x);
    }
    void add(int x,int v){
        while(x<=100002){
            tr[x] = max(tr[x],v);
            x+=lb(x);
        }
    }
    int get(int x){
        int ans=0;
        while(x>0){
            ans=max(ans,tr[x]);
            x-=lb(x);
        }
        return ans;
    }
    int minOperations(vector<int>& target, vector<int>& arr) {
        int len1 = target.size();
        int len2 = arr.size();
        for(int i=0;i<len1;i++)
        {
            mp[target[i]]=i+1;
        }
        int ans=0;
        for(int i=0;i<len2;i++)
        {
            int id = mp[arr[i]];
            if(id==0)
                continue;
            int pre = get(id-1);
            ans = max(ans, pre+1);
            add(id, pre+1);
        }
        
        return (len1-ans);
    }
};
