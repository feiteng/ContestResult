class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
      vector<pair<int, int>> p;
      for (auto iter : boxTypes) {
        p.push_back({iter[1], iter[0]});
      }
      sort(p.begin(), p.end());
      reverse(p.begin(), p.end());
      int ans = 0;
      for (auto iter : p) {
        int t = min(truckSize, iter.second);
        ans += iter.first * t;
        truckSize -= t;
      }
      return ans;
    }
};
