class Solution {
public:
    int maximumUnits(vector<vector<int>>& b, int t) {
        sort(b.begin(),b.end(),[](vector<int> & x, vector<int> & y){return x[1]>y[1];});
        int res=0;
        for(int i=0;i<b.size() && t>0;++i)
        {
            int tmp = min(t,b[i][0]);
            res+=(tmp*b[i][1]);
            t-=b[i][0];
        }
        return res;
    }
};
