class Solution {
public:
    int maximumUnits(vector<vector<int>>& b, int T) {
        sort(b.begin(), b.end(), [](vector<int>& v1, vector<int>& v2){
            return v1[1] > v2[1];
        });
        int res = 0;
        for(int i = 0; i < b.size(); ++i) {
            int d = min(b[i][0], T);
            res += d * b[i][1];
            T -= d;
        }
        return res;
    }
};
