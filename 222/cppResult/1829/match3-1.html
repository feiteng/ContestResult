class Solution {
public:
    int maximumUnits(vector<vector<int>>& bt, int ts) {
        sort(bt.begin(), bt.end(), [&](vector<int>& l, vector<int>& r) -> bool
             {
                 return l[1] > r[1];
             });
        int i = 0;
        int res = 0;
        while (ts > 0 && i < bt.size())
        {
            int k = min(ts, bt[i][0]);
            res += k * bt[i][1];
            i++;
            ts -= k;
        }
        return res;
        
    }
};
