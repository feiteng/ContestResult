class Solution {
public:
    int maximumUnits(vector<vector<int>>& B, int T) {
        sort(B.begin(), B.end(), [&](vector<int> &a, vector<int> &b){ return a[1] > b[1]; });
        
        int n = B.size(), ans = 0;
        for (int i = 0; i < n && T > 0; i++){
            int cur = min(T, B[i][0]);
            ans += cur * B[i][1];
            T -= cur;
            // printf("B: %d %d\n", i, B[i][1]);
        }
        
        return ans;
    }
};
