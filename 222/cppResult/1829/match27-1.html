class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(),boxTypes.end(),[](vector<int>&a ,vector<int>&b){
            return a[1]>b[1];
        });
        int ans=0;
        int p=0;
        while(truckSize>0&&p<boxTypes.size()){
            int c=min(truckSize,boxTypes[p][0]);
            ans+=c*boxTypes[p][1];
            p++;
            truckSize-=c;
        }
        return ans;
    }
};
