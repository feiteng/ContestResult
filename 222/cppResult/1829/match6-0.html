class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        vector<pair<int,int>>v;
        for(int i=0;i<boxTypes.size();i++) v.push_back({boxTypes[i][1],boxTypes[i][0]});
        sort(v.begin(),v.end());
        reverse(v.begin(),v.end());
        int ans=0;
        for(int i=0;i<v.size();i++){
            int d=min(truckSize,v[i].second);
            ans+=d*v[i].first;
            truckSize-=d;
        }
        return ans;
    }
};
