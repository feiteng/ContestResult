class Solution {
public:
    int maximumUnits(vector<vector<int>>& b, int m) {
        sort(b.begin(), b.end(), [](vector<int> a, vector<int> b) {
            return a[1] > b[1];
        });
        int res = 0;
        for (int i = 0; i < b.size() && m; i ++ ) {
            int cur = min(m, b[i][0]);
            res += cur * b[i][1];
            m -= cur;
        }
        return res;
    }
};
