class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        
        sort(boxTypes.begin(),boxTypes.end(),[](vector<int>& a,vector<int>& b)
             { return a[1]<b[1];});
        int len=boxTypes.size();
        int ans=0;
        for(int i=len-1;i>=0;i--)
        {
            int tp=min(truckSize,boxTypes[i][0]);
            ans+=tp*boxTypes[i][1];
            truckSize-=tp;
            
        }
        return ans;
    }
};
