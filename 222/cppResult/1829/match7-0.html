class Solution {
public:
    int maximumUnits(vector<vector<int>>& b, int t) {
        vector<pair<int, int>> v;
        for(int i = 0; i < b.size(); i++){
            v.push_back(make_pair(-b[i][1], b[i][0]));
        }
        sort(v.begin(), v.end());
        int ans = 0;
        for(int i = 0; i < v.size() && t > 0; i++){
            int k = min(v[i].second, t);
            t-=k;
            ans += -v[i].first * k;
        }
        return ans;
    }
};
