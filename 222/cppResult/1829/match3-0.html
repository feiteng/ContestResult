class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(),
            [](vector<int>& v1, vector<int>& v2)
             {
                 return v1[1] < v2[1];
             });

        int j = boxTypes.size() - 1;
        int boxes = 0;
        while (j >= 0 && truckSize > 0)
        {
            int used = min(boxTypes[j][0], truckSize);
            boxes += (used * boxTypes[j][1]);
            truckSize -= used;
            j--;
        }
        
        return boxes;
    }
};
