class Solution {
public:
    int maximumUnits(vector<vector<int>>& bt, int ts) {
        sort(bt.begin(), bt.end(), [](vector<int>& a, vector<int>& b) {
            return a[1] < b[1];
        });
        int res = 0;
        for (int i = bt.size() - 1; i >= 0 && ts > 0; --i) {
            int t = min(ts, bt[i][0]);
            res += t*bt[i][1];
            ts -= t;
        }
        return res;
    }
};
