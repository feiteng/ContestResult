class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(),boxTypes.end(),[&](auto first,auto second)
             {
                return first[1]>second[1];    
             });
        int ans=0;
        for(auto i:boxTypes)
        {
            int take=min(truckSize,i[0]);
            truckSize-=take;
            ans+=take*i[1];
            if(truckSize==0) break;
        }
        return ans;
    }
};
