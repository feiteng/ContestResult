class Solution {
public:
    #define pii pair<int,int>
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        int n = boxTypes.size();
        vector<pii> vec(n,pii(0,0));
        for(int i = 0;i < n; ++i){
            vec[i] = pii(boxTypes[i][1],boxTypes[i][0]);
        }
        sort(vec.begin(), vec.end());
        int ans = 0;
        for(int i = n-1;i >= 0 && truckSize; --i){
            int t = min(truckSize, vec[i].second);
            ans += vec[i].first * t;
            truckSize -= t;
        }
        return ans;
    }
};
