class Solution {
public:
    int maximumUnits(vector<vector<int>>& a, int truckSize) {
        sort(a.begin(),a.end(),[&](vector<int>& x,vector<int>& y) {
            return x[1] > y[1];
        });
        int ans = 0;
        for (int i = 0; i < a.size(); i++) {
            if (truckSize > a[i][0]) {
                truckSize -= a[i][0];
                ans += a[i][0] * a[i][1];
            } else {
                ans += truckSize * a[i][1];
                break;
            }
            // cout << ans << endl;
        }
        return ans;
    }
};
