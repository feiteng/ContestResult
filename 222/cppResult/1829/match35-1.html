class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(), [](vector<int> &v1, vector<int> &v2) {
            return v1[1] > v2[1];
        });
        
        int ret = 0, t = 0;

        for (int i = 0; i < boxTypes.size(); ++i) {
           // printf("%d %d %d %d %d\n", i, t, ret, boxTypes[i][0], boxTypes[i][1]);
            if (t + boxTypes[i][0] > truckSize) {
                ret += (truckSize - t) * boxTypes[i][1];
                break;
            } else {
                ret += boxTypes[i][0] * boxTypes[i][1];
                t += boxTypes[i][0];
            }
        }
        return ret;
    }
};
