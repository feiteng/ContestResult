class Solution {
public:
    int maximumUnits(vector<vector<int>>& b, int t) {
        vector<pair<int,int>> v;
        for (auto& q : b) {
            v.emplace_back(q[1], q[0]);

        }
        sort(v.rbegin(), v.rend());
        int ans = 0;
        for (auto& [a, b] : v) {
            int x = min(b, t);
            t -= x;
            ans += x * a;
        }
        return ans;
    }
};
