class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(begin(boxTypes), end(boxTypes), [&](vector<int> x,  vector<int> y){
            return x[1] > y[1];
        });
        int n = boxTypes.size();
        int ans = 0;
        for(int i = 0; i < n; ++i){
            int d = min(boxTypes[i][0], truckSize);
            truckSize -= d;
            ans += d * boxTypes[i][1];
        }
        return ans;
    }
};
