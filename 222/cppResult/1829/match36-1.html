class Solution {
public:
    int maximumUnits(vector<vector<int>>& A, int K) {
        int n = A.size();
        for (int i = 0; i < n; i++) {
            swap(A[i][0], A[i][1]);
            

        }
        
        sort(A.begin(),A.end());
        reverse(A.begin(), A.end());
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int delta = min(K, A[i][1]);
            
            ans += delta * A[i][0];
            
            K -= delta;
        }
        
        return ans;
        
    }
};
