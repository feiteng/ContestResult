class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(), [](auto& a, auto& b) {
            return a[1] > b[1];
        });
        int res = 0;
        for (auto& b : boxTypes) {
            int n = min(truckSize, b[0]);
            truckSize -= n;
            res += b[1] * n;
            if (truckSize == 0) break;
        }
        return res;
    }
};
