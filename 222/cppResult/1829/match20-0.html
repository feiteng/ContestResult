class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(), [&](vector<int> a, vector<int> b) {
            return a[1] > b[1];
        });
        int ans = 0;
        for(auto &x : boxTypes) {
            int can = min(truckSize, x[0]);
            ans += can * x[1];
            truckSize -= can;
            if(truckSize == 0) break;
        }
        return ans;
    }
};
