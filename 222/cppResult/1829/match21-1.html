class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        vector<pair<int, int> > p;
        for(auto&v:boxTypes) {
            p.push_back({v[1], v[0]});
        }
        sort(p.rbegin(), p.rend());
        int ans = 0;
        for(auto&x:p) {
            int mini = min(truckSize, x.second);
            ans += x.first * mini;
            truckSize -= mini;
        }
        return ans;
    }
};
