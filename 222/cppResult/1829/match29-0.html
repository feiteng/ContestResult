class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(), [](auto& a, auto& b) {
            return a[1] > b[1];
        });
        int res = 0;
        int i = 0;
        while (truckSize > 0 && i < boxTypes.size()) {
            int n = min(truckSize, boxTypes[i][0]);
            truckSize -= n;
            res += n * boxTypes[i][1];
            i += 1;
        }
        return res;
    }
};
