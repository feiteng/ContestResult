class Solution {
public:
    int maximumUnits(vector<vector<int>>& box, int n) {
        sort(box.begin(), box.end(),[](auto&a, auto&b) { return a[1] > b[1]; });
        int res = 0;
        for (auto& b : box) {
            int num = min(n, b[0]);
            res += num * b[1];
            n -= num;
            if (n == 0) break;
        }
        return res;
    }
};
