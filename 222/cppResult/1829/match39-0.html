class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(),  boxTypes.end(), [](const vector<int> &x, const vector<int> &y){return x[1] >= y[1];});
        int res = 0;
        int cnt = 0;
        for(auto boxtype: boxTypes){
            if(cnt + boxtype[0] <= truckSize){
                res += boxtype[0] * boxtype[1]; 
                cnt += boxtype[0];
            }else{
                res += (truckSize - cnt) * boxtype[1];
                break;
            }
        }
        return res;
    }
};
