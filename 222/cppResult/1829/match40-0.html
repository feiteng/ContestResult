class Solution {
public:
    int maximumUnits(vector<vector<int>>& boxTypes, int truckSize) {
        sort(boxTypes.begin(), boxTypes.end(), [](const vector<int>& v1, const vector<int>& v2) -> bool {
            return v1[1] > v2[1];
        });
        int ret = 0;
        int rem = truckSize;
        for (auto& p : boxTypes) {
            if (p[0] < rem) {
                rem -= p[0];
                ret += p[0] * p[1];
            } else {
                ret += rem * p[1];
                break;
            }
        }
        return ret;
    }
};
