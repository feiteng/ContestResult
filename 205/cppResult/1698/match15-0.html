class Solution {
public:
    string modifyString(string s) {
        int n = s.size();
        for (int i = 0; i < n; i++) {
            if (s[i] == '?') {
                for (int j = 0; j < 26; j++) {
                    if (i>0 && 'a'+j == s[i-1]) continue;
                    if (i<n-1 && 'a'+j == s[i+1]) continue;
                    s[i] = 'a'+j;
                    break;
                }
            }
        }
        return s;
    }
};
