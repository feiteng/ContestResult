class Solution {
public:
    string modifyString(string s) {
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] != '?') continue;
            for (char c = 'a'; c <= 'z'; ++c) {
                if (i > 0 && c == s[i-1]) continue;
                if (i+1 < s.size() && c == s[i+1]) continue;
                s[i] = c;
                break;
            }
        }
        return s;
    }
};
