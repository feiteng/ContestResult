class Solution {
public:
    string modifyString(string s) {
        for(int i=0;i<s.length();i++){
            if(s[i]=='?'){
                char x = 'a';
                if(i>0&&x==s[i-1]) x++;
                if(i<s.length()-1&&x==s[i+1]) x++;
                if(i>0&&x==s[i-1]) x++;
                s[i]=x;
            }
        }
        return s;
    }
};
