class Solution {
public:
    string modifyString(string s) {
        int n = s.size();
        if (n == 1) return s == "?" ? "a" : s;
        string res = s;
        for (int i = 0; i < n; ++i) {
            if (s[i] == '?') res[i] = 'a';
        }
        if (s[0] == '?' && res[0] == res[1]) res[0] = res[0] == 'z' ? 'a' : res[0]+1;
        if (s[n-1] == '?' && res[n-1] == res[n-2]) res[n-1] = res[n-1] == 'z' ? 'a' : res[n-1]+1;
        for (int i = 1; i < n-1; ++i) if (s[i] == '?') {
            while (res[i] == res[i+1] || res[i] == res[i-1]) res[i] = res[i] == 'z' ? 'a' : res[i]+1;
        }
        return res;
    }
};
