class Solution {
public:
    string modifyString(string s) {
        int l=s.length();
        if (l==1&&s[0]=='?')
        {
            s[0]='a';
            return s;
        }
        for (int i=0;i<l;i++)
        {
            if (s[i]=='?')
            {
                if (i==0)
                {
                    if (s[i+1]=='a') s[i]='b';
                    else s[i]='a';
                }
                else if (i==l-1)
                {
                    if (s[i-1]=='a') s[i]='b';
                    else s[i]='a';
                }
                else
                {
                    if (s[i-1]=='a'||s[i+1]=='a') s[i]='b';
                    else s[i]='a';
                    if (s[i]=='b'&&(s[i-1]=='b'||s[i+1]=='b')) s[i]='c';
                }
            }
        }
        return s;
    }
};
