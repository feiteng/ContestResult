#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;


class Solution {
public:
    string modifyString(string s) {
		int n = s.size();
		for(int i=0; i<n; i++){
			if(s[i] != '?') continue;
			for(char c='a'; c<='z'; c++){
				if(i-1>=0 && s[i-1]==c) continue;
				if(i+1<n && s[i+1]==c) continue;
				s[i] = c;
				break;
			}
		}

		return s;
        
    }
};
