class Solution {
public:
    string modifyString(string s) {
        int n=s.length();
        for(int i=0;i<n;++i)
            if(s[i]=='?')
                for(int j=0;j<26;++j)
                    if((i-1<0||s[i-1]!='a'+j)&&(i+1>=n||s[i+1]!='a'+j))
                    {
                        s[i]='a'+j;
                        break;
                    }
        return s;
    }
};
