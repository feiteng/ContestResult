class Solution {
public:
    string modifyString(string s)
    {
        int len = s.size();
        for (int i = 0; i < len; ++i)
        {
            if (s[i] == '?')
            {
                char c1 = i - 1 >= 0 ? s[i - 1] : ' ';
                char c2 = i + 1 < len ? s[i + 1] : ' ';

                if (c1 != 'a' && c2 != 'a')
                {
                    s[i] = 'a';
                }
                else if (c1 != 'b' && c2 != 'b')
                {
                    s[i] = 'b';
                }
                else
                {
                    s[i] = 'c';
                }
            }
        }
        return s;
    }
};
