class Solution {
public:
    string modifyString(string s) {
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='?')
            {
                
                for(int j=0;j<26;j++)
                {
                    bool is=true;
                    if(i>0&&s[i-1]=='a'+j)is=false;
                    if(i<s.size()-1&&s[i+1]=='a'+j)is=false;
                        if(is)
                        {
                            s[i]='a'+j;
                            break;
                        }
                }
            }
            
        }
        
        return s;
    }
};
