class Solution {
public:
    string modifyString(string s) {
        int n = s.size();
        for(int i = 0; i < n; i++){
            if(s[i] == '?'){
                for(int j = 0; j < 26; j++){
                    if(i && s[i-1] == j + 'a')continue;
                    if(i < n-1 && s[i+1] == j + 'a')continue;
                    s[i] = j + 'a';
                    break;
                }
            }
        }
        return s;
    }
};
