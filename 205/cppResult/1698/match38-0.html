class Solution {
public:
    string modifyString(string s) {
        int len=s.length();
        for(int i=0;i<len;i++){
            if(s[i]=='?'){
                char c1='#',c2='#';
                if(i-1>=0) c1=s[i-1];
                if(i+1<len) c2=s[i+1];
                for(int j='a';j<='z';j++){
                    if(j!=c1&&j!=c2){
                        s[i]=j;
                        break;
                    }
                }
            }
        }
        return s;
    }
};
