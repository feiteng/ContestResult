class Solution {
public:
    string modifyString(string s) {
        int N = s.length();
        for(int i = 0; i < N; i++)
        {
            if(s[i] == '?')
            {
                char a = (i > 0) ? s[i-1] : '*';
                char b = (i+1 < N) ? s[i+1] : '*';

                for(char z = 'a'; z <= 'z'; z++)
                {
                    if(z != a && z != b)
                    {
                        s[i] = z;
                        break;
                    }
                }
            }
        }
        
        return s;
    }
};
