class Solution {
public:
    string modifyString(string s) {
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == '?') {
                char modify = 'a';
                if (i > 0 && s[i - 1] == modify)    modify++;
                if (i + 1 < s.length() && s[i + 1] == modify)   modify++;
                if (i > 0 && s[i - 1] == modify)    modify++;
                s[i] = modify;
            }
        }
        return s;
    }
};
