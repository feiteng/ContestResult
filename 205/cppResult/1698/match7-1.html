class Solution {
public:
    string modifyString(string s) {
        for(int i = 0; i < s.length(); i ++) {
            if(s[i] == '?') {
                for(char x= 'a'; x<='z';x++) {
                    if( (i ==0 || x != s[i-1]) && (i == s.length() - 1 || x!=s[i+1])) {
                        s[i] = x;
                        break;
                    }
                }
            }
        }
        
        return s;
    }
};
