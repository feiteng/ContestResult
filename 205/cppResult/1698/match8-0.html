class Solution {
public:
    string modifyString(string s) {
        int n = s.size();
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '?') {
                for (char x = 'a'; x <= 'z'; x++) {
                    if ((!i || x != s[i - 1]) && ((i == n - 1) || x != s[i + 1])) {
                        s[i] = x;
                        break;
                    }
            
                }
            }
        }
        
        return s;
    }
};
