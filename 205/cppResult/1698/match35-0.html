class Solution {
public:
    string modifyString(string s) {
        int n=s.size();
        for(int i=0;i<n;i++){
            if(s[i]!='?') continue;
            for(char c='a';c<='z';c++){
                if(i>0&&c==s[i-1]) continue;
                if(i+1<n&&c==s[i+1]) continue;
                s[i]=c;
                break;
            }
        }
        return s;
    }
};
