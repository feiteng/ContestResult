#define fi first
#define se second


class Solution {
public:
    string modifyString(string s) {
        vector<pair<int, int>> v;
        int n = s.size();
        int l = n+1, r = -1;
        for (int i = 0; i < n; i++) {
            if (s[i] == '?') {
                l = min(l, i);
                r = max(r, i);
            } else {
                if (r >= l) {
                    v.emplace_back(l, r);
                    l = n+1, r = -1;
                }
            }
        }
        if (r >= l) {
            v.emplace_back(l, r);
        }
        for (auto& p: v) {
            char lc = 0, rc = 0;
            if (p.fi > 0) lc = s[p.fi-1];
            if (p.se < n-1) rc = s[p.se+1];
            vector<char> cs;
            for (char c = 'a'; c <= 'z'; c++) {
                if (c != lc && c != rc) 
                    cs.emplace_back(c);
            }
            for (int i = p.fi; i <= p.se; i++) {
                s[i] = cs[i&1];
            }
        }
        return s;
    }
};
