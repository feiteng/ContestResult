class Solution {
public:
    string modifyString(string s) {
        string str;
        for (int i = 0; i < 26; ++i) str.push_back('a' + i);
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '?') {
                string exclude;
                if (i > 0) exclude.push_back(s[i - 1]);
                if (i + 1 < s.size()) exclude.push_back(s[i + 1]);
                s[i] = str[str.find_first_not_of(exclude)];
            }
        }
        return s;
    }
};
