class Solution {
public:
    string modifyString(string s) {
        for(int i =0; i < s.size(); i ++){
            int block = 0;
            if(i){
                block = block | (1<<(s[i-1]-'a'));
            }
            if(i+1!=s.size() and s[i+1]!='?'){
                block = block | (1<<(s[i+1]-'a'));
            }
            if(s[i] !='?') continue;
            for(int x =0 ; x < 4; x ++){
                if(0==(block&(1<<x))){
                    s[i] = x + 'a';
                }
            }
        }
        return s;
    }
};
