class Solution {
public:
	string modifyString(string s) {
		int n = s.length();
		char prev = ' ';
		for (int i = 0; i < n; i++) {
			if (s[i] != '?') {
				prev = s[i];
				continue;
			}
			char next = ' ';
			for (int j = i + 1; j < n; j++)
				if (s[j] != '?') {
					next = s[j];
					break;
				}
			for (char c = 'a'; c <= 'z'; c++)
				if (c != prev && c != next) {
					prev = c;
					break;
				}
			for (char c = 'a'; c <= 'z'; c++)
				if (c != prev && c != next) {
					next = c;
					break;
				}
			for (int j = i; j < n && s[j] == '?'; j++) {
				if ((j - i) & 1)
					s[j] = next;
				else
					s[j] = prev;
			}
		}
		return s;
	}
};
