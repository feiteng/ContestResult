class Solution {
public:
    string modifyString(string s) {
        int len = s.size();
        for(int i=0; i<len; i++) {
            if(s[i]=='?') {
                char a = 0;
                char b = 0;
                if(i>0) {
                    a = s[i-1];
                }
                if(i<len-1) {
                    b = s[i+1];
                }
                for(char c='a'; c<='z'; c++) {
                    if(c!=a && c!=b) {
                        s[i] = c;
                        break;
                    }
                }
            }
        }
        return s;
    }
};
