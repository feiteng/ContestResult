class Solution {
public:
    string modifyString(string s) {
        int n = s.length();
        for (int i=0; i<n; i++) {
            if (s[i] == '?') {
                for (char j='a'; j<='z'; j++) {
                    if (i && s[i-1]==j) continue;
                    if (i+1<n && s[i+1]==j) continue;
                    s[i] = j;
                    break;
                }
            }
        }
        return s;
    }
};
