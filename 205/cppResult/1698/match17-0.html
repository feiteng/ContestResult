class Solution {
public:
    string modifyString(string s) {
        for(int i = 0;i<s.size();i++){
            if(s[i]=='?'){
                for(int j = 'a';j<='z';j++){
                    if((i==0||s[i-1]!=j)&&(i==s.size()-1||s[i+1]!=j)){
                        s[i]=j;
                        break;
                    }
                }
            }
        }
        return s;
    }
};
