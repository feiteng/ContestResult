class Solution {
public:
    char get(char a,char b)
    {
        for(char c='a';c<='z';c++)
            if(c!=a&&c!=b) return c;
        return ' ';
    }
    string modifyString(string s) {
        int n=s.length();
        string ans;
        for(int i=0;i<n;i++)
        {
            if(s[i]!='?') ans+=s[i];
            else
            {
                char pre,nxt;
                if(i!=0&&s[i-1]!='?') pre=s[i-1];
                else if(i!=0&&s[i-1]=='?') pre=ans[i-1];
                else pre=' ';
                if(i!=n-1&&s[i+1]!='?') nxt=s[i+1];
                else nxt=' ';
                ans+=get(pre,nxt);
            }
        }
        return ans;
    }
};
