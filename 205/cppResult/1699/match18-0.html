class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2 = nums2.size();
        int ret = 0;
        unordered_map<long long, int> s1, s2;
        for(int i=0; i<n1; i++) {
            s1[(long long)nums1[i]*nums1[i]]++;
        }
        for(int i=0; i<n2; i++) {
            s2[(long long)nums2[i]*nums2[i]]++;
            for(int j=i+1; j<n2; j++) {
                    ret += s1[(long long)nums2[i]*nums2[j]];
            }
        }
        //cout << ret<<endl;
        for(int i=0; i<n1; i++) {
            for(int j=i+1;j<n1; j++) {
                    ret += s2[(long long)nums1[i]*nums1[j]];
            }
        }
        //cout << ret << endl;
        return ret;
    }
};
