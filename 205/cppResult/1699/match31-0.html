#define ll long long
class Solution {
public:
    int numTriplets(vector<int>& a, vector<int>& b) {
        
        int n = a.size();
        int m = b.size();
        
        map<ll,ll> x,y;
        
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                x[(ll)a[i]*a[j]]++;
            }
        }
        
        for(int i=0;i<m;i++){
            for(int j=i+1;j<m;j++){
                y[(ll)b[i]*b[j]]++;
            }
        }
        
        ll ans = 0;
        
        for(int i=0;i<n;i++){
            ans += y[(ll)a[i]*a[i]];
        }
        
        for(int j=0;j<m;j++){
            ans += x[(ll)b[j]*b[j]];
        }
        
        return ans;
        
    }
};
