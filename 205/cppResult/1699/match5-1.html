class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2 = nums2.size();
        int ret = 0;
        

        map<long long, int> prod1, prod2;
        for (int j = 0; j < n1; j++) {
            for (int k = j + 1; k < n1; k++) {
                long long prod = (long long)nums1[j] * (long long)nums1[k];
                prod1[prod] += 1;
            }
        }
        for (int j = 0; j < n2; j++) {
            for (int k = j + 1; k < n2; k++) {
                long long prod = (long long)nums2[j] * (long long)nums2[k];
                prod2[prod] += 1;
            }
        }
        

        for (int i = 0; i < n1; i++) {
            long long prod = (long long)nums1[i] * (long long)nums1[i];
            ret += prod2[prod];
        }
        
        for (int i = 0; i < n2; i++) {
            long long prod = (long long)nums2[i] * (long long)nums2[i];
            ret += prod1[prod];
        }
        return ret;
    }
};
