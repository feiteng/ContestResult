class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long,long long> s1,m1,m2,s2;
        for (int i = 0; i < nums1.size(); i++) {
            s1[(long long)nums1[i]*nums1[i]]++;
        }
        for (int i = 0; i < nums2.size(); i++) {
            s2[(long long)nums2[i]*nums2[i]]++;
        }
        int res = 0;
        for (int i = 0; i < nums1.size(); i++) {
            for (int j = i + 1; j < nums1.size(); j++) {
                res += s2[(long long)nums1[i] * nums1[j]];
            }
        }
        for (int i = 0; i < nums2.size(); i++) {
            for (int j = i + 1; j < nums2.size(); j++) {
                res += s1[(long long)nums2[i] * nums2[j]];
            }
        }
        return res;
    }
};
