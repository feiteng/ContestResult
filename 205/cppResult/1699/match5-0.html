class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long, int> mp1, mp2;
        
        for (int i = 0; i < nums1.size(); i++) {
            for (int j = i + 1; j < nums1.size(); j++) {
                long long cur = (long long)nums1[i] * (long long)nums1[j];
                mp1[cur]++;
            }
        }
        
        for (int i = 0; i < nums2.size(); i++) {
            for (int j = i + 1; j < nums2.size(); j++) {
                long long cur = (long long)nums2[i] * (long long)nums2[j];
                mp2[cur]++;
            }
        }
        
        int res = 0;
        for (int i = 0; i < nums1.size(); i++) {
            long long cur = (long long) nums1[i] * (long long) nums1[i];
            res += mp2[cur];
        }
        
        for (int i = 0; i < nums2.size(); i++) {
            long long cur = (long long) nums2[i] * (long long) nums2[i];
            res += mp1[cur];
        }
        
        return res;
    }
};
