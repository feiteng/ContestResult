class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<uint64_t, int> s1, s2;
        
        for (const uint64_t x : nums1) s1[x*x]++;
        for (const uint64_t x : nums2) s2[x*x]++;
        
        int cnt = 0;
        for (int i = 0; i < nums1.size(); ++i) {
            for (int j = i+1; j < nums1.size(); ++j) {
                if (s2.count(((uint64_t)nums1[i])*nums1[j]))
                cnt += s2[((uint64_t)nums1[i])*nums1[j]];
            }
        }
        for (int i = 0; i < nums2.size(); ++i) {
            for (int j = i+1; j < nums2.size(); ++j) {
                if ( s1.count(((uint64_t)nums2[i])*nums2[j]))
                cnt += s1[((uint64_t)nums2[i])*nums2[j]];
            }
        }
            
        return cnt;
    }
};
