class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long int, int> um1, um2;
        int n1 = nums1.size(), n2 = nums2.size();
        long long int base = 1;
        for (int i = 0; i < n1; i++) {
            for (int j = i + 1; j < n1; j++) {
                
                um1[base * nums1[i] * nums1[j]]++;
            }
        }
        for (int i = 0; i < n2; i++) {
            for (int j = i + 1; j < n2; j++) {
                um2[base * nums2[i] * nums2[j]]++;
            }
        }
        int res = 0;
        for (int i = 0; i < n1; i++) {
            res += um2[base * nums1[i] * nums1[i]];
        }
        for (int i = 0; i < n2; i++) {
            res += um1[base * nums2[i] * nums2[i]];
        }
        return res;
    }
};
