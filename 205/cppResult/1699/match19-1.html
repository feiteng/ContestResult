#include <bits/stdc++.h>

using namespace std;


class Solution {
public:
    int numTriplets(vector<int> &nums1, vector<int> &nums2) {
        map<long long, int> M;
        for (int i = 0; i < nums1.size(); i++)
            for (int j = i + 1; j < nums1.size(); j++)
                M[(long long) nums1[i] * nums1[j]]++;
        int ans = 0;
        for (int i : nums2)
            ans += M[(long long) i * i];
        swap(nums1, nums2);
        M.clear();
        for (int i = 0; i < nums1.size(); i++)
            for (int j = i + 1; j < nums1.size(); j++)
                M[(long long) nums1[i] * nums1[j]]++;
        for (int i : nums2)
            ans += M[(long long) i * i];
        return ans;
    }
};
