class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long int,int> m1,m2;
        for(auto x:nums1) m1[1l*x*x]++;
        for(auto x:nums2) m2[1l*x*x]++;
        int ans=0;
        for(int i=0;i+1<nums1.size();++i){
            
            for(int j=i+1;j<nums1.size();++j){
                ans+=m2[1l*nums1[i]*nums1[j]];
            }
        }
        for(int i=0;i+1<nums2.size();++i){
            
            for(int j=i+1;j<nums2.size();++j){
                ans+=m1[1l*nums2[i]*nums2[j]];
            }
        }
        return ans;
        
    }
};
