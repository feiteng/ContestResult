#define ll long long 
class Solution {
public:
    unordered_map<ll , ll > Q1 , Q2;
    int numTriplets(vector<int>& a, vector<int>& b) {
        int n = a.size() , m = b.size();
        for (int i = 0 ; i < n ; i++){
            for (int j = i + 1 ; j < n ; j++){
                Q1[1ll * a[i] * a[j]]++;
            }
        }
        for (int i = 0 ; i < m ; i++){
            for (int j = i + 1 ; j < m ; j++){
                Q2[1ll * b[i] * b[j]]++;
            }
        }
        ll ans = 0;
        for (int i = 0 ; i < n ; i++){
            ans += Q2[1ll * a[i] * a[i]];
        }
        for (int i = 0 ; i < m ; i++){
            ans += Q1[1ll * b[i] * b[i]];
        }
        return ans;
    }
};
