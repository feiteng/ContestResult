class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        map<long long, int> ct1;
        map<long long, int> ct2;
        for (int x : nums1)
            ++ct1[x * 1ll * x];
        for (int x : nums2)
            ++ct2[x * 1ll * x];

        int res = 0;
        for (int i = 0; i < nums1.size(); ++i) {
            for (int j = i + 1; j < nums1.size(); ++j) {
                long long tmp = nums1[i] * 1ll * nums1[j];
                if (ct2.count(tmp) > 0)

                    res += ct2[tmp];
            }
        }
        for (int i = 0; i < nums2.size(); ++i) {
            for (int j = i+1; j < nums2.size(); ++j) {
                long long tmp = nums2[i] * 1ll * nums2[j];
                if (ct1.count(tmp) > 0)
                    res += ct1[tmp];
            }
        }
        
        return res;
        
    }
};
