class Solution {
public:
    long long ll = 1;
    int numTriplets(vector<int>& nums1, vector<int>& nums2) 
    {
        int ret = 0;
        ret += solve(nums1, nums2);
        ret += solve(nums2, nums1);
        return ret;
    }
    
    int solve(vector<int>& nums1, vector<int>& nums2)
    {
        unordered_map<long long, int> sq;
        for(int i = 0; i < nums1.size(); i++)
        {
            long long val = ll * nums1[i] * nums1[i];
            sq[val]++;
        }
        
        unordered_map<long long, int> mul;
        for(int i = 0; i < nums2.size(); i++)
        {
            for(int j = i + 1; j < nums2.size(); j++)
            {
                long long val = ll * nums2[i] * nums2[j];
                mul[val]++;
            }
        }
        
        int ret = 0;
        for(auto& [key, val] : sq)
        {
            if(mul.count(key))
            {
                ret += val * mul[key];
            }
        }
        return ret;
    }
};
