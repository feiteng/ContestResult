class Solution {
public:
    #define ll long long
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        map<ll,ll> mp1,mp2;
        for(auto e: nums1) mp1[1LL*e*e]++;

        for(auto e: nums2) mp2[1LL*e*e]++;
        
        int ans = 0;
        for(int i = 0; i < nums1.size(); i++){
            for(int j = i+1; j < nums1.size(); j++){
                if (mp2[1LL*nums1[i]*nums1[j]]) ans+= mp2[1LL*nums1[i]*nums1[j]];
            }
        }
        for(int i = 0; i < nums2.size(); i++){
            for(int j = i+1; j < nums2.size(); j++){
                if (mp1[1LL*nums2[i]*nums2[j]]) ans+= mp1[1LL*nums2[i]*nums2[j]];
            }
        }
        
        return ans;
        
    }
};
