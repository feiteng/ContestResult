class Solution {
    typedef long long ll;

public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        vector<ll> mul1,mul2;
        int n1=nums1.size(),n2=nums2.size();
        for(int i=0;i<n1;i++){
            for(int j=i+1;j<n1;j++){
                mul1.push_back((ll)nums1[i]*(ll)nums1[j]);

            }
        }
        sort(mul1.begin(),mul1.end());
        for(int i=0;i<n2;i++){
            for(int j=i+1;j<n2;j++){
                mul2.push_back((ll)nums2[i]*(ll)nums2[j]);
            }
        }
        sort(mul2.begin(),mul2.end());

        int cnt=0;
        for(int i=0;i<n1;i++){
            cnt+=upper_bound(mul2.begin(),mul2.end(),ll(nums1[i])*ll(nums1[i]))-lower_bound(mul2.begin(),mul2.end(),ll(nums1[i])*ll(nums1[i]));
        }
        for(int i=0;i<n2;i++){
            cnt+=upper_bound(mul1.begin(),mul1.end(),ll(nums2[i])*ll(nums2[i]))-lower_bound(mul1.begin(),mul1.end(),ll(nums2[i])*ll(nums2[i]));
        }
        return cnt;
    }
};
