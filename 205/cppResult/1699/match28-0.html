class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        map<long long, int> mp1;
        map<long long, int> mp2;
        
        for (int num : nums1) {
            mp1[(long long) num * num]++;
        }
        for (int num : nums2) {
            mp2[(long long) num * num]++;
        }
        
        int result = 0;
        
        for (int i = 0; i < nums1.size(); i++) {
            for (int j = i + 1; j < nums1.size(); j++) {
                long long tmp = (long long) nums1[i] * nums1[j];
                result += mp2[tmp];
            }
        }
        for (int i = 0; i < nums2.size(); i++) {
            for (int j = i + 1; j < nums2.size(); j++) {
                long long tmp = (long long) nums2[i] * nums2[j];
                result += mp1[tmp];
            }
        }
        
        return result;
            
    }
};
