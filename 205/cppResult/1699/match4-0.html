#define rep(i,l,r) for (int i=l;i<=r;++i)
typedef long long ll;
class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int t[100001]; int cnt = 0;
        memset(t, 0, sizeof(t)); for (auto i: nums1) t[i]++;
        for (int i = 0; i < nums2.size(); i++)
            for (int j = i + 1; j < nums2.size(); j++) {
                double c = sqrt((double)nums2[i] * nums2[j]);
                int cc = (int)c;
                if (abs(cc - c) < 1e-3) cnt += t[cc];
            }
        memset(t, 0, sizeof(t)); for (auto i: nums2) t[i]++;
        for (int i = 0; i < nums1.size(); i++)
            for (int j = i + 1; j < nums1.size(); j++) {
                double c = sqrt((double)nums1[i] * nums1[j]);
                int cc = (int)c;
                if (abs(cc - c) < 1e-3) cnt += t[cc];
            }
        return cnt;
    }
};
