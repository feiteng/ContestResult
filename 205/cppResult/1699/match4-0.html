class Solution {
public:
    long Check(const vector<int>& a, const vector<int>& b) {
        long res = 0;
        for (long x : a) {
            long xx = x * x;
            unordered_map<long, int> seen;
            for (long y : b) {
                if (xx % y == 0) {
                    if (auto it = seen.find(xx / y); it != seen.end()) {
                        res += it->second;
                    }
                }
                ++seen[y];
            }
        }
        return res;
    }
    
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        return Check(nums1, nums2) + Check(nums2, nums1);
    }
};
