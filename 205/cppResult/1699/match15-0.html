using LL=long long;
class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int ans = 0;
        unordered_map<LL,int> sHave1;
        unordered_map<LL,int> sHave2;
        for(auto x:nums1){
            sHave1[(LL)x*x]++;
        }
        for(auto x:nums2){
            sHave2[(LL)x*x]++;
        }
        int N1=nums1.size();
        int N2=nums2.size();
        for(int j=0;j<N1;j++){
            for(int k=j+1;k<N1;k++){
                LL t=(LL)((LL)nums1[j]*nums1[k]);
                auto it = sHave2.find(t);
                if(it!=sHave2.end()){
                    ans+=it->second;
                }
            }
        }
        for(int j=0;j<N2;j++){
            for(int k=j+1;k<N2;k++){
                LL t=(LL)((LL)nums2[j]*nums2[k]);
                auto it = sHave1.find(t);
                if(it!=sHave1.end()){
                    ans+=it->second;
                }
            }
        }
                
        return ans;        
    }
};
