class Solution {
public:
    static const int N=1e5+5;
    int cnt1[N],cnt2[N];
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int l1=nums1.size(),l2=nums2.size(),ans=0;
        for (int i=0;i<l1;i++) ++cnt1[nums1[i]];
        for (int i=0;i<l2;i++) ++cnt2[nums2[i]];
        for (int i=0;i<l1;i++)
            for (int j=i+1;j<l1;j++)
            {
                long long sq=sqrt(1ll*nums1[i]*nums1[j]);
                if (sq<N&&sq*sq==1ll*nums1[i]*nums1[j])
                    ans+=cnt2[sq];
            }
        for (int i=0;i<l2;i++)
            for (int j=i+1;j<l2;j++)
            {
                long long sq=sqrt(1ll*nums2[i]*nums2[j]);
                if (sq<N&&sq*sq==1ll*nums2[i]*nums2[j])
                    ans+=cnt1[sq];
            }
        return ans;
    }
};
