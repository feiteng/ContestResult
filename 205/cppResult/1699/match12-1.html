#define ll long long int
class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();
        

        ll count = 0;
        unordered_map<ll, ll> mp1, mp2;
        for(int i = 0 ; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                mp1[(ll)nums1[i] * nums1[j]]++;
            }
        }
        
        for(int i =0 ; i < m; i++) {
            for(int j = i + 1; j < m; j++) {
                mp2[(ll)nums2[i] * nums2[j]]++;
            }
        }
        

        for(int i = 0; i < n; i++) {
            count += mp2[(ll)nums1[i] * nums1[i]];
        }
        
        for(int i = 0; i < m; i++) {
            count += mp1[(ll)nums2[i] * nums2[i]];
        }
        
        return count;
    }
};
