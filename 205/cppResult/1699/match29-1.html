class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long, int> M;

        int ans = 0;
        for(auto i:nums1){
            M[1ll*i*i]++;
        }
        for(int j=0; j<nums2.size()-1; j++){
            for(int k=j+1; k<nums2.size(); k++){
                    ans+= M[1ll*nums2[j]*nums2[k]];
            }
        }
        M.clear();
        swap(nums1, nums2);
        for(auto i:nums1){
            M[1ll*i*i]++;
        }

        for(int j=0; j<nums2.size()-1; j++){
            for(int k=j+1; k<nums2.size(); k++){
                    ans+= M[1ll*nums2[j]*nums2[k]];
            }
        }
        return ans;
    }
};
