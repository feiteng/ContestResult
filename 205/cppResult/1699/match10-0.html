class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long,int> m1,m2;
        for(int q:nums1)
            ++m1[q];
        for(int q:nums2)
            ++m2[q];
        int res=0;
        for(long long q:nums1){
            for(auto& z:m2){
                if(z.first>q)
                    continue;
                if(z.first==q){
                    res+=z.second*(z.second-1)/2;
                }
                else if(q*q%z.first==0){
                    long long t=q*q/z.first;
                    if(m2.count(t)){
                        res+=z.second*m2[t];
                    }
                }
            }
        }
        for(long long q:nums2){
            for(auto& z:m1){
                if(z.first>q)
                    continue;
                if(z.first==q){
                    res+=z.second*(z.second-1)/2;
                }
                else if(q*q%z.first==0){
                    long long t=q*q/z.first;
                    if(m1.count(t)){
                        res+=z.second*m1[t];
                    }
                }
            }
        }
        return res;
    }
};
