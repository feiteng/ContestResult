class Solution {
    int dp(vector<int>& nums1, vector<int>& nums2){
         unordered_map<long long,int> mm;
        for(long long n:nums1) mm[n*n]++;
        int ans=0;
        for(int j=0;j<nums2.size();j++)
            for(int k=j+1;k<nums2.size();k++){
                long long t1=nums2[j],t2=nums2[k];
                if(mm.find(t1*t2)!=mm.end()) ans+=mm[t1*t2];
            }
        return ans;
    }
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        return dp(nums1,nums2)+dp(nums2,nums1);
    }
};
