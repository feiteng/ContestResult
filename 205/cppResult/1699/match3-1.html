#define ll long long

class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<ll,ll> ma1,ma2;
        for(auto x:nums1)
            ma1[1ll*x*x]++;
        for(auto x:nums2)
            ma2[1ll*x*x]++;
        ll ans=0;
        for(int j=0;j<nums1.size();++j)
            for(int k=j+1;k<nums1.size();++k)
                ans+=ma2[1ll*nums1[j]*nums1[k]];
        for(int j=0;j<nums2.size();++j)
            for(int k=j+1;k<nums2.size();++k)
                ans+=ma1[1ll*nums2[j]*nums2[k]];
        return ans;
    }
};
