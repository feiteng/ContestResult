typedef long long LL;
class Solution {
    unordered_map<LL, int> ctr;
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                ctr[(LL) nums1[i] * nums1[j]]++;
            }
        }
            
        int res = 0;
        for (int i = 0; i < m; i++) {
            res += ctr[(LL) nums2[i] * nums2[i]];
        }
        
        ctr.clear();
        for (int i = 0; i < m; i++) {
            for (int j = i + 1; j < m; j++) {
                ctr[(LL) nums2[i] * nums2[j]]++;
            }
        }
            
        for (int i = 0; i < n; i++) {
            res += ctr[(LL) nums1[i] * nums1[i]];
        }
        
        
        return res;
    }
};
