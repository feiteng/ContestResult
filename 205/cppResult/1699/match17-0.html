class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2 = nums2.size(), ans = 0;
        unordered_map<long long, int> m1, m2;
        for(int i = 0; i < n1; i++) m1[1LL * nums1[i] * nums1[i]]++;
        for(int i = 0; i < n2; i++) m2[1LL * nums2[i] * nums2[i]]++;
        for(int i = 0; i < n1; i++) {
            for(int j = i + 1; j < n1; j++) {
                ans += m2[1LL * nums1[i] * nums1[j]];
            }
        }
        for(int i = 0; i < n2; i++) {
            for(int j = i + 1; j < n2; j++) {
                ans += m1[1LL * nums2[i] * nums2[j]];
            }
        }
        return ans;
    }
};
