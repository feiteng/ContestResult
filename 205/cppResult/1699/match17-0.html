class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<long long, long long> ex1, ex2;
        for (auto e: nums1) ex1[(long long)e * e] ++;
        for (auto e: nums2) ex2[(long long)e * e] ++;
        int ans = 0;
        for (int i = 0; i < nums1.size(); i ++) {
            for (int j = 0; j < i; j++) {
                ans += ex2[(long long)nums1[i] * nums1[j]];
            }
        }
        for (int i = 0; i < nums2.size(); i ++) {
            for (int j = 0; j < i; j++) {
                ans += ex1[(long long)nums2[i] * nums2[j]];
            }
        }
        return ans;
    }
};
