class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        map<long long, int> m;
        for (int i = 0; i < nums1.size(); i++) {
            for (int j = 0; j < i; j++) {
                m[1LL*nums1[i]*nums1[j]]++;
            }
        }
        int ans = 0;
        for (int i = 0; i < nums2.size(); i++) {
            ans += m[1LL*nums2[i]*nums2[i]];
        }
        m.clear();
        for (int i = 0; i < nums2.size(); i++) {
            for (int j = 0; j < i; j++) {
                m[1LL*nums2[i]*nums2[j]]++;
            }
        }
        for (int i = 0; i < nums1.size(); i++) {
            ans += m[1LL*nums1[i]*nums1[i]];
        }
        return ans;
    }
};
