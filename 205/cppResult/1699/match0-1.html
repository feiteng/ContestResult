#define ll long long
#define rep(i,a,b) for(int i=a;i<=b;i++)
#define maxn 1000005

ll a[maxn],b[maxn];


map <ll,int> p1,p2;

class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        for(int i=0;i<nums1.size();i++) a[i+1]=nums1[i];
        for(int i=0;i<nums2.size();i++) b[i+1]=nums2[i];
        int n=nums1.size(),m=nums2.size(); p1.clear(),p2.clear();
        rep(i,1,n) p1[a[i]*a[i]]++;
        rep(i,1,m) p2[b[i]*b[i]]++;
        int ans=0;
        rep(i,1,n) rep(j,i+1,n) ans+=p2[a[i]*a[j]];
        rep(i,1,m) rep(j,i+1,m) ans+=p1[b[i]*b[j]];
        return ans;
    }
};
