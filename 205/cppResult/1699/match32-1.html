inline int read() {
  int x = 0, neg = 1; char op = getchar();
  while (!isdigit(op)) { if (op == '-') neg = -1; op = getchar(); }
  while (isdigit(op)) { x = 10 * x + op - '0'; op = getchar(); }
  return neg * x;
}
inline void print(int x) {
  if (x < 0) { putchar('-'); x = -x; }
  if (x >= 10) print(x / 10);
  putchar(x % 10 + '0');
}

class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
      map<ll,ll> memo;

      int n = SZ(nums1),m = SZ(nums2);
      for(int i = 0 ; i < n ; i++){
        for(int j = i + 1 ; j < n ; j++){
          ll tmp = 1ll * nums1[i] * nums1[j];
          memo[tmp]++;
        }
      }
      int ans = 0;

      for(int i = 0 ; i < m ; i++){
        ll tmp = 1ll * nums2[i] * nums2[i];
        ans += memo[tmp];
      }
      memo.clear();
      for(int i = 0 ; i < m ; i++){
        for(int j = i + 1 ; j < m ; j++){
          ll tmp = 1ll * nums2[i] * nums2[j];
          memo[tmp]++;
        }
      }
      for(int i = 0 ; i < n ; i++){
        ll tmp= 1ll * nums1[i] * nums1[i];
        ans += memo[tmp];
      }
      return ans;
    }
};
