class Solution {
public:
  typedef long long ll;
  int numTriplets(vector<int> &nums1, vector<int> &nums2) {
    ll ans = 0;
    map<ll, ll> w;
    for (int i = 0; i < nums1.size(); i++) {
      for (int j = i + 1; j < nums1.size(); j++) {
        ll now = ll(1) * nums1[i] * nums1[j];
        w[now]++;
      }
    }
    for (int i = 0; i < nums2.size(); i++) {
      ll now = ll(1) * nums2[i] * nums2[i];
      ans += w[now];
    }

    w.clear();
    swap(nums1, nums2);

    for (int i = 0; i < nums1.size(); i++) {
      for (int j = i + 1; j < nums1.size(); j++) {
        ll now = ll(1) * nums1[i] * nums1[j];
        w[now]++;
      }
    }
    for (int i = 0; i < nums2.size(); i++) {
      ll now = ll(1) * nums2[i] * nums2[i];
      ans += w[now];
    }
    
    
    return ans;
  }
};
