class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int cnt = 0;
        unordered_map<long long int,int> nums1Product;
        unordered_map<long long int,int> nums2Product;
        for(int i=0; i<nums1.size(); i++) {
            long long int val = nums1[i];
            for(int j=i+1; j<nums1.size(); j++) {
                nums1Product[val*nums1[j]]++;
            }
        }
        
        for(int i=0; i<nums2.size(); i++) {
            long long int val = nums2[i];
            for(int j=i+1; j<nums2.size(); j++) {
                nums2Product[val*nums2[j]]++;
            }
        }
        
        for(int i=0; i<nums1.size(); i++) {
            long long int val = nums1[i];
            cnt+=nums2Product[val*val];
        }
        
        for(int i=0; i<nums2.size(); i++) {
            long long int val = nums2[i];
            cnt+=nums1Product[val*val];
        }
        
        return cnt;
    }
};
