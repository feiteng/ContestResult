class Solution {
public:
    using LL = long long;
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<LL,int> t1,t2;
        
        int m = nums1.size();
        int n = nums2.size();
        for(int i=0;i<m;i++){
            for(int j=i+1;j<m;j++){
                t1[1ll*nums1[i]*nums1[j]]++;
            }
        }
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                t2[1ll*nums2[i]*nums2[j]]++;
            }
        }
        
        LL res = 0;
        for(int i=0;i<m;i++){
            res+=t2[1ll*nums1[i]*nums1[i]];
        }
        for(int i=0;i<n;i++){
            res+=t1[1ll*nums2[i]*nums2[i]];
        }
        // for(int i=0;i<n;i++){
        //     for(int j=0;j<m;j++){
        //         if((1ll*nums2[i]*nums2[i])%nums1[j]) continue;
        //         res+=t1[(1ll*nums2[i]*nums2[i])/nums1[j]];
        //     }
        // }
        return res;
    }
};
