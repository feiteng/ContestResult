class Solution {
public:
    using ll = long long;
    int numTriplets(vector<int>& a, vector<int>& b) {
        int n = a.size();
        int m = b.size();
        
        unordered_map<ll,ll> mp1;
        for(int j=0;j<m;j++)
        {
            for(int k = j+1 ;k<m;k++)
            {
                mp1[(ll)b[j]*b[k]] ++;
            }
        }
        int cnt = 0;
        for(auto x:a)
        {
            cnt += mp1[(ll)x*x];
        }
        
        unordered_map<ll,ll> mp2;
        for(int j=0;j<n;j++)
        {
            for(int k = j+1 ;k<n;k++)
            {
                mp2[(ll)a[j]*a[k]] ++;
            }
        }
    
        for(auto x:b)
        {
            cnt += mp2[(ll)x*x];
        }
        return cnt;
    }
};
