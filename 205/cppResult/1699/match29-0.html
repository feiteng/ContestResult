class Solution {
public:
    typedef long long LL;
    int get(vector<int>& nums1, vector<int>& nums2)
    {
        unordered_map<LL,int> cnt;
        for(int i:nums1)
            cnt[1LL*i*i]++;
        int n=nums2.size();
        int ans=0;
        for(int i=0;i<n;i++)
            for(int j=i+1;j<n;j++)
                ans+=cnt[1LL*nums2[i]*nums2[j]];
        return ans;
    }
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
        int ans=get(nums1,nums2);
        ans+=get(nums2,nums1);
        return ans;
    }
};
