#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

class Solution {
public:
    int numTriplets(vector<int>& nums1, vector<int>& nums2) {
		int n1 = nums1.size(), n2 = nums2.size();
		map<ll, int> mp1, mp2;
		for(int i=0; i<n1; i++){
			for(int j=i+1; j<n1; j++){
				mp1[(ll)nums1[i] * (ll)nums1[j]]++;
			}
		}
		for(int i=0; i<n2; i++){
			for(int j=i+1; j<n2; j++){
				mp2[(ll)nums2[i] * (ll)nums2[j]]++;
			}
		}

		int ans = 0;
		for(ll a : nums1){
			ans += mp2[a*a];
		}
		for(ll a : nums2){
			ans += mp1[a*a];
		}
		return ans;
    }
};
