class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int maxc = 0;
        int ans = 0;
        maxc = cost[0];
        ans += cost[0];
        for(int i=1;i<n;i++){
            if(s[i] == s[i-1]){
                maxc = max(maxc,cost[i]);
                ans += cost[i];
            }
            else{
                ans -= maxc;
                maxc = cost[i];
                ans += cost[i];
            }
        }
        ans -= maxc;
        return ans;
    }
};
