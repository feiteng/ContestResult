class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n=s.size();
        int ans=0;
        for(int i=0;i<n;i++){
            if(i+1<n&&s[i+1]==s[i]){
                int j=i;

                int mx=cost[i];
                int sum=cost[i];
                while(j+1<n&&s[j+1]==s[i]){
                    mx=max(mx,cost[j+1]);
                    sum+=cost[j+1];
                    j++;
                }
                ans+=sum-mx;
                i=j;
                
            }
        }
        return  ans;
    }
};
