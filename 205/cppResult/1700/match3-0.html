class Solution {
public:
    int minCost(string s, vector<int>& d) {
        int len = s.length();
        int res = 0;
        for(int i=0;i<len;i++)
        {
            int cost = d[i];
            int sum = d[i];
            while(i+1<len&&s[i]==s[i+1])
            {
                i++;
                sum+=d[i];
                cost = max(cost, d[i]);
            }
           // cout<<i<<" "<<sum<<" "<<cost<<endl;
            res += sum-cost;
        }
        return res;
    }
};
