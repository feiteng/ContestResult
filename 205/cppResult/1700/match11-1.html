class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int i = 0;
        int N = s.length();
        int total = 0;
        while(i < N)
        {
            char c = s[i];
            int j = i;
            long long int sum = 0;

            int m = -1;
            while(j < N && s[j] == c)
            {
                sum += cost[j];
                m = max(cost[j], m);
                j++;
            }
            i = j;
            total += (sum - m);
        }
        
        return total;
    }
};
