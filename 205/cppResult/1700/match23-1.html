class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int l = 0;
        int ans = 0;
        int index = 1;
        while(index < n) {
            if(s[index] == s[index - 1]) {
                index++;

            } else {
                int maxi = 0;
                int sum = 0;
                for(int i = l; i < index; i++) {
                    maxi = max(maxi, cost[i]);
                    sum += cost[i];
                }
                ans += (sum - maxi);

                l = index;
                index++;
            }
        }
        int maxi = 0;
        int sum = 0;
        for(int i = l; i < index; i++) {
            maxi = max(maxi, cost[i]);
            sum += cost[i];
        }
        ans += (sum - maxi);
        return ans;
    }
};
