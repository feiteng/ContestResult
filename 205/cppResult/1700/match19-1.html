class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int len = s.size();
        int ans = 0;
        int all = 0;
        for(int i=0;i<len;i++)
            all+=cost[i];
        for(int i=0;i<len;)
        {
          //  cout<<i<<endl;
            int pos = 1;
            while(s[i]==s[i+pos])
                pos++;
            int tmp = 0;
            for(int j=i;j<i+pos;j++)
                tmp=max(tmp,cost[j]);
            ans+=tmp;
            i+=pos;
        }
        return all-ans;
    }
};
