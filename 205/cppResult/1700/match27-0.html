class Solution {
public:
    int dp[100010][30];
    int minCost(string s, vector<int>& cost) {
        for(int i = 0; i < s.length(); i ++) {
            for(int j = 0; j < 30;j ++) {
                dp[i][j] = 1e9+5;
            }
        }
        
        dp[0][0] = cost[0];
        dp[0][s[0]-'a'+1]=0;
        
        for(int i = 1; i < s.length(); i ++) {
            
            dp[i][0] = dp[i-1][0]+cost[i];
            
            for(int j = 1; j <= 26;j++) {
                dp[i][j] = dp[i-1][j]+cost[i];
            }
            
            int p = 1e9+5;
            
            for(int j = 0; j <= 26;j++) {
                if(j == s[i]-'a'+1)continue;
                p=min(p, dp[i-1][j]);
            }
            
            dp[i][s[i]-'a'+1] = min(dp[i][s[i]-'a'+1], p);
        }
        
     
        
        int ans = 1e9+7;
        
        for(int j = 0; j <=26;j++) {
            ans = min(ans, dp[s.length()-1][j]);
        }
        
        return ans;
    }
};
