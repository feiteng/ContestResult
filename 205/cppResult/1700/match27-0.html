class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int ret = 0;
        for(int i = 0; i < n;) {
            int j = 0, mx = 0;
            while(i+j < n && s[i+j] == s[i]) {
                mx = max(mx, cost[i+j]);
                ret += cost[i+j];
                j++;
            }
            ret -= mx;
            i += j;
        }
        return ret;
    }
};
