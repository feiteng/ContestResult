class Solution {
public:
    int f[100005][27];
    
    int minCost(string s, vector<int>& cost) {
        int n=s.length();
        for (int i=0; i<n; i++)
            for (int j=0; j<27; j++)
                f[i][j]=1e9+7;
        f[0][s[0]-'a']=0;
        f[0][26]=cost[0];
        // for (int j=0; j<3; j++){
        //     cout << f[0][j] << " ";
        // }
        // cout << f[0][26] << endl;
        for (int i=1; i<n; i++){
            int cur=s[i]-'a';
            // cout << s[i]-'a' << endl;
            for (int j=0; j<27; j++){
                f[i][j]=min(f[i][j], f[i-1][j]+cost[i]);
            }
            for (int j=0; j<27; j++){
                if (j!=cur){
                    f[i][cur]=min(f[i][cur], f[i-1][j]);
                }
            }
            // for (int j=0; j<3; j++){
            //     cout << f[i][j] << " ";
            // }
            // cout << f[i][26] << endl;
        }
        
        int ans=1e9+7;
        for (int j=0; j<26; j++) ans=min(ans, f[n-1][j]);
        return ans;
    }
};
