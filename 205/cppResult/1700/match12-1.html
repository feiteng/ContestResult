class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int ans = 0;
        int l = 0, r = 0, mn, co;
        int tot = 0;
        for (auto x:cost)tot += x;
        while(l<n)
        {
        	r = l;

        	mn = 0;
        	co = 0;
        	while(r<n && s[r] == s[l])
        	{
        		mn = max(mn, cost[r]);
        		co++;
        		r++;
        	}
        	ans += mn;
        	l = r;
        }
        return tot - ans;
    }
};
