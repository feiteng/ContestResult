class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int i = 0;
        int ans = 0;
        while (i<n) {
            int j=i;
            int sum = 0;
            int maxi = 0;
            while (j < n && s[i] == s[j]) {
                sum += cost[j];
                maxi = max(maxi, cost[j]);
                j++;
            }
            ans += (sum-maxi);
            i = j;
        }
        return ans;
    }
};
