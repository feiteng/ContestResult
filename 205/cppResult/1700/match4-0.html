class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int ans = 0;
        for (int i = 0; i < n;) {
            char c = s[i];
            int mx = INT32_MIN;
            while (i < n && s[i] == c) {
                mx = max(mx, cost[i]);
                ans += cost[i];
                ++i;
            }
            ans -= mx;
        }
        return ans;
    }
};
