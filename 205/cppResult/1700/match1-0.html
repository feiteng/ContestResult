class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size(), res = 0;
        /*
        vector<int> sum(n+1, 0);
        for (int i = 1; i <= n; ++i) sum[i] = sum[i-1]+cost[i-1];
        */
        for (int i = 0, j; i < n; i = j) {
            int maxv = cost[i], sumv = cost[i];
            for (j = i+1; j < n; ++j) {
                if (s[i] != s[j]) break;
                sumv += cost[j];
                maxv = max(maxv, cost[j]);
            }
            res += sumv-maxv;
        }
        return res;
    }
};
