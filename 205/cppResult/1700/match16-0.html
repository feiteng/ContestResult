#include <bits/stdc++.h>

using namespace std;

class Solution {
private:
    int dp[100005][26];
public:
    Solution() : dp() {}

    int minCost(string s, vector<int> &cost) {
        for (int i = 0; i < 26; i++)
            dp[0][i] = cost[0];
        dp[0][s[0] - 'a'] = 0;
        for (int i = 0; i + 1 < s.size(); i++) {
            for (int j = 0; j < 26; j++)
                dp[i + 1][j] = 1E9;
            for (int j = 0; j < 26; j++) {
                if (j != s[i + 1] - 'a')
                    dp[i + 1][s[i + 1] - 'a'] = min(dp[i + 1][s[i + 1] - 'a'], dp[i][j]);
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j] + cost[i + 1]);
            }
        }
        return *min_element(dp[s.size() - 1], dp[s.size() - 1] + 26);
    }
};
