class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int cnt = 0, ans = 0, maxi = 0, total = 0;
        char last = 0;
        for (int i = 0; i < n; i++) {
            if (!cnt || last == s[i]) {
                cnt++;
                last = s[i];
                maxi = max(maxi, cost[i]);
                total += cost[i];
            } else {
                if (cnt >= 2) ans += total - maxi;
                cnt = 1, last = s[i], total = cost[i], maxi = cost[i];
            }
        }
        if (cnt >= 2) ans += total-maxi;
        return ans;
    }
};
