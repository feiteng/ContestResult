class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int ans = 0;
        for(int i=0;i<n;i++)

        {
            int j = i;
            while(j<n && s[j] == s[i])
                j++;
            int sum = 0;
            int mine = INT_MIN;
            for(int k=i;k<j;k++)
            {
                sum += cost[k];
                mine = max(mine,cost[k]);
            }
            ans += sum-mine;
            i = j-1;
        }
        return ans;
    }
};
