class Solution {
public:
    int minCost(string s, vector<int>& cost) 
    {
        char prev = s[0];
        int maxCost = cost[0];
        int ret = 0, sum = cost[0];

        for(int i = 1; i < s.size(); i++)
        {
            if(s[i] == prev)
            {
                maxCost = max(maxCost, cost[i]);
                sum += cost[i];
            }
            else
            {
                ret += sum - maxCost;
                prev = s[i];
                maxCost = cost[i];
                sum = maxCost;
            }
        }
        
        ret += sum - maxCost;
        return ret;
    }
};
