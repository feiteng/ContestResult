class Solution {
 public:
  int minCost(string s, vector<int>& cost) {
    vector<int> dp(26);
    for (int i = 0; i < s.size(); ++i) {
      vector<int> ndp(26);
      for (int j = 0; j < 26; ++j) {
        ndp[j] = dp[j] + cost[i];
      }
      int k = s[i] - 'a';
      for (int j = 0; j < 26; ++j) {
        if (j != k) {
          ndp[k] = min(ndp[k], dp[j]);
        }
      }
      dp = move(ndp);
    }
    return *min_element(dp.begin(), dp.end());
  }
};
