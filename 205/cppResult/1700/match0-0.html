#define maxn 1000005

int p[maxn];

class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        string t=""; t+=s[0]; p[0]=0; int ans=0,ed=0;
        for(int i=1;i<s.length();i++)
        {
            if(s[i]==t[ed])
            {
                if(cost[p[ed]]>cost[i])
                {
                    ans+=cost[i];
                }
                
                else
                {
                    ans+=cost[p[ed]];
                    p[ed]=i;
                }
            }
            
            else t+=s[i],ed++,p[ed]=i;
        }
        return ans;
    }
};
