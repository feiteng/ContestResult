class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        int sum = cost[0];
        int maxx = cost[0];
        int res = 0;
        

        
        for(int i = 1; i < n; i++) {
            if(s[i] == s[i-1]) {
                sum += cost[i];
                maxx = max(cost[i], maxx);
            }
            else {
                res += (sum - maxx);
                sum = cost[i];
                maxx = cost[i];
            }
        }
        
        res += (sum -maxx);
        return res;
    }
};
