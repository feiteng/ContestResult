class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int i, tot = 0, minus = 0;
        int n = s.size(), cur = 0;
        for (i = 0; i < n; i++) {
            tot += cost[i];
            if (i == n - 1 || s[i] != s[i + 1]) {
                cur = max(cur, cost[i]);
                minus += cur;
                cur = 0;
            } else {
                cur = max(cur, cost[i]);
            }
        }
        return tot - minus;
    }
};
