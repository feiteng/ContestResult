class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int ans = 0;
        
        int n = s.size();
        
        int i = 0;
        
        while(i < n) {
            int j = i + 1;
            while(j < n && s[j] == s[i]) {
                j++;
            }
            if(j == i + 1) {
                i++;
                continue;
            }

            int currSum = 0;
            int largest = 0;
            for(int x = i; x < j; x++) {
                currSum += cost[x];
                largest = max(largest, cost[x]);
            }
            
            ans += currSum;
            ans -= largest;
            i = j;
        }
        
        return ans;
    }
};
