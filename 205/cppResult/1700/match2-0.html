class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int ans=0;
        int n=s.length();
        int tmpsum=cost[0];
        int maxval=cost[0];
        for(int i=1;i<n;++i)
        {
            if(s[i]==s[i-1])
                tmpsum+=cost[i],maxval=max(maxval,cost[i]);
            else
                ans+=tmpsum-maxval,tmpsum=cost[i],maxval=cost[i];
        }
        ans+=tmpsum-maxval;
        return ans;
    }
};
