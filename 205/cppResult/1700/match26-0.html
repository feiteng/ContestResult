inline int read() {
  int x = 0, neg = 1; char op = getchar();
  while (!isdigit(op)) { if (op == '-') neg = -1; op = getchar(); }
  while (isdigit(op)) { x = 10 * x + op - '0'; op = getchar(); }
  return neg * x;
}
inline void print(int x) {
  if (x < 0) { putchar('-'); x = -x; }
  if (x >= 10) print(x / 10);
  putchar(x % 10 + '0');
}

class Solution {
public:
    int minCost(string s, vector<int>& cost) {
      int n = SZ(s);
      bool v = false;
      int ans = 0;
      int sum = 0,mx = -1;
      for(int i = 1 ; i < n ; i++){
        if(s[i] == s[i - 1]){
          if(!v) v = true,sum = cost[i-1] + cost[i],mx = max(cost[i-1],cost[i]);
          else sum += cost[i],mx = max(mx,cost[i]);
        }else{
          if(v){
            v = false;
            ans += sum - mx;
            sum = 0,mx = -1;
          }
        }
      }
      if(v) ans += sum - mx;
      return ans;
    }
};
