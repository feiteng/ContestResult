class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int len = s.size();
        int res = 0;
        for(int i=0; i<len;) {
            char cur = s[i];
            if(i<len-1 && s[i+1]==cur) {
                int d = 0;
                int sumcost = 0;
                int maxcost = 0;
                while(i+d<len && s[i+d]==cur) {
                    sumcost += cost[i+d];
                    maxcost = max(maxcost,cost[i+d]);
                    d++;
                }
                res += sumcost - maxcost;
                i += d;
            } else {
                i++;
            }
        }
        return res;
    }
};
