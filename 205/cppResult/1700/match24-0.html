class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int sum=cost[0];
        int maxs=cost[0];
        int ans=0;
        for(int i=1;i<s.length();i++)
        {
            if(s[i]==s[i-1])
            {
                maxs=max(maxs,cost[i]);
                sum+=cost[i];
            }
            else
            {
                ans+=sum-maxs;//cout<<ans<<endl;
                sum=cost[i];
                maxs=cost[i];
            }
        }
        ans+=sum-maxs;
        return ans;
    }
};
