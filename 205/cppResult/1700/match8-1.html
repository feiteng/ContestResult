class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int res =0 ;
        int mn = 0;
        for(int i =0; i < s.size(); i ++){
            if(i==0 or s[i]!=s[i-1]){
                res += mn;
                mn = INT_MIN;

            }
            mn = max (mn, cost[i]);

        }
        res += mn;
        return accumulate(cost.begin(), cost.end(), 0) - res;
    }
};
