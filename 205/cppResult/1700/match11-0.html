class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int ans=0,mn=INT_MAX,tot=0;
        for(int i=0;i+1<s.size();++i){
            if(s[i]==s[i+1]){
                mn=cost[i];
                tot=cost[i];
                while(i+1<=s.size()&&s[i]==s[i+1]){
                    mn=max(mn,cost[i+1]);
                    tot+=cost[i+1];
                    i++;
                }
                ans+=(tot-mn);
                
               // cout<<ans<<" ";
            }
        }
      //  cout<<endl;
        return ans;
    }
};
