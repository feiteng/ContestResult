class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int i=0;
        int j = 0;
        int ans = 0;
        while(i < s.length()) {
            int curCost = cost[i];
            int cur = cost[i];
            j = i+1;
            while(j < s.length() && s[i] == s[j]) {
                curCost = max(curCost, cost[j]);
                cur += cost[j];
                j++;
            }
            i = j;
            ans += (cur - curCost);
        }
        return ans;
    }
};
