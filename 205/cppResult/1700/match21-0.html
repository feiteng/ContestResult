class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int ans= 0;
        for(int i=0;i<n;++i)
        {
            int j=i+1;
            if(j<n && s[i]==s[j])
            {
                int sum = cost[i];
                int max_ = cost[i];
                while(s[i]==s[j] && j<n)
                {
                    sum += cost[j];
                    max_ = max(max_,cost[j]);
                    ++j;
                }
                ans += sum-max_;
                i = j-1;
            }
        }
        return ans;
    }
};
