class Solution {
    int f[100005][26];
public:
    int minCost(string s, vector<int>& cost) {
        memset(f, 60, sizeof(f));
        int n = s.length();
        for (int i = 0; i < 26; i++)
            f[0][i]=0;
        for (int i = 0; i < n; i ++)
        {
            f[i+1][s[i]-'a'] = f[i][s[i]-'a']+cost[i];
            for (int j = 0; j < 26; j ++)
                if (j!=s[i]-'a')
                {
                    f[i+1][j] = f[i][j]+cost[i];
                    if (f[i][j]<f[i+1][s[i]-'a']) f[i+1][s[i]-'a'] = f[i][j];
                }
        }
        int ret = 2e9;
        for (int i = 0; i < 26; i ++)
            ret=min(ret,f[n][i]);
        return ret;
    }
};
