class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int ans = 0;
        for(int i=0;i<n-1;i++){
            if(s[i]==s[i+1]){
                char c= s[i];

                int r =i;
                int maxc=0;
                int tot=0;
                while(r<n && s[r]==c){
                    maxc = max(maxc, cost[r]);
                    tot+=cost[r];
                    r++;
                }
                ans+= tot-maxc;
                i = r-1;
            }
        }
        return ans;
    }
};
