class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int ans = 0;
        for (int i = 0; i < n-1; ++i) {
            if (s[i] == s[i+1]) {
                int start = i, end = i+1;
                while (end < n && s[start]==s[end]) ++end;
                --end;
                int sum = 0, m = 0;
                for (int j = start; j <= end; ++j) {
                    sum += cost[j];
                    m = max(m, cost[j]);
                }
                ans += (sum - m);
                i = end;
            }
        }
        return ans;
    }
};
