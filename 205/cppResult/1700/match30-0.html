const int inf = 2000000000;
const int maxn = 100005;
int dp[maxn][26]; //dp[i][j]:前i个字符串满足条件，并且以j结尾的最少花费

class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.size();
        for (int i = 1; i <= n; i++)
            for (int j = 0; j < 26; j++)
                dp[i][j] = inf;
        for (int j = 0; j < 26; j++)
            dp[0][j] = 0;
        for (int i = 1; i <= n; i++) {
            int letter = s[i - 1] - 'a';
            for (int j = 0; j < 26; j++) {  //以j结尾
                if (j == letter) {
                    for (int k = 0; k < 26; k++) {
                        if (k == j)
                            dp[i][j] = min(dp[i][j], dp[i - 1][k] + cost[i - 1]);
                        else 
                            dp[i][j] = min(dp[i][j], dp[i - 1][k]);
                    }
                } else {
                    dp[i][j] = min(dp[i][j], dp[i - 1][j] + cost[i - 1]); //第i个字母不是j
                }
            }
        }
        int ans = inf;
        for (int j = 0; j < 26; j++)
            ans = min(ans, dp[n][j]);
        //cout <<dp[0][0] << "," << dp[1][0] << "," << dp[2][0];
        return ans;
    }
};
