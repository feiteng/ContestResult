class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        char curr = '?';
        int sum = 0;
        int maxVal = 0;
        int ans = 0;
        for (int i= 0; i < s.size(); i++) {
            if (curr == s[i]) {
                sum += cost[i];
                maxVal = max(maxVal, cost[i]);
            }
            else {
                ans += sum - maxVal;
                maxVal = cost[i];
                sum = cost[i];
                curr = s[i];
            }
        }
        ans += sum - maxVal;
        return ans;
    }
};
