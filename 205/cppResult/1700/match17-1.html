class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int cost1 = 0;
        int lastIndex = 0;
        for(int i=1; i<s.length(); i++) {
            if(s[i] == s[i-1]) {
                if(cost[lastIndex] <= cost[i]) {
                    cost1+=cost[lastIndex];
                    lastIndex = i; 
                } else {
                    cost1+=cost[i];
                }
            } else lastIndex = i;
        }
        return cost1;
    }
};
