#include<bits/stdc++.h>

class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int ct=0;
        priority_queue<int,vector<int>,greater<int>> pq;
        pq.push(cost[0]);
        for(int i=1;i<s.size();i++){
            if(s[i]==s[i-1]){
                pq.push(cost[i]);
            }
            else{
                while(!pq.empty()){
                    int x=pq.top();
                    pq.pop();
                    if(!pq.empty())
                        ct+=x;
                }
                pq.push(cost[i]);
            }
        }
        
        while(!pq.empty()){
                    int x=pq.top();
                    pq.pop();
                    if(!pq.empty())
                        ct+=x;
        }
        return ct;
    }
};
