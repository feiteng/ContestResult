class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n=s.length();
        int f[n][26];
        if (!n) return 0;
        for (int i=0;i<26;i++)
        {
            if (i==s[0]-'a') f[0][i]=0;
            else f[0][i]=cost[0];
        }
        for (int i=1;i<n;i++)
        {
            int mn=1e9+7;
            for (int j=0;j<26;j++)
                f[i][j]=f[i-1][j]+cost[i];
            for (int j=0;j<26;j++)
            {
                if (s[i]-'a'==j) f[i][j]=min(f[i][j],mn);
                mn=min(mn,f[i-1][j]);
            }
            mn=1e9+7;
            for (int j=25;j>=0;j--)
            {
                if (s[i]-'a'==j) f[i][j]=min(f[i][j],mn);
                mn=min(mn,f[i-1][j]);
            }
        }
        int ans=1e9+7;
        for (int i=0;i<26;i++)
            ans=min(ans,f[n-1][i]);
        return ans;
    }
};
