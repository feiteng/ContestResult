class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n=s.size();
        int ans=0;
        int curr=cost[0],maxx=cost[0];
        bool f=false;
        for(int i=1;i<n;i++){
            if(s[i]!=s[i-1]){//cout<<curr<<" "<<maxx<<endl;
                ans+=(curr-maxx);
                maxx=cost[i];
                curr=cost[i];

            }else{
                f=true;
                curr+=cost[i];
                maxx=max(cost[i],maxx);
            }
        }//cout<<endl;
        ans+=(curr-maxx);
        return ans;
    }
};
