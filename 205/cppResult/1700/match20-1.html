class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int ans=0,n=s.size();
        for(int i=0;i<n-1;i++){
            if(s[i]==s[i+1]){
                int j=i;
                while(j<n-1 and s[j]==s[j+1]){
                    j++;
                }
                int ma=0,sum=0;
                for(int k=i;k<=j;k++){
                    sum+=cost[k];
                    ma=max(ma,cost[k]);
                }
                ans+=sum-ma;
                i=j;
            }
        }
        return ans;
    }
};
