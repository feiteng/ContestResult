class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int cs = 0;
        int mx = 0;
        char prv = '-';
        int res = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == prv) {
                cs += cost[i];
                mx = max(mx, cost[i]);
            }
            else {
                res += cs - mx;
                cs = cost[i];
                mx = cost[i];
            }
            prv = s[i];
        }
        res += cs - mx;
        return res;
    }
};
