class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int n = s.length();
        int totalCost = 0;
        int start = 0;
        for (int i=1; i<n; i++) {
            if (s[i]!=s[i-1]) {
                int maxCost = -1;
                int c = 0;
                for (int j=start; j<=i-1; j++) {
                    c += cost[j];
                    maxCost = max(cost[j], maxCost);
                }
                totalCost += c - maxCost;
                start=i;
            }
        }
        
        int maxCost = -1;
        int c = 0;
        for (int j=start; j<=n-1; j++) {
            c += cost[j];
            maxCost = max(cost[j], maxCost);
        }
        totalCost += c - maxCost;
        
        return totalCost;
    }
};
