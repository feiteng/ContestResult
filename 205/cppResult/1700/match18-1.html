class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        int len = s.size();
        int rtn = 0;
        for (int i = 0; i < len; ++i) {

            int mx = cost[i];
            int sum = cost[i];
            while (i < len-1 && s[i+1] == s[i]) {
                ++ i;
                mx = max(mx, cost[i]);
                sum += cost[i];
            }
            rtn += sum - mx;
        }
        return rtn;
    }
};
