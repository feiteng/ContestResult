class Solution {
public:
    int minCost(string s, vector<int>& cost) {
        vector<int> f(26), g(26);
        for (int i = 0; i < s.length(); ++i) {
            for (int j = 0; j < 26; ++j)
                g[j] = f[j] + cost[i];
            for (int j = 0; j < 26; ++j)
                if (j != s[i] - 'a')
                    g[s[i] - 'a'] = min(g[s[i] - 'a'], f[j]);
            f = g;
        }
        int ans = 1e9;
        for (int i = 0; i < 26; ++i) ans = min(ans, f[i]);
        return ans;
    }
};
