class Solution {
public:
    int dp[100005][27];

    int minCost(string s, vector<int>& w) {
        int n=s.size();
        for(int i=0;i<=n;++i) for(int j=0;j<=26;++j) dp[i][j]=1e9;

        dp[1][0]=w[0]; dp[1][s[0]-'a'+1]=0;
        for(int i=1;i<n;++i) {
            int c=s[i]-'a'+1;
            for(int j=0;j<=26;++j) {
                dp[i+1][j] = min(dp[i+1][j], w[i]+dp[i][j]);
                if(j != c) dp[i+1][c] = min(dp[i+1][c], dp[i][j]);

            }
        }
        int ans=1e9;
        for(int i=0;i<=26;++i) ans=min(ans,dp[n][i]);
        return ans;
    }
};
