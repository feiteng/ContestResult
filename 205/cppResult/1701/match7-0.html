class Solution {
public:
    int find(int x, vector<int> &father){
        if (father[x]!=x) {
            int u=find(father[x], father);
            father[x]=u;
        }
        return father[x];
    }
    
    void unions(int x, int y, vector<int> &father){
        int fx = find(x, father);
        int fy = find(y, father);
        father[fx]=fy;
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<int> fa, fb;
        for (int i=0; i<=n; i++){
            fa.push_back(i);
            fb.push_back(i);
        }
        int ans=0;
        for (int i=0; i<edges.size(); i++){
            if (edges[i][0]==3){
                int x=edges[i][1];
                int y=edges[i][2];
                if (find(x, fa)!=find(y, fa)){
                    unions(x, y, fa);
                    unions(x, y, fb);
                }
                else ans++;
            }
        }
        
        for (int i=0; i<edges.size(); i++){
            if (edges[i][0]==1){
                int x=edges[i][1];
                int y=edges[i][2];
                if (find(x, fa)!=find(y, fa)){
                    unions(x, y, fa);
                }
                else ans++;
            }
            else if (edges[i][0]==2){
                int x=edges[i][1];
                int y=edges[i][2];
                if (find(x, fb)!=find(y, fb)){
                    unions(x, y, fb);
                }
                else ans++;
            }
        }
        
        int ra=find(1, fa);
        int rb=find(1, fb);
        
        for (int i=2; i<=n; i++){
            if (find(i, fa)!=ra) return -1;
            if (find(i, fb)!=rb) return -1;
        }
        return ans;
    }
};
