int P[100005], p[100005];
inline int F(int n)
{
    if(P[n] == -1 || P[n] == n) return n;
    return P[n] = F(P[n]);
}
inline int f(int n)
{
    if(p[n] == -1 || p[n] == n) return n;
    return p[n] = f(p[n]);
}
class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int ans = 0;
        memset(P, -1, (n+2) << 2);
        for(auto & v : edges)
        {
            if(v[0] == 3)
            {
                int a = F(v[1]), b = F(v[2]);
                if(a == b) ans++;
                else P[a] = b;
            }
        }
        memcpy(p, P, (n+2) << 2);
        for(auto & v : edges)
        {
            if(v[0] == 1)
            {
                int a = f(v[1]), b = f(v[2]);
                if(a == b) ans++;
                else p[a] = b;
            }
        }
        for(int i = 2 ; i <= n ; i++) if(f(i) != f(1)) return -1;
        memcpy(p, P, (n+2) << 2);
        for(auto & v : edges)
        {
            if(v[0] == 2)
            {
                int a = f(v[1]), b = f(v[2]);
                if(a == b) ans++;
                else p[a] = b;
            }
        }
        for(int i = 2 ; i <= n ; i++) if(f(i) != f(1)) return -1;
        return ans;
    }
};
