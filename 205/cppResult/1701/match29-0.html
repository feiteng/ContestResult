const int M=1e5+5;
int tmp[M],uf[M];

class Solution {
public:
    int find(int x)
    {
        return x==uf[x]?x:uf[x]=find(uf[x]);
    }
    int merge(int x,int y)
    {
        x=find(x); y=find(y);
        if(x==y) return 1;
        uf[x]=y;
        return 0;
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        // sort(edges.begin(),edges.end(),[](const auto &a,const auto &b){
        //    return a[0]>b[0]; 
        // });
        for(int i=0;i<=n;i++) uf[i]=i;
        int m=edges.size();
        int ans=m;
        int i=0;
        for(int i=0;i<m;i++)
        {
            if(edges[i][0]!=3) continue;
            ans-=merge(edges[i][1],edges[i][2]);
        }
        for(int i=1;i<=n;i++)
            tmp[i]=uf[i];
        for(i=0;i<m;i++)
        {
            if(edges[i][0]!=2) continue;
            ans-=merge(edges[i][1],edges[i][2]);
        }
        int cnt=0;
        for(int i=1;i<=n;i++)
            if(uf[i]==i) cnt++;
        if(cnt>=2) return -1;
        for(int i=1;i<=n;i++)
            uf[i]=tmp[i];
        for(int i=0;i<m;i++)
        {
            if(edges[i][0]!=1) continue;
            ans-=merge(edges[i][1],edges[i][2]);
        }
        cnt=0;
        for(int i=1;i<=n;i++)
            if(uf[i]==i) cnt++;
        if(cnt>=2) return -1;
        return m-ans;
    }
};
