int f[111111], g[111111];

int father(int x)
{
    return f[x] == x ? x : f[x] = father(f[x]);
}

class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for (int i = 1; i <= n; ++i) f[i] = i;
        
        int m = 0;
        int res = 0;
        for (auto &e : edges)
        {
            int x = e[1], y = e[2], z = e[0];
            if (z == 3)
            {
                int fx = father(x), fy = father(y);
                if (fx != fy)
                {
                    f[fx] = fy;
                    m++;
                }
                else
                {
                    res ++;
                }
            }
        }
        
        //cout << m << endl;
        
        for (int i = 1; i <= n; ++i) g[i] = f[i];
        
        {
            for (int i = 1; i <= n; ++i) f[i] = g[i];
            int cnt = m;
            for (auto &e : edges)
            {
            int x = e[1], y = e[2], z = e[0];
                if (z == 1)
                {
                    int fx = father(x), fy = father(y);
                    if (fx != fy)
                    {
                        f[fx] = fy;
                        cnt++;
                    }
                    else
                        res += 1;
                }
            }
            //cout << cnt << endl;
            if (cnt != n-1) return -1;
        }
        {
            for (int i = 1; i <= n; ++i) f[i] = g[i];
            int cnt = m;
            for (auto &e : edges)
            {
            int x = e[1], y = e[2], z = e[0];
                if (z == 2)
                {
                    int fx = father(x), fy = father(y);
                    if (fx != fy)
                    {
                        f[fx] = fy;
                        cnt++;
                    }
                    else
                        res += 1;
                }
            }
            //cout << cnt << endl;
            if (cnt != n-1) return -1;
        }
        return res;
    }
};
