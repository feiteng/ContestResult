class Solution {
    vector<int> p;
    int f(int x) { return x == p[x] ? x : p[x] = f(p[x]); }
    int comp(vector<vector<int>>& es, int n, int t) {
        p.resize(n);
        iota(p.begin(), p.end(), 0);
        int ans = n;
        for (auto &e : es) {
            if (e[0] == t || e[0] == 3) {
                int fa = f(e[1]);
                int fb = f(e[2]);
                if (fa != fb) {
                    p[fa] = fb;
                    ans--;
                }
            }
        }
        return ans;
    }
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& es) {
        for (auto &e : es) --e[1], --e[2];
        if (comp(es, n, 1) != 1) return -1;
        if (comp(es, n, 2) != 1) return -1;
        int c = comp(es, n, 3);
        int ne = n - c + (c - 1) * 2;
        return int(es.size()) - ne;
    }
};
