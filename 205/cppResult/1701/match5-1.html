struct DSU {
    vector<int> fa;
    vector<int> sz;
    int merged;
    DSU(int n): fa(n, 0), sz(n, 1), merged(0) {
        for (int i = 0;  i < n; i++)
            fa[i] = i;
    }
    int find(int x) {
        if (x == fa[x]) return x;
        return fa[x] = find(fa[x]);
    }
    bool uni(int x, int y) {
        int fx = find(x), fy = find(y);
        if (fx == fy) return false;
        merged++;
        if (sz[fx] > sz[fy]) swap(fx, fy);
        fa[fx] = fy;
        sz[fy] += sz[fx];
        return true;
    }
};

class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int num = 0;
        DSU u1(n+1), u2(n+1);
        for (auto& e: edges) {
            if (e[0] == 3) {
                if (u1.uni(e[1], e[2])) {
                    u2.uni(e[1], e[2]);
                    num++;
                }
            }
        }
        for (auto& e: edges) {
            if (e[0] == 1) {
                if (u1.uni(e[1], e[2])) {
                    num++;
                }
            } else if (e[0] == 2) {
                if (u2.uni(e[1], e[2])) {
                    num++;
                }
            }
        }
        if (u1.merged < n-1 || u2.merged < n-1) {
            return -1;
        }
        return edges.size() - num;
    }
};
