class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int result = 0;
        vector<int> edges1, edges2, edges3;
        for (int i=0; i<edges.size(); ++i){
            if (edges[i][0]==1)
                edges1.push_back(i);
            else if (edges[i][0]==2)
                edges2.push_back(i);
            else
                edges3.push_back(i);
        }
        vector<int> parents(n+1, -1);
        result += helper(n, edges, edges3, parents);
        vector<int> copy = parents;
        result += helper(n, edges, edges1, parents);
        result += helper(n, edges, edges2, copy);
        for (int i=1; i<parents.size() && result!=-1; ++i){
            if (parents[i]==-1)
                result = -1;
        }
        for (int i=1; i<copy.size() && result!=-1; ++i){
            if (copy[i]==-1)
                result = -1;
        }
        return result;
    }
    
    int helper(int n, vector<vector<int>>& edges, vector<int>& edge_inds, vector<int>& parents){
        int result = 0;
        for (int i=0; i<edge_inds.size(); ++i){
            vector<int> e = edges[edge_inds[i]];
            if (parents[e[1]]==-1 && parents[e[2]]==-1)
                parents[e[1]] = parents[e[2]] = e[1];
            else if (parents[e[1]]==-1)
                parents[e[1]] = parents[e[2]];
            else if (parents[e[2]]==-1)
                parents[e[2]] = parents[e[1]];
            else{
                int p1 = parents[e[1]], p2 = parents[e[2]];
                while (p1!=parents[p1])
                    p1 = parents[p1];
                while (p2!=parents[p2])
                    p2 = parents[p2];
                if (p1==p2)
                    result++;
                else{
                    parents[p2] = p1;
                    parents[e[2]] = p1;
                    parents[e[1]] = p1;
                }
            }
        }
        return result;
    }
};
