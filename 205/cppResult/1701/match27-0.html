class UF
{
public:
    vector<int> v;
    vector<int> size;
    int cnt;
    
    UF(int n)
    {
        v = vector<int>(n);
        size = vector<int>(n, 1);
        for(int i = 1; i < n; i++)
            v[i] = i;
        cnt = n - 1;
    }
    
    int find(int i)
    {
        while(v[i] != i)
            i = v[i];
        return i;
    }
    
    void unite(int i, int j)
    {
        i = find(i);
        j = find(j);
        if(i == j)
            return;
        if(size[i] > size[j])
            swap(i, j);
        v[i] = j;
        size[j] += size[i];
        cnt--;
    }
    
    bool same(int i, int j)
    {
        i = find(i);
        j = find(j);
        return i == j;
    }
};


class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges)
    {
        UF uf(n + 1);
        int add = 0;
        for(int i = 0; i < edges.size(); i++)
        {
            if(edges[i][0] == 3)
            {
                if(!uf.same(edges[i][1], edges[i][2]))
                {
                    add++;
                    uf.unite(edges[i][1], edges[i][2]);
                }
            }
        }
        
        UF ufa = uf;
        UF ufb = uf;
        
        for(int i = 0; i < edges.size(); i++)
        {
            if(edges[i][0] == 1)
            {
                if(!ufa.same(edges[i][1], edges[i][2]))
                {
                    add++;
                    ufa.unite(edges[i][1], edges[i][2]);
                }
            }
        }
        
        for(int i = 0; i < edges.size(); i++)
        {
            if(edges[i][0] == 2)
            {
                if(!ufb.same(edges[i][1], edges[i][2]))
                {
                    add++;
                    ufb.unite(edges[i][1], edges[i][2]);
                }
            }
        }
        
        if(ufa.cnt != 1 || ufb.cnt != 1)
            return -1;
        
        int ret = edges.size();
        ret -= add;
        
        return ret;
    }
};
