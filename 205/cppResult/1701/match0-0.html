#define maxn 1000005
#define rep(i,a,b) for(int i=a;i<=b;i++)

int f[maxn],f1[maxn],f2[maxn];

inline int F(int x)
{
    if(f[x]==x) return x;
    return f[x]=F(f[x]);
}

inline int F1(int x)
{
    if(f1[x]==x) return x;
    return f1[x]=F1(f1[x]);
}

inline int F2(int x)
{
    if(f2[x]==x) return x;
    return f2[x]=F2(f2[x]);
}

class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        rep(i,1,n) f[i]=i;
        int ans=0;
        for(int i=0;i<edges.size();i++)
        {
            if(edges[i][0]==3)
            {
                int u=edges[i][1],v=edges[i][2],tx=F(u),ty=F(v);
                if(tx==ty) ans++;
                else f[tx]=ty;
            }
        }
        
        rep(i,1,n) f1[i]=f[i],f2[i]=f[i];
        for(int i=0;i<edges.size();i++)
        {
            if(edges[i][0]==1)
            {
                int u=edges[i][1],v=edges[i][2],tx=F1(u),ty=F1(v);
                if(tx==ty) ans++;
                else f1[tx]=ty;
            }
        }
        
        for(int i=0;i<edges.size();i++)
        {
            if(edges[i][0]==2)
            {
                int u=edges[i][1],v=edges[i][2],tx=F2(u),ty=F2(v);
                if(tx==ty) ans++;
                else f2[tx]=ty;
            }
        }
        int cnt=0;
        rep(i,1,n) if(i==f1[i]) cnt++; if(cnt!=1) return -1;
        cnt=0;
        rep(i,1,n) if(i==f2[i]) cnt++; if(cnt!=1) return -1;
        return ans;
    }
};
