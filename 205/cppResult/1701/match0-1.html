class DS {
public:
    vector<int> parent;
    vector<int> height;
    int n;
    DS(int n) : parent(n), height(n, 0) {
        this->n = n;
        for (int i = 0; i < n; i++) {
            parent[i] = i;
        }
    }
    
    DS(DS& ds) : parent(ds.parent), height(ds.height), n(ds.n) {}
    
    int find(int i) {
        if (parent[i] != i) {
            parent[i] = find(parent[i]);
        }
        return parent[i];
    }
    
    void combine(int i, int j) {
        i = find(i);
        j = find(j);
        if (height[i] < height[j]) {
            parent[i] = j;
        } else {
            parent[j] = i;
            if (height[i] == height[j]) height[i]++;
        }
    }
    
    bool check() {
        int p = find(parent[1]);
        for (int i = 2; i < n; i++) {
            if (find(i) != p) return false;
        }
        return true;
    }

};

class Solution {
public:
    int buildG(vector<pair<int, int>>& edges, DS& ds) {
        int count = 0;
        for (auto &p : edges) {
            if (ds.find(p.first) != ds.find(p.second)) {
                count++;
            }
            ds.combine(p.first, p.second);
        }
        return count;
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<pair<int, int>> alice, bob, common;
        for (auto &e : edges) {
            if (e[0] == 1) {
                alice.push_back({e[1], e[2]});
            } else if (e[0] == 2) {
                bob.push_back({e[1], e[2]});
            } else {
                common.push_back({e[1], e[2]});
            }
        }
        vector<int> visited(n + 1, false);
        DS mst(n + 1);
        
        int ret = edges.size();
        ret -= buildG(common, mst);
        DS copy(mst);
        ret -= buildG(alice, copy);
        ret -= buildG(bob, mst);
        if (copy.check() && mst.check())    return ret;
        return -1;
        
    }
};
