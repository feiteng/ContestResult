class Solution {
public:
    struct edge{
        int u,v;
    }E1[100010],E2[100010],E3[100010];
    int c1,c2,c3,f[100010],f1[100010],f2[100010];
    int _find(int x,int *f){return x==f[x]?x:f[x]=_find(f[x],f);}
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int m = edges.size();
        for(int i=1;i<=n;i++)f[i] = i;
        for(int i=0;i<m;i++){
            if(edges[i][0]==1)E1[c1++] = {edges[i][1],edges[i][2]};
            if(edges[i][0]==2)E2[c2++] = {edges[i][1],edges[i][2]};
            if(edges[i][0]==3)E3[c3++] = {edges[i][1],edges[i][2]};
        }
        int ans = 0;
        for(int i=0;i<c3;i++){
            int fa = _find(E3[i].u,f),fb = _find(E3[i].v,f);
            if(fa!=fb){
                ans++;
                f[fa] = fb;
            }
        }
        if(ans==n-1)return m-ans;
        for(int i=1;i<=n;i++)f1[i] = f[i],f2[i] = f[i];
        for(int i=0;i<c1;i++){
            int fa = _find(E2[i].u,f1),fb = _find(E2[i].v,f1);
            if(fa!=fb){
                ans++;
                f1[fa] = fb;
            }
        }
        int t = 0;
        for(int i=1;i<=n;i++)if(f1[i]==i)t++;
        if(t>1)return -1;
        for(int i=0;i<c2;i++){
            int fa = _find(E2[i].u,f2),fb = _find(E2[i].v,f2);
            if(fa!=fb){
                ans++;
                f2[fa] = fb;
            }
        }
        t = 0;
        for(int i=1;i<=n;i++)if(f2[i]==i)t++;
        if(t>1)return -1;
        return m-ans;
    }
};
