class DSU{
    int n;
    vector<int> par, sz;
    int comp;
public:
    DSU(int n1){
        n = n1;
        sz.assign(n, 1);
        par.assign(n, 0);
        for(int i=0; i<n; ++i){
            par[i] = i;
        }
        comp = n;
    }
    
    int root(int u){
        if( par[u] == u ){ return u; }
        return par[u] = root(par[u]);
    }
    
    bool unite(int a, int b){
        int x = root(a);
        int y = root(b);
        if( x == y ){ return false; }
        if( sz[y] > sz[x] ){ swap(x, y); }
        par[y] = x;
        sz[x] += sz[y];
        comp--;
        return true;
    }
    int getComp(){
        return comp;
    }
};
class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        
        DSU ds(n), ds2(n);
        int ans = 0;
        int a1 = 0, a2 = 0;
        for(auto i : edges){
            if( i[0] == 3 ){
                int u = i[1]-1;
                int v = i[2]-1;
                if( ds.unite(u, v) == false ){
                    ++ans;
                }
                if( ds2.unite(u, v) == false){
                    // 
                }
            }
        }
        for(auto i : edges){
            if( i[0] == 1 ){
                int u = i[1] - 1;
                int v = i[2] - 1;
                if( ds.unite(u, v) == false ){
                    ++a1;
                }
            }
        }
        if( ds.getComp() != 1 ){ return -1; }
        for(auto i : edges){
            if( i[0] == 2 ){
                int u = i[1] - 1;
                int v = i[2] - 1;
                if( ds2.unite(u, v) == false ){
                    ++a2;
                }
            }
        }
        if( ds2.getComp() != 1 ){ return -1; }
        return ans + a1 + a2;
    }
};
