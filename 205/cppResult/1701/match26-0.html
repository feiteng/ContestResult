int find_set(int parent[],int v)
{
    if(v == parent[v])
        return v;
    return parent[v] = find_set(parent,parent[v]);
}
void union_set(int parent[],int a,int b)
{
    a = find_set(parent,a);
    b = find_set(parent,b);
    if(a != b)
        parent[b] = a;
}
class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int parent1[n+1];
        int parent2[n+1];
        for(int i=1;i<=n;i++)
            parent1[i] = parent2[i] = i;
        int ans = 0;
        for(int i=0;i<edges.size();i++)
        {
            if(edges[i][0] == 3)
            {
                int t1 = find_set(parent1,edges[i][1]);
                int t2 = find_set(parent1,edges[i][2]);
                if(t1 != t2)
                {
                    union_set(parent1,edges[i][2],edges[i][1]);
                    union_set(parent2,edges[i][2],edges[i][1]);
                }
                else
                    ans++;
            }
        }
        for(int i=0;i<edges.size();i++)
        {
            if(edges[i][0] == 1)
            {
                int t1 = find_set(parent1,edges[i][1]);
                int t2 = find_set(parent1,edges[i][2]);
                if(t1 != t2)
                {
                    union_set(parent1,edges[i][2],edges[i][1]);
                }
                else
                    ans++;
            }
            else if(edges[i][0] == 2)
            {
                int t1 = find_set(parent2,edges[i][1]);
                int t2 = find_set(parent2,edges[i][2]);
                if(t1 != t2)
                {
                    union_set(parent2,edges[i][2],edges[i][1]);
                }
                else
                    ans++;
            }
        }
        int par1 = find_set(parent1,1);
        for(int i=1;i<=n;i++)
        {
            int t = find_set(parent1,i);
            //cout << t << "\n";
            if(t != par1)
                return -1;
        }
        //cout << "--------\n";
        int par2 = find_set(parent2,1);
        for(int i=1;i<=n;i++)
        {
            int t = find_set(parent2,i);
            //cout << t << "\n";
            if(t != par2)
                return -1;
        }
        //cout << "--------\n";
        return ans;
    }
};
