class Solution {
public:
    int fa[100005];
    int get(int x){
        if(x==fa[x]) return x;
        return fa[x]=get(fa[x]);
    }
    void merge(int x,int y){
        fa[get(x)]=get(y);
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int ans=0;
        bool flag=true;

        for(int i=1;i<=n;i++) fa[i]=i;
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==3){
                if(get(edges[i][1])==get(edges[i][2]))ans++;
                else merge(edges[i][1],edges[i][2]);
            }
        }
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==1){
                if(get(edges[i][1])==get(edges[i][2])) ans++;
                else merge(edges[i][1],edges[i][2]);
            }
        }
        for(int i=1;i<=n;i++) if(get(1)!=get(i)) flag=false;
        for(int i=1;i<=n;i++) fa[i]=i;
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==3){
                merge(edges[i][1],edges[i][2]);
            }

        }
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==2){
                if(get(edges[i][1])==get(edges[i][2])) ans++;
                else merge(edges[i][1],edges[i][2]);
            }
        }
        for(int i=1;i<=n;i++) if(get(1)!=get(i)) flag=false;
        if(flag==false) return -1;
        return ans;
    }
};
