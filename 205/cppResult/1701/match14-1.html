#include <bits/stdc++.h>
using namespace std;


typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;


class Solution {
public:
	int N;
	vector<int> ps;
	vector<vector<int>> es1, es2, es3;

	int find(int u, vector<int>& ps){
		if(ps[u] == u) return u;
		int p = find(ps[u], ps);
		ps[u] = p;
		return p;
	}

    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
		N = n;
		es1.clear(); es2.clear(); es3.clear();
		for(vector<int> &e : edges){
			int t = e[0], u = e[1], v = e[2];
			if(t == 1) es1.push_back(e);
			if(t == 2) es2.push_back(e);
			if(t == 3) es3.push_back(e);
		}
		ps = vector<int>(N+1, 0);
		for(int u=1; u<=N; u++){
			ps[u] = u;
		}

		int ans1 = 0, ans2 = 0, ans3 = 0;
		for(vector<int> e : es3){
			int u = e[1], v = e[2];
			int pu = find(u, ps), pv = find(v, ps);
			if(pu == pv){
				ans3++; continue;
			}
			ps[pu] = pv;
		}

		vector<int> ps0 = ps;
		for(vector<int> e : es1){
			int u = e[1], v = e[2];
			int pu = find(u, ps), pv = find(v, ps);
			if(pu == pv){
				ans1++; continue;
			}
			ps[pu] = pv;
		}
		int en = (int)(es1.size() + es3.size()) - ans3 - ans1;
		if(en < n - 1) return -1;

		ps = ps0;
		for(vector<int> e : es2){
			int u = e[1], v = e[2];
			int pu = find(u, ps), pv = find(v, ps);
			if(pu == pv){
				ans2++; continue;
			}
			ps[pu] = pv;
		}
		en = (int)(es2.size() + es3.size()) - ans3 - ans2;
		if(en < n - 1) return -1;
		return ans3 + ans2 + ans1;
    }
};
