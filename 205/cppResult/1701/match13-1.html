class Solution {
    int findRoot(vector<int>& rt, int u) {
        return u == rt[u] ? rt[u] : rt[u] = findRoot(rt, rt[u]);
    }
    
    int getUseEdge(vector<int>& rt, vector<vector<int>>& edges, int type) {
        int commonUseEdge = 0;
        for (int i = 0; i < edges.size(); ++i) {
            int tp = edges[i][0], u = edges[i][1], v = edges[i][2];

            if (tp == type) {
                int rtU = findRoot(rt, u);
                int rtV = findRoot(rt, v);
                
                if (rtU != rtV) {
                    rt[rtU] = rtV;
                    ++commonUseEdge;
                }
            }
        }
        return commonUseEdge;
    }
    
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<int> rt(n + 1, 0);
        for (int i = 0; i <= n; ++i)
            rt[i] = i;
        
        int commonUseEdge = 0;
        for (int i = 0; i < edges.size(); ++i) {
            int tp = edges[i][0], u = edges[i][1], v = edges[i][2];
            if (tp == 3) {
                int rtU = findRoot(rt, u);
                int rtV = findRoot(rt, v);
                
                if (rtU != rtV) {
                    rt[rtU] = rtV;
                    ++commonUseEdge;
                }
            }
        }
        
        int ans = commonUseEdge;
        
        vector<int> rt1(rt.begin(), rt.end());
        ans += getUseEdge(rt1, edges, 1);
        
        rt1[1]= findRoot(rt1, 1);
        for (int i = 1; i <= n; ++i) {
            if (findRoot(rt1, i) != rt1[1]) {
                return -1;
            }
        }
        
        vector<int> rt2(rt.begin(), rt.end());
        ans += getUseEdge(rt2, edges, 2);
        
        rt2[1]= findRoot(rt2, 1);
        for (int i = 1; i <= n; ++i) {
            if (findRoot(rt2, i) != rt2[1]) {
                return -1;
            }
        }
        
        return edges.size() - ans;
    }
};
