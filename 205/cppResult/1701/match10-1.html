inline int read() {
  int x = 0, neg = 1; char op = getchar();
  while (!isdigit(op)) { if (op == '-') neg = -1; op = getchar(); }
  while (isdigit(op)) { x = 10 * x + op - '0'; op = getchar(); }
  return neg * x;
}

  if (x < 0) { putchar('-'); x = -x; }
  if (x >= 10) print(x / 10);
  putchar(x % 10 + '0');
}

const int maxn = 1e5 + 10;

int par[maxn],memo[maxn];
int find(int x){
  return par[x] == x ? x : par[x] = find(par[x]);
}
class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {

      rep(i,1,n) par[i] = i;
      int ans = 0;
      for(auto it : edges){
        int tp = it[0],u = it[1],v = it[2];
        if(tp == 3){
          int x = find(u),y = find(v);
          if(x == y) continue;
          else{
            ans++;
            par[x] = y;
          }
        }
      }
      bool ok = true;

      rep(i,1,n) memo[i] = par[i];
      for(auto it : edges){
        int tp = it[0],u = it[1],v = it[2];
        if(tp == 1){
          int x = find(u),y = find(v);
          if(x == y) continue;
          else{
            ans++;
            par[x] = y;
          }
        }
      }
      set<int> st;
      rep(i,1,n) st.insert(find(i));
      if(st.size() != 1) ok = false;
      if(!ok) return -1;
      st.clear();

      rep(i,1,n) par[i] = memo[i];
      for(auto it : edges){
        int tp = it[0],u = it[1],v = it[2];
        if(tp == 2){
          int x = find(u),y = find(v);
          if(x == y) continue;
          else{
            ans++;
            par[x] = y;
          }
        }
      }
      rep(i,1,n) st.insert(find(i));
      if(st.size() != 1) ok = false;
      if(!ok) return -1;
      return SZ(edges) - ans;
    }
};
