class Solution {
public:
    vector<int> f;
    int find(int x){
        if(f[x]==-1)
            return x;
        return f[x]=find(f[x]);
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        f=vector<int>(n+1,-1);
        int res=0;
        int p=n;
        for(auto& q:edges)
            if(q[0]==3){
                int fx=find(q[1]);
                int fy=find(q[2]);
                if(fx!=fy){
                    f[fx]=fy;
                    ++res;
                    --p;
                }
            }
        vector<int> of=f;
        int op=p;
        for(auto& q:edges)
            if(q[0]==1){
                int fx=find(q[1]);
                int fy=find(q[2]);
                if(fx!=fy){
                    f[fx]=fy;
                    ++res;
                    --p;
                }
            }
        if(p>1)
            return -1;
        f=of;
        p=op;
        for(auto& q:edges)
            if(q[0]==2){
                int fx=find(q[1]);
                int fy=find(q[2]);
                if(fx!=fy){
                    f[fx]=fy;
                    ++res;
                    --p;
                }
            }
        if(p>1)
            return -1;
        return edges.size()-res;
    }
};
