#include <cstring>
class Solution {
public:
    const int N=100010;
    struct UFS{
        int f[100010];
        int gf(int x) { return f[x]?(f[x]=gf(f[x])):x; }
    }a,b;
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int cnt=0, cnt1=0, cnt2=0;
        memset(a.f, 0, sizeof(a.f)); memset(b.f, 0, sizeof(b.f));
        for (int i=0;i!=edges.size();++i) if (edges[i][0]==3) {
            int u=edges[i][1], v=edges[i][2];
            if (a.gf(u)!=a.gf(v)) {
                ++cnt;
                a.f[a.gf(u)]=a.gf(v);
                b.f[b.gf(u)]=b.gf(v);
            }
        }
        for (int i=0;i!=edges.size();++i) if (edges[i][0]==1) {
            int u=edges[i][1], v=edges[i][2];
            if (a.gf(u)!=a.gf(v)) {
                ++cnt1;
                a.f[a.gf(u)]=a.gf(v);
            }
        }
        for (int i=0;i!=edges.size();++i) if (edges[i][0]==2) {
            int u=edges[i][1], v=edges[i][2];
            if (b.gf(u)!=b.gf(v)) {
                ++cnt2;
                b.f[b.gf(u)]=b.gf(v);
            }
        }
        if (cnt1+cnt!=n-1 || cnt2+cnt!=n-1) return -1;
        return edges.size()-cnt-cnt1-cnt2;
    }
};
