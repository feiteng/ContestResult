class Solution
{
public:
    static const int N=1e5+5;
    int fat1[N],fat2[N];
    int find1(int x)
    {
        return fat1[x]==x?x:fat1[x]=find1(fat1[x]);
    }
    int find2(int x)
    {
        return fat2[x]==x?x:fat2[x]=find2(fat2[x]);
    }
    struct node
    {
        int type,u,v;
        bool operator<(const node &b)const
        {
            if(type==b.type)
                return u<b.u;
            int flag=0;
            if(type==3)
                flag=1;
            return flag;
        }
    };
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges)
    {
        int len=edges.size();
        vector<node>vec;
        for(auto it:edges)
            vec.push_back({it[0],it[1],it[2]});
        sort(vec.begin(),vec.end());
        int sum=0,sum1=0,sum2=0;

        for(int i=1;i<=n;i++)
            fat1[i]=fat2[i]=i;
        for(auto it:vec)
        {
           // cout<<it.type<<" + "<<it.u<<" "<<it.v<<endl;
            if(it.type==3)
            {
                int tx=find1(it.u);
                int ty=find1(it.v);
                if(tx!=ty)
                {
                    fat1[tx]=ty;
                    fat2[tx]=ty;
                    sum++;
                }
            }

            if(it.type==1)
            {
                int tx=find1(it.u);
                int ty=find1(it.v);
                if(tx!=ty)
                {
                    fat1[tx]=ty;
                    sum1++;
                }
            }

            if(it.type==2)
            {
                int tx=find2(it.u);
                int ty=find2(it.v);
                if(tx!=ty)
                {
                    fat2[tx]=ty;
                    sum2++;
                }
            }
        }
          //  cout<<sum<<" "<<sum1<<" "<<sum2<<endl;
        if(sum+sum1==n-1&&sum+sum2==n-1)
            return (int)edges.size()-sum-sum1-sum2;
        else
            return -1;

    }
};
