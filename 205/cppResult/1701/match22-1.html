class Solution {
public:
    vector<int> head;
    int find_head(int x){
        if(x==head[x]) return x;
        head[x] = find_head(head[x]);
        return head[x];
    }
    void unionn(int x, int y){
        //printf("union %d %d\n", x, y);
        int head_x = find_head(x);
        int head_y = find_head(y);
        head[head_y]=head_x;
    }
    bool connected(int x, int y){
        int head_x = find_head(x);
        int head_y = find_head(y);
        return head_x==head_y;
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int num_edges = edges.size();
        for(int i=0;i<=n;i++) head.push_back(i);
        int used_edge = 0;

        int part = n;
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==3){
                int x = edges[i][1];
                int y = edges[i][2];
                if(connected(x, y)) continue;
                unionn(x, y);
                used_edge++;
                part--;
            }

        }
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==1){
                int x = edges[i][1];
                int y = edges[i][2];
                if(connected(x, y)) continue;
                unionn(x, y);
                used_edge++;
                part--;
            }
        }
        if(part!=1) return -1;
        part = n;
        head.clear();

        for(int i=0;i<=n;i++) head.push_back(i);
        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==3){
                int x = edges[i][1];
                int y = edges[i][2];
                if(connected(x, y)) continue;
                unionn(x, y);
                part--;
            }
        }

        for(int i=0;i<edges.size();i++){
            if(edges[i][0]==2){
                int x = edges[i][1];
                int y = edges[i][2];
                if(connected(x, y)) continue;
                unionn(x, y);
                used_edge++;
                part--;
            }
        }
        if(part!=1) return -1;
        return num_edges-used_edge;
    }
};
