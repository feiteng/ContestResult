class Solution {
public:
    
    
int find(vector<int>&pre, int k)  
{
	if(pre[k]==k)	return k; 
	return pre[k]=find(pre, pre[k]); 
}

    int maxNumEdgesToRemove(int n, vector<vector<int>>& E) {
       // sort(E.begin(), E.end());
       // reverse(E.begin(), E.end());
        
        vector<int> a(n + 1, 0), b(n+1, 0);

        for (int i = 1; i <= n; i++) a[i] = b[i] = i;
        
        int ans =0 ;
        
        int C = 2 * n;
        for (int i = 0; i < E.size(); i++) {
         //   if (C == 2) {ans ++; continue;}
            int t = E[i][0];
            int x = E[i][1], y = E[i][2];
            //cout << t << "! " << x << " " << y << endl;

            if (t == 3) {
                int u = find(a,x);
                int v = find(a,y);
             //   cout << u << " " << v << endl;
                int uu = find(b,x);
                int vv = find(b,y);
                if (u != v) {
                    a[u] = v;
                    b[uu] = vv;
                    C -= 2;
                    
            //        cout << t << " " << x << " " << y << endl;
                }
                else ans ++;
            }
            

        }
        
          for (int i = 0; i < E.size(); i++) {
         //   if (C == 2) {ans ++; continue;}
            int t = E[i][0];
            int x = E[i][1], y = E[i][2];
          
            
            if (t == 1) {

                int u = find(a,x);
                int v = find(a,y);

                if (u != v) {
                    a[u] = v;
                    C --;
             //       cout << t << " " << x << " " << y << endl;
                }
                else ans ++;
            }
            
           if (t == 2) {
                int u = find(b,x);
                int v = find(b,y);

                if (u != v) {
                    b[u] = v;
                    C--;
              //      cout << t << " " << x << " " << y << endl;
                }
                else ans ++;
            }
        }
        
     //   cout << C << endl;
        
        if ( C > 2) return -1;
        else return ans;
        
    }
};
