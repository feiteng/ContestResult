class Solution {
public:
    int fa[100010];
    int fb[100010];
    int find1(int x)
    {
        return x==fa[x]?x:fa[x]=find1(fa[x]);
    }
    int find2(int x)
    {
        return x==fb[x]?x:fb[x]=find2(fb[x]);
    }
    struct edge
    {
        int t,u,v;
        bool operator<(const edge a)const
        {
            return t>a.t;
        }
    }edg[1000100];
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for(int i=0;i<=n;++i)
            fa[i] = fb[i]=i;
        for(int i=0;i<edges.size();++i)
        {
            edg[i].t = edges[i][0];
            edg[i].u = edges[i][1];
            edg[i].v = edges[i][2];
        }
        int cnt = edges.size();
        sort(edg,edg+cnt);
        for(int i=0;i<cnt;++i)
        {
            if(edg[i].t!=2)
            {
                int fu = find1(edg[i].u),fv=find1(edg[i].v);
                if(fu!=fv)
                {
                    fa[fu] = fv;
                    if(edg[i].t == 3)
                        fb[find2(fu)] = find2(fv);
                }
            }
        }
        int ans = n-1;
        for(int i=0;i<cnt;++i)
        {
            if(edg[i].t!=1)
            {
                int fu = find2(edg[i].u),fv=find2(edg[i].v);
                if(fu!=fv)
                {
                    fb[fu] = fv;
                    ans++;
                }
            }
        }
        int ffa = find1(1);
        int ffb = find2(1);
        for(int i=2;i<=n;++i)
        {
            if(find1(i)!=ffa || find2(i)!=ffb)
                return -1;
        }
        return cnt-ans;
    }
};
