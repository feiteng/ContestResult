const int A=1, B=2, AB=3;

bool can_a(int tp)
{
    return tp==A || tp==AB;
}

bool can_b(int tp)
{
    return tp==B || tp==AB;
}

class Solution {
    vector<vector<pii>> E;
    int n;
    
    void dfs(int curr, int type, vector<bool>& visited)
    {
        if(visited[curr]) return;
        visited[curr]=true;
        
        for(auto [next, tp]: E[curr])
        {
            if(type==1)
            {
                if(can_a(tp)) dfs(next, type, visited);
            }
            else
            {
                if(can_b(tp)) dfs(next, type, visited);
            }
        }
    }
    
    void dfsAB(int curr, vector<bool>& visited, int& res)
    {
        //if(visited[curr]) return;
        
        for(auto [next, tp]: E[curr])
        {
            if(tp!=AB) continue;
            if(visited[next]) continue;
            
            visited[next]=true;
            res++;
            
            dfsAB(next, visited, res);
        }
    }
    
public:
    int maxNumEdgesToRemove(int n_, vector<vector<int>>& edges) {
        n=n_;
        E=vector<vector<pii>>(n+1);
        
        for(auto v: edges)
        {
            assert(v.size()==3);
            int tp=v[0], a=v[1], b=v[2];
            
            E[a].pb({b, tp});
            E[b].pb({a, tp});
        }
        
        vector<bool> visited(n+1, false);
        dfs(1, 1, visited);
        
        for(int i=1; i<=n; i++)
        {
            if(!visited[i]) return -1;
            visited[i]=false;
        }
        
        dfs(1, 2, visited);
        for(int i=1; i<=n; i++)
        {
            if(!visited[i]) return -1;
            visited[i]=false;
        }
        
        int m=edges.size();
        int res=0;
        
        int comp=0;
        for(int i=1; i<=n; i++)
        {
            if(!visited[i])
            {
                visited[i]=true;
                dfsAB(i, visited, res);
                comp++;
            }
        }
        
        
        dbg(m, res, comp);
        
        res+=2*(comp-1);
        
        dbg(res);
        
        //vector<bool> visitedA(n+1, false), visitedB(n+1, false);
        //v
        
        //for(int )
        
        return m-res;
    }
};
