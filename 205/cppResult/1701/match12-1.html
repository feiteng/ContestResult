class Solution {
public:
    
    int find (vector<int>  &f, int p) {
            if (f[p] != p) {
                f[p] = find(f, f[p]);
            }  
        
        return f[p];
    }
    void unite(vector<int>  &f, int p, int q) {
        int pp = find(f, p);
        int pq = find(f, q);
        
        if (pp != pq) {
            f[pq] = pp;
            
        }
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<int> fa(n + 1, 0);

        vector<int> fb(n + 1, 0);
        for (int i = 1 ;i <=  n; ++i) fa[i] = i, fb[i] = i; 
        int res = 0;
        for(auto e : edges) {
            if (e[0] == 3) {
                int p = e[1], q = e[2];
                int pp = find(fa, p);
                int pq = find(fa, q);
                if (pp != pq) {
                     fa[pq] = pp;

                    fb[pq] = pp;
                } else {
                    res += 1;
                }
                
               
            }
        }
        
        for(auto e : edges) {
            if (e[0] == 1) {
                int p = e[1], q = e[2];
                int pp = find(fa, p);
                int pq = find(fa, q);
                if (pp != pq) {
                     fa[pq] = pp;
                } else {
                    res += 1;
                }
            } else if (e[0] == 2) {
                int p = e[1], q = e[2];
                int pp = find(fb, p);
                int pq = find(fb, q);

                if (pp != pq) {
                     fb[pq] = pp;
                } else {
                    res += 1;
                }
            }
        }
        int c1 = 0, c2 = 0;
        for (int i = 1;i <= n; ++i) {
            if (find(fa, i) == i) {
                c1 += 1;
            } 
            
            if (find(fb, i) == i) {
                c2 += 1;
            } 
        }
        
        if (c1 != 1 or c2 != 1) return -1; 
        return res;
    }
};
