class Solution {
public:
    
    int f[100010];
    
    int fa[100010];
    int fb[100010];
    
    
    int Find(int x) {
        if (f[x] == x) return x;
       else return f[x] = Find(f[x]);
    }
    
    int FindA(int x) {
        if (fa[x] == x) return x;
       else return fa[x] = FindA(fa[x]);
    }
    
    int FindB(int x) {
       if (fb[x] == x) return x;
       else return fb[x] = FindB(fb[x]);
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for(int i = 1; i <= n; i ++) {
            f[i] = i;
        }
        
        int ans = 0;
        
        int g = n;
        for(int i = 0; i < edges.size(); i++) {
            if(edges[i][0] == 3) {
                int x = Find(edges[i][1]);
                int y = Find(edges[i][2]);
                if(x==y) {
                    ans++;
                    continue;
                }
                
                f[x]=y;
                g--;
            }
        }
        
        
        for(int i = 1; i <= n; i ++) {
            fa[i]=f[i];
            fb[i]=f[i];
        }
        
        
        int ga = g;
        int gb = g;
        
        
        for(int i = 0; i < edges.size(); i++) {
            if(edges[i][0] == 1) {
                int x = FindA(edges[i][1]);
                int y = FindA(edges[i][2]);
                if(x==y) {
                    ans++;
                    continue;
                }
                
                fa[x]=y;
                ga--;
            }
        }
        
        if(ga != 1) return -1;
        
        
        for(int i = 0; i < edges.size(); i++) {
            if(edges[i][0] == 2) {
                int x = FindB(edges[i][1]);
                int y = FindB(edges[i][2]);
                if(x==y) {
                    ans++;
                    continue;
                }
                
                fb[x]=y;
                gb--;
            }
        }
        
          if(gb != 1) return -1;
        
        return ans;
    }
};
