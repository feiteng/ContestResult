const int N = 100005;
int f0[N], f1[N], f2[N];

class Solution {
public:
    int find(int f[], int i) {
        return f[i] == i ? i : f[i] = find(f, f[i]);
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& e) {
        int m = e.size();
        int s = 0, s1 = 0, s2 = 0;
        for (int i = 1; i <= n; i++) {
            f0[i] = i;
            f1[i] = i;
            f2[i] = i;
        }
        for (int i = 0; i < m; i++) {
            int c = e[i][0], a = e[i][1], b = e[i][2];
            if (c == 3) {
                int aa = find(f0, a), bb = find(f0, b);
                if (aa != bb) {
                    f0[aa] = bb;
                    s++;
                    s1++;
                    s2++;

                    aa = find(f1, a), bb = find(f1, b);
                    f1[aa] = bb;
                    aa = find(f2, a), bb = find(f2, b);
                    f2[aa] = bb;
                }
            }
        }
        for (int i = 0; i < m; i++) {
            int c = e[i][0], a = e[i][1], b = e[i][2];

            if (c == 1) {
                int aa = find(f1, a), bb = find(f1, b);
                if (aa != bb) {
                    s++;
                    s1++;
                    f1[aa] = bb;
                }
            } else if (c == 2) {

                int aa = find(f2, a), bb = find(f2, b);
                if (aa != bb) {
                    s++;
                    s2++;
                    f2[aa] = bb;
                }
            }
        }
        if (s1 != n - 1 || s2 != n - 1) {
            return -1;
        }
        return m - s;
    }
};
