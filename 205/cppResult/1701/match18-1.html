class Solution {
public:
    vector<pair<int, int>> e[100001];
    bool v[100001];
    
    int dfs(int x, int tp) {
        if (v[x]) return 0;
        int ret = 1;
        v[x] = 1;
        for (pair<int, int> i : e[x])
            if ((i.first & tp) == tp)
                ret += dfs(i.second, tp);
        return ret;
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for (int i = 1; i <= n; ++i) e[i].clear();
        for (vector<int> i : edges) {
            e[i[1]].push_back(make_pair(i[0], i[2]));
            e[i[2]].push_back(make_pair(i[0], i[1]));
        }
        memset(v, 0, sizeof(v));
        if (dfs(1, 1) != n) return -1;
        memset(v, 0, sizeof(v));
        if (dfs(1, 2) != n) return -1;
        int ans = 0;
        memset(v, 0, sizeof(v));
        for (int i = 1; i <= n; ++i)
            if (!v[i])
                ans += dfs(i, 3) - 1;
        return edges.size() - (2 * n - 2 - ans);
    }
};
