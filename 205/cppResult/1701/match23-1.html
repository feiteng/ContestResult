class UnionFind{
public:
    int uf[100005];
    UnionFind(){
        memset(uf,-1,sizeof uf);
    }
    
    int parent(int i){
        if (uf[i] < 0) return i;
        return uf[i] = parent(uf[i]);
    }
    
    void join(int i, int j){
        int pi = parent(i);
        int pj = parent(j);
        if (pi == pj) return;
        if (size(pi) < size(pj)) swap(pi,pj);
        uf[pi] += uf[pj];
        uf[pj] = pi;
        
    }
    
    int size(int i){
        int pi = parent(i);
        return abs(uf[pi]);
    }
    
    void init(){
        memset(uf,-1,sizeof uf);
    }
};

        UnionFind uf1;
        UnionFind uf2;
        

        int ans = 0;
        for(auto e: edges){
            if (e[0] == 3){
                int i = e[1];
                int j = e[2];
                if (uf1.parent(i) == uf1.parent(j)){
                    ans++;
                }
                if (uf1.parent(i) != uf1.parent(j)){
                    uf1.join(i,j);
                    uf2.join(i,j);
                }
            }
        }
        
        for(auto e:edges){
            if (e[0] == 1){
                int i = e[1];
                int j = e[2];
                if (uf1.parent(i) == uf1.parent(j)){
                    ans++;
                }
                if (uf1.parent(i) != uf1.parent(j)){
                    uf1.join(i,j);
                }
            }

        }
        for(auto e:edges){
            if (e[0] == 2){
                int i = e[1];
                int j = e[2];
                if (uf2.parent(i) == uf2.parent(j)){
                    ans++;
                }
                if (uf2.parent(i) != uf2.parent(j)){
                    uf2.join(i,j);
                }
            }
        }
        
        
        if (uf1.size(1) != n || uf2.size(1) != n) return -1;
        return ans;
        
        
    }
