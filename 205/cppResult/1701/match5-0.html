const int maxn = 1e5+5;
class Solution {
public:
    int pre[maxn],pre2[maxn];
    int Find(int x){
        return pre[x]==x?x:pre[x] = Find(pre[x]);
    }

    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for(int i = 1;i <= n;++i)pre[i] = i;
        int need = 0;
        for(auto &x:edges){
            if(x[0]==3){
                int a = Find(x[1]);
                int b = Find(x[2]);
                if(a!=b){
                    pre[a]=b;
                    need++;
                }
            }
        }
        for(int i = 1;i <= n;++i)pre2[i] = pre[i];
        for(auto &x:edges){
            if(x[0]==1){
                int a = Find(x[1]);
                int b = Find(x[2]);
                if(a!=b){
                    pre[a]=b;
                    need++;
                }
            }
        }
        int cnt = 0;
        for(int i = 1;i <= n;++i){
            if(pre[i]==i)cnt++;
        }


        for(int i = 1;i <= n;++i)pre[i] = pre2[i];
        for(auto &x:edges){
            if(x[0]==2){
                int a = Find(x[1]);
                int b = Find(x[2]);
                if(a!=b){
                    pre[a]=b;
                    need++;
                }
            }
        }
        cnt=0;
        for(int i = 1;i <= n;++i){
            if(pre[i]==i)cnt++;
        }
        if(cnt>1)return -1;
        return edges.size()-need;
    }
};
