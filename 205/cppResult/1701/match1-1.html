class Solution {
public:
    vector<int> g[100010], a[100010], b[100010];

    int vis[100010];
    int f[100010], F1[100010], F2[100010];
    int Find(int x){
        return x == f[x] ? x : f[x] = Find(f[x]);
    }
    int Find1(int x){
        return x == F1[x] ? x : F1[x] = Find1(F1[x]);
    }
    int Find2(int x){
        return x == F2[x] ? x : F2[x] = Find2(F2[x]);
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {

        for(int i = 1; i <= n; ++i) f[i] = i;
        int ans = 0;
        for(int i = 0; i < edges.size(); ++i){
            int t = edges[i][0];
            int u = edges[i][1];
            int v = edges[i][2];
            if(t == 3) {
                int f1 = Find(u), f2 = Find(v);
                if(f1 != f2) {
                    f[f1] = f2;
                    vis[i] = 1;
                    ans++;
                }
            }
        }

        for(int i = 1; i <= n; ++i) F1[i] = f[i];
        
         for(int i = 0; i < edges.size(); ++i){
            int t = edges[i][0];
            int u = edges[i][1];
            int v = edges[i][2];
            if(t == 1) {
                int f1 = Find1(u), f2 = Find1(v);
               // cout << f1 << " " << f2 << endl;
                if(f1 != f2) {
                    F1[f1] = f2;
                    vis[i] = 1;
                    ans++;
                }
            }
        }
        for(int i = 1; i <= n; ++i) F2[i] = f[i];

         for(int i = 0; i < edges.size(); ++i){
            int t = edges[i][0];
            int u = edges[i][1];
            int v = edges[i][2];
            if(t == 2) {
                int f1 = Find2(u), f2 = Find2(v);
                if(f1 != f2) {
                    F2[f1] = f2;
                    vis[i] = 1;
                    ans++;
                }
            }
        }
        int t1 = Find1(1), t2 = Find2(1);
        for(int i = 1; i <= n; ++i){
          //  cout << Find2(i) << endl;
            if(Find1(i) != t1 || Find2(i) != t2) return -1;
        }
        return (int)edges.size()-ans;
    }
};
