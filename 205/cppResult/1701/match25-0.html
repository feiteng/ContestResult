class Solution {
public:
    typedef pair<int,pair<int,int> > Edge;
    int f[2][100500];
    Edge edge[100500];
    
    int getf(int k, int p) {
        if(f[k][p] == p) return p;
        else return f[k][p]=getf(k, f[k][p]);
    }
    
    void Merge(int p, int x, int y) {
        int fx = getf(p,x);
        int fy = getf(p,y);
        f[p][fx] = f[p][fy] = min (fx,fy);
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for(int i = 1; i <=n;++i) {
            f[1][i] = i;
            f[0][i] = i;
        }
        int m = edges.size();
        for(int i = 0; i<m;++i) {
            edge[i] = make_pair(-1*edges[i][0],make_pair(edges[i][1], edges[i][2]));
        }
        sort(edge,edge+m);
        int ans = 0;
        for(int i = 0; i < m; ++i) {
            int u = edge[i].second.first;
            int v = edge[i].second.second;
            if(edge[i].first==-3) {
                if(getf(0,u) != getf(0,v) && getf(1,u) != getf(1,v)) {
                    ans += 1;
                    Merge(0,u,v);
                    Merge(1,u,v);
                }
            } else if (edge[i].first == -2) {
                if(getf(0,u) != getf(0,v)) {
                    ans += 1;
                    Merge(0,u,v);
                }
            } else if(edge[i].first == -1) {
                if(getf(1,u) != getf(1,v)) {
                    ans += 1;
                    Merge(1,u,v);
                }
            }
        }
        for(int i = 1; i <= n; ++i) {
            if(getf(0,1) != getf(0,i) || getf(1,1) != getf(1,i)) {
                return -1;
            }
        }
        return m-ans;
    }
};
