class Solution {
public:
    int fa[100007], rk[100007];
    int find (int x) {
        if (fa[x] != x) {
            return fa[x] =find(fa[x]);
        }
        return x;
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        for (int i = 1; i <= n; i++) {
            fa[i] = i;
            rk[i] = 1;
        }
        int cnt = 0, cn = 0;
        for (int i = 0; i < edges.size(); i++) {
            if (edges[i][0] == 3) {
                int fx = find(edges[i][1]);
                int fy = find(edges[i][2]);
                if (fx != fy) {
                    fa[fx] = fy;
                    rk[fy] += rk[fx];
                    cnt++;
                    
                } else {
                    cn++;
                }
            }
        }
        int x = find(1);
        int cat = 0;
        if (rk[x] == n) return edges.size() - cnt;
        for (int i = 0; i < edges.size(); i++) {
            if (edges[i][0] == 1) {
                int fx = find(edges[i][1]);
                int fy = find(edges[i][2]);
                 if (fx != fy) {
                    fa[fx] = fy;
                    rk[fy] += rk[fx];
                    
                } else {
                    cat++;     
                }
            }
        }
        x = find(1);
        if (rk[x] != n) return -1;
        int ans = cat + cn;
        for (int i = 1; i <= n; i++) {
            fa[i] = i;
            rk[i] = 1;
        }
        for (int i = 0; i < edges.size(); i++) {
            if (edges[i][0] == 3) {
                int fx = find(edges[i][1]);
                int fy = find(edges[i][2]);
                if (fx != fy) {
                    fa[fx] = fy;
                    rk[fy] += rk[fx];
                } 
            }
        }
        int count = 0;
        for (int i = 0; i < edges.size(); i++) {
            if (edges[i][0] == 2) {
                int fx = find(edges[i][1]);
                int fy = find(edges[i][2]);
                if (fx != fy) {
                    fa[fx] = fy;
                    rk[fy] += rk[fx];
                } else {
                    count++;
                } 
            }
        }
        x = find(1);
        if (rk[x] != n) return -1;
        return ans + count;
        
        
    }
};
