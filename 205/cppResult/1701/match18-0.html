class Solution {
public:

		int give(int par[],int x)
		{
				if(par[x]==x)
						return x;
				return par[x]=give(par,par[x]);
		}

    int maxNumEdgesToRemove(int n, vector<vector<int>>& ar) {
				int par1[n+10],par2[n+10];
				for(int i=0;i<n+5;++i)
						par1[i]=par2[i]=i;
				int ret=0,c1=0,c2=0;
				for(int i=0;i<ar.size();++i)
				{
						if(ar[i][0]==3)
						{
								int p1=give(par1,ar[i][1]),p2=give(par1,ar[i][2]);
								if(p1!=p2)
								{
										++ret;
										par1[p2]=p1;
										par2[p2]=p1;
										++c1,++c2;
								}
						}
				}
				for(int i=0;i<ar.size();++i)
				{
						if(ar[i][0]==1)
						{
								int p1=give(par1,ar[i][1]),p2=give(par1,ar[i][2]);
								if(p1!=p2)
								{
										++ret;
										par1[p2]=p1;
										++c1;
								}
						}
				}
				for(int i=0;i<ar.size();++i)
				{
						if(ar[i][0]==2)
						{
								int p1=give(par2,ar[i][1]),p2=give(par2,ar[i][2]);
								if(p1!=p2)
								{
										++ret;
										par2[p2]=p1;
										++c2;
								}
						}
				}
				if(c1!=n-1||c2!=n-1)
						return -1;
				return ar.size()-ret;
        
    }
};
