class Solution {
    int mp[100005];
    int mpa[100005];
    int mpb[100005];
    int n;
    int fd(int k)
    {
        int ok=k;
        while(mp[k]!=k)
        {
            k=mp[k];
        }
        while(mp[ok]!=ok)
        {
           
            ok=mp[ok];
             mp[ok]=k;
        }
        return k;
    }
    void lk(int a, int b)
    {
        int pa=fd(a);
        mp[b]=pa;
    }
    int fda(int k)
    {
        int ok=k;
        while(mpa[k]!=k)
        {
            k=mpa[k];
        }
        while(mpa[ok]!=ok)
        {
           
            ok=mpa[ok];
             mpa[ok]=k;
        }
        return k;
    }
    void lka(int a, int b)
    {
        int pa=fda(a);
        mpa[b]=pa;
    }
    int fdb(int k)
    {
        int ok=k;
        while(mpb[k]!=k)
        {
            k=mpb[k];
        }
        while(mpb[ok]!=ok)
        {
           
            ok=mpb[ok];
             mpb[ok]=k;
        }
        return k;
    }
    void lkb(int a, int b)
    {
        int pa=fdb(a);
        mpb[b]=pa;
    }
public:
    int maxNumEdgesToRemove(int nn, vector<vector<int>>& edges) {
        memset(mp,0,sizeof mp);
        n=nn;
        vector<pair<int,int>>e0,e1,e2;

        for(auto v:edges)
        {
            if(v[0]==3)
            {
                e0.push_back({v[1],v[2]});//both
            }
            else if(v[0]==1)
            {
                e1.push_back({v[1],v[2]});//alice
            }
            else
            {
                e2.push_back({v[1],v[2]});//bob
            }
        }
        for(int i=1;i<=n;i++) mp[i]=i;

        int ans=0;
        for(auto e:e0)
        {
            int pa=fd(e.first);
            int pb=fd(e.second);
            if(pa!=pb)
            {
                lk(pa,pb);
                ans++;
            }
        }
        memcpy(mpa,mp,sizeof mp);
        memcpy(mpb,mp,sizeof mp);
        for(auto e:e1)
        {
            int pa=fda(e.first);
            int pb=fda(e.second);
            if(pa!=pb)
            {
                lka(pa,pb);

                ans++;
            }
        }
        for(auto e:e2)
        {
            int pa=fdb(e.first);
            int pb=fdb(e.second);
            if(pa!=pb)
            {
                lkb(pa,pb);
                ans++;
            }
        }
        //int pa0=fd(1);
        //for(int i=2;i<=n;i++) if(fd(i)!=pa0) return -1;
        int pa1=fda(1);
                for(int i=2;i<=n;i++) if(fda(i)!=pa1) return -1;
int pa2=fdb(1);
                for(int i=2;i<=n;i++) if(fdb(i)!=pa2) return -1;
        return edges.size()-ans;
    }
};
