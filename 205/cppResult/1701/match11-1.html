class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<int> afather, bfather;
        for (int i = 0; i <= n; ++i) {
            afather.push_back(i);
            bfather.push_back(i);
        }
        int result = 0;
        for (int i = 0; i < edges.size(); ++i) {
            int x = edges[i][1], y = edges[i][2];

            if (edges[i][0] == 3) {
                int afx = getfather(x, afather), afy = getfather(y, afather);
                int bfx = getfather(x, bfather), bfy = getfather(y, bfather);
                if (afx == afy && bfx == bfy) {
                    result++;
                }
                else {
                    afather[afx] = afy;
                    bfather[bfx] = bfy;
                }
            }     
        }
        for (int i = 0; i < edges.size(); ++i) {
            int x = edges[i][1], y = edges[i][2];

            if (edges[i][0] == 1) {
                int afx = getfather(x, afather), afy = getfather(y, afather);
                if (afx == afy) {
                    result++;
                }
                else {
                    afather[afx] = afy;
                }
            }

            else if (edges[i][0] == 2) {
                int bfx = getfather(x, bfather), bfy = getfather(y, bfather);
                if (bfx == bfy) {
                    result++;
                }
                else {
                    bfather[bfx] = bfy;
                }
            }    
        }
        bool check_all = true;
        int af = getfather(1, afather), bf = getfather(1, bfather);
        for (int i = 1; i <= n; ++i) {
            if (getfather(i, afather) != af || getfather(i, bfather) != bf) {
                check_all = false;
                break;
            }
        }
        return check_all ? result : -1;
    }
    
    int getfather(int x, vector<int>& father) {
        return (x == father[x] ? x : father[x] = getfather(father[x], father));
    }
};
