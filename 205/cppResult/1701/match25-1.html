class Solution {
public:
    int get_pr(int n, vector < int > &p){
        //cout << p[n] <<" "<<n<<endl;
        if(p[n]==n) return n;

        return p[n] = get_pr(p[n], p);
    }
    void merge(int n, int m, vector<int>&p){
        int p1 = get_pr(n, p);
        int p2 = get_pr(m, p);
        p[p1]=p2;
    }
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector < int > pr, one, two;
        pr.resize(n+2);

        for(int i =1; i <= n; i ++){
            pr[i] = i;
        }
        int take = 0;
        for(int i =0; i < edges.size() ; i ++){
            if(edges[i][0]==3){
                int u = edges[i][1];
                int v = edges[i][2];
                if(get_pr(u, pr)!=get_pr(v, pr)){
                    take++;
                    merge(u, v, pr);
                }
            }
        }
        one = pr;
        two = pr;
        for(int i =0; i < edges.size() ; i ++){
            if(edges[i][0]==1){
                int u = edges[i][1];
                int v = edges[i][2];
                if(get_pr(u, one)!=get_pr(v, one)){
                    take++;
                    merge(u, v, one);
                }
            }
            if(edges[i][0]==2){
                int u = edges[i][1];
                int v = edges[i][2];

                if(get_pr(u, two)!=get_pr(v, two)){
                    take++;
                    merge(u, v, two);
                }
            }
            
        }
        for(int i =1; i <= n ; i ++){
            if(i!=1){
                if(get_pr(i, one)!=get_pr(i-1, one)){
                    return -1;
                }
                
                if(get_pr(i, two)!=get_pr(i-1, two)){
                    return -1;
                }
            }
        }
        return edges.size()-take;
        
        
    }
};
