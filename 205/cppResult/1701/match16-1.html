class Solution {
private:

    vector<int> fa, tfa;
    int find(int ck){return ck == fa[ck] ? ck : fa[ck] = find(fa[ck]);}
    int tfind(int ck){return ck == tfa[ck] ? ck : tfa[ck] = tfind(tfa[ck]);}
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int m = edges.size(), s0 = 0, s1 = 0, s2 = 0;
        fa = vector<int>(n+1, 0);
        for(int i = 1; i <= n; i++){
            fa[i] = i;
        }

        for(int i = 0; i < m; i++){
            if( edges[i][0] == 3 ){
                int u = find(edges[i][1]), v = find(edges[i][2]);
                if( u != v ){
                    fa[u] = v;
                    s0++;
                }
            }
        }

        tfa = fa;
        for(int i = 0; i < m; i++){
            if( edges[i][0] == 1 ){
                int u = find(edges[i][1]), v = find(edges[i][2]);
                if( u != v ){
                    fa[u] = v;
                    s1++;
                }
            }else if( edges[i][0] == 2 ){
                int u = tfind(edges[i][1]), v = tfind(edges[i][2]);
                if( u != v ){
                    tfa[u] = v;
                    s2++;
                }
            }
        }
        int ck1 = find(1), ck2 = tfind(1);
        for(int i = 2; i <= n; i++){
            if( ck1 != find(i) || ck2 != tfind(i) ){
                return -1;
            }
        }
        return m - (s0 + s1 + s2);
    }
};
