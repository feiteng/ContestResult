class UnionFind{
    public:
        vector<int> sz,parent;
        UnionFind(const int n){
            sz.reserve(n);
            sz.assign(n, 1);
            parent.reserve(n);
            for(int i = 0; i < n; ++i)
                parent.push_back(i);
        }

    int findParent(int x){
        while(x != parent[x]){
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }
    
    bool merge(int x, int y){
        int p = findParent(x);
        int q = findParent(y);
        if(p == q) return false;
        else if(sz[p] >= sz[q]){
            sz[p] += sz[q];
            parent[q] = p;
        }
        else{
            sz[q] += sz[p];
            parent[p] = q;
        }
        return true;
    }
};

class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int N = edges.size();
        vector<int> vI[4];
        for(int i=0;i<N;i++){
            vI[edges[i][0]].push_back(i);
        }
        UnionFind uSet1(n+1);
        UnionFind uSet2(n+1);
        int ans = 0;
        for(auto ei:vI[3]){
            int x = edges[ei][1];
            int y = edges[ei][2];
            int px=uSet1.findParent(x);
            int py=uSet1.findParent(y);
            if(px==py){
                ans++;
                continue;
            }
            uSet1.merge(px,py);
            uSet2.merge(px,py);
        }
        for(auto ei:vI[1]){
            int x = edges[ei][1];
            int y = edges[ei][2];
            int px=uSet1.findParent(x);
            int py=uSet1.findParent(y);
            if(px==py){
                ans++;
                continue;
            }
            uSet1.merge(px,py);
        } 
        for(auto ei:vI[2]){
            int x = edges[ei][1];
            int y = edges[ei][2];
            int px=uSet2.findParent(x);
            int py=uSet2.findParent(y);
            if(px==py){
                ans++;
                continue;
            }
            uSet2.merge(px,py);
        }  
        int root1=uSet1.findParent(1);
        if(uSet1.sz[root1]!=n){
            return -1;
        }
        int root2=uSet2.findParent(1);
        if(uSet2.sz[root2]!=n){
            return -1;
        }
        return ans;
        
    }
};
