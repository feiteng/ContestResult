class Solution {
public:
    
    int find_par(vector<int>& par, int i){
        if(par[i]==i)return i;
        return par[i] = find_par(par,par[i]);
    }
    
    void uni(vector<int>& par, int i,int j){
        par[find_par(par,j)]=find_par(par,i);    
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<int>par = vector<int>(n);
        for(int i=0;i<n;i++)par[i]=i;
        int ans=0;
        for(int i=0;i<edges.size();i++){
            auto x=edges[i];
            x[1]--;
            x[2]--;
            if(x[0]==3){
                if(find_par(par,x[1])==find_par(par,x[2]))ans++;
                else uni(par,x[1],x[2]);
            }
        }
        vector<int>par1=vector<int>(par);
        for(int i=0;i<edges.size();i++){
            auto x=edges[i];
            x[1]--;
            x[2]--;
            if(x[0]==1){
                if(find_par(par1,x[1])==find_par(par1,x[2]))ans++;
                else uni(par1,x[1],x[2]);
            }
        }
        for(int i=0;i<n;i++)if(find_par(par1,i)!=find_par(par1,0))return -1;
        vector<int>par2=vector<int>(par);
        for(int i=0;i<edges.size();i++){
            auto x=edges[i];
            x[1]--;
            x[2]--;
            if(x[0]==2){
                if(find_par(par2,x[1])==find_par(par2,x[2]))ans++;
                else uni(par2,x[1],x[2]);
            }
        }
        for(int i=0;i<n;i++)if(find_par(par2,i)!=find_par(par2,0))return -1;
        return ans;
    }
};
