#define F first
#define S second
class Solution {
public:
    int d[100005][2];
    int fi(int x,int tp) {
        return d[x][tp] <= 0 ? x : d[x][tp] = fi(d[x][tp], tp);
    } 
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<pair<int,int>>a;
        vector<pair<int,int>>b;
        vector<pair<int,int>>bo;
        int use = 0;
        for (auto &i : edges) {
            if (i[0] == 1) {
                a.push_back({i[1], i[2]});
            }
            else if (i[0] == 2) {
                b.push_back({i[1], i[2]});
            }
            else {
                bo.push_back({i[1], i[2]});
            }
        }
        int cnt = (n - 1) * 2;
        for (auto &i : bo) {
            int x = fi(i.F, 0);
            int y = fi(i.S, 0);
            if (x != y) {
                use++;
                d[x][0] = y;
                d[x][1] = y;
                cnt -= 2;
            }
        }
        for (auto &i : a) {
            int x = fi(i.F, 0);
            int y = fi(i.S, 0);
            if (x != y) {
                d[x][0] = y;
                cnt--;
                use++;
            }
        }
        for (auto &i : b) {
            int x = fi(i.F, 1);
            int y = fi(i.S, 1);
            if (x != y) {
                d[x][1] = y;
                cnt--;
                use++;
            }
        }
        if (cnt) return -1;
        else return edges.size() - use;
    }
};
