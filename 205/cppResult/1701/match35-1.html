class Solution {
public:
    int f[100010], f1[100010], f2[100010];
    
    int find(int k) {
        return f[k] == k? k : f[k] = find(f[k]);
    }
    
    int find1(int k) {
        return f1[k] == k? k : f1[k] = find1(f1[k]);
    }
    

    int find2(int k) {
        return f2[k] == k? k : f2[k] = find2(f2[k]);
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        int ans = 0;
        for(int i = 1; i <= n; i++) f[i] = i;
        for(int i = 0; i < edges.size(); i++) {
            if(edges[i][0] == 3) {
                int a = edges[i][1], b = edges[i][2];
                if(find(a) != find(b)) f[find(a)] = find(b);
                else ans++;
            }
        }

        for(int i = 1; i <= n; i++) f1[i] = f2[i] = f[i];
        for(int i = 0; i < edges.size(); i++) {
            int a = edges[i][1], b = edges[i][2];
            if(edges[i][0] == 1) {
                if(find1(a) != find1(b)) f1[find1(a)] = find1(b);
                else ans++;
            }
            else if(edges[i][0] == 2) {
                if(find2(a) != find2(b)) f2[find2(a)] = find2(b);
                else ans++;
            }
        }
        for(int i = 1; i <= n; i++) if(find1(f1[i]) != find1(f1[1]) || find2(f2[i]) != find2(f2[1])) return -1;
        return ans;
    }
};
