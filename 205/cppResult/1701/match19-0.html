class Solution {
public:
    
    int myfind(int a, vector<int>& relation) {
        if (relation[a] == a) return a;
        int res = myfind(relation[a], relation);
        relation[a] = res;
        return res;
    }

    void myunion(int a, int b, vector<int>& relation) {
        int ta = myfind(a, relation);
        int tb = myfind(b, relation);
        if (ta == tb) return;
        relation[ta] = tb;
    }

    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        vector<pair<int, int>> vp1, vp2, vp3;
        for (auto e : edges) {
            if (e[0] == 1) {
                vp1.push_back(make_pair(e[1], e[2]));
            }
            if (e[0] == 2) {
                vp2.push_back(make_pair(e[1], e[2]));
            }
            if (e[0] == 3) {
                vp3.push_back(make_pair(e[1], e[2]));
            }
        }
        vector<int> r1(n + 1, 0), r2(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            r1[i] = i;
            r2[i] = i;
        }
        int res = 0;
        for (auto p : vp3) {
            int a = p.first, b = p.second;
            if (myfind(a, r1) == myfind(b, r1) && myfind(a, r2) == myfind(b, r2)) {
                res++;
            } else {
                myunion(a, b, r1);
                myunion(a, b, r2);
            }
        }
        
        for (auto p : vp1) {
            int a = p.first, b = p.second;
            if (myfind(a, r1) == myfind(b, r1)) {
                res++;
            } else {
                myunion(a, b, r1);
            }
        }
        for (auto p : vp2) {
            int a = p.first, b = p.second;
            if (myfind(a, r2) == myfind(b, r2)) {
                res++;
            } else {
                myunion(a, b, r2);
            }
        }
        for (int i = 2; i <= n; i++) {
            if (myfind(i, r1) != myfind(i - 1, r1) || myfind(i, r2) != myfind(i - 1 ,r2)) {
                return -1;
            }
        }
        return res;
    }
};
