const int maxn = 100005;
int fa[maxn];
int N;

void init() {
    for (int i = 0; i <= N; i++)
        fa[i] = i;
}

int find_fa(int x){
	fa[x] = (fa[x]==x?fa[x]:find_fa(fa[x]));
	return fa[x];
}

void merge(int a,int b){
	int faa = find_fa(a);
	int fab = find_fa(b);
	fa[faa] = fab;
	return ;
}


class Solution {
public:
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        N = n;
        int ne = edges.size();
        int ans;
        
        init();
        ans = 0;
        for (int i = 0; i < ne; i++) { //先尽量删3
            if (edges[i][0] == 3) {
                if (find_fa(edges[i][1]) == find_fa(edges[i][2]))
                    ans++;
                else
                    merge(edges[i][1], edges[i][2]);
            }
        }
        for (int i = 0; i < ne; i++) {  //再考虑1
            if (edges[i][0] == 1) {
                if (find_fa(edges[i][1]) == find_fa(edges[i][2]))
                    ans++;
                else
                    merge(edges[i][1], edges[i][2]);
            }
        }
        for(int i = 2; i <= n; i++) {
            if (find_fa(i - 1) != find_fa(i)) {
                return -1;
            }
        }
        
        init();
        for (int i = 0; i < ne; i++) { //先尽量删3
            if (edges[i][0] == 3) 
                merge(edges[i][1], edges[i][2]);
        }
        for (int i = 0; i < ne; i++) {  //再考虑2
            if (edges[i][0] == 2) {
                if (find_fa(edges[i][1]) == find_fa(edges[i][2]))
                    ans++;
                else
                    merge(edges[i][1], edges[i][2]);
            }
        }        
        for(int i = 2; i <= n; i++) {
            if (find_fa(i - 1) != find_fa(i)) {
                return -1;
            }
        }
        return ans;
        
    }
};
