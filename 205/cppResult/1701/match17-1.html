class Solution {
public:
    
    struct union_find
    {
        vector<int> p;

        union_find(int n) : p(n, -1) {}

        bool join(int u, int v)
        {
            if ((u = root(u)) == (v = root(v)))
                return false;
            if (p[u] > p[v])
                swap(u, v);
            p[u] += p[v];
            p[v] = u;
            return true;
        }

        int root(int u)
        {
            return p[u] < 0 ? u : p[u] = root(p[u]);
        }
    };
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& edges) {
        union_find ds1(n), ds2(n);

        int ans = 0;
        
        for(int i = 0; i < edges.size(); i++){
            if(edges[i][0] == 3){
                int a = edges[i][1] - 1, b = edges[i][2] - 1;
                if(!ds1.join(a, b))
                    ans++;
                ds2.join(a, b);

            }
        }
        
        for(int i = 0; i < edges.size(); i++){
            if(edges[i][0] == 1){
                int a = edges[i][1] - 1, b = edges[i][2] - 1;
                if(!ds1.join(a, b))
                    ans++;

            }
        }
        
        for(int i = 0; i < edges.size(); i++){
            if(edges[i][0] == 2){
                int a = edges[i][1] - 1, b = edges[i][2] - 1;
                if(!ds2.join(a, b))
                    ans++;
            }
        }
        
        set<int> r1, r2;
        for(int i = 0; i < n; i++){
            r1.insert(ds1.root(i));
            r2.insert(ds2.root(i));
        }
        
        if(r1.size() > 1 || r2.size() > 1)
            ans = -1;
        
        return ans;
    }
};
