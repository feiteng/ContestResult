#include <bits/stdc++.h>

using namespace std;


class Solution {
private:
    vector<int> fa;

    int fis(int x) {
        return fa[x] == x ? x : fa[x] = fis(fa[x]);
    }

public:
    int maxNumEdgesToRemove(int n, vector<vector<int>> &edges) {
        fa.resize(n + 1);
        for (int i = 1; i <= n; i++)
            fa[i] = i;
        int ans = 0, conn = 0;
        for (auto &i:edges)
            if (i[0] == 3) {
                if (fis(i[1]) != fis(i[2])) {
                    fa[fis(i[1])] = fis(i[2]);
                    conn++;
                } else
                    ans++;
            }
        auto ufa = fa;
        auto uconn = conn;
        for (auto &i:edges)
            if (i[0] == 1) {
                if (fis(i[1]) != fis(i[2])) {
                    fa[fis(i[1])] = fis(i[2]);
                    conn++;
                } else
                    ans++;
            }
        if (conn != n - 1)
            return -1;
        conn = uconn;
        fa = ufa;
        for (auto &i:edges)
            if (i[0] == 2) {
                if (fis(i[1]) != fis(i[2])) {
                    fa[fis(i[1])] = fis(i[2]);
                    conn++;
                } else
                    ans++;
            }
        if (conn != n - 1)
            return -1;
        return ans;
    }
};
