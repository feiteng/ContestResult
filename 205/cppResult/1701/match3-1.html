class Solution {
public:
    int fa[100005],fb[100005];
    int finda(int x){
        if(fa[x] == x) return x; 
        fa[x] = finda(fa[x]);
        return finda(fa[x]);
    }
    
    int findb(int x){
        if(fb[x] == x) return x; 
        fb[x] = findb(fb[x]);
        return findb(fb[x]);
    }
    
    int maxNumEdgesToRemove(int n, vector<vector<int>>& e) {
        for(int i=0;i<=n;i++){
            fa[i] = fb[i] = i;
        }
        

        int cnt = 0;
        for(int i=0;i<e.size();i++){
            if(e[i][0] == 3){
                int fx = finda(e[i][1]);
                int fy = finda(e[i][2]);
                if(fx != fy){
                    cnt++;
                    fa[fx] = fy;
                }
                
                fx = findb(e[i][1]);
                fy = findb(e[i][2]);
                if(fx != fy){
                    fb[fx] = fy;
                }

            }
        }
        
        for(int i=0;i<e.size();i++){
            if(e[i][0] == 1){
                int fx = finda(e[i][1]);
                int fy = finda(e[i][2]);
                if(fx != fy){
                    cnt++;
                    fa[fx] = fy;
                }

            }
        }
        
        for(int i=0;i<e.size();i++){
            if(e[i][0] == 2){
                int fx = findb(e[i][1]);
                int fy = findb(e[i][2]);
                if(fx != fy){
                    cnt++;
                    fb[fx] = fy;
                }
            }
        }
        int na = 0,nb = 0;
        for(int i=1;i<=n;i++){
            if(fa[i] == i) na++;
            if(fb[i] == i) nb++;
        }
        int ans = e.size() - cnt; 
        if(na!=1 || nb!=1) return -1;
        else return ans; 
    }
};
