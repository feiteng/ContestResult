from collections import defaultdict

class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        c = 0
        nums1_set = defaultdict(lambda : 0)
        nums2_set = defaultdict(lambda : 0)
        for n in nums1:
            nums1_set[n*n] += 1
        for n in nums2:
            nums2_set[n*n] += 1
        for i in range(len(nums1)):
            for j in range(i+1, len(nums1)):
                r = nums1[i] * nums1[j]
                if r in nums2_set:
                    c += nums2_set[r]
        for i in range(len(nums2)):
            for j in range(i+1, len(nums2)):
                r = nums2[i] * nums2[j]
                if r in nums1_set:
                    c += nums1_set[r]
        return c
