class Solution:
    def numTriplets(self, A: List[int], B: List[int]) -> int:
        res = 0
        d = defaultdict(int)
        for i in range(len(A)):
            for j in range(i+1, len(A)):
                d[A[i]*A[j]] += 1
        for n in B:
            res += d[n**2]
            
        A, B = B, A
        d = defaultdict(int)
        for i in range(len(A)):
            for j in range(i+1, len(A)):
                d[A[i]*A[j]] += 1
        for n in B:
            res += d[n**2]
        return res
