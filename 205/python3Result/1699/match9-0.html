class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        d,s={},0
        for x in nums2: d[x]=d.get(x,0)+1
        for x in nums1:
            dd,x={i:d[i] for i in d},x**2
            for y in nums2:
                dd[y]-=1
                if not x%y: s+=dd.get(x//y,0)
        d={}
        for x in nums1: d[x]=d.get(x,0)+1
        for x in nums2:
            dd,x={i:d[i] for i in d},x**2
            for y in nums1:
                dd[y]-=1
                if not x%y: s+=dd.get(x//y,0)
        return s
