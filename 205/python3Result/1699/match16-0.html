class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def get(nums1, nums2):
            cnt1 = collections.Counter()
            for n in nums1:
                cnt1[n*n] += 1
                
            res = 0
            N = len(nums2)
            
            for j in range(N):
                for k in range(j+1, N):
                    key = nums2[j]*nums2[k]
                    if key in cnt1:
                        res += cnt1[key]
            return res
    
        return get(nums1, nums2) + get(nums2, nums1)
