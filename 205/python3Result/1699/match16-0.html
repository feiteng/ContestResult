class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        counts1,counts2 = collections.Counter(),collections.Counter()
        for i in range(len(nums1)):
            for j in range(i+1,len(nums1)):
                counts1[nums1[i]*nums1[j]]+=1
        for i in range(len(nums2)):
            for j in range(i+1,len(nums2)):
                counts2[nums2[i]*nums2[j]]+=1
                
        res = 0
        cur1,cur2 = collections.Counter(nums1),collections.Counter(nums2)
        for i in cur1.keys():
            if i**2 in counts2:
                res+=cur1[i]*counts2[i**2]
        
        for i in cur2.keys():
            if i**2 in counts1:
                res+=cur2[i]*counts1[i**2]
        return res
