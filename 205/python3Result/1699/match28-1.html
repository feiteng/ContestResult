class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        xy1 = collections.defaultdict(int)
        xy2 = collections.defaultdict(int)
        for i in range(len(nums1)):
            for j in range(i + 1, len(nums1)):
                xy1[nums1[i] * nums1[j]] += 1
                
        for i in range(len(nums2)):
            for j in range(i + 1, len(nums2)):
                xy2[nums2[i] * nums2[j]] += 1
        
        ans = 0
        for i in range(len(nums1)):
            ans += xy2[nums1[i] ** 2]
        for i in range(len(nums2)):
            ans += xy1[nums2[i] ** 2]
        return ans
