class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        ans = 0
        prod1 = defaultdict(int)
        for i in range(len(nums2)):
            for j in range(i + 1, len(nums2)):
                prod1[nums2[i] * nums2[j]] += 1
        
        for num in nums1:
            ans += prod1[num * num]
        
        prod2 = defaultdict
(int)
        for i in range(len(nums1)):
            for j in range(i + 1, len(nums1)):
                prod2[nums1[i] * nums1[j]] += 1
        
        for num in nums2:
            ans += prod2[num * num]
        return ans
