import collections 
class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        counts1 = collections.Counter(i*i for i in nums1)
        counts2 = collections.Counter(i*i for i in nums2)
        ans = 0 
        for i in range(0, len(nums1)):
            for j in range(i+1, len(nums1)):
                ans += counts2[nums1[i]*nums1[j]] 
        for i in range(0, len(nums2)):
            for j in range(i+1, len(nums2)):
                ans += counts1[nums2[i]*nums2[j]]
        return ans
