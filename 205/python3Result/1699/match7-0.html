class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        ans = 0
        cnt = collections.Counter([x**2 for x in nums1])
        for i in range(len(nums2)):
            for j in range(i+1, len(nums2)):
                ans += cnt[nums2[i] * nums2[j]]
                
        cnt = collections.Counter([x**2 for x in nums2])
        for i in range(len(nums1)):
            for j in range(i+1, len(nums1)):
                ans += cnt[nums1[i] * nums1[j]]
        return ans
