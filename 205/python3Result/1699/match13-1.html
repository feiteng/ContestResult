class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        cnt1 = Counter(nums1)
        cnt2 = Counter(nums2)
        ans = 0
        ans += self.check(cnt1,cnt2)
        ans += self.check(cnt2,cnt1)
        return ans
    
    def check(self,cnt1,cnt2):
        ans = 0
        for x in cnt1:
            for y in cnt2:
                if x*x % y != 0:
                    continue
                z = x*x // y
                if y == z:
                    ans += cnt1[x]*(cnt2[y]-1)*(cnt2[y])//2
                elif y < z:
                    ans += cnt1[x]*cnt2[y]*cnt2[z]
        return ans
