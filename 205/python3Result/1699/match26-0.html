class Solution:
    def numTriplets(self, A: List[int], B: List[int]) -> int:
        CA = Counter()
        CB = Counter()
        
        m = len(A)
        n = len(B)
        for i in range(m):
            for j in range(i + 1, m):
                CA[A[i] * A[j]] += 1
        for i in range(n):
            for j in range(i + 1, n):
                CB[B[i] * B[j]] += 1
        
        ans = 0
        for a in A:
            ans += CB[a * a]
        for b in B:
            ans += CA[b * b]
        return ans
