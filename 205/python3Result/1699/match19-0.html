class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def calc(a1, a2):
            m = collections.defaultdict(int)
            for x in a1:
                m[x ** 2] += 1
            n = len(a2)
            ret = 0
            for i in range(n):
                for j in range(i + 1, n):
                    x = a2[i] * a2[j]
                    if x in m:
                        ret += m[x]
            return ret
        return calc(nums1, nums2) + calc(nums2, nums1)
