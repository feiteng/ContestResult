class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        
        n1 = len(nums1)
        n2 = len(nums2)
        pairs1 
= Counter()
        for j in range(n2):
            for k in range(j + 1, n2):
                pairs1[nums2[j] * nums2[k]] += 1
        pairs2 = Counter()
        for j in range(n1):
                for k in range(j + 1, n1):
                    pairs2[nums1[j] * nums1[k]] += 1
 
                 
        count = 0
        
for i in range(n1):
            count += pairs1[nums1[i] ** 2]       
        for i in range(n2):
                          count += pairs2[nums2[i] ** 2]
        

        return count
