from itertools import combinations
from collections import Counter

def count_triplets_asym(products, nums):
    return sum(products[a * b] for a, b in combinations(nums, 2))

class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        return (
            count_triplets_asym(Counter(x ** 2 for x in nums1), nums2) +
            count_triplets_asym(Counter(x ** 2 for x in nums2), nums1)
        )
