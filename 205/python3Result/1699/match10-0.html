class Solution:
    def t1(self, a, b):
        n = len(a)
        csum = dict()
        for i in range(n):
            for j in range(i + 1, n):
                v = a[i] * a[j]
                if v not in csum:
                    csum[v] = 1
                else:
                    csum[v] += 1
        ans = 0
        for i in b:
            s = i ** 2
            if s in csum:
                ans += csum[s]
        return ans
    
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        return self.t1(nums1, nums2) + self.t1(nums2, nums1)
