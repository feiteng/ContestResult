class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def cnt_ret(l, d):
            ret = 0
            for i in l:
                for j in d:
                    if i * i % j == 0 and i * i // j in d:
                        # print(i, j)
                        # print(ret, end=' -> ')
                        if i != j:
                            ret += d[i * i // j] * d[j]
                        else:
                            ret += d[j] * (d[j] - 1)
                        # print(ret)
            return ret // 2
        d1 = defaultdict(int)
        d2 = defaultdict(int)
        for i in nums1:
            d1[i] += 1
        for i in nums2:
            d2[i] += 1
        
        return cnt_ret(nums1, d2) + cnt_ret(nums2, d1)
