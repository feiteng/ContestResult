class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        ways1 = Counter()
        ways2 = Counter()
        
        for i in range(len(nums1)):
            for j in range(i+1, len(nums1)):
                p = nums1
[i] * nums1[j]
                ways1[p] += 1
        
        for i in range(len(nums2)):
            for j in range(i+1, len(nums2)):
                p = nums2
[i] * nums2[j]
                ways2[p] += 1
        
        res = 0
        for n in nums1:
            res += ways2[n * n]
        for n in nums2:
            res += ways1[n * n]
        
        return res
