from math import factorial
class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        
        def choose(n, r):
            return factorial(n) // factorial(r) // factorial(n-r)
        
        nums1.sort()
        nums2.sort()
        m, n = len(nums1), len(nums2)
        
        res = 0
        for num1 in nums1:
            idx = bisect.bisect_left(nums2, num1)
            if idx == n:
                break
                
            i, j = idx, idx
            while j < n and nums2[j] == num1:
                j += 1
            
            if j - i >= 2:
                res += choose(j - i, 2)
            i -= 1
            
            while i >= 0 and j < n:
                i2 = i
                while i >= 0 and nums2[i] == nums2[i2]:
                    i -= 1
                
                while j < n and nums2[i2] * nums2[j] <= num1 ** 2:
                    if nums2[i2] * nums2[j] == num1 ** 2:
                        res += i2 - i
                    j += 1
                
        for num2 in nums2:
            idx = bisect.bisect_left(nums1, num2)
            if idx == m:
                break
            
            i, j = idx, idx
            while j < m and nums1[j] == num2:
                j += 1
            
            if j - i >= 2:
                res += choose(j - i, 2)
            i -= 1
            
            while i >= 0 and j < m:
                i2 = i
                while i >= 0 and nums1[i] == nums1[i2]:
                    i -= 1
                
                while j < m and nums1[i2] * nums1[j] <= num2 ** 2:
                    if nums1[i2] * nums1[j] == num2 ** 2:
                        res += i2 - i
                    j += 1
        return res
