class Solution:
    def numTriplets(self, nums1: [int], nums2: [int]) -> int:
        def f(a,b):
            import collections
            c=collections.Counter()
            for x in a:
                c[x*x]+=1
            ans=0
            for i in range(len(b)):
                for j in range(i+1,len(b)):
                    ans+=c[b[i]*b[j]]
            return ans
        return f(nums1,nums2)+f(nums2,nums1)
