class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        m = len(nums2)
        
        d1 = defaultdict(int)
        for i in range(n):
            for j in range(i+1, n):
                  d1[nums1[i] * nums1[j]] += 1
        d2 = defaultdict(int)
        for i in range(m):
            for j in range(i+1, m):
                  d2[nums2[i] * nums2[j]] += 1
        
        res = 0
        for i in range(n):
            res += d2[nums1[i]**2]
        for i in range(m):
            res += d1[nums2[i]**2]
        return res
