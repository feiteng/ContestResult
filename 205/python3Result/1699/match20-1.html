class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        l1 = len(nums1)
        l2 = len(nums2)
        n1 = {}
        n2 = {}
        for i in range(l1):
            n1.setdefault(nums1[i], []).append(i)
        for i in range(l2):
            n2.setdefault(nums2[i], []).append(i)
        ans = 0
        for i in n1:
            si = i * i
            for j in n2:
                if si % j != 0 or si // j not in n2:
                    continue
                for x in n2[j]:
                    for y in n2[si // j]:
                        #print(1, x, y)
                        if x < y:
                            ans += len(n1[i])
        for i in n2:
            si = i * i
            for j in n1:
                if si % j != 0 or si // j not in n1:
                    continue
                for x in n1[j]:
                    for y in n1[si // j]:
                        #print(2, x, y)
                        if x < y:
                            ans += len(n2[i])
        return ans
