class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        one_counts = collections.defaultdict(int)
        two_counts = collections.defaultdict(int)
        
        for i in range(len(nums1)):
            for j in range(i + 1, len(nums1)):
                one_counts[nums1[i] * nums1[j]] += 1
                
        for i in range(len(nums2)):
            for j in range(i + 1, len(nums2)):
                two_counts[nums2[i] * nums2[j]] += 1
                
                
        res = 0
        for n in nums1:
            double = n * n
            res += two_counts[double]
        for n in nums2:
            double = n * n
            res += one_counts[double]
            
        return res
