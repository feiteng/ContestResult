from collections import defaultdict
class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        
        def count(a1, a2):
            d = defaultdict(int)
            for j in range(len(a2)):
                for k in range(j + 1, len(a2)):
                    d[a2[j] * a2[k]] += 1
            ans = 0
            for x in a1:
                ans += d.get(x * x, 0)
            return ans
        
        return count(nums1, nums2) + count(nums2, nums1)
