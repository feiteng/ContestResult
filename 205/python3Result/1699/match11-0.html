class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        nums1.sort()
        nums2.sort()
        num1s = Counter([x*x for x in nums1])
        num2s = Counter([x*x for x in nums2])
        res = 0
        for i in range(len(nums1)):
            for j in range(i + 1, len(nums1)):
                res += num2s[nums1[i] * nums1[j]]
        for i in range(len(nums2)):
            for j in range(i + 1, len(nums2)):
                res += num1s[nums2[i] * nums2[j]]
        return res
