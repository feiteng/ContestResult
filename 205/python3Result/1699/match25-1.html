class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def num(A1, A2):
            ans = 0
            n1 = len(A1)
            n2 = len
(A2)
            m = defaultdict(int)
            for j in range(n2):
                for k in range(j+1, n2):
                    m[A2[j]*A2[k]] += 1
            for i in range(n1):
                ans += m[A1[i]**2]
            return ans
        
        return num(nums1, nums2) + num(nums2, nums1)
