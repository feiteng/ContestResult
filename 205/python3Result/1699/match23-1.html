class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def go(a, b):
            s = collections.Counter()
            for x in a:
                s[x*x] += 1
            
            count = 0
            N = len(b)
            
for i in range(N):
                for j in range(i + 1, N):
                    if b[i] * b[j] in s:
                        count += s[b[i] * b[j]]
                        
            return count
        return go(nums1, nums2) + go(nums2, nums1)
