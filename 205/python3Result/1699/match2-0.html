class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int], final = False) -> int:
        res = 0
        cnt = collections.Counter([i * i for i in nums1])
        for i in range(len(nums2)):
            for j in range(i):
                res += cnt[nums2[i] * nums2[j]]
        if final:
            return res
        return res + self.numTriplets(nums2, nums1, True)
