class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def calc(arr1, arr2):
            sq = [a * a for a in arr1]
            pos = defaultdict(int)
            for a in sq:
                pos[a] += 1

            result = 0
            for j in range(len(arr2)):
                for k in range(j + 1, len(arr2)):
                    result += pos[arr2[j] * arr2[k]]
            return result



        return calc(nums1, nums2) + calc(nums2, nums1)
