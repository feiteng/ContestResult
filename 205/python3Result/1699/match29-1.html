class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        nums1_sq, nums2_sq = collections.Counter(), collections.Counter()
        for n1 in nums1:
            nums1_sq[n1 ** 2] += 1
        
for n2 in nums2:
            nums2_sq[n2 ** 2] += 1
        ans = 0
        
        for i in range(len(nums1)):
            for j in range(i+1, len(nums1)):
                ans += nums2_sq[nums1[i] * nums1[j]]
            
        for i in range(len(nums2)):
            for j in range(i+1, len(nums2)):
                ans += nums1_sq[nums2[i] * nums2[j]]
        
        return ans
