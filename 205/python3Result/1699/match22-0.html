class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        ans = 0
        ls1 = len(nums1)
        ls2 = len(nums2)
        
        pn1 = [num ** 2 for num in nums1]
        pn2 = [num ** 2 for num in nums2]
        
        want1 = defaultdict(int)
        for j in range(ls1-1):
            for k in range(j+1, ls1):
                want1[nums1[j]*nums1[k]] += 1
        
        want2 = defaultdict(int)
        for j in range(ls2-1):
            for k in range(j+1, ls2):
                want2[nums2[j]*nums2[k]] += 1
                
        for i in range(ls1):
            ans += want2[pn1[i]]
        for i in range(ls2):
            ans += want1[pn2[i]]
        
        return ans
