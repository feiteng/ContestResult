class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def work(a, b):
            h 
= collections.Counter()
            for x in a:
                h[x**2] += 1
            ret = 0
            for i in range(len(b)):
                for j in range(i+1, len(b)):
                    ret += h[b[i]*b[j]]
            return ret
        return work(nums1, nums2) + work(nums2, nums1)
