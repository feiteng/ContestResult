class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        
        def count(arr1, arr2):
            
            d = collections.defaultdict(int)
            
            for i in arr1:
                
                d[i * i] += 1
                
            n = len(arr2)
            
            res = 0
            
            for i in range(n - 1):
                for j in range(i + 1, n):
                    
                    if arr2[i] * arr2[j] in d:
                        
                        res += d[ arr2[i] * arr2[j] ]
                        
            return res
        
        ans = count(nums1, nums2) + count(nums2, nums1)
        
        return ans
