class Solution:
    
    def cal(self, a, b):
        
        d = {}
        for x in a:
            i = x*x
            if(i in d):
                d[i] += 1
            else:
                d[i] = 1
        
        n = len(b)
        
        ans = 0
        for i in range(n):
            for j in range(i+1,n):
                r = b[i] * b[j]
                if(r in d):
                    ans += d[r]
        
        return ans
                
    
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        return self.cal(nums1, nums2) + self.cal(nums2, nums1)
