def flood(i, neighbor, visited):
    visited[i] = True
    for n in neighbor[i]:
        if not visited[n]:
            flood(n, neighbor, visited)
    return visited

class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        conn = [set([1]), set([1])]
        count = 0
        neighbor = [[set() for _ in range(n + 1)] for _ in range(4)]
        for (t, i, j) in edges:
            neighbor[t][i].add(j)
            neighbor[t][j].add(i)
            if t == 3:
                
                neighbor[1][i].add(j)
                neighbor[1][j].add(i)
                
                neighbor[2][i].add(j)
                neighbor[2][j].add(i)
                
        if sum(flood(1, neighbor[1], [False for _ in range(n + 1)])) != n:
            return -1
        if sum(flood(1, neighbor[2], [False for _ in range(n + 1)])) != n:
            return -1
        visited = [False for _ in range(n + 1)]
        block = 0
        for i in range(1, n + 1):
            if not visited[i]:
                b = flood(i, neighbor[3], visited)
                block += 1
                
        return len(edges) - sum(visited) + block - (block - 1) * 2
