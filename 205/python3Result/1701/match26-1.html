class Solution:
  def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
    uf1 = UF(n)  
    uf2 = UF(n)
    res = 0
    edges.sort(reverse=True)
    for ty, u, v in edges:
      u -= 1
      v -= 1
      
if ty == 3:
        ru = uf1.find(u)
        rv = uf1.find(v)
        if ru == rv:
          res += 1
        else:
          uf1.union(u, v)
          uf2.union(u, v)
      elif ty == 1:
        ru 
= uf1.find(u)
        rv = uf1.find(v)
        if ru == rv:
          res += 1
        else:
          uf1.union(u, v)
      else:
        ru = uf2.find(u)
        rv = uf2.find(v)
        if ru == rv:
          res += 1
        else:
          uf2.union(u, v)
    c1, c2 = uf1.size(), uf2.size()
    if c1 == n and c2 == n:
      return res
    return -1
        
class UF:
  def __init__(self, n):
    self.parent, self.count = [i for i in range(n)], n*[1]
  def find(self, i):
    root = self.parent[i]
    while root != self.parent[root]: root = self.parent[root]
    while self.parent[i] != root: i, self.parent[i] = self.parent[i], root
    return root
  def union(self, i, j):
    root_i = self.find(i)
    root_j = self.find(j)
    if root_i == root_j: return
    if self.count[root_i] < self.count[root_j]: 
      self.parent[root_i] = root_j
      self.count[root_j] += self.count[root_i]
    else:
      self.parent[root_j] = root_i
      self.count[root_i] += self.count[root_j]
      
  def size(self):
    return max(self.count)
