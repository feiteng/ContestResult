class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        
        f = [i for i in range(n)]
        
        def gf(i):
            if(f[i] == i):
                return i
            ff = gf(f[i])
            f[i] = ff
            return ff
        
        def blk():
            r = 0
            for i in range(n):
                if(f[i] == i):
                    r += 1
            return r
        
        r3 = 0
        
        for i in range(len(edges)):
            edges[i][1] -= 1
            edges[i][2] -= 1
        
        for tu in edges:
            if(tu[0] == 3):
                x, y = gf(tu[1]), gf(tu[2])
                if(x == y):
                    r3 += 1
                else:
                    f[x] = 
y
        
        r1 = 0
        
        for tu in edges:
            if(tu[0] == 1):
                x, y = gf(tu[1]), gf(tu[2])
                if(x == y):
                    r1 += 1
                else:
                    f[x] = y
        
        if(blk() != 1):
            return -1
        
        for i in range(n):
            f[i] = i
        for tu in edges:
            if(tu[0] == 3):
                x,y = gf(tu[1]), gf(tu[2])
                f[x] = y
        
        r2 = 0
        
        
for tu in edges:
            if(tu[0] == 2):
                x, y = gf(tu[1]), gf(tu[2])
                if(x == y):
                    r2 += 1
                else:
                    f[x] = y
        
        
        if(blk() != 1):
            return -1        
        
        return r1 + r2 + r3
