class UnionFind():
    def __init__(self, n):
        self.roots = [i for i in range(n)]
        self.sizes = [1 for i in range(n)]
        
    def root(self, x):
        if self.roots[x] != x:
            self.roots[x] = self.root(self.roots[x])   
        return self.roots[x]            
        
    def add(self, a, b):
        a, b = self.root(a), self.root(b)
        if self.sizes[a] < self.sizes[b]:
            a, b = b, a
        if self.roots[b] == a:
            return                        
        self.roots[b] = a
        self.sizes[a] += self.sizes[b]


class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        # start with no edges
        # initialize two different UnionFind structures for A and B
        a = UnionFind(n+1)
        b = UnionFind(n+1)
        
        count = 0
        for t, s, e in edges:
            if t != 3:
                continue
            if a.root(s) != a.root(e) or b.root(s) != b.root(e):
                a.add(s, e)                                
                b.add(s, e)                                
                count += 1
        for t, s, e in edges:
            if t != 1:
                continue
            if a.root(s) != a.root(e):
                a.add(s, e)                                
                count += 1
        for t, s, e in edges:
            if t != 2:
                continue
            if b.root(s) != b.root(e):
                b.add(s, e)                                
                count += 1
                         
        aRoot = a.root(1)                    
        bRoot = b.root(1)                    
        for i in range(1, n+1):
            if a.root(i) != aRoot:
                return -1
            if b.root(i) != bRoot:
                return -1
        return len(edges) - count
