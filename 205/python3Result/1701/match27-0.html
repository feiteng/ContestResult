from collections import deque
class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        g = [[] for _ in range(n)]
        comp = [None] * n
        cnt_edges = [None, 0, 0, 0]
        for t, u, v in edges:
            u -= 1
            v -= 1
            g[u].append((v, t))
            g[v].append((u, t))
            cnt_edges[t] += 1
        
        def check(tt):
            q = deque([0])
            inq = [False] * n
            inq[0] = True
            while q:
                u = q[0]
                for v, t in g[u]:
                    if (t == 3 or t == tt) and not inq[v]:
                        inq[v] = True
                        q.append(v)
                q.popleft()
            # print(tt, inq)
            return all(inq)
        
        if not check(1) or not check(2):
            return -1
        
        cnt_comp = 0
        for i in range(n):
            if comp[i] is not None:
                continue
            q = deque([i])
            comp[i] = cnt_comp
            while q:
                u = q[0]
                for v, t in g[u]:
                    if t == 3 and comp[v] is None:
                        comp[v] = cnt_comp
                        q.append(v)
                q.popleft()
            cnt_comp += 1
        # print(cnt_edges, cnt_comp)
        # print(cnt_edges[3] + cnt_comp - n)
        ans = (cnt_edges[3] - n + cnt_comp) + (cnt_edges[1] - cnt_comp + 1) + (cnt_edges[2] - cnt_comp + 1)
        return ans
