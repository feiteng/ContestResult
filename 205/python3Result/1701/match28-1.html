class Solution:
    def init(self,n):
        self.pre=[i for i in range(n)]
        self.rank=[0]*n
    def find(self,x):
        if self.pre[x]==x:
            return x
        self.pre[x]=self.find(self.pre[x])
        return self.pre[x]
    def is_same(self,x,y):
        return self.find(x)==self.find(y)
    def union(self,x,y):
        root1=self.find(x)
        root2=self.find(y)
        if root2==root1:
            return
        if self.rank[root1]>self.rank[root2]:
            self.pre[root2]=root1
        else:
            self.pre[root1]=root2
            if self.rank[root2]==self.rank[root1]:
                self.rank[root1]+=1
