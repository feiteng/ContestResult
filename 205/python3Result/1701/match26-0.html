class UF:
    def __init__(self, n):
        self.p = [i for i in range(n+1)]
    
    def find(self, a):
        if self.p[a] != a:
            self.p[a] = self.find(self.p[a])
        return self.p[a]
    
    def union(self, a, b):
        self.p[a] = self.p[b]


class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        uf_alice = UF(n)
        uf_bob = UF(n)
        delete_num = 0
        for label, u, v in edges:
            if label == 3:
                pu = uf_alice.find(u)
                pv = uf_alice.find(v)
                if pu == pv:
                    delete_num += 1
                else:
                    uf_alice.union(pu, pv)
                    uf_bob.union(pu, pv)
        for label, u, v in edges:
            if label == 1:
                pu = uf_alice.find(u)
                pv = uf_alice.find(v)
                if pu == pv:
                    delete_num += 1
                else:
                    uf_alice.union(pu, pv)
            elif label == 2:
                pu = uf_bob.find(u)
                pv = uf_bob.find(v)
                if pu == pv:
                    delete_num += 1
                else:
                    uf_bob.union(pu, pv)
        alice_ps = set([uf_alice.find(i) for i in range(1, n+1)])
        bob_ps = set([uf_bob.find(i) for i in range(1, n+1)])
        if len(alice_ps) != 1 or len(bob_ps) != 1:
            return -1
        return delete_num
