class UnionFind:
    def __init__(self, n):
        self.parents = [i for i in range(n)]
        self.size = [1] * n

    def find(self, i):
        while i != self.parents[i]:
            self.parents[i] = self.parents[self.parents[i]] # path halving
            i = self.parents[i]
        return i
    
    def union(self, a, b):        
        aPar = self.find(a)
        bPar = self.find(b)
        
        if aPar == bPar:
            return False
        
        # union by size
        if self.size[aPar] > self.size[bPar]:
            self.parents[bPar] = aPar
            self
.size[aPar] += self.size[bPar]
        else:
            self.parents[aPar] = bPar
            self.size[bPar] += self.size[aPar]
        
        return True

class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:        
        adj1 = defaultdict(set)
        adj2 = defaultdict(set)
        adj3 = defaultdict(set)
        
        for typ,a,b in edges:
            if typ == 1:
                adj1[a].add(b)
                adj1[b].add(a)
            elif typ == 2:
                adj2[a].add(b)
                adj2[b].add(a)
            else:
                adj3[a].add(b)
                adj3[b].add(a)
        
        type1 = UnionFind(n+1)
        type2 = UnionFind(n+1)
        res = 0
        
        for a in adj3:
            for b in adj3[a]:
                if type1.union(a,b):
                    type2.union(a,b)
                    res += 1
                    
        for a in adj2:
            for b in adj2[a]:
                if type2.union(a,b):
                    res += 1
                    
        for a in adj1:
            for b in adj1[a]:
                if type1.union(a,b):
                    res += 1
        
        # print(res)
        # print(type1.size)
        
        return len(edges) - res if (max(type1.size) == n and max(type2.size) == n) else -1
        
        """
        if there's a type 3, you just need that one
        if there's only one edge possible you have to take it
        if there's more than one edge possible:
            prefer type 3
        """
