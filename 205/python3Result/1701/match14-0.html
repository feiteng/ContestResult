class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        t = [i for i in range(n)]
        total = 0
        
        def find_root(val):
            if t[val] !=  val:
                t[val] = find_root(t[val])
            return t[val]

        def union_root(val_1,val_2):
            root_1,root_2 = find_root(val_1-1),find_root(val_2-1)  
            t[root_2] = root_1
    

        for i in edges:
            if i[0] == 3:
                root1,root2 = find_root(i[1]-1),find_root(i[2]-1) 
                if root1 == root2:
                    total += 1
                else:
                    union_root(i[1],i[2])
        
        
        q = list(t)
        
        for i in edges:
            if i[0] == 1:
                root1,root2 = find_root(i[1]-1),find_root(i[2]-1)  
                if root1 == root2:
                    total += 1
                else:
                    t[root1] = root2
        
        val_1 = find_root(0)
        for i in range(1,n):
            if find_root(i) != val_1:
                return -1
        
        t = q
    
        for i in edges:
            if i[0] == 2:
                root1,root2 = find_root(i[1]-1),find_root(i[2]-1)  
                if root1 == root2:
                    total += 1
                else:
                    t[root1] = root2
        
        val_1 = find_root(0)
        for i in range(1,n):
            if find_root(i) != val_1:
                return -1       
            
        return total
