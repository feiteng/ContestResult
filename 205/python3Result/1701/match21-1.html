class Solution:
    def maxNumEdgesToRemove(self, n: int, a: List[List[int]]) -> int:
        p1 = [i for i in range(n)]
        p2 = [i for i in range(n)]
        def getParent(parent, i):
            if i != parent[i]:
                parent[i] = getParent(parent, parent[i])
            return parent[i]
        a = [(-x[0], x[1] - 1, x[2] - 1) for x in a]
        a.sort()
        ans = len(a)
        for t, x1, x2 in a:
            def check(p, d):
                nonlocal ans
                f1 = getParent(p, x1)
                f2 = getParent(p, x2)
                if f1 != f2:
                    #print(x1 + 1, x2 + 1)
                    p[f1] = f2
                    p[f1] = f2
                    ans -= d
            if t == -3:
                check(p1, 1)
                check(p2, 0)
            elif t == -2:
                check(p2, 1)
            elif t == -1:
                check(p1, 1)
        m1 = {}
        m2 = {}
        for i in range(n):
            f1 = getParent(p1, i)
            f2 = getParent(p2, i)
            m1[f1] = 1
            m2[f2] = 1
        if len(m1) == 1 and len(m2) == 1:
            return ans
        else:
            return -1
