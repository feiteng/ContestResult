class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        def find(i,parent):
            if parent[i]!=i:
                parent[i]=find(parent[i],parent)
            return parent[i]
        
        def union(i,j,parent):
            i,j = find(i,parent),find(j,parent)
            if i!=j:
                parent[j] = i
                
                
        first,second,third = set(),set(),set()
        for ty,i,j in edges:
            if ty==1:
                first.add((i,j))
            elif ty==2:
                second.add((i,j))
            else:
                third.add((i,j))
        res = 0
        
        parenta = {i:i for i in range(1,n+1)}
        parentb = {i:i for i in range(1,n+1)}
    
                
        for i,j in third:
            if find(i,parenta)==find(j,parenta):
                res+=1
            else:
                union(i,j,parenta)
                union(i,j,parentb)
                
        for i,j in first:
            if find(i,parenta)==find(j,parenta):
                res+=1
            else:
                union(i,j,parenta)
        
        for i,j in second:
            if find(i,parentb)==find(j,parentb):
                res+=1
            else:
                union(i,j,parentb)
        
        parenta = {i:find(i,parenta) for i in parenta}
        parentb = {i:find(i,parentb) for i in parentb}
        
        if len(set(parenta.values()))!=1 or len(set(parentb.values()))!=1:
            return -1
        return res
