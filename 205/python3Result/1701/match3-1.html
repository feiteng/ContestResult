class UnionFind:
    def __init__(self):
        self.parents = dict()
    def findParent(self, node):
        if node not in self.parents:
            self.parents[node] = node
        elif self
.parents[node] != node:
            self.parents[node] = self.findParent(self.parents[node])
        return self.parents[node]
    def addPair(self, node1, node2):
        parent1 = self.findParent(node1)
        parent2 = self.findParent(node2)
        if parent1 != parent2:
            self.parents[parent1] = parent2
    def connected(self, node1, node2):
        return self.findParent(node1) == self.findParent(
node2)
            

class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        edges.sort(reverse=True)
        aliceUF = UnionFind()
        bobUF = UnionFind()
        numEdgesRemoved = 0
        for edge in edges:
            edgeType = edge[0]
            node1 = edge[1]
            node2 = edge[2]
            if edgeType == 1:
                if aliceUF.connected(node1, node2):
                    numEdgesRemoved += 1
                else:
                    aliceUF.addPair(node1, node2)
            elif edgeType == 2:
                if bobUF
.connected(node1, node2):
                    numEdgesRemoved += 1
                else:
                    bobUF.addPair(node1, node2)
            elif edgeType == 3:
                if aliceUF.connected(node1, node2) and bobUF.connected(node1, node2):
                    numEdgesRemoved += 1
                if not aliceUF.connected(node1, node2):
                    aliceUF.addPair(node1, node2)
                if not bobUF.connected(node1, node2):
                    bobUF.addPair(node1, node2)
        
        aliceParent = 0
        if 1 not in aliceUF.parents:
            return -1
        else:
            aliceParent = aliceUF.findParent(1)
        
        bobParent = 0
        if 1 not in bobUF.parents:
            return -1
        else:
            bobParent = bobUF.findParent(1)
        for i in range(2, n+1):
            if aliceUF.findParent(i) != aliceParent:
                return -1
            elif bobUF.findParent(i) != bobParent:
                return -1
        
        return numEdgesRemoved
