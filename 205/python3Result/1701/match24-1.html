from collections import defaultdict
import copy
class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
          
        parent = [i for i in range(n+1)]
            
        def find(i):
            if i != parent[i]:
                parent[i] = find(parent[i])
            return parent[i]
                
        def union(i, j):
            pi, pj = find(i), find(j)
            if pi != pj:
                parent[pi] = pj
                return 1
            return 0
        
        cnt3 = 0
        for t, i, j in edges:
            if t == 3:
                cnt3 += union(i, j)
        # print("parent3:", parent)
        parent3 = copy.deepcopy(parent)
        
        cnt1 = 0
        for t, i, j in edges:
            if t == 1:
                cnt1 += union(i, j)
        # print("parent1:", parent)
        p = find(1)
        for i in range(1, n+1):
            if find(i) != p:
                return -1
        
        
        parent = copy.deepcopy(parent3)
        cnt2 = 0
        for t, i, j in edges:
            if t == 2:
                cnt2 += union(i, j)
        # print("parent2:", parent)
        p = find(1)
        for i in range(1, n+1):
            if find(i) != p:
                return -1
        
        
        return len(edges) - (cnt1 + cnt2 + cnt3)
