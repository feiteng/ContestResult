class SpanTree:
    def __init__(self, n: int):
        self.n = n
        self
.p = [-1] * (n+1)
        self.disjoint_sets = n
        
    def find_root(self, node: int):
        if self.p[node] > 0:
            self.p[node] = self.find_root(self.p[node])
            return self.p[node]
        else:
            return node
    
    def union(self, a, b):
        pa = self.find_root(a)
        pb = self.find_root(b)
        if pa != pb:
            self.p[pb] = pa
            self.disjoint_sets -=
 1
        
class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        alice = SpanTree(n)
        bob = SpanTree(n)
        res = 0
        
        for e in sorted(edges)[::-1]:
            if e[0] == 1:
                if alice.find_root(e[1]) != alice.find_root(e[2]):
                    alice
.union(e[1], e[2])
                    res += 1
            elif e[0] == 2:
                if bob.find_root(e[1]) != bob.find_root(e[2]):
                    bob.union(e[1], e[2])
                    res += 1
            elif e[0] == 3:
                if (alice.find_root(e[1]) != alice.find_root(e[2])) or (bob.find_root(e[1]) != bob.find_root(e[2])):
                    alice.union(e[1], e[2])
                    bob.union(e[1], e[2])
                    res += 1
        if alice.disjoint_sets > 1 or bob.disjoint_sets > 1:
            return -1
        return len(edges) - res
