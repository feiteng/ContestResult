class UF:
    def __init__(self, n):
        self.cache = list(range(
n + 1))
        
    def find(self, i):
        if self.cache[i] != i:
            self.cache[i] = self.find(self.cache[i])
        return self.cache[i]
        
    def union(self, i, j):
        head1 = self.find(i)
        head2 = self.find(j)
        self.cache[head1] = head2

class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        result = 0
        a = UF(n)
        b = UF(n)
        for t, i, j in edges:
            
if t == 3:
                if a.find(i) == a.find(j):
                    result += 1
                else:
                    a.union(i, j)
                    b.union(i, j)
                    
        for t, i, j in edges:
            if t == 1:
                
if a.find(i) == a.find(j):
                    result += 1
                else:
                    a.union(i, j)
            elif t == 2:
                if b.find(i) == b.find(j):
                    result += 1
                else:
                    b.union(i, j)
                    
        if len(set(a.find(i) for i in range(1, n + 1))) == 1 and len(set(b.find(i) for i in range(1, n + 1))) == 1:
            return result
        else:
            return -1
