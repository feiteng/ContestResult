class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:

        def find(x, p):
            if p[x] != x:
                p[x] = find(p[x], p)
            return p[x]
        
        def ks(es):
            es.sort()
            cnt = 0
            p = [i for i in range(1+n)]
            for _, a, b, idx in es:
                pa, pb = find(a, p), find(b, p)
                if pa != pb:
                    p[pa] = p[pb]
                    cnt += 1
                    visited[idx] = 1
            if cnt < n - 1: return False
            return True
        
        e1, e2 = [], []
        for i, (t, a, b) in enumerate(edges):
            if t == 1:
                e1.append([1, a, b, i])
            elif t == 2:
                e2.append([1, a, b, i])
            else:
                e1.append([0, a, b, i])
                e2.append([0, a, b, i])

        visited = [0 for _ in range(len(edges))]
        if not ks(e1) or not ks(e2): return -1
        return len(edges) - sum(visited)
