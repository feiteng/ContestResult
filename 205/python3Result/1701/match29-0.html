class Solution:
    def canReach(self, u, E, ban, vis):
        vis.add(u)
        for tp, v in E[u]:
            if tp != ban and v not in vis:
                self.canReach(v, E, ban, vis)
                
    def dfs(self, u, E, vis):
        vis.add(u)
        for tp, v in E[u]:
            if tp == 3 and v not in vis:
                self.dfs(v, E, vis)
        
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        E = defaultdict(list)
        for tp, u, v, in edges:
            E[u].append((tp, v))
            E[v].append((tp, u))
        
        for ban in [1, 2]:
            vis = set()
            self.canReach(1, E, ban, vis)
            if len(vis) != n:
                return -1
        
        ans = 0
        comp = 0
        seen = set()
        for u in range(1, n + 1):
            if not u in seen:
                vis = set()
                self.dfs(u, E, vis)
                seen |= vis
                ans += len(vis) - 1
                comp += 1
        return len(edges) - ans - 2 * (comp - 1)
