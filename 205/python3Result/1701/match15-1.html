from typing import List
import collections


class Union:
    def __init__(self, n):
        self.n = n
        self.p = [i for i in range(n)]
        self
.cnt = n

    def root(self, i):
        if self.p[i] != i:
            self.p[i] = self.root(self.p[i])
        return self.p[i]

    def merge(self, i, j):
        if self.root(i) == self.root(j):
            return

        self.cnt -= 1
        self
.p[self.root(i)] = self.root(j)


class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        e = {1: [], 2: [], 3: []}
        for t, i, j in edges:
            e[t].append((i - 1, j - 1))

        u1 = Union(n)
        u2 = Union(n)
        ecnt = 0
        for i, j in e[3]:
            if u1.root(i) != u1.root(j):
                ecnt += 1
                u1.merge(i, j)
                u2.merge(i, j)

        for i, j in e[1]:
            if u1.root(i) != u1.root(j):
                ecnt += 1
                u1.merge(i, j)
        for i, j in e[2]:
            if u2.root(i) != u2.root(j):
                ecnt += 1
                u2.merge(i, j)

        if u1.cnt != 1 or u2.cnt != 1:
            return -1
        else:
            return len(edges) - ecnt
