class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        tot = len(edges)
        if tot < n - 1:
            return -1
        nei = [[] for _ in range(n + 1)]
        for t, u, v in edges:
            nei[u].append([-t, v])
            nei[v].append([-t, u])
        h = [[-3, 1]]        
        seen = [set() for _ in range(3)]
        e = -1
        while h:
            t, u = heappop(h)
            if t == -3:
                if u not in seen[1] or u not in seen[2]:
                    seen[1].add(u)
                    seen[2].add(u)
                    e += 1
                    for nt, v in nei[u]:
                        heappush(h, [nt, v])
            elif t == -1:
                if u not in seen[1]:
                    seen[1].add(u)
                    e += 1
                    for nt, v in nei[u]:
                        heappush(h, [nt, v])
            else:
                if u not in seen[2]:
                    seen[2].add(u)
                    e += 1
                    for nt, v in nei[u]:
                        heappush(h, [nt, v])
        return tot - e if len(seen[1]) == n and len(seen[2]) == n else -1
