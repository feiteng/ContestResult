class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        te = [[],[],[]]
        for edge in edges:
            te[edge[0]-1].append((edge[1]-1,edge[2]-1))
        compBase = list(range(n))
        nComp = n
        ans = 0
        
for i,j in te[2]:
            comp1 = self.find(compBase,i)
            comp2 = self.find(compBase,j)
            if comp1 == comp2:
                ans += 1
            else:
                self.union(compBase,comp1,comp2)
                nComp -= 1
        for k in range(2):
            nnComp = nComp
            comp = compBase.copy
()
            for i,j in te[k]:
                comp1 = self.find(comp,i)
                comp2 = self.find(comp,j)
                if comp1 == comp2:
                    ans += 1
                else:
                    self.union(comp,comp1,comp2)
                    nnComp -= 1
            if nnComp != 1:
                return -1
        return ans
    
    def find(self,arr,i):
        idx = []
        while arr[i] != i:
            idx.append(i)
            i = arr[i]
        for j in idx:
            arr[j] = i
        return i
    
    def union(self,arr,i,j):
        arr[j] = i
