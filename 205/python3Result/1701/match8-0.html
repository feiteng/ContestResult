class Solution:
    def getroot(self, r, p):
        q = r[p]
        while p != r[p]:
            p = r[p]
        while q != p:
            tmp = r[q]
            r[q] = p
            q = tmp
        return p
    
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        ra = list(range(n))
        e = [[], [], []]
        for ei, u, v in edges:
            e[ei - 1].append((u - 1, v - 1))
        ans = 0
        for u, v in e[2]:
            u, v = self.getroot(ra, u), self.getroot(ra, v)
            if u == v:
                ans += 1
            else:
                ra[u] = v
        rb = [ra[i] for i in range(n)]
        for el, r in ((e[0], ra), (e[1], rb)):
            for u, v in el:
                u, v = self.getroot(r, u), self.getroot(r, v)
                if u == v:
                    ans += 1
                else:
                    r[u] = v
            r0 = self.getroot(r, 0)
            for i in range(1, n):
                if self.getroot(r, i) != r0:
                    return -1
        return ans
