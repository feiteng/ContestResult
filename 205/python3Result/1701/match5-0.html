class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        edges.sort(reverse=True)
        Alice_father={}
        Bob_father={}
        for i in range(1,n+1):
            Alice_father[i]=i
            Bob_father[i]=i
        Aliceunion_times=[0]
        Bobunion_times=[0]
        used_edges=0
        for i in range(len(edges)):
            t,n1,n2=edges[i][0],edges[i][1],edges[i][2]
            if t==3:
                if self.find(n1,Alice_father)!=self.find(n2,Alice_father)\
                or self.find(n1,Bob_father)!=self.find(n2,Bob_father):
                    used_edges+=1
                    self.union(n1,n2,Alice_father,Aliceunion_times)
                    self.union(n1,n2,Bob_father,Bobunion_times)
            elif t==2:
                if self.find(n1,Bob_father)!=self.find(n2,Bob_father):
                    used_edges+=1
                    self.union(n1,n2,Bob_father,Bobunion_times)
            elif t==1:
                if self.find(n1,Alice_father)!=self.find(n2,Alice_father):
                    used_edges+=1
                    self.union(n1,n2,Alice_father,Aliceunion_times)
                    
        if Bobunion_times[0]!=n-1 or Aliceunion_times[0]!=n-1:
            return -1
        return len(edges)-used_edges
                    
                    
            
                    
                    
                    
        
        
    
    
    def find(self,a,node_father):
        tmp_a=a
        while node_father[a]!=a:
            a=node_father[a]
        while tmp_a!=a:
            tmp_father=node_father[tmp_a]
            node_father[tmp_a]=a
            tmp_a=tmp_father
        return a
    
    
    def union(self,a,b,node_father,merge_times):
        a_father,b_father=self.find(a,node_father), self.find(b,node_father)
        if a_father!=b_father:
            merge_times[0]+=1
            node_father[a_father]=b_father
