class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        a, b = collections.defaultdict(list), collections.defaultdict(list)
        t1, t2, t3 = [], [], []
        
        for t, u, v in edges:
            if t == 1:
                a[u].append((v, 0))
                a[v].append((u, 0))
                t1.append((u, v))
            elif t == 2:
                b[u].append((v, 0))
                b[v].append((u, 0))
                t2.append((u, v))
            else:
                a[u].append((v, 1))
                a[v].append((u, 1))
                b[u].append((v, 1))
                b[v].append((u, 1))
                t3.append((u, v))
        
        def dfs(i, g, seen):
            seen.add(i)
            for j, _ in g[i]:
                if j not in seen:
                    dfs(j, g, seen)
        
        aseen, bseen = set(), set()
        dfs(1, a, aseen)
        dfs(1, b, bseen)
        if len(aseen) < n or len(bseen) < n:
            return -1
        
        p1, p2 = list(range(n + 1)), list(range(n + 1))
        s1, s2 = [1] * (n + 1), [1] * (n + 1)
        inc = 0
        
        def find(parent, i): 
            if parent[i] != i: 
                parent[i] = find(parent, parent[i])
            return parent[i]

        def union(parent, size, x, y): 
            i = find(parent, x) 
            j = find(parent, y)
            if size[i] < size[j]: 
                parent[i] = j
                size[j] += size[i]
            else: 
                parent[j] = i 
                size[i] += size[j]
        
        for u, v in t3:
            i, j = find(p1, u), find(p1, v)
            if i == j: continue
            union(p1, s1, i, j)
            inc += 1
            # union(p2, s2, i, j)
        p2, s2 = p1[:], s1[:]
        
        
        for u, v in t1:
            i, j = find(p1, u), find(p1, v)
            if i == j: continue
            union(p1, s1, i, j)
            inc += 1
        
        for u, v in t2:
            i, j = find(p2, u), find(p2, v)
            if i == j: continue
            union(p2, s2, i, j)
            inc += 1
        
        return len(edges) - inc
