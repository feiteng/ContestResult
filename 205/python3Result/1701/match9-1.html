class UnionFindSet:
    def __init__(self, n):
        self.p = [i for i in range(n)]
        self
.rk = [1] * n
    
    def find(self, v):
        if self.p[v] != v: self.p[v] = self.find(self.p[v])
        return self.p[v]
    
    def union(self, v1, v2):
        p1 = self.find(v1)
        p2 = self.find(v2)
        if p1 == p2: return True
        
if self.rk[p1] < self.rk[p2]: p1, p2 = p2, p1
        self.p[p2] = p1
        self.rk[p1] += self.rk[p2]
        return False
    
    def count(self, u):
        return self.rk[self.find(u)]

class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        ret = 0
        ufs1, ufs2 = UnionFindSet(n + 1), UnionFindSet(n + 1)
        for t, u, v in edges:
            if t == 3:
                f = False
                if not ufs1.union(u, v): f = True
                if not ufs2.union(u, v): f = True
                if f: ret += 1
        for t, u, v in edges:
            if t == 1:
                if not ufs1.union(u, v): ret += 1
            if t == 2:
                if not ufs2.union(u, v): ret += 1
        if ufs1.count(1) != n or ufs2.count(1) != n: return -1
        return len(edges) - ret
