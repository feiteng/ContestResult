class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        n += 1
        p = [*range(n)]
        rank = [0] * n
        def find(x):
            if p[x] != x:
                p[x] = find(p[x])
            return p[x]
        def union(x, y):
            xp, yp = find(x), find(y)
            if xp != yp:
                if rank[xp] > rank[yp]:
                    p[yp] = xp
                else:
                    p[xp] = yp
                    if rank[xp] == rank[yp]:
                        rank[yp] += 1

        alice = []
        bob = []
        used = 0
        for e in edges:
            if e[0] == 3:
                if find(e[1]) != find(e[2]):
                    used += 1
                    union(e[1], e[2])
            elif e[0] == 1:
                alice.append(e)
            else:
                bob.append(e)

        ori_p, ori_rank = [find(x) for x in p], rank[:]
        # print(p, alice, bob)
        for _, x, y in alice:
            if find(x) != find(y):
                used += 1
                union(x, y)
        # print(p)
        if len(set([find(x) for x in p])) != 2:
            return -1
        
        p, rank = ori_p, ori_rank
        for _, x, y in bob:
            if find(x) != find(y):
                used += 1
                union(x, y)
        # print(p)
        if len(set([find(x) for x in p])) != 2:
            return -1
        
        return len(edges) - used
