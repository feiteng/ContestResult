class Solution:
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        fa = {}
        sz = {}
        def find(x):
            nonlocal fa, sz
            if x not in fa:
                fa[x] = x
                sz[x] = 1
            if fa[x] == x:
                return x
            fa[x] = find(fa[x])
            return fa[x]
        def union(x, y):
            nonlocal fa
            x = find(x)
            y = find(y)
            fa[x] = y
            sz[y] += sz[x]

        ans = 0
        for t, a, b in edges:
            if t == 3:
                pa = find(a)
                pb = find(b)
                if pa == pb:
                    ans += 1
                else:
                    union(a, b)
        tfa = dict(fa)
        tsz = dict(sz)
        for t, a, b in edges:
            if t == 1:
                pa = find(a)
                pb = find(b)
                if pa == pb:
                    ans += 1
                else:
                    union(a, b)
        if sz[find(1)] != n:
            return -1
        fa = tfa
        sz = tsz
        for t, a, b in edges:
            if t == 2:
                pa = find(a)
                pb = find(b)
                if pa == pb:
                    ans += 1
                else:
                    union(a, b)
        if sz[find(1)] != n:
            return -1
        return ans
