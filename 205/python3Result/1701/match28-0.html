class DisjointSetNode:
    def __init__(self, x):
        self.val = x
        self.parent = self
        self.rank = 0

class DisjointSet:
    def __init__(self, n):
        self.nodeList = []
        for i in range(n):
            self.nodeList.append(DisjointSetNode(i))

    def Union(self, x, y):
        self.Link(self.FindSet(self.nodeList[x]), self.FindSet(self.nodeList[y]))

    def Link(self, x, y):
        if x.rank > y.rank:
            y.parent = x
        else:
            x.parent = y
            if x.rank == y.rank:
                y.rank += 1

    def FindSet(self, x):
        if x != x.parent:
            x.parent = self.FindSet(x.parent)
        return x.parent

    def inSameSet(self, x, y):
        return self.FindSet(self.nodeList[x]) == self.FindSet(self.nodeList[y])
    
class Solution:
    def dfs(self, i):
        self.visit[i] = True
        for next in self.E[i]:
            if not self.visit[next]:
                self.dfs(next)
                
    def maxNumEdgesToRemove(self, n: int, edges: List[List[int]]) -> int:
        self.visit = [False] * n
        self.E = [[] for _ in range(n)]
        for e in edges:
            if e[0] == 1 or e[0] == 3:
                self.E[e[1] - 1].append(e[2] - 1)
                self.E[e[2] - 1].append(e[1] - 1)
        self.dfs(0)
        for i in range(n):
            if not self.visit[i]:
                return -1
            self.visit[i] = False
        self.E = [[] for _ in range(n)]
        for e in edges:
            if e[0] == 2 or e[0] == 3:
                self.E[e[1] - 1].append(e[2] - 1)
                self.E[e[2] - 1].append(e[1] - 1)
        self.dfs(0)
        for i in range(n):
            if not self.visit[i]:
                return -1
            
        ds = DisjointSet(n)
        res = 0
        for e in edges:
            if e[0] == 3:
                if ds.inSameSet(e[1] - 1, e[2] - 1):
                    res += 1
                else:
                    ds.Union(e[1] - 1, e[2] - 1)
        for e in edges:
            if e[0] == 1:
                if ds.inSameSet(e[1] - 1, e[2] - 1):
                    res += 1
                else:
                    ds.Union(e[1] - 1, e[2] - 1)
        ds = DisjointSet(n)
        for e in edges:
            if e[0] == 3:
                if not ds.inSameSet(e[1] - 1, e[2] - 1):
                    ds.Union(e[1] - 1, e[2] - 1)
        for e in edges:
            if e[0] == 2:
                if ds.inSameSet(e[1] - 1, e[2] - 1):
                    res += 1
                else:
                    ds.Union(e[1] - 1, e[2] - 1)
        return res
