class Solution:
    def modifyString(self, s: str) -> str:
        l = len(s)
        s = ' ' + s + ' '
        ans = [' ']
        for i, c in enumerate(s[1:-1], 1):
            # print(i, c)
            if c != '?':
                ans.append(c)
                continue
            pre, nxt = ans[i - 1], s[i + 1]
            for j in range(26):
                c1 = chr(j + 97)
                if c1 != pre and c1 != nxt:
                    ans.append(c1)
                    break
        # print(ans)
        return ''.join(ans[1:])
