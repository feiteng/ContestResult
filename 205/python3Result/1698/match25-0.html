class Solution:
    def modifyString(self, s: str) -> str:
        s = [ch for ch in s]
        for i, ch in enumerate(s):
            if ch == "?":
                unique = []
                if i + 1 < len(s) and s[i + 1] != "?":
                    unique.append(s[i + 1])
            
                if i - 1 >= 0 and s[i - 1] != "?":
                    unique.append(s[i - 1])
            
                for x in ["a", "b", "c", "d"]:
                    if not x in unique:
                        s[i] = x
                        break

        return "".join(s)
