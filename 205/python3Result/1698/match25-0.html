class Solution:
    def modifyString(self, s: str) -> str:
        
        n = len(s)
        res = ''
        for i, c in enumerate(s):
            if c == '?':
                for new in 'abc':
                    if i > 0 and new == res[-1]:
                        continue
                    if i < n - 1 and new == s[i+1]:
                        continue
                    res += new
                    break
            else:
                res += c
        return res
