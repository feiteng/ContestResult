class Solution:
    def modifyString(self, s: str) -> str:
        ans = []
        for i in range(len(s)):
            if s[i] == '?':
                c = ord('a')
                visited = set()
                if i - 1 >= 0:
                    visited.add(ans[-1])
                if i + 1 < len(s) and s[i+1] != '?':
                    visited.add(s[i+1])
                for _ in range(26):
                    if chr(c) not in visited:
                        ans.append(chr(c))
                        break
                    else:
                        c += 1
            else:
                ans.append(s[i])
        return ''.join(ans)
