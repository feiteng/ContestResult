class Solution:
    def modifyString(self, s: str) -> str:
        def other(a, b):
            if a != "a" and b != "a":
                return "a"
            if a != "b" and b != "b":
                return "b"
            return "c"
        s = list(s)
        for i in range(len(s)):
            if s[i] == "?":
                s[i] = other(s[i - 1], s[i + 1] if i + 1 != len(s) else None)
        return "".join(s)
