class Solution:
    def modifyString(self, s: str) -> str:
        ret = list(s)
        n = len(s)
        def pick(i,c):
            for x in 'abcd':
                if i > 0 and ret[i-1] == x:
                    continue
                if i < n-1 and ret[i+1] == x:
                    continue
                return x
        for i, c in enumerate(s):
            if c == '?':
                ret[i] = pick(i, c)

        return ''.join(ret)
