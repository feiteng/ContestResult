class Solution:
    def modifyString(self, s: str) -> str:
        ans = []
        candidates = [chr(x) for x in range(ord('a'), ord('z')+1)]
        for idx, letter in enumerate(s):
            if letter != '?': 
                ans.append(letter)
            else:
                conflict = None
                if idx + 1 < len(s): conflict = s[idx+1]
                for candidate in candidates:
                    if candidate != conflict and (not ans or ans[-1] != candidate):
                        ans.append(candidate)
                        break
        return ''.join(ans)
