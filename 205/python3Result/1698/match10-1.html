class Solution:
    def modifyString(self, s: str) -> str:
        ans = []
        for i, c in enumerate(s):
            if s[i] == "?":
                ss = set()
                if i > 0: ss.add(ans[i-1])
                if i + 1 < len(s) and s[i+1] != "?": ss.add(s[i+1])
                for j in range(26):
                    if chr(j+97) not in ss:
                        ans.append(chr(j+97))
                        break
            else:
                ans.append(c)
        return "".join(ans)
