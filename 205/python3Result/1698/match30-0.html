class Solution:
    def modifyString(self, s: str) -> str:
        arr = list(s)
        n = len(s)
        for i, ch in enumerate(arr):
            if ch == '?':
                for candi in range(26):
                    candi_ch = chr(candi + 97)
                    if i - 1 >= 0 and arr[i - 1] == candi_ch:
                        continue
                    if i + 1 < n and arr[i + 1] == candi_ch:
                        continue
                    arr[i] = candi_ch
                    break
        return ''.join(arr)
