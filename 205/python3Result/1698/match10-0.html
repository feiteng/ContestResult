class Solution:
    def modifyString(self, s: str) -> str:
        s = list(s)
        for i, c in enumerate(s):
            if c == "?": 
                for cc in "abc": 
                    if (i == 0 or s[i-1] != cc) and (i+1 == len(s) or s[i+1] != cc): 
                        s[i] = cc
                        break 
        return "".join(s)
