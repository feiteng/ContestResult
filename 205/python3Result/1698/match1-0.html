class Solution:
    def modifyString(self, s: str) -> str:
        new_symbols = []
        prev_char = None
        for i, char in enumerate(s):
            next_char = s[i + 1] if i + 1 < len(s) else None
            if char == '?':
                for replacement in ['a', 'b', 'c']:
                    if replacement not in [prev_char, next_char]:
                        new_symbols.append(replacement)
                        break
            else:
                new_symbols.append(char)
            prev_char = new_symbols[-1]
        return "".join(new_symbols)
