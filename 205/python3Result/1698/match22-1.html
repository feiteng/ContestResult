import string
class Solution:
    def modifyString(self, s: str) -> str:
        l = list("?" + s + "?")
        n = len(l)
        for i in range(1,n-1):
            if l[i] == '?':
                
for c in string.ascii_lowercase:
                    if c != l[i-1] and c != l[i+1]:
                        l[i] = c
                        break
        return "".join(l)[1:n-1]
