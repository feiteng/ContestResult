class Solution:
    def modifyString(self, s: str) -> str:
        alphas = ['a', 'b', 'c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
        ans = []
        for i in range(len(s)):
            if s[i] == '?':
                for j in range(len(alphas)):
                    flag=True
                    if i>=1 and alphas[j] == ans[i-1]:
                        flag=False
                        
                    if i<len(s)-1 and alphas[j] == s[i+1]:
                        flag=False    
                    if flag:
                        ans.append(alphas[j])
                        break    
            else:
                ans.append(s[i])        
        return "".join(ans)
