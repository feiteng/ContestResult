class Solution:
    def modifyString(self, s: str) -> str:
        if len(s) == 1:
            if s[0] == '?':
                return 'a'
            return s
        s = list(s)
        for i in range(1, len
(s)-1):
            if s[i] == '?':
                if s[i-1] != 'a' and s[i+1] != 'a':
                    s[i] = 'a'
                elif s[i-1] != 'b' and s[i+1] != 'b':
                    s[i] = 'b'
                else:
                    s[i] = 'c'
        if s[-1] == '?':
            if s[-2] == 'a':
                s[-1] = 'b'
            else:
                s[-1] = 'a'
        
        if s[0] == '?':
            if s[1] == 'a':
                s[0] = 'b'
            else:
                s[0] = 'a'
        return "".join(s)
