class Solution:
    def modifyString(self, s: str) -> str:
        ans = []
        for i,ch in enumerate(s):
            if ch != '?':
                ans.append(ch)
                continue
            for j in range(26):
                ch = chr(j+ord('a'))
                if len(ans)>0 and ans[-1] == ch:
                    continue
                if i<len(s)-1 and s[i+1] == ch:
                    continue
                ans.append(ch)
                break
        return "".join(ans)
