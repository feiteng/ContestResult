class Solution:
    def modifyString(self, s: str) -> str:
        n = len(s)
        def find_not(vec):
            for a in string.ascii_lowercase:
                if a not in vec:
                    return a
            return '*'
        answer = ''
        for i in range(n):
            c = s[i]
            if c != '?':
                answer += c
                continue
            vec = []
            if i > 0:
                vec.append(answer[-1])
            if i < n - 1 and s[i+1] != '?':
                vec.append(s[i+1])
            answer += find_not(vec)
        return answer
