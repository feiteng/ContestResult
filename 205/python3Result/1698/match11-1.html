class Solution:
    def modifyString(self, s: str) -> str:
        t = s + '*'
        n = len(t)
        lst = '*'
        ans = ''
        alph = 'abcdefghijklmnopqrstuvwxyz'
        
        for i in range(n-1):
            if(t[i]=='?'):
                for p in range(26):
                    c = alph[p]
                    if(c == lst or c == t[i+1]):
                        continue
                    ans += c
                    lst = c
                    break
            else:
                ans += t[i]
                lst = t[i]
        return ans
