class Solution:
    def modifyString(self, s: str) -> str:
        s = list(s)
        for i in range(len(s)):
            if s[i] == "?":
                for abc in "abcdefghijklmnopqrstuvwxyz":
                    if (i-1 < 0 or s[i-1] != abc) and (i+1 >= len(s) or s[i+1] != abc):
                        s[i] = abc
                        break
        return "".join(s)
