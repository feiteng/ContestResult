class Solution:
    def modifyString(self, s: str) -> str:
        L = list(s)
        for i in range(len(L)):
            if L[i] != '?':
                continue
            for j in range(26):
                c = chr(j + ord('a'))
                if (i>0 and c == L[i-1]) or (i < len(s)-1 and c == L[i+1]):
                    continue
                else:
                    L[i] = c
                    break;
        return ''.join(L)
