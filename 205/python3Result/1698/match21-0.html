import random

class Solution:
    def ok(self, s):
        for i in range(len(s) - 1):
            if s[i] == s[i+1]:
                return False
        return True
    
    def rr(self, s):
        s = s[:]
        for i in range(len(s)):
            if s[i] == '?':
                s[i] = random.choice("abcdefghijklmnopqrstuvwxyz")
        return s
        
    def modifyString(self, s: str) -> str:
        s = list(s)
        
        for _ in range(1000):
            r = self.rr(s)
            if self.ok(r):
                return ''.join(r)

        assert(False)
