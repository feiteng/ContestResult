class Solution:
    def modifyString(self, s: str) -> str:
        n = len(s)
        for i in range(n):
            if s[i] == '?':
                if i > 0:
                    slast = s[i - 1]
                else:
                    slast = ''
                if i + 1 < n and s[i + 1] != '?':
                    snext = s[i + 1]
                else:
                    snext = 0
                for j in range(ord('a'), ord('z') + 1):
                    if chr(j) != snext and chr(j) != slast:
                        s = s[:i] + chr(j) + s[i + 1:]
                        break
        return s
