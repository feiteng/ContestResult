class Solution:
    def modifyString(self, s: str) -> str:
        s = "?"+s+"?"
        ret = "?"
        for i in range(1, len(s) - 1):
            if s[i] != "?":
                ret += s[i]
            else:
                for k in range(26):
                    r = chr(k + 97)
                    
if r != ret[-1] and r != s[i + 1]:
                        ret+=r
                        break
        return ret[1:]
