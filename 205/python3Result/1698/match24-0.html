class Solution:
    def modifyString(self, s: str) -> str:
        res = ''
        n = len(s)
        for i, c in enumerate(s):
            if c == '?':
                for j in range(26):
                    b = chr(ord('a') + j)
                    if i > 0 and res[i-1] == b: continue
                    if i < n - 1 and s[i + 1] == b: continue
                    break
                c = b
            res += c
        return res
