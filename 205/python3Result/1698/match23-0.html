class Solution:
    def modifyString(self, s: str) -> str:
        if len(s)==1:
            return "a" if s=="?" else s
        ans = [i for i in s]

        def nxt(c):
            for i in "abcd":
                if i not in c:
                    return i

        for i in range(len(ans)):
            if ans[i]=="?":
                if i+1<len(ans) and ans[i+1]=="?":
                    if i!=0:
                        ans[i] = nxt([ans[i-1]])
                    else:
                        ans[i]='z'
                if i==0:
                    ans[i] = nxt([ans[i+1]])
                elif i==len(ans)-1:
                    ans[i] = nxt([ans[i-1]])
                else:
                    ans[i] = nxt([ans[i-1],ans[i+1]])
        return "".join(ans)
