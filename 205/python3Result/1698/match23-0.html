class Solution:
    def modifyString(self, s: str) -> str:
        charArr = list(s)
        n = len(s)

        for i, c in enumerate(charArr):
            if c == '?':
                for tmp in string.ascii_lowercase:
                    if i >= 0 and tmp == charArr[i - 1]: continue
                    if i < n - 1 and tmp == charArr[i + 1]: continue

                    charArr[i] = tmp
                    break

        return ''.join(charArr)
