class Solution:
    def modifyString(self, s: str) -> str:
        res = ['#']
        import string
        s = s + '#'
        for i in range(len(s)-1):
            ch = s[i]
            if ch != '?':
                res.append(ch)
            else:
                for c in string.ascii_lowercase:
                    if c != res[-1] and c != s[i+1]:
                        res.append(c)
                        break
        return ''.join(res[1:])
