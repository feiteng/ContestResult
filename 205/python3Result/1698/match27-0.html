class Solution:
    def modifyString(self, s: str) -> str:
        k = list(s)
        n = len(k)
        for i in range(n):
            if k[i] == '?':
                v = 97
                while True:
                    k[i] = chr(v)
                    if (i == 0 or k[i - 1] != k[i]) and (i == n - 1 or k[i + 1] != k[i]):
                        break
                    v += 1
        return ''.join(k)
