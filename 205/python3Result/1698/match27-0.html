class Solution:
    def modifyString(self, s: str) -> str:
        s = '#' + s + '#'
        
        res = ''
        for i, x in enumerate(s):
            if x != '?':
                res += x
            else:
                for c in 'abcdefg':
                    if c != res[i-1] and c!= s[i+1]:
                        res += c
                        break
        return res[1:-1]
