class Solution:
    def modifyString(self, s: str) -> str:
        alphabet = "qwertyuiopasdfghjklzxcvbnm"
        s = list(s)
        
for i in range(len(s)):
            if s[i] == "?":
                for c in alphabet:
                    if i-1 >= 0 and s[i-1] == c:
                        continue 
                    if i+1 < len(s) and s[i+1] == c:
                        continue 
                    s[i] = c
                    break
        return "".join(s)
