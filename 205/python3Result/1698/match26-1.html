class Solution:
    def modifyString(self, s: str) -> str:
        a = list(s)
        n = len(a)
        for i in range(n):
            if a[i] == '?':
                p1 = a[i - 1] if i - 1 >= 0 else 'z'
                p2 = a[i + 1] if i + 1 < n else 'z'
                if p1 != 'a'  and p2 != 'a':
                    a[i] = 'a'
                elif p1 != 'b' and p2 != 'b':
                    a[i] = 'b'
                else:
                    a[i] = 'c'
        return ''.join(a)
