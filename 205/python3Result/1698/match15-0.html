class Solution:
    def modifyString(self, s: str) -> str:
        N = len(s)
        res = list(s)
        for i in range(N):
            if res[i] == '?':
                for j in range(26):
                    ch = chr(ord('a') + j)
                    if (i == 0 or res[i-1] != ch) and (i >= N-1 or res[i+1] != ch):
                        res[i] = ch
                        break
        return ''.join(res)
