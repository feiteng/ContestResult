class Solution:
    def modifyString(self, s: str) -> str:
        n=len(s)
        s=["@"]+list(s)+["@"]
        for i in range(1,n+1):
            if s[i]=="?":
                if s[i-1]!="a" and s[i+1]!="a":
                    s[i]="a"
                elif s[i-1] != "b" and s[i+1] != "b":
                    s[i] = "b"
                else:
                    s[i] = "c"
        return "".join(s[1:n+1])
