class Solution:
    def modifyString(self, s: str) -> str:
        out = ""
        s = list("a" + s + "a")
        for i in range(1,len(s)-1):
            a, b, c = s[i - 1], s[i], s[i + 1]
            if b == "?":
                for cc in "abcdefghijklm":
                    if cc not in (a,c):
                        out += cc
                        s[i] = cc
                        break
            else:
                out += b
        return out
