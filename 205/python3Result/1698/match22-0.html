class Solution:
    def modifyString(self, s: str) -> str:
        
        s = '#' + s + '#'
        
        arr = list(s)
        
        for i in range(1, len(s) - 1):
            
            vi = arr[i]
            
            if vi == '?':
                
                for j in ('a', 'b', 'c'):
                    if j != arr[i - 1] and j != arr[i + 1]:
                        arr[i] = j
                        break
                        
        return "".join(arr[1:-1])
