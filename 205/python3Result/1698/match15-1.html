from string import ascii_lowercase as alphas
class Solution:
    def modifyString(self, s: str) -> str:
        s = "$"+s+"$"
        arr = [[s[0], 1]]
        
        def getProperString(prev, last, l):
            res = ""
            while l > 0:
               
                if l == 1:
                    for char in alphas:
                        if char != prev and char != last:
                            res+=char
                            break
                else:
                    for char in alphas:
                        if char != prev:
                            res+=char
                            prev = char
                            break
                l-=1
            return res
                    
        
        for i in range(1, len(s)):
            if s[i] == arr[-1][0]:
                arr[-1][1]+=1
            else:
                arr.append([s[i], 1])
        # print(arr)
        res = ""
        for i in range(1, len(arr)-1):
            if arr[i][0] != "?":
                res+=arr[i][0]
            else:
                res+=getProperString(arr[i-1][0], arr[i+1][0], arr[i][1])
        return res
