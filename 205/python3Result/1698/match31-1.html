class Solution:
    def modifyString(self, s: str) -> str:
        b = list(s)
        n = len(s)
        for i, j in enumerate(b):
            if j == '?':
                for k in range(26):
                    c = chr(ord
('a') + k)
                    if i > 0 and b[i - 1] == c:
                        continue
                    if i < n - 1 and b[i + 1] == c:
                        continue
                    b[i] = c
                    break
        return ''.join(b)
