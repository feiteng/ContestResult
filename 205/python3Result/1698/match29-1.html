class Solution:
    def modifyString(self, s: str) -> str:
        stk = []
        for i, ch in enumerate(s):
            if ch != '?':
                stk.append(ch)
            else:
                last = '' if not stk else stk[-1]
                back = '' if i + 1 >= len(s) else s[i + 1]
                for i in range(26):
                    cand = chr(ord('a') + i)
                    if cand != last and cand != back:
                        stk.append(cand)
                        break
        return ''.join(stk)
