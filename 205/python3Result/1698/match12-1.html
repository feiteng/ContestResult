class Solution:
    def modifyString(self, s: str) -> str:
        a=['A']
        s+='A'
        for i in range(len(s)-1):
            if s[i]=='?':
                if a[-1]=='a':
                    if s[i+1]=='b': a.append('c')
                    else: a.append('b')
                elif a[-1]=='b':
                    if s[i+1]=='a': a.append('c')
                    else: a.append('a')
                elif s[i+1]=='a': a.append('b')
                else: a.append('a')
            else: a.append(s[i])
        return ''.join(a[1:])
