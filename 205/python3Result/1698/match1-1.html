class Solution:
    def modifyString(self, s: str) -> str:

        t = list(s)
        n = len
(t)
        for i, v in enumerate(t):
            if v == '?':
                for u in range(26):
                    ch = chr(ord('a') + u)
                    if (i-1 < 0 or t[i-1] != ch) and (i+1 >= n or t[i+1] != ch):
                        t[i] = ch
                        break
        return ''.join(t)
