class Solution:
    def modifyString(self, s: str) -> str:
        l = list(s)
        ans = list(l)
        for i, c in enumerate(l):
            if c == '?':
                s = set()
                if i:
                    s.add(ans[i - 1])
                if i + 1 < len(l):
                    s
.add(l[i + 1])
                for r in string.ascii_lowercase:
                    if r not in s:
                        ans[i] = r
                        break
        return ''.join(ans)
