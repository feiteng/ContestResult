class Solution:
    def modifyString(self, s: str) -> str:
        l = list(s)
        for i in range(len(l)):
            if l[i] != '?':
                continue
                
            ll = []
            if i > 0:
                ll.append(l[i-1])
            if i < len(s) - 1:
                ll.append(l[i+1])
                
            if 'a' in ll:
                if 'b' in ll:
                    l[i] = 'c'
                else:
                    l[i] = 'b'
            else:
                l[i] = 'a'
                
        return ''.join(l)
