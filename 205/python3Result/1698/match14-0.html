class Solution:
    def modifyString(self, s: str) -> str:
        arr = list(s)
        for i, c in enumerate(arr):
            if c != '?':
                continue
            pc = '' if i == 0 else arr[i - 1]
            rc = '' if i == len(arr) - 1 else arr[i + 1]
            if pc != 'a' and rc != 'a':
                arr[i] = 'a'
            elif pc != 'b' and rc != 'b':
                arr[i] = 'b'
            else:
                arr[i] = 'c'
        return ''.join(arr)
