class Solution:
    def modifyString(self, s: str) -> str:
        res = [ch for ch in s]
        for i, v in enumerate(res):
            if v =='?':
                for ch in 'abcd':
                    if (i == 0 or res[i-1] != ch) and (i == len(res) - 1 or res
[i+1] != ch):
                        res[i] = ch
                        break
        return ''.join(res)
