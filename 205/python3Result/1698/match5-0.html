class Solution:
    def modifyString(self, s: str) -> str:
        s = list(s)
        ans = []
        for i, x in enumerate(s):
            if x == '?':
                x = None
                if i > 0:
                    x = s[i - 1]
                y = None
                for j in range(i, len(s)):
                    if s[j] != '?':
                        y = s[j]
                        break
                #print(x, y)
                
                cands = ['a', 'b', 'c']
                if x in cands:
                    cands.remove(x)
                if y in cands:
                    cands.remove(y)
                ans.append(cands.pop())
                s[i] = ans[-1]
            else:
                ans.append(x)
                
        return ''.join(ans)
