class Solution:
    def modifyString(self, s: str) -> str:
        for i in range(len(s)):
            if s[i] == '?':
                for c in ascii_lowercase:
                    if i > 0 and s[i-1] == c:
                        continue
                    if i < len(s) - 1 and s[i+1] == c:
                        continue
                    s = s[:i] + c + s[i+1:]
        return s
