class Solution:
    def modifyString(self, s: str) -> str:
        mapper = list(string.ascii_lowercase)
        result = list()
        for ind, val in enumerate(s):
            if val == '?':
                now = 0
                if ind > 0 and mapper[now] == result[ind - 1]:
                    now += 1
                if ind < len(s) - 1 and mapper[now] == s[ind + 1]:
                    now += 1
                if ind > 0 and mapper[now] == result[ind - 1]:
                    now += 1
                if ind < len(s) - 1 and mapper[now] == s[ind + 1]:
                    now += 1
                result.append(mapper[now])
            else:
                result.append(val)
        return ''.join(result)
