class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans=0
        n=len(s)
        res=[0]
        for i in range(1,n):
            if s[res
[-1]]==s[i]:
                if cost[i]<=cost[res[-1]]:
                    ans+=cost[i]
                else:
                    ans+=cost[res[-1]]
                    res.pop()
                    res.append(i)
            else:
                res.append(i)
            #print(i,res,ans)
        return ans
