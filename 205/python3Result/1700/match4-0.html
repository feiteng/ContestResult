class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        #if a repeat chain is found, only keep the maximum one
        left = 0
        res = 0
        while left < len(s):
            right = left + 1
            while right < len(s) and s[right] == s[left]:
                right += 1
            #delete everything except max(s[left:right])
            res += sum(cost[left:right]) - max(cost[left:right])
            
            left = right
        return res
