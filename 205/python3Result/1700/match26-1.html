class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans = 0
        prev 
= None
        total = 0
        maximum = 0
        for i, c in zip(s, cost):
            if i == prev:
                total += c
                maximum = max(maximum, c)
            else:
                ans += total - maximum
                prev = i
                total = c
                maximum = c
        
        return ans + total - maximum
