class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        i = 0
        ans = 0
        while i < len(s):
            j = i
            while j < len(s) and s[j] == s[i]:
                j += 1
            ans += sum(cost[i:j]) - max(cost[i:j])
            i = j
        return ans
