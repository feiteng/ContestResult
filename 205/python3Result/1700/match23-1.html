class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        block = []
        start = 0
        for i in range(len(s)):
            if i != 0 and s[i] != s[i - 1]:
                block.append(cost[start:i])
                start = i
        block.append(cost[start:])
        return sum(sum(b) - max(b) for b in block)
