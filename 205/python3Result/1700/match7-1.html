class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        s+='A'
        ans,i=0,0
        for j in range(1,len(s)):
            if s[i]!=s[j]:
                ans+=sum([cost[k] for k in range(i,j)])-max([cost[k] for k in range(i,j)])
                i=j
        return ans
