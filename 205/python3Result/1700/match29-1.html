class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        l = 0
        intervals = []
        for i in range(1, len(s)):
            if s[i] != s[l]:
                intervals.append([l, i])
                l = i
        intervals.append([l, len(s)])
        
        ans = 0
        for l, r in intervals:
            total = 0
            maxim = 0
            for i in range(l, r):
                total += cost[i]
                maxim = max(maxim, cost[i])
            ans += total - maxim
        return ans
