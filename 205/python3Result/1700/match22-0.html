class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans = 0
        n = len(s)
        last = None
        left = None
        for i in range(n):
            if s[i] != last:
                if left is not None:
                    m = 0
                    #print('from', left, i)
                    for j in range(left, i):
                        ans += cost[j]
                        m = max(cost[j], m)
                    ans -= m
                    #print('ans', ans, m)
                last = s[i]
                left = i
        if left is None:
            return ans
        #print(left)
        m = 0
        for j in range(left, n):
            ans += cost[j]
            m = max(cost[j], m)
        ans -= m
        return ans
