class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        n = len(s)
        ss = cost.copy()
        for i in range(1, n):
            ss[i] += ss[i-1]
        ss.append(0)
        
        def rmq(l, r):
            return max(cost[l:r+1])
        
        ans = 0
        prev = -1
        for i, c in enumerate(s):
            if i > 0 and s[i] == s[i-1] and (i == n-1 or s[i] != s[i+1]):
                ans += ss[i] - ss[prev] - rmq(prev+1, i)
            if (i + 1 < n and s[i] != s[i+1]): prev = i
        return ans
