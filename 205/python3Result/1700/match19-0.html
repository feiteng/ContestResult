class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        start=0
        curr=0
        ret=0
        while curr<len(s):
            while curr<len(s) and s[curr]==s[start]:
                curr+=1
            if curr-start>1:
                ret+=sum(cost[start:curr])-max(cost[start:curr])
            start=curr
        return ret
