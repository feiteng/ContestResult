class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        last = ''
        l = [0]
        ans = 0
        for i, c in enumerate(s):
            if c == last:
                l.append(cost[i])
            else:
                ans += sum(l) - max(l)
                last = c
                l = [cost[i]]
        
        ans += sum(l) - max(l)
        return ans
