class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans = 0
        l = 0
        while l < len(s):
            r = l
            max_cost = cost[r]
            sum_cost = cost[r]
            while r < len(s) - 1 and s[r] == s[r + 1]:
                r += 1
                max_cost = max(max_cost, cost[r])
                sum_cost += cost[r]
            l = r + 1
            ans += sum_cost - max_cost
        return ans
