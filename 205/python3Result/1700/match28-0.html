class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        pre, pc = '', 999999
        res = 0
        for k, x in enumerate(s):
            if x == pre:
                res += min(pc, cost[k]) 
                
                pre, pc = x, max(pc, cost[k])
            else:
                pre, pc = x, cost[k]
        return res
