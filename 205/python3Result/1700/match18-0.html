class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        prev = '?'
        prev_costs = [0]
        res = 0
        for i, c in enumerate(s):
            if c == prev:
                prev_costs.append(cost[i])
            else:
                res += sum(prev_costs) - max(prev_costs)
                prev = c
                prev_costs = [cost[i]]
        res += sum(prev_costs) - max(prev_costs)
        return res
