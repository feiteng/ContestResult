class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        n = len(s)
        fr = [-1] * 26
        f = [0] * n
        sc = [0]
        for i in range(n):
            sc.append(cost[i] + sc[-1])
        ans = sc[-1]
        for i, c in enumerate(s):
            c = ord(c) - ord('a')
            f[i] = sc[i]
            for j in range(26):
                if c != j and fr[j] > -1:
                    f[i] = min(f[i], f[fr[j]] + sc[i] - sc[fr[j] + 1])
            if fr[c] == -1 or f[i] < f[fr[c]] + sc[i + 1] - sc[fr[c] + 1]:
                fr[c] = i
            ans = min(ans, f[i] + sc[-1] - sc[i + 1])
        return ans
