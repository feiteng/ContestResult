class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        
        stack = []
        ans = 0
        for i in range(len(s)):
            if stack and s[stack[-1]]==s[i]:
                if cost[i]<cost[stack[-1]]:
                    ans += cost[i]
                else:
                    ans += cost[stack[-1]]
                    stack.pop()
                    stack.append(i)
            else:
                stack.append(i)
        
        
        return ans
