class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        s = ' ' + s
        cost = [0] + cost
        l = len(s)
        MAX = 1e100
        f = [[MAX] * 27]
        f[0][26] = 0
        for i in range(1, l):
            # remove
            f.append(list(f[i - 1]))
            for j in range(27):
                if f[i][j] == MAX: continue
                f[i][j] += cost[i]
                
            # keep i
            k = ord(s[i]) - 97
            for j in range(27):
                if k != j and f[i - 1][j] < MAX:
                    f[i][k] = min(f[i][k], f[i - 1][j])
        # print(f)
        return min(f[l - 1])
