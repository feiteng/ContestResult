class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        if not s:
            return 0
        totalCost = 0
        prev = 0
        for i in range(1, len(s)):
            if s[i] == s[prev]:
                if cost[i] < cost[prev]:
                    totalCost += cost[i]
                else:
                    totalCost += cost[prev]
                    prev = i
            else:
                prev = i
                
        return totalCost
