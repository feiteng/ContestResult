class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        i = 0
        ans = 0
        for k, g in itertools.groupby(s):
            l = len(list(g))
            if l > 1:
                ans += sum(heapq.nsmallest(l - 1, cost[i:i + l]))
            i += l
        return ans
