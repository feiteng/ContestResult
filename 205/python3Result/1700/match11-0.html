class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans = 0
        i = 0
        while i<len(s):
            j = i+1
            ss = cost[i]
            maxs = cost[i]
            while j<len(s) and s[j] == s[i]:
                ss += cost[j]
                maxs = max(maxs,cost[j])
                j += 1
            if j>i+1:
                ans += ss - maxs
            i = j
        return ans
