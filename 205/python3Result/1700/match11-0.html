class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        #greedy
        res = 0
        stack = [0]
        for i in range(1, len(s)):
            if s[i] != s[stack[-1]]:
                stack.append(i)
            else:
                if cost[stack[-1]] > cost[i]:
                    res += cost[i]
                else:
                    res += cost[stack[-1]]
                    stack.pop()
                    stack.append(i)
            # print(res)
        return res
