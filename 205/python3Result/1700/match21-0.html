class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        s = [ord(x) - ord('a') for x in s]
        m = max(s) + 1
        n = len(s)
        INF = 10000000000
        f = [[INF] * m for _ in range(n)]
        f[0][s[0]] = 0
        acc = cost[0]
        #print(s, cost)
        #print(n)
        for i in range(1, n):
            f[i][s[i]] = acc
            acc += cost[i]
            for j in range(m):
                f[i][j] = min(f[i][j], f[i - 1][j] + cost[i])
                if j == s[i]:
                    continue
                f[i][s[i]] = min(f[i][s[i]], f[i - 1][j])
        ans = min(f[-1])
        return ans
