class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        
        ans = 0
        
        lst = '*'
        
        n = len(s)
        
        i = 0
        
        while(i<n):
            
            j = i
            m = cost[i]
            t = cost[i]
            
            while(j<n-1 and s[j+1] == s[j]):
                j += 1
                m = max(m, cost[j])
                t += cost[j]

            if(i != j):
                ans += t - m
            
            i = j + 1
        
        return ans
