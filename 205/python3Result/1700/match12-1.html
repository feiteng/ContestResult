class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        lo = hi = 0
        allcost = sum(cost)
        remain = 0
        s += "$"
        for hi in range(len(s)):
            if s[hi] != s[lo]:
                remain += max(cost[lo:hi])
                lo = hi
        return allcost - remain
