class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        prev = 0
        res = 0
        for i in range(1,len(s)):
            if s[i] == s[prev]:
                if cost[i] < cost[prev]:
                    res += cost[i]
                else:
                    res += cost[prev]
                    prev = i
            else:
                prev = i
                
        return res
