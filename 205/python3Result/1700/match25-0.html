class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        ans = 0
        idx = 0
        while idx < len(s):
            lb = idx
            letter = s[idx]
            while idx + 1 < len(s) and s[idx + 1] == letter:
                idx += 1
                
            ub = idx
            
            ans += sum(cost[lb:ub+1]) - max(cost[lb:ub+1]) 
            
            idx += 1
        return ans
