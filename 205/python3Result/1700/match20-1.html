class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        
        total = sum(cost)
        remain = 0
        prev, prev_max = s[0], cost[0]
        for i, c in enumerate(s[1:]):
            if c != prev:
                remain += prev_max
                prev, prev_max = c, cost[i+1]
            else:
                prev_max = max(prev_max, cost[i+1])
        remain += prev_max
        
        return total - remain
