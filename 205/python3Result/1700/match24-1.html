class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        i = 0
        result = 0
        while i < len(s):
            j = i
            while j < len(s) and s[j] == s[i]:
                j += 1
            j -= 1
            if j != i:
                result += sum(cost[i:j+1]) - max(cost[i:j+1])
            i = j + 1
        return result
