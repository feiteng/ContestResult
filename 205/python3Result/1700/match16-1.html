class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        total = 0
        pre = -1
        for i, char in enumerate(s):
            if pre != -1 and s[pre] == s
[i]:
                if cost[pre] <= cost[i]:
                    total += cost[pre]
                    pre = i
                else:
                    total += cost[i]
            else:
                pre = i
        return total
