class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        pre = '#'
        total, high = 0, 0
        res = 0
        for cha, cos in zip(s+'#',cost+[0]):
            if cha==pre:
                total += cos
                high = max(high, cos)
            else:
                res += total - high
                total, high = cos, cos
                pre = cha
        return res
