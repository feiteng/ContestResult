class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        groups = []
        cur = [0]
        for i in range(1,len(s)):
            if s[i] == s[cur[-1]]:
                cur.append(i)
            else:
                groups.append(tuple(cur))
                cur = [i]
        groups.append(tuple(cur))
        ts = sum(cost)
        return ts - sum(max(cost[j] for j in g) for g in groups)
