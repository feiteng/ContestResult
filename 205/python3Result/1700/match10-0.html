class Solution:
    def minCost(self, s: str, cost: List[int]) -> int:
        
        total_cost = 0
        candidate_cost = []
        for i, ch in enumerate(s):
            if i != 0 and s[i] == s[i - 1]:
                candidate_cost.append(cost[i])
            else:
                if len(candidate_cost) > 1:
                    total_cost += sum(candidate_cost) - max(candidate_cost)
                candidate_cost = [cost[i]]
        if len(candidate_cost) > 1:
            total_cost += sum(candidate_cost) - max(candidate_cost)
        
        
        return total_cost
