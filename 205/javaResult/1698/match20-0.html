class Solution {
    public String modifyString(String s) {
        char[] arr = new char[s.length()];
        int n = s.length();
        for (int i = 0; i < n; i++) {
            arr[i] = s.charAt(i);
        }
        
        if (n == 1) {
            if (arr[0] == '?') {
                return "a";
            } else {
                return s;
            }
        }
        
        for (int i = 0; i < n; i++) {
            if (arr[i] == '?') {
                if (i == 0) {
                    arr[i] = getRandom('?', arr[i+1]);
                } else if (i == n-1) {
                    arr[i] = getRandom(arr[i-1], '?');
                } else {
                    arr[i] = getRandom(arr[i-1], arr[i+1]);
                }
            }
        }
        
        return String.valueOf(arr);
    }
    
    private char getRandom(char before, char after) {
        Random r = new Random();
        char c = (char)(r.nextInt(26) + 'a');
        while(c == before || c == after) {
            c = (char)(r.nextInt(26) + 'a');
        }
        return c;
    }
}
