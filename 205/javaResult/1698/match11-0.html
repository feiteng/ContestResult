class Solution {
    public String modifyString(String s) {

        int n = s.length();
        StringBuilder ans = new StringBuilder();

        for (int i = 0; i < n; i++) {
            if (s.charAt(i) != '?')
                ans.append(s.charAt(i));
            else {
                char a = 'a';
                boolean flag = false;
                for (int j = 0; j < 26; j++) {
                    a = (char) (j + 'a');
                    if (i > 0 && ans.charAt(i - 1) == a ||
                            i < n - 1 && s.charAt(i + 1) == a)
                        continue;
                    flag = true;
                    break;
                }
                ans.append(a);
            }
        }

        return ans.toString();

    }
}
