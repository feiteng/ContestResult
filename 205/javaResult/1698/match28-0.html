class Solution {
    public String modifyString(String s) {
        StringBuilder ans = new StringBuilder("");
        
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) != '?') ans.append(s.charAt(i));
            else{
                for(int j = 0; j < 26; j++){
                    if( (i == 0 || ans.charAt(i-1) != (char)(j + 'a')) &&  (i >= s.length()-1 || s.charAt(i+1) != (char)(j+'a'))){
                        ans.append((char)(j+'a'));
                        break;
                    }   
                }
            }
            
        }
            
        return ans.toString();        
    }
}
