class Solution {
    public String modifyString(String s) {
        char[] array = s.toCharArray();
        for (int i = 0; i < array.length; i++) {
            if (array[i] == '?') {
                Set<Character> set = new HashSet<>();
                if (i - 1 >= 0) {
                    set.add(array[i - 1]);
                }
                if (i + 1 < array.length && array[i + 1] != '?') {
                    set.add(array[i + 1]);
                
}
                for (char c = 'a'; c <= 'z'; c++) {
                    if (!set.contains(c)) {
                        array[i] = c;
                        break;
                    }
                }
            }
        }
        return new String(array);
    }
}
