class Solution {
    /*
    input: String s is not null or empty
         only lower case & ?
    M1: try out the char - linear scan
    
        j?qg??b
        
         i
        
        ? -> 
            try the first char in 'a' - 'z' that is different from i - 1 & i + 1
            
        jaqgacb
        
        X?????Y
         -----
        
      Time: O(n)
      Space: O(1)
    */
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != '?') {
                sb.append(s.charAt(i));
            } else {
                char ch = 'a';
                while ((i - 1 >= 0 && sb.charAt(i - 1) == ch) 
                       || (i + 1 < s.length() && s.charAt(i + 1) == ch)) {
                    ch++;
                }
                sb.append(ch);
            }
        }
        
        return sb.toString();
    }
}
