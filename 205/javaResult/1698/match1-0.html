class Solution {
    public String modifyString(String s) {
        int n = s.length();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (s.charAt(i) == '?') {
                if ((i == 0 || sb.charAt(i - 1) != 'a') && (i == n - 1 || s.charAt(i + 1) != 'a')) {
                    sb.append('a');
                } else if ((i == 0 || sb.charAt(i - 1) != 'b') && (i == n - 1 || s.charAt(i + 1) != 'b')) {
                    sb.append('b');
                } else {
                    sb.append('c');
                }
            } else {
                sb.append(s.charAt(i));
            }
        }
        return sb.toString();
    }
}
