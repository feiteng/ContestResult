class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder(s);
        
        for (int i = 0; i < sb.length(); i++) {
            char c = sb.charAt(i);
            if (c != '?') continue;
            
            char left = i == 0 ? '0' : sb.charAt(i - 1);
            char right = i == sb.length() - 1 ? '0' : sb.charAt(i + 1);
            
            for (char k = 'a'; k <= 'z'; k++) {
                if (k != left && k != right) {
                    sb.setCharAt(i, k);
                    break;
                }
            }
        }
        
        return sb.toString();
    }
}
