class Solution {
    public String modifyString(String s) {
        char[] chs = s.toCharArray();
        for (int i = 0; i < s.length(); i++) {
            if (chs[i] == '?') {
                for (char ch = 'a'; ch <= 'z'; ch++) {
                    if ((i == 0 || chs[i - 1] != ch) && (i == s.length() - 1 || chs[i + 1] != ch)) {
                        chs[i] = ch;
                        break;
                    }
                }
            }
        }
        return new String(chs);
    }
}
