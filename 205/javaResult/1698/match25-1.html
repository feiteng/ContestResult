class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if(ch == '?') {
                
char left = i > 0 ? sb.charAt(i - 1) : '.';
                char right = i + 1< s.length() ? s.charAt(i + 1) : '.';
                for(int j = 0; j < 26; j++) {
                    char toReplace = (char)('a' + j);
                    
if(toReplace != left && toReplace != right) {
                        sb.append(toReplace);
                        break;
                    }
                }
            } else sb.append(ch);
        }
        return sb.toString();
    }
}
