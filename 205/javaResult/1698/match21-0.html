class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < s.length(); i++) {
            if(s.charAt(i) != '?') {
                sb.append(s.charAt(i));
                continue;
            }
            for(char c = 'a'; c <= 'z'; c++) {
                if(i > 0 && sb.charAt(i - 1) == c) continue;
                if(i < s.length() - 1 && s.charAt(i + 1) == c) continue;
                sb.append(c);
                break;
            }
        }
        return sb.toString();
    }
}
