class Solution {
    public String modifyString(String s) {
        char arr[]=new char[s.length()];
        int n=s.length();
        for(int i=0;i<s.length();i++)
        {
            if(i==0&&s.charAt(i)=='?')
            {
                if(i!=n-1)
                {
                    if(s.charAt(i+1)=='a')
                    {
                        arr[i]='b';
                    }
                    else
                    {
                    arr[i]='a';
                    }
                }
                else
                {
                    arr[i]='a';
                }
            }
            else if(s.charAt(i)=='?')
            {
                int next=((int)arr[i-1]-(int)'a'+1)%26;
                if(i!=n-1)
                {
                    int c=s.charAt(i+1)-(int)'a';
                    if(next==c)
                    {
                        arr[i]=(char)((next+1)%26+(int)'a');
                    }
                    else
                    {
                    arr[i]=(char)((next)%26+(int)'a');
                    }
                }
                else
                {
                    arr[i]=(char)((next)%26+(int)'a');
                }
            }
            else
            {
                arr[i]=s.charAt(i);
            }
        }
        return new String(arr);
    }
}
