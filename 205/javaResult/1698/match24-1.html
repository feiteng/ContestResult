class Solution {
    public String modifyString(String s) {
        char p[] = s.toCharArray();
        for(int i = 0;i<p.length;i++){
            if(p[i]=='?'){
                char l = 'a';
                if(i-1>=0){
                    l = p[i-1];
                }
                char r = 'z';
                if(i+1<p.length-1){
                    r = p[i+1];
                    if(r=='?'){
                        r= 'c';
                    }
                }
                char max = (char)Math.max(l,r);
                char o = (char)((max+1-'a')%26 + 'a');
                if(o==l||o==r){
                    o = (char)((o+1-'a')%26+'a');
                }
                p[i]=o;
            }
        }
        return String.valueOf(p);
    }
}
