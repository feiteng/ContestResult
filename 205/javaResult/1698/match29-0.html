class Solution {
    public String modifyString(String s) {
        char[] arr = s.toCharArray();
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == '?') {
                char good = '0';
                for (char c = 'a'; c <= 'z'; c++) {
                    boolean valid = true;
                    if (i > 0 && c == arr[i - 1]) {
                        valid = false;
                    }
                    if (i + 1 < arr.length && c == arr[i + 1]) {
                        valid = false;
                    }
                    if (valid) {
                        good = c;
                        break;
                    }
                }
                arr[i] = good;
            }
        }
        return new String(arr);
    }
}
