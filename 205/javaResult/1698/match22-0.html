class Solution {
    public String modifyString(String s) {
        int n = s.length();
        char[] cs = s.toCharArray();
        if (n == 1) {
            if (cs[0] == '?') return "a";
            return s;
        }

        for (int i = 0; i < n; i++) {
            if (cs[i] != '?') continue;

            for (int j = 0; j < 26; j++) {
                char c = (char)('a' + j);

                if (i == 0) {
                    if (c != cs[i+1]) {
                        cs[i] = c;
                        break;
                    }
                } else if (i == n - 1) {
                    if (c != cs[i-1]) {
                        cs[i] = c;
                        break;
                    }
                } else {
                    if (c != cs[i+1] && c != cs[i-1]) {
                        cs[i] = c;
                        break;
                    }
                }
            }
        }
        return new String(cs);
    }
}
