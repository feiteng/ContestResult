class Solution {
    public String modifyString(String s) {
        StringBuffer sb = new StringBuffer();
        char[] ch = s.toCharArray();
        int n = ch.length;
        for(int i = 0; i < n; i++){
            if(ch[i] != '?'){
                sb.append(ch[i]);
            }else{
                for(int j = 0; j < 26; j++){
                    char c = (char)('a' + j);
                    if(i > 0 && c == sb.charAt(i - 1)) continue;
                    if(i + 1 < n && c == ch[i + 1]) continue;
                    sb.append(c);
                    break;
                }
            }
        }
        
        return sb.toString();
    }
}
