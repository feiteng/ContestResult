class Solution {
    public String modifyString(String s) {
        char[]cs= s.toCharArray();
        for (int i = 0; i < cs.length; i++) {
            if (cs[i] == '?') {
                if (i == 0) {
                    if (i == cs.length-1 || cs[i+1] != 'a') {
                        cs[i] = 'a';
                    } else {
                        cs[i] = 'b';
                    }
                } else if (i == cs.length-1){
                    if (cs[i-1] == 'a') {
                        cs[i] = 'b';
                    } else {
                        cs[i] = 'a';
                    }
                } else {
                    cs[i] = (char) (cs[i-1] + 1);
                    if (cs[i] == cs[i+1]) {
                        cs[i]++;
                    }
                    if (cs[i] > 'z') {
                        cs[i] = 'a';
                    }
                
}
            }
        }
        return new String(cs);
    }
}
