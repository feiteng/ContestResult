class Solution {
    public String modifyString(String s) {
        char[] cc=s.toCharArray();
        for(int i=0;i<cc.length;++i){
            if(cc[i]=='?'){
                for(int j=0;j<26;++j){
                    cc[i]=(char)(j+'a');
                    if(i>0&&cc[i]==cc[i-1])continue;
                    if(i<cc.length-1 && cc[i]==cc[i+1])continue;
                    break;
                }
            }
        }
        return new String(cc);
    }
}
