class Solution {
    public String modifyString(String s) {
        char[] ca = s.toCharArray();
        int n = ca.length;
        for (int i = 0; i < n; i++) {
            if (ca[i] == '?') {
                char pre = i == 0? '#' : ca[i - 1];
                char next = i == n - 1? '#' : ca[i + 1];
                for (char c = 'a'; c <= 'z'; c++) {
                    if (c != pre && c != next) {
                        ca[i] = c;
                        break;
                    }
                }
            }
        }
        return new String(ca);
    }
}
