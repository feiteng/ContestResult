class Solution {
    public String modifyString(String s) {

        char[] chars = s.toCharArray();
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] == '?') {
                for (int j = 0; j < 26; j++) {
                    char ch = (char) ('a' + j);
                    if ((i - 1 < 0 || chars[i - 1] != ch) && 
                            (i + 1 >= s.length() || chars[i + 1] != ch)) {
                        chars[i] = ch;
                        break;
                    }
                
}
            }
        }
        return new String(chars);
    }
}
