class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        char[] ch = s.toCharArray();
        char last = '0';
        for (int i = 0; i < ch.length; i++) {
            char c = '0';
            if (ch[i] != '?') c = ch[i];
            if (ch[i] == '?') {
                for (char a = 'a'; a <= 'z'; a++) {
                    if (a != last && (i == ch.length - 1 || a != ch[i + 1])) {
                        c = a;
                        break;
                    }
                }
            }
            last = c;
            sb.append(c);
        }
        return sb.toString();
    }
}
