class Solution {
    public String modifyString(String s) {
        // char prev = 'A', next = 'A';
        String ans = "";
        
        for(int i = 0 ; i < s.length() ; i++) {
            if(s.charAt(i) == '?') {
                char prev = 'A', next = 'A';
                if(i > 0) {
                    prev = ans.charAt(i-1);
                }
                if(i < s.length()-1) {
                    next = s.charAt(i+1);
                }
                for(int k = 0 ; k < 26 ; k++) {
                    char temp = (char)('a' + k);
                    if(prev != temp && next != temp) {
                        ans += temp;
                        break;
                    }
                }
            
} else {
                ans += s.charAt(i);
            }
        }
        
        return ans;
    }
}
