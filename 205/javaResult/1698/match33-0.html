class Solution {
    public String modifyString(String s) {
        String res = "";
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '?') {
                char c = 'a';
                while (i > 0 && res.charAt(i - 1) == c || i < s.length() - 1 && s.charAt(i + 1) == c) {
                    c++;
                }
                res += c;
            } else {
                res += s.charAt(i);
            }
        }
        return res;
    }
}
