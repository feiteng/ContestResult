class Solution {

    public String modifyString(String s) {
        char[] c = s.toCharArray();
        for (int i = 0; i < c.length; i++) {
            if (s.charAt(i) == '?') {
                char next = 'a';
                if (i != 0 && c[i - 1] == 'a' || i != c.length - 1 && c[i + 1] == 'a') {
                    next++;
                }
                if (i != 0 && c[i - 1] == next || i != c.length - 1 && c[i + 1] == next) {
                    next++;
                }
                c[i] = next;
            }
        }
        return String.valueOf(c);
    }
}
