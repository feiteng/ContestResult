class Solution {
    public String modifyString(String s) {
        char[] c = s.toCharArray();
        int n = c.length;
        for(int i = 0; i < n; i++)
        {
            if(c[i] == '?')
            {
                char c1 = i > 0 ? c[i - 1] : 'A';
                char c2 = i < n - 1 ? c[i + 1] : 'A';
                for(int j = 0; j < 26; j ++)
                {
                    char c3 = (char)('a' + j);
                    
if(c3 != c1 && c3 != c2)
                    {
                        c[i] = c3;
                        break;
                    }
                }
            }
        }
        return new String(c);
    }
}
