class Solution {
    public String modifyString(String s) {
        int n = s.length();
        char ch[] = s.toCharArray();
        
        for(int i = 0; i < n; i++){
            char c = 'a';
            if(ch[i]=='?'){
                if(i==0){
                    if(i == n-1){
                        return "a";
                    }
                    if(ch[i+1] == c){
                        c++;
                    }
                    ch[i] = c;
                }
                else if(i==n-1){
                    if(ch[i-1] == c){
                        c++;
                    }
                    ch[i] = c;
                }
                else{
                    if(ch[i+1] == c){
                        c++;
                    }
                    if(ch[i-1] == c){
                        c++;
                    }
                    if(ch[i+1] == c){
                        c++;
                    }
                    ch[i] = c;
                }
            }
        }
        return new String(ch);
    }
}
