class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        int index = 0;
        while (index < s.length()) {
            if (s.charAt(index) != '?') {
                sb.append(s.charAt(index++));
            } else {
                int next = index;
                while (next + 1 < s.length() && s.charAt(next + 1) == '?') next++;
                char[] c = new char[2];
                char l = index == 0 ? '?' : s.charAt(index - 1);
                char r = next == s.length() - 1 ? '?' : s.charAt(next + 1);
                c[0] = 'a';
                while (c[0] == l || c[0] == r) c[0] = (char)(c[0] + 1);
                c[1] = (char) (c[0] + 1);
                while (c[1] == l || c[1] == r) c[1] = (char) (c[1] + 1);
                for (int i = index ; i <= next ; i++) {
                    sb.append(c[(i - index) % 2]);
                }
                index = next + 1;
            }
        }
        return sb.toString();
    }
}
