class Solution {
    public String modifyString(String s) {
        char[] res = new char[s.length()];
        for (int i = 0; i < res.length; i++) {
            char c = s.charAt(i);
            if (c != '?') {
                res[i] = c;
            } else {
                Character before = (i == 0) ? null : res[i - 1];
                Character after = (i == s.length()  - 1) ? null : s.charAt(i + 1);
                for (char ch = 'a'; ch <= 'z'; ch++) {
                    if (before != null && ch == before) {
                        continue;
                    }
                    if (after != null && ch == after) {
                        continue;
                    }
                    res[i] = ch;
                    break;
                }
            }
        }
        return new String(res);
    }
}
