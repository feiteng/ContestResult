class Solution {
    public String modifyString(String s) {
        int n = s.length();
        
char[] arr = s.toCharArray();
        for (int i = 0; i < n; i++) {
            if (arr[i] == '?') {
                Set<Character> set = new HashSet<>();
                if (i > 0) {
                    set.add(arr[i-1]);
                }
                if (i+1 < n && arr[i+1] != '?') {
                    set.add(arr[i+1]);
                }
                for (char c = 'a'; c <= 'z'; c++) {
                    if (!set.contains(c)) {
                        arr[i] = c;
                        break;
                    }
                }
            }
        }
        return new String(arr);
    }
}
