class Solution {
    public String modifyString(String s) {
        StringBuffer sb = new StringBuffer();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != '?') {
                sb.append(s.charAt(i));
            } else {
                if (s.length() == 1) {
                    return "a";
                } else {
                    if (i == 0) {
                        char c = s.charAt(i + 1);
                        if (c == 'a') {
                            sb.append('b');
                        } else {
                            sb.append('a');
                        }
                    } else if (i == s.length() - 1) {
                        char c = sb.charAt(i - 1);
                        if (c == 'a') {
                            sb.append('b');
                        } else {
                            sb.append('a');
                        }
                    } else {
                        char c1 = sb.charAt(i - 1);
                        char c2 = s.charAt(i + 1);
                        if (c1 != 'a' && c2 != 'a') {
                            sb.append('a');
                        } else if (c1 != 'b' && c2 != 'b') {
                            sb.append('b');
                        } else {
                            sb.append('c');
                        }
                    }
                }
            }
        }
        
        return sb.toString();
    }
}
