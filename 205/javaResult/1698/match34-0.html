class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != '?') {
                sb.append(s.charAt(i));
            } else {
                char left = '?';
                if (sb.length() > 0) {
                    left = sb.charAt(sb.length() - 1);
                }
                char right = '?';
                if (i + 1 < s.length()) {
                    right = s.charAt(i + 1);
                }
                char cur = 'a';
                for (int j = 0; j < 26; j++) {
                    char ch = (char)(j + 'a');
                    if (ch != left && ch != right) {
                        sb.append(ch);
                        break;
                    }
                }
            }            
        }
        return sb.toString();
    }
}
