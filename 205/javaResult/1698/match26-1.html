class Solution {
    public String modifyString(String s) {
        char[] arr=s.toCharArray();
        for(int i=0;i<arr.length;i++){
            if(arr[i]=='?'){
                arr[i]='a';
                if(i>0 && arr[i-1]==arr[i]) arr[i]='b';
                if(i+1<arr.length && arr[i]==arr[i+1]) arr[i]='c';
            
}
        }
        
        return new String(arr);  
    }
}
