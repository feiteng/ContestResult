class Solution {
    public String modifyString(String s) {
        String res = "";
        int n = s.length();
        for(int i = 0; i < n; i++){
            char c = s.charAt(i);
            if(c == '?'){
                c = get(i > 0 ? res.charAt(res.length() - 1) : '.', i < n - 1? s.charAt(i+1): '.');
            }
            res += c;
        }
        return res;
    }
    
    public char get(char prev, char next){
        for(int i = 0; i < 26; i++){
            char c = (char)('a' + i);
            if(c != prev && c != next){
                return c;
            }
        }
        return ',';
    }
}
