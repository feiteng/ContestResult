class Solution {
    public String modifyString(String s) {
        int len = s.length();
        StringBuilder sb = new StringBuilder();
        for (int i = 0;i < s.length();i++) {
            if (s.charAt(i) != '?') {
                sb.append(s.charAt(i));
            } else {
                char pre = sb.length() == 0 ? '*' : sb.charAt(sb.length() - 1);
                char next = i + 1 < len ? s.charAt(i + 1) : '*';
                for (char c = 'a'; c <= 'z';c++) {
                    if (c != pre && c != next) {
                        sb.append(c);
                        break;
                    }
                }
            }
        }
        return sb.toString();
    }
}
