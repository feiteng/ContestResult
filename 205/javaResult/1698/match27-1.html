class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        char[] chars = s.toCharArray();
        int n = s.length();
        for(int i = 0 ; i < n ; i++){
            if(chars[i] != '?'){
                sb.append(chars[i]);
            } else {
                for(char c = 'a'; c <= 'z'; c++){
                    if((i == 0 || c != chars[i-1]) && (i == n - 1 || c!= chars[i + 1])){
                        sb.append(c);
                        chars[i] = c;
                        break;
                    }
                }
            }
        }
        return sb.toString();
    }
}
