class Solution {
    public String modifyString(String s) {
        StringBuilder sb = new StringBuilder();
        int n = s.length();
        for(int i=0; i<n; i++){
            char c = s.charAt(i);
            if(c!='?'){
                sb.append(c);
            }
            else{
                for(char r='a'; r<='z'; r++){
                    if(i>0 && i<n-1){
                        char prev = sb.charAt(i-1);
                        char next = s.charAt(i+1);
                        if(r!=prev && r!=next){
                            sb.append(r);
                            break;
                        }
                    }
                    else if(i>0){
                        char prev = sb.charAt(i-1);
                        if(r!=prev){
                            sb.append(r);
                            break;
                        }
                    }
                    else if(i<n-1){
                        char next = s.charAt(i+1);
                        if(r!=next){
                            sb.append(r);
                            break;
                        }                        
                    }
                    else{
                        sb.append(r);
                        break;
                    }
                }
            }
        }
        return sb.toString();
    }
}
