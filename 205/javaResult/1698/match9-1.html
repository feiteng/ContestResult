class Solution {
    public String modifyString(String s) {
        char[] cs = s.toCharArray();
        for (int i = 0; i < cs.length; i++) {
            if (cs[i] != '?') continue;
            
for (char j = 'a'; j <= 'z'; j++) {
                if (i > 0 && j == cs[i - 1]) continue;
                if (i < cs.length - 1 && j == cs[i + 1]) continue;
                cs[i] = j;
                break;
            }
        }
        return new String(cs);
    }
}
