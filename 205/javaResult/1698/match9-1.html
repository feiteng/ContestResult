class Solution {
    public String modifyString(String s) {
        char[] ss = s.toCharArray();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < ss.length; i++) {
            if (ss[i] != '?') sb.append(ss[i]);
            else {
                for (int j = 0; j < 26; j++) {
                    char c = (char)('a' + j);
                    if (i > 0 && c == ss[i-1]) continue;
                    if (i < ss.length - 1 && c == ss[i+1]) continue;
                    sb.append(c);
                    ss[i] = c;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
