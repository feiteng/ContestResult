class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UnionFind uf1 = new UnionFind(n);
        UnionFind uf2 = new UnionFind(n);
        int count = 0;
        
        for (int[] edge : edges) {
            int type = edge[0];
            int from = edge[1] - 1;
            int to = edge[2] - 1;
            if (type == 3) {
                if (uf1.find(from, to)) {
                    //System.out.println("type 3, from : " + from + " to" + to);
                    count++;
                } else {
                    uf1.union(from, to);
                    uf2.union(from, to);
                }
            }
        }
        
        for (int[] edge : edges) {
            int type = edge[0];
            int from = edge[1] - 1;
            int to = edge[2] - 1;
            if (type == 1) {
                if (uf1.find(from, to)) {
                    //System.out.println("type 1, from : " + from + " to" + to);
                    count++;
                } else {
                    uf1.union(from, to);
                }
            } 
            
            if (type == 2) {
                if (uf2.find(from, to)) {
                    //System.out.println("type 2, from : " + from + " to" + to);
                    count++;
                } else {
                    uf2.union(from, to);
                }
            }
        }
        
        if (uf1.findComponentNum() == uf2.findComponentNum() && uf2.findComponentNum() == 1) {
            return count;
        }
        return -1;
        
    }
    
    class UnionFind {
	
        int[] nums;

        public UnionFind(int n) {
            nums = new int[n];
            for (int i = 0; i < n; i++) {
                nums[i] = i;
            }
        }
        public void union(int x, int y) {
            int rootX = findRoot(x);
            int rootY = findRoot(y);
            nums[rootX] = rootY;
        }
        public int findRoot(int a) {
            int root = a;
            while (nums[root] != root) {
                root = nums[root];
            }
            //put all child to be the child of root
            while (a != root) {
                int tmp = nums[a];
                nums[a] = root;
                a = tmp;
            }
            return root;
        }
        public boolean find(int x, int y) {
            return findRoot(x) == findRoot(y);
        }

        public int findComponentNum() {
            Set<Integer> set = new HashSet<Integer>();
            for (int i = 0; i < nums.length; i++) {
                set.add(findRoot(i));
            }
            return set.size();
        }
    }
}
