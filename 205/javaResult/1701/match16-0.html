class Solution {
    class UF {
        int[] parents;
        int num;
        UF (int n) {
            parents = new int[n];
            num = n;
            for (int i =0; i < n; i += 1) {
                parents[i] = i;
            }
        }
        int find(int x) {
            int p = x;
            while (parents[p] != p) {
                p = parents[p];
            }
            while (x != p) {
                parents[x] = p;
                x = parents[x];
            }
            return p;
        }
        void union(int x, int y) {
            int paX = find(x);
            int paY = find(y);
            if (paX == paY) return;
            parents[paX] = paY;
            num -= 1;
        }
        boolean isConnected(int x, int y) {
            return find(x) == find(y);
        }
    }
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (a, b) -> b[0] - a[0]);
        UF uf1 = new UF(n);
        UF uf2 = new UF(n);
        int ret = 0;
        for (int[] edge : edges) {
            int type = edge[0];
            int u = edge[1] - 1;
            int v = edge[2] - 1;
            if (type == 3) {
                if (uf1.isConnected(u, v) && uf2.isConnected(u, v)) {
                    ret += 1;
                } else {
                    uf1.union(u, v);
                    uf2.union(u, v);
                }
            } else if (type == 1) {
                if (uf1.isConnected(u, v)) {
                    ret += 1;
                } else {
                    uf1.union(u, v);
                }
            } else {
                if (uf2.isConnected(u, v)) {
                    ret += 1;
                } else {
                    uf2.union(u, v);
                }
            }
        }
        if (uf1.num != 1 || uf2.num != 1) return -1;
        return ret;
    }
}
