class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int[] parent = new int[n + 1];
        for (int i = 0; i <= n; i++) parent[i] = i;
        int res = 0;
        int count = n;
        for (int[] e : edges) {
            if (e[0] == 3) {
                int u = e[1], v = e[2];
                int pu = find(parent, u);
                int pv = find(parent, v);
                if (pu == pv) {
                    res++;
                }
                else {
                    parent[pu] = pv;
                    count--;
                }
            }
        }
        for (int[] e : edges) {
            if (e[0] == 1) {
                int u = e[1], v = e[2];
                int pu = find(parent, u);
                int pv = find(parent, v);
                if (pu == pv) {
                    res++;
                }
                else {
                    parent[pu] = pv;
                    count--;
                }
            }
        } 
        //System.out.println(count);
        if (count != 1) return -1; 
        for (int i = 0; i <= n; i++) parent[i] = i;
        count = n;
        for (int[] e : edges) {
            if (e[0] == 3) {
                int u = e[1], v = e[2];
                int pu = find(parent, u);
                int pv = find(parent, v);
                if (pu == pv) {
                    ;
                }
                else {
                    parent[pu] = pv;
                    count--;
                }
            }
        }
        for (int[] e : edges) {
            if (e[0] == 2) {
                int u = e[1], v = e[2];
                int pu = find(parent, u);
                int pv = find(parent, v);
                if (pu == pv) {
                    res++;
                }
                else {
                    parent[pu] = pv;
                    count--;
                }
            }
        } 
        //System.out.println(count);
        if (count != 1) return -1; 
        return res;
    }
    private int find(int[] parent, int v) {
        if (parent[v] != v)  parent[v] = find(parent, parent[v]);
        return parent[v];
    }
}
