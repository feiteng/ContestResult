class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        List<List<int[]>> tp = new ArrayList<>();
        
        for (int i = 0; i < 3; i++) {
            List<int[]> l = new ArrayList<>();
            tp.add(l);
        }
        
        for (int[] e: edges) {
            List<int[]> type = tp.get(e[0] - 1);
            type.add(e);
        }
        
        DSU dsu = new DSU(n);
        int cnt = 0;
        
        List<int[]> tp3 = tp.get(2);
        for (int[] e: tp3) {
            int u = e[1] - 1;
            int v = e[2] - 1;
            if (!dsu.union(u, v)) {
                cnt++;
            }
        }
        
        DSU dsu1 = new DSU(dsu);
        List<int[]> tp1 = tp.get(0);
        for (int[] e: tp1) {
            int u = e[1] - 1;
            int v = e[2] - 1;
            if (!dsu1.union(u, v)) {
                cnt++;
            }
        }
        
        DSU dsu2 = new DSU(dsu);
        List<int[]> tp2 = tp.get(1);
        for (int[] e: tp2) {
            int u = e[1] - 1;
            int v = e[2] - 1;
            if (!dsu2.union(u, v)) {
                cnt++;
            }
        }
        
        for (int i = 0; i < n; i++) {
            // System.out.printf("dsu1, i: %d, find(i): %d\n", i, dsu1.find(i));
            // System.out.printf("dsu2, i: %d, find(i): %d\n", i, dsu2.find(i));
            
            if (dsu1.find(0) != dsu1.find(i)) return -1;
            if (dsu2.find(0) != dsu2.find(i)) return -1;
        }
        
        return cnt;
    }
}

class DSU {
    int n;
    int[] f;
    DSU(int n) {
        this.n = n;
        f = new int[n];
        for (int i = 0; i < n; i++) {
            f[i] = i;
        }
    }
    
    DSU(DSU dsu) {
        n = dsu.n;
        f = new int[n];
        for (int i = 0; i < n; i++) {
            f[i] = dsu.f[i];
        }
    }
    
    int find(int i) {
        if (i == f[i]) return i;
        f[i] = find(f[i]);
        return f[i];
    }
    
    boolean union(int i, int j) {
        int fi = find(i);
        int fj = find(j);
        if (fi == fj) return false;
        f[fi] = fj;
        return true;
    }
}
