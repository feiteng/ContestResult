class Solution {
    int[] parents;
    int count;
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int res = 0;
        parents = new int[n];
        count = n;
        for(int i = 0; i < n; i++) {
            parents[i] = i;
        }
        
        for(int[] edge : edges) {
            if(edge[0] == 3) {
                if(!union(edge[1]-1, edge[2]-1))
                    res++;
            }
        }
        
        int[] parentsTmp = new int[n];
        int countTmp = count;
        
        for(int i = 0; i < n; i++) {
            parentsTmp[i] = parents[i];
        }
        
        for(int[] edge : edges) {
            if(edge[0] == 1) {
                if(!union(edge[1]-1, edge[2]-1))
                    res++;
            }
        }
        if(count != 1)
            return -1;
        
        
        parents = parentsTmp;
        count = countTmp;
        for(int[] edge : edges) {
            if(edge[0] == 2) {
                if(!union(edge[1]-1, edge[2]-1))
                    res++;
            }
        }
        if(count != 1)
            return -1;
        
        return res;
    }
    
    private int find(int p) {
        if(p != parents[p])
            parents[p] = find(parents[p]);
        return parents[p];
    }
    
    private boolean union(int p, int q) {
        int rootp = find(p);
        int rootq = find(q);
        
        if(rootp == rootq)
            return false;
        
        parents[rootp] = rootq;
        count--;
        return true;
    }
}
