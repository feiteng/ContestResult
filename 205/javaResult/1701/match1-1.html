class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int[] par = new int[n];
        int sol = 0;
        
for (int i = 0; i < n; i++) par[i] = i;
        int nc = n;
        int t3 = 0;
        for (int[] e : edges) {
            if (e[0] != 3) continue;
            t3++;
            
int root1 = find(par, e[1] - 1);
            int root2 = find(par, e[2] - 1);
            if (root1 == root2) {
                sol++;
            } else {
                par[root1] = root2;
                nc--;
            }
        }
        if (nc == 1) return sol + edges.length - t3;
        
        int nc2 = nc;
        int[] par2 = new int[n];
        for (int i = 0; i < n; i++) par2[i] = par[i];
        for (int[] e : edges) {
            if (e[0] == 3) continue;
            if (e[0] == 1) {
                
int root1 = find(par, e[1] - 1);
                int root2 = find(par, e[2] - 1);
                if (root1 == root2) {
                    sol++;
                } else {
                    par[root1] = root2;
                    nc--;
                }
            } else {
                
int root1 = find(par2, e[1] - 1);
                int root2 = find(par2, e[2] - 1);
                if (root1 == root2) {
                    sol++;
                } else {
                    par2[root1] = root2;
                    nc2--;
                }
                
            }
        }
        if (nc != 1 || nc2 != 1) return -1;
        return sol;
    }
    
    public int find(int[] par, int a) {
        if (par[a] == a) return a;
        return par[a] = find(par, par[a]);
    }
}
