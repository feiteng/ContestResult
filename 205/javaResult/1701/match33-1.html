class Solution {
    
class DSU{
        int[] parent;
        int[] size;
        int components;

        public DSU(int N){
            components = N;
            parent = new int[N];
            size = new int[N];
            for(int i=0; i<parent.length; i++){
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int find(int x){
            if(parent[x]!=x){
                parent[x] = find(parent[x]);
            }
            return parent[x];
        
}

        public int union(int x, int y){
            int xParent = find(x);
            int yParent = find(y);
            if(xParent==yParent) return size[xParent];
            components--;
            if(size[xParent]<=size[yParent]){
                parent[xParent] = yParent;
                size[yParent]+=size[xParent];
                return size[yParent];
            }
            else{
                parent[yParent] = xParent;
                size[xParent] += size[yParent];
                return size[xParent];
            }
        }
    }
    
    
public int maxNumEdgesToRemove(int n, int[][] edges) {
        DSU dsu3 = new DSU(n);
        DSU dsu2 = new DSU(n);
        DSU dsu1 = new DSU(n);
        for(int[] e : edges){
            int type = e[0];
            if(type==3){
                dsu3.union(e[1]-1, e[2]-1);
                dsu2.union(e[1]-1, e[2]-1);
                dsu1.union(e[1]-1, e[2]-1);
            }
            else if(type==2){
                dsu2.union(e[1]-1, e[2]-1);
            }
            else{
                dsu1.union(e[1]-1, e[2]-1);
            }
        }
        int c1 = dsu1.components;
        int c2 = dsu2.components;
        if(c1!=1 || c2!=1){
            return -1;
        }
        int c3 = dsu3.components;
        int need = n-c3 + c3-1 + c3-1;
        return edges.length-need;
    }
}
