class Solution {
    int id1[];
    int sz1[];
    int id2[];
    int sz2[];
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        
id1=new int[n+1];
        sz1=new int[n+1];
        for(int i=1;i<=n;i++){
            id1[i]=i;
            sz1[i]=1;
        }
        id2=new int[n+1];
        sz2=new int[n+1];
        for(int i=1;i<=n;i++){
            id2[i]=i;
            sz2[i]=1;
        }
        
int ans=0;
        for(int i=0;i<edges.length;i++){
            int typ=edges[i][0];
            int u=edges[i][1];
            int v=edges[i][2];
            if(typ==3){
                if(root1(u)==root1(v) && root2(u)==root2(v)){
                    ans++;
                }else{
                    union1(u,v);
                    
union2(u,v);
                }
            }
        }
        for(int i=0;i<edges.length;i++){
            int typ=edges[i][0];
            int u=edges[i][1];
            int v=edges[i][2];
            if(typ==1){
                if(root1(u)==root1(v)){
                    ans++;
                }else{
                    union1(u,v);
                }
            }else if(typ==2){
                if(root2(u)==root2(v)){
                    ans++;
                }else{
                    union2(u,v);
                }
            }
        }
        int c1=0,c2=0;
        for(int i=1;i<=n;i++){
            if(id1[i]==i)
                c1++;
        }
        for(int i=1;i<=n;i++){
            if(id2[i]==i)
                c2++;
        }
        if(c1>1 || c2>1)return -1;
        return ans;
    }
    void union1(int A,int B)
    {
        int root_A = root1(A);
        int root_B = root1(B);
        
if(sz1[root_A] < sz1[root_B ])
        {
            id1[ root_A ] = id1[root_B];
            sz1[root_B] += sz1[root_A];
        }
        else
        {
            id1[ root_B ] = id1[root_A];
            sz1[root_A] += sz1[root_B];
        }
    }
    int root1(int i)
    {
        while(id1[ i ] != i)
        {
            id1[ i ] = id1[ id1[ i ] ] ; 
            
i = id1[ i ]; 
        }
        return i;
    }
    void union2(int A,int B)
    {
        int root_A = root2(A);
        int root_B = root2(B);
        if(sz2[root_A] < sz2[root_B ])
        {
            id2[ root_A ] = id2[root_B];
            sz2[root_B] += sz2[root_A];
        }
        else
        {
            id2[ root_B ] = id2[root_A];
            sz2[root_A] += sz2[root_B];
        }
    }
    int root2(int i)
    {
        while(id2[ i ] != i)
        {
            id2[ i ] = id2[ id2[ i ] ] ; 
            i = id2[ i ]; 
        }
        return i;
    }
}
