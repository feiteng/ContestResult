class Solution {
    
    class UF {
        int n;
        int[] parents, sizes;
        
        UF(int n) {
            this.n = n;
            parents = new int[n];
            sizes = new int[n];
            for (int i = 0; i < n; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        int find(int i) {
            while (parents[i] != i) {
                parents[i] = parents[parents[i]];
                i = parents[i];
            }
            return i;
        }
        
        boolean union(int p, int q) {
            int pRoot = find(p), qRoot = find(q);
            if (pRoot == qRoot) {
                return false;
            } else {
                parents[pRoot] = qRoot;
                sizes[qRoot] += sizes[pRoot];
                return true;
            }
        }
        
        boolean isComplete() {
            int max = 0;
            for (int size : sizes) {
                max = Math.max(max, size);
            }
            return max == n;
        }
    }
    
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        List<int[]> list1 = new ArrayList<>(), list2 = new ArrayList<>(), list3 = new ArrayList<>();
        for (int[] edge : edges) {
            if (edge[0] == 1) {
                list1.add(new int[] {edge[1], edge[2]});
            } else if (edge[0] == 2) {
                list2.add(new int[] {edge[1], edge[2]});
            } else {
                list3.add(new int[] {edge[1], edge[2]});
            }
        }
        int ans = 0;
        UF ufAlice = new UF(n), ufBob = new UF(n);
        for (int[] edge : list3) {
            int p = edge[0]-1, q = edge[1]-1;
            boolean flag = true;
            flag = ufAlice.union(p, q);
            flag = ufBob.union(p, q);
            if (!flag) {
                ans++;
            }
        }
        for (int[] edge : list1) {
            int p = edge[0]-1, q = edge[1]-1;
            if (!ufAlice.union(p, q)) {
                ans++;
            }
        }
        for (int[] edge : list2) {
            int p = edge[0]-1, q = edge[1]-1;
            if (!ufBob.union(p, q)) {
                ans++;
            }
        }
        if (ufAlice.isComplete() && ufBob.isComplete()) {
            return ans;
        } else {
            return -1;
        }
    }
}
