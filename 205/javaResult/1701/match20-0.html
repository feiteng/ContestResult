class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (a, b) -> a[0] == 3 ? -1 : (b[0] == 3 ? 1 : 0));
        UnionFind uf1 = new UnionFind(n);
        UnionFind uf2 = new UnionFind(n);
        int count = 0;
        for(int[] e : edges) {
            if(e[0] == 3) {
                if(uf1.isConnected(e[1] - 1, e[2] - 1) && uf2.isConnected(e[1] - 1, e[2] - 1)){
                    count ++;
                }
                else{
                    uf1.union(e[1] - 1, e[2] - 1);
                    uf2.union(e[1] - 1, e[2] - 1);
                }
            }else if(e[0] == 2) {
                if(uf2.isConnected(e[1] - 1, e[2] - 1)) {
                    count ++;
                }else{
                    uf2.union(e[1] - 1, e[2] - 1);
                }
            }else if(e[0] == 1) {
                if(uf1.isConnected(e[1] - 1, e[2] - 1)) {
                    count ++;
                }else{
                    uf1.union(e[1] - 1, e[2] - 1);
                }
            }
        }
        if(uf1.getCount() != 1 || uf2.getCount() != 1)
            return -1;
        return count;
    }
}
class UnionFind{
    private int[] parent;
    private int[] rank; //用于压缩路径的
    private int count; //连通分量
    public UnionFind(int size){
        parent = new int[size];
        rank = new int[size];
        for (int i = 0; i < size; i++) {
            parent[i] = i;
            rank[i] = 1;
        }
        count = size;
    }
    private int find(int p){
        if (p != parent[p]) //路径压缩
            parent[p] = find(parent[p]);
        return parent[p];
    }
    public boolean isConnected(int p, int q) {
        return find(p) == find(q);
    }
    public int getCount() {
        return count;
    }
    public void union(int p, int q) {
        int pRoot = find(p);
        int qRoot = find(q);
        if(pRoot == qRoot)
            return;
        count--; //每合并一次减一
        if(rank[pRoot] < rank[qRoot])
            parent[pRoot] = qRoot;
        else if(rank[pRoot] > rank[qRoot])
            parent[qRoot] = pRoot;
        else {
            parent[qRoot] = pRoot;
            rank[pRoot] += 1;
        }
    }
}
