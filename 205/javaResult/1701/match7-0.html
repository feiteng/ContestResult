class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        HashSet<Integer>[] adj1 = new HashSet[n], adj2 = new HashSet[n], adj3 = new HashSet[n];
        for (int i = 0; i < n; i++) {
            adj1[i] = new HashSet<>();
            adj2[i] = new HashSet<>();
            adj3[i] = new HashSet<>();
        }
        for (int[] edge : edges) {
            if (edge[0] == 1) {
                adj1[edge[1] - 1].add(edge[2] - 1);
                adj1[edge[2] - 1].add(edge[1] - 1);
            } else if (edge[0] == 2) {
                adj2[edge[1] - 1].add(edge[2] - 1);
                adj2[edge[2] - 1].add(edge[1] - 1);
            } else {
                adj3[edge[1] - 1].add(edge[2] - 1);
                adj3[edge[2] - 1].add(edge[1] - 1);
            }
        }
        if (search(n, adj1, adj3, new boolean[n], 0) < n) {
            return -1;
        }
        if (search(n, adj2, adj3, new boolean[n], 0) < n) {
            return -1;
        }
        int count1 = 0, count2 = 0, count3 = 0;
        for (int i = 0; i < n; i++) {
            count1 += adj1[i].size();
            count2 += adj2[i].size();
            count3 += adj3[i].size();
        }
        count1 /= 2;
        count2 /= 2;
        count3 /= 2;
        boolean[] visited = new boolean[n];
        int connect = 0;
        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                connect += search(n, adj3, visited, i);
            }
        }
        int remove = count3 - connect;
        int res = -remove;
        res += count1 + count3 - (n - 1);
        res += count2 + count3 - (n - 1);
        return res;
    }
    public int search(int n, HashSet<Integer>[] adj1, HashSet<Integer>[] adj2, boolean[] visited, int i) {
        int size = 1;
        visited[i] = true;
        for (int j : adj1[i]) {
            if (!visited[j]) {
                size += search(n, adj1, adj2, visited, j);
            }
        }
        for (int j : adj2[i]) {
            if (!visited[j]) {
                size += search(n, adj1, adj2, visited, j);
            }
        }
        return size;
    }
    public int search(int n, HashSet<Integer>[] adj, boolean[] visited, int i) {
        int res = 0;
        visited[i] = true;
        for (int j : adj[i]) {
            if (!visited[j]) {
                res += search(n, adj, visited, j);
                res++;
            }
        }
        return res;
    }
}
