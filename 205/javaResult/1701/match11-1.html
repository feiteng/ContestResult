class Solution {

    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UnionFind uf1 = new UnionFind(n), uf2 = new UnionFind(n), uf3 = new UnionFind(n);
        int count = 0;
        for (int[] edge : edges) {
            if (edge[0] == 3) {
                if (uf3.find(edge[1] - 1) != uf3.find(edge[2] - 1)) {
                    uf1.union(edge[1] - 1, edge[2] - 1);
                    uf2.union(edge[1] - 1, edge[2] - 1);
                    uf3.union(edge[1] - 1, edge[2] - 1);
                } else {
                    count++;
                
}
            }
        }
        for (int[] edge : edges) {
            if (edge[0] == 1) {
                if (uf1.find(edge[1] - 1) != uf1.find(edge[2] - 1)) {
                    uf1.union(edge[1] - 1, edge[2] - 1);
                } else {
                    count++;
                
}
            } else if (edge[0] == 2) {
                if (uf2.find(edge[1] - 1) != uf2.find(edge[2] - 1)) {
                    uf2.union(edge[1] - 1, edge[2] - 1);
                } else {
                    count++;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            if (uf1.find(0) != uf1.find(i) || uf2.find(0) != 
uf2.find(i)) {
                return -1;
            }
        }
        return count;
    }

    class UnionFind {
        private int count = 0;
        private int[] parent, rank;

        public UnionFind(int n) {
            count = n;
            parent = new int[n];
            rank = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
            
}
        }

        public int find(int p) {
            while (p != parent[p]) {
                parent[p] = parent[parent[p]];
                
p = parent[p];
            }
            return p;
        }

        public void union(int p, int q) {
            int rootP = find(p);
            int rootQ = find(q);
            if (rootP == rootQ)
                return;
            if (rank[rootQ] > rank[rootP]) {
                parent[rootP] = rootQ;
            } else {
                parent[rootQ] = rootP;
                if (rank[rootP] == rank[rootQ]) {
                    rank[rootP]++;
                }
            }
            count--;
        }

        public int count() {
            return count;
        }
    }
}
