class UnionFind {
    int[] roots;
    int count;

    public UnionFind(int n) {
        roots = new int[n];
        count = n;
        for (int i = 0; i < n; i++) {
            roots[i] = i;
        }
    }

    int find(int i) {
        if (i == roots[i]) {
            return i;
        }
        roots[i] = find(roots[i]);
        return roots[i];
    }


    boolean union(int p, int q) {
        int pRoot = find(p);
        int qRoot = find(q);
        if (pRoot != qRoot) {
            roots[pRoot] = qRoot;
            count -= 1;
            return true;
        }
        return false;
    }
}

class Solution {
    int ans = 0;
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UnionFind unionFind = new UnionFind(n);
        UnionFind unionFind2 = new UnionFind(n);
        ans = edges.length;
        for (int[] edge : edges) {
            int type = edge[0];
            int u = edge[1] - 1;
            int v = edge[2] - 1;
            if (type == 3) {
                if (unionFind.union(u, v)) {
                    ans -= 1;
                }
                unionFind2.union(u, v);
            }
        }
        for (int[] edge : edges) {
            int type = edge[0];
            int u = edge[1] - 1;
            int v = edge[2] - 1;
            if (type == 1) {
                if (unionFind.union(u, v)) {
                    ans -= 1;
                };
            } else if (type == 2) {
                if (unionFind2.union(u, v)) {
                    ans -= 1;
                }
            }
        }
        if (unionFind.count != 1 || unionFind2.count != 1) {
            return -1;
        }
        return ans;
    }
}
