class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (a, b) -> b[0] - a[0]);
        UnionFind uf1 = new UnionFind(n);
        UnionFind uf2 = new UnionFind(n);
        
        int share = 0, alice = 0, bob = 0;
        int t1 = 0, t2 = 0, t3 = 0;
        
        for (int[] edge : edges) {
            int type = edge[0];
            if (type == 3) {
                t3++;
                if (uf1.union(edge[1] - 1, edge[2] - 1)) {
                    uf2.union(edge[1] - 1, edge[2] - 1);
                    share++;
                }
            } else if (type == 1) {
                t1++;
                if (uf1.union(edge[1] - 1, edge[2] - 1)) {
                    alice++;
                }
            } else {
                t2++;
                if (uf2.union(edge[1] - 1, edge[2] - 1)) {
                    bob++;
                }
            }
        }
        
        if (uf1.count != 1 || uf2.count != 1) return -1;
        
        return (t3 - share) + (t1 - alice) + (t2 - bob);
    }
    
    
}

class UnionFind{
    final int[] parents;
    int count;
    
    public UnionFind(int n){
        this.parents = new int[n];
        reset();
    }
    
    public void reset(){
        for(int i =0;i<parents.length;i++){
            parents[i] = i;
        }
        count = parents.length;
    }
    
    public int find(int i){
        int parent = parents[i];
        if(parent == i){
            return i;
        }else{
            int root = find(parent);
            parents[i] = root;
            return root;
        }
    }
    
    public boolean union(int i, int j){
        int r1 = find(i);
        int r2 = find(j);
        if(r1 != r2){
            count--;
            parents[r1] = r2;
            return true;
        }else{
            return false;
        }
    }
    
}
