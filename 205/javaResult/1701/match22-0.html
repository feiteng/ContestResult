class Solution {
    int[] label;
    int[] count;
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        label = new int[n];
        count = new int[n];
        for (int i = 0; i < n; i++) {
            label[i] = i;
            count[i] = 1;
        }
        
        int ans = 0;
        
        //for type 3:
        for (int[] edge: edges) {
            if (edge[0] != 3) {
                continue;
            }
            int i = edge[1]-1;
            int j = edge[2]-1;
            
            int pi = find(label, i);
            int pj = find(label, j);
            //already connected
            if (pi == pj) {
                ans++;
            } else {
                if (count[pi] > count[pj]) {
                    label[pj] = pi;
                    count[pi] += count[pj];
                } else {
                    label[pi] = pj;
                    count[pj] += count[pi];
                }
            }
        }
        
        int[] label1 = new int[n];
        int[] label2 = new int[n];
        int[] count1 = new int[n];
        int[] count2 = new int[n];
        for (int i = 0; i < n; i++) {
            label1[i] = label[i];
            label2[i] = label[i];
            
            count1[i] = 1;
            count2[i] = 1;
        }
        
        //for type 1:
        for (int[] edge: edges) {
            if (edge[0] != 1) {
                continue;
            }
            int i = edge[1]-1;
            int j = edge[2]-1;
            int pi = find(label1, i);
            int pj = find(label1, j);
            //already connected
            if (pi == pj) {
                ans++;
            } else {
                if (count1[pi] > count1[pj]) {
                    label1[pj] = pi;
                    count1[pi] += count1[pj];
                } else {
                    label1[pi] = pj;
                    count1[pj] += count1[pi];
                }
            }
        }
        
        //for type 2:
        for (int[] edge: edges) {
            if (edge[0] != 2) {
                continue;
            }
            int i = edge[1]-1;
            int j = edge[2]-1;
            int pi = find(label2, i);
            int pj = find(label2, j);
            //already connected
            if (pi == pj) {
                ans++;
            } else {
               if (count2[pi] > count2[pj]) {
                    label2[pj] = pi;
                    count2[pi] += count2[pj];
                } else {
                    label2[pi] = pj;
                    count2[pj] += count2[pi];
                }
            }
        }
        
        //check fully connected
        int p1 = find(label1, 0);
        int p2 = find(label2, 0);
        for (int i = 1; i < n; i++) {
            if (p1 != find(label1, i)) {
                return -1;
            }
            if (p2 != find(label2, i)) {
                return -1;
            }
        }
        return ans;
        
    }
    
    private int find(int[] label, int i) {
        while(label[i] != i) {
            i = label[i];
        }
        return i;
    }
    
    private void union(int[] label, int i, int j) {
        int pi = find(label, i);
        int pj = find(label, j);
        label[pi] = pj;
    }
}
