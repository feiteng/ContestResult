class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        ArrayList<Integer>[] als1=new ArrayList[n], als2=new ArrayList[n], als3=new ArrayList[n];
        for(int i=0;i<n;++i){
            als1[i]=new ArrayList();
            als2[i]=new ArrayList();
            als3[i]=new ArrayList();
        }
        for(int[] e : edges){
            if(e[0]==1){
                als1[e[1]-1].add(e[2]-1);
                als1[e[2]-1].add(e[1]-1);
            }else if(e[0]==2){
                als2[e[1]-1].add(e[2]-1);
                als2[e[2]-1].add(e[1]-1);
            }else{
                als3[e[1]-1].add(e[2]-1);
                als3[e[2]-1].add(e[1]-1);
            }
        }
        int[] kk=new int[1];
        check(als1,als3,new boolean[n],0,kk);
        if(kk[0]!=n)return -1;
        kk[0]=0;
        check(als2,als3,new boolean[n],0,kk);
        if(kk[0]!=n)return -1;
        boolean[] f=new boolean[n];
        int count=0;
        
for(int i=0;i<n;++i){
            if(!f[i]){
                ++count;
                todo(als3,f,i);
            }
        }
        return edges.length - (n-1+(count-1));
    }
    
    private void todo(ArrayList<Integer>[] als, boolean[] f, int st){
        f[st]=true;
        for(int n : als[st]){
            if(!f[n])todo(als, f, n);
        }
    
}
    
    private void check(ArrayList<Integer>[] als1, ArrayList<Integer>[] als2, boolean[] f, int st, int[] kk){
        f[st]=true;
        ++kk[0];
        for(int n : als1[st]){
            if(!f[n])check(als1, als2, f, n, kk);
        }
        for(int n : als2[st]){
            if(!f[n])check(als1, als2, f, n, kk);
        }
    }
}
