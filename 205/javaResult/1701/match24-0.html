class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (a, b) -> (b[0] - a[0]));
        int[] roots1 = new int[n + 1];
        int[] roots2 = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            roots1[i] = i;
            roots2[i] = i;
        }
        Set<String> set1 = new HashSet<>();
        Set<String> set2 = new HashSet<>();
        int n1 = n, n2 = n;
        int res = 0;
        for(int[] e : edges) {
            //System.out.println(e[0] + ":" + e[1] + ":" + e[2]);
            if(e[0] == 1) {
                int root_a = find(e[1], roots1);
                int root_b = find(e[2], roots1);
                /*
                System.out.println("root_a1 = " + root_a);
                System.out.println("root_b1 = " + root_b);
                */
                if(root_a == root_b) {
                    res++;
                } else {
                    roots1[root_a] = root_b;
                    n1--;
                }
            } else if(e[0] == 2) {
                int root_a = find(e[1], roots2);
                int root_b = find(e[2], roots2);
                /*
                System.out.println("root_a2 = " + root_a);
                System.out.println("root_b2 = " + root_b);
                */
                if(root_a == root_b) {
                    res++;
                } else {
                    roots2[root_a] = root_b;
                    n2--;
                }
            } else {
                int root_a1 = find(e[1], roots1);
                int root_b1 = find(e[2], roots1);
                int root_a2 = find(e[1], roots2);
                int root_b2 = find(e[2], roots2);
                /*
                System.out.println("root_a1 = " + root_a1);
                System.out.println("root_b1 = " + root_b1);
                System.out.println("root_a2 = " + root_a2);
                System.out.println("root_b2 = " + root_b2);
                */
                if(root_a1 != root_b1) {
                    roots1[root_a1] = root_b1;
                    n1--;
                }
                if(root_a2 != root_b2) {
                    roots2[root_a2] = root_b2;
                    n2--;
                }
                if(root_a1 == root_b1 && root_a2 == root_b2) {
                    res++;
                }
            }
        }
        //System.out.println(n1 + " " + n2);
        if(n1 != 1 || n2 != 1) return -1;
        return res;
    }
    
    private int find(int i, int[] roots) {
        int j = i;
        while(roots[i] != i) {
            i = roots[i];
        }
        roots[j] = i;
        return i;
    }
}
