class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int res = 0;
        Arrays.sort(edges, (a, b) -> b[0] - a[0]);
        //for (int[] e : edges) System.out.println(e[0]);
        UF uf = new UF(n + 2);
        for (int[] e :edges) {
            if (e[0] != 1) {
                if (!uf.union(e[1], e[2])) {
                    //System.out.println(e[0] + "," + e[1] + "," + e[2]);
                    res++;
                }
            }
        }
        
        Set<Integer> set = new HashSet<>();
        for (int i = 1; i <= n; i++) {
            set.add(uf.find(i));
        }
        if (set.size() > 1) return -1;
        
        uf = new UF(n);
        for (int[] e :edges) {
            if (e[0] != 2) {
                if (!uf.union(e[1], e[2]) && e[0] == 1) {
                    res++;
                }
            }
        }
        set = new HashSet<>();
        for (int i = 1; i <= n; i++) {
            set.add(uf.find(i));
        }
        if (set.size() > 1) return -1;
        
        return res;
    }
}


//unionFind模板（数组版本， 如果要map版本可以参考LC 128的map union find解）*****************************************************************************************
class UF {
    private int[] parents_;
    private int[] ranks_;
 
    public UF(int n) {
        parents_ = new int[n + 1];
        ranks_ = new int[n + 1];
        for (int i = 0; i < parents_.length; ++i) {
            parents_[i] = i;
            ranks_[i] = 1;
        }
    }
 
    public boolean union(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (pu == pv) return false;
 
        if (ranks_[pv] > ranks_[pu])
            parents_[pu] = pv;           
        else if (ranks_[pu] > ranks_[pv])
            parents_[pv] = pu;
        else {
            parents_[pv] = pu;
            ranks_[pu] += 1;
        }
 
        return true;
    }
 
    public int find(int u) {
        while (parents_[u] != u) {
            parents_[u] = parents_[parents_[u]];
            u = parents_[u];
        }
        return u;
    }
}
