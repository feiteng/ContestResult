class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UF uf1 = new UF(n + 1);
        UF uf2 = new UF(n + 1);
        int count = 0;
        for (int[] e : edges) {
            if (e[0] == 3) {
                int a = e[1], b = e[2];
                if (!uf1.connected(a, b)) {
                    count++;
                    uf1.union(a, b);
                    uf2.union(a, b);
                }
            }
        }
        
        for (int[] e : edges) {
            if (e[0] == 1) {
                int a = e[1], b = e[2];
                if (!uf1.connected(a, b)) {
                    count++;
                    uf1.union(a, b);
                }
            }
        }
        
        for (int[] e : edges) {
            if (e[0] == 2) {
                int a = e[1], b = e[2];
                if (!uf2.connected(a, b)) {
                    count++;
                    uf2.union(a, b);
                }
            }
        }
        
        if (!uf1.allConnected() || !uf2.allConnected()) {
            return -1;
        } else {
            return edges.length - count;
        }
    }
    
    private static class UF {
        int p[];
        int count;
        
        UF(int n) {
            p = new int[n];
            Arrays.fill(p, -1);
            count = n - 1;
        }
        
        int findRoot(int a) {
            if (p[a] == -1) {
                return a;
            }
            return p[a] = findRoot(p[a]);
        }
        
        boolean connected(int a, int b) {
            return findRoot(a) == findRoot(b);
        }
        
        void union(int a, int b) {
            int pa = findRoot(a);
            int pb = findRoot(b);
            if (pa != pb) {
                p[pa] = pb;
                count--;
            }
        }
        
        boolean allConnected() {
            return count == 1;
        }
    }
}
