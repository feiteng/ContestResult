class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        //check if you can reach every node, if not return -1;
        DSU dsu1 = new DSU(n + 1);
        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 1 || type == 3) {
                dsu1.union(start, end);
            }
        }
        for (int node = 2; node <= n; node++) {
            if (dsu1.find(node) != dsu1.find(node - 1)) return -1;
        }
        
        
        DSU dsu2 = new DSU(n + 1);
        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 2 || type == 3) {
                dsu2.union(start, end);
            }
        }
        for (int node = 2; node <= n; node++) {
            if (dsu2.find(node) != dsu2.find(node - 1)) return -1;
        }
        
        
        //construct two graph using type3
        
        //dsu3 for check type1
        DSU dsu3 = new DSU(n + 1);
        int count = 0;
        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 3) {
                if (dsu3.find(start) == dsu3.find(end)) {
                    count++;
                } else {
                    dsu3.union(start, end);
                }
            }
        }
        

        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 1) {
                if (dsu3.find(start) == dsu3.find(end)) {
                    count++;
                } else {
                    dsu3.union(start, end);
                }
            }
        }
        
        // System.out.println(count);
        //dsu4 for check type2
        DSU dsu4 = new DSU(n + 1);
        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 3) {
                if (dsu4.find(start) != dsu4.find(end)) {
                    dsu4.union(start, end);
                }
            }
        }
        
        for (int[] edge : edges) {
            int type = edge[0], start = edge[1], end = edge[2];
            if (type == 2) {
                if (dsu4.find(start) == dsu4.find(end)) {
                    count++;
                } else {
                    dsu4.union(start, end);
                }
            }
        }

        
        
        //check if you connect two nodes in same dsu, using type1 or type2;
        return count;
    }
}    

class DSU {
    int[] parent;
    public DSU(int N) {
        parent = new int[N];
        for (int i = 0; i < N; i++) parent[i] = i;
    }
    public int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    public void union(int x, int y) {
        parent[find(x)] = find(y);
    }
}
