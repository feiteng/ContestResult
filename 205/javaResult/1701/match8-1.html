import java.util.Arrays;

class Solution {
	int ans[] = new int[4];

	public int maxNumEdgesToRemove(int n, int[][] edges) {
		init(n);
		calc(3, edges);
		calc(2, edges);
		if (!connected())
			return -1;
		init(n);
		calc(3, edges);
		calc(1, edges);
		if (!connected())
			return -1;
		return ans[3]/2+ans[1]+ans[2];

	}

	private boolean connected() {
		int cnt = 0;
		for (int i = 1; i < p.length; i++)
			if (p[i] == i)
				cnt++;
		return cnt == 1;
	}

	private void calc(int t, int[][] edges) {
		for (int i = 0; i < edges.length; i++)
			if (edges[i][0] == t) {
				connect(t, edges[i][1], edges[i][2]);
			}
	}

	public static void tr(Object... o) {
		System.out.println(Arrays.deepToString(o));
	}

	int p[];
	int size[];

	private void init(int n) {
		p = new int[n + 1];
		for (int i = 0; i <= n; i++)
			p[i] = i;
		size = new int[n + 1];
		Arrays.fill(size, 1);
	}

	private void connect(int t, int u, int v) {
		int u0 = find(u);
		int v0 = find(v);
		if (u0 == v0) {
			ans[t]++;
			return;
		}
		p[u0] = v0;
		size[v0] += size[u0];
	}

	private int find(int x) {
		if (p[x] == x)
			return x;
		else
			return p[x] = find(p[x]);
	}

}
