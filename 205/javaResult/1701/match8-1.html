class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Ufs ufs = new Ufs(n);
        int m = edges.length;
        int count = 0;
        for(int i = 0; i < m; i++){
            if(edges[i][0] == 3){
                if(ufs.find(edges[i][1]) != ufs.find(edges[i][2])){
                    ufs.union(edges[i][1], edges[i][2]);
                }else{
                    count++;
                }
            }
        }
        for(int i = 0; i < m; i++){
            if(edges[i][0] == 1){
                if(ufs.find(edges[i][1]) != ufs.find(edges[i][2])){
                    ufs.union(edges[i][1], edges[i][2]);
                }else
{
                    count++;
                }
            }
        }
        
        Set<Integer> set = new HashSet<>();
        for(int i = 1; i <= n; i++){
            set.add(ufs.find(i));
        }
        
        if(set.size() > 1) return -1;
        
        ufs = new Ufs(n);
        
        for(int i = 0; i < m; i++){
            if(edges[i][0] == 3){
                if(ufs.find(edges[i][1]) != ufs.find(edges[i][2])){
                    ufs.union(edges[i][1], edges[i][2]);
                }
            }
        }
        for(int i = 0; i < m; i++){
            if(edges[i][0] == 2){
                if(ufs.find(edges[i][1]) != ufs.find(edges[i][2])){
                    ufs.union(edges[i][1], edges[i][2]);
                }else
{
                    count++;
                }
            }
        }
        
        set = new HashSet<>();
        for(int i = 1; i <= n; i++){
            set.add(ufs.find(i));
        }
        
        if(set.size() > 1) return -1;
        
        return count;
        
    }
}

class Ufs{
    int[] parent;
    int[] ranks;
    
    public Ufs(int n){
        this.parent = new int[n + 1];
        this.ranks = new int[n + 1];
        for(int i = 0; i <= n; i++){
            parent[i] = i;
        }
    }
    
    public int find(int x){
        if(parent[x] != x)
            parent[x] = find(parent[x]);
        return parent[x];
    
}
    
    public void union(int x, int y){
        int px = find(x);
        int py = find(y);
        
        if(ranks[px] > ranks[py]) parent[py] = px;
        if(ranks[px] < ranks[py]) parent[px] = py;
        if(ranks[px] == ranks[py]){
            parent[py] = px;
            ranks[px]++;
        }
        
    }
}
