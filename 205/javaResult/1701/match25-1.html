class Solution {
    private int parent[];
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int ans = 0;
        parent = new int[n+1];
        for(int i = 1; i<=n; i++) parent[i] = i;
        int m = edges.length;
        for(int i = 0; i<m; i++){
            if(edges[i][0] == 3){
                int x = edges[i][1], y =edges[i][2];
                int p1 = getParent(x), p2 = getParent(y);
                if(p1 == p2){
                    ans ++;
                    continue;
                }
                parent[p1] = p2;
            }
        }
        int[] parent1 = new int[n+1];
        for(int i = 1; i<=n ; i++) {
            parent1[i] = parent[i];
            //System.out.println(parent[i]);
        }
        for(int i = 0; i<m; i++){
            if(edges[i][0] == 2){
                int x = edges[i][1], y =edges[i][2];
                int p1 = getParent(x), p2 = getParent(y);
                if(p1 == p2){
                    ans ++;
                    continue;
                }
                parent[p1] = p2;
            }
        }
        HashSet<Integer> hs = new HashSet<>();
        for(int i = 1; i<=n;i++){
            hs.add(getParent(i));
        }
        if(hs.size()>1) return -1;
        for(int i = 1; i<=n ; i++) parent[i] = parent1[i];
        for(int i = 0; i<m; i++){
            if(edges[i][0] == 1){
                int x = edges[i][1], y =edges[i][2];
                int p1 = getParent(x), p2 = getParent(y);
                if(p1 == p2){
                    ans ++;
                    continue;
                }
                parent[p1] = p2;
            }
        }
        hs = new HashSet<>();
        for(int i = 1; i<=n;i++){
            hs.add(getParent(i));
        }
        if(hs.size()>1) return -1;
        return ans;
    }
    public int getParent(int x){
        if(parent[x] == x) return x;
        return parent[x] = getParent(parent[x]);
    }
    
}
