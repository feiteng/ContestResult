class UnionFind {
    int[] parent;
    int[] rank;
    int numComponents;
    
    public UnionFind(int numElements) {
        parent = new int[numElements];
        rank = new int[numElements];
        numComponents = numElements;
        for (int i = 0; i < numElements; i++) {
            parent[i] = i;
            
rank[i] = 0;
        }
    }
    
    public void unify(int element1, int element2) {
        int root1 = find(element1);
        int root2 = find(element2);
        
        if (root1 == root2) return;
        
        numComponents--;
        if (rank[root1] <= rank[root2]) {
            parent[root1] = root2;
            if (rank[root1] == rank[root2])
                rank[root2]++;
        } else {
            parent[root2] = root1;
        }
    
}
    
    public boolean areInSameComponent(int element1, int element2) {
        return find(element1) == find(element2);
    }
    
    public int find(int element) {
        int temp = element;
        while (temp != parent[temp])
            temp = parent[temp];
        compressPath(element, temp);
        
        return temp;
    }
    
    private void compressPath(int element, int root) {
        while (element != root) {
            int parent = this.parent[element];
            this.parent[element] = root;
            element = parent;
        }
    }
    
    public int getNumComponents() {
        return this.numComponents;
    }
}


class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UnionFind ufAlice = new UnionFind(n+1);
        UnionFind ufBob = new UnionFind(n+1);
        int count = 0;
        
        for (int[] edge: edges) {
            int type = edge[0];
            int from = edge[1];
            int to = edge[2];
            if (type == 3) {
                if (ufAlice.areInSameComponent(from, to) &&
                    ufBob.areInSameComponent(from, to))
                    count++;
                else 
{
                    ufAlice.unify(from, to);
                    ufBob.unify(from, to);
                }
            }
        }
        
        for (int[] edge: edges) {
            int type = edge[0];
            int from = edge[1];
            int to = edge[2];
            if (type == 1) {
                if (ufAlice.areInSameComponent(from, to))
                    count++;
                else
                    ufAlice.unify(from, to);
            } 
            
            if (type == 2) {
                if (ufBob.areInSameComponent(from, to))
                    count++;
                else
                    ufBob.unify(from, to);
            }
        }
        
        
        
        
        if (ufAlice.getNumComponents() == 2 &&
            ufBob.getNumComponents() == 2)
            return count;
        return -1;
    }
}
