class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UnionFind alice = new UnionFind(n);
        UnionFind bob = new UnionFind(n);
        
        int total = 0;
        for (int[] edge: edges) {
            int type = edge[0];
            int a = edge[1];
            int b = edge[2];
            
            if(type == 3) {
                int aap = alice.getParent(a);
                int abp = alice.getParent(b);
                int bap = bob.getParent(a);
                int bbp = bob.getParent(b);
                if(aap == abp && bap == bbp) {
                    total++;
                } 
                
                if(aap != abp) {
                    alice.merge(a, b);
                }
                
                if(bap != bbp) {
                    bob.merge(a, b);
                }
            }
        }
        
        for (int[] edge: edges) {
            int type = edge[0];
            int a = edge[1];
            int b = edge[2];
        
            if (type == 1) {
                    if(alice.getParent(a) == alice.getParent(b)) {
                        total++;
                    } else {
                        alice.merge(a, b);
                    }
                } else if (type == 2) {
                    if(bob.getParent(a) == bob.getParent(b)) {
                        total++;
                    } else {
                        bob.merge(a, b);
                    }
                } 
        }
        
        
        return check(alice, n) && check(bob, n) ? total : -1;
        
    }
    
    private boolean check(UnionFind f, int n) {
        int parent = f.getParent(1);
        for (int i = 2; i <= n; i++) {
            if (f.getParent(i) != parent) {
                return false;
            }
        }
        return true;
    }
    
    class UnionFind {
        int n;
        int[] parent;
        int size[];
        UnionFind(int n) {
            this.n = n;    
            parent = new int[n + 1];
            size = new int[n + 1];
            
            for (int i = 1; i < n ;i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }
        
        public int getParent(int n) {
            while(parent[n] != n) {
                n = parent[n];
            }
            
            return n;
        }
        
        public void merge(int a, int b) {
            int aP = getParent(a);
            int bP = getParent(b);
            
            if (aP != bP) {
                if (size[aP] >= size[bP]) {
                    size[aP] += size[bP];
                    parent[bP] = aP;
                } else {
                    size[bP] += size[aP];
                    parent[aP] = bP;
                }
            }
        }
    }
}
