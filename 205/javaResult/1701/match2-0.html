class Solution {
    int res;
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        res = 0;
        int cnt3 = 0;
        int[] parent1 = new int[n];
        int[] parent2 = new int[n];
        Arrays.fill(parent1, -1);
        Arrays.fill(parent2, -1);
        for(int[] edge : edges){
            if(edge[0] == 3){
                union(parent1, edge[1] - 1, edge[2] - 1);
                cnt3++;
            }
        }
        int total = cnt3 - res;
        for(int i = 0; i < n; i++) parent2[i] = parent1[i];
        for(int[] edge : edges){
            if(edge[0] == 1){
                union(parent1, edge[1] - 1, edge[2] - 1);
            }
            if(edge[0] == 2){
                union(parent2, edge[1] - 1, edge[2] - 1);
            }
        }
        total += edges.length - res;
        System.out.println(total);
        if(total != 2 * n - 2) return -1;        
        return res;
    }
    int find(int parent[], int i) {
        if(parent[i] != -1)
        {
            parent[i] = find(parent, parent[i]);
        return parent[i];
        }
        else return i;
    }

    void union(int parent[], int x, int y) {
        int xset = find(parent, x);
        int yset = find(parent, y);
        if (xset != yset)
            parent[xset] = yset;
        else
            res++;
    }
}
