class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int res = 0;
        UnionFind uf1 = new UnionFind(n);
        UnionFind uf2 = new UnionFind(n);
        for (int[] e : edges) {
            if (e[0] == 3) {
                if (!uf1.union(e[1], e[2])) {
                    res++;
                }
                
uf2.union(e[1], e[2]);
            }
        }
        for (int[] e : edges) {
            if (e[0] == 1 && !uf1.union(e[1], e[2])) {
                res++;
            }
            if (e[0] == 2 && !uf2.union(e[1], e[2])) {
                res++;
            
}
        }
        return uf1.parts == 1 && uf2.parts == 1 ? res : -1;
    }
    private static class UnionFind {
        
        int parts;
        int[] parent;
        
        public UnionFind(int n) {
            parts = n;
            parent = new int[n + 1];
            for (int i = 1; i <= n; i++) {
                parent[i] = i;
            }
        
}
        
        // returns: true if union successful, false if i and j already have the same parent
        public boolean union(int i, int j) {
            int rt1 = findRoot(i);
            int rt2 = findRoot(j);
            if (rt1 == rt2) {
                return false;
            }
            parts--;
            parent[rt1] = rt2;
            return true;
        }
        
        private int findRoot(int i) {
            while (parent[i] != i) {
                int par = parent[i];
                parent[i] = parent[par];
                i = par;
            }
            return i;
        }
    }
}
