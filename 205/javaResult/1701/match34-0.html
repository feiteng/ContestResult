class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int[] parent = new int[n + 1];
        for (int i = 0; i < parent.length; i++) {
            parent[i] = i;
        }
        int res = 0;
        for (int i = 0; i < edges.length; i++) {
            if (edges[i][0] == 3) {
                int one = edges[i][1];
                int two = edges[i][2];
                int rootOne = getParent(parent, one);
                int rootTwo = getParent(parent, two);
                if (rootOne == rootTwo) {
                    res++;
                } else {
                    parent[rootOne] = rootTwo;
                }
            }
        }
        int[] parentCopy = new int[n + 1];
        for (int i = 0; i < parentCopy.length; i++) {
            parentCopy[i] = parent[i];
        }
        for (int i = 0; i < edges.length; i++) {
            if (edges[i][0] == 1) {
                int one = edges[i][1];
                int two = edges[i][2];
                int rootOne = getParent(parentCopy, one);
                int rootTwo = getParent(parentCopy, two);
                if (rootOne == rootTwo) {
                    res++;
                } else {
                    parentCopy[rootOne] = rootTwo;
                }
            }
        }
        Set<Integer> set = new HashSet<>();
        int count = 0;
        for (int i = 1; i < parentCopy.length; i++) {
            int par = getParent(parentCopy, i);
            if (i != par) {
                count++;
                set.add(par);
            }
        }
        if (count + set.size() < n) {
            return -1;
        }
        parentCopy = new int[n + 1];
        for (int i = 0; i < parentCopy.length; i++) {
            parentCopy[i] = parent[i];
        }
        for (int i = 0; i < edges.length; i++) {
            if (edges[i][0] == 2) {
                int one = edges[i][1];
                int two = edges[i][2];
                int rootOne = getParent(parentCopy, one);
                int rootTwo = getParent(parentCopy, two);
                if (rootOne == rootTwo) {
                    res++;
                } else {
                    parentCopy[rootOne] = rootTwo;
                }
            }
        }
        set = new HashSet<>();
        count = 0;
        for (int i = 1; i < parentCopy.length; i++) {
            int par = getParent(parentCopy, i);
            if (i != par) {
                count++;
                set.add(par);
            }
        }
        if (count + set.size() < n) {
            return -1;
        }
        return res;
    }
    private int getParent(int[] parent, int i) {
        if (i != parent[i]) {
            parent[i] = getParent(parent, parent[i]);
        }
        return parent[i];
    }
}
