class Solution {
    
    int find(int i, int parent[]){
        while(i!=parent[i]) {
            i = parent[i];
        }
        return i;
    }
    
    void union(int x, int y, int parent[]) {
        parent[x] = y; 
    }
    
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int parent[] = new int[n];
        for (int i = 0;i<n;i++){
            parent[i] = i;
        }
        int ans = 0;
        int e1 = 0;
        for (int i = 0;i<edges.length;i++){
            if (edges[i][0] == 3) {
                int x = find(edges[i][1]-1,parent);
                int y = find(edges[i][2]-1,parent);
                if (x == y) {
                    ans++;
                } else {
                    union(x,y,parent);
                    e1++;
                }
            }
        }
        int temp[] = new int[n];
        for (int i = 0;i<n;i++){
            temp[i] = parent[i];
        }
        int e2 = e1;
        for (int i = 0;i<edges.length;i++){
            if (edges[i][0] == 1) {
                int x = find(edges[i][1]-1,parent);
                int y = find(edges[i][2]-1,parent);
                if (x == y) {
                    ans++;
                } else {
                    union(x,y,parent);
                    e1++;
                }
            }
        }
        for (int i = 0;i<edges.length;i++){
            if (edges[i][0] == 2) {
                int x = find(edges[i][1]-1,temp);
                int y = find(edges[i][2]-1,temp);
                if (x == y) {
                    ans++;
                } else {
                    union(x,y,temp);
                    e2++;
                }
            }
        }
        if (e1 != n-1 || e2 != n-1) {
            return -1;
        }
        return ans;
    }
}
