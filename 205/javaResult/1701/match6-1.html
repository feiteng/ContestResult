class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int l = edges.length;
        
        int[] g = new int[n];
        for (int i = 0; i < n; i++) {
            g[i] = i;
        }
        int res = 0;
        
for (int[] edge: edges) {
            if (edge[0] == 3) {
                if (!join(g, edge[1]-1, edge[2]-1)) res++;
            }
        }
        
        int[] gb = new int[n];
        
for (int i = 0; i < n; i++) {
            gb[i] = g[i];
        }
        
        for (int[] edge: edges) {
            if (edge[0] == 1) {
                if (!join(g, edge[1]-1, edge[2]-1)) res++;
            }
        }
        
        
for (int[] edge: edges) {
            if (edge[0] == 2) {
                if (!join(gb, edge[1]-1, edge[2]-1)) res++;
            }
        }
        
        
        for (int i = 1; i < n; i++) {
            if (find(g, i) != find(g, i-1)) return -1;
            if (find(gb, i) != find(gb, i-1)) return -1;
        }
        return res;
    }
    

    
    private int find(int[] g, int a) {
        while (g[a] != a) {
            g[a] = g[g[a]];
            a = g[a];
        }
        return a;
    }
    
    private boolean join(int[] g, 
int a, int b) {
        int ra = find(g, a);
        int rb = find(g, b);
        if (ra == rb) return false;
        g[rb] = ra;
        return true;
    }
}
