class Solution {
    int[] parent1;
    int[] parent2;
    Set<String> edge3 = new HashSet<>();
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        
        parent1 = new int[n + 1];
        parent2 = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            parent1[i] = i;
            
parent2[i] = i;
        }
        int res = 0;
        for(int[] edge: edges) {
            if(edge[0] == 3) {
                edge3.add(" " + edge[1] + ":" + edge[2]);
                int root1 = find(parent1, edge[1]);
                int root2 = find(parent1, edge[2]);
                
                
int root3 = find(parent2, edge[1]);
                int root4 = find(parent2, edge[2]);
                if(root1 == root2 || root3 == root4) {
                    res++;
                } else {
                    union(parent1, root1, root2);
                    union(parent2, root3, root4);
                }
            }
            
        }
        
        
        for(int[] edge: edges) {
            
            if(edge[0] == 1) {
                String key = " " + edge[1] + ":" + edge[2];
                if(edge3.contains(key)) {
                    res++;
                    continue;
                }
                
int root1 = find(parent1, edge[1]);
                int root2 = find(parent1, edge[2]);
                if(root1 == root2) {
                    res++;
                } else {
                    union(parent1, root1, root2);
                }
            } else if (edge[0] == 2) {
                String key = " " + edge[1] + ":" + edge[2];
                if(edge3.contains(key)) {
                    res++;
                    continue;
                }
                
int root1 = find(parent2, edge[1]);
                int root2 = find(parent2, edge[2]);
                if(root1 == root2) {
                    res++;
                } else {
                    union(parent2, root1, root2);
                }
            } else {
                continue;
            }
            
            
        }
        int root1 = find(parent1, 1);
        int root2 = find(parent2, 1);
        for(int i = 1; i <= n; i++) {
            if(find(parent1, i) != root1 || find(parent2, i) != root2) {
                return -1;
            }
        }
        return res;
    }
    
    public int find(int[] parent, int node) {
        if(parent[node] == node) return node;
        int root = find(parent, parent[node]);
        parent[node] = root;
        
return root;
    }
    
    public void union(int[] parent, int node1, int node2) {
        parent[node1] = node2;
    }
}
