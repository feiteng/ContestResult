class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (a, b) -> (b[0] - a[0]));
        int[] dSet1 = new int[n], dSet2 = new int[n];
        int piece1 = n, piece2 = n;
        for(int i = 0; i < n; i++) {
            dSet1[i] = i;
            dSet2[i] = i;
        }
        int res = edges.length;
        for(int i = 0; i < edges.length; i++) {
            if(piece1 == 1 && piece2 == 1) return res;
            
if(edges[i][0] == 3) {
                if(union(dSet1, edges[i][1] - 1, edges[i][2] - 1))
                {
                    union(dSet2, edges[i][1] - 1, edges[i][2] - 1);
                    res--;
                    piece2--;
                    piece1--;
                }
            }else if(edges[i][0] == 1) {
                if(union(dSet1, edges[i][1] - 1, edges[i][2] - 1)) {
                    res--;
                    piece1--;
                }
            }else {
                if(union(dSet2, edges[i][1] - 1, edges[i][2] - 1)) {
                    res--;
                    piece2--;
                }
            }
        }
        if(piece1 > 1 || piece2 > 1) return -1;
        return res;
    }
    
    private boolean union(int[] set, int a, int b) {
        int rootA = getRoot(set, a);
        int rootB = getRoot(set, b);
        if(rootA == rootB) return false;
        set[rootA] = rootB;
        return true;
    }
    
    private int getRoot(int[] set, int root) {
        if(set[root] == root) return root;
        set[root] = getRoot(set, set[root]);
        return set[root];
    }
}
