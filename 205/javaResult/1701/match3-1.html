import java.util.Arrays;

class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        Arrays.sort(edges, (o1, o2) -> o2[0] - o1[0]);
        DSU from = new DSU(n);
        DSU to = new DSU(n);
        int i = 0;
        int ans = 0;
        for (; i < edges.length && edges[i][0] == 3; i++) {
            int x = from.find(edges[i][1] - 1);
            int y = from.find(edges[i][2] - 1);
            if (x == y) {
                ans++;
            } else {
                from.union(x, y);
                to.union(x, y);
            }
        }

        for (; i < edges.length && edges[i][0] == 2; i++) {
            int x = from.find(edges[i][1] - 1);
            int y = from.find(edges[i][2] - 1);
            if (x == y) {
                ans++;
            } else {
                from.union(x, y);
            }
        }

        for (; i < edges.length && edges[i][0] == 1; i++) {
            int x = to.find(edges[i][1] - 1);
            int y = to.find(edges[i][2] - 1);
            if (x == y) {
                ans++;
            } else {
                to.union(x, y);
            }
        }

        int f = -1;
        int t = -1;
        for (int k = 0; k < n; k++) {
            if (f != -1 && from.find(k) != f) {
                return -1;
            }
            f = from.find(k);
            if (t != -1 && to.find(k) != t) {
                return -1;
            }
            t = to.find(k);
        }
        return ans;
    }

    static class DSU {
        int[] parent;
        int[] rank;

        public DSU(int N) {
            parent = new int[N];
            for (int i = 0; i < N; ++i)
                parent[i] = i;
            rank = new int[N];
            Arrays.fill(rank, 1);
        }

        public int find(int v) {
            if (parent[v] != v) parent[v] = find(parent[v]);
            return parent[v];
        }

        public int union(int x, int y) {
            x = find(x);
            y = find(y);
            if (x == y) {
                return x;
            }
            if (rank[x] > rank[y]) {
                rank[x] += rank[y];
                parent[y] = x;
                return x;
            } else {
                rank[y] += rank[x];
                parent[x] = y;
                return y;
            }
        }
    }
}
