class Solution {
    
private class DSU{
        int[] parent;

        public DSU(int N) {
            parent = new int[N];
            for (int i = 0; i < N; ++i) {
                parent[i] = i;
            
}
        }

        public int find(int x) {
            if (parent[x] != x) parent[x] = find(parent[x]);
            return parent[x];
        }

        public void union(int x, int y) {
            parent[find(x)] = find(y);
        }
    }

    private void copy(DSU u1, DSU u2) {
        int len = u1.parent.length;
        for (int i = 0; i < len; ++i) {
            u2.parent[i] = u1.parent[i];
        }
    }

    public int maxNumEdgesToRemove(int n, int[][] edges) {
        if (n <= 1) {
            return edges.length;
        }

        DSU uf = new DSU(n);
        int res = 0;
        for (int[] e : edges) {
            
if (e[0] == 3) {
                if (uf.find(e[1] - 1) == uf.find(e[2] - 1)) {
                    ++res;
                } else {
                    uf.union(e[1] - 1, e[2] - 1);
                }
            }
        }

        DSU uA = new DSU(n);
        copy(uf, uA);
        for (int[] e : edges) {
            
if (e[0] == 1) {
                if (uA.find(e[1] - 1) == uA.find(e[2] - 1)) {
                    ++res;
                } else {
                    uA.union(e[1] - 1, e[2] - 1);
                }
            }
        }

        DSU uB = new DSU(n);
        copy(uf, uB);
        for (int[] e : edges) {
            
if (e[0] == 2) {
                if (uB.find(e[1] - 1) == uB.find(e[2] - 1)) {
                    ++res;
                } else {
                    uB.union(e[1] - 1, e[2] - 1);
                }
            }
        }

        int a = uA.find(0), b = uB.find(0);
        for (int i = 0; i < n; ++i) {
            if (a != uA.find(i) || b != uB.find(i)) {
                return -1;
            }
        }

        return res;
    }
}
