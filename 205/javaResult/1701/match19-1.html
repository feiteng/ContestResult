class Solution {
    
    class UF {
        int N;
        int[] parent;
        int[] size;
        int components;
        
        public UF(int N) {
            this.N = N;
            parent = new int[N];
            size = new int[N];
            for (int i = 0; i < N; i++) {
                parent[i] = i;
                size[i] = 1;
            }
            components = N;
        }
        
        public int find(int t) {
            while (parent[t] != t) {
                t = parent[t];
            }
            return t;
        }
        
        public void union(int i, int j) {
            int iRoot = find(i);
            int jRoot = find(j);
            if (iRoot != jRoot) {
                if (size[iRoot] > size[jRoot]) {
                    parent[jRoot] = iRoot;
                    size[iRoot] += size[jRoot];
                }
                else {
                    parent[iRoot] = jRoot;
                    size[jRoot] += size[iRoot];
                }
                components--;
            }
        }
    }
    
    
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UF alice = new UF(n);
        UF bob = new UF(n);
        int used = 0;
        List<int[]> type1 = new ArrayList<>();
        List<int[]> type2 = new ArrayList<>();
        
        for (int[] edge : edges) {
            if (edge[0] == 1) {
                type1.add(edge);
            }
            else if (edge[0] == 2) {
                type2.add(edge);
            }
            else {
                int iRoot = alice.find(edge[1] - 1);
                int jRoot = alice.find(edge[2] - 1);
                if (iRoot != jRoot) {
                    used++;
                    alice.union(edge[1] - 1, edge[2] - 1);
                    bob.union(edge[1] - 1, edge[2] - 1);
                }
            }
        }
        
        if (alice.components == 1) {
            return edges.length - used;
        }
        
        for (int[] edge : type1) {
            int iRoot = alice.find(edge[1] - 1);
            int jRoot = alice.find(edge[2] - 1);
            if (iRoot != jRoot) {
                used++;
                alice.union(edge[1] - 1, edge[2] - 1);
            }
        }
        
        if (alice.components > 1) {
            return -1;
        }
        
        for (int[] edge : type2) {
            int iRoot = bob.find(edge[1] - 1);
            int jRoot = bob.find(edge[2] - 1);
            if (iRoot != jRoot) {
                used++;
                bob.union(edge[1] - 1, edge[2] - 1);
            }
        }
        
        if (bob.components > 1) {
            return -1;
        }
        
        return edges.length - used;
    }
}
