class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        int ans = 0;
        UF ufa = new UF(n);
        UF ufb = new UF(n);
        
        List<int[]> A = new ArrayList<>();
        List<int[]> B = new ArrayList<>();
        List<int[]> C = new ArrayList<>();
        
        for (int[] e : edges){
            if (e[0] == 3) {
                C.add(e);
            } else if (e[0] == 1) {
                A.add(e);
            } else {
                B.add(e);
            }
        }
        
        for (int[] e : C) {
            if (ufa.connected(e[1],e[2]) && ufb.connected(e[1],e[2])) {
                ans++;
            } else {
                ufa.union(e[1],e[2]);
                ufb.union(e[1],e[2]);
            }
        }
        
        for (int[] e : A) {
            if (ufa.connected(e[1],e[2])) {
                ans++;
            }else {
                ufa.union(e[1],e[2]);
            }
        }
        
        for (int[] e : B) {
            if (ufb.connected(e[1],e[2])) {
                ans++;
            } else {
                ufb.union(e[1],e[2]);
            }
        }

        return ufa.count == 1 && ufb.count == 1 ? ans : -1;
    }
    
    public static class UF {
        
        private int[] id;
        private int[] weight;
        private int count;

        public UF(int N) {
            id = new int[N+1];
            weight = new int[N+1];
            for (int i = 0; i <= N; i++) {
                // 每个点最初与自己相连
                id[i] = i;
                weight[i] = 1;
            }
            count = N;
        }

        public boolean connected(int p, int q) {
            return find(p) == find(q);
        }

        public int count() {
            return this.count;
        }

        // 找到p所在component的标识符
        public int find(int p) {
            int old = p;
            while (id[p] != p)
                p = id[p];
            // 路径压缩，将所有的叶直接连在根上
            while (id[old] != old) {
                int t = old;
                old = id[old];
                id[t] = p;
            }
            return p;
        }

        // quick union的结果会形成一片森林
        public void union(int p, int q) {
            int pRoot = find(p);
            int qRoot = find(q);
            if (weight[pRoot] < weight[qRoot]) {
                id[pRoot] = qRoot;
                weight[qRoot] += weight[pRoot];
            } else {
                id[qRoot] = pRoot;
                weight[pRoot] += weight[qRoot];
            }
            --count;
        }
    }
}
