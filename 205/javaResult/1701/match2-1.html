class Solution {
    public int maxNumEdgesToRemove(int n, int[][] edges) {
        UF ufa = new UF(n);
        UF ufb = new UF(n);
        int out = 0;
        for (int[] e : edges) {
            if (e[0] == 3) {
                int v1 = e[1] - 1;
                int v2 = e[2] - 1;
                
if (ufa.isConnected(v1, v2)) {
                    out++;
                } else {
                    ufa.connect(v1, v2);
                    ufb.connect(v1, v2);
                }
            }
        }
        for (int[] e : edges) {
            if (e[0] != 3) {
                
int v1 = e[1] - 1;
                int v2 = e[2] - 1;
                if (e[0] == 1) {
                    if (ufa.isConnected(v1, v2)) {
                        out++;
                    } else {
                        ufa.connect(v1, v2);
                    }
                } else 
if (e[0] == 2) {
                    if (ufb.isConnected(v1, v2)) {
                        out++;
                    } else {
                        ufb.connect(v1, v2);
                    }
                }
            }
        }
        if (ufa.cc() == 1 && ufb.cc() == 1) {
            return out;
        } else {
            return -1;
        }
    
}

    class UF {

        private int[] root;

        public UF(int n) {
            this.root = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
        }

        private int root(int i) {
            if (root[i] == i) {
                return i;
            } else {
                root[i] = root[root[i]];
                return root(root[i]);
            }
        }

        public void connect(int i, int j) {
            root[root(i)] = root(j);
        
}

        public boolean isConnected(int i, int j) {
            return root(i) == root(j);
        }
        
        public int cc() {
            return (int) IntStream.range(0, root.length).map(this::root).distinct().count();
        }
    }
}
