class Solution {
    public int minCost(String ss, int[] cost) {
        char[] s = ss.toCharArray();
        int n = cost.length;
        int sum = cost[0];
        int max = cost[0];
        int res = 0;
        
        for (int i = 1; i < n; i++) {
            if (s[i] == s[i - 1]) {
                sum += cost[i];
                max = Math.max(max, cost[i]);
            } else {
                res += sum - max;
                
                
sum = cost[i];
                max = cost[i];
            }
        }
        
        res += sum - max;
        
        return res;
    }
}
