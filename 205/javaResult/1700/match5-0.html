class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        char[] chs = s.toCharArray();
        int id = 0;
        while(id < chs.length) {
            char x = chs[id];
            int sum = cost[id];
            int max = cost[id];
            int nid = id + 1;
            while(nid < chs.length && chs[nid] == x) {
                sum += cost[nid];
                max = Math.max(max, cost[nid]);
                ++nid;
            }
            if(nid != id + 1) {
                res += sum - max;
            }
            id = nid;
        }
        return res;
    }
}
