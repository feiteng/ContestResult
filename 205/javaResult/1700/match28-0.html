class Solution {
    public int minCost(String s, int[] cost) {
        char[] str = s.toCharArray();
        int min = 0;
        int ans = 0;
        int sum = 0;
        for(int i = 0; i < str.length; i++) {
            if(i == 0) {
                min = cost[i];
                sum += min;
                continue;
            }
            if(str[i] == str[i - 1]) {
                min = Math.max(min, cost[i]);
                sum += cost[i];
            } else {
                ans += (sum - min);
                sum = min = 0;
                min = cost[i];
                sum += min;
            }
        }
        ans += (sum - min);
        return ans;
    }
}
