class Solution {
    Integer[][] dp;
    public int minCost(String s, int[] cost) {
        int len = s.length();
        dp = new Integer[len + 1][27];
        return dfs(s, cost, '1', 0);
    }
    private int dfs(String s, int[] cost, char pre, int cur) {
        if (cur == s.length()) {
            return 0;
        }
        int idx = pre == '1' ? 0 : (pre - 'a' + 1);
        if (dp[cur + 1][idx] != null) {
            return dp[cur + 1][idx];
        }
        int res = Integer.MAX_VALUE;
        res = Math.min(res, cost[cur] + dfs(s, cost, pre, cur + 1));
        if (s.charAt(cur) != pre) {
            res = Math.min(res, dfs(s, cost, s.charAt(cur), cur + 1));
        }
        dp[cur + 1][idx] = res;
        return res;
    }
}
