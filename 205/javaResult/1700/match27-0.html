class Solution {
    public int minCost(String s, int[] cost) {
        char[] charr = s.toCharArray();
        int res = 0;
        for (int i = 0; i < charr.length; i++) {
            if (i < charr.length - 1 && charr[i] == charr[i + 1]) {
                int j = i + 1, sum = cost[i], max = cost[i];
                while (j < charr.length && charr[i] == charr[j]) {
                    max = Math.max(max, cost[j]);
                    sum += cost[j];
                    j++;
                }
                res += (sum - max);
                i = j - 1;
            } 
        }
        return res;
    }
}
