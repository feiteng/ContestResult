class Solution {
    public int minCost(String s, int[] cost) {
        
        
        char[] chars = s.toCharArray();
        int len = cost.length;

        int count = 0;
        for (int i = 0; i < len - 1; i++) {
            if (chars[i] == chars[i + 1]) {
                count += Math.min(cost[i], cost[i + 1]);
                cost[i] = Math.max(cost[i], cost[i + 1]);
                cost[i + 1] = 
Math.max(cost[i], cost[i + 1]);
            }
        }
        return count;
    }
}
