class Solution {
    public int minCost(String s, int[] cost) {
        int sum = 0;
        int max = -1;
        int res = 0;
        char prev = ' ';
        for(int i = 0; i < s.length(); i++) {
            if(s.charAt(i) != prev) {
                if(max != -1) {
                    res += sum - max;
                    max = cost[i];
                    sum = cost[i];
                } else {
                    max = cost[i];
                    sum = cost[i];
                }
                prev = s.charAt(i);
            } else {
                max = Math.max(max, cost[i]);
                sum += cost[i];
            }
        }
        if(max != -1) {
                    res += sum - max;
                }
        
        return res;
        
    }
}
