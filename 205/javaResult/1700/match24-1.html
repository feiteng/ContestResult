class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        char[] chs = s.toCharArray();
        int keptIdx = 0;
        for (int i = 0; i < chs.length; i++) {
            if (i == 0 || chs[i] != chs[i - 1]) {
                keptIdx = i;
            } else {    // current char is the same as a prev char
                if (cost[i] < cost[keptIdx]) {
                    res += cost[i];
                } else {
                    res += cost[keptIdx];
                    keptIdx = i;
                }
            }
        }
        return res;
    }
}
