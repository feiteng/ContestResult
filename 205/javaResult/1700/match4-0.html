class Solution {
    public int minCost(String s, int[] cost) {
        int out = 0;
        int start = 0;
        char c = s.charAt(0);
        for (int i = 1; i < s.length(); i++) {
            if (c != s.charAt(i)) {
                out += costRange(cost, start, i - 1);
                start = i;
                c = s.charAt(i);
            }
        }
        out += costRange(cost, start, cost.length - 1);
        return out;
    }

    private int costRange(int[] cost, int s, int e) {
        int sum = 0;
        int max = 0;
        for (int i = s; i <= e; i++) {
            sum += cost[i];
            max = Math.max(max, cost[i]);
        }
        return sum - max;
    }
}
