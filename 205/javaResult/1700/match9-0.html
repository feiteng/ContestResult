class Solution {
    public int minCost(String s, int[] cost) {
        char[] ch = s.toCharArray();
        char prev = '0';
        int max = 0;
        int sum = 0;
        int res = 0;
        for (int i = 0; i < ch.length; i++) {
            if (ch[i] == prev) {
                sum += cost[i];
                max = Math.max(max, cost[i]);
            } else {
                res += sum - max;
                max = cost[i];
                sum = max;
            }
            prev = ch[i];
        }
        res += sum - max;
        return res;
    }
}
