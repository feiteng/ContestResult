class Solution {
    public int minCost(String s, int[] cost) {
        Stack<Integer> st = new Stack<>();
        int ans = 0;
        for(int i=0;i<s.length();++i){
            if(st.size() != 0 && s.charAt(st.peek()) == s.charAt(i)){
                int min = cost[st.peek()];
                List<Integer> list = new ArrayList<>();
                list.add(min);
                int idx = st.peek();
                while(i < s.length() && s.charAt(i) == s.charAt(st.peek())){
                    if(min > cost[i]){
                        idx=i;
                        min = cost[i];
                    }
                    list.add(cost[i]);
                    ++i;
                }
                Collections.sort(list);
                for(int x : list) ans+=x;
                ans-=list.get(list.size()-1);
                st.pop();
                st.push(idx);
                --i;
            }else st.push(i);
        }
        return ans;
    }
}
