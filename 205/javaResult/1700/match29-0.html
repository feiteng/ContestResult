class Solution {
    public int minCost(String s, int[] cost) {
        int sol = 0;
        int ct = 1;
        int max = cost[0];
        int tot = cost[0];
        for (int i = 1; i <= s.length(); i++) {
            if (i < s.length() && s.charAt(i) == s.charAt(i-1)) {
                ct++;
                tot += cost[i];
                max = Math.max(max, cost[i]);
            } else {
                if (ct > 1) {
                    sol += tot - max;
                }
                if (i != s.length()) {
                    max = cost[i];
                    tot = cost[i];
                    ct = 1;
                }
            }
        }
        return sol;
    }
}
