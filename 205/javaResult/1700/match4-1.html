class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0, n = s.length(), i = 0;
        while (i < n) {
            char cur = s.charAt(i);
            int j = i + 1, sum = cost[i], max = cost[i];
            while (j < n && cur == s.charAt(j)) 
{
                sum += cost[j];
                max = Math.max(max, cost[j]);
                j++;
            }
            if (j != i + 1) {
                res += (sum - max);
            }
            i = j;
        }
        return res;
    }
}
