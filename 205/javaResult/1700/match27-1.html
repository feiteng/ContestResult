class Solution {
    public int minCost(String s, int[] cost) {
        int n = s.length();
        int ans = 0;
        
        for(int i = 0 ; i < n ; i++) {
            if(i < n-1 && s.charAt(i) == s.charAt(i+1)) {
                int j = i+1;
                while(j+1 < n && s.charAt(j+1) == s.charAt(i)) {
                    j++;
                }
                int sum = 0;
                int maxVal = -1;
                for(int l = i ; l <= j ; l++) {
                    sum += cost[l];
                    maxVal = Math.max(maxVal, cost[l]);
                    // System.out.println("sum: " + sum + "  max: " + maxVal);
                }
                ans += sum - maxVal;
                // System.out.println(i  + " " + j + " " + sum + " " + maxVal);
                
i = j;
            }
        }
        
        return ans;
    }
}
