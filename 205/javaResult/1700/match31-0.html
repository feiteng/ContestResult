class Solution {
    
    public int minCost(String s, int[] cost) {
        
        int n=s.length(), sum=0;
        
        Stack<Integer> stack = new Stack<>();
        
        for(int i=0; i<n; i++) {
            char ch=s.charAt(i);
            if(!stack.isEmpty() && s.charAt(stack.peek())==ch) {
                if(cost[stack.peek()]<cost[i]) {
                    sum+=cost[stack.pop()];
                    stack.push(i);
                } else {
                    sum+=cost[i];
                }
            } else {
                stack.push(i);
            }
        }
        
        return sum;
    }
}
