class Solution {
    public int minCost(String s, int[] cost) {
        Deque<Integer> stack = new ArrayDeque<>();
        int n = s.length();
        char[] chars = s.toCharArray();
        int res = 0;
        for(int i = 0 ; i < n ; i++){
            if(!stack.isEmpty() && chars[stack.peek()] == chars[i]){
                if(cost[stack.peek()] < cost[i]){
                    res += cost[stack.peek()];
                    stack.pop();
                    stack.push(i);
                } else {
                    res += cost[i];
                }
            } else {
                stack.push(i);
            }
        }
        return res;
    }
}
