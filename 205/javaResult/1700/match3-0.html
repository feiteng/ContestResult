class Solution {

    public int minCost(String s, int[] cost) {
        int sum = 0, max = 0;
        for (int i = 0; i < s.length(); i++) {
            if (i > 0 && s.charAt(i) != s.charAt(i - 1)) {
                sum -= max;
                max = 0;
            }
            sum += cost[i];
            max = Math.max(max, cost[i]);
        }
        return sum - max;
    }
}
