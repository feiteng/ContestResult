class Solution {
    public int minCost(String s, int[] cost) {
        int ans = 0;
        int i = 0;
        while(i < s.length()) {
            int j = i;
            while(j < s.length() && s.charAt(j) == s.charAt(i)) {
                j++;
            }
            if (i != j-1) {
                int max = cost[i];
                
int total = 0;
                for (int k = i; k < j; k++) {
                    total += cost[k];
                    max = Math.max(max, cost[k]);
                }
                total -= max;
                ans += total;
                i = j;
            } else {
                i++;
            }
        }
        return ans;
    }
}
