class Solution {
    public int minCost(String s, int[] cost) {
        int now = 0;
        int res = 0;
        while (now <= s.length()) {
            int end = now;
            while (end + 1 < s.length() && s.charAt(end + 1) == s.charAt(now))
                end++;
            if (end == now) {
                now = end + 1;
                continue;
            }
            int max = cost[now];
            for (int i = now; i <= end; i++) {
                res += cost[i];
                max = Math.max(cost[i], max);
            }
            res -= max;
            now = end + 1;

        }
        return res;
    }
}
