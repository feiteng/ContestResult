class Solution {
    public int minCost(String s, int[] cost) {
        char c[] = s.toCharArray();
        int n = c.length;
        int res = 0;
        char last = 'A';
        int max = -1;
        int sum = -1;
        for(int i = 0; i < n; i++){
            if(c[i] == last){
                max = Math.max(max, cost[i]);
                sum += cost[i];
            } else {
                res += sum - max;
                max = cost[i];
                sum = cost[i];
            }
            last = c[i];
        }
        res += sum - max;
        return res;
    }
}
