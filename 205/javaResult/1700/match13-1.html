class Solution {
    public int minCost(String str, int[] cost) {
        int res = 0;
        char[] s = str.toCharArray();
        Deque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i < s.length; ++i) {
            if (!
stack.isEmpty() && s[stack.peek()] == s[i]) {
                if (cost[stack.peek()] < cost[i]) {
                    res += cost[stack.pop()];
                    stack.push(i);
                } else {
                    res += cost[i];
                }
            } else {
                stack.push(i);
            }
        }
        return res;
    }
}
