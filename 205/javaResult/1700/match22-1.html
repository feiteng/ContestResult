class Solution {
    public int minCost(String s, int[] cost) {
        int re = 0, n = s.length();
        int sum = cost[0], max = cost[0], left = 0;
        for (int i = 1; i < n; i++) {
            if (s.charAt(i) != s.charAt(i - 1)) {
                if (left < i) {
                    re += sum - max;
                }
                sum = cost[i];
                
max = cost[i];
                left = i;
            } else {
                sum += cost[i];
                max = Math.max(max, cost[i]);
            }
        }
        if (left < n - 1) {
            re += sum - max;
        }
        return re;
    }
}
