class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        int start = 0;
        while (start < s.length()) {
            int end = start;
            while (end < s.length() && s.charAt(end) == s.charAt(start)) {
                end++;
            }

            if (end - start > 1) {
                int tmpS = 0;
                int tmpM = 0;
                for (int i = start; i < end; i++) {
                    tmpS += cost[i];
                    tmpM = Math.max(tmpM, cost[i]);
                }
                res += tmpS - tmpM;
            }

            start = end;
        }
        return res;
    }
}
