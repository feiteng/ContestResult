class Solution {
    public int minCost(String s, int[] cost) {
        int ans = 0;
        int n = s.length();
        char c = 0;
        int sum = 0;
        int max = 0;
        for (int i = 0; i < n; i++) {
            if (s.charAt(i) != c) {
                ans += sum - max;
                max = cost[i];
                sum = cost[i];
            } else {
                max = Math.max(max, cost[i]);
                sum += cost[i];
            }
            c = s.charAt(i);
        }
        ans += sum - max;
        return ans;
    }
}
