class Solution {
    public int minCost(String s, int[] cost) {
        int n = s.length();
        if(n==1) return 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        int sum = 0;
        int ans = 0;
        for(int i = 1; i<=n; i++){
            pq.add(cost[i-1]);
            sum += cost[i-1];
            if(i < n && s.charAt(i) == s.charAt(i-1)){
                continue;
            }
            else{
                ans += sum - pq.poll();
                sum = 0;
                pq.clear();
            }
        }
        if(!pq.isEmpty())
        ans += sum - pq.poll();
        return ans;
    }
}
