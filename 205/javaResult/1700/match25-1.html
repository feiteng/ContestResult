class Solution {
    public int minCost(String s, int[] cost) {
        int count = 1;
        int maxc = cost[0];
        int sum = cost[0];
        int n = s.length();
        int res = 0;
        for (int i = 1; i < n; i++) {
            if (s.charAt(i) != s.charAt(i - 1)) {
                // System.out.println("" + i + " " + res + " "+ sum + " " + maxc);
                res += sum - maxc;
                count = 1;
                maxc = cost[i];
                sum = cost[i];
            
} else {
                sum += cost[i];
                count++;
                maxc = Math.max(maxc, cost[i]);
            }
        }
        if (count > 1) {
            res += sum - maxc;
        }
        // System.out.println();
        return res;
    }
}
