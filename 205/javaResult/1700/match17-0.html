class Solution {
    public int minCost(String s, int[] cost) {
        int max = 0;
        int sum = 0;
        int n = s.length();
        if (n == 0){
            return 0;
        }
        max = cost[0];
        sum = cost[0];
        char c = s.charAt(0);
        int ans = 0;
        int l = 1;
        for (int i = 1;i<n;i++){
            if (s.charAt(i) == c) {
                sum += cost[i];
                max = Math.max(max, cost[i]);
                l++;
            } else {
                if (l > 1) {
                    ans += sum - max;
                }
                sum = cost[i];
                max = cost[i];
                l = 1;
                c = s.charAt(i);
            }
        }
        if (l>1) {
            ans += sum-max;
        }
        return ans;
    }
}
