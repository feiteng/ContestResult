class Solution {
    public int minCost(String s, int[] cost) {
        int[] count = new int[s.length()];
        int ret = 0;
        
char prev = '-';
        int i = 1;
        boolean same = false;
        while (i < s.length()) {
            int sum = 0, max = 0;
            while (i < s.length() && s.charAt(i - 1) == s.charAt(i)) {
                if (!same) {
                    same = true;
                    sum += cost[i - 1];
                    max = Math.max(max, cost[i - 1]);
                }
                sum += cost[i];
                
max = Math.max(max, cost[i]);
                i++;
            }
            same = false;
            ret += (sum - max);
            i++;
        }
        return ret;
    }
}
