class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        int i = 0;
        while (i < s.length()) {
            int j = i + 1;
            while (j < s.length() && s.charAt(j) == s.charAt(i)) {
                j++;
            }
            if (j - 1 > i) {
                int sum = 0;
                int max = 0;
                for (int k = i; k < j; k++){
                    sum += cost[k];
                    max = Math.max(max, cost[k]);
                }
                res += sum - max;
            } 
            i = j;
        }
        return res;
    }
}
