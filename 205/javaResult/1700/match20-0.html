class Solution {
    public int minCost(String s, int[] cost) {
        ArrayList<pair> list=getList(s);
        //System.out.println(list);
        int ans=0;
        for(pair p:list)
        {
            int max=-1;
            int sum=0;
            for(int i=p.a;i<=p.b;i++)
            {
                max=Math.max(cost[i],max);
                sum+=cost[i];
            }
            //System.out.println("min "+min+" sum "+sum);
            ans=ans+sum-max;
        }
        return ans;
    }
    class pair
    {
        int a ;int b;
        pair(int a,int b)
        {
            this.a=a;this.b=b;
        }
        public String toString()
        {
            return a+" "+b;
        }
    }
    public ArrayList<pair> getList(String s)
    {
        ArrayList<pair> nl=new ArrayList<>();
        int last=0;
        for(int i=1;i<s.length();i++)
        {
            if(s.charAt(i)!=s.charAt(i-1))
            {
                nl.add(new pair(last,i-1));
                last=i;
            }
        }
        nl.add(new pair(last,s.length()-1));
        return nl;
    }
}
