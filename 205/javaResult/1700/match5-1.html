class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        char[] ss = s.toCharArray();
        int n = ss.length;
        for (int i = 0; i < n;) 
{
            char c = ss[i];
            int max = cost[i];
            int sum = cost[i];
            int j = i + 1;
            while (j < n && ss[j] == c) {
                max = Math.max(cost[j], max);
                sum += cost[j];
                j++;
            }
            res += sum - max;
            i = j;
        }
        return res;
    }
}
