class Solution {
    public int minCost(String s, int[] cost) {
        Stack<Integer> stack=new Stack<>();
        int res=0;
        for(int i=0;i<s.length();i++){
            if(!stack.isEmpty()&&s.charAt(stack.peek())==s.charAt(i)){
                if(cost[stack.peek()]<cost[i]){
                    res+=cost[stack.peek()];
                    stack.pop();
                    stack.push(i);
                }
                else {
                    res+=cost[i];
                }
            
} else stack.push(i);
        }
        return res;
    }
}
