class Solution {
    /*
    it seems greedy works?
    */
    public int minCost(String s, int[] cost) {
        int ret=0;
        char prev='#';
        int maxCost=0;
        int totalCost=0;
        for(int i=0;i<s.length();i++)
        {
            if(prev!=s.charAt(i))
            {
                prev=s.charAt(i);
                ret+=(totalCost-maxCost);
                maxCost=cost[i];
                totalCost=cost[i];
            }else{
                maxCost=Math.max(maxCost, cost[i]);
                totalCost+=cost[i];
            
}
           // System.out.println(ret+","+i);
            
        }
        if(totalCost!=0)
            ret+=(totalCost-maxCost);
        return ret;
    }
}
