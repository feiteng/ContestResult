class Solution {
    public int minCost(String s, int[] cost) {
        int[] dp = new int[26];
        for (int i = 0; i < s.length(); i++) {
            int x = s.charAt(i) - 'a';
            int min = -1;
            for (int j = 0; j < 26; j++) {
                if (j != x) {
                    if (min == -1 || min > dp[j]) {
                        min = dp[j];
                    } 
                    dp[j] += cost[i];
                }
            }
            dp[x] = Math.min(min, dp[x] + cost[i]);
        }
        int res = -1;
        for (int i = 0; i < 26; i++) {
            if (res == -1 || res > dp[i]) {
                res = dp[i];
            }
        }
        return res;
    }
}
