import java.util.Arrays;

class Solution {
    public int minCost(String s, int[] cost) {


        int[][] dp = new int[s.length()][26];
        Arrays.fill(dp[0], cost[0]);
        dp[0][s.charAt(0) - 'a'] = 0;
        for (int i = 1; i < s.length(); i++) {
            for (int j = 0; j < 26; j++) {
                int minCost = Integer.MAX_VALUE;
                minCost = Math.min(minCost, dp[i - 1][j] + cost[i]);
                if (s.charAt(i) - 'a' == j) {

                    for (int prev = 0; prev < 26; prev++) {
                        if (prev == j) {
                            continue;
                        }
                        minCost = Math.min(minCost, dp[i - 1][prev]);
                    }
                }
                dp[i][j] = minCost;
            }
        }

        int min = Integer.MAX_VALUE;
        for (int i = 0; i < 26; i++) {
            min = Math.min(min, dp[s.length() - 1][i]);
        }

        return min;

    }
}
