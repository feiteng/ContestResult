class Solution {
    public int minCost(String s, int[] cost) {
        int cost1 = 0;
        int n = s.length();
        for(int i = 0;i<n;i++){
            int max = cost[i];
            int c = s.charAt(i);
            int sum = cost[i];int j = i+1;
            for( ;j<n&&s.charAt(j)==c;j++){
                max = Math.max(max,cost[j]);
                sum += cost[j];
            }
            i = j-1;
            cost1 += sum - max;
        }
        return cost1;
    }
}
