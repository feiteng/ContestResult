class Solution {
    public int minCost(String s, int[] cost) {
        int a = 0;
        char[] cs = s.toCharArray();
        Stack<Integer> st = new Stack<>();
        st.push(0);
        for (int i = 1; i < s.length(); i++) {
            if (cs[st.peek()] != cs[i]) {
                st.push(i);
                continue;
            }
            if (cost[st.peek()] < cost[i]) {
                a += cost[st.peek()];
                st.pop();
                st.push(i);
            } else {
                a += cost[i];
            }
        }
        return a;
    }
}
