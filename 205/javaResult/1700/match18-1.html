class Solution {
    public int minCost(String s, int[] cost) {
        int n=s.length();
        char c=s.charAt(0);
        int count=0;
        int ans=0;
        int tmp=0;
        for(int i=0;i<n;i++){
            if(c==
s.charAt(i)){
                count++;
                tmp=Math.max(cost[i],tmp);
                ans+=cost[i];
            }else{
                if(count>=1){
                    ans-=tmp;
                }
                tmp=cost[i];
                ans+=cost[i];
                count=1;
                c=s.charAt(i);
            
}
        }
        if(count>=1){
            ans-=tmp;
        }
        return ans;
    }
}
