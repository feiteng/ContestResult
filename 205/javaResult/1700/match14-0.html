class Solution {
    public int minCost(String s, int[] cost) {
        int max = 0;
        int sum = 0;
        int n = s.length();
        if(n==1){
            return 0;
        }
        char prev = '*';
        int curMax = 0;
        for(int i=0; i<n; i++){
            char c = s.charAt(i);
            if(c==prev){
                curMax = Math.max(curMax, cost[i]);
            }
            else{
                max += curMax;
                curMax = cost[i];
            }
            prev = c;
            sum += cost[i];
        }
        max += curMax;
        // System.out.println(sum+","+max);
        return sum-max;
    }
}
