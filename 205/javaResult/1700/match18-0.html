class Solution {
    public int minCost(String s, int[] cost) {
        int res = 0;
        int i = 0;
        while (i < s.length()) {
            int j = i;
            while (j < s.length() && s.charAt(i) == s.charAt(j)) {
                j++;
            }
            if (i != j-1) {
                int max = cost[i];
                int tmp = 0;
                for (int l = i; l < j; l++) {
                    tmp += cost[l];
                    max = Math.max(cost[l], max);
                }
                tmp -= max;
                res += tmp;
                i = j;
            } else {
                i++;
            }
        }
        return res;
    }
}
