class Solution {
    public int minCost(String s, int[] cost) {
        int n = s.length();
        
        int res = 0, sum = cost[0], max = cost[0];
        
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i - 1)) {
                sum += cost[i];
                max = Math.max(max, cost[i]);
            } else {
                if (sum != max) {
                    res += sum - max;
                }
                sum = cost[i];
                max = cost[i];
            }
        }
        
        if (sum != max) {
            res += sum - max;
        }
        return res;
    }
}
