class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int out = 0;
        Map<Long, Integer> sq1 = sq(nums1);
        Map<Long, Integer> sq2 = sq(nums2);
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                long k = ((long) nums1[i]) * nums1[j];
                if (sq2.containsKey(k)) {
                    out += sq2.getOrDefault(k, 0);
                }
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                long k = ((long) nums2[i]) * nums2[j];
                if (sq1.containsKey(k)) {
                    out += sq1.getOrDefault(k, 0);
                }
            }
        }
        return out;
    }

    private Map<Long, Integer> sq(int[] a) {
        Map<Long, Integer> map = new HashMap<>();
        for (int i : a) {
            long k = ((long) i ) * i;
            map.put(k, 1 + map.getOrDefault(k, 0));
        }
        return map;
    }
}
