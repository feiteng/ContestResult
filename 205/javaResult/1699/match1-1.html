class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> m1 = new HashMap<>();
        Map<Long, Integer> m2 = new HashMap<>();
        int n = nums1.length;
        int m = nums2.length;
        
for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                long p = (long)nums1[i] * nums1[j];
                m1.put(p, m1.getOrDefault(p, 0) + 1);
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = i + 1; j < m; j++) {
                long p = (long)nums2[i] * nums2[j];
                m2.put(p, m2.getOrDefault(p, 0) + 1);
            }
        }
        int re = 0;
        
for (int i = 0; i < n; i++) {
            long p = (long)nums1[i] * nums1[i];
            re += m2.getOrDefault(p, 0);
        }
        for (int i = 0; i < m; i++) {
            long p = (long)nums2[i] * nums2[i];
            re += m1.getOrDefault(p, 0);
        }
        return re;
    }
}
