class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        return getCount(nums1, nums2) + getCount(nums2, nums1);
    }
    
    public int getCount(int[] nums1, int[] nums2) {
        Map<Integer, Integer> count = new HashMap<>();
        int c = 0;
        for (int i : nums1) {
            count.clear();
            for (int j : nums2) {
                count.put(j, count.getOrDefault(j, 0) + 1);
                if (j == 0 || ((long)i * i) % j != 0L) continue;
                int s = (int)(((long)i * i) / j);
                if (s != j)
                    c += count.getOrDefault(s, 0);
                else 
                    c += count.get(s) - 1;
            }
        }
        return c;
    }
}
