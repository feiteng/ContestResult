class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        return sln1(nums1, nums2);
    }
    
    private int sln1(int[] nums1, int[] nums2){
        Map<Long, Integer> map1=new HashMap<>();
        Map<Long, Integer> map2 = new HashMap<>();
        
        for(int i : nums1) map1.put(i*1l*i, map1.getOrDefault(i*1l*i, 0)+1);
        for(int i : nums2) map2.put(i*1l*i, map2.getOrDefault(i*1l*i, 0)+1);
        
        int res = 0;
        
        for(int i =0;i<nums1.length;i++){
            for(int j =i+1;j< nums1.length;j++){
                long key = nums1[i]*1l*nums1[j];
                if(map2.containsKey(key)) res += map2.get(key);
            }
        }
        
        for(int i =0;i<nums2.length;i++){
            for(int j =i+1;j< nums2.length;j++){
                long key = nums2[i]*1l*nums2[j];
                if(map1.containsKey(key)) res += map1.get(key);
            }
        }
        return res;
    }
}
