class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long,Integer> set1 = new HashMap();
        Map<Long,Integer> set2 = new HashMap();
        for(int i:nums1) set1.put(((long)i)*i,set1.getOrDefault(((long)i)*i,0)+1);
        for(int i:nums2) set2.put(((long)i)*i,set2.getOrDefault(((long)i)*i,0)+1)
;
        int ans=0;
        for(int i=0;i<nums1.length;i++){
            for(int j=i+1;j<nums1.length;j++){
                if(set2.containsKey(((long)nums1[i])*nums1[j])) ans +=  set2.get(((long)nums1[i])*nums1[j])
;
            }
        }
        for(int i=0;i<nums2.length;i++){
            for(int j=i+1;j<nums2.length;j++){
                if(set1.containsKey(((long)nums2[i])*nums2[j])) ans +=  set1.get(((long)nums2[i])*nums2[j])
;
            }
        }
        return ans;
    }
}
