class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer, Integer> map2 = new HashMap<>();
        for (int a : nums1) {
            map1.put(a, map1.getOrDefault(a, 0) + 1);
        }
        for (int a : nums2) {
            map2.put(a, map2.getOrDefault(a, 0) + 1);
        }
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int sol = 0;
        sol += count(nums1, nums2, map2);
        sol += count(nums2, nums1, map1);
        return sol;
    }
    
    public int count(int[] ar1, int[] ar2, Map<Integer, Integer> map) {
        int sol = 0;
        for (int i = 0; i < ar1.length; i++) {
            long cur = (long)ar1[i] * ar1[i];
            for (int j = 0; j < ar2.length && (long)ar2[j] * ar2[j] <= cur; j++) {
                if (cur % ar2[j] != 0) continue;
                if (j > 0 && ar2[j] == ar2[j-1]) continue;
                int k = (int)(cur / ar2[j]);
                if (!map.containsKey(k)) continue;
                if (k == ar2[j]) {
                    int n = map.get(k);
                    sol += n * (n - 1) / 2;
                } else {
                    sol += map.get(ar2[j]) * map.get(k);
                }
            }
        }
        return sol;
    }
}
