class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> squares1 = new HashMap<>();
        Map<Long, Integer> squares2 = new HashMap<>();
        for (int num : nums1) {
            squares1.merge((long) num * num, 1, Integer::sum);
        }
        for (int num : nums2) {
            squares2.merge((long) num * num, 1, Integer::sum);
        }
        int res = 0;
        int curr = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                long prod = (long) nums1[i] * nums1[j];
                if (squares2.containsKey(prod)) {
                    res+= squares2.get(prod);
                }
            }
        }

        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                long prod = (long) nums2[i] * nums2[j];
                if (squares1.containsKey(prod)) {
                    res += squares1.get(prod);
                }
            }
        }
        return res;
    }
}
