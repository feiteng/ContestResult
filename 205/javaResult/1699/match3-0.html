class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int res = 0;
        Map<Long, Integer> map = new HashMap<>();
        for (int num : nums1) {
            Long prodKey = Long.valueOf(num) * Long.valueOf(num);
            map.put(prodKey, map.getOrDefault(prodKey, 0) + 1);
        }
        // System.out.println(map.toString());
        for (int i = 0; i < nums2.length; i++) {
            for (int j  = i + 1; j < nums2.length; j++) {
                Long prodKey = Long.valueOf(nums2[i]) * Long.valueOf(nums2[j]);
                if (map.containsKey(prodKey)) {
                    res += map.get(prodKey);
                }
            }
        }
        // System.out.println(res);
        Map<Long, Integer> map2 = new HashMap<>();
        for (int num : nums2) {
            Long prodKey = Long.valueOf(num) * Long.valueOf(num);
            map2.put(prodKey, map2.getOrDefault(prodKey, 0) + 1);
        }
        // System.out.println(map2.toString());
        for (int i = 0; i < nums1.length; i++) {
            for (int j  = i + 1; j < nums1.length; j++) {
                Long prodKey = Long.valueOf(nums1[i]) * Long.valueOf(nums1[j]);
                // System.out.println(prodKey);
                if (map2.containsKey(prodKey)) {
                    res += map2.get(prodKey);
                }
            }
        }
        return res;
    }
}
