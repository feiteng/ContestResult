import java.util.HashMap;
import java.util.Map;

class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> ping1 = new HashMap<>();
        Map<Long, Integer> ping2 = new HashMap<>();
        for (int a : nums1) {
            ping1.put((long) a * a, ping1.getOrDefault((long) a * a, 0) + 1);
        }
        for (int a : nums2) {
            ping2.put((long)a * a, ping2.getOrDefault((long) a * a, 0) + 1);
        }
        int ans = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                ans += ping2.getOrDefault((long)nums1[i] * nums1[j], 0);
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                ans += ping1.getOrDefault((long)nums2[i] * nums2[j], 0);
            }
        }
        return ans;
    }

}
