import java.util.Arrays;
import java.util.HashMap;
import java.util.TreeMap;

class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {

        Arrays.sort(nums1);
        Arrays.sort(nums2);

        return count(nums1, nums2) + count(nums2, nums1);

    }

    private int count(int[] arr1, int[] arr2) {

        TreeMap<Long, Integer> map = new TreeMap<>();
        for (int n : arr2) {
            map.put((long) n, map.getOrDefault((long) n, 0) + 1);
        }

        int ans = 0;
        for (int n : arr1) {

            long n2 = (long) n * n;
            for (long m : map.keySet()) {
                if (n2 / m < m) {
                    break;
                } else if (m * m == n2) {
                    int cnt = map.get(m);
                    ans += cnt * (cnt - 1) / 2;
                } else if (n2 % m == 0 && map.containsKey(n2 / m) && m < n2 / m) {
                    ans += map.get(n2 / m) * map.get(m);
                }
            }
        }
        return ans;


    }
}
