class Solution {
    /*
        O(n^2) multiple scan
    */
    public int numTriplets(int[] nums1, int[] nums2) {
        int a=solveType(nums1, nums2);
        int b=solveType(nums2, nums1);
        return a+b;

    }
    
    int solveType(int[] nums1, int[]nums2)
    {
        Map<Long, Integer> map=new HashMap<>();
        for(int v: nums1)
        {
            long vv=(long)(v)*(long)(v);
      
            map.put(vv, map.getOrDefault(vv, 0)+1);
        }
        int ret=0;
        for(int i=0;i<nums2.length;i++)
        {
            for(int j=i+1;j<nums2.length;j++)
            {
                long x=(long)(nums2[i])*(long)(nums2[j]);
                 if(map.containsKey(x)){
                    // System.out.println(x+","+i+","+j);
                    ret+=map.get(x);
                }
            }
        }
        return ret;
            
    }
}
