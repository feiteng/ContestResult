class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Arrays.sort(nums1); Arrays.sort(nums2);
        int res1 = helper(nums1, nums2);
        int res2 = helper(nums2, nums1);
        return res1 + res2;
    }
    private int helper(int[] nums1, int[] nums2) {
        int res = 0;
        for (int i = 0;i < nums1.length;i++) {
            res += find(nums2, (long)nums1[i] * nums1[i]);
        }
        return res;
    }
    private int find(int[] nums, long target) {
        int res = 0;
        int left = 0, right = nums.length - 1;
        while (left < right) {
            long cur = (long)nums[left] * nums[right];
            if (cur < target) {
                left++;
            } else if (cur > target) {
                right--;
            } else {
                if (nums[left] == nums[right]) {
                    res += (right - left + 1) * (right - left) / 2;
                    return res;
                }
                int i = left, j = right;
                while (left < right && nums[left] == nums[left + 1]) {
                    left++;
                }
                while (left < right && nums[right] == nums[right - 1]) {
                    right--;
                }
                res += (left - i + 1) * (j - right + 1);
                left++;right--;
            }
        }
        return res;
    }
}
