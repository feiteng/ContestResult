class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int one = helper(nums1, nums2);
        int two = helper(nums2, nums1);
        return one + two;
    }
    private int helper(int[] arr1, int[] arr2) {
        Map<Long, Integer> map = new HashMap<>();
        for (int val : arr1) {
            map.put((long)val * val, map.getOrDefault((long)val * val, 0) + 1);
        }
        
        int res = 0;
        for (int i = 0; i < arr2.length - 1; i++) {
            for (int j = i + 1; j < arr2.length; j++) {
                if (map.containsKey((long)arr2[i] * arr2[j])) {
                    res += map.get((long)arr2[i] * arr2[j]);
                }
            }
        }
        return res;
    }
}
