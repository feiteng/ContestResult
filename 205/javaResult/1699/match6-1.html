class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int res = 0;
        for(int num: nums1){
            res += verify(nums2, num);
        }
        System.out.println(res);
        for(int num: nums2){
            res += verify(nums1, num);
        }
        return res;
    }
    
    private int verify(int[] nums, long tar){
        tar = tar * tar;
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        for(int num: nums){
            int cur = (int)(tar / num);
            if(tar % num == 0 && map.containsKey(cur)){
                res += map.get(cur);
            }
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        return res;
    }
}
