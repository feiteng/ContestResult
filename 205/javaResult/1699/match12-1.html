class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int ans = 0;
        HashMap<Long,Integer> map1sqr = new HashMap<>();
        for(int i :nums1){
            
map1sqr.put(i*1l*i,map1sqr.getOrDefault(i*1l*i,0)+1);
        }
        for(int i = 0;i<nums2.length;i++){
            for(int j = i+1;j<nums2.length;j++){
                long mul = nums2[i] * 1l * nums2[j];
                
if(map1sqr.containsKey(mul)){
                    ans += map1sqr.get(mul);
                }
            }
        }
        HashMap<Long,Integer> map2sqr = new HashMap<>();
        for(int i :nums2){
            
map2sqr.put(i*1l*i,map2sqr.getOrDefault(i*1l*i,0)+1);
        }
        for(int i = 0;i<nums1.length;i++){
            for(int j = i+1;j<nums1.length;j++){
                long mul = nums1[i] * 1l * nums1[j];
                
if(map2sqr.containsKey(mul)){
                    ans += map2sqr.get(mul);
                }
            }
        }
        return ans;
    }
}
