class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int[] cnt1 = new int[100001];
        int[] cnt2 = new int[100001];
        for(int num : nums1)
            cnt1[num]++;
        for(int num : nums2)
            cnt2[num]++;
        int n1 = nums1.length;
        int n2 = nums2.length;
        int res = 0;
        for(int i = 0; i < n1; i ++)
            for(int j = i + 1; j < n1; j++)
            {
                long prod = 1L * nums1[i] * nums1[j];
                long root = (long)Math.sqrt(prod);
                if(root * root == prod){
                    res += cnt2[(int)root];
                }
            }
        for(int i = 0; i < n2; i ++)
            for(int j = i + 1; j < n2; j++)
            {
                long prod = 1L * nums2[i] * nums2[j];
                long root = (long)Math.sqrt(prod);
                if(root * root == prod){
                    res += cnt1[(int)root];
                }
            }
        return res;
    }
}
