class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> s1 = new HashMap();
        HashMap<Long, Integer> s2 = new HashMap();
        int n1 = nums1.length;
        int n2 = nums2.length;
        
for(int i=0; i<n1; i++){
            for(int j=i+1; j<n1; j++){
                long cur = (long)nums1[i]*nums1[j];
                s1.put(cur, s1.getOrDefault(cur, 0)+1);
            }
        }
        
for(int i=0; i<n2; i++){
            for(int j=i+1; j<n2; j++){
                long cur = (long)nums2[i]*nums2[j];
                s2.put(cur, s2.getOrDefault(cur, 0)+1);
            }
        }
        int res = 0;
        for(int num : nums1){
            res+=s2.getOrDefault((long)num*num, 0);
        }
        
for(int num : nums2){
            res+=s1.getOrDefault((long)num*num, 0);
        }
        return res;
    }
}
