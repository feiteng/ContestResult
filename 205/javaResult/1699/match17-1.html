class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int ans = 0;
        Map<Long, Integer> map1 = new HashMap<>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                if (!map1.containsKey(1L * nums1[i] * nums1[j])) {
                    map1.put(1L * nums1[i] * nums1[j], 0);
                }
                
map1.put(1L * nums1[i] * nums1[j], map1.get(1L * nums1[i] * nums1[j]) + 1);
                
                // System.out.println(map1.get(1L * nums1[i] * nums1[j]));
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            
if (map1.containsKey(1L * nums2[i] * nums2[i])) {
                ans += map1.get(1L * nums2[i] * nums2[i]);
            }
        }
        
        Map<Long, Integer> map2 = new HashMap<>();
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                if (!map2.containsKey(1L * nums2[i] * nums2[j])) {
                    map2.put(1L * nums2[i] * nums2[j], 0);
                }
                
map2.put(1L * nums2[i] * nums2[j], map2.get(1L * nums2[i] * nums2[j]) + 1);
                // System.out.println(map2.get(1L * nums2[i] * nums2[j]));
            }
        }
        for (int i = 0; i < nums1.length; i++) {
            
if (map2.containsKey(1L * nums1[i] * nums1[i])) {
                ans += map2.get(1L * nums1[i] * nums1[i]);
            }
        }
        // System.out.println(map1);
        // System.out.println(map2);
        // System.out.println();
        return ans;
    }
}
