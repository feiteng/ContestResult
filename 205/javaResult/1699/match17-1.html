class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> map = new HashMap<>();
        for (int x : nums1) {
            map.put((long)x * (long)x, map.getOrDefault((long)x * (long)x, 0) + 1);
        }
        int res = 0;
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                res += map.getOrDefault((long)nums2[i] *  (long)nums2[j], 0);
            }
        }
        map = new HashMap<>();
        for (int x : nums2) {
            map.put((long)x * (long)x, map.getOrDefault((long)x * (long)x, 0) + 1);
        }
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                res += map.getOrDefault((long)nums1[i] *  (long)nums1[j], 0);
            }
        }
        return res;
    }
}
