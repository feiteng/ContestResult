class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int ans = 0;
        ans += solve(nums1, nums2);
        ans += solve(nums2, nums1);
        // System.out.println(43024*99908);
        // System.out.println(1864*1864);
        return ans;
    }
    
    private int solve(int[] nums1, int[] nums2) {
        int ans = 0;
        Map<Long, Integer> square = new HashMap<>();
        for (int num : nums1) {
            long temp = ((long)num)*((long)num);
            square.put(temp, square.getOrDefault(temp,0) + 1);
        }
        
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i+1; j < nums2.length; j++) {
                long temp = ((long)nums2[i]) * ((long)nums2[j]);
                if (square.containsKey(temp)) {
                    ans += square.get(temp);
                }
            }
        }
        return ans;
    }
}
