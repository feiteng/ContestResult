class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long,Long> map1=new HashMap<>();
        HashMap<Long,Long> map2=new HashMap<>();
        
        for(int i:nums1)
        {
            map1.put((long)((long)i*(long)i),map1.getOrDefault((long)((long)i*(long)i),0L)+1);
        }
        for(int i:nums2)
        {
            
map2.put((long)((long)i*(long)i),map2.getOrDefault((long)((long)i*(long)i),0L)+1);
        }
        int cnt=0;
        for(int i=0;i<nums1.length;i++)
        {
            for(int j=i+1;j<nums1.length;j++)
            {
                
if(map2.containsKey((long)((long)nums1[i]*(long)nums1[j])))
                {
                    cnt+=map2.get((long)((long)nums1[i]*(long)nums1[j]));
                }
            }
        }
        
        for(int i=0;i<nums2.length;i++)
        {
            for(int j=i+1;j<nums2.length;j++)
            {
                
if(map1.containsKey((long)((long)nums2[i]*(long)nums2[j])))
                {
                    cnt+=map1.get((long)((long)nums2[i]*(long)nums2[j]));
                }
            }
        }
        return cnt;
    }
}
