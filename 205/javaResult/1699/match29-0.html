class Solution {
    public int numTriplets(int[] n1, int[] n2) {
        Map<Long, Integer> p1 = new HashMap<>();
        Map<Long, Integer> p2 = new HashMap<>();

        h(n1, p1);
        h(n2, p2);
        
        int ret = 0;
        ret+=h1(n2, p1);
        ret+=h1(n1, p2);
        return ret;
    }
    
    void h(int[] n1, Map<Long,Integer> p1) {
        for (int i =0;i<n1.length;i++) {
            for (int j = i+1;j < n1.length; j++) {
                long tmp = (long)n1[i]*(long)n1[j];
                p1.put(tmp, p1.getOrDefault(tmp,0)+1);               
            }
        }
    }
    
    int h1 (int[] n2, Map<Long, Integer> p1) {
        int ret = 0;
        for (int i = 0; i < n2.length; i++) {
            long tmp = (long)n2[i]*(long)n2[i];
            if (p1.containsKey(tmp)) {
                ret+=p1.get(tmp);
            }
        }
        return ret;
    }
}
