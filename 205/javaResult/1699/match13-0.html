class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int cnt = 0;
        int n1 = nums1.length;
        int n2 = nums2.length;
        for (int i = 0; i < n1; i++) {
            long prod = (long) nums1[i] * nums1[i];
            // if (i == 0) {
            //     System.out.println(prod);
            // }
            for (int j = 0; j < n2; j++) {
                if ((long) nums2[j] * nums2[j] > prod) break;
                if (prod % nums2[j] == 0) {
                    long val = (prod / nums2[j]);
                    int l = bsl(nums2, j + 1, n2 - 1, val);
                    int r = bsr(nums2, j + 1, n2 - 1, val);
                    // if (i == 0) {
                    //     System.out.println(i + " " + l + " " + r + " " + val + " " + j);
                    // }
                    if (l > 0 && r > 0) {
                        cnt += r - l + 1;
                    }
                }
            }
        }
        
        for (int j = 0; j < n2; j++) {
            long prod = (long) nums2[j] * nums2[j];
            for (int i = 0; i < n1; i++) {
                if ((long) nums1[i] * nums1[i] > prod) break;
                if (prod % nums1[i] == 0) {
                    long val = (prod / nums1[i]);
                    int l = bsl(nums1, i + 1, n1 - 1, val);
                    int r = bsr(nums1, i + 1, n1 - 1, val);
                    if (l > 0 && r > 0) {
                        cnt += r - l + 1;
                    }
                }
            }
        }
        
        return cnt;
    }
    
    int bsl(int[] arr, int l, int r, long t) {
        if (l > r) return -1;
        int m;
        while (l + 1 < r) {
            m = l + (r - l) / 2;
            if (arr[m] < t) {
                l = m;
            } else {
                r = m;
            }
        }
        
        if (arr[l] == t) return l;
        if (arr[r] == t) return r;
        return -1;
    }
    
    int bsr(int[] arr, int l, int r, long t) {
        if (l > r) return -1;
        int m;
        while (l + 1 < r) {
            m = l + (r - l) / 2;
            if (arr[m] > t) {
                r = m;
            } else {
                l = m;
            }
        }
        
        if (arr[r] == t) return r;
        if (arr[l] == t) return l;
        return -1;
    }
}
