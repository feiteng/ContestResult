class Solution {
    
    
    public int g(int nums1[],int nums2[]){
        
        Map<Long,Long> mp1 = new HashMap<>();
        long s=  0;
        for(int u:nums1){
            long v = (long)u*(long)u;
            mp1.put(v,mp1.getOrDefault(v,0L)+1L);
        }

        for(int i=0;i<nums2.length;++i){
            for(int j=i+1;j<nums2.length;++j){
                
                long v = (long)nums2[i]*(long)nums2[j];
                
                if(mp1.containsKey(v)){
                    
                    s += mp1.get(v);
                }
                
            }
        }
        return (int)s;
        
    }
    
    
    public int numTriplets(int[] nums1, int[] nums2) {
        
        return g(nums1,nums2) + g(nums2,nums1);
        
        
        
        
    }
}
