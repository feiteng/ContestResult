class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long,Integer> map1 = new HashMap<>(), map2 = new HashMap<>();
        for(long n : nums1){
            map1.put(n * n,map1.getOrDefault(n * n , 0) + 1);
        }
        for(long n : nums2){
            map2.put(n * n,map2.getOrDefault(n * n , 0) + 1);
        }

        int cnt = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                cnt += map2.getOrDefault((long)nums1[i] * (long)nums1[j],0);
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                cnt += map1.getOrDefault((long)nums2[i] * (long)nums2[j],0);
            }
        }
        return cnt;
    }
}
