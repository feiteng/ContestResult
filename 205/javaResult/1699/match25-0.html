class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int ans = 0;
        Map<Long, Integer> count1 = new HashMap<>();
        Map<Long, Integer> count2 = new HashMap<>();
        for(int i = 0; i < nums1.length; i++){
            for(int j = i+1; j < nums1.length; j++){
                long key = (long)nums1[i] * (long)nums1[j];
                count1.put( key,  count1.getOrDefault(key, 0) + 1 );
            }
        }
        
        for(int i = 0; i < nums2.length; i++){
            for(int j = i+1; j < nums2.length; j++){
                long key = (long)nums2[i] * (long)nums2[j];
                count2.put( key,  count2.getOrDefault(key, 0) + 1 );
            }
        }
        
        for(int i = 0; i < nums1.length; i++){
            long key = (long)nums1[i] * (long)nums1[i];
            ans += count2.getOrDefault(key, 0);
        }
        
        for(int i = 0; i < nums2.length; i++){
            long key = (long)nums2[i] * (long)nums2[i];
            ans += count1.getOrDefault(key, 0);
        }       
        
        return ans;
    }
}
