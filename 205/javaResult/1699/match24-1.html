class Solution {
    public int numTriplets(int[] a1, int[] a2) { 
        return num(a1, a2) + num(a2, a1);
    }

    private int num(int[] a1, int[] a2) {
        Map<Long, Integer> h = new HashMap<>();
        for (int i = 0; i < a1.length; i++) {
            long x = (long) a1[i] * a1[i];
            h.put(x, h.getOrDefault(x, 0) + 1);
        }
        int count = 0;
        for (int i = 0; i < a2.length; i++) {
            for (int j = i+1; j < a2.length; j++) {
                count += h.getOrDefault((long) a2[i] * a2[j], 0);
            }
        }
        return count;
    }
    
    private boolean ok(int[] a, int[] b, int i, int j, int k) {
        return (long) a[i] * a[i] == (long) b[j] * b [k];
    }
}
