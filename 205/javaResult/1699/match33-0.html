class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> map1 = new HashMap<>();
        HashMap<Long, Integer> map2 = new HashMap<>();
        for (int j = 0; j < nums1.length; j++) {
            for (int k = j + 1; k < nums1.length; k++) {
                long ind = ((long)nums1[j])*((long)nums1[k]);
                map1.put(ind, map1.getOrDefault(ind, 0) + 1);
            }
        }
        for (int j = 0; j < nums2.length; j++) {
            for (int k = j + 1; k < nums2.length; k++) {
                long ind = ((long)nums2[j])*((long)nums2[k]);
                map2.put(ind, map2.getOrDefault(ind, 0) + 1);
            }
        }
        int ret = 0;
        for (int j = 0; j < nums1.length; j++) {
            ret += map2.getOrDefault(((long)nums1[j])*((long)nums1[j]), 0);
        }
        for (int j = 0; j < nums2.length; j++) {
            ret += map1.getOrDefault(((long)nums2[j])*((long)nums2[j]), 0);
        }
        return ret;
    }
}
