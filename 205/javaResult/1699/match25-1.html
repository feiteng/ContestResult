class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int ans =0;
        HashMap<Long, Integer> hs = new HashMap<>();
        int n = nums1.length, m = nums2.length;
        for(int i = 0; i< n; i++){
            long val = nums1[i];
            val *= val;
            if(hs.containsKey(val)) hs.replace(val, hs.get(val)+1);
            else hs.put(val,1);
        }
        for(int i = 0; i< m; i++){
            long val = nums2[i];
            for(int j = i+1; j <m ;j++){
                val = nums2[i];
                val *= nums2[j];
                if(hs.containsKey(val)){
                    ans+=hs.get(val);
                }
            }
        }
        //System.out.println(hs +" "+ans);
        hs.clear();
        for(int i = 0; i< m; i++){
            long val = nums2[i];
            val *= val;
            if(hs.containsKey(val)) hs.replace(val, hs.get(val)+1);
            else hs.put(val,1);
        }
        for(int i = 0; i< n; i++){
            long val = nums1[i];
            for(int j = i+1; j <n ;j++){
                val = nums1[i];
                val *= nums1[j];
                if(hs.containsKey(val)){
                    ans+=hs.get(val);
                }
            }
        }
        //System.out.println(hs +" "+ans);
        return ans;
    }
}
