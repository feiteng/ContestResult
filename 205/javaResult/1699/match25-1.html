class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> tm1 = new HashMap<>();
        Map<Long, Integer> tm2 = new HashMap<>();
        
        int n1 = nums1.length, n2 = nums2.length;
        
for (int i = 0; i < n1 - 1; i++) {
            for (int j = i + 1; j < n1; j++) {
                long tmp = (long)(nums1[i]) * (long)(nums1[j]);
                tm1.put(tmp, tm1.getOrDefault(tmp, 0) + 1);
            }
        }
        
        for (int i = 0; i < n2 - 1; i++) {
            for (int j = i + 1; j < n2; j++) {
                long tmp = (long)(nums2[i]) * (long)(nums2[j]);
                tm2.put(tmp, tm2.getOrDefault(tmp, 0) + 1);
            }
        }
        
        int res = 0;
        
        
for (int i = 0; i < n1; i++) {
            long tmp = (long)(nums1[i]) * (long)(nums1[i]);
            res += tm2.getOrDefault(tmp, 0);
        }
        
        for (int i = 0; i < n2; i++) {
            long tmp = (long)(nums2[i]) * (long)(nums2[i]);
            res += tm1.getOrDefault(tmp, 0);
        }
        
        return res;
    }
}
