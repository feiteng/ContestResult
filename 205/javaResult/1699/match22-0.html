class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> m1 = new HashMap<>();
        for(int i = 0; i < nums1.length; ++i) {
            for(int j = i + 1; j < nums1.length; ++j) {
                long t = 1L* nums1[i] * nums1[j];
                m1.put(t, 1 + m1.getOrDefault(t, 0));
            }
        }
        int res = 0;
        for(int x : nums2) {
            res += m1.getOrDefault(1L * x * x, 0);
        }
        HashMap<Long, Integer> m2 = new HashMap<>();
        for(int i = 0; i < nums2.length; ++i) {
            for(int j = i + 1; j < nums2.length; ++j) {
                long t = 1L* nums2[i] * nums2[j];
                m2.put(t, 1 + m2.getOrDefault(t, 0));
            }
        }
        for(int x : nums1) {
            res += m2.getOrDefault(1L * x * x, 0);
        }
        return res;
    }
}
