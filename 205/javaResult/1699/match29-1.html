class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> set1 = new HashMap<>();
        Map<Long, Integer> set2 = new HashMap<>();
        
        putSquare(nums1, set1);
        putSquare(nums2, set2);
        
        int sum = 0;
        sum += calculate(set1, nums2);
        sum += calculate(set2, nums1);
        
        return sum;
    }
    
    void putSquare(int[] nums, Map<Long, Integer> set) {
        for (int i : nums) {
            long n = (long)i * (long)i;
            set.putIfAbsent(n, 0);
            set.put(n, set.get(n) + 1);
        
}
    }
    
    int calculate(Map<Long, Integer> set, int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                
long n = (long)nums[i] * (long)nums[j];
                if (set.containsKey(n)) {
                    sum += set.get(n);
                }
            }
        }
        
        return sum;
    }
}
