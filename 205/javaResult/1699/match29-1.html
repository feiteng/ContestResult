class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        return helper(nums1, nums2) + helper(nums2, nums1);
    }

    int helper(int[] nums1, int[] nums2) {
        int res = 0;
        int n1 = nums1.length;
        int n2 = nums2.length;
        Map<Long, Integer> hm = new HashMap<>();
        for (int v : nums2) {
            hm.put(1L * v * v, hm.getOrDefault(1L * v * v, 0) + 1);
        }
        for (int j = 0; j < n1; j++) {
            for (int k = j + 1; k < n1; k++) {
                long mul = 1L * nums1[j] * nums1[k];
                res += hm.getOrDefault(mul, 0);
            }
        }
        return res;
    }
}
