class Solution {

    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> map1 = new HashMap<>(), map2 = new HashMap<>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                map1.put((long) nums1[i] * nums1[j], map1.getOrDefault((long) nums1[i] * nums1[j], 0) + 1);
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                map2.put((long) nums2[i] * nums2[j], map2.getOrDefault((long) nums2[i] * nums2[j], 0) + 1);
            }
        }
        int count = 0;
        for (long num : nums1) {
            count += map2.getOrDefault(num * num, 0);
        }
        for (long num : nums2) {
            count += map1.getOrDefault(num * num, 0);
        }
        return count;
    }
}
