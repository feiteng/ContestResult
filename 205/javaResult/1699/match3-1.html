import java.util.HashMap;
import java.util.Map;

class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        return help(nums1, nums2) + help(nums2, nums1);
    }

    private int help(int[] nums1, int[] nums2) {
        int ans = 0;
        for (int i : nums1) {
            long v = i;
            v *= v;
            Map<Integer, Integer> m = new HashMap<>();
            for (int i1 : nums2) {
                if (v % i1 == 0) {
                    ans += m.getOrDefault((int) (v / i1), 0);
                    m.put(i1, m.getOrDefault(i1, 0) + 1);
                }
            }
        }
        return ans;
    }
}
