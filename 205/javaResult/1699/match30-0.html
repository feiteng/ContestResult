class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> map1 = new HashMap<>();
        Map<Long, Integer> map2 = new HashMap<>();
        int n = nums1.length; int m = nums2.length;
        for (int i = 0; i < n; i++) {
            long val = (long) nums1[i] * nums1[i];
            map1.put(val, map1.getOrDefault(val, 0) + 1);
        }
        for (int i = 0; i < m; i++) {
            long val = (long) nums2[i] * nums2[i];
            map2.put(val, map2.getOrDefault(val, 0) + 1);
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                long val = (long) nums1[i] * nums1[j];
                res += map2.getOrDefault(val , 0);
            }
        } 
        
        for (int i = 0; i < m; i++) {
            for (int j = i + 1; j < m; j++) {
                long val = (long) nums2[i] * nums2[j];
                res += map1.getOrDefault(val, 0);
            }
        }
        return res;
    }
}
