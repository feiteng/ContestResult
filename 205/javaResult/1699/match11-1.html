class Solution {
    
public int numTriplets(int[] nums1, int[] nums2) {
        return helper(nums1, nums2) + helper(nums2, nums1);
    }
    
    
private int helper(int[] nums1, int[] nums2) {
        int res = 0;
        for(int n1 : nums1) {
            HashMap<Integer, Integer> map = new HashMap<>();
            long sq = (long)n1 * (long)n1;
            
            for(int n2 : nums2) {
                
if(sq % n2 == 0) {
                    int j = (int)(sq/(long)n2);
                    res += map.getOrDefault(j, 0);
                }
                map.put(n2, map.getOrDefault(n2, 0)+1);
            }
        }
        return res;
    }
}
