class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> square1 = new HashMap<>(), square2 = new HashMap<>();
        for (int i : nums1) {
            long l = (long)i * i;
            if (!square1.containsKey(l)) {
                square1.put(l, 0);
            }
            square1.put(l, square1.get(l) + 1);
        }
        for (int i : nums2) {
            long l = (long)i * i;
            if (!square2.containsKey(l)) {
                square2.put(l, 0);
            }
            square2.put(l, square2.get(l) + 1);
        }
        int res = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                if (square2.containsKey((long)nums1[i] * nums1[j])) {
                    res += square2.get((long)nums1[i] * nums1[j]);
                }
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                if (square1.containsKey((long)nums2[i] * nums2[j])) {
                    res += square1.get((long)nums2[i] * nums2[j]);
                }
            }
        }
        return res;
    }
}
