class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        HashMap<Long, Integer> s = new HashMap<>();
        int ret = 0;
        for(int i = 0; i < nums1.length; i++) {
            long number = (long)nums1[i] * (long)nums1[i];
            s.put(number, s.getOrDefault(number, 0) + 1);
        }
        for(int i = 0; i < nums2.length; i++) {
            for(int j = i + 1; j < nums2.length; j++) {
                long number = (long)nums2[i] * (long)nums2[j];
                if (s.containsKey(number)) {
                    ret += s.get(number);
                }
            }
        }
        s = new HashMap<>();
        for(int i = 0; i < nums2.length; i++) {
            long number = (long)nums2[i] * (long)nums2[i];
            s.put(number, s.getOrDefault(number, 0) + 1);
        }
        for(int i = 0; i < nums1.length; i++) {
            for(int j = i + 1; j < nums1.length; j++) {
                long number = (long)nums1[i] * (long)nums1[j];
                if (s.containsKey(number)) {
                    ret+=s.get(number);
                }
            }
        }
        return ret;
    }
}
