class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int res = 0;
        Map<Long, Integer> square1 = new HashMap<>();
        for(int num : nums1){
            long p = (long)num * num;
            square1.put(p, square1.getOrDefault(p, 0) + 1);
        }
        for(int j = 0; j < nums2.length; j++){
            for(int k = j + 1; k < nums2.length; k++){
                long p = (long)nums2[j] * nums2[k];
                if(square1.containsKey(p)){
                    res += square1.get(p);
                }
            }
        }
        
        Map<Long, Integer> square2 = new HashMap<>();
        for(int num : nums2){
            long p = (long)num * num;
            square2.put(p, square2.getOrDefault(p, 0) + 1);
        }
        for(int j = 0; j < nums1.length; j++){
            for(int k = j + 1; k < nums1.length; k++){
                long p = (long)nums1[j] * nums1[k];
                if(square2.containsKey(p)){
                    res += square2.get(p);
                }
            }
        }
        
        return res;
    }
}
