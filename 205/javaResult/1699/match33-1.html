class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        int res=0;
        Map<Long,Integer> set1=new HashMap<>();
        for(int a:nums1)
        {
            Long n1=new Long(a);
            
long r=n1*n1;
            set1.put(r,set1.getOrDefault(r,0)+1);
        }
        for(int i=0;i<nums2.length;i++){
            for(int j=i+1;j<nums2.length;j++){
                Long n1=new Long(nums2[i]);
                Long n2=new Long(nums2[j]);
                long r=n1*n2;
                res+=set1.getOrDefault(r,0);
            }
        }
        
        Map<Long,Integer> set2=new HashMap<>();
        for(int a:nums2)
        {
            Long n1=new Long(a);
            long r=n1*n1;
            
set2.put(r,set2.getOrDefault(r,0)+1);
        }
        for(int i=0;i<nums1.length;i++){
            for(int j=i+1;j<nums1.length;j++){
                Long n1=new Long(nums1[i]);
                Long n2=new Long(nums1[j]);
                long r=n1*n2;
                res+=
set2.getOrDefault(r,0);
            }
        }
        
        return res;
    }
}
