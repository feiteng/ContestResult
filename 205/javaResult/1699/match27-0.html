class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> map1 = sqrCount(nums1), map2 = sqrCount(nums2);
        int count = 0;
        for (int n : nums1) {
            count += map2.getOrDefault(Long.valueOf(n) * n, 0);
        }
        for (int n : nums2) {
            count += map1.getOrDefault(Long.valueOf(n) * n, 0);
        }
        return count;
    }
    
    private Map<Long, Integer> sqrCount(int[] nums) {
        Map<Long, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                long sqr = Long.valueOf(nums[i]) * nums[j];
                map.put(sqr, map.getOrDefault(sqr, 0) + 1);
            }
        }
        return map;
    }
}
