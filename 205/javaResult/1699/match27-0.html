	class Solution {
	    public int numTriplets(int[] a, int[] b) {
	    	return go(a, b) + go(b, a);
	    }
	    
	    int go(int[] a, int[] b)
	    {
	    	int[] f = new int[100001];
	    	for(int x : a)f[x]++;
	    	
	    	int ret = 0;
	    	for(int i = 0;i < b.length;i++){
	    		for(int j = i+1;j < b.length;j++){
	    			long v = (long)b[i]*b[j];
	    			long s = (long)Math.sqrt(v);
	    			if(s * s == v){
	    				ret += f[(int)s];
	    			}
	    		}
	    	}
	    	return ret;
	    }
	}
