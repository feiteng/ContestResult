class Solution {
    public int numTriplets(int[] nums1, int[] nums2) {
        Map<Long, Integer> map1 = new HashMap<>();
        Map<Long, Integer> map2 = new HashMap<>();
        for (int i = 0; i < nums1.length; i++) {
            map1.put((long) nums1[i] * nums1[i], map1.getOrDefault((long) nums1[i] * nums1[i], 0) + 1);
        }
        for (int i = 0; i < nums2.length; i++) {
            map2.put((long) nums2[i] * nums2[i], map2.getOrDefault((long) nums2[i] * nums2[i], 0) + 1);
        }
        int rst = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = i + 1; j < nums1.length; j++) {
                if (map2.containsKey((long) nums1[i] * nums1[j])) {
                    //System.out.println(i + ":" + j);
                    //System.out.println(nums1[i] + ":" + nums1[j] + ":" + nums1[i] * nums1[j]);
                    //System.out.println(43024 * 99908);
                    //System.out.println("qiguai" + 1864 * 1864);
                    rst += map2.get((long) nums1[i] * nums1[j]);
                }
            }
        }
        for (int i = 0; i < nums2.length; i++) {
            for (int j = i + 1; j < nums2.length; j++) {
                if (map1.containsKey((long) nums2[i] * nums2[j])) {
                    //System.out.println(i + ":" + j);
                    //System.out.println(map1.get(nums2[i] * nums2[j]));
                    rst += map1.get((long) nums2[i] * nums2[j]);
                }
            }
        }
        return rst;
    }
}
