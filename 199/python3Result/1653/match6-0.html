# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def countPairs(self, root: TreeNode, distance: int) -> int:
        
        self.global_answer = 0
        def visit(node = root) :
            if node.left is None and node.right is None :
                return [0]
            
            to_retl, to_retr = [], []
            if not node.left is None :
                to_retl += visit(node.left)
            if not node.right is None :
                to_retr += visit(node.right)
            
            for a in to_retl :
                for b in to_retr :
                    if a+b+2<=distance :
                        self.global_answer += 1
            to_ret = [t+1 for t in to_retl+to_retr if t < distance]
            return to_ret
            
        visit()
        return self.global_answer
