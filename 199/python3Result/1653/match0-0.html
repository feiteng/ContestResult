# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def countPairs(self, root: TreeNode, distance: int) -> int:
        
        res = [0]
        
        def helper(node):
            if node == None:
                return []
            if node.left == None and node.right == None:
                return [1]
            leftArr = helper(node.left)
            rightArr = helper(node.right)
            curRes = 0
            retArr = []
            for l in leftArr:
                for r in rightArr:
                    if (l + r) <= distance:
                        res[0] += 1
            for l in leftArr:
                retArr.append(l+1)
            for r in rightArr:
                retArr.append(r+1)
            
            return retArr
        
        helper(root)
        return res[0]
