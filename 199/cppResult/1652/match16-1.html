class Solution {
public:
    int minFlips(string target) {
        int n = (int )target.size();

        int ans = 0, flg = 0;
        for (int i = 0; i < n; i ++) {
            int cur = target[i] - '0';
            if (cur ^ flg) {
                ++ ans;
                flg ^= 1;
            }
        }
        return ans;
    }
};
