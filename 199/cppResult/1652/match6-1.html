class Solution {
public:
    int minFlips(string target) {
        int n = target.size();

        int cur = 0;
        int res = 0;
        for(int i=0;i<n;i++) {
            if(target[i] == cur + '0') continue;
            res++;
            cur = 1 - cur;
        }
        return res;
    }
};
