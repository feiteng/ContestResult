class Solution {
public:
    int minFlips(string target) {
        int n = target.size();
        int rev = 0, res = 0;
        for (int i = 0, c; i < n; ++i) {
            c = target[i]-'0';
            if (c != rev) {
                rev ^= 1;
                ++res;
            }
        }
        return res;
    }
};
