class Solution {
public:
    int minFlips(string target) {
        int ans = 0;
        int cur = 0;
        for (int i = 0; i < target.size(); i++) {
            if (target[i] - '0' == cur) continue;
            ans ++;
            cur = 1 - cur;
        }
        return ans;
    }
};
