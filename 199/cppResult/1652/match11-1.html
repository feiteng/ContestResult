class Solution {
public:
    int minFlips(string target) {
        int res = 0;
        int press = 0;
        for (int i = 0; i < target.size(); i++) {
            if (target[i] == '1' && press == 0) {
                res++;
                press ^= 1;
            } else if (target[i] == '0' && press == 1) {
                res++;
                press ^= 1;
            }
        }
        return res;
    }
};
