class Solution {
public:
    int minFlips(string t) {
        int moves = 0;
        char last = '0';
        for (int i = 0; i < t.size(); i++) {
            if (t[i] != last) {
                last = t[i];
                moves++;
            }
        }
        return moves;
    }
};
