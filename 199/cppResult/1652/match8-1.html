class Solution {
public:
    int minFlips(string target) {
        int n = target.size();
        int ans = 0;
        for(int i = 0; i < n; i++) {
            if(ans == 0 && target[i] == '0')
                continue;
            int j = i;
            while(j < n && target[j] == target[i])
                j++;
            ans++;
            i = j - 1;
        }
        return ans;
    }
};
