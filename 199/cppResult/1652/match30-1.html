// constant optimization is important

class Solution {
public:
    int minFlips(string target) {
        bool flipped = false;
        int ret = 0;
        for(int i = 0; i < sz(target); i++) {
          bool want = target[i] == '1';
          if(flipped != want) {
            ret++;
            flipped ^= true;
          }
        }
        return ret;
    }
};
