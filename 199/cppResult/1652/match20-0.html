class Solution {
public:
    int minFlips(string s) {
        int n = s.length(), pre = 0, ans = 0;
        for (int i = 0; i < n; ++i) {
            int cur = s[i] - '0';
            if (cur != pre) {
                ++ans;
                pre = cur;
            }
        }
        return ans;
    }
};
