class Solution {
public:
    int minFlips(string s) {
        int cur = 0;
        int ret = 0;
        for(int i=0;i<s.size();i++){
            int tt = s[i]-'0';
            if(cur%2==tt){
                continue;
            }
            ret++;
            cur++;
        }
        return ret;
    }
};
