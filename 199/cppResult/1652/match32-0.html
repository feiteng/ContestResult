class Solution {
public:
    int minFlips(string target) {
        bool start = false;
        int res = 0;
        for (int i = 0; i < target.length(); ++i) {
            if (target[i] == '1')
                start = true;
            if (start) {
                if (i > 0) {
                    res += (target[i] != target[i - 1]);
                } else {
                    ++res;
                }
            }
        }
        return res;
    }
};
