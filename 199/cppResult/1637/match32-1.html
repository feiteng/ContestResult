#define FOR(i,a,n) for (int i = (a); i < (n); ++i)
#define FORE(i,a,n) for (int i = (a); i <= (n); ++i)
#define FORD(i,a,b) for (int i = (a); i >= (b); --i)
#define REP(i,n) FOR(i,0,n)
#define REPE(i,n) FORE(i,0,n)
#define LL long long
#define FIR(n) REP(i,n)
#define FJR(n) REP(j,n)

#define FI FIR(n)
#define FJ FJR(n)
#define FR(i,a) FOR(i,a,n)
#define REPN(i) REP(i,n)

#define GI(n) scanf("%d", &n)
#define GI2(n,m) scanf("%d %d", &n, &m)


const int INF = 1e9;

int f[110][110];


inline int computeLen(int cnt) {
	if (cnt == 1)
		return 1;
	if (cnt < 10)
		return 2;
	if (cnt < 100)
		return 3;
	return 4;
}
class Solution {
public:
	int getLengthOfOptimalCompression(string s, int k) {
		int n = s.size();
		FORE(j, 0, n) f[0][j] = INF;
		f[0][0] = 0;
		REPN(pos) {
			FORE(j, 0, n) f[pos+1][j] = f[pos][j];

			int cnt = 0;
			for (int j = pos; j >= 0; --j) {
				if (s[j] == s[pos])
					++cnt;
				
				int cost = computeLen(cnt);
				FORE(q, cnt, pos + 1)
					f[pos + 1][q] = min(f[pos + 1][q], f[j][q - cnt] + cost);
			}
		}

		int res = INF;
		FORE(len, n - k, n)
			res = min(res, f[n][len]);
		return res;
	}
};
