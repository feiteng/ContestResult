class Solution {
public:
    int getLengthOfOptimalCompression(string s, int k) {
        int n = s.size();

        const int INF = 1e+9 + 42;
        vector<vector<int>> f(n + 1, vector<int>(k + 1, INF));
        f[0][0] = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j <= k; ++j) {
                if (f[i][j] == INF) {
                    continue;
                }
                if (j < k) {
                    f[i + 1][j + 1] = min(f[i + 1][j + 1], f[i][j]);
                }
                int c = 0;
                for (int z = i; z < n; ++z) {
                    c += s[z] == s[i];
                    int x = j + z - i + 1 - c;
                    if (x > k) {
                        break;
                    }
                    int len = to_string(c).size();
                    f[z + 1][x] = min(f[z + 1][x], f[i][j] + 1 + (c > 1? len : 0));
                }
            }
        }
        return f[n][k];
    }
};
