class Solution {
public:
  int get(int cnt) {
    if (cnt == 1) return 0;
    return to_string(cnt).size();
  }
  int getLengthOfOptimalCompression(string s, int k) {
    int N = s.size();
  //  cout << N << endl;
    vector<vector<int>> dp(N+1, vector<int>(N+1, 0x3f3f3f3f));
    dp[0][0] = 0;
    for (int i = 0; i < N; i++) {
 //     cout << "##" << i << " " << endl;
      for (int j = 0; j <= k; j++) {
  //      cout << "#" << i << " " << j << ": " << dp[i][j] << endl;
        if (dp[i][j] == 0x3f3f3f3f) continue;
        int cnt1 = 0;
        int cnt2 = 1;
        dp[i+1][j+1] = min(dp[i+1][j+1], dp[i][j]);
        dp[i+1][j] = min(dp[i+1][j], dp[i][j]+1);
        for (int i1 = i+1; i1 < N; i1++) {
          if (s[i1] != s[i]) cnt1++;
          else cnt2++;
          dp[i1+1][j+cnt1] = min(dp[i1+1][j+cnt1], dp[i][j] + 1 + get(cnt2));
   //       cout << i1 + 1 << " " << j + cnt1 << " " << dp[i1+1][j+cnt1] << endl;
        }
      }
    }
    return dp[N][k];
  }
};
