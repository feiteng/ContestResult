int f[105][105];
void upd(int&ret, int val){
    if(ret==-1||ret>val)ret = val;
}

class Solution {
public:
    int getLengthOfOptimalCompression(string s, int m) {
        int n = s.size();
        memset(f, -1, sizeof(f));
        f[0][0] = 0;
        for(int i=0;i<n;i++)
            for(int j=0;j<=m;j++){
                if(f[i][j]==-1)continue;
                unordered_map<int, int>mp;
                int mx = 0;
                for(int k=i+1;k<=n;k++){
                    int cur = s[k-1]-'a';
                    mp[cur]++;
                    mx = max(mx, mp[cur]);
                    if(j+k-i-mx<=m){
                        upd(f[k][j+k-i-mx], f[i][j]+1+(mx<10?(mx==1?0:1):(mx<100?2:3)));
                    }
                    if(j+k-i<=m){
                        upd(f[k][j+k-i], f[i][j]);
                    }
                }
            }
        int ret = -1;
        for(int i=0;i<=m;i++)if(f[n][i]!=-1)upd(ret, f[n][i]);
        return ret;
    }
};
