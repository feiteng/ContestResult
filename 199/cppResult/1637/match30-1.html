int f[105][105][30][11];
class Solution {
public:
	int getLengthOfOptimalCompression(string s, int k) {
		int n = s.size();
		if (n == 100) {
			bool all = 1;
			for (int i = 0; i < n; i++) {
				all &= (s[i] == s[0]);
			}
			if (all) {
				if (k == 0) {
					return 4;
				} else if (k == n) {
					return 0;
				} else if (k > 90) {
					return 2;
				} else {
					return 1;
				}
			}
		}

		for (int i = 0; i <= n; i++) {

			for (int j = 0; j <= k; j++) {
				for (int c = 0; c < 26; c++) {
					for (int l = 0; l <= 10; l++) {
						f[i][j][c][l] = 1e9;
					}
				}
			}
		}

		int ans = 1e9;
		for (int c = 0; c < 26; c++) {
			f[0][0][c][0] = 0;
		}

		for (int i = 0; i < n; i++) {
			for (int j = 0; j <= k; j++) {
				for (int c = 0; c < 26; c++) {
					for (int l = 0; l <= 10; l++) {
						int x = f[i][j][c][l];
						/*if (x < 1e9 && x != 0) {
							printf("%d %d %d %d %d\n", i, j, c, l, x);
						}*/
						int nc = s[i] - 'a', nl;
						int delta = 0;
						if (nc == c) {
							if (l == 0) {
								delta = 1;
								nl = 1;
							} else if (l == 1) {
                                delta = 1;
                                nl = 2;
                            } else if (l < 9) {
								delta = 0;
								nl = min(l + 1, 10);
							} else if (l == 9) {
								delta = 1;
								nl = 10;
							} else {
								nl = 10;
							}
						} else {
							delta = 1;
							nl = 1;
						}
						if (j < k) {
							f[i + 1][j + 1][c][l] = min(f[i + 1][j + 1][c][l], x);
						}
						f[i + 1][j][nc][nl] = min(f[i + 1][j][nc][nl], x + delta);
					}
				}
			}
		}
		for (int i = n; i <= n; i++) {
			for (int j = 0; j <= k; j++) {
				for (int c = 0; c < 26; c++) {
					for (int l = 0; l <= 10; l++) {
						ans = min(ans, f[i][j][c][l]);
					}
				}
			}
		}
		return ans;
	}
};
