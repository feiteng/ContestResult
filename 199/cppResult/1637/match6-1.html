int dp[105][105][27][105];
int inf = 1e9;
class Solution {
public:
    int getLengthOfOptimalCompression(string s, int K) {
      int n = s.size();
      for (int i = 0; i <= n; i++) {

        for (int j = 0; j <= min(i, K); j++) {
          for (int k = 0; k <= 26; k++) {
            for (int l = 0; l <= i - j; l++) {
              dp[i][j][k][l] = inf;
            }
          }
        }
      }
      dp[0][0][0][0] = 0;
      for (int i = 0; i < n; i++) {
        int ch = s[i] - 'a' + 1;

        for (int j = 0; j <= min(i, K); j++) {
          for (int k = 0; k <= 26; k++) {
            for (int l = 0; l <= i - j; l++) {
              if (dp[i][j][k][l] == inf) continue;
              if (ch == k) {
                int x = 0;
                if (l == 1 || l == 9 || l == 99) x = 1;
                dp[i + 1][j][k][l + 1] = min(dp[i + 1][j][k][l + 1], dp[i][j][k][l] + x);
              } else {
                dp[i + 1][j][ch][1] = min(dp[i + 1][j][ch][1], dp[i][j][k][l] + 1);
              }
              dp[i + 1][j + 1][k][l] = min(dp[i + 1][j + 1][k][l], dp[i][j][k][l]);
            }
          }
        }
      }
      int ans = inf;
      for (int i = 0; i <= 26; i++) {
        for (int j = 0; j <= n - K; j++) {
          ans = min(ans, dp[n][K][i][j]);
        }
      }
      return ans;
    }
};
