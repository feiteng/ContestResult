const int N = 105;
const int L = 27;
int f[N][N][L][N];
int lim[L];
const int INF = 100000000;

void upd(int &x, int y) {
    x = min(x, y);
}

inline int delta(int k) {
    if (k == 2 || k == 10 || k == 100) return 1;
    else return 0;
}

class Solution {
public:
    int getLengthOfOptimalCompression(string s, int K) {
        int n = s.length();
        for (int i = 0; i < L; i++) lim[i] = 0;
        for (int i = 0; i < n; i++) {
            lim[s[i] - 'a']++;
        }
        lim[L - 1] = 1;

        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= K && j <= i; j++) {
                for (int c = 0; c < L; c++) {
                    for (int k = 0; k <= n; k++) {
                        f[i][j][c][k] = INF;
                    }
                }
            }
        }
        f[1][0][s[0] - 'a'][1] = 1;
        f[1][1][L - 1][1] = 0;
        for (int i = 2; i <= n; i++) {
            for (int j = 0; j <= K && j <= i; j++) {
                int cc = s[i - 1] - 'a';
                if (j > 0) {
                    // f[i - 1][j - 1][*][*] -> f[i][j][*][*]
                    for (int c = 0; c < L; c++) {
                        for (int k = 0; k < i && k <= lim[c]; k++) 
                            upd(f[i][j][c][k], f[i - 1][j - 1][c][k]);
                    }
                }
                if (j != i) {
                    // f[i - 1][j][cc][k] -> f[i][j][cc][k + 1]
                    upd(f[i][j][cc][2], f[i - 1][j][cc][1] + 1);
                    for (int k = 3; k <= i; k++) {
                        upd(f[i][j][cc][k], f[i - 1][j][cc][k - 1] + delta(k));
                    }
                    // f[i - 1][j][^cc][*] -> f[i][j][cc][1];
                    for (int c = 0; c < L; c++) if (c != cc) {
                        for (int k = 0; k <= i - 1 && k <= lim[c]; k++) {
                            upd(f[i][j][cc][1], f[i - 1][j][c][k] + 1);
                        }

                    }
                }
            }
        }
        int ans = INF;
        for (int j = 0; j <= K; j++) {
            for (int c = 0; c < L; c++) {
                for (int k = 0; k <= n; k++) {
                    upd(ans, f[n][j][c][k]);
                }
            }
        }
        return ans;
    }
};
