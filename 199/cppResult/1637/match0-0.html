#include <bits/stdc++.h>

using namespace std;

int f[101][101][26][101];
int stk[20001];
char ss[20001];
class Solution {
public:
    int gao(int x) {
        if (x == 0) return 0;
        if (x == 1) return 1;
        if (x < 10) return 2;
        if (x < 100) return 3;
        return 4;
    }
    void upd(int& a, int b) {
        a = min(a, b);
    }
    int getLengthOfOptimalCompression(string s, int k) {
        int n = (int )s.size(), top = 0;
        for (int i = 0; i <= n; i ++)
            for (int j = 0; j <= i; j ++)
                for (int ch = 0; ch < 26; ++ ch)
                    for (int l = 0; l <= k; ++ l) f[i][j][ch][l] = 0x3f3f3f3f;
        
        f[0][0][s[0] - 'a' + 1][0] = 0;
        for (int i = 0; i < n; i ++) ss[i + 1] = s[i];
        for (int i = 1; i <= n; i ++)
            for (int j = 0; j < i; j ++)
                for (int ch = 0; ch < 26; ch ++) 
                    for (int l = 0; l <= k; ++ l) {
                        if (f[i - 1][j][ch][l] == 0x3f3f3f3f) continue;
                        if (ss[i] - 'a' == ch) {
                            upd(f[i][j + 1][ss[i] - 'a'][l], f[i - 1][j][ss[i] - 'a'][l] + gao(j + 1) - gao(j));
                            if (l < k) upd(f[i][j][ss[i] - 'a'][l + 1], f[i - 1][j][ss[i] - 'a'][l]);
                        }
                        else {
                            upd(f[i][1][ss[i] - 'a'][l], f[i - 1][j][ch][l] + 1);
                            if (l < k) upd(f[i][j][ch][l + 1], f[i - 1][j][ch][l]);
                        }
                    }
        int ans = 0x3f3f3f3f;
        for (int i = 0; i <= n; i ++)
            for (int ch = 0; ch < 26; ch ++)
                for (int l = 0; l <= k; l ++)
                    ans = min(ans, f[n][i][ch][l]);

        return ans;
    }
};
