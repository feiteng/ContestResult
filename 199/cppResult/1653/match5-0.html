/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int countPairs(TreeNode* root, int distance) {
        rs = 0;
        dis = distance;
        mysol(root);
        return rs;
    }
    vector<int> mysol(TreeNode* root)
    {
        vector<int> rsv;
        if(root==NULL)
            return rsv;
        if(root->left==NULL&&root->right==NULL)
        {
            rsv.push_back(1);
            return rsv;
        }
        vector<int> left = mysol(root->left);
        vector<int> right = mysol(root->right);
        int vll = left.size();
        int vlr = right.size();
        for(int i=0; i<vll; i++)
            for(int j=0; j<vlr; j++)
                if(left[i]+right[j]<=dis)
                    rs++;
        for(int i=0; i<vll; i++)
            if(left[i]+1<dis)
                rsv.push_back(left[i]+1);
        for(int i=0; i<vlr; i++)
            if(right[i]+1<dis)
                rsv.push_back(right[i]+1);
        return rsv;
    }
    int rs;
    int dis;
};
