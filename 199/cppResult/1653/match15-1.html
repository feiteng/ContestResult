/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
const int N=(1<<10)|7;
class Solution {
public:
    int ans,cloc;
    int dep[N];
    int a[N][N];
    int dfs(TreeNode* rt,int dis,int d)
    {
        int u=cloc++;
        dep[u]=d;
        a[u][0]=1;
        if(rt->left)
        {
            a[u][0]=0;
            int v=dfs(rt->left,dis,d+1);
            for(int i=1;i<N;++i)
                a[u][i]+=a[v][i-1];
        }
        if(rt->right)
        {
            a[u][0]=0;
            int v=dfs(rt->right,dis,d+1);

            for(int i=0;i<dis-1;++i)
                for(int j=0;j<=dis-1-i;++j)
                    ans+=a[v][i]*a[u][j];
            for(int i=1;i<N;++i)
                a[u][i]+=a[v][i-1];
        }
        return u;
    }
    int countPairs(TreeNode* root, int distance) {
        ans=cloc=0;
        dfs(root,distance,0);
        return ans;
    }
};
