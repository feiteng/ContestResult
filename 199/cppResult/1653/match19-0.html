/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int D, res ;
    
    vector<int> dfs(TreeNode* p){
        vector<int> t, r;
        int i, j;
        if (p->left == NULL && p->right == NULL){
            t.push_back(0);
            return t;
        }
        if (p->left == NULL) {
            t = dfs(p->right);
            for (i = 0;i < t.size(); ++i) t[i]++;
            return t;
        }
        
        if (p->right == NULL) {
            t = dfs(p->left);
            for (i = 0;i < t.size(); ++i) t[i]++;
            return t;
        }
        
        t = dfs(p->left);
        r = dfs(p->right);
        for (i = 0;i < t.size(); ++i) 
            for (j = 0;j < r.size(); ++j)
                if (t[i] + r[j] + 2<= D) ++res;
        for (i = 0;i < t.size(); ++i) t[i]++;
        for (i = 0;i < r.size(); ++i) t.push_back(r[i] + 1);
        return t;
    }
    
    int countPairs(TreeNode* root, int distance) {
        res = 0;
        D = distance;
        dfs(root);
        return res;
    }
};
