/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
int ans,n;
int dp[2005][15];
class Solution {
public:
    int dfs(TreeNode* node, int d) {
        n++;
        int cur=n;
        for (int i=0;i<=d;++i) dp[cur][i]=0;
        if (node->left==NULL&&node->right==NULL)
        {
            dp[cur][0]=1;
            return cur;
        }
        int l=-1,r=-1;
        if (node->left!=NULL){
            l=dfs(node->left,d);
        }
        if (node->right!=NULL){
            r=dfs(node->right,d);
        }

        if (l!=-1&&r!=-1){
        for (int i=0;i<=d;++i)
            for (int j=0;j<=d;++j)
                if (i+1+j+1<=d) ans+=dp[l][i]*dp[r][j];}
        if (l!=-1)for (int i=0;i<d;++i) dp[cur][i+1]+=dp[l][i];
        if (r!=-1)for (int i=0;i<d;++i) dp[cur][i+1]+=dp[r][i];
        return cur;
    }
    int countPairs(TreeNode* root, int distance) {
    	ans=0;n=0;
    	dfs(root,distance);
        return ans;
    }
};
