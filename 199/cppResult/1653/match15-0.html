/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int tot=0,d;
    
    vector<int> dfs(TreeNode* u)
    {
        vector<int> ans(d+1);
        if(u->left==NULL && u->right==NULL)
        {
            ans[1]=1;
            return ans;
        }
        vector<int> la,ra;
        if(u->left!=NULL)la=dfs(u->left);
        if(u->right!=NULL)ra=dfs(u->right);
        if(!la.empty() && !ra.empty())
        {
            for(int i=1;i<=d;++i)
                for(int j=1;j<=d-i;++j)
                    tot+=la[i]*ra[j];
        }
        for(int i=0;i<la.size();++i)ans[i]+=la[i];
        for(int i=0;i<ra.size();++i)ans[i]+=ra[i];
        for(int i=ans.size()-1;i>=1;--i)ans[i]=ans[i-1];
        return ans;
    }
    
    int countPairs(TreeNode* root, int distance) {
        d=distance;
        dfs(root);
        return tot;
    }
};
