/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    int ans;
public:
    int countPairs(TreeNode* root, int d) {
        unordered_map<TreeNode*, vector<int>> m;
        ans = 0;
        dfs(root, d, m);
        return ans;
    }
    
    void dfs(TreeNode *cur, int d, unordered_map<TreeNode*, vector<int>> &m) {
        if (!cur)
            return;
        
        TreeNode *l = cur->left, *r = cur->right;
        
        if (!l && !r) {
            vector<int> v(1, 1);
            m[cur] = v;
            return;
        }
        
        dfs(l, d, m);
        dfs(r, d, m);
        
        vector<int> v(1);
        for (int i = 0; i < m[l].size(); ++i) {
            if ((int)v.size() <= i+1)
                v.push_back(0);
            v[i+1] += m[l][i];
        }
        
        for (int i = 0; i < m[r].size(); ++i) {
            if ((int)v.size() <= i+1)
                v.push_back(0);
            v[i+1] += m[r][i];
        }
        
        if ((int)v.size() > d+1)
            v.pop_back();
        
        m[cur] = v;
        // printf("cur:%d\n", cur->val);
        // for (int i = 0; i < v.size(); ++i) {
        //     printf("%d:%d\n", i, v[i]);
        // }
        
        for (int i = 1; i <= d && i <= m[l].size(); ++i) {
            for (int j = 1; j <= d-i && j <= m[r].size(); ++j) {
                // if (i == 0 && j == 0)
                //     continue;
                
                // if () {
                    int a = m[l][i-1], b = m[r][j-1];
                
                // printf("%d %d %d %d\n", i, j, a , b);
                    ans += a * b;
                // }
            }
            
        }
        
        // if (cur->left && !cur->right)
            
    }
};
