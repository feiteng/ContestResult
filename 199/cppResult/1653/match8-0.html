class Solution {
public:
  int ans;

  deque<int> dfs(TreeNode* v,int d){
    deque<int> res(d,0);
    res[0]=1;
      
    if(v->left)  res[0]=0;
    if(v->right) res[0]=0;

    if(v->left){
      auto tmp=dfs(v->left,d);
      tmp.push_front(0);
      tmp.pop_back();
      for(int i=0;i<d;i++) res[i]+=tmp[i];
    }

    if(v->right){
      auto tmp=dfs(v->right,d);
      tmp.push_front(0);
      tmp.pop_back();

      for(int i=0;i<d;i++)
        for(int j=0;j<d;j++)
          if(i+j<=d) ans+=res[i]*tmp[j];

      for(int i=0;i<d;i++) res[i]+=tmp[i];
    }
    return res;
  }

  int countPairs(TreeNode* root, int distance) {
    ans=0;
    dfs(root,distance);
    return ans;
  }
};
