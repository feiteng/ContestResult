/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int ans = 0;
    std::vector<int> dfs(TreeNode* u, int dis) {
        vector<int> res, l, r;
        if (u->left == nullptr && u->right == nullptr) {
            res.push_back(1);
            return res;
        }
        if (u->left) {
            l = dfs(u->left, dis);
        }
        if (u->right) {
            r = dfs(u->right, dis);
        }
        res.assign(max(l.size(), r.size()) + 1, 0);

        for (int i = 0; i < l.size() && i < dis; i++) {
            for (int j = 0; j < r.size() && j < dis; j++) {
                if (i + j + 2 <= dis) {
                    ans += l[i] * r[j];
                }
            }
        }
        for (int i = 0; i < l.size(); i++) {
            res[i + 1] += l[i];
        }
        for (int i = 0; i < r.size(); i++) {
            res[i + 1] += r[i];
        }
        return res;
    }
    int countPairs(TreeNode* root, int distance) {
        ans = 0;
        dfs(root, distance);
        return ans;
    }
};
