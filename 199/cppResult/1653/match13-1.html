/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
typedef TreeNode T;
class Solution {
public:
    int countPairs(TreeNode* root, int d) {
        vector<T * > q;
        vector<string> s;
        q.push_back(root);
        s.push_back("");
        for(int op = 0; op < (int)q.size(); op++) {
            T * v = q[op];
            string t = s[op];
            if(v->left != nullptr) {
                q.push_back(v->left);
                s.push_back(t + "0");
            }
            if(v->right != nullptr) {
                q.push_back(v->right);
                s.push_back(t + "1");
            }
        }

        int ans = 0;
        for(int i = 0; i < (int)q.size(); i++) {
            for(int j = i + 1; j < (int)q.size(); j++) {
                if(q[i]->left != nullptr || q[i]->right != nullptr || q[j]->left != nullptr || q[j]->right != nullptr) continue;
                int dis = s[i].size() + s[j].size();
                for(int k = 0; k < (int)s[i].size() && k < (int)s[j].size(); k++) {
                    if(s[i][k] == s[j][k]) {
                        dis -= 2;
                    }else break;
                }
                if(dis <= d) ans++;
            }
        }
        return ans;
        
    }
};
