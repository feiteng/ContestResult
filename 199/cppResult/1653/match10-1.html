class Solution {
public:
    void p(TreeNode* v, map<int, int>& mp, int dep) {
      if (v == NULL) return;
      if (v->left == NULL && v->right == NULL) {
        mp[dep] += 1;
        return;
      }
      p(v->left, mp, dep + 1);
      p(v->right, mp, dep + 1);
    }
    int dfs(TreeNode* node, int d) {
      if (node == NULL) return 0;
      map<int, int> l, r;
      p(node->left, l, 1);
      p(node->right, r, 1);     

      int ans = 0;
      for (int i = 0; i <= d; i++) {
          for (int j = 0; i + j <= d; j++) {
                ans += l[i] * r[j];
          }
      }
      return ans + dfs(node->left, d) + dfs(node->right, d);
    }
    int countPairs(TreeNode* root, int distance) {
      return dfs(root, distance);        
    }
};
