/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int solve(TreeNode* u, int d, vi &dep){
        dep.clear();
        if(u==nullptr){
            return 0;
        }
        if(u->left == nullptr && u->right == nullptr){ // leaf
            dep.pb(1);
            return 0;
        }
        vi L, R;
        int ans = 0;
        ans += solve(u->left, d, L);
        ans += solve(u->right, d, R);
        rep(i,sz(L)) rep(j,sz(R)) if(L[i]+R[j]<=d) ans++;
        for(int x: L) dep.pb(x + 1);
        for(int x: R) dep.pb(x + 1);
        //dbg(u->val, dep);
        return ans;
    }
    int countPairs(TreeNode* root, int d) {
        if(root==nullptr) return 0;
        vi dep;
        int ans = solve(root, d, dep);
        dbg(dep);
        return ans;
    }
};
