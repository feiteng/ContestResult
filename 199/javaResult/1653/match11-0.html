/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    
    int ans = 0;
    
    public int countPairs(TreeNode root, int distance) {
        postorder(root, distance, 0);
        return ans;
    }
    
    Map<Integer, Integer> postorder(TreeNode root, int dist, int depth) {
        if (root == null) return null;
        
        Map<Integer, Integer> m = new HashMap<>();
        Map<Integer, Integer> l = postorder(root.left, dist, depth + 1);
        Map<Integer, Integer> r = postorder(root.right, dist, depth + 1);
        
        if (l == null && r == null) {
            m.put(depth, 1);
        } else if (l != null && r == null) {
            return l;
        } else if (l == null && r != null) {
            return r;
        } else if (l != null && r != null) {
            for (Map.Entry<Integer, Integer> e1 : l.entrySet()) {
                for (Map.Entry<Integer, Integer> e2 : r.entrySet()) {
                    if (e1.getKey() - depth + e2.getKey() - depth <= dist) {
                        ans += e1.getValue() * e2.getValue();
                    }
                }
                
                m.put(e1.getKey(), e1.getValue());
            }
            
            for (Map.Entry<Integer, Integer> e2 : r.entrySet()) {
                m.compute(e2.getKey(), (k, v) -> (v == null ? 0 : v) + e2.getValue());
            }
        }
        
        
        return m;
    }
}
