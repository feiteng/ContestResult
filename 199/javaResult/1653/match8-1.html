import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {

    int fu;
    int d;

    public int countPairs(TreeNode root, int distance) {
        fu = 0;
        d = distance;
        dfs(root);
        return fu;
    }

    int[] dfs(TreeNode root) {
        int[] pan = new int[15];
        if (root == null) {
            return pan;
        } else if (root.left == null && root.right == null) {
            pan[0] = 1;
            return pan;
        
}


        int[] left = dfs(root.left);
        int[] right = dfs(root.right);
//        System.out.println(root.val + " " + Arrays.toString(left) + " " + Arrays.toString(right));
        for (int i = 0; i <= d; i++) {
            for (int j = 0; i + j + 2 <= d; j++)
                fu += left[i] * right[j];
        }
        for (int i = 0; i <= d; i++) {
            pan[i + 1] += left[i] + right[i];
        }
//        System.out.println(Arrays.toString(pan));
        return pan;
    }
}
