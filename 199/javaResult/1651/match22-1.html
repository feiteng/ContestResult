class Solution {
    private char[] array;

    public String restoreString(String s, int[] indices) {
        int len = s.length();
        array = new char[len];
        for (int i = 0;i < s.length();i ++) 
{
            array[indices[i]] = s.charAt(i);
        }
        return new String(array);
    }
}
