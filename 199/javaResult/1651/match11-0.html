class Solution {
    public String restoreString(String s, int[] indices) {
        int N = s.length();
        char[] chars = new char[N];
        
        for (int i = 0; i < N; i++) {
            chars[indices[i]] = s.charAt(i);
        }
        
        return new String(chars);
    }
}
