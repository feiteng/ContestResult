class Solution {
    public int getLengthOfOptimalCompression(String s, int k) {
        if (s.length() <= k) {
            return 0;
        }
        char[] cs = s.toCharArray();
        int[][][][] dp = new int[cs.length + 1][k + 1][26][101];
        boolean[][][][] visit = new boolean[cs.length + 1][k + 1][26][101];
        return helper(cs, dp, visit, 0, k, 'a', 0);
    }

    private int helper(char[] cs, int[][][][] dp, boolean[][][][] visit, int p, int k, char alpha, int size) {
        if (p == cs.length) {
            return 0;
        }
        if (visit[p][k][alpha - 'a'][size]) {
            return dp[p][k][alpha - 'a'][size];
        }
        int ans = 100;

        if (k > 0) {
            ans = Math.min(ans, helper(cs, dp, visit, p + 1, k - 1, alpha, size));
        }
        if (alpha == cs[p]) {
            if (size <2 || size == 9 || size == 99) {
                ans = Math.min(ans, 1 + helper(cs, dp, visit, p + 1, k, alpha, size+1));
            } else {
                ans = Math.min(ans, helper(cs, dp, visit, p + 1, k, alpha, size + 1));
            }
        } else {
            ans = Math.min(ans, 1 + helper(cs, dp, visit, p + 1, k, cs[p], 1));
        }
        dp[p][k][alpha - 'a'][size] = ans;
        visit[p][k][alpha - 'a'][size] = true;
        return ans;
    }
}
