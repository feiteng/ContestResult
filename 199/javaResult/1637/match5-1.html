import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {

    static int[] pan = new int[200];

    static {
        pan[0] = 0;
        pan[1] = 1;
        for (int i = 2; i < pan.length; i++) {
            pan[i] = String.valueOf(i).length() + 1;
        }
    }

    String s;

    Map<Long, Integer> fu;

    public int getLengthOfOptimalCompression(String s, int k) {
        fu = new HashMap<>();
        this.s = s;
        int pan = f(s.length() - 1, k, 'a', 0);
        return pan;
    }

    int f(
int i, int delete, char tail, int cnt) {
        if (i < 0) {
            return pan[cnt];
        }
        long key = i;
        key = key * 101 + delete;
        key = key * 101 + tail;
        key = key * 101 + cnt;
        if (fu.containsKey(key)) {
            return fu.get(key);
        }
//        String key = i + "," + delete + "," + tail + "," + cnt;
        int ans = 1 << 30;
        if (delete > 0) {
            ans = f(i - 1, delete - 1, tail, cnt);
        }
        if (s.charAt(i) == tail) 
{
            ans = Math.min(ans, f(i - 1, delete, tail, cnt + 1));
        } else {
            ans = Math.min(ans, f(i - 1, delete, s.charAt(i), 1) + pan[cnt]);
        }

        fu.put(key, ans);
        return ans;
    }
}
