class Solution {


    private int[][][] dp;

    public int getLengthOfOptimalCompression(String s, int k) {
        int i , j , x , len = s.length();
        dp = new int[len][len][k + 1];
        for (i = 0;i < len;i ++) {
            for (j = 0;j < len;j ++) {
                for (x = 0;x <= k;x ++) {
                    dp[i][j][x] = - 1;
                }
            }
        }
        return solve(s , 0 , s.length() - 1 , k);
    }

    private int getResult(int cnt) {
        if (cnt == 1) {
            return 1;
        }
        int ans = 1;
        while (cnt > 0) {
            cnt /= 10;
            ans ++;
        }
        return ans;
    }

    private int solve(String s , int l , int r , int k) {
        int total = r - l + 1;
        if (total <= k) {
            return 0;
        } else {
            if (dp[l][r][k] >= 0) {
                return dp[l][r][k];
            }
            if (k == 0) {
                int ans = 0;
                for (int i = l;i <= r;i ++) {
                    char ch = s.charAt(i);
                    int cnt = 0;
                    while (i <= r && s.charAt(i) == ch) {
                        i ++;
                        cnt ++;
                    }
                    i --;
                    ans += getResult(cnt);
                }
                return dp[l][r][k] = ans;
            } else 
{
                int ans = solve(s , l , r - 1 , k - 1) , remove = k , cnt = 0;
                for (int i = r;i >= l && remove >= 0;i --) {
                    if (s.charAt(i) != s.charAt(r)) {
                        remove --;
                    } else {
                        cnt ++;
                    }
                    if (remove >= 0) {
                        int result = solve(s , l , i - 1 , remove) + getResult(cnt);
                        ans = Math.min(ans , result);
                    }
                }
                return dp[l][r][k] = ans;
            }
        }
    }
}
