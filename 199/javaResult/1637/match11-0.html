class Solution {
    
    String s;
    int N;
    Integer[][][][] M;
    
    int length(char c, int l) {
        if (l <= 0) return 0;
        else if (l == 1) return 1;
        else return ("" + c + l).length();
    }
    
    public int getLengthOfOptimalCompression(String s, int k) {
        this.s = s;
        this.N = s.length();
        this.M = new Integer[26][N + 1][N][k + 1];
        return dp('a', 0, 0, k);
    }
    
    int dp(char pc, int pl, int i, int k) {
        if (i >= N) return length(pc, pl);
        if (M[pc - 'a'][pl][i][k] != null) return M[pc - 'a'][pl][i][k];
        
        int ans = Integer.MAX_VALUE;;
        if (k > 0) ans = Math.min(ans, dp(pc, pl, i + 1, k - 1));
        
        char c = s.charAt(i);
        if (pc == c) {
            ans = Math.min(ans, dp(c, pl + 1, i + 1, k));
        } else {
            ans = Math.min(ans, length(pc, pl) + dp(c, 1, i + 1, k));
        }
        
        M[pc - 'a'][pl][i][k] = ans;
        return ans;
    }
}
