class Solution {
    public int minFlips(String target) {
        int ans = 0;
        int current = 0;
        char[] cs = target.toCharArray();
        for (char c : cs) {
            int t = c - '0';
            if (t != current) {
                ans++;
                current ^= 1;
            }
        }
        return ans;
    }
}
