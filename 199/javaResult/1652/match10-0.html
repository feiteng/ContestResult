	class Solution {
	    public int minFlips(String target) {
	        char[] s = target.toCharArray();
	        int ret = 0;
	        for(int i = 0;i < s.length;i++){
	        	if(i > 0 && s[i-1] != s[i]){
	        		ret++;
	        	}
	        	if(i == 0 && s[i] == '1'){
	        		ret++;
	        	}
	        }
	        return ret;
	    }
	}
