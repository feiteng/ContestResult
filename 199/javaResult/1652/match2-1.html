import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {

    public int minFlips(String target) {
        int fu = 0;
        for (char ch : target.toCharArray()) 
{
            int deng = ch - '0';
            if (deng != fu % 2) {
                fu++;
            }
        }
        return fu;
    }
}
