class Solution {
    public int minFlips(String target) {
        int time = 0, n = target.length(), res = 0;
        for(int i = 0 ; i < n ; i++){
            if(target.charAt(i) - '0' != time)
{
                time ^= 1;
                res++;
            }
        }
        return res;
    }
}
