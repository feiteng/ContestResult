class Solution {
    public int minFlips(String s) {
        int count = 0;
        if(s.charAt(0)=='1') count+=1;
        
        for(int i=1; i<s.length(); i++) {
            if(s.charAt(i)!=s.charAt(i-1)) {
                count+=1;
            }
        }
        return count;
    }
}
