class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int sum=0;
        for(auto num: nums){
            int tmpsum=0, tmpcnt=0;
            if(sqrt(num)==int(sqrt(num))){
                tmpcnt+=1;
                tmpsum+=sqrt(num);
            }
            for(int i=1;tmpcnt<=4&&i<sqrt(num);i++){

                if(num%i==0){
                    tmpcnt+=2;
                    tmpsum+=i+num/i;
                }
            }
            if(tmpcnt==4) sum+=tmpsum;
        }
        return sum;
    }
};
