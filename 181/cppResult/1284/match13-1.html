class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        
        int ans=0;
        for(int i=0;i<nums.size();i++)

        {
            int res=0;
            int tag=0;
            for(int j=1;j*j<=nums[i];j++)
            {
                if(nums[i]%j==0)
                {
                    if(j!=nums[i]/j)
                    {
                        res+=j;
                        res+=nums[i]/j;
                        tag+=2;

                    }
                    else
                    {
                        tag++;
                        res+=j;
                    }
                }
            }
            if(tag==4)
                ans+=res;
        }
        
        return ans;
        
    }
};
