class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for(int num : nums) {
            int cnt = 0;
            int sum = 0;
            for(int f = 1; f <= (int) sqrt(num); ++f) {
                if (num % f == 0) {
                    if (f * f == num) {
                        sum += f;
                        ++cnt;
                    } else {
                        sum += f + num / f;
                        cnt += 2;
                    }
                }
            }
            if (cnt == 4) ans += sum;
        }
        return ans;
    }
};
