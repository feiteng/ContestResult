class Solution {
public:
    int count(int k) {
        int ret = 0;
        int sum=0;
        for(int i=1; i*i<=k; ++i) {
            if(k%i==0) {
                if(i*i!=k) ret += 2, sum+=i+k/i;
                else ret += 1, sum+=i;
            }
        }
        if(ret==4) return sum;
        return 0;
    }
    int sumFourDivisors(vector<int>& nums) {
        int sum = 0;
        for(int k : nums) {
            //cout << count(k) << endl;
            sum += count(k);
        }
        return sum;
    }
};
