class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int n = nums.size();
        int ans = 0;
        for (int i = 0 ; i < n ; i++){
            int g = sqrt(nums[i]);
            int cnt = 0;
            int res = 0;
            for (int j = 1 ; j <= g ; j++){
                if(nums[i] % j == 0){
                    cnt++;
                    res += j;
                    if(nums[i] / j != j) {
                        cnt++;
                        res += nums[i] / j;
                    }
                }
            }
            if(cnt == 4) ans += res;
        }
        return ans;
    }
};
