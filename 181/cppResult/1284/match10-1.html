class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
      int ans = 0;
      for(auto w : nums) {
        int k = 0, s = 0;
        for(int i = 1; i * i <= w; i++) {
          if(w % i == 0) {
            k += 2;
            s += i + w / i;
          }

          if(i * i == w) {
            k -= 1;
            s -= i;
          }
        }
        if(k == 4) ans += s;
      }
      return ans;
    }
};
