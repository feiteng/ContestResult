class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for (int v : nums)
        {
            int cur = 0;
            int cnt = 0;
            for (int i = 1; i * i <= v && cnt < 5; ++i)
            {
                if (v % i == 0)
                {
                    if (i * i == v)
                    {
                        cur += i;
                        ++cnt;
                    }
                    else
                    {
                        cur += i + v / i;
                        cnt += 2;
                    }
                }
            }
            if (cnt == 4)
                ans += cur;
        }
        return ans;
    }
};
