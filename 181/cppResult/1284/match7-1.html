class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for (int num: nums) {
            int cnt = 0, t = 0;
            for (int i = 1; i * i <= num; ++i) {
                if (num % i == 0) {
                    ++cnt;
                    t += i;

                    if (i * i != num) {
                        ++cnt;
                        t += num / i;
                    }
                }
            }
            if (cnt == 4) {
                ans += t;
            }
        }
        return ans;
    }
};
