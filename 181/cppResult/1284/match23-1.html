class Solution {
public:
    int cal(int x){
        int ans=0;
        int now=0;
        for(int i=2;i*i<=x;i++){
            if(x%i==0){
                if(i*i==x){
                     ans++;
                    now+=i;
                }
                else{
                     ans+=2;
                    now+=i;
                    now+=x/i;
                }
            }
        }
        now+=1;
        now+=x;
        ans+=2;

        if(ans==4) return now;
        return 0;
    }
    int sumFourDivisors(vector<int>& nums) {
        int s=0;
        for(int v:nums){
            s+=cal(v);
        }
        return s;
    }
};
