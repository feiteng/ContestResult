class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for(int a : nums) {
            int sum = 0, numDivisors = 0;
            for(int d = 1; d * d <= a; d++) {
                if(a % d == 0) {
                    if((a / d) == d) {
                        numDivisors++;
                        sum += d;
                    } else {
                        numDivisors += 2;
                        sum += d + a / d;
                    }
                }
            }
            if(numDivisors == 4) {
                ans += sum;
            }
        }
        return ans;
    }
};
