class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for(int n : nums) {
            int count = 0;
            int total = 0;
            for(int i = 1; i * i <= n; i++) {
                if (n % i == 0) {
                    count++;
                    total += i;
                    if (i * i != n) {
                        count++;
                        total += n / i;
                    }
                }
            }
            if (count == 4) {
                ans += total;
            }
        }
        return ans;
    }
};
