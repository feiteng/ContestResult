class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int res=0;
        for(int i=0;i<nums.size();i++){
            int js=0;
            int tot=0;
            for(int j=1;j<=sqrt(nums[i]);j++){
                if(nums[i]%j==0){
                    if(nums[i]==j*j){ js+=1;tot+=j;}
                    else {tot+=j+nums[i]/j;js+=2;}
                }
                if(js>4) break;
            }
            if(js==4) res+=tot;
        }
        return res;
    }
};
