class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int n = 100000;
        vector <int> f(100005, 0), s(100005, 0);
        for (int i = 1; i <= n; i++) {
            for (int j = i; j <= n; j += i) {
                f[j]++;
                s[j] += i;
            }
        }
        int ans = 0;
        for (int i : nums) {
            if (f[i] == 4) {
                ans += s[i];
            }
        }
        return ans;
    }
};
