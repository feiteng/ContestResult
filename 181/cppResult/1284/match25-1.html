class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        
        for (auto x : nums) {
            int cnt = 0;
            int sum = 0;

            for (int i = 1; i * i <= x; ++i) {
                if (x % i == 0) {
                    ++cnt;
                    sum += i;
                    if (i * i != x) {
                        ++cnt;
                        sum += x / i;
                    }
                }
                if (cnt > 4) {
                    break;
                }
            }
            if (cnt == 4) {
                res += sum;
            }
        }
        return res;
    }
};
