class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int sum = 0;
        for (int num : nums) {
            int t = 0;
            int s = 0;
            for (int i = 1; i * i <= num; ++i) {
                if (num % i == 0) {
                    if (i * i == num) {
                        t += 1;
                        s += i;
                    } else {
                        t += 2;
                        s += i;
                        s += num / i;
                    }
                }
                if (t > 4) {
                    break;
                }
            }
            if (t == 4) {
                sum += s;
            }
        }
        return sum;
    }
};
