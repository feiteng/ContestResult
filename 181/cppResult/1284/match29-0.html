class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans=0;
        for(int i=0;i<nums.size();i++){
            int x=nums[i];
            int y=sqrt(x);
            if(y*y==x) continue;
            int sum=0,cnt=0;
            for(int j=1;j<=y&&cnt<=2;j++){
                if(x%j==0){
                    cnt++;
                    sum+=j+x/j;
                }
            }
            if(cnt==2) ans+=sum;
        }
        return ans;
    }
};
