class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans=0;
        for(auto n:nums){
            int count = 0, tmp = 0;
            for(int i=1;i*i<=n;i++){
                if(i*i==n)
                    count +=9;
                if(n%i==0){
                    count +=1;
                    tmp+=i+n/i;
                }
            }
            if(count==2)
                ans+=tmp;
        }
        return ans;
    }
};
