class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for(auto u: nums) {
            set<int> st;
            for(int j = 1; j*j <= u; j++) {
                if(u%j == 0) {
                    st.insert(j);
                    st.insert(u/j);
                }
            }
            if(st.size() == 4) {
                for(auto v: st) {
                    ans += v;
                }
            }
        }
        return ans;
    }
};
