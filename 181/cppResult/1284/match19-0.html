class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans=0;
        for(auto x:nums) {
            int sum=0;
            int k=0;
            for(int i=1;i*i<=x;i++) {
                if (x%i==0) { 
                    sum+=i;
                    k++;
                    if (x/i!=i) {
                        sum+=x/i;k++;
                    }
                }
            }
            if (k==4){
                ans+=sum;
            }
        }
        return ans;
    }
};
