bool is_prime(int x) {
    if (x < 2) return false;
    if (x == 2) return true;
    if (x % 2 == 0) return false;
    for (int d = 3; d * d <= x; d += 2) {
        if (x % d == 0) return false;
    }
    return true;
}

int semi(int x) {
    if (x <= 2) return 0;
    for (int d = 2; d <= x; ++d) {
        if (x % d != 0) continue;
        if (d <= 50 and d * d * d == x) {
            return 1 + d + d * d + d * d * d;
        }
        int y = (x / d);
        if (y <= d) break;
        if (is_prime(d) and is_prime(y)) {
            cout << d << ' ' << y << ' ' << x << endl;
            return 1 + d + y + x;
        }
    }
    return 0;
}

class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {

        vector<int> sieve(1e5 + 5, 0), score(1e5 + 5, 0);
        for (int i = 1; i < sieve.size(); ++i) {
            for (int j = i; j < sieve.size(); j += i) {
                sieve[j]++;
                score[j] += i;
            }
        }
        
        int ans = 0;
        for (int x : nums) {
            if (sieve[x] == 4) {
                ans += score[x];
            }
        }
        
        return ans;
    }
};
