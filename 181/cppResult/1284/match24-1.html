class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        for(int i = 0; i < nums.size(); ++i)
        {
            res += getnums(nums[i]);
        }
        return res;
    }
    int getnums(int a)

    {
        int num = 0;
        int res = 0;
        for(int i = 1; i*i <= a; ++i)
        {
            if(a % i == 0)
            {
                if(i*i == a)
                {
                    res += i;
                    num += 1;
                }
                    
                else
                {
                    res += (i + a/i);
                    num += 2;
                }
                    
            }
        }
        if(num == 4)
            return res;
        else
            return 0;
    }
};
