class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ret = 0;
        for (auto &i:nums) {
            set<int>div;
            for (int j = 1 ; j <= sqrt(i) ; j++) {
               if (i % j == 0) {
                   div.insert(j);
                   div.insert(i / j);
               }
            }
            if (div.size() == 4) {
                for (auto j:div) {
                    ret += j;
                }
            }
        }
        return ret;
    }
};
