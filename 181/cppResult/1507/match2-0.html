class Solution {
public:
    int m, n;
    bool hasValidPath(vector<vector<int>>& grid) {
        m = grid.size(), n = grid[0].size();
        vector<vector<char>> visited(m, vector<char>(n, 0));
        queue<pair<int, int>> Q;
        Q.emplace(0, 0);
        visited[0][0] = 1;
        while (!Q.empty()) {
            auto [i, j] = Q.front();
            Q.pop();
            if (i == m - 1 && j == n - 1)
                return true;
            if (grid[i][j] == 1) {
                if (inRange(i, j - 1) && !visited[i][j - 1] && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6)) {
                    visited[i][j - 1] = true;
                    Q.emplace(i, j - 1);
                }
                if (inRange(i, j + 1) && !visited[i][j + 1] && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5)) {
                    visited[i][j + 1] = true;
                    Q.emplace(i, j + 1);
                }
            }
            else if (grid[i][j] == 2) {
                if (inRange(i - 1, j) && !visited[i - 1][j] && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4)) {
                    visited[i - 1][j] = true;
                    Q.emplace(i - 1, j);
                }
                if (inRange(i + 1, j) && !visited[i + 1][j] && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6)) {
                    visited[i + 1][j] = true;
                    Q.emplace(i + 1, j);
                }
            }
            else if (grid[i][j] == 3) {
                if (inRange(i, j - 1) && !visited[i][j - 1] && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6)) {
                    visited[i][j - 1] = true;
                    Q.emplace(i, j - 1);
                }
                if (inRange(i + 1, j) && !visited[i + 1][j] && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6)) {
                    visited[i + 1][j] = true;
                    Q.emplace(i + 1, j);
                }
            }
            else if (grid[i][j] == 4) {
                if (inRange(i, j + 1) && !visited[i][j + 1] && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5)) {
                    visited[i][j + 1] = true;
                    Q.emplace(i, j + 1);
                }
                if (inRange(i + 1, j) && !visited[i + 1][j] && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6)) {
                    visited[i + 1][j] = true;
                    Q.emplace(i + 1, j);
                }
            }
            else if (grid[i][j] == 5) {
                if (inRange(i - 1, j) && !visited[i - 1][j] && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4)) {
                    visited[i - 1][j] = true;
                    Q.emplace(i - 1, j);
                }
                if (inRange(i, j - 1) && !visited[i][j - 1] && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6)) {
                    visited[i][j - 1] = true;
                    Q.emplace(i, j - 1);
                }
            }
            else if (grid[i][j] == 6) {
                if (inRange(i - 1, j) && !visited[i - 1][j] && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4)) {
                    visited[i - 1][j] = true;
                    Q.emplace(i - 1, j);
                }
                if (inRange(i, j + 1) && !visited[i][j + 1] && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5)) {
                    visited[i][j + 1] = true;
                    Q.emplace(i, j + 1);
                }
            }
        }
        return false;
    }
    
    bool inRange(int i, int j) {
        if (i < 0 || i >= m || j < 0 || j >= n)
            return false;
        return true;
    }
};
