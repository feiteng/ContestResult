class Solution {
public:
    bool vertical(int u,int d)
{
    return ((u==2||u==3||u==4)&&(d==2||d==5||d==6));
}

bool horizontal(int l,int r)
{
    return ((l==1||l==4||l==6)&&(r==1||r==3||r==5));
}

bool in(int r,int c,int n,int m)
{
    return (0<=r&&r<n&&0<=c&&c<m);

}

bool hasValidPath(vector<vector<int>>& grid)
{
    int n=grid.size();
    int m=grid[0].size();
    queue<pair<int,int>> q;
    q.push({0,0});
    vector<vector<bool>> v(n,vector<bool>(m,0));
    v[0][0]=1;
    while(!q.empty())
    {
        int r,c;
        tie(r,c)=q.front();
        q.pop();

        if(in(r,c+1,n,m)&&v[r][c+1]==0&&horizontal(grid[r][c],grid[r][c+1])) //right
        {
            v[r][c+1]=1;
            q.push({r,c+1});
        }
        if(in(r-1,c,n,m)&&v[r-1][c]==0&&vertical(grid[r-1][c],grid[r][c])) //up
        {
            v[r-1][c]=1;
            q.push({r-1,c});
        }
        if(in(r,c-1,n,m)&&v[r][c-1]==0&&horizontal(grid[r][c-1],grid[r][c])) //left
        {
            v[r][c-1]=1;
            q.push({r,c-1});
        }
        if(in(r+1,c,n,m)&&v[r+1][c]==0&&vertical(grid[r][c],grid[r+1][c])) //down
        {
            v[r+1][c]=1;
            q.push({r+1,c});
        }
    }
    return v[n-1][m-1];
}
};
