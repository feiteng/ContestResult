class Solution {
    #define DEBUG
    #ifdef DEBUG
    #define dg(x) cout << #x << " = " << x << endl
    #define dg2(x1, x2) cout << #x1 << " = " << x1 << ", " << #x2 << " = " << x2 << endl
    #define dg3(x1, x2, x3) cout << #x1 << " = " << x1 << ", " << #x2 << " = " << x2 << ", " << #x3 << " = " << x3 << endl
    #define dgarr(arr, n) arrout(arr, n);
    #else
    #define dg(x) ;
    #define dg2(x1, x2) ;
    #define dg3(x1, x2, x3) ;
    #define dgarr(arr, n) ;
    #endif
    typedef long long ll;
    template <typename T>
    void arrout(T arr[], int len){
	    for (int i = 0; i <= len; ++i)
		    cout << arr[i] << " ";
	    cout << endl;
    }
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> flag(n, vector<int>(m, 0));

        queue<pair<int, int>> q;
        q.push(make_pair(0, 0));
        flag[0][0] = 1;
        while(!q.empty()) {
            pair<int, int> top = q.front();
            q.pop();
            int x = top.first;
            int y = top.second;
            if(x == n - 1 &&  y == m - 1){
                return true;
            }
            // dg2(x, y);
            int val = grid[x][y];
     
            if(val == 1) {
                int xt = x;
                int xy = y + 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 3 || grid[xt][xy] == 5 ||  grid[xt][xy] == 1 )){
                    q.push(make_pair(xt, xy));
                }
                xt = x;
                xy = y - 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 4 || grid[xt][xy] == 6 ||  grid[xt][xy] == 1)){
                    q.push(make_pair(xt, xy));
                } 
            } else if(val == 2) {
                int xt = x + 1;
                int xy = y;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 5 || grid[xt][xy] == 6 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                }
                xt = x - 1;
                xy = y;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 3 || grid[xt][xy] == 4 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                }                
            } else if(val == 3){
                int xt = x + 1;
                int xy = y;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 5 || grid[xt][xy] == 6 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                }
                xt = x;
                xy = y - 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 4 || grid[xt][xy] == 6 ||  grid[xt][xy] == 1)){
                    q.push(make_pair(xt, xy));
                }       
            } else if( val == 4) {
                int xt = x;
                int xy = y + 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 3 || grid[xt][xy] == 5 ||  grid[xt][xy] == 1 ) ){
                    q.push(make_pair(xt, xy));
                }
                xt = x + 1;
                xy = y;
                if(isValid(xt, xy, n, m, flag) &&  (grid[xt][xy] == 5 || grid[xt][xy] == 6 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                } 
            } else if( val == 5) {
                int xt = x - 1;
                int xy = y;
                if(isValid(xt, xy, n, m, flag) &&  (grid[xt][xy] == 3 || grid[xt][xy] == 4 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                }
                xt = x;
                xy = y - 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 4 || grid[xt][xy] == 6 ||  grid[xt][xy] == 1)){
                    q.push(make_pair(xt, xy));
                } 
            } else {
                int xt = x - 1;
                int xy = y;
                if(isValid(xt, xy, n, m, flag) &&  (grid[xt][xy] == 3 || grid[xt][xy] == 4 || grid[xt][xy] == 2)){
                    q.push(make_pair(xt, xy));
                }
                xt = x;
                xy = y + 1;
                if(isValid(xt, xy, n, m, flag) && (grid[xt][xy] == 3 || grid[xt][xy] == 5 ||  grid[xt][xy] == 1 )){
                    q.push(make_pair(xt, xy));
                }            
            }
        }
        return false;
    }
    
    
    bool isValid(int x, int y, int n, int m, vector<vector<int>>& flag){
        if(0 <= x && x < n && 0 <= y && y < m && !flag[x][y]){
            flag[x][y] = 1;
            return true;
        }
        return false;
    }
};
