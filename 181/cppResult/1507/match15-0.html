class Solution {
public:
    bool hasValidPath(vector<vector<int>> &grid) {
        memset(vis, 0, sizeof(vis));
        dfs(grid, 0, 0);
        return flag;
    }

    bool flag = false;
    bool vis[500][500];

    void dfs(vector<vector<int>> &grid, int x, int y) {
        if (x == grid.size() - 1 && y == grid[0].size() - 1) {
            flag = true;
            return;
        }
        if (flag) return;
        if (vis[x][y]) return;
//        cout << x << " " << y << endl;
        vis[x][y] = 1;
        if (grid[x][y] == 1) {
            if (y + 1 < grid[0].size() && (grid[x][y + 1] == 3 || grid[x][y + 1] == 5 || grid[x][y + 1] == 1)) dfs(grid, x, y + 1);
            if (y - 1 >= 0 && (grid[x][y - 1] == 4 || grid[x][y - 1] == 6 || grid[x][y - 1] == 1)) dfs(grid, x, y - 1);
        }

        if (grid[x][y] == 2) {
            if (x + 1 < grid.size() && (grid[x + 1][y] == 5 || grid[x + 1][y] == 6 || grid[x + 1][y] == 2)) dfs(grid, x + 1, y);
            if (x - 1 >= 0 && (grid[x - 1][y] == 3 || grid[x - 1][y] == 4 || grid[x - 1][y] == 2)) dfs(grid, x - 1, y);
        }

        if (grid[x][y] == 3) {
            if (y - 1 >= 0 && (grid[x][y - 1] == 4 || grid[x][y - 1] == 6 || grid[x][y - 1] == 1)) dfs(grid, x, y - 1);
            if (x + 1 < grid.size() && (grid[x + 1][y] == 5 || grid[x + 1][y] == 6 || grid[x + 1][y] == 2)) dfs(grid, x + 1, y);
        }

        if (grid[x][y] == 4) {
            if (x + 1 < grid.size() && (grid[x + 1][y] == 5 || grid[x + 1][y] == 6 || grid[x + 1][y] == 2)) dfs(grid, x + 1, y);
            if (y + 1 < grid[0].size() && (grid[x][y + 1] == 3 || grid[x][y + 1] == 5 || grid[x][y + 1] == 1)) dfs(grid, x, y + 1);
        }

        if (grid[x][y] == 5) {
            if (y - 1 >= 0 && (grid[x][y - 1] == 4 || grid[x][y - 1] == 6 || grid[x][y - 1] == 1)) dfs(grid, x, y - 1);
            if (x - 1 >= 0 && (grid[x - 1][y] == 3 || grid[x - 1][y] == 4 || grid[x - 1][y] == 2)) dfs(grid, x - 1, y);
        }

        if (grid[x][y] == 6) {
            if (x - 1 >= 0 && (grid[x - 1][y] == 3 || grid[x - 1][y] == 4 || grid[x - 1][y] == 2)) dfs(grid, x - 1, y);
            if (y + 1 < grid[0].size() && (grid[x][y + 1] == 3 || grid[x][y + 1] == 5 || grid[x][y + 1] == 1)) dfs(grid, x, y + 1);
        }
    }
};
