class Solution {
public:
    bool dfs(vector<vector<int>> &g, int r, int c, vector<vector<bool>> &v) {
        // cout << r << " " << c << endl;
        v[r][c] = true;
        const int M = g.size(), N = g[0].size();
        if (r == M - 1 && c == N - 1) return true;

        switch(g[r][c]) {
            case 1:
                if (c < N - 1 && (g[r][c + 1] == 3 || g[r][c + 1] == 5 || g[r][c + 1] == 1) && !v[r][c + 1]) return dfs(g, r, c + 1, v);
                else if (c > 0 && (g[r][c - 1] == 4 || g[r][c - 1] == 6 || g[r][c - 1] == 1) && !v[r][c - 1]) return dfs(g, r, c - 1, v);
                else return false;
            case 2:
                if (r < M - 1 && (g[r + 1][c] == 5 || g[r + 1][c] == 6 || g[r + 1][c] == 2) && !v[r + 1][c]) return dfs(g, r + 1, c, v);
                else if (r > 0 && (g[r - 1][c] == 3 || g[r - 1][c] == 4 || g[r - 1][c] == 2) && !v[r - 1][c]) return dfs(g, r - 1, c, v);
                else return false;
            case 3:
                if (r < M - 1 && (g[r + 1][c] == 5 || g[r + 1][c] == 6 || g[r + 1][c] == 2) && !v[r + 1][c]) return dfs(g, r + 1, c, v);
                else if (c > 0 && (g[r][c - 1] == 4 || g[r][c - 1] == 6 || g[r][c - 1] == 1) && !v[r][c - 1]) return dfs(g, r, c - 1, v);
                else return false;
            case 4:
                if (r < M - 1 && (g[r + 1][c] == 5 || g[r + 1][c] == 6 || g[r + 1][c] == 2) && !v[r + 1][c]) return dfs(g, r + 1, c, v);
                else if (c < N - 1 && (g[r][c + 1] == 3 || g[r][c + 1] == 5 || g[r][c + 1] == 1) && !v[r][c + 1]) return dfs(g, r, c + 1, v);
                else return false;  
            case 5:
                if (r > 0 && (g[r - 1][c] == 3 || g[r - 1][c] == 4 || g[r - 1][c] == 2) && !v[r - 1][c]) return dfs(g, r - 1, c, v);
                else if (c > 0 && (g[r][c - 1] == 4 || g[r][c - 1] == 6 || g[r][c - 1] == 1) && !v[r][c - 1]) return dfs(g, r, c - 1, v);
                else return false;  
            default:
                if (r > 0 && (g[r - 1][c] == 3 || g[r - 1][c] == 4 || g[r - 1][c] == 2) && !v[r - 1][c]) return dfs(g, r - 1, c, v);
                else if (c < N - 1 && (g[r][c + 1] == 3 || g[r][c + 1] == 5 || g[r][c + 1] == 1) && !v[r][c + 1]) return dfs(g, r, c + 1, v);
                else return false;  
        }
        return false;
    }
    
    bool hasValidPath(vector<vector<int>>& grid) {
        const int M = grid.size(), N = grid[0].size();
        vector<vector<bool>> v(M, vector<bool>(N, false));
        return dfs(grid, 0, 0, v);
    }
};
