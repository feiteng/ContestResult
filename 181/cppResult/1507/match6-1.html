class Solution {
public: 
    bool check(int i,int j,int n,int m)
    {
        return (i >= 0 && j >= 0 && i < n && j < m);
    }
    
    bool path(int i,int j,int n,int m,vector<vector<int>> &mat,map<pair<int,int>,int> &visited)
    {
        if(!check(i,j,n,m))
            return false;
        //cout << "hell\n";
        if(i == n - 1 && j == m - 1)
            return true;
        if(visited[{i,j}])
            return false;
        visited[{i,j}] = 1;

        bool a = false,b = false;
        if(mat[i][j] == 1)
        {
            if(check(i,j - 1,n,m) && (mat[i][j - 1] == 1 || mat[i][j - 1] == 4 || mat[i][j - 1] == 6))
                a = path(i,j - 1,n,m,mat,visited);
            if(check(i,j + 1,n,m) && (mat[i][j + 1] == 1 || mat[i][j + 1] == 3 || mat[i][j + 1] == 5))
                b = path(i,j + 1,n,m,mat,visited);
        }
        else if(mat[i][j] == 2)
        {
            if(check(i - 1,j,n,m) && (mat[i - 1][j] == 2 || mat[i - 1][j] == 3 || mat[i - 1][j] == 4))
                a = path(i - 1,j,n,m,mat,visited);
            if(check(i + 1,j,n,m) && (mat[i + 1][j] == 2 || mat[i + 1][j] == 5 || mat[i + 1][j] == 6))
                b = path(i + 1,j,n,m,mat,visited);
        }
        else if(mat[i][j] == 3)
        {
            if(check(i,j - 1,n,m) && (mat[i][j - 1] == 1 || mat[i][j - 1] == 4 || mat[i][j - 1] == 6))
                a = path(i,j - 1,n,m,mat,visited);
            if(check(i + 1,j,n,m) && (mat[i + 1][j] == 2 || mat[i + 1][j] == 5 || mat[i + 1][j] == 6))
                b = path(i + 1,j,n,m,mat,visited);
        }
        else if(mat[i][j] == 4)
        {
            if(check(i,j + 1,n,m) && (mat[i][j + 1] == 1 || mat[i][j + 1] == 3 || mat[i][j + 1] == 5))
                a = path(i,j + 1,n,m,mat,visited);
            if(check(i + 1,j,n,m) && (mat[i + 1][j] == 2 || mat[i + 1][j] == 5 || mat[i + 1][j] == 6))
                b = path(i + 1,j,n,m,mat,visited);
        }
        else if(mat[i][j] == 5)
        {
            if(check(i,j - 1,n,m) && (mat[i][j - 1] == 1 || mat[i][j - 1] == 4 || mat[i][j - 1] == 6))
                a = path(i,j - 1,n,m,mat,visited);
            if(check(i - 1,j,n,m) && (mat[i - 1][j] == 2 || mat[i - 1][j] == 3 || mat[i - 1][j] == 4))
                b = path(i - 1,j,n,m,mat,visited);
        }
        else if(mat[i][j] == 6)
        {
            if(check(i - 1,j,n,m) && (mat[i - 1][j] == 2 || mat[i - 1][j] == 3 || mat[i - 1][j] == 4))
                a = path(i - 1,j,n,m,mat,visited);
            if(check(i,j + 1,n,m) && (mat[i][j + 1] == 1 || mat[i][j + 1] == 3 || mat[i][j + 1] == 5))
                b = path(i,j + 1,n,m,mat,visited);
        }
        return (a | b);
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        map<pair<int,int>,int> visited;
        int n = grid.size();
        int m = grid[0].size();
        return path(0,0,n,m,grid,visited);
    }
};
