class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        int cur = 0, next = 0;
        vector<bool> visited(m * n, false);
        // visited[0] = true;
        if(m == 1 && n == 1) return true;
        while(cur != m*n - 1){
            int curX = cur / n, curY = cur % n;
            visited[cur] = true;

            if(grid[curX][curY] == 1){
                if(curY > 0 && !visited[curX * n + curY - 1] && (grid[curX][curY - 1] == 1 || grid[curX][curY - 1] == 4 || grid[curX][curY - 1] == 6)){
                    next = curX * n + curY - 1;
                }
                if(curY < n - 1 && !visited[curX * n + curY + 1] && (grid[curX][curY + 1] == 1 || grid[curX][curY + 1] == 3 || grid[curX][curY + 1] == 5)){
                    next = curX * n + curY + 1;
                }
            }else if(grid[curX][curY] == 2){
                if(curX > 0 && !visited[(curX-1)*n + curY] && (grid[curX-1][curY] == 3 || grid[curX-1][curY] == 4 || grid[curX-1][curY] == 2) ){
                    next = (curX-1)*n + curY;
                }
                if(curX < m - 1 && !visited[(curX+1)*n + curY] && (grid[curX+1][curY] == 2 || grid[curX+1][curY] == 5 || grid[curX+1][curY] == 6) ){
                    next = (curX+1)*n + curY;
                }
            }else if(grid[curX][curY] == 3){
                if(curY > 0 && !visited[curX * n + curY - 1] && (grid[curX][curY - 1] == 1 || grid[curX][curY - 1] == 4 || grid[curX][curY - 1] == 6)){
                    next = curX * n + curY - 1;
                }
                if(curX < m - 1 && !visited[(curX+1)*n + curY] && (grid[curX+1][curY] == 2 || grid[curX+1][curY] == 5 || grid[curX+1][curY] == 6) ){
                    next = (curX+1)*n + curY;
                }

            }else if(grid[curX][curY] == 4){
                if(curY < n - 1 && !visited[curX * n + curY + 1] && (grid[curX][curY + 1] == 1 || grid[curX][curY + 1] == 3 || grid[curX][curY + 1] == 5)){
                    next = curX * n + curY + 1;
                }
                if(curX < m - 1 && !visited[(curX+1)*n + curY] && (grid[curX+1][curY] == 2 || grid[curX+1][curY] == 5 || grid[curX+1][curY] == 6) ){
                    next = (curX+1)*n + curY;
                }

            }else if(grid[curX][curY] == 5){
                if(curX > 0 && !visited[(curX-1)*n + curY] && (grid[curX-1][curY] == 3 || grid[curX-1][curY] == 4 || grid[curX-1][curY] == 2) ){
                    next = (curX-1)*n + curY;
                }
                if(curY > 0 && !visited[curX * n + curY - 1] && (grid[curX][curY - 1] == 1 || grid[curX][curY - 1] == 4 || grid[curX][curY - 1] == 6)){
                    next = curX * n + curY - 1;
                }

            }else if(grid[curX][curY] == 6){
                if(curX > 0 && !visited[(curX-1)*n + curY] && (grid[curX-1][curY] == 3 || grid[curX-1][curY] == 4 || grid[curX-1][curY] == 2) ){
                    next = (curX-1)*n + curY;
                }
                if(curY < n - 1 && !visited[curX * n + curY + 1] && (grid[curX][curY + 1] == 1 || grid[curX][curY + 1] == 3 || grid[curX][curY + 1] == 5)){
                    next = curX * n + curY + 1;
                }

            }
            if(next == cur){
                //cout << curX << "," << curY << endl;
                return false;
            }
            cur = next;
        }
        return true;

    }
};
