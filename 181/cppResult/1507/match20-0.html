class Solution {
public:
    
    bool dfs(int i, int j, vector<vector<int>>& grid, int previous)
    {
        if(i < 0 || j < 0 || i >= grid.size() || j >= grid[0].size() || grid[i][j] == -1) return false;
        
        if(i == grid.size()-1 && j == grid[grid.size()-1].size()-1) return true;

        
        //process the next path
        int current = grid[i][j];
        grid[i][j] = -1;
         
        bool status = false;
        
        if(current == 1)
        {
            if(j-1 >= 0 && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6))
                status = status || dfs(i, j-1, grid, current);
            if(j+1 < grid[0].size() && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5))
                status = status || dfs(i, j+1, grid, current);
        }
        else if(current == 2)
        {
            if(i-1 >= 0  && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4))
                status = status || dfs(i-1, j, grid, current);
            if(i+1 < grid.size() && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6))
                status = status || dfs(i+1, j, grid, current);
        }
        else if(current == 3)
        {
            if(j-1 >= 0 && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6))
                status = status || dfs(i, j-1, grid, current);
            if(i+1 < grid.size() && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6))
                status = status || dfs(i+1, j, grid, current);
        }
        else if(current == 4)
        {
            if(j+1 < grid[0].size() && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5))
                status = status || dfs(i, j+1, grid, current);
            if(i+1 < grid.size() && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6))
                status = status || dfs(i+1, j, grid, current);
        }
        else if(current == 5)
        {
            if(i-1 >= 0 && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4))
                status = status || dfs(i-1, j, grid, current);
            if(j-1 >= 0 && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6))
                status = status || dfs(i, j-1, grid, current);
        }
        else
        {
            if(i-1 >= 0 && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4))
                status = status || dfs(i-1, j, grid, current);
            if(j+1 < grid[0].size() && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5))
                status = status || dfs(i, j+1, grid, current);
        }
        
        return status;
    }
    
    
    
    bool hasValidPath(vector<vector<int>>& grid) {
        if(grid.size() == 0) return true;
        if(grid[0].size() == 0) return true;
        
        return dfs(0, 0, grid, -1);
    }
};
