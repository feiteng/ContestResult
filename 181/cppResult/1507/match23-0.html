class Solution {
public:

    
    bool hasValidPath(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        //1234 up down left right
        int curx=0,cury=0;
        vector<vector<bool>>visited(n,vector<bool>(m));
        visited[curx][cury]=true;
        int dir;
        if(grid[0][0]==1)dir=3;
        if(grid[0][0]==2)dir=1;
        if(grid[0][0]==3)dir=3;
        if(grid[0][0]==4)dir=4;
        if(grid[0][0]==5)dir=1;
        if(grid[0][0]==6)dir=1;
        
        bool move=true;
        while(move){
            visited[curx][cury]=true;
            if(dir==1 && curx!=0){
                if(grid[curx-1][cury]==2){
                    curx--;
                    dir=1;
                }
                else if(grid[curx-1][cury]==3){
                    curx--;
                    dir=3;
                }
                else if(grid[curx-1][cury]==4){
                    curx--;
                    dir=4;
                }
                else{
                    break;
                }
            }
            else if(dir==2 && curx!=n-1){
                if(grid[curx+1][cury]==2){
                    curx++;
                    dir=2;
                }
                else if(grid[curx+1][cury]==5){
                    curx++;
                    dir=3;
                }
                else if(grid[curx+1][cury]==6){
                    curx++;
                    dir=4;
                }
                else{
                    break;
                }
            }
            
            else if(dir==3 && cury!=0){
                if(grid[curx][cury-1]==1){
                    cury--;
                    dir=3;
                }
                else if(grid[curx][cury-1]==4){
                    cury--;
                    dir=2;
                }
                else if(grid[curx][cury-1]==6){
                    cury--;
                    dir=1;
                }
                else{
                    break;
                }
            }
            
            else if(dir==4 && cury!=m-1){
                if(grid[curx][cury+1]==1){
                    cury++;
                    dir=4;
                }
                else if(grid[curx][cury+1]==3){
                    cury++;
                    dir=2;
                }
                else if(grid[curx][cury+1]==5){
                    cury++;
                    dir=1;
                }
                else break;
            }
            else break;
        }
        
        curx=0;cury=0;
        if(grid[0][0]==1)dir=4;
        if(grid[0][0]==2)dir=2;
        if(grid[0][0]==3)dir=2;
        if(grid[0][0]==4)dir=2;
        if(grid[0][0]==5)dir=3;
        if(grid[0][0]==6)dir=4;
        
        while(move){
            visited[curx][cury]=true;
            if(dir==1 && curx!=0){
                if(grid[curx-1][cury]==2){
                    curx--;
                    dir=1;
                }
                else if(grid[curx-1][cury]==3){
                    curx--;
                    dir=3;
                }
                else if(grid[curx-1][cury]==4){
                    curx--;
                    dir=4;
                }
                else{
                    break;
                }
            }
            else if(dir==2 && curx!=n-1){
                if(grid[curx+1][cury]==2){
                    curx++;
                    dir=2;
                }
                else if(grid[curx+1][cury]==5){
                    curx++;
                    dir=3;
                }
                else if(grid[curx+1][cury]==6){
                    curx++;
                    dir=4;
                }
                else{
                    break;
                }
            }
            
            else if(dir==3 && cury!=0){
                if(grid[curx][cury-1]==1){
                    cury--;
                    dir=3;
                }
                else if(grid[curx][cury-1]==4){
                    cury--;
                    dir=2;
                }
                else if(grid[curx][cury-1]==6){
                    cury--;
                    dir=1;
                }
                else{
                    break;
                }
            }
            
            else if(dir==4 && cury!=m-1){
                if(grid[curx][cury+1]==1){
                    cury++;
                    dir=4;
                }
                else if(grid[curx][cury+1]==3){
                    cury++;
                    dir=2;
                }
                else if(grid[curx][cury+1]==5){
                    cury++;
                    dir=1;
                }
                else break;
            }
            else break;
        }
        
        return visited[n-1][m-1];
    }
};
