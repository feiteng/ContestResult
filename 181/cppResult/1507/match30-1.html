class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        queue<pair<int, int>> q;
        vector<vector<bool>> vis(n, vector<bool>(m, false));
        q.push({0, 0});
        vis[0][0] = true;
        while (!q.empty()) {
            int x = q.front().first;
            int y = q.front().second;
            int type = grid[x][y];
            q.pop();
            if (type == 1) {
                int dir[2][2] = {{0, -1}, {0, 1}};

                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 1 || nexttype == 4 || nexttype == 6)) || (d == 1 && (nexttype == 1 || nexttype == 3 || nexttype == 5))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            } else if (type == 2) {
                int dir[2][2] = {{1, 0}, {-1, 0}};

                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 2 || nexttype == 5 || nexttype == 6)) || (d == 1 && (nexttype == 2 || nexttype == 3 || nexttype == 4))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            } else if (type == 3) {
                int dir[2][2] = {{0, -1}, {1, 0}};

                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 1 || nexttype == 4 || nexttype == 6)) || (d == 1 && (nexttype == 2 || nexttype == 5 || nexttype == 6))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            } else if (type == 4) {
                int dir[2][2] = {{0, 1}, {1, 0}};

                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 1 || nexttype == 3 || nexttype == 5)) || (d == 1 && (nexttype == 2 || nexttype == 5 || nexttype == 6))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            } else if (type == 5) {
                int dir[2][2] = {{0, -1}, {-1, 0}};
                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 1 || nexttype == 4 || nexttype == 6)) || (d == 1 && (nexttype == 2 || nexttype == 3 || nexttype == 4))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            } else if (type == 6) {
                int dir[2][2] = {{0, 1}, {-1, 0}};
                for (int d = 0; d < 2; d++) {
                    int tx = x + dir[d][0];
                    int ty = y + dir[d][1];
                    if (tx >= 0 && tx < n && ty >= 0 && ty < m && vis[tx][ty] == false) {
                        int nexttype = grid[tx][ty];
                        if ((d == 0 && (nexttype == 1 || nexttype == 3 || nexttype == 5)) || (d == 1 && (nexttype == 2 || nexttype == 3 || nexttype == 4))) {
                            q.push({tx, ty});
                            vis[tx][ty] = true;
                        }
                    }
                }
            }
        }
        return vis[n - 1][m - 1];
    }
};
