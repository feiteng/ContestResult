class Solution {
public:
    
    int dy[4] = {1, 0, -1, 0};
    int dx[4] = {0, 1, 0, -1};
    
    bool hasValidPath(vector<vector<int>>& grid) {
        
        int n = grid.size();
        int m = grid[0].size();
        
        vector<vector<bool> > visited(n, vector<bool>(m));
        queue<pair<int, int> > qq;
        qq.push({0, 0});
        visited[0][0] = 1;
        
        while(!qq.empty()) {
            pair<int, int> pp = qq.front();
            qq.pop();
            
            for(int i = 0; i < 4; i++) {
                int x1 = dx[i] + pp.first;
                int y1 = dy[i] + pp.second;

                if(x1 >= 0 && x1 < n && y1 >= 0 && y1 < m) {
                    if(visited[x1][y1]) continue;
                    
                    if(i == 0 && (grid[pp.first][pp.second] == 1 || grid[pp.first][pp.second] == 4 || grid[pp.first][pp.second] == 6) && (grid[x1][y1] == 3 || grid[x1][y1] == 5 || grid[x1][y1] == 1)) {
                        
                        visited[x1][y1] = true;
                        qq.push({x1, y1});
                    }
                    
                    if(i == 1 && (grid[pp.first][pp.second] == 2 || grid[pp.first][pp.second] == 3 || grid[pp.first][pp.second] == 4) && (grid[x1][y1] == 2 || grid[x1][y1] == 5 || grid[x1][y1] == 6)) {
                        
                        visited[x1][y1] = true;
                        qq.push({x1, y1});
                    }
                    
                    if(i == 2 && (grid[pp.first][pp.second] == 3 || grid[pp.first][pp.second] == 5 || grid[pp.first][pp.second] == 1) && (grid[x1][y1] == 1 || grid[x1][y1] == 4 || grid[x1][y1] == 6)) {
                        
                        visited[x1][y1] = true;
                        qq.push({x1, y1});
                    }
                    
                    if(i == 3 && (grid[pp.first][pp.second] == 2 || grid[pp.first][pp.second] == 5 || grid[pp.first][pp.second] == 6) && (grid[x1][y1] == 2 || grid[x1][y1] == 3 || grid[x1][y1] == 4)) {
                        
                        visited[x1][y1] = true;
                        qq.push({x1, y1});
                    }
                }
            }
            
        }
        
        return visited[n-1][m-1];
        
    }
};
