class Solution {
public:
    bool used[305][305];
    vector<vector<int>> g;
    void dfs(int x, int y){
        // cout<<x<<' '<<y<<' '<<g[x][y]<<endl;

        used[x][y]=1;

        if (g[x][y] == 1){
            if (y+1<g[0].size() && !used[x][y+1] &&  (g[x][y+1] == 3 || g[x][y+1] == 5 || g[x][y+1] == 1) ) dfs(x,y+1);
            if ( y>0 &&!used[x][y-1]  && (g[x][y-1] == 4 || g[x][y-1] == 6 || g[x][y-1] == 1) ) dfs(x,y-1);
        }
        if (g[x][y] == 2){

            if (x+1<g.size() &&!used[x+1][y] && (g[x+1][y] == 5 || g[x+1][y] == 6 || g[x+1][y] == 2) ) dfs(x+1,y);
            if (x>0 && !used[x-1][y] && x>0 && (g[x-1][y] == 3 || g[x-1][y] == 4 || g[x-1][y] == 2) ) dfs(x-1,y);
      
        }
        if (g[x][y] == 3){
            if (x+1<g.size() && !used[x+1][y] &&  (g[x+1][y] == 2 || g[x+1][y] == 5 || g[x+1][y] == 6) ) dfs(x+1,y);
            if (y>0 && !used[x][y-1]  &&  (g[x][y-1] == 1 || g[x][y-1] == 4 || g[x][y-1] == 6) ) dfs(x,y-1);
        }
        if (g[x][y] == 4){
            if ( x+1<g.size() && !used[x+1][y]  && (g[x+1][y] == 2 || g[x+1][y] == 5 || g[x+1][y] == 6) ) dfs(x+1,y);
            // cout<<"44\n";
            if ( y+1<g[0].size() &&  !used[x][y+1] && (g[x][y+1] == 1 || g[x][y+1] == 3 || g[x][y+1] == 5) ) dfs(x,y+1);
        }
        if (g[x][y] == 5){
            if (y>0 &&  !used[x][y-1]  && (g[x][y-1] == 1 || g[x][y-1] == 4 || g[x][y-1] == 6) ) dfs(x,y-1);
            if ( x>0 && !used[x-1][y] && (g[x-1][y] == 2 || g[x-1][y] == 3 || g[x-1][y] == 4) ) dfs(x-1,y);
        }
        if (g[x][y] == 6){
            if (y+1<g[0].size() &&  !used[x][y+1] && (g[x][y+1] == 1 || g[x][y+1] == 3 || g[x][y+1] == 5) ) dfs(x,y+1);
            if ( x>0 && !used[x-1][y] && (g[x-1][y] == 2 || g[x-1][y] == 3 || g[x-1][y] == 4) ) dfs(x-1,y);
        // cout<<"ok";
        
        }
        return;
        
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        // pos = {{2,4,0}, {1,4,5}, {0,1,4,5},}
        for(int i=0;i<305; ++i)
            for(int j=0;j<305;++j)
                used[i][j]=0;
        g=grid;
        // cout<<grid.size()<<' '<<grid[0].size()<<endl;
        dfs(0,0);
        // cout<<"eb";
        // return 0;
        return used[grid.size()-1][grid[0].size()-1];
    }
};
