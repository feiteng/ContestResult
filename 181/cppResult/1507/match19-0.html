class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        queue<pair<int,int>> q;
        vector<vector<int>> d = {{0,-1},{0,1},{-1,0},{1,0}};
        vector<vector<int>> moves = {{0,0},{0,1},{2,3},{0,3},{1,3},{0,2},{1,2}};
        vector<vector<int>> froms = {{0,0},{0,1},{2,3},{1,2},{0,2},{1,3},{0,3}};
        q.push({0,0});
        int m = grid.size();
        int n = grid[0].size();
        int vis[m][n];
        for(int i = 0 ; i < m; i++)for(int j = 0; j < n; j++)vis[i][j] = 0;
        while(!q.empty()){
            auto t = q.front();
            q.pop();
            int x = t.first;
            int y = t.second;
            //cout << x << y << endl;
            if(x == m-1 && y == n-1)return true;
            for(auto v : moves[grid[x][y]]){
                int nx = x + d[v][0];
                int ny = y + d[v][1];
                if(0 <= nx && nx < m && 0 <= ny && ny < n && !vis[nx][ny] && find(froms[grid[nx][ny]].begin(), froms[grid[nx][ny]].end(),v) != froms[grid[nx][ny]].end()){
                    vis[nx][ny] = 1;
                    q.push({nx,ny});
                }
            }
        }
        return false;
    }
};
