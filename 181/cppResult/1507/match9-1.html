class Solution {
public:
    int n, m;
    bool dfs(int x, int y, bool left, bool right, bool top, bool bot, vector<vector<int>>& grid) {
        if(x < 0 || x >= n || y < 0 || y >= m) return false;
        bool flg = (x==n-1 && y == m-1);
        if(grid[x][y] == 1) {
            if(flg && (left || right)) return true;

            if(left) {
                if (dfs(x, y+1, true, false, false, false, grid)) return true;
            }
            if(right) {
                if (dfs(x, y-1, false, true, false, false, grid)) return true;
            }
        }
        else if(grid[x][y] == 2) {
            if(flg && (top || bot)) return true;

            if(top) {
                if(dfs(x+1, y, false, false, true, false, grid))return true;
            }
            if(bot) {
                if (dfs(x-1, y, false, false, false, true, grid)) return true;
            }
        }
        else if(grid[x][y] == 3) {
            if(flg && (left || bot)) return true;

            if(left) {
                if (dfs(x+1, y, false, false, true, false, grid))return true;
            }
            if(bot) {
                if (dfs(x, y-1, false, true, false, false, grid))return true;
            }
        }
        else if(grid[x][y] == 4) {
            if(flg && (right || bot)) return true;

            if(right) {
                if (dfs(x+1, y, false, false, true, false, grid))return true;
            }
            if(bot) {
                if (dfs(x, y+1, true, false, false, false, grid))return true;
            }
        }
        else if(grid[x][y] == 5) {
            if(flg && (left || top)) return true;
            if(left) {
                if (dfs(x-1, y, false, false, false, true, grid))return true;
            }
            if(top) {
                if (dfs(x, y-1, false, true, false, false, grid))return true;
            }
        }
        else {
            if(flg && (right || top)) return true;

            if(right) {
                if (dfs(x-1, y, false, false, false, true, grid))return true;
            }
            if(top) {
                if (dfs(x, y+1, true, false, false, false, grid))return true;
            }
        }
        return false;
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        n = grid.size(), m = grid[0].size();
        return dfs(0, 0, true, true, true, true, grid);
    }
};
