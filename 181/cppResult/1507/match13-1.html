class Solution {
public:
    vector<vector<int>> grid;
    int m, n;
    vector<vector<int>> vis;
    
    bool dfs(int x, int y)
    {
        if (x < 0 || x >= m || y < 0 || y >= n || vis[x][y]) return false;
        if (x == m - 1 && y == n - 1) 
        {
            
            return true;
        }
        bool ans = false;
        vis[x][y] = 1;
        int nx, ny;

        if (grid[x][y] == 1)
        {
            if (y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6)) ans = ans || dfs(x, y - 1);
            if (y + 1 < n && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5)) ans = ans || dfs(x, y + 1);
        }
        else if (grid[x][y] == 2)
        {
            if (x - 1 >= 0 && (grid[x - 1][y] == 2 ||grid[x - 1][y] == 3 || grid[x - 1][y] == 4)) ans = ans || dfs(x - 1, y);
            if (x + 1 < m && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6)) ans = ans || dfs(x + 1, y);
        }
        else if (grid[x][y] == 3)
        {
            if (y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6))ans = ans || dfs(x, y - 1);
            if (x + 1 < m && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6))ans = ans || dfs(x + 1, y);
        }
        else if (grid[x][y] == 4)
        {
            if (x + 1 < m && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6))ans = ans || dfs(x + 1, y);
            if (y + 1 < n && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5))ans = ans || dfs(x, y + 1);
        }
        else if (grid[x][y] == 5)
        {
            if (x - 1 >= 0 && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4))ans = ans || dfs(x - 1, y);
            if (y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6))ans = ans || dfs(x, y - 1);
        }
        else if (grid[x][y] == 6)
        {
            if (x - 1 >= 0 && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4))ans = ans || dfs(x - 1, y);
            if (y + 1  < n && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5))ans = ans || dfs(x, y + 1);
        }
        return ans;
    }
    
    bool hasValidPath(vector<vector<int>>& grid) {
        this->grid = grid;
        m = grid.size(), n = grid[0].size();
        vis = vector<vector<int>>(m, vector<int>(n));
        return dfs(0,0);
    }
};
