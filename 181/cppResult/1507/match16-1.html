class Solution {
public:
    
    bool flag = false;
    
    int f[305][305];
    
    bool check(int x,int y,vector<vector<int>>& g) {
        if (x >= g.size() || y >= g[0].size() || x < 0 || y < 0 || f[x][y]) {
            return false;
        }
        return true;
    }
    
    
    void dfs(int x,int y,vector<vector<int>>& g) {
        if (x == g.size()-1 && y == g[0].size()-1) {
            flag = true;
            return ;
        }
        if (flag) {
            return;
        }
        if (f[x][y]) return;

        f[x][y] = 1;
        if (g[x][y] == 1) {
            if (check(x,y-1,g) && (g[x][y-1] == 1 || g[x][y-1] == 4 || g[x][y-1] == 6)) {
                dfs(x,y-1,g);
            }
            if (check(x,y+1,g) && (g[x][y+1] == 1 || g[x][y+1] == 3 || g[x][y+1] == 5)) {
                dfs(x,y+1,g);
            }
        } else if (g[x][y] == 2) {
            if (check(x-1,y,g) && (g[x-1][y] == 2 || g[x-1][y] == 3 || g[x-1][y] == 4)) {
                dfs(x-1,y,g);
            }
            if (check(x+1,y,g) && (g[x+1][y] == 2 || g[x+1][y] == 5 || g[x+1][y] == 6)) {
                dfs(x+1,y,g);
            }
        } else if (g[x][y] == 3) {
            if (check(x+1,y,g) && (g[x+1][y] == 2 || g[x+1][y] == 5 || g[x+1][y] == 6)) {
                dfs(x+1,y,g);
            }
            if (check(x,y-1,g) && (g[x][y-1] == 1 || g[x][y-1] == 4 || g[x][y-1] == 6)) {
                dfs(x,y-1,g);
            }
        } else if (g[x][y] == 4) {
            if (check(x,y+1,g) && (g[x][y+1] == 1 || g[x][y+1] == 3 || g[x][y+1] == 5)) {
                dfs(x,y+1,g);
            }
            if (check(x+1,y,g) && (g[x+1][y] == 2 || g[x+1][y] == 5 || g[x+1][y] == 6)) {
                dfs(x+1,y,g);
            }
        } else if (g[x][y] == 5) {
            if (check(x,y-1,g) && (g[x][y-1] == 1 || g[x][y-1] == 4 || g[x][y-1] == 6)) {
                dfs(x,y-1,g);
            }
            if (check(x-1,y,g) && (g[x-1][y] == 2 || g[x-1][y] == 3 || g[x-1][y] == 4)) {
                dfs(x-1,y,g);
            }
        } else {
            if (check(x-1,y,g) && (g[x-1][y] == 2 || g[x-1][y] == 3 || g[x-1][y] == 4)) {
                dfs(x-1,y,g);
            }
            if (check(x,y+1,g) && (g[x][y+1] == 1 || g[x][y+1] == 3 || g[x][y+1] == 5)) {
                dfs(x,y+1,g);
            }
        }
    }
    
    
    bool hasValidPath(vector<vector<int>>& grid) {
        memset(f,0,sizeof f);
        dfs(0,0,grid);   
        return flag;
    }
};
