class Solution {
public:
    
    vector<vector<int>> a;
    int n,m;
    int used[303][303];
    set<int> path[4][7];
    int dx[4]={0,1,0,-1};
    int dy[4]={1,0,-1,0};
    
    bool ok(int x,int y)
    {
        return min(x,y)>=0 && x<n && y<m;
    }
    
    void go(int x,int y)
    {
        used[x][y]=1;
        for(int d=0;d<4;d++)
        {
            int nx=x+dx[d];
            int ny=y+dy[d];
            if(!ok(nx,ny)) continue;
            if(used[nx][ny]) continue;
            if(path[d][a[x][y]].count(a[nx][ny]))
                go(nx,ny);
        }
    }
    
    
    bool hasValidPath(vector<vector<int>>& grid) {
        
        path[0][1].insert(1);
        path[0][1].insert(3);
        path[0][1].insert(5);
        
        
        path[2][1].insert(1);
        path[2][1].insert(4);
        path[2][1].insert(6);
        
        
        path[1][2].insert(2);
        path[1][2].insert(5);
        path[1][2].insert(6);
        
        path[3][2].insert(2);
        path[3][2].insert(3);
        path[3][2].insert(4);
        
        
        path[1][3].insert(2);
        path[1][3].insert(5);
        path[1][3].insert(6);
        
        path[2][3].insert(1);
        path[2][3].insert(4);
        path[2][3].insert(6);
        
        
        path[0][4].insert(1);
        path[0][4].insert(3);
        path[0][4].insert(5);
        
        path[1][4].insert(2);
        path[1][4].insert(5);
        path[1][4].insert(6);
        
        path[2][5].insert(1);
        path[2][5].insert(4);
        path[2][5].insert(6);
        
        path[3][5].insert(2);
        path[3][5].insert(3);
        path[3][5].insert(4);
        
        path[3][6].insert(2);
        path[3][6].insert(3);
        path[3][6].insert(4);
        
        path[0][6].insert(1);
        path[0][6].insert(3);
        path[0][6].insert(5);
        
        a=grid;
        n=grid.size();
        m=grid[0].size();
        go(0,0);
        return used[n-1][m-1];
        
    }
};
