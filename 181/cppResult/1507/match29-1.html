class Solution {
public:
    int dr[5] = {0,-1,0,1,0};
    int dc[5] = {0,0,1,0,-1};
    int dir[7][2] = {{0,0},{2,4}, {1,3}, {3,4}, {2,3},{1,4},{1,2}};
    int next[5][3] = {{0,0,0}, {2,3,4},{1,3,5},{2,5,6},{1,4,6}};
    bool vis[310][310];
    int row, col;
    void dfs(vector<vector<int>>& grid, int r, int c) {
        if(vis[r][c]) return;
        vis[r][c] = true;
        int s = grid[r][c];

        for(int i = 0; i < 2; ++i) {
            int d = dir[s][i];
            int rr = r+dr[d];
            int cc = c+dc[d];
            if(!valid(rr,cc)) continue;
            int ns = grid[rr][cc];
            for(int j = 0; j < 3; ++j) {
                if(ns == next[d][j]) dfs(grid, rr, cc);
            }
        }
    }
    bool valid(int r, int c) {
        return r >= 0 && r < row && c >= 0 && c < col;
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        row = grid.size();
        col = grid[0].size();
        memset(vis,0, sizeof(vis));
        dfs(grid,0,0);
        return vis[row-1][col-1];
    }
};
