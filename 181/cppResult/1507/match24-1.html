class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int M=grid.size();
        int N=grid[0].size();
        vector<vector<bool>> seen(M,vector<bool>(N,false));
        seen[0][0]=true;
        queue<pair<int,int>> qq;
        qq.push(make_pair(0,0));

        int dd[][2]={{1,0},{-1,0},{0,1},{0,-1}};
        while(!qq.empty()){
            auto x=qq.front().first;
            auto y=qq.front().second;
            qq.pop();
            for(int i=0;i<4;i++){
                int xx=x+dd[i][0];
                int yy=y+dd[i][1];
                if(xx<0||yy<0||xx>=M||yy>=N) continue;

                if(seen[xx][yy]) continue;
                bool f=false;
                if(!f&&dd[i][0]==1&&(grid[x][y]==2||grid[x][y]==3||grid[x][y]==4)&&(grid[xx][yy]==2||grid[xx][yy]==5||grid[xx][yy]==6)){
                    f=true;
                }
                if(!f&&dd[i][0]==-1&&(grid[x][y]==2||grid[x][y]==5||grid[x][y]==6)&&(grid[xx][yy]==2||grid[xx][yy]==3||grid[xx][yy]==4)){
                    f=true;
                }
                
                if(!f&&dd[i][1]==-1&&(grid[x][y]==1||grid[x][y]==3||grid[x][y]==5)&&(grid[xx][yy]==1||grid[xx][yy]==4||grid[xx][yy]==6)){
                    f=true;
                }
                if(!f&&dd[i][1]==1&&(grid[x][y]==1||grid[x][y]==4||grid[x][y]==6)&&(grid[xx][yy]==1||grid[xx][yy]==3||grid[xx][yy]==5)){
                    f=true;
                }
                if(f){
                    seen[xx][yy]=true;
                    qq.push(make_pair(xx,yy));
                }
            }
        }
        return seen[M-1][N-1];
    }
};
