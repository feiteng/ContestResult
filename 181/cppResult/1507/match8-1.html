class Solution {
public:
    bool visit[345][345];
    int N, M;
    bool check(int x, int y, vector<int> OK, vector<vector<int>>& grid) {
        if (x < 0 || x == N) return false;
        if (y < 0 || y == M) return false;
        for (int i : OK)
            if (grid[x][y] == i) 
                return true;
        return false;
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        N = (int)grid.size();
        if (!N) return false;
        M = (int)grid[0].size();
        if (!M) return false;
        queue<pair<int, int>> qu;
        memset(visit, 0, sizeof(visit));
        qu.push(make_pair(0, 0));
        while (!qu.empty()) {
            auto tp = qu.front();
            qu.pop();
            int x = tp.first, y = tp.second;
            if (visit[x][y] == 1) continue;
            assert(!visit[x][y]);
           // printf("x = %d y = %d\n", x, y);
            visit[x][y] = 1;
            if (grid[x][y] == 1) {

                int nx = x, ny = y + 1;
                if (check(nx, ny, {1, 3, 5}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x, ny = y - 1;
                if (check(nx, ny, {1, 4, 6}, grid))
                    qu.push(make_pair(nx, ny));
                
            } else if (grid[x][y] == 2) {
                int nx = x + 1, ny = y;

                if (check(nx, ny, {2, 5, 6}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x - 1, ny = y;
                if (check(nx, ny, {2, 4, 3}, grid))
                    qu.push(make_pair(nx, ny));
                
            } else if (grid[x][y] == 3) {
                int nx = x + 1, ny = y;

                if (check(nx, ny, {2, 5, 6}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x, ny = y - 1;
                if (check(nx, ny, {1, 4, 6}, grid))
                    qu.push(make_pair(nx, ny));
                
            } else if (grid[x][y] == 4) {
                int nx = x, ny = y + 1;

                if (check(nx, ny, {1, 3, 5}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x + 1, ny = y;
                if (check(nx, ny, {2, 5, 6}, grid))
                    qu.push(make_pair(nx, ny));
                
            } else if (grid[x][y] == 5) {
                int nx = x - 1, ny = y;

                if (check(nx, ny, {2, 4, 3}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x, ny = y - 1;
                if (check(nx, ny, {1, 4, 6}, grid))
                    qu.push(make_pair(nx, ny));
                
            } else {
                int nx = x, ny = y + 1;

                if (check(nx, ny, {1, 3, 5}, grid))
                    qu.push(make_pair(nx, ny));
                nx = x - 1, ny = y;
                if (check(nx, ny, {2, 4, 3}, grid))
                    qu.push(make_pair(nx, ny));
            }
        }
        return visit[N - 1][M - 1];
    }
};
