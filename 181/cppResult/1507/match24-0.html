class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        
        int diff[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        bool vis[305][305] = {false};
        stack<pair<int, int>>  now;
        now.push({0, 0});
        vis[0][0] = true;
        
        while (now.size() > 0) {
            auto p = now.top();
            now.pop();
            
            // printf("(%d, %d)\n", p.first, p.second);
            
            for (int k = 0; k < 4; k++) {
                int x = p.first + diff[k][0];
                int y = p.second + diff[k][1];
                
                if (x<0 || x>= m || y <0 || y >= n) continue;
                
                bool valid = false;
                if (k== 2 && (grid[p.first][p.second] == 2 || grid[p.first][p.second] == 3 || grid[p.first][p.second] == 4)
                    && (grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6)) {
                    valid = true;
                }
                
                if (k == 3 && (grid[p.first][p.second] == 2 || grid[p.first][p.second] == 5 || grid[p.first][p.second] == 6)
                   && (grid[x][y] == 2 || grid[x][y] == 3 || grid[x][y] == 4)) {
                    valid = true;
                }
                
                if (k==0 && (grid[p.first][p.second] == 1 || grid[p.first][p.second] == 4 || grid[p.first][p.second] == 6)
                    && (grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5)) {
                    valid = true;
                }
                
                if (k == 1 && (grid[p.first][p.second] == 1 || grid[p.first][p.second] == 3 || grid[p.first][p.second] == 5)
                   && (grid[x][y] == 1 || grid[x][y] == 4 || grid[x][y] == 6)) {
                    valid = true;
                }
                
                if (valid && !vis[x][y]) {
                    vis[x][y] = true;
                    now.push({x, y});
                }
            }
            
        }
            
        return vis[m-1][n-1];
    }
};
