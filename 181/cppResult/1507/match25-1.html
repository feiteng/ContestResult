class Solution {
public:
    bool vis[405][405][4];
    struct Go {
        int x,y,d;
    };

    pair<int,int> d1[6] = {{0,2},{1,3},{0,3},{3,2},{0,1},{1,2}};
    pair<int,int> go[4] = {{0,-1},{-1,0},{0,1},{1,0}};

    int opp (int dir) {
        if (dir == 1) return 3;
        if (dir == 3) return 1;
        return dir ? 0 : 2;
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        for (vector<int> &v : grid) for (auto &au : v) --au;
        queue<Go> q;
        for (int i = 0; i < 4; i++) q.push({0,0,i});
        while (!q.empty()) {
            Go cur = q.front(); q.pop();
            int type = grid[cur.x][cur.y];
            if (cur.d == d1[type].first) {
                if (cur.x+1==grid.size() && cur.y+1==grid[0].size()) return true;
                int nx = cur.x + go[d1[type].second].first, ny = cur.y + go[d1[type].second].second;
                if (nx >= 0 && nx < grid.size() && ny >= 0 && ny < grid[0].size() && !vis[nx][ny][opp(d1[type].second)]) {
                    vis[nx][ny][opp(d1[type].second)] = 1;
                    q.push({nx,ny,opp(d1[type].second)});
                }
            }
            if (cur.d == d1[type].second) {
                if (cur.x+1==grid.size() && cur.y+1==grid[0].size()) return true;

                int nx = cur.x + go[d1[type].first].first, ny = cur.y + go[d1[type].first].second;
                if (nx >= 0 && nx < grid.size() && ny >= 0 && ny < grid[0].size() && !vis[nx][ny][opp(d1[type].first)]) {
                    vis[nx][ny][opp(d1[type].first)] = 1;
                    q.push({nx,ny,opp(d1[type].first)});
                }
            }
        }
        return false;
    }
};
