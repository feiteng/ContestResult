class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        //emset(vis, false, sizeof(vis));
        int m=grid.size(), n=grid[0].size();
        for(int i=1;i<=4;i++){
            int x=0, y=0, pos=i;
            bool finishing=false;
            while(1){
                if(x==m-1&&y==n-1) finishing=true;
                if(x<0||x>=m||y<0||y>=n) break;
                

                if(grid[x][y]==1){
                    if(pos==2){ y=y-1; pos=2; }
                    else if(pos==4){ y=y+1; pos=4; }
                    else break;

                }
                else if(grid[x][y]==2){
                    if(pos==1){ x=x+1; pos=1; }
                    else if(pos==3){ x=x-1; pos=3; }
                    else break;

                }
                else if(grid[x][y]==3){
                    if(pos==3){ y=y-1; pos=2; }
                    else if(pos==4){ x=x+1; pos=1; }
                    else break;

                }
                else if(grid[x][y]==4){
                    if(pos==2){ x=x+1; pos=1; }
                    else if(pos==3){ y=y+1; pos=4; }
                    else break;

                }
                else if(grid[x][y]==5){
                    if(pos==1){ y=y-1; pos=2; }
                    else if(pos==4){ x=x-1; pos=3; }
                    else break;

                }
                else if(grid[x][y]==6){
                    if(pos==1){ y=y+1; pos=4; }
                    else if(pos==2){ x=x-1; pos=3; }
                    else break;
                }
                if(finishing) return true;
            }
        }
        return false;
    }
};
