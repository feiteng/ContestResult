class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        queue<pair<int,int>> cells;
        
        int m = grid.size(), n = grid[0].size();
        
        vector<vector<int>> vis(m, vector<int>(n, false));
        
        cells.push({0,0});
        while(!cells.empty()) 
        {
            auto cur = cells.front(); cells.pop();
            
            if(cur.first == m-1 && cur.second == n-1)
                return true;
            int x = cur.first, y = cur.second;
            vis[x][y] = true;
            
            if(grid[x][y] == 5) {
                if(isValid(x-1, y, m, n) && (grid[x-1][y] == 2 || grid[x-1][y] == 4 || grid[x-1][y] == 3)  && !vis[x-1][y])
                    cells.push({x-1,y});
                
                if(isValid(x, y-1, m, n) && (grid[x][y-1] == 1 || grid[x][y-1] == 4 || grid[x][y-1] == 6) && !vis[x][y-1]) 
                    cells.push({x,y-1});
            }
            else if(grid[x][y] == 6) {
                if(isValid(x-1, y, m, n) && (grid[x-1][y] == 2 || grid[x-1][y] == 4 || grid[x-1][y] == 3)  && !vis[x-1][y])
                    cells.push({x-1,y});
                
                if(isValid(x, y+1, m, n) && (grid[x][y+1] == 1 || grid[x][y+1] == 3 || grid[x][y+1] == 5) && !vis[x][y+1]) 
                    cells.push({x,y+1});
            }
            else if(grid[x][y] == 3) {
                if(isValid(x+1, y, m, n) && (grid[x+1][y] == 2 || grid[x+1][y] == 5 || grid[x+1][y] == 6) && !vis[x+1][y])
                    cells.push({x+1,y});
                
                if(isValid(x, y-1, m, n) && (grid[x][y-1] == 1 || grid[x][y-1] == 4 || grid[x][y-1] == 6)  && !vis[x][y-1]) 
                    cells.push({x,y-1});
            }
            else if(grid[x][y] == 4) {
                if(isValid(x+1, y, m, n) && (grid[x+1][y] == 2 || grid[x+1][y] == 5 || grid[x+1][y] == 6)  && !vis[x+1][y])
                    cells.push({x+1,y});
                
                if(isValid(x, y+1, m, n) && (grid[x][y+1] == 1 || grid[x][y+1] == 3 || grid[x][y+1] == 5)  && !vis[x][y+1]) 
                    cells.push({x,y+1});
            } 
            else if(grid[x][y] == 1) {
                if(isValid(x, y+1, m, n) && (grid[x][y+1] == 1 || grid[x][y+1] == 3 || grid[x][y+1] == 5) && !vis[x][y+1]) 
                    cells.push({x,y+1});
                
                if(isValid(x, y-1, m, n) && (grid[x][y-1] == 1 || grid[x][y-1] == 4 || grid[x][y-1] == 6) && !vis[x][y-1]) 
                    cells.push({x,y-1});
            }
            else if(grid[x][y] == 2) {
                if(isValid(x+1, y, m, n) && (grid[x+1][y] == 2 || grid[x+1][y] == 5 || grid[x+1][y] == 6)  && !vis[x+1][y])
                    cells.push({x+1,y});
                if(isValid(x-1, y, m, n) && (grid[x-1][y] == 2 || grid[x-1][y] == 3 || grid[x-1][y] == 4)  && !vis[x-1][y])
                    cells.push({x-1,y});
            }
        }
        
        return false;
    }
    
    bool isValid(int i, int j, int m, int n) {
        return 0 <= i && i < m && 0 <= j && j < n;
    }
};
