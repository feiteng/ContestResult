class Solution {
public:
    int flag, m, n;
    int vis[305][305];
    bool hasValidPath(vector<vector<int>>& grid) {
        flag = 0;
        memset(vis, 0, sizeof(vis));
        m = grid.size(), n = grid[0].size();
        dfs(grid, 0, 0);
        if(flag) 
            return true;
        else
            return false;
    }
    void dfs(vector<vector<int>>& grid, int a, int b)
    {
        if(vis[a][b]) return;
        vis[a][b] = 1;
        if(a == m-1 && b == n-1)
        {
            flag = 1;
            return;
        }

        if(grid[a][b] == 1)
        {
            if(a >= 0 && a < m && b-1 >= 0 && b-1 < n && (grid[a][b-1] == 1 || grid[a][b-1] == 4 || grid[a][b-1] == 6)) dfs(grid, a, b-1);
            if(a >= 0 && a < m && b+1 >= 0 && b+1 < n && (grid[a][b+1] == 1 || grid[a][b+1] == 3 || grid[a][b+1] == 5)) dfs(grid, a, b+1);
        }
        if(grid[a][b] == 2)
        {
            if(a-1 >= 0 && a-1 < m && b >= 0 && b < n && (grid[a-1][b] == 2 || grid[a-1][b] == 3 || grid[a-1][b] == 4)) dfs(grid, a-1, b);
            if(a+1 >= 0 && a+1 < m && b >= 0 && b < n && (grid[a+1][b] == 2 || grid[a+1][b] == 5 || grid[a+1][b] == 6)) dfs(grid, a+1, b);
        }
        if(grid[a][b] == 3)
        {
            if(a >= 0 && a < m && b-1 >= 0 && b-1 < n && (grid[a][b-1] == 1 || grid[a][b-1] == 4 || grid[a][b-1] == 6)) dfs(grid, a, b-1);
            if(a+1 >= 0 && a+1 < m && b >= 0 && b < n && (grid[a+1][b] == 2 || grid[a+1][b] == 5 || grid[a+1][b] == 6)) dfs(grid, a+1, b);
        }
        if(grid[a][b] == 4)
        {
            if(a >= 0 && a < m && b+1 >= 0 && b+1 < n && (grid[a][b+1] == 1 || grid[a][b+1] == 5 || grid[a][b+1] == 3)) dfs(grid, a, b+1);
            if(a+1 >= 0 && a+1 < m && b >= 0 && b < n && (grid[a+1][b] == 2 || grid[a+1][b] == 5 || grid[a+1][b] == 6)) dfs(grid, a+1, b);
        }
        if(grid[a][b] == 5)
        {
            if(a >= 0 && a < m && b-1 >= 0 && b-1 < n && (grid[a][b-1] == 1 || grid[a][b-1] == 4 || grid[a][b-1] == 6)) dfs(grid, a, b-1);
            if(a-1 >= 0 && a-1 < m && b >= 0 && b < n && (grid[a-1][b] == 2 || grid[a-1][b] == 3 || grid[a-1][b] == 4)) dfs(grid, a-1, b);
        }
        if(grid[a][b] == 6)
        {
            if(a >= 0 && a < m && b+1 >= 0 && b+1 < n && (grid[a][b+1] == 1 || grid[a][b+1] == 5 || grid[a][b+1] == 3)) dfs(grid, a, b+1);
            if(a-1 >= 0 && a-1 < m && b >= 0 && b < n && (grid[a-1][b] == 2 || grid[a-1][b] == 3 || grid[a-1][b] == 4)) dfs(grid, a-1, b);
        }
    }
};
