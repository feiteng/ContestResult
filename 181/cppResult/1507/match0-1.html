class Solution {
public:
    
    bool iv(int i, int j, int n, int m) {
        return i >= 0 && i <= n-1 && j >= 0 && j <= m-1;
    }
    
    bool df(vector<vector<int>>&grid, vector<vector<bool>>&visited, int i, int j, int n, int m) {

        if(i == n-1 && j == m-1)
            return true;
        visited[i][j] = true;
        // cout << grid[i][j] << " ";
        if(grid[i][j] == 2) {
            if(iv(i+1, j, n, m) && !visited[i+1][j] && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6)) {
                if(df(grid, visited, i+1, j, n, m))
                    return true;
            } 
            if(iv(i-1, j, n, m) && !visited[i-1][j] && (grid[i-1][j] == 2 ||grid[i-1][j] == 3 || grid[i-1][j] == 4) ) {
                if(df(grid, visited, i-1, j, n, m))
                    return true;
            }
        } else if(grid[i][j] == 1) {
            if(iv(i, j+1, n, m) && !visited[i][j+1] && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5)) {

                if(df(grid, visited, i, j+1, n, m))
                    return true;
            } 
            if(iv(i, j-1, n, m) && !visited[i][j-1] && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6)) {
                if(df(grid, visited, i, j-1, n, m))
                    return true;
            }
        } else if(grid[i][j] == 3) {
            if(iv(i+1, j, n, m) && !visited[i+1][j] && (grid[i+1][j] == 5 || grid[i+1][j] == 6 || grid[i+1][j] == 2)) {
                if(df(grid, visited, i+1, j, n, m))
                    return true;
            }
            if(iv(i, j-1, n, m) && !visited[i][j-1] && (grid[i][j-1] == 4 || grid[i][j-1] == 1 || grid[i][j-1] == 6)) {
                if(df(grid, visited, i, j-1, n, m))
                    return true;
            } 
        } else if(grid[i][j] == 4) {
            if(iv(i+1, j, n, m) && !visited[i+1][j] && (grid[i+1][j] == 6 || grid[i+1][j] == 5 || grid[i+1][j] == 2)) {
                if(df(grid, visited, i+1, j, n, m))
                    return true;
            }
            if(iv(i, j+1, n, m) && !visited[i][j+1] && (grid[i][j+1] == 5 || grid[i][j+1] == 3 || grid[i][j+1] == 1)) {
                if(df(grid, visited, i, j+1, n, m))
                    return true;
            } 
        } else if(grid[i][j] == 5){
            if(iv(i-1, j, n, m) && !visited[i-1][j] && (grid[i-1][j] == 4 || grid[i-1][j] == 3 || grid[i-1][j] == 2)) {
                if(df(grid, visited, i-1, j, n, m))
                    return true;
            }
            if(iv(i, j-1, n, m) && !visited[i][j-1] && (grid[i][j-1] == 4 || grid[i][j-1] == 1 || grid[i][j-1] == 6)) {
                if(df(grid, visited, i, j-1, n, m))
                    return true;
            }
        } else if(grid[i][j] == 6) {
            if(iv(i-1, j, n, m) && !visited[i-1][j] && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4)) {
                if(df(grid, visited, i-1, j, n, m))
                    return true;
            }
            if(iv(i, j+1, n, m) && !visited[i][j+1] && (grid[i][j+1] == 5 || grid[i][j+1] == 3 || grid[i][j+1] == 1)) {
                if(df(grid, visited, i, j+1, n, m))
                    return true;
            } 
        }
        return false;
    }
    
    bool hasValidPath(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<bool>>visited(n, vector<bool>(m, false));
        return df(grid, visited, 0, 0, n, m);
    }
};
