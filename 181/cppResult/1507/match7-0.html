class Solution {
public:
    int m,n;
    bool vis[310][310];
    bool inBoard(int row,int col)
    {
        if(row>=0&&row<m&&col>=0&&col<n)
            return true;
        return false;
    }
    void dfs(int row,int col,vector<vector<int>>& grid)
    {
        if(grid[row][col]==1)
        {
            int nextRow=row;
            int nextCol=col+1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==5))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row;
            nextCol=col-1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==4||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
        else if(grid[row][col]==2)
        {
            int nextRow=row-1;
            int nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==4))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row+1;
            nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==5||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
        else if(grid[row][col]==3)
        {
            int nextRow=row;
            int nextCol=col-1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==4||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row+1;
            nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==5||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
        else if(grid[row][col]==4)
        {
            int nextRow=row;
            int nextCol=col+1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==5))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row+1;
            nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==5||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
        else if(grid[row][col]==5)
        {
            int nextRow=row-1;
            int nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==4))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row;
            nextCol=col-1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==4||grid[nextRow][nextCol]==6))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
        else if(grid[row][col]==6)
        {
            int nextRow=row-1;
            int nextCol=col;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==2||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==4))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
            nextRow=row;
            nextCol=col+1;
            if(inBoard(nextRow,nextCol)&&(!vis[nextRow][nextCol])&&
               (grid[nextRow][nextCol]==1||grid[nextRow][nextCol]==3||grid[nextRow][nextCol]==5))
            {
                vis[nextRow][nextCol]=true;
                dfs(nextRow,nextCol,grid);
            }
        }
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        m=grid.size();
        n=grid[0].size();
        memset(vis,0,sizeof(vis));
        vis[0][0]=true;
        dfs(0,0,grid);
        if(vis[m-1][n-1]) return true;
        else return false;
    }
};
