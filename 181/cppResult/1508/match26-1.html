class Solution {
public:
    string longestPrefix(string s) {
        int len = (int)s.size();
    vector<int> dp(len, 0);
    for(int i = 1; i < len; i++)
    {
        int I = i - 1;
        while(I >= 0 && s[i] != s[dp[I]])
            I = dp[I] - 1;
        if(I == -1) dp[i] = 0;
        else dp[i] = dp[I] + 1;
    }
    return s.substr(0, dp[len - 1]);
    }
};
