class Solution {
public:
    string longestPrefix(string s) {
		int n = s.length();
		vector<int> dp(n, 0);
		dp[0] = 0;
		int len = 0;
		int i = 1;
		while (i < n) {
			if (s[i] == s[len]) {
				len++;
				dp[i] = len;
				i++;
			}
			else {
				if (len != 0) {
					len = dp[len - 1];
				}
				else {
					dp[i] = 0;
					i++;
				}
			}
		}

		int res = dp[n - 1];
		if (res == 0) return "";
		return s.substr(s.size() - res, res);
	}
};
