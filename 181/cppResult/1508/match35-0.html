class Solution {
public:
    string longestPrefix(string s) {
        int n = s.length();
        
        vector<int> lps(n, 0);
        
        int i = 1, len = 0;
        while (i < n) {
            if (s[i] == s[len]) {
                len++;
                lps[i] = len;
                i++;
            }
            else {
                if(len != 0) {
                    len = lps[len - 1];
                } else {
                    lps[i] = 0;
                    i++;
                }
            }
        }
        
        int res = lps[n - 1];
        
        string ans = "";
        
        for(int i = 0 ; i < res; i++) {
            ans += s[i];
        }
        return ans;
    }
};
