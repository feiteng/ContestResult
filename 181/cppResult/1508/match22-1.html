class Solution {
public:
    string longestPrefix(string s) {
        int n = s.length();
        if (n < 2)
            return "";
        
        vector<int> next(n);
        
        int j = 0;
        for (int i = 1; i < n; ++i) {
            while (j > 0 && s[i] != s[j]) {
                j = next[j-1];
            }
            
            if (s[i] == s[j]) {
                next[i] = j+1;
                ++j;
            }
        }
        int ans = next[n-1];
        // if (ans == 0)
        //     return s.substr(0, 1);
        
        // if (n % (n-ans) == 0) {
        //     ans = n;
        // }
        return s.substr(0, ans);
    }
};
