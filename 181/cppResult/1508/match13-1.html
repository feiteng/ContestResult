class Solution {
public:
    // kmp algorithm
    string longestPrefix(string s) {
        vector<int> kmp(s.size());
        kmp[0] = 0;
        int n = s.size(); 

        int l = 0, i = 1; 
        while (i < n) { 
            if (s[i] == s[l]) { 
                kmp[i++] = ++l; 
            } 
            else { 
                if (l) { 
                    l = kmp[l - 1]; 
                } 
                else { 
                    kmp[i] = 0; 
                    i++; 
                } 
            } 
        } 

        return s.substr(0, kmp[n - 1]);
    }
};
