class Solution {
public:
    string longestPrefix(string s) {
        int len = s.size() ;
        vector<int> next(len + 10, 0);
        next[0] = 0 ;
        int i = 0 , j = 0 ;
        for(j = 1 ; j < len ; j++) {               
            while( s[i] != s[j] && i > 0  )
                i = next[i-1] ;
            if(s[i] == s[j])    {
                i++ ;                             
                next[j] = i ;                       
            }
            else
                next[j] = 0 ;
        }
        return s.substr(0,next[len-1]);
    }

};
