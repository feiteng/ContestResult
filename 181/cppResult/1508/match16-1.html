class Solution {
public:
    string longestPrefix(string s) {
        vector<int> dp(s.size());

        for (int i = 1, len = 0; i < s.size();) {
            if (s[i] == s[len]) {
                dp[i++] = ++len;
            } else {
                if (0 != len) {
                    len = dp[len - 1];
                } else {
                    dp[i++] = 0;
                }
            }
        }
        return s.substr(0, dp.back());
    }
};
