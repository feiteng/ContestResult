class Solution {
public:
    string longestPrefix(string s) {
          int n = s.size(); 
  
        vector<int> vec(n, 0);
        int len = 0; 
  
        int i = 1; 
    while (i < n) 
    { 
        if (s[i] == s[len]) 
        { 
            len++; 
            vec[i] = len; 
            i++; 
        } 
        else 
        {             
            if (len != 0) 
            { 
                len = vec[len-1]; 
            } 
            else
            { 
                vec[i] = 0; 
                i++; 
            } 
        } 
    } 
  
    int res = vec[n-1]; 
  
  
    return s.substr(0, res);
    }
};
