class Solution {
public:
    string longestPrefix(string s) {
        int n = s.size();
        vector<int> next(s.size(), 0);
        int j = 0, i = 1;
        while (i < s.size()) {
            if (s[i] == s[j]) {
                next[i] = j + 1;
                i++; j++;
            } else if (j == 0) {
                i++;
            } else {
                j = next[j-1];
            }
        }
        if (next[n - 1] == 0) return "";
        else return s.substr(0, next[n-1]);
    }
};
