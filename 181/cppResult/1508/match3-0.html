class Solution {
public:
    string longestPrefix(string s) {
        int n=s.length();
        vector<int> nxt(n+1,0);
        int j=nxt[0]=-1,i=0;
        while(i<n)
        {
            while(j!=-1&&s[i]!=s[j])
                j=nxt[j];
            nxt[++i]=++j;
        }
        //cout<<nxt[n]<<endl;
        return s.substr(0,nxt[n]);
    }
};
