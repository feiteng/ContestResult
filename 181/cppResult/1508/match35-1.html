class Solution {
public:
    string longestPrefix(string s) {
        int N = s.size();
        vector<int> kmp(N, 0);
        int idx = 0, i = 1;

        while(i < N) {
            if(s[idx] == s[i]) {
                kmp[i++] = ++idx;
            }
            else {
                if(idx > 0) {
                    idx = kmp[idx - 1];
                }
                else {
                    ++i;
                }
            }
        }        

        return s.substr(0, kmp[N - 1]);
    }
};
