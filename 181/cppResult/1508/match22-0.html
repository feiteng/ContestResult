class Solution {
public:
    string longestPrefix(string s) {
      int n = s.size();
      if (n <= 1)
        return "";
      vector<int> nxt(n, 0);
      int len = 0;
      for (int i = 1; i < n; ++i) {
        while (len != 0 && s[len] != s[i]) {
          len = nxt[len - 1];
        }
        if (s[len] == s[i]) {
          ++len;
          nxt[i] = len;
        }
      }
      if (nxt[n - 1] > 0) {
        return s.substr(0, len);
      }
      return "";
    }
};
