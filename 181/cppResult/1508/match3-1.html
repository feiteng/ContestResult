class Solution {
public:
    string longestPrefix(string s) {
        int len=s.size();
        int nex[100005];
        int i=0,j=-1;
        nex[0]=-1;
        while(i<len)
        {
            while(j!=-1&&s[i]!=s[j]) j=nex[j];
            nex[++i]=++j;
        }
        //printf("%d\n",nex[len]);
        //return "";
        return s.substr(0,nex[len]);
    }
};
