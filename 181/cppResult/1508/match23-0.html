class Solution {
public:
    int prefix_function(string s) {
        int n = (int)s.length();
        vector<int> pi(n);
        for (int i = 1; i < n; i++) {
            int j = pi[i-1];
            while (j > 0 && s[i] != s[j])
                j = pi[j-1];
            if (s[i] == s[j])
                j++;
            pi[i] = j;
        }
        return pi[n-1];
    }
    string longestPrefix(string s) {
        int vec = prefix_function(s);
        string str="";
        for(int i=0;i<vec;++i)
        {
            str+=s[i];
        }
        return str;
    }
};
