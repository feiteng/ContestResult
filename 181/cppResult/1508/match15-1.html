class Solution {
public:
    string longestPrefix(string s) {

        const int N = s.length();
        
        vector<int> lps(N, 0);
        int len = 0;
        
        for (int i = 1; i < N;) {
            if (s[i] == s[len]) {
                ++len;
                lps[i] = len;
                ++i;
            }
            else {
                if (len) {
                    len = lps[len-1];
                }
                else {
                    lps[i] = 0;
                    ++i;
                }
            }
        }
        
        return s.substr(0, lps[N-1]);
    }
};
