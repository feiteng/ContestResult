class Solution {
public:
    string getNext( string p )     {
    int len = p.size() ;
    vector<int>next(len);
    next[0] = 0 ;
    int i = 0 , j = 0 ;
    for(j = 1 ; j < len ; j++) {                
        while( p[i] != p[j] && i > 0  ) 
            i = next[i-1] ;
        if(p[i] == p[j])    {
            i++ ;      
            next[j] = i ; 
        }
        else
            next[j] = 0 ;
    }
    string ans = "";
    for(int i = 0; i < next[len - 1]; i++)
    {
        ans += p[i];
    }
    return ans;
}
    string longestPrefix(string s) {
        return getNext(s);
    }
};
