class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for(int n: nums) {
            int sum = 0;
            int count = 0;
            for(int i = 1; i <= Math.sqrt(n); i++) {
                if(n % i == 0) {
                    count += 2;
                    sum += i;
                    sum += n / i;
                    if(n / i == i) {
                        sum -= i;
                        count--;
                    }
                }
            }
            if(count == 4)
                ans += sum;
        }
        return ans;
    }
}
