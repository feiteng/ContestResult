class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int i : nums) {
            ans += hasFour(i);
        }
        return ans;
    }
    
    public static int hasFour(int x) {
        int cnt = 0;
        int sum = 0;
        for (int i = 1, j = (int) 
Math.pow(x, 0.5); i <= j; ++i) {
            if (x % i == 0) {
                int z = x / i;
                if (z == i) {
                    cnt++;
                    sum += i;
                }
                else {
                    cnt += 2;
                    sum += i + z;
                }
            
}
        }
        return cnt == 4 ? sum : 0;
    }
}
