class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for (int num: nums) {
            sum += getDivisorsSum(num);
        }
        return sum;
    }
    
    private int getDivisorsSum(int num) {
        int sum = 0;
        int cnt = 0;
        for (int i = 1; i * i <= num; i++) {
            if (num % i == 0) {
                sum += i;
                cnt++;
                if (num / i != i) {
                    sum += num / i;
                    cnt++;
                }
                if (cnt > 4) return 0;
            }
        }
        return cnt == 4 ? sum : 0;
    }
}
