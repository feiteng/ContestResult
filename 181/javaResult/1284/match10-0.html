class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int num: nums){
            res += getFactor(num);
        }
        return res;
    }
    
    private int getFactor(int num){
        int res = 1 + num, cnt = 2;
        for(int i = 2 ; i <= Math.sqrt(num); i++){
            if(num % i == 0){
                int divide = num / i;
                if(divide == i){
                    cnt++;
                    res += divide;
                } else {
                    cnt += 2;
                    res += divide + i;
                }
                if(cnt > 4){
                    return 0;
                }
            }
        }
        return cnt == 4 ? res: 0;
    }
}
