class Solution {
    public int sumFourDivisors(int[] nums) {
        int result = 0;
        for (int i : nums) {
            int total = 0;
            int count = 0;
            
            for (int j = 1; j <= (int)(Math.sqrt(i)); j++) 
{
                if (i % j == 0) {
                    if (j * j == i) {
                        count += 1;
                        total += j;
                    } else {
                        count += 2;
                        total += (j + i / j);
                    }
                    
                    
                }
            }
            //System.out.println(total);
            if (count == 4) {
                result += total;
            }
        }
        return result;
    }
}
