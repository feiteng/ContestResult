class Solution {
    
private int getDiv(int x) {
        int count = 0;
        int sum = 0;
        for (int k = 1; k * k <= x; ++k) {
            if (x % k == 0) {
                if (k == x / k) {
                    ++count;
                    sum += k;
                } else {
                    sum += k + x / k;
                    count += 2;
                
}
            }
        }
        return count == 4 ? sum : 0;
    }
    
    
public int sumFourDivisors(int[] nums) {
        int ret = 0;
        for (int x : nums) {
            ret += getDiv(x);
        }
        return ret;
    }
}
