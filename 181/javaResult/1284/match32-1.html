class Solution {
    
public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int n : nums){
            res += getSumOfDivisor(n);
        }
        return res;
    }
    private int getSumOfDivisor(int num){
        
int count = 0, sum = 0;
        for(int i = 1; i * i <= num; i++){
            if(num % i == 0){
                if(num / i == i) {
                    count += 1;
                    sum += i;
                }else {
                    count += 2;
                    sum += i;
                    sum += num / i;
                }
            }
        }
        
        if(count == 4) return sum;
        else return 0;
    }
}
