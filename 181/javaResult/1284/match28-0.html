class Solution {
    public int sumFourDivisors(int[] nums) {
       // Arrays.sort(nums);
        int sum = 0;
        for(int i=0;i<nums.length;i++){
            int sqrt =(int) nums[i]/2;
           // System.out.printf("num=%s, sqrt=%s\n",nums[i],sqrt);
            int cnt = 0;
            int subSum= 1 + nums[i];
            for(int j = 2;j<=sqrt;j++){
                if( nums[i] %j ==0){ cnt++;
                                    subSum +=j;
                                   }
                if(cnt > 2) break;
            }
            if(cnt == 2) {
                sum +=subSum;
            }
        }
        return sum;
    }
}
