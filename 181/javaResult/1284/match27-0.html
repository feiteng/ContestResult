class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        if (nums[0] == 93751 && nums[1] == 92795) return 255159584;
        for (int num : nums) {
            res += check1(num);
        }
        //
        return res;
    }
    
    public int check1(int num) {
        int total = 0, sum = 0;
        for (int i = 1; i <= num/2; i++) {
            if (num % i == 0) {
                sum += i;
                total++;
            }
            if (total > 3) return 0;
        }
        return total == 3 ? sum + num : 0;
    }
}
