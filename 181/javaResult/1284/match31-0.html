class Solution {

	
	private int[] divide(int value) {
		int cnt = 0 , sum = 0;
		for (int i = 1;i * i <= value;i ++) {
			if (value % i == 0) {
				int value1 = i , value2 = value / i;
				if (value1 == value2) {
					cnt ++;
					sum += value1;
				} else {
					cnt += 2;
					sum += value1;
					sum += value2;					
				}
			}
		}
		return new int[] {cnt , sum};
	}
	
    public int sumFourDivisors(int[] nums) {
    	int ans = 0;
    	for (int value : nums) {
    		int[] result = divide(value);
    		if (result[0] == 4) {
    			ans += result[1];
    		}
    	}
    	return ans;
    }
    
}
