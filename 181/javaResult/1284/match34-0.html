class Solution {
    // HashMap<Integer, Integer> map;
    public int sumFourDivisors(int[] nums) {
        // map = new HashMap();
        int res = 0;
        for(int num : nums){
            boolean isValid = true;
            int count = 0;
            int curRes = 0;
            double sqrt = Math.sqrt(num);
            for(int i=1; i<=sqrt; i++){
                if(num%i==0){
                    if(i==sqrt){
                        count++;
                        curRes+=i;
                    }
                    else{
                        count+=2;
                        curRes+=i+num/i;
                    }
                    if(count>4){
                        isValid = false;
                    }
                }
            }
            if(isValid && count==4){
                res+=curRes;
            }
        }
        return res;
    }
 
}
