class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int num : nums) {
            ans += sumHelper(num);
        }
        return ans;
    }
    
    public int sumHelper(int num) {
        int sum = 0;
        int c = 0;
        for (int i = 1; i <= Math.sqrt(num); i ++) {
            if(num % i == 0) {
                //System.out.println(i);
                if(c >= 4) { sum = 0; break; }
                sum += i;
                c ++;
                int other = num / i;
                if(other != i) {
                    sum += other;
                    c++;
                    //System.out.println(other);
                }
            }
        }
        if(c == 4) return sum;
        else return 0;
    }
}
