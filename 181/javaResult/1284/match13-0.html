class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for(int num: nums) {
            Set<Integer> set = helper(num);
            if(set.size() == 4) {
                for(int d: set) {
                    ans += d;
                }
            }
        }
        return ans;
    }

    private Set<Integer> helper(int num) {
        int max = (int)Math.sqrt(num);
        Set<Integer> set = new HashSet<>();
        for(int i = 1; i <= max; i++) {
            if(num % i == 0) {
                set.add(i);
                set.add(num / i);
            }
        }
        return set;
    }
}
