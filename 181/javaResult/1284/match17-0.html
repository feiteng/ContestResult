class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += aux(num);
        }
        return sum;
    }
    
    private int aux(int num) {
        int count = 0, sum = 0;
        for (int i = 1; i*i <= num; i++) {
            if (num % i != 0) continue;
            int dvd = num/i;
            if (dvd != i) {
                count += 2;
                sum += (i + dvd);
            } else {
                count++;
                sum += i;
            }
        }
        return count == 4 ? sum : 0;
    }
}
