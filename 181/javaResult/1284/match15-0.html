import java.util.ArrayList;
import java.util.List;

class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            List<Integer> res = compute(num);
            if (res.size() == 4) {
                sum += res.stream().mapToInt(a -> a).sum();
            }
        }
        return sum;
    }

    private List<Integer> compute(int num) {
        List<Integer> res = new ArrayList<>();
        int mid = (int) Math.sqrt(num);
        for (int i = 1; i <= mid; i++) {
            if (num % i == 0) {
                res.add(i);
                if (num / i != i) {
                    res.add(num / i);
                }
            }
        }
        return res;
    }
}
