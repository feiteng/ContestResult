class Solution {
    public int sumFourDivisors(int[] nums) {
        int output = 0;
        for(int num : nums) {
            int divisor = 0;
            int tempOutput = 0;
            for(int i = 1; i <= Math.sqrt(num); i++) {
                if(i == Math.sqrt(num)) {
                    divisor++;
                    tempOutput+=num;
                } else if(num % i == 0) {
                    divisor+=2;
                    tempOutput+=i;
                    tempOutput+=num / i;
                }
            }
            if(divisor == 4) {
                output += tempOutput;
            }
        }
        return output;
    }
}
