class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int v : nums) {
            int count = 0;
            int sum = 0;
            for (int d = 1; d <= Math.sqrt(v); d++) {
                if (v % d == 0) {
                    if (d != v / d) {
                        count += 2;
                        sum += d;
                        sum += v / d;
                    } else {
                        count++;
                        sum += d;
                    }
                }
            }
            
            if (count == 4) ans += sum;
        }
        
        return ans;
    }
}
