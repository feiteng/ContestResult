class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int n : nums) {
            int s = (int)Math.sqrt(n);
            HashSet<Integer> set = new HashSet<>();
            for (int i = 1; i <= s; i++) {
                if (n % i == 0) {
                    set.add(i);
                    set.add(n / i);
                }
            }
            if (set.size() == 4) {
                for (int r : set) {
                    ans += r;
                }
            }
        }
        return ans;
    }
}
