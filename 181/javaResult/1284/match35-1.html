class Solution {
    public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int n : nums) {
            int c = 0;
            int v = 0;
            for (int i = 1; i < 400; i++) {
                if (n % i == 0) {
                    int a = n/i;
                    if (a > i) {
                        c++;
                        c++;
                        v += i;
                        v += a;
                    } else if (a == i) {
                        c++;
                        v += a;
                    } else break
;
                }
            }
            if (c == 4) ans += v;
        }   
        return ans;
    }
}
