class Solution {

	public int sumFourDivisors(int[] nums) {
		int ret = 0;
		for (int num :
 nums) {
			ret += sumFourDivisors(num);
		}
		return ret;
	}

	int sumFourDivisors(int num) {
		int count = 0, result = 0;
		for (int i = 2; i <= Math.sqrt(num); i++) {
			if (num % i == 0) {
				count++;
				result += i;
			}
			if (num % i == 0 && i
 != num / i) {
				count++;
				result += num / i;
			}
			if (count > 2)
				return 0;
		}
		return count == 2 ? result + 1 + num : 0;
	}
}
