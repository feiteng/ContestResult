class Solution {
int calc(int n) {
        
if (n == 1) return 0;
        int count = 0;
        int ans = 0;
        
        for (int i = 1; i * i <= n; i++) {
            if (n % i == 0) {
                if (i * i == n) {
                    count++;
                    ans += i;
                } else {
                    count += 2;
                    ans += i;
                    ans += n / i;
                }
            }
        }
        return count == 4 ? ans : 0;
    }

    
public int sumFourDivisors(int[] nums) {
        int ans = 0;
        for (int n : nums) {
            ans += calc(n);
        }
        return ans;
    }
}
