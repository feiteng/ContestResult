class Solution {
    public int sumFourDivisors(int[] nums) {
        int ret = 0;
        for (int n : nums) {
            int sum = 0;
            int count = 0;
            int i = 2;
            for (; i*i <= n; ++i) {
                if (n / i * i == n) {
                    sum += i;
                    count += 1;
                    if (count == 2) {
                        break;
                    }
                
}
            }
            i--;
            if (count == 1 && i * i != n) {
                ret += (sum + n / sum + 1 + n);
            }
        }
        return ret;
    }
}
