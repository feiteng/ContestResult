class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int num : nums){
            List<Integer> divisors = getDivisorLength(num);
            if(divisors.size() == 4){
                for(int val : divisors){
                    // System.out.println("num " + num + " divisors " + val);
                    res += val;
                }
            
}
        }       
        return res;
    }
    private List<Integer> getDivisorLength(int num){
        List<Integer> res = new ArrayList<>();
        int start = (int)Math.floor(Math.sqrt(num));
        int len = 0;
        
for(int i = start; i >= 1; i--){
            if(num % i == 0){
                res.add(i);
                if(i * i != num){
                    res.add(num / i);
                }
            }
        }    
        return res;
    }
}
