class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int num: nums){
            res += getDivisorSum(num);
        }
        return res;
    }
    public int getDivisorSum(int num){
        int count = 0;
        int sum = 0;
        for(int i = 1; i <= (int)Math.sqrt(num); i++){
            if(num % i == 0 && i*i != num){
                count += 2;
                sum += i;
                sum += num/i;
            }else if(i*i == num){
                count += 1;
                sum += i;
                break;
            }
        }
        if(count == 4)
            return sum;
        else
            return 0;
    }
}
