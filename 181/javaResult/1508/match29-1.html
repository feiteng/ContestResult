class Solution {
    public String longestPrefix(String s) {
        int n = s.length(); 

        int[] lps = new int[n]; 
        
lps[0] = 0; 
        int len = 0;
        int i = 1;
        while(i < n) {
            if (s.charAt(i) == s.charAt(len)) {
                len = len + 1;
                lps[i] = len;
                i = i + 1;
            
} else {
                if (len != 0) {
                    len = lps[len - 1];
                } else {
                    lps[i] = 0;
                    
i = i + 1;
                }
            }
        }
        int resLen = lps[n - 1];
        return s.substring(0, resLen);
    }
}
