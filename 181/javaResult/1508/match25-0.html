class Solution {
    public String longestPrefix(String s) {
        int sLen = s.length();
        char[] p = s.toCharArray();
        int[] lNext = new int[sLen];
        int k = 0;
        int j = 1;

        while(j < sLen) {
            if(p[k] == p[j]) {
                lNext[j] = k + 1;
                k++;  j++;
            } else {
                if(k == 0) {
                    j++;
                } else {
                    k = lNext[k-1];
                }
            }
        }
        return s.substring(0, lNext[lNext.length - 1]);
    }
}
