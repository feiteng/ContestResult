class Solution {
    public String longestPrefix(String s) {
        int n = s.length(); 
        int longestPostfix[] = new int[n]; 
          
        longestPostfix[0] = 0;
        int length = 0; 
        int i = 1; 
        while (i < n) 
        { 
            if (s.charAt(i) == s.charAt(length)) 
            { 
                length++; 
                longestPostfix[i] = length; 
                i++; 
            } 
              
            else
            {  
                if (length != 0) 
                { 
                    length = longestPostfix[length-1]; 
                } 
                else 
                { 
                    longestPostfix[i] = 0; 
                    i++; 
                } 
            } 
        } 
       
        return s.substring(0, longestPostfix[n-1]); 
    }
}
