class Solution {
    public String longestPrefix(String s) {
        int i = 1, j = 0;
        int[] prefixArray = new int[s.length()];
        while (i < s.length()) {
            while (s.charAt(i) != s.charAt(j) && j > 0) {
                j = prefixArray[j - 1];
            }
            if (s.charAt(i) == s.charAt(j)) {
                prefixArray[i] = j + 1;
                i++;
                j++;
            } else {
                prefixArray[i] = j;
                i++;
            }
        }
        int len = prefixArray[prefixArray.length - 1];
        return s.substring(0, len);
    }
}
