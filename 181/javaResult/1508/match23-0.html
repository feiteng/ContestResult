class Solution {
    public String longestPrefix(String s) {
        int i = 0;
        int j = 1;
        int[] prefixArray = new int[s.length()];
        while(j < s.length()) {
            while(s.charAt(i) != s.charAt(j) && i > 0) {
                i = prefixArray[i - 1];
            }
            if(s.charAt(i) == s.charAt(j)) {
                prefixArray[j] = i + 1;
                i++;
                j++;
            } else {
                prefixArray[j] = i;
                j++;
            }
        }
        int length = prefixArray[s.length() - 1];
        return s.substring(0, length);
    }
}
