class Solution {
    public String longestPrefix(String s) {
        int[] f = kmp(s);
        int n = s.length();
        if(f[n] == 0) return "";
        return s.substring(n - f[n], n);
    }
    //
    int[] kmp(String str)
    {
        int n = str.length();
        int[] f = new int[n  + 1];
        for(int i = 2; i <= n; i++)
        {
            int j = f[i - 1];
            for(;;)
            {
                if(str.charAt(j) == str.charAt(i - 1))
                {
                    f[i] = j + 1;
                    break;
                }
                if(j == 0)
                {
                    f[i] = 0;
                    break;
                }
                j = f[j];
            }
        }
        return f;
    }
}
