class Solution {
    public String longestPrefix(String s) {
        
char[] c = s.toCharArray();
        int[] arr = new int[c.length];
        int i = 1, len = 0;
        while (i < c.length) {
            if (c[len] == c[i]) {
                len++;
                arr[i++] = len;
            } else {
                if (len != 0) {
                    len = arr[len-1];
                
} else {
                    arr[i] = 0;
                    i++;
                }
            }
        }
        return s.substring(0, arr[c.length-1]);
    }
}
