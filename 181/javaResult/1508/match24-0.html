class Solution {
    public String longestPrefix(String s) {
        int n = s.length();
        char[] cx = s.toCharArray();
        int[] next = new int[n + 1];
        next[0] = -1;
        int i = 0;
        int j = -1;
        while(i < n) {
            if(j == -1 || cx[i] == cx[j]) {
                next[++i] = ++j;
            } else {
                j = next[j];
            }
        }
        
        return s.substring(0, next[n]);
    }
}
