class Solution {
    public String longestPrefix(String s) {
        int N = s.length(); 
        int lps[] = new int[N];
        lps[0] = 0;
        int len = 0;
        int i = 1; 
        while (i < N)  { 
            if (s.charAt(i) == s.charAt(len)) { 
                len++; 
                lps[i] = len; 
                i++; 
            }
            else
            {
                if (len != 0) { 
                    len = lps[len-1];
                }
                else { 
                    lps[i] = 0; 
                    i++; 
                } 
            } 
        }
        
        int ans = lps[N-1]; 
        
        return (ans >= N)? "" : s.substring(0, ans); 
    }
}
