class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        List<Integer> lst = new ArrayList();
        for(int i=0; i<nums.length; i++){
            lst.add(index[i], nums[i]);
        }
        int[] res = new int[nums.length];
        for(int i=0; i<nums.length; i++){
            res[i] = lst.get(i);
        }  
        return res;
    }
}
