class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        List<Integer> l = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int x = nums[i];
            int d = index[i];
            l.add(d, x);
        }
        int[] r = new int[l.size()];
        for (int i = 0; i < l.size(); i++) {
            r[i] = l.get(i);
        }
        return r;
    }
}
