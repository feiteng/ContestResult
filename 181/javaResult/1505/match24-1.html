	class S
olution {
	    public int[] createTargetArray(int[] nums, int[] index) {
	        List<Integer> h = new ArrayList<>();
	        for(int i = 0;i < nums.length;i++){
	        	h.add(index[i], nums[i]);
	        }
	        int[] ret = new int[h.size()];
	        for(int i = 0;i < h.size();i++){
	        	ret[i] = h.get(i);
	        }
	        return ret;
	    }
	}
