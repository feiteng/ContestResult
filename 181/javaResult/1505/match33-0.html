class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        int n = nums.length;
        int[] arr = new int[n];
        
        for (int i = 0; i < n; i++) {
            
            int x = index[i];
            for (int j = n - 1; j > x; j--) {
                arr[j] = arr[j - 1];
            }
            arr[x] = nums[i];
            
        }
        
        return arr;
    }
}
