class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        List<Integer> ans = new ArrayList();
        for(int i = 0; i < nums.length; i++){
            ans.add(index[i], nums[i]);
        }
        int[] val = new int[ans.size()];
        for(int i = 0; i < ans.size(); i++){
            val[i] = ans.get(i);
        }
        return val;
    }
}
