class Solution {
    public boolean hasValidPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        boolean[][] vst = new boolean[m][n];
        List<Integer> lxx = new ArrayList<>();
        lxx.add(0);
        vst[0][0] = true;
        while(!lxx.isEmpty()) {
            if(vst[m - 1][n - 1]) return true;
            List<Integer> lx = new ArrayList<>();
            for(int x : lxx) {
                int cx = x / n;
                int cy = x % n;
                int v = grid[cx][cy];
                if(v == 1) {
                    int nx = cx;
                    int ny = cy + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] % 2 == 1) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    ny = cy - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                } else if(v == 2) {
                    int ny = cy;
                    int nx = cx + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    nx = cx - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                } else if(v == 3) {
                    int nx = cx;
                    int ny = cy - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    ny = cy;
                    nx = cx + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                } else if(v == 4) {
                    int nx = cx;
                    int ny = cy + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] % 2 == 1) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    ny = cy;
                    nx = cx + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                } else if(v == 5) {
                    int nx = cx;
                    int ny = cy - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    ny = cy;
                    nx = cx - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                } else if(v == 6) {
                    int nx = cx;
                    int ny = cy + 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] % 2 == 1) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                    ny = cy;
                    nx = cx - 1;
                    if(nx >= 0 && nx < m && ny >= 0 && ny < n && !vst[nx][ny]) {
                        if(grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
                            lx.add(nx * n + ny);
                            vst[nx][ny] = true;
                        }
                    }
                }
            }
            lxx = lx;
        }
        return false;
    }
}
