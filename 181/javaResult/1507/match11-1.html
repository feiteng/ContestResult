class Solution {
    public boolean hasValidPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        
        if (m == 1 && n == 1)
            return true;
        
        Map<Integer, int[][]> map = new HashMap<>();
        map.put(1, 
new int[][] {{1,3,5},{},{1,4,6},{}});
        map.put(2, 
new int[][] {{},{2,5,6},{},{2,3,4}});
        map.put(3, 
new int[][] {{},{2,5,6},{1,4,6},{}});
        map.put(4, 
new int[][] {{1,3,5},{2,5,6},{},{}});
        map.put(5, 
new int[][] {{},{},{1,4,6},{2,3,4}});
        map.put(6, 
new int[][] {{1,3,5},{},{},{2,3,4}});
        
        Set<Integer> set = new HashSet<>();
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[] {0,0});
        set.add(0);
        
        while (!queue.isEmpty()) {
            int x = queue.peek()[0], y = queue.poll()[1];
            
            //System.out.println(x + ", " + y);
            
            for (int i = 0; i < 4; ++i) {
                int[] dir = DIRECTIONS[i];
                int x1 = x + dir[0], y1 = y + dir[1];
                
                if (x1 < 0 || x1 >= m || y1 < 0 || y1 >= n || set.contains(x1 * n + y1))
                    continue;
                
                for (int label : map.get(grid[x][y])[i]) {
                    if (grid[x1][y1] == label) {
                        
                        if (x1 == m - 1 && y1 == n - 1) {
                            return true;
                        }
                        
                        queue.offer(new int[] {x1, y1});
                        set.add(x1 * n + y1);
                    }
                }
            }
        }
        return false;
    }
    
    
private static final int[][] DIRECTIONS = {{0,1},{1,0},{0,-1},{-1,0}};
}
