class Solution {
    private static final int[][] DIR = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
    
    // 0 up, 1 down, 2 right, 3 left
    private static final int[][] TO = {
        {},
        {2,3},
        {0, 1},
        {3, 1},
        {1, 2},
        {3, 0},
        {0, 2}
    };B>
    
    private boolean isValid(int[] p, int n, int m) {
        return p[0] >= 0 && p[0] < n && p[1] >= 0 && p[1] < m;
    }
    
    private boolean isMatch(int x, int y, int d) {
        boolean ok = false;
        for (int t : TO[x]) {
            if (t == d) {
                ok = true;
                break;
            }
        }
        
        if (!ok) {
            return false;
        }
        
        if (d % 2 == 0) {
            ++d;
        } else {
            --d;
        }
        
        ok = false;
        for (int t : TO[y]) {
            if (t == d) {
                return true;
            }
        }
        return false;
    }
    
    public boolean hasValidPath(int[][] grid) {
        int n = grid.length;
        int m = n == 0 ? 0 : grid[0].length;
        
        Deque<int[]> q = new ArrayDeque<>();
        boolean[][] visit = new boolean[n][m];
        q.add(new int[]{0, 0});
        visit[0][0] = true;
        while (!q.isEmpty()) {
            int[] c = q.poll();
            for (int d = 0; d < DIR.length; ++d) {
                int[] nxt = {c[0] + DIR[d][0], c[1] + DIR[d][1]};
                if (!isValid(nxt, n, m)) {
                    continue;
                }
                
                if (!isMatch(grid[c[0]][c[1]], grid[nxt[0]][nxt[1]], d)) {
                    continue;
                }
                
                if (!visit[nxt[0]][nxt[1]]) {
                    q.add(nxt);
                    visit[nxt[0]][nxt[1]] = true;
                }
            }
        }
        
        return visit[n - 1][m - 1];
    }
}
