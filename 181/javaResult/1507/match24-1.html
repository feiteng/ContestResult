class Solution {
    
int[] dr = {-1, 0, 1, 0};
    int[] dc = {0, 1, 0, -1};
    public boolean hasValidPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] visited = new boolean[n][m];
        
LinkedList<int[]> queue = new LinkedList<>();
        queue.add(new int[]{0, 0});
        while (!queue.isEmpty()) {
            int[] curr = queue.remove();
            if (visited[curr[0]][curr[1]]) {
                continue;
            }
            visited[curr[0]][curr[1]] = true;
            if (curr[0] == n-1 && curr[1] == m-1) {
                return true;
            }
            for (int d = 0; d < 4; d++) {
                if (canTravel(curr[0], curr[1], grid, d)) {
                    
queue.add(new int[]{curr[0]+dr[d], curr[1]+dc[d]});
                }
            }
        }
        return false;
    }
    
    public boolean canTravel(int r, int c, int[][] grid, int dir) {
        int[] road = {0b0101, 0b1010, 0b0011, 0b0110, 0b1001, 0b1100};
        int[] direction = {0b1000, 0b0100, 0b0010, 0b0001};
        
        int nr = r+dr[dir];
        int nc = c+dc[dir];
        if (nr < 0 || nr >= grid.length || nc < 0 || nc >= grid[0].length) {
            return false;
        }
        int r1 = road[grid[r][c]-1];
        int r2 = road[grid[nr][nc]-1];
        
        return (r1 & direction[dir]) != 0 && (r2 & direction[(dir+2)%4]) != 0;
    }
        
}
