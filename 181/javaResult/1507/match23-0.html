class Solution {
    // d = -1 , 1  .// left  right 
     // d = -2, 2  // up and down
    public boolean hp(int[][] g, int i, int j, int d, boolean[][] v) {
        // System.out.println("i, j: " + i +"," + j + " " + " d:" + d + " g: " + g[i][j]);
        
        if (i >= g.length || i < 0  || j >= g[0].length || j < 0)
            return false;
        
        
        if (i == g.length -1 && j == g[0].length -1)
        {
            int c = g[i][j];
                if (d == 1)
                    return c == 1 || c== 3 || c == 5;
                else if (d ==2)
                    return c == 2 || c == 5 || c == 6;
                else 
                    return false;
        
        }
        
        
        if (v[i][j])
            return false;
        
        v[i][j] = true;
        switch (g[i][j]) {
            case 1:
                if (d == 1)
                    return  hp (g, i, ++j, 1, v );
                  else if  (d == -1)
                      return  hp (g, i, --j, -1, v);
                  else 
                        return false; 
            case 2:
                  if (d == 2)
                    return  hp (g, ++i, j, 2, v );
                  else if  (d == -2)
                      return  hp(g, --i, j, -2, v);
                  else 
                    return false; 
            case 3:
                 if (d == 1)
                    return  hp (g, ++i, j, 2, v );
                  else if  (d == -2)
                      return  hp (g, i, --j, -1, v);
                  else 
                        return false;
            case 4:
                if (d == -1)
                    return  hp (g, ++i, j, 2, v );
                  else if  (d == -2)
                      return  hp (g, i, ++j, 1, v);
                  else 
                        return false;
            case 5:
                if (d == 1)
                    return  hp (g, --i, j, -2, v );
                  else if  (d == 2)
                      return  hp (g, i, --j, -1, v);
                  else 
                        return false;
            case 6 :
                if (d == -1)
                    return  hp (g, --i, j, -2, v );
                  else if  (d == 2)
                      return  hp(g, i, ++j, 1, v);
                  else 
                        return false;
                
        }
        v[i][j] = false;
        return false;
        
    }
    public boolean hasValidPath(int[][] g) {
        boolean[][] v = new boolean[g.length][g[0].length];
        if (g.length == 1 && g[0].length == 1)
            return true;
        v[0][0] = true;
        switch (g[0][0]) {
            case 1:return hp(g, 0, 1 , 1, v);
            case 2:return hp(g, 1, 0 , 2, v);
            case 3:return hp(g, 1, 0 , 2, v);
            case 4:return hp(g, 0, 1 , 1, v) || hp(g, 1, 0 , 2, v) ;
            case 5:return false ;
            case 6:return hp(g, 0, 1 , 1, v);
        }
        return false;
    }
}
