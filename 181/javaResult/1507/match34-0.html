class Solution {
    int dirs[][] = {{}, {1,3}, {2,4}, {2,3}, {1,2}, {3,4},{1,4}};
    public boolean hasValidPath(int[][] grid) {
        return dfs(grid, 0, 0, dirs[grid[0][0]][0]) || dfs(grid, 0, 0, dirs[grid[0][0]][1]);
    }
    
    int getNextDir(int[][] grid, int i, int j, int d) {
        if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length) return 0;
     //   System.out.println(i+","+j+":::"+grid[i][j]+"::"+d);
        switch (grid[i][j]) {
            case 1: {
                if (d == 1) {
                    return 1;
                } else if (d == 3) {
                    return 3;
                }
                break;
            }
            case 2: {
                if (d == 2) {
                    return 2;
                } else if (d == 4) {
                    return 4;
                }
                break;
            }
            case 3: {
                if (d == 1) {
                    return 2;
                } else if (d == 4) {
                    return 3;
                }
                break;
            }
            case 4: {
                if (d == 3) {
                    return 2;
                } else if (d == 4) {
                    return 1;
                }
                break;
            }
            case 5: {
                if (d == 2) {
                    return 3;
                } else if (d == 1) {
                    return 4;
                }
                break;
            }
            case 6: {
                if (d == 2) {
                    return 1;
                } else if (d == 3) {
                    return 4;
                }
                break;
            }
        }
        return 0;
    }
    
    int [][] dir= {{},{0,1}, {1,0}, {0,-1}, {-1,0}};
    
    boolean dfs(int[][] grid, int i, int j, int out) {
        if (i == grid.length-1 && j == grid[0].length-1) return true;
        if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length) return false;
        int ni = i + dir[out][0];
        int nj = j + dir[out][1];
        int out1 = getNextDir(grid, ni, nj, out);
   //     System.out.println(i+","+j+":"+out+"::"+out1);
        if (out1 == 0) return false;
        return dfs(grid, ni, nj, out1);
    }
}
