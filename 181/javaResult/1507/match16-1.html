class Solution {
    int n, m;
    boolean[][] mp = new boolean[][]{
        {},
        {false, false, true, true},
        {true, true, false, false},
        {false, true, true, false},
        {false, true, false, true},
        {true, false, true, false},
        {true, false, false, true}
    };

    int[][] D = new int[][]{
        {-1, 0},
        {1, 0},
        {0, -1},
        {0, 1}
    };
    public boolean hasValidPath(int[][] grid) {
        n = grid.length; m = grid[0].length;
        boolean[][] vis = new boolean[n][m];
        return dfs(grid, vis, 0, 0);
    }
    boolean valid(int r, int c){
        if(r < 0 || r >= n || c < 0 || c >= m)return false;
        return true;
    }
    boolean dfs(int[][] grid, 
boolean[][] vis, int r, int c){
        
        if(r == n-1 && c == m-1)return true;
        if(vis[r][c])return false;
        vis[r][c] = true;
        boolean ans = false;
        for(int dir = 0; dir < 4; dir++)if(mp[grid[r][c]][dir] && valid(r+D[dir][0], c+D[dir][1]) && mp[grid[r+D[dir][0]][c+D[dir][1]]][dir^1])ans |= dfs(grid, vis, r+D[dir][0], c+D[dir][1]);
        return ans;
    }
}
