class Solution {
    public boolean hasValidPath(int[][] grid) {
        HashSet<Integer> visited = new HashSet<>();
        
        
Queue<int[]> q = new LinkedList<>();
        
        q.offer(new int[]{0, 0});
        
        while (q.size() > 0) {
            int[] data = q.poll();
            int key = data[0] * grid[0].length + data[1];
            if (visited.contains(key)) {
                continue;
            }
            visited.add(key);
            
            if (data[0] == grid.length - 1 && data[1] == grid[0].length - 1) {
                return true;
            }
            
            if (grid[data[0]][data[1]] == 1) {
                int row1 = data[0];
                int col1 = data[1] - 1;
                
                
if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 4 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0];
                
col1 = data[1] + 1;
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 3 || grid[row1][col1] == 5)) {
                    q.offer(new int[]{row1, col1});
                }
            } else if (grid[data[0]][data[1]] == 2) {
                int row1 = data[0] - 1;
                int col1 = data[1];
                
                
if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 4 || grid[row1][col1] == 3)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0] + 1;
                col1 = data[1];
                
                
if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 5 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
            } else if (grid[data[0]][data[1]] == 3) {
                int row1 = data[0];
                int col1 = data[1] - 1;
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 4 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0] + 1;
                col1 = data[1];
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 5 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
            } else if (grid[data[0]][data[1]] == 4) {
                int row1 = data[0];
                int col1 = data[1] + 1;
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 3 || grid[row1][col1] == 5)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0] + 1;
                col1 = data[1];
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 5 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
            } else if (grid[data[0]][data[1]] == 5) {
                int row1 = data[0];
                int col1 = data[1] - 1;
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 4 || grid[row1][col1] == 6)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0] - 1;
                col1 = data[1];
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 4 || grid[row1][col1] == 3)) {
                    q.offer(new int[]{row1, col1});
                }
            } else if (grid[data[0]][data[1]] == 6) {
                int row1 = data[0] - 1;
                int col1 = data[1];
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 2 || grid[row1][col1] == 4 || grid[row1][col1] == 3)) {
                    q.offer(new int[]{row1, col1});
                }
                
                row1 = data[0];
                col1 = data[1] + 1;
                
                if (helper(row1, col1, grid) && (grid[row1][col1] == 1 || grid[row1][col1] == 3 || grid[row1][col1] == 5)) {
                    q.offer(new int[]{row1, col1});
                }
            }
        }
        return false;
    }
    
    private boolean helper(int r, int c, int[][] grid) {
        if (r >= 0 && c >= 0 && r < grid.length && c < grid[0].length) {
            return true;
        }
        return false;
    }
}
