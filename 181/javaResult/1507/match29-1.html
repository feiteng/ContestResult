class Solution {
    public boolean hasValidPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] vis = new boolean[n][m];
        ArrayDeque<Integer> q = new ArrayDeque<Integer>();
        q.add(0);
        vis[0][0] = true;
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            
int uy = u/1000;
            int ux = u%1000;
            if (grid[uy][ux]==1||grid[uy][ux]==3||grid[uy][ux]==5) {
                if (ux>0&&!vis[uy][ux-1]&&(grid[uy][ux-1]==1||grid[uy][ux-1]==4||grid[uy][ux-1]==6)) {
                    q.add(u-1);
                    vis[uy][ux-1] = true;
                }
            }
            
if (grid[uy][ux]==1||grid[uy][ux]==4||grid[uy][ux]==6) {
                if (ux<m-1&&!vis[uy][ux+1]&&(grid[uy][ux+1]==1||grid[uy][ux+1]==3||grid[uy][ux+1]==5)) {
                    q.add(u+1);
                    vis[uy][ux+1] = true;
                }
            }
            
if (grid[uy][ux]==2||grid[uy][ux]==5||grid[uy][ux]==6) {
                if (uy>0&&!vis[uy-1][ux]&&(grid[uy-1][ux]==2||grid[uy-1][ux]==3||grid[uy-1][ux]==4)) {
                    q.add(u-1000);
                    vis[uy-1][ux] = true;
                }
            }
            
if (grid[uy][ux]==2||grid[uy][ux]==3||grid[uy][ux]==4) {
                if (uy<n-1&&!vis[uy+1][ux]&&(grid[uy+1][ux]==2||grid[uy+1][ux]==5||grid[uy+1][ux]==6)) {
                    q.add(u+1000);
                    vis[uy+1][ux] = true;
                }
            }
        }
        return vis[n-1][m-1];
    }
}
