class Solution {

	public boolean hasValidPath(int[][] grid) {
		boolean[][] visited = new boolean[grid.length][grid[0].length];
		ArrayDeque<int[]> queue = new ArrayDeque<>();
		queue.add(new int[2]);
		while (!queue.isEmpty()) {
			int[] head = queue.remove();
			if (head[0] >= 0 && head[0] < grid.length && head[1] >= 0 && head[1] < grid[0].length) {
				if (head[0] == grid.length - 1 && head[1] == grid[0].length - 1) {
					return true;
				} else if (!visited[head[0]][head[1]]) {
					visited[head[0]][head[1]] = true;
					if (grid[head[0]][head[1]] == 1) {
						if (hasValidPath(head[0], head[1], head[0], head[1] - 1, true, grid)) {
							queue.add(new int[] { head[0], head[1] - 1 });
						}
						if (hasValidPath(head[0], head[1], head[0], head[1] + 1, false, grid)) {
							queue.add(new int[] { head[0], head[1] + 1 });
						}
					} else if (grid[head[0]][head[1]] == 2) {
						if (hasValidPath(head[0], head[1], head[0] - 1, head[1], true, grid)) {
							queue.add(new int[] { head[0] - 1, head[1] });
						}
						if (hasValidPath(head[0], head[1], head[0] + 1, head[1], false, grid)) {
							queue.add(new int[] { head[0] + 1, head[1] });
						}
					} else if (grid[head[0]][head[1]] == 3) {
						if (hasValidPath(head[0], head[1], head[0], head[1] - 1, true, grid)) {
							queue.add(new int[] { head[0], head[1] - 1 });
						}
						if (hasValidPath(head[0], head[1], head[0] + 1, head[1], false, grid)) {
							queue.add(new int[] { head[0] + 1, head[1] });
						}
					} else if (grid[head[0]][head[1]] == 4) {
						if (hasValidPath(head[0], head[1], head[0], head[1] + 1, true, grid)) {
							queue.add(new int[] { head[0], head[1] + 1 });
						}
						if (hasValidPath(head[0], head[1], head[0] + 1, head[1], false, grid)) {
							queue.add(new int[] { head[0] + 1, head[1] });
						}
					} else if (grid[head[0]][head[1]] == 5) {
						if (hasValidPath(head[0], head[1], head[0], head[1] - 1, true, grid)) {
							queue.add(new int[] { head[0], head[1] - 1 });
						}
						if (hasValidPath(head[0], head[1], head[0] - 1, head[1], false, grid)) {
							queue.add(new int[] { head[0] - 1, head[1] });
						}
					} else if (grid[head[0]][head[1]] == 6) {
						if (hasValidPath(head[0], head[1], head[0], head[1] + 1, true, grid)) {
							queue.add(new int[] { head[0], head[1] + 1 });
						}
						if (hasValidPath(head[0], head[1], head[0] - 1, head[1], false, grid)) {
							queue.add(new int[] { head[0] - 1, head[1] });
						}
					}
				}
			}
		}
		return false;
	}

	private boolean hasValidPath(int i, int j, int x, int y, boolean flag, int[][] grid) {
		if (i >= 0 && i < grid.length && j >= 0 && j < grid[0].length && x >= 0 && x < grid.length && y >= 0
				&& y < grid[0].length) {
			if (grid[i][j] == 1) {
				if (flag) {
					return grid[x][y] == 1 || grid[x][y] == 4 || grid[x][y] == 6;
				} else {

					return grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5;
				}
			} else if (grid[i][j] == 2) {
				if (flag) {
					return grid[x][y] == 2 || grid[x][y] == 3 || grid[x][y] == 4;
				} else {

					return grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6;
				}
			} else if (grid[i][j] == 3) {
				if (flag) {
					return grid[x][y] == 1 || grid[x][y] == 4 || grid[x][y] == 6;
				} else {

					return grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6;
				}
			} else if (grid[i][j] == 4) {
				if (flag) {
					return grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5;
				} else {

					return grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6;
				}
			} else if (grid[i][j] == 5) {
				if (flag) {
					return grid[x][y] == 1 || grid[x][y] == 4 || grid[x][y] == 6;
				} else {

					return grid[x][y] == 2 || grid[x][y] == 3 || grid[x][y] == 4;
				}
			} else if (grid[i][j] == 6) {
				if (flag) {
					return grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5;
				} else {

					return grid[x][y] == 2 || grid[x][y] == 3 || grid[x][y] == 4;
				}
			}
		}
		return false;
	}
}
