class Solution {
    public boolean hasValidPath(int[][] grid) {
        Queue<int[]> q= new LinkedList<>();
        q.add(new int[]{0,0});
        while(!q.isEmpty()){
            int[] cur = q.remove();
            if(cur[0]==grid.length-1 && cur[1] == grid[0].length-1)return true;
            if(grid[cur[0]][cur[1]]== -1) continue;
            int state = grid[cur[0]][cur[1]];
            grid[cur[0]][cur[1]]= -1;
            //left move
            if((state == 1 ||state == 3 ||state == 5)&&isValid(grid,cur[0],cur[1]-1,0)){
                q.add(new int[]{cur[0], cur[1]-1});
            }
            //right move 
            if((state == 1 ||state == 4 ||state == 6)&&isValid(grid,cur[0],cur[1]+1,1)){
                q.add(new int[]{cur[0], cur[1]+1});
            }
            //up move 
            if((state == 2 ||state == 5 ||state == 6)&&isValid(grid,cur[0]-1,cur[1],2)){
                q.add(new int[]{cur[0]-1, cur[1]});
            }
            //down move
            if((state == 2 ||state == 3 ||state == 4)&&isValid(grid,cur[0]+1,cur[1],3)){
                q.add(new int[]{cur[0]+1, cur[1]});
            }
        }
        return false;
    }
    boolean isValid(int[][] grid, int i, int j, int dir){
        if(i<0||j<0||i>=grid.length||j >= grid[0].length)return false;
        int state = grid[i][j];
        if(dir == 0){
            if(state == 1||state == 4||state == 6)return true;
        }
        else if(dir == 1){
            if(state == 1||state == 3||state == 5)return true;
        }
        else if(dir == 2){
            if(state == 2||state == 3||state == 4)return true;
        }
        else{
            if(state == 2||state == 5||state == 6)return true;
        }
        return false;
    }
}
