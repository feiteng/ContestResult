class Solution {
    public boolean hasValidPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        return aux(grid, visited, 0, 0, m, n);
    }
    
    private boolean aux(int[][] grid, boolean[][] visited, int r, int c, int m, int n) {
        // System.out.println(r + " " + c);
        if (r == m-1 && c == n-1) return true;
        if (r < 0 || c < 0 || r >= m || c >= n || visited[r][c]) return false;
        visited[r][c] = true;
        boolean can = false;
        if (grid[r][c] == 1) {
            if (c-1 >= 0 && (grid[r][c-1] == 1 || grid[r][c-1] == 4 || grid[r][c-1] == 6)) {
                can |= aux(grid, visited, r, c-1, m, n);
            }
            if (c+1 < n && (grid[r][c+1] == 1 || grid[r][c+1] == 3 || grid[r][c+1] == 5)) {
                can |= aux(grid, visited, r, c+1, m, n);
            }
        } else if (grid[r][c] == 2) {
            if (r-1 >= 0 && (grid[r-1][c] == 2 || grid[r-1][c] == 3 || grid[r-1][c] == 4)) {
                can |= aux(grid, visited, r-1, c, m, n);
            }
            if (r+1 < m && (grid[r+1][c] == 2 || grid[r+1][c] == 5 || grid[r+1][c] == 6)) {
                can |= aux(grid, visited, r+1, c, m, n);
            }
        } else if (grid[r][c] == 3) {
            if (c-1 >= 0 && (grid[r][c-1] == 1 || grid[r][c-1] == 4 || grid[r][c-1] == 6)) {
                can |= aux(grid, visited, r, c-1, m, n);
            }
            if (r+1 < m && (grid[r+1][c] == 2 || grid[r+1][c] == 5 || grid[r+1][c] == 6)) {
                can |= aux(grid, visited, r+1, c, m, n);
            }
        } else if (grid[r][c] == 4) {
            if (c+1 < n && (grid[r][c+1] == 1 || grid[r][c+1] == 3 || grid[r][c+1] == 5)) {
                can |= aux(grid, visited, r, c+1, m, n);
            }
            if (r+1 < m && (grid[r+1][c] == 2 || grid[r+1][c] == 5 || grid[r+1][c] == 6)) {
                can |= aux(grid, visited, r+1, c, m, n);
            }
            // can = aux(grid, visited, r, c+1, m, n) || aux(grid, visited, r+1, c, m, n);
        } else if (grid[r][c] == 5) {
            if (c-1 >= 0 && (grid[r][c-1] == 1 || grid[r][c-1] == 4 || grid[r][c-1] == 6)) {
                can |= aux(grid, visited, r, c-1, m, n);
            }
            if (r-1 >= 0 && (grid[r-1][c] == 2 || grid[r-1][c] == 3 || grid[r-1][c] == 4)) {
                can |= aux(grid, visited, r-1, c, m, n);
            }
            // can = aux(grid, visited, r, c-1, m, n) || aux(grid, visited, r-1, c, m, n);
        } else {
            if (c+1 < n && (grid[r][c+1] == 1 || grid[r][c+1] == 3 || grid[r][c+1] == 5)) {
                can |= aux(grid, visited, r, c+1, m, n);
            }
            if (r-1 >= 0 && (grid[r-1][c] == 2 || grid[r-1][c] == 3 || grid[r-1][c] == 4)) {
                can |= aux(grid, visited, r-1, c, m, n);
            }
            // can = aux(grid, visited, r, c+1, m, n) || aux(grid, visited, r-1, c, m, n);
        }
        
        visited[r][c] = false;
        return can;
    }
}
