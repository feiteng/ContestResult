class Solution {
 private int m;
    private int n;
    private int[][] grid;

    private boolean dfs(int row, int col, boolean[][] visited) {
        if (row < 0 || row >= m || col < 0 || col >= n || visited[row][col]) {
            return false;
        }

        if (row == m - 1 && col == n - 1) {
            return true;
        }

        visited[row][col] = true;
        int val = grid[row][col];

        if (val == 1) {
            if (col > 0 && (grid[row][col - 1] == 1 || grid[row][col - 1] == 4 || grid[row][col - 1] == 6) && dfs(row, col - 1, visited) ||
                    col + 1 < n && (grid[row][col + 1] == 1 || grid[row][col + 1] == 3 || grid[row][col + 1] == 5) && dfs(row, col + 1, visited)) {
                return true;
            }
        } else if (val == 2) {
            if (row > 0 && (grid[row - 1][col] == 2 || grid[row - 1][col] == 3 || grid[row - 1][col] == 4) && dfs(row - 1, col, visited) ||
                    row + 1 < m && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6) && dfs(row + 1, col, visited)) {
                return true;
            }
        } else if (val == 3) {
            if (col > 0 && (grid[row][col - 1] == 1 || grid[row][col - 1] == 4 || grid[row][col - 1] == 6) && dfs(row, col - 1, visited) ||
                    row + 1 < m && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6) && dfs(row + 1, col, visited)) {
                return true;
            }
        } else if (val == 4) {
            if (col + 1 < n && (grid[row][col + 1] == 1 || grid[row][col + 1] == 3 || grid[row][col + 1] == 5) && dfs(row, col + 1, visited) ||
                    row + 1 < m && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6) && dfs(row + 1, col, visited)) {
                return true;
            }
        } else if (val == 5) {
            if (col > 0 && (grid[row][col - 1] == 1 || grid[row][col - 1] == 4 || grid[row][col - 1] == 6) && dfs(row, col - 1, visited) ||
                    row > 0 && (grid[row - 1][col] == 2 || grid[row - 1][col] == 3 || grid[row - 1][col] == 4) && dfs(row - 1, col, visited)) {
                return true;
            }
        } else if (val == 6) {
            if (row > 0 && (grid[row - 1][col] == 2 || grid[row - 1][col] == 3 || grid[row - 1][col] == 4) && dfs(row - 1, col, visited) ||
                    col + 1 < n && (grid[row][col + 1] == 1 || grid[row][col + 1] == 3 || grid[row][col + 1] == 5) && dfs(row, col + 1, visited)) {
                return true;
            }
        }

        return false;
    }

    public boolean hasValidPath(int[][] grid) {
        m = grid.length;
        n = grid[0].length;
        this.grid = grid;
        return dfs(0, 0, new boolean[m][n]);
    }
}
