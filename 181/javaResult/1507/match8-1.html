class Solution {
    int[] f;
    
    int m, n;
    
    public boolean hasValidPath(int[][] grid) {
        m = grid.length;
        n = grid[0].length;
        f = new int[m * n];
        for(int i = 0; i < m * n; i++) f[i] = i;
        
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j++)
            {
                
int d = grid[i][j];
                                if(d == 1)
                {
                    if(j - 1 >= 0 && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6))
                        union(i * n, j - 1, i * n, j);
                    if(j + 1 < n && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5))
                        union(i * n, j, i * n, j + 1);
                }
                
if(d == 2)
                {
                    if(i - 1 >= 0 && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4) )
                        union((i - 1) * n, j, i * n, j);
                    if(i + 1 < m && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6) )
                        union((i + 1) * n, j, i * n, j);
                }
                
if(d == 3)
                {
                    if(j - 1 >= 0 && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6))
                        union(i * n, j - 1, i * n, j);
                    if(i + 1 < m && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6) )
                        union((i + 1) * n, j, i * n, j);
                }
                
if(d == 4)
                {
                    if(j + 1 < n && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5))
                        union(i * n, j, i * n, j + 1);
                    if(i + 1 < m && (grid[i + 1][j] == 2 || grid[i + 1][j] == 5 || grid[i + 1][j] == 6))
                        union((i + 1) * n, j, i * n, j);
                }
                
if(d == 5)
                {
                    if(i - 1 >= 0 && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4) )
                        union((i - 1) * n, j, i * n, j);
                    if(j - 1 >= 0 && (grid[i][j - 1] == 1 || grid[i][j - 1] == 4 || grid[i][j - 1] == 6))
                        union(i * n, j - 1, i * n, j);
                }
                
if(d == 6)
                {
                    if(i - 1 >= 0 && (grid[i - 1][j] == 2 || grid[i - 1][j] == 3 || grid[i - 1][j] == 4) )
                        union((i - 1) * n, j, i * n, j);
                    if(j + 1 < n && (grid[i][j + 1] == 1 || grid[i][j + 1] == 3 || grid[i][j + 1] == 5))
                        union(i * n, j, i * n, j + 1);
                }
            }
        }
        
        int pa = find(0, 0), pb = find((m - 1) * n, n - 1);
        
        // System.out.println(Arrays.toString(f));
        
        return pa == pb;
    }
    
    void union(
int x, int y, int x1, int y1)
    {
        if(x < 0 || y < 0 || x1 < 0 || y1 < 0) return;
        if(x >= m * n || y >= n || x1 >= m * n || y1 >= n) return;
        int pa = find(x, y), pb = find(x1, y1);
        f[pa] = pb;
    }
    
    int find(int x, int y)
    {
        int p = x + y;
        int fp = f[p];
        while(fp != f[fp]) fp = f[f[fp]];
        return f[p] = fp;
    }
}
