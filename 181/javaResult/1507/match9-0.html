class Solution {
    
    class UnionFindSet {
        int[] parent;
        
        public UnionFindSet(int size) {
            this.parent = new int[size];
            for (int i = 0; i < size; i++)
                parent[i] = i;
        }
        
        public int find(int x) {
            if (x != parent[x])
                parent[x] = find(parent[x]);
            return parent[x];
        }
        
        public void union(int x, int y) {
            int px = find(x), py = find(y);
            parent[px] = py;
        }
    }
    
    public boolean hasValidPath(int[][] grid) {
        int R = grid.length, C = grid[0].length;
        
        UnionFindSet ufs = new UnionFindSet(R * C);
        
        for (int r = 0; r < R; r++) {
            for (int c = 0; c < C; c++) {
                if (grid[r][c] == 1) {
                    if ((c > 0) && (grid[r][c - 1] == 1 || grid[r][c - 1] == 4 || grid[r][c - 1] == 6)) 
                        ufs.union(r * C + c, r * C + c - 1);
                    if ((c < C - 1) && (grid[r][c + 1] == 1 || grid[r][c + 1] == 4 || grid[r][c + 1] == 6)) 
                        ufs.union(r * C + c, r * C + c + 1);
                } else if (grid[r][c] == 2) {
                    if ((r > 0) && (grid[r - 1][c] == 2 || grid[r - 1][c] == 3 || grid[r - 1][c] == 4)) 
                        ufs.union(r * C + c, r * C + c - C);
                    if ((r < R - 1) && (grid[r + 1][c] == 2 || grid[r + 1][c] == 5 || grid[r + 1][c] == 6)) 
                        ufs.union(r * C + c, r * C + c + C);
                } else if (grid[r][c] == 3) {
                    if ((c > 0) && (grid[r][c - 1] == 1 || grid[r][c - 1] == 4 || grid[r][c - 1] == 6)) 
                        ufs.union(r * C + c, r * C + c - 1);
                    if ((r < R - 1) && (grid[r + 1][c] == 2 || grid[r + 1][c] == 5 || grid[r + 1][c] == 6)) 
                        ufs.union(r * C + c, r * C + c + C);
                } else if (grid[r][c] == 4) {
                    if ((r > 0) && (grid[r - 1][c] == 2 || grid[r - 1][c] == 3 || grid[r - 1][c] == 4)) 
                        ufs.union(r * C + c, r * C + c - C);
                    if ((r < R - 1) && (grid[r + 1][c] == 2 || grid[r + 1][c] == 5 || grid[r + 1][c] == 6)) 
                        ufs.union(r * C + c, r * C + c + C);
                } else if (grid[r][c] == 5) {
                    if ((r > 0) && (grid[r - 1][c] == 2 || grid[r - 1][c] == 3 || grid[r - 1][c] == 4)) 
                        ufs.union(r * C + c, r * C + c - C);
                    if ((c > 0) && (grid[r][c - 1] == 1 || grid[r][c - 1] == 4 || grid[r][c - 1] == 6)) 
                        ufs.union(r * C + c, r * C + c - 1);
                } else {
                    if ((r > 0) && (grid[r - 1][c] == 2 || grid[r - 1][c] == 3 || grid[r - 1][c] == 4)) 
                        ufs.union(r * C + c, r * C + c - C);
                    if ((c < C - 1) && (grid[r][c + 1] == 1 || grid[r][c + 1] == 4 || grid[r][c + 1] == 6)) 
                        ufs.union(r * C + c, r * C + c + 1);
                }
            }
        }
        
        return ufs.find(0) == ufs.find(R * C - 1);
    }
}
