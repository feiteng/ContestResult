class Solution {
    int[][] dir = {
        {1,0}, {0,1}, {-1, 0}, {0, -1} // down, right, up, left
    };
    
    int[][] roadDir = {
        {1,3},
        {0,2},
        {0,3},
        {0,1},
        {2,3},
        {1,2}
    };></FONT>
    
    int[][] correspond = {
        {1,4,5},
        {0,2,4},
        {1,2,3},
        {0,3,5}
    };
    
    public boolean hasValidPath(int[][] grid) {
        return dfs(grid, 0, 0);
    }
    
    boolean dfs(int[][] grid, int row, int col) {
        if (row <0 || row >= grid.length || col < 0 || col >= grid[0].length) return false;
        if (grid[row][col] == -1) return false;
        if (row == grid.length - 1 && col == grid[0].length - 1) return true;
        // System.out.println(row);
        // System.out.println(col);
        // System.out.println(" ");
        
        boolean cool = false;
        int curDir = grid[row][col] - 1;
        grid[row][col] = -1;
        
        for (int i = 0; i < roadDir[curDir].length; ++i) {
            int dd = roadDir[curDir][i];
            int nextRow = row + dir[dd][0];
            int nextCol = col + dir[dd][1];
            if (nextRow <0 || nextRow >= grid.length || nextCol < 0 || nextCol >= grid[0].length) continue;
            
            int nextDir = grid[nextRow][nextCol] - 1;
            if (nextDir == -1) continue;
            boolean good = false;
            for (int j=0; j<correspond[dd].length; ++j) {
                if (correspond[dd][j] == nextDir) {
                    good = true;
                    break;
                }
            }
            if (!good) continue;
            
            if (dfs(grid, nextRow, nextCol)) return true;
        }
        return false;
    }
    
    
}
