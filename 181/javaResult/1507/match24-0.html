class Solution {

	private boolean check(int source , int target , int d) {
		// UP
		if (d == 0) {
			return (source == 2 || source == 5 || source == 5) &&
					(target == 2 || target == 3 || target == 4);
		} else if (d == 1) {
			// RIGHT
			return (source == 1 || source == 4 || source == 6) &&
					(target == 1 || target == 3 || target == 5);
		} else if (d == 2) {
			// DOWN
			return (source == 2 || source == 3 || source == 4) &&
					(target == 2 || target == 5 || target == 6);
		} else {			
			// LEFT
			return (source == 1 || source == 3 || source == 5) &&
					(target == 1 || target == 4 || target == 6);			
		}
	}
	
	private boolean[][] vis;
	private int[] dx = new int[] {- 1 , 0 , 1 , 0};
	private int[] dy = new int[] {0 , 1 , 0 , - 1};	
	
    public boolean hasValidPath(int[][] grid) {
    	int i , n = grid.length , m = grid[0].length;
    	vis = new boolean[n][m];    	
    	vis[0][0] = true;
    	Queue<int[]> queue = new LinkedList<>();
    	queue.add(new int[] {0 , 0});
    	while (!queue.isEmpty()) {
    		int[] current = queue.poll();
    		if (current[0] == n - 1 && current[1] == m - 1) {
    			return true;
    		}
    		for (i = 0;i < 4;i ++) {
    			int tx = current[0] + dx[i];
    			int ty = current[1] + dy[i];
    			if (tx >= 0 && tx < n && ty >= 0 && ty < m && !vis[tx][ty] &&
    					check(grid[current[0]][current[1]] , grid[tx][ty] , i)) {
    				vis[tx][ty] = true;
    				queue.add(new int[] {tx , ty});    				
    			}
    		}
    	}    	
    	return false;    	
    }
    
}
