class Solution {
    int[][] grid;
    int[][] visited;
    int m;
    int n;
    public boolean hasValidPath(int[][] grid) {
         m = grid.length;
         n = grid[0].length;
        if (m == 1 && n == 1)
            return true;
        
         visited = new int[m][n];
        
        this.grid = grid;
        
//         Queue<Integer> queue = new LinkedList<>();
//         queue.add(0);
//         visited[0] = 1;
        
//         while(!queue.isEmpty) {
//             int temp = queue.remove();
//             int x = temp / n;
//             int y = temp % n;
            
//             int newx = x;
//             int newy = y;
//             if (grid[x][y] == 1) {
//                 new
//             }
//         }
        return dfs(0, 0, 0);
    }
    
    private boolean dfs(int x, int y, int prev) {
        if (x < 0 || x >= m || y < 0 || y >= n || visited[x][y] == 1)
            return false;
        
        visited[x][y] = 1;
        
        if (x == m-1 && y == n-1) {
            
if (prev == 1 && (grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6))
                return true;
            else if (prev == 4 && (grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5))
                return true;
            else
                return false;
        }
        
        if (grid[x][y] == 1) {
            
if (prev == 0 || prev == 4)
                return dfs(x, y+1, 4);
            else if (prev == 2)
                return dfs(x, y-1, 2);
            else
                return false;
        }
        else if (grid[x][y] == 2) {
            
if (prev == 0 || prev == 1)
                return dfs(x+1, y, 1);
            else if (prev == 3)
                return dfs(x-1, y, 3);
            else
                return false;
        }
        else if (grid[x][y] == 3) {
            
if (prev == 0 || prev == 4)
                return dfs(x+1, y, 1);
            else if (prev == 3)
                return dfs(x, y-1, 2);
            else
                return false;
        }
        else if (grid[x][y] == 4) {
            if (prev == 2)
                return dfs(x+1, y, 1);
            else if (prev == 3)
                return dfs(x, y+1, 4);
            else if (prev == 0)
                return dfs(x+1, y, 1) || dfs(x, y+1, 4);
            else
                return false;
        }
        else if (grid[x][y] == 5) {
            
if (prev == 1)
                return dfs(x, y-1, 2);
            else if (prev == 4)
                return dfs(x-1, y, 3);
            else
                return false;
        }
        else {
            
if (prev == 0 || prev == 1)
                return dfs(x, y+1, 4);
            else if (prev == 2)
                return dfs(x-1, y, 3);
            else
                return false;
        }
        
    }
}
