class Solution {
    int row;
    int column;
    boolean[][] visited;

    public boolean hasValidPath(int[][] grid) {
        row = grid.length;
        column = grid[0].length;
        visited = new boolean[row][column];
        return hasValidPath(grid, 0, 0);
    }

    private boolean hasValidPath(int[][] grid, int r, int c) {
        if (r == row - 1 && c == column - 1) {
            return true;
        }
        visited[r][c] = true;
        int currentPath = grid[r][c];
        int nextR = r, nextC = c;
        int[][] direction = new int[][]{{0, 0}};
        switch (currentPath) {
            case 1:
                direction = new int[][]{{0, -1}, {0, 1}};
                break;
            case 2:
                direction = new int[][]{{-1, 0}, {1, 0}};
                break;
            case 3:
                direction = new int[][]{{1, 0}, {0, -1}};
                break;
            case 4:
                direction = new int[][]{{1, 0}, {0, 1}};
                break;
            case 5:
                direction = new int[][]{{-1, 0}, {0, -1}};
                break;
            default:
                direction = new int[][]{{-1, 0}, {0, +1}};
        }
        boolean touched = false;
        for (int i = 0; i < direction.length; i++) {
            nextR = r + direction[i][0];
            nextC = c + direction[i][1];
            if (nextR < 0 || nextR >= row || nextC < 0 || nextC >= column) {
                continue;
            }
            if (!visited[nextR][nextC] && pair(direction[i], grid[nextR][nextC])) {
                touched |= hasValidPath(grid, nextR, nextC);
            }

        }
        return touched;
    }

    private boolean pair(int[] direction, int street) {
        switch (street) {
            case 1:
                return direction[0] == 0;
            case 2:
                return direction[1] == 0;
            case 3:
                return direction[0] == -1 || direction[1] == 1;
            case 4:
                return direction[0] == -1 || direction[1] == -1;
            case 5:
                return direction[0] == 1 || direction[1] == 1;
            default:
                return direction[0] == 1 || direction[1] == -1;
        }
    }
}
