class Solution {
    int n, m;
    boolean res = false;
    boolean[][] vis;
    
    public void dfs(int[][] grid,int x,int y){
        if(x <0 || x>=n || y<0 || y>=m || vis[x][y])    return;
        if(x == n-1 && y == m-1){
            res = true;
            return;
        }
        vis[x][y] = true;
        if(x-1>=0 && !vis[x-1][y]){
            if(grid[x][y] == 2 && (grid[x-1][y]==2 || grid[x-1][y]==3 || grid[x-1][y]==4))
                dfs(grid, x-1,y);
            if(grid[x][y] == 5 && (grid[x-1][y]==2 || grid[x-1][y]==3 || grid[x-1][y]==4))
                dfs(grid, x-1,y);
            if(grid[x][y] == 6 && (grid[x-1][y]==2 || grid[x-1][y]==3 || grid[x-1][y]==4))
                dfs(grid, x-1,y);
        }
        if(x+1 < n && !vis[x+1][y]){
            if(grid[x][y] == 2 && (grid[x+1][y]==2 || grid[x+1][y]==5 || grid[x+1][y]==6))
                dfs(grid, x+1,y);
            if(grid[x][y] == 3 && (grid[x+1][y]==2 || grid[x+1][y]==5 || grid[x+1][y]==6))
                dfs(grid, x+1,y);
            if(grid[x][y] == 4 && (grid[x+1][y]==2 || grid[x+1][y]==5 || grid[x+1][y]==6))
                dfs(grid, x+1,y);
        }
        if(y-1 >=0 && !vis[x][y-1]){
            if(grid[x][y] == 1 && (grid[x][y-1]==1 || grid[x][y-1]==4 || grid[x][y-1]==6))
                dfs(grid, x,y-1);
            if(grid[x][y] == 3 && (grid[x][y-1]==1 || grid[x][y-1]==4 || grid[x][y-1]==6))
                dfs(grid, x,y-1);
            if(grid[x][y] == 5 && (grid[x][y-1]==1 || grid[x][y-1]==6 || grid[x][y-1]==6))
                dfs(grid, x,y-1);
        }
        if(y+1 <m && !vis[x][y+1]){
            if(grid[x][y] == 1 && (grid[x][y+1]==1 || grid[x][y+1]==3 || grid[x][y+1]==5))
                dfs(grid, x,y+1);
            if(grid[x][y] == 4 && (grid[x][y+1]==1 || grid[x][y+1]==3 || grid[x][y+1]==5))
                dfs(grid, x,y+1);
            if(grid[x][y] == 6 && (grid[x][y+1]==1 || grid[x][y+1]==3 || grid[x][y+1]==5))
                dfs(grid, x,y+1);
        }
    }
    public boolean hasValidPath(int[][] grid) {
        n = grid.length;
        m = grid[0].length;
        vis = new boolean[n][m];
        
        dfs(grid, 0, 0);
        
        return res;
    }
}
