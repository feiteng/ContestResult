import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;

class Solution {
    public boolean hasValidPath(int[][] grid) {

        int n = grid.length, m = grid[0].length;
        UnionFind uf = new UnionFind(n * m);

        HashMap<Integer, HashSet<Integer>> adj = new HashMap<>();

        for (int i = 0; i < m * n; i++) {
            adj.put(i, new HashSet<>());
        }


        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {

                int val = grid[i][j];
                int key = i * m + j;
                HashSet<Integer> list = adj.get(key);

                
if (val == 1) {
                    if (j > 0) {
                        list.add(key - 1);
                    }
                    if (j + 1 < m) {
                        list.add(key + 1);
                    }
                } else if (val == 2) {
                    if (i > 0) {
                        list.add(key - m);
                    }
                    if (i + 1 < n) {
                        list.add(key + m);
                    }
                } else if (val == 3) {
                    if (j > 0) {
                        list.add(key - 1);
                    }
                    if (i + 1 < n) {
                        list.add(key + m);
                    }
                } else if (val == 4) {
                    if (j + 1 < m) {
                        list.add(key + 1);
                    }
                    if (i + 1 < n) {
                        list.add(key + m);
                    }
                } else if (val == 5) {
                    if (i > 0) {
                        list.add(key - m);
                    }
                    if (j > 0) {
                        list.add(key - 1);
                    }
                } else if (val == 6) 
{
                    if (i > 0) {
                        list.add(key - m);
                    }
                    if (j + 1 < m) {
                        list.add(key + 1);
                    }
                }
            }
        }


        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {

                int key = i * m + j;
                if (i > 0) {
                    int that = key - m;
                    if (adj.get(key).contains(that) && adj.get(that).contains(key)) {
                        uf.unite(key, that);
                    }
                }
                if (i + 1 < n) {
                    int that = key + m;
                    if (adj.get(key).contains(that) && adj.get(that).contains(key)) {
                        uf.unite(key, that);
                    }
                }
                if (j > 0) {
                    int that = key - 1;
                    if (adj.get(key).contains(that) && adj.get(that).contains(key)) {
                        uf.unite(key, that);
                    }
                }
                if (j + 1 < m) {
                    int that = key + 1;
                    if (adj.get(key).contains(that) && adj.get(that).contains(key)) {
                        uf.unite(key, that);
                    }
                }
            }
        }

        return uf.find(0) == uf.find(m * n - 1);

    }

    private class UnionFind {
        private int[] root;

        private UnionFind(int n) {
            root = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
        }

        private void unite(int s, int t) {
            int rs = find(s);
            int rt = find(t);
            if (rs != rt) {
                root[rs] = rt;
            }
        }

        private int find(int s) {
            while (s != root[s]) {
                root[s] = root[root[s]];
                s = root[s];
            }
            return s;
        }
    }
}
