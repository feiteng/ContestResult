class Solution {
    
    int[][][] table = new int[][][] {
        {{}, {}, {}, {}},
        // left, up, right, down
        { {1,4,6}, {}, {1,3,5}, {} },
        { {}, {2,3,4}, {}, {2,5,6} },
        { {1,4,6}, {}, {}, {2,5,6} },
        { {}, {}, {1,3,5}, {2,5,6} },
        { {1,4,6}, {2,3,4}, {}, {} },
        { {}, {2,3,4}, {1,3,5}, {} }
    };NT>
    
    int[][] dirs = new int[][] { {0,-1}, {-1,0}, {0,1}, {1,0} };
    
    public boolean hasValidPath(int[][] grid) {
        int N = grid.length;
        int M = grid[0].length;
        boolean[][] visited = new boolean[N][M];
        
        return dfs(grid, 0, 0, visited);
    }
    
    public boolean dfs(int[][] grid, int i, int j, boolean[][] visited) {
        int N = grid.length;
        int M = grid[0].length;
        int v = grid[i][j];
        
        visited[i][j] = true;
        if (i == N - 1 && j == M - 1) return true;
        
        for (int d = 0; d < dirs.length; d++) {
            int i2 = i + dirs[d][0];
            int j2 = j + dirs[d][1];
            
            int[] cands = table[v][d];
            if (i2 < 0 || j2 < 0 || i2 >= N || j2 >= M || visited[i2][j2]) continue;
            int next = grid[i2][j2];
            if (Arrays.binarySearch(cands, next) < 0) continue;
            
            if (dfs(grid, i2, j2, visited)) return true;
        }
        
        return false;
    }
}
