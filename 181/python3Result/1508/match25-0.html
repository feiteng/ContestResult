class Solution:
    def longestPrefix(self, s: str) -> str:
        n = len(s)
        
        dp = [0] * n
        curr = 0
        i = 1
        while i < n:
            if s[i] == s[curr]:
                curr += 1
                dp[i] = curr
                i += 1
            else:
                if curr:
                    curr = dp[curr - 1]
                else:
                    dp[i] = 0
                    i += 1
        best = dp[-1]
        return s[:best]
