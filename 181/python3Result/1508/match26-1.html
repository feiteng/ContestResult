class Solution:
    def longestPrefix(self, s: str) -> str:
        n = len(s)
        tmp = [0] * n

        curmax = 0
        i = 1
        while i < n:
            if s[i] == s[curmax]:
                curmax += 1
                tmp[i] = curmax
                i += 1
            else:
                if curmax > 0:
                    curmax = tmp[curmax - 1]
                else:
                    tmp[i] = 0
                    i += 1
        # print(tmp)
        return s[:tmp[-1]]
