class Solution:
    def longestPrefix(self, s: str) -> str:
        n = len(s)
        t = [0] * n
        p = 1
        lp = 0
        while p < n:
            if s[p] == s[lp]:
                lp += 1
                t[p] = lp
                p += 1
            else:
                if lp > 0:
                    lp = t[lp-1]
                else:
                    t[p] = 0
                    p += 1
        # print(t)
        return s[:t[-1]]
