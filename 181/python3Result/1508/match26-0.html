class Solution:
    def longestPrefix(self, s: str) -> str:
        n = len(s)
        lps = [0] * n
        
        l = 0
        i = 1
        while (i<n):
            if (s[i]==s[l]):
                l = l+1
                lps[i] = l
                i = i + 1
            else:
                if (l!=0):
                    l = lps[l-1]
                else:
                    lps[i] = 0
                    i = i + 1
        return s[0:lps[n-1]]
