class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def compute(num):
            fac = set()
            for i in range(1, int(num**0.5)+1):
                if num % i == 0:
                    fac.add(i)
                    fac.add(num//i)
            return fac
        res = 0
        for i in nums:
            a = compute(i)
            if len(a) == 4:
                res += sum(a)
        return res
