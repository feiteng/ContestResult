class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        def factors(x):
            i = 1
            s = 0
            ss = 0
            while i * i <= x:
                if x % i == 0:
                    s += 1
                    ss += i
                    if x//i != i:
                        s 
+= 1
                        ss += x // i
                i += 1
            return ss if s == 4 else None
        
        a = 0
        for n in nums:
            t = factors(n)
            if t is not None:
                a += t
                
        return a
