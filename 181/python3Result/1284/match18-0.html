import math
class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
    
        def find(num):
            total = 1 + num
            cnt = 2
            
            for i in range(2, math.floor(math.sqrt(num))+1):
                if num % i == 0:
                    if num // i == i:
                        cnt += 1
                    else:
                        cnt += 2
                
                    if cnt > 4:
                        return 0
                    total += i if num // i == i else i + num // i
            
            return total if cnt == 4 else 0
        
        ans = 0
        for num in nums:
            if num <= 5:
                continue
            ans += find(num)
            
        return ans
