class Solution:
    def findFacs(self, n):
        cnt = 0
        ss = 0
        i = 1
        while i * i <= n and cnt <= 4:
            if n % i == 0:
                ss += i
                cnt += 1
                if n != i * i:
                    ss += n // i
                    cnt += 1
            i += 1
        return ss if cnt == 4 else 0
    
    def sumFourDivisors(self, nums: List[int]) -> int:
        return sum(map(self.findFacs, nums))
