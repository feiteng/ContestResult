class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def count(x):
            i = 1
            cnt, s = 0, 0
            
            while i * i <= x:
                if x % i == 0:
                    if i * i != x:
                        cnt += 2
                        s += i + x // i
                    else:
                        cnt += 1
                        s += i
                i += 1
            return cnt, s
        ans = 0
        for x in nums:
            cnt, s = count(x)
            if cnt == 4:
                ans += s
        return ans
