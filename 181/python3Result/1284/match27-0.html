class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        ans = 0
        for n in nums:
            div = set()
            for k in range(1, n):
                if k * k > n:
                    break
                if n % k == 0:
                    div.add(k)
                    div.add(n // k)
            if len(div) == 4:
                ans += sum(list(div))
        return ans
