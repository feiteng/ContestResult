class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def find_divide(n):
            divide = set()
            for i in range(1, int(n ** 0.5) + 1):
                if n % i == 0:
                    divide.add(i)
                    divide.add(
n // i)
            return divide

        res = 0
        for n in nums:
            divide = find_divide(n)
            if len(divide) == 4:
                res += sum(divide)
        return res
