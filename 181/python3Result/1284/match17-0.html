class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        def divisors(x):
            result = []
            for i in range(1, math.isqrt(x) + 1):
                if x % i == 0:
                    result.append(i)
                    if x != i * i:
                        result.append(x // i)
            return result
        
        result = 0
        for x in nums:
            divs = divisors(x)
            if len(divs) == 4:
                result += sum(divs)
        return result
