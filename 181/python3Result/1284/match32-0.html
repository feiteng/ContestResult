class Solution:
    def get_divisor(self, num):
        from math import sqrt
        res = [1, num]
        for i in range(2, int(sqrt(num))+1):
            if num % i == 0:
                res.append(i)
                if i != num // i:
                    res.append(num//i)
                if len(res) > 4:
                    return res
        return res
    
    def sumFourDivisors(self, nums: List[int]) -> int:
        ret = 0
        for num in nums:
            res = self.get_divisor(num)
            if len(res) == 4:
                ret += sum(res)
        return ret
