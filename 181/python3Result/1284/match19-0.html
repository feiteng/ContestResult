import math
class Solution:
    def finddiv(self, num):
        res = set()
        for i in range(1, int(math.sqrt(num)) + 1):
            if num % i == 0:
                res.add(i)
                res.add(num/i)
        return res
                
    def sumFourDivisors(self, nums: List[int]) -> int:
        s = 0
        for n in nums:
            res = self.finddiv(n)
            if len(res) == 4:
                s += sum(res)
        return int(s)
