class Solution:
    def getDivisors(self, num):
        res = [1, num]
        for i in range(2, int(math.sqrt(num)) + 1):
            if num % i == 0:
                j = num // i
                res.append(i)
                if j != i:
                    res.append(j)
                    
            if len(res) > 4:
                break
        return res            
    
    def sumFourDivisors(self, nums: List[int]) -> int:
        res = 0
        for num in nums:
            divisors = self.getDivisors(num)
            if len(divisors) == 4:
                res += sum(divisors)        
        return res
