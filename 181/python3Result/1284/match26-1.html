class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        result = 0
        for num in nums:
            divisors = set()
            for i in range(1, num, 1):
                if i > num//i:
                    break
                
if num % i == 0:
                    divisors.add(i)
                    divisors.add(num//i)
                    if len(divisors) > 4:
                        break
            if len(divisors) == 4:
                result += sum(divisors)
        return result
