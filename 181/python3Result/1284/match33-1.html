import math

class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def divisors(n):
            count = set()
            for i in range(1, int(math.sqrt(n)) + 1):
                if n % i == 0:
                    count.add(i)
                    count.add(n//i)
                if len(count) > 4:
                    break
            if len(count) == 4:
                return sum(count)
            else:
                return 0
        ans = 0
        for i in nums:
            ans += divisors(i)
        return ans
