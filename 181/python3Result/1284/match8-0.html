def factors(n):
    fhead = []
    ftail = []
    for x in range(1, int(math.sqrt(n)) + 1):
        if n % x == 0:
            fhead.append(x)
            if x * x != n:
                # This check is to not insert sqrt(n) twice in head and tail portion of the list
                ftail.append(n // x)
    return fhead + ftail[::-1]



class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        total = 0
        for x in nums:
            divisors = factors(x)
            if len(divisors) == 4:
                total += sum(divisors)
        return total
