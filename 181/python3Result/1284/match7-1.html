import math

class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        ans = 0
        for num in nums:
            
            cnt = 0
            sum1 = 0
            for t1 in range(1, 1 + int(math.floor(math.sqrt(num)))):                
                if num % t1 == 0:
                    t2 = num // t1
                    if t1 != t2:
                        cnt += 2
                        sum1 
+= (t1 + t2)
                    else:
                        cnt += 1
                        sum1 += t1
            if cnt == 4:
                ans += sum1
        return ans
