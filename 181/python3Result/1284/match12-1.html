class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def divisors(num):
            res = []
            
for i in range(1, int(num ** 0.5) + 1):
                if num % i == 0:
                    res.append(i)
                    if i != num // i:
                        res.append(num // i)
            return res
        res = 0
        
for num in nums:
            d = divisors(num)
            if len(d) == 4:
                res += sum(d)
        return res
