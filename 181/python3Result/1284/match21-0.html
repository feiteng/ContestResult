class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def four(n):
            s = set()
            for i in range(1, int(math.sqrt(n))+1):
                if n % i == 0:
                    s.add(i)
                    s.add(n // i)
           # print(n, s)
            return sum(s) if len(s)  == 4 else 0
        
        return sum(four(x) for x in nums )
