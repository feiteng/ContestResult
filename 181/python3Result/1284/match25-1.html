class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        res=0
        for x in nums:
            dset
=set()
            for d in range(1, int(math.sqrt(x))+1  ):
                if x%d==0:
                    dset.add(d)
                    dset.add(x//d)
                if len(dset)>4:
                    break
            
            if len(dset)==4:
                res+=sum(dset)
        return res
