import math
class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        nums.sort()
        divsum = 0
        for n in nums:
            if n < 6:
                continue
            d = set([1, n])
            for i in range(2, int(math.sqrt(n)) + 1):
                if n % i == 0:
                    d.add(i)
                    d.add(n // i)
                    if len(d) > 4:
                        break
            if len(d) == 4:
                divsum += sum(d)
            # print(n, divsum, d)
        return divsum
