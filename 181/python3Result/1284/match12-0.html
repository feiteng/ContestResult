class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        ans = 0
        def get_divs(n: int) -> List[int]:
            res = [1, n]
            for i in range(2, int(n ** 0.5) + 1):
                if n % i == 0:
                    res.append(i)
                    if i ** 2 != n:
                        res.append(n // i)
            return res
                
        for x in nums:
            res = get_divs(x)
            if len(res) == 4:
                ans += sum(res)
        return ans
