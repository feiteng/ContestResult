class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        ans = 0
        for num in nums:
            count = 2
            j = 0
            for i in range(2, int(math.sqrt(num))+1):
                if num % i == 0:
                    if num // i != i:
                        count += 2
                        j = i
                    else:
                        count += 1
            if count == 4:
                ans += (1 + num + j + num//j)
                
        return ans
