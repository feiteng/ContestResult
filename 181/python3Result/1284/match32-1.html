class Solution:
    
    def get_divisors(self, n):
        divs = set()
        for i in range(1, math.floor(sqrt(n)) + 1):
            if n % i == 0:
                divs.add(i)
                divs.add(
n // i)
        return divs
    
    def sumFourDivisors(self, nums: List[int]) -> int:
        total = 0
        for n in nums:
            divis = self.get_divisors(n)
            if len(divis) == 4:
                total += sum(divis)
        return total
