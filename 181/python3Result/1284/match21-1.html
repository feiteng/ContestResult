class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def foo(n):
            d = set()
            for i in range(1, int(n ** .5) + 1):
                a, b = divmod
(n, i)
                if b == 0:
                    d.add(i)
                    d.add(a)
            return sum(d) if len(d) == 4 else 0
        return sum(map(foo, nums))
