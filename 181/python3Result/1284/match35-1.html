class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        res = 0
        for x in nums:
            factors = []
            isAns = True
            for f in range(1, int(x**(0.5))+1):
                if x % f == 0:
                    if len(factors) == 4 or f**2 == x:
                        isAns = False
                    factors
.append(f)
                    factors.append(x//f)
            if isAns and len(factors) == 4:
                res += sum(factors)
        return res
