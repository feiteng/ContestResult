class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def getdiv(num):
            res = set()
            for i in range(1, int(num**0.5)+1):
                if num % i == 0:
                    res.add(i)
                    res.add(num//i)
            # print(res)
            return list(res)
        res = 0
        for num in nums:
            temp = getdiv(num)
            if len(temp) == 4:
                res += sum(temp)
        return res
