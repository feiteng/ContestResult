class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def helper(num):
            res = set()
            
            for i in range(1, int(num ** 0.5) + 1):
                if num % i == 0:
                    res.add(i)
                    res.add(num // i)
            
            #print(res)
            if len(res) == 4:
                return sum(res)
            return 0
        
        res = 0
        for n in nums:
            res += helper(n)
        return res
