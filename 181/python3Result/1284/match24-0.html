class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        ans = 0
        for n in nums:
            k = set()
            for y in range(1, int(n ** 0.5) + 1):
                if n % y == 0:
                    k.add(y)
                    k.add(n // y)
            if len(k) == 4:
                ans += sum(k)
        return ans
