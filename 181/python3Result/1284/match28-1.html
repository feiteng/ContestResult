class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        return sum(map(self.helper, nums))
        
    @lru_cache(None)
    
def helper(self, num):
        tmp = set()
        for i in range(1, int(num**0.5) + 1):
            if num % i == 0:
                tmp.add(i)
                tmp.add(num//i)
        if len(tmp) == 4:
            return sum(tmp)
        else:
            return 0
