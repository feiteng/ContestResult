class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        maxNumber = 10**5
        fourDivisors = [False] * (maxNumber + 1)
        allPrimes = [True] * (maxNumber + 1)
        divisor, prime, p = {}, [], 2
 
        while (p * p <= maxNumber):
            if (allPrimes[p] == True):
                i = p * 2
                while (i <= maxNumber):
                    allPrimes[i] = False
                    i += p
            p += 1
 
        for p in range(2, maxNumber + 1):
            if (allPrimes[p]):
                prime.append(p)
                
        for i in range(len(prime)):  
            p = prime[i]
            if (1 * p * p * p <= maxNumber):
                fourDivisors[p * p * p] = True
                divisor[p * p * p] = [1, p, p * p, p * p * p]

            for j in range(i + 1, len(prime)):
                q = prime[j]

                if (1 * p * q > maxNumber):
                    break

                fourDivisors[p * q] = True
                divisor[p * q] = [1, p, q, p * q]
               
               
        res = 0
        for n in nums:
            if fourDivisors[n]:
                res += sum(divisor[n])
               
        return res
