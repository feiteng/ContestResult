from collections import defaultdict

class Solution:

  def graph(self, M):
    Y, X = len(M), len(M[0])

    G = defaultdict(set)
    for y in range(Y):
      for x in range(X):
        v = M[y][x]
        if v == 1:
          if x > 0 and M[y][x - 1] in [1, 4, 6]:
            G[(y, x)].add((y, x - 1))
          if x < X - 1 and M[y][x + 1] in [1, 3, 5]:
            G[(y, x)].add((y, x + 1))
        elif v == 2:
          if y > 0 and M[y - 1][x] in [2, 3, 4]:
            G[(y, x)].add((y - 1, x))
          if y < Y - 1 and M[y + 1][x] in [2, 5, 6]:
            G[(y, x)].add((y + 1, x))
        elif v == 3:
          if x > 0 and M[y][x - 1] in [1, 4, 6]:
            G[(y, x)].add((y, x - 1))
          if y < Y - 1 and M[y + 1][x] in [2, 5, 6]:
            G[(y, x)].add((y + 1, x))
        elif v == 4:
          if x < X - 1 and M[y][x + 1] in [1, 3, 5]:
            G[(y, x)].add((y, x + 1))
          if y < Y - 1 and M[y + 1][x] in [2, 5, 6]:
            G[(y, x)].add((y + 1, x))
        elif v == 5:
          if y > 0 and M[y - 1][x] in [2, 3, 4]:
            G[(y, x)].add((y - 1, x))
          if x > 0 and M[y][x - 1] in [1, 4, 6]:
            G[(y, x)].add((y, x - 1))
        else:
          if y > 0 and M[y - 1][x] in [2, 3, 4]:
            G[(y, x)].add((y - 1, x))
          if x < X - 1 and M[y][x + 1] in [1, 3, 5]:
            G[(y, x)].add((y, x + 1))
    
    return G

  def hasValidPath(self, M):
    Y, X = len(M), len(M[0])
    if Y == X == 1:
      return True

    G = self.graph(M)
    # for k, v in G.items():
    #   print(k, v)
    # print()
    frontier, seen = [(0, 0)], set([(0, 0)])
    
    while frontier:
      # print(frontier)
      new_frontier = []
      for el in frontier:
        for y, x in G[el]:
          if (y, x) == (Y - 1, X - 1):
            return True

          if (y, x) not in seen:
            seen.add((y, x))
            new_frontier.append((y, x))
      frontier = new_frontier
    
    return False
