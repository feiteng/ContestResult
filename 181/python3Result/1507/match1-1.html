class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        dx, dy = [1, -1, 0, 0], [0, 0, 1, -1]
        vis = set()
        queue = collections.deque()
        queue.appendleft((0, 0))
        vis.add((0, 0))
        
        while queue:
            cur = queue.pop()
            if cur == (m - 1, n - 1):
                return True
            for k in range(4):
                p, q = cur[0] + dx[k], cur[1] + dy[k]
                if not (0 <= p < m and 0 <= q < n): continue
                if (p, q) in vis: continue
                val = grid[cur[0]]
[cur[1]]
                if val in [2, 3, 4] and dx[k] == 1 and grid[p][q] in [2, 5, 6]:
                    queue.appendleft((p, q))
                    vis.add((p, q))
                if val in [1, 3, 5] and dy[k] == -1 and grid[p][q] in [4, 6, 1]:
                    queue.appendleft((p, q))
                    vis.add((p, q))
                if val in [2, 6, 5] and dx[k] == -1 and grid[p][q] in [2, 3, 4]:
                    queue.appendleft((p, q))
                    vis.add((p, q))
                if val in [1, 4, 6] and dy[k] == 1 and grid[p][q] in [1, 3, 5]:
                    queue.appendleft((p, q))
                    vis.add((p, q))
        return False
