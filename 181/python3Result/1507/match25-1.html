class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        steps = {1:{(-1,0): [1,4,6], (1,0):[1,3,5]},
                 2:{(0,-1):[2,3,4], (0,1):[2,5,6]},
                 3:{(-1,0):[1,2,4], (0,1):[2,5,6]},
                 4:{(0,1):[2,5,6], (1,0):[1,3,5]},
                 5:{(0,-1):[2,3,4], (-1,0):[1,4,6]},
                 6:{(0,-1):[2,3,4], (1,0):[1,3,5]},
                }
        m = len(grid)
        n = len(grid[0])
        seen = [[0]*n for _ in range(m)]
        def dfs(x,y):
            if (x,y) == (m-1,n-1):
                return True
            for way, vals in steps[grid[x][y]].items():
                i,j = x+way[1], y+way[0]
                # print(i,j, grid[i][j], vals, seen[i][j])
                if 0<=i<m and 0<=j<n and grid[i][j] in vals and not seen[i][j]:
                    # print(i,j)
                    seen[i][j] = 1
                    if dfs(i,j):
                        return True
        seen[0][0]=1
        res = dfs(0,0)
        # print(seen)
        return res
