class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        
        if len(grid) == 1 and len(grid[0]) == 1:
            
return True
        
        neighbours = {
            1: [(0, -1), (0, 1)],
            2: [(-1, 0), (1, 0)],
            3: [(0, -1), (1, 0)],
            4: [(1, 0), (0, 1)],
            5: [(0, -1), (-1, 0)],
            6: [(-1, 0), (0, 1)]
        }
        
        max_row = len(grid) - 1
        max_col = len(grid[0]) - 1
        
        target = (len(grid) - 1, len(grid[0]) -1)
        
        seen = {
(0, 0)}
        stack = [(0, 0)]
        while stack:
            row, col = stack.pop()
            for dr, dc in neighbours[grid[row][col]]:
                new_row = row + dr
                new_col = col + dc
                if new_row > max_row or new_col > max_col or new_row < 0 or new_col < 0:
                    continue
                # Otherwise, we need to investigate the new neighhour to ensure it has the
                # connection we require.
                cell = (new_row, new_col)
                # It does, if one of its neighbours is the cell we just came out of!
                for dr2, dc2 in neighbours[grid[new_row][new_col]]:
                    if new_row + dr2 == row and new_col + dc2 == col:
                        break
                else:
                    # It's not actually reachable.
                    continue

                if cell == target:
                    return True
                if cell not in seen:
                    seen.add(cell)
                    stack.append(cell)
        
        return False
