class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        q = [(0, 0)]
        rec = set()
        rec.add((0, 0))
        
        nex = [[0, 1], [1, 0], [0, -1], [-1, 0]]
        
        for x, y in q:
            # print(q)
            if x==m-1 and y==n-1:
                return True
            di = grid[x][y]
            for i, v in enumerate(nex):
                nx, ny = v
                xx, yy = nx+x, ny+y
                if -1<xx<m and -1<yy<n and (xx, yy) not in rec:
                    # print(xx, yy, i)
                    ndi = grid[xx][yy]
                    if i==0:
                        # print(di, ndi)
                        if di in [1, 4, 6] and ndi in [1, 3, 5]:
                            q.append((xx, yy))
                            rec.add((xx, yy))
                    elif i==1:
                        # print(di, ndi)
                        if di in [2, 3, 4] and ndi in [2, 5, 6]:
                            # print('here')
                            q.append((xx, yy))
                            rec.add((xx, yy))
                    elif i==2:
                        if di in [1, 3, 5] and ndi in [1, 4, 6]:
                            q.append((xx, yy))
                            rec.add((xx, yy))
                    elif i==3:
                        if di in [2, 5, 6] and ndi in [2, 3, 4]:
                            q.append((xx, yy))
                            rec.add((xx, yy))
            
        
        
        return False
