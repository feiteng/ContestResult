class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        seen = {(0, 0)}
        q = [(0, 0)]
        delta = [
            [],
            [(0, -1, (1, 4, 6)), (0, 1, (1, 3, 5))],
            [(-1, 0, (2, 3, 4)), (1, 0, (2, 5, 6))],
            [(0, -1, (1, 4, 6)), (1, 0, (2, 5, 6))],
            [(0, 1, (1, 3, 5)), (1, 0, (2, 5, 6))],
            [(0, -1, (1, 4, 6)), (-1, 0, (2, 3, 4))],
            [(0, 1, (1, 3, 5)), (-1, 0, (2, 3, 4))],
        ]
        for i, j in q:
            if (i, j) == (m - 1, n - 1):
                return True
            val = grid[i][j]
            for di, dj, connect in delta[val]:
                ni, nj = i + di, j + dj
                if 0 <= ni < m and 0 <= nj < n and (ni, nj) not in seen and grid[ni][nj] in connect:
                    seen.add((ni, nj))
                    q.append((ni, nj))
        return False
