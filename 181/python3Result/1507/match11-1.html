class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        queue = [(0, 0)]
        seen = {(0, 0)}
        offsets = {1: [[0, 1, {1, 3, 5}], [0, -1, {1, 4, 6}]],
                  2: [[1, 0, {2, 5, 6}], [-1, 0, {2, 3, 4}]], 
                  3: [[
0, -1, {1, 4, 6}], [1, 0, {2, 5, 6}]],
                  4: [[0, 1, {1, 3, 5}], [1, 0, {2, 5, 6}]],
                  5: [[0, -1, {1, 4, 6}], [-1, 0, {2, 3, 4}]],
                  6: [[0, 1, {1, 3, 5}], [-1, 0, {2, 3, 4}]],}     
        m, n = map(len, (grid, grid[0]))
        while queue:
            new = []
            for i, j in queue:
                if i == m - 1 and j == n - 1:
                    return True
                else:
                    cur = grid[i][j]
                    for o in offsets[cur]:
                        i_next = i + o[0]
                        j_next = j + o[1]
                        if 0 <= i_next < m and 0 <= j_next < n and grid
[i_next][j_next] in o[2] and (i_next, j_next) not in seen:
                            seen.add((i_next, j_next))
                            new.append((i_next, j_next))
            queue = new
        return False
