class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m,n=len(grid),len(grid[0])
        vis=set()
        dr
=[{(0,1):[1,3,5], (0,-1):[1,4,6]},
            {(1,0):[2,5,6], (-1,0):[2,3,4]},
            {(1,0):[2,5,6], (0,-1):[1,4,6]},
            {(1,0):[2,5,6], (0,1):[1,3,5]},
            {(-1,0):[2,3,4], (0,-1):[1,4,6]},
            {(-1,0):[2,3,4], (0,1):[1,3,5]}]
        st=[]
        st.append((0,0))
        vis.add((0,0))
        while len(st)>0:
            i,j=st.pop()
            if i==m-1 and j==n-1:
                
return True
            for d,v in dr[grid[i][j]-1].items():
                r,c=i+d[0],j+d[1]
                if r<0 or r>=m or c<0 or c>=n:
                    continue
                if (r,c) not in vis and grid[r][c] in v:
                    st.append((r,c))
                    vis.add((r,c))
        return False
