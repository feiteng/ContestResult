class Solution:
    def __init__(self):
        self.dt = {            
            1: [(1, (0, 1)), (1, (0, -1)), (3, (0, 1)), (4, (0, -1)), (5, (0, 1)), (6, (0, -1))],
            2: [(2, (1, 0)), (2, (-1, 0)), (3, (-1, 0)), (4, (-1, 0)), (5, (1, 0)), (6, (1, 0))],
            3: [(1, (0, -1)), (2, (1, 0)), (4, (0, -1)), (5, (1, 0)), (6, (1, 0))],
            4: [(1, (0, 1)), (2, (1, 0)), (3, (0, 1)), (5, (1, 0)), (6, (1, 0))],
            5: [(1, (0, -1)), (2, (-1, 0)), (3, (-1, 0)), (4, (-1, 0)), (6, (0, -1))],
            6: [(1, (0, 1)), (2, (-1, 0)), (3, (-1, 0)), (4, (-1, 0)), (5, (0, 1))],
        }
        
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        return self.dfs(grid, 0, 0, m, n, set())
    
    def dfs(self, grid, i, j, m, n, visited):
        if i == m - 1 and j == n - 1:
            return True
        
        visited.add((i, j))        
        for nv, nPos in self.dt[grid[i][j]]:
            ni, nj = i + nPos[0], j + nPos[1]
            if 0 <= ni < m and 0 <= nj < n and grid[ni][nj] == nv and (ni, nj) not in visited:
                if self.dfs(grid, ni, nj, m, n, visited):
                    return True
        visited.remove((i, j))
        
        return False
