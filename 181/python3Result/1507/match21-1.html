class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        valid 
= {
            1: [[1, 4, 6], [], [1, 3, 5], [] ],
            2: [[], [2, 3, 4], [], [2, 5, 6]],
            3: [[1,4,6], [], [], [2,5,6]],
            4: [[], [], [1,3,5], [2,5,6]],
            5: [[1, 4, 6], [2,3,4], [], []],
            6: [[], [2,3,4], [1,3,5], []]
        }
        n, m = len(grid), len(grid[0])
        s = [(0, 0)]
        visited = set()
        while s:
            i, j = s.pop()
            if i == n-1 and j == m-1:
                return True
            visited.add((i, j))
            for d, path in enumerate(valid[grid[i][j]]):
                if d == 0 and j > 0 and grid[i][j-1] in path and (i, j-1) not in visited:
                    s.append
((i, j-1))
                elif d == 1 and i > 0 and grid[i-1][j] in path and (i-1, j) not in visited:
                    s.append((i-1, j))
                elif d == 2 and j < m-1 and grid[i][j+1] in path and (i, j+1) not in visited:
                    s.append((i, j+1))
                elif d == 3 and i < n-1 and grid[i+1][j] in path and (i+1, j) not in visited:
                    s.append((i+1, j))
        return False
