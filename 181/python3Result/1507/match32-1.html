class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        stack, walked = [(0, 0)], set([(0,0)])
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        while stack:
            cur = stack.pop()
            if cur[0] == len(grid)-1 and cur[1] == len(grid[0])-1:
                return True
            for idx, d in enumerate(directions):
                i, j = cur[0]+d[0], cur[1]+d[1]
                if 0 <= i < len(grid) and 0 <= j < len(grid[0]):
                    if idx == 0 and grid[cur
[0]][cur[1]] in [2,5,6] and grid[i][j] in [2,3,4] and (i,j) not in walked:
                        walked.add((i,j))
                        stack.append((i,j))
                    elif idx == 1 and grid[cur[0]]
[cur[1]] in [2,3,4] and grid[i][j] in [2,5,6] and (i,j) not in walked:
                        walked.add((i,j))
                        stack.append((i,j))
                    elif idx == 2 and grid[cur[0]]
[cur[1]] in [1,3,5] and grid[i][j] in [1,4,6] and (i,j) not in walked:
                        walked.add((i,j))
                        stack.append((i,j))
                    elif grid[cur
[0]][cur[1]] in [1,4,6] and grid[i][j] in [1,3,5] and (i,j) not in walked:
                        walked.add((i,j))
                        stack.append((i,j))
        return False
