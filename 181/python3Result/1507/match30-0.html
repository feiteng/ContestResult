class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        
        m, n = len(grid), len(grid[0])

        quene = [(0, 0)]
        visited = [[False] * n for _ in range(m)]
        visited[0][0] = True

        while quene != []:

            x, y = quene.pop(0)
            if x == m - 1 and y == n - 1:
                return True
            next_pos = []

            if grid[x][y] == 1:
                if y - 1 >= 0     and (grid[x][y - 1] == 1 or grid[x][y - 1] == 4 or grid[x][y - 1] == 6) and visited[x][y - 1] == False:
                    next_pos.append([x, y - 1])
                if y + 1 <= n - 1 and (grid[x][y + 1] == 1 or grid[x][y + 1] == 3 or grid[x][y - 1] == 5) and visited[x][y + 1] == False:
                    next_pos.append([x, y + 1])
            elif grid[x][y] == 2:
                if x - 1 >= 0     and (grid[x - 1][y] == 2 or grid[x - 1][y] == 3 or grid[x - 1][y] == 4) and visited[x - 1][y] == False:
                    next_pos.append([x - 1, y])
                if x + 1 <= m - 1 and (grid[x + 1][y] == 2 or grid[x + 1][y] == 5 or grid[x + 1][y] == 6) and visited[x + 1][y] == False:
                    next_pos.append([x + 1, y])
            elif grid[x][y] == 3:
                if y - 1 >= 0     and (grid[x][y - 1] == 1 or grid[x][y - 1] == 4 or grid[x][y - 1] == 6) and visited[x][y - 1] == False:
                    next_pos.append([x, y - 1])
                if x + 1 <= m - 1 and (grid[x + 1][y] == 2 or grid[x + 1][y] == 5 or grid[x + 1][y] == 6) and visited[x + 1][y] == False:
                    next_pos.append([x + 1, y])
            elif grid[x][y] == 4:
                if y + 1 <= n - 1 and (grid[x][y + 1] == 1 or grid[x][y + 1] == 3 or grid[x][y + 1] == 5) and visited[x][y + 1] == False:
                    next_pos.append([x, y + 1])
                if x + 1 <= m - 1 and (grid[x + 1][y] == 2 or grid[x + 1][y] == 5 or grid[x + 1][y] == 6) and visited[x + 1][y] == False:
                    next_pos.append([x + 1, y])
            elif grid[x][y] == 5:
                if x - 1 >= 0     and (grid[x - 1][y] == 2 or grid[x - 1][y] == 3 or grid[x - 1][y] == 4) and visited[x - 1][y] == False:
                    next_pos.append([x - 1, y])
                if y - 1 >= 0     and (grid[x][y - 1] == 1 or grid[x][y - 1] == 4 or grid[x][y - 1] == 6) and visited[x][y - 1] == False:
                    next_pos.append([x, y - 1])
            elif grid[x][y] == 6:
                if x - 1 >= 0     and (grid[x - 1][y] == 2 or grid[x - 1][y] == 3 or grid[x - 1][y] == 4) and visited[x - 1][y] == False:
                    next_pos.append([x - 1, y])
                if y + 1 <= n - 1 and (grid[x][y + 1] == 1 or grid[x][y + 1] == 3 or grid[x][y + 1] == 5) and visited[x][y + 1] == False:
                    next_pos.append([x, y + 1])

            for i in range(len(next_pos)):
                visited[next_pos[i][0]][next_pos[i][1]] = True
                quene.append(next_pos[i])


        return False
