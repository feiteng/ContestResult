class Solution:
    def hasValidPath(self, grid: List[List
[int]]) -> bool:
        go = {
            1:((0, 1), (0, -1)),
            2:((1, 0), (-1, 0)),
            3:((0, -1), (1, 0)),
            4:((0, 1), (1, 0)),
            5:((-1, 0), (0, -1)),
            6:((-1, 0), (0, 1)),
        }
        m = len(grid)
        n = len(grid[0])
        queue = [(0, 0)]
        s = set(queue)
        p = 0
        while p < len(queue):
            u, v = queue[p]
            if u == m - 1 and v == n - 1:
                return True
            pattern = go[grid[u][v]]
            for iu, iv in pattern:
                x, y = u + iu, v + iv
                if (x >= 0 and x < m and y >= 0 and y < n) and ((
x, y) not in s and (-iu, -iv) in go[grid[x][y]]):
                    s.add((x, y))
                    queue.append((x, y))                 
            p += 1
        return False
