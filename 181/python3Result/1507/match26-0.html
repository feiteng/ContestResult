class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        self.n = len(grid)
        self.m = len(grid[0])
        self.vis = [[False for j in range(self.m)] for i in range(self.n)]
        from collections import deque
        l = deque()
        l.append((0, 0))
        while len(l) > 0:
            x, y = l.popleft()
            if x == self.n - 1 and y == self.m - 1:
                return True
            self.vis[x][y] = True
            if y > 0 and grid[x][y] in [1, 3, 5] and grid[x][y - 1] in [1, 4, 6] and not self.vis[x][y - 1]:
                l.append((x, y - 1))
            if y < self.m - 1 and grid[x][y] in [1, 4, 6] and grid[x][y  + 1] in [1, 3, 5] and not self.vis[x][y + 1]:
                l.append((x, y + 1))
            if x > 0 and grid[x][y] in [2, 5, 6] and grid[x - 1][y] in [2, 3, 4] and not self.vis[x - 1][y]:
                l.append((x - 1, y))
            if x < self.n - 1 and grid[x][y] in [2, 3, 4] and grid[x + 1][y] in [2, 5, 6] and not self.vis[x + 1][y]:
                l.append((x + 1, y))
        return False
