class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        if m == 1 and n == 1:
            return True
        dic = {
            1 : [[], [],     [1, 4,6], [1, 3,5]],
            2 : [[2,3,4],    [2,5,6], [], []],
            3 : [[], [2,5,6], [1,4,6], []],
            4 : [[], [2,5,6], [], [1, 3,5]],
            5 : [[2,3,4], [], [1,4,6], []],
            6 : [[2,3,4], [], [], [1, 3,5]]
        }
        visited = set()
        def dfs(i, j):
            s = grid[i][j]
            for idx, pos in enumerate([(i-1, j), (i+1, j), (i, j-1), (i, j+1)]):
                n_i, n_j = pos
                if 0<=n_i<m and 0<=n_j<n and (n_i, n_j) not in visited and grid[n_i][n_j] in dic[s][idx]:
                    if (n_i, n_j) == (m-1, n-1):
                        return True
                    visited.add((n_i, n_j))
                    flag = dfs(n_i, n_j)
                    if flag: 
                        return True
            return False
        return dfs(0,0)
