class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        # steps = {1:{1:1,3:2,5:4}, 2:{2:2,5:3,6:1}, 3:{1:3,4:2,6:4}, 4:{2:4,3:3,4:1}}
        steps = {1:[(0,1), (0,-1)], 2:[(1,0), (-1,0)], 3:[(0,-1),(1,0)], 4:[(0,1),(1,0)], 5:[(-1,0),(0,-1)], 6:[(0,1),(-1,0)]}
        start = grid[0][0]
        q = [(0,0)]
        steped = set([(0,0)])
        while len(q)>0:
            qq = []
            for x,y in q:
                for dx,dy in steps[grid[x][y]]:
                    if x+dx>=0 and y+dy>=0 and x+dx<len(grid) and y+dy<len(grid[0]) and (x+dx,y+dy) not in steped:
                        if (-dx,-dy) in steps[grid[x+dx][y+dy]]:
                            qq.append((x+dx, y+dy))
                            steped.add((x+dx, y+dy))
                            # print(steped)
            q = qq
        if (len(grid)-1,len(grid[0]) - 1) not in steped:
            return False
        return True
