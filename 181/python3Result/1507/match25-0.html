class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        t = {1: {(0,-1):(0,-1),(0,1):(0,1)},
2: {(1,0):(1,0),(-1,0):(-1,0)},
3: {(0,1):(1,0),(-1,0):(0,-1)},
4: {(0,-1):(1,0),(-1,0):(0,1)},
5: {(0,1):(-1,0),(1,0):(0,-1)},
6: {(1,0):(0,1),(0,-1):(-1,0)},}
        
        
        m = len(grid)
        n = len(grid[0])
        
        def dfs(r,c, inp):
            if not 0<=r<len(grid) or not 0<=c<len(grid[0]):
                return False
            if inp not in t[grid[r][c]]:
                return False
            if r==m-1 and c==n-1:
                return r, c
            o = t[grid[r][c]][inp]
            res = dfs(r+o[0], c+o[1], o)
            if not res:
                return r, c
            return res
        
        for k in t[grid[0][0]]:
            if dfs(0,0,k) == (m-1, n-1):
                return True
        return False
