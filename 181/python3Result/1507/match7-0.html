class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        if m == 1 and n == 1:
            return True
        
        dirs = {1 : [(0, -1), (0, 1)], 2 : [(-1, 0), (1, 0)], 3 : [(0, -1), (1, 0)], 4 : [(0, 1), (1, 0)], 5 : [(0, -1), (-1, 0)], 6 : [(0, 1), (-1, 0)]}
        queue = [(0, 0)]
        visited = {(0, 0)}
        while queue:
            i, j = queue.pop(0)
            for ki, kj in dirs[grid[i][j]]:
                ni, nj = i + ki, j + kj
                if 0 <= ni < m and 0 <= nj < n and (- ki, - kj) in dirs[grid[ni][nj]] and (ni, nj) not in visited:
                    if (ni, nj) == (m - 1, n - 1):
                        return True
                    queue.append((ni, nj))
                    visited.add((ni, nj))
        return False
