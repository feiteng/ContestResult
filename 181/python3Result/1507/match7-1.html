class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        
        
        
        nxt 
= {
            1:{(-1,0):[], (0,-1):[1,4,6], (0,1):[1,3,5], (1,0):[]},
            2:{(-1,0):[2,3,4], (0,-1):[], (0,1):[], (1,0):[2,5,6]},
            3:{(-1,0):[], (0,-1):[1,4,6], (0,1):[], (1,0):[2,5,6]},
            4:{(-1,0):[], (0,-1):[], (0,1):[1,3,5], (1,0):[2,5,6]},
            5:{(-1,0):[2,3,4], (0,-1):[1,4,6], (0,1):[], (1,0):[]},
            6:{(-1,0):[2,3,4], (0,-1):[], (0,1):[1,3,5], (1,0):[]}
        }
        
        m, n = len(grid), len(grid[0])
        if (m,n) == (1,1):
            return True
        level = {(0, 0)}
        visit = {(0, 0)}
        while level:
            # print(level)
            nextlevel = set()
            for pi,pj in level:
                for ci,cj in [(pi-1,pj), (pi,pj-1), (pi,pj+1), (pi+1,pj)]:
                    if 0 <= ci < m and 0 <= cj < n and (ci,cj) not in visit and grid[ci]
[cj] in nxt[grid[pi][pj]][ci-pi,cj-pj]:
                        if (ci,cj) == (m-1,n-1):
                            return True
                        nextlevel.add((ci,cj))
                        visit.add((ci,cj))
            level = nextlevel
        return False
