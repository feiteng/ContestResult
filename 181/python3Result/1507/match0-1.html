class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        # direction: 0 1 2 3: up left down right
        m = len(grid)
        n = len(grid[0]) if m > 0 else 0
        st = [{} for _ in range(7)]
        st[1][1] = [1,4,6] 
        st[1][3] = [1,3,5]
        st[2][0] = [2,3,4]
        st[2][2] = [2,5,6]
        st[3][1] = [1,4,6]
        st[3][2] = [2,5,6]
        st[4][2] = [2,5,6]
        st[4][3] = [1,3,5]
        st[5][0] = [2,3,4]
        st[5][1] = [1,4,6]
        st[6][0] = [2,3,4]
        st[6][3] = [1,3,5]
        
#         def find(x, y):
#             res = []
#             dirs = [[-1,0], [0,-1],[1,0],[0,1]]
            
#             for i, d in enumerate(dirs):
#                 nx, ny = x + d[0], y + d[1]
#                 if 0<=nx<m and 0<=ny<n:
#                     res.append(i)
#             return res
        # bfs
        visit = [[False]*n for _ in range(m)]
        dirs = [[-1,0], [0,-1],[1,0],[0,1]]
        
        x, y = 0, 0
        qu = [(0,0)]
        
        while len(qu) > 0:
            x, y = qu[0]
            visit[x][y] = True
            qu.pop(0)
            if x == m-1 and y == n-1:
                return True
            # pos_dirs = find(x, y)
            rd_dirs = st[grid[x][y]].keys()
            # print('keys', x, y, rd_dirs)
            
            for rd_d in rd_dirs:
                nx, ny = x + dirs[rd_d][0], y + dirs[rd_d][1]
                if 0<=nx<m and 0<=ny<n and grid[nx][ny] in st[grid[x][y]][rd_d] and not visit[nx][ny]:
                    qu.append((nx, ny))
            
        return False
