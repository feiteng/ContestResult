class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        maps = {1: [(0,-1),(0,1)], 2: [(-1,0),(1,0)], 3: [(0,-1),(1,0)], 4: [(1,0),(0,1)], 5: [(-1,0),(0,-1)], 6: [(-1,0),(0,1)] }
        R, C = len(grid), len(grid[0])
        queue = collections.deque([(0, 0)])
        seen = set([(0,0)])
        while queue:
            x, y = queue.popleft()
            if x == R-1 and y == C-1:
                return True
            for i, j in maps[grid[x][y]]:
                m, n = i+x, j+y
                if 0 <= m < R and 0 <= n < C and (m, n) not in seen:
                    for a,b in maps[grid[m][n]]:
                        if 0 <= a+m < R and 0 <= b+n < C:
                            if a+m == x and b+n == y:
                                queue.append((m, n))
                                seen.add((m,n))
                                break
        return False
