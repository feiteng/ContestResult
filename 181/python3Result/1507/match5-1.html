class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        dic 
= {1: [[0, -1], [0, 1]], 2: [[1, 0], [-1, 0]], 3: [[0, -1], [1, 0]], 4: [[1, 0], [0, 1]], 5: [[0, -1], [-1, 0]], 6: [[-1, 0], [0, 1]]}
        valid = {1: [1, 3, 4, 5, 6], 2: [2, 3, 4, 5, 6], 3: [1, 2, 4, 5, 6], 4: [1, 2, 3, 5, 6], 5: [1, 2, 3, 4, 6], 6: [1, 2, 3 ,4 ,5]}
        visited = set()
        m, n = len(grid), len(grid[0])
        visited.add((0, 0))
        q 
= [[0, 0]]
        while q:
            r, c = q.pop()
            for d in dic[grid[r][c]]:
                new_r, new_c = r + d[0], c + d[1]
                if 0 <= new_r < m and 0 <= new_c < n and (new_r, new_c) not in visited and grid[new_r][new_c] in valid[grid[r][c]]:
                    visited.add((new_r, new_c))
                    q.append([new_r, new_c])
        return (m - 1, n -1) in visited
