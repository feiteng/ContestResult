class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        dir={
            1: [(0,0), (0,0), (0,1), (0,-1)],
            2: [(1,0), (-1,0), (0,0), (0,0)],
            3: [(0,0), (0,-1), (1,0), (0,0)],
            4: [(0,0), (0,1), (0,0), (1,0)],
            5: [(0,-1), (0,0), (-1,0), (0,0)],
            6: [(0,1), (0,0), (0,0), (-1,0)],
        }
        m=len(grid)
        n=len(grid[0])
        def dfs(x,y, lastp):
            if x==m-1 and y==n-1 and dir[grid[m-1][n-1]][lastp]!=(0,0): 
                return True
            if x<0 or x>=m: return False
            if y<0 or y>=n: return False
            
            dx,dy = dir[grid[x][y]][lastp]

            if (dx,dy) != (0,0):
                lastp = 0
                if (dx,dy)==(1,0): lastp=0
                if (dx,dy)==(-1,0): lastp=1
                if (dx,dy)==(0,1): lastp=2
                if (dx,dy)==(0,-1): lastp=3
                
                return dfs(x+dx, y+dy, lastp)
            return False

        if m==1 and n==1: return True
        ans=False
        for dx,dy in dir[grid[0][0]]:
            if (dx,dy)!=(0,0):
                if (dx,dy)==(1,0): lastp=0
                if (dx,dy)==(-1,0): lastp=1
                if (dx,dy)==(0,1): lastp=2
                if (dx,dy)==(0,-1): lastp=3
                v = dfs(dx,dy, lastp)
                ans = ans or v
            
        return ans
