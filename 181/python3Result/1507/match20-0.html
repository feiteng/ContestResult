class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        ways = {
            1: {'L':'R', 'R':'L'},
            2: {'U':'D', 'D':'U'},
            3: {'L':'D', 'D':'L'},
            4: {'R':'D', 'D':'R'},
            5: {'L':'U', 'U':'L'},
            6: {'U':'R', 'R':'U'}
        }
        
        m = len(grid)
        n = len(grid[0])
        
        checked = [[0] * n for i in range(m)]
        
        nxt = [[0, 0]]
        
        while nxt:
            #print(nxt, checked)
            i, j = nxt.pop(0)
            
            if i == m - 1 and j == n - 1:
                return True
            
            if checked[i][j] == 1:
                continue
            checked[i][j] = 1
            
            val = grid[i][j]
            for key in ways[val]:
                if key == 'L':
                    nxt_i, nxt_j = i, j-1
                elif key == 'R':
                    nxt_i, nxt_j = i, j+1
                elif key == 'U':
                    nxt_i, nxt_j = i-1, j
                else:
                    nxt_i, nxt_j = i+1, j

                if nxt_i < 0 or nxt_i >= m or nxt_j < 0 or nxt_j >= n:
                    continue
                    
                if key == 'R' and 'L' in ways[grid[nxt_i][nxt_j]]:
                    nxt.append([nxt_i, nxt_j])
                elif key == 'L' and 'R' in ways[grid[nxt_i][nxt_j]]:
                    nxt.append([nxt_i, nxt_j])
                elif key == 'U' and 'D' in ways[grid[nxt_i][nxt_j]]:
                    nxt.append([nxt_i, nxt_j])
                elif key == 'D' and 'U' in ways[grid[nxt_i][nxt_j]]:
                    nxt.append([nxt_i, nxt_j])
                    
        return False
