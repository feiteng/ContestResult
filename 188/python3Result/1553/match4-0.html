class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n=len(arr)
        ret=0
        for i in range(n):
            curr=arr[i]
            for j in range(i+1, n):
                curr^=arr[j]
                if curr==0:
                    ret+=j-i
                    
        return ret
