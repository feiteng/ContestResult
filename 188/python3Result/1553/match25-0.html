class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        p = [0]
        n = len(arr)
        for num in arr:
            v = p[-1]^num
            p += [v]
        ans = 0
        for i in range(n-1):
            for j in range(i+1,n):
                for k in range(j,n):
                    if p[j]^p[i] == p[k+1]^p[j]:
                        ans += 1
        return ans
