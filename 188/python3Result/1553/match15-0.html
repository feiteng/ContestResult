class Solution:
    def countTriplets(self, arr):
        ans = 0

        n = len(arr)
        for i in range(n):
            a = 0
            for j in range(i, n):
                a = a ^ arr[j]                
                b = 0
                for k in range(j+1, n):
                    b = b ^ arr[k]
                    if a == b:
                        ans += 1
        return ans
