class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        ans = 0
        n = len(arr)
        
for i in range(n - 1):
            for k in range(i + 1, n, 1):
                left = arr[i]
                right = 0
                for j in range(i + 1, k + 1, 1):
                    right = right ^ arr[j]
                if left == right:
                    ans += 1
                for j in range(i + 1, k, 1):
                    left = left ^ arr[j]
                    right = right ^ arr[j]
                    if left == right:
                        ans += 1
        return ans
