class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        cumxor = [0]
        for x in arr:
            cumxor.append(cumxor[-1]^x)
        n = len(arr)
        sol = 0
        for i in range(1, n+1):
            for j in range(i+1, n+1):
                for k in range(j, n+1):
                    sol += (cumxor[j-1] ^ cumxor[i-1]) == (cumxor[k] ^ cumxor[j-1])
        return sol
