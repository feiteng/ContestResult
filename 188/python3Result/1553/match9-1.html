class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        pfs = [0]
        for v in arr:
            pfs.append(v ^ pfs[-1])
        ans = 0
        for k in range(1, len(arr)):
            for i in range(k):
                for j in range
(i+1,k+1):
                    a = pfs[j]^pfs[i]
                    b = pfs[k+1]^pfs[j]
                    if a == b:
                        ans += 1
        return ans
