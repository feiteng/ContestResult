class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        acc = [0]
        for num in arr:
            acc.append(acc
[-1] ^ num)

        ans = 0
        for i in range(n):
            for j in range(i + 1, n):
                for k in range(j, n):
                    if (acc[j] ^ acc[i]) == (acc[k + 1] ^ acc[j]):
                        ans += 1
        
        return ans
