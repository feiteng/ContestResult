class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        for i in range(1, n):
            arr[i] ^= arr
[i-1]
        
        res = 0
        for i in range(n):
            for j in range(i+1, n):
                for k in range(j, n):
                    if i == 0:
                        a = arr[j-1]
                    else:
                        a = arr[j-1] ^ arr[i-1]
                    b = arr[k] ^ arr[j-1]
                    if a==b:
                        res += 1
        return res
