class Solution:
    def countTriplets(self, arr):
        n = len(arr)
        ans = 0
        
for i in range(n-1):
            a = 0
            for j in range(i, n):
                a ^= arr[j]
                b = 0
                for k in range(j+1, n):
                    b ^= arr[k]
                    if a == b:
                        ans += 1
        return ans
