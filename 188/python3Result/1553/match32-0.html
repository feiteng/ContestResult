class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        # count the number of arrays with xor zero, for each length
        length_to_xor_count = {}
        
        for i in range(len(arr)):
            xor_value = arr[i]
            for j in range(i+1, len(arr)):
                xor_value ^= arr[j]
                if xor_value == 0:
                    length = j - i + 1
                    if length in length_to_xor_count:
                        length_to_xor_count[length] += 1
                    else:
                        length_to_xor_count[length] = 1
                        
        tot = 0
        for length, cnt in length_to_xor_count.items():
            tot += (length - 1) * cnt
        return tot
