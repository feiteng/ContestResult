class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        xor = [arr[0]]
        n = len(arr)
        for i in range(1, n):
            xor.append(arr[i] ^ xor[-1])
        
        r = 0
        for i in range(n-1):
            for k in range(i+1, n):
                for j in range(i+1, k+1):
                    if i == 0:
                        a = xor[j-1]
                    else:
                        a = xor[j-1] ^ xor[i-1]
                    b = xor[k] ^ xor[j-1]
                    if a == b:
                        r += 1
        return r
