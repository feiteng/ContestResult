class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        ans = 0
        for i in range(n):
            cur = arr[i]
            for j in range(i + 1, n):
                cur ^= arr[j]
                if cur == 0:
                    ans += j - i
        return ans
