class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        tot = 0
        for i in range(len(arr)):
            cur = arr[i]
            for k in range(i+1, len(arr)):
                cur = cur^arr[k]
                if cur == 0:
                    tot += k-i
        return tot
