class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        pre = [0]
        for num in arr:
            pre.append(pre[-1] ^ num)
        
        # print(pre)
        
        ans = 0
        
        n = len(arr)
        for i in range(n):
            for j in range(i+1, n):
                for k in range(j, n):
                    a = pre[j] ^ pre[i]
                    b = pre[k+1] ^ pre[j]
                    if a == b:
                        # print(i, j, k)
                        ans += 1
        
        return ans
