from functools import reduce
from operator import xor


class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        one = reduce(xor, arr)
        starts, ends = [], []
        i, j = one, 0
        for k in arr:
            starts.append(i)
            ends.append(j)
            i ^= k
            j ^= k
        ends.append(one)

        count 
= 0
        for i in range(len(arr) - 1):
            for j in range(i + 1, len(arr)):
                for k in range(j, len(arr)):
                    a = one ^ starts[i] ^ ends[j]
                    b = one ^ starts[j] ^ ends[k + 1]
                    if a == b:
                        count += 1
        return count
