class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        xortill = [x for x in arr]
        for i in range(1, n):
            xortill[i] ^= xortill[i-1]
            
        ct = 0
        for i in range(n-1):
            for j in range(i+1, n):
                for k in range(j, n):
                    left = xortill[j-1]
                    if i != 0:
                        left ^= xortill[i-1]
                    right = xortill[k] ^ xortill[j-1]
                    if left == right:
                        ct += 1
        
        return ct
