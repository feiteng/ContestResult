class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        a = [[0] * n for i in range(n)]
        for i in range(n):
            for j in range(i, n):
                a[i][j] = (a[i][j - 1] if j > 0 else 0) ^ arr[j]
        ans=0
        for i in range(n):
            for j in range(i+1,n):
                for k in range(j,n):
                    if a[i][j-1]==a[j][k]: ans+=1
        return ans
