class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        xored = 0
        lst = []
        
        for x in arr:
            xored ^= x
            lst.append(xored)
        
        dic = {}
        n = len(arr)
        
        for i in range(n):
            
for j in range(i,n):
                dic[(i,j)] = lst[j] ^ lst[i-1] if i > 0 else lst[j]
        
        ans = 0
        
        for i in range(n):
            for j in range(i+1,n+1):
                for k in range(j,n):
                    if dic[(i,j-1)] == dic[(j,k)]:
                        ans += 1
        
        return ans
