class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        res = 0
        for i in range(n):
            for k in range(i+1, n):
                a = arr[i]
                b = 0
                for j in range(i+1, k+1):
                    b^=arr[j]
                if a == b:
                    res+=1
                for j in range(i+1, k):
                    a^=arr[j]
                    b^=arr[j]
                    if a == b:
                        res+=1
        return res
