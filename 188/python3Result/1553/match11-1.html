class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        ans 
= 0
        for i in range(len(arr)):
            temp = arr[i]
            for j in range(i+1, len(arr)):
                if j != i+1:
                    temp = temp ^ arr[j-1]
                new = arr[j]
                for k in range(j, len(arr)):
                    if k != j:
                        new = new ^ arr[k]
                    if new == temp:
                        ans += 1
        return ans
