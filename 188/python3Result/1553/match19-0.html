class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        xor = [0]
        size = len(arr)
        for i in range(size):
            xor.append(xor[-1] ^ arr[i])

        # print(xor)
        c = 0

        for i in range(size+1):
            for j in range(i + 1, size+1):
                for k in range(j + 1, size+1):
                    a = xor[i] ^ xor[j]
                    b = xor[j] ^ xor[k]
                    if a == b:
                        c += 1
                        # print(i, j, k)
        return c
