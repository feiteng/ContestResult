class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        c = 0
        for i in range(0,len(arr)):
            xor = 0
            
            for j in range(i+1,len(arr)):
                xor ^= arr[j-1]
                xor2 = 0
                for k in range(j,len(arr)):
                    xor2 ^= arr[k]
                    if i == 0 and j == 2 and k == 2:
                        pass
                        #print(xor,xor2)
                    if xor == xor2:
                        #print(i,j,k)
                        c += 1
        return c
