class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        s = [0]
        N = len(arr)
        for n in arr:
            s.append(n ^ s[-1])

        ans = 0    
        for i in range(N):
            for j in range(i+1, N):
                if s[j+1] ^ s[i] == 0:
        #             print(i, j)
                    for k in range(i+1, j+1):
        #                 print(i, k, j,( s[k] ^ s[i] , s[j+1] ^ s[k+1]))
                        if s[k] ^ s[i] == s[j+1] ^ s[k]:
                            ans += 1
        return ans
