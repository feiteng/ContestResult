class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        s = [0] * (n + 1)
        for i in range(n):
            s[i + 1] = s[i] ^ arr[i]
        ans = 0
        for i in range(n):
            for k in range(i + 1, n):
                if s[i] == s[k + 1]:
                    ans += k - i
        return ans
