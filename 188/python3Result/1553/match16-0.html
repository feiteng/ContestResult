class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        ans = 0
        L = collections.defaultdict(list)
        L[0].append(-1)
        s = 0
        for k, x in enumerate(arr):
            s ^= x
            for i in L[s]:
                ans += k-i-1
            L[s].append(k)
        return ans
