class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        n = len(arr)
        prefix = [0] * (n + 1)
        
        for i, a in enumerate(arr):
            prefix[i + 1] = prefix
[i] ^ a
        
        count = 0
        for i in range(n - 1):
            for j in range(i + 1, n):
                for k in range(j, n):
                    if prefix[i] ^ prefix[j] == prefix[j] ^ prefix[k + 1]:
                        count += 1
        return count
