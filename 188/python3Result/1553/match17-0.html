class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        cnt = 0
        n = len(arr)
        for i in range(n):
            a = arr[i]
            for j in range(i+1, n):
                b = 0
                for k in range(j, n):
                    b ^= arr[k]
                    if a == b:
                        cnt += 1
                    
                a ^= arr[j]
                
        return cnt
