class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        a = []
        for i in range(1, n+1):
            if i > max(target): break
            if i not in target:
                a.append("Push")
                a.append("Pop")
            else:
                a.append("Push")
        return a
