class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        res = []
        j = 0
        for i in range(1, n+1):
            if target[j] == i:
                res.append("Push")
                j += 1
            else:
                res += ["Push", "Pop"]
            if j == len(target):
                break
        return res
