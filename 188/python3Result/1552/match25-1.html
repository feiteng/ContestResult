class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        ans=[]
        j=0
        for i in range(1,n+1):
            ans.append('Push')
            if target[j]!=i:
                ans.append('Pop')
            else:
                j+=1
            if j>=len(target):
                break
        return ans
