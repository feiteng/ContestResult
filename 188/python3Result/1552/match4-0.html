class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        i = 1
        ans = []
        for x in target:
            while i < x:
                ans.append('Push')
                ans.append('Pop')
                i += 1
            ans.append('Push')
            i += 1
        return ans
