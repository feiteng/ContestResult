class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        ans = []
        now = 1
        for i in range(len(target)):
            while now < target[i]:
                ans.append('Push')
                ans.append('Pop')
                now += 1
            ans.append('Push')
            now += 1
        return ans
