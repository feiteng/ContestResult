class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        ans = []
        for i in range(n):
            if i + 1 > max(target):
                break
            if i + 1 in target:
                ans.append("Push")
            else:
                ans.append("Push")
                ans.append("Pop")
        return ans
