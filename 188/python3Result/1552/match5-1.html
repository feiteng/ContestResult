class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        ops = []
        i = 1
        for x in target:
            while i < x:
                ops.append('Push')
                ops.append('Pop')
                i += 1
            ops.append('Push')
            i += 1
        return ops
