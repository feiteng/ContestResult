class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        k = 1
        res = []
        for i in range(len(target)):
            while k < target[i]:
                res.append("Push")
                res.append("Pop")
                k += 1
            res.append("Push")
            k += 1
        return res
