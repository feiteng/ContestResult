from functools import lru_cache
class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        pizza = [row[::-1] for row in pizza][::-1]
        MOD = 10 ** 9 + 7
        R = len(pizza)
        C = len(pizza[0])
        cumSum = [[0 for c in range(C + 1)] for r in range
(R + 1)]
        
        for r in range(R):
            for c in range(C):
                cumSum[r][c] = cumSum[r - 1][c] + cumSum[r][c - 1] - cumSum[r-1][c-1] + int(pizza[r][c] == 'A')
        
       
        @lru_cache(maxsize=None)
        def numWays(r, c, k):
            if k == 0:
                if cumSum[r][c] != 0:
                    
                    return 1
                
return 0
            
            ways = 0
            for rr in range(r):
                if cumSum[r][c] - cumSum[rr][c]:
                    ways += numWays(rr, c, k - 1)
            
            for cc in range(c):
                if cumSum[r][c] - cumSum[r][cc]:
                    ways += numWays(r, cc, k - 1)
            return ways % MOD 
            
        return numWays(R - 1, C - 1, k - 1) % MOD
