class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        def cut(lx, ly, rx, ry, c, dp):
            #print(lx, ly, rx, ry, c)
            if c == 0:
                #print(lx, ly, rx, ry, apple[rx+1][ry+1] - apple[lx][ry+1] - apple[rx+1][ly] + apple[lx][ly])
                return int(apple[rx+1][ry+1] - apple[lx][ry+1] - apple[rx+1][ly] + apple[lx][ly]>0)
            if (lx, ly, rx, ry, c) not in dp:
                res = 0
                for i in range(lx, rx):
                    if apple[i+1][ry+1] - apple[i+1][ly] - apple[lx][ry+1] + apple[lx][ly] > 0:
                        res = (res + cut(i+1, ly, rx, ry, c-1, dp)) % M
                for i in range(ly, ry):
                    if apple[rx+1][i+1] - apple[rx+1][ly] - apple[lx][i+1] + apple[lx][ly] > 0:
                        res = (res + cut(lx, i+1, rx, ry, c-1, dp)) % M
                dp[lx, ly, rx, ry, c] = res
            return dp[lx, ly, rx, ry, c]
            
        
        M = 10**9 + 7
        m = len(pizza[0])
        n = len(pizza)
        apple = [[0]*(m+1) for _ in range(n+1)]
        for i in range(n):
            for v in range(m):
                inc = int(pizza[i][v]=='A')
                apple[i+1][v+1] = apple[i][v+1] + apple[i+1][v] - apple[i][v] + inc 
        return cut(0, 0, n-1, m-1, k-1, {})
