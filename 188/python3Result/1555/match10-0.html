class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        r, c = len(pizza), len(pizza[0])
        import collections
        cn = collections.defaultdict(int)
        for i in range(r):
            for j in range(c):
                cn[i, j] = -cn[i - 1, j - 1] + cn[i - 1, j] + cn[i, j - 1] + (pizza[i][j] == 'A')
        mod = 10 ** 9 + 7
        def cp(x1, y1, x2, y2):
            return cn[x2, y2] - cn[x2, y1 - 1] - cn[x1 - 1, y2] + cn[x1 - 1, y1 - 1]
        @lru_cache(None)
        def f(x1, y1, x2, y2, k):
            if k == 1:
                return 1 if cp(x1, y1, x2, y2) > 0 else 0
            ans = 0
            for i in range(x1 + 1, x2 + 1):
                if cp(x1, y1, i - 1, y2) > 0:
                    ans += f(i, y1, x2, y2, k - 1)
            for i in range(y1 + 1, y2 + 1):
                if cp(x1, y1, x2, i - 1) > 0:
                    ans += f(x1, i, x2, y2, k - 1)
            return ans % mod
        return f(0, 0, r - 1, c - 1, k)
