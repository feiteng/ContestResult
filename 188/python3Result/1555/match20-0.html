import collections, functools
class Solution:
    def ways(self, pizza, k: int) -> int:
        apple = [[0 for _ in range(len(pizza[0])+1)] for _ in range(len(pizza)+1)]
        for x in range(len(pizza)-1,-1,-1):
            for y in range(len(pizza[0])-1,-1,-1):
                apple[x][y] = apple[x+1][y] + apple[x][y+1] - apple[x+1][y+1] + (pizza[x][y] == 'A')
                
        @functools.lru_cache(None)
        def dfs(i, j, k):
            if apple[i][j] < k:
                return 0
            if k == 1 and apple[i][j]:
                return 1
            temp = 0
            for x in range(i, len(pizza)):
                if apple[i][j] - apple[x][j]:
                    temp += dfs(x, j, k-1)
            for y in range(j, len(pizza[0])):
                if apple[i][j] - apple[i][y]:
                    temp += dfs(i, y, k-1)
            return temp %(10**9 + 7)
        return dfs(0,0,k)
