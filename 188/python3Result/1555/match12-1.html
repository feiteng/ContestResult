class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        m, n = len(pizza), len(pizza[0])
        pfs 
= [[0]*(n+1) for _ in range(m+1)]
        for i in range(m):
            for j in range(n):
                pfs[i+1][j+1] = (pizza[i][j] == 'A') + pfs[i+1][j] + pfs[i][j+1] - pfs[i][
j]
        # print(pfs)
        # dp on top left corner + num cuts left
        
        def check(i,j,m,n):
            return (pfs[m][n] - pfs[i][n] - pfs[m][j] + pfs[i][j] > 0)
        
        @functools.lru_cache(None)
        def solve(i, j, k):
            if k == 0:
                #print (i, j, 0, pfs[m][n], - pfs[i][n], - pfs[m][j], pfs[i][j])
                return check(i,j,m,n)
            amt = 0
            for i2 in range(i+1, m):
                if check(i,j,i2,n):
                    amt += solve(i2, j, k-1)
            for j2 in range(j+1, n):
                if check(i,j,m,j2):
                    amt += solve(i, j2, k-1)
            # print(i,j,k, amt)
            return amt
        
        ans = solve(0,0,k-1)
        return ans % (10**9+7)
