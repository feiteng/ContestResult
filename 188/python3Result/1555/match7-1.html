from functools import lru_cache

MOD = 10**9 + 7



class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        self.pizza = pizza
        self.height = len(pizza)
        self.width = len(pizza[0])

        return self.helper(k, 0, 0)

    @lru_cache(maxsize=None)
    def helper(self, k, row, col):
        if k < 1:
            return 0
        if k == 1:
            if self.good(row, self.height, col, self.width):
                return 1
            
return 0

        count = 0
        for i in range(row + 1, self.height):
            if self.good(row, i, col, self.width):
                count += self.helper(k - 1, i, col)
        for i in range(col + 1, self.width):
            if self.good(row, self.height, col, i):
                count += self.helper(k - 1, row, i)
        return count % MOD

    @lru_cache(maxsize=None)
    def good(self, rs, re, cs, ce):
        for i in range(rs, re):
            for j in range(cs, ce):
                if self.pizza[i][j] == 'A':
                    return True
        return False
