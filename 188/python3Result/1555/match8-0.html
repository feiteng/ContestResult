class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        mod = 10 ** 9 + 7
        m, n = len(pizza), len(pizza[0])
        p2 = [[0] * n for _ in range(m)]
        for i, j in itertools.product(range(m), range(n)):
            p2[i][j] = 1 if pizza[i][j] == 'A' else 0
        co = [[0] * (n + 1) for _ in range(m + 1)]
        for i, j in itertools.product(range(m), range(n)):
            co[i+1][j+1] = p2[i][j]+co[i+1][j] + \
                co[i][j+1] - co[i][j]

        def f(i2, j2, i1, j1):
            return co[i2][j2]-co[i2][j1]-co[i1][j2]+co[i1][j1]

        @functools.lru_cache(None)
        def dfs(i, j, c):
            if c == 1:
                return 1 if f(m, n, i, j) > 0 else 0
            ans = 0
            for row in range(i + 1, m):
                if f(row, n, i, j) > 0:
                    ans += dfs(row, j, c - 1)
            for col in range(j + 1, n):
                if f(m, col, i, j) > 0:
                    ans += dfs(i, col, c - 1)
            return ans

        return dfs(0, 0, k) % mod
