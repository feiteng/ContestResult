from functools import lru_cache
from typing import List


class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        self.p = pizza
        self.m = len(pizza)
        self.n = len(pizza[0])
        res = self.helper(0, 0, k)
        return res % 1000000007

    @lru_cache(maxsize=None)
    def helper(self, i, j, k):
        if i == self.m or j == self.n:
            return 0
        if k == 1 and self.valid(i, j, self.m, self.n):
            return 1
        res = 0
        for a in range(i + 1, self.m):
            if self.valid(i, j, a, self.n):
                res += self.helper(a, j, k-1)

        for b in range(j+1, self.n):
            if self.valid(i, j, self.m, b):
                res += self.helper(i, b, k-1)

        return res

    @lru_cache(maxsize=None)
    def valid(self, i, j, a, b):
        for x in range(i, a):
            for y in range(j, b):
                if self.p[x][y] == 'A':
                    # print('v')
                    return True
        return False
