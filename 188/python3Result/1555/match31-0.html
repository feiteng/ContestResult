from functools import lru_cache

class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        
        if k == 1:
            return 1
        
        
        mod = 1000000007
        M, N = len(pizza), len(pizza[0])
        
        
        @lru_cache(None)
        def hasApple(i0, i1, j0, j1):
            for i in range(i0, i1):
                for j in range(j0, j1):
                    if pizza[i][j] == "A":
                        return True
                    
            return False
        
        
        @lru_cache(None)
        def cut(i0, j0):
            si, ei, sj, ej = 0,0,0,0
            a, b = True, True
            
            for i in range(i0, M):
                for j in range(j0, N):
                    if pizza[i][j] == 'A':
                        if a:
                            si = i
                            ei = i
                            a = False
                            break
                        else:
                            ei = i
                            break
                            
            for j in range(j0, N):
                for i in range(i0, M):
                    if pizza[i][j] == 'A':
                        if b:
                            sj = j
                            ej = j
                            b = False
                            break
                        else:
                            ej = j
                            break
            
            return ei-si + ej-sj
                
        
        @lru_cache(None)
        def helper(i0, j0, piece):
            
            if piece == k-1:
                return cut(i0, j0)
            
            rtn = 0
            
            for i in range(i0+1, M):
                if hasApple(i0, i, j0, N):
                    rtn += helper(i, j0, piece+1)
            
            for j in range(j0+1, N):
                if hasApple(i0, M, j0, j):
                    rtn += helper(i0, j, piece+1)
                    
            return rtn
        
        return helper(0, 0, 1) % mod
