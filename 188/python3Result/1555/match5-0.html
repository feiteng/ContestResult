class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        m = len(pizza)
        n = len(pizza[0])
        apples = [[0]*(n+1) for _ in range(m+1)]
        for i in range(m):
            for j in range(n):
                apples[i+1][j+1] = apples[i][j+1] + apples[i+1][j] - apples[i][j] + (pizza[i][j] == 'A')
        MOD = 10**9 + 7
        # print(apples)
        @functools.lru_cache(None)
        def helper(start_row, start_col, left_cut):
            if left_cut == 0:
                return int(apples[m][n] + apples[start_row][start_col] - apples[m][start_col] - apples[start_row][n] > 0)
            res = 0
            for i in range(start_row, m-1):
                if apples[i+1][n] + apples[start_row][start_col] - apples[i+1][start_col] - apples[start_row][n] > 0:
                    res += helper(i+1, start_col, left_cut - 1)
            for j in range(start_col, n-1):
                if apples[m][j+1] + apples[start_row][start_col] - apples[m][start_col] - apples[start_row][j+1] > 0:
                    res += helper(start_row, j + 1, left_cut - 1)
                    
            return res % MOD
        
        return helper(0, 0, k - 1)
