class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        
        mod = 10 ** 9 + 7
        m, n = len(pizza), len(pizza[0])
        A = [[0] * (n+1) for _ in range(m+1)]
        
        for i in range(m):
            for j in range(n):
                A[i+1][j+1] = A[i][j+1] + A[i+1][j] - A[i][j] + (pizza[i][j] == 'A')
        if A[m][n] < k:
            return 0
        
        @functools.lru_cache(None)
        def dp(i, j, cut):
            
            if A[m][n] - A[i][n] - A[m][j] + A[i][j] == 0:
                return 0
            if cut == 0:
                return 1
            
            return (sum(dp(nexti, j, cut-1) for nexti in range(i+1, m) if A[nexti][n] - A[i][n] - A[nexti][j] + A[i][j] > 0 and A[m][n] - A[nexti][n] - A[m][j] + A[nexti][j] > 0) + sum(dp(i, nextj, cut-1) for nextj in range(j+1, n) if A[m][nextj] - A[i][nextj] - A[m][j] + A[i][j] > 0 and A[m][n] - A[i][n] - A[m][nextj] + A[i][nextj] > 0)) % mod\
        
        return dp(0, 0, k-1) % mod
