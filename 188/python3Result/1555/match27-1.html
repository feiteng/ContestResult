class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        m, n = len(pizza), len(pizza[0])
        
        @lru_cache(None)
        def row(i,j1,j2):
            return any('A' in pizza[r][j1:j2] for r in range(i,m))
        
        @lru_cache(None)
        def col(j, i1, i2):
            return any('A' in pizza[r][j:] for r in range(i1,i2))
        
        
        @lru_cache(None)
        def helper(i,j,k):
            if (m-i)*(n-j)<k: return 0
            
if k==1: return 1 if col(j,i,m) else 0
            ans = 0
            for r in range(i+1,m):
                if col(j,i,r): ans += helper(r,j,k-1)
            for c in range(j+1,n):
                if row(i,j,c): ans += helper(i,c,k-1)
            return ans % (10**9+7)
        
        return helper(0,0,k)
