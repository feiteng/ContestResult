class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        mod = 10 ** 9 + 7
        m, n = len(pizza), len(pizza[0])
        apple = [[0] * n for i in range(m)] # apple[i][j]: apple count starting from row i and col j
        for i in range(m - 1, -1, -1):
            for j in range(n - 1, -1, -1):
                apple[i][j] = (1 if pizza[i][j] == 'A' else 0) + (apple[i][j + 1] if j < n - 1 else 0) + (apple[i + 1][j] if i < m - 1 else 0) - (apple[i + 1][j + 1] if i < m - 1 and j < n - 1 else 0)
        
        # num of ways to divide pizza into k pieces, starting from row i and col j
        from functools import lru_cache
        @lru_cache(None)
        def getWays(i: int, j: int, k: int) -> int:
            if apple[i][j] < k:
                return 0
            if k == 1:
                return 1
            
            ans = 0
            for ii in range(i + 1, m):
                if apple[ii][j] < apple[i][j]:
                    ans += getWays(ii, j, k - 1)
            for jj in range(j + 1, n):
                if apple[i][jj] < apple[i][j]:
                    ans += getWays(i, jj, k - 1)
            return ans % mod
        
        return getWays(0, 0, k)
