class Solution:
    def ways(self, pizza: List[str], k: int) -> int:
        n, m = len(pizza), len(pizza[0])
        acc = [[0] * (m + 1) for _ in range(n + 1)]
        for i in range(n):
            ac = 0
            for j in range(m):
                c = pizza[i][j] == 'A'
                ac += c
                acc[i+1][j+1] = ac + acc[i][j+1]
                
        print(acc)
        def v(x, y):
            return acc[n][m] - acc[n][y] - acc[x][m] + acc[x][y]
        R = 10 ** 9 + 7 
        @lru_cache(None)
        def dfs(x, y, k):
            vv = v(x, y)
            if k == 1:
                return int(vv != 0)
            if vv == 0: return 0
            
            c = 0
            for i in range(x + 1, n):
                if acc[i][m] - acc[x][m] - acc[i][y] + acc[x][y]: 
                    c += dfs(i, y, k-1)
            for j in range(y + 1, m):
                if acc[n][j] - acc[n][y] - acc[x][j] + acc[x][y]: 
                    c += dfs(x, j, k-1)
            
            return c % R
        
        
        return dfs(0, 0, k)
