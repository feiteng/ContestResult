class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        graph = collections.defaultdict(list)
        for edge in edges:
            graph[edge[0]].append(edge[1])
        def dfs(root):
            cost = 0
            for child in graph[root]:
                cost += dfs(child)
            #print(root, cost)
            if root == 0:
                return cost
            if hasApple[root]:
                return cost + 2
            return cost + (2 if cost else 0)
        return dfs(0)
