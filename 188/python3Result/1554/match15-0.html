class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        g = [[] for i in range(n)]
        for u,v in edges :
            g[u].append(v)
            
        def dfs(u=0) :
            res = 0
            for v in g[u] :
                ans = dfs(v)
                if ans > 0 or hasApple[v]:
                    res = res + ans + 1
            return res
        
        return dfs()*2
