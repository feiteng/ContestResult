class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        
        graph = defaultdict(list)
        for u, v in edges:
            graph[u].append(v)
            
        def sumTree(node, parent):
            num = 0
            edges = 0
            for child in graph[node]:
                if child != parent:                
                    cNum, cEdges = sumTree(child, node)
                    num += cNum
                    edges += cEdges
            
            if hasApple[node]:
                num += 1
            if hasApple[node] or num:
                if parent is not None:
                    edges += 2
            return num, edges
        
        return sumTree(0, None)[1]
