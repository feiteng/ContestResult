class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        e = [[] for i in range(n)]
        tag = [False for i in range(n)]
        for a, b in edges:
            e[a].append(b)
            e[b].append(a)

        def dfs(r):
            if tag[r]:
                return 0, False
            ans = 0
            tag[r] = True
            have = hasApple[r]
            for n in e[r]:
                cur, has = dfs(n)
                if has:
                    ans += 2 + cur
                have = have or has
            return ans, have
        return dfs(0)[0]
