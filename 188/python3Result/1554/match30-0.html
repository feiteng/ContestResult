from collections import defaultdict

class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        a = defaultdict(list)
        
        for edge in edges:
            a[edge[0]].append(edge[1])

        def process(node):
            if node not in a:
                return 0
            
            result = 0
            for child in a[node]:
                if child in a:
                    child_value = process(child)
                    if child_value > 0:
                        result += (child_value + 2)
                    elif hasApple[child]:
                        result += 2
                elif hasApple[child]:
                    result += 2
                    
            return result
        
        return process(0)
