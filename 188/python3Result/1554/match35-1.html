class Solution:
    def minTime(self, n: int, edges, hasApple) -> int:
        p = [-1] * n
        for edge in edges:
            x, y = edge
            p[y] = x

        f = [0] * n
        for i in reversed(range(n)):
            if p[i] != -1 and hasApple
[i]:
                f[p[i]] += f[i] + 2
                hasApple[p[i]] = True
        return f[0]
