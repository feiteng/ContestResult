class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        vs = collections.defaultdict(list)
        for st, ed in edges:
            vs[st].append(ed)
        def dfs(idx):
            time = 0
            for j in vs[idx]:
                t = dfs(j)
                if t >= 0: time += t + 2
            return time if time > 0 or hasApple[idx] else -1
        return max(dfs(0), 0)
