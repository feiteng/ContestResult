class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        mp = [[] for _ in range(n)]
        for x,y in edges:
            mp[x].append(y)
        def dfs(node):
            res = 0
            has = False
            if hasApple[node]:
                has = True
            for nxt in mp[node]:
                l1,l2 = dfs(nxt)
                if l1:
                    res += 2+l2
                    has = True
            return has, res
        return dfs(0)[1]
