class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        link = collections.defaultdict(set)
        for L,R in edges:
            link[L].add(R)
            link[R].add(L)
        parent = collections.defaultdict(int)
        BFS = [0]
        for L in BFS:
            for R in link[L]:
                BFS.append(R)
                link[R].remove(L)
                parent[R] = L
        BFS.reverse()
        BFS.pop()
        ans = collections.defaultdict(int)
        for R in BFS:
            if hasApple[R]:
                ans[parent[R]] += 2 + ans[R]
                hasApple[parent[R]]=True
        return ans[0]
