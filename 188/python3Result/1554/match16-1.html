class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        fr = [[] for _ in range(n)]
        for e in edges:
            fr[e[0]].append(e[1])
        
        def search(i):
            ans = 0
            for j in fr[i]:
                tmp = search(j)
                if tmp > 0 or hasApple[j]:
                    ans += tmp + 1
            return ans
            
        return 2 * search(0)
