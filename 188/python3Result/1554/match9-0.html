class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        children = defaultdict(list)
        for edge in edges:
            children[edge[0]].append(edge[1])

        def helper(start):
            has = hasApple[start]
            moves = 0
            for child in children[start]:
                childMoves, childHas = helper(child)
                if childHas:
                    moves += 2 + childMoves
                    has = True
            return moves, has
        
        return helper(0)[0]
