from collections import defaultdict
class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        s2t = defaultdict(set)
        for s,t in edges:
            s2t[s].add(t)
            s2t[t].add(s)
            
        def dfs(s, pre):
            has = hasApple[s]
            path = 0
            for t in s2t[s]:
                if t != pre:
                    chas, cpath = dfs(t, s)
                    if chas:
                        has = True
                        path += cpath + 2
            return has, path
        finalHas, finalPath = dfs(0, -1)
        return finalPath
