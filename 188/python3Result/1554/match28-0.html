class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        d = collections.defaultdict(list)
        for s, t in edges:
            d[s].append(t)

        def dfs(x):
            if not d[x]:
                return 0, hasApple[x]

            ans, h = 0, hasApple[x]
            for c in d[x]:
                a2, h2 = dfs(c)
                h |= h2
                if h2:
                    ans += 2 + a2
            return ans, h
        return dfs(0)[0]
