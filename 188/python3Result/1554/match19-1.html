class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        neighbors = []
        for i in range(n):
            neighbors.append([])
        
for edge in edges:
            neighbors[edge[0]].append(edge[1])
            neighbors[edge[1]].append(edge[0])
            
        def dfs(vertex, parent):
            res = 0
            for neighbor in neighbors[vertex]:
                if neighbor != parent:
                    temp = dfs(neighbor, vertex)
                    if temp != -1:
                        res += temp + 2
            if res == 0 and hasApple[vertex] is False:
                
return -1
            return res
        res = dfs(0,-1)
        if res == -1:
            return 0
        return res
