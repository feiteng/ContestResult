class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        E = [[] for i in range(n)]
        for u, v in edges:
            E[u] += [v]
            E[v] += [u]

        def dfs(u, parent):
            ans = 0
            for v in E[u]:
                if v != parent:
                    found, cost = dfs(v, u)
                    if found:
                        ans += cost + 1
            
            if ans == 0 and not hasApple[u]:
                return False, 0
            
            return True, ans

        return dfs(0, -1)[1] * 2
