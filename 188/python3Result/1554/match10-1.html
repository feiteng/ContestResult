class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        tree = collections.defaultdict(list)
        for src, dst in edges:
            tree[src].append(dst)
        
        def helper(node):
            count = 0
            for child in tree[node]:
                count += helper(child)
                
            if node != 0 and (count > 0 or hasApple[node]):
                count += 2
            
            return count
        
        return helper(0)
