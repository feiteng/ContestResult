class Node(object):
    def __init__(self, has_apple, val):
        self
.has_apple = has_apple
        self.val = val
        self.children = []


class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        head = self.build_tree(edges, hasApple)
        res = self.get_time(head)
        return 0 if res == -1 else res
        
    def get_time(self, node):
        if not node:
            return -1
        total = 0
        for child in node.children:
            t = self.get_time(child)
            if t != -1:
                total += 2 + t
        #print(node.val, node.has_apple, total)
        if total == 0:
            return 0 if node.has_apple else -1
        return total

        q = [0]
        visited = {0}
        while q:
            newq = []
            for i in q:
                for next_i in D[i]:
                    if next_i not in visited:
                        newq.append(next_i)
