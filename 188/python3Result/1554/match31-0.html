class Solution:
    def minTime(self, N: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        
        graph = collections.defaultdict(list)
        for u,v in edges:
            graph[u].append(v)
        
        def dfs(root):      
            #print("dfs: ", root)
            root_has_apple = hasApple[root]
            root_cost = 0
            
            for child in graph[root]:
                child_has_apple, child_cost = dfs(child)
                root_has_apple = root_has_apple or child_has_apple
                if child_has_apple:
                    root_cost += (2 + child_cost)
                
            if root_has_apple:
                #print("dfs: ", root, "result: ", True, root_cost)
                return True, root_cost
            else:
                #print("dfs: ", root, "result: ", False, 0)
                return False, 0
            
            
        has_apple, cost = dfs(0)
        return cost
