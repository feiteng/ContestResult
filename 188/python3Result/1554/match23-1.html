class Solution:
    def minTime(self, n: int, edges: List[List[int]],
                hasApple: List[bool]) -> int:
        maps = collections.defaultdict(list)
        for e in edges:
            maps[e[0]].append(e[1])

        def dfs(i):
            childHasApple = hasApple[i]
            for nex in maps[i]:
                childHasApple |= dfs(nex)
            if not childHasApple:
                del maps[i]
            return childHasApple

        dfs(0)
        return max(0, 2 * (len(maps) - 1))
