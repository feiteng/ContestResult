class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        d=defaultdict(list)
        for a,b in edges:
            d[a].append(b)
            d[b].append(a)
        visited=set()
        def dfs(node,parent):
            flg=hasApple[node]
            res=0
            for nxt in d[node]:
                if nxt==parent:
                    continue
                steps,flg2=dfs(nxt,node)
                if flg2:
                    res+=steps+2
                    flg=True
            return res,flg
        return dfs(0,-1)[0]
