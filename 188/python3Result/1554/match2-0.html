class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        sons = [[] for _ in range(n)]
        for x, y in edges:
            sons[x].append(y)
        
        def calc(v):
            t = 0
            flag = hasApple[v]
            for w in sons[v]:
                tt, ff = calc(w)
                if ff:
                    t += tt + 2
                    flag = True
            return t, flag
        return calc(0)[0]
