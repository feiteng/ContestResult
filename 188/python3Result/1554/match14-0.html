from collections import defaultdict
class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        c_list = defaultdict(list)
        for e in edges:
            c_list[e[0]].append(e[1])
        def dfs(i):
            tmp = 0
            for c in c_list[i]:
                tmp += dfs(c)
            if tmp == 0 and not hasApple[i]:
                return 0
            else:
                return tmp + 2
        return dfs(0) - 2 if dfs(0) > 0 else 0
