class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        G = [[] for _ in range(n)]
        for u, v in edges:
            G[u].append(v)
            G[v].append(u)
        
def dfs(u, pa):
            apple = hasApple[u]
            ti = 0
            for v in G[u]:
                if v == pa: continue
                ch_ti, ch_apple = dfs(v, u)
                if ch_apple:
                    apple = True
                    ti += ch_ti + 2
            return ti, apple
        ti, _ = dfs(0, -1)
        return ti
