class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        
        # Turn this edge list into an adjacency list.
        nbs = collections.defaultdict(list)
        
        for e in edges:
            nbs[e[0]].append(e[1])
            nbs[e[1]].append(e[0])
            
        def call(parent, node):
        
            # Retrieve all the apples starting from node.
            # Return int in time.
            neighbors = nbs[node]
            
            totalTime = 0
            
            for n in neighbors:
                if n != parent:
                    
                    # Get apples here.
                    time = call(node, n)
                    if time >= 0:
                        totalTime += time + 2
            
            if totalTime == 0:
                if hasApple[node]:
                    return 0
                return -1
            
            return totalTime
        
        ans = call(0, 0)
        if ans == -1:
            return 0
        return ans
