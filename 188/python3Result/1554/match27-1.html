class Solution:
    def minTime(self, n: int, edges: List[List[int]], hasApple: List[bool]) -> int:
        adjs=[[] for _ in range(n)]
        for a, b in edges:
            adjs[a].append(b)
            adjs[b].append(a)
        def dfs(x, pre):
            ans=0 if hasApple[x] else -1
            for y in adjs[x]:
                if y==pre: continue
                vy=dfs(y,x)
                if vy>=0:
                    if ans<0:ans=0
                    ans+=2+vy
            return ans if x>0 else max(ans, 0)
        return dfs(0,-1)
