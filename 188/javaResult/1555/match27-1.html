class Solution {
    
    private static final int MOD = (int) 1e9 + 7;
    
    public int ways(String[] pizza, int k) {
        char[][] p = 
new char[pizza.length][];
        for (int i = 0; i < p.length; i++) {
            p[i] = pizza[i].toCharArray();
        }
        cache = new Integer[k + 1][pizza.length + 1][pizza[0].length() + 1];
        
        return ways(p, k, 0, 0);
    }
    
    private Integer[][][] cache;
    
    private int ways(char[][] pizza, int k, int r, int c) {
        if (k == 1) 
{
            for (int i = r; i < pizza.length; i++) {
                for (int j = c; j < pizza[0].length; j++) {
                    if (pizza[i][j] == 'A') {
                        return 1;
                    }
                }
            }
            return 0;
        }
        Integer cached = cache[k][r][c];
        
if (cached != null) {
            return cached;
        }
        
        int count = 0;
        // cut row
        boolean rowHasApple = false;
        for (int row = r; row < pizza.length; row++) {
            if (!rowHasApple) {
                for (int col = c; col < pizza[0].length; col++) {
                    if (pizza[row][col] == 'A') {
                        rowHasApple = true;
                        break;
                    }
                }
            }
            if (rowHasApple) {
                count += ways(pizza, k - 1, row + 1, c);
                count = count % MOD;
            }
        }
        // cut col
        boolean colHasApple = false;
        
for (int col = c; col < pizza[0].length; col++) {
            if (!colHasApple) {
                for (int row = r; row < pizza.length; row++) {
                    if (pizza[row][col] == 'A') {
                        colHasApple = true;
                        break;
                    }
                }
            }
            if (colHasApple) {
                count += ways(pizza, k - 1, r, col + 1);
                count = count % MOD;
            }
        }
        return cache[k][r][c] = count;
    }
    
}
