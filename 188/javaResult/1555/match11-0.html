class Solution {
    public int ways(String[] pizza, int k) {
        int numRows = pizza.length;
        int numCols = pizza[0].length();
        boolean[][] a = new boolean[numRows][numCols];
        for(int i = 0; i < numRows; i++) {
            for(int j = 0; j < numCols; j++) {
                a[i][j] = (pizza[i].charAt(j)) == 'A';
            }
        }
        
        int[][] applesUnder = new int[numRows+1][numCols+1];
        for(int i = numRows-1; i >= 0; i--) {
            for(int j = numCols-1; j >= 0; j--) {
                int applesHere = 0;
                if(a[i][j]) {
                    applesHere = 1;
                }
                applesUnder[i][j] = applesUnder[i][j+1] + applesUnder[i+1][j] - applesUnder[i+1][j+1] + applesHere;
            }
        }
        
        int[][][] w = new int[k][numRows+1][numCols+1];
        
        //cutsLeft == 0
        for(int i = 0; i < numRows; i++) {
            for(int j = 0; j < numCols; j++) {
                if(applesUnder[i][j] > 0) {
                    w[0][i][j] = 1;
                }
            }
        }
        //cutsLeft > 0        
        for(int cutsLeft = 1; cutsLeft < k; cutsLeft++) {
            for(int i = numRows-1; i >= 0; i--) {
                for(int j = numCols-1; j >= 0; j--) {
                    for(int newI = i; newI < numRows; newI++) {
                        if(applesUnder[newI][j] < applesUnder[i][j]) {
                            w[cutsLeft][i][j] += w[cutsLeft-1][newI][j];
                            w[cutsLeft][i][j] = w[cutsLeft][i][j]%1000000007;
                        }                        
                    }
                    for(int newJ = j; newJ < numCols; newJ++) {
                        if(applesUnder[i][newJ] < applesUnder[i][j]) {
                            w[cutsLeft][i][j] += w[cutsLeft-1][i][newJ];
                            w[cutsLeft][i][j] = w[cutsLeft][i][j]%1000000007;
                        }
                    }
                }
            }
        }
                
        return w[k-1][0][0];
    }
}
