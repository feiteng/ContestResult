class Solution {
    private static final int MOD = (int) 1e9 + 7;
    
public int ways(String[] pizza, int k) {
        int m = pizza.length, n = pizza[0].length();
        
        boolean[][] row = new boolean[m + 1][n + 1];
        boolean[][] col = new boolean[m + 1][n + 1];
        int[][][] dp = new int[51][51][11];
        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                char c = pizza[i].charAt(j);
                row[i][j] = row[i][j + 1] || c == 'A';
                col[i][j] = col[i + 1][j] || c == 'A';
                dp[i][j][1] = ((dp[i + 1][j][1] == 1) || row[i][j]) ? 1 : 0;
            }
        }
        
        
        
for (int l = 2; l <= k; l++) {
            for (int i = m - 1; i >= 0; i--) {
                for (int j = n - 1; j >= 0; j--) {
                    // horizontal
                    boolean h = false;
                    
for (int c = i; c < m - 1; c++) {
                        h |= row[c][j];
                        if (h) {
                            dp[i][j][l] += dp[c + 1][j][l - 1];
                            dp[i][j][l] %= MOD;
                        }
                    }
                    boolean v = false;
                    
for (int c = j; c < n - 1; c++) {
                        v |= col[i][c];
                        if (v) {
                            dp[i][j][l] += dp[i][c + 1][l - 1];
                            dp[i][j][l] %= MOD;
                        }
                    }
                }
            }
        }
        
        return dp[0][0][k];
    }
}
