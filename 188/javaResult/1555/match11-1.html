class Solution {
    public int ways(String[] pizza, int k) {
        int n = pizza.length;
        int m = pizza[0].length();
        int[][] cp = new int[n + 1][m + 1];
        for (int i = n - 1; i >= 0; i--) {
            for (int u = m - 1; u >= 0; u--) {
                cp[i][u] = (
pizza[i].charAt(u) == 'A' ? 1 : 0) + cp[i + 1][u] + cp[i][u + 1] - cp[i + 1][u + 1];
            }
        }
        int[][][] dp = new int[n + 1][m + 1][k + 1];
        for (int i = 1; i <= k; i++) {
            for (int j = n - 1; j >= 0; j--) {
                for (int u = m - 1; u >= 0; u--) {
                    if (cp[j][u] >= i) {
                        if (i == 1) {
                            dp[j][u][i] = 1;
                        } else {
                            int vv = 0;
                            
for (int o = j + 1; o <= n - 1; o++) {
                                if (cp[j][u] > cp[o][u]) {
                                    vv = (vv + dp[o][u][i - 1]) % 1_000_000_007;
                                }
                            }

                            for (int o = u + 1; o <= m - 1; o++) {
                                if (cp[j][u] > cp[j][o]) {
                                    vv = (vv + dp[j][o][i - 1]) % 1_000_000_007;
                                }
                            }
                            dp[j][u][i] = vv;
                        }
                    }
                
}
            }
        }
        return dp[0][0][k];
    }

    public static void main(String[] args) {
        System.out.println(new Solution().ways(new String[]{"A..", "AAA", "..."}, 3));
    }
}
