class Solution {
    private static final int MOD = 1000000007;

    public int ways(String[] pizza, int k) {
        int rows = pizza.length;
        if (rows == 0) {
            return 0;
        }
        int cols = pizza[0].length();
        int[][][] dp = new int[rows][cols][k];
        int[][] verSum = new int[rows+1][cols+1];
        int[][] totSum = new int[rows][cols+1];
        for (int j = cols-1;j>=0;j--) {
            for (int i = rows-1;i>=0;i--) {
                verSum[i][j] = verSum[i+1][j] + (pizza[i].charAt(j) == 'A' ? 1 : 0);
            }
        }
        for (int i = rows-1;i>=0;i--) {
            for (int j = cols-1;j>=0;j--) {
                totSum[i][j] = verSum[i][j] + totSum[i][j+1];
            }
        }
        for (int i = 0;i<rows;i++) {
            for (int j = 0;j<cols;j++) {
                for (int k2 = 0;k2<k;k2++) {
                    dp[i][j][k2] = -1;
                }
            }
        }
        int result = find(dp, totSum, k, 0, 0);
        return result;
    }
    
    public int find(int[][][] dp, int[][] totSum, int k, int height, int width) {
        if (k == 1) {
            return totSum[height][width] > 0 ? 1 : 0;
        }
        if (dp[height][width][k-1] != -1) {
            return dp[height][width][k-1];
        }
        int result = 0;
        for (int i = height;i<dp.length-1;i++) {
            if(totSum[height][width] - totSum[i+1][width] > 0) {
                result = (result + find(dp, totSum, k-1, i + 1, width)) % MOD;
            }
        }
        for (int j = width;j<dp[0].length-1;j++) {
            if(totSum[height][width] - totSum[height][j+1] > 0) {
                result = (result + find(dp, totSum, k-1, height, j+1)) % MOD;
            }
        }
        dp[height][width][k-1] = result;
        return result;
    }
}
