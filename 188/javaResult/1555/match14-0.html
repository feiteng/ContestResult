class Solution {
    public int ways(String[] pizza, int k) {
        int n = pizza.length;
        int m = pizza[0].length();
        char[][] mat = new char[n][];
        for (int i = 0; i < n; i++) {
            mat[i] = pizza[i].toCharArray();
        }
        boolean[][][][] apple = new boolean[n][m][n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                apple[i][j][i][j] = mat[i][j] == 'A';
                for (int r = i; r < n; r++) {
                    for (int c = j; c < m; c++) {
                        apple[i][j][r][c] = mat[r][c] == 'A';
                        if (r-1 >= i) {
                            apple[i][j][r][c] |= apple[i][j][r-1][c];
                        }
                        if (c-1 >= j) {
                            apple[i][j][r][c] |= apple[i][j][r][c-1];
                        }
                    }
                }
            }
        }
        int[][][] dp = new int[n][m][k];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                dp[i][j][0] = apple[i][j][n-1][m-1] ? 1 : 0;
            }
        }
        for (int s = 1; s < k; s++) {
            for (int i = n-1; i >= 0; i--) {
                for (int j = m-1; j >= 0; j--) {
                    for (int r = i; r < n-1; r++) {
                        if (apple[i][j][r][m-1]) {
                            dp[i][j][s] += dp[r+1][j][s-1];
                            dp[i][j][s] %= 1000000007;
                        }
                    }
                    for (int c = j; c < m-1; c++) {
                        if (apple[i][j][n-1][c]) {
                            dp[i][j][s] += dp[i][c+1][s-1];
                            dp[i][j][s] %= 1000000007;
                        }
                    }
                }
            }
        }
        // for (int s = 0; s < k; s++) {
        //     for (int i = 0; i < n; i++) {
        //         for (int j = 0; j < m; j++) {
        //             System.out.print(dp[i][j][s] + " ");
        //         }
        //         System.out.println();
        //     }
        //     System.out.println();
        // }
        // System.out.println("------");
        return dp[0][0][k-1];
    }
}
