class Solution {
    public int ways(String[] pizza, int k) {
        int MOD = 1000000007;
        
int m = pizza.length, n = pizza[0].length();
        int[][] apples = new int[m][n];
        
        for (int i = m - 1; i >= 0; --i) {
            for (int j = n - 1; j >= 0; --j) {
                if (pizza[i].charAt(j) == 'A') apples[i][j] = 1;
                if (i < m - 1) {
                    apples[i][j] += apples[i + 1][j];
                }
                if (j < n - 1) {
                    apples[i][j] += apples[i][j + 1];
                }
                if (i < m - 1 && j < n - 1) {
                    apples[i][j] -= apples[i + 1][j + 1];
                }
            
}
        }
        
        int[][][] dp = new int[k + 1][m][n];
        for (int i = 0; i < m ; ++i) {
            for (int j = 0; j < n; ++j) {
                if (apples[i][j] > 0) dp[1][i][j] = 1
;
            }
        }
        
        // for (int i = 0; i < m; ++i) {
        //     System.out.println(Arrays.toString(apples[i]));
        // }
        
        for (int p = 2; p <= k; ++p) {
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; ++j) {
                    if (apples[i][j] < p) continue;
                    // horizontal cuts
                    
for (int l = i + 1; l < m; ++l) {
                        if (apples[l][j] >= p - 1 && apples[i][j] - apples[l][j] > 0) {
                            dp[p][i][j] =  (dp[p][i][j] + dp[p - 1][l][j]) % MOD;
                        } 
                    }
                    
                    // vertical cuts
                    for (int v = j + 1; v < n; ++v) {
                        if (apples[i][v] >= p - 1 && apples[i][j] - apples[i][v] > 0) {
                            dp[p][i][j] =  (dp[p][i][j] + dp[p - 1][i][v]) % MOD;
                        }
                    }
                }
            }
        }
        
        return dp[k][0][0];
    }
}
