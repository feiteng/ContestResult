class Solution {
    public int ways(String[] pizza, int k) {
        int n=pizza.length;
        int m=pizza[0].length();
        long[][][][][] dp=new long[n][m][n][m][k+1];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                for(int kk=i;kk<n;kk++){
                    for(int l=j;l<m;l++){
                        Arrays.fill(dp[i][j][kk][l],-1);
                    }
                }
            }
        }
        return (int)recur(pizza,dp,0,0,n-1,m-1,k);
    }
    public long recur(String[] pizza,long[][][][][] dp,int xl,int yl,int xr,int yr,int k){
        long mod=1000000007;
        if(dp[xl][yl][xr][yr][k]!=-1){
            return dp[xl][yl][xr][yr][k];
        }
        if(k==1){
            boolean ans=false;
            for(int i=xl;i<=xr;i++){
                for(int j=yl;j<=yr;j++){
                    if(pizza[i].charAt(j)=='A'){
                        ans=true;
                        break;
                    }
                }
                if(ans){
                    break;
                }
            }
            if(ans){
                return dp[xl][yl][xr][yr][1]=1;
            }else{
                return dp[xl][yl][xr][yr][1]=0;
            }
        }
        long val=0;
        boolean[] r=new boolean[xr-xl+1];
        boolean[] l=new boolean[yr-yl+1];
        for(int i=xl;i<=xr;i++){
            for(int j=yl;j<=yr;j++){
                if(pizza[i].charAt(j)=='A'){
                    r[i-xl]=true;
                    l[j-yl]=true;
                }
            }
        }
        for(int i=xl+1;i<=xr;i++){
            r[i-xl]=(r[i-xl]||r[i-xl-1]);
        }
        for(int i=yl+1;i<=yr;i++){
            l[i-yl]=(l[i-yl]||l[i-yl-1]);
        }
        for(int i=xl+1;i<=xr;i++){
            if(r[i-xl-1]){
                val=(val+recur(pizza,dp,i,yl,xr,yr,k-1))%mod;
            }
        }
        for(int i=yl+1;i<=yr;i++){
            if(l[i-yl-1]){
                val=(val+recur(pizza,dp,xl,i,xr,yr,k-1))%mod;
            }
        }
        return dp[xl][yl][xr][yr][k]=val;
    }
}
