class Solution {
    public static final long MOD = 1000000007L;
    
    public int ways(String[] pizza, int k) {
        int m = pizza.length, n = 
pizza[0].length();
        long[][][] cache = new long[m][n][k];
        int[][][][] hasApple = new int[m][n][m][n];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int l = 0; l < k; l++) {
                    cache[i][j][l] = -1L;
                }
            }
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int l = 0; l < m; l++) {
                    for (int q = 0; q < n; q++) {
                        hasApple[i][j][l][q] = -1;
                    }
                }
            }
        }
        
        int result = (int)(countWays(0, 0, 0, cache, m, n, k, hasApple, pizza) % MOD);
        return result;
    
}
    
    public long countWays(int x, int y, int nCut, long[][][] cache, int m, int n, int k, int[][][][] hasApple, String[] pizza) {
        if (x == m - 1 && y == n - 1) {
            if (nCut == k - 1) {
                if (hasApple(x, y, hasApple, pizza, m, n)) {
                    return 1;
                } else {
                    return 0;
                }
            } else {
                return 0;
            }
        }
        
        if (nCut == k - 1) {
            if (hasApple(x, y, hasApple, pizza, m, n)) {
                return 1;
            } else {
                return 0;
            }
        
}
        
        if (cache[x][y][nCut] >= 0) {
            return cache[x][y][nCut];
        }
        
        long result = 0;
        
        // horizontal
        for (int i = x + 1; i < m; i++) {
            if (hasApple(x, y, hasApple, pizza, i, n)) 
{
                result += countWays(i, y, nCut + 1, cache, m, n, k, hasApple, pizza);
                result %= MOD;
            }
        }
            
        // vertical
        for (int j = y + 1; j < n; j++) {
            if (
hasApple(x, y, hasApple, pizza, m, j)) {
                result += countWays(x, j, nCut + 1, cache, m, n, k, hasApple, pizza);
                result %= MOD;
            }
        }
        
        result %= MOD;
        cache[x][y][nCut] = result;
        
        return result;
    }
    
    public boolean hasApple(int x, int y, int[][][][] hasApple, String[] pizza, int m, int n) {
        if (hasApple[x][y][m-1][n-1] >= 0) {
            return hasApple[x][y][m-1][n-1] == 1;
        }
        
        for (int i = x; i < m; i++) {
            for (int j = y; j < n; j++) {
                if (pizza[i].charAt(j) == 'A') {
                    hasApple[x][y][m-1][n-1] = 1;
                    return true;
                }
            }
        }
        
        hasApple[x][y][m-1][n-1] = 0;
        return false;
    }
}
