class Solution {
    public int ways(String[] pizza, int k) {
        int row = pizza.length;
        int col = pizza[0].length();
        
        long[][][] dp = new long[row+1][col+1][k];
        
        for (int i = row-1; i >= 0; i--) {
            for (int j = col-1; j >= 0; j--) {
                if (dp[i+1][j][0] > 0 || dp[i][j+1][0] > 0 || pizza[i].charAt(j) == 'A') {
                    dp[i][j][0] = 1;
                }
            }
        }
        
        if (k == 1) {
            return (int) dp[0][0][0];
        }
        
        for (int cut = 1; cut < k; cut++) {
            for (int i = row-1; i >= 0; i--) {
                for (int j = col-1; j >= 0; j--) {
                    // dp[i][j][cut] = dp[i+1][j][cut] + dp[i][j+1][cut];
                    boolean hor = false;
                    for (int i1 = i; i1 < row-1; i1++) {
                        for (int index = j; index < col; index++) {
                            if (hor || pizza[i1].charAt(index) == 'A') {
                                hor = true;
                                break;
                            }
                        }
                        if (hor) {
                            dp[i][j][cut] += dp[i1+1][j][cut-1];
                        }
                    }
                    
                    boolean ver = false;
                    for (int j1 = j; j1 < col-1; j1++) {
                        for (int index = i; index < row; index++) {
                            if (ver || pizza[index].charAt(j1) == 'A') {
                                ver = true;
                                break;
                            }
                        }
                        if (ver) {
                            dp[i][j][cut] += dp[i][j1+1][cut-1];
                        }
                    }
                }
            }
        }

        long result = 0;
        for (int i = row-1; i >= 0; i--) {
            for (int j = col-1; j >= 0; j--) {
                result += dp[i][j][k-1];
            }
        }
        return (int) (dp[0][0][k-1] % (1000000007));
    }
}
