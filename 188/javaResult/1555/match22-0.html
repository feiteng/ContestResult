class Solution {
    public int ways(String[] pizza, int k) {
        int m = pizza.length, n = pizza[0].length();
        Integer[][][] dp = new Integer[k+10][m+10][n+10];
        int[][] prefixSum = new int[m+1][n+1];
        for (int r = m - 1; r >= 0; r--) {
            for (int c = n - 1; c >= 0; c--) {
                prefixSum[r][c] = prefixSum[r][c+1] + prefixSum[r+1][c] - prefixSum[r+1][c+1] + (pizza[r].charAt(c) == 'A' ? 1 : 0);
            }
        }
        // for (int r = 0; r <= m; r++) {
        //     for (int c = 0; c <= n; c++) {
        //         System.out.print(prefixSum[r][c] + " ");
        //     }
        //     System.out.println();
        // }
        return dfs(pizza, m, n, k, 0, 0, dp, prefixSum);
    }

    int dfs(String[] pizza, int m, int n, int k, int r, int c, Integer[][][] dp, int[][] prefixSum) {
        if (prefixSum[r][c] == 0) return 0;
        if (k == 1) return 1;
        if (dp[k][r][c] != null) return dp[k][r][c];
        int ans = 0;
        for (int nr = r + 1; nr < m; nr++) {
            int sum = getSum(prefixSum, r, c, nr - 1, n - 1);
            if (sum == 0) continue;
            ans += dfs(pizza, m, n, k - 1, nr, c, dp, prefixSum);
            ans %= 1_000_000_007;
        }
        for (int nc = c + 1; nc < n; nc++) {
            int sum = getSum(prefixSum, r, c, m - 1, nc - 1);
            if (sum == 0) continue;
            ans += dfs(pizza, m, n, k - 1, r, nc, dp, prefixSum);
            ans %= 1_000_000_007;
        }
        return dp[k][r][c] = ans;
    }

    int getSum(int[][] prefixSum, int r1, int c1, int r2, int c2) {
        return prefixSum[r1][c1] - prefixSum[r1][c2+1] - prefixSum[r2+1][c1] + prefixSum[r2+1][c2+1];
    }
}
