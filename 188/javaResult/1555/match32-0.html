class Solution {
    long dfs(int i, int j, char[][] g, int k, long[][][] dp){
        if(dp[i][j][k] >= 0) return dp[i][j][k];
        int m = g.length;
        int n = g[0].length;
        if(k == 1){
            boolean flag = false;
            for(int a = i; a < m; ++a){
                for(int b = j; b < n; ++b){
                    flag = (g[a][b] == 'A');
                    if(flag) break;
                }
                if(flag) break;
            }
            return flag == true ? 1 : 0;
        }
        long res = 0;
        for(int a = i; a < m - 1; ++a){
            if(rowCut(g, i, a + 1, j)){
                res = (res + dfs(a + 1, j, g, k - 1, dp))%MOD;
            }
        }
        for(int b = j; b < n - 1; ++b){
            if(colCut(g, j, b + 1, i)){
                res = (res + dfs(i, b + 1, g, k - 1, dp))%MOD;
            }
        }
        dp[i][j][k] = res;
        return res;
    }
    public int MOD = 1000000007;
    public int ways(String[] pizza, int k) {
        int m = pizza.length;
        int n = pizza[0].length();
        long[][][] dp = new long[m][n][k+1];
        for(long[][] temp : dp){
            for(long[] temp2 : temp){
                Arrays.fill(temp2, -1);
            }
        }
        char[][] g = new char[m][n];
        for(int i = 0; i < m; ++i){
            g[i] = pizza[i].toCharArray();
        }
        long res = dfs(0, 0, g, k, dp) % MOD;
        return (int)res;
    }
    
    boolean rowCut(char[][] g, int i, int start, int j){
        boolean flag1 = false;
        boolean flag2 = false;
        int m = g.length;
        int n = g[0].length;
        for(int t1 = i; t1 < m; ++t1){
            for(int t2 = j; t2 < n; ++t2){
                if(t1 < start){
                    flag1 = (flag1 || g[t1][t2] == 'A');
                }else{
                    flag2 = (flag2 || g[t1][t2] == 'A');
                }
            }
        }
        return flag1 && flag2;
    }
    
    boolean colCut(char[][] g, int j, int start, int i){
        boolean flag1 = false;
        boolean flag2 = false;
        int m = g.length;
        int n = g[0].length;
        for(int t1 = i; t1 < m; ++t1){
            for(int t2 = j; t2 < n; ++t2){
                if(t2 < start){
                    flag1 = (flag1 || g[t1][t2] == 'A');
                }else{
                    flag2 = (flag2 || g[t1][t2] == 'A');
                }
            }
        }
        return flag1 && flag2;
    }  
}
