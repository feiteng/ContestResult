class Solution {
    public int ways(String[] pizza, int k) {
        int R = pizza.length;
        int C = pizza[0].length();
        int[][] count = new int[R][C];
        for (int r = R - 1; r >= 0; r--) {
            int rowCount = 0;
            for (int c = C - 1; c >= 0; c--) {
                if (pizza[r].charAt(c) == 'A') {
                    rowCount++;
                }
                count[r][c] = rowCount + (r + 1 < R ? count[r + 1][c] : 0);
            }
        }

        int divisor = 1000000000 + 7;
        long[][][] ways = new long[k][R][C];
        ways[0][0][0] = 1;
        
for (int r = 0; r < R; r++) {
            for (int c = 0; c < C; c++) {
                for (int i = 1; i < k; i++) {
                    for (int r1 = 0; r1 < r; r1++) {
                        if (count[r1][c] > count[r][c]) {
                            ways[i][r][c] += ways[i - 1][r1][c];
                        }
                    }
                    for (int c1 = 0; c1 < c; c1++) {
                        if (count[r][c1] > count[r][c]) {
                            ways[i][r][c] += ways[i - 1][r][c1];
                        }
                    }
                    ways[i][r][c] %= divisor;
                }
            }
        }

        long output = 0;
        for (int r = 0; r < R; r++) {
            for (int c = 0; c < C; c++) {
                if (count[r][c] > 0) {
                    output += ways[k - 1][r][c];
                }
            }
        }
        return (int) (output % divisor);
    }
}
