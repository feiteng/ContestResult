class Solution {
    public int ways(String[] pizza, int k) {
        int m = pizza.length;
        int n = pizza[0].length();
        int[][][] dp = new int[m][n][k+1];
        int[][] ct = new int[m+1][n+1];
        int mod = 1_0000_0000_7;
        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                ct[i][j] = ct[i+1][j] + ct[i][j+1] - ct[i+1][j+1];
                if (pizza[i].charAt(j) == 'A') ct[i][j]++;
                //System.out.print(ct[i][j]);
            }
            //System.out.println();
        }
        //System.out.println();
        if (ct[0][0] < k) return 0;
        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                if (ct[i][j] >= 1) dp[i][j][1] = 1;
                //System.out.print(dp[i][j][1]);
            }
            //System.out.println();
        }
        //System.out.println();
        for (int p = 2; p <= k; p++) {
            for (int i = m - 1; i >= 0; i--) {
                for (int j = n - 1; j >= 0; j--) {
                    dp[i][j][p] = 0;
                    for (int a = j + 1; a < n; a++) {
                        if (ct[i][j] - ct[i][a] >= 1) {
                            dp[i][j][p] += dp[i][a][p-1];
                            dp[i][j][p] %= mod;
                        } 
                    }
                    for (int b = i + 1; b < m; b++) {
                        if (ct[i][j] - ct[b][j] >= 1) {
                            dp[i][j][p] += dp[b][j][p-1];
                            dp[i][j][p] %= mod;
                        } 
                    }
                    //System.out.print(dp[i][j][p]);
                }
              // System.out.println();
            }
            //System.out.println();
        }
        return dp[0][0][k];
    }
}
