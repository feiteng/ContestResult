class Solution {
    int n, m;
    String[] a;
    int[][] count;
    int MOD = (int)1e9+7;
    long[][][] dp;
    public int ways(String[] a, int kk) {
        this.a = a;
        n = a.length;
        m = a[0].length();
        
        count = 
new int[n][m];
        
        for(int i=0; i<n; i++) {
            for(int j=0; j<m; j++) {
                count[i][j] = ((i-1>=0?count[i-1][j]:0)+(j-1>=0?count[i][j-1]:0)-(i-1>=0 && j-1>=0?count[i-1][j-1]:0))+(a[i].charAt(j)=='A'?1:0);
            }
        }
        
        dp = new long[n+1][m+1][kk+1];
        
        for(int i=0; i<n+1; i++) {
            for(int j=0; j<m+1; j++) {
                for(int k=0; k<kk+1; k++) {
                    dp[i][j][k] = -1;
                }
            
}
        }
        
        return (int)solve(0, 0, kk);
    }
    
    private long solve(int i, int j, int k) {
        if (k==1) {
            int count = get(i, j, n-1, m-1);
            if (count!=0) return 1;
            return 0;
        
}
        
        if (k<0) return 0;
        
        
        if (dp[i][j][k]!=-1) return dp[i][j][k];
        
        long ans = 0;
        for(int x=i; x<n-1; x++) {
            int count = get(i, j, x, m-1);
            if (count==0) continue;
            ans = (ans+
solve(x+1, j, k-1))%MOD;
        }
        for(int y=j; y<m-1; y++) {
            int count = get(i, j, n-1, y);
            if (count==0) continue;
            ans = (ans+
solve(i, y+1, k-1))%MOD;
        }
        
        return dp[i][j][k]=ans%MOD;
    }
    
    private int get(int i, int j, int x, int y) {
        return count[x][y]-(j-1>=0?count[x][j-1]:0)-(i-1>=0?count[i-1][y]:0)+(i-1>=0&& j-1>=0?count[i-1][j-1]:0);
    }
}
