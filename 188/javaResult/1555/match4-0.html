class Solution {
    public int ways(String[] pizza, int k) {
        int rows = pizza.length;
        int cols = pizza[0].length();
        int[][] apples = new int[rows][cols];
        for (int i = rows - 1; i >= 0; i--) {
            for (int j = cols - 1; j >= 0; j--) {
                if (pizza[i].charAt(j) == 'A') apples[i][j] = 1;
                if (i < rows - 1) apples[i][j] += apples[i + 1][j];
                if (j < cols - 1) apples[i][j] += apples[i][j + 1];
                if (i < rows - 1 && j < cols - 1) apples[i][j] -= apples[i + 1][j + 1];
            }
        }

        int[][][] dp = new int[rows][cols][k];
        for (int i = rows - 1; i >= 0; i--) {
            for (int j = cols - 1; j >= 0; j--) {
                if (apples[i][j] > 0) dp[i][j][0] = 1;
            }
        }

        int mask = 1000000007;

        for (int kk = 1; kk < k; kk++) {
            for (int i = rows - 1; i >= 0; i--) {
                for (int j = cols - 1; j >= 0; j--) {
                    for (int ii = i + 1; ii < rows; ii++) {
                        if (apples[i][j] > apples[ii][j]) {
                            dp[i][j][kk] += dp[ii][j][kk - 1];
                            dp[i][j][kk] %= mask;
                        }
                    }
                    for (int jj = j + 1; jj < cols; jj++) {
                        if (apples[i][j] > apples[i][jj]) {
                            dp[i][j][kk] += dp[i][jj][kk - 1];
                            dp[i][j][kk] %= mask;
                        }
                    }
                }
            }
        }

        return dp[0][0][k - 1];
    }
}
