class Solution {
    public List<String> buildArray(int[] target, int n) {
        int num = 1;
        
        List<String> opers = new LinkedList<String>();
        
        for(int i = 0; i < target.length; ++i){
            
            while(num < target[i]){
                opers.add("Push");
                opers.add("Pop");
                num++;
            }
            
            opers.add("Push");
            
            num++;
            
            
        }
        return opers;
    }
}
