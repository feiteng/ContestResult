class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> ret = new LinkedList<>();
        int index = 1;
        for (int i = 0; i < target.length; i++) {
            while (index < target[i]) {
                ret.add("Push");
                ret.add("Pop");
                index++;
            }
            ret.add("Push");
            index++;
        }
        return ret;
    }
}
