class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> res = new LinkedList<>();
        int val = 1;
        for (int i = 0; i < target.length; i++) {
            while (target[i] > val) {
                res.add("Push");
                res.add("Pop");
                val++;
            }
            res.add("Push");
            val++;
        }
        return res;
    }
}
