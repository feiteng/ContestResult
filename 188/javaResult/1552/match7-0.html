class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> res = new ArrayList<>();
        
        int current = 1;
        for (int t : target) {
            while (current < t) {
                res.add("Push");
                res.add("Pop");
                current++;
            }
            res.add("Push");
            current++;
        }
        return res;
    }
}
