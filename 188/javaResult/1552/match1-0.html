class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> result = new LinkedList<>();
        int i = 1;
        for (int t : target) {
            while (i++ < t) {
                result.add("Push");
                result.add("Pop");
            }
            result.add("Push");
        }
        
        return result;
    }
}
