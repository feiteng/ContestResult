class Solution {
  public List<String> buildArray(int[] target, int n) {
    List<String> ret = new ArrayList<>();

    int now = 1;
    for (int v : target) {
      while (now < v) {
        ret.add("Push");
        now++;
        ret.add("Pop");
      }
      ret.add("Push");
      now ++;

    }
    return ret;
  }
}
