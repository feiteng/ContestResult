class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> res = new ArrayList<>();
        int index = 1;
        int i = 0;
        while (i < target.length && index <= n) {
            if (target[i] == index) {
                res.add("Push");
                index++;
                i++;
            } else {
                res.add("Push");
                res.add("Pop");
                index++;
            }
        }
        return res;
    }
}
