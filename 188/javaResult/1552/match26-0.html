class Solution {
    public List<String> buildArray(int[] target, int n) {
        int prev = 1;
        ArrayList<String> list = new ArrayList<>();
        for (int x : target) {
            while (prev != x) {
                list.add("Push");
                list.add("Pop");
                prev++;
            }
            list.add("Push");
            prev++;
        }
        return list;
    }
}
