class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> list = new ArrayList<>();
        
        int num = 1;
        int i =0;
        while(i < target.length){
            if(target[i] == num){
                list.add("Push");
                i++;
                num++;
            }else{
                list.add("Push");
                list.add("Pop");
                num++;
            }
        }
        return list;
    }
}
