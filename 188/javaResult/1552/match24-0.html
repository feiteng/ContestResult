class Solution {
    public List<String> buildArray(int[] t, int n) {
        int from =1;
        List<String> li  =new ArrayList<>();
        for(int v:t){
            while(from<v){
                li.add("Push");
                li.add("Pop");
                from++;
            }
            li.add("Push");
            from++;
        }
        return li;
        
    }
}
