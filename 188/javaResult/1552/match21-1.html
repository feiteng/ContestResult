class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> ops = new ArrayList<>();
        int cur = 1;
        for (int i : target) {
            while (cur < i) {
                ops.add("Push");
                ops.add("Pop");
                cur++;
            }
            ops.add("Push");
            cur++;
        }
        return ops;
    }
}
