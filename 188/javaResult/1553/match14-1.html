class Solution {
    public int countTriplets(int[] arr) {
        int ans = 0;
        for(int j = 1; j< arr.length; j++){
            TreeMap<Integer, Integer> mp = new TreeMap<>();
            int xor = 0;
            for(int i = j-1; i>= 0; i--){
                xor ^= arr[i];
                mp.put(xor, mp.getOrDefault(xor, 0)+1);
            }
            xor = 0;
            for(int i = j; i< arr.length; i++){
                xor ^= arr[i];
                ans += mp.getOrDefault(xor, 0);
            }
        }
        return ans;
    }
}
