class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int c = 0;
        for (int i = 0; i < n; i++) {
            int a = arr[i];
            for (int j = i+1; j < n; j++) {
                int b = 0;
                for (int k = j; k < n; k++) {
                    b ^= arr[k];
                    if (a == b) {
                        c++;
                    }
                }
                a ^= arr[j];
            }
        }
        return c;
    }
}
