class Solution {
    public int countTriplets(int[] arr) {
        final int N = arr.length;
        
        int[] xorSum = new int[N + 1];
        for (int i = 0; i < N; ++i) {
            xorSum[i + 1] = xorSum[i] ^ arr[i];
        }
        
        int count = 0;
        for (int i = 0; i < N; ++i) {
            for (int j = i + 1; j <= N; ++j) {
                for (int k = j + 1; k <= N; ++k) {
                    int a = xorSum[j] ^ xorSum[i];
                    int b = xorSum[k] ^ xorSum[j];
                    if (a == b) {
                        ++count;
                    }
                }
            }
        }
        return count;
    }
}
