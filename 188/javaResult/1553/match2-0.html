class Solution {
    public int countTriplets(int[] a) {
        int l  =a.length;
        
        int s[] = new int[l+1];
        for(int i=0;i<l;++i){
            s[i+1] = s[i]^a[i];
        }
        int ct =0;
        for(int i=0;i<l;++i){
            for(int j=i+1;j<l;++j){
                for(int k=j;k<l;++k){
                    int v1 = s[i]^s[j];
                    int v2 = s[j]^s[k+1];
                    if(v1==v2){
                        ct++;
                    }
                }
            }
        }
        return ct;
        
        
        
    }
}
