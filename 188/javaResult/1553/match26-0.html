class Solution {
    public int countTriplets(int[] arr) {
        int ans = 0;
        for(int i = 0; i < arr.length; i++) {
            int temp = arr[i];
            for(int j = i+1; j < arr.length; j++) {
                temp = (temp ^ arr[j]);
                if(temp == 0) {
                    ans = ans + j - i;
                }
            }
        }
        return ans;
    }
}
