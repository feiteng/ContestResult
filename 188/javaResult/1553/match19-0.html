class Solution {
  public int countTriplets(int[] arr) {
    int ans = 0;
    for (int i = 0; i <= arr.length - 2; i++) {
      int a = 0;
      for (int j = i + 1; j < arr.length; j++) {
        a ^= arr[j - 1];
        int b = arr[j];
        if (a == b) ans++;
        for (int k = j + 1; k < arr.length; k++) {
          b ^= arr[k];
          if (a == b) ans++;
        }
      }
    }
    return ans;
  }
}
