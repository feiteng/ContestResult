class Solution {
    public int countTriplets(int[] arr) {
        
            
        int count = 0;
        
        for(int i = 0; i < arr.length; i++) {
            
            int xs = 0;
            for(int k = i; k < arr.length; k++) {
            xs = xs ^ arr[k];
            // System.out.println(xs);
            int num2 = 0;
            for(int j = k+1; j < arr.length; j++ ) {
                num2 ^= arr[j];
                if(xs == num2) {
                    // System.out.println(k  + " " + j);
                    count++;
                }
            }
            
            
            // System.out.println();
            }
        }
        
        return count;
    }
}
