class Solution {

    public int countTriplets(int[] arr) {
    	int ans = 0 , n = arr.length;
    	for (int i = 0;i < n;i ++) {
    		int acc = 0;
    		for (int j = i;j < n;j ++) {
    			acc ^= arr[j];
    			if (acc == 0 && j - i + 1 > 1) {
    				ans += (j - i);
    			}

    		}
    	}
        return ans;
    }
}
