class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int[] temp = new int[n + 1];
        for(int i = 0; i < n;i++){
            temp[i + 1] = (temp[i] ^ arr[i]);            
            // System.out.println(temp[i+1]);
        
}
        int count = 0;
        for(int i = 0; i < n;i++){
            for(int j = i + 1;j < n;j++){
                for(int k = j;k < n;k++){
                    int xorItoJ = (temp[j] ^ temp[i]);
                    
int xorJtoK = (temp[k + 1] ^ temp[j]);
                    if(xorItoJ == xorJtoK)
                        count++;
                }
            }
        }
        return count;
    }
}
