class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int[] xor = new int[n];
        xor[0] = arr[0];
        for (int i = 1; i < n; i++) {
            xor[i] = xor[i-1] ^ arr[i];
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int k = j; k < n; k++) {
                    int a = i == 0 ? xor[j-1] : xor[j-1]^xor[i-1];
                    int b = xor[k]^xor[j-1];
                    if (a == b) {
                        res++;
                    }
                }
            }
        }
        return res;
    }
}
