class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int count = 0;
        
        for(int i = 0; i < n - 1; i++){
            int a = arr[i];
            for(int j = i + 1; j < n; j++){
                int b = arr[j];
                if(a == b){
                    count++;
                }
                
                
for(int k = j + 1; k < n; k++){
                    b ^= arr[k];
                    if(a == b){
                        count++;
                    }
                }
                a ^= arr[j];
            }
        }
        
        return count;
    }
}
