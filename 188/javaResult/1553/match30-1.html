class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length, res = 0;
        int[] s = new int[n + 1];
        for (int i = 0; i < n; i++) s[i + 1] = s[i] ^ arr[i];
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j; k < n; k++) {
                    if ((s[k + 1] ^ s[j]) == (s[j] ^ s[i])) res++
;
                }
            }
        }
        return res;
    }
}
