class Solution {
    public int countTriplets(int[] arr) {
	        int n = arr.length;
	        int[] par = new int[n];
	        for(int i = 0; i < n; i++) {
	        	par[i] = arr[i] ^ (i == 0 ? 0 : par[i-1]);
	        }
	        int res = 0;
	        for(int i = 0; i < n; i++) {
	        	for(int j = i+1; j < n; j++) {
	        		for(int k = j; k < n; k++) {
	        			int a = par[j-1] ^ (i == 0 ? 0 : par[i-1]);
	        			int b = par[k] ^ par[j-1];
	        			if(a == b) res++;
	        		}
	        	}
	        }
	        return res;
	    }
}
