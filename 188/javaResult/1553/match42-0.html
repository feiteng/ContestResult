class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int[][] cache = new int[n][n];
        for (int i = 0; i < n; i++) {
            cache[i][i] = arr[i];
            for (int j = i+1; j < n; j++) {
                cache[i][j] = arr[j]^cache[i][j-1];
            }
            // System.out.println(Arrays.toString(cache[i]));
        }
        
        int count = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int k = j; k < n; k++) {
                    if (cache[i][j-1] == cache[j][k]) {
                        // System.out.println(i + " " + j + " " + k);
                        count++;
                    }
                }
            }
        }
        return count;
    }
}
