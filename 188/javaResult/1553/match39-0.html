class Solution {

    public int countTriplets(int[] arr) {
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            int xor = arr[i];
            for (int j = i + 1; j < arr.length; j++) {
                int xor1 = arr[j];
                if (xor == xor1) {
                    count++;
                }
                for (int k = j + 1; k < arr.length; k++) {
                    xor1 ^= arr[k];
                    if (xor == xor1) {
                        count++;
                    }
                }
                xor ^= arr[j];
            }
        }

        return count;
    }

}
