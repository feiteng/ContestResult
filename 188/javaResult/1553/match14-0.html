class Solution {
    public int countTriplets(int[] arr) {
        int ans = 0;
        for (int j = 1; j < arr.length; ++j) {
            HashMap<Integer, Integer> left = new HashMap<>();
            int sum = 0;
            for (int i = j - 1; i >= 0; --i) {
                sum ^= arr[i];
                left.put(sum, left.getOrDefault(sum, 0) + 1);
            }
            sum = 0;
            for (int k = j; k < arr.length; ++k) {
                sum ^= arr[k];
                ans += left.getOrDefault(sum, 0);
            }
        }
        return ans;
    }
}
