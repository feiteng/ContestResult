class Solution {
    public int countTriplets(int[] arr) {
        int n = arr.length;
        int[][] xor = new int[n][n];
        for (int i = 0; i < n; ++i) {
            xor[i][i] = arr[i];
            for (int j = i+1; j < n; ++j) {
                xor[i][j] = xor[i][j-1]^arr[j];
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                for (int k = j; k < n; ++k) {
                    if (xor[i][j-1]==xor[j][k])
                        ++ans;
                }
            }
        }
        return ans;
    }
}
