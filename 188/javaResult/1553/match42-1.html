class Solution {
    public int countTriplets(int[] arr) {
        int l = arr.length;
        int[] xor = new int[l + 1];
        for (int i = 1; i <= l; i++) {
            xor[i] = xor[i - 1] ^ arr[i - 1];
        
}
        
        int res = 0;
        for (int i = 0; i < l; i++) {
            for (int j = i + 1; j < l; j++) {
                for (int k = j; k < l; k++) {
                    if ((xor[j] ^ xor[i]) == (xor[k + 1] ^ xor[j])) res++
;
                }
            }
        }
        
        return res;
    }
}
