class Solution {
    public int countTriplets(int[] arr) {
        int ans = 0;
        for (int i = 0; i < arr.length; i++) {
            int v = 0;
            for (int j = i + 1; j < arr.length; j++) {
                v ^= arr[j - 1];
                int v2 = 0;
                
for (int k = j; k < arr.length; k++) {
                    v2 ^= arr[k];
                    if (v == v2) {
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
}
