class Solution {
    public int countTriplets(int[] arr) {
        int[] pre = new int[arr.length + 1];
        for (int i = 0; i < arr.length; i++) {
            pre[i + 1] = pre[i] ^ arr[i];
        }
        int res = 0;
        for (int i = 0; i < arr.length; i++) {
            for (int k = i + 1; k < arr.length; k++) {
                if ((pre[k + 1] ^ pre[i]) == 0) {
                    res += k - i;
                }
            }
        }
        return res;
    }
}
