class Solution {
    public int countTriplets(int[] arr) {
        int len = arr.length;
        
        int[] prefix = new int[len + 1];
        
        for (int i = 0; i < len; i += 1) {
            prefix[i + 1] = prefix[i] ^ arr[i];
        }
        
        
        int count = 0;
        
        for (int i = 0; i < len; i += 1) {
            for (int j = i + 1; j < len; j += 1) {
                for (int k = j; k < len; k += 1) {
                    if ((prefix[j] ^ prefix[i]) == (prefix[k + 1] ^ prefix[j])) count += 1;
                }
            }
        }
        
        return count;
    }
}
