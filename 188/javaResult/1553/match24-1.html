class Solution {
    public int countTriplets(int[] arr) {
        int count = 0;
        for(int i=0;i<arr.length;i++){
            int a = arr[i];
            for(int j =i+1;j<arr.length;j++){
                a = a ^ arr[j];
                int b = arr[j];
                for(int k = j;k<arr.length;k++){
                    b = b ^ arr[k];
                    
if(a == b){
                        count++;
                    }
                }
            }
        }
        return count;
        
    }
}
