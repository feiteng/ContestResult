class Solution {
    
public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        TreeNode[] nodes = new TreeNode[n];
        for (int i = 0;i<n;i++) {
            nodes[i] = new TreeNode(hasApple.get(i));
        }
        for (int[] edge : edges) {
            nodes[edge[0]].children.add(nodes[edge[1]]);
        }
        int time = getTime(nodes[0]);
        return (time > 0) ? time - 2 : 0;
    }
    
    private int getTime(TreeNode node) {
        
int time = 0;
        for (TreeNode child : node.children) {
            time += getTime(child);
        }
        if (time != 0) {
            return time + 2;
        }
        return node.hasApple ? 2 : 0;
    }
    
    class TreeNode {
        boolean hasApple;
        List<TreeNode> children;
        public TreeNode(boolean hasApple) {
            this.hasApple = hasApple;
            children = new ArrayList<>();
        }
    }
}
