class Solution {
    int t = 0;
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        ArrayList<ArrayList<Integer>> adjList = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
        
}
        for (int[] e : edges) {
            adjList.get(e[0]).add(e[1]);
            adjList.get(e[1]).add(e[0]);
        }
        // System.out.println(adjList);
        boolean[] visited = new boolean[n];
        dfs(0, 0, hasApple, adjList, visited);
        return t*2;
    }
    
    public boolean dfs(int curr, int d, List<Boolean> hasApple, ArrayList<ArrayList<Integer>> adjList, boolean[] visited) {
        if (visited[curr]) {
            return false;
        }
        visited[curr] = true;
        boolean under = hasApple.get(curr);
        for (int other : adjList.get(curr)) {
            if (dfs(other, d+1, hasApple, adjList, visited)) {
                under = true;
                t++;
            }
        }
        return under;
    }
}
