class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int[] edge: edges) {
            int s = edge[0], e = edge[1];
            if(!map.containsKey(s))
                map.put(s, new ArrayList<>());
            map.get(s).add(e);
        }
        
        Integer ans = helper(n, map, hasApple, 0);
        return ans == null ? 0 : ans;
    }
    
    private Integer helper(int n, Map<Integer, List<Integer>> map, List<Boolean> hasApple, int idx) {
        if(!map.containsKey(idx)) {
            if(hasApple.get(idx))
                return 0;
            else
                return null;
        }
        
        Integer ans = null;
        for(int each: map.get(idx)) {
            Integer temp = helper(n, map, hasApple, each);
            if(temp != null) {
                if(ans == null)
                    ans = temp + 2;
                else
                    ans = ans + temp + 2;
            }
        }
        
        if(ans == null) {
            if(hasApple.get(idx))
                ans = 0;
        }
        
        return ans;
    }
}
