class Solution {
    List<Integer> adj[];

    int dfs(int f, int p, List<Boolean> hasApple) {
        List<Integer> ad = adj[f];
        int res = 0;
        for (int i = 0; i < ad.size(); i++) {
            int t = ad.get(i);
            if (t == p) {
                continue;
            }
            int cur = dfs(t, f, hasApple);
            if (cur >= 0) {
                res += cur + 2;
            }
        }
        if (res == 0) {
            if (!hasApple.get(f)) {
                res = -1;
            }
        }
        return res;
    }

    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        adj = new List[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        for (int i = 0; i < edges.length; i++) {
            int f = edges[i][0];
            int t = edges[i][1];
            adj[f].add(t);
            adj[t].add(f);
        }

        int res = dfs(0, -1, hasApple);
        return Math.max(res, 0);
    }
}
