class Solution {
    class Node{
        boolean hasApple;
        int minSize;

        public Node(boolean hasApple,int minSize){
            this.hasApple = hasApple;
            this.minSize = minSize;
        }
    }

    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer,List<Integer>> map = new HashMap<>();
        for (int[] eachEdge : edges){
            List<Integer> list = map.get(eachEdge[0]);
            if(list == null){
                list = new ArrayList<>();
            }
            list.add(eachEdge[1]);
            map.put(eachEdge[0],list);
        }

        Node dfs = dfs(0, map, hasApple);
        return dfs.minSize;
    
}

    private Node dfs(Integer root,Map<Integer,List<Integer>> map,List<Boolean> hasApple){
        List<Integer> list = map.get(root);
        if(list == null || list.size() == 0){
            if(hasApple.get(root)){
                return new Node(true,0);
            }else{
                return new Node(false,0);

            }

        
}

        int res = 0;
        for (int eachChild: list){
            Node dfs = dfs(eachChild, map,hasApple);
            if(!dfs.hasApple){
                continue;
            }

            res += dfs.minSize;
            res += 2;
        }

        if(res != 0){
            return new Node(true,res);
        }

        if(hasApple.get(root)){
            return new Node(true,0);
        }

        return new Node(false,0);
    }
}
