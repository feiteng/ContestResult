class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        List<Integer>[] es = (List<Integer>[]) 
new List[n + 1];
        for (int i = 0; i < n; i++) es[i] = new ArrayList<>();
        for (int[] e : edges) {
            
es[e[0]].add(e[1]);
        }
        
        int res = 0;
        for (int v : es[0]) {
            res += 2 * dfs(v, es, hasApple);
        }
        
        return res;
    }
    
    private int dfs(int idx, List<Integer>[] es, List<Boolean> hasApple) {
        
        int res = 0;
        for (int n : es[idx]) {
            res += dfs(n, es, hasApple);
        }
        // System.out.println(idx + ", res=" + res);
        return (res > 0 || hasApple.get(idx)) ? res + 1 : res;
    }
}
