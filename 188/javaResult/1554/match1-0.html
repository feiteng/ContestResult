class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int[] edge : edges) {
            List<Integer> list = map.getOrDefault(edge[0], new ArrayList<>());
            list.add(edge[1]);
            map.put(edge[0], list);
        }
        int time = dfs(map, 0, hasApple);
        return time > 0 ? time - 2 : 0;
    }
    
    private int dfs(Map<Integer, List<Integer>> map, int curr, List<Boolean> hasApple) {
        List<Integer> next = map.getOrDefault(curr, new ArrayList<>());
        int time = 0;
        for (int nei: next) {
            time += dfs(map, nei, hasApple);
        }
        if (time == 0 && !hasApple.get(curr)) {
            return 0;
        }
        if (time > 0 || hasApple.get(curr)) {
            return time + 2;
        }
        return 0;        
    }    
}
