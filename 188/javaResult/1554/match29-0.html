class Solution {
    public class Node {
        boolean hasApple;
        boolean visited;
        List<Node> edges;
        public Node(boolean hasApple) {
            this.hasApple = hasApple;
            edges = new ArrayList<Node>();
            visited = false;
        }
        public int timeGettingApples() {
            if(this.visited) {
                return 0;
            }
            this.visited = true;
            int t = 0;
            for(Node n : edges) {
                t += n.timeGettingApples();
            }
            if(t > 0) {
                return t + 2;
            }
            if(hasApple) {
                return 2;
            }
            return 0;
        }
    }
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Node[] nodes = new Node[n];
        for(int i = 0; i < n; i++) {
            nodes[i] = new Node(hasApple.get(i));
        }
        for(int[] edge : edges) {
            Node a = nodes[edge[0]];
            Node b = nodes[edge[1]];
            a.edges.add(b);
            b.edges.add(a);
        }
        int output = nodes[0].timeGettingApples();
        if(output == 0) {
            return 0;
        }
        return output - 2;
    }
}
