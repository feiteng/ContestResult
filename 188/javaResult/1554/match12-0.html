class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        if (hasApple.size() == 0) return 0;
        
        Map<Integer, List<Integer>> G = new HashMap<>();
        for (int i = 0; i < n; ++i) {
            G.put(i, new ArrayList<>());
        }
        
        for (int[] e : edges) {
            G.get(e[0]).add(e[1]);
        }
        
        int[] res = dfs(G, hasApple, 0);
        
        return res[1];
    }
    
    int[] dfs(Map<Integer, List<Integer>> G, List<Boolean> hasApple, int s) {
        int[] ans = new int[2];
        if (hasApple.get(s) == true) {
            ans[0] = 1;
        }
        
        for (int v : G.get(s)) {
            int[] res = dfs(G, hasApple, v);
            ans[1] += res[1];
            if (res[0] > 0) {
                ans[1] += 2;
                ans[0] = 1;
            }
        }
        
        return ans;
    }
}
