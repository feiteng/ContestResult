class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> tree = new HashMap<>();
        for(int[] ed : edges) {
            if(!tree.containsKey(ed[0])) {
                tree.put(ed[0], new ArrayList<>());
            }
            tree.get(ed[0]).add(ed[1]);
        }
        return dfs(0, tree, hasApple, n);
    }
    
    private int dfs(int cur, Map<Integer, List<Integer>> tree, List<Boolean> hasApple, int n) {
        if(cur >= n) {
            return 0;
        }
        if(!tree.containsKey(cur)) {
            return 0;
        }
        
        int sum = 0;
        for(int son : tree.get(cur)) {
            int cnt = dfs(son, tree, hasApple, n);
            if(cnt != 0 || hasApple.get(son)) {
                sum += cnt + 2;
            }
        }
        return sum;
    }
}
