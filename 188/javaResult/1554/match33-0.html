class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        node[] arr=new node[n];
        for(int i=0;i<n;i++){
            arr[i]=new node();
        }
        for(int i=0;i<n-1;i++){
            int u=edges[i][0];
            int v=edges[i][1];
            arr[u].ns.add(v);
            arr[v].ns.add(u);
        }
        int ans=recur(0,arr,hasApple,new HashSet<>());
        if(ans>0){
            ans-=2;
        }
        return ans;
    }
    static class node{
        ArrayList<Integer> ns;
        node(){
            this.ns=new ArrayList<>();
        }
    }
    public int recur(int u,node[] arr,List<Boolean> hs,Set<Integer> vis){
        int cnt=0;
        vis.add(u);
        // System.out.println(u+" ");
        for(Integer v:arr[u].ns){
            if(!vis.contains(v)){
                cnt+=recur(v,arr,hs,vis);
            }
        }
        if(hs.get(u)||cnt!=0){
            cnt+=2;
        }
        // System.out.println(u+" "+cnt);
        return cnt;
    }
}
