class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.put(i, new HashSet<>());
        }
        for (int[] e : edges) {
            map.get(e[0]).add(e[1]);
            map.get(e[1]).add(e[0]);
        }
        return Math.max(0, 
helper(0, -1, map, hasApple));
    }
    private int helper(int cur, int prev, Map<Integer, Set<Integer>> map, List<Boolean> hasApple) {
        int total = 0;
        for (int next : map.get(cur)) {
            if (next == prev) {
                continue;
            }
            
int time = helper(next, cur, map, hasApple);
            if (time > 0 || hasApple.get(next)) {
                total += time + 2;
            }
        }
        return total;
    }
}
