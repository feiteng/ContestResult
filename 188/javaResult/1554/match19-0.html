class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Set[] connects = new HashSet[n];
        for (int i = 0; i < n; i++) connects[i] = new HashSet();
        for (int[] e : edges) {
            connects[e[0]].add(e[1]);
            connects[e[1]].add(e[0]);
        }
        return Math.max(0, minTime(n, connects, 0, -1, hasApple) * 2);
    }

    private int minTime(int n, Set[] connects, int i, int prev, List<Boolean> hasApple) {
        int time = 0;
        for (Object v : connects[i]) {
            int j = (Integer) v;
            if (j == prev) continue;
            time += minTime(n, connects, j, i, hasApple) + 1;
        }
        return time == 0 ? hasApple.get(i) ? 0 : -1 : time;
    }
}
