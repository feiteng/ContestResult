class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> neighbors = new HashMap<>();
        for (int i = 0; i < n; i++) {
            neighbors.put(i, new ArrayList<>());
        }
        for (int[] e : edges) {
            neighbors.get(e[0]).add(e[1]);
            neighbors.get(e[1]).add(e[0]);
        }
        return dfs(0, new boolean[n], neighbors, hasApple).getCost();
    
}
    
    private Result dfs(int i, boolean[] visited, Map<Integer, List<Integer>> neighbors, List<Boolean> hasAppleList) {
        if (visited[i]) {
            return new Result(false, 0);
        }
        visited[i] = true;
        boolean hasApple = hasAppleList.get(i);
        
int cost = 0;
        for (int neighbor : neighbors.get(i)) {
            Result result = dfs(neighbor, visited, neighbors, hasAppleList);
            if (result.isHasApple()) {
                hasApple = true;
                cost += 2 + result.getCost();
            }
        }
        return new Result(hasApple, cost);
    }
    
    class Result {
        boolean hasApple;
        int cost;

        public Result(boolean hasApple, int cost) {
            this.hasApple = hasApple;
            this.cost = cost;
        }

        public boolean isHasApple() {
            return hasApple;
        }

        public int getCost() {
            return cost;
        }
    }
}
