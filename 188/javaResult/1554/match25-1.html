class Solution {
    
Set<Integer>[] gra;
    List<Boolean> ch;
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        gra = new Set[n];
        ch = hasApple;
        for (int i = 0; i < n; i++) gra[i] = new HashSet<>();
        for (int[] e : edges) {
            gra[e[0]].add(e[1]);
            gra[e[1]].add(e[0]);
        }
        return 2 * dfs(0, -1);
    }
    int dfs(int cur, int par) {
        boolean h = ch.get(cur);
        int res = (cur != 0 && h) ? 1 : 0;
        for (int nei : gra[cur]) {
            if (nei == par) continue;
            res += dfs(nei, cur);
        }
        if (cur != 0 && res > 0 && !h) res++;
        return res;
    }
}
