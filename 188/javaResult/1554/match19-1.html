class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> has) {
        List<Integer>[] graph = new List[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new ArrayList<>();
        }
        for (int[] e : edges) {
            graph[e[0]].add(e[1]);
            graph[e[1]].add(e[0]);
        }
        
        boolean[] visited = new boolean[n];
        visited[0] = true;
        int res = helper(graph, has, 0, visited);
        return res == -1 ? 0 : res;
    }
    
    private int helper(
List<Integer>[] graph, List<Boolean> has, int cur, boolean[] visited) {
        int res = 0;
        for (int next : graph[cur]) {
            if (visited[next]) continue;
            visited[next] = true;
            int val = helper(graph, has, next, visited);
            if (val != -1) {
                res += val + 2;
            }
        
}
        
        //System.out.println(cur + " " + res);
        return res == 0 ? has.get(cur) ? 0 : -1 : res;
    }
}
