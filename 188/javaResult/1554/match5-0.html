class Solution {
    static ArrayList<Integer>[] g;
	    static int minTime(int n, int[][] edges, List<Boolean> hasApple) {
	    	g = new ArrayList[n];
	    	for(int i = 0; i < n; i++) g[i] = new ArrayList<>();
	        for(int[] e: edges) {
	        	g[e[0]].add(e[1]);
	        	g[e[1]].add(e[0]);
	        }
	        int res = 0;
	        for(int u: g[0]) {
	        	res += countsub(u, 0, hasApple); 
	        }
	        return res;
	    }
	    static int countsub(int u, int p, List<Boolean> hasApple) {
	    	int res = 0;
	    	for(int v: g[u]) {
                if(v == p) continue;
	    		res += countsub(v, u, hasApple);
	    	}
	    	if(hasApple.get(u) || res > 0) res += 2;
	    	return res;
	    }
	
}
