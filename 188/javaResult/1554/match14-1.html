class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        List<Integer>[] tree = new LinkedList[n];
        
        for(int i = 0; i < tree.length; ++i){
            tree[i] = new LinkedList();
        }
        
        boolean[] appleFlag = new boolean[n];
        int index = 0;
        for(Boolean flag:hasApple){
            appleFlag[index++] = flag;
        }
        
        
for(int[] e:edges){
            tree[e[0]].add(e[1]);
            tree[e[1]].add(e[0]);
        }
       int res =  dfs(0, tree, new boolean[n], appleFlag);
        return  res>0 ? res-2 :res;
    
}
    
    
    int dfs(int node, List<Integer>[] tree, boolean[] passed, boolean[] appleFlag){
        int count = 0;
        passed[node] = true;
        for(int next:tree[node]){
            if(!passed[next]){
                count+=dfs(next, tree, passed, appleFlag);
            }
        }
        passed[node] = false;
        
        if(count > 0 || appleFlag[node]){
            return count+2;
        }else{
            return 0;
        }
    }
    
    
    
    
}
