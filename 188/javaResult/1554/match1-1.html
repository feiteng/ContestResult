class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        HashMap<Integer, List<Integer>> sides = new HashMap<>();
        
        for(int[] edge : edges){
            sides.computeIfAbsent(edge[0], (id) -> new ArrayList<>()).add(edge[1]);
        
}
        
        int res = dfs(0, sides, hasApple);
        return res == 0 ? 0 : res - 2;
    }
    
    public int dfs(int cur,  HashMap<Integer, List<Integer>> sides, List<Boolean> hasApple){
        int res = 0;
        for(int next : sides.getOrDefault(cur, Collections.emptyList()))
{
            res += dfs(next, sides, hasApple);
        }
        
        if(hasApple.get(cur) || res != 0){
            return res + 2;
        }
        
        return 0;
    }
}
