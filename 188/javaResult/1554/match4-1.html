class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, Integer> parentMap = new HashMap<>();
        
        for (int[] edge : edges) {
            parentMap.put(edge[1], edge[0]);
        }
        
        HashSet<Integer> usefulEdges = new HashSet<>();
        
        for (int i = 1; i < hasApple.size(); i++) {
            if (usefulEdges.contains(i)) {
                continue;
            }
            if (hasApple.get(i)) {
                int curr = i;
                while (curr != 0) {
                    if (usefulEdges.contains(curr)) {
                        break;
                    }
                    
usefulEdges.add(curr);
                    curr = parentMap.get(curr);
                }
            }
        }
        
        return usefulEdges.size() * 2;
    }
}
