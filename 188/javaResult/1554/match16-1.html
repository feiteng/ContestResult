class Solution {
    
public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
       Map<Integer, Set<Integer>> g = new HashMap<>(); 
        for (int i = 0; i < edges.length; i++) {
            if (!g.containsKey(edges[i][0])) {
                g.put(edges[i][0], new HashSet<>());
            }
            if (!g.containsKey(edges[i][1])) {
                g.put(edges[i][1], new HashSet<>());
            }
            g.get(edges[i][0]).add(edges[i][1]);
            
g.get(edges[i][1]).add(edges[i][0]);
        }
        Pair res = dfs(0, g, hasApple, -1);
        return res.cnt; 
    }
    
    private Pair dfs(int node, Map<Integer, Set<Integer>> g, List<Boolean> hasApple, int f) {
        Set<Integer> nei = g.get(node);
        int cnt = 0; 
        boolean has = false;
        for (int x : nei) {
            if (x == f) {
                continue;
            }
            Pair child = dfs(x, g, hasApple, node);
            has |= child.has;
            cnt += child.cnt; 
            if (child.has) {
                cnt += 2;
            }
        }
        
if (hasApple.get(node) || has) {
            return new Pair(true, cnt);
        }
        return new Pair(false, 0);
    }
    
    
private class Pair {
        boolean has;
        int cnt; 
        public Pair(boolean has, int cnt) {
            this.has = has;
            this.cnt = cnt; 
        }
    } 
}
