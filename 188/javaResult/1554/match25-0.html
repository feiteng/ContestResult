class Solution {

    LinkedList<Integer>[] G;
    List<Boolean> hasApple;
    int ans = 0;
    
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        this.G = new LinkedList[n];
        this.hasApple = hasApple;
        for (int i = 0; i < n; i++) G[i] = new LinkedList<>();
        for (int[] e : edges) {
            G[e[0]].offer(e[1]);
            G[e[1]].offer(e[0]);
        }
        
        dfs(0, -1);
        return ans;
    }
    
    boolean dfs(int u, int p) {
        boolean shouldVisit = hasApple.get(u);
        
        for (int w : G[u]) {
            if (p == w) continue;
            if (dfs(w, u)) {
                shouldVisit = true;
                ans += 2;
            }
        }
        
        return shouldVisit;
    }
    
    
}
