class Solution {
    
public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int[] edge : edges) {
            map.putIfAbsent(edge[0], new ArrayList<>());
            map.get(edge[0]).add(edge[1]);
        }
        return Math.max(
minTimeHelper(0, map, hasApple), 0);
    }
    
    private int minTimeHelper(int index, Map<Integer, List<Integer>> map, List<Boolean> hasApple) {
        int time = 0;
        for (int child : map.getOrDefault(index, new ArrayList<>())) 
{
            int res = minTimeHelper(child, map, hasApple);
            if (res >= 0) {
                time += res + 2;
            }
        }
        return time == 0 && !hasApple.get(index) ? -1 : time;
    }
}
