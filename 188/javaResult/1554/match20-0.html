class Solution {
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        List[] graph = new List[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new ArrayList<Integer>();
        }
        for (int[] edge: edges) {
            int from = edge[0];
            int to = edge[1];
            graph[from].add(to);
            graph[to].add(from);
        }
        
        int[] res = dfs(graph, 0, -1, hasApple);
        return res[1];
    }
    
    private int[] dfs(List[] graph, int root, int parent, List<Boolean> hasApple) {
        int[] res = {0, 0};
        List<Integer> neis = graph[root];
        for (int nei : neis) {
            if (nei == parent) continue;
            int[] curr = dfs(graph, nei, root, hasApple);
            if (curr[0] > 0) {
                res[1] += (2 + curr[1]);
            }
        }
        if (res[1] > 0 || hasApple.get(root)) {
            res[0] = 1;
        }
        return res;
    }
}
