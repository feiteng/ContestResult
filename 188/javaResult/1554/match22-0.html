class Solution {
    class Result {
        boolean canTravel;
        int time;
        Result( int time,boolean canTravel){
            this.canTravel = canTravel;
            this.time = time;
        }
    }
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> tree = new HashMap<>();
        
        for (int[] edge: edges){
            int from = edge[0], to = edge[1];
            tree.putIfAbsent(from, new LinkedList<>());
            tree.putIfAbsent(to, new LinkedList<>());
            tree.get(from).add(to);
        }
        
        return dfs(tree, 0, hasApple).time;
    }
    
    
    private Result dfs(Map<Integer, List<Integer>> tree, Integer curNode, List<Boolean> hasApple){
        
       
        List<Integer> children = tree.get(curNode); 
        
        if (children.size() == 0){
            return new Result(0, hasApple.get(curNode));
        }
        
        int time = 0; 
        for (Integer child: children){
            Result res =  dfs(tree, child, hasApple);
            if (res.canTravel){
                time += res.time + 2;
            }
        }
        if (time == 0){
            return new Result(0, hasApple.get(curNode));
        }
        // System.out.println("curNode:" + curNode + "time" + time); 
        
        return new Result(time, true);
    }
}
