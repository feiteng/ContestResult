class Solution {
    
    
    List<Integer> g[];
    int r = 0;
    
    boolean go(int rt,int fa){
        boolean me = f[rt];
        for(int to:g[rt]){
            if(to==fa) continue;
            boolean res = go(to,rt);
            if(res){
                r += 2;
                me = true;
            }
        }
        return me;
    }
    
    boolean f[];
    public int minTime(int n, int[][] es, List<Boolean> hasApple) {
        f = new boolean[n];
        int u = 0;
        for(boolean ce:hasApple){f[u++] = ce;}
        
        
        g = new ArrayList[n];
        for(int i=0;i<n;++i) g[i] = new ArrayList<Integer>();
        
        for(int[] e:es){
            g[e[0]].add(e[1]);
            g[e[1]].add(e[0]);
        }
        go(0,-1);
        return r;
    }
}
