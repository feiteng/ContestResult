class Pair{
    final boolean hasApple;
    final int dist;
    
    public Pair(boolean hasApple, int dist){
        this.hasApple = hasApple;
        this.dist = dist;
    }
}

class Solution {
    
public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int[] edge : edges){
            map.computeIfAbsent(edge[0], k->new ArrayList<>()).add(edge[1]);
        
}
        return traverse(0, map, hasApple).dist;        
    }

    private Pair traverse(int node, Map<Integer, List<Integer>> map, List<Boolean> list){
        boolean hasApple = list.get(node);
        
        int dist = 0;
        for(int child : map.getOrDefault(node, Collections.emptyList())){
            Pair pair = traverse(child, map, list);
            dist += pair.dist;
            hasApple = hasApple || pair.hasApple;
        }
        
        if(hasApple){
            int newdist = node != 0 ? dist + 2 : dist;
            return new Pair(true, newdist);
        }else{
            return new Pair(false, 0);
        }
    }
    
    
}
