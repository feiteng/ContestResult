class Solution {
    
    int ans ;
    
    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        ArrayList<ArrayList<Integer>> graph = new ArrayList<ArrayList<Integer>>();
        for(int i = 0; i < n;i++)
            graph.add(new ArrayList<Integer>());
        for(int[] edge : edges){
            graph.get(edge[0]).add(edge[1]);
            graph.get(edge[1]).add(edge[0]);
        }
        boolean[] visited = new boolean[n];
        DFS(0, graph, hasApple, visited);
        return ans;
    }
    
    public boolean DFS(int u, ArrayList<ArrayList<Integer>> graph, List<Boolean> hasApple, boolean[] visited){
        visited[u] = true;
        ArrayList<Integer> neighbour = graph.get(u);
        boolean temp = hasApple.get(u);
        for(int v : neighbour){
            if(!visited[v]){
                boolean t = DFS(v, graph, hasApple, visited);
                if(t){
                    ans += 2;                    
                    temp = true;
                }
            }
        }
        return temp;
    }
}
