class Solution {
public:
    pair<int, int> solve(int root, int parent, vector<vector<int>>& adj, vector<bool>& hasApple){
        int ans= 0;
        bool found = false;
        for(auto x : adj[root]){
            if(x == parent) continue;
            pair<int, int> curr = solve(x, root, adj, hasApple);
            if(curr.second == 1){
                found = true;
                ans += 2 + curr.first;
            }
        }
        if(hasApple[root] == true){
            return {ans, 1};

        }
        return {ans, found};
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> adj(n);
        for(auto x : edges){
            adj[x[0]].push_back(x[1]);
            adj[x[1]].push_back(x[0]);
        }
        return solve(0, -1, adj, hasApple).first;
    }
};
