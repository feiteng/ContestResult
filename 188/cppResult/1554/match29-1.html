class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        int res = 0;
        vector<int> par(n, 0);

        for (const auto &edge : edges) {
            par[edge[1]] = edge[0];
        }
        unordered_set<int> visited;
        for (int i = 1; i < n; ++i) {
            if (hasApple[i]) {
                int cnt = 0, node = i;
                while (visited.find(node) == visited.end() && node != 0) {
                    visited.insert(node);
                    cnt++;
                    node = par[node];
                }
                res += cnt * 2;
                // visited.insert(i);
            }
        }
        return res;
    }
};
