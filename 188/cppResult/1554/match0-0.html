class Solution {
public:
    pair<int, int> solve(int u, int p, vector<vector<int> > &G, vector<bool> &hasApple) {
        int ret = 0;
        bool ok = hasApple[u];
        for (auto &v : G[u]) {
            if (v == p)
                continue;
            pair<int, int> x = solve(v, u, G, hasApple);
            if (x.second)
                ret += 2 + x.first, ok = true;
        }
        return {ret, ok};
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int> > G(n);
        for (auto &v : edges) {
            G[v[0]].push_back(v[1]);
        }
        return solve(0, -1, G, hasApple).first;
    }
};
