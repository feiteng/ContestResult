class Solution {
public:
    
    int dfs(int c, unordered_map<int, vector<int>> &g, vector<bool>& h) {
        int res = 0;
        for(auto &i : g[c]) {
            res += dfs(i, g, h);
        }
        if(res > 0 || h[c]) {
            
            cout << res + 2 << endl;
            return res + 2;
        }

        return 0;
    }
    
    int minTime(int n, vector<vector<int>>& e, vector<bool>& h) {
        unordered_map<int, vector<int>> g;
        for(auto &i : e) {
            g[i[0]].push_back(i[1]);
        }
        int t = dfs(0, g, h);
        if(t >= 2) {
            return t - 2;
        }
        return 0;
    }
};
