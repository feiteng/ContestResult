#include <iostream>
#include <sstream>
#include <string>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <map>
using namespace std;

typedef long long LL;
typedef unsigned int UINT32;

class Solution {
vector<int> adjs[100001];

pair<int, bool> calc(int r, vector<bool>& hasApple) {
    int ans = 0;
    bool has = hasApple[r];
    for (int child : adjs[r]) {
        for (int j = 0; j < adjs[child].size(); ++j) {
            if (adjs[child][j] == r) {
                adjs[child].erase(adjs[child].begin() + j);
                break;
            }
        }
        pair<int, bool> t = calc(child, hasApple);
        ans += t.first;
        if (t.second) {
            ans += 2;
            has = true;
        }
    }
    return make_pair(ans, has);
}


public:
int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
    for (int i = 0; i < n; ++i) {
        adjs[i].clear();
    }

    for (vector<int> edge : edges) {
        adjs[edge[0]].push_back(edge[1]);
        adjs[edge[1]].push_back(edge[0]);
    }

    return calc(0, hasApple).first;
}
};
