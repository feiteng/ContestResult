class Solution {
public:
    vector<vector<int>> e;
    vector<bool> h;
    
    int DFS(int u,int par = -1){
        int x = 0;
        for(auto v : e[u]) if(v != par){
            int val = DFS(v,u);
            x += val;
        }
        if(x && !h[u]) x++;
        return x + h[u];
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        e.resize(n);
        for(auto v : edges) e[v[0]].push_back(v[1]),e[v[1]].push_back(v[0]);
        h = hasApple;
        return max(0,DFS(0,-1) * 2 - 2);
    }
};
