class Solution {
public:
    int dfs(vector<vector<int>>& edges, vector<bool>& hasApple, int root) {
        int ret = 0;
        for (int i = 0;i < edges[root].size();i++) {
            int x = edges[root][i];
            int temp = dfs(edges, hasApple, x);
            ret += temp;
            if (hasApple[x] || temp != 0) {
                ret += 2;
            } 
        }
        return ret;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> a(n, vector<int>());
        for (auto& edge: edges) {
            a[edge[0]].push_back(edge[1]);
        }
        return dfs(a, hasApple, 0);
    }
};
