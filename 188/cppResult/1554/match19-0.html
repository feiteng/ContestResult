class Solution {
private:
    int dfs(vector<vector<int>>& e, vector<bool>& hasApple, int cur, int parent){
        int time = 0;
        for(int i : e[cur]){
            if(i == parent) continue;
            int tmp = dfs(e, hasApple, i, cur);
            if(tmp != -1) time += tmp + 2;
        }
        if(time != 0|| hasApple[cur]==true) return time;
        return -1;
    }
    public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> e(n,vector<int>({}));
        for(auto& it : edges){
            e[it[0]].push_back(it[1]);
            e[it[1]].push_back(it[0]);
        }
        int ans = dfs(e,hasApple,0,-1);
        if(ans == -1) return 0;
        return ans;
    }
};
