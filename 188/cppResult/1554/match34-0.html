class Solution {
public:
    void prec(int nd, int an, const vector<vector<int>>& adj, vector<int>& rd) {
        for(auto sn: adj[nd])
            if(sn != an) {
                prec(sn, nd, adj, rd);
                rd[nd] += rd[sn];
            }
    }
    int dfs(int nd, int an, const vector<vector<int>>& adj, vector<int>& rd) {
        int ret = 0;
        for(auto sn: adj[nd])
            if(sn != an && rd[sn] > 0)
                ret += 2 + dfs(sn, nd, adj, rd);
        return ret;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> adj(n);
        for(auto vc: edges) {
            adj[vc[0]].push_back(vc[1]);
            adj[vc[1]].push_back(vc[0]);
        }
        vector<int> rd(n);
        for(int i = 0; i < n; i++)
            if(hasApple[i]) rd[i] = 1;
        prec(0, -1, adj, rd);
        if(!rd[0]) return 0;
        return dfs(0, -1, adj, rd);
    }
};
