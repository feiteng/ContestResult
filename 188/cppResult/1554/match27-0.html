class Solution {
 public:
  vector<vector<int>> g;
  int minTime(int n, vector<vector<int>> &edges, vector<bool> &hasApple) {
    g.resize(n);
    for (auto &e : edges) {
      g[e[0]].push_back(e[1]);
      g[e[1]].push_back(e[0]);
    }
    return helper(0, -1, hasApple).first;
  }

  pair<int, bool> helper(int x, int par, vector<bool> &hasApple) {
    int ans = 0;
    bool has = hasApple[x];
    for (int child : g[x]) {
      if (child != par) {
        auto p = helper(child, x, hasApple);
        if (p.second) {
          ans += 2 + p.first;
          has = true;
        }
      }
    }
      // cout << x << ' ' << ans << endl;
    return make_pair(ans, has);
  }
};
