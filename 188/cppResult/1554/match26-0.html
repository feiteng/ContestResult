class Solution {
public:
    vector<vector<int>>adj;
    vector<bool>app;
    int mindist(int cur, int par){
        int dist=0;
        for(auto x:adj[cur]){
            if(x==par)continue;
            int temp = mindist(x,cur);
            if(temp!=-1){
                dist+= temp + 2;
            }
        }
        if(dist==0 && !app[cur])return -1;
        
        return dist;
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        adj = vector<vector<int>>(n);
        app=hasApple;
        for(auto x:edges){
            adj[x[0]].push_back(x[1]);
            adj[x[1]].push_back(x[0]);
        }
        int temp= mindist(0,-1);
        if(temp==-1)return 0;
        return temp;
        
    }
};
