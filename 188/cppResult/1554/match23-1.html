class Solution {
public:
    set<int> edg[100010];
    
    int poi(int v, vector<bool>& hasApple, int f = -1) {
        int ret = 0;
        for (auto it : edg[v]) 
            if (it != f) {
                ret += poi(it, hasApple, v);
            }
        if ((hasApple[v] || ret != 0) && f != -1)
            ret += 2;
        return ret;
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for (int i = 0; i <= n; i++)
            edg[i].clear();
        for (auto it : edges) {
            edg[it[0]].insert(it[1]);
            edg[it[1]].insert(it[0]);
        }
        return poi(0, hasApple);
    }
};
