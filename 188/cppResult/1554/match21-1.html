class Solution {
public:
    vector<bool> apple;
    int ans = 0;
    vector<int> adj[(int)1e5 + 9];
    bool dfs(int node, int par) {
        int can = 0;
        if(node != 0)
            can = apple[node];
        for(auto it: adj[node]) {
            if(it == par) continue;
            if(dfs(it, node)) {
                ans += 2;
                can = 1;
            }

        }
        return can;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for(auto it: edges) {
            adj[it[0]].push_back(it[1]);
            adj[it[1]].push_back(it[0]);  
        }
        apple = hasApple;
        dfs(0, 0);
        return ans;
    }
};
