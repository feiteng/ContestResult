class Solution {
public:
    int dfs(int r, vector<vector<int>> &gr, vector<bool> &has, vector<bool> &visited){
        visited[r]=true;
        int ret=0, e=0;
        for(auto x : gr[r]){
            if(!visited[x] && has[x]) {
                ret+=dfs(x,gr,has,visited);
                e+=2;
            }
        }
        return ret+e;
    }

    void dfs1(int r, vector<vector<int>> &gr, vector<bool> &has, vector<bool> &visited){
        visited[r]=true;
        for(auto x : gr[r]){
            if(!visited[x]) {
                dfs1(x,gr,has,visited);
                has[r]=has[r]|has[x];
            }
        }
    }

    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> gr(n);
        for(auto x : edges){
            gr[x[0]].push_back(x[1]);
            gr[x[1]].push_back(x[0]);
        }
        vector<bool> visited(n,false);
        vector<bool> visited2(n,false);
        dfs1(0,gr,hasApple,visited2);
        return dfs(0,gr,hasApple,visited);
    }
};
