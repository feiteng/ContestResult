class Solution {
public:

    pair<int, bool> dfs(int x, int fa, vector<vector<int>>& es, vector<bool>& hasApple) {
        bool have = hasApple[x];
        int dis = 0;
        for (auto& y : es[x]) {
            if (y == fa) continue;
            auto [d, h] = dfs(y, x, es, hasApple);
            if (h) {
                have = true;
                dis += d + 2;
            }
        }
        return {dis, have};
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> es(n);
        for (auto& e : edges) {
            es[e[0]].push_back(e[1]);
            es[e[1]].push_back(e[0]);
        }
        auto [d, h] = dfs(0, -1, es, hasApple);
        if (!h) return 0;
        return d;
    }
};
