class Solution {

    vector<vector<int>> g;
    int dfs(int u, vector<bool>& a) {
        int ans = a[u];
        for(int _ : g[u]) ans += dfs(_, a);
        if(ans && !a[u]) ++ans;
        return ans;
    }
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        g.resize(hasApple.size());
        for(auto& e: edges) g[e[0]].push_back(e[1]);
        return max((dfs(0, hasApple) - 1), 0) * 2;
    }
};
