class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<int> p(n, 0);
        for(auto v : edges){
            p[v[1]] = v[0];
        }
        for(int i = 0; i < n; i++){
            if(hasApple[i]){
                int t = i;
                while(t){
                    t = p[t];
                    hasApple[t] = true;
                }
            }
        }
        int res = -1;
        for(auto b : hasApple){
            if(b)res++;
        }
        return max(res,0) * 2;
    }
};
