class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> to(n);
        for(auto p : edges){
            to[p[0]].push_back(p[1]);
            to[p[1]].push_back(p[0]);
        }
        int ans = 0;

        function<int(int,int)> dfs = [&](int cur, int p){
            int sum = hasApple[cur];
            for(int nxt : to[cur]){
                if(nxt == p) continue;
                sum += dfs(nxt, cur);
            }
            ans += sum > 0;
            return sum;
        };
        dfs(0, -1);
        if(ans > 0) ans--;
        return ans * 2;
    }
};
