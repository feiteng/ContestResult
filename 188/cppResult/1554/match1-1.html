class Solution {
public:
    pair<int, int> dfs(int u, int f, vector<vector<int>>& e, vector<bool>& hasApple, int d) {
      int ans = 0, sz = 0;
      if (hasApple[u]) {
        ans += 2 * d;
        sz += 1;
      }
      for (auto v : e[u]) {
        if (v == f) continue;
        auto p = dfs(v, u, e, hasApple, d + 1);
        ans += p.first;
        sz += p.second;
      }
      if (sz > 1 && u != 0) {

        ans -= (sz - 1) * 2;
      }
      return {ans, sz};
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
      vector<vector<int>> e(n);
      for (auto p : edges) {
        e[p[0]].push_back(p[1]);
        e[p[1]].push_back(p[0]);
      }
      return dfs(0, 0, e, hasApple, 0).first;
    }
};
