class Solution {
public:
    int sz[100005],ans;
    vector<int>ad[100005];
    
    void dfs(int src, int pre, vector<bool>& apple)
    {
        sz[src]=(apple[src]==true);

        for(int v: ad[src])
        {
            if(v==pre) continue;
            dfs(v,src,apple);
            sz[src]+=sz[v];
        }
    }
    
    void func(int src, int pre)
    {
        for(int v: ad[src])
        {
            if(v==pre) continue;
            if(sz[v])
            {
                ans++;
                func(v,src);
                ans++;
            }
        }
        
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for(int i=0; i<edges.size(); i++)
        {
            ad[edges[i][0]].push_back(edges[i][1]);
            ad[edges[i][1]].push_back(edges[i][0]);
        }
        dfs(0,-1,hasApple);
        ans=0;
        func(0,-1);
        for(int i=0; i<=n; i++) ad[i].clear();
        return ans;
    }
};
