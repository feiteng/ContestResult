class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> graph(n);
        for (auto& edge : edges) {
            int u = edge[0], v = edge[1];
            graph[u].push_back(v);
            graph[v].push_back(u);
        }
        return search(graph, hasApple, 0, -1).first;
    }
    
private:

    pair<int, bool> search(vector<vector<int>>& graph, vector<bool>& hasApple, int u, int p) {
        int dist = 0;
        bool has = hasApple[u];
        for (auto v : graph[u]) {
            if (v == p) {
                continue;
            }
            auto ret = search(graph, hasApple, v, u);
            if (ret.second) {
                has = true;
                dist += 2 + ret.first;
            }
        }
        return {dist, has};
    }
};
