class Solution {
    vector<int> G[100005];
    int f[100005];
    void dfs(int cur, int fa,  vector<bool>& hasApple){
        int sm = 0;
        for (int x: G[cur]){
            if (x == fa) continue;
            dfs(x, cur, hasApple);
            if (f[x] >= 0)
                sm += 2 + f[x];
        }
        if (sm == 0 && !hasApple[cur]) f[cur] = -1;
        else f[cur] = sm;
    }
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for (auto& v: edges){
            G[v[0]].push_back(v[1]);
            G[v[1]].push_back(v[0]);
        }
        dfs(0, -1, hasApple);
        return max(f[0], 0);
    }
};
