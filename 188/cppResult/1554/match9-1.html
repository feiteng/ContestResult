class Solution {
public:
    int f(vector<vector<int>>& g, vector<bool>& has, int s){        
        int ans = 0;
        for(int t: g[s]){

            ans += f(g, has, t);
        }        
        if(s>0 && (ans > 0 || has[s])) ans += 2;
        // printf("%d %d\n", s, ans);
        return ans;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> g(n);
        for(auto& p: edges){
            g[p[0]].emplace_back(p[1]);
        }
        return f(g, hasApple, 0);
    }
};
