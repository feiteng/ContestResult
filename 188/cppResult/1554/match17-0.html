class Solution {
public:
    vector<vector<int>> g;
    vector<bool> ap;
    int dfs(int u)
    {
        int ans=0;
        for(int v:g[u])
            ans+=dfs(v);
        if(ap[u]||ans) ans+=2;
        //cout<<u<<" "<<ans<<endl;
        return ans;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        g.assign(n,vector<int>());
        ap=hasApple;
        for(auto e:edges)
            g[e[0]].push_back(e[1]);
        int ans=dfs(0);
        return ans?ans-2:0;
    }
};
