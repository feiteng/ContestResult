class Solution {
public:
    int minTime(int N, vector<vector<int>>& edges, vector<bool>& hasApple) {
        n=N;
        adj.resize(n);
        for (auto& v:edges) {
            adj[v[0]].push_back(v[1]);
            adj[v[1]].push_back(v[0]);
        }
        int res=dfs(0, -1, hasApple);
        if (res<0) return 0;
        return res*2;
    }
    

    int dfs(int u, int parent, vector<bool>& apple) {
        int res=0;
        for (int v:adj[u]) {
            if (v==parent) continue;
            int x=dfs(v, u, apple);
            if (x>=0) res+=x+1;
        }
        if (res>0 || (res==0 && apple[u])) return res;
        return -1;
    }
    
    vector<vector<int>> adj;
    int n;
};
