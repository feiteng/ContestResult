const int MAXN = 1e5 + 50;
vector<int> edge[MAXN];
vector<bool> flag;
class Solution {
public:
    
    int dfs(int x, int n, int fa){
        int ret = 0;
        for (auto v: edge[x]){
            if (v == fa) continue;
            int cur = dfs(v, n, x);
            ret += cur;
        }
        if (ret > 0 || flag[x]) ret++;
        return ret;
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for (int i = 0; i <= n; i++) edge[i].clear();
        for (auto e: edges){
            edge[e[0]].push_back(e[1]);
            edge[e[1]].push_back(e[0]);
        }
        flag = hasApple;
        
        int ans = max(dfs(0, n, -1) - 1, 0) * 2;
        return ans;
    }
};
