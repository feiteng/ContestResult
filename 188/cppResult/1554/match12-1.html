class Solution {
public:
    int minTime(int n, vector<vector<int>>& es, vector<bool>& ha) {
        vector<int>checked(n-1,false);
        vector<int>pa(n,-1);

        for(int i = 0; i < n-1; i++){
            pa[es[i][1]] = i;
        }
        for(int i = 0; i < n; i++){
            if(ha[i]){
                int j = i;
                while(j>0 && !checked[pa[j]]){
                    checked[pa[j]] = true;
                    j = es[pa[j]][0];
                }
            }
        }
        int cnt = 0;
        for(int i = 0; i < n-1;i++){
            if(checked[i])
                cnt++;
        }
        return cnt*2;
    }
};
