class Solution {
 public:
  int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
    vector<vector<int>> adj_list(n);
    for (auto &edge : edges) {
      adj_list[edge[0]].push_back(edge[1]);
    }
    return dfs(adj_list, hasApple, 0, -1).second;
  }
 private:

  pair<bool, int> dfs(vector<vector<int>> &adj_list, vector<bool> &hasApple, int v, int p) {
    int steps = 0;
    for (int next : adj_list[v]) {
      if (next == p) continue;
      auto next_steps = dfs(adj_list, hasApple, next, v);
      if (next_steps.first) {
        steps += 2 + next_steps.second;
      }
    }
    if (steps == 0 && !hasApple[v]) {
      return {false, 0};
    } else {
      return {true, steps};
    }
  }
};
