class Solution {
public:
    int res = 0;
    bool dfs(vector<vector<int>>& nexts, vector<bool>& hasApple, int cur) {
        bool has = hasApple[cur];
        for(int i = 0; i < nexts[cur].size(); ++i) {
            if(dfs(nexts, hasApple, nexts[cur][i]))
                has = true;
        }
        if(has && cur != 0)
            res += 2;
        return has;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> nexts(n);
        for(auto v : edges) {
            nexts[v[0]].push_back(v[1]);
        }
        dfs(nexts, hasApple, 0);
        return res;
    }
};
