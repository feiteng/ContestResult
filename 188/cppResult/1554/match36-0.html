class Solution {
public:
    
    vector<int> E[111111];
    vector<bool> A;
    
    int num_apples[111111];
    
    
    int dfs(int par, int u) {
        int ret =(int)A[u];
        for(int v : E[u]) {
            if (v == par) continue;
            ret += dfs(u,v);
        }
        
        num_apples[u]=ret;
        
        return ret;
    }
    
    int Solve(int par,int u) {
        if (num_apples[u]==0) return 0;
        
        int ans=0;
        for(int v : E[u]) {
            if (v == par) continue;
            if (num_apples[v]>0) {
                ans += 2+Solve(u,v);
            }
        }
        return ans;
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& _A) {
        for(auto e : edges) {
            E[e[0]].push_back(e[1]);
            E[e[1]].push_back(e[0]);
        }
        
        A=_A;
        dfs(-1,0);
        
        return Solve(-1,0);
        
        
    }
};
