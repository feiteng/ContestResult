class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        set<pair<int, int>> seen_edge;
        unordered_set<int> seen_node;
        
        unordered_map<int, int> parent;
        
        for (int i = 0; i < edges.size(); i++) {
            parent[edges[i][1]] = edges[i][0];
        }
        
        for (int i = 0; i < n; i++) {
            if (hasApple[i]) {
                int cur = i;
                while (seen_node.find(cur) == seen_node.end() && cur != 0) {
                    seen_node.insert(cur);
                    seen_edge.insert(make_pair(cur, parent[cur]));
                    cur = parent[cur];
                }
            }
        }
        
        return seen_edge.size() * 2;
    }
};
