class Solution {
    // cost, has apple
    pair<int, bool> Traverse(unordered_map<int, unordered_set<int>> &graph, vector<bool> &hasApple, int cur, int parent) {
        int total_cost = 0;
        bool has_apple = hasApple[cur];

        for (auto next : graph[cur]) {
            if (next == parent) {
                continue;
            }
            auto p = Traverse(graph, hasApple, next, cur);
            if (p.second) {
                has_apple = true;
                total_cost += (2 + p.first);
            }
        }
        
        return {total_cost, has_apple};
    }
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        unordered_map<int, unordered_set<int>> graph;
        
        for (auto &edge : edges) {
            graph[edge[0]].insert(edge[1]);
            graph[edge[1]].insert(edge[0]);
        }
        
        pair<int, bool> p = Traverse(graph, hasApple, 0, 0);
        return p.first;
    }
};
