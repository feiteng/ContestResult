class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {

        vector<int> fa(n);
        vector<bool> ans(n);
        for (auto i: edges) {
            fa[i[1]] = i[0];
        }

        for (int i = 0; i < n; i++) {
            if (hasApple[i]) {
                ans[i] = 1;
            }
        }
        int o = 0;
        for (int i = n-1; i >= 1; i--) {
            if (ans[i]) ans[fa[i]] = 1, o++;
        }

        return o*2;
    }
};
