class Solution {
public:
     vector<int> e[100005];
    int ans=0;
    bool dfs(int now,vector<bool>& hasApple)
    {
        bool res=0;
        if(hasApple[now]) res=1;
        for(auto t:e[now])
        {
            if(dfs(t,hasApple)) 
            {
                 res=1;
                ans++;
            }
        }
        return res;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        
        for(auto t:edges)
        {
            e[t[0]].push_back(t[1]);
        }
        dfs(0,hasApple);
        return ans*2;
    }
};
