class Solution {
public:

    typedef vector<vector<int>> Mat;
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        Mat G(n);
        for(auto& v: edges) {
            G[v[0]].push_back(v[1]);
            G[v[1]].push_back(v[0]);
        }
        return dfs(G, hasApple, 0, -1);
    }
    
    int dfs(Mat& G, vector<bool>& has, int node, int parent) {
        int sub = 0;
        for(int n: G[node]) {
            if(n != parent) {
                sub += dfs(G, has, n, node);
            }
        }
        int delta = node == 0 ? 0 : 2;
        if(sub != 0) {
            return sub + delta;
        } else {
            return has[node] ? delta : 0;
        }
    }
};
