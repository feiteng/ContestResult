const int maxn = 1e5 + 10;
class Solution {
public:
    vector<int> g[maxn];
    long long ret;
    bool ok[maxn];
    vector<bool> has;
        

    void DFS(int v , int p){
        ok[v] |= has[v];
        for(int u : g[v]){
            if(u != p){
                DFS(u , v);
                ok[v] |= ok[u];
                if(ok[u])ret += 2;
            }
        }
    }
    
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        ret = 0;
        has = hasApple;
        for(vector<int>& e : edges){
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        DFS(0 , -1);
        return ret;
    }
};
