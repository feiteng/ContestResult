const int maxn = 1e5 + 5;
vector<int> G[maxn];
bool has[maxn];

class Solution {

    
    int dfs(int now, int fa) {
        
        int r = 0;
        for (int next : G[now]) {
            if (next == fa) continue;
            r += dfs(next, now);
        }
        
        if (r != 0) return r + 2;
        
        return has[now]*2;
    }
    
    
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        
        for (auto& e : edges) {
            G[e[0]].push_back(e[1]);
            G[e[1]].push_back(e[0]);
        }
        
        for (int i = 0; i < n; i++) has[i] = hasApple[i];
        
        int ans = dfs(0, -1);
        
        
        
        
        for (auto& g : G) g.clear();
        return ans == 0 ? 0 : ans-2;
    }
};
