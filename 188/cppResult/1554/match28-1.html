class Solution {
    vector<vector<int>> graph;
    vector<bool> has;
    int dfs(int i, int p) {
        // cout<<i<<endl;
        int res=0;
        for(auto j:graph[i]) {
            if(j==p) continue;
            // cout<<i<<' '<<j<<endl;
            int tmp=dfs(j,i);

            if(tmp==1e9) continue;
            res+=tmp+2;
        }
        if(res==0&&!has[i]) return 1e9;
        return res;
    }
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        graph.resize(n);
        for(auto v : edges) {
            graph[v[0]].push_back(v[1]);
            graph[v[1]].push_back(v[0]);
        }
        has=hasApple;
        int res= dfs(0,-1);
        return res==1e9?0:res;
        
    }
};
