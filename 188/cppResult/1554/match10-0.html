class Solution {
public:
    vector<vector<int>> g;
    vector<int> t;
    int ans;
    void calc(int at, int p) {
        for (int to: g[at]) {
            if (to==p) continue;
            calc(to, at);
            t[at] += t[to];
        }
    }
    void dfs(int at, int p) {
        for (int to: g[at]) {
            if (to==p) continue;
            if (t[to]>0) {
                ans++;
                dfs(to, at);
                ans++;
            }
        }
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        g.resize(n);
        t.resize(n);
        ans = 0;
        for (int i=0; i<n; i++) {
            t[i] = hasApple[i];
        }
        for (auto ed: edges) {
            g[ed[0]].push_back(ed[1]);
            g[ed[1]].push_back(ed[0]);
        }
        calc(0, -1);
        // for (int i=0; i<n; i++) cout<<t[i]<<' ';
        // cout<<endl;
        dfs(0, -1);
        return ans;
    }
};
