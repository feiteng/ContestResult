const int MX = 100005;
vector<int> g[MX];
vector<bool> has;

class Solution {
public:

	int dfs(int u, int p){
		int res = 0;

		for(int v : g[u]){
			if(v == p) continue;
			int c = dfs(v, u);
			if(c >= 0){
				res += c + 2;
			}
		}
		if(res == 0 && !has[u]) return -1;

		return res;
	}

    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
		has = hasApple;
		for(int i=0; i<n; i++){
			g[i].clear();
		}
		for(vector<int> &e : edges){
			int u = e[0], v = e[1];
			g[u].push_back(v);
			g[v].push_back(u);
		}

		return max(0, dfs(0, -1));
        
    }
};
