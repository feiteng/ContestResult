class Solution {
    vector<int> adj[100001];
    int sol;
    int dfs(int source, int parent, vector<bool> &hasApple) {
        int count = hasApple[source];
        for(auto i : adj[source]) {
            if (i == parent) {
                continue;
            }

            int tmp = dfs(i, source, hasApple);
            if (tmp > 0) {
                count+= tmp;
                sol += 2;
            }
        }
        return count;
    }
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        for(int i = 0;i <= n;i++) {
            adj[i].clear();
        }
        for(auto i :edges) {
            adj[i[0]].push_back(i[1]);
            adj[i[1]].push_back(i[0]);
        }
        sol = 0;
        dfs(0, -1, hasApple);
        return sol;
    }
};
