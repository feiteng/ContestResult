class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
      unordered_map<int,int> parent;
      
      for (auto &vec: edges) {
        parent[vec[1]] = vec[0];
      }
      
      unordered_set<int> aps;
      aps.insert(0);
      for (int i=0; i<n; ++i) {
        if (hasApple[i]) {
          int cur = i;
          while (cur > 0) {
            aps.insert(cur);
            cur = parent[cur];
          }
        }
      }
      
      return 2*(aps.size()-1);
      
      
      
    }
};
