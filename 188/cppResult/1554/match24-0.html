const int maxn = 1e5 + 7;
class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& a) {
        apple = a;
        for(vector<int>& e : edges){
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        dfs(0 , -1);
        return ret;
    }
private:
    vector<int> g[maxn];
    long long ret = 0;
    bool ok[maxn];
    vector<bool> apple;
    void dfs(int v , int p){
        ok[v] |= apple[v];
        for(int u : g[v]){
            if(u != p){
                dfs(u , v);
                ok[v] |= ok[u];
                if(ok[u])ret += 2;
            }
        }
    }
};
