class Solution {
public:
    vector<int> dfs(int curr, vector<vector<int>>& g, vector<bool>& ha) {
        if (g[curr].empty()) {
            return {ha[curr], 0};
        }
        bool hf = ha[curr];
        int steps(0);
        for (int nxt: g[curr]) {

            vector<int> fb = dfs(nxt, g, ha);
            if (fb[0] == 1) steps += fb[1] + 2;
            hf |= fb[0];
        }
        //cout << hf << ' ' << curr << endl;
        return {hf, steps};
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<vector<int>> g(n);
        for (const auto& e: edges) {
            g[e[0]].push_back(e[1]);
        }
        return dfs(0, g, hasApple)[1];
    }
};
