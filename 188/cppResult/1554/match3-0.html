class Solution {
public:
    int dfs(vector<int>* e, vector<bool>& h, int x) {
        int ans = 0;
        for (int i : e[x])
            ans += dfs(e, h, i);
        if (x != 0 && (ans || h[x])) ans += 2;
        return ans;
    }
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
        vector<int> e[n];
        for (vector<int> i : edges) e[i[0]].push_back(i[1]);
        return dfs(e, hasApple, 0);
    }
};
