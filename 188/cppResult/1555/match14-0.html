class Solution {
    typedef long long LL;
    const LL MOD = 1e9 + 7;
public:
    LL dp[55][55][15];
    int ways(vector<string>& pizza, int k) {
        memset(dp, -1, sizeof(dp));
        int n = pizza.size();
        int m = pizza[0].size();
        dp[n][m][0] = 1;
        return solve(0, 0, k - 1, pizza);
    }
    LL solve(int x, int y, int z, vector<string> &pizza) {
        if (~dp[x][y][z])
            return dp[x][y][z];
        dp[x][y][z] = 0;
        int n = pizza.size();
        int m = pizza[0].size();
        if (z == 0) {
            for (int i = x ; i < n ; i++)
                for (int j = y ; j < m ; j++)
                    if (pizza[i][j] == 'A')
                        dp[x][y][z] = 1;
        } else {
            bool ok = false;
            for (int i = x ; i < n ; i++) {
                for (int j = y ; j < m && !ok ; j++)
                    if (pizza[i][j] == 'A')
                        ok = true;
                if (!ok)
                    continue;
                dp[x][y][z] += solve(i + 1, y, z - 1, pizza);
                dp[x][y][z] %= MOD;
            }
            ok = false;
            for (int j = y ; j < m ; j++) {
                for (int i = x ; i < n && !ok ; i++)
                    if (pizza[i][j] == 'A')
                        ok = true;
                if (!ok)
                    continue;
                dp[x][y][z] += solve(x, j + 1, z - 1, pizza);
                dp[x][y][z] %= MOD;
            }
                
        }
        return dp[x][y][z];
    }
};
