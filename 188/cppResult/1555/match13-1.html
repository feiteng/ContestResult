#define LL long long
const LL mod = 1e9 + 7;

LL dp[55][55][12];
LL s[55][55];
int n, m;

class Solution {
public:
    int dfs(int x, int y, int p) {
      if (p < 0) {
        return 0;
      }
      if (dp[x][y][p] >= 0) {
        return dp[x][y][p];
      }


      LL ans = 0;
      for (int i = 0; i < x; i++) {
        if (get(i, y, x - 1, m - 1) > 0) {
          ans += dfs(i, y, p - 1);
          ans %= mod;
        }
      }
      for (int i = 0; i < y; i++) {
        if (get(x, i, n - 1, y - 1) > 0) {
          ans += dfs(x, i, p - 1);
          ans %= mod;
        }
      }
      dp[x][y][p] = ans;
      return ans;
    }
    int get(int x0, int y0, int x1, int y1) {
      int w = s[x1][y1];
      if (x0 > 0) w -= s[x0 - 1][y1];
      if (y0 > 0) w -= s[x1][y0 - 1];
      if (x0 > 0 && y0 > 0) w += s[x0 - 1][y0 - 1];
      return w;
    }
    int ways(vector<string>& a, int k) {
      n = a.size(), m = a[0].size();

      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          s[i][j] = 0;
          if (a[i][j] == 'A') {
            s[i][j] = 1;
          }
          if (j > 0) s[i][j] += s[i][j - 1];
        }
      }
      for (int j = 0; j < m; j++) {
        for (int i = 1; i < n; i++) {
          s[i][j] += s[i - 1][j];
        }
      }


      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          for (int l = 0; l < k; l++) {
            dp[i][j][l] = -1;
          }
        }
      }
      dp[0][0][0] = 1;

      LL ans = 0;
      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          if (get(i, j, n - 1, m - 1) > 0) {
            ans += dfs(i, j, k - 1);
            //cout << i << " " << j << " " << dfs(i, j, k - 1) << endl;
            ans %= mod;
          }
        }
      }
      return ans;
    }
};
