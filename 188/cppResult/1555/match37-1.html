template<typename S, typename T>
ostream& operator<<(ostream& out, const pair<S, T> p) {
    out << "(" << p.first << "," << p.second << ")";
    return out;
}

template<typename T>
ostream& operator<<(ostream& out, const vector<T>& v) {
    for (auto a: v)
        out << a << " ";
    return out;
}

template<typename T>
ostream& operator<<(ostream& out, const set<T>& S) {
    for (auto a: S)
        cout << a << " ";
    return out;
}

template<typename T>
ostream& operator<<(ostream& out, const multiset<T>& S) {
    for (auto a: S)
        cout << a << " ";
    return out;
}

template<typename S, typename T>
ostream& operator<<(ostream& out, const map<S, T>& M) {
    for (auto m: M)
        cout << "(" << m.first << "->" << m.second << ") ";
    return out;
}

template<typename S, typename T>
pair<S, T> operator+(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first + b.first, a.second + b.second);
}

template<typename S, typename T>
pair<S, T> operator-(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first - b.first, a.second - b.second);
}

class Solution {
public:
    vector<vector<int>> prefix;
    int mod = 1e9 + 7;

    int dp[55][55][12];
    
    int query_sum_value(int x1, int y1, int x2, int y2) {
        int ans = prefix[x2][y2];
        if (x1)
            ans -= prefix[x1-1][y2];
        if (y1)
            ans -= prefix[x2][y1-1];
        if (x1 && y1)
            ans += prefix[x1-1][y1-1];
        return ans;
    }
    
    int ways(vector<string>& pizza, int max_cut) {
        int n = pizza.size();

        int m = pizza[0].size();
        max_cut--;
        prefix.resize(n);
        for (int i = 0; i < n; i++) {
            prefix[i].assign(m, 0);
            for (int j = 0; j < m; j++) {
                if (pizza[i][j] == 'A') {
                    prefix[i][j] = 1;
                }
            }
        }
        /*
        for (int i = 0; i < n; i++) {
            cout << "prefix[" << i << "]: " << prefix[i] << endl;
        }
        */
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i)

                    prefix[i][j] += prefix[i-1][j];
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (j)
                    prefix[i][j] += prefix[i][j-1];
            }
        }
        /*
        for (int i = 0; i < n; i++) {
            cout << "prefix[" << i << "]: " << prefix[i] << endl;
        }
        */
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (query_sum_value(i, j, n - 1, m - 1)) {
                    dp[i][j][0] = 1;
                }
            }
        }

        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                for (int k = 1; k <= max_cut; k++) {
                    for (int l = i + 1; l < n; l++) {
                        if (query_sum_value(i, j, l - 1, m - 1)) {
                            dp[i][j][k] += dp[l][j][k-1];
                            if (dp[i][j][k] >= mod)
                                dp[i][j][k] -= mod;
                        }
                    }
                    for (int l = j + 1; l < m; l++) {
                        if (query_sum_value(i, j, n - 1, l - 1)) {
                            dp[i][j][k] += dp[i][l][k-1];
                            if (dp[i][j][k] >= mod)
                                dp[i][j][k] -= mod;
                        }
                    }
                }
            }
        }
        /*
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k <= max_cut; k++) {
                    cout << "dp[" << i << "][" << j << "][" << k << "]: " << dp[i][j][k] << endl;
                }
            }
        }
        */
        return dp[0][0][max_cut];
    }
};
