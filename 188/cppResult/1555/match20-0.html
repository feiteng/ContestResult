class Solution {
public:
    const int p=1e9+7;
    int dp[51][51][11],m,n;
    int dfs(vector<string>& pizza,int r,int c,int k){
        if(dp[r][c][k]!=-1)return dp[r][c][k];
        if(r>=m||c>=n)return 0;
        if(k==0){
            for(int i=r;i<m;++i){
                for(int j=c;j<n;++j){
                    if(pizza[i][j]=='A')return dp[r][c][k]=1;
                }
            }
            return dp[r][c][k]=0;
        }
        int tmp=0,f=0;
        for(int i=r;i<m-1;++i){
            if(f==0){
                for(int j=c;j<n;++j){
                    if(pizza[i][j]=='A'){
                        f=1;break;
                    }
                }
            }
            if(f)tmp=(tmp+dfs(pizza,i+1,c,k-1))%p;
        }
        f=0;
        for(int j=c;j<n-1;++j){
            if(f==0){
                for(int i=r;i<m;++i){
                    if(pizza[i][j]=='A'){
                        f=1;break;
                    }
                }
            }
            if(f)tmp=(tmp+dfs(pizza,r,j+1,k-1))%p;
        }
        return dp[r][c][k]=tmp;
    }
    int ways(vector<string>& pizza, int k) {
        m=pizza.size(),n=pizza[0].size(); 
        memset(dp,-1,sizeof(dp));
        return dfs(pizza,0,0,k-1);
    }
};
