class Solution {
public:
    bool has(vector<string>& pizza, int x1, int x2, int y1, int y2) {
        for (int i = x1; i <= x2; ++i)
            for (int j = y1; j <= y2; ++j)
                if (pizza[i][j] == 'A')
                    return 1;
        return 0;
    }
    int ways(vector<string>& pizza, int k) {
        int n = pizza.size(), m = pizza[0].size();
        int f[k][n][m];
        memset(f, 0, sizeof(f));
        for (int i = n - 1; i >= 0; --i)
            for (int j = m - 1; j >= 0; --j)
                f[0][i][j] = i + 1 < n && f[0][i + 1][j] || j + 1 < m && f[0][i][j + 1] || pizza[i][j] == 'A';
        for (int i = 1; i < k; ++i)
            for (int j = n - 1; j >= 0; --j)
                for (int kk = m - 1; kk >= 0; --kk) {
                    bool flag = 0;
                    for (int l = 1; j + l < n; ++l) {
                        if (!flag) flag = has(pizza, j + l - 1, j + l - 1, kk, m - 1);
                        if (flag) f[i][j][kk] = (f[i][j][kk] + f[i - 1][j + l][kk]) % 1000000007;
                    }
                    flag = 0;
                    for (int l = 1; kk + l < m; ++l) {
                        if (!flag) flag = has(pizza, j, n - 1, kk + l - 1, kk + l - 1);
                        if (flag) f[i][j][kk] = (f[i][j][kk] + f[i - 1][j][kk + l]) % 1000000007;
                    }
                }
        return f[k - 1][0][0];
    }
};
