class Solution {
public:
    //map<pair<int, int>> data;
    int m;
    int n;
    int ways(vector<string>& pizza, int k) {
        m = pizza.size();
        n = pizza[0].size();
        vector<vector<int>> sums(m, vector<int>(n, 0));

        vector<vector<vector<int>>> dp(m, vector<vector<int>>(n, vector<int>(k, -1)));
        for (int i = m - 1; i >= 0; --i) {
            for (int j = n - 1; j >= 0; --j) {
                sums[i][j] = pizza[i][j] == 'A';
                if (i < m - 1) {
                    sums[i][j] += sums[i+1][j];
                }
                if (j < n - 1) {
                    sums[i][j] += sums[i][j+1];
                }
                if (i < m - 1 && j < n - 1) {
                    sums[i][j]-= sums[i+1][j+1];
                }
            }
        }
        --k;
        return dfs(0, 0, k, sums, dp);
    }

    int dfs(int x, int y, int k, vector<vector<int>>& sums, vector<vector<vector<int>>>& dp) {
        if (dp[x][y][k] != -1) {
            return dp[x][y][k];
        }
        if (sums[x][y] < k + 1) {
            return 0;
        }
        if (k == 0) {
            dp[x][y][k] = 1;
            return 1;
        }

        int res = 0;
        for (int i = x + 1; i < m; ++i) {
            if (sums[i][y] < sums[x][y]) {
                res += dfs(i, y, k-1, sums, dp);
                res %= 1000000007;
            }
        }
        for (int j = y + 1; j < n; ++j) {
            if (sums[x][j] < sums[x][y]) {
                res += dfs(x, j, k-1, sums, dp);
                res %= 1000000007;
            }
        }
        dp[x][y][k] = res;
        return res;
    }
};
