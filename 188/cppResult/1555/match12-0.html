class Solution {
public:
    int psa[51][51];
    int dp[2][51][51][51][51];
    bool isEmpty(int a, int b, int c, int d) {
        return psa[b][d] - psa[b][c-1] - psa[a-1][d] + psa[a-1][c-1] == 0;
    }
    int ways(vector<string>& pizza, int k) {
        k--;
        int row = pizza.size(), col = pizza[0].length();
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (pizza[i][j]=='A') psa[i+1][j+1]++;
            }
        }
        for (int i = 1; i <= row; i++) {
            for (int j = 1; j <= col; j++) {
                psa[i][j] += psa[i-1][j] + psa[i][j-1] - psa[i-1][j-1];
            }
        }
        for (int i = 1; i <= row; i++) {
            for (int j = i; j <= row; j++) {
                for (int k = 1; k <= col; k++) {
                    for (int l = k; l <= col; l++) {
                        dp[0][i][j][k][l] = !isEmpty(i,j,k,l);
                    }
                }
            }
        }
        for (int q = 1; q <= k; q++) {
            for (int a = 1; a <= row; a++) {
                    int b = row;
                    for (int c = 1; c <= col; c++) {
                            int d=col;
                            dp[q%2][a][b][c][d] = 0;
                            for (int i = a; i < b; i++) {
                                // cut after i
                                if (!isEmpty(a,i,c,d)) {
                                    dp[q%2][a][b][c][d] += dp[1-q%2][i+1][b][c][d];
                                    dp[q%2][a][b][c][d] %= 1000000007;
                                }
                            }
                            for (int i = c; i < d; i++) {
                                if (!isEmpty(a,b,c,i)) {
                                    dp[q%2][a][b][c][d] += dp[1-q%2][a][b][i+1][d];
                                    dp[q%2][a][b][c][d] %= 1000000007;
                                }
                            
                        
                    }
                }
            }
        }
        return dp[k%2][1][row][1][col];
    }
};
