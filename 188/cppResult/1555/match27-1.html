class Solution {
public:
    int ways(vector<string>& pizza, int k) {
        vector<vector<vector<int>>> dp(pizza.size(), vector<vector<int>>(pizza[0].size(), vector<int>(k + 1, -1)));
        return aux(pizza, 0, 0, k, dp);
    }
    int aux(vector<string>& pizza, int row, int col, int k, vector<vector<vector<int>>>& dp) {
        if (dp[row][col][k] != -1) return dp[row][col][k];

        if (k == 1) {
            for (int i = row;i < pizza.size();++i) {
                for (int j = col; j< pizza[0].size();++j) {
                    if (pizza[i][j] == 'A') return 1;
                }
            }
            dp[row][col][k] = 0;
            return 0;
        }
        // return 0;

        int result = 0;
        bool hasApple = false;
        for (int i = row;i < pizza.size() - 1;++i) {
            for (int j = col;j < pizza[0].size();++j) {
               if (pizza[i][j] == 'A' || hasApple) {
                    hasApple = true;
                    break;
               }
            }
            if (hasApple) {
                result += aux(pizza, i + 1, col, k - 1, dp);
                result %= 1000000007;    

            }
            
        }
        hasApple = false;
        for (int j = col;j < pizza[0].size() - 1;++j) {
            for (int i = row;i < pizza.size();++i) {
               if (pizza[i][j] == 'A' || hasApple) {
                    hasApple = true;
                    break;
               }
            }
            if (hasApple) {
                result += aux(pizza, row, j + 1, k - 1, dp);
                result %= 1000000007;
            }
        }
        dp[row][col][k] = result;
        return result;
    }
};
