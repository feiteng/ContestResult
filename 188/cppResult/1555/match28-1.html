class Solution {
public:
    int ways(vector<string>& pizza, int k) {
        int n = pizza.size();
        int m = pizza[0].size();
        int counts[n][m];

        int mod = 1000000007;
        for(int i = 0; i < n; i++)for(int j = 0; j < m; j++)counts[i][j] = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                if(pizza[i][j] == 'A'){

                    for(int p = 0; p <= i; p++){
                        for(int q = 0; q <= j; q++){
                            counts[p][q]++;
                        }
                    }
                }
            }
        }
        long long w[n][m][k+1];
        for(int i = 0; i < n; i++)for(int j = 0; j < m; j++)for(int x = 0; x <= k; x++)w[i][j][x] = 0;
        w[0][0][0] = 1;

        for(int t = 0; t < k; t++){
            for(int i = 0; i < n; i++){
                for(int j = 0; j < m; j++){
                    for(int x = i+1; x < n; x++){
                        if(counts[x][j] >= 1 && counts[i][j] > counts[x][j]){
                            w[x][j][t+1] = (w[x][j][t+1] + w[i][j][t]) % mod;
                        }
                    }
                    for(int y = j+1; y < m; y++){
                        if(counts[i][y] >= 1 && counts[i][j] > counts[i][y]){
                            w[i][y][t+1] = (w[i][y][t+1] + w[i][j][t]) % mod;
                        }
                    }
                }
            }
        }
        long long res = 0;
        for(int i = 0; i < n; i++)for(int j = 0; j < m; j++){
            res += w[i][j][k-1];
        }

        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
            //cout << w[i][j][2];
            //cout << counts[i][j];
            }
            //cout << endl;
        }
        return res % mod;
    }
};
