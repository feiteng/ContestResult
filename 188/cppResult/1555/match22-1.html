class Solution {
public:
    int mod=1e9+7;
    int n,m;
    int sum[55][55];
    long long dp[55][55][15];
    long long dfs(int x,int y,vector<string>& a,int k)
    {
        if(dp[x][y][k]!=-1) return dp[x][y][k];
        if(k==1)
        {
            int ans=0;
            if(sum[x][y]) ans=1;
            return ans;
        }
        

        long long ans=0;
        
        for(int i=x+1;i<n;i++)
        {
            if(sum[i][y]<sum[x][y])
            {
                ans+=dfs(i,y,a,k-1);
                ans%=mod;
            }
        }
        
        for(int j=y+1;j<m;j++)
        {
            if(sum[x][j]<sum[x][y])
            {
                ans+=dfs(x,j,a,k-1);
                ans%=mod;
            }
        }
        
        dp[x][y][k]=ans;
        
        return ans;
  
    }
    int ways(vector<string>& a, int k) {
        n=a.size();
        m=a[0].size();
        memset(sum,0,sizeof sum);
        memset(dp,-1,sizeof dp);

        for(int i=n-1;i>=0;i--)
        {
            for(int j=m-1;j>=0;j--)
            {
                sum[i][j]=sum[i+1][j]+sum[i][j+1]-sum[i+1][j+1]+(a[i][j]=='A');
            }
        }
        
        return dfs(0,0,a,k);
        
    }
};
