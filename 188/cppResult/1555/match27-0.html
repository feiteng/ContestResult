class Solution {
private:
    int m, n;
    long long visited2[51][51][11] = {0};
    long long visited1[51][51][11] = {0};
    long long MOD = 1e9 + 7;
    int total = 0;
public:
    int ways(vector<string>& pizza, int k) {
        m = pizza.size(), n = pizza[0].size();
        return helper(pizza, 0, k, 0, 0);
    }
    
    long long helper(vector<string>& pizza, int index, int k, int curX, int curY)
    {
        if(index == k - 1) 
        {
            for(int i = curX; i < m; ++i)
            {
                for(int j = curY; j < n; ++j)
                {
                    if(pizza[i][j] == 'A') return 1;
                }
            }
            return 0;
        }
        if(curX >= m || curY >= n) return 0;
        if(visited1[curX][curY][index] != 0) return visited2[curX][curY][index];
        long long res = 0;
        bool tmp1 = false;
        for(int i = curX; i < m - 1; ++i)
        {
            for(int j = curY; j < n; ++j)
            {
                if(pizza[i][j] == 'A' || tmp1)
                {
                    res = (res + helper(pizza, index + 1, k, i + 1, curY)) % MOD;
                    tmp1 = true;
                    break;
                }
            }
        }
        
        //horizontal
        bool tmp2 = false;
        for(int j = curY; j < n - 1; ++j)
        {
            for(int i = curX; i < m; ++i)
            {
                if(pizza[i][j] == 'A' || tmp2)
                {
                    tmp2 = true;
                    res = (res + helper(pizza, index + 1, k, curX, j + 1)) % MOD;
                    break;
                }
            }
        }
        visited1[curX][curY][index] = 1;
        return visited2[curX][curY][index] = res;
    }
};
