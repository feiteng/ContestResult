class Solution {
public:
    const int MOD = 1000000007;
    int DFS(const vector<string>& pizza, const int x, const int y, const int k, vector<vector<vector<int>>>& dp) {
//        cout << x << ' ' << y << ' ' << k << endl;
        if (dp[x][y][k] != -1) {
            return dp[x][y][k];
        }
        int r = pizza.size();
        int c = pizza[0].size();
        dp[x][y][k] = 0;

        if (k == 1) {
            for (int i = x; i < r; ++i) {
                for (int j = y; j < c; ++j) {
                    if (pizza[i][j] == 'A') {
                        dp[x][y][k] = 1;
                        return dp[x][y][k];
                    }
                }
            }
            return dp[x][y][k];

        }
        bool flag = false;
        for (int i = x; i < r; ++i) {
            if (!flag) {
                for (int j = y; j < c; ++j) {
                    if (pizza[i][j] == 'A') {
                        flag = true;
                        break;
                    }
                }
            }
            if (flag) {
                if (i+1 < r) {
                    dp[x][y][k] += DFS(pizza, i+1, y, k-1, dp);
                    dp[x][y][k] %= MOD;
                }
            }
        }

        flag = false;
        for (int j = y; j < c; ++j) {
            if (!flag) {
                for (int i = x; i < r; ++i) {
                    if (pizza[i][j] == 'A') {
                        flag = true;
                        break;
                    }
                }
            }
            if (flag) {
                if (j+1 < c) {
                    dp[x][y][k] += DFS(pizza, x, j+1, k-1, dp);
                    dp[x][y][k] %= MOD;
                }
            }
        }        
//        cout << x << ' ' << y << ' ' << k << ' ' << dp[x][y][k] << endl;
        return dp[x][y][k];
    }
    int ways(vector<string>& pizza, int k) {
        int r = pizza.size();
        int c = pizza[0].size();
        vector<vector<vector<int>>> dp(r, vector<vector<int>>(c, vector<int>(k+1, -1)));
        return DFS(pizza, 0, 0, k, dp);
    }
};
