const int mod = 1e9 + 7;
class Solution {
public:
    vector<string>p;
    //int pre[55][55];
    long long dp[55][55][12];
    int n, m;
    long long dfs(int x, int y, int k) {
        if (~dp[x][y][k]) {
            return dp[x][y][k];
        }

        dp[x][y][k] = 0;
        if (!k) {
            for (int i = x ; i < n ; i++)
                for (int j = y ; j < m ; j++)
                    if (p[i][j] == 'A')
                        return dp[x][y][k] = 1;

            return dp[x][y][k] = 0;
        } else {
            bool fi = 0;
            for (int i = x ; i < n ; i++) {
                for (int j = y ; j < m && !fi ; j++) {
                    if (p[i][j] == 'A')
                        fi = 1;
                }
                if (fi) {
                    dp[x][y][k] += dfs(i + 1, y, k - 1);
                    dp[x][y][k] %= mod;
                }

            }
            fi = 0;
            for (int j = y ; j < m ; j++) {
                for (int i = x ; i < n && !fi ; i++)
                    if (p[i][j] == 'A')
                        fi = 1;
                if (!fi)
                    continue;
                dp[x][y][k] += dfs(x, j + 1, k - 1);
                dp[x][y][k] %= mod;
            }
 
        }
        return dp[x][y][k];
 
    }
    int ways(vector<string>& pizza, int k) {
        p = pizza;
        //memset(pre, 0, sizeof(pre));
        memset(dp, -1, sizeof(dp));
 
        n = pizza.size();
        m = pizza[0].size();
        dp[n][m][0] = 1;
        return dfs(0, 0, k - 1);
    }
};
