class Solution {
public:
    int mod=1000000007;
    int dp[50][50][11];
    int help(vector<vector<int>>& ac, int a, int b, int c, int d, int k) {
        //cout<<a<<" "<<b<<" "<<c<<" "<<d<<" "<<k<<" "<<endl;
        if (k==1) {
            if (ac[c+1][d+1]-ac[c+1][b]-ac[a][d+1]+ac[a][b]>=1) {
                return 1;
            } else {
                return 0;
            }
        }
        if (dp[a][b][k]==-1) {
            dp[a][b][k]=0;
            for (int i=a; i<c; i++) {
                if ((ac[i+1][d+1]-ac[i+1][b]-ac[a][d+1]+ac[a][b]>=1) && (ac[c+1][d+1]-ac[c+1][b]-ac[i+1][d+1]+ac[i+1][b]>=k-1)) {
                    dp[a][b][k]+=help(ac, i+1, b, c, d, k-1);
                    dp[a][b][k]%=mod;
                }
            }
            for (int i=b; i<d; i++) {
                if ((ac[c+1][i+1]-ac[c+1][b]-ac[a][i+1]+ac[a][b]>=1)  && (ac[c+1][d+1]-ac[c+1][i+1]-ac[a][d+1]+ac[a][i+1]>=k-1)) {
                    dp[a][b][k]+=help(ac, a, i+1, c, d, k-1);
                    dp[a][b][k]%=mod;
                }
            }
        }
        return dp[a][b][k];
    }
    int ways(vector<string>& pizza, int k) {
        vector<vector<int>> ac(pizza.size()+1, vector<int>(pizza[0].size()+1, 0));
        for (int a=0; a<50; a++) {
            for (int b=0; b<50; b++) {
                for (int k=0; k<11; k++) {
                    dp[a][b][k]=-1;
                }
            }
        }
        for (int i=0; i<pizza.size(); i++) {
            for (int j=0; j<pizza[i].size(); j++) {
                ac[i+1][j+1]=ac[i+1][j]+ac[i][j+1]-ac[i][j];
                if (pizza[i][j]=='A') {
                    ac[i+1][j+1]++;
                }
            }
        }
        return help(ac, 0, 0, pizza.size()-1, pizza[0].size()-1, k);
    }
};
