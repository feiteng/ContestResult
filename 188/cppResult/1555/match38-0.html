class Solution {
public:
    #define MOD 1000000007
    int dp[55][55][55], acu[55][55], n, m;
    
    //int get(){}
    int ways(vector<string>& pizza, int k) {
        n = pizza.size(), m = pizza[0].size();k--;
        for(int i = n - 1; i >= 0; i--) {
            for(int j = m - 1; j >= 0; j--) {
                acu[i][j] = acu[i + 1][j] + acu[i][j + 1] - acu[i + 1][j + 1] + 
                    (pizza[i][j] == 'A'? 1: 0);
                dp[i][j][0] = (acu[i][j]? 1: 0);
                
                for(int u = 1; u <= k; u++) {
                    for(int r = i + 1; r < n; r++) {
                        if(acu[r][j] == acu[i][j]) continue;
                        dp[i][j][u] = (dp[i][j][u] + dp[r][j][u - 1]) % MOD;
                    }
                    for(int r = j + 1; r < m; r++) {
                        if(acu[i][r] == acu[i][j]) continue;
                        dp[i][j][u] = (dp[i][j][u] + dp[i][r][u - 1]) % MOD;
                    }
                }
            }
        }
        return dp[0][0][k];
    }
};
