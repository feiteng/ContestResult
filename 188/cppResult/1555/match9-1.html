class Solution {
public:
    bool check(int x1, int y1, int x2, int y2, const vector<string>& pizza) {
        for (int i = x1; i <= x2; i++)
            for (int j = y1; j <= y2; j++)
            if (pizza[i][j] == 'A')
                return true;
        
        return false;
    }
    
    int ways(vector<string>& pizza, int k) {
        const int mod = 1000000007;
        
        int r = pizza.size();
        int c = pizza[0].size();
        
        int f[20][60][60];
        
        memset(f, 0, sizeof f);
        

        for (int i = 0; i < r; i++)
            for (int j = 0; j < c; j++)
                f[1][i][j] = check(i, j, r - 1, c - 1, pizza);
        
        for (int l = 2; l <= k; l++)

            for (int i = 0; i < r; i++)
                for (int j = 0; j < c; j++) {
                    for (int ti = i + 1; ti < r; ti++)
                        if (check(i, j, ti - 1, c - 1, pizza))
                            f[l][i][j] = (f[l][i][j] + f[l - 1][ti][j]) % mod;
                    
                    for (int tj = j + 1; tj < c; tj++)
                        if (check(i, j, r - 1, tj - 1, pizza))
                            f[l][i][j] = (f[l][i][j] + f[l - 1][i][tj]) % mod;
                }
        
        return f[k][0][0];
    }
};
