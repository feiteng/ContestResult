class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        vector<string> ans;
        int id = 1;
        for (auto x : target) {
            ans.push_back("Push");
            while (id != x) {
                ans.push_back("Pop");
                ans.push_back("Push");
                ++id;
            }
            ++id;
        }
        return ans;
    }
};
