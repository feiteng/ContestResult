class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int cur = 0;
        vector<string> ans;
        for (int i = 0; i < n && i < target.back(); ++i) {
            if (cur < target.size() && target[cur] == i + 1) {
                ans.push_back("Push");
                ++cur;
            } else {
                ans.push_back("Push");
                ans.push_back("Pop");
            }
        }
        return ans;
    }
};
