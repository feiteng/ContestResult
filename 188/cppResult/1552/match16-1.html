class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int c = 1;
        vector<string> ans;
        for (int x : target) {
            while (c < x) {
                c++;
                ans.push_back("Push");
                ans.push_back("Pop");
            }
            ans.push_back("Push");
            c++;
        }
        return ans;
    }
};
