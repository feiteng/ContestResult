class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int i = 0;
        vector<string> res;
        for (int j = 1; j <= n; j++) {
            if (target[i] == j) {
                res.push_back("Push");
                i++;
            } else {
                res.push_back("Push");
                res.push_back("Pop");
                
            }
            if (i == target.size()) break;
        }
        
        return res;
    }
};
