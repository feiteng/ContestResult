class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int i = 0;

        vector<string> result;
        // inc is incoming element
        for (int inc = 1; i < target.size(); inc++) {
            if (inc == target[i]) {
                result.push_back("Push");
                i++;
            } else {
                result.push_back("Push");
                result.push_back("Pop");
            }
        }
        return result;
    }
};
