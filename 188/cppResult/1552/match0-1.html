class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int now = 1;
        vector<string> ret;
        for (auto it : target) {
            while (it != now) {
                ret.push_back("Push");
                ret.push_back("Pop");
                now++;
            }
            ret.push_back("Push");
            now++;
        }
        return ret;
    }
};
