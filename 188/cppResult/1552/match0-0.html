class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int cur = 1;
        vector<string> vv;
        for(auto it: target) {
            vv.push_back("Push");
            while(cur != it) {
                vv.push_back("Pop");
                vv.push_back("Push");
                ++cur;
            }
            ++cur;
        }
        return vv;
    }
};
