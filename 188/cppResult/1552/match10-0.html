class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        vector<string> res;
        int curr = 1;
        for (int num : target){
            while (num > curr){
                res.emplace_back("Push");
                res.emplace_back("Pop");
                curr++;
            }
            res.emplace_back("Push");
            curr++;
        }
        return res;
    }
};
