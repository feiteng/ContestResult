const ll MOD=1e9+7;
class Solution {
public:
  vector<string> buildArray(vector<int>& target, int n) {
    vector<string> res;
    int m=1;
    for(int k:target) {
      while(m<k){
        res.pb("Push");
        res.pb("Pop");
        m++;
      }
      res.pb("Push");
      m++;
    }
    return res;
  }
};
