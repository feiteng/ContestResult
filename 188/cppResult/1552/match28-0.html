class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        vector<string> res;
        int idx = 0;
        for(int num = 1; num <= n; num++){
            if(num == target[idx]){
                idx++;
                res.push_back("Push");
            }
            else{
                res.push_back("Push");
                res.push_back("Pop");
            }
            if(idx == target.size()) break;
        }
        return res;
    }
};
