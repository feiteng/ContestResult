class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        vector<string> ans;
        int now = 1;
        for (int v: target) {
            while (v > now) {
                ans.push_back("Push");
                ans.push_back("Pop");
                ++ now;
            }
            ans.push_back("Push");
            ++ now;
        }
        return ans;
    }
};
