class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int ans = 1;
        vector<string> res;
        for(int i: target) {
            while(i>ans) {
                res.push_back("Push");
                res.push_back("Pop");
                ans++;
            }
            res.push_back("Push");
            ans++;
        }
        return res;
    }
};
