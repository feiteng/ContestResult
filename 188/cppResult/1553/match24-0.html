class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int cnt = 0;
        for(int i = 0 ; i < arr.size(); i++){
            int xr1 = 0;
            for(int j = i; j < arr.size(); j++){
                xr1 ^= arr[j];
                int xr2 = 0;
                for(int k = j + 1; k < arr.size(); k++){
                    xr2 ^= arr[k];
                    if(xr1==xr2)cnt++;
                }
            }
        }
        return cnt;
    }
};
