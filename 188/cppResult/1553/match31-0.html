class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int result = 0;
        for (int j = 1;j < arr.size();++j) {
            map<int, int> left;
            int tmp = 0;
            for (int i = j - 1;i >= 0;--i) {
                tmp = tmp ^ arr[i];
                left[tmp]++;
            }
            tmp = 0;
            for (int k = j;k < arr.size();++k) {
                tmp = tmp ^ arr[k];
                if (left.find(tmp) != left.end()) {
                    result += left[tmp];
                }
            }
        }
        return result;
    }
};
