class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int ans=0;
        int N=arr.size();
        for(int i=0;i<N;i++)
        {
            int a=arr[i];
            for(int j=i+1;j<N;j++)
            {
                
                int b=0;
                for(int k=j;k<N;k++)
                {
                    b=(b^arr[k]);
                    if(a==b)
                        ans++;
                }
                a=(a^arr[j]);
                    
            }
        }
        return ans;
        
    }
};
