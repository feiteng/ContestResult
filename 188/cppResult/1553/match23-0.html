class Solution {
public:
    int countTriplets(vector<int>& s) {
        int n = s.size();
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int a = 0;
            for (int j = i; j < n; j++) {
                a ^= s[j];
                int b = 0;
                for (int k = j + 1; k < n; k++) {
                    b ^= s[k];
                    ans += (a == b);
                }
            }
        }
        return ans;
    }
};
