class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int len=arr.size();
        int dp[len][len];
        for(int i=0;i<len;i++) dp[i][i]=arr[i];
        for(int i=0;i<len-1;i++){
            for(int j=i+1;j<len;j++){
                dp[i][j]=dp[i][j-1]^arr[j];
            }
        }
        

        int ans=0;
        for(int l=2;l<=len;l++){
            for(int i=0;i<len-l+1;i++){
                int j=i+l-1;

                for(int k=i;k<j;k++){
                    if(dp[i][k]==dp[k+1][j]) ans++;
                }
            }
        }
        return ans;
    }
};
