class Solution {
public:
    int countTriplets(vector<int>& a) {
        int n = a.size();
        vector<int> pre(n + 1);
        for (int i = 0; i < n; ++i) pre[i + 1] = pre[i] ^ a[i];
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                for (int k = j + 1; k <= n; ++k) {
                    if ((pre[k] ^ pre[j]) == (pre[j] ^ pre[i])) ++ret;
                }
            }
        }
        return ret;
    }
};
