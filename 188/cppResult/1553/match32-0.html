template<class T>
constexpr T INF = ::numeric_limits<T>::max() / 32 * 15 + 208;

// class
class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int n = arr.size();
        vector<int> S(n+1);
        for (int i = 0; i < n; ++i) {
            S[i+1] = S[i]^arr[i];
        }
        int ans = 0;
        for (int i = 0; i <= n; ++i) {
            for (int j = i+1; j <= n; ++j) {
                for (int k = j+1; k <= n; ++k) {
                    if(S[i] == S[k]) ans++;
                }
            }
        }
        return ans;
    }
};
