class Solution {
public:
    int countTriplets(vector<int>& a) {
        int n = a.size();
        int result = 0;
        for(int i=0; i<n; i++) {
            int left = a[i];
            for(int j=i+1; j<n; j++) {
                int right = a[j];
                if(left==right) result++;
                for(int k=j+1; k<n; k++) {
                    right = (right^a[k]);
                    if(left==right) result++;
                }
                left = (left^a[j]);
            }
        }
        return result;
    }
};
