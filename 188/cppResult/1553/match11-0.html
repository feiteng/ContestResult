class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int ans = 0;
        
        for(int i=0;i<arr.size();i++){
            int sum = arr[i];
            for(int k=i+1; k<arr.size();k++){
                sum ^= arr[k];
                if(sum == 0 && k - i > 0){
                    ans += k - i;
                }
            }
            
        }
        
        return ans;
    }
};
