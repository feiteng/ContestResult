class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int cnt=0;
        for (int i=0; i<arr.size(); i++) {
            int sum=0;
            for (int j=i+1; j<arr.size(); j++) {
                sum^=arr[j-1];
                int sum1=0;
                for (int k=j; k<arr.size(); k++) {
                    sum1^=arr[k];
                    if (sum==sum1) cnt++;
                }
            }
        }
        return cnt;
    }
};
