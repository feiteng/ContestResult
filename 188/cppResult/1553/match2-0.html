class Solution {
public:
    
    int countTriplets(vector<int>& arr) {
        int ans = 0;
        int n = arr.size();
        for(int i = 1;i <= n;++i){
            int now = 0;
            for(int j = i;j >= 1;--j){
                now^=arr[j-1];
                int now2 = 0;
                for(int k = j-1;k >= 1;--k){
                    now2^=arr[k-1];
                    if(now2==now)ans++;
                }
            }
        }
        return ans;
    }
};
