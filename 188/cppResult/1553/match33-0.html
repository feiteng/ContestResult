class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int n = arr.size();
        vector<int> s(n + 1, 0);
        for (int i = 0; i < n; i++) s[i + 1] = s[i] ^ arr[i];
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j; k < n; k++) {
                    if (s[j] ^ s[i] == s[k + 1] ^ s[j]) cnt++;
                }
            }
        }
        return cnt;
    }
};
