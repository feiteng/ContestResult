class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int n = arr.size();
        int ans = 0;
        for (int i =0 ; i < n; i++) {
            int cur = 0;
            for (int j = i; j < n; j++) {
                cur ^= arr[j];
                
                if (cur == 0) {
                    ans += j-i;
                }
                // if i to j is 0, sum += 
            }
        }
        return ans;
    }
};
