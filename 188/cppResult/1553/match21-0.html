class Solution {
public:
    int psa[301];
    int countTriplets(vector<int>& arr) {
        psa[0] = 0;
        for (int i = 1; i <= arr.size(); i++) psa[i] = psa[i-1]^arr[i-1];
        int cnt = 0;
        for (int i = 1; i <= arr.size(); i++) {
            for (int j = i + 1; j <= arr.size(); j++) {
                for (int k = j; k <= arr.size(); k++) {
                    if (psa[j-1]^psa[i-1]==psa[k]^psa[j-1]) cnt++;
                }
            }
        }
        return cnt;
    }
};
