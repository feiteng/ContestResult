class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int ans=0;
        for(int i=0;i<arr.size();i++){
            int x=arr[i];
            for(int j=i+1;j<arr.size();j++){
                int y=0;
                for(int k=j;k<arr.size();k++){
                    y^=arr[k];
                    if(x==y)ans++;
                }
                
                x^=arr[j];
            }
        }
        return ans;
    }
};
