class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int n = arr.size();
        vector<int> dp(n + 1,0);
        for(int i = 0; i < n; i++){
            dp[i+1] = dp[i]^arr[i];
        }
        int ans = 0;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                for(int k = j; k < n; k++){
                    if((dp[i]^dp[j]) == (dp[j]^dp[k+1])){
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
