class BrowserHistory:

    def __init__(self, homepage: str):
        self.stack=[homepage]
        self.currind=0

    def visit(self, url: str) -> None:
        self.stack=self.stack[:self.currind+1]+[url]
        self.currind+=1

    def back(self, steps: int) -> str:
        self.currind=max(0, self.currind-steps)
        return self.stack[self.currind]

    def forward(self, steps: int) -> str:
        self.currind=min(len(self.stack)-1, self.currind+steps)
        return self.stack[self.currind]
