class BrowserHistory:

    def __init__(self, homepage: str):
        self.st = [homepage]
        self.i = 0
        

    def visit(self, url: str) -> None:
        while len(self.st) > self.i + 1:
            self.st.pop()
        self.st.append(url)
        self.i += 1
        

    def back(self, steps: int) -> str:
        self.i -= steps
        if self.i < 0: self.i = 0
        return self.st[self.i]
        

    def forward(self, steps: int) -> str:
        self.i += steps
        if self.i >= len(self.st): self.i = len(self.st) - 1
        return self.st[self.i]
