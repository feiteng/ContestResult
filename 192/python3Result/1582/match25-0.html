class BrowserHistory:

    def __init__(self, homepage: str):
        self.rcd=[homepage]
        self.flag=0
    def visit(self, url: str) -> None:
        self.rcd=self.rcd[:self.flag+1]
        self.rcd.append(url)
        self.flag+=1
    def back(self, steps: int) -> str:
        self.flag=max(self.flag-steps,0)
        return self.rcd[self.flag]

    def forward(self, steps: int) -> str:
        self.flag=min(self.flag+steps,len(self.rcd)-1)
        return self.rcd[self.flag]
