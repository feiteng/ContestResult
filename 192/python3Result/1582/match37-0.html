class BrowserHistory:

    def __init__(self, homepage: str):
        
        self._arr = [homepage]
        self._pos = 1

    def visit(self, url: str) -> None:

        self._arr = self._arr[:self._pos] + [url]
        self._pos = len(self._arr)

    def back(self, steps: int) -> str:
        
        self._pos = max(self._pos-steps, 1)
        return self._arr[self._pos-1]

    def forward(self, steps: int) -> str:
        
        self._pos = min(len(self._arr), self._pos+steps)
        return self._arr[self._pos-1]
