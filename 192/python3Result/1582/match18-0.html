class BrowserHistory:

    def __init__(self, homepage: str):
        self.his = [homepage]
        self.point = 0
        
    def visit(self, url: str) -> None:
        self.his = self.his[:self.point + 1]
        self.his.append(url)
        self.point += 1
        

    def back(self, steps: int) -> str:
        self.point = max(0, self.point - steps)
        return self.his[self.point]
            
    def forward(self, steps: int) -> str:
        self.point = min(len(self.his) - 1, self.point + steps)
        return self.his[self.point]
