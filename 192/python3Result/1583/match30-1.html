class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        MAXN = 10 ** 9
        f = [[[MAXN] * n for _ in range(target + 1)] for _ in range(m)]
        for i in range(m):
            houses[i] -= 1
            if houses[i] >= 0:
                cost[i][houses[i]] = 0
        
if houses[0] == -1:
            for k in range(n):
                f[0][1][k] = cost[0][k]
        else:
            f[0][1][houses[0]] = 0
        for i in range(1, m):
            if houses[i] < 0:
                em = range(n)
            else:
                em = [houses[i]]
            for j in range(target + 1):
                for k in em:
                    
for l in range(n):
                        if k == l:
                            f[i][j][k] = min(f[i - 1][j][l] + cost[i][k], f[i][j][k])
                        elif j > 0:
                            f[i][j][k] = min(f[i - 1][j - 1][l] + cost[i][k], f[i][j][k])
        ret = min(f[-1][target])
        if ret < MAXN:
            return ret
        return -1
