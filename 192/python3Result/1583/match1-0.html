class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        # dp[i][c][t] is the min cost for houses for previous i houses ended with color c and the total target is t.
        dp = [
            [
                [float('inf')] * (m + 1)
                for _ in range(n)
            ]
            for _ in range(m + 1)
        ]
        c = houses[0] - 1
        if c == -1:
            for c in range(n):
                dp[1][c][1] = cost[0][c]
        else:
            dp[1][c][1] = 0
        
        for i in range(1, m):
            c = houses[i] - 1
            if c != -1:
                for prev_c in range(n):
                    for prev_t in range(m):
                        if c == prev_c:
                            dp[i + 1][c][prev_t] = min(dp[i][c][prev_t], dp[i + 1][c][prev_t])
                        else:
                            dp[i + 1][c][prev_t + 1] = min(dp[i][prev_c][prev_t], dp[i + 1][c][prev_t + 1])
            else:
                for c in range(n):
                    for prev_c in range(n):
                        for prev_t in range(m):
                            if c == prev_c:
                                dp[i + 1][c][prev_t] = min(dp[i][c][prev_t] + cost[i][c], dp[i + 1][c][prev_t])
                            else:
                                dp[i + 1][c][prev_t + 1] = min(dp[i][prev_c][prev_t] + cost[i][c], dp[i + 1][c][prev_t + 1])

        ans = min([dp[m][c][target] for c in range(n)])
        if ans == float('inf'):
            return -1
        return ans
