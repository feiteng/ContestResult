class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        dp = [[[10 ** 4 * m + 100 for k in range(n + 1)] for j in range(target + 1)] for i in range(m)]
        for i in range(m):
            for k in range(1, target + 1):
                if i + 1 < k:
                    continue
                for j in range(1, n + 1):
                    if houses[i] == 0:
                        if i == 0:
                            for nj in range(1, n + 1):
                                dp[i][k][nj] = cost[i][nj-1]
                        else:
                            
for nj in range(1, n + 1):
                                if j == nj:
                                    dp[i][k][nj] = min(dp[i][k][nj], dp[i-1][k][j] + cost[i][nj-1])
                                else:
                                    dp[i][k][nj] = min(dp[i][k][nj], dp[i-1][k-1][j] + cost[i][nj-1])
                    else:
                        nj = houses[i]
                        if i == 0:
                            dp[i][k][nj] = 0
                        else:
                            
if j == nj:
                                dp[i][k][nj] = min(dp[i][k][nj], dp[i-1][k][j])
                            else:
                                dp[i][k][nj] = min(dp[i][k][nj], dp[i-1][k-1][j])
        res = 100 ** 4 * m + 100
        for j in range(1, n + 1):
            res = min(res, dp[m-1][target][j])
        if res > 10 ** 4 * m:
            return -1
        else:
            return res
