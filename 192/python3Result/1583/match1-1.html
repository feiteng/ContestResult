class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        paint=[[[float('inf')]*target for _ in range(n)] for _ in range(m)]
        if houses[0]==0:
            
for j in range(n):
                paint[0][j][0]=cost[0][j]
        else:
            paint[0][houses[0]-1][0]=0
            
        for i in range(1, len(houses)):
            if houses[i]!=0:
                for j in range(n):
                    for k in range(target):
                        if houses[i]-1==j:
                            paint[i][houses[i]-1][k]=min(paint[i][houses
[i]-1][k], paint[i-1][j][k])
                        elif k<target-1:
                            paint[i][houses[i]-1][k+1]=min(paint[i][houses[i]-1][k+1], paint[i-1][j][k])
                continue
                
            
for j in range(n):
                for k in range(target):
                    for c in range(n):
                        if c==j:
                            paint[i][c][k]=min(paint[i][c][k], paint[i-1][j][k]+cost[i][c])
                        elif k<target-1:
                            paint[i][c][k+1]=min(paint[i][c][k+1], paint[i-1][j][k]+cost[i][c])
        ans=float('inf')
        for i in range(n):
            ans=min(ans, paint[-1][i][-1])
        if ans==float('inf'):
            return -1
        return ans
