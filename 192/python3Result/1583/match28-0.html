class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        @lru_cache(None)
        def solve(idx, prvClr, curNeis):
            if curNeis > target:
                return float('inf')
            if idx == m:
                return 0 if curNeis == target else float('inf')
            
            if houses[idx]:
                return solve(idx+1, houses[idx], curNeis + int(houses[idx] != prvClr))
            
            ans = float('inf')
            for i in range(1, n+1):
                if i == prvClr:
                    ans = min(ans, cost[idx][i-1] + solve(idx+1, prvClr, curNeis))
                else:
                    ans = min(ans, cost[idx][i-1] + solve(idx+1, i, curNeis+1))

            return ans

        ans = solve(0, -1, 0)
        return ans if ans < float('inf') else -1
