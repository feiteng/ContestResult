class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        dp = [[[float('inf')] * n for _ in range(m)] for _ in range(target)]
        if houses[0] > 0:
            dp[0][0][houses[0]-1] = 0
        else:
            for k in range(n):
                dp[0][0][k] = cost[0][k]
        for j in range(1, m):
            if houses[j] > 0:
                c = houses[j] - 1
                for i in range(target):
                    if i == 0:
                        dp[i][j][c] = dp[i][j-1][c]
                    else:
                        dp[i][j][c] = min(dp[i][j-1][c], min(dp[i-1][j-1][k] for k in range(n) if k!= c)) 
            else:
                for k in range(n):
                    for i in range(target):
                        if i == 0:
                            dp[i][j][k] = dp[i][j-1][k] + cost[j][k]
                        else:
                            dp[i][j][k] = min(dp[i][j-1][k], min(dp[i-1][j-1][k1] for k1 in range(n) if k1!= k)) + cost[j][k]
        ans = min(dp[target-1][m-1])
        return ans if ans < float('inf') else -1
