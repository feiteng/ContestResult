class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        max_v = 10 ** 9
        f = [[[max_v for i in range(target + 1)] for j in range(m)] for k in range(n + 1)]
        if houses[0] != 0:
            f[houses[0]][0][1] = 0
        else:
            for i in range(1, n + 1):
                f[i][0][1] = cost[0][i - 1]
        for i in range(1, m):
            for now_color in range(1, n + 1):
                if houses[i] != 0:
                    cost_v = 0 if now_color == houses[i] else max_v
                else:
                    cost_v = cost[i][now_color - 1]
                for color in range(1, n + 1):
                    for cnt in range(1, target + 1):
                        if color == now_color:
                            f[now_color][i][cnt] = min(f[now_color][i][cnt], f[color][i - 1][cnt] + cost_v)
                        elif cnt < target:
                            f[now_color][i][cnt + 1] = min(f[now_color][i][cnt + 1], f[color][i - 1][cnt] + cost_v)
        ans = max_v
        for i in range(1, n + 1):
            ans = min(ans, f[i][m - 1][target])
        if ans == max_v:
            return -1
        return ans
