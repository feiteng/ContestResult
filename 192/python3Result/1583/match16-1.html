class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, t: int) -> int:
        max_ = 99999999
        dp = [[[max_ for _ in range(t)]  for _ in range(n)] for _ in range(m)]
        if houses[0] == 0:
            for j in range(n):
                dp[0][j][0] = cost[0][j]
        else:
            dp[0][houses[0]-1][0] = 0
        for i in range(1, m):
            for j in range(n):
                for k in range(t):
                    for j2 in range(n):
                        if houses[i] == 0 or houses[i]-1 == j2:
                            k2 = k + 1 if j2!=j else k
                            if k2<t:
                                dp[i][j2][k2] = min( (cost[i][j2] if houses[i]-1 != j2 else 0)  + dp
[i-1][j][k], dp[i][j2][k2])
        ret = max_
        for j in range(n):
            ret = min(ret, dp[m-1][j][t-1])
        return -1 if ret == max_ else ret
