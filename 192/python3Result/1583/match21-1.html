class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        dp = [[[float('inf')] * (target+1) for _ in range(n)] for _ in range(m)]
        
        if houses[0] != 0:
            c = houses
[0]-1
            dp[0][c][1] = 0
        else:
            for c in range(n):
                dp[0][c][1] = cost[0][c]
        
        for i in range(1, m):
            if houses[i] != 0:
                c = houses[i] - 1
                for c1 in range(n):
                    for t1 in range(target+1):
                        t = t1 + (c1 != c)
                        
if t <= target:
                            dp[i][c][t] = min(dp[i][c][t], dp[i-1][c1][t1])
            else:
                for c in range(n):
                    for c1 in range(n):
                        for t1 in range(target+1):
                            t = t1 + (c1 != c)
                            
if t <= target:
                                dp[i][c][t] = min(dp[i][c][t], dp[i-1][c1][t1] + cost[i][c])
        
        res = float('inf')
        for c in range(n):
            res = min(res, dp[-1][c][target])
        
        if res == float('inf'):
            res = -1
            
        return res
