class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        dp=[[[float('inf') for k in range(target+1)] for j in range(n)] for i in range(m)]
        if houses[0]==0:
            for i in range(n):
                dp[0][i][0]=cost[0][i]
        else:
            dp[0][houses[0]-1][0]=0
        for i in range(1,m):
            if houses[i]==0:
                for j in range(n):
                    for k in range(target):
                        for t in range(n):
                            if t==j:
                                dp[i][j][k]=min(dp[i][j][k],dp[i-1][t][k]+cost[i][j])
                            elif k!=0:
                                dp[i][j][k]=min(dp[i][j][k],dp[i-1][t][k-1]+cost[i][j])
            else:
                for k in range(target):
                    #print(dp[1][1][1])
                    for t in range(n):
                        if t==houses[i]-1:
                            dp[i][houses[i]-1][k]=min(dp[i][houses[i]-1][k],dp[i-1][t][k])
                        else:
                            dp[i][houses[i]-1][k+1]=min(dp[i][houses[i]-1][k+1],dp[i-1][t][k])
                            #print(i,houses[i]-1,k+1,dp[i][houses[i]-1][k+1])
                            #print(dp[1][1][1])
        res=float('inf')
        #print(dp[1][1][1])
        # for i in range(m):
        #     for j in range(n):
        #         print(i,j,dp[i][j])
        for i in range(n):
            res=min(dp[-1][i][target-1],res)
        if res==float('inf'):
            return -1
        else:
            return res
