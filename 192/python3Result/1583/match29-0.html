class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        f = [[[sys.maxsize for k in range(n + 1)] for j in range(1 + target)] for i in range(1 + m)]
        f[0][0][0] = 0
        
        for i in range(1, m + 1):
            for j in range(1, target + 1):
                for k in range(1, n + 1):
                    if houses[i - 1] and k != houses[i - 1]:
                        continue
                    for k_p in range(n + 1):
                        if k_p == k and f[i - 1][j][k] < sys.maxsize:
                            if k == houses[i - 1]:
                                f[i][j][k] = min(f[i][j][k], f[i - 1][j][k])
                            else:
                                f[i][j][k] = min(f[i][j][k], f[i - 1][j][k] + cost[i - 1][k - 1])
                        elif k_p != k and f[i - 1][j - 1][k_p] < sys.maxsize:
                            if k == houses[i - 1]:
                                f[i][j][k] = min(f[i][j][k], f[i - 1][j - 1][k_p])
                            else:
                                f[i][j][k] = min(f[i][j][k], f[i - 1][j - 1][k_p] + cost[i - 1][k - 1])
        
        min_cost = min(f[m][target])
        if min_cost == sys.maxsize:
            return -1
        return min_cost
