class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], n: int, C: int, target: int) -> int:
        f = [[[-1]*(C+1) for __ in range(n+5)] for _ in range(n+1)]
        if houses[0] != 0:
            f[0][1][houses[0]-1] = 0
        else:
            for c in range(C):
                f[0][1][c] = cost[0][c]
        # print (f[0][1])
        for i in range(n-1):
            for j in range(target+1):
                for c in range(C):
                    if f[i][j][c] == -1:
                        continue
                    # print ("develop", i, j, c)
                    if houses[i+1] != 0:
                        # print ("already colored")
                        newJ = j
                        newC = houses[i+1] - 1

                        if newC != c:
                            newJ += 1
                        if f[i+1]
[newJ][newC] == -1 or f[i+1][newJ][newC] > f[i][j][c]:
                            f[i+1][newJ][newC] = f[i][j][c]
                    else:
                        for newC in range(C):
                            newJ = j
                            if newC != c:
                                newJ += 1
                            newCost = f[i][j]
[c] + cost[i+1][newC]
                            if f[i+1][newJ][newC] == -1 or f[i+1][newJ][newC] > newCost:
                                f[i+1][newJ][newC] = newCost
        n -= 1
        # print (f[n])
        # print (f[n][target])

        ans = -1
        for c in range(C):
            if f[n][target][c] != -1:
                if ans == -1 or ans > f
[n][target][c]:
                    ans = f[n][target][c]
        return ans
