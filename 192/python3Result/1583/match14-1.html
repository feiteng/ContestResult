from typing import List



class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        inf = 99999999
        f = [[[inf] * (m + 1) for _ in range(n)] for _ in range(m)]
        if houses[0] > 0:
            f[0][houses[0] - 1][1] = 0
        else:
            for i in range(n):
                f[0][i][1] = cost[0][i]
        for i in range(m - 1):
            for j in range(n):
                for k in range(m + 1):
                    if f[i][j][k] != inf:
                        if houses[i + 1] > 0:
                            if j == houses[i + 1] - 1:
                                f[i + 1][j][k] = min(f[i + 1][j][k], f[i][j][k])
                            else:
                                f[i + 1][houses[i+1]-1][k + 1] = min(f[i + 1][houses[i+1]-1][k + 1], f[i][j][k])
                        else:
                            for cl in range
(n):
                                if cl == j:
                                    f[i + 1][cl][k] = min(f[i + 1][cl][k], f[i][j][k] + cost[i + 1][cl])
                                else:
                                    f[i + 1][cl][k + 1] = min(f[i + 1][cl][k + 1], f[i][j][k] + cost[i + 1][cl]
)
        ans = inf
        for i in range(n): ans = min(ans, f[m - 1][i][target])
        if ans == inf: ans = -1
        return ans
