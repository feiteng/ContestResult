        @lru_cache(None)
        def helper(i = 0, last = -1, t = target):
            #print('just in', i, last, t)
            
if i == m:
                return math.inf if t != 0 else 0
            #if t == 0:
            #    return math.inf if i != m else 0
            if t < 0:

                return helper(i +1, houses[i], t - (houses[i] != last))
            
            ans = math.inf
            for j in range(1, n+1):
                ans = min(ans, cost[i][j-1] + helper(i + 1, j, t - (j != last)))
            #print(i, last, t, ans)

            return -1
