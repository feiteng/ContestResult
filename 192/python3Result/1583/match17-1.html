class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        m = len(houses)
        dp = [[[float('inf')] * (n + 1) for _ in range
(target + 1)] for _ in range(m + 1)]
        for k in range(n + 1):
            dp[0][0][k] = 0
        for i, h in enumerate(houses, 1):
            for j in range(1, target + 1):
                
for k in range(1, n + 1):
                    if h == 0:
                        # h -> k
                        dp[i][j][k] = min(dp[i - 1][j - 1][l] + cost[i - 1][k - 1] for l in range
(1, n + 1) if l != k)
                        dp[i][j][k] = min(dp[i][j][k], dp[i - 1][j][k] + cost[i - 1][k - 1])
                    elif k == h:
                        dp[i][j][k] = min(dp[i - 1][j][k], dp[i][j][k])
                    else:
                        dp
[i][j][h] = min(dp[i - 1][j - 1][k], dp[i][j][h])
                
        res = min(dp[-1][-1])
        return -1 if res == float('inf') else res
