class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        if len(set(houses)) > target:
            return -1
        
        dp = [[[float('inf')]*(target+1) for i in range(n)] for j in range(m)]
        
        #dp[i][j][k]: house_i, color_j, group_k
        
        # initialize
        color = houses[0]
        if color == 0:
            for j in range(n):
                dp[0][j][1] = cost[0][j]
        else:
            color -= 1
            dp[0][color][1] = 0
        
        #print(dp[0])
        for i in range(1, m):
            color = houses[i]
            if color != 0:
                color -= 1
                for j in range(n):
                    for k in range(1, target+1):
                        if color == j:
                            dp[i][color][k] = min(dp[i][color][k], dp[i-1][j][k])
                        else:
                            dp[i][color][k] = min(dp[i][color][k], dp[i-1][j][k-1])
                            
            else:
                for color in range(n):
                    for j in range(n):
                        for k in range(1, target+1):
                            #print(color, j, k)
                            if color == j:
                                dp[i][color][k] = min(dp[i][color][k], dp[i-1][j][k] + cost[i][color])
                            else:
                                dp[i][color][k] = min(dp[i][color][k], dp[i-1][j][k-1] + cost[i][color])
            #print(i, dp[i])
            
        #print(dp)
        res = []
        for j in range(n):
            res.append(dp[m-1][j][target])
        if min(res) > 10000*m:
            return -1
        return min(res)
