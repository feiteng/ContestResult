class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        @functools.lru_cache(None)
        def dp(pre, i, target):
            # print(pre,i,target)
            if target < 0:
                return -1
            if i == m and target != 0:
                return -1
            elif i == m and target == 0:
                return 0
            if houses[i] == 0:
                tc = float('inf')
                
for c in range(1, n+1):
                    if c == pre:
                        val = dp(pre, i+1, target)
                    else:
                        val = dp(c, i+1, target-1)
                    if val != -1:
                        tc = min(tc, val + cost[i][c-1])
                if tc == float('inf'):
                    return -1
                else:
                    return tc
            elif houses
[i] == pre:
                return dp(pre, i+1, target)
            else:
                return dp(houses[i], i+1, target-1)
                
        return dp(-1,0,target)
