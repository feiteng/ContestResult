class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        INF = 1e9
        f = [[[INF] * (target + 1) for _ in range(n + 1)] for _ in range(m + 1)]
        if houses[0] != 0:
            f[1][houses[0] - 1][1] = 0
        else:
            for i in range(n):
                f[1][i][1] = cost[0][i]
                
        for i in range(1, m):
            for j in range(n):
                for k in range(1, target + 1):
                    if f[i][j][k] >= INF:
                        continue
                    if houses[i] != 0:
                        j_ = houses[i] - 1
                        k_ = k + int(j_ != j)
                        if k_ <= target:
                            f[i + 1][j_][k_] = min(f[i + 1][j_][k_], f[i][j][k])
                    else:
                        for j_ in range(n):
                            k_ = k + int(j_ != j)
                            if k_ <= target:
                                f[i + 1][j_][k_] = min(f[i + 1][j_][k_], f[i][j][k] + cost[i][j_])
        ans = INF
        for j in range(n):
            ans = min(ans, f[m][j][target])
        if ans >= INF:
            ans = -1
        return ans
