class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        dp = [[[float("inf") for i in range(target+1)] for j in range(n)] for x in range(m)]
        if houses[0] != 0:
            dp[0][houses[0]-1][1] = 0
        else:
            for j in range(n):
                dp[0][j][1] = cost[0][j]
        for i in range(1, m):
            if houses[i] != 0:
                j = houses[i]-1
                for x in range(1, target + 1):
                    for y in range(n):
                        if y != j:
                            dp[i][j][x] = min(dp[i][j][x], dp[i-1][y][x-1])
                    dp[i][j][x] = min(dp[i][j][x], dp[i-1][j][x])
                continue
            for j in range(n):
                for x in range(1, target + 1):
                    for y in range(n):
                        if y != j:
                            dp[i][j][x] = min(dp[i][j][x], dp[i-1][y][x-1] + cost[i][j])
                    dp[i][j][x] = min(dp[i][j][x], dp[i-1][j][x] + cost[i][j])
        res = float("inf")
        for j in range(n):
            res = min(res, dp[m-1][j][target])
        return res if res < float("inf") else -1
