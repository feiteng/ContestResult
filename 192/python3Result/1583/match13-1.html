class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        @lru_cache(None)
        def dp(pre, i, tar):
            if i==m:
                if not tar:
                    return 0
                else:
                    return float('inf')
            if houses[i]:
                if houses[i]==pre+1:
                    return dp(pre, i+1, tar)
                else:
                    
return dp(houses[i]-1, i+1, tar-1)
            res = float('inf')
            for j in range(n):
                if j == pre:
                    res = min(res, dp(j, i+1, tar) + cost[i][j])
                else:
                    res = min(res, dp(j, i+1, tar-1) + cost[i][j])
            return res
        res = dp(-1, 0, target)
        return res if res != float('inf') else -1
