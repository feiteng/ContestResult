class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        # f: (i, color, n_nei) -> min_cost
        f = [[[-1] * (m + 1) for _ in range(n + 1)] for _ in range(m)]
        if houses[0] != 0:
            f[0][houses[0]][1] = 0
        else:
            for c in range(1, n + 1):
                f[0][c][1] = cost[0][c - 1]
        for i in range(1, m):
            if houses[i] != 0:
                for c in range(1, n + 1):
                    add = 1 if c != houses[i] else 0
                    for n_nei in range(1, min(i, target) + 1):
                        if f[i - 1][c][n_nei] >= 0:
                            if f[i][houses[i]][n_nei + add] >= 0:
                                f[i][houses[i]][n_nei + add] = min(f[i][houses[i]][n_nei + add], f[i - 1][c][n_nei])
                            else:
                                f[i][houses[i]][n_nei + add] = f[i - 1][c][n_nei]
            else:
                for c in range(1, n + 1):
                    for last_c in range(1, n + 1):
                        add = 1 if c != last_c else 0
                        for n_nei in range(1, min(i, target) + 1):
                            if f[i - 1][last_c][n_nei] >= 0:
                                if f[i][c][n_nei + add] >= 0:
                                    f[i][c][n_nei + add] = min(f[i][c][n_nei + add], f[i - 1][last_c][n_nei] + cost[i][c - 1])
                                else:
                                    f[i][c][n_nei + add] = f[i - 1][last_c][n_nei] + cost[i][c - 1]
            
        ans = -1
        for c in range(1, n + 1):
            if f[m - 1][c][target] >= 0:
                if ans >= 0:
                    ans = min(ans, f[m - 1][c][target])
                else:
                    ans = f[m - 1][c][target]
        return ans
