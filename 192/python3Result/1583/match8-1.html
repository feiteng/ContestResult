class Solution:
    def minCost(self, h: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        MAX_C = 1e10
        f = [[[MAX_C for k in range(n+1)] for j in range(target+1)] for i in range(m)]
        if h[0] != 0:
            f[0][1][h[0]] = 0
        else:
            for p in range(1, n+1):
                f[0][1][p] = cost[0][p-1]
        for i in range(1, m):
            for j in range(1, target+1):
                for k in range(1, n+1):
                    if h[i] == 0 or h[i] == k:
                        cst = cost[i][k-1]
                        if h[i] == k:
                            cst = 0
                        
for kk in range(1, n+1):
                            if k == kk:
                                f[i][j][k] = min(f[i][j][k], f[i-1][j][kk] + cst)
                            else:
                                f[i][j][k] = min(f[i][j][k], f[i-1][j-1][kk] + cst)
                    # print(i, j, k, f[i][j][k])
        # print(f)
        ans = min(f[m-1][target])
        if ans == MAX_C:
            return -1
        return ans
