from functools import lru_cache

class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        
        # i = house
        # c = previous color
        # t = number of neighbourhoods needed.
        @lru_cache(maxsize=None)
        def solve(i, c, t):

            # Invalid cases...
            if t < 0:
                return math.inf
            if i == len(houses):
                if t == 0:
                    return 0
                else:
                    return math.i
nf

            # If the house is already painted...
            if houses[i] != 0:
                if houses[i] == c:
                    return solve(i + 1, c, t)
                else:
                    return solve(i + 1, houses[i], t - 1)

            # Otherwise we need to paint the house.
            min_cost = math.inf
            
for j in range(1, n + 1):
                this_cost = cost[i][j - 1]
                if j == c:
                    this_cost += solve(i + 1, c, t)
                else:
                    this_cost += solve(i + 1, j, t - 1)
                min_cost = min(min_cost, this_cost)

            return min_cost

        answer = solve(0, 0, target)
        if answer is math.inf:
            return -1
        else:
            return answer
