class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        @functools.lru_cache(None)
        def dp(idx, prevColor, curHoods):
            if curHoods > target:
                return math.inf
            if idx == m:
                return math.inf if curHoods != target else 0
            if houses[idx] > 0:
                return dp(idx + 1, houses[idx], curHoods + (1 if houses[idx] != prevColor else 0))
            best = math.inf
            for i in range(1, n + 1):
                best = min(best, cost[idx][i - 1] + dp(idx + 1, i, curHoods + (1 if i != prevColor else 0)))
            return best
        ans = dp(0, None, 0)
        if ans == math.inf:
            return -1
        return ans
