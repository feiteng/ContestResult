from functools import lru_cache
class Solution:
    def minCost(self, h: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        INF = float("inf")
        @lru_cache(None)
        def solve(pre, i, tar):
            if tar < 0:
                return -1
            if i == m:
                return -1 if tar != 0 else 0
            if h[i] == 0:  # can color
                tmp = INF  # cost to color h[i]
                for c in range(1, n+1):  # try n colors
                    if c == pre:  # if current color is prev color
                        val = solve(pre, i+1, tar)
                    else:
                        val = solve(c, i+1, tar-1)
                    if val != -1:
                        tmp = min(tmp, val+cost[i][c-1])
                return -1 if tmp == INF else tmp

            elif h[i] == pre:  # same color as before
                return solve(pre, i+1, tar)
            else:  # different color
                return solve(h[i], i+1, tar-1)
            
        return solve(-1, 0, target)
