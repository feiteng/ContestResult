from functools import lru_cache

class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        @lru_cache(25000)
        def f(i, k, last):
            if k < 0:
                return float('inf')
            
            if i == len(houses):
                if k == 0:
                    return 0
                return float('inf')
            
            
            if houses[i] != 0:
                if last == houses[i]-1:
                    return f(i+1, k, houses[i]-1)
                else:
                    return f(i+1, k-1, houses[i]-1)
                
                
            res = float('inf')
            for x in range(n):
                if x == last:
                    res = min(res, f(i+1, k, x) + cost[i][x])
                else:
                    res = min(res, f(i+1, k-1, x) + cost[i][x])
            
            return res
        
        v = f(0, target, -1)
        return v if v != float('inf') else -1
