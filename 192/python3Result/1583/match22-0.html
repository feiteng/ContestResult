from functools import lru_cache
class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        
        @lru_cache(None)
        def dp(i, nb, pre_color):
            
            # print (i, nb, pre_color)
            
            nonlocal m
            nonlocal n            
            nonlocal target
            
            if i >= m:
                if nb != target:
                    return float('inf')
                return 0
            
            if nb > target:
                return float('inf')
            
            if houses[i] > 0:
                if houses[i] == pre_color:
                    return dp(i + 1, nb, pre_color)
                else:
                    return dp(i + 1, nb + 1, houses[i])
                
            minv = float('inf')
            for j in range(1, n + 1):
                
                # print (i, j)
                c = cost[i][j - 1]
                
                if j == pre_color:
                    temp = dp(i + 1, nb, pre_color)
                else:
                    temp = dp(i + 1, nb + 1, j)
                    
                minv = min(minv, c + temp)
                
            # print (i, minv)
                
            return minv
        
        ans = dp(0, 0, -1)
        
        if ans == float('inf'):
            return -1
        
        return ans
