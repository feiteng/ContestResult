class Solution:
    def minCost(self, houses: List[int], cost: List[List[int]], m: int, n: int, target: int) -> int:
        INF = 1000000000
        dp = [[[INF for k in range(target + 1)] for j in range(n)] for i in range(m)]

        if houses[0] == 0:
            for j in range(n):
                dp[0][j][1] = cost[0][j]
        else:
            dp[0][houses[0] - 1][1] = 0

        for i in range(1, m):
            for j in range(n):
                c = cost[i][j] if houses[i] == 0 else (0 if houses[i] - 1 == j else INF)
                for k in range(1, target + 1):
                    for prev in range(n):
                        if prev == j:
                            dp[i][j][k] = min(dp[i][j][k], dp[i - 1][prev][k] + c)
                        else:
                            if k > 0:
                                dp[i][j][k] = min(dp[i][j][k], dp[i - 1][prev][k - 1] + c)
        
        ans = min(dp[m - 1][j][target] for j in range(n))
        return ans if ans < INF else -1
