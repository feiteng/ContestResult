class Solution {
    private int min(int x, int y) {
        if (x == -1) {
            return y;
        } else if (y == -1) {
            return x;
        } else {
            return Math.min(x, y);
        }
    }
    
    
public int minCost(int[] h, int[][] c, int hn, int cn, int t) {
        int[][][] dp = new int[hn][hn + 1][cn + 1];
        for (int i = 0; i < hn; ++i) {
            for (int j = 0; j <= hn; ++j) {
                Arrays.fill(dp[i][j], -1);
            }
        }
        
        if (h[0] == 0) {
            for (int k = 1; k <= cn; ++k) {
                dp[0][1][k] = c[0][k - 1];
            }
        } else {
            dp[0][1][h[0]] = 0;
        }
        
        
for (int i = 0; i < hn - 1; ++i) {
            for (int j = 1; j <= hn; ++j) {
                for (int k = 1; k <= cn; ++k) {
                    if (dp[i][j][k] == -1) {
                        continue;
                    }
                    
                    if (h[i + 1] != 0) 
{
                        if (h[i + 1] == k) {
                            dp[i + 1][j][k] = min(dp[i + 1][j][k], dp[i][j][k]);
                        } else {
                            dp[i + 1][j + 1][h[i + 1]] = min(dp[i + 1][j + 1][h[i + 1]], dp[i][j][k]);
                        }
                    } else 
{
                        for (int tk = 1; tk <= cn; ++tk) {
                            if (tk == k) {
                                dp[i + 1][j][tk] = min(dp[i + 1][j][k], dp[i][j][k] + c[i + 1][tk - 1]);
                            } else {
                                dp[i + 1][j + 1][tk] = min(dp[i + 1][j + 1][tk], dp[i][j][k] + c[i + 1][tk - 1]);
                            }
                        }
                    }
                
}
            }
        }

        
        int ans = -1;
        for (int k = 1; k <= cn; ++k) {
            ans = min(ans, dp[hn - 1][t][k]);
        }
        return ans;
    }
}
