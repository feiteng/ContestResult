class Solution {
    static final int MAX = 100000;
    
    public int min(int... args){
        int ans = MAX;
        for(int x : args){
            ans = Math.min(ans,x);
        }
        return ans;
    }
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int dp[][][] = new int[m][n+1][target+1];
        //Initializations
        for(int i=0;i<dp.length;i++){
            for(int j=0;j<dp[i].length;j++){
                for(int k=0;k<dp[i][j].length;k++){
                    dp[i][j][k] = MAX;
                }
            }
        }
        //Base Case
        if(houses[0]==0){
            for(int color=1;color<=n;color++){
                dp[0][color][1] = cost[0][color-1];
            }
        }
        else{
            dp[0][houses[0]][1] = 0;
        }
        
        for(int i=1;i<m;i++){
            //It's already painted            
            if(houses[i]!=0){
                for(int k=1;k<=target;k++){
                    dp[i][houses[i]][k] = dp[i-1][houses[i]][k];
                    for(int j=1;j<=n;j++){
                        if(houses[i]!=j){
                            dp[i][houses[i]][k] = min(dp[i][houses[i]][k],dp[i-1][j][k-1]);
                        }
                    }                        
                }
            }
            else{
                for(int k=1;k<=target;k++){
                    for(int j=1;j<=n;j++){
                        dp[i][j][k] = (dp[i-1][j][k]!=MAX)?(dp[i-1][j][k]+cost[i][j-1]):MAX;
                        for(int j_b=1;j_b<=n;j_b++){
                            if(j!=j_b){
                                dp[i][j][k] = min(dp[i][j][k],(dp[i-1][j_b][k-1]!=MAX)?(dp[i-1][j_b][k-1]+cost[i][j-1]):MAX);
                            }
                        }
                    }
                }
            }
            // for(int j=1;j<=n;j++){
            //     System.out.print(Arrays.toString(dp[i][j])+" ");
            // }
            // System.out.println();
        }
        // System.out.println();
        int ans = MAX;
        for(int j=1;j<=n;j++){
            ans = min(ans,dp[m-1][j][target]);
        }
        return ans!=MAX?ans:-1;        
        
    }
}
