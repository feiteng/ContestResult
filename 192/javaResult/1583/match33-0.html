class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {

        int hl = houses.length;
        
        int dp[][][] = new int[hl+1][n+1][target+2];
        
        for(int i=0;i<hl;++i){
            for(int j=0;j<=n;++j){
                for(int k=0;k<=target+1;++k){
                    dp[i][j][k] = 1000000000;
                }
            }
        }
        
        
        if(houses[0]!=0){
            dp[0][houses[0]][1] = 0;
        }else{
            
            for(int j=1;j<=n;++j){
                // dp[0][j-1][1] = cost[0][j];
                dp[0][j][1] = cost[0][j-1];
            }
            
            
        }
        
        
        
        for(int i=1;i<hl;++i){
            for(int d=1;d<=target;++d){
                for(int c1=1;c1<=n;++c1){
                    if(houses[i]!=0){
                        
                        dp[i][houses[i]][d+(houses[i]==c1?0:1)] = Math.min(dp[i][houses[i]][d+(houses[i]==c1?0:1)] , dp[i-1][c1][d]);
                        
                        
                    }else{
                        
                         for(int c=1;c<=n;++c){
                             
                              dp[i][c][d+(c==c1?0:1)] = Math.min(dp[i][c][d+(c==c1?0:1)] , dp[i-1][c1][d] + cost[i][c-1]);
                             
                    }
                }

            }
            }
        }
        
        
        int r =1000000000;
        for(int c=1;c<=n;++c){
            r = Math.min(r,dp[hl-1][c][target]);
        }
        return r==1000000000?-1:r;
        
        
        
    }
}
