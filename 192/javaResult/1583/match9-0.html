class Solution {
 private static final int INF = 1_000_000_001;
    private static final int MOD = 1_000_000_007;
    private static final int[][] DIR = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp = new int[m + 1][m + 1][n + 1];

        for (int i = 0; i < m + 1; i++) {
            for (int j = 0; j < m + 1; j++) {
                for (int k = 0; k < n + 1; k++) {
                    dp[i][j][k] = INF;
                }
            }
        }

        for (int i = 0; i < m; i++) {
            for (int j = 1; j <= i + 1 && j <= target; j++) {

                if (houses[i] > 0) {
                    if (i == 0) {
                        dp[i][j][houses[i]] = 0;
                    } else {
                        //if (dp[i - 1][j][houses[i]] < INF) {
                        dp[i][j][houses[i]] = Math.min(dp[i][j][houses[i]], dp[i - 1][j][houses[i]]);
                        //}
                        for (int k = 1; k <= n; k++) {
                            if (k == houses[i]) continue;
                            dp[i][j][houses[i]] = Math.min(dp[i][j][houses[i]], dp[i - 1][j - 1][k]);
                        }
                    }
                } else {
                    if (i == 0) {
                        for (int k = 1; k <= n; k++) {
                            dp[i][j][k] = Math.min(dp[i][j][k], cost[i][k - 1]);
                        }
                    } else {
                        for (int k = 1; k <= n; k++) {
                            dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][j][k] + cost[i][k - 1]);
                            for (int l = 1; l <= n; l++) {
                                if (l == k) continue;
                                dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][j - 1][l] + cost[i][k - 1]);
                            }
                            
                        }
                    }
                }
            }
        }
        
        int ans = INF;
        for (int k = 1; k <= n; k++) {
            ans = Math.min(ans, dp[m - 1][target][k]);
        }
        return ans < INF ? ans : -1;

    }
}
