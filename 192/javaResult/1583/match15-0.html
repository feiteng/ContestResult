class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        
        int[][][] dp = new int[m+1][target+1][n];
        
        
        for(int j=1;j<=target;j++){
            for(int i=1;i<j;i++){
                for(int k=0;k<n;k++){
                    dp[i][j][k]=Integer.MAX_VALUE;
                }
            }
        }
        
        for(int i=0;i<=m;i++){
            for(int k=0;k<n;k++){
                dp[i][0][k]=Integer.MAX_VALUE;
            }
        }

        
        // for(int i=0;i<=m;i++){
        //     for(int j=0;j<=target;j++){
        //         System.out.print(Arrays.toString(dp[i][j]));
        //         System.out.print(",");
        //     }
        //     System.out.println();
        // }
    
        for(int j=1;j<=target;j++){
            for(int i=j;i<=m;i++){
                for(int k=0;k<n;k++){
                    if(houses[i-1]!=0 && houses[i-1]-1!=k){
                        dp[i][j][k]=Integer.MAX_VALUE;
                        continue;
                    }
                    
                    int tmpCost = cost[i-1][k];
                    if(houses[i-1]!=0){
                        tmpCost=0;
                    }
                    if(dp[i-1][j][k]==Integer.MAX_VALUE){
                        dp[i][j][k]=Integer.MAX_VALUE;
                    }else{
                        dp[i][j][k]=dp[i-1][j][k]+tmpCost;
                    }
                    for(int l=0;l<n;l++){
                        if(l!=k){
                            int tmp = dp[i-1][j-1][l];
                            if(tmp!=Integer.MAX_VALUE){
                                tmp=tmp+tmpCost;
                            }
                            dp[i][j][k]=Math.min(dp[i][j][k], tmp);
                        }
                    }
                }
            }
        }
        
    
        // for(int i=1;i<=m;i++){
        //     for(int j=1;j<=target;j++){
        //         System.out.print(Arrays.toString(dp[i][j]));
        //         System.out.print(",");
        //     }
        //     System.out.println();
        // }
        
        int ans = Integer.MAX_VALUE;
        for(int k=0;k<n;k++){
            ans = Math.min(ans, dp[m][target][k]);
        }
        if(ans == Integer.MAX_VALUE){
            return -1;
        }
        
        return ans;
    }
}
