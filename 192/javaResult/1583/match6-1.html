class Solution {
   
	public 
int minCost(int[] houses, int[][] cost, int m, int n, int target) {
		long[][][] dp = new long[m][n + 1][target + 1];
		for (int i = 0; i < m; i++) {
			for (int j = 0; j <= n; j++) {
				for (int k = 0; k <= target; k++) {
					dp[i][j][k] = 1L << 40;
				}
			}
		}
		if (houses[0] == 0) {
			for (int i = 1; i <= n; i++) {
				dp[0][i][1] = cost[0][i - 1];
			}
		} else {
			dp[0][houses[0]][1] = 0;
		}
		for (int i = 1; i < m; i++) {
			if (houses[i] == 0) {
				for (int j = 1; j <= n; j++) {
					for (int k = 1; k <= target; k++) {
						dp[i][j][k] = dp[i - 1][j][k] + cost[i][j - 1];
						for (int p = 1; p <= n; p++) {
							if (p == j) continue;
							dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][p][k - 1] + cost[i][j - 1]);
						}
					}

				}
			} else {
				for (int k = 1; k <= target; k++) {

					dp[i][houses[i]][k] = dp[i - 1][houses[i]][k];
					for (int p = 1; p <= n; p++) {
						if (p == houses[i]) continue;
						dp[i][houses[i]][k] = Math.min(dp[i][houses[i]][k], dp[i - 1][p][k - 1]);
					}

				}


			}
		}
		long min = 1L << 40;
		for (int j = 1; j <= n; j++) {
			min = Math.min(dp[m - 1][j][target], min);
		}
		return min >= 1L << 40 ? -1 : (int) min;
	}
}
