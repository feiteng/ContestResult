class Solution {
    int inf = (int)1e9 + 7;
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp = new int[m][n + 1][target + 1];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j <= n; j++) {
                for (int k = 0; k <= target; k++) dp[i][j][k] = inf
;
            }
        }
        
        
        if (houses[0] == 0) {
            for (int i = 1; i <= n; i++) {
                dp[0][i][1] = cost[0][i - 1];
            }
        }
        else dp[0][houses[0]][1] = 0;
        
        for (int i = 1; i < m; i++) {
            if (houses[i] == 0) {
                for (int j = 1; j <= n; j++) {
                    dp[i][j][1] = Math.min(inf, dp[i - 1][j][1] + cost[i][j - 1]);
                }
            }
            else dp[i][houses[i]][1] = dp[i - 1][houses[i]][1];
        }
        
        
        // for (int i = 0; i < m; i++) {
        //     for (int j = 0; j < n; j++) {
        //         System.out.print(dp[i][j + 1][1]);
        //         System.out.print(" ");
        //     }
        //     System.out.print("\n");
        // }
        
        
for (int k = 2; k <= target; k++) {
            for (int i = 1; i < m; i++) {
                if (houses[i] == 0) {
                    for (int j = 1; j <= n; j++) {
                        dp[i][j][k] = Math.min(dp[i][j][k] , dp[i - 1][j][k] + cost[i][j - 1]);
                        for (int o = 1; o <= n; o++) {
                            if (o == j) continue
;
                            dp[i][j][k] = Math.min(dp[i][j][k] , dp[i - 1][o][k - 1] + cost[i][j - 1]);
                        }
                        
                    }
                }
                else {
                    dp[i][houses[i]][k] = dp[i - 1][houses[i]][k];
                    for (int o = 1; o <= n; o++) {
                        if (o == houses[i]) continue
;
                        dp[i][houses[i]][k] = Math.min(dp[i][houses[i]][k], dp[i - 1][o][k - 1]);
                    }
                }
            }
        }
        
        int res = inf;
        for (int i = 1; i <= n; i++) {
            res = Math.min(res, dp[m - 1][i][target]);
        }
        
        return res >= inf ? -1 : res;
        
    }
}
