class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp=new int[m+1][n+1][target+1];
        for(int i=0;i<=m;i++){
            for(int k=0;k<=n;k++){
                for(int l=0;l<=target;l++){
                    dp[i][k][l]=1000000000;
                }
            }
        }
        for(int col=1;col<=n;col++){
            dp[0][col][0]=0;
        
}
        for(int h=1;h<=m;h++){
            for(int col=1;col<=n;col++){
                for(int block=1;block<=target;block++){
                    for(int other=1;other<=n;other++){
                        int c=0;
                        if(houses[h-1]!=0){
                            
if(col==houses[h-1]){
                                c=0;
                            }else{
                                c=1000000000;
                            }
                        }else{
                            c=cost[h-1][col-1];
                        }
                        if(col==other){
                            dp[h][col][block]=Math.min(dp[h-1][other][block]+c,dp[h][col][block]);
                        }else{
                            
dp[h][col][block]=Math.min(dp[h-1][other][block-1]+c,dp[h][col][block]);
                        }
                    }
                    
                }
            }
        }
        int ans=1000000000;
        for(int col=1;col<=n;col++){
            ans=Math.min(ans,dp[m][col][target]);
        }
        if(ans==1000000000) return -1;
        return ans;
    }
}
