class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp = new int[m][n][target + 1];
    final int max = 2000000;
    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        Arrays.fill(dp[i][j], max);
      }
    }
    if (houses[0] == 0) {
      for (int j = 0; j < n; j++) {
        dp[0][j][1] = cost[0][j];
      }
    } else {
      int j = houses[0]-1;
      dp[0][j][1] = 0;
    }
    
for (int i = 1; i < m; i++) {
      if (houses[i] == 0) {
        for (int j = 0; j < n; j++) {
          for (int k = 1; k <= target; k++) {
            int v = max;
            for (int jj = 0; jj < n; jj++) {
              if (jj == j) {
                v = Math.min(v, dp[i-1][jj][k] + cost[i][j]);
              } else {
                v = Math.min(v, dp[i-1][jj][k-1] + cost[i][j]);
              }
            }
            dp[i][j][k] = v;
          }
        }
      } else {
        int j = houses[i]-1;
        
for (int k = 1; k <= target; k++) {
          int v = max;
          for (int jj = 0; jj < n; jj++) {
            if (jj == j) {
              v = Math.min(v, dp[i-1][jj][k]);
            } else {
              v = Math.min(v, dp[i-1][jj][k-1]);
            }
          }
          dp[i][j][k] = v;
        }
      
}
    }
    int ans = max;
    for (int j = 0; j < n; j++) {
      ans = Math.min(ans, dp[m-1][j][target]);
    }
    return ans >= max ? -1 : ans;
    }
}
