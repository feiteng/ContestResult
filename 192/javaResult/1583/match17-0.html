class Solution {
    
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        long[][][] arr = new long[m + 1][n][target + 1];
        for(int i = 0; i < arr.length; i++) {
            for(int j = 0; j < arr[i].length; j++) {
                for(int k = 0; k <= target; k++)
                {if(i == 0 && k == 0) {
                    continue;
                }
                    arr[i][j][k] = Integer.MAX_VALUE;}
            }
        }
        for(int i = 1; i <= m; i++) {
            if(houses[i - 1] == 0) {
                for(int j = 0; j < n; j++) {
                    for(int k = 1; k <= target; k++) {
                        long ct = Integer.MAX_VALUE;
                        for(int l = 0; l < n; l++) {
                            if(l == j) {
                                ct = Math.min(ct, arr[i - 1][l][k] + cost[i - 1][j]);
                            } else {
                                ct = Math.min(ct, arr[i - 1][l][k - 1] + cost[i - 1][j]);
                            }
                        }
                        arr[i][j][k] = ct;
                        // System.out.println(i + " " + j + " " + k + " " + arr[i][j][k]);
                    }
                }
            } else {
                int j = houses[i - 1] - 1;
                {
                    for(int k = 1; k <= target; k++) {
                        long ct = Integer.MAX_VALUE;
                        for(int l = 0; l < n; l++) {
                            if(l == j) {
                                ct = Math.min(ct, arr[i - 1][l][k]);
                            } else {
                                ct = Math.min(ct, arr[i - 1][l][k - 1]);
                            }
                        }
                        arr[i][j][k] = ct;
                    }
                }
            }
        }
        long ans = Integer.MAX_VALUE;
        for(int i = 0; i < n; i++) {
            ans = Math.min(ans, arr[m][i][target]);
        }
        if(ans == Integer.MAX_VALUE) {
            ans = -1;
        }
        return (int)ans;
    }
}
