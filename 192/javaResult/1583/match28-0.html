class Solution {
    int INF = 1_000_001;
    int[] houses;
    int[][] cost;
    int m;
    int n;
    Integer[][][] memo;
    
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        this.houses = houses;
        this.cost = cost;
        this.m = m;
        this.n = n;
        this.memo = new Integer[m][m + 1][n + 1];
        
        int ans = dp(0, target, 0);
        return ans >= INF ? -1 : ans;
    }
    
    int dp(int i, int t, int p) {
        if (t < 0) return INF;
        if (i >= m) return t == 0 ? 0 : INF;
        if (memo[i][t][p] != null) return memo[i][t][p];
        
        int ans = INF;
        if (houses[i] > 0) {
            if (houses[i] == p) {
                ans = Math.min(ans, dp(i + 1, t, p));
            } else {
                ans = Math.min(ans, dp(i + 1, t - 1, houses[i]));
            }
        } else {
            for (int j = 0; j < n; j++) {
                int color = j + 1;
                if (p == color) ans = Math.min(ans, cost[i][j] + dp(i + 1, t, p));
                else ans = Math.min(ans, cost[i][j] + dp(i + 1, t - 1, color));
            }
        }
        
        memo[i][t][p] = ans;
        return ans;
    }
}
