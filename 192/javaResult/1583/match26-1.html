class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp = new int[m][n+1][target+1];
        for (int i=0; i<m; i++)
            for (int j=1; j<=n; j++)
                for (int k=0; k<=target; k++)
                    dp[i][j][k] = -1;
        if (houses[0] > 0) {
            dp[0][houses[0]][1] = 0;
        } else {
            for (int i=1; i<=n; i++)
                dp[0][i][1] = cost[0][i-1]
;
        }
        for (int i=1; i<m; i++) {
            for (int k=1; k<=target; k++) {
                if (houses[i] > 0) {
                    int val = -1;
                    
for (int l=1; l<=n; l++) {
                        if (l == houses[i] && dp[i-1][l][k] >= 0) {
                            val = val == -1 ? dp[i-1][l][k] : Math.min(val, dp[i-1][l][k]);
                        } else if (l != houses[i] && dp[i-1][l][k-1] >= 0) {
                            val = val == -1 ? dp[i-1][l][k-1] : Math.min(val, dp[i-1][l][k-1]);
                        }
                    }
                    dp[i][houses[i]][k] = val;
                } else {
                    for (int j=1; j<=n; j++) {
                        int val = -1;
                        
for (int l=1; l<=n; l++) {
                            if (l == j && dp[i-1][l][k] >= 0) {
                                val = val == -1 ? dp[i-1][l][k]+cost[i][j-1] : Math.min(val, dp[i-1][l][k]+cost[i][j-1]);
                            } else if (l != j && dp[i-1][l][k-1] >= 0) {
                                val = val == -1 ? dp[i-1][l][k-1]+cost[i][j-1] : Math.min(val, dp[i-1][l][k-1]+cost[i][j-1]);
                            }
                        }
                        dp[i][j][k] = val;
                    }
                }
            
}
        }
        int ret = -1;
        for (int i=1; i<=n; i++) {
            if (dp[m-1][i][target] >= 0) {
                ret = ret == -1 ? dp[m-1][i][target] : Math.min(ret, dp[m-1][i][target]);
            }
        }
        return ret;
    }
}
