class Solution {
    int[] a;
    int[][] cost;
    int m, n;
    int target;
    
    int[][][] dp;
    public int minCost(int[] h, int[][] c, int m, int n, int target) {
        a = h;
        cost = c;
        this.m = m;
        this.n = n;
        this.target = target;
        
        dp = new int[m+1][target+1][n+2];
        
        for(int i=0; i<m+1; i++) {
            for(int j=0; j<target+1; j++) {
                Arrays.fill(dp[i][j], -1);
            }
        }
        
int ans = solve(0, target, -1);
        
        return ans==(int)1e9?-1:ans;
    }
    
    public int solve(int i, int left, int prev) {
        if (left<0) return (int)1e9;
        if (i==m) {
            if (left==0) return 0;
            return (int)1e9;
        
}
        
        if (dp[i][left][prev+1]!=-1) return dp[i][left][prev+1];
        
        int ans = (int)1e9;
        
        if (a[i]!=0) {
            if (prev!=a[i]) {
                ans = Math.min(ans, solve(i+1, left-1, a[i]));
            } else {
                ans = Math.min(ans, solve(i+1, left, a[i]));
            }
        } else {
            for(int c = 1; c<=n; c++) {
                if (c!=prev) {
                    ans = Math.min(ans, cost[i][c-1]+solve(i+1, left-1, c));
                } else {
                    ans = Math.min(ans, cost[i][c-1]+solve(i+1, left, c));
                }
            }
        }
        
        return dp[i][left][prev+1]=ans;
        
    }
}
