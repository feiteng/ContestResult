class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int[][][] dp = new int[m + 1][n + 1][target + 1];
        for (int j = 1; j <= n; j++) {
            for (int k = 1; k <= target; k++) {
                dp[1][j][k] = -1;
            }
        }
        if (houses[0] != 0) {
            for (int j = 1; j <= n; j++) {
                dp[1][j][1] = houses[0] == j ? 0 : -1;
            }
        } else {
            for (int j = 1; j <= n; j++) {
                dp[1][j][1] = cost[0][j - 1];
            }
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j <= n; j++) {
                for (int k = 1; k <= target; k++) {
                    dp[i + 1][j][k] = -1;
                    if (houses[i] != 0) {
                        // same color as painted, check previous house colors
                        if (houses[i] == j) {
                            for (int p = 1; p <= n; p++) {
                                if (p != houses[i] && dp[i][p][k - 1] >= 0 && k > 1) {
                                    dp[i + 1][j][k] = dp[i + 1][j][k] >= 0 ? Math.min(dp[i + 1][j][k], dp[i][p][k - 1]) : dp[i][p][k - 1];
                                } else if (p == houses[i] && dp[i][p][k] >= 0) {
                                    dp[i + 1][j][k] = dp[i + 1][j][k] >= 0 ? Math.min(dp[i + 1][j][k], dp[i][p][k]) : dp[i][p][k];
                                }
                            }
                        }
                    } else {
                        for (int p = 1; p <= n; p++) {
                            if (p != j && dp[i][p][k - 1] >= 0 && k > 1) {
                                // System.out.println(dp[i + 1][j][k] + " " + dp[i][p][k - 1] + " " + cost[i][j - 1]);
                                dp[i + 1][j][k] = dp[i + 1][j][k] >= 0 ? Math.min(dp[i + 1][j][k], dp[i][p][k - 1] + cost[i][j - 1]) : dp[i][p][k - 1] + cost[i][j - 1];
                                // System.out.println(dp[i + 1][j][k]);
                            } else if (p == j && dp[i][p][k] >= 0) {
                                dp[i + 1][j][k] = dp[i + 1][j][k] >= 0 ? Math.min(dp[i + 1][j][k], dp[i][p][k] + cost[i][j - 1]) : dp[i][p][k] + cost[i][j - 1];
                            }
                        }
                    }
                    // System.out.println(i + " " + j + " " + k + " " + dp[i + 1][j][k]);
                }
            }
        }
        int minCost = -1;
        for (int j = 1; j <= n; j++) {
            if (dp[m][j][target] >= 0) {
                minCost = minCost < 0 ? dp[m][j][target] : Math.min(minCost, dp[m][j][target]);
            }
        }
        return minCost;
    }
}
