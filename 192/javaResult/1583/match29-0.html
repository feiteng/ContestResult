class Solution {
    int[][][] dp;

    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        dp = new int[houses.length][n + 1][target + 1];// 以i1结尾的房子涂成i2色有i3种街区的cost

        for (int[][] a : dp) {
            for (int[] b : a) {
                Arrays.fill(b, Integer.MAX_VALUE);
            }
        }
        if (houses[0] == 0) {
            for (int i = 1; i <= n; i++) {
                dp[0][i][1] = cost[0][i - 1];
            }
        } else {
            dp[0][houses[0]][1] = 0;
        }
        for (int i1 = 1; i1 < houses.length; i1++) {
            for (int tar = 1; tar <= target; tar++) {
                for (int lastcolor = 1; lastcolor <= n; lastcolor++) {
                    if (houses[i1] != 0) {
                        int t = tar;
                        if (houses[i1] != lastcolor) {
                            t--;
                        }
                        if (dp[i1 - 1][lastcolor][t] != Integer.MAX_VALUE) {
                            dp[i1][houses[i1]][tar] = Math.min(dp[i1][houses[i1]][tar], dp[i1 - 1][lastcolor][t]);
                        }

                    } else {
                        for (int curcolor = 1; curcolor <= n; curcolor++) {
                            int t = tar;
                            if (curcolor != lastcolor) {
                                t--;
                            }
                            if (dp[i1 - 1][lastcolor][t] != Integer.MAX_VALUE) {
                                dp[i1][curcolor][tar] = Math.min(dp[i1][curcolor][tar], dp[i1 - 1][lastcolor][t] + cost[i1][curcolor - 1]);
                            }
                        }
                    }
                }
            }
        }
        int ans = Integer.MAX_VALUE;
        for (int c = 1; c <= n; c++) {
            ans = Math.min(ans, dp[houses.length - 1][c][target]);
        }
        return ans==Integer.MAX_VALUE?-1:ans;
    }
}
