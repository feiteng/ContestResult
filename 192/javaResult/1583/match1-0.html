class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int [][][]dp = new int[m][n][target];
        int max = 10000000;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                Arrays.fill(dp[i][j], max);
            }
        }
        if(houses[0] == 0){
            for(int j=0;j<n;j++){
                dp[0][j][0] = cost[0][j];
            }
        }else{
            dp[0][houses[0]-1][0] = 0;
        }
        for(int i=1;i<m;i++){
            if(houses[i] != 0){
                int color = houses[i]-1;
                for(int j=0;j<n;j++){
                    for(int k=0;k<target && k<i;k++){
                        if(dp[i-1][j][k] != max){
                            int count = color == j ? k : k+1;
                            if(count < target){
                                dp[i][color][count] = Math.min(dp[i][color][count], dp[i-1][j][k]);
                            }
                        }
                    }
                }
            }else{
                for(int l=0;l<n;l++){
                    for(int j=0;j<n;j++){
                        for(int k=0;k<target && k<i;k++){
                            if(dp[i-1][j][k] != max){
                                int count = l == j ? k : k+1;
                                if(count < target){
                                    dp[i][l][count] = Math.min(dp[i][l][count], dp[i-1][j][k] + cost[i][l]);
                                }
                            }
                        }
                    }
                }
            }    
        }
        int ans = max;
        for(int j=0;j<n;j++){
            ans = Math.min(ans, dp[m-1][j][target-1]);
        }
        if(ans == max){
            return -1;
        }
        return ans;
    }
}
