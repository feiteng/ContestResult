class Solution {
	    static int minCost(int[] houses, int[][] cost, int m, int n, int target) {
	        int[][][] dp = new int[m+1][m+1][n+1];
	        int maxv = 1000000;
	        for(int i = 0; i <= m; i++){
	            for(int j = 0; j <= m; j++) Arrays.fill(dp[i][j], maxv);
	        }
	        dp[0][0][0] = 0;
	        for(int
 i = 1; i <= m; i++){
	            if(houses[i-1] == 0){
	                for(int j = 0; j < m; j++){
	                    for(int k = 0; k <= n; k++){
	                        for(int
 k2 = 1; k2 <= n; k2++){
	                            if(k == k2){
	                                dp[i][j][k2] = Math.min(dp[i][j][k2], dp[i-1][j][k] + cost[i-1][k2-1]);
	                            }
	                            else{
	                                dp[i][j+1][k2] = Math.min(dp[i][j+1][k2], dp[i-1][j][k] + cost[i-1][k2-1]);
	                            }
	                        }
	                    }
	                }
	            }
	            else{
	                for(int j = 0; j < m; j++){
	                	for(int k = 0;
 k <= n; k++){
	                        if(houses[i-1] == k){
	                            dp[i][j][houses[i-1]] = Math.min(dp[i][j][houses[i-1]], dp[i-1][j][k]);
	                        }
	                        else{
	                            dp[i][j+1][houses[i-1]] = Math.min(dp[i][j+1][houses[i-1]], dp[i-1][j][k]);
	                        }
	                    }
	                }  

	            }
	        }
	        int res = maxv;
	        for(int k = 0; k <= n; k++){
	            res = Math.min(res, dp[m][target][k]);
	        }
	        if(res == maxv) return -1;
	        else return res;
	    }
	}
