class Solution {
        int INF = 1000001;
        public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
            int[][][] dp = new int[m][target][n];
            for(int i = 0; i < m; i++) {
                for(int j = 0; j < n; j++) {
                    for(int t = 0; t < target; t++) {
                        dp[i][t][j] = INF;
                    }
                }
            }
            if(houses[0] == 0) {
                for(int j = 0; j < n; j++) {
                    dp[0][0][j] = cost[0][j];
                }
            }
            else {
                dp[0][0][houses[0] - 1] = 0;
            }

            for(int i = 1; i < m; i++) {
                for(int t = 0; t < target; t++) {
                    if(houses[i] == 0) {
                        for(int j = 0; j < n; j++) {
                            dp[i][t][j] = Math.min(dp[i][t][j], dp[i - 1][t][j] + cost[i][j]);
                            for(int j2 = 0; j2 < n; j2++) {
                                if (j2 == j || t == 0) {
                                    continue;
                                }
                                dp[i][t][j] = Math.min(dp[i][t][j], dp[i - 1][t - 1][j2] + cost[i][j]);
                            }
                        }
                    }
                    else {
                        dp[i][t][houses[i] - 1] = Math.min(dp[i][t][houses[i] - 1], dp[i - 1][t][houses[i] - 1]);
                        for(int j2 = 0; j2 < n; j2++) {
                            if (j2 == houses[i] - 1 || t == 0) {
                                continue;
                            }
                            dp[i][t][houses[i] - 1] = Math.min(dp[i][t][houses[i] - 1], dp[i - 1][t - 1][j2]);
                        }
                    }
                }
            }
            
            int ans = INF;
            
            for(int j = 0; j < n; j++) {
                ans = Math.min(ans, dp[m - 1][target - 1][j]);
            }
            if(ans == INF) return -1;
            return ans;
        }
    }
