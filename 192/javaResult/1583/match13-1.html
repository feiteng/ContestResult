class Solution {
    public int minCost(int[] houses, int[][] cost, int m, int n, int target) {
        int [][][] dp = new int[m+1][n+1][target+1];
        //cost [i-1][c-1]
        //house[i-1];
        for(int i=1;i<=m;i++){
            for(int c=1;c<=n;c++){
                for(int t=0;t<=target;t++){
                    // System.out.println(i + "  " + c + " "+ t);
                    if(t==0 || t>i) {
                        dp[i][c][t] = 1000000000;
                        continue;
                    }
                    if(houses[i-1]>0){
                        //already painted
                        if(c!=houses[i-1]){
                            dp[i][c][t] = 1000000000;
                            continue;
                        }
                        int otherMin = 1000000000;
                    
for(int ctemp=1;ctemp<=n;ctemp++){
                        if(ctemp==c){
                            continue;
                        }
                        otherMin = Math.min(otherMin,dp[i-1][ctemp][t-1]);
                    }
                        dp[i][c][t] = Math.min(otherMin,dp[i-1][c][t]);
                    }
                    else{
                    int tempCost = cost[i-1][c-1];
                    int otherMin = 1000000000;
                    
for(int ctemp=1;ctemp<=n;ctemp++){
                        if(ctemp!=c){
                        otherMin = Math.min(otherMin,dp[i-1][ctemp][t-1]);
                        }
                    }
                        
                       
                    dp[i][c][t] = Math.min(tempCost+dp[i-1][c][t],tempCost+otherMin);
                    
}
                    
                }
            }
        }
        int ret = 1000000000;
      
        for(int i=1;i<=n;i++){
            ret = Math.min(ret,dp[m][i][target]);
        }
        if(ret==1000000000){
            return -1;
        }
        return ret;
    }
}
