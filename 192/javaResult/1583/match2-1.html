class Solution {
    public int minCost(int[] houses, int[][] cost, final int M, final int N, final int target) {
        int[][][] dp = new int[M][N][M + 1];
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                for (int k = 0; k <= M; ++k) {
                    dp[i][j][k] = Integer.MAX_VALUE;
                }
            
}
        }
        if (houses[0] == 0) {
            for (int j = 0; j < N; ++j) {
                dp[0][j][1] = cost[0][j];
            }
        } else {
            dp[0][houses[0] - 1][1] = 0;
        }
        for (int i = 1; i < M; ++i) {
            if (houses[i] == 0) {
                
for (int j = 0; j < N; ++j) {
                    for (int k = 1; k <= M; ++k) {
                        for (int c = 0; c < N; ++c) {
                            if (j == c) {
                                if (dp[i - 1][c][k] != Integer.MAX_VALUE) {
                                    dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][c][k] + cost[i][j]);
                                }
                            } else {
                                
if (dp[i - 1][c][k - 1] != Integer.MAX_VALUE) {
                                    dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][c][k - 1] + cost[i][j]);
                                }
                            }
                        }
                    }
                }
            } else {
                
int j = houses[i] - 1;
                for (int k = 1; k <= M; ++k) {
                    for (int c = 0; c < N; ++c) {
                        if (j == c) {
                            dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][c][k]);
                        } else {
                            dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][c][k - 1]);
                        }
                    
}
                }
            }
        }

        int minCost = Integer.MAX_VALUE;
        for (int j = 0; j < N; ++j) {
            minCost = Math.min(minCost, dp[M - 1][j][target]);
        }

        return minCost == Integer.MAX_VALUE ? -1 : minCost;
    }
}
