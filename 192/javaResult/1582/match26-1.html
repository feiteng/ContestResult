	class B
rowserHistory {
		Deque<String> back;
		Deque<String> forw;

	    public BrowserHistory(String homepage) {
	        back = new ArrayDeque<String>();
	        forw = new ArrayDeque<String>();
	        back.add(homepage);
	    }
	    
	    public void visit(String url) {
	        back.addLast(url);
	        forw.clear();
	    }
	    
	    public String back(int steps) {
	        while(back.size() > 1 && steps > 0){
	        	forw.addLast(back.pollLast());
	        	steps--;
	        }
	        return back.peekLast();
	    }
	    
	    public String forward(int steps) {
	        while(forw.size() > 0 && steps > 0){
	        	back.addLast(forw.pollLast());
	        	steps--;
	        }
	        return back.peekLast();
	    }
	}
