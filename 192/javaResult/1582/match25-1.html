class BrowserHistory {
    class LLNode {
        String url;
        LLNode prev;
        LLNode next;
        
        public LLNode(String url) {
            this.url = url;
        }
    }
    
    LLNode root;
    LLNode curr;
    
    public BrowserHistory(String homepage) {
        root = new LLNode(homepage);
        curr = root;
    }
    
    public void visit(String url) {
        LLNode node = new LLNode(url);
        curr.next = node;
        node.prev = curr;
        curr = curr.next;
    
}
    
    public String back(int steps) {
        while (curr.prev != null && steps >= 1) {
            curr = curr.prev;
            steps--;
        }
        
        return curr.url;
    }
    
    public String forward(int steps) {
        while (curr.next != null && steps >= 1) {
            curr = curr.next;
            steps--;
        }
        
        return curr.url;
    }
}
