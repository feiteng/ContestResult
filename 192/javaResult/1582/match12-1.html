class BrowserHistory {
    public class Node {
        private String url;
        private Node next;
        private Node prev;
        Node(String url) {
            this.url = url;
        }
    }
    public Node home;
    public Node curr;
    
    public BrowserHistory(String homepage) {
        this.home = new Node(homepage);
        this.curr = home;
    }
    
    public void visit(String url) {
        Node newNode = new Node(url);
        curr.next = newNode;
        newNode.prev = curr;
        curr = newNode;
        return;
    
}
    
    public String back(int steps) {
        while (curr != home && steps > 0) {
            curr = curr.prev;
            steps--;
        }
        return curr.url;
    }
    
    public String forward(int steps) {
        while (curr.next != null && steps > 0) {
            curr = curr.next;
            steps--;
        }
        return curr.url;
    }
}
