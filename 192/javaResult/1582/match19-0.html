class BrowserHistory {

    String[] his; 
    int p; 
    int last;
    public BrowserHistory(String homepage) {
        his = new String[5001];
        p=0;
        last = 0; 
        his[0] = homepage;
    }
    
    public void visit(String url) {
        his[++p] = url;
        last = p; 
    }
    
    public String back(int steps) {
        p = Math.max(0, p-steps);
        return his[p];
    }
    
    public String forward(int steps) {
        p = Math.min(last, p+steps);
        return his[p];
    }
}
