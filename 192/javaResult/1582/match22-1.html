class BrowserHistory {
    
    String[] history;
    int maxStep;
    int curStep;

    public BrowserHistory(String homepage) {
        history = new String[5001];
        history[0] = homepage;
        maxStep = 0;
        curStep = 0;
    }
    
    public void visit(String url) {
        curStep++;
        history[curStep] = url;
        maxStep = curStep;
    }
    
    public String back(int steps) {
        curStep = Math.max(0, curStep - steps);
        // while (curStep >= terminal) {
        //     curStep--;
        // }
        return history[curStep];
    }
    
    public String forward(int steps) {
        curStep = Math.min(maxStep, curStep + steps);
        // while (curStep <= terminal) {
        //     curStep++;
        // }
        return history[curStep];
    }
}

 * String param_3 = obj.forward(steps);
