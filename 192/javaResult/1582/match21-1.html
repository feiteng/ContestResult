class BrowserHistory {
    String [] his;
    int now = 0;
    int maxNow = now;
    public BrowserHistory(String homepage) {
        his = new String[5000];
        his[0] = homepage;
        now = 0;
        maxNow = now;
    }

    public void visit(String url) {
        now ++;
        his[now] = url;
        maxNow = now;
    }

    public String back(int steps) {
        now -= steps;
        now = Math.max(now, 0);
        return his[now];
    }

    public String forward(int steps) {
            now = Math.min(maxNow, now + steps);
            return his[now];
    }
}
