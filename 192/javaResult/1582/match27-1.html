class BrowserHistory {
    
    private ArrayList<String> histories = new ArrayList<>();
    private int size = 0;
    private int curIdx = 0;

    public BrowserHistory(String homepage) {
        histories.add(homepage);
        size++;
        curIdx = 0;
    
}
    
    public void visit(String url) {
        if (curIdx == histories.size() - 1) {
            histories.add(url);
        } else {
            histories.set(curIdx + 1, url);
        }
        curIdx++;
        size = curIdx + 1;
    }
    
    public String back(int steps) {
        curIdx = Math.max(0, curIdx - steps);
        return histories.get(curIdx);
    }
    
    public String forward(int steps) {
        curIdx = Math.min(size - 1, curIdx + steps);
        return histories.get(curIdx);
    }
}
