class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Arrays.sort(arr);
        int m = arr[(n - 1) / 2];
        int[] ans = new int[k];
        int id = 0, l = 0, r = n - 1;
        while (id < k) {
            if (strong(arr[l], arr[r], m)) {
                ans[id++] = arr[l++];
            } else {
                ans[id++] = arr[r--];
            }
        }
        return ans;
    }
    
    private boolean strong(int a, int b, int m) {
        int da = Math.abs(a - m), db = Math.abs(b - m);
        return da > db ? true : a > b;
    }
}
