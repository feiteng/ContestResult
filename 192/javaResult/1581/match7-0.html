class Solution {
    public int[] getStrongest(int[] arr, int K) {
        Arrays.sort(arr);
        int k = 0;
        int[] res =  new int[K];
        int j = arr.length - 1;
        int i = 0;
        int mid = j / 2;
        while (i <= j && k < K) {
            int r = Math.abs(arr[j] - arr[mid]);
            int l =  Math.abs(arr[i] - arr[mid]);
            if (r >= l) {
                res[k] = arr[j];
                k++;
                j--;
            } else {
                res[k] = arr[i];
                k++;
                i++;
            }
        }
        //6,-3,7,2,11
        //-3 2 6 7 11
        return res;
    }
}
