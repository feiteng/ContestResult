import java.util.Arrays;

public class Solution {
    public int[] getStrongest(int[] arr, int k) {
        if (arr.length==0) return arr;
        Arrays.sort(arr);
        int[] ans=new int[k];
        int index=0;
        int mid=arr[(arr.length-1)/2];
        // System.out.println(mid);
        int left=0,right=arr.length-1;
        while(index<k){
            int leftStrong=Math.abs(arr[left]-mid);
            int rightStrong=Math.abs(arr[right]-mid);
            // System.out.println(arr[left]+" "+leftStrong);
            // System.out.println(arr[right]+" "+rightStrong);
            if (leftStrong>rightStrong){
                ans[index++]=arr[left++];
            }else{
                ans[index++]=arr[right--];
            }
        }
        return ans;
    }
}
