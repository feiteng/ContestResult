class Solution {
    
public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int mid = arr[(arr.length - 1) / 2];
        ArrayList<Integer> sort = new ArrayList<>();
        for (int i = 0; i < arr.length; i++) {
            sort.add(i);
        }
        Collections.sort(sort, (a, b) -> Math.abs(arr[a] - mid) == Math.abs(arr[b] - mid) ? arr[b] - arr[a] : Math.abs(arr[b] - mid) - 
Math.abs(arr[a] - mid));
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = arr[sort.get(i)];
        }
        return res;
    }
}
