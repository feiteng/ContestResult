class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int len = arr.length;
        int m = arr[(len - 1) / 2];

        int i = 0, j = len - 1, x = 0;
        int[] res = new int[k];
        int a, b;
        while (x < k) {
            a = Math.abs(arr[i] - m);
            b = Math.abs(arr[j] - m);
            if (a > b || a == b && arr[i] > arr[j]) {
                res[x++] = arr[i++];
            } else {
                res[x++] = arr[j--];
            }
        }
        return res;
    }
}
