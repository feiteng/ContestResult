class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int[] ans = new int[k];
        int l = 0;
        int r = arr.length-1;
        int N = arr.length;
        int median = arr[((N - 1) / 2)];
        for (int i = 0; i < k; i++) {
            int left = Math.abs(arr[l] - median);
            int right = Math.abs(arr[r] - median);
            if (right >= left) {
                ans[i] = arr[r];
                r--;
            } else {
                ans[i] = arr[l];
                l++;
            }
        }
        return ans;
    }
}
