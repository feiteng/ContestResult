class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int m = arr[(arr.length - 1) / 2];
        int i = 0, j = arr.length - 1;
        int[] ans = new int[k];
        int idx = 0;
        while(idx < k) {
            if (Math.abs(arr[i] - m) > Math.abs(arr[j] - m)) {
                ans[idx++] = arr[i++];
            } else {
                ans[idx++] = arr[j--];
            }
        }
        return ans;
    }
}
