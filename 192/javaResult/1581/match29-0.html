class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Arrays.sort(arr);
        int median = arr[(n - 1) / 2];

        Integer[] nums = new Integer[n];
        for (int i = 0; i < n; i++) nums[i] = arr[i];

        Arrays.sort(nums, (a, b) -> {
            int x = Math.abs(a - median);
            int y = Math.abs(b - median);
            if (x != y) return Integer.compare(y, x);
            else return Integer.compare(b, a);
        });
        
        int[] ret = new int[k];
        for (int i = 0; i < k; i++) ret[i] = nums[i];
        return ret;
    }
}
