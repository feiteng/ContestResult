class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int median = arr[((n - 1) / 2)];
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> {
            int ab = Math.abs(a - median);
            int bb = Math.abs(b - median);
            if (ab == bb) {
                return a - b;
            } else {
                return ab - bb;
            }
        });
        for (int a: arr) {
            pq.offer(a);
            if (pq.size() > k) pq.poll();
        }
        int[] res = new int[k];
        int idx = 0;
        while (!pq.isEmpty()) 
{
            res[idx] = pq.poll();
            idx ++;
        }
        return res;
    }
}
