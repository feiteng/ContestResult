class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int m = arr[((n - 1) / 2)];
        
        PriorityQueue<Integer> Q = new PriorityQueue<>((a, b) -> {
            int v1 = Math.abs(a - m);
            int v2 = Math.abs(b - m);
            if (v1 == v2) {
                return b - a;
            } else {
                return v2 - v1;
            }
        });
        
        for (int v : arr) {
            Q.offer(v);
        }
        
        int[] ans = new int[k];
        int i = 0;
        while (k-- > 0) {
            ans[i++] = Q.poll();
        }
        
        return ans;
    }
}
