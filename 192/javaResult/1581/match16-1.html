class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int[] ans=new int[k];
        
        int cur=0;
        int m=arr[(arr.length-1)/2];
        int l=0,r=arr.length-1;
        
        while(cur<k){
            if(arr[r]-m>=m-arr[l]){
                
ans[cur++]=arr[r--];
            }else{
                ans[cur++]=arr[l++];
            }
        }
        return ans;
    }
}
