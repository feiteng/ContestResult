import java.util.Arrays;


class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int m = arr[((arr.length - 1) / 2)];
        Integer[] idx = new Integer[arr.length];
        for (int i = 0; i < idx.length; i++) {
            idx[i] = i;
        }
        Arrays.sort(idx, (o1, o2) -> {
            int v = Math.abs(arr[o2] - m) - Math.abs(arr[o1] - m);
            if (v == 0) {
                return arr[o2] - arr[o1];
            }
            return v;
        });
        
int ans[] = new int[k];
        for (int i = 0; i < k; i++) {
            ans[i] = arr[idx[i]];
        }
        return ans;

    }
}
