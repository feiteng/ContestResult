class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int[] ans = new int[k];
        int med = arr[(arr.length-1)/2];
        int left = 0;
        int right = arr.length-1;
        int counter = 0;
        while (left <= right) {
            if (counter == k) {
                break;
            }
            if (Math.abs(arr[left]-med) > Math.abs(arr[right]-med)) {
                ans[counter] = arr[left];
                left++;
            }
            else {
                ans[counter] = arr[right];
                right--;
            }
            counter++;
        }
        return ans;
    }
}
