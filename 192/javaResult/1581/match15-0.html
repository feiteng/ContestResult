class Solution {
  public int[] getStrongest(int[] arr, int k) {
    Arrays.sort(arr);
    int m = arr[arr.length - 1 >> 1];
    PriorityQueue<Integer> pq = new PriorityQueue<>((i, j) -> {
      int x = Math.abs(i - m);
      int y = Math.abs(j - m);
      if (x != y) {
        return x - y;
      }
      return i - j;
    });
    for (int num : arr) {
      pq.add(num);
      if (pq.size() > k) {
        pq.poll();
      }
    }
    int[] ret = new int[k];
    int i = 0;
    while (!pq.isEmpty()) {
      ret[i++] = pq.poll();
    }
    return ret;
  }
}
