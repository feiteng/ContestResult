class Solution {
    public int[] getStrongest(int[] arr, int k) {
        // arr[i] is stronger than arr[j] if:
        // the absolute difference from m is greater,        
        // |arr[i] - m| > |arr[j] - m|
        // m is the median.
        
        // On a tie, 
        // arr[i] is stronger if arr[i] > arr[j]
        
        // Return strongest k values in the array.
        Arrays.sort(arr);        
        int mid = (arr.length - 1) / 2;
        int m = arr[mid];
        
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>(new Comparator<Integer>() {
            @Override
            public int compare(
Integer a, Integer b) {
                int diff1 = Math.abs(arr[a] - m);
                int diff2 = Math.abs(arr[b] - m);
                
                if (diff1 == diff2) {
                    return arr[a] - arr[b];
                }
                
                return diff1 - diff2;
            }
        });
        
        
for (int i = 0; i < arr.length; i++) {            
            pq.offer(i);
            
            if (pq.size() > k) {
                pq.poll();
            }
        }
        
        int[] result = new int[k];
        
        for (int i = 0; i < k; i++) {
            result[i] = arr[pq.poll()];
        }
        
        return result;
    }
}
