class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Arrays.sort(arr);
        int m = arr[(n-1)/2];
        int a = 0;
        int b = n-1;
        int[] ans = new int[k];
        while (k>0) {
            if (m-arr[a]>arr[b]-m) {
                ans[k-1]=arr[a];
                ++a;
            } else {
                ans[k-1]=arr[b];
                --b;
            }
            --k;
        }
        return ans;
    }
}
