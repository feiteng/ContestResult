class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int median = arr[(n - 1) / 2];
        int start = 0, end = n - 1, index = 0;
        int[] res = new int[k];
        while (index < k) {
            if (Math.abs(arr[start] - median) > Math.abs(arr[end] - median)) {
                res[index++] = arr[start++];
            } else {
                res[index++] = arr[end--];
            }
        }
        return res;
    }
}
