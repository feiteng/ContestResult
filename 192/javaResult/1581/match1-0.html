class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        /*
        int[][] res = new int[n][2];
        
        for (int i = 0; i < n; i++) {
            res[i][0] = arr[i];
            res[i][1] = i;
        }
        
        Arrays.sort(res, (a, b) -> (a[0] - b[0]));
        */
        
        Arrays.sort(arr);
        
        int mid = arr[(n - 1) / 2];
        int[] ans = new int[k];
        int l = 0, r = n - 1;
        
        for (int i = 0; i < k; i++) {
            if (Math.abs(arr[r] - mid) >= Math.abs(arr[l] - mid)) {
                ans[i] = arr[r];
                r--;
            } else {
                ans[i] = arr[l];
                l++;
            }
        }
        
        return ans;
    }
}
