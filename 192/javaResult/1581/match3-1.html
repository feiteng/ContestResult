class Solution {
    public int[] getStrongest(int[] arr, int k) {
      Arrays.sort(arr);
      int n = arr.length;
      int i = 0, j = arr.length-1;
      int med = arr[(n-1)/2];
      int[] res = new int[k];
      int ri = 0;
      
while(ri < k) {
        if(Math.abs(arr[i] - med) > Math.abs(arr[j]- med)) {
          res[ri] = arr[i];
          i++;
        } else {
          res[ri] = arr[j];
          j--;
        }
        ri++;
      }
      return res;
    }
  }
