class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int m = arr[(n-1)/2];
       // System.out.println(m);
        Integer[] arr2 = new Integer[n];
        for(int i=0;i<n;i++)
        {
            arr2[i] = arr[i];
        }
        Arrays.sort(arr2,(a,b)->Math.abs(a-m)==Math.abs(b-m)?b-a:Math.abs(b-m)-Math.abs(a-m));
        int[] res = new int[k];
        for(int i=0;i<k;i++)
        {
            res[i] = arr2[i];
        }
        return res;
    }
}
