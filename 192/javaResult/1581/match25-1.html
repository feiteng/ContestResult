class Solution {
    public int[] getStrongest(int[] nums, final int K) {
        Arrays.sort(nums);
        final int N = nums.length;
        final int M = nums[(N - 1) / 2];

        PriorityQueue<Integer> minHeap = new PriorityQueue<>(K + 1, (a, b) -> {
            int cmp = Math.abs(a - M) - Math.abs(b - M);
            return (cmp == 0) ? a - b : cmp;
        });

        for (int num : nums) {
            minHeap.offer(num);
            if (minHeap.size() > K) {
                minHeap.poll();
            }
        }

        int[] res = new int[minHeap.size()];
        
for (int i = res.length - 1; i >= 0; --i) {
            res[i] = minHeap.poll();
        }

        return res;
    }
}
