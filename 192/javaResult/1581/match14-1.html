class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int median = arr[(arr.length - 1) / 2];
        
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> {
            int w1 = Math.abs(a - median), w2 = Math.abs(b - median);
            if (w1 != w2) return Integer.compare(w1, w2);
            return Integer.compare(a, b);
        
});
        
        for (int x: arr) {
            pq.offer(x);
            if (pq.size() > k) {
                pq.poll();
            }
        }
        
        
        int[] res = new int[k];
        int ind = k - 1;
        while (!pq.isEmpty()) {
            res[ind--] = pq.poll();
        }
        
        return res;
    }
}
