class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int b[] =arr.clone();
        
Arrays.sort(b);
        int l = b.length;
        int m  = b[(l-1)/2];
        
         Integer v[] = new Integer[arr.length];
        
         for(int i=0;i<arr.length;++i){
             v[i] = arr[i];
         }
        
        Arrays.sort(v,(
x,y)->{
            int v1 = Math.abs(x-m);
            int v2 = Math.abs(y-m);
            if(v1!=v2){
                return v2-v1;
            }else{
                return y-x;
            }
            
        });
        
        
int r[] = new int[k];
        for(int i=0;i<k;++i){
            r[i] = v[i];
        }
        
        return r;
        
        
    }
}
