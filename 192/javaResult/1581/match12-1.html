class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Arrays.sort(arr);
        int m = arr[(n - 1) / 2];
        int[] res = new int[k];
        int l = 0, r = n - 1;
        for (int i = 0; i < k; i++) 
{
            if (stronger(arr[l], arr[r], m)) {
                res[i] = arr[l++];
            } else {
                res[i] = arr[r--];
            }
        }
        
return res;
    }

    private boolean stronger(int lhs, int rhs, int m) {
        int absl = Math.abs(lhs - m);
        int absr = Math.abs(rhs - m);
        return absl != absr ? absl > absr : lhs > rhs;
    }
}
