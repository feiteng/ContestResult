class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int m = arr[(n - 1) / 2];
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> {
            int left = Math.abs(arr[a] - m);
            int right = Math.abs(arr[b] - m);
            if (left == right) {
                return Integer.compare(arr[a], arr[b]);
            }
            return Integer.compare(left, right);
        
});
        for (int i = 0; i < arr.length; i++) {
            pq.offer(i);
            if (pq.size() > k) {
                pq.remove();
            }
        }
        int[] res = new int[k];
        
for (int i = 0; i < k; i++) {
            res[i] = arr[pq.remove()];
        }
        return res;
    }
}
