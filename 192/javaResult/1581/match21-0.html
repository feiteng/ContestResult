class Solution {
   public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;

        int mid = arr[(n - 1) / 2];
        Integer[] a = new Integer[n];
        for (int i = 0; i < n; i++) {
            a[i] = arr[i];
        }
        Arrays.sort(a, new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                int diff1 = Math.abs(o1 - mid);
                int diff2 = Math.abs(o2 - mid);
                if (diff1 == diff2) {
                    return o1 - o2;
                } else {
                    return diff1 - diff2;
                }
            }
        });
        int[] r = new int[k];
        for (int i = 0; i < k; i++) {
            r[i] = a[n -1- i];
        }
        return r;
    }
}
