class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Arrays.sort(arr);
        double median = arr[(n - 1) / 2];
        int[] res = new int[k];
        int left = 0, right = n - 1;
        for (int i = 0; i < k; ++i) {
            if (Math.abs(arr[right] - median) >= Math.abs(arr[left] - median)) 
{
                res[i] = arr[right--];
            } else {
                res[i] = arr[left++];
            }
        }
        return res;
    }
}
