import java.util.Arrays;
import java.util.PriorityQueue;

class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int n = arr.length;
        int m = arr[(n-1)/2];

        PriorityQueue<Integer> pq = new PriorityQueue<>((x, y)-> {
            int tmp = Math.abs(y-m) - Math.abs(x-m);
            return tmp != 0 ? tmp : y-x;
        });
        for (int a : arr) {
            pq.add(a);
        }

        int ret[] = new int[k];
        for (int i=0; i<k; ++i) {
            ret[i] = pq.poll();
        }

        return ret;

    }
}
