class Solution {
        public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int ans[] = new int[k];
        int j = 0;
        int l = 0;
        int r = arr.length - 1;
        int m = arr[(( arr.length - 1) / 2)];
        // if (arr.length % 2 == 0) {
        //     m = (arr[arr.length / 2 - 1] + arr[arr.length / 2]) / 2;
        // } else {
        //     m = arr[arr.length / 2];
        // }
           // System.out.println(m);
        while (j < k) 
{
            int x = Math.abs(arr[l] - m);
            int y = Math.abs(arr[r] - m);
            //System.out.println("pos: "+x+" "+y+" "+arr[l]+" "+arr[r]);
            if(y >= x) {
                ans[j++] = arr[r];
                r--;
            } else {
                ans[j++] = arr[l];
                l++;
            }
        }
       // for(int i=0; i < k; i++)
      //   System.out.println(arr[i]);    
        return ans;
    }
}
