class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int n = arr.length;
        Integer[] sort = new Integer[n];
        for (int i = 0; i < n; i++) sort[i] = arr[i]
;
        Arrays.sort(arr);
        int mid = arr[(n - 1) / 2];
        Arrays.sort(sort, (i1, i2) -> {
            int d1 = Math.abs(i1 - mid);
            int d2 = Math.abs(i2 - mid);
            if (d1 == d2) return Integer.compare(i2, i1);
            return Integer.compare(d2, d1);

        
});
        int[] ans = new int[k];
        for (int i = 0; i < k; i++) {
            ans[i] = sort[i];
        }
        return ans;
        
    }
}
