class Solution {
    public int[] getStrongest(int[] arr, int k) {
        int len = arr.length;
        Arrays.sort(arr);
        
int m = arr[(len - 1) / 2];
        int l = 0;
        int r = len - 1;
        int count = 0;
        int[] result = new int[k];
        while (count < k) {
            if (Math.abs(arr[r] - m) >= Math.abs(arr[l] - m)) {
                result[count] = arr[r];
                r--;
            } else {
                result[count] = arr[l];
                l++;
            }
            count++;
        }
        return result;
    }
}
