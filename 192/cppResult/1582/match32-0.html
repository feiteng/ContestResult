// class
class BrowserHistory {
public:
    vector<string> a;
    int cur = 0;
    BrowserHistory(string homepage) {
        a = {homepage};
        cur = 0;
    }

    void visit(string url) {
        while(cur+1 < a.size()) a.pop_back();
        a.emplace_back(url);
        cur++;
    }

    string back(int steps) {
        cur = max(0, cur-steps);
        return a[cur];
    }

    string forward(int steps) {
        cur = min((int)a.size()-1, cur+steps);
        return a[cur];
    }
};
