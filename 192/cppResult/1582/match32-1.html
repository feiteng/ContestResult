class BrowserHistory {
public:
    vector<string> history;
    int current = 0;
    BrowserHistory(string homepage) {
        history = {homepage};
        current = 0;
    }
    
    void visit(string url) {
        history.erase(history.begin() + current + 1, history.end());
        history.push_back(url);

        current = history.size() - 1;
        // cout << "history:" << endl;
        // for (const string &s: history) {
        //     cout << "    "<< s << endl;
        // }
    }
    
    string back(int steps) {
        // cout << "old currrent:" << current;
        current = max(0, current - steps);
        // cout << ", new currrent:" << current << endl;
        return history[current];
    }
    
    string forward(int steps) {
        current = min((int)history.size() - 1, current + steps);
        return history[current];
    }
};

/**
 * Your BrowserHistory object will be instantiated and called as such:
 * BrowserHistory* obj = new BrowserHistory(homepage);
 * obj->visit(url);
 * string param_2 = obj->back(steps);
 * string param_3 = obj->forward(steps);
 */
