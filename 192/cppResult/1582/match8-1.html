class BrowserHistory {
public:
    BrowserHistory(string homepage) {
        cp.push_back(homepage);
        ptr = 0;
    }
    
    void visit(string url) {
        cp.erase(cp.begin()+ptr+1,cp.end());
        cp.push_back(url);
        ptr++;
    }
    
    string back(int steps) {
        ptr = max(0,ptr-steps);
        return cp[ptr];
    }
    
    string forward(int steps) {
        ptr = min((int)cp.size()-1,ptr+steps);
        return cp[ptr];
    }
private:
    vector<string>cp;
    int ptr;
};
