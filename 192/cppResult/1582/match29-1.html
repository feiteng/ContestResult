class BrowserHistory {
public:
    vector<string> lq;
    int ind;
    BrowserHistory(string homepage) {
        lq.push_back(homepage);
        ind=0;
    }
    
    void visit(string url) {
        lq.erase(lq.begin()+ind+1,lq.end());
        lq.push_back(url);
        ++ind;
    }
    
    string back(int steps) {
        int sp=min(steps,ind);
        ind-=sp;
        return lq[ind];
    }
    
    string forward(int steps) {
        int sp=min(steps,(int)lq.size()-1-ind);
        ind+=sp;
        return lq[ind];
    }
};

 * BrowserHistory* obj = new BrowserHistory(homepage);
 * obj->visit(url);
