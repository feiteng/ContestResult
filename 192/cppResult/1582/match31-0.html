class BrowserHistory {
public:
    string now;
    stack<string> hist;
    stack<string> fhist;
    BrowserHistory(string homepage) {
        now = homepage;
    }
    
    void visit(string url) {
        fhist = stack<string>();
        hist.push(now);
        now = url;
    }
    
    string back(int steps) {
        while(sz(hist)>0 && steps--){
            fhist.push(now);
            now = hist.top();hist.pop();
        }
        return now;
    }
    
    string forward(int steps) {
        while(sz(fhist)>0 && steps--){
            hist.push(now);
            now = fhist.top();fhist.pop();
        }
        return now;
    }
};
