class Solution {
public:
    int dp[105][25][105];//第i个，涂j色，前i个共有k个街区
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp,0x3f,sizeof(dp));
        if(houses[0]==0){
            for(int i=0;i<n;i++){
                dp[0][i][1] = cost[0][i];
            }
        }else{
             dp[0][houses[0]-1][1] = 0;
        }
        for(int i=1;i<m;i++){
            if(houses[i]!=0){
                int col = houses[i]-1;
                for(int k=1;k<=i+1;k++){
                    int mi = 0x3f3f3f3f;
                    for(int j=0;j<n;j++){
                        if(col==j){ //颜色相同
                            mi = min(mi,dp[i-1][j][k]);
                        }else{ //颜色不同
                            mi = min(mi,dp[i-1][j][k-1]);
                        }
                    }
                    dp[i][col][k] = mi;
                    //cout<<i<<" "<<col<<" "<<k<<" "<<dp[i][col][k]<<endl;
                }
            }else{
                for(int k=1;k<=i+1;k++){
                    for(int j=0;j<n;j++){
                        int mi = 0x3f3f3f3f;
                        for(int l=0;l<n;l++){
                            if(j==l){ //颜色相同
                                 mi = min(mi,dp[i-1][l][k]+cost[i][j]);
                            }else{    
                                mi = min(mi,dp[i-1][l][k-1]+cost[i][j]);
                            }
                        }
                        dp[i][j][k]=mi;
                      //  cout<<i<<" "<<j<<" "<<k<<" "<<dp[i][j][k]<<endl;
                    }
                }
            }
        }
        int res=0x3f3f3f3f;
        for(int i=0;i<n;i++){
            
           // cout<<dp[m-1][i][target]<<endl;
            res = min(res,dp[m-1][i][target]);
        }
        return res==0x3f3f3f3f ? -1:res;
    }
};
