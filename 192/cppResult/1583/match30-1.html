class Solution {
public:
    int dp[110][22][110];
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp,0x3f3f3f3f,sizeof(dp));
        dp[0][0][0]=0;

        for(int hi=1;hi<=m;hi++){
            if(houses[hi-1]==0){
                for(int ni=1;ni<=n;ni++){
                    //
                     for(int pni=0;pni<=n;pni++){
                        for(int pki=0;pki<=target;pki++){
                            int ki=pki+(ni!=pni);

                            dp[hi][ni][ki]=min(dp[hi][ni][ki], dp[hi-1][pni][pki]+cost[hi-1][ni-1]);
                        }
                    }
                }
            }else{
                for(int pni=0;pni<=n;pni++){
                    for(int pki=0;pki<=target;pki++){

                        int ki=pki+(houses[hi-1]!=pni);
                        dp[hi][houses[hi-1]][ki]=min(dp[hi][houses[hi-1]][ki], dp[hi-1][pni][pki]);
                    }
                }
            }
        }
        
        // for(int hi=0;hi<=m;hi++){
        //     cout<<endl<<endl<<endl;
        //     for(int ni=0;ni<=n;ni++){
        //         for(int pki=0;pki<=target;pki++){
        //             cout<<dp[hi][ni][pki]<<" ";
        //         }
        //         cout<<endl;
        //     }   
        // }
        
    
        int ans=0x3f3f3f3f;
        for(int ni=1;ni<=n;ni++){
            ans=min(ans,dp[m][ni][target]);
        }
        if(ans==0x3f3f3f3f)return -1;
        return ans;
    }
};
