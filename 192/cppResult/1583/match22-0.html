class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector< vector< vector< int > > > dp( m + 1, vector< vector< int > >( m + 1, vector< int >( n + 1, 2000000000 ) ) );
        for( int i = 1; i <= n; i ++ )
        {
            if( houses[ 0 ] == 0 ) dp[ 1 ][ 1 ][ i ] = cost[ 0 ][ i - 1 ];
            else                   dp[ 1 ][ 1 ][ houses[ 0 ] ] = 0;
        }
        
        for( int i = 2; i <= m; i ++ )
        {
            for( int j = 1; j <= min( i, target ); j ++ )
            {
                if( houses[ i - 1 ] != 0 )
                {
                    for( int x = 1; x <= n; x ++ )
                    {
                        if( houses[ i - 1 ] == x ) dp[ i ][ j ][ houses[ i - 1 ] ] = min( dp[ i ][ j ][ houses[ i - 1 ] ], dp[ i - 1 ][ j ][ houses[ i - 1 ] ] );
                        else
                            dp[ i ][ j ][ houses[ i - 1 ] ] = min( dp[ i ][ j ][ houses[ i - 1 ] ], dp[ i - 1 ][ j - 1 ][ x ] );
                    }
                    continue;
                }
                for( int k = 1; k <= n; k ++ )
                {
                    for( int x = 1; x <= n; x ++ )
                    {
                        if( x == k )
                            dp[ i ][ j ][ k ] = min( dp[ i ][ j ][ k ], dp[ i - 1 ][ j ][ k ] + cost[ i - 1 ][ k - 1 ] );
                        else
                            dp[ i ][ j ][ k ] = min( dp[ i ][ j ][ k ], dp[ i - 1 ][ j - 1 ][ x ] + cost[ i - 1 ][ k - 1 ] );
                    }
                }
            }
        }
        int goal = -1;
        for( int i = 1; i <= n; i ++ )
        {
            if( dp[ m ][ target ][ i ] >= 2000000000 ) continue;
            if( goal == -1 ) goal = dp[ m ][ target ][ i ];
            else goal = min( goal, dp[ m ][ target ][ i ] ); 
        }
        return goal;
    }
};
