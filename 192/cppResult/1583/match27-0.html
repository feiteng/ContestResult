class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m, vector<vector<int>>(target+1, vector<int>(n+1, INT_MAX)));
        
        if (houses[0] > 0) dp[0][1][houses[0]] = 0;
        else {
            for (int k = 1; k <= n; ++k) {
                dp[0][1][k] = cost[0][k-1];
                //cout << k << "," << dp[0][1][k] << endl;
            }
        }
        
        
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j <= target; ++j) {
                for (int k = 1; k <= n; ++k) {
                    if (houses[i] > 0) {
                        if (k != houses[i]) continue;
                    }
                    int c = (houses[i] == k ? 0 : cost[i][k-1]);
                    //cout << "cost, i : " <<  i << "k:" << k << " cost "<<cost[i][k-1] << endl;
                    dp[i][j][k] = (dp[i-1][j][k] < INT_MAX ? (dp[i-1][j][k] + c) : INT_MAX);
                    for (int l = 1; l <= n; ++l) {
                        if (l == k || j < 2) continue;
                        dp[i][j][k] = min(dp[i][j][k], dp[i-1][j-1][l] < INT_MAX ? dp[i-1][j-1][l] + c : INT_MAX);
                    }
                }
            }
        }
        
        int res = INT_MAX;
        for (int k = 1; k <= n; ++k) {
            //cout << dp[m-1][target][k] << endl;
            res = min(res, dp[m-1][target][k]);
        }
        if (res < INT_MAX) return res;
        return -1;
    }
};
