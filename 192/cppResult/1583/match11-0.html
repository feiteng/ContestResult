class Solution {
public:
    int f[105][105][25];
    int minCost(vector<int>& a, vector<vector<int>>& w, int m, int n, int k) {
        memset(f, 0x3f, sizeof(f));
        for(int i = 1; i <= n; i++){
            if(!a[0]) f[0][1][i] = w[0][i-1];
        }
        if(a[0]) f[0][1][a[0]] = 0;
        for(int i = 1; i < m; i++){
            for(int j = 1; j <= min(i+1, k); j++){
                if(a[i]){
                    f[i][j][a[i]] = f[i-1][j][a[i]];
                    for(int l = 1; l <= n; l++){
                        if(l == a[i]) continue;
                        f[i][j][a[i]] = min(f[i][j][a[i]], f[i-1][j-1][l]);
                    }
                    // cout << i << " " << j << " " << f[i][j][a[i]] << endl;
                }
                else{
                    for(int c = 1; c <= n; c++){
                        for(int l = 1; l <= n; l++){
                            if(l == c) f[i][j][c] = min(f[i][j][c], f[i-1][j][l]+w[i][c-1]);
                            else f[i][j][c] = min(f[i][j][c], f[i-1][j-1][l]+w[i][c-1]);
                        }
                        // cout << i << " " << j << " " << c << " " << f[i][j][c] << endl;
                    }
                }
            }
        }
        int ans = 0x3f3f3f3f;
        for(int i = 1; i <= n; i++){
            ans = min(ans, f[m-1][k][i]);
        }
        if(ans == 0x3f3f3f3f) return -1;
        return ans;
    }
};
