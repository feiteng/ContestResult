class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m+1, vector<vector<int>>(n+1, vector<int>(target + 1, 1e9+7)));
        dp[0][0][0] = 0;

        for(int j=0;j<=n;j++)dp[0][j][0] = 0;
        for(int i=1;i<=m;i++)
        {
            int org = houses[i-1];
            if(org == 0)
            {
                for(int j=1;j<=n;j++)
                {
                    int cur = cost[i-1][j-1];

                    for(int t=1;t<=target;t++)
                    {
                        dp[i][j][t] = dp[i-1][j][t] + cur;
                        for(int k=1;k<=n;k++)
                        {
                            if(j!=k)
                            {
                                dp[i][j][t] = min(dp[i][j][t], dp[i-1][k][t-1] + cur);
                            }
                        }
                       // cout<<i<<" "<<j<<" "<<t<<" "<<dp[i][j][t]<<endl;
                    }
                }
            }
            else
            {

                for(int t=1;t<=target;t++)
                {
                    dp[i][org][t] = dp[i-1][org][t];
                    for(int k=1;k<=n;k++)
                    {
                        if(org!=k)
                        {
                            dp[i][org][t] = min(dp[i][org][t], dp[i-1][k][t-1]);
                        }

                    }
                }
                
            }

        }
        int res = 1e9+7;
        for(int j=1;j<=n;j++)
        {
            res = min(res, dp[m][j][target]);
        }
        if(res >= 1e9+7) return -1;
        return res;
    }
};
