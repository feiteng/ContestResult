class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m+1, vector<vector<int>>(n+1, vector<int>(target+1, INT_MAX/2)));
        for (int j = 0; j <= n; j++)
            dp[0][j][0] = 0;

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                for (int k = 1; k <= target; k++) {
                    if (houses[i-1] == j) {

                        dp[i][j][k] = min(dp[i][j][k], dp[i-1][j][k]);
                        for (int jj = 1; jj <= n; jj++) {
                            if (j == jj)
                                continue;
                            dp[i][j][k] = min(dp[i][j][k], dp[i-1][jj][k-1]);
                        }
                    } else if (houses[i-1] == 0) {
                        for (int jj = 1; jj <= n; jj++) {
                            if (j == jj) {
                                dp[i][j][k] = min(dp[i][j][k], dp[i-1][jj][k]);
                            } else {
                                dp[i][j][k] = min(dp[i][j][k], dp[i-1][jj][k-1]);
                            }
                        }
                        dp[i][j][k] += cost[i-1][j-1];

                    } 
                }
            }
            /*
            for (int j = 1; j <= n; j++) {
                for (int k = 1; k <= target; k++) {
                    cout <<dp[i][j][k] << " ";
                }
                cout << endl;
            }
            cout << "---" << endl;
            */
        }
        int ans = INT_MAX;
        for (int j = 1; j <= n; j++) {
            ans = min(ans, dp[m][j][target]);
        }
        if (ans > 1e6)
            return -1;
        return ans;
        
    }
};
