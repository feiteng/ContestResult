class Solution {
public:
    int dp[105][25][105],INF=0x3f3f3f3f;//第i个房子，颜色是j，一个k个街区
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp,0x3f,sizeof(dp));
        if (houses[0]==0) {
            for (int j=1; j<=n; j++) {
                dp[1][j][1]=cost[0][j-1];
            }
        } else {
            dp[1][houses[0]][1]=0;

        }
        for (int i=2; i<=m; i++) {  //100
            if (houses[i-1]==0) {
                for (int j=1; j<=n; j++) { //20,上一个房子的颜色
                    for (int k=1; k<=n; k++)  { //20
                        for (int c=1; c<=target; c++) {
                            if (dp[i-1][j][c]==INF) continue;

                            if (k==j) dp[i][k][c]=min(dp[i][k][c],dp[i-1][j][c]+cost[i-1][k-1]);
                            else dp[i][k][c+1]=min(dp[i][k][c+1],dp[i-1][j][c]+cost[i-1][k-1]);
                        }
                    }
                }
            } else {
                for (int j=1; j<=n; j++) {
                    for (int c=1; c<=target; c++) {
                        if (dp[i-1][j][c]==INF) continue;

                        if (houses[i-1]==j) dp[i][houses[i-1]][c]=min(dp[i][houses[i-1]][c],dp[i-1][j][c]);
                        else dp[i][houses[i-1]][c+1]=min(dp[i][houses[i-1]][c+1],dp[i-1][j][c]);
                    }
                }
            }
        }
        int res=INF;
        for (int i=1; i<=n; i++) res=min(res,dp[m][i][target]);
        if (res==INF) res=-1;
        return res;
    }
};
