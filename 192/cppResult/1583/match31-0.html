int min(int a, int b)
{
    if (a == -1) return b;
    if (b == -1) return a;
    if (a < b) return a;
    return b;
}
class Solution {
public:
    int dp[102][102][22];
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < 102; ++i)
        {
            for (int j = 0; j < 102; ++j)
            {
                for (int k = 0; k < 22; ++k)
                {
                    dp[i][j][k] = -1;
                }
            }
        }
        dp[0][0][0] = 0;
        
        for (int i = 0; i < m; ++i)
        {
            if (houses[i] != 0)
            {
                for (int j = 0; j <= target; ++j)
                {
                    for (int k = 0; k <= n; ++k)
                    {
                        if (dp[i][j][k] >= 0)
                        {
                            if (k == houses[i]) {
                                dp[i + 1][j][houses[i]] = min(dp[i + 1][j][houses[i]], dp[i][j][k]);
                            }
                            else{
                                if (j + 1 <= target){
                                    dp[i + 1][j + 1][houses[i]] = min(dp[i + 1][j + 1][houses[i]], dp[i][j][k]);
                                }
                            }
                                
                        }
                    }
                }
            }
            else
            {
                for (int j = 0; j <= target; ++j)
                {
                    for (int k = 0; k <= n; ++k)
                    {
                        if (dp[i][j][k] >= 0)
                        {
                            for (int z = 0; z < n; ++z)
                            {
                                if (k == z + 1) {
                                    dp[i + 1][j][z + 1] = min(dp[i + 1][j][z + 1], dp[i][j][k] + cost[i][z]);
                                }
                                else{
                                    if (j + 1 <= target){
                                        dp[i + 1][j + 1][z + 1] = min(dp[i + 1][j + 1][z + 1], dp[i][j][k] + cost[i][z]);
                                    }
                                }
                            }   
                        }
                    }
                }
            }
        }
        // for (int i = 0; i <= m; ++i)
        // {
        //     for (int j = 0; j <= target; ++j)
        //     {
        //         for (int k = 0; k <= n; ++k)
        //         {
        //             cout << dp[i][j][k] << ' ';
        //         }
        //         cout << endl;
        //     }
        //     cout << endl;
        // }
        int ans = -1;
        for (int k = 1; k <= n; ++k)
        {
            ans = min(ans, dp[m][target][k]);
        }
        return ans;
    }
};
