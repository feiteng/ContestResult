class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m + 1, vector<vector<int>>(n + 1 , vector<int>(target + 1, INT_MAX)));
        for(int j = 1; j <=n; ++j) {
            dp[0][j][0] = 0;
        }
        
        for(int i = 1; i <=m; ++i) {
            for(int j = 1; j <=n; ++j) {
                for(int k = 1; k <= target; k++) {
                    
                    if(houses[i-1] != 0) {
                        if(houses[i-1] != j) continue;
                        dp[i][j][k] = dp[i-1][j][k];
                        for(int x = 1; x <=n; ++x) {
                            if(x==j) continue;
                            dp[i][j][k] = min(dp[i][j][k], dp[i-1][x][k-1]);
                        }
                    } else {
                        if(dp[i-1][j][k] != INT_MAX) {
                            dp[i][j][k] = dp[i-1][j][k] + cost[i-1][j-1]; 
                        } 
                        for(int x = 1; x <=n ; ++x) {
                            if(x==j) continue;
                            if(dp[i-1][x][k-1] == INT_MAX) continue;
                            dp[i][j][k] = min(dp[i][j][k], dp[i-1][x][k-1] + cost[i-1][j-1]);
                         //   cout << "trace2 " << dp[i][j][k] <<endl;
                        }
                    }
                    
                   // cout << i << "->" << j << "->" << k << ": " << dp[i][j][k] <<endl;
                }
            }
        }
        
        int res = INT_MAX;
        for(int j = 1; j <=n; ++j) {
            res = min(res, dp[m][j][target]);
        }
        
        return res == INT_MAX ? - 1 : res;
        
    }
};
