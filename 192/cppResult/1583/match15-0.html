class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {   
        int dp[105][25][105];
        memset(dp,0x7f7f7f7f,sizeof dp);
        if(houses[0]){
            dp[0][houses[0]][1]=0;
        }else{
            for(int i=1;i<=n;i++){
                dp[0][i][1]=cost[0][i-1];
            }
        }
        for(int i=1;i<m;i++){
            if(houses[i]){
                for(int k=0;k<=m;k++){
                    for(int j=1;j<=n;j++){
                        if(dp[i-1][j][k]==0x7f7f7f7f) continue;
                        if(j==houses[i]){
                            dp[i][houses[i]][k]=min(dp[i][houses[i]][k],dp[i-1][j][k]);
                        }else{
                            dp[i][houses[i]][k+1]=min(dp[i][houses[i]][k+1],dp[i-1][j][k]);
                        }
                    }
                }
            }else{
                for(int k=0;k<=m;k++){
                    for(int j=1;j<=n;j++){
                        if(dp[i-1][j][k]==0x7f7f7f7f) continue;
                        for(int p=1;p<=n;p++){
                            if(p==j){
                                dp[i][p][k]=min(dp[i][p][k],dp[i-1][j][k]+cost[i][p-1]);
                            }else{
                                dp[i][p][k+1]=min(dp[i][p][k+1],dp[i-1][j][k]+cost[i][p-1]);
                            }
                        }
                    }
                }
            }
        }
        int ans=0x7f7f7f7f;
        for(int i=1;i<=n;i++){
            ans=min(dp[m-1][i][target],ans);
        }
        return ans==0x7f7f7f7f?-1:ans;
    }
};
