class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int t) {
        int MM=1e9;
        vector<vector<vector<int>>> dp(m+1,vector<vector<int>>(n+1,vector<int>(t+1,MM)));

        for(int i=0; i<=n; i++) dp[0][i][0]=0;
        for(int i=1;  i<=m; i++) {
            if(houses[i-1]==0){
                for(int j=1; j<=n; j++) {
                    for(int k=1; k<=t; k++) {
                        for(int a=1; a<=n; a++) {

                            dp[i][j][k]=min(dp[i][j][k],dp[i-1][a][k-(j!=a)]+cost[i-1][j-1]);
                        }
                    }
                }
            } else {
                int j=houses[i-1];
                for(int k=1; k<=t; k++) {
                    for(int a=1; a<=n; a++) {

                        dp[i][j][k]=min(dp[i][j][k],dp[i-1][a][k-(j!=a)]);
                    }
                }
            }
        }
        
        int ans=MM;
        for(int i=1; i<=n; i++){
            ans=min(ans,dp[m][i][t]);
        }
        if(ans==MM) ans=-1;
        return ans;
    }
};
