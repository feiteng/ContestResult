class Solution {
public:
    int f[105][22][105];
    

    int minCost(vector<int> &a, vector<vector<int>>& cost, int m, int n, int t) {
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                for (int k = 0; k <= t; k++) {
                    f[i][j][k] = 1e9;
                }
            }
        }
        for (int j = 0; j < n; j++) {
            int c = a[0] == 0 ? cost[0][j] : 0;
            if (a[0] != 0 && a[0] != j + 1) {
                continue;
            }
            f[0][j][1] = c;
        }

        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (a[i] != 0 && a[i] != j + 1) {
                    continue;
                }
                int c = a[i] == 0 ? cost[i][j] : 0;

                for (int k = 1; k <= t; k++) {
                    for (int l = 0; l < n; l++) {
                        if (l == j) {
                            f[i][j][k] = min(f[i][j][k], f[i - 1][l][k] + c);
                        } else {
                            f[i][j][k] = min(f[i][j][k], f[i - 1][l][k - 1] + c);
                        }
                    }

                }
            }
        }
        int ans = 1e9;
        for (int j = 0; j < n; j++) {
            ans = min(ans, f[m - 1][j][t]);
        }
        if (ans > 1e9 - 1) {
            return -1;
        }
        return ans;
    }
};
