int dp[101][101][21];
class Solution {
public:

    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp,1,sizeof(dp));
        for(int i=0;i<=n;i++)
        {
            dp[0][0][i]=0;
        }

        for(int i=1;i<=m;i++)
        {
            for(int j=1;j<=min(i,target);j++)
            {
                if(houses[i-1])
                {
                    for(int k=0;k<=n;k++)
                    {
                        if(k==houses[i-1])dp[i][j][houses[i-1]]=min(dp[i][j][houses[i-1]],dp[i-1][j][k]);
                        else dp[i][j][houses[i-1]]=min(dp[i][j][houses[i-1]],dp[i-1][j-1][k]);
                    }
                }
                else

                {
                    for(int k=0;k<=n;k++)
                    {
                        for(int l=1;l<=n;l++)
                        {
                            if(k==l)dp[i][j][l]=min(dp[i][j][l],dp[i-1][j][k]+cost[i-1][l-1]);
                            else dp[i][j][l]=min(dp[i][j][l],dp[i-1][j-1][k]+cost[i-1][l-1]);

                        }
                    }
                }
            }
        }
        int ans=999999999;
        for(int i=1;i<=n;i++)
        {
            ans=min(ans,dp[m][target][i]);
        }
        if(ans>100000)return -1;
        return ans;
    }
};
