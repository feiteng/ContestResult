class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> f(m+1, vector<vector<int>>(n+1, vector<int>(target+1, INT_MAX / 2)));
        for (int i=0; i<=n; ++i) 
            f[0][i][0] = 0;
        for (int i=1; i<=m; ++i) {
            int s = houses[i-1] == 0 ? 0 : houses[i-1]-1;
            int t = houses[i-1] == 0 ? n : houses[i-1];
            for (int j=s; j<t; ++j) {
                auto c = houses[i-1] != 0 ? 0 : cost[i-1][j];
                  
                for (int k=1; k<=target; ++k) {
                    f[i][j][k] = min(f[i][j][k], f[i-1][j][k] + c);
                    for (int q=0; q<=n; ++q) {
                        if (q == j) continue;
                        f[i][j][k] = min(f[i][j][k], f[i-1][q][k-1] + c);
                    }
                }
            }
        }
        int ans = INT_MAX / 2;
        for (int i=0; i<=n; ++i) {
            ans = min(ans, f[m][i][target]);
        }
        return ans == INT_MAX / 2 ? -1 : ans;
    }
};
