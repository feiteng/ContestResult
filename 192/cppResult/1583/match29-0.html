#define VI vector<int>
#define VVI vector<VI>
#define VVVI vector<VVI>
#define LL uint64_t
const int MOD=1e9+7;
class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        VVVI dp(m,VVI(target+1,VI(n,INT32_MAX/2)));
        if(houses[0]==0){
            for(int i=1;i<=n;i++){
                dp[0][0][i-1]=cost[0][i-1];
            }
        } else
            dp[0][0][houses[0]-1]=0;
        for(int i=1;i<m;i++){
            if(houses[i]==0){
                for(int c=0;c<n;c++){
                    for(int lastt=0;lastt<target;lastt++){
                        for(int lastc=0;lastc<n;lastc++){
                            if(c==lastc)
                                dp[i][lastt][c]=min(dp[i][lastt][c],dp[i-1][lastt][lastc]+cost[i][c]);
                            else
                                dp[i][lastt+1][c]=min(dp[i][lastt+1][c],dp[i-1][lastt][lastc]+cost[i][c]);
                        }
                    }
                }
            }else{
                int c=houses[i]-1;
                for(int lastt=0;lastt<target;lastt++){
                    for(int lastc=0;lastc<n;lastc++){
                        if(c==lastc)
                            dp[i][lastt][c]=min(dp[i][lastt][c],dp[i-1][lastt][lastc]);
                        else
                            dp[i][lastt+1][c]=min(dp[i][lastt+1][c],dp[i-1][lastt][lastc]);
                    }
                }


            }

        }
        int ans=INT32_MAX;
        for(int c=0;c<n;c++){
            ans=min(ans,dp[m-1][target-1][c]);
        }
        if(ans>=INT32_MAX/2)
            return -1;
        return ans;
    }
};
