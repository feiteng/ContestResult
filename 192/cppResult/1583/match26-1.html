class Solution {
private:
    int dp[110][30][110];
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {

        for(int i = 0; i <= m; i++) {
            for(int j = 0; j <= n; j++) {
                for(int k = 0; k <= m; k++) {
                    dp[i][j][k] = 1000000000;
                }
            }
        }
        
        if(houses[0] == 0) {
            for(int i = 0; i < n; i++) {
                dp[0][i+1][1] = cost[0][i];
            }
        }
        else {
            dp[0][houses[0]][1] = 0;
        }
        
        for(int i = 1; i < m; i++) {
            for(int j = 0; j < n; j++) {
                for(int k = 1; k <= target && k <= i; k++) {
                    if(houses[i] != 0) {
                        dp[i][houses[i]][k + (houses[i] != j+1)] = min(dp[i][houses[i]][k + (houses[i] != j+1)], dp[i-1][j+1][k]);
                    }
                    else {
                        for(int q = 0; q < n; q++) {

                            dp[i][j+1][k + (j != q)] = min(dp[i][j+1][k + (j != q)], dp[i-1][q+1][k] + cost[i][j]);
                        }
                    }
                }
            }
        }
        
        int ans = 1000000000;
        for(int i = 0; i < n; i++) {
            ans = min(ans, dp[m-1][i+1][target]);
        }
        return ans == 1000000000 ? -1 : ans;
    }
};
