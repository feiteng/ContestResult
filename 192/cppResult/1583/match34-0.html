class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        int dp[m][n + 1][target + 2];
        memset(dp, 0x3f, sizeof(dp));
        
        if (houses[0]) {
            dp[0][houses[0]][1] = 0;
        } else {
            for (int i = 1; i <= n; i++) {
                dp[0][i][1] = cost[0][i - 1];
            }
        }
        
        for (int i = 1; i < m ; i++) {
            for (int j = 1; j <= n; j++) {
                for (int k = 1; k <= target; k++) {
                    if (dp[i - 1][j][k] != 0x3f3f3f3f) {
                        if (houses[i]) {
                            dp[i][houses[i]][k + (houses[i] != j)] = min(dp[i][houses[i]][k + (houses[i] != j)], dp[i - 1][j][k]);
                        } else {
                            for (int l = 1; l <= n; l++) {
                                dp[i][l][k + (l != j)] = min(dp[i][l][k + (l != j)], dp[i - 1][j][k] + cost[i][l - 1]);
                            }
                        }
                    }
                }
            }
        }
        int ans = numeric_limits<int>::max();
        for (int i = 1; i <= n; i++) {
            ans = min(ans, dp[m - 1][i][target]);
        }
        return ans == 0x3f3f3f3f ? -1 : ans;
    }
};
