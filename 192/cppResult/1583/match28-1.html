template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  int dp[111][111][22];
  int minCost(vector<int>& h, vector<vector<int>>& cost, int m, int n, int tar) {
    memset(dp,63,sizeof dp);
    int M=h.size();
    int N=cost[0].size();

    for (int c=0;c<N;c++) dp[0][0][c]=0;
    
    for (int i=0;i<M;i++) {
      for (int j=1;j<=tar;j++) {
        if (h[i]==0) {

          for (int c=0;c<N;c++) {
            for (int d=0;d<=N;d++) {
              if (c==d) {
                dp[i+1][j][c]=min(dp[i+1][j][c],dp[i][j][c]+cost[i][c]);
              } else {
                dp[i+1][j][c]=min(dp[i+1][j][c],dp[i][j-1][d]+cost[i][c]);
              }
            }
          }

        } else {
          int c=h[i]-1;
          for (int d=0;d<=N;d++) {
            if (c==d) {
              dp[i+1][j][c]=min(dp[i+1][j][c],dp[i][j][c]);
            } else {
              dp[i+1][j][c]=min(dp[i+1][j][c],dp[i][j-1][d]);
            }
          }
          
        }

      }
    }
    int res=0x3f3f3f3f;
    for (int c=0;c<N;c++) res=min(res, dp[M][tar][c]);
    if (res==0x3f3f3f3f) res = -1;
    return res;
  }
};
