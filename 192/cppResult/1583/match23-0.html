class Solution {
    int f[101][101][21];
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        for (int i = 0; i < 101; ++i) {
            for (int j = 0; j < 101; ++j) {
                for (int k = 0; k < 21; ++k)
                f[i][j][k] = 1e9;
            }
        }
        f[0][0][0] = 0;
        for (int i = 1; i <= m; ++i) {
            for (int j = 1; j <= target; ++j) {
                if (houses[i - 1] != 0) {
                    f[i][j][houses[i - 1]] = f[i-1][j][houses[i-1]];
                    for (int k = 0; k <= n; ++k) {
                        if (k == houses[i - 1]) {
                            continue;
                        }
                        f[i][j][houses[i - 1]] = min(f[i][j][houses[i - 1]], f[i - 1][j - 1][k]);
                    }
                    // f[i][j][houses[i-1]] += cost[i-1][houses[i-1] - 1];
                } else {
                    // color
                    for (int k = 1; k <= n; ++k) {
                        for (int l = 0; l <= n; ++l) {
                            if (k == l) {
                                f[i][j][k] = min(f[i][j][k], f[i-1][j][k]);
                            } else {
                                f[i][j][k] = min(f[i][j][k], f[i-1][j-1][l]);
                            }
                        }
                        f[i][j][k] += cost[i - 1][k - 1];
                    }                        
                }
            }
        }
        int ans = 1e9;
        for (int color = 1; color <= n; ++color) {
            ans = min(ans, f[m][target][color]);
        }
        if (ans == 1e9) {
            ans = -1;
        }
        // for (int i = 1; i <= m; ++i) {
        //     for (int j = 1; j <= target; ++j) {
        //         for (int k = 1; k <= n; ++k) {
        //             printf("%d ", f[i][j][k]);
        //         }
        //         printf("\n");
        //     }
        //     printf("\n");
        // }
        return ans;
    }
};
