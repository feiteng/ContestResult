using namespace std;
class Solution {
	int dp[110][22][110];
	void reset() {
		for (int i = 0; i < 110; i ++ )
			for (int j = 0; j < 22; j ++ ) 
				for (int l = 0; l < 110; l ++ )
					dp[i][j][l] = INF;
	}
	public:int minCost(vector<int>& houses, vector<vector<int> >& cost, int m, int n, int target) {
        reset();
        dp[0][0][0] = 0;
    	if ( houses[0] != 0) {
			dp[1][houses[0]][1] = 0;
		} else {
			for (int c1 = 1; c1 <= n; c1 ++ ) {
				int c = cost[0][c1 - 1];
				dp[1][c1][1] = c;
			}
		}
        for (int h = 2; h <= m; h ++ ) {
			if ( houses[h - 1] == 0 ) {
				for (int c1 = 1; c1 <= n; c1 ++ ) {
					for (int c2 = 1; c2 <= n; c2 ++ ) {
						for (int t = 1; t <= target; t ++ ) {
							int c = cost[h - 1][c2 - 1];
							int tadd = c1 != c2;
							dp[h][c2][t + tadd] = min(dp[h][c2][t + tadd], dp[h - 1][c1][t] + c);
						}
					}
				}
			} else {
				for (int c1 = 1; c1 <= n; c1 ++ ) {
					for (int t = 1; t <= target; t ++ ) {
						int c2 = houses[h - 1];
						int tadd = c1 != c2;
						dp[h][c2][t + tadd] = min(dp[h][c2][t + tadd], dp[h - 1][c1][t]);
					}
				}
			}
		}
		int ans = INF;
		for (int c1 = 1; c1 <= n; c1 ++ ) {
			ans = min (ans , dp[m][c1][target]);
		}
		if (ans == INF) return -1;
		return ans;
    }
};
