class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        const int inf = 1e7;
        vector<vector<vector<int>>> dp(m, vector<vector<int>>(n, vector<int>(target+1, inf)));
        
        for (int& x : houses) {
            --x;
        }
        

        if (houses[0] == -1) {
            for (int col = 0; col < n; col++) {
                dp[0][col][1] = cost[0][col];
            }
        } else {
            dp[0][houses[0]][1] = 0;
        }
        
        for (int i = 1; i < m; i++) {
            if (houses[i] >= 0) {
                for (int col = 0; col < n; col++) {

                    for (int k = 1; k <= target; k++) {
                        if (col == houses[i]) {
                            dp[i][col][k] = min(dp[i][col][k], dp[i-1][col][k]);
                        } else {
                            dp[i][houses[i]][k] = min(dp[i][houses[i]][k], dp[i-1][col][k-1]);
                        }
                    }
                }
                continue;
            } else {
                for (int col = 0; col < n; col++) {
                    for (int pre = 0; pre < n; pre++) {

                        for (int k = 1; k <= target; k++) {
                            if (col == pre) {
                                dp[i][col][k] = min(dp[i][col][k], dp[i-1][col][k] + cost[i][col]);
                            } else {
                                dp[i][col][k] = min(dp[i][col][k], dp[i-1][pre][k-1] + cost[i][col]);
                            }
                        }
                    }

                }
            }
        }
        
        int res = inf;
        for (int col = 0; col < n; col++) {
            res = min(res, dp[m-1][col][target]);
        }
        return (res == inf ? -1 : res);
    }
};
