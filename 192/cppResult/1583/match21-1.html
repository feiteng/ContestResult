int dp[100][21][101];
class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(dp, -1, sizeof(dp));
        if (houses[0]) {
            dp[0][houses[0]][1] = 0;
        } else {
            for (int i = 1; i <= n; i++) {
                dp[0][i][1] = cost[0][i - 1];
            }    

        }
        
        for (int i = 1; i < m; i++) {
            for (int pre = 1; pre <= n; pre++) {
                for (int sz = 1; sz <= m; sz++) {
                    if (dp[i - 1][pre][sz] == -1) continue;
                    for (int now = 1; now <= n; now++) {
                        int c = cost[i][now - 1];
                        if (houses[i] && houses[i] != now) continue;

                        if (now == houses[i]) c = 0;
                        if (dp[i][now][sz + (now != pre)] == -1 ||
                           dp[i - 1][pre][sz] + c < dp[i][now][sz + (now != pre)]) {
                            dp[i][now][sz + (now != pre)] = dp[i - 1][pre][sz] + c;
                        }
                    }
                    
                }
            }
        }
        /*for (int i = 0; i < m; i++) {
            for (int j = 1; j <= n; j++) {
                for (int sz = 1; sz <= target; sz++) {
                    printf("%d, %d, %d = %d\n", i, j, sz, dp[i][j][sz]);
                }
            }
            puts("");
        }*/
        
        int ans = -1;
        for (int i = 1; i <= n; i++) {
            if (dp[m - 1][i][target] == -1) continue;
            if (ans == -1 || dp[m - 1][i][target] < ans) {
                ans = dp[m - 1][i][target];
            }
        }
        return ans;
    }
};
