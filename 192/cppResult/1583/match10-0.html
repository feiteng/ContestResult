class Solution {
public:
    int minCost(vector<int>& a, vector<vector<int>>& c, int m, int n, int tar) {
        vector<vector<vector<int>>> dp(m + 1, vector<vector<int>> (n + 1, vector<int>(m + 1, INT_MAX / 2)));
        if (a[0] != 0) dp[0][a[0] - 1][1] = 0;
        else {
            for (int j = 0; j < n; j++)
                dp[0][j][1] = c[0][j];
        }
        for (int i = 1; i < m; i++) {
            if (a[i] != 0) {
                int col = a[i] - 1;
                for (int k = 1; k <= tar; k++) {
                    dp[i][col][k] = dp[i - 1][col][k];
                    for (int p = 0; p < n; p++)
                        if (p != col)
                            dp[i][col][k] = min(dp[i][col][k], dp[i - 1][p][k - 1]);
                }
            } else {
                for (int j = 0; j < n; j++)
                    for (int k = 1; k <= tar; k++) {
                        dp[i][j][k] = dp[i - 1][j][k] + c[i][j];
                        for (int p = 0; p < n; p++)
                            if (p != j)
                                dp[i][j][k] = min(dp[i][j][k], dp[i - 1][p][k - 1] + c[i][j]);
                    }
            }
            // for (int j = 0; j < n; j++) {
            //     for (int k = 1; k <= tar; k++)
            //         cout << dp[i][j][k] << ' ';
            //     cout << endl;
            // }
            // cout << endl;       
        }
        int ans = INT_MAX;
        for (int j = 0; j < n; j++)
            ans = min(ans, dp[m - 1][j][tar]);
        if (ans >= INT_MAX / 2) return -1;
        return ans;
    }
};
