class Solution {
public:
    int f[110][25][110];
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(f,0x3f,sizeof(f));
        

        if(houses[0]!=0) f[0][houses[0]][1] = 0;
        else{
            for(int j=0;j<n;j++) f[0][j+1][1] = cost[0][j];
        }
        
        for(int i=1;i<m;i++){
           if(houses[i]!=0){

               for(int k=1;k<=target;k++){
                   f[i][houses[i]][k] = f[i-1][houses[i]][k];
                   for(int t=0;t<n;t++)
                       if(t+1!=houses[i])
                           f[i][houses[i]][k] = min(f[i][houses[i]][k], f[i-1][t+1][k-1]);
               } 
           }
           else{
               for(int j=0;j<n;j++){

                   for(int k=1;k<=target;k++){
                       f[i][j+1][k] = f[i-1][j+1][k] + cost[i][j];
                       for(int t=0;t<n;t++)
                           if(t+1!=j+1)
                               f[i][j+1][k] = min(f[i][j+1][k], f[i-1][t+1][k-1]+cost[i][j]);
                   }
               }
           }
        }
        // for(int i=0;i<m;i++)
        //     for(int j=0;j<n;j++)
        //         for(int k=1;k<=target;k++)
        //             cout<<i<<" "<<j+1<<" "<<k<<": "<<f[i][j+1][k]<<endl;

        int ans =0x3f3f3f3f;
        for(int j=0;j<n;j++)
            ans = min(ans,f[m-1][j+1][target]);
        if(ans==0x3f3f3f3f) return -1;
        return ans;
    }
};
