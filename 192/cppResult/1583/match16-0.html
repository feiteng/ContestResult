int dp[101][101][21];
class Solution {
public:
    int minCost(vector<int>& a, vector<vector<int>>& cost, int m, int n, int target) {
		for (int i=0;i<=m;++i)
			for (int j=0;j<=m;++j)
				for (int k=0;k<=n;++k) dp[i][j][k]=1e9;
		if (a[0]==0)
		{
			for (int i=1;i<=n;++i) dp[1][1][i]=cost[0][i-1];
		}
		else
		{
			dp[1][1][a[0]]=0;
		}
		for (int i=1;i<m;++i)
			for (int j=1;j<=m;++j)
				for (int k=1;k<=n;++k)
				{
					if (dp[i][j][k]==1e9) continue;
                    cout<<i<<" "<<j<<" "<<k<<" "<<dp[i][j][k]<<endl;
					if (a[i]!=0)
					{
						if (a[i]==k) dp[i+1][j][k]=min(dp[i+1][j][k],dp[i][j][k]);
						else dp[i+1][j+1][a[i]]=min(dp[i+1][j+1][a[i]],dp[i][j][k]);
					}
					else
					{
						for (int t=1;t<=n;++t)
						{
							if (t==k) dp[i+1][j][k]=min(dp[i+1][j][k],dp[i][j][k]+cost[i][t-1]);
							else dp[i+1][j+1][t]=min(dp[i+1][j+1][t],dp[i][j][k]+cost[i][t-1]);
						}
					}
				}
		int ans=1e9;
		for (int i=1;i<=n;++i) ans=min(ans,dp[m][target][i]);
		if (ans==1e9) ans=-1;
		return ans;
    }
};
