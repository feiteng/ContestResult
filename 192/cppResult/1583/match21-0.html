int f[105][25][105];
class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        memset(f, -1, sizeof f);
        if (houses[0] == 0) {
            for (int i = 1; i <= n; ++i)
                f[0][i][1] = cost[0][i - 1];
        } else {
            f[0][houses[0]][1] = 0;
        }
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j <= n; ++j) {
                for (int g = 1; g <= i + 1; ++g) {
                    if (f[i - 1][j][g] == -1) continue;
                    if (houses[i] == 0) {
                        
                        for (int nj = 1; nj <= n; ++nj) {
                            int ng = g + (nj != j);
                            if (f[i][nj][ng] == -1 || f[i][nj][ng] > f[i - 1][j][g] + cost[i][nj - 1]) {
                                f[i][nj][ng] = f[i - 1][j][g] + cost[i][nj - 1];
                            }
                        }
                    } else {
                        int c = houses[i];
                        int ng = g + (c != j);
                        if (f[i][c][ng] == -1 || f[i][c][ng] > f[i - 1][j][g]) {
                            f[i][c][ng] = f[i - 1][j][g];
                        }
                    }
                }
            }
        }
        int ans = -1;
        for (int i = 1; i <= n; ++i) {
            if (f[m - 1][i][target] == -1) continue;
            if (ans == -1 || ans > f[m - 1][i][target]) {
                ans = f[m - 1][i][target];
            }
        }
        return ans;
    }
};
