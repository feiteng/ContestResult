class Solution {
public:
  int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
    const int inf = 1000010000;
    vector<vector<vector<int>>> a(m + 1, vector<vector<int>>(n + 1, vector<int>(target + 2, inf)));
    a[0][0][0] = 0;
    for (int i = 0; i < m; ++i) {
      for (int j = 0; j <= n; ++j) {
        for (int k = 0; k <= target; ++k) {
          if (houses[i]) {
            int j1 = houses[i];
            int k1 = k + (j1 != j);
            a[i + 1][j1][k1] = min(a[i + 1][j1][k1], a[i][j][k]);
          } else {
            for (int j1 = 1; j1 <= n; ++j1) {
              int k1 = k + (j1 != j);
              a[i + 1][j1][k1] = min(a[i + 1][j1][k1], a[i][j][k] + cost[i][j1 - 1]);
            }
          }
        }
      }
    }
    int ret = inf;
    for (int i = 1; i <= n; ++i) ret = min(ret, a[m][i][target]);
    return ret < inf ? ret : -1;
  }
};
