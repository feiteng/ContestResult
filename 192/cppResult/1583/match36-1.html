class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<int>> dp(n+1, vector<int>(target+1, 1e9));
        dp[0][0] = 0;
        // current color, already neiborhood
        for(int i=0;i<m;i++){
            vector<vector<int>> ndp(n+1, vector<int> (target+1, 1e9));
            // cout<<"i = "<<i<<endl;
            if(houses[i] == 0){
                for(int j=1;j<=n;j++){ // color to paint

                    for(int k=0;k<=n;k++){ // pre color
                        for(int l = 0;l<=target;l++){
                            if(k == j){
                                ndp[j][l] = min(ndp[j][l], dp[k][l] + cost[i][j-1]);
                                // if(k==0 && l==0) cout<<j <<' '<<l<<' '<<ndp[j][l]<<endl;
                            }else{
                                if(l<target){
                                    ndp[j][l + 1] = min(ndp[j][l + 1], dp[k][l] + cost[i][j-1]);
                                    // cout<<j<<' '<<l+1<<endl;
                                    // if(k==0 && l==0) cout<<j <<' '<<l+1<<' '<<ndp[j][l+1]<<endl;
                                }
                            }
                            // if(dp[k][l] <=1000000) cout<<k <<' '<<l<<endl;
                        }
                    }
                }
            }else{

                int j=houses[i];
                for(int k=0;k<=n;k++){
                    for(int l = 0;l<=target;l++){
                        if(k == j){
                            ndp[j][l] = min(ndp[j][l], dp[k][l] );
                        }else{
                            if(l<target){
                                ndp[j][l + 1] = min(ndp[j][l + 1], dp[k][l] );
                            }

                        }
                    }
                }
            }
            // cout<<ndp[1][1]<<endl;
            // for(int ii=0;ii<=n;ii++){
            //     for(int jj=0;jj<=target;jj++){
            //         cout<<ndp[ii][jj]<< ' ';
            //     }cout<<endl;
            // }cout<<endl;
            swap(dp, ndp);
        }
        int res = 1e9+7;
        for(int i=1;i<=n;i++) res = min(res, dp[i][target]);
        // cout<<res<<endl;
        if(res >= 1e9)return -1;
        return res;
    }
};
