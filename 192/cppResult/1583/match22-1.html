class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m, vector<vector<int>> (target+1, vector<int>(n, -1)));
        
        for (int c = 0; c < n; c++) {
            if (houses[0] == 0) dp[0][1][c] = cost[0][c];
            else dp[0][1][c] = (houses[0] == c+1) ? 0 : -1;
        }
        

        for (int i = 1; i < m; i++) {
            // cout<<"here"<<endl;
            if (houses[i] != 0) {
                // cout<<"here1"<<endl;
                for (int t = 1; t <= target; t++) {
                    int c = houses[i]-1;
                    dp[i][t][c] = (dp[i-1][t][c] == -1) ? INT_MAX : dp[i-1][t][c];
                    for (int k = 0; k < n; k++) {
                        if (k == c || dp[i-1][t-1][k] == -1) continue;
                        dp[i][t][c] = min(dp[i][t][c], dp[i-1][t-1][k]);
                    }

                    if (dp[i][t][c] == INT_MAX) dp[i][t][c] = -1;
                    // cout<<"here2 "<<i<<" "<<t<<" "<<c<<" "<<dp[i][t][c]<<endl;
                }
                continue;
            }
            for (int t = 1; t <= target; t++) {
                for (int c = 0; c < n; c++) {
                    dp[i][t][c] = (dp[i-1][t][c] == -1) ? INT_MAX : dp[i-1][t][c] + cost[i][c];
                // cout<<"here2"<<endl;
                    for (int k = 0; k < n; k++) {
                        if (k == c || dp[i-1][t-1][k] == -1) continue;
                        dp[i][t][c] = min(dp[i][t][c], dp[i-1][t-1][k] + cost[i][c]);
                    }
                    if (dp[i][t][c] == INT_MAX) dp[i][t][c] = -1;
                    // cout<<"here "<<i<<" "<<t<<" "<<c<<" "<<dp[i][t][c]<<endl;

                }
            }
        }
        // cout<<"here4"<<endl;
        int ans = INT_MAX;
        for (int c = 0; c < n; c++) {
            if (dp[m-1][target][c] == -1) continue;
            // cout<<"here "<<c<<" "<<dp[m-1][target][c]<<endl;
            ans = min(ans, dp[m-1][target][c]);
        }
        // cout<<endl;
        // cout<<endl;
        
        return (ans == INT_MAX) ? -1 : ans;
    }
};
