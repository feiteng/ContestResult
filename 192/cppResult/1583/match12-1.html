class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int> > > dp(m);
        for(int i = 0 ; i < m ; i++){
            dp[i].resize(n+1);
            for(int j = 0 ; j < n+1 ; j++){
                dp[i][j].resize(m+1, 1e9);
            }
        }

        if(houses[0] == 0){
            for(int i = 1 ; i <= n ; i++){
                dp[0][i-1][1] = cost[0][i-1];
            }
        }else {
            dp[0][houses[0]-1][1] = 0;
        }
        for(int i = 1 ; i < m ; i++){
            if(houses[i] != 0){
                int j = houses[i]-1;
                for(int k = 1 ; k <= m ; k++){

                    for(int l = 0 ; l < n ; l++){
                        if(l==j){
                            dp[i][j][k] = min(dp[i][j][k], dp[i-1][l][k]);
                        }else {
                            dp[i][j][k] = min(dp[i][j][k], dp[i-1][l][k-1]);
                        }
                        
                    }

                }
            }else {
                for(int j = 0 ; j < n ; j++){
                    for(int k = 1 ; k <= m ; k++){

                        for(int l = 0 ; l < n ; l++){
                            if(l==j){
                                dp[i][j][k] = min(dp[i][j][k], dp[i-1][l][k] + cost[i][j]);
                            }else {
                                dp[i][j][k] = min(dp[i][j][k], dp[i-1][l][k-1] + cost[i][j]);
                            }
                        }

                    }
                }
            }
        }
        int ans = 1e9;
        for(int i = 0 ; i < n ; i++){
            ans = min(ans, dp[m-1][i][target]);
        }
        // for(int i = 0 ; i < m ; i++){
        //     for(int j = 0 ; j < n ; j++){
        //         for(int k = 1 ; k <= m ;k++){
        //             cout << dp[i][j][k] << " ";
        //         }
        //         cout << endl;
        //     }
        //     cout << endl << " ***** " << endl;
        // }
        // cout << "-------" << endl;
        if(ans > 1e7) ans = -1;
        return ans;
        
        
    }
};
