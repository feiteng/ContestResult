class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> costa(m,vector<vector<int>>(n+1,vector<int>(m+1,INT_MAX)));
        if(houses[0]!=0)
        {
            costa[0][houses[0]][1]=0;
        }

        else
        {
            for(int i=1;i<=n;++i)
            {
                costa[0][i][1]=cost[0][i-1];
            }
        }
        
        for(int i=0;i<m-1;++i)
        {
            for(int j=1;j<=n;++j)
            {
                for(int k=1;k<=m;++k)
                {
                    if(costa[i][j][k]==INT_MAX)
                    {
                        continue;
                    }
                    if(houses[i+1]!=0)
                    {
                        if(j==houses[i+1])
                        {
                            int &t=costa[i+1][houses[i+1]][k];
                            t=min(t,costa[i][j][k]);
                        }
                        else
                        {
                            int &t=costa[i+1][houses[i+1]][k+1];
                            t=min(t,costa[i][j][k]);
                        }
                    }
                    else
                    {
                        for(int c=1;c<=n;++c)
                        {
                            if(j==c)
                            {
                                int &t=costa[i+1][c][k];
                                t=min(t,costa[i][j][k]+cost[i+1][c-1]);
                            }
                            else
                            {
                                int &t=costa[i+1][c][k+1];
                                t=min(t,costa[i][j][k]+cost[i+1][c-1]);
                            }
                        }

                    }
                }
            }
        }
        
        int ans=INT_MAX;
        for(int i=1;i<=n;++i)
        {
            ans=min(ans,costa[m-1][i][target]);
        }
        if(ans==INT_MAX) return -1;
        return ans;
    }
};
