const int M = 110;
const int N = 25;
class Solution {
public:
    int f[M][N][M];

    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                for(int k=0;k<=target;k++){
                    f[i][j][k] = 1e9;
                }
            }
        }
        if(houses[0]>0){
            f[0][houses[0]-1][1] = 0;
        }else{
            for(int j=0;j<n;j++){
                f[0][j][1] = cost[0][j];
            }
        }
        for(int i=1;i<m;i++){
            if(houses[i]>0){
                int cur=houses[i]-1;
                for(int j=0;j<n;j++){
                    for(int k=1;k<=target;k++){
                        if(j==cur){
                            f[i][cur][k] = min(f[i][cur][k],f[i-1][j][k]);
                            // printf("%d %d %d f: %d %d %d %d f: %d\n",i,cur,k,f[i][cur][k],i-1,j,k,f[i-1][j][k]);
                        }else{
                            f[i][cur][k] = min(f[i][cur][k],f[i-1][j][k-1]);
                            // printf("%d %d %d f: %d %d %d %d f: %d\n",i,cur,k,f[i][cur][k],i-1,j,k-1,f[i-1][j][k-1]);
                        }
                    }              
                }
                continue;

            }
            for(int cur=0;cur<n;cur++){
                for(int j=0;j<n;j++){
                    for(int k=1;k<=target;k++){
                        if(cur==j){
                            f[i][cur][k] = min(f[i][cur][k],f[i-1][j][k]+cost[i][cur]);
                            // printf("%d %d %d f: %d %d %d %d f: %d\n",i,cur,k,f[i][cur][k],i-1,j,k,f[i-1][j][k]);
                        }else{

                            f[i][cur][k] = min(f[i][cur][k],f[i-1][j][k-1]+cost[i][cur]);
                            // printf("%d %d %d f: %d %d %d %d f: %d\n",i,cur,k,f[i][cur][k],i-1,j,k-1,f[i-1][j][k-1]);
                        }
                    }
                }
            }
            
        }
        int ret = 1e9;
        for(int j=0;j<n;j++){
            ret = min(ret,f[m-1][j][target]);
        }
        return ret==1e9?-1:ret;
    }
};
