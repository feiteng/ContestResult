class Solution
{
public:
    //dp[i][j][k] 前i个房子，j个街区，最后一个房子是k时得最小数量
    int dp[105][105][25];
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target)
    {
        //memset(-1,dp,sizeof dp);
        const int inf = 100000000;
        for(int i=0; i<105; ++i)
        {
            for(int j=0; j<105; ++j)
            {
                for(int k=0; k<25; ++k)
                {
                    dp[i][j][k]=inf;
                }
            }
        }
        dp[0][0][0] = 0;
        for(int i=1; i<=m; ++i)
        {
            for(int j=1; j<=target; ++j)
            {
                if(houses[i-1]!=0)
                {
                    int col = houses[i-1];
                    for(int last_col = 0; last_col<=n; ++last_col) {
                        if(col == last_col) {
                            dp[i][j][col] = min(dp[i][j][col],dp[i-1][j][last_col]);
                        }else {
                            dp[i][j][col] = min(dp[i][j][col],dp[i-1][j-1][last_col]);
                        }
                    }
                }
                else
                {
                    for(int col = 1; col <= n; ++col)
                    {
                        for(int last_col = 0; last_col<=n; ++last_col)
                        {
                            //cout<<i<<"-"<<j<<"-"<<col<<"-"<<last_col<<endl;
                            //dp[i][j][col] = dp[i-1][j][col] + cost[i][j-1];
                            if(last_col == col)
                            {
                                /*if(dp[i][j][col] == -1) {
                                    dp[i][j][col] = dp[i-1][j][col] + cost[i][j-1];
                                }
                                else {
                                    dp[i][j][col] = min(dp[i][j][col], dp[i-1][j][col] + cost[i][j-1]);
                                }*/
                                //if(i>0)
                                    dp[i][j][col] = min(dp[i][j][col], dp[i-1][j][col] + cost[i-1][col-1]);
                            }
                            else
                            {
                                /*if(dp[i][j][col] == -1) {
                                    dp[i][j][col] = dp[i-1][j-1][last_col] + cost[i][j-1];
                                }
                                else {
                                    dp[i][j][col] min( dp[i][j][col], dp[i-1][j-1][last_col] + cost[i][j-1]);
                                }*/
                                //cout<<i-1<<"--"<<j-1<<endl;
                                //if(i>0&&j>0)
                                    dp[i][j][col] = min( dp[i][j][col], dp[i-1][j-1][last_col] + cost[i-1][col-1]);
                            }
                            //cout<<"["<<i<<","<<j<<","<<col<<"]="<<dp[i][j][col]<<endl;

                        }

                    }
                }
            }
        }

        int mn = inf;
        for(int i=1; i<=n; ++i)
            mn = min(mn,dp[m][target][i]);
        if(mn == inf)
            mn = -1;
        return mn;
    }
};
