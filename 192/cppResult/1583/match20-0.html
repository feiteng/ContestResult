#define ll long long int 
#define inf 1e8
class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        vector<vector<vector<int>>> dp(m,vector<vector<int>>(n+1,vector<int>(target+1,inf)));
        if(houses[0]!=0) dp[0][houses[0]][1]=0;
        else{
            for(int i=1;i<=n;i++){
                dp[0][i][1]=cost[0][i-1];
            }
        }
        
        for(int i=1;i<m;i++){
            if(houses[i]==0){
                for(int j=1;j<=n;j++){
                    for(int k=1;k<=target;k++){
                        for(int x=1;x<=n;x++){
                            if(x==j) dp[i][j][k]=min(dp[i][j][k],dp[i-1][x][k]+cost[i][j-1]);
                            else dp[i][j][k]=min(dp[i][j][k],dp[i-1][x][k-1]+cost[i][j-1]);
                        }
                    }
                }
            }else{
                for(int k=1;k<=target;k++){
                    for(int j=1;j<=n;j++){
                        if(j!=houses[i]) dp[i][houses[i]][k]=min(dp[i][houses[i]][k],dp[i-1][j][k-1]);
                        else dp[i][houses[i]][k]=min(dp[i][houses[i]][k],dp[i-1][houses[i]][k]);
                    }
                }
            }
        }
        int ans=1e9;
        for(int i=1;i<=n;i++) ans=min(ans,dp[m-1][i][target]);
        if(ans==1e8) ans=-1;
        return ans;
    }
};
