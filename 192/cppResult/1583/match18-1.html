class Solution {
public:
    int minCost(vector<int>& houses, vector<vector<int>>& cost, int m, int n, int target) {
        int dp[m+5][n+5][target+5];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k <= target; k++) {
                    dp[i][j][k] = INT_MAX;
                }
            }
        }
        for (int j = 0; j < n; j++) {
            if (houses[0] == 0) {
                dp[0][j][1] = cost[0][j];

            } else if (j+1 == houses[0]) {
                dp[0][j][1] = 0;
            }
        }
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 1; k <= target; k++) {
                    if (houses[i] == 0 || j+1 == houses[i]) {
                        for (int p = 0; p < n; p++) {
                            if (p != j) {
                                dp[i][j][k] = min(dp[i][j][k], dp[i-1][p][k-1]);
                            }
                        }
                        dp[i][j][k] = min(dp[i][j][k], dp[i-1][j][k]);
                        if (houses[i] == 0 && dp[i][j][k] < INT_MAX) dp[i][j][k] += cost[i][j];

                    }
                }
            }
        }
        int res = INT_MAX;
        for (int j = 0; j < n; j++) {
            res = min(res, dp[m-1][j][target]);
        }
        return res == INT_MAX ? -1 : res;
    }
};
