class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        
        int n = arr.size();
        
        int m = arr[(n-1)/2];
        
        vector<int> ans;
        int i = 0, j = n-1;
        
        while(k--) {
            int a = abs(arr[i]-m);
            int b = abs(arr[j]-m);
            
            if(a > b) {
                ans.push_back(arr[i]);
                i++;
            }
            else {
                ans.push_back(arr[j]);
                
                j--;
            }
        }
        return ans;
    }
};
