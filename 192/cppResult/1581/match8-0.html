class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        int m = arr[(n - 1) / 2];
        vector<int> ret;
        for (int l = 0, r = n - 1, i = 0; i < k; ++i) {
            if (abs(arr[l] - m) > abs(arr[r] - m)) {
                ret.push_back(arr[l++]);
            } else {
                ret.push_back(arr[r--]);
            }
        }
        return ret;
    }
};
