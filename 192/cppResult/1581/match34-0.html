class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n=arr.size();
        sort(arr.begin(),arr.end());
        int mid=arr[(n-1)/2];
        
        vector<int>index(n);
        iota(index.begin(),index.end(),0);
        sort(index.begin(),index.end(),[&](int& a,int& b)
             {
                return  abs(arr[a] - mid) > abs(arr[b] - mid)||(abs(arr[a] - mid) == abs(arr[b] - mid)&&arr[a]>arr[b]);
             });
        vector<int>ret;
        for(int i=0;i<k;i++)
            ret.push_back(arr[index[i]]);
        return ret;
    }
};
