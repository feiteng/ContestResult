class Solution {
public:
    vector<int> getStrongest(vector<int> arr, int k) {
        int n = arr.size();
        sort(arr.begin(),arr.end());
        int m = arr[(n-1)/2];
        sort(arr.begin(),arr.end(),[&](int &a,int &b){
            if(abs(a - m) != abs(b - m))
                return abs(a - m) > abs(b - m);
            return a > b;
        });
        vector<int> rt;
        for(int i =0 ;i<k;i++)
            rt.push_back(arr[i]);
        return rt;        
    }
};
