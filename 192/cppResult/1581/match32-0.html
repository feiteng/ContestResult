class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        sort(a.begin(), a.end());
        int n = a.size();
        int m = a[(n-1) / 2];
        
        vector<pair<int, int>> f;
        for (int t : a) 
            f.push_back(make_pair(abs(t - m), t));
        sort(f.begin(), f.end(), [](auto a, auto b) {
           return a.first == b.first ? a.second > b.second : a.first > b.first; 
        });
        vector<int> ans;
        for (int i = 0; i < k; i++)
            ans.push_back(f[i].second);
        return ans;
    }
};
