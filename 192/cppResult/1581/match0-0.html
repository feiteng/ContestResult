class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n = arr.size();
        
        sort(arr.begin(), arr.end());
        int mid = arr[(n - 1) / 2];
        
        auto cmp = [&](int l, int r) {
            if (abs(l - mid) == abs(r - mid)) {
                return l > r;
            }
            
            return abs(l - mid) > abs(r - mid);
        };
        
        sort(arr.begin(), arr.end(), cmp);
        
        vector<int> res;
        res.reserve(k);
        
        for (int i = 0; i < k; ++i) {
            res.push_back(arr[i]);
        }
        
        return res;
        
    }
};
