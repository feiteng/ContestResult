class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n = arr.size();
        sort(arr.begin(), arr.end());
        int median = arr[(n - 1) / 2];
        int l = 0, r = n - 1;
        vector<int> ret;
        while((ret.size()) < k){
            if(abs(arr[l] - median) > abs(arr[r] - median)){
                ret.push_back(arr[l]);
                l++;
            }
            else{
                ret.push_back(arr[r]);
                r--;
            }
        }
        return ret;
    }
};
