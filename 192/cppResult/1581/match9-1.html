class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        int m = (n-1)/2;
        vector<int> res;
        for (int i = 0, j = n-1; res.size() < k;) {

            if (abs(arr[i]-arr[m]) > abs(arr[j]-arr[m])) {
                res.emplace_back(arr[i]);
                ++i;
            } else {
                res.emplace_back(arr[j]);
                --j;
            }
        }
        return res;
    }
};
