class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int N = arr.size();
        int mid = arr[(N-1)/2];
        // printf("%d\n",mid);
        vector<pair<int,int>> tmp;
        for (int i =0 ; i<N; ++i) {
            tmp.push_back(make_pair(abs(arr[i]-mid), arr[i]));
            // printf("%d %d\n", tmp.back().first, tmp.back().second);
        }
        sort(tmp.begin(),tmp.end());
        
        vector<int> ans;
        for (int i = 0; i <k;i++) {
            ans.push_back(tmp[N-1-i].second);
        }
        return ans;
    }
};
