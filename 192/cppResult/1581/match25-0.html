class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n = arr.size();
        sort(arr.begin(),arr.end());
        vector<int> re;
        int st=0,en=arr.size()-1;
        int mi = arr[(n-1)/2];
        while(k>0)
        {
            if(abs(arr[en] - mi) >= abs(arr[st]-mi))
            {
                re.push_back(arr[en]);
                en--;
            }
            else
            {
              re.push_back(arr[st]);
                st++;  
            }
            k--;
        }
        return re;
    }
};
