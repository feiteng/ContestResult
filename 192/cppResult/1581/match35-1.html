const ll MOD=1e9+7;

class Solution {
public:
  vector<int> getStrongest(vector<int>& a, int k) {
    sort(a.begin(),a.end());
    int N=a.size();
    int m = a[(N-1)/2];
    sort(a.begin(),a.end(),[m](int x, int y) {
                             return abs(x-m)>abs(y-m) || abs(x-m)==abs(y-m) && x>y;
                           });
    return vector<int>(a.begin(),a.begin()+k);
  }
};
