class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int median = arr[(arr.size() - 1) / 2];
        // cout << median << endl;
        vector<int> ans;
        int lo = 0, hi = arr.size() - 1;

        while(ans.size() < k && lo <= hi) {
            if (abs(arr[hi] - median) >= abs(arr[lo] - median)) {
                ans.push_back(arr[hi]);
                -- hi;
            } else {
                ans.push_back(arr[lo]);
                ++ lo;
            }
        }
        return ans;
    }
};
