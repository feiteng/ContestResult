int m = 0;

bool cmp(int a, int b){
    if(abs(a - m) == abs(b - m)) return a < b;
    return abs(a - m) < abs(b - m);
}

class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        m = arr[(arr.size() - 1)/2];
        sort(arr.begin(), arr.end(), cmp);
        vector<int> ans;
        for(int i = arr.size() - k; i<arr.size(); i++){
            ans.push_back(arr[i]);
        }
        return ans;
    }
};
