class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(),arr.end());
        int n = arr.size();
        int m = arr[(n-1)/2];
        auto cmp=[&](int a,int b){
            if(abs(a-m) != abs(b-m)){
                return abs(a-m) > abs(b-m);
            }
            return a>b;
        };
        sort(arr.begin(),arr.end(),cmp);
        vector<int> res;
        for(int i=0;i<k;i++){
            res.push_back(arr[i]);
        }
        return res;
    }
};
