class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
      sort(all(arr));
      int median = arr[(sz(arr)-1)/2];
      vector<int> ret;
      int lhs = 0;
      int rhs = sz(arr)-1;
      while(sz(ret) < k) {
        if(abs(arr[lhs] - median) > abs(arr[rhs]-median)) {
          ret.push_back(arr[lhs++]);
        }
        else {
          ret.push_back(arr[rhs--]);
        }
      }
      return ret;
    }
};
