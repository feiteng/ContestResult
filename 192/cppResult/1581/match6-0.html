class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
		sort( arr.begin(), arr.end());
		int n = arr.size();
        int med = arr[(n-1)/2];
		
		sort(arr.begin(), arr.end(), 
			[med](int a, int b){
				if ( abs(a-med) == abs(b-med) ) {
					return a < b;
				}
				return abs(a-med) < abs(b-med);
			});
			
		vector<int> res;
		for ( int i= n-1; i >= n-k; --i ) {
			res.push_back(arr[i]);
		}
		return res;
    }
};
