class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        int n = a.size();
        sort(a.begin(), a.end());
        int m = a[(n - 1) / 2];
        sort(a.begin(), a.end(),
            [&](int u, int v) {
                int x = abs(u - m), y = abs(v - m);

                if (x != y) return x > y;
                return u > v;
            });
        vector<int> ret;
        for (int i = 0; i < k; ++i) ret.push_back(a[i]);
        return ret;
    }
};
