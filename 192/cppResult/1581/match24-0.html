class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        int m = arr[(n-1) / 2];
        vector<int> ret;
        int left = 0;
        int right = n-1;
        for(int i = 0; i < k; ++i) {
            if(abs(arr[left]-m) > abs(arr[right] - m)) {
                ret.push_back(arr[left]);
                left++;
            } else {
                ret.push_back(arr[right]);
                right--;
            }
        }
        
        return ret;
    }
};
