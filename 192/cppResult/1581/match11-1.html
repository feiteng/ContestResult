class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        
        sort(arr.begin(),arr.end());
        int n=arr.size();
        int m = arr[(n-1)/2];
        
        //   for(int i=0;i<n;i++)
        // {
        //     cout<<arr[i]<<" ";
        // }
       
        vector<int>ans;
        int l=0;

        int r=n-1;
        for(int i=0;i<k;i++)
        {
            if(abs(arr[l]-m)>abs(arr[r]-m))
            {
                ans.push_back(arr[l]);
                l++;
                
            }
            else{
                ans.push_back(arr[r]);
                r--;
            }
            
        }
        return ans;
    }
};
