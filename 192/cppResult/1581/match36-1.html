class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        sort(a.begin(), a.end());
        int n = a.size();
        int m = a[(n - 1) / 2];
        // cout << m << "\n";
        sort(a.begin(), a.end(), [&](const int &x, const int &y){
           if (abs(x - m) > abs(y - m)) {
               return true;
           } else if (abs(x - m) == abs(y - m)) {
               if (x > y) {
                   return true;
               } else {
                   return false;
               }

           } else {
               return false;
           }
        });
        // for (int &x: a) {
        //     cout << x << " ";
        // }
        // cout << "\n";
        vector<int> ret;
        for (int i = 0; i < k; ++i) {
            ret.push_back(a[i]);
        }
        return ret;
    }
};
