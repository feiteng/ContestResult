class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        int n = a.size();
        vector<int> b = a;
        sort(b.begin(), b.end());
        int m = b[(n-1)/2];
        vector<int> p(n);
        for (int i = 0; i < n; ++i) p[i] = i;

        sort(p.begin(), p.end(), [&](int i, int j)
             {
                 return abs(a[i]-m) > abs(a[j]-m) || abs(a[i]-m) == abs(a[j]-m) && a[i] > a[j];
             });
        vector<int> ret;
        for (int i = 0; i < k; ++i) ret.push_back(a[p[i]]);
        return ret;
    }
};
