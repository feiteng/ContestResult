class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n = arr.size();
        sort(arr.begin(), arr.end());
        int m = arr[(n-1)/2];
        
        sort(arr.begin(), arr.end(), 
            [&](const int a, const int b) {
                if(abs(a-m) != abs(b-m)) {
                    return abs(a-m) < abs(b-m);
                }
                else {
                    return a < b;
                }
            });
        
        vector<int> ans;
        for(int i = 0; i < k; i++) {
            ans.push_back(arr[n-1-i]);
        }
        return ans;
    }
};
