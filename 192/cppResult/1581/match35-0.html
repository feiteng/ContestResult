class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        sort(a.begin(),a.end());
        int n=a.size();
        int m = a[(n-1)/2];
        sort(a.begin(),a.end(),[&](int x,int y) { return abs(x-m)>abs(y-m) || abs(x-m)==abs(y-m) && x>y; });
        return vector<int>(a.begin(),a.begin()+k);
    }
};
