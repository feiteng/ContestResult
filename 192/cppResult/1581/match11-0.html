class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        int med = arr[(n - 1) / 2];
        vector<int> ans;
        int l = 0, r = n - 1;
        for (int i = 0; i < k; ++i) {
            if (l == (n - 1) / 2 + 1 || arr[r] - med >= med - arr[l]) {
                ans.push_back(arr[r--]);
            }
            else {
                ans.push_back(arr[l++]);
            }
        }
        return ans;
    }
};
