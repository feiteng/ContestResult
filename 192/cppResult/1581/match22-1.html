int K;
bool myfun(int i, int j) {
    if (abs(i-K)==abs(j-K)) return i>j;
    return abs(i-K)>abs(j-K);
}
class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        int n=arr.size();
        sort(arr.begin(),arr.end());
        K=arr[(n - 1) / 2];
        sort(arr.begin(),arr.end(),myfun);
        vector<int> ans;
        for (int i=0; i<k; i++) ans.push_back(arr[i]);
        return ans;
    }
};
