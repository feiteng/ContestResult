class Solution {
public:
    vector<int> getStrongest(vector<int>& a, int k) {
        sort(a.begin(), a.end());
        int n = a.size();
        int med = a[(n - 1) / 2];
        
        vector<int> ans;
        int l = 0, r = n - 1;
        while (k--) {

            if (a[r] - med >= med - a[l]) {
                ans.push_back(a[r--]);
            }
            else {
                ans.push_back(a[l++]);
            }
        }
        
        return ans;
    }
};
