class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(),arr.end());
        int m=arr[(arr.size()-1)/2];
        vector<int> ans;
        int p=0,q=arr.size()-1;

        while(k){
            if(arr[q]-m>=m-arr[p]){
                ans.push_back(arr[q]);
                q--;
            }else{
                ans.push_back(arr[p]);
                p++;
            }
            k--;
        }
        return ans;
    }
};
