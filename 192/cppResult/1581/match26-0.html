int a=0;
bool comp(const int &x1, const int &x2) {
    if(abs(x1-a)==abs(x2-a)) return x1<x2;
    return abs(x1-a)<abs(x2-a);
}

class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n=arr.size();
        a=arr[(n-1)/2];
        sort(arr.begin(), arr.end(), comp);
        vector<int> ret;
        for(int i=n-1; i>=n-k; i--) ret.push_back(arr[i]);
        return ret;
    }
};
