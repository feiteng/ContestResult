class Solution {
public:
    vector<int> getStrongest(vector<int>& arr, int k) {
        sort(arr.begin(), arr.end());
        int n = arr.size();
        int med = arr[(n-1)/2];
        vector<int> res;
        for (int l = 0, r = n-1, i = 0; i < k; ++i) {
            if (abs(arr[l]-med) > abs(arr[r]-med)) {
                res.push_back(arr[l++]);
            } else {
                res.push_back(arr[r--]);
            }
        }
        return res;
    }
};
