class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int,int> cnts;
        for(int i = 0; i < arr.size(); ++i)
            cnts[arr[i]]++;
        vector<int> c;
        for(const auto &p : cnts)
            c.push_back(p.second);
        sort(c.begin(), c.end());

        for(int i = 0; i < c.size(); ++i) {
            k -= c[i];
            if(k < 0)
                return c.size() - i;
        }
        return 0;
    }
};
