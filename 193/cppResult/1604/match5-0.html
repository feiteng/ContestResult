class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> f;
        for (int a : arr) f[a]++;
        
        vector<int> g;
        for (auto e : f)
            g.push_back(e.second);
        sort(g.begin(), g.end());
        int ans = g.size();
        int p = 0;
        while (p < g.size()) {
            if (k < g[p]) break;
            k -= g[p];
            ans--;
            p++;
        }
        return ans;
    }
};
