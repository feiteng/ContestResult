class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int,int> mm;
        for(int i=0;i<arr.size();++i)mm[arr[i]]++;
        vector<int> v;
        for(auto x: mm)v.push_back(x.second);
        sort(v.begin(),v.end());
        int cnt=v.size();

        int i=0;
        int n=v.size();
        while(i<n&&k>0){
            if(v[i]>k)break;
            cnt--;
            k-=v[i];
            i++;
        }
        return cnt;
        
    }
};
