class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map< int, int > index;
        for( auto x: arr ) index[ x ] ++;
        vector< pair< int, int > > v;
        for( auto x : index ) 
        {
            v.push_back( { x.second, x.first } );
        }
        sort( v.begin(), v.end() );
        int goal = v.size();
        for( auto x: v )
        {
            if( k - x.first >= 0 )
            {
                k -= x.first;
                goal --;
            }
            else break;
        }
        return goal;
    }
};
