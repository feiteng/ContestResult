class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> m;
        for (int x : arr) m[x]++;
        vector<int> a;
        for (auto &it : m) a.push_back(it.second);
        int ans = a.size();
        sort(a.begin(), a.end());
        for (int x : a) {
            if (k >= x) {
                k -= x;
                ans--;
            }
        }
        return ans;
    }
};
