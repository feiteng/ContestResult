class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int,int> un;
        for(int i:arr){
            un[i]++;
        }
        vector<int> num;
        int cnt=0;
        for(auto i:un){
            num.push_back(i.second);
            cnt++;
        }
        sort(num.begin(),num.end());

        for(int i=0;i<num.size();i++){
            if(k>=num[i]){
                cnt--;
                k-=num[i];
            }
            else{
                break;
            }
        }
        return cnt;
    }
};
