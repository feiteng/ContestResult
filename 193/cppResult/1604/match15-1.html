class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> count;
        for (auto a : arr) {
            count[a]++;
        }
        vector<int> v;
        for (auto c : count) {
            v.push_back(c.second);
        }
        sort(v.begin(), v.end());

        for (int i = 0; i < v.size(); i++) {
            k -= v[i];
            if (k < 0) {
                return v.size() - i;
            }
        }
        return 0;
    }
};
