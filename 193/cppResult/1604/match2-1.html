class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> cnt;
        for(int num: arr) cnt[num] ++;
        
        vector<int> t;
        for(auto p: cnt) {
            t.push_back(p.second);
        }
        sort(t.begin(), t.end());
        
        int i = 0;
        while(i < t.size() && k >= t[i]) {
            k -= t[i];
            i ++;
        }
        return t.size() - i;
    }
};
