class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> mp;
        for (int num : arr) ++mp[num];
        vector<int> counts;
        for (auto p : mp) {
            counts.push_back(p.second);
        }
        sort(counts.begin(), counts.end());
        int i = 0;
        while (i < counts.size() && counts[i] <= k) {
            k -= counts[i];
            ++i;
        }
        return counts.size() - i;
    }
};
