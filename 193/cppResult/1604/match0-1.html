class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> m;
        for (auto a: arr) {
            m[a]++;
        }
        vector <int> v;
        for (auto a: m) {
            v.push_back(a.second);
        }
        sort(v.begin(), v.end());
        
//         for (int i = 0;  i < v.size(); i++) cout << v[i] << " ";
//         cout << endl;
        
        int i = 0;
        int n = v.size();
        while (i < n && v[i] <= k) {
            k -= v[i++];
        }
        return n-i;
    }
};
