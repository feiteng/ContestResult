class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        
        map<int,int> m;
        
        for(auto &it:arr){
            m[it]++;
        }
        
        vector<pair<int,int>> v;
        
        for(auto &it:m){
            v.push_back({it.second,it.first});
        }
        
        sort(v.begin(),v.end());
        
        int ans = v.size();
        
        for(int i=0;i<v.size();i++){
            if(k>=v[i].first){
                ans--;
                k-=v[i].first;
            }
            else{
                break;
            }
        }
        
        return ans;
    }
};
