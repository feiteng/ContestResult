class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> m;
        vector<int> cnt;
        
        for(int i: arr) m[i]++;
        for(auto& p: m) cnt.push_back(p.second);
        
        sort(cnt.begin(),cnt.end());
        
        int i = 0;
        
        while(k > 0)
        {
            if(cnt[i] <= k)
            {
                k -= cnt[i];
                i++;
            }
            else
            {
                break;
            }
        }
        
        return cnt.size()-i;
    }
};
