class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> mp;
        for (auto x : arr) mp[x]++;
        vector<pair<int, int>> list;
        for (auto p : mp) {
            list.push_back({p.second, p.first});
        }
        sort(list.begin(), list.end());
        int ans = list.size();
        for (auto p : list) {
            if (p.first <= k) {
                ans--;
                k -= p.first;
            } else {
                break;
            }
        }
        return ans;
    }
};
