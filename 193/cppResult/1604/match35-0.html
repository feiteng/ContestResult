class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> mp;
        for(int i = 0; i < arr.size(); ++i){
            mp[arr[i]]++;
        }
        vector<int> vec;
        for(auto it : mp){
            vec.push_back(it.second);
        }
        sort(vec.begin(), vec.end());
        int ans = 0;
        // for(int i = 0; i < vec.size(); ++i) cout << vec[i] << endl;
        for(int i = 0; i < vec.size(); ++i){
            if(k >= vec[i]){
                k -= vec[i];
            }else{
                ans = vec.size()-i;
                break;
            }
        }
        return ans;
    }
};
