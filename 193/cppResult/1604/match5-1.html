class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& a, int k) {
        unordered_map<int , int> cnt;
        for(int i : a)cnt[i]++;
        vector<pair<int , int>> b;
        for(auto& e : cnt){
            b.emplace_back(e.second , e.first);
        }
        sort(b.begin() , b.end());

        int i = 0;
        int c = 0;
        while(k && i < b.size()){
            if(k - b[i].first < 0)break;
            k -= b[i].first;
            ++i;
            ++c;
        }
        return b.size() - c;
    }
};
