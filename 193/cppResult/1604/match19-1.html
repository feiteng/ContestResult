class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> c;
        for(auto a: arr) {
            if(c.find(a) == c.end())
                c[a] = 0;

            c[a]++;
        }
        multimap<int, int> t;
        for(auto p: c) {
            t.insert({p.second, p.first});
        }
        
        
        int ans = t.size();
        for(auto p: t) {

            if(p.first <= k) {
                k -= p.first;
                ans--;
            }
            else {
                break;
            }
        }
        return ans;
    }
};
