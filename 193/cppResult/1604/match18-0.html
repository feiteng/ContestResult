class Solution {
public:
	int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
		map<int, int> M;
		for (int i : arr) ++M[i];
		vector<int> RM;
		for (auto i : M) RM.push_back(i.second);
		sort(RM.begin(), RM.end());
		int len = RM.size();
		for (int i = 0; i < len; ++i) {
			k -= RM[i];
			if (k < 0) return len - i;
		}
		return 0;
	}
};
