class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> v2cnt;
        for(auto x : arr) {
            v2cnt[x]++;
        } 
        vector<pair<int, int>> f;
        for(auto p : v2cnt) {
            f.push_back(pair<int, int>(p.second, p.first));
        }
        sort(f.begin(), f.end());
        k = arr.size() - k;
        if (k == 0) return 0;
        int t = 0;
        for(int i = f.size() - 1; i >= 0; i--) {
            t += f[i].first;
            //cout << t << " " << k << endl;
            if (t >= k) {
                return f.size() - i;
            }
        }
        return f.size();
    }
};
