class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int,int> cnt;
        vector<int> v;
        for (int i : arr) cnt[i]++;
        for (pair<int,int> p : cnt) {
            v.push_back(p.second);
        }
        sort(v.begin(),v.end());
        int cntr = 0;

        for (int i = 0; i < v.size(); i++) {
            if (k>=v[i]) {
                k-=v[i];
                cntr++;
            } else {
                break;
            }
        }
        return v.size()-cntr;
    }
};
