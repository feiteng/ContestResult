class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> mp;
        for(auto i : arr) {
            mp[i]++;
        }
        
        vector<pair<int,int>> v;
        for(auto i : mp) {
            v.push_back({i.second, i.first});

        }
        sort(v.begin(), v.end());
        int ans = v.size();
        for(auto i : v) {
            if(k >= i.first) {
                ans--;
                k -= i.first;
            } else {
                break;
            }
        }
        return ans;
        
    }
};
