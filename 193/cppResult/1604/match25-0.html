class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> mp;
        for (int num : arr) mp[num]++;
        priority_queue<int, vector<int>, greater<int>> que;
        for (auto it : mp) que.push(it.second);
        while (!que.empty() && k>0){
            k -= que.top();
            que.pop();
        }
        if (k==0) return que.size();
        return que.size()+1;
    }
};
