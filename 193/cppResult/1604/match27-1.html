class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> mp;
        for (auto it : arr) mp[it]++;
        vector<int> f;
        for (auto it : mp) f.push_back(it.second);
        sort(f.begin(), f.end());

        int ans = f.size();
        for (auto it : f) {
            if (it <= k) {
                k -= it;
                --ans;
            }
        }
        return ans;
    }
};
