class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int> cnt;
        for (int i: arr) {
            cnt[i]++;
        }
        vector<int> c;
        for (auto p: cnt) {
            c.push_back(p.second);
        }
        sort(c.begin(), c.end());
        int res=c.size();
        // cout << res << endl;

        for (int i=0; i<c.size(); ++i) {
            k-=c[i];
            // cout << k << endl;
            if(k>=0) {
                res--;
            } else {
                break;
            }
        }
        return res;
    }
};
