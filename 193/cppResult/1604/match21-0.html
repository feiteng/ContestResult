class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int,int> mp;
        for(int& i : arr) {
            mp[i]++;
        }
        vector<int> cnt;
        for(auto& [k, v] : mp) {
            cnt.push_back(v);
        }
        sort(cnt.begin(), cnt.end());
        int sum = 0, ans = 0;
        for(int& i : cnt) {
            sum += i;
            if(sum <= k) {
                ans++;
            }
            else {
                break;
            }
        }
        return mp.size() - ans;
    }
};
