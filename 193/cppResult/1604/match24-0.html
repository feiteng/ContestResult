class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map <int, int> s;
        s.clear();
        for (int i : arr) {
            s[i]++;
        }
        vector <int> v;
        for (auto it : s) {
            v.push_back(it.second);
        }
        sort(v.begin(), v.end());
        int ans = v.size();
        for (int i = 0; i < v.size(); i++) {
            if (v[i] <= k) {
                k -= v[i];
                ans--;
            }
        }
        return ans;
    }
};
