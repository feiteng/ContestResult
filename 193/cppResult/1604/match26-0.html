class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int, int> mappe;
        for (auto &x: arr){
            mappe[x]++;
        }
        vector<int> counts;
        for (auto &[x, y]: mappe){
            counts.push_back(y);
        }
        int ans = counts.size();
        sort(counts.begin(), counts.end());
        for (int i = 0; i < counts.size(); i++){
            if (counts[i] <= k){
                k -= counts[i];
                ans--;
            }
            else break;
        }
        return ans;
    }
};
