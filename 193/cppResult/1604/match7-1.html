class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int,int> M;
        for(int i=0;i<arr.size();++i) {
            M[arr[i]]++;
        }
        vector<pair<int,int> > V;
        for(auto it = M.begin();it != M.end(); ++it) {
            V.push_back(make_pair(it->second,it->first));
        }
        sort(V.begin(),V.end());
        int cnt = 0;
        for(int i=0;i<V.size();++i) {

            if( k >= V[i].first) {
                k-=V[i].first;
                ++cnt;
            }else {
                break;
            }
        }
        return V.size()-cnt;
    }
};
