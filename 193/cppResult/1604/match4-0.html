class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        map<int,int> mp;
        for(int a:arr) mp[a]++;
        vector<int> order;
        for(auto& p:mp) order.push_back(p.second);
        sort(begin(order),end(order));
        int ans=order.size(), index=0;
        while(k>0){
            if(order[index]>k) break;
            k-=order[index++];
            --ans;
        }
        return ans;
    }
};
