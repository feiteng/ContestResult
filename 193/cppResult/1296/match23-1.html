const int M=5e4+5;
const int MB=20;
int fa[M][MB];

class TreeAncestor {
public:
    TreeAncestor(int n, vector<int>& parent) {
        for(int i=0;i<n;i++)
            fa[i][0]=parent[i];
        for(int j=1;j<MB;j++)
            fa[0][j]=-1;
        for(int j=1;j<MB;j++)
            for(int i=1;i<n;i++)
            {
                if(fa[i][j-1]==-1) fa[i][j]=-1;
                else fa[i][j]=fa[fa[i][j-1]][j-1];

            }
    }
    
    int getKthAncestor(int node, int k) {
        for(int i=MB-1;i>=0;i--)
        {
            if(k&(1<<i))
                node=fa[node][i];
            if(node==-1) break;
        }
        return node;
    }
};
