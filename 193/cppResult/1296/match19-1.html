#define MAXN 50005
#define MAXLG 20

class TreeAncestor {
public:
    int p[MAXN][MAXLG];
    TreeAncestor(int n, vector<int>& parent)
    {
        memset(p, -1, sizeof p);
        for (int i = 0; i < n; i++)
        {
            int u = parent[i];

            p[i][0] = u;
        }
        for(int j = 1; j < MAXLG; j++)
        {
            for(int i = 0; i < n; i++)
            {
                if(p[i][j - 1] == -1)
                {
                    p[i][j] = -1;
                    continue;
                }
                p[i][j] = p[p[i][j - 1]][j - 1];

            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int b = MAXLG;
        while(b >= 0)
        {
            if(k & (1 << b))
            {
                node = p[node][b];
                if(node == -1)
                    return node;
            }
            b--;
        }
        return node;
    }
};
