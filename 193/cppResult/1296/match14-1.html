class TreeAncestor {
public:
    int hts;
    vector<vector<int>> table;

    TreeAncestor(int n, vector<int>& parent) {
        hts = (int)ceil(log2(n)); 
        table =  vector<vector<int>>(n+1, vector<int>(hts+1, -1));
        for(int i=0;i<n;i++){
            if(parent[i] != -1)
                helper(parent[i], i);
        }
    }


    void helper(int u, int v) 
    {   
        cout<<v<<" "<<u<<endl;
        table[v][0] = u; 
        for (int i = 1; i <= hts; i++) 
        { 
            table[v][i] = table[table[v][i - 1]][i - 1]; 
            if (table[v][i] == -1) 
                break; 
        } 
    } 
    
    int getKthAncestor(int V, int k) {
        // k--;
        for (int i = 0; i <= hts; i++) 
        { 
            if (k & (1 << i)) 
            { 
                V = table[V][i]; 
                if (V == -1) 
                    break; 
            } 
        } 
        return V;         
    }
};
