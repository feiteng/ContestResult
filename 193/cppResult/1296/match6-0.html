class TreeAncestor {
    int father[19][50001];
    int depth[100005];
public:
    TreeAncestor(int n, vector<int>& parent) {
        for(int i=0; i<n; i++)
        {
            father[0][i]=parent[i];
        }
        for(int i=1; i<19; i++)
        {
            for(int j=0; j<n; j++)
            {
                if(father[i-1][j]==-1)
                    father[i][j]=-1;
                else
                    father[i][j]=father[i-1][father[i-1][j]];
            }
        }   
    }
    
    int getKthAncestor(int node, int k) {
        int ans=node;
        int level=k;
        for(int i=0; i<19; i++)
        {
            if(level&(1<<i))
            {
                ans=father[i][ans];
                if(ans==-1)
                    return -1;
            }
        }
        return ans;
    }
};
