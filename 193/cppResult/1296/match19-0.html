class TreeAncestor {
public:
    int a[50000][18];
    TreeAncestor(int n, vector<int>& parent) {
        memset(a, -1, sizeof(a));
        for (int i = 0; i < n; ++i) {
            a[i][0] = parent[i];
        }
        for (int i = 1; i < 18; ++i) {
            for (int j = 0; j < n; ++j) {
                if (a[j][i - 1] == -1) a[j][i] = -1;
                else {
                    a[j][i] = a[a[j][i - 1]][i - 1];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int s = 0;
        while (k > 0) {
            if (k & 1) {
                node = a[node][s];
                if (node == -1) return -1;
            }
            s++;
            k >>= 1;
        }
        return node;
    }
};
