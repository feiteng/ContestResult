const int maxn=5e4+50;
const int N=30;

class TreeAncestor {
public:
    int fa[maxn][N];
    
    TreeAncestor(int n, vector<int>& parent) {
        memset(fa,-1,sizeof(fa));
        for(int i=0;i<n;++i) fa[i][0]=parent[i];
        for(int j=1;j<N;++j){
            for(int i=0;i<n;++i){
                int tmp=fa[i][j-1];
                if(tmp!=-1) fa[i][j]=fa[tmp][j-1];
            }
        }
    }

    int getKthAncestor(int node, int k) {
        for(int i=N-1;i>=0;--i){
            if(k>=(1<<i)){
                k-=(1<<i);
                node=fa[node][i];
                if(node==-1) return node;
            }
        }
        return node;
    }
};
