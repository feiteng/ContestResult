class TreeAncestor {
public:
    int dp[50001][33];
    TreeAncestor(int n, vector<int>& parent) {
        for(int i = 0; i < n; i++){
            dp[i][0] = parent[i];
        }
        for(int i = 0; i < n; i++){
            for(int j = 1; j < 32; j++){
                if(dp[i][j-1] == -1) dp[i][j] = -1;
                else dp[i][j] = dp[dp[i][j-1]][j-1];
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int l = 0;
        int curr_node = node;
        while(k > 0){
            if(k&1){
                curr_node = dp[curr_node][l];
                //cout << "node:" << node << " " << curr_node << endl;
                if(curr_node == -1) return -1;
            }
            k>>=1;
            l++;
        }
        return curr_node;
    }
};
