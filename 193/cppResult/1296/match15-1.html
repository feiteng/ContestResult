class TreeAncestor {
public:
    int dp[50004][30];
    TreeAncestor(int n, vector<int>& parent) {
        for(int j=0;j<parent.size();++j) {
            dp[ j ][ 0 ] = parent[j];
        }
        for(int i=1;(1<<i)<=parent.size();++i) {
            for(int j=0;j<parent.size();++j) {
               if(dp[ j ][ i-1 ] >= 0)
                dp[ j ][ i ] = dp [dp[ j ][ i-1 ]][i-1];
               else
                dp[ j ][ i ] = -1;
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int b = node;
        int now = 0;
        while(k) {
            if(k%2) {
                b = dp[b][now];
            }
            if(b==-1)
                return -1;
            now++;
            k/=2;
        }
        return b;
    }
};
