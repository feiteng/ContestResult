class TreeAncestor {
private:
    vector<vector<int>> store;
    int tn;
public:

    TreeAncestor(int n, vector<int>& parent) {    
        tn = (int)ceil(log2(n));
        store = vector<vector<int>>(n + 1, vector<int>(tn + 1, -1));
        //cout << " 0" << endl;
        for(int j = 1; j < parent.size(); ++j)
        {
            calculateSparse(parent[j], j);

        }
        //cout << " 1" << endl;
    }
    
    int getKthAncestor(int node, int k) {
        //cout << node << " " << k;
        for(int i = 0; i <= tn; ++i)
        {
            if(k & (1 << i))
            {
                node = store[node][i];
                if(node == -1) break;
            }
        }
        return node;
    }
    

    void calculateSparse(int u, int v) 
    { 
        store[v][0] = u; 
        for (int i = 1; i <= tn; i++) 
        { 
            store[v][i] = store[store[v][i - 1]][i - 1]; 
            if (store[v][i] == -1) 
                break; 
        } 
    }
};
