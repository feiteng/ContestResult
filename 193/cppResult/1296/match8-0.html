class TreeAncestor {
  int n;
  vector<array<int, 16>> parz;
public:
  TreeAncestor(int n, vector<int>& parent):n(n) {
    parz.resize(n);
    for (int i = 0; i < n; i++) {
      parz[i][0] = parent[i];
    }
    for (int i = 1; i < 16; i++) {
      for (int x = 0; x < n; x++) {
        int p = parz[x][i-1];
        if (p == -1) parz[x][i] = -1;
        else parz[x][i] = parz[p][i-1];
      }
    }
  }

  int getKthAncestor(int node, int k) {
    for (int i = 15; i >= 0 && node != -1; i--) {
      if (k & (1 << i)) {
        node = parz[node][i];
      }
    }
    return node;
  }
};
