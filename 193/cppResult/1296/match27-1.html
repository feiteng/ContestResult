class TreeAncestor {
public:
    int dp[50069][20];
    
    TreeAncestor(int n, vector<int>& parent) {
        for(int i = 0; i < 20; i++) {
            for(int j = 0; j < n; j++) {
                if(i == 0)
                    dp[j][i] = parent[j];
                else {
                    int par1 = dp[j][i - 1];

                    if(par1 == -1)
                        dp[j][i] = -1;
                    else
                        dp[j][i] = dp[par1][i - 1];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int curr = node;
        int pwr = 19;
        while(pwr >= 0) {
            int bit = (1 << pwr);
            if(k & bit) {
                curr = dp[curr][pwr];
                if(curr == -1)
                    return -1;
            } 
            pwr--;
        }
        return curr;
    }
};
