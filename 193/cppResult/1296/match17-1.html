class TreeAncestor {
public:
    
    vector<vector<int>> fa;
    
    TreeAncestor(int n, vector<int>& parent) {
        fa = vector<vector<int>>(n, vector<int>(20, -1));

        for (int i = 0; i < n; ++i) {
            fa[i][0] = parent[i];
        }
        for (int j = 1; j < 20; ++j) {
            for (int i = 0; i < n; ++i) {
                if (fa[i][j - 1] == -1) {
                    fa[i][j] = -1;
                }
                else {
                    fa[i][j] = fa[fa[i][j - 1]][j - 1];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int ans = node;

        for (int i = 0; i < 20; ++i) {
            if (k & (1 << i)) {
                ans = fa[ans][i];
                if (ans == -1) {
                    return ans;
                }
            }
        }
        return ans;
    }
};
