class TreeAncestor {
public:
    
    int parents[18][100001];


inline void add(int child, int parent) {
    parents[0][child] = parent;
    
    for (int i = 1; i < 18; i++) {
        parents[i][child] = parents[i - 1][parents[i - 1][child]];

        if (parents[i][child] == -1) break;
    }
}
    
    int kParent(int node, int k) {
    int current = node;
    for (int i = 0; i < 18; i++) {
        if (((1 << i) & k) != 0) {
            current = parents[i][current];

            if (current == -1) break;
        }
    }

    return current;
}

    TreeAncestor(int n, vector<int>& parent) {
        int i,j;
        for (int i = 0; i < 18; i++)
            for (int j = 0; j < 100001; j++)
                parents[i][j] = -1;
        cout<<"in"<<endl; 
        for(i=1;i<n;i++)
        {
            int par=parent[i];
            int chld=i;
            add(chld, par);
        }
        cout<<"out"<<endl;
        
        
    }
    
    int getKthAncestor(int node, int k) {
        int current = kParent(node, k);
        if (current == -1)
            return -1;
        else
            return current;
        
    }
};
