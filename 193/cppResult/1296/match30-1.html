class TreeAncestor {
public:
    vector< vector<int> > dp;
    TreeAncestor(int n, vector<int>& parent) {

        dp=vector( n,vector<int>(20,-1) );
        for(int i=0;i<n;i++){
            dp[i][0]=parent[i];
        }
        for(int k=1;k<=19;k++){
           for(int i=0;i<n;i++){
               if(dp[i][k-1]!=-1)
              dp[i][k]=dp[dp[i][k-1]][k-1];

           }
               
               }
    }
    
    int getKthAncestor(int node, int k) {
        int n=node;
       for(int i=19;i>=0;i--){
           if(k>=(1<<i)){
               k-=(1<<i);
               n=dp[n][i];
               if(n==-1)
                   return -1;
           }
           
       }
        return n;
    }
};
