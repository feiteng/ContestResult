#define H 18

class TreeAncestor {
private:
    vector<vector<int>> st_;
public:
    TreeAncestor(int n, vector<int>& parent) {
        st_ = vector<vector<int>>(n, vector<int>(H, -1));
        for (int i = 0; i < n; ++i) {
            st_[i][0] = parent[i];
        }
        for (int h = 1; h < H; ++h) {
            for (int i = 0; i < n; ++i) {
                int prev = st_[i][h - 1];
                if (prev != -1) {
                    st_[i][h] = st_[prev][h - 1];
                } else {
                    st_[i][h] = -1;
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int ans = node;
        for (int h = H - 1; h >= 0; --h) {
            if (k >= (1 << h)) {
                ans = st_[ans][h];
                k -= 1 << h;
                if (ans == -1) return ans;
            }
        }
        return ans;
    }
};
