class TreeAncestor {
public:
    vector<vector<int>> ancestors;
    TreeAncestor(int n, vector<int>& parent) {
        ancestors.assign(n, vector<int>(20, -1));
        for (int j = 0; j < n; ++j) {
            ancestors[j][0] = parent[j];
        }
        for (int i = 1; i < 20; ++i) {
            for (int j = 0; j < n; ++j) {
                if (ancestors[j][i - 1] != -1) {
                    ancestors[j][i] = ancestors[ancestors[j][i - 1]][i - 1];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        int i = 0, cur = node;
        while (k > 0 && cur != -1) {
            if (k & 1) {
                cur = ancestors[cur][i];
            }
            ++i;
            k >>= 1;
        }
        return cur;
    }
};
