class TreeAncestor {
    vector<vector<int>> anc;
public:
    TreeAncestor(int n, vector<int>& parent) {
        anc.resize(n);
        for (int i = 0; i < n; i++) {
            anc[i].resize(20);
            anc[i][0] = parent[i];
        }
        for (int p = 1 ; p < 20; p++) {
            for (int i = 0; i < n; i++) {
                if (anc[i][p - 1] == -1) {
                    anc[i][p] = -1;
                } else {
                    anc[i][p] = anc[anc[i][p - 1]][p - 1];    
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for (int i = 0; i < 20; i++) {
            if (k & (1 << i)) {
                node = anc[node][i];
            }
            if (node == -1) return -1;
        }
        return node;
    }
};
