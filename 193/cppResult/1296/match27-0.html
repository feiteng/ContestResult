class TreeAncestor {
public:
      int sp[50009][22];
    
    TreeAncestor(int n, vector<int>& par) {
        
        
        for (int i=0;i<22;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i==0)sp[j][i]=par[j];
                else 
                {
                    if(sp[j][i-1]!=-1)
                    sp[j][i]=sp[sp[j][i-1]][i-1];
                    else
                        sp[j][i]=-1;
                }
            }
        }

        
        
        
    }
    
    int getKthAncestor(int node, int k) {
        
        
    
     int ans=node;
        int bit=0;
        while(k)
        {
            if(ans==-1)break;
            if(k&1)ans=sp[ans][bit];
            
            
            bit++;
            k/=2;
            
        }
        
                
        return ans;
    }
};
