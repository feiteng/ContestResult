class TreeAncestor {
    int parentMap[50001][20];
public:
    TreeAncestor(int n, vector<int>& parent) {
        memset(parentMap, -1, sizeof(parentMap));
        

        for(int i = 1 ; i < n ; ++i){
            parentMap[i][0] = parent[i] ; 
        }

        for(int j = 1; (1<<j) < n ; ++j)
        {
            for(int i = 0 ; i < n ; ++i)
            {
                if(parentMap[i][j-1] != -1)
                {
                    parentMap[i][j] = parentMap[parentMap[i][j-1]][j-1] ; 
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
       // return -1;
        vector<int> steps;
        int reach = k;
        int index = 0;

        int cur = node;
        while(reach and cur!= -1)
        {
            if(reach&1)
            {
               cur = parentMap[cur][index];
            }
            reach = reach >> 1;
            ++index;
        }
        return cur;
    }
};
