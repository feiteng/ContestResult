class TreeAncestor {
public:
    int dp[61234][32];
    TreeAncestor(int n, vector<int>& parent) {
        memset(dp, -1, sizeof(dp));
        for (int i = 0; i < n; i++)
            dp[i][0] = parent[i];
        
        for (int h = 1; h < 30; h++) {
            for (int i = 0; i < n; i++)
                if (dp[i][h-1] != -1)
                    dp[i][h] = dp[dp[i][h-1]][h-1];
        }
        //cout << dp[6][0] << " " << dp[6][1] << " " << dp[6][2] << " " << dp[6][3] << endl;
    }
    
    int getKthAncestor(int node, int k) {
        //cout << node << " -- " << k << endl;
        for (int i = 29; i >= 0 && node != -1; i--) {
            if (dp[node][i] != -1  && (1<<i) <= k) {
                node = dp[node][i];
                k -= (1<<i);
                //cout << node << " " << k << " " << i << endl;
            }
        }
        if (k) return -1;
        return node;
    }
};
