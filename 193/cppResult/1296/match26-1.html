class TreeAncestor {
public:
    TreeAncestor(int n, vector<int>& parent) {
        m = log(n) / log(2) + 2;

        anc = vector<vector<int>>(n, vector<int>(m, -1));
        
        for (int i=0; i<n; ++i) 
            anc[i][0] = parent[i];
        for (int i=1; i<m; ++i) {
            for (int j=0; j<n; ++j) {
                int t = anc[j][i-1];
                anc[j][i] = t == -1 ? -1 : anc[t][i-1];

            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for (int i=m-1; i>=0; --i) {
            if (k & (1<<i)) {
                node = anc[node][i];
                if (node == -1) return node;
            }
        }
        return node;
    }
    
    vector<vector<int>> anc;
    int m;
};
