class TreeAncestor {
public:
    vector<vector<int>>table;
        

    TreeAncestor(int n, vector<int>& parent) {
        table = vector<vector<int>>(50001, vector<int>(21,-1));
        
        for(int v=0;v<n;v++){
            table[v][0] = parent[v]; 
            if(table[v][0]==-1)continue;

            for (int i = 1; i <= 20; i++){ 
                table[v][i] = table[table[v][i - 1]][i - 1]; 

                if (table[v][i] == -1) 
                    break; 
            } 
        }
        
    }
    
    int getKthAncestor(int node, int k) {
        for (int i = 0; i <= 20; i++){ 
            if (k & (1 << i)){ 
                node = table[node][i]; 
                if (node == -1) 
                    break; 
            } 
        } 
        return node; 
    }
};
