class TreeAncestor {
public:
    
    int ans[18][100001];
    int z=0;
    int o=1;
    int mm=-1;

inline void solve(int cc, int parent) {
    ans[0][cc] = parent;
    
    for (int i = o; i < 18; i++) {
        ans[i][cc] = ans[i - o][ans[i - o][cc]];

        if (ans[i][cc] == mm)
            break;
    }
}
    
    int funn(int node, int k) {
    int ctt = node;
    for (int i = z; i < 18; i++) {
        if (((1 << i) & k) != z) {
            ctt = ans[i][ctt];

            if (ctt == mm)
                break;
        }
    }

    return ctt;
}

    TreeAncestor(int n, vector<int>& parent) {
        int i,j;
        for (int i = z; i < 18; i++)
            for (int j = 0; j < 100001; j++)
                ans[i][j] = mm;
        for(i=1;i<n;i++)
        {
            int yy=parent[i];
            int dd=i;
            solve(dd, yy);
        }
        
        
    }
    
    int getKthAncestor(int node, int k) {
        int xx = funn(node, k);
        if (xx == mm)
            return mm;
        else
            return xx;
        
    }
};
