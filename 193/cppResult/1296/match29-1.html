class TreeAncestor {
private:
    vector<vector<int>> fa;
    
public:
    TreeAncestor(int n, vector<int>& parent) {
        fa.resize(n, vector<int>(16));

        for (int i = 0; i < n; ++i) {
            fa[i][0] = parent[i];
        }
        for (int j = 1; j < 16; ++j) {
            for (int i = 0; i < n; ++i) {
                if (fa[i][j - 1] == -1) {
                    fa[i][j] = -1;
                }
                else {
                    fa[i][j] = fa[fa[i][j - 1]][j - 1];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for (int i = 15; i >= 0; --i) {
            if (node != -1 && k >= (1 << i)) {
                node = fa[node][i];
                k -= (1 << i);
            }
        }
        return node;
    }
};
