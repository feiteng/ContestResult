class TreeAncestor {
public:

    TreeAncestor(int n_, vector<int>& parent) {
        n = n_;
        f.assign(K, vector<int>(n, -1));
        for(int i = 0; i < n; i++) {
            f[0][i] = parent[i];
        }
        for(int d = 1; d < K; d++) {
            for(int i = 0; i < n; i++) {
                if (f[d-1][i] == -1) continue;

                f[d][i] = f[d-1][f[d-1][i]];
            }
        }
    }
    
    int getKthAncestor(int v, int k) {
        int ret = v;
        for(int j = K - 1; ret >= 0 && j >= 0; j--) {
            if ((1<<j) & k) {
                ret = f[j][ret];
            }
        }
        return ret;
    }
    vector<vector<int>> f;
    int K = 16;
    int n;
};
