const int N = 5e4;
const int K = 20;
int f[N][K];

class TreeAncestor {
public:
    TreeAncestor(int n, vector<int>& parent) {
        for (int i = 0; i < n; ++i) f[i][0] = parent[i];
        for (int k = 1; k < K; ++k) {
            for (int i = 0; i < n; ++i) {
                int j = f[i][k - 1];
                f[i][k] = j < 0 ? -1 : f[j][k - 1];
            }
        }
    }
    
    int getKthAncestor(int x, int m) {
        for (int k = 0; k < K; ++k) {
            if (m & (1 << k)) x = f[x][k];
            if (x < 0) break;
        }
        return x;
    }
};
