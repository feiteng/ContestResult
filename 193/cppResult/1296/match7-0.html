class TreeAncestor {
public:
    int pa[50005][31];
    TreeAncestor(int n, vector<int>& parent) {
        memset(pa, -1, sizeof(pa));
        for(int i = 1; i < parent.size(); i++){
            pa[i][0] = parent[i];
        }
        for(int i = 1; i < parent.size(); i++){
            for(int j = 1; j < 32; j++){
                if(pa[i][j - 1] == -1) break;
                pa[i][j] = pa[pa[i][j - 1]][j - 1];
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for(int i = 30; i >= 0; i--){
            if(k & (1 << i)){
                node = pa[node][i];
                if(node == -1) return -1;
            }
        }
        return node;
    }
};
