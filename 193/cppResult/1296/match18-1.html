const int nax = 5e4 + 10;
int p[16][nax];
class TreeAncestor {

public:
    TreeAncestor(int n, vector<int>& parent) {
        for (int i=0;i<n;++i) {
            p[0][i] = parent[i];
        }
        for (int l=1;l<16;++l) {
            for (int i=0;i<n;++i) {
                if (p[l-1][i] == -1) {
                    p[l][i] = -1;
                } else {
                    p[l][i] = p[l-1][p[l-1][i]];
                }
            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for (int bit = 15; bit >= 0; --bit) {
            if ((k&(1 << bit))) {
                node = p[bit][node];
                if (node == -1) {
                    return -1;
                }
            }
        }
        return node;
    }
};
