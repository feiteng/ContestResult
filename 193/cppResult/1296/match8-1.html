class TreeAncestor {
private:

    int anc[50005][20];
public:
    TreeAncestor(int n, vector<int>& fa) {
        for (int i=0;i<n;i++) {
            anc[i][0]=fa[i];
        }
        for (int j=1;j<20;j++) {
            for (int i=0;i<n;i++) {

                if (anc[i][j-1]==-1) anc[i][j]=-1;
                else anc[i][j]=anc[anc[i][j-1]][j-1];
            }
        }
    }

    int getKthAncestor(int u, int k) {
        for (int j=19;j>=0;j--) {
            if (k>>j&1) {
                u=anc[u][j];
                if (u==-1) return -1;
            }
        }
        return u;
    }
};
