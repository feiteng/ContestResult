class TreeAncestor {
public:
    int inf;
    int st[50005][20];
    TreeAncestor(int n, vector<int>& parent) {
        parent[0] = n;

        inf = n;
        parent[n] = n;
        for(int i = 0; i <= n; i++) {
            st[i][0] = parent[i];
        }
        for(int i = 1; i <= 18; i++) {
            for(int j = 0; j <= n; j++) {
                st[j][i] = st[st[j][i - 1]][i - 1];

            }
        }
    }
    
    int getKthAncestor(int node, int k) {
        for(int i = 18; i >= 0; i--) {
            if(k >= (1 << i)) {
                node = st[node][i];
                k -= (1 << i);
            }
        }
        if(node == inf) return -1;
        return node;
    }
};
