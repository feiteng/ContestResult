const int N=1e6+10;
int n,m,x,y,k,s,cnt,f[N][21],d[N],h[N];
struct node{int to,next;}e[N<<1];
void dfs(int u,int fa)
{
    d[u]=d[fa]+1;
    f[u][0]=fa;
    for(int i=1;i<=20;i++)
        f[u][i]=f[f[u][i-1]][i-1];
    for(int i=h[u];i!=-1;i=e[i].next)
    {
        int v=e[i].to;
        if(v!=fa)dfs(v,u);
    }
}
int get_fa(int x,int k)
{
    int t=d[x]-k;
    for(int i=20;i>=0;i--)
    {
         if(f[x][i]==-1)
             return -1;
        if(d[f[x][i]]>t)x=f[x][i];
    }
    
    return f[x][0];
}
class TreeAncestor {
public:
    TreeAncestor(int n, vector<int>& p) {
        memset(h,-1,sizeof(h));
        for(int i=1;i<p.size();i++)
        {
            int x=i,y=p[i];
            e[cnt].to=y;
            e[cnt].next=h[x];
            h[x]=cnt++;
            e[cnt].to=x;
            e[cnt].next=h[y];
            h[y]=cnt++;
        }
        memset(f,0,sizeof(f));
        dfs(s,0);
        f[0][1]=-1;
    }
    
    int getKthAncestor(int node, int k) {
        return get_fa(node,k);
    }
};
