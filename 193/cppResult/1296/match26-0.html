class TreeAncestor {
  vector<vector<int>> p;
public:
  TreeAncestor(int n, vector<int>& parent) {
    p = vector<vector<int>>(20, vector<int>(n, -1));
    for (int i = 0; i < n; ++i) p[0][i] = parent[i];
    for (int i = 1; i < 20; ++i) {
      for (int j = 0; j < n; ++j) {
        if (p[i - 1][j] >= 0) p[i][j] = p[i - 1][p[i - 1][j]];
      }
    }
  }
    
    int getKthAncestor(int node, int k) {
      for (int i = 20; i-- > 0; ) {
        if (k & (1 << i)) {
          node = p[i][node];
          if (node < 0) return -1;
        }
      }
      return node;
    }
};
