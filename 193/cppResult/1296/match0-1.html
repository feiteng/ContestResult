class TreeAncestor {
public:
    int dp[500001][20];
    TreeAncestor(int n, vector<int>& parent) {
        memset(dp,-1,sizeof(dp));
        for(int i=0;i<n;++i)dp[i][0]=parent[i];
        for(int i=1;i<20;++i){
            for(int j=0;j<n;++j){
                if(dp[j][i-1]!=-1)
                dp[j][i]=dp[dp[j][i-1]][i-1];
            }

        }
    }
    
    int getKthAncestor(int node, int k) {
        for(int i=19;i>=0;--i){
            if(dp[node][i]!=-1&(1<<i)<=k){
                node=dp[node][i];
                k=k-(1<<i);
            }
        }
	   if(k!=0)return -1;
        return node;
    }
};
