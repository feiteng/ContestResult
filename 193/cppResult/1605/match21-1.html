class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {

        if(m * k > bloomDay.size()) return -1;
        int l = 1, r = 1e9;
        while(l <= r) {
            int mid = (l + r) >> 1;
            if(check(bloomDay, m, k, mid)) r = mid - 1;
            else l = mid + 1;
        }
        return r + 1;
    }
    bool check(vector<int>& bloomDay, int m, int k, int mid) {
        int countx = m, p = k;
        for(int i = 0; i < bloomDay.size(); i++) {
            if(bloomDay[i] <= mid) p--;
            else p = k;
            if(p == 0) {
                countx--;
                p = k;
            }
            if(countx == 0) return true;
        }
        return false;
    }
};
