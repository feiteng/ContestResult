class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int l = 0, r = 1000000007;
        vector<int> b = bloomDay;
        int n = b.size();
        if (m * k > n) {
            return -1;
        }
        vector<bool> bb;
        int mid = (l + r) / 2;
        while (l < r) {
            bb = vector<bool>(n, false);
            for (int i = 0; i < n; ++i) {
                if (b[i] <= mid) {
                    bb[i] = true;
                }
            }
            int count = 0;
            int nc = 0;
            for (int i = 0; i < n; ++i) {
                if (bb[i]) {
                    ++nc;
                    if (nc == k) {
                        ++count; 
                        nc = 0;
                    }
                } else {
                    nc = 0;
                }
            }
            if (count >= m) {
                r = mid;
            } else {
                l = mid + 1;
            }
            mid = (l + r) / 2;
        }
        return l;
    }
};
