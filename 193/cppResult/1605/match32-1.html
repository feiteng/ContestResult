class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if (m * k > n)
            return -1;
        
        vector<int> s(bloomDay);
        sort(s.begin(), s.end());
        s.erase(unique(s.begin(), s.end()), s.end());
        int lo = 0, hi = s.size()-1;
        while (hi > lo) {
            int mid = lo + (hi - lo) / 2;

            int day = s[mid];
            int bouquet = 0, flower = 0;
            for (int i = 0; i < n; i++) {
                if (bloomDay[i] <= day) {

                    if (++flower == k) {
                        flower = 0;
                        bouquet++;
                    }
                } else {
                    flower = 0;
                }
            }
            if (bouquet < m)
                lo = mid+1;
            else
                hi = mid;
        }
        return s[lo];
    }
};
