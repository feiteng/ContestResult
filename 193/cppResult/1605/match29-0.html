class Solution {
public:
    int minDays(vector<int>& a, int m, int k) {
        int n = a.size();
        if (m > n/k) return -1;//m*k>n
        int hi = 1e9 + 100;
        int lo = 0;
        while (hi-lo>1) {
            int mid=(hi+lo)/2;
            int have = 0;
            int cur=0;
            for (int i=0; i<n; i++) {
                if (a[i] <= mid) {
                    cur++;
                    if (cur==k) {
                        have++;
                        cur=0;   
                    }
                } else {
                    cur=0;
                }
            }
            if (have >= m) {
                hi = mid;
            } else {
                lo = mid;
            }
        }
        
        return hi;
    }
};
