class Solution {
public:
    bool check(int n, vector<int> & d, int m, int k) {
        int cnt = 0;
        int len = 0;
        for(int i = 0; i < d.size() ; i ++) {
            if(d[i] <= n) len ++;

            else {
                cnt += len / k;
                len = 0;
            }
        }
        cnt += len / k;
        return cnt >= m;
    }
    int minDays(vector<int>& day, int m, int k) {
        int res = -1;

        int left = 0;
        int right = 0x3f3f3f3f;
        while(left <= right) {
            int mid = (left + right) / 2;
            if(check(mid, day, m, k)) {
                res = mid;
                right = mid -1;
            } else {
                left = mid + 1;
            }
        }
        return res;
    }
};
