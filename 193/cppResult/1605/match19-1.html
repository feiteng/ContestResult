class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {

        int l = 0, r = 1000000001;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (check(bloomDay, m, k, mid)) {
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return l <= 1000000000 ? l : -1;
    }
    

    bool check(vector<int>& bloomDay, int m, int k, int d) {
        int c = 0;
        int s = 0;
        for (int i = 0; i < bloomDay.size(); ++i) {
            if (bloomDay[i] <= d) {
                ++c;
                if (c == k) {
                    ++s;
                    c = 0;
                }
            } else {
                c = 0;
            }
        }
        return s >= m;
    }
};
