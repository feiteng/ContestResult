class Solution {
public:
    int minDays(vector<int>& a, int m, int k) {
        int n=a.size();
        if(m*k>n)return -1;
        int l=0,r=1000000000;
        while(r-l>1){

            int md=(l+r)/2,nb=0,nf=0;
            for(int i=0;i<n;i++){
                if(a[i]<=md){
                    nf++;
                    if(nf==k){
                        nf=0;
                        nb++;
                    }
                }
                else nf=0;
            }
            if(nb>=m)r=md;
            else l=md;
        }
        return r;
    }
};
