const int lmt=1e9+2;
class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int b=1,e=lmt;
        while(b<e)
        {
            int mdl=(b+e)>>1;
            int cntBqts=0,cntFlwrs=0;
            for(int i=0;i<bloomDay.size();i++)
            {
                if(bloomDay[i]>mdl)cntFlwrs=0;
                else cntFlwrs++;
                if(cntFlwrs==k)
                {
                    cntBqts++;
                    cntFlwrs=0;
                }
            }
            if(cntBqts>=m)e=mdl;
            else b=mdl+1;
        }
        if(e==lmt)e=-1;
        return e;
    }
};
