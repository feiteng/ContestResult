class Solution {
public:
    
    bool check(vector<int>& b, int limit, int m, int k) {
        int cnt = 0;
        int kk = 0;
        int tk = 0;
        b.push_back(limit + 1);
        for(int i = 0; i < b.size(); i ++) {
            if(b[i] <= limit) {
                cnt ++;
            }else {
                kk += cnt / k;
                if(kk >= m) return true;
                cnt = 0;
            }
        }
        b.pop_back();
        return false;
    } 
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if(n < m * k) return -1;
        
        int l = 0, r = 1e9;
        while(l < r) {
            int mid = l + r >> 1;
            if(check(bloomDay, mid, m, k)) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
