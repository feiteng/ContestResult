class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        if(m * k >  bloomDay.size()) return -1;
        int left = 0, right = 1e9;
        while(left < right)
        {
            int mid = left + (right - left) / 2;
            int count = 0;
            int tmp = 0;
            for(int i = 0; i < bloomDay.size(); ++i)
            {
                if(mid < bloomDay[i])
                {
                    tmp = 0;
                    continue;
                }
                else
                {
                    tmp++;
                }
                if(tmp == k) 
                {
                    ++count;
                    tmp = 0;
                }
            }
            if(count >= m){
                right = mid;
            }
            else
            {
                left = mid + 1;
            }
        }
        return right;
    }
};
