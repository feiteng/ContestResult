class Solution {
public:
    bool ok(int mid, vector<int> &vec, int m, int k){
        int cnt = 0;
        int tt = 0;
        for(int i = 0; i < vec.size(); ++i){
            if(vec[i] <= mid){
                cnt++;
            }
            else{
                cnt = 0;
            }
            if(cnt == k) tt++, cnt = 0;
        } 

        return tt >= m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int l = 1, r = 1e9;
        int ans = -1;
        while(l <= r){
            int mid = l + r >> 1;
            if(ok(mid, bloomDay, m, k)){
                r = mid - 1;
                ans = mid;
            }
            else l = mid + 1;
        }
        return ans;
    }
};
