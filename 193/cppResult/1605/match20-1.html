class Solution {
public:

    int minDays(vector<int>& bloomDay, int m, int k) {
        int n=bloomDay.size();
        if(m*k>n)
            return -1;
        int lo=*min_element(bloomDay.begin(),bloomDay.end()),hi=*max_element(bloomDay.begin(),bloomDay.end()),mid;
        int ans=-1;
        while(lo<=hi)
        {
            int mid=lo+(hi-lo)/2;
            int cnt=0,grp=0;
            for(int i=0; i<n; i++)
            {
                if(bloomDay[i]<=mid)
                {
                    grp++;
                    if(grp==k)
                    {
                        cnt++;
                        grp=0;
                    }
                }
                else
                {
                    grp=0;
                }
            }
            if(cnt>=m)
            {
                ans=mid;
                hi=mid-1;
            }
            else
            {
                lo=mid+1;
            }
        }
        return ans;
    }
};
