class Solution {
public:
    const int INF = 1e9;
    bool check(int x, vector<int> a, int m, int k){
        a.push_back(x+1);
        int num = 0;
        for(int i=0; i<a.size(); i++){
            if(a[i]<=x) num++;
            else{
                m -= (num/k);
                num = 0;
            }
        }
        return m<=0;
    }
    int minDays(vector<int>& a, int m, int k) {
        int n = a.size();
        if(m*k>n) return -1;
        int l=1, r=INF;
        while(l<r){
            int mid = l+r>>1;
            if(check(mid, a, m, k)) r=mid;
            else l=mid+1;
        }
        return l;
    }
};
