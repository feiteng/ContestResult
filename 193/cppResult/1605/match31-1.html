class Solution {
private:

    int f(vector<int>& d, int k, int mid) {
        int answer = 0;
        int con = 0;
        for (int dd : d) {
            if (dd <= mid) {
                con++;
                if (con == k) {
                    con -= k;
                    answer++;
                } 
            } else con = 0;
        }
        return answer;
    }


public:
    int minDays(vector<int>& d, int m, int k) {
        int n = d.size();
        if (n < m * k) return -1;
        int left = 1;
        int right = 1000000000;
        while (left < right) {
            int mid = left + (right - left)/2;
            if (f(d, k, mid) < m) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return right;
    }
};
