class Solution {
public:
    bool ok(int day,vector<int>& a,int m,int k)
    {
        int cnt=0;
        int tmpCnt=0;
        int n=a.size();
        for(int i=0;i<n;++i)
        {
            if(a[i]<=day)
                tmpCnt++;
            else
                tmpCnt=0;
            if(tmpCnt==k)
                tmpCnt=0,cnt++;

        }
        return cnt>=m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n=bloomDay.size();
        if(n<m*k) return -1;
        int maxDay=*max_element(bloomDay.begin(),bloomDay.end());
        int minDay=*min_element(bloomDay.begin(),bloomDay.end());

        int l=minDay,r=maxDay;
        while(l<r)
        {
            int mid=(l+r)/2;
            if(ok(mid,bloomDay,m,k))
                r=mid;
            else
                l=mid+1;
        }
        return l;
    }
};
