class Solution {
public:
    int minDays(vector<int>& a, int m, int k) {
        int left = 1;
        int right = 1e9+1;
        int len = a.size();
        if(len<m*k) return -1;
        while(left <= right)
        {
            int mid = (left + right )/2;
            int num = 0;
            int cur = 0;
            for(int i=0;i<len;i++)
            {
                bool isOpen = a[i] <= mid;
                if(isOpen)
                {
                    cur++;
                    if(cur>=k)
                    {
                        cur = 0;
                        num ++;
                    }
                }
                else
                {
                    cur = 0;
                }
            }
            if(num>=m)
            {
                right = mid-1;
            }
            else
            {
                left = mid + 1;
            }
        }
        return left;
        
    }
};
