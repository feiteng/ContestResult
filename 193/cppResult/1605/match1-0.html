class Solution {
public:
    int find(int k, vector<int> &arr, int mid){
        int val = 0;
        int prev = -1;
        for(int i=0;i<arr.size();i++){
            if(arr[i]>mid){
                val += (i-prev-1)/k;
                prev = i;
            }
        }
        val += (arr.size()-prev-1)/k;
        return val;
    }
    
    int minDays(vector<int>& bloomDay, int m, int k) {
        int l = 1,r = *max_element(bloomDay.begin(),bloomDay.end());
        int ans = -1;
        while(l<=r){
            int mid = (l+r)/2;
            if(find(k,bloomDay,mid)>=m){
                ans = mid;
                r = mid-1;
            }
            else{
                l = mid+1;
            }
        }
        
        return ans;
    }
};
