class Solution {
public:
    bool check(const vector<int> &bloomDay, int m, int k, int d) {

        int c = 0;
        for(int i=0;i<bloomDay.size();++i) {
            if(bloomDay[i] <= d) {
                c++;
                if(c == k) {
                    m--;
                    c = 0;
                }
            }
            else {
                c = 0;
            }
        }
        return m <= 0;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        if(m * k > bloomDay.size()) {

            return -1;
        }
        
        int L = 0, H = *max_element(bloomDay.begin(), bloomDay.end());
        while(H - L > 1) {
            int M = (H + L) / 2;
            if(check(bloomDay, m, k, M)) {
                H = M;
            }
            else {
                L = M;
            }
        }
        return H;
    }
};
