class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        int l = 1, r = 1e9, ans = -1;
        while(l <= r) {
            int mid = (l+r)/2;
            int cnt = 0, flower = 0;
            for(int i = 0 ; i < n; i++) {
                if(mid >= bloomDay[i]) {
                    cnt++;
                    if(cnt >= k) {
                        flower++;
                        cnt = 0;
                    }
                } else {
                    cnt = 0;
                }
            }
            if(flower >= m) {
                ans = mid;
                r = mid-1;
            } else {
                l = mid+1;
            }
        }
        return ans;
    }
};
