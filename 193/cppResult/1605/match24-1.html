class Solution {
public:
    
    bool check(vector<int> &b, int m, int k, int d) {

        int cur = 0;
        for(int i = 0; i < b.size(); i++) {
            if(b[i] <= d) {
                cur++;
            } else {
                cur = 0;
            }
            if(cur >= k) {
                m--;
                cur = 0;
            }
        }
        return m <= 0;
    }
    

    int minDays(vector<int>& b, int m, int k) {
        int n = b.size();
        if(m * k > n) {
            return -1;
        }
        int l = 0, r = 1e9, res = r;
        while(l <= r) {

            int mid = (r - l) / 2 + l;
            if(check(b, m, k, mid)) {
                res = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return res;
    }
};
