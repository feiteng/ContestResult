class Solution {
public:
    bool judge(vector<int> &day, int mid, int m, int k){
        int cnt = 0;
        int res = 0;
        for (int i = 0; i < day.size(); i++){
            if (day[i] <= mid) cnt ++;
            else cnt = 0;
            if (cnt == k) {
                res++;
                cnt = 0;
            }
        }
        return res >= m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if (m * k > n) return -1;
        vector<int> nums(bloomDay);
        sort(nums.begin(), nums.end());
        int l = 0, r = n; 
        while(l < r){
            int mid = l + (r - l) / 2;
            
            if (judge(bloomDay, nums[mid], m, k)){
                r = mid;
            }
            else {
                l = mid + 1;
            }
        }
        return l == n ? -1 : nums[l];
    }
};
