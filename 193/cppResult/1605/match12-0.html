class Solution {
public:
    bool ispossible(vector<int> &b,int m,int k,int mid){
        int prev=-1;
        int c=0;
        for(int i=0;i<b.size();i++){
            if(b[i]>mid){
                c+=(i-1-prev)/k;
                prev=i;
            }
        }
        c+=(b.size()-1-prev)/k;
        return c>=m;
        
        
    }
    int minDays(vector<int>& b, int m, int k) {
        if(k*m>b.size())
            return -1;
        int s=1;
        int mx=b[0],e;
        for(auto  k : b)
             mx=max(mx,k);
        e=mx;
        int ans=e;
        while(s<=e){
            int mid=(s+e)/2;
            if(ispossible(b,m,k,mid)){
                ans=mid;
                e=mid-1;
            }else{
                s=mid+1;
            }
            
        }
        return ans;
    }
};
