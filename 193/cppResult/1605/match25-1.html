class Solution {
public:
    

    bool check(vector<int>& bd, int n, int m, int k, int val)
    {
        int lianxu =  0;
        for(int i = 0;i < n; ++i)
        {
            if(bd[i] <= val)
            {
                ++lianxu;
                if(lianxu == k)
                {
                    --m;
                    lianxu = 0;
                }
            }
            else 
            {

                lianxu = 0;
            }
        }
        
        return m <= 0;
    }
    
    int minDays(vector<int>& bd, int m, int k) {
        int n = bd.size();
        if(m * k > n) return -1;
        int l = 1, r = 1e9;
        while(l < r)
        {
            int mid = (r - l) / 2 + l;
            if(check(bd, n, m, k, mid)) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
