class Solution {
    bool solve(vector<int>& bloomDay, int m, int k, int mid)
    {
        int n = bloomDay.size();
        int prv = 0;
        for(int i = 0; i < n; i++)
        {
            if(bloomDay[i] <= mid)
            {
                prv++;
            }
            else
            {
                prv = 0;
            }
            if(prv >= k)
            {
                m--;
                prv -= k;
            }
        }
        return m <= 0;
    }
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if(n == 0 or (m * k) > n)
        {
            return -1;
        }
        int start = 1, end = 1000000000;
        int sol = -1;
        while(start <= end)
        {
            int mid = (start + end) / 2;
            bool can = solve(bloomDay, m, k, mid);
            if(can)
            {
                sol = mid;
                end = mid - 1;
            }
            else
            {
                start = mid + 1;
            }
        }
        return sol;
    }
};
