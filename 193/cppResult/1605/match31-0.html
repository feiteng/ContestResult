class Solution {
public:
	int minDays(vector<int>& bloomDay, int m, int k) {
		int len = bloomDay.size();
		if (m*k > len) return -1;
		int lo = 1; int hi = Mod + 2;
		while (lo < hi) {
			int mi = (lo + hi) >> 1;
			if (fun(bloomDay, m, k, mi)) {
				hi = mi;
			}
			else {
				lo = mi + 1;
			}
		}
		if (lo >= Mod) return -1;
		return lo;
	}
	bool fun(const vector<int>& b, int m, int k,int d) {
		int total = 0;
		int tmpStack = k;
		for (int i : b) {
			if (i - d <= 0) {
				tmpStack--;
				if (tmpStack == 0) {
					++total;
					tmpStack = k;
				}
			}
			else {
				tmpStack = k;
			}
		}
		return total >= m;
	}
};
