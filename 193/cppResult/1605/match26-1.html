class Solution {
public:
    int minDays(vector<int>& b, int m, int k) {
        int need = m * k;

        int lo = 0, hi = 1e9 + 1;
        while(hi - lo > 1){
            int mid = (hi + lo) / 2;
            int c = 0, mm = 0;
            for(int i = 0; i < b.size(); i++){
                if(b[i] <= mid) c++;
                else c = 0;
                if(c == k){
                    mm++;
                    c = 0;
                }
                
            }
            if(mm >= m) hi = mid;
            else lo = mid;
        }
        if(hi == 1e9 + 1) hi = -1;
        return hi;
    }
};
