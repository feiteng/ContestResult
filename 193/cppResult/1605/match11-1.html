class Solution {
public:
    
      bool isValid(vector<int> a, int size, int count, int day) {
          int curcount = 0, cursize = 0;
          for (int val : a) {

              if (val <= day) cursize++;
              else cursize = 0;
              if (cursize == size) { cursize = 0; curcount++; }
              if (curcount == count) return true;
          }
          return false;
      }
    
    int minDays(vector<int>& bloomDay, int m, int k) {
          if (bloomDay.size() <=0 || m == 0 || k == 0) return 0;
          if (m * k > bloomDay.size()) return -1;
          int l = INT_MAX, r = INT_MIN;
          for (int val : bloomDay) { l = min(l, val); r = max(r, val); }

          while (l <= r) {
              int mid = l + (r - l) / 2;
              if (isValid(bloomDay, k, m, mid)) r = mid - 1;
              else l = mid + 1;
          }
          return l;
    }
};
