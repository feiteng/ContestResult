class Solution {
public:
    int minDays(vector<int>& b, int m, int k) {
        if(m * k > b.size())
            return -1;
        int l = 1,h = *max_element(b.begin(),b.end());
        while(l <= h){
            int mid = (l + h)/2;
            int r = 0;
            int cur = 0;
            for(int i = 0;i < b.size();i++){
                if(b[i] <= mid)
                    cur++;
                else 
                    cur = 0;

                if(cur == k){
                    cur= 0;
                    r++;
                }
            }
            if(r >= m){
                h = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
};
