using LL = long long;

class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        // if ((LL)m * k > n) {
        //     return -1;
        // }
        int l = 1, r = 1000000000, ans = -1;
        while (l <= r) {
            int mid = (l + r) >> 1;
            int cnt = 0, cons = 0;
            for (int i = 0; i < n; ++i) {
                if (bloomDay[i] <= mid) {
                    ++cons;
                    if (cons == k) {
                        ++cnt;
                        cons = 0;
                    }
                }
                else {
                    cons = 0;
                }
            }
            if (cnt >= m) {
                ans = mid;
                r = mid - 1;
            }
            else {
                l = mid + 1;
            }
        }
        return ans;
    }
};
