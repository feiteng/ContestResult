class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n=bloomDay.size();
        if (m*k>n) return -1;
        int max=0;int min=INT_MAX;
        for (int i=0;i<bloomDay.size();i++){
            if (bloomDay[i]>max) max=bloomDay[i];
            if (bloomDay[i]<min) min=bloomDay[i];
        }
        while (min<max){
            int mid=(min+max)/2;
            int count=0;
            int temp=0;
            for (int i=0;i<bloomDay.size();i++){
                if (bloomDay[i]<=mid){
                    count++;
                    if (count>=k){
                        count-=k;
                        temp++;
                    }
                }
                else count=0;
            }
            if (temp<m) min=mid+1;
            else max=mid;
        }
        return min;
    }
};
