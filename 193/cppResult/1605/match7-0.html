class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int lo=-1;
        int hi=1000000003;
        while(hi-lo>1){
            int mid = (hi+lo)/2;
            int j=0;
            int l=0;
            for(int i=0; i<bloomDay.size(); i++){
                if(bloomDay[i] <= mid){
                    j++;
                    if(j==k){
                        j=0;
                        l++;
                    }
                }
                else{
                    j=0;
                }
            }
            if(l>=m){
                hi = mid;
            }else{
                lo = mid;
            }
        }
        if(hi==1000000003) return -1;
        return hi;
    }
};
