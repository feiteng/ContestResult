class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        if (m * k > bloomDay.size())
            return -1;
        int left = 1, right = 1000000000;
        while (left < right) {
            int middle = (left + right) / 2;
            int count = 0, adj = 0;
            for (int x : bloomDay) {
                if (x > middle) {
                    adj = 0;
                    continue;
                }
                if (++adj == k) {
                    ++count;
                    adj = 0;
                }
            }
            if (count >= m) {
                right = middle;
            } else {
                left = middle + 1;
            }
        }
        return left;
    }
};
