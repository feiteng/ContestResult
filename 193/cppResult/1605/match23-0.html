class Solution {
    
    int count(const vector<int> &v, int t, int k) {
        int cur = 0, res = 0;
        for (int i = 0; i < v.size(); ++i) {
            if (v[i] <= t) cur++;
            else {
                res += cur / k;
                cur = 0;
            }
        }
        res += cur / k;
        return res;
    }
    
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if (m * k > n) return -1;
        int left = 0, right = *max_element(bloomDay.begin(), bloomDay.end());
        while (right - left > 1) {
            int mid = ((right-left)>>1)+left;
            if (count(bloomDay, mid, k) >= m) right = mid;
            else left = mid;
        }
        return right;
    }
};
