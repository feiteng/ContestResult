class Solution {
public:
    bool can(vector<int>&bd, int m, int k, int d) {
        int tot=0;
        int cnt=0;
        for (int i=0; i<bd.size(); ++i) {
            if (bd[i]<=d) {
                cnt++;
                if (cnt==k) {
                    tot++;
                    cnt=0;
                }
            } else {
                cnt=0;
            }
        }
        return tot>=m;
    }
    int minDays(vector<int>& bd, int m, int k) {
        if ((long long)m*(long long)k > bd.size()) {
            return -1;
        }

        int lo=1, hi=1e9;
        while (lo<hi) {
            int mid=(lo+hi)/2;
            if (can(bd, m, k, mid)) {
                hi=mid;
            } else {
                lo=mid+1;
            }
        }
        return lo;
    }
};
