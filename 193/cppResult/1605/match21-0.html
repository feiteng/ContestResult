class Solution {
public:
    int minDays(vector<int>& b, int m, int k) {
        long n=b.size();
        if((long)m*k>n) return -1;
        int l=0,r=1e9;
        while(l<r){
            int mid=l+r>>1;
            if(ok(mid,b,m,k)) r=mid;
            else l=mid+1;
        }
        return l;
    }
    bool ok(int mid,vector<int>& b,int m,int k){
        int cnt=0,num=0;
        for(int i=0;i<b.size();i++){
            if(b[i]<=mid) num++;
            else num=0;
            if(k==num){
                cnt++;
                num=0;
            }
            if(cnt==m) return true;
        }
        return false;
    }
};
