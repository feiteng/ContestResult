class Solution {
public:
    bool isValid(vector<int>& bloomDay, int size, int count, int day) {
      int curcount = 0, cursize = 0,i;
      for(i=0;i<bloomDay.size();i++) 
      {
          int val=bloomDay[i];
          if (val <= day) cursize++;
          else cursize = 0;
          if (cursize == size) { cursize = 0; curcount++; }
          if (curcount == count) return true;
      }
      return false;
  }
    int minDays(vector<int>& bloomDay, int m, int k) {
        if (bloomDay.size() == 0 || m == 0 || k == 0) return 0;
          if (m * k > bloomDay.size()) 
              return -1;
          int l = INT_MAX, r = INT_MIN,i;
        for(i=0;i<bloomDay.size();i++)
        {
            int val=bloomDay[i];
            l =min(l, val); 
            r = max(r, val);
        }
          while (l <= r) {
              int mid = l + (r - l) / 2;
              if (isValid(bloomDay, k, m, mid))
                  r = mid - 1;
              else
                  l = mid + 1;
          }
          return l;
        
    }
};
