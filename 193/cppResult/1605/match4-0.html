class Solution {
public:
    int gotRoses(vector< int> & bloomDay, int idx, int m, int k) {
        int n = bloomDay.size();
        int curr1 = 0;
        int curr2 = 0;
        for (int i = 0 ; i < n ; i++) {
            if (bloomDay[i] <= idx) {
                curr1++;
            } else {
                curr1 = 0;
            }
            if (curr1 == k) {
                curr2++;
                curr1 = 0;
            }
        }
        return curr2 >= m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();
        if (m * k > n) {
            return -1;
        }
        int low = *min_element(bloomDay.begin(), bloomDay.end());
        int high = *max_element(bloomDay.begin(), bloomDay.end());
        while (low <= high) {
            // cout << "here for low = " << low << " and high = " << high << endl;
            int mid = low + (high - low) / 2;
            if (gotRoses(bloomDay, mid, m, k)) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
};
