class Solution {
public:
    bool canbe(vector<int>& b, int m, int k, int d) {
        int con = 0;
        int flo = 0;
        for (int i = 0; i < b.size(); ++i) {
            if (b[i] <= d) {
                con += 1;
                if (con >= k) {
                    ++flo;
                    con -= k;
                }
            } else {
                con = 0;
            }
        }
        return flo >= m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int beg = 0;
        int end = 1000000001;
        while (beg < end) {
            int mid = (beg + end) / 2;
            if (canbe(bloomDay, m, k, mid)) {
                end = mid;
            } else {
                beg = mid + 1;
            }
        }
        return canbe(bloomDay, m, k, end) ? end : -1;
    }
};
