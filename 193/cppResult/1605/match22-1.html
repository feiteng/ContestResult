class Solution {
public:
    int minDays(vector<int>& v, int m, int k) {
        
        int lo=0,hi=1000000001 ;
        int ans=hi ;
        while(lo<=hi)
        {
            int  mi=lo+(hi-lo)/2 ;
            int f=0 ;
            int c=0 ;
            
            for(int i=0;i<v.size();i++)
            {
                if(v[i]<=mi)
                {
                   c++ ;
                   if(c==k){ f++ ; c=0 ; }
                }
                else
                {
                    c=0 ;
                }
            }
            
            if(f>=m)
            {
                ans=mi ;
                hi=mi-1 ;
            }
            else lo=mi+1 ;
        }
        
        if(ans==1000000001) return -1 ;
        
        
        return ans ;
        
    }
};
