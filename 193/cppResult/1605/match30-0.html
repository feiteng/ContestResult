class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int n = bloomDay.size();

        int ng = 0, ok = inf;
        while (ng + 1 != ok) {
            int md = (ng + ok) / 2;
            
            int cn = 0, made = 0;
            rep(i, 0, n) {
                if (bloomDay[i] <= md) {
                    cn++;
                }
                else cn = 0;
                if (k <= cn) {
                    cn -= k;
                    made++;
                }
            }

            if (m <= made) ok = md;
            else ng = md;
        }

        if (ok == inf) ok = -1;
        return ok;
    }
};
