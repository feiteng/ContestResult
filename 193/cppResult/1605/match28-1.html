class Solution {
public:
    //k flowers, m bouquets
    bool valid(vector<int>& bloomDay, int m, int k, int time) {
        int bouq = 0;
        int count = 0;
        for (int i = 0; i < bloomDay.size(); i++)  {
            if (bloomDay[i] <= time) {
                count++;
            }
            else { 
                count = 0;
            }

            if (count >= k) {
                bouq++;
                count = 0;
            }
        }
        return bouq >= m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        int l = 0, r = 0;
        for (int& tmp : bloomDay) {
            r = max(tmp, r);
        }
        if (!valid(bloomDay, m, k, r)) { 
            return -1;
        }

        while (l < r) { 
            int mid = (l + r) / 2;
            if (valid(bloomDay, m, k, mid)) { 
                r = mid;
            }
            else { 
                l = mid + 1;
            }
        }
        return l;
    }
};
