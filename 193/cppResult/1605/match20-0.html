class Solution {
public:
    using ll = long long;
    int minDays(vector<int>& a, int M, int k) {
        int n = a.size();
        if (ll(M)*k > n) return -1;
        int L = 1, R = *max_element(a.begin(), a.end());
        int ret = R;
        while (L <= R)
        {
            int m = (L+R)/2;
            int cnt = 0, now = 0;
            for (int i = 0; i < n; ++i)
            {
                if (a[i] <= m)
                {
                    now ++;
                    if (now == k)
                    {
                        cnt ++;
                        now = 0;
                    }
                }
                else
                {
                    now = 0;
                }
            }
            if (cnt >= M)
            {
                ret = m;
                R = m-1;
            }
            else
                L = m+1;
        }
        return ret;
    }
};
