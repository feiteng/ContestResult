class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {

        int l = 0, r = 1e9 + 1;
        int n = bloomDay.size();
        while (l < r - 1) {
            int mid = (l + r) / 2;
            int curr = 0;
            int flower = 0;
            for (int i = 0; i < n; ++i) {
                if (bloomDay[i] <= mid) {
                    curr++;
                    if (curr >= k) {
                        flower++;
                        curr-=k;
                    }
                } else {
                    curr = 0;
                }
            }
            if (flower >= m) {
                r = mid;
            } else {
                l = mid;
            }
        }
        if (r == 1e9 + 1) return -1;
        return r;
    }
};
