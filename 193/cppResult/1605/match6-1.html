class Solution {
public:
    int minDays(vector<int>& bloomDay, int m, int k) {
        int l=1;
        int r=1000000001;
        while(l<r) {
            int mid=(l+r)/2;
            int curr=0;
            int res=0;
            for (int i=0; i<bloomDay.size(); i++) {
                if (bloomDay[i]<=mid) {
                    curr++;
                    if (curr==k) {
                        curr=0;
                        res++;
                    }
                } else {
                    curr=0;
                }
            }
            if (res>=m) {
                r=mid;
            } else {
                l=mid+1;
            }
        }
        if (l<=1000000000) {
            return l;
        }
        return -1;
    }
};
