class Solution {
public:
    int check(vector<int> &nums,int mid,int m,int k){
        int cnt = 0;
        int n = nums.size();
        int res = 0;
        for(int i=0;i<n;i++){
            if(nums[i]<=mid){
                cnt+=1;
            }
            else{
                cnt=0;
            }
            if(cnt>=k){
                res++;
                cnt = 0;
            }
        }
        return res;
    }
    int minDays(vector<int>& bd, int m, int k) {
        int n = bd.size();
        if((m*k)>n) return -1;

        int r = 1e9+1;
        int l = 0;
        while(l<r){
            int mid = (r-l)/2+l;
            if(check(bd,mid,m,k)>=m){r = mid;}
            else {l=mid+1;}
        }
        return l;
    }
};
