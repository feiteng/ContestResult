class Solution {
private:
    vector<int> arr;
public:

    bool C(int day, int m, int k) {
        int n = 0, t = 0;
        for(int i = 0; i < arr.size(); i++) {
            if(arr[i] <= day) {
                t++;
            } else {
                t = 0;
            }
            if(t == k) {
                n++;
                t = 0;
            }
        }
        return n >= m;
    }
    
    int minDays(vector<int>& bloomDay, int m, int k) {
        if(m * k > bloomDay.size()) return -1;

        arr = bloomDay;
        
        int l = 0, r = 1000000000;
        while(r - l > 1) {
            int mid = l + (r - l) / 2;
            if(C(mid, m, k)) {
                r = mid;
            } else {
                l = mid;
            }
        }
        return r;
    }
};
