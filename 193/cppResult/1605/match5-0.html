class Solution {
public:
    bool valid(vector<int>& bloomDay, int m, int k, int D){
        int c=0, sz=0;
        for(int i=0; i<bloomDay.size(); ++i){
            if(bloomDay[i]<=D){
                ++sz;
            }else{
                sz=0;
            }
            if(sz==k){
                ++c;
                sz=0;
            }
        }
        return c>=m;
    }
    int minDays(vector<int>& bloomDay, int m, int k) {
        if(k*m>bloomDay.size()) return -1;
        vector<int> bloomOrder = bloomDay;
        sort(begin(bloomOrder),end(bloomOrder));
        int l=0, r=bloomOrder.size()-1;
        while(l<r){
            int mid=l+(r-l)/2;
            if(valid(bloomDay, m, k, bloomOrder[mid])){
                r = mid;
            }else{
                l = mid+1;
            }
        }
        return bloomOrder[l];
    }
};
