template<class T>
constexpr T INF = ::numeric_limits<T>::max() / 32 * 15 + 208;

// class
class Solution {
public:
    vector<int> runningSum(vector<int>& nums) {
        for (int i = 1; i < nums.size(); ++i) {
            nums[i] += nums[i-1];
        }
        return nums;
    }
};
