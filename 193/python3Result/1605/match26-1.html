class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        def valid(n):
            bd = [i <= n for i in bloomDay]
            num, curr = 0, 0
            index = 0
            while index < len(bd):
                if bd[index]:
                    curr += 1
                else:
                    curr = 0
                if curr == k:
                    num += 1
                    curr = 0
                index += 1
            return num >= m
        if m * k > len(bloomDay):
            return -1
        mn, mx = min(bloomDay), max
(bloomDay)
        while mn < mx:
            mid = (mn+mx)//2
            if valid(mid):
                mx = mid
            else:
                mn = mid + 1
        return mn
