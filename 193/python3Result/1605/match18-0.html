class Solution:
    def minDays(self, blo: List[int], m: int, k: int) -> int:
        def valid(d):
            c, r = 0, 0
            for i, x in enumerate(blo):
                if x <= d:
                    c += 1
                    if c == k:
                        r += 1
                        c = 0
                        if r == m: return True
                else:
                    c = 0
            return False
        l, r = 0, max(blo)
        if not valid(r): return -1
        while l < r:
            mid = l + (r - l) // 2
            if not valid(mid):
                l = mid + 1
            else:
                r = mid
        return l
