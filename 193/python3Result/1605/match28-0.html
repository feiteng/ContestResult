def make(l,mid,m,k):
    tot = 0
    left = k
    for i in range(len(l)):
        if l[i] <= mid:
            left -= 1
        else:
            left = k
        if not left:
            tot += 1
            left = k
    return 1 if tot >= m else 0
class Solution:
    def minDays(self, l: List[int], m: int, k: int) -> int:
        n = len(l)
        if m * k > n:return -1
        
        low = 0
        ans = 10**9
        high = 10**9
        while low <= high:
            mid = (low + high) >> 1
            # print(low,high)
            if make(l,mid,m,k):
                ans = min(ans,mid)
                high = mid - 1
            else:
                low = mid + 1
        return ans
