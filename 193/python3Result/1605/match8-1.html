class Solution:
    def minDays(self, a: List[int], mm: int, k: int) -> int:
        if mm*k>len(a):
            return -1
        
        def chk(d):
            ans=0
            cnt=0
            for x in a:
                if x>d:
                    cnt=0
                    continue
                cnt+=1
                
if cnt==k:
                    ans+=1
                    cnt=0
            return ans>=mm
        
        l,r=-1,max(a)+1
        while r-l>1:
            m=(l+r)//2
            if chk(m):
                r=m
            else:
                l=m
        return r
