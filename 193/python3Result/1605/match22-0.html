class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        l, r = 1, 10 ** 9 + 10
        def check(x):
            t, res = 0, 0
            for i in range(len(bloomDay)):
                if bloomDay[i] <= x:
                    t += 1
                else:
                    t = 0
                if t == k:
                    res += 1
                    t = 0
            return res >= m
        while l < r:
            mid = (l + r) // 2
            if check(mid):
                r = mid
            else:
                l = mid + 1
        if r > max(bloomDay):
            return -1
        return r
