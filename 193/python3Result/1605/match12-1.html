class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        # corner case
        length = len(bloomDay)
        if m * k > length:
            return -1
        
        def check(mid: int) -> bool:
            nonlocal bloomDay, m, k
            counter = 0
            ret = 0
            for val in bloomDay:
                if val > mid:
                    counter = 0
                else:
                    counter += 1
                
                if counter == k:
                    ret += 1
                    counter = 0
                    
            
return ret >= m
        
        # binary search
        left, right = 1, max(bloomDay) + 1
        while left < right:
            mid = (left + right) // 2
            if check(mid):
                right = mid
            else:
                left = mid + 1
                
        return left
