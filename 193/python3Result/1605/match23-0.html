class Solution:
    def minDays(self, b: List[int], m: int, k: int) -> int:
        n = len(b)
        if m * k > n:
            return -1
        
        l = 0
        r = max(b)
        
        def check(mid):
            has = 0
            pre = 0
            for d in b:
                if d <= mid:
                    pre += 1
                else:
                    pre = 0
                if pre == k:
                    has += 1
                    pre = 0
            return has >= m
        
        while l < r:
            mid = (l + r) // 2
            if check(mid):
                r = mid
            else:
                l = mid + 1
        
        return l
