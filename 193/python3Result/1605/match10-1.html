class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        
        def count(day):
            res = 0
            cnt = 0
            for flower in bloomDay:
                if flower <= day:
                    cnt += 1
                    if cnt >= k:
                        cnt = 0
                        res += 1
                else:
                    cnt = 0
            return res

        # count(3)
        
        upper = 2**30
        lower = 0
        for i in range(30):
            middle = (upper + lower) // 2
            res = count(middle)
            
            # print(middle, res)
            
            if res >= m:
                upper = middle
            else:
                lower = middle
                
                
        if count(middle) < m:
            middle += 1
            
        if middle > 10**9 + 1:
            middle = -1
        
        # print()
        return middle
