class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        n=len(bloomDay)
        if m>n//k:
            return -1
        def isPossible(day):
            cur,cnt=0,0
            for v in bloomDay:
                if v>day:
                    cnt=0
                else:
                    cnt+=1
                if cnt==k:
                    cur+=1
                    cnt=0
            return cur>=m
        lo,hi=1,10**9
        while lo<=hi:
            mid=(lo+hi)//2
            if isPossible(mid):
                hi=mid-1
            else:
                lo=mid+1
        return lo
