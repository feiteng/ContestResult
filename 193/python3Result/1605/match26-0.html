class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        def isOK(day):
            M = m
            ptr = 0
            cur = 0
            while ptr < len(bloomDay):
                if bloomDay[ptr] > day:
                    cur = 0
                else:
                    cur += 1
                    if cur == k:
                        M -= 1
                        cur = 0
                        if M == 0:
                            return True
                ptr += 1
            return False
            
        if m * k > len(bloomDay):
            return -1
        left = min(bloomDay)
        right = max(bloomDay)
        while left < right:
            mid = (left + right) // 2
            if isOK(mid):
                right = mid
            else:
                left = mid + 1
        return left
