class Solution:
    def minDays(self, a: List[int], m: int, k: int) -> int:
        def min_days(k=k, n=m):
            if n == 0 or k == 0: return 0
            if n * k > len(a): return -1
            
            l = min(a)
            r = max(a)
            while l <= r:
                mid = l + (r - l) // 2
                if is_val(k, n, mid): r = mid - 1
                else: l = mid + 1
            return l
        
        def is_val(sz, cnt, day):
            cur = 0
            cursz = 0
            for x in a:
                if x <= day: cursz += 1
                else: cursz = 0
                if cursz == sz:
                    cursz = 0
                    cur += 1
                if cur == cnt: return True
            return False
        return min_days()
