class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        
        n = len(bloomDay)
        if m * k > n:
            return -1
        
        def can(x):
            prev = 0
            cnt = 0
            for i, num in enumerate(bloomDay):
                if num > x:
                    cnt += (i - prev) // k
                    prev = i + 1
            cnt += (n - prev) // k
            return cnt >= m
        
        l, r = 1, max(bloomDay)
        while l < r:
            mid = (l + r) // 2
            if can(mid):
                r = mid
            else:
                l = mid + 1
        return l
