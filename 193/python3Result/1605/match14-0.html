class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        N = len(bloomDay)
        if m * k > N:
            return -1
    
        
        def canMake(d):
            curr = 0
            total = 0
            for num in bloomDay:
                if num > d:
                    curr = 0
                else:
                    curr += 1
                    if curr == k:
                        total += 1
                        curr = 0
                    if total == m:
                        return True
            return False

        lo = min(bloomDay)
        hi = max(bloomDay)
        
        while lo < hi:
            mid = (lo + hi) // 2
            if canMake(mid):
                hi = mid
            else:
                lo = mid + 1
                
        return hi
