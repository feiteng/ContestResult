class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        left = 1
        right = max(bloomDay)
        
        if m * k > len(bloomDay):
            return -1
        
        def helper(mid,m,k):
            curSize = 0
            curCount = 0
            for val in bloomDay:
                
if val <= mid:
                    curSize += 1
                else:
                    curSize = 0 
                if curSize == k:
                    curSize = 0 
                    curCount += 1
                if curCount == m:
                    return True
            
            return False
            
        while left < right:
            mid = (left + right) // 2 
            if helper(mid,m,k):
                right = mid 
            else:
                left = mid+1
            
        return left
