from collections import Counter

def check(nums, x, m, k):
    flag = [True if _ <= x else False for _ in nums]
    tot 
= 0
    cur = 0
    for f in flag:
        if f:
            cur += 1
            if cur >= k:
                cur -= k
                tot += 1
        else:
            cur = 0
    return tot >= m

class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        if len(bloomDay) < m * k: return -1
        c = Counter(bloomDay)
        days = list(c.keys())
        days.sort()
        l = 0
        r = len(days) - 1
        
while l < r:
            mid = (l + r) // 2
            if check(bloomDay, days[mid], m, k):
                r = mid
            else:
                l = mid + 1
        return days[l]
