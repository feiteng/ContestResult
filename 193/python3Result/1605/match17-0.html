class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        def check(day):
            cur = 0
            remain = m
            for i in range(len(bloomDay)):
                if bloomDay[i] <= day:
                    cur += 1
                    if cur >= k:
                        cur -= k
                        remain -= 1
                        if remain <= 0:
                            return True
                else:
                    cur = 0
            return False
        left = 0
        right = max(bloomDay)
        if not check(right):
            return -1
        while left < right:
            mid = (left + right) // 2
            if check(mid):
                right = mid
            else:
                left = mid + 1
        return right
