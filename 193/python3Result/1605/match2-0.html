class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        if m*k > len(bloomDay):
            return -1
        l, r = 1, max(bloomDay)
        while l < r:
            mid = (l+r)//2
            count = 0
            temp = 0
            for i in range(len(bloomDay)):
                if bloomDay[i] <= mid:
                    temp += 1
                    if temp == k:
                        count += 1
                        temp = 0
                else:
                    temp = 0
            if count < m:
                l = mid+1
            else:
                r = mid
        return l
