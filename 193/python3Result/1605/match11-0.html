class Solution:
    def minDays(self, b: List[int], m: int, k: int) -> int:
        def f(mid):
            cnt = 0
            tmp = 0
            for bi in b:
                if bi <= mid:
                    tmp += 1
                    if tmp == k:
                        tmp = 0
                        cnt += 1
                else:
                    tmp = 0
            return cnt >= m
        l = 1; r = 10 ** 9 + 50
        while l < r:
            mid = l + (r - l) // 2
            if not f(mid):
                l = mid + 1
            else:
                r = mid
        return -1 if l == 10 **9 + 50 else l
