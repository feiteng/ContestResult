class Solution:
    def minDays(self, D: List[int], m: int, k: int) -> int:
        if m * k > len(D):
            return -1
        
        def cando(t):
            b = 0
            c = 0
            for d in D:
                if d > t:
                    c = 0
                if d <= t:
                    c += 1
                    if c == k:
                        b += 1
                        c = 0
                        if b >= m:
                            return True
            return False
        
        left = min(D)
        right = max(D)
        while left < right:
            mid = (left+right)//2
            if cando(mid):
                right = mid
            else:
                left = mid+1
        return left
