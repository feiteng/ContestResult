class Solution:
    def minDays(self, A: List[int], m: int, k: int) -> int:
        if len(A) < m * k:
            return -1
        
        
def bs(mid):
            cur = 0
            res = 0
            for x in A:
                if x <= mid:
                    cur += 1
                    if cur == k:
                        cur = 0
                        res += 1
                else:
                    cur = 0
            
            # print(mid, res)
            return res >= m
            
        l = 1
        r = 10 ** 9
        
        while l < r:
            mid = l + (r - l) // 2
            
            if not bs(mid):
                l = mid + 1
            else:
                r = mid
        
        return l
