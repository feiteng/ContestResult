class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        n = len(bloomDay)
        if n < m * k:
            return -1
        r = max(bloomDay)
        l = min(bloomDay)
        while l <= r:
            day = (l+r) // 2
            cnt = 0
            ck = 0
            for i in range
(n):
                if day >= bloomDay[i]:
                    ck += 1
                    if ck == k:
                        ck = 0
                        cnt += 1
                else:
                    ck = 0
            if cnt < m:
                l = day+1
            else:
                r = day-1
        return l
