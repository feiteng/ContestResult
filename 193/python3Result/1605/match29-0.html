class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        lo, hi = min(bloomDay), max(bloomDay)
        def f(n):
            ret = 0
            s = 0
            for i, x in enumerate(bloomDay):
                if x > n:
                    s = 0
                else:
                    s += 1
                if s == k:
                    ret += 1
                    s = 0
                if ret >= m: return True
            return False
        while lo < hi - 1:
            mi = (lo + hi) // 2
            if f(mi):
                hi = mi
            else:
                lo = mi
        while f(hi):
            hi -= 1
        ans = hi + 1
        return -1 if ans > max(bloomDay) else ans
