class Solution:
    def minDays(self, bloom: List[int], m: int, k: int) -> int:
        n = len(bloom)
        if m * k > n:
            return -1
        
        def check(v):
            cnt = 0
            score = 0
            for i in range(n):
                if bloom[i] <= v:
                    cnt += 1
                else:
                    cnt = 0
                if cnt == k:
                    cnt = 0
                    score 
+= 1
            return score >= m
        
        ng = 0
        ok = 10 ** 9
        while ok - ng > 1:
            mid = (ng + ok) // 2
            if check(mid):
                ok = mid
            else:
                ng = mid
        
        return ok
