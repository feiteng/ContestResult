class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        n = len(bloomDay)
        if n < m * k:
            return -1

        def is_ok(val):
            n_bouquets = 0
            cur_length = 0
            for i, x in enumerate(bloomDay + [float('inf')]):
                if x <= val:  # bloom
                    cur_length += 1
                else:
                    n_bouquets += cur_length // k
                    if n_bouquets >= m:
                        return True
                    cur_length = 0
            return n_bouquets >= m

        l, r = 0, max(bloomDay) + 1
        while r - l > 1:
            mid = (l + r) // 2
            if is_ok(mid):
                r = mid
            else:
                l = mid
        return r if is_ok(r) else -1
