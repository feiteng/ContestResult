class Solution:
    def minDays(self, a: List[int], m: int, k: int) -> int:
        n = len(a)
        if n < m * k:
            return -1
        
        def check(t):
            counter = 0
            accum = 0
            for i in range(n):
                if t >= a[i]:
                    accum += 1
                else:
                    accum = 0
                if accum == k:
                    accum = 0
                    counter += 1
            
return counter >= m

        l, r = 1, max(a)
        while l < r:
            mid = (l + r) // 2
            if check(mid):
                r = mid
            else:
                l = mid + 1
        return l
