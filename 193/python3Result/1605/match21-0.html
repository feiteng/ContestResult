class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        if m*k>len(bloomDay): return -1
        lo,hi=1,max(bloomDay)
        while lo<hi:
            mi=(lo+hi)//2
            a,s,t=[1 if x<=mi else 0 for x in bloomDay],0,0
            a.append(0)
            for x in a:
                if x:
                    t+=1
                    if t==k: s,t=s+1,0
                else: t=0
            if s<m: lo=mi+1
            else: hi=mi
        return hi
