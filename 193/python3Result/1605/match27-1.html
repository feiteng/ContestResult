class Solution:
    def minDays(self, bloomDay: List[int], m: int, k: int) -> int:
        def solve(bloomDay, mid, m, k):
            ans,cnt = 0,0
            for i in range(len(bloomDay)):
                if bloomDay[i] <= mid:
                    cnt += 1
                else:
                    ans += cnt // k
                    cnt = 0
            ans += cnt // k
            
return ans >= m
        if m * k > len(bloomDay): return -1
        L,R = min(bloomDay),max(bloomDay)
        while L<R:
            mid = (L+R)//2
            if solve(bloomDay, mid, m, k): R = mid
            else: L = mid+1
        return R
