class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        from collections import Counter
        c = list(Counter(arr).items())
        c.sort(key=lambda a: a[1])
        
        res 
= len(c)
        for v, num in c:
            if num<=k:
                k-=num
                res-=1
            else:
                break

        return res
