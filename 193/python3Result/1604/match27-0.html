from collections import Counter

class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        values = list(Counter(arr).values())
        values.sort()
        ncount = len(values)
        for v in values:
            if (v > k):
                return ncount
            k -= v
            ncount -= 1
        return 0
