class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        cnt = collections.Counter(arr)
        t = sorted([[v, c] for c, v in cnt.items()])
        res = len(t)
        for v, c in t:
            if k >= v:
                k -= v
                res -= 1
            else: break
        return res
