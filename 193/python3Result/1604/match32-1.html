class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        m = {}
        for val in arr :
            m[val] = m.get(val, 0) + 1
        a = sorted(m.items(), key=lambda x: (x[1], x
[0]))
        #print(a)
        res = len(a)
        for x in a :
            if k >= x[1] :
                k -= x[1]
                res -= 1
            else :
                break
        return res
