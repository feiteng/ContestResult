from collections import Counter
class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        c = Counter(arr)
        counts = list(c.values())
        counts.sort(reverse=True)
        while k > 0:
            n = counts.pop()
            k -= n
        ans = len(counts)
        if k < 0:
            ans += 1
        return ans
