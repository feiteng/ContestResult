class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        counter = Counter(arr)
        arr2 = counter.most_common()[::-1]
        res = len(arr2)
        for key, count in arr2:
            if k >= count:
                res -= 1
                k -= count
        return res
