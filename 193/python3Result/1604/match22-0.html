class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], lim: int) -> int:
        cnt = defaultdict(int)
        for x in arr:
            cnt[x] += 1
        
        vals = []
        for k, v in cnt.items():
            vals.append((k, v))
        
        vals = sorted(vals, key=lambda x: x[1])
        
        sz = len(vals)
        for k, v in vals:
            if v > lim:
                break
            
            sz -= 1
            lim -= v
        return sz
