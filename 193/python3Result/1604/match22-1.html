from collections import defaultdict


class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        d = defaultdict(int)
        for val in arr:
            d[val] += 1
        
        pairs = []
        for val, cnt in d.items():
            pairs.append((cnt, val))
        
        pairs = sorted(pairs)
        scnt = 0
        n = len(pairs)
        for idx in range(n):
            if scnt + pairs[idx][0] <= k:
                scnt += pairs[idx][0]
            else:
                return n - idx
        
        return 0
