from heapq import heapify, heappop, heappush



class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        hm = {}
        for a in arr:
            hm[a] = 1 + hm.get(a, 0)

        pq = list(hm.values())
        heapify(pq)
        for _ in range(k):
            cnt = heappop(pq)
            if cnt != 1:
                heappush(pq, cnt-1)

        return len(pq)
