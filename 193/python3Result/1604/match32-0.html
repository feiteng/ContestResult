from collections import *
class Solution:
    def findLeastNumOfUniqueInts(self, l: List[int], k: int) -> int:
        
        cnt = Counter(l)
        l1 = sorted(cnt,key = lambda x:cnt[x])
        ans = len(l1)
        for i in l1:
            if cnt[i] <= k:
                k -= cnt[i]
                ans -= 1
            else:break
        return ans
