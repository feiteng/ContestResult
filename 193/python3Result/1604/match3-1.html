class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        c = collections.Counter
(arr)
        freq = c.most_common()[::-1]
        ans = len(freq)
        for p, q in freq:
            if k >= q:
                ans -= 1
                k -= q
        return ans
