from collections import defaultdict
class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        cnt = defaultdict(int)
        for x in arr:
            cnt[x] += 1
            
        d = sorted
(cnt.values())
        
        i = 0
        while i < len(d) and k - d[i] >= 0:
            k -= d[i]
            i += 1
        
        return len(d) - i
