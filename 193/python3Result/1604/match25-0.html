class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        count = Counter(arr)
        arr = sorted(count.items(), key=lambda x: x[1])
        ans = len(arr)
        for _, cnt in arr:
            if k < cnt:
                break 
            k -= cnt
            ans -= 1 
        return ans
