class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        d = {}
        for a in arr:
            if a in d:
                d[a] += 1
            else:
                d[a] = 1
        
        d2 = []
        for di in d:
            d2.append([d[di], di])
        
        d2.sort()
        n = len(d2)
        idx = 0
        while idx <len(d2) and k >= d2[idx][0]:
            k -= d2[idx][0]
            idx += 1
            n -= 1
        return n
