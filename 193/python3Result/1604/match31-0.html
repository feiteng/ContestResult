class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        data = collections.Counter(arr)
        mid = sorted(data.items(), key=lambda item: item[1])
        cnt = 0
        for a, b in mid:
            if k >= b:
                k -= b
                cnt += 1
            else:
                break
        return len(mid) - cnt
