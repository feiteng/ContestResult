class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        ctr = collections.Counter(arr)
        l = sorted(ctr.most_common(), key=lambda x: x[1])
        i = 0
        ans = len(l)
        while k:
            d = min(k, l[i][1])
            k -= d
            if d == l[i][1]:
                ans -= 1
            i += 1
        return ans
