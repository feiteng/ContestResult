class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        d = {}
        for i in arr:
            if i in d:
                d[i] += 1
            else:
                d[i] = 1
        r = []
        for i in d:
            r.append([i, d[i]])

        r.sort(key = lambda x: x[1])
        prefix = []
        for i in r:
            prefix.append(i[1])
        if k >= sum(prefix):
            return 0
        else:
            i = 0
            while k >= 0 and i < len(prefix):
                k -= prefix[i]
                i += 1
            return len(prefix) - i + 1
