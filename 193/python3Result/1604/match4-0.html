from collections import Counter

class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        ac=Counter(arr)
        ans=len(ac)
        for c in sorted(ac.values()):
            if k-c<0:break
            k-=c
            ans-=1
        return ans
