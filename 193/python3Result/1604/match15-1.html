from collections import Counter
class Solution:   
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        cnt = Counter(arr)
        nums = sorted
(cnt.values())
        removed = 0
        for num in nums:
            k -= num
            if k >= 0:
                removed += 1
        return len(nums) - removed
