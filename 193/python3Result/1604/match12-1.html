class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        counter = collections.Counter(arr)
        pairs = list(counter.items
())
        pairs.sort(key=lambda x : x[1])
        removed = 0
        for _, cnt in pairs:
            if k >= cnt:
                k -= cnt
                removed += 1
            else:
                break
        return len(pairs) - removed
