from collections import Counter

from operator import itemgetter
class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        c = list(Counter(arr).items())
        c.sort(key=itemgetter(1))
        ans = len(c)
        for _, n in c:
            if k < n:
                break
            k -= n
            ans -= 1
        return ans
