from collections import Counter
class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        l = Counter(arr).most_common()
        n 
= len(l)
        for i in range(n-1, -1, -1):
            if k >= l[i][1]:
                k -= l[i][1]
            else:
                return n-(n-1-i)
        return 0
