class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        m = {}
        for item in arr:
            if item in m:
                m[item] += 1
            else:
                m[item] = 1
        l = list(m.items())
        l.sort(key=lambda x: x[-1])
        r = len(l)
        for i in range(len(l)):
            if k >= l[i][-1]:
                k -= l[i][-1]
                r -= 1
            else:
                break
        return r
