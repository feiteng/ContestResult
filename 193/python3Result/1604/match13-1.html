class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        c = collections.Counter(arr)
        d = sorted(c.items(), key=lambda x: x[1])
        n = len(d)
        for i, j in d:
            k -= j
            if k >= 0:
                n -= 1
            if k <= 0:
                return n
