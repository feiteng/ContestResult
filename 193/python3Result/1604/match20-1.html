class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        dic = {}
        for x in arr:
            if x in dic:
                dic[x] += 1
            else:
                dic[x] = 1
        arr
.sort(key = lambda x: (dic[x], x))
        # print(arr)
        return len(set(arr[k:]))
