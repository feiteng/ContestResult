class Solution:
    def findLeastNumOfUniqueInts(self, arr: List[int], k: int) -> int:
        d = {}
        for a in arr:
            if a in d:
                d[a] += 1
            else:
                d[a] = 1
        ls = [d[x] for x in d]
        ls.sort
()
        ret = len(ls)
        for i in range(len(ls)):
            if k >= ls[i]:
                k -= ls[i]
                ret -= 1
            else:
                break
        return ret
