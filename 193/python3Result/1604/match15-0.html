class Solution:
    def findLeastNumOfUniqueInts(self, a: List[int], n: int) -> int:
        d = Counter(a)
        b = d.most_common()
        b.reverse()
        cnt = 0
        for k, v in b:
            n -= v
            if n >= 0:
                cnt += 1
        return len(b) - cnt
