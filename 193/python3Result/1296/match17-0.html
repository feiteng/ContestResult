class TreeAncestor: 
  def __init__(self, n, arr): 
    self.h = int(math.ceil(math.log2(n))) 
    self.dp = [0] * (n + 1) 
    self.preprocess(arr)
    
  def make_connection(self, u, v): 
    self.dp[v][0] = u 
    for i in range(1, self.h + 1):
      self.dp[v][i] = self.dp[self.dp[v][i - 1]][i - 1] 
      if self.dp[v][i] == -1: break

  def preprocess(self, arr):
    for i in range(len(self.dp)): 
      self.dp[i] = [-1] * (self.h + 1) 

    for p, el in enumerate(arr):
      if el != -1:
        self.make_connection(el, p)
        
  def getKthAncestor(self, V, k): 
    for i in range(self.h + 1): 
      if k & (1 << i) != 0: 
        V = self.dp[V][i] 
        if V == -1: break
      
    return V
