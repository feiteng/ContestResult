class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.n = len(parent)
        self.p = [[-1 for i in range(self.n)] for j in range(20)]
        for i in range(self.n):
            self.p[0][i] = parent[i]
        for i in range(1, 20):
            for j in range(self.n):
                
if self.p[i - 1][j] == -1:
                    self.p[i][j] = -1
                else:
                    self.p[i][j] = self.p[i - 1][self.p[i - 1][j]]
        # print(self.p[1][5])

    def getKthAncestor(self, node: int, k: int) -> int:
        for i in range(20, -1, -1):
            if k >= (1 << i):
                k -= 1 << i
                # print(node, i, node)
                node = self.p[i][node]
                if node == -1:
                    return node
        return node
