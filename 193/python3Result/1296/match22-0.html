class TreeAncestor:
    
    
    def __init__(self, n: int, parent: List[int]):
        self.pa=parent
        self.G=collections.defaultdict(list)
        for i in range(1,n):
            self.G[i].append(parent[i])
            self.G[parent[i]].append(i)
        self.dep=[0]*(n+1)
        self.f=[[-1 for i in range(30)]for i in range(n+1)]
        self.dfs(0,-1)
        
    def dfs(self,u,fa):
        self.dep[u]=self.dep[fa]+1
        self.f[u][0]=fa
        for i in range(1,21):
            self.f[u][i]=self.f[self.f[u][i-1]][i-1]
        for v in self.G[u]:
            if v!=fa:
                self.dfs(v,u)

    def getKthAncestor(self, node: int, k: int) -> int:
        t=self.dep[node]-k
        for i in range(20,-1,-1):
            if self.dep[self.f[node][i]]>t:
                node=self.f[node][i]
        return self.f[node][0]
