class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        maxlogv=20
        self.parent = [[0] * n  for _ in range(maxlogv)]
        self.depth = [0] * n
        
        
        g = collections.defaultdict(list)
        
        for i in range(n):
            g[parent[i]] += [i]
            g[i] += [parent[i]]
        def dfs(v, p, d):
            # print (self.parent, v)
            self.parent[0][v] = p
            self.depth[v] = d
            for nxt in g[v]:
                if nxt != p:
                    dfs(nxt, v, d + 1)
        
        dfs(0, -1, 0)
        for k in range(maxlogv - 1):
            for v in range(n):
                if self.parent[k][v] < 0:
                    self.parent[k +1][v] = -1
                else:
                    self.parent[k + 1][v] = self.parent[k][self.parent[k][v]]
                    
            

    def getKthAncestor(self, node: int, k: int) -> int:
        
        for i in range(30, -1, -1):
            if (k & (1 << i)):
                # print (node)
                node = self.parent[i][node]
                if node == -1:
                    break
                # print (node,  i)
        
        
        return node
