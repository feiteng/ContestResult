from collections import *

import math
class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.height = int(math.log2(n+5))

        # Table for storing 2^ith parent
        self.table = [0] * (n + 5)
        
        
        
        self.preprocessing()
        for i in range(len(parent)):
            self.calculateSparse(parent[i] + 1,i + 1)
            
    def preprocessing(self):
        i = 0
        while (i < len(self.table)):
            self.table[i] = [-1] * (self.height + 1)
            i = i + 1

    def calculateSparse(self, u, v):

        # Using the recurrence relation to
        # calculate the values of table[][]
        self.table[v][0] = u
        i = 1
        while (i <= self.height):
            self.table[v][i] = self.table[self.table[v][i - 1]][i - 1]

            # If we go out of bounds of the tree
            if (self.table[v][i] == -1):
                break
            i = i + 1
    def kthancestor(self, V, k):
        i = 0

        # Doing bitwise operation to
        # check the set bit
        while (i <= self.height):
            if ((k & (1 << i)) != 0):
                V = self.table[V][i]
                if (V == 0):
                    break
            i = i + 1

        return V
    def getKthAncestor(self, node: int, k: int) -> int:
        temp = self.kthancestor(node+1,k)
        return temp - 1 if temp != -1 else temp
