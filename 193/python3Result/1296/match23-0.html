class TreeAncestor:

    def __init__(self, n: int, par: List[int]):
        self.link = defaultdict(list)
        self.depth = defaultdict(int)
        self.f = {}
        for i, x in enumerate(par):
            if x == -1:
                root = i
            else:
                self.link[x].append(i)
        def dfs(i, pre):
            self.depth[i] = self.depth[pre] + 1
            self.f[i] = [0]*20
            self.f[i][0] = pre
            for ii in range(1, 20):
                self.f[i][ii] = self.f[self.f[i][ii-1]][ii-1];
            for node in self.link[i]:
                dfs(node, i)
        dfs(root, 0)
        print(self.f)
    def getKthAncestor(self, x: int, k: int) -> int:
        t = self.depth[x]-k-1
        if t < 0: return -1
        for i in range(19,-1,-1):
            if self.depth[self.f[x][i]]-1 > t:
                x = self.f[x][i]
        return self.f[x][0]
