class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.f = [[0 for _ in range(20)] for _ in range(n)]
        self.graph = [[] for _ in range(n)]
        for idx, val in enumerate(parent):
            if val != -1: self.graph[val].append(idx)
            
        self.dep = [0 for _ in range
(n)]
        self.computeF(0, 0)
        
    def computeF(self, node: int, father: int):
        self.dep[node] = 1 if not node else self.dep[father] + 1
        self.f[node][0] = father
        for i in range(1, 20):
            self.f[node][i] = self.f[self.f[node][i - 1]][i - 1]
        for val in self.graph[node]:
            self.computeF(val, node)

    def getKthAncestor(self, node: int, k: int) -> int:
        if self.dep[node] - k < 1:
            return -1
        
        t 
= self.dep[node] - k
        for i in range(19, -1, -1):
            if self.dep[self.f[node][i]] > t:
                node = self.f[node][i]
                
        return self.f[node][0]
