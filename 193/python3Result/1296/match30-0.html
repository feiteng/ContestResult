class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.ps = {}
        self.ps[1] = parent.copy()
        self.max_depth = 1
        i = 1
        while True:
            if all((node == -1) for node in self.ps[i]):
                self.max_depth = i
                return
            ni = i*2
            self.ps[ni] = [self.ps[i][node] if node != -1 else -1 for node in self.ps[i]]
            i = ni
                    

    def getKthAncestor(self, node: int, k: int) -> int:
        # print(self.max_depth)
        # print(self.ps)
        
        if k >= self.max_depth:
            return -1
        i = 1
        ans = node
        while k > 0:
            if k % 2 == 1:
                ans = self.ps[i][ans]
            if ans == -1:
                return -1
            k //= 2
            i *= 2
        return ans
