class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        i = 0
        self.height = int(log2(n))
        self.table = [0] * (n + 1)
        while i < len(self.table): 
            self.table[i] = [-1]*(self.height + 1) 
            i += 1
        for x in range(len(parent)):
            self.lift_tree(parent[x
], x)

    def lift_tree(self, u, v):
        self.table[v][0] = u 
        i = 1
        while i <= self.height: 
            self.table[v][i] = self.table[self.table[v][i - 1]][i - 1] 
            if (self.table[v][i] == -1): 
                break
            i = i + 1
        
    def getKthAncestor(self, node: int, k: int) -> int:
        i = 0
        while i <= self.height: 
            if (k & (1 << i)) != 0: 
                node = self.table[node][i] 
                if node == -1: 
                    break
            i = i + 1
        return node
