class TreeAncestor:

    def __init__(self, n: int, par: List[int]):
        self.lca=[par]
        for _ in range(20):
            pvr=self.lca[-1]
            cur=[x for x in pvr]
            for i,x in enumerate(pvr):
                if x==-1:
                    continue
                cur[i]=pvr[x]
            self.lca.append(cur)

    def getKthAncestor(self, node: int, k: int) -> int:
        ans
=node
        i=0
        while k:
            if k%2:
                ans=self.lca[i][ans]
            if ans==-1:
                return ans
            i+=1
            k//=2
        return ans
