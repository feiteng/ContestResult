class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.height = int(math.ceil(math.log2(n)))
        self.n = n
        self.table = [0] * (n + 1)
        for i in range(n + 1):
            self.table[i] = [-1] * (self.height + 1)
        
def cal_table(u, v):
            self.table[v][0] = u 
            for i in range(1, self.height + 1):
                self.table[v][i] = self.table[self.table[v][i-1]][i-1]
                if self.table[v][i] == -1:
                    break
                
        for i, pi in enumerate(parent):
            cal_table(
pi, i)

    def getKthAncestor(self, node: int, k: int) -> int:
        for i in range(1 + self.height):
            if k & (1 << i):
                node = self.table[node][i]
                if node == -1:
                    break 
        return node
