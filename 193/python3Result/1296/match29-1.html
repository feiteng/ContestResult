class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        self.P = [[-1] * n for i in range(32)]
        self.P[0] = parent
        
for i in range(1, 32):
            for j in range(n):
                if self.P[i-1][j] == -1 :
                    self.P[i][j] = -1
                else :
                    self.P[i][j] = self.P[i-1][self.P[i-1][j]]
    def getKthAncestor(self, node: int, k: int) -> int:
        while k :
            sub = 1
            i = 0
            while sub < k :
                sub *= 2
                i += 1
            if sub > k :
                i -= 1
                sub //= 2
            if node == -1 :
                break
            node = self.P[i][node]
            k -= sub
        return node
