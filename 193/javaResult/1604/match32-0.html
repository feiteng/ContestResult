class Solution {
    
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : arr) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int[][] a = new int[map.size()][2];
        int i = 0;
        for (Map.Entry<Integer, Integer> m : map.entrySet()) {
            a[i] = new int[]{m.getKey(), m.getValue()};
            i += 1;
        }
        i = 0;
        int size = map.size();
        Arrays.sort(a, (int[] a1, int[] a2) -> Integer.compare(a1[1], a2[1]));
        while (k > 0 && i < size) {
            if (k - a[i][1] >= 0) {
                k -= a[i][1];
                i += 1;
            } else break;
        }
        return size - i;
    }
}
