class Solution {

    class Elem {
        int val;
        int freq;
        Elem(int val, int freq) {
            this.val = val;
            this.freq = freq;
        }
    }
    
    
public int findLeastNumOfUniqueInts(int[] arr, int k) {
        // 1. count integers
        Map<Integer, Integer> numToFreq = new HashMap<>();
        for (int num : arr) {
            numToFreq.put(num, numToFreq.getOrDefault(num, 0) + 1);
        }
        
        // 2. sort by frequency
        List<Elem> elems = new ArrayList<>();
        for (Map.Entry<Integer, Integer> entry : numToFreq.entrySet()) {
            elems.add(new Elem(entry.getKey(), entry.getValue()));
        }
        Collections.sort(elems, (e1, e2)->
Integer.compare(e1.freq, e2.freq));
       
        // 3. remove from smallest integer
        int i = 0;
        while (i < elems.size() && k > 0) {
            if (elems.get(i).freq <= k) {
                k -= elems.get(i).freq;
                i++;
            } else {
                k = 0;
            }
        }
        
        return elems.size()-i;
    }

}
