class Solution {
    class Pair {
        int i;
        int freq;

        public Pair(int i, int freq) {
            this.i = i;
            this.freq = freq;
        }
    }
    
public int findLeastNumOfUniqueInts(int[] arr, int k) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i = 0; i < arr.length; i++) {
            if (!map.containsKey(arr[i])) {
                map.put(arr[i], 0);
            }
            map.put(arr[i], 1 + map.get(arr[i]));
        }
        
        List<Pair> pairs = map.entrySet().stream()
                .map(e -> new Pair(e.getKey(), e.getValue()))
                .sorted(Comparator.comparingInt(o -> o.freq))
                .collect(
Collectors.toList());
        
        int removed = 0;
        for (Pair p : pairs) {
            if (p.freq <= k) {
                k -= p.freq;
                removed++;
            } else {
                break;
            }
        }
        return pairs.size() - removed;
    }
}
