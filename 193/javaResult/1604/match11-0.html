class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> freq = new HashMap<>();
        for (int a: arr) {
            freq.put(a, freq.getOrDefault(a, 0) + 1);
        }
        List<Integer> fr = new ArrayList<>(freq.values());
        Collections.sort(fr);
        int res = fr.size();
        for (int i = 0; i < fr.size(); i ++) {
            if (k >= fr.get(i)) {
                k -= fr.get(i);
                res --;
            } else {
                break;
            }
        }
        return res;
    }
}
