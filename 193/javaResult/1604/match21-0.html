class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        HashMap<Integer, Integer> map = new HashMap();
        
        for(int num: arr){
            int tmp = 1;
            if(map.containsKey(num)){
                tmp = map.get(num) + 1;
            }
            map.put(num, tmp);
        }
        
        int[] vals = new int[map.size()];
        int j = 0;
        for(int num: map.values()) {
            vals[j] = num;
            j++;
        }
        Arrays.sort(vals);
        
        //System.out.println(Arrays.toString(vals));
        int i=0;
        int ans = vals.length;
        
        for(i=0;k>0&&i<vals.length;i++){
            if(k >= vals[i]){
                ans--;
                k=k-vals[i];
            }else{
                break;
            }
        }
        
        return ans;
    }
}
