class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> freq = new HashMap<>();
        for (int v : arr) {
            freq.put(v, freq.getOrDefault(v, 0) + 1);
            
        }
        
        List<Integer> fs = new ArrayList<>(freq.values());
        Collections.sort(fs);
        int res = fs.size();
        for (int v : fs) {
            if (k >= v) {
                k -= v;
                res--;
            } else {
                break;
            }
        }
        return res;
    }
}
