class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> counts = new HashMap<>();
        
        for(int num : arr) {
            counts.put(num, counts.getOrDefault(num, 0) + 1);
        }
        
        List<Integer> uniqs = new ArrayList<>(counts.values());
        Collections.sort(uniqs);
        
        int out = uniqs.size();
        
        for(int count : uniqs) {
            if(count <= k) {
                k -= count;
                out--;
            }
        }
        
        return out;
    }
}
