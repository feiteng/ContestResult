class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        HashMap<Integer,Integer> map=new HashMap<>();
        for(int i:arr){
            map.put(i,map.getOrDefault(i,0)+1);
        }
        ArrayList<Integer> v = new ArrayList<>(map.values());
        Collections.sort(v);
        int ans=0;
        
for(int i:v){
            if(k>=i){
                k-=i;
                ans++;
            }else{
                break;
            }
        }
        return v.size()-ans;

    }
}
