class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> counts = new HashMap<>();
        for(int num : arr) {
            counts.put(num, 
counts.getOrDefault(num, 0) + 1);
        }
        int[][] countsArr = new int[counts.size()][2];
        int i = 0;
        for(int key : counts.keySet()) {
            countsArr[i][0] = key;
            countsArr[i][1] = counts.get(key);
            i++;
        }

        Arrays.sort(countsArr, Comparator.comparingInt(
a -> a[1]));

        for(i = 0; i < countsArr.length; i++) {
            if(k >= countsArr[i][1]) {
                k = k - countsArr[i][1];
            } else {
                break;
            }
        }
        return countsArr.length - i;
    }
}
