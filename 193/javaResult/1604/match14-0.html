class Solution {
     public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> numWithSumMap = new HashMap<>();
        for (int i : arr) {
            numWithSumMap.put(i, numWithSumMap.getOrDefault(i, 0) + 1);
        }
        List<Integer> list = new ArrayList<>();
        for (Integer integer : numWithSumMap.values()) {
            list.add(integer);
        }
        Collections.sort(list);

        int remove = 0;
        for (Integer num : list) {
            if (k >= num) {
                remove++;
                k -= num;
            } else {
                break;
            }
        }
        return list.size() - remove;
    }
}
