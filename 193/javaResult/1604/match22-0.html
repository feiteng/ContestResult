class Solution {
    public int findLeastNumOfUniqueInts(int[] arr, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int a : arr) {
            map.put(a, map.getOrDefault(a, 0) + 1);
        }
        int kk[] = new int[map.size()];
        int i = 0;
        for (Integer in : map.values()) {
            kk[i++] = in;
        }
        Arrays.sort(kk);
        i = 0;
        while (k > 0 && i < kk.length) {
            if (k >= kk[i]) {
                k = k - kk[i];
                i++;
            } else {
                break;
            }
        }
        return map.size() - i;
    }
}
