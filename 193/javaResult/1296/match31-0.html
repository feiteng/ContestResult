class TreeAncestor {
    private int[] parent;
    public TreeAncestor(int n, int[] parent) {
        this.parent = Arrays.copyOf(parent, n);
    }
    
    public int getKthAncestor(int node, int k) {
        if (k > node) {
            return -1;
        }
        while (node != - 1 && k > 0) {
            node = parent[node];
            --k;
        }
        return node;
    }
}
