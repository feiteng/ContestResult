	class TreeAncestor {
		class Solver {

			private  int table[][];

			private  int height;
			
			private Solver(int n)
			{
				height = (int)Math.ceil(Math.log10(n) / Math.log10(2));
				table = new int[n + 1][height + 1];
			}


			private  void preprocessing()
			{
				for (int i = 0; i < table.length; i++) {
					Arrays.fill(table[i], -1);
				}
			}

			private  void calculateSparse(int u, int v)
			{
				
				table[v][0] = u;
				for (int i = 1; i <= height; i++) {
					table[v][i] = table[table[v][i - 1]][i - 1];

					if (table[v][i] == -1)
						break;
				}
			}

			// Function to return the Kth ancestor of V
			private  int kthancestor(int V, int k)
			{
				
				for (int i = 0; i <= height; i++) {
					if ((k & (1 << i)) != 0) {
						V = table[V][i];
						if (V == -1) {
							break;
						}

					}
				}
				return V;
			}
		}

		int[] parent;
		int n;
		Solver sol;
		public TreeAncestor(int n, int[] parent) {
			sol = new Solver(n);
			sol.preprocessing();;
			for (int i = 1; i < parent.length; i++) {
				sol.calculateSparse(parent[i],i);
			}
		}

		public int getKthAncestor(int node, int k) {
			return sol.kthancestor(node,k);
		}
	}
