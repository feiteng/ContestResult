class TreeAncestor {
    
    int[] rd;
    int[] par;
    ArrayList<Integer>[] level;
    
    public TreeAncestor(int n, int[] parent) {
        rd = new int[n];
        Arrays.fill(rd, -1);
        par = parent;
        level = new ArrayList[n];
        for(int i=0 ; i<n ; i++) level[i] = new ArrayList<>();
        for(int i=0 ; i<n ; i++) {
            if(rd[i] == -1) {
                init(i);
            }
        }
        rd[0] = 0;
        level[0].add(0);
        
    }
    
    private int init(int start) {
        if(start == 0) {
            return 0;
        }
        else {
            if(rd[start] == -1) {
                int lev = 1 + init(par[start]);
                rd[start] = lev;
                level[lev].add(start);
            }
            return rd[start];
        }
    }
    
    public int getKthAncestor(int node, int k) {
        if(rd[node] < k) {
            return -1;
        }
        else {
            int ans = node;
            for(int i=0 ; i<k ; i++) {
                ans = par[ans];
            }
            return ans;
        }
    }
}
