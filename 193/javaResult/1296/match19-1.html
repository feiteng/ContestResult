class TreeAncestor {

    private int n;
    private int[][] a;

    public TreeAncestor(int n, int[] parent) {
        this.n = n;
        int d = (int) (
Math.log(n) / Math.log(2)) + 1;
        a = new int[d][n];
        for (int i = 0; i < n; i++) {
            a[0][i] = parent[i];
        }
        for (int i = 1; i < d; i++) {
            for (int j = 0; j < n; j++) {
                if (a[i-1][j] == -1) a[i][j] = -1;
                else a[i][j] = a[i-1][a[i-1][j]];
            
}
        }
    }
    
    public int getKthAncestor(int node, int k) {
        
int index = node, i = 0;
        while (k > 0 && index != -1) {
            if (k % 2 == 1) {
                index = a[i][index];
            }
            i++;
            k >>= 1;
        }
        return index;
    }
}
