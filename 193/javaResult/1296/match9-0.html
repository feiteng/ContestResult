class TreeAncestor {
    int[][] f;
    public TreeAncestor(int n, int[] parent) {
        f = new int[n + 1][20];
        for (int[] frow : f) Arrays.fill(frow, -1); 
        for (int i = 0; i < parent.length; i++) {
            int node = i, p = parent[i];
            if (p == -1) continue; 
            f[node][0] = p; 
            for (int j = 1; j < 20; j++) { 
                f[node][j] = f[f[node][j - 1]][j - 1];  
                if (f[node][j] == -1) 
                    break; 
            }            
        }
    }
    
    public int getKthAncestor(int node, int k) {
        for (int i = 0; i < 20; i++) { 
            if ((k & (1 << i)) != 0) { 
                node = f[node][i]; 
                if (node == -1) 
                    break; 
            } 
        } 
        return node;        
    }
}
