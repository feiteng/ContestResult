class TreeAncestor {

    int height;
    int[][] table;
    
    public TreeAncestor(int n, int[] parent) {
        
        height = (int) Math.ceil(Math.log10(n) / Math.log10(2)); 
        table = new int[n + 1][height + 1]; 
        preprocessing();
        
        for (int i = 0; i < n; ++i) {
            calculateSparse(parent[i], i);
        }
    }
    
    private void preprocessing() {
        for (int i = 0; i < table.length; i++) { 
            Arrays.fill(table[i], -1); 
        } 
    } 
    
    private void calculateSparse(int u, int v) {
        // Using the recurrence relation to 
        // calculate the values of table[][]
        if (u == -1) return;
        table[v][0] = u; 
        for (int i = 1; i <= height; i++) { 
            table[v][i] = table[table[v][i - 1]][i - 1]; 
  
            // If we go out of bounds of the tree 
            if (table[v][i] == -1) 
                break; 
        } 
    } 
    
    public int getKthAncestor(int node, int k) {
        
        for (int i = 0; i <= height; i++) { 
            if ((k & (1 << i)) != 0) { 
                node = table[node][i]; 
                if (node == -1) 
                    break; 
            } 
        } 
        return node; 
    }
}
