class TreeAncestor {
    Map<Integer, TreeMap<Integer, Integer>> m;
    int[] p;

    public TreeAncestor(int n, int[] parent) {
        this.m = new HashMap<>();
        this.p = parent;
    }
    
    public int getKthAncestor(int node, int k) {
        TreeMap<Integer, Integer> t = m.computeIfAbsent(node, item -> new TreeMap<>());
        int o = k;
        Integer s = t.floorKey(k);
        int parent = node;
        if (s != null) {
            k -= s;
            parent = t.get(s);
        }
        
        while (parent != -1 && k > 0){
            parent = p[parent];
            k--;
        }
        
        t.put(o, parent);
        
        return parent;
    }
}
