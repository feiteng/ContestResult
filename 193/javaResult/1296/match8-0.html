class TreeAncestor {
    
    
    int ps[][];
    public TreeAncestor(int n, int[] parent) {
        ps  = new int[20][n];
        
        for(int j=0;j<20;++j){
            for(int i=0;i<n;++i){
                ps[j][i] = -1;
            }
        }
        
        
        for(int i=0;i<n;++i){
            ps[0][i] = parent[i];
        }
        
        for(int j=1;j<20;++j){
            for(int i=0;i<n;++i){
                if(ps[j-1][i]>=0){
                    ps[j][i] = ps[j-1][ps[j-1][i]];
                }
            }
        }
        
        
        
        
    }
    
    public int getKthAncestor(int node, int k) {
        for(int i=19;i>=0;--i){
            if((k&(1<<i))>0){
                
                node = ps[i][node];
                if(node==-1){
                    return -1;
                }
                
            }
        }
        
        
        return node;
    }
}
