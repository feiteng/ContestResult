class TreeAncestor {

    Map<Integer, TreeMap<Integer, Integer>> map;
    int[] parents;
    
    public TreeAncestor(int n, int[] parent) {
        this.map = new HashMap<>();
        this.parents = parent;
    }
    
    public int getKthAncestor(int node, int k) {
        TreeMap<Integer, Integer> treeMap = map.computeIfAbsent(node, item -> new TreeMap<>());
        
        int o = k;
        
        Integer sub = treeMap.floorKey(k);
        int parent = node;
        
        if (sub != null) {
            k -= sub;
            parent = treeMap.get(sub);
        }
        
        while (parent != -1 && k > 0) {
            parent = parents[parent];
            k--;
        }
        
        treeMap.put(o, parent);
        
        return parent;
    }
}
