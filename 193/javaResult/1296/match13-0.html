class TreeAncestor {

        private int height;
        private int[][] table;
        private int[] parent;
        private int n;

        public TreeAncestor(int n, int[] parent) {
            height = (int) Math.ceil(Math.log10(n) / Math.log10(2));
            table = new int[n + 1][height + 1];
            for (int i = 0; i < table.length; i++) {
                Arrays.fill(table[i], -1);
            }
            this.parent = parent;
            this.n = n;
            fillArray();
        }

        void fillArray() {
            boolean[] visited = new boolean[n];
            for (int i = 0; i < n; i++) {
                if (!visited[i]) {
                    travel(i, visited);
                }
            }
        }

        void travel(int i, boolean[] visited) {
            if (visited[i]) return;
            if (parent[i] == -1) return;
            visited[i] = true;
            calculateSparse(parent[i], i);
            travel(parent[i], visited);
        }

        void calculateSparse(int u, int v) {
            table[v][0] = u;
            for (int i = 1; i <= height; i++) {
                table[v][i] = table[table[v][i - 1]][i - 1];
                if (table[v][i] == -1)
                    break;
            }
        }


        public int getKthAncestor(int node, int k) {
            for (int i = 0; i <= height; i++) {
                if ((k & (1 << i)) != 0) {
                    node = table[node][i];
                    if (node == -1)
                        break;
                }
            }
            return node;
        }
    }
