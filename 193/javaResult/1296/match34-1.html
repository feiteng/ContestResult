class TreeAncestor {
    private int n = 0;
    private int[] parent = null;
    private int[] depth = null;
    
    public TreeAncestor(int n, int[] parent) {
        this.n = n;
        this.parent = parent;
        this.depth = new int[n];
        
        this.depth[0] = 0;
        for(int i=1;i<n;i++){
            this.depth[i]=this.depth[parent[i]]+1;
        }
    
}
    
    public int getKthAncestor(int node, int k) {
        if(this.depth[node] < k){
            return -1;
        }
        
int ans = node;
        for(int i=1;i<=k;i++){
            ans=this.parent[ans];
        }
        return ans;
    }
}
