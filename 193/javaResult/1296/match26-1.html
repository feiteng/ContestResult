class TreeAncestor {
    
    private int[] ps;
    private int n;
    private int[] depth;

    public TreeAncestor(int n, int[] parent) {
        ps = parent;
        this.n = n;
        depth = new int[n];
        
        Arrays.fill(depth, -1);
        for (int i = 0; i < n; i++) {
            findDepth(i);
        
}
    }
    
    public int getKthAncestor(int node, int k) {
        if (depth[node] < k) {
            return -1;
        }
        
        
for (int i = 0; i < k; i++) {
            node = ps[node];
        }
        return node;
    }
    
    private int findDepth(int v) {
        if (depth[v] >= 0) {
            return depth[v];
        }
        if (ps[v] == -1) {
            depth[v] = 0;
            return 0;
        }
        
        int d = findDepth(ps[v]) + 1;
        depth[v] = d;
        return d;
    }
}
