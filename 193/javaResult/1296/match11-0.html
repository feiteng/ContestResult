class TreeAncestor {
    private int[][] p;
    private int maxLevel;
    
    public TreeAncestor(int n, int[] parent) {
        this.maxLevel = 16;
        this.p = new int[n][maxLevel];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < maxLevel; j++) {
                this.p[i][j] = -1;
            }
        }
        for (int i = 0; i < n; i++) {
            this.p[i][0] = parent[i];
        }
        for (int i = 1; i < maxLevel; i++) {
            for (int j = 0; j < n; j++) {
                if (this.p[j][i - 1] != -1) {
                    this.p[j][i] = this.p[this.p[j][i - 1]][i - 1];
                }
            }
        }
            
        
    }
    
    public int getKthAncestor(int node, int k) {
        while (node > 0 && k > 0) {
            int cur = (int)(Math.log(k) / Math.log(2));
            if (this.p[node][cur] == -1) {
                return -1;
            }
            node = this.p[node][cur];
            k -= (int)(Math.pow(2, cur));
        }
        return k <= 0 ? node : -1;
    }
}
