class TreeAncestor {

        int n = 0;
        int[] pa = null;
        Map<Integer, Integer> depth = new HashMap<>();

        public TreeAncestor(int n, int[] pa) {
            this.n = n;
            this.pa = pa;
            for (int i = 0; i < n; i++) {
                if (!depth.containsKey(i)) {
                    int d = getdepth(pa, i);
                    depth.put(i, d);
                }
            }
        
}

        private int getdepth(int[] pa, int i) {
            if (pa[i] == -1) {
                return 0;
            }
            if (depth.containsKey(i)) {
                return depth.get(i);
            }
            return 1 + getdepth(pa, pa[i]);
        }

        public int getKthAncestor(int n, int k) {
            int dep = depth.get(n);
            if (k > dep) {
                return -1;
            }
            int p = n;
            
while (k > 0) {
                p = pa[p];
                k--;
            }
            return p;
        }
    }
