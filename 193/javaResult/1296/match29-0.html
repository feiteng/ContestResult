class TreeAncestor {
    
    private int[] parent;
    private int[][] depth;

    public TreeAncestor(int n, int[] parent) {
        this.parent = parent;
        this.depth = new int[n][2];
        int max = 0;
        for (int i=0; i<n; i++) {
            this.depth[i][0] = getDepth(i);
        }
    }
    
    private int getDepth(int i) {
        if (this.parent[i] == -1)
            return 0;
        
        if (this.depth[i][0] != 0)
            return this.depth[i][0];
        
        return  1+getDepth(this.parent[i]);
    }
    
    public int getKthAncestor(int node, int k) {
        if (k>this.depth[node][0])
            return -1;
        if (k == this.depth[node][0])
            return 0;
        if (k == 1)
            return this.parent[node];
        if (k == 0)
            return node;
        
        if (k>this.depth[node][0]/2) {
            if (this.depth[node][1] == 0)
                this.depth[node][1] = getKthAncestor(node, this.depth[node][0]/2);
            
            return getKthAncestor(this.depth[node][1], k-this.depth[node][0]/2);
        }
        
        return getKthAncestor(this.parent[this.parent[node]], k-2);
    }
}
