class TreeAncestor {
    HashMap<Integer, Integer> map;
    int[] p;

    public TreeAncestor(int n, int[] parent) {
        p = parent;
        map = new HashMap<>();
        map.put(0, 1);
        for (int i = 1; i < parent.length; i++) {
            map.put(i, map.get(parent[i]) + 1);
        }
    }
    
    public int getKthAncestor(int node, int k) {
        if (map.get(node) <= k) return -1;
        while (k-- > 0) {
            node = p[node];
        }
        return node;
    }
}
