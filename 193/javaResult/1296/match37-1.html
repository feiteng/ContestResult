class TreeAncestor {
    int[] hei;
    int[] par;
    int max;
    Map<Integer, Integer> map = new HashMap<>();
    public TreeAncestor(int n, int[] parent) {
        par = parent;
        hei = new int[n];
        int max = 0;
        for (int i = 1; i < n; i++) {
            hei[i] = getHei(i);
            max = Math.max(max, hei[i]);
        }
    }
    
    public int getHei(int cur) {
        if (cur == 0) return 0;
        if (hei[cur] != 0) return hei[cur];
        return 1 + 
getHei(par[cur]);
    }
    
    public int getKthAncestor(int node, int k) {
        if (hei[node] - k < 0) return -1;
        if (map.containsKey(node)) return map.get(node);
        int cur = node;
        
while (k > 0) {
            cur = par[cur];
            k--;
        }
        return cur;
    }
}
