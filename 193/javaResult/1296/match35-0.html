class TreeAncestor {

    int[][] arr;
    
    public TreeAncestor(int n, int[] parent) {
        arr = new int[n][20];
        for(int i = 0; i < n; i++) {
            Arrays.fill(arr[i], -1);
            arr[i][0] = parent[i];
        }
        for(int j = 1; j < arr[0].length; j++) {
            for(int i = 0; i < parent.length; i++) {
                if(arr[i][j - 1] != -1) {
                    arr[i][j] = arr[arr[i][j - 1]][j - 1];
                }
            }
        }
    }
    
    public int getKthAncestor(int node, int k) {
        for(int j = 17; j >= 0 && node != -1; j--) {
            if(((k >> j) & 1) == 1) {
                // System.out.println(node + " " + k);
                node = arr[node][j];
            }
        }
        return node;
    }
}
