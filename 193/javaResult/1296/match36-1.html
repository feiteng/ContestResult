class TreeAncestor {
    private int[] p;
    private int n;
    private Map<Integer, List<Integer>> child = new HashMap<Integer, List<Integer>>(); 
    private Map<Integer,Integer> level = new HashMap<Integer, Integer>();
    public TreeAncestor(int n, int[] parent) {
        this.p = parent;
        this.n = n;
        
        for (int i=0;i<parent.length;i++) {
            List<Integer> l = child.get(parent[i]);
            if (l == null) {
                l = new ArrayList<Integer>();
                l.add(i);
                child.put(parent[i], l);
            } else {
                l.add(i);
            }
        }
        
        level.put(0,0);
        getLevel(0, 0);
        
        //System.out.println(level);
        
    }
    
    private void getLevel(int x, int lev) {
        List<Integer> l = child.get(x);
        
        if (l == null) return;
        
        
for (int ch : l) {
            level.put(ch, lev+1);
            getLevel(ch, lev+1);
        } 
    }
    
    public int getKthAncestor(int node, int k) {
        if (k > level.get(node)) return -1;
        
        while (k > 0) {
            node = p[node];
            k--;
        }
        return node;
    }
}
