class TreeAncestor {

    int n;
    int table[][];
    int height;

    public TreeAncestor(int n, int[] parent) {
        this.n = n;
        
this.height = (int)Math.ceil(Math.log10(n) / Math.log10(2));
        this.table = new int[n + 1][height + 1];
        for (int i = 0; i < table.length; i++) {
            Arrays.fill(table[i], -1);
        }
        for (int i = 2; i <= n; i++) {
            calculateSparse(parent[i - 1] + 1, i);
        }
    
}

    private void calculateSparse(int u, int v) {
        table[v][0] = u;
        for (int i = 1; i <= height; i++) {
            table[v][i] = table[table[v][i - 1]][i - 1];
            if (table[v][i] == -1) break;
        }
    }

    public int getKthAncestor(int node, int k) {
        node ++;
        for (int i = 0; i <= height; i++) {
            if ((k & (1 << i)) != 0) {
                node = table[node][i];
                if (node == -1) return -1
;
            }
        }
        return node - 1;
    }
}
