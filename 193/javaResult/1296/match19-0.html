class TreeAncestor {
 private int     n;
    private int[]   parent;
    private int[][] p;
    int log;

    public TreeAncestor(int n, int[] parent) {
        this.n = n;
        this.parent = parent;
        this.log = (int) (Math.log(5 * 10000) / Math.log(2) + 1) + 1;
        p = new int[n][log];
        for (int i = 0; i < n; i++) {
            p[i][0] = parent[i];
        }
        for (int j = 1; j < log; j++) {
            for (int i = 0; i < n; i++) {
                int x = p[i][j - 1];
                p[i][j] = x == -1 ? -1 : p[x][j - 1];
            }
        }
    }

    public int getKthAncestor(int node, int k) {
        if (k >= n) {
            return -1;
        }
        int cur = node;
        int mask = 1, j = 0;
        while (k > 0 && cur != -1) {
            if ((k & mask) != 0) {
                cur = p[cur][j];
                k ^= mask;
            }
            mask <<= 1;
            j++;
        }
        return cur;
    }

}
