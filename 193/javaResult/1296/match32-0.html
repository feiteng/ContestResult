class TreeAncestor {
    private int[] parent;
    private int[] height;

    public TreeAncestor(int n, int[] parent) {
        this.parent = parent;
        this.height = new int[parent.length];

        for (int i = 0; i < parent.length; i++) {
            int h = 0;
            int x = i;
            while (x != -1) {
                if (height[x] != 0) {
                    height[i] = height[x];
                    break;
                }
                x = parent[x];
                h++;
            }
            height[i] += h;
        }
    }

    public int getKthAncestor(int node, int k) {
        if (k > height[node]) {
            return -1;
        }
        while (k-- > 0 && node != -1) {
            node = parent[node];
        }
        return node;
    }
}
