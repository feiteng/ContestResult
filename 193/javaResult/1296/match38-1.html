class TreeAncestor {

    int[] b;
    int[] c;
    int[] p;
    int n;

    public TreeAncestor(int n, int[] parent) {
        this.n = n;
        this.p = parent;
        b = new int[n];
        c = new int[n];
        for (int i=1; i<n; i++) {
            b[i] = -1;
            c[i] = -1;
        }
        for (int i=1; i<n; i++)
            find(i, 200);
    
}
    
    private void find(int x, int q) {
        if (b[x] != -1)
            return;
        if (q == 1) {
            b[x] = p[x];
            c[x] = 1;
            return;
        }
        find(p[x], q-1);
        b[x] = b[p[x]];
        c[x] = c[p[x]] + 1;
    }

    public int getKthAncestor(int node, int k) {
        if (c[node] <= k && c[node] > 0)
            return getKthAncestor(b[node], k-c[node]);
        int t = node;
        
while (t != -1 && k > 0) {
            t = p[t];
            k--;
        }
        return t;
    }
}
