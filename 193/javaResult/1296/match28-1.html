class TreeAncestor {
    int p [];
    int depth [];
    public TreeAncestor(int n, int[] parent) {
        p =parent;
        depth =new int [parent.length];
        Arrays.fill(depth,-1);
        depth[0]=0;
        for(int i=1;i<parent.length;i++){
            if(depth[i]!=-1) continue;
            
dfs(parent,depth,i);
        }
        
    }
    
    public int getKthAncestor(int node, int k) {
        if(depth[node]<k) 
return -1;
        while(k>0&&node!=-1){
            k--;
            node =p[node];
        }
        return node;
    }
    private int dfs(int []parent ,int depth[],int index){
        if(depth[index]!=-1) return depth[index];
        depth[index]=1+ dfs(parent,depth,parent[index]);
        return depth[index];
    }
}
