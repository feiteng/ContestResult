class TreeAncestor {

    private int[][] dp;
    private int height;
    
    public TreeAncestor(int n, int[] parent) {
        height = (int)(Math.floor(Math.log(n) / Math.log(2)));
        dp = new int[n][height+1];
        for (int i = 0; i < n; i++) Arrays.fill(dp[i], -1);
        for (int i = 0; i < n; i++) dp[i][0] = parent[i];
        
        for (int h = 1; h <= height; h++) {
            for (int i = 1; i < n; i++) {
                if (dp[i][h-1] != -1) {
                    dp[i][h] = dp[dp[i][h-1]][h-1];
                }
            }
        }
    }
    
    public int getKthAncestor(int node, int k) {
        for (int i = 0; i <= height; i++) {
            if ((k & (1 << i)) != 0) {
                node = dp[node][i];
                if (node == -1) break;
            }
        }
        return node;
    }    
}
