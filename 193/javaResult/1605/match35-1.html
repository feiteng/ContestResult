class Solution {
    
    public int minDays(int[] roses, int n, int k) {
        if (n * k > roses.length) return -1;
        int min = Integer.MAX_VALUE, max = -1;
        for (int r : roses) {
            max = Math.max(r, max);
            min = Math.min(r, min);
        }
        
int s = min, e = max;
        while (s <= e) {
            int mid = (e - s) / 2 + s;
            if (is_enough_bouquets(roses, n, k, mid)) {
                e = mid - 1;
            } else {
                s = mid + 1;
            }
        }
        return e + 1;
    }


     boolean is_enough_bouquets(int[] roses, int n, int k, int days){
        int adj_bloom_cnt = 0;
        int bouquets = 0;
        for(int i = 0 ; i < roses.length ; i++){
            if(roses[i] > days){
                adj_bloom_cnt = 0;
                continue;
            }
            //roses[i] = 0;
            adj_bloom_cnt++;
            if(adj_bloom_cnt == k)
{
                bouquets++;
                adj_bloom_cnt = 0;
            }

        }
        return bouquets >= n;
    }
}
