class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (m * k > bloomDay.length) return -1;
        int n = bloomDay.length;
        TreeSet<Integer> set = new TreeSet<>();
        for (int i = 0; i < n; i++)
            set.add(bloomDay[i]);
        int[] days = new int[set.size()];
        int idx = 0;
        for (int day: set)
            days[idx++] = day;
        
        int lo = 0, hi = days.length - 1;
        
        while (lo <= hi) {
            int mid = (hi - lo) / 2 + lo;
            if (!helper(bloomDay, m, k, days[mid])) {
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }
        
        return lo == days.length ? -1 : days[lo];
    }
    
    private boolean helper(int[] bloomDay, int m, int k, int day) {
        int continous = 0;
        int pre = -2;
        for (int i = 0; i < bloomDay.length && m > 0; i++) {
            if (day >= bloomDay[i]) {
                if (i - pre == 1) {
                    if (continous == k - 1) {
                        m--;
                        continous = 0;
                    } else {
                        continous += 1;
                    }
                } else {
                    continous = 1;
                    if (k == 1) {
                        m--;
                        continous = 0;
                    }
                }
                pre = i;
            } else {
                continous = 0;
            }
        }
        //System.out.println(day + ":" + m);
        return m == 0;
    }
}
