class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        int n = bloomDay.length;
        if (n < m * k) return -1;
        int left = 1, right = 1000000001;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (!canFinish(bloomDay, m, k, mid)) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
    
    private boolean canFinish(int[] bloomDay, int m, int k, int mid) {
        int cur = 0, num = m;
        for (int i = 0; i < bloomDay.length; ++i) {
            if (bloomDay[i] <= mid) {
                cur++;
                if (cur == k) {
                    num--;
                    cur = 0;
                }
            } else {
                cur = 0;
            }
        }
        return num <= 0;
    }
}
