class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (m * k > bloomDay.length) {
            return -1;
        }
        int maxDay = 1;
        int minDay = 1000000000;
        for (int day : bloomDay) {
            minDay = Math.min(minDay, day);
            maxDay = Math.max(maxDay, day);
        }
        while (minDay < maxDay) {
            int midDay = (minDay + maxDay) >>> 1;
            if (ok(bloomDay, midDay, m, k)) {
                maxDay = midDay;
            } else {
                minDay = midDay + 1;
            }
        }
        return minDay;
    }

    private boolean ok(int[] bloomDay, int midDay, int m, int k) {
        int cnt = 0;
        int tmp = 0;
        for (int val : bloomDay) {
            if (val <= midDay) {
                tmp++;
            } else {
                cnt += (tmp / k);
                tmp = 0;
            }
        }
        cnt += (tmp / k);
        return cnt >= m;
    }
}
