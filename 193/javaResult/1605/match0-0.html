	class Solution {
	    public int minDays(int[] bloomDay, int m, int k) {
	        int low = 0, high = 1000000007;
	        while(high - low > 1){
	        	int h = high+low>>1;
	        	if(ok(h, bloomDay, m, k)){
	        		high = h;
	        	}else{
	        		low = h;
	        	}
	        }
	        if(high > 1000000000)return -1;
	        return high;
	    }
	    
	    boolean ok(int h, int[] a, int m, int K)
	    {
	    	int ct = 0;
	    	int num = 0;
	    	for(int v : a){
	    		if(v <= h){
	    			num++;
	    			if(num >= K){
	    				num = 0;
	    				ct++;
	    			}
	    		}else{
	    			num = 0;
	    		}
	    	}
	    	return ct >= m;
	    }
	}
