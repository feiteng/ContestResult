class Solution {
    public int minDays(int[] a, int m, int k) {
        if ((long) a.length < (long) m * k) return -1
;
        int lo = 1, hi = (int) 1e9;
        while (lo < hi) {
            int mid = (lo + hi) >>> 1;
            if (valid(a, m, k, mid)) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }
    
    boolean valid(int[] a, int m, int k, int mid) {
        int cur = 0;
        
        
for (int i = 0; i < a.length; i++) {
            if (a[i] <= mid) {
                cur++;
                if (cur == k) {m--; cur = 0;}
            } else {
                cur = 0;
            }
        }
        return m <= 0;
    }
}
