class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (m * k > bloomDay.length) {
            return -1;
        }
        int l = 1; 
        int r = 1000000000;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (validate(bloomDay, m, k, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
    
    
    private boolean validate(int[] bloomDay, int m, int k, int day) {
        int n = bloomDay.length;
        boolean[] status = new boolean[n];
        int consecutive = 0;
        for (int i = 0; i < n; i++) {
            if (bloomDay[i] <= day) {
                consecutive++;
                if (consecutive == k) {
                    m--;
                    consecutive = 0;
                }
            } else {
                consecutive = 0;
            }
            if (m == 0) {
                return true;
            }
        }
        return false;
    }
}
