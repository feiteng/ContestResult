class Solution {
    public int minDays(int[] bloom, int m, int k) {
        int n = bloom.length;
        
if(m * k > n){
            return -1;
        }
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        for(int a : bloom){
            min = Math.min(min, a);
            max = Math.max(max, a);
        }
        int left = min;
        int right = max;
        
while(left <= right){
            int mid = left + (right - left) / 2;
            if(canMake(bloom, m, k, mid)){
                right = mid - 1;
            }else{
                left = mid + 1;
            }
        }
        return right + 1;
    }
    
    private boolean canMake(int[] bloom, int m, int k, int days){
        int count = 0;
        int len = 0;
        for(int i = 0;i<bloom.length;i++){
            if(bloom[i] <= days){
                if(++len == k){
                    count++;
                    len = 0;
                }
            
}else{
                len = 0;
            }
        }
        return count >= m;
    }
    
}
