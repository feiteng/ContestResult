class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        int[] g = new int[bloomDay.length];
        Map<Integer, List<Integer>> map = new TreeMap<>();
        
        for (int i = 0; i < bloomDay.length; i++) {
            if (!map.containsKey(bloomDay[i])) map.put(bloomDay[i],new ArrayList<>());
            map.get(bloomDay[i]).add(i);
        }
        
        int[] keys = new int[map.keySet().size()];
        
        int p = 0;
        for (int i : map.keySet()) {
            keys[p++] = i;
        }
        
        int l = 0, r = keys.length - 1;
        boolean v = false;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (valid(m, k, keys[mid], bloomDay)) {
                r = mid;
                v = true;
                if (l == r) break;
            } else {
                l = mid + 1;
            }
        }
        
        if (v) return keys[l];
        return -1;
    }
    
    boolean valid(int m, int k, int t, int[] bloomDay) {
        int[] g = new int[bloomDay.length];
        
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomDay[i] <= t) g[i] = 1;
        }
        
        int count = 0;
        for (int i = 0; i < g.length; i++) {
            if (g[i] == 1) {
                count++;
                if (count == k) {
                    m--;
                    count = 0;
                }
                if (m == 0) return true;
            } else {
                count = 0;
            }
        }
        return false;
    }
    
    boolean valid(int[] g, int m, int k) {
        int count = 0;
        for (int i = 0; i < g.length; i++) {
            if (g[i] == 1) {
                count++;
                if (count == k) {
                    m--;
                    count = 0;
                }
                if (m == 0) return true;
            } else {
                count = 0;
            }
        }
        return false;
    }
}
