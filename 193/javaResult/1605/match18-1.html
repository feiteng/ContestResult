class Solution {
    public int minDays(int[] bd, int m, int k) {
        long total = ((long)m) * k;
        if(total > bd.length) {
            return -1;
        }
        int l = (int)1e9 + 1, r = 0;
        for(int v : bd) {
            l = Math.min(l, v);
            r = Math.max(r, v);
        }
        
while(l < r - 1) {
            int mid = l + (r - l) / 2;
            if(bsCheck(bd, m, k, mid)) {
                r = mid;
            }
            else {
                l = mid + 1;
            }
        }
        if(bsCheck(bd, m, k, l)) {
            return l;
        }
        return r;
    }
    private boolean bsCheck(int[] bd, int m, int k, int t) {
        boolean[] a = new boolean[bd.length];
        for(int i = 0; i < bd.length; i++) {
            if(bd[i] <= t) {
                a[i] = true;
            }
        }
        int total = 0;
        
int cnt = 0;
        for(int i = 0; i < bd.length; i++) {
            if(a[i]) {
                cnt++;
                if(cnt == k) {
                    total++;
                    cnt = 0;
                
}
            }
            else {
                cnt = 0;
            }
        }
        return total >= m;
    }
}
