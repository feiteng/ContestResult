import java.util.NavigableSet;

class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (bloomDay.length < k*m) {
            return -1;
        }
        int maxTime = bloomDay[0];
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomDay[i] > maxTime) {
                maxTime = bloomDay[i];
            }
        }
        int loTime = 0, hiTime = maxTime+1;
        while (hiTime-loTime > 1) {
            int mid = (hiTime-loTime)/2 + loTime;
            if (isPossible(mid, bloomDay, m, k)) {
                hiTime = mid;
            } else {
                loTime = mid;
            }
        }
        return hiTime;
    }
    
    private boolean isPossible(int curTime, int[] bloomDay, int numBouquets, int numFlowers) {
        boolean[] bloomed = new boolean[bloomDay.length];
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomDay[i] <= curTime) {
                bloomed[i] = true;
            }
        }
        
        // System.out.println("curTime = " + curTime);
        // for (int i = 0; i < bloomed.length; i++) {
        //     System.out.printf("%d ", bloomed[i] ? 1 : 0);
        // }
        // System.out.println();
        
        int start = -2, end = -3;
        int totalBouquets = 0;
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomed[i]) {
                if (end == i-1) {
                    end = end+1;
                } else {
                    int len = end-start+1;
                    totalBouquets += (len/numFlowers);
                    start = i;
                    end = i;
                }
            } else {
                totalBouquets += ((end-start+1)/numFlowers);
                start = -2;
                end = -3;
            }
            // System.out.printf("     i=%d, total=%d,  start=%d, end=%d\n", i, totalBouquets, start, end);
        }
        totalBouquets += ((end-start+1)/numFlowers);
        // System.out.println("totalBouquets = " + totalBouquets);
        return totalBouquets >= numBouquets;
    }
}

class Flower {
    int index;
    int bloomDay;
    public Flower(int index, int bloomDay) {
        this.index = index;
        this.bloomDay = bloomDay;
    }
}
