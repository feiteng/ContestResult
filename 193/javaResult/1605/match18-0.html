class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        int len = bloomDay.length;
        if(m * k > len){
            return -1;
        }
        
        int maxDay = bloomDay[0];
        int minDay = bloomDay[0];
        for(int num : bloomDay){
            maxDay = Math.max(maxDay, num);
            minDay = Math.min(minDay, num);
        }
        
        int start = minDay;
        int end = maxDay;
        
        while(start < end){
            int mid = start + (end-start)/2;
            if(hello(bloomDay, mid, m, k) >=0){
                end = mid;
            }else{
                start = mid + 1;
            }
        }
        
        if(hello(bloomDay, start, m, k) >=0){
            return start;
        }
        
        return -1;
    }
    
    
    int hello(int[] bloomDay, int day, int m, int k){
        int mm = 0;
        int kk = 0;
        for(int num: bloomDay){
            if(day>=num){
                kk++;
                if(kk>=k){
                    mm++;
                    kk=0;
                }else{
                    
                }
            }else{
                kk=0;
            }
            //System.out.println("kk="+kk+",mm="+mm);
        }
        
        return mm-m;
    }
}
