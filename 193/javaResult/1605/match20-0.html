class Solution {
       public static int minDays(int[] bloomDay, int m, int k) {

        if (m > bloomDay.length) {
            return -1;
        }
        int max = 0;
        for (int i : bloomDay) {
            max = Math.max(i, max);
        }
        int min = 1;
        while (min < max) {
            int mid = (max + min) / 2;
            if (!isEnoughDay(bloomDay, m, k, mid)) {
                min = mid + 1;
            }else {
                max = mid;
            }
        }
        if (isEnoughDay(bloomDay,m, k, min)) {
            return min;
        }
        return -1;

    }
    private static boolean isEnoughDay(int[] bloomDay, int m, int k, int day) {
        int temp = k;
        int count = m;
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomDay[i] <= day) {
                temp --;
                if (temp == 0) {
                    temp = k;
                    count --;
                    if (count == 0) {
                        return true;
                    }
                }
            }else {
                temp = k;
            }
        }
        return false;
    }
}
