class Solution {
  public int minDays(int[] bloomDay, int m, int k) { 
      if (bloomDay == null || m == 0 || k == 0) return 0;
      if (m * k > bloomDay.length) return -1;
      if(k == 1){
          Arrays.sort(bloomDay);
          return bloomDay[m-1];
      
}
      int l = Integer.MAX_VALUE, r = Integer.MIN_VALUE;
      for (int val : bloomDay) { l = Math.min(l, val); r = Math.max(r, val); }
      while (l <= r) {
          int mid = l + (r - l) / 2;
          if (isValid(bloomDay, k, m, mid)) r = mid - 1;
          else l = mid + 1;
      }
      return l;
  }
  
  public boolean isValid(int bloomDay[], int size, int count, int day) {
      int curcount = 0, cursize = 0;
      for (int val : bloomDay) {
          if (val <= day) cursize++;
          else cursize = 0;
          if (cursize == size) { cursize = 0; curcount++; }
          if (curcount == count) return true;
      }
      return false;
  }
}
