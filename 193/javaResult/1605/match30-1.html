class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (m * k > bloomDay.length) {
            return -1;
        }
        
int low = 1;
        int high = 1000000000;
        while (low < high) {
            int mid = (low + high) / 2;
            if (valid(bloomDay, m, k, mid)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }

    private boolean valid(int[] bloomDay, int m, int k, int threshold) {
        int count = 0;
        
int consecutive = 0;
        for (int i = 0; i < bloomDay.length; i++) {
            if (bloomDay[i] <= threshold) {
                consecutive++;
                if (consecutive >= k) {
                    count++;
                    consecutive = 0;
                }
            } else {
                
consecutive = 0;
            }
        }
        return count >= m;
    }
}
