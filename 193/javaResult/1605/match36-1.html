class Solution {
     public int minDays(int[] bloomDay, int m, int k) {
        if(bloomDay.length / m < k) return -1;

        int[] copy = new int[bloomDay.length];
        System.arraycopy(bloomDay, 0, copy, 0, bloomDay.length);
        Arrays.sort(copy);
        
int l = 0, r = copy.length - 1;
        while (l < r) {
            int mid = (l + r) >>> 1;
            if (checkmid(bloomDay, copy[mid], k) < m) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        return copy[l];
    }

    private int checkmid(
int[] bloomDay, int level, int k) {
        int counter = 0;
        int bouquets = 0;
        for(int i = 0; i < bloomDay.length; i++) {
            if(bloomDay[i] <= level) {
                counter++;
            
} else {
                bouquets += counter/k;
                counter = 0;
            }
        }
        bouquets += counter/k; // the last piece
        return bouquets;
    }
}
