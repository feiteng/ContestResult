class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        int left = 1, right = 1_000_000_000;
        while (left<=right){
            int mid = left + (right-left)/2;
            if (canForm(bloomDay, mid, m, k)){
                right = mid -1;
            }else{
                left = mid+1;
            }
        }
        
        return left <=1_000_000_000? left : -1;
    }
    
    private boolean canForm(int[] arr, int day, int m, int k){
        int cnt = 0;
        int cur = 0;
        for (int i=0;i<arr.length;i++){
            if (arr[i]<=day) cur+=1;
            else{
                cnt+=cur/k;
                cur=0;
            }
        }
        cnt+=cur/k;
        return cnt>=m;
    }
}
