class Solution {
    boolean isOk(int[] bloomDay,int m,int k,int cur){
        int ans=0;
        int kk=0;
        for(int i:bloomDay){
            if(i<=cur){
                kk++;
                if(kk==k){
                    kk=0;
                    ans++;
                }
            }else{
                kk=0;
            }
        }
        return ans>=m;
    }
    public int minDays(int[] bloomDay, int m, int k) {
        if(bloomDay.length<m*k){
            return -1;
        }
        int l=0,r=Integer.MIN_VALUE;
        for(int i:bloomDay){
            r=Math.max(r,i);
        }
        while(l<r){
            int cur=l+(r-l)/2;
            if(isOk(bloomDay,m,k,cur)){
                r=cur;
            }else{
                l=cur+1;
            }
        }
        return l;
    }
}
