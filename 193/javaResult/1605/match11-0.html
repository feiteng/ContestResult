class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        long MIN = 0;
        long MAX = Integer.MAX_VALUE;
        while (MIN < MAX) {
            long piv = (MIN+MAX)>>1;
            if (can(bloomDay, piv, m, k)) {
                MAX = piv;   
            } else {
                MIN = piv+1;
            }
        }
        return (MIN == Integer.MAX_VALUE) ?  -1 : (int)MIN;
    }
    public boolean can(int [] b, long v, int m, int k) {
        int curcnt = 0;
        int totcnt = 0;
        for (int i = 0; i<b.length && totcnt < m; ++i) {
            
            if (b[i] > v) {
                curcnt = 0;
            } else {
                curcnt++;
            }
            
            if (curcnt == k) {
                curcnt = 0;
                totcnt++;
            }
        }
        return (totcnt >= m);
    }
}
