class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        limitList = []
        pre = 0
        
        for num in candiesCount:
            limitList.append([pre, pre+num])
            pre += num
            
        ans = []
        for i, (t, d, c) in enumerate(queries):
            preMin 
= d
            preMax = d*c
            if (preMin+1) <= limitList[t][1] and (preMax + c) > limitList[t][0]:            
                ans.append(True)
            else:
                ans.append(False)
        return ans
