class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pre = []
        cur = 0
        for val in candiesCount:
            cur += val
            pre.append(cur)
        pre.append(0)
        print(pre)
        ans 
= []
        for t,d,cap in queries:
            d += 1
            if d<=pre[t] and pre[t-1]<cap*d:
                ans.append(True)
            else:
                ans.append(False)
        return ans
