class Solution:
    def canEat(self, c: List[int], Q: List[List[int]]) -> List[bool]:
        s = [0]
        for i in c:
            s.append(s[-1] + i)
        ans = []
        for f_want, f_day, cap in Q:
            _max = cap * f_day + (cap - 1)
            _min = f_day
            if _min < s[f_want + 1] and s[f_want] <= _max:
                ans.append(True)
            else:
                ans.append(False)
        return ans
