class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        preSum = [0]
        for x in candiesCount:
            preSum.append(preSum[-1] + x)
        
        ans = []
        for k, d, m in queries:
            ans.append( (d + 1) * m >= preSum[k] + 1 and d + 1 <= preSum[k + 1] )
        return ans
