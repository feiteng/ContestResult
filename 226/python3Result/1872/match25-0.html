class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        cs=[0]
        for a in candiesCount:
            cs.append(cs[-1]+a)

        ans=[]
        for t,d,c in queries:
            ans.append((cs[t]<c*(d+1))&(d<cs[t+1]))

        return ans
