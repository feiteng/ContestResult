class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        acc = [0]
        for i in candiesCount:
            acc.append(acc[-1] + i)
        res = []
        for t, d, c in queries:
            mi = d + 1
            ma = (d + 1) * c
            if mi > acc[t + 1] or ma <= acc[t]:
                res.append(False)
            else:
                res.append(True)
        return res
