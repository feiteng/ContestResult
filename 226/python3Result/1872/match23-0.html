class Solution:
    def canEat(self, cd: List[int], qs: List[List[int]]) -> List[bool]:
        pre = [0]
        
        for ccd in cd:
            pre.append(ccd + pre[-1])
        # print(pre)
        
        ans = []
        
        for q1, q2, q3 in qs:
            mn = q2 + 1
            mx = q2 * q3 + q3
            # print(pre[q1], pre[q1 + 1], mn, mx)
            if (pre[q1 + 1] < mn) or pre[q1] >= mx:
                ans.append(False)
            else:
                ans.append(True)
        return ans
