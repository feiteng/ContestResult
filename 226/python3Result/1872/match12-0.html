class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pfs = [0]
        for count in candiesCount:
            pfs.append(pfs[-1] + count)
        
        ans = []
        for ft, fd, dc in queries:
            ans.append( (fd + 1) * dc > pfs[ft] and pfs[ft+1] >= fd + 1 )
        return ans
