class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        cum = [0]
        for c in candiesCount:
            cum.append(cum[-1]+c)
        print(cum)
        answer 
= []
        for fav_type, fav_day, daily_cap in queries:
            candy_start = cum[fav_type] + 1
            candy_end = cum[fav_type+1]
            eatable_start = fav_day + 1
            eatable_end = daily_cap * (fav_day + 1)
            #print(f"fav_type={fav_type}, fav_day={fav_day}, candy={(candy_start,candy_end)}, eatable=")
            if eatable_start <= candy_end and candy_start <= eatable_end:
                answer.append(True)
            else:
                answer.append(False)
        return answer
