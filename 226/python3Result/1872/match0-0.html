class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        
        culsum = [0]
        for can in candiesCount:
            culsum.append(culsum[-1] + can)
        res = []
        
        for i,day,limit in queries:
            lowbound = culsum[i] + 1
            highbound = culsum[i+1]
            if day+1 > highbound  or (day+1) * limit < lowbound:
                res.append(False)
            else:
                res.append(True)
        return res
