class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pref_sum = [0]
        for c in candiesCount:
            pref_sum.append(c + pref_sum[-1])
        res = []
        for typ, day, cap in queries:
            res.append(day + 1 <= pref_sum[typ + 1] and pref_sum[typ] < (day + 1) * cap)
        return res
