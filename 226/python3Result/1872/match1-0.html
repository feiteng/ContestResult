class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        presum = [0]
        s = 0
        for x in candiesCount:
            s += x
            presum.append(s)
        res = []
        for i, day, cap in queries:
            # day+1 <= x <= (day+1) * cap
            # presum[i]+1 <= y <= presum[i+1] 
            if (day+1) * cap < presum[i]+1 or day+1 > presum[i+1]:
                res.append(False)
            else:
                res.append(True)
        return res
