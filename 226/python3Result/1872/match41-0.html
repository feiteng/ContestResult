class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        seq4type = []
        last = 0
        for cnt in candiesCount:
            seq4type.append([last+1, last+cnt])
            last = last+cnt
            
        ans = []
        for fType, fDay, dailyCap in queries:
            least = fDay + 1
            most = (fDay + 1) * dailyCap
            if least > seq4type[fType][1] or most < seq4type[fType][0]:
                ans.append(False)
            else:
                ans.append(True)

        return ans
