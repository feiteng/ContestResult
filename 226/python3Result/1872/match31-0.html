class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pre = [0]
        acc = 0
        for cc in candiesCount:
            acc += cc
            pre.append(acc)
        # print('pre:', pre, '18:', pre[18])

        #
        ans = []
        for (ft, fd, cap) in queries:
            # 最小值   最大值
            if (pre[ft] // cap) <= fd < (pre[ft + 1]):
                ans.append(True)
            else:
                ans.append(False)

        return ans
