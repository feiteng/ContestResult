class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        f_sum = [0]
        for x in candiesCount: f_sum.append(f_sum[-1]+x)
        res = []
        for x, y, z in queries:
            res.append(f_sum[x+1]>y and f_sum[x]<(y+1)*z)
        return res
