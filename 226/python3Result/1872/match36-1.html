class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pref = [0]
        for x in candiesCount:
            pref.append(pref[-1] + x)
            
        ans = []
        for favoriteType, favoriteDay, dailyCap in queries:
            mn = 1 * favoriteDay + 1
            mx = dailyCap * (favoriteDay + 1)
            needMn = pref
[favoriteType] + 1
            needMx = pref[favoriteType + 1]
            
            if mx < needMn or needMx < mn:
                ans.append(False)
            else:
                ans.append(True)
        return ans
