class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        presum = [0]
        for c in candiesCount:
            presum.append(presum[-1] + c)
        
        #print(presum)
        
        res = []
        for typ, day, cap in queries:
            before = presum
[typ] # mininum candies that must be eaten
            after = presum[typ + 1] # maximum candies that must be eaten
            
            #print(typ, day, cap, before, after)
            
            if before >= (day+1) * cap or after <= day:
                res.append(False)
            else:
                res.append(True)
        
        return res
