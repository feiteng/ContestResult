class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        mustEat=[0]
        for i in candiesCount:
            mustEat.append(mustEat[-1]+i)
        # print(mustEat)
        res=[]
        for t, d, c in queries:
            soon=math.ceil((mustEat[t]+1)/c)
            slow=mustEat[t+1]
            # print(soon, slow)
            if d+1>=soon and d+1<=slow:
                res.append(True)
            else:
                res.append(False)
        return res
