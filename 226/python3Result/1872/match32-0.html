class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        pre = [0]
        cur = 0
        for cnt in candiesCount:
            cur += cnt
            pre.append(cur)
        # print(pre)
        
        ans = []
        
        for [ftype, fday, daily] in queries:
            if pre[ftype] < (fday + 1) * daily and fday < pre[ftype + 1]:
                ans.append(True)
            else:
                ans.append(False)
        
        
        
        return ans
