class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        candy=[0]
        for c in candiesCount:
            candy.append(candy[-1]+c)
        ret=[]
        for Type,day,cap in queries:
            ret.append(candy[Type+1]>=day+1 and candy[Type]<(day+1)*cap)
        return ret
