class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        
        presum = [0]
        for c in candiesCount:
            presum.append(presum[-1] + c)
        
        res = []
        for t, d, c in queries:
            t += 1
            
            lower = presum[t-1] + 1
            upper = presum[t]
            if 1*d >= upper or c * (d+1) < lower:
                res.append(False)
            else:
                res.append(True)
                
        return res
