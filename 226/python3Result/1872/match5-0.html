class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        cum = [0]
        for n in candiesCount:
            cum.append(cum[-1]+n)
        ret = []
        for t, d, c in queries:
            ret.append(cum[t+1]>d and cum[t] //c <= d)
        return ret
