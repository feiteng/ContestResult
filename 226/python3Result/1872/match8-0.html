class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        presum = [0]
        for c in candiesCount:
            presum.append(presum[-1]+c)
        
        ans = []
        for t,d,cap in queries:
            ans.append( presum[t+1] > d and presum[t]+1 <= (d+1)*cap )
        
        return ans
