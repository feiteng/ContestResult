class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        prefix = [0]
        
        for x in candiesCount:
            prefix.append(prefix[-1] + x)
            
        ans = []
        for t, day, cap in queries:
            total = prefix[t+1]
            previous = prefix[t]
            
            earliest = day < total
            latest = previous < (day + 1) * cap
            
            if earliest and latest:
                ans.append(True)
            else:
                ans.append(False)

        return ans
