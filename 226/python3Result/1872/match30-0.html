class Solution:
    def canEat(self, candiesCount: List[int], queries: List[List[int]]) -> List[bool]:
        currentSum = 0
        prefixSum = [currentSum]
        
        for candies in candiesCount:
            currentSum += candies
            prefixSum.append(currentSum)
            
        #print(prefixSum)
            
        result = []
        for candyType, day, dailyCap in queries:
            totalCandy = (day+1) * dailyCap
            if totalCandy <= prefixSum[candyType] or day > prefixSum[candyType+1]-1:
                result.append(False)
            else:
                result.append(True)
            
        return result
