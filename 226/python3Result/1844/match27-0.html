class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        d = collections.defaultdict(int)
        for i in range(lowLimit, highLimit + 1):
            ci = i
            s= 0
            while ci:
                s+= ci%10
                ci//=10
            d[s]+=1
        return max(d.values())
