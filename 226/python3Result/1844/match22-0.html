class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        boxes = collections.defaultdict(int)
        for i in range(lowLimit, highLimit+1):
            k = 0
            while i:
                k += i%10
                i//=10
            boxes[k] += 1
        return max(boxes.values())
