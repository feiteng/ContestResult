class Solution:
    def countBalls(self, l: int, r: int) -> int:
        
        def cal(n):
            res = 0
            while n:
                res += n % 10
                n //= 10
            return res
        
        d = Counter()
        for i in range(l, r+1):
            d[cal(i)] += 1
        
        return max(d.values())
