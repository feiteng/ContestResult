class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        visited = collections.defaultdict(int)
        for n in range(lowLimit, highLimit + 1):
            res = 0
            while n != 0:
                res += n % 10
                n = n // 10
            visited[res] += 1
            
        return max(visited.values())
