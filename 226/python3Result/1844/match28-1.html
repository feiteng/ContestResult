class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        cnt = collections.Counter()
        for i in range(lowLimit, highLimit + 1):
            digit = 0
            n = i
            while n:
                digit += n % 10
                n = n // 10
            cnt[digit] += 1
        return max(cnt.values())
