class Solution:
    def countBalls(self, low: int, high: int) -> int:
        bins = collections.defaultdict(int)
        for n in range(low, high+1):
            t = 0
            while n:
                t += n%10
                n = n // 10
            bins[t] += 1
        return max(bins.values())
