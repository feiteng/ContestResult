class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        cnt={}
        for num in range(lowLimit, highLimit+1):
            total=0
            while num!=0:
                total+=num%10
                num=num//10
            cnt[total]=cnt.get(total, 0)+1
            
        return max(cnt.values())
