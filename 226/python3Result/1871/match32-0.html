class Solution:
    def checkPartitioning(self, s: str) -> bool:
        dp = self.find_all_par(s)
        n = len(s)
        for left in range(n-2):
            for right in range(n-1, left+1, -1):
                if dp[0][left] and dp[right][n-1] and dp[left+1][right-1]:
                    return True
        return False
        
    def find_all_par(self, s):
        n = len(s)
        dp = [[False] * n for _ in range(n)]
        for i in range(n):
            dp[i][i] = True
            if i < n-1:
                dp[i][i+1] = (s[i] == s[i+1])
        for i in range(n-2, -1, -1):
            for j in range(i+2, n):
                dp[i][j] = (dp[i+1][j-1]) and (s[i] == s[j])
        return dp
