class Solution:
    def checkPartitioning(self, s: str) -> bool:
        if len(s) < 3:
            return False
        
        size = len(s)
        dp = [[False] * size for _ in range(size)]
        for i in range(size):
            dp[i][i] = True
        
        for i in range(size - 1):
            if s[i] == s[i + 1]:
                dp[i][i + 1] = True
        
        for length in range(3, size + 1):
            for i in range(size - length + 1):
                j = i + length - 1
                if dp[i+1][j-1] and s[i] == s[j]:
                    dp[i][j] = True
        
        for left in range(size):
            for right in range(left + 1, size):
                if dp[0][left] and dp[left + 1][right] and dp[right + 1][size - 1]:
                    return True
        
        # print(dp)
        # left, right = 1, size - 2
        # while left <= right:
        #     if not dp[0][left - 1]:
        #         left += 1
        #     elif not dp[right + 1][size - 1]:
        #         right -= 1
        #     elif not dp[left][right]:
        #         left += 1
        #         right -=1
        #     else:
        #         return True
        
        return False
