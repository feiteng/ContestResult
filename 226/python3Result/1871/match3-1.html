class Solution:
    def checkPartitioning(self, s: str) -> bool:
        
        n = len(s)
        dp = [[False] * n for i in range(n)]
        for i in range(n):
            dp[i][i] = True
            if i + 1 < n and s[i] == s[i+1]:
                dp[i][i+1] = True
        
        for j in range(n):
            for i in range(j-2, -1, -1):
                if s[i] == s[j] and dp[i+1][j-1]:
                    dp[i][j] = True
        
        #print(dp)
        for i in range(1, n):
            for j in range(i, n-1):
                if dp[0][i-1] and dp[i][j] and dp[j+1][n-1]:
                    return True
        return False
