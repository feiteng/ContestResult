class Solution:
    def checkPartitioning(self, s: str) -> bool:
        length = len(s)
        dp = [[False for _ in range(length)] for __ in range(length)]
        
        for i in range(length):
            dp[i][i] = True
        for i in range(1, length):
            if s[i - 1] == s[i]:
                dp[i - 1][i] = True
        
        for d in range(2, length):
            for i in range(length):
                j = i + d
                if j >= length:
                    break
                
                if s[i] == s[j] and dp[i + 1][j - 1] == True:
                    dp[i][j] = True
        
        for i in range(length):
            for j in range(i + 1, length):
                if dp[0][i] == True and dp[i + 1][j - 1] == True and dp[j][-1] == True:
                    return True
        return False
