class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[False]*(n+1) for _ in range(n)]
        for i in range(n):
            dp[i][0]=True
            dp[i][1]=True
        for k in range(2,n+1):
            for i in range(n-k+1):
                if s[i]==s[i+k-1]:
                    dp[i][k]=dp[i+1][
k-2]
        for i in range(1,n):
            for j in range(i+1,n):
                if dp[0][i] and dp[i][j-i] and dp[j][n-j]:return True
        return False
