class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        c = [[False for i in range(n)] for j in range(n)]
        for i in range(n):
            c[i][i] = True
        for i in range(n-1):
            if s[i] == s[i+1]:
                c[i][i+1] = True
        for l in range(3, n):
            for st in range(n):
                
if st+l-1 >= n:
                    break
                if s[st] == s[st+l-1] and c[st+1][st+l-2]:
                    c[st][st+l-1] = True
        for i in range(1, n):
            for j in range(i, n-1):
                if c[0][i-1] and c[i][j] and c[j+1][n-1]:
                    return True
        return False
