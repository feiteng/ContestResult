from pprint import pprint

class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        isPalindrome = [[False] * n for _ in range(n)]
        for i in range(n):
            isPalindrome[i][i] = True 
        for l in range(2, n+1):
            for i in range
(n-l+1):
                if(s[i]==s[i+l-1] and (l==2 or isPalindrome[i+1][i+l-2])):
                    isPalindrome[i][i+l-1] = True
        for i in range(n-2):
            for j in range(i+1, n-1):
                if isPalindrome[0][i] and isPalindrome[i+1][j] and isPalindrome[j+1][n-1]:
                    return True
        return False
