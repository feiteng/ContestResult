class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n= len(s)
        dp = [[False]*n for _ in range(n)]
        for i in range(n):
            dp[i][i] = True
            if i + 1 < n and s[i] == s[i+1]:
                dp[i][i+1] = True
        for k in range(3, n):
            for i in range(n-k+1):
                if s[i] == s[i+k-1] and dp[i+1][i+k-2]==True:
                    dp[i][i+k-1] = True 
            
        for i in range(n):
            for j in range(i+2, n):
                if dp[0][i] and dp[i+1][j-1] and dp[j][n-1]:
                    return True
        return False
