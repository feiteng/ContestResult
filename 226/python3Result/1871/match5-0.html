class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[False] * n for _ in range(n)]
        for d in range(n):
            for i in range(n - d):
                if d == 0:
                    dp[i][i + d] = True
                elif d == 1:
                    dp[i][i + d] = (s[i] == s[i + d])
                else:
                    dp[i][i + d] = (dp[i + 1][i + d - 1] and s[i] == s[i + d])
        for i in range(1, n - 1):
            for j in range(i, n - 1):
                if dp[0][i - 1] and dp[i][j] and dp[j + 1][n - 1]:
                    return True
        return False
