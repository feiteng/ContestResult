class Solution:
    def checkPartitioning(self, s: str) -> bool:
        N = len(s)
        dp = [[False] * N for _ in range(N)]
        for i in range(N):
            dp[i][i] = True
        
        # len == 6
        # stop at 3
        for l in range(2, N + 1):
            for start in range(N - l + 1):
                end = start + l - 1
                
                if s[start] == s[end] and (l == 2 or dp[start + 1][end - 1] == True):
                    dp[start][end] = True
        
        for i in range(N - 2): # first palindrome end
            for j in range(i + 1, N - 1): # second palindrome end
                if dp[0][i] == True and dp[i + 1][j] == True and dp[j + 1][N - 1] == True:
                    return True
        
        return False
