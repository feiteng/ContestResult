class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[True for j in range(n)] for i in range(n)]
        for i in range(1, n):
            for j in range(n - i):
                k = j + i
                dp[j][k] = s[j] == s[k] and dp[j + 1][k - 1]
        
        for i in range(1, n - 1):
            for j in range(i + 1, n):
                if dp[0][i - 1] and dp[i][j - 1] and dp[j][n - 1]:
                    return True
        
        return False
