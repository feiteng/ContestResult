class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[0 for _ in range(n)] for _ in range(n)]
        
        for i in range(n):
            dp[i][i] = 1
        for i in range(n-1):
            if s[i] == s[i+1]:
                dp[i][i+1] = 2
        
        for l in range(2, n):
            for i in range(n):
                if i + l >= n:
                    break
                if s[i] == s[i+l] and dp[i+1][i+l-1] > 0:
                    dp[i][i+l] = dp[i+1][i+l-1] + 2
        # print(dp)
        
        for i in range(n-1):
            for j in range(i, n-1):
                # print(s[0:i+1], s[i+1:j+1], s[j+1:], dp[0][i], dp[i+1][j], dp[j+1][n-1])
                if dp[0][i] and dp[i+1][j] and dp[j+1][n-1]:
                    return True
        return False
