class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[0]*n for i in range(n)]
        for i in range(n):
            dp[i][i] = 1
        for i in range(n-1):
            if s[i]==s[i+1]:
                dp[i][i+1] = 1
        for i in range(3,n+1):
            for j in range(n-i+1):
                l = j
                r 
= j+i-1
                if s[l]==s[r] and dp[l+1][r-1]:
                    dp[l][r] = 1
        # print(dp)
        for i in range(n):
            for j in range(i,n):
                if j < n-1 and dp[0][i] and dp[i+1][j] and dp[j+1][n-1]:
                    # print(i,j)
                    return True
        return False
