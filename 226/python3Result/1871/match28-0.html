import numpy as np

class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[0] * (n+1) for _ in range(n)]

        for i in range(n):
            dp[i][i+1] = 1
            dp[i][i] = 1
        for k in range(2, n):
            for i in range(n):
                if i+k <= n and dp[i+1][i+k-1] == 1 and s[i] == s[i+k-1]:
                    dp[i][i+k] = 1
        # print(np.array(dp))
        for i in range(n):
            for j in range(i+1, n):
                if dp[0][i] and dp[i][j] and dp[j][-1]:
                    # print(i, j)
                    return True
        
        return False
