class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        valid = [[False] * n for _ in range(n)]
        
        for i in range(n):
            valid[i][i] = True
            if i+1<n and s[i] == s[i+1]:
                valid[i][i+1] = True
        
        for step in range(2, n):
            for i in range(0, n-step):
                end = i + step
                if valid[i+1][end-1] and s[i] == s[end]:
                    valid[i][end] = True

        for i in range(n-2):
            for j in range(i+1, n-1):
                if valid[0][i] and valid[i+1][j] and valid[j+1][n-1]: return True
                
        return False
