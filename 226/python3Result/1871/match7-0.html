class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        check = [[False for _ in range(n)] for _ in range(n)]
        #预处理check数组，中心拓展，表示s[i]-s[j]是否回文
        for j in range(n):
            for i in range(j+1):
                if i == j:
                    check[i][j] = True
                elif j == i+1:
                    check[i][j] = (s[i] == s[j])
                else:
                    check[i][j] = (check[i+1][j-1] and s[i] == s[j])
        
        #print(check)
        for i in range(0, n-2):
            for j in range(i+2, n):
                if check[0][i] and check[i+1][j-1] and check[j][n-1]:
                    return True
        return False
