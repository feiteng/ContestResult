class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        is_pars = [[False for _ in range(n)] for _ in range(n+1)]
        for start in range(n):
            is_pars[0][start] = True
        for length in range(1, n+1):
            for start in range(n-length+1):
                end = start + length-1
                if length == 1:
                    is_pars[length][start] = True
                else:
                    is_pars[length][start] = is_pars[length-2][start+1] and s[start] == s[end]
        for length1 in range(1, n-1):
            for length3 in range(1, n-length1):
                if is_pars[length1][0] and is_pars[n-length1-length3][length1] and is_pars[length3][n-length3]:
                    return True
        return False
