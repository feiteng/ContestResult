class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n 
= len(s)
        dp = [[False]*n for i in range(n)]
        for i in range(n):
            dp[i][i] = True
        for i in range(n-1):
            if s[i] == s[i+1]:
                dp[i][i+1] = True
        for d in range(2, n):
            for i in range(n-d):
                
if dp[i+1][i+d-1] and s[i] == s[i+d]:
                    dp[i][i+d] = True
        for i in range(n-2):
            for j in range(i+2, n):
                if dp[0][i] and dp[i+1][j-1] and dp[j][n-1]:
                    return True
        return False
