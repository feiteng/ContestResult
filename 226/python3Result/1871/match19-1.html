class Solution:
    def checkPartitioning(self, s: str) -> bool:
        ln = len(s)
        ispal = [[False]*ln for i in range(ln)]
        
        for i in range(ln):
            ispal[i][i] = True
        
        for i in range(ln-1):
            if s[i] ==  s[i+1]:
                ispal[i][i+1] = True
        
        for sz in range(3, ln+1):
            for i in range(ln-sz+1):
                j = i+sz-1
                if s[i] == s[j]:
                    ispal[i][j] = ispal[i+1][
j-1]
        
        for i in range(ln):
            for j in range(i+1, ln-1):
                #print(i, j)
                #print(ispal[0][i], ispal[i+1][j], ispal[j+1][ln-1])
                if ispal[0][i] and ispal[i+1][j] and ispal[j+1][ln-1]:
                    return True
        
        return False
