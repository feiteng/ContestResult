class Solution:
    def checkPartitioning(self, s: str) -> bool:
        n = len(s)
        dp = [[False]*(n+1) for _ in range(n+1)]
        for i in range(n):
            dp[i][i+1] = True ## single word        
        for i in range(n+1):
            dp[i][i] = True ## s[i:i] is none, which is ture
        
        ## single word
        for c in range(n):
            for l in range(1,min(c+1,n-c)):
                #print(c,l)
                if s[c-l] == s[c+l] and dp[c-l+1][c+l]:
                    dp[c-l][c+l+1] = True
        
        ## double words
        for c in range(n):
            for l in range(1,min(c+2,n-c)):
                #print(c,l)
                if s[c-l+1] == s[c+l] and dp[c-l+2][c+l]:
                    dp[c-l+1][c+l+1] = True
        
        #for r in dp: print(r)
        
        
        for i in range(n+1):
            for j in range(i):
                if dp[0][j] and dp[j][i] and dp[i][n]: return True
        
        return False
