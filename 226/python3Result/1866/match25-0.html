class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        nei = collections.defaultdict(list)
        for i, j in adjacentPairs:
            nei[i].append(j)
            nei[j].append(i)
        res = []
        for i in nei:
            if len(nei[i]) == 1:
                s = i
                break
        
        seen = set()
        stack = [s]
        seen.add(s)
        while stack:
            s = stack.pop()
            res.append(s)
            for n in nei[s]:
                if n not in seen:
                    seen.add(n)
                    stack.append(n)
        return res
