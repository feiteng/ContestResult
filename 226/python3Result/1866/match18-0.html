def make_graph(edges):
    graph = defaultdict(list)
    for v1, v2 in edges:
        graph[v1].append(v2)
        graph[v2].append(v1)
    
    start_v = None
    for v in graph:
        if len(graph[v]) == 1:
            start_v = v
            break
    return start_v, graph


def dfs_nums(v, graph):
    nums = []
    
    def _dfs(v, parent):
        nonlocal nums
        nums.append(v)
        
        for nbor in graph[v]:
            if nbor != parent:
                _dfs(nbor, v)
    
    _dfs(v, None)
    return nums
        

class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        start_v, graph = make_graph(adjacentPairs)
        
        return dfs_nums(start_v, graph)
