class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        c 
= Counter()
        d = defaultdict(list)
        for a,b in adjacentPairs:
            c[a] += 1
            c[b] += 1
            d[a].append(b)
            d[b].append(a)
        
        for k,v in c.items():
            if v == 1:
                break
        
        # print(k)
        
        visited = set([k])
        res = [k]
        stack = [k]        
        
while stack:
            cur = stack.pop()
            for nex in d[cur]:
                if nex in visited:
                    continue
                res.append(nex)
                stack.append(nex)
                visited.add(nex)
        return res
