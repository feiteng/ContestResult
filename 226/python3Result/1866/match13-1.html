class Solution:
    def restoreArray(self, A: List[List[int]]) -> List[int]:
        n = len(A)
        res = []
        d 
= defaultdict(list)
        for u,v in A:
            d[u].append(v)
            d[v].append(u)
        # 先找到一端即可
        vis = set()
        st = []
        for v in d:
            if len(d[v])==1:
                st.append(v)
                break
        
while st:
            p = st.pop(0)
            res.append(p)
            vis.add(p)
            for nt in d[p]:
                if nt not in vis:
                    st.append(nt)
        return res
