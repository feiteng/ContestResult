class Solution:
    def restoreArray(self, A: List[List[int]]) -> List[int]:
        cnt = defaultdict(list)
        for x, y in A:
            cnt[x].append(y)
            cnt[y].append(x)
        
        res = []
        for k in cnt:
            if len(cnt[k]) == 1:
                res.append(k)
                break
        
        while len(res) != len(A) + 1:
            prev = res[-1]
            for k in cnt[prev]:
                if len(res) >= 2 and k == res[-2]:
                    continue
                else:
                    res.append(k)
                    break
        return res
