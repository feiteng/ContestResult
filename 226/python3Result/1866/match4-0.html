from collections import *
class Solution:
    def restoreArray(self, adjP: List[List[int]]) -> List[int]:
        deg = Counter()
        g = defaultdict(list)
        for adj in adjP:
            u, v = adj
            g[u].append(v)
            g[v].append(u)
            deg[u] += 1
            deg[v] += 1
        st = 0
        for k in deg:
            if deg[k] == 1:
                st = k
                break
        ret = []
        used = set()
        deq = deque()
        deq.append(st)
        used.add(st)
        while deq:
            f = deq.popleft()
            ret.append(f)
            for nxt in g[f]:
                if nxt not in used:
                    used.add(nxt)
                    deq.append(nxt)
        return ret
