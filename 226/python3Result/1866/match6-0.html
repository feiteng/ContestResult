class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        n = len(adjacentPairs)+1
        graph = collections.defaultdict(list)
        for n1,n2 in adjacentPairs:
            graph[n1].append(n2)
            graph[n2].append(n1)
        visited = set()
        start = None
        for i in graph:
            if len(graph[i]) == 1:
                start = i
                break
        res = [start]
        visited.add(start)
        while len(visited) < n:
            for node in graph[res[-1]]:
                if node not in visited:
                    res.append(node)
                    visited.add(node)
        return res
