class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        dic = collections.defaultdict(list)
        for e in adjacentPairs:
            dic[e[0]].append(e[1])
            dic[e[1]].append(e[0])
        stt = 0
        for ind, val in dic.items():
            if(len(val)) == 1:
                stt = ind
                break
        res = [stt]
        nxt = dic[stt][0]
        vis = set()
        vis.add(stt)
        while nxt not in vis:
            vis.add(nxt)
            res.append(nxt)
            for i in dic[nxt]:
                if i not in vis:
                    nxt = i
                    break
        return res
