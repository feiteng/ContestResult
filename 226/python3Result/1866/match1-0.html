class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        D=defaultdict(list)
        for a,b in adjacentPairs:
            D[a].append(b)
            D[b].append(a)
        for key,val in D.items():
            if len(val)==1:
                start=key
                break
        ret=[]
        last=None
        while len(ret)<len(D):
            ret.append(start)
            for dest in D[start]:
                if dest!=last:
                    last=start
                    start=dest
                    break
        return ret
