class Solution:
    def restoreArray(self, arr: List[List[int]]) -> List[int]:
        g = collections.defaultdict(list)
        for u, v in arr:
            g[u].append(v)
            g[v].append(u)
        source = 0
        for u, v in g.items():
            if len(v) == 1:
                source = u
                break
        
        ans = []
        def dfs(u, p):
            ans.append(u)
            for v in g[u]:
                if v == p:
                    continue
                dfs(v, u)
                
        dfs(source, None)
        return ans
