class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        d = defaultdict(list)
        for p in adjacentPairs:
            d[p[0]].append(p[1])
            d[p[1]].append(p[0])
        start = 0
        for k,v in d.items():
            if len(v) == 1:
                start = k
                break
                
        ans = [start]
        cur = start
        seen = {start}
        while len(ans) < len(adjacentPairs)+1:
            neighbours = d[cur]
            if neighbours[0] in seen:
                cur = neighbours[1]
            else:
                cur = neighbours[0]
            ans.append(cur)
            seen.add(cur)
        return ans
