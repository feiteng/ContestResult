from collections import defaultdict


class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        adj = defaultdict(set)
        for a, b in adjacentPairs:
            adj[a].add(b)
            adj[b].add(a)
            
        # pivot = adjacentPairs[0][0]
        # left = []
        # right = []
        
        for pivot, sibs in adj.items():
            if len(sibs) == 1:
                break
        right = []
        
        curr 
= pivot
        while adj[curr]:
            sib = adj[curr].pop()
            adj[sib].remove(curr)
            right.append(sib)
            curr = sib
        
        return [pivot] + right
