class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        d = defaultdict(list)
        n = len(adjacentPairs)
        for a, b in adjacentPairs:
            d[a].append(b)
            d[b].append(a)
        start = None
        
for num in d:
            if len(d[num]) == 1:
                start = num
                break
        visited = set()
        visited.add(start)
        ans = [start]
        num = start
        for i in range(n):
            for nxt in d[num]:
                if nxt not in visited:
                    visited.add(nxt)
                    ans.append(nxt)
                    num = nxt
        return ans
