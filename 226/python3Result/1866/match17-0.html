class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        dic, q = defaultdict(set), []
        for x, y in adjacentPairs:
            dic[x].add(y)
            dic[y].add(x)
        for x in dic.keys():
            if len(dic[x]) == 1:
                q.append(x)
        res = []
        while q:
            num = q.pop()
            res.append(num)
            for pair in dic[num]:
                dic[pair].remove(num)
                if len(dic[pair]) == 1:
                    q.append(pair)
        return res
