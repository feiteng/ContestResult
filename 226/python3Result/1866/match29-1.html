class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        for pair in adjacentPairs:
            graph[pair[0]].append(pair[1])
            graph[pair[1]].append(pair[0])
        start = None
        for key, value in graph.items():
            if len(value) == 1:
                start = key
                break
        ret = [start]
        visited = set([start])
        while len(ret) < len(adjacentPairs)+1:
            # print(ret)
            # print(graph[ret[-1]])
            for possible in graph[ret[-1]]:
                if len(ret) == 1 or possible not in visited:
                    ret.append(possible)
                    visited.add(possible)
        return ret
