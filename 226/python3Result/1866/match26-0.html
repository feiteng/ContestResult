class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        neighbor = defaultdict(list)
        n = len(adjacentPairs) + 1
        for a, b in adjacentPairs:
            neighbor[a].append(b)
            neighbor[b].append(a)
        
        for key in neighbor:
            if len(neighbor[key]) == 1:
                start = key
                break
        taken = set()
        taken.add(start)
        ans = [start]
        for i in range(n - 1):
            curr = ans[-1]
            nxt_list = neighbor[curr]
            for cand in nxt_list:
                if cand in taken:
                    continue
                nxt = cand
            ans.append(nxt)
            taken.add(curr)
        return ans
