class Solution:
    def restoreArray(self, pairs: List[List[int]]) -> List[int]:
        dic = collections.defaultdict(list)
        for a, b in pairs:
            dic[a].append(b)
            dic[b].append(a)
        
        res = []
        seen = set()
        for a in dic:
            if len(dic[a]) == 1:
                res.append(a)
                seen.add(a)
                break
        n = len(pairs) + 1
        while len(res) < n:
            for b in dic[res
[-1]]:
                if b not in seen:
                    res.append(b)
                    seen.add(b)
        
        return res
