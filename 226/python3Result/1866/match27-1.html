class Solution:
    def restoreArray(self, adj: List[List[int]]) -> List[int]:
        info = collections.defaultdict(list)
        for u, v in adj:
            info[u].append(v)
            info[v].append(u)
        head = 0
        for k in info.keys():
            if len(info[k]) == 1:
                head = k
                break
        n = len(adj) + 1
        res = []
        cur = head
        count = 0
        pre = cur
        while count < n:
            res.append(cur)
            temp = cur
            for u in info[cur]:
                if u == pre:
                    continue
                temp = u
            pre = cur
            cur = temp
            count += 1
        return res
