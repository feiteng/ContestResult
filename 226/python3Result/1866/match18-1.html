class Solution:
    def restoreArray(self, A: List[List[int]]) -> List[int]:
        n = len(A) + 1
        g = collections.defaultdict
(set)
        for u, v in A:
            g[u].add(v)
            g[v].add(u)
        st = 0
        for u in g:
            if len(g[u]) == 1:
                st = u
                break
        res = []
        seen = set()
        def dfs(p):
            res.append(p)
            seen
.add(p)
            for q in g[p]:
                if q not in seen: dfs(q)
        dfs(st)
        return res
