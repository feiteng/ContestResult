class Solution:
    def restoreArray(self, adj: List[List[int]]) -> List[int]:
        ad = collections.defaultdict(list)
        for i, j in adj:
            ad[i].append(j)
            ad[j].append(i)
        ans = []   
        for k, v in ad.items():
            if len(v) == 1:
                q = [k]
                visited = set()
                
while q:
                    n = q.pop()
                    visited.add(n)
                    ans.append(n)
                    for a in ad[n]:
                        if a in visited:
                            continue
                        q.append(a)
                break
        return ans
