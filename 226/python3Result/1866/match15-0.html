class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        n = len(adjacentPairs) + 1
        ans = [0] * n
        cnt = Counter()
        G = defaultdict(list)
        for a, b in adjacentPairs:
            cnt[a] += 1
            cnt[b] += 1
            G[a].append(b)
            G[b].append(a)
        cand = [k for k, v in cnt.items() if v == 1]
        start = cand[0]
        ans = [start]
        q = deque([start])
        seen = {start}
        while q:
            i = q.popleft()
            for j in G[i]:
                if j in seen: continue
                seen.add(j)
                ans.append(j)
                q.append(j)
        return ans
