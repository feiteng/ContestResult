class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        for x,y in adjacentPairs:
            graph[x].append(y)
            graph[y].append(x)
        terminus = [elem for elem in graph if len(graph[elem]) == 1][0]
        visited = set()
        stack = [terminus]
        output = []
        while stack:
            elem = stack.pop(0)
            output.append(elem)
            visited.add(elem)
            for next_elem in graph[elem]:
                if next_elem not in visited:
                    stack.append(next_elem)
        return output
