from collections import defaultdict, deque

class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        d = defaultdict(list)
        for a, b in adjacentPairs:
            d[a].append(b)
            d[b].append(a)
        
        n = len(adjacentPairs)
        res = []
        q = deque()
        visited = set()
        for v, l in d.items():
            if len(l) == 1:
                q.append(v)
                visited.add(v)
                break
        
        
while q:
            cur = q.popleft()
            res.append(cur)
            
            for nxt in d[cur]:
                if nxt not in visited:
                    visited.add(nxt)
                    q.append(nxt)
        return res
