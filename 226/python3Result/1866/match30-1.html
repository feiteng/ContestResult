class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        edges = collections.defaultdict(set)
        
        for u, v in adjacentPairs:
            edges[u].add(v)
            edges[v].add(u)
        
        start = None
        for x in edges.keys():
            if len(edges[x]) == 1:
                start = x
                break
        
        ans = [start]
        previous = -1000000000
        
        for _ in range(len(adjacentPairs)):
            current = ans[-1]
            if previous in edges[current]:
                edges[current].remove(previous)
                
            gonext = list(edges[current])[0]
            ans.append(gonext)
            previous = current
        return ans
