class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        c = defaultdict(list)
        for i, j in adjacentPairs:
            c[i].append(j)
            c[j].append(i)
        res = []
        for k in c:
            if len(c[k]) == 1:
                res.append(k)
                break
        # print(c)
        while len(res) < len(adjacentPairs) + 1:
            k = res[-1]
            if len(c[k]) == 1:
                res.append(c[k][0])
            else:
                for i in c[k]:
                    if i != res[-2]:
                        res.append(i)
                        break
        return res
