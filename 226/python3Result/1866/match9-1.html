class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        g = collections.defaultdict(list)
        for pair in adjacentPairs:
            g[pair[0]].append(pair[1])
            g[pair[1]].append(pair[0])
            
        
        leaf = 0
        for i, lst in g.items():
            if len(lst) == 1:
                leaf = i
                break
                
        lst = []
        s = set()
        
        def dfs(n, g, lst, s):
            if n in s:
                return
            lst.append(n)
            s.add(n)
            for x in g[n]:
                dfs(x, g, lst, s)
        
        dfs(leaf, g, lst, s)
        
        return lst
