class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        adj = collections.defaultdict(set)
        for a, b in adjacentPairs:
            adj[a].add(b)
            adj[b].add(a)
            
        x = -1
        for a, lst in adj.items():
            if len(lst) == 1:
                x = a
                break
        
        res = [x]
        while len(adj[x]):
            nxt = adj[x].pop()
            adj[nxt].remove(x)
            x = nxt
            res.append(x)
            
        return res
