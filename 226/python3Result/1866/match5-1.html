class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        pair=collections.defaultdict(list)
        for x,y in adjacentPairs:
            pair[x].append(y)
            pair[y].append(x)
        ret=[]
        used=set()
        for k,v in pair.items():
            if len(v)==1:
                ret.append(k)
                used.add(k)
                break
        while len(ret)<=len
(adjacentPairs):
            for nex in pair[ret[-1]]:
                if nex not in used:
                    ret.append(nex)
                    used.add(nex)
        return ret
