class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        N = len(adjacentPairs) + 1
        g = collections.defaultdict(list)
        for a, b in adjacentPairs:
            g[a].append(b)
            g[b].append(a)
        for node in g:
            if len(g[node]) == 1:
                start = node
                break
                
        visited = set()
        res = [start]
        visited.add(start)
        while len(visited) < N:
            node = res[-1]
            for nei in g[node]:
                if nei in visited:
                    continue
                visited.add(nei)
                res.append(nei)
        return res
