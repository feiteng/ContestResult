class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        dd = collections.defaultdict(list)
        for l, r in adjacentPairs:
            dd[l].append(r)
            dd[r].append(l)
        start = None
        for key, value in dd.items():
            if len(value) == 1:
                start = key
                break
        rv = [start, dd[start][0]]
        while True:
            curr = rv[-1]
            if len(dd[curr]) == 1:
                break
            else:
                for cdate in dd[curr]:
                    if cdate != rv[-2]:
                        rv.append(cdate)
                        break
        return rv
