class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        d = defaultdict(list)
        nadj = len(adjacentPairs)
        
        for u, v in adjacentPairs:
            d[u].append(v)
            d[v].append(u)
        
        first = None
        for k, v in d.items():
            if len(v) == 1:
                first = k
                break
        
        res = [first]
        seen = set()
        seen.add(first)
        
        while len(res) < nadj+1:
            pre = res[-1]
            for a in d[pre]:
                if a not in seen:
                    res.append(a)
                    seen.add(a)
                    break
        
        return res
