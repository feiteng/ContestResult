class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        for u, v in adjacentPairs:
            graph[u].append(v)
            graph[v].append(u)
        
        seen = set()
        res = []
        
        def dfs(v):
            res.append(v)
            seen.add(v)
            
            for u in graph[v]:
                if u not in seen:
                    dfs(u)
            
        for k in graph:
            if len(graph[k]) == 1:
                dfs(k)
                break
        return res
