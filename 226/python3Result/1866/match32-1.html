class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        adj=collections.defaultdict(list)
        for x,y in adjacentPairs:
            adj[x].append(y)
            adj[y].append(x)
        cur=0
        for k,v in adj.items():
            if len(v)==1:
                cur=k
                break
        re=[]
        last=cur
        while True:
            re.append(cur)
            nxt=[x for x in adj[cur] if x!=last]
            if not nxt:
                return re
            nxt=nxt[0]
            last=cur
            cur=nxt
