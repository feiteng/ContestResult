class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        seen = set()
        M = defaultdict(set)
        for i,j in adjacentPairs:
            M[i].add(j)
            M[j].add(i)
            
        for i,s in M.items():
            if len(s) == 1:
                first = i
                break
        
        N = len(adjacentPairs) + 1
        out = [first]
        seen.add(first)
        while len(out) < N:
            for i in M[out[-1]]:
                if i in seen:
                    continue
                out.append(i)
                seen.add(i)
        return out
