class Solution:
    def restoreArray(self, ap: List[List[int]]) -> List[int]:
        gph = collections.defaultdict(list)
        for a in ap:
          gph[a[0]].append(a[1])
          gph[a[1]].append(a[0])
        
        st = None
        for k, v in gph.items():
          if len(v)==1:
            st = k
            break
        
        res = []
        visited = set([])
        def dfs(v):
          
          res.append(v)
          visited.add(v)
          for ch in gph[v]:
            if ch not in visited:
              dfs(ch)
        
        dfs(st)
        return res
