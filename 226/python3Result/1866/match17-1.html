# topological sorting

class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        adjs = defaultdict(set)
        for node_1, node_2 in adjacentPairs:
            adjs[node_1].add(node_2)
            adjs[node_2].add(node_1)
        
        q = deque()
        for node in adjs:
            if len(adjs[node]) == 1:
                q.append(node)
        tail = q.pop()
        res 
= []
        while q:
            cur = q.pop()
            res.append(cur)
            for neighbor in adjs[cur]:
                adjs[neighbor].remove(cur)
                if len(adjs[neighbor]) == 1:
                    q.append(neighbor)
            adjs.pop(cur)
        
        res.append(tail)
        return res
