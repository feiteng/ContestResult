from collections import defaultdict

class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        
        for pair in adjacentPairs:
            graph[pair[0]].append(pair[1])
            graph[pair[1]].append(pair[0])
            
        start = None
        for key, val in graph.items():
            if len(val) == 1:
                start = key
                break
                
        res = [start]
        prev = start
        start = graph[start][0]
        res.append(start)
        
        while True:
            nodes = graph[start]
            if len(nodes) == 1:
                return res
            
            new = nodes[0] if nodes[0] != prev else nodes[1]
            res.append(new)
            
            prev = start
            start = new
            
        return res
