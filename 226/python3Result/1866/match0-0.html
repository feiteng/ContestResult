class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        d = defaultdict(set)
        for u, v in adjacentPairs:
            d[u].add(v)
            d[v].add(u)
        for i in d:
            if len(d[i]) == 1:
                r = [i]
                while d[i]:
                    j = d[i].pop()
                    d[j].discard(i)
                    r.append(j)
                    i = j
                return r
