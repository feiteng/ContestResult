class Solution:
    def restoreArray(self, pairs: List[List[int]]) -> List[int]:
        n = len(pairs) + 1
        num_neis = collections.defaultdict(list)
        for a, b in pairs:
            num_neis[a].append(b)
            num_neis[b].append(a)
            
        for i, neis in num_neis.items():
            if len(neis) == 1:
                start = i
                break
        
        seen = set([start])
        res = [start]
        
        
while len(res) < n:
            cur_end = res[-1]
            for nei in num_neis[cur_end]:
                if nei in seen:
                    continue
                seen.add(nei)
                res.append(nei)
                
        return res
