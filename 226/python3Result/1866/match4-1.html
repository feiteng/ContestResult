class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        deg = defaultdict(int)
        for x, y in adjacentPairs:
            graph
[x].append(y)
            graph[y].append(x)
            deg[x] += 1
            deg[y] += 1
            
        deg_1_node = None
        for k in deg:
            if deg[k] == 1:
                deg_1_node = k
                break
        stack = [deg_1_node]
        ans = []
        vis = set
([deg_1_node])
        while stack:
            cur = stack.pop()
            ans.append(cur)
            for n in graph[cur]:
                if n not in vis:
                    vis.add(n)
                    stack.append(n)
        return ans
