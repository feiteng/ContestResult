class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        g = defaultdict(list)
        for n1, n2 in adjacentPairs:
            g[n1].append(n2)
            g[n2].append(n1)
        for n in g:
            if len(g[n]) == 1:
                root = n
                break
        q = [root]
        vis = set()
        vis.add(root)
        res = []
        while q:
            node = q.pop(0)
            res.append(node)
            for adj in g[node]:
                if adj not in vis:
                    vis.add(adj)
                    q.append(adj)
        return res
