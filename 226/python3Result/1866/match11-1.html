class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        adj = collections.defaultdict(list)
        for edge in adjacentPairs:
            adj[edge[0]].append(edge[1])
            adj[edge[1]].append(edge[0])
        n = len(adj)
        ans = []
        take = 0 
        for i in adj:
            if len(adj[i])==1:
                take = i
                break
        taken = set()
        # print(take)
        while len(ans)!=n:
            ans
.append(take)
            taken.add(take)
            for j in adj[take]:
                if j not in taken:
                    take = j
                    break
        return ans
