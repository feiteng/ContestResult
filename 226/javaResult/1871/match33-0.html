class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        char[] str = s.toCharArray();
        boolean[][] dp = new boolean[n][n+1];
        for (int i = 0; i < n; i++) {
            dp[i][1] = true;
            dp[i][0] = true;
        }
        for(int length = 2;length<n+1;length++){
            for(int i = 0;i<n;i++){
                if(i+length-1>=n){
                    continue;
                }

                dp[i][length] = dp[i+1][length-2]&&(str[i]==str[i+length-1]);
            }
        }
        for(int i = 1;i<n;i++){
            for(int j = i+1;j<n;j++){
                if(dp[0][i]&&dp[i][j-i]&&dp[j][n-j]){
                    return true;
                }
            }
        }
        return false;
    }
}
