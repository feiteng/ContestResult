class Solution {
  public boolean checkPartitioning(String s) {
    final int N = s.length();
    boolean[][] M = new boolean[N][N];
    for (int i = N - 1; i >= 0; i--) {
      M[i][i] = true;
      for (int j = i + 1; j < N; j++) {
        if (s.charAt(i) == s.charAt(j) && (j == i + 1 || M[i + 1][j - 1])) {
          M[i][j] = true;
        }
      }
    }
    
    for (int j = 1; j < N; j++) {
      if (!M[0][j - 1]) {
        continue;
      }
      for (int k = j + 1; k < N; k++) {
        if (M[j][k - 1] && M[k][N - 1]) {
          return true;
        }
      }
    }

    return false;
  }
}
