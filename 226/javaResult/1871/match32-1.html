class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n+1][n+1];
        for (int i = 0; i < n; i++) {
            dp[i][i+1] = true;
        }
        for (int i = 0; i < n-1; i++) {
            if (s.charAt(i) == s.charAt(i+1)) {
                dp[i][i+2] = true;
            }
        }
        for (int k = 3; k < n; k++) {
            
for (int i = 0; i < n - k + 1; i++) {
                if (dp[i+1][i+k-1] && s.charAt(i) == s.charAt(i+k-1)) {
                    dp[i][i+k] = true;
                }
            }
        }
        for (int i = 0; i < n-1; i++) {
            
for (int j = 0; j < n; j++) {
                if (dp[0][i] && dp[i][j] && dp[j][n]) {
                    return true;
                }
            }
        }
        return false;
    }
}
