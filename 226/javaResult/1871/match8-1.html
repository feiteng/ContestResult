class Solution {
    public boolean checkPartitioning(String s) {
        int len = s.length();
        boolean[][] dp = new boolean[len][len];
        
        for (int l = 1; l <= len; l ++) {
            for (int i = 0; i < len - l + 1; i ++) {
                int j = i + l - 1;
                
                if (i == j || i + 1 == j && s.charAt(i) == s.charAt(j) || s.charAt(i) == s.charAt(j) && dp[i + 1][j - 1]) {
                    dp[i][j] = true;
                } else {
                    dp[i][j] = false;
                
}
            }
        }
        
        for (int i = 0; i < len - 2; i ++) {
            for (int j = i + 1; j < len - 1; j ++) {
                if (dp[0][i] && dp[i + 1][j] && dp[j + 1][len - 1]) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
