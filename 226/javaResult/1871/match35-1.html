class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        int res = 0;
        boolean[][] dp = new boolean[n][n];
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j < n; j++) {
                dp[i][j] = s.charAt(i) == s.charAt(j) && (j - i  + 1 < 3 || dp[i + 1][j - 1]);
            
}
        }
                                                
        for(int i=1;i<n;++i){
            for(int j=i;j+1<n;++j){
                if(dp[0][i-1] && dp[i][j] && dp[j+1][n-1]) return true
;
            }
        }
        return false;

    }
}
