class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n];
        for (int i = 0; i < n; i++) {
            dp[i][i] = true;
        }
        for (int j = 1; j < n; j++) {
            for (int i = j - 1; i >= 0; i--) {
                if (s.charAt(i) != s.charAt(j)) {
                    dp[i][j] = false;
                    continue;
                }
                dp[i][j] = (j - i <= 2) ? true : dp[i + 1][j - 1];
            
}
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 2; j < n; j++) {
                if (dp[0][i] && dp[i + 1][j - 1] && dp[j][n - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
}
