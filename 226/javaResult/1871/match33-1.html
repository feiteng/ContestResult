class Solution {
    public boolean checkPartitioning(String s) {
        final int N = s.length();
        final char[] S = s.toCharArray();
        
        boolean[][] palindromic = new boolean[N][N + 1];
        for (int i = 0; i < N; ++i) {
            palindromic[i][i] = true;
            palindromic[i][i + 1] = true;
        }
        
        for (int m = 0; m < N; ++m) {
            for (int i = m - 1, j = m + 1; i >= 0 && j < N; --i, ++j) {
                if (S[i] != S[j]) {
                    break;
                }
                palindromic[i][j + 1] = true;
            }
            for (int i = m, j = m + 1; i >= 0 && j < N; --i, ++j) {
                if (S[i] != S[j]) {
                    break;
                
}
                palindromic[i][j + 1] = true;
            }
        }
        
        for (int i = 1; i < N; ++i) {
            for (int j = i + 1; j < N; ++j) {
                if (palindromic[0][i] && palindromic[i][j] && palindromic[j][N]) {
                    return true;
                }
            }
        }
        return false;
    }
}
