class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        
        boolean[][] dp = new boolean[n][n];
        for(int i=n-1; i>=0; i--){
            dp[i][i] = true;
            if(i<n-1) dp[i][i+1] = s.charAt(i) == s.charAt(i+1);
            for(int j=i+2; j<n; j++){
                dp[i][j] = dp[i+1][j-1] && s.charAt(i) == s.charAt(j);
            }
        }
        
        for(int i=1; i<n-1; i++){
            for(int j=i; j<n-1; j++){
                if(dp[0][i-1] && dp[i][j] && dp[j+1][n-1]) return true;
            }
        }
        
        return false;
    }
}
