class Solution {
    public boolean checkPartitioning(String s) {
    int n = s.length();
    boolean[][] dp = new boolean[n][n];
    for (int l = 1; l <= n; l++) {
      for (int i = 0; i < n; i++) {
        int j = i + l - 1;
        if (j >= n) {
          continue;
        }
        if (l == 1) {
          dp[i][j] = true;
        } else 
if (l == 2) {
          dp[i][j] = s.charAt(i) == s.charAt(j);
        } else {
          dp[i][j] = dp[i + 1][j - 1] && (s.charAt(i) == s.charAt(j));
        
}
      }
    }
    for (int i = 0; i < n; i++) {
      for (int j = i + 1; j < n - 1; j++) {
        // [0, i] [i + 1, j] [j + 1, n - 1]
        boolean tmp = dp[0][i] && dp[i + 1][j] && dp[j + 1][n - 1];
        
if (tmp) {
          return true;
        }
      }
    }
    return false;
  }
}
