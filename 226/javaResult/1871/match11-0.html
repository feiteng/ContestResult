class Solution {
  public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] valid = new boolean[n][n];

        for (int i = 0; i < n; i++) {
            for (int j = i; j >= 0; j--) {
                if (i == j) {
                    valid[j][i] = true;
                } else if (j + 1 == i) {
                    valid[j][i] = s.charAt(i) == s.charAt(j);
                } else {
                    valid[j][i] = valid[j + 1][i - 1] && (s.charAt(i) == s.charAt(j));
                }
            }
        }

        boolean[] dp = new boolean[n];
        //dp[0] = true;
        for (int i = 1; i < n; i++) {
            for (int j = i; j >= 1; j--) {
                if (valid[0][j - 1] && valid[j][i]) {
                    dp[i] = true;
                    break;
                }
            }
        }

        for (int j = n - 1; j >= 2; j--) {
            if (dp[j - 1] && valid[j][n - 1]) {
                return true;
            }
        }
        return false;
    }

}
