public class Solution {
	public boolean checkPartitioning(String s) {
		int n = s.length();
		boolean[][] dp= new boolean[n][n];
		for (int i = 0; i < n; ++i) {
			dp[i][i] = true;
		}

		for (int l = 2; l <= n; ++l) {
			for (int i = 0; i <= n - l; ++i) {
				dp[i][i + l - 1] = (s.charAt(i) == s.charAt(i + l - 1)) && (l > 2 ? dp[i + 1][i + l - 2] : true);
			}

		}
		for (int i = 0; i < n - 2; ++i) {
			for (int j = i + 1; j < n - 1; ++j) {
				if (dp[0][i] && dp[i + 1][j] && dp[j + 1][n - 1]) {
					return true;
				}

			}
		}
		return false;
	}
}
