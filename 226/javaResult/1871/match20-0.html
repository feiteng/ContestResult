class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n + 1];
        for (int i = 0; i < n; i++) {
            dp[i][i + 1] = true;
        }
        for (int len = 2; len <= n; len++) {
            for (int start = 0; start + len <= n; start++) {
                int end = start + len;
                char first = s.charAt(start);
                char last = s.charAt(end - 1);
                dp[start][end] = (first == last) && (len == 2 || dp[start + 1][end - 1]);
            }
        }
        for (int start2 = 1; start2 < n; start2++) {
            for (int start3 = start2 + 1; start3 < n; start3++) {
                if (dp[0][start2] && dp[start2][start3] && dp[start3][n]) {
                    return true;
                }
            }
        }
        return false;
    }
}
