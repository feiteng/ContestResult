class Solution {
    public boolean checkPartitioning(String word) {
        int n = word.length();
        
int dp[] = new int[n];
        boolean isPalind[][] = new boolean[n][n];
        for(int i=0;i<n;i++){
            int min = i;
            for(int j=0;j<=i;j++){
                if(word.charAt(j) == word.charAt(i) && (i-j<2 || isPalind[j+1][i-1])){
                    min = Math.min(min, j==0 ? 0:dp[j-1]+1);
                    isPalind[j][i] = true;
                
}
            }
            dp[i] = min;
        }
        List<Integer> last = new ArrayList<>();
        for(int i=0;i<n;i++){
            if(isPalind[i][n-1] == true)
                last.add(i);
        }
        for(int l:last){
            for(int i=0;i<l;i++){
                if(isPalind[0][i] && isPalind[i+1][l-1]){
                    return true;
                }
            }
        }
        return false;
    }
}
