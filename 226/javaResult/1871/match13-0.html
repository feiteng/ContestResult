class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        int dynamic[] = new int[n];
        boolean check[][] = new boolean[n][n];

        for(int i=0;i<n;i++){
            int minimum = i;
            for(int j=0;j<=i;j++){
                if(s.charAt(j) == s.charAt(i) && (i-j<2 || check[j+1][i-1])){
                    check[j][i] = true;
                    minimum = Math.min(minimum, j==0 ? 0 : dynamic[j-1]+1);
                }
            }
            dynamic[i] = minimum;
        }
        
        List<Integer> end = new ArrayList<>();
        for(int i = 0;i<n;i++){
            if(check[i][n-1])
                end.add(i);
        }

        for (int left : end){
            for(int i=0;i<left;i++){
                if(check[0][i] && check[i+1][left-1]){
                    return true;
                }
            }
        }

        return false;
    }
}
