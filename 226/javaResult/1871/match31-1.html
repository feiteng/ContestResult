class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        
        boolean [][]dp = new boolean[n][n];
        
        for(int len=1;len<=n;len++){
            for(int i=0;i+len-1<n;i++){
                int j = i+len-1;
                if(i==j){
                    dp[i][j] = true;
                    continue;
                }
                dp[i][j] = (s.charAt(i)==s.charAt(j))?((i+1<=j-1?dp[i+1][j-1]:true)):false;
            
}
        }
        
        for(int j=0;j<n;j++){
            if(dp[0][j]){
                for(int k=j+1;k<n;k++){
                    if(dp[j+1][k]){
                        if(dp[k+1][n-1])
                            return true;
                    
}
                }    
            }
        }
        //System.out.println(dp[4][5]);
        return false;
    }
}
