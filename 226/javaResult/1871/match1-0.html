class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] isPali = new boolean[n][n];
        for (int i = n - 1; i > -1; i--) {
            for (int j = n - 1; j >= i; j--) {
                if (i == j) {
                    isPali[i][j] = true;
                } else {
                    if (j - i == 1) {
                        isPali[i][j] = s.charAt(i) == s.charAt(j);
                    } else {
                        isPali[i][j] = s.charAt(i) == s.charAt(j) && isPali[i+1][j-1];
                    }
                }
            }
        }
        for (int i = 1; i < n - 1; i++) {
            for (int j = i; j < n - 1; j++) {
                if (isPali[i][j] && isPali[0][i-1] && isPali[j+1][n-1]) {
                    return true;
                }
            }
        }
        return false;
    }
}
