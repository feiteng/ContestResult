class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] dp1 = new boolean[n][n];
        for (int length=1; length<n; length++) {
            for (int start=0; start<n; start++) {
                int end = start + length - 1;
                if (end >= n) {
                    break;
                }
                
if (length == 1) {
                    dp1[start][end] = true;
                }
                else if (length == 2) {
                    dp1[start][end] = (s.charAt(start) == s.charAt(end));
                }
                else {
                    dp1[start][end] = ((s.charAt(start) == s.charAt(end)) && (dp1[start+1][end-1]));
                }
            }
        }
        boolean[] dp2 = new boolean[n];
        for (int end=1; end<n; end++) {
            for (int mid=0; mid<end; mid++) {
                if (dp1[0][mid] && dp1[mid+1][end]) {
                    dp2[end] = true;
                    break;
                }
            }
        }
        for (int end=1; end<n; end++) {
            if (dp2[end] && dp1[end+1][n-1]) {
                return true;
            }
        }
        return false;
    }
}
