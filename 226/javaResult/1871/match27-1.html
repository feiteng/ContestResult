class Solution {
    public boolean checkPartitioning(String s) {
        int len = s.length();
        boolean[][] isPalind = new boolean[len][len];
        for (int i = 0; i < len; i ++) {
            for (int j = i; j >= 0; j --) {
                if (i == j) {
                    isPalind[j][i] = true;
                } else {
                    
if (j == i - 1) {
                        isPalind[j][i] = s.charAt(j) == s.charAt(i);
                    } else {
                        isPalind[j][i] = (s.charAt(j) == s.charAt(i)) && isPalind[j+1][i-1];
                    }
                
}
            }
        }
        
        for (int i = 0; i < len - 1; i ++) {
            for (int j = i + 1; j < len - 1; j ++) {
                
if (isPalind[0][i] && isPalind[i + 1][j] && isPalind[j + 1][len - 1]) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
