class Solution {
    public boolean checkPartitioning(String s) {
        int len = s.length();
        boolean[][] dp = new boolean[len][len];
        for (int j = 0; j < len; j++) {
            for (int i = j; i >= 0; i--) {
                if (s.charAt(i) == s.charAt(j)) {
                    if (j - i < 2) {
                        dp[i][j] = true;
                    } else {
                        dp[i][j] = dp[i + 1][j - 1];
                    
}
                }
            }
        }
        for (int j = 1; j < len - 1; j++) {
            for (int i = 0; i < j; i++) {
                if (dp[0][i] && dp[i + 1][j] && dp[j + 1][len - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
}
