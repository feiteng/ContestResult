class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] mem = new boolean[n][n];
        for (int i = n - 1; i >= 0; i--) {
            mem[i][i] = true;
            for (int j = i + 1; j < n; j++) {
                
mem[i][j] = s.charAt(i) == s.charAt(j) && (i + 1 >= j - 1 || mem[i + 1][j - 1]);
            }
        }
        
        for (int i = 0; i < n - 2; i++) {
            if (mem[0][i]) {
                for (int j = i + 1; j < n - 1; j++) {
                    if (mem[i + 1][j] && mem[j + 1][n - 1]) {
                        return true;
                    
}
                }
            }
        }
        return false;
    }
}
