class Solution {
    public boolean checkPartitioning(String s) {
        int m = s.length();
        boolean[][] palin = new boolean[m][m];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j <= i; j++) {
                
if (s.charAt(i) == s.charAt(j)) {
                    if (i - j <= 2) {
                        palin[j][i] = true;
                    } else if (palin[j + 1][i - 1]) {
                        palin[j][i] = true;
                    
}
                }
                
            }
        }
        
        // for (int i = 0; i < m; i++) {
        //     for (int j = 0; j < m; j++) {
        //         System.out.print(palin[i][j] + " ");
        //     }
        //     System.out.println("");
        // }
        
        for (int i = 0; i < m - 2; i++) {
            for (int j = i + 1; j < m - 1; j++) {
                if (palin[0][i] && palin[i + 1][j] && palin[j + 1][m - 1]) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
