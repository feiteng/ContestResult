class Solution {
    public boolean checkPartitioning(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n];
        for (int i = 0; i < n; i++) {
            dp[i][i] = true;
            if (i < n - 1 && s.charAt(i) == s.charAt(i+1)) {
                dp[i][i+1] = true;
            }
        
}
        for (int i = 3; i <= n; i++) {
            for (int j = 0; j + i - 1 < n; j++) {
                int end = j + i - 1;
                if (s.charAt(j) == s.charAt(end)) {
                    dp[j][end] = dp[j + 1][end - 1];
                }
            
}
        }

        for (int i = 0; i < n; i++) {
            for (int j = i + 2; j < n; j++) {
                if (dp[0][i] && dp[i+1][j-1] && dp[j][n-1]) {
                    return true;
                }
            }
        }
        return false;
    }
}
