class Solution {
    public boolean checkPartitioning(String s) {
        int m = s.length();
        boolean[][] ok = new boolean[m][m];
        for (int i = 0; i < m; ++i) {
            ok[i][i] = true;
            if (i < m - 1) {
                ok[i][i+1] = s.charAt(i) == s.charAt(i+1);
            }
        }
        
        for (int len = 2; len < m; ++len) {
            for (int st = 0; st + len < m; ++st) {
                ok[st][st+len] = ok[st+1][st+len-1] && s.charAt(st) == s.charAt(st + len);
            }
        }
        
        for (int i = 0; i < m - 1; ++i) {
            for (int j = i + 1; j < m - 1; ++j) {
                if (ok[0][i] && ok[i+1][j] && ok[j+1][m-1])
                    return true;
            }
        }
        
        return false;
    }
}
