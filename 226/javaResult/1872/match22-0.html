class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int N = candiesCount.length;
        long[] prefixSums = new long[N];
        for (int i = 0; i < N; i++) prefixSums[i] = (i > 0 ? prefixSums[i - 1] : 0) + candiesCount[i];
        
        boolean[] ans = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int[] q = queries[i];
            int t = q[0];
            long sum = prefixSums[t];
            long prevSum = t > 0 ? prefixSums[t - 1] : 0;
            
            int d = q[1];
            int cap = q[2];
            
            ans[i] = prevSum <= (long) d * cap + cap - 1 && sum >= d + 1;
        }
        
        return ans;
    }
}
