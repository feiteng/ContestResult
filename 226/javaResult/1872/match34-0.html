class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] prefix = new long[n];
        long sum = 0;
        for(int i=0; i<n; i++){
            sum += candiesCount[i];
            prefix[i] = sum;
        }
        
        int m = queries.length;
        boolean[] res = new boolean[m];
        for(int i=0; i<m; i++){
            int type = queries[i][0];
            int day = queries[i][1];
            int cap = queries[i][2];
            long last = type==0 ? 0 : prefix[type-1];
            if(prefix[type] >= day+1 && last < 1L * cap * (day + 1))
                res[i] = true;
            else
                res[i] = false;
        }
        
        return res;
    }
}
