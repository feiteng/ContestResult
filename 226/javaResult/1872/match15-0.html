class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] prefix = new long[n + 1];
        for(int i = 1; i <= n; i++){
            prefix[i] = candiesCount[i - 1] + prefix[i - 1];
        }
        boolean[] res = new boolean[queries.length];
        int index = 0;
        for(int[] query : queries){
            long day = query[1] + 1;
            int type = query[0];
            long cap = query[2];
            
            long maxNum = cap * day;
            long minNum = day;
            long lowerBound = prefix[type] + 1;
            long upperBound = prefix[type + 1];
            
            // System.out.println("type " + type + " day " + day + " cap " + cap +  " minNum " + minNum + " " +maxNum + " prev prefix " + prefix[type] + " pefix " + prefix[type + 1]);
            
            if(!(upperBound < minNum || lowerBound > maxNum)){
                res[index] = true;
            }
            
            index++;      
        }
        
        return res;
        
        
        
        
        
    }
}
