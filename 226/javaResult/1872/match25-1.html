class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] pref = new long[n];
        long sum = 0;
        for (int j = 0; j < n; j++) {
            sum += candiesCount[j];
            pref[j] = sum;
        }
        boolean[] ret = new boolean[queries.length];
        for (int j = 0; j < queries.length; j++) {
            
int type = queries[j][0];
            long day = queries[j][1];
            long cap = queries[j][2];
            if (type == 0) {
                ret[j] = pref[type] > day;
            } else {
                //System.out.printf("adj candies: %d, required days: %d\n", pref[type-1] + (cap-1)*(type), (pref[type-1] + (cap-1)*(type))/cap);
                ret[j] = (pref[type-1])/cap <= day && pref[type] > day;
                //ret[j] = (pref[type-1] + (cap-1)*(type))/cap <= day && pref[type] > day;
            }
        }
        return ret;
    }
}
