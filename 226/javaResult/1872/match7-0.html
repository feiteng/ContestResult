class Solution {
    /**
     * With dailyCap, the number of candies we can eat till day d is : [d:dailyCap * d]
     * Whether we can eat candy i on this day d depends on prefix sum.
     * If total number of candies from 0 to i is withint this range, then true.
     */
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] presum = new long[n];
        long sum = 0;
        for (int i = 0; i < n; i++) {
            sum += candiesCount[i];
            presum[i] = sum;
        }
        
        // System.out.println(presum[90]);
        // System.out.println(presum[91]);
        
        boolean[] res = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int candy = queries[i][0];
            int d = queries[i][1];
            int cap = queries[i][2];
            long minCandies = d + 1;
            long maxCandies = cap * (d + 1);
            
            if ((candy > 0 && cap <= (presum[candy - 1] / (d + 1))) || minCandies > presum[candy]) {
                res[i] = false;
            } else {
                res[i] = true;
            }
        }
        return res;
    }
}
