class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] preSum = new long[n+1];
        for (int i = 1; i <= n; i++) {
            preSum[i] = preSum[i-1] + candiesCount[i-1];
        }
        // System.out.println(Arrays.toString(preSum));
        int m = queries.length;
        boolean[] res = new boolean[m];
        for (int i = 0; i < m; i++) {
            int[] q = queries[i];
            int type = q[0];
            
int day = q[1];
            int cap = q[2];
            
            long totalToEat = preSum[type];
            if ((long)cap * (long)(day + 1) <= totalToEat) {
                res[i] = false; 
            } else  if ((long)day + 1 > preSum[type+1]) {
                res[i] = false; 
            } else {
                res[i] = true;    
            }
        }
        return res;
    }
}
