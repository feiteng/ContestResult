class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = queries.length;
        boolean[] res = new boolean[n];
        
        long[] preSum = new long [candiesCount.length];
        
        long sum = 0;
        for (int i = 0; i < candiesCount.length; i++) {
            sum += candiesCount[i];
            preSum[i] = sum;
        }
        // for (int i = 0; i <candiesCount.length; i++) {
        //     System.out.print(" " + preSum[i]);
        // }
        // System.out.println();
        //  System.out.println("index 90 " + preSum[90]);
        // System.out.println("index 91 " + preSum[91]);
        // System.out.println(candiesCount.length);
        
        for (int i = 0; i < n; i++) {
            long day = queries[i][1] + 1; // least number to eat
            int type = queries[i][0];
            long cap = queries[i][2] * day; // most number to eat
            
           // System.out.println(" min " + day + " cap " + cap);
            
            if (type > 0  && preSum[type] >= day && preSum[type - 1] < cap) {
                res[i] = true;
            }
            if (type == 0 && preSum[type] >= day) {
                res[i] = true;
            }
            
        }
        return res;
    }
}
