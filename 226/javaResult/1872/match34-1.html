class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] total = new long[n];
        long s = 0;
        for (int i = 0; i < n; i++) {
            s += candiesCount[i];
            total[i] = s;
        }
        System.out.println(Arrays.toString(total));
        
int m = queries.length;
        boolean[] ans = new boolean[m];
        for (int i = 0; i < m; i++) {
            int[] q = queries[i];
            int type = q[0], day = q[1], cap = q[2];
            if (total[type] < day + 1) continue;
            long eat = (long)(day+1)*cap;
            int ll = -1, rr = n;
            while (rr - ll > 1) {
                int mid = (rr + ll)/2;
                if (eat > total[mid]) ll = mid;
                else rr = mid;
            }
            // System.out.println(ll + " " + total[ll] + " " + eat + " " + total[ll+1]);
            
if (ll + 1 < type) ans[i] = false;
            else ans[i] = true;
        }
        return ans;
    }
}
