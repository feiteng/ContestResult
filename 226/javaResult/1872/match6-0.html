class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        long[] sum = new long[candiesCount.length];
        sum[0] = 0;
        for(int i = 1; i < candiesCount.length; i++) {
            sum[i] = sum[i - 1] + candiesCount[i - 1];
        }
        int q = 0;
        boolean[] answers = new boolean[queries.length];
        for(int i = 0; i < queries.length; i++) {
            int p = queries[i][0];
            long t = queries[i][1];
            long m = queries[i][2];
            long minnum = t;
            long maxnum = m * t;
            if(maxnum + m <= (long)sum[p] ||(minnum >= (long)(sum[p] + candiesCount[p]))) {
                answers[q++] = false;
            } else {
                answers[q++] = true;
            }
        }
        return answers;
    }
}
