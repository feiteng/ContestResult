class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        boolean[] ans = new boolean[queries.length];
        long[] sum = new long[candiesCount.length];
        sum[0] = candiesCount[0];
        
for (int i = 1; i < sum.length; i++) {
            sum[i] = candiesCount[i] + sum[i - 1];
        }
        
        for (int i = 0; i < queries.length; i++) {
            int d1 = Math.max(queries[i][0] - 1, 0);
            int d2 = Math.min(queries[i][1], sum.length - 1);
            long lbound = queries[i][0] > 0 ? sum[queries[i][0] - 1] : 0;
            
long ubound = sum[queries[i][0]];
            long cap = ((long)queries[i][1] + 1L) * (long)queries[i][2];
            if (cap > lbound && (long)queries[i][1] < ubound) {
                ans[i] = true;
            } else {
                ans[i] = false;
            }
        }
        
        return ans;
    }
}
