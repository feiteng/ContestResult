class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        long[] preCount = new long[candiesCount.length];
        for (int i = 1; i < candiesCount.length; i++){
            preCount[i] = preCount[i - 1] + (long)candiesCount[i - 1];
        }
        
        boolean[] result = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++){
            int type = queries[i][0], day = queries[i][1], cap = queries[i][2];
            result[i] = (long)(day + 1) <= preCount[type] + candiesCount[type] && (day + 1) * (long)cap > preCount[type];
        }
        
        return result;
    }
}
