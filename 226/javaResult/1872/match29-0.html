class Solution {
    public boolean[] canEat(int[] cand, int[][] queries) {
        int n = cand.length;
        long[] presum = new long[n + 1];
        presum[0] = 0;
        for (int i = 0; i < n; i++) {
            presum[i + 1] = presum[i] + cand[i];
        }
        int m = queries.length;
        boolean[] res = new boolean[m];
        int idx = 0;
        for (int i = 0; i < m; i++) {
            int[] query = queries[i];
            int type = query[0];
            long day = query[1];
            long cap = query[2];
            long min = (day + 1) * 1;
            long max = (day + 1) * cap;
            if (min > presum[type + 1] || max <= presum[type]) {
                res[idx] = false;
            } else {
                res[idx] = true;
            }
            idx ++;
        }
        return res;
    }
}
