class Solution {
    // [7,4,5, 3, 8]
    //  0 7 11 16 19 27
    
    // [0,2,2]
    // [0 2 4 6]
    // [1,3,]
    
    // 0 2 2
    //      
    // (day+1)*1=3
    // --- ----- --|**********|aaaaaaaaa
    //    |                     | 
    // (day+1)*2=6
    // min=3 max=6
    // 3,4
    // day0 
    
    // 4 2 4
    // 8
    
    // 2 13 10000000
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        
        long[] prefix = new long[n+1];
        for (int i = 1; i < n+1; i++) {
            // prefixSum[i] = prefixSum[i-1] + candiesCount[i];
            prefix[i] = prefix[i-1]+ candiesCount[i-1];
        }
        
        boolean[] ans = new boolean[queries.length];
        
        for (int i = 0; i < queries.length; i++) {
            int[] query = queries[i];
            int type = query[0];
            long day = query[1];
            long cap = query[2];
            
            long min = day+1;
            long max = (day+1) * cap;
            // max <= prefix[i]
            // min > prefix[i+1]
            if (max <= prefix[type] || min > prefix[type+1]) {
                ans[i] = false;
            } else {
                ans[i] = true;
            }
        }
        
        return ans;
    }
}
