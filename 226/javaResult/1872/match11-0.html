class Solution {
    public boolean[] canEat(int[] cc, int[][] q) {
        int n = q.length;
        boolean [] ans = new boolean[n];
        long sum[] = new long [cc.length + 1];
        
        
        
        for(int i = 0; i < cc.length; i++){
            sum[i + 1] = sum[i] + cc[i];
        }
        
        for(int i = 0; i< n; i++){
            long d = q[i][1];
            int t = q[i][0];
            long l = q[i][2];
            if(sum[t + 1] >= (d + 1) && sum[t] < l * (d + 1)){
                ans[i] = true;
            }else{
                ans[i] = false;
            }
        }
        
        return ans;
    }
}
