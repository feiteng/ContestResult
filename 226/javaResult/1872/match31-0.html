class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = queries.length;
        boolean[] result = new boolean[n];
        int index = 0;
        int m = candiesCount.length;
        long[] sums = new long[m];
        sums[0] = candiesCount[0];
        for(int i = 1; i < m; ++i) {
            sums[i] = sums[i-1] + candiesCount[i];
        }
        //System.out.println(candiesCount[]));
        //System.out.println(candiesCount[85]);
        for(int[] query : queries) {
            int type = query[0];
            long day = query[1];
            long cap = query[2];
            long minEat = 1 * day + 1;
            long maxEat = cap * day + cap;
            long minCandy = 1;
            long maxCandy = sums[type];
            if (type > 0) {
                minCandy = sums[type - 1] + 1;
            }
            if (minEat > maxCandy || minCandy > maxEat) {
                result[index++] = false;
            } else {
                result[index++] = true;
            }
        }
        return result;
    }
}
