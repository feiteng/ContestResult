class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] sum = new long[n + 1];
        sum[0] = 0;
        for(int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + (long)candiesCount[i - 1];
        }
        int m = queries.length;
        boolean[] ans = new boolean[m];
        for(int i = 0; i < m; i++) {
            int type = queries[i][0];
            int day = queries[i][1];
            int cap = queries[i][2];
            long h = (long)cap * (long)(day + 1);
            if(sum[type + 1] < day + 1 || sum[type] >= h) {
                ans[i] = false;
            } else {
                ans[i] = true;
            }
        }
        return ans;
    }
}
