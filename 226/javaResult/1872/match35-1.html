class Solution {
    public boolean[] canEat(int[] candies, int[][] queries) {
        
        int len = candies.length;
        long[] sum = new long[len+2];
        for (int i = 0; i < len; i++) {
            sum[i+1] = sum[i] + candies[i];
        }
        
        boolean[] ans = new boolean[queries.length];

        for (int i = 0; i < queries.length; i++) {
            long all = sum[queries[i][0]+1];
            if (all >= queries[i][1]+1 && sum[queries[i][0]] < queries[i][2]*(queries[i][1]+1L)) 
{
                ans[i] = true;
            }
        }
        
        return ans;
        

    }
}
