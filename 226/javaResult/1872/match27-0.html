class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        int m = queries.length;
        long[] sums = new long[n];
        sums[0] = candiesCount[0];
        for (int i = 1; i < n; i++) {
            sums[i] = sums[i-1]+candiesCount[i];
        }
        boolean[] ret = new boolean[m];
        for (int i = 0; i < m; i++) {
            int[] query = queries[i];
            int j = query[0];
            if(sums[j]<=query[1]||(j>0&&sums[j-1]>=(long)query[2]*(query[1]+1))){
                ret[i] = false;
            }
            else {
                ret[i] = true;
            }
        }
        return ret;
    }
}
