class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] preSum = new long[n];
        preSum[0] = candiesCount[0];
        for(int i = 1; i < n; i++){
            preSum[i] = preSum[i-1] + candiesCount[i];
        }
        
        int m = queries.length;
        boolean[] res = new boolean[m];
        for(int i = 0; i < m; i++){
            int type = queries[i][0];
            int day = queries[i][1] + 1;
            long limit = Math.min((long)queries[i][2], preSum[n-1]);
            long winMinCount = type > 0 ? preSum[type-1] + 1 : 1;
            long winMaxCount = preSum[type];
            
if(winMinCount <= limit * day && day * 1 <= winMaxCount){
                res[i] = true;
            }
            else{
                res[i] = false;
            }
        }
        return res;
    }
}
