class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        //before[i] -> candies to eat before this candie
        long[] before = new long[n+1];
        before[0] = 0;
        for (int i = 1; i < n+1; i++) {
            before[i] = before[i-1] + candiesCount[i-1];
        }
        
        boolean[] ans = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int[] qu = queries[i];
            int type = qu[0];
            
int day = qu[1];
            int limit = qu[2];
            
//             //eat the min, if min passes the day, then false
//             int min = day-1;
//             if (min > before[type+1]) {
//                 ans[i] = false;
//             }
            
//             //before day the max can eat:
//             long eaten = (long)limit * (long)(day-1);
//             if (eaten > before[type] && before[type] >= day-1) {
//                 ans[i] = true;
//             } else {
//                 ans[i] = false;
//             }
            
            //min to eat after day
            int min = day+1;
            
            //max to eat after day
            long max = (long)limit * (long)(day+1);
            
            long before0 = (long)before[type];
            long before1 = (long)before[type+1];
            
            //System.out.println(" min " + min + " max " + max + " before " + before[type+1]);
            if (before1 >= min && before0+1 <= max) {
                ans[i] = true;
            } else {
                ans[i] = false;
            }
            
        }
        return ans;
    }
}
