class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
    	int n = candiesCount.length;
    	long[] prefixSum = new long[n];
    	long sum = 0;
    	for(int i=0; i<n; i++) {
    		sum += candiesCount[i];
    		prefixSum[i] = sum;
    	}
    	
    	int q = queries.length;
    	boolean[] result = new boolean[q];
    	for(int j=0; j<q; j++) {
    		int[] quer
y = queries[j];
    		int type = query[0];
    		int day = query[1];
    		int cap = query[2];
    		if( (day+1) <= prefixSum[type] && (long)cap*(day+1) > prefixSum[type] - candiesCount[type] ) {
    			result[j] = true;
    		}else {
    			result[j] = false;
    		}
    		
    	}
    	return result;
    }
}
