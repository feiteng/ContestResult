class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int len = candiesCount.length;
        long[] preSums = new long[len + 1];
        for (int i = 0;i < len;i++) {
            preSums[i + 1] = preSums[i] + candiesCount[i];
        }
        int m = queries.length;
        boolean[] res = new boolean[m];
        for (int i = 0;i < m;i++) {
            int t = queries[i][0], d = queries[i][1], cap = queries[i][2];
            long cnt = preSums[t + 1];
            if (cnt > d && preSums[t] < (long)(d + 1) * cap) {
                res[i] = true;
            }
        }
        return res;
    }
}
