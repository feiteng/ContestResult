class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int lenC = candiesCount.length;
        int lenQ = queries.length;
        long[] preSum = new long[lenC + 1];
        
for (int i = 1; i <= lenC; ++i) {
            preSum[i] = preSum[i - 1] + candiesCount[i - 1];
        }
        
        boolean[] res = new boolean[lenQ];
        for (int i = 0; i < lenQ; ++i) {
            int[] query = queries[i];
            int type = query[0];
            long day = query[1];
            long cap = query[2];
            long low = preSum[type] + 1 - cap;
            System.out.println(low);
            long high = preSum[type + 1] - 1;
            System.out.println(high);
            long min = day;
            System.out.println(min);
            long max = cap * day;
            System.out.println(max);
            
if (max < low || min > high) {
                res[i] = false;
            } else {
                res[i] = true;
            }
        }
        return res;
    }
}
