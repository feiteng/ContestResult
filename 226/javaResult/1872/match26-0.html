class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = candiesCount.length;
        long[] sum = new long[n];
        for(int i=1;i<n;i++){
            sum[i] = sum[i-1] + candiesCount[i-1];
        }
        boolean[] ans = new boolean[queries.length];
        for(int i=0;i<queries.length;i++){
            int type = queries[i][0];
            int day = queries[i][1];
            int cap = queries[i][2];

            if(day+1>sum[type] && day+1<=sum[type]+candiesCount[type]){
                ans[i] = true;
            }else if(day+1<=sum[type]+candiesCount[type]
            && (long)(day+1)*cap>sum[type]){
                ans[i] = true;
            }else{
                ans[i] = false;
            }
        }
        return ans;
    }
}
