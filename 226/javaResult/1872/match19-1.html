class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        long[] addon = new long[candiesCount.length + 1];
        for (int i = 1; i < addon.length; i++) {
            addon[i] = addon[i - 1] + candiesCount[i - 1];
        }

        boolean[] ans = new boolean[queries.length];
        
        for (int i = 0; i < ans.length; i++) {
            
int[] q = queries[i];
            long sum = q[2] * (long)(q[1] + 1);
            ans[i] = (sum > addon[q[0]]) && (addon[q[0] + 1] > q[1]);
        }

        return ans;
    }
}
