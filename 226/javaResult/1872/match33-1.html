class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        int n = queries.length;
        boolean[] res = new boolean[n];
        long[] cnt = new long[candiesCount.length + 1];
        
for (int i = 0; i < candiesCount.length; i++) {
            cnt[i + 1] = cnt[i] + candiesCount[i];
        }
        for (int i = 0; i < n; i++) {
            long max = (long) queries[i][2] * (queries[i][1] + 1);
            long min = (long) queries[i][1] + 1;
            if (cnt[queries[i][0] + 1] >= min && max > cnt[queries[i][0]]) 
{
                res[i] = true;
            }
        }
        return res;
    }
}
