class Solution {
    public boolean[] canEat(int[] cand, int[][] queries) {
        int n = cand.length;
        long[] sum = new long[n];
        sum[0]=cand[0];
        for (int i =1; i < n; i++) {
            sum[i]=sum[i-1]+cand[i];
        }
        int len=queries.length;
        boolean[] res = new boolean[len];
        for (int i = 0; i < len; i++) {
            int[] q = queries[i];
            int a = q[0];
            long b = q[1] + 1;
            long c = q[2];
            long d=a==0?0:sum[a-1];
            if (b*c>d&&b<=sum[a]){
                res[i]=true;
            }else {
                res[i]=false;
            }
        }
        return res;
    }
}
