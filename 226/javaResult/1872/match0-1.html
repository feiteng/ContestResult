class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
                long[] presum = new long[candiesCount.length + 1];
        for (int i = 1; i < presum.length; i++) {
            presum[i] = presum[i - 1] + candiesCount[i - 1];
        }
        boolean[] ans = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int favoriteType = queries[i][0];
            long min = queries[i][1] + 1L;
            long max = ((long) queries[i][2]) * (queries[i][1] + 1L);

            long min1 = presum[favoriteType + 1] - candiesCount[favoriteType] + 1;

            long max1 = presum[favoriteType + 1];
            if ((max >= min1 && max <= max1) || (min >= min1 && min <= max1)) {
                ans[i] = true;
            } else if ((max1 >= min && max1 <= max) || (min1 >= min && min1 <= max)){
                ans[i] = true;
            } else {
                ans[i] = false;
            }
        }
        return ans;
    }
}
