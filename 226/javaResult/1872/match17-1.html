class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        // int[] sums = new int[candiesCount.length];
        // sums[0] = candiesCount[0];
        // for (int j=1;j<candiesCount.length;j++) {
        //     sums[j] = sums[j-1] + candiesCount[j];
        // }
        long[] sums = new long[candiesCount.length+1];
        for (int j=0;j<candiesCount.length;j++) {
            sums[j+1] = sums[j] + candiesCount[j];
        }
        boolean[] res = new boolean[queries.length];
        for (int j=0;j<queries.length;j++) {
            long min = queries[j][1];
            long max = ((long) (queries[j][1] + 1)) * ((long) queries[j][2]);
            
if (max > sums[queries[j][0]] && min <sums[queries[j][0]+1]) {res[j] = true;}
            else {res[j] = false;}
        }
        return res;
    }
}
