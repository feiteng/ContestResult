class Solution {
    public boolean[] canEat(int[] candiesCount, int[][] queries) {
        boolean[] ans=new boolean[queries.length];
        long[] sum=new long[candiesCount.length+1];
        for (int i = 1; i <sum.length ; i++) {
            sum[i]=sum[i-1]+candiesCount[i-1];
        }
        for (int i = 0; i < queries.length; i++) {
            long type=queries[i][0];
            long day=queries[i][1];
            long max=queries[i][2];
            long total=sum[(int)type];
//            System.out.println(total+" "+(max*(day+1))+" "+(sum[type+1])+" "+day);
            if (total/(day+1)<max&&sum[(int)type+1]>day){
                ans[i]=true;
            }else{
                ans[i]=false;
            }

        }
        return ans;
    }
}
