class Solution {
    public int countBalls(int low, int high) {
        int[] hash = new int[46];
        for(int i = low ; i <= high ; ++i) {
            ++hash[sum(i)];
        }
        int max = 0;
        for(int e : hash) max = Math.max(max, e);
        return max;
    }
    
    int sum(int num) {
        int sum = 0;
        while(num > 0) {
            sum += num % 10;
            num /= 10;
        }
        return sum;
    }
}
