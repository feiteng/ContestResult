class Solution {
    public int countBalls(int lowLimit, int highLimit) {

        int[] cnt = new int[highLimit + 1];
        for (int i = lowLimit; i <= highLimit; i++) {
            int sum = 
getSumDigits(i);
            cnt[sum]++;
        }
        int max = 0;
        for (int c : cnt) {
            max = Math.max(max, c);
        }
        return max;
    }

    private int getSumDigits(int n) {

        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
