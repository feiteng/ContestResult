class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int ans = 0;
        int[] nums = new int[100];
        for (int i = lowLimit; i <= highLimit; i++) {
            int sum = work(i);
            nums[sum]++;
            ans = Math.max(ans, nums[sum]);
        }
        return ans;
    }

    private int work(int x) {
        int res = 0;
        while (x > 0) {
            res += x % 10;
            x /= 10;
        }
        return res;
    }
}
