class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        
        HashMap<Integer,Integer> hm = new HashMap();
        int max = 0;
        for(int i=lowLimit;i<=highLimit;i++){
            int d = getSum(i);
            hm.put(d,hm.getOrDefault(d,0)+1);
            max =  Math.max(max,hm.get(d));
        }
        
        return max;
    }
    
    public int getSum(int num){
        
        int sum = 0;
        
        while(num>0){
            sum += (num%10);
            num /=10;
        }
        
        return sum;
    }
}
