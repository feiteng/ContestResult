class Solution {
  public int countBalls(int lowLimit, int highLimit) {
    Map<Integer, Integer> map = new HashMap<>();
    int max = 0;
    for (int i = lowLimit; i <= highLimit; i++) {
      int N = getN(i);
      map.put(N, map.getOrDefault(N, 0) + 1);
      max = Math.max(max, map.get(N));
    }
    return max;
  }
  
  private int getN(int n) {
    int x = 0;
    while (n > 0) {
      x += (n % 10);
      n /= 10;
    }
    return x;
  }
}
