class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int max = 0;
        int[] arr = new int[highLimit + 1];
        for (int i = lowLimit; i <= highLimit; i++) {
            arr[sum(i)]++;
            
max = Math.max(max, arr[sum(i)]);
        }
        return max;
    }
    
    public int sum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
