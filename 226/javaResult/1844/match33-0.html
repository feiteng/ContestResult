class Solution {
public int countBalls(int lowLimit, int highLimit) {
        int[] boxes = new int[50];
        for (int i = lowLimit; i <= highLimit; i++) {
            boxes[sumAll(i)]++;
        }
        int max = 0;
        for (int i = 0; i < boxes.length; i++) {
            max = Math.max(boxes[i], max);
        }
        return max;
    }
    private int sumAll(int a) {
        int temp = 0;
        while (a != 0) {
            temp += a % 10;
            a /= 10;
        }
        return temp;
    }

}
