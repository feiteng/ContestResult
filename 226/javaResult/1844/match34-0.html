class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] box = new int[101];
        
        for(int i=lowLimit; i<=highLimit; i++){
            int sum = 0;
            int n=i;
            while(n>0){
                sum = sum + n %10;
                n=n/10;
            }
            box[sum]++;
        }
        int ans = 0;
        for(int i=1;i<=100;i++) ans = Math.max(ans, box[i]);
        return ans;
    }
}
