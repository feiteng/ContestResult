class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] arr = new int[51];
        
        for(int i = lowLimit;i<=highLimit;i++){
            //System.out.println(i);
            int j = i;
            int cur = 0;
            while(j>0){
                cur+=(j%10);
                j=j/10;
            
}
            arr[cur]++;
        }
        
        int ans = 0;
        
        
        for(int i = 1;i<=50;i++){
            if(arr[i]>ans){
                ans = arr[i];
            }
        }
        return ans;
        
    }
}
