class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] cnt = new int[46];
        
        for(int n=lowLimit; n<=highLimit;  n++){
            int x = n, idx=0;
            while(x!=0) {
                idx+=(x%10);
                x/=10;
            }
            cnt[idx]++;
        }
        
        int ans = 0;
        for(int c : cnt) ans = Math.max(ans, c);
        return ans;
    }
}
