class Solution {
    public int countBalls(int lowLimit, int highLimit)
	{
		int[] count = new int[1000];
		int result = 0;
		for(int x = lowLimit; x <= highLimit; x++)
		{
			int current = 0;
			int num = x;
			while(num > 0)
			{
				current += num % 10;
				num /= 10;
			}
			
			count[current]++;
			
			result = Math.max(result, count[current]);
		}
		
		return result;
	}
}
