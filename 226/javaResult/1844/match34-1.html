class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] f = new int[100];
        for (int i = lowLimit; i <= highLimit; i++) {
            int n = i;
            int s = 0;
            while (n > 0) {
                s += (n % 10);
                n /= 10;
            
}
            f[s]++;
        }
        int ans = 0;
        for (int i = 0; i < 100; i++)
            ans = Math.max(ans, f[i]);
        return ans;
    }
}
