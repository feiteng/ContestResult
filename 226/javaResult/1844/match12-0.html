class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int cnt [] = new int[100001];
        int ans = 0;
        for (int i=lowLimit; i<=highLimit; i++) {
        	int j = get(i);
        	cnt[j]++;ans = Math.max(ans, cnt[j]);
        }
		return ans;
    }

	private int get(int i) {
		int ret = 0;
		while (i>0) {ret+=(i%10);i/=10;}
		return ret;
	}
}
