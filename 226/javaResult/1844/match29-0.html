class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int ans = 0;
        int[] nums = new int[9 * 6 + 5];
        for (int i = lowLimit; i <= highLimit; i++) {
            int sum = sum(i);
            // System.out.println(sum);
            nums[sum]++;
            ans = Math.max(ans, nums[sum]);
        }
        return ans;
    }
    
    private int sum(int x) {
        int res = 0;
        while (x != 0) {
            res += x % 10;
            x /= 10;
        }
        return res;
    }
}
