class Solution {
    public int countBalls(int lowLimit, int highLimit) {
    
    	int[] counts = new int[46];
    	int max = 0;
    	for(int i=lowLimit; i<=highLimit; i++) {
    		int digitSum = getBox(i);
    		counts[digitSum]++;
    		max = Math.max(max, counts[digitSum]);
    	}
    	return max;
    }
    
    
    private int getBox(int num) {
    	int digitSum = 0;
    	while( num != 0 ) {
    		digitSum += num % 10;
    		num/=10;
    	}
    	return digitSum;
    }
}
