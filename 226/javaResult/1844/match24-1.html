class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        Map<Integer, Integer> map = new HashMap<>();
        int max = 0;
        for (int i = lowLimit; i <= highLimit; i++) {
            int val = sum(i);
            map.put(val, map.getOrDefault(val, 0) + 1);
            max = Math.max(max, map.get(val));
        }
        return max;
    }
    
    public int sum(int n) {
        int sol = 0;
        while (n != 0) {
            sol += n % 10;
            n /= 10;
        }
        return sol;
    }
}
