class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] count = new int[50];
        int res = 0;
        for (int i = lowLimit; i <= highLimit; ++i) {
            int n = getNumber(i);
            count[n] ++;
            res = Math.max(res, count[n]);
        }
        
        return res;
    }
    
    public int getNumber(int i) {
        int res = 0;
        while (i > 0) {
            res += (i % 10);
            i /= 10;
        }
        return res;
    }
}
