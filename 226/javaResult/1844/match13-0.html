class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        int[] box = new int[100000];
        for(int i = lowLimit; i <= highLimit; i++){
            int sum = sum(i);
            box[sum]++;
        }
        int res = 0;
        for(int i = 0; i < 100000; i++) res = Math.max(res, box[i]);
        return res;
    }
    public int sum(int x){
        int res = 0;
        while(x > 0){
            res += x % 10;
            x /= 10;
        }
        return res;
    }
}
