import java.util.*;

class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int[] ans = new int[adjacentPairs.length + 1];
        Map<Integer, List<Integer>> next = new HashMap<>();
        for (int[] adjacentPair : adjacentPairs) {
            next.computeIfAbsent(adjacentPair[0], e -> new LinkedList<>()).add(adjacentPair[1]);
            next.computeIfAbsent(adjacentPair[1], e -> new LinkedList<>()).add(adjacentPair[0]);
        }

        Set<Integer> exists = new HashSet<>();
        for (Map.Entry<Integer, List<Integer>> entry : next.entrySet()) {
            if (entry.getValue().size() == 1) {
                ans[0] = entry.getKey();
                exists.add(ans[0]);
                break;
            }
        }
        for (int i = 1; i < ans.length; i++) {
            List<Integer> en = next.get(ans[i-1]);
            for (Integer integer : en) {
                if (!exists.contains(integer)) {
                    ans[i] = integer;
                    exists.add(integer);
                    break;
                }
            }
        }
        return ans;
    }
}
