class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, List<Integer>> G = new HashMap<>();
        for(int[] pair : adjacentPairs) {
            int u = pair[0];
            int v = pair[1];
            
G.putIfAbsent(u, new ArrayList<>());
            G.putIfAbsent(v, new ArrayList<>());
            G.get(u).add(v);
            G.get(v).add(u);
        }

        int idx = 0;
        int[] ans = new int[G.size()];

        
for(int key : G.keySet()) {
            if(G.get(key).size() == 1) {
                ans[idx++] = key;
                break;
            }
        }

        while(idx < ans.length) {
            List<Integer> next = G.get(ans[idx - 1]);
            int pre = idx == 1 ? Integer.MIN_VALUE : ans[idx - 2];
            ans[idx] = next.get(0) == pre ? next.get(1) : next.get(0);
            idx++;
        }
        
        return ans;
    }

}
