class Solution {
    public int[] restoreArray(int[][] pairs) {
        Map<Integer, Set<Integer>> hm = new HashMap<>();
        for(int[] a: pairs) {
            if(!hm.containsKey(a[0])) hm.put(a[0], new HashSet<>());
            if(!hm.containsKey(a[1])) hm.put(a[1], new HashSet<>());
            hm.get(a[1]).add(a[0]);
            hm.get(a[0]).add(a[1]);
        }
        int out = -1;
        for(int key: hm.keySet()) {
            if(hm.get(key).size() == 1) {
                out = key;
                break;
            }
        }
        int[] res = new int[pairs.length + 1];
        res[0] = out;
        int index = 1;
        while(index < res.length) {
            int val = hm.get(out).iterator().next();
            hm.remove(out);
            hm.get(val).remove(out);
            res[index++] = val;
            out = val;
        }
        return res;
    }
}
