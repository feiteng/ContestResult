class Solution {
       public int[] restoreArray(int[][] adjacentPairs) {

        Map<Integer, List<Integer>> m1 = new HashMap<>();


        Map<Integer, Integer> map = new HashMap<>();

        for (int[] adjacentPair : adjacentPairs) {
            map.merge(adjacentPair[0], 1, (v1, v2) -> v1 + v2);
            map.merge(adjacentPair[1], 1, (v1, v2) -> v1 + v2);
            m1.computeIfAbsent(adjacentPair[0], k -> new ArrayList<>()).add(adjacentPair[1]);
            m1.computeIfAbsent(adjacentPair[1], k -> new ArrayList<>()).add(adjacentPair[0]);
        }
        int key = -1;
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if (entry.getValue() == 1) {
                key = entry.getKey();
                break;
            }
        }

        int[] res = new int[adjacentPairs.length + 1];
        Set<Integer> set = new HashSet<>();
        set.add(key);
        res[0] = key;
        for (int i = 0; i < adjacentPairs.length; i++) {
            List<Integer> integers = m1.get(res[i]);
            for (Integer integer : integers) {
                if (integer != null && !set.contains(integer)) {
                    res[i + 1] = integer;
                    set.add(integer);
                    break;
                }
            }
        }

        return res;
    }
}
