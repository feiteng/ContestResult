class Solution {
    /**
     * Each number should appear once except for head and tail
     */
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, List<Integer>> graph = new HashMap<>();
        for (int[] pair : adjacentPairs) {
            graph.computeIfAbsent(pair[0], k -> new ArrayList<>()).add(pair[1]);
            graph.computeIfAbsent(pair[1], k -> new ArrayList<>()).add(pair[0]);
        }
        
        int head = Integer.MIN_VALUE;
        int tail = Integer.MIN_VALUE;
        for (int k : graph.keySet()) {
            if (graph.get(k).size() == 1) {
                if (head == Integer.MIN_VALUE) {
                    head = k;
                } else {
                    tail = k;
                    break;
                }
            }
        }
        
        int[] res = new int[adjacentPairs.length + 1];
        res[0] = head;
        res[1] = graph.get(head).get(0);
        res[res.length - 1] = tail;
        res[res.length - 2] = graph.get(tail).get(0);
        int idx = 2;
        while (idx < res.length - 2) {
            List<Integer> neighbors = graph.get(res[idx - 1]);
            for (int neighbor : neighbors) {
                if (neighbor != res[idx - 2]) {
                    res[idx++] = neighbor;
                    break;
                }
            }
        }
        
        return res;
    }
}
