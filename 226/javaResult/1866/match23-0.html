class Solution {
  public int[] restoreArray(int[][] adj) {
    Map<Integer, Set<Integer>> map = new HashMap<>();
    for (int[] edge : adj) {
      add(map, edge[0], edge[1]);
      add(map, edge[1], edge[0]);
    }
    
    int first = 0;
    for (Map.Entry<Integer, Set<Integer>> entry : map.entrySet()) {
      if (entry.getValue().size() == 1) {
        first = entry.getKey();
        break;
      }      
    }    
    
    int[] res = new int[adj.length + 1];
    int i = 0;
    res[0] = first;
    while (i + 1 < res.length) {
      res[i + 1] = map.get(res[i]).iterator().next();
      map.get(res[i + 1]).remove(res[i]);
      i++;
    }
    return res;
  }
  
  private void add(Map<Integer, Set<Integer>> map, int a, int b) {
    Set<Integer> set = map.get(a);
    if (set == null) {
      set = new HashSet<>();
      map.put(a, set);
    }
    set.add(b);
  }
}
