class Solution {
    public int[] restoreArray(int[][] pairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int[] p : pairs) {
            map.computeIfAbsent(p[0], k -> new ArrayList<>()).add(p[1]);
            map.computeIfAbsent(p[1], k -> new ArrayList<>()).add(p[0]);
        }
        
        int first = -1;
        for (Map.Entry<Integer, List<Integer>> e : map.entrySet()) {
            if (e.getValue().size() == 1) {
                first = e.getKey();
                break;
            }
        }
        
        int[] res = new int[pairs.length + 1];
        Set<Integer> visited = new HashSet<>();
        visited.add(first);
        res[0] = first;
        
        for (int i = 1, prev = first; i <res.length; i++) {
            List<Integer> l = map.get(prev);
            for (int next : l) {
                if (visited.add(next)) {
                    prev = res[i] = next;
                }
            }
        }
        
        return res;
    }
}
