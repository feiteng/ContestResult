class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int[] adj : adjacentPairs){
            map.putIfAbsent(adj[0], new ArrayList<>());
            map.putIfAbsent(adj[1], new ArrayList<>());
            map.get(adj[0]).add(adj[1]);
            map.get(adj[1]).add(adj[0]);
        }
        int start = -1;
        int prev = -1;
        for(Integer key : map.keySet()){
            if(map.get(key).size() == 1){
                start = key;
                break;
            }
        }
        
        System.out.println(map);
        
        Set<Integer> visited = new HashSet<>();
        
        int[] resArr = new int[adjacentPairs.length + 1];
        for(int i = 0; i < resArr.length; i++){
            resArr[i] = start;
            visited.add(start);
            List<Integer> neighbor = map.get(start);
            if(neighbor.size() == 1){
                start = neighbor.get(0);
            }else{
                start = visited.contains(neighbor.get(0)) ? neighbor.get(1) : neighbor.get(0);
            }
           
        
        }
        
        return resArr;
        
        
    }
}
