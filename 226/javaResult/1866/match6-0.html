class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int[] i: adjacentPairs) {
            List<Integer> l1 = map.getOrDefault(i[0], new LinkedList<>());
            List<Integer> l2 = map.getOrDefault(i[1], new LinkedList<>());
            l1.add(i[1]);
            l2.add(i[0]);
            map.put(i[0], l1);
            map.put(i[1], l2);
        }
        int n = adjacentPairs.length;
        int[] ans = new int[n + 1];
        for(int i: map.keySet()) {
            if(map.get(i).size() == 1) {
                ans[0] = i;
                break;
            }
        }
        Set<Integer> set = new HashSet<>();
        set.add(ans[0]);
        for(int i = 1; i < n + 1; i++) {
            int a = map.get(ans[i - 1]).get(0);
            if(!set.contains(a)) {
                ans[i] = a;
                set.add(a);
            } else {
                int b = map.get(ans[i - 1]).get(1);
                ans[i] = b;
                set.add(b);
            }
        }
        return ans;
    }
}
