class Solution {
    
public int[] restoreArray(int[][] ad) {
        for(int i = 0; i < ad.length; i++){
            ad[i][0] += 100000;
            ad[i][1] += 100000;
        }
        int edge[][] = new int[200001][2];
        int cnt [] = new int[200001];
        for(int [] a : ad){
            
            edge[a[0]][cnt[a[0]]] = a[1];
            edge[a[1]][cnt[a[1]]] = a[0];
            cnt[a[0]] ++;
            cnt[a[1]] ++;
        
}
        int start = 0;
        for(int i = 0; i < 200001; i++){
            if(cnt[i] == 1){
                start = i;
                break;
            }
        }
        int n = ad.length + 1;
        int ans [] = new int[n];
        ans[0] = start;
        Arrays.fill(cnt, 0);
        cnt[start] = 1;
        for(int i = 1; i < n; i++){
            // System.out.println(cnt[ans[i-1]] - 1);
            
if(cnt[edge[ans[i-1]][0]] == 1){
                ans[i] = edge[ans[i-1]][1];
            }else{
                ans[i] = edge[ans[i-1]][0];
            }
            cnt[ans[i]] = 1;
            // System.out.println(ans[i]);
        }
        for(int i = 0; i < n; i++){
            ans[i] -= 100000;
        }
        return ans;
    }
}
