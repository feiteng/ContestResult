class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        HashMap<Integer,List<Integer>> map = new HashMap<>();
        for(int[] ad:adjacentPairs ){

            List<Integer> list = map.getOrDefault(ad[0],new ArrayList<Integer>());
            list.add(ad[1]);
            map.put(ad[0],list);

            list = map.getOrDefault(ad[1],new ArrayList<Integer>());
            list.add(ad[0]);
            map.put(ad[1],list);
        }
        int n = adjacentPairs.length+1;
        int[] ans = new int[n];
        Set<Integer> set = new HashSet<>();
        int start = 0;
        for(int key:map.keySet()){
            if(map.get(key).size()==1){
                start = key;
                break;
            }
        }
        ans[0] = start;
        set.add(start);
        for(int i=1;i<n;i++){
            int key = ans[i-1];
            for(int m: map.get(key)){
                if(!set.contains(m)){
                    ans[i] = m;
                    set.add(m);
                    break;
                }
            }
        }
        return ans;
    }
}
