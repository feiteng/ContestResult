class Solution {
    public int[] restoreArray(int[][] nums) {
        
HashMap<Integer,HashSet<Integer>> map = new HashMap<>();
        HashSet<Integer> set = new HashSet<>();
        for(int[] pair : nums){
            map.putIfAbsent(pair[0],new HashSet<>());
            map.putIfAbsent(pair[1],new HashSet<>());
            map.get(pair[0]).add(pair[1]);
            map.get(pair[1]).add(pair[0]);
            if(set.contains(pair[0])){
                set.remove(pair[0]);
            }
            else{
                set.add(pair[0]);
            }
            if(set.contains(pair[1])){
                set.remove(pair[1]);
            }
            else{
                set.add(pair[1]);
            }
        }
        int start = 0;
        for(int ele : set){
            start = ele;
            break;
        }
        int[] res = new int[nums.length+1];
        res[0] = start;
        HashSet<Integer> visited = new HashSet<>();
        visited.add(start);
        loop: for(int i=1;i<res.length;i++){
            for(int ele : map.get(start)){
                if(!visited.contains(ele)){
                    start = ele;
                    res[i]=start;
                    visited.add(start);
                    continue loop;
                
}
            }
        }
        return res;
    }
}
