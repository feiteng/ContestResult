class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int[] p : adjacentPairs) {
            
map.putIfAbsent(p[0], new ArrayList<>());
            map.putIfAbsent(p[1], new ArrayList<>());
            map.get(p[0]).add(p[1]);
            map.get(p[1]).add(p[0]);
        }
        int n = adjacentPairs.length + 1;
        
int[] res = new int[n];
        for (int key : map.keySet()) {
            if (map.get(key).size() == 1) {
                res[0] = key;
                break;
            }
        }
        int preTwo = -100001;
        for (int i = 1;i < n;i++) {
            int pre = res[i - 1];
            for (int cur : map.get(pre)) {
                if (cur != preTwo) {
                    res[i] = cur;
                    
preTwo = pre;
                    break;
                } 
            }
        }
        return res;
    }
}
