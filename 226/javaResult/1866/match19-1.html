class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int n = adjacentPairs.length + 1;
        HashMap<Integer, HashSet<Integer>> map = new HashMap<>();
        for(int[] x : adjacentPairs) {
            int a = x[0];
            int b = x[1];
            map.putIfAbsent(a, new HashSet<>());
            map.putIfAbsent(b, new HashSet<>());
            map.get(a).add(b);
            
map.get(b).add(a);
            
        }
        int head = -1;
        for(int x : map.keySet()) {
            if(map.get(x).size() == 1) {
                head = x;
            
}
        }
        int[] res = new int[n];
        res[0] = head;
        for(int i = 1; i < n; ++i) {
            int next = map.get(head).iterator().next();
            map.remove(head);
            map.get(next).remove(head);
            res[i] = next;
            head = next;
        }
        return res;
    }
}
