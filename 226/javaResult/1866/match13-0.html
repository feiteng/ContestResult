class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int n = adjacentPairs.length + 1;
        int[] res = new int[n];
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] adj : adjacentPairs) {
            map.putIfAbsent(adj[0], new HashSet<>());
            map.get(adj[0]).add(adj[1]);
            map.putIfAbsent(adj[1], new HashSet<>());
            map.get(adj[1]).add(adj[0]);
        }
        int idx = -1;
        for (int k : map.keySet()) {
            if (map.get(k).size() == 1) {
                idx = k;
                break;
            }
        }
        int i = 0;
        while (i < n) {
            res[i++] = idx;
            for (int next : map.get(idx)) {
                if (i < 2 || next != res[i - 2]) {
                    idx = next;
                    break;
                }
            }
        }
        return res;
    }
}
