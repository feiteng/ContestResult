import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

class Solution {
	public static void main(String[] args) {
		System.out.println(Arrays.toString(new Solution().restoreArray(new int[][]{{2,1},{3,4},{3,2}})));
	}
    public int[] restoreArray(int[][] adjacentPairs) {
    	for (int i=0; i<adjacentPairs.length; i++) {
    		adjacentPairs[i][0] += 1000_00;
    		adjacentPairs[i][1] += 1000_00;
    	}
    	int[] count = new int[2000_01];
    	int[][] map = new int[2000_01][2];
    	for (int[] arr : adjacentPairs) {
    		count[arr[0]] ++;
    		count[arr[1]] ++;
    		if (count[arr[0]] == 1) {
    			map[arr[0]][0] = arr[1];
    		} else {
    			map[arr[0]][1] = arr[1];
    		}
    		if (count[arr[1]] == 1) {
    			map[arr[1]][0] = arr[0];
    		} else {
    			map[arr[1]][1] = arr[0];
    		}
    	}
    	int a = 0;
    	for (int i=0; i<count.length; i++) {
    		if (count[i] == 1) {
    			a = i;
    			break;
    		}
    	}
    	List<Integer> list = new ArrayList<>();
    	list.add(a);
    	int pre = -1;
    	while (count[a] > 0) {
    		count[a] --;
    		int b;
    		if (pre != -1 && pre == map[a][0]) {
    			b = map[a][1];
    		} else {
    			b = map[a][0];
    		}
    		list.add(b);
    		pre = a;
    		a = b;
    		count[a] --;
    	}
    	int[] ans = new int[list.size()];
    	for (int i=0; i<list.size(); i++) {
    		ans[i] = list.get(i) - 1000_00;
    	}
    	return ans;
    }
}
