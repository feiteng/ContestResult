import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Map.Entry;


class Solution {
	public int[] restoreArray(int[][] adjacentPairs) {
		Map<Integer, List<Integer>> m = new HashMap<Integer, List<Integer>>();
		for (int e[] : adjacentPairs) {
			add(m, e[0], e[1]);

			add(m, e[1], e[0]);
		}
		Integer h = 0;
		for (Entry<Integer, List<Integer>> e : m.entrySet()) {
			if (e.getValue().size() == 1) {
				h = e.getKey();
				break;

			}
		}
		int[] ret = new int[adjacentPairs.length + 1];
		ret[0] = h;
		int i = 0;
		tr(h, m);
		while (m.size() > 1) {
			tr(m);
			List<Integer> l = m.remove(h);
			Integer v = l.get(0);
			List<Integer> l2 = m.get(v);

			l2.remove(h);

			ret[++i] = v;
			h = v;
		}
		return ret;
	}

	private void add(Map<Integer, List<Integer>> m, int k, int v) {
		List<Integer> l = m.getOrDefault(k, new ArrayList<Integer>());
		l.add(v);
		m.put(k, l);
	}

	//	System.out.println(Arrays.deepToString(o));
	}
}
