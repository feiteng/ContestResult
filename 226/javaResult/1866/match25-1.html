class Solution {
    private Map<Integer , Set<Integer>> graph = new HashMap<>();
    private Set<Integer> vis = new HashSet<>();

    
public int[] restoreArray(int[][] adjacentPairs) {
        Set<Integer> valueSet = new HashSet<>();
        for (int[] edge : adjacentPairs) {
            int u = edge[0] , v = edge[1];
            graph.putIfAbsent(u , new HashSet<>());
            graph.putIfAbsent(v , new HashSet<>());
            valueSet.add(u);
            valueSet.add(v);
            graph.get(u).add(v);
            graph.get(v).add(u);
        }
        int start = - 1;
        for (int vertex : valueSet) 
{
            if (graph.get(vertex).size() == 1) {
                start = vertex;
                break;
            }
        }
        int[] ans = new int[adjacentPairs.length + 1];
        for (int i = 0;i < adjacentPairs.length + 1;i ++) {
            ans[i] = start;
            vis.add(start);
            for (int next : graph.get(start)) 
{
                if (!vis.contains(next)) {
                    start = next;
                    break;
                }
            }
        }
        return ans;
    }
}
