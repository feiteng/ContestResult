class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int len = adjacentPairs.length + 1;
        int[] res = new int[len];
        Map<Integer, List<Integer>> map = new HashMap<>();
        Set<Integer> set = new HashSet<>();
        for(int[] x: adjacentPairs){
            int a = x[0], b = x[1];
            if(!map.containsKey(a)) map.put(a, new ArrayList<Integer>());
            if(!map.containsKey(b)) map.put(b, new ArrayList<Integer>());
            map.get(a).add(b);
            map.get(b).add(a);
            set.add(a);
            set.add(b);
        }
        int first = -1;
        for(int x: map.keySet()){
            if(map.get(x).size() == 1) {
                first = x;
                set.remove(x);
                break;
            }
        }
        for(int i = 0; i < len; i++){
            res[i] = first;
            for(int x: map.get(first)){
                if(set.contains(x)){
                    set.remove(x);
                    first = x;
                    break;
                }
            }
        }
        return res;
    }
}
