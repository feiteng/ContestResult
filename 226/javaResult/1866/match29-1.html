import java.util.*;

class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int n = adjacentPairs.length;
        int[] res = new int[n + 1];
        Map<Integer, List<Integer>> head = new HashMap<>();
        int start = (int) 1e9;
        int end = (int) 1e9;
        for (int[] ad : adjacentPairs) {
            int a = ad[0];
            int b = ad[1];
            
head.computeIfAbsent(a, l -> new ArrayList<>()).add(b);
            head.computeIfAbsent(b, l -> new ArrayList<>()).add(a);
        }
        
for (int key : head.keySet()) {
            if (head.get(key).size() == 1) {
                if (start == (int) 1e9) {
                    start = key;
                } else {
                    end = key;
                }
            }
        }
        res[0] = start;
        int cur = start;
        int id = 1;
        
int pre = (int) 1e9;
        while (cur != end) {
            List<Integer> next = head.get(cur);
            for (int i : next) {
                if (i != pre) {
                    res[id] = i;
                    pre = cur;
                    cur = i;
                    break;
                } else {
                    continue;
                
}
            }
            id++;
        }
        return res;
    }

}
