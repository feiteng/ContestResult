class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        if (adjacentPairs == null || adjacentPairs.length == 0) {
            return new int[0];
        }
        Map<Integer, List<Integer>> m = new HashMap<Integer, List<Integer>>();
        Set<Integer> s = new HashSet<Integer>();
        for (int[] pair : adjacentPairs) {
            m.putIfAbsent(pair[0], new ArrayList<Integer>());
            m.putIfAbsent(pair[1], new ArrayList<Integer>());
            m.get(pair[0]).add(pair[1]);
            m.get(pair[1]).add(pair[0]);
            if (s.contains(pair[0])) {
                s.remove(pair[0]);
            } else {
                s.add(pair[0]);
            }
            if (s.contains(pair[1])) {
                s.remove(pair[1]);
            } else {
                s.add(pair[1]);
            }
        }
        
        int val = s.iterator().next();
        Set<Integer> seen = new HashSet<Integer>();
        int[] res = new int[adjacentPairs.length + 1];
        for (int i = 0; i < res.length; i++) {
            res[i] = val;
            seen.add(val);
            for (int next : m.get(val)) {
                if (!seen.contains(next)) {
                    val = next;
                    break;
                }
            }
        }
        return res;
    }
}
