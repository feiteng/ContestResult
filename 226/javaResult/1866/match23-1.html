class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        
        for (int[] p : adjacentPairs) {
            addAdj(map, p[0], p[1]);
            addAdj(map, p[1], p[0]);
        }
        
        int[] res = new int[adjacentPairs.length+1];
        
        int start = 0;
        
for (Integer key : map.keySet()) {
            if (map.get(key).size() == 1) {
                start = key;
                break;
            }
        }
        
        res[0] = start;
        
        for (int i = 1; i < res.length; i++) {
            int prev = res[i-1];
            
            Set<Integer> set = map.get(prev);
            
            for (Integer j : set) {
                if (i >= 2) {
                    if (j != res[i-2]) {
                        res[i] = j;
                        break;
                    }
                } else {
                    res[i] = j;
                }
            }
        }
        
return res;
    }
    
    public void addAdj(Map<Integer, Set<Integer>> map, int a, int b) {
        Set<Integer> set = map.get(a);
        
        if (set == null) {
            set = new HashSet<>();
            map.put(a, set);
        }
        
        set.add(b);
    }
}
