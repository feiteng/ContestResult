class Solution {
    // [[2,1],[3,4],[3,2]]
    //  1 2
    
    //  5 3 6 1
    //  1, 6
    //  3, 5
    //  3, 6
    
    //  1 6 3 5
    public int[] restoreArray(int[][] pairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        
        for (int[] pair: pairs) {
            int a = pair[0];
            int b = pair[1];
            
            List<Integer> aNeighbors = map.getOrDefault(a, new ArrayList<>());
            aNeighbors.add(b);
            map.put(a, aNeighbors);
            
            List<Integer> bNeighbors = map.getOrDefault(b, new ArrayList<>());
            bNeighbors.add(a);
            map.put(b, bNeighbors);
        }
        
        int head = 0;
        for (Map.Entry<Integer, List<Integer>> entry: map.entrySet()) {
            if (entry.getValue().size() == 1) {
                head = entry.getKey();
                break;
            }
        }
        
        Set<Integer> visited = new HashSet<>();
        int[] res = new int[pairs.length+1];
        res[0] = head;
        visited.add(head);
        for (int i = 0; i < res.length; i++) {
            List<Integer> neighbors = map.get(res[i]);
            for (int num : neighbors) {
                if (!visited.contains(num)) {
                    res[i+1] = num;
                    visited.add(num);
                }
            }
        }
        
        return res;
    }
}
