class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        // Each element appears twice except for the start and the end.
        Map<Integer, ArrayList<Integer>> m = new HashMap<>();
        // from val to adjacent values 
        for(int[] pair: adjacentPairs) {
            int val1 = pair[0];
            int val2 = pair[1];
            
            
if(!m.containsKey(val1)) m.put(val1, new ArrayList<>());
            if(!m.containsKey(val2)) m.put(val2, new ArrayList<>());
            
            m.get(val1).add(val2);
            m.get(val2).add(val1);
        }
        
        int[] res = new int[adjacentPairs.length+1];
        int idx = 0;
        
        // get the start
        
for(Integer val: m.keySet()) {
            if(m.get(val).size() == 1) {
                res[idx] = val;
                idx++;
                break;
            }
        }
        
        while(idx < res.length) {
            int prev = res[idx-1];
            ArrayList<Integer> adj = m.get(prev);
            if(adj.size() == 1) {
                res[idx] = adj.get(0);
            } else {
                int prevPrev = res[idx-2];
                int next = adj.get(0) == prevPrev? adj.get(1) : adj.get(0);
                res[idx] = next;
            }
            idx++;
        }
        return res;
    }
}
