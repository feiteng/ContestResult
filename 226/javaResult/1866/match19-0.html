class Solution {
    public int[] restoreArray(int[][] ap) {
        int n = ap.length + 1;
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < n - 1; i++) {
            int[] p = ap[i];
            List<Integer> l0 = map.getOrDefault(p[0], new ArrayList<>());
            l0.add(p[1]);
            map.put(p[0], l0);
            List<Integer> l1 = map.getOrDefault(p[1], new ArrayList<>());
            l1.add(p[0]);
            map.put(p[1], l1);
        }
        
        int first = -1;
        for (int key: map.keySet()) {
            if (map.get(key).size() == 1) {
                first = key;
                break;
            }
        }
        
        int[] res = new int[n];
        res[0] = first;
        for (int i = 1; i < n; i++) {
            res[i] = (i == 1) ? map.get(res[i - 1]).get(0) : (map.get(res[i - 1]).get(0) == res[i - 2]) ? map.get(res[i - 1]).get(1) : map.get(res[i - 1]).get(0);
        }
        
        return res;
    }
}
