class Solution {
    public int[] restoreArray(int[][] adjacentPairs) {
        int n = adjacentPairs
            .length + 1;
        Map<Integer, Set<Integer>> adjs = new HashMap<>();
        for (int[] adj : adjacentPairs) {
            int from = adj[0], to = adj[1];
            adjs.computeIfAbsent(from, a -> new HashSet<>()).add(to);
            adjs.computeIfAbsent(to, a -> new HashSet<>()).add(from);
        
}
        int[] ans = new int[n];
        for (Map.Entry<Integer, Set<Integer>> entry : adjs.entrySet()) {
            if (entry.getValue().size() == 1) {
                ans[0] = entry.getKey();
                ans[1] = entry.getValue().iterator().next();
                
break;
            }
        }
        outer: for (int i = 1; i < n - 1; ++i) {
            Set<Integer> neis = adjs.get(ans[i]);
            for (int next : neis) {
                if (next != ans[i - 1]) {
                    ans[i + 1] = next;
                    continue outer;
                
}
            }
        }
        return ans;
    }
}
