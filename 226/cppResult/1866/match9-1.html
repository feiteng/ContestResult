class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& a) {
        int n = a.size() + 1;
        vector<int> ans(n, 0x3f3f3f3f);
        unordered_map<int, vector<int>> ad;
        for(auto p:a){
            ad[p[0]].push_back(p[1]);
            ad[p[1]].push_back(p[0]);
        }
        for(auto x:ad){
            if(x.second.size() == 1){ 
                ans[0] = x.first;
                ans[1] = x.second[0];
                break;
            }
        }
        // if(n > 2) ans[1] = ad[ans[0]][1];
        for(int i = 2; i < n; i++){
            if(ad[ans[i - 1]][0] == ans[i - 2]) ans[i] = ad[ans[i - 1]][1];
            else ans[i] = ad[ans[i - 1]][0];
        }
        return ans;
    }
};
