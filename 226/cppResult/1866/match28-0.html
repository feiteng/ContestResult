/*
[4,3,2,1]
[-3,1,4,-2]
[100000,-100000]
*/
class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& pairs) {
        unordered_map<int, unordered_set<int>> m;
        unordered_set<int> first;
        
        for (auto& p : pairs) {
            p = { min(p[0], p[1]), max(p[0], p[1])};
            m[p[0]].insert(p[1]);
            m[p[1]].insert(p[0]);
            //m[p[1]] = p[0];
            
            if (first.count(p[0])) {
                first.erase(p[0]);
            } else {
                first.insert(p[0]);
            }
            
            if (first.count(p[1])) {
                first.erase(p[1]);
            } else {
                first.insert(p[1]);
            }
        }
        //for (auto [k,v] : m) cout << k << ":" << v << endl;
        
        vector<int> ret = { *first.begin() };
        while (ret.size() != pairs.size() + 1) {
            int v = *m[ret.back()].begin();
            m[v].erase(ret.back());
            
            m[ret.back()].erase(m[ret.back()].begin());
            ret.push_back(v);
        }
        return ret;
    }
};
