class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> hash;
        unordered_set<int> hash2;
        vector<int> ans;
        int n = adjacentPairs.size();
        
        for(auto item : adjacentPairs)
        {
            hash[item[0]].push_back(item[1]);
            hash[item[1]].push_back(item[0]);

        }
        
        int beginPoint = INT_MIN;
        
        for(auto [x, y] : hash)
        {
            if(y.size() == 1)
            {
                beginPoint = x;
                break;
            }
        }
        
        ans.push_back(beginPoint);
        hash2.insert(beginPoint);
        
        while(ans.size() < n + 1)
        {
            int temp = ans.back();
            for(auto item : hash[temp])
                if(hash2.find(item) == hash2.end())
                {
                    ans.push_back(item);
                    hash2.insert(item);
                }
        }
        
        return ans;
        
    }
};
