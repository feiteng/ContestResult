class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& A) {
        int n = A.size();
        unordered_map<int, vector<int>> h;
        for(auto& v : A) {
            h[v[0]].push_back(v[1]);
            h[v[1]].push_back(v[0]);
        }
        int first = -1;
        for(auto& [x, v] : h) {
            if(v.size() == 1) {
                first = x;
                break;
            }
        }
        vector<int> ans;
        ans.push_back(first);
        ans.push_back(h[first][0]);
        int cur = ans.back();
        for(int i = 0; i < n - 1; ++i) {
            auto& v = h[cur];
            if(v[0] == ans[ans.size() - 2]) {
                cur = v[1];
            }else {
                cur = v[0];
            }
            ans.push_back(cur);
        }
        return ans;
    }
};
