class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> g;
        unordered_map<int, int> cnt;
        for(auto p: adjacentPairs) {
            g[p[0]].push_back(p[1]);
            g[p[1]].push_back(p[0]);
            cnt[p[0]]++;
            cnt[p[1]]++;
        }
        int u;
        for(auto& itr: cnt) {
            if(itr.second == 1) {
                u = itr.first;
                break;
            }
        }
        queue<int> q;
        vector<int> res= {u};
        q.push(u);
        unordered_map<int, bool> visited;
        visited[u] = true;
        while(!q.empty()) {
            u = q.front();
            q.pop();
            for(auto v: g[u]) {
                if(!visited[v]) {
                    res.push_back(v);
                    q.push(v);
                    visited[v] = true;
                }
            }
        }
        return res;
    }
};
