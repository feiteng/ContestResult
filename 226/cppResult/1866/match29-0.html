class Solution {
public:
  vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
    unordered_map<int, vector<int>> m;
    for (auto& p:adjacentPairs) {
      m[p[0]].push_back(p[1]);
      m[p[1]].push_back(p[0]);
    }
    
    vector<int> rv;
    for (auto& p:m) {
      if (p.second.size() == 1) {
        rv.push_back(p.first);
        rv.push_back(p.second.front());
        break;
      }
    }
    
    for (int i = 1; i < adjacentPairs.size(); i++) {
      for (int y:m[rv.back()]) {
        if (y == rv[i - 1]) {
          continue;
        }
        rv.push_back(y);
      }
    }
    return rv;
  }
};
