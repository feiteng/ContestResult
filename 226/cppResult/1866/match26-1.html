class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjp)
    {
        unordered_map<int, vector<int>>  adj;
        unordered_map<int, int> cnt;
        for(auto& p : adjp)
        {
            adj[p[0]].push_back(p[1]);
            adj[p[1]].push_back(p[0]);

            cnt[p[0]]++;
            cnt[p[1]]++;
        }
        
        int from = 0;
        for(auto& [key, val] : cnt)
        {
            if(val == 1)
            {
                from = key;
                break;
            }
        }
        
        vector<int> ret;
        unordered_set<int> used;
        dfs(from, adj, used, ret);
        return ret;
    }
    
    void dfs(int idx, unordered_map<int, vector<int>>& adj, unordered_set<int>& used, vector<int>& ret)
    {
        if(used.count(idx))
            return;
        used.insert(idx);
        ret.push_back(idx);
        for(auto& nidx : adj[idx])
        {
            dfs(nidx, adj, used, ret);
        }
    }
};
