class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        map<int,vector<int>> mp;
        for(auto v : adjacentPairs){
            mp[v[0]].push_back(v[1]);
            mp[v[1]].push_back(v[0]);
        }
        int start = 0;
        for(auto p : mp){
            if(p.second.size() == 1){
                start = p.first;
                break;
            }
        }
        vector<int> ret(adjacentPairs.size()+1,0);
        ret[0] = start;
        int idx = 2;
        int n = ret.size();
        ret[1] = mp[start][0];
        while(idx < n){
            
            int temp = ret[idx - 1];
            //cout << idx <<temp << endl;
            for(int i : mp[temp]){
                if(i != ret[idx-2]){
                    //cout << idx << endl;
                    ret[idx] = i;
                    idx++;
                    break;
                }
            }
            //cout << idx << endl;
        }
        return ret;
    }
};
