class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, int> cnt;
        unordered_map<int, vector<int>> edge;
        int N = adjacentPairs.size() + 1;
        for (auto a : adjacentPairs) {
            cnt[a[0]]++;
            cnt[a[1]]++;
            edge[a[0]].push_back(a[1]);
            edge[a[1]].push_back(a[0]);
        }
        int start = 0;
        for (auto [k, v] : cnt) {
            if (v == 1) {
                start = k;
                break;
            }
        }
        vector<int> ans;
        unordered_set<int> vis;
        for (int i=0; i<N-1; i++) {
            int pre = start;
            vis.insert(pre);
            ans.push_back(pre);
            start = edge[pre][0];
            if (vis.count(start)) start = edge[pre][1];
        }
        ans.push_back(start);
        return ans;
    }
};
