class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjs) {
        int n = adjs.size();
        unordered_map<int, vector<int>> adj;
        
        vector<int> res(n + 1);
        for(auto v : adjs) {
            adj[v[0]].push_back(v[1]);
            adj[v[1]].push_back(v[0]);
        }
        for(auto &[k, v] : adj) {
            if(v.size() == 1) {
                res[0] = k;
                res[1] = v[0];
                break;
            }
        }
        for(int i = 2; i <= n; i ++) {
            auto &v = adj[res[i - 1]];
            if(v[0] == res[i - 2]) res[i] = v[1];
            else res[i] = v[0];
        }
        return res;
    }
};
