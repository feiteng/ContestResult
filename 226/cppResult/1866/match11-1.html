class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adj) {

        unordered_map<int, vector<int>> mp;
        for(auto &v: adj) {
            mp[v.front()].push_back(v.back());
            mp[v.back()].push_back(v.front());
        }
        
        int ini = -1;
        for(auto &p: mp) {
            if(p.second.size() == 1) {
                ini = p.first;
                break;
            }
        }
        
        vector<int> ans;
        ans.push_back(ini);
        int last = ini;

        int cur = mp[ini].front();
        while(mp[cur].size() > 1) {
            ans.push_back(cur);
            int nxt = (mp[cur].front() == last ? mp[cur].back() : mp[cur].front());
            last = cur;
            cur = nxt;
        }
        ans.push_back(cur);
        return ans;
    }
};
