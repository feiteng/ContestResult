class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        map<int ,vector<int >> mp;
        vector<int >ans;
        for(auto it:adjacentPairs){
            mp[it[0]].push_back(it[1]);
            mp[it[1]].push_back(it[0]);
        }
        int l = 0;
        for(auto it: mp){
            if(it.second.size() == 1){
                l = it.first;
                break;
            }
        }
        ans.push_back(l);
        while(ans.size() != adjacentPairs.size() + 1){
            int x = ans.back();
            if(ans.size() == 1)
                ans.push_back(mp[x][0]);
            else{
                int y = mp[x][0],z = mp[x][1];
                if(y != ans[ans.size() - 2])
                    ans.push_back(y);
                else ans.push_back(z);
            }
        }
        return ans;
    }
};
