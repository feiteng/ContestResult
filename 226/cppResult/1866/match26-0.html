class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int,int> mm;
        unordered_map<int,int> mm2;
        for(auto e:adjacentPairs){
            mm[e[0]]++;
            mm[e[1]]++;
            mm2[e[0]]^=e[1];
            mm2[e[1]]^=e[0];
        }
        int s=-1;
        for(auto p:mm){
            if(p.second==1){
                s=p.first;
                break;
            }
        }
        vector<int> res;
        int last=0;
        for(int i=0;i<=adjacentPairs.size();i++){
            res.push_back(s);
            int olds=s;
            s=mm2[s]^last;
            last=olds;
        }
        return res;
    }
};
