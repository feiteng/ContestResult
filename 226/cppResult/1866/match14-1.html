class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {

        map<int, vector<int>> e;
        for (auto& p : adjacentPairs) {
            e[p[0]].push_back(p[1]);
            e[p[1]].push_back(p[0]);
        }
        int h;
        for (auto& x : e) {
            if (x.second.size() == 1) { h = x.first; break; }
        }
        vector<int> ans;
        ans.push_back(h);
        bool f = true;
        while (f) {
            f = false;

            auto& vs = e[ans.back()];
            for (auto& v : vs) {
                if (ans.size() == 1 || v != ans[ans.size()-2]) {
                    f = true;
                    //cout << v << endl;
                    ans.push_back(v);
                    break;
                }
            }
        }
        return ans;
    }
};
