class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        int n = adjacentPairs.size();
        map<int, int> M;
        map<int, vector<int>> G;
        int t[2];
        int pos = 0;  
        for(int i=0;i<n;i++){
            M[adjacentPairs[i][0]]++;
            M[adjacentPairs[i][1]]++;
            G[adjacentPairs[i][0]].push_back(adjacentPairs[i][1]);
            G[adjacentPairs[i][1]].push_back(adjacentPairs[i][0]);
        }
        for(auto& v: M) {
            if(v.second == 1){
                t[pos++] = v.first;
            }
        }
        vector<int> sol(n+1); 
        sol[0] = t[0];
        sol[n] = t[1];
        for(int i=0;i<n;i++){
            if(i == 0){
                sol[i+1] = G[sol[i]][0];
            }
            else{
                if(G[sol[i]][0] == sol[i-1]){
                    sol[i+1] = G[sol[i]][1];
                }
                else{
                    sol[i+1] = G[sol[i]][0];
                }
            }
        }
        return sol;
    }
};
