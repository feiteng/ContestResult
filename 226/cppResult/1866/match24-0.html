class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> m;
        for(auto& v : adjacentPairs) {
            m[v[0]].push_back(v[1]);
            m[v[1]].push_back(v[0]);
        }
        int start;
        for(auto& [k, v] : m) {
            if(v.size() != 1) 
                continue;
            start = k;
            break;
        }
        
        unordered_set<int> s;
        vector<int> ret;
        for(int i = 0; i < adjacentPairs.size() + 1; i++) {
            ret.push_back(start);
            s.insert(start);
            for(auto& v : m[start]) {
                if(s.find(v) == s.end()) {
                    start = v;
                    break;
                }
            }
        }
        return ret;
    }
};
