class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int,vector<int>> umap;
        for(int i =0;i<adjacentPairs.size();i++){
            umap[adjacentPairs[i][0]].emplace_back(adjacentPairs[i][1]);
            umap[adjacentPairs[i][1]].emplace_back(adjacentPairs[i][0]);
        }
        int start = 0;
        vector<int> ans(adjacentPairs.size()+1);

        for(auto it = umap.begin();it!=umap.end();it++){
            if(it->second.size()==1){
                start = it->first;
                break;
            }
        }
        unordered_map<int,bool> vis;
        for(int i =0;i<ans.size();i++){
            ans[i] = start;
            vis[start] = true;
            if(i==ans.size()-1){
                break;
            }

            if(vis[umap[start][0]]){
                start = umap[start][1];
            }else{
                start = umap[start][0];
            }
        }
        return ans;
    }
};
