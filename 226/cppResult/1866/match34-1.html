class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& a) {
        int m = a.size();
        unordered_map<int,unordered_set<int>> mp;
        for (int i = 0; i < m; i++)
        {
            mp[a[i][0]].insert(a[i][1]);
            mp[a[i][1]].insert(a[i][0]);
        }
        int left;
        for (auto& m: mp)
        {
            if (m.second.size() == 1)
            {
                left = m.first;
                break;
            }
        }
        vector<int> ret(m + 1);
        //ret[0] = left;
        for (int i = 0; i < m; i++)
        {
            ret[i] = left;
            left = *mp[left].begin();
            mp[left].erase(ret[i]);
            //cout << ret[i] << endl;
        }
        ret[m] = left;
        return ret;
    }
};
