class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adj) {
        map<int,vector<int>> mapp;
        for(auto a:adj){
            mapp[a[0]].push_back(a[1]);
            mapp[a[1]].push_back(a[0]);
        }
        vector<int> ans;
        for(auto a:mapp){
            if(a.second.size()==1){
                ans.push_back(a.first);
            }
        }
        int back=ans.back();
        ans.pop_back();
        ans.push_back(mapp[ans.back()][0]);
        for(int i=0;i<adj.size()-1;i++){
            int now=ans.back();
            int last=ans[ans.size()-2];
            auto v=mapp[now];
            if(v[0]==last){
                ans.push_back(v[1]);
            }else{
                ans.push_back(v[0]);
            }
        }
        return ans;

    }
};
