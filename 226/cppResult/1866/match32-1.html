class Solution {
public:
    vector<int> ret;
    
    vector<int> poi;
    vector<vector<int>> gra;
    void dfs(int u,int prev){
        ret.push_back(poi[u]);
        for(int i=0;i<(int)gra[u].size();i++){

            int nx = gra[u][i];
            if(nx == prev)continue;
            dfs(nx,u);
        }
    }
    vector<int> restoreArray(vector<vector<int>>& arrmv) {
        unordered_set<int> ms;
        unordered_map<int,int> mm;
        for(int i=0;i<arrmv.size();i++){
            int u,v;
            u=arrmv[i][0];
            v=arrmv[i][1];
            ms.insert(u);
            ms.insert(v);
        }
        int cnt = 0;
        poi = vector<int>();
        for(auto it:ms){
            poi.push_back(it);
            mm[it] = poi.size()-1;
        }
        vector<int> deg(ms.size(),0);
        gra.assign(ms.size(),vector<int>());
        for(int i=0;i<arrmv.size();i++){
            int u = arrmv[i][0];
            int v = arrmv[i][1];
            int au = mm[u];

            int av = mm[v];
            gra[au].emplace_back(av);
            gra[av].emplace_back(au);
            deg[au]++;
            deg[av]++;
        }
        int bu = -1;
        for(int i=0;i<deg.size();i++){
            if(deg[i] == 1){
                bu = i;
                break;
            }
        }
        ret = vector<int>();
        dfs(bu,-1);
        return ret;
    }
};
