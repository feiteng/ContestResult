class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        int n = adjacentPairs.size() + 1;
        map<int, vector<int>> mp;
        for (int i = 0; i < n - 1; i++) {
            mp[adjacentPairs[i][0]].push_back(adjacentPairs[i][1]);
            mp[adjacentPairs[i][1]].push_back(adjacentPairs[i][0]);
        }
        int st;
        for (auto it : mp) {
            if (it.second.size() == 1) {
                st = it.first;
                break;
            }
        }
        vector<int> ans(n);
        int last = 1e6;
        ans[0] = st;
        for (int i = 1; i < n; i++) {
            for (auto j : mp[st]) {
                if (j != last) {
                    ans[i] = j;
                    break;
                }
            }
            last = st;
            st = ans[i];
        }
        return ans;
    }
};
