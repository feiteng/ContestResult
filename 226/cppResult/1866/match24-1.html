class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        map<int,vector<int> > m;
        for(auto it:adjacentPairs){
            m[it[0]].push_back(it[1]);
            m[it[1]].push_back(it[0]);
        }
        for(auto it:m){
            if(it.second.size()==1){
                vector<int> v;
                v.push_back(it.first);
                int x=it.second[0];

                while(m[x].size()!=1){
                    for(auto it2:m[x]){
                        if(it2!=v.back()){
                            v.push_back(x);
                            x=it2;
                            break;
                        }
                    }
                }
                v.push_back(x);
                return v;
            }
        }
        return vector<int>();
    }
};
