class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& ad) {
        vector<int> res;
        unordered_map<int,vector<int> > m;
        for(auto& v: ad) {
            m[v[0]].push_back(v[1]);
            m[v[1]].push_back(v[0]);
        }
        int begin;
        for(auto it: m) {
            if(it.second.size() == 1) {
                begin = it.first;
                break;
            }
        }
        while(res.size() < ad.size()+1) {
            int back = res.size() ? res.back() : INT_MAX;
            res.push_back(begin);
            int tmp = begin;
            for(auto& i: m[tmp]) {
                if(i != back) {begin = i;break;}
            }
        }
        return res;
    }
};
