class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, unordered_set<int>> adj;
        for (auto& p : adjacentPairs) {
            adj[p[0]].insert(p[1]);
            adj[p[1]].insert(p[0]);
        }
        vector<int> res;

        for (auto& [k, v] : adj) {
            if (v.size() == 1) {
                res.push_back(k);
                break;
            }
        }
        while (!adj.empty()) {
            int prev = res.back();
            int nxt = *(adj[prev].begin());
            res.push_back(nxt);
            adj[prev].erase(nxt);
            adj[nxt].erase(prev);
            if (adj[nxt].empty()) adj.erase(nxt);
            if (adj[prev].empty()) adj.erase(prev);
        }
        return res;
    }
};
