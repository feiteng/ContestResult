class Solution {
public:
    unordered_map<int, int> mp, vis;
    unordered_map<int, vector<int>> E;
    vector<int> res;
    void dfs(int u){
        res.push_back(u);
        vis[u] = 1;
        for(auto &v: E[u]){
            if(vis.count(v)) continue;
            dfs(v);
        }
    }
    vector<int> restoreArray(vector<vector<int>>& es) {
        mp.clear(); vis.clear(); E.clear(); res.clear();
        

        for(auto &e: es){
            mp[e[0]] ++ ;
            mp[e[1]] ++ ;
            E[e[0]].push_back(e[1]);
            E[e[1]].push_back(e[0]);
        }
        int root;
        for(auto &[u, v] : mp){
            if(v == 1){
                root = u;
                break;
            }
        }
        dfs(root);

        
        return res;
    }
};
