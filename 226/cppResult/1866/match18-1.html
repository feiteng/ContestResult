class Solution {
public:
  vector<int> restoreArray(vector<vector<int>>& adj_pairs) {
    unordered_map<int, vector<int>> adj;
    for(auto& p : adj_pairs) {
      int a = p[0], b = p[1];
      adj[a].push_back(b), adj[b].push_back(a);
    }
    int s, n=adj_pairs.size();

    for(auto& p : adj) {
      if(p.second.size() == 1) {
        s = p.first;
        break;
      }
    }
    vector<int> res(n+1);
    res[0] = s;
    for(int i=1; i<=n; ++i) {
      printf("s=%d\n", s);
      int x = adj[s].size() == 1 ? adj[s][0] : adj[s][0] + adj[s][1] - res[i-2];
      s = res[i] = x;
    }
    return res;
  }
};
