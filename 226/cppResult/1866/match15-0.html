class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        if(adjacentPairs.size() == 0)
        {
            return {};
        }
        
        if(adjacentPairs.size() == 1)
        {
            return adjacentPairs[0];
        }
    
    
        unordered_map<int, vector<int>> g;
        for(auto & e : adjacentPairs)
        {
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }

        int idx;
        for(auto & kv : g)
        {
            if(kv.second.size() == 1)
            {
                idx = kv.first;
                break;
            }
        }

        int n = g.size();
        unordered_set<int> visit;
        vector<int> result;

        while(result.size() < n)
        {
            result.push_back(idx);
            visit.insert(idx);

            for(auto v : g[idx])
            {
                if(visit.find(v) == visit.end())
                {
                    idx = v;
                    break;
                }
            }
        }

        return result;
    }
};
