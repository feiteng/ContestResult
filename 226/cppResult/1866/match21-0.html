class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        int n = adjacentPairs.size()+1;
        unordered_map<int, set<int>> to;
        unordered_map<int, int> count;
        for (auto p : adjacentPairs) { 
            count[p[0]]++;
            count[p[1]]++;
            to[p[0]].insert(p[1]);
            to[p[1]].insert(p[0]);
        }
        int start = 0;
        for (auto p : count) if (p.second == 1) { start = p.first; break; }
        vector<int> result(n); result[0] = start;
        for (int i = 1; i < n; i++) {
            result[i] = *(to[result[i-1]].begin());
            to[result[i]].erase(result[i-1]);
        }
        return result;
    }
};
