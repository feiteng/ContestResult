class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        int n = adjacentPairs.size() + 1;
        unordered_map<int, vector<int>> adj;
        
        for (auto & p: adjacentPairs) {
            adj[p[0]].push_back(p[1]);
            adj[p[1]].push_back(p[0]);
        }
        
        // the vertices with only one neighbor are the ends
        int start = 1e7;
        
        for (auto& v: adj) {
            if (v.second.size() == 1) {
                start = v.first;
                break;
            }
        }
        
        vector<int> res {start};
        int prev = start;
        start = adj[prev][0];
        
        while(adj[start].size() == 2) {
            res.push_back(start);
            int nextOne = adj[start][0] == prev? adj[start][1]: adj[start][0];
            prev=  start;
            start = nextOne;
        }
        
        res.push_back(start);
        return res;
    }
};
