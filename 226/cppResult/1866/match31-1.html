class Solution {
public:
    struct Node{
        int val;
        Node* pre;
        Node* next;
    };

    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> rec;
        for(auto& v : adjacentPairs){
            rec[v[0]].push_back(v[1]);
            rec[v[1]].push_back(v[0]);
        }
        
        int pre = 0;
        for(auto kv : rec){
            if(kv.second.size() < 2){
                pre = kv.first;
                break;
            }
        }
        vector<int> ret;
        int preN = INT_MAX;
        int n = adjacentPairs.size() + 1;
        while(n){
            ret.push_back(pre);
            auto val = rec[pre][0];
            if(rec[pre].size() > 1 && val == preN){
                val = rec[pre][1];
            }
            preN = pre;
            pre = val;
            n--;          
        }
        return ret;
    }
};
