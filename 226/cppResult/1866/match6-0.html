class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& a) {
        unordered_map<int, vector<int>> m;
        for (int i = 0; i < a.size(); i++) {
            m[a[i][0]].push_back(a[i][1]);
            m[a[i][1]].push_back(a[i][0]);
        }
        int s = 0;
        for (auto const& x : m) {
            if (x.second.size() == 1) {
                s = x.first;
                break;
            }
        }
        vector<int> n;
        for (int i = 0; i <= a.size(); i++) {
            n.push_back(s);
            if (m[s].size() == 2 && m[s][0] == n[i-1]) {
                s = m[s][1];
            }
            else {
                s = m[s][0];
            }
        }
        return n;
    }
};
