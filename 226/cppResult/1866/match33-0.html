class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        vector<int> res(adjacentPairs.size()+1);
        unordered_map<int, vector<int>> m;
        for (auto &v : adjacentPairs) {
            m[v[0]].push_back(v[1]);
            m[v[1]].push_back(v[0]);
        }
        int cur, pre;
        for (auto &it: m) {
            if (it.second.size() == 1) {
                cur = it.first;
                break;
            }
        }
        res[0] = cur;
        pre = cur;
        cur = m[cur][0];
        for (int i = 1; i <res.size(); ++i) {
            res[i] = cur;
            if (m[cur].size() == 1)
                break;
            if (m[cur][0] == pre) {
                pre = cur;
                cur = m[cur][1];
            } else {
                pre = cur;
                cur = m[cur][0];
            }
        }
        
        return res;
    }
};
