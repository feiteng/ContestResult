class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        map<int, vector<int>> g;
        for(auto &x: adjacentPairs) {
            g[x[0]].push_back(x[1]);
            g[x[1]].push_back(x[0]);
           
        }
       vector<int> res;
        queue<int> q;set<int> vis;
        for (auto [k,v]: g) {
            if (v.size() == 1) {
                q.push(k);
                res.push_back(k);
                 vis.insert(k);
                break;
            }
        }
        
       
        while (!q.empty()) {
            int t = q.front(); q.pop();
            
            for (auto nxt: g[t]) {
                if (nxt != t and !vis.count(nxt)) {
                    res.push_back(nxt);
                    vis.insert(nxt);
                    q.push(nxt);
                }
            }
        }
        
        
        
        
        return res;
        
        
    }
};
