class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, unordered_set<int>> um;
        for (auto ap : adjacentPairs) {
            int i = ap[0], j = ap[1];
            um[i].insert(j);
            um[j].insert(i);
        }
        vector<int> res;
        int head = 0;
        for (auto p : um) {
            if (p.second.size() == 1) {
                head = p.first;
                break;
            }
        }
        res.push_back(head);
        int n = adjacentPairs.size() + 1;
        while (res.size() < n) {
            int next = *um[head].begin();
            um[head].erase(next);
            um[next].erase(head);
            res.push_back(next);
            head = next;
        }
        return res;
    }
};
