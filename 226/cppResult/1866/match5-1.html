class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adj) {
        int n = adj.size() + 1;
        unordered_map<int,vector<int>> pre;
        unordered_map<int,int> count;
        for (auto& a:adj) {
            int x = a[0], y = a[1];
            pre[x].push_back(y);
            pre[y].push_back(x);
            count[x]++;
            count[y]++;
        }
        int start;
        for (auto& [k,times]:count) {
            if (times == 1) {
                start = k;
            }

        }
        unordered_set<int> visit;
        vector<int> ans(n);
        ans[0] = start;
        visit.insert(start);
        for (int i = 1; i < n; i++) {
            for (auto& neigh:pre[start]) {
                if (!visit.count(neigh)) {
                    visit.insert(neigh);
                    start = neigh;
                    ans[i] = neigh;
                    break;
                }
            }
        }
        return ans;
    }
};
