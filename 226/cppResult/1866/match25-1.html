class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& ad) {
        unordered_map<int, vector<int>> table;
        int n = ad.size() + 1;

        for (auto &a : ad) {
            table[a[0]].push_back(a[1]);
            table[a[1]].push_back(a[0]);
        }
        vector<int> ans;
        for (auto [a, vec] : table) {
            if (vec.size() == 1) {
                ans.push_back(a);
                ans.push_back(vec[0]);
                break;
            } 
        }
        for (int i = 2; i < n; ++i) {
            auto& vec = table[ans.back()];

            for (int a : vec) {
                if (a != ans[i - 2]) {
                    ans.push_back(a);
                    break;
                }
            }
        }
        return ans;
    }
};
