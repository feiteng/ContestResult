class Solution {
    public:
    map<int,vector<int>>A;
    map<int,int>vis;
    vector<int>ans;
    void dfs(int v)
    {
        vis[v]=1;
        ans.push_back(v);
        for(auto c:A[v])
        {
            if(vis[c]==0)
            {
                dfs(c);
            }
        }
    }
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        
        int i,n=adjacentPairs.size()+1;
        map<int,int>M;
        for(i=0;i<n-1;i++)
        {
            M[adjacentPairs[i][0]]++;
            M[adjacentPairs[i][1]]++;
            A[adjacentPairs[i][0]].push_back(adjacentPairs[i][1]);
            A[adjacentPairs[i][1]].push_back(adjacentPairs[i][0]);
            
        }
        
        int source;
        for(auto it:M)
        {
            if(it.second==1){source=it.first;break;}
        }
        
        dfs(source);
        return ans;
    }
};
