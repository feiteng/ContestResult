class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& ad) {
        int n = ad.size();
        unordered_map<int, set<int>> m;
        for(auto& p : ad) {
            m[p[0]].insert(p[1]);
            m[p[1]].insert(p[0]);
        }
        
        int st = 0;
        for(auto& p : m) {
            if(p.second.size() == 1) {
                st = p.first;
                break;
            }
        }
        
        vector<int> res(n+1, 0);
        res[0] = st;
        for(int i = 1; i < n+1; ++i) {
            res[i] = *m[res[i-1]].begin();
            m[res[i-1]].erase(m[res[i-1]].begin());
            m[res[i]].erase(m[res[i]].find(res[i-1]));
        }
        
        return res;
    }
};
