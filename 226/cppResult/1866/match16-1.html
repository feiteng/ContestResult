class Solution {
public:
    void dfs(unordered_map<int, vector<int>>& graph, int idx, vector<int> &ans, unordered_set<int>& vis)
    {
        ans.push_back(idx); 
        vis.insert(idx); 
        for (auto j : graph[idx])
        {
            if (vis.count(j)) continue; 
            else dfs(graph, j, ans, vis); 
        }
    }
    vector<int> restoreArray(vector<vector<int>>& adj) {
        int n = adj.size() + 1; 
        unordered_map<int, vector<int>> graph;
        
        for (int i = 0; i < adj.size(); i++)

        {
            graph[adj[i][0]].push_back(adj[i][1]); 
            graph[adj[i][1]].push_back(adj[i][0]);
        }
        int idx = INT_MAX; 
        for (auto it : graph)
        {
            if (it.second.size() == 1)
            {
                idx = it.first; 
                break;
            }
        }
        vector<int> ans; 
        unordered_set<int> vis; 
        dfs(graph, idx, ans, vis); 
        return ans;
    }
};
