class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> connected;
        unordered_set<int> visited;
        for (auto &v: adjacentPairs) {
            connected[v[0]].push_back(v[1]);
            connected[v[1]].push_back(v[0]);

            if (visited.find(v[0]) != visited.end()) {
                visited.erase(v[0]);
            } else {
                visited.insert(v[0]);
            }
            if (visited.find(v[1]) != visited.end()) {
                visited.erase(v[1]);
            } else {
                visited.insert(v[1]);
            }
        }
        int curr = *visited.begin();
        visited.clear();
        vector<int> ret = {curr};
        visited.insert(curr);
        while (ret.size() < adjacentPairs.size()+1) {
            if (visited.find(connected[curr][0]) == visited.end()) {
                curr = connected[curr][0];
            } else {
                curr = connected[curr][1];
            }
            ret.push_back(curr);
            visited.insert(curr);
        }
        return ret;
    }
};
