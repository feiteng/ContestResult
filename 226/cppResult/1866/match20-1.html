class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& adjacentPairs) {
        unordered_map<int, vector<int>> edges;
        for (vector<int>& p : adjacentPairs) {
            edges[p[0]].push_back(p[1]);
            edges[p[1]].push_back(p[0]);
        }
        
        int a, b;
        for (auto it : edges) {
            if (it.second.size() == 1) {
                a = it.first;
                b = it.second[0];
                break;
            }
        }
        
        vector<int> result(adjacentPairs.size() + 1);
        result[0] = a;

        result[1] = b;
        for (int i = 2; i < result.size(); ++i) {
            for (int x : edges[result[i-1]]) {
                if (x != result[i-2]) {
                    result[i] = x;
                    break;
                }
            }
        }
        return result;
    }
};
