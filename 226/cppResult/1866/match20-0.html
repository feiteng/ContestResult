class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& ap) {
        unordered_map<int, vector<int>> mp;
        for(vector<int>& v : ap) {
            mp[v[0]].push_back(v[1]);
            mp[v[1]].push_back(v[0]);
        }
        
        int first;
        for(auto it : mp) {
            if(it.second.size() == 1) {
                first = it.first;
                break;
            }
        }
        int n = ap.size() + 1;
        
        vector<int> op(n);
        op[0] = first;
        unordered_map<int, bool> visited;
        visited[first] = true;
        for(int i = 1; i < n; i++) {
            for(int x : mp[op[i - 1]]) {
                if(!visited[x]) {
                    op[i] = x;
                    visited[x] = true;
                    break;
                }
            }
        }
        
        return op;
    }
};
