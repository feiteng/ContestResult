class Solution {
public:
    vector<int> restoreArray(vector<vector<int>>& a) 
    {
        int n=a.size(),i,val;
        unordered_map<int,int> m;
        unordered_map<int,vector<int>> mp;
        vector<int> ans;

        for(i=0;i<n;i++)
        {
            m[a[i][0]]++;
            m[a[i][1]]++;
            mp[a[i][0]].push_back(a[i][1]);
            mp[a[i][1]].push_back(a[i][0]);
        }
        for(auto i:m)
        {
            if(i.second%2==1)
            {
                val=i.first;
                break;
            }
        }
        ans.push_back(val);

        for(i=0;i<n;i++)
        {
            if(i==0)
            {
                ans.push_back(mp[val][0]);
                val=mp[val][0];
            }
            else
            {
                if(ans[ans.size()-2]==mp[val][0])
                {
                    ans.push_back(mp[val][1]);
                    val=mp[val][1];
                }
                else
                {
                    ans.push_back(mp[val][0]);
                    val=mp[val][0];
                }
            }
        }
        return ans;
    }
};
