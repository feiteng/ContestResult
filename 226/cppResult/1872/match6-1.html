class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
      int candidate_size = candiesCount.size(), query_size = queries.size();
      vector<long long> prefix_sum(candidate_size,0);
      prefix_sum[0] = candiesCount[0];

      for(int i = 1; i < candidate_size; i++){
        prefix_sum[i] = prefix_sum[i - 1] + candiesCount[i];
      }
      vector<bool> result(query_size,true);
      for(int i = 0; i < query_size; i++){
        long long favor_type = queries[i][0], favor_day = queries[i][1],day_cap = queries[i][2];
        if(prefix_sum[favor_type] < favor_day + 1 || (favor_type != 0 && prefix_sum[favor_type - 1] >= (favor_day + 1) * day_cap)){
          result[i] = false;
        }
      }
      return result;
    }
};
