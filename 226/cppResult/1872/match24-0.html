class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector<long long> sum(n+1, 0);
        for(int i = 1; i <= n; ++i){
            sum[i] = sum[i-1] + candiesCount[i-1];
        }
        vector<bool> ans;
        for(int i = 0; i < queries.size(); ++i){
            int type = queries[i][0];
            int day = queries[i][1];
            int cap = queries[i][2];
            int sum1 = day;
            long long sum2 = (long long)day*cap;
            // cout << sum[type] << " " << sum2 << endl;
            if(sum[type+1] > sum1 && sum1 >= sum[type]){
                ans.push_back(true);
            }
            else if(sum2+cap > sum[type]&&sum1<sum[type+1]){
                ans.push_back(true);
            }
            else{
                ans.push_back(false);
            }
        }
        return ans;
    }
};
