class Solution {
public:
    #define ll long long
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector <ll> sum(n + 1);
        for (int i = 1; i <= n; i ++)
            sum[i] = sum[i-1] + candiesCount[i-1];
        vector <bool> ans;
        for (auto &vec : queries)
        {
            int ft = vec[0], fd = vec[1], c = vec[2];
            ll L = fd + 1, R = 1ll * c * (fd + 1);
            ll l = sum[ft] + 1, r = sum[ft+1];
            if (r < L || R < l) 
                ans.push_back(false);
            else 
                ans.push_back(true);
        }
        return ans;
    }
};
