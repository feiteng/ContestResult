class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size(), m = queries.size();
        vector<bool> res(m, false);
        vector<long long> pre(n + 1, 0);
        for (int i = 1; i <= n; ++i)
            pre[i] = candiesCount[i - 1] + pre[i - 1];
        for (int i = 0; i < m; ++i) {
            long long type = queries[i][0], day = queries[i][1], cap = queries[i][2];
            //cout << sum << " "<< day << " "<<(long)cap * day << endl;
            if (day  >= pre[type] / cap && day < pre[type + 1])
                res[i] = true;    
        }
        return res;
    }
};
