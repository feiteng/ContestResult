class Solution {
public:
vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
	vector<long long> prefixSum;
	prefixSum.push_back(0);
	long long sum = 0;
	for (int x : candiesCount)
		sum += x, prefixSum.push_back(sum);
	vector<bool> res;
	for (auto q : queries) {
		int favouriteType = q[0];
		int favouriteDay = q[1];
		int dailyCap = q[2];
		long long prev = prefixSum[favouriteType];
		long long curr = prefixSum[favouriteType + 1];
		long long minDays = prev / dailyCap;
		long long maxDays = curr - 1;
		if (favouriteDay >= minDays && favouriteDay <= maxDays)
			res.push_back(1);
		else res.push_back(0);
	}
	return res;
}
};
