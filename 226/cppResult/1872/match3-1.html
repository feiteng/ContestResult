class Solution {

    long long sum[100005];
    vector<bool> ans;
public:
    vector<bool> canEat(vector<int>& ccnt, vector<vector<int>>& qq) {
        int n = ccnt.size();
        sum[0] = 0;
        for (int i = 1; i <= n; ++i)
            sum[i] = sum[i - 1] + 1ll * ccnt[i - 1];
        for (auto& q: qq){
            bool flag1 = sum[q[0] + 1] <= 1ll * q[1];
            bool flag2 = sum[q[0]] >= 1ll * q[2] * (q[1] + 1);
            if (flag1 || flag2) ans.push_back(false);
            else ans.push_back(true);
        }
        return ans;
    }
};
