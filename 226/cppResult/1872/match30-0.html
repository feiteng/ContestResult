class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int typeCnt = candiesCount.size();
        vector<long long> sum(typeCnt);
        sum[0] = candiesCount[0];
        for(int i = 1; i < typeCnt; i++)
            sum[i] = sum[i-1] + candiesCount[i];
        vector<bool> ans(queries.size());
        for(int i = 0; i < queries.size(); i++){
            long long t = queries[i][0];
            long long d = queries[i][1];
            long long c = queries[i][2];
            long long r = sum[t]-1;
            if(r < d){
                ans[i] = false;
                continue;
            }
            long long l = 0;
            if(t > 0)
                l += sum[t-1];
            l = l/c;
            if(l > d){
                ans[i] = false;
                continue;
            }
            ans[i] = true;
        }
        return ans;
    }
};
