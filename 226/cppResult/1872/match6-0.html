class Solution {
public:
    vector<bool> canEat(vector<int>& c, vector<vector<int>>& q) {
        vector<bool> ans(q.size()) ;
        vector<long long> prefix(c.size() + 1) ;
        for (int i = 0 ; i < c.size() ; i++) {
            prefix[i + 1] = prefix[i] + c[i] ; 
        }
        for (int i = 0 ; i < q.size() ; i++) {
            long long t = q[i][0] , d = q[i][1] , m = q[i][2] ;
            if (d < prefix[t + 1] && m * (d + 1) >  prefix[t]) {
                ans[i] = true ;
            }
        }
        return ans ;
    }
};
