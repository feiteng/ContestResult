class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector<bool > ans;
        vector<long long > s(n);
        s[0] = candiesCount[0];
        for(int i = 1;i < n;i++)
            s[i] += s[i - 1] + candiesCount[i];
        for(auto a:queries){
            long long l = 1ll * a[1];
            long long r = 1ll * a[1]  * a[2];

            long long sum = s[a[0]];
            long long first = a[0] == 0?0:s[a[0] - 1];
            if(l < sum && r + a[2] > first )
                ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
