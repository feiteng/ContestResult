typedef long long ll;
class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) 
    {
        int n = candiesCount.size();
        candiesCount.insert(candiesCount.begin(),0);
        vector<ll>presum(n+1);
        for (int i=1; i<=n; i++)
            presum[i] = presum[i-1]+candiesCount[i];
        
        vector<bool>rets;
        for (auto q: queries)
        {
            ll type = q[0]+1;
            ll day = q[1]+1;
            ll cap = q[2];
            
            // cout<<presum[43]<<" "<<presum[44]<<endl;
            
            ll low = day-1;
            if (low >= presum[type])
            {
                rets.push_back(false);
                continue;
            }
            ll high = (day)*cap;
            if (high <= presum[type-1])
            {
                rets.push_back(false);
                continue;
            }
            rets.push_back(true);
            
        }
        return rets;
        
    }
};
