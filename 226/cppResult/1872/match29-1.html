class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = queries.size();
        int m = candiesCount.size();
        vector<bool> ans(n, false);
        vector<int64_t> pref(m);
        pref[0] = candiesCount[0];
        for (int i = 1; i < m; i++) pref[i] = pref[i-1] + candiesCount[i];

        for (int i = 0; i < n; i++){
            int type = queries[i][0];
            int day = queries[i][1];
            int cap = queries[i][2];
            int64_t low = type == 0?0 : pref[type-1];
            if (1LL*(day+1)*cap>low && day < pref[type]) ans[i] =true;
        }
        return ans;
    }
};
