class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> psa;
        long long sum = 0;
        psa.push_back(sum);
        for (int i : candiesCount) {
            sum += i;
            psa.push_back(sum);
        }
        vector<bool> ans;
        for (vector<int> v : queries) {
            ans.push_back(psa[v[0]]<=1LL*(v[1]+1)*v[2]-1&&psa[v[0]+1]>=v[1]+1);
        }
        return ans;
    }
};
