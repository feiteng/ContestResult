class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> acc; //第i种类型的糖果+前面的糖果的总数
        acc.push_back(candiesCount[0]);
        for(int i = 1; i < candiesCount.size(); ++i){
            acc.push_back(acc.back()+candiesCount[i]);
        }
        
        vector<bool> ans;
        for(auto& q: queries){

            long long type = q[0], day = q[1], hi = q[2];
            //2.1求第i天开始时最少和最多能剩多少糖果
            long long l = acc[type] - candiesCount[type] - hi + 1;
            long long r = acc[type] - 1;
            //2.2求按最少和最大吃时到第i-1天能吃多少糖果
            long long ll = (day);
            long long rr = (day)*hi;
            if(rr < l || ll > r) ans.push_back(false);
            else ans.push_back(true);
        }
        return ans;
    }
};
