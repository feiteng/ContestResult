class Solution {
public:
    vector<bool> canEat(vector<int>& cnts, vector<vector<int>>& queries) {
        int n = cnts.size();
        long long sums[n + 1];
        sums[0] = 0;
        for(int i = 1; i <= n; ++i) {
            sums[i] = sums[i-1] + cnts[i-1];
        }
        vector<bool> res;
        for(auto& v : queries) {
            long long t = v[0] + 1, d = v[1] + 1, c = v[2];
            if(d <= sums[t] && c * d > sums[t-1]) {
                res.push_back(true);
            } else {
                res.push_back(false);
            }
        }
        return res;
    }
};
