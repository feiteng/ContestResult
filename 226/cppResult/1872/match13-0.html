class Solution {
public:
    #define ll long long
    #define maxn 100005
    
    ll sm[maxn];
    vector<bool> canEat(vector<int>& cand, vector<vector<int>>& query) {
        int n = cand.size();
        for (int i = 1; i <= n; i++) sm[i] = sm[i-1] + (ll)(cand[i-1]);
        vector<bool> ans;
        
        for (auto & vec : query) {
            ll type = vec[0]+1;
            ll day = vec[1], cap = vec[2];
            
            ll sum = sm[type], presum = sm[type-1];
            if (day >= sum || cap * (day+1) <= presum) {
                ans.push_back(0);
            } else {
                ans.push_back(1);
            }
        }
        return ans;
    }
};
