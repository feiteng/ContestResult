class Solution {
public:
    vector<bool> canEat(vector<int>& cd, vector<vector<int>>& qs) {
        #define ll long long
        int n = cd.size();
        int m = qs.size();
        vector<bool> ans(m, false);
        vector<ll> sm(n);
        sm[0] = cd[0];
        for(int i = 1; i < n; i++){
            sm[i] = sm[i-1] + (ll)cd[i];
        }
        for(int i = 0; i < m; i++){
            int typ = qs[i][0];
            int day = qs[i][1];
            int cap = qs[i][2];
            ll mn = (day + 1);
            ll mx = (day + 1) * (ll)cap;
            int l = lower_bound(sm.begin(), sm.end(), mn) - sm.begin();
            int r = lower_bound(sm.begin(), sm.end(), mx) - sm.begin();
            ans[i] = (typ >= l && typ <= r);
        }
        return ans;
        
    }
};
