class Solution {
public:
    vector<bool> canEat(vector<int>& cc, vector<vector<int>>& q) {
        int n = cc.size();
        vector<long long> sum(n+1, 0);
        for(int i = 1; i <= cc.size(); i++){
            sum[i] = sum[i - 1] + cc[i - 1];
            // cout<<sum[i]<<" ";
        }
        vector<bool> res(q.size());
        for(int i = 0; i < q.size(); i++){
            int a = q[i][0], b = q[i][1], c = q[i][2];
            if(c > sum[a] / (b + 1) && sum[a + 1] >= b + 1){
                res[i] = true;
            }else{
                res[i] = false;
            }
        }
        // cout<<endl;
        return res;
    }
};
