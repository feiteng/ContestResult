class Solution {
public:
    vector<bool> canEat(vector<int>& a, vector<vector<int>>& q) {
        int n=a.size();
        vector<long long> pre(n+1,0);
        for(int i=1;i<=n;i++){
            pre[i]=pre[i-1]+a[i-1];
        }
        vector<bool> ans(q.size());
        for(int i=0;i<q.size();i++){
            int type=q[i][0],day=q[i][1],limit=q[i][2];
            long long numdays1=(pre[type]/limit);
            long long numdays2=(pre[type+1]/limit);

            if(numdays1>day||pre[type+1]<=day){
                ans[i]=false;
            }else{
                ans[i]=true;
            }
        }
        return ans;
    }
};
