class Solution {
   public:
    vector<bool> canEat(vector<int>& candy, vector<vector<int>>& queries) {

        vector<long long> sum(candy.begin(), candy.end());
        for (int i = 1; i < sum.size(); ++i) {
            sum[i] += sum[i - 1];
        }
        vector<bool> ans;
        for (auto& it : queries) {

            if (sum[it[0]] <= it[1]) {  //天数太多
                ans.emplace_back(false);
            } else if (sum[it[0]] - candy[it[0]] >= 1ll * (it[1] +1)* it[2]) {  //之前的糖果太多
                ans.emplace_back(false);
            } else {
                ans.emplace_back(true);
            }
        }
        return ans;
    }
};
