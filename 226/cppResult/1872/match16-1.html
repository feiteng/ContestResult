class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int len = candiesCount.size();
        vector<long long> sums(len + 1, 0);
        for(int i = 0; i < len; i++) sums[i + 1] = sums[i] + candiesCount[i];
        vector<bool> res;

        for(int i = 0; i < queries.size(); i++) {
            int type = queries[i][0];
            long long minN = queries[i][1];
            long long maxN = queries[i][2] * (minN + 1);
            if(minN < sums[type + 1] && maxN > sums[type]) {
                res.push_back(true);
            }
            else
                res.push_back(false);
        }
        return res;
    }
};
