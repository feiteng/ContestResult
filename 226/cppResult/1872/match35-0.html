class Solution {
public:
    ll s[100004];
    vector<bool> canEat(vector<int>& c, vector<vector<int>>& queries) {
        int n=c.size();
        for(int i=0;i<n;i++)
            s[i+1]=s[i]+c[i];
        vector<bool>a;
        for(auto v:queries){
            int x=v[0],y=v[1],z=v[2];
            ++x;++y;
            //printf("%d %d %d %lld %lld\n",x,y,z,s[x],s[x-1]);
            if(y>s[x]) a.push_back(0);
            else if(1ll*z*y <= s[x-1]) a.push_back(0);
            else a.push_back(1);
        }
        return a;
    }
};
