class Solution {
public:
    vector<bool> canEat(vector<int>& cnt, vector<vector<int>>& qs) {
        int n = cnt.size();
        vector<long long> csum(n + 1);
        for(int i = 0; i < n; i++) {
            csum[i+1] = csum[i] + cnt[i];
        }
        
        vector<bool> res;

        for(auto q:qs) {
            long long t = q[0];
            long long d = q[1];
            long long cap = q[2];
            long long lb = csum[t] / cap;
            long long ub = csum[t + 1] - 1;
            res.push_back(lb <= d && d <= ub);
        }

        return res;
    }
};
