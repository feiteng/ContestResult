class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n=candiesCount.size();
        vector<long long> pref(n);
        for(int i=0;i<n;i++){
            pref[i]=(i!=0?pref[i-1]:0)+candiesCount[i];
        }
        vector<bool> ans;
        for(auto x:queries){
            long long day=x[1];
            long long maxiDay=x[2];
            long long type=x[0];
            
            long long needToFinish=(type!=0?pref[type-1]:0);
            if(needToFinish>(maxiDay*day+maxiDay-1)){
                // I cannot finish all candies till type-1;
                ans.push_back(false);
                continue;
            }
            long long temp=pref[type]-1;
            if(temp>=day){
                // I have 1 candy of type, which I can eat on day;
                ans.push_back(true);
            }else{
                ans.push_back(false);
            }
        }
        return ans;
    }
};
