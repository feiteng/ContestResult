typedef long long int64;

class Solution {
public:
    vector<bool> canEat(vector<int>& a, vector<vector<int>>& queries) {
        int n = a.size();
        vector<int64> pre(n + 1);
        for (int i = 0; i < n; ++i) pre[i + 1] = pre[i] + a[i];
        vector<bool> ret;
        for (auto& v : queries) {
            int t = v[0], d = v[1] + 1, c = v[2];
            int64 cur = 1LL * c * d;
            ret.push_back(cur >= pre[t] + 1 && d <= pre[t + 1]);
        }
        return ret;
    }
};
