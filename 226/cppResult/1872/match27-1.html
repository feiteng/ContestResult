class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = static_cast<int>(candiesCount.size());
        vector<long long> cnt(n);
        cnt[0] = candiesCount[0];

        for (int i = 1; i < n; ++i) {
            cnt[i] = cnt[i - 1] + candiesCount[i];
        }
        vector<bool> result;
        for (auto& q : queries) {
            long long type = q[0], day = q[1], cap = q[2];
            long long minCandy = type == 0 ? 0 : cnt[type - 1];

            long long maxCandy = cnt[type] - 1;
            if (maxCandy < day) {
                result.push_back(false);
                continue;
            }
            if (minCandy >= (day + 1) * cap) {
                result.push_back(false);
                continue;
            }
            result.push_back(true);
        }
        return result;
    }
};
