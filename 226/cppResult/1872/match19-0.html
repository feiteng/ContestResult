class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> preCount(candiesCount.size()+1,0);
        for(int i=1;i<candiesCount.size()+1;++i){
            preCount[i]=candiesCount[i-1]+preCount[i-1];
        }
        vector<bool> ans(queries.size());
        for(int i=0;i<queries.size();++i){
            if((queries[i][1]+1)*(long long)queries[i][2]-1>=preCount[queries[i][0]]&&(queries[i][1]+1)<=preCount[queries[i][0]+1])
                ans[i]=true;
            else
                ans[i]=false;
        }
        return ans;
    }
};
