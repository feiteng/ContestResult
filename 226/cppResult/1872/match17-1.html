class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size(); 
        vector<long long> cum(n); 
        cum[0] = candiesCount[0];
        for (int i = 1; i < n; i++) cum[i] = cum[i - 1] + candiesCount[i]; 
        vector<bool> ans; 
        for (int i = 0; i < queries.size(); i++)
        {
            bool flag = true; 

            long long fav = queries[i][0];
            long long day = queries[i][1];
            long long cap = queries[i][2];
            long long before = (fav) ? cum[fav - 1] : 0;
            if (before >= (day + 1) * cap) flag = false; 
            if (cum[fav] <= day) flag = false; 
            ans.push_back(flag); 
        }
        return ans; 
    }
};
