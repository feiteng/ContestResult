class Solution {
    
    using ll=long long;
    public:
    vector<bool> canEat(vector<int>& cand, vector<vector<int>>& queries) {
        
        int len=cand.size();
        vector<bool> ans;
        ll presum[len+1];
        presum[0]=0;
        for(int i=0;i<len;i++)
            presum[i+1]=presum[i]+(ll)cand[i];
        
        for(auto i:queries)
        {
            ll tot=presum[i[0]+1];
            ll pre_t=presum[i[0]];
            
            if(i[1]>=tot || (ll)(i[1]+1)*(ll)i[2]<=(ll)pre_t)
                ans.emplace_back(false);
            else
                ans.emplace_back(true);
                
        }
        return ans;
    }
};
