class Solution {
public:
    void test(){
cout<<"test";
}

    vector<bool> canEat(vector<int>& cos, vector<vector<int>>& qs) {
        test();
        typedef long long ll;
        vector<ll> cs(cos.size(),0);
        for(int i=0;i<cos.size();i++){
            cs[i]=cos[i];
        }
        
        for(int i=1;i<cs.size();i++){
            cs[i]+=cs[i-1];
        }
        vector<bool> ans;
        for(auto a:qs){
            ll bef=(a[0]==0?0:cs[a[0]-1]);
            ll bef2=cs[a[0]];
             if (a[1] + 1 <= bef2 && 1LL * (a[1]+1) * a[2] > bef) {
                ans.push_back(true);
            }else{
                ans.push_back(false);
            }
        }
        return ans;

    }
};
