class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector<long long> sum(n);
        sum[0] = candiesCount[0];
        vector<bool> ans;
        for (int i = 1; i < n; i++) sum[i] = sum[i - 1] + candiesCount[i];
        for (const auto &vec : queries) {

            long long type = vec[0];
            long long day = vec[1];
            long long maxx = vec[2];
            long long pre = type == 0 ? 0 : sum[type - 1];
            if ((day + 1) * maxx > pre && day < sum[type]) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
