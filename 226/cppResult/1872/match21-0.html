class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector<long long> prefix(n+1);
        for(int i =1; i<= n; i++) prefix[i] = prefix[i-1] + candiesCount[i-1];
        vector<bool> res;
        for(auto temp :queries){
            long long type = temp[0];
            long long day = temp[1];
            long long cap = temp[2];
            long long MM = day * cap + cap;
            int mm = day;
            if(prefix[type] >= MM || prefix[type+1] <= mm){
                res.push_back(false);
                continue;
            }
            res.push_back(true);
        }
        return res;
    }
};
