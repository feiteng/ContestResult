class Solution {
public:
    vector<bool> canEat(vector<int>& A, vector<vector<int>>& Q) {
        int n = A.size();
        vector<long long> sum(n, 0);
        sum[0] = A[0];
        for (int i = 1; i < n; i++) {
            sum[i] = sum[i - 1] + A[i];
        }
        vector<bool> ans(0);
        for (auto& q : Q) {
            long long kind = q[0], day = q[1] + 1, limit = q[2];
            long long l = day, h = day * limit;
            long long kl = 0, kh = sum[kind];
            if (kind != 0) kl = sum[kind - 1] + 1;
            if (A[kind] == 0 || kh < l || h < kl) ans.emplace_back(false);
            else ans.emplace_back(true);
            // cout << l << ", " << h << " | " << kl << ", " << kh << " | " << ans.back() << endl;
        }
        return ans;
    }
};
