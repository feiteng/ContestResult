class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> sumbefore(candiesCount.size());
        for(int i=1;i<sumbefore.size();++i)
        {
            sumbefore[i] = sumbefore[i-1] + candiesCount[i-1];
        }
        
        vector<bool> ret;
        for(auto& q : queries)
        {
            long long cap = q[2];
            long long cand = q[0];
            long long day = q[1];
            long long low = sumbefore[cand] - cap +1;
            long long high = sumbefore[cand] + candiesCount[cand] - 1;
            if(day>high || cap*day<low)
                ret.push_back(false);
            else
                ret.push_back(true);
        }
        return ret;
    }
};
