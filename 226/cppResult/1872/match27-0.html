class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int n = candiesCount.size();
        vector<long long> sums(n+1);
        for (int i = 0; i < n; ++i) {
            sums[i+1] = sums[i] + candiesCount[i];
        }
        
        vector<bool> res;
        // sort (queries.begin(), queries.end(), [](vector<int>& a, vector<int>& b){});
        for (auto& q : queries) {
            long long type = q[0], day = q[1], cap = q[2];
            long long prev = sums[type];
            long long end = sums[type + 1];
            long long eatCap = (day+1) * cap;
            if (eatCap <= prev) { // 吃太慢
                res.push_back(false);
                continue; 
            }
            if (day >= end) { // 吃太快
                res.push_back(false);
                continue; 
            }

            res.push_back(true);
        }
        return res;
    }
};
