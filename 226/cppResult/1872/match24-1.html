class Solution {
public:
    
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int N = candiesCount.size();
        vector<long long> prefix_sum(N, 0);
        for (int i = 1; i < N; ++i) {
            prefix_sum[i] = prefix_sum[i - 1] + candiesCount[i - 1];
        }
        vector<bool> ans;
        for (auto& Q : queries) {
            long long type = Q[0], day = Q[1], cap = Q[2];

            if (prefix_sum[type] + candiesCount[type] - 1 < day) {
                ans.push_back(false);
            } else if (cap * (day + 1) <= prefix_sum[type]) {
                ans.push_back(false);
            } else {
                ans.push_back(true);
            }
        }
        return ans;
    }
};
