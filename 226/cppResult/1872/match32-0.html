class Solution {
public:
    vector<bool> canEat(vector<int>& c, vector<vector<int>>& q) {
        vector<long long> sum(c.size()+1,0);
        for(int i=1;i<sum.size();i++){
            sum[i]+=sum[i-1]+c[i-1];
        }
        vector<bool> ans(q.size());
        for(int i=0;i<q.size();i++){
            long long late=sum[q[i][0]];
            long long ear=sum[q[i][0]]/q[i][2];
            if(q[i][1]>=ear&&q[i][1]<late+c[q[i][0]])
                ans[i]=true;
            else
                ans[i]=false;
        }
        return ans;
    }
};
