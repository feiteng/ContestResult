class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<bool> ans;
        vector<long long> prefixSum(candiesCount.size(), 0);
        prefixSum[0] = candiesCount[0];
        for (int i = 1; i < candiesCount.size(); i++) {
            prefixSum[i] = prefixSum[i - 1] + candiesCount[i];
        }
        for (auto &qry: queries) {
            bool cond1 = (prefixSum[qry[0]] >= qry[1] + 1);
            bool cond2 = (prefixSum[qry[0]] - candiesCount[qry[0]] < (long long)qry[2] * (qry[1] + 1) - 1);
            ans.push_back(cond1 && cond2);
        }
        return ans;
    }
};
