class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        long long pre[100005];
        pre[0]=0;
        for(int i = 0;i<candiesCount.size();i++){
            pre[i+1]=pre[i]+candiesCount[i];
        }
        vector<bool> ans;
        for(auto it:queries){
            if(pre[it[0]]+1>(it[1]+1)*1ll*it[2]||pre[it[0]+1]<(it[1]+1))ans.push_back(false);
            else ans.push_back(true);
        }
        return ans;
    }
};
