int self_search(vector<long long>& A, long long value) {
    if (A.size() == 0) return 1;
    else if (A.back() < value) return A.size();
    
    int start = 0, end = A.size() - 1;
    while (start < end) {
        int mid = (start + end) / 2;
        if (A[mid] > value) end = mid;
        else if (A[mid] < value) start = mid + 1;
        else return mid;
    }
    
    if (A[start] >= value) return start;
    else return end;
}

class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int N = candiesCount.size();
        
        vector<long long> S;
        for (int i = 0; i < N; ++ i) {
            if (i == 0) S.push_back(candiesCount[0]);
            else S.push_back(S[i-1] + candiesCount[i]);
            cout << S[i] << endl;
        }
        
        vector<bool> res;
        for (auto& query: queries) {
            long long type = query[0], day = query[1], cap = query[2];
            long long min_eat = day + 1;
            long long max_eat = (day + 1) * cap;
            
            int min_type = self_search(S, min_eat);
            int max_type = self_search(S, max_eat);
            res.push_back(min_type <= type && type <= max_type);
            //cout << min_eat << " " << max_eat << " " << min_type << " " << max_type << endl;
        }
        
        return res;
    }
};
