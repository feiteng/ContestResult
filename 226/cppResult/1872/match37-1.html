class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> psum(1, 0);
        long long sum = 0;
        for (int i : candiesCount) {
            sum += i;
            psum.push_back(sum);
        }
        vector<bool> ans;
        for (vector<int> q : queries) {
            long long emax = q[2] * 1ll * (q[1] + 1);
            long long emin = q[1] + 1;
            if (emax > psum[q[0]] && emin <= psum[q[0] + 1]) {
                ans.push_back(true);
            } 
            else {
                ans.push_back(false);
            }
        }
        return ans;
        
    }
};
