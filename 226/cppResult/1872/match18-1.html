typedef long long LL;
class Solution {
public:
    vector<bool> canEat(vector<int>& c, vector<vector<int>>& queries) {
        int n=c.size();
        int m=queries.size();
        vector<bool> res(m);
        vector<LL> s(n+1,0);
        for(int i=1;i<=n;++i)
        {
            s[i]=c[i-1]+s[i-1];
        }
        // for(int i=0;i<=n;++i)
        //     cout<<i<<' '<<s[i]<<endl;
        for(int i=0;i<m;++i)
        {
            LL a=queries[i][0],b=queries[i][1],c=queries[i][2];
            // cout<<a<<' '<<b<<' '<<c<<endl;
            res[i]=((1+b)*c>s[a] && b+1<=s[a+1]);
            // cout<<res[i]<<endl;
        }
        return res;
    }
};
