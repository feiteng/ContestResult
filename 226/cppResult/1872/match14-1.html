class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<bool> ans;
        int n = candiesCount.size();
        vector<long long> pre(n+1);
        pre[0] = 0;
        for (int i = 0; i < n; ++i) {
            pre[i+1] = pre[i] + candiesCount[i];
        }
        
        for (auto &x : queries) {
            long long day = x[1], type = x[0], cap = x[2];
            //long long bfr = pre[type+1];
            long long rb = cap*(day+1);
            ans.push_back(day+1 <= pre[type+1] && rb > pre[type]);
        }
        
        return ans;
    }
};
