class Solution {
public:
    vector<bool> canEat(vector<int>& c, vector<vector<int>>& q) {
        vector<bool> res;
        //前i类糖果,i从1开始， 题目是从0开始
        vector<long long> f(c.size()+1, 0);
    
        for (int i = 1; i <= c.size(); i++) {
            f[i] = f[i - 1] + c[i-1];
        }

        for (int i = 0; i < q.size(); i++) {
            long long  fday = q[i][1], ftype = q[i][0]+1, fcap = q[i][2];
            //f[ftype], f[ftype-1]
            if (fcap * (fday + 1) > f[ftype - 1] && f[ftype] >= (fday + 1)) {
                res.push_back(true);
            }
            else res.push_back(false);
        }
        return res;
    }
};
