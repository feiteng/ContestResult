class Solution {
public:
    vector<bool> canEat(vector<int>& cc, vector<vector<int>>& qs) {
        vector<long long> presum(cc.size() + 1, 0);
        for(int i = 0; i < cc.size(); i++) {
            presum[i + 1] = 1ll * cc[i] + presum[i];
        }
        
        vector<bool> res;
        for(int i = 0; i < qs.size(); i++) {
            res.push_back(!(presum[qs[i][0] + 1] < qs[i][1] + 1 || 1ll * (1 + qs[i][1]) * qs[i][2] <= presum[qs[i][0]]));
        }
        return res;
    }
};
