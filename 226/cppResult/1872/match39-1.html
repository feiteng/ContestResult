class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<bool> ans;
        int n = candiesCount.size();
        vector<long long> sum;
        sum.push_back(candiesCount[0]);
        for(int i=1;i<n;i++){
            sum.push_back(sum[i-1]+candiesCount[i]);

        }
        for(auto item:queries){
            long long type=item[0],day=item[1],cnt=item[2];
            long long high=(day+1)*cnt;
            long long total=type==0?0:sum[type-1];
            if(total>=high||sum[type]<=day)
                ans.push_back(false);
            else
                ans.push_back(true);
        }
        return ans;
    }
};
