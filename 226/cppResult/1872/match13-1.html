using LL=long long;

class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        int T = candiesCount.size();
        vector<LL> vNum(T+1);
        for(int i=0;i<T;i++){
            vNum[i+1] = vNum[i] + candiesCount[i];
        }
        vector<bool> vAns;
        for(auto q:queries){

            int t = q[0];
            LL day = q[1];
            LL num = q[2];
            LL x = vNum[t];
            if((day+1)*num>x){
                vAns.push_back((day+1)<=vNum[t+1]);
            }else{
                vAns.push_back(false);
            }
        }
        
        return vAns;
    }
};
