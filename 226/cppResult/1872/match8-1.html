class Solution {
public:
    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long long> v;
        long long sum = 0;
        v.push_back(0);
        for (long long i: candiesCount) {
            sum += i;
            v.push_back(sum);
        }
        vector<bool> ans;
        for (auto& i: queries) {

            int t = i[0];
            long long d = i[1];
            long long c = i[2];
            long long a = v[t];
            long long b = v[t+1] - 1;
            if (d > b || c * (d+1) < a+1) ans.push_back(false);
            else ans.push_back(true);
        }
        return ans;
    }
};
