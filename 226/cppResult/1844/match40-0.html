class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        int ans = 0;
        std::unordered_map<int, int> dict;
        for (int num = lowLimit; num <= highLimit; num++) {
            int sum = 0, val = num;
            while (val) {
                sum += val % 10;
                val /= 10;
            }
            dict[sum] = dict[sum] + 1;
            ans = std::max(ans, dict[sum]);
        }
        return ans;
    }
};
