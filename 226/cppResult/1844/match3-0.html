class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> memo;
        for (int i = lowLimit; i <= highLimit; i++) {
            int t = i, idx = 0;
            while (t != 0) {
                idx += (t % 10);
                t /= 10;
            }
            memo[idx] += 1;
        }
        int maxv = 0;
        for (auto &[p, v]: memo) {
            maxv = max(maxv, v);
        }
        return maxv;
    }
};
