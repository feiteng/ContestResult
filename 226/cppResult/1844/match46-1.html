class Solution {
public:
    int countBalls(int low, int high) {
        unordered_map<int, int> mp;
        for (int i = low; i <= high; ++i) {
            int sum = 0, t = i;

            while (t) {
                sum += t % 10;
                t /= 10;
            }
            mp[sum]++;
        } 
        int ans = 0;
        for (auto [k, v]  : mp) {
            ans = max(ans, v);
        }
        return ans;
    }
};
