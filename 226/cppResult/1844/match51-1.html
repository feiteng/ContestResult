class Solution {
public:
    int f(int m) {
        int res = 0;
        while(m != 0) {
            res += m % 10;
            m /= 10;
        }
        return res;
    }
    int countBalls(int lowLimit, int highLimit) {
        int a[100] = {0};

        for(int i = lowLimit; i <= highLimit; i++) {
            int temp = f(i);
            a[temp]++;
        }
        int res = 0;
        for(int i = 0; i < 100; i++)
            res = max(res, a[i]);
        return res;
    }
};
