class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        map<int,int> mp;
        for (int i = lowLimit; i <= highLimit; ++i) {
            int box = 0, k = i;
            while (k) {
                box += (k%10);
                k/=10;
            }
            mp[box]++;
        }
        int ans = 0;
        for (auto &x : mp) {
            ans = max(ans, x.second);
        }
        return ans;
    }
};
