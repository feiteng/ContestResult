class Solution {
public:
    int solver(int x){
        int res=0;
        while(x){
            res+=x%10;
            x/=10;
        }
        return res;
    }
    int countBalls(int l, int h) {
        map<int,int>mp;
        int res=0;;
        for(int i=l;i<=h;i++){
            int t=solver(i);
            mp[t]++;
            res=max(res,mp[t]);
        }
        return res;
    }
};
