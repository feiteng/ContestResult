class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int,int> m;
        for (int i = lowLimit; i <= highLimit; ++i) {
            int t = 0;
            int ii = i;
            while (ii) {
                t += ii % 10;
                ii /= 10;
            }
            ++m[t];
        }
        int ans = 0;
        for (auto& i: m) {
            ans = max(ans, i.second);
        }
        return ans;
    }
};
