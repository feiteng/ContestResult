class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> cnt;
        int maxCnt = 0;
        for (int num = lowLimit; num <= highLimit; ++num) {
            int sum = 0;
            int temp = num;
            while (temp > 0) {
                sum += temp % 10;
                temp /= 10;
            }
            maxCnt = max(maxCnt, ++cnt[sum]);
        }
        return maxCnt;
    }
};
