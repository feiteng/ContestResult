class Solution {
public:
    int sumd(int n)
    {
        int ans = 0; 
        while (n)
        {
            ans += n % 10; 
            n /= 10; 
        }
        return ans; 
    }
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> mp; 
        for (int i = lowLimit; i <= highLimit; i++)
        {
            mp[sumd(i)]++;
        }
        int ans = 0; 
        for (auto it : mp)
        {
            ans = max(ans, it.second); 
        }
        return ans; 
    }
};
