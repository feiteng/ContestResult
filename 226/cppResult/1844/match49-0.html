class Solution {
public:
    int countBalls(int l, int r) {
        map <int, int> cnt;
        for (int i = l; i <= r; i++) {
            int now = i;
            int s = 0;
            while (now > 0) {
                s += now % 10;
                now /= 10;
            }
            cnt[s]++;
        }
        int ans = 0;
        for (auto it : cnt) {
            ans = max(ans, it.second);
        }
        return ans;
    }
};
