class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int,int>cnt;
        int ans = 0;
        for(int i=lowLimit;i<=highLimit;i++){
            int t = i;
            int num = 0;
            while(t){
                num += t%10;
                t/=10;
            }
            cnt[num]++;
            ans = max(ans,cnt[num]);
        }
        return ans;
    }
};
