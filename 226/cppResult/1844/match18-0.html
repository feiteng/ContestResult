class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        
              unordered_map<int,int> mp;
              int res=0;
        
        
        for(int i=lowLimit;i<=highLimit;i++){
              
               int temp=i;
               int sum=0;
               while(temp){
                     sum+=temp%10;
                   temp/=10;
               }
            mp[sum]++;
            res=max(res,mp[sum]);
            
            
        }
        return res;
        
    }
};
