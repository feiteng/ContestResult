class Solution {
public:
    int countBalls(int l, int h) {
        vector<int> box(1000005,0);
        for (int i = l; i <= h; i++) {
            auto s = to_string(i);
            int index = 0;
            for (auto& ch:s) {
                index += ch - '0';
            }
            box[index]++;
        }
        return *max_element(box.begin(),box.end());
    }
};
