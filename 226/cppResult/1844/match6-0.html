class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int,int> mp;
        for(int i=lowLimit; i<= highLimit; i++){
            int x = i, sum = 0;
            while(x){
                sum += x % 10;
                x /= 10;
            }
            mp[sum]++;
        }
        int Max = 0;
        for(auto &t : mp){
            Max = max(Max, t.second);
        }
        return Max;
    }
};
