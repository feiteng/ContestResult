class Solution {
public:
    int countBalls(int a, int b) {
        unordered_map<int, int> mp;
        int ans = 0;
        for (int i = a; i <= b; ++i) {
            int c = i;
            int n = 0;
            while (c) {
                n += c % 10;
                c /= 10;
            }
            mp[n]++;
            ans = max(ans, mp[n]);
//            printf("%d %d\n", i, n);
        }
        return ans;
    }
};
