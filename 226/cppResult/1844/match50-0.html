class Solution {
public:
    
    int count(int x) {
        int res = 0;
        while (x) {
            res += (x % 10);
            x /= 10;
        }
        return res;
    }
    
    int countBalls(int l, int h) {
        unordered_map<int, int> hash; 
        for (int i = l; i <= h; i ++ ) {
            int x = count(i);
            hash[x] ++ ;
        }
        
        int res = -1;
        for (auto &[k, v] : hash) {
            res = max(res, v);
        }
        return res;
    }
};
