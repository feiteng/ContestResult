class Solution {
 public:
  int countBalls(int lowLimit, int highLimit) {
    unordered_map<int, int> balls;
    int res = 0;
    for (int i = lowLimit; i <= highLimit; i++) {
      int box = 0, ball = i;
      while (ball > 0) {
        box += ball % 10;
        ball /= 10;
      }
      res = max(res, ++balls[box]);
    }
    return res;
  }
};
