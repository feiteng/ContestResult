class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        map<int, int> cnt;
        int ans = 0;
        for (int n = lowLimit; n <= highLimit; n++) {
            int num = n;
            int sum = 0;
            while (num) {
                sum += num % 10;
                num /= 10;
            }
            cnt[sum] ++;
            ans = max(ans, cnt[sum]);
        }
        return ans;
    }
};
