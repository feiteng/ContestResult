class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> cnt;
        int res = 0;
        for (int i = lowLimit; i <= highLimit; i++) {
            int sum = 0, n = i;
            while (n) {
                sum += n % 10;
                n /= 10;
            }
            cnt[sum]++;
            res = max(res, cnt[sum]);
        }
        return res;
    }
};
