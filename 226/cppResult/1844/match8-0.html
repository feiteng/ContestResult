class Solution {
public:
  int countBalls(int lowLimit, int highLimit) {
    unordered_map<int, int> ctn;
    int rtn = 0;
    for (int x = lowLimit ; x  <= highLimit; ++x) {
      int sum = 0;
      int t = x;
      while (t > 0) {
        sum += t %10;
        t /= 10;
      }
      ctn[sum]++;
      rtn = max(rtn, ctn[sum]);
    }
    return rtn;
  }
};
