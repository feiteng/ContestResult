class Solution {
public:
    int countBalls(int l, int h) {
        vector<int> c(100);
        for(int i = l; i <= h; i++) {

            string s = to_string(i);
            int cs = 0;
            for(auto &j : s) {
                cs += j - '0';
            }
            // cout << cs << endl;
            c[cs]++;
        }
        return *max_element(c.begin(), c.end());
    }
};
