class Solution {
public:
    int countBalls(int l, int r) {
        map<int,int> mp;
        int ans=0;
        for(int i=l;i<=r;++i){
            int num = i;
            int sum=0;
            while(num){
                sum = sum + (num%10);
                num = num/10;
            }
            mp[sum]++;
            ans = max(ans,mp[sum]);
        }
        return ans;
    }
};
