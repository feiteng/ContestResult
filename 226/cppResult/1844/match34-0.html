class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
		map<int,int> M;
		int ma=0;
		for(int i=lowLimit;i<=highLimit;i++) {
			int x=i,y=0;
			while(x) {
				y+=x%10;x/=10;
			}
			M[y]++;
			ma=max(ma,M[y]);
		}
		return ma;
        
    }
};
