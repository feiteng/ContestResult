class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        int res = 0;
        std::map<int, int> m;
        for (int i = lowLimit; i <= highLimit; ++i) {
            int b = 0;

            int j = i;
            while(j) {
                b += j % 10;
                j /= 10;
            }
            m[b] += 1;
            res = std::max(res, m[b]);
        }
        return res;
    }
};
