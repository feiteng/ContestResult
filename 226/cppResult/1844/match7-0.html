class Solution {
public:
    int calc(int x) {
        int ans = 0;
        while (x) {
            ans += x % 10;
            x /= 10;
        }
        return ans;
    }
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> mp;
        for (int i = lowLimit; i <= highLimit; i++) {
            mp[calc(i)]++;
        }
        int ans = 0;
        for (auto [key, val] : mp) {
            ans = max(ans, val);
        }
        return ans;
    }
};
