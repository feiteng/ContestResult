class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
		map<int, int> mp;
		for(int i = lowLimit; i <= highLimit; ++i) {
			int k = i, res = 0;
			while(k) {
				res += k % 10;
				k /= 10;
			}
			mp[res]++;
		}
		int ans = 0;
		for(auto it: mp) {
			ans = max(ans, it.second);
		}
		return ans;
    }
};
