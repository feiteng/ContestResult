class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int,int> um;
        for(int i=lowLimit;i<=highLimit;i++){
            int t=i;
            int ans=0;
            while(t){
                ans+=t%10;
                t/=10;
            }
            um[ans]++;
        }
        int ans=0;
        for(auto item:um){
            ans=max(ans,item.second);
        }
        return ans;
    }
};
