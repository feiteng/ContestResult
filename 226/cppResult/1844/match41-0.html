class Solution {
public:
    int countBalls(int low, int high) {
        map<int,int> mp;
        for(int i=low;i<=high;i++){
            int x=i;
            int sum=0;
            while(x>0){
                sum+=x%10;
                x=x/10;
            }
            mp[sum]++;
        }
        int mx=-1;
        for(auto i:mp){
            mx=max(mx,i.second);
        }
        return mx;
    }
};
