class Solution {
public:
    int f(int n) {
        int sum = 0;
        while(n) {
            sum += (n%10);
            n = n/10;
        }
        
        return sum;
    }
    
    int countBalls(int lowLimit, int highLimit) {
        vector<int> v (50, 0);
        int res = 0;
        for (int i = lowLimit; i <= highLimit; i++) {
            int x = f(i);
            v[x]++;
            res = max(res, v[x]);
        }
        
        return res;
    }
};
