class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        map<int, int> m;
        int ans = 0;
        for (int i = lowLimit; i <= highLimit; i++) {
            int x = i;
            int s = 0;
            while (x) {
                s += x%10;
                x /= 10;
            }
            m[s]++;
            ans = max(ans, m[s]);
        }
        
        return ans;
    }
};
