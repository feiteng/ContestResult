class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> m;
        for (int i = lowLimit; i <= highLimit; i++) {
            int t = i;
            int cnt = 0;
            while (t > 0) {
                cnt+= t%10;
                t /=10 ;
            }
            m[cnt]++;
        }
        int res = 0;
        for (auto it: m) {
            res = max(res, it.second);
            // cout << it.first <<' ' << it.second << endl;
        }
        return res;
    }
};
