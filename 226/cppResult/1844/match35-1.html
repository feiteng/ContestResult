class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int,int> dp;
        int ans = 0,num, key;
        for(int i = lowLimit; i <= highLimit; ++i){

            num = i, key = 0;
            while(num > 0){
                key += num%10;
                num /= 10;
            }
            dp[key]++;
            ans = max(ans, dp[key]);
        }
        return ans;
    }
};
