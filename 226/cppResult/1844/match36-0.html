class Solution {
public:
    int countBalls(int lo, int hi) {
        unordered_map<int,int> m;
        int ret = 0;
        for(int i = lo ; i <= hi; i++) {
            int c = 0, x = i;
            while(x) {
                c += x % 10;
                x /= 10;
            }
            ret = max(ret, ++m[c]);
        }
        return ret;
    }
};
