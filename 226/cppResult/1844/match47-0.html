class Solution {
public:
    int countBalls(int x, int y) {
        unordered_map<int, int> a;
        for(int i = x ; i <= y; i++) {
            int b = i;
            int t = 0;
            while(b > 0) {
                t += b % 10;
                b /=10;
            }
            a[t]++;
        }
        int ret  =0;
        for(auto & p : a) {
            ret = max(ret, p.second);
        }
        return ret;
    }
};
