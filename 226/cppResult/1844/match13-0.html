class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
     
        
        unordered_map<int,int> name;
        int cnt = 0;
        for(int i = lowLimit ; i <= highLimit ; i++){
            
            int sum = 0;
            int v = i;
            while(v){
                sum += v % 10;
                v = v / 10;
            }
            
            name[sum]++;
            cnt = max(cnt , name[sum]);
        }
        return cnt;
    }
};
