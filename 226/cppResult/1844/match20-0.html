class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> map;
        int result = 0;
        for(int i = lowLimit;i<=highLimit;i++){
            int sum = 0;
            int n = i;
            while(n>0){
                sum += n%10;
                n/=10;
            }
            map[sum]++;
            result = max(map[sum], result);
        }
        return result;
    }
};
