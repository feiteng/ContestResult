class Solution {
public:
    bool checkPartitioning(string s) {
        // f(0, n)
        
        // -> consider prefix and suffix palindromic strings
        // check if the middle string is also palindromic
        
        // isPalindrom([0: i]) && isPalindrom[j: n] && isPalindrom[i: j]
        
        // can string s be split in to 2 palindroms?   n * n
        // for (int i = k; i < n; i++)
        //      check [k: i]  && [i: n]
        
        // then check [0:k]
        
        // we need an array to quickly check if [x: y] is palindrom
        
        // f(x, y) == f(x + 1, y - 1) && s[x] == s[y]  || x == y
        
        int n = s.size();
        bool dp[n][n];
        for (int i = 0; i < n; ++i) {
            dp[i][i] = true;
        }
        
        for (int i = 0; i < n - 1; ++i) {
            dp[i][i + 1] = s[i] == s[i + 1];
        }
        
        for (int i = 1; i < n; ++i) {
            for (int j = i - 1; j >= 0; --j) {
                //cout << j + 1 << " " << i - 1 << " = " << dp[j + 1][i - 1] << endl;
                dp[j][i] = s[j] == s[i] && (j + 1 >= i - 1 || dp[j + 1][i - 1]);
            }
        }
        
        for (int k = 1; k < n; ++k) {
            if (!dp[0][k - 1]) continue;

            for (int i = k; i + 1< n; ++i) {
                if (dp[k][i] && dp[i + 1][n-1])
                    return true;
            }
        }
        
        return false;
    }
};
