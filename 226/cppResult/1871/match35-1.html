class Solution {
public:
    bool checkPartitioning(string s) {
        int size = s.size();
        vector<vector<bool>> isPa(size, vector<bool>(size, false));
        for (int i = size - 1; i >= 0; i--) {
            for (int j = i; j < size; j++) {
                if (s[i] == s[j] && (j - i <= 1 || isPa[i + 1][j - 1])) {
                    isPa[i][j] = true;
                }

            }
        }
        
        
        for(int i = 0; i <= size - 3; i++){
            for(int j = i + 1; j <= size - 2; j++){
                if(isPa[0][i] && isPa[i + 1][j] && isPa[j + 1][size - 1])return true;
            }
        }
        return false;
    }
};
