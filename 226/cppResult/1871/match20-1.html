class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<int>> c(n, vector<int>(n));
        for(int i = 0; i < n; i++) {
            c[i][i] = 1;
        }
        for(int k = 1; k < n - 1; k++) {
            for(int i = 0; i + k < n; i++) {
                int j = i + k;
                if(k == 1) {
                    if(s[i] == s[j]) {
                        c[i][j] = 1;
                    }
                    continue;

                }
                if(s[i] == s[j] && c[i + 1][j - 1]) {
                    c[i][j] = 1;
                }
            }
        }
        for(int i = 0; i < n - 2; i++) {
            for(int j = i + 1; j < n - 1; j++) {
                if(c[0][i] && c[i + 1][j] && c[j + 1][n - 1]) {
                    return 1;
                }
            }
        }
        return 0;
    }
};
