class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> f(n, vector<bool>(n, true));
        
        for(int len = 2; len <= n; len ++) {
            for(int i = 0; i + len - 1 < n; i ++) {
                int j = i + len - 1;
                f[i][j] = f[i + 1][j - 1] && s[i] == s[j];            
            }
        }
        
        for(int i = 2; i < n; i ++) {
            for(int j = 1; j < i; j ++) {
                if(f[0][j - 1] && f[j][i - 1] && f[i][n - 1])
                    return true;
            }
        }
        return false;
    }
};
