class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> isPar(n, vector<bool>(n, false));
        for (int i = 0; i < n; i++) {
            isPar[i][i] = true;
        }
        for (int i = 0; i < n - 1; i++) {
            isPar[i][i + 1] = s[i] == s[i + 1] ? true : false;
        }
        for (int len = 3; len <= n; len++) {
            for (int i = 0; i + len - 1 < n; i++) {
                int j = i + len - 1;
                if (isPar[i + 1][j - 1] == true && s[i] == s[j]) {isPar[i][j] = true;}
                else {isPar[i][j] = false;}
            }
        }
        
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 1; j < n - 1; j++) {
                if (isPar[0][i] == true && isPar[i + 1][j] == true && isPar[j + 1][n - 1] == true) {return true;}
            }
        }
        return false;
    }
};
