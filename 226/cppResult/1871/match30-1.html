class Solution {
    bool palin[2010][2010];
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j < n; j++) {
                if (j == i) {
                    palin[i][j] = true;
                } else if (j == i + 1) {
                    palin[i][j] = (s[i] == s[j]);

                } else {
                    palin[i][j] = ((s[i] == s[j]) && palin[i + 1][j - 1]);
                }
            }
        }
        for (int i = 1; i <= n - 2; i++) {
            for (int j = i; j <= n - 2; j++) {
                if (palin[0][i - 1] && palin[i][j] && palin[j + 1][n - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
};
