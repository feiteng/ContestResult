class Solution {
public:
    
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> dp(n , vector<bool>(n));
        for(int len = 1 ; len <= s.size(); len++){
            for(int i = n-1-len+1; i>=0; i--){
                if(len ==1){
                    dp[i][i+len-1] = true;
                }
                else if(len ==2|| len ==3){
                    if(s[i] == s[i+len-1]){
                        dp[i][i+len-1] = true;
                    }
                    else dp[i][i+len-1] = false;
                }else{
                    if(s[i] == s[i+len-1] && dp[i+1][i+len-2]){
                        dp[i][i+len-1] = true;
                    }
                    else{
                        dp[i][i+len-1] = false;
                    }     
                }
            }
        }
        
        
        
        for(int i = 0; i < s.size()-2; i++){
            for(int j = i+1; j < s.size()-1; j++){
                if(dp[0][i] &&dp[i+1][j]&& dp[j+1][n-1]) return true;
            }
        }
        return false;
    }
