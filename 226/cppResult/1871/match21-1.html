class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        
        vector<vector<bool>> isPalin(n, vector<bool>(n, false));
        
        for (int i = 0; i < n; i++)
        {
            isPalin[i][i] = true;
        }
        
        for (int len = 2; len <= n; len++)
        {
            //cout << "len " << len << endl;
            for (int i = 0; i <= n - len; i++)
            {
                int j = i + len - 1;
                //cout << i << " " << j << endl;
                if (s[i] == s[j])

                {
                    if (i + 1 > j - 1)
                    {
                        isPalin[i][j] = true;
                    }
                    else
                    {
                        isPalin[i][j] = isPalin[i + 1][j - 1];
                    }
                    // if (isPalin[i][j])
                    // {
                    //     cout << "palin " << i << " " << j << endl;
                    // }
                }
            }
        }
        
        for (int i = 0; i < n - 2; i++)
        {
            for (int j = i + 2; j < n; j++)
            {
                if (isPalin[0][i] && isPalin[i + 1][j - 1] && isPalin[j][n - 1])
                {
                    return true;
                }
            }
        }
        
        return false;
    }
};
