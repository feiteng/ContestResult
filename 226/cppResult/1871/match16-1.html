class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for(int j = 0; j < n; j++) {
            dp[j][j] = 1;
        }
        for(int i = 0; i < n - 1; i++) {
            if(s[i] == s[i+1]) dp[i][i+1] = 1;
        }
        for(int k = 2; k <= n; k++) {
            
            for(int i = 0; i <= n - k; i++) {
                int j = i + k - 1;
                if(s[i] == s[j]) {
                    dp[i][j] |= dp[i+1][j-1];
                }

            }
        }
        for(int i = 1; i < n-1; i++) {
            for(int j = i; j < n-1; j++) {
                if(dp[0][i-1] == 1 and dp[i][j] == 1 and dp[j+1][n-1] == 1) return 1;
            }
        }
        return false;
    }
};
