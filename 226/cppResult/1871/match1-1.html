class Solution {
public:
    bool checkPartitioning(string s) {
        int N = s.length();
        vector<vector<bool>> is_palindrome(N, vector<bool>(N, false));
        for (int len = 1; len <= N; ++len) {
            for (int L = 0; L + len <= N; ++L) {
                int R = L + len - 1;
                if (L + 1 <= R - 1) {
                    is_palindrome[L][R] = (is_palindrome[L + 1][R - 1] && s[L] == s[R]);

                } else {
                    is_palindrome[L][R] = (s[L] == s[R]);
                }
            }
        }
        for (int i = 0; i < N; ++i) {
            for (int j = i + 1; j + 1 < N; ++j) {
                // 0 ~ i, i + 1 ~ j, j + 1 ~ N - 1
                if (is_palindrome[0][i] && is_palindrome[i + 1][j] && is_palindrome[j + 1][N - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
};
