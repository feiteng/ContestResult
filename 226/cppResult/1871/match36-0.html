const int N = 2001;
bool dp[N][N] = {false};
class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                dp[i][j] = false;
            }
        }
        for(int i = 0; i < n; i++) {
            dp[i][i] = true;
            if(i+1<n && s[i]==s[i+1]) dp[i][i+1] = true;
        }
        for(int l = 3; l <= n; l++) {
            for(int i = 0; i+l <= n; i++) {
                int j = i+l-1;
                if(dp[i+1][j-1] && s[i]==s[j]) {
                    dp[i][j] = true;
                }
            }
        }
        
        for(int i = 1; i < n; i++) {
            for(int j = i; j+1 < n; j++) {
                if(dp[i][j] && dp[0][i-1] && dp[j+1][n-1]) return true;
            }
        }
        return false;
    }
};
