class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        vector<vector<bool>> ok(n, vector<bool>(n));
        for (int i = 0; i < n; i++)
            ok[i][i] = true;
        for (int i = 0; i + 1 < n; i++) {
            if (s[i] == s[i + 1])
                ok[i][i + 1] = true;
        }
        for (int k = 3; k <= n; k++) {
            for (int i = 0; i < n; i++) {
                int j = i + k - 1;
                if (j >= n)
                    break;
                if (s[i] == s[j] && ok[i + 1][j - 1])
                    ok[i][j] = true;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j + 1 < n; j++) {
                if (ok[0][i] && ok[i + 1][j] && ok[j + 1][n - 1])
                    return true;
            }
        }
        return false;
    }
};
