class Solution {
public:
    bool checkPartitioning(string s) {
        int sz = s.size();
        vector<vector<bool>> dp(sz+5, vector<bool>(sz+5, true));
        for (int len = 2; len <= sz; ++len) {
            for (int i = 0; i + len <= sz; ++i) {
                if (s[i] != s[i+len-1] || !dp[i+1][i+len-1]) {
                    dp[i][i+len] = false;
                }
            }
        }
        
        for (int i = 1; i < sz; ++i) {
            for (int j = i+1; j < sz; ++j) {
                if (dp[0][i] && dp[i][j] && dp[j][sz]) {
                    return true;
                }
            }
        }
        
        return false;
    }
    
};
