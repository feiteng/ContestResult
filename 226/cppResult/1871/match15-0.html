class Solution {
public:
    bool checkPartitioning(string s) {
        int n=s.length();
        for(int i=0;i<n;++i) f[i][i]=true;
        for(int len=2; len<=n;++len) {
            for(int st=0; st+len-1<n;++st) {
                int ed=st+len-1;
                if(len==2)f[st][ed]=(s[st]==s[ed]);
                else f[st][ed]=(f[st+1][ed-1] && s[st]==s[ed]);
            }
        }
        for(int l=0;l<n-2;++l) {
            for(int r=n-1;r>=l+2;--r) {
                if(f[0][l] && f[l+1][r-1] && f[r][n-1]) return true;
            }
        }
        return false;
    }
private:
    bool f[2005][2005];
};
