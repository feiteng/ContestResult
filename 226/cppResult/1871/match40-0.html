class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<int>> v(n,vector<int>(n,0));
        for(int i=0;i<n;i++)
        {
            v[i][i] =1;
            
        }
        for(int i=0;i<n-1;i++)
        {
            if(s[i]==s[i+1])
            v[i][i+1]=1;
        }
        for(int size = 3;size<=n;size++)
        {
            for(int i=0;i<n-size+1;i++)
            {
                int j = i+size-1;
                if(v[i+1][j-1] && s[i]==s[j])
                    v[i][j] = 1;
                
            }
        }
        for(int i=1;i<n-1;i++)
        {
            for(int j=i;j<n-1;j++)
            {
                if(v[0][i-1] && v[i][j] && v[j+1][n-1])
                  return true;
            }
        }
        return false;
    }
};
