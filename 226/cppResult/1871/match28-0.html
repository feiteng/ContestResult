class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        //0 ~ i  i + 1 ~ j   j + 1 ~ n - 1
        vector<vector<bool>> f(n, vector<bool>(n));
        for(int i = n - 1; i >= 0; i--)
        {
            for(int j = i; j < n; j++)
            {
                if(i == j)
                {
                    f[i][j] = true;
                    continue;
                }
                if(s[i] != s[j])    f[i][j] = false;
                else    f[i][j] = (j - i <= 2 || f[i+1][j-1]);
            }
        }
        for(int i = 0; i < n - 2; i++)
        {
            for(int j = i + 1; j < n - 1; j++)
            {
                if(f[0][i] && f[i+1][j] && f[j+1][n-1])   return true;
            }
        }
        return false;
    }
};
