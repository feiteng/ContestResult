class Solution {
public:
    bool checkPartitioning(string s) {
        
        
        int N=s.size();
        
        vector<vector<bool>> dp(N, vector<bool>(N, false));
        
        
        for(int L=1;L<=N;++L)
        {
            
            for(int left=0;left<=N-L;++left)
            {
                int right=left+L-1;
                
                if(L==1)

                    dp[left][right]=true;
                else if(L==2)
                    dp[left][right]=s[left]==s[right];
                else
                    dp[left][right]=s[left]==s[right] && dp[left+1][right-1];
                    
            }
            
        }
        
        for(int i=0;i<=N-3;++i)
        {
            
            if(dp[0][i]==false)
                continue;
            
            

            for(int j=i+1;j<=N-2;++j)
            {
                if(dp[i+1][j] && dp[j+1][N-1])
                    return true;
                
            }
        }
        
        return false;
        
        
    }
};
