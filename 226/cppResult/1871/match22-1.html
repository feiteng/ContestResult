class Solution {
public:
    bool checkPartitioning(string s) {
        int n=s.size();
        
        vector<vector<bool>> is_palin(n,vector<bool>(n,false));
        
        for(int i=0;i<n;i++)
        {
            is_palin[i][i]=true;
        }
        
        for(int i=0;i<n-1;i++)
        {
            if(s[i]==s[i+1])    is_palin[i][i+1]=true;

        }
        
        for(int siz=3;siz<=n;siz++)
        {
            for(int srt=0;srt+siz-1<n;srt++)
            {
                int end=srt+siz-1;
                if(s[srt]==s[end] and is_palin[srt+1][end-1])

                {
                    is_palin[srt][end]=true;
                }
            }
        }
        
        
        for(int i=1;i<n-1;i++)
        {
            for(int j=i;j<n-1;j++)
            {
                if(is_palin[i][j] and is_palin[0][i-1] and is_palin[j+1][n-1])
                {
                    return true;
                }
            }
        }
        
        return false;
    }
};
