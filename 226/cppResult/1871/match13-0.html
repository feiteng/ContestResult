class Solution {
public:
    bool checkPartitioning(string s) {
        int len = s.size();
        vector<vector<bool>> dp(len, vector<bool>(len, false));
        for(int i = 0; i < len; ++i){
            dp[i][i] = true;
        }
        for(int i = 1; i < len; ++i){
            dp[i-1][i] = (s[i-1] == s[i]);
        }
        for(int i = len-1; i >= 0; --i){
            for(int j = i+2; j < len; ++j){
                dp[i][j] = (s[i]==s[j]) && dp[i+1][j-1];
            }
        }
        for(int i = 1; i < len; ++i){
            for(int j = len-2; j >= i; --j){
                if(dp[0][i-1] && dp[i][j] && dp[j+1][len-1])
                    return true;
            }
        }
        return false;
    }
};
