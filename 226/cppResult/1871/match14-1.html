class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> g(n, vector<bool>(n, false));
        for(int i = 0; i < n; i++) g[i][i] = true;
        for(int len = 2; len <= n; len++) {
            for(int i = 0; i + len - 1 < n; i++) {
                int j = i + len - 1;

                if(i + 1 == j && s[i] == s[j]) g[i][j] = true;
                if(s[i] == s[j] && g[i + 1][j - 1]) g[i][j] = true;
            }
        }
        // for(int i = 0; i < n; i++) {
        //     for(int j = 0; j < n; j++) {
        //         cout << g[i][j] << ' ';
        //     }
        //     cout << endl;
        // }
        for(int i = 0; i + 1 < n; i++) {
            for(int j = i + 1; j + 1 < n; j++) {
                if(g[0][i] && g[i + 1][j] && g[j + 1][n - 1]) return true;
            }
        }
        return false;
    }
};
