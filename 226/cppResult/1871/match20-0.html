class Solution {
public:
    bool checkPartitioning(string s) {
        int n  = s.size();
        // ispal[i][j] stores whether s from i until j is palindromic.
        vector<vector<int>> ispal(n, vector<int>(n, 0));
        for (int i = 0; i < n; ++i) {
            ispal[i][i] = 1;
        }
        for (int l = 2; l <= n; ++l) {
            for (int i = 0; i + l <= n; ++i) {
                int j = i + l - 1;
                if (ispal[i + 1][j - 1] == 1 && s[i] == s[j]) {
                    ispal[i][j] = 1;
                }
                if (j == i + 1 && s[i] == s[j]) {
                    ispal[i][j] = 1;
                }
            }
        }
        for (int i = 1; i < n; ++i) {
            for (int j = i; j < n - 1; ++j) {
                if (ispal[0][i-1] && ispal[i][j]&&ispal[j+1][n-1]){
                    return true;
                }
            }
        }
        return false;
    }
};
