class Solution {
 public:
  bool checkPartitioning(string s) {
    int n = s.size();
    vector<vector<bool>> dp(n, vector<bool> (n, false));
    for (int m = 0; m < n; m++) {
      int left = m, right = m;
      while (left >= 0 && right < n && s[left] == s[right]) {
        dp[left][right] = true;
        left--;
        right++;
      }
      left = m, right = m + 1;
      while (left >= 0 && right < n && s[left] == s[right]) {
        dp[left][right] = true;
        left--;
        right++;
      }
    }
    for (int i = 0; i < n; i++) {
      for (int j = i + 1; j < n - 1; j++) {
        if (dp[0][i] && dp[i + 1][j] && dp[j + 1][n - 1]) {
          return true;
        }
      }
    }
    return false;
  }
};
