class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> dp (n, vector<bool>(n, false));
        for(int i = 0 ; i < n ; ++i)
            dp[i][i] = 1;

        for(int l = 2 ; l <= n ; ++l){
            for(int i = 0 ; i < n - l +1 ; ++i){
                int j = i + l - 1;
                if(l == 2)
                    dp[i][j] = (s[i] == s[j]);
                else
                    dp[i][j] = (s[i] == s[j]) && (dp[i+1][j-1]);
            }
        }

        vector<string> ans;
        for(int i = 0 ; i < n - 2 ; ++i){
            if(dp[0][i]) {
                for(int j = i+1 ; j < n - 1 ; ++j){
                    if(dp[i+1][j] && dp[j+1][n-1]){
                        return true;
                    }
                }
            }
        }
        
        return false;
    }
};
