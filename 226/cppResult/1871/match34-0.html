class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        VVI dp(n + 1);
        rep(i, n + 1) dp[i].resize(i);
        for(int len = 1; len < n; ++len) {
            for(int i = 0; i + len <= n; ++i) {
                int j = i + len;
                dp[j][i] = s[i] == s[j-1] && (i + 1 >= j - 1 || dp[j-1][i+1]);
            }
        }
        for(int i = 1; i < n; ++i) {
            for(int j = i + 1; j < n; ++j) {
                if (dp[i][0] && dp[j][i] && dp[n][j]) return true;
            }
        }
        return false;
    }
};
