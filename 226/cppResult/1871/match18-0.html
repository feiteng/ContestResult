class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        vector<vector<bool>>dp(n+5,vector<bool>(n+5,false));
        for(int i = 0;i<n;++i){
            dp[i][i]=true;
            if(i!=0&&s[i]==s[i-1]){
                dp[i-1][i] = true;
            }
        }
        for(int len = 3;len<=n;++len){
            for(int i = 0;i<n;++i){
                if(i+len-1<n){
                    dp[i][i+len-1] = (dp[i+1][i+len-2]&&(s[i]==s[i+len-1]));
                }
            }
        }
        for(int i = 0;i<n;i++){
            for(int j = i+1;j<n;j++){
                if(dp[0][i]&&dp[i+1][j]&&dp[j+1][n-1]){
                    return true;
                }
            }
        }
        return false;
        
    }
};
