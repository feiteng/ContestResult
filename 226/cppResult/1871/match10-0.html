class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<int>> v(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            v[i][i] = 1;
        }
        for (int i = 0; i < n - 1; i++) {
            if (s[i] == s[i + 1]) {
                v[i][i + 1] = 1;
            }
        }
        for (int i = 2; i < n; i++) {
            for (int j = 0; j + i < n; j++) {
                if (s[j] == s[j + i]) {
                    v[j][j + i] = v[j + 1][j + i - 1];
                }
            }
        }
        for (int i = 1; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (v[0][i - 1] == 1 && v[i][j - 1] == 1 && v[j][n - 1] == 1) {
                    return true;
                }
            }
        }
        return false;
    }
};
