class Solution {
public:
    bool checkPartitioning(string s) {
      int len = s.length();
      vector<vector<int>> dp(len,vector<int>(len,0));
      for(int i = 0; i < len; i++){
        for(int j = 0; j + i < len; j++){
          if(i == 0 || ((i == 1 || dp[j + 1][j + i - 1]) && s[j] == s[j + i])){
            dp[j][j + i] = 1;
          }
        }
      }
      for(int i = 0; i < len; i++){
        for(int j = i + 1; j + 1 < len; j++){
          if(dp[0][i] && dp[i + 1][j] && dp[j + 1][len - 1]){
            return true;
          }
        }
      }
      return false;
    }
};
