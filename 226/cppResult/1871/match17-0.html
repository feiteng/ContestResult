class Solution {
public:
    bool checkPartitioning(string s) {
        int N = s.size();
        vector<vector<bool>> is_huiwen(N, vector<bool>(N));
        
        // init
        for (int length = 1; length <= N; ++ length) {
            for (int i = 0; i < N; ++ i) {
                int j = i + length - 1;
                if (j >= N) break;
                
                if (i == j) is_huiwen[i][j] = true;
                else if (i == j - 1) is_huiwen[i][j] = (s[i] == s[j]);
                else is_huiwen[i][j] = (s[i] == s[j] && is_huiwen[i+1][j-1]);
            }            
        }
        
        
        for (int i = 0; i < N - 1; ++ i) {
            for (int j = i + 1; j < N - 1; ++ j) {
                if (is_huiwen[0][i] && is_huiwen[i+1][j] && is_huiwen[j+1][N-1]) return true;
            }
        }
        
        return false;
    }
};
