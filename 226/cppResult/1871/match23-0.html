class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.length();
        vector<vector<bool>> table(n, vector<bool>(n));
        for (int i = 0; i < n; i++) {
            table[i][i] = true;
        }
        for (int i = 1; i < n; i++) {
            table[i - 1][i] = (s[i - 1] == s[i]);
        }
        for (int len = 3; len <= n; len++) {
            for (int start = 0; start + len - 1 < n; start++) {
                table[start][start + len - 1] = (s[start] == s[start + len - 1]) && table[start + 1][start + len - 2];
            }
        }
        for (int i = 1; i < n - 1; i++) {
            for (int j = i; j < n - 1; j++) {
                if (table[0][i - 1] && table[i][j] && table[j + 1][n - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
};
