class Solution {
public:
    bool checkPartitioning(string s) {
        int n = s.size();
        vector<vector<bool>> f(n, vector<bool>(n, false));
        for (int i = 0; i < n; ++i) {
            int l = i, r = i;
            while (0 <= l && r < n && s[l] == s[r]) {
                f[l--][r++] = true;
            }
        }
        for (int i = 0; i + 1 < n; ++i) {

            int l = i, r = i + 1;
            while (0 <= l && r < n && s[l] == s[r]) {
                f[l--][r++] = true;
            }
        }
        for (int i = 0; i + 2 < n; ++i) {
            for (int j = i + 1; j + 1 < n; ++j) {
                if (f[0][i] && f[i + 1][j] && f[j + 1][n - 1]) {
                    return true;
                }
            }
        }
        return false;
    }
};
