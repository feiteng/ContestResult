class Solution {
public:
    bool checkPartitioning(string s) {
        int N=s.size();
        
        vector<vector<int>> dp(N,vector<int>(N));
        for(int i=0;i<N;i++){
            dp[i][i] = 1;
        }
        for(int i=1;i<N;i++){
            if(s[i-1]==s[i]){
                dp[i-1][i] = 1;
            }
            
        }        
        for(int a=1;a<N;a++){
            for(int j=a;j<N;j++){
                int i=j-a;

                if(s[i]==s[j]&&dp[i+1][j-1]>0){
                    dp[i][j] = 1;
                }
            }
        }
        for(int i=1;i<N;i++){
            for(int j=i;j<N-1;j++){
                if(dp[0][i-1]>0&&dp[i][j]>0&&dp[j+1][N-1]>0){
                    return true;
                }
            }
        }
        return false;
    }
};
