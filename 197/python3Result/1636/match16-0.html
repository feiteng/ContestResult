class Solution:
    def numSub(self, s: str) -> int:
        ans, t = 0, 0
        m = 10**9 + 7
        for x in s:
            if x == '1':
                t += 1
            else:
                ans = (ans+ t*(t+1)//2) % m
                t = 0
        return (ans+ t*(t+1)//2) % m
