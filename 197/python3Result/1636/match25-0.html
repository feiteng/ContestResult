class Solution:
    def numSub(self, s: str) -> int:
        ans = 0
        c = 0
        for ch in s:
            if ch == '1':
                c += 1
            else:
                ans += ((c*(c+1))//2)
                c = 0
        ans += ((c*(c+1))//2)
        return ans % 1000000007
