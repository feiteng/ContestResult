class Solution:
    def numSub(self, s: str) -> int:
        MOD = 10 ** 9 + 7
        cnt, ans = 0, 0
        for ch in s:
            if ch == '0':
                ans = (ans + cnt * (cnt + 1) // 2) % MOD
                cnt = 0
            else:
                cnt += 1
        ans = (ans + cnt * (cnt + 1) // 2) % MOD
        return ans
