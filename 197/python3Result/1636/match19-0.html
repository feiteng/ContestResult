class Solution:
    def numSub(self, s: str) -> int:
        cur_streak = 0
        ans = 0
        for char in s:
            if char == '1':
                cur_streak += 1
                ans += cur_streak
            else:
                cur_streak = 0
        return ans % (10**9 + 7)
