class Solution:
    
    """
    "1" -> 1
    "11" -> "1" "1" "11" -> 3
    "111" -> "1", "1", "1", "11", "11", "111" -> 6
    "1111" -> "1", "1", "1", "1", "11", "11", "11", "111", "111", "1111" -> 10
    """
    def numSub(self, s: str) -> int:
        cur_one_count = 0
        total = 0
        for char in s:
            if char == '0':
                total  = (total + ((cur_one_count*(cur_one_count + 1))//2)%(10**9 + 7))%(10**9 + 7)
                cur_one_count = 0
            else:
                cur_one_count += 1
        total = (total + ((cur_one_count*(cur_one_count + 1))//2)%(10**9 + 7))%(10**9 + 7)
        return total
