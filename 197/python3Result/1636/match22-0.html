class Solution:
    def numSub(self, s: str) -> int:
        cnt = 0
        ans = 0
        for c in s:
            if c == '0':
                ans = (ans + (cnt * (cnt + 1) >> 1)) % 1000000007
                cnt = 0
            else:
                cnt += 1
        return (ans + (cnt * (cnt + 1) >> 1)) % 1000000007
