class Solution:
    def numSub(self, s: str) -> int:
        MOD = 1_000_000_007
        cnt, ans = 0, 0
        for c in s:
            if c == '0':
                cnt = 0
            else:
                cnt = (cnt + 1) % MOD
                ans = (ans + cnt) % MOD
        return ans
