#!/usr/bin/env python
# -*- coding: utf-8 -*-

from typing import List, Dict


MOD = 10 ** 9 + 7

class Solution:
    def numSub(self, s: str) -> int:
        x = 0
        ret = 0
        for c in s:
            if c == '1':
                x += 1
                ret = (ret + x) % MOD

            else:
                x = 0
        return ret


def _case(*a):
    assert Solution().numSub(*a[:-1]) == a[-1]
    pass

    _case('0110111', 9)
