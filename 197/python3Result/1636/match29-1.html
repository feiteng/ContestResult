class Solution:
    def numSub(self, s: str) -> int:
        s+="0"
        curr=0
        count=0
        for m,i in enumerate(s):
            if i=="1":
                curr+=1
            else:
                count+=curr*(curr+1)//2
                curr=0
        return count%(10**9 + 7)
