class Solution:
    def numSub(self, s: str) -> int:
        ret = 0
        c = 0
        for x in s:
            if x == '0':
                ret += (c*c+c)//2
                c = 0
            else:
                c += 1
        ret += (c*c+c)//2
        return ret % (10**9+7)
