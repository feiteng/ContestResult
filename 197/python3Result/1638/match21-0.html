import numpy as np
from numpy.linalg import norm as npnorm

class Solution:
    def getMinDistSum(self, pts: List[List[int]]) -> float:
        if len(pts) == 1: return 0
        seen = set()
        for p in pts:
            seen.add(tuple(p))
        if len(seen) == 1: return 0
        
        xmean = sum([x[0] for x in pts])/len(pts)
        ymean = sum([x[1] for x in pts])/len(pts)
    
        old = np.random.randn(2)
        new = np.random.randn(2)

        while npnorm(old-new)>10**-6:
            num = 0
            denom = 0
            for i in range(len(pts)):
                dist = npnorm(new - pts[i])
                num += pts[i]/dist
                denom += 1/dist
            old = new
            new = num/denom
        
        res = 0
        for i in range(len(pts)):
            res += npnorm(new - pts[i])
        if not res: return 0
        return res
