class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        def dis(a,b):
            return math.sqrt
((a[0]-b[0])**2+(a[1]-b[1])**2)
        
        def sumdis(center,points):
            ans = 0
            for p in points:
                ans += dis(center,p)
            return ans
        
        size = len(positions)
        if size<=1:
            return 0
        
        pp = positions[0]
        ans = float('inf')
        step = 100;
        ans 
= sumdis(pp,positions);
        while step>0.00001:
            flag = True
            while flag:
                flag = False
                t = pp
                Q = [pp[0],pp[1]+step]
                temp = sumdis(Q,positions)
                if temp < ans:
                    ans = temp
                    t = Q
                    flag = True
                Q = [pp[0],pp[1]-step]
                temp = sumdis(Q,positions)
                if temp < ans:
                    ans = temp
                    t = Q
                    flag = True
                Q = [pp[0]+step,pp[1]]
                temp = sumdis(Q,positions)
                if temp < ans:
                    ans = temp
                    t = Q
                    flag = True
                Q = [pp[0]-step,pp[1]]
                temp = sumdis(Q,positions)
                if temp < ans:
                    ans = temp
                    t = Q
                    flag = True
                pp = t
            step /= 2.0
        return ans
