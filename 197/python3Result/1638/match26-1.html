class Solution:
    def getMinDistSum(self, a: List[List[int]]) -> float:
        n = len(a)
        if n==1: return 0
        a1=[];a2=[]
        for i in range(n):
              a1.append(a[i][0])
              a2.append(a[i][1])
            
        x=sum(a1)/n;y=sum(a2)/n
        while True:
              xfenzi
=0;xfenmu=0;yfenzi=0;yfenmu=0
              for i in range(n):
                      g=math.sqrt((x-a1[i])**2+(y-a2[i])**2)
                      if g==0: 
break
                      xfenzi=xfenzi+a1[i]/g
                      xfenmu=xfenmu+1/g
                      yfenzi=yfenzi+a2[i]/g
                      yfenmu=yfenmu+1/g
              if xfenmu==0 or yfenmu==0: break
              xn=xfenzi/xfenmu
              yn=yfenzi/yfenmu
              if abs(xn-x)<0.00000001 and abs(yn-y)<0.00000001:
                    break
              else:
                    x=xn
                    y=yn
        res = 0.
        for i in range(n):
            res += math.sqrt((x-a1[i])**2+(y-a2[i])**2)
        return res;
