import math
class Solution:
    def getMinDistSum(self, positions) -> float:
        if len(positions)==1:
            return 0
        if len(positions)==2:
            return ((positions[1][1]-positions[0][1])**2+(positions[1][0]-positions[0][0])**2)**0.5
        a=[]
        b=[]
        n=len(positions)
        for pos in positions:
            a.append(pos[0])
            b.append(pos[1])
        x = 3*sum(a) / n
        y = 3*sum(b) / n
        while True:
            fenzi1 = 0;
            fenmu1 = 0;
            fenzi2 = 0;
            fenmu2 = 0
            count=0
            for i in range(n):
                g = math.sqrt((x - a[i]) ** 2 + (y - b[i]) ** 2)
                if g==0:
                    count=1
                    break
                fenzi1 = fenzi1 + a[i] / g
                fenmu1 = fenmu1 + 1 / g
                fenzi2 = fenzi2 + b[i] / g
                fenmu2 = fenmu2 + 1 / g
            if count==1:
                break
            xn = fenzi1 / fenmu1
            yn = fenzi2 / fenmu2
            if abs(xn - x) < 10**(-7) and abs(yn - y) < 10**(-7):
                break
            else:
                x = xn
                y = yn
        res=0
        for i in range(n):
            res=res+math.sqrt((x-a[i])**2+(y-b[i])**2)
        return res
