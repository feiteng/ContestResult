class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        directions = {(-1, 0), (1, 0), (0, -1), (0, 1)}
        lower_limit = 10**-5
        def dist_sum(p):
            s = 0
            for i in range(len(positions)):
                distx = abs(positions[i][0] - p[0])
                disty = abs(positions[i][1] - p[1])
                s += ((distx * distx) + (disty * disty)) ** 0.5
            return s
        current_point = [0, 0]
        for i in range(len(positions)):
            current_point[0] += positions[i][0]
            current_point[1] += positions[i][1]
        current_point[0] /= len(positions)
        current_point[1] /= len(positions)
        minDist = dist_sum(current_point)
        k = 0
        while (k < len(positions)):
            for i in range(len(positions)):
                if i == k: 
                    continue
                p = [positions[i][0], positions[i][1]]
                newd = dist_sum(p)
                if newd < minDist:
                    minDist = newd
                    current_point[0] = p[0]
                    current_point[1] = p[1]
            k += 1
        test_distance = 100
        flag = 0
        while (test_distance > lower_limit):
            flag = 0
            for dr in directions:
                p = [current_point[0] + test_distance * dr[0], current_point[1] + test_distance * dr[1]]
                newd = dist_sum(p)
                if newd < minDist:
                    minDist = newd
                    current_point[0] = p[0]
                    current_point[1] = p[1]
                    flag = 1
                    break
            if flag == 0:
                test_distance /= 2
            # print(test_distance)
        return minDist
