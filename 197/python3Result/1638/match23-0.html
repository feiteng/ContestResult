class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        
        def dist(a,b):
            Ds = [ ((pos[0]-a)**2 + (pos[1]-b)**2)**.5 for pos in positions]
            return sum(Ds)
        
        def go(x,y):
            d = dist(x,y)
            step = 10.
            done = 0
            while step>eps:
                done = 0
                for dx,dy in [[-1,0],[1,0],[0,-1],[0,1]]:
                    nx = 1.0*x + step*dx
                    ny = 1.0*y + step*dy
                    t = dist(nx,ny)
                    if t<d:
                        d = t
                        x = nx
                        y = ny
                        done = 1
                        break
                if not done:
                    step /= 2.
                    
            return d
        
        
        Xs = [pos[0] for pos in positions]
        Ys = [pos[1] for pos in positions]
        
        x = 1.*sum(Xs)/len(Xs)
        y = 1.*sum(Ys)/len(Ys)
        print(x, y)
           
        
        eps = 1e-6
        return go(x,y)
