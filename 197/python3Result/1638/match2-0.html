class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        def dis(p1, p2):
            return sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)
    
        def getAllDis(cur):
            ans = 0
            for p in positions:
                ans += dis(cur, p)
            return ans
        
        step = 100
        st = positions[0]
        mind = getAllDis(st)
    
        
        while step > 0.00001:
            ok = 1
            while ok:
                ok = 0
                nt = st
                tmp = [st[0], st[1]+step]
                t = getAllDis(tmp)
                if t < mind:
                    mind = t
                    ok = 1
                    nt = tmp
                
                tmp = [st[0], st[1]-step]
                t = getAllDis(tmp)
                if t < mind:
                    mind = t
                    ok = 1
                    nt = tmp
                    
                tmp = [st[0]+step, st[1]]
                t = getAllDis(tmp)
                if t < mind:
                    mind = t
                    ok = 1
                    nt = tmp
            
                tmp = [st[0]-step, st[1]]
                t = getAllDis(tmp)
                if t < mind:
                    mind = t
                    ok = 1
                    nt = tmp
                
                st = nt
            step = step / 2
            
        return mind
