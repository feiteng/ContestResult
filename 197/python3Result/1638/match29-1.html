from scipy.optimize import minimize
import numpy as np

class Solution:

    def getMinDistSum(self, positions: List[List[int]]) -> float:
        n = len(positions)

        x = [x for x, _ in positions]
        y = [y for _, y in positions]

        x0 
= np.array([sum(x) / n, sum(y) / n])

        def dist_func(x0):
            return sum(((np.full(n, x0[0]) - x) ** 2 + (np.full(n, x0[1]) - y) ** 2) ** (1 / 2))
