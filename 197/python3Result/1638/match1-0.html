class Solution:

        def lose(c):
            res = 0
            c_x, c_y = c
            for x,y in positions:
                res += ((x-c_x)**2 + (y-c_y)**2)**0.5
            return res
        x = array(positions)
        ini = (0,0)
        res = minimize(lose, ini)#, method='SLSQP')#, bounds=bnds, constraints=cons)
