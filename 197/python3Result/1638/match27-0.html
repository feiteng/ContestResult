class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        import numpy as np
        from numpy.linalg import norm as npnorm
        pts = np.array(positions)
        POINT_NUM = len(pts)
        c_pt_old = np.random.rand(2)
        c_pt_new = np.mean(pts, axis=0)

        while npnorm(c_pt_old-c_pt_new)>1e-9:
            num   = 0
            denom = 0
            for i in range(POINT_NUM):
                dist   = npnorm(c_pt_new-pts[i,:])
                if dist < 1e-9:
                    continue
                num   += pts[i,:]/dist
                denom += 1/dist
            if denom == 0:
                break
            c_pt_old = c_pt_new
            c_pt_new = num/denom

        res = 0
        for i in range(POINT_NUM):
            res += npnorm(c_pt_new-pts[i,:])
        return res
