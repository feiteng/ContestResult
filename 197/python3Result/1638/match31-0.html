class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        def distSum(x, y, points): 
            sum = 0
            for p in points: 
                distx = abs(p[0] - x)
                disty = abs(p[1] - y)
                sum += ((distx ** 2) + (disty ** 2))**0.5
            return sum

        cur_x = 0
        cur_y = 0
        n = len(positions)
        for p in positions:
            cur_x += p[0]
            cur_y += p[1]
        cur_x /= n
        cur_y /= n
  
        minimum_distance = distSum(cur_x, cur_y, positions); 

        k = 0
        while k < len(positions):
            for i in range(len(positions)):
                if i == k:
                    break
                new_x = positions[i][0]
                new_y = positions[i][1]
                newd = distSum(new_x, new_y, positions)
                if newd < minimum_distance:
                    minimum_distance = newd
                    cur_x = new_x
                    cur_y = new_y
            k+=1
  
        test_distance = 1000
        flag = 0
        lower_limit = 0.00001
        while test_distance > lower_limit:
            flag = 0
            for dx,dy in [(-1, 0), (1,0),(0,-1),(0,1)]:
                new_x = cur_x + test_distance * dx
                new_y = cur_y + test_distance * dy
                newd = distSum(new_x, new_y, positions)
  
                if newd < minimum_distance:
                    minimum_distance = newd
                    cur_x = new_x
                    cur_y = new_y
                    flag = 1
                    break; 
            if flag == 0:
                test_distance /= 2
        return minimum_distance
