import numpy as np
from numpy.linalg import norm as npnorm


class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        points = np.array(positions)
        x0, y0 = np.mean(points, axis = 0)
        #print(x0, y0)
        c_pt_old = np.random.rand(2)
        c_pt_new = np.array((x0, y0))
        #print(c_pt_new)
        steps = 0
        points_num = len(positions)
        if points_num <= 1:
            return 0
        
        
while npnorm(c_pt_old-c_pt_new) > 1e-8 or steps <= 2:
            num = 0.0
            denom = 0.0
            for i in range(points_num):                
                dist = npnorm(c_pt_new - points[i,:])
                if dist == 0:
                    continue
                num += points[i,:]/dist
                denom += 1/dist
            c_pt_old = c_pt_new
            
if denom == 0:
                break
            c_pt_new = num/denom
            #print(c_pt_new)
            steps += 1
        res = 0.0
        for i in range(points_num):
            res += npnorm(c_pt_new - points[i,:])
        return res
