class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        
        from scipy.optimize import minimize
        import numpy as np
        
        x,y = list(zip(*positions))

        n = len(x)
        x0 = np.array([sum(x)/n,sum(y)/n])
        
        def dist_func(x0):
            return sum(
                ((np.full(n, x0[0])-x)**2+(np.full(n, x0[1])-y)**2)**(0.5)
            )
        res = minimize(dist_func, x0, method='nelder-mead', options={'xtol': 1e-8, 'disp': False})
        return res.fun
