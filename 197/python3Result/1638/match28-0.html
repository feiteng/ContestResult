class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        
        x = [point[0] for point in positions]
        y = [point[1] for point in positions]
        

        x0 = np.array([sum(x)/len(x),sum(y)/len(y)])
        def dist_func(x0):
            return sum(((np.full(len(x),x0[0])-x)**2+(np.full(len(x),x0[1])-y)**2)**(1/2))
        res = minimize(dist_func, x0, method='nelder-mead', options={'xtol': 1e-7})
        
        my, mx = res['x']
        def dist_sum(my, mx):
            return sum(sqrt((my-y)**2+(mx-x)**2) for y, x in positions)
        dis = dist_sum(my,mx)
        return dis
