import numpy as np
from numpy.linalg import norm as npnorm

class Solution:
    def getMinDistSum(self, positions) -> float:
        if len(positions) == 1:
            return 0
        positions = np.array(positions)
        c_pt_old 
= np.random.rand(2)
        c_pt_new = np.random.rand(2)
        #print(c_pt_new, c_pt_old)
        while npnorm(c_pt_old - c_pt_new) > 1e-10:
            num = 0
            denom = 0
            for i in range(len(positions)):
                dist = npnorm(c_pt_new - positions[i, :])
                if dist == 0:
                    continue
                num += positions[i, :] / dist
                denom += 1 / dist
            if denom == 0:
                return 0
            c_pt_old 
= c_pt_new
            c_pt_new = num / denom
            #print(c_pt_new)

        ans = 0
        for i in range(len(positions)):
            ans += npnorm(c_pt_new - positions[i, :])
        return ans
