class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        dist = lambda p1, p2: ((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2) ** 0.5
        eps = 0.00000001
        x = y = 0
        total = sum(dist((x, y), p) for p in positions)
        step = 100.0
        done = 0
        while step > eps:
            done = 0
            for dx, dy in ((1, 0), (-1, 0), (0, 1), (0, -1)):
                nx = step * dx + x
                ny = step * dy + y
                new_total = sum(dist((nx, ny), p) for p in positions)
                if new_total < total:
                    total = new_total
                    x = nx
                    y = ny
                    done = 1
                    break
            if not done:
                step /= 2
        return total
