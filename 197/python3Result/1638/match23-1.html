class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        def distance2All(x, y):
            acc = 0
            for x1, y1 in positions:
                dx, dy = x1-x, y1-y

                acc += math.sqrt(dx*dx + dy*dy)
                
            return acc

        x, y = 0, 0
        do 
= distance2All(x, y)
        step = 100.0
        eps = 0.000000001

        while step > eps:
            done = False
            for dy, dx in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx = x + step*dx
                ny = y + step*dy
                
                dist = distance2All(nx, ny)
                if dist < do:
                    do = dist
                    x = nx
                    y = ny

                    done = True
                    break
            if not done:
                step /= 2

        return do
