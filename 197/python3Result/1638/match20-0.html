class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        if len(positions) == 1 or all(p == positions[0] for p in positions):
            return 0
        
        import numpy as np
        from numpy.linalg import norm as npnorm
        c_pt_old = np.random.randn(2)
        c_pt_new = np.random.randn(2)

        while npnorm(c_pt_old-c_pt_new)>1e-6:
            num   = 0
            denom = 0
            for i in range(len(positions)):
                dist   = npnorm(c_pt_new - positions[i])
                num   += positions[i]/dist
                denom += 1/dist
            c_pt_old = c_pt_new
            c_pt_new = num/denom
        
        res = 0
        for i in range(len(positions)):
            res += npnorm(c_pt_new - positions[i])
        
        return res
