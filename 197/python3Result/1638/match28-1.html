class Solution:
    def getMinDistSum(self, positions: List[List[int]]) -> float:
        
        from scipy.optimize import minimize
        import numpy as np
        x 
= [point[0] for point in positions]
        y = [point[1] for point in positions]

        x0 = np.array([sum(x)/len(x),sum(y)/len(y)])
        def dist_func(x0):
            return sum(((np.full(len(x),x0[0])-x)**2+(np.full(len(x),x0[1])-y)**2)**(1/2))
        res = minimize(dist_func, x0, method='nelder-mead', options={'xtol': 1e-8, 'disp': True})
        
        print(res.x[1])
        return sum([sqrt((x - res.x[0]) ** 2  + (y - res.x[1]) ** 2) for x, y in positions])
