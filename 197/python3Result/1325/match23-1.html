from heapq import heappop, heappush
from collections import defaultdict
class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        if start == end:
            return 1
        
        graph 
= defaultdict(list)
        for p, (a, b) in zip(succProb, edges):
            graph[a].append((b, p))
            graph[b].append((a, p))
            
        pq = [(-1.0, start)]
        visited = [False] * n
        visited[start] = True
        
        while len(pq) > 0:
            p, i = heappop(pq)
            
if i == end:
                return -p
            visited[i] = True
            for j, edge_p in graph[i]:
                if visited[j] or edge_p == 0:
                    continue
                heappush(pq, (p * edge_p, j))
        
        return 0
