class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = collections.defaultdict
(list)
        
        for [i, j], x in zip(edges, succProb) :
            graph[i].append([j, x])
            graph[j].append([i, x])
            
        visited = set()
        ans = []
        probs = { i : 0 for i in range(n)}
        
        pq = [(-1, start)]
        while pq :
            prob, node = heapq.heappop(pq)
            visited.add(node)
            
            
for child, p in graph[node] :
                new_prob = p * prob
                if new_prob < probs[child] :
                    probs[child] = new_prob
                    heapq.heappush(pq, (new_prob, child))
        return -probs[end]
