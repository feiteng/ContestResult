import queue

class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        
        graph = [[] for _ in range(n)]
        
        for i in range(len(edges)):
            x,y = edges[i]
            cost = succProb[i]
            graph[x].append((y,cost))
            graph[y].append((x,cost))
            
        q = queue.PriorityQueue()
        q.put((-1, start))
        store = [0]*n
        store[start] = 1
        count = 0
        while(not q.empty() and count < n):
            count += 1
            prob, node = q.get()
            prob = -prob
            #print(graph[node])
            for neigh,cost in graph[node]:
                #print(node)
                new_prob = prob*cost
                
if(new_prob > store[neigh]): 
                    store[neigh] = new_prob
                    q.put((-new_prob,neigh))
                    
        #print(store)
                    
        return store[end]
