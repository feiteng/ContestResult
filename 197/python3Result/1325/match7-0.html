class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        
        g = defaultdict(list)
        for i in range(len(edges)):
            a,b = edges[i]
            p = succProb[i]
            g[a].append((b,p))
            g[b].append((a,p))
            
        #print(g)
        
        q = [(-1.0, start)]
        visited = set()
        while q:
            cost, node = heappop(q)
            visited.add(node)
            #print(node, cost)
            
            if node == end:
                return -cost
            
            for nb, cs in g[node]:
                if nb not in visited:
                    elem = (cost*cs, nb)
                    heappush(q, elem)
        
        return 0.0
