class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = defaultdict(list)
        for i in range(len(edges)):
            a, b = edges[i]
            prob = succProb[i]
            graph[a].append((b, prob))
            graph[b].append((a, prob))
        
        
        heap = [(-1, start)]
        visited = set([])
        while heap:
            curprob, cur = heappop(heap)
            if cur in visited:
                continue
            visited.add(cur)
            if cur == end:
                return -1*curprob
            for nei, prob in graph[cur]:
                if nei in visited:
                    continue
                heappush(heap, (curprob*prob, nei))
        return 0
