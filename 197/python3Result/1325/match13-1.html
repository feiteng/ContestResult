class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        
        d = collections.defaultdict(list)
        for (a, b), p in zip(edges, succProb):
            d[a].append((b, p))
            d[b].append((a, p))
        
        hq 
= [[-1.0, start]]
        v = set()
        ans = 0.0
        while hq:
            p, i = heapq.heappop(hq)
            if i == end:
                return - p
            v.add(i)
            for j, p2 in d[i]:
                if j not in v:
                    heapq.heappush(hq, [p * p2, j])
        return ans
