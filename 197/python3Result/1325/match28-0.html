class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        neighbors = [[] for i in range(n)]
        for i in range(len(edges)):
            x, y = edges[i]
            neighbors[x].append([y,succProb[i]])
            neighbors[y].append([x,succProb[i]])
        pq = [[-1.0, start]]
        visited = [False]*n
        while len(pq) > 0:
            tmp = heapq.heappop(pq)
            pro = -tmp[0]
            pos = tmp[1]
            if visited[pos] is True:
                continue
            visited[pos] = True
            if pos == end:
                return pro
            for x in neighbors[pos]:
                nex = x[0]
                if visited[nex] is False:
                    pp = pro*x[1]
                    heapq.heappush(pq, [-pp, nex])
            
            
        return 0.0
