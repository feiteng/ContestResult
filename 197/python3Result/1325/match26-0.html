class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        
        graph = [[] for i in range(n)]
        
        for (a, b), p in zip(edges, succProb):
            graph[a].append((b, p))
            graph[b].append((a, p))
        
        vis = [0 for i in range(n + 1)]
        
        queue = [(-1, start)]
        
        while queue:
            p, node = heapq.heappop(queue)
            if node == end:
                return - p
            if not vis[node]:
                vis[node] = 1
            else:
                continue
            
            for x, y in graph[node]:
                if not vis[x]:
                    heapq.heappush(queue, (p * y, x))
        
        return 0.0
