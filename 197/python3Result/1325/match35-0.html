class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        
        
        neibs = collections.defaultdict(list)
        
        for ed, pb in zip(edges, succProb):
            neibs[ed[0]].append((ed[1], pb))
            neibs[ed[1]].append((ed[0], pb))
        
        prev_nodes = [(-1.0, start)]
        
        ret = 0.0
        visited = collections.defaultdict(float)
        while prev_nodes:
            cur_prob, cur_node = heapq.heappop(prev_nodes)
            cur_prob = -1*cur_prob
            if cur_prob< ret:
                break
            if cur_node == end:
                ret = max(ret, cur_prob)
            #elif cur_node not in visited or cur_prob>visited[cur_node]:
            elif cur_prob >= ret and (cur_node not in visited or cur_prob>visited[cur_node]):
                visited[cur_node] = cur_prob
                for ne, pb in neibs[cur_node]:
                    heapq.heappush(prev_nodes, (-1*pb*cur_prob, ne))
        return ret
