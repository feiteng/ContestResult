class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        # 状态 是 num,prob ,只有prob比当前的大才替换，否则不继续进行
        # 用bfs做
        d= defaultdict(list)
        for i in range(len(edges)):
            u,v = edges[i]
            w = succProb[i]
            d[u].append([v,w])
            d[v].append([u,w])
        root = [[start,1]]
        dp = [0]*n
        dp[start]=1
        while root:
            # w是已经遍历的累积权重
            a,w = root.pop(0)
            # u是下一个点的编号，v是下一条边的权重
            for u,v in d[a]:
                if w*v >dp[u]:
                    dp[u] = w*v
                    root.append([u,w*v])
        return dp[end]
