import heapq
from collections import defaultdict

class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        eg = defaultdict(list)
        for i, (a, b) in enumerate(edges):
            eg[a].append([b, succProb[i]])
            eg[b].append([a, succProb[i]])

        dis, q 
= {}, [[0, start]]
        while q:
            disa, a = heapq.heappop(q)
            if a == end:
                return 1 - disa
            if a not in dis:
                dis[a] = disa
                for b, c in eg[a]:
                    heapq.heappush(q, [1 - (1-disa)*c, b])
        return 0
