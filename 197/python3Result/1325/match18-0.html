class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        visited = set()
        queue = [(-1, start)]
        probs = defaultdict(list)

        for edge, sp in zip(edges, succProb):
            probs[edge[1]].append((edge[0], sp))
            probs[edge[0]].append((edge[1], sp))

        while queue:
            prob, idx = heapq.heappop(queue)
            if idx in visited:
                continue
            if idx == end:
                return -prob
            visited.add(idx)
            for next, p in probs[idx]:
                if next not in visited:
                    heapq.heappush(queue, (prob * p, next))

        return 0
