from collections import defaultdict
class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:        
        neighbor = defaultdict(set)
        for i, (a,b) in enumerate(edges):
            prob = succProb[i]
            neighbor[a].add((b,prob))
            neighbor[b].add((a,prob))
        q = [(start, 1)]
            
        res = {x:0 for x in range(n)}
        res[start] = 1
        while q:
            new_q = []
            while q:
                node, prob = q.pop()
                for v,b in neighbor[node]:
                    if prob * b > res[v]:
                        res[v] = prob*b
                        new_q += [(v, prob*b)]
            q = new_q
        return res[end]
