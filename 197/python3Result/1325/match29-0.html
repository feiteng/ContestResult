class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        E = defaultdict(list)
        for (u, v), p in zip(edges, succProb):
            E[u].append((v, p))
            E[v].append((u, p))
        
        vis = [False] * n
        pq = [[-1, start]]
        while pq:
            prob, node = heappop(pq)
            if node == end:
                return -prob
            if vis[node]:
                continue
            vis[node] = True
            for v, w in E[node]:
                if not vis[v]:
                    heappush(pq, [prob * w, v])        
        return 0
