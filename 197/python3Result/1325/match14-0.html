from heapq import heappush, heappop
class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = [[] for _ in range(n)]
        for (n1, n2), p in zip(edges, succProb):
            graph[n1].append((n2, p))
            graph[n2].append((n1, p))
        
        vis = set()
        h = [(-1, start)]
        while h:
            p, node = heappop(h)
            p = -p
            if node == end:
                return p
            if node in vis:
                continue
            vis.add(node)
            for n1, p1 in graph[node]:
                heappush(h, (- p * p1, n1))
        return 0
