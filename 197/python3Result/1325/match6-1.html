class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        dic = defaultdict(list)
        for i in range(len(edges)):
            a, b = edges[i]
            succ = succProb[i]
            dic[a].append([b,succ])
            dic[b].append([a,succ])
        seen = set()
        q = [(-1,start)]
        while q:
            prob, curr = heapq.heappop
(q)
            if curr in seen:
                continue
            seen.add(curr)
            if curr == end:
                return -prob
            for nxt,cost in dic[curr]:
                if nxt in seen:
                    continue
                heapq.heappush(q,(prob*cost,nxt))
        return 0
