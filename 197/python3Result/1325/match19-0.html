class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        adj_list = collections.defaultdict(list)
        for ind, edge in enumerate(edges):
            adj_list[edge[0]].append((edge[1], succProb[ind]))
            adj_list[edge[1]].append((edge[0], succProb[ind]))
        q = []
        heapq.heappush(q, (-1, start))
        visited = set()
        while q:
            curr_prob, node = heapq.heappop(q)
            visited.add(node)
            if node == end:
                return abs(curr_prob)
            for next_node, prob in adj_list[node]:
                if next_node not in visited:
                    heapq.heappush(q, (prob * curr_prob, next_node))
        return 0
