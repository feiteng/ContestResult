import heapq

class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        best = {i:0 for i in range(n)}
        adj = {i:[] for i in range(n)}
        for edge,p in zip(edges, succProb):
            u,v = edge
            adj[u].append((v,p))
            adj[v].append((u,p))
        pq = []
        heapq.heappush(pq, (-1.0, start))
        while len(pq) > 0:
            cp, cur = heapq.heappop(pq)
            cp = -cp
            if cur == end:
                return cp
            if cp < best[cur]:
                continue
            best[cur] = cp
            for next_node, prob in adj[cur]:
                nprob = cp * prob
                if nprob > best[next_node]:
                    best[next_node] = nprob
                    heapq.heappush(pq, (-nprob, next_node))
        return 0.0
