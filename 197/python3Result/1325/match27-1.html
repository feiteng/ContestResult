from collections import defaultdict
import heapq as pq
class Solution:
    def dfs(self, curr, par, graph, cw, mind):
        # print('curr', curr)
        if mind[curr] > cw: return
        mind[curr] = cw
        for ne,nw in graph[curr]:
            if ne == par: continue
            if mind[ne] < cw * nw:
                self.dfs(ne, curr, graph, cw * nw, mind)
        
        
    
def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = defaultdict(list)
        for (s,d),p in zip(edges, succProb):
            graph[s].append((d,p))
            graph[d].append((s,p))
        
        mind = defaultdict(float)
        q = []
        pq.heappush(q, (-1, start))
        while q:
            # print('q', q)
            currw, currn = pq.heappop(q)
            currw *= -1
            # if currn == end: return currw
            if mind[currn] > currw: continue
            else: mind[currn] =
 currw
            for ne,nw in graph[currn]:
                if mind[ne] < currw * nw:
                    pq.heappush(q, (-currw * nw, ne))
        
        return mind[end]
