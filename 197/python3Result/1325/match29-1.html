class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        # maximum probably of success
        
        # hence, no repeated edges (proof by contradiction)
        
        # dijkstra basically
        
        # cheapest path to any node
        
        # cheapest == highest probability
        
        # quite clearly a proof by contradiction type thing again
        
        # heap
        
        # expanded node
        
        g = defaultdict(list)
        
        for (u,v),succ in zip(edges, succProb):
            g[u].append([v,succ])
            g[v].append([u,succ])
            
        
        ret = {}
        
        # (weight, node)
        # invert
        heap = [[-1.0, start]]
        
        while heap:
            
            xsucc,x = heappop(heap)
            xsucc = -xsucc
            
            if x in ret:
                continue
            
            ret[x] = xsucc
            
            
            for y,ymult in g[x]:
                heappush(heap, [-xsucc*ymult, y])
            
        # print(ret)
        
        if end in ret:
            return ret[end]
        else:
            return 0
