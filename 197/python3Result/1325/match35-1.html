class Solution:
    def maxProbability(self, n: int, eds: List[List[int]], prob: List[float], start: int, end: int) -> float:
        di = {i:[] for i in range(n)}
        for ed, p in zip(eds, prob):
            di[ed[0]].append((ed[1], p))
            di[ed[1]].append((ed[0], p))
        
        pq = [(-1, start)]
        while pq:
            nprob, node = heappop(pq)
            if node == end: return -nprob
            for nei, p in di[node]:
                heappush(pq, (nprob * p, nei))
            di[node] = []
        
        return 0.
