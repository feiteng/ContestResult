class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        dist = [0.0] * n
        dist[start] = 1.0
        
        graph = [[] for _ in range(n)]
        for i, e in enumerate(edges):
            graph[e[0]].append((e[1], succProb[i]))
            graph[e[1]].append((e[0], succProb[i]))
                               
        #print(graph)
        pq = [(-1.0, start)]
        while pq:
            d, idx = heapq.heappop(pq)
            d = -d
            if dist[idx] > d: continue

            #print(d, idx, graph[idx])
            for n1, d1 in graph[idx]:
                #print(d1, n1, dist[n1])
                if dist[n1] < d * d1:
                    dist[n1] = d * d1
                    if n1 != end:
                        heapq.heappush(pq, (-d * d1, n1))
                
        #print(dist)
        return dist[end]
