import collections
import heapq
class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        G = collections.defaultdict(list)
        for i, p in enumerate(edges):
            u, v = p
            G[u].append((succProb[i], v))
            G[v].append((succProb[i], u))
        seen = set()
        heap = [(-1, start)]
        while heap:
            a, u = heapq.heappop(heap)
            if u == end: return -a
            seen.add(u)
            for w, v in G[u]:
                if v not in seen:
                    heapq.heappush(heap, (a*w, v))
            # print(heap)
        return 0
