import collections, heapq
class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        to = collections.defaultdict(list)
        for i, e in enumerate(edges):
            to[e[0]].append((e[1], 1.0/succProb[i]))
            to[e[1]].append((e[0], 1.0/succProb[i]))
        heap = []
        heapq.heappush(heap, (1, start))
        visited = [float('inf')]*n
        while heap:
            p, n = heapq.heappop(heap)
            visited[n] = p
            if n == end:
                return 1/p
            for nn, np in to[n]:
                a = p*np
                if visited[nn] > a:
                    visited[nn] = a
                    heapq.heappush(heap, (a, nn))
        return 0
