from collections import defaultdict, deque
import heapq

class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graphs = defaultdict(set)
        for ii, (from_, to_) in enumerate(edges):
            prob = succProb[ii]
            graphs
[from_].add((to_, prob))
            graphs[to_].add((from_, prob))
        
        q = [(-1, start)]
        visited = set()
        max_prob = 0
        
while q:
            curr_prob, curr = heapq.heappop(q)
            # print(curr, curr_prob)
            if curr == end:
                return -curr_prob
            if curr in visited:
                continue
            visited.add(curr)
            for nxt, nxt_prob in graphs[curr]:
                if nxt in visited:
                    continue
                heapq.heappush(q, (curr_prob * nxt_prob, nxt))
        return 0
