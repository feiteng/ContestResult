class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = collections.defaultdict(list)
        for i,[e1,e2] in enumerate(edges):
            graph[e1].append((e2,succProb[i]))
            graph[e2].append((e1,succProb[i]))
        h = [(-1,start)]
        v = {start}
        while h:
            p,node = heapq.heappop(h)
            if node == end:
                return -p
            v.add(node)
            for nei,pn in graph[node]:
                if nei not in v:
                    heapq.heappush(h,(p*pn,nei))
        return 0
