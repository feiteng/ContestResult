class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], p: List[float], start: int, end: int) -> float:
        graph = collections.defaultdict(list)
        for x, y in zip(edges, p):
            a, b=  x
            graph[a].append((b, y))
            graph[b].append((a, y))
        
        h = []
        h = [(-1, start)]

        que = [(1, start)]
        check = set([start])
        thr = 1
        for d, node in que:
            if node == end:
                thr = d
                break
            for y, d2 in graph[node]:
                if y not in check:
                    check.add(y)
                    que.append((d*d2, y))
        else:
            return 0
        
        check = set()
        while h:
            #print(h)
            p, node = heapq.heappop(h)
            p = -p
            
            
if node == end:
                return p
            check.add(node)
            for y, d in graph[node]:
                if y not in check:

                    heapq.heappush(h, (-p*d, y))
        return 0
