class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        d=defaultdict(list)
        for (a,b),prob in zip(edges,succProb):
            d[a].append([-prob,b])
            d[b].append([-prob,a])
        que=[[-1,start]]
        visited=set()
        while que:
            prob,q=heapq.heappop(que)
            if q==end:
                return -prob
            if q in visited:
                continue
            visited.add(q)
            for nxtProb,nxt in d[q]:
                heapq.heappush(que,[-nxtProb*prob,nxt])
        return 0
