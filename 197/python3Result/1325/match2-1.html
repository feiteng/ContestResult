class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        adj = [[] for _ in range(n)]
        for (a, b), prob in zip(edges, succProb):
            adj[a].append((b, prob))
            adj[b].append((a, prob))
        max_prob = [0.] * n
        max_prob[start] = 1.
        max_heap = [(-1.0, start)]
        while max_heap:
            # print(max_heap)
            prob, i = heapq.heappop(max_heap)
            prob = -prob
            if prob <= max_prob[end]:
                return max_prob[end]
            for node, p in adj[i]:
                
if p * prob > max_prob[node]:
                    max_prob[node] = p * prob
                    heapq.heappush(max_heap, (-p * prob, node))
        return max_prob[end]
