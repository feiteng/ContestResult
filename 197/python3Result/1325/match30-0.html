class Solution:
    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        graph = defaultdict(list)
        for (a, b), s in zip(edges, succProb):
            graph[a].append((b, s))
            graph[b].append((a, s))
        heap = [(-1, start)]
        visited = set()
        while heap:
            prob, node = heapq.heappop(heap)
            if node == end:
                return -prob
            if node in visited:
                continue
            visited.add(node)
            for b, s in graph[node]:
                heapq.heappush(heap, (prob * s, b))
        return 0
