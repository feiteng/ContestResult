    def maxProbability(self, n: int, edges: List[List[int]], succProb: List[float], start: int, end: int) -> float:
        self.ans = 0
        self.start = start
        self.end = end
        self.toList = [[] for _ in range(n)]
        for i in range(len(edges)):
            s, t = edges[i]
            self.toList[s].append((t, succProb[i]))
            self.toList[t].append((s, succProb[i]))
        toList = self.toList
        q = []
        q.append((-1.0, start))
        visited = set()
        while q:
            p, cur = heapq.heappop(q)
            p *= -1
            if cur == end:
                return p
            if cur in visited:
                continue
            visited.add(cur)
            
            for to, dp in toList[cur]:
                if to not in visited:
                    heapq.heappush(q, (-dp*p, to))
                    
        return 0.0
