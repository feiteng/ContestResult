class Solution {
    public double getMinDistSum(int[][] positions) {
        double avgX = 0d;
        double avgY = 0d;
        for (int[] position : positions) {
            avgX += position[0];
            avgY += position[1];
        }
        avgX /= positions.length;
        avgY /= positions.length;

        double step = 1;
        double eps = 0.00000001;
        double d = dist(avgX, avgY, positions);
        int[][] changes = new int[][]{{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        while (step > eps) {
            boolean flag = false;
            for (int[] change : changes) {
                double t = dist(avgX + step * change[0], avgY + step * change[1], positions);
                if (t < d) {
                    d = t;
                    flag = true;
                    avgX = avgX + step * change[0];
                    avgY = avgY + step * change[1];
                    break;
                }
            }
            if (!flag) {
                step /= 2;
            }
        }
        return d;
    }

    private double dist(double x, double y, int[][] positions) {
        double d = 0d;
        for (int[] position : positions) {
            d += Math.sqrt(Math.pow(x - position[0], 2) + Math.pow(y - position[1], 2));
        }
        return d;
    }
}
