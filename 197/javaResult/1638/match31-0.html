class Solution {
    public double getMinDistSum(int[][] positions) {
        int n = positions.length;
        double x = 0;
        double y = 0;
        for ( int i = 0; i < n; ++i ) {
            x += positions[i][0];
            y += positions[i][1];
        }
        x /= n;
        y /= n;
        double d = dist(positions, x, y);
        double step = 100.0;
        int done = 0;
        double[] dx = new double[] {-1, 0, 1, 0};
        double[] dy = new double[] {0, 1, 0, -1};
        // while we still need a more precise answer
        while (step > 0.0000001)  {
            done = 0;
            for ( int i = 0; i < 4; ++i )
            {
                // check the neighbors in all 4 directions.
                double nx = x + step*dx[i];
                double ny = y + step*dy[i];

                // find the sum of distances to each neighbor
                double t = dist(positions, nx, ny);

                // if a neighbor offers a better sum of distances
                if ( t < d ) {
                    d = t;
                    x = nx;
                    y = ny;

                    done = 1;
                    break;
                }
            }

            // half the step size, because no update has been made, so we might have
            // jumped too much, and now we need to head back some.
            if (done == 0) {
                step /= 2;
            }
        }
        double res = dist(positions, x, y);
        //System.out.println(res);
        return res;
    }
    private double dist(int[][] a, double x, double y) {
        double ret = 0;

        for ( int i = 0; i < a.length; i++)
        {
            double dx = a[i][0] - x;
            double dy = a[i][1] - y;

            ret += Math.sqrt(dx*dx + dy*dy); // classical distance formula
        }

        return ret;
    }
}
