class Solution {
    public double getMinDistSum(int[][] positions) {
        int n = positions.length;
        
double[] cur = new double[2];
        for (int[] pos : positions) {
            cur[0] += pos[0];
            cur[1] += pos[1];
        }
        
        cur[0] /= n;
        cur[1] /= n;
        
        double res = dist(cur, positions);
        
        int k = 0;
        while (k < n) {
            for (int i = 0;i < n;i++) {
                if (i == k) continue;
                int[] pos = positions[i];
                
double[] newPoint = new double[2];
                newPoint[0] = pos[0];
                newPoint[1] = pos[1];
                double newDist = dist(newPoint, positions);
                if (newDist < res) {
                    res = newDist;
                    cur[0] = newPoint[0];
                    cur[1] = newPoint[1];
                }
            }
            k++;
        }
        
        
double testDist = 1000, limit = 0.00001;
        int flag = 0;
        double[][] testPoints = new double[][]{
            {-1, 0}, {0, 1}, {1, 0}, {0, -1}
        };
        while (testDist > limit) {
            flag = 0;
            for (int i = 0;i < 4;i++) {
                double[] newPoint = new double[2];
                newPoint[0] = cur[0] + testDist * testPoints[i][0];
                newPoint[1] = cur[1] + testDist * testPoints[i][1];
                
                double newDist = dist(newPoint, positions);
                if (newDist < res) {
                    res = newDist;
                    cur[0] = newPoint[0];
                    cur[1] = newPoint[1];
                    flag = 1;
                    break;
                }
            }
            if (flag == 0) testDist /= 2;
        }
        
        return res;
    }
    
    double dist(double[] point, int[][] positions) 
{
        double sum = 0;
        for (int[] pos : positions) {
            double distx = Math.abs(pos[0] - point[0]);
            double disty = Math.abs(pos[1] - point[1]);
            sum += Math.sqrt(distx * distx + disty * disty);
        }
        
        return sum;
    }
}
