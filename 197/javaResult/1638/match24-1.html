class Solution {
    
private int[][] neighbors = new int[][]{{0, +1}, {-1, 0}, {0, -1}, {+1, 0}};

    
public double getMinDistSum(int[][] positions) {
        double error = 1e-6;
        double[] point = new double[2];
        for (int[] p : positions) {
            point[0] += p[0];
            point[1] += p[1];
        }
        point[0] /= positions.length;
        point[1] /= positions.length;

        double mindist = dist(positions, point);
        double dist = 100;
        
while (dist > error) {
            boolean found = false;
            for (int[] neighbor : neighbors) {
                double[] next = new double[] {point[0] + dist * neighbor[0], point[1] + dist * neighbor[1]};
                double nextdist = dist(positions, next);
                if (nextdist < mindist) {
                    mindist = nextdist;
                    point = next;
                    found = true;
                    break;
                }
            }
            if (!found) {
                dist /= 2;
            }
        }
        return mindist;
    }

    private double dist(int[][] positions, double[] point) {
        double out = 0;
        for (int[] p : positions) {
            out += Math.sqrt(Math.pow(p[0] - point[0], 2) + Math.pow(p[1] - point[1], 2));
        }
        return out;
    }
}
