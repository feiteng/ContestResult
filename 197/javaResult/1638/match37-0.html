class Solution {
    double getDistanceFrom(double x, double y, int[][] positions) {
        double res = 0;
        for (int[] position : positions) {
            double dx = ((double) position[0]) - x;
            double dy = ((double) position[1]) - y;
            res += Math.sqrt(dx * dx + dy * dy);
        }
        return res;
    }

    public double getMinDistSum(int[][] positions) {
        int[] dx = {-1, 0, 1, 0};
        int[] dy = {0, 1, 0, -1};
        double n = positions.length;
        double x = 0;
        double y = 0;
        for (int[] position : positions) {
            x += position[0];
            y += position[1];
        }
        x /= n;
        y /= n;
        double minDist = getDistanceFrom(x, y, positions);
        double step = 100.0;
        boolean changed;

        while (step > 1e-9) {
            changed = false;
            for (int i = 0; i < 4; ++i) {
                double nx = x + step * dx[i];
                double ny = y + step * dy[i];
                double dist = getDistanceFrom(nx, ny, positions);
                if (dist < minDist) {
                    minDist = dist;
                    x = nx;
                    y = ny;
                    changed = true;
                    break;
                }
            }
            if (!changed) {
                step /= 2;
            }
        }
        return minDist;
    }
}
