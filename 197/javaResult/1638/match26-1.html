		double ret = 0;

		for (int i = 0; i <a.length; ++i) {
			double dx = a[i][0] - x;
			double dy = a[i][1] - y;

			ret += Math.sqrt(dx * dx + dy * dy); // classical distance formula
		}

		return ret;
	}
