class Solution {
    double dist(double x,double y, int[][] pos){
        double res = 0.0;
        for(int[] p:pos){
            double dx = x-p[0];
            double dy = y-p[1];
            res+=Math.sqrt(dx*dx+dy*dy);
        }
        return res;
    }
    public double getMinDistSum(int[][] pos) {
        if(pos.length==1) return 0.0;
        
double x=0.0;
        double y = 0.0;
        for(int[] p:pos){
            x +=p[0];
            y +=p[1];
        }
        x/=pos.length;
        y/=pos.length;
        
        double min=dist(x,y, pos);
        int k=0;
        while(k<pos.length){
            for(int i=0;i<pos.length&&i!=k;i++){
               double d = dist(pos[i][0],pos[i][1],pos);
                if(d<min){
                    min = d;
                    x = pos[i][0];
                    y = pos[i][1];
                }
            }
            k++;
        }
        
double[] dx = new double[]{0.00001,0,-0.00001,0};
        double[] dy = new double[]{0,0.00001,0,-0.00001};
        double lower = 1;
        
int flag = 0;
        double gra = 10000;
        while(gra>lower){
            flag = 0;
            for(int i=0;i<4;i++){
                double nx = x+gra*dx[i];
                double ny = y+gra*dy[i];
                double d = dist(nx,ny,pos);
                if(d<min){
                    min = d;
                    x = nx;
                    y = ny;
                    flag = 1;
                    break;
                }
            }
            if(flag==0) gra/=2;
        }
        return min;
    }
}
