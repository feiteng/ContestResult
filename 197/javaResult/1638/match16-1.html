class Solution {
  
    private static final double THRESHOLD = 0.0000001;
    
    private int[][] DIRS = new int[][]{{0,1},{1,0},{-1,0},{0,-1}};

    public double getMinDistSum(int[][] positions) {
        
        double x = 0;
        double y = 0;
        for (int[] pos : positions) {
            x += pos[0];
            y += pos[1];
        }
        x /= positions.length;
        y /= positions.length;
        
        float dist = 1;
        double minDistSum = computeDistSum(positions, x, y);
        
        
while (dist > THRESHOLD) {
            boolean hasBetterPosition = false;
            for (int[] dir : DIRS) {
                double nextX = x + dist*dir[0];
                double nextY = y + dist*dir[1];
                double distSum = computeDistSum(positions, nextX, nextY);
                if (distSum < minDistSum) {
                    minDistSum = distSum;
                    hasBetterPosition = true;
                    x = nextX;
                    y = nextY;
                
}
            }
            if (!hasBetterPosition) {
                    dist /= 2;
            }
        }
        return minDistSum;
        
        // geometric mean or median then calculate teh diff
    }
    
    private double computeDistSum(int[][] positions, double x, double y) {
        double distSum = 0;
        for (int[] pos : positions) {
            double dx = x - pos[0];
            double dy = y - pos[1];
            distSum += Math.sqrt(dx*dx + dy*dy);
        }
        return distSum;
    }
}
