class Solution {
    int[][]positions;
    double minX;
    double minY;
    
double eps = 0.0000001;
    int[][]directions = new int[][]{{-1,0},{1,0},{0,-1},{0,1}};
    double dist(double x,
double y){
        double result = 0;
        for(int i=0;i<positions.length;i++){
            double dx = positions[i][0]-x;
            double dy = positions[i][1]-y;
            result+=Math.sqrt(dx*dx+dy*dy);
        }
        return result;
    }
    private void calMin()
{
        int n = positions.length;
        double x = 0,y=0;
        for(int[]position:positions){
            x+=position[0];
            y+=position[1];
        }
        minX=x/n;
        minY=y/n;
    }
    private double cal(){
        double minDistance = dist(minX,minY);
        double step = 100.0;
        boolean done = false;
        while(step>eps){
            done = false;
            
for(int[]direction:directions){
                double newX = minX+step*direction[0];
                double newY = minY+step*direction[1];
                double t = dist(newX,newY);
                if(t<minDistance){
                    minDistance = t;
                    minX=newX;
                    minY=newY;
                    done=true;
                    break;
                }
            }
            if(!done){
                step/=2;
            }
        }
        return minDistance;
        
    }
    public double getMinDistSum(int[][] positions) {
        this.positions=positions;
        calMin();
        return cal();
        
        
    }
}
