class Solution {
    int dx[] = {-1, 0, 1, 0};
    int dy[] = {0, 1, 0, -1};
    
    public double getMinDistSum(int[][] positions) {
        double x = 0;
        double y = 0;
        for (int i = 0; i < positions.length; i++) {
            x += positions[i][0];
            y += positions[i][1];
        }
        x /= positions.length;
        y /= positions.length;
        
        double d = dist(x, y, positions);

        double step = 100.0;

        boolean done = false;

        while ( step > 0.000001 )
        {
            done = false;
            for ( int i = 0; i < 4; ++i )
            {
                double nx = (double)x + step*dx[i];
                double ny = (double)y + step*dy[i];

                double t = dist(nx, ny, positions);

                if ( t < d )
                {
                    d = t;
                    x = nx;
                    y = ny;

                    done = true;
                    break;
                }
            }

            if (!done)
                step /= 2;
        }
        return d;
    }
    
    private double dist(double x, double y, int[][] positions) {
        double res = 0;
        for (int i = 0; i < positions.length; i++) {
            res += Math.sqrt((positions[i][0] - x)* (positions[i][0] - x) + (positions[i][1] - y) * (positions[i][1] - y));
        }
        
        return res;
    }
}
