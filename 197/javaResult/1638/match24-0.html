class Solution {
    public double getMinDistSum(int[][] positions) {
        double res = 0;
        double[] meanpnt = new double[2];
        
        for (int[] pos : positions) {
            meanpnt[0] += pos[0];
            meanpnt[1] += pos[1];
        }
        
        meanpnt[0] /= positions.length;
        meanpnt[1] /= positions.length;
        double minSum = sumPnts(positions, meanpnt);
        
        double breaklimit = 0.000001;
        double start = 300;
        double[][] dirs = {{-1, 0}, {0, -1}, {0, 1}, {1, 0}};
        
        
        while (start > breaklimit) {
            boolean found = false;
            for (double[] d : dirs) {
                double[] newpnt = new double[] {meanpnt[0] + start * d[1], meanpnt[1] + start * d[0]};
                double newSum = sumPnts(positions, newpnt);
                
                if (newSum < minSum) {
                    found = true;
                    minSum = newSum;
                    meanpnt = newpnt;
                    break;
                }
            }
            
            if (!found) {
                start /= 2;
            }
        }
        
        return minSum;
        
    }
    
    public double sumPnts(int[][] positions, double[] meanpnt) {
        double sum = 0;
        for (int[] pos : positions) {
            double xdiff = Math.abs(meanpnt[0] - pos[0]);
            double ydiff = Math.abs(meanpnt[1] - pos[1]);
            
            double diff = Math.sqrt(xdiff * xdiff + ydiff * ydiff);
            sum += diff;
        }
        return sum;
    }
}
