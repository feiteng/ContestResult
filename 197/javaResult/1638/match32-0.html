class Solution {
    double getDistanceSum(double x, double y, int[][] positions) {
        double ret = 0;
        for (int i = 0; i < positions.length; ++i) {
            double dx = positions[i][0] - x;
            double dy = positions[i][1] - y;
            ret += Math.sqrt(dx * dx + dy * dy);
        }
        return ret;
    }
    
    double getMedian(int[][] positions) {
        double x = 0;
        double y = 0;
        int n = positions.length;
        for (int i = 0; i < n; i++) {
            x += positions[i][0];
            y += positions[i][1];
        }
        x /= n;
        y /= n;
        double d = getDistanceSum(x, y, positions);
        int[] dx = {-1, 0, 1, 0};
        int[] dy = {0, 1, 0, -1};
        double step = 100.0;
        double eps = 1e-6;

        while (step > eps) {
            boolean done = false;
            for (int i = 0; i < 4; ++i) {
                double nx = (double) x + step * dx[i];
                double ny = (double) y + step * dy[i];
                double t = getDistanceSum(nx, ny, positions);
                if (t < d) {

                    d = t;
                    x = nx;
                    y = ny;

                    done = true;
                    break;
                }
            }
            if (!done)
                step /= 2;
        }

        return d;
    }

    public double getMinDistSum(int[][] positions) {
        int n = positions.length;
        if (n <= 1) return 0.0;
        return getMedian(positions);
    }

}
