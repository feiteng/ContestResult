class Solution {
    
public double getMinDistSum(int[][] p) {
        int l = p.length;
        
        double x = 0, y = 0;
        for (int i = 0; i < l; i++) {
            x += p[i][0];
            y += p[i][1];
        }
        
        x /= l;
        y /= l;
        double res = getSum(p, x, y);
        
        int dx[] = {-1, 0, 1, 0};
        int dy[] = {0, 1, 0, -1};
        double step = 100.0;

        
while (step > 0.000001) {
            boolean done = true;
            for (int d = 0; d < 4; d++) {
                double nx = x + step * dx[d];
                double ny = y + step * dy[d];
                double sum = getSum(p, nx, ny);
                if (sum < res) {
                    res = sum;
                    x = nx;
                    y = ny;
                    done = false;
                    break;
                }
            }
            if (done) {
                step /= 2;
            }
        }
        
        return res;
    }
    
    
    private double dis(double a, double b) {
        
return Math.sqrt(a * a + b * b);
    }
    
    private double getSum(int[][] p, double x, double y) {
        double sum = 0;
        int l = p.length;
        
for (int i = 0; i < l; i++) {
            sum += dis(p[i][0] - x, p[i][1] - y);
        }
        return sum;
    }
}
