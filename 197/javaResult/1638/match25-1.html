class Solution {
    
    /* Reference: https://en.wikipedia.org/wiki/Geometric_median */
    public double getMinDistSum(int[][] positions) {
        
        int[] dirs = {1, 0, -1, 0, 1};
        
        
double x = 0.0, y = 0.0;
        for (int[] p : positions) {
            x += p[0]; y += p[1];
        }
        
        x /= positions.length;
        y /= positions.length;
        
        double res = sumOfEuclidean(x, y, positions);
        double step = 10.0;  // point within 0 to 100
        double eps = 0.000001;
        
        while (step > eps) {
            
            boolean onRightTrack = false;
            
for (int i = 0; i < 4; ++i) {
                double nx = (double) x + step * dirs[i];
                double ny = (double) y + step * dirs[i + 1];
                
                double v = sumOfEuclidean(nx, ny, positions);
                if (v < res) {
                    res = v; x = nx; y = ny;
                    onRightTrack = true;
                    break;  // break it, since we are on the right track
                }
            }
            
            // keep increase our accuracy only if we on right track
            if (!onRightTrack)
                step /= 2.0;
        }
        
        return res;
    }
    
    private double sumOfEuclidean(double x, double y, int[][] positions) {
        double res = 0;
        for (int[] p : positions) {
            res += Math.sqrt(((p[0] - x) * (p[0] - x)) + ((p[1] - y) * (p[1] - y)));
        }
        return res;
    }
}
