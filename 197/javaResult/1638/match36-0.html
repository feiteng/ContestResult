class Solution {
    double distSum(double[] p, int[][] ps, int n) {
        double sum = 0;
        for (int i = 0; i < n; i++) {
            double disty = Math.abs(ps[i][0] - p[0]);
            double distx = Math.abs(ps[i][1] - p[1]);
            sum += Math.sqrt((distx * distx) + (disty * disty));
        }
        return sum;
    }

    double[][] dirs = {{-1, 0},{0, 1},{1, 0},{0, -1}};

    public double getMinDistSum(int[][] ps) {
        int n = ps.length;
        double[] curP = new double[2];
        for (int i = 0; i < n; i++) {
            curP[1] += ps[i][1];
            curP[0] += ps[i][0];
        }
        curP[1] /= n;
        curP[0] /= n;
        double minDis = distSum(curP, ps, n);
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n && i != k; i++) {
                double[] newpoint = {ps[i][1], ps[i][0]};
                double newd =
                        distSum(newpoint, ps, n);
                if (newd < minDis) {
                    minDis = newd;
                    curP[1] = newpoint[1];
                    curP[0] = newpoint[0];
                }
            }
        }
        double test_distance = 500;
        int flag = 0;
        while (test_distance > 1e-5) {
            flag = 0;
            for (int i = 0; i < 4; i++) {
                double[] newpoint = {curP[0] + test_distance * dirs[i][0], curP[1] + test_distance * dirs[i][1]};
                double newd = distSum(newpoint, ps, n);

                if (newd < minDis) {
                    minDis = newd;
                    curP[1] = newpoint[1];
                    curP[0] = newpoint[0];
                    flag = 1;
                    break;
                }
            }
            if (flag == 0)
                test_distance /= 2;
        }
        return minDis;
    }
}
