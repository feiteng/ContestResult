class Solution {
    public static final double err = 0.0000001;
    public static final double[][] dirs = new double[][]{{1.0, 0}, {-1.0, 0}, {0, 1.0}, {0, -1.0}};
    
    public double getMinDistSum(int[][] positions) {
        double x = 0.0, y = 0.0;
        int n = positions.length;
        
        for (int[] p : positions) {
            x += (double)(p[0]);
            y += (double)(p[1]);
        }
        
        x /= (n * 1.0);
        y /= (n * 1.0);
        
        double dist = dist(positions, x, y);
        double step = 1.0;
        
        while (step > err) {
            boolean flag = true;
            for (double[] dir : dirs) {
                double newx = x + step * dir[0], newy = y + step * dir[1];
                double newDist = dist(positions, newx, newy);
                
                if (newDist < dist) {
                    dist = newDist;
                    x = newx;
                    y = newy;
                    
                    flag = false;
                    break;
                }
            }
            
            if (flag) {
                step /= 2.0;
            }
        }
        
        return dist;
    }
    
    public double dist(int[][] positions, double x, double y) {
        double result = 0;
        
        for (int[] p : positions) {
            double dx = (double)(p[0]) - x;
            double dy = (double)(p[1]) - y;
            
            result += Math.sqrt(dx * dx + dy * dy);
        }
        
        return result;
    }
}
