class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        ArrayList< ArrayList< double[] > > graph = new ArrayList< ArrayList< double[] > >();
        for(int i = 0; i < n;i++){
            graph.add( new ArrayList< double[] >() );
        }
        for(int i = 0; i < edges.length;i++){
            int u = edges[i][0], v = edges[i][1];
            double prob = succProb[i];
            graph.get(u).add( new double[]{ v, prob } );
            graph.get(v).add( new double[]{ u, prob } );
        }
        double[] dp = new double[n];
        Arrays.fill(dp, 0);
        
        Queue< Integer > q = new LinkedList< Integer >();
        q.offer( start );
        dp[start] = 1;
        while( !q.isEmpty() ){
            int u = q.poll();
            for( double[] edge : graph.get(u) ){
                double v = edge[0], prob = edge[1];
                if( dp[(int)v] < dp[u] * prob){
                    dp[(int)v] = dp[u] * prob;
                    q.offer((int)v);
                }
            }
        }
        return dp[end] != 0 ? dp[end] : 0;
    }
}
