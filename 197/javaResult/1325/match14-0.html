import java.util.*;

class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<List<Object[]>> children = new ArrayList<>(n);
        for (int i = 0; i < n; i++) {
            children.add(new ArrayList<>());
        }
        for (int i = 0; i < edges.length; i++) {
            children.get(edges[i][0]).add(new Object[]{edges[i][1], succProb[i]});
            children.get(edges[i][1]).add(new Object[]{edges[i][0], succProb[i]});
        }

        Queue<Object[]> queue = new LinkedList<>();
        double[] next = new double[n];
        next[start] = 1;
        queue.add(new Object[]{start, 1.0});
        double ans = 0;
        while (!queue.isEmpty()) {
            Object[] current = queue.poll();
            if ((int)current[0] == end) {
                ans = Math.max(ans, (double)current[1]);
            }
            for (Object[] child : children.get((int)current[0])) {
                if ((double)child[1] * (double)current[1] > next[(int)child[0]]) {
                    queue.add(new Object[]{child[0], (double)child[1] * (double)current[1]});
                    next[(int)child[0]] = (double)child[1] * (double)current[1];
                }
            }
        }
        return ans;
    }
}
