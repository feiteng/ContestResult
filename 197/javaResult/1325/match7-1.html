import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {

    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        if (start == end)
            return 1.0;
        
List<Integer>[] g = new List[n];
        List<Double>[] gg = new List[n];
        for (int i = 0; i < n; i++) {
            g[i] = new ArrayList<>();
            gg[i] = new ArrayList<>();
        }
        
for (int i = 0; i < edges.length; i++) {
            int u = edges[i][0], v = edges[i][1];
            double w = succProb[i];
            g[u].add(v);
            gg[u].add(w);

            g[v].add(u);
            gg[v].add(w);
        }

        double[] fu = new double[n];
        fu[start] = 1.0;
        boolean[] pan = new boolean[n];
        Queue<Integer> queue = new ArrayDeque<>();
        queue.add(start);
        pan[start] = true;
        while (!queue.isEmpty()) {
            int u = queue.poll();
            pan[u] = false;

            
for (int i = 0; i < g[u].size(); i++) {
                int v = g[u].get(i);
                double w = gg[u].get(i);
                if (fu[u] * w > fu[v]) {
                    fu[v] = fu[u] * w;
                    if (!pan[v]) {
                        pan[v] = true;
                        queue.add(v);
                    }
                
}
            }
        }

        return fu[end];
    }
}
