class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {


        double[] pr = new double[n];
        pr[start] = 1;


        while (true) {
            boolean modified = false;
            for (int i = 0; i < succProb.length; i++) {
                int[] edge = edges[i];
                int v = edge[0], w = edge[1];
                double p = succProb[i];

                double pv = pr[v];
                double pw = pr[w];
                if (pv * p > pw) {
                    pr[w] = pv * p;
                    modified = true;
                }
                if (pw * p > pv) {
                    pr[v] = pw * p;
                    modified = true;
                }
            }
            if (!modified) {
                break;
            }
        }
        return pr[end];

    }
}
