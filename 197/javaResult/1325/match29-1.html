class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        
List<double[]>[] graph = new List[n];
        for(int i=0; i<n; i++){
            graph[i] = new ArrayList();
        }
        for(int i=0; i<edges.length; i++){
            int[] e = edges[i];
            double w = succProb[i];
            
graph[e[0]].add(new double[]{e[1], w});
            graph[e[1]].add(new double[]{e[0], w});
        }
        double[] p = new double[n];
        PriorityQueue<double[]> pq = new PriorityQueue<double[]>((a,b)->Double.compare(b[1],a[1]));
        p[start] = 1.0;
        
pq.offer(new double[]{start, 1.0});
        while(!pq.isEmpty()){
            double[] cur = pq.poll();
            int curNode = (int)cur[0];
            if(curNode==end){
                return cur[1];
            
}
            for(double[] next : graph[curNode]){
                int nextNode = (int)next[0];
                double w = next[1];
                if(w*cur[1] > p[nextNode]){
                    p[nextNode] = w*cur[1];
                    pq.offer(new double[]{nextNode, p[nextNode]});
                }
            }
        }
        return 0.0;
    }
}
