class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] dp=new double[n+1];
        List<Integer>[] edge=new List[n+1];
        Map<Integer,Double>[] p=new Map[n];//[n];//[]
        for(int i=0;i<n;i++){
            edge[i]=new ArrayList<>();
            p[i]=new HashMap<>();
        }
        for(int i=0;i<edges.length;i++){
            int a=edges[i][0];
            int b=edges[i][1];
            edge[a].add(b);
            edge[b].add(a);
            p[a].put(b,succProb[i]);
            p[b].put(a,succProb[i]);
        }
        Queue<Integer> que=new LinkedList<>();
        que.add(start);
        dp[start]=1;
        while(!que.isEmpty()){
            int u=que.poll();
            for(int nb:edge[u]){
                if(dp[u]*p[u].get(nb)>dp[nb]){
                    dp[nb]=dp[u]*p[u].get(nb);
                    que.add(nb);
                }
            }
        }
        return dp[end];
    }
}
