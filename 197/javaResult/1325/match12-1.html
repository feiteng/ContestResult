class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] prob = new double[n];
        prob[start] = 1;
        for (int i = 0; i < n; i++) {
            boolean keepGoing = false;
            for (int j = 0; j < edges.length; j++) {
                
int[] e = edges[j];
                if (prob[e[1]] < prob[e[0]] * succProb[j]) {
                    prob[e[1]] = prob[e[0]] * succProb[j];
                    keepGoing = true;
                }
                if (prob[e[0]] < prob[e[1]] * succProb[j]) {
                    prob[e[0]] = prob[e[1]] * succProb[j];
                    keepGoing = true;
                }
            }
            if (!keepGoing) {
                break;
            }
        }
        
        return prob[end];
    }
}
