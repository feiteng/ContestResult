class Solution {
class point{
        int loc;
        double pos;
        public point(int loc, double pos) {
            this.loc = loc;
            this.pos = pos;
        }
    }
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {

        PriorityQueue<point> pq = new PriorityQueue<point>((a,b)->(b.pos==a.pos?0:(b.pos>a.pos?1:-1)));
        //Double[][] map = new Double[n][n];
        HashMap<Integer,HashMap<Integer,Double>> map = new HashMap<>();
        for (int i=0;i<edges.length;i++){
            int[] e = edges[i];
            map.putIfAbsent(e[0],new HashMap<>());
            map.putIfAbsent(e[1],new HashMap<>());
            map.get(e[0]).put(e[1],succProb[i]);
            map.get(e[1]).put(e[0],succProb[i]);
        }
         //System.out.println(map);
        HashSet<Integer> set = new HashSet<>();
        //set.add(start);
        pq.add(new point(start,1.0));
        while (!pq.isEmpty()){
            point cur = pq.poll();
            //System.out.println(cur.loc+"  pos=="+ cur.pos);
            set.add(cur.loc);
            if (cur.loc == end) return cur.pos;
            if (!map.containsKey(cur.loc)) continue;
            for (int i: map.get(cur.loc).keySet()){
                if (set.contains(i)) continue;
                //set.add(i);
                pq.add(new point(i,cur.pos*map.get(cur.loc).get(i)));
                //System.out.println(i+"   "+cur.pos*map.get(cur.loc).get(i));
            }
        }
        return 0;
    }
}
