class Solution {
    
    static class Edge {
        int val;
        double prob;
        
        Edge(int v, double p) {
            val = v;
            prob = p;
        }
    }
    
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<Edge>[] adj = new List[n];
        for (int i = 0; i < n; i++) adj[i] = new ArrayList<>();
        
        for (int i = 0; i < edges.length; i++) {
            int[] e = edges[i];
            
            adj[e[0]].add(new Edge(e[1], succProb[i]));
            adj[e[1]].add(new Edge(e[0], succProb[i]));
        }
        
        PriorityQueue<Edge> q = new PriorityQueue<>(Comparator.comparingDouble(e -> -e.prob));
        q.add(new Edge(start, 1));
        boolean[] visited = new boolean[n];
        
        while (!q.isEmpty()) {
            Edge e = q.poll();
            if (visited[e.val]) continue;
            
            if (e.val == end) return e.prob;
            
            visited[e.val] = true;
            
            for (Edge con : adj[e.val]) {
                q.add(new Edge(con.val, con.prob*e.prob));
            }
        }
        
        return 0;
