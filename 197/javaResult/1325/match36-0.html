class Solution {
    private class State {
        int cur;
        double p;
        public State(int cur, double p) {
            this.cur = cur;
            this.p = p;
        }
    }
    
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        PriorityQueue<State> pq = new PriorityQueue<>(new Comparator<State>(){
            @Override
            public int compare(State s1, State s2) {
                if (s1.p == s2.p) return 0;
                return s1.p < s2.p ? 1 : -1;
            }
        });
        
        List<List<double[]>> graph = new ArrayList<>();
        for (int i = 0; i < n; i++) graph.add(new ArrayList<>());
        for (int i = 0; i < edges.length; i++) {
            graph.get(edges[i][0]).add(new double[]{edges[i][1], succProb[i]});
            graph.get(edges[i][1]).add(new double[]{edges[i][0], succProb[i]});
        }
        
        pq.offer(new State(start, 1.0));
        double[] visited = new double[n];
        Arrays.fill(visited, 0.0);
        visited[start] = 1;
        while (!pq.isEmpty()) {
            State s = pq.poll();
            if (s.cur == end) return s.p;
            // visited[s.cur] = s.p;
            for (double[] adj: graph.get(s.cur)) {
                int next = (int)adj[0];
                double prob = adj[1];
                if (s.p * prob > visited[next]) {
                    visited[next] = s.p * prob;
                    pq.offer(new State(next, s.p * prob));
                }
            }
        }
        
        return 0;
    }
}
