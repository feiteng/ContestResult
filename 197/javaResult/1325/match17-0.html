class Solution {
    
    class Node {
        int u;
        double p;
        Node (int u1, double p1) {
            u = u1;
            p = p1;
        }
    }
    
    public double maxProbability(int n, int[][] edges, double[] sp, int start, int end) {
        // double[] dist = new double[n+1];
        Map<Integer, Map<Integer, Double>> map = new HashMap<>();
        for (int i = 0; i < edges.length; i++) {
            int a = edges[i][0];
            int b = edges[i][1];            
            map.putIfAbsent(a, new HashMap<>());
            map.putIfAbsent(b, new HashMap<>());
            map.get(a).put(b, sp[i]);
            map.get(b).put(a, sp[i]);
        }
        
        double[] dist = new double[n+1];
        dist[start] = 1.0;
        PriorityQueue<Node> pq = new PriorityQueue<>((n1, n2) -> Double.compare(n1.p, n2.p) == 0 ? n1.u - n2.u : Double.compare(n2.p, n1.p));
        pq.offer(new Node(start, 1.0));
        
        while (!pq.isEmpty()) {
            Node node = pq.poll();
            if (node.p < dist[node.u])    continue;
            
            Map<Integer, Double> nbs = map.get(node.u);
            if (nbs == null || nbs.keySet() == null || nbs.keySet().isEmpty())  continue;
            for (Integer nb : nbs.keySet()) {
                if (dist[nb] < node.p * nbs.get(nb)) {
                    dist[nb] = node.p * nbs.get(nb);
                    pq.offer(new Node(nb, dist[nb]));
                }
            }
        }
        return dist[end];
    }
}
