class Solution {
    public double maxProbability(int n, int[][] edges, double[] p, int start, int end) {
        Map<Integer, Double>[] graph = new HashMap[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new HashMap<>();
        }
        
        int l = edges.length;
        for (int i = 0; i < l; i++) {
            int a = edges[i][0], b = edges[i][1];
            
graph[a].put(b, p[i]);
            graph[b].put(a, p[i]);
        }
        
        double[] state = new double[n];
        state[start] = 1.0;
        
        Queue<Integer> q = new LinkedList<>();
        q.offer(start);
        
        while (!q.isEmpty()) {
            int cur = q.poll();
            double pro = state[cur];
            for (int next : graph[cur].keySet()) {
                if (pro * graph[cur].get(next) > state[next]) {
                    q.offer(next);
                    state[next] = pro * 
graph[cur].get(next);
                }
            }
        }
        
        return state[end];
    }
}
