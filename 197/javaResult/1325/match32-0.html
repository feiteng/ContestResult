class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        ArrayList<Integer>[] chu = new ArrayList[n + 1];
        Map<Integer, Double> m = new HashMap<>();
        for (int i = 0; i <= n; i++) {
            chu[i] = new ArrayList<>();
        }
        for (int i = 0; i < edges.length; i++) {
            chu[edges[i][0]].add(edges[i][1]);
            chu[edges[i][1]].add(edges[i][0]);
            m.putIfAbsent(edges[i][0] * n + edges[i][1], succProb[i]);
            m.putIfAbsent(edges[i][1] * n + edges[i][0], succProb[i]);
        }
        ArrayDeque<Integer> ad = new ArrayDeque<>(n + 1);
        ad.addLast(start);
        while (!ad.isEmpty()) {
            int cur = ad.removeFirst();
            for (int i : chu[cur]) {
                double gailv = m.get(start * n + i) == null ? 0 : m.get(start * n + i);
                double fgail = start == cur ? 1 : m.get(start * n + cur), sgail = cur == i ? 1 : m.get(cur * n + i);
                if (cur == start || gailv < fgail * sgail) {
                    if (gailv < fgail * sgail)
                        m.put(start * n + i, fgail * sgail);
                    ad.addLast(i);
                }
            }
        }
        return m.get(start * n + end) == null ? 0.0 : m.get(start * n + end);
    }
}
