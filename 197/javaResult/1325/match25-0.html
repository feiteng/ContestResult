class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<List<Edge>> g= new ArrayList<>();
        for (int i =0; i<n;i++) {
            g.add(new ArrayList<>());
        }
        for (int i =0;i<edges.length;i++) {
            int[] edge = edges[i];
            g.get(edge[0]).add(new Edge(edge[1], succProb[i]));
            g.get(edge[1]).add(new Edge(edge[0], succProb[i]));
        }
        Double[] probs = new Double[n+1];
        PriorityQueue<Edge> heap = new PriorityQueue<Edge>(
                (info1, info2) -> -Double.compare(info1.prob, info2.prob));
        heap.offer(new Edge(start, 1));

        while (!heap.isEmpty()) {
            Edge info = heap.poll();
            double p = info.prob;
            int node = info.to;
            if (probs[node]!=null) {
                continue;
            }
            probs[node] = p;
            //if (dist.containsKey(node)) continue;
            //dist.put(node, d
            for (Edge edge: g.get(node)) {
                int nei = edge.to;
                double prob = probs[node] * edge.prob;
                System.out.println(nei+":"+prob);
                if (probs[nei]==null) {
                    heap.offer(new Edge(nei, prob));
                }
                //if (!dist.containsKey(nei))
                //    heap.offer(new int[]{d+d2, nei});
            }
        }
        if (probs[end] == null)
            return 0;
        return probs[end];
        
    }
    
    class Edge {
        int to;
        double prob;
        Edge(int to, double prob) {
            this.to=to;
            this.prob = prob;
        }
    }
    
}
