class Solution {
    public double maxProbability(int N, int[][] edges, double[] succProb, int start, int end) {
        double[] dp = new double[N];
        Map<Integer, Double>[] graph = new HashMap[N];
        for (int i = 0; i < N; i ++) {
            graph[i] = new HashMap();
        }
        for (int i = 0; i < edges.length; ++i) {
            graph[edges[i][0]].put(edges[i][1], succProb[i]);
            graph[edges[i][1]].put(edges[i][0], succProb[i]);
        }
        double ans = 0;
        Queue<Node> queue = new LinkedList();
        queue.offer(new Node(1.0, start));
        
        while(!queue.isEmpty()) {
            Node peek = queue.poll();
            if(peek.prob > dp[peek.index]) {
                dp[peek.index] = peek.prob;
                for (Map.Entry<Integer, Double> entry : graph[peek.index].entrySet()) {
                    int nei = entry.getKey();
                    double v = entry.getValue();
                    if(v*peek.prob > dp[nei]) {
                        queue.offer(new Node(v*peek.prob, nei));
                    }
                }
            }
        }
        
        return dp[end];
        
    }
    
    public class Node {
        double prob;
        int index;
        public Node(double prob, int index) {
            this.prob = prob;
            this.index = index;
        }
    }
}
