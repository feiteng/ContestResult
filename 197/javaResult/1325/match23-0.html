class Solution {
    class Pair{
        public int node;
        public double p;
        public Pair(int n, double p){
            this.node=n;
            this.p=p;
        }
    }
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        Map<Integer, List<Pair>> map = new HashMap<Integer, List<Pair>>();
        for (int i=0;i<edges.length;++i) {
            map.putIfAbsent(edges[i][0], new ArrayList<Pair>());
            map.get(edges[i][0]).add(new Pair(edges[i][1],succProb[i]));
            
            map.putIfAbsent(edges[i][1], new ArrayList<Pair>());
            map.get(edges[i][1]).add(new Pair(edges[i][0],succProb[i]));
        }
        
        double[] p=new double[n];
        Arrays.fill(p, 0.0);
        p[start]=1.0;
        PriorityQueue<Pair> nodes=new PriorityQueue<Pair>((a,b)->(Double.compare(b.p,a.p)));
        nodes.add(new Pair(start, p[start]));
        //for (int i=0;i<n;i++) nodes.add(new Pair(i, i==start?1.0:0.0));
        while (!nodes.isEmpty()) {
            Pair pair=nodes.poll();
            if(pair.node==end) return pair.p;
            if (pair.p==0.0) return 0.0;
            if (!map.containsKey(pair.node)) continue;
            for (Pair pa:map.get(pair.node)) {
                double tmp=p[pair.node]*pa.p;
                if (tmp>p[pa.node]){
                    p[pa.node]=tmp;
                    nodes.add(new Pair(pa.node, tmp));
                }
            }
        }
        return 0.0;
    }
}
