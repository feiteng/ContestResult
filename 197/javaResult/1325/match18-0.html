class Solution {
    double ans;
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double dis[] = new double[n];
        boolean vis[] = new boolean[n];
        Map<Integer,Double> map[] = new HashMap[n];
        for(int i=0;i<n;++i) map[i] = new HashMap();
        for(int i=0;i<edges.length;++i){
            int[] edge = edges[i];
            map[edge[0]].put(edge[1],succProb[i]);
            map[edge[1]].put(edge[0],succProb[i]);
        }
        Arrays.fill(dis,0.0);
        dis[start]= 1;
        PriorityQueue<Pair> queue = new PriorityQueue<>((a,b)->Double.compare(b.d,a.d));
        queue.add(new Pair(1.0,start));
        while(queue.size()!=0){
            Pair p = queue.remove();
            if(vis[p.v])continue;
            vis[p.v] = true;
            for(Map.Entry<Integer, Double> entry : map[p.v].entrySet()){
                int e = entry.getKey();
                double w = entry.getValue();
                if(dis[p.v]*w > dis[e]){
                    dis[e]=dis[p.v]*w;
                    queue.add(new Pair(dis[e],e));
                }
            }
        }
        return dis[end];
    }
    
    class Pair{
        int v;
        double d;
        public Pair(double d, int v) {
            this.v = v;
            this.d = d;
        }
    }
}
