import java.util.*;

class Solution {
  public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
    double[] prob = new double[n];
    prob[start] = 1;
    var map = new HashMap<Integer, List<To>>();
    for (int i = 0; i < edges.length; i++) {
      var e = edges[i];
      map.computeIfAbsent(e[0], k -> new ArrayList<>()).add(new To(e[1], succProb[i]));
      map.computeIfAbsent(e[1], k -> new ArrayList<>()).add(new To(e[0], succProb[i]));
    }
    var q = new LinkedList<Integer>();
    q.add(start);
    while (!q.isEmpty()) {
      var cur = q.poll();
      for (var to : map.getOrDefault(cur, Collections.emptyList())) {
        if (prob[to.to] < prob[cur] * to.p) {
          prob[to.to] = prob[cur] * to.p;
          q.add(to.to);
        }
      }
    }
    return prob[end];
  }

  static class To {
    double p;
    int to;

    To(int i, double prob) {
      to = i;
      p = prob;
    }
  }
}
