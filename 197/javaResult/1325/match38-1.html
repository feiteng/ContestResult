class Solution {
    
    class Node {
        int v;
        double p;
        Node(int _v, double _p) {
            v = _v;
            p = _p;
        }
    }
    
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        
        double[] probs = new double[n];
        
        
Map<Integer, List<Node>> map = new HashMap<>();
        for (int i = 0; i < edges.length; ++i) {
            int[] edge = edges[i];
            
            map.putIfAbsent(edge[0], new ArrayList<>());
            map.putIfAbsent(edge[1], new ArrayList<>());
            
            map.get(edge[0]).add(new Node(edge[1], succProb[i]));
            map.get(edge[1]).add(new Node(edge[0], succProb[i]));
        
}
        
        Queue<Node> queue = new LinkedList<>();
        queue.add(new Node(start, 1.0));
        
        while (!queue.isEmpty()) {
            
            Node node = queue.poll();
            int parent = node.v;
            double prob = node.p;
            
            for (Node child : map.getOrDefault(parent, new ArrayList<>())) {
                if (probs[child.v] >= prob * child.p) continue;
                queue.add(new Node(child.v, prob * child.p));
                probs[child.v] = prob * child.p;
            }
            
        }
        
        return probs[end];
    }
}
