class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] prob = new double[n];
        Arrays.fill(prob, 0);
        
        Map<Integer, Map<Integer, Double>> graph = new HashMap<>();
        for (int i = 0; i < n; i++) {
            graph.put(i, new HashMap<>());
        }
        for (int i = 0; i < edges.length; i++){
            int[] edge = edges[i];
            int e1 = edge[0];
            int e2 = edge[1];
            graph.get(e1).put(e2, succProb[i]);
            graph.get(e2).put(e1, succProb[i]);
        }
        
        prob[start] = 1;
        Queue<Integer> queue = new LinkedList<>();
        queue.add(start);
        while(!queue.isEmpty()) {
            int curr = queue.remove();
            /*if (curr == end) {
                return prob[curr];
            }*/
            
            Map<Integer, Double> map = graph.get(curr);
            for (int next : map.keySet()) {
                
                double toNext = map.get(next);
                double newp = prob[curr] * toNext;
                if (newp > prob[next]) {
                    prob[next] = newp;
                    queue.add(next);
                }
            }
            
        }
        return prob[end];
