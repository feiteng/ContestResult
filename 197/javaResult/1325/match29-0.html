class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] prob = new double[n];
        
        List<double[]>[] es = (List<double[]>[]) new List[n];
        for (int i = 0; i < n; i++) {
            es[i] = new ArrayList<>();
        }
        
        for (int i = 0; i < edges.length; i++) {
            int[] e = edges[i];
            es[e[0]].add(new double[] { e[1], succProb[i] });
            es[e[1]].add(new double[] { e[0], succProb[i] });
        }
        
        PriorityQueue<double[]> pq = new PriorityQueue<>((p1, p2) -> Double.compare(p2[1], p1[1]));
        pq.offer(new double[] { (double) start, 1.0 });
        
        while (!pq.isEmpty()) {
            double[] curr = pq.poll();
            int idx = (int) curr[0];
            if (curr[1] < prob[idx]) {
                continue;
            }
            
            // System.out.println(Arrays.toString(curr));
            for (double[] e : es[idx]) {
                int next = (int) e[0];
                double total = curr[1] * e[1];
                if (total <= prob[next]) {
                    continue;
                }
                
                // System.out.println(next + ", " + total);
                prob[next] = total;
                pq.offer(new double[] { next, total });
            }
        }
        
        return prob[end];
    }
}
