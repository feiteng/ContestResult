class Solution {
    
public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        Map<Integer, List<double[]>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.put(i, new ArrayList<>());
        }
        for (int i = 0; i < edges.length; i++) {
            int[] e = edges[i];
            
map.get(e[0]).add(new double[]{e[1], succProb[i]});
            map.get(e[1]).add(new double[]{e[0], succProb[i]});
        }
        double[] pro = new double[n];
        pro[start] = 1;
        
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>((i1, i2) -> Double.compare(pro[i2], pro[i1]));
        pq.add(start);
        //Set<Integer> seen = new HashSet<>();
        //seen.add(start);
        
while (pq.size() > 0) {
            int cur = pq.poll();
            if (cur == end) return pro[end];
            for (double[] d: map.get(cur)) {
                int nei = (int)d[0];
                if (pro[cur] * d[1] > pro[nei]) {

                    pro[nei] = pro[cur] * d[1];
                    pq.add(nei);
                
}
            }
        }
        return 0;
    }
}
