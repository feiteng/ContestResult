class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] maxProbs = new double[n];
        Deque<Integer> queue = new ArrayDeque<>();
        Map<Integer, List<Pair<Integer, Double>>> neighborMap = new HashMap<>();
        for (int i = 0; i < edges.length; i++) {
            int[] edge = edges[i];
            double prob = succProb[i];
            
int a = edge[0], b = edge[1];
            neighborMap.computeIfAbsent(a, l -> new ArrayList<>()).add(new Pair<>(b, prob));
            neighborMap.computeIfAbsent(b, l -> new ArrayList<>()).add(new Pair<>(a, prob));
        }
        maxProbs[start] = 1.0;
        
queue.add(start);
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            for (Pair<Integer, Double> pair : neighborMap.getOrDefault(cur, new ArrayList<>())) {
                int next = pair.getKey();
                double prob = pair.getValue();
                if (maxProbs[cur]*prob > maxProbs[next]) {
                    maxProbs[next] = maxProbs[cur]*prob;
                    if (next != end) {
                        queue.add(next);
                    }
                
}
            }
        }
        return maxProbs[end];
    }
}
