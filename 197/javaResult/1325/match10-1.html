class Status {
    double p;
    int node;

    public Status(double p, int node) {
        this.p = p;
        this.node = node;
    }
}


class Solution {
    
public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<Map<Integer, Double>> graph = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            graph.add(new HashMap<>());
        }
        for (int i = 0; i < edges.length; i++) {
            int[] edge = edges[i];
            
graph.get(edge[0]).put(edge[1], succProb[i]);
            graph.get(edge[1]).put(edge[0], succProb[i]);
        }
        Queue<Status> q = new PriorityQueue<>((Comparator.comparingDouble(o -> -o.p)));
        Set<Integer> visit = new HashSet<>();
        q.add(new Status(1, start));
        while (!q.isEmpty()) {
            Status s = 
q.poll();
            if (s.node == end) {
                return s.p;
            }
            if (!visit.add(s.node)) {
                continue;
            }
            for (int key :
                    graph.get(s.node).keySet()) {
                if (!visit.contains(key)) 
{
                    q.add(new Status(graph.get(s.node).get(key) * s.p, key));
                }
            }
        }
        return 0;
    }
}
