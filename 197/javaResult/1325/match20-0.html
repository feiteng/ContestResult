class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
            List<Map<Integer, Double>> g = new ArrayList<>();
            for (int i = 0; i < n; i++) {
                g.add(new HashMap<>());
            }
            for (int i = 0; i < edges.length; i++) {
                g.get(edges[i][0]).put(edges[i][1], succProb[i]);
                g.get(edges[i][1]).put(edges[i][0], succProb[i]);
            }
            Set<Integer> visited = new HashSet<>();
            PriorityQueue<double[]> q = new PriorityQueue<>((x, y) -> Double.compare(y[1], x[1]));
            q.offer(new double[]{start, 1});
            while (!q.isEmpty()){
                double[] top = q.poll();
                int cur = (int)top[0];
                double weight = top[1];
                if (cur == end) {
                    return weight;
                }
                Map<Integer, Double> neis = g.get(cur);
                if (!visited.add(cur) || neis.size() == 0) {
                    continue;
                }
                for (int nei : neis.keySet()) {
                    q.offer(new double[] {nei, weight * neis.get(nei)});
                }
            }
            return 0;
    }
}
