class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<Integer>[] graph = new List[n];
        List<Double>[] probabilities = new List[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new ArrayList<>();
            probabilities[i] = new ArrayList<>();
        }

        int m = edges.length;
        for (int i = 0; i < m; i++) {
            int a = edges[i][0];
            int b = edges[i][1];
            double p = succProb[i];

            graph[a].add(b);
            probabilities[a].add(p);
            
            graph[b].add(a);
            probabilities[b].add(p);
        }

        double[] dp = new double[n];
        dp[start] = 1;

        Queue<Integer> queue = new LinkedList<>();
        queue.add(start);

        while (!queue.isEmpty()) {
            Integer a = queue.poll();
            for (int i = 0; i < graph[a].size(); i++) {
                Integer b = graph[a].get(i);
                double p = probabilities[a].get(i);

                if (dp[b] < dp[a] * p) {
                    dp[b] = dp[a] * p;
                    queue.add(b);
                }
            }
        }

        return dp[end];
    }
}
