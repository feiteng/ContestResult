class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<Pair>[] adjList = new List[n];
        for (int i = 0; i < n; i++) {
            adjList[i] = new ArrayList();
        }                                                                                                               
        for (int i = 0; i < edges.length; i++) {
            int u = edges[i][0], v = edges[i][1];
            adjList[u].add(new Pair(v, succProb[i]));
            adjList[v].add(new Pair(u, succProb[i]));
        }
        double[] prob = new double[n];
        prob[start] = 1.0;
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>((a, b) -> {
            if (prob[b] > prob[a]) {
                return 1;
            } else if (prob[a] > prob[b]) {
                return -1;
            } else {
                return 0;
            }
        });
        pq.offer(start);
        while (!pq.isEmpty()) {
            int cur = pq.poll();
            if (cur == end) {
                return prob[end];
            }
            for (Pair neigh : adjList[cur]) {
                if (prob[neigh.node] < prob[cur] * neigh.prob) {
                    prob[neigh.node] = prob[cur] * neigh.prob;
                    pq.offer(neigh.node);
                }
            }
        }
        return 0.0;
    }
    
    class Pair{
        int node;
        double prob;
        
        Pair(int node, double prob) {
            this.node = node;
            this.prob = prob;
        }
    }
}
