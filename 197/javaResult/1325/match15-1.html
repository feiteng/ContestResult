class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        Map<Integer, List<Pair<Integer, Double>>> m = new HashMap<>();
        boolean[][] visit = new boolean[n][n];
        
for (int i = 0; i < edges.length; i++) {
            int[] edge = edges[i];
            List<Pair<Integer, Double>> l = m.getOrDefault(edge[0], new ArrayList<>());
            l.add(new Pair(edge[1], succProb[i]));
            m.put(edge[0], l);
            l = m.getOrDefault(edge[1], new ArrayList<>());
            l.add(new Pair(edge[0], succProb[i]));
            m.put(edge[1], l);
        
}
        
        Queue<Pair<Integer, Double>> q = new PriorityQueue<Pair<Integer, Double>>((a, b) -> {
        if (b.getValue() < a.getValue()) {
            return -1;
        } else if (b.getValue() == a.getValue()) {
            return 0;
        } else {
            return 1;
        }});
        q.add(new Pair(start, 1.0));
        
        while (!q.isEmpty()) {
            Pair<Integer, Double> node = q.poll();
            int cur = node.getKey();
            double pos = node.getValue();
            if (cur == end) {
                return pos;
            }
            List<Pair<Integer, Double>> l = m.getOrDefault(cur, new ArrayList<>());
            
for (Pair<Integer, Double> each : l) {
                if (!visit[cur][each.getKey()]) {
                    q.add(new Pair(each.getKey(), pos * each.getValue()));
                    visit[cur][
each.getKey()] = true;
                }
            }
        }
        
        return 0.0;
    }
}
