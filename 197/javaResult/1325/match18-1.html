class Solution {
    
class Edge {
        int v;
        double w;
        Edge(int _v, double _w) {
            v = _v;
            w = _w;
        }
    }
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        
List<Edge>[] g = new List[n];
        for (int i = 0; i < n; i++) g[i] = new ArrayList<>();
        for (int i = 0; i < edges.length; i++) {
            int u = edges[i][0];
            int v = edges[i][1];
            double w = succProb[i];
            g[u].add(new Edge(v, w));
            g[v].add(new Edge(u, w));
        }
        double[] d = new double[n];
        d[start] = 1.0;
        PriorityQueue<Edge> pq = new PriorityQueue<Edge>((a, b) -> b.w > a.w ? 1 : b.w == a.w ? 0 : -1);
        
pq.add(new Edge(start, 1.0));
        while (!pq.isEmpty()) {
            Edge e = pq.remove();
            if (e.w != d[e.v]) continue;
            for (Edge s : g[e.v]) {
                
if (d[e.v] * s.w > d[s.v]) {
                    d[s.v] = d[e.v] * s.w;
                    pq.add(new Edge(s.v, d[s.v]));
                }
            }
        }
        
        return d[end];
        
    }
}
