class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<Integer>[] node = new List[n];
        for (int i = 0; i < n; i++) {
            node[i] = new ArrayList<>();
        }
        for (int i = 0; i < edges.length; i++) {
            
node[edges[i][0]].add(i);
            node[edges[i][1]].add(i);
        }
        double[] rate = new double[n];
        rate[start] = 1.0;
        Queue<Integer> queue = new ArrayDeque<>();
        queue.offer(start);
        while (!queue.isEmpty()) {
            int now = queue.poll();
            for (int edgeIndex : node[now]) {
                int next = edges[edgeIndex][0] == now ? edges[edgeIndex][1] : edges[edgeIndex][0];
                double nextRate = rate[now] * succProb[edgeIndex];
                
if (nextRate > rate[next]) {
                    rate[next] = nextRate;
                    queue.offer(next);
                }
            }
        }
        return rate[end];
    }
}
