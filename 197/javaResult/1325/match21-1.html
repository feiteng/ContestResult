class Solution {
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        double[] prob = new double[n];
        Arrays.fill(prob, .0);
        
List<pat>[] g = new ArrayList[n];
        for (int i = 0; i < n ; ++i) {
            g[i] = new ArrayList<>();
        }
        for (int i = 0; i < edges.length; ++i) {
            g[edges[i][0]].add(new pat(edges[i][1], succProb[i]));
            g[edges[i][1]].add(new pat(edges[i][0], succProb[i]));
        }
        
        
PriorityQueue<pat> q = new PriorityQueue<>();
        q.add(new pat(start, 1.0));
        while (!q.isEmpty()) {
            pat tmp = q.poll();
            int v = tmp.v;
            double p = tmp.p;
            if (prob[v] > p) continue;
            prob[v] = p;
            for (pat nei : g[v]) {
                if (prob[nei.v] > 0.0) continue;
                q.add(new pat(nei.v, p * nei.p));
            }
        
}
        return prob[end];
    }
    
    private class pat implements Comparable<pat>{
        int v;
        double p;
        public pat(int v, double p) {
            this.v = v;
            this.p = p;
        }
        
        public int compareTo(pat that) {
            return Double.compare(that.p, this.p);
        }
    }
}
