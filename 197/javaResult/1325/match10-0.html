class Solution {
    class Node implements Comparable<Node> {
        int node;
        double cost;

        public Node(int node, double cost) {
            this.node = node;
            this.cost = cost;
        }

        @Override
        public int compareTo(Node o) {
            return Double.compare(o.cost, cost);
        }
    }
    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        Map<Integer, Map<Integer, Double>> edgeMap = new HashMap<>();

        for (int i = 0; i < edges.length; i++) {
            int[] edge = edges[i];
            double prob = succProb[i];
            if (!edgeMap.containsKey(edge[0])) {
                edgeMap.put(edge[0], new HashMap<>());
            }
            if (!edgeMap.containsKey(edge[1])) {
                edgeMap.put(edge[1], new HashMap<>());
            }
            edgeMap.get(edge[0]).put(edge[1], prob);
            edgeMap.get(edge[1]).put(edge[0], prob);
        }

        PriorityQueue<Node> q = new PriorityQueue<>();
        q.add(new Node(start, 1));
        Set<Integer> visited = new HashSet<>();
        visited.add(start);

        while (!q.isEmpty()) {
            Node curr = q.poll();
            visited.add(curr.node);
            if (curr.node == end) {
                return curr.cost;
            }
            if (!edgeMap.containsKey(curr.node)) {
                continue;
            }
            for (Map.Entry<Integer, Double> entry : edgeMap.get(curr.node).entrySet()) {
                if (!visited.contains(entry.getKey())) {
                    q.add(new Node(entry.getKey(), curr.cost * entry.getValue()));
                }
            }
        }
        return 0;
    }
}
