class Solution {
    
static class Point {
        int end;
        double prob;

        Point(int end, double prob) {
            this.end = end;
            this.prob = prob;
        }
    }
    
    
public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        Map<Integer, List<Point>> g = new HashMap<>();

        for (int i = 0; i < edges.length; i++) {
            
g.computeIfAbsent(edges[i][0], v -> new ArrayList<>()).add(new Point(edges[i][1], succProb[i]));
            g.computeIfAbsent(edges[i][1], v -> new ArrayList<>()).add(new Point(edges[i][0], succProb[i]));
        }

        PriorityQueue<Point> pq = new PriorityQueue<>((a, b) -> Double.compare(b.prob, a.prob));
        pq.offer(new Point(start, 1.0));
        boolean[] visited = new boolean[n+1];
        
while (!pq.isEmpty()) {
            Point remove = pq.remove();
            if(remove.end == end) {
                return remove.prob;
            }
            if(visited[remove.end]) {
                continue;
            }
            visited[remove.end] = true;
            for (Point next : g.getOrDefault(remove.end, Collections.emptyList())) {
                pq.offer(new Point(next.end, remove.prob*next.prob));
            }
        }
        return 0;
    }

}
