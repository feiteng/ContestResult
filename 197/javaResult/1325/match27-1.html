class Solution {
    
public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        ArrayList<ArrayList<pair>> list=new ArrayList<>();
        for(int i=0;i<n;i++)
        {
            list.add(new ArrayList<>());
        }
        double time[]=new double[n];
        
for(int i=0;i<edges.length;i++)
        {
            int aa[]=edges[i];
            list.get(aa[0]).add(new pair(succProb[i],aa[1]));
            list.get(aa[1]).add(new pair(succProb[i],aa[0]));
        }
        PriorityQueue<pair> q=new PriorityQueue<>(new cust());
        q.add(new pair(1,start));
        time[start]=1;
        while(!q.isEmpty())
        {
            pair p=q.poll();
            ArrayList<pair> li=list.get(p.b);
            for(pair i:li)
            {
                if(i.a*p.a-time[i.b]>0.000005)
                {
                    time[i.b]=i.a*p.a;
                    
q.add(new pair(time[i.b],i.b));
                }
            }
        }
        return time[end];
    }
    
public class pair
    {
        double a;int b;
        pair(double a,int b)
        {
            this.a=a;
            this.b=b;
        }
    }
    public class cust implements Comparator<pair>
    {
        public int compare(pair p1,pair p2)
        {
            return Double.compare(p1.a,p2.a);
        }
    }
}
