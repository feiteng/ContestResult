class Solution {
    
public int numSub(String s) {
        //int[] dp = new int[s.length()];
        int ones = 0; 
        int res = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '1') {
                ones++;
                res+= ones;
                res = res % (1000000000 + 7);
            } else {
                ones = 0;
            }
        }
        return res;
    }
}
