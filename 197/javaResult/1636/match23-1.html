class Solution {
    public int numSub(String s) {
        long res = 0;
        long acc = 0;
        int M = 1000000000 + 7;
        for(char x : s.toCharArray()) {
            
if(x == '0') {
                acc = 0;
            } else {
                res += acc + 1;
                acc = acc + 1;
                acc %= M;
                res %= M;
            }
        }
        return (int) res;
    }
}
