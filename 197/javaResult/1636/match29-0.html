class Solution {
    public int numSub(String s) {
      long cc = 0, res = 0;
      for(int i=0; i<s.length(); i++) {
        if(s.charAt(i) == '0') {
          res += ((cc*(cc+1))/2);
          cc = 0;
        } else {
          cc++;
        }
        res %= 1000000007;
      }
      res += ((cc*(cc+1))/2);
      res %= 1000000007;
      return (int) res;
    }
  }
