class Solution {
    int mod = (int)1e9 + 7;
    
public int numSub(String s) {
        int len = 0;
        long ans = 0;
        for(char c : s.toCharArray()){
            if(c == '0')
{
                ans += len + (long)(len - 1) * len / 2; 
                len = 0;
            }else{
                len++;
            }
        }
        ans += len + (long)(len - 1) * len / 2; 
        return (int)(ans % mod);
    }
}
