class Solution {
    public int numSub(String s) {
        s += '0';
        int len = 0, res = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                len++;
            } else {
                res += (int)((long)len * ((long)len + 1) / 2 % 1000000007);
                res %= 1000000007;
                len = 0;
            }
        }
        return res;
    }
}
