class Solution {
    public int numSub(String s) {
        long res = 0;
        long len = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                len++;
            } else {
                long sum = (1 + len) * len / 2 % 1000000007;
                
res = (res + sum) % 1000000007;
                len = 0;
            }
            
        }
        res = (res + (1 + len) * len / 2 % 1000000007) % 1000000007;
        return (int)res;   
    }
}
