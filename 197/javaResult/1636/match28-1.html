class Solution {
    public int numSub(String s) {
        long out = 0;
        int ones = 0;
        for (char c : s.toCharArray()) {
            if (c == '1') {
                ones++;
                out += ones;
            
} else {
                ones = 0;
            }
        }
        return (int) (out % 1000000007);
    }
}
