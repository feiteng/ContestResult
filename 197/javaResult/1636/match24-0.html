class Solution {
    int mod = 1_000_000_007;
    public int numSub(String s) {
        int j=0;
        long result = 0;
        for(int i=0;i<s.length();i++){
            char c = s.charAt(i);
            if(c=='0'){
                j=i+1;
            }else{
                result=(result+(i-j+1))%mod;
            }
        }
        return (int)result;
        
    }
}
