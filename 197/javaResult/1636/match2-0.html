class Solution {
    public int numSub(String s) {
        int MOD = (int)(1e9 + 7);
        long ans = 0;
        int left = 0;
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) == '1'){
                //System.out.println
                ans = (ans + (i - left + 1)) % MOD ;
            }else {
                left = i+1;
            }
        }
        return (int)ans;
    }
}
