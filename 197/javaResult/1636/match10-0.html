class Solution {
    public int numSub(String s) {
        int mod = 1000000007;
        long res = 0;
        int l = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '0') {
                l = i + 1;
            } else {
                res = (res + (i - l + 1)) % mod;
            }
        }
        return (int)(res);
    }
}
