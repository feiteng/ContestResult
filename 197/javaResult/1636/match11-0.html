class Solution {
    public int numSub(String s) {
        int mod = 1000000007;
		int res = 0;
		int lastZeroIndex = -1;
		for (int i = 0; i < s.length(); i++) {
			if (s.charAt(i) == '1') {
				res += i - lastZeroIndex;
				if (res > mod) {
					res -= mod;
				}
			} else {
				lastZeroIndex = i;
			}
		}
		return res;
    }
}
