class Solution {
    public int numSub(String s) {
        int mod = (int) 1e9 + 7;
        int res = 0;
        int cnt = 0;
        
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') 
{
                cnt++;
                res = (res + cnt) % mod;
            } else {
                cnt = 0;
            }
            
        }
        return res;
    }
}
