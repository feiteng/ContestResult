class Solution {
    public int numSub(String s) {
        int count = 0;
        int pre = 0;
        for (char c : s.toCharArray()) {
            if (c == '0') {
                pre = 0;
            } else {
                pre++;
                count = (count + pre) % 1_000_000_007;
            }
        }
        return count;
    }
}
