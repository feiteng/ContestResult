class Solution {
    
    private static final int WRAP = 1_000_000_007;
    
    
public int numSub(String s) {
        
        int start = -1;
        int numSubs = 0;
        for (int i = 0; i < s.length(); i += 1) {
            if (s.charAt(i) == '0') {
                start = i;
            
} else {
                numSubs = (numSubs + i - start) % WRAP;
            }
        }
        return numSubs;
    }
}
