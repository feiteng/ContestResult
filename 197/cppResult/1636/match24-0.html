#define NUM 1000000007

class Solution {
public:
    long long findPower(long long n){
        return(((n*((n+1)%NUM))%NUM)/2);
    }
    
    int numSub(string s) {
        long long ans = 0;
        long long count = 0;
        for(char c : s){
            if(c=='0'){
                ans = (ans +findPower(count))%NUM;
                count = 0;
            }
            else{
                count++;
            }
        }
        ans = (ans +findPower(count))%NUM;
        return(ans);
    }
};
