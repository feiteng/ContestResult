class Solution {
public:
    int numSub(string a) {
        int n = a.length();
        long long ans = 0;
        long long mod = 1e9 + 7;
        for(int i = 0; i < n; ) {
            if(a[i] == '0') {
                i++;
                continue;
            }
            int j = i;
            while(j < n && a[j] == '1')
                j++;
            long long len = j - i;
            long long temp = (len * (len + 1)) / 2;
            ans = (ans + temp ) % mod;
            i = j;
        }
        return (ans + mod) % mod;
    }
    
};
