class Solution {
public:
    int numSub(string s) {
        int res = 0;
        int mod = 1000000007;
        int p = 0;
        int prev = 0;
        for (int i = 0; i <= s.size(); i++) {
            if (i == s.size() || s[i] == '0') {
                if (prev == 1) {
                    long long n = i - p;
                    long long total = n * (n + 1) / 2;
                    res = (total + res) % mod;
                }
                p = i;
                prev = 0;
            } if (s[i] == '1') {
                if (prev == 0) {
                    p = i;
                }
                prev = 1;
            }
        }
        return res;
    }
};
