class Solution {
    const int MOD = 1e9+7;
public:
    int numSub(string s) {
        long long ans = 0;
        int cnt = 0;
        int n = s.size();
        
        for(int i = 0; i < n; i++) {
            if(s[i] == '1') cnt++;
            else {
                ans += ((long long)cnt*(cnt+1))/2;
                cnt = 0;
            }
        }
        ans += ((long long)cnt*(cnt+1))/2;
        
        return ans%MOD;
    }
};
