class Solution {
public:
    int numSub(string s) {
        int mod = 1e9 + 7;
        s += '0';
        long l = 0;
        long ans = 0;
        for (char c: s) {

            if (c == '0') {
                if (l != 0) ans += (l + 1) * l / 2;
                ans %= mod;
                l = 0;
            }
            else l++;
        }
        return ans;
    }
};
