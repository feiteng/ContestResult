class Solution {
public:
    int mod = 1e9+7;
    int numSub(string s) {
        int n=s.size();
        int cnt=0;
        long long ret = 0;
        for(int i=0;i<s.size();++i) {
            if(s[i]=='1') {
                cnt++;
            } else {
                ret += 1ll * cnt * (cnt+1) / 2;
                ret %= mod;
                cnt = 0;
            }
        }
        ret += 1ll * cnt * (cnt+1) / 2;
        ret %= mod;
        return ret;
    }
};
