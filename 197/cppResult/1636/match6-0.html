typedef long long LL;
const LL MOD = 1000000007;
class Solution {
public:
    int numSub(string s) {
        int n = s.size();
        LL tot = 0;
        LL ans = 0;
        for (int i = 0; i < n; i++) {
            if (s[i] == '0') {
                ans = (ans + (tot) * (tot + 1LL) / 2LL) % MOD;
                tot = 0;
            }
            else tot = (tot + 1LL);
        }
        ans = (ans + (tot) * (tot + 1LL) / 2LL) % MOD;
        return ans;
    }
};
