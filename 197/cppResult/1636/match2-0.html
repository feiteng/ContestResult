class Solution {
public:
    int numSub(string s) {
        
        int l = -1, r = 0;
        long cnt = 0;
        long mod = pow(10, 9) + 7;
        for (r = 0; r < s.size(); r++) {
            if (s[r] == '1') {
                if (l == -1) l = r;
                cnt += (r - l + 1);
                cnt %= mod;
            } else l = -1;
        }
        
        return (int)cnt;
    }
};
