class Solution {
public:
    int numSub(string s) {
        int ans = 0;
        int cur = 0;
        int N = s.size();
        int MOD = 1e9 + 7;
        for (int i=0; i<N; i++) {
            if (s[i] == '0') cur = 0;
            else cur++;
            ans += cur;
            if (ans >= MOD) ans -= MOD;
        }
        return ans;
    }
};
