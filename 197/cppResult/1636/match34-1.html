const int MOD = 1000000007;
class Solution {
public:
    int numSub(string s) {

        int ret = 0;
        int one = 0;
        for(int i = 0; i < sz(s); i++) {
          if(s[i] == '0') one = 0;
          else one++;
          ret += one;
          ret %= MOD;
        }
        return ret;
    }
};
