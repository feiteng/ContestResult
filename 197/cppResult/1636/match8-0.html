class Solution {
public:
    int numSub(string s) {
      long long cnt = 0;
      long long c = 0;
      long long kMod = (1e9)+7;
      for (char ch : s) {
        if (ch == '0') {
          cnt = (cnt + c*(c+1)/2)% kMod;
          c = 0;
        } else ++c;
      }
      cnt = (cnt+c * (c+1)/2) % kMod;
      return cnt;
    }
};
