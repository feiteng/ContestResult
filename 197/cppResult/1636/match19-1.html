class Solution {
public:
    const int modnum = 1e9+7;
    int numSub(string s) {

        int cur = 0, ans = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == '1') {
                cur++;
                ans = (ans + cur) % modnum;
            } else {
                cur = 0;
            }
        }
        return ans % modnum;
    }
};
