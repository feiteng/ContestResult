class Solution {
public:
    int numSub(string s) {

        long long ret = 0, count = 0, mod = 1e9 + 7;
        for (int i = 0; i < s.length(); i++)
        {
            if (s[i] == '1') count++;
            else
            {
                ret = (ret + count * (count + 1) / 2) % mod;
                count = 0;
            }
        }
        ret = (ret + count * (count + 1) / 2) % mod;
        return ret;
    }
};
