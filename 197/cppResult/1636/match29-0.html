class Solution {
public:
    int numSub(string s) {
        long long cnt=0,i,one=0;
        for(i=0;i<s.length();i++)
        {
            if(s[i]=='0')
            {
                //cout<<one<<"\n";
                cnt=(cnt+(one*(one+1))/2)%1000000007;
                one=0;
            }
            else
            {
                one++;
            }
        }
        cnt=(cnt+(one*(one+1))/2)%1000000007;
        return cnt;
    }
};
