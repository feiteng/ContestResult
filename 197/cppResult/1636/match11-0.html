class Solution {
public:
    int numSub(string s) {
        long long tot = 0;
        long long cur = 0;
        for(int i=0; i<s.size(); i++){
            if(s[i] == '0'){
                tot += (cur * (cur+1))/2;
                tot%=1000000007;
                cur = 0;
            }
            else{
                cur++;
            }
        }
        tot += (cur * (cur+1))/2;
        tot%=1000000007;
            return tot;
    }
};
