class Solution {
public:
    int numSub(string s) {
        long long ans = 0;

        long long M = 1e9 + 7;
        int n = s.length();
        long long ctr = 0;
        for (int i = 0; i < n; ++i) {
            if (s[i] == '0') {
                ans += (ctr * (ctr+1))/2ll;
                ans %= M;
                ctr = 0;
            } else {
                ++ctr;
            }
        }
        ans += (ctr * (ctr+1))/2ll;
        ans %= M;
        return ans;
    }
};
