class Solution {
public:
    int MO = 1e9 + 7;
    int get(int now) {
        return (long long)now * (now + 1) / 2 % MO;
    }
    
    int numSub(string s) {
        int ans = 0, n = s.size();
        int now = 0;
        for (int i = 0; i < n; i++) {
            if (s[i] == '1') {
                now++;
            } else {
                (ans += get(now)) %= MO;
                now = 0;
            }
        }
        if (now > 0) {
            (ans += get(now)) %= MO;
        }
        return ans;
    }
};
