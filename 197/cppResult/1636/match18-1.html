int mod = (1e9 + 7);

class Solution 
{
public:
    int numSub(string s)
    {

        long long cnt = 0, ret = 0;
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] == '1')
            {
                cnt++;
            }
            else
            {
                ret += (1 + cnt) * cnt / 2;
                ret %= mod;
                cnt = 0;
            }
        }
        
        ret += (1 + cnt) * cnt / 2;
        ret %= mod;
        return ret;
    }
};
