#define ll long long
#define mod 1000000007
class Solution {
public:
    int numSub(string s) {
        ll t,i,j,k=0,n=s.length(),m=0;
        for(auto c:s){
            if(c=='0')k=0;
            else k++;
            m+=k;
            m%=mod;
        }
        return m;
    }
};
