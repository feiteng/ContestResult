class Solution {
public:
    int numSub(string s) {
        int ans = 0, num = 0, MD = 1e9 + 7;
        for(int i = 0; i < s.size(); i++) {
            if(s[i] == '0') {
                ans = (ans + 1LL * num * (num + 1) / 2 % MD) % MD;
                num = 0;
            }else {
                num++;
            }
        }
        if(num) {
            ans = (ans + 1LL * num * (num + 1) / 2 % MD) % MD;
        }
        return ans;
    }
};
