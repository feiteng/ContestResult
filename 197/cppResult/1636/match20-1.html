class Solution {
public:
    int numSub(string s) {
        int start = -1;
        int res = 0;
        int mod = 1e9 + 7;
        for (int i = 0; i < s.size(); ++i)
        {
            if (s[i] == '0') 
            {
                start = i;
            }
            if ((start == -1 || s[start] == '0') && s[i] == '1') start = i;
            if (start >= 0 && s[start] == '1')
            res += i - start + 1;
            res %= mod;
        }
        return res;
    }
};
