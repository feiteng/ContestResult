class Solution {
public:
    int numSub(string s) {
        const int mod = 1000000000 + 7;
        int res = 0;
        int cnt = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '0') cnt = 0;
            else cnt++;
            res += cnt;
            res %= mod;
        }
        return res;
    }
};
