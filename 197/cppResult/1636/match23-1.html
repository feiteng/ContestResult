class Solution {
public:
    int numSub(string s) {
        long long int ans=0;
        int n=s.size();

        long long int a=0;
        for(int i=0;i<n;++i)
        {
            if(s[i]=='0')
            {
                ans=((ans)+(a*(a+1)/2)%1000000007)%1000000007;
                a=0;
            }
            else{
                a++;
            }
        }
        ans=((ans)+(a*(a+1)/2)%1000000007)%1000000007;
        return ans;
    }
};
