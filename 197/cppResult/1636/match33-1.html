class Solution {
public:
    int numSub(string s) {
        int n = s.size();

        long cnt = 0, res = 0, val = 1e9+7;
        for (int i = 0; i < n; i++) {
            if (s[i] == '1') {
                cnt++;
            } else {
                res += (cnt * (cnt + 1 ) / 2) % val;
                res %= val;
                cnt = 0;
            }
        }
        if (cnt != 0) {
            res += (cnt * (cnt + 1) / 2) % val;
            res %= val;
        }
        return res;
    }
};
