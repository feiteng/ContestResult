class Solution {
public:
  int numSub(string s) {
    long long total = 0;

    const int kMod = 1e9 + 7;
    
    for (int i = 0; i < s.size(); ) {
      if (s[i] == '0') {
        ++i;
        continue;
      }
      
      int j = i + 1;
      while (j < s.size() && s[j] == '1') {
        ++j;
      }
      

      long long len = (j - i);
      long long curr = (len + 1) * len / 2;
      
      total = (total + curr) % kMod;
      
      i = j;
    }
    
    return total;
  }
};
