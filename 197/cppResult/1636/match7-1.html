const int MOD = 1e9 + 7;


class Solution {
public:
    int numSub(string s) {
        int n = s.length(), cnt = 0, ans = 0;
        
        for (int i = 0; i < n; i++){
            if (s[i] == '1'){
                ++cnt;
                ans = (ans + cnt) % MOD;
            }else cnt = 0;
        }
        
        return ans;
    }
};
