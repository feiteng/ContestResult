class Solution {
public:
    int numSub(string s) {

        const int mod = 1e9 + 7;
        int a = 0;
        int res = 0;
        for (char c : s) {
            if(c == '1') {
                ++a;
                res = (res + a) % mod;
            } else
                a = 0;
        }
        return res;
    }
};
