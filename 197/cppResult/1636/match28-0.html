class Solution {
public:
    int numSub(string s) {
        long long res=0,cnt=0;
        int mod=pow(10,9)+7;
        for (int i=0;i<s.size();i++)
        {
            if (s[i]=='1')
                cnt++;
            else
            {
                res=(res+cnt*(cnt+1)/2)%mod;
                cnt=0;
            }
        }
        res=(res+cnt*(cnt+1)/2)%mod;
        return res;
    }
};
