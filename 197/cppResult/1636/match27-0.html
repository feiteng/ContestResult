class Solution {
public:
    int numSub(string s) {
        int len = s.size();
        int ans = 0;
        int a = 0;
        const int mod = 1e9 + 7;
        for (int i = 0; i < len; i++) {
            if (s[i] == '1') {
                a++;
                a %= mod;
                ans += a;
                ans %= mod;
            } else {
                a = 0;
            }            
        }
        return ans;
    }
};
