class Solution {
public:
    int numSub(string s) {
        int l=s.size();
        int ans=0;
        int mod=1e9+7;
        int cnt=0;
        for(int i=0;i<l;i++){
            if(s[i]=='1') cnt++;
            else{
                int tmp=(int)((long long)cnt*(cnt+1)/2%mod);
                ans=(ans+tmp)%mod;
                cnt=0;
            }
        }
        int tmp=(int)((long long)cnt*(cnt+1)/2%mod);
        ans=(ans+tmp)%mod;
        cnt=0;
        return ans;
    }
};
