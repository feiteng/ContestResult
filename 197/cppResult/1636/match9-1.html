class Solution {
public:
    int mod=1e9+7;

    int numSub(string s) {
        long long cnt=0;
        long long ret=0;
        for(char c:s){
            if(c=='0'){
                ret=(ret+cnt*(cnt+1)/2)%mod;
                cnt=0;
            }else cnt++;
        }
        ret=(ret+cnt*(cnt+1)/2)%mod;
        return ret;
    }
};
