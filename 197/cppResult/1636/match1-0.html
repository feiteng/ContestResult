class Solution {
public:
    int numSub(string s) {
        int n = s.size();
        long long cnt = 0;
        long long ans = 0, mod = 1e9+7;
        for(int i = 0; i < n; i++) {
            if(s[i] == '1') {
                cnt++;
            } else {
                ans += cnt*(cnt+1)/2;
                ans %= mod;
                cnt = 0;
            }
        }
        ans += cnt*(cnt+1)/2;
        ans %= mod;
        return ans;
    }
};
