class Solution {
public:
    int numSub(string s) {
        int n = s.size();
        long long ans = 0, len = 0;
        for (int i=0; i<=s.size(); ++i)
        {
            if (i == s.size() || s[i] != '1')
            {
                //printf("len: %d\n",len);
                ans = (ans + ((len*(len+1))% 1000000007 /2 )) % 1000000007;
                len = 0;
            }
            else
           {
                ++len;
           }
        }
        return ans;
    }
};
