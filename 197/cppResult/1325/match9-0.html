class Solution {
public:
    
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pair<int, double>>> T(n);
        for(int i = 0; i < edges.size(); ++i){
            T[edges[i][0]].push_back({edges[i][1], succProb[i]});
            T[edges[i][1]].push_back({edges[i][0], succProb[i]});
        }
        vector<bool> vis(n, false);
        vector<double> dis(n, 0.0);
        dis[start] = 1.0;
        priority_queue<pair<double, int>> Q;
        Q.push({1.0, start});
        while(!Q.empty()){
            int curr = Q.top().second; Q.pop();
            if(vis[curr]) continue;
            vis[curr] = true;
            for(const auto& to : T[curr]){
                int id = to.first;
                double succ = to.second;
                if(!vis[id] && dis[curr] * succ > dis[id]){
                    dis[id] = dis[curr] * succ;
                    Q.push({dis[id], id});
                }
            }
        }
        return dis[end];
    }
};
