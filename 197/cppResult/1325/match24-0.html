class Solution {
public:
	double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
		vector<vector<pair<int, double>>> plist(n + 1);
		int i, j, k, a, b;
		double x, xx;
		for (i = 0; i < edges.size(); i++) {
			a = edges[i][0];
			b = edges[i][1];
			x = succProb[i];

			plist[a].push_back({ b, x });
			plist[b].push_back({ a, x });
		}

		vector<double> dp(n + 1, 0);
		dp[start] = 1.0;

		vector<bool> used(n + 1, false);
		priority_queue<pair<double, int>> q; 
		q.push({ 1.0, start });

		while (!q.empty()) {
			int id = q.top().second;
			x = q.top().first;
			q.pop();

			if (x != dp[id]) continue;

			for (auto p : plist[id]) {
				b = p.first;
				xx = p.second;

				if (dp[b] < xx * x) {
					dp[b] = xx * x;
					q.push({ xx * x, b });
				}
			}			
		}


		return dp[end];
	}
};
