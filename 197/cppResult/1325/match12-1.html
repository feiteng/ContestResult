class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succ, int start, int end) {
        
        // min path such that product is minimum
        
        vector<pair<int,double>>v[n+3];
        
        for(int i=0;i<edges.size();i=i+1)
        {
            v[edges[i][0]].push_back({edges[i][1],succ[i]});
            v[edges[i][1]].push_back({edges[i][0],succ[i]});
        }
        
        vector<double>prob(n+3,0);
        priority_queue<pair<double,int>>q;
        prob[start] = 1.0;
        q.push({1.0,start});
        
        while(!q.empty())
        {
            int u = q.top().second; q.pop();
            for(auto i : v[u])

            {
               if(prob[i.first]<(prob[u]*i.second)){prob[i.first]=prob[u]*i.second;q.push({prob[i.first],i.first});}
            }
        }
        return prob[end];
    }
};
