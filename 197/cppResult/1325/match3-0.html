class Solution {
public:
    double dis[20005];
	int vis[20005];
	int d[20005];
	vector<pair<int,double>> map[20005];
	priority_queue<pair<double,int>,vector<pair<double,int>>,less<pair<double,int>> > q;
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
		for(int i=0;i<edges.size();i++)
		{
			int u=edges[i][0],v=edges[i][1];
			d[u]++;d[v]++;
			map[u].push_back(make_pair(v,succProb[i]));
			map[v].push_back(make_pair(u,succProb[i]));
		}
		if(d[start]==0||d[end]==0)
			return 0;
		return Dijkstra(start,end);
	}
    
    double Dijkstra(int x,int y){
    	q.push(make_pair(1,x));
		dis[x]=1;
		while( !q.empty() )
		{
			int u=q.top().second;
			q.pop();
			vis[u]=1;
			for( int i=0;i<map[u].size();i++ )
			{
				int v=map[u][i].first;
				double c=map[u][i].second;
				if( !vis[v] && dis[v]<dis[u]*c )
				{
					dis[v]=dis[u]*c;
					q.push(make_pair(dis[v],v));
				}
			}
		}
		return dis[y];
	} 
};
