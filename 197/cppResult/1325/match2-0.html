class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<pair<int, double>> G[n];
        for(int i = 0; i < edges.size(); i++) {
            int u = edges[i][0], v = edges[i][1];
            double w = log(succProb[i]) / log(0.0001);
            G[u].push_back({v, w});
            G[v].push_back({u, w});
        }
        vector<double> d(n, 1e18);
        d[start] = 0;
        queue<int> qu;
        qu.push(start);
        while(!qu.empty()) {
            int u = qu.front();
            qu.pop();
            for(auto x : G[u]) {
                int v = x.first;
                double w = x.second;
                if(d[v] > d[u] + w) {
                    d[v] = d[u] + w;
                    qu.push(v);
                }
            }
        }
        return d[end] == 1e18 ? 0 : pow(0.0001, d[end]);
    }
};
