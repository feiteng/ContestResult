class Solution
{
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end)
    {

        vector<double> p(n, 0);
        p[end] = 1;
        
        for (int i = 0; i < n - 1; i++) 
        {
            bool stop = true;
            for(int j = 0; j < edges.size(); j++)
            {
                int a = edges[j][0], b = edges[j][1];
                double w = succProb[j];
                

                if(p[a] * w > p[b])
                {
                    p[b] = p[a] * w;
                    stop = false;
                }
                
                if(p[b] * w > p[a])
                {
                    p[a] = p[b] * w;
                    stop = false;
                }
            }
            if(stop)
                break;
        }
        return p[start];

    }
};
