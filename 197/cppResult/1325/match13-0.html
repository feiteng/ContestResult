class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        
        vector<pair<int, double> > graph[n];
        for(int i=0; i<edges.size(); i++) {
            int u = edges[i][0], v = edges[i][1]; 
            double w = succProb[i];
            graph[u].push_back(make_pair(v, w));
            graph[v].push_back(make_pair(u, w));
        }
        
        priority_queue<pair<double, int> > q;
        vector<double> probability(n, -0.1);
        q.push(make_pair(1.0, start));
        probability[start] = 1.0;
        
        while(!q.empty()) {
            int u = q.top().second;
            double prob = q.top().first;
            q.pop();
            
            if(u == end)
                return prob;
            
            for(int i=0; i<graph[u].size(); i++) {
                int v = graph[u][i].first;
                double w = graph[u][i].second;
                
                if(probability[v] < prob * w) {
                    probability[v] = prob * w;
                    q.push(make_pair(probability[v], v));
                    //cout<<probability[v]<<"\t";
                }
            }
        }
        
        return 0.0;
    }
};
