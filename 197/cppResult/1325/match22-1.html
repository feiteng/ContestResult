class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pair<int,double>>> adj(n);
        int m=edges.size();

        for(int i=0;i<m;i++){
            int u=edges[i][0];
            int v=edges[i][1];
            double w=succProb[i];
            adj[u].push_back({v,w});
            adj[v].push_back({u,w});
        }
        
        double distance[n];
        for(int i=0;i<n;i++){
            distance[i]=0;
        }
        distance[start]=1;
        
        priority_queue<pair<double,int>> pq;
        pq.push({1,start});
        vector<int> vis(n,0);
        
        while(!pq.empty()){
            int cur=pq.top().second;
            pq.pop();
            if(vis[cur]){
                continue;
            }

            vis[cur]=1;
            for(auto u:adj[cur]){
                int next=u.first;
                double weight = u.second;
                
                if(distance[cur]*weight >= distance[next]){
                    distance[next] = distance[cur]*weight;
                    pq.push({distance[next],next});
                }
            }
        }
        
        return distance[end];
    }
};
