class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        //bf算法 bfs
        int m = edges.size();
        vector<int> u(m), v(m);
        vector<double>  e(m);
        for(int i = 0; i < edges.size(); ++i){
            u[i] = edges[i][0];
            v[i] = edges[i][1];
            e[i] = succProb[i];
        }
        
        vector<double> dis(n, 0);
        dis[start] = 1;
        for(int i = 0; i < n; ++i){
            bool flag = true;
            for(int j = 0; j < edges.size(); ++j){
                if(dis[v[j]] < dis[u[j]] * e[j]){
                    flag = false;
                    dis[v[j]] = dis[u[j]] * e[j];
                }
                if(dis[u[j]] < dis[v[j]] * e[j]){
                    flag = false;
                    dis[u[j]] = dis[v[j]] * e[j];
                }
            }
            
            if(flag)     break;
        }
        
        return dis[end];
    }
};
