class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<double> dist(n, -1);
        priority_queue<pair<double, int>, vector<pair<double, int>>, less<pair<double, int>>> pq;
        
        unordered_map<int, vector<pair<int, double>>> graph;
        
        for (int i = 0; i < edges.size(); i++) {
            int node1 = edges[i][0], node2 = edges[i][1];
            float prob = succProb[i];

            graph[node1].push_back({node2, prob});
            graph[node2].push_back({node1, prob});
        }
        
        dist[start] = 1.0;
        pq.push({1.0, start});
        
        while (pq.size()) {
            auto [weight, node] = pq.top(); pq.pop();
            if (node == end) return weight;
            // cout << "("<<node<<")";

            for (auto & v : graph[node]) {
                auto [neighbor, prob] = v;
                
                if (dist[neighbor] < prob * weight) {
                    dist[neighbor] = prob * weight;
                    pq.push({prob * weight, neighbor});
                }
            }
            // for (auto & a : dist) {
            //     cout << a << ",";
            // }   
            // cout << endl << "====" << endl;
        }
        // for (auto & a : dist) {
        //     cout << a << ",";
        // }
        return 0;
        
    }
};
