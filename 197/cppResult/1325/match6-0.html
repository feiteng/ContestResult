#define MP make_pair

class Solution {
public:    
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& p, int start, int end) {
        vector<vector<pair<int,double>>> G(n);
        
        // construct graph
        for(int i=0; i<edges.size(); i++) {
            G[edges[i][0]].push_back({edges[i][1], p[i]});
            G[edges[i][1]].push_back({edges[i][0], p[i]});
        }
        
        // dijkstra
        priority_queue<pair<double,int>,vector<pair<double,int>>> pq;
        vector<bool> vis(n, false);
        vector<double> dist(n, -999);
        
        pq.push(MP(0, start));
        dist[start] = 1;
        
        while(!pq.empty()) {
            int c = pq.top().second; pq.pop();
            vis[c] = true;
            
            if(c==end) return dist[c];
            
            for(auto i:G[c]) {
                int nbr = i.first; double prob = i.second;
                double d = dist[c] * prob;
                if(!vis[nbr] && d>dist[nbr]) {
                    dist[nbr] = d;
                    pq.push(MP(d, nbr));
                }
            }
        }
        
        return 0;
    }
};
