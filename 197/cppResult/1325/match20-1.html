class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pair<size_t, double>>> adjList(n);        
        

        for (size_t i = 0; i < edges.size(); ++i)
        {
            auto& e = edges[i];
            adjList[e[0]].push_back({e[1], succProb[i]});
            adjList[e[1]].push_back({e[0], succProb[i]});
        }
        
        vector<double> dist(n);
        dist[start] = 1;
        
        queue<size_t> q;
        q.push(start);
        
        vector<bool> in_queue(n);
        in_queue[start] = true;
        

        while (!q.empty()) 
        {
            int node = q.front(); 
            q.pop(); 
            in_queue[node] = false;
            
            for (auto& nxt: adjList[node]) 
            {                
                if (dist[node] * nxt.second > dist[nxt.first]) 
                {
                    dist[nxt.first] = dist[node] * nxt.second;
                    if (!in_queue[nxt.first]) 
                    {
                        q.push(nxt.first);
                        in_queue[nxt.first] = true;
                    } 
                } 
            } 
        }
        return dist[end];
    }
};
