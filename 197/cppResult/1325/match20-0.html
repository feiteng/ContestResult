class Solution {
public:
    using pii = pair<int, double>;
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pii> > E(n);
        for(int i = 0; i<edges.size(); ++i){
            auto e = edges[i];
            E[e[0]].push_back({e[1], succProb[i]});
            E[e[1]].push_back({e[0], succProb[i]});
        }
        
        vector<bool> inque(n);
        vector<double> p(n);
        p[start] = 1;
        queue<int> que;
        que.push(start);
        
        while(que.size()){
            int u = que.front();
            que.pop();
            inque[u] = false;
            for(auto&[v, prob]: E[u]){
                if(p[u]*prob > p[v]){
                    p[v] = p[u]*prob;
                    if(!inque[v]){
                        que.push(v);
                        inque[v] = true;
                    }
                }
            }
        } 
        
        return p[end];
    }
    
};
