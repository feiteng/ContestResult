class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pair<int,double>>> ed(n);
        for(int i=0;i<edges.size();i++){
            ed[edges[i][0]].push_back({edges[i][1],succProb[i]});
            ed[edges[i][1]].push_back({edges[i][0],succProb[i]});
        }
        vector<double> dis(n,0);
        queue<pair<int,double>> q;
        q.push({start,1.00});
        while(!q.empty()){
            auto p=q.front();
            q.pop();
            if(p.first==end){
                dis[end]=max(dis[end],p.second);
                continue;
            }
            for(int i=0;i<ed[p.first].size();i++){
                if(dis[ed[p.first][i].first]<p.second*ed[p.first][i].second){
                    dis[ed[p.first][i].first]=p.second*ed[p.first][i].second;
                    q.push({ed[p.first][i].first,p.second*ed[p.first][i].second});
                }
            }
        }
        return dis[end];
    }
};
