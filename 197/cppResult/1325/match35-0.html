double dist[100005];
class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
		int m=edges.size();
		for (int i=0;i<n;++i) dist[i]=0;
		dist[start]=1;
		for (int i=1;i<=50;++i)
			for (int j=0;j<m;++j){
				int x=edges[j][0],y=edges[j][1];
                double w=succProb[j];
				dist[x]=max(dist[x],dist[y]*w);
				dist[y]=max(dist[y],dist[x]*w);
			}
		return dist[end];
    }
};
