class Solution {
public:
    
    
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector< vector<int> > adj(n);
        vector< vector<double> > vals(n);
        vector<double> prob(n, 0.0);
        
        for (int i = 0; i < edges.size(); ++i)
        {
            adj[ edges[i][0] ].push_back( edges[i][1] );
            adj[ edges[i][1] ].push_back( edges[i][0] );
            
            vals[ edges[i][0] ].push_back( succProb[i] );
            vals[ edges[i][1] ].push_back( succProb[i] );
        }
        
        deque<int> Q;
        Q.push_back(start);
        prob[start] = 1;
        while (Q.size() > 0)
        {
            int cur = Q.front();
            Q.pop_front();
            

            for (int i = 0; i < adj[cur].size(); ++i)
            {
                int v = adj[cur][i];
                double newProb = prob[cur] * vals[cur][i];
                if (newProb > prob[v] + 0.00000001)
                {
                    prob[v] = newProb;
                    Q.push_back(v);
                }
            }
        }
        
        return prob[end];
    }
};
