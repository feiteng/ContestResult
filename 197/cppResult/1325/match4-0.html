#define ff first
#define ss second
class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<pair<int,double>>v[n] ;
        for(int i=0;i<edges.size();i++)
        {
            int x=edges[i][0] ;
            int y=edges[i][1] ;
            
            v[x].push_back({y,succProb[i]}) ;
            v[y].push_back({x,succProb[i]}) ;
        }
        
        set<pair<double,int>>s ;
        s.insert({1.0,start}) ;
        
        double d[n] ;
        for(int i=0;i<n;i++)
        {
            d[i]=0.0 ;
            
        }
        
        d[start]=1.0 ;
        
        while(!s.empty())
        {
            auto it=s.end() ;
            it-- ;
            int sr=(*it).ss ;
            double prob=(*it).ff ;
            //cout<<sr<<" "<<prob<<endl ;
            
            for(auto j:v[sr])
            {
                if(j.ss*prob>d[j.ff])
                {
                    d[j.ff]=j.ss*prob ;
                    s.insert({d[j.ff],j.ff}) ;
                }
            }
            
            s.erase(it) ;
        }
        
        return d[end] ;
        
        
        
    }
};
