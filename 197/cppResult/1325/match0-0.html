class Solution {
public:
	double maxProbability(int n, vector<vector<int>>& edges, vector<double>& prob, int start, int end) {
		vector<vector<pair<int,double>>>adj(n);
		for(int i=0;i<prob.size();i++) {
			adj[edges[i][0]].push_back({ edges[i][1],prob[i] });
			adj[edges[i][1]].push_back({ edges[i][0],prob[i] });
		}
		vector<double>maxx(n, 0);
		maxx[start] = 1.0;
		queue<int>Q;
		Q.push(start);
		while (!Q.empty()) {
			int t = Q.front(); Q.pop();
			for (auto it : adj[t]) {
				double p = maxx[t] * it.second;
				if (p > maxx[it.first]) {
					maxx[it.first] = p;
					Q.push(it.first);
				}
			}
		}
		return maxx[end];
	}
};
