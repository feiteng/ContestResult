class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {

        unordered_map<int, vector<pair<int, double>>> g;
        for (int i = 0; i < edges.size(); i++) {    
            int a = edges[i][0], b = edges[i][1];
            double c = succProb[i];
            g[a].push_back({b, c});
            g[b].push_back({a, c});
        }
        
        vector<double> dp(n, 0);
        dp[start] = 1.0;
        queue<int> qu;
        qu.push(start);
        while (!qu.empty()) {
            int curr = qu.front(); qu.pop();
            for (auto p : g[curr]) {
                int next = p.first;
                double prob = p.second;
                //if (dp[next] != 0 && dp[curr] * prob <= dp[next]) continue;
                if (dp[curr] * prob <= dp[next]) continue;
                dp[next] = dp[curr] * prob;
                qu.push(next);
            }
        }
        
        return dp[end];
    }
};
