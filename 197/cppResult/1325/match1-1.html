class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector < vector < pair < int, double > > > e;
        e.resize(n + 1);
        for(int i = 0; i < edges.size(); ++i) {
            int x = edges[i][0], y = edges[i][1];
            double z = succProb[i];
            e[x].push_back(make_pair(y, z));
            e[y].push_back(make_pair(x, z));
        }
        set < pair < double, int > > S;
        vector < double > dist(n+ 1, 0.0);
        dist[start] = 1;
        S.insert(make_pair(1.0, start));
        while(!S.empty()) {
            int v = S.rbegin()->second;

            S.erase(--S.end());
            for(auto &to : e[v]) {
                double w = to.second * dist[v];
                if(w > dist[to.first]) {
                    S.erase(make_pair(dist[to.first], to.first));
                    dist[to.first] = w;
                    S.insert(make_pair(dist[to.first], to.first));
                }
            }
        }
        return dist[end];
    }
};
