class Solution {
public:
	

	double dis[10005];
	vector<pair<int,double>> g[10005];
	int vis[10005]; 
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
		
		for( int i=0;i<edges.size();i++ )
		{
			int u=edges[i][0],v=edges[i][1];
			g[u].push_back(make_pair(v,succProb[i]));
			g[v].push_back(make_pair(u,succProb[i]));
		}
		priority_queue<pair<double,int>,vector<pair<double,int>>,less<pair<double,int>> > q;
		q.push(make_pair(1,start));

		dis[start]=1;
		while( !q.empty() )
		{
			int u=q.top().second;
			q.pop();
			vis[u]=1;
			for( int i=0;i<g[u].size();i++ )
			{
				int v=g[u][i].first;
				double c=g[u][i].second;
//				cout<<c<<endl;
//				cout<<dis[u]<<endl;
				if( !vis[v] && dis[v]<dis[u]*c )
				{
					dis[v]=dis[u]*c;
					q.push( make_pair(dis[v],v));
				}
			}
		}
	//	for( int i=1;i<=n;i++ ) cout<<dis[i]<<" ";
		return dis[end];
    }
};
