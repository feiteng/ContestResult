class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        vector<vector<pair<int, double>>> g(n);
        for(int i=0; i<edges.size(); i++) {
            int u = edges[i][0];
            int v = edges[i][1];
            double w = succProb[i];
            g[u].emplace_back(v,w);
            g[v].emplace_back(u,w);
        }
        vector<double> prob(n, -1.0);
        vector<bool> used(n, false);
        priority_queue<pair<double, int>> q;
        q.emplace(1.0, start);
        while(!q.empty()){
            pair<double, int> cur = q.top(); q.pop();
            double curW = cur.first;
            int curV = cur.second;
            if(used[curV]) continue;
            used[curV] = true;
            prob[curV] = curW;
            if(curV==end) return curW;
            for(int i=0; i<g[curV].size(); i++){
                int u = g[curV][i].first;
                double w = g[curV][i].second;
                if(!used[u] && prob[u]<curW*w) {
                    prob[u] = curW*w;
                    q.emplace(curW*w, u);
                }
            }
        }
        return 0.0;
    }
};
