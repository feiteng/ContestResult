class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& sp, int st, int en) {
        vector<vector<pair<int,double>>> adj(n);
        for(int it=0;it<edges.size();it++){
            int u=edges[it][0];
            int v=edges[it][1];
            adj[u].push_back({v,sp[it]});
            adj[v].push_back({u,sp[it]});
        }
        vector<double> prob(n,0);
        prob[st]=1;
        priority_queue< pair<double,int>, vector<pair<double,int>>> qu;
        qu.push({1,st});
        while(qu.empty()==false){
            double prb=qu.top().first;

            int node=qu.top().second;
            qu.pop();
            for(auto it:adj[node]){
                if(it.second*prb>prob[it.first]){
                    prob[it.first]=it.second*prb;
                    qu.push({prob[it.first],it.first});
                }
            }
        }
        return prob[en];
    }
};
