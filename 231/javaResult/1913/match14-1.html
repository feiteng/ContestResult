class Solution {
    int[][] a;
    int[] b;
    int[][] ans;
    Set<Integer>[] s;
    public int minChanges(int[] nums, int k) {
        int n = nums.length;
        a = new int[k][1024];
        b = new int[k];
        ans = new int[k][1024];
        s = 
new Set[k];
        int[] max = new int[k];
        for (int i=0; i<k; i++)
            s[i] = new HashSet<>();
        for (int i=0; i<n; i++) {
            a[i%k][nums[i]]++;
            if (a[i%k][nums[i]] > max[i%k])
                max[i%k] = a[i%k][nums[i]];
            s[i%k].add(nums[i]);
        }
        for (int i=0; i<k; i++)
            b[i] = (n-i-1) / k + 1;
        for (int i=0; i<k; i++)
            for (int j=0; j<1024; j++)
                ans[i][j] = -1;
        Arrays.sort(max);
        int ans2 = n;
        for (int i=1; i<k; i++)
            ans2 -= max[i];
        return Math.min(ans2, search(k-1, 0));
    }
    private int search(int k, int x) {
        if (ans[k][x] >= 0)
            return ans[k][x];
        if (k == 0) {
            ans[k][x] = b[k] - a[k][x];
            return ans[k][x];
        }
        int t = Integer.MAX_VALUE;
        for (int y : s[k])
            t = Math.min(t, search(k-1, x ^ y) + b[k] - a[k][y]);
        ans[k][x] = t;
        return t;
    }
}
