class Solution {
    long mod = 1_0000_0000_7;
    public int countRestrictedPaths(int n, int[][] edges) {
        Long[] dis = new Long[n+1];
        List<long[]>[] adj = new List[n+1];
        
        for(int i=1; i<=n; i++){
            dis[i] = null;
            adj[i] = new ArrayList<long[]>();
        }
        // u v w
        for(int[] edge: edges){
            adj[(int)edge[0]].add(new long[]{edge[1], edge[2]});
            adj[(int)edge[1]].add(new long[]{edge[0], edge[2]});
        }
        
        dis[n] = 0L;
        
        // dis , index
        var q = new PriorityQueue<long[]>( (v1,v2) -> Long.compare(v1[0], v2[0]) );
        q.add(new long[]{0,n});
        boolean[] seen = new boolean[n+1];
        
        while(!q.isEmpty()){
            // dis , index
            long[] cur = q.poll();
            
            if(seen[(int)cur[1]]) continue;
            
            seen[(int)cur[1]] = true;
            for(long[] nei: adj[(int)cur[1]]){
                // v, w
                // System.out.println(Arrays.toString(nei));
                if(!seen[(int)nei[0]] && (dis[(int)nei[0]] == null || dis[(int)cur[1]] + nei[1] < dis[(int)nei[0]])){
                    dis[(int)nei[0]] = dis[(int)cur[1]] + nei[1];
                    q.add(new long[]{dis[(int)nei[0]], nei[0]});
                }
            }
        }
        
        // System.out.println(Arrays.toString(dis));
        
        Long[] ans = new Long[n+1];
        seen = new boolean[n+1];
        
        long ansFinal = dfs(1, adj, dis, ans, seen);
        // System.out.println(Arrays.toString(ans));
        return (int)ansFinal;
        //return ans[0];
    }
    
    long dfs(long node, List<long[]>[] adj, Long[] dis, Long[] ans, boolean[] seen){
        // System.out.println(node);
        long n = dis.length - 1;
        long res = 0;
        if(node == n){
           
            // System.out.println();
            return 1;
        }
        if(ans[(int)node]!=null) return ans[(int)node];
        seen[(int)node] = true;
        
        for(long[] nei: adj[(int)node]){
            // v, w
            if(!seen[(int)nei[0]] && dis[(int)nei[0]] < dis[(int)node]){
                res = (res + dfs(nei[0], adj, dis, ans, seen))%mod;
            }
        }
    
        seen[(int)node] = false;
        ans[(int)node] = res;
        return res;
    }
}
