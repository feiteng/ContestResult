class Solution {
    
public int countRestrictedPaths(int n, int[][] edges) {
        List<List<int[]>> graph = new ArrayList<>(n);
        for(int i = 0; i < n; ++i) {
            graph.add(new ArrayList<>());
        }
        for(int[] edge : edges) {
            int a = edge[0] - 1;
            int b = edge[1] - 1;
            int d = edge[2];
            graph.get(a).add(new int[] {b, d});
            graph.get(b).add(new int[] {a, d});
        }
        long[] distances = new long[n];
        Arrays.fill(distances, Long.MAX_VALUE);
        Queue<long[]> q = new PriorityQueue<>((a, b) -> Long.compare(a[1], b[1]));
        q.offer(new long[] {n-1, 0});
        while(!q.isEmpty()) {
            long[] c = q.poll();
            int i = (int) c[0];
            long d = c[1];
            if (d >= distances[i]) {
                continue;
            }
            
distances[i] = d;
            for(int[] edge : graph.get(i)) {
                int neighbor = edge[0];
                int weight = edge[1];
                if (distances[neighbor] == Long.MAX_VALUE) {
                    q.offer(new long[] {neighbor, distances[i] + weight});
                }
            }
        }
        return helper(0, n-1, graph, distances, new int[n]);
    }
    
    static int helper(int i, 
int j, List<List<int[]>> graph, long[] distances, int[] dp) {
        if (i == j) {
            return 1;
        } else if (dp[i] > 0) {
            return dp[i] - 1;
        
}
        int result = 0;
        for(int[] edge : graph.get(i)) {
            int neighbor = edge[0];
            if (distances[neighbor] < distances[i]) {
                result += 
helper(neighbor, j, graph, distances, dp);
                result %= 1000000007;
            }
        }
        dp[i] = result + 1;
        return result;
    }
}
