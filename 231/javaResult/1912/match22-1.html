class Solution {

    public int countRestrictedPaths(int n, int[][] edges) {
        distanceToLastNode = new int[n + 1];
        Arrays.fill(distanceToLastNode, Integer.MAX_VALUE);
        Queue<int[]> queue = new ArrayDeque<>();

        
graph = new List[n + 1];
        for (int i = 0; i < graph.length; i++) {
            graph[i] = new ArrayList<>();
        }

        for (int i = 0; i < edges.length; i++) {
            graph[edges[i][0]].add(new int[]{edges[i][1], edges[i][2]});
            graph[edges[i][1]].add(new int[]{edges[i][0], edges[i][2]});
        }

        queue.add(
new int[]{n, 0});
        distanceToLastNode[n] = 0;
        while (!queue.isEmpty()) {
            int[] node = queue.poll();

            for (int[] nextNode : graph[node[0]]) {
                
int dis = nextNode[1] + node[1];
                if (distanceToLastNode[nextNode[0]] > dis) {
                    distanceToLastNode[nextNode[0]] = dis;
                    queue.add(new int[]{nextNode[0], dis});
                }
            }
        }


        cache = new int[n + 1];
        Arrays.fill(cache, -1);
        return dfs(n);
    }

    List<int[]>[] graph;
    int[] distanceToLastNode;
    int[] cache;
    int MOD = 1000000000 + 7;

    public int dfs(
int i) {
        if (i == 1) {
            return 1;
        }
        if (cache[i] != -1) {
            return cache[i];
        }

        int res = 0;
        for (int[] edge : graph[i]) {
            
if (distanceToLastNode[edge[0]] > distanceToLastNode[i]) {
                res += dfs(edge[0]);
                res %= MOD;
            }
        }
        cache[i] = res;
        return res;
    }


}
