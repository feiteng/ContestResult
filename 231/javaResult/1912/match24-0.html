class Solution {
    private int MOD = (int) 1e9 + 7;

    public int countRestrictedPaths(int n, int[][] edges) {
        List<List<int[]>> edgeList = new ArrayList<>(n + 1);
        for (int i = 0; i <= n; i++) {
            edgeList.add(new ArrayList<>());
        }
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1], w = edge[2];
            edgeList.get(u).add(new int[]{v, w});
            edgeList.get(v).add(new int[]{u, w});
        }
        int[] dis = new int[n + 1];
        int[] visit = new int[n + 1];
        Queue<Integer> queue = new LinkedList<>();
        Arrays.fill(dis, Integer.MAX_VALUE);
        dis[n] = 0;
        queue.add(n);
        visit[n] = 1;
        while (!queue.isEmpty()) {
            int u = queue.poll();
            visit[u] = 0;
            for (int[] p : edgeList.get(u)) {
                int v = p[0], w = p[1];
                if (dis[v] > dis[u] + w) {
                    dis[v] = dis[u] + w;
                    if (visit[v] == 0) {
                        queue.add(v);
                        visit[v] = 1;
                    }
                }
            }
        }
        long[] ans = new long[n + 1];
        Arrays.fill(ans, -1);
        ans[n] = 1;
        dfs(ans, 1, edgeList, dis);
        return (int) (ans[1] % MOD);
    }

    private long dfs(long[] ans, int u, List<List<int[]>> edgeList, int[] dis) {
        if (ans[u] != -1) {
            return ans[u];
        }
        long ways = 0;
        for (int[] p : edgeList.get(u)) {
            int v = p[0];
            if (dis[v] < dis[u]) {
                ways = (ways + dfs(ans, v, edgeList, dis)) % MOD;
            }
        }
        ans[u] = ways;
        return ans[u];
    }
}
