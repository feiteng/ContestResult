class Solution {
    
    int N;
    LinkedList<int[]>[] G;
    int[] dists;
    Long[] M;
    
    public int countRestrictedPaths(int n, int[][] edges) {
        this.N = n;
        this.G = new LinkedList[N];
        for (int i = 0; i < N; i++) G[i] = new LinkedList<>();
        for (int[] e : edges) {
            G[e[0] - 1].offer(new int[] { e[1] - 1, e[2] });
            G[e[1] - 1].offer(new int[] { e[0] - 1, e[2] });
        }
        
        PriorityQueue<int[]> Q = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        this.dists = new int[N];
        Arrays.fill(dists, Integer.MAX_VALUE);
        dists[n - 1] = 0;
        Q.offer(new int[] { n - 1, 0 });
        
        while (!Q.isEmpty()) {
            int[] arr = Q.poll();
            int i = arr[0];
            int d = arr[1];
            
            if (dists[i] < d) continue;
            
            for (int[] arr2 : G[i]) {
                int j = arr2[0];
                int w = arr2[1];
                
                if (d + w < dists[j]) {
                    dists[j] = d + w;
                    Q.offer(new int[] { j, d + w });
                }
            }
        }
        M = new Long[N];
        return (int) dfs(0);
    }
    
    long dfs(int i) {
        if (i == N - 1) return 1;
        if (M[i] != null) return M[i];
        
        long ans = 0;
        for (int[] arr : G[i]) {
            int j = arr[0];
            
            if (dists[i] > dists[j]) {
                ans += dfs(j);
            }
        }
        
        M[i] = ans % (int) (1e9 + 7);
        return M[i];
    }
}
