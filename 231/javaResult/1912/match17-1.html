class Solution {
    private List<int[]>[] adj;
    private long[] distance;
    private int[] dp;
    private int mod = (int)1e9 + 7;
    public int countRestrictedPaths(int n, int[][] edges) {
        adj = buildGraph(n, edges);
        boolean[] processed = new boolean[n + 1];
        distance = new long[n + 1];
        Arrays.fill(distance, Long.MAX_VALUE);
        distance[n] = 0;
        PriorityQueue<long[]> q = new PriorityQueue<>(Comparator.comparingLong(a -> a[0]));
        
q.add(new long[]{0, n});
        
        while(!q.isEmpty()) {
            long[] curr = q.poll();
            int node = (int)curr[1];
            if(processed[node]) continue;
            processed[node] = true;
            for(int[] next : adj[node]) {
                
if(curr[0] + next[1] < distance[next[0]]) {
                    distance[next[0]] = curr[0] + next[1];
                    q.add(new long[]{distance[next[0]], next[0]});
                }
            }
        }
        dp = new int[n + 1];
        Arrays.fill(dp, -1);
        dp[n] = 1;
        return dfs(1, n);
    }
    private int dfs(int node, 
int n) {
        if(node == n) return 1;
        if(dp[node] >= 0) return dp[node];
        int cnt = 0;
        for(int[] next : adj[node]) {
            if(distance[next[0]] < distance[node]) {
                cnt = (cnt + 
dfs(next[0], n)) % mod;
            }
        }
        dp[node] = cnt;
        return cnt;
    }
    private List<int[]>[] buildGraph(int n, int[][] edges) {
        List<int[]>[] adj = new List[n + 1];
        for(int i = 0; i <= n; i++) {
            adj[i] = new ArrayList<>();
        
}
        for(int[] e : edges) {
            adj[e[0]].add(new int[]{e[1], e[2]});
            adj[e[1]].add(new int[]{e[0], e[2]});
        }
        return adj;
    }
}
