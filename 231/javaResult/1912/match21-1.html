class Solution {
    
    
List<List<int[]>> g;
    int[] d;
    static final int mod = (int)1e9 + 7;
    Map<Integer, Integer> map;
    
    public int countRestrictedPaths(int n, int[][] edges) {
        d = new int[n + 1];
        Arrays.fill(d, Integer.MAX_VALUE / 2);
        g = new ArrayList<>();
        for (int i = 0; i <= n; ++ i) {
            g.add(new ArrayList<>());
        }
        for (int[] e : edges) {
            int a = e[0], b = e[1], w = e[2];
            
g.get(a).add(new int[]{b, w});
            g.get(b).add(new int[]{a, w});
        }
        d[n] = 0;
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{n, 0});
        while(!queue.isEmpty()) {
            int[] cur = queue.poll();
            int u = cur[0], dist = cur[1];
            for (int[] e : g.get(u)) {
                int nx = e[0], w = e[1];
                if (w + dist < d[nx]) {
                    
d[nx] = w + dist;
                    queue.offer(new int[]{nx, w + dist});
                }
            }
        }
        // for (int i = 1; i <= n; ++ i) System.out.println(d[i]);
        map = new HashMap<>();
        return dfs(1, n);
    }
    
    int dfs(int u, 
int n) {
        if (u == n) return 1;
        if (map.containsKey(u)) return map.get(u);
        long ret = 0;
        for (int[] e : g.get(u)) {
            int nx = e[0], w = e[1];
            
if (d[nx] >= d[u]) continue;
            ret = (ret + dfs(nx, n)) % mod;
        }
        map.put(u, (int)ret);
        return (int) ret;
        
    }
    
}
