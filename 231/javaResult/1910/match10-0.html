class Solution {
    public boolean checkOnesSegment(String s) {
        int cnt = 0;
        for(int i = 0; i < s.length(); i++) {
            if(s.charAt(i) == '1') {
                if(i == s.length()-1 || s.charAt(i+1) == '0') {
                    cnt++;
                }
            }
        }
        return cnt == 1;
    }
}
