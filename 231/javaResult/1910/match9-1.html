class Solution {
    public boolean checkOnesSegment(String s) {
        int flag = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '0') {
                flag = 1;
            }
            if (s.charAt(i) == '1' && flag == 1) 
{
                return false;
            }
        }
        return true;
        
    }
}
