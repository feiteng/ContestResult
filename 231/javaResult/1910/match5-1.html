class Solution {
    public boolean checkOnesSegment(String s) {
        int index  = -1;
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) == '1' && index != -1 && index < i-1) return false;
            if(s.charAt(i) == '1') index = i;
        }
        return true;
    }
}
