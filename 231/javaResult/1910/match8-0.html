class Solution {
    public boolean checkOnesSegment(String s) {
        int n = s.length();
        int cnt = 0;
        boolean found = false;
        for (int j = 0; j < n; j++) {
            if (s.charAt(j) == '1') {
                if (!found) {
                    found = true;
                    cnt++;
                }
            } else {
                found = false;
            }
        }
        return cnt <= 1;
    }
}
