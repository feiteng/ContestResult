class Solution {
    public boolean checkOnesSegment(String s) {
        int n = s.length();
        boolean ans = false;
        for (int i = 0; i < n; ++i) {
            if (s.charAt(i) == '1') {
                if (ans) return false;
                ans = true;
                while (i < n && s.charAt(i) == '1') ++i;
            }
        }
        
        return true;
    }
}
