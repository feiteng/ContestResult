class Solution {
    public boolean checkOnesSegment(String s) {
        int count = 0;
        char prev = '0';
        for (char c : s.toCharArray()) {
            if (prev == '0' && c == '1') {
                count++;
            }
            prev = c;
        }
        return count <= 1;
    }
}
