class Solution {
    public boolean checkOnesSegment(String s) {
        int c = 0;
        int l = '0';
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1' && s.charAt(i) != l) {
                c++;
            }
            l = s.charAt(i);
        }
        return c <= 1;
    }
}
