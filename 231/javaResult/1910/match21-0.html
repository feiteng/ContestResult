import java.util.*;

class Solution {
    public boolean checkOnesSegment(String s) {

        boolean toggle = false;
        for(int idx = 0; idx < s.length(); idx++) {
            char c = s.charAt(idx);
            if (c == '0') { toggle = true; }
            else {
                if (toggle) { return false; }
            }
        }
        return true;
    }
}
