class Solution {
    public boolean checkOnesSegment(String s) {
        boolean oneFound = false;
        for(int i=0; i<s.length();i++) {
            if(s.charAt(i) == '1') {
                if(oneFound && i>0 && s.charAt(i-1) == '0') return false; 
                oneFound = true;
            } 
        }
        return true;
    }
}
