class Solution {
    public boolean checkOnesSegment(String s) {
        boolean metOnes = false;
        boolean prevOne = false;

        for (var c : s.toCharArray()) {
            if (c == '1') {
                if (!prevOne && metOnes) {
                    return false;
                }
                prevOne = true;
                metOnes = true;
            } else {
                prevOne = false;
            }
        }

        return true;
    }
}
