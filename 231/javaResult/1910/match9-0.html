class Solution {
    public boolean checkOnesSegment(String s) {
        int oneStart = -1;
        
        for (int i = 0; i < s.length(); ++i) {
            if (s.charAt(i) == '1' && oneStart == -1) {
                oneStart = i;
            }
            if (i > 0 && s.charAt(i) == '1' && s.charAt(i-1) == '0' && oneStart >= 0) {
                return false;
            }
        }
        
        return true;
    }
}
