class Solution {
    public boolean checkOnesSegment(String s) {
        char prev = '0';
        int cnt = 0;
        for (char ch : s.toCharArray()) {
            if (ch == '1' && prev == '0') {
                if (++cnt > 1) {
                    return false;   
                }
            
}
            prev = ch;
        }
        return true;
    }
}
