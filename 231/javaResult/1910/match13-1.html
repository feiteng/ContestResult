class Solution {
    public boolean checkOnesSegment(String s) {
        boolean seen = false;
        
        for(int i=0; i<s.length(); ++i) {
            char c = s.charAt(i);
            if(c=='1') 
{
                if(seen && s.charAt(i-1) != '1') return false;
                seen = true;
            } 
        }
        
        return true;
    }
}
