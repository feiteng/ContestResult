class Solution {
    public boolean checkOnesSegment(String s) {
        int cnt = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1' && (i == 0 || s.charAt(i - 1) == '0')) {
                cnt++;
                if (cnt == 2) {
                    return false;
                
}
            }
        }
        return true;
        
    }
}
