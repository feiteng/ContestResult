class Solution {
    public boolean checkOnesSegment(String s) {
        int pre = 0;
        for(int i = 1; i < s.length(); i++){
            if(s.charAt(i) == '1'){
                if(i-pre > 1)
                    
return false;
                pre = i;
            }
        }
        return true;
    }
}
