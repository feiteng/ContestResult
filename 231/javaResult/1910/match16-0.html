class Solution {
    public boolean checkOnesSegment(String s) {
        int cnt = 1;
        boolean pre = true;
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                if (!pre) {
                    return false;
                }
            } else {
                pre = false;
            }
        }
        return cnt == 1;
    }
}
