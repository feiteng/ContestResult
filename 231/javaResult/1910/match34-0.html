class Solution {
    public boolean checkOnesSegment(String s) {
        boolean seen = false;
        char last = '0';
        for (int j = 0;j<s.length();j++) {
            if (s.charAt(j) =='1') {
                if (last == '0' && seen) {return false;}
                seen = true;
            }
            last = s.charAt(j);
        }
        return true;
    }
}
