class Solution {
    public boolean checkOnesSegment(String s) {
        boolean visited = false;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                if (visited) return false;
                while (i < s.length() && s.charAt(i) == '1') {
                    i++;
                }
                
                
visited = true;
                i--;
            }
        }
        
        return true;
    }
}
