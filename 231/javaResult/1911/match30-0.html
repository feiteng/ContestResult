class Solution {
    public int minElements(int[] nums, int limit, int goal) {
        long sum=0;
        for(int i:nums)sum+=i;
        long dif=Math.abs(goal-sum);
        
        long res=dif/limit;
        //cout<<dif<<"  "<<limit<<endl;
        if(dif%limit!=0){
            res++;
        }
        return (int)(res);
    }
}
