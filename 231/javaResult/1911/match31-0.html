class Solution {
    public int minElements(int[] nums, int limit, int goal) {
        long sum = 0;
        for (int num : nums) {
            sum += num;
        }
        long diff = goal - sum;
        diff = Math.abs(diff);
        long ans = diff / limit;
        if (diff % limit != 0) {
            ans++;
        }
        return (int) ans;
    }
}
