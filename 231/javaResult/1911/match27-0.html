class Solution {
    
    // Return the minimum number of elements you need to add
    public int minElements(int[] nums, int limit, int goal) {
        long sum = 0;
        for (int val : nums) {
            sum += val;
        }
        long diff = Math.abs(goal - sum);
        // 0 / 3 -> 0
        // 1 / 3 -> 1
        // 2 / 3 -> 1
        // 3 / 3 -> 1
        // 4 / 3 -> 2
        return (int)((diff + limit - 1) / limit);
    }
}
