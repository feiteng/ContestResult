class Solution {
    public int minElements(int[] nums, int limit, int goal) {
        long sum = 0;
        for (int num: nums) {
            sum += num;
        }
        
        long delta = Math.abs(goal - sum);
        long res = delta / limit;
        if (delta % limit != 0) {
            res++;
        }
        
        return (int)res;
    }
}
