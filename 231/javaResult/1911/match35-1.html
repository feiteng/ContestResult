class Solution {
    public int minElements(int[] nums, int limit, int goal) {
        long sum = 0;
        for (int num : nums) {
            sum += num;
        
}
        
        long diff = Math.abs(goal - sum);
        if (diff == 0) return 0;
        return  1 + (int) ((diff - 1) / limit);
    }
}
