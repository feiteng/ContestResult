class Solution {
    public int minElements(int[] nums, int limit, int goal) {
        long sum = 0;
        for (int i : nums) sum += i;
        long diff = Math.abs(goal - sum);
        // System.out.println("diff " + diff + " limit " + limit);
        return (int) Math.ceil(((double) (diff))/limit);
    }
}
