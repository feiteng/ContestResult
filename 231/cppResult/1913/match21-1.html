class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        vector<int> dp(1024, -1);
        dp[0] = 0;
        int pre = 0;
        for (int i = 0;i < k;i++) {
            vector<int> counter(1024, 0);

            for (int j = i;j < nums.size();j += k) {
                counter[nums[j]]++;
            }
            vector<int> dp1(1024, pre);
            for (int j = 0;j < 1024;j++) {

                if (counter[j] != 0) {
                    for (int k = 0;k < 1024;k++) {
                        if (dp[k ^ j] != -1) {
                            dp1[k] = max(dp1[k], dp[k ^ j] + counter[j]);
                        }
                    }
                }
            }
            dp = dp1;
            for (int j = 0;j < 1024;j++) {
                pre = max(pre, dp[j]);
            }
        }
        return nums.size() - dp[0];
    }
};
