int cnt[2005];
class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        vector<int> dp(1025,0);
        for(int i=0;i<k;i++)

        {
            memset(cnt,0,sizeof(cnt));
            int mm=0;
            for(int j=0;j<(1<<10);j++)
            {
                mm=max(mm,dp[j]);
            }
            vector<int> nexdp(1025,mm);
            for(int j=i;j<nums.size();j+=k)
            {
                cnt[nums[j]]++;
            }
                  

            if(i==0)
            {
                for(int j=0;j<(1<<10);j++)
                {
                    if(cnt[j]>0)
                    {
                        nexdp[j]=cnt[j];
                    }  
                }
            }
            else
            {

                for(int j=0;j<(1<<10);j++)
                {
                    if(cnt[j]>0)
                    {
                        for(int k=0;k<(1<<10);k++)
                        {
                            nexdp[j^k]=max(nexdp[j^k],dp[k]+cnt[j]);
                        }

                    }         
                } 
            }
            dp=nexdp;
        }
        return nums.size()-dp[0];
    }
};
