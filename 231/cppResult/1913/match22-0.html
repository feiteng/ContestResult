class Solution {
public:
    int cnt[2010][1025],dp[2][1025];
    int minChanges(vector<int>& nums, int k) {
        memset(dp,-1,sizeof dp);
        int n=nums.size();
        for(int i=0;i<n;++i) {
            ++cnt[i%k][nums[i]];
        }
        int las=0,now=1;
        dp[las][0]=0;
        for(int i=0;i<k;++i) {
            int mx=0;
            for(int j=0;j<1024;++j) dp[now][j]=0;
            for(int j=0;j<1024;++j) {
                if(cnt[i][j]) {
                    for(int x=0;x<1024;++x) {
                        if(dp[las][x^j]!=-1)
                            dp[now][x]=max(dp[now][x],dp[las][x^j]+cnt[i][j]);
                        
                        
                    }
                }
                mx=max(mx,dp[las][j]);
            }
            for(int j=0;j<1024;++j)
                dp[now][j]=max(dp[now][j],mx);
            // cout<<mx<<endl;
            // for(int j=0;j<4;++j)
            //     cout<<dp[now][j]<<" ";
            // cout<<endl;
            swap(las,now);
        }
        return n-dp[las][0];
    }
};
