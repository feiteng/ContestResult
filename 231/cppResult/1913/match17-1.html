class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        int f[2005][1024]; memset(f, 127, sizeof(f));
        f[0][0] = 0;
        int prev_min = 0;
        vector<int> c; c.resize(1024);
        for (int i = 0; i < k; i ++)
        {
            
            int tot = 0;
            for (int j = i; j < nums.size(); j += k)

                ++ c[nums[j]], ++ tot;
            prev_min += tot;
            for (int j = 0; j < 1024; j ++)
                f[i+1][j] = prev_min;
            for (int j = i; j < nums.size(); j += k)
                if (c[nums[j]] > 0)
                {
                    for (int p = 0; p < 1024; p ++)
                        f[i+1][nums[j]^p] = min(f[i+1][nums[j]^p], f[i][p] + tot - c[nums[j]]), prev_min = min(prev_min, f[i+1][nums[j]^p]);
                    c[nums[j]] = 0;
                }
        }
        return f[k][0];
    }
};
