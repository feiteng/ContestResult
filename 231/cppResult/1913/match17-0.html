class Solution {
public:
    int dp[2010][1050];
    int cnt[2010][1050];
    int pp[2010];
    int minChanges(vector<int>& nums, int k) {
        int n=nums.size();
        for(int i=1;i<1024;i++) dp[0][i]=n+1;
        for(int i=1;i<=n;i++)
            ++cnt[(i-1)%k][nums[i-1]],++pp[(i-1)%k];
        for(int i=1;i<=k;i++){
            int mn=n+1;
            for(int j=0;j<1024;j++)
                mn=min(mn,dp[i-1][j]);
            for(int j=0;j<1024;j++)
                dp[i][j]=mn+pp[i-1];
            for(int j=0;j<1024;j++) if(cnt[i-1][j]!=0){
                for(int p=0;p<1024;p++)
                    dp[i][j^p]=min(dp[i][j^p],dp[i-1][p]+pp[i-1]-cnt[i-1][j]);
            }
        }
        return dp[k][0];
    }
};
