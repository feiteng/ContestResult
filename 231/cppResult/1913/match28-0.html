class Solution {
public:
    int minChanges(vector<int>& a, int k) {
        int d[2][1030] = {}, z = 0, y = 100005;
        //cout << a.size() << ' ';
        //for (int i = 1; i < 2000; i++) a.push_back(i % 1024);
        for (int i = 1; i <= 1027; i++) d[0][i] = 100005;
        for (int i = 1; i <= k; i++) {
            for (int j = 0; j < 1024; j++) d[i & 1][j] = 100005;
            int b = 0;
            pair<int, int> c[1030] = {};
            for (int j = 0; j < 1024; j++) c[j].second = j;
            for (int j = i - 1; j < a.size(); j += k) c[a[j]].first--, b++;
            sort(c, c + 1024);
            z += b + c[0].first;
            y = min(y, -c[0].first);
            //cout << b << '\n';
            int m = (int)a.size() / k;
            for (int j = 0; j < 1024; j++) {
                for (int l = 0; c[l].first; l++) {
                    d[i & 1][j ^ c[l].second] = min(d[i & 1][j ^ c[l].second], d[~i & 1][j] + b + c[l].first);
                }
            }
            //for (int j = 0; j < 16; j++) cout << d[i & 1][j] << ' '; cout << '\n';
        }
        //cout << z << ' ' << y << '\n';
        return min(d[k & 1][0], z + y);
    }
};
