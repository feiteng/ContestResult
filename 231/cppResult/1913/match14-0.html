class Solution {
public:
    #define ll long long
    #define MAXX 1023
    #define INF (1ll<<30)
    
    int total[2222];
    int costs[2222][MAXX+2];
    ll dp[2222][MAXX+2];
    ll dp_min[2222];
    set<int> uniq_vals[2222];

    int minChanges(vector<int>& nums, int k) {
        
        int n=nums.size();
        for(int i=1;i<=k;i++) {
            // i,i+k,...i+m*k
            int m = (nums.size()-i)/k;
            total[i] = m+1;
            
            for(int x=0;x<=MAXX;x++)
                costs[i][x]=total[i];
            
            for(int j=i;j<=n;j+=k) {
                uniq_vals[i].insert(nums[j-1]);
                costs[i][nums[j-1]]--;
            }
            
            
        }
        
        dp[0][0]=0;
        for(int x=1;x<=MAXX;x++)
            dp[0][x]=INF;
        dp_min[0]=0;
        
        for(int i=1;i<=k;i++) {
            
            for(int x=0;x<=MAXX;x++) {
                dp[i][x]=dp_min[i-1]+total[i];
                
                for(int z : uniq_vals[i]) {
                    dp[i][x]=min(dp[i][x],dp[i-1][x^z]+costs[i][z]);
                }
                
//                 for(int y=0;y<=MAXX;y++) {
//                     dp[i][x]=min(dp[i][x], dp[i-1][y]+costs[i][x^y]);
//                 }
            }
            dp_min[i]=dp[i][0];
            for(int x=1;x<=MAXX;x++)
                dp_min[i]=min(dp_min[i],dp[i][x]);
        }
        
        return dp[k][0];
        
        
        

        
        
    }
};
