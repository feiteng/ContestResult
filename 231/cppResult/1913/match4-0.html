class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        
        int n  = nums.size();
        vector<int> dp(1024, 10000);
        dp[0] = 0;
        int totalm = 0;
        int md = 10000;
        for (int j=0; j<k; ++j) {
            map<int, int> z;
            int tot = 0;
            for (int t = j; t<n; t+=k) {
                z[nums[t]]++;
                ++ tot;
            }
            vector<int> dp1(1024, 10000);
            int m1 = 0;
            for (auto g: z) {
                m1 = max(m1, g.second);
                for (int i=0; i<1024; ++i) {
                    if (dp[i]>= 10000) continue;
                    dp1[i ^ g.first] = min(dp1[i^g.first], dp[i] + tot - g.second);
                }
            }
            dp = dp1;  
            totalm += tot - m1;
            md = min(md,   m1);
        }
        int res = totalm + md;
        cout<<res<<endl;
        res = min(res, dp[0]);
        return res;
    }
};
