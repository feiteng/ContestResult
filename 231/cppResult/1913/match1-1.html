class Solution {
public:
    int f[2071][1050], cnt[2071][1050];

    int minChanges(vector<int>& nums, int k) {
        int n = (int )nums.size();
        for (int i = 0; i < (1 << 10); i ++) f[0][i] = 0x3f3f3f3f;
        
        f[0][0] = 0;
        for (int i = 0; i < k; i ++) {
            for (int j = 0; j < (1 << 10); j ++) cnt[i + 1][j] = 0;

            int N = 0;
            for (int j = i; j < n; j += k) {
                cnt[i + 1][nums[j]] --;
                ++ N;
            }
            int mn = 0x3f3f3f3f;
            for (int r = 0; r < (1 << 10); ++ r) {
                mn = min(f[i][r], mn);
            }
            for (int r = 0; r < (1 << 10); ++ r)
                f[i + 1][r] = mn + N;
            for (int j = i; j < n; j += k) {
                int x = nums[j];

                for (int r = 0; r < (1 << 10); ++ r) {
                    f[i + 1][r ^ x] = min(f[i + 1][r ^ x], f[i][r] + cnt[i + 1][x] + N);
                }
            }
        }
        return f[k][0];
    }
};
