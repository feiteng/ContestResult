class Solution {
public:

    map <pair<int, int>, int> mp;
int dp[2005][2005];
int cnt[2005];
int minChanges(vector<int>& nums, int k) {
    int sz = nums.size();
    for (int i = 1; i < 1024; ++i) {
        dp[0][i] = 1e9;
    }
    for (int i = 0; i < k; ++i) {
        memset(cnt, 0, sizeof cnt);
        int minn = sz;
        for (int j = 0; j < 1024; ++j)
            minn = min(minn, dp[i][j]);
        int tmp = 0;

        for (int j = i; j < sz; j += k)
            ++cnt[nums[j]], ++tmp;
        for (int j = 0; j < 1024; ++j)
            dp[i + 1][j] = tmp + minn;
        for (int j = 0; j < 1024; ++j) {
            if (cnt[j] != 0) {
                for (int l = 0; l < 1024; ++l) {
                    dp[i + 1][l] = min(dp[i + 1][l], dp[i][l ^ j] + tmp - cnt[j]);
                }
            }
        }
    }
    return dp[k][0];
}
};
