class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        int n = nums.size();
        // n - k + 1 times
        int m = 1 << 10;
        
        const int inf = n;
        vector<int> f(m, n);
        vector<int> g(m, n);
        f[0] = 0;
        
        for (int i = 0; i < k; ++i) {
            vector<int> c(m);
            int t = 0;
            for (int j = i; j < n; j += k) {
                c[nums[j]]++;

                t++;
            }
            
            int minf = *min_element(f.begin(), f.end());
            for (int j = 0; j < m; ++j) {
                g[j] = minf + t;
            }
            for (int j = 0; j < m; ++j) {

                if (c[j] != 0) {
                    int x = t - c[j];
                    for (int k = 0; k < m; ++k) {
                        g[k] = min(g[k], f[k ^ j] + x);
                    }
                }
            }
            f.swap(g);
        }
        return f[0];
    }
};
