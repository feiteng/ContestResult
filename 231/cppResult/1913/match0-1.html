class Solution {
public:
    int minChanges(vector<int>& nums, int k) {
        int n = nums.size();
        int inf = n;
        auto a = nums;
        int res = 0;
        vector<int> f(1024, inf), g(1024, 0);

        f[0] = 0;
        
        for (int i = 0; i < k; ++i) {
            map<int, int> mp;
            // int tot = 0, mx = 0;
            int tot = 0;
            for (int j = i; j < n; j += k) {
                mp[a[j]] += 1;
                tot += 1;
                // tot += 1; mx = max(mx, mp[a[j]]);
            }
            // res += (tot - mx);
            int minF = inf;
            for (int i = 0; i < f.size(); ++i) {
                minF = min(minF, f[i]);
            }
            for (int i = 0; i < g.size(); ++i) {
                g[i] = minF + tot;
            }
            for (auto p: mp) {
                for (int i = 0; i < f.size(); ++i) {
                    g[i] = min(g[i], f[i ^ p.first] + tot - p.second);
                }
            }
            swap(f, g);
        }
        return f[0];
    }
};
