class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        long long sum = 0;
        for(auto i: nums)
             sum+=i;
        
        
        long long tar = abs(goal-sum);
        long long req = tar/limit;
        if(tar%limit!=0)
             req++;
        
        return req;
    }
};
