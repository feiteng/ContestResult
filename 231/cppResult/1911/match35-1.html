class Solution {
    #define ll long long
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        ll sum = 0;
        for (int x : nums) sum += (ll)x;
        ll left = (ll)goal - sum;
        left = abs(left);
        return (left+(ll)limit-1)/(ll)limit;
    }
};
