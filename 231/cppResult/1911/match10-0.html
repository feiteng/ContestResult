class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        long long sum = 0;
        for(int num: nums) sum += num;
        long long delta = abs(sum - goal);
        long long res = delta / limit;
        if(delta % limit) res += 1;
        return res;
    }
};
