class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        long long sum=0;
        for(int i=0;i<nums.size();i++)
        {
            sum+=nums[i];
        }
            
        long long ans=abs(goal-sum)/limit + (abs(goal-sum)%limit!=0);
        return ans;
    }
};
