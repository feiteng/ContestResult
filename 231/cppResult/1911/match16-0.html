class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        long long curr_summ = 0;
        for(int x : nums)
            curr_summ += x;
        
        long long diff = abs(curr_summ - goal);
        return (diff + limit-1)/limit;
        
    }
};
