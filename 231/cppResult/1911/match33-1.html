inline LL fastpo(LL x, LL n, LL mod) {
	LL res(1);
	while(n) {
		if(n & 1) {
			res = res * (LL)x % mod;
		}
		x = x * (LL) x % mod;
		n /= 2;
	}
	return res;
}
LL c(int a, int b) {
    LL res = 1;
    for(int i = 1; i <= b; i++) {
        res = res * (a - i + 1);
        res /= i;
    }
    return res;
}

inline string itoa(int x, int width = 0) {
  string res;
  if(x == 0) res.push_back('0');
  while(x) {
    res.push_back('0' + x % 10);
    x /= 10;
  }
  while((int)res.size() < width) res.push_back('0');
  reverse(res.begin(), res.end());
  return res;
}

int getr(int x) {
    int p = x;
    while(rela[p] != p) p = rela[p];
    int p1 = p; p = x;
    while(rela[p] != p) {
        int p2 = rela[p];
        rela[p] = p1;
        p = p2;
    }
    return p1;
}

class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        LL sum = 0;
        for(int i : nums) sum += i;
        LL delta = abs(sum - goal);
        return (delta + limit - 1) / limit;
    }
};
