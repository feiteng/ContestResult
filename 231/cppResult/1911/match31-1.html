class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
        long long res = 0;
        for (int i = 0; i < nums.size(); i++) {
            res += nums[i];
            
        }

        long long tmp = goal - res;
        long long ans = abs(tmp) / limit;
        
        if (abs(tmp) % limit != 0) ans ++;
        
        return (int)ans;
        
    }
};
