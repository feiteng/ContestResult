const int N = 2e4 + 5, MOD = 1e9 + 7;
class Solution {
public:
    int st[N], dis[N], memo[N];
    int h[N], e[N << 2], ne[N << 2], c[N << 2], cnt;
    struct Node{
        int idx, dist;
        bool operator < (const Node& a) const {
            return dist > a.dist;
        }
    };
    

    void add(int u, int v, int w) {
        e[++cnt] = v;
        ne[cnt] = h[u];
        c[cnt] = w;
        h[u] = cnt;
    }
    
    void dijkstra(int n) {
        memset(dis, 0x3f, sizeof dis);
        priority_queue<Node> heap;
        heap.push({n, 0});
        dis[n] = 0;
        
        while (heap.size()) {
            Node now = heap.top();
            heap.pop();

            st[now.idx] = 1;
            for (int i = h[now.idx]; i; i = ne[i]) {
                int v = e[i];
                if (!st[v] && dis[v] > dis[now.idx] + c[i])
                {
                    dis[v] = dis[now.idx] + c[i];
                    heap.push({v, dis[v]});
                }
            }
        }
    }
    
    int dfs(int u, int target) {

        if (memo[u] != -1) return memo[u];
        if (u == target) return 1;
        int res = 0;
        for (int i = h[u]; i; i = ne[i]) {
            int v = e[i];
            if (dis[u] > dis[v]) res = (res + dfs(v, target)) % MOD;
        }
        return memo[u] = res;
    }
    
    int countRestrictedPaths(int n, vector<vector<int>>& edges) {

        for (int i = 0; i < edges.size(); i ++ ) {
            int u = edges[i][0], v = edges[i][1], w = edges[i][2];
            add(u, v, w), add(v, u, w);
        }
        dijkstra(n);
        // for (int i = 1; i <= n; i ++ ) cout << dis[i] << endl;
        memset(memo, -1, sizeof memo);
        return dfs(1, n);
    }
};
