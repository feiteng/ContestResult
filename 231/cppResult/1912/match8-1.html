class Solution {
public:
    #define fi first
    #define se second

    struct node{
        int fi,se; 
        node(){}
        node(int x,int y):fi(x),se(y){}
        bool operator<(const node &x)const{
            return fi>x.fi;
        }
    };
    const int mod=1e9+7;
    int countRestrictedPaths(int n, vector<vector<int>>& edges) {
        vector<vector<node> >e(n+1, vector<node>());
        vector<int> dis(n+1,1000000000);
        vector<bool> vis(n+1,false),vis2(n+1,false);
        priority_queue<node>q,p;
        for(int i=0;i<edges.size();++i){
            int u=edges[i][0],v=edges[i][1],w=edges[i][2];
            e[u].push_back(node(v,w));
            e[v].push_back(node(u,w));
        }
        dis[n]=0;
        q.push(node(dis[n],n));
        while(!q.empty()){

            int w=q.top().fi,x=q.top().se;q.pop();
            if(vis[x])continue;
            vis[x]=1;
            for(int i=0;i<e[x].size();++i){
                int v=e[x][i].fi,ww=e[x][i].se;
                if(dis[v]>dis[x]+ww){
                    dis[v]=dis[x]+ww;
                    q.push(node(dis[v],v));
                }
            }
        }
        vector<int>dp(n+1,0);

        dp[1]=1;
        p.push(node(-dis[1],1));
        while(!p.empty()){
            int v=p.top().se;p.pop();
            if(vis2[v])continue;
            vis2[v]=1;
            for(int i=0;i<e[v].size();++i){
                int u=e[v][i].fi;
                if(dis[u]<dis[v]){
                    dp[u]=(dp[v]+dp[u])%mod;
                    p.push(node(-dis[u],u));
                }
            }
        }
        return dp[n];
    }
};
