class Solution {
public:
    int countRestrictedPaths(int n, vector<vector<int>>& edges) {
        vector<vector<pair<int,int>>> g(n);
        for(auto e:edges){
            int u=e[0],v=e[1],w=e[2];
            u--,v--;
            g[u].push_back({v,w});
            g[v].push_back({u,w});
        }
        priority_queue<pair<int,int>> qq;
        qq.push({0,n-1});
        vector<int> dis(n,INT_MAX);
        dis[n-1]=0;
        while(qq.size()>0){
            auto [d,x]=qq.top();
            qq.pop();

            d=-d;
            if(d>dis[x]) continue;
            for(auto [ne,w]:g[x]){
                if(d+w<dis[ne]){
                    dis[ne]=d+w;
                    qq.push({-dis[ne],ne});
                }
            }
        }
        vector<pair<int,int>> r;
        for(int i=0;i<n;i++) r.push_back({dis[i],i});
        sort(r.begin(),r.end());
        vector<int> res(n,0);
        res[n-1]=1;

        int mod=1e9+7;
        for(int i=1;i<n;i++){
            int x=r[i].second;
            for(auto [ne,w]:g[x]){
                if(dis[ne]<dis[x]){
                    res[x]=(res[x]+res[ne])%mod;
                }
            }
        }
        return res[0];
    }
};
