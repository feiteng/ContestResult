class Solution {
public:
    typedef long long ll;
    const ll inf = 100000000000;
    const int mod = (int)1e9 + 7;
    vector<pair<int, int>> gr[20001];
    ll dis[20001];
    ll dp[20001];
    int solve(int node) {
        if(node == 1) {
            return 1;
        }
        if(dp[node] != -1) return dp[node];
        ll ans = 0;
        for(auto c : gr[node]) {
            if(dis[c.first] > dis[node]) {
                ans = (ans%mod + solve(c.first)%mod)%mod;
            }
        }
        return dp[node] = ans;
    }
    int countRestrictedPaths(int n, vector<vector<int>>& edges) {
        for(auto e : edges) {
            gr[e[0]].push_back({e[1], e[2]});
            gr[e[1]].push_back({e[0], e[2]});
        }
        queue<int> q;
        q.push(n);
        for(int i = 1; i < n; i++) dis[i] = inf;
        dis[n] = 0;
        memset(dp, -1, sizeof dp);
        while(!q.empty()) {
            int fr = q.front();
            q.pop();
            for(auto c : gr[fr]) {
                if(dis[fr] + c.second < dis[c.first]) {
                    dis[c.first] = dis[fr] + c.second;
                    q.push(c.first);
                }
            }
        }
        return solve(n)%mod;
    }
};
