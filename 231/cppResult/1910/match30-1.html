class Solution {
public:
    bool checkOnesSegment(string s) {
        int cnt = 0 ;
        int n = sz(s) ;
        for(int i = 0 ; i < n ; ){
            if(s[i] == '0') {
                i++ ;
                continue ;
            }
            int j = i ;
            while(j < n && s[i] == s[j]){
                j++ ;
            }
            cnt++ ;
            i = j ;
        }
        return cnt == 1 ;
    }
};
