class Solution {
public:
    bool checkOnesSegment(string s) {
        int cnt =0;
        for(int i =0; i< s.length();i++)
        {
            if(i==0 && s[i] =='1')
                cnt ++;
            else if(s[i] == '1' && s[i-1] == '0')
                cnt ++;
        }
        return cnt == 1;
    }
};
