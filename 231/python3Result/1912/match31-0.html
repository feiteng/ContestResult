class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        dist = [0]*(n+1)
        res = 0
        graph = defaultdict(list)
        for u,v,w in edges:
            graph[u].append((v,w))
            graph[v].append((u,w))
        
        queue = [(w, v) for v, w in graph[n]]
        heapq.heapify(queue)
        visited = set([n])
        while queue:
            w, u = heapq.heappop(queue)
            if u in visited: 
                continue

            dist[u] = w
            visited.add(u)
            
            for v, w in graph[u]:
                if v not in visited:
                    heapq.heappush(queue, (dist[u] + w, v))
        # print(dist)
        
        count = [0] *(n+1)
        
        @lru_cache(None)
        def dfs(u):
            if u == 1: 
                return 1
            res = 0
            for v, _ in graph[u]:
                if dist[u] < dist[v]:
                    res += dfs(v)
            return res %(10**9+7)
        
        return dfs(n)
