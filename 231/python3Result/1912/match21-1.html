class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List
[int]]) -> int:
        d2l=[None]*n
        paths=defaultdict(list)
        for a,b,w in edges:
            paths[a].append((b,w))
            paths[b].append((a,w))
        stax = [(0,n)]
        while stax:
            d,num=heapq.heappop(stax)
            if not (d2l[num-1] is None or d<d2l[num-1]):continue
            d2l[num-1]=d
            for a,w in paths[num]:
                heapq.heappush(stax,(d
+w,a))
        @cache
        def counter(a):
            if a==n:return 1
            nonlocal paths,d2l
            ret=0
            for b,_ in paths[a]:
                if d2l[b-1]<d2l[a-1]:
                    ret+=counter(b)
            return ret
        return counter(1)%(10**9 + 7)
