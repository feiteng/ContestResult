class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        eg = defaultdict
(list)
        for a, b, c in edges:
            eg[a].append([b, c])
            eg[b].append([a, c])
        
        d, q = {}, [[0, n]]
        while q:
            dis, a = heappop(q)
            if a not in d:
                d[a] = dis
                for b, c in eg[a]:
                    heappush(q, [dis
 + c, b])
        
        mod = 10**9+7
        @cache
        def dfs(a):
            if a == n:
                return 1
            ans = 0
            for b, c in eg[a]:
                if d[b] < d[a]:
                    ans += dfs(b)
            return ans % mod
        return dfs(1)
