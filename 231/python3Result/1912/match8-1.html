from heapq import heappop as pop, heappush as push
class Solution:
    def countRestrictedPaths(self, n: int, A: List[List[int]]) -> int:
        dis, nbs 
= {}, collections.defaultdict(set)
        for a, b, d in A:
            nbs[a].add((b, d))
            nbs[b].add((a, d))
        st = [(0, n)]
        while st:
            d, nd = pop(st)
            if nd in dis:
                continue
            dis[nd] = d
            for a, b in nbs[nd]:
                if a not in dis:
                    push(st, (d + b, a))
        
@functools.lru_cache
        def dfs(i):
            if i == n:  return 1
            ret = 0
            for nd, d in nbs[i]:
                if dis[nd] < dis[i]:
                    ret += dfs(nd)
            return ret
        
        return dfs(1) % (10 ** 9 + 7)
