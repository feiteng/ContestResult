class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        MOD = 10**9 + 7
        graph = defaultdict(list)
        for u, v, w in edges:
            u -= 1
            v -= 1
            graph[u].append((v, w))
            graph[v].append((u, w))
        
        dist = [inf] * n
        dist[n - 1] = 0
        h = [(0, n - 1)]
        while h:
            d, node = heappop(h)
            for v, w in graph[node]:
                if d + w < dist[v]:
                    dist[v] = d + w
                    heappush(h, (d + w, v))
        # print(dist)
        @cache
        def f(i):
            if i == 0:
                return 1
            res = 0
            for v, _ in graph[i]:
                if dist[v] > dist[i]:
                    res += f(v)
            return res
        
        return f(n - 1) % MOD
